38/*****************************************************************************/
39std::string ConjugateGradientParameters::blasTranslator(const BLASKernel value) {
40 std::string s;
41 switch (value) {
42 case ConjugateGradientParameters::GTSAM: s = "GTSAM" ; break;
43 default: s = "UNDEFINED" ; break;
44 }
45 return s;
46}
47
48/*****************************************************************************/
49ConjugateGradientParameters::BLASKernel ConjugateGradientParameters::blasTranslator(
50 const std::string &src) {
51 std::string s = src;
52 // Convert to upper case
53 std::transform(first: s.begin(), last: s.end(), result: s.begin(), unary_op: ::toupper);
54 if (s == "GTSAM") return ConjugateGradientParameters::GTSAM;
55
56 /* default is SBM */
57 return ConjugateGradientParameters::GTSAM;
58}
