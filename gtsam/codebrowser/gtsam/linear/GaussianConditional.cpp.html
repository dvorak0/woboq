65 /* ************************************************************************ */
66 GaussianConditional GaussianConditional::FromMeanAndStddev(Key key,
67 const Vector& mu,
68 double sigma) {
69 // |Rx - d| = |x - mu|/sigma
70 const Matrix R = Matrix::Identity(rows: mu.size(), cols: mu.size());
71 const Vector& d = mu;
72 return GaussianConditional(key, d, R,
73 noiseModel::Isotropic::Sigma(dim: mu.size(), sigma));
74 }

76 /* ************************************************************************ */
77 GaussianConditional GaussianConditional::FromMeanAndStddev(
78 Key key, const Matrix& A, Key parent, const Vector& b, double sigma) {
79 // |Rx + Sy - d| = |x-(Ay + b)|/sigma
80 const Matrix R = Matrix::Identity(rows: b.size(), cols: b.size());
81 const Matrix S = -A;
82 const Vector& d = b;
83 return GaussianConditional(key, d, R, parent, S,
84 noiseModel::Isotropic::Sigma(dim: b.size(), sigma));
85 }

87 /* ************************************************************************ */
88 GaussianConditional GaussianConditional::FromMeanAndStddev(
89 Key key, const Matrix& A1, Key parent1, const Matrix& A2, Key parent2,
90 const Vector& b, double sigma) {
91 // |Rx + Sy + Tz - d| = |x-(A1 y + A2 z + b)|/sigma
92 const Matrix R = Matrix::Identity(rows: b.size(), cols: b.size());
93 const Matrix S = -A1;
94 const Matrix T = -A2;
95 const Vector& d = b;
96 return GaussianConditional(key, d, R, parent1, S, parent2, T,
97 noiseModel::Isotropic::Sigma(dim: b.size(), sigma));
98 }

101 void GaussianConditional::print(const string &s, const KeyFormatter& formatter) const {
102 cout << (s.empty() ? "" : s + " ") << "p(";
103 for (const_iterator it = beginFrontals(); it != endFrontals(); ++it) {
104 cout << formatter(*it) << (nrFrontals() > 1 ? " " : "");
105 }
106
107 if (nrParents()) {
108 cout << " |";
109 for (const_iterator it = beginParents(); it != endParents(); ++it) {
110 cout << " " << formatter(*it);
111 }
112 }
113 cout << ")" << endl;
114
115 cout << formatMatrixIndented(label: " R = ", matrix: R()) << endl;
116 for (const_iterator it = beginParents() ; it != endParents() ; ++it) {
117 cout << formatMatrixIndented(label: " S[" + formatter(*it) + "] = ", matrix: getA(variable: it)) << endl;
118 }
119 cout << formatMatrixIndented(label: " d = ", matrix: getb(), makeVectorHorizontal: true) << "\n";
120 if (nrParents() == 0) {
121 const auto mean = solve(parents: {}); // solve for mean.
122 mean.print(str: " mean", formatter);
123 }
124 cout << " logNormalizationConstant: " << std::fixed << std::setprecision(4)
125 << -negLogConstant() << endl;
126 if (model_)
127 model_->print(name: " Noise model: ");
128 else
129 cout << " No noise model" << endl;
130 }

132 /* ************************************************************************* */
133 bool GaussianConditional::equals(const GaussianFactor& f, double tol) const {
134 if (const GaussianConditional* c = dynamic_cast<const GaussianConditional*>(&f)) {
135 // check if the size of the parents_ map is the same
136 if (parents().size() != c->parents().size())
137 return false;
138
139 // check if R_ and d_ are linear independent
140 for (DenseIndex i = 0; i < Ab_.rows(); i++) {
141 list<Vector> rows1, rows2;
142 rows1.push_back(x: Vector(R().row(i)));
143 rows2.push_back(x: Vector(c->R().row(i)));
144
145 // check if the matrices are the same
146 // iterate over the parents_ map
147 for (const_iterator it = beginParents(); it != endParents(); ++it) {
148 const_iterator it2 = c->beginParents() + (it - beginParents());
149 if (*it != *(it2))
150 return false;
151 rows1.push_back(x: row(A: getA(variable: it), j: i));
152 rows2.push_back(x: row(A: c->getA(variable: it2), j: i));
153 }
154
155 Vector row1 = concatVectors(vs: rows1);
156 Vector row2 = concatVectors(vs: rows2);
157 if (!linear_dependent(vec1: row1, vec2: row2, tol))
158 return false;
159 }
160
161 // check if sigmas are equal
162 if ((model_ && !c->model_) || (!model_ && c->model_)
163 || (model_ && c->model_ && !model_->equals(expected: *c->model_, tol)))
164 return false;
165
166 return true;
167 } else {
168 return false;
169 }
170 }

172 /* ************************************************************************* */
173 double GaussianConditional::logDeterminant() const {
174 if (get_model()) {
175 Vector diag = R().diagonal();
176 get_model()->whitenInPlace(v&: diag);
177 return diag.unaryExpr(func: [](double x) { return log(x: x); }).sum();
178 } else {
179 return R().diagonal().unaryExpr(func: [](double x) { return log(x: x); }).sum();
180 }
181 }

183 /* ************************************************************************* */
184 // normalization constant = 1.0 / sqrt((2*pi)^n*det(Sigma))
185 // neg-log = 0.5 * n*log(2*pi) + 0.5 * log det(Sigma)
186 double GaussianConditional::negLogConstant() const {
187 constexpr double log2pi = 1.8378770664093454835606594728112;
188 size_t n = d().size();
189 // Sigma = (R'R)^{-1}, det(Sigma) = det((R'R)^{-1}) = det(R'R)^{-1}
190 // log det(Sigma) = -log(det(R'R)) = -2*log(det(R))
191 // Hence, log det(Sigma)) = -2.0 * logDeterminant()
192 // which gives neg-log = 0.5*n*log(2*pi) + 0.5*(-2.0 * logDeterminant())
193 // = 0.5*n*log(2*pi) - (0.5*2.0 * logDeterminant())
194 // = 0.5*n*log(2*pi) - logDeterminant()
195 return 0.5 * n * log2pi - logDeterminant();
196 }

198 /* ************************************************************************* */
199 // density = k exp(-error(x))
200 // log = log(k) - error(x)
201 double GaussianConditional::logProbability(const VectorValues& x) const {
202 return -(negLogConstant() + error(c: x));
203 }

205 double GaussianConditional::logProbability(const HybridValues& x) const {

219 VectorValues GaussianConditional::solve(const VectorValues& x) const {
220 // Concatenate all vector values that correspond to parent variables
221 const Vector xS = x.vector(keys: KeyVector(beginParents(), endParents()));
222
223 // Update right-hand-side
224 const Vector rhs = d() - S() * xS;
225
226 // Solve matrix
227 const Vector solution = R().triangularView<Eigen::Upper>().solve(other: rhs);
228
229 // Check for indeterminant solution
230 if (solution.hasNaN()) {
231 throw IndeterminantLinearSystemException(keys().front());
232 }
233
234 // Insert solution into a VectorValues
235 VectorValues result;
236 DenseIndex vectorPosition = 0;
237 for (const_iterator frontal = beginFrontals(); frontal != endFrontals(); ++frontal) {
238 result.emplace(j: *frontal, args: solution.segment(start: vectorPosition, n: getDim(variable: frontal)));
239 vectorPosition += getDim(variable: frontal);
240 }
241
242 return result;
243 }

245 /* ************************************************************************* */
246 VectorValues GaussianConditional::solveOtherRHS(
247 const VectorValues& parents, const VectorValues& rhs) const {
248 // Concatenate all vector values that correspond to parent variables
249 Vector xS = parents.vector(keys: KeyVector(beginParents(), endParents()));
250
251 // Instead of updating getb(), update the right-hand-side from the given rhs
252 const Vector rhsR = rhs.vector(keys: KeyVector(beginFrontals(), endFrontals()));
253 xS = rhsR - S() * xS;
254
255 // Solve Matrix
256 Vector soln = R().triangularView<Eigen::Upper>().solve(other: xS);
257
258 // Scale by sigmas
259 if (model_)
260 soln.array() *= model_->sigmas().array();
261
262 // Insert solution into a VectorValues
263 VectorValues result;
264 DenseIndex vectorPosition = 0;
265 for (const_iterator frontal = beginFrontals(); frontal != endFrontals(); ++frontal) {
266 result.emplace(j: *frontal, args: soln.segment(start: vectorPosition, n: getDim(variable: frontal)));
267 vectorPosition += getDim(variable: frontal);
268 }
269
270 return result;
271 }

273 /* ************************************************************************* */
274 void GaussianConditional::solveTransposeInPlace(VectorValues& gy) const {
275 Vector frontalVec = gy.vector(keys: KeyVector(beginFrontals(), endFrontals()));
276 frontalVec = R().transpose().triangularView<Eigen::Lower>().solve(other: frontalVec);
277
278 // Check for indeterminate solution
279 if (frontalVec.hasNaN()) throw IndeterminantLinearSystemException(this->keys().front());
280
281 for (const_iterator it = beginParents(); it!= endParents(); it++)
282 gy[*it].noalias() += -1.0 * getA(variable: it).transpose() * frontalVec;
283
284 // Scale by sigmas
285 if (model_)
286 frontalVec.array() *= model_->sigmas().array();
287
288 // Write frontal solution into a VectorValues
289 DenseIndex vectorPosition = 0;
290 for (const_iterator frontal = beginFrontals(); frontal != endFrontals(); ++frontal) {
291 gy[*frontal] = frontalVec.segment(start: vectorPosition, n: getDim(variable: frontal));
292 vectorPosition += getDim(variable: frontal);
293 }
294 }

296 /* ************************************************************************ */
297 JacobianFactor::shared_ptr GaussianConditional::likelihood(
298 const VectorValues& frontalValues) const {
299 // Error is |Rx - (d - Sy - Tz - ...)|^2
300 // so when we instantiate x (which has to be completely known) we beget:
301 // |Sy + Tz + ... - (d - Rx)|^2
302 // The noise model just transfers over!
303
304 // Get frontalValues as vector
305 const Vector x =
306 frontalValues.vector(keys: KeyVector(beginFrontals(), endFrontals()));
307
308 // Copy the augmented Jacobian matrix:
309 auto newAb = Ab_;
310
311 // Restrict view to parent blocks
312 newAb.firstBlock() += nrFrontals_;
313
314 // Update right-hand-side (last column)
315 auto last = newAb.matrix().cols() - 1;
316 const auto RR = R().triangularView<Eigen::Upper>();
317 newAb.matrix().col(i: last) -= RR * x;
318
319 // The keys now do not include the frontal keys:
320 KeyVector newKeys;
321 newKeys.reserve(n: nrParents());
322 for (auto&& key : parents()) newKeys.push_back(x: key);
323
324 // Hopefully second newAb copy below is optimized out...
325 return std::make_shared<JacobianFactor>(args&: newKeys, args&: newAb, args: model_);
326 }

329 JacobianFactor::shared_ptr GaussianConditional::likelihood(
330 const Vector& frontal) const {
331 if (nrFrontals() != 1)
332 throw std::invalid_argument(
333 "GaussianConditional Single value likelihood can only be invoked on "
334 "single-variable conditional");
335 VectorValues values;
336 values.insert(j: keys_[0], value: frontal);
337 return likelihood(frontalValues: values);
338 }

341 VectorValues GaussianConditional::sample(const VectorValues& parentsValues,
342 std::mt19937_64* rng) const {
343 if (nrFrontals() != 1) {
344 throw std::invalid_argument(
345 "GaussianConditional::sample can only be called on single variable "
346 "conditionals");
347 }
348
349 VectorValues solution = solve(x: parentsValues);
350 Key key = firstFrontalKey();
351
352 // Check if rng is nullptr, then assign default
353 rng = (rng == nullptr) ? &kRandomNumberGenerator : rng;
354
355 // The vector of sigma values for sampling.
356 // If no model, initialize sigmas to 1, else to model sigmas
357 const Vector& sigmas = (!model_) ? Vector::Ones(newSize: rows()) : model_->sigmas();
358 solution[key] += Sampler::sampleDiagonal(sigmas, rng);
359 return solution;
360 }

364 throw std::invalid_argument(
365 "sample() can only be invoked on no-parent prior");
366 VectorValues values;
367 return sample(parentsValues: values, rng);
