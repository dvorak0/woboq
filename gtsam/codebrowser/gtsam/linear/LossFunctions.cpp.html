36Vector Base::weight(const Vector& error) const {
37 const size_t n = error.rows();
38 Vector w(n);
39 for (size_t i = 0; i < n; ++i)
40 w(i) = weight(distance: error(i));
41 return w;
42}
43
44Vector Base::sqrtWeight(const Vector &error) const {
45 return weight(error).cwiseSqrt();
46}
47
48// The following three functions re-weight block matrices and a vector
49// according to their weight implementation
50
51void Base::reweight(Vector& error) const {
52 if (reweight_ == Block) {
53 const double w = sqrtWeight(distance: error.norm());
54 error *= w;
55 } else {
56 error.array() *= weight(error).cwiseSqrt().array();
57 }
58}
59
60// Reweight n block matrices with one error vector
61void Base::reweight(vector<Matrix> &A, Vector &error) const {
62 if ( reweight_ == Block ) {
63 const double w = sqrtWeight(distance: error.norm());
64 for(Matrix& Aj: A) {
65 Aj *= w;
66 }
67 error *= w;
68 }
69 else {
70 const Vector W = sqrtWeight(error);
71 for(Matrix& Aj: A) {
72 vector_scale_inplace(v: W,A&: Aj);
73 }
74 error = W.cwiseProduct(other: error);
75 }
76}
77
78// Reweight one block matrix with one error vector
79void Base::reweight(Matrix &A, Vector &error) const {
80 if ( reweight_ == Block ) {
81 const double w = sqrtWeight(distance: error.norm());
82 A *= w;
83 error *= w;
84 }
85 else {
86 const Vector W = sqrtWeight(error);
87 vector_scale_inplace(v: W,A);
88 error = W.cwiseProduct(other: error);
89 }
90}
91
92// Reweight two block matrix with one error vector
93void Base::reweight(Matrix &A1, Matrix &A2, Vector &error) const {
94 if ( reweight_ == Block ) {
95 const double w = sqrtWeight(distance: error.norm());
96 A1 *= w;
97 A2 *= w;
98 error *= w;
99 }
100 else {
101 const Vector W = sqrtWeight(error);
102 vector_scale_inplace(v: W,A&: A1);
103 vector_scale_inplace(v: W,A&: A2);
104 error = W.cwiseProduct(other: error);
105 }
106}
107
108// Reweight three block matrix with one error vector
109void Base::reweight(Matrix &A1, Matrix &A2, Matrix &A3, Vector &error) const {
110 if ( reweight_ == Block ) {
111 const double w = sqrtWeight(distance: error.norm());
112 A1 *= w;
113 A2 *= w;
114 A3 *= w;
115 error *= w;
116 }
117 else {
118 const Vector W = sqrtWeight(error);
119 vector_scale_inplace(v: W,A&: A1);
120 vector_scale_inplace(v: W,A&: A2);
121 vector_scale_inplace(v: W,A&: A3);
122 error = W.cwiseProduct(other: error);
123 }
124}

140Fair::Fair(double c, const ReweightScheme reweight) : Base(reweight), c_(c) {
141 if (c_ <= 0) {
142 throw runtime_error("mEstimator Fair takes only positive double in constructor.");
143 }
144}
145
146double Fair::weight(double distance) const {
147 return 1.0 / (1.0 + std::abs(x: distance) / c_);
148}
149
150double Fair::loss(double distance) const {
151 const double absError = std::abs(x: distance);
152 const double normalizedError = absError / c_;
153 const double c_2 = c_ * c_;
154 return c_2 * (normalizedError - std::log1p(x: normalizedError));
155}

158{ cout << s << "fair (" << c_ << ")" << endl; }

173Huber::Huber(double k, const ReweightScheme reweight) : Base(reweight), k_(k) {
174 if (k_ <= 0) {
175 throw runtime_error("mEstimator Huber takes only positive double in constructor.");
176 }
177}
178
179double Huber::weight(double distance) const {
180 const double absError = std::abs(x: distance);
181 return (absError <= k_) ? (1.0) : (k_ / absError);
182}
183
184double Huber::loss(double distance) const {
185 const double absError = std::abs(x: distance);
186 if (absError <= k_) { // |x| <= k
187 return distance*distance / 2;
188 } else { // |x| > k
189 return k_ * (absError - (k_/2));
190 }
191}

244Tukey::Tukey(double c, const ReweightScheme reweight) : Base(reweight), c_(c), csquared_(c * c) {
245 if (c <= 0) {
246 throw runtime_error("mEstimator Tukey takes only positive double in constructor.");
247 }

249
250double Tukey::weight(double distance) const {
251 if (std::abs(x: distance) <= c_) {
252 const double one_minus_xc2 = 1.0 - distance*distance/csquared_;
253 return one_minus_xc2 * one_minus_xc2;
254 }
255 return 0.0;
256}
257
258double Tukey::loss(double distance) const {
259 double absError = std::abs(x: distance);
260 if (absError <= c_) {
261 const double one_minus_xc2 = 1.0 - distance*distance/csquared_;
262 const double t = one_minus_xc2*one_minus_xc2*one_minus_xc2;
263 return csquared_ * (1 - t) / 6.0;
264 } else {
265 return csquared_ / 6.0;
266 }
267}

316GemanMcClure::GemanMcClure(double c, const ReweightScheme reweight)
317 : Base(reweight), c_(c) {
318}
319
320double GemanMcClure::weight(double distance) const {
321 const double c2 = c_*c_;
322 const double c4 = c2*c2;
323 const double c2error = c2 + distance*distance;
324 return c4/(c2error*c2error);
325}
326
327double GemanMcClure::loss(double distance) const {
328 const double c2 = c_*c_;
329 const double error2 = distance*distance;
330 return 0.5 * (c2 * error2) / (c2 + error2);
331}

350DCS::DCS(double c, const ReweightScheme reweight)
351 : Base(reweight), c_(c) {
352}
353
354double DCS::weight(double distance) const {
355 const double e2 = distance*distance;
356 if (e2 > c_)
357 {
358 const double w = 2.0*c_/(c_ + e2);
359 return w*w;
360 }
361
362 return 1.0;
363}
364
365double DCS::loss(double distance) const {
366 // This is the simplified version of Eq 9 from (Agarwal13icra)
367 // after you simplify and cancel terms.
368 const double e2 = distance*distance;
369 const double e4 = e2*e2;
370 const double c2 = c_*c_;
371
372 return (c2*e2 + c_*e4) / ((e2 + c_)*(e2 + c_));
373}

393L2WithDeadZone::L2WithDeadZone(double k, const ReweightScheme reweight)
394 : Base(reweight), k_(k) {
395 if (k_ <= 0) {
396 throw runtime_error("mEstimator L2WithDeadZone takes only positive double in constructor.");

399
400double L2WithDeadZone::weight(double distance) const {
401 // note that this code is slightly uglier than residual, because there are three distinct
402 // cases to handle (left of deadzone, deadzone, right of deadzone) instead of the two
403 // cases (deadzone, non-deadzone) in residual.
404 if (std::abs(x: distance) <= k_) return 0.0;
405 else if (distance > k_) return (-k_+distance)/distance;
406 else return (k_+distance)/distance;
407}
408
409double L2WithDeadZone::loss(double distance) const {
410 const double abs_error = std::abs(x: distance);
411 return (abs_error < k_) ? 0.0 : 0.5*(k_-abs_error)*(k_-abs_error);
412}

415 std::cout << s << ": L2WithDeadZone (" << k_ << ")" << std::endl;

433AsymmetricTukey::AsymmetricTukey(double c, const ReweightScheme reweight) : Base(reweight), c_(c), csquared_(c * c) {
434 if (c <= 0) {
435 throw runtime_error("mEstimator AsymmetricTukey takes only positive double in constructor.");
436 }
437}
438
439double AsymmetricTukey::weight(double distance) const {
440 distance = -distance;
441 if (distance >= 0.0) {
442 return 1.0;
443 } else if (distance > -c_) {
444 const double one_minus_xc2 = 1.0 - distance * distance / csquared_;
445 return one_minus_xc2 * one_minus_xc2;
446 }
447 return 0.0;
448}
449
450double AsymmetricTukey::loss(double distance) const {
451 distance = -distance;
452 if (distance >= 0.0) {
453 return distance * distance / 2.0;
454 } else if (distance >= -c_) {
455 const double one_minus_xc2 = 1.0 - distance * distance / csquared_;
456 const double t = one_minus_xc2 * one_minus_xc2 * one_minus_xc2;
457 return csquared_ * (1 - t) / 6.0;
458 }
459 return csquared_ / 6.0;
460}

463 std::cout << s << ": AsymmetricTukey (" << c_ << ")" << std::endl;
464}
