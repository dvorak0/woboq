22#include <gtsam/inference/Key.h>

26namespace gtsam {
27
28class GaussianFactorGraph;
29class Ordering;
30
31/// One SlotEntry stores the slot index for a variable, as well its dim.
32struct GTSAM_EXPORT SlotEntry {
33 Key key;
34 size_t dimension;
35 SlotEntry(Key _key, size_t _dimension) : key(_key), dimension(_dimension) {}
36 std::string toString() const;
37 friend bool operator<(const SlotEntry& p, const SlotEntry& q) {
38 return p.key < q.key;
39 }
40 static bool Zero(const SlotEntry& p) { return p.dimension==0;}
41};
42
43/**
44 * Scatter is an intermediate data structure used when building a HessianFactor
45 * incrementally, to get the keys in the right order. In spirit, it is a map
46 * from global variable indices to slot indices in the union of involved
47 * variables. We also include the dimensionality of the variable.
48 */
49class Scatter : public FastVector<SlotEntry> {
50 public:
51 /// Default Constructor
52 GTSAM_EXPORT Scatter() {}
53
54 /// Construct from gaussian factor graph, without ordering
55 GTSAM_EXPORT explicit Scatter(const GaussianFactorGraph& gfg);
56
57 /// Construct from gaussian factor graph, with (partial or complete) ordering
58 GTSAM_EXPORT explicit Scatter(const GaussianFactorGraph& gfg, const Ordering& ordering);
59
60 /// Add a key/dim pair
61 GTSAM_EXPORT void add(Key key, size_t dim);
62
63 private:
64 /// Find the SlotEntry with the right key (linear time worst case)
65 iterator find(Key key);
66};
67
68} // \ namespace gtsam
