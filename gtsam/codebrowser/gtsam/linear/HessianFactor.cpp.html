44/* ************************************************************************* */
45void HessianFactor::Allocate(const Scatter& scatter) {
46 gttic(HessianFactor_Allocate);
47
48 // Allocate with dimensions for each variable plus 1 at the end for the information vector
49 const size_t n = scatter.size();
50 keys_.resize(new_size: n);
51 FastVector<DenseIndex> dims(n + 1);
52 DenseIndex slot = 0;
53 for(const SlotEntry& slotentry: scatter) {
54 keys_[slot] = slotentry.key;
55 dims[slot] = slotentry.dimension;
56 ++slot;
57 }
58 dims.back() = 1;
59 info_ = SymmetricBlockMatrix(dims);
60}

74/* ************************************************************************* */
75HessianFactor::HessianFactor(Key j, const Matrix& G, const Vector& g, double f)
76 : GaussianFactor(KeyVector{j}), info_(Dims{static_cast<Key>(G.cols()), 1}) {
77 if (G.rows() != G.cols() || G.rows() != g.size())
78 throw invalid_argument(
79 "Attempting to construct HessianFactor with inconsistent matrix and/or vector dimensions");
80 info_.setDiagonalBlock(I: 0, xpr: G);
81 linearTerm() = g;
82 constantTerm() = f;
83}
84
85/* ************************************************************************* */
86// error is 0.5*(x-mu)'*inv(Sigma)*(x-mu) = 0.5*(x'*G*x - 2*x'*G*mu + mu'*G*mu)
87// where G = inv(Sigma), g = G*mu, f = mu'*G*mu = mu'*g
88HessianFactor::HessianFactor(Key j, const Vector& mu, const Matrix& Sigma)
89 : GaussianFactor(KeyVector{j}), info_(Dims{static_cast<Key>(Sigma.cols()), 1}) {
90 if (Sigma.rows() != Sigma.cols() || Sigma.rows() != mu.size())
91 throw invalid_argument(
92 "Attempting to construct HessianFactor with inconsistent matrix and/or vector dimensions");
93 info_.setDiagonalBlock(I: 0, xpr: Sigma.inverse()); // G
94 linearTerm() = info_.diagonalBlock(J: 0) * mu; // g
95 constantTerm() = mu.dot(other: linearTerm().col(i: 0)); // f
96}

99HessianFactor::HessianFactor(Key j1, Key j2, const Matrix& G11,
100 const Matrix& G12, const Vector& g1, const Matrix& G22, const Vector& g2,

104 info_.setDiagonalBlock(I: 0, xpr: G11);
105 info_.setOffDiagonalBlock(I: 0, J: 1, xpr: G12);

107 linearTerm() << g1, g2;
108 constantTerm() = f;

111/* ************************************************************************* */
112HessianFactor::HessianFactor(Key j1, Key j2, Key j3, const Matrix& G11,
113 const Matrix& G12, const Matrix& G13, const Vector& g1, const Matrix& G22,
114 const Matrix& G23, const Vector& g2, const Matrix& G33, const Vector& g3,
115 double f) :
116 GaussianFactor(KeyVector{j1,j2,j3}), info_(
117 Dims{static_cast<Key>(G11.cols()),static_cast<Key>(G22.cols()),static_cast<Key>(G33.cols()),1}) {
118 if (G11.rows() != G11.cols() || G11.rows() != G12.rows()
119 || G11.rows() != G13.rows() || G11.rows() != g1.size()
120 || G22.cols() != G12.cols() || G33.cols() != G13.cols()
121 || G22.cols() != g2.size() || G33.cols() != g3.size())
122 throw invalid_argument(
123 "Inconsistent matrix and/or vector dimensions in HessianFactor constructor");
124 info_.setDiagonalBlock(I: 0, xpr: G11);
125 info_.setOffDiagonalBlock(I: 0, J: 1, xpr: G12);
126 info_.setOffDiagonalBlock(I: 0, J: 2, xpr: G13);
127 info_.setDiagonalBlock(I: 1, xpr: G22);
128 info_.setOffDiagonalBlock(I: 1, J: 2, xpr: G23);
129 info_.setDiagonalBlock(I: 2, xpr: G33);
130 linearTerm() << g1, g2, g3;
131 constantTerm() = f;
132}

145/* ************************************************************************* */
146HessianFactor::HessianFactor(const KeyVector& js,
147 const std::vector<Matrix>& Gs, const std::vector<Vector>& gs, double f) :
148 GaussianFactor(js), info_(_getSizeHFVec(m: gs), true) {
149 // Get the number of variables
150 size_t variable_count = js.size();
151
152 // Verify the provided number of entries in the vectors are consistent
153 if (gs.size() != variable_count
154 || Gs.size() != (variable_count * (variable_count + 1)) / 2)
155 throw invalid_argument(
156 "Inconsistent number of entries between js, Gs, and gs in HessianFactor constructor.\nThe number of keys provided \
157 in js must match the number of linear vector pieces in gs. The number of upper-diagonal blocks in Gs must be n*(n+1)/2");
158
159 // Verify the dimensions of each provided matrix are consistent
160 // Note: equations for calculating the indices derived from the "sum of an arithmetic sequence" formula
161 for (size_t i = 0; i < variable_count; ++i) {
162 DenseIndex block_size = gs[i].size();
163 // Check rows
164 for (size_t j = 0; j < variable_count - i; ++j) {
165 size_t index = i * (2 * variable_count - i + 1) / 2 + j;
166 if (Gs[index].rows() != block_size) {
167 throw invalid_argument(
168 "Inconsistent matrix and/or vector dimensions in HessianFactor constructor");
169 }
170 }
171 // Check cols
172 for (size_t j = 0; j <= i; ++j) {
173 size_t index = j * (2 * variable_count - j + 1) / 2 + (i - j);
174 if (Gs[index].cols() != block_size) {
175 throw invalid_argument(
176 "Inconsistent matrix and/or vector dimensions in HessianFactor constructor");
177 }
178 }
179 }
180
181 // Fill in the blocks
182 size_t index = 0;
183 for (size_t i = 0; i < variable_count; ++i) {
184 for (size_t j = i; j < variable_count; ++j) {
185 if (i == j) {
186 info_.setDiagonalBlock(I: i, xpr: Gs[index]);
187 } else {
188 info_.setOffDiagonalBlock(I: i, J: j, xpr: Gs[index]);
189 }
190 index++;
191 }
192 info_.setOffDiagonalBlock(I: i, J: variable_count, xpr: gs[i]);
193 }
194 constantTerm() = f;
195}

199void _FromJacobianHelper(const JacobianFactor& jf, SymmetricBlockMatrix& info) {
200 gttic(HessianFactor_fromJacobian);
201 const SharedDiagonal& jfModel = jf.get_model();
202 auto A = jf.matrixObject().full();
203 if (jfModel) {
204 if (jf.get_model()->isConstrained())
205 throw invalid_argument(
206 "Cannot construct HessianFactor from JacobianFactor with constrained noise model");
207
208 auto D = (jfModel->invsigmas().array() * jfModel->invsigmas().array()).matrix().asDiagonal();
209
210 info.setFullMatrix(A.transpose() * D * A);
211 } else {
212 info.setFullMatrix(A.transpose() * A);
213 }
214}
215}

224/* ************************************************************************* */
225HessianFactor::HessianFactor(const GaussianFactor& gf) :
226 GaussianFactor(gf) {
227 // Copy the matrix data depending on what type of factor we're copying from
228 if (const JacobianFactor* jf = dynamic_cast<const JacobianFactor*>(&gf)) {
229 info_ = SymmetricBlockMatrix::LikeActiveViewOf(other: jf->matrixObject());
230 _FromJacobianHelper(jf: *jf, info&: info_);
231 } else if (const HessianFactor* hf = dynamic_cast<const HessianFactor*>(&gf)) {
232 info_ = hf->info_;
233 } else {
234 throw std::invalid_argument(
235 "In HessianFactor(const GaussianFactor& gf), gf is neither a JacobianFactor nor a HessianFactor");
236 }
237}

240HessianFactor::HessianFactor(const GaussianFactorGraph& factors,
241 const Scatter& scatter) {
242 gttic(HessianFactor_MergeConstructor);
243
244 Allocate(scatter);
245
246 // Form A' * A
247 gttic(update);
248 info_.setAllZero();
249 for(const auto& factor: factors)
250 if (factor)
251 factor->updateHessian(keys: keys_, info: &info_);
252 gttoc(update);
253}

293void HessianFactor::hessianDiagonalAdd(VectorValues &d) const {
294 for (DenseIndex j = 0; j < (DenseIndex)size(); ++j) {
295 auto result = d.emplace(j: keys_[j], args: info_.diagonal(J: j));
296 if(!result.second) {
297 // if emplace fails, it returns an iterator to the existing element, which we add to:
298 result.first->second += info_.diagonal(J: j);
299 }
300 }

331/* ************************************************************************* */
332double HessianFactor::error(const VectorValues& c) const {
333 // error 0.5*(f - 2*x'*g + x'*G*x)
334 const double f = constantTerm();
335 if (empty()) {
336 return 0.5 * f;
337 }
338 double xtg = 0, xGx = 0;
339 // extract the relevant subset of the VectorValues
340 // NOTE may not be as efficient
341 const Vector x = c.vector(keys: keys());
342 xtg = x.dot(other: linearTerm().col(i: 0));
343 auto AtA = informationView();
344 xGx = x.transpose() * AtA * x;
345 return 0.5 * (f - 2.0 * xtg + xGx);
346}

348/* ************************************************************************* */
349void HessianFactor::updateHessian(const KeyVector& infoKeys,
350 SymmetricBlockMatrix* info) const {
351 assert(info);
352 gttic(updateHessian_HessianFactor);
353 const DenseIndex nrVariablesInThisFactor = size();
354
355 vector<DenseIndex> slots(nrVariablesInThisFactor + 1);
356 for (DenseIndex j = 0; j < nrVariablesInThisFactor; ++j)
357 slots[j] = Slot(keys: infoKeys, key: keys_[j]);
358 slots[nrVariablesInThisFactor] = info->nBlocks() - 1;
359
360 // Apply updates to the upper triangle
361 // Loop over this factor's blocks with indices (i,j)
362 // For every block (i,j), we determine the block (I,J) in info.
363 for (DenseIndex j = 0; j <= nrVariablesInThisFactor; ++j) {
364 const DenseIndex J = slots[j];
365 info->updateDiagonalBlock(I: J, xpr: info_.diagonalBlock(J: j));
366 for (DenseIndex i = 0; i < j; ++i) {
367 const DenseIndex I = slots[i];
368 assert(i < j);
369 assert(I != J);
370 info->updateOffDiagonalBlock(I, J, xpr: info_.aboveDiagonalBlock(I: i, J: j));
371 }
372 }
373}

383/* ************************************************************************* */
384void HessianFactor::multiplyHessianAdd(double alpha, const VectorValues& x,
385 VectorValues& yvalues) const {
386
387 // Create a vector of temporary y values, corresponding to rows i
388 vector<Vector> y;
389 y.reserve(n: size());
390 for (const_iterator it = begin(); it != end(); it++)
391 y.push_back(x: Vector::Zero(size: getDim(variable: it)));
392
393 // Accessing the VectorValues one by one is expensive
394 // So we will loop over columns to access x only once per column
395 // And fill the above temporary y values, to be added into yvalues after
396 for (DenseIndex j = 0; j < (DenseIndex) size(); ++j) {
397 // xj is the input vector
398 Vector xj = x.at(j: keys_[j]);
399 DenseIndex i = 0;
400 for (; i < j; ++i)
401 y[i] += info_.aboveDiagonalBlock(I: i, J: j) * xj;
402
403 // blocks on the diagonal are only half
404 y[i] += info_.diagonalBlock(J: j) * xj;
405 // for below diagonal, we take transpose block from upper triangular part
406 for (i = j + 1; i < (DenseIndex) size(); ++i)
407 y[i] += info_.aboveDiagonalBlock(I: j, J: i).transpose() * xj;
408 }
409
410 // copy to yvalues
411 for (DenseIndex i = 0; i < (DenseIndex) size(); ++i) {
412 const auto [it, didNotExist] = yvalues.tryInsert(j: keys_[i], value: Vector());
413 if (didNotExist)
414 it->second = alpha * y[i]; // init
415 else
416 it->second += alpha * y[i]; // add
417 }
418}

436/* ************************************************************************* */
437Vector HessianFactor::gradient(Key key, const VectorValues& x) const {
438 const Factor::const_iterator it_i = find(key);
439 const DenseIndex I = std::distance(first: begin(), last: it_i);
440
441 // Sum over G_ij*xj for all xj connecting to xi
442 Vector b = Vector::Zero(size: x.at(j: key).size());
443 for (Factor::const_iterator it_j = begin(); it_j != end(); ++it_j) {
444 const DenseIndex J = std::distance(first: begin(), last: it_j);
445
446 // Obtain Gij from the Hessian factor
447 // Hessian factor only stores an upper triangular matrix, so be careful when i>j
448 const Matrix Gij = info_.block(I, J);
449 // Accumulate Gij*xj to gradf
450 b += Gij * x.at(j: *it_j);
451 }
452 // Subtract the linear term gi
453 b += -linearTerm(j: it_i);
454 return b;
455}

458std::shared_ptr<GaussianConditional> HessianFactor::eliminateCholesky(const Ordering& keys) {
459 gttic(HessianFactor_eliminateCholesky);
460
461 GaussianConditional::shared_ptr conditional;
462
463 try {
464 // Do dense elimination
465 size_t nFrontals = keys.size();
466 assert(nFrontals <= size());
467 info_.choleskyPartial(nFrontals);
468
469 // TODO(frank): pre-allocate GaussianConditional and write into it
470 VerticalBlockMatrix Ab = info_.split(nFrontals);
471 conditional = std::make_shared<GaussianConditional>(args&: keys_, args&: nFrontals, args: std::move(Ab));
472
473 // Erase the eliminated keys in this factor
474 keys_.erase(first: begin(), last: begin() + nFrontals);
475 } catch (const CholeskyFailed&) {
476#ifndef NDEBUG
477 cout << "Partial Cholesky on HessianFactor failed." << endl;
478 keys.print("Frontal keys ");
479 print("HessianFactor:");
480#endif
481 throw IndeterminantLinearSystemException(keys.front());
482 }
483
484 // Return result
485 return conditional;
486}

488/* ************************************************************************* */
489VectorValues HessianFactor::solve() {
490 gttic(HessianFactor_solve);
491
492 // Do Cholesky Factorization
493 const size_t n = size();
494 assert(size_t(info_.nBlocks()) == n + 1);
495 info_.choleskyPartial(nFrontals: n);
496 auto R = info_.triangularView(I: 0, J: n);
497 auto eta = linearTerm();
498
499 // Solve
500 const Vector solution = R.solve(other: eta);
501
502 // Parse into VectorValues
503 VectorValues delta;
504 std::size_t offset = 0;
505 for (DenseIndex j = 0; j < (DenseIndex)n; ++j) {
506 const DenseIndex dim = info_.getDim(block: j);
507 delta.emplace(j: keys_[j], args: solution.segment(start: offset, n: dim));
508 offset += dim;
509 }
510
511 return delta;
512}

514/* ************************************************************************* */
515std::pair<std::shared_ptr<GaussianConditional>, std::shared_ptr<HessianFactor> >
516EliminateCholesky(const GaussianFactorGraph& factors, const Ordering& keys) {
517 gttic(EliminateCholesky);
518
519 // Build joint factor
520 HessianFactor::shared_ptr jointFactor;
521 try {
522 Scatter scatter(factors, keys);
523 jointFactor = std::make_shared<HessianFactor>(args: factors, args&: scatter);
524 } catch (std::invalid_argument&) {
525 throw InvalidDenseElimination(
526 "EliminateCholesky was called with a request to eliminate variables that are not\n"
527 "involved in the provided factors.");
528 }
529
530 // Do dense elimination
531 auto conditional = jointFactor->eliminateCholesky(keys);
532
533 // Return result
534 return make_pair(x&: conditional, y&: jointFactor);
535}

537/* ************************************************************************* */
538std::pair<std::shared_ptr<GaussianConditional>,
539 std::shared_ptr<GaussianFactor> > EliminatePreferCholesky(
540 const GaussianFactorGraph& factors, const Ordering& keys) {
541 gttic(EliminatePreferCholesky);
542
543 // If any JacobianFactors have constrained noise models, we have to convert
544 // all factors to JacobianFactors. Otherwise, we can convert all factors
545 // to HessianFactors. This is because QR can handle constrained noise
546 // models but Cholesky cannot.
547 if (hasConstraints(factors))
548 return EliminateQR(factors, keys);
549 else
550 return EliminateCholesky(factors, keys);
551}
