12/**

14 * @brief A factor with a quadratic error function - a Gaussian
15 * @brief GaussianFactor

19// \callgraph
20
21#pragma once
22
23#include <gtsam/inference/Factor.h>
24#include <gtsam/base/Matrix.h>
25#include <gtsam/base/Testable.h>
26
27namespace gtsam {
28
29 // Forward declarations
30 class VectorValues;
31 class Scatter;
32 class SymmetricBlockMatrix;
33
34 /**
35 * An abstract virtual base class for JacobianFactor and HessianFactor. A GaussianFactor has a
36 * quadratic error function. GaussianFactor is non-mutable (all methods const!). The factor value
37 * is exp(-0.5*||Ax-b||^2) */
38 class GTSAM_EXPORT GaussianFactor : public Factor
39 {
40 public:
41 typedef GaussianFactor This; ///< This class
42 typedef std::shared_ptr<This> shared_ptr; ///< shared_ptr to this class
43 typedef Factor Base; ///< Our base class
44
45 /// @name Standard Constructors
46 /// @{
47
48 /** Default constructor creates empty factor */
49 GaussianFactor() {}
50
51 /** Construct from container of keys. This constructor is used internally from derived factor
52 * constructors, either from a container of keys or from a boost::assign::list_of. */
53 template<typename CONTAINER>
54 GaussianFactor(const CONTAINER& keys) : Base(keys) {}
55
56 /// @}
57 /// @name Testable
58 /// @{
59
60 /// print with optional string
61 void print(
62 const std::string& s = "",
63 const KeyFormatter& formatter = DefaultKeyFormatter) const override = 0;
64
65 /// assert equality up to a tolerance
66 virtual bool equals(const GaussianFactor& lf, double tol = 1e-9) const = 0;
67
68 /// @}
69 /// @name Standard Interface
70 /// @{
71
72 /**
73 * In Gaussian factors, the error function returns either the negative log-likelihood, e.g.,
74 * 0.5*(A*x-b)'*D*(A*x-b)
75 * for a \class JacobianFactor, or the negative log-density, e.g.,
76 * 0.5*(A*x-b)'*D*(A*x-b) - log(k)
77 * for a \class GaussianConditional, where k is the normalization constant.
78 */
79 virtual double error(const VectorValues& c) const;
80
81 /**
82 * The Factor::error simply extracts the \class VectorValues from the
83 * \class HybridValues and calculates the error.
84 */
85 double error(const HybridValues& c) const override;
86
87 /** Return the dimension of the variable pointed to by the given key iterator */
88 virtual DenseIndex getDim(const_iterator variable) const = 0;
89
90 /**
91 * Return a dense \f$ [ \;A\;b\; ] \in \mathbb{R}^{m \times n+1} \f$
92 * Jacobian matrix, augmented with b with the noise models baked
93 * into A and b. The negative log-likelihood is
94 * \f$ \frac{1}{2} \Vert Ax-b \Vert^2 \f$. See also
95 * GaussianFactorGraph::jacobian and GaussianFactorGraph::sparseJacobian.
96 */
97 virtual Matrix augmentedJacobian() const = 0;
98
99 /**
100 * Return the dense Jacobian \f$ A \f$ and right-hand-side \f$ b \f$,
101 * with the noise models baked into A and b. The negative log-likelihood
102 * is \f$ \frac{1}{2} \Vert Ax-b \Vert^2 \f$. See also
103 * GaussianFactorGraph::augmentedJacobian and
104 * GaussianFactorGraph::sparseJacobian.
105 */
106 virtual std::pair<Matrix,Vector> jacobian() const = 0;
107
108 /** Return the augmented information matrix represented by this GaussianFactor.
109 * The augmented information matrix contains the information matrix with an
110 * additional column holding the information vector, and an additional row
111 * holding the transpose of the information vector. The lower-right entry
112 * contains the constant error term (when \f$ \delta x = 0 \f$). The
113 * augmented information matrix is described in more detail in HessianFactor,
114 * which in fact stores an augmented information matrix.
115 */
116 virtual Matrix augmentedInformation() const = 0;
117
118 /** Return the non-augmented information matrix represented by this
119 * GaussianFactor.
120 */
121 virtual Matrix information() const = 0;
122
123 /// Return the diagonal of the Hessian for this factor
124 VectorValues hessianDiagonal() const;
125
126 /// Add the current diagonal to a VectorValues instance
127 virtual void hessianDiagonalAdd(VectorValues& d) const = 0;
128
129 /// Raw memory access version of hessianDiagonal
130 virtual void hessianDiagonal(double* d) const = 0;
131
132 /// Return the block diagonal of the Hessian for this factor
133 virtual std::map<Key,Matrix> hessianBlockDiagonal() const = 0;
134
135 /** Clone a factor (make a deep copy) */
136 virtual GaussianFactor::shared_ptr clone() const = 0;
137
138 /**
139 * Construct the corresponding anti-factor to negate information
140 * stored stored in this factor.
141 * @return a HessianFactor with negated Hessian matrices
142 */
143 virtual GaussianFactor::shared_ptr negate() const = 0;
144
145 /** Update an information matrix by adding the information corresponding to this factor
146 * (used internally during elimination).
147 * @param scatter A mapping from variable index to slot index in this HessianFactor
148 * @param info The information matrix to be updated
149 */
150 virtual void updateHessian(const KeyVector& keys,
151 SymmetricBlockMatrix* info) const = 0;
152
153 /// @}
154 /// @name Operator interface
155 /// @{
156
157 /// y += alpha * A'*A*x
158 virtual void multiplyHessianAdd(double alpha, const VectorValues& x, VectorValues& y) const = 0;
159
160 /// A'*b for Jacobian, eta for Hessian
161 virtual VectorValues gradientAtZero() const = 0;
162
163 /// Raw memory access version of gradientAtZero
164 virtual void gradientAtZero(double* d) const = 0;
165
166 /// Gradient wrt a key at any values
167 virtual Vector gradient(Key key, const VectorValues& x) const = 0;
168
169 /// @}
170 /// @name Advanced Interface
171 /// @{
172
173 // Determine position of a given key
174 template <typename CONTAINER>
175 static DenseIndex Slot(const CONTAINER& keys, Key key) {
176 return std::find(keys.begin(), keys.end(), key) - keys.begin();
177 }
178
179 /// @}
180
181 private:
182#if GTSAM_ENABLE_BOOST_SERIALIZATION
183 /** Serialization function */
184 friend class boost::serialization::access;
185 template<class ARCHIVE>
186 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
187 ar & BOOST_SERIALIZATION_BASE_OBJECT_NVP(Base);
188 }
189#endif
190 }; // GaussianFactor
191
192/// traits
193template<>
194struct traits<GaussianFactor> : public Testable<GaussianFactor> {
195};
