28/* ************************************************************************* */
29GaussianDensity GaussianDensity::FromMeanAndStddev(Key key, const Vector& mean,
30 double sigma) {
31 return GaussianDensity(key, mean, Matrix::Identity(rows: mean.size(), cols: mean.size()),
32 noiseModel::Isotropic::Sigma(dim: mean.size(), sigma));
33}

35/* ************************************************************************* */
36void GaussianDensity::print(const string& s,
37 const KeyFormatter& formatter) const {
38 cout << s << ": density on ";
39 for (const_iterator it = beginFrontals(); it != endFrontals(); ++it) {
40 cout << "[" << formatter(*it) << "] ";
41 }
42 cout << endl;
43 gtsam::print(v: mean(), s: "mean: ");
44 gtsam::print(A: covariance(), s: "covariance: ");
45 if (model_) model_->print(name: "Noise model: ");
46}
47
48/* ************************************************************************* */
49Vector GaussianDensity::mean() const {
50 VectorValues soln = this->solve(parents: VectorValues());
51 return soln[firstFrontalKey()];
52}
53
54/* ************************************************************************* */
55Matrix GaussianDensity::covariance() const { return information().inverse(); }
