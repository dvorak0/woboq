31 /**
32 * GaussianBayesNet is a Bayes net made from linear-Gaussian conditionals.
33 * @ingroup linear
34 */
35 class GTSAM_EXPORT GaussianBayesNet: public BayesNet<GaussianConditional>
36 {
37 public:
38
39 typedef BayesNet<GaussianConditional> Base;
40 typedef GaussianBayesNet This;
41 typedef GaussianConditional ConditionalType;
42 typedef std::shared_ptr<This> shared_ptr;
43 typedef std::shared_ptr<ConditionalType> sharedConditional;
44
45 /// @name Standard Constructors
46 /// @{
47
48 /** Construct empty bayes net */
49 GaussianBayesNet() {}
50
51 /** Construct from iterator over conditionals */
52 template <typename ITERATOR>
53 GaussianBayesNet(ITERATOR firstConditional, ITERATOR lastConditional)
54 : Base(firstConditional, lastConditional) {}
55
56 /** Construct from container of factors (shared_ptr or plain objects) */
57 template <class CONTAINER>
58 explicit GaussianBayesNet(const CONTAINER& conditionals) {
59 push_back(conditionals);
60 }
61
62 /** Implicit copy/downcast constructor to override explicit template
63 * container constructor */
64 template <class DERIVEDCONDITIONAL>
65 explicit GaussianBayesNet(const FactorGraph<DERIVEDCONDITIONAL>& graph)
66 : Base(graph) {}
67
68 /**
69 * Constructor that takes an initializer list of shared pointers.
70 * BayesNet bn = {make_shared<Conditional>(), ...};
71 */
72 template <class DERIVEDCONDITIONAL>
73 GaussianBayesNet(
74 std::initializer_list<std::shared_ptr<DERIVEDCONDITIONAL> > conditionals)
75 : Base(conditionals) {}
76
77 /// @}
78
79 /// @name Testable
80 /// @{
81
82 /** Check equality */
83 bool equals(const This& bn, double tol = 1e-9) const;
84
85 /// Check exact equality.
86 friend bool operator==(const GaussianBayesNet& lhs,
87 const GaussianBayesNet& rhs) {
88 return lhs.isEqual(other: rhs);
89 }
90
91 /// print graph
92 void print(
93 const std::string& s = "",
94 const KeyFormatter& formatter = DefaultKeyFormatter) const override {
95 Base::print(s, formatter);
96 }
97
98 /// @}
99
100 /// @name Standard Interface
101 /// @{
102
103 /// Sum error over all variables.
104 double error(const VectorValues& x) const;
105
106 /// Sum logProbability over all variables.
107 double logProbability(const VectorValues& x) const;
108
109 /**
110 * Calculate probability density for given values `x`:
111 * exp(logProbability)
112 * where x is the vector of values.
113 */
114 double evaluate(const VectorValues& x) const;
115
116 /// Evaluate probability density, sugar.
117 double operator()(const VectorValues& x) const {
118 return evaluate(x);
119 }
120
121 /// Solve the GaussianBayesNet, i.e. return \f$ x = R^{-1}*d \f$, by
122 /// back-substitution
123 VectorValues optimize() const;
124
125 /// Version of optimize for incomplete BayesNet, given missing variables
126 VectorValues optimize(const VectorValues& given) const;
127
128 /**
129 * Sample using ancestral sampling
130 * Example:
131 * std::mt19937_64 rng(42);
132 * auto sample = gbn.sample(&rng);
133 */
134 VectorValues sample(std::mt19937_64* rng = nullptr) const;
135
136 /**
137 * Sample from an incomplete BayesNet, given missing variables
138 * Example:
139 * std::mt19937_64 rng(42);
140 * VectorValues given = ...;
141 * auto sample = gbn.sample(given, &rng);
142 */
143 VectorValues sample(const VectorValues& given, std::mt19937_64* rng = nullptr) const;
144
145 /**
146 * Return ordering corresponding to a topological sort.
147 * There are many topological sorts of a Bayes net. This one
148 * corresponds to the one that makes 'matrix' below upper-triangular.
149 * In case Bayes net is incomplete any non-frontal are added to the end.
150 */
151 Ordering ordering() const;
152
153 ///@}
154
155 ///@name Linear Algebra
156 ///@{
157
158 /**
159 * Return (dense) upper-triangular matrix representation
160 * Will return upper-triangular matrix only when using 'ordering' above.
161 * In case Bayes net is incomplete zero columns are added to the end.
162 */
163 std::pair<Matrix, Vector> matrix(const Ordering& ordering) const;
164
165 /**
166 * Return (dense) upper-triangular matrix representation
167 * Will return upper-triangular matrix only when using 'ordering' above.
168 * In case Bayes net is incomplete zero columns are added to the end.
169 */
170 std::pair<Matrix, Vector> matrix() const;
171
172 /**
173 * Optimize along the gradient direction, with a closed-form computation to perform the line
174 * search. The gradient is computed about \f$ \delta x=0 \f$.
175 *
176 * This function returns \f$ \delta x \f$ that minimizes a reparametrized problem. The error
177 * function of a GaussianBayesNet is
178 *
179 * \f[ f(\delta x) = \frac{1}{2} |R \delta x - d|^2 = \frac{1}{2}d^T d - d^T R \delta x +
180 * \frac{1}{2} \delta x^T R^T R \delta x \f]
181 *
182 * with gradient and Hessian
183 *
184 * \f[ g(\delta x) = R^T(R\delta x - d), \qquad G(\delta x) = R^T R. \f]
185 *
186 * This function performs the line search in the direction of the gradient evaluated at \f$ g =
187 * g(\delta x = 0) \f$ with step size \f$ \alpha \f$ that minimizes \f$ f(\delta x = \alpha g)
188 * \f$:
189 *
190 * \f[ f(\alpha) = \frac{1}{2} d^T d + g^T \delta x + \frac{1}{2} \alpha^2 g^T G g \f]
191 *
192 * Optimizing by setting the derivative to zero yields \f$ \hat \alpha = (-g^T g) / (g^T G g)
193 * \f$. For efficiency, this function evaluates the denominator without computing the Hessian
194 * \f$ G \f$, returning
195 *
196 * \f[ \delta x = \hat\alpha g = \frac{-g^T g}{(R g)^T(R g)} \f] */
197 VectorValues optimizeGradientSearch() const;
198
199 /** Compute the gradient of the energy function, \f$ \nabla_{x=x_0} \left\Vert \Sigma^{-1} R x -
200 * d \right\Vert^2 \f$, centered around \f$ x = x_0 \f$. The gradient is \f$ R^T(Rx-d) \f$.
201 *
202 * @param x0 The center about which to compute the gradient
203 * @return The gradient as a VectorValues */
204 VectorValues gradient(const VectorValues& x0) const;
205
206 /** Compute the gradient of the energy function, \f$ \nabla_{x=0} \left\Vert \Sigma^{-1} R x - d
207 * \right\Vert^2 \f$, centered around zero. The gradient about zero is \f$ -R^T d \f$. See also
208 * gradient(const GaussianBayesNet&, const VectorValues&).
209 *
210 * @param [output] g A VectorValues to store the gradient, which must be preallocated, see
211 * allocateVectorValues */
212 VectorValues gradientAtZero() const;
213
214 /**
215 * Computes the determinant of a GassianBayesNet. A GaussianBayesNet is an upper triangular
216 * matrix and for an upper triangular matrix determinant is the product of the diagonal
217 * elements. Instead of actually multiplying we add the logarithms of the diagonal elements and
218 * take the exponent at the end because this is more numerically stable.
219 * @param bayesNet The input GaussianBayesNet
220 * @return The determinant */
221 double determinant() const;
222
223 /**
224 * Computes the log of the determinant of a GassianBayesNet. A GaussianBayesNet is an upper
225 * triangular matrix and for an upper triangular matrix determinant is the product of the
226 * diagonal elements.
227 * @param bayesNet The input GaussianBayesNet
228 * @return The determinant */
229 double logDeterminant() const;
230
231 /**
232 * @brief Get the negative log of the normalization constant corresponding
233 * to the joint Gaussian density represented by this Bayes net.
234 *
235 * @return double
236 */
237 double negLogConstant() const;
238
239 /**
240 * Backsubstitute with a different RHS vector than the one stored in this BayesNet.
241 * gy=inv(R*inv(Sigma))*gx
242 */
243 VectorValues backSubstitute(const VectorValues& gx) const;
244
245 /**
246 * Transpose backsubstitute with a different RHS vector than the one stored in this BayesNet.
247 * gy=inv(L)*gx by solving L*gy=gx.
248 * gy=inv(R'*inv(Sigma))*gx
249 * gz'*R'=gx', gy = gz.*sigmas
250 */
251 VectorValues backSubstituteTranspose(const VectorValues& gx) const;
252
253 /// @}
254 /// @name HybridValues methods.
255 /// @{
256
257 using Base::evaluate; // Expose evaluate(const HybridValues&) method..
258 using Base::logProbability; // Expose logProbability(const HybridValues&) method..
259 using Base::error; // Expose error(const HybridValues&) method..
260
261 /// @}
262

269 ar & BOOST_SERIALIZATION_BASE_OBJECT_NVP(Base);
270 }
271#endif
272 };
