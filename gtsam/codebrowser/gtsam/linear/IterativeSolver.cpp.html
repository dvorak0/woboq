61/*****************************************************************************/
62IterativeOptimizationParameters::Verbosity IterativeOptimizationParameters::verbosityTranslator(
63 const string &src) {
64 string s = src;
65 // Convert to upper case
66 std::transform(first: s.begin(), last: s.end(), result: s.begin(), unary_op: ::toupper);
67 if (s == "SILENT")
68 return IterativeOptimizationParameters::SILENT;
69 else if (s == "COMPLEXITY")
70 return IterativeOptimizationParameters::COMPLEXITY;
71 else if (s == "ERROR")
72 return IterativeOptimizationParameters::ERROR;
73 /* default is default */
74 else
75 return IterativeOptimizationParameters::SILENT;
76}

78/*****************************************************************************/
79string IterativeOptimizationParameters::verbosityTranslator(
80 IterativeOptimizationParameters::Verbosity verbosity) {
81 if (verbosity == SILENT)
82 return "SILENT";
83 else if (verbosity == COMPLEXITY)
84 return "COMPLEXITY";
85 else if (verbosity == ERROR)
86 return "ERROR";
87 else
88 return "UNKNOWN";
89}

116/****************************************************************************/
117void KeyInfo::initialize(const GaussianFactorGraph &fg) {
118 const map<Key, size_t> colspec = fg.getKeyDimMap();
119 const size_t n = ordering_.size();
120 size_t start = 0;
121
122 for (size_t i = 0; i < n; ++i) {
123 const Key key = ordering_[i];
124 const auto it_key = colspec.find(x: key);
125 if (it_key==colspec.end())
126 throw std::runtime_error("KeyInfo: Inconsistency in key-dim map");
127 const size_t dim = it_key->second;
128 this->emplace(args: key, args: KeyInfoEntry(i, dim, start));
129 start += dim;
130 }
131 numCols_ = start;
132}
133
134/****************************************************************************/
135vector<size_t> KeyInfo::colSpec() const {
136 std::vector<size_t> result(size(), 0);
137 for ( const auto &item: *this ) {
138 result[item.second.index] = item.second.dim;
139 }
140 return result;
141}
142
143/****************************************************************************/
144VectorValues KeyInfo::x0() const {
145 VectorValues result;
146 for ( const auto &item: *this ) {
147 result.emplace(j: item.first, args: Vector::Zero(size: item.second.dim));
148 }
149 return result;
150}
151
152/****************************************************************************/
153Vector KeyInfo::x0vector() const {
154 return Vector::Zero(size: numCols_);
155}
