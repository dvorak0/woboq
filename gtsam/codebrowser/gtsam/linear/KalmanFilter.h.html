12/**
13 * @file KalmanFilter.h
14 * @brief Simple linear Kalman filter implemented using factor graphs, i.e.,
15 * performs Cholesky or QR-based SRIF (Square-Root Information Filter).
16 * @date Sep 3, 2011
17 * @authors Stephen Williams, Frank Dellaert
18 */
19
20#pragma once
21
22#include <gtsam/linear/GaussianDensity.h>
23#include <gtsam/linear/GaussianFactorGraph.h>
24#include <gtsam/linear/NoiseModel.h>
25
26#ifndef KALMANFILTER_DEFAULT_FACTORIZATION
27#define KALMANFILTER_DEFAULT_FACTORIZATION QR
28#endif
29
30namespace gtsam {
31
32/**
33 * Kalman Filter class
34 *
35 * Maintains a Gaussian density under linear-Gaussian motion and
36 * measurement models using the square-root information form.
37 *
38 * The filter is functional; it does not maintain internal state. Instead:
39 * - Use `init()` to create an initial filter state,
40 * - Call `predict()` and `update()` to create new states.
41 */
42class GTSAM_EXPORT KalmanFilter {
43 public:
44 /**
45 * @enum Factorization
46 * @brief Specifies the factorization variant to use.
47 */
48 enum Factorization { QR, CHOLESKY };
49
50 /**
51 * @typedef State
52 * @brief The Kalman filter state, represented as a shared pointer to a
53 * GaussianDensity.
54 */
55 typedef GaussianDensity::shared_ptr State;
56
57 private:
58 const size_t n_; ///< Dimensionality of the state.
59 const Matrix I_; ///< Identity matrix of size \f$ n \times n \f$.
60 const GaussianFactorGraph::Eliminate
61 function_; ///< Elimination algorithm used.
62
63 /**
64 * Solve the factor graph.
65 * @param factorGraph The Gaussian factor graph to solve.
66 * @return The resulting Kalman filter state.
67 */
68 State solve(const GaussianFactorGraph& factorGraph) const;
69
70 /**
71 * Fuse two states.
72 * @param p The prior state.
73 * @param newFactor The new factor to incorporate.
74 * @return The resulting fused state.
75 */
76 State fuse(const State& p, GaussianFactor::shared_ptr newFactor) const;
77
78 public:
79 /**
80 * Constructor.
81 * @param n Dimensionality of the state.
82 * @param method Factorization method (default: QR unless compile-flag set).
83 */
84 KalmanFilter(size_t n,
85 Factorization method = KALMANFILTER_DEFAULT_FACTORIZATION)
86 : n_(n),
87 I_(Matrix::Identity(rows: n_, cols: n_)),
88 function_(method == QR
89 ? GaussianFactorGraph::Eliminate(EliminateQR)
90 : GaussianFactorGraph::Eliminate(EliminateCholesky)) {}
91
92 /**
93 * Create the initial state (prior density at time \f$ k=0 \f$).
94 *
95 * In Kalman Filter notation:
96 * - \f$ x_{0|0} \f$: Initial state estimate.
97 * - \f$ P_{0|0} \f$: Initial covariance matrix.
98 *
99 * @param x0 Estimate of the state at time 0 (\f$ x_{0|0} \f$).
100 * @param P0 Covariance matrix (\f$ P_{0|0} \f$), given as a diagonal Gaussian
101 * model.
102 * @return Initial Kalman filter state.
103 */
104 State init(const Vector& x0, const SharedDiagonal& P0) const;
105
106 /**
107 * Create the initial state with a full covariance matrix.
108 * @param x0 Initial state estimate.
109 * @param P0 Full covariance matrix.
110 * @return Initial Kalman filter state.
111 */
112 State init(const Vector& x0, const Matrix& P0) const;
113
114 /**
115 * Print the Kalman filter details.
116 * @param s Optional string prefix.
117 */
118 void print(const std::string& s = "") const;
119
120 /**
121 * Return the step index \f$ k \f$ (starts at 0, incremented at each predict
122 * step).
123 * @param p The current state.
124 * @return Step index.
125 */
126 static Key step(const State& p) { return p->firstFrontalKey(); }
127
128 /**
129 * Predict the next state \f$ P(x_{k+1}|Z^k) \f$.
130 *
131 * In Kalman Filter notation:
132 * - \f$ x_{k+1|k} \f$: Predicted state.
133 * - \f$ P_{k+1|k} \f$: Predicted covariance.
134 *
135 * Motion model:
136 * \f[
137 * x_{k+1} = F \cdot x_k + B \cdot u_k + w
138 * \f]
139 * where \f$ w \f$ is zero-mean Gaussian noise with covariance \f$ Q \f$.
140 *
141 * @param p Previous state (\f$ x_k \f$).
142 * @param F State transition matrix (\f$ F \f$).
143 * @param B Control input matrix (\f$ B \f$).
144 * @param u Control vector (\f$ u_k \f$).
145 * @param modelQ Noise model (\f$ Q \f$, diagonal Gaussian).
146 * @return Predicted state (\f$ x_{k+1|k} \f$).
147 */
148 State predict(const State& p, const Matrix& F, const Matrix& B,
149 const Vector& u, const SharedDiagonal& modelQ) const;
150
151 /**
152 * Predict the next state with a full covariance matrix.
153 *
154 *@note Q is normally derived as G*w*G^T where w models uncertainty of some
155 * physical property, such as velocity or acceleration, and G is derived from
156 * physics. This version allows more realistic models than a diagonal matrix.
157 *
158 * @param p Previous state.
159 * @param F State transition matrix.
160 * @param B Control input matrix.
161 * @param u Control vector.
162 * @param Q Full covariance matrix (\f$ Q \f$).
163 * @return Predicted state.
164 */
165 State predictQ(const State& p, const Matrix& F, const Matrix& B,
166 const Vector& u, const Matrix& Q) const;
167
168 /**
169 * Predict the next state using a GaussianFactor motion model.
170 * @param p Previous state.
171 * @param A0 Factor matrix.
172 * @param A1 Factor matrix.
173 * @param b Constant term vector.
174 * @param model Noise model (optional).
175 * @return Predicted state.
176 */
177 State predict2(const State& p, const Matrix& A0, const Matrix& A1,
178 const Vector& b, const SharedDiagonal& model = nullptr) const;
179
180 /**
181 * Update the Kalman filter with a measurement.
182 *
183 * Observation model:
184 * \f[
185 * z_k = H \cdot x_k + v
186 * \f]
187 * where \f$ v \f$ is zero-mean Gaussian noise with covariance R.
188 * In this version, R is restricted to diagonal Gaussians (model parameter)
189 *
190 * @param p Previous state.
191 * @param H Observation matrix.
192 * @param z Measurement vector.
193 * @param model Noise model (diagonal Gaussian).
194 * @return Updated state.
195 */
196 State update(const State& p, const Matrix& H, const Vector& z,
197 const SharedDiagonal& model) const;
198
199 /**
200 * Update the Kalman filter with a measurement using a full covariance matrix.
201 * @param p Previous state.
202 * @param H Observation matrix.
203 * @param z Measurement vector.
204 * @param R Full covariance matrix.
205 * @return Updated state.
206 */
207 State updateQ(const State& p, const Matrix& H, const Vector& z,
208 const Matrix& R) const;
209
210 /**
211 * Return the dimensionality of the state.
212 * @return Dimensionality of the state.
213 */
214 size_t dim() const { return n_; }
215};
216
217} // namespace gtsam
