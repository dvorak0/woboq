32/**************************************************************************************************/
33// Just taking system [A|b]
34SubgraphSolver::SubgraphSolver(const GaussianFactorGraph &Ab,
35 const Parameters &parameters, const Ordering& ordering) :
36 parameters_(parameters) {
37 const auto [Ab1, Ab2] = splitGraph(gfg: Ab);
38 if (parameters_.verbosity())
39 cout << "Split A into (A1) " << Ab1.size() << " and (A2) " << Ab2.size()
40 << " factors" << endl;
41
42 auto Rc1 = *Ab1.eliminateSequential(ordering, function: EliminateQR);
43 auto xbar = Rc1.optimize();
44 pc_ = std::make_shared<SubgraphPreconditioner>(args: Ab2, args&: Rc1, args&: xbar);
45}
46
47/**************************************************************************************************/
48// Taking eliminated tree [R1|c] and constraint graph [A2|b2]
49SubgraphSolver::SubgraphSolver(const GaussianBayesNet &Rc1,
50 const GaussianFactorGraph &Ab2,
51 const Parameters &parameters)
52 : parameters_(parameters) {
53 auto xbar = Rc1.optimize();
54 pc_ = std::make_shared<SubgraphPreconditioner>(args: Ab2, args: Rc1, args&: xbar);
55}
56
57/**************************************************************************************************/
58// Taking subgraphs [A1|b1] and [A2|b2]
59// delegate up
60SubgraphSolver::SubgraphSolver(const GaussianFactorGraph &Ab1,
61 const GaussianFactorGraph &Ab2,
62 const Parameters &parameters,
63 const Ordering &ordering)
64 : SubgraphSolver(*Ab1.eliminateSequential(ordering, function: EliminateQR), Ab2,
65 parameters) {}

67/**************************************************************************************************/
68VectorValues SubgraphSolver::optimize() const {
69 VectorValues ybar = conjugateGradients<SubgraphPreconditioner, VectorValues,
70 Errors>(Ab: *pc_, x: pc_->zero(), parameters: parameters_);
71 return pc_->x(y: ybar);
72}
73
74VectorValues SubgraphSolver::optimize(const GaussianFactorGraph &gfg,
75 const KeyInfo &keyInfo, const map<Key, Vector> &lambda,
76 const VectorValues &initial) {
77 return VectorValues();
78}
79/**************************************************************************************************/
80pair<GaussianFactorGraph, GaussianFactorGraph> //
81SubgraphSolver::splitGraph(const GaussianFactorGraph &factorGraph) {
82
83 /* identify the subgraph structure */
84 const SubgraphBuilder builder(parameters_.builderParams);
85 auto subgraph = builder(factorGraph);
86
87 /* build factor subgraph */
88 return splitFactorGraph(factorGraph, subgraph);
89}
