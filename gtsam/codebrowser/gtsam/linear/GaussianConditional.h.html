32 /**
33 * A GaussianConditional functions as the node in a Bayes network.
34 * It has a set of parents y,z, etc. and implements a Gaussian probability density p(x | y, z) on x.
35 * The negative log-density is given by \f$ \frac{1}{2} |Rx - (d - Sy - Tz - ...)|^2 \f$
36 * The mean of the conditional density is \f$ R^{-1}(d - Sy - Tz - ...) \f$.
37 * The covariance of the conditional density is given by the noise model and is constrained to be diagonal.
38 * @ingroup linear
39 */
40 class GTSAM_EXPORT GaussianConditional :
41 public JacobianFactor,
42 public Conditional<JacobianFactor, GaussianConditional>
43 {
44 public:
45 typedef GaussianConditional This; ///< Typedef to this class
46 typedef std::shared_ptr<This> shared_ptr; ///< shared_ptr to this class
47 typedef JacobianFactor BaseFactor; ///< Typedef to our factor base class
48 typedef Conditional<BaseFactor, This> BaseConditional; ///< Typedef to our conditional base class
49
50 /// @name Constructors
51 /// @{
52
53 /** default constructor needed for serialization */
54 GaussianConditional() {}
55
56 /** constructor with no parents |Rx-d| */
57 GaussianConditional(Key key, const Vector& d, const Matrix& R,
58 const SharedDiagonal& sigmas = SharedDiagonal());
59
60 /** constructor with only one parent |Rx+Sy-d| */
61 GaussianConditional(Key key, const Vector& d, const Matrix& R, Key parent1,
62 const Matrix& S,
63 const SharedDiagonal& sigmas = SharedDiagonal());
64
65 /** constructor with two parents |Rx+Sy+Tz-d| */
66 GaussianConditional(Key key, const Vector& d, const Matrix& R, Key parent1,
67 const Matrix& S, Key parent2, const Matrix& T,
68 const SharedDiagonal& sigmas = SharedDiagonal());

70 /** Constructor with arbitrary number of frontals and parents.
71 * @tparam TERMS A container whose value type is std::pair<Key, Matrix>, specifying the
72 * collection of keys and matrices making up the conditional. */
73 template<typename TERMS>
74 GaussianConditional(const TERMS& terms,
75 size_t nrFrontals, const Vector& d,
76 const SharedDiagonal& sigmas = SharedDiagonal());
77
78 /**
79 * @brief Constructor with an arbitrary number of keys, where the augmented matrix
80 * is given all together instead of in block terms.
81 *
82 * @tparam KEYS Type of the keys container.
83 * @param keys Container of keys.
84 * @param nrFrontals Number of frontal variables.
85 * @param augmentedMatrix The augmented matrix containing the coefficients.
86 * @param sigmas Optional noise model (default is an empty SharedDiagonal).
87 */
88 template <typename KEYS>
89 GaussianConditional(const KEYS& keys, size_t nrFrontals,
90 const VerticalBlockMatrix& augmentedMatrix,
91 const SharedDiagonal& sigmas = SharedDiagonal());
92
93 /**
94 * @brief Constructor with an arbitrary number of keys, where the augmented matrix
95 * is given all together instead of in block terms, using move semantics for efficiency.
96 *
97 * @tparam KEYS Type of the keys container.
98 * @param keys Container of keys.
99 * @param nrFrontals Number of frontal variables.
100 * @param augmentedMatrix The augmented matrix containing the coefficients (moved).
101 * @param sigmas Optional noise model (default is an empty SharedDiagonal).
102 */
103 template <typename KEYS>
104 GaussianConditional(const KEYS& keys, size_t nrFrontals,
105 VerticalBlockMatrix&& augmentedMatrix,
106 const SharedDiagonal& sigmas = SharedDiagonal());

108 /// Construct from mean `mu` and standard deviation `sigma`.
109 static GaussianConditional FromMeanAndStddev(Key key, const Vector& mu,
110 double sigma);
111
112 /// Construct from conditional mean `A1 p1 + b` and standard deviation.
113 static GaussianConditional FromMeanAndStddev(Key key, const Matrix& A,
114 Key parent, const Vector& b,
115 double sigma);
116
117 /// Construct from conditional mean `A1 p1 + A2 p2 + b` and standard
118 /// deviation `sigma`.
119 static GaussianConditional FromMeanAndStddev(Key key, //
120 const Matrix& A1, Key parent1,
121 const Matrix& A2, Key parent2,
122 const Vector& b, double sigma);
123
124 /// Create shared pointer by forwarding arguments to fromMeanAndStddev.
125 template<typename... Args>
126 static shared_ptr sharedMeanAndStddev(Args&&... args) {
127 return std::make_shared<This>(FromMeanAndStddev(std::forward<Args>(args)...));
128 }
129
130 /** Combine several GaussianConditional into a single dense GC. The conditionals enumerated by
131 * \c first and \c last must be in increasing order, meaning that the parents of any
132 * conditional may not include a conditional coming before it.
133 * @param firstConditional Iterator to the first conditional to combine, must dereference to a
134 * shared_ptr<GaussianConditional>.
135 * @param lastConditional Iterator to after the last conditional to combine, must dereference
136 * to a shared_ptr<GaussianConditional>. */
137 template<typename ITERATOR>
138 static shared_ptr Combine(ITERATOR firstConditional, ITERATOR lastConditional);

152 /// @}
153 /// @name Standard Interface
154 /// @{
155
156 /**
157 * @brief Return the negative log of the normalization constant.
158 *
159 * normalization constant k = 1.0 / sqrt((2*pi)^n*det(Sigma))
160 * -log(k) = 0.5 * n*log(2*pi) + 0.5 * log det(Sigma)
161 *
162 * @return double
163 */
164 double negLogConstant() const override;
165
166 /**
167 * Calculate log-probability log(evaluate(x)) for given values `x`:
168 * -error(x) - 0.5 * n*log(2*pi) - 0.5 * log det(Sigma)
169 * where x is the vector of values, and Sigma is the covariance matrix.
170 * This differs from error as it is log, not negative log, and it
171 * includes the normalization constant.
172 */
173 double logProbability(const VectorValues& x) const;
174
175 /**
176 * Calculate probability density for given values `x`:
177 * exp(logProbability(x)) == exp(-GaussianFactor::error(x)) / sqrt((2*pi)^n*det(Sigma))
178 * where x is the vector of values, and Sigma is the covariance matrix.
179 */
180 double evaluate(const VectorValues& x) const;
181
182 /// Evaluate probability density, sugar.
183 double operator()(const VectorValues& x) const {
184 return evaluate(x);
185 }
186
187 /**
188 * Solves a conditional Gaussian and writes the solution into the entries of
189 * \c x for each frontal variable of the conditional. The parents are
190 * assumed to have already been solved in and their values are read from \c x.
191 * This function works for multiple frontal variables.
192 *
193 * Given the Gaussian conditional with log likelihood \f$ |R x_f - (d - S x_s)|^2 \f$,
194 * where \f$ f \f$ are the frontal variables and \f$ s \f$ are the separator
195 * variables of this conditional, this solve function computes
196 * \f$ x_f = R^{-1} (d - S x_s) \f$ using back-substitution.
197 *
198 * @param parents VectorValues containing solved parents \f$ x_s \f$.
199 */
200 VectorValues solve(const VectorValues& parents) const;
201
202 VectorValues solveOtherRHS(const VectorValues& parents, const VectorValues& rhs) const;
203
204 /** Performs transpose backsubstition in place on values */
205 void solveTransposeInPlace(VectorValues& gy) const;
206
207 /** Convert to a likelihood factor by providing value before bar. */
208 JacobianFactor::shared_ptr likelihood(
209 const VectorValues& frontalValues) const;
210
211 /** Single variable version of likelihood. */
212 JacobianFactor::shared_ptr likelihood(const Vector& frontal) const;

214 /**
215 * Sample from conditional, zero parent version
216 * Example:
217 * std::mt19937_64 rng(42);
218 * auto sample = gc.sample(&rng);
219 */
220 VectorValues sample(std::mt19937_64* rng = nullptr) const;
221
222 /**
223 * Sample from conditional, given missing variables
224 * Example:
225 * std::mt19937_64 rng(42);
226 * VectorValues given = ...;
227 * auto sample = gc.sample(given, &rng);
228 */
229 VectorValues sample(const VectorValues& parentsValues,
230 std::mt19937_64* rng = nullptr) const;

232 /// @}
233 /// @name Linear algebra.
234 /// @{
235
236 /** Return a view of the upper-triangular R block of the conditional */
237 constABlock R() const { return Ab_.range(startBlock: 0, endBlock: nrFrontals()); }
238
239 /** Get a view of the parent blocks. */
240 constABlock S() const { return Ab_.range(startBlock: nrFrontals(), endBlock: size()); }
241
242 /** Get a view of the S matrix for the variable pointed to by the given key iterator */
243 constABlock S(const_iterator it) const { return BaseFactor::getA(variable: it); }
244
245 /** Get a view of the r.h.s. vector d */
246 const constBVector d() const { return BaseFactor::getb(); }
247
248 /**
249 * @brief Compute the determinant of the R matrix.
250 *
251 * The determinant is computed in log form using logDeterminant for
252 * numerical stability and then exponentiated.
253 *
254 * Note, the covariance matrix \f$ \Sigma = (R^T R)^{-1} \f$, and hence
255 * \f$ \det(\Sigma) = 1 / \det(R^T R) = 1 / determinant()^ 2 \f$.
256 *
257 * @return double
258 */
259 inline double determinant() const { return exp(x: logDeterminant()); }
260
261 /**
262 * @brief Compute the log determinant of the R matrix.
263 *
264 * For numerical stability, the determinant is computed in log
265 * form, so it is a summation rather than a multiplication.
266 *
267 * Note, the covariance matrix \f$ \Sigma = (R^T R)^{-1} \f$, and hence
268 * \f$ \log \det(\Sigma) = - \log \det(R^T R) = - 2 logDeterminant() \f$.
269 *
270 * @return double
271 */
272 double logDeterminant() const;
273
