53/*****************************************************************************/
54VectorValues PCGSolver::optimize(const GaussianFactorGraph &gfg,
55 const KeyInfo &keyInfo, const std::map<Key, Vector> &lambda,
56 const VectorValues &initial) {
57 /* build preconditioner */
58 preconditioner_->build(gfg, info: keyInfo, lambda);
59
60 /* apply pcg */
61 GaussianFactorGraphSystem system(gfg, *preconditioner_, keyInfo, lambda);
62 Vector x0 = initial.vector(keys: keyInfo.ordering());
63 const Vector sol = preconditionedConjugateGradient(system, initial: x0, parameters: parameters_);
64
65 return buildVectorValues(v: sol, keyInfo);
66}

76/*****************************************************************************/
77void GaussianFactorGraphSystem::residual(const Vector &x, Vector &r) const {
78 /* implement b-Ax, assume x and r are pre-allocated */
79
80 /* reset r to b */
81 getb(b&: r);
82
83 /* substract A*x */
84 Vector Ax = Vector::Zero(rows: r.rows(), cols: 1);
85 multiply(x, y&: Ax);
86 r -= Ax;
87}
88
89/*****************************************************************************/
90void GaussianFactorGraphSystem::multiply(const Vector &x, Vector& AtAx) const {
91 /* implement A^T*(A*x), assume x and AtAx are pre-allocated */
92
93 // Build a VectorValues for Vector x
94 VectorValues vvX = buildVectorValues(v: x, keyInfo: keyInfo_);
95
96 // VectorValues form of A'Ax for multiplyHessianAdd
97 VectorValues vvAtAx = keyInfo_.x0(); // crucial for performance
98
99 // vvAtAx += 1.0 * A'Ax for each factor
100 gfg_.multiplyHessianAdd(alpha: 1.0, x: vvX, y&: vvAtAx);
101
102 // Make the result as Vector form
103 AtAx = vvAtAx.vector(keys: keyInfo_.ordering());
104}
105
106/*****************************************************************************/
107void GaussianFactorGraphSystem::getb(Vector &b) const {
108 /* compute rhs, assume b pre-allocated */
109
110 // Get whitened r.h.s (A^T * b) from each factor in the form of VectorValues
111 VectorValues vvb = gfg_.gradientAtZero();
112
113 // Make the result as Vector form
114 b = -vvb.vector(keys: keyInfo_.ordering());
115}

117/**********************************************************************************/
118void GaussianFactorGraphSystem::leftPrecondition(const Vector &x,
119 Vector &y) const {
120 // For a preconditioner M = L*L^T
121 // Calculate y = L^{-1} x
122 preconditioner_.solve(y: x, x&: y);
123}

125/**********************************************************************************/
126void GaussianFactorGraphSystem::rightPrecondition(const Vector &x,
127 Vector &y) const {
128 // For a preconditioner M = L*L^T
129 // Calculate y = L^{-T} x
130 preconditioner_.transposeSolve(y: x, x&: y);
131}

144/**********************************************************************************/
145VectorValues buildVectorValues(const Vector &v, const Ordering &ordering,
146 const map<Key, size_t> & dimensions) {
147 VectorValues result;
148
149 DenseIndex offset = 0;
150 for (size_t i = 0; i < ordering.size(); ++i) {
151 const Key key = ordering[i];
152 map<Key, size_t>::const_iterator it = dimensions.find(x: key);
153 if (it == dimensions.end()) {
154 throw invalid_argument(
155 "buildVectorValues: inconsistent ordering and dimensions");
156 }
157 const size_t dim = it->second;
158 result.emplace(j: key, args: v.segment(start: offset, n: dim));
159 offset += dim;
160 }
161
162 return result;
163}
164
165/**********************************************************************************/
166VectorValues buildVectorValues(const Vector &v, const KeyInfo &keyInfo) {
167 VectorValues result;
168 for ( const KeyInfo::value_type &item: keyInfo ) {
169 result.emplace(j: item.first, args: v.segment(start: item.second.start, n: item.second.dim));
170 }
171 return result;
172}
