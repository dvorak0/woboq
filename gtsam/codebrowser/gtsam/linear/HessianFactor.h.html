36 /**
37 * @brief A Gaussian factor using the canonical parameters (information form)
38 *
39 * HessianFactor implements a general quadratic factor of the form
40 * \f[ E(x) = 0.5 x^T G x - x^T g + 0.5 f \f]
41 * that stores the matrix \f$ G \f$, the vector \f$ g \f$, and the constant term \f$ f \f$.
42 *
43 * When \f$ G \f$ is positive semidefinite, this factor represents a Gaussian,
44 * in which case \f$ G \f$ is the information matrix \f$ \Lambda \f$,
45 * \f$ g \f$ is the information vector \f$ \eta \f$, and \f$ f \f$ is the residual
46 * sum-square-error at the mean, when \f$ x = \mu \f$.
47 *
48 * Indeed, the negative log-likelihood of a Gaussian is (up to a constant)
49 * @f$ E(x) = 0.5(x-\mu)^T P^{-1} (x-\mu) @f$
50 * with @f$ \mu @f$ the mean and @f$ P @f$ the covariance matrix. Expanding the product we get
51 * @f[
52 * E(x) = 0.5 x^T P^{-1} x - x^T P^{-1} \mu + 0.5 \mu^T P^{-1} \mu
53 * @f]
54 * We define the Information matrix (or Hessian) @f$ \Lambda = P^{-1} @f$
55 * and the information vector @f$ \eta = P^{-1} \mu = \Lambda \mu @f$
56 * to arrive at the canonical form of the Gaussian:
57 * @f[
58 * E(x) = 0.5 x^T \Lambda x - x^T \eta + 0.5 \mu^T \Lambda \mu
59 * @f]
60 *
61 * This factor is one of the factors that can be in a GaussianFactorGraph.
62 * It may be returned from NonlinearFactor::linearize(), but is also
63 * used internally to store the Hessian during Cholesky elimination.
64 *
65 * This can represent a quadratic factor with characteristics that cannot be
66 * represented using a JacobianFactor (which has the form
67 * \f$ E(x) = \Vert Ax - b \Vert^2 \f$ and stores the Jacobian \f$ A \f$
68 * and error vector \f$ b \f$, i.e. is a sum-of-squares factor). For example,
69 * a HessianFactor need not be positive semidefinite, it can be indefinite or
70 * even negative semidefinite.
71 *
72 * If a HessianFactor is indefinite or negative semi-definite, then in order
73 * for solving the linear system to be possible,
74 * the Hessian of the full system must be positive definite (i.e. when all
75 * small Hessians are combined, the result must be positive definite). If
76 * this is not the case, an error will occur during elimination.
77 *
78 * This class stores G, g, and f as an augmented matrix HessianFactor::matrix_.
79 * The upper-left n x n blocks of HessianFactor::matrix_ store the upper-right
80 * triangle of G, the upper-right-most column of length n of HessianFactor::matrix_
81 * stores g, and the lower-right entry of HessianFactor::matrix_ stores f, i.e.
82 * \code
83 HessianFactor::matrix_ = [ G11 G12 G13 ... g1
84 0 G22 G23 ... g2
85 0 0 G33 ... g3
86 : : : :
87 0 0 0 ... f ]
88 \endcode
89 Blocks can be accessed as follows:
90 \code
91 G11 = info(begin(), begin());
92 G12 = info(begin(), begin()+1);
93 G23 = info(begin()+1, begin()+2);
94 g2 = linearTerm(begin()+1);
95 f = constantTerm();
96 .......
97 \endcode
98 */

116 /** Construct a unary factor. G is the quadratic term (Hessian matrix), g
117 * the linear term (a vector), and f the constant term. The quadratic
118 * error is:
119 * 0.5*(f - 2*x'*g + x'*G*x)
120 */
121 HessianFactor(Key j, const Matrix& G, const Vector& g, double f);
122
123 /** Construct a unary factor, given a mean and covariance matrix.
124 * error is 0.5*(x-mu)'*inv(Sigma)*(x-mu)
125 */
126 HessianFactor(Key j, const Vector& mu, const Matrix& Sigma);
127
128 /** Construct a binary factor. Gxx are the upper-triangle blocks of the
129 * quadratic term (the Hessian matrix), gx the pieces of the linear vector
130 * term, and f the constant term.
131 * JacobianFactor error is \f[ 0.5* (Ax-b)' M (Ax-b) = 0.5*x'A'MAx - x'A'Mb + 0.5*b'Mb \f]
132 * HessianFactor error is \f[ 0.5*(x'Gx - 2x'g + f) = 0.5*x'Gx - x'*g + 0.5*f \f]
133 * So, with \f$ A = [A1 A2] \f$ and \f$ G=A'*M*A = [A1';A2']*M*[A1 A2] \f$ we have
134 \code
135 n1*n1 G11 = A1'*M*A1
136 n1*n2 G12 = A1'*M*A2
137 n2*n2 G22 = A2'*M*A2
138 n1*1 g1 = A1'*M*b
139 n2*1 g2 = A2'*M*b
140 1*1 f = b'*M*b
141 \endcode
142 */
143 HessianFactor(Key j1, Key j2,
144 const Matrix& G11, const Matrix& G12, const Vector& g1,
145 const Matrix& G22, const Vector& g2, double f);
146
147 /** Construct a ternary factor. Gxx are the upper-triangle blocks of the
148 * quadratic term (the Hessian matrix), gx the pieces of the linear vector
149 * term, and f the constant term.
150 */
151 HessianFactor(Key j1, Key j2, Key j3,
152 const Matrix& G11, const Matrix& G12, const Matrix& G13, const Vector& g1,
153 const Matrix& G22, const Matrix& G23, const Vector& g2,
154 const Matrix& G33, const Vector& g3, double f);

225 /** Return the constant term \f$ f \f$ as described above
226 * @return The constant term \f$ f \f$
227 */
228 double constantTerm() const {
229 const auto view = info_.diagonalBlock(J: size());
230 return view(0, 0);
231 }

233 /** Return the constant term \f$ f \f$ as described above
234 * @return The constant term \f$ f \f$

236 double& constantTerm() { return info_.diagonalBlock(J: size())(0, 0); }

238 /** Return the part of linear term \f$ g \f$ as described above corresponding to the requested variable.
239 * @param j Which block row to get, as an iterator pointing to the slot in this factor. You can
240 * use, for example, begin() + 2 to get the 3rd variable in this factor.
241 * @return The linear term \f$ g \f$ */
242 SymmetricBlockMatrix::constBlock linearTerm(const_iterator j) const {
243#ifndef NDEBUG
244 if(empty()) throw;
245#endif
246 return info_.aboveDiagonalBlock(I: j - begin(), J: size());
247 }
248
249 /** Return the complete linear term \f$ g \f$ as described above.
250 * @return The linear term \f$ g \f$ */
251 SymmetricBlockMatrix::constBlock linearTerm() const {
252#ifndef NDEBUG
253 if(empty()) throw;
254#endif
255 // get the last column (except the bottom right block)
256 return info_.aboveDiagonalRange(i_startBlock: 0, i_endBlock: size(), j_startBlock: size(), j_endBlock: size() + 1);
257 }

264#endif

275 /** Return the augmented information matrix represented by this GaussianFactor.
276 * The augmented information matrix contains the information matrix with an
277 * additional column holding the information vector, and an additional row
278 * holding the transpose of the information vector. The lower-right entry
279 * contains the constant error term (when \f$ \delta x = 0 \f$). The
280 * augmented information matrix is described in more detail in HessianFactor,
281 * which in fact stores an augmented information matrix.
282 *
283 * For HessianFactor, this is the same as info() except that this function
284 * returns a complete symmetric matrix whereas info() returns a matrix where
285 * only the upper triangle is valid, but should be interpreted as symmetric.
286 * This is because info() returns only a reference to the internal
287 * representation of the augmented information matrix, which stores only the
288 * upper triangle.
289 */
290 Matrix augmentedInformation() const override;

384/**
385* Densely partially eliminate with Cholesky factorization. JacobianFactors are
386* left-multiplied with their transpose to form the Hessian using the conversion constructor
387* HessianFactor(const JacobianFactor&).
388*
389* If any factors contain constrained noise models, this function will fail because our current
390* implementation cannot handle constrained noise models in Cholesky factorization. The
391* function EliminatePreferCholesky() automatically does QR instead when this is the case.
392*
393* Variables are eliminated in the order specified in \c keys.
394*
395* @param factors Factors to combine and eliminate
396* @param keys The variables to eliminate and their elimination ordering
397* @return The conditional and remaining factor
398*
399* \ingroup LinearSolving */
400GTSAM_EXPORT std::pair<std::shared_ptr<GaussianConditional>, std::shared_ptr<HessianFactor> >
401 EliminateCholesky(const GaussianFactorGraph& factors, const Ordering& keys);
402
403/**
404* Densely partially eliminate with Cholesky factorization. JacobianFactors are
405* left-multiplied with their transpose to form the Hessian using the conversion constructor
406* HessianFactor(const JacobianFactor&).
407*
408* This function will fall back on QR factorization for any cliques containing JacobianFactor's
409* with constrained noise models.
410*
411* Variables are eliminated in the order specified in \c keys.
412*
413* @param factors Factors to combine and eliminate
414* @param keys The variables to eliminate and their elimination ordering
415* @return The conditional and remaining factor
416*
417* \ingroup LinearSolving */
418GTSAM_EXPORT std::pair<std::shared_ptr<GaussianConditional>, std::shared_ptr<GaussianFactor> >
419 EliminatePreferCholesky(const GaussianFactorGraph& factors, const Ordering& keys);
