19#pragma once
20
21#include <gtsam/linear/NoiseModel.h>
22
23#include <random>
24
25namespace gtsam {
26
27/**
28 * Sampling structure that keeps internal random number generators for
29 * diagonal distributions specified by NoiseModel
30 */
31class GTSAM_EXPORT Sampler {
32 protected:
33 /** noiseModel created at generation */
34 noiseModel::Diagonal::shared_ptr model_;
35
36 /** generator */
37 mutable std::mt19937_64 generator_;
38
39 public:
40 typedef std::shared_ptr<Sampler> shared_ptr;
41
42 /// @name constructors
43 /// @{
44
45 /**
46 * Create a sampler for the distribution specified by a diagonal NoiseModel
47 * with a manually specified seed
48 *
49 * NOTE: do not use zero as a seed, it will break the generator
50 */
51 explicit Sampler(const noiseModel::Diagonal::shared_ptr& model,
52 uint_fast64_t seed = 42u);
53
54 /**
55 * Create a sampler for a distribution specified by a vector of sigmas
56 * directly
57 *
58 * NOTE: do not use zero as a seed, it will break the generator
59 */
60 explicit Sampler(const Vector& sigmas, uint_fast64_t seed = 42u);
61
62 /// @}
63 /// @name access functions
64 /// @{
65
66 size_t dim() const { return model_->dim(); }
67
68 Vector sigmas() const { return model_->sigmas(); }
69
70 const noiseModel::Diagonal::shared_ptr& model() const { return model_; }
71
72 /// @}
73 /// @name basic functionality
74 /// @{
75
76 /// sample from distribution
77 Vector sample() const;
78
79 /// sample with given random number generator
80 static Vector sampleDiagonal(const Vector& sigmas, std::mt19937_64* rng);
81 /// @}
82
83 protected:
84 /** given sigmas for a diagonal model, returns a sample */
85 Vector sampleDiagonal(const Vector& sigmas) const;
86};
87
88} // namespace gtsam
