12 /**
13 * @file RegularHessianFactor.h
14 * @brief Specialized HessianFactor class for regular problems (fixed-size blocks).
15 * @details This factor is specifically designed for quadratic cost functions
16 * where all variables involved have the same, fixed dimension `D`.
17 * It stores the Hessian and gradient terms and provides optimized
18 * methods for operations like Hessian-vector products, crucial for
19 * iterative solvers like Conjugate Gradient. It ensures that all
20 * involved blocks have the expected dimension `D`.
21 * @author Sungtae An
22 * @author Frank Dellaert
23 * @date March 2014
24 */
25

36 /**
37 * @brief A HessianFactor where all variables have the same dimension D.
38 * @details Represents a quadratic factor \f$ E(x) = 0.5 x^T G x - g^T x + 0.5 f \f$,
39 * corresponding to a Gaussian probability density \f$ P(x) \propto \exp(-E(x)) \f$.
40 * Here, G is the Hessian (or information matrix), g is the gradient vector,
41 * and f is a constant term (negative log-likelihood at x=0).
42 *
43 * This class is templated on the dimension `D` and enforces that all variables
44 * associated with this factor have this dimension during construction.
45 * It inherits from HessianFactor but provides specialized, efficient implementations
46 * for operations like `multiplyHessianAdd` using raw memory access, assuming
47 * a regular block structure. This can significantly improve performance in
48 * iterative optimization algorithms when dealing with many variables of the
49 * same type and dimension (e.g., Pose3 variables in SLAM).
50 *
51 * It can be constructed directly from Hessian blocks, from a
52 * RegularJacobianFactor, or by linearizing a NonlinearFactorGraph.
53 *
54 * @tparam D The dimension of each variable block involved in the factor.
55 */
56 template<size_t D>
57 class RegularHessianFactor : public HessianFactor {
58
59 public:
60
61 typedef Eigen::Matrix<double, D, 1> VectorD;
62 typedef Eigen::Matrix<double, D, D> MatrixD;
63
64 /**
65 * @brief Construct an n-way factor from supplied components.
66 * @details Represents the energy function \f$ E(x) = 0.5 \sum_{i,j} x_i^T G_{ij} x_j - \sum_i g_i^T x_i + 0.5 f \f$.
67 * Note that the keys in `js` must correspond order-wise to the blocks in `Gs` and `gs`.
68 * @param js Vector of keys involved in the factor.
69 * @param Gs Vector of upper-triangular Hessian blocks (row-major order, e.g., G11, G12, G13, G22, G23, G33 for a ternary factor).
70 * @param gs Vector of gradient vector blocks (-J^T b).
71 * @param f Constant term \f$ 0.5*f = 0.5*\|b\|^2 \f$.
72 */
73 RegularHessianFactor(const KeyVector& js,
74 const std::vector<Matrix>& Gs, const std::vector<Vector>& gs, double f) :
75 HessianFactor(js, Gs, gs, f) {
76 checkInvariants();
77 }
78
79 /** Construct a binary factor. Gxx are the upper-triangle blocks of the
80 * quadratic term (the Hessian matrix), gx the pieces of the linear vector
81 * term, and f the constant term.
82 */
83 RegularHessianFactor(Key j1, Key j2, const MatrixD& G11, const MatrixD& G12,
84 const VectorD& g1, const MatrixD& G22, const VectorD& g2, double f) :
85 HessianFactor(j1, j2, G11, G12, g1, G22, g2, f) {
86 }
87
88 /** Construct a ternary factor. Gxx are the upper-triangle blocks of the
89 * quadratic term (the Hessian matrix), gx the pieces of the linear vector
90 * term, and f the constant term.
91 */
92 RegularHessianFactor(Key j1, Key j2, Key j3,
93 const MatrixD& G11, const MatrixD& G12, const MatrixD& G13, const VectorD& g1,
94 const MatrixD& G22, const MatrixD& G23, const VectorD& g2,
95 const MatrixD& G33, const VectorD& g3, double f) :
96 HessianFactor(j1, j2, j3, G11, G12, G13, g1, G22, G23, g2, G33, g3, f) {
97 }
98
99 /**
100 * @brief Constructor with an arbitrary number of keys and the augmented information matrix
101 * specified as a block matrix.
102 * @details The augmented information matrix contains the Hessian blocks G and the
103 * gradient blocks g, arranged as \f$ [ G, -g ; -g^T, f ] \f$.
104 * @param keys Container of keys specifying the variables involved and their order.
105 * @param augmentedInformation The augmented information matrix [H -g; -g' f], laid out as a SymmetricBlockMatrix.
106 */
107 template<typename KEYS>
108 RegularHessianFactor(const KEYS& keys,
109 const SymmetricBlockMatrix& augmentedInformation) :
110 HessianFactor(keys, augmentedInformation) {
111 checkInvariants();
112 }
113
114 /**
115 * @brief Construct a RegularHessianFactor from a RegularJacobianFactor.
116 * @details Computes \f$ G = J^T J \f$, \f$ g = J^T b \f$, and \f$ f = b^T b \f$.
117 * Requires that the JacobianFactor also has regular block sizes matching `D`.
118 * @param jf The RegularJacobianFactor to convert.
119 */
120 RegularHessianFactor(const RegularJacobianFactor<D>& jf)
121 : HessianFactor(jf) {
122 }
123
124 /**
125 * @brief Construct from a GaussianFactorGraph combined using a Scatter.
126 * @details This is typically used internally by optimizers. It sums the Hessian
127 * contributions from multiple factors in the graph.
128 * @param factors The graph containing factors to combine.
129 * @param scatter A Scatter structure indicating the layout of variables.
130 */
131 RegularHessianFactor(const GaussianFactorGraph& factors,
132 const Scatter& scatter)
133 : HessianFactor(factors, scatter) {
134 checkInvariants();
135 }
136
137 /**
138 * @brief Construct from a GaussianFactorGraph.
139 * @details This is typically used internally by optimizers. It sums the Hessian
140 * contributions from multiple factors in the graph. Assumes keys are ordered 0..n-1.
141 * @param factors The graph containing factors to combine.
142 */
143 RegularHessianFactor(const GaussianFactorGraph& factors)
144 : HessianFactor(factors) {
145 checkInvariants();
146 }
147
148 private:
149
150 /**
151 * @brief Check if the factor has regular block structure.
152 * @details Verifies that the dimensions of the stored augmented information matrix
153 * `info_` correspond to the expected size based on the number of keys and the
154 * fixed block dimension `D`. Throws `std::invalid_argument` if the structure
155 * is not regular. This is called internally by constructors.
156 */
157 void checkInvariants() const {
158 if (info_.cols() != 0 && // Allow zero-key factors (e.g. priors on anchor nodes)
159 info_.cols() != 1 + (info_.nBlocks() - 1) * static_cast<DenseIndex>(D))
160 throw std::invalid_argument(
161 "RegularHessianFactor constructor was given non-regular factors or "
162 "incorrect template dimension D");
163 }
164
165 // Use Eigen magic to access raw memory for efficiency in Hessian products
166 typedef Eigen::Map<VectorD> DMap;
167 typedef Eigen::Map<const VectorD> ConstDMap;
168
169 // Scratch space for multiplyHessianAdd to avoid re-allocation
170 // According to link below this is thread-safe.
171 // http://stackoverflow.com/questions/11160964/multiple-copies-of-the-same-object-c-thread-safe
172 mutable std::vector<VectorD, Eigen::aligned_allocator<VectorD>> y_;
173
174 public:
175
176 /**
177 * @brief Multiply the Hessian part of the factor times a VectorValues `x` and add the result to `y`.
178 * @details Computes `y += alpha * H * x`.
179 * @param alpha Scalar multiplier.
180 * @param x VectorValues containing the vector to multiply with.
181 * @param[in,out] y VectorValues to store the result (accumulates).
182 */
183 void multiplyHessianAdd(double alpha, const VectorValues& x,
184 VectorValues& y) const override {
185 // Note: This implementation just calls the base class. The raw memory versions
186 // below are specifically optimized for the regular structure of this class.
187 // Consider using those directly or ensuring the base class implementation
188 // is efficient enough for your use case if calling this version.
189 HessianFactor::multiplyHessianAdd(alpha, x, y);
190 }
191
192 /**
193 * @brief Multiply the Hessian part of the factor times a raw vector `x` and add the result to `y`. (Raw memory version)
194 * @details Computes `y += alpha * H * x`, optimized for regular block structure.
195 * Assumes `x` and `yvalues` are pre-allocated, contiguous memory blocks
196 * where variable `j` corresponds to memory chunk `x + keys_[j] * D`
197 * and `yvalues + keys_[j] * D`.
198 * @param alpha Scalar multiplier.
199 * @param x Raw pointer to the input vector data.
200 * @param[in,out] yvalues Raw pointer to the output vector data (accumulates).
201 */
202 void multiplyHessianAdd(double alpha, const double* x,
203 double* yvalues) const {
204 // Ensure scratch space is properly sized
205 const size_t n = size();
206 if (y_.size() != n) {
207 y_.resize(n);
208 }
209 for (VectorD& yi : y_)
210 yi.setZero();
211
212 // Loop over columns (j) of the Hessian H=info_
213 // Accessing x only once per column j
214 // Fill temporary y_ vector corresponding to rows i
215 for (DenseIndex j = 0; j < static_cast<DenseIndex>(n); ++j) {
216 Key key_j = keys_[j];
217 ConstDMap xj(x + key_j * D); // Map to the j-th block of x
218
219 DenseIndex i = 0;
220 // Off-diagonal blocks G_ij * x_j for i < j
221 for (; i < j; ++i) {
222 y_[i] += info_.aboveDiagonalBlock(I: i, J: j) * xj;
223 }
224 // Diagonal block G_jj * x_j
225 y_[i] += info_.diagonalBlock(j) * xj;
226 // Off-diagonal blocks G_ij * x_j for i > j (using transpose G_ji^T * x_j)
227 for (i = j + 1; i < static_cast<DenseIndex>(n); ++i) {
228 y_[i] += info_.aboveDiagonalBlock(I: j, J: i).transpose() * xj;
229 }
230 }
231
232 // Add accumulated results from y_ to the output yvalues
233 for (DenseIndex i = 0; i < static_cast<DenseIndex>(n); ++i) {
234 Key key_i = keys_[i];
235 DMap map_yi(yvalues + key_i * D); // Map to the i-th block of yvalues
236 map_yi += alpha * y_[i];
237 }
238 }
239
240 /**
241 * @brief Multiply the Hessian part of the factor times a raw vector `x` and add the result to `y`.
242 * @details Computes `y += alpha * H * x`, optimized for regular block structure,
243 * but handles potentially non-contiguous or scattered memory layouts
244 * for `x` and `yvalues` as defined by the `offsets` vector.
245 * `offsets[k]` should give the starting index of variable `k` in the
246 * raw memory arrays. `offsets[k+1] - offsets[k]` should equal `D`.
247 * @param alpha Scalar multiplier.
248 * @param x Raw pointer to the input vector data.
249 * @param[in,out] yvalues Raw pointer to the output vector data (accumulates).
250 * @param offsets Vector mapping variable keys to their starting index in `x` and `yvalues`.
251 */
252 void multiplyHessianAdd(double alpha, const double* x, double* yvalues,
253 const std::vector<size_t>& offsets) const {
254 // Ensure scratch space is properly sized
255 const size_t n = size();
256 if (y_.size() != n) {
257 y_.resize(n);
258 }
259 for (VectorD& yi : y_)
260 yi.setZero();
261
262 // Loop over columns (j) of the Hessian H=info_
263 for (DenseIndex j = 0; j < static_cast<DenseIndex>(n); ++j) {
264 Key key_j = keys_[j];
265 size_t offset_j = offsets[key_j];
266 // Ensure block size matches D (redundant if checkInvariants worked, but safe)
267 size_t dim_j = offsets[key_j + 1] - offset_j;
268 if (dim_j != D) throw std::runtime_error("RegularHessianFactor::multiplyHessianAdd: Mismatched dimension in offset map.");
269 ConstDMap xj(x + offset_j); // Map to the j-th block of x using offset
270
271 DenseIndex i = 0;
272 // Off-diagonal blocks G_ij * x_j for i < j
273 for (; i < j; ++i) {
274 y_[i] += info_.aboveDiagonalBlock(I: i, J: j) * xj;
275 }
276 // Diagonal block G_jj * x_j
277 y_[i] += info_.diagonalBlock(j) * xj;
278 // Off-diagonal blocks G_ij * x_j for i > j (using transpose G_ji^T * x_j)
279 for (i = j + 1; i < static_cast<DenseIndex>(n); ++i) {
280 y_[i] += info_.aboveDiagonalBlock(I: j, J: i).transpose() * xj;
281 }
282 }
283
284 // Add accumulated results from y_ to the output yvalues using offsets
285 for (DenseIndex i = 0; i < static_cast<DenseIndex>(n); ++i) {
286 Key key_i = keys_[i];
287 size_t offset_i = offsets[key_i];
288 size_t dim_i = offsets[key_i + 1] - offset_i;
289 if (dim_i != D) throw std::runtime_error("RegularHessianFactor::multiplyHessianAdd: Mismatched dimension in offset map.");
290 DMap map_yi(yvalues + offset_i); // Map to the i-th block of yvalues using offset
291 map_yi += alpha * y_[i];
292 }
293 }
294
295 /**
296 * @brief Return the diagonal of the Hessian for this factor (Raw memory version).
297 * @details Adds the diagonal elements of the Hessian matrix \f$ H = G \f$ associated
298 * with this factor to the pre-allocated raw memory block `d`.
299 * Assumes `d` has the standard contiguous layout `d + keys_[i] * D`.
300 * @param[in,out] d Raw pointer to the memory block where Hessian diagonals should be added.
301 */
302 void hessianDiagonal(double* d) const override {
303 // Loop over all variables (diagonal blocks) in the factor
304 const size_t n = size();
305 for (DenseIndex pos = 0; pos < static_cast<DenseIndex>(n); ++pos) {
306 Key j = keys_[pos];
307 // Get the diagonal block G_jj and add its diagonal elements to d
308 DMap(d + D * j) += info_.diagonal(J: pos);
309 }
310 }
311
312 /**
313 * @brief Add the gradient vector \f$ -g \f$ (gradient at zero) to a raw memory block `d`.
314 * @details Adds the gradient vector \f$ -g \f$ (where the factor represents
315 * \f$ 0.5 x^T G x - g^T x + 0.5 f \f$) to the pre-allocated raw
316 * memory block `d`. Assumes `d` has the standard contiguous layout
317 * `d + keys_[i] * D`.
318 * @param[in,out] d Raw pointer to the memory block where the gradient should be added.
319 */
320 void gradientAtZero(double* d) const override {
321 // The linear term g is stored as the last block column of info_ (negated).
322 // We add (-g) to d.
323 const size_t n = size();
324 for (DenseIndex pos = 0; pos < static_cast<DenseIndex>(n); ++pos) {
325 Key j = keys_[pos];
326 // info_.aboveDiagonalBlock(pos, n) accesses the block corresponding to g_j
327 DMap(d + D * j) += info_.aboveDiagonalBlock(I: pos, J: n);
328 }
329 }
