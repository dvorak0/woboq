35/**
36 * \brief Compute maximum Eigenpair with power method
37 *
38 * References :
39 * 1) G. Golub and C. V. Loan, Matrix Computations, 3rd ed. Baltimore, Johns
40 * Hopkins University Press, 1996, pp.405-411
41 * 2) Rosen, D. and Carlone, L., 2017, September. Computational
42 * enhancements for certifiably correct SLAM. In Proceedings of the
43 * International Conference on Intelligent Robots and Systems.
44 * 3) Yulun Tian and Kasra Khosoussi and David M. Rosen and Jonathan P. How,
45 * 2020, Aug, Distributed Certifiably Correct Pose-Graph Optimization, Arxiv
46 * 4) C. de Sa, B. He, I. Mitliagkas, C. Ré, and P. Xu, “Accelerated
47 * stochastic power iteration,” in Proc. Mach. Learn. Res., no. 84, 2018, pp.
48 * 58–67
49 *
50 * It performs the following iteration: \f$ x_{k+1} = A * x_k \f$
51 * where A is the aim matrix we want to get eigenpair of, x is the
52 * Ritz vector
53 *
54 * Template argument Operator just needs multiplication operator
55 *
56 */
57template <class Operator>
58class PowerMethod {
59 protected:
60 /**
61 * Const reference to an externally-held matrix whose minimum-eigenvalue we
62 * want to compute
63 */
64 const Operator &A_;
65
66 const int dim_; // dimension of Matrix A
67
68 size_t nrIterations_; // number of iterations
69
70 double ritzValue_; // Ritz eigenvalue
71 Vector ritzVector_; // Ritz eigenvector
72
73 public:
74 /// @name Standard Constructors
75 /// @{
76
77 /// Construct from the aim matrix and intial ritz vector
78 explicit PowerMethod(const Operator &A,
79 const std::optional<Vector> initial = {})
80 : A_(A), dim_(A.rows()), nrIterations_(0) {
81 Vector x0;
82 x0 = initial ? *initial : Vector::Random(size: dim_);
83 x0.normalize();
84
85 // initialize Ritz eigen value
86 ritzValue_ = 0.0;
87
88 // initialize Ritz eigen vector
89 ritzVector_ = powerIteration(x0);
90 }
91
92 /**
93 * Run power iteration to get ritzVector with previous ritzVector x, and
94 * return A * x / || A * x ||
95 */
96 Vector powerIteration(const Vector &x) const {
97 Vector y = A_ * x;
98 y.normalize();
99 return y;
100 }
101
102 /**
103 * Run power iteration to get ritzVector with previous ritzVector x, and
104 * return A * x / || A * x ||
105 */
106 Vector powerIteration() const { return powerIteration(ritzVector_); }
107
108 /**
109 * After Perform power iteration on a single Ritz value, check if the Ritz
110 * residual for the current Ritz pair is less than the required convergence
111 * tol, return true if yes, else false
112 */
113 bool converged(double tol) const {
114 const Vector x = ritzVector_;
115 // store the Ritz eigen value
116 const double ritzValue = x.dot(A_ * x);
117 const double error = (A_ * x - ritzValue * x).norm();
118 return error < tol;
119 }
120
121 /// Return the number of iterations
122 size_t nrIterations() const { return nrIterations_; }
123
124 /**
125 * Start the power/accelerated iteration, after performing the
126 * power/accelerated iteration, calculate the ritz error, repeat this
127 * operation until the ritz error converge. If converged return true, else
128 * false.
129 */
130 bool compute(size_t maxIterations, double tol) {
131 // Starting
132 bool isConverged = false;
133
134 for (size_t i = 0; i < maxIterations && !isConverged; i++) {
135 ++nrIterations_;
136 // update the ritzVector after power iteration
137 ritzVector_ = powerIteration();
138 // update the ritzValue
139 ritzValue_ = ritzVector_.dot(A_ * ritzVector_);
140 isConverged = converged(tol);
141 }
142
143 return isConverged;
144 }
145
146 /// Return the eigenvalue
147 double eigenvalue() const { return ritzValue_; }
148
149 /// Return the eigenvector
150 Vector eigenvector() const { return ritzVector_; }
151};
