39/**
40 * The mEstimator name space contains all robust error functions.
41 * It mirrors the exposition at
42 * https://members.loria.fr/MOBerger/Enseignement/Master2/Documents/ZhangIVC-97-01.pdf
43 * which talks about minimizing \sum \rho(r_i), where \rho is a loss function of choice.
44 *
45 * To illustrate, let's consider the least-squares (L2), L1, and Huber estimators as examples:
46 *
47 * Name Symbol Least-Squares L1-norm Huber
48 * Loss \rho(x) 0.5*x^2 |x| 0.5*x^2 if |x|<k, 0.5*k^2 + k|x-k| otherwise
49 * Derivative \phi(x) x sgn(x) x if |x|<k, k sgn(x) otherwise
50 * Weight w(x)=\phi(x)/x 1 1/|x| 1 if |x|<k, k/|x| otherwise
51 *
52 * With these definitions, D(\rho(x), p) = \phi(x) D(x,p) = w(x) x D(x,p) = w(x) D(L2(x), p),
53 * and hence we can solve the equivalent weighted least squares problem \sum w(r_i) \rho(r_i)
54 *
55 * Each M-estimator in the mEstimator name space simply implements the above functions.
56 */
57// clang-format on

60/**
61 * Pure virtual class for all robust error function classes.
62 *
63 * It provides the machinery for block vs scalar reweighting strategies, in
64 * addition to defining the interface of derived classes.
65 */
66class GTSAM_EXPORT Base {
67 public:
68 /** the rows can be weighted independently according to the error
69 * or uniformly with the norm of the right hand side */
70 enum ReweightScheme { Scalar, Block };
71 typedef std::shared_ptr<Base> shared_ptr;
72
73 protected:
74 /// Strategy for reweighting \sa ReweightScheme
75 ReweightScheme reweight_;
76
77 public:
78 Base(const ReweightScheme reweight = Block) : reweight_(reweight) {}
79 virtual ~Base() {}
80
81 /// Returns the reweight scheme, as explained in ReweightScheme
82 ReweightScheme reweightScheme() const { return reweight_; }
83
84 /**
85 * This method is responsible for returning the total penalty for a given
86 * amount of error. For example, this method is responsible for implementing
87 * the quadratic function for an L2 penalty, the absolute value function for
88 * an L1 penalty, etc.
89 *
90 * TODO(mikebosse): When the loss function has as input the norm of the
91 * error vector, then it prevents implementations of asymmeric loss
92 * functions. It would be better for this function to accept the vector and
93 * internally call the norm if necessary.
94 *
95 * This returns \f$\rho(x)\f$ in \ref mEstimator
96 */
97 virtual double loss(double distance) const { return 0; }
98
99 /**
100 * This method is responsible for returning the weight function for a given
101 * amount of error. The weight function is related to the analytic derivative
102 * of the loss function. See
103 * https://members.loria.fr/MOBerger/Enseignement/Master2/Documents/ZhangIVC-97-01.pdf
104 * for details. This method is required when optimizing cost functions with
105 * robust penalties using iteratively re-weighted least squares.
106 *
107 * This returns w(x) in \ref mEstimator
108 */
109 virtual double weight(double distance) const = 0;
110
111 virtual void print(const std::string &s) const = 0;
112 virtual bool equals(const Base &expected, double tol = 1e-8) const = 0;
113
114 double sqrtWeight(double distance) const { return std::sqrt(x: weight(distance)); }
115
116 /** produce a weight vector according to an error vector and the implemented
117 * robust function */
118 Vector weight(const Vector &error) const;
119
120 /** square root version of the weight function */
121 Vector sqrtWeight(const Vector &error) const;
122
123 /** reweight block matrices and a vector according to their weight
124 * implementation */
125 void reweight(Vector &error) const;
126 void reweight(std::vector<Matrix> &A, Vector &error) const;
127 void reweight(Matrix &A, Vector &error) const;
128 void reweight(Matrix &A1, Matrix &A2, Vector &error) const;
129 void reweight(Matrix &A1, Matrix &A2, Matrix &A3, Vector &error) const;
130
131 private:
132#if GTSAM_ENABLE_BOOST_SERIALIZATION
133 /** Serialization function */
134 friend class boost::serialization::access;
135 template <class ARCHIVE>
136 void serialize(ARCHIVE &ar, const unsigned int /*version*/) {
137 ar &BOOST_SERIALIZATION_NVP(reweight_);
138 }
139#endif
140};

142/** "Null" robust loss function, equivalent to a Gaussian pdf noise model, or
143 * plain least-squares (non-robust).
144 *
145 * This model has no additional parameters.
146 *
147 * - Loss \f$ \rho(x) = 0.5 x² \f$
148 * - Derivative \f$ \phi(x) = x \f$
149 * - Weight \f$ w(x) = \phi(x)/x = 1 \f$
150 */
151class GTSAM_EXPORT Null : public Base {
152 public:
153 typedef std::shared_ptr<Null> shared_ptr;
154
155 Null(const ReweightScheme reweight = Block) : Base(reweight) {}
156 ~Null() override {}
157 double weight(double /*error*/) const override { return 1.0; }
158 double loss(double distance) const override { return 0.5 * distance * distance; }
159 void print(const std::string &s) const override;
160 bool equals(const Base & /*expected*/, double /*tol*/) const override { return true; }
161 static shared_ptr Create();
162
163 private:
164#if GTSAM_ENABLE_BOOST_SERIALIZATION
165 /** Serialization function */
166 friend class boost::serialization::access;
167 template <class ARCHIVE>
168 void serialize(ARCHIVE &ar, const unsigned int /*version*/) {
169 ar &BOOST_SERIALIZATION_BASE_OBJECT_NVP(Base);
170 }
171#endif
172};

394/** DCS implements the Dynamic Covariance Scaling robust error model
395 * from the paper Robust Map Optimization (Agarwal13icra).
396 *
397 * Under the special condition of the parameter c == 1.0 and not
398 * forcing the output weight s <= 1.0, DCS is similar to Geman-McClure.
399 *
400 * This model has a scalar parameter "c" (with "units" of squared error).
401 *
402 * - Loss \rho(x) = (c²x² + cx⁴)/(x²+c)² (for any "x")
403 * - Derivative \phi(x) = 2c²x/(x²+c)²
404 * - Weight w(x) = \phi(x)/x = 2c²/(x²+c)² if x²>c, 1 otherwise
405 */

434/** L2WithDeadZone implements a standard L2 penalty, but with a dead zone of
435 * width 2*k, centered at the origin. The resulting penalty within the dead
436 * zone is always zero, and grows quadratically outside the dead zone. In this
437 * sense, the L2WithDeadZone penalty is "robust to inliers", rather than being
438 * robust to outliers. This penalty can be used to create barrier functions in
439 * a general way.
440 *
441 * This model has a scalar parameter "k".
442 *
443 * - Loss \f$ \rho(x) = 0 \f$ if |x|<k, 0.5(k-|x|)² otherwise
444 * - Derivative \f$ \phi(x) = 0 \f$ if |x|<k, (-k+x) if x>k, (k+x) if x<-k
445 * - Weight \f$ w(x) = \phi(x)/x = 0 \f$ if |x|<k, (-k+x)/x if x>k, (k+x)/x if x<-k
446 */
447class GTSAM_EXPORT L2WithDeadZone : public Base {
448 protected:
449 double k_;
450
451 public:
452 typedef std::shared_ptr<L2WithDeadZone> shared_ptr;
453
454 L2WithDeadZone(double k = 1.0, const ReweightScheme reweight = Block);
455 double weight(double distance) const override;
456 double loss(double distance) const override;
457 void print(const std::string &s) const override;
458 bool equals(const Base &expected, double tol = 1e-8) const override;
459 static shared_ptr Create(double k, const ReweightScheme reweight = Block);
460 double modelParameter() const { return k_; }
461
462 private:
