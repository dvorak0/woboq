51 /**
52 * noiseModel::Base is the abstract base class for all noise models.
53 *
54 * Noise models must implement a 'whiten' function to normalize an error vector,
55 * and an 'unwhiten' function to unnormalize an error vector.
56 */
57 class GTSAM_EXPORT Base {
58
59 public:
60 typedef std::shared_ptr<Base> shared_ptr;
61
62 protected:
63
64 size_t dim_;
65
66 public:
67
68 /// primary constructor @param dim is the dimension of the model
69 Base(size_t dim = 1):dim_(dim) {}
70 virtual ~Base() {}
71
72 /// true if a constrained noise model, saves slow/clumsy dynamic casting
73 virtual bool isConstrained() const { return false; } // default false
74
75 /// true if a unit noise model, saves slow/clumsy dynamic casting
76 virtual bool isUnit() const { return false; } // default false
77
78 /// Dimensionality
79 inline size_t dim() const { return dim_;}
80
81 virtual void print(const std::string& name = "") const = 0;
82
83 virtual bool equals(const Base& expected, double tol=1e-9) const = 0;
84
85 /// Calculate standard deviations
86 virtual Vector sigmas() const;
87
88 /// Whiten an error vector.
89 virtual Vector whiten(const Vector& v) const = 0;
90
91 /// Whiten a matrix.
92 virtual Matrix Whiten(const Matrix& H) const = 0;
93
94 /// Unwhiten an error vector.
95 virtual Vector unwhiten(const Vector& v) const = 0;
96
97 /// Squared Mahalanobis distance v'*R'*R*v = <R*v,R*v>
98 virtual double squaredMahalanobisDistance(const Vector& v) const;
99
100 /// Mahalanobis distance
101 virtual double mahalanobisDistance(const Vector& v) const {
102 return std::sqrt(x: squaredMahalanobisDistance(v));
103 }
104
105 /// loss function, input is Mahalanobis distance
106 virtual double loss(const double squared_distance) const {
107 return 0.5 * squared_distance;
108 }
109
110 virtual void WhitenSystem(std::vector<Matrix>& A, Vector& b) const = 0;
111 virtual void WhitenSystem(Matrix& A, Vector& b) const = 0;
112 virtual void WhitenSystem(Matrix& A1, Matrix& A2, Vector& b) const = 0;
113 virtual void WhitenSystem(Matrix& A1, Matrix& A2, Matrix& A3, Vector& b) const = 0;
114
115 /** in-place whiten, override if can be done more efficiently */
116 virtual void whitenInPlace(Vector& v) const {
117 v = whiten(v);
118 }
119
120 /** in-place unwhiten, override if can be done more efficiently */
121 virtual void unwhitenInPlace(Vector& v) const {
122 v = unwhiten(v);
123 }
124
125 /** in-place whiten, override if can be done more efficiently */
126 virtual void whitenInPlace(Eigen::Block<Vector>& v) const {
127 v = whiten(v);
128 }
129
130 /** in-place unwhiten, override if can be done more efficiently */
131 virtual void unwhitenInPlace(Eigen::Block<Vector>& v) const {
132 v = unwhiten(v);
133 }
134
135 /** Useful function for robust noise models to get the unweighted but whitened error */
136 virtual Vector unweightedWhiten(const Vector& v) const {
137 return whiten(v);
138 }
139
140 /** get the weight from the effective loss function on residual vector v */
141 virtual double weight(const Vector& v) const { return 1.0; }
142
143 private:
144#if GTSAM_ENABLE_BOOST_SERIALIZATION
145 /** Serialization function */
146 friend class boost::serialization::access;
147 template<class ARCHIVE>
148 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
149 ar & BOOST_SERIALIZATION_NVP(dim_);
150 }
151#endif
152 };

156 /**
157 * Gaussian implements the mathematical model
158 * |R*x|^2 = |y|^2 with R'*R=inv(Sigma)
159 * where
160 * y = whiten(x) = R*x
161 * x = unwhiten(x) = inv(R)*y
162 * as indeed
163 * |y|^2 = y'*y = x'*R'*R*x
164 * Various derived classes are available that are more efficient.
165 * The named constructors return a shared_ptr because, when the smart flag is true,
166 * the underlying object might be a derived class such as Diagonal.
167 */
168 class GTSAM_EXPORT Gaussian: public Base {
169
170 protected:
171
172 /** Matrix square root of information matrix (R) */
173 std::optional<Matrix> sqrt_information_;
174
175 private:
176
177 /**
178 * Return R itself, but note that Whiten(H) is cheaper than R*H
179 */
180 const Matrix& thisR() const {
181 // should never happen
182 if (!sqrt_information_) throw std::runtime_error("Gaussian: has no R matrix");
183 return *sqrt_information_;
184 }
185
186 /// Compute the log of |R|. Used for computing log(|Î£|)
187 virtual double logDetR() const;
188
189 public:
190
191 typedef std::shared_ptr<Gaussian> shared_ptr;
192
193 /** constructor takes square root information matrix */
194 Gaussian(size_t dim = 1,
195 const std::optional<Matrix>& sqrt_information = {})
196 : Base(dim), sqrt_information_(sqrt_information) {}
197
198 ~Gaussian() override {}
199
200 /**
201 * A Gaussian noise model created by specifying a square root information matrix.
202 * @param R The (upper-triangular) square root information matrix
203 * @param smart check if can be simplified to derived class
204 */
205 static shared_ptr SqrtInformation(const Matrix& R, bool smart = true);
206
207 /**
208 * A Gaussian noise model created by specifying an information matrix.
209 * @param M The information matrix
210 * @param smart check if can be simplified to derived class
211 */
212 static shared_ptr Information(const Matrix& M, bool smart = true);
213
214 /**
215 * A Gaussian noise model created by specifying a covariance matrix.
216 * @param covariance The square covariance Matrix
217 * @param smart check if can be simplified to derived class
218 */
219 static shared_ptr Covariance(const Matrix& covariance, bool smart = true);
220
221 void print(const std::string& name) const override;
222 bool equals(const Base& expected, double tol=1e-9) const override;
223 Vector sigmas() const override;
224 Vector whiten(const Vector& v) const override;
225 Vector unwhiten(const Vector& v) const override;
226
227 /**
228 * Multiply a derivative with R (derivative of whiten)
229 * Equivalent to whitening each column of the input matrix.
230 */
231 Matrix Whiten(const Matrix& H) const override;
232
233 /**
234 * In-place version
235 */
236 virtual void WhitenInPlace(Matrix& H) const;
237
238 /**
239 * In-place version
240 */
241 virtual void WhitenInPlace(Eigen::Block<Matrix> H) const;
242
243 /**
244 * Whiten a system, in place as well
245 */
246 void WhitenSystem(std::vector<Matrix>& A, Vector& b) const override;
247 void WhitenSystem(Matrix& A, Vector& b) const override;
248 void WhitenSystem(Matrix& A1, Matrix& A2, Vector& b) const override;
249 void WhitenSystem(Matrix& A1, Matrix& A2, Matrix& A3, Vector& b) const override;
250
251 /**
252 * Apply appropriately weighted QR factorization to the system [A b]
253 * Q' * [A b] = [R d]
254 * Dimensions: (r*m) * m*(n+1) = r*(n+1), where r = min(m,n).
255 * This routine performs an in-place factorization on Ab.
256 * Below-diagonal elements are set to zero by this routine.
257 * @param Ab is the m*(n+1) augmented system matrix [A b]
258 * @return Empty SharedDiagonal() noise model: R,d are whitened
259 */
260 virtual std::shared_ptr<Diagonal> QR(Matrix& Ab) const;
261
262 /// Return R itself, but note that Whiten(H) is cheaper than R*H
263 virtual Matrix R() const { return thisR();}
264
265 /// Compute information matrix
266 virtual Matrix information() const;
267
268 /// Compute covariance matrix
269 virtual Matrix covariance() const;
270
271 /// Compute the log of |Î£|
272 double logDeterminant() const;
273
274 /**
275 * @brief Compute the negative log of the normalization constant
276 * for a Gaussian noise model k = 1/\sqrt(|2ÏÎ£|).
277 *
278 * @return double
279 */
280 double negLogConstant() const;
281
282 private:
283#if GTSAM_ENABLE_BOOST_SERIALIZATION
284 /** Serialization function */
285 friend class boost::serialization::access;
286 template<class ARCHIVE>
287 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
288 ar & BOOST_SERIALIZATION_BASE_OBJECT_NVP(Base);
289 ar & BOOST_SERIALIZATION_NVP(sqrt_information_);
290 }
291#endif
292 }; // Gaussian

296 /**
297 * A diagonal noise model implements a diagonal covariance matrix, with the
298 * elements of the diagonal specified in a Vector. This class has no public

313

315 virtual double logDetR() const override;

319 Diagonal();

322

324

326 * A diagonal noise model created by specifying a Vector of sigmas, i.e.
327 * standard deviations, the diagonal of the square root covariance matrix.
328 */
329 static shared_ptr Sigmas(const Vector& sigmas, bool smart = true);
330
331 /**
332 * A diagonal noise model created by specifying a Vector of variances, i.e.
333 * i.e. the diagonal of the covariance matrix.
334 * @param variances A vector containing the variances of this noise model
335 * @param smart check if can be simplified to derived class
336 */

338
339 /**

345 void print(const std::string& name) const override;

392 /**
393 * A Constrained constrained model is a specialization of Diagonal which allows
394 * some or all of the sigmas to be zero, forcing the error to be zero there.
395 * All other Gaussian models are guaranteed to have a non-singular square-root
396 * information matrix, but this class is specifically equipped to deal with
397 * singular noise models, specifically: whiten will return zero on those
398 * components that have zero sigma *and* zero error, unchanged otherwise.
399 *
400 * While a hard constraint may seem to be a case in which there is infinite error,
401 * we do not ever produce an error value of infinity to allow for constraints
402 * to actually be optimized rather than self-destructing if not initialized correctly.
403 */
404 class GTSAM_EXPORT Constrained : public Diagonal {
405 protected:
406
407 // Sigmas are contained in the base class
408 Vector mu_; ///< Penalty function weight - needs to be large enough to dominate soft constraints
409
410 /**
411 * Constructor that prevents any inf values
412 * from appearing in invsigmas or precisions.
413 * Allows for specifying mu.
414 */
415 Constrained(const Vector& mu, const Vector& sigmas);
416
417 public:
418
419 typedef std::shared_ptr<Constrained> shared_ptr;
420
421 /**
422 * protected constructor takes sigmas.
423 * prevents any inf values
424 * from appearing in invsigmas or precisions.
425 * mu set to large default value (1000.0)
426 */
427 Constrained(const Vector& sigmas = Z_1x1);
428
429 ~Constrained() override {}
430
431 /// true if a constrained noise mode, saves slow/clumsy dynamic casting
432 bool isConstrained() const override { return true; }
433
434 /// Return true if a particular dimension is free or constrained
435 bool constrained(size_t i) const;
436
437 /// Access mu as a vector
438 const Vector& mu() const { return mu_; }
439
440 /**
441 * A diagonal noise model created by specifying a Vector of
442 * standard devations, some of which might be zero
443 */
444 static shared_ptr MixedSigmas(const Vector& mu, const Vector& sigmas);
445
446 /**
447 * A diagonal noise model created by specifying a Vector of
448 * standard devations, some of which might be zero
449 */
450 static shared_ptr MixedSigmas(const Vector& sigmas);
451
452 /**
453 * A diagonal noise model created by specifying a Vector of
454 * standard devations, some of which might be zero
455 */
456 static shared_ptr MixedSigmas(double m, const Vector& sigmas);
457
458 /**
459 * A diagonal noise model created by specifying a Vector of
460 * standard devations, some of which might be zero
461 */
462 static shared_ptr MixedVariances(const Vector& mu, const Vector& variances);
463 static shared_ptr MixedVariances(const Vector& variances);
464
465 /**
466 * A diagonal noise model created by specifying a Vector of
467 * precisions, some of which might be inf
468 */
469 static shared_ptr MixedPrecisions(const Vector& mu, const Vector& precisions);
470 static shared_ptr MixedPrecisions(const Vector& precisions);
471
472 /**
473 * The squaredMahalanobisDistance function for a constrained noisemodel,
474 * for non-constrained versions, uses sigmas, otherwise
475 * uses the penalty function with mu
476 */
477 double squaredMahalanobisDistance(const Vector& v) const override;
478
479 /** Fully constrained variations */
480 static shared_ptr All(size_t dim) {
481 return shared_ptr(new Constrained(Vector::Constant(size: dim, value: 1000.0), Vector::Constant(size: dim,value: 0)));
482 }
483
484 /** Fully constrained variations */
485 static shared_ptr All(size_t dim, const Vector& mu) {
486 return shared_ptr(new Constrained(mu, Vector::Constant(size: dim,value: 0)));
487 }
488
489 /** Fully constrained variations with a mu parameter */
490 static shared_ptr All(size_t dim, double mu) {
491 return shared_ptr(new Constrained(Vector::Constant(size: dim, value: mu), Vector::Constant(size: dim,value: 0)));
492 }
493
494 void print(const std::string& name) const override;
495
496 /// Calculates error vector with weights applied
497 Vector whiten(const Vector& v) const override;
498
499 /// Whitening functions will perform partial whitening on rows
500 /// with a non-zero sigma. Other rows remain untouched.
501 Matrix Whiten(const Matrix& H) const override;
502 void WhitenInPlace(Matrix& H) const override;
503 void WhitenInPlace(Eigen::Block<Matrix> H) const override;
504
505 /**
506 * Apply QR factorization to the system [A b], taking into account constraints
507 * Q' * [A b] = [R d]
508 * Dimensions: (r*m) * m*(n+1) = r*(n+1), where r = min(m,n).
509 * This routine performs an in-place factorization on Ab.
510 * Below-diagonal elements are set to zero by this routine.
511 * @param Ab is the m*(n+1) augmented system matrix [A b]
512 * @return diagonal noise model can be all zeros, mixed, or not-constrained
513 */
514 Diagonal::shared_ptr QR(Matrix& Ab) const override;
515
516 /**
517 * Returns a Unit version of a constrained noisemodel in which
518 * constrained sigmas remain constrained and the rest are unit scaled
519 */
520 shared_ptr unit() const;
521
522 private:

661 /**
662 * Base class for robust error models
663 * The robust M-estimators above simply tell us how to re-weight the residual, and are
664 * isotropic kernels, in that they do not allow for correlated noise. They also have no way
665 * to scale the residual values, e.g., dividing by a single standard deviation.
666 * Hence, the actual robust noise model below does this scaling/whitening in sequence, by
667 * passing both a standard noise model and a robust estimator.
668 *
669 * Taking as an example noise = Isotropic::Create(d, sigma), we first divide the residuals
670 * uw = |Ax-b| by sigma by "whitening" the system (A,b), obtaining r = |Ax-b|/sigma, and
671 * then we pass the now whitened residual 'r' through the robust M-estimator.
672 * This is currently done by multiplying with sqrt(w), because the residuals will be squared
673 * again in error, yielding 0.5 \sum w(r)*r^2.
674 *
675 * In other words, while sigma is expressed in the native residual units, a parameter like
676 * k in the Huber norm is expressed in whitened units, i.e., "nr of sigmas".
677 */
678 class GTSAM_EXPORT Robust : public Base {
679 public:
680 typedef std::shared_ptr<Robust> shared_ptr;
681
682 protected:
683 typedef mEstimator::Base RobustModel;
684 typedef noiseModel::Base NoiseModel;
685
686 const RobustModel::shared_ptr robust_; ///< robust error function used
687 const NoiseModel::shared_ptr noise_; ///< noise model used
688
689 public:
690
691 /// Default Constructor for serialization
692 Robust() {}
693
694 /// Constructor
695 Robust(const RobustModel::shared_ptr robust, const NoiseModel::shared_ptr noise)
696 : Base(noise->dim()), robust_(robust), noise_(noise) {}
697
698 /// Destructor
699 ~Robust() override {}
700
701 void print(const std::string& name) const override;
702 bool equals(const Base& expected, double tol=1e-9) const override;
703
704 /// Return the contained robust error function
705 const RobustModel::shared_ptr& robust() const { return robust_; }
706
707 /// Return the contained noise model
708 const NoiseModel::shared_ptr& noise() const { return noise_; }
709
710 // Functions below are dummy but necessary for the noiseModel::Base
711 inline Vector whiten(const Vector& v) const override
712 { Vector r = v; this->WhitenSystem(b&: r); return r; }
713 inline Matrix Whiten(const Matrix& A) const override
714 { Vector b; Matrix B=A; this->WhitenSystem(A&: B,b); return B; }
715 inline Vector unwhiten(const Vector& /*v*/) const override
716 { throw std::invalid_argument("unwhiten is not currently supported for robust noise models."); }
717 /// Compute loss from the m-estimator using the Mahalanobis distance.
718 double loss(const double squared_distance) const override {
719 return robust_->loss(distance: std::sqrt(x: squared_distance));
720 }
721
722 // NOTE: This is special because in whiten the base version will do the reweighting
723 // which is incorrect!
724 double squaredMahalanobisDistance(const Vector& v) const override {
725 return noise_->squaredMahalanobisDistance(v);
726 }
727
728 // These are really robust iterated re-weighting support functions
729 virtual void WhitenSystem(Vector& b) const;
730 void WhitenSystem(std::vector<Matrix>& A, Vector& b) const override;
731 void WhitenSystem(Matrix& A, Vector& b) const override;
