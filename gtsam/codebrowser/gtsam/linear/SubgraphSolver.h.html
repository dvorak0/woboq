46/**
47 * This class implements the linear SPCG solver presented in Dellaert et al in
48 * IROS'10.
49 *
50 * Given a linear least-squares problem \f$ f(x) = |A x - b|^2 \f$. We split the
51 * problem into \f$ f(x) = |A_t - b_t|^2 + |A_c - b_c|^2 \f$ where \f$ A_t \f$
52 * denotes the "tree" part, and \f$ A_c \f$ denotes the "constraint" part.
53 *
54 * \f$A_t \f$ is factorized into \f$ Q_t R_t \f$, and we compute
55 * \f$ c_t = Q_t^{-1} b_t \f$, and \f$ x_t = R_t^{-1} c_t \f$ accordingly.
56 *
57 * Then we solve a reparametrized problem
58 * \f$ f(y) = |y|^2 + |A_c R_t^{-1} y -\bar{b_y}|^2 \f$,
59 * where \f$ y = R_t(x - x_t) \f$, and \f$ \bar{b_y} = (b_c - A_c x_t) \f$
60 *
61 * In the matrix form, it is equivalent to solving
62 * \f$ [A_c R_t^{-1} ; I ] y = [\bar{b_y} ; 0] \f$.
63 * We can solve it with the least-squares variation of the conjugate gradient
64 * method.
65 *
66 * To use it in nonlinear optimization, please see the following example
67 *
68 * LevenbergMarquardtParams parameters;
69 * parameters.linearSolverType = NonlinearOptimizerParams::CONJUGATE_GRADIENT;
70 * parameters.iterativeParams = std::make_shared<SubgraphSolverParameters>();
71 * LevenbergMarquardtOptimizer optimizer(graph, initialEstimate, parameters);
72 * Values result = optimizer.optimize();
73 *
74 * \nosubgrouping
75 */
76class GTSAM_EXPORT SubgraphSolver : public IterativeSolver {
77 public:
78 typedef SubgraphSolverParameters Parameters;
79
80 protected:
81 Parameters parameters_;
82 std::shared_ptr<SubgraphPreconditioner> pc_; ///< preconditioner object
83
84 public:
85 /// @name Constructors
86 /// @{
87 /**
88 * Given a gaussian factor graph, split it into a spanning tree (A1) + others
89 * (A2) for SPCG Will throw exception if there are ternary factors or higher
90 * arity, as we use Kruskal's algorithm to split the graph, treating binary
91 * factors as edges.
92 */
93 SubgraphSolver(const GaussianFactorGraph &A, const Parameters &parameters,
94 const Ordering &ordering);
95
96 /**
97 * The user specifies the subgraph part and the constraints part.
98 * May throw exception if A1 is underdetermined. An ordering is required to
99 * eliminate Ab1. We take Ab1 as a const reference, as it will be transformed
100 * into Rc1, but take Ab2 as a shared pointer as we need to keep it around.
101 */
102 SubgraphSolver(const GaussianFactorGraph &Ab1, const GaussianFactorGraph &Ab2,
103 const Parameters &parameters, const Ordering &ordering);
104 /**
105 * The same as above, but we assume A1 was solved by caller.
106 * We take two shared pointers as we keep both around.
107 */
108 SubgraphSolver(const GaussianBayesNet &Rc1, const GaussianFactorGraph &Ab2,
109 const Parameters &parameters);
110
111 /// Destructor
112 ~SubgraphSolver() override {}
113
114 /// @}
115 /// @name Implement interface
116 /// @{
117
118 /// Optimize from zero
119 VectorValues optimize() const;
120
121 /// Interface that IterativeSolver subclasses have to implement
122 VectorValues optimize(const GaussianFactorGraph &gfg,
123 const KeyInfo &keyInfo,
124 const std::map<Key, Vector> &lambda,
125 const VectorValues &initial) override;
126
127 /// @}
128 /// @name Implement interface
129 /// @{
130
131 /// Split graph using Kruskal algorithm, treating binary factors as edges.
132 std::pair<GaussianFactorGraph, GaussianFactorGraph> splitGraph(
133 const GaussianFactorGraph &gfg);
134
135 /// @}
136};
