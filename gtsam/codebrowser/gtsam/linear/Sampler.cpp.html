25/* ************************************************************************* */
26Sampler::Sampler(const noiseModel::Diagonal::shared_ptr& model,
27 uint_fast64_t seed)
28 : model_(model), generator_(seed) {
29 if (!model) {
30 throw std::invalid_argument("Sampler::Sampler needs a non-null model.");
31 }
32}
33
34/* ************************************************************************* */
35Sampler::Sampler(const Vector& sigmas, uint_fast64_t seed)
36 : model_(noiseModel::Diagonal::Sigmas(sigmas, smart: true)), generator_(seed) {}
37
38/* ************************************************************************* */
39Vector Sampler::sampleDiagonal(const Vector& sigmas, std::mt19937_64* rng) {
40 size_t d = sigmas.size();
41 Vector result(d);
42 for (size_t i = 0; i < d; i++) {
43 double sigma = sigmas(i);
44
45 // handle constrained case separately
46 if (sigma == 0.0) {
47 result(i) = 0.0;
48 } else {
49 std::normal_distribution<double> dist(0.0, sigma);
50 result(i) = dist(*rng);
51 }
52 }
53 return result;
54}
55
56/* ************************************************************************* */
57Vector Sampler::sampleDiagonal(const Vector& sigmas) const {
58 return sampleDiagonal(sigmas, rng: &generator_);
59}
60
61/* ************************************************************************* */
62Vector Sampler::sample() const {
63 assert(model_.get());
64 const Vector& sigmas = model_->sigmas();
65 return sampleDiagonal(sigmas);
66}
