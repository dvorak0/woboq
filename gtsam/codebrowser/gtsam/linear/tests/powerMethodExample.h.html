34/* ************************************************************************* */
35inline GaussianFactorGraph createSparseGraph() {
36 using symbol_shorthand::X;
37 // Let's make a scalar synchronization graph with 4 nodes
38 GaussianFactorGraph fg;
39 auto model = noiseModel::Unit::Create(dim: 1);
40 for (size_t j = 0; j < 3; j++) {
41 fg.add(key1: X(j), A1: -I_1x1, key2: X(j: j + 1), A2: I_1x1, b: Vector1::Zero(), model);
42 }
43 fg.add(key1: X(j: 3), A1: -I_1x1, key2: X(j: 0), A2: I_1x1, b: Vector1::Zero(), model); // extra row
44
45 return fg;
46}
47
48/* ************************************************************************* */
49inline GaussianFactorGraph createDenseGraph() {
50 using symbol_shorthand::X;
51 // Let's make a scalar synchronization graph with 10 nodes
52 GaussianFactorGraph fg;
53 auto model = noiseModel::Unit::Create(dim: 1);
54 // Iterate over nodes
55 for (size_t j = 0; j < 10; j++) {
56 // Each node has an edge with all the others
57 for (size_t i = 1; i < 10; i++)
58 fg.add(key1: X(j), A1: -I_1x1, key2: X(j: (j + i) % 10), A2: I_1x1, b: Vector1::Zero(), model);
59 }
60
61 return fg;
62}
