134TEST (Serialization, linear_factors) {
135 VectorValues values;
136 values.insert(j: 0, value: (Vector(1) << 1.0).finished());
137 values.insert(j: 1, value: Vector2(2.0,3.0));
138 values.insert(j: 2, value: Vector2(4.0,5.0));
139 EXPECT(equalsObj<VectorValues>(values));
140 EXPECT(equalsXML<VectorValues>(values));
141 EXPECT(equalsBinary<VectorValues>(values));
142
143 Key i1 = 4, i2 = 7;
144 Matrix A1 = I_3x3, A2 = -1.0 * I_3x3;
145 Vector b = Vector::Ones(newSize: 3);
146 SharedDiagonal model = noiseModel::Diagonal::Sigmas(sigmas: Vector3(1.0, 2.0, 3.0));
147 JacobianFactor jacobianfactor(i1, A1, i2, A2, b, model);
148 EXPECT(equalsObj(jacobianfactor));
149 EXPECT(equalsXML(jacobianfactor));
150 EXPECT(equalsBinary(jacobianfactor));
151
152 HessianFactor hessianfactor(jacobianfactor);
153 EXPECT(equalsObj(hessianfactor));
154 EXPECT(equalsXML(hessianfactor));
155 EXPECT(equalsBinary(hessianfactor));
156}

199TEST(Serialization, gaussian_bayes_net) {
200 // Create an arbitrary Bayes Net
201 GaussianBayesNet gbn;
202 gbn += GaussianConditional::shared_ptr(new GaussianConditional(
203 0, Vector2(1.0, 2.0), (Matrix2() << 3.0, 4.0, 0.0, 6.0).finished(), 3,
204 (Matrix2() << 7.0, 8.0, 9.0, 10.0).finished(), 4,
205 (Matrix2() << 11.0, 12.0, 13.0, 14.0).finished()));
206 gbn += GaussianConditional::shared_ptr(new GaussianConditional(
207 1, Vector2(15.0, 16.0), (Matrix2() << 17.0, 18.0, 0.0, 20.0).finished(),
208 2, (Matrix2() << 21.0, 22.0, 23.0, 24.0).finished(), 4,
209 (Matrix2() << 25.0, 26.0, 27.0, 28.0).finished()));
210 gbn += GaussianConditional::shared_ptr(new GaussianConditional(
211 2, Vector2(29.0, 30.0), (Matrix2() << 31.0, 32.0, 0.0, 34.0).finished(),
212 3, (Matrix2() << 35.0, 36.0, 37.0, 38.0).finished()));
213 gbn += GaussianConditional::shared_ptr(new GaussianConditional(
214 3, Vector2(39.0, 40.0), (Matrix2() << 41.0, 42.0, 0.0, 44.0).finished(),
215 4, (Matrix2() << 45.0, 46.0, 47.0, 48.0).finished()));
216 gbn += GaussianConditional::shared_ptr(new GaussianConditional(
217 4, Vector2(49.0, 50.0), (Matrix2() << 51.0, 52.0, 0.0, 54.0).finished()));
218
219 std::string serialized = serialize(input: gbn);
220 GaussianBayesNet actual;
221 deserialize(serialized, output&: actual);
222 EXPECT(assert_equal(gbn, actual));
223}

226TEST (Serialization, gaussian_bayes_tree) {
227 const Key x1=1, x2=2, x3=3, x4=4;
228 const Ordering chainOrdering {x2, x1, x3, x4};
229 const SharedDiagonal chainNoise = noiseModel::Isotropic::Sigma(dim: 1, sigma: 0.5);
230 const GaussianFactorGraph chain = {
231 std::make_shared<JacobianFactor>(args: x2, args&: (Matrix(1, 1) << 1.).finished(), args: x1, args&: (Matrix(1, 1) << 1.).finished(), args&: (Vector(1) << 1.).finished(), args: chainNoise),
232 std::make_shared<JacobianFactor>(args: x2, args&: (Matrix(1, 1) << 1.).finished(), args: x3, args&: (Matrix(1, 1) << 1.).finished(), args&: (Vector(1) << 1.).finished(), args: chainNoise),
233 std::make_shared<JacobianFactor>(args: x3, args&: (Matrix(1, 1) << 1.).finished(), args: x4, args&: (Matrix(1, 1) << 1.).finished(), args&: (Vector(1) << 1.).finished(), args: chainNoise),
234 std::make_shared<JacobianFactor>(args: x4, args&: (Matrix(1, 1) << 1.).finished(), args&: (Vector(1) << 1.).finished(), args: chainNoise)};
235
236 GaussianBayesTree init = *chain.eliminateMultifrontal(ordering: chainOrdering);
237 GaussianBayesTree expected = *chain.eliminateMultifrontal(ordering: chainOrdering);
238 GaussianBayesTree actual;
239
240 std::string serialized = serialize(input: init);
241 deserialize(serialized, output&: actual);
242 EXPECT(assert_equal(expected, actual));
243}
