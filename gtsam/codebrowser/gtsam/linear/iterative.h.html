27 /**
28 * Method of conjugate gradients (CG) template
29 * "System" class S needs gradient(S,v), e=S*v, v=S^e
30 * "Vector" class V needs dot(v,v), -v, v+v, s*v
31 * "Vector" class E needs dot(v,v)
32 * @param Ab, the "system" that needs to be solved, examples below
33 * @param x is the initial estimate
34 * @param steepest flag, if true does steepest descent, not CG
35 * */
36 template<class S, class V, class E>
37 V conjugateGradients(const S& Ab, V x,
38 const ConjugateGradientParameters &parameters, bool steepest = false);
39
40 /**
41 * Helper class encapsulating the combined system |Ax-b_|^2
42 * Needed to run Conjugate Gradients on matrices
43 * */
44 class GTSAM_EXPORT System {
45
46 private:
47 const Matrix& A_;
48 const Vector& b_;
49
50 public:
51
52 System(const Matrix& A, const Vector& b) :
53 A_(A), b_(b) {
54 }
55
56 /** Access A matrix */
57 const Matrix& A() const { return A_; }
58
59 /** Access b vector */
60 const Vector& b() const { return b_; }
61
62 /** Apply operator A'*e */
63 Vector operator^(const Vector& e) const {
64 return A_.transpose() * e;
65 }
66
67 /**
68 * Print with optional string
69 */
70 void print (const std::string& s = "System") const;
71
72 /** gradient of objective function 0.5*|Ax-b_|^2 at x = A_'*(Ax-b_) */
73 Vector gradient(const Vector& x) const {
74 return A().transpose() * (A() * x - b());
75 }
76
77 /** Apply operator A */
78 Vector operator*(const Vector& x) const {
79 return A() * x;
80 }
81
82 /** Apply operator A in place */
83 void multiplyInPlace(const Vector& x, Vector& e) const {
84 e = A() * x;
85 }
86
87 /** x += alpha* A'*e */
88 void transposeMultiplyAdd(double alpha, const Vector& e, Vector& x) const {
89 x += alpha * A().transpose() * e;
90 }
91 };
