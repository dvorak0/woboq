21#include <gtsam/linear/GaussianFactorGraph.h>

23#include <gtsam/inference/JunctionTree.h>

25namespace gtsam {
26
27 // Forward declarations
28 class GaussianEliminationTree;
29
30 /**
31 * A junction tree specialized to Gaussian factors, i.e., it is a cluster tree with Gaussian
32 * factors stored in each cluster. It can be eliminated into a Gaussian Bayes tree with the same
33 * structure, which is essentially doing multifrontal sparse matrix factorization.
34 *
35 * \ingroup Multifrontal
36 * \nosubgrouping
37 */
38 class GTSAM_EXPORT GaussianJunctionTree :
39 public JunctionTree<GaussianBayesTree, GaussianFactorGraph> {
40 public:
41 typedef JunctionTree<GaussianBayesTree, GaussianFactorGraph> Base; ///< Base class
42 typedef GaussianJunctionTree This; ///< This class
43 typedef std::shared_ptr<This> shared_ptr; ///< Shared pointer to this class
44
45 /**
46 * Build the elimination tree of a factor graph using pre-computed column structure.
47 * @param factorGraph The factor graph for which to build the elimination tree
48 * @param structure The set of factors involving each variable. If this is not
49 * precomputed, you can call the Create(const FactorGraph<DERIVEDFACTOR>&)
50 * named constructor instead.
51 * @return The elimination tree
52 */
53 GaussianJunctionTree(const GaussianEliminationTree& eliminationTree);
54 };
55
56}
