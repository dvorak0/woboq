36 /**
37 * VectorValues represents a collection of vector-valued variables associated
38 * each with a unique integer index. It is typically used to store the variables
39 * of a GaussianFactorGraph. Optimizing a GaussianFactorGraph or GaussianBayesNet
40 * returns this class.
41 *
42 * For basic usage, such as receiving a linear solution from gtsam solving functions,
43 * or creating this class in unit tests and examples where speed is not important,
44 * you can use a simple interface:
45 * - The default constructor VectorValues() to create this class
46 * - insert(Key, const Vector&) to add vector variables
47 * - operator[](Key) for read and write access to stored variables
48 * - \ref exists (Key) to check if a variable is present
49 * - Other facilities like iterators, size(), dim(), etc.
50 *
51 * Example:
52 * \code
53 VectorValues values;
54 values.emplace(3, Vector3(1.0, 2.0, 3.0));
55 values.emplace(4, Vector2(4.0, 5.0));
56 values.emplace(0, (Vector(4) << 6.0, 7.0, 8.0, 9.0).finished());
57
58 // Prints [ 3.0 4.0 ]
59 gtsam::print(values[1]);
60
61 // Prints [ 8.0 9.0 ]
62 values[1] = Vector2(8.0, 9.0);
63 gtsam::print(values[1]);
64 \endcode
65 *
66 * <h2>Advanced Interface and Performance Information</h2>
67 *
68 * Access is through the variable Key j, and returns a SubVector,
69 * which is a view on the underlying data structure.
70 *
71 * This class is additionally used in gradient descent and dog leg to store the gradient.
72 * @ingroup linear
73 */

138 /** Check whether a variable with key \c j exists. */

141 /**
142 * Read/write access to the vector value with key \c j, throws
143 * std::out_of_range if \c j does not exist, identical to operator[](Key).
144 */
145 Vector& at(Key j) {
146 iterator item = find(j);
147 if (item == end())
148 throw std::out_of_range(
149 "Requested variable '" + DefaultKeyFormatter(j) + "' is not in this VectorValues.");
150 else
151 return item->second;
152 }

159 const_iterator item = find(j);
160 if (item == end())
161 throw std::out_of_range(
162 "Requested variable '" + DefaultKeyFormatter(j) + "' is not in this VectorValues.");
163 else
164 return item->second;
165 }

180 /** Insert a vector \c value with key \c j. Throws an invalid_argument exception if the key \c
181 * j is already used.
182 * @param value The vector to be inserted.
183 * @param j The index with which the value will be associated. */
184 iterator insert(const std::pair<Key, Vector>& key_value);

186 /** Emplace a vector \c value with key \c j. Throws an invalid_argument exception if the key \c
187 * j is already used.
188 * @param value The vector to be inserted.
189 * @param j The index with which the value will be associated. */
190 template<class... Args>
191 inline std::pair<VectorValues::iterator, bool> emplace(Key j, Args&&... args) {
192#if ! defined(GTSAM_USE_TBB) || defined (TBB_GREATER_EQUAL_2020)
193 return values_.emplace(std::piecewise_construct, std::forward_as_tuple(args&: j), std::forward_as_tuple(args...));
194#else
195 return values_.insert({j, Vector(std::forward<Args>(args)...)});
196#endif
197 }
198
199 /** Insert a vector \c value with key \c j. Throws an invalid_argument exception if the key \c
200 * j is already used.
201 * @param value The vector to be inserted.
202 * @param j The index with which the value will be associated. */
203 iterator insert(Key j, const Vector& value) {
204 return insert(key_value: {j, value});
205 }
206
207 /** Insert all values from \c values. Throws an invalid_argument exception if any keys to be
208 * inserted are already used. */
209 VectorValues& insert(const VectorValues& values);
210
211 /** insert that mimics the STL map insert - if the value already exists, the map is not modified
212 * and an iterator to the existing value is returned, along with 'false'. If the value did not
213 * exist, it is inserted and an iterator pointing to the new element, along with 'true', is
214 * returned. */
215 inline std::pair<iterator, bool> tryInsert(Key j, const Vector& value) {
216#ifdef TBB_GREATER_EQUAL_2020
217 return values_.emplace(j, value);
218#else
219 return values_.insert(x: {j, value});
220#endif
221 }
222
223 /// insert_or_assign that mimics the STL map insert_or_assign - if the value already exists, the
224 /// map is updated, otherwise a new value is inserted at j.
225 void insert_or_assign(Key j, const Vector& value) {
226 if (!tryInsert(j, value).second) {
227 (*this)[j] = value;
228 }
229 }

231 /** Erase the vector with the given key, or throw std::out_of_range if it does not exist */

235 DefaultKeyFormatter(var) +

276 /** Access a vector that is a subset of relevant keys. */
277 template <typename CONTAINER>
278 Vector vector(const CONTAINER& keys) const {
279 DenseIndex totalDim = 0;
280 FastVector<const Vector*> items;
281 items.reserve(n: keys.end() - keys.begin());
282 for (Key key : keys) {
283 const Vector* v = &at(j: key);
284 totalDim += v->size();
285 items.push_back(x: v);
286 }
287
288 Vector result(totalDim);
289 DenseIndex pos = 0;
290 for (const Vector* v : items) {
291 result.segment(start: pos, n: v->size()) = *v;
292 pos += v->size();
293 }
294
295 return result;
296 }
