48 /** A Bayes tree representing a Gaussian density */
49 class GTSAM_EXPORT GaussianBayesTree :
50 public BayesTree<GaussianBayesTreeClique>
51 {
52 private:
53 typedef BayesTree<GaussianBayesTreeClique> Base;
54
55 public:
56 typedef GaussianBayesTree This;
57 typedef std::shared_ptr<This> shared_ptr;
58
59 /** Default constructor, creates an empty Bayes tree */
60 GaussianBayesTree() {}
61
62 /** Check equality */
63 bool equals(const This& other, double tol = 1e-9) const;
64
65 /** Recursively optimize the BayesTree to produce a vector solution. */
66 VectorValues optimize() const;
67
68 /**
69 * Optimize along the gradient direction, with a closed-form computation to perform the line
70 * search. The gradient is computed about \f$ \delta x=0 \f$.
71 *
72 * This function returns \f$ \delta x \f$ that minimizes a reparametrized problem. The error
73 * function of a GaussianBayesNet is
74 *
75 * \f[ f(\delta x) = \frac{1}{2} |R \delta x - d|^2 = \frac{1}{2}d^T d - d^T R \delta x +
76 * \frac{1}{2} \delta x^T R^T R \delta x \f]
77 *
78 * with gradient and Hessian
79 *
80 * \f[ g(\delta x) = R^T(R\delta x - d), \qquad G(\delta x) = R^T R. \f]
81 *
82 * This function performs the line search in the direction of the gradient evaluated at \f$ g =
83 * g(\delta x = 0) \f$ with step size \f$ \alpha \f$ that minimizes \f$ f(\delta x = \alpha g)
84 * \f$:
85 *
86 * \f[ f(\alpha) = \frac{1}{2} d^T d + g^T \delta x + \frac{1}{2} \alpha^2 g^T G g \f]
87 *
88 * Optimizing by setting the derivative to zero yields \f$ \hat \alpha = (-g^T g) / (g^T G g)
89 * \f$. For efficiency, this function evaluates the denominator without computing the Hessian
90 * \f$ G \f$, returning
91 *
92 * \f[ \delta x = \hat\alpha g = \frac{-g^T g}{(R g)^T(R g)} \f] */
93 VectorValues optimizeGradientSearch() const;
94
95 /** Compute the gradient of the energy function, \f$ \nabla_{x=x_0} \left\Vert \Sigma^{-1} R x -
96 * d \right\Vert^2 \f$, centered around \f$ x = x_0 \f$. The gradient is \f$ R^T(Rx-d) \f$.
97 *
98 * @param x0 The center about which to compute the gradient
99 * @return The gradient as a VectorValues */
100 VectorValues gradient(const VectorValues& x0) const;
101
102 /** Compute the gradient of the energy function, \f$ \nabla_{x=0} \left\Vert \Sigma^{-1} R x - d
103 * \right\Vert^2 \f$, centered around zero. The gradient about zero is \f$ -R^T d \f$. See also
104 * gradient(const GaussianBayesNet&, const VectorValues&).
105 *
106 * @return A VectorValues storing the gradient. */
107 VectorValues gradientAtZero() const;
108
109 /** 0.5 * sum of squared Mahalanobis distances. */
110 double error(const VectorValues& x) const;
111
112 /** Computes the determinant of a GassianBayesTree, as if the Bayes tree is reorganized into a
113 * matrix. A GassianBayesTree is equivalent to an upper triangular matrix, and for an upper
114 * triangular matrix determinant is the product of the diagonal elements. Instead of actually
115 * multiplying we add the logarithms of the diagonal elements and take the exponent at the end
116 * because this is more numerically stable. */
117 double determinant() const;
118
119 /** Computes the determinant of a GassianBayesTree, as if the Bayes tree is reorganized into a
120 * matrix. A GassianBayesTree is equivalent to an upper triangular matrix, and for an upper
121 * triangular matrix determinant is the product of the diagonal elements. Instead of actually
122 * multiplying we add the logarithms of the diagonal elements and take the exponent at the end
123 * because this is more numerically stable. */
124 double logDeterminant() const;
125
126 /** Return the marginal on the requested variable as a covariance matrix. See also
127 * marginalFactor(). */
128 Matrix marginalCovariance(Key key) const;
129 };
