40 struct GTSAM_EXPORT SubgraphPreconditionerParameters : public PreconditionerParameters {
41 typedef std::shared_ptr<SubgraphPreconditionerParameters> shared_ptr;
42 SubgraphPreconditionerParameters(const SubgraphBuilderParameters &p = SubgraphBuilderParameters())
43 : builderParams(p) {}
44 SubgraphBuilderParameters builderParams;
45 };

47 /**
48 * Subgraph conditioner class, as explained in the RSS 2010 submission.
49 * Starting with a graph A*x=b, we split it in two systems A1*x=b1 and A2*x=b2
50 * We solve R1*x=c1, and make the substitution y=R1*x-c1.
51 * To use the class, give the Bayes Net R1*x=c1 and Graph A2*x=b2.
52 * Then solve for yhat using CG, and solve for xhat = system.x(yhat).
53 */
54 class GTSAM_EXPORT SubgraphPreconditioner : public Preconditioner {
55
56 public:
57 typedef std::shared_ptr<SubgraphPreconditioner> shared_ptr;
58
59 private:
60 GaussianFactorGraph Ab2_;
61 GaussianBayesNet Rc1_;
62 VectorValues xbar_; ///< A1 \ b1
63 Errors b2bar_; ///< A2*xbar - b2
64
65 KeyInfo keyInfo_;
66 SubgraphPreconditionerParameters parameters_;
67
68 public:
69
70 SubgraphPreconditioner(const SubgraphPreconditionerParameters &p = SubgraphPreconditionerParameters());
71
72 /**
73 * Constructor
74 * @param Ab2: the Graph A2*x=b2
75 * @param Rc1: the Bayes Net R1*x=c1
76 * @param xbar: the solution to R1*x=c1
77 */
78 SubgraphPreconditioner(const GaussianFactorGraph& Ab2, const GaussianBayesNet& Rc1, const VectorValues& xbar,
79 const SubgraphPreconditionerParameters &p = SubgraphPreconditionerParameters());
80
81 ~SubgraphPreconditioner() override {}
82
83 /** print the object */
84 void print(const std::string& s = "SubgraphPreconditioner") const;
85
86 /** Access Ab2 */
87 const GaussianFactorGraph& Ab2() const { return Ab2_; }
88
89 /** Access Rc1 */
90 const GaussianBayesNet& Rc1() const { return Rc1_; }
91
92 /** Access b2bar */
93 const Errors b2bar() const { return b2bar_; }
94
95 /**
96 * Add zero-mean i.i.d. Gaussian prior terms to each variable
97 * @param sigma Standard deviation of Gaussian
98 */
99
100 /* x = xbar + inv(R1)*y */
101 VectorValues x(const VectorValues& y) const;
102
103 /* A zero VectorValues with the structure of xbar */
104 VectorValues zero() const {
105 return VectorValues::Zero(other: xbar_);
106 }
107
108 /**
109 * Add constraint part of the error only
110 * y += alpha*inv(R1')*A2'*e2
111 * Takes a range indicating e2 !!!!
112 */
113 void transposeMultiplyAdd2(double alpha, Errors::const_iterator begin,
114 Errors::const_iterator end, VectorValues& y) const;
115
116 /* error, given y */
117 double error(const VectorValues& y) const;
118
119 /** gradient = y + inv(R1')*A2'*(A2*inv(R1)*y-b2bar) */
120 VectorValues gradient(const VectorValues& y) const;
121
122 /** Apply operator A */
123 Errors operator*(const VectorValues& y) const;
124
125 /** Apply operator A in place: needs e allocated already */
126 void multiplyInPlace(const VectorValues& y, Errors& e) const;
127
128 /** Apply operator A' */
129 VectorValues operator^(const Errors& e) const;
130
131 /**
132 * Add A'*e to y
133 * y += alpha*A'*[e1;e2] = [alpha*e1; alpha*inv(R1')*A2'*e2]
134 */
135 void transposeMultiplyAdd(double alpha, const Errors& e, VectorValues& y) const;
136
137 /*****************************************************************************/
138 /* implement virtual functions of Preconditioner */
139
140 /// implement x = R^{-1} y
141 void solve(const Vector& y, Vector &x) const override;
142
143 /// implement x = R^{-T} y
144 void transposeSolve(const Vector& y, Vector& x) const override;
145
146 /// build/factorize the preconditioner
147 void build(
148 const GaussianFactorGraph &gfg,
149 const KeyInfo &info,
150 const std::map<Key,Vector> &lambda
151 ) override;
152 /*****************************************************************************/
153 };
