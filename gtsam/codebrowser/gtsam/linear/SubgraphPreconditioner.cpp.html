37/* ************************************************************************* */
38static Vector getSubvector(const Vector &src, const KeyInfo &keyInfo,
39 const KeyVector &keys) {
40 /* a cache of starting index and dim */
41 vector<std::pair<size_t, size_t> > cache;
42 cache.reserve(n: 3);
43
44 /* figure out dimension by traversing the keys */
45 size_t dim = 0;
46 for (const Key &key : keys) {
47 const KeyInfoEntry &entry = keyInfo.find(x: key)->second;
48 cache.emplace_back(args: entry.start, args: entry.dim);
49 dim += entry.dim;
50 }
51
52 /* use the cache to fill the result */
53 Vector result(dim);
54 size_t index = 0;
55 for (const auto &p : cache) {
56 result.segment(start: index, n: p.second) = src.segment(start: p.first, n: p.second);
57 index += p.second;
58 }
59
60 return result;
61}
62
63/*****************************************************************************/
64static void setSubvector(const Vector &src, const KeyInfo &keyInfo,
65 const KeyVector &keys, Vector &dst) {
66 size_t index = 0;
67 for (const Key &key : keys) {
68 const KeyInfoEntry &entry = keyInfo.find(x: key)->second;
69 const size_t keyDim = entry.dim;
70 dst.segment(start: entry.start, n: keyDim) = src.segment(start: index, n: keyDim);
71 index += keyDim;
72 }
73}

75/* ************************************************************************* */
76// Convert any non-Jacobian factors to Jacobians (e.g. Hessian -> Jacobian with
77// Cholesky)
78static GaussianFactorGraph convertToJacobianFactors(
79 const GaussianFactorGraph &gfg) {
80 GaussianFactorGraph result;
81 for (const auto &factor : gfg)
82 if (factor) {
83 auto jf = std::dynamic_pointer_cast<JacobianFactor>(r: factor);
84 if (!jf) {
85 jf = std::make_shared<JacobianFactor>(args&: *factor);
86 }
87 result.push_back(factor: jf);
88 }
89 return result;
90}

96/* ************************************************************************* */
97SubgraphPreconditioner::SubgraphPreconditioner(const GaussianFactorGraph& Ab2,
98 const GaussianBayesNet& Rc1, const VectorValues& xbar, const SubgraphPreconditionerParameters &p) :
99 Ab2_(convertToJacobianFactors(gfg: Ab2)), Rc1_(Rc1), xbar_(xbar),
100 b2bar_(-Ab2_.gaussianErrors(x: xbar)), parameters_(p) {
101}

103/* ************************************************************************* */
104// x = xbar + inv(R1)*y
105VectorValues SubgraphPreconditioner::x(const VectorValues& y) const {
106 return xbar_ + Rc1_.backSubstitute(gx: y);
107}

109/* ************************************************************************* */
110double SubgraphPreconditioner::error(const VectorValues& y) const {
111 Errors e = createErrors(V: y);
112 VectorValues x = this->x(y);
113 Errors e2 = Ab2_.gaussianErrors(x);
114 return 0.5 * (dot(a: e, b: e) + dot(a: e2,b: e2));
115}

117/* ************************************************************************* */
118// gradient is y + inv(R1')*A2'*(A2*inv(R1)*y-b2bar),
119VectorValues SubgraphPreconditioner::gradient(const VectorValues &y) const {
120 VectorValues x = Rc1_.backSubstitute(gx: y); /* inv(R1)*y */
121 Errors e = Ab2_ * x - b2bar_; /* (A2*inv(R1)*y-b2bar) */
122 VectorValues v = VectorValues::Zero(other: x);
123 Ab2_.transposeMultiplyAdd(alpha: 1.0, e, x&: v); /* A2'*(A2*inv(R1)*y-b2bar) */
124 return y + Rc1_.backSubstituteTranspose(gx: v);
125}
126
127/* ************************************************************************* */
128// Apply operator A, A*y = [I;A2*inv(R1)]*y = [y; A2*inv(R1)*y]
129Errors SubgraphPreconditioner::operator*(const VectorValues &y) const {
130 Errors e = createErrors(V: y);
131 VectorValues x = Rc1_.backSubstitute(gx: y); /* x=inv(R1)*y */
132 Errors e2 = Ab2_ * x; /* A2*x */
133 e.splice(position: e.end(), x&: e2);
134 return e;
135}

137/* ************************************************************************* */
138// In-place version that overwrites e
139void SubgraphPreconditioner::multiplyInPlace(const VectorValues& y, Errors& e) const {
140
141 Errors::iterator ei = e.begin();
142 for(const auto& key_value: y) {
143 *ei = key_value.second;
144 ++ei;
145 }
146
147 // Add A2 contribution
148 VectorValues x = Rc1_.backSubstitute(gx: y); // x=inv(R1)*y
149 Ab2_.multiplyInPlace(x, e: ei); // use iterator version
150}

152/* ************************************************************************* */
153// Apply operator A', A'*e = [I inv(R1')*A2']*e = e1 + inv(R1')*A2'*e2
154VectorValues SubgraphPreconditioner::operator^(const Errors& e) const {
155
156 Errors::const_iterator it = e.begin();
157 VectorValues y = zero();
158 for(auto& key_value: y) {
159 key_value.second = *it;
160 ++it;
161 }
162 transposeMultiplyAdd2(alpha: 1.0, begin: it, end: e.end(), y);
163 return y;
164}

166/* ************************************************************************* */
167// y += alpha*A'*e
168void SubgraphPreconditioner::transposeMultiplyAdd
169(double alpha, const Errors& e, VectorValues& y) const {
170
171 Errors::const_iterator it = e.begin();
172 for(auto& key_value: y) {
173 const Vector& ei = *it;
174 key_value.second += alpha * ei;
175 ++it;
176 }
177 transposeMultiplyAdd2(alpha, begin: it, end: e.end(), y);
178}
179
180/* ************************************************************************* */
181// y += alpha*inv(R1')*A2'*e2
182void SubgraphPreconditioner::transposeMultiplyAdd2 (double alpha,
183 Errors::const_iterator it, Errors::const_iterator end, VectorValues& y) const {
184
185 // create e2 with what's left of e
186 // TODO can we avoid creating e2 by passing iterator to transposeMultiplyAdd ?
187 Errors e2;
188 while (it != end) e2.push_back(x: *(it++));
189
190 VectorValues x = VectorValues::Zero(other: y); // x = 0
191 Ab2_.transposeMultiplyAdd(alpha: 1.0,e: e2,x); // x += A2'*e2
192 y += alpha * Rc1_.backSubstituteTranspose(gx: x); // y += alpha*inv(R1')*x
193}

201/*****************************************************************************/
202void SubgraphPreconditioner::solve(const Vector &y, Vector &x) const {
203 assert(x.size() == y.size());
204
205 /* back substitute */
206 for (auto it = std::make_reverse_iterator(i: Rc1_.end()); it != std::make_reverse_iterator(i: Rc1_.begin()); ++it) {
207 auto& cg = *it;
208 /* collect a subvector of x that consists of the parents of cg (S) */
209 const KeyVector parentKeys(cg->beginParents(), cg->endParents());
210 const KeyVector frontalKeys(cg->beginFrontals(), cg->endFrontals());
211 const Vector xParent = getSubvector(src: x, keyInfo: keyInfo_, keys: parentKeys);
212 const Vector rhsFrontal = getSubvector(src: y, keyInfo: keyInfo_, keys: frontalKeys);
213
214 /* compute the solution for the current pivot */
215 const Vector solFrontal = cg->R().triangularView<Eigen::Upper>().solve(
216 other: rhsFrontal - cg->S() * xParent);
217
218 /* assign subvector of sol to the frontal variables */
219 setSubvector(src: solFrontal, keyInfo: keyInfo_, keys: frontalKeys, dst&: x);
220 }
221}

223/*****************************************************************************/
224void SubgraphPreconditioner::transposeSolve(const Vector &y, Vector &x) const {
225 /* copy first */
226 assert(x.size() == y.size());
227 std::copy(first: y.data(), last: y.data() + y.rows(), result: x.data());
228
229 /* in place back substitute */
230 for (const auto &cg : Rc1_) {
231 const KeyVector frontalKeys(cg->beginFrontals(), cg->endFrontals());
232 const Vector rhsFrontal = getSubvector(src: x, keyInfo: keyInfo_, keys: frontalKeys);
233 const Vector solFrontal =
234 cg->R().transpose().triangularView<Eigen::Lower>().solve(
235 other: rhsFrontal);
236
237 // Check for indeterminant solution
238 if (solFrontal.hasNaN())
239 throw IndeterminantLinearSystemException(cg->keys().front());
240
241 /* assign subvector of sol to the frontal variables */
242 setSubvector(src: solFrontal, keyInfo: keyInfo_, keys: frontalKeys, dst&: x);
243
244 /* substract from parent variables */
245 for (auto it = cg->beginParents(); it != cg->endParents(); it++) {
246 const KeyInfoEntry &entry = keyInfo_.find(x: *it)->second;
247 Eigen::Map<Vector> rhsParent(x.data() + entry.start, entry.dim, 1);
248 rhsParent -= Matrix(cg->getA(variable: it)).transpose() * solFrontal;
249 }
250 }
251}

253/*****************************************************************************/
254void SubgraphPreconditioner::build(const GaussianFactorGraph &gfg, const KeyInfo &keyInfo, const std::map<Key,Vector> &lambda)
255{
256 /* identify the subgraph structure */
257 const SubgraphBuilder builder(parameters_.builderParams);
258 auto subgraph = builder(gfg);
259
260 keyInfo_ = keyInfo;
261
262 /* build factor subgraph */
263 auto gfg_subgraph = buildFactorSubgraph(gfg, subgraph, clone: true);
264
265 /* factorize and cache BayesNet */
266 Rc1_ = *gfg_subgraph.eliminateSequential();
267}
