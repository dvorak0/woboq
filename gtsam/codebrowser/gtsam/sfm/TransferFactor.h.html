29/**
30 * Base class that holds the EdgeKeys and provides the getMatrices method.
31 */
32template <typename F>
33class TransferEdges {
34 protected:
35 EdgeKey edge1_, edge2_; ///< The two EdgeKeys.
36 uint32_t c_; ///< The transfer target
37
38 public:
39 TransferEdges(EdgeKey edge1, EdgeKey edge2)
40 : edge1_(edge1), edge2_(edge2), c_(ViewC(edge1, edge2)) {}
41
42 /// Returns the view A index based on the EdgeKeys
43 static size_t ViewA(const EdgeKey& edge1, const EdgeKey& edge2) {
44 size_t c = ViewC(edge1, edge2);
45 return (edge1.i() == c) ? edge1.j() : edge1.i();
46 }
47
48 /// Returns the view B index based on the EdgeKeys
49 static size_t ViewB(const EdgeKey& edge1, const EdgeKey& edge2) {
50 size_t c = ViewC(edge1, edge2);
51 return (edge2.i() == c) ? edge2.j() : edge2.i();
52 }
53
54 /// Returns the view C index based on the EdgeKeys
55 static size_t ViewC(const EdgeKey& edge1, const EdgeKey& edge2) {
56 if (edge1.i() == edge2.i() || edge1.i() == edge2.j())
57 return edge1.i();
58 else if (edge1.j() == edge2.i() || edge1.j() == edge2.j())
59 return edge1.j();
60 else
61 throw std::runtime_error(
62 "EssentialTransferFactorK: No common key in edge keys.");
63 }
64
65 /// Create Matrix3 objects based on EdgeKey configurations.
66 std::pair<Matrix3, Matrix3> getMatrices(const F& F1, const F& F2) const {
67 // Determine whether to transpose F1
68 const Matrix3 Fca =
69 edge1_.i() == c_ ? F1.matrix() : F1.matrix().transpose();
70
71 // Determine whether to transpose F2
72 const Matrix3 Fcb =
73 edge2_.i() == c_ ? F2.matrix() : F2.matrix().transpose();
74
75 return {Fca, Fcb};
76 }
77};

79/**
80 * Binary factor in the context of Structure from Motion (SfM).
81 * It is used to transfer transfer corresponding points from two views to a
82 * third based on two fundamental matrices. The factor computes the error
83 * between the transferred points `pa` and `pb`, and the actual point `pc` in
84 * the target view. Jacobians are done using numerical differentiation.
85 */
86template <typename F>
87class TransferFactor : public NoiseModelFactorN<F, F>, public TransferEdges<F> {
88 public:
89 using Base = NoiseModelFactorN<F, F>;
90 using Triplet = std::tuple<Point2, Point2, Point2>;
91
92 protected:
93 std::vector<Triplet> triplets_; ///< Point triplets.
94
95 public:
96 /**
97 * @brief Constructor that accepts a vector of point triplets.
98 *
99 * @param edge1 First EdgeKey specifying F1: (a, c) or (c, a).
100 * @param edge2 Second EdgeKey specifying F2: (b, c) or (c, b).
101 * @param triplets A vector of triplets containing (pa, pb, pc).
102 * @param model An optional SharedNoiseModel that defines the noise model
103 * for this factor. Defaults to nullptr.
104 */
105 TransferFactor(EdgeKey edge1, EdgeKey edge2,
106 const std::vector<Triplet>& triplets,
107 const SharedNoiseModel& model = nullptr)
108 : Base(model, edge1, edge2),
109 TransferEdges<F>(edge1, edge2),
110 triplets_(triplets) {}
111
112 /// Vector of errors returns 2*N vector.
113 Vector evaluateError(const F& F1, const F& F2,
114 OptionalMatrixType H1 = nullptr,
115 OptionalMatrixType H2 = nullptr) const override {
116 std::function<Vector(const F&, const F&)> errorFunction = [&](const F& f1,
117 const F& f2) {
118 Vector errors(2 * triplets_.size());
119 size_t idx = 0;
120 auto [Fca, Fcb] = this->getMatrices(f1, f2);
121 for (const auto& [pa, pb, pc] : triplets_) {
122 errors.segment<2>(start: idx) = EpipolarTransfer(Fca, pa, Fcb, pb) - pc;
123 idx += 2;
124 }
125 return errors;
126 };
127
128 if (H1) *H1 = numericalDerivative21(errorFunction, F1, F2);
129 if (H2) *H2 = numericalDerivative22(errorFunction, F1, F2);
130 return errorFunction(F1, F2);
131 }
132};

134/**
135 * @class EssentialTransferFactor
136 * @brief Transfers points between views using essential matrices with a shared
137 * calibration.
138 *
139 * This factor is templated on the calibration class K and extends
140 * the TransferFactor for EssentialMatrices. It involves two essential matrices
141 * and a shared calibration object (K). The evaluateError function calibrates
142 * the image points, calls the base class's transfer method, and computes the
143 * error using bulk numerical differentiation.
144 */
145template <typename K>
146class EssentialTransferFactor : public TransferFactor<EssentialMatrix> {
147 using EM = EssentialMatrix;
148 using Triplet = std::tuple<Point2, Point2, Point2>;
149 std::shared_ptr<K> calibration_; ///< Shared pointer to calibration object
150
151 public:
152 using Base = TransferFactor<EM>;
153 using This = EssentialTransferFactor<K>;
154 using shared_ptr = std::shared_ptr<This>;
155
156 /**
157 * @brief Constructor that accepts a vector of point triplets and a shared
158 * calibration.
159 *
160 * @param edge1 First EdgeKey specifying E1: (a, c) or (c, a)
161 * @param edge2 Second EdgeKey specifying E2: (b, c) or (c, b)
162 * @param triplets A vector of triplets containing (pa, pb, pc)
163 * @param calibration Shared pointer to calibration object
164 * @param model An optional SharedNoiseModel
165 */
166 EssentialTransferFactor(EdgeKey edge1, EdgeKey edge2,
167 const std::vector<Triplet>& triplets,
168 const std::shared_ptr<K>& calibration,
169 const SharedNoiseModel& model = nullptr)
170 : Base(edge1, edge2, triplets, model), calibration_(calibration) {}
171
172 /// Transfer points pa and pb to view c and evaluate error.
173 Vector2 TransferError(const Matrix3& Eca, const Point2& pa,
174 const Matrix3& Ecb, const Point2& pb,
175 const Point2& pc) const {
176 const Point2 pA = calibration_->calibrate(pa);
177 const Point2 pB = calibration_->calibrate(pb);
178 const Point2 pC = EpipolarTransfer(Fca: Eca, pa: pA, Fcb: Ecb, pb: pB);
179 return calibration_->uncalibrate(pC) - pc;
180 }
181
182 /// Evaluate error function
183 Vector evaluateError(const EM& E1, const EM& E2,
184 OptionalMatrixType H1 = nullptr,
185 OptionalMatrixType H2 = nullptr) const override {
186 std::function<Vector(const EM&, const EM&)> errorFunction =
187 [&](const EM& e1, const EM& e2) {
188 Vector errors(2 * this->triplets_.size());
189 size_t idx = 0;
190 auto [Eca, Ecb] = this->getMatrices(e1, e2);
191 for (const auto& [pa, pb, pc] : this->triplets_) {
192 errors.segment<2>(start: idx) = TransferError(Eca, pa, Ecb, pb, pc);
193 idx += 2;
194 }
195 return errors;
196 };
197
198 // Compute error
199 Vector errors = errorFunction(E1, E2);
200
201 // Compute Jacobians if requested
202 if (H1) *H1 = numericalDerivative21(h: errorFunction, x1: E1, x2: E2);
203 if (H2) *H2 = numericalDerivative22(h: errorFunction, x1: E1, x2: E2);
204
205 return errors;
206 }
207};

209/**
210 * @class EssentialTransferFactorK
211 * @brief Transfers points between views using essential matrices, optimizes for
212 * calibrations of the views, as well. Note that the EssentialMatrixFactor4 does
213 * something similar but without transfer.
214 *
215 * @note Derives calibration keys from edges, and uses symbol 'k'.
216 *
217 * This factor is templated on the calibration class K and extends
218 * the TransferFactor for EssentialMatrices. It involves two essential matrices
219 * and three calibration objects (Ka, Kb, Kc). The evaluateError
220 * function calibrates the image points, calls the base class's transfer method,
221 * and computes the error using bulk numerical differentiation.
222 */
223template <typename K>
224class EssentialTransferFactorK
225 : public NoiseModelFactorN<EssentialMatrix, EssentialMatrix, K, K, K>,
226 TransferEdges<EssentialMatrix> {
227 using EM = EssentialMatrix;
228 using Triplet = std::tuple<Point2, Point2, Point2>;
229 std::vector<Triplet> triplets_; ///< Point triplets
230
231 public:
232 using Base = NoiseModelFactorN<EM, EM, K, K, K>;
233 using This = EssentialTransferFactorK<K>;
234 using shared_ptr = std::shared_ptr<This>;
235
236 /**
237 * @brief Constructor that accepts a vector of point triplets.
238 *
239 * @note Calibrations are assumed all different, keys are derived from edges.
240 *
241 * @param edge1 First EdgeKey specifying E1: (a, c) or (c, a)
242 * @param edge2 Second EdgeKey specifying E2: (b, c) or (c, b)
243 * @param triplets A vector of triplets containing (pa, pb, pc)
244 * @param model An optional SharedNoiseModel
245 */
246 EssentialTransferFactorK(EdgeKey edge1, EdgeKey edge2,
247 const std::vector<Triplet>& triplets,
248 const SharedNoiseModel& model = nullptr)
249 : Base(model, edge1, edge2,
250 Symbol('k', ViewA(edge1, edge2)), // calibration key for view a
251 Symbol('k', ViewB(edge1, edge2)), // calibration key for view b
252 Symbol('k', ViewC(edge1, edge2))), // calibration key for target c
253 TransferEdges<EM>(edge1, edge2),
254 triplets_(triplets) {}
255
256 /**
257 * @brief Constructor that accepts a vector of point triplets.
258 *
259 * @note Calibrations are assumed all same, using given key `keyK`.
260 *
261 * @param edge1 First EdgeKey specifying E1: (a, c) or (c, a)
262 * @param edge2 Second EdgeKey specifying E2: (b, c) or (c, b)
263 * @param keyK Calibration key for all views.
264 * @param triplets A vector of triplets containing (pa, pb, pc)
265 * @param model An optional SharedNoiseModel
266 */
267 EssentialTransferFactorK(EdgeKey edge1, EdgeKey edge2, Key keyK,
268 const std::vector<Triplet>& triplets,
269 const SharedNoiseModel& model = nullptr)
270 : Base(model, edge1, edge2, keyK, keyK, keyK),
271 TransferEdges<EM>(edge1, edge2),
272 triplets_(triplets) {}
273
274 /// Transfer points pa and pb to view c and evaluate error.
275 Vector2 TransferError(const Matrix3& Eca, const K& Ka, const Point2& pa,
276 const Matrix3& Ecb, const K& Kb, const Point2& pb,
277 const K& Kc, const Point2& pc) const {
278 const Point2 pA = Ka.calibrate(pa);
279 const Point2 pB = Kb.calibrate(pb);
280 const Point2 pC = EpipolarTransfer(Fca: Eca, pa: pA, Fcb: Ecb, pb: pB);
281 return Kc.uncalibrate(pC) - pc;
282 }
283
284 /// Evaluate error function
285 Vector evaluateError(const EM& E1, const EM& E2, const K& Ka, const K& Kb,
286 const K& Kc, OptionalMatrixType H1 = nullptr,
287 OptionalMatrixType H2 = nullptr,
288 OptionalMatrixType H3 = nullptr,
289 OptionalMatrixType H4 = nullptr,
290 OptionalMatrixType H5 = nullptr) const override {
291 std::function<Vector(const EM&, const EM&, const K&, const K&, const K&)>
292 errorFunction = [&](const EM& e1, const EM& e2, const K& kA,
293 const K& kB, const K& kC) {
294 Vector errors(2 * triplets_.size());
295 size_t idx = 0;
296 auto [Eca, Ecb] = this->getMatrices(e1, e2);
297 for (const auto& [pa, pb, pc] : triplets_) {
298 errors.segment<2>(start: idx) =
299 TransferError(Eca, Ka: kA, pa, Ecb, Kb: kB, pb, Kc: kC, pc);
300 idx += 2;
301 }
302 return errors;
303 };
304
305 // Compute error
306 Vector errors = errorFunction(E1, E2, Ka, Kb, Kc);
307
308 // Compute Jacobians if requested
309 if (H1) *H1 = numericalDerivative51(errorFunction, E1, E2, Ka, Kb, Kc);
310 if (H2) *H2 = numericalDerivative52(errorFunction, E1, E2, Ka, Kb, Kc);
311 if (H3) *H3 = numericalDerivative53(errorFunction, E1, E2, Ka, Kb, Kc);
312 if (H4) *H4 = numericalDerivative54(errorFunction, E1, E2, Ka, Kb, Kc);
313 if (H5) *H5 = numericalDerivative55(errorFunction, E1, E2, Ka, Kb, Kc);
314
315 return errors;
316 }
317
318 /// Return the dimension of the factor
319 size_t dim() const override { return 2 * triplets_.size(); }
320};
