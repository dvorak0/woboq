168TEST(ShonanAveraging3, CheckWithEigen) {
169 // control randomness
170 static std::mt19937 rng(0);
171 Vector descentDirection = Vector::Random(size: 15); // for use below
172 const Values randomRotations = kShonan.initializeRandomly(rng);
173 Values random = ShonanAveraging3::LiftTo<Rot3>(p: 3, values: randomRotations);
174
175 // Optimize
176 const Values Qstar3 = kShonan.tryOptimizingAt(p: 3, initial: random);
177
178 // Compute Eigenvalue with Spectra solver
179 double lambda = kShonan.computeMinEigenValue(values: Qstar3);
180
181 // Check Eigenvalue with slow Eigen version, converts matrix A to dense matrix!
182 const Matrix S = ShonanAveraging3::StiefelElementMatrix(values: Qstar3);
183 auto A = kShonan.computeA(S);
184 bool computeEigenvectors = false;
185 Eigen::EigenSolver<Matrix> eigenSolver(Matrix(A), computeEigenvectors);
186 auto lambdas = eigenSolver.eigenvalues().real();
187 double minEigenValue = lambdas(0);
188 for (int i = 1; i < lambdas.size(); i++)
189 minEigenValue = min(a: lambdas(i), b: minEigenValue);
190
191 // Compute Eigenvalue with Accelerated Power method
192 double lambdaAP = kShonan.computeMinEigenValueAP(values: Qstar3);
193
194 // Actual check
195 EXPECT_DOUBLES_EQUAL(0, lambda, 1e-11);
196 EXPECT_DOUBLES_EQUAL(0, minEigenValue, 1e-11);
197 EXPECT_DOUBLES_EQUAL(0, lambdaAP, 1e-11);
198
199
200 // Construct test descent direction (as minEigenVector is not predictable
201 // across platforms, being one from a basically flat 3d- subspace)
202
203 // Check descent
204 Values initialQ4 =
205 ShonanAveraging3::LiftwithDescent(p: 4, values: Qstar3, minEigenVector: descentDirection);
206 EXPECT_LONGS_EQUAL(5, initialQ4.size());
207
208 Matrix expected(4, 4);
209#if __APPLE__
210 expected << 0.145767, -0.938445, 0.135713, -0.282233, //
211 0.780348, -0.0104323, 0.266238, 0.565743, //
212 -0.383624, 0.0434887, 0.917211, 0.0983088, //
213 -0.471849, -0.342523, -0.263482, 0.768514;
214#elif __linux__
215 expected << 0.100724, -0.987231, 0.104092, 0.0662867, //
216 0.571527, 0.0292782, 0.226546, -0.788147, //
217 -0.349294, 0.064102, 0.93465, 0.0177471, //
218 0.735667, 0.142857, 0.253519, 0.611649;
219#elif _WIN32
220 expected << 0.0825862, -0.645931, 0.271896, 0.708537, //
221 0.927285, -0.0156335, 0.291603, -0.234236, //
222 -0.36419, -0.132115, 0.831933, -0.39724, //
223 0.0262425, 0.751715, 0.385912, 0.534143;
224#endif
225
226 EXPECT(assert_equal(SOn(expected), initialQ4.at<SOn>(0), 1e-5));
227}

261TEST(ShonanAveraging3, runKlaus) {
262 using namespace klaus;
263
264 // Initialize a Shonan instance without the Karcher mean
265 ShonanAveraging3::Parameters parameters;
266 parameters.setKarcherWeight(0);
267
268 // Load 3 pose example taken in Klaus by Shicong
269 static const ShonanAveraging3 shonan =
270 fromExampleName(name: "Klaus3.g2o", parameters);
271
272 // Check nr poses
273 EXPECT_LONGS_EQUAL(3, shonan.nrUnknowns());
274
275 // Colmap uses the Y-down vision frame, and the first 3 rotations are close to
276 // identity. We check that below. Note tolerance is quite high.
277 static const Rot3 identity;
278 EXPECT(assert_equal(identity, wR0, 0.2));
279 EXPECT(assert_equal(identity, wR1, 0.2));
280 EXPECT(assert_equal(identity, wR2, 0.2));
281
282 // Get measurements
283 const Rot3 R01 = shonan.measured(k: 0);
284 const Rot3 R12 = shonan.measured(k: 1);
285 const Rot3 R02 = shonan.measured(k: 2);
286
287 // Regression test to make sure data did not change.
288 EXPECT(assert_equal(Rot3(0.9995433591728293, -0.022048798853273946,
289 -0.01796327847857683, 0.010210006313668573),
290 R01));
291
292 // Check Colmap solution agrees OK with relative rotation measurements.
293 EXPECT(assert_equal(R01, wR0.between(wR1), 0.1));
294 EXPECT(assert_equal(R12, wR1.between(wR2), 0.1));
295 EXPECT(assert_equal(R02, wR0.between(wR2), 0.1));
296
297 // Run Shonan (with prior on first rotation)
298 auto initial = shonan.initializeRandomly(rng&: kPRNG);
299 auto result = shonan.run(initialEstimate: initial, pMin: 5);
300 EXPECT_DOUBLES_EQUAL(0, shonan.cost(result.first), 1e-2);
301 EXPECT_DOUBLES_EQUAL(-9.2259161494467889e-05, result.second,
302 1e-4); // Regression
303
304 // Get Shonan solution in new frame R (R for result)
305 const Rot3 rR0 = result.first.at<Rot3>(j: 0);
306 const Rot3 rR1 = result.first.at<Rot3>(j: 1);
307 const Rot3 rR2 = result.first.at<Rot3>(j: 2);
308
309 // rR0 = rRw * wR0 => rRw = rR0 * wR0.inverse()
310 // rR1 = rRw * wR1
311 // rR2 = rRw * wR2
312
313 const Rot3 rRw = rR0 * wR0.inverse();
314 EXPECT(assert_equal(rRw * wR1, rR1, 0.1))
315 EXPECT(assert_equal(rRw * wR2, rR2, 0.1))
316}

363TEST(ShonanAveraging2, noisyToyGraphWithHuber) {
364 // Load 2D toy example
365 auto lmParams = LevenbergMarquardtParams::CeresDefaults();
366 string g2oFile = findExampleDataFile(name: "noisyToyGraph.txt");
367 ShonanAveraging2::Parameters parameters(lmParams);
368 auto measurements = parseMeasurements<Rot2>(filename: g2oFile);
369 parameters.setUseHuber(true);
370 parameters.setCertifyOptimality(false);
371
372 string parameters_print =
373 " ShonanAveragingParameters: \n alpha: 0\n beta: 1\n gamma: 0\n "
374 "useHuber: 1\n";
375 assert_print_equal(expected: parameters_print, actual: parameters);
376
377 ShonanAveraging2 shonan(measurements, parameters);
378 EXPECT_LONGS_EQUAL(4, shonan.nrUnknowns());
379
380 // Check graph building
381 NonlinearFactorGraph graph = shonan.buildGraphAt(p: 2);
382 EXPECT_LONGS_EQUAL(6, graph.size());
383
384 // test that each factor is actually robust
385 for (size_t i=0; i<=4; i++) { // note: last is the Gauge factor and is not robust
386 const auto &robust = std::dynamic_pointer_cast<noiseModel::Robust>(
387 r: graph.at<NoiseModelFactor>(i)->noiseModel());
388 // we expect the factors to be use a robust noise model
389 // (in particular, Huber)
390 EXPECT(robust);
391 }
392
393 // test result
394 auto initial = shonan.initializeRandomly(rng&: kPRNG);
395 auto result = shonan.run(initialEstimate: initial, pMin: 2,pMax: 2);
396 EXPECT_DOUBLES_EQUAL(0.0008211, shonan.cost(result.first), 1e-6);
397 EXPECT_DOUBLES_EQUAL(0, result.second, 1e-10); // certificate!

401// Test alpha/beta/gamma prior weighting.
402TEST(ShonanAveraging3, PriorWeights) {
403 auto lmParams = LevenbergMarquardtParams::CeresDefaults();
404 ShonanAveraging3::Parameters params(lmParams);
405 EXPECT_DOUBLES_EQUAL(0, params.alpha, 1e-9);
406 EXPECT_DOUBLES_EQUAL(1, params.beta, 1e-9);
407 EXPECT_DOUBLES_EQUAL(0, params.gamma, 1e-9);
408 double alpha = 100.0, beta = 200.0, gamma = 300.0;
409 params.setAnchorWeight(alpha);
410 params.setKarcherWeight(beta);
411 params.setGaugesWeight(gamma);
412 EXPECT_DOUBLES_EQUAL(alpha, params.alpha, 1e-9);
413 EXPECT_DOUBLES_EQUAL(beta, params.beta, 1e-9);
414 EXPECT_DOUBLES_EQUAL(gamma, params.gamma, 1e-9);
415 params.setKarcherWeight(0);
416 static const ShonanAveraging3 shonan = fromExampleName(name: "Klaus3.g2o", parameters: params);
417 for (auto i : {0,1,2}) {
418 const auto& m = shonan.measurement(k: i);
419 auto isotropic =
420 std::static_pointer_cast<noiseModel::Isotropic>(r: m.noiseModel());
421 CHECK(isotropic != nullptr);
422 EXPECT_LONGS_EQUAL(3, isotropic->dim());
423 EXPECT_DOUBLES_EQUAL(0.2, isotropic->sigma(), 1e-9);
424 }
425 auto I = genericValue(v: Rot3());
426 Values initial{{0, I}, {1, I}, {2, I}};
427 EXPECT_DOUBLES_EQUAL(3.0756, shonan.cost(initial), 1e-4);
428 auto result = shonan.run(initialEstimate: initial, pMin: 3, pMax: 3);
429 EXPECT_DOUBLES_EQUAL(0.0015, shonan.cost(result.first), 1e-4);
430}
