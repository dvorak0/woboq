60TEST(ShonanFactor3, evaluateError) {
61 auto model = noiseModel::Isotropic::Sigma(dim: 3, sigma: 1.2);
62 for (const size_t p : {5, 4, 3}) {
63 Matrix M = Matrix::Identity(rows: p, cols: p);
64 M.topLeftCorner(cRows: 3, cCols: 3) = submanifold::R1.matrix();
65 SOn Q1(M);
66 M.topLeftCorner(cRows: 3, cCols: 3) = submanifold::R2.matrix();
67 SOn Q2(M);
68 auto factor = ShonanFactor3(1, 2, Rot3(::so3::R12.matrix()), p, model);
69 Matrix H1, H2;
70 factor.evaluateError(x: Q1, x: Q2, H&: H1, H&: H2);
71
72 // Test derivatives
73 Values values;
74 values.insert(j: 1, val: Q1);
75 values.insert(j: 2, val: Q2);
76 EXPECT_CORRECT_FACTOR_JACOBIANS(factor, values, 1e-7, 1e-5);
77 }
78}

81TEST(ShonanFactor3, equivalenceToSO3) {
82 using namespace ::submanifold;
83 auto R12 = ::so3::R12.retract(v: Vector3(0.1, 0.2, -0.1));
84 auto model = noiseModel::Isotropic::Sigma(dim: 3, sigma: 1.2);
85 auto factor3 = FrobeniusBetweenFactor<SO3>(1, 2, R12, model);
86 auto factor4 = ShonanFactor3(1, 2, Rot3(R12.matrix()), 4, model);
87 const Matrix3 E3(factor3.evaluateError(x: R1, x: R2).data());
88 const Matrix43 E4(
89 factor4.evaluateError(x: SOn(Q1.matrix()), x: SOn(Q2.matrix())).data());
90 EXPECT(assert_equal((Matrix)E4.topLeftCorner<3, 3>(), E3, 1e-9));
91 EXPECT(assert_equal((Matrix)E4.row(3), Matrix13::Zero(), 1e-9));
92}
