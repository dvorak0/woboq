32typedef Eigen::MatrixX2i CorrespondenceIndices; // N x 2 array
33
34// Output of detections in an image.
35// Coordinate system convention:
36// 1. The x coordinate denotes the horizontal direction (+ve direction towards
37// the right).
38// 2. The y coordinate denotes the vertical direction (+ve direction downwards).
39// 3. Origin is at the top left corner of the image.
40struct Keypoints {
41 // The (x, y) coordinates of the features, of shape Nx2.
42 Eigen::MatrixX2d coordinates;
43
44 // Optional scale of the detections, of shape N.
45 // Note: gtsam::Vector is typedef'd for Eigen::VectorXd.
46 std::optional<gtsam::Vector> scales;
47
48 /// Optional confidences/responses for each detection, of shape N.
49 std::optional<gtsam::Vector> responses;
50
51 Keypoints(const Eigen::MatrixX2d &_coordinates)
52 : coordinates(_coordinates) {}
53};
54
55using KeypointsVector = std::vector<Keypoints>;
56// Mapping from each image pair to (N,2) array representing indices of matching
57// keypoints.
58using MatchIndicesMap = std::map<IndexPair, CorrespondenceIndices>;
59
60/**
61 * @brief Creates a list of tracks from 2d point correspondences.
62 *
63 * Creates a disjoint-set forest (DSF) and 2d tracks from pairwise matches.
64 * We create a singleton for union-find set elements from camera index of a
65 * detection and the index of that detection in that camera's keypoint list,
66 * i.e. (i,k).
67 *
68 * @param Map from (i1,i2) image pair indices to (K,2) matrix, for K
69 * correspondence indices, from each image.
70 * @param Length-N list of keypoints, for N images/cameras.
71 */
72std::vector<SfmTrack2d> GTSAM_EXPORT tracksFromPairwiseMatches(
73 const MatchIndicesMap& matches, const KeypointsVector& keypoints,
74 bool verbose = false);
75
