28namespace gtsam {

30/**
31 * ShonanFactor is a BetweenFactor that moves in SO(p), but will
32 * land on the SO(d) sub-manifold of SO(p) at the global minimum. It projects
33 * the SO(p) matrices down to a Stiefel manifold of p*d matrices.
34 */
35template <size_t d>
36class GTSAM_EXPORT ShonanFactor : public NoiseModelFactorN<SOn, SOn> {
37 Matrix M_; ///< measured rotation between R1 and R2
38 size_t p_, pp_; ///< dimensionality constants
39 std::shared_ptr<Matrix> G_; ///< matrix of vectorized generators
40
41 // Select Rot2 or Rot3 interface based template parameter d
42 using Rot = typename std::conditional<d == 2, Rot2, Rot3>::type;
43
44public:
45
46 // Provide access to the Matrix& version of evaluateError:
47 using NoiseModelFactor2<SOn, SOn>::evaluateError;
48
49 /// @name Constructor
50 /// @{
51
52 /// Constructor. Note we convert to d*p-dimensional noise model.
53 /// To save memory and mallocs, pass in the vectorized Lie algebra generators:
54 /// G = std::make_shared<Matrix>(SOn::VectorizedGenerators(p));
55 ShonanFactor(Key j1, Key j2, const Rot &R12, size_t p,
56 const SharedNoiseModel &model = nullptr,
57 const std::shared_ptr<Matrix> &G = nullptr);
58
59 /// @}
60 /// @name Testable
61 /// @{
62
63 /// print with optional string
64 void
65 print(const std::string &s,
66 const KeyFormatter &keyFormatter = DefaultKeyFormatter) const override;
67
68 /// assert equality up to a tolerance
69 bool equals(const NonlinearFactor &expected,
70 double tol = 1e-9) const override;
71
72 /// @}
73 /// @name NoiseModelFactorN methods
74 /// @{
75
76 /// Error is Frobenius norm between Q1*P*R12 and Q2*P, where P=[I_3x3;0]
77 /// projects down from SO(p) to the Stiefel manifold of px3 matrices.
78 Vector evaluateError(const SOn& Q1, const SOn& Q2, OptionalMatrixType H1, OptionalMatrixType H2) const override;
79 /// @}
80
81private:
82 /// Calculate Jacobians if asked, Only implemented for d=2 and 3 in .cpp
83 void fillJacobians(const Matrix &M1, const Matrix &M2,
84 OptionalMatrixType H1,
85 OptionalMatrixType H2) const;
86};
87
88// Explicit instantiation for d=2 and d=3 in .cpp file:
89using ShonanFactor2 = ShonanFactor<2>;
90using ShonanFactor3 = ShonanFactor<3>;
91
92} // namespace gtsam
