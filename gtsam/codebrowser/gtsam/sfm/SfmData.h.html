19#pragma once

21#include <gtsam/geometry/Cal3Bundler.h>
22#include <gtsam/geometry/PinholeCamera.h>
23#include <gtsam/nonlinear/NonlinearFactorGraph.h>
24#include <gtsam/nonlinear/Values.h>
25#include <gtsam/sfm/SfmTrack.h>
26
27#include <string>
28#include <vector>
29
30namespace gtsam {
31
32/// Define the structure for the camera poses
33typedef PinholeCamera<Cal3Bundler> SfmCamera;
34
35/**
36 * @brief SfmData stores a bunch of SfmTracks
37 * @ingroup sfm
38 */
39struct GTSAM_EXPORT SfmData {
40 std::vector<SfmCamera> cameras; ///< Set of cameras
41
42 std::vector<SfmTrack> tracks; ///< Sparse set of points
43
44 /// @name Create from file
45 /// @{
46
47 /**
48 * @brief Parses a bundler output file and return result as SfmData instance.
49 * @param filename The name of the bundler file
50 * @param data SfM structure where the data is stored
51 * @return true if the parsing was successful, false otherwise
52 */
53 static SfmData FromBundlerFile(const std::string& filename);
54
55 /**
56 * @brief Parse a "Bundle Adjustment in the Large" (BAL) file and return
57 * result as SfmData instance.
58 * @param filename The name of the BAL file.
59 * @return SfM structure where the data is stored.
60 */
61 static SfmData FromBalFile(const std::string& filename);
62
63 /// @}
64 /// @name Standard Interface
65 /// @{
66
67 /// Add a track to SfmData
68 void addTrack(const SfmTrack& t) { tracks.push_back(x: t); }
69
70 /// Add a camera to SfmData
71 void addCamera(const SfmCamera& cam) { cameras.push_back(x: cam); }
72
73 /// The number of reconstructed 3D points
74 size_t numberTracks() const { return tracks.size(); }
75
76 /// The number of cameras
77 size_t numberCameras() const { return cameras.size(); }
78
79 /// The track formed by series of landmark measurements
80 const SfmTrack& track(size_t idx) const { return tracks[idx]; }
81
82 /// The camera pose at frame index `idx`
83 const SfmCamera& camera(size_t idx) const { return cameras[idx]; }
84
85 /// Getters
86 const std::vector<SfmCamera>& cameraList() const { return cameras; }
87 const std::vector<SfmTrack>& trackList() const { return tracks; }
88
89 /**
90 * @brief Create projection factors using keys i and P(j)
91 *
92 * @param model a noise model for projection errors
93 * @return NonlinearFactorGraph
94 */
95 NonlinearFactorGraph generalSfmFactors(
96 const SharedNoiseModel& model = noiseModel::Isotropic::Sigma(dim: 2,
97 sigma: 1.0)) const;
98
99 /**
100 * @brief Create factor graph with projection factors and gauge fix.
101 *
102 * Note: pose keys are simply integer indices, points use Symbol('p', j).
103 *
104 * @param model a noise model for projection errors
105 * @param fixedCamera which camera to fix, if any (use std::nullopt if none)
106 * @param fixedPoint which point to fix, if any (use std::nullopt if none)
107 * @return NonlinearFactorGraph
108 */
109 NonlinearFactorGraph sfmFactorGraph(
110 const SharedNoiseModel& model = noiseModel::Isotropic::Sigma(dim: 2, sigma: 1.0),
111 std::optional<size_t> fixedCamera = 0,
112 std::optional<size_t> fixedPoint = 0) const;
113
114 /// @}
115 /// @name Testable
116 /// @{
117
118 /// print
119 void print(const std::string& s = "") const;
120
121 /// assert equality up to a tolerance
122 bool equals(const SfmData& sfmData, double tol = 1e-9) const;
123
124 /// @}
125 /// @name Serialization
126 /// @{
127
128#if GTSAM_ENABLE_BOOST_SERIALIZATION
129 friend class boost::serialization::access;
130 /** Serialization function */
131 template <class Archive>
132 void serialize(Archive& ar, const unsigned int /*version*/) {
133 ar& BOOST_SERIALIZATION_NVP(cameras);
134 ar& BOOST_SERIALIZATION_NVP(tracks);
135 }
136#endif
137
138
139 /// @}
140};
141
142/// traits
143template <>
144struct traits<SfmData> : public Testable<SfmData> {};
145
146/**
147 * @brief This function parses a "Bundle Adjustment in the Large" (BAL) file and
148 * returns the data as a SfmData structure. Mainly used by wrapped code.
149 * @param filename The name of the BAL file.
150 * @return SfM structure where the data is stored.
151 */
152GTSAM_EXPORT SfmData readBal(const std::string& filename);
153
154/**
155 * @brief This function writes a "Bundle Adjustment in the Large" (BAL) file
156 * from a SfmData structure
157 * @param filename The name of the BAL file to write
158 * @param data SfM structure where the data is stored
159 * @return true if the parsing was successful, false otherwise
160 */
161GTSAM_EXPORT bool writeBAL(const std::string& filename, const SfmData& data);
162
163/**
164 * @brief This function writes a "Bundle Adjustment in the Large" (BAL) file
165 * from a SfmData structure and a value structure (measurements are the same as
166 * the SfM input data, while camera poses and values are read from Values)
167 * @param filename The name of the BAL file to write
168 * @param data SfM structure where the data is stored
169 * @param values structure where the graph values are stored (values can be
170 * either Pose3 or PinholeCamera<Cal3Bundler> for the cameras, and should be
171 * Point3 for the 3D points). Note: assumes that the keys are "i" for pose i
172 * and "Symbol::('p',j)" for landmark j.
173 * @return true if the parsing was successful, false otherwise
174 */
175GTSAM_EXPORT bool writeBALfromValues(const std::string& filename,
176 const SfmData& data, const Values& values);
177
178/**
179 * @brief This function converts an openGL camera pose to an GTSAM camera pose
180 * @param R rotation in openGL
181 * @param tx x component of the translation in openGL
182 * @param ty y component of the translation in openGL
183 * @param tz z component of the translation in openGL
184 * @return Pose3 in GTSAM format
185 */
186GTSAM_EXPORT Pose3 openGL2gtsam(const Rot3& R, double tx, double ty, double tz);
187
188/**
189 * @brief This function converts a GTSAM camera pose to an openGL camera pose
190 * @param R rotation in GTSAM
191 * @param tx x component of the translation in GTSAM
192 * @param ty y component of the translation in GTSAM
193 * @param tz z component of the translation in GTSAM
194 * @return Pose3 in openGL format
195 */
196GTSAM_EXPORT Pose3 gtsam2openGL(const Rot3& R, double tx, double ty, double tz);
197
198/**
199 * @brief This function converts a GTSAM camera pose to an openGL camera pose
200 * @param PoseGTSAM pose in GTSAM format
201 * @return Pose3 in openGL format
202 */
203GTSAM_EXPORT Pose3 gtsam2openGL(const Pose3& PoseGTSAM);
204
205/**
206 * @brief This function creates initial values for cameras from db.
207 *
208 * No symbol is used, so camera keys are simply integer indices.
209 *
210 * @param SfmData
211 * @return Values
212 */
213GTSAM_EXPORT Values initialCamerasEstimate(const SfmData& db);
214
215/**
216 * @brief This function creates initial values for cameras and points from db
217 *
218 * Note: Pose keys are simply integer indices, points use Symbol('p', j).
219 *
220 * @param SfmData
221 * @return Values
222 */
223GTSAM_EXPORT Values initialCamerasAndPointsEstimate(const SfmData& db);
