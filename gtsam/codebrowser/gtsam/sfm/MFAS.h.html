31/**
32 The MFAS class to solve a Minimum feedback arc set (MFAS)
33 problem. We implement the solution from:
34 Kyle Wilson and Noah Snavely, "Robust Global Translations with 1DSfM",
35 Proceedings of the European Conference on Computer Vision, ECCV 2014
36
37 Given a weighted directed graph, the objective in a Minimum feedback arc set
38 problem is to obtain a directed acyclic graph by removing
39 edges such that the total weight of removed edges is minimum.
40
41 Although MFAS is a general graph problem and can be applied in many areas,
42 this classed was designed for the purpose of outlier rejection in a
43 translation averaging for SfM setting. For more details, refer to the above
44 paper. The nodes of the graph in this context represents cameras in 3D and the
45 edges between them represent unit translations in the world coordinate frame,
46 i.e w_aZb is the unit translation from a to b expressed in the world
47 coordinate frame. The weights for the edges are obtained by projecting the
48 unit translations in a projection direction.
49 @ingroup sfm
50*/
51class GTSAM_EXPORT MFAS {
52 public:
53 // used to represent edges between two nodes in the graph. When used in
54 // translation averaging for global SfM
55 using KeyPair = std::pair<Key, Key>;
56 using TranslationEdges = std::vector<BinaryMeasurement<Unit3>>;
57
58 private:
59 // edges with a direction such that all weights are positive
60 // i.e, edges that originally had negative weights are flipped
61 std::map<KeyPair, double> edgeWeights_;
62
63 public:
64 /**
65 * @brief Construct from the weighted directed edges
66 * between the nodes. Each node is identified by a Key.
67 * @param edgeWeights: weights of edges in the graph
68 */
69 MFAS(const std::map<KeyPair, double> &edgeWeights)
70 : edgeWeights_(edgeWeights) {}
71
72 /**
73 * @brief Constructor to be used in the context of translation averaging.
74 * Here, the nodes of the graph are cameras in 3D and the edges have a unit
75 * translation direction between them. The weights of the edges is computed by
76 * projecting them along a projection direction.
77 * @param relativeTranslations translation directions between the cameras
78 * @param projectionDirection direction in which edges are to be projected
79 */
80 MFAS(const TranslationEdges &relativeTranslations,
81 const Unit3 &projectionDirection);
82
83 /**
84 * @brief Computes the 1D MFAS ordering of nodes in the graph
85 * @return orderedNodes: vector of nodes in the obtained order
86 */
87 KeyVector computeOrdering() const;
88
89 /**
90 * @brief Computes the outlier weights of the graph. We define the outlier
91 * weight of a edge to be zero if the edge is an inlier and the magnitude of
92 * its edgeWeight if it is an outlier. This function internally calls
93 * computeOrdering and uses the obtained ordering to identify outlier edges.
94 * @return outlierWeights: map from an edge to its outlier weight.
95 */
96 std::map<KeyPair, double> computeOutlierWeights() const;
97};
98
99typedef std::map<std::pair<Key, Key>, double> KeyPairDoubleMap;
