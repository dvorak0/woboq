31// Set up an optimization problem for the unknown translations Ti in the world
32// coordinate frame, given the known camera attitudes wRi with respect to the
33// world frame, and a set of (noisy) translation directions of type Unit3,
34// w_aZb. The measurement equation is
35// w_aZb = Unit3(Tb - Ta) (1)
36// i.e., w_aZb is the translation direction from frame A to B, in world
37// coordinates. Although Unit3 instances live on a manifold, following
38// Wilson14eccv_1DSfM.pdf error we compute the *chordal distance* in the
39// ambient world coordinate frame.
40//
41// It is clear that we cannot recover the scale, nor the absolute position,
42// so the gauge freedom in this case is 3 + 1 = 4. We fix these by taking fixing
43// the translations Ta and Tb associated with the first measurement w_aZb,
44// clamping them to their initial values as given to this method. If no initial
45// values are given, we use the origin for Tb and set Tb to make (1) come
46// through, i.e.,
47// Tb = s * wRa * Point3(w_aZb) (2)
48// where s is an arbitrary scale that can be supplied, default 1.0. Hence, two
49// versions are supplied below corresponding to whether we have initial values
50// or not.
51class GTSAM_EXPORT TranslationRecovery {
52 public:
53 using KeyPair = std::pair<Key, Key>;
54 using TranslationEdges = std::vector<BinaryMeasurement<Unit3>>;
55
56 private:
57 // Translation directions between camera pairs.
58 TranslationEdges relativeTranslations_;
59
60 // Parameters.
61 LevenbergMarquardtParams lmParams_;
62
63 const bool use_bilinear_translation_factor_ = false;
64
65 public:
66 /**
67 * @brief Construct a new Translation Recovery object
68 *
69 * @param lmParams parameters for optimization.
70 */
71 TranslationRecovery(const LevenbergMarquardtParams &lmParams,
72 bool use_bilinear_translation_factor = false)
73 : lmParams_(lmParams),
74 use_bilinear_translation_factor_(use_bilinear_translation_factor) {}
75
76 /**
77 * @brief Default constructor.
78 */
79 TranslationRecovery() = default;
80
81 /**
82 * @brief Build the factor graph to do the optimization.
83 *
84 * @param relativeTranslations unit translation directions between
85 * translations to be estimated
86 * @return NonlinearFactorGraph
87 */
88 NonlinearFactorGraph buildGraph(
89 const std::vector<BinaryMeasurement<Unit3>> &relativeTranslations) const;
90
91 /**
92 * @brief Add 3 factors to the graph:
93 * - A prior on the first point to lie at (0, 0, 0)
94 * - If betweenTranslations is non-empty, between factors provided by it.
95 * - If betweenTranslations is empty, a prior on scale of the first
96 * relativeTranslations edge.
97 *
98 * @param relativeTranslations unit translation directions between
99 * translations to be estimated
100 * @param scale scale for first relative translation which fixes gauge.
101 * @param graph factor graph to which prior is added.
102 * @param betweenTranslations relative translations (with scale) between 2
103 * points in world coordinate frame known a priori.
104 * @param priorNoiseModel the noise model to use with the prior.
105 */
106 void addPrior(
107 const std::vector<BinaryMeasurement<Unit3>> &relativeTranslations,
108 const double scale,
109 const std::vector<BinaryMeasurement<Point3>> &betweenTranslations,
110 NonlinearFactorGraph *graph,
111 const SharedNoiseModel &priorNoiseModel =
112 noiseModel::Isotropic::Sigma(dim: dim: 3, sigma: sigma: 0.01)) const;
113
114 /**
115 * @brief Create random initial translations.
116 *
117 * @param relativeTranslations unit translation directions between
118 * translations to be estimated
119 * @param betweenTranslations relative translations (with scale) between 2
120 * points in world coordinate frame known a priori.
121 * @param rng random number generator
122 * @param initialValues (optional) initial values from a prior
123 * @return Values
124 */
125 Values initializeRandomly(
126 const std::vector<BinaryMeasurement<Unit3>> &relativeTranslations,
127 const std::vector<BinaryMeasurement<Point3>> &betweenTranslations,
128 std::mt19937 *rng, const Values &initialValues = Values()) const;
129
130 /**
131 * @brief Version of initializeRandomly with a fixed seed.
132 *
133 * @param relativeTranslations unit translation directions between
134 * translations to be estimated
135 * @param betweenTranslations relative translations (with scale) between 2
136 * points in world coordinate frame known a priori.
137 * @param initialValues (optional) initial values from a prior
138 * @return Values
139 */
140 Values initializeRandomly(
141 const std::vector<BinaryMeasurement<Unit3>> &relativeTranslations,
142 const std::vector<BinaryMeasurement<Point3>> &betweenTranslations,
143 const Values &initialValues = Values()) const;
144
145 /**
146 * @brief Build and optimize factor graph.
147 *
148 * @param relativeTranslations the relative translations, in world coordinate
149 * frames, vector of BinaryMeasurements of Unit3, where each key of a
150 * measurement is a point in 3D. If a relative translation magnitude is zero,
151 * it is treated as a hard same-point constraint (the result of all nodes
152 * connected by a zero-magnitude edge will be the same).
153 * @param scale scale for first relative translation which fixes gauge.
154 * The scale is only used if betweenTranslations is empty.
155 * @param betweenTranslations relative translations (with scale) between 2
156 * points in world coordinate frame known a priori. Unlike
157 * relativeTranslations, zero-magnitude betweenTranslations are not treated as
158 * hard constraints.
159 * @param initialValues initial values for optimization. Initializes randomly
160 * if not provided.
161 * @return Values
162 */
163 Values run(
164 const TranslationEdges &relativeTranslations, const double scale = 1.0,
165 const std::vector<BinaryMeasurement<Point3>> &betweenTranslations = {},
166 const Values &initialValues = Values()) const;
167
168 /**
169 * @brief Simulate translation direction measurements
170 *
171 * @param poses SE(3) ground truth poses stored as Values
172 * @param edges pairs (a,b) for which a measurement w_aZb will be generated.
173 * @return TranslationEdges vector of binary measurements where the keys are
174 * the cameras and the measurement is the simulated Unit3 translation
175 * direction between the cameras.
176 */
177 static TranslationEdges SimulateMeasurements(
178 const Values &poses, const std::vector<KeyPair> &edges);
179};
180} // namespace gtsam
