25namespace gtsam {
26/**
27 * The ShonanGaugeFactor creates a constraint on a single SO(n) to avoid moving
28 * in the stabilizer.
29 *
30 * Details: SO(p) contains the p*3 Stiefel matrices of orthogonal frames: we
31 * take those to be the 3 columns on the left.
32 * The P*P skew-symmetric matrices associated with so(p) can be partitioned as
33 * (Appendix B in the ECCV paper):
34 * | [w] -K' |
35 * | K [g] |
36 * where w is the SO(3) space, K are the extra Stiefel diemnsions (wormhole!)
37 * and (g)amma are extra dimensions in SO(p) that do not modify the cost
38 * function. The latter corresponds to rotations SO(p-d), and so the first few
39 * values of p-d for d==3 with their corresponding dimensionality are {0:0, 1:0,
40 * 2:1, 3:3, ...}
41 *
42 * The ShonanGaugeFactor adds a unit Jacobian to these extra dimensions,
43 * essentially restricting the optimization to the Stiefel manifold.
44 */
45class GTSAM_EXPORT ShonanGaugeFactor : public NonlinearFactor {
46 // Row dimension, equal to the dimensionality of SO(p-d)
47 size_t rows_;
48
49 /// Constant Jacobian
50 std::shared_ptr<JacobianFactor> whitenedJacobian_;
51
52public:
53 /**
54 * Construct from key for an SO(p) matrix, for base dimension d (2 or 3)
55 * If parameter gamma is given, it acts as a precision = 1/sigma^2, and
56 * the Jacobian will be multiplied with 1/sigma = sqrt(gamma).
57 */
58 ShonanGaugeFactor(Key key, size_t p, size_t d = 3,
59 std::optional<double> gamma = {})
60 : NonlinearFactor(KeyVector{key}) {
61 if (p < d) {
62 throw std::invalid_argument("ShonanGaugeFactor must have p>=d.");
63 }
64 // Calculate dimensions
65 size_t q = p - d;
66 size_t P = SOn::Dimension(n: p); // dimensionality of SO(p)
67 rows_ = SOn::Dimension(n: q); // dimensionality of SO(q), the gauge
68
69 // Create constant Jacobian as a rows_*P matrix: there are rows_ penalized
70 // dimensions, but it is a bit tricky to find them among the P columns.
71 // The key is to look at how skew-symmetric matrices are laid out in SOn.h:
72 // the first tangent dimension will always be included, but beyond that we
73 // have to be careful. We always need to skip the d top-rows of the skew-
74 // symmetric matrix as they below to K, part of the Stiefel manifold.
75 Matrix A(rows_, P);
76 A.setZero();
77 double invSigma = gamma ? std::sqrt(x: *gamma) : 1.0;
78 size_t i = 0, j = 0, n = p - 1 - d;
79 while (i < rows_) {
80 A.block(startRow: i, startCol: j, blockRows: n, blockCols: n) = invSigma * Matrix::Identity(rows: n, cols: n);
81 i += n;
82 j += n + d; // skip d columns
83 n -= 1;
84 }
85 // TODO(frank): assign the right one in the right columns
86 whitenedJacobian_ =
87 std::make_shared<JacobianFactor>(args&: key, args&: A, args: Vector::Zero(size: rows_));
88 }
89
90 /// Destructor
91 ~ShonanGaugeFactor() override {}
92
93 /// Calculate the error of the factor: always zero
94 double error(const Values &c) const override { return 0; }
95
96 /// get the dimension of the factor (number of rows on linearization)
97 size_t dim() const override { return rows_; }
98
99 /// linearize to a GaussianFactor
100 std::shared_ptr<GaussianFactor> linearize(const Values &c) const override {
101 return whitenedJacobian_;
102 }
103};
