107/**
108 * Class that implements Shonan Averaging from our ECCV'20 paper.
109 * Note: The "basic" API uses all Rot values (Rot2 or Rot3, depending on value
110 * of d), whereas the different levels and "advanced" API at SO(p) needs Values
111 * of type SOn<Dynamic>.
112 *
113 * The template parameter d can be 2 or 3.
114 * Both are specialized in the .cpp file.
115 *
116 * If you use this code in your work, please consider citing our paper:
117 * Shonan Rotation Averaging, Global Optimality by Surfing SO(p)^n
118 * Frank Dellaert, David M. Rosen, Jing Wu, Robert Mahony, and Luca Carlone,
119 * European Computer Vision Conference, 2020.
120 * You can view our ECCV spotlight video at https://youtu.be/5ppaqMyHtE0
121 */
122template <size_t d>
123class GTSAM_EXPORT ShonanAveraging {
124 public:
125 using Sparse = Eigen::SparseMatrix<double>;
126
127 // Define the Parameters type and use its typedef of the rotation type:
128 using Parameters = ShonanAveragingParameters<d>;
129 using Rot = typename Parameters::Rot;
130
131 // We store SO(d) BetweenFactors to get noise model
132 using Measurements = std::vector<BinaryMeasurement<Rot>>;
133
134 private:
135 Parameters parameters_;
136 Measurements measurements_;
137 size_t nrUnknowns_;
138 Sparse D_; // Sparse (diagonal) degree matrix
139 Sparse Q_; // Sparse measurement matrix, == \tilde{R} in Eriksson18cvpr
140 Sparse L_; // connection Laplacian L = D - Q, needed for optimality check
141
142 /**
143 * Build 3Nx3N sparse matrix consisting of rotation measurements, arranged as
144 * (i,j) and (j,i) blocks within a sparse matrix.
145 */
146 Sparse buildQ() const;
147
148 /// Build 3Nx3N sparse degree matrix D
149 Sparse buildD() const;
150
151 public:
152 /// @name Standard Constructors
153 /// @{
154
155 /// Construct from set of relative measurements (given as BetweenFactor<Rot3>
156 /// for now) NoiseModel *must* be isotropic.
157 ShonanAveraging(const Measurements &measurements,
158 const Parameters &parameters = Parameters());
159
160 /// @}
161 /// @name Query properties
162 /// @{
163
164 /// Return number of unknowns
165 size_t nrUnknowns() const { return nrUnknowns_; }
166
167 /// Return number of measurements
168 size_t numberMeasurements() const { return measurements_.size(); }
169
170 /// k^th binary measurement
171 const BinaryMeasurement<Rot> &measurement(size_t k) const {
172 return measurements_[k];
173 }
174
175 /**
176 * Update factors to use robust Huber loss.
177 *
178 * @param measurements Vector of BinaryMeasurements.
179 * @param k Huber noise model threshold.
180 */
181 Measurements makeNoiseModelRobust(const Measurements &measurements,
182 double k = 1.345) const {
183 Measurements robustMeasurements;
184 for (auto &measurement : measurements) {
185 auto model = measurement.noiseModel();
186 const auto &robust =
187 std::dynamic_pointer_cast<noiseModel::Robust>(model);
188
189 SharedNoiseModel robust_model;
190 // Check if the noise model is already robust
191 if (robust) {
192 robust_model = model;
193 } else {
194 // make robust
195 robust_model = noiseModel::Robust::Create(
196 robust: noiseModel::mEstimator::Huber::Create(k), noise: model);
197 }
198 BinaryMeasurement<Rot> meas(measurement.key1(), measurement.key2(),
199 measurement.measured(), robust_model);
200 robustMeasurements.push_back(meas);
201 }
202 return robustMeasurements;
203 }
204
205 /// k^th measurement, as a Rot.
206 const Rot &measured(size_t k) const { return measurements_[k].measured(); }
207
208 /// Keys for k^th measurement, as a vector of Key values.
209 const KeyVector &keys(size_t k) const { return measurements_[k].keys(); }
210
211 /// @}
212 /// @name Matrix API (advanced use, debugging)
213 /// @{
214
215 Sparse D() const { return D_; } ///< Sparse version of D
216 Matrix denseD() const { return Matrix(D_); } ///< Dense version of D
217 Sparse Q() const { return Q_; } ///< Sparse version of Q
218 Matrix denseQ() const { return Matrix(Q_); } ///< Dense version of Q
219 Sparse L() const { return L_; } ///< Sparse version of L
220 Matrix denseL() const { return Matrix(L_); } ///< Dense version of L
221
222 /// Version that takes pxdN Stiefel manifold elements
223 Sparse computeLambda(const Matrix &S) const;
224
225 /// Dense versions of computeLambda for wrapper/testing
226 Matrix computeLambda_(const Values &values) const {
227 return Matrix(computeLambda(values));
228 }
229
230 /// Dense versions of computeLambda for wrapper/testing
231 Matrix computeLambda_(const Matrix &S) const {
232 return Matrix(computeLambda(S));
233 }
234
235 /// Compute A matrix whose Eigenvalues we will examine
236 Sparse computeA(const Values &values) const;
237
238 /// Version that takes pxdN Stiefel manifold elements
239 Sparse computeA(const Matrix &S) const;
240
241 /// Dense version of computeA for wrapper/testing
242 Matrix computeA_(const Values &values) const {
243 return Matrix(computeA(values));
244 }
245
246 /// Project to pxdN Stiefel manifold
247 static Matrix StiefelElementMatrix(const Values &values);
248
249 /**
250 * Compute minimum eigenvalue for optimality check.
251 * @param values: should be of type SOn
252 */
253 double computeMinEigenValue(const Values &values,
254 Vector *minEigenVector = nullptr) const;
255
256 /**
257 * Compute minimum eigenvalue with accelerated power method.
258 * @param values: should be of type SOn
259 */
260 double computeMinEigenValueAP(const Values &values,
261 Vector *minEigenVector = nullptr) const;
262
263 /// Project pxdN Stiefel manifold matrix S to Rot3^N
264 Values roundSolutionS(const Matrix &S) const;
265
266 /// Create a VectorValues with eigenvector v_i
267 static VectorValues TangentVectorValues(size_t p, const Vector &v);
268
269 /// Calculate the riemannian gradient of F(values) at values
270 Matrix riemannianGradient(size_t p, const Values &values) const;
271
272 /**
273 * Lift up the dimension of values in type SO(p-1) with descent direction
274 * provided by minEigenVector and return new values in type SO(p)
275 */
276 static Values LiftwithDescent(size_t p, const Values &values,
277 const Vector &minEigenVector);
278
279 /**
280 * Given some values at p-1, return new values at p, by doing a line search
281 * along the descent direction, computed from the minimum eigenvector at p-1.
282 * @param values should be of type SO(p-1)
283 * @param minEigenVector corresponding to minEigenValue at level p-1
284 * @return values of type SO(p)
285 */
286 Values initializeWithDescent(
287 size_t p, const Values &values, const Vector &minEigenVector,
288 double minEigenValue, double gradienTolerance = 1e-2,
289 double preconditionedGradNormTolerance = 1e-4) const;
290 /// @}
291 /// @name Advanced API
292 /// @{
293
294 /**
295 * Build graph for SO(p)
296 * @param p the dimensionality of the rotation manifold to optimize over
297 */
298 NonlinearFactorGraph buildGraphAt(size_t p) const;
299
300 /**
301 * Create initial Values of type SO(p)
302 * @param p the dimensionality of the rotation manifold
303 * @param rng random number generator
304 */
305 Values initializeRandomlyAt(size_t p, std::mt19937 &rng) const;
306
307 /// Version of initializeRandomlyAt with fixed random seed.
308 Values initializeRandomlyAt(size_t p) const;
309
310 /**
311 * Calculate cost for SO(p)
312 * Values should be of type SO(p)
313 */
314 double costAt(size_t p, const Values &values) const;
315
316 /**
317 * Given an estimated local minimum Yopt for the (possibly lifted)
318 * relaxation, this function computes and returns the block-diagonal elements
319 * of the corresponding Lagrange multiplier.
320 */
321 Sparse computeLambda(const Values &values) const;
322
323 /**
324 * Compute minimum eigenvalue for optimality check.
325 * @param values: should be of type SOn
326 * @return minEigenVector and minEigenValue
327 */
328 std::pair<double, Vector> computeMinEigenVector(const Values &values) const;
329
330 /**
331 * Check optimality
332 * @param values: should be of type SOn
333 */
334 bool checkOptimality(const Values &values) const;
335
336 /**
337 * Try to create optimizer at SO(p)
338 * @param p the dimensionality of the rotation manifold to optimize over
339 * @param initial initial SO(p) values
340 * @return lm optimizer
341 */
342 std::shared_ptr<LevenbergMarquardtOptimizer> createOptimizerAt(
343 size_t p, const Values &initial) const;
344
345 /**
346 * Try to optimize at SO(p)
347 * @param p the dimensionality of the rotation manifold to optimize over
348 * @param initial initial SO(p) values
349 * @return SO(p) values
350 */
351 Values tryOptimizingAt(size_t p, const Values &initial) const;
352
353 /**
354 * Project from SO(p) to Rot2 or Rot3
355 * Values should be of type SO(p)
356 */
357 Values projectFrom(size_t p, const Values &values) const;
358
359 /**
360 * Project from SO(p)^N to Rot2^N or Rot3^N
361 * Values should be of type SO(p)
362 */
363 Values roundSolution(const Values &values) const;
364
365 /// Lift Values of type T to SO(p)
366 template <class T>
367 static Values LiftTo(size_t p, const Values &values) {
368 Values result;
369 for (const auto& it : values.extract<T>()) {
370 result.insert(it.first, SOn::Lift(p, it.second.matrix()));
371 }
372 return result;
373 }
374
375 /// @}
376 /// @name Basic API
377 /// @{
378
379 /**
380 * Calculate cost for SO(3)
381 * Values should be of type Rot3
382 */
383 double cost(const Values &values) const;
384
385 /**
386 * Initialize randomly at SO(d)
387 * @param rng random number generator
388 * Example:
389 * std::mt19937 rng(42);
390 * Values initial = initializeRandomly(rng, p);
391 */
392 Values initializeRandomly(std::mt19937 &rng) const;
393
394 /// Random initialization for wrapper, fixed random seed.
395 Values initializeRandomly() const;
396
397 /**
398 * Optimize at different values of p until convergence.
399 * @param initial initial Rot3 values
400 * @param pMin value of p to start Riemanian staircase at (default: d).
401 * @param pMax maximum value of p to try (default: 10)
402 * @return (Rot3 values, minimum eigenvalue)
403 */
404 std::pair<Values, double> run(const Values &initialEstimate, size_t pMin = d,
405 size_t pMax = 10) const;
406 /// @}
407
408 /**
409 * Helper function to convert measurements to robust noise model
410 * if flag is set.
411 *
412 * @tparam T the type of measurement, e.g. Rot3.
413 * @param measurements vector of BinaryMeasurements of type T.
414 * @param useRobustModel flag indicating whether use robust noise model
415 * instead.
416 */
417 template <typename T>
418 inline std::vector<BinaryMeasurement<T>> maybeRobust(
419 const std::vector<BinaryMeasurement<T>> &measurements,
420 bool useRobustModel = false) const {
421 return useRobustModel ? makeNoiseModelRobust(measurements) : measurements;
422 }
423};
