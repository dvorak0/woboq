49/* ************************************************************************* */
50template <size_t d>
51ShonanAveragingParameters<d>::ShonanAveragingParameters(
52 const LevenbergMarquardtParams &_lm, const std::string &method,
53 double optimalityThreshold, double alpha, double beta, double gamma)
54 : lm(_lm),
55 optimalityThreshold(optimalityThreshold),
56 alpha(alpha),
57 beta(beta),
58 gamma(gamma),
59 useHuber(false),
60 certifyOptimality(true) {
61 // By default, we will do conjugate gradient
62 lm.linearSolverType = LevenbergMarquardtParams::Iterative;
63
64 // Create subgraph builder parameters
65 SubgraphBuilderParameters builderParameters;
66 builderParameters.skeletonType = SubgraphBuilderParameters::KRUSKAL;
67 builderParameters.skeletonWeight = SubgraphBuilderParameters::EQUAL;
68 builderParameters.augmentationWeight = SubgraphBuilderParameters::SKELETON;
69 builderParameters.augmentationFactor = 0.0;
70
71 // Choose optimization method
72 if (method == "SUBGRAPH") {
73 lm.iterativeParams =
74 std::make_shared<SubgraphSolverParameters>(args&: builderParameters);
75 } else if (method == "SGPC") {
76 lm.iterativeParams = std::make_shared<PCGSolverParameters>(
77 args: std::make_shared<SubgraphPreconditionerParameters>(args&: builderParameters));
78 } else if (method == "JACOBI") {
79 lm.iterativeParams = std::make_shared<PCGSolverParameters>(args: std::make_shared<BlockJacobiPreconditionerParameters>());
80 } else if (method == "QR") {
81 lm.setLinearSolverType("MULTIFRONTAL_QR");
82 } else if (method == "CHOLESKY") {
83 lm.setLinearSolverType("MULTIFRONTAL_CHOLESKY");
84 } else {
85 throw std::invalid_argument("ShonanAveragingParameters: unknown method");
86 }
87}

95// Calculate number of unknown rotations referenced by measurements
96// Throws exception of keys are not numbered as expected (may change in future).
97template <size_t d>
98static size_t NrUnknowns(
99 const typename ShonanAveraging<d>::Measurements &measurements) {
100 Key maxKey = 0;
101 std::set<Key> keys;
102 for (const auto &measurement : measurements) {
103 for (const Key &key : measurement.keys()) {
104 maxKey = std::max(a: key, b: maxKey);
105 keys.insert(x: key);
106 }
107 }
108 size_t N = keys.size();
109 if (maxKey != N - 1) {
110 throw std::invalid_argument(
111 "As of now, ShonanAveraging expects keys numbered 0..N-1");
112 }
113 return N;
114}

116/* ************************************************************************* */
117template <size_t d>
118ShonanAveraging<d>::ShonanAveraging(const Measurements &measurements,
119 const Parameters &parameters)
120 : parameters_(parameters),
121 measurements_(measurements),
122 nrUnknowns_(NrUnknowns<d>(measurements)) {
123 for (const auto &measurement : measurements_) {
124 const auto &model = measurement.noiseModel();
125 if (model && model->dim() != SO<d>::dimension) {
126 measurement.print("Factor with incorrect noise model:\n");
127 throw std::invalid_argument(
128 "ShonanAveraging: measurements passed to "
129 "constructor have incorrect dimension.");
130 }
131 }
132 Q_ = buildQ();
133 D_ = buildD();
134 L_ = D_ - Q_;
135}

137/* ************************************************************************* */
138template <size_t d>
139NonlinearFactorGraph ShonanAveraging<d>::buildGraphAt(size_t p) const {
140 NonlinearFactorGraph graph;
141 auto G = std::make_shared<Matrix>(args: SO<-1>::VectorizedGenerators(n: p));
142
143 for (const auto &measurement : measurements_) {
144 const auto &keys = measurement.keys();
145 const auto &Rij = measurement.measured();
146 const auto &model = measurement.noiseModel();
147 graph.emplace_shared<ShonanFactor<d>>(keys[0], keys[1], Rij, p, model, G);
148 }
149 // Possibly add Karcher prior
150 if (parameters_.beta > 0) {
151 const size_t dim = SOn::Dimension(n: p);
152 graph.emplace_shared<KarcherMeanFactor<SOn>>(args: graph.keys(), args: dim);
153 }
154 // Possibly add gauge factors - they are probably useless as gradient is zero
155 if (parameters_.gamma > 0 && p > d + 1) {
156 for (auto key : graph.keys())
157 graph.emplace_shared<ShonanGaugeFactor>(key, p, d, parameters_.gamma);
158 }
159 return graph;
160}

163template <size_t d>
164double ShonanAveraging<d>::costAt(size_t p, const Values &values) const {
165 const NonlinearFactorGraph graph = buildGraphAt(p);
166 return graph.error(values);

170template <size_t d>
171std::shared_ptr<LevenbergMarquardtOptimizer>
172ShonanAveraging<d>::createOptimizerAt(size_t p, const Values &initial) const {
173 // Build graph
174 NonlinearFactorGraph graph = buildGraphAt(p);
175
176 // Anchor prior is added here as depends on initial value (and cost is zero)
177 if (parameters_.alpha > 0) {
178 const size_t dim = SOn::Dimension(n: p);
179 const auto [i, value] = parameters_.anchor;
180 auto model = noiseModel::Isotropic::Precision(dim, precision: parameters_.alpha);
181 graph.emplace_shared<PriorFactor<SOn>>(i, SOn::Lift(p, value.matrix()),
182 model);
183 }
184 // Optimize
185 return std::make_shared<LevenbergMarquardtOptimizer>(graph, initial,
186 parameters_.lm);
187}

198// Project to pxdN Stiefel manifold
199template <size_t d>
200Matrix ShonanAveraging<d>::StiefelElementMatrix(const Values &values) {
201 const size_t N = values.size();
202 const size_t p = values.at<SOn>(j: 0).rows();
203 Matrix S(p, N * d);
204 for (const auto& it : values.extract<SOn>()) {
205 S.middleCols<d>(it.first * d) =
206 it.second.matrix().leftCols<d>(); // project Qj to Stiefel
207 }
208 return S;
209}

236/* ************************************************************************* */
237template <size_t d>
238static Matrix RoundSolutionS(const Matrix &S) {
239 const size_t N = S.cols() / d;
240 // First, compute a thin SVD of S
241 Eigen::JacobiSVD<Matrix> svd(S, Eigen::ComputeThinV);
242 const Vector sigmas = svd.singularValues();
243
244 // Construct a diagonal matrix comprised of the first d singular values
245 using DiagonalMatrix = Eigen::DiagonalMatrix<double, d>;
246 DiagonalMatrix Sigma_d;
247 Sigma_d.diagonal() = sigmas.head<d>();
248
249 // Now, construct a rank-d truncated singular value decomposition for S
250 Matrix R = Sigma_d * svd.matrixV().leftCols<d>().transpose();
251
252 // Count the number of blocks whose determinants have positive sign
253 size_t numPositiveBlocks = 0;
254 for (size_t i = 0; i < N; ++i) {
255 // Compute the determinant of the ith dxd block of R
256 double determinant = R.middleCols<d>(d * i).determinant();
257 if (determinant > 0) ++numPositiveBlocks;
258 }
259
260 if (numPositiveBlocks < N / 2) {
261 // Less than half of the total number of blocks have the correct sign.
262 // To reverse their orientations, multiply with a reflection matrix.
263 DiagonalMatrix reflector;
264 reflector.setIdentity();
265 reflector.diagonal()(d - 1) = -1;
266 R = reflector * R;
267 }
268
269 return R;
270}

272/* ************************************************************************* */
273template <>
274Values ShonanAveraging<2>::roundSolutionS(const Matrix &S) const {
275 // Round to a 2*2N matrix
276 Matrix R = RoundSolutionS<2>(S);
277
278 // Finally, project each dxd rotation block to SO(2)
279 Values values;
280 for (size_t j = 0; j < nrUnknowns(); ++j) {
281 const Rot2 Ri = Rot2::atan2(y: R(1, 2 * j), x: R(0, 2 * j));
282 values.insert(j, val: Ri);
283 }
284 return values;
285}

287template <>
288Values ShonanAveraging<3>::roundSolutionS(const Matrix &S) const {
289 // Round to a 3*3N matrix
290 Matrix R = RoundSolutionS<3>(S);
291
292 // Finally, project each dxd rotation block to SO(3)
293 Values values;
294 for (size_t j = 0; j < nrUnknowns(); ++j) {
295 const Rot3 Ri = Rot3::ClosestTo(M: R.middleCols<3>(startCol: 3 * j));
296 values.insert(j, val: Ri);
297 }
298 return values;
299}

302template <size_t d>
303Values ShonanAveraging<d>::roundSolution(const Values &values) const {
304 // Project to pxdN Stiefel manifold...
305 Matrix S = StiefelElementMatrix(values);
306 // ...and call version above.

310/* ************************************************************************* */
311template <size_t d>
312double ShonanAveraging<d>::cost(const Values &values) const {
313 NonlinearFactorGraph graph;
314 for (const auto &measurement : measurements_) {
315 const auto &keys = measurement.keys();
316 const auto &Rij = measurement.measured();
317 const auto &model = measurement.noiseModel();
318 graph.emplace_shared<FrobeniusBetweenFactor<SO<d>>>(
319 keys[0], keys[1], SO<d>(Rij.matrix()), model);
320 }
321 // Finally, project each dxd rotation block to SO(d)
322 Values result;
323 for (const auto& it : values.extract<Rot>()) {
324 result.insert(it.first, SO<d>(it.second.matrix()));
325 }
326 return graph.error(values: result);
327}

330// Get kappa from noise model
331template <typename T, size_t d>
332static double Kappa(const BinaryMeasurement<T> &measurement,
333 const ShonanAveragingParameters<d> &parameters) {
334 const auto &isotropic = std::dynamic_pointer_cast<noiseModel::Isotropic>(
335 measurement.noiseModel());
336 double sigma;
337 if (isotropic) {
338 sigma = isotropic->sigma();
339 } else {
340 const auto &robust = std::dynamic_pointer_cast<noiseModel::Robust>(
341 measurement.noiseModel());
342 // Check if noise model is robust
343 if (robust) {
344 // If robust, check if optimality certificate is expected
345 if (parameters.getCertifyOptimality()) {
346 throw std::invalid_argument(
347 "Certification of optimality does not work with robust cost.");
348 } else {
349 // Optimality certificate not required, so setting default sigma
350 sigma = 1;
351 }
352 } else {
353 throw std::invalid_argument(
354 "Shonan averaging noise models must be isotropic (but robust losses "
355 "are allowed).");
356 }
357 }
358 return 1.0 / (sigma * sigma);
359}

361/* ************************************************************************* */
362template <size_t d>
363Sparse ShonanAveraging<d>::buildD() const {
364 // Each measurement contributes 2*d elements along the diagonal of the
365 // degree matrix.
366 static constexpr size_t stride = 2 * d;
367
368 // Reserve space for triplets
369 std::vector<Eigen::Triplet<double>> triplets;
370 triplets.reserve(n: stride * measurements_.size());
371
372 for (const auto &measurement : measurements_) {
373 // Get pose keys
374 const auto &keys = measurement.keys();
375
376 // Get kappa from noise model
377 double kappa = Kappa<Rot, d>(measurement, parameters_);
378
379 const size_t di = d * keys[0], dj = d * keys[1];
380 for (size_t k = 0; k < d; k++) {
381 // Elements of ith block-diagonal
382 triplets.emplace_back(args: di + k, args: di + k, args&: kappa);
383 // Elements of jth block-diagonal
384 triplets.emplace_back(args: dj + k, args: dj + k, args&: kappa);
385 }
386 }
387
388 // Construct and return a sparse matrix from these triplets
389 const size_t dN = d * nrUnknowns();
390 Sparse D(dN, dN);
391 D.setFromTriplets(begin: triplets.begin(), end: triplets.end());
392
393 return D;
394}

396/* ************************************************************************* */
397template <size_t d>
398Sparse ShonanAveraging<d>::buildQ() const {
399 // Each measurement contributes 2*d^2 elements on a pair of symmetric
400 // off-diagonal blocks
401 static constexpr size_t stride = 2 * d * d;
402
403 // Reserve space for triplets
404 std::vector<Eigen::Triplet<double>> triplets;
405 triplets.reserve(n: stride * measurements_.size());
406
407 for (const auto &measurement : measurements_) {
408 // Get pose keys
409 const auto &keys = measurement.keys();
410
411 // Extract rotation measurement
412 const auto Rij = measurement.measured().matrix();
413
414 // Get kappa from noise model
415 double kappa = Kappa<Rot, d>(measurement, parameters_);
416
417 const size_t di = d * keys[0], dj = d * keys[1];
418 for (size_t r = 0; r < d; r++) {
419 for (size_t c = 0; c < d; c++) {
420 // Elements of ij block
421 triplets.emplace_back(di + r, dj + c, kappa * Rij(r, c));
422 // Elements of ji block
423 triplets.emplace_back(dj + r, di + c, kappa * Rij(c, r));
424 }
425 }
426 }
427
428 // Construct and return a sparse matrix from these triplets
429 const size_t dN = d * nrUnknowns();
430 Sparse Q(dN, dN);
431 Q.setFromTriplets(begin: triplets.begin(), end: triplets.end());
432
433 return Q;
434}

436/* ************************************************************************* */
437template <size_t d>
438Sparse ShonanAveraging<d>::computeLambda(const Matrix &S) const {
439 // Each pose contributes 2*d elements along the diagonal of Lambda
440 static constexpr size_t stride = d * d;
441
442 // Reserve space for triplets
443 const size_t N = nrUnknowns();
444 std::vector<Eigen::Triplet<double>> triplets;
445 triplets.reserve(n: stride * N);
446
447 // Do sparse-dense multiply to get Q*S'
448 auto QSt = Q_ * S.transpose();
449
450 for (size_t j = 0; j < N; j++) {
451 // Compute B, the building block for the j^th diagonal block of Lambda
452 const size_t dj = d * j;
453 Matrix B = QSt.middleRows(startRow: dj, n: d) * S.middleCols<d>(dj);
454
455 // Elements of jth block-diagonal
456 for (size_t r = 0; r < d; r++)
457 for (size_t c = 0; c < d; c++)
458 triplets.emplace_back(args: dj + r, args: dj + c, args: 0.5 * (B(r, c) + B(c, r)));
459 }
460
461 // Construct and return a sparse matrix from these triplets
462 Sparse Lambda(d * N, d * N);
463 Lambda.setFromTriplets(begin: triplets.begin(), end: triplets.end());
464 return Lambda;
465}

493// Perturb the initial initialVector by adding a spherically-uniformly
494// distributed random vector with 0.03*||initialVector||_2 magnitude to
495// initialVector
496// ref : Part III. C, Rosen, D. and Carlone, L., 2017, September. Computational
497// enhancements for certifiably correct SLAM. In Proceedings of the
498// International Conference on Intelligent Robots and Systems.
499static Vector perturb(const Vector &initialVector) {
500 // generate a 0.03*||x_0||_2 as stated in David's paper
501 int n = initialVector.rows();
502 Vector disturb = Vector::Random(size: n);
503 disturb.normalize();
504
505 double magnitude = initialVector.norm();
506 Vector perturbedVector = initialVector + 0.03 * magnitude * disturb;
507 perturbedVector.normalize();
508 return perturbedVector;
509}

511/* ************************************************************************* */
512/// MINIMUM EIGENVALUE COMPUTATIONS
513// Alg.6 from paper Distributed Certifiably Correct Pose-Graph Optimization,
514// it takes in the certificate matrix A as input, the maxIterations and the
515// minEigenvalueNonnegativityTolerance is set to 1000 and 10e-4 ad default,
516// there are two parts
517// in this algorithm:
518// (1) compute the maximum eigenpair (\lamda_dom, \vect{v}_dom) of A by power
519// method. if \lamda_dom is less than zero, then return the eigenpair. (2)
520// compute the maximum eigenpair (\theta, \vect{v}) of C = \lamda_dom * I - A by
521// accelerated power method. Then return (\lamda_dom - \theta, \vect{v}).
522static bool PowerMinimumEigenValue(
523 const Sparse &A, const Matrix &S, double &minEigenValue,
524 Vector *minEigenVector = 0, size_t *numIterations = 0,
525 size_t maxIterations = 1000,
526 double minEigenvalueNonnegativityTolerance = 10e-4) {
527
528 // a. Compute dominant eigenpair of S using power method
529 PowerMethod<Sparse> pmOperator(A);
530
531 const bool pmConverged = pmOperator.compute(
532 maxIterations, tol: 1e-5);
533
534 // Check convergence and bail out if necessary
535 if (!pmConverged) return false;
536
537 const double pmEigenValue = pmOperator.eigenvalue();
538
539 if (pmEigenValue < 0) {
540 // The largest-magnitude eigenvalue is negative, and therefore also the
541 // minimum eigenvalue, so just return this solution
542 minEigenValue = pmEigenValue;
543 if (minEigenVector) {
544 *minEigenVector = pmOperator.eigenvector();
545 minEigenVector->normalize(); // Ensure that this is a unit vector
546 }
547 return true;
548 }
549
550 const Sparse C = pmEigenValue * Matrix::Identity(rows: A.rows(), cols: A.cols()).sparseView() - A;
551 const std::optional<Vector> initial = perturb(initialVector: S.row(i: 0));
552 AcceleratedPowerMethod<Sparse> apmShiftedOperator(C, initial);
553
554 const bool minConverged = apmShiftedOperator.compute(
555 maxIterations, tol: minEigenvalueNonnegativityTolerance / pmEigenValue);
556
557 if (!minConverged) return false;
558
559 minEigenValue = pmEigenValue - apmShiftedOperator.eigenvalue();
560 if (minEigenVector) {
561 *minEigenVector = apmShiftedOperator.eigenvector();
562 minEigenVector->normalize(); // Ensure that this is a unit vector
563 }
564 if (numIterations) *numIterations = apmShiftedOperator.nrIterations();
565 return true;
566}

568/** This is a lightweight struct used in conjunction with Spectra to compute
569 * the minimum eigenvalue and eigenvector of a sparse matrix A; it has a single
570 * nontrivial function, perform_op(x,y), that computes and returns the product
571 * y = (A + sigma*I) x */
572struct MatrixProdFunctor {
573 using Scalar = double;
574
575 // Const reference to an externally-held matrix whose minimum-eigenvalue we
576 // want to compute
577 const Sparse &A_;
578
579 // Spectral shift
580 double sigma_;
581
582 // Constructor
583 explicit MatrixProdFunctor(const Sparse &A, double sigma = 0)
584 : A_(A), sigma_(sigma) {}
585
586 int rows() const { return A_.rows(); }
587 int cols() const { return A_.cols(); }
588
589 // Matrix-vector multiplication operation
590 void perform_op(const double *x, double *y) const {
591 // Wrap the raw arrays as Eigen Vector types
592 Eigen::Map<const Vector> X(x, rows());
593 Eigen::Map<Vector> Y(y, rows());
594
595 // Do the multiplication using wrapped Eigen vectors
596 Y = A_ * X + sigma_ * X;
597 }
598};

600/// Function to compute the minimum eigenvalue of A using Lanczos in Spectra.
601/// This does 2 things:
602///
603/// (1) Quick (coarse) eigenvalue computation to estimate the largest-magnitude
604/// eigenvalue (2) A second eigenvalue computation applied to A-sigma*I, where
605/// sigma is chosen to make the minimum eigenvalue of A the extremal eigenvalue
606/// of A-sigma*I
607///
608/// Upon completion, this returns a boolean value indicating whether the minimum
609/// eigenvalue was computed to the required precision -- if so, its sets the
610/// values of minEigenValue and minEigenVector appropriately
611
612/// Note that in the following function signature, S is supposed to be the
613/// block-row-matrix that is a critical point for the optimization algorithm;
614/// either S (Stiefel manifold) or R (block rotations). We use this to
615/// construct a starting vector v for the Lanczos process that will be close to
616/// the minimum eigenvector we're looking for whenever the relaxation is exact
617/// -- this is a key feature that helps to make this method fast. Note that
618/// instead of passing in all of S, it would be enough to pass in one of S's
619/// *rows*, if that's more convenient.
620
621// For the defaults, David Rosen says:
622// - maxIterations refers to the max number of Lanczos iterations to run;
623// ~1000 should be sufficiently large
624// - We've been using 10^-4 for the nonnegativity tolerance
625// - for numLanczosVectors, 20 is a good default value
626
627static bool SparseMinimumEigenValue(
628 const Sparse &A, const Matrix &S, double *minEigenValue,
629 Vector *minEigenVector = 0, size_t *numIterations = 0,
630 size_t maxIterations = 1000,
631 double minEigenvalueNonnegativityTolerance = 10e-4,
632 Eigen::Index numLanczosVectors = 20) {
633 // a. Estimate the largest-magnitude eigenvalue of this matrix using Lanczos
634 MatrixProdFunctor lmOperator(A);
635 Spectra::SymEigsSolver<MatrixProdFunctor> lmEigenValueSolver(
636 lmOperator, 1, std::min(a: numLanczosVectors, b: A.rows()));
637 lmEigenValueSolver.init();
638
639 const int lmConverged =
640 lmEigenValueSolver.compute(selection: Spectra::SortRule::LargestMagn, maxit: maxIterations,
641 tol: 1e-4, sorting: Spectra::SortRule::LargestMagn);
642
643 // Check convergence and bail out if necessary
644 if (lmConverged != 1) return false;
645
646 const double lmEigenValue = lmEigenValueSolver.eigenvalues()(0);
647
648 if (lmEigenValue < 0) {
649 // The largest-magnitude eigenvalue is negative, and therefore also the
650 // minimum eigenvalue, so just return this solution
651 *minEigenValue = lmEigenValue;
652 if (minEigenVector) {
653 *minEigenVector = lmEigenValueSolver.eigenvectors(nvec: 1).col(i: 0);
654 minEigenVector->normalize(); // Ensure that this is a unit vector
655 }
656 return true;
657 }
658
659 // The largest-magnitude eigenvalue is positive, and is therefore the
660 // maximum eigenvalue. Therefore, after shifting the spectrum of A
661 // by -2*lmEigenValue (by forming A - 2*lambda_max*I), the shifted
662 // spectrum will lie in the interval [minEigenValue(A) - 2* lambda_max(A),
663 // -lambda_max*A]; in particular, the largest-magnitude eigenvalue of
664 // A - 2*lambda_max*I is minEigenValue - 2*lambda_max, with corresponding
665 // eigenvector v_min
666
667 MatrixProdFunctor minShiftedOperator(A, -2 * lmEigenValue);
668
669 Spectra::SymEigsSolver<MatrixProdFunctor> minEigenValueSolver(
670 minShiftedOperator, 1, std::min(a: numLanczosVectors, b: A.rows()));
671
672 // If S is a critical point of F, then S^T is also in the null space of S -
673 // Lambda(S) (cf. Lemma 6 of the tech report), and therefore its rows are
674 // eigenvectors corresponding to the eigenvalue 0. In the case that the
675 // relaxation is exact, this is the *minimum* eigenvalue, and therefore the
676 // rows of S are exactly the eigenvectors that we're looking for. On the
677 // other hand, if the relaxation is *not* exact, then S - Lambda(S) has at
678 // least one strictly negative eigenvalue, and the rows of S are *unstable
679 // fixed points* for the Lanczos iterations. Thus, we will take a slightly
680 // "fuzzed" version of the first row of S as an initialization for the
681 // Lanczos iterations; this allows for rapid convergence in the case that
682 // the relaxation is exact (since are starting close to a solution), while
683 // simultaneously allowing the iterations to escape from this fixed point in
684 // the case that the relaxation is not exact.
685 Vector v0 = S.row(i: 0).transpose();
686 Vector perturbation(v0.size());
687 perturbation.setRandom();
688 perturbation.normalize();
689 Vector xinit = v0 + (.03 * v0.norm()) * perturbation; // Perturb v0 by ~3%
690
691 // Use this to initialize the eigensolver
692 minEigenValueSolver.init(init_resid: xinit.data());
693
694 // Now determine the relative precision required in the Lanczos method in
695 // order to be able to estimate the smallest eigenvalue within an *absolute*
696 // tolerance of 'minEigenvalueNonnegativityTolerance'
697 const int minConverged = minEigenValueSolver.compute(
698 selection: Spectra::SortRule::LargestMagn, maxit: maxIterations,
699 tol: minEigenvalueNonnegativityTolerance / lmEigenValue,
700 sorting: Spectra::SortRule::LargestMagn);
701
702 if (minConverged != 1) return false;
703
704 *minEigenValue = minEigenValueSolver.eigenvalues()(0) + 2 * lmEigenValue;
705 if (minEigenVector) {
706 *minEigenVector = minEigenValueSolver.eigenvectors(nvec: 1).col(i: 0);
707 minEigenVector->normalize(); // Ensure that this is a unit vector
708 }
709 if (numIterations) *numIterations = minEigenValueSolver.num_iterations();
710 return true;
711}

764template <size_t d>
765VectorValues ShonanAveraging<d>::TangentVectorValues(size_t p,
766 const Vector &v) {
767 VectorValues delta;
768 // Create a tangent direction xi with eigenvector segment v_i
769 const size_t dimension = SOn::Dimension(n: p);
770 double sign0 = pow(x: -1.0, y: round(x: (p + 1) / 2) + 1);
771 for (size_t i = 0; i < v.size() / d; i++) {
772 // Assumes key is 0-based integer
773 const auto v_i = v.segment<d>(d * i);
774 Vector xi = Vector::Zero(size: dimension);
775 double sign = sign0;
776 for (size_t j = 0; j < d; j++) {
777 xi(j + p - d - 1) = sign * v_i(d - j - 1);
778 sign = -sign;
779 }
780 delta.insert(j: i, value: xi);
781 }
782 return delta;
783}

785/* ************************************************************************* */
786template <size_t d>
787Matrix ShonanAveraging<d>::riemannianGradient(size_t p,
788 const Values &values) const {
789 Matrix S_dot = StiefelElementMatrix(values);
790 // calculate the gradient of F(Q_dot) at Q_dot
791 Matrix euclideanGradient = 2 * (L_ * (S_dot.transpose())).transpose();
792 // cout << "euclidean gradient rows and cols" << euclideanGradient.rows() <<
793 // "\t" << euclideanGradient.cols() << endl;
794
795 // project the gradient onto the entire euclidean space
796 Matrix symBlockDiagProduct(p, d * nrUnknowns());
797 for (size_t i = 0; i < nrUnknowns(); i++) {
798 // Compute block product
799 const size_t di = d * i;
800 const Matrix P = S_dot.middleCols<d>(di).transpose() *
801 euclideanGradient.middleCols<d>(di);
802 // Symmetrize this block
803 Matrix S = .5 * (P + P.transpose());
804 // Compute S_dot * S and set corresponding block
805 symBlockDiagProduct.middleCols<d>(di) = S_dot.middleCols<d>(di) * S;
806 }
807 Matrix riemannianGradient = euclideanGradient - symBlockDiagProduct;
808 return riemannianGradient;
809}

820/* ************************************************************************* */
821template <size_t d>
822Values ShonanAveraging<d>::initializeWithDescent(
823 size_t p, const Values &values, const Vector &minEigenVector,
824 double minEigenValue, double gradienTolerance,
825 double preconditionedGradNormTolerance) const {
826 double funcVal = costAt(p: p - 1, values);
827 double alphaMin = 1e-2;
828 double alpha =
829 std::max(a: 1024 * alphaMin, b: 10 * gradienTolerance / fabs(x: minEigenValue));
830 vector<double> alphas;
831 vector<double> fvals;
832 // line search
833 while ((alpha >= alphaMin)) {
834 Values Qplus = LiftwithDescent(p, values, minEigenVector: alpha * minEigenVector);
835 double funcValTest = costAt(p, values: Qplus);
836 Matrix gradTest = riemannianGradient(p, values: Qplus);
837 double gradTestNorm = gradTest.norm();
838 // Record alpha and funcVal
839 alphas.push_back(x: alpha);
840 fvals.push_back(x: funcValTest);
841 if ((funcVal > funcValTest) && (gradTestNorm > gradienTolerance)) {
842 return Qplus;
843 }
844 alpha /= 2;
845 }
846
847 auto fminIter = min_element(first: fvals.begin(), last: fvals.end());
848 auto minIdx = distance(first: fvals.begin(), last: fminIter);
849 double fMin = fvals[minIdx];
850 double aMin = alphas[minIdx];
851 if (fMin < funcVal) {
852 Values Qplus = LiftwithDescent(p, values, minEigenVector: aMin * minEigenVector);
853 return Qplus;
854 }
855
856 return LiftwithDescent(p, values, minEigenVector: alpha * minEigenVector);
857}

890template <size_t d>
891std::pair<Values, double> ShonanAveraging<d>::run(const Values &initialEstimate,
892 size_t pMin,
893 size_t pMax) const {
894 if (pMin < d) {
895 throw std::runtime_error("pMin is smaller than the base dimension d");
896 }
897 Values Qstar;
898 Values initialSOp = LiftTo<Rot>(pMin, initialEstimate); // lift to pMin!
899 for (size_t p = pMin; p <= pMax; p++) {
900 // Optimize until convergence at this level
901 Qstar = tryOptimizingAt(p, initial: initialSOp);
902 if (parameters_.getUseHuber() || !parameters_.getCertifyOptimality()) {
903 // in this case, there is no optimality certification
904 if (pMin != pMax) {
905 throw std::runtime_error(
906 "When using robust norm, Shonan only tests a single rank. Set pMin = pMax");
907 }
908 const Values SO3Values = roundSolution(values: Qstar);
909 return {SO3Values, 0};
910 } else {
911 // Check certificate of global optimality
912 Vector minEigenVector;
913 double minEigenValue = computeMinEigenValue(values: Qstar, minEigenVector: &minEigenVector);
914 if (minEigenValue > parameters_.optimalityThreshold) {
915 // If at global optimum, round and return solution
916 const Values SO3Values = roundSolution(values: Qstar);
917 return {SO3Values, minEigenValue};
918 }
919
920 // Not at global optimimum yet, so check whether we will go to next level
921 if (p != pMax) {
922 // Calculate initial estimate for next level by following minEigenVector
923 initialSOp =
924 initializeWithDescent(p: p + 1, values: Qstar, minEigenVector, minEigenValue);
925 }
926 }
927 }
928 throw std::runtime_error("Shonan::run did not converge for given pMax");
929}
