76/**
77 * equals with a tolerance
78 */
79template <class MATRIX>
80bool equal_with_abs_tol(const Eigen::DenseBase<MATRIX>& A, const Eigen::DenseBase<MATRIX>& B, double tol = 1e-9) {
81
82 const size_t n1 = A.cols(), m1 = A.rows();
83 const size_t n2 = B.cols(), m2 = B.rows();
84
85 if(m1!=m2 || n1!=n2) return false;
86
87 for(size_t i=0; i<m1; i++)
88 for(size_t j=0; j<n1; j++) {
89 if(!fpEqual(A(i,j), B(i,j), tol, false)) {
90 return false;
91 }
92 }
93 return true;
94}

164/**
165 * extract submatrix, slice semantics, i.e. range = [i1,i2[ excluding i2
166 * @param A matrix
167 * @param i1 first row index
168 * @param i2 last row index + 1
169 * @param j1 first col index
170 * @param j2 last col index + 1
171 * @return submatrix A(i1:i2-1,j1:j2-1)
172 */
173template<class MATRIX>
174Eigen::Block<const MATRIX> sub(const MATRIX& A, size_t i1, size_t i2, size_t j1, size_t j2) {
175 size_t m=i2-i1, n=j2-j1;
176 return A.block(i1,j1,m,n);
177}

179/**
180 * insert a submatrix IN PLACE at a specified location in a larger matrix
181 * NOTE: there is no size checking
182 * @param fullMatrix matrix to be updated
183 * @param subMatrix matrix to be inserted
184 * @param i is the row of the upper left corner insert location
185 * @param j is the column of the upper left corner insert location
186 */
187template <typename Derived1, typename Derived2>
188void insertSub(Eigen::MatrixBase<Derived1>& fullMatrix, const Eigen::MatrixBase<Derived2>& subMatrix, size_t i, size_t j) {
189 fullMatrix.block(i, j, subMatrix.rows(), subMatrix.cols()) = subMatrix;
190}

224/// Reshape functor
225template <int OutM, int OutN, int OutOptions, int InM, int InN, int InOptions>
226struct Reshape {
227 //TODO replace this with Eigen's reshape function as soon as available. (There is a PR already pending : https://bitbucket.org/eigen/eigen/pull-request/41/reshape/diff)
228 typedef Eigen::Map<const Eigen::Matrix<double, OutM, OutN, OutOptions> > ReshapedType;
229 static inline ReshapedType reshape(const Eigen::Matrix<double, InM, InN, InOptions> & in) {
230 return in.data();
231 }
232};

234/// Reshape specialization that does nothing as shape stays the same (needed to not be ambiguous for square input equals square output)
235template <int M, int InOptions>
236struct Reshape<M, M, InOptions, M, M, InOptions> {
237 typedef const Eigen::Matrix<double, M, M, InOptions> & ReshapedType;
238 static inline ReshapedType reshape(const Eigen::Matrix<double, M, M, InOptions> & in) {
239 return in;
240 }
241};

243/// Reshape specialization that does nothing as shape stays the same
244template <int M, int N, int InOptions>
245struct Reshape<M, N, InOptions, M, N, InOptions> {
246 typedef const Eigen::Matrix<double, M, N, InOptions> & ReshapedType;
247 static inline ReshapedType reshape(const Eigen::Matrix<double, M, N, InOptions> & in) {
248 return in;
249 }
250};

252/// Reshape specialization that does transpose
253template <int M, int N, int InOptions>
254struct Reshape<N, M, InOptions, M, N, InOptions> {
255 typedef typename Eigen::Matrix<double, M, N, InOptions>::ConstTransposeReturnType ReshapedType;
256 static inline ReshapedType reshape(const Eigen::Matrix<double, M, N, InOptions> & in) {
257 return in.transpose();
258 }
259};

261template <int OutM, int OutN, int OutOptions, int InM, int InN, int InOptions>
262inline typename Reshape<OutM, OutN, OutOptions, InM, InN, InOptions>::ReshapedType reshape(const Eigen::Matrix<double, InM, InN, InOptions> & m){
263 static_assert(InM * InN == OutM * OutN);
264 return Reshape<OutM, OutN, OutOptions, InM, InN, InOptions>::reshape(m);

429/**
430 * Functor that implements multiplication of a vector b with the inverse of a
431 * matrix A. The derivatives are inspired by Mike Giles' "An extended collection
432 * of matrix derivative results for forward and reverse mode algorithmic
433 * differentiation", at https://people.maths.ox.ac.uk/gilesm/files/NA-08-01.pdf
434 */
435template <int N>
436struct MultiplyWithInverse {
437 typedef Eigen::Matrix<double, N, 1> VectorN;
438 typedef Eigen::Matrix<double, N, N> MatrixN;
439
440 /// A.inverse() * b, with optional derivatives
441 VectorN operator()(const MatrixN& A, const VectorN& b,
442 OptionalJacobian<N, N* N> H1 = {},
443 OptionalJacobian<N, N> H2 = {}) const {
444 const MatrixN invA = A.inverse();
445 const VectorN c = invA * b;
446 // The derivative in A is just -[c[0]*invA c[1]*invA ... c[N-1]*invA]
447 if (H1)
448 for (size_t j = 0; j < N; j++)
449 H1->template middleCols<N>(N * j) = -c[j] * invA;
450 // The derivative in b is easy, as invA*b is just a linear map:
451 if (H2) *H2 = invA;
452 return c;
453 }
454};
455
456/**
457 * Functor that implements multiplication with the inverse of a matrix, itself
458 * the result of a function f. It turn out we only need the derivatives of the
459 * operator phi(a): b -> f(a) * b
460 */
461template <typename T, int N>
462struct MultiplyWithInverseFunction {
463 inline constexpr static auto M = traits<T>::dimension;
464 typedef Eigen::Matrix<double, N, 1> VectorN;
465 typedef Eigen::Matrix<double, N, N> MatrixN;
466
467 // The function phi should calculate f(a)*b, with derivatives in a and b.
468 // Naturally, the derivative in b is f(a).
469 typedef std::function<VectorN(
470 const T&, const VectorN&, OptionalJacobian<N, M>, OptionalJacobian<N, N>)>
471 Operator;
472
473 /// Construct with function as explained above
474 MultiplyWithInverseFunction(const Operator& phi) : phi_(phi) {}
475
476 /// f(a).inverse() * b, with optional derivatives
477 VectorN operator()(const T& a, const VectorN& b,
478 OptionalJacobian<N, M> H1 = {},
479 OptionalJacobian<N, N> H2 = {}) const {
480 MatrixN A;
481 phi_(a, b, {}, A); // get A = f(a) by calling f once
482 const MatrixN invA = A.inverse();
483 const VectorN c = invA * b;
484
485 if (H1) {
486 Eigen::Matrix<double, N, M> H;
487 phi_(a, c, H, {}); // get derivative H of forward mapping
488 *H1 = -invA* H;
489 }
490 if (H2) *H2 = invA;
491 return c;
492 }
493
494 private:
495 const Operator phi_;
496};
