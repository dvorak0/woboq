53/* ************************************************************************* */
54void TimingOutline::add(size_t usecs, size_t usecsWall) {
55 t_ += usecs;
56 tWall_ += usecsWall;
57 tIt_ += usecs;
58 double secs = (double(usecs) / 1000000.0);
59 t2_ += secs * secs;
60 ++n_;
61}

74/* ************************************************************************* */
75size_t TimingOutline::time() const {
76 size_t time = 0;
77 bool hasChildren = false;
78 for(const ChildMap::value_type& child: children_) {
79 time += child.second->time();
80 hasChildren = true;
81 }
82 if (hasChildren)
83 return time;
84 else
85 return t_;
86}

88/* ************************************************************************* */
89void TimingOutline::print(const std::string& outline) const {
90 std::string formattedLabel = label_;
91 std::replace(first: formattedLabel.begin(), last: formattedLabel.end(), old_value: '_', new_value: ' ');
92 std::cout << outline << "-" << formattedLabel << ": " << self() << " CPU ("
93 << n_ << " times, " << wall() << " wall, " << secs() << " children, min: "
94 << min() << " max: " << max() << ")\n";
95 // Order children
96 ChildOrder childOrder;
97 for(const ChildMap::value_type& child: children_) {
98 childOrder[child.second->myOrder_] = child.second;
99 }
100 // Print children
101 for(const ChildOrder::value_type& order_child: childOrder) {
102 std::string childOutline(outline);
103 childOutline += "| ";
104 order_child.second->print(outline: childOutline);
105 }
106 std::cout.flush();
107}

153void TimingOutline::print2(const std::string& outline,
154 const double parentTotal) const {
155 const int w1 = 24, w2 = 2, w3 = 6, w4 = 8, precision = 2;
156 const double selfTotal = self(), selfMean = (n_ > 0) ? selfTotal / double(n_) : 0.0;
157 const double childTotal = secs();
158
159 // compute standard deviation
160 const double selfStd = (n_ > 0) ? sqrt(x: t2_ / double(n_) - selfMean * selfMean) : 0.0;
161 const std::string label = outline + label_ + ": ";
162
163 if (n_ == 0) {
164 std::cout << label << std::fixed << std::setprecision(precision)
165 << childTotal << " seconds" << std::endl;
166 } else {
167 std::cout << std::setw(w1 + outline.length()) << label;
168 std::cout << std::setiosflags(std::ios::right) << std::setw(w2) << n_
169 << " (times), " << std::setiosflags(std::ios::right) << std::fixed
170 << std::setw(w3) << std::setprecision(precision) << selfMean
171 << " (mean), " << std::setiosflags(std::ios::right) << std::fixed
172 << std::setw(w3) << std::setprecision(precision) << selfStd << " (std),"
173 << std::setiosflags(std::ios::right) << std::fixed << std::setw(w4)
174 << std::setprecision(precision) << selfTotal << " (total),";
175
176 if (parentTotal > 0.0)
177 std::cout << std::setiosflags(std::ios::right) << std::fixed
178 << std::setw(w3) << std::setprecision(precision)
179 << 100.0 * selfTotal / parentTotal << " (%)";
180
181 std::cout << std::endl;
182 }
183
184 for(const ChildMap::value_type& child: children_) {
185 std::string childOutline(outline);
186 if (n_ == 0) {
187 child.second->print2(outline: childOutline, parentTotal: childTotal);
188 } else {
189 childOutline += " ";
190 child.second->print2(outline: childOutline, parentTotal: selfTotal);
191 }
192 }
193}

196const std::shared_ptr<TimingOutline>& TimingOutline::child(size_t child,
197 const std::string& label, const std::weak_ptr<TimingOutline>& thisPtr) {
198#if GTSAM_USE_BOOST_FEATURES
199 assert(thisPtr.lock().get() == this);
200 std::shared_ptr<TimingOutline>& result = children_[child];
201 if (!result) {
202 // Create child if necessary
203 result.reset(p: new TimingOutline(label, child));
204 ++this->lastChildOrder_;
205 result->myOrder_ = this->lastChildOrder_;
206 result->parent_ = thisPtr;
207 }
208 return result;
209#else
210 assert(thisPtr.lock().get() == this);
211 std::shared_ptr<TimingOutline>& result = children_[child];
212 if (!result) {
213 // Create child if necessary
214 result.reset(new TimingOutline(label, child));
215 ++this->lastChildOrder_;
216 result->myOrder_ = this->lastChildOrder_;
217 result->parent_ = thisPtr;
218 }
219 return result;
220#endif
221}

223/* ************************************************************************* */
224void TimingOutline::tic() {
225#if GTSAM_USE_BOOST_FEATURES
226#ifdef GTSAM_USING_NEW_BOOST_TIMERS
227 assert(timer_.is_stopped());
228 timer_.start();
229#else
230 assert(!timerActive_);
231 timer_.restart();
232 *timerActive_ = true;
233#endif
234
235#ifdef GTSAM_USE_TBB
236 tbbTimer_ = tbb::tick_count::now();
237#endif
238#else // GTSAM_USE_BOOST_FEATURES
239 assert(!timer_active_);
240 cpu_timer_start_ = std::clock();
241 wall_timer_start_ = std::chrono::steady_clock::now();
242 timer_active_ = true;
243#endif // GTSAM_USE_BOOST_FEATURES
244}

246/* ************************************************************************* */
247void TimingOutline::toc() {
248#if GTSAM_USE_BOOST_FEATURES
249
250#ifdef GTSAM_USING_NEW_BOOST_TIMERS
251
252 assert(!timer_.is_stopped());
253 timer_.stop();
254 size_t cpuTime = (timer_.elapsed().user + timer_.elapsed().system) / 1000;
255# ifndef GTSAM_USE_TBB
256 size_t wallTime = timer_.elapsed().wall / 1000;
257# endif
258
259#else
260
261 assert(timerActive_);
262 double elapsed = timer_.elapsed();
263 size_t cpuTime = size_t(elapsed * 1000000.0);
264 *timerActive_ = false;
265# ifndef GTSAM_USE_TBB
266 size_t wallTime = cpuTime;
267# endif
268
269#endif
270
271#ifdef GTSAM_USE_TBB
272 size_t wallTime = size_t(
273 (tbb::tick_count::now() - tbbTimer_).seconds() * 1e6);
274#endif
275
276 add(usecs: cpuTime, usecsWall: wallTime);
277#else // GTSAM_USE_BOOST_FEATURES
278 assert(timer_active_);
279
280 // measure CPU time
281 const std::clock_t cpu_end = std::clock();
282 const double cpu_secs = static_cast<double>(cpu_end - cpu_timer_start_) / CLOCKS_PER_SEC;
283 const size_t cpuTime = static_cast<size_t>(cpu_secs * 1000000.0);
284
285 // measure wall time
286 const auto wall_end = std::chrono::steady_clock::now();
287 const size_t wallTime =
288 std::chrono::duration_cast<std::chrono::microseconds>(
289 wall_end - wall_timer_start_)
290 .count();
291
292 add(cpuTime, wallTime);
293 timer_active_ = false;
294#endif // GTSAM_USE_BOOST_FEATURES
295}

297/* ************************************************************************* */
298void TimingOutline::finishedIteration() {
299 if (tIt_ > tMax_)
300 tMax_ = tIt_;
301 if (tMin_ == 0 || tIt_ < tMin_)
302 tMin_ = tIt_;
303 tIt_ = 0;
304 for(ChildMap::value_type& child: children_) {
305 child.second->finishedIteration();
306 }
307}

309/* ************************************************************************* */
310size_t getTicTocID(const char *descriptionC) {
311 const std::string description(descriptionC);
312 // Global (static) map from strings to ID numbers and current next ID number
313 static size_t nextId = 0;
314 static gtsam::FastMap<std::string, size_t> idMap;
315
316 // Retrieve or add this string
317 auto it = idMap.find(x: description);
318 if (it == idMap.end()) {
319 it = idMap.insert(x: {description, nextId}).first;
320 ++nextId;
321 }
322
323 // Return ID
324 return it->second;
325}

328void tic(size_t id, const char *labelC) {
329 const std::string label(labelC);
330 std::shared_ptr<TimingOutline> node = //
331 gCurrentTimer.lock()->child(child: id, label, thisPtr: gCurrentTimer);
332 gCurrentTimer = node;
333 node->tic();
334}

336/* ************************************************************************* */
337void toc(size_t id, const char *labelC) {
338 const std::string label(labelC);
339 std::shared_ptr<TimingOutline> current(gCurrentTimer.lock());
340 if (id != current->id_) {
341 gTimingRoot->print();
342 throw std::invalid_argument(
343 "gtsam timing: Mismatched tic/toc: gttoc(\"" + label +
344 "\") called when last tic was \"" + current->label_ + "\".");
345 }
346 if (!current->parent_.lock()) {
347 gTimingRoot->print();
348 throw std::invalid_argument(
349 "gtsam timing: Mismatched tic/toc: extra gttoc(\"" + label +
350 "\"), already at the root");
351 }
352 current->toc();
353 gCurrentTimer = current->parent_;
354}
