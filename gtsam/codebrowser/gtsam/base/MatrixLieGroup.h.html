27 // Helper to compute product of compile-time dimensions, returning Dynamic if either is Dynamic.
28 constexpr int product(int a, int b) {
29 return (a == Eigen::Dynamic || b == Eigen::Dynamic) ? Eigen::Dynamic : a * b;
30 }

32 // Helper to compute the matrix of vectorized generators for fixed-size groups.
33 template<class Class, int D, int N>
34 auto computeVectorizedGenerators() {
35 static_assert(D != Eigen::Dynamic && N != Eigen::Dynamic,
36 "This helper is only for fixed-size Lie groups.");
37 Eigen::Matrix<double, N* N, D> P;
38 for (int i = 0; i < D; ++i) {
39 const auto G_i = Class::Hat(Class::TangentVector::Unit(D, i));
40 P.col(i) = Eigen::Map<const Eigen::Matrix<double, N* N, 1>>(G_i.data());
41 }
42 return P;
43 }
44 } // namespace internal

46 /// A CRTP helper class that implements matrix Lie group methods.
47 /// To use, derive from MatrixLieGroup<Class,D,N> instead of LieGroup<Class,D>.
48 /// Your class must implement a `matrix()` method and static `Hat()/Vee()` methods.
49 template<class Class, int D, int N>
50 struct MatrixLieGroup : public LieGroup<Class, D> {
51 using Base = LieGroup<Class, D>;
52 using Base::dimension;
53 using Base::Dim;
54 using Base::dim;
55 using ChartJacobian = typename Base::ChartJacobian;
56 using Jacobian = typename Base::Jacobian;
57 using TangentVector = typename Base::TangentVector;
58
59 /**
60 * Vectorize the matrix representation of a Lie group element.
61 * The derivative `H` is the `(N*N) x D` Jacobian of this vectorization map.
62 * It is given by the formula `H = (I_N ⊗ T) * P`, where `T` is the `N x N`
63 * matrix of this group element, `⊗` is the Kronecker product, and `P` is
64 * the `(N*N) x D` matrix whose columns are the vectorized Lie algebra
65 * generators `vec(Hat(e_j))`. This can be computed efficiently via
66 * block-wise multiplication.
67 */
68 Eigen::Matrix<double, internal::product(a: N, b: N), 1> vec(
69 OptionalJacobian<internal::product(a: N, b: N), D> H = {}) const {
70 const auto& derived = static_cast<const Class&>(*this);
71 const auto& T = derived.matrix();
72
73 if (H) {
74 if constexpr (N != Eigen::Dynamic && D != Eigen::Dynamic) { // Fixed-size case
75 const auto& P = VectorizedGenerators();
76 for (int i = 0; i < N; ++i) {
77 H->block(i * N, 0, N, D) = T * P.block(i * N, 0, N, D);
78 }
79 }
80 else { // Dynamic-size case
81 const size_t n = T.rows();
82 const size_t d = derived.dim();
83 H->resize(n * n, d);
84
85 // Create P, the matrix of vectorized generators, on the fly.
86 Eigen::Matrix<double, Eigen::Dynamic, Eigen::Dynamic> P(n * n, d);
87 for (size_t j = 0; j < d; ++j) {
88 const auto G_j = Class::Hat(TangentVector::Unit(d, j));
89 P.col(i: j) = Eigen::Map<const Eigen::Matrix<double, Eigen::Dynamic, 1>>(
90 G_j.data(), n * n);
91 }
92
93 // Apply the formula H = (I_n ⊗ T) * P.
94 for (size_t i = 0; i < n; ++i) {
95 H->block(i * n, 0, n, d) = T * P.block(startRow: i * n, startCol: 0, blockRows: n, blockCols: d);
96 }
97 }
98 }
99
100 if constexpr (N != Eigen::Dynamic) { // Fixed-size case
101 return Eigen::Map<const Eigen::Matrix<double, N* N, 1>>(T.data());
102 }
103 else { // Dynamic-size case
104 return Eigen::Map<const Eigen::Matrix<double, Eigen::Dynamic, 1>>(
105 T.data(), T.size());
106 }
107 }

109 /**
110 * A generic implementation of AdjointMap for matrix Lie groups.
111 * The Adjoint map `Ad_g` is the tangent map of the conjugation `C_g(x) = g*x*g.inverse()`
112 * at the identity. For matrix Lie groups, `Ad_g(v) = g*v*g.inverse()` where `v` is an
113 * element of the Lie algebra. The columns of the Adjoint matrix are
114 * `vee(g * Hat(e_i) * g.inverse())` for each basis vector `e_i`.
115 * This method can be overridden by derived classes with a more efficient,
116 * closed-form solution.
117 */
118 Jacobian AdjointMap() const {
119 const auto& m = static_cast<const Class&>(*this);
120 size_t d = D;
121 if constexpr (D == Eigen::Dynamic) d = m.dim();
122 Jacobian adj(d, d);
123 const auto T_mat = m.matrix();
124 const auto T_inv_mat = m.inverse().matrix();
125 for (size_t i = 0; i < d; i++) {
126 // TangentVector::Unit(d, i) works for both fixed and dynamic size vectors.
127 const auto G_i = Class::Hat(TangentVector::Unit(d, i));
128 adj.col(i) = Class::Vee(T_mat * G_i * T_inv_mat);
129 }
130 return adj;
131 }
132
133 private:
134 /// Pre-compute and store vectorized generators for fixed-size groups.
135 inline static const Eigen::Matrix<double, internal::product(a: N, b: N), D>&
136 VectorizedGenerators() {
137 static const auto P =
138 internal::computeVectorizedGenerators<Class, D, N>();
139 return P;
140 }
141 };

145 /// Adds LieAlgebra, Hat, Vee, and Vec to LieGroupTraits
146 template <class Class, int N> struct MatrixLieGroupTraits : LieGroupTraits<Class> {
147 using LieAlgebra = typename Class::LieAlgebra;
148 using TangentVector = typename LieGroupTraits<Class>::TangentVector;
149
150 static LieAlgebra Hat(const TangentVector& v) {
151 return Class::Hat(v);
152 }
153
154 static TangentVector Vee(const LieAlgebra& X) {
155 return Class::Vee(X);
156 }
157
158 /// Vectorize the matrix representation of a Lie group element.
159 static Eigen::Matrix<double, product(a: N, b: N), 1> Vec(
160 const Class& m,
161 OptionalJacobian<product(a: N, b: N),
162 LieGroupTraits<Class>::dimension> H = {}) {
163 return m.vec(H);
164 }
165 };
166
167 /// Both LieGroupTraits and Testable
168 template<class Class, int N> struct MatrixLieGroup : MatrixLieGroupTraits<Class, N>, Testable<Class> {};

173 * Matrix Lie Group Concept
174 */
175 template<typename T>
176 class IsMatrixLieGroup : public IsLieGroup<T> {
177 public:
178 using LieAlgebra = typename traits<T>::LieAlgebra;
179 using TangentVector = typename traits<T>::TangentVector;
180
181 GTSAM_CONCEPT_USAGE(IsMatrixLieGroup) {
182 // hat and vee
183 X = traits<T>::Hat(xi);
184 xi = traits<T>::Vee(X);
185 // vec
186 (void)traits<T>::Vec(g);
187 }
188 private:
189 T g;
190 LieAlgebra X;
191 TangentVector xi;
192 };

194 /**
195 * Three term approximation of the Baker-Campbell-Hausdorff formula
196 * In non-commutative Lie groups, when composing exp(Z) = exp(X)exp(Y)
197 * it is not true that Z = X+Y. Instead, Z can be calculated using the BCH
198 * formula: Z = X + Y + [X,Y]/2 + [X-Y,[X,Y]]/12 - [Y,[X,[X,Y]]]/24
199 * http://en.wikipedia.org/wiki/Baker-Campbell-Hausdorff_formula
200 */
201 /// AGC: bracket() only appears in Rot3 tests, should this be used elsewhere?
202 template<class T>
203 T BCH(const T& X, const T& Y) {
204 static const double _2 = 1. / 2., _12 = 1. / 12., _24 = 1. / 24.;
205 T X_Y = bracket(X, Y);
206 return T(X + Y + _2 * X_Y + _12 * bracket(X - Y, X_Y) - _24 * bracket(Y, bracket(X, X_Y)));
207 }

217 /**
218 * Exponential map given exponential coordinates
219 * class T needs a constructor from Matrix.
220 * @param x exponential coordinates, vector of size n
221 * @ return a T
222 */
223 template <class T>
224 T expm(const Vector& x, int K = 7) {
225 const Matrix xhat = T::Hat(x);
226 return T(expm(A: xhat, K));
227 }
