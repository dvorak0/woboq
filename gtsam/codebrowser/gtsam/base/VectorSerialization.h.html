29namespace boost {
30namespace serialization {
31
32// split version - copies into an STL vector for serialization
33template <class Archive>
34void save(Archive& ar, const gtsam::Vector& v, unsigned int /*version*/) {
35 const size_t size = v.size();
36 ar << BOOST_SERIALIZATION_NVP(size);
37 ar << make_nvp(n: "data", v: make_array(t: v.data(), s: v.size()));
38}
39
40template <class Archive>
41void load(Archive& ar, gtsam::Vector& v, unsigned int /*version*/) {
42 size_t size;
43 ar >> BOOST_SERIALIZATION_NVP(size);
44 v.resize(size);
45 ar >> make_nvp(n: "data", v: make_array(t: v.data(), s: v.size()));
46}
47
48// split version - copies into an STL vector for serialization
49template <class Archive, int D>
50void save(Archive& ar, const Eigen::Matrix<double, D, 1>& v,
51 unsigned int /*version*/) {
52 ar << make_nvp("data", make_array(v.data(), v.RowsAtCompileTime));
53}
54
55template <class Archive, int D>
56void load(Archive& ar, Eigen::Matrix<double, D, 1>& v,
57 unsigned int /*version*/) {
58 ar >> make_nvp("data", make_array(v.data(), v.RowsAtCompileTime));
59}

61} // namespace serialization
