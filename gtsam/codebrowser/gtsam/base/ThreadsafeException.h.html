38/// Base exception type that uses tbb_allocator if GTSAM is compiled with TBB.
39template<class DERIVED>
40class ThreadsafeException:
41public std::exception
42{
43private:
44 typedef std::exception Base;
45#ifdef GTSAM_USE_TBB
46protected:
47 typedef std::basic_string<char, std::char_traits<char>,
48 tbb::tbb_allocator<char> > String;
49 typedef tbb::tbb_allocator<char> Allocator;
50#else
51protected:
52 typedef std::string String;
53 typedef std::allocator<char> Allocator;
54#endif
55
56protected:
57 bool dynamic_; ///< Whether this object was moved
58 mutable std::optional<String> description_; ///< Optional description
59
60 /// Default constructor is protected - may only be created from derived classes
61 ThreadsafeException() :
62 dynamic_(false) {
63 }
64
65 /// Copy constructor is protected - may only be created from derived classes
66 ThreadsafeException(const ThreadsafeException& other) :
67 Base(other), dynamic_(false) {
68 }
69
70 /// Construct with description string
71 ThreadsafeException(const std::string& description) :
72 dynamic_(false), description_(
73 String(description.begin(), description.end())) {
74 }
75
76 /// Default destructor doesn't have the noexcept
77 ~ThreadsafeException() noexcept override {
78 }
79
80public:
81 const char* what() const noexcept override {
82 return description_ ? description_->c_str() : "";
83 }
84};
