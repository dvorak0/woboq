19#pragma once
20
21#include <gtsam/base/FastDefaultAllocator.h>
22#if GTSAM_ENABLE_BOOST_SERIALIZATION
23#include <boost/serialization/nvp.hpp>
24#include <boost/serialization/map.hpp>
25#endif
26#include <map>

28namespace gtsam {
29
30/**
31 * FastMap is a thin wrapper around std::map that uses the boost
32 * fast_pool_allocator instead of the default STL allocator. This is just a
33 * convenience to avoid having lengthy types in the code. Through timing,
34 * we've seen that the fast_pool_allocator can lead to speedups of several
35 * percent.
36 * @ingroup base
37 */
38template<typename KEY, typename VALUE>
39class FastMap : public std::map<KEY, VALUE, std::less<KEY>,
40 typename internal::FastDefaultAllocator<std::pair<const KEY, VALUE> >::type> {
41
42public:
43
44 typedef std::map<KEY, VALUE, std::less<KEY>,
45 typename internal::FastDefaultAllocator<std::pair<const KEY, VALUE> >::type > Base;
46
47 /** Default constructor */
48 FastMap() {}
49
50 /** Constructor from a range, passes through to base class */
51 template<typename INPUTITERATOR>
52 explicit FastMap(INPUTITERATOR first, INPUTITERATOR last) : Base(first, last) {}
53
54 /** Copy constructor from another FastMap */
55 FastMap(const FastMap<KEY,VALUE>& x) : Base(x) {}
56
57 FastMap& operator=(const FastMap<KEY,VALUE>& x) = default;
58
59 /** Copy constructor from the base map class */
60 FastMap(const Base& x) : Base(x) {}
61
62 /** Conversion to a standard STL container */
63 operator std::map<KEY,VALUE>() const {
64 return std::map<KEY,VALUE>(this->begin(), this->end());
65 }
66
67 /** Handy 'insert' function for Matlab wrapper */
68 bool insert2(const KEY& key, const VALUE& val) { return Base::insert({key, val}).second; }
69
70 /** Handy 'exists' function */
71 bool exists(const KEY& e) const { return this->find(e) != this->end(); }
72
73private:
74#if GTSAM_ENABLE_BOOST_SERIALIZATION
75 /** Serialization function */
76 friend class boost::serialization::access;
77 template<class ARCHIVE>
78 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
79 ar & BOOST_SERIALIZATION_BASE_OBJECT_NVP(Base);
80 }
81#endif
82};
83
84}
