42 /**
43 * This class stores a dense matrix and allows it to be accessed as a collection of blocks. When
44 * constructed, the caller must provide the dimensions of the blocks.
45 *
46 * The block structure is symmetric, but the underlying matrix does not necessarily need to be.
47 *
48 * This class also has a parameter that can be changed after construction to change the apparent
49 * matrix view. firstBlock() determines the block that appears to have index 0 for all operations
50 * (except re-setting firstBlock()).
51 *
52 * @ingroup base */
53 class GTSAM_EXPORT SymmetricBlockMatrix
54 {
55 public:
56 typedef SymmetricBlockMatrix This;
57 typedef Eigen::Block<Matrix> Block;
58 typedef Eigen::Block<const Matrix> constBlock;
59
60 protected:
61 Matrix matrix_; ///< The full matrix
62 FastVector<DenseIndex> variableColOffsets_; ///< the starting columns of each block (0-based)
63
64 DenseIndex blockStart_; ///< Changes apparent matrix view, see main class comment.
65
66 public:
67 /// Construct from an empty matrix (asserts that the matrix is empty)
68 SymmetricBlockMatrix();
69
70 /// Construct from a container of the sizes of each block.
71 template<typename CONTAINER>
72 SymmetricBlockMatrix(const CONTAINER& dimensions, bool appendOneDimension = false) :
73 blockStart_(0)
74 {
75 fillOffsets(dimensions.begin(), dimensions.end(), appendOneDimension);
76 matrix_.resize(rows: variableColOffsets_.back(), cols: variableColOffsets_.back());
77 assertInvariants();
78 }
79
80 /// Construct from iterator over the sizes of each vertical block.
81 template<typename ITERATOR>
82 SymmetricBlockMatrix(ITERATOR firstBlockDim, ITERATOR lastBlockDim, bool appendOneDimension = false) :
83 blockStart_(0)
84 {
85 fillOffsets(firstBlockDim, lastBlockDim, appendOneDimension);
86 matrix_.resize(rows: variableColOffsets_.back(), cols: variableColOffsets_.back());
87 assertInvariants();
88 }
89
90 /// Construct from a container of the sizes of each vertical block and a pre-prepared matrix.
91 template<typename CONTAINER>
92 SymmetricBlockMatrix(const CONTAINER& dimensions, const Matrix& matrix, bool appendOneDimension = false) :
93 blockStart_(0)
94 {
95 matrix_.resize(rows: matrix.rows(), cols: matrix.cols());
96 matrix_.triangularView<Eigen::Upper>() = matrix.triangularView<Eigen::Upper>();
97 fillOffsets(dimensions.begin(), dimensions.end(), appendOneDimension);
98 if(matrix_.rows() != matrix_.cols())
99 throw std::invalid_argument("Requested to create a SymmetricBlockMatrix from a non-square matrix.");
100 if(variableColOffsets_.back() != matrix_.cols())
101 throw std::invalid_argument("Requested to create a SymmetricBlockMatrix with dimensions that do not sum to the total size of the provided matrix.");
102 assertInvariants();
103 }
104
105 /// Copy the block structure, but do not copy the matrix data. If blockStart() has been
106 /// modified, this copies the structure of the corresponding matrix view. In the destination
107 /// SymmetricBlockMatrix, blockStart() will be 0.
108 static SymmetricBlockMatrix LikeActiveViewOf(const SymmetricBlockMatrix& other);
109
110 /// Copy the block structure, but do not copy the matrix data. If blockStart() has been
111 /// modified, this copies the structure of the corresponding matrix view. In the destination
112 /// SymmetricBlockMatrix, blockStart() will be 0.
113 static SymmetricBlockMatrix LikeActiveViewOf(const VerticalBlockMatrix& other);
114
115 /// Row size
116 DenseIndex rows() const { assertInvariants(); return variableColOffsets_.back() - variableColOffsets_[blockStart_]; }
117
118 /// Column size
119 DenseIndex cols() const { return rows(); }
120
121 /// Block count
122 DenseIndex nBlocks() const { return nActualBlocks() - blockStart_; }
123
124 /// Number of dimensions for variable on this diagonal block.
125 DenseIndex getDim(DenseIndex block) const {
126 return calcIndices(iBlock: block, jBlock: block, blockRows: 1, blockCols: 1)[2];
127 }

129 /// @name Block getter methods.
130 /// @{
131
132 /// Get a copy of a block (anywhere in the matrix).
133 /// This method makes a copy - use the methods below if performance is critical.
134 Matrix block(DenseIndex I, DenseIndex J) const;
135
136 /// Return the J'th diagonal block as a self adjoint view.
137 Eigen::SelfAdjointView<Block, Eigen::Upper> diagonalBlock(DenseIndex J) {
138 return block_(iBlock: J, jBlock: J).selfadjointView<Eigen::Upper>();
139 }
140
141 /// Return the J'th diagonal block as a self adjoint view.
142 Eigen::SelfAdjointView<constBlock, Eigen::Upper> diagonalBlock(DenseIndex J) const {
143 return block_(iBlock: J, jBlock: J).selfadjointView<Eigen::Upper>();
144 }
145
146 /// Get the diagonal of the J'th diagonal block.
147 Vector diagonal(DenseIndex J) const {
148 return block_(iBlock: J, jBlock: J).diagonal();
149 }
150
151 /// Get block above the diagonal (I, J).
152 constBlock aboveDiagonalBlock(DenseIndex I, DenseIndex J) const {
153 assert(I < J);
154 return block_(iBlock: I, jBlock: J);
155 }
156
157 /// Return the square sub-matrix that contains blocks(i:j, i:j).
158 Eigen::SelfAdjointView<constBlock, Eigen::Upper> selfadjointView(
159 DenseIndex I, DenseIndex J) const {
160 assert(J > I);
161 return block_(iBlock: I, jBlock: I, blockRows: J - I, blockCols: J - I).selfadjointView<Eigen::Upper>();
162 }
163
164 /// Return the square sub-matrix that contains blocks(i:j, i:j) as a triangular view.
165 Eigen::TriangularView<constBlock, Eigen::Upper> triangularView(DenseIndex I,
166 DenseIndex J) const {
167 assert(J > I);
168 return block_(iBlock: I, jBlock: I, blockRows: J - I, blockCols: J - I).triangularView<Eigen::Upper>();
169 }
170
171 /// Get a range [i,j) from the matrix. Indices are in block units.
172 constBlock aboveDiagonalRange(DenseIndex i_startBlock,
173 DenseIndex i_endBlock,
174 DenseIndex j_startBlock,
175 DenseIndex j_endBlock) const {
176 assert(i_startBlock < j_startBlock);
177 assert(i_endBlock <= j_startBlock);
178 return block_(iBlock: i_startBlock, jBlock: j_startBlock, blockRows: i_endBlock - i_startBlock,
179 blockCols: j_endBlock - j_startBlock);
180 }
181
182 /// Get a range [i,j) from the matrix. Indices are in block units.
183 Block aboveDiagonalRange(DenseIndex i_startBlock, DenseIndex i_endBlock,
184 DenseIndex j_startBlock, DenseIndex j_endBlock) {
185 assert(i_startBlock < j_startBlock);
186 assert(i_endBlock <= j_startBlock);
187 return block_(iBlock: i_startBlock, jBlock: j_startBlock, blockRows: i_endBlock - i_startBlock,
188 blockCols: j_endBlock - j_startBlock);
189 }

191 /// @}
192 /// @name Block setter methods.
193 /// @{
194
195 /// Set a diagonal block. Only the upper triangular portion of `xpr` is evaluated.
196 template <typename XprType>
197 void setDiagonalBlock(DenseIndex I, const XprType& xpr) {
198 block_(iBlock: I, jBlock: I).triangularView<Eigen::Upper>() = xpr.template triangularView<Eigen::Upper>();
199 }
200
201 /// Set an off-diagonal block. Only the upper triangular portion of `xpr` is evaluated.
202 template <typename XprType>
203 void setOffDiagonalBlock(DenseIndex I, DenseIndex J, const XprType& xpr) {
204 assert(I != J);
205 if (I < J) {
206 block_(iBlock: I, jBlock: J) = xpr;
207 } else {
208 block_(iBlock: J, jBlock: I) = xpr.transpose();
209 }
210 }
211
212 /// Increment the diagonal block by the values in `xpr`. Only reads the upper triangular part of `xpr`.
213 template <typename XprType>
214 void updateDiagonalBlock(DenseIndex I, const XprType& xpr) {
215 // TODO(gareth): Eigen won't let us add triangular or self-adjoint views
216 // here, so we do it manually.
217 auto dest = block_(iBlock: I, jBlock: I);
218 assert(dest.rows() == xpr.rows());
219 assert(dest.cols() == xpr.cols());
220 for (DenseIndex col = 0; col < dest.cols(); ++col) {
221 for (DenseIndex row = 0; row <= col; ++row) {
222 dest(row, col) += xpr(row, col);
223 }
224 }
225 }
226
227 /// Update an off diagonal block.
228 /// NOTE(emmett): This assumes noalias().
229 template <typename XprType>
230 void updateOffDiagonalBlock(DenseIndex I, DenseIndex J, const XprType& xpr) {
231 assert(I != J);
232 if (I < J) {
233 block_(iBlock: I, jBlock: J).noalias() += xpr;
234 } else {
235 block_(iBlock: J, jBlock: I).noalias() += xpr.transpose();
236 }
237 }

277 /// Retrieve or modify the first logical block, i.e. the block referenced by block index 0.
278 /// Blocks before it will be inaccessible, except by accessing the underlying matrix using
279 /// matrix().
280 DenseIndex& blockStart() { return blockStart_; }

283 /// it will be inaccessible, except by accessing the underlying matrix using matrix().
284 DenseIndex blockStart() const { return blockStart_; }

286 /**
287 * Given the augmented Hessian [A1'A1 A1'A2 A1'b
288 * A2'A1 A2'A2 A2'b
289 * b'A1 b'A2 b'b]
290 * on x1 and x2, does partial Cholesky in-place to obtain [R Sd;0 L] such that
291 * R'R = A1'A1
292 * R'Sd = [A1'A2 A1'b]
293 * L'L is the augmented Hessian on the the separator x2
294 * R and Sd can be interpreted as a GaussianConditional |R*x1 + S*x2 - d]^2
295 */
296 void choleskyPartial(DenseIndex nFrontals);
297
298 /**
299 * After partial Cholesky, we can optionally split off R and Sd, to be interpreted as
300 * a GaussianConditional |R*x1 + S*x2 - d]^2. We leave the symmetric lower block L in place,
301 * and adjust block_start so now *this refers to it.
302 */
303 VerticalBlockMatrix split(DenseIndex nFrontals);

307 /// Number of offsets in the full matrix.
308 DenseIndex nOffsets() const {
309 return variableColOffsets_.size();
310 }
311
312 /// Number of actual blocks in the full matrix.
313 DenseIndex nActualBlocks() const {
314 return nOffsets() - 1;
315 }
316
317 /// Get an offset for a block index (in the active view).
318 DenseIndex offset(DenseIndex block) const {
319 assert(block >= 0);
320 const DenseIndex actual_index = block + blockStart();
321 assert(actual_index < nOffsets());
322 return variableColOffsets_[actual_index];
323 }
324
325 /// Get an arbitrary block from the matrix. Indices are in block units.
326 constBlock block_(DenseIndex iBlock, DenseIndex jBlock,
327 DenseIndex blockRows = 1, DenseIndex blockCols = 1) const {
328 const std::array<DenseIndex, 4> indices =
329 calcIndices(iBlock, jBlock, blockRows, blockCols);
330 return matrix_.block(startRow: indices[0], startCol: indices[1], blockRows: indices[2], blockCols: indices[3]);
331 }
332
333 /// Get an arbitrary block from the matrix. Indices are in block units.
334 Block block_(DenseIndex iBlock, DenseIndex jBlock, DenseIndex blockRows = 1,
335 DenseIndex blockCols = 1) {
336 const std::array<DenseIndex, 4> indices =
337 calcIndices(iBlock, jBlock, blockRows, blockCols);
338 return matrix_.block(startRow: indices[0], startCol: indices[1], blockRows: indices[2], blockCols: indices[3]);
339 }
340
341 /// Get the full matrix as a block.
342 constBlock full() const {
343 return block_(iBlock: 0, jBlock: 0, blockRows: nBlocks(), blockCols: nBlocks());
344 }
345
346 /// Get the full matrix as a block.
347 Block full() {
348 return block_(iBlock: 0, jBlock: 0, blockRows: nBlocks(), blockCols: nBlocks());
349 }
350
351 /// Compute the indices into the underlying matrix for a given block.
352 std::array<DenseIndex, 4> calcIndices(DenseIndex iBlock, DenseIndex jBlock,
353 DenseIndex blockRows,
354 DenseIndex blockCols) const {
355 assert(blockRows >= 0);
356 assert(blockCols >= 0);
357
358 // adjust indices to account for start and size of blocks
359 const DenseIndex denseI = offset(block: iBlock);
360 const DenseIndex denseJ = offset(block: jBlock);
361 const DenseIndex denseRows = offset(block: iBlock + blockRows) - denseI;
362 const DenseIndex denseCols = offset(block: jBlock + blockCols) - denseJ;
363 return {._M_elems: {denseI, denseJ, denseRows, denseCols}};
364 }
365
366 void assertInvariants() const
367 {
368 assert(matrix_.rows() == matrix_.cols());
369 assert(matrix_.cols() == variableColOffsets_.back());
370 assert(blockStart_ < (DenseIndex)variableColOffsets_.size());
371 }
372
373 template<typename ITERATOR>
374 void fillOffsets(ITERATOR firstBlockDim, ITERATOR lastBlockDim, bool appendOneDimension)
375 {
376 variableColOffsets_.resize((lastBlockDim-firstBlockDim) + 1 + (appendOneDimension ? 1 : 0));
377 variableColOffsets_[0] = 0;
378 DenseIndex j=0;
379 for(ITERATOR dim=firstBlockDim; dim!=lastBlockDim; ++dim) {
380 variableColOffsets_[j+1] = variableColOffsets_[j] + *dim;
381 ++ j;
382 }
383 if(appendOneDimension)
384 {
385 variableColOffsets_[j+1] = variableColOffsets_[j] + 1;
386 ++ j;
387 }
388 }

394#if GTSAM_ENABLE_BOOST_SERIALIZATION
395 /** Serialization function */
396 friend class boost::serialization::access;
397 template<class ARCHIVE>
398 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
399 // Fill in the lower triangle part of the matrix, so boost::serialization won't
400 // complain about uninitialized data with an input_stream_error exception
401 // http://www.boost.org/doc/libs/1_37_0/libs/serialization/doc/exceptions.html#stream_error
402 matrix_.triangularView<Eigen::Lower>() = matrix_.triangularView<Eigen::Upper>().transpose();
403 ar & BOOST_SERIALIZATION_NVP(matrix_);
404 ar & BOOST_SERIALIZATION_NVP(variableColOffsets_);
405 ar & BOOST_SERIALIZATION_NVP(blockStart_);
406 }
407#endif
