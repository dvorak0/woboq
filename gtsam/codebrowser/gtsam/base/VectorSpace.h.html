23/// VectorSpaceTraits Implementation for Fixed sizes
24template<class Class, int N>
25struct VectorSpaceImpl {
26
27 /// @name Manifold
28 /// @{
29 typedef Eigen::Matrix<double, N, 1> TangentVector;
30 typedef OptionalJacobian<N, N> ChartJacobian;
31 typedef Eigen::Matrix<double, N, N> Jacobian;
32 static int GetDimension(const Class&) { return N;}
33
34 static TangentVector Local(const Class& origin, const Class& other,
35 ChartJacobian H1 = {}, ChartJacobian H2 = {}) {
36 if (H1) *H1 = - Jacobian::Identity();
37 if (H2) *H2 = Jacobian::Identity();
38 Class v = other - origin;
39 return v.vector();
40 }
41
42 static Class Retract(const Class& origin, const TangentVector& v,
43 ChartJacobian H1 = {}, ChartJacobian H2 = {}) {
44 if (H1) *H1 = Jacobian::Identity();
45 if (H2) *H2 = Jacobian::Identity();
46 return origin + v;
47 }
48
49 /// @}
50
51 /// @name Lie Group
52 /// @{
53
54 typedef Eigen::Matrix<double, N, 1> LieAlgebra;
55
56 static TangentVector Logmap(const Class& m, ChartJacobian Hm = {}) {
57 if (Hm) *Hm = Jacobian::Identity();
58 return m.vector();
59 }
60
61 static Class Expmap(const TangentVector& v, ChartJacobian Hv = {}) {
62 if (Hv) *Hv = Jacobian::Identity();
63 return Class(v);
64 }
65
66 static Class Compose(const Class& v1, const Class& v2, ChartJacobian H1 = {},
67 ChartJacobian H2 = {}) {
68 if (H1) *H1 = Jacobian::Identity();
69 if (H2) *H2 = Jacobian::Identity();
70 return v1 + v2;
71 }
72
73 static Class Between(const Class& v1, const Class& v2, ChartJacobian H1 = {},
74 ChartJacobian H2 = {}) {
75 if (H1) *H1 = - Jacobian::Identity();
76 if (H2) *H2 = Jacobian::Identity();
77 return v2 - v1;
78 }
79
80 static Class Inverse(const Class& v, ChartJacobian H = {}) {
81 if (H) *H = - Jacobian::Identity();
82 return -v;
83 }
84
85 static LieAlgebra Hat(const TangentVector& v) { return v; }
86
87 static TangentVector Vee(const LieAlgebra& X) { return X; }
88
89 static Jacobian AdjointMap(const Class& /*m*/) {
90 return Jacobian::Identity();
91 }
92 /// @}
93};

95/// VectorSpaceTraits implementation for dynamic types.
96template<class Class>
97struct VectorSpaceImpl<Class,Eigen::Dynamic> {
98
99 /// @name Group
100 /// @{
101 static Class Compose(const Class& v1, const Class& v2) { return v1+v2;}
102 static Class Between(const Class& v1, const Class& v2) { return v2-v1;}
103 static Class Inverse(const Class& m) { return -m;}
104 /// @}
105
106 /// @name Manifold
107 /// @{
108 typedef Eigen::VectorXd TangentVector;
109 typedef OptionalJacobian<Eigen::Dynamic,Eigen::Dynamic> ChartJacobian;
110 static int GetDimension(const Class& m) { return m.dim();}
111
112 static Eigen::MatrixXd Eye(const Class& m) {
113 int dim = GetDimension(m);
114 return Eigen::MatrixXd::Identity(rows: dim, cols: dim);
115 }
116
117 static TangentVector Local(const Class& origin, const Class& other,
118 ChartJacobian H1 = {}, ChartJacobian H2 = {}) {
119 if (H1) *H1 = - Eye(m: origin);
120 if (H2) *H2 = Eye(m: other);
121 Class v = other - origin;
122 return v.vector();
123 }
124
125 static Class Retract(const Class& origin, const TangentVector& v,
126 ChartJacobian H1 = {}, ChartJacobian H2 = {}) {
127 if (H1) *H1 = Eye(m: origin);
128 if (H2) *H2 = Eye(m: origin);
129 return origin + v;
130 }
131
132 /// @}
133
134 /// @name Lie Group
135 /// @{
136
137 static TangentVector Logmap(const Class& m, ChartJacobian Hm = {}) {
138 if (Hm) *Hm = Eye(m);
139 return m.vector();
140 }
141
142 static Class Expmap(const TangentVector& v, ChartJacobian Hv = {}) {
143 Class result(v);
144 if (Hv) *Hv = Eye(m: v);
145 return result;
146 }
147
148 static Class Compose(const Class& v1, const Class& v2, ChartJacobian H1,
149 ChartJacobian H2 = {}) {
150 if (H1) *H1 = Eye(m: v1);
151 if (H2) *H2 = Eye(m: v2);
152 return v1 + v2;
153 }
154
155 static Class Between(const Class& v1, const Class& v2, ChartJacobian H1,
156 ChartJacobian H2 = {}) {
157 if (H1) *H1 = - Eye(m: v1);
158 if (H2) *H2 = Eye(m: v2);
159 return v2 - v1;
160 }
161
162 static Class Inverse(const Class& v, ChartJacobian H) {
163 if (H) *H = -Eye(m: v);
164 return -v;
165 }
166
167 static Eigen::MatrixXd AdjointMap(const Class& m) { return Eye(m); }
168 /// @}
169};

171/// Requirements on type to pass it to Manifold template below
172template<class Class>
173struct HasVectorSpacePrereqs {
174
175 inline constexpr static auto dim = Class::dimension;
176
177 Class p, q;
178 Vector v;
179
180 GTSAM_CONCEPT_USAGE(HasVectorSpacePrereqs) {
181 p = Class::Identity(); // identity
182 q = p + p; // addition
183 q = p - p; // subtraction
184 v = p.vector(); // conversion to vector
185 q = p + v; // addition of a vector on the right
186 }
187};

189/// A helper that implements the traits interface for *classes* that define vector spaces
190/// To use this for your class, define:
191/// template<> struct traits<Class> : public VectorSpaceTraits<Class> {};
192/// The class needs to support the requirements defined by HasVectorSpacePrereqs above
193template<class Class>
194struct VectorSpaceTraits: VectorSpaceImpl<Class, Class::dimension> {
195
196 // Check that Class has the necessary machinery
197GTSAM_CONCEPT_ASSERT(HasVectorSpacePrereqs<Class>);
198
199 typedef vector_space_tag structure_category;
200
201 /// @name Group
202 /// @{
203 typedef additive_group_tag group_flavor;
204 static Class Identity() { return Class::Identity();}
205 /// @}
206
207 /// @name Manifold
208 /// @{
209 inline constexpr static auto dimension = Class::dimension;
210 typedef Class ManifoldType;
211 /// @}
212};

218/// A helper that implements the traits interface for scalar vector spaces. Usage:
219/// template<> struct traits<Type> : public ScalarTraits<Type> { };
220template<typename Scalar>
221struct ScalarTraits : VectorSpaceImpl<Scalar, 1> {
222
223 typedef vector_space_tag structure_category;
224
225 /// @name Testable
226 /// @{
227 static void Print(Scalar m, const std::string& str = "") {
228 gtsam::print(m, str);
229 }
230 static bool Equals(Scalar v1, Scalar v2, double tol = 1e-8) {
231 return std::abs(v1 - v2) < tol;
232 }
233 /// @}
234
235 /// @name Group
236 /// @{
237 typedef additive_group_tag group_flavor;
238 static Scalar Identity() { return 0;}
239 /// @}
240
241 /// @name Manifold
242 /// @{
243 typedef Scalar ManifoldType;
244 inline constexpr static auto dimension = 1;
245 typedef Eigen::Matrix<double, 1, 1> TangentVector;
246 typedef OptionalJacobian<1, 1> ChartJacobian;
247
248 static TangentVector Local(Scalar origin, Scalar other,
249 ChartJacobian H1 = {}, ChartJacobian H2 = {}) {
250 if (H1) (*H1)[0] = -1.0;
251 if (H2) (*H2)[0] = 1.0;
252 TangentVector result;
253 result(0) = other - origin;
254 return result;
255 }
256
257 static Scalar Retract(Scalar origin, const TangentVector& v,
258 ChartJacobian H1 = {}, ChartJacobian H2 = {}) {
259 if (H1) (*H1)[0] = 1.0;
260 if (H2) (*H2)[0] = 1.0;
261 return origin + v[0];
262 }
263 /// @}
264
265 /// @name Lie Group
266 /// @{
267 static TangentVector Logmap(Scalar m, ChartJacobian H = {}) {
268 if (H) (*H)[0] = 1.0;
269 return Local(origin: 0, other: m);
270 }
271
272 static Scalar Expmap(const TangentVector& v, ChartJacobian H = {}) {
273 if (H) (*H)[0] = 1.0;
274 return v[0];
275 }
276 // AdjointMap for ScalarTraits is inherited from VectorSpaceImpl<Scalar, 1>
277 /// @}
278};

290// traits for any fixed double Eigen matrix
291template<int M, int N, int Options, int MaxRows, int MaxCols>
292struct traits<Eigen::Matrix<double, M, N, Options, MaxRows, MaxCols> > :
293 internal::VectorSpaceImpl<
294 Eigen::Matrix<double, M, N, Options, MaxRows, MaxCols>, M * N> {
295
296 typedef vector_space_tag structure_category;
297 typedef Eigen::Matrix<double, M, N, Options, MaxRows, MaxCols> Fixed;
298
299 /// @name Testable
300 /// @{
301 static void Print(const Fixed& m, const std::string& str = "") {
302 gtsam::print(A: Eigen::MatrixXd(m), s: str);
303 }
304 static bool Equals(const Fixed& v1, const Fixed& v2, double tol = 1e-8) {
305 return equal_with_abs_tol(v1, v2, tol);
306 }
307 /// @}
308
309 /// @name Group
310 /// @{
311 typedef additive_group_tag group_flavor;
312 static Fixed Identity() { return Fixed::Zero();}
313 /// @}
314
315 /// @name Manifold
316 /// @{
317 inline constexpr static auto dimension = M * N;
318 typedef Fixed ManifoldType;
319 typedef Eigen::Matrix<double, dimension, 1> TangentVector;
320 typedef Eigen::Matrix<double, dimension, dimension> Jacobian;
321 typedef OptionalJacobian<dimension, dimension> ChartJacobian;
322
323 static TangentVector Local(const Fixed& origin, const Fixed& other,
324 ChartJacobian H1 = {}, ChartJacobian H2 = {}) {
325 if (H1) (*H1) = -Jacobian::Identity();
326 if (H2) (*H2) = Jacobian::Identity();
327 TangentVector result;
328 Eigen::Map<Fixed>(result.data()) = other - origin;
329 return result;
330 }
331
332 static Fixed Retract(const Fixed& origin, const TangentVector& v,
333 ChartJacobian H1 = {}, ChartJacobian H2 = {}) {
334 if (H1) (*H1) = Jacobian::Identity();
335 if (H2) (*H2) = Jacobian::Identity();
336 return origin + Eigen::Map<const Fixed>(v.data());
337 }
338 /// @}
339
340 /// @name Lie Group
341 /// @{
342 static TangentVector Logmap(const Fixed& m, ChartJacobian H = {}) {
343 if (H) *H = Jacobian::Identity();
344 TangentVector result;
345 Eigen::Map<Fixed>(result.data()) = m;
346 return result;
347 }
348
349 static Fixed Expmap(const TangentVector& v, ChartJacobian H = {}) {
350 Fixed m;
351 m.setZero();
352 if (H) *H = Jacobian::Identity();
353 return m + Eigen::Map<const Fixed>(v.data());
354 }
355
356 // AdjointMap for fixed-size Eigen matrices is inherited from
357 // internal::VectorSpaceImpl< Eigen::Matrix<double, M, N, ...> , M*N >
358 /// @}
359};

364// traits for dynamic Eigen matrices
365template<int M, int N, int Options, int MaxRows, int MaxCols>
366struct DynamicTraits {
367
368 typedef vector_space_tag structure_category;
369 typedef Eigen::Matrix<double, M, N, Options, MaxRows, MaxCols> Dynamic;
370
371 /// @name Testable
372 /// @{
373 static void Print(const Dynamic& m, const std::string& str = "") {
374 gtsam::print(A: Eigen::MatrixXd(m), s: str);
375 }
376 static bool Equals(const Dynamic& v1, const Dynamic& v2,
377 double tol = 1e-8) {
378 return equal_with_abs_tol(v1, v2, tol);
379 }
380 /// @}
381
382 /// @name Group
383 /// @{
384 typedef additive_group_tag group_flavor;
385 static Dynamic Identity() {
386 throw std::runtime_error("Identity not defined for dynamic types");
387 }
388 /// @}
389
390 /// @name Manifold
391 /// @{
392 inline constexpr static auto dimension = Eigen::Dynamic;
393 typedef Eigen::VectorXd TangentVector;
394 typedef Eigen::MatrixXd Jacobian;
395 typedef OptionalJacobian<dimension, dimension> ChartJacobian;
396 typedef Dynamic ManifoldType;
397
398 static int GetDimension(const Dynamic& m) {
399 return m.rows() * m.cols();
400 }
401
402 static Jacobian Eye(const Dynamic& m) {
403 int dim = GetDimension(m);
404 return Eigen::Matrix<double, dimension, dimension>::Identity(rows: dim, cols: dim);
405 }
406
407 static TangentVector Local(const Dynamic& m, const Dynamic& other, //
408 ChartJacobian H1 = {}, ChartJacobian H2 = {}) {
409 if (H1) *H1 = -Eye(m);
410 if (H2) *H2 = Eye(m);
411 TangentVector v(GetDimension(m));
412 Eigen::Map<Dynamic>(v.data(), m.rows(), m.cols()) = other - m;
413 return v;
414 }
415
416 static Dynamic Retract(const Dynamic& m, const TangentVector& v, //
417 ChartJacobian H1 = {}, ChartJacobian H2 = {}) {
418 if (H1) *H1 = Eye(m);
419 if (H2) *H2 = Eye(m);
420 return m + Eigen::Map<const Dynamic>(v.data(), m.rows(), m.cols());
421 }
422 /// @}
423
424 /// @name Lie Group
425 /// @{
426 using LieAlgebra = Dynamic;
427
428 static TangentVector Logmap(const Dynamic& m, ChartJacobian H = {}) {
429 if (H) *H = Eye(m);
430 TangentVector result(GetDimension(m));
431 Eigen::Map<Dynamic>(result.data(), m.rows(), m.cols()) = m;
432 return result;
433 }
434
435 static Dynamic Expmap(const TangentVector& /*v*/, ChartJacobian H = {}) {
436 static_cast<void>(H);
437 throw std::runtime_error("Expmap not defined for dynamic types");
438 }
439
440 static Dynamic Inverse(const Dynamic& m, ChartJacobian H = {}) {
441 if (H) *H = -Eye(m);
442 return -m;
443 }
444
445 static Dynamic Compose(const Dynamic& v1, const Dynamic& v2,
446 ChartJacobian H1 = {}, ChartJacobian H2 = {}) {
447 if (H1) *H1 = Eye(m: v1);
448 if (H2) *H2 = Eye(m: v1);
449 return v1 + v2;
450 }
451
452 static Dynamic Between(const Dynamic& v1, const Dynamic& v2,
453 ChartJacobian H1 = {}, ChartJacobian H2 = {}) {
454 if (H1) *H1 = -Eye(m: v1);
455 if (H2) *H2 = Eye(m: v1);
456 return v2 - v1;
457 }
458
459 static LieAlgebra Hat(const TangentVector& v) {
460 return v;
461 }
462
463 static TangentVector Vee(const LieAlgebra& X) {
464 return X;
465 }
466
467 static Jacobian AdjointMap(const Dynamic& m) { return Eye(m); }
468 /// @}
469
470};

492/// Vector Space concept
493template<typename T>
494class IsVectorSpace: public IsLieGroup<T> {
495public:
496
497 typedef typename traits<T>::structure_category structure_category_tag;
498
499 GTSAM_CONCEPT_USAGE(IsVectorSpace) {
500 static_assert(
501 (std::is_base_of<vector_space_tag, structure_category_tag>::value),
502 "This type's trait does not assert it as a vector space (or derived)");
503 r = p + q;
504 r = -p;
505 r = p - q;
506 }
507
508private:
509 T p, q, r;
510};
