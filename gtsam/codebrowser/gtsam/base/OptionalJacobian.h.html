31/**
32 * OptionalJacobian is an Eigen::Ref like class that can take be constructed using
33 * either a fixed size or dynamic Eigen matrix. In the latter case, the dynamic
34 * matrix will be resized.
35 * Below this class, a dynamic version is also implemented.
36 */
37template<int Rows, int Cols>
38class OptionalJacobian {
39
40public:
41
42 /// Jacobian size type
43 /// TODO(frank): how to enforce RowMajor? Or better, make it work with any storage order?
44 typedef Eigen::Matrix<double, Rows, Cols> Jacobian;
45
46private:
47
48 Eigen::Map<Jacobian> map_; /// View on constructor argument, if given
49
50 // Trick from http://eigen.tuxfamily.org/dox/group__TutorialMapClass.html
51 // uses "placement new" to make map_ usurp the memory of the fixed size matrix
52 void usurp(double* data) {
53 new (&map_) Eigen::Map<Jacobian>(data);
54 }
55
56 // Private and very dangerous constructor straight from memory
57 OptionalJacobian(double* data) : map_(nullptr) {
58 if (data) usurp(data);
59 }
60
61 template<int M, int N>
62 friend class OptionalJacobian;
63
64public:
65
66 /// Default constructor
67 OptionalJacobian() :
68 map_(nullptr) {
69 }
70
71 /// Default constructor with nullptr_t
72 /// To guide the compiler when nullptr
73 /// is passed to args of the type OptionalJacobian
74 OptionalJacobian(std::nullptr_t /*unused*/) :
75 map_(nullptr) {
76 }
77
78 /// Constructor that will usurp data of a fixed-size matrix
79 OptionalJacobian(Jacobian& fixed) :
80 map_(nullptr) {
81 usurp(data: fixed.data());
82 }
83
84 /// Constructor that will usurp data of a fixed-size matrix, pointer version
85 OptionalJacobian(Jacobian* fixedPtr) :
86 map_(nullptr) {
87 if (fixedPtr)
88 usurp(data: fixedPtr->data());
89 }
90
91 /// Constructor that will resize a dynamic matrix (unless already correct)
92 OptionalJacobian(Eigen::MatrixXd& dynamic) :
93 map_(nullptr) {
94 dynamic.resize(rows: Rows, cols: Cols); // no malloc if correct size
95 usurp(data: dynamic.data());
96 }
97
98 /// Constructor that will resize a dynamic matrix (unless already correct)
99 OptionalJacobian(Eigen::MatrixXd* dynamic) :
100 map_(nullptr) {
101 if (dynamic) {
102 dynamic->resize(rows: Rows, cols: Cols); // no malloc if correct size
103 usurp(data: dynamic->data());
104 }
105 }
106
107 /**
108 * @brief Constructor from an Eigen::Ref *value*. Will not usurp if dimension is wrong
109 * @note This is important so we don't overwrite someone else's memory!
110 */
111 template<class MATRIX>
112 OptionalJacobian(Eigen::Ref<MATRIX> dynamic_ref) :
113 map_(nullptr) {
114 if (dynamic_ref.rows() == Rows && dynamic_ref.cols() == Cols && !dynamic_ref.IsRowMajor) {
115 usurp(data: dynamic_ref.data());
116 } else {
117 throw std::invalid_argument(
118 std::string("OptionalJacobian called with wrong dimensions or "
119 "storage order.\n"
120 "Expected: ") +
121 "(" + std::to_string(val: Rows) + ", " + std::to_string(val: Cols) + ")");
122 }
123 }
124
125 /// Constructor with std::nullopt just makes empty
126 OptionalJacobian(std::nullopt_t /*none*/) :
127 map_(nullptr) {
128 }
129
130 /// Constructor compatible with old-style derivatives
131 OptionalJacobian(const std::optional<std::reference_wrapper<Eigen::MatrixXd>> optional) :
132 map_(nullptr) {
133 if (optional) {
134 optional->get().resize(rows: Rows, cols: Cols);
135 usurp(data: optional->get().data());
136 }
137 }
138 /// Constructor that will usurp data of a block expression
139 /// TODO(frank): unfortunately using a Map makes usurping non-contiguous memory impossible
140 // template <typename Derived, bool InnerPanel>
141 // OptionalJacobian(Eigen::Block<Derived,Rows,Cols,InnerPanel> block) : map_(nullptr) { ?? }
142
143 /// Return true if allocated, false if default constructor was used
144 operator bool() const {
145 return map_.data() != nullptr;
146 }
147
148 /// De-reference, like boost optional
149 Eigen::Map<Jacobian>& operator*() {
150 return map_;
151 }
152
153 /// operator->()
154 Eigen::Map<Jacobian>* operator->() {
155 return &map_;
156 }
157
158 /// Access M*N sub-block if we are allocated, otherwise none
159 /// TODO(frank): this could work as is below if only constructor above worked
160 // template <int M, int N>
161 // OptionalJacobian<M, N> block(int startRow, int startCol) {
162 // if (*this)
163 // OptionalJacobian<M, N>(map_.block<M, N>(startRow, startCol));
164 // else
165 // return OptionalJacobian<M, N>();
166 // }
167
168 /// Access Rows*N sub-block if we are allocated, otherwise return an empty OptionalJacobian
169 /// The use case is functions with arguments that are dissected, e.g. Pose3 into Rot3, Point3
170 /// TODO(frank): ideally, we'd like full block functionality, but see note above.
171 template <int N>
172 OptionalJacobian<Rows, N> cols(int startCol) {
173 if (*this)
174 return OptionalJacobian<Rows, N>(&map_(0,startCol));
175 else
176 return OptionalJacobian<Rows, N>();
177 }
178
179 /// Access M*Cols sub-block if we are allocated, otherwise return empty OptionalJacobian
180 /// The use case is functions that create their return value piecemeal by calling other functions
181 /// TODO(frank): Unfortunately we assume column-major storage order and hence this can't work
182 /// template <int M> OptionalJacobian<M, Cols> rows(int startRow) { ?? }
183};
184
185// The pure dynamic specialization of this is needed to support
186// variable-sized types. Note that this is designed to work like the
187// boost optional scheme from GTSAM 3.
188template<>
189class OptionalJacobian<Eigen::Dynamic, Eigen::Dynamic> {
190
191public:
192
193 /// Jacobian size type
194 typedef Eigen::MatrixXd Jacobian;
195
196private:
197
198 Jacobian* pointer_; /// View on constructor argument, if given
199
200public:
201
202 /// Default constructor
203 OptionalJacobian() :
204 pointer_(nullptr) {
205 }
206
207 /// Construct from pointer to dynamic matrix
208 OptionalJacobian(Jacobian* pointer) : pointer_(pointer) {}
209
210 /// Construct from refrence to dynamic matrix
211 OptionalJacobian(Jacobian& dynamic) : pointer_(&dynamic) {}
212
213 /// Constructor with std::nullopt just makes empty
214 OptionalJacobian(std::nullopt_t /*none*/) :
215 pointer_(nullptr) {
216 }
217
218 /// Constructor for optional matrix reference
219 OptionalJacobian(const std::optional<std::reference_wrapper<Eigen::MatrixXd>> optional) :
220 pointer_(nullptr) {
221 if (optional) pointer_ = &((*optional).get());
222 }
223
224 /// Return true if allocated, false if default constructor was used
225 operator bool() const {
226 return pointer_!=nullptr;
227 }
228
229 /// De-reference, like boost optional
230 Jacobian& operator*() {
231 return *pointer_;
232 }
233
234 /// TODO: operator->()
235 Jacobian* operator->(){ return pointer_; }
236};

238// forward declare
239template <typename T> struct traits;
240
241/**
242 * @brief: meta-function to generate Jacobian
243 * @param T return type
244 * @param A argument type
245 */
246template <class T, class A>
247struct MakeJacobian {
248 typedef Eigen::Matrix<double, traits<T>::dimension, traits<A>::dimension> type;
249};
250
251/**
252 * @brief: meta-function to generate JacobianTA optional reference
253 * Used mainly by Expressions
254 * @param T return type
255 * @param A argument type
256 */
257template<class T, class A>
258struct MakeOptionalJacobian {
259 typedef OptionalJacobian<traits<T>::dimension,
260 traits<A>::dimension> type;
261};
