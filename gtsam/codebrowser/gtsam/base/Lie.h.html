33/// A CRTP helper class that implements Lie group methods
34/// Prerequisites: methods operator*, inverse, and AdjointMap, as well as a
35/// ChartAtOrigin struct that will be used to define the manifold Chart
36/// To use, simply derive, but also say "using LieGroup<Class,N>::inverse"
37/// For derivative math, see doc/math.pdf
38template <class Class, int N>
39struct LieGroup {
40
41 inline constexpr static auto dimension = N;
42 typedef OptionalJacobian<N, N> ChartJacobian;
43 typedef Eigen::Matrix<double, N, N> Jacobian;
44 typedef Eigen::Matrix<double, N, 1> TangentVector;
45
46 /// Static method to get the dimension (compile-time or dynamic)
47 static constexpr int Dim() { return N; }
48
49 /// Provided fixed dimension in dim() if needed
50 template <int M = N>
51 std::enable_if_t<M != Eigen::Dynamic, int> dim() const {
52 return N;
53 }
54
55 const Class & derived() const {
56 return static_cast<const Class&>(*this);
57 }
58
59 Class compose(const Class& g) const {
60 return derived() * g;
61 }
62
63 Class between(const Class& g) const {
64 return derived().inverse() * g;
65 }
66
67 Class compose(const Class& g, ChartJacobian H1,
68 ChartJacobian H2 = {}) const {
69 if (H1) *H1 = g.inverse().AdjointMap();
70 if (H2) *H2 = identityMatrix();
71 return derived() * g;
72 }
73
74 Class between(const Class& g, ChartJacobian H1,
75 ChartJacobian H2 = {}) const {
76 Class result = derived().inverse() * g;
77 if (H1) *H1 = - result.inverse().AdjointMap();
78 if (H2) *H2 = identityMatrix();
79 return result;
80 }
81
82 Class inverse(ChartJacobian H) const {
83 if (H) *H = - derived().AdjointMap();
84 return derived().inverse();
85 }
86
87 /// expmap as required by manifold concept
88 /// Applies exponential map to v and composes with *this
89 Class expmap(const TangentVector& v) const {
90 return compose(Class::Expmap(v));
91 }
92
93 /// logmap as required by manifold concept
94 /// Applies logarithmic map to group element that takes *this to g
95 TangentVector logmap(const Class& g) const {
96 return Class::Logmap(between(g));
97 }
98
99 /// expmap with optional derivatives
100 Class expmap(const TangentVector& v, //
101 ChartJacobian H1, ChartJacobian H2 = {}) const {
102 Jacobian D_g_v;
103 Class g = Class::Expmap(v,H2 ? &D_g_v : 0);
104 Class h = compose(g); // derivatives inlined below
105 if (H1) *H1 = g.inverse().AdjointMap();
106 if (H2) *H2 = D_g_v;
107 return h;
108 }
109
110 /// logmap with optional derivatives
111 TangentVector logmap(const Class& g, //
112 ChartJacobian H1, ChartJacobian H2 = {}) const {
113 Class h = between(g); // derivatives inlined below
114 Jacobian D_v_h;
115 TangentVector v = Class::Logmap(h, (H1 || H2) ? &D_v_h : 0);
116 if (H1) *H1 = - D_v_h * h.inverse().AdjointMap();
117 if (H2) *H2 = D_v_h;
118 return v;
119 }
120
121 /// Retract at origin: possible in Lie group because it has an identity
122 static Class Retract(const TangentVector& v) {
123 return Class::ChartAtOrigin::Retract(v);
124 }
125
126 /// LocalCoordinates at origin: possible in Lie group because it has an identity
127 static TangentVector LocalCoordinates(const Class& g) {
128 return Class::ChartAtOrigin::Local(g);
129 }
130
131 /// Retract at origin with optional derivative
132 static Class Retract(const TangentVector& v, ChartJacobian H) {
133 return Class::ChartAtOrigin::Retract(v,H);
134 }
135
136 /// LocalCoordinates at origin with optional derivative
137 static TangentVector LocalCoordinates(const Class& g, ChartJacobian H) {
138 return Class::ChartAtOrigin::Local(g,H);
139 }
140
141 /// retract as required by manifold concept: applies v at *this
142 Class retract(const TangentVector& v) const {
143 return compose(Class::ChartAtOrigin::Retract(v));
144 }
145
146 /// localCoordinates as required by manifold concept: finds tangent vector between *this and g
147 TangentVector localCoordinates(const Class& g) const {
148 return Class::ChartAtOrigin::Local(between(g));
149 }
150
151 /// retract with optional derivatives
152 Class retract(const TangentVector& v, //
153 ChartJacobian H1, ChartJacobian H2 = {}) const {
154 Jacobian D_g_v;
155 Class g = Class::ChartAtOrigin::Retract(v, H2 ? &D_g_v : 0);
156 Class h = compose(g); // derivatives inlined below
157 if (H1) *H1 = g.inverse().AdjointMap();
158 if (H2) *H2 = D_g_v;
159 return h;
160 }
161
162 /// localCoordinates with optional derivatives
163 TangentVector localCoordinates(const Class& g, //
164 ChartJacobian H1, ChartJacobian H2 = {}) const {
165 Class h = between(g); // derivatives inlined below
166 Jacobian D_v_h;
167 TangentVector v = Class::ChartAtOrigin::Local(h, (H1 || H2) ? &D_v_h : 0);
168 if (H1) *H1 = - D_v_h * h.inverse().AdjointMap();
169 if (H2) *H2 = D_v_h;
170 return v;
171 }
172
173 private:
174
175 // Helper to get identity matrix of correct size for static or dynamic N
176 Jacobian identityMatrix() const {
177 if constexpr (N == Eigen::Dynamic) {
178 return Jacobian::Identity(derived().dim(), derived().dim());
179 } else {
180 return Jacobian::Identity();
181 }
182 }
183};

190/// A helper class that implements the traits interface for GTSAM lie groups.
191/// To use this for your gtsam type, define:
192/// template<> struct traits<Class> : public internal::LieGroupTraits<Class> {};
193/// Assumes existence of: identity, dimension, localCoordinates, retract,
194/// and additionally Logmap, Expmap, AdjointMap, compose, between, and inverse
195template<class Class>
196struct LieGroupTraits : public GetDimensionImpl<Class, Class::dimension> {
197 using structure_category = lie_group_tag;
198
199 /// @name Group
200 /// @{
201 using group_flavor = multiplicative_group_tag;
202 static Class Identity() { return Class::Identity(); }
203 /// @}
204
205 /// @name Manifold
206 /// @{
207 using ManifoldType = Class;
208 // Note: Class::dimension can be an int or Eigen::Dynamic.
209 // GetDimensionImpl handles resolving this to a static value or providing GetDimension(obj).
210 inline constexpr static auto dimension = Class::dimension;
211 using TangentVector = Eigen::Matrix<double, dimension, 1>;
212 using ChartJacobian = OptionalJacobian<dimension, dimension>;
213
214 static TangentVector Local(const Class& origin, const Class& other,
215 ChartJacobian H1 = {}, ChartJacobian H2 = {}) {
216 return origin.localCoordinates(other, H1, H2);
217 }
218
219 static Class Retract(const Class& origin, const TangentVector& v,
220 ChartJacobian H = {}, ChartJacobian Hv = {}) {
221 return origin.retract(v, H, Hv);
222 }
223 /// @}
224
225 /// @name Lie Group
226 /// @{
227 static TangentVector Logmap(const Class& m, ChartJacobian Hm = {}) {
228 return Class::Logmap(m, Hm);
229 }
230
231 static Class Expmap(const TangentVector& v, ChartJacobian Hv = {}) {
232 return Class::Expmap(v, Hv);
233 }
234
235 static Class Compose(const Class& m1, const Class& m2, //
236 ChartJacobian H1 = {}, ChartJacobian H2 = {}) {
237 return m1.compose(m2, H1, H2);
238 }
239
240 static Class Between(const Class& m1, const Class& m2, //
241 ChartJacobian H1 = {}, ChartJacobian H2 = {}) {
242 return m1.between(m2, H1, H2);
243 }
244
245 static Class Inverse(const Class& m, //
246 ChartJacobian H = {}) {
247 return m.inverse(H);
248 }
249
250 static Eigen::Matrix<double, dimension, dimension> AdjointMap(const Class& m) {
251 // This assumes that the Class itself provides a member function `AdjointMap()`
252 // For dynamically-sized types (dimension == Eigen::Dynamic),
253 // m.AdjointMap() must return a gtsam::Matrix of the correct runtime dimensions.
254 return m.AdjointMap();
255 }
256 /// @}
257};

270/** Compute l0 s.t. l2=l1*l0 */
271template<class Class>
272inline Class between_default(const Class& l1, const Class& l2) {
273 return l1.inverse().compose(l2);
274}
275
276/** Log map centered at l0, s.t. exp(l0,log(l0,lp)) = lp */
277template<class Class>
278inline Vector logmap_default(const Class& l0, const Class& lp) {
279 return Class::Logmap(l0.between(lp));
280}
281
282/** Exponential map centered at l0, s.t. exp(t,d) = t*exp(d) */
283template<class Class>
284inline Class expmap_default(const Class& t, const Vector& d) {
285 return t.compose(Class::Expmap(d));
286}

288/**
289 * Lie Group Concept
290 */
291template<typename T>
292class IsLieGroup: public IsGroup<T>, public IsManifold<T> {
293public:
294 // Concept marker: allows checking IsLieGroup<T>::value in templates
295 static constexpr bool value =
296 std::is_base_of_v<lie_group_tag, typename traits<T>::structure_category>;
297
298 using structure_category_tag = typename traits<T>::structure_category;
299 using ManifoldType = typename traits<T>::ManifoldType;
300 using TangentVector = typename traits<T>::TangentVector;
301 using ChartJacobian = typename traits<T>::ChartJacobian;
302
303 GTSAM_CONCEPT_USAGE(IsLieGroup) {
304 static_assert(
305 value,
306 "This type's trait does not assert it is a Lie group (or derived)");
307
308 // group operations with Jacobians
309 g = traits<T>::Compose(g, h, Hg, Hh);
310 g = traits<T>::Between(g, h, Hg, Hh);
311 g = traits<T>::Inverse(g, Hg);
312 // log and exp map without Jacobians
313 g = traits<T>::Expmap(v);
314 v = traits<T>::Logmap(g);
315 // log and exponential map with Jacobians
316 g = traits<T>::Expmap(v, Hg);
317 v = traits<T>::Logmap(g, Hg);
318 // AdjointMap
319 *Hg = traits<T>::AdjointMap(g);
320 }
321private:
322 T g, h;
323 TangentVector v;
324 ChartJacobian Hg, Hh;
325};

327/**
328 * Linear interpolation between X and Y by coefficient t. Typically t \in [0,1],
329 * but can also be used to extrapolate before pose X or after pose Y.
330 */
331template <typename T>
332T interpolate(const T& X, const T& Y, double t,
333 typename MakeOptionalJacobian<T, T>::type Hx = {},
334 typename MakeOptionalJacobian<T, T>::type Hy = {},
335 typename MakeOptionalJacobian<T, double>::type Ht = {}) {
336 if (Hx || Hy || Ht) {
337 typename MakeJacobian<T, T>::type between_H_x, log_H, exp_H, compose_H_x;
338 const T between =
339 traits<T>::Between(X, Y, between_H_x); // between_H_y = identity
340 typename traits<T>::TangentVector delta = traits<T>::Logmap(between, log_H);
341 const T Delta = traits<T>::Expmap(t * delta, exp_H);
342 const T result = traits<T>::Compose(
343 X, Delta, compose_H_x); // compose_H_xinv_y = identity
344
345 if (Hx) *Hx = compose_H_x + t * exp_H * log_H * between_H_x;
346 if (Hy) *Hy = t * exp_H * log_H;
347 if (Ht) *Ht = delta;
348 return result;
349 }
350 return traits<T>::Compose(
351 X, traits<T>::Expmap(t * traits<T>::Logmap(traits<T>::Between(X, Y))));
352}

354/**
355 * Functor for transforming covariance of T.
356 * T needs to satisfy the Lie group concept.
357 */
358template<class T>
359class TransformCovariance
360{
361private:
362 typename T::Jacobian adjointMap_;
363public:
364 explicit TransformCovariance(const T &X) : adjointMap_{X.AdjointMap()} {}
365 typename T::Jacobian operator()(const typename T::Jacobian &covariance)
366 { return adjointMap_ * covariance * adjointMap_.transpose(); }
367};
