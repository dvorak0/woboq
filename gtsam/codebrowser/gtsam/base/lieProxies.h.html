14 * @brief Provides convenient mappings of common member functions for testing
15 * @author Alex Cunningham

18#pragma once
19
20#include <gtsam/global_includes.h>
21
22/**
23 * Global functions in a separate testing namespace
24 *
25 * These should not be used outside of tests, as they are just remappings
26 * of the original functions. We use these to avoid needing to do
27 * too much std::bind magic or writing a bunch of separate proxy functions.
28 *
29 * Don't expect all classes to work for all of these functions.
30 */
31namespace gtsam {
32namespace testing {
33
34 /** binary functions */
35 template<class T>
36 T between(const T& t1, const T& t2) { return t1.between(t2); }
37
38 template<class T>
39 T compose(const T& t1, const T& t2) { return t1.compose(t2); }
40
41 /** unary functions */
42 template<class T>
43 T inverse(const T& t) { return t.inverse(); }
44
45 /** rotation functions */
46 template<class T, class P>
47 P rotate(const T& r, const P& pt) { return r.rotate(pt); }
48
49 template<class T, class P>
50 P unrotate(const T& r, const P& pt) { return r.unrotate(pt); }
51
52} // \namespace testing
53} // \namespace gtsam
