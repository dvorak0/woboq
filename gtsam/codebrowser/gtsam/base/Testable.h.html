14 * @brief Concept check for values that can be used in unit tests
15 * @author Frank Dellaert
16 *
17 * The necessary functions to implement for Testable are defined
18 * below with additional details as to the interface.
19 * The concept checking function will check whether or not
20 * the function exists in derived class and throw compile-time errors.

22 * print with optional string naming the object
23 * void print(const std::string& name) const = 0;
24 *
25 * equality up to tolerance
26 * tricky to implement, see PriorFactor for an example

50 /**
51 * A testable concept check that should be placed in applicable unit
52 * tests and in generic algorithms.
53 *
54 * See macros for details on using this structure
55 * @ingroup base
56 * @tparam T is the objectype this constrains to be testable - assumes print() and equals()
57 */
58 template <class T>
59 class IsTestable {
60 T t;
61 bool r1,r2;
62 public:
63
64 GTSAM_CONCEPT_USAGE(IsTestable) {
65 // check print function, with optional string
66 traits<T>::Print(t, std::string());
67 traits<T>::Print(t);
68
69 // check print, with optional threshold
70 double tol = 1.0;
71 r1 = traits<T>::Equals(t,t,tol);
72 r2 = traits<T>::Equals(t,t);
73 }
74 }; // \ Testable

83 /** Call equal on the object */
84 template<class T>
85 inline bool equal(const T& obj1, const T& obj2, double tol) {
86 return traits<T>::Equals(obj1,obj2, tol);
87 }

89 /** Call equal without tolerance (use default tolerance) */
90 template<class T>
91 inline bool equal(const T& obj1, const T& obj2) {
92 return traits<T>::Equals(obj1,obj2);

95 /**
96 * This template works for any type with equals
97 */
98 template<class V>
99 bool assert_equal(const V& expected, const V& actual, double tol = 1e-9) {
100 if (traits<V>::Equals(actual,expected, tol))
101 return true;
102 printf(format: "Not equal:\n");
103 traits<V>::Print(expected,"expected:\n");
104 traits<V>::Print(actual,"actual:\n");
105 return false;
106 }
107
108 /**
109 * Template to create a binary predicate
110 */
111 template<class V>
112 struct equals : public std::function<bool(const V&, const V&)> {
113 double tol_;
114 equals(double tol = 1e-9) : tol_(tol) {}
115 bool operator()(const V& expected, const V& actual) {
116 return (traits<V>::Equals(actual, expected, tol_));
117 }
118 };
119
120 /**
121 * Binary predicate on shared pointers
122 */
123 template<class V>
124 struct equals_star : public std::function<bool(const std::shared_ptr<V>&, const std::shared_ptr<V>&)> {
125 double tol_;
126 equals_star(double tol = 1e-9) : tol_(tol) {}
127 bool operator()(const std::shared_ptr<V>& expected, const std::shared_ptr<V>& actual) {
128 if (!actual && !expected) return true;
129 return actual && expected && traits<V>::Equals(*actual,*expected, tol_);
130 }
131 };

133 /// Requirements on type to pass it to Testable template below
134 template<typename T>
135 struct HasTestablePrereqs {
136
137 GTSAM_CONCEPT_USAGE(HasTestablePrereqs) {
138 t->print(str);
139 b = t->equals(*s,tol);
140 }
141
142 T *t, *s; // Pointer is to allow abstract classes
143 bool b;
144 double tol;
145 std::string str;
146 };
147
148 /// A helper that implements the traits interface for GTSAM types.
149 /// To use this for your gtsam type, define:
150 /// template<> struct traits<Type> : public Testable<Type> { };
151 template<typename T>
152 struct Testable {
153
154 // Check that T has the necessary methods
155 GTSAM_CONCEPT_ASSERT(HasTestablePrereqs<T>);
156
157 static void Print(const T& m, const std::string& str = "") {
158 m.print(str);
159 }
160 static bool Equals(const T& m1, const T& m2, double tol = 1e-8) {
161 return m1.equals(m2, tol);
162 }
163 };
