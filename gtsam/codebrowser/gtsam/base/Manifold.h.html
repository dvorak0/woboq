31/// tag to assert a type is a manifold
32struct manifold_tag {};
33
34/**
35 * A manifold defines a space in which there is a notion of a linear tangent space
36 * that can be centered around a given point on the manifold. These nonlinear
37 * spaces may have such properties as wrapping around (as is the case with rotations),
38 * which might make linear operations on parameters not return a viable element of
39 * the manifold.
40 *
41 * We perform optimization by computing a linear delta in the tangent space of the
42 * current estimate, and then apply this change using a retraction operation, which
43 * maps the change in tangent space back to the manifold itself.
44 *
45 * There may be multiple possible retractions for a given manifold, which can be chosen
46 * between depending on the computational complexity. The important criteria for
47 * the creation for the retract and localCoordinates functions is that they be
48 * inverse operations.
49 *
50 */
51
52template <typename T> struct traits;

56/// Requirements on type to pass it to Manifold template below
57template<class Class>
58struct HasManifoldPrereqs {
59
60 inline constexpr static auto dim = Class::dimension;
61
62 Class p, q;
63 Eigen::Matrix<double, dim, 1> v;
64 OptionalJacobian<dim, dim> Hp, Hq, Hv;
65
66 GTSAM_CONCEPT_USAGE(HasManifoldPrereqs) {
67 v = p.localCoordinates(q);
68 q = p.retract(v);
69 }
70};
71
72/// Traits to get dimension, supporting both fixed and dynamic
73template<class Class, int N>
74struct GetDimensionImpl {
75 // Get dimension at compile-time for fixed-size manifolds, and at
76 // run-time for dynamic-size manifolds.
77 static int GetDimension(const Class& m) {
78 if constexpr (N == Eigen::Dynamic) {
79 return m.dim();
80 } else {
81 return N;
82 }
83 }
84};
85
86/// A helper that implements the traits interface for GTSAM manifolds.
87/// To use this for your class type, define:
88/// template<> struct traits<Class> : public internal::ManifoldTraits<Class> { };
89template<class Class>
90struct ManifoldTraits: GetDimensionImpl<Class, Class::dimension> {
91
92 // Check that Class has the necessary machinery
93 GTSAM_CONCEPT_ASSERT(HasManifoldPrereqs<Class>);
94
95 // Dimension of the manifold
96 inline constexpr static auto dimension = Class::dimension;
97
98 // Typedefs required by all manifold types.
99 typedef Class ManifoldType;
100 typedef manifold_tag structure_category;
101 typedef Eigen::Matrix<double, dimension, 1> TangentVector;
102
103 // Local coordinates
104 static TangentVector Local(const Class& origin, const Class& other) {
105 return origin.localCoordinates(other);
106 }
107
108 // Retraction back to manifold
109 static Class Retract(const Class& origin, const TangentVector& v) {
110 return origin.retract(v);
111 }
112};
113
114/// Both ManifoldTraits and Testable
115template<class Class> struct Manifold: ManifoldTraits<Class>, Testable<Class> {};
116
117} // \ namespace internal

119/// Check invariants for Manifold type
120template<typename T>
121GTSAM_CONCEPT_REQUIRES(IsTestable<T>, bool) //
122check_manifold_invariants(const T& a, const T& b, double tol=1e-9) {
123 typename traits<T>::TangentVector v0 = traits<T>::Local(a,a);
124 typename traits<T>::TangentVector v = traits<T>::Local(a,b);
125 T c = traits<T>::Retract(a,v);
126 return v0.norm() < tol && traits<T>::Equals(b,c,tol);
127}

129/// Manifold concept
130template<typename T>
131class IsManifold {
132
133public:
134
135 using structure_category_tag = typename traits<T>::structure_category;
136 static inline constexpr int dim = traits<T>::dimension;
137 using ManifoldType = typename traits<T>::ManifoldType;
138 using TangentVector = typename traits<T>::TangentVector;
139 // Concept marker: allows checking IsManifold<T>::value in templates
140 static constexpr bool value =
141 std::is_base_of_v<manifold_tag, structure_category_tag>;
142
143 GTSAM_CONCEPT_USAGE(IsManifold) {
144 static_assert(
145 value,
146 "This type's structure_category trait does not assert it as a manifold (or derived)");
147 if constexpr (dim != Eigen::Dynamic) {
148 static_assert(TangentVector::SizeAtCompileTime == dim);
149 }
150
151 // make sure Chart methods are defined
152 v = traits<T>::Local(p, q);
153 q = traits<T>::Retract(p, v);
154 }
155
156private:
157
158 TangentVector v;
159 ManifoldType p, q;
160};
161
162/// Give fixed size dimension of a type, fails at compile time if dynamic
163template<typename T>
164struct FixedDimension {
165 using value_type = const int;
166 static inline constexpr int value = traits<T>::dimension;
167 static_assert(value != Eigen::Dynamic,
168 "FixedDimension instantiated for dynamically-sized type.");
169};
