30namespace gtsam {
31
32 /**
33 * This is the base class for any type to be stored in Values.
34 * Note: As of GTSAM 4.0, Value types should no longer derive from Value or
35 * DerivedValue. Use type traits instead.
36 * See https://bitbucket.org/gtborg/gtsam/wiki/Migrating%20from%20GTSAM%203.X%20to%20GTSAM%204.0#markdown-header-custom-value-types
37 * for current usage and migration details.
38 */
39 class GTSAM_EXPORT Value {
40 public:
41 // todo - not sure if valid
42 Value() = default;
43 Value(const Value& other) = default;
44
45 /** Clone this value in a special memory pool, must be deleted with Value::deallocate_, *not* with the 'delete' operator. */
46 virtual Value* clone_() const = 0;
47
48 /** Deallocate a raw pointer of this value */
49 virtual void deallocate_() const = 0;
50
51 /** Clone this value (normal clone on the heap, delete with 'delete' operator) */
52 virtual std::shared_ptr<Value> clone() const = 0;
53
54 /** Compare this Value with another for equality. */
55 virtual bool equals_(const Value& other, double tol = 1e-9) const = 0;
56
57 /** Print this value, for debugging and unit tests */
58 virtual void print(const std::string& str = "") const = 0;
59
60 /** Return the dimensionality of the tangent space of this value. This is
61 * the dimensionality of \c delta passed into retract() and of the vector
62 * returned by localCoordinates().
63 * @return The dimensionality of the tangent space
64 */
65 virtual size_t dim() const = 0;
66
67 /** Increment the value, by mapping from the vector delta in the tangent
68 * space of the current value back to the manifold to produce a new,
69 * incremented value.
70 * @param delta The delta vector in the tangent space of this value, by
71 * which to increment this value.
72 */
73 virtual Value* retract_(const Vector& delta) const = 0;
74
75 /** Compute the coordinates in the tangent space of this value that
76 * retract() would map to \c value.
77 * @param value The value whose coordinates should be determined in the
78 * tangent space of the value on which this function is called.
79 * @return The coordinates of \c value in the tangent space of \c this.
80 */
81 virtual Vector localCoordinates_(const Value& value) const = 0;
82
83 /** Assignment operator */
84 virtual Value& operator=(const Value& /*rhs*/) {
85 //needs a empty definition so recursion in implicit derived assignment operators work
86 return *this;
87 }
88
89 /** Cast to known ValueType */
90 template<typename ValueType>
91 const ValueType& cast() const;
92
93 /** Virutal destructor */
94 virtual ~Value() {}
95
96 private:
97 /** Empty serialization function.
98 *
99 * There are two important things that users need to do to serialize derived objects in Values successfully:
100 * (Those derived objects are stored in Values as pointer to this abstract base class Value)
101 *
102 * 1. All DERIVED classes derived from Value must put the following line in their serialization function:
103 * \code
104 ar & boost::serialization::make_nvp("DERIVED", boost::serialization::base_object<Value>(*this));
105 \endcode
106 * or, alternatively
107 * \code
108 ar & BOOST_SERIALIZATION_BASE_OBJECT_NVP(Value);
109 \endcode
110 * See: http://www.boost.org/doc/libs/release/libs/serialization/doc/serialization.html#runtimecasting
111 *
112 * 2. The source module that includes archive class headers to serialize objects of derived classes
113 * (boost/archive/text_oarchive.h, for example) must *export* all derived classes, using either
114 * BOOST_CLASS_EXPORT or BOOST_CLASS_EXPORT_GUID macros:
115 \code
116 BOOST_CLASS_EXPORT(DERIVED_CLASS_1)
117 BOOST_CLASS_EXPORT_GUID(DERIVED_CLASS_2, "DERIVED_CLASS_2_ID_STRING")
118 \endcode
119 * See: http://www.boost.org/doc/libs/release/libs/serialization/doc/serialization.html#derivedpointers
120 * http://www.boost.org/doc/libs/release/libs/serialization/doc/serialization.html#export
121 * http://www.boost.org/doc/libs/release/libs/serialization/doc/serialization.html#instantiation\
122 * http://www.boost.org/doc/libs/release/libs/serialization/doc/special.html#export
123 * http://www.boost.org/doc/libs/release/libs/serialization/doc/traits.html#export
124 * The last two links explain why these export lines have to be in the same source module that includes
125 * any of the archive class headers.
126 * */
127#if GTSAM_ENABLE_BOOST_SERIALIZATION
128 friend class boost::serialization::access;
129 template<class ARCHIVE>
130 void serialize(ARCHIVE & /*ar*/, const unsigned int /*version*/) {
131 }
132#endif
133
134 };
135
136} /* namespace gtsam */

140#endif
