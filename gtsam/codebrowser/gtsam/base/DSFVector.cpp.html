28DSFBase::DSFBase(const size_t numNodes) {
29 v_ = std::make_shared < V > (args: numNodes);
30 int index = 0;
31 for (V::iterator it = v_->begin(); it != v_->end(); it++, index++)
32 *it = index;
33}

35/* ************************************************************************* */
36DSFBase::DSFBase(const std::shared_ptr<V>& v_in) {
37 v_ = v_in;
38 int index = 0;
39 for (V::iterator it = v_->begin(); it != v_->end(); it++, index++)
40 *it = index;
41}

43/* ************************************************************************* */
44size_t DSFBase::find(size_t key) const {
45 // follow parent pointers until we reach set representative
46 size_t parent = (*v_)[key];
47 if (parent != key)
48 parent = find(key: parent); // recursive call
49 (*v_)[key] = parent; // path compression
50 return parent;
51}

53/* ************************************************************************* */
54void DSFBase::merge(const size_t& i1, const size_t& i2) {
55 (*v_)[find(key: i2)] = find(key: i1);
56}

58/* ************************************************************************* */
59DSFVector::DSFVector(const size_t numNodes) :
60 DSFBase(numNodes) {
61 keys_.reserve(n: numNodes);
62 for (size_t index = 0; index < numNodes; index++)
63 keys_.push_back(x: index);
64}

66/* ************************************************************************* */
67DSFVector::DSFVector(const std::vector<size_t>& keys) :
68 DSFBase(1 + *std::max_element(first: keys.begin(), last: keys.end())), keys_(keys) {
69}

71/* ************************************************************************* */
72DSFVector::DSFVector(const std::shared_ptr<V>& v_in,
73 const std::vector<size_t>& keys) :
74 DSFBase(v_in), keys_(keys) {
75 assert(*(std::max_element(keys.begin(), keys.end()))<v_in->size());

78/* ************************************************************************* */
79bool DSFVector::isSingleton(const size_t& label) const {
80 bool result = false;
81 for(size_t key: keys_) {
82 if (find(key) == label) {
83 if (!result) // find the first occurrence
84 result = true;
85 else
86 return false;
87 }
88 }
89 return result;
90}

93std::set<size_t> DSFVector::set(const size_t& label) const {
94 std::set < size_t > set;
95 for(size_t key: keys_)

98 return set;
99}
