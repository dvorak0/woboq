28 /// An shorthand alias for accessing the \::type inside std::enable_if that can be used in a template directly
29 template<bool B, class T = void>
30 using enable_if_t = typename std::enable_if<B, T>::type;
31}

33namespace gtsam {
34
35 /**
36 * Add our own `make_shared` as a layer of wrapping on `std::make_shared`
37 * This solves the problem with the stock `make_shared` that custom alignment is not respected, causing SEGFAULTs
38 * at runtime, which is notoriously hard to debug.
39 *
40 * Explanation
41 * ===============
42 * The template `needs_eigen_aligned_allocator<T>::value` will evaluate to `std::true_type` if the type alias
43 * `_eigen_aligned_allocator_trait = void` is present in a class, which is automatically added by the
44 * `GTSAM_MAKE_ALIGNED_OPERATOR_NEW` macro.
45 *
46 * This function declaration will only be taken when the above condition is true, so if some object does not need to
47 * be aligned, `gtsam::make_shared` will fall back to the next definition, which is a simple wrapper for
48 * `std::make_shared`.
49 *
50 * @tparam T The type of object being constructed
51 * @tparam Args Type of the arguments of the constructor
52 * @param args Arguments of the constructor
53 * @return The object created as a std::shared_ptr<T>
54 */
55 template<typename T, typename ... Args>
56 gtsam::enable_if_t<needs_eigen_aligned_allocator<T>::value, std::shared_ptr<T>> make_shared(Args &&... args) {
57 return std::allocate_shared<T>(Eigen::aligned_allocator<T>(), std::forward<Args>(args)...);
58 }
59
60 /// Fall back to the boost version if no need for alignment
61 template<typename T, typename ... Args>
62 gtsam::enable_if_t<!needs_eigen_aligned_allocator<T>::value, std::shared_ptr<T>> make_shared(Args &&... args) {
63 return std::make_shared<T>(std::forward<Args>(args)...);
64 }
