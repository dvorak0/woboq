32gtsam::GaussianFactorGraph makeTestGaussianFactorGraph() {
33 using namespace gtsam;
34 using namespace symbol_shorthand;
35
36 GaussianFactorGraph gfg;
37 Matrix I = I_2x2;
38 Vector2 b(0, 0);
39 const SharedDiagonal model = noiseModel::Diagonal::Sigmas(sigmas: Vector2(0.5, 0.5));
40 gfg.emplace_shared<JacobianFactor>(args: X(j: 1), args&: I, args: X(j: 2), args&: I, args&: b, args: model);
41 gfg.emplace_shared<JacobianFactor>(args: X(j: 1), args&: I, args: X(j: 3), args&: I, args&: b, args: model);
42 gfg.emplace_shared<JacobianFactor>(args: X(j: 1), args&: I, args: X(j: 4), args&: I, args&: b, args: model);
43 gfg.emplace_shared<JacobianFactor>(args: X(j: 2), args&: I, args: X(j: 3), args&: I, args&: b, args: model);
44 gfg.emplace_shared<JacobianFactor>(args: X(j: 2), args&: I, args: X(j: 4), args&: I, args&: b, args: model);
45 gfg.emplace_shared<JacobianFactor>(args: X(j: 3), args&: I, args: X(j: 4), args&: I, args&: b, args: model);
46
47 return gfg;
48}

50gtsam::NonlinearFactorGraph makeTestNonlinearFactorGraph() {
51 using namespace gtsam;
52 using namespace symbol_shorthand;
53
54 NonlinearFactorGraph nfg;
55 const SharedDiagonal model = noiseModel::Diagonal::Sigmas(sigmas: Vector2(0.5, 0.5));
56 nfg.emplace_shared<BetweenFactor<Rot3>>(args: X(j: 1), args: X(j: 2), args: Rot3(), args: model);
57 nfg.emplace_shared<BetweenFactor<Rot3>>(args: X(j: 1), args: X(j: 3), args: Rot3(), args: model);
58 nfg.emplace_shared<BetweenFactor<Rot3>>(args: X(j: 1), args: X(j: 4), args: Rot3(), args: model);
59 nfg.emplace_shared<BetweenFactor<Rot3>>(args: X(j: 2), args: X(j: 3), args: Rot3(), args: model);
60 nfg.emplace_shared<BetweenFactor<Rot3>>(args: X(j: 2), args: X(j: 4), args: Rot3(), args: model);
61 nfg.emplace_shared<BetweenFactor<Rot3>>(args: X(j: 3), args: X(j: 4), args: Rot3(), args: model);
62
63 return nfg;
64}

66/* ************************************************************************* */
67TEST(kruskal, GaussianFactorGraph) {
68 using namespace gtsam;
69
70 // Create factor graph.
71 const auto g = makeTestGaussianFactorGraph();
72
73 // Assign weights to all the edges in the graph.
74 const auto weights = std::vector<double>(g.size(), 1.0);
75
76 const auto mstEdgeIndices = utils::kruskal(fg: g, weights);
77
78 EXPECT(mstEdgeIndices[0] == 0);
79 EXPECT(mstEdgeIndices[1] == 1);
80 EXPECT(mstEdgeIndices[2] == 2);
81}
