35TEST(SymmetricBlockMatrix, ReadBlocks)
36{
37 // On the diagonal
38 Matrix expected1 = (Matrix(2, 2) <<
39 22, 23,
40 23, 29).finished();
41 Matrix actual1 = testBlockMatrix.diagonalBlock(J: 1);
42 EXPECT(assert_equal(expected1, actual1));
43
44 // Above the diagonal
45 Matrix expected2 = (Matrix(3, 2) <<
46 4, 5,
47 10, 11,
48 16, 17).finished();
49 Matrix actual2 = testBlockMatrix.aboveDiagonalBlock(I: 0, J: 1);

54TEST(SymmetricBlockMatrix, WriteBlocks)
55{
56 // On the diagonal
57 Matrix expected1 = testBlockMatrix.diagonalBlock(J: 1);
58 SymmetricBlockMatrix bm1 = SymmetricBlockMatrix::LikeActiveViewOf(other: testBlockMatrix);
59
60 bm1.setDiagonalBlock(I: 1, xpr: expected1);
61 Matrix actual1 = bm1.diagonalBlock(J: 1);
62 EXPECT(assert_equal(expected1, actual1));
63
64 // Above the diagonal
65 Matrix expected2 = testBlockMatrix.aboveDiagonalBlock(I: 0, J: 1);
66 SymmetricBlockMatrix bm2 = SymmetricBlockMatrix::LikeActiveViewOf(other: testBlockMatrix);
67 bm2.setOffDiagonalBlock(I: 0, J: 1, xpr: expected2);
68 Matrix actual2 = bm2.aboveDiagonalBlock(I: 0, J: 1);
69 EXPECT(assert_equal(expected2, actual2));
70
71 // Below the diagonal
72 Matrix expected3 = testBlockMatrix.aboveDiagonalBlock(I: 0, J: 1).transpose();
73 SymmetricBlockMatrix bm3 = SymmetricBlockMatrix::LikeActiveViewOf(other: testBlockMatrix);
74 bm3.setOffDiagonalBlock(I: 1, J: 0, xpr: expected3);
75 Matrix actual3 = bm3.aboveDiagonalBlock(I: 0, J: 1).transpose();
76 EXPECT(assert_equal(expected3, actual3));
77}

100TEST(SymmetricBlockMatrix, expressions)
101{
102 const std::vector<size_t> dimensions{2, 3, 1};
103 SymmetricBlockMatrix expected1(dimensions, (Matrix(6, 6) <<
104 0, 0, 0, 0, 0, 0,
105 0, 0, 0, 0, 0, 0,
106 0, 0, 4, 6, 8, 0,
107 0, 0, 0, 9, 12, 0,
108 0, 0, 0, 0, 16, 0,
109 0, 0, 0, 0, 0, 0).finished());
110
111 SymmetricBlockMatrix expected2(dimensions, (Matrix(6, 6) <<
112 0, 0, 10, 15, 20, 0,
113 0, 0, 12, 18, 24, 0,
114 0, 0, 0, 0, 0, 0,
115 0, 0, 0, 0, 0, 0,
116 0, 0, 0, 0, 0, 0,
117 0, 0, 0, 0, 0, 0).finished());
118
119 Matrix a = (Matrix(1, 3) << 2, 3, 4).finished();
120 Matrix b = (Matrix(1, 2) << 5, 6).finished();
121
122 SymmetricBlockMatrix bm1(dimensions);
123 bm1.setZero();
124 bm1.diagonalBlock(J: 1).rankUpdate(u: a.transpose());
125 EXPECT(assert_equal(Matrix(expected1.selfadjointView()), bm1.selfadjointView()));
126
127 SymmetricBlockMatrix bm2(dimensions);
128 bm2.setZero();
129 bm2.updateOffDiagonalBlock(I: 0, J: 1, xpr: b.transpose() * a);
130 EXPECT(assert_equal(Matrix(expected2.selfadjointView()), bm2.selfadjointView()));
131
132 SymmetricBlockMatrix bm3(dimensions);
133 bm3.setZero();
134 bm3.updateOffDiagonalBlock(I: 1, J: 0, xpr: a.transpose() * b);
135 EXPECT(assert_equal(Matrix(expected2.selfadjointView()), bm3.selfadjointView()));
136
137 SymmetricBlockMatrix bm4(dimensions);
138 bm4.setZero();
139 bm4.updateDiagonalBlock(I: 1, xpr: expected1.diagonalBlock(J: 1));
140 EXPECT(assert_equal(Matrix(expected1.selfadjointView()), bm4.selfadjointView()));
141
142 SymmetricBlockMatrix bm5(dimensions);
143 bm5.setZero();
144 bm5.updateOffDiagonalBlock(I: 0, J: 1, xpr: expected2.aboveDiagonalBlock(I: 0, J: 1));
145 EXPECT(assert_equal(Matrix(expected2.selfadjointView()), bm5.selfadjointView()));
146
147 SymmetricBlockMatrix bm6(dimensions);
148 bm6.setZero();

150 EXPECT(assert_equal(Matrix(expected2.selfadjointView()), bm6.selfadjointView()));
151}

154TEST(SymmetricBlockMatrix, inverseInPlace) {
155 // generate an invertible matrix
156 const Vector3 a(1.0, 0.2, 2.0), b(0.3, 0.8, -1.0), c(0.1, 0.2, 0.7);
157 Matrix inputMatrix(3, 3);
158 inputMatrix.setZero();
159 inputMatrix += a * a.transpose();
160 inputMatrix += b * b.transpose();
161 inputMatrix += c * c.transpose();
162 const Matrix expectedInverse = inputMatrix.inverse();
163
164 const std::vector<size_t> dimensions{2, 1};
165 SymmetricBlockMatrix symmMatrix(dimensions, inputMatrix);
166 // invert in place
167 symmMatrix.invertInPlace();
168 EXPECT(assert_equal(expectedInverse, symmMatrix.selfadjointView()));
169}
