26/* ************************************************************************* */
27double f(const Vector2& x) {
28 assert(x.size() == 2);
29 return sin(x: x(0)) + cos(x: x(1));
30}
31
32/* ************************************************************************* */
33//
34TEST(testNumericalDerivative, numericalGradient) {
35 Vector2 x(1, 1.1);
36
37 Vector expected(2);
38 expected << cos(x: x(0)), -sin(x: x(1));
39
40 Vector actual = numericalGradient<Vector2>(h: f, x);
41
42 EXPECT(assert_equal(expected, actual, 1e-5));
43}

55}

57/* ************************************************************************* */
58double f2(const Vector2& x) {
59 assert(x.size() == 2);
60 return sin(x: x(0)) * cos(x: x(1));
61}
62
63/* ************************************************************************* */
64//
65TEST(testNumericalDerivative, numericalHessian2) {
66 Vector2 v(0.5, 1.0);
67 Vector2 x(v);
68
69 Matrix expected = (Matrix(2, 2) << -cos(x: x(1)) * sin(x: x(0)), -sin(x: x(1))
70 * cos(x: x(0)), -cos(x: x(0)) * sin(x: x(1)), -sin(x: x(0)) * cos(x: x(1))).finished();
71
72 Matrix actual = numericalHessian(f: f2, x);
73
74 EXPECT(assert_equal(expected, actual, 1e-5));
75}

79 return sin(x: x1) * cos(x: x2);
80}

98}

108 return sin(x: x) * cos(x: y) * z * z;
109}

141Vector6 f6(const double x1, const double x2, const double x3, const double x4,
142 const double x5, const double x6) {
143 Vector6 result;
144 result << sin(x: x1), cos(x: x2), x3 * x3, x4 * x4 * x4, sqrt(x: x5), sin(x: x6) - cos(x: x6);
145 return result;
146}

152 return result;
153}
