19#include <CppUnitLite/TestHarness.h>
20#include <gtsam/base/VerticalBlockMatrix.h>
21
22#include<list>
23#include<vector>
24
25using namespace gtsam;
26
27const std::vector<size_t> dimensions{3, 2, 1};
28
29//*****************************************************************************
30TEST(VerticalBlockMatrix, Constructor1) {
31 VerticalBlockMatrix actual(dimensions,6);
32 EXPECT_LONGS_EQUAL(6,actual.rows());
33 EXPECT_LONGS_EQUAL(6,actual.cols());
34 EXPECT_LONGS_EQUAL(3,actual.nBlocks());
35}
36
37//*****************************************************************************
38TEST(VerticalBlockMatrix, Constructor2) {
39 VerticalBlockMatrix actual(dimensions,
40 (Matrix(6, 6) << 1, 2, 3, 4, 5, 6, //
41 2, 8, 9, 10, 11, 12, //
42 3, 9, 15, 16, 17, 18, //
43 4, 10, 16, 22, 23, 24, //
44 5, 11, 17, 23, 29, 30, //
45 6, 12, 18, 24, 30, 36).finished());
46 EXPECT_LONGS_EQUAL(6,actual.rows());
47 EXPECT_LONGS_EQUAL(6,actual.cols());
48 EXPECT_LONGS_EQUAL(3,actual.nBlocks());
49}
50
51//*****************************************************************************
52TEST(VerticalBlockMatrix, Constructor3) {
53 VerticalBlockMatrix actual(dimensions.begin(),dimensions.end(),6);
54 EXPECT_LONGS_EQUAL(6,actual.rows());
55 EXPECT_LONGS_EQUAL(6,actual.cols());
56 EXPECT_LONGS_EQUAL(3,actual.nBlocks());
57}
58
59//*****************************************************************************
60int main() {
61 TestResult tr;
62 return TestRegistry::runAllTests(result&: tr);
63}
