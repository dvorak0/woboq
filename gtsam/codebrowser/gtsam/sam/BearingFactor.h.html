12/**
13 * @file BearingFactor.h
14 * @brief Serializable factor induced by a bearing measurement
15 * @date July 2015
16 * @author Frank Dellaert
17 **/
18
19#pragma once
20
21#include <gtsam/nonlinear/ExpressionFactor.h>
22
23namespace gtsam {
24
25// forward declaration of Bearing functor, assumed partially specified
26template <typename A1, typename A2>
27struct Bearing;
28
29/**
30 * Binary factor for a bearing measurement
31 * Works for any two types A1,A2 for which the functor Bearing<A1,A2>() is
32 * defined
33 * @ingroup sam
34 */
35template <typename A1, typename A2,
36 typename T = typename Bearing<A1, A2>::result_type>
37struct BearingFactor : public ExpressionFactorN<T, A1, A2> {
38 typedef ExpressionFactorN<T, A1, A2> Base;
39
40 /// default constructor
41 BearingFactor() {}
42
43 /// primary constructor
44 BearingFactor(Key key1, Key key2, const T& measured,
45 const SharedNoiseModel& model)
46 : Base({key1, key2}, model, measured) {
47 this->initialize(expression(keys: {key1, key2}));
48 }
49
50 // Return measurement expression
51 Expression<T> expression(const typename Base::ArrayNKeys &keys) const override {
52 Expression<A1> a1_(keys[0]);
53 Expression<A2> a2_(keys[1]);
54 return Expression<T>(Bearing<A1, A2>(), a1_, a2_);
55 }
56
57 /// print
58 void print(const std::string& s = "",
59 const KeyFormatter& kf = DefaultKeyFormatter) const override {
60 std::cout << s << "BearingFactor" << std::endl;
61 Base::print(s, kf);
62 }
63
64 Vector evaluateError(const A1& a1, const A2& a2,
65 OptionalMatrixType H1 = OptionalNone, OptionalMatrixType H2 = OptionalNone) const {
66 std::vector<Matrix> Hs(2);
67 const auto &keys = Factor::keys();
68 const Vector error = unwhitenedError(
69 {{keys[0], genericValue(a1)}, {keys[1], genericValue(a2)}},
70 Hs);
71 if (H1) *H1 = Hs[0];
72 if (H2) *H2 = Hs[1];
73 return error;
74 }
75
76
77 private:
78#if GTSAM_ENABLE_BOOST_SERIALIZATION
79 friend class boost::serialization::access;
80 template <class ARCHIVE>
81 void serialize(ARCHIVE& ar, const unsigned int /*version*/) {
82 ar& boost::serialization::make_nvp(
83 "Base", boost::serialization::base_object<Base>(*this));
84 }
85#endif
86}; // BearingFactor
87
88/// traits
89template <typename A1, typename A2, typename T>
90struct traits<BearingFactor<A1, A2, T> >
91 : public Testable<BearingFactor<A1, A2, T> > {};
92
93} // namespace gtsam
