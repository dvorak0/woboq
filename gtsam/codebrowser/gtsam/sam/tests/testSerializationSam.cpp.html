91TEST(RangeFactor, EqualsAfterDeserializing) {
92 // Check that the same results are obtained after deserializing:
93 Pose3 body_P_sensor_3D(Rot3::RzRyRx(x: -M_PI_2, y: 0.0, z: -M_PI_2),
94 Point3(0.25, -0.10, 1.0));
95 RangeFactorWithTransform<Pose3, Point3> factor3D_1(
96 poseKey, pointKey, rangeMmeasurement, rangeNoiseModel, body_P_sensor_3D),
97 factor3D_2;
98
99 // check with Equal() trait:
100 gtsam::serializationTestHelpers::roundtripXML(input: factor3D_1, output&: factor3D_2);
101 CHECK(assert_equal(factor3D_1, factor3D_2));
102
103 const Pose3 pose(Rot3::RzRyRx(x: 0.2, y: -0.3, z: 1.75), Point3(1.0, 2.0, -3.0));
104 const Point3 point(-2.0, 11.0, 1.0);
105 const Values values = {{poseKey, genericValue(v: pose)},
106 {pointKey, genericValue(v: point)}};
107
108 const Vector error_1 = factor3D_1.unwhitenedError(x: values);
109 const Vector error_2 = factor3D_2.unwhitenedError(x: values);
110 CHECK(assert_equal(error_1, error_2));
111}
