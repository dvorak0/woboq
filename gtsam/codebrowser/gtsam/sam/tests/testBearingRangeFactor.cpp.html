29namespace {
30Key poseKey(1);
31Key pointKey(2);
32} // namespace
33
34/* ************************************************************************* */
35TEST(BearingRangeFactor, 2D) {
36 typedef BearingRangeFactor<Pose2, Point2> BearingRangeFactor2D;
37 SharedNoiseModel model2D(noiseModel::Isotropic::Sigma(dim: 2, sigma: 0.5));
38 BearingRangeFactor2D factor2D(poseKey, pointKey, 1, 2, model2D);
39
40 // Set the linearization point
41 Values values;
42 values.insert(j: poseKey, val: Pose2(1.0, 2.0, 0.57));
43 values.insert(j: pointKey, val: Point2(-4.0, 11.0));
44
45 EXPECT_CORRECT_EXPRESSION_JACOBIANS(factor2D.expression({poseKey, pointKey}),
46 values, 1e-7, 1e-5);
47 EXPECT_CORRECT_FACTOR_JACOBIANS(factor2D, values, 1e-7, 1e-5);
48}
49
50/* ************************************************************************* */
51TEST(BearingRangeFactor, 3D) {
52 typedef BearingRangeFactor<Pose3, Point3> BearingRangeFactor3D;
53 SharedNoiseModel model3D(noiseModel::Isotropic::Sigma(dim: 3, sigma: 0.5));
54
55 const Unit3 bearing = Pose3().bearing(point: Point3(1, 0, 0));
56 const double range = 1.0;
57 BearingRangeFactor3D factor3D(poseKey, pointKey, bearing, range, model3D);
58
59 // Set the linearization point
60 Values values;
61 values.insert(j: poseKey, val: Pose3());
62 values.insert(j: pointKey, val: Point3(1, 0, 0));
63
64 // Check that the error is zero at the linearization point
65 Vector actualError = factor3D.unwhitenedError(x: values);
66 EXPECT(assert_equal(Vector::Zero(actualError.size()), actualError, 1e-9));
67
