22#pragma once
23
24#include <gtsam/navigation/NavState.h>
25#include <gtsam/navigation/PreintegrationBase.h>
26
27namespace gtsam {
28
29/**
30 * IMU pre-integration on NavState manifold.
31 * This corresponds to the original RSS paper (with one difference: V is rotated)
32 */
33class GTSAM_EXPORT ManifoldPreintegration : public PreintegrationBase {
34 protected:
35
36 /**
37 * Pre-integrated navigation state, from frame i to frame j
38 * Note: relative position does not take into account velocity at time i, see deltap+, in [2]
39 * Note: velocity is now also in frame i, as opposed to deltaVij in [2]
40 */
41 NavState deltaXij_;
42 Matrix3 delRdelBiasOmega_; ///< Jacobian of preintegrated rotation w.r.t. angular rate bias
43 Matrix3 delPdelBiasAcc_; ///< Jacobian of preintegrated position w.r.t. acceleration bias
44 Matrix3 delPdelBiasOmega_; ///< Jacobian of preintegrated position w.r.t. angular rate bias
45 Matrix3 delVdelBiasAcc_; ///< Jacobian of preintegrated velocity w.r.t. acceleration bias
46 Matrix3 delVdelBiasOmega_; ///< Jacobian of preintegrated velocity w.r.t. angular rate bias
47
48 /// Default constructor for serialization
49 ManifoldPreintegration() {
50 resetIntegration();
51 }
52
53public:
54 /// @name Constructors
55 /// @{
56
57 /**
58 * Constructor, initializes the variables in the base class
59 * @param p Parameters, typically fixed in a single application
60 * @param bias Current estimate of acceleration and rotation rate biases
61 */
62 ManifoldPreintegration(const std::shared_ptr<Params>& p,
63 const imuBias::ConstantBias& biasHat = imuBias::ConstantBias());
64
65 /// @}
66
67 /// @name Basic utilities
68 /// @{
69 /// Re-initialize PreintegratedMeasurements
70 void resetIntegration() override;
71
72 /// @}
73
74 /// @name Instance variables access
75 /// @{
76 NavState deltaXij() const override { return deltaXij_; }
77 Rot3 deltaRij() const override { return deltaXij_.attitude(); }
78 Vector3 deltaPij() const override { return deltaXij_.position(); }
79 Vector3 deltaVij() const override { return deltaXij_.velocity(); }
80
81 Matrix3 delRdelBiasOmega() const { return delRdelBiasOmega_; }
82 Matrix3 delPdelBiasAcc() const { return delPdelBiasAcc_; }
83 Matrix3 delPdelBiasOmega() const { return delPdelBiasOmega_; }
84 Matrix3 delVdelBiasAcc() const { return delVdelBiasAcc_; }
85 Matrix3 delVdelBiasOmega() const { return delVdelBiasOmega_; }
86
87 /// @name Testable
88 /// @{
89 bool equals(const ManifoldPreintegration& other, double tol) const;
90 /// @}
91
92 /// @name Main functionality
93 /// @{
94
95 /// Update preintegrated measurements and get derivatives
96 /// It takes measured quantities in the j frame
97 /// Modifies preintegrated quantities in place after correcting for bias and possibly sensor pose
98 /// NOTE(frank): implementation is different in two versions
99 void update(const Vector3& measuredAcc, const Vector3& measuredOmega, const double dt,
100 Matrix9* A, Matrix93* B, Matrix93* C) override;
101
102 /// Given the estimate of the bias, return a NavState tangent vector
103 /// summarizing the preintegrated IMU measurements so far
104 /// NOTE(frank): implementation is different in two versions
105 Vector9 biasCorrectedDelta(const imuBias::ConstantBias& bias_i,
106 OptionalJacobian<9, 6> H = {}) const override;
107
108 /** Dummy clone for MATLAB */
109 virtual std::shared_ptr<ManifoldPreintegration> clone() const {
110 return std::shared_ptr<ManifoldPreintegration>();
111 }
112
113 /// @}
114
115private:
116#if GTSAM_ENABLE_BOOST_SERIALIZATION
117 /** Serialization function */
118 friend class boost::serialization::access;
119 template<class ARCHIVE>
120 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
121 namespace bs = ::boost::serialization;
122 ar & BOOST_SERIALIZATION_BASE_OBJECT_NVP(PreintegrationBase);
123 ar & BOOST_SERIALIZATION_NVP(deltaXij_);
124 ar & BOOST_SERIALIZATION_NVP(delRdelBiasOmega_);
125 ar & BOOST_SERIALIZATION_NVP(delPdelBiasAcc_);
126 ar & BOOST_SERIALIZATION_NVP(delPdelBiasOmega_);
127 ar & BOOST_SERIALIZATION_NVP(delVdelBiasAcc_);
128 ar & BOOST_SERIALIZATION_NVP(delVdelBiasOmega_);
129 }
130#endif
131};
132
133} /// namespace gtsam
