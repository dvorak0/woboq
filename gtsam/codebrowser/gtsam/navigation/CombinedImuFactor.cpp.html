82#define D_R_R(H) (H)->block<3,3>(0,0)
83#define D_R_t(H) (H)->block<3,3>(0,3)
84#define D_R_v(H) (H)->block<3,3>(0,6)
85#define D_t_R(H) (H)->block<3,3>(3,0)
86#define D_t_t(H) (H)->block<3,3>(3,3)
87#define D_t_v(H) (H)->block<3,3>(3,6)
88#define D_v_R(H) (H)->block<3,3>(6,0)
89#define D_v_t(H) (H)->block<3,3>(6,3)
90#define D_v_v(H) (H)->block<3,3>(6,6)
91#define D_a_a(H) (H)->block<3,3>(9,9)
92#define D_g_g(H) (H)->block<3,3>(12,12)

94//------------------------------------------------------------------------------
95template <class PreintegrationType>
96void PreintegratedCombinedMeasurementsT<
97 PreintegrationType>::integrateMeasurement(const Vector3& measuredAcc,
98 const Vector3& measuredOmega,
99 double dt) {
100 if (dt <= 0) {
101 throw std::runtime_error(
102 "PreintegratedCombinedMeasurements::integrateMeasurement: dt <=0");
103 }
104
105 // Update preintegrated measurements.
106 Matrix9 A; // Jacobian wrt preintegrated measurements without bias (df/dx)
107 Matrix93 B, C; // Jacobian of state wrpt accel bias and omega bias.
108 PreintegrationType::update(measuredAcc, measuredOmega, dt, &A, &B, &C);
109
110 // Update preintegrated measurements covariance: as in [2] we consider a first
111 // order propagation that can be seen as a prediction phase in an EKF
112 // framework. In this implementation, in contrast to [2], we consider the
113 // uncertainty of the bias selection and we keep correlation between biases
114 // and preintegrated measurements
115
116 // Single Jacobians to propagate covariance
117 Matrix3 theta_H_omega = C.topRows<3>();
118 Matrix3 pos_H_acc = B.middleRows<3>(startRow: 3);
119 Matrix3 vel_H_acc = B.bottomRows<3>();
120
121 // overall Jacobian wrt preintegrated measurements (df/dx)
122 Eigen::Matrix<double, 15, 15> F;
123 F.setZero();
124 F.block<9, 9>(startRow: 0, startCol: 0) = A;
125 F.block<3, 3>(startRow: 0, startCol: 12) = theta_H_omega;
126 F.block<3, 3>(startRow: 3, startCol: 9) = pos_H_acc;
127 F.block<3, 3>(startRow: 6, startCol: 9) = vel_H_acc;
128 F.block<6, 6>(startRow: 9, startCol: 9) = I_6x6;
129
130 // Update the uncertainty on the state (matrix F in [4]).
131 preintMeasCov_ = F * preintMeasCov_ * F.transpose();
132
133 // propagate uncertainty
134 // TODO(frank): use noiseModel routine so we can have arbitrary noise models.
135 const Matrix3& aCov = this->p().accelerometerCovariance;
136 const Matrix3& wCov = this->p().gyroscopeCovariance;
137 const Matrix3& iCov = this->p().integrationCovariance;
138
139 // first order uncertainty propagation
140 // Optimized matrix mult: (1/dt) * G * measurementCovariance * G.transpose()
141 Eigen::Matrix<double, 15, 15> G_measCov_Gt;
142 G_measCov_Gt.setZero(rows: 15, cols: 15);
143
144 // BLOCK DIAGONAL TERMS
145 D_R_R(&G_measCov_Gt) =
146 (theta_H_omega * (wCov / dt) * theta_H_omega.transpose());
147
148 D_t_t(&G_measCov_Gt) =
149 (pos_H_acc * (aCov / dt) * pos_H_acc.transpose()) + (dt * iCov);
150
151 D_v_v(&G_measCov_Gt) = (vel_H_acc * (aCov / dt) * vel_H_acc.transpose());
152
153 D_a_a(&G_measCov_Gt) = dt * this->p().biasAccCovariance;
154 D_g_g(&G_measCov_Gt) = dt * this->p().biasOmegaCovariance;
155
156 // OFF BLOCK DIAGONAL TERMS
157 D_t_v(&G_measCov_Gt) = (pos_H_acc * (aCov / dt) * vel_H_acc.transpose());
158 D_v_t(&G_measCov_Gt) = (vel_H_acc * (aCov / dt) * pos_H_acc.transpose());
159
160 preintMeasCov_.noalias() += G_measCov_Gt;
161}

185//------------------------------------------------------------------------------
186template <class PIM>
187Vector CombinedImuFactorT<PIM>::evaluateError(const Pose3& pose_i,
188 const Vector3& vel_i, const Pose3& pose_j, const Vector3& vel_j,
189 const imuBias::ConstantBias& bias_i, const imuBias::ConstantBias& bias_j,
190 OptionalMatrixType H1, OptionalMatrixType H2,
191 OptionalMatrixType H3, OptionalMatrixType H4,
192 OptionalMatrixType H5, OptionalMatrixType H6) const {
193
194 // error wrt bias evolution model (random walk)
195 Matrix6 Hbias_i, Hbias_j;
196 Vector6 fbias = traits<imuBias::ConstantBias>::Between(v1: bias_j, v2: bias_i,
197 H1: H6 ? &Hbias_j : 0, H2: H5 ? &Hbias_i : 0).vector();
198
199 Matrix96 D_r_pose_i, D_r_pose_j, D_r_bias_i;
200 Matrix93 D_r_vel_i, D_r_vel_j;
201
202 // error wrt preintegrated measurements
203 Vector9 r_Rpv = pim_.computeErrorAndJacobians(pose_i, vel_i, pose_j, vel_j,
204 bias_i, H1 ? &D_r_pose_i : 0, H2 ? &D_r_vel_i : 0, H3 ? &D_r_pose_j : 0,
205 H4 ? &D_r_vel_j : 0, H5 ? &D_r_bias_i : 0);
206
207 // if we need the jacobians
208 if (H1) {
209 H1->resize(rows: 15, cols: 6);
210 H1->block<9, 6>(startRow: 0, startCol: 0) = D_r_pose_i;
211 // adding: [dBiasAcc/dPi ; dBiasOmega/dPi]
212 H1->block<6, 6>(startRow: 9, startCol: 0).setZero();
213 }
214 if (H2) {
215 H2->resize(rows: 15, cols: 3);
216 H2->block<9, 3>(startRow: 0, startCol: 0) = D_r_vel_i;
217 // adding: [dBiasAcc/dVi ; dBiasOmega/dVi]
218 H2->block<6, 3>(startRow: 9, startCol: 0).setZero();
219 }
220 if (H3) {
221 H3->resize(rows: 15, cols: 6);
222 H3->block<9, 6>(startRow: 0, startCol: 0) = D_r_pose_j;
223 // adding: [dBiasAcc/dPj ; dBiasOmega/dPj]
224 H3->block<6, 6>(startRow: 9, startCol: 0).setZero();
225 }
226 if (H4) {
227 H4->resize(rows: 15, cols: 3);
228 H4->block<9, 3>(startRow: 0, startCol: 0) = D_r_vel_j;
229 // adding: [dBiasAcc/dVi ; dBiasOmega/dVi]
230 H4->block<6, 3>(startRow: 9, startCol: 0).setZero();
231 }
232 if (H5) {
233 H5->resize(rows: 15, cols: 6);
234 H5->block<9, 6>(startRow: 0, startCol: 0) = D_r_bias_i;
235 // adding: [dBiasAcc/dBias_i ; dBiasOmega/dBias_i]
236 H5->block<6, 6>(startRow: 9, startCol: 0) = Hbias_i;
237 }
238 if (H6) {
239 H6->resize(rows: 15, cols: 6);
240 H6->block<9, 6>(startRow: 0, startCol: 0).setZero();
241 // adding: [dBiasAcc/dBias_j ; dBiasOmega/dBias_j]
242 H6->block<6, 6>(startRow: 9, startCol: 0) = Hbias_j;
243 }
244
245 // overall error
246 Vector r(15);
247 r << r_Rpv, fbias; // vector of size 15
248 return r;
249}
