26//------------------------------------------------------------------------------
27NavState NavState::Create(const Rot3& R, const Point3& t, const Velocity3& v,
28 OptionalJacobian<9, 3> H1, OptionalJacobian<9, 3> H2,
29 OptionalJacobian<9, 3> H3) {
30 Matrix3 Rt;
31 if (H2 || H3) Rt = R.transpose();
32 if (H1) *H1 << I_3x3, Z_3x3, Z_3x3;
33 if (H2) *H2 << Z_3x3, Rt, Z_3x3;
34 if (H3) *H3 << Z_3x3, Z_3x3, Rt;
35 return NavState(R, t, v);
36}

38NavState NavState::FromPoseVelocity(const Pose3& pose, const Vector3& vel,
39 OptionalJacobian<9, 6> H1, OptionalJacobian<9, 3> H2) {
40 if (H1)
41 *H1 << I_3x3, Z_3x3, Z_3x3, I_3x3, Z_3x3, Z_3x3;
42 if (H2)
43 *H2 << Z_3x3, Z_3x3, pose.rotation().transpose();
44 return NavState(pose, vel);

69Vector3 NavState::bodyVelocity(OptionalJacobian<3, 9> H) const {
70 const Rot3& nRb = R_;
71 const Vector3& n_v = v_;
72 Matrix3 D_bv_nRb;
73 Vector3 b_v = nRb.unrotate(p: n_v, H1: H ? &D_bv_nRb : 0);
74 if (H)
75 *H << D_bv_nRb, Z_3x3, I_3x3;
76 return b_v;
77}

90//------------------------------------------------------------------------------
91NavState::Vector25 NavState::vec(OptionalJacobian<25, 9> H) const {
92 const Matrix5 T = this->matrix();
93 if (H) {
94 H->setZero();
95 auto R = T.block<3, 3>(startRow: 0, startCol: 0);
96 H->block<3, 1>(startRow: 0, startCol: 1) = -R.col(i: 2);
97 H->block<3, 1>(startRow: 0, startCol: 2) = R.col(i: 1);
98 H->block<3, 1>(startRow: 5, startCol: 0) = R.col(i: 2);
99 H->block<3, 1>(startRow: 5, startCol: 2) = -R.col(i: 0);
100 H->block<3, 1>(startRow: 10, startCol: 0) = -R.col(i: 1);
101 H->block<3, 1>(startRow: 10, startCol: 1) = R.col(i: 0);
102 H->block<3, 3>(startRow: 15, startCol: 3) = R;
103 H->block<3, 3>(startRow: 20, startCol: 6) = R;
104 }
105 return Eigen::Map<const Vector25>(T.data());
106}

134// See [this document](doc/Jacobians.md) for details.
135NavState NavState::Expmap(const Vector9& xi, OptionalJacobian<9, 9> Hxi) {
136 // Get angular velocity w and components rho (for t) and nu (for v) from xi
137 Vector3 w = xi.head<3>(), rho = xi.segment<3>(start: 3), nu = xi.tail<3>();
138
139 // Instantiate functor for Dexp-related operations:
140 const so3::DexpFunctor local(w);
141
142 // Compute rotation using Expmap
143#ifdef GTSAM_USE_QUATERNIONS
144 const Rot3 R = traits<gtsam::Quaternion>::Expmap(w);
145#else
146 const Rot3 R(local.expmap());
147#endif
148
149 // Compute translation and velocity. See Pose3::Expmap
150 Matrix3 H_t_w, H_v_w;
151 const Vector3 t = local.Jacobian().applyLeft(v: rho, Hw: Hxi ? &H_t_w : nullptr);
152 const Vector3 v = local.Jacobian().applyLeft(v: nu, Hw: Hxi ? &H_v_w : nullptr);
153
154 if (Hxi) {
155 const Matrix3 Jr = local.Jacobian().right();
156 // We are creating a NavState, so we still need to chain H_t_w and H_v_w
157 // with R^T, the Jacobian of Navstate::Create with respect to both t and v.
158 const Matrix3 Rt = R.transpose();
159 *Hxi << Jr, Z_3x3, Z_3x3, // Jr here *is* the Jacobian of expmap
160 Rt * H_t_w, Jr, Z_3x3, //
161 Rt * H_v_w, Z_3x3, Jr;
162 // In the last two rows, Jr = R^T * Jl, see Barfoot eq. (8.83).
163 // Jl is the left Jacobian of SO(3) at w.
164 }
165
166 return NavState(R, t, v);
167}

169//------------------------------------------------------------------------------
170Vector9 NavState::Logmap(const NavState& state, OptionalJacobian<9, 9> Hstate) {
171 if (Hstate) *Hstate = LogmapDerivative(xi: state);
172
173 const Vector3 phi = Rot3::Logmap(R: state.rotation());
174 const Vector3& p = state.position();
175 const Vector3& v = state.velocity();
176 const double t = phi.norm();
177 if (t < 1e-8) {
178 Vector9 log;
179 log << phi, p, v;
180 return log;
181
182 } else {
183 const Matrix3 W = skewSymmetric(w: phi / t);
184
185 const double Tan = tan(x: 0.5 * t);
186 const Vector3 Wp = W * p;
187 const Vector3 Wv = W * v;
188 const Vector3 rho = p - (0.5 * t) * Wp + (1 - t / (2. * Tan)) * (W * Wp);
189 const Vector3 nu = v - (0.5 * t) * Wv + (1 - t / (2. * Tan)) * (W * Wv);
190 Vector9 log;
191 // Order is , p, v
192 log << phi, rho, nu;
193 return log;
194 }
195}

197//------------------------------------------------------------------------------
198Matrix9 NavState::AdjointMap() const {
199 const Matrix3 R = R_.matrix();
200 Matrix3 A = skewSymmetric(w: t_) * R;
201 Matrix3 B = skewSymmetric(w: v_) * R;
202 // Eqn 2 in Barrau20icra
203 Matrix9 adj;
204 adj << R, Z_3x3, Z_3x3, A, R, Z_3x3, B, Z_3x3, R;
205 return adj;
206}

208//------------------------------------------------------------------------------
209Vector9 NavState::Adjoint(const Vector9& xi_b, OptionalJacobian<9, 9> H_state,
210 OptionalJacobian<9, 9> H_xib) const {
211 const Matrix9 Ad = AdjointMap();
212
213 // Jacobians
214 if (H_state) *H_state = -Ad * adjointMap(xi: xi_b);
215 if (H_xib) *H_xib = Ad;
216
217 return Ad * xi_b;
218}

220//------------------------------------------------------------------------------
221Matrix9 NavState::adjointMap(const Vector9& xi) {
222 Matrix3 w_hat = skewSymmetric(wx: xi(0), wy: xi(1), wz: xi(2));
223 Matrix3 v_hat = skewSymmetric(wx: xi(3), wy: xi(4), wz: xi(5));
224 Matrix3 a_hat = skewSymmetric(wx: xi(6), wy: xi(7), wz: xi(8));
225 Matrix9 adj;
226 adj << w_hat, Z_3x3, Z_3x3, v_hat, w_hat, Z_3x3, a_hat, Z_3x3, w_hat;
227 return adj;
228}

230//------------------------------------------------------------------------------
231Vector9 NavState::adjoint(const Vector9& xi, const Vector9& y,
232 OptionalJacobian<9, 9> Hxi,
233 OptionalJacobian<9, 9> H_y) {
234 if (Hxi) {
235 Hxi->setZero();
236 for (int i = 0; i < 9; ++i) {
237 Vector9 dxi;
238 dxi.setZero();
239 dxi(i) = 1.0;
240 Matrix9 Gi = adjointMap(xi: dxi);
241 Hxi->col(i) = Gi * y;
242 }
243 }
244
245 const Matrix9& ad_xi = adjointMap(xi);
246 if (H_y) *H_y = ad_xi;
247
248 return ad_xi * y;
249}

258//------------------------------------------------------------------------------
259Matrix9 NavState::LogmapDerivative(const Vector9& xi) {
260 const Vector3 w = xi.head<3>();
261 Vector3 rho = xi.segment<3>(start: 3);
262 Vector3 nu = xi.tail<3>();
263
264 // Instantiate functor for Dexp-related operations:
265 const so3::DexpFunctor local(w);
266
267 // Call Jacobian().applyLeft to get its Jacobians
268 Matrix3 H_t_w, H_v_w;
269 local.Jacobian().applyLeft(v: rho, Hw: H_t_w);
270 local.Jacobian().applyLeft(v: nu, Hw: H_v_w);
271
272 // Multiply with R^T to account for NavState::Create Jacobian.
273 const Matrix3 Rt = local.expmap().transpose();
274 const Matrix3 Qt = Rt * H_t_w;
275 const Matrix3 Qv = Rt * H_v_w;
276
277 // Now compute the blocks of the LogmapDerivative Jacobian
278 const Matrix3 Jw = Rot3::LogmapDerivative(x: w);
279 const Matrix3 Qt2 = -Jw * Qt * Jw;
280 const Matrix3 Qv2 = -Jw * Qv * Jw;
281
282 Matrix9 J;
283 J << Jw, Z_3x3, Z_3x3,
284 Qt2, Jw, Z_3x3,
285 Qv2, Z_3x3, Jw;
286 return J;
287}

296Matrix5 NavState::Hat(const Vector9& xi) {
297 Matrix5 X;
298 const double wx = xi(0), wy = xi(1), wz = xi(2);
299 const double px = xi(3), py = xi(4), pz = xi(5);
300 const double vx = xi(6), vy = xi(7), vz = xi(8);
301 X << 0., -wz, wy, px, vx,
302 wz, 0., -wx, py, vy,
303 -wy, wx, 0., pz, vz,
304 0., 0., 0., 0., 0.,
305 0., 0., 0., 0., 0.;
306 return X;

330//------------------------------------------------------------------------------
331NavState NavState::retract(const Vector9& xi, //
332 OptionalJacobian<9, 9> H1, OptionalJacobian<9, 9> H2) const {
333 Rot3 nRb = R_;
334 Point3 n_t = t_, n_v = v_;
335 Matrix3 D_bRc_xi, D_R_nRb, D_t_nRb, D_v_nRb;
336 const Rot3 bRc = Rot3::Expmap(v: dR(v: xi), H: H2 ? &D_bRc_xi : 0);
337 const Rot3 nRc = nRb.compose(g: bRc, H1: H1 ? &D_R_nRb : 0);
338 const Point3 t = n_t + nRb.rotate(p: dP(v: xi), H1: H1 ? &D_t_nRb : 0);
339 const Point3 v = n_v + nRb.rotate(p: dV(v: xi), H1: H1 ? &D_v_nRb : 0);
340 if (H1) {
341 *H1 << D_R_nRb, Z_3x3, Z_3x3, //
342 // Note(frank): the derivative of n_t with respect to xi is nRb
343 // We pre-multiply with nRc' to account for NavState::Create
344 // Then we make use of the identity nRc' * nRb = bRc'
345 nRc.transpose() * D_t_nRb, bRc.transpose(), Z_3x3,
346 // Similar reasoning for v:
347 nRc.transpose() * D_v_nRb, Z_3x3, bRc.transpose();
348 }
349 if (H2) {
350 *H2 << D_bRc_xi, Z_3x3, Z_3x3, //
351 Z_3x3, bRc.transpose(), Z_3x3, //
352 Z_3x3, Z_3x3, bRc.transpose();
353 }
354 return NavState(nRc, t, v);
355}

357//------------------------------------------------------------------------------
358Vector9 NavState::localCoordinates(const NavState& g, //
359 OptionalJacobian<9, 9> H1, OptionalJacobian<9, 9> H2) const {
360 Matrix3 D_dR_R, D_dt_R, D_dv_R;
361 const Rot3 dR = R_.between(g: g.R_, H1: H1 ? &D_dR_R : 0);
362 const Point3 dP = R_.unrotate(p: g.t_ - t_, H1: H1 ? &D_dt_R : 0);
363 const Vector dV = R_.unrotate(p: g.v_ - v_, H1: H1 ? &D_dv_R : 0);
364
365 Vector9 xi;
366 Matrix3 D_xi_R;
367 xi << Rot3::Logmap(R: dR, H: (H1 || H2) ? &D_xi_R : 0), dP, dV;
368 if (H1) {
369 *H1 << D_xi_R * D_dR_R, Z_3x3, Z_3x3, //
370 D_dt_R, -I_3x3, Z_3x3, //
371 D_dv_R, Z_3x3, -I_3x3;
372 }
373 if (H2) {
374 *H2 << D_xi_R, Z_3x3, Z_3x3, //
375 Z_3x3, dR.matrix(), Z_3x3, //
376 Z_3x3, Z_3x3, dR.matrix();
377 }
378
379 return xi;
380}

383// sugar for derivative blocks
384#define D_R_R(H) (H)->block<3,3>(0,0)
385#define D_R_t(H) (H)->block<3,3>(0,3)
386#define D_R_v(H) (H)->block<3,3>(0,6)
387#define D_t_R(H) (H)->block<3,3>(3,0)
388#define D_t_t(H) (H)->block<3,3>(3,3)
389#define D_t_v(H) (H)->block<3,3>(3,6)
390#define D_v_R(H) (H)->block<3,3>(6,0)
391#define D_v_t(H) (H)->block<3,3>(6,3)
392#define D_v_v(H) (H)->block<3,3>(6,6)

394//------------------------------------------------------------------------------
395NavState NavState::update(const Vector3& b_acceleration, const Vector3& b_omega,
396 const double dt, OptionalJacobian<9, 9> F, OptionalJacobian<9, 3> G1,
397 OptionalJacobian<9, 3> G2) const {
398
399 Vector9 xi;
400 Matrix39 D_xiP_state;
401 Vector3 b_v = bodyVelocity(H: F ? &D_xiP_state : 0);
402 double dt22 = 0.5 * dt * dt;
403
404 // Integrate on tangent space. TODO(frank): coriolis?
405 dR(v&: xi) << dt * b_omega;
406 dP(v&: xi) << dt * b_v + dt22 * b_acceleration;
407 dV(v&: xi) << dt * b_acceleration;
408
409 // Bring back to manifold
410 Matrix9 D_newState_xi;
411 NavState newState = retract(xi, H1: F, H2: G1 || G2 ? &D_newState_xi : 0);
412
413 // Derivative wrt state is computed by retract directly
414 // However, as dP(xi) also depends on state, we need to add that contribution
415 if (F) {
416 F->middleRows<3>(startRow: 3) += dt * D_t_t(F) * D_xiP_state;
417 }
418 // derivative wrt acceleration
419 if (G1) {
420 // D_newState_dPxi = D_newState_xi.middleCols<3>(3)
421 // D_dPxi_acc = dt22 * I_3x3
422 // D_newState_dVxi = D_newState_xi.rightCols<3>()
423 // D_dVxi_acc = dt * I_3x3
424 // *G2 = D_newState_acc = D_newState_dPxi * D_dPxi_acc + D_newState_dVxi * D_dVxi_acc
425 *G1 = D_newState_xi.middleCols<3>(startCol: 3) * dt22
426 + D_newState_xi.rightCols<3>() * dt;
427 }
428 // derivative wrt omega
429 if (G2) {
430 // D_newState_dRxi = D_newState_xi.leftCols<3>()
431 // D_dRxi_omega = dt * I_3x3
432 // *G1 = D_newState_omega = D_newState_dRxi * D_dRxi_omega
433 *G2 = D_newState_xi.leftCols<3>() * dt;
434 }
435 return newState;
436}

438//------------------------------------------------------------------------------
439
440// Because our navigation frames are placed on a spinning Earth, we experience two apparent forces on our inertials
441// Let Omega be the Earth's rotation rate in the navigation frame
442// Coriolis acceleration = -2 * (omega X n_v)
443// Centrifugal acceleration (secondOrder) = -omega X (omega X n_t)
444// We would also experience a rotation of (omega*dt) over time - so, counteract by compensating rotation by (-omega * dt)
445// Integrate centrifugal & coriolis accelerations to yield position, velocity perturbations
446Vector9 NavState::coriolis(double dt, const Vector3& omega, bool secondOrder,
447 OptionalJacobian<9, 9> H) const {
448 Rot3 nRb = R_;
449 Point3 n_t = t_, n_v = v_;
450
451 const double dt2 = dt * dt;
452 const Vector3 omega_cross_vel = omega.cross(other: n_v);
453
454 Vector9 n_xi;
455 // Coriolis (first order) acceleration corrections
456 dR(v&: n_xi) << ((-dt) * omega);
457 dP(v&: n_xi) << ((-dt2) * omega_cross_vel); // NOTE(luca): we got rid of the 2 wrt INS paper
458 dV(v&: n_xi) << ((-2.0 * dt) * omega_cross_vel);
459
460 // Centrifugal (second order) acceleration corrections
461 Matrix3 D_c2_nt; // To store Jacobian (if needed/desired)
462 if (secondOrder) {
463 const Vector3 omega_cross2_t = doubleCross(p: omega, q: n_t, H1: nullptr, H2: H ? &D_c2_nt : nullptr);
464 dP(v&: n_xi) -= (0.5 * dt2) * omega_cross2_t;
465 dV(v&: n_xi) -= dt * omega_cross2_t;
466 }
467
468 // Transform correction from navigation frame -> body frame and get Jacobians
469 Vector9 xi;
470 Matrix3 D_dR_R, D_dP_R, D_dV_R;
471 dR(v&: xi) = nRb.unrotate(p: dR(v&: n_xi), H1: H ? &D_dR_R : 0);
472 dP(v&: xi) = nRb.unrotate(p: dP(v&: n_xi), H1: H ? &D_dP_R : 0);
473 dV(v&: xi) = nRb.unrotate(p: dV(v&: n_xi), H1: H ? &D_dV_R : 0);
474
475 // Assemble Jacobians
476 if (H) {
477 H->setZero();
478 const Vector3 omega_b = nRb.unrotate(p: omega);
479 const Matrix3 D_c1_b = skewSymmetric(w: omega_b);
480 H->setZero();
481 D_R_R(H) << D_dR_R;
482 D_t_v(H) << (-dt2) * D_c1_b;
483 D_t_R(H) << D_dP_R;
484 D_v_v(H) << (-2.0 * dt) * D_c1_b;
485 D_v_R(H) << D_dV_R;
486 if (secondOrder) {
487 Matrix3 D_c2_b = D_c1_b * D_c1_b;
488 D_t_t(H) -= (0.5 * dt2) * D_c2_b;
489 D_v_t(H) -= dt * D_c2_b;
490 }
491 }
492 return xi;
493}

495//------------------------------------------------------------------------------
496Vector9 NavState::correctPIM(const Vector9& pim, double dt,
497 const Vector3& n_gravity, const std::optional<Vector3>& omegaCoriolis,
498 bool use2ndOrderCoriolis, OptionalJacobian<9, 9> H1,
499 OptionalJacobian<9, 9> H2) const {
500 const Rot3& nRb = R_;
501 const Velocity3& n_v = v_; // derivative is Ri !
502 const double dt22 = 0.5 * dt * dt;
503
504 Vector9 xi;
505 Matrix3 D_dP_Ri1, D_dP_Ri2, D_dP_nv, D_dV_Ri;
506 dR(v&: xi) = dR(v: pim);
507 dP(v&: xi) = dP(v: pim)
508 + dt * nRb.unrotate(p: n_v, H1: H1 ? &D_dP_Ri1 : 0, H2: H2 ? &D_dP_nv : 0)
509 + dt22 * nRb.unrotate(p: n_gravity, H1: H1 ? &D_dP_Ri2 : 0);
510 dV(v&: xi) = dV(v: pim) + dt * nRb.unrotate(p: n_gravity, H1: H1 ? &D_dV_Ri : 0);
511
512 if (omegaCoriolis) {
513 xi += coriolis(dt, omega: *omegaCoriolis, secondOrder: use2ndOrderCoriolis, H: H1);
514 }
515
516 if (H1 || H2) {
517 Matrix3 Ri = nRb.matrix();
518
519 if (H1) {
520 if (!omegaCoriolis)
521 H1->setZero(); // if coriolis H1 is already initialized
522 D_t_R(H1) += dt * D_dP_Ri1 + dt22 * D_dP_Ri2;
523 D_t_v(H1) += dt * D_dP_nv * Ri;
524 D_v_R(H1) += dt * D_dV_Ri;
525 }
526 if (H2) {
527 H2->setIdentity();
528 }
529 }
530
531 return xi;
532}
