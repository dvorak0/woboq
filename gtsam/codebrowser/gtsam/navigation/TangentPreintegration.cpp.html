32//------------------------------------------------------------------------------
33void TangentPreintegration::resetIntegration() {
34 deltaTij_ = 0.0;
35 preintegrated_.setZero();
36 preintegrated_H_biasAcc_.setZero();
37 preintegrated_H_biasOmega_.setZero();
38}

40//------------------------------------------------------------------------------
41bool TangentPreintegration::equals(const TangentPreintegration& other,
42 double tol) const {
43 return p_->equals(other: *other.p_, tol) && std::abs(x: deltaTij_ - other.deltaTij_) < tol
44 && biasHat_.equals(expected: other.biasHat_, tol)
45 && equal_with_abs_tol(A: preintegrated_, B: other.preintegrated_, tol)
46 && equal_with_abs_tol(A: preintegrated_H_biasAcc_,
47 B: other.preintegrated_H_biasAcc_, tol)
48 && equal_with_abs_tol(A: preintegrated_H_biasOmega_,
49 B: other.preintegrated_H_biasOmega_, tol);
50}

53// See extensive discussion in ImuFactor.lyx
54Vector9 TangentPreintegration::UpdatePreintegrated(const Vector3& a_body,
55 const Vector3& w_body, double dt, const Vector9& preintegrated,
56 OptionalJacobian<9, 9> A, OptionalJacobian<9, 3> B,
57 OptionalJacobian<9, 3> C) {
58 const auto theta = preintegrated.segment<3>(start: 0);
59 const auto position = preintegrated.segment<3>(start: 3);
60 const auto velocity = preintegrated.segment<3>(start: 6);
61
62 // This functor allows for saving computation when exponential map and its
63 // derivatives are needed at the same location in so<3>
64 so3::DexpFunctor local(theta);
65
66 // Calculate exact mean propagation
67 Matrix3 w_tangent_H_theta, invH;
68 const Vector3 w_tangent = // angular velocity mapped back to tangent space
69 local.InvJacobian().applyRight(v: w_body, Hw: A ? &w_tangent_H_theta : 0, Hv: C ? &invH : 0);
70 const Matrix3 nRb = local.expmap(); // nRb: rotation of body in nav frame
71 const Vector3 a_nav = nRb * a_body;
72 const double dt22 = 0.5 * dt * dt;
73
74 Vector9 preintegratedPlus;
75 preintegratedPlus << // new preintegrated vector:
76 theta + w_tangent * dt, // theta
77 position + velocity * dt + a_nav * dt22, // position
78 velocity + a_nav * dt; // velocity
79
80 if (A) {
81 // Exact derivative of nRb*a with respect to theta:
82 const Matrix3 a_nav_H_theta = nRb * skewSymmetric(w: -a_body) * local.rightJacobian();
83
84 A->setIdentity();
85 A->block<3, 3>(startRow: 0, startCol: 0).noalias() += w_tangent_H_theta * dt; // theta
86 A->block<3, 3>(startRow: 3, startCol: 0) = a_nav_H_theta * dt22; // position wrpt theta...
87 A->block<3, 3>(startRow: 3, startCol: 6) = I_3x3 * dt; // .. and velocity
88 A->block<3, 3>(startRow: 6, startCol: 0) = a_nav_H_theta * dt; // velocity wrpt theta
89 }
90 if (B) {
91 B->block<3, 3>(startRow: 0, startCol: 0) = Z_3x3;
92 B->block<3, 3>(startRow: 3, startCol: 0) = nRb * dt22;
93 B->block<3, 3>(startRow: 6, startCol: 0) = nRb * dt;
94 }
95 if (C) {
96 C->block<3, 3>(startRow: 0, startCol: 0) = invH * dt;
97 C->block<3, 3>(startRow: 3, startCol: 0) = Z_3x3;
98 C->block<3, 3>(startRow: 6, startCol: 0) = Z_3x3;
99 }
100
101 return preintegratedPlus;
102}

104//------------------------------------------------------------------------------
105void TangentPreintegration::update(const Vector3& measuredAcc,
106 const Vector3& measuredOmega, const double dt, Matrix9* A, Matrix93* B,
107 Matrix93* C) {
108 // Correct for bias in the sensor frame
109 Vector3 acc = biasHat_.correctAccelerometer(measurement: measuredAcc);
110 Vector3 omega = biasHat_.correctGyroscope(measurement: measuredOmega);
111
112 // Possibly correct for sensor pose by converting to body frame
113 Matrix3 D_correctedAcc_acc, D_correctedAcc_omega, D_correctedOmega_omega;
114 if (p().body_P_sensor) {
115 std::tie(args&: acc, args&: omega) = correctMeasurementsBySensorPose(
116 unbiasedAcc: acc, unbiasedOmega: omega, correctedAcc_H_unbiasedAcc: D_correctedAcc_acc, correctedAcc_H_unbiasedOmega: D_correctedAcc_omega,
117 correctedOmega_H_unbiasedOmega: D_correctedOmega_omega);
118 }
119
120 // Do update
121 deltaTij_ += dt;
122 preintegrated_ = UpdatePreintegrated(a_body: acc, w_body: omega, dt, preintegrated: preintegrated_, A, B, C);
123
124 if (p().body_P_sensor) {
125 // More complicated derivatives in case of non-trivial sensor pose
126 *C *= D_correctedOmega_omega;
127 if (!p().body_P_sensor->translation().isZero())
128 *C += *B * D_correctedAcc_omega;
129 *B *= D_correctedAcc_acc; // NOTE(frank): needs to be last
130 }
131
132 // new_H_biasAcc = new_H_old * old_H_biasAcc + new_H_acc * acc_H_biasAcc
133 // where acc_H_biasAcc = -I_3x3, hence
134 // new_H_biasAcc = new_H_old * old_H_biasAcc - new_H_acc
135 preintegrated_H_biasAcc_ = (*A) * preintegrated_H_biasAcc_ - (*B);
136
137 // new_H_biasOmega = new_H_old * old_H_biasOmega + new_H_omega * omega_H_biasOmega
138 // where omega_H_biasOmega = -I_3x3, hence
139 // new_H_biasOmega = new_H_old * old_H_biasOmega - new_H_omega
140 preintegrated_H_biasOmega_ = (*A) * preintegrated_H_biasOmega_ - (*C);
141}

144Vector9 TangentPreintegration::biasCorrectedDelta(
145 const imuBias::ConstantBias& bias_i, OptionalJacobian<9, 6> H) const {
146 // We correct for a change between bias_i and the biasHat_ used to integrate
147 // This is a simple linear correction with obvious derivatives
148 const imuBias::ConstantBias biasIncr = bias_i - biasHat_;
149 const Vector9 biasCorrected = preintegrated()
150 + preintegrated_H_biasAcc_ * biasIncr.accelerometer()
151 + preintegrated_H_biasOmega_ * biasIncr.gyroscope();
152
153 if (H) {
154 (*H) << preintegrated_H_biasAcc_, preintegrated_H_biasOmega_;
155 }
156 return biasCorrected;
157}

172Vector9 TangentPreintegration::Compose(const Vector9& zeta01,
173 const Vector9& zeta12, double deltaT12, OptionalJacobian<9, 9> H1,
174 OptionalJacobian<9, 9> H2) {
175 const auto t01 = zeta01.segment<3>(start: 0);
176 const auto p01 = zeta01.segment<3>(start: 3);
177 const auto v01 = zeta01.segment<3>(start: 6);
178
179 const auto t12 = zeta12.segment<3>(start: 0);
180 const auto p12 = zeta12.segment<3>(start: 3);
181 const auto v12 = zeta12.segment<3>(start: 6);
182
183 Matrix3 R01_H_t01, R12_H_t12;
184 const Rot3 R01 = Rot3::Expmap(v: t01, H: R01_H_t01);
185 const Rot3 R12 = Rot3::Expmap(v: t12, H: R12_H_t12);
186
187 Matrix3 R02_H_R01, R02_H_R12; // NOTE(frank): R02_H_R12 == Identity
188 const Rot3 R02 = R01.compose(g: R12, H1: R02_H_R01, H2: R02_H_R12);
189
190 Matrix3 t02_H_R02;
191 Vector9 zeta02;
192 const Matrix3 R = R01.matrix();
193 zeta02 << Rot3::Logmap(R: R02, H: t02_H_R02), // theta
194 p01 + v01 * deltaT12 + R * p12, // position
195 v01 + R * v12; // velocity
196
197 if (H1) {
198 H1->setIdentity();
199 D_R_R(H1) = t02_H_R02 * R02_H_R01 * R01_H_t01;
200 D_t_R(H1) = R * skewSymmetric(w: -p12) * R01_H_t01;
201 D_t_v(H1) = I_3x3 * deltaT12;
202 D_v_R(H1) = R * skewSymmetric(w: -v12) * R01_H_t01;
203 }
204
205 if (H2) {
206 H2->setZero();
207 D_R_R(H2) = t02_H_R02 * R02_H_R12 * R12_H_t12;
208 D_t_t(H2) = R;
209 D_v_v(H2) = R;
210 }
211
212 return zeta02;
213}

215//------------------------------------------------------------------------------
216void TangentPreintegration::mergeWith(const TangentPreintegration& pim12,
217 Matrix9* H1, Matrix9* H2) {
218 if (!matchesParamsWith(other: pim12)) {
219 throw std::domain_error(
220 "Cannot merge pre-integrated measurements with different params");
221 }
222
223 if (params()->body_P_sensor) {
224 throw std::domain_error(
225 "Cannot merge pre-integrated measurements with sensor pose yet");
226 }
227
228 const double t01 = deltaTij();
229 const double t12 = pim12.deltaTij();
230 deltaTij_ = t01 + t12;
231
232 const Vector9 zeta01 = preintegrated();
233 Vector9 zeta12 = pim12.preintegrated(); // will be modified.
234
235 const imuBias::ConstantBias bias_incr_for_12 = biasHat() - pim12.biasHat();
236 zeta12 += pim12.preintegrated_H_biasOmega_ * bias_incr_for_12.gyroscope()
237 + pim12.preintegrated_H_biasAcc_ * bias_incr_for_12.accelerometer();
238
239 preintegrated_ = TangentPreintegration::Compose(zeta01, zeta12, deltaT12: t12, H1, H2);
240
241 preintegrated_H_biasAcc_ = (*H1) * preintegrated_H_biasAcc_
242 + (*H2) * pim12.preintegrated_H_biasAcc_;
243
244 preintegrated_H_biasOmega_ = (*H1) * preintegrated_H_biasOmega_
245 + (*H2) * pim12.preintegrated_H_biasOmega_;
246}
