26Vector AttitudeFactor::attitudeError(const Rot3& nRb,
27 OptionalJacobian<2, 3> H) const {
28 if (H) {
29 Matrix23 D_nRotated_R;
30 Matrix22 D_e_nRotated;
31 Unit3 nRotated = nRb.rotate(p: bMeasured_, HR: D_nRotated_R);
32 Vector e = nRef_.errorVector(q: nRotated, H_p: {}, H_q: D_e_nRotated);
33
34 (*H) = D_e_nRotated * D_nRotated_R;
35 return e;
36 } else {
37 Unit3 nRotated = nRb * bMeasured_;
38 return nRef_.errorVector(q: nRotated);
39 }
40}
