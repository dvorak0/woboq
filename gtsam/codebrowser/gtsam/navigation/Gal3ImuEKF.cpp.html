26Gal3ImuEKF::Gal3ImuEKF(const Gal3& X0, const Covariance& P0,
27 const std::shared_ptr<PreintegrationParams>& p,
28 Mode mode)
29 : Base(X0, P0), params_(p), mode_(mode) {
30 // Build process noise Q_ = block_diag(Cg, Ca, Ci, 0)
31 Q_.setZero();
32 Q_.template block<3, 3>(startRow: 0, startCol: 0) = p->gyroscopeCovariance;
33 Q_.template block<3, 3>(startRow: 3, startCol: 3) = p->accelerometerCovariance;
34 Q_.template block<3, 3>(startRow: 6, startCol: 6) = p->integrationCovariance;
35}

37Gal3 Gal3ImuEKF::Dynamics(const Vector3& g_n, const Gal3& X,
38 const Vector3& omega_b, const Vector3& f_b, double dt,
39 Mode mode, OptionalJacobian<10, 10> A) {
40 if (dt <= 0.0) {
41 throw std::invalid_argument("Gal3ImuEKF::Dynamics: dt must be positive");
42 }
43
44 // Calculate W, phi, and U
45 const Gal3 W = (mode == NO_TIME) ? TimeZeroingGravity(g_n, dt)
46 : CompensatedGravity(g_n, dt, t_k: X.time());
47 const Gal3 U = Imu(omega_b, f_b, dt);
48
49 const Gal3 X_next = Base::Dynamics(W, X, U, A);
50 if (A && mode == TRACK_TIME_WITH_COVARIANCE) {
51 // Extra column from state-dependent left factor W(t_k):
52 // right-trivialized increment at W due to Î´t is e_t := [0; 0; -g dt; 0] in
53 // the navigation frame
54 Vector e_t(10);
55 e_t.setZero();
56 e_t.segment<3>(start: 6) = -g_n * dt; // p-block (indices 6..8)
57
58 // Bring to the right-side through the adjoint of X_next and add on to A:
59 A->col(i: 9) += X_next.inverse().Adjoint(xi_base: e_t);
60 }
61 return X_next;
62}
63
64void Gal3ImuEKF::predict(const Vector3& omega_b, const Vector3& f_b,
65 double dt) {
66 if (dt <= 0.0) {
67 throw std::invalid_argument("Gal3ImuEKF::predict: dt must be positive");
68 }
69
70 // Calculate next state, with covariance
71 Gal3::Jacobian A;
72 X_ = Dynamics(g_n: params_->n_gravity, X: X_, omega_b, f_b, dt, mode: mode_, A);
73
74 // Scale continuous-time process noise to the discrete interval [t, t+dt]
75 Covariance Qdt = Q_ * dt;
76
77 // Update covariance
78 P_ = A * P_ * A.transpose() + Qdt;
79}

81const std::shared_ptr<PreintegrationParams>& Gal3ImuEKF::params() const {

83}

85const Vector3& Gal3ImuEKF::gravity() const { return params_->n_gravity; }

87const Gal3ImuEKF::Covariance& Gal3ImuEKF::processNoise() const { return Q_; }
