33namespace gtsam {
34
35/**
36 * PreintegrationBase is the base class for PreintegratedMeasurements
37 * (in ImuFactor) and CombinedPreintegratedMeasurements (in CombinedImuFactor).
38 * It includes the definitions of the preintegrated variables and the methods
39 * to access, print, and compare them.
40 */
41class GTSAM_EXPORT PreintegrationBase {
42 public:
43 typedef imuBias::ConstantBias Bias;
44 typedef PreintegrationParams Params;
45
46 protected:
47 std::shared_ptr<Params> p_;
48
49 /// Acceleration and gyro bias used for preintegration
50 Bias biasHat_;
51
52 /// Time interval from i to j
53 double deltaTij_;
54
55 /// Default constructor for serialization
56 PreintegrationBase() {}
57
58 /// Virtual destructor for serialization
59 virtual ~PreintegrationBase() {}
60
61 public:
62 /// @name Constructors
63 /// @{
64
65 /**
66 * Constructor, initializes the variables in the base class
67 * @param p Parameters, typically fixed in a single application
68 * @param bias Current estimate of acceleration and rotation rate biases
69 */
70 PreintegrationBase(const std::shared_ptr<Params>& p,
71 const imuBias::ConstantBias& biasHat = imuBias::ConstantBias());
72
73 /// @}
74
75 /// @name Basic utilities
76 /// @{
77 /// Re-initialize PreintegratedMeasurements
78 virtual void resetIntegration() = 0;
79
80 /// @name Basic utilities
81 /// @{
82 /// Re-initialize PreintegratedMeasurements and set new bias
83 void resetIntegrationAndSetBias(const Bias& biasHat);
84
85 /// check parameters equality: checks whether shared pointer points to same Params object.
86 bool matchesParamsWith(const PreintegrationBase& other) const {
87 return p_.get() == other.p_.get();
88 }
89
90 /// shared pointer to params
91 const std::shared_ptr<Params>& params() const {
92 return p_;
93 }
94
95 /// const reference to params
96 Params& p() const {
97 return *p_;
98 }
99
100 /// @}
101
102 /// @name Instance variables access
103 /// @{
104 const imuBias::ConstantBias& biasHat() const { return biasHat_; }
105 double deltaTij() const { return deltaTij_; }
106
107 virtual Vector3 deltaPij() const = 0;
108 virtual Vector3 deltaVij() const = 0;
109 virtual Rot3 deltaRij() const = 0;
110 virtual NavState deltaXij() const = 0;
111
112 // Exposed for MATLAB
113 Vector6 biasHatVector() const { return biasHat_.vector(); }
114 /// @}
115
116 /// @name Testable
117 /// @{
118 GTSAM_EXPORT friend std::ostream& operator<<(std::ostream& os, const PreintegrationBase& pim);
119 virtual void print(const std::string& s="") const;
120 /// @}
121
122 /// @name Main functionality
123 /// @{
124
125 /**
126 * Subtract estimate and correct for sensor pose
127 * Compute the derivatives due to non-identity body_P_sensor (rotation and centrifugal acc)
128 * Ignore D_correctedOmega_measuredAcc as it is trivially zero
129 */
130 std::pair<Vector3, Vector3> correctMeasurementsBySensorPose(
131 const Vector3& unbiasedAcc, const Vector3& unbiasedOmega,
132 OptionalJacobian<3, 3> correctedAcc_H_unbiasedAcc = {},
133 OptionalJacobian<3, 3> correctedAcc_H_unbiasedOmega = {},
134 OptionalJacobian<3, 3> correctedOmega_H_unbiasedOmega = {}) const;
135
136 /**
137 * Update preintegrated measurements and get derivatives
138 * It takes measured quantities in the j frame
139 * Modifies preintegrated quantities in place after correcting for bias and possibly sensor pose
140 */
141 virtual void update(const Vector3& measuredAcc, const Vector3& measuredOmega,
142 const double dt, Matrix9* A, Matrix93* B, Matrix93* C) = 0;
143
144 /// Version without derivatives
145 virtual void integrateMeasurement(const Vector3& measuredAcc,
146 const Vector3& measuredOmega, const double dt);
147
148 /// Given the estimate of the bias, return a NavState tangent vector
149 /// summarizing the preintegrated IMU measurements so far
150 virtual Vector9 biasCorrectedDelta(const imuBias::ConstantBias& bias_i,
151 OptionalJacobian<9, 6> H = {}) const = 0;
152
153 /// Predict state at time j
154 NavState predict(const NavState& state_i, const imuBias::ConstantBias& bias_i,
155 OptionalJacobian<9, 9> H1 = {},
156 OptionalJacobian<9, 6> H2 = {}) const;
157
158 /// Calculate error given navStates
159 Vector9 computeError(const NavState& state_i, const NavState& state_j,
160 const imuBias::ConstantBias& bias_i,
161 OptionalJacobian<9, 9> H1, OptionalJacobian<9, 9> H2,
162 OptionalJacobian<9, 6> H3) const;
163
164 /**
165 * Compute errors w.r.t. preintegrated measurements and jacobians
166 * wrt pose_i, vel_i, bias_i, pose_j, bias_j
167 */
168 Vector9 computeErrorAndJacobians(const Pose3& pose_i, const Vector3& vel_i,
169 const Pose3& pose_j, const Vector3& vel_j,
170 const imuBias::ConstantBias& bias_i,
171 OptionalJacobian<9, 6> H1 = {}, OptionalJacobian<9, 3> H2 = {},
172 OptionalJacobian<9, 6> H3 = {}, OptionalJacobian<9, 3> H4 = {},
173 OptionalJacobian<9, 6> H5 = {}) const;
174
175 private:
176#if GTSAM_ENABLE_BOOST_SERIALIZATION
177 /** Serialization function */
178 friend class boost::serialization::access;
179 template<class ARCHIVE>
180 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
181 ar & BOOST_SERIALIZATION_NVP(p_);
182 ar & BOOST_SERIALIZATION_NVP(biasHat_);
183 ar & BOOST_SERIALIZATION_NVP(deltaTij_);
184 }
185#endif
186
187 public:
188 GTSAM_MAKE_ALIGNED_OPERATOR_NEW
189};
190
191} /// namespace gtsam
