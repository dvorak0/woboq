50//------------------------------------------------------------------------------
51void PreintegratedAhrsMeasurements::integrateMeasurement(
52 const Vector3& measuredOmega, double deltaT) {
53 // 1. integrate (handles bias + body_P_sensor rotation internally)
54 // Fr is the Jacobian of the new preintegrated rotation w.r.t. the previous one.
55 Matrix3 Fr;
56 PreintegratedRotation::integrateGyroMeasurement(measuredOmega, biasHat: biasHat_,
57 deltaT, F: &Fr);
58
59 // 2. Calculate noise in the body frame
60 Matrix3 SigmaBody = p().gyroscopeCovariance;
61 if (p().body_P_sensor) {
62 const Matrix3& bRs = p().body_P_sensor->rotation().matrix(); // body‚Üêsensor
63 SigmaBody = bRs * SigmaBody * bRs.transpose();
64 }
65
66 // First order uncertainty propagation:
67 // new_cov = Fr * old_cov * Fr.transpose() + new_noise
68 // The deltaT allows to pass from continuous time noise to discrete time
69 // noise. Comparing with the IMUFactor.cpp implementation, the latter is an
70 // approximation for C * (wCov / dt) * C.transpose(), with C \approx I * dt.
71 preintMeasCov_ = Fr * preintMeasCov_ * Fr.transpose() + SigmaBody * deltaT;
72}

74//------------------------------------------------------------------------------
75Rot3 PreintegratedAhrsMeasurements::predict(
76 const Rot3& Ri, const Vector3& bias, gtsam::OptionalJacobian<3, 3> H1,
77 gtsam::OptionalJacobian<3, 3> H2) const {
78 // Use H2 as an in/out parameter to hold the Jacobian of the bias-corrected
79 // rotation w.r.t. the bias increment. This is an efficient C++ pattern.
80 const Vector3 biasOmegaIncr = bias - biasHat_;
81 const Rot3 biascorrected = this->biascorrectedDeltaRij(biasOmegaIncr, H: H2);
82
83 // We handle the common case of no Coriolis correction first.
84 if (!p().omegaCoriolis) {
85 // ---- FAST PATH ----
86 // In this path, the Jacobian of compose wrt its second argument is
87 // identity, so H2 already holds the final Jacobian wrt bias.
88 return Ri.compose(g: biascorrected, H1);
89
90 } else {
91 // ---- SLOW PATH ----
92 // Calculate Coriolis effects and its derivative w.r.t. Ri.
93 Matrix3 D_coriolis_Ri;
94 const Vector3 coriolis =
95 integrateCoriolis(Ri, H: H1 ? &D_coriolis_Ri : nullptr);
96
97 // Compose bias and Coriolis corrections.
98 Matrix3 D_finalDelta_biasDelta, D_finalDelta_coriolis;
99 const Rot3 finalDeltaRij =
100 biascorrected.expmap(v: -coriolis, H1: H2 ? &D_finalDelta_biasDelta : nullptr,
101 H2: H1 ? &D_finalDelta_coriolis : nullptr);
102
103 // Predict final orientation, getting the direct Jacobian wrt Ri.
104 const Rot3 predicted_Rj = Ri.compose(g: finalDeltaRij, H1);
105
106 // Augment Jacobians with the indirect paths.
107 if (H1) {
108 // Add in indirect path: Ri -> coriolis -> finalDeltaRij
109 // H1 is now D_predicted_Rj_wrt_Ri
110 *H1 -= D_finalDelta_coriolis * D_coriolis_Ri;
111 }
112
113 if (H2) {
114 // Apply the chain rule for the bias Jacobian, updating H2 in-place.
115 // H2 was D_biascorrected_wrt_bias, now it is D_predicted_Rj_wrt_bias
116 *H2 = D_finalDelta_biasDelta * (*H2);
117 }
118
119 return predicted_Rj;
120 }
121}

123//------------------------------------------------------------------------------
124Vector3 PreintegratedAhrsMeasurements::computeError(
125 const Rot3& Ri, const Rot3& Rj, const Vector3& bias,
126 gtsam::OptionalJacobian<3, 3> H1, gtsam::OptionalJacobian<3, 3> H2,
127 gtsam::OptionalJacobian<3, 3> H3) const {
128 // Predict orientation at time j
129 Matrix3 D_predict_Ri, D_predict_bias;
130 Rot3 predicted_Rj = predict(Ri, bias, H1: H1 ? &D_predict_Ri : nullptr,
131 H2: H3 ? &D_predict_bias : nullptr);
132
133 // Compute the error vector: log(Rj.inverse() * predicted_Rj)
134 Matrix3 D_error_Rj, D_error_predict;
135 Vector3 error = Rj.logmap(g: predicted_Rj, H1: H2 ? &D_error_Rj : nullptr,
136 H2: H1 || H3 ? &D_error_predict : nullptr);
137
138 // Jacobians using the chain rule
139 if (H1) *H1 = D_error_predict * D_predict_Ri;
140 if (H2) *H2 = D_error_Rj;
141 if (H3) *H3 = D_error_predict * D_predict_bias;
142
143 return error;
144}

198//------------------------------------------------------------------------------
199Rot3 AHRSFactor::predict(const Rot3& Ri, const Vector3& bias,
200 const PreintegratedAhrsMeasurements& pim,
201 const Vector3& omegaCoriolis,
202 const std::optional<Pose3>& body_P_sensor) {
203 auto p = std::make_shared<PreintegratedAhrsMeasurements::Params>(args&: pim.p());
204 p->omegaCoriolis = omegaCoriolis;
205 p->body_P_sensor = body_P_sensor;
206 PreintegratedAhrsMeasurements newPim = pim;
207 newPim.p_ = p;
208 return newPim.predict(Ri, bias);
209}
