57/**
58 * Scenario with constant twist 3D trajectory.
59 * Note that a plane flying level does not feel any acceleration: gravity is
60 * being perfectly compensated by the lift generated by the wings, and drag is
61 * compensated by thrust. The accelerometer will add the gravity component back
62 * in, as modeled by the specificForce method in ScenarioRunner.
63 */
64class GTSAM_EXPORT ConstantTwistScenario : public Scenario {
65 public:
66 /// Construct scenario with constant twist [w,v]
67 ConstantTwistScenario(const Vector3& w, const Vector3& v,
68 const Pose3& nTb0 = Pose3())
69 : twist_((Vector6() << w, v).finished()), a_b_(w.cross(other: v)), nTb0_(nTb0) {}
70
71 Pose3 pose(double t) const override;
72 Vector3 omega_b(double t) const override;
73 Vector3 velocity_n(double t) const override;
74 Vector3 acceleration_n(double t) const override;
75
76 private:
77 const Vector6 twist_;
78 const Vector3 a_b_; // constant centripetal acceleration in body = w_b * v_b
79 const Pose3 nTb0_;
80};

102/**
103 * @brief A scenario defined by discrete ground-truth measurements over time.
104 *
105 * This class stores time-stamped ground truth values for pose, angular
106 * velocity, linear velocity, and linear acceleration. When queried for a
107 * value at a specific time `t`, it performs linear interpolation for vector
108 * quantities and manifold interpolation for poses. If `t` is outside the
109 * range of the provided timestamps, the value at the nearest endpoint is
110 * returned (i.e., the data is clamped).
111 */
112class GTSAM_EXPORT DiscreteScenario : public Scenario {
113 public:
114 /**
115 * Constructor.
116 * @param poses Map of timestamps to ground truth poses (nTb).
117 * @param angularVelocities_b Map of timestamps to ground truth angular velocities in body frame.
118 * @param velocities_n Map of timestamps to ground truth linear velocities in nav frame.
119 * @param accelerations_n Map of timestamps to ground truth linear accelerations in nav frame.
120 */
121 DiscreteScenario(const map<double, Pose3>& poses,
122 const map<double, Vector3>& angularVelocities_b,
123 const map<double, Vector3>& velocities_n,
124 const map<double, Vector3>& accelerations_n)
125 : poses_(poses),
126 angularVelocities_b_(angularVelocities_b),
127 velocities_n_(velocities_n),
128 accelerations_n_(accelerations_n) {
129 if (poses_.empty() || angularVelocities_b_.empty() ||
130 velocities_n_.empty() || accelerations_n_.empty()) {
131 throw invalid_argument(
132 "Input maps for DiscreteScenario cannot be empty.");
133 }
134
135 // Since std::map is sorted, the first element has the smallest key
136 // and the last element has the largest key. We can access the last
137 // element's key efficiently using rbegin().
138 double min_t = poses_.begin()->first;
139 double max_t = poses_.rbegin()->first;
140
141 min_t = min(a: min_t, b: angularVelocities_b_.begin()->first);
142 max_t = max(a: max_t, b: angularVelocities_b_.rbegin()->first);
143
144 min_t = min(a: min_t, b: velocities_n_.begin()->first);
145 max_t = max(a: max_t, b: velocities_n_.rbegin()->first);
146
147 min_t = min(a: min_t, b: accelerations_n_.begin()->first);
148 max_t = max(a: max_t, b: accelerations_n_.rbegin()->first);
149
150 // Calculate and assign the duration.
151 t_ = max_t - min_t;
152 }

154 /**
155 * @brief Named constructor to create a scenario from a CSV file.
156 *
157 * The CSV file should contain a header row followed by data rows.
158 * All timestamps will be normalized so that the first timestamp in the file
159 * corresponds to t=0 for the scenario.
160 *
161 * CSV is expected to contain the following columns:
162 * t,q_w,q_x,q_y,q_z,v_x,v_y,v_z,p_x,p_y,p_z,w_x,w_y,w_z,a_x,a_y,a_z
163 * Other columns will be ignored.
164 *
165 * @param csv_filepath Path to the CSV file.
166 * @return A constructed DiscreteScenario.
167 * @throws runtime_error if file cannot be opened or is malformed.
168 */
169 static DiscreteScenario FromCSV(const string& csv_filepath);

179 /// @brief The duration of time between the first timestamp and the last.
180 double duration() const;
181
182 private:
183 /// Map from timestamp to pose.
184 map<double, Pose3> poses_;
185 /// Map from timestamp to angular velocity in body frame.
186 map<double, Vector3> angularVelocities_b_;
187 /// Map from timestamp to velocity in navigation frame.
188 map<double, Vector3> velocities_n_;

194 /**
195 * @brief Internal helper to interpolate values in a time-stamped map.
196 *
197 * This function finds the two values that bracket the given time `t` and
198 * performs interpolation. It uses gtsam::interpolate, which correctly
199 * handles manifold types like Pose3 as well as vector types.
200 *
201 * @tparam T The type of the value to interpolate (e.g., Pose3, Vector3).
202 * @param values The map of time-stamped values.
203 * @param t The time at which to interpolate.
204 * @return The interpolated value.
205 */
206 template <typename T>
207 T interpolate(const map<double, T>& values, double t) const {
208 // Find the first element with a timestamp >= t
209 auto it2 = values.lower_bound(t);
210
211 // If t is before or at the very first measurement, return the first value.
212 if (it2 == values.begin()) {
213 return it2->second;
214 }
215
216 // If t is after the very last measurement, return the last value.
217 if (it2 == values.end()) {
218 return values.rbegin()->second;
219 }
220
221 // Standard case: t is between it1 and it2.
222 auto it1 = prev(it2);
223
224 const double t1 = it1->first;
225 const T& value1 = it1->second;
226 const double t2 = it2->first;
227 const T& value2 = it2->second;
228
229 const double dt = t2 - t1;
230 // Avoid division by zero if timestamps are identical
231 if (abs(x: dt) < 1e-9) {
232 return value1;
233 }
234
235 // Calculate interpolation fraction
236 const double alpha = (t - t1) / dt;
237
238 // Use GTSAM's manifold interpolation
239 return gtsam::interpolate<T>(value1, value2, alpha);
240 }
