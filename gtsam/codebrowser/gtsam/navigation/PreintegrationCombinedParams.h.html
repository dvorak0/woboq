31namespace gtsam {
32
33/// Parameters for pre-integration using PreintegratedCombinedMeasurements:
34/// Usage: Create just a single Params and pass a shared pointer to the
35/// constructor
36struct GTSAM_EXPORT PreintegrationCombinedParams : PreintegrationParams {
37 Matrix3 biasAccCovariance; ///< continuous-time "Covariance" describing
38 ///< accelerometer bias random walk
39 Matrix3 biasOmegaCovariance; ///< continuous-time "Covariance" describing
40 ///< gyroscope bias random walk
41
42 /// Default constructor makes uninitialized params struct.
43 /// Used for serialization.
44 PreintegrationCombinedParams()
45 : biasAccCovariance(I_3x3),
46 biasOmegaCovariance(I_3x3) {
47#ifdef GTSAM_ALLOW_DEPRECATED_SINCE_V43
48 biasAccOmegaInt.setZero();
49#endif
50 }
51
52 /// See two named constructors below for good values of n_gravity in body
53 /// frame
54 PreintegrationCombinedParams(const Vector3& n_gravity_)
55 : PreintegrationParams(n_gravity_),
56 biasAccCovariance(I_3x3),
57 biasOmegaCovariance(I_3x3) {
58#ifdef GTSAM_ALLOW_DEPRECATED_SINCE_V43
59 biasAccOmegaInt.setZero();
60#endif
61 }
62
63 // Default Params for a Z-down navigation frame, such as NED: gravity points
64 // along positive Z-axis
65 static std::shared_ptr<PreintegrationCombinedParams> MakeSharedD(
66 double g = 9.81) {
67 return std::shared_ptr<PreintegrationCombinedParams>(
68 new PreintegrationCombinedParams(Vector3(0, 0, g)));
69 }
70
71 // Default Params for a Z-up navigation frame, such as ENU: gravity points
72 // along negative Z-axis
73 static std::shared_ptr<PreintegrationCombinedParams> MakeSharedU(
74 double g = 9.81) {
75 return std::shared_ptr<PreintegrationCombinedParams>(
76 new PreintegrationCombinedParams(Vector3(0, 0, -g)));
77 }
78
79 void print(const std::string& s = "") const override;
80 bool equals(const PreintegratedRotationParams& other,
81 double tol) const override;
82
83 void setBiasAccCovariance(const Matrix3& cov) { biasAccCovariance = cov; }
84 void setBiasOmegaCovariance(const Matrix3& cov) { biasOmegaCovariance = cov; }
85
86 const Matrix3& getBiasAccCovariance() const { return biasAccCovariance; }
87 const Matrix3& getBiasOmegaCovariance() const { return biasOmegaCovariance; }
88
89#ifdef GTSAM_ALLOW_DEPRECATED_SINCE_V43
90 Matrix6 biasAccOmegaInt;
91 /// @deprecated: biasAccOmegaInt is no longer used. Use a prior on first bias instead.
92 void setBiasAccOmegaInit(const Matrix6& cov) {
93 std::cerr << "Warning: setBiasAccOmegaInit() is deprecated and no longer used." << std::endl;
94 biasAccOmegaInt = cov;
95 }
96 /// @deprecated: biasAccOmegaInt is no longer used. Use a prior on first bias instead.
97 const Matrix6& getBiasAccOmegaInit() const {
98 std::cerr << "Warning: getBiasAccOmegaInit() is deprecated and no longer used." << std::endl;
99 return biasAccOmegaInt;
100 }
101#endif
102
103 private:
104#if GTSAM_ENABLE_BOOST_SERIALIZATION
105 /** Serialization function */
106 friend class boost::serialization::access;
107 template <class ARCHIVE>
108 void serialize(ARCHIVE& ar, const unsigned int /*version*/) {
109 namespace bs = ::boost::serialization;
110 ar& BOOST_SERIALIZATION_BASE_OBJECT_NVP(PreintegrationParams);
111 ar& BOOST_SERIALIZATION_NVP(biasAccCovariance);
112 ar& BOOST_SERIALIZATION_NVP(biasOmegaCovariance);
113#ifdef GTSAM_ALLOW_DEPRECATED_SINCE_V43
114 ar& BOOST_SERIALIZATION_NVP(biasAccOmegaInt);
115#endif
116 }
117#endif
118
119 public:
120 GTSAM_MAKE_ALIGNED_OPERATOR_NEW
121};
122
123} // namespace gtsam
