26/**
27 * Prior on position in a Cartesian frame.
28 * If there exists a non-zero lever arm between body frame and GPS
29 * antenna, instead use GPSFactorArm.
30 * Possibilities include:
31 * ENU: East-North-Up navigation frame at some local origin
32 * NED: North-East-Down navigation frame at some local origin
33 * ECEF: Earth-centered Earth-fixed, origin at Earth's center
34 * See Farrell08book or e.g. http://www.dirsig.org/docs/new/coordinates.html
35 * @ingroup navigation
36 */
37class GTSAM_EXPORT GPSFactor: public NoiseModelFactorN<Pose3> {
38
39private:
40
41 typedef NoiseModelFactorN<Pose3> Base;
42
43 Point3 nT_; ///< Position measurement in cartesian coordinates (navigation frame)
44
45public:
46
47 // Provide access to the Matrix& version of evaluateError:
48 using Base::evaluateError;
49
50 /// shorthand for a smart pointer to a factor
51 typedef std::shared_ptr<GPSFactor> shared_ptr;
52
53 /// Typedef to this class
54 typedef GPSFactor This;
55
56 /** default constructor - only use for serialization */
57 GPSFactor(): nT_(0, 0, 0) {}
58
59 ~GPSFactor() override {}
60
61 /**
62 * @brief Constructor from a measurement in a Cartesian frame.
63 * Use GeographicLib to convert from geographic (latitude and longitude) coordinates
64 * @param key of the Pose3 variable that will be constrained
65 * @param gpsIn measurement already in correct coordinates
66 * @param model Gaussian noise model
67 */
68 GPSFactor(Key key, const Point3& gpsIn, const SharedNoiseModel& model) :
69 Base(model, key), nT_(gpsIn) {
70 }
71
72 /// @return a deep copy of this factor
73 gtsam::NonlinearFactor::shared_ptr clone() const override {
74 return std::static_pointer_cast<gtsam::NonlinearFactor>(
75 r: gtsam::NonlinearFactor::shared_ptr(new This(*this)));
76 }
77
78 /// print
79 void print(const std::string& s = "", const KeyFormatter& keyFormatter =
80 DefaultKeyFormatter) const override;
81
82 /// equals
83 bool equals(const NonlinearFactor& expected, double tol = 1e-9) const override;
84
85 /// vector of errors
86 Vector evaluateError(const Pose3& nTb, OptionalMatrixType H) const override;
87
88 /// return the measurement, in the navigation frame
89 inline const Point3 & measurementIn() const {
90 return nT_;
91 }
92
93 /**
94 * Convenience function to estimate state at time t, given two GPS
95 * readings (in local NED Cartesian frame) bracketing t
96 * Assumes roll is zero, calculates yaw and pitch from NED1->NED2 vector.
97 */
98 static std::pair<Pose3, Vector3> EstimateState(double t1, const Point3& NED1,
99 double t2, const Point3& NED2, double timestamp);
100
101private:
102
103#if GTSAM_ENABLE_BOOST_SERIALIZATION ///
104 /// Serialization function
105 friend class boost::serialization::access;
106 template<class ARCHIVE>
107 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
108 // NoiseModelFactor1 instead of NoiseModelFactorN for backward compatibility
109 ar
110 & boost::serialization::make_nvp(n: "NoiseModelFactor1",
111 v&: boost::serialization::base_object<Base>(d&: *this));
112 ar & BOOST_SERIALIZATION_NVP(nT_);
113 }
114#endif
115};

117/**
118 * Version of GPSFactor (for Pose3) with lever arm between GPS and Body frame.
119 * Because the actual location of the antenna depends on both position and
120 * attitude, providing a lever arm makes components of the attitude observable
121 * and accounts for position measurement vs state discrepancies while turning.
122 * @ingroup navigation
123 */
124class GTSAM_EXPORT GPSFactorArm: public NoiseModelFactorN<Pose3> {
125
126private:
127
128 typedef NoiseModelFactorN<Pose3> Base;
129
130 Point3 nT_; ///< Position measurement in cartesian coordinates (navigation frame)
131 Point3 bL_; ///< bL_ is a lever arm in the body frame, denoting the 3D
132 ///< position of the GPS antenna in the body frame
133
134public:
135
136 // Provide access to the Matrix& version of evaluateError:
137 using Base::evaluateError;
138
139 /// shorthand for a smart pointer to a factor
140 typedef std::shared_ptr<GPSFactorArm> shared_ptr;
141
142 /// Typedef to this class
143 typedef GPSFactorArm This;
144
145 /// default constructor - only use for serialization
146 GPSFactorArm():nT_(0, 0, 0), bL_(0, 0, 0) {}
147
148 ~GPSFactorArm() override {}
149
150 /** Constructor from a measurement in a Cartesian frame.
151 * @param key key of the Pose3 variable related to this measurement
152 * @param gpsIn gps measurement, in Cartesian navigation frame
153 * @param leverArm translation from the body frame origin to the gps antenna, in body frame
154 * @param model Gaussian noise model
155 */
156 GPSFactorArm(Key key, const Point3& gpsIn, const Point3& leverArm, const SharedNoiseModel& model) :
157 Base(model, key), nT_(gpsIn), bL_(leverArm) {
158 }
159
160 /// @return a deep copy of this factor
161 gtsam::NonlinearFactor::shared_ptr clone() const override {
162 return std::static_pointer_cast<gtsam::NonlinearFactor>(
163 r: gtsam::NonlinearFactor::shared_ptr(new This(*this)));
164 }
165
166 /// print
167 void print(const std::string& s = "", const KeyFormatter& keyFormatter =
168 DefaultKeyFormatter) const override;
169
170 /// equals
171 bool equals(const NonlinearFactor& expected, double tol = 1e-9) const override;
172
173 /// vector of errors
174 Vector evaluateError(const Pose3& nTb, OptionalMatrixType H) const override;
175
176 /// return the measurement, in the navigation frame
177 inline const Point3 & measurementIn() const {
178 return nT_;
179 }
180
181 /// return the lever arm, a position in the body frame
182 inline const Point3 & leverArm() const {
183 return bL_;
184 }
185
186};
