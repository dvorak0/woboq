18#pragma once

20#include <gtsam/geometry/Pose3.h>
21#include <gtsam/navigation/NavState.h>
22#include <gtsam/nonlinear/NonlinearFactor.h>
23
24namespace gtsam {
25
26/**
27 * Prior on height in a cartesian frame.
28 * Receive barometric pressure in kilopascals
29 * Model with a slowly moving bias to capture differences
30 * between the height and the standard atmosphere
31 * https://www.grc.nasa.gov/www/k-12/airplane/atmosmet.html
32 * @ingroup navigation
33 */
34class GTSAM_EXPORT BarometricFactor : public NoiseModelFactorN<Pose3, double> {
35 private:
36 typedef NoiseModelFactorN<Pose3, double> Base;
37
38 double nT_; ///< Height Measurement based on a standard atmosphere
39
40 public:
41
42 // Provide access to the Matrix& version of evaluateError:
43 using Base::evaluateError;
44
45 /// shorthand for a smart pointer to a factor
46 typedef std::shared_ptr<BarometricFactor> shared_ptr;
47
48 /// Typedef to this class
49 typedef BarometricFactor This;
50
51 /** default constructor - only use for serialization */
52 BarometricFactor() : nT_(0) {}
53
54 ~BarometricFactor() override {}
55
56 /**
57 * @brief Constructor from a measurement of pressure in KPa.
58 * @param key of the Pose3 variable that will be constrained
59 * @param key of the barometric bias that will be constrained
60 * @param baroIn measurement in KPa
61 * @param model Gaussian noise model 1 dimension
62 */
63 BarometricFactor(Key key, Key baroKey, const double& baroIn,
64 const SharedNoiseModel& model)
65 : Base(model, key, baroKey), nT_(heightOut(n: baroIn)) {}
66
67 /// @return a deep copy of this factor
68 gtsam::NonlinearFactor::shared_ptr clone() const override {
69 return std::static_pointer_cast<gtsam::NonlinearFactor>(
70 r: gtsam::NonlinearFactor::shared_ptr(new This(*this)));
71 }
72
73 /// print
74 void print(
75 const std::string& s = "",
76 const KeyFormatter& keyFormatter = DefaultKeyFormatter) const override;
77
78 /// equals
79 bool equals(const NonlinearFactor& expected,
80 double tol = 1e-9) const override;
81
82 /// vector of errors
83 Vector evaluateError(const Pose3& p, const double& b,
84 OptionalMatrixType H, OptionalMatrixType H2) const override;
85
86 inline const double& measurementIn() const { return nT_; }
87
88 inline double heightOut(double n) const {
89 // From https://www.grc.nasa.gov/www/k-12/airplane/atmosmet.html
90 return (std::pow(x: n / 101.29, y: 1. / 5.256) * 288.08 - 273.1 - 15.04) /
91 -0.00649;
92 }
93
94 inline double baroOut(const double& meters) const {
95 double temp = 15.04 - 0.00649 * meters;
96 return 101.29 * std::pow(x: ((temp + 273.1) / 288.08), y: 5.256);
97 }
98
99 private:
100#if GTSAM_ENABLE_BOOST_SERIALIZATION ///
101 /// Serialization function
102 friend class boost::serialization::access;
103 template <class ARCHIVE>
104 void serialize(ARCHIVE& ar, const unsigned int /*version*/) {
105 // NoiseModelFactor1 instead of NoiseModelFactorN for backward compatibility
106 ar& boost::serialization::make_nvp(
107 n: "NoiseModelFactor1",
108 v&: boost::serialization::base_object<Base>(d&: *this));
109 ar& BOOST_SERIALIZATION_NVP(nT_);
110 }
111#endif
112};
