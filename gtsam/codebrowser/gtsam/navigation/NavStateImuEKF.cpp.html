26NavStateImuEKF::NavStateImuEKF(const NavState& X0, const Covariance& P0,
27 const std::shared_ptr<PreintegrationParams>& p)
28 : Base(X0, P0), params_(p) {
29 // Build process noise Q_ = block_diag(Cg, Ci, Ca)
30 Q_.setZero();
31 Q_.template block<3, 3>(startRow: 0, startCol: 0) = p->gyroscopeCovariance;
32 Q_.template block<3, 3>(startRow: 3, startCol: 3) = p->integrationCovariance;
33 Q_.template block<3, 3>(startRow: 6, startCol: 6) = p->accelerometerCovariance;
34}

36NavState NavStateImuEKF::Imu(const Vector3& omega_b, const Vector3& f_b,
37 double dt) {
38 const Vector3 phi_b = omega_b * dt;
39 const so3::DexpFunctor local(phi_b);
40 const Matrix3 J_l = local.Jacobian().left();
41 const Matrix3 N_l = local.Gamma().left();
42 return {Rot3::Expmap(v: phi_b), N_l * f_b * dt * dt, J_l * f_b * dt};
43}

45NavState NavStateImuEKF::Dynamics(const Vector3& g_n, const NavState& X,
46 const Vector3& omega_b, const Vector3& f_b,
47 double dt, OptionalJacobian<9, 9> A) {
48 if (dt <= 0.0) {
49 throw std::invalid_argument(
50 "NavStateImuEKF::Dynamics: dt must be positive");
51 }
52
53 // Calculate W, phi, and U
54 const NavState W = Gravity(g_n, dt);
55 NavState::AutonomousFlow phi{.dt: dt}; // Φ: velocity acts on position
56 const NavState U = Imu(omega_b, f_b, dt);
57
58 return Base::Dynamics(W, phi, X, U, A);
59}
60
61void NavStateImuEKF::predict(const Vector3& omega_b, const Vector3& f_b,
62 double dt) {
63 if (dt <= 0.0) {
64 throw std::invalid_argument("NavStateImuEKF::predict: dt must be positive");
65 }
66
67 // Calculate W, phi, and U
68 const NavState W = Gravity(g_n: params_->n_gravity, dt);
69 NavState::AutonomousFlow phi{.dt: dt}; // Φ: velocity acts on position
70 const NavState U = Imu(omega_b, f_b, dt);
71
72 // Scale continuous-time process noise to the discrete interval [t, t+dt]
73 Covariance Qdt = Q_ * dt;
74
75 // EKF predict
76 Base::predict(W, phi, U, Q: Qdt);
77}
