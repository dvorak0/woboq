26/**
27 * @class AttitudeFactor
28 *
29 * @brief Base class for an attitude factor that constrains the rotation between
30 * body and navigation frames.
31 *
32 * This factor enforces that when the measured direction in the body frame
33 * (e.g., from an IMU accelerometer) is rotated into the navigation frame using the
34 * rotation variable, it should align with a known reference direction in the
35 * navigation frame (e.g., gravity vector).
36 *
37 * Mathematically, the error is zero when:
38 * nRb * bMeasured == nRef
39 *
40 * This is useful for incorporating absolute orientation measurements into the
41 * factor graph.
42 *
43 * @ingroup navigation
44 */
45class AttitudeFactor {
46 protected:
47 Unit3 nRef_, bMeasured_; ///< Position measurement in
48
49 public:
50 /** default constructor - only use for serialization */
51 AttitudeFactor() {}
52
53 /**
54 * @brief Constructor
55 * @param nRef Reference direction in the navigation frame (e.g., gravity).
56 * @param bMeasured Measured direction in the body frame (e.g., from IMU
57 * accelerometer). Default is Unit3(0, 0, 1).
58 */
59 AttitudeFactor(const Unit3& nRef, const Unit3& bMeasured = Unit3(0, 0, 1))
60 : nRef_(nRef), bMeasured_(bMeasured) {}
61
62 /** vector of errors */
63 Vector attitudeError(const Rot3& p, OptionalJacobian<2, 3> H = {}) const;
64
65 const Unit3& nRef() const { return nRef_; }
66 const Unit3& bMeasured() const { return bMeasured_; }
67
68#ifdef GTSAM_ALLOW_DEPRECATED_SINCE_V43
69 [[deprecated("Use nRef() instead")]]
70 const Unit3& nZ() const { return nRef_; }
71
72 [[deprecated("Use bMeasured() instead")]]
73 const Unit3& bRef() const { return bMeasured_; }
74#endif
75
76#if GTSAM_ENABLE_BOOST_SERIALIZATION
77 /** Serialization function */
78 friend class boost::serialization::access;
79 template <class ARCHIVE>
80 void serialize(ARCHIVE& ar, const unsigned int /*version*/) {
81 ar& boost::serialization::make_nvp(n: "nRef_", v&: nRef_);
82 ar& boost::serialization::make_nvp(n: "bMeasured_", v&: bMeasured_);
83 }
84#endif
85};

87/**
88 * Version of AttitudeFactor for Rot3
89 * @ingroup navigation
90 */
91class GTSAM_EXPORT Rot3AttitudeFactor : public NoiseModelFactorN<Rot3>,
92 public AttitudeFactor {
93 typedef NoiseModelFactorN<Rot3> Base;
94
95 public:
96 // Provide access to the Matrix& version of evaluateError:
97 using Base::evaluateError;
98
99 /// shorthand for a smart pointer to a factor
100 typedef std::shared_ptr<Rot3AttitudeFactor> shared_ptr;
101
102 /// Typedef to this class
103 typedef Rot3AttitudeFactor This;
104
105 /** default constructor - only use for serialization */
106 Rot3AttitudeFactor() {}
107
108 ~Rot3AttitudeFactor() override {}
109
110 /**
111 * @brief Constructor
112 * @param key of the Rot3 variable that will be constrained
113 * @param nRef reference direction in navigation frame
114 * @param model Gaussian noise model
115 * @param bMeasured measured direction in body frame (default Z-axis)
116 */
117 Rot3AttitudeFactor(Key key, const Unit3& nRef, const SharedNoiseModel& model,
118 const Unit3& bMeasured = Unit3(0, 0, 1))
119 : Base(model, key), AttitudeFactor(nRef, bMeasured) {}
120
121 /// @return a deep copy of this factor
122 gtsam::NonlinearFactor::shared_ptr clone() const override {
123 return std::static_pointer_cast<gtsam::NonlinearFactor>(
124 r: gtsam::NonlinearFactor::shared_ptr(new This(*this)));
125 }
126
127 /** print */
128 void print(const std::string& s = "", const KeyFormatter& keyFormatter =
129 DefaultKeyFormatter) const override;
130
131 /** equals */
132 bool equals(const NonlinearFactor& expected,
133 double tol = 1e-9) const override;
134
135 /** vector of errors */
136 Vector evaluateError(const Rot3& nRb, OptionalMatrixType H) const override {
137 return attitudeError(p: nRb, H);
138 }
139
140 private:
141#if GTSAM_ENABLE_BOOST_SERIALIZATION
142 /** Serialization function */
143 friend class boost::serialization::access;
144 template <class ARCHIVE>
145 void serialize(ARCHIVE& ar, const unsigned int /*version*/) {
146 // NoiseModelFactor1 instead of NoiseModelFactorN for backward compatibility
147 ar& boost::serialization::make_nvp(
148 n: "NoiseModelFactor1", v&: boost::serialization::base_object<Base>(d&: *this));
149 ar& boost::serialization::make_nvp(
150 n: "AttitudeFactor",
151 v&: boost::serialization::base_object<AttitudeFactor>(d&: *this));
152 }
153#endif
154
155 public:
156 GTSAM_MAKE_ALIGNED_OPERATOR_NEW
157};

167class GTSAM_EXPORT Pose3AttitudeFactor : public NoiseModelFactorN<Pose3>,
168 public AttitudeFactor {
169 typedef NoiseModelFactorN<Pose3> Base;
170
171 public:
172 // Provide access to the Matrix& version of evaluateError:
173 using Base::evaluateError;
174
175 /// shorthand for a smart pointer to a factor
176 typedef std::shared_ptr<Pose3AttitudeFactor> shared_ptr;
177
178 /// Typedef to this class
179 typedef Pose3AttitudeFactor This;
180
181 /** default constructor - only use for serialization */
182 Pose3AttitudeFactor() {}
183
184 ~Pose3AttitudeFactor() override {}
185
186 /**
187 * @brief Constructor
188 * @param key of the Pose3 variable that will be constrained
189 * @param nRef reference direction in navigation frame
190 * @param model Gaussian noise model
191 * @param bMeasured measured direction in body frame (default Z-axis)
192 */
193 Pose3AttitudeFactor(Key key, const Unit3& nRef, const SharedNoiseModel& model,
194 const Unit3& bMeasured = Unit3(0, 0, 1))
195 : Base(model, key), AttitudeFactor(nRef, bMeasured) {}
196
197 /// @return a deep copy of this factor
198 gtsam::NonlinearFactor::shared_ptr clone() const override {
199 return std::static_pointer_cast<gtsam::NonlinearFactor>(
200 r: gtsam::NonlinearFactor::shared_ptr(new This(*this)));
201 }
202
203 /** print */
204 void print(const std::string& s = "", const KeyFormatter& keyFormatter =
205 DefaultKeyFormatter) const override;
206
207 /** equals */
208 bool equals(const NonlinearFactor& expected,
209 double tol = 1e-9) const override;
210
211 /** vector of errors */
212 Vector evaluateError(const Pose3& nTb, OptionalMatrixType H) const override {
213 Vector e = attitudeError(p: nTb.rotation(), H);
214 if (H) {
215 Matrix H23 = *H;
216 *H = Matrix::Zero(rows: 2, cols: 6);
217 H->block<2, 3>(startRow: 0, startCol: 0) = H23;
218 }
219 return e;
220 }
221
222 private:
223#if GTSAM_ENABLE_BOOST_SERIALIZATION
224 /** Serialization function */
225 friend class boost::serialization::access;
226 template <class ARCHIVE>
227 void serialize(ARCHIVE& ar, const unsigned int /*version*/) {
228 // NoiseModelFactor1 instead of NoiseModelFactorN for backward compatibility
229 ar& boost::serialization::make_nvp(
230 n: "NoiseModelFactor1", v&: boost::serialization::base_object<Base>(d&: *this));
231 ar& boost::serialization::make_nvp(
232 n: "AttitudeFactor",
233 v&: boost::serialization::base_object<AttitudeFactor>(d&: *this));
234 }
235#endif
236
237 public:
238 GTSAM_MAKE_ALIGNED_OPERATOR_NEW
239};
