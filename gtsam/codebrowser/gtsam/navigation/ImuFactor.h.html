61/**
62 * PreintegratedImuMeasurements accumulates (integrates) the IMU measurements
63 * (rotation rates and accelerations) and the corresponding covariance matrix.
64 * The measurements are then used to build the Preintegrated IMU factor.
65 * Integration is done incrementally (ideally, one integrates the measurement
66 * as soon as it is received from the IMU) so as to avoid costly integration
67 * at time of factor construction.
68 *
69 * @ingroup navigation
70 */
71template <class PreintegrationType>
72class GTSAM_EXPORT PreintegratedImuMeasurementsT: public PreintegrationType {
73
74 template <class PIM> friend class ImuFactorT;
75 template <class PIM> friend class ImuFactor2T;
76
77protected:
78
79 Matrix9 preintMeasCov_; ///< COVARIANCE OF: [PreintROTATION PreintPOSITION PreintVELOCITY]
80 ///< (first-order propagation from *measurementCovariance*).
81
82public:
83
84 /// Default constructor for serialization and wrappers
85 PreintegratedImuMeasurementsT() {
86 this->resetIntegration();
87 }
88
89 /**
90 * Constructor, initializes the class with no measurements
91 * @param p Parameters, typically fixed in a single application
92 * @param biasHat Current estimate of acceleration and rotation rate biases
93 */
94 PreintegratedImuMeasurementsT(const std::shared_ptr<PreintegrationParams>& p,
95 const imuBias::ConstantBias& biasHat = imuBias::ConstantBias()) :
96 PreintegrationType(p, biasHat) {
97 this->resetIntegration();
98 }
99
100/**
101 * Construct preintegrated directly from members: base class and preintMeasCov
102 * @param base PreintegrationType instance
103 * @param preintMeasCov Covariance matrix used in noise model.
104 */
105 PreintegratedImuMeasurementsT(const PreintegrationType& base, const Matrix9& preintMeasCov)
106 : PreintegrationType(base),
107 preintMeasCov_(preintMeasCov) {
108 this->PreintegrationType::resetIntegration();
109 }
110
111 /// Virtual destructor
112 ~PreintegratedImuMeasurementsT() override {
113 }
114
115 /// print
116 void print(const std::string& s = "Preintegrated Measurements:") const override;
117
118 /// equals
119 bool equals(const PreintegratedImuMeasurementsT<PreintegrationType>& expected, double tol = 1e-9) const;
120
121 /// Re-initialize PreintegratedImuMeasurements
122 void resetIntegration() override;
123
124 /**
125 * Add a single IMU measurement to the preintegration.
126 * Both accelerometer and gyroscope measurements are taken to be in the sensor
127 * frame and conversion to the body frame is handled by `body_P_sensor` in
128 * `PreintegrationParams`.
129 *
130 * @param measuredAcc Measured acceleration (as given by the sensor)
131 * @param measuredOmega Measured angular velocity (as given by the sensor)
132 * @param dt Time interval between this and the last IMU measurement
133 */
134 void integrateMeasurement(const Vector3& measuredAcc,
135 const Vector3& measuredOmega, const double dt) override;
136
137 /// Add multiple measurements, in matrix columns
138 void integrateMeasurements(const Matrix& measuredAccs, const Matrix& measuredOmegas,
139 const Matrix& dts);
140
141 /// Return pre-integrated measurement covariance
142 Matrix preintMeasCov() const { return preintMeasCov_; }
143
144 /// Merge in a different set of measurements and update bias derivatives accordingly
145 /// This method is specific to TangentPreintegration backend.
146 template <typename PB = PreintegrationType,
147 // This method is only callable when PreintegrationType is TangentPreintegration.
148 typename = typename std::enable_if<std::is_same<PB, TangentPreintegration>::value>::type>
149 void mergeWith(const PreintegratedImuMeasurementsT<TangentPreintegration>& pim12, Matrix9* H1, Matrix9* H2) {
150 // The `this->PreintegrationType::mergeWith` implies calling TangentPreintegration's mergeWith.
151 // Since pim12 is PreintegratedImuMeasurementsT<TangentPreintegration>, it is a TangentPreintegration.
152 this->PreintegrationType::mergeWith(pim12, H1, H2);
153 // NOTE(gareth): Temporary P is needed as of Eigen 3.3
154 const Matrix9 P = *H1 * preintMeasCov_ * H1->transpose();
155 preintMeasCov_ = P + *H2 * pim12.preintMeasCov_ * H2->transpose();
156 }
157
158 private:
159#if GTSAM_ENABLE_BOOST_SERIALIZATION ///
160 /// Serialization function
161 friend class boost::serialization::access;
162 template<class ARCHIVE>
163 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
164 namespace bs = ::boost::serialization;
165 ar & BOOST_SERIALIZATION_BASE_OBJECT_NVP(PreintegrationType);
166 ar & BOOST_SERIALIZATION_NVP(preintMeasCov_);
167 }
168#endif
169};

175/**
176 * ImuFactor is a 5-ways factor involving previous state (pose and velocity of
177 * the vehicle at previous time step), current state (pose and velocity at
178 * current time step), and the bias estimate. Following the preintegration
179 * scheme proposed in [2], the ImuFactor includes many IMU measurements, which
180 * are "summarized" using the PreintegratedImuMeasurements class.
181 * Note that this factor does not model "temporal consistency" of the biases
182 * (which are usually slowly varying quantities), which is up to the caller.
183 * See also CombinedImuFactor for a class that does this for you.
184 *
185 * @ingroup navigation
186 */
187template <class PIM = PreintegratedImuMeasurements>
188class GTSAM_EXPORT ImuFactorT: public NoiseModelFactorN<Pose3, Vector3, Pose3, Vector3,
189 imuBias::ConstantBias> {
190private:
191
192 typedef ImuFactorT<PIM> This;
193 typedef NoiseModelFactorN<Pose3, Vector3, Pose3, Vector3,
194 imuBias::ConstantBias> Base;
195
196 PIM pim_;
197
198public:
199
200 // Provide access to the Matrix& version of evaluateError:
201 using Base::evaluateError;
202
203 /** Shorthand for a smart pointer to a factor */
204 typedef std::shared_ptr<This> shared_ptr;
205
206
207 /** Default constructor - only use for serialization */
208 ImuFactorT() {}
209
210 /**
211 * Constructor
212 * @param pose_i Previous pose key
213 * @param vel_i Previous velocity key
214 * @param pose_j Current pose key
215 * @param vel_j Current velocity key
216 * @param bias Previous bias key
217 * @param preintegratedMeasurements The preintegreated measurements since the
218 * last pose.
219 */
220 ImuFactorT(Key pose_i, Key vel_i, Key pose_j, Key vel_j, Key bias,
221 const PIM& preintegratedMeasurements)
222 : Base(noiseModel::Gaussian::Covariance(covariance: preintegratedMeasurements.preintMeasCov()),
223 pose_i, vel_i, pose_j, vel_j, bias),
224 pim_(preintegratedMeasurements) {}
225
226 ~ImuFactorT() override {
227 }
228
229 /// @return a deep copy of this factor
230 gtsam::NonlinearFactor::shared_ptr clone() const override {
231 return std::make_shared<This>(*this);
232 }
233
234 /// @name Testable
235 /// @{
236 void print(const std::string& s = "", const KeyFormatter& keyFormatter =
237 DefaultKeyFormatter) const override;
238 bool equals(const NonlinearFactor& expected, double tol = 1e-9) const override;
239 /// @}
240
241 /** Access the preintegrated measurements. */
242
243 const PIM& preintegratedMeasurements() const {
244 return pim_;
245 }
246
247 /** implement functions needed to derive from Factor */
248
249 /// vector of errors
250 Vector evaluateError(const Pose3& pose_i, const Vector3& vel_i,
251 const Pose3& pose_j, const Vector3& vel_j,
252 const imuBias::ConstantBias& bias_i, OptionalMatrixType H1, OptionalMatrixType H2,
253 OptionalMatrixType H3, OptionalMatrixType H4, OptionalMatrixType H5) const override;
254
255 /// Merge two pre-integrated measurement classes
256 template <typename MethodPIMArg = PIM,
257 // This method is only callable when PIM is PreintegratedImuMeasurementsT<TangentPreintegration>.
258 typename = typename std::enable_if<
259 std::is_same<MethodPIMArg, PreintegratedImuMeasurementsT<TangentPreintegration>>::value
260 >::type
261 >
262 static MethodPIMArg Merge(
263 const MethodPIMArg& pim01,
264 const MethodPIMArg& pim12
265 ) {
266 // When this template is instantiated:
267 // 1. MethodPIMArg = PIM. It's mirrored to avoid error C7637 from strict compilers.
268 // 2. The SFINAE condition ensures MethodPIMArg IS PreintegratedImuMeasurementsT<TangentPreintegration>.
269 // So, arguments are const PreintegratedImuMeasurementsT<TangentPreintegration>&
270 // and return is PreintegratedImuMeasurementsT<TangentPreintegration>.
271
272 if (!pim01.matchesParamsWith(pim12))
273 throw std::domain_error(
274 "Cannot merge PreintegratedImuMeasurements with different params");
275
276 if (pim01.p_->body_P_sensor)
277 throw std::domain_error(
278 "Cannot merge PreintegratedImuMeasurements with sensor pose yet");
279
280 // the bias for the merged factor will be the bias from 01
281 MethodPIMArg pim02 = pim01;
282
283 Matrix9 H1, H2;
284 pim02.mergeWith(pim12, &H1, &H2);
285
286 return pim02;
287 }
288
289 /// Merge two factors
290 template <
291 typename MethodPIMArg = PIM,
292 // This method is only callable when PIM is PreintegratedImuMeasurementsT<TangentPreintegration>.
293 typename = typename std::enable_if<
294 std::is_same<MethodPIMArg, PreintegratedImuMeasurementsT<TangentPreintegration>>::value
295 >::type
296 >
297 static typename ImuFactorT<MethodPIMArg>::shared_ptr Merge(
298 const typename ImuFactorT<MethodPIMArg>::shared_ptr& f01,
299 const typename ImuFactorT<MethodPIMArg>::shared_ptr& f12
300 ) {
301 // When this template is instantiated:
302 // 1. MethodPIMArg = PIM. It's mirrored to avoid error C7637 from strict compilers.
303 // 2. The SFINAE condition ensures MethodPIMArg IS PreintegratedImuMeasurementsT<TangentPreintegration>.
304 // So, ImuFactorT<MethodPIMArg> is effectively ImuFactorT<PIM>, which is `This`.
305 // The signature effectively becomes:
306 // static typename This::shared_ptr Merge(const typename This::shared_ptr&, const typename This::shared_ptr&)
307
308 // IMU bias keys must be the same.
309 if (f01->template key<5>() != f12->template key<5>())
310 throw std::domain_error("ImuFactor::Merge: IMU bias keys must be the same");
311
312 // expect intermediate pose, velocity keys to matchup.
313 if (f01->template key<3>() != f12->template key<1>() || f01->template key<4>() != f12->template key<2>())
314 throw std::domain_error(
315 "ImuFactor::Merge: intermediate pose, velocity keys need to match up");
316
317 // return new factor
318 auto pim02 = This::Merge(f01->preintegratedMeasurements(), f12->preintegratedMeasurements());
319
320 return std::make_shared<This>( // `This` is ImuFactorT<MethodPIMArg> (i.e. ImuFactorT<PIM>)
321 f01->template key<1>(), // P0
322 f01->template key<2>(), // V0
323 f12->template key<3>(), // P2
324 f12->template key<4>(), // V2
325 f01->template key<5>(), // B
326 pim02);
327 }
328
329 private:
330 /** Serialization function */
331#if GTSAM_ENABLE_BOOST_SERIALIZATION
332 friend class boost::serialization::access;
333 template<class ARCHIVE>
334 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
335 // NoiseModelFactor5 instead of NoiseModelFactorN for backward compatibility
336 ar & boost::serialization::make_nvp("NoiseModelFactor5",
337 boost::serialization::base_object<Base>(*this));
338 ar & BOOST_SERIALIZATION_NVP(pim_);
339 }
340#endif
341};
