12/**
13 * @file ConstantVelocityFactor.h
14 * @brief Maintain a constant velocity motion model between two NavStates
15 * @author Asa Hammond
16 */
17
18#pragma once
19
20#include <gtsam/navigation/NavState.h>
21#include <gtsam/nonlinear/NonlinearFactor.h>
22
23namespace gtsam {
24
25/**
26 * Binary factor for applying a constant velocity model to a moving body represented as a NavState.
27 * The only measurement is dt, the time delta between the states.
28 */
29class ConstantVelocityFactor : public NoiseModelFactorN<NavState, NavState> {
30 double dt_;
31
32 public:
33 using Base = NoiseModelFactor2<NavState, NavState>;
34
35 // Provide access to the Matrix& version of evaluateError:
36 using Base::evaluateError;
37
38 public:
39 ConstantVelocityFactor(Key i, Key j, double dt, const SharedNoiseModel &model)
40 : NoiseModelFactorN<NavState, NavState>(model, i, j), dt_(dt) {}
41 ~ConstantVelocityFactor() override {}
42
43 /**
44 * @brief Calculate error: (x2 - x1.update(dt)))
45 * where X1 and X1 are NavStates and dt is
46 * the time difference in seconds between the states.
47 * @param x1 NavState for key a
48 * @param x2 NavState for key b
49 * @param H1 optional jacobian in x1
50 * @param H2 optional jacobian in x2
51 * @return * Vector
52 */
53 gtsam::Vector evaluateError(const NavState &x1, const NavState &x2,
54 OptionalMatrixType H1, OptionalMatrixType H2) const override {
55 // only used to use update() below
56 static const Vector3 b_accel{0.0, 0.0, 0.0};
57 static const Vector3 b_omega{0.0, 0.0, 0.0};
58
59 Matrix99 predicted_H_x1;
60 NavState predicted = x1.update(b_acceleration: b_accel, b_omega, dt: dt_, F: H1 ? &predicted_H_x1 : nullptr, G1: {}, G2: {});
61
62 Matrix99 error_H_predicted;
63 Vector9 error = predicted.localCoordinates(g: x2, H1: H1 ? &error_H_predicted : nullptr, H2);
64
65 if (H1) {
66 *H1 = error_H_predicted * predicted_H_x1;
67 }
68 return error;
69 }
70};
71
72} // namespace gtsam
