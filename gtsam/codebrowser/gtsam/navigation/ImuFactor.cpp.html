58//------------------------------------------------------------------------------
59template <class PreintegrationType>
60void PreintegratedImuMeasurementsT<PreintegrationType>::integrateMeasurement(
61 const Vector3& measuredAcc, const Vector3& measuredOmega, double dt) {
62 if (dt <= 0) {
63 throw std::runtime_error(
64 "PreintegratedImuMeasurements::integrateMeasurement: dt <=0");
65 }
66
67 // Update preintegrated measurements (also get Jacobian)
68 Matrix9 A; // overall Jacobian wrt preintegrated measurements (df/dx)
69 Matrix93 B, C; // Jacobian of state wrpt accel bias and omega bias respectively.
70 PreintegrationType::update(measuredAcc, measuredOmega, dt, &A, &B, &C);
71
72 // first order covariance propagation:
73 // as in [2] we consider a first order propagation that can be seen as a
74 // prediction phase in EKF
75
76 // propagate uncertainty
77 // TODO(frank): use noiseModel routine so we can have arbitrary noise models.
78 const Matrix3& aCov = this->p().accelerometerCovariance;
79 const Matrix3& wCov = this->p().gyroscopeCovariance;
80 const Matrix3& iCov = this->p().integrationCovariance;
81
82 // (1/dt) allows to pass from continuous time noise to discrete time noise
83 // Update the uncertainty on the state (matrix A in [4]).
84 preintMeasCov_ = A * preintMeasCov_ * A.transpose();
85 // These 2 updates account for uncertainty on the IMU measurement (matrix B in [4]).
86 preintMeasCov_.noalias() += B * (aCov / dt) * B.transpose();
87 preintMeasCov_.noalias() += C * (wCov / dt) * C.transpose();
88
89 // NOTE(frank): (Gi*dt)*(C/dt)*(Gi'*dt), with Gi << Z_3x3, I_3x3, Z_3x3 (9x3 matrix)
90 preintMeasCov_.block<3, 3>(startRow: 3, startCol: 3).noalias() += iCov * dt;
91}

93//------------------------------------------------------------------------------
94template <class PreintegrationType>
95void PreintegratedImuMeasurementsT<PreintegrationType>::integrateMeasurements(
96 const Matrix& measuredAccs, const Matrix& measuredOmegas,
97 const Matrix& dts) {
98 assert(
99 measuredAccs.rows() == 3 && measuredOmegas.rows() == 3 && dts.rows() == 1);
100 assert(dts.cols() >= 1);
101 assert(measuredAccs.cols() == dts.cols());
102 assert(measuredOmegas.cols() == dts.cols());
103 size_t n = static_cast<size_t>(dts.cols());
104 for (size_t j = 0; j < n; j++) {
105 integrateMeasurement(measuredAcc: measuredAccs.col(i: j), measuredOmega: measuredOmegas.col(i: j), dt: dts(0, j));
106 }
107}

142 const imuBias::ConstantBias& bias_i, OptionalMatrixType H1,
