20#pragma once
21
22#include <gtsam/navigation/LeftLinearEKF.h> // Include the base class
23#include <gtsam/navigation/NavState.h>
24#include <gtsam/navigation/PreintegrationParams.h>
25
26namespace gtsam {
27
28/// Specialized EKF for IMU-driven NavState on SE_2(3)
29class GTSAM_EXPORT NavStateImuEKF : public LeftLinearEKF<NavState> {
30 public:
31 using Base = LeftLinearEKF<NavState>;
32 using TangentVector = typename Base::TangentVector; // Vector9
33 using Jacobian = typename Base::Jacobian; // 9x9
34 using Covariance = typename Base::Covariance; // 9x9
35
36 /**
37 * Construct with initial state/covariance and preintegration params (for
38 * gravity and IMU covariances)
39 * @param X0 Initial NavState.
40 * @param P0 Initial covariance in tangent space at X0.
41 * @param params Preintegration parameters providing gravity and options.
42 */
43 NavStateImuEKF(const NavState& X0, const Covariance& P0,
44 const std::shared_ptr<PreintegrationParams>& params);
45
46 /// Calculate W (gravity-only left composition, world-frame increments)
47 static NavState Gravity(const Vector3& g_n, double dt) {
48 return {Rot3(), g_n * (0.5 * dt * dt), g_n * dt};
49 }
50
51 /// Calculate U from raw IMU (no gravity): body-frame increments
52 /// We do an explicit closed-form integration based on SO(3) kernels
53 static NavState Imu(const Vector3& omega_b, const Vector3& f_b, double dt);
54
55 /**
56 * @brief Compute the dynamics of the system.
57 *
58 * This function computes the next state of the system based on the current
59 * state, gravity, body angular velocity, body specific force, and time step.
60 * The dynamics are defined as:
61 * X_{k+1} = f(X_k; g, omega_b, f_b, dt)
62 * = W(g, dt) \phi_dt(X_k) U(omega_b, f_b, dt)
63 * where W, \phi, and U are the gravity, (autonomous) position update, and
64 * IMU increment functions, respectively.
65 *
66 * @param g_n Gravity vector in the navigation frame.
67 * @param X Current NavState.
68 * @param omega_b Body angular velocity measurement (rad/s).
69 * @param f_b Body specific force measurement (m/s^2).
70 * @param dt Time step in seconds.
71 * @param A Optional Jacobian of the dynamics with respect to the state.
72 * @return The next NavState after applying the dynamics.
73 */
74 static NavState Dynamics(const Vector3& g_n, const NavState& X,
75 const Vector3& omega_b, const Vector3& f_b,
76 double dt, OptionalJacobian<9, 9> A = {});
77
78 /**
79 * @brief Predict the next state using gyro and accelerometer measurements.
80 *
81 * This method updates the state of the system based on the provided body
82 * angular velocity (omega_b) and body specific force (f_b)
83 * measurements over a given time step.
84 *
85 * @param omega_b Body angular velocity measurement (rad/s).
86 * @param f_b Body specific force measurement (m/s^2).
87 * @param dt Time step in seconds.
88 */
89 void predict(const Vector3& omega_b, const Vector3& f_b, double dt);
90
91 /// Accessors
92 const std::shared_ptr<PreintegrationParams>& params() const;
93 const Vector3& gravity() const;
94 const Covariance& processNoise() const;
95
96 private:
97 std::shared_ptr<PreintegrationParams> params_;
98 Covariance Q_ = Covariance::Zero();
99};
100
101} // namespace gtsam
