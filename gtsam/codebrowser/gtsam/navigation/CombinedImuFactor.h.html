56/**
57 * PreintegratedCombinedMeasurements integrates the IMU measurements
58 * (rotation rates and accelerations) and the corresponding covariance matrix.
59 * The measurements are then used to build the CombinedImuFactor. Integration
60 * is done incrementally (ideally, one integrates the measurement as soon as
61 * it is received from the IMU) so as to avoid costly integration at time of
62 * factor construction.
63 *
64 * @ingroup navigation
65 */
66template <class PreintegrationType>
67class GTSAM_EXPORT PreintegratedCombinedMeasurementsT : public PreintegrationType {
68 public:
69 typedef PreintegrationCombinedParams Params;
70
71 protected:
72 /* Covariance matrix of the preintegrated measurements
73 * COVARIANCE OF: [PreintROTATION PreintPOSITION PreintVELOCITY BiasAcc
74 * BiasOmega] (first-order propagation from *measurementCovariance*).
75 * PreintegratedCombinedMeasurements also include the biases and keep the
76 * correlation between the preintegrated measurements and the biases
77 */
78 Eigen::Matrix<double, 15, 15> preintMeasCov_;
79
80 template <class PIM> friend class CombinedImuFactorT;
81
82 public:
83 /// @name Constructors
84 /// @{
85
86 /// Default constructor only for serialization and wrappers
87 PreintegratedCombinedMeasurementsT() { this->resetIntegration(); }
88
89 /**
90 * Default constructor, initializes the class with no measurements
91 * @param p Parameters, typically fixed in a single application
92 * @param biasHat Current estimate of acceleration and rotation rate biases
93 * @param preintMeasCov Covariance matrix used in noise model.
94 */
95 PreintegratedCombinedMeasurementsT(
96 const std::shared_ptr<Params>& p,
97 const imuBias::ConstantBias& biasHat = imuBias::ConstantBias(),
98 const Eigen::Matrix<double, 15, 15>& preintMeasCov =
99 Eigen::Matrix<double, 15, 15>::Zero())
100 : PreintegrationType(p, biasHat), preintMeasCov_(preintMeasCov) {
101 this->PreintegrationType::resetIntegration();
102 }
103
104 /**
105 * Construct preintegrated directly from members: base class and
106 * preintMeasCov
107 * @param base PreintegrationType instance
108 * @param preintMeasCov Covariance matrix used in noise model.
109 */
110 PreintegratedCombinedMeasurementsT(
111 const PreintegrationType& base,
112 const Eigen::Matrix<double, 15, 15>& preintMeasCov)
113 : PreintegrationType(base), preintMeasCov_(preintMeasCov) {
114 this->PreintegrationType::resetIntegration();
115 }
116
117 /// Virtual destructor
118 ~PreintegratedCombinedMeasurementsT() override {}
119
120 /// @}
121
122 /// @name Basic utilities
123 /// @{
124
125 /// Re-initialize PreintegratedCombinedMeasurements
126 void resetIntegration() override;
127
128 /// const reference to params, shadows definition in base class
129 Params& p() const { return *std::static_pointer_cast<Params>(this->p_); }
130 /// @}
131
132 /// @name Access instance variables
133 /// @{
134 /// Return pre-integrated measurement covariance
135 Matrix preintMeasCov() const { return preintMeasCov_; }
136 /// @}
137
138 /// @name Testable
139 /// @{
140 /// print
141 void print(
142 const std::string& s = "Preintegrated Measurements:") const override;
143 /// equals
144 bool equals(const PreintegratedCombinedMeasurementsT<PreintegrationType>& expected,
145 double tol = 1e-9) const;
146 /// @}
147
148 /// @name Main functionality
149 /// @{
150
151 /**
152 * Add a single IMU measurement to the preintegration.
153 * Both accelerometer and gyroscope measurements are taken to be in the sensor
154 * frame and conversion to the body frame is handled by `body_P_sensor` in
155 * `PreintegrationParams`.
156 *
157 * @param measuredAcc Measured acceleration (as given by the sensor)
158 * @param measuredOmega Measured angular velocity (as given by the sensor)
159 * @param dt Time interval between two consecutive IMU measurements
160 */
161 void integrateMeasurement(const Vector3& measuredAcc,
162 const Vector3& measuredOmega,
163 const double dt) override;
164
165 /// @}
166
167#ifdef GTSAM_ALLOW_DEPRECATED_SINCE_V43
168/// @deprecated: biasAccOmegaInt is no longer used. Use a prior on first bias instead.
169 void resetIntegration(const gtsam::Matrix6& Q_init) {
170 std::cerr << "Warning: setBiasAccOmegaInit() is deprecated and no longer used." << std::endl;
171 PreintegrationType::resetIntegration();
172 preintMeasCov_.setZero();
173 }
174#endif
175
176 private:
177#if GTSAM_ENABLE_BOOST_SERIALIZATION ///
178 /// Serialization function
179 friend class boost::serialization::access;
180 template <class ARCHIVE>
181 void serialize(ARCHIVE& ar, const unsigned int /*version*/) {
182 namespace bs = ::boost::serialization;
183 ar& BOOST_SERIALIZATION_BASE_OBJECT_NVP(PreintegrationType);
184 ar& BOOST_SERIALIZATION_NVP(preintMeasCov_);
185 }
186#endif
187
188 public:
189 GTSAM_MAKE_ALIGNED_OPERATOR_NEW
190};

195/**
196 * CombinedImuFactor is a 6-ways factor involving previous state (pose and
197 * velocity of the vehicle, as well as bias at previous time step), and current
198 * state (pose, velocity, bias at current time step). Following the pre-
199 * integration scheme proposed in [2], the CombinedImuFactor includes many IMU
200 * measurements, which are "summarized" using the
201 * PreintegratedCombinedMeasurements class. There are 3 main differences wrpt
202 * the ImuFactor class: 1) The factor is 6-ways, meaning that it also involves
203 * both biases (previous and current time step).Therefore, the factor internally
204 * imposes the biases to be slowly varying; in particular, the matrices
205 * "biasAccCovariance" and "biasOmegaCovariance" described the random walk that
206 * models bias evolution. 2) The preintegration covariance takes into account
207 * the noise in the bias estimate used for integration. 3) The covariance matrix
208 * of the PreintegratedCombinedMeasurements preserves the correlation between
209 * the bias uncertainty and the preintegrated measurements uncertainty.
210 *
211 * @ingroup navigation
212 */
213template <class PIM = PreintegratedCombinedMeasurements>
214class GTSAM_EXPORT CombinedImuFactorT
215 : public NoiseModelFactorN<Pose3, Vector3, Pose3, Vector3,
216 imuBias::ConstantBias, imuBias::ConstantBias> {
217 public:
218 private:
219 typedef CombinedImuFactorT<PIM> This;
220 typedef NoiseModelFactorN<Pose3, Vector3, Pose3, Vector3,
221 imuBias::ConstantBias, imuBias::ConstantBias>
222 Base;
223
224 PIM pim_;
225
226 public:
227 // Provide access to Matrix& version of evaluateError:
228 using Base::evaluateError;
229
230 /** Shorthand for a smart pointer to a factor */
231 typedef std::shared_ptr<This> shared_ptr;
232
233 /** Default constructor - only use for serialization */
234 CombinedImuFactorT() {}
235
236 /**
237 * Constructor
238 * @param pose_i Previous pose key
239 * @param vel_i Previous velocity key
240 * @param pose_j Current pose key
241 * @param vel_j Current velocity key
242 * @param bias_i Previous bias key
243 * @param bias_j Current bias key
244 * @param PreintegratedCombinedMeasurements Combined IMU measurements
245 */
246 CombinedImuFactorT(
247 Key pose_i, Key vel_i, Key pose_j, Key vel_j, Key bias_i, Key bias_j,
248 const PIM& preintegratedMeasurements)
249 : Base(noiseModel::Gaussian::Covariance(covariance: preintegratedMeasurements.preintMeasCov()),
250 pose_i, vel_i, pose_j, vel_j, bias_i, bias_j),
251 pim_(preintegratedMeasurements) {}
252
253 ~CombinedImuFactorT() override {}
254
255 /// @return a deep copy of this factor
256 gtsam::NonlinearFactor::shared_ptr clone() const override {
257 return std::make_shared<This>(*this);
258 }
259
260 /** implement functions needed for Testable */
261
262 /// @name Testable
263 /// @{
264 /// print
265 void print(const std::string& s = "", const KeyFormatter& keyFormatter =
266 DefaultKeyFormatter) const override;
267
268 /// equals
269 bool equals(const NonlinearFactor& expected,
270 double tol = 1e-9) const override;
271 /// @}
272
273 /** Access the preintegrated measurements. */
274
275 const PIM& preintegratedMeasurements() const {
276 return pim_;
277 }
278
279 /** implement functions needed to derive from Factor */
280
281 /// vector of errors
282 Vector evaluateError(const Pose3& pose_i, const Vector3& vel_i,
283 const Pose3& pose_j, const Vector3& vel_j,
284 const imuBias::ConstantBias& bias_i,
285 const imuBias::ConstantBias& bias_j,
286 OptionalMatrixType H1, OptionalMatrixType H2,
287 OptionalMatrixType H3, OptionalMatrixType H4,
288 OptionalMatrixType H5,
289 OptionalMatrixType H6) const override;
290
291 private:
292#if GTSAM_ENABLE_BOOST_SERIALIZATION
293 /** Serialization function */
294 friend class boost::serialization::access;
295 template <class ARCHIVE>
296 void serialize(ARCHIVE& ar, const unsigned int /*version*/) {
297 // NoiseModelFactor6 instead of NoiseModelFactorN for backward compatibility
298 ar& boost::serialization::make_nvp(
299 "NoiseModelFactor6", boost::serialization::base_object<Base>(*this));
300 ar& BOOST_SERIALIZATION_NVP(pim_);
301 }
302#endif
303
304 public:
305 GTSAM_MAKE_ALIGNED_OPERATOR_NEW
306};
