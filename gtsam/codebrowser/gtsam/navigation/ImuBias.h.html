27namespace gtsam {
28
29/// All bias models live in the imuBias namespace
30namespace imuBias {
31
32class GTSAM_EXPORT ConstantBias {
33private:
34 Vector3 biasAcc_; ///< The units for stddev are σ = m/s² or m √Hz/s²
35 Vector3 biasGyro_; ///< The units for stddev are σ = rad/s or rad √Hz/s
36
37public:
38 /// dimension of the variable - used to autodetect sizes
39 static const size_t dimension = 6;
40
41 /// @name Standard Constructors
42 /// @{
43
44 ConstantBias() :
45 biasAcc_(0.0, 0.0, 0.0), biasGyro_(0.0, 0.0, 0.0) {
46 }
47
48 ConstantBias(const Vector3& biasAcc, const Vector3& biasGyro) :
49 biasAcc_(biasAcc), biasGyro_(biasGyro) {
50 }
51
52 explicit ConstantBias(const Vector6& v) :
53 biasAcc_(v.head<3>()), biasGyro_(v.tail<3>()) {
54 }
55
56 /// @}
57
58 /** return the accelerometer and gyro biases in a single vector */
59 Vector6 vector() const {
60 Vector6 v;
61 v << biasAcc_, biasGyro_;
62 return v;
63 }
64
65 /** get accelerometer bias */
66 const Vector3& accelerometer() const {
67 return biasAcc_;
68 }
69
70 /** get gyroscope bias */
71 const Vector3& gyroscope() const {
72 return biasGyro_;
73 }
74
75 /** Correct an accelerometer measurement using this bias model, and optionally compute Jacobians */
76 Vector3 correctAccelerometer(const Vector3& measurement,
77 OptionalJacobian<3, 6> H1 = {},
78 OptionalJacobian<3, 3> H2 = {}) const {
79 if (H1) (*H1) << -I_3x3, Z_3x3;
80 if (H2) (*H2) << I_3x3;
81 return measurement - biasAcc_;
82 }
83
84 /** Correct a gyroscope measurement using this bias model, and optionally compute Jacobians */
85 Vector3 correctGyroscope(const Vector3& measurement,
86 OptionalJacobian<3, 6> H1 = {},
87 OptionalJacobian<3, 3> H2 = {}) const {
88 if (H1) (*H1) << Z_3x3, -I_3x3;
89 if (H2) (*H2) << I_3x3;
90 return measurement - biasGyro_;
91 }
92
93 /// @name Testable
94 /// @{
95
96 /// ostream operator
97 GTSAM_EXPORT friend std::ostream& operator<<(std::ostream& os,
98 const ConstantBias& bias);
99
100 /// print with optional string
101 void print(const std::string& s = "") const;
102
103 /** equality up to tolerance */
104 inline bool equals(const ConstantBias& expected, double tol = 1e-5) const {
105 return equal_with_abs_tol(A: biasAcc_, B: expected.biasAcc_, tol)
106 && equal_with_abs_tol(A: biasGyro_, B: expected.biasGyro_, tol);
107 }
108
109 /// @}
110 /// @name Group
111 /// @{
112
113 /** identity for group operation */
114 static ConstantBias Identity() {
115 return ConstantBias();
116 }
117
118 /** inverse */
119 inline ConstantBias operator-() const {
120 return ConstantBias(-biasAcc_, -biasGyro_);
121 }
122
123 /** addition of vector on right */
124 ConstantBias operator+(const Vector6& v) const {
125 return ConstantBias(biasAcc_ + v.head<3>(), biasGyro_ + v.tail<3>());
126 }
127
128 /** addition */
129 ConstantBias operator+(const ConstantBias& b) const {
130 return ConstantBias(biasAcc_ + b.biasAcc_, biasGyro_ + b.biasGyro_);
131 }
132
133 /** subtraction */
134 ConstantBias operator-(const ConstantBias& b) const {
135 return ConstantBias(biasAcc_ - b.biasAcc_, biasGyro_ - b.biasGyro_);
136 }
137
138 /// @}
139 /// @name Manifold
140 /// @{
141
142 /// The retract function
143 ConstantBias retract(const Vector6& v) const {
144 return ConstantBias(biasAcc_ + v.head<3>(), biasGyro_ + v.tail<3>());
145 }
146
147 /// The local coordinates function
148 Vector6 localCoordinates(const ConstantBias& other) const {
149 return other.vector() - vector();
150 }
151
152 /// @}
153
154private:
155
156 /// @name Advanced Interface
157 /// @{
158
159#if GTSAM_ENABLE_BOOST_SERIALIZATION
160 /** Serialization function */
161 friend class boost::serialization::access;
162 template<class ARCHIVE>
163 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
164 ar & BOOST_SERIALIZATION_NVP(biasAcc_);
165 ar & BOOST_SERIALIZATION_NVP(biasGyro_);
166 }
167#endif
168
169
170public:
171 GTSAM_MAKE_ALIGNED_OPERATOR_NEW
172 /// @}
173
174}; // ConstantBias class
175} // namespace imuBias
176
177template<>
178struct traits<imuBias::ConstantBias> : public internal::VectorSpace<
179 imuBias::ConstantBias> {
180};

182} // namespace gtsam
