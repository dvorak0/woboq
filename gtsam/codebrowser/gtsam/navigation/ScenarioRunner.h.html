27// Convert covariance to diagonal noise model, if possible, otherwise throw
28static noiseModel::Diagonal::shared_ptr Diagonal(const Matrix& covariance) {
29 bool smart = true;
30 auto model = noiseModel::Gaussian::Covariance(covariance, smart);
31 auto diagonal = std::dynamic_pointer_cast<noiseModel::Diagonal>(r: model);
32 if (!diagonal)
33 throw std::invalid_argument("ScenarioRunner::Diagonal: not a diagonal");
34 return diagonal;
35}

37/*
38 * Simple class to test navigation scenarios.
39 * Takes a trajectory scenario as input, and can generate IMU measurements
40 */
41class GTSAM_EXPORT ScenarioRunner {
42 public:
43 typedef imuBias::ConstantBias Bias;
44 typedef std::shared_ptr<PreintegrationParams> SharedParams;
45
46 protected:
47 const Scenario& scenario_;
48 const SharedParams p_;
49 const double imuSampleTime_, sqrt_dt_;
50 const Bias estimatedBias_;
51
52 // Create two samplers for acceleration and omega noise
53 Sampler gyroSampler_, accSampler_;
54
55 public:
56 ScenarioRunner(const Scenario& scenario, const SharedParams& p,
57 double imuSampleTime = 1.0 / 100.0, const Bias& bias = Bias())
58 : scenario_(scenario),
59 p_(p),
60 imuSampleTime_(imuSampleTime),
61 sqrt_dt_(std::sqrt(x: imuSampleTime)),
62 estimatedBias_(bias),
63 // NOTE(duy): random seeds that work well:
64 gyroSampler_(Diagonal(covariance: p->gyroscopeCovariance), 10),
65 accSampler_(Diagonal(covariance: p->accelerometerCovariance), 29284) {}
66
67 // NOTE(frank): hardcoded for now with Z up (gravity points in negative Z)
68 // also, uses g=10 for easy debugging
69 const Vector3& gravity_n() const { return p_->n_gravity; }
70
71 const Scenario& scenario() const { return scenario_; }
72
73 // A gyro simply measures angular velocity in body frame
74 Vector3 actualAngularVelocity(double t) const { return scenario_.omega_b(t); }
75
76 // An accelerometer measures acceleration in body, but not gravity
77 Vector3 actualSpecificForce(double t) const {
78 Rot3 bRn(scenario_.rotation(t).transpose());
79 return scenario_.acceleration_b(t) - bRn * gravity_n();
80 }
81
82 // Angular velocity measured by gyroscope, corrupted by bias and noise
83 Vector3 measuredAngularVelocity(double t) const {
84 return actualAngularVelocity(t) + estimatedBias_.gyroscope() +
85 gyroSampler_.sample() / sqrt_dt_;
86 }
87
88 /// Specific force measured by accelerometer, corrupted by bias and noise
89 Vector3 measuredSpecificForce(double t) const {
90 return actualSpecificForce(t) + estimatedBias_.accelerometer() +
91 accSampler_.sample() / sqrt_dt_;
92 }
93
94 /// The IMU sample time (i.e. the time between two IMU measurements)
95 double imuSampleTime() const { return imuSampleTime_; }
96
97 /// Integrate measurements for T seconds into a PIM
98 PreintegratedImuMeasurements integrate(double T,
99 const Bias& estimatedBias = Bias(),
100 bool corrupted = false) const;
101
102 /// Predict predict given a PIM
103 NavState predict(const PreintegratedImuMeasurements& pim,
104 const Bias& estimatedBias = Bias()) const;
105
106 /// Compute a Monte Carlo estimate of the predict covariance using N samples
107 Matrix9 estimateCovariance(double T, size_t N = 1000,
108 const Bias& estimatedBias = Bias()) const;
109
110 /// Estimate covariance of sampled noise for sanity-check
111 Matrix6 estimateNoiseCovariance(size_t N = 1000) const;
112};

114/*
115 * Simple class to test navigation scenarios with CombinedImuMeasurements.
116 * Takes a trajectory scenario as input, and can generate IMU measurements
117 */
118class GTSAM_EXPORT CombinedScenarioRunner : public ScenarioRunner {
119 public:
120 typedef std::shared_ptr<PreintegrationCombinedParams> SharedParams;
121
122 private:
123 const SharedParams p_;
124 const Eigen::Matrix<double, 15, 15> preintMeasCov_;
125
126 public:
127 CombinedScenarioRunner(const Scenario& scenario, const SharedParams& p,
128 double imuSampleTime = 1.0 / 100.0,
129 const Bias& bias = Bias(),
130 const Eigen::Matrix<double, 15, 15>& preintMeasCov =
131 Eigen::Matrix<double, 15, 15>::Zero())
132 : ScenarioRunner(scenario, static_cast<ScenarioRunner::SharedParams>(p),
133 imuSampleTime, bias),
134 p_(p),
135 preintMeasCov_(preintMeasCov) {}
136
137 /// Integrate measurements for T seconds into a PIM
138 PreintegratedCombinedMeasurements integrate(
139 double T, const Bias& estimatedBias = Bias(),
140 bool corrupted = false) const;
141
142 /// Predict predict given a PIM
143 NavState predict(const PreintegratedCombinedMeasurements& pim,
144 const Bias& estimatedBias = Bias()) const;
145
146 /// Compute a Monte Carlo estimate of the predict covariance using N samples
147 Eigen::Matrix<double, 15, 15> estimateCovariance(
148 double T, size_t N = 1000, const Bias& estimatedBias = Bias()) const;
149};
