32Rot3 Scenario::rotation(double t) const {

42Vector3 Scenario::velocity_b(double t) const {
43 const Rot3 nRb = rotation(t);
44 return nRb.transpose() * velocity_n(t);
45}
46
47Vector3 Scenario::acceleration_b(double t) const {
48 const Rot3 nRb = rotation(t);
49 return nRb.transpose() * acceleration_n(t);
50}

52Pose3 ConstantTwistScenario::pose(double t) const {
53 return nTb0_ * Pose3::Expmap(xi: twist_ * t);
54}
55
56Vector3 ConstantTwistScenario::omega_b(double t) const {
57 return twist_.head<3>();
58}
59
60Vector3 ConstantTwistScenario::velocity_n(double t) const {
61 return rotation(t).matrix() * twist_.tail<3>();
62}
63
64Vector3 ConstantTwistScenario::acceleration_n(double t) const {
65 return rotation(t) * a_b_;
66}

68Pose3 AcceleratingScenario::pose(double t) const {
69 return Pose3(nRb_.expmap(v: omega_b_ * t), p0_ + v0_ * t + a_n_ * t * t / 2.0);
70}

104DiscreteScenario DiscreteScenario::FromCSV(const string& csv_filepath) {
105 ifstream file(csv_filepath);
106 if (!file.is_open()) {
107 throw runtime_error("DiscreteScenario::FromCSV: Could not open file " +
108 csv_filepath);
109 }
110
111 // Header parsing
112 string header_line;
113 if (!getline(is&: file, str&: header_line)) {
114 throw runtime_error(
115 "DiscreteScenario::FromCSV: CSV file is empty or contains no header.");
116 }
117
118 map<string, int> columnIndex;
119 stringstream header_ss(header_line);
120 string column_name;
121 int index = 0;
122 while (getline(in&: header_ss, str&: column_name, delim: ',')) {
123 // Basic trim for whitespace, in case of " col1, col2 "
124 column_name.erase(pos: 0, n: column_name.find_first_not_of(s: " \t\r\n"));
125 column_name.erase(pos: column_name.find_last_not_of(s: " \t\r\n") + 1);
126 columnIndex[column_name] = index++;
127 }
128
129 // Define and validate required columns
130 // These are the header names we expect to find in the CSV. Based on EuRoC MAV
131 // dataset format.
132 const vector<string> required_columns = {
133 "t", "p_x", "p_y", "p_z", "q_w", "q_x", "q_y", "q_z",
134 "v_x", "v_y", "v_z", "w_x", "w_y", "w_z", "a_x", "a_y",
135 "a_z"};
136
137 for (const auto& col : required_columns) {
138 if (columnIndex.find(x: col) == columnIndex.end()) {
139 throw runtime_error(
140 "DiscreteScenario::FromCSV: Missing required column header '" + col +
141 "' in file " + csv_filepath);
142 }
143 }
144
145 // Temporary storage for data
146 struct DataPoint {
147 double t;
148 Pose3 pose;
149 Vector3 omega_b, velocity_n, acceleration_n;
150 };
151 vector<DataPoint> data_points;
152
153 // Read data rows
154 string data_line;
155 int line_number = 1; // Header was line 1
156 while (getline(is&: file, str&: data_line)) {
157 line_number++;
158 stringstream data_ss(data_line);
159 vector<string> values;
160 string value;
161 while (getline(in&: data_ss, str&: value, delim: ',')) {
162 values.push_back(x: value);
163 }
164
165 if (values.size() != columnIndex.size()) {
166 throw runtime_error(
167 "DiscreteScenario::FromCSV: Malformed data at line " +
168 to_string(val: line_number) + ". Expected " + to_string(val: columnIndex.size()) +
169 " columns, but found " + to_string(val: values.size()) + ".");
170 }
171
172 try {
173 DataPoint dp;
174 // Use the columnIndex map to get data by name, not by fixed position
175 dp.t = stod(str: values[columnIndex.at(k: "t")]);
176
177 double px = stod(str: values[columnIndex.at(k: "p_x")]);
178 double py = stod(str: values[columnIndex.at(k: "p_y")]);
179 double pz = stod(str: values[columnIndex.at(k: "p_z")]);
180
181 double qw = stod(str: values[columnIndex.at(k: "q_w")]);
182 double qx = stod(str: values[columnIndex.at(k: "q_x")]);
183 double qy = stod(str: values[columnIndex.at(k: "q_y")]);
184 double qz = stod(str: values[columnIndex.at(k: "q_z")]);
185
186 dp.velocity_n << stod(str: values[columnIndex.at(k: "v_x")]),
187 stod(str: values[columnIndex.at(k: "v_y")]),
188 stod(str: values[columnIndex.at(k: "v_z")]);
189
190 dp.omega_b << stod(str: values[columnIndex.at(k: "w_x")]),
191 stod(str: values[columnIndex.at(k: "w_y")]),
192 stod(str: values[columnIndex.at(k: "w_z")]);
193
194 dp.acceleration_n << stod(str: values[columnIndex.at(k: "a_x")]),
195 stod(str: values[columnIndex.at(k: "a_y")]),
196 stod(str: values[columnIndex.at(k: "a_z")]);
197
198 dp.pose = Pose3(Rot3::Quaternion(w: qw, x: qx, y: qy, z: qz), Point3(px, py, pz));
199 data_points.push_back(x: dp);
200
201 } catch (const std::invalid_argument& e) {
202 throw runtime_error(
203 "DiscreteScenario::FromCSV: Non-numeric data at line " +
204 to_string(val: line_number) + ". " + e.what());
205 } catch (const std::out_of_range& e) {
206 // This can be triggered by stod or by map::at if something is wrong
207 throw runtime_error(
208 "DiscreteScenario::FromCSV: Data conversion error at line " +
209 to_string(val: line_number) + ". " + e.what());
210 }
211 }
212
213 if (data_points.empty()) {
214 throw runtime_error(
215 "DiscreteScenario::FromCSV: No data points loaded from file.");
216 }
217
218 // Normalize timestamps and populate maps
219 const double t0 = data_points.front().t;
220 map<double, Pose3> poses;
221 map<double, Vector3> angularVelocities_b, velocities_n, accelerations_n;
222
223 for (const auto& dp : data_points) {
224 const double normalized_t = dp.t - t0;
225 poses[normalized_t] = dp.pose;
226 angularVelocities_b[normalized_t] = dp.omega_b;
227 velocities_n[normalized_t] = dp.velocity_n;
228 accelerations_n[normalized_t] = dp.acceleration_n;
229 }
230
231 return DiscreteScenario(poses, angularVelocities_b, velocities_n,
232 accelerations_n);
233}
