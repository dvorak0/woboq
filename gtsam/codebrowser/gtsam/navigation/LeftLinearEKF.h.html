31namespace gtsam {
32
33/**
34 * @class LeftLinearEKF
35 * @brief EKF on a Lie group with a general left–linear prediction model.
36 *
37 * Discrete step: x⁺ = W · φ(x) · U, with W,U ∈ G and φ ∈ Aut(G).
38 * For left-invariant error, the state-independent linearization is
39 * A = Ad_{U^{-1}} · Φ where Φ := dφ|_e (right-trivialized). The left factor
40 * W cancels in A and does not appear there.
41 */
42template <typename G>
43class LeftLinearEKF : public LieGroupEKF<G> {
44 public:
45 using Base = LieGroupEKF<G>;
46 static constexpr int Dim = Base::Dim; ///< Compile-time dimension of G.
47 using TangentVector = typename Base::TangentVector;
48 using Jacobian = typename Base::Jacobian;
49 using Covariance = typename Base::Covariance;
50
51 LeftLinearEKF(const G& X0, const Covariance& P0) : Base(X0, P0) {}
52
53 /**
54 * @brief SFINAE template to check if a type satisfies the automorphism
55 * concept. Specifically, it checks for the existence of:
56 * - G operator()(const G&) const
57 * - Jacobian dIdentity() const
58 */
59 template <typename Phi>
60 struct is_automorphism
61 : std::conjunction<
62 std::is_invocable_r<G, Phi, const G&>,
63 std::is_same<decltype(std::declval<Phi>().dIdentity()), Jacobian>> {
64 };
65
66 /**
67 * General left–linear dynamics.
68 * Returns W · φ(X) · U, and optional Jacobian A = Ad_{U^{-1}} Φ, Φ := dφ|_e.
69 */
70 template <class Phi, typename = std::enable_if_t<is_automorphism<Phi>::value>>
71 static G Dynamics(const G& W, const Phi& phi, const G& X, const G& U,
72 OptionalJacobian<Dim, Dim> A = {}) {
73 return traits<G>::Compose(W, Dynamics<Phi>(phi, X, U, A));
74 }
75
76 /**
77 * Left–linear dynamics with W=I.
78 * Returns φ(X) · U, and optional Jacobian A = Ad_{U^{-1}} Φ, Φ := dφ|_e.
79 */
80 template <class Phi, typename = std::enable_if_t<is_automorphism<Phi>::value>>
81 static G Dynamics(const Phi& phi, const G& X, const G& U,
82 OptionalJacobian<Dim, Dim> A = {}) {
83 if (A) {
84 const G U_inv = traits<G>::Inverse(U);
85 *A = traits<G>::AdjointMap(U_inv) * phi.dIdentity();
86 }
87 return traits<G>::Compose(phi(X), U);
88 }
89
90 /**
91 * General left–linear prediction, updates filter state as follows:
92 * X⁺ = W · φ(X) · U
93 * P⁺ = A P Aᵀ + Q with A = Ad_{U^{-1}} Φ, Φ := dφ|_e.
94 */
95 template <class Phi, typename = std::enable_if_t<is_automorphism<Phi>::value>>
96 void predict(const G& W, const Phi& phi, const G& U, const Covariance& Q) {
97 Jacobian A;
98 this->X_ = this->Dynamics(W, phi, this->X_, U, A);
99 this->P_ = A * this->P_ * A.transpose() + Q;
100 }
101
102 /**
103 * Special case of predict with W=I, updates filter state as follows:
104 * Update: X⁺ = φ(X) · U
105 * Covariance: P⁺ = A P Aᵀ + Q with A = Ad_{U^{-1}} Φ, Φ := dφ|_e.
106 */
107 template <class Phi, typename = std::enable_if_t<is_automorphism<Phi>::value>>
108 void predict(const Phi& phi, const G& U, const Covariance& Q) {
109 Jacobian A;
110 this->X_ = this->Dynamics(phi, this->X_, U, A);
111 this->P_ = A * this->P_ * A.transpose() + Q;
112 }
113};
