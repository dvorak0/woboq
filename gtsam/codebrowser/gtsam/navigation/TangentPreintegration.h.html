18#pragma once
19
20#include <gtsam/navigation/PreintegrationBase.h>
21
22namespace gtsam {
23
24/**
25 * Integrate on the 9D tangent space of the NavState manifold.
26 * See extensive discussion in ImuFactor.lyx
27 */
28class GTSAM_EXPORT TangentPreintegration : public PreintegrationBase {
29 protected:
30
31 /**
32 * Preintegrated navigation state, as a 9D vector on tangent space at frame i
33 * Order is: theta, position, velocity
34 */
35 Vector9 preintegrated_;
36 Matrix93 preintegrated_H_biasAcc_; ///< Jacobian of preintegrated_ w.r.t. acceleration bias
37 Matrix93 preintegrated_H_biasOmega_; ///< Jacobian of preintegrated_ w.r.t. angular rate bias
38
39 /// Default constructor for serialization
40 TangentPreintegration() {
41 resetIntegration();
42 }
43
44public:
45 /// @name Constructors/destructors
46 /// @{
47
48 /**
49 * Constructor, initializes the variables in the base class
50 * @param p Parameters, typically fixed in a single application
51 * @param bias Current estimate of acceleration and rotation rate biases
52 */
53 TangentPreintegration(const std::shared_ptr<Params>& p,
54 const imuBias::ConstantBias& biasHat = imuBias::ConstantBias());
55
56 /// Virtual destructor
57 ~TangentPreintegration() override {
58 }
59
60 /// @}
61
62 /// @name Basic utilities
63 /// @{
64 /// Re-initialize PreintegratedMeasurements
65 void resetIntegration() override;
66
67 /// @}
68
69 /// @name Instance variables access
70 /// @{
71 Vector3 deltaPij() const override { return preintegrated_.segment<3>(start: 3); }
72 Vector3 deltaVij() const override { return preintegrated_.tail<3>(); }
73 Rot3 deltaRij() const override { return Rot3::Expmap(v: theta()); }
74 NavState deltaXij() const override { return NavState().retract(v: preintegrated_); }
75
76 const Vector9& preintegrated() const { return preintegrated_; }
77 Vector3 theta() const { return preintegrated_.head<3>(); }
78 const Matrix93& preintegrated_H_biasAcc() const { return preintegrated_H_biasAcc_; }
79 const Matrix93& preintegrated_H_biasOmega() const { return preintegrated_H_biasOmega_; }
80
81 /// @name Testable
82 /// @{
83 bool equals(const TangentPreintegration& other, double tol) const;
84 /// @}
85
86 /// @name Main functionality
87 /// @{
88
89 // Update integrated vector on tangent manifold preintegrated with acceleration
90 // Static, functional version.
91 static Vector9 UpdatePreintegrated(const Vector3& a_body,
92 const Vector3& w_body, const double dt,
93 const Vector9& preintegrated,
94 OptionalJacobian<9, 9> A = {},
95 OptionalJacobian<9, 3> B = {},
96 OptionalJacobian<9, 3> C = {});
97
98 /// Update preintegrated measurements and get derivatives
99 /// It takes measured quantities in the j frame
100 /// Modifies preintegrated quantities in place after correcting for bias and possibly sensor pose
101 /// NOTE(frank): implementation is different in two versions
102 void update(const Vector3& measuredAcc, const Vector3& measuredOmega,
103 const double dt, Matrix9* A, Matrix93* B, Matrix93* C) override;
104
105 /// Given the estimate of the bias, return a NavState tangent vector
106 /// summarizing the preintegrated IMU measurements so far
107 /// NOTE(frank): implementation is different in two versions
108 Vector9 biasCorrectedDelta(const imuBias::ConstantBias& bias_i,
109 OptionalJacobian<9, 6> H = {}) const override;
110
111 // Compose the two pre-integrated 9D-vectors zeta01 and zeta02, with derivatives
112 static Vector9 Compose(const Vector9& zeta01, const Vector9& zeta12,
113 double deltaT12,
114 OptionalJacobian<9, 9> H1 = {},
115 OptionalJacobian<9, 9> H2 = {});
116
117 /// Merge in a different set of measurements and update bias derivatives accordingly
118 /// The derivatives apply to the preintegrated Vector9
119 void mergeWith(const TangentPreintegration& pim, Matrix9* H1, Matrix9* H2);
120 /// @}
121
122 /** Dummy clone for MATLAB */
123 virtual std::shared_ptr<TangentPreintegration> clone() const {
124 return std::shared_ptr<TangentPreintegration>();
125 }
126
127 /// @}
128
129private:
130#if GTSAM_ENABLE_BOOST_SERIALIZATION
131 /** Serialization function */
132 friend class boost::serialization::access;
133 template<class ARCHIVE>
134 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
135 namespace bs = ::boost::serialization;
136 ar & BOOST_SERIALIZATION_BASE_OBJECT_NVP(PreintegrationBase);
137 ar & BOOST_SERIALIZATION_NVP(preintegrated_);
138 ar & BOOST_SERIALIZATION_NVP(preintegrated_H_biasAcc_);
139 ar & BOOST_SERIALIZATION_NVP(preintegrated_H_biasOmega_);
140 }
141#endif
142
143public:
144 GTSAM_MAKE_ALIGNED_OPERATOR_NEW
145};
146
147} /// namespace gtsam
