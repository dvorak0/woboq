13 * @file LieGroupEKF.h

18 * Lie group. It provides predict methods with state-dependent dynamics
19 * functions. Please use the InvariantEKF class for prediction via group
20 * composition.

38/**
39 * @class LieGroupEKF
40 * @brief Extended Kalman Filter on a Lie group G, derived from ManifoldEKF
41 *
42 * @tparam G Lie group type (must satisfy LieGroup concept).
43 *
44 * This filter specializes ManifoldEKF for Lie groups, offering predict methods
45 * with state-dependent dynamics functions.
46 * Use the InvariantEKF class for prediction via group composition.
47 * For details on how static and dynamic dimensions are handled, please refer to
48 * the `ManifoldEKF` class documentation.
49 */
50template <typename G>
51class LieGroupEKF : public ManifoldEKF<G> {
52 public:
53 using Base = ManifoldEKF<G>; ///< Base class type
54 static constexpr int Dim = Base::Dim; ///< Compile-time dimension of G.
55 using TangentVector =
56 typename Base::TangentVector; ///< Tangent vector type for G.
57 /// Jacobian for group operations (Adjoint, Expmap derivatives, F).
58 using Jacobian = typename Base::Jacobian; // Eigen::Matrix<double, Dim, Dim>
59 using Covariance =
60 typename Base::Covariance; // Eigen::Matrix<double, Dim, Dim>
61
62 /**
63 * Constructor: initialize with state and covariance.
64 * @param X0 Initial state on Lie group G.
65 * @param P0 Initial covariance in the tangent space at X0.
66 */
67 LieGroupEKF(const G& X0, const Covariance& P0) : Base(X0, P0) {
68 static_assert(IsLieGroup<G>::value,
69 "Template parameter G must be a GTSAM Lie Group");
70 }
71
72 /**
73 * SFINAE check for correctly typed state-dependent dynamics function.
74 * Signature: TangentVector f(const G& X, OptionalJacobian<Dim, Dim> Df)
75 * Df = d(xi)/d(local(X))
76 */
77 template <typename Dynamics>
78 using enable_if_dynamics =
79 std::enable_if_t<!std::is_convertible_v<Dynamics, TangentVector> &&
80 std::is_invocable_r_v<TangentVector, Dynamics, const G&,
81 OptionalJacobian<Dim, Dim>&>>;
82
83 /**
84 * Predict mean and Jacobian A with state-dependent dynamics:
85 * xi = f(X_k, Df) (Compute tangent vector dynamics and Jacobian
86 * Df) U = Expmap(xi * dt, Dexp) (Compute motion increment U and Expmap
87 * Jacobian Dexp) X_{k+1} = X_k * U (Predict next state) A =
88 * Ad_{U^{-1}} + Dexp * Df * dt (Compute full state transition Jacobian)
89 *
90 * @tparam Dynamics Functor signature: TangentVector f(const G&,
91 * OptionalJacobian<Dim,Dim>&)
92 * @param f Dynamics functor returning tangent vector xi and its Jacobian Df
93 * w.r.t. local(X).
94 * @param dt Time step.
95 * @param A OptionalJacobian to store the computed state transition Jacobian
96 * A.
97 * @return Predicted state X_{k+1}.
98 */
99 template <typename Dynamics, typename = enable_if_dynamics<Dynamics>>
100 G predictMean(Dynamics&& f, double dt,
101 OptionalJacobian<Dim, Dim> A = {}) const {
102 Jacobian Df, Dexp; // Eigen::Matrix<double, Dim, Dim>
103
104 if constexpr (std::is_same_v<G, Matrix>) {
105 // Specialize to Matrix case as its Expmap is not defined.
106 TangentVector xi = f(this->X_, A ? &Df : nullptr);
107 const Matrix nextX = traits<Matrix>::Retract(
108 m: this->X_, v: xi * dt, H1: A ? &Dexp : nullptr); // just addition
109 if (A) {
110 const Matrix I_n = Matrix::Identity(this->n_, this->n_);
111 *A = I_n + Dexp * Df * dt; // AdjointMap is always identity for Matrix
112 }
113 return nextX;
114 } else {
115 TangentVector xi =
116 f(this->X_, A ? &Df : nullptr); // xi and Df = d(xi)/d(localX)
117 G U = traits<G>::Expmap(xi * dt, A ? &Dexp : nullptr);
118 if (A) {
119 // State transition Jacobian for left-invariant EKF:
120 *A = traits<G>::Inverse(U).AdjointMap() + Dexp * Df * dt;
121 }
122 return this->X_.compose(U);
123 }
124 }
125
126 /**
127 * Predict step with state-dependent dynamics:
128 * Uses predictMean to compute X_{k+1} and A, then updates covariance.
129 * X_{k+1}, A = predictMean(f, dt)
130 * P_{k+1} = A P_k A^T + Q
131 *
132 * @tparam Dynamics Functor signature: TangentVector f(const G&,
133 * OptionalJacobian<Dim,Dim>&)
134 * @param f Dynamics functor.
135 * @param dt Time step.
136 * @param Q Process noise covariance.
137 */
138 template <typename Dynamics, typename = enable_if_dynamics<Dynamics>>
139 void predict(Dynamics&& f, double dt, const Covariance& Q) {
140 Jacobian A;
141 if constexpr (Dim == Eigen::Dynamic) {
142 A.resize(this->n_, this->n_);
143 }
144 this->X_ = predictMean(std::forward<Dynamics>(f), dt, A);
145 this->P_ = A * this->P_ * A.transpose() + Q;
146 }
147
148 /**
149 * SFINAE check for state- and control-dependent dynamics function.
150 * Signature: TangentVector f(const G& X, const Control& u,
151 * OptionalJacobian<Dim, Dim> Df)
152 */
153 template <typename Control, typename Dynamics>
154 using enable_if_full_dynamics = std::enable_if_t<
155 std::is_invocable_r_v<TangentVector, Dynamics, const G&, const Control&,
156 OptionalJacobian<Dim, Dim>&>>;
157
158 /**
159 * Predict mean and Jacobian A with state and control input dynamics:
160 * Wraps the dynamics function and calls the state-only predictMean.
161 * xi = f(X_k, u, Df)
162 *
163 * @tparam Control Control input type.
164 * @tparam Dynamics Functor signature: TangentVector f(const G&, const
165 * Control&, OptionalJacobian<Dim,Dim>&)
166 * @param f Dynamics functor.
167 * @param u Control input.
168 * @param dt Time step.
169 * @param A Optional pointer to store the computed state transition Jacobian
170 * A.
171 * @return Predicted state X_{k+1}.
172 */
173 template <typename Control, typename Dynamics,
174 typename = enable_if_full_dynamics<Control, Dynamics>>
175 G predictMean(Dynamics&& f, const Control& u, double dt,
176 OptionalJacobian<Dim, Dim> A = {}) const {
177 return predictMean(
178 [&](const G& X, OptionalJacobian<Dim, Dim> Df) { return f(X, u, Df); },
179 dt, A);
180 }
181
182 /**
183 * Predict step with state and control input dynamics:
184 * Wraps the dynamics function and calls the state-only predict.
185 * xi = f(X_k, u, Df)
186 *
187 * @tparam Control Control input type.
188 * @tparam Dynamics Functor signature: TangentVector f(const G&, const
189 * Control&, OptionalJacobian<Dim,Dim>&)
190 * @param f Dynamics functor.
191 * @param u Control input.
192 * @param dt Time step.
193 * @param Q Process noise covariance.
194 */
195 template <typename Control, typename Dynamics,
196 typename = enable_if_full_dynamics<Control, Dynamics>>
197 void predict(Dynamics&& f, const Control& u, double dt, const Covariance& Q) {
198 return predict(
199 [&](const G& X, OptionalJacobian<Dim, Dim> Df) { return f(X, u, Df); },
200 dt, Q);
201 }
202
203 /**
204 * Predict using a precomputed group increment U and its Jacobian J_UX.
205 *
206 * Contract:
207 * - Input state X_k on G with covariance P_k in local coordinates
208 * - Precomputed increment U = U(X_k) in G
209 * - Jacobian J_UX = d(u_left)/d(local(X)) at X_k, where u_left = Log(U)
210 * - Process noise Q expressed in the same coordinates as u_left
211 *
212 * Update performed:
213 * - X_{k+1} = X_k âˆ˜ U
214 * - A = Ad_{U^{-1}} + J_UX
215 * - P_{k+1} = A P_k A^T + Q
216 *
217 * Notes:
218 * This API is intended for custom integrators that construct U(X) directly
219 * (e.g., second-order kinematics), which may not be expressible as an Euler
220 * step on the tangent used by predict/predictMean.
221 */
222 void predictWithCompose(const G& U, const Jacobian& J_UX,
223 const Covariance& Q) {
224 Jacobian A_local;
225 if constexpr (std::is_same_v<G, Matrix>) {
226 const Matrix I_n = Matrix::Identity(this->n_, this->n_);
227 A_local = I_n + J_UX;
228 this->X_ = traits<Matrix>::Retract(m: this->X_, v: U);
229 } else {
230 A_local = traits<G>::Inverse(U).AdjointMap() + J_UX;
231 this->X_ = this->X_.compose(U);
232 }
233 this->P_ = A_local * this->P_ * A_local.transpose() + Q;
234 }
235
236}; // LieGroupEKF
