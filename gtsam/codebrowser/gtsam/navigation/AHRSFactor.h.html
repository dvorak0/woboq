31/**
32 * PreintegratedAHRSMeasurements accumulates (integrates) the gyroscope
33 * measurements (rotation rates) and the corresponding covariance matrix.
34 * Can be built incrementally so as to avoid costly integration at time of
35 * factor construction.
36 *
37 * @section math_notes Mathematical Formulation
38 *
39 * The preintegrated rotation is updated incrementally with each gyroscope
40 * measurement. Given a gyroscope measurement \f$ \omega_k \f$ at time \f$ t_k \f$,
41 * the preintegrated rotation \f$ \Delta R_{ij} \f$ from time \f$ t_i \f$ to \f$ t_j \f$
42 * is the product of many small rotations:
43 * \f[
44 * \Delta R_{ij} = \prod_{k=i}^{j-1} \text{Exp}((\omega_k - b_g) \Delta t)
45 * \f]
46 * where \f$ b_g \f$ is the gyroscope bias, and \f$ \text{Exp}(\cdot) \f$ is the
47 * exponential map from \f$ \mathbb{R}^3 \f$ to SO(3).
48 *
49 * This class also propagates the covariance of the preintegrated rotation.
50 */
51class GTSAM_EXPORT PreintegratedAhrsMeasurements
52 : public PreintegratedRotation {
53 protected:
54 Vector3 biasHat_; ///< Angular rate bias values used during preintegration.
55 Matrix3 preintMeasCov_; ///< Covariance matrix of the preintegrated
56 ///< measurements (first-order propagation from
57 ///< *measurementCovariance*)
58
59 friend class AHRSFactor;
60
61 public:
62 /// Default constructor, only for serialization and wrappers
63 PreintegratedAhrsMeasurements() {}
64
65 /**
66 * Default constructor, initialize with no measurements
67 * @param bias Current estimate of rotation rate biases
68 */
69 PreintegratedAhrsMeasurements(const std::shared_ptr<Params>& p,
70 const Vector3& biasHat)
71 : PreintegratedRotation(p), biasHat_(biasHat) {
72 resetIntegration();
73 }
74
75 /**
76 * Non-Default constructor, initialize with measurements
77 * @param p: Parameters for AHRS pre-integration
78 * @param bias_hat: Current estimate of rotation rate biases
79 * @param deltaTij: Delta time in pre-integration
80 * @param deltaRij: Delta rotation in pre-integration
81 * @param delRdelBiasOmega: Jacobian of rotation wrt. to gyro bias
82 * @param preint_meas_cov: Pre-integration covariance
83 */
84 PreintegratedAhrsMeasurements(const std::shared_ptr<Params>& p,
85 const Vector3& bias_hat, double deltaTij,
86 const Rot3& deltaRij,
87 const Matrix3& delRdelBiasOmega,
88 const Matrix3& preint_meas_cov)
89 : PreintegratedRotation(p, deltaTij, deltaRij, delRdelBiasOmega),
90 biasHat_(bias_hat),
91 preintMeasCov_(preint_meas_cov) {}
92
93 Params& p() const { return *std::static_pointer_cast<Params>(r: p_); }
94 const Vector3& biasHat() const { return biasHat_; }
95 const Matrix3& preintMeasCov() const { return preintMeasCov_; }
96
97 /// print
98 void print(const std::string& s = "Preintegrated Measurements: ") const;
99
100 /// equals
101 bool equals(const PreintegratedAhrsMeasurements& expected,
102 double tol = 1e-9) const;
103
104 /// Reset integrated quantities to zero
105 void resetIntegration();
106
107 /**
108 * Add a single gyroscope measurement to the preintegration.
109 * Measurements are taken to be in the sensor
110 * frame and conversion to the body frame is handled by `body_P_sensor` in
111 * `PreintegratedRotationParams` (if provided).
112 *
113 * @param measuredOmega Measured angular velocity (as given by the sensor)
114 * @param deltaT Time step
115 */
116 void integrateMeasurement(const Vector3& measuredOmega, double deltaT);
117
118 /**
119 * Predict the orientation at time j, given orientation and bias at time i.
120 * @param Ri orientation at time i
121 * @param bias gyroscope bias
122 * @param H1 optional 3x3 Jacobian wrt Ri
123 * @param H2 optional 3x3 Jacobian wrt bias
124 * @return predicted orientation at time j
125 */
126 Rot3 predict(const Rot3& Ri, const Vector3& bias,
127 gtsam::OptionalJacobian<3, 3> H1 = {},
128 gtsam::OptionalJacobian<3, 3> H2 = {}) const;
129
130 /**
131 * Calculate the error between the predicted and actual rotation.
132 * @param Ri The orientation at time i
133 * @param Rj The orientation at time j
134 * @param bias The gyroscope bias
135 * @param H1 Optional Jacobian of the error with respect to Ri
136 * @param H2 Optional Jacobian of the error with respect to Rj
137 * @param H3 Optional Jacobian of the error with respect to bias
138 * @return A 3D vector containing the rotation error.
139 */
140 Vector3 computeError(const Rot3& Ri, const Rot3& Rj, const Vector3& bias,
141 gtsam::OptionalJacobian<3, 3> H1 = {},
142 gtsam::OptionalJacobian<3, 3> H2 = {},
143 gtsam::OptionalJacobian<3, 3> H3 = {}) const;
144
145 /// @deprecated constructor, but used in tests.
146 PreintegratedAhrsMeasurements(const Vector3& biasHat,
147 const Matrix3& measuredOmegaCovariance)
148 : PreintegratedRotation(std::make_shared<Params>()), biasHat_(biasHat) {
149 p_->gyroscopeCovariance = measuredOmegaCovariance;
150 resetIntegration();
151 }
152
153 private:
154#if GTSAM_ENABLE_BOOST_SERIALIZATION
155 /** Serialization function */
156 friend class boost::serialization::access;
157 template <class ARCHIVE>
158 void serialize(ARCHIVE& ar, const unsigned int /*version*/) {
159 ar& BOOST_SERIALIZATION_BASE_OBJECT_NVP(PreintegratedRotation);
160 ar& BOOST_SERIALIZATION_NVP(p_);
161 ar& BOOST_SERIALIZATION_NVP(biasHat_);
162 }
163#endif
164};

166/**
167 * An AHRSFactor is a three-way factor that is based on the preintegrated
168 * gyroscope measurements.
169 *
170 * @section math_notes Mathematical Formulation
171 *
172 * The factor relates the orientation at two time steps, \f$ R_i \f$ and \f$ R_j \f$,
173 * and the gyroscope bias \f$ b_g \f$. The error function is given by:
174 * \f[
175 * e(R_i, R_j, b_g) = \text{Log}\left( (\Delta \tilde{R}_{ij}(b_g))^{-1} R_i^{-1} R_j \right)
176 * \f]
177 * where \f$ \Delta \tilde{R}_{ij}(b_g) \f$ is the preintegrated rotation corrected
178 * for the current estimate of the gyroscope bias, and \f$ \text{Log}(\cdot) \f$ is
179 * the logarithmic map from SO(3) to \f$ \mathbb{R}^3 \f$.
180 *
181 * The preintegrated rotation \f$ \Delta R_{ij} \f$ is calculated as:
182 * \f[
183 * \Delta R_{ij} = \prod_{k=i}^{j-1} \text{Exp}((\omega_k - \hat{b}_g) \Delta t)
184 * \f]
185 * where \f$ \hat{b}_g \f$ is the bias estimate used for preintegration. The
186 * bias-corrected preintegrated rotation \f$ \Delta \tilde{R}_{ij}(b_g) \f$ is
187 * then approximated using a first-order expansion:
188 * \f[
189 * \Delta \tilde{R}_{ij}(b_g) \approx \Delta R_{ij} \text{Exp}(J_b (b_g - \hat{b}_g))
190 * \f]
191 * where \f$ J_b \f$ is the Jacobian of the preintegrated rotation with respect
192 * to the gyroscope bias.
193 */
194class GTSAM_EXPORT AHRSFactor : public NoiseModelFactorN<Rot3, Rot3, Vector3> {
195 typedef AHRSFactor This;
196 typedef NoiseModelFactorN<Rot3, Rot3, Vector3> Base;
197
198 PreintegratedAhrsMeasurements _PIM_;
199
200 public:
201 // Provide access to the Matrix& version of evaluateError:
202 using Base::evaluateError;
203

205#if !defined(_MSC_VER) && __GNUC__ == 4 && __GNUC_MINOR__ > 5
206 typedef typename std::shared_ptr<AHRSFactor> shared_ptr;
207#else
208 typedef std::shared_ptr<AHRSFactor> shared_ptr;
209#endif
