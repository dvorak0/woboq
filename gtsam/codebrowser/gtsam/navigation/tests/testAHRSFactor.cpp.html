56PreintegratedAhrsMeasurements integrateMeasurements(
57 const Vector3& biasHat, const list<Vector3>& measuredOmegas,
58 const list<double>& deltaTs) {
59 PreintegratedAhrsMeasurements result(biasHat, I_3x3);
60
61 list<Vector3>::const_iterator itOmega = measuredOmegas.begin();
62 list<double>::const_iterator itDeltaT = deltaTs.begin();
63 for (; itOmega != measuredOmegas.end(); ++itOmega, ++itDeltaT) {
64 result.integrateMeasurement(measuredOmega: *itOmega, deltaT: *itDeltaT);
65 }
66
67 return result;
68}

72TEST(AHRSFactor, PreintegratedAhrsMeasurements) {
73 // Linearization point
74 Vector3 biasHat(0, 0, 0); ///< Current estimate of angular rate bias
75
76 // Measurements
77 Vector3 measuredOmega(M_PI / 100.0, 0.0, 0.0);
78 double deltaT = 0.5;
79
80 // Expected preintegrated values
81 Rot3 expectedDeltaR1 = Rot3::Roll(t: 0.5 * M_PI / 100.0);
82
83 // Actual preintegrated values
84 PreintegratedAhrsMeasurements actual1(biasHat, kMeasuredOmegaCovariance);
85 actual1.integrateMeasurement(measuredOmega, deltaT);
86
87 EXPECT(assert_equal(expectedDeltaR1, Rot3(actual1.deltaRij()), 1e-6));
88 DOUBLES_EQUAL(deltaT, actual1.deltaTij(), 1e-6);
89
90 // Check the covariance
91 Matrix3 expectedMeasCov = kMeasuredOmegaCovariance * deltaT;
92 EXPECT(assert_equal(expectedMeasCov, actual1.preintMeasCov(), 1e-6));
93
94 // Integrate again
95 Rot3 expectedDeltaR2 = Rot3::Roll(t: 2.0 * 0.5 * M_PI / 100.0);
96
97 // Actual preintegrated values
98 PreintegratedAhrsMeasurements actual2 = actual1;
99 actual2.integrateMeasurement(measuredOmega, deltaT);
100
101 EXPECT(assert_equal(expectedDeltaR2, Rot3(actual2.deltaRij()), 1e-6));
102 DOUBLES_EQUAL(deltaT * 2, actual2.deltaTij(), 1e-6);
103}

164 std::function<Vector3(const Rot3&, const Rot3&, const Vector3&)> f =

171 (void)pim.computeError(Ri, Rj, bias, H1, H2, H3);

177
178 // Compare
179 EXPECT(assert_equal(H1_numerical, H1, 1e-6));
180 EXPECT(assert_equal(H2_numerical, H2, 1e-6));
181 EXPECT(assert_equal(H3_numerical, H3, 1e-6));

269TEST(AHRSFactor, PartialDerivativeLogmap) {
270 // Linearization point
271 Vector3 thetaHat(0.1, 0.1, 0); ///< Current estimate of rotation rate bias
272
273 auto f = [thetaHat](const Vector3 deltaTheta) {
274 return Rot3::Logmap(
275 R: Rot3::Expmap(v: thetaHat).compose(g: Rot3::Expmap(v: deltaTheta)));
276 };
277
278 // Compute numerical derivatives
279 Vector3 deltaTheta(0, 0, 0);
280 Matrix expectedH = numericalDerivative11<Vector3, Vector3>(h: f, x: deltaTheta);
281
282 const Vector3 x = thetaHat; // parametrization of so(3)
283 const Matrix3 X = skewSymmetric(w: x); // element of Lie algebra so(3): X = x^
284 double norm = x.norm();
285 const Matrix3 actualH =
286 I_3x3 + 0.5 * X +
287 (1 / (norm * norm) - (1 + cos(x: norm)) / (2 * norm * sin(x: norm))) * X * X;
288
289 // Compare Jacobians
290 EXPECT(assert_equal(expectedH, actualH));
291}

513TEST(AHRSFactor, bodyPSensorWithBias) {
514 using noiseModel::Diagonal;
515
516 int numRotations = 10;
517 const Vector3 noiseBetweenBiasSigma(3.0e-6, 3.0e-6, 3.0e-6);
518 SharedDiagonal biasNoiseModel = Diagonal::Sigmas(sigmas: noiseBetweenBiasSigma);
519
520 // Measurements in the sensor frame:
521 const double omega = 0.1;
522 const Vector3 realOmega(omega, 0, 0);
523 const Vector3 realBias(1, 2, 3); // large !
524 const Vector3 measuredOmega = realOmega + realBias;
525
526 auto p = std::make_shared<PreintegratedAhrsMeasurements::Params>();
527 p->body_P_sensor = Pose3(Rot3::Yaw(M_PI_2), Point3(0, 0, 0));
528 p->gyroscopeCovariance = 1e-8 * I_3x3;
529 double deltaT = 0.005;
530
531 // Specify noise values on priors
532 const Vector3 priorNoisePoseSigmas(0.001, 0.001, 0.001);
533 const Vector3 priorNoiseBiasSigmas(0.5e-1, 0.5e-1, 0.5e-1);
534 SharedDiagonal priorNoisePose = Diagonal::Sigmas(sigmas: priorNoisePoseSigmas);
535 SharedDiagonal priorNoiseBias = Diagonal::Sigmas(sigmas: priorNoiseBiasSigmas);
536
537 // Create a factor graph with priors on initial pose, velocity and bias
538 NonlinearFactorGraph graph;
539 Values values;
540
541 graph.addPrior(key: R(j: 0), prior: Rot3(), model: priorNoisePose);
542 values.insert(j: R(j: 0), val: Rot3());
543
544 // The key to this test is that we specify the bias, in the sensor frame, as
545 // known a priori. We also create factors below that encode our assumption
546 // that this bias is constant over time In theory, after optimization, we
547 // should recover that same bias estimate
548 graph.addPrior(key: B(j: 0), prior: realBias, model: priorNoiseBias);
549 values.insert(j: B(j: 0), val: realBias);
550
551 // Now add IMU factors and bias noise models
552 const Vector3 zeroBias(0, 0, 0);
553 for (int i = 1; i < numRotations; i++) {
554 PreintegratedAhrsMeasurements pim(p, realBias);
555 for (int j = 0; j < 200; ++j)
556 pim.integrateMeasurement(measuredOmega, deltaT);
557
558 // Create factors
559 graph.emplace_shared<AHRSFactor>(args: R(j: i - 1), args: R(j: i), args: B(j: i - 1), args&: pim);
560 graph.emplace_shared<BetweenFactor<Vector3> >(args: B(j: i - 1), args: B(j: i), args: zeroBias,
561 args&: biasNoiseModel);
562
563 values.insert(j: R(j: i), val: Rot3());
564 values.insert(j: B(j: i), val: realBias);
565 }
566
567 // Finally, optimize, and get bias at last time step
568 LevenbergMarquardtParams params;
569 // params.setVerbosityLM("SUMMARY");
570 Values result = LevenbergMarquardtOptimizer(graph, values, params).optimize();
571 const Vector3 biasActual = result.at<Vector3>(j: B(j: numRotations - 1));
572
573 // Bias should be a self-fulfilling prophesy:
574 EXPECT(assert_equal(realBias, biasActual, 1e-3));
575
576 // Check that the successive rotations are all `omega` apart:
577 for (int i = 0; i < numRotations; i++) {
578 Rot3 expectedRot = Rot3::Pitch(t: omega * i);
579 Rot3 actualRot = result.at<Rot3>(j: R(j: i));
580 EXPECT(assert_equal(expectedRot, actualRot, 1e-3));
581 }
