28// Create a 2D GPS measurement function that returns the predicted measurement h
29// and Jacobian H. The predicted measurement h is the translation of the state X.
30// (Copied from IEKF_SE2Example.cpp)
31Vector2 h_gps(const Pose2& X, OptionalJacobian<2, 3> H = {}) {
32 return X.translation(Hself: H);
33}

36TEST(IEKF_Pose2, PredictUpdateSequence) {
37 // GIVEN: Initial state, covariance, noises, motions, measurements
38 // (from IEKF_SE2Example.cpp)
39 Pose2 X0(0.0, 0.0, 0.0);
40 Matrix3 P0 = Matrix3::Identity() * 0.1;
41
42 Matrix3 Q = (Vector3(0.05, 0.05, 0.001)).asDiagonal();
43 Matrix2 R = I_2x2 * 0.01;
44
45 Pose2 U1(1.0, 1.0, 0.5), U2(1.0, 1.0, 0.0);
46
47 Vector2 z1, z2;
48 z1 << 1.0, 0.0;
49 z2 << 1.0, 1.0;
50
51 // Create the filter
52 InvariantEKF<Pose2> ekf(X0, P0);
53 EXPECT(assert_equal(X0, ekf.state()));
54 EXPECT(assert_equal(P0, ekf.covariance()));
55
56 // --- First Prediction ---
57 ekf.predict(U: U1, Q);
58
59 // Calculate expected state and covariance
60 Pose2 X1_expected = X0.compose(g: U1);
61 Matrix3 Ad_U1_inv = U1.inverse().AdjointMap();
62 Matrix3 P1_expected = Ad_U1_inv * P0 * Ad_U1_inv.transpose() + Q;
63
64 // Verify
65 EXPECT(assert_equal(X1_expected, ekf.state(), 1e-9));
66 EXPECT(assert_equal(P1_expected, ekf.covariance(), 1e-9));
67
68
69 // --- First Update ---
70 ekf.update(h&: h_gps, z: z1, R);
71
72 // Calculate expected state and covariance (manual Kalman steps)
73 Matrix H1; // H = dh/dlocal(X) -> 2x3
74 Vector2 z_pred1 = h_gps(X: X1_expected, H: H1);
75 Vector2 y1 = z_pred1 - z1; // Innovation
76 Matrix S1 = H1 * P1_expected * H1.transpose() + R;
77 Matrix K1 = P1_expected * H1.transpose() * S1.inverse(); // Gain (3x2)
78 Vector3 delta_xi1 = -K1 * y1; // Correction (tangent space)
79 Pose2 X1_updated_expected = X1_expected.retract(v: delta_xi1);
80 Matrix3 I_KH1 = Matrix3::Identity() - K1 * H1;
81 Matrix3 P1_updated_expected = I_KH1 * P1_expected; // Standard form P = (I-KH)P
82
83 // Verify
84 EXPECT(assert_equal(X1_updated_expected, ekf.state(), 1e-9));
85 EXPECT(assert_equal(P1_updated_expected, ekf.covariance(), 1e-9));
86
87
88 // --- Second Prediction ---
89 ekf.predict(U: U2, Q);
90
91 // Calculate expected state and covariance
92 Pose2 X2_expected = X1_updated_expected.compose(g: U2);
93 Matrix3 Ad_U2_inv = U2.inverse().AdjointMap();
94 Matrix3 P2_expected = Ad_U2_inv * P1_updated_expected * Ad_U2_inv.transpose() + Q;
95
96 // Verify
97 EXPECT(assert_equal(X2_expected, ekf.state(), 1e-9));
98 EXPECT(assert_equal(P2_expected, ekf.covariance(), 1e-9));
99
100
101 // --- Second Update ---
102 ekf.update(h&: h_gps, z: z2, R);
103
104 // Calculate expected state and covariance (manual Kalman steps)
105 Matrix H2; // 2x3
106 Vector2 z_pred2 = h_gps(X: X2_expected, H: H2);
107 Vector2 y2 = z_pred2 - z2; // Innovation
108 Matrix S2 = H2 * P2_expected * H2.transpose() + R;
109 Matrix K2 = P2_expected * H2.transpose() * S2.inverse(); // Gain (3x2)
110 Vector3 delta_xi2 = -K2 * y2; // Correction (tangent space)
111 Pose2 X2_updated_expected = X2_expected.retract(v: delta_xi2);
112 Matrix3 I_KH2 = Matrix3::Identity() - K2 * H2;
113 Matrix3 P2_updated_expected = I_KH2 * P2_expected; // Standard form
114
115 // Verify
116 EXPECT(assert_equal(X2_updated_expected, ekf.state(), 1e-9));
117 EXPECT(assert_equal(P2_updated_expected, ekf.covariance(), 1e-9));
118
119}

121// Define simple dynamics and measurement for a 2x2 Matrix state
122namespace invariant_ekf_example {
123 Matrix f(const Matrix& p, const Vector& vTangent, double dt) {
124 return traits<Matrix>::Retract(m: p, v: vTangent * dt);
125 }
126 double h(const Matrix& p, OptionalJacobian<-1, -1> H = {}) {
127 if (H) {
128 H->resize(rows: 1, cols: p.size());
129 (*H) << 1.0, 0.0, 0.0, 1.0; // Assuming 2x2
130 }
131 return p(0, 0) + p(1, 1); // trace !
132 }
133} // namespace invariant_ekf_example

135TEST(InvariantEKF_DynamicMatrix, PredictAndUpdate) {
136 // --- Setup ---
137 Matrix p0Matrix = (Matrix(2, 2) << 1.0, 2.0, 3.0, 4.0).finished();
138 Matrix p0Covariance = I_4x4 * 0.01;
139 Vector velocityTangent = (Vector(4) << 0.5, 0.1, -0.1, -0.5).finished();
140 double dt = 0.1;
141 Matrix Q = I_4x4 * 0.001;
142 Matrix R = Matrix::Identity(rows: 1, cols: 1) * 0.005;
143
144 InvariantEKF<Matrix> ekf(p0Matrix, p0Covariance);
145 EXPECT_LONGS_EQUAL(4, ekf.state().size());
146 EXPECT_LONGS_EQUAL(4, ekf.dimension());
147
148 // --- Predict ---
149 ekf.predict(u: velocityTangent, dt, Q);
150
151 // Verification for Predict
152 Matrix pPredictedExpected = invariant_ekf_example::f(p: p0Matrix, vTangent: velocityTangent, dt);
153 Matrix pCovariancePredictedExpected = p0Covariance + Q;
154 EXPECT(assert_equal(pPredictedExpected, ekf.state(), 1e-9));
155 EXPECT(assert_equal(pCovariancePredictedExpected, ekf.covariance(), 1e-9));
156
157 // --- Update ---
158 // Use the state after prediction for the update step
159 Matrix pStateBeforeUpdate = ekf.state();
160 Matrix pCovarianceBeforeUpdate = ekf.covariance();
161
162 double zTrue = invariant_ekf_example::h(p: pStateBeforeUpdate);
163 double zObserved = zTrue - 0.03; // Simulated measurement with some error
164
165 ekf.update(h&: invariant_ekf_example::h, z: zObserved, R);
166
167 // Verification for Update (Manual Kalman Steps)
168 Matrix H(1, 4);
169 double zPredictionManual = invariant_ekf_example::h(p: pStateBeforeUpdate, H);
170 double innovationY_tangent = zObserved - zPredictionManual;
171 Matrix S = H * pCovarianceBeforeUpdate * H.transpose() + R;
172 Matrix kalmanGainK = pCovarianceBeforeUpdate * H.transpose() * S.inverse();
173 Vector deltaXiTangent = kalmanGainK * innovationY_tangent;
174 Matrix pUpdatedExpected = traits<Matrix>::Retract(m: pStateBeforeUpdate, v: deltaXiTangent);
175 Matrix I_KH = I_4x4 - kalmanGainK * H;
176 Matrix pUpdatedCovarianceExpected = I_KH * pCovarianceBeforeUpdate * I_KH.transpose() + kalmanGainK * R * kalmanGainK.transpose();
177
178 EXPECT(assert_equal(pUpdatedExpected, ekf.state(), 1e-9));
179 EXPECT(assert_equal(pUpdatedCovarianceExpected, ekf.covariance(), 1e-9));
180}
