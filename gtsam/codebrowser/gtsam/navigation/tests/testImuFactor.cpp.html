73TEST(ImuFactor, Accelerating) {
74 const double a = 0.2, v = 50;
75
76 // Set up body pointing towards y axis, and start at 10,20,0 with velocity going in X
77 // The body itself has Z axis pointing down
78 const Rot3 nRb(Point3(0, 1, 0), Point3(1, 0, 0), Point3(0, 0, -1));
79 const Point3 initial_position(10, 20, 0);
80 const Vector3 initial_velocity(v, 0, 0);
81
82 const AcceleratingScenario scenario(nRb, initial_position, initial_velocity,
83 Vector3(a, 0, 0));
84
85 const double T = 3.0; // seconds
86 ScenarioRunner runner(scenario, testing::Params(), T / 10);
87
88 PreintegratedImuMeasurements pim = runner.integrate(T);
89 EXPECT(assert_equal(scenario.pose(T), runner.predict(pim).pose(), 1e-9));
90
91 Matrix9 estimatedCov = runner.estimateCovariance(T, N: 100);
92 EXPECT(assert_equal(estimatedCov, pim.preintMeasCov(), 0.1));
93}

96TEST_PIM(ImuFactor, PreintegratedMeasurements) {
97 // Measurements
98 const double a = 0.1, w = M_PI / 100.0;
99 Vector3 measuredAcc(a, 0.0, 0.0);
100 Vector3 measuredOmega(w, 0.0, 0.0);
101 double deltaT = 0.5;
102
103 // Expected pre-integrated values
104 Vector3 expectedDeltaR1(w * deltaT, 0.0, 0.0);
105 Vector3 expectedDeltaP1(0.5 * a * deltaT*deltaT, 0, 0);
106 Vector3 expectedDeltaV1(0.05, 0.0, 0.0);
107
108 // Actual pre-integrated values
109 PIM actual(testing::Params());
110 actual.integrateMeasurement(measuredAcc, measuredOmega, deltaT);
111 EXPECT(assert_equal(Rot3::Expmap(expectedDeltaR1), actual.deltaRij()));
112 EXPECT(assert_equal(expectedDeltaP1, actual.deltaPij()));
113 EXPECT(assert_equal(expectedDeltaV1, actual.deltaVij()));
114 DOUBLES_EQUAL(0.5, actual.deltaTij(), 1e-9);
115
116 // Check derivatives of computeError
117 Bias bias(Vector3(0.2, 0, 0), Vector3(0.1, 0, 0.3)); // Biases (acc, rot)
118 NavState x1, x2 = actual.predict(x1, bias);
119
120 {
121 Matrix9 aH1, aH2;
122 Matrix96 aH3;
123 actual.computeError(x1, x2, bias, aH1, aH2, aH3);
124 std::function<Vector9(const NavState&, const NavState&, const Bias&)> f =
125 std::bind(&PreintegrationBase::computeError, actual,
126 std::placeholders::_1, std::placeholders::_2, std::placeholders::_3,
127 nullptr, nullptr, nullptr);
128 EXPECT(assert_equal(numericalDerivative31(f, x1, x2, bias), aH1, 1e-9));
129 EXPECT(assert_equal(numericalDerivative32(f, x1, x2, bias), aH2, 1e-9));
130 EXPECT(assert_equal(numericalDerivative33(f, x1, x2, bias), aH3, 1e-9));
131 }
132
133 // Integrate again
134 Vector3 expectedDeltaR2(2.0 * 0.5 * M_PI / 100.0, 0.0, 0.0);
135 Vector3 expectedDeltaP2(0.025 + expectedDeltaP1(0) + 0.5 * 0.1 * 0.5 * 0.5, 0, 0);
136 Vector3 expectedDeltaV2 = Vector3(0.05, 0.0, 0.0) +
137 Rot3::Expmap(v: expectedDeltaR1) * measuredAcc * 0.5;
138
139 // Actual pre-integrated values
140 actual.integrateMeasurement(measuredAcc, measuredOmega, deltaT);
141 EXPECT(assert_equal(Rot3::Expmap(expectedDeltaR2), actual.deltaRij()));
142 EXPECT(assert_equal(expectedDeltaP2, actual.deltaPij()));
143 EXPECT(assert_equal(expectedDeltaV2, actual.deltaVij()));
144 DOUBLES_EQUAL(1.0, actual.deltaTij(), 1e-9);
145}

219TEST_PIM(ImuFactor, ErrorAndJacobians) {
220 using namespace common;
221 PIM pim(testing::Params());
222
223 pim.integrateMeasurement(measuredAcc, measuredOmega, deltaT);
224 EXPECT(assert_equal(state2, pim.predict(state1, kZeroBias)));
225
226 // Create factor
227 ImuFactorT<PIM> factor(X(j: 1), V(j: 1), X(j: 2), V(j: 2), B(j: 1), pim);
228
229 // Expected error
230 Vector expectedError(9);
231 expectedError << 0, 0, 0, 0, 0, 0, 0, 0, 0;
232 EXPECT(
233 assert_equal(expectedError,
234 factor.evaluateError(x1, v1, x2, v2, kZeroBias)));
235
236 Values values;
237 values.insert(j: X(j: 1), val: x1);
238 values.insert(j: V(j: 1), val: v1);
239 values.insert(j: X(j: 2), val: x2);
240 values.insert(j: V(j: 2), val: v2);
241 values.insert(j: B(j: 1), val: kZeroBias);
242 EXPECT(assert_equal(expectedError, factor.unwhitenedError(values)));
243
244 // Make sure linearization is correct
245 double diffDelta = 1e-7;
246 EXPECT_CORRECT_FACTOR_JACOBIANS(factor, values, diffDelta, 1e-3);
247
248 // Actual Jacobians
249 Matrix H1a, H2a, H3a, H4a, H5a;
250 (void) factor.evaluateError(x1, v1, x2, v2, kZeroBias, H1a, H2a, H3a, H4a,
251 H5a);
252
253 // Make sure rotation part is correct when error is interpreted as axis-angle
254 // Jacobians are around zero, so the rotation part is the same as:
255 Matrix H1Rot3 = numericalDerivative11<Rot3, Pose3>(
256 [&](const Pose3& pose_i) {
257 return Rot3::Expmap(v: factor.evaluateError(pose_i, v1, x2, v2, kZeroBias).head(3));
258 },
259 x1);
260 EXPECT(assert_equal(H1Rot3, H1a.topRows(3)));
261
262 Matrix H3Rot3 = numericalDerivative11<Rot3, Pose3>(
263 [&](const Pose3& pose_j) {
264 return Rot3::Expmap(v: factor.evaluateError(x1, v1, pose_j, v2, kZeroBias).head(3));
265 },
266 x2);
267 EXPECT(assert_equal(H3Rot3, H3a.topRows(3)));
268
269 // Evaluate error with wrong values
270 Vector3 v2_wrong = v2 + Vector3(0.1, 0.1, 0.1);
271 values.update(j: V(j: 2), val: v2_wrong);
272 expectedError << 0, 0, 0, 0, 0, 0, -0.0724744871, -0.040715657, -0.151952901;
273 EXPECT(
274 assert_equal(expectedError,
275 factor.evaluateError(x1, v1, x2, v2_wrong, kZeroBias), 1e-2));
276 EXPECT(assert_equal(expectedError, factor.unwhitenedError(values), 1e-2));
277
278 // Make sure the whitening is done correctly
279 Matrix cov = pim.preintMeasCov();
280 Matrix R = RtR(A: cov.inverse());
281 Vector whitened = R * expectedError;
282 EXPECT(assert_equal(0.5 * whitened.squaredNorm(), factor.error(values), 1e-4));
283
284 // Make sure linearization is correct
285 EXPECT_CORRECT_FACTOR_JACOBIANS(factor, values, diffDelta, 1e-3);
286}

422TEST(ImuFactor, fistOrderExponential) {
423 // Linearization point
424 Vector3 biasOmega(0, 0, 0); // Current estimate of rotation rate bias
425
426 // Measurements
427 Vector3 measuredOmega(0.1, 0, 0);
428 double deltaT = 1.0;
429
430 // change w.r.t. linearization point
431 double alpha = 0.0;
432 Vector3 deltaBiasOmega;
433 deltaBiasOmega << alpha, alpha, alpha;
434
435 const Matrix3 Jr = Rot3::ExpmapDerivative(
436 x: (measuredOmega - biasOmega) * deltaT);
437
438 Matrix3 delRdelBiasOmega = -Jr * deltaT; // the delta bias appears with the minus sign
439
440 const Matrix expectedRot = Rot3::Expmap(
441 v: (measuredOmega - biasOmega - deltaBiasOmega) * deltaT).matrix();
442
443 const Matrix3 hatRot =
444 Rot3::Expmap(v: (measuredOmega - biasOmega) * deltaT).matrix();
445 const Matrix3 actualRot = hatRot
446 * Rot3::Expmap(v: delRdelBiasOmega * deltaBiasOmega).matrix();
447
448 // This is a first order expansion so the equality is only an approximation
449 EXPECT(assert_equal(expectedRot, actualRot));
450}

678TEST_PIM(ImuFactor, bodyPSensorWithBias) {
679 gttic(bodyPSensorWithBias);
680 using noiseModel::Diagonal;
681 typedef Bias Bias;
682
683 int numPoses = 10;
684 Vector6 noiseBetweenBiasSigma;
685 noiseBetweenBiasSigma << Vector3(2.0e-5, 2.0e-5, 2.0e-5), Vector3(3.0e-6,
686 3.0e-6, 3.0e-6);
687 SharedDiagonal biasNoiseModel = Diagonal::Sigmas(sigmas: noiseBetweenBiasSigma);
688
689 // Measurements
690 // Sensor frame is z-down
691 // Gyroscope measurement is the angular velocity of sensor w.r.t nav frame in sensor frame
692 Vector3 measuredOmega(0, 0.01, 0);
693 // Acc measurement is acceleration of sensor in the sensor frame, when stationary,
694 // table exerts an equal and opposite force w.r.t gravity
695 Vector3 measuredAcc(0, 0, -kGravity);
696
697 auto p = testing::Params();
698 p->n_gravity = Vector3(0, 0, -kGravity);
699 p->body_P_sensor = Pose3(Rot3::Ypr(y: 0, p: 0, M_PI), Point3(0,0,0));
700 p->accelerometerCovariance = 1e-7 * I_3x3;
701 p->gyroscopeCovariance = 1e-8 * I_3x3;
702 p->integrationCovariance = 1e-9 * I_3x3;
703 double deltaT = 0.005;
704
705 // Specify noise values on priors
706 Vector6 priorNoisePoseSigmas(
707 (Vector(6) << 0.001, 0.001, 0.001, 0.01, 0.01, 0.01).finished());
708 Vector3 priorNoiseVelSigmas((Vector(3) << 0.1, 0.1, 0.1).finished());
709 Vector6 priorNoiseBiasSigmas(
710 (Vector(6) << 0.1, 0.1, 0.1, 0.5e-1, 0.5e-1, 0.5e-1).finished());
711 SharedDiagonal priorNoisePose = Diagonal::Sigmas(sigmas: priorNoisePoseSigmas);
712 SharedDiagonal priorNoiseVel = Diagonal::Sigmas(sigmas: priorNoiseVelSigmas);
713 SharedDiagonal priorNoiseBias = Diagonal::Sigmas(sigmas: priorNoiseBiasSigmas);
714 Vector3 zeroVel(0, 0, 0);
715
716 // Create a factor graph with priors on initial pose, velocity and bias
717 NonlinearFactorGraph graph;
718 Values values;
719
720 graph.addPrior(key: X(j: 0), prior: Pose3(), model: priorNoisePose);
721 values.insert(j: X(j: 0), val: Pose3());
722
723 graph.addPrior(key: V(j: 0), prior: zeroVel, model: priorNoiseVel);
724 values.insert(j: V(j: 0), val: zeroVel);
725
726 // The key to this test is that we specify the bias, in the sensor frame, as known a priori
727 // We also create factors below that encode our assumption that this bias is constant over time
728 // In theory, after optimization, we should recover that same bias estimate
729 Bias priorBias(Vector3(0, 0, 0), Vector3(0, 0.01, 0)); // Biases (acc, rot)
730 graph.addPrior(key: B(j: 0), prior: priorBias, model: priorNoiseBias);
731 values.insert(j: B(j: 0), val: priorBias);
732
733 // Now add IMU factors and bias noise models
734 Bias zeroBias(Vector3(0, 0, 0), Vector3(0, 0, 0));
735 for (int i = 1; i < numPoses; i++) {
736 PIM pim(p, priorBias);
737 for (int j = 0; j < 200; ++j)
738 pim.integrateMeasurement(measuredAcc, measuredOmega, deltaT);
739
740 // Create factors
741 using FACTOR = ImuFactorT<PIM>;
742 graph.emplace_shared<FACTOR>(X(j: i - 1), V(j: i - 1), X(j: i), V(j: i), B(j: i - 1), pim);
743 graph.emplace_shared<BetweenFactor<Bias> >(args: B(j: i - 1), args: B(j: i), args&: zeroBias, args&: biasNoiseModel);
744
745 values.insert(j: X(j: i), val: Pose3());
746 values.insert(j: V(j: i), val: zeroVel);
747 values.insert(j: B(j: i), val: priorBias);
748 }
749
750 // Finally, optimize, and get bias at last time step
751 Values result = LevenbergMarquardtOptimizer(graph, values).optimize();
752 Bias biasActual = result.at<Bias>(j: B(j: numPoses - 1));
753
754 // And compare it with expected value (our prior)
755 Bias biasExpected(Vector3(0, 0, 0), Vector3(0, 0.01, 0));
756 EXPECT(assert_equal(biasExpected, biasActual, 1e-3));
757}
