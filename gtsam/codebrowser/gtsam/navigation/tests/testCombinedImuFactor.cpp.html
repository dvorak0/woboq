37namespace combined {
38// Create default parameters with Z-down and above noise parameters
39static std::shared_ptr<PreintegratedCombinedMeasurements::Params> Params(
40 const Matrix3& biasAccCovariance = Matrix3::Zero(),
41 const Matrix3& biasOmegaCovariance = Matrix3::Zero()) {
42 auto p = PreintegratedCombinedMeasurements::Params::MakeSharedD(g: kGravity);
43 p->gyroscopeCovariance = kGyroSigma * kGyroSigma * I_3x3;
44 p->accelerometerCovariance = kAccelSigma * kAccelSigma * I_3x3;
45 p->integrationCovariance = 0.0001 * I_3x3;
46 p->biasAccCovariance = biasAccCovariance;
47 p->biasOmegaCovariance = biasOmegaCovariance;
48 return p;
49}
50} // namespace combined

53TEST_PIM(CombinedImuFactor, PreintegratedMeasurements ) {
54 // Linearization point
55 Bias bias(Vector3(0, 0, 0), Vector3(0, 0, 0)); ///< Current estimate of acceleration and angular rate biases
56
57 // Measurements
58 Vector3 measuredAcc(0.1, 0.0, 0.0);
59 Vector3 measuredOmega(M_PI / 100.0, 0.0, 0.0);
60 double deltaT = 0.5;
61 double tol = 1e-6;
62
63 auto p = combined::Params();
64
65 // Actual preintegrated values
66 PIM expected1(p, bias);
67 expected1.integrateMeasurement(measuredAcc, measuredOmega, deltaT);
68
69 CombinedPIM actual1(p, bias);
70
71 actual1.integrateMeasurement(measuredAcc, measuredOmega, deltaT);
72
73 EXPECT(assert_equal(Vector(expected1.deltaPij()), actual1.deltaPij(), tol));
74 EXPECT(assert_equal(Vector(expected1.deltaVij()), actual1.deltaVij(), tol));
75 EXPECT(assert_equal(expected1.deltaRij(), actual1.deltaRij(), tol));
76 DOUBLES_EQUAL(expected1.deltaTij(), actual1.deltaTij(), tol);
77}

81TEST_PIM(CombinedImuFactor, ErrorWithBiases ) {
82 Bias bias(Vector3(0.2, 0, 0), Vector3(0, 0, 0.3)); // Biases (acc, rot)
83 Bias bias2(Vector3(0.2, 0.2, 0), Vector3(1, 0, 0.3)); // Biases (acc, rot)
84 Pose3 x1(Rot3::Expmap(v: Vector3(0, 0, M_PI / 4.0)), Point3(5.0, 1.0, -50.0));
85 Vector3 v1(0.5, 0.0, 0.0);
86 Pose3 x2(Rot3::Expmap(v: Vector3(0, 0, M_PI / 4.0 + M_PI / 10.0)),
87 Point3(5.5, 1.0, -50.0));
88 Vector3 v2(0.5, 0.0, 0.0);
89
90 auto p = combined::Params();
91 p->omegaCoriolis = Vector3(0,0.1,0.1);
92 PIM pim(p, Bias(Vector3(0.2, 0.0, 0.0), Vector3(0.0, 0.0, 0.0)));
93
94 // Measurements
95 Vector3 measuredOmega;
96 measuredOmega << 0, 0, M_PI / 10.0 + 0.3;
97 Vector3 measuredAcc =
98 x1.rotation().unrotate(p: -p->n_gravity) + Vector3(0.2, 0.0, 0.0);
99 double deltaT = 1.0;
100 double tol = 1e-6;
101
102 pim.integrateMeasurement(measuredAcc, measuredOmega, deltaT);
103
104 CombinedPIM combined_pim(p, Bias(Vector3(0.2, 0.0, 0.0), Vector3(0.0, 0.0, 0.0)));
105
106 combined_pim.integrateMeasurement(measuredAcc, measuredOmega, deltaT);
107
108 // Create factor
109 ImuFactorT<PIM> imuFactor(X(j: 1), V(j: 1), X(j: 2), V(j: 2), B(j: 1), pim);
110
111 CombinedImuFactorT<CombinedPIM> combinedFactor(X(j: 1), V(j: 1), X(j: 2), V(j: 2), B(j: 1), B(j: 2),
112 combined_pim);
113
114 Vector errorExpected = imuFactor.evaluateError(x1, v1, x2, v2, bias);
115 Vector errorActual = combinedFactor.evaluateError(x1, v1, x2, v2, bias,
116 bias2);
117 EXPECT(assert_equal(errorExpected, errorActual.head(9), tol));
118
119 // Expected Jacobians
120 Matrix H1e, H2e, H3e, H4e, H5e;
121 (void) imuFactor.evaluateError(x1, v1, x2, v2, bias, H1e, H2e, H3e, H4e, H5e);
122
123 // Actual Jacobians
124 Matrix H1a, H2a, H3a, H4a, H5a, H6a;
125 (void) combinedFactor.evaluateError(x1, v1, x2, v2, bias, bias2, H1a, H2a,
126 H3a, H4a, H5a, H6a);
127
128 EXPECT(assert_equal(H1e, H1a.topRows(9)));
129 EXPECT(assert_equal(H2e, H2a.topRows(9)));
130 EXPECT(assert_equal(H3e, H3a.topRows(9)));
131 EXPECT(assert_equal(H4e, H4a.topRows(9)));
132 EXPECT(assert_equal(H5e, H5a.topRows(9)));
133}

137TEST(CombinedImuFactor, FirstOrderPreIntegratedMeasurements) {
138 auto p = combined::Params();
139 testing::SomeMeasurements measurements;
140
141 auto preintegrated = [&](const Vector3& a, const Vector3& w) {
142 PreintegratedImuMeasurements pim(p, Bias(a, w));
143 testing::integrateMeasurements(measurements, pim: &pim);
144 return pim.preintegrated();
145 };
146
147 // Actual pre-integrated values
148 PreintegratedCombinedMeasurementsT<TangentPreintegration> pim(p);
149 testing::integrateMeasurements(measurements, pim: &pim);
150
151 EXPECT(assert_equal(numericalDerivative21<Vector9, Vector3, Vector3>(preintegrated, Z_3x1, Z_3x1),
152 pim.preintegrated_H_biasAcc()));
153 EXPECT(assert_equal(numericalDerivative22<Vector9, Vector3, Vector3>(preintegrated, Z_3x1, Z_3x1),
154 pim.preintegrated_H_biasOmega(), 1e-3));
155}

158TEST_PIM(CombinedImuFactor, PredictPositionAndVelocity) {
159 const Bias bias(Vector3(0, 0.1, 0), Vector3(0, 0.1, 0)); // Biases (acc, rot)
160
161 auto p = combined::Params();
162
163 // Measurements
164 const Vector3 measuredOmega(0, 0.1, 0); // M_PI/10.0+0.3;
165 const Vector3 measuredAcc(0, 1.1, -kGravity);
166 const double deltaT = 0.01;
167
168 CombinedPIM pim(p, bias);
169
170 for (int i = 0; i < 100; ++i)
171 pim.integrateMeasurement(measuredAcc, measuredOmega, deltaT);
172
173 // Create factor
174 const noiseModel::Gaussian::shared_ptr combinedmodel =
175 noiseModel::Gaussian::Covariance(covariance: pim.preintMeasCov());
176 const CombinedImuFactorT<CombinedPIM> Combinedfactor(X(j: 1), V(j: 1), X(j: 2), V(j: 2), B(j: 1), B(j: 2), pim);
177
178 // Predict
179 const NavState actual = pim.predict(NavState(), bias);
180 const Pose3 expectedPose(Rot3(), Point3(0, 0.5, 0));
181 const Vector3 expectedVelocity(0, 1, 0);
182 EXPECT(assert_equal(expectedPose, actual.pose()));
183 EXPECT(assert_equal(Vector(expectedVelocity), Vector(actual.velocity())));
184}

250// Test that the covariance values for the ImuFactor and the CombinedImuFactor
251// (top-left 9x9) are the same
252TEST_PIM(CombinedImuFactor, SameCovariance) {
253 // IMU measurements and time delta
254 Vector3 accMeas(0.1577, -0.8251, 9.6111);
255 Vector3 omegaMeas(-0.0210, 0.0311, 0.0145);
256 double deltaT = 0.01;
257
258 // Assume zero bias
259 imuBias::ConstantBias currentBias;
260
261 // Define params for ImuFactor
262 auto params = PreintegrationParams::MakeSharedU();
263 params->setAccelerometerCovariance(pow(x: 0.01, y: 2) * I_3x3);
264 params->setGyroscopeCovariance(pow(x: 1.75e-4, y: 2) * I_3x3);
265 params->setIntegrationCovariance(pow(x: 0, y: 2) * I_3x3);
266 params->setOmegaCoriolis(Vector3::Zero());
267
268 // The IMU preintegration object for ImuFactor
269 PIM pim(params, currentBias);
270 pim.integrateMeasurement(accMeas, omegaMeas, deltaT);
271
272 // Define params for CombinedImuFactor
273 auto combined_params = PreintegrationCombinedParams::MakeSharedU();
274 combined_params->setAccelerometerCovariance(pow(x: 0.01, y: 2) * I_3x3);
275 combined_params->setGyroscopeCovariance(pow(x: 1.75e-4, y: 2) * I_3x3);
276 // Set bias integration covariance explicitly to zero
277 combined_params->setIntegrationCovariance(Z_3x3);
278 combined_params->setOmegaCoriolis(Z_3x1);
279
280 // The IMU preintegration object for CombinedImuFactor
281 CombinedPIM cpim(combined_params, currentBias);
282 cpim.integrateMeasurement(accMeas, omegaMeas, deltaT);
283
284 // Assert if the noise covariance
285 EXPECT(assert_equal(pim.preintMeasCov(),
286 cpim.preintMeasCov().block(0, 0, 9, 9)));
287}

290// Runners currently still use the integration type based on the compile flag.
291// So we use the default CombinedScenarioRunner below.
292TEST(CombinedImuFactor, Accelerating) {
293 const double a = 0.2, v = 50;
294
295 // Set up body pointing towards y axis, and start at 10,20,0 with velocity
296 // going in X The body itself has Z axis pointing down
297 const Rot3 nRb(Point3(0, 1, 0), Point3(1, 0, 0), Point3(0, 0, -1));
298 const Point3 initial_position(10, 20, 0);
299 const Vector3 initial_velocity(v, 0, 0);
300
301 const AcceleratingScenario scenario(nRb, initial_position, initial_velocity,
302 Vector3(a, 0, 0));
303
304 const double T = 3.0; // seconds
305
306 CombinedScenarioRunner runner(scenario, combined::Params(), T / 10);
307
308 PreintegratedCombinedMeasurements pim = runner.integrate(T);
309 EXPECT(assert_equal(scenario.pose(T), runner.predict(pim).pose(), 1e-9));
310
311 auto estimatedCov = runner.estimateCovariance(T, N: 100);
312 Eigen::Matrix<double, 15, 15> expected = pim.preintMeasCov();
313 EXPECT(assert_equal(estimatedCov, expected, 0.1));
314}
