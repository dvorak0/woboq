19#include <gtsam/geometry/Rot3.h>
20#include <gtsam/linear/NoiseModel.h>
21#include <gtsam/navigation/NavState.h>
22#include <gtsam/navigation/expressions.h>
23#include <gtsam/nonlinear/Values.h>
24#include <gtsam/slam/expressions.h>
25
26#include <CppUnitLite/TestHarness.h>
27
28using namespace gtsam;
29
30// A NavState unknown expression wXb with key 42
31Expression<NavState> wXb_(42);
32
33/* ************************************************************************* */
34// Example: absolute position measurement
35TEST(Expressions, Position) {
36 auto absolutePosition_ = position(X: wXb_);
37
38 // Test with some values
39 Values values;
40 values.insert(j: 42, val: NavState(Rot3(), Point3(1, 2, 3), Velocity3(4, 5, 6)));
41 EXPECT(assert_equal(Point3(1, 2, 3), absolutePosition_.value(values)));
42}
43
44/* ************************************************************************* */
45// Example: velocity measurement in body frame
46TEST(Expressions, Velocity) {
47 // We want to predict h(wXb) = velocity in body frame
48 // h(wXb) = bRw(wXb) * wV(wXb)
49 auto bodyVelocity_ = unrotate(x: attitude(X: wXb_), p: velocity(X: wXb_));
50
51 // Test with some values
52 Values values;
53 values.insert(j: 42, val: NavState(Rot3(), Point3(1, 2, 3), Velocity3(4, 5, 6)));
54 EXPECT(assert_equal(Velocity3(4, 5, 6), bodyVelocity_.value(values)));
55}
56
57/* ************************************************************************* */
58int main() {
59 TestResult tr;
60 return TestRegistry::runAllTests(result&: tr);
61}
