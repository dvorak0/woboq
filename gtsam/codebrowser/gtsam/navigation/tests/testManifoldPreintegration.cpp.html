31TEST(ManifoldPreintegration, BiasCorrectionJacobians) {
32 testing::SomeMeasurements measurements;
33
34 std::function<Rot3(const Vector3&, const Vector3&)> deltaRij =
35 [&](const Vector3& a, const Vector3& w) {
36 ManifoldPreintegration pim(testing::Params(), Bias(a, w));
37 testing::integrateMeasurements(measurements, pim: &pim);
38 return pim.deltaRij();
39 };
40
41 std::function<Point3(const Vector3&, const Vector3&)> deltaPij =
42 [&](const Vector3& a, const Vector3& w) {
43 ManifoldPreintegration pim(testing::Params(), Bias(a, w));
44 testing::integrateMeasurements(measurements, pim: &pim);
45 return pim.deltaPij();
46 };
47
48 std::function<Vector3(const Vector3&, const Vector3&)> deltaVij =
49 [&](const Vector3& a, const Vector3& w) {
50 ManifoldPreintegration pim(testing::Params(), Bias(a, w));
51 testing::integrateMeasurements(measurements, pim: &pim);
52 return pim.deltaVij();
53 };
54
55 // Actual pre-integrated values
56 ManifoldPreintegration pim(testing::Params());
57 testing::integrateMeasurements(measurements, pim: &pim);
58
59 EXPECT(
60 assert_equal(numericalDerivative21(deltaRij, kZero, kZero),
61 Matrix3(Z_3x3)));
62 EXPECT(
63 assert_equal(numericalDerivative22(deltaRij, kZero, kZero),
64 pim.delRdelBiasOmega(), 1e-3));
65
66 EXPECT(
67 assert_equal(numericalDerivative21(deltaPij, kZero, kZero),
68 pim.delPdelBiasAcc()));
69 EXPECT(
70 assert_equal(numericalDerivative22(deltaPij, kZero, kZero),
71 pim.delPdelBiasOmega(), 1e-3));
72
73 EXPECT(
74 assert_equal(numericalDerivative21(deltaVij, kZero, kZero),
75 pim.delVdelBiasAcc()));
76 EXPECT(
77 assert_equal(numericalDerivative22(deltaVij, kZero, kZero),
78 pim.delVdelBiasOmega(), 1e-3));
79}

82TEST(ManifoldPreintegration, computeError) {
83 ManifoldPreintegration pim(testing::Params());
84 NavState x1, x2;
85 imuBias::ConstantBias bias;
86 Matrix9 aH1, aH2;
87 Matrix96 aH3;
88 pim.computeError(state_i: x1, state_j: x2, bias_i: bias, H1: aH1, H2: aH2, H3: aH3);
89 std::function<Vector9(const NavState&, const NavState&,
90 const imuBias::ConstantBias&)>
91 f = std::bind(f: &ManifoldPreintegration::computeError, args&: pim,
92 args: std::placeholders::_1, args: std::placeholders::_2,
93 args: std::placeholders::_3, args: nullptr, args: nullptr,
94 args: nullptr);
95 // NOTE(frank): tolerance of 1e-3 on H1 because approximate away from 0
96 EXPECT(assert_equal(numericalDerivative31(f, x1, x2, bias), aH1, 1e-9));
97 EXPECT(assert_equal(numericalDerivative32(f, x1, x2, bias), aH2, 1e-9));
98 EXPECT(assert_equal(numericalDerivative33(f, x1, x2, bias), aH3, 1e-9));
99}

102TEST(ManifoldPreintegration, CompareWithPreintegratedRotation) {
103 // Create a PreintegratedRotation object
104 auto p = std::make_shared<PreintegratedRotationParams>();
105 PreintegratedRotation pim(p);
106
107 // Integrate a single measurement
108 const double omega = 0.1;
109 const Vector3 trueOmega(omega, 0, 0);
110 const Vector3 biasOmega(1, 2, 3);
111 const Vector3 measuredOmega = trueOmega + biasOmega;
112 const double deltaT = 0.5;
113
114 // Check the accumulated rotation.
115 Rot3 expected = Rot3::Roll(t: omega * deltaT);
116 const Vector biasOmegaHat = biasOmega;
117 pim.integrateGyroMeasurement(measuredOmega, biasHat: biasOmegaHat, deltaT);
118 EXPECT(assert_equal(expected, pim.deltaRij(), 1e-9));
119
120 // Now do the same for a ManifoldPreintegration object
121 imuBias::ConstantBias biasHat {Z_3x1, biasOmega};
122 ManifoldPreintegration manifoldPim(testing::Params(), biasHat);
123 manifoldPim.integrateMeasurement(measuredAcc: Z_3x1, measuredOmega, dt: deltaT);
124 EXPECT(assert_equal(expected, manifoldPim.deltaRij(), 1e-9));
125
126 // Check their internal Jacobians are the same:
127 EXPECT(assert_equal(pim.delRdelBiasOmega(), manifoldPim.delRdelBiasOmega()));
128
129 // Let's check the derivatives a delta away from the bias hat
130 const double delta = 0.05;
131 const Vector3 biasOmegaIncr(delta, 0, 0);
132 imuBias::ConstantBias bias_i {Z_3x1, biasOmegaHat + biasOmegaIncr};
133
134 // Check PreintegratedRotation::biascorrectedDeltaRij.
135 // Note that biascorrectedDeltaRij expects the biasHat to be subtracted already
136 Matrix3 H;
137 Rot3 corrected = pim.biascorrectedDeltaRij(biasOmegaIncr, H);
138 EQUALITY(Vector3(-deltaT * delta, 0, 0), expected.logmap(corrected));
139 const Rot3 expected2 = Rot3::Roll(t: (omega - delta) * deltaT);
140 EXPECT(assert_equal(expected2, corrected, 1e-9));
141
142 // Check ManifoldPreintegration::biasCorrectedDelta.
143 // Note that, confusingly, biasCorrectedDelta will subtract biasHat inside
144 Matrix96 H2;
145 Vector9 biasCorrected = manifoldPim.biasCorrectedDelta(bias_i, H: H2);
146 Vector9 expected3;
147 expected3 << Rot3::Logmap(R: expected2), Z_3x1, Z_3x1;
148 EXPECT(assert_equal(expected3, biasCorrected, 1e-9));
149
150 // Check that this one is sane:
151 auto g = [&](const Vector3& b) {
152 return manifoldPim.biasCorrectedDelta(bias_i: {Z_3x1, b}, H: {});
153 };
154 EXPECT(assert_equal<Matrix>(numericalDerivative11<Vector9, Vector3>(g, bias_i.gyroscope()),
155 H2.rightCols<3>()));
156}
