18#include <gtsam/base/numericalDerivative.h>

25using namespace gtsam;

27TEST(LeftLinearEKF, WPhiU_matches_IMU_dynamics_with_gravity) {
28 const double dt = 1e-2; // 10 ms
29 const Vector3 n_gravity(0, 0, -9.81);
30 const Vector3 gyro(0.01, -0.02, 0.03); // rad/s (body)
31 const Vector3 accel(0.10, -0.05, 0.20); // m/s^2 (body specific force)
32
33 // Initial state X0 = (R, p, v)
34 const Rot3 R0 = Rot3::RzRyRx(x: 0.10, y: -0.20, z: 0.05);
35 const Point3 p0(1.0, 2.0, 3.0);
36 const Vector3 v0(0.5, -0.3, 0.8);
37 const NavState X0(R0, p0, v0);
38
39 // Build W (gravity-only left composition, world-frame increments)
40 const Point3 pW = 0.5 * n_gravity * dt * dt;
41 const Vector3 vW = n_gravity * dt;
42 const NavState W(Rot3(), pW, vW);
43
44 // Î¦ functor: velocity acts on position
45 NavState::AutonomousFlow phi{.dt: dt};
46
47 // Build U from raw IMU (no gravity): body-frame increments
48 const Rot3 dR = Rot3::Expmap(v: gyro * dt);
49 const Vector3 dp_body = accel * (0.5 * dt * dt);
50 const Vector3 dv_body = accel * dt;
51 const NavState U(dR, dp_body, dv_body);
52
53 // Run LeftLinearEKF predict with Q = 0
54 NavState::Jacobian A;
55 using EKF = LeftLinearEKF<NavState>;
56 NavState Xp = EKF::Dynamics(W, phi, X: X0, U, A);
57
58 // Closed-form expected result of NavState IMU dynamics with gravity
59 const Rot3 R_expected = R0 * dR;
60 // v + R a dt + g dt:
61 const Vector3 v_expected = v0 + R0.rotate(p: dv_body) + n_gravity * dt;
62 // p + v dt + R(0.5 a dt^2) + 0.5 g dt^2:
63 const Point3 p_expected = pW + p0 + v0 * dt + R0.rotate(p: dp_body);
64 const NavState X_expected(R_expected, p_expected, v_expected);
65 CHECK(assert_equal(X_expected, Xp));
66
67 // Check A against numerical derivative
68
69 auto numericalA = numericalDerivative11<NavState, NavState>(
70 h: [&](const NavState& X) { return EKF::Dynamics(W, phi, X, U, A: {}); }, x: X0);
71 CHECK(assert_equal(numericalA, A, 1e-9));
72
73 // Initialize filter
74 EKF::Covariance P0 = I_9x9;
75 EKF ekf(X0, P0);
76
77 // Run LeftLinearEKF predict with Q = I
78 EKF::Covariance Q = Z_9x9;
79 ekf.predict(W, phi, U, Q);
80 CHECK(assert_equal(X_expected, ekf.state()));
81 CHECK(assert_equal<NavState::Jacobian>(A * A.transpose(), ekf.covariance()));
82}
83
84int main() {
85 TestResult tr;
86 return TestRegistry::runAllTests(result&: tr);
87}
