48TEST(NavStateImuEKF, DefaultProcessNoiseFromParams) {
49 using namespace nontrivial_navstate_example;
50
51 // GIVEN params with specific covariances
52 auto params = PreintegrationParams::MakeSharedU(g: 9.81);
53 Matrix3 Cg = (Matrix3() << 0.01, 0, 0, 0, 0.02, 0, 0, 0, 0.03).finished();
54 Matrix3 Ci = (Matrix3() << 0.001, 0, 0, 0, 0.002, 0, 0, 0, 0.003).finished();
55 Matrix3 Ca = (Matrix3() << 0.1, 0, 0, 0, 0.2, 0, 0, 0, 0.3).finished();
56 params->setGyroscopeCovariance(Cg);
57 params->setIntegrationCovariance(Ci);
58 params->setAccelerometerCovariance(Ca);
59
60 Matrix9 P0 = I_9x9 * 0.01;
61 NavStateImuEKF ekf(X0, P0, params);
62
63 Matrix9 Q = Matrix9::Zero();
64 Q.block<3, 3>(startRow: 0, startCol: 0) = Cg;
65 Q.block<3, 3>(startRow: 3, startCol: 3) = Ci;
66 Q.block<3, 3>(startRow: 6, startCol: 6) = Ca;
67 EXPECT(assert_equal(Q, ekf.processNoise(), 1e-12));
68}

88TEST(NavStateImuEKF, PredictMatchesExplicitIntegration) {
89 using namespace nontrivial_navstate_example;
90 double dt = 10;
91
92 // Explicit integration from paper
93 const Vector3 phi_b = omega_b * dt;
94 const so3::DexpFunctor local(phi_b);
95 const Matrix3 J_l = local.Jacobian().left();
96 const Matrix3 N_l = local.Gamma().left();
97
98 const NavState U{Rot3::Expmap(v: phi_b), // R_new
99 N_l * f_b * dt * dt, // p_new
100 J_l * f_b * dt}; // v_new
101
102 // Check against static IMU function:
103 const NavState U_static = NavStateImuEKF::Imu(omega_b, f_b, dt);
104 EXPECT(assert_equal(U, U_static, 1e-9));
105
106 // Check against Gal3::Expmap
107 Vector10 xi;
108 xi << omega_b, f_b, Vector3::Zero(), 1.0;
109 const Gal3 T = Gal3::Expmap(xi: xi * dt);
110 const NavState U_gal3{T.rotation(), T.position(), T.velocity()};
111 EXPECT(assert_equal(U_gal3, U, 1e-9));
112
113 // Explicit integration, combined with gravity and v0 * dt boost
114 const Vector3& g_n = params->n_gravity;
115 const Rot3 R_new = R0 * U.attitude();
116 const Vector3 v_new = v0 + g_n * dt + R0 * U.velocity();
117 const Point3 p_new = p0 + g_n * (0.5 * dt * dt) + v0 * dt + R0 * U.position();
118 const NavState X_explicit(R_new, p_new, v_new);
119
120 // Increment-based integration should match exactly
121 auto params = PreintegrationParams::MakeSharedU(g: 9.81);
122 NavStateImuEKF ekf(X0, I_9x9 * 1e-3, params);
123 NavState X_next = NavStateImuEKF::Dynamics(g_n, X: X0, omega_b, f_b, dt);
124 EXPECT(assert_equal(X_explicit, X_next, 1e-12));
125}

128// Check Jacobian for world-position measurement h(X)=position(X).
129TEST(NavStateImuEKF, PositionMeasurementJacobian) {
130 using namespace nontrivial_navstate_example;
131
132 // GIVEN a nontrivial state X0
133 const NavState& X = X0;
134
135 // Analytic Jacobian H = dh/d local(X) for h(X)=position(X)
136 // With left-invariant chart and NavState compose, dp is in body frame:
137 // p' = p + R * dp => H = [0, R, 0]
138 Matrix39 H;
139 H.setZero();
140 H.block<3, 3>(startRow: 0, startCol: 3) = X.attitude().matrix();
141
142 // Numerical Jacobian via central differencing
143 std::function<Point3(const NavState&)> h = [](const NavState& Xq) {
144 return Xq.position();
145 };
146 Matrix39 expected = numericalDerivative11<Point3, NavState>(h, x: X);
147
148 EXPECT(assert_equal(expected, H, 1e-6));
149}

152// Sanity-check a single position update using updateWithVector.
153// Verifies delta_xi = K * innovation and covariance reduction in pos block.
154TEST(NavStateImuEKF, PositionUpdateSanity) {
155 using namespace nontrivial_navstate_example;
156
157 // GIVEN an EKF with diagonal covariance (no cross-terms)
158 Matrix9 P0 = Matrix9::Zero();
159 P0.block<3, 3>(startRow: 0, startCol: 0) = I_3x3 * 1e-3; // rot
160 P0.block<3, 3>(startRow: 3, startCol: 3) = I_3x3 * 1.0; // pos
161 P0.block<3, 3>(startRow: 6, startCol: 6) = I_3x3 * 0.5; // vel
162 auto params = PreintegrationParams::MakeSharedU(g: 9.81);
163 NavStateImuEKF ekf(X0, P0, params);
164
165 // BEFORE update: capture state and covariance
166 const NavState X_before = ekf.state();
167 const Matrix9 P_prior = ekf.covariance();
168
169 // Position measurement: z = p + d
170 const Vector3 d(1.0, -2.0, 0.5);
171 const Point3 p_true =
172 X_before.position() + d; // pretend ground truth is offset
173 const Vector3 z = Vector3(p_true.x(), p_true.y(), p_true.z());
174
175 // Predicted measurement and Jacobian H = [0 R 0]
176 const Vector3 prediction =
177 Vector3(X_before.position().x(), X_before.position().y(),
178 X_before.position().z());
179 Matrix39 H;
180 H.setZero();
181 const Matrix3 Rworld = X_before.attitude().matrix();
182 H.block<3, 3>(startRow: 0, startCol: 3) = Rworld;
183
184 // Reasonable measurement noise
185 const double sigma = 0.1; // meters
186 Matrix3 Rmeas = I_3x3 * (sigma * sigma);
187
188 // Manually compute K and delta_xi expected
189 const Matrix3 S = H * P_prior * H.transpose() + Rmeas;
190 const Matrix93 K = P_prior * H.transpose() * S.inverse();
191 const Vector3 innovation =
192 z - prediction; // Vector measurement: y = z - prediction
193 const Vector9 delta_expected = K * innovation;
194
195 // WHEN performing the EKF update
196 ekf.updateWithVector(prediction, H, z, R: Rmeas);
197
198 // THEN: delta_xi applied equals expected (within tolerance)
199 const NavState& X_after = ekf.state();
200 const Vector9 delta_ekf = X_before.localCoordinates(g: X_after);
201 EXPECT(assert_equal(delta_expected, delta_ekf, 1e-9));
202
203 // AND: position moved toward z in world frame approximately by R*dp
204 const Vector3 dp_body = delta_expected.segment<3>(start: 3);
205 const Point3 p_expected = X_before.position() + Point3(Rworld * dp_body);
206 EXPECT(assert_equal(p_expected, X_after.position(), 1e-9));
207
208 // AND: covariance position block decreased
209 const Matrix9 P_post = ekf.covariance();
210 const double trace_pos_prior = P_prior.block<3, 3>(startRow: 3, startCol: 3).trace();
211 const double trace_pos_post = P_post.block<3, 3>(startRow: 3, startCol: 3).trace();
212 CHECK(trace_pos_post < trace_pos_prior);
213
214 // Compare EKF position update to solving an equivalent GaussianFactorGraph.
215 using symbol_shorthand::X;
216 const Key key = X(j: 0);
217 GaussianFactorGraph gfg;
218
219 // Prior as JacobianFactor using full Gaussian covariance:
220 // || I * x - 0||_{P^{-1}}
221 Vector b = Vector::Zero(size: 9);
222 gfg.add(key1: key, A1: I_9x9, b, model: noiseModel::Diagonal::Variances(variances: P_prior.diagonal()));
223
224 // Measurement as JacobianFactor using full Gaussian covariance:
225 // || H * x -innovation ||_{R^{-1}}
226 Vector b_meas = innovation; // 3x1
227 gfg.add(key1: key, A1: H, b: b_meas, model: noiseModel::Isotropic::Sigma(dim: 3, sigma));
228
229 // Solve for MAP delta
230 VectorValues delta_map = gfg.optimize();
231 const Vector delta_graph = delta_map.at(j: key);
232
233 // THEN: Graph solution equals EKF correction
234 EXPECT(assert_equal(delta_graph, delta_ekf, 1e-9));
235}
