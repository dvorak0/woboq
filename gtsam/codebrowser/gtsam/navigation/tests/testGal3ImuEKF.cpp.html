44Rot3 R0 = Rot3::RzRyRx(x: 0.1, y: -0.2, z: 0.3);

52Vector3 f_b(0.5, -0.3, 0.2);

57TEST(Gal3ImuEKF, DefaultProcessNoiseFromParams) {
58 using namespace nontrivial_gal3_example;
59
60 // GIVEN params with specific covariances
61 auto params = PreintegrationParams::MakeSharedU(g: 9.81);
62 Matrix3 Cg = (Matrix3() << 0.01, 0, 0, 0, 0.02, 0, 0, 0, 0.03).finished();
63 Matrix3 Ci = (Matrix3() << 0.001, 0, 0, 0, 0.002, 0, 0, 0, 0.003).finished();
64 Matrix3 Ca = (Matrix3() << 0.1, 0, 0, 0, 0.2, 0, 0, 0, 0.3).finished();
65 params->setGyroscopeCovariance(Cg);
66 params->setIntegrationCovariance(Ci);
67 params->setAccelerometerCovariance(Ca);
68
69 Matrix10 P0 = I_10x10 * 0.01;
70 Gal3ImuEKF ekf(X0, P0, params);
71
72 Matrix10 Q = Matrix10::Zero();
73 Q.block<3, 3>(startRow: 0, startCol: 0) = Cg;
74 Q.block<3, 3>(startRow: 6, startCol: 6) = Ci;
75 Q.block<3, 3>(startRow: 3, startCol: 3) = Ca;
76 EXPECT(assert_equal(Q, ekf.processNoise(), 1e-12));
77}

99TEST(Gal3ImuEKF, PredictMatchesExplicitIntegration) {
100 using namespace nontrivial_gal3_example;
101 double dt = 10;
102
103 // Explicit integration from paper
104 const Vector3 phi_b = omega_b * dt;
105 const so3::DexpFunctor local(phi_b);
106 const Matrix3 J_l = local.Jacobian().left();
107 const Matrix3 N_l = local.Gamma().left();
108
109 const Gal3 U{Rot3::Expmap(v: phi_b), N_l * f_b * dt * dt, J_l * f_b * dt, dt};
110
111 // Check against static IMU function:
112 const Gal3 U_static = Gal3ImuEKF::Imu(omega_b, f_b, dt);
113 EXPECT(assert_equal(U, U_static, 1e-9));
114
115 // Check against Gal3::Expmap
116 Vector10 xi;
117 xi << omega_b, f_b, Vector3::Zero(), 1.0;
118 const Gal3 U_gal3 = Gal3::Expmap(xi: xi * dt);
119 EXPECT(assert_equal(U_gal3, U, 1e-9));
120
121 // Explicit integration, combined with gravity and v0 * dt boost
122 const Vector3& g_n = params->n_gravity;
123 const Rot3 R_new = R0 * U.attitude();
124 const Vector3 v_new = v0 + g_n * dt + R0 * U.velocity();
125 const Point3 p_new = p0 + g_n * (0.5 * dt * dt) + v0 * dt + R0 * U.position();
126 const Gal3 X_explicit(R_new, p_new, v_new, t0 + dt);
127
128 // Increment-based integration should match exactly
129 auto params = PreintegrationParams::MakeSharedU(g: 9.81);
130 Gal3ImuEKF ekf(X0, I_10x10 * 1e-3, params);
131 Gal3 X_next = Gal3ImuEKF::Dynamics(g_n, X: X0, omega_b, f_b, dt);
132 EXPECT(assert_equal(X_explicit, X_next, 1e-12));
133}

136// Test predict with a scenario that can be integrated exactly.
137TEST(Gal3ImuEKF, PredictMatchesScenario) {
138 // --- Scenario: camera orbiting a fixed point ---
139 // measurements in the body/sensor frame, assumed FRD
140 double radius = 30.0;
141 double angular_velocity = M_PI; // rad/sec (half-turn per second)
142 Vector3 w_b(0, 0, angular_velocity); // body yaw rate
143 Vector3 v_n(radius * angular_velocity, 0, 0); // world-frame velocity
144 ConstantTwistScenario scenario(w_b, v_n);
145
146 // using NED coordinates as the navigation frame
147 auto params = PreintegrationParams::MakeSharedD(g: 10); // use 10 m/s^2 gravity
148
149 // EKF setup with initial state and covariance, FRD frame aligned with NED
150 // (i.e., looking north)
151 const Gal3 X0 = scenario.gal3(t: 0.0); // Get state at t=0
152 Matrix P0 = Matrix::Identity(rows: 10, cols: 10) * 0.1; // std ~ 0.316 on each component
153 P0(9, 9) = 1e-6; // small variance for time
154
155 ScenarioRunner runner(scenario, params, 1.0);
156 Gal3ImuEKF ekf(X0, P0, params);
157
158 // Predict every 0.5 second for 2 seconds
159 double T = 2.0, dt = 0.5;
160 size_t N = T / dt;
161 Vector3 omega_b = runner.actualAngularVelocity(t: 0.0);
162 Vector3 f_b = runner.actualSpecificForce(t: 0.0);
163 for (size_t i = 0; i < N; i++) {
164 ekf.predict(omega_b, f_b, dt);
165 // Check that predicted state matches ground truth
166 double t = dt * (i + 1);
167 EXPECT(assert_equal(scenario.gal3(t), ekf.state(), 1e-9));
168 }
169}

172// Check Jacobian for world-position measurement h(X)=position(X).
173TEST(Gal3ImuEKF, PositionMeasurementJacobian) {
174 using namespace nontrivial_gal3_example;
175
176 // GIVEN a nontrivial state X0
177 const Gal3& X = X0;
178
179 // Analytic Jacobian H = dh/d local(X) for h(X)=position(X)
180 // With left-invariant chart and NavState compose, dp is in body frame:
181 // p' = p + R * dp => H = [0, R, 0]
182 Matrix310 H;
183 H.setZero();
184 H.block<3, 3>(startRow: 0, startCol: 6) = X.attitude().matrix();
185 H.block<3, 1>(startRow: 0, startCol: 9) = X.velocity();
186
187 // Numerical Jacobian via central differencing
188 std::function<Point3(const Gal3&)> h = [](const Gal3& Xq) {
189 return Xq.position();
190 };
191 Matrix310 expected = numericalDerivative11<Point3, Gal3>(h, x: X);
192
193 EXPECT(assert_equal(expected, H, 1e-6));
194}

196// Sanity-check a single position update using updateWithVector. Verifies
197// delta_xi = K * innovation and covariance reduction in pos block.
198TEST(Gal3ImuEKF, PositionUpdateSanity) {
199 using namespace nontrivial_gal3_example;
200
201 // GIVEN an EKF with diagonal covariance (no cross-terms)
202 Matrix10 P0 = Matrix10::Zero();
203 P0.block<3, 3>(startRow: 0, startCol: 0) = I_3x3 * 1e-3; // rot
204 P0.block<3, 3>(startRow: 3, startCol: 3) = I_3x3 * 0.5; // vel
205 P0.block<3, 3>(startRow: 6, startCol: 6) = I_3x3 * 1.0; // pos
206 P0(9, 9) = 1e-6;
207 auto params = PreintegrationParams::MakeSharedU(g: 9.81);
208 Gal3ImuEKF ekf(X0, P0, params);
209
210 // BEFORE update: capture state and covariance
211 const Gal3 X_before = ekf.state();
212 const Matrix10 P_prior = ekf.covariance();
213
214 // Position measurement: z = p + d
215 const Vector3 d(1.0, -2.0, 0.5);
216 const Point3 p_true =
217 X_before.position() + d; // pretend ground truth is offset
218 const Vector3 z = Vector3(p_true.x(), p_true.y(), p_true.z());
219
220 // Predicted measurement and Jacobian H = [0 R 0]
221 const Vector3 prediction =
222 Vector3(X_before.position().x(), X_before.position().y(),
223 X_before.position().z());
224 Matrix310 H;
225 H.setZero();
226 const Matrix3 Rworld = X_before.attitude().matrix();
227 H.block<3, 3>(startRow: 0, startCol: 6) = Rworld;
228 H.block<3, 1>(startRow: 0, startCol: 9) = X_before.velocity();
229
230 // Reasonable measurement noise
231 const double sigma = 0.1; // meters
232 Matrix3 Rmeas = I_3x3 * (sigma * sigma);
233
234 // Manually compute K and delta_xi expected
235 const Matrix3 S = H * P_prior * H.transpose() + Rmeas;
236 const Matrix10x3 K = P_prior * H.transpose() * S.inverse();
237 const Vector3 innovation =
238 z - prediction; // Vector measurement: y = z - prediction
239 const Vector10 delta_expected = K * innovation;
240
241 // WHEN performing the EKF update
242 ekf.updateWithVector(prediction, H, z, R: Rmeas);
243
244 // THEN: delta_xi applied equals expected (within tolerance)
245 const Gal3& X_after = ekf.state();
246 const Vector10 delta_ekf = X_before.localCoordinates(g: X_after);
247 EXPECT(assert_equal(delta_expected, delta_ekf, 1e-9));
248 //
249 // AND: position moved toward z in world frame approximately by R*dp
250 const Vector3 dp_body = delta_expected.segment<3>(start: 6);
251 const double dt_correction = delta_expected(9);
252 const Point3 p_expected = X_before.position() + Point3(Rworld * dp_body) +
253 X_before.velocity() * dt_correction;
254 // const Point3 p_expected = X_before.position() + Point3(Rworld * dp_body);
255
256 EXPECT(assert_equal(p_expected, X_after.position(), 1e-9));
257
258 // // AND: covariance position block decreased
259 const Matrix10 P_post = ekf.covariance();
260 const double trace_pos_prior = P_prior.block<3, 3>(startRow: 6, startCol: 6).trace();
261 const double trace_pos_post = P_post.block<3, 3>(startRow: 6, startCol: 6).trace();
262 CHECK(trace_pos_post < trace_pos_prior);
263 //
264 // Compare EKF position update to solving an equivalent GaussianFactorGraph.
265 using symbol_shorthand::X;
266 const Key key = X(j: 0);
267 GaussianFactorGraph gfg;
268
269 // Prior as JacobianFactor using full Gaussian covariance:
270 // || I * x - 0||_{P^{-1}}
271 Vector b = Vector::Zero(size: 10);
272 gfg.add(key1: key, A1: I_10x10, b, model: noiseModel::Diagonal::Variances(variances: P_prior.diagonal()));
273
274 // Measurement as JacobianFactor using full Gaussian covariance:
275 // || H * x -innovation ||_{R^{-1}}
276 Vector b_meas = innovation; // 3x1
277 gfg.add(key1: key, A1: H, b: b_meas, model: noiseModel::Isotropic::Sigma(dim: 3, sigma));
278
279 // Solve for MAP delta
280 VectorValues delta_map = gfg.optimize();
281 const Vector delta_graph = delta_map.at(j: key);
282
283 // THEN: Graph solution equals EKF correction
284 EXPECT(assert_equal(delta_graph, delta_ekf, 1e-9));
285}

288// Verify dynamics W and U.
289TEST(Gal3ImuEKF, PredictWithWandU) {
290 using namespace nontrivial_gal3_example;
291
292 double dt = 0.1;
293
294 const Gal3 W = Gal3ImuEKF::CompensatedGravity(g_n: params->n_gravity, dt, t_k: t0);
295 const Gal3 U = Gal3ImuEKF::Imu(omega_b, f_b, dt);
296
297 // Compute dynamics
298 Matrix10 A_ekf;
299 Gal3 X_predicted = Gal3ImuEKF::Base::Dynamics(W, X: X0, U, A: A_ekf);
300
301 // New state: W*X0*U
302 Gal3 X_expected = W * X0 * U;
303 EXPECT(assert_equal(X_expected, X_predicted, 1e-12));
304
305 // Check times
306 EXPECT_DOUBLES_EQUAL(t0, X0.time(), 1e-12);
307 EXPECT_DOUBLES_EQUAL(t0 + dt, X_predicted.time(), 1e-12);
308
309 // Expected: J = Ad_U^(-1)
310 Matrix10 A_expected = U.inverse().AdjointMap();
311 EXPECT(assert_equal(A_expected, A_ekf, 1e-9));
312}

346// Verify W, U match the Exp (G,N) function
347TEST(Gal3ImuEKF, FormulationsMatchMatrixExponential) {
348 using namespace nontrivial_gal3_example;
349
350 const double dt = 10;
351 const Vector3& g = params->n_gravity;
352
353 // Create tangent vectors for G, N,
354
355 // W Matrix W
356 Gal3::TangentVector xiG = Gal3::TangentVector::Zero();
357 xiG.segment<3>(start: 3) = g;
358
359 // N is given by a dt contribution
360 Gal3::TangentVector xiN = Gal3::TangentVector::Zero();
361 xiN(9) = 1.0;
362
363 // Compare W
364 Gal3 W_expected = Gal3::Expmap(xi: (xiG - xiN) * dt);
365 Gal3 W_actual = Gal3ImuEKF::TimeZeroingGravity(g_n: g, dt);
366 EXPECT(assert_equal(W_expected, W_actual, 1e-9));
367
368 // Compare U
369 // U is given by exp((w^ - b^ + N)*dt)
370 Vector3 rho(0.0, 0.0, 0.0);
371 Gal3::TangentVector w;
372 w << omega_b, f_b, rho, 0;
373
374 Gal3 U_expected = Gal3::Expmap(xi: (w + xiN) * dt);
375 Gal3 U_actual = Gal3ImuEKF::Imu(omega_b, f_b, dt);
376 EXPECT(assert_equal(U_expected, U_actual, 1e-5));
377}

380// Check that TimeZeroingGravity and CompensatedGravity match the "correction:
381// math" formulas from the paper.
382TEST(Gal3ImuEKF, TestCorrectionFormulas) {
383 using namespace nontrivial_gal3_example;
384 const Vector3 g_n = params->n_gravity;
385 double dt = 0.1;
386 double t_k = 5.0;
387
388 // Check that TimeZeroingGravity matches the math
389 auto timeZeroingGravity = [g_n](double dt) -> Gal3 {
390 const Gal3 G = Gal3ImuEKF::Gravity(g_n, dt);
391 const Gal3 C{Rot3(), Z_3x1, Z_3x1, -dt};
392 return G * C;
393 };
394 Gal3 W_math = timeZeroingGravity(dt);
395 Gal3 W_actual = Gal3ImuEKF::TimeZeroingGravity(g_n, dt);
396 EXPECT(assert_equal(W_math, W_actual, 1e-9));
397
398 // Check that CompensatedGravity matches the math
399 auto compensatedGravity = [g_n](double dt, double t_k) -> Gal3 {
400 const Gal3 G = Gal3ImuEKF::Gravity(g_n, dt);
401 const Gal3 C{Rot3(), -t_k * G.velocity() - 2.0 * G.translation(), Z_3x1, 0};
402 return G * C;
403 };
404 Gal3 W_comp_math = compensatedGravity(dt, t_k);
405 Gal3 W_comp_actual = Gal3ImuEKF::CompensatedGravity(g_n, dt, t_k);
406 EXPECT(assert_equal(W_comp_math, W_comp_actual, 1e-9));
407}
