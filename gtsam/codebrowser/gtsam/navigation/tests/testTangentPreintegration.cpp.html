37TEST(TangentPreintegration, UpdateEstimate1) {
38 TangentPreintegration pim(testing::Params());
39 const Vector3 acc(0.1, 0.2, 10), omega(0.1, 0.2, 0.3);
40 Vector9 zeta;
41 zeta.setZero();
42 Matrix9 aH1;
43 Matrix93 aH2, aH3;
44 pim.UpdatePreintegrated(a_body: acc, w_body: omega, dt: kDt, preintegrated: zeta, A: aH1, B: aH2, C: aH3);
45 EXPECT(assert_equal(numericalDerivative31(f, zeta, acc, omega), aH1, 1e-9));
46 EXPECT(assert_equal(numericalDerivative32(f, zeta, acc, omega), aH2, 1e-9));
47 EXPECT(assert_equal(numericalDerivative33(f, zeta, acc, omega), aH3, 1e-9));
48}

51TEST(TangentPreintegration, UpdateEstimate2) {

56 Matrix9 aH1;
57 Matrix93 aH2, aH3;
58 pim.UpdatePreintegrated(a_body: acc, w_body: omega, dt: kDt, preintegrated: zeta, A: aH1, B: aH2, C: aH3);

62 EXPECT(assert_equal(numericalDerivative33(f, zeta, acc, omega), aH3, 1e-9));

66TEST(ImuFactor, BiasCorrectionJacobians) {
67 testing::SomeMeasurements measurements;
68
69 std::function<Vector9(const Vector3&, const Vector3&)> preintegrated =
70 [&](const Vector3& a, const Vector3& w) {
71 TangentPreintegration pim(testing::Params(), Bias(a, w));
72 testing::integrateMeasurements(measurements, pim: &pim);
73 return pim.preintegrated();
74 };
75
76 // Actual pre-integrated values
77 TangentPreintegration pim(testing::Params());
78 testing::integrateMeasurements(measurements, pim: &pim);
79
80 EXPECT(
81 assert_equal(numericalDerivative21(preintegrated, kZero, kZero),
82 pim.preintegrated_H_biasAcc()));
83 EXPECT(
84 assert_equal(numericalDerivative22(preintegrated, kZero, kZero),
85 pim.preintegrated_H_biasOmega(), 1e-3));
86}

89TEST(TangentPreintegration, computeError) {
90 TangentPreintegration pim(testing::Params());
91 NavState x1, x2;
92 imuBias::ConstantBias bias;
93 Matrix9 aH1, aH2;
94 Matrix96 aH3;
95 pim.computeError(state_i: x1, state_j: x2, bias_i: bias, H1: aH1, H2: aH2, H3: aH3);
96 std::function<Vector9(const NavState&, const NavState&,
97 const imuBias::ConstantBias&)>
98 f = std::bind(f: &TangentPreintegration::computeError, args&: pim,
99 args: std::placeholders::_1, args: std::placeholders::_2,
100 args: std::placeholders::_3, args: nullptr, args: nullptr,
101 args: nullptr);
102 // NOTE(frank): tolerance of 1e-3 on H1 because approximate away from 0
103 EXPECT(assert_equal(numericalDerivative31(f, x1, x2, bias), aH1, 1e-9));
104 EXPECT(assert_equal(numericalDerivative32(f, x1, x2, bias), aH2, 1e-9));
105 EXPECT(assert_equal(numericalDerivative33(f, x1, x2, bias), aH3, 1e-9));
106}

109TEST(TangentPreintegration, Compose) {
110 testing::SomeMeasurements measurements;
111 TangentPreintegration pim(testing::Params());
112 testing::integrateMeasurements(measurements, pim: &pim);
113
114 std::function<Vector9(const Vector9&, const Vector9&)> f =
115 [pim](const Vector9& zeta01, const Vector9& zeta12) {
116 return TangentPreintegration::Compose(zeta01, zeta12, deltaT12: pim.deltaTij());
117 };
118
119 // Expected merge result
120 TangentPreintegration expected_pim02(testing::Params());
121 testing::integrateMeasurements(measurements, pim: &expected_pim02);
122 testing::integrateMeasurements(measurements, pim: &expected_pim02);
123
124 // Actual result
125 Matrix9 H1, H2;
126 TangentPreintegration actual_pim02 = pim;
127 actual_pim02.mergeWith(pim, H1: &H1, H2: &H2);
128
129 const Vector9 zeta = pim.preintegrated();
130 const Vector9 actual_zeta =
131 TangentPreintegration::Compose(zeta01: zeta, zeta12: zeta, deltaT12: pim.deltaTij());
132 EXPECT(assert_equal(expected_pim02.preintegrated(), actual_zeta, 1e-7));
133 EXPECT(assert_equal(numericalDerivative21(f, zeta, zeta), H1, 1e-7));
134 EXPECT(assert_equal(numericalDerivative22(f, zeta, zeta), H2, 1e-7));
135}

138TEST(TangentPreintegration, MergedBiasDerivatives) {
139 testing::SomeMeasurements measurements;
140
141 auto f = [&](const Vector3& a, const Vector3& w) {
142 TangentPreintegration pim02(testing::Params(), Bias(a, w));
143 testing::integrateMeasurements(measurements, pim: &pim02);
144 testing::integrateMeasurements(measurements, pim: &pim02);
145 return pim02.preintegrated();
146 };
147
148 // Expected merge result
149 TangentPreintegration expected_pim02(testing::Params());
150 testing::integrateMeasurements(measurements, pim: &expected_pim02);
151 testing::integrateMeasurements(measurements, pim: &expected_pim02);
152
153 EXPECT(assert_equal(numericalDerivative21<Vector9, Vector3, Vector3>(f, Z_3x1, Z_3x1),
154 expected_pim02.preintegrated_H_biasAcc()));
155 EXPECT(assert_equal(numericalDerivative22<Vector9, Vector3, Vector3>(f, Z_3x1, Z_3x1),
156 expected_pim02.preintegrated_H_biasOmega(), 1e-7));
157}
