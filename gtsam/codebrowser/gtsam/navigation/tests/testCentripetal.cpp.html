35// For offset IMUs, acceleration is given by
36// fb + omega × (lever arm) + (omega × (omega × lever arm))
37// = fb + doublecross(w0, lever arm)
38// We then can rotate by body -> sensor frame to ensure it is in the sensor
39// measurement Angular velocity is also rotated by body -> sensor frame
40static std::pair<Vector3, Vector3> computeSensorMeasurements(
41 const Vector3& fb, const Vector3& wb, const Pose3& bTs,
42 const imuBias::ConstantBias& bias = imuBias::ConstantBias()) {
43 // Force on sensor as if in the same orientation as body
44 const Vector3 f_b_sensor = fb + gtsam::doubleCross(p: wb, q: bTs.translation());
45
46 // Find body -> sensor frame given by sRb = bRs^T
47 const Rot3 sRb = bTs.rotation().inverse();
48
49 // Rotate forces and angular velocities to sensor frame
50 const Vector3 f_sensor = sRb * f_b_sensor + bias.accelerometer();
51 const Vector3 w_sensor = sRb * wb + bias.gyroscope();
52
53 return std::make_pair(x: f_sensor, y: w_sensor);
54}

57namespace gravity {
58const double g = 9.81;
59const Vector3 g_n = Vector3(0, 0, -g);
60} // namespace gravity
61
62/* ************************************************************************* */
63/// Simple helper function that integrates a single IMU
64template <typename PIM>
65static NavState integrate(const ConstantTwistScenario& scenario,
66 const Pose3& bTs, const imuBias::ConstantBias& bias) {
67 using namespace gravity;
68
69 // Create preintegration parameters for the IMU
70 auto p = PreintegrationParams::MakeSharedU(g);
71 p->setBodyPSensor(bTs);
72
73 PIM pim(p, bias);
74
75 // Integrate for 5 seconds at 10 Hz
76 const double dt = 0.1;
77 const double T = 5.0;
78 for (int k = 0; k * dt < T; k++) {
79 const double t = k * dt;
80
81 // Force at body frame = a - g_b
82 const Vector3 fb =
83 scenario.acceleration_b(t) - scenario.rotation(t).unrotate(p: g_n);
84 const Vector3 wb = scenario.omega_b(t);
85
86 // Compute sensor measurements using the helper function
87 auto [f_sensor, w_sensor] = computeSensorMeasurements(fb, wb, bTs, bias);
88
89 // Integrate IMU
90 pim.integrateMeasurement(f_sensor, w_sensor, dt);
91 }
92
93 // Predict the final state
94 return pim.predict(NavState(), bias);
95}

98namespace car_in_circle {
99// We set up a scenario where a car is driving in a 10m circle at 10 m/s with
100// a yaw rate of 1 rad/s
101const double v = 10, w = 1;
102const Vector3 W(0, 0, w), V(v, 0, 0);
103const ConstantTwistScenario scenario(W, V);
104} // namespace car_in_circle
105
106/* ************************************************************************* */
107namespace aligned_imus {
108// Set up the three IMUs; imu0 at body frame, imu1 3m left of body frame, imu2
109// 3m right in same orientation Using FRD; right +3m, left -3m
110const double b = 3.0; // Lever arm in m
111const Pose3 bTs_imu0; // IMU at body frame
112const Pose3 bTs_imu1({}, Point3(0, -b, 0)); // IMU 1 3m left of body frame
113const Pose3 bTs_imu2({}, Point3(0, b, 0)); // IMU 2 3m right of body
114} // namespace aligned_imus
115
116/* ************************************************************************* */
117namespace rotated_imus {
118// Set up the three IMUs; imu0 at body frame, imu1 3m left of body frame, imu2
119// 3m right. Set IMU on left to be rotated on yaw by -90, IMU on right by +90
120const double b = 3.0; // Lever arm in m
121const Pose3 bTs_imu0; // IMU at body frame
122const Pose3 bTs_imu1(Rot3::Rz(t: -M_PI_2),
123 Point3(0, -b, 0)); // IMU 1 3m left of body frame
124const Pose3 bTs_imu2(Rot3::Rz(M_PI_2),
125 Point3(0, b, 0)); // IMU 2 3m right of body frame
126} // namespace rotated_imus
127
128/* ************************************************************************* */
129TEST_PIM(CentripetalCompensation, Circle3IMUsSameOrientation) {
130 using namespace aligned_imus;
131 using namespace car_in_circle;
132
133 const imuBias::ConstantBias zero; // zero bias
134
135 // Integrate imu0 to get state0
136 auto state0 = integrate<PIM>(scenario, bTs_imu0, zero);
137
138 // Integrate imu1 and imu2 should yield the same result
139 EXPECT(assert_equal(state0, integrate<PIM>(scenario, bTs_imu1, zero), 1e-6));
140 EXPECT(assert_equal(state0, integrate<PIM>(scenario, bTs_imu2, zero), 1e-6));
141}
