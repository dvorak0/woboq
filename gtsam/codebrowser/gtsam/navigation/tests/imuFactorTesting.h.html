42// Realistic MEMS white noise characteristics. Angular and velocity random walk
43// expressed in degrees respectively m/s per sqrt(hr).
44auto radians = [](double t) { return t * M_PI / 180; };
45[[maybe_unused]] static const double kGyroSigma =

47[[maybe_unused]] static const double kAccelSigma = 0.1 / 60; // 10 cm VRW

60struct ImuMeasurement {
61 ImuMeasurement(const Vector3& acc, const Vector3& gyro, double dt)
62 : acc(acc), gyro(gyro), dt(dt) {}
63 const Vector3 acc, gyro;
64 const double dt;
65};
66
67template <typename PIM>
68void integrateMeasurements(const vector<ImuMeasurement>& measurements,
69 PIM* pim) {
70 for (const auto& m : measurements)
71 pim->integrateMeasurement(m.acc, m.gyro, m.dt);
72}
73
74struct SomeMeasurements : vector<ImuMeasurement> {

89// Macro to test ImuFactor with both Manifold and Tangent preintegration
90// In the tests below the selected PreintegratedImuMeasurementsT is available
91// as `PIM`, and the combined version as `CombinedPIM`.
92#define TEST_PIM(testGroup, testName) \
93 template <class PIM, class CombinedPIM> \
94 void testGroup##testName##Helper(TestResult& result_, \
95 const std::string& name_); \
96 TEST(testGroup, testName) { \
97 using M = ManifoldPreintegration; \
98 using PM = PreintegratedImuMeasurementsT<M>; \
99 using CM = PreintegratedCombinedMeasurementsT<M>; \
100 using T = TangentPreintegration; \
101 using PT = PreintegratedImuMeasurementsT<T>; \
102 using CT = PreintegratedCombinedMeasurementsT<T>; \
103 testGroup##testName##Helper<PM, CM>(result_, this->name_); \
104 testGroup##testName##Helper<PT, CT>(result_, this->name_); \
105 } \
106 template <class PIM, class CombinedPIM> \
107 void testGroup##testName##Helper(TestResult& result_, \
108 const std::string& name_)
109} // namespace
