36//***************************************************************************
37bool BarometricFactor::equals(const NonlinearFactor& expected,
38 double tol) const {
39 const This* e = dynamic_cast<const This*>(&expected);
40 return e != nullptr && Base::equals(f: *e, tol) &&
41 traits<double>::Equals(v1: nT_, v2: e->nT_, tol);
42}

44//***************************************************************************
45Vector BarometricFactor::evaluateError(const Pose3& p, const double& bias,
46 OptionalMatrixType H,
47 OptionalMatrixType H2) const {
48 Matrix tH;
49 Vector ret = (Vector(1) << (p.translation(Hself: tH).z() + bias - nT_)).finished();
50 if (H) (*H) = tH.block<1, 6>(startRow: 2, startCol: 0);
51 if (H2) (*H2) = (Matrix(1, 1) << 1.0).finished();
52 return ret;
53}
