12/**
13 * @file ManifoldEKF.h
14 * @brief Extended Kalman Filter base class on a generic manifold M
15 *
16 * This file defines the ManifoldEKF class template for performing prediction
17 * and update steps of an Extended Kalman Filter on states residing in a
18 * differentiable manifold. It relies on the manifold's retract and
19 * localCoordinates operations.

21 * @date April 24, 2025
22 * @authors Scott Baker, Matt Kielo, Frank Dellaert
23 */

39/**
40 * @class ManifoldEKF
41 * @brief Extended Kalman Filter on a generic manifold M
42 *
43 * @tparam M Manifold type (must satisfy Manifold concept).
44 *
45 * This filter maintains a state X in the manifold M and covariance P in the
46 * tangent space at X.
47 * Prediction requires providing the predicted next state and the state
48 * transition Jacobian F. Updates apply a measurement function h and correct the
49 * state using the tangent space error.
50 *
51 * **Handling Static and Dynamic Dimensions:**
52 * The filter supports manifolds M with either a compile-time fixed dimension or
53 * a runtime dynamic dimension. This is determined by
54 * `gtsam::traits<M>::dimension`.
55 * - If `dimension` is an integer (e.g., 3, 6), it's a fixed-size manifold.
56 * - If `dimension` is `Eigen::Dynamic`, it's a dynamically-sized manifold. In
57 * this case, `gtsam::traits<M>::GetDimension(const M&)` must be available to
58 * retrieve the actual dimension at runtime. The internal protected member `n_`
59 * stores this runtime dimension. Covariance matrices (e.g., `P_`, method
60 * argument `Q`) and Jacobians (e.g., method argument `F`) are typed using
61 * `Covariance` and `Jacobian` typedefs, which are specializations of
62 * `Eigen::Matrix<double, Dim, Dim>`, where `Dim` is `traits<M>::dimension`.
63 * For dynamically-sized manifolds (`Dim == Eigen::Dynamic`), these Eigen types
64 * represent dynamically-sized matrices.
65 */
66template <typename M>
67class ManifoldEKF {
68 public:
69 /// Compile-time dimension of the manifold M.
70 static constexpr int Dim = traits<M>::dimension;
71
72 /// Tangent vector type for the manifold M.
73 using TangentVector = typename traits<M>::TangentVector;
74 /// Covariance matrix type (P, Q).
75 using Covariance = Eigen::Matrix<double, Dim, Dim>;
76 /// State transition Jacobian type (F).
77 using Jacobian = Eigen::Matrix<double, Dim, Dim>;
78
79 /**
80 * Constructor: initialize with state and covariance.
81 * @param X0 Initial state on manifold M.
82 * @param P0 Initial covariance in the tangent space at X0
83 */
84 ManifoldEKF(const M& X0, const Covariance& P0) : X_(X0) {
85 static_assert(IsManifold<M>::value,
86 "Template parameter M must be a GTSAM Manifold.");
87
88 if constexpr (Dim == Eigen::Dynamic) {
89 n_ = traits<M>::GetDimension(X0);
90 // Validate dimensions of initial covariance P0.
91 if (P0.rows() != n_ || P0.cols() != n_) {
92 throw std::invalid_argument(
93 "ManifoldEKF: Initial covariance P0 dimensions (" +
94 std::to_string(P0.rows()) + "x" + std::to_string(P0.cols()) +
95 ") do not match state's tangent space dimension (" +
96 std::to_string(val: n_) + ").");
97 }
98 } else {
99 n_ = Dim;
100 }
101
102 P_ = P0;
103 }
104
105 virtual ~ManifoldEKF() = default;
106
107 /// @return current state estimate on manifold M.
108 const M& state() const { return X_; }
109
110 /// @return current covariance estimate.
111 const Covariance& covariance() const { return P_; }
112
113 /// @return runtime dimension of the manifold.
114 int dimension() const { return n_; }
115
116 /**
117 * Basic predict step: Updates state and covariance given the predicted
118 * next state and the state transition Jacobian F.
119 * X_{k+1} = X_next
120 * P_{k+1} = F P_k F^T + Q
121 * where F = d(local(X_{k+1})) / d(local(X_k)) is the Jacobian of the
122 * state transition in local coordinates around X_k.
123 *
124 * @param X_next The predicted state at time k+1 on manifold M.
125 * @param F The state transition Jacobian.
126 * @param Q Process noise covariance matrix.
127 */
128 void predict(const M& X_next, const Jacobian& F, const Covariance& Q) {
129 if constexpr (Dim == Eigen::Dynamic) {
130 if (F.rows() != n_ || F.cols() != n_ || Q.rows() != n_ ||
131 Q.cols() != n_) {
132 throw std::invalid_argument(
133 "ManifoldEKF::predict: Dynamic F/Q dimensions must match state "
134 "dimension " +
135 std::to_string(val: n_) + ".");
136 }
137 }
138
139 X_ = X_next;
140 P_ = F * P_ * F.transpose() + Q;
141 }
142
143 /**
144 * Measurement update: Corrects the state and covariance using a
145 * pre-calculated predicted measurement and its Jacobian.
146 *
147 * @tparam Measurement Type of the measurement vector (e.g., VectorN<m>,
148 * Vector).
149 * @param prediction Predicted measurement.
150 * @param H Jacobian of the measurement function h w.r.t. local(X), H =
151 * dh/dlocal(X).
152 * @param z Observed measurement.
153 * @param R Measurement noise covariance.
154 */
155 template <typename Measurement>
156 void update(
157 const Measurement& prediction,
158 const Eigen::Matrix<double, traits<Measurement>::dimension, Dim>& H,
159 const Measurement& z,
160 const Eigen::Matrix<double, traits<Measurement>::dimension,
161 traits<Measurement>::dimension>& R) {
162 static_assert(IsManifold<Measurement>::value,
163 "Template parameter Measurement must be a GTSAM Manifold for "
164 "LocalCoordinates.");
165
166 static constexpr int MeasDim = traits<Measurement>::dimension;
167
168 if constexpr (MeasDim == Eigen::Dynamic) {
169 int m_runtime = traits<Measurement>::GetDimension(z);
170 if (traits<Measurement>::GetDimension(prediction) != m_runtime) {
171 throw std::invalid_argument(
172 "ManifoldEKF::update: Dynamic measurement 'prediction' and 'z' "
173 "have different dimensions.");
174 }
175 if (H.rows() != m_runtime || H.cols() != n_ || R.rows() != m_runtime ||
176 R.cols() != m_runtime) {
177 throw std::invalid_argument(
178 "ManifoldEKF::update: Jacobian H or Noise R dimensions mismatch "
179 "for dynamic measurement.");
180 }
181 } else {
182 if constexpr (Dim == Eigen::Dynamic) {
183 if (H.cols() != n_) {
184 throw std::invalid_argument(
185 "ManifoldEKF::update: Jacobian H columns must match state "
186 "dimension " +
187 std::to_string(val: n_) + ".");
188 }
189 }
190 }
191
192 // Innovation: y = z - h(x_pred). In tangent space: local(h(x_pred), z)
193 typename traits<Measurement>::TangentVector innovation =
194 traits<Measurement>::Local(prediction, z);
195
196 // Innovation covariance: S = H P H^T + R
197 // S will be Eigen::Matrix<double, MeasDim, MeasDim>
198 Eigen::Matrix<double, MeasDim, MeasDim> S = H * P_ * H.transpose() + R;
199
200 // Kalman Gain: K = P H^T S^-1
201 // K will be Eigen::Matrix<double, Dim, MeasDim>
202 Eigen::Matrix<double, Dim, MeasDim> K = P_ * H.transpose() * S.inverse();
203
204 // Correction vector in tangent space of M: delta_xi = K * innovation
205 const TangentVector delta_xi =
206 K * innovation; // delta_xi is Dim x 1 (or n_ x 1 if dynamic)
207
208 // Update state using retract: X_new = retract(X_old, delta_xi)
209 X_ = traits<M>::Retract(X_, delta_xi);
210
211 // Update covariance using Joseph form for numerical stability
212 Jacobian I_n; // Eigen::Matrix<double, Dim, Dim>
213 if constexpr (Dim == Eigen::Dynamic) {
214 I_n = Jacobian::Identity(n_, n_);
215 } else {
216 I_n = Jacobian::Identity();
217 }
218
219 // I_KH will be Eigen::Matrix<double, Dim, Dim>
220 Jacobian I_KH = I_n - K * H;
221 P_ = I_KH * P_ * I_KH.transpose() + K * R * K.transpose();
222 }
223
224 /**
225 * Measurement update: Corrects the state and covariance using a measurement
226 * model function.
227 *
228 * @tparam Measurement Type of the measurement vector.
229 * @tparam MeasurementFunction Functor/lambda providing measurement and its
230 * Jacobian. Signature: `Measurement h(const M& x, Jac& H_jacobian)` where H =
231 * d(h)/d(local(X)).
232 * @param h Measurement model function.
233 * @param z Observed measurement.
234 * @param R Measurement noise covariance.
235 */
236 template <typename Measurement, typename MeasurementFunction>
237 void update(MeasurementFunction&& h, const Measurement& z,
238 const Eigen::Matrix<double, traits<Measurement>::dimension,
239 traits<Measurement>::dimension>& R) {
240 static_assert(IsManifold<Measurement>::value,
241 "Template parameter Measurement must be a GTSAM Manifold.");
242
243 static constexpr int MeasDim = traits<Measurement>::dimension;
244
245 int m_runtime;
246 if constexpr (MeasDim == Eigen::Dynamic) {
247 m_runtime = traits<Measurement>::GetDimension(z);
248 } else {
249 m_runtime = MeasDim;
250 }
251
252 // Predict measurement and get Jacobian H = dh/dlocal(X)
253 Matrix H(m_runtime, n_);
254 Measurement prediction = h(X_, H);
255
256 // Call the other update function
257 update(prediction, H, z, R);
258 }
259
260 /// Convenience bridge for wrappers: vector measurement update calling
261 /// update<Vector>. This overload exists to avoid templates in wrappers. It
262 /// validates sizes and forwards to the templated update with Measurement =
263 /// gtsam::Vector (dynamic size).
264 void updateWithVector(const gtsam::Vector& prediction, const Matrix& H,
265 const gtsam::Vector& z, const Matrix& R) {
266 // Basic dimension checks for dynamic-sized measurement
267 const int m = static_cast<int>(prediction.size());
268 if (static_cast<int>(z.size()) != m) {
269 throw std::invalid_argument(
270 "ManifoldEKF::updateWithVector: prediction and z must have same "
271 "length.");
272 }
273 if (H.rows() != m || H.cols() != n_) {
274 throw std::invalid_argument(
275 "ManifoldEKF::updateWithVector: H must be m x n where m = "
276 "measurement size and n = state dimension.");
277 }
278 if (R.rows() != m || R.cols() != m) {
279 throw std::invalid_argument(
280 "ManifoldEKF::updateWithVector: R must be m x m where m = "
281 "measurement size.");
282 }
283
284 // Forward to templated update with Measurement = Vector
285 this->template update<Vector>(prediction, H, z, R);
286 }
287
288 protected:
289 M X_; ///< Manifold state estimate.
290 Covariance P_; ///< Covariance (Eigen::Matrix<double, Dim, Dim>).
291 int n_; ///< Runtime tangent space dimension of M.
292};
