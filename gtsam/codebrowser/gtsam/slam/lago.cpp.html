45/* ************************************************************************* */
46/**
47 * Compute the cumulative orientation (without wrapping) wrt the root of a
48 * spanning tree (tree) for a node (nodeKey). The function starts at the nodes and
49 * moves towards the root summing up the (directed) rotation measurements.
50 * Relative measurements are encoded in "deltaThetaMap".
51 * The root is assumed to have orientation zero.
52 */
53static double computeThetaToRoot(const Key nodeKey,
54 const PredecessorMap& tree, const key2doubleMap& deltaThetaMap,
55 const key2doubleMap& thetaFromRootMap) {
56
57 double nodeTheta = 0;
58 Key key_child = nodeKey; // the node
59 Key key_parent = 0; // the initialization does not matter
60 while (1) {
61 // We check if we reached the root
62 if (tree.at(k: key_child) == key_child) // if we reached the root
63 break;
64 // we sum the delta theta corresponding to the edge parent->child
65 nodeTheta += deltaThetaMap.at(k: key_child);
66 // we get the parent
67 key_parent = tree.at(k: key_child); // the parent
68 // we check if we connected to some part of the tree we know
69 if (thetaFromRootMap.find(x: key_parent) != thetaFromRootMap.end()) {
70 nodeTheta += thetaFromRootMap.at(k: key_parent);
71 break;
72 }
73 key_child = key_parent; // we move upwards in the tree
74 }
75 return nodeTheta;
76}

78/* ************************************************************************* */
79key2doubleMap computeThetasToRoot(const key2doubleMap& deltaThetaMap,
80 const PredecessorMap& tree) {
81
82 key2doubleMap thetaToRootMap;
83
84 // Orientation of the root
85 thetaToRootMap.emplace(args: kAnchorKey, args: 0.0);
86
87 // for all nodes in the tree
88 for(const auto& [nodeKey, _]: deltaThetaMap) {
89 // compute the orientation wrt root
90 const double nodeTheta = computeThetaToRoot(nodeKey, tree, deltaThetaMap,
91 thetaFromRootMap: thetaToRootMap);
92 thetaToRootMap.emplace(args: nodeKey, args: nodeTheta);
93 }
94 return thetaToRootMap;
95}

97/* ************************************************************************* */
98void getSymbolicGraph(
99/*OUTPUTS*/vector<size_t>& spanningTreeIds, vector<size_t>& chordsIds,
100 key2doubleMap& deltaThetaMap,
101 /*INPUTS*/const PredecessorMap& tree, const NonlinearFactorGraph& g) {
102
103 // Get keys for which you want the orientation
104 size_t id = 0;
105 // Loop over the factors
106 for(const std::shared_ptr<NonlinearFactor>& factor: g) {
107 if (factor->keys().size() == 2) {
108 Key key1 = factor->keys()[0];
109 Key key2 = factor->keys()[1];
110 // recast to a between
111 std::shared_ptr<BetweenFactor<Pose2> > pose2Between =
112 std::dynamic_pointer_cast<BetweenFactor<Pose2> >(r: factor);
113 if (!pose2Between)
114 continue;
115 // get the orientation - measured().theta();
116 double deltaTheta = pose2Between->measured().theta();
117 // insert (directed) orientations in the map "deltaThetaMap"
118 bool inTree = false;
119 if (tree.at(k: key1) == key2) { // key2 -> key1
120 deltaThetaMap.insert(x: pair<Key, double>(key1, -deltaTheta));
121 inTree = true;
122 } else if (tree.at(k: key2) == key1) { // key1 -> key2
123 deltaThetaMap.insert(x: pair<Key, double>(key2, deltaTheta));
124 inTree = true;
125 }
126 // store factor slot, distinguishing spanning tree edges from chordsIds
127 if (inTree == true)
128 spanningTreeIds.push_back(x: id);
129 else
130 // it's a chord!
131 chordsIds.push_back(x: id);
132 }
133 id++;
134 }
135}

138// Retrieve the deltaTheta and the corresponding noise model from a BetweenFactor<Pose2>
139static void getDeltaThetaAndNoise(NonlinearFactor::shared_ptr factor,
140 Vector& deltaTheta, noiseModel::Diagonal::shared_ptr& model_deltaTheta) {
141
142 // Get the relative rotation measurement from the between factor
143 std::shared_ptr<BetweenFactor<Pose2> > pose2Between =
144 std::dynamic_pointer_cast<BetweenFactor<Pose2> >(r: factor);
145 if (!pose2Between)
146 throw invalid_argument(
147 "buildLinearOrientationGraph: invalid between factor!");
148 deltaTheta = (Vector(1) << pose2Between->measured().theta()).finished();
149
150 // Retrieve the noise model for the relative rotation
151 SharedNoiseModel model = pose2Between->noiseModel();
152 std::shared_ptr<noiseModel::Diagonal> diagonalModel =
153 std::dynamic_pointer_cast<noiseModel::Diagonal>(r: model);
154 if (!diagonalModel)
155 throw invalid_argument("buildLinearOrientationGraph: invalid noise model "
156 "(current version assumes diagonal noise model)!");
157 Vector std_deltaTheta = (Vector(1) << diagonalModel->sigma(i: 2)).finished(); // std on the angular measurement
158 model_deltaTheta = noiseModel::Diagonal::Sigmas(sigmas: std_deltaTheta);
159}

161/* ************************************************************************* */
162GaussianFactorGraph buildLinearOrientationGraph(
163 const vector<size_t>& spanningTreeIds, const vector<size_t>& chordsIds,
164 const NonlinearFactorGraph& g, const key2doubleMap& orientationsToRoot,
165 const PredecessorMap& tree) {
166
167 GaussianFactorGraph lagoGraph;
168 Vector deltaTheta;
169 noiseModel::Diagonal::shared_ptr model_deltaTheta;
170
171 // put original measurements in the spanning tree
172 for(const size_t& factorId: spanningTreeIds) {
173 const KeyVector& keys = g[factorId]->keys();
174 Key key1 = keys[0], key2 = keys[1];
175 getDeltaThetaAndNoise(factor: g[factorId], deltaTheta, model_deltaTheta);
176 lagoGraph.add(key1, A1: -I_1x1, key2, A2: I_1x1, b: deltaTheta, model: model_deltaTheta);
177 }
178 // put regularized measurements in the chords
179 for(const size_t& factorId: chordsIds) {
180 const KeyVector& keys = g[factorId]->keys();
181 Key key1 = keys[0], key2 = keys[1];
182 getDeltaThetaAndNoise(factor: g[factorId], deltaTheta, model_deltaTheta);
183 double key1_DeltaTheta_key2 = deltaTheta(0);
184 ///cout << "REG: key1= " << DefaultKeyFormatter(key1) << " key2= " << DefaultKeyFormatter(key2) << endl;
185 double k2pi_noise = key1_DeltaTheta_key2 + orientationsToRoot.at(k: key1)
186 - orientationsToRoot.at(k: key2); // this coincides to summing up measurements along the cycle induced by the chord
187 double k = std::round(x: k2pi_noise / (2 * M_PI));
188 //if (k2pi_noise - 2*k*M_PI > 1e-5) cout << k2pi_noise - 2*k*M_PI << endl; // for debug
189 Vector deltaThetaRegularized = (Vector(1)
190 << key1_DeltaTheta_key2 - 2 * k * M_PI).finished();
191 lagoGraph.add(key1, A1: -I_1x1, key2, A2: I_1x1, b: deltaThetaRegularized, model: model_deltaTheta);
192 }
193 // prior on the anchor orientation
194 lagoGraph.add(key1: kAnchorKey, A1: I_1x1, b: (Vector(1) << 0.0).finished(), model: priorOrientationNoise);
195 return lagoGraph;
196}

198/* ************************************************************************* */
199static PredecessorMap findOdometricPath(
200 const NonlinearFactorGraph& pose2Graph) {
201
202 PredecessorMap tree;
203 Key minKey = kAnchorKey; // this initialization does not matter
204 bool minUnassigned = true;
205
206 for(const std::shared_ptr<NonlinearFactor>& factor: pose2Graph) {
207
208 Key key1 = std::min(a: factor->keys()[0], b: factor->keys()[1]);
209 Key key2 = std::max(a: factor->keys()[0], b: factor->keys()[1]);
210 if (minUnassigned) {
211 minKey = key1;
212 minUnassigned = false;
213 }
214 if (key2 - key1 == 1) { // consecutive keys
215 tree.emplace(args&: key2, args&: key1);
216 if (key1 < minKey)
217 minKey = key1;
218 }
219 }
220 tree.emplace(args&: minKey, args: kAnchorKey);
221 tree.emplace(args: kAnchorKey, args: kAnchorKey); // root
222 return tree;
223}

225/*****************************************************************************/
226PredecessorMap findMinimumSpanningTree(
227 const NonlinearFactorGraph& pose2Graph) {
228 // Compute the minimum spanning tree
229 const auto edgeWeights = std::vector<double>(pose2Graph.size(), 1.0);
230 const auto mstEdgeIndices =
231 utils::kruskal(fg: pose2Graph, weights: edgeWeights);
232
233 // Create a PredecessorMap 'predecessorMap' such that:
234 // predecessorMap[key2] = key1, where key1 is the 'parent' node for key2 in
235 // the spanning tree
236 PredecessorMap predecessorMap;
237 std::map<Key, bool> visitationMap;
238 std::stack<std::pair<Key, Key>> stack;
239
240 stack.push(x: {kAnchorKey, kAnchorKey});
241 while (!stack.empty()) {
242 auto [u, parent] = stack.top();
243 stack.pop();
244 if (visitationMap[u]) continue;
245 visitationMap[u] = true;
246 predecessorMap[u] = parent;
247 for (const auto& edgeIdx : mstEdgeIndices) {
248 const auto v = pose2Graph[edgeIdx]->front();
249 const auto w = pose2Graph[edgeIdx]->back();
250 if ((v == u || w == u) && !visitationMap[v == u ? w : v]) {
251 stack.push(x: {v == u ? w : v, u});
252 }
253 }
254 }
255
256 return predecessorMap;
257}

259/* ************************************************************************* */
260// Return the orientations of a graph including only BetweenFactors<Pose2>
261static VectorValues computeOrientations(const NonlinearFactorGraph& pose2Graph,
262 bool useOdometricPath) {
263 gttic(lago_computeOrientations);
264
265 PredecessorMap tree;
266 // Find a minimum spanning tree
267 if (useOdometricPath)
268 tree = findOdometricPath(pose2Graph);
269 else
270 tree = findMinimumSpanningTree(pose2Graph);
271
272 // Create a linear factor graph (LFG) of scalars
273 key2doubleMap deltaThetaMap;
274 vector<size_t>
275 spanningTreeIds; // ids of between factors forming the spanning tree T
276 vector<size_t>
277 chordsIds; // ids of between factors corresponding to chordsIds wrt T
278 getSymbolicGraph(spanningTreeIds, chordsIds, deltaThetaMap, tree, g: pose2Graph);
279
280 // temporary structure to correct wraparounds along loops
281 key2doubleMap orientationsToRoot = computeThetasToRoot(deltaThetaMap, tree);
282
283 // regularize measurements and plug everything in a factor graph
284 GaussianFactorGraph lagoGraph = buildLinearOrientationGraph(
285 spanningTreeIds, chordsIds, g: pose2Graph, orientationsToRoot, tree);
286
287 // Solve the LFG
288 VectorValues orientationsLago = lagoGraph.optimize();
289
290 return orientationsLago;
291}

293/* ************************************************************************* */
294VectorValues initializeOrientations(const NonlinearFactorGraph& graph,
295 bool useOdometricPath) {
296 // We "extract" the Pose2 subgraph of the original graph: this
297 // is done to properly model priors and avoiding operating on a larger graph
298 NonlinearFactorGraph pose2Graph = initialize::buildPoseGraph<Pose2>(graph);
299
300 // Get orientations from relative orientation measurements
301 return computeOrientations(pose2Graph, useOdometricPath);
302}

304/* ************************************************************************* */
305Values computePoses(const NonlinearFactorGraph& pose2graph,
306 VectorValues& orientationsLago) {
307 gttic(lago_computePoses);
308
309 // Linearized graph on full poses
310 GaussianFactorGraph linearPose2graph;
311
312 // We include the linear version of each between factor
313 for(const std::shared_ptr<NonlinearFactor>& factor: pose2graph) {
314
315 std::shared_ptr<BetweenFactor<Pose2> > pose2Between =
316 std::dynamic_pointer_cast<BetweenFactor<Pose2> >(r: factor);
317
318 if (pose2Between) {
319 Key key1 = pose2Between->keys()[0];
320 double theta1 = orientationsLago.at(j: key1)(0);
321 double s1 = sin(x: theta1);
322 double c1 = cos(x: theta1);
323
324 Key key2 = pose2Between->keys()[1];
325 double theta2 = orientationsLago.at(j: key2)(0);
326
327 double linearDeltaRot = theta2 - theta1
328 - pose2Between->measured().theta();
329 linearDeltaRot = Rot2(linearDeltaRot).theta(); // to normalize
330
331 double dx = pose2Between->measured().x();
332 double dy = pose2Between->measured().y();
333
334 Vector globalDeltaCart = //
335 (Vector(2) << c1 * dx - s1 * dy, s1 * dx + c1 * dy).finished();
336 Vector b = (Vector(3) << globalDeltaCart, linearDeltaRot).finished(); // rhs
337 Matrix J1 = -I_3x3;
338 J1(0, 2) = s1 * dx + c1 * dy;
339 J1(1, 2) = -c1 * dx + s1 * dy;
340 // Retrieve the noise model for the relative rotation
341 std::shared_ptr<noiseModel::Diagonal> diagonalModel =
342 std::dynamic_pointer_cast<noiseModel::Diagonal>(
343 r: pose2Between->noiseModel());
344
345 linearPose2graph.add(key1, A1: J1, key2, A2: I_3x3, b, model: diagonalModel);
346 } else {
347 throw invalid_argument(
348 "computeLagoPoses: cannot manage non between factor here!");
349 }
350 }
351 // add prior
352 linearPose2graph.add(key1: kAnchorKey, A1: I_3x3, b: Vector3(0.0, 0.0, 0.0), model: priorPose2Noise);
353
354 // optimize
355 VectorValues posesLago = linearPose2graph.optimize();
356
357 // put into Values structure
358 Values initialGuessLago;
359 for(const VectorValues::value_type& it: posesLago) {
360 Key key = it.first;
361 if (key != kAnchorKey) {
362 const Vector& poseVector = it.second;
363 Pose2 poseLago = Pose2(poseVector(0), poseVector(1),
364 orientationsLago.at(j: key)(0) + poseVector(2));
365 initialGuessLago.insert(j: key, val: poseLago);
366 }
367 }
368 return initialGuessLago;
369}

371/* ************************************************************************* */
372Values initialize(const NonlinearFactorGraph& graph, bool useOdometricPath) {
373 gttic(lago_initialize);
374
375 // We "extract" the Pose2 subgraph of the original graph: this
376 // is done to properly model priors and avoiding operating on a larger graph
377 NonlinearFactorGraph pose2Graph = initialize::buildPoseGraph<Pose2>(graph);
378
379 // Get orientations from relative orientation measurements
380 VectorValues orientationsLago = computeOrientations(pose2Graph,
381 useOdometricPath);
382
383 // Compute the full poses
384 return computePoses(pose2graph: pose2Graph, orientationsLago);
385}

387/* ************************************************************************* */
388Values initialize(const NonlinearFactorGraph& graph,
389 const Values& initialGuess) {
390 Values initialGuessLago;
391
392 // get the orientation estimates from LAGO
393 VectorValues orientations = initializeOrientations(graph);
394
395 // for all nodes in the tree
396 for(const VectorValues::value_type& it: orientations) {
397 Key key = it.first;
398 if (key != kAnchorKey) {
399 const Pose2& pose = initialGuess.at<Pose2>(j: key);
400 const Vector& orientation = it.second;
401 Pose2 poseLago = Pose2(pose.x(), pose.y(), orientation(0));
402 initialGuessLago.insert(j: key, val: poseLago);
403 }
404 }
405 return initialGuessLago;
406}
