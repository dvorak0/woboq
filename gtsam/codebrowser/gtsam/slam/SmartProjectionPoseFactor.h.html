24namespace gtsam {

36/**
37 * This factor assumes that camera calibration is fixed, and that
38 * the calibration is the same for all cameras involved in this factor.
39 * The factor only constrains poses (variable dimension is 6).
40 * This factor requires that values contains the involved poses (Pose3).
41 * If the calibration should be optimized, as well, use SmartProjectionFactor instead!
42 * @ingroup slam
43 */
44template <class CALIBRATION>
45class SmartProjectionPoseFactor
46 : public SmartProjectionFactor<PinholePose<CALIBRATION> > {
47 private:
48 typedef PinholePose<CALIBRATION> Camera;
49 typedef SmartProjectionFactor<Camera> Base;
50 typedef SmartProjectionPoseFactor<CALIBRATION> This;
51
52protected:
53
54 std::shared_ptr<CALIBRATION> K_; ///< calibration object (one for all cameras)
55
56public:
57
58 /// shorthand for a smart pointer to a factor
59 typedef std::shared_ptr<This> shared_ptr;
60
61 /**
62 * Default constructor, only for serialization
63 */
64 SmartProjectionPoseFactor() {}
65
66 /**
67 * Constructor
68 * @param sharedNoiseModel isotropic noise model for the 2D feature measurements
69 * @param K (fixed) calibration, assumed to be the same for all cameras
70 * @param params parameters for the smart projection factors
71 */
72 SmartProjectionPoseFactor(
73 const SharedNoiseModel& sharedNoiseModel,
74 const std::shared_ptr<CALIBRATION> K,
75 const SmartProjectionParams& params = SmartProjectionParams())
76 : Base(sharedNoiseModel, params), K_(K) {
77 }
78
79 /**
80 * Constructor
81 * @param sharedNoiseModel isotropic noise model for the 2D feature measurements
82 * @param K (fixed) calibration, assumed to be the same for all cameras
83 * @param body_P_sensor pose of the camera in the body frame (optional)
84 * @param params parameters for the smart projection factors
85 */
86 SmartProjectionPoseFactor(
87 const SharedNoiseModel& sharedNoiseModel,
88 const std::shared_ptr<CALIBRATION> K,
89 const std::optional<Pose3> body_P_sensor,
90 const SmartProjectionParams& params = SmartProjectionParams())
91 : SmartProjectionPoseFactor(sharedNoiseModel, K, params) {
92 this->body_P_sensor_ = body_P_sensor;
93 }
94
95 /** Virtual destructor */
96 ~SmartProjectionPoseFactor() override {
97 }
98
99 /**
100 * print
101 * @param s optional string naming the factor
102 * @param keyFormatter optional formatter useful for printing Symbols
103 */
104 void print(const std::string& s = "", const KeyFormatter& keyFormatter =
105 DefaultKeyFormatter) const override {
106 std::cout << s << "SmartProjectionPoseFactor, z = \n ";
107 Base::print("", keyFormatter);
108 }
109
110 /// equals
111 bool equals(const NonlinearFactor& p, double tol = 1e-9) const override {
112 const This *e = dynamic_cast<const This*>(&p);
113 return e && Base::equals(p, tol);
114 }
115
116 /**
117 * error calculates the error of the factor.
118 */
119 double error(const Values& values) const override {
120 if (this->active(values)) {
121 return this->totalReprojectionError(cameras(values));
122 } else { // else of active flag
123 return 0.0;
124 }
125 }
126
127 /** return calibration shared pointers */
128 inline const std::shared_ptr<CALIBRATION> calibration() const {
129 return K_;
130 }
131
132 /**
133 * Collect all cameras involved in this factor
134 * @param values Values structure which must contain camera poses corresponding
135 * to keys involved in this factor
136 * @return vector of Values
137 */
138 typename Base::Cameras cameras(const Values& values) const override {
139 typename Base::Cameras cameras;
140 for (const Key& k : this->keys_) {
141 const Pose3 world_P_sensor_k =
142 Base::body_P_sensor_ ? values.at<Pose3>(j: k) * *Base::body_P_sensor_
143 : values.at<Pose3>(j: k);
144 cameras.emplace_back(world_P_sensor_k, K_);
145 }
146 return cameras;
147 }
148
149 private:
150
151#if GTSAM_ENABLE_BOOST_SERIALIZATION ///
152 /// Serialization function
153 friend class boost::serialization::access;
154 template<class ARCHIVE>
155 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
156 ar & BOOST_SERIALIZATION_BASE_OBJECT_NVP(Base);
157 ar & BOOST_SERIALIZATION_NVP(K_);
158 }
159#endif
160};
161// end of class declaration
162
163/// traits
164template<class CALIBRATION>
165struct traits<SmartProjectionPoseFactor<CALIBRATION> > : public Testable<
166 SmartProjectionPoseFactor<CALIBRATION> > {
167};
