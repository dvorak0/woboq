18#pragma once

20#include <gtsam/nonlinear/NonlinearFactor.h>
21#include <gtsam/geometry/CalibratedCamera.h>

23namespace gtsam {
24
25/**
26 * Non-linear factor for a constraint derived from a 2D measurement.
27 * The calibration and pose are assumed known.
28 * @ingroup slam
29 */
30template<class CAMERA>
31class TriangulationFactor: public NoiseModelFactorN<Point3> {
32
33public:
34
35 /// CAMERA type
36 using Camera = CAMERA;
37 /// shorthand for measurement type, e.g. Point2 or StereoPoint2
38 using Measurement = typename CAMERA::Measurement;
39
40protected:
41
42 /// shorthand for base class type
43 using Base = NoiseModelFactorN<Point3>;
44
45 /// shorthand for this class
46 using This = TriangulationFactor<CAMERA>;
47
48 // Keep a copy of measurement and calibration for I/O
49 const CAMERA camera_; ///< CAMERA in which this landmark was seen
50 const Measurement measured_; ///< 2D measurement
51
52 // verbosity handling for Cheirality Exceptions
53 const bool throwCheirality_; ///< If true, rethrows Cheirality exceptions (default: false)
54 const bool verboseCheirality_; ///< If true, prints text for Cheirality exceptions (default: false)
55
56public:
57 EIGEN_MAKE_ALIGNED_OPERATOR_NEW
58
59 /// shorthand for a smart pointer to a factor
60 using shared_ptr = std::shared_ptr<This>;
61
62 // Provide access to the Matrix& version of evaluateError:
63 using NoiseModelFactor1<Point3>::evaluateError;
64
65 /// Default constructor
66 TriangulationFactor() :
67 throwCheirality_(false), verboseCheirality_(false), measured_() {
68 }
69
70 /**
71 * Constructor with exception-handling flags
72 * @param camera is the camera in which unknown landmark is seen
73 * @param measured is the 2 dimensional location of point in image (the measurement)
74 * @param model is the standard deviation
75 * @param pointKey is the index of the landmark
76 * @param throwCheirality determines whether Cheirality exceptions are rethrown
77 * @param verboseCheirality determines whether exceptions are printed for Cheirality
78 */
79 TriangulationFactor(const CAMERA& camera, const Measurement& measured,
80 const SharedNoiseModel& model, Key pointKey, bool throwCheirality = false,
81 bool verboseCheirality = false) :
82 Base(model, pointKey), camera_(camera), measured_(measured), throwCheirality_(
83 throwCheirality), verboseCheirality_(verboseCheirality) {
84 if (model && model->dim() != traits<Measurement>::dimension)
85 throw std::invalid_argument(
86 "TriangulationFactor must be created with "
87 + std::to_string(val: (int) traits<Measurement>::dimension)
88 + "-dimensional noise model.");
89 }
90
91 /** Virtual destructor */
92 ~TriangulationFactor() override {
93 }
94
95 /// @return a deep copy of this factor
96 gtsam::NonlinearFactor::shared_ptr clone() const override {
97 return std::static_pointer_cast<gtsam::NonlinearFactor>(
98 r: gtsam::NonlinearFactor::shared_ptr(new This(*this)));
99 }
100
101 /**
102 * print
103 * @param s optional string naming the factor
104 * @param keyFormatter optional formatter useful for printing Symbols
105 */
106 void print(const std::string& s = "", const KeyFormatter& keyFormatter =
107 DefaultKeyFormatter) const override {
108 std::cout << s << "TriangulationFactor,";
109 camera_.print("camera");
110 traits<Measurement>::Print(measured_, "z");
111 Base::print(s: "", keyFormatter);
112 }
113
114 /// equals
115 bool equals(const NonlinearFactor& p, double tol = 1e-9) const override {
116 const This *e = dynamic_cast<const This*>(&p);
117 return e && Base::equals(f: p, tol) && this->camera_.equals(e->camera_, tol)
118 && traits<Measurement>::Equals(this->measured_, e->measured_, tol);
119 }
120
121 /// Evaluate error h(x)-z and optionally derivatives
122 Vector evaluateError(const Point3& point, OptionalMatrixType H2) const override {
123 try {
124 return traits<Measurement>::Local(measured_, camera_.project2(point, OptionalNone, H2));
125 } catch (CheiralityException& e) {
126 if (H2)
127 *H2 = Matrix::Zero(traits<Measurement>::dimension, 3);
128 if (verboseCheirality_)
129 std::cout << e.what() << ": Landmark "
130 << DefaultKeyFormatter(this->key()) << " moved behind camera"
131 << std::endl;
132 if (throwCheirality_)
133 throw e;
134 return camera_.defaultErrorWhenTriangulatingBehindCamera();
135 }
136 }
137
138 /// thread-safe (?) scratch memory for linearize
139 mutable VerticalBlockMatrix Ab;
140 mutable Matrix A;
141 mutable Vector b;
142
143 /**
144 * Linearize to a JacobianFactor, does not support constrained noise model !
145 * \f$ Ax-b \approx h(x+\delta x)-z = h(x) + A \delta x - z \f$
146 * Hence \f$ b = z - h(x) = - \mathtt{error\_vector}(x) \f$
147 */
148 std::shared_ptr<GaussianFactor> linearize(const Values& x) const override {
149 // Only linearize if the factor is active
150 if (!this->active(x))
151 return std::shared_ptr<JacobianFactor>();
152
153 // Allocate memory for Jacobian factor, do only once
154 if (Ab.rows() == 0) {
155 std::vector<size_t> dimensions(1, 3);
156 Ab = VerticalBlockMatrix(dimensions, traits<Measurement>::dimension, true);
157 A.resize(traits<Measurement>::dimension,3);
158 b.resize(traits<Measurement>::dimension);
159 }
160
161 // Would be even better if we could pass blocks to project
162 const Point3& point = x.at<Point3>(key());
163 b = traits<Measurement>::Local(camera_.project2(point, {}, A), measured_);
164 if (noiseModel_)
165 this->noiseModel_->WhitenSystem(A, b);
166
167 Ab(0) = A;
168 Ab(1) = b;
169
170 return std::make_shared<JacobianFactor>(this->keys_, Ab);
171 }
172
173 /** return the measurement */
174 const Measurement& measured() const {
175 return measured_;
176 }
177
178 /** return verbosity */
179 inline bool verboseCheirality() const {
180 return verboseCheirality_;
181 }
182
183 /** return flag for throwing cheirality exceptions */
184 inline bool throwCheirality() const {
185 return throwCheirality_;
186 }
187
188private:
189
190#if GTSAM_ENABLE_BOOST_SERIALIZATION ///
191 /// Serialization function
192 friend class boost::serialization::access;
193 template<class ARCHIVE>
194 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
195 ar & BOOST_SERIALIZATION_BASE_OBJECT_NVP(Base);
196 ar & BOOST_SERIALIZATION_NVP(camera_);
197 ar & BOOST_SERIALIZATION_NVP(measured_);
198 ar & BOOST_SERIALIZATION_NVP(throwCheirality_);
199 ar & BOOST_SERIALIZATION_NVP(verboseCheirality_);
200 }
201#endif
202};
203} // \ namespace gtsam
