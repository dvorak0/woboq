15/**
16 * Factor on unknown rotation iRC that relates two incremental rotations
17 * c1Rc2 = iRc' * i1Ri2 * iRc
18 * Which we can write (see doc/math.lyx)
19 * e^[z] = iRc' * e^[p] * iRc = e^([iRc'*p])
20 * with z and p measured and predicted angular velocities, and hence
21 * p = iRc * z
22 */
23class RotateFactor: public NoiseModelFactorN<Rot3> {
24
25 Point3 p_, z_; ///< Predicted and measured directions, p = iRc * z
26
27 typedef NoiseModelFactorN<Rot3> Base;
28 typedef RotateFactor This;
29
30public:
31
32 // Provide access to the Matrix& version of evaluateError:
33 using Base::evaluateError;
34
35 /// Constructor
36 RotateFactor(Key key, const Rot3& P, const Rot3& Z,
37 const SharedNoiseModel& model) :
38 Base(model, key), p_(Rot3::Logmap(R: P)), z_(Rot3::Logmap(R: Z)) {
39 }
40
41 /// @return a deep copy of this factor
42 gtsam::NonlinearFactor::shared_ptr clone() const override {
43 return std::static_pointer_cast<gtsam::NonlinearFactor>(
44 r: gtsam::NonlinearFactor::shared_ptr(new This(*this))); }
45
46 /// print
47 void print(const std::string& s = "",
48 const KeyFormatter& keyFormatter = DefaultKeyFormatter) const override {
49 Base::print(s);
50 std::cout << "RotateFactor:]\n";
51 std::cout << "p: " << p_.transpose() << std::endl;
52 std::cout << "z: " << z_.transpose() << std::endl;
53 }
54
55 /// vector of errors returns 2D vector
56 Vector evaluateError(const Rot3& R, OptionalMatrixType H) const override {
57 // predict p_ as q = R*z_, derivative H will be filled if not none
58 Point3 q = R.rotate(p: z_,H1: H);
59 // error is just difference, and note derivative of that wrpt q is I3
60 return (Vector(3) << q.x()-p_.x(), q.y()-p_.y(), q.z()-p_.z()).finished();
61 }
62
63};

65/**
66 * Factor on unknown rotation iRc that relates two directions c
67 * Directions provide less constraints than a full rotation
68 */
69class RotateDirectionsFactor: public NoiseModelFactorN<Rot3> {
70
71 Unit3 i_p_, c_z_; ///< Predicted and measured directions, i_p = iRc * c_z
72
73 typedef NoiseModelFactorN<Rot3> Base;
74 typedef RotateDirectionsFactor This;
75
76public:
77
78 // Provide access to the Matrix& version of evaluateError:
79 using Base::evaluateError;
80
81 /// Constructor
82 RotateDirectionsFactor(Key key, const Unit3& i_p, const Unit3& c_z,
83 const SharedNoiseModel& model) :
84 Base(model, key), i_p_(i_p), c_z_(c_z) {
85 }
86
87 /// Initialize rotation iRc such that i_p = iRc * c_z
88 static Rot3 Initialize(const Unit3& i_p, const Unit3& c_z) {
89 gtsam::Quaternion iRc;
90 // setFromTwoVectors sets iRc to (a) quaternion which transform c_z into i_p
91 iRc.setFromTwoVectors(a: c_z.unitVector(), b: i_p.unitVector());
92 return Rot3(iRc);
93 }
94
95 /// @return a deep copy of this factor
96 gtsam::NonlinearFactor::shared_ptr clone() const override {
97 return std::static_pointer_cast<gtsam::NonlinearFactor>(
98 r: gtsam::NonlinearFactor::shared_ptr(new This(*this))); }
99
100 /// print
101 void print(const std::string& s = "",
102 const KeyFormatter& keyFormatter = DefaultKeyFormatter) const override {
103 Base::print(s);
104 std::cout << "RotateDirectionsFactor:" << std::endl;
105 i_p_.print(s: "p");
106 c_z_.print(s: "z");
107 }
108
109 /// vector of errors returns 2D vector
110 Vector evaluateError(const Rot3& iRc, OptionalMatrixType H) const override {
111 Unit3 i_q = iRc * c_z_;
112 Vector error = i_p_.errorVector(q: i_q, H_p: {}, H_q: H);
113 if (H) {
114 Matrix DR;
115 iRc.rotate(p: c_z_, HR: DR);
116 *H = (*H) * DR;
117 }
118 return error;
119 }
120
121 GTSAM_MAKE_ALIGNED_OPERATOR_NEW
122};
