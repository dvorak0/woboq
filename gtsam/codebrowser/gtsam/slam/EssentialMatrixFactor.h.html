31/**
32 * Factor that evaluates epipolar error p'Ep for given essential matrix
33 */
34class EssentialMatrixFactor : public NoiseModelFactorN<EssentialMatrix> {
35 Vector3 vA_, vB_; ///< Homogeneous versions, in ideal coordinates
36
37 typedef NoiseModelFactorN<EssentialMatrix> Base;
38 typedef EssentialMatrixFactor This;
39
40 public:
41 // Provide access to the Matrix& version of evaluateError:
42 using Base::evaluateError;
43
44 /**
45 * Constructor
46 * @param key Essential Matrix variable key
47 * @param pA point in first camera, in calibrated coordinates
48 * @param pB point in second camera, in calibrated coordinates
49 * @param model noise model is about dot product in ideal, homogeneous
50 * coordinates
51 */
52 EssentialMatrixFactor(Key key, const Point2& pA, const Point2& pB,
53 const SharedNoiseModel& model)
54 : Base(model, key) {
55 vA_ = EssentialMatrix::Homogeneous(p: pA);
56 vB_ = EssentialMatrix::Homogeneous(p: pB);
57 }
58
59 /**
60 * Constructor
61 * @param key Essential Matrix variable key
62 * @param pA point in first camera, in pixel coordinates
63 * @param pB point in second camera, in pixel coordinates
64 * @param model noise model is about dot product in ideal, homogeneous
65 * coordinates
66 * @param K calibration object, will be used only in constructor
67 */
68 template <class CALIBRATION>
69 EssentialMatrixFactor(Key key, const Point2& pA, const Point2& pB,
70 const SharedNoiseModel& model,
71 std::shared_ptr<CALIBRATION> K)
72 : Base(model, key) {
73#ifndef NDEBUG
74 if (!K) throw;
75#endif
76 vA_ = EssentialMatrix::Homogeneous(p: K->calibrate(pA));
77 vB_ = EssentialMatrix::Homogeneous(p: K->calibrate(pB));
78 }
79
80 /// @return a deep copy of this factor
81 gtsam::NonlinearFactor::shared_ptr clone() const override {
82 return std::static_pointer_cast<gtsam::NonlinearFactor>(
83 r: gtsam::NonlinearFactor::shared_ptr(new This(*this)));
84 }
85
86 /// print
87 void print(
88 const std::string& s = "",
89 const KeyFormatter& keyFormatter = DefaultKeyFormatter) const override {
90 Base::print(s);
91 std::cout << " EssentialMatrixFactor with measurements\n ("
92 << vA_.transpose() << ")' and (" << vB_.transpose() << ")'"
93 << std::endl;
94 }
95
96 /// vector of errors returns 1D vector
97 Vector evaluateError(const EssentialMatrix& E,
98 OptionalMatrixType H) const override {
99 Vector error(1);
100 error(0) = E.error(vA: vA_, vB: vB_, H);
101 return error;
102 }
103
104 public:
105 GTSAM_MAKE_ALIGNED_OPERATOR_NEW
106};

176 /*
177 * Vector of errors returns 2D vector
178 * @param E essential matrix
179 * @param d inverse depth d
180 */
181 Vector evaluateError(const EssentialMatrix& E, const double& d,
182 OptionalMatrixType DE,
183 OptionalMatrixType Dd) const override {
184 // We have point x,y in image 1
185 // Given a depth Z, the corresponding 3D point P1 = Z*(x,y,1) = (x,y,1)/d
186 // We then convert to second camera by P2 = 1R2'*(P1-1T2)
187 // The homogeneous coordinates of can be written as
188 // 2R1*(P1-1T2) == 2R1*d*(P1-1T2) == 2R1*((x,y,1)-d*1T2)
189 // where we multiplied with d which yields equivalent homogeneous
190 // coordinates. Note that this is just the homography 2R1 for d==0 The point
191 // d*P1 = (x,y,1) is computed in constructor as dP1_
192
193 // Project to normalized image coordinates, then uncalibrate
194 Point2 pn(0, 0);
195 if (!DE && !Dd) {
196 Point3 _1T2 = E.direction().point3();
197 Point3 d1T2 = d * _1T2;
198 Point3 dP2 = E.rotation().unrotate(p: dP1_ - d1T2); // 2R1*((x,y,1)-d*1T2)
199 pn = PinholeBase::Project(pc: dP2);
200
201 } else {
202 // Calculate derivatives. TODO if slow: optimize with Mathematica
203 // 3*2 3*3 3*3
204 Matrix D_1T2_dir, DdP2_rot, DP2_point;
205
206 Point3 _1T2 = E.direction().point3(H: D_1T2_dir);
207 Point3 d1T2 = d * _1T2;
208 Point3 dP2 = E.rotation().unrotate(p: dP1_ - d1T2, H1: DdP2_rot, H2: DP2_point);
209
210 Matrix23 Dpn_dP2;
211 pn = PinholeBase::Project(pc: dP2, Dpoint: Dpn_dP2);
212
213 if (DE) {
214 Matrix DdP2_E(3, 5);
215 DdP2_E << DdP2_rot, -DP2_point * d * D_1T2_dir; // (3*3), (3*3) * (3*2)
216 *DE = f_ * Dpn_dP2 * DdP2_E; // (2*3) * (3*5)
217 }
218
219 if (Dd) // efficient backwards computation:
220 // (2*3) * (3*3) * (3*1)
221 *Dd = -f_ * (Dpn_dP2 * (DP2_point * _1T2));
222 }
223 Point2 reprojectionError = pn - pn_;
224 return f_ * reprojectionError;
225 }
226
227 public:
228 GTSAM_MAKE_ALIGNED_OPERATOR_NEW
229};

289 /*
290 * Vector of errors returns 2D vector
291 * @param E essential matrix
292 * @param d inverse depth d
293 */
294 Vector evaluateError(const EssentialMatrix& E, const double& d,
295 OptionalMatrixType DE,
296 OptionalMatrixType Dd) const override {
297 if (!DE) {
298 // Convert E from body to camera frame
299 EssentialMatrix cameraE = cRb_ * E;
300 // Evaluate error
301 return Base::evaluateError(E: cameraE, d, OptionalNone, Dd);
302 } else {
303 // Version with derivatives
304 Matrix D_e_cameraE, D_cameraE_E; // 2*5, 5*5
305 EssentialMatrix cameraE = E.rotate(cRb: cRb_, HE: D_cameraE_E);
306 // Using the pointer version of evaluateError since the Base class
307 // (EssentialMatrixFactor2) does not have the matrix reference version of
308 // evaluateError
309 Vector e = Base::evaluateError(E: cameraE, d, DE: &D_e_cameraE, Dd);
310 *DE = D_e_cameraE * D_cameraE_E; // (2*5) * (5*5)
311 return e;
312 }
313 }
314
315 public:
316 GTSAM_MAKE_ALIGNED_OPERATOR_NEW
317};

320/**
321 * Binary factor that optimizes for E and calibration K using the algebraic
322 * epipolar error (K^-1 pA)'E (K^-1 pB). The calibration is shared between two
323 * images.
324 *
325 * Note: As correspondences between 2d coordinates can only recover 7 DoF,
326 * this factor should always be used with a prior factor on calibration.
327 * Even with a prior, we can only optimize 2 DoF in the calibration. So the
328 * prior should have a noise model with very low sigma in the remaining
329 * dimensions. This has been tested to work on Cal3_S2. With Cal3Bundler, it
330 * works only with a strong prior (low sigma noise model) on all degrees of
331 * freedom.
332 */
333template <class CALIBRATION>
334class EssentialMatrixFactor4
335 : public NoiseModelFactorN<EssentialMatrix, CALIBRATION> {
336 private:
337 Point2 pA_, pB_; ///< points in pixel coordinates
338
339 typedef NoiseModelFactorN<EssentialMatrix, CALIBRATION> Base;
340 typedef EssentialMatrixFactor4 This;
341
342 static constexpr int DimK = FixedDimension<CALIBRATION>::value;
343 typedef Eigen::Matrix<double, 2, DimK> JacobianCalibration;
344
345 public:
346 // Provide access to the Matrix& version of evaluateError:
347 using Base::evaluateError;
348
349 /**
350 * Constructor
351 * @param keyE Essential Matrix aEb variable key
352 * @param keyK Calibration variable key (common for both cameras)
353 * @param pA point in first camera, in pixel coordinates
354 * @param pB point in second camera, in pixel coordinates
355 * @param model noise model is about dot product in ideal, homogeneous
356 * coordinates
357 */
358 EssentialMatrixFactor4(Key keyE, Key keyK, const Point2& pA, const Point2& pB,
359 const SharedNoiseModel& model = nullptr)
360 : Base(model, keyE, keyK), pA_(pA), pB_(pB) {}
361
362 /// @return a deep copy of this factor
363 gtsam::NonlinearFactor::shared_ptr clone() const override {
364 return std::static_pointer_cast<gtsam::NonlinearFactor>(
365 r: gtsam::NonlinearFactor::shared_ptr(new This(*this)));
366 }
367
368 /// print
369 void print(
370 const std::string& s = "",
371 const KeyFormatter& keyFormatter = DefaultKeyFormatter) const override {
372 Base::print(s);
373 std::cout << " EssentialMatrixFactor4 with measurements\n ("
374 << pA_.transpose() << ")' and (" << pB_.transpose() << ")'"
375 << std::endl;
376 }
377
378 /**
379 * @brief Calculate the algebraic epipolar error pA' (K^-1)' E K pB.
380 *
381 * @param E essential matrix for key keyE
382 * @param K calibration (common for both images) for key keyK
383 * @param H1 optional jacobian of error w.r.t E
384 * @param H2 optional jacobian of error w.r.t K
385 * @return * Vector 1D vector of algebraic error
386 */
387 Vector evaluateError(const EssentialMatrix& E, const CALIBRATION& K,
388 OptionalMatrixType HE,
389 OptionalMatrixType HK) const override {
390 // converting from pixel coordinates to normalized coordinates cA and cB
391 JacobianCalibration cA_H_K; // dcA/dK
392 JacobianCalibration cB_H_K; // dcB/dK
393 Point2 cA = K.calibrate(pA_, HK ? &cA_H_K : 0, OptionalNone);
394 Point2 cB = K.calibrate(pB_, HK ? &cB_H_K : 0, OptionalNone);
395
396 // convert to homogeneous coordinates
397 Vector3 vA = EssentialMatrix::Homogeneous(p: cA);
398 Vector3 vB = EssentialMatrix::Homogeneous(p: cB);
399
400 if (HK) {
401 // compute the jacobian of error w.r.t K
402
403 // error function f = vA.T * E * vB
404 // H2 = df/dK = vB.T * E.T * dvA/dK + vA.T * E * dvB/dK
405 // where dvA/dK = dvA/dcA * dcA/dK, dVB/dK = dvB/dcB * dcB/dK
406 // and dvA/dcA = dvB/dcB = [[1, 0], [0, 1], [0, 0]]
407 Matrix DynamicH_K = vB.transpose() * E.matrix().transpose().leftCols<2>() * cA_H_K +
408 vA.transpose() * E.matrix().leftCols<2>() * cB_H_K; // (1*2) * (2*DimK)
409 *HK = DynamicH_K;
410 }
411
412 Vector error(1);
413 error(0) = E.error(vA, vB, H: HE);
414
415 return error;
416 }
417
418 public:
419 GTSAM_MAKE_ALIGNED_OPERATOR_NEW
420};

423/**
424 * Binary factor that optimizes for E and two calibrations Ka and Kb using the
425 * algebraic epipolar error (Ka^-1 pA)'E (Kb^-1 pB). The calibrations are
426 * assumed different for the two images, but if you use the same key for Ka and
427 * Kb, the sum of the two K Jacobians equals that of the K Jacobian for
428 * EssentialMatrix4. If you know there is a single global calibration, use
429 * that factor instead.
430 *
431 * Note: see the comment about priors from EssentialMatrixFactor4: even stronger
432 * caveats about having priors on calibration apply here.
433 */
434template <class CALIBRATION>
435class EssentialMatrixFactor5
436 : public NoiseModelFactorN<EssentialMatrix, CALIBRATION, CALIBRATION> {
437 private:
438 Point2 pA_, pB_; ///< points in pixel coordinates
439
440 typedef NoiseModelFactorN<EssentialMatrix, CALIBRATION, CALIBRATION> Base;
441 typedef EssentialMatrixFactor5 This;
442
443 static constexpr int DimK = FixedDimension<CALIBRATION>::value;
444 typedef Eigen::Matrix<double, 2, DimK> JacobianCalibration;
445
446 public:
447 // Provide access to the Matrix& version of evaluateError:
448 using Base::evaluateError;
449
450 /**
451 * Constructor
452 * @param keyE Essential Matrix aEb variable key
453 * @param keyKa Calibration variable key for camera A
454 * @param keyKb Calibration variable key for camera B
455 * @param pA point in first camera, in pixel coordinates
456 * @param pB point in second camera, in pixel coordinates
457 * @param model noise model is about dot product in ideal, homogeneous
458 * coordinates
459 */
460 EssentialMatrixFactor5(Key keyE, Key keyKa, Key keyKb, const Point2& pA,
461 const Point2& pB,
462 const SharedNoiseModel& model = nullptr)
463 : Base(model, keyE, keyKa, keyKb), pA_(pA), pB_(pB) {}
464

466 gtsam::NonlinearFactor::shared_ptr clone() const override {

468 r: gtsam::NonlinearFactor::shared_ptr(new This(*this)));

471 /// print

476 std::cout << " EssentialMatrixFactor5 with measurements\n ("

481 /**
482 * @brief Calculate the algebraic epipolar error pA' (Ka^-1)' E Kb pB.
483 *
484 * @param E essential matrix for key keyE
485 * @param Ka calibration for camera A for key keyKa
486 * @param Kb calibration for camera B for key keyKb
487 * @param H1 optional jacobian of error w.r.t E
488 * @param H2 optional jacobian of error w.r.t Ka
489 * @param H3 optional jacobian of error w.r.t Kb
490 * @return * Vector 1D vector of algebraic error
491 */
492 Vector evaluateError(const EssentialMatrix& E, const CALIBRATION& Ka,
493 const CALIBRATION& Kb, OptionalMatrixType HE,
494 OptionalMatrixType HKa,
495 OptionalMatrixType HKb) const override {
496 // converting from pixel coordinates to normalized coordinates cA and cB
497 JacobianCalibration cA_H_Ka; // dcA/dKa
498 JacobianCalibration cB_H_Kb; // dcB/dKb
499 Point2 cA = Ka.calibrate(pA_, HKa ? &cA_H_Ka : 0, OptionalNone);
500 Point2 cB = Kb.calibrate(pB_, HKb ? &cB_H_Kb : 0, OptionalNone);
501
502 // Convert to homogeneous coordinates.
503 Vector3 vA = EssentialMatrix::Homogeneous(p: cA);
504 Vector3 vB = EssentialMatrix::Homogeneous(p: cB);
505
506 if (HKa) {
507 // Compute the jacobian of error w.r.t Ka.
508 Matrix DynamicHka = vB.transpose() * E.matrix().transpose().leftCols<2>() * cA_H_Ka;
509 *HKa = DynamicHka;
510 }
511
512 if (HKb) {
513 // Compute the jacobian of error w.r.t Kb.
514 Matrix DynamicHkb = vA.transpose() * E.matrix().leftCols<2>() * cB_H_Kb;
515 *HKb = DynamicHkb;
516 }
517
518 Vector error(1);
519 error(0) = E.error(vA, vB, H: HE);
520
521 return error;
522 }
523
524 public:
525 GTSAM_MAKE_ALIGNED_OPERATOR_NEW
526};
