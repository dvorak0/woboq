12/**
13 * @file SmartProjectionRigFactor.h
14 * @brief Smart factor on poses, assuming camera calibration is fixed.
15 * Same as SmartProjectionPoseFactor, except:
16 * - it is templated on CAMERA (i.e., it allows cameras beyond pinhole)
17 * - it admits a different calibration for each measurement (i.e., it
18 * can model a multi-camera rig system)
19 * - it allows multiple observations from the same pose/key (again, to
20 * model a multi-camera system)
21 * @author Luca Carlone
22 * @author Frank Dellaert
23 */

40/**
41 * This factor assumes that camera calibration is fixed (but each measurement
42 * can be taken by a different camera in the rig, hence can have a different
43 * extrinsic and intrinsic calibration). The factor only constrains poses
44 * (variable dimension is 6 for each pose). This factor requires that values
45 * contains the involved poses (Pose3). If all measurements share the same
46 * calibration (i.e., are from the same camera), use SmartProjectionPoseFactor
47 * instead! If the calibration should be optimized, as well, use
48 * SmartProjectionFactor instead!
49 *
50 * <b>Note on Template Parameter `CAMERA`:</b>
51 * While this factor is templated on `CAMERA` to allow for generality (e.g.,
52 * `SphericalCamera`), the current internal implementation for linearization
53 * (specifically, methods like `createHessianFactor` involving Schur complement
54 * calculations inherited or adapted from base classes) has limitations. It
55 * implicitly assumes that the CAMERA only has a Pose3 unknown.
56 *
57 * Consequently:
58 * - This factor works correctly with `CAMERA` types where this is the case,
59 * such as `PinholePose<CALIBRATION>` or `SphericalCamera`.
60 * - Using `CAMERA` types where `dimension != 6`, such as
61 * `PinholeCamera<CALIBRATION>` (which has dimension `6 + CalDim`), will lead
62 * to compilation errors due to matrix dimension mismatches.
63 *
64 * Therefore, for standard pinhole cameras within a fixed rig, use
65 * `PinholePose<CALIBRATION>` as the `CAMERA` template parameter when defining
66 * the `CameraSet` passed to this factor's constructor.
67 *
68 * TODO(dellaert): Refactor the internal linearization logic (e.g., in
69 * `createHessianFactor`) to explicitly compute Jacobians with respect to the
70 * 6-DoF body pose by correctly applying the chain rule, rather than relying on
71 * `traits<CAMERA>::dimension` for downstream calculations.
72 *
73 * @ingroup slam
74 */
75template <class CAMERA>
76class SmartProjectionRigFactor : public SmartProjectionFactor<CAMERA> {
77 private:
78 typedef SmartProjectionFactor<CAMERA> Base;
79 typedef SmartProjectionRigFactor<CAMERA> This;
80 typedef typename CAMERA::CalibrationType CALIBRATION;
81 typedef typename CAMERA::Measurement MEASUREMENT;
82 typedef typename CAMERA::MeasurementVector MEASUREMENTS;
83
84 static const int DimPose = 6; ///< Pose3 dimension
85 static const int ZDim = 2; ///< Measurement dimension
86
87 protected:
88 /// vector of keys (one for each observation) with potentially repeated keys
89 KeyVector nonUniqueKeys_;
90
91 /// cameras in the rig (fixed poses wrt body and intrinsics, for each camera)
92 std::shared_ptr<typename Base::Cameras> cameraRig_;
93
94 /// vector of camera Ids (one for each observation, in the same order),
95 /// identifying which camera took the measurement
96 FastVector<size_t> cameraIds_;
97
98 public:
99 EIGEN_MAKE_ALIGNED_OPERATOR_NEW
100
101 typedef CAMERA Camera;
102 typedef CameraSet<CAMERA> Cameras;
103
104 /// shorthand for a smart pointer to a factor
105 typedef std::shared_ptr<This> shared_ptr;
106
107 /// Default constructor, only for serialization
108 SmartProjectionRigFactor() {}
109
110 /**
111 * Constructor
112 * @param sharedNoiseModel isotropic noise model for the 2D feature
113 * measurements
114 * @param cameraRig set of cameras (fixed poses wrt body and intrinsics) in
115 * the camera rig
116 * @param params parameters for the smart projection factors
117 */
118 SmartProjectionRigFactor(
119 const SharedNoiseModel& sharedNoiseModel,
120 const std::shared_ptr<Cameras>& cameraRig,
121 const SmartProjectionParams& params = SmartProjectionParams())
122 : Base(sharedNoiseModel, params), cameraRig_(cameraRig) {
123 // throw exception if configuration is not supported by this factor
124 if (Base::params_.degeneracyMode != gtsam::ZERO_ON_DEGENERACY)
125 throw std::runtime_error(
126 "SmartProjectionRigFactor: "
127 "degeneracyMode must be set to ZERO_ON_DEGENERACY");
128 if (Base::params_.linearizationMode != gtsam::HESSIAN)
129 throw std::runtime_error(
130 "SmartProjectionRigFactor: "
131 "linearizationMode must be set to HESSIAN");
132 }
133
134 /**
135 * add a new measurement, corresponding to an observation from pose "poseKey"
136 * and taken from the camera in the rig identified by "cameraId"
137 * @param measured 2-dimensional location of the projection of a
138 * single landmark in a single view (the measurement)
139 * @param poseKey key corresponding to the body pose of the camera taking the
140 * measurement
141 * @param cameraId ID of the camera in the rig taking the measurement (default
142 * 0)
143 */
144 void add(const MEASUREMENT& measured, const Key& poseKey,
145 const size_t& cameraId = 0) {
146 // store measurement and key
147 this->measured_.push_back(measured);
148 this->nonUniqueKeys_.push_back(poseKey);
149
150 // also store keys in the keys_ vector: these keys are assumed to be
151 // unique, so we avoid duplicates here
152 if (std::find(this->keys_.begin(), this->keys_.end(), poseKey) ==
153 this->keys_.end())
154 this->keys_.push_back(poseKey); // add only unique keys
155
156 // store id of the camera taking the measurement
157 cameraIds_.push_back(x: cameraId);
158 }
159
160 /**
161 * Variant of the previous "add" function in which we include multiple
162 * measurements
163 * @param measurements vector of the 2m dimensional location of the projection
164 * of a single landmark in the m views (the measurements)
165 * @param poseKeys keys corresponding to the body poses of the cameras taking
166 * the measurements
167 * @param cameraIds IDs of the cameras in the rig taking each measurement
168 * (same order as the measurements)
169 */
170 void add(const MEASUREMENTS& measurements, const KeyVector& poseKeys,
171 const FastVector<size_t>& cameraIds = FastVector<size_t>()) {
172 if (poseKeys.size() != measurements.size() ||
173 (poseKeys.size() != cameraIds.size() && cameraIds.size() != 0)) {
174 throw std::runtime_error(
175 "SmartProjectionRigFactor: "
176 "trying to add inconsistent inputs");
177 }
178 if (cameraIds.size() == 0 && cameraRig_->size() > 1) {
179 throw std::runtime_error(
180 "SmartProjectionRigFactor: "
181 "camera rig includes multiple camera "
182 "but add did not input cameraIds");
183 }
184 for (size_t i = 0; i < measurements.size(); i++) {
185 add(measurements[i], poseKeys[i],
186 cameraIds.size() == 0 ? 0 : cameraIds[i]);
187 }
188 }
189
190 /// return (for each observation) the (possibly non unique) keys involved in
191 /// the measurements
192 const KeyVector& nonUniqueKeys() const { return nonUniqueKeys_; }
193
194 /// return the calibration object
195 const std::shared_ptr<Cameras>& cameraRig() const { return cameraRig_; }
196
197 /// return the calibration object
198 const FastVector<size_t>& cameraIds() const { return cameraIds_; }

200 /**
201 * print
202 * @param s optional string naming the factor
203 * @param keyFormatter optional formatter useful for printing Symbols
204 */
205 void print(
206 const std::string& s = "",
207 const KeyFormatter& keyFormatter = DefaultKeyFormatter) const override {
208 std::cout << s << "SmartProjectionRigFactor: \n ";
209 for (size_t i = 0; i < nonUniqueKeys_.size(); i++) {
210 std::cout << "-- Measurement nr " << i << std::endl;
211 std::cout << "key: " << keyFormatter(nonUniqueKeys_[i]) << std::endl;
212 std::cout << "cameraId: " << cameraIds_[i] << std::endl;
213 (*cameraRig_)[cameraIds_[i]].print("camera in rig:\n");
214 }
215 Base::print("", keyFormatter);
216 }

218 /// equals
219 bool equals(const NonlinearFactor& p, double tol = 1e-9) const override {
220 const This* e = dynamic_cast<const This*>(&p);
221 return e && Base::equals(p, tol) && nonUniqueKeys_ == e->nonUniqueKeys() &&
222 cameraRig_->equals(*(e->cameraRig())) &&
223 std::equal(cameraIds_.begin(), cameraIds_.end(),
224 e->cameraIds().begin());
225 }
226
227 /**
228 * Collect all cameras involved in this factor
229 * @param values Values structure which must contain body poses corresponding
230 * to keys involved in this factor
231 * @return vector of cameras
232 */
233 typename Base::Cameras cameras(const Values& values) const override {
234 typename Base::Cameras cameras;
235 cameras.reserve(nonUniqueKeys_.size()); // preallocate
236 for (size_t i = 0; i < nonUniqueKeys_.size(); i++) {
237 const typename Base::Camera& camera_i = (*cameraRig_)[cameraIds_[i]];
238 const Pose3 world_P_sensor_i =
239 values.at<Pose3>(j: nonUniqueKeys_[i]) // = world_P_body
240 * camera_i.pose(); // = body_P_cam_i
241 cameras.emplace_back(world_P_sensor_i,
242 make_shared<typename CAMERA::CalibrationType>(
243 camera_i.calibration()));
244 }
245 return cameras;
246 }

248 /**
249 * error calculates the error of the factor.
250 */
251 double error(const Values& values) const override {
252 if (this->active(values)) {
253 return this->totalReprojectionError(this->cameras(values));
254 } else { // else of active flag
255 return 0.0;
256 }
257 }
258
259 /**
260 * Compute jacobian F, E and error vector at a given linearization point
261 * @param values Values structure which must contain camera poses
262 * corresponding to keys involved in this factor
263 * @return Return arguments are the camera jacobians Fs (including the
264 * jacobian with respect to both body poses we interpolate from), the point
265 * Jacobian E, and the error vector b. Note that the jacobians are computed
266 * for a given point.
267 */
268 void computeJacobiansWithTriangulatedPoint(typename Base::FBlocks& Fs,
269 Matrix& E, Vector& b,
270 const Cameras& cameras) const {
271 if (!this->result_) {
272 throw("computeJacobiansWithTriangulatedPoint");
273 } else { // valid result: compute jacobians
274 b = -cameras.reprojectionError(*this->result_, this->measured_, Fs, E);
275 for (size_t i = 0; i < Fs.size(); i++) {
276 const Pose3& body_P_sensor = (*cameraRig_)[cameraIds_[i]].pose();
277 const Pose3 world_P_body = cameras[i].pose() * body_P_sensor.inverse();
278 Eigen::Matrix<double, DimPose, DimPose> H;
279 world_P_body.compose(g: body_P_sensor, H1: H);
280 Fs.at(i) = Fs.at(i) * H;
281 }
282 }
283 }

285 /// linearize and return a Hessianfactor that is an approximation of error(p)
286 std::shared_ptr<RegularHessianFactor<DimPose> > createHessianFactor(
287 const Values& values, const double& lambda = 0.0,
288 bool diagonalDamping = false) const {
289 // we may have multiple observation sharing the same keys (e.g., 2 cameras
290 // measuring from the same body pose), hence the number of unique keys may
291 // be smaller than nrMeasurements
292 size_t nrUniqueKeys =
293 this->keys_
294 .size(); // note: by construction, keys_ only contains unique keys
295
296 Cameras cameras = this->cameras(values);
297
298 // Create structures for Hessian Factors
299 std::vector<size_t> js;
300 std::vector<Matrix> Gs(nrUniqueKeys * (nrUniqueKeys + 1) / 2);
301 std::vector<Vector> gs(nrUniqueKeys);
302
303 if (this->measured_.size() != cameras.size()) // 1 observation per camera
304 throw std::runtime_error(
305 "SmartProjectionRigFactor: "
306 "measured_.size() inconsistent with input");
307
308 // triangulate 3D point at given linearization point
309 this->triangulateSafe(cameras);
310
311 if (!this->result_) { // failed: return "empty/zero" Hessian
312 if (this->params_.degeneracyMode == ZERO_ON_DEGENERACY) {
313 for (Matrix& m : Gs) m = Matrix::Zero(rows: DimPose, cols: DimPose);
314 for (Vector& v : gs) v = Vector::Zero(size: DimPose);
315 return std::make_shared<RegularHessianFactor<DimPose> >(this->keys_,
316 Gs, gs, 0.0);
317 } else {
318 throw std::runtime_error(
319 "SmartProjectionRigFactor: "
320 "only supported degeneracy mode is ZERO_ON_DEGENERACY");
321 }
322 }
323
324 // compute Jacobian given triangulated 3D Point
325 typename Base::FBlocks Fs;
326 Matrix E;
327 Vector b;
328 this->computeJacobiansWithTriangulatedPoint(Fs, E, b, cameras);
329
330 // Whiten using noise model
331 this->noiseModel_->WhitenSystem(E, b);
332 for (size_t i = 0; i < Fs.size(); i++) {
333 Fs[i] = this->noiseModel_->Whiten(Fs[i]);
334 }
335
336 const Matrix3 P = Base::Cameras::PointCov(E, lambda, diagonalDamping);
337
338 // Build augmented Hessian (with last row/column being the information
339 // vector) Note: we need to get the augumented hessian wrt the unique keys
340 // in key_
341 SymmetricBlockMatrix augmentedHessianUniqueKeys =
342 Base::Cameras::template SchurComplementAndRearrangeBlocks<3, 6, 6>(
343 Fs, E, P, b, nonUniqueKeys_, this->keys_);
344
345 return std::make_shared<RegularHessianFactor<DimPose> >(
346 this->keys_, augmentedHessianUniqueKeys);
347 }
348
349 /**
350 * Linearize to Gaussian Factor (possibly adding a damping factor Lambda for
351 * LM)
352 * @param values Values structure which must contain camera poses and
353 * extrinsic pose for this factor
354 * @return a Gaussian factor
355 */
356 std::shared_ptr<GaussianFactor> linearizeDamped(
357 const Values& values, const double& lambda = 0.0) const {
358 // depending on flag set on construction we may linearize to different
359 // linear factors
360 switch (this->params_.linearizationMode) {
361 case HESSIAN:
362 return this->createHessianFactor(values, lambda);
363 default:
364 throw std::runtime_error(
365 "SmartProjectionRigFactor: unknown linearization mode");
366 }
367 }
368
369 /// linearize
370 std::shared_ptr<GaussianFactor> linearize(
371 const Values& values) const override {
372 return this->linearizeDamped(values);
373 }

384 // ar& BOOST_SERIALIZATION_NVP(cameraIds_);
385 }
386#endif
