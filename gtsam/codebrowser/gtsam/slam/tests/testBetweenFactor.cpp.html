22/**
23 * This TEST should fail. If you want it to pass, change noise to 0.
24 */
25TEST(BetweenFactor, Rot3) {
26 Rot3 R1 = Rot3::Rodrigues(wx: 0.1, wy: 0.2, wz: 0.3);
27 Rot3 R2 = Rot3::Rodrigues(wx: 0.4, wy: 0.5, wz: 0.6);
28 Rot3 noise = Rot3(); // Rot3::Rodrigues(0.01, 0.01, 0.01); // Uncomment to make unit test fail
29 Rot3 measured = R1.between(g: R2)*noise ;
30
31 BetweenFactor<Rot3> factor(R(j: 1), R(j: 2), measured, Isotropic::Sigma(dim: 3, sigma: 0.05));
32 Matrix actualH1, actualH2;
33 Vector actual = factor.evaluateError(x: R1, x: R2, H&: actualH1, H&: actualH2);
34
35 Vector expected = Rot3::Logmap(R: measured.inverse() * R1.between(g: R2));
36 EXPECT(assert_equal(expected,actual/*, 1e-100*/)); // Uncomment to make unit test fail
37
38 Matrix numericalH1 = numericalDerivative21<Vector3, Rot3, Rot3>(
39 h: [&factor](const Rot3& r1, const Rot3& r2) {return factor.evaluateError(x: r1, x: r2);},
40 x1: R1, x2: R2, delta: 1e-5);
41 EXPECT(assert_equal(numericalH1,actualH1, 1E-5));
42
43 Matrix numericalH2 = numericalDerivative22<Vector3,Rot3,Rot3>(
44 h: [&factor](const Rot3& r1, const Rot3& r2) {return factor.evaluateError(x: r1, x: r2);},
45 x1: R1, x2: R2, delta: 1e-5);
46 EXPECT(assert_equal(numericalH2,actualH2, 1E-5));
47}

74TEST(BetweenFactor, Point3Jacobians) {
75 SharedNoiseModel model = noiseModel::Isotropic::Sigma(dim: 3, sigma: 1.0);
76 Point3 measured(1, 2, 3);
77 BetweenFactor<Point3> factor(1, 2, measured, model);
78
79 Values values;
80 values.insert(j: 1, val: Point3(0, 0, 0));
81 values.insert(j: 2, val: Point3(1, 2, 3));
82 Vector3 error = factor.evaluateError(x: Point3(0, 0, 0), x: Point3(1, 2, 3));
83 EXPECT(assert_equal<Vector3>(Vector3::Zero(), error, 1e-9));
84 EXPECT_CORRECT_FACTOR_JACOBIANS(factor, values, 1e-7, 1e-5);
85}
