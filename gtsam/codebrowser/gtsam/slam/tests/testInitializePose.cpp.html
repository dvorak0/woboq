34TEST(InitializePose3, computePoses2D) {
35 const string g2oFile = findExampleDataFile(name: "noisyToyGraph.txt");
36 bool is3D = false;
37 const auto [inputGraph, posesInFile] = readG2o(g2oFile, is3D);
38
39 auto priorModel = noiseModel::Unit::Create(dim: 3);
40 inputGraph->addPrior(key: 0, prior: posesInFile->at<Pose2>(j: 0), model: priorModel);
41
42 auto poseGraph = initialize::buildPoseGraph<Pose2>(graph: *inputGraph);
43
44 auto I = genericValue(v: Rot3());
45 Values orientations;
46 for (size_t i : {0, 1, 2, 3})
47 orientations.insert(j: i, val: posesInFile->at<Pose2>(j: i).rotation());
48 const Values poses = initialize::computePoses<Pose2>(initialRot: orientations, posegraph: &poseGraph);
49
50 // posesInFile is seriously noisy, so we check error of recovered poses
51 EXPECT_DOUBLES_EQUAL(0.0810283, inputGraph->error(poses), 1e-6);
52}

55TEST(InitializePose3, computePoses3D) {
56 const string g2oFile = findExampleDataFile(name: "Klaus3");
57 bool is3D = true;
58 const auto [inputGraph, posesInFile] = readG2o(g2oFile, is3D);
59
60 auto priorModel = noiseModel::Unit::Create(dim: 6);
61 inputGraph->addPrior(key: 0, prior: posesInFile->at<Pose3>(j: 0), model: priorModel);
62
63 auto poseGraph = initialize::buildPoseGraph<Pose3>(graph: *inputGraph);
64
65 auto I = genericValue(v: Rot3());
66 Values orientations;
67 for (size_t i : {0, 1, 2})
68 orientations.insert(j: i, val: posesInFile->at<Pose3>(j: i).rotation());
69 Values poses = initialize::computePoses<Pose3>(initialRot: orientations, posegraph: &poseGraph);
70 EXPECT(assert_equal(*posesInFile, poses, 0.1)); // TODO(frank): very loose !!
71}
