32/* ************************************************************************* */
33TEST( EssentialMatrixConstraint, test ) {
34 // Create a factor
35 Key poseKey1(1);
36 Key poseKey2(2);
37 Rot3 trueRotation = Rot3::RzRyRx(x: 0.15, y: 0.15, z: -0.20);
38 Point3 trueTranslation(+0.5, -1.0, +1.0);
39 Unit3 trueDirection(trueTranslation);
40 EssentialMatrix measurement(trueRotation, trueDirection);
41 SharedNoiseModel model = noiseModel::Isotropic::Sigma(dim: 5, sigma: 0.25);
42 EssentialMatrixConstraint factor(poseKey1, poseKey2, measurement, model);
43
44 // Create a linearization point at the zero-error point
45 Pose3 pose1(Rot3::RzRyRx(x: 0.00, y: -0.15, z: 0.30), Point3(-4.0, 7.0, -10.0));
46 Pose3 pose2(
47 Rot3::RzRyRx(x: 0.179693265735950, y: 0.002945368776519, z: 0.102274823253840),
48 Point3(-3.37493895, 6.14660244, -8.93650986));
49
50 Vector expected = Z_5x1;
51 Vector actual = factor.evaluateError(x: pose1,x: pose2);
52 CHECK(assert_equal(expected, actual, 1e-8));
53
54 // Calculate numerical derivatives
55 Matrix expectedH1 = numericalDerivative11<Vector5, Pose3>(
56 h: [&factor, &pose2](const Pose3& p1) {return factor.evaluateError(x: p1, x: pose2);},
57 x: pose1);
58
59 Matrix expectedH2 = numericalDerivative11<Vector5, Pose3>(
60 h: [&factor, &pose1](const Pose3& p2) {return factor.evaluateError(x: pose1, x: p2);},
61 x: pose2);
62
63 // Use the factor to calculate the derivative
64 Matrix actualH1;
65 Matrix actualH2;
66 factor.evaluateError(x: pose1, x: pose2, H&: actualH1, H&: actualH2);
67
68 // Verify we get the expected error
69 CHECK(assert_equal(expectedH1, actualH1, 1e-5));
70 CHECK(assert_equal(expectedH2, actualH2, 1e-5));
71}
72
73/* ************************************************************************* */
74int main() {
75 TestResult tr;
76 return TestRegistry::runAllTests(result&: tr);
77}
