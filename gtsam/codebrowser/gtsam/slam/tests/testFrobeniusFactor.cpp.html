65TEST(FrobeniusPriorSO3, ClosestTo) {
66 // Example top-left of SO(4) matrix not quite on SO(3) manifold
67 Matrix3 M;
68 M << 0.79067393, 0.6051136, -0.0930814, //
69 0.4155925, -0.64214347, -0.64324489, //
70 -0.44948549, 0.47046326, -0.75917576;
71
72 SO3 expected = SO3::ClosestTo(M);
73
74 // manifold optimization gets same result as SVD solution in ClosestTo
75 NonlinearFactorGraph graph;
76 graph.emplace_shared<FrobeniusPrior<SO3> >(args: 1, args&: M);
77
78 Values initial;
79 initial.insert(j: 1, val: SO3(I_3x3));
80 auto result = GaussNewtonOptimizer(graph, initial).optimize();
81 EXPECT_DOUBLES_EQUAL(0.0, graph.error(result), 1e-6);
82 EXPECT(assert_equal(expected, result.at<SO3>(1), 1e-6));
83}

85/* ************************************************************************* */
86TEST(FrobeniusPriorSO3, ChordalL2mean) {
87 // See Hartley13ijcv:
88 // Cost function C(R) = \sum FrobeniusPrior(R,R_i)
89 // Closed form solution = ClosestTo(C_e), where
90 // C_e = \sum R_i !!!!
91
92 // We will test by computing mean of R1=exp(v1) R1^T=exp(-v1):
93 using namespace ::so3;
94 SO3 expected; // identity
95 Matrix3 M = R1.matrix() + R1.matrix().transpose();
96 EXPECT(assert_equal(expected, SO3::ClosestTo(M), 1e-6));
97 EXPECT(assert_equal(expected, SO3::ChordalMean({ R1, R1.inverse() }), 1e-6));
98
99 // manifold optimization gets same result as ChordalMean
100 NonlinearFactorGraph graph;
101 graph.emplace_shared<FrobeniusPrior<SO3> >(args: 1, args: R1.matrix());
102 graph.emplace_shared<FrobeniusPrior<SO3> >(args: 1, args: R1.matrix().transpose());
103
104 Values initial;
105 initial.insert<SO3>(j: 1, val: R1.inverse());
106 auto result = GaussNewtonOptimizer(graph, initial).optimize();
107 EXPECT_DOUBLES_EQUAL(0.0, graph.error(result), 0.1); // Why so loose?
108 EXPECT(assert_equal(expected, result.at<SO3>(1), 1e-5));
109}

307TEST(FrobeniusBetweenFactorNLSimilarity2, Optimization) {
308 using namespace ::sim2;
309 auto factor = FrobeniusBetweenFactorNL<Similarity2>(1, 2, P1.between(g: P2));
310
311 NonlinearFactorGraph graph;
312 graph.add(factorOrContainer: factor);
313 graph.add(factorOrContainer: PriorFactor<Similarity2>(1, P1, noiseModel::Constrained::All(dim: 4)));
314
315 Values initial;
316 initial.insert(j: 1, val: P1);
317 initial.insert(j: 2, val: P2.retract(v: Vector4::Constant(value: 0.1)));
318
319 double initial_error = graph.error(values: initial);
320
321 GaussNewtonParams params;
322 // params.setVerbosity("ERROR");
323 GaussNewtonOptimizer optimizer(graph, initial, params);
324 Values result = optimizer.optimize();
325
326 double final_error = graph.error(values: result);
327
328 EXPECT(final_error < initial_error);
329}
