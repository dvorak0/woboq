102TEST(PlanarProjectionFactor1, Jacobian) {
103 // Verify Jacobians with numeric derivative
104 std::default_random_engine rng(42);
105 std::uniform_real_distribution<double> dist(-0.3, 0.3);
106 SharedNoiseModel model = noiseModel::Diagonal::Sigmas(sigmas: Vector2(1, 1));
107 // center of the random camera poses
108 Pose3 centerOffset(
109 Rot3(0, 0, 1,//
110 -1, 0, 0, //
111 0, -1, 0),
112 Vector3(0, 0, 0)
113 );
114
115 for (int i = 0; i < 1000; ++i) {
116 Point3 landmark(2 + dist(rng), dist(rng), dist(rng));
117 Point2 measured(200 + 100 * dist(rng), 200 + 100 * dist(rng));
118 Pose3 offset = centerOffset.compose(
119 g: Pose3(
120 Rot3::Ypr(y: dist(rng), p: dist(rng), r: dist(rng)),
121 Point3(dist(rng), dist(rng), dist(rng))));
122 Cal3DS2 calib(200, 200, 0, 200, 200, -0.2, 0.1);
123 PlanarProjectionFactor1 factor(X(j: 0), landmark, measured, offset, calib, model);
124 Pose2 pose(dist(rng), dist(rng), dist(rng));
125 Matrix H1;
126 factor.evaluateError(x: pose, H&: H1);
127 auto expectedH1 = numericalDerivative11<Vector, Pose2>(
128 h: [&factor](const Pose2& p) {
129 return factor.evaluateError(wTb: p, HwTb: {});},
130 x: pose);
131 CHECK(assert_equal(expectedH1, H1, 5e-6));
132 }
133}

136TEST(PlanarProjectionFactor1, Solve) {
137 // Example localization
138
139 SharedNoiseModel pxModel = noiseModel::Diagonal::Sigmas(sigmas: Vector2(1, 1));
140 // pose model is wide, so the solver finds the right answer.
141 SharedNoiseModel xNoise = noiseModel::Diagonal::Sigmas(sigmas: Vector3(10, 10, 10));
142
143 // landmarks
144 Point3 l0(1, 0.1, 1);
145 Point3 l1(1, -0.1, 1);
146
147 // camera pixels
148 Point2 p0(180, 0);
149 Point2 p1(220, 0);
150
151 // body
152 Pose2 x0(0, 0, 0);
153
154 // camera z looking at +x with (xy) antiparallel to (yz)
155 Pose3 c0(
156 Rot3(0, 0, 1, //
157 -1, 0, 0, //
158 0, -1, 0), //
159 Vector3(0, 0, 0));
160 Cal3DS2 calib(200, 200, 0, 200, 200, 0, 0);
161
162 NonlinearFactorGraph graph;
163 graph.add(factorOrContainer: PlanarProjectionFactor1(X(j: 0), l0, p0, c0, calib, pxModel));
164 graph.add(factorOrContainer: PlanarProjectionFactor1(X(j: 0), l1, p1, c0, calib, pxModel));
165 graph.add(factorOrContainer: PriorFactor<Pose2>(X(j: 0), x0, xNoise));
166
167 Values initialEstimate;
168 initialEstimate.insert(j: X(j: 0), val: x0);
169
170 // run the optimizer
171 LevenbergMarquardtOptimizer optimizer(graph, initialEstimate);
172 Values result = optimizer.optimize();
173
174 // verify that the optimizer found the right pose.
175 CHECK(assert_equal(x0, result.at<Pose2>(X(0)), 2e-3));
176
177 // covariance
178 Marginals marginals(graph, result);
179 Matrix cov = marginals.marginalCovariance(variable: X(j: 0));
180 CHECK(assert_equal((Matrix33() << //
181 0.000012, 0.000000, 0.000000, //
182 0.000000, 0.001287, -.001262, //
183 0.000000, -.001262, 0.001250).finished(), cov, 3e-6));
184
185 // pose stddev
186 Vector3 sigma = cov.diagonal().cwiseSqrt();
187 CHECK(assert_equal((Vector3() << //
188 0.0035,
189 0.0359,
190 0.0354
191 ).finished(), sigma, 1e-4));
192
193}

297 );
298
299 for (int i = 0; i < 1000; ++i) {
300 Point3 landmark(2 + dist(rng), dist(rng), dist(rng));
301 Point2 measured(200 + 100 * dist(rng), 200 + 100 * dist(rng));
302 Pose3 offset = centerOffset.compose(
303 g: Pose3(
304 Rot3::Ypr(y: dist(rng), p: dist(rng), r: dist(rng)),
305 Point3(dist(rng), dist(rng), dist(rng))));
306 Cal3DS2 calib(200, 200, 0, 200, 200, -0.2, 0.1);
307
308 PlanarProjectionFactor3 factor(X(j: 0), C(j: 0), K(j: 0), landmark, measured, model);
309
310 Pose2 pose(dist(rng), dist(rng), dist(rng));
311
312 // actual H
313 Matrix H1, H2, H3;
314 factor.evaluateError(x: pose, x: offset, x: calib, H&: H1, H&: H2, H&: H3);
315

317 h: [&factor](const Pose2& p, const Pose3& o, const Cal3DS2& c) {
318 return factor.evaluateError(wTb: p, bTc: o, calib: c, HwTb: {}, HbTc: {}, Hcalib: {});},
319 x1: pose, x2: offset, x3: calib);
320 Matrix expectedH2 = numericalDerivative32<Vector, Pose2, Pose3, Cal3DS2>(
321 h: [&factor](const Pose2& p, const Pose3& o, const Cal3DS2& c) {
322 return factor.evaluateError(wTb: p, bTc: o, calib: c, HwTb: {}, HbTc: {}, Hcalib: {});},
323 x1: pose, x2: offset, x3: calib);
324 Matrix expectedH3 = numericalDerivative33<Vector, Pose2, Pose3, Cal3DS2>(
325 h: [&factor](const Pose2& p, const Pose3& o, const Cal3DS2& c) {
326 return factor.evaluateError(wTb: p, bTc: o, calib: c, HwTb: {}, HbTc: {}, Hcalib: {});},
327 x1: pose, x2: offset, x3: calib);

329 CHECK(assert_equal(expectedH2, H2, 5e-6));

335TEST(PlanarProjectionFactor3, SolveOffset) {
336 // Example localization
337 SharedNoiseModel pxModel = noiseModel::Diagonal::Sigmas(sigmas: Vector2(1, 1));
338 SharedNoiseModel xNoise = noiseModel::Diagonal::Sigmas(sigmas: Vector3(0.01, 0.01, 0.01));
339 // offset model is wide, so the solver finds the right answer.
340 SharedNoiseModel cNoise = noiseModel::Diagonal::Sigmas(sigmas: Vector6(10, 10, 10, 10, 10, 10));
341 SharedNoiseModel kNoise = noiseModel::Diagonal::Sigmas(sigmas: Vector9(0.001, 0.001, 0.001, 0.001, 0.001, 0.001, 0.001, 0.001, 0.001));
342
343 // landmarks
344 Point3 l0(1, 0, 1);
345 Point3 l1(1, 0, 0);
346 Point3 l2(1, -1, 1);
347 Point3 l3(2, 2, 1);
348
349 // camera pixels
350 Point2 p0(200, 200);
351 Point2 p1(200, 400);
352 Point2 p2(400, 200);
353 Point2 p3(0, 200);
354
355 // body
356 Pose2 x0(0, 0, 0);
357
358 // camera z looking at +x with (xy) antiparallel to (yz)
359 Pose3 c0(
360 Rot3(0, 0, 1, //
361 -1, 0, 0, //
362 0, -1, 0), //
363 Vector3(0, 0, 1)); // note z offset
364 Cal3DS2 calib(200, 200, 0, 200, 200, 0, 0);
365
366 NonlinearFactorGraph graph;
367 graph.add(factorOrContainer: PlanarProjectionFactor3(X(j: 0), C(j: 0), K(j: 0), l0, p0, pxModel));
368 graph.add(factorOrContainer: PlanarProjectionFactor3(X(j: 0), C(j: 0), K(j: 0), l1, p1, pxModel));
369 graph.add(factorOrContainer: PlanarProjectionFactor3(X(j: 0), C(j: 0), K(j: 0), l2, p2, pxModel));
370 graph.add(factorOrContainer: PlanarProjectionFactor3(X(j: 0), C(j: 0), K(j: 0), l3, p3, pxModel));
371 graph.add(factorOrContainer: PriorFactor<Pose2>(X(j: 0), x0, xNoise));
372 graph.add(factorOrContainer: PriorFactor<Pose3>(C(j: 0), c0, cNoise));
373 graph.add(factorOrContainer: PriorFactor<Cal3DS2>(K(j: 0), calib, kNoise));
374
375 Values initialEstimate;
376 initialEstimate.insert(j: X(j: 0), val: x0);
377 initialEstimate.insert(j: C(j: 0), val: c0);
378 initialEstimate.insert(j: K(j: 0), val: calib);
379
380 // run the optimizer
381 LevenbergMarquardtOptimizer optimizer(graph, initialEstimate);
382 Values result = optimizer.optimize();
383
384 // verify that the optimizer found the right pose.
385 CHECK(assert_equal(x0, result.at<Pose2>(X(0)), 2e-3));
386
387 // verify the camera is pointing at +x
388 Pose3 cc0 = result.at<Pose3>(j: C(j: 0));
389 CHECK(assert_equal(c0, cc0, 5e-3));
390
391 // verify the calibration
392 CHECK(assert_equal(calib, result.at<Cal3DS2>(K(0)), 2e-3));
393
394 Marginals marginals(graph, result);
395 Matrix x0cov = marginals.marginalCovariance(variable: X(j: 0));
396
397 // narrow prior => ~zero cov
398 CHECK(assert_equal(Matrix33::Zero(), x0cov, 1e-4));
399
400 Matrix c0cov = marginals.marginalCovariance(variable: C(j: 0));
401
402 // invert the camera offset to get covariance in body coordinates
403 Matrix66 HcTb = cc0.inverse().AdjointMap().inverse();
404 Matrix c0cov2 = HcTb * c0cov * HcTb.transpose();
405
406 // camera-frame stddev
407 Vector6 c0sigma = c0cov.diagonal().cwiseSqrt();
408 CHECK(assert_equal((Vector6() << //
409 0.009,
410 0.011,
411 0.004,
412 0.012,
413 0.012,
414 0.011
415 ).finished(), c0sigma, 1e-3));
416
417 // body frame stddev
418 Vector6 bTcSigma = c0cov2.diagonal().cwiseSqrt();
419 CHECK(assert_equal((Vector6() << //
420 0.004,
421 0.009,
422 0.011,
423 0.012,
424 0.012,
425 0.012
426 ).finished(), bTcSigma, 1e-3));
427
428 // narrow prior => ~zero cov
429 CHECK(assert_equal(Matrix99::Zero(), marginals.marginalCovariance(K(0)), 3e-3));
430}
