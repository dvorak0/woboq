112 // Project two landmarks into two cameras

116 SmartFactor factor(model, sharedK);
117 factor.add(measured: level_uv, key: x1);
118 factor.add(measured: level_uv_right, key: x2);
119
120 Values values; // it's a pose factor, hence these are poses
121 values.insert(j: x1, val: cam1.pose());
122 values.insert(j: x2, val: cam2.pose());
123
124 double actualError = factor.error(values);
125 double expectedError = 0.0;
126 EXPECT_DOUBLES_EQUAL(expectedError, actualError, 1e-7);
127
128 SmartFactor::Cameras cameras = factor.cameras(values);
129 double actualError2 = factor.totalReprojectionError(cameras);
130 EXPECT_DOUBLES_EQUAL(expectedError, actualError2, 1e-7);
131
132 // Calculate expected derivative for point (easiest to check)
133 std::function<Vector(Point3)> f = //
134 std::bind(f: &SmartFactor::whitenedError<Point3>, args&: factor, args&: cameras, args: std::placeholders::_1);
135
136 // Calculate using computeEP
137 Matrix actualE;
138 factor.triangulateAndComputeE(E&: actualE, values);
139
140 // get point
141 auto point = factor.point();
142 CHECK(point);
143
144 // calculate numerical derivative with triangulated point
145 Matrix expectedE = sigma * numericalDerivative11<Vector, Point3>(h: f, x: *point);
146 EXPECT(assert_equal(expectedE, actualE, 1e-7));
147
148 // Calculate using reprojectionError
149 SmartFactor::Cameras::FBlocks F;
150 Matrix E;
151 Vector actualErrors = factor.unwhitenedError(cameras, point: *point, optArgs&: F, optArgs&: E);
152 EXPECT(assert_equal(expectedE, E, 1e-7));
153
154 EXPECT(assert_equal(Z_4x1, actualErrors, 1e-7));
155
156 // Calculate using computeJacobians
157 Vector b;
158 SmartFactor::FBlocks Fs;
159 factor.computeJacobians(Fs, E, b, cameras, point: *point);
160 double actualError3 = b.squaredNorm();
161 EXPECT(assert_equal(expectedE, E, 1e-7));
162 EXPECT_DOUBLES_EQUAL(expectedError, actualError3, 1e-6);
163}

221

223 KeyVector views {x1, x2, x3};
224

226 params.setRankTolerance(1.0);
227 params.setDegeneracyMode(IGNORE_DEGENERACY);

243 graph.push_back(factor: smartFactor1);
244 graph.push_back(factor: smartFactor2);
245 graph.push_back(factor: smartFactor3);
246 graph.addPrior(key: x1, prior: wTb1, model: noisePrior);
247 graph.addPrior(key: x2, prior: wTb2, model: noisePrior);
248
249 // Check errors at ground truth poses
250 Values gtValues;
251 gtValues.insert(j: x1, val: wTb1);
252 gtValues.insert(j: x2, val: wTb2);
253 gtValues.insert(j: x3, val: wTb3);
254 double actualError = graph.error(values: gtValues);
255 double expectedError = 0.0;
256 DOUBLES_EQUAL(expectedError, actualError, 1e-7)
257
258 Pose3 noise_pose = Pose3(Rot3::Ypr(y: -M_PI / 100, p: 0., r: -M_PI / 100),
259 Point3(0.1, 0.1, 0.1));
260 Values values;
261 values.insert(j: x1, val: wTb1);
262 values.insert(j: x2, val: wTb2);
263 // initialize third pose with some noise, we expect it to move back to
264 // original pose3
265 values.insert(j: x3, val: wTb3 * noise_pose);
266
267 LevenbergMarquardtParams lmParams;
268 Values result;
269 LevenbergMarquardtOptimizer optimizer(graph, values, lmParams);
270 result = optimizer.optimize();
271 EXPECT(assert_equal(wTb3, result.at<Pose3>(x3)));
272}
