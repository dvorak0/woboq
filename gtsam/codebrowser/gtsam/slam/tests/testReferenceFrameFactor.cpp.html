55Vector evaluateError_(const PointReferenceFrameFactor& c,
56 const Point2& global, const Pose2& trans, const Point2& local) {
57 return Vector(c.evaluateError(x: global, x: trans, x: local));
58}
59TEST( ReferenceFrameFactor, jacobians ) {
60
61 // from examples below
62 Point2 local(2.0, 3.0), global(-1.0, 2.0);
63 Pose2 trans(1.5, 2.5, 0.3);
64
65 PointReferenceFrameFactor tc(lA1, tA1, lB1);
66 Matrix actualDT, actualDL, actualDF;
67 tc.evaluateError(x: global, x: trans, x: local, H&: actualDF, H&: actualDT, H&: actualDL);
68
69 Matrix numericalDT, numericalDL, numericalDF;
70 numericalDF = numericalDerivative31<Vector,Point2,Pose2,Point2>(
71 h: std::bind(f&: evaluateError_, args&: tc, args: std::placeholders::_1, args: std::placeholders::_2, args: std::placeholders::_3),
72 x1: global, x2: trans, x3: local, delta: 1e-5);
73 numericalDT = numericalDerivative32<Vector,Point2,Pose2,Point2>(
74 h: std::bind(f&: evaluateError_, args&: tc, args: std::placeholders::_1, args: std::placeholders::_2, args: std::placeholders::_3),
75 x1: global, x2: trans, x3: local, delta: 1e-5);
76 numericalDL = numericalDerivative33<Vector,Point2,Pose2,Point2>(
77 h: std::bind(f&: evaluateError_, args&: tc, args: std::placeholders::_1, args: std::placeholders::_2, args: std::placeholders::_3),
78 x1: global, x2: trans, x3: local, delta: 1e-5);
79
80 EXPECT(assert_equal(numericalDF, actualDF));
81 EXPECT(assert_equal(numericalDL, actualDL));
82 EXPECT(assert_equal(numericalDT, actualDT));
83}

86TEST( ReferenceFrameFactor, jacobians_zero ) {
87
88 // get values that are ideal
89 Pose2 trans(2.0, 3.0, 0.0);
90 Point2 global(5.0, 6.0);
91 Point2 local = trans.transformFrom(point: global);
92
93 PointReferenceFrameFactor tc(lA1, tA1, lB1);

95 expCost = Z_2x1;
96 EXPECT(assert_equal(expCost, actCost, 1e-5));
97

99 tc.evaluateError(x: global, x: trans, x: local, H&: actualDF, H&: actualDT, H&: actualDL);
100

103 h: std::bind(f&: evaluateError_, args&: tc, args: std::placeholders::_1, args: std::placeholders::_2, args: std::placeholders::_3),
104 x1: global, x2: trans, x3: local, delta: 1e-5);
105 numericalDT = numericalDerivative32<Vector,Point2,Pose2,Point2>(
106 h: std::bind(f&: evaluateError_, args&: tc, args: std::placeholders::_1, args: std::placeholders::_2, args: std::placeholders::_3),
107 x1: global, x2: trans, x3: local, delta: 1e-5);
108 numericalDL = numericalDerivative33<Vector,Point2,Pose2,Point2>(

112 EXPECT(assert_equal(numericalDF, actualDF));
113 EXPECT(assert_equal(numericalDL, actualDL));

118TEST( ReferenceFrameFactor, converge_trans ) {
119
120 // initial points
121 Point2 local1(2.0, 2.0), local2(4.0, 5.0),
122 global1(-1.0, 5.0), global2(2.0, 3.0);
123 Pose2 transIdeal(7.0, 3.0, M_PI/2);
124
125 // verify direction
126 EXPECT(assert_equal(local1, transIdeal.transformFrom(global1)));
127 EXPECT(assert_equal(local2, transIdeal.transformFrom(global2)));
128
129 // choose transform
130 // Pose2 trans = transIdeal; // ideal - works
131 // Pose2 trans = transIdeal * Pose2(0.1, 1.0, 0.00); // translation - works
132 // Pose2 trans = transIdeal * Pose2(10.1, 1.0, 0.00); // large translation - works
133 // Pose2 trans = transIdeal * Pose2(0.0, 0.0, 0.1); // small rotation - works
134 Pose2 trans = transIdeal * Pose2(-200.0, 100.0, 1.3); // combined - works
135 // Pose2 trans = transIdeal * Pose2(-200.0, 100.0, 2.0); // beyond pi/2 - fails
136
137 NonlinearFactorGraph graph;
138 graph.emplace_shared<PointReferenceFrameFactor>(args&: lB1, args&: tA1, args&: lA1);
139 graph.emplace_shared<PointReferenceFrameFactor>(args&: lB2, args&: tA1, args&: lA2);
140
141 // hard constraints on points
142 double error_gain = 1000.0;
143 graph.emplace_shared<NonlinearEquality<gtsam::Point2> >(args&: lA1, args&: local1, args&: error_gain);
144 graph.emplace_shared<NonlinearEquality<gtsam::Point2> >(args&: lA2, args&: local2, args&: error_gain);
145 graph.emplace_shared<NonlinearEquality<gtsam::Point2> >(args&: lB1, args&: global1, args&: error_gain);
146 graph.emplace_shared<NonlinearEquality<gtsam::Point2> >(args&: lB2, args&: global2, args&: error_gain);
147
148 // create initial estimate
149 Values init;
150 init.insert(j: lA1, val: local1);
151 init.insert(j: lA2, val: local2);
152 init.insert(j: lB1, val: global1);
153 init.insert(j: lB2, val: global2);
154 init.insert(j: tA1, val: trans);
155
156 // optimize
157 LevenbergMarquardtOptimizer solver(graph, init);
158 Values actual = solver.optimize();
159
160 Values expected;
161 expected.insert(j: lA1, val: local1);
162 expected.insert(j: lA2, val: local2);
163 expected.insert(j: lB1, val: global1);
164 expected.insert(j: lB2, val: global2);
165 expected.insert(j: tA1, val: transIdeal);
166
167 EXPECT(assert_equal(expected, actual, 1e-4));
168}

171TEST( ReferenceFrameFactor, converge_local ) {
172
173 // initial points
174 Point2 global(-1.0, 2.0);
175 // Pose2 trans(1.5, 2.5, 0.3); // original
176 // Pose2 trans(1.5, 2.5, 1.0); // larger rotation
177 Pose2 trans(1.5, 2.5, 3.1); // significant rotation
178
179 Point2 idealLocal = trans.transformFrom(point: global);
180
181 // perturb the initial estimate
182 // Point2 local = idealLocal; // Ideal case - works
183 // Point2 local = idealLocal + Point2(1.0, 0.0); // works
184 Point2 local = idealLocal + Point2(-10.0, 10.0); // works
185
186 NonlinearFactorGraph graph;
187 double error_gain = 1000.0;
188 graph.emplace_shared<PointReferenceFrameFactor>(args&: lB1, args&: tA1, args&: lA1);
189 graph.emplace_shared<NonlinearEquality<gtsam::Point2> >(args&: lB1, args&: global, args&: error_gain);
190 graph.emplace_shared<NonlinearEquality<gtsam::Pose2> >(args&: tA1, args&: trans, args&: error_gain);
191
192 // create initial estimate
193 Values init;
194 init.insert(j: lA1, val: local);
195 init.insert(j: lB1, val: global);
196 init.insert(j: tA1, val: trans);
197
198 // optimize
199 LevenbergMarquardtOptimizer solver(graph, init);
200 Values actual = solver.optimize();
201
202 CHECK(actual.exists(lA1));
203 EXPECT(assert_equal(idealLocal, actual.at<Point2>(lA1), 1e-5));
204}

207TEST( ReferenceFrameFactor, converge_global ) {
208
209 // initial points
210 Point2 local(2.0, 3.0);
211 // Pose2 trans(1.5, 2.5, 0.3); // original
212 // Pose2 trans(1.5, 2.5, 1.0); // larger rotation
213 Pose2 trans(1.5, 2.5, 3.1); // significant rotation
214
215 Point2 idealForeign = trans.inverse().transformFrom(point: local);
216
217 // perturb the initial estimate
218 // Point2 global = idealForeign; // Ideal - works
219 // Point2 global = idealForeign + Point2(1.0, 0.0); // simple - works
220 Point2 global = idealForeign + Point2(10.0, -10.0); // larger - works
221
222 NonlinearFactorGraph graph;
223 double error_gain = 1000.0;
224 graph.emplace_shared<PointReferenceFrameFactor>(args&: lB1, args&: tA1, args&: lA1);
225 graph.emplace_shared<NonlinearEquality<gtsam::Point2> >(args&: lA1, args&: local, args&: error_gain);
226 graph.emplace_shared<NonlinearEquality<gtsam::Pose2> >(args&: tA1, args&: trans, args&: error_gain);
227
228 // create initial estimate
229 Values init;
230 init.insert(j: lA1, val: local);
231 init.insert(j: lB1, val: global);
232 init.insert(j: tA1, val: trans);
233
234 // optimize
235 LevenbergMarquardtOptimizer solver(graph, init);
236 Values actual = solver.optimize();
237
238 // verify
239 CHECK(actual.exists(lB1));
240 EXPECT(assert_equal(idealForeign, actual.at<Point2>(lB1), 1e-5));
241}
