33static Symbol x0('x', 0), x1('x', 1), x2('x', 2), x3('x', 3);
34static SharedNoiseModel model(noiseModel::Isotropic::Sigma(dim: 6, sigma: 0.1));
35
36namespace simple {
37// We consider a small graph:
38// symbolic FG
39// x2 0 1
40// / | \ 1 2
41// / | \ 2 3
42// x3 | x1 2 0
43// \ | / 0 3
44// \ | /
45// x0
46//
47static Point3 p0 = Point3(0,0,0);
48static Rot3 R0 = Rot3::Expmap( v: ( Vector(3) << 0.0,0.0,0.0 ).finished() );
49static Point3 p1 = Point3(1,2,0);
50static Rot3 R1 = Rot3::Expmap( v: ( Vector(3) << 0.0,0.0,1.570796 ).finished() );
51static Point3 p2 = Point3(0,2,0);
52static Rot3 R2 = Rot3::Expmap( v: ( Vector(3) << 0.0,0.0,3.141593 ).finished() );
53static Point3 p3 = Point3(-1,1,0);
54static Rot3 R3 = Rot3::Expmap( v: ( Vector(3) << 0.0,0.0,4.712389 ).finished() );
55
56static Pose3 pose0 = Pose3(R0,p0);
57static Pose3 pose1 = Pose3(R1,p1);
58static Pose3 pose2 = Pose3(R2,p2);
59static Pose3 pose3 = Pose3(R3,p3);
60
61NonlinearFactorGraph graph() {
62 NonlinearFactorGraph g;
63 g.add(factorOrContainer: BetweenFactor<Pose3>(x0, x1, pose0.between(g: pose1), model));
64 g.add(factorOrContainer: BetweenFactor<Pose3>(x1, x2, pose1.between(g: pose2), model));
65 g.add(factorOrContainer: BetweenFactor<Pose3>(x2, x3, pose2.between(g: pose3), model));
66 g.add(factorOrContainer: BetweenFactor<Pose3>(x2, x0, pose2.between(g: pose0), model));
67 g.add(factorOrContainer: BetweenFactor<Pose3>(x0, x3, pose0.between(g: pose3), model));
68 g.addPrior(key: x0, prior: pose0, model);
69 return g;
70}
71
72NonlinearFactorGraph graph2() {
73 NonlinearFactorGraph g;
74 g.add(factorOrContainer: BetweenFactor<Pose3>(x0, x1, pose0.between(g: pose1), noiseModel::Isotropic::Precision(dim: 6, precision: 1.0)));
75 g.add(factorOrContainer: BetweenFactor<Pose3>(x1, x2, pose1.between(g: pose2), noiseModel::Isotropic::Precision(dim: 6, precision: 1.0)));
76 g.add(factorOrContainer: BetweenFactor<Pose3>(x2, x3, pose2.between(g: pose3), noiseModel::Isotropic::Precision(dim: 6, precision: 1.0)));
77 // random pose, but zero information
78 auto noise_zero_info = noiseModel::Isotropic::Precision(dim: 6, precision: 0.0);
79 g.add(factorOrContainer: BetweenFactor<Pose3>(
80 x2, x0, Pose3(Rot3::Ypr(y: 0.1, p: 0.0, r: 0.1), Point3(0.0, 0.0, 0.0)),
81 noise_zero_info));
82 // random pose, but zero information
83 g.add(factorOrContainer: BetweenFactor<Pose3>(
84 x0, x3, Pose3(Rot3::Ypr(y: 0.5, p: -0.2, r: 0.2), Point3(10, 20, 30)),
85 noise_zero_info));
86 g.addPrior(key: x0, prior: pose0, model);
87 return g;
88}

173TEST( InitializePose3, iterationGradient ) {
174 NonlinearFactorGraph pose3Graph = InitializePose3::buildPose3graph(graph: simple::graph());
175
176 // Wrong initial guess - initialization should fix the rotations
177 Rot3 Rpert = Rot3::Expmap(v: Vector3(0.01, 0.01, 0.01));
178 Values givenPoses;
179 givenPoses.insert(j: x0,val: simple::pose0);
180 givenPoses.insert(j: x1,val: (simple::pose0).compose( g: Pose3(Rpert,Point3(0,0,0)) ));
181 givenPoses.insert(j: x2, val: (simple::pose0).compose( g: Pose3(Rpert.inverse(),Point3(0,0,0)) ));
182 givenPoses.insert(j: x3, val: (simple::pose0).compose( g: Pose3(Rpert,Point3(0,0,0)) ));
183
184 size_t maxIter = 1; // test gradient at the first iteration
185 bool setRefFrame = false;
186 Values orientations = InitializePose3::computeOrientationsGradient(pose3Graph, givenGuess: givenPoses, maxIter, setRefFrame);
187
188 Matrix M0 = (Matrix(3,3) << 0.999435813876064, -0.033571481675497, 0.001004768630281,
189 0.033572116359134, 0.999436104312325, -0.000621610948719,
190 -0.000983333645009, 0.000654992453817, 0.999999302019670).finished();

193

195 0.010943459008004, 0.999898317528125, -0.009143047050380,
196 -0.008336465609239, 0.009234508232789, 0.999922610604863).finished();
197 Rot3 R1Expected = Rot3(M1);
198 EXPECT(assert_equal(R1Expected, orientations.at<Rot3>(x1), 1e-5));
199
200 Matrix M2 = (Matrix(3,3) << 0.998936644682875, 0.045376417678595, -0.008158469732553,
201 -0.045306446926148, 0.998936408933058, 0.008566024448664,
202 0.008538487960253, -0.008187284445083, 0.999930028850403).finished();
203 Rot3 R2Expected = Rot3(M2);

214TEST( InitializePose3, orientationsGradient ) {
215 NonlinearFactorGraph pose3Graph = InitializePose3::buildPose3graph(graph: simple::graph());
216
217 // Wrong initial guess - initialization should fix the rotations
218 Rot3 Rpert = Rot3::Expmap(v: Vector3(0.01, 0.01, 0.01));
219 Values givenPoses;
220 givenPoses.insert(j: x0,val: simple::pose0);
221 givenPoses.insert(j: x1,val: (simple::pose0).compose( g: Pose3(Rpert,Point3(0,0,0)) ));
222 givenPoses.insert(j: x2, val: (simple::pose0).compose( g: Pose3(Rpert.inverse(),Point3(0,0,0)) ));
223 givenPoses.insert(j: x3, val: (simple::pose0).compose( g: Pose3(Rpert,Point3(0,0,0)) ));
224 // do 10 gradient iterations
225 bool setRefFrame = false;
226 Values orientations = InitializePose3::computeOrientationsGradient(pose3Graph, givenGuess: givenPoses, maxIter: 10, setRefFrame);
227
228 // const Key keyAnchor = symbol('Z', 9999999);
229 // givenPoses.insert(keyAnchor,simple::pose0);
230 // string g2oFile = "/home/aspn/Desktop/toyExample.g2o";
231 // writeG2o(pose3Graph, givenPoses, g2oFile);
232
233 const string matlabResultsfile = findExampleDataFile(name: "simpleGraph10gradIter");
234 bool is3D = true;
235 const auto [matlabGraph, matlabValues] = readG2o(g2oFile: matlabResultsfile, is3D);
236
237 Rot3 R0Expected = matlabValues->at<Pose3>(j: 1).rotation();
238 EXPECT(assert_equal(R0Expected, orientations.at<Rot3>(x0), 1e-4));
239
240 Rot3 R1Expected = matlabValues->at<Pose3>(j: 2).rotation();
241 EXPECT(assert_equal(R1Expected, orientations.at<Rot3>(x1), 1e-4));
242
243 Rot3 R2Expected = matlabValues->at<Pose3>(j: 3).rotation();
244 EXPECT(assert_equal(R2Expected, orientations.at<Rot3>(x2), 1e-3));
245
246 Rot3 R3Expected = matlabValues->at<Pose3>(j: 4).rotation();
247 EXPECT(assert_equal(R3Expected, orientations.at<Rot3>(x3), 1e-4));
248}
