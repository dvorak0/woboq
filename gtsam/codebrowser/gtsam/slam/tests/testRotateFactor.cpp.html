24static const double kDegree = M_PI / 180;

26//*************************************************************************
27// Create some test data
28// Let's assume IMU is aligned with aero (X-forward,Z down)
29// And camera is looking forward.
30static const Point3 cameraX(0, 1, 0), cameraY(0, 0, 1), cameraZ(1, 0, 0);
31static const Rot3 iRc(cameraX, cameraY, cameraZ);
32
33// Now, let's create some rotations around IMU frame
34static const Unit3 p1(1, 0, 0), p2(0, 1, 0), p3(0, 0, 1);
35static const Rot3 i1Ri2 = Rot3::AxisAngle(axis: p1, angle: 1), //
36i2Ri3 = Rot3::AxisAngle(axis: p2, angle: 1), //
37i3Ri4 = Rot3::AxisAngle(axis: p3, angle: 1);
38
39// The corresponding rotations in the camera frame
40static const Rot3 c1Zc2 = iRc.inverse() * i1Ri2 * iRc, //
41c2Zc3 = iRc.inverse() * i2Ri3 * iRc, //
42c3Zc4 = iRc.inverse() * i3Ri4 * iRc;
43
44// The corresponding rotated directions in the camera frame
45static const Unit3 z1 = iRc.inverse() * p1, //
46z2 = iRc.inverse() * p2, //
47z3 = iRc.inverse() * p3;
48
49typedef noiseModel::Isotropic::shared_ptr Model;
50
51//*************************************************************************
52TEST (RotateFactor, checkMath) {
53 EXPECT(assert_equal(c1Zc2, Rot3::AxisAngle(z1, 1)));
54 EXPECT(assert_equal(c2Zc3, Rot3::AxisAngle(z2, 1)));
55 EXPECT(assert_equal(c3Zc4, Rot3::AxisAngle(z3, 1)));
56}
57
58//*************************************************************************
59TEST (RotateFactor, test) {
60 Model model = noiseModel::Isotropic::Sigma(dim: 3, sigma: 0.01);
61 RotateFactor f(1, i1Ri2, c1Zc2, model);
62 EXPECT(assert_equal(Z_3x1, f.evaluateError(iRc), 1e-8));
63
64 Rot3 R = iRc.retract(v: Vector3(0.1, 0.2, 0.1));
65#if defined(GTSAM_ROT3_EXPMAP) || defined(GTSAM_USE_QUATERNIONS)
66 Vector expectedE = Vector3(-0.0248752, 0.202981, -0.0890529);
67#else
68 Vector expectedE = Vector3(-0.0246305, 0.20197, -0.08867);
69#endif
70 EXPECT( assert_equal(expectedE, f.evaluateError(R), 1e-5));
71
72 Matrix actual, expected;
73 // Use numerical derivatives to calculate the expected Jacobian
74 {
75 expected = numericalDerivative11<Vector3, Rot3>(
76 h: [&f](const Rot3& r) { return f.evaluateError(x: r); }, x: iRc);
77 f.evaluateError(x: iRc, H&: actual);
78 EXPECT(assert_equal(expected, actual, 1e-9));
79 }
80 {
81 expected = numericalDerivative11<Vector3, Rot3>(
82 h: [&f](const Rot3& r) { return f.evaluateError(x: r); }, x: R);
83 f.evaluateError(x: R, H&: actual);
84 EXPECT(assert_equal(expected, actual, 1e-9));
85 }
86}

88//*************************************************************************
89TEST (RotateFactor, minimization) {
90 // Let's try to recover the correct iRc by minimizing
91 NonlinearFactorGraph graph;
92 Model model = noiseModel::Isotropic::Sigma(dim: 3, sigma: 0.01);
93 graph.emplace_shared<RotateFactor>(args: 1, args: i1Ri2, args: c1Zc2, args&: model);
94 graph.emplace_shared<RotateFactor>(args: 1, args: i2Ri3, args: c2Zc3, args&: model);
95 graph.emplace_shared<RotateFactor>(args: 1, args: i3Ri4, args: c3Zc4, args&: model);
96
97 // Check error at ground truth
98 Values truth;
99 truth.insert(j: 1, val: iRc);
100 EXPECT_DOUBLES_EQUAL(0, graph.error(truth), 1e-8);
101
102 // Check error at initial estimate
103 Values initial;
104 Rot3 initialE = iRc.retract(v: kDegree * Vector3(20, -20, 20));
105 initial.insert(j: 1, val: initialE);
106
107#if defined(GTSAM_ROT3_EXPMAP) || defined(GTSAM_USE_QUATERNIONS)
108 EXPECT_DOUBLES_EQUAL(3545.40, graph.error(initial), 1);
109#else
110 EXPECT_DOUBLES_EQUAL(3349, graph.error(initial), 1);
111#endif
112
113 // Optimize
114 LevenbergMarquardtParams parameters;
115 //parameters.setVerbosity("ERROR");
116 LevenbergMarquardtOptimizer optimizer(graph, initial, parameters);
117 Values result = optimizer.optimize();
118
119 // Check result
120 Rot3 actual = result.at<Rot3>(j: 1);
121 EXPECT(assert_equal(iRc, actual,1e-1));
122
123 // Check error at result
124 EXPECT_DOUBLES_EQUAL(0, graph.error(result), 1e-4);
125}

198TEST(RotateDirectionsFactor, Initialization) {
199 // Create a gravity vector in a nav frame that has Z up
200 const Point3 n_gravity(0, 0, -10);
201 const Unit3 n_p(-n_gravity);
202
203 // NOTE(frank): avoid singularities by using 85/275 instead of 90/270
204 std::vector<double> angles = {0, 45, 85, 135, 180, 225, 275, 315};
205 for (double yaw : angles) {
206 const Rot3 nRy = Rot3::Yaw(t: yaw * kDegree);
207 for (double pitch : angles) {
208 const Rot3 yRp = Rot3::Pitch(t: pitch * kDegree);
209 for (double roll : angles) {
210 const Rot3 pRb = Rot3::Roll(t: roll * kDegree);
211
212 // Rotation from body to nav frame:
213 const Rot3 nRb = nRy * yRp * pRb;
214 const Vector3 rpy = nRb.rpy() / kDegree;
215
216 // Simulate measurement of IMU in body frame:
217 const Point3 b_acc = nRb.unrotate(p: -n_gravity);
218 const Unit3 b_z(b_acc);
219
220 // Check initialization
221 const Rot3 actual_nRb = RotateDirectionsFactor::Initialize(i_p: n_p, c_z: b_z);
222 const Vector3 actual_rpy = actual_nRb.rpy() / kDegree;
223 EXPECT_DOUBLES_EQUAL(rpy.x(), actual_rpy.x(), 1e-5);
224 EXPECT_DOUBLES_EQUAL(rpy.y(), actual_rpy.y(), 1e-5);
225 }
226 }
227 }
