36static Symbol x0('x', 0), x1('x', 1), x2('x', 2), x3('x', 3);
37static SharedNoiseModel model(noiseModel::Isotropic::Sigma(dim: 3, sigma: 0.1));
38
39namespace simpleLago {
40// We consider a small graph:
41// symbolic FG
42// x2 0 1
43// / | \ 1 2
44// / | \ 2 3
45// x3 | x1 2 0
46// \ | / 0 3
47// \ | /
48// x0
49//
50
51static Pose2 pose0 = Pose2(0.000000, 0.000000, 0.000000);
52static Pose2 pose1 = Pose2(1.000000, 1.000000, 1.570796);
53static Pose2 pose2 = Pose2(0.000000, 2.000000, 3.141593);
54static Pose2 pose3 = Pose2(-1.000000, 1.000000, 4.712389);
55
56NonlinearFactorGraph graph() {
57 NonlinearFactorGraph g;
58 g.add(factorOrContainer: BetweenFactor<Pose2>(x0, x1, pose0.between(g: pose1), model));
59 g.add(factorOrContainer: BetweenFactor<Pose2>(x1, x2, pose1.between(g: pose2), model));
60 g.add(factorOrContainer: BetweenFactor<Pose2>(x2, x3, pose2.between(g: pose3), model));
61 g.add(factorOrContainer: BetweenFactor<Pose2>(x2, x0, pose2.between(g: pose0), model));
62 g.add(factorOrContainer: BetweenFactor<Pose2>(x0, x3, pose0.between(g: pose3), model));
63 g.addPrior(key: x0, prior: pose0, model);
64 return g;
65}
66}

69TEST(Lago, findMinimumSpanningTree) {
70 NonlinearFactorGraph g = simpleLago::graph();
71 auto gPlus = initialize::buildPoseGraph<Pose2>(graph: g);
72 lago::PredecessorMap tree = lago::findMinimumSpanningTree(pose2Graph: gPlus);
73
74 /* We should recover the following spanning tree:
75 x2
76 / \
77 / \
78 x3 x1
79 /
80 /
81 x0
82 |
83 a
84 */
85 using initialize::kAnchorKey;
86 EXPECT_LONGS_EQUAL(kAnchorKey, tree[kAnchorKey]);
87 EXPECT_LONGS_EQUAL(kAnchorKey, tree[x0]);
88 EXPECT_LONGS_EQUAL(x0, tree[x1]);
89 EXPECT_LONGS_EQUAL(x1, tree[x2]);
90 EXPECT_LONGS_EQUAL(x2, tree[x3]);
91}

111TEST(Lago, orientationsOverSpanningTree) {
112 NonlinearFactorGraph g = simpleLago::graph();
113 auto gPlus = initialize::buildPoseGraph<Pose2>(graph: g);
114 lago::PredecessorMap tree = lago::findMinimumSpanningTree(pose2Graph: gPlus);
115
116 // check the tree structure
117 using initialize::kAnchorKey;
118
119 EXPECT_LONGS_EQUAL(kAnchorKey, tree[x0]);
120 EXPECT_LONGS_EQUAL(x0, tree[x1]);
121 EXPECT_LONGS_EQUAL(x1, tree[x2]);
122 EXPECT_LONGS_EQUAL(x2, tree[x3]);
123
124 lago::key2doubleMap expected;
125 expected[x0] = 0;
126 expected[x1] = M_PI / 2; // edges traversed: x0->x1
127 expected[x2] = M_PI; // edges traversed: x0->x1->x2
128 expected[x3] = 3 * M_PI / 2; // edges traversed: x0->x1->x2->x3
129
130 lago::key2doubleMap deltaThetaMap;
131 vector<size_t>
132 spanningTreeIds; // ids of between factors forming the spanning tree T
133 vector<size_t>
134 chordsIds; // ids of between factors corresponding to chordsIds wrt T
135 lago::getSymbolicGraph(spanningTreeIds, chordsIds, deltaThetaMap, tree,
136 g: gPlus);
137
138 lago::key2doubleMap actual;
139 actual = lago::computeThetasToRoot(deltaThetaMap, tree);
140
141 DOUBLES_EQUAL(expected[x0], actual[x0], 1e-6);
142 DOUBLES_EQUAL(expected[x1], actual[x1], 1e-6);
143 DOUBLES_EQUAL(expected[x2], actual[x2], 1e-6);
144 DOUBLES_EQUAL(expected[x3], actual[x3], 1e-6);
145}

148TEST( Lago, regularizedMeasurements ) {
149 NonlinearFactorGraph g = simpleLago::graph();
150 auto gPlus = initialize::buildPoseGraph<Pose2>(graph: g);
151 lago::PredecessorMap tree = lago::findMinimumSpanningTree(pose2Graph: gPlus);
152
153 lago::key2doubleMap deltaThetaMap;
154 vector<size_t> spanningTreeIds; // ids of between factors forming the spanning tree T
155 vector<size_t> chordsIds; // ids of between factors corresponding to chordsIds wrt T
156 lago::getSymbolicGraph(spanningTreeIds, chordsIds, deltaThetaMap, tree, g: gPlus);
157
158 lago::key2doubleMap orientationsToRoot = lago::computeThetasToRoot(deltaThetaMap, tree);
159
160 GaussianFactorGraph lagoGraph = lago::buildLinearOrientationGraph(spanningTreeIds, chordsIds, g: gPlus, orientationsToRoot, tree);
161 std::pair<Matrix,Vector> actualAb = lagoGraph.jacobian();
162 // jacobian corresponding to the orientation measurements (last entry is the prior on the anchor and is disregarded)
163 Vector actual = (Vector(5) << actualAb.second(0),actualAb.second(1),actualAb.second(2),actualAb.second(3),actualAb.second(4)).finished();
164 // this is the whitened error, so we multiply by the std to unwhiten
165 actual = 0.1 * actual;
166 // Expected regularized measurements (same for the spanning tree, corrected for the chordsIds)
167 Vector expected = (Vector(5) << M_PI/2, M_PI/2, M_PI/2, 0 , -M_PI).finished();
168
169 EXPECT(assert_equal(expected, actual, 1e-6));
170}

251TEST( Lago, smallGraphValues ) {
252
253 // we set the orientations in the initial guess to zero
254 Values initialGuess;
255 initialGuess.insert(j: x0,val: Pose2(simpleLago::pose0.x(),simpleLago::pose0.y(),0.0));
256 initialGuess.insert(j: x1,val: Pose2(simpleLago::pose1.x(),simpleLago::pose1.y(),0.0));
257 initialGuess.insert(j: x2,val: Pose2(simpleLago::pose2.x(),simpleLago::pose2.y(),0.0));
258 initialGuess.insert(j: x3,val: Pose2(simpleLago::pose3.x(),simpleLago::pose3.y(),0.0));
259
260 // lago does not touch the Cartesian part and only fixed the orientations
261 Values actual = lago::initialize(graph: simpleLago::graph(), initialGuess);
262
263 // we are in a noiseless case
264 Values expected;
265 expected.insert(j: x0,val: simpleLago::pose0);
266 expected.insert(j: x1,val: simpleLago::pose1);
267 expected.insert(j: x2,val: simpleLago::pose2);
268 expected.insert(j: x3,val: simpleLago::pose3);
269
270 EXPECT(assert_equal(expected, actual, 1e-6));
271}

290TEST( Lago, largeGraphNoisy_orientations ) {
291
292 string inputFile = findExampleDataFile(name: "noisyToyGraph");
293 const auto [g, initial] = readG2o(g2oFile: inputFile);
294
295 // Add prior on the pose having index (key) = 0
296 NonlinearFactorGraph graphWithPrior = *g;
297 noiseModel::Diagonal::shared_ptr priorModel = noiseModel::Diagonal::Variances(variances: Vector3(1e-2, 1e-2, 1e-4));
298 graphWithPrior.addPrior(key: 0, prior: Pose2(), model: priorModel);
299
300 VectorValues actualVV = lago::initializeOrientations(graph: graphWithPrior);
301 Values actual;
302 Key keyAnc = symbol(c: 'Z',j: 9999999);
303 for(VectorValues::const_iterator it = actualVV.begin(); it != actualVV.end(); ++it ){
304 Key key = it->first;
305 if (key != keyAnc){
306 Vector orientation = actualVV.at(j: key);
307 Pose2 poseLago = Pose2(0.0,0.0,orientation(0));
308 actual.insert(j: key, val: poseLago);
309 }
310 }
311 string matlabFile = findExampleDataFile(name: "orientationsNoisyToyGraph");
312 const auto [gmatlab, expected] = readG2o(g2oFile: matlabFile);
313
314 for(const auto& key_pose: expected->extract<Pose2>()){
315 const Key& k = key_pose.first;
316 const Pose2& pose = key_pose.second;
317 EXPECT(assert_equal(pose, actual.at<Pose2>(k), 1e-5));
318 }
319}
