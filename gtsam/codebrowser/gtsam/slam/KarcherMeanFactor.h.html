28/**
29 * Optimize for the Karcher mean, minimizing the geodesic distance to each of
30 * the given Lie groups elements, by constructing a factor graph out of simple
31 * PriorFactors.
32 */
33template <class T>
34typename std::enable_if<traits<T>::IsLieGroup, T>::type
35FindKarcherMean(const std::vector<T, Eigen::aligned_allocator<T>> &elements);
36
37/// FindKarcherMean version from initializer list
38template <class T>
39typename std::enable_if<traits<T>::IsLieGroup, T>::type
40FindKarcherMean(std::initializer_list<T> &&elements);
41
42/**
43 * The KarcherMeanFactor creates a constraint on all SO(n) variables with
44 * given keys that the Karcher mean (see above) will stay the same. Note the
45 * mean itself is irrelevant to the constraint and is not a parameter: the
46 * constraint is implemented as enforcing that the sum of local updates is
47 * equal to zero, hence creating a rank-dim constraint. Note it is implemented
48 * as a soft constraint, as typically it is used to fix a gauge freedom.
49 * */
50template <class T> class KarcherMeanFactor : public NonlinearFactor {
51 // Compile time dimension: can be -1
52 inline constexpr static auto D = traits<T>::dimension;
53
54 // Runtime dimension: always >=0
55 size_t d_;
56
57 /// Constant Jacobian made of d*d identity matrices
58 std::shared_ptr<JacobianFactor> whitenedJacobian_;
59
60public:
61 /**
62 * Construct from given keys.
63 * If parameter beta is given, it acts as a precision = 1/sigma^2, and
64 * the Jacobian will be multiplied with 1/sigma = sqrt(beta).
65 */
66 template <typename CONTAINER>
67 KarcherMeanFactor(const CONTAINER &keys, int d = D,
68 std::optional<double> beta = {});
69
70 /// Destructor
71 ~KarcherMeanFactor() override {}
72
73 /// Calculate the error of the factor: always zero
74 double error(const Values &c) const override { return 0; }
75
76 /// get the dimension of the factor (number of rows on linearization)
77 size_t dim() const override { return d_; }
78
79 /// linearize to a GaussianFactor
80 std::shared_ptr<GaussianFactor> linearize(const Values &c) const override {
81 return whitenedJacobian_;
82 }
83};
