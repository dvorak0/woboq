11/**
12 * JacobianFactor for Schur complement that uses the "Nullspace Trick" by
13 * Mourikis et al.
14 *
15 * This trick is equivalent to the Schur complement, but can be faster.
16 * In essence, the linear factor |E*dp + F*dX - b|, where p is point and X are
17 * poses, is multiplied by Enull, a matrix that spans the left nullspace of E,
18 * i.e., The mx3 matrix is analyzed with SVD as E = [Erange Enull]*S*V (mxm *
19 * mx3 * 3x3) where Enull is an m x (m-3) matrix Then Enull'*E*dp = 0, and
20 * |Enull'*E*dp + Enull'*F*dX - Enull'*b| == |Enull'*F*dX - Enull'*b|
21 * Normally F is m x 6*numKeys, and Enull'*F yields an (m-3) x 6*numKeys matrix.
22 *
23 * The code below assumes that F is block diagonal and is given as a vector of
24 * ZDim*D blocks. Example: m = 4 (2 measurements), Enull = 4*1, F = 4*12 (for
25 * D=6) Then Enull'*F = 1*4 * 4*12 = 1*12, but each 1*6 piece can be computed as
26 * a 1x2 * 2x6 multiplication.
27 */
28template<size_t D, size_t ZDim>
29class JacobianFactorSVD: public RegularJacobianFactor<D> {
30
31 typedef RegularJacobianFactor<D> Base;
32 typedef Eigen::Matrix<double, ZDim, D> MatrixZD; // e.g 2 x 6 with Z=Point2
33 typedef std::pair<Key, Matrix> KeyMatrix;
34
35public:
36
37 /// Default constructor
38 JacobianFactorSVD() {
39 }
40
41 /// Empty constructor with keys.
42 JacobianFactorSVD(const KeyVector& keys,
43 const SharedDiagonal& model = SharedDiagonal())
44 : Base() {
45 Matrix zeroMatrix = Matrix::Zero(rows: 0, cols: D);
46 Vector zeroVector = Vector::Zero(size: 0);
47 std::vector<KeyMatrix> QF;
48 QF.reserve(n: keys.size());
49 for(const Key& key: keys)
50 QF.push_back(x: KeyMatrix(key, zeroMatrix));
51 JacobianFactor::fillTerms(QF, zeroVector, model);
52 }
53
54 /**
55 * @brief Construct a new JacobianFactorSVD object, createing a reduced-rank
56 * Jacobian factor on the CameraSet.
57 *
58 * @param keys keys associated with F blocks.
59 * @param Fblocks CameraSet derivatives, ZDim*D blocks of block-diagonal F
60 * @param Enull a reduced point derivative
61 * @param b right-hand side
62 * @param model noise model
63 */
64 JacobianFactorSVD(
65 const KeyVector& keys,
66 const std::vector<MatrixZD, Eigen::aligned_allocator<MatrixZD> >& Fblocks,
67 const Matrix& Enull, const Vector& b,
68 const SharedDiagonal& model = SharedDiagonal())
69 : Base() {
70 size_t numKeys = Enull.rows() / ZDim;
71 size_t m2 = ZDim * numKeys - 3; // TODO: is this not just Enull.rows()?
72 // PLAIN nullptr SPACE TRICK
73 // Matrix Q = Enull * Enull.transpose();
74 // for(const KeyMatrixZD& it: Fblocks)
75 // QF.push_back(KeyMatrix(it.first, Q.block(0, 2 * j++, m2, 2) * it.second));
76 // JacobianFactor factor(QF, Q * b);
77 std::vector<KeyMatrix> QF;
78 QF.reserve(n: numKeys);
79 for (size_t k = 0; k < Fblocks.size(); ++k) {
80 Key key = keys[k];
81 QF.emplace_back(
82 key, (Enull.transpose()).block(startRow: 0, startCol: ZDim * k, blockRows: m2, blockCols: ZDim) * Fblocks[k]);
83 }
84 JacobianFactor::fillTerms(QF, Enull.transpose() * b, model);
85 }
86};
