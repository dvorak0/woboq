45/**
46 * Find the full path to an example dataset distributed with gtsam. The name
47 * may be specified with or without a file extension - if no extension is
48 * given, this function first looks for the .graph extension, then .txt. We
49 * first check the gtsam source tree for the file, followed by the installed
50 * example dataset location. Both the source tree and installed locations
51 * are obtained from CMake during compilation.
52 * @return The full path and filename to the requested dataset.
53 * @throw std::invalid_argument if no matching file could be found using the
54 * search process described above.
55 */
56GTSAM_EXPORT std::string findExampleDataFile(const std::string& name);

64/// Indicates how noise parameters are stored in file
65enum NoiseFormat {
66 NoiseFormatG2O, ///< Information matrix I11, I12, I13, I22, I23, I33
67 NoiseFormatTORO, ///< Information matrix, but inf_ff inf_fs inf_ss inf_rr inf_fr inf_sr
68 NoiseFormatGRAPH, ///< default: toro-style order, but covariance matrix !
69 NoiseFormatCOV, ///< Covariance matrix C11, C12, C13, C22, C23, C33
70 NoiseFormatAUTO ///< Try to guess covariance matrix layout
71};
72
73/// Robust kernel type to wrap around quadratic noise model
74enum KernelFunctionType {
75 KernelFunctionTypeNONE, KernelFunctionTypeHUBER, KernelFunctionTypeTUKEY
76};

78/**
79 * Parse variables in a line-based text format (like g2o) into a map.
80 * Instantiated in .cpp Pose2, Point2, Pose3, and Point3.
81 * Note the map keys are integer indices, *not* gtsam::Keys. This is is
82 * different below where landmarks will use L(index) symbols.
83 */
84template <typename T>
85GTSAM_EXPORT std::map<size_t, T> parseVariables(const std::string &filename,
86 size_t maxIndex = 0);
87
88/**
89 * Parse binary measurements in a line-based text format (like g2o) into a
90 * vector. Instantiated in .cpp for Pose2, Rot2, Pose3, and Rot3. The rotation
91 * versions parse poses and extract only the rotation part, using the marginal
92 * covariance as noise model.
93 */
94template <typename T>
95GTSAM_EXPORT std::vector<BinaryMeasurement<T>>
96parseMeasurements(const std::string &filename,
97 const noiseModel::Diagonal::shared_ptr &model = nullptr,
98 size_t maxIndex = 0);
99
100/**
101 * Parse BetweenFactors in a line-based text format (like g2o) into a vector of
102 * shared pointers. Instantiated in .cpp T equal to Pose2 and Pose3.
103 */
104template <typename T>
105GTSAM_EXPORT std::vector<typename BetweenFactor<T>::shared_ptr>
106parseFactors(const std::string &filename,
107 const noiseModel::Diagonal::shared_ptr &model = nullptr,
108 size_t maxIndex = 0);

139/// Return type for load functions, which return a graph and initial values. For
140/// landmarks, the gtsam::Symbol L(index) is used to insert into the Values.
141/// Bearing-range measurements also refer to landmarks with L(index).
142using GraphAndValues =
143 std::pair<NonlinearFactorGraph::shared_ptr, Values::shared_ptr>;
144
145/**
146 * Load TORO 2D Graph
147 * @param dataset/model pair as constructed by [dataset]
148 * @param maxIndex if non-zero cut out vertices >= maxIndex
149 * @param addNoise add noise to the edges
150 * @param smart try to reduce complexity of covariance to cheapest model
151 */
152GTSAM_EXPORT GraphAndValues load2D(
153 std::pair<std::string, SharedNoiseModel> dataset, size_t maxIndex = 0,
154 bool addNoise = false,
155 bool smart = true, //
156 NoiseFormat noiseFormat = NoiseFormatAUTO,
157 KernelFunctionType kernelFunctionType = KernelFunctionTypeNONE);
158
159/**
160 * Load TORO/G2O style graph files
161 * @param filename
162 * @param model optional noise model to use instead of one specified by file
163 * @param maxIndex if non-zero cut out vertices >= maxIndex
164 * @param addNoise add noise to the edges
165 * @param smart try to reduce complexity of covariance to cheapest model
166 * @param noiseFormat how noise parameters are stored
167 * @param kernelFunctionType whether to wrap the noise model in a robust kernel
168 * @return graph and initial values
169 */
170GTSAM_EXPORT GraphAndValues
171load2D(const std::string& filename, SharedNoiseModel model = SharedNoiseModel(),
172 size_t maxIndex = 0, bool addNoise = false, bool smart = true,
173 NoiseFormat noiseFormat = NoiseFormatAUTO, //
174 KernelFunctionType kernelFunctionType = KernelFunctionTypeNONE);

181/**
182 * @brief This function parses a g2o file and stores the measurements into a
183 * NonlinearFactorGraph and the initial guess in a Values structure
184 * @param filename The name of the g2o file\
185 * @param is3D indicates if the file describes a 2D or 3D problem
186 * @param kernelFunctionType whether to wrap the noise model in a robust kernel
187 * @return graph and initial values
188 */
189GTSAM_EXPORT GraphAndValues
190readG2o(const std::string& g2oFile, const bool is3D = false,
191 KernelFunctionType kernelFunctionType = KernelFunctionTypeNONE);
192
193/**
194 * @brief This function writes a g2o file from
195 * NonlinearFactorGraph and a Values structure
196 * @param filename The name of the g2o file to write
197 * @param graph NonlinearFactor graph storing the measurements
198 * @param estimate Values
199 *
200 * Note:behavior change in PR #471: to be consistent with load2D and load3D, we
201 * write the *indices* to file and not the full Keys. This change really only
202 * affects landmarks, which get read as indices but stored in values with the
203 * symbol L(index).
204 */
205GTSAM_EXPORT void writeG2o(const NonlinearFactorGraph& graph,
206 const Values& estimate, const std::string& filename);
