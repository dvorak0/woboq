15/**
16 * Factor to measure a planar landmark from a given pose
17 */
18class GTSAM_EXPORT OrientedPlane3Factor: public NoiseModelFactorN<Pose3, OrientedPlane3> {
19 protected:
20 OrientedPlane3 measured_p_;
21 typedef NoiseModelFactorN<Pose3, OrientedPlane3> Base;
22
23 public:
24
25 // Provide access to the Matrix& version of evaluateError:
26 using NoiseModelFactor2<Pose3, OrientedPlane3>::evaluateError;
27
28 /// Constructor
29 OrientedPlane3Factor() {
30 }
31 ~OrientedPlane3Factor() override {}
32
33 /** Constructor with measured plane (a,b,c,d) coefficients
34 * @param z measured plane (a,b,c,d) coefficients as 4D vector
35 * @param noiseModel noiseModel Gaussian noise model
36 * @param poseKey Key or symbol for unknown pose
37 * @param landmarkKey Key or symbol for unknown planar landmark
38 * @return the transformed plane
39 */
40 OrientedPlane3Factor(const Vector4& z, const SharedGaussian& noiseModel,
41 Key poseKey, Key landmarkKey)
42 : Base(noiseModel, poseKey, landmarkKey), measured_p_(z) {}
43
44 /// print
45 void print(const std::string& s = "OrientedPlane3Factor",
46 const KeyFormatter& keyFormatter = DefaultKeyFormatter) const override;
47
48 /// evaluateError
49 Vector evaluateError(
50 const Pose3& pose, const OrientedPlane3& plane,
51 OptionalMatrixType H1, OptionalMatrixType H2) const override;
52};
