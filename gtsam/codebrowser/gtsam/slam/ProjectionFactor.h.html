30namespace gtsam {
31
32 /**
33 * Non-linear factor for a constraint derived from a 2D measurement.
34 * The calibration is known here.
35 * The main building block for visual SLAM.
36 * @ingroup slam
37 */
38 template <class POSE = Pose3, class LANDMARK = Point3,
39 class CALIBRATION = Cal3_S2>
40 class GenericProjectionFactor: public NoiseModelFactorN<POSE, LANDMARK> {
41 protected:
42
43 // Keep a copy of measurement and calibration for I/O
44 Point2 measured_; ///< 2D measurement
45 std::shared_ptr<CALIBRATION> K_; ///< shared pointer to calibration object
46 std::optional<POSE> body_P_sensor_; ///< The pose of the sensor in the body frame
47
48 // verbosity handling for Cheirality Exceptions
49 bool throwCheirality_; ///< If true, rethrows Cheirality exceptions (default: false)
50 bool verboseCheirality_; ///< If true, prints text for Cheirality exceptions (default: false)
51
52 public:
53
54 /// shorthand for base class type
55 typedef NoiseModelFactorN<POSE, LANDMARK> Base;
56
57 // Provide access to the Matrix& version of evaluateError:
58 using Base::evaluateError;
59
60 /// shorthand for this class
61 typedef GenericProjectionFactor<POSE, LANDMARK, CALIBRATION> This;
62
63 /// shorthand for a smart pointer to a factor
64 typedef std::shared_ptr<This> shared_ptr;
65
66 /// Default constructor
67 GenericProjectionFactor() :
68 measured_(0, 0), throwCheirality_(false), verboseCheirality_(false) {
69 }
70
71 /**
72 * Constructor
73 * TODO: Mark argument order standard (keys, measurement, parameters)
74 * @param measured is the 2 dimensional location of point in image (the measurement)
75 * @param model is the standard deviation
76 * @param poseKey is the index of the camera
77 * @param pointKey is the index of the landmark
78 * @param K shared pointer to the constant calibration
79 * @param body_P_sensor is the transform from body to sensor frame (default identity)
80 */
81 GenericProjectionFactor(const Point2& measured, const SharedNoiseModel& model,
82 Key poseKey, Key pointKey, const std::shared_ptr<CALIBRATION>& K,
83 std::optional<POSE> body_P_sensor = {}) :
84 Base(model, poseKey, pointKey), measured_(measured), K_(K), body_P_sensor_(body_P_sensor),
85 throwCheirality_(false), verboseCheirality_(false) {}
86
87 /**
88 * Constructor with exception-handling flags
89 * TODO: Mark argument order standard (keys, measurement, parameters)
90 * @param measured is the 2 dimensional location of point in image (the measurement)
91 * @param model is the standard deviation
92 * @param poseKey is the index of the camera
93 * @param pointKey is the index of the landmark
94 * @param K shared pointer to the constant calibration
95 * @param throwCheirality determines whether Cheirality exceptions are rethrown
96 * @param verboseCheirality determines whether exceptions are printed for Cheirality
97 * @param body_P_sensor is the transform from body to sensor frame (default identity)
98 */
99 GenericProjectionFactor(const Point2& measured, const SharedNoiseModel& model,
100 Key poseKey, Key pointKey, const std::shared_ptr<CALIBRATION>& K,
101 bool throwCheirality, bool verboseCheirality,
102 std::optional<POSE> body_P_sensor = {}) :
103 Base(model, poseKey, pointKey), measured_(measured), K_(K), body_P_sensor_(body_P_sensor),
104 throwCheirality_(throwCheirality), verboseCheirality_(verboseCheirality) {}
105
106 /** Virtual destructor */
107 ~GenericProjectionFactor() override {}
108
109 /// @return a deep copy of this factor
110 gtsam::NonlinearFactor::shared_ptr clone() const override {
111 return std::static_pointer_cast<gtsam::NonlinearFactor>(
112 r: gtsam::NonlinearFactor::shared_ptr(new This(*this))); }
113
114 /**
115 * print
116 * @param s optional string naming the factor
117 * @param keyFormatter optional formatter useful for printing Symbols
118 */
119 void print(const std::string& s = "", const KeyFormatter& keyFormatter = DefaultKeyFormatter) const override {
120 std::cout << s << "GenericProjectionFactor, z = ";
121 traits<Point2>::Print(m: measured_);
122 if(this->body_P_sensor_)
123 this->body_P_sensor_->print(" sensor pose in body frame: ");
124 Base::print("", keyFormatter);
125 }
126
127 /// equals
128 bool equals(const NonlinearFactor& p, double tol = 1e-9) const override {
129 const This *e = dynamic_cast<const This*>(&p);
130 return e
131 && Base::equals(p, tol)
132 && traits<Point2>::Equals(v1: this->measured_, v2: e->measured_, tol)
133 && this->K_->equals(*e->K_, tol)
134 && ((!body_P_sensor_ && !e->body_P_sensor_) || (body_P_sensor_ && e->body_P_sensor_ && body_P_sensor_->equals(*e->body_P_sensor_)));
135 }
136
137 /// Evaluate error h(x)-z and optionally derivatives
138 Vector evaluateError(const Pose3& pose, const Point3& point,
139 OptionalMatrixType H1, OptionalMatrixType H2) const override {
140 try {
141 if(body_P_sensor_) {
142 if(H1) {
143 gtsam::Matrix H0;
144 PinholeCamera<CALIBRATION> camera(pose.compose(*body_P_sensor_, H0), *K_);
145 Point2 reprojectionError(camera.project(point, H1, H2, {}) - measured_);
146 *H1 = *H1 * H0;
147 return reprojectionError;
148 } else {
149 PinholeCamera<CALIBRATION> camera(pose.compose(*body_P_sensor_), *K_);
150 return camera.project(point, H1, H2, {}) - measured_;
151 }
152 } else {
153 PinholeCamera<CALIBRATION> camera(pose, *K_);
154 return camera.project(point, H1, H2, {}) - measured_;
155 }
156 } catch( CheiralityException& e) {
157 if (H1) *H1 = Matrix::Zero(rows: 2,cols: 6);
158 if (H2) *H2 = Matrix::Zero(rows: 2,cols: 3);
159 if (verboseCheirality_)
160 std::cout << e.what() << ": Landmark "<< DefaultKeyFormatter(this->key2()) <<
161 " moved behind camera " << DefaultKeyFormatter(this->key1()) << std::endl;
162 if (throwCheirality_)
163 throw CheiralityException(this->key2());
164 }
165 return Vector2::Constant(2.0 * K_->fx());
166 }
167
168 /** return the measurement */
169 const Point2& measured() const {
170 return measured_;
171 }
172
173 /** return the calibration object */
174 const std::shared_ptr<CALIBRATION> calibration() const {
175 return K_;
176 }
177
178 /** return the (optional) sensor pose with respect to the vehicle frame */
179 const std::optional<POSE>& body_P_sensor() const {
180 return body_P_sensor_;
181 }
182
183 /** return verbosity */
184 inline bool verboseCheirality() const { return verboseCheirality_; }
185
186 /** return flag for throwing cheirality exceptions */
187 inline bool throwCheirality() const { return throwCheirality_; }
188
189 private:
190
191#if GTSAM_ENABLE_BOOST_SERIALIZATION ///
192 /// Serialization function
193 friend class boost::serialization::access;
194 template<class ARCHIVE>
195 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
196 ar & BOOST_SERIALIZATION_BASE_OBJECT_NVP(Base);
197 ar & BOOST_SERIALIZATION_NVP(measured_);
198 ar & BOOST_SERIALIZATION_NVP(K_);
199 ar & BOOST_SERIALIZATION_NVP(body_P_sensor_);
200 ar & BOOST_SERIALIZATION_NVP(throwCheirality_);
201 ar & BOOST_SERIALIZATION_NVP(verboseCheirality_);
202 }
203#endif
204
205 public:
206 GTSAM_MAKE_ALIGNED_OPERATOR_NEW
207};
208

211 struct traits<GenericProjectionFactor<POSE, LANDMARK, CALIBRATION> > :
212 public Testable<GenericProjectionFactor<POSE, LANDMARK, CALIBRATION> > {
213 };
