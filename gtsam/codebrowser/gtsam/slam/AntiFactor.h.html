12/**

16#pragma once
17
18#include <ostream>
19
20#include <gtsam/nonlinear/NonlinearFactor.h>
21#include <gtsam/linear/GaussianFactor.h>
22
23namespace gtsam {
24
25 /**
26 * A class for downdating an existing factor from a graph. The AntiFactor returns the same
27 * linearized Hessian matrices of the original factor, but with the opposite sign. This effectively
28 * cancels out any affects of the original factor during optimization."
29 * @ingroup slam
30 */
31 class AntiFactor: public NonlinearFactor {
32
33 private:
34
35 typedef AntiFactor This;
36 typedef NonlinearFactor Base;
37 typedef NonlinearFactor::shared_ptr sharedFactor;
38
39 sharedFactor factor_;
40
41 public:
42
43 // shorthand for a smart pointer to a factor
44 typedef std::shared_ptr<AntiFactor> shared_ptr;
45
46 /** default constructor - only use for serialization */
47 AntiFactor() {}
48
49 /** constructor - Creates the equivalent AntiFactor from an existing factor */
50 AntiFactor(NonlinearFactor::shared_ptr factor) : Base(factor->keys()), factor_(factor) {}
51
52 ~AntiFactor() override {}
53
54 /// @return a deep copy of this factor
55 gtsam::NonlinearFactor::shared_ptr clone() const override {
56 return std::static_pointer_cast<gtsam::NonlinearFactor>(
57 r: gtsam::NonlinearFactor::shared_ptr(new This(*this))); }
58
59 /** implement functions needed for Testable */
60
61 /** print */
62 void print(const std::string& s, const KeyFormatter& keyFormatter = DefaultKeyFormatter) const override {
63 std::cout << s << "AntiFactor version of:" << std::endl;
64 factor_->print(s, keyFormatter);
65 }
66
67 /** equals */
68 bool equals(const NonlinearFactor& expected, double tol=1e-9) const override {
69 const This *e = dynamic_cast<const This*> (&expected);
70 return e != nullptr && Base::equals(f: *e, tol) && this->factor_->equals(f: *e->factor_, tol);
71 }
72
73 /** implement functions needed to derive from Factor */
74
75 /**
76 * Calculate the error of the factor
77 * For the AntiFactor, this will have the same magnitude of the original factor,
78 * but the opposite sign.
79 */
80 double error(const Values& c) const override { return -factor_->error(c); }
81
82 /** get the dimension of the factor (same as the original factor) */
83 size_t dim() const override { return factor_->dim(); }
84
85 /**
86 * Checks whether this factor should be used based on a set of values.
87 * The AntiFactor will have the same 'active' profile as the original factor.
88 */
89 bool active(const Values& c) const override { return factor_->active(c); }
90
91 /**
92 * Linearize to a GaussianFactor. The AntiFactor always returns a Hessian Factor
93 * with the same Hessian matrices as the original factor (even if the original factor
94 * returns a Jacobian instead of a full Hessian), but with the opposite sign. This
95 * effectively cancels the effect of the original factor on the factor graph.
96 */
97 std::shared_ptr<GaussianFactor> linearize(const Values& c) const override {
98
99 // Generate the linearized factor from the contained nonlinear factor
100 GaussianFactor::shared_ptr gaussianFactor = factor_->linearize(c);
101
102 // return the negated version of the factor
103 return gaussianFactor->negate();
104 }
105
106
107 private:
108
109#if GTSAM_ENABLE_BOOST_SERIALIZATION
110 /** Serialization function */
111 friend class boost::serialization::access;
112 template<class ARCHIVE>
113 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
114 ar & boost::serialization::make_nvp(n: "AntiFactor",
115 v&: boost::serialization::base_object<Base>(d&: *this));
116 ar & BOOST_SERIALIZATION_NVP(factor_);
117 }
118#endif
119 }; // \class AntiFactor
120
121} /// namespace gtsam
