22namespace gtsam {

24/**
25 * Transform function that must be specialized specific domains
26 * @tparam T is a Transform type
27 * @tparam P is a point type
28 */
29template<class T, class P>
30P transform_point(
31 const T& trans, const P& global,
32 OptionalMatrixType Dtrans,
33 OptionalMatrixType Dglobal) {
34 return trans.transformFrom(global, Dtrans, Dglobal);
35}
36
37/**
38 * A constraint between two landmarks in separate maps
39 * Templated on:
40 * Point : Type of landmark
41 * Transform : Transform variable class
42 *
43 * The transform is defined as transforming global to local:
44 * l = lTg * g
45 *
46 * The Point and Transform concepts must be Lie types, and the transform
47 * relationship "Point = transformFrom(Transform, Point)" must exist.
48 *
49 * To implement this function in new domains, specialize a new version of
50 * Point transform_point<Transform,Point>(transform, global, Dtrans, Dglobal)
51 * to use the correct point and transform types.
52 *
53 * This base class should be specialized to implement the cost function for
54 * specific classes of landmarks
55 */
56template<class POINT, class TRANSFORM>
57class ReferenceFrameFactor : public NoiseModelFactorN<POINT, TRANSFORM, POINT> {
58protected:
59 /** default constructor for serialization only */
60 ReferenceFrameFactor() {}
61
62public:
63 typedef NoiseModelFactorN<POINT, TRANSFORM, POINT> Base;
64 typedef ReferenceFrameFactor<POINT, TRANSFORM> This;
65
66 // Provide access to the Matrix& version of evaluateError:
67 using Base::evaluateError;
68
69 typedef POINT Point;
70 typedef TRANSFORM Transform;
71
72 /**
73 * General constructor with arbitrary noise model (constrained or otherwise)
74 */
75 ReferenceFrameFactor(Key globalKey, Key transKey, Key localKey, const noiseModel::Base::shared_ptr& model)
76 : Base(model,globalKey, transKey, localKey) {}
77
78 /**
79 * Construct a hard frame of reference reference constraint with equal mu values for
80 * each degree of freedom.
81 */
82 ReferenceFrameFactor(double mu, Key globalKey, Key transKey, Key localKey)
83 : Base(globalKey, transKey, localKey, Point().dim(), mu) {}
84
85 /**
86 * Simple soft constraint constructor for frame of reference, with equal weighting for
87 * each degree of freedom.
88 */
89 ReferenceFrameFactor(Key globalKey, Key transKey, Key localKey, double sigma = 1e-2)
90 : Base(noiseModel::Isotropic::Sigma(dim: traits<POINT>::dimension, sigma),
91 globalKey, transKey, localKey) {}
92
93 ~ReferenceFrameFactor() override{}
94
95 NonlinearFactor::shared_ptr clone() const override {
96 return std::static_pointer_cast<NonlinearFactor>(
97 r: NonlinearFactor::shared_ptr(new This(*this))); }
98
99 /** Combined cost and derivative function using boost::optional */
100 Vector evaluateError(const Point& global, const Transform& trans, const Point& local,
101 OptionalMatrixType Dforeign, OptionalMatrixType Dtrans,
102 OptionalMatrixType Dlocal) const override {
103 Point newlocal = transform_point<Transform,Point>(trans, global, Dtrans, Dforeign);
104 if (Dlocal) {
105 *Dlocal = -1* Matrix::Identity(traits<Point>::dimension, traits<Point>::dimension);
106 }
107 return traits<Point>::Local(local,newlocal);
108 }
109
110 void print(const std::string& s="",
111 const gtsam::KeyFormatter& keyFormatter = DefaultKeyFormatter) const override {
112 std::cout << s << ": ReferenceFrameFactor("
113 << "Global: " << keyFormatter(this->key1()) << ","
114 << " Transform: " << keyFormatter(this->key2()) << ","
115 << " Local: " << keyFormatter(this->key3()) << ")\n";
116 this->noiseModel_->print(" noise model");
117 }
118
119 // access - convenience functions
120 Key global_key() const { return this->key1(); }
121 Key transform_key() const { return this->key2(); }
122 Key local_key() const { return this->key3(); }
123
124private:
125#if GTSAM_ENABLE_BOOST_SERIALIZATION
126 /** Serialization function */
127 friend class boost::serialization::access;
128 template<class ARCHIVE>
129 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
130 ar & boost::serialization::make_nvp("NonlinearFactor3",
131 boost::serialization::base_object<Base>(*this));
132 }
133#endif
134};
