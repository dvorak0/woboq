28/**
29 * @brief Convert a possibly robust noise model to an isotropic model for
30 * Frobenius factors.
31 *
32 * This function is used to convert a noise model, which may be robust, into an
33 * isotropic noise model suitable for Frobenius factors. If the input noise
34 * model is null, it returns an n-dimensional isotropic noise model with
35 * sigma=1.0. If the input noise model is isotropic, it extends it to the
36 * desired dimension. If the noise model is robust, the sigmas of the underlying
37 * noise model are used. If the noise model is not isotropic and `defaultToUnit`
38 * is false, an exception is thrown.
39 *
40 * @param model The input noise model (possibly robust).
41 * @param dimension The desired dimension for the isotropic model.
42 * @param defaultToUnit If true, fallback to unit if conversion is not possible.
43 * @throws std::runtime_error if model not isotropic and defaultToUnit =false.
44 * @return An isotropic (possibly robust) noise model.
45 */
46GTSAM_EXPORT SharedNoiseModel ConvertNoiseModel(const SharedNoiseModel& model,
47 size_t n,
48 bool defaultToUnit = true);
49
50/**
51 * @brief Ensure a noise model has the correct dimension for a given type.
52 *
53 * If the model is already of dimension Dim, it is returned as-is.
54 * Otherwise, ConvertNoiseModel is called to convert it.
55 * Asserts that the model's dimension matches T's expected dimension before
56 * conversion.
57 *
58 * @tparam T The type whose dimension is checked.
59 * @tparam Dim The required dimension.
60 * @param model The input noise model.
61 * @return A noise model of dimension Dim.
62 */
63template <class T, size_t Dim>
64inline SharedNoiseModel ConvertModel(const SharedNoiseModel& model) {
65 if (!model || model->dim() == Dim) {
66 return model;
67 }
68 if (model->dim() != T::dimension) {
69 throw std::runtime_error(
70 "Noise model dimension does not match expected dimension for T.");
71 }
72 return ConvertNoiseModel(model, n: Dim);
73}

75/**
76 * FrobeniusPrior calculates the Frobenius norm between a given matrix and an
77 * element of SO(3) or SO(4).
78 */
79template <class T>
80class FrobeniusPrior : public NoiseModelFactorN<T> {
81 GTSAM_CONCEPT_ASSERT(IsMatrixLieGroup<T>);
82 inline constexpr static auto N = T::LieAlgebra::RowsAtCompileTime;
83 inline constexpr static auto Dim = N * N;
84 using MatrixNN = Eigen::Matrix<double, N, N>;
85 Eigen::Matrix<double, Dim, 1> vecM_; ///< vectorized matrix to approximate
86
87 public:
88 // Provide access to the Matrix& version of evaluateError:
89 using NoiseModelFactor1<T>::evaluateError;
90
91 EIGEN_MAKE_ALIGNED_OPERATOR_NEW
92
93 /// Constructor
94 FrobeniusPrior(Key j, const MatrixNN& M,
95 const SharedNoiseModel& model = nullptr)
96 : NoiseModelFactorN<T>(ConvertModel<T, Dim>(model), j) {
97 vecM_ << Eigen::Map<const Matrix>(M.data(), Dim, 1);
98 }
99
100 /// Error is just Frobenius norm between T element and vectorized matrix M.
101 Vector evaluateError(const T& g, OptionalMatrixType H) const override {
102 return traits<T>::Vec(g, H) -
103 vecM_; // Jacobian is computed only when needed.
104 }
105};

107/**
108 * FrobeniusFactor calculates the Frobenius norm between rotation matrices.
109 * The template argument can be any fixed-size SO<N>.
110 */
111template <class T>
112class FrobeniusFactor : public NoiseModelFactorN<T, T> {
113 GTSAM_CONCEPT_ASSERT(IsMatrixLieGroup<T>);
114 inline constexpr static auto N = T::LieAlgebra::RowsAtCompileTime;
115 inline constexpr static auto Dim = N * N;
116
117 public:
118 // Provide access to the Matrix& version of evaluateError:
119 using NoiseModelFactor2<T, T>::evaluateError;
120
121 /// Constructor
122 FrobeniusFactor(Key j1, Key j2, const SharedNoiseModel& model = nullptr)
123 : NoiseModelFactorN<T, T>(ConvertModel<T, Dim>(model), j1, j2) {}
124
125 /// Error is just Frobenius norm between rotation matrices.
126 Vector evaluateError(const T& T1, const T& T2, OptionalMatrixType H1,
127 OptionalMatrixType H2) const override {
128 Vector error = traits<T>::Vec(T2, H2) - traits<T>::Vec(T1, H1);
129 if (H1) *H1 = -*H1;
130 return error;
131 }
132};

134/**
135 * FrobeniusBetweenFactorNL is a BetweenFactor that evaluates the Frobenius
136 * norm of the rotation error between measured and predicted (rather than the
137 * Logmap of the error). This factor is only defined for fixed-dimension
138 * types, that are matrix Lie groups.
139 *
140 * This version is called NL, because it minimizes |inv(T2)*T1*T12_ - I|_F
141 * as opposed to the (historically older) FrobeniusBetweenFactor, that
142 * minimizes
143 * ||T2 - T1*T12_||_F. This only holds for certain groups, e.g., not Sim(3).
144 */
145template <class T>
146class FrobeniusBetweenFactorNL : public NoiseModelFactorN<T, T> {
147 GTSAM_CONCEPT_ASSERT(IsMatrixLieGroup<T>);
148 inline constexpr static auto N = T::LieAlgebra::RowsAtCompileTime;
149 inline constexpr static auto Dim = N * N;
150 static_assert(N > 0, "The Lie algebra dimension N must be greater than 0.");
151
152 protected:
153 T T12_; ///< measured rotation between T1 and T2
154
155 using MatrixN = Eigen::Matrix<double, N, N>;
156 using VectorD = Eigen::Matrix<double, Dim, 1>;
157
158 public:
159 // Provide access to the Matrix& version of evaluateError:
160 using NoiseModelFactor2<T, T>::evaluateError;
161
162 EIGEN_MAKE_ALIGNED_OPERATOR_NEW
163
164 /// @name Constructor
165 /// @{
166
167 /// Construct from two keys and measured rotation
168 FrobeniusBetweenFactorNL(Key j1, Key j2, const T& T12,
169 const SharedNoiseModel& model = nullptr)
170 : NoiseModelFactorN<T, T>(ConvertModel<T, Dim>(model), j1, j2),
171 T12_(T12) {}
172
173 /// @}
174 /// @name Testable
175 /// @{
176
177 /// print with optional string
178 void print(const std::string& s, const KeyFormatter& keyFormatter =
179 DefaultKeyFormatter) const override {
180 std::cout << s << "FrobeniusBetweenFactorNL<" << demangle(name: typeid(T).name())
181 << ">(" << keyFormatter(this->key1()) << ","
182 << keyFormatter(this->key2()) << ")\n";
183 traits<T>::Print(T12_, " T12: ");
184 this->noiseModel_->print(" noise model: ");
185 }
186
187 /// assert equality up to a tolerance
188 bool equals(const NonlinearFactor& expected,
189 double tol = 1e-9) const override {
190 auto e = dynamic_cast<const FrobeniusBetweenFactorNL*>(&expected);
191 return e != nullptr && NoiseModelFactorN<T, T>::equals(*e, tol) &&
192 traits<T>::Equals(this->T12_, e->T12_, tol);
193 }
194
195 /// @}
196 /// @name NoiseModelFactorN methods
197 /// @{
198
199 /// Error is |inv(T2)*T1*T12_ - I|_F.
200 Vector evaluateError(const T& T1, const T& T2, OptionalMatrixType H1,
201 OptionalMatrixType H2) const override {
202 // predict T2*T1
203 typename T::Jacobian H_T21_T2;
204 const T hatT21 = traits<T>::Between(T2, T1, H1 ? &H_T21_T2 : nullptr);
205
206 // Calculate \hat T21 * T12_, which is predicted to be I_NxN
207 typename T::Jacobian H_pred_hat;
208 const T pred = traits<T>::Compose(hatT21, T12_, H1 ? &H_pred_hat : nullptr);
209
210 // Move to constructor
211 const MatrixN I = MatrixN::Identity();
212 const VectorD vecI = Eigen::Map<const VectorD>(I.data());
213
214 // Calculate error
215 Eigen::Matrix<double, Dim, T::dimension> H_vec_pred;
216 Vector error = traits<T>::Vec(pred, H1 ? &H_vec_pred : nullptr) - vecI;
217
218 // Do chain rule
219 const auto H_error_hat21 = H_vec_pred * H_pred_hat;
220 if (H1) *H1 = H_error_hat21; // H_pred_T1 is identity
221 if (H2) *H2 = H_error_hat21 * H_T21_T2;
222 return error;
223 }
224 /// @}
225};

227/**
228 * FrobeniusBetweenFactor uses ||T2 - T1*T12_||_F, which only works if the
229 * Frobenius error is invariant to multiplying with an arbitrary element T.
230 */
231template <class T>
232class FrobeniusBetweenFactor : public FrobeniusBetweenFactorNL<T> {
233 inline constexpr static auto N = T::LieAlgebra::RowsAtCompileTime;
234 inline constexpr static auto Dim = N * N;
235
236 typename T::Jacobian T2hat_H_T1_; ///< fixed derivative of T2hat wrpt T1
237
238 public:
239 // Provide access to the Matrix& version of evaluateError:
240 using NoiseModelFactor2<T, T>::evaluateError;
241
242 EIGEN_MAKE_ALIGNED_OPERATOR_NEW
243
244 /// Construct from two keys and measured rotation
245 FrobeniusBetweenFactor(Key j1, Key j2, const T& T12,
246 const SharedNoiseModel& model = nullptr)
247 : FrobeniusBetweenFactorNL<T>(j1, j2, T12, model),
248 T2hat_H_T1_(traits<T>::AdjointMap(traits<T>::Inverse(T12))) {}
249
250 /// Error is Frobenius norm between T1*T12 and T2.
251 Vector evaluateError(const T& T1, const T& T2, OptionalMatrixType H1,
252 OptionalMatrixType H2) const override {
253 const T T2hat = traits<T>::Compose(T1, this->T12_);
254 Eigen::Matrix<double, Dim, T::dimension> vec_H_T2hat;
255 Vector error = traits<T>::Vec(T2, H2) -
256 traits<T>::Vec(T2hat, H1 ? &vec_H_T2hat : nullptr);
257 if (H1) *H1 = -vec_H_T2hat * T2hat_H_T1_;
258 return error;
259 }
260};
