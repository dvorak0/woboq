25SharedNoiseModel ConvertNoiseModel(const SharedNoiseModel &model,
26 size_t dimension, bool defaultToUnit) {
27 double sigma = 1.0;
28 noiseModel::mEstimator::Base::shared_ptr robustLoss;
29
30 if (model != nullptr) {
31 // Unwrap robust model if present and remember its robust loss
32 SharedNoiseModel baseModel = model;
33 if (auto robust = std::dynamic_pointer_cast<noiseModel::Robust>(r: model)) {
34 robustLoss = robust->robust();
35 baseModel = robust->noise();
36 }
37
38 // Use smart constructor to check for isotropy
39 Vector sigmas = baseModel->sigmas();
40 auto smartModel = noiseModel::Diagonal::Sigmas(sigmas, smart: true);
41 auto isotropic =
42 std::dynamic_pointer_cast<noiseModel::Isotropic>(r: smartModel);
43
44 if (isotropic) {
45 sigma = isotropic->sigma();
46 } else if (!defaultToUnit) {
47 throw std::runtime_error("Can only convert isotropic rotation noise");
48 }
49 }
50
51 auto isoModel = noiseModel::Isotropic::Sigma(dim: dimension, sigma);
52 if (robustLoss) {
53 return noiseModel::Robust::Create(robust: robustLoss, noise: isoModel);
54 } else {
55 return isoModel;
56 }
57}
