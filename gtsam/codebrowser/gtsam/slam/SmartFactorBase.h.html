40/**
41 * @brief Base class for smart factors.
42 * This base class has no internal point, but it has a measurement, noise model
43 * and an optional sensor pose.
44 * This class mainly computes the derivatives and returns them as a variety of
45 * factors. The methods take a CameraSet<CAMERA> argument and the value of a
46 * point, which is kept in the derived class.
47 *
48 * @tparam CAMERA should behave like a PinholeCamera.
49 */
50template<class CAMERA>
51class SmartFactorBase: public NonlinearFactor {
52
53private:
54 typedef NonlinearFactor Base;
55 typedef SmartFactorBase<CAMERA> This;
56 typedef typename CAMERA::Measurement Z;
57 typedef typename CAMERA::MeasurementVector ZVector;
58
59public:
60
61 static const int Dim = traits<CAMERA>::dimension; ///< Camera dimension
62 static const int ZDim = traits<Z>::dimension; ///< Measurement dimension
63 typedef Eigen::Matrix<double, ZDim, Dim> MatrixZD; // F blocks (derivatives wrpt camera)
64 typedef std::vector<MatrixZD, Eigen::aligned_allocator<MatrixZD> > FBlocks; // vector of F blocks
65
66protected:
67 /**
68 * As of Feb 22, 2015, the noise model is the same for all measurements and
69 * is isotropic. This allows for moving most calculations of Schur complement
70 * etc. to be easily moved to CameraSet, and also agrees pragmatically
71 * with what is normally done.
72 */
73 SharedIsotropic noiseModel_;
74
75 /**
76 * Measurements for each of the m views.
77 * We keep a copy of the measurements for I/O and computing the error.
78 * The order is kept the same as the keys that we use to create the factor.
79 */
80 ZVector measured_;
81
82 std::optional<Pose3>
83 body_P_sensor_; ///< Pose of the camera in the body frame
84
85 // Cache for Fblocks, to avoid a malloc ever time we re-linearize
86 mutable FBlocks Fs;
87
88 public:
89 GTSAM_MAKE_ALIGNED_OPERATOR_NEW
90
91 /// shorthand for a smart pointer to a factor.
92 typedef std::shared_ptr<This> shared_ptr;
93
94 /// The CameraSet data structure is used to refer to a set of cameras.
95 typedef CameraSet<CAMERA> Cameras;

97 /// Default Constructor, for serialization.
98 SmartFactorBase() {}

100 /// Construct with given noise model and optional arguments.
101 SmartFactorBase(const SharedNoiseModel& sharedNoiseModel,
102 std::optional<Pose3> body_P_sensor = {},
103 size_t expectedNumberCameras = 10)
104 : body_P_sensor_(body_P_sensor), Fs(expectedNumberCameras) {
105
106 if (!sharedNoiseModel)
107 throw std::runtime_error("SmartFactorBase: sharedNoiseModel is required");
108
109 SharedIsotropic sharedIsotropic = std::dynamic_pointer_cast<
110 noiseModel::Isotropic>(r: sharedNoiseModel);
111
112 if (!sharedIsotropic)
113 throw std::runtime_error("SmartFactorBase: needs isotropic");
114
115 noiseModel_ = sharedIsotropic;
116 }
117
118 /// Virtual destructor, subclasses from NonlinearFactor.
119 ~SmartFactorBase() override {
120 }

122 /**
123 * Add a new measurement and pose/camera key.
124 * @param measured is the 2m dimensional projection of a single landmark
125 * @param key is the index corresponding to the camera observing the landmark
126 */
127 void add(const Z& measured, const Key& key) {
128 if(std::find(keys_.begin(), keys_.end(), key) != keys_.end()) {
129 throw std::invalid_argument(
130 "SmartFactorBase::add: adding duplicate measurement for key.");
131 }
132 this->measured_.push_back(measured);
133 this->keys_.push_back(key);
134 }
135
136 /// Add a bunch of measurements, together with the camera keys.
137 void add(const ZVector& measurements, const KeyVector& cameraKeys) {
138#ifndef NDEBUG
139 if (measurements.size() != cameraKeys.size()) {
140 throw std::runtime_error("Number of measurements and camera keys do not match");
141 }
142#endif
143 for (size_t i = 0; i < measurements.size(); i++) {
144 this->add(measurements[i], cameraKeys[i]);
145 }
146 }
147
148 /**
149 * Add an entire SfM_track (collection of cameras observing a single point).
150 * The noise is assumed to be the same for all measurements.
151 */
152 template<class SFM_TRACK>
153 void add(const SFM_TRACK& trackToAdd) {
154 for (size_t k = 0; k < trackToAdd.numberMeasurements(); k++) {
155 this->measured_.push_back(trackToAdd.measurements[k].second);
156 this->keys_.push_back(trackToAdd.measurements[k].first);
157 }
158 }
159
160 /// Return the dimension (number of rows!) of the factor.
161 size_t dim() const override { return ZDim * this->measured_.size(); }
162
163 /// Return the 2D measurements (ZDim, in general).
164 const ZVector& measured() const { return measured_; }
165
166 /// Collect all cameras: important that in key order.
167 virtual Cameras cameras(const Values& values) const {
168 Cameras cameras;
169 for(const Key& k: this->keys_) {
170 cameras.push_back(values.at<CAMERA>(k));
171 }
172 return cameras;
173 }

176 * print
177 * @param s optional string naming the factor
178 * @param keyFormatter optional formatter useful for printing Symbols
179 */
180 void print(const std::string& s = "", const KeyFormatter& keyFormatter =
181 DefaultKeyFormatter) const override {
182 std::cout << s << "SmartFactorBase, z = \n";
183 for (size_t k = 0; k < measured_.size(); ++k) {
184 std::cout << "measurement " << k<<", px = \n" << measured_[k] << "\n";
185 noiseModel_->print(name: "noise model = ");
186 }
187 if(body_P_sensor_)
188 body_P_sensor_->print(s: "body_P_sensor_:\n");
189 Base::print(s: "", keyFormatter);

192 /// equals
193 bool equals(const NonlinearFactor& p, double tol = 1e-9) const override {
194 if (const This* e = dynamic_cast<const This*>(&p)) {
195 // Check that all measurements are the same.
196 for (size_t i = 0; i < measured_.size(); i++) {
197 if (!traits<Z>::Equals(this->measured_.at(i), e->measured_.at(i), tol))
198 return false;
199 }
200 // If so, check base class.
201 return Base::equals(f: p, tol);
202 } else {
203 return false;
204 }
205 }

207 /** Compute reprojection errors [h(x)-z] = [cameras.project(p)-z] and
208 * derivatives. This is the error before the noise model is applied.
209 * The templated version described above must finally get resolved to this
210 * function.
211 */
212 template <class POINT>
213 Vector unwhitenedError(
214 const Cameras& cameras, const POINT& point,
215 typename Cameras::FBlocks* Fs = nullptr, //
216 Matrix* E = nullptr) const {
217 // Reproject, with optional derivatives.
218 Vector error = cameras.reprojectionError(point, measured_, Fs, E);
219
220 // Apply chain rule if body_P_sensor_ is given.
221 if (body_P_sensor_ && Fs) {
222 const Pose3 sensor_P_body = body_P_sensor_->inverse();
223 constexpr int camera_dim = traits<CAMERA>::dimension;
224 constexpr int pose_dim = traits<Pose3>::dimension;
225
226 for (size_t i = 0; i < Fs->size(); i++) {
227 const Pose3 world_P_body = cameras[i].pose() * sensor_P_body;
228 Eigen::Matrix<double, camera_dim, camera_dim> J;
229 J.setZero();
230 Eigen::Matrix<double, pose_dim, pose_dim> H;
231 // Call compose to compute Jacobian for camera extrinsics
232 world_P_body.compose(g: *body_P_sensor_, H1: H);
233 // Assign extrinsics part of the Jacobian
234 J.template block<pose_dim, pose_dim>(0, 0) = H;
235 Fs->at(i) = Fs->at(i) * J;
236 }
237 }
238
239 // Correct the Jacobians in case some measurements are missing.
240 correctForMissingMeasurements(cameras, error, Fs, E);
241
242 return error;
243 }
244
245 /**
246 * An overload of unwhitenedError. This allows
247 * end users to provide optional arguments that are l-value references
248 * to the matrices and vectors that will be used to store the results instead
249 * of pointers.
250 */
251 template<class POINT, class ...OptArgs, typename = std::enable_if_t<sizeof...(OptArgs)!=0>>
252 Vector unwhitenedError(
253 const Cameras& cameras, const POINT& point,
254 OptArgs&&... optArgs) const {
255 return unwhitenedError(cameras, point, (&optArgs)...);
256 }

258 /**
259 * This corrects the Jacobians for the case in which some 2D measurement is
260 * missing (nan). In practice, this does not do anything in the monocular
261 * case, but it is implemented in the stereo version.
262 */
263 virtual void correctForMissingMeasurements(
264 const Cameras& cameras, Vector& ue,
265 typename Cameras::FBlocks* Fs = nullptr,
266 Matrix* E = nullptr) const {}
267
268 /**
269 * An overload of correctForMissingMeasurements. This allows
270 * end users to provide optional arguments that are l-value references
271 * to the matrices and vectors that will be used to store the results instead
272 * of pointers.
273 */
274 template<class ...OptArgs>
275 void correctForMissingMeasurements(
276 const Cameras& cameras, Vector& ue,
277 OptArgs&&... optArgs) const {
278 correctForMissingMeasurements(cameras, ue, (&optArgs)...);
279 }

281 /**
282 * Calculate vector of re-projection errors [h(x)-z] = [cameras.project(p) -
283 * z], with the noise model applied.
284 */
285 template<class POINT>
286 Vector whitenedError(const Cameras& cameras, const POINT& point) const {
287 Vector error = cameras.reprojectionError(point, measured_);
288 if (noiseModel_)
289 noiseModel_->whitenInPlace(v&: error);
290 return error;
291 }
292
293 /**
294 * Calculate the error of the factor.
295 * This is the log-likelihood, e.g. \f$ 0.5(h(x)-z)^2/\sigma^2 \f$ in case of
296 * Gaussian. In this class, we take the raw prediction error \f$ h(x)-z \f$,
297 * ask the noise model to transform it to \f$ (h(x)-z)^2/\sigma^2 \f$, and
298 * then multiply by 0.5. Will be used in "error(Values)" function required by
299 * NonlinearFactor base class
300 */
301 template<class POINT>
302 double totalReprojectionError(const Cameras& cameras,
303 const POINT& point) const {
304 Vector error = whitenedError(cameras, point);
305 return 0.5 * error.dot(other: error);
306 }
307
308 /// Computes Point Covariance P from the "point Jacobian" E.
309 static Matrix PointCov(const Matrix& E) {
310 return (E.transpose() * E).inverse();
311 }
312
313 /**
314 * Compute F, E, and b (called below in both vanilla and SVD versions), where
315 * F is a vector of derivatives wrpt the cameras, and E the stacked
316 * derivatives with respect to the point. The value of cameras/point are
317 * passed as parameters.
318 */
319 template<class POINT>
320 void computeJacobians(FBlocks& Fs, Matrix& E, Vector& b,
321 const Cameras& cameras, const POINT& point) const {
322 // Project into Camera set and calculate derivatives
323 // As in expressionFactor, RHS vector b = - (h(x_bar) - z) = z-h(x_bar)
324 // Indeed, nonlinear error |h(x_bar+dx)-z| ~ |h(x_bar) + A*dx - z|
325 // = |A*dx - (z-h(x_bar))|
326 b = -unwhitenedError(cameras, point, &Fs, &E);
327 }
328
329 /**
330 * SVD version that produces smaller Jacobian matrices by doing an SVD
331 * decomposition on E, and returning the left nulkl-space of E.
332 * See JacobianFactorSVD for more documentation.
333 * */
334 template<class POINT>
335 void computeJacobiansSVD(FBlocks& Fs, Matrix& Enull,
336 Vector& b, const Cameras& cameras, const POINT& point) const {
337
338 Matrix E;
339 computeJacobians(Fs, E, b, cameras, point);
340
341 static const int N = FixedDimension<POINT>::value; // 2 (Unit3) or 3 (Point3)
342
343 // Do SVD on A.
344 Eigen::JacobiSVD<Matrix> svd(E, Eigen::ComputeFullU);
345 size_t m = this->keys_.size();
346 Enull = svd.matrixU().block(startRow: 0, startCol: N, blockRows: ZDim * m, blockCols: ZDim * m - N); // last ZDim*m-N columns
347 }

349 /// Linearize to a Hessianfactor.
350 // TODO(dellaert): Not used/tested anywhere and not properly whitened.
351 std::shared_ptr<RegularHessianFactor<Dim> > createHessianFactor(
352 const Cameras& cameras, const Point3& point, const double lambda = 0.0,
353 bool diagonalDamping = false) const {
354
355 Matrix E;
356 Vector b;
357 computeJacobians(Fs, E, b, cameras, point);
358
359 // build augmented hessian
360 SymmetricBlockMatrix augmentedHessian = Cameras::SchurComplement(Fs, E, b);
361
362 return std::make_shared<RegularHessianFactor<Dim> >(keys_,
363 augmentedHessian);
364 }
365
366 /**
367 * Add the contribution of the smart factor to a pre-allocated Hessian,
368 * using sparse linear algebra. More efficient than the creation of the
369 * Hessian without preallocation of the SymmetricBlockMatrix
370 */
371 void updateAugmentedHessian(const Cameras& cameras, const Point3& point,
372 const double lambda, bool diagonalDamping,
373 SymmetricBlockMatrix& augmentedHessian,
374 const KeyVector allKeys) const {
375 Matrix E;
376 Vector b;
377 computeJacobians(Fs, E, b, cameras, point);
378 Cameras::UpdateSchurComplement(Fs, E, b, allKeys, keys_, augmentedHessian);
379 }
380
381 /// Whiten the Jacobians computed by computeJacobians using noiseModel_
382 void whitenJacobians(FBlocks& F, Matrix& E, Vector& b) const {
383 noiseModel_->WhitenSystem(A&: E, b);
384 // TODO make WhitenInPlace work with any dense matrix type
385 for (size_t i = 0; i < F.size(); i++)
386 F[i] = noiseModel_->Whiten(H: F[i]);
387 }
388
389 /// Return Jacobians as RegularImplicitSchurFactor with raw access
390 std::shared_ptr<RegularImplicitSchurFactor<CAMERA> > //
391 createRegularImplicitSchurFactor(const Cameras& cameras, const Point3& point,
392 double lambda = 0.0, bool diagonalDamping = false) const {
393 Matrix E;
394 Vector b;
395 FBlocks F;
396 computeJacobians(F, E, b, cameras, point);
397 whitenJacobians(F, E, b);
398 Matrix P = Cameras::PointCov(E, lambda, diagonalDamping);
399 return std::make_shared<RegularImplicitSchurFactor<CAMERA> >(keys_, F, E,
400 P, b);
401 }
402
403 /// Return Jacobians as JacobianFactorQ.
404 std::shared_ptr<JacobianFactorQ<Dim, ZDim> > createJacobianQFactor(
405 const Cameras& cameras, const Point3& point, double lambda = 0.0,
406 bool diagonalDamping = false) const {
407 Matrix E;
408 Vector b;
409 FBlocks F;
410 computeJacobians(F, E, b, cameras, point);
411 const size_t M = b.size();
412 Matrix P = Cameras::PointCov(E, lambda, diagonalDamping);
413 SharedIsotropic n = noiseModel::Isotropic::Sigma(dim: M, sigma: noiseModel_->sigma());
414 return std::make_shared<JacobianFactorQ<Dim, ZDim> >(keys_, F, E, P, b, n);
415 }
416
417 /**
418 * Return Jacobians as JacobianFactorSVD.
419 * TODO(dellaert): lambda is currently ignored
420 */
421 std::shared_ptr<JacobianFactor> createJacobianSVDFactor(
422 const Cameras& cameras, const Point3& point, double lambda = 0.0) const {
423 size_t m = this->keys_.size();
424 FBlocks F;
425 Vector b;
426 const size_t M = ZDim * m;
427 Matrix E0(M, M - 3);
428 computeJacobiansSVD(F, E0, b, cameras, point);
429 SharedIsotropic n = noiseModel::Isotropic::Sigma(dim: M - 3,
430 sigma: noiseModel_->sigma());
431 return std::make_shared<JacobianFactorSVD<Dim, ZDim> >(keys_, F, E0, b, n);
432 }

434 /// Create BIG block-diagonal matrix F from Fblocks
435 static void FillDiagonalF(const FBlocks& Fs, Matrix& F) {
436 size_t m = Fs.size();
437 F.resize(rows: ZDim * m, cols: Dim * m);
438 F.setZero();
439 for (size_t i = 0; i < m; ++i)
440 F.block<ZDim, Dim>(ZDim * i, Dim * i) = Fs.at(i);
441 }

444 Pose3 body_P_sensor() const{
445 if(body_P_sensor_)
446 return *body_P_sensor_;
447 else
448 return Pose3(); // if unspecified, the transformation is the identity
