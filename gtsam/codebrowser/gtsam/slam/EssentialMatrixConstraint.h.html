19#pragma once
20
21#include <gtsam/nonlinear/NonlinearFactor.h>
22#include <gtsam/geometry/EssentialMatrix.h>
23
24namespace gtsam {
25
26/**
27 * Binary factor between two Pose3 variables induced by an EssentialMatrix measurement
28 * @ingroup slam
29 */
30class GTSAM_EXPORT EssentialMatrixConstraint: public NoiseModelFactorN<Pose3, Pose3> {
31
32private:
33
34 typedef EssentialMatrixConstraint This;
35 typedef NoiseModelFactorN<Pose3, Pose3> Base;
36
37 EssentialMatrix measuredE_; /** The measurement is an essential matrix */
38
39public:
40
41 // Provide access to the Matrix& version of evaluateError:
42 using Base::evaluateError;
43
44 // shorthand for a smart pointer to a factor
45 typedef std::shared_ptr<EssentialMatrixConstraint> shared_ptr;
46
47 /** default constructor - only use for serialization */
48 EssentialMatrixConstraint() {
49 }
50
51 /**
52 * Constructor
53 * @param key1 key for first pose
54 * @param key2 key for second pose
55 * @param measuredE measured EssentialMatrix
56 * @param model noise model, 5D !
57 */
58 EssentialMatrixConstraint(Key key1, Key key2,
59 const EssentialMatrix& measuredE, const SharedNoiseModel& model) :
60 Base(model, key1, key2), measuredE_(measuredE) {
61 }
62
63 ~EssentialMatrixConstraint() override {
64 }
65
66 /// @return a deep copy of this factor
67 gtsam::NonlinearFactor::shared_ptr clone() const override {
68 return std::static_pointer_cast<gtsam::NonlinearFactor>(
69 r: gtsam::NonlinearFactor::shared_ptr(new This(*this)));
70 }
71
72 /** implement functions needed for Testable */
73
74 /** print */
75 void print(const std::string& s = "",
76 const KeyFormatter& keyFormatter = DefaultKeyFormatter) const override;
77
78 /** equals */
79 bool equals(const NonlinearFactor& expected, double tol = 1e-9) const override;
80
81 /** implement functions needed to derive from Factor */
82
83 /** vector of errors */
84 Vector evaluateError(const Pose3& p1, const Pose3& p2,
85 OptionalMatrixType Hp1, OptionalMatrixType Hp2) const override;
86
87 /** return the measured */
88 const EssentialMatrix& measured() const {
89 return measuredE_;
90 }
91
92private:
93
94#if GTSAM_ENABLE_BOOST_SERIALIZATION
95 /** Serialization function */
96 friend class boost::serialization::access;
97 template<class ARCHIVE>
98 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
99 // NoiseModelFactor2 instead of NoiseModelFactorN for backward compatibility
100 ar
101 & boost::serialization::make_nvp(n: "NoiseModelFactor2",
102 v&: boost::serialization::base_object<Base>(d&: *this));
103 ar & BOOST_SERIALIZATION_NVP(measuredE_);
104 }
105#endif
106
107public:
108 GTSAM_MAKE_ALIGNED_OPERATOR_NEW
109};
110// \class EssentialMatrixConstraint
111
112}/// namespace gtsam
