24//***************************************************************************
25Vector OrientedPlane3Factor::evaluateError(const Pose3& pose,
26 const OrientedPlane3& plane, OptionalMatrixType H1,
27 OptionalMatrixType H2) const {
28 Matrix36 predicted_H_pose;
29 Matrix33 predicted_H_plane, error_H_predicted;
30
31 OrientedPlane3 predicted_plane = plane.transform(xr: pose,
32 Hp: H2 ? &predicted_H_plane : nullptr, Hr: H1 ? &predicted_H_pose : nullptr);
33
34 Vector3 err = predicted_plane.errorVector(
35 other: measured_p_, H1: (H1 || H2) ? &error_H_predicted : nullptr);
36
37 // Apply the chain rule to calculate the derivatives.
38 if (H1) {
39 *H1 = error_H_predicted * predicted_H_pose;
40 }
41 if (H2) {
42 *H2 = error_H_predicted * predicted_H_plane;
43 }
44
45 return err;
46}

65//***************************************************************************
66Vector OrientedPlane3DirectionPrior::evaluateError(
67 const OrientedPlane3& plane, OptionalMatrixType H) const {
68 Unit3 n_hat_p = measured_p_.normal();
69 Unit3 n_hat_q = plane.normal();
70 Matrix2 H_p;
71 Vector e = n_hat_p.errorVector(q: n_hat_q, H_p: {}, H_q: H ? &H_p : nullptr);
72 if (H) {
73 H->resize(rows: 2, cols: 3);
74 *H << H_p, Z_2x1;
75 }
76 return e;
77}
