12/**
13 * @file lago.h
14 * @brief Initialize Pose2 in a factor graph using LAGO
15 * (Linear Approximation for Graph Optimization). see papers:
16 *
17 * L. Carlone, R. Aragues, J. Castellanos, and B. Bona, A fast and accurate
18 * approximation for planar pose graph optimization, IJRR, 2014.
19 *
20 * L. Carlone, R. Aragues, J.A. Castellanos, and B. Bona, A linear approximation
21 * for graph-based simultaneous localization and mapping, RSS, 2011.
22 *
23 * @param graph: nonlinear factor graph (can include arbitrary factors but we assume
24 * that there is a subgraph involving Pose2 and betweenFactors). Also in the current
25 * version we assume that there is an odometric spanning path (x0->x1, x1->x2, etc)
26 * and a prior on x0. This assumption can be relaxed by using the extra argument
27 * useOdometricPath = false, although this part of code is not stable yet.
28 * @return Values: initial guess from LAGO (only pose2 are initialized)
29 *
30 * @author Luca Carlone
31 * @author Frank Dellaert
32 * @date May 14, 2014
33 */
34
35#pragma once
36
37#include <gtsam/nonlinear/NonlinearFactorGraph.h>
38#include <gtsam/linear/GaussianFactorGraph.h>
39#include <gtsam/linear/VectorValues.h>
40
41namespace gtsam {
42namespace lago {
43
44typedef std::map<Key, double> key2doubleMap;
45typedef std::map<Key, Key> PredecessorMap;
46
47/**
48 * Compute the cumulative orientations (without wrapping)
49 * for all nodes wrt the root (root has zero orientation).
50 */
51GTSAM_EXPORT key2doubleMap computeThetasToRoot(
52 const key2doubleMap& deltaThetaMap, const PredecessorMap& tree);
53
54/**
55 * Given a factor graph "g", and a spanning tree "tree", select the nodes belonging
56 * to the tree and to g, and stores the factor slots corresponding to edges in the
57 * tree and to chordsIds wrt this tree.
58 * Also it computes deltaThetaMap which is a fast way to encode relative
59 * orientations along the tree: for a node key2, s.t. tree[key2]=key1,
60 * the value deltaThetaMap[key2] is relative orientation theta[key2]-theta[key1]
61 */
62GTSAM_EXPORT void getSymbolicGraph(
63/*OUTPUTS*/std::vector<size_t>& spanningTreeIds, std::vector<size_t>& chordsIds,
64 key2doubleMap& deltaThetaMap,
65 /*INPUTS*/const PredecessorMap& tree, const NonlinearFactorGraph& g);
66
67/** Linear factor graph with regularized orientation measurements */
68GTSAM_EXPORT GaussianFactorGraph buildLinearOrientationGraph(
69 const std::vector<size_t>& spanningTreeIds,
70 const std::vector<size_t>& chordsIds, const NonlinearFactorGraph& g,
71 const key2doubleMap& orientationsToRoot, const PredecessorMap& tree);
72
73/** Given a "pose2" factor graph, find its minimum spanning tree.
74 * Note: All 'Pose2' Between factors are given equal weightage.
75 * Note: Assumes all the edges (factors) are Between factors.
76 */
77GTSAM_EXPORT PredecessorMap findMinimumSpanningTree(
78 const NonlinearFactorGraph& pose2Graph);
79
80/** LAGO: Return the orientations of the Pose2 in a generic factor graph */
81GTSAM_EXPORT VectorValues initializeOrientations(
82 const NonlinearFactorGraph& graph, bool useOdometricPath = true);
83
84/** Return the values for the Pose2 in a generic factor graph */
85GTSAM_EXPORT Values initialize(const NonlinearFactorGraph& graph,
86 bool useOdometricPath = true);
87
88/** Only correct the orientation part in initialGuess */
89GTSAM_EXPORT Values initialize(const NonlinearFactorGraph& graph,
90 const Values& initialGuess);
91
92} // end of namespace lago
93} // end of namespace gtsam
