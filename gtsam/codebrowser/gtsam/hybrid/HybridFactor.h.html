41/**
42 * Base class for *truly* hybrid probabilistic factors
43 *
44 * Examples:
45 * - HybridNonlinearFactor
46 * - HybridGaussianFactor
47 * - HybridGaussianConditional
48 *
49 * @ingroup hybrid
50 */
51class GTSAM_EXPORT HybridFactor : public Factor {
52 public:
53 /// Enum to help with categorizing hybrid factors.
54 enum class Category { None, Discrete, Continuous, Hybrid };
55
56 private:
57 /// Record what category of HybridFactor this is.
58 Category category_ = Category::None;
59
60 protected:
61 // Set of DiscreteKeys for this factor.
62 DiscreteKeys discreteKeys_;
63 /// Record continuous keys for book-keeping
64 KeyVector continuousKeys_;
65
66 public:
67 // typedefs needed to play nice with gtsam
68 typedef HybridFactor This; ///< This class
69 typedef std::shared_ptr<HybridFactor>
70 shared_ptr; ///< shared_ptr to this class
71 typedef Factor Base; ///< Our base class
72
73 /// @name Standard Constructors
74 /// @{
75
76 /** Default constructor creates empty factor */
77 HybridFactor() = default;
78
79 /**
80 * @brief Construct hybrid factor from continuous keys.
81 *
82 * @param keys Vector of continuous keys.
83 */
84 explicit HybridFactor(const KeyVector &keys);
85
86 /**
87 * @brief Construct hybrid factor from discrete keys.
88 *
89 * @param keys Vector of discrete keys.
90 */
91 explicit HybridFactor(const DiscreteKeys &discreteKeys);
92
93 /**
94 * @brief Construct a new Hybrid Factor object.
95 *
96 * @param continuousKeys Vector of keys for continuous variables.
97 * @param discreteKeys Vector of keys for discrete variables.
98 */
99 HybridFactor(const KeyVector &continuousKeys,
100 const DiscreteKeys &discreteKeys);
101
102 /// @}
103 /// @name Testable
104 /// @{
105
106 /// equals
107 virtual bool equals(const HybridFactor &lf, double tol = 1e-9) const;
108
109 /// print
110 void print(
111 const std::string &s = "HybridFactor\n",
112 const KeyFormatter &formatter = DefaultKeyFormatter) const override;
113
114 /// @}
115 /// @name Standard Interface
116 /// @{
117
118 /// True if this is a factor of discrete variables only.
119 bool isDiscrete() const { return category_ == Category::Discrete; }
120
121 /// True if this is a factor of continuous variables only.
122 bool isContinuous() const { return category_ == Category::Continuous; }
123
124 /// True is this is a Discrete-Continuous factor.
125 bool isHybrid() const { return category_ == Category::Hybrid; }
126
127 /// Return the number of continuous variables in this factor.
128 size_t nrContinuous() const { return continuousKeys_.size(); }
129
130 /// Return the discrete keys for this factor.
131 const DiscreteKeys &discreteKeys() const { return discreteKeys_; }
132
133 /// Return only the continuous keys for this factor.
134 const KeyVector &continuousKeys() const { return continuousKeys_; }
135
136 /// Compute tree of linear errors.
137 virtual AlgebraicDecisionTree<Key> errorTree(
138 const VectorValues &values) const = 0;
139
140 /// Restrict the factor to the given discrete values.
141 virtual std::shared_ptr<Factor> restrict(
142 const DiscreteValues &discreteValues) const = 0;
143
144 /// @}
145
146 private:
147#if GTSAM_ENABLE_BOOST_SERIALIZATION
148 /** Serialization function */
149 friend class boost::serialization::access;
150 template <class ARCHIVE>
151 void serialize(ARCHIVE &ar, const unsigned int /*version*/) {
152 ar &BOOST_SERIALIZATION_BASE_OBJECT_NVP(Base);
153 ar &BOOST_SERIALIZATION_NVP(category_);
154 ar &BOOST_SERIALIZATION_NVP(discreteKeys_);
155 ar &BOOST_SERIALIZATION_NVP(continuousKeys_);
156 }
157#endif
158};
