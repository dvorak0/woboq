22/* ************************************************************************ */
23DecisionTreeFactor DiscreteFactorFromErrors(
24 const DiscreteKeys& discreteKeys,
25 const AlgebraicDecisionTree<Key>& errors) {
26 double min_log = errors.min();
27 AlgebraicDecisionTree<Key> potentials(
28 errors, [&min_log](const double x) { return exp(x: -(x - min_log)); });
29
30 return DecisionTreeFactor(discreteKeys, potentials);
31}

33/* ************************************************************************ */
34DecisionTreeFactor DiscreteBoundaryFactor::ComputeDiscreteBoundary(
35 const DiscreteKeys& dkeys,
36 const HybridNonlinearFactor::FactorValuePairs& factors,
37 const gtsam::Values& values) {
38 auto calculateError =
39 [&values](const NonlinearFactorValuePair& pair) -> double {
40 if (pair.first) {
41 // `error` has the following contributions:
42 // - the scalar is the sum of all mode-dependent constants (including log
43 // normalization constant)
44 // - factor->error(initial) is the error on the initial values
45 return pair.second + pair.first->error(c: values);
46 } else {
47 // If the factor has been pruned, return infinite error
48 return std::numeric_limits<double>::infinity();
49 }
50 };
51 AlgebraicDecisionTree<Key> errors(factors, calculateError);
52 return DiscreteFactorFromErrors(discreteKeys: dkeys, errors);
53}
