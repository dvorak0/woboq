25/* ************************************************************************* */
26std::set<DiscreteKey> HybridFactorGraph::discreteKeys() const {
27 std::set<DiscreteKey> keys;
28 for (auto& factor : factors_) {
29 if (auto p = std::dynamic_pointer_cast<DiscreteFactor>(r: factor)) {
30 for (const DiscreteKey& key : p->discreteKeys()) {
31 keys.insert(x: key);
32 }
33 } else if (auto p = std::dynamic_pointer_cast<HybridFactor>(r: factor)) {
34 for (const DiscreteKey& key : p->discreteKeys()) {
35 keys.insert(x: key);
36 }
37 }
38 }
39 return keys;
40}
41
42/* ************************************************************************* */
43KeySet HybridFactorGraph::discreteKeySet() const {
44 KeySet keys;
45 std::set<DiscreteKey> key_set = discreteKeys();
46 std::transform(first: key_set.begin(), last: key_set.end(),
47 result: std::inserter(x&: keys, i: keys.begin()),
48 unary_op: [](const DiscreteKey& k) { return k.first; });
49 return keys;
50}
51
52/* ************************************************************************* */
53const KeySet HybridFactorGraph::continuousKeySet() const {
54 KeySet keys;
55 for (auto& factor : factors_) {
56 if (auto p = std::dynamic_pointer_cast<HybridFactor>(r: factor)) {
57 for (const Key& key : p->continuousKeys()) {
58 keys.insert(x: key);
59 }
60 } else if (auto p = std::dynamic_pointer_cast<GaussianFactor>(r: factor)) {
61 keys.insert(first: p->keys().begin(), last: p->keys().end());
62 } else if (auto p = std::dynamic_pointer_cast<NonlinearFactor>(r: factor)) {
63 keys.insert(first: p->keys().begin(), last: p->keys().end());
64 }
65 }
66 return keys;
67}
