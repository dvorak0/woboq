42Ordering HybridGaussianISAM::GetOrdering(
43 HybridGaussianFactorGraph& factors,
44 const HybridGaussianFactorGraph& newFactors) {
45 // Get all the discrete keys from the factors
46 const KeySet allDiscrete = factors.discreteKeySet();
47
48 // Create KeyVector with continuous keys followed by discrete keys.
49 KeyVector newKeysDiscreteLast;
50 const KeySet newFactorKeys = newFactors.keys();
51 // Insert continuous keys first.
52 for (auto& k : newFactorKeys) {
53 if (!allDiscrete.exists(e: k)) {
54 newKeysDiscreteLast.push_back(x: k);
55 }
56 }
57 // Insert discrete keys at the end
58 std::copy(first: allDiscrete.begin(), last: allDiscrete.end(),
59 result: std::back_inserter(x&: newKeysDiscreteLast));
60
61 const VariableIndex index(factors);
62
63 // Get an ordering where the new keys are eliminated last
64 Ordering ordering = Ordering::ColamdConstrainedLast(
65 variableIndex: index, constrainLast: KeyVector(newKeysDiscreteLast.begin(), newKeysDiscreteLast.end()),
66 forceOrder: true);
67 return ordering;
68}

70/* ************************************************************************* */
71void HybridGaussianISAM::updateInternal(
72 const HybridGaussianFactorGraph& newFactors,
73 HybridBayesTree::Cliques* orphans, const std::optional<size_t>& maxNrLeaves,
74 const std::optional<Ordering>& ordering,
75 const HybridBayesTree::Eliminate& function) {
76 // Remove the contaminated part of the Bayes tree
77 BayesNetType bn;
78 const KeySet newFactorKeys = newFactors.keys();
79 if (!this->empty()) {
80 KeyVector keyVector(newFactorKeys.begin(), newFactorKeys.end());
81 this->removeTop(keys: keyVector, bn: &bn, orphans);
82 }
83
84 // Add the removed top and the new factors
85 HybridGaussianFactorGraph factors;
86 factors.push_back(container: bn);
87 factors.push_back(container: newFactors);
88
89 // Add the orphaned subtrees
90 for (const sharedClique& orphan : *orphans) {
91 factors.emplace_shared<BayesTreeOrphanWrapper<Node>>(args: orphan);
92 }
93
94 const VariableIndex index(factors);
95 Ordering elimination_ordering;
96 if (ordering) {
97 elimination_ordering = *ordering;
98 } else {
99 elimination_ordering = GetOrdering(factors, newFactors);
100 }
101
102 // eliminate all factors (top, added, orphans) into a new Bayes tree
103 HybridBayesTree::shared_ptr bayesTree = factors.eliminateMultifrontal(
104 ordering: elimination_ordering, function, variableIndex: std::cref(t: index));
105
106 if (maxNrLeaves) {
107#if GTSAM_HYBRID_TIMING
108 gttic_(HybridBayesTreePrune);
109#endif
110 bayesTree->prune(maxNumberLeaves: *maxNrLeaves);
111#if GTSAM_HYBRID_TIMING
112 gttoc_(HybridBayesTreePrune);
113#endif
114 }
115
116 // Re-add into Bayes tree data structures
117 this->roots_.insert(position: this->roots_.end(), first: bayesTree->roots().begin(),
118 last: bayesTree->roots().end());
119 this->nodes_.insert(first: bayesTree->nodes().begin(), last: bayesTree->nodes().end());
120}

122/* ************************************************************************* */
123void HybridGaussianISAM::update(const HybridGaussianFactorGraph& newFactors,
124 const std::optional<size_t>& maxNrLeaves,
125 const std::optional<Ordering>& ordering,
126 const HybridBayesTree::Eliminate& function) {
127 Cliques orphans;
128 this->updateInternal(newFactors, orphans: &orphans, maxNrLeaves, ordering, function);
129}
