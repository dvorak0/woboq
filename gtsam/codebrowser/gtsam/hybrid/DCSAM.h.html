41/**
42 * @brief Class which implements Discrete-Continuous Smoothing And Mapping,
43 * as detailed in Doherty22ral (https://arxiv.org/abs/2204.11936).
44 *
45 * Performs hybrid optimization by alternatively optimizing
46 * for continous and discrete variables.
47 *
48 */
49class GTSAM_EXPORT DCSAM {
50 private:
51 /// The factor graph for all continuous factors
52 NonlinearFactorGraph nfg_;
53 /// The factor graph for all discrete factors
54 DiscreteFactorGraph dfg_;
55 /// The factor graph for hybrid factors
56 HybridNonlinearFactorGraph hfg_;
57
58 // TODO(Varun): Allow for using other continuous and discrete solvers
59
60 /// ISAM2 optimizer for continuous optimization
61 ISAM2 isam_;
62
63 /// The current best estimate for continuous values
64 Values currContinuous_;
65 /// The current best estimate for discrete values
66 DiscreteValues currDiscrete_;
67
68 public:
69 DCSAM();
70
71 explicit DCSAM(const ISAM2Params &isam_params);
72
73 /**
74 * For this solver, runs an iteration of alternating minimization between
75 * discrete and continuous variables, adding any user-supplied factors (with
76 * initial guess) first.
77 *
78 * 1. Adds new discrete factors (if any) as supplied by a user to the
79 * discrete factor graph, then adds any discrete-continuous factors to the
80 * discrete factor graph, appropriately initializing their continuous
81 * variables to those from the last solve and any supplied by the initial
82 * guess.
83 *
84 * 2. Update the solution for the discrete variables.
85 *
86 * 3. For all new discrete-continuous factors to be passed to the continuous
87 * solver, update/set the latest discrete variables (prior to adding).
88 *
89 * 4. In one step: add new factors, new values, and earmarked old factor keys
90 * to iSAM. Specifically, loop over DC factors already in iSAM, updating their
91 * discrete information, then call isam_.update() with the (initialized) new
92 * DC factors, any new continuous factors, and the initial guess to be
93 * supplied.
94 *
95 * 5. Calculate the latest continuous variables from iSAM.
96 *
97 * 6. Update the discrete factors in the discrete factor graph dfg_ with the
98 * latest information from the continuous solve.
99 *
100 * @param graph - a HybridNonlinearFactorGraph containing all factors to add.
101 * @param initialGuessContinuous - an initial guess for any new
102 * continuous keys that appear in the updated factors
103 * (or if one wants to force override previously obtained continuous values).
104 * @param initialGuessDiscrete - Initial guess for discrete variables.
105 */
106 void update(const HybridNonlinearFactorGraph &graph,
107 const HybridValues &initialGuess = HybridValues());
108
109 /**
110 * Inline convenience function to allow "skipping" the initial guess for
111 * continuous variables while adding an initial guess for discrete variables.
112 */
113 inline void update(const HybridNonlinearFactorGraph &graph,
114 const DiscreteValues &initialGuessDiscrete) {
115 update(graph, initialGuess: HybridValues(VectorValues(), initialGuessDiscrete));
116 }
117
118 /**
119 * Simply used to call `update` without any new factors. Runs an iteration of
120 * optimization.
121 */
122 void update();
123
124 /**
125 * This is the primary function used to extract an estimate from the solver.
126 * Internally, calls `isam_.calculateEstimate()` and `dfg_.optimize()`
127 * to obtain an estimate for the continuous (resp. discrete) variables
128 * and packages them into a `HybridValues`.
129 *
130 * @return a HybridValues object containing an estimate
131 * of the most probable assignment to the continuous (HybridValues.nonlinear)
132 * and discrete (HybridValues.discrete) variables.
133 */
134 HybridValues calculateEstimate() const;
135
136 /**
137 * Used to obtain the marginals from the solver.
138 *
139 * NOTE: not obviously correct (see DCSAM.cpp implementation) at the moment.
140 * Should perhaps retrieve the marginals for the factor graph obtained as
141 * isam_.getFactorsUnsafe() and `dfg_` rather than taking as a parameter?
142 * I think this was originally intended to mimic the gtsam `Marginals` class.
143 *
144 * @param graph
145 * @param continuousEst
146 * @param dfg
147 */
148 // DCMarginals getMarginals(const NonlinearFactorGraph &graph,
149 // const Values &continuousEst,
150 // const DiscreteFactorGraph &dfg);
151
152 const DiscreteFactorGraph &getDiscreteFactorGraph() const { return dfg_; }
153
154 const NonlinearFactorGraph &getNonlinearFactorGraph() const {
155 return isam_.getFactorsUnsafe();
156 }
157
158 protected:
159 /**
160 * Add factors in `dfg` to member discrete factor graph `dfg_`,
161 * then optimize for most probable explanation and
162 * update the current discrete estimate.
163 *
164 * @param dfg - A discrete factor graph containing the factors to add
165 * @param discreteValues - An assignment to the continuous variables
166 * (or subset thereof).
167 */
168 void updateDiscrete(const DiscreteFactorGraph &dfg = DiscreteFactorGraph(),
169 const DiscreteValues &discreteVals = DiscreteValues());
170
171 /**
172 * Given the latest discrete values (currDiscrete_),
173 * a set of new factors (newFactors),
174 * and an initial guess for any new keys (initialGuess),
175 * this method
176 * 1. Selects the appropriate continuous factors from the hybridfactors
177 * 2. Marks any affected keys as such
178 * 3. Calls `isam_.update` with the new factors and initial guess.
179 *
180 * See implementation for more detail.
181 */
182 void updateContinuous(const NonlinearFactorGraph &newFactors,
183 const Values &initialGuess);
184
185 /**
186 * Solve for discrete variables given continuous variables.
187 *
188 * Internally, this method computes DiscreteBoundaryFactors from the hybrid
189 * factors using the current continuous estimate `currContinuous_`, and then
190 * calls `dfg_.optimize()` to get the most probable estimate.
191 *
192 * @return An assignment (DiscreteValues) to the discrete variables
193 * in the graph.
194 */
195 DiscreteValues solveDiscrete() const;
196};
