49/**
50 * @brief Main elimination function for HybridGaussianFactorGraph.
51 *
52 * @param factors The factor graph to eliminate.
53 * @param keys The elimination ordering.
54 * @return The conditional on the ordering keys and the remaining factors.
55 * @ingroup hybrid

59EliminateHybrid(const HybridGaussianFactorGraph& factors, const Ordering& keys);

71template <>
72struct EliminationTraits<HybridGaussianFactorGraph> {
73 typedef Factor FactorType; ///< Type of factors in factor graph
74 typedef HybridGaussianFactorGraph
75 FactorGraphType; ///< Type of the factor graph (e.g.
76 ///< HybridGaussianFactorGraph)
77 typedef HybridConditional
78 ConditionalType; ///< Type of conditionals from elimination
79 typedef HybridBayesNet
80 BayesNetType; ///< Type of Bayes net from sequential elimination
81 typedef HybridEliminationTree
82 EliminationTreeType; ///< Type of elimination tree
83 typedef HybridBayesTree BayesTreeType; ///< Type of Bayes tree
84 typedef HybridJunctionTree JunctionTreeType; ///< Type of Junction tree
85 /// The default dense elimination function
86 static std::pair<std::shared_ptr<ConditionalType>,
87 std::shared_ptr<FactorType>>
88 DefaultEliminate(const FactorGraphType& factors, const Ordering& keys) {
89 return EliminateHybrid(factors, keys);
90 }
91 /// The default ordering generation function
92 static Ordering DefaultOrderingFunc(
93 const FactorGraphType& graph,
94 std::optional<std::reference_wrapper<const VariableIndex>>) {
95 return HybridOrdering(graph);
96 }
97};

99/**
100 * Hybrid Gaussian Factor Graph
101 * -----------------------
102 * This is the linearized version of a hybrid factor graph.
103 *
104 * @ingroup hybrid
105 */
106class GTSAM_EXPORT HybridGaussianFactorGraph
107 : public HybridFactorGraph,
108 public EliminateableFactorGraph<HybridGaussianFactorGraph> {
109 protected:
110 /// Check if FACTOR type is derived from GaussianFactor.
111 template <typename FACTOR>
112 using IsGaussian = typename std::enable_if<
113 std::is_base_of<GaussianFactor, FACTOR>::value>::type;
114
115 public:
116 using Base = HybridFactorGraph;
117 using This = HybridGaussianFactorGraph; ///< this class
118 ///< for elimination
119 using BaseEliminateable = EliminateableFactorGraph<This>;
120 using shared_ptr = std::shared_ptr<This>; ///< shared_ptr to This
121
122 using Values = gtsam::Values; ///< backwards compatibility
123 using Indices = KeyVector; ///< map from keys to values
124
125 /// @name Constructors
126 /// @{
127
128 /// @brief Default constructor.
129 HybridGaussianFactorGraph() = default;
130
131 /** Construct from container of factors (shared_ptr or plain objects) */
132 template <class CONTAINER>
133 explicit HybridGaussianFactorGraph(const CONTAINER& factors)
134 : Base(factors) {}
135
136 /**
137 * Construct from an initializer lists of GaussianFactor shared pointers.
138 * Example:
139 * HybridGaussianFactorGraph graph = { factor1, factor2, factor3 };
140 */
141 HybridGaussianFactorGraph(std::initializer_list<sharedFactor> factors)
142 : Base(factors) {}
143
144 /**
145 * Implicit copy/downcast constructor to override explicit template container
146 * constructor. In BayesTree this is used for:
147 * `cachedSeparatorMarginal_.reset(*separatorMarginal)`
148 * */
149 template <class DERIVEDFACTOR>
150 HybridGaussianFactorGraph(const FactorGraph<DERIVEDFACTOR>& graph)
151 : Base(graph) {}
152
153 /// @}
154 /// @name Testable
155 /// @{
156
157 void print(
158 const std::string& s = "HybridGaussianFactorGraph",
159 const KeyFormatter& keyFormatter = DefaultKeyFormatter) const override;
160
161 /**
162 * @brief Print the errors of each factor in the hybrid factor graph.
163 *
164 * @param values The HybridValues for the variables used to compute the error.
165 * @param str String that is output before the factor graph and errors.
166 * @param keyFormatter Formatter function for the keys in the factors.
167 * @param printCondition A condition to check if a factor should be printed.
168 */
169 void printErrors(
170 const HybridValues& values,
171 const std::string& str = "HybridGaussianFactorGraph: ",
172 const KeyFormatter& keyFormatter = DefaultKeyFormatter,
173 const std::function<bool(const Factor* /*factor*/,
174 double /*whitenedError*/, size_t /*index*/)>&
175 printCondition =
176 [](const Factor*, double, size_t) { return true; }) const;
177
178 // bool equals(const This& fg, double tol = 1e-9) const override;
179
180 /// @}
181 /// @name Standard Interface
182 /// @{
183
184 /// Expose error(const HybridValues&) method.
185 using Base::error;
186
187 /**
188 * @brief Compute error for each discrete assignment,
189 * and return as a tree.
190 *
191 * Error \f$ e = \Vert x - \mu \Vert_{\Sigma} \f$.
192 *
193 * @param continuousValues Continuous values at which to compute the error.
194 * @return AlgebraicDecisionTree<Key>
195 */
196 AlgebraicDecisionTree<Key> errorTree(
197 const VectorValues& continuousValues) const;
198
199 /**
200 * @brief Compute the unnormalized posterior probability for a continuous
201 * vector values given a specific assignment.
202 *
203 * @return double
204 */
205 double probPrime(const HybridValues& values) const;
206
207 /**
208 * @brief Computer posterior P(M|X=x) when all continuous values X are given.
209 * This is efficient as this simply probPrime normalized.
210 *
211 * @note Not a DiscreteConditional as the cardinalities of the DiscreteKeys,
212 * which we would need, are hard to recover.
213 *
214 * @param continuousValues Continuous values x to condition on.
215 * @return DecisionTreeFactor
216 */
217 AlgebraicDecisionTree<Key> discretePosterior(
218 const VectorValues& continuousValues) const;
219
220 /**
221 * @brief Create a decision tree of factor graphs out of this hybrid factor
222 * graph.
223 *
224 * For example, if there are two hybrid factors, one with a discrete key A
225 * and one with a discrete key B, then the decision tree will have two levels,
226 * one for A and one for B. The leaves of the tree will be the Gaussian
227 * factors that have only continuous keys.
228 */
229 HybridGaussianProductFactor collectProductFactor() const;
230
231 /**
232 * @brief Eliminate the given continuous keys.
233 *
234 * @param keys The continuous keys to eliminate.
235 * @return The conditional on the keys and a factor on the separator.
236 */
237 std::pair<std::shared_ptr<HybridConditional>, std::shared_ptr<Factor>>
238 eliminate(const Ordering& keys) const;
239 /// @}
240
241 /**
242 @brief Get the GaussianFactorGraph at a given discrete assignment. Note this
243 * corresponds to the Gaussian posterior p(X|M=m, Z=z) of the continuous
244 * variables X given the discrete assignment M=m and whatever measurements z
245 * where assumed in the creation of the factor Graph.
246 *
247 * @note Be careful, as any factors not Gaussian are ignored.
248 *
249 * @param assignment The discrete value assignment for the discrete keys.
250 * @return Gaussian factors as a GaussianFactorGraph
251 */
252 GaussianFactorGraph choose(const DiscreteValues& assignment) const;
253
254 /// Syntactic sugar for choose
255 GaussianFactorGraph operator()(const DiscreteValues& assignment) const {
256 return choose(assignment);
257 }
258
259 /**
260 * @brief Helper method to get all the discrete factors
261 * as a DiscreteFactorGraph.
262 *
263 * @return DiscreteFactorGraph
264 */
265 DiscreteFactorGraph discreteFactors() const;
266};
