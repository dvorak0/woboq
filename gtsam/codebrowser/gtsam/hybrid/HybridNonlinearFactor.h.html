23#include <gtsam/hybrid/HybridGaussianFactor.h>
24#include <gtsam/hybrid/HybridValues.h>
25#include <gtsam/nonlinear/NonlinearFactor.h>

31namespace gtsam {
32
33/// Alias for a NoiseModelFactor shared pointer and double scalar pair.
34using NonlinearFactorValuePair =
35 std::pair<NoiseModelFactor::shared_ptr, double>;
36
37/**
38 * @brief Implementation of a discrete-conditioned hybrid factor.
39 *
40 * Implements a joint discrete-continuous factor where the discrete variable
41 * serves to "select" a hybrid component corresponding to a NoiseModelFactor.
42 *
43 * This class stores all factors as HybridFactors which can then be typecast to
44 * one of (NoiseModelFactor, GaussianFactor) which can then be checked to
45 * perform the correct operation.
46 *
47 * In factor graphs the error function typically returns 0.5*|h(x)-z|^2, i.e.,
48 * the negative log-likelihood for a Gaussian noise model.
49 * In hybrid factor graphs we allow *adding* an arbitrary scalar dependent on
50 * the discrete assignment.
51 * For example, adding a 70/30 mode probability is supported by providing the
52 * scalars $-log(.7)$ and $-log(.3)$.
53 * Note that adding a common constant will not make any difference in the
54 * optimization, so $-log(70)$ and $-log(30)$ work just as well.
55 *
56 * @ingroup hybrid
57 */
58class GTSAM_EXPORT HybridNonlinearFactor : public HybridFactor {
59 public:
60 using Base = HybridFactor;
61 using This = HybridNonlinearFactor;
62 using shared_ptr = std::shared_ptr<HybridNonlinearFactor>;
63 using sharedFactor = std::shared_ptr<NoiseModelFactor>;
64
65 /**
66 * @brief typedef for DecisionTree which has Keys as node labels and
67 * pairs of NoiseModelFactor & an arbitrary scalar as leaf nodes.
68 */
69 using FactorValuePairs = DecisionTree<Key, NonlinearFactorValuePair>;
70
71 private:
72 /// Decision tree of nonlinear factors indexed by discrete keys.
73 FactorValuePairs factors_;
74
75 /// HybridFactor method implementation. Should not be used.
76 AlgebraicDecisionTree<Key> errorTree(
77 const VectorValues& continuousValues) const override {
78 throw std::runtime_error(
79 "HybridNonlinearFactor::error does not take VectorValues.");
80 }
81
82 public:
83 /// @name Constructors
84 /// @{
85
86 /// Default constructor, mainly for serialization.
87 HybridNonlinearFactor() = default;
88
89 /**
90 * @brief Construct a new HybridNonlinearFactor on a single discrete key,
91 * providing the factors for each mode m as a vector of factors ϕ_m(x).
92 * The value ϕ(x,m) for the factor is simply ϕ_m(x).
93 *
94 * @param discreteKey The discrete key for the "mode", indexing components.
95 * @param factors Vector of gaussian factors, one for each mode.
96 */
97 HybridNonlinearFactor(
98 const DiscreteKey& discreteKey,
99 const std::vector<NoiseModelFactor::shared_ptr>& factors);
100
101 /**
102 * @brief Construct a new HybridNonlinearFactor on a single discrete key,
103 * including a scalar error value for each mode m. The factors and scalars are
104 * provided as a vector of pairs (ϕ_m(x), E_m).
105 * The value ϕ(x,m) for the factor is now ϕ_m(x) + E_m.
106 *
107 * @param discreteKey The discrete key for the "mode", indexing components.
108 * @param pairs Vector of gaussian factor-scalar pairs, one per mode.
109 */
110 HybridNonlinearFactor(const DiscreteKey& discreteKey,
111 const std::vector<NonlinearFactorValuePair>& pairs);
112
113 /**
114 * @brief Construct a new HybridNonlinearFactor on a several discrete keys M,
115 * including a scalar error value for each assignment m. The factors and
116 * scalars are provided as a DecisionTree<Key> of pairs (ϕ_M(x), E_M).
117 * The value ϕ(x,M) for the factor is again ϕ_m(x) + E_m.
118 *
119 * @param discreteKeys Discrete variables and their cardinalities.
120 * @param factors The decision tree of nonlinear factor/scalar pairs.
121 */
122 HybridNonlinearFactor(const DiscreteKeys& discreteKeys,
123 const FactorValuePairs& factors);
124
125 /**
126 * @brief Compute error of the HybridNonlinearFactor as a tree.
127 *
128 * @param continuousValues The continuous values for which to compute the
129 * error.
130 * @return AlgebraicDecisionTree<Key> A decision tree with the same keys
131 * as the factor, and leaf values as the error.
132 */
133 AlgebraicDecisionTree<Key> errorTree(const Values& continuousValues) const;
134
135 /**
136 * @brief Compute error of factor given both continuous and discrete values.
137 *
138 * @param continuousValues The continuous Values.
139 * @param discreteValues The discrete Values.
140 * @return double The error of this factor.
141 */
142 double error(const Values& continuousValues,
143 const DiscreteValues& assignment) const;
144
145 /**
146 * @brief Compute error of factor given hybrid values.
147 *
148 * @param values The continuous Values and the discrete assignment.
149 * @return double The error of this factor.
150 */
151 double error(const HybridValues& values) const override;
152
153 /**
154 * @brief Get the dimension of the factor (number of rows on linearization).
155 * Returns the dimension of the first component factor.
156 * @return size_t
157 */
158 size_t dim() const;
159
160 /// @}
161 /// @name Testable
162 /// @{
163
164 /// print to stdout
165 void print(const std::string& s = "", const KeyFormatter& keyFormatter =
166 DefaultKeyFormatter) const override;
167
168 /// Check equality
169 bool equals(const HybridFactor& other, double tol = 1e-9) const override;
170
171 /// @}
172 /// @name Standard API
173 /// @{
174
175 /// Getter for NonlinearFactor decision tree
176 const FactorValuePairs& factors() const { return factors_; }
177
178 /// Linearize specific nonlinear factors based on the assignment in
179 /// discreteValues.
180 GaussianFactor::shared_ptr linearize(const Values& continuousValues,
181 const DiscreteValues& assignment) const;
182
183 /// Linearize all the continuous factors to get a HybridGaussianFactor.
184 std::shared_ptr<HybridGaussianFactor> linearize(
185 const Values& continuousValues) const;
186
187 /// Prune this factor based on the discrete probabilities.
188 HybridNonlinearFactor::shared_ptr prune(
189 const DecisionTreeFactor& discreteProbs) const;
190
191 /// Restrict the factor to the given discrete values.
192 std::shared_ptr<Factor> restrict(
193 const DiscreteValues& assignment) const override;
194
195 /// @}
196
197 private:
198 /// Helper struct to assist private constructor below.
199 struct ConstructorHelper;
200
201 // Private constructor using ConstructorHelper above.
202 HybridNonlinearFactor(const ConstructorHelper& helper);
203};
204
205// traits
206template <>
207struct traits<HybridNonlinearFactor> : public Testable<HybridNonlinearFactor> {
208};
