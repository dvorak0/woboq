29void DCSAM::update(const HybridNonlinearFactorGraph &hnfg,
30 const HybridValues &initialGuess) {
31 // First things first: combine currContinuous_ estimate with
32 // the new continuous values from initialGuess to
33 // produce the full continuous variable state.
34 currContinuous_.insert_or_assign(values: initialGuess.nonlinear());
35
36 // Also combine currDiscrete_ estimate with new discrete values from
37 // initialGuess to give a full discrete variable state.
38 for (const auto &kv : initialGuess.discrete()) {
39 // This will update the element with key `kv.first` if one exists, or add a
40 // new element with key `kv.first` if not.
41 currDiscrete_[kv.first] = kv.second;
42 }
43
44 // We individually record the continuous-only and discrete-only factors
45 // so we can separately record the hybrid factors and get
46 // corresponding continuous/discrete factors given
47 // the current best discrete/continuous estimates.
48 NonlinearFactorGraph continuousFactors;
49 DiscreteFactorGraph discreteFactors;
50
51 for (auto &factor : hnfg) {
52 if (auto hybrid_factor = std::dynamic_pointer_cast<HybridFactor>(r: factor)) {
53 if (auto nonlinear_mixture =
54 std::dynamic_pointer_cast<HybridNonlinearFactor>(r: hybrid_factor)) {
55 hfg_.push_back(factor: nonlinear_mixture);
56 }
57
58 } else if (auto cont_factor =
59 std::dynamic_pointer_cast<NonlinearFactor>(r: factor)) {
60 continuousFactors.push_back(factor: cont_factor);
61
62 } else if (auto discrete_factor =
63 std::dynamic_pointer_cast<DiscreteFactor>(r: factor)) {
64 discreteFactors.push_back(factor: discrete_factor);
65 }
66 }
67
68 // Record the discrete factors in `dfg_`.
69 updateDiscrete(dfg: discreteFactors, discreteVals: initialGuess.discrete());
70
71 // Only the initialGuess needs to be provided for the continuous solver
72 // (not the entire continuous state).
73 updateContinuous(newFactors: continuousFactors, initialGuess: initialGuess.nonlinear());
74
75 currContinuous_ = isam_.calculateEstimate();
76
77 // Discrete info will be updated when we call update/solveDiscrete again
78}
79
80void DCSAM::update() { update(hnfg: HybridNonlinearFactorGraph()); }
81
82void DCSAM::updateDiscrete(const DiscreteFactorGraph &dfg,
83 const DiscreteValues &discreteVals) {
84 for (auto &factor : dfg) {
85 dfg_.push_back(factor);
86 }
87
88 // Update current discrete state estimate.
89 DiscreteValues mpe = solveDiscrete();
90 // Update the current discrete estimate with the latest solution.
91 for (auto &&kv : mpe) {
92 currDiscrete_[kv.first] = kv.second;
93 }
94}

96void DCSAM::updateContinuous(const NonlinearFactorGraph &newFactors,
97 const Values &initialGuess) {
98 // Initialize continuous factors
99 NonlinearFactorGraph graph(newFactors);
100
101 // Given the best discrete values estimate,
102 // we get the corresponding continuous factors.
103 for (auto &factor : hfg_) {
104 if (auto nonlinear_mixture =
105 std::dynamic_pointer_cast<HybridNonlinearFactor>(r: factor)) {
106 auto sharedContinuous = nonlinear_mixture->factors()(currDiscrete_).first;
107 graph.push_back(factor: sharedContinuous);
108 // Record the continuous factor
109 nfg_.push_back(factor: sharedContinuous);
110 }
111 }
112
113 // Mark all affected variables, which are the continuous variables
114 ISAM2UpdateParams updateParams;
115 FastMap<FactorIndex, KeySet> newAffectedKeys;
116 for (size_t j = 0; j < nfg_.size(); j++) {
117 auto continuousFactor = nfg_[j];
118 for (const Key &k : continuousFactor->keys()) {
119 newAffectedKeys[j].insert(x: k);
120 }
121 }
122 updateParams.newAffectedKeys = std::move(newAffectedKeys);
123
124 // Solve for continuous values
125 isam_.update(newFactors: graph, newTheta: initialGuess, updateParams);
126}

128DiscreteValues DCSAM::solveDiscrete() const {
129 // First copy the recorded discrete factors
130 DiscreteFactorGraph dfg(dfg_);
131
132 // Next we add the hybrid factors as DiscreteBoundary factors
133 // This step ensures we use the latest currContinuous_ values
134 // to compute the discrete factors
135 for (auto &factor : hfg_) {
136 auto nonlinear_mixture =
137 std::dynamic_pointer_cast<HybridNonlinearFactor>(r: factor);
138 if (nonlinear_mixture) {
139 auto discreteFactor = std::make_shared<DiscreteBoundaryFactor>(
140 args: nonlinear_mixture->discreteKeys(), args: nonlinear_mixture->factors(),
141 args: currContinuous_);
142 dfg.push_back(factor: discreteFactor);
143 }
144 }
145
146 // Finally we optimize for the MPE
147 DiscreteValues discreteVals = dfg.optimize();
148 return discreteVals;
149}
150
151HybridValues DCSAM::calculateEstimate() const {
152 // NOTE: if we have these cached from solves, we could presumably just return
153 // the cached values.
154 Values continuousVals = isam_.calculateEstimate();
155 DiscreteValues discreteVals = solveDiscrete();
156 HybridValues values(VectorValues(), discreteVals, continuousVals);
157 return values;
158}
