31/* *******************************************************************************/
32static Y add(const Y& y1, const Y& y2) {
33 GaussianFactorGraph result = y1.first;
34 result.push_back(container: y2.first);
35 return {result, y1.second + y2.second};
36}

38/* *******************************************************************************/
39HybridGaussianProductFactor operator+(const HybridGaussianProductFactor& a,
40 const HybridGaussianProductFactor& b) {
41 return a.empty() ? b : HybridGaussianProductFactor(a.apply(g: b, op: add));
42}

70/* *******************************************************************************/
71void HybridGaussianProductFactor::print(const std::string& s,
72 const KeyFormatter& formatter) const {
73 KeySet keys;
74 auto printer = [&](const Y& y) {
75 if (keys.empty()) keys = y.first.keys();
76 return "Graph of size " + std::to_string(val: y.first.size()) +
77 ", scalar sum: " + std::to_string(val: y.second);
78 };
79 Base::print(s, labelFormatter: formatter, valueFormatter: printer);
80 if (!keys.empty()) {
81 std::cout << s << " Keys:";
82 for (auto&& key : keys) std::cout << " " << formatter(key);
83 std::cout << "." << std::endl;
84 }
85}

88bool HybridGaussianProductFactor::equals(

90 return Base::equals(other, compare: [tol](const Y& a, const Y& b) {
91 return a.first.equals(fg: b.first, tol) && std::abs(x: a.second - b.second) < tol;
92 });

95/* *******************************************************************************/
96HybridGaussianProductFactor HybridGaussianProductFactor::removeEmpty() const {
97 auto emptyGaussian = [](const Y& y) {
98 bool hasNull =
99 std::any_of(first: y.first.begin(), last: y.first.end(),
100 pred: [](const GaussianFactor::shared_ptr& ptr) { return !ptr; });
101 return hasNull ? Y{GaussianFactorGraph(), 0.0} : y;
102 };
103 return {Base(*this, emptyGaussian)};
104}
