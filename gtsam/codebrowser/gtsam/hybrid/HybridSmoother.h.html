12/**
13 * @file HybridSmoother.h
14 * @brief An incremental smoother for hybrid factor graphs
15 * @author Varun Agrawal
16 * @date October 2022
17 */
18
19#include <gtsam/discrete/DiscreteFactorGraph.h>
20#include <gtsam/hybrid/HybridBayesNet.h>
21#include <gtsam/hybrid/HybridGaussianFactorGraph.h>
22#include <gtsam/hybrid/HybridNonlinearFactorGraph.h>
23
24#include <optional>
25
26namespace gtsam {
27
28class GTSAM_EXPORT HybridSmoother {
29 private:
30 HybridNonlinearFactorGraph allFactors_;
31 Values linearizationPoint_;
32
33 HybridBayesNet hybridBayesNet_;
34 /// The threshold above which we make a decision about a mode.
35 std::optional<double> marginalThreshold_;
36 DiscreteValues fixedValues_;
37
38 public:
39 /**
40 * @brief Constructor
41 *
42 * @param removeDeadModes Flag indicating whether to remove dead modes.
43 * @param marginalThreshold The threshold above which a mode gets assigned a
44 * value and is considered "dead". 0.99 is a good starting value.
45 */
46 HybridSmoother(const std::optional<double> marginalThreshold = {})
47 : marginalThreshold_(marginalThreshold) {}
48
49 /// Return fixed values:
50 const DiscreteValues& fixedValues() const { return fixedValues_; }
51
52 /**
53 * Re-initialize the smoother from a new hybrid Bayes Net.
54 */
55 void reInitialize(HybridBayesNet&& hybridBayesNet);
56
57 /**
58 * Re-initialize the smoother from
59 * a new hybrid Bayes Net (non rvalue version).
60 */
61 void reInitialize(HybridBayesNet& hybridBayesNet);
62
63 /**
64 * Given new factors, perform an incremental update.
65 * The relevant densities in the `hybridBayesNet` will be added to the input
66 * graph (fragment), and then eliminated according to the `ordering`
67 * presented. The remaining factor graph contains hybrid Gaussian factors
68 * that are not connected to the variables in the ordering, or a single
69 * discrete factor on all discrete keys, plus all discrete factors in the
70 * original graph.
71 *
72 * \note If maxNrLeaves is given, we look at the discrete factor resulting
73 * from this elimination, and prune it and the Gaussian components
74 * corresponding to the pruned choices.
75 *
76 * @param graph The new factors, should be linear only
77 * @param maxNrLeaves The maximum number of leaves in the new discrete factor,
78 * if applicable
79 * @param givenOrdering The (optional) ordering for elimination, only
80 * continuous variables are allowed
81 */
82 void update(const HybridNonlinearFactorGraph& graph, const Values& initial,
83 std::optional<size_t> maxNrLeaves = {},
84 const std::optional<Ordering> givenOrdering = {});
85
86 /**
87 * @brief Get an elimination ordering which eliminates continuous
88 * and then discrete.
89 *
90 * Expects `factors` to already have the necessary conditionals
91 * which were connected to the variables in the newly added factors.
92 * Those variables should be in `newFactorKeys`.
93 *
94 * @param factors All the new factors and connected conditionals.
95 * @param newFactorKeys The keys/variables in the newly added factors.
96 * @return Ordering
97 */
98 Ordering getOrdering(const HybridGaussianFactorGraph& factors,
99 const KeySet& newFactorKeys);
100
101 /**
102 * @brief Add conditionals from previous timestep as part of liquefication.
103 *
104 * @param graph The new factor graph for the current time step.
105 * @param hybridBayesNet The hybrid bayes net containing all conditionals so
106 * far.
107 * @param ordering The elimination ordering.
108 * @return std::pair<HybridGaussianFactorGraph, HybridBayesNet>
109 */
110 std::pair<HybridGaussianFactorGraph, HybridBayesNet> addConditionals(
111 const HybridGaussianFactorGraph& graph,
112 const HybridBayesNet& hybridBayesNet) const;
113
114 /**
115 * @brief Get the hybrid Gaussian conditional from
116 * the Bayes Net posterior at `index`.
117 *
118 * @param index Indexing value.
119 * @return HybridGaussianConditional::shared_ptr
120 */
121 HybridGaussianConditional::shared_ptr gaussianMixture(size_t index) const;
122
123 /// Return the Bayes Net posterior.
124 const HybridBayesNet& hybridBayesNet() const;
125
126 /// Optimize the hybrid Bayes Net, taking into accound fixed values.
127 HybridValues optimize() const;
128
129 /**
130 * @brief Relinearize the nonlinear factor graph with
131 * the latest stored linearization point.
132 *
133 * @param givenOrdering An optional elimination ordering.
134 */
135 void relinearize(const std::optional<Ordering> givenOrdering = {});
136
137 /// Return the current linearization point.
138 Values linearizationPoint() const;
139
140 /// Return all the recorded nonlinear factors
141 HybridNonlinearFactorGraph allFactors() const;
142
143 private:
144 /// Helper to compute the ordering if ordering is not given.
145 Ordering maybeComputeOrdering(const HybridGaussianFactorGraph& updatedGraph,
146 const std::optional<Ordering> givenOrdering);
147
148 /**
149 * @brief Remove fixed discrete values for discrete keys
150 * introduced in `newFactors`, and reintroduce discrete factors
151 * with marginalThreshold_ as the probability value.
152 *
153 * @param graph The factor graph with previous conditionals added in.
154 * @param newFactors The new factors added to the smoother,
155 * used to check if a fixed discrete value has been reintroduced.
156 * @return HybridGaussianFactorGraph
157 */
158 HybridGaussianFactorGraph removeFixedValues(
159 const HybridGaussianFactorGraph& graph,
160 const HybridGaussianFactorGraph& newFactors);
161};
162
163} // namespace gtsam
