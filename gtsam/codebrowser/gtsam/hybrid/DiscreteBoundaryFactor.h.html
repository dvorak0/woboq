26/**
27 * @brief Take negative log-values, shift them so that the minimum value is 0,
28 * and then exponentiate to create a TableFactor (not normalized yet!).
29 *
30 * @param discreteKeys The discrete keys for the resulting DecisionTreeFactor
31 * @param errors DecisionTree of (unnormalized) errors.
32 * @return TableFactor::shared_ptr
33 */
34GTSAM_EXPORT DecisionTreeFactor DiscreteFactorFromErrors(
35 const DiscreteKeys& discreteKeys, const AlgebraicDecisionTree<Key>& errors);
36
37/**
38 * A discrete probabilistic factor which computes its values from the
39 * continuous factors connected to it and a set of values.
40 *
41 * Helps in computing the discrete probabilities at the boundary of
42 * discrete and continuous factors.
43 *
44 * @ingroup discrete
45 */
46class GTSAM_EXPORT DiscreteBoundaryFactor : public DecisionTreeFactor {
47 protected:
48 /**
49 * @brief Helper method to compute the discrete values from continuous values.
50 *
51 * @param dkeys The discrete keys for this factor.
52 * @param factors A decision tree of nonlinear factors and associated scalars,
53 * which is used to compute the discrete values.
54 * @param values The continuous values at which to compute the probabilities.
55 * @return DecisionTreeFactor
56 */
57 static DecisionTreeFactor ComputeDiscreteBoundary(
58 const DiscreteKeys& dkeys,
59 const HybridNonlinearFactor::FactorValuePairs& factors,
60 const gtsam::Values& values);
61
62 public:
63 // typedefs needed to play nice with gtsam
64 typedef DiscreteBoundaryFactor This;
65 typedef DecisionTreeFactor Base; ///< Typedef to base class
66 typedef std::shared_ptr<DiscreteBoundaryFactor> shared_ptr;
67
68 /// @name Standard Constructors
69 /// @{
70
71 /** Default constructor for I/O */
72 DiscreteBoundaryFactor() {}
73
74 /** Constructor from DiscreteKeys and AlgebraicDecisionTree */
75 DiscreteBoundaryFactor(const DiscreteKeys& keys, const ADT& potentials)
76 : DecisionTreeFactor(keys, potentials) {}
77
78 /** Constructor from DiscreteKeys and AlgebraicDecisionTree */
79 DiscreteBoundaryFactor(const DiscreteKeys& dkeys,
80 const HybridNonlinearFactor::FactorValuePairs& factors,
81 const gtsam::Values& continuousVals)
82 : DecisionTreeFactor(
83 ComputeDiscreteBoundary(dkeys, factors, values: continuousVals)) {}
84
85 /// @}
86 /// @name Testable
87 /// @{
88
89 /// Print
90 void print(
91 const std::string& s = "DiscreteBoundaryFactor:\n",
92 const KeyFormatter& formatter = DefaultKeyFormatter) const override;
93
94 /// @}
95 /// @name Standard Interface
96 /// @{
97
98 using DecisionTreeFactor::operator*;
99
100 /// multiply with a scalar
101 DiscreteFactor::shared_ptr operator*(double s) const override;
102
103 /// multiply two factors
104 DiscreteBoundaryFactor operator*(const DiscreteBoundaryFactor& f) const;
105
106 using DecisionTreeFactor::operator/;
107
108 /**
109 * @brief Divide by factor f (safely).
110 * Division of a factor \f$f(x, y)\f$ by another factor \f$g(y, z)\f$
111 * results in a function which involves all keys
112 * \f$(\frac{f}{g})(x, y, z) = f(x, y) / g(y, z)\f$
113 *
114 * @param f The DecisinTreeFactor to divide by.
115 * @return DecisionTreeFactor
116 */
117 DiscreteBoundaryFactor operator/(const DiscreteBoundaryFactor& f) const;
118
119 /// @}
120 /// @name Advanced Interface
121 /// @{
122
123 /**
124 * Apply unary operator (*this) "op" f
125 * @param op a unary operator that operates on AlgebraicDecisionTree
126 */
127 DiscreteBoundaryFactor apply(Unary op) const;
128
129 /**
130 * Apply unary operator (*this) "op" f
131 * @param op a unary operator that operates on AlgebraicDecisionTree. Takes
132 * both the assignment and the value.
133 */
134 DiscreteBoundaryFactor apply(UnaryAssignment op) const;
135
136 /**
137 * Apply binary operator (*this) "op" f
138 * @param f the second argument for op
139 * @param op a binary operator that operates on AlgebraicDecisionTree
140 */
141 DiscreteBoundaryFactor apply(const DiscreteBoundaryFactor& f,
142 Binary op) const;
143
144 /// @}
145
146 private:
147#if GTSAM_ENABLE_BOOST_SERIALIZATION
148 /** Serialization function */
149 friend class boost::serialization::access;
150 template <class ARCHIVE>
151 void serialize(ARCHIVE& ar, const unsigned int /*version*/) {
152 ar& BOOST_SERIALIZATION_BASE_OBJECT_NVP(Base);
153 }
154#endif
155};
156
157// traits
158template <>
159struct traits<DiscreteBoundaryFactor>
160 : public Testable<DiscreteBoundaryFactor> {};
161} // namespace gtsam
