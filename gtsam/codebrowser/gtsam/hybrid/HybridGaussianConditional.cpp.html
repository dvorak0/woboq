38GaussianConditional::shared_ptr checkConditional(
39 const GaussianFactor::shared_ptr &factor) {
40 if (auto conditional =
41 std::dynamic_pointer_cast<GaussianConditional>(r: factor)) {
42 return conditional;
43 } else {
44 throw std::logic_error(
45 "A HybridGaussianConditional unexpectedly contained a non-conditional");
46 }
47}

49/* *******************************************************************************/
50/**
51 * @brief Helper struct for constructing HybridGaussianConditional objects
52 *
53 * This struct contains the following fields:
54 * - nrFrontals: Optional size_t for number of frontal variables
55 * - pairs: FactorValuePairs for storing conditionals with their negLogConstant
56 * - minNegLogConstant: minimum negLogConstant, computed here, subtracted in
57 * constructor
58 */
59struct HybridGaussianConditional::Helper {
60 FactorValuePairs pairs;
61 std::optional<size_t> nrFrontals = {};
62 double minNegLogConstant = std::numeric_limits<double>::infinity();
63
64 using GC = GaussianConditional;
65 using P = std::vector<std::pair<Vector, double>>;
66
67 /// Construct from a vector of mean and sigma pairs, plus extra args.
68 template <typename... Args>
69 explicit Helper(const DiscreteKey &mode, const P &p, Args &&...args) {
70 nrFrontals = 1;
71 std::vector<GaussianFactorValuePair> fvs;
72 std::vector<GC::shared_ptr> gcs;
73 fvs.reserve(n: p.size());
74 gcs.reserve(n: p.size());
75 for (auto &&[mean, sigma] : p) {
76 auto gaussianConditional =
77 GC::sharedMeanAndStddev(std::forward<Args>(args)..., mean, sigma);
78 double value = gaussianConditional->negLogConstant();
79 minNegLogConstant = std::min(a: minNegLogConstant, b: value);
80 fvs.emplace_back(gaussianConditional, value);
81 gcs.push_back(gaussianConditional);
82 }
83
84 pairs = FactorValuePairs({mode}, fvs);
85 }
86
87 /// Construct from tree of GaussianConditionals.
88 explicit Helper(const Conditionals &conditionals) {
89 auto func = [this](const GC::shared_ptr &gc) -> GaussianFactorValuePair {
90 if (!gc) return {nullptr, std::numeric_limits<double>::infinity()};
91 if (!nrFrontals) nrFrontals = gc->nrFrontals();
92 double value = gc->negLogConstant();
93 minNegLogConstant = std::min(a: minNegLogConstant, b: value);
94 return {gc, value};
95 };
96 pairs = FactorValuePairs(conditionals, func);
97 if (!nrFrontals.has_value()) {
98 throw std::runtime_error(
99 "HybridGaussianConditional: need at least one frontal variable. "
100 "Provided conditionals do not contain any frontal variables.");
101 }
102 }
103
104 /// Construct from tree of factor/scalar pairs.
105 explicit Helper(const FactorValuePairs &pairs) : pairs(pairs) {
106 auto func = [this](const GaussianFactorValuePair &pair) {
107 if (!pair.first) return;
108 auto gc = checkConditional(factor: pair.first);
109 if (!nrFrontals) nrFrontals = gc->nrFrontals();
110 minNegLogConstant = std::min(a: minNegLogConstant, b: pair.second);
111 };
112 pairs.visit(f: func);
113 if (!nrFrontals.has_value()) {
114 throw std::runtime_error(
115 "HybridGaussianConditional: need at least one frontal variable. "
116 "Provided conditionals do not contain any frontal variables.");
117 }
118 }
119};

121/* *******************************************************************************/
122HybridGaussianConditional::HybridGaussianConditional(
123 const DiscreteKeys &discreteParents, Helper &&helper, bool pruned)
124 : BaseFactor(discreteParents,
125 FactorValuePairs(
126 [&](const GaussianFactorValuePair
127 &pair) { // subtract minNegLogConstant
128 return GaussianFactorValuePair{
129 pair.first, pair.second - helper.minNegLogConstant};
130 },
131 std::move(helper.pairs))),
132 BaseConditional(*helper.nrFrontals),
133 negLogConstant_(helper.minNegLogConstant),
134 pruned_(pruned) {}

182/* *******************************************************************************/
183size_t HybridGaussianConditional::nrComponents() const {
184 size_t total = 0;
185 factors().visit(f: [&total](auto &&node) {
186 if (node.first) total += 1;
187 });
188 return total;

191/* *******************************************************************************/
192GaussianConditional::shared_ptr HybridGaussianConditional::choose(
193 const DiscreteValues &discreteValues) const {
194 try {
195 auto &[factor, _] = factors()(discreteValues);
196 if (!factor) return nullptr;
197
198 auto conditional = checkConditional(factor);
199 return conditional;
200 } catch (const std::out_of_range &e) {
201 GTSAM_PRINT(*this);
202 GTSAM_PRINT(discreteValues);
203 throw std::runtime_error(
204 "HybridGaussianConditional::choose: discreteValues does not contain "
205 "all discrete parents.");
206 }
207}

209/* *******************************************************************************/
210bool HybridGaussianConditional::equals(const HybridFactor &lf,
211 double tol) const {
212 const This *e = dynamic_cast<const This *>(&lf);
213 if (e == nullptr) return false;
214
215 // Factors existence and scalar values are checked in BaseFactor::equals.
216 // Here we check additionally that the factors *are* conditionals
217 // and are equal.
218 auto compareFunc = [tol](const GaussianFactorValuePair &pair1,
219 const GaussianFactorValuePair &pair2) {
220 auto c1 = std::dynamic_pointer_cast<GaussianConditional>(r: pair1.first),
221 c2 = std::dynamic_pointer_cast<GaussianConditional>(r: pair2.first);
222 return (!c1 && !c2) || (c1 && c2 && c1->equals(cg: *c2, tol));
223 };
224 return Base::equals(lf: *e, tol) && factors().equals(other: e->factors(), compare: compareFunc);
225}

227/* *******************************************************************************/
228void HybridGaussianConditional::print(const std::string &s,
229 const KeyFormatter &formatter) const {
230 std::cout << (s.empty() ? "" : s + "\n");
231 BaseConditional::print(s: "", formatter);
232 std::cout << " Discrete Keys = ";
233 for (auto &dk : discreteKeys()) {
234 std::cout << "(" << formatter(dk.first) << ", " << dk.second << "), ";
235 }
236 std::cout << std::endl
237 << " logNormalizationConstant: " << std::fixed
238 << std::setprecision(4) << -negLogConstant() << std::endl
239 << std::endl;
240 factors().print(
241 s: "", labelFormatter: [&](Key k) { return formatter(k); },
242 valueFormatter: [&](const GaussianFactorValuePair &pair) -> std::string {
243 RedirectCout rd;
244 if (auto gf =
245 std::dynamic_pointer_cast<GaussianConditional>(r: pair.first)) {
246 gf->print("", formatter);
247 return rd.str();
248 } else {
249 return "nullptr";
250 }
251 });
252}

254/* ************************************************************************* */
255KeyVector HybridGaussianConditional::continuousParents() const {
256 // Get all parent keys:
257 const auto range = parents();
258 KeyVector continuousParentKeys(range.begin(), range.end());
259 // Loop over all discrete keys:
260 for (const auto &discreteKey : discreteKeys()) {
261 const Key key = discreteKey.first;
262 // remove that key from continuousParentKeys:
263 continuousParentKeys.erase(first: std::remove(first: continuousParentKeys.begin(),
264 last: continuousParentKeys.end(), value: key),
265 last: continuousParentKeys.end());
266 }
267 return continuousParentKeys;
268}

270/* ************************************************************************* */
271bool HybridGaussianConditional::allFrontalsGiven(
272 const VectorValues &given) const {
273 for (auto &&kv : given) {
274 if (given.find(j: kv.first) == given.end()) {
275 return false;
276 }
277 }
278 return true;
279}

281/* ************************************************************************* */
282std::shared_ptr<HybridGaussianFactor> HybridGaussianConditional::likelihood(
283 const VectorValues &given) const {
284 if (!allFrontalsGiven(given)) {
285 throw std::runtime_error(
286 "HybridGaussianConditional::likelihood: given values are missing some "
287 "frontals.");
288 }
289
290 const DiscreteKeys discreteParentKeys = discreteKeys();
291 const KeyVector continuousParentKeys = continuousParents();
292 const HybridGaussianFactor::FactorValuePairs likelihoods(
293 factors(),
294 [&](const GaussianFactorValuePair &pair) -> GaussianFactorValuePair {
295 if (auto conditional =
296 std::dynamic_pointer_cast<GaussianConditional>(r: pair.first)) {
297 const auto likelihood_m = conditional->likelihood(frontalValues: given);
298 // pair.second == conditional->negLogConstant() - negLogConstant_
299 return {likelihood_m, pair.second};
300 } else {
301 return {nullptr, std::numeric_limits<double>::infinity()};
302 }
303 });
304 return std::make_shared<HybridGaussianFactor>(args: discreteParentKeys,
305 args: likelihoods);
306}

314/* *******************************************************************************/
315HybridGaussianConditional::shared_ptr HybridGaussianConditional::prune(
316 const DiscreteConditional &discreteProbs) const {
317 // Find keys in discreteProbs.keys() but not in this->keys():
318 std::set<Key> mine(this->keys().begin(), this->keys().end());
319 std::set<Key> theirs(discreteProbs.keys().begin(),
320 discreteProbs.keys().end());
321 std::vector<Key> diff;
322 std::set_difference(first1: theirs.begin(), last1: theirs.end(), first2: mine.begin(), last2: mine.end(),
323 result: std::back_inserter(x&: diff));
324
325 // Find maximum probability value for every combination of *our* keys.
326 auto max = discreteProbs.max(keys: Ordering(diff));
327
328 // Check the max value for every combination of our keys.
329 // If the max value is 0.0, we can prune the corresponding conditional.
330 bool allPruned = true;
331 auto pruner =
332 [&](const Assignment<Key> &choices,
333 const GaussianFactorValuePair &pair) -> GaussianFactorValuePair {
334 // If this choice is zero probability or Gaussian is null, return infinity
335 if (!pair.first || max->evaluate(values: choices) == 0.0) {
336 return {nullptr, std::numeric_limits<double>::infinity()};
337 } else {
338 allPruned = false;
339 // Add negLogConstant_ back so that the minimum negLogConstant in the
340 // HybridGaussianConditional is set correctly.
341 return {pair.first, pair.second + negLogConstant_};
342 }
343 };
344
345 FactorValuePairs prunedConditionals = factors().apply(op: pruner);
346 if (allPruned) return nullptr;
347 return std::make_shared<HybridGaussianConditional>(args: discreteKeys(),
348 args&: prunedConditionals, args: true);
349}

351/* *******************************************************************************/
352double HybridGaussianConditional::logProbability(
353 const HybridValues &values) const {
354 auto [factor, _] = factors()(values.discrete());
355 auto conditional = checkConditional(factor);
356 return conditional->logProbability(x: values.continuous());
357}
