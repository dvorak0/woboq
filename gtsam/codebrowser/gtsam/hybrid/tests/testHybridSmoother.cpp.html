47// Test approximate inference with an additional pruning step.
48TEST(HybridSmoother, IncrementalSmoother) {
49 using namespace estimation_fixture;
50
51 size_t K = 5;
52
53 // Switching example of robot moving in 1D
54 // with given measurements and equal mode priors.
55 HybridNonlinearFactorGraph graph;
56 Values initial;
57 Switching switching = InitializeEstimationProblem(
58 K, between_sigma: 1.0, measurement_sigma: 0.1, measurements, transitionProbabilityTable: "1/1 1/1", graph: &graph, initial: &initial);
59
60 HybridSmoother smoother;
61 constexpr size_t maxNrLeaves = 5;
62
63 // Loop over timesteps from 1...K-1
64 for (size_t k = 1; k < K; k++) {
65 if (k > 1) graph.push_back(factor: switching.modeChain.at(i: k - 1)); // Mode chain
66 graph.push_back(factor: switching.binaryFactors.at(i: k - 1)); // Motion Model
67 graph.push_back(factor: switching.unaryFactors.at(i: k)); // Measurement
68
69 initial.insert(j: X(j: k), val: switching.linearizationPoint.at<double>(j: X(j: k)));
70
71 smoother.update(graph, initial, maxNrLeaves);
72
73 // Clear all the factors from the graph
74 graph.resize(size: 0);
75 }
76
77 auto& hybridBayesNet = smoother.hybridBayesNet();
78#ifdef GTSAM_DT_MERGING
79 EXPECT_LONGS_EQUAL(11, hybridBayesNet.at(5)->asDiscrete()->nrValues());
80#else
81 EXPECT_LONGS_EQUAL(16, hybridBayesNet.at(5)->asDiscrete()->nrValues());
82#endif
83
84 // Get the continuous delta update as well as
85 // the optimal discrete assignment.
86 HybridValues delta = hybridBayesNet.optimize();
87
88 // Check discrete assignment
89 DiscreteValues expected_discrete;
90 for (size_t k = 0; k < K - 1; k++) {
91 expected_discrete[M(j: k)] = discrete_seq[k];
92 }
93 EXPECT(assert_equal(expected_discrete, delta.discrete()));
94
95 // Update nonlinear solution and verify
96 Values result = initial.retract(delta: delta.continuous());
97 Values expected_continuous;
98 for (size_t k = 0; k < K; k++) {
99 expected_continuous.insert(j: X(j: k), val: measurements[k]);
100 }
101 EXPECT(assert_equal(expected_continuous, result));
102}
