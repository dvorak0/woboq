79TEST(HybridGaussianFactorGraph, eliminateFullMultifrontalCLG) {
80 HybridGaussianFactorGraph hfg;
81
82 // Prior on x0
83 hfg.add(factorOrContainer: JacobianFactor(X(j: 0), I_3x3, Z_3x1));
84 // Factor between x0-x1
85 hfg.add(factorOrContainer: JacobianFactor(X(j: 0), I_3x3, X(j: 1), -I_3x3, Z_3x1));
86
87 // Hybrid factor P(x1|c1)
88 hfg.add(factorOrContainer: HybridGaussianFactor(m1, two_component_fixture::components(key: X(j: 1))));
89 // Prior factor on c1
90 hfg.add(factorOrContainer: DecisionTreeFactor(m1, {2, 8}));
91
92 // Get a constrained ordering keeping c1 last
93 auto ordering_full = HybridOrdering(graph: hfg);
94
95 // Returns a Hybrid Bayes Tree with distribution P(x0|x1)P(x1|c1)P(c1)
96 HybridBayesTree::shared_ptr hbt = hfg.eliminateMultifrontal(ordering: ordering_full);
97
98 EXPECT_LONGS_EQUAL(3, hbt->size());
99}

102// Check assembling the Bayes Tree roots after we do partial elimination
103TEST(HybridGaussianFactorGraph, eliminateFullMultifrontalTwoClique) {
104 HybridGaussianFactorGraph hfg;
105
106 hfg.add(factorOrContainer: JacobianFactor(X(j: 0), I_3x3, X(j: 1), -I_3x3, Z_3x1));
107 hfg.add(factorOrContainer: JacobianFactor(X(j: 1), I_3x3, X(j: 2), -I_3x3, Z_3x1));
108
109 hfg.add(factorOrContainer: HybridGaussianFactor(m0, two_component_fixture::components(key: X(j: 0))));
110 hfg.add(factorOrContainer: HybridGaussianFactor(m1, two_component_fixture::components(key: X(j: 2))));
111
112 hfg.add(factorOrContainer: DecisionTreeFactor({m1, m2}, "1 2 3 4"));
113
114 hfg.add(factorOrContainer: JacobianFactor(X(j: 3), I_3x3, X(j: 4), -I_3x3, Z_3x1));
115 hfg.add(factorOrContainer: JacobianFactor(X(j: 4), I_3x3, X(j: 5), -I_3x3, Z_3x1));
116
117 hfg.add(factorOrContainer: HybridGaussianFactor(m3, two_component_fixture::components(key: X(j: 3))));
118 hfg.add(factorOrContainer: HybridGaussianFactor(m2, two_component_fixture::components(key: X(j: 5))));
119
120 auto ordering_full =
121 Ordering::ColamdConstrainedLast(graph: hfg, constrainLast: {M(j: 0), M(j: 1), M(j: 2), M(j: 3)});
122
123 const auto [hbt, remaining] = hfg.eliminatePartialMultifrontal(ordering: ordering_full);
124
125 // 9 cliques in the bayes tree and 0 remaining variables to eliminate.
126 EXPECT_LONGS_EQUAL(9, hbt->size());
127 EXPECT_LONGS_EQUAL(0, remaining->size());
128
129 /*
130 (Fan) Explanation: the Junction tree will need to re-eliminate to get to the
131 marginal on X(1), which is not possible because it involves eliminating
132 discrete before continuous. The solution to this, however, is in Murphy02.
133 TLDR is that this is 1. expensive and 2. inexact. nevertheless it is doable.
134 And I believe that we should do this.
135 */
136}

155TEST(HybridGaussianFactorGraph, Switching) {
156 auto N = 12;
157 auto hfg = makeSwitchingChain(K: N);
158
159 // X(5) will be the center, X(1-4), X(6-9)
160 // X(3), X(7)
161 // X(2), X(8)
162 // X(1), X(4), X(6), X(9)
163 // M(5) will be the center, M(1-4), M(6-8)
164 // M(3), M(7)
165 // M(1), M(4), M(2), M(6), M(8)
166 // auto ordering_full =
167 // Ordering(KeyVector{X(1), X(4), X(2), X(6), X(9), X(8), X(3), X(7),
168 // X(5),
169 // M(1), M(4), M(2), M(6), M(8), M(3), M(7), M(5)});
170 Ordering ordering;
171
172 {
173 std::vector<int> naturalX(N);
174 std::iota(first: naturalX.begin(), last: naturalX.end(), value: 1);
175 std::vector<Key> ordX;
176 std::transform(first: naturalX.begin(), last: naturalX.end(), result: std::back_inserter(x&: ordX),
177 unary_op: [](int x) { return X(j: x); });
178
179 auto [ndX, lvls] = makeBinaryOrdering(input&: ordX);
180 std::copy(first: ndX.begin(), last: ndX.end(), result: std::back_inserter(x&: ordering));
181 }
182 {
183 std::vector<int> naturalC(N - 1);
184 std::iota(first: naturalC.begin(), last: naturalC.end(), value: 1);
185 std::vector<Key> ordC;
186 std::transform(first: naturalC.begin(), last: naturalC.end(), result: std::back_inserter(x&: ordC),
187 unary_op: [](int x) { return M(j: x); });
188
189 const auto [ndC, lvls] = makeBinaryOrdering(input&: ordC);
190 std::copy(first: ndC.begin(), last: ndC.end(), result: std::back_inserter(x&: ordering));
191 }
192
193 const auto [hbt, remaining] = hfg->eliminatePartialMultifrontal(ordering);
194
195 // 12 cliques in the bayes tree and 0 remaining variables to eliminate.
196 EXPECT_LONGS_EQUAL(12, hbt->size());
197 EXPECT_LONGS_EQUAL(0, remaining->size());
198}

345namespace optimize_fixture {
346HybridGaussianFactorGraph GetGaussianFactorGraph(size_t N) {
347 Switching s(N);
348 HybridGaussianFactorGraph graph;
349
350 for (size_t i = 0; i < N - 1; i++) {
351 graph.push_back(factor: s.linearBinaryFactors.at(i));
352 }
353 for (size_t i = 0; i < N; i++) {
354 graph.push_back(factor: s.linearUnaryFactors.at(i));
355 }
356 for (size_t i = 0; i < N - 1; i++) {
357 graph.push_back(factor: s.modeChain.at(i));
358 }
359
360 return graph;
361}

365// Test for optimizing a HybridBayesTree with a given assignment.
366TEST(HybridBayesTree, OptimizeAssignment) {
367 using namespace optimize_fixture;
368
369 size_t N = 4;
370
371 HybridGaussianISAM isam;
372
373 // Add the 3 hybrid factors, x0-x1, x1-x2, x2-x3
374 // Then add the Gaussian factors, 1 prior on X(0),
375 // 3 measurements on X(1), X(2), X(3)
376 // Finally add the discrete factors
377 // m0, m1-m0, m2-m1
378 HybridGaussianFactorGraph graph1 = GetGaussianFactorGraph(N);
379
380 isam.update(newFactors: graph1);
381
382 DiscreteValues assignment;
383 assignment[M(j: 0)] = 1;
384 assignment[M(j: 1)] = 1;
385 assignment[M(j: 2)] = 1;
386
387 VectorValues delta = isam.optimize(assignment);
388
389 // The linearization point has the same value as the key index,
390 // e.g. X(1) = 1, X(2) = 2,
391 // but the factors specify X(k) = k-1, so delta should be -1.
392 VectorValues expected_delta;
393 expected_delta.insert(key_value: make_pair(x: X(j: 0), y: -Vector1::Ones()));
394 expected_delta.insert(key_value: make_pair(x: X(j: 1), y: -Vector1::Ones()));
395 expected_delta.insert(key_value: make_pair(x: X(j: 2), y: -Vector1::Ones()));
396 expected_delta.insert(key_value: make_pair(x: X(j: 3), y: -Vector1::Ones()));
397
398 EXPECT(assert_equal(expected_delta, delta));
399
400 Switching s(N);
401 // Create ordering.
402 Ordering ordering;
403 for (size_t k = 0; k < s.K; k++) ordering.push_back(x: X(j: k));
404
405 const auto [hybridBayesNet, remainingFactorGraph] =
406 s.linearizedFactorGraph().eliminatePartialSequential(ordering);
407
408 GaussianBayesNet gbn = hybridBayesNet->choose(assignment);
409 VectorValues expected = gbn.optimize();
410
411 EXPECT(assert_equal(expected, delta));
412}

415// Test for optimizing a HybridBayesTree.
416TEST(HybridBayesTree, Optimize) {
417 using namespace optimize_fixture;
418
419 size_t N = 4;
420
421 HybridGaussianISAM isam;
422 // Add the 3 hybrid factors, x0-x1, x1-x2, x2-x3
423 // Then add the Gaussian factors, 1 prior on X(0),
424 // 3 measurements on X(1), X(2), X(3)
425 // Finally add the discrete factors
426 // m0, m1-m0, m2-m1
427 HybridGaussianFactorGraph graph1 = GetGaussianFactorGraph(N);
428
429 isam.update(newFactors: graph1);
430
431 HybridValues delta = isam.optimize();
432
433 Switching s(N);
434 // Create ordering.
435 Ordering ordering;
436 for (size_t k = 0; k < s.K; k++) ordering.push_back(x: X(j: k));
437
438 const auto [hybridBayesNet, remainingFactorGraph] =
439 s.linearizedFactorGraph().eliminatePartialSequential(ordering);
440
441 DiscreteFactorGraph dfg = remainingFactorGraph->discreteFactors();
442
443 // Add the probabilities for each branch
444 DiscreteKeys discrete_keys = {{M(j: 0), 2}, {M(j: 1), 2}, {M(j: 2), 2}};
445 vector<double> probs = {0.012519475, 0.041280228, 0.075018647, 0.081663656,
446 0.037152205, 0.12248971, 0.07349729, 0.08};
447 dfg.emplace_shared<DecisionTreeFactor>(args&: discrete_keys, args&: probs);
448
449 DiscreteValues expectedMPE = dfg.optimize();
450 VectorValues expectedValues = hybridBayesNet->optimize(assignment: expectedMPE);
451
452 EXPECT(assert_equal(expectedMPE, delta.discrete()));
453 EXPECT(assert_equal(expectedValues, delta.continuous()));
454}
