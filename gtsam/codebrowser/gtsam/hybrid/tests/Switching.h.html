45/**
46 * @brief Create a switching system chain. A switching system is a continuous
47 * system which depends on a discrete mode at each time step of the chain.
48 *
49 * @param K The number of chain elements.
50 * @param x The functional to help specify the continuous key.
51 * @param m The functional to help specify the discrete key.
52 * @return HybridGaussianFactorGraph::shared_ptr
53 */
54inline HybridGaussianFactorGraph::shared_ptr makeSwitchingChain(
55 size_t K, std::function<Key(int)> x = X, std::function<Key(int)> m = M,
56 const std::string &transitionProbabilityTable = "0 1 1 3") {
57 HybridGaussianFactorGraph hfg;
58
59 hfg.add(factorOrContainer: JacobianFactor(x(1), I_3x3, Z_3x1));
60
61 // x(1) to x(n+1)
62 for (size_t k = 1; k < K; k++) {
63 DiscreteKeys dKeys{{m(k), 2}};
64 std::vector<GaussianFactor::shared_ptr> components;
65 components.emplace_back(
66 args: new JacobianFactor(x(k), I_3x3, x(k + 1), I_3x3, Z_3x1));
67 components.emplace_back(
68 args: new JacobianFactor(x(k), I_3x3, x(k + 1), I_3x3, Vector3::Ones()));
69 hfg.add(factorOrContainer: HybridGaussianFactor({m(k), 2}, components));
70
71 if (k > 1) {
72 hfg.add(factorOrContainer: DecisionTreeFactor({{m(k - 1), 2}, {m(k), 2}},
73 transitionProbabilityTable));
74 }
75 }
76
77 return std::make_shared<HybridGaussianFactorGraph>(args: std::move(hfg));
78}

80/**
81 * @brief Return the ordering as a binary tree such that all parent nodes are
82 * above their children.
83 *
84 * This will result in a nested dissection Bayes tree after elimination.
85 *
86 * @param input The original ordering.
87 * @return std::pair<KeyVector, std::vector<int>>
88 */
89inline std::pair<KeyVector, std::vector<int>> makeBinaryOrdering(
90 std::vector<Key> &input) {
91 KeyVector new_order;
92
93 std::vector<int> levels(input.size());
94 std::function<void(std::vector<Key>::iterator, std::vector<Key>::iterator,
95 int)>
96 bsg = [&bsg, &new_order, &levels, &input](
97 std::vector<Key>::iterator begin,
98 std::vector<Key>::iterator end, int lvl) {
99 if (std::distance(first: begin, last: end) > 1) {
100 std::vector<Key>::iterator pivot =
101 begin + std::distance(first: begin, last: end) / 2;
102
103 new_order.push_back(x: *pivot);
104 levels[std::distance(first: input.begin(), last: pivot)] = lvl;
105 bsg(begin, pivot, lvl + 1);
106 bsg(pivot + 1, end, lvl + 1);
107 } else if (std::distance(first: begin, last: end) == 1) {
108 new_order.push_back(x: *begin);
109 levels[std::distance(first: input.begin(), last: begin)] = lvl;
110 }
111 };
112
113 bsg(input.begin(), input.end(), 0);
114 std::reverse(first: new_order.begin(), last: new_order.end());
115
116 return {new_order, levels};
117}

120using MotionModel = BetweenFactor<double>;
121
122// Test fixture with switching network.
123/// ϕ(X(0)) .. ϕ(X(k),X(k+1)) .. ϕ(X(k);z_k) .. ϕ(M(0)) .. ϕ(M(K-3),M(K-2))
124struct Switching {
125 public:
126 size_t K;
127 DiscreteKeys modes;
128 HybridNonlinearFactorGraph unaryFactors, binaryFactors, modeChain;
129 HybridGaussianFactorGraph linearUnaryFactors, linearBinaryFactors;
130 Values linearizationPoint;
131
132 /**
133 * @brief Create with given number of time steps.
134 *
135 * @param K The total number of timesteps.
136 * @param between_sigma The stddev between poses.
137 * @param prior_sigma The stddev on priors (also used for measurements).
138 * @param measurements Vector of measurements for each timestep.
139 */
140 Switching(size_t K, double between_sigma = 1.0, double prior_sigma = 0.1,
141 std::vector<double> measurements = {},
142 std::string transitionProbabilityTable = "1/2 3/2")
143 : K(K) {
144 using noiseModel::Isotropic;
145
146 // Create DiscreteKeys for K-1 binary modes.
147 for (size_t k = 0; k < K - 1; k++) {
148 modes.emplace_back(args: M(j: k), args: 2);
149 }
150
151 // If measurements are not provided, we just have the robot moving forward.
152 if (measurements.size() == 0) {
153 for (size_t k = 0; k < K; k++) {
154 measurements.push_back(x: k);
155 }
156 }
157
158 // Create hybrid factor graph.
159
160 // Add a prior ϕ(X(0)) on X(0).
161 unaryFactors.emplace_shared<PriorFactor<double>>(
162 args: X(j: 0), args&: measurements.at(n: 0), args: Isotropic::Sigma(dim: 1, sigma: prior_sigma));
163
164 // Add "motion models" ϕ(X(k),X(k+1),M(k)).
165 for (size_t k = 0; k < K - 1; k++) {
166 auto motion_models = motionModels(k, sigma: between_sigma);
167 binaryFactors.emplace_shared<HybridNonlinearFactor>(args&: modes[k],
168 args&: motion_models);
169 }
170
171 // Add measurement factors ϕ(X(k);z_k).
172 auto measurement_noise = Isotropic::Sigma(dim: 1, sigma: prior_sigma);
173 for (size_t k = 1; k < K; k++) {
174 unaryFactors.emplace_shared<PriorFactor<double>>(args: X(j: k), args&: measurements.at(n: k),
175 args&: measurement_noise);
176 }
177
178 // Add "mode chain" ϕ(M(0)) ϕ(M(0),M(1)) ... ϕ(M(K-3),M(K-2))
179 modeChain = createModeChain(transitionProbabilityTable);
180
181 // Create the linearization point.
182 for (size_t k = 0; k < K; k++) {
183 linearizationPoint.insert<double>(j: X(j: k), val: static_cast<double>(k + 1));
184 }
185
186 linearUnaryFactors = *unaryFactors.linearize(continuousValues: linearizationPoint);
187 linearBinaryFactors = *binaryFactors.linearize(continuousValues: linearizationPoint);
188 }
189
190 // Create motion models for a given time step
191 static std::vector<NoiseModelFactor::shared_ptr> motionModels(
192 size_t k, double sigma = 1.0) {
193 auto noise_model = noiseModel::Isotropic::Sigma(dim: 1, sigma);
194 auto still =
195 std::make_shared<MotionModel>(args: X(j: k), args: X(j: k + 1), args: 0.0, args&: noise_model),
196 moving =
197 std::make_shared<MotionModel>(args: X(j: k), args: X(j: k + 1), args: 1.0, args&: noise_model);
198 return {still, moving};
199 }
200
201 /**
202 * @brief Add "mode chain" to HybridNonlinearFactorGraph from M(0) to M(K-2).
203 * E.g. if K=4, we want M0, M1 and M2.
204 *
205 * @param fg The factor graph to which the mode chain is added.
206 */
207 HybridNonlinearFactorGraph createModeChain(
208 std::string transitionProbabilityTable = "1/2 3/2") {
209 HybridNonlinearFactorGraph chain;
210 chain.emplace_shared<DiscreteDistribution>(args&: modes[0], args: "1/1");
211 for (size_t k = 0; k < K - 2; k++) {
212 auto parents = {modes[k]};
213 chain.emplace_shared<DiscreteConditional>(args&: modes[k + 1], args&: parents,
214 args&: transitionProbabilityTable);
215 }
216 return chain;
217 }
218
219 /// Get the full linear factor graph.
220 HybridGaussianFactorGraph linearizedFactorGraph() const {
221 HybridGaussianFactorGraph graph;
222 graph.push_back(container: linearUnaryFactors);
223 graph.push_back(container: linearBinaryFactors);
224 graph.push_back(container: modeChain);
225 return graph;
226 }
227
228 /// Get all the nonlinear factors.
229 HybridNonlinearFactorGraph nonlinearFactorGraph() const {
230 HybridNonlinearFactorGraph graph;
231 graph.push_back(container: unaryFactors);
232 graph.push_back(container: binaryFactors);
233 graph.push_back(container: modeChain);
234 return graph;
235 }
236};
