48/*
49 * Test full DCSAM solve on DCMixtureFactor for 1D case.
50 *
51 * This is essentially identical to the `DCContinuousFactor` test above,
52 but the
53 * solution is obtained by calling the `DCSAM::update` functions (rather
54 than
55 * implemented manually as above).
56 */
57TEST(DCSAM, SimpleMixtureFactor) {
58 using namespace discrete_mixture_fixture;
59
60 // Make a symbol for a single continuous variable and add to KeyVector
61 KeyVector keys;
62 keys.push_back(x: x1);
63
64 std::vector<NonlinearFactorValuePair> factorComponents{
65 {f1(), priorNoise1()->negLogConstant()},
66 {fNullHypo(), priorNoiseNullHypo()->negLogConstant()}};
67
68 HybridNonlinearFactor dcMixture(dk, factorComponents);
69
70 // Make an empty hybrid factor graph
71 HybridNonlinearFactorGraph hfg;
72
73 hfg.push_back(factor: dcMixture);
74
75 // Let's make an initial guess
76 Values initialGuessContinuous;
77 double initVal = -2.5;
78 initialGuessContinuous.insert(j: x1, val: initVal);
79
80 // We also need an initial guess for the discrete variables
81 // (this will only be used if it is needed by your factors),
82 // here it is ignored.
83 DiscreteValues initialGuessDiscrete;
84 initialGuessDiscrete[dk.first] = 0;
85
86 // Let's make a solver
87 DCSAM dcsam;
88
89 HybridValues initialGuess(initialGuessDiscrete, initialGuessContinuous);
90
91 // Add the HybridFactorGraph to DCSAM
92 dcsam.update(graph: hfg, initialGuess);
93
94 // Solve
95 HybridValues dcvals = dcsam.calculateEstimate();
96
97 // Run another iteration
98 dcsam.update();
99
100 // Update HybridValues
101 dcvals = dcsam.calculateEstimate();
102
103 // Ensure that the prediction is correct
104 const size_t mpeD = dcvals.discrete().at(k: dk.first);
105 EXPECT_LONGS_EQUAL(0, mpeD);
106}

121/**
122 * This is a basic octagonal pose graph SLAM test
123 * to verify that DCSAM works on standard SLAM examples.
124 */
125TEST(DCSAM, SimpleSlamBatch) {
126 using namespace slam_fixture;
127
128 // Make a hybrid factor graph
129 HybridNonlinearFactorGraph graph;
130
131 // Values for initial guess
132 Values initialGuessContinuous;
133
134 Key x0 = X(j: 0);
135
136 PriorFactor<Pose2> p0(x0, pose0, prior_noise);
137 graph.push_back(factor: p0);
138
139 initialGuessContinuous.insert(j: x0, val: pose0);
140
141 // Setup dcsam
142 DCSAM dcsam;
143
144 Pose2 odom(pose0);
145 for (size_t i = 0; i < 7; i++) {
146 Key xi = X(j: i);
147 Key xj = X(j: i + 1);
148
149 Pose2 meas = dx * noise;
150
151 BetweenFactor<Pose2> bw(xi, xj, dx * noise, meas_noise);
152 graph.push_back(factor: bw);
153
154 odom = odom * meas;
155 initialGuessContinuous.insert(j: xj, val: odom);
156 }
157
158 BetweenFactor<Pose2> bw(x0, X(j: 7), dx * noise, meas_noise);
159
160 HybridValues initialGuess(DiscreteValues(), initialGuessContinuous);
161 dcsam.update(graph, initialGuess);
162 HybridValues dcvals = dcsam.calculateEstimate();
163
164 // Values from the DCSAM repo
165 Values expectedContinuous;
166 expectedContinuous.insert<Pose2>(
167 j: X(j: 0), val: Pose2(-3.53708e-33, -8.97559e-34, 4.1955e-33));
168 expectedContinuous.insert<Pose2>(j: X(j: 1), val: Pose2(1, 0.0141421, 0.795398));
169 expectedContinuous.insert<Pose2>(j: X(j: 2), val: Pose2(1.6899, 0.738184, 1.5908));
170 expectedContinuous.insert<Pose2>(j: X(j: 3), val: Pose2(1.65576, 1.7377, 2.38619));
171 expectedContinuous.insert<Pose2>(j: X(j: 4), val: Pose2(0.918069, 2.41298, -3.10159));
172 expectedContinuous.insert<Pose2>(j: X(j: 5), val: Pose2(-0.0805657, 2.35886, -2.30619));
173 expectedContinuous.insert<Pose2>(j: X(j: 6), val: Pose2(-0.740961, 1.60781, -1.5108));
174 expectedContinuous.insert<Pose2>(j: X(j: 7), val: Pose2(-0.66688, 0.61046, -0.715398));
175
176 EXPECT(assert_equal(expectedContinuous, dcvals.nonlinear(), 1e-5));
177}

179/**
180 * This is a basic qualitative octagonal pose graph SLAM test
181 * to verify that DCSAM works on standard SLAM examples
182 * in the *incremental* setting.
183 */
184TEST(DCSAM, SimpleSlamIncremental) {
185 using namespace slam_fixture;
186
187 // Make a factor graph
188 HybridNonlinearFactorGraph graph;
189
190 // Values for initial guess
191 Values initialGuessContinuous;
192
193 Key x0 = X(j: 0);
194
195 PriorFactor<Pose2> p0(x0, pose0, prior_noise);
196
197 initialGuessContinuous.insert(j: x0, val: pose0);
198 graph.push_back(factor: p0);
199
200 // Setup dcsam
201 DCSAM dcsam;
202 HybridValues initialGuess(DiscreteValues(), initialGuessContinuous);
203 dcsam.update(graph, initialGuess);
204
205 graph.resize(size: 0);
206 initialGuessContinuous.clear();
207
208 Pose2 odom(pose0);
209
210 for (size_t i = 0; i < 7; i++) {
211 Key xi = X(j: i);
212 Key xj = X(j: i + 1);
213
214 Pose2 meas = dx * noise;
215
216 BetweenFactor<Pose2> bw(xi, xj, dx * noise, meas_noise);
217 graph.push_back(factor: bw);
218
219 odom = odom * meas;
220 initialGuessContinuous.insert(j: xj, val: odom);
221 initialGuess =
222 HybridValues(initialGuess.discrete(), initialGuessContinuous);
223 dcsam.update(graph, initialGuess);
224
225 graph.resize(size: 0);
226 initialGuessContinuous.clear();
227 }
228
229 BetweenFactor<Pose2> bw(x0, X(j: 7), dx * noise, meas_noise);
230
231 dcsam.update(graph, initialGuess: HybridValues(initialGuess.discrete(), Values()));
232 HybridValues dcvals = dcsam.calculateEstimate();
233
234 // Values from the DCSAM repo
235 Values expectedContinuous;

237 j: X(j: 0), val: Pose2(-3.53708e-33, -8.97559e-34, 4.1955e-33));
238 expectedContinuous.insert<Pose2>(j: X(j: 1), val: Pose2(1, 0.0141421, 0.795398));
239 expectedContinuous.insert<Pose2>(j: X(j: 2), val: Pose2(1.6899, 0.738184, 1.5908));
240 expectedContinuous.insert<Pose2>(j: X(j: 3), val: Pose2(1.65576, 1.7377, 2.38619));
241 expectedContinuous.insert<Pose2>(j: X(j: 4), val: Pose2(0.918069, 2.41298, -3.10159));
242 expectedContinuous.insert<Pose2>(j: X(j: 5), val: Pose2(-0.0805657, 2.35886, -2.30619));
243 expectedContinuous.insert<Pose2>(j: X(j: 6), val: Pose2(-0.740961, 1.60781, -1.5108));
244 expectedContinuous.insert<Pose2>(j: X(j: 7), val: Pose2(-0.66688, 0.61046, -0.715398));

249/**
250 * This test is a sanity check that DCSAM works correctly for solely
251 discrete problems
252 *
253 */
254TEST(DCSAM, SimpleDiscrete) {
255 // Create a DCSAM instance
256 DCSAM dcsam;
257
258 // Make an empty hybrid factor graph
259 HybridNonlinearFactorGraph hfg;
260
261 // We'll make a variable with 2 possible assignments
262 const size_t cardinality = 2;
263 DiscreteKey dk(D(j: 1), cardinality);
264 const std::vector<double> probs{0.1, 0.9};
265
266 // Make a discrete prior factor and add it to the graph
267 DecisionTreeFactor dpf(dk, probs);
268 hfg.push_back(factor: dpf);
269
270 // Initial guess for discrete values (only used in certain circumstances)
271 DiscreteValues initialGuessDiscrete;
272 initialGuessDiscrete[dk.first] = 1;
273
274 // Update DCSAM with the new factor
275 dcsam.update(graph: hfg, initialGuessDiscrete);
276
277 // Solve
278 HybridValues dcvals = dcsam.calculateEstimate();
279
280 // Get the most probable estimate
281 const size_t mpeD = dcvals.discrete().at(k: dk.first);
282
283 // Ensure that the prediction is correct
284 EXPECT_LONGS_EQUAL(1, mpeD);
285}

287/**
288 * This is a basic octagonal pose graph SLAM test with two
289 * *semantic* landmarks (aka landmarks with unknown measurement
290 correspondences)
291 * to verify that DCSAM works on standard SLAM examples in
292 * the *incremental* setting
293 */
294TEST(DCSAM, SimpleSemanticSlam) {
295 using namespace slam_fixture;
296
297 // Make a factor graph
298 HybridNonlinearFactorGraph hfg;
299
300 // Values for initial guess
301 Values initialGuessContinuous;
302 // Initial guess for discrete values (only used in certain circumstances)
303 DiscreteValues initialGuessDiscrete;
304
305 Key x0 = X(j: 0);
306 Key l1 = L(j: 1);
307 Key lc1 = C(j: 1);
308 // Create a discrete key for landmark 1 class with cardinality 2.
309 DiscreteKey lm1_class(lc1, 2);
310
311 const double circumradius = (std::sqrt(x: 4 + 2 * std::sqrt(x: 2))) / 2.0;
312 Point2 landmark1(circumradius, circumradius);
313
314 auto prior_lm_noise = noiseModel::Isotropic::Sigma(dim: 2, sigma: prior_sigma);
315
316 // 0.1 rad std on bearing, 10cm on range
317 auto br_noise = noiseModel::Isotropic::Sigma(dim: 2, sigma: 0.1);
318
319 std::vector<double> prior_lm1_class;
320 prior_lm1_class.push_back(x: 0.9);
321 prior_lm1_class.push_back(x: 0.1);
322
323 PriorFactor<Pose2> p0(x0, pose0, prior_noise);
324 // PriorFactor<Point2> pl1(l1, landmark1, prior_lm_noise);
325
326 // add prior on discrete landmark
327 DecisionTreeFactor plc1(lm1_class, prior_lm1_class);
328
329 initialGuessContinuous.insert(j: x0, val: pose0);
330 initialGuessContinuous.insert(j: l1, val: landmark1);
331
332 // Set initial guess for discrete class var (ignored internally; only used for
333 // MaxMixtures and SumMixtures)
334 initialGuessDiscrete[lm1_class.first] = 0;
335
336 hfg.push_back(factor: p0);
337 hfg.push_back(factor: plc1);
338
339 // Setup dcsam
340 DCSAM dcsam;
341 HybridValues initialGuess(initialGuessDiscrete, initialGuessContinuous);
342 dcsam.update(graph: hfg, initialGuess);
343
344 HybridValues dcval_start = dcsam.calculateEstimate();
345
346 hfg.resize(size: 0);
347 initialGuessContinuous.clear();
348 initialGuessDiscrete.clear();
349
350 Pose2 odom(pose0);
351 for (size_t i = 0; i < 7; i++) {
352 Key xi = X(j: i);
353 Key xj = X(j: i + 1);
354
355 Pose2 meas = dx * noise;
356
357 BetweenFactor<Pose2> bw(xi, xj, meas, meas_noise);
358 hfg.push_back(factor: bw);
359
360 // Add bearing range measurement to landmark in center
361 Rot2 bearing1 = Rot2::fromDegrees(theta: 67.5);
362 const double range1 = circumradius;
363 hfg.push_back(
364 factor: BearingRangeFactor<Pose2, Point2>(xi, l1, bearing1, range1, br_noise));
365
366 // Add semantic measurement to landmark in center
367 // For the first couple measurements, pick class=0, later pick class=1
368 std::vector<double> semantic_meas;
369 if (i < 2) {
370 semantic_meas.push_back(x: 0.9);
371 semantic_meas.push_back(x: 0.1);
372 } else {
373 semantic_meas.push_back(x: 0.1);
374 semantic_meas.push_back(x: 0.9);
375 }
376 DecisionTreeFactor dpf(lm1_class, semantic_meas);
377 hfg.push_back(factor: dpf);
378
379 odom = odom * meas;
380 initialGuessContinuous.insert(j: xj, val: odom);
381 initialGuess = HybridValues(DiscreteValues(), initialGuessContinuous);
382 dcsam.update(graph: hfg, initialGuess);
383
384 HybridValues dcvals = dcsam.calculateEstimate();
385
386 hfg.resize(size: 0);
387 initialGuessContinuous.clear();
388 }
389
390 BetweenFactor<Pose2> bw(x0, X(j: 7), dx * noise, meas_noise);
391
392 hfg.push_back(factor: bw);
393 dcsam.update(graph: hfg, initialGuess: HybridValues(initialGuess.discrete(), Values()));
394
395 HybridValues dcvals = dcsam.calculateEstimate();
396
397 // Values copied from the DCSAM repo
398 DiscreteValues dv;
399 dv[lc1] = 1;
400
401 Values nonlinear;
402 nonlinear.insert(j: L(j: 1), val: Point2(0.501542, 1.20546));
403 nonlinear.insert(j: X(j: 0), val: Pose2(0.00324255, 0.00959038, -0.0170418));
404 nonlinear.insert(j: X(j: 1), val: Pose2(1.04653, 0.020688, 0.826187));
405 nonlinear.insert(j: X(j: 2), val: Pose2(1.7428, 0.798357, 1.6467));
406 nonlinear.insert(j: X(j: 3), val: Pose2(1.64173, 1.8505, 2.47626));
407 nonlinear.insert(j: X(j: 4), val: Pose2(0.802545, 2.48895, -2.98168));
408 nonlinear.insert(j: X(j: 5), val: Pose2(-0.240369, 2.30667, -2.15643));
409 nonlinear.insert(j: X(j: 6), val: Pose2(-0.819911, 1.4135, -1.3317));
410 nonlinear.insert(j: X(j: 7), val: Pose2(-0.689416, 0.344402, -0.456848));
411
412 HybridValues expectedValues(VectorValues(), dv, nonlinear);
413 EXPECT(assert_equal(expectedValues, dcvals, 1e-5));
414}
