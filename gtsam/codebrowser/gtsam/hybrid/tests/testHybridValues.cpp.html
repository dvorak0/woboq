33using namespace std;
34using namespace gtsam;
35
36static const HybridValues kExample{ {{99, Vector2(2, 3)}}, {{100, 3}}, Values{{22, genericValue(v: Pose2(1,2,3))}} };
37
38/* ************************************************************************* */
39TEST(HybridValues, Basics) {
40 HybridValues values;
41 values.insert(j: 99, value: Vector2(2, 3));
42 values.insert(j: 100, value: 3);
43 values.insertNonlinear(j: 22, value: Pose2(1, 2, 3));
44 EXPECT(assert_equal(kExample, values));
45 EXPECT(assert_equal(values.at(99), Vector2(2, 3)));
46 EXPECT(assert_equal(values.atDiscrete(100), int(3)));
47
48 HybridValues values2;
49 values2.insert(j: 100, value: 3);
50 values2.insert(j: 99, value: Vector2(2, 3));
51 EXPECT(assert_equal(kExample, values2));
52}
53
54/* ************************************************************************* */
55// Check insert
56TEST(HybridValues, Insert) {
57 HybridValues actual;
58 EXPECT(assert_equal({{}, {{100, 3}}}, //
59 actual.insert(DiscreteValues{{100, 3}})));
60 EXPECT(assert_equal(kExample, //
61 actual.insert(VectorValues{{99, Vector2(2, 3)}})));
62 EXPECT(assert_equal(kExample, //
63 actual.insert(Values{{22, genericValue(Pose2(1, 2, 3))}})));
64 HybridValues actual2;
65 EXPECT(assert_equal(kExample, actual2.insert(kExample)));
66}
67
68/* ************************************************************************* */
69// Check update.
70TEST(HybridValues, Update) {
71 HybridValues actual(kExample);
72 EXPECT(assert_equal({{{99, Vector2(2, 3)}}, {{100, 2}}},
73 actual.update(DiscreteValues{{100, 2}})));
74 EXPECT(assert_equal({{{99, Vector1(4)}}, {{100, 2}}},
75 actual.update(VectorValues{{99, Vector1(4)}})));
76 EXPECT(assert_equal({{{99, Vector1(4)}}, {{100, 2}}, Values{{22, genericValue(Pose2(4, 5, 6))}}},
77 actual.update(Values{{22, genericValue(Pose2(4, 5, 6))}})));
78 HybridValues actual2(kExample);
79 EXPECT(assert_equal(kExample, actual2.update(kExample)));
80}
