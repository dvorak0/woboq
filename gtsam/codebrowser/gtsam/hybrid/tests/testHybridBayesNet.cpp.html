123// Test API for a tiny hybrid Bayes net.
124TEST(HybridBayesNet, Tiny) {
125 auto bayesNet = tiny::createHybridBayesNet(); // P(z|x,mode)P(x)P(mode)
126 EXPECT_LONGS_EQUAL(3, bayesNet.size());
127
128 const VectorValues vv{{Z(j: 0), Vector1(5.0)}, {X(j: 0), Vector1(5.0)}};
129 HybridValues zero{vv, {{M(j: 0), 0}}}, one{vv, {{M(j: 0), 1}}};
130
131 // Check Invariants for components
132 HybridGaussianConditional::shared_ptr hgc = bayesNet.at(i: 0)->asHybrid();
133 GaussianConditional::shared_ptr gc0 = hgc->choose(discreteValues: zero.discrete()),
134 gc1 = hgc->choose(discreteValues: one.discrete());
135 GaussianConditional::shared_ptr px = bayesNet.at(i: 1)->asGaussian();
136 GaussianConditional::CheckInvariants(conditional: *gc0, values: vv);
137 GaussianConditional::CheckInvariants(conditional: *gc1, values: vv);
138 GaussianConditional::CheckInvariants(conditional: *px, values: vv);
139 HybridGaussianConditional::CheckInvariants(conditional: *hgc, values: zero);
140 HybridGaussianConditional::CheckInvariants(conditional: *hgc, values: one);
141
142 // choose
143 GaussianBayesNet expectedChosen;
144 expectedChosen.push_back(factor: gc0);
145 expectedChosen.push_back(factor: px);
146 auto chosen0 = bayesNet.choose(assignment: zero.discrete());
147 auto chosen1 = bayesNet.choose(assignment: one.discrete());
148 EXPECT(assert_equal(expectedChosen, chosen0, 1e-9));
149
150 // logProbability
151 const double logP0 = chosen0.logProbability(x: vv) + log(x: 0.4); // 0.4 is prior
152 const double logP1 = chosen1.logProbability(x: vv) + log(x: 0.6); // 0.6 is prior
153 EXPECT_DOUBLES_EQUAL(logP0, bayesNet.logProbability(zero), 1e-9);
154 EXPECT_DOUBLES_EQUAL(logP1, bayesNet.logProbability(one), 1e-9);
155
156 // evaluate
157 EXPECT_DOUBLES_EQUAL(exp(logP0), bayesNet.evaluate(zero), 1e-9);
158
159 // optimize
160 EXPECT(assert_equal(one, bayesNet.optimize()));
161 EXPECT(assert_equal(chosen0.optimize(), bayesNet.optimize(zero.discrete())));
162
163 // sample. Not deterministic !!! TODO(Frank): figure out why
164 // std::mt19937_64 rng(42);
165 // EXPECT(assert_equal({{M(0), 1}}, bayesNet.sample(&rng).discrete()));
166
167 // prune
168 auto pruned = bayesNet.prune(maxNrLeaves: 1);
169 CHECK(pruned.at(0)->asHybrid());
170 EXPECT_LONGS_EQUAL(1, pruned.at(0)->asHybrid()->nrComponents());
171 EXPECT(!pruned.equals(bayesNet));
172
173 // error
174 const double error0 = chosen0.error(x: vv) + gc0->negLogConstant() -
175 px->negLogConstant() - log(x: 0.4);
176 const double error1 = chosen1.error(x: vv) + gc1->negLogConstant() -
177 px->negLogConstant() - log(x: 0.6);
178 // print errors:
179 EXPECT_DOUBLES_EQUAL(error0, bayesNet.error(zero), 1e-9);
180 EXPECT_DOUBLES_EQUAL(error1, bayesNet.error(one), 1e-9);
181 EXPECT_DOUBLES_EQUAL(error0 + logP0, error1 + logP1, 1e-9);
182
183 // errorTree
184 AlgebraicDecisionTree<Key> expected(M(j: 0), error0, error1);
185 EXPECT(assert_equal(expected, bayesNet.errorTree(vv)));
186
187 // discretePosterior
188 // We have: P(z|x,mode)P(x)P(mode). When we condition on z and x, we get
189 // P(mode|z,x) \propto P(z|x,mode)P(x)P(mode)
190 // Normalizing this yields posterior P(mode|z,x) = {0.8, 0.2}
191 double q0 = std::exp(x: logP0), q1 = std::exp(x: logP1), sum = q0 + q1;
192 AlgebraicDecisionTree<Key> expectedPosterior(M(j: 0), q0 / sum, q1 / sum);
193 EXPECT(assert_equal(expectedPosterior, bayesNet.discretePosterior(vv)));
194
195 // toFactorGraph
196 auto fg = bayesNet.toFactorGraph(measurements: {{Z(j: 0), Vector1(5.0)}});
197 EXPECT_LONGS_EQUAL(3, fg.size());
198
199 // Create the product factor for eliminating x0:
200 HybridGaussianFactorGraph factors_x0;
201 factors_x0.push_back(factor: fg.at(i: 0));
202 factors_x0.push_back(factor: fg.at(i: 1));
203 auto productFactor = factors_x0.collectProductFactor();
204
205 // Check that scalars are 0 and 1.79 (regression)
206 EXPECT_DOUBLES_EQUAL(0.0, productFactor({{M(0), 0}}).second, 1e-9);
207 EXPECT_DOUBLES_EQUAL(1.791759, productFactor({{M(0), 1}}).second, 1e-5);
208
209 // Call eliminate and check scalar:
210 auto result = factors_x0.eliminate(keys: {X(j: 0)});
211 auto df = std::dynamic_pointer_cast<DecisionTreeFactor>(r: result.second);
212
213 // Check that the ratio of probPrime to evaluate is the same for all modes.
214 std::vector<double> ratio(2);
215 ratio[0] = std::exp(x: -fg.error(values: zero)) / bayesNet.evaluate(values: zero);
216 ratio[1] = std::exp(x: -fg.error(values: one)) / bayesNet.evaluate(values: one);
217 EXPECT_DOUBLES_EQUAL(ratio[0], ratio[1], 1e-8);
218
219 // Better and more general test:
220 // Since (M, x) \propto P(M,x|z) the discretePosteriors should agree
221 q0 = std::exp(x: -fg.error(values: zero));
222 q1 = std::exp(x: -fg.error(values: one));
223 sum = q0 + q1;
224 EXPECT(assert_equal(expectedPosterior, {M(0), q0 / sum, q1 / sum}));
225 VectorValues xv{{X(j: 0), Vector1(5.0)}};
226 auto fgPosterior = fg.discretePosterior(continuousValues: xv);
227 EXPECT(assert_equal(expectedPosterior, fgPosterior));
228}

357// Test Bayes net error and log-probability
358TEST(HybridBayesNet, Error) {
359 using namespace hbn_error;
360
361 HybridBayesNet::shared_ptr posterior =
362 s.linearizedFactorGraph().eliminateSequential();
363 EXPECT_LONGS_EQUAL(5, posterior->size());
364
365 // Optimize
366 HybridValues delta = posterior->optimize();
367
368 // Verify discrete posterior at optimal value sums to 1.
369 auto discretePosterior = posterior->discretePosterior(continuousValues: delta.continuous());
370 EXPECT_DOUBLES_EQUAL(1.0, discretePosterior.sum(), 1e-9);
371
372 // Regression test on discrete posterior at optimal value.
373 std::vector<double> leaves = {0.095516068, 0.31800092, 0.27798511, 0.3084979};
374 AlgebraicDecisionTree<Key> expected(s.modes, leaves);
375 EXPECT(assert_equal(expected, discretePosterior, 1e-6));
376
377 // Verify logProbability computation and check specific logProbability value
378 const HybridValues hybridValues{delta.continuous(), discrete_values};
379 double logProbability = 0;
380 logProbability += posterior->at(i: 0)->asHybrid()->logProbability(values: hybridValues);
381 logProbability += posterior->at(i: 1)->asHybrid()->logProbability(values: hybridValues);
382 logProbability += posterior->at(i: 2)->asHybrid()->logProbability(values: hybridValues);
383 logProbability +=
384 posterior->at(i: 3)->asDiscrete()->logProbability(x: hybridValues);
385 logProbability +=
386 posterior->at(i: 4)->asDiscrete()->logProbability(x: hybridValues);
387 EXPECT_DOUBLES_EQUAL(logProbability, posterior->logProbability(hybridValues),
388 1e-9);
389
390 double negLogConstant = posterior->negLogConstant(discrete: discrete_values);
391
392 // The sum of all the mode densities
393 double normalizer =
394 AlgebraicDecisionTree<Key>(posterior->errorTree(continuousValues: delta.continuous()),
395 [](double error) { return exp(x: -error); })
396 .sum();
397
398 // Check agreement with discrete posterior
399 double density = exp(x: logProbability + negLogConstant) / normalizer;
400 EXPECT_DOUBLES_EQUAL(density, discretePosterior(discrete_values), 1e-6);
401}

531 s.modes,

555 std::dynamic_pointer_cast<DecisionTreeFactor::ADT>(
556 r: pruned_discrete_conditionals);
557
558 // The checker functor verifies the values for us.
559 discrete_conditional_tree->apply(op: checker);
560}

564TEST(HybridBayesNet, Sampling) {
565 HybridNonlinearFactorGraph nfg;
566
567 auto noise_model = noiseModel::Diagonal::Sigmas(sigmas: Vector1(1.0));
568 nfg.emplace_shared<PriorFactor<double>>(args: X(j: 0), args: 0.0, args&: noise_model);
569
570 auto zero_motion =
571 std::make_shared<BetweenFactor<double>>(args: X(j: 0), args: X(j: 1), args: 0, args&: noise_model);
572 auto one_motion =
573 std::make_shared<BetweenFactor<double>>(args: X(j: 0), args: X(j: 1), args: 1, args&: noise_model);
574 nfg.emplace_shared<HybridNonlinearFactor>(
575 args: DiscreteKey(M(j: 0), 2),
576 args: std::vector<NoiseModelFactor::shared_ptr>{zero_motion, one_motion});
577
578 DiscreteKey mode(M(j: 0), 2);
579 nfg.emplace_shared<DiscreteDistribution>(args&: mode, args: "1/1");
580
581 Values initial;
582 double z0 = 0.0, z1 = 1.0;
583 initial.insert<double>(j: X(j: 0), val: z0);
584 initial.insert<double>(j: X(j: 1), val: z1);
585
586 // Create the factor graph from the nonlinear factor graph.
587 HybridGaussianFactorGraph::shared_ptr fg = nfg.linearize(continuousValues: initial);
588 // Eliminate into BN
589 HybridBayesNet::shared_ptr bn = fg->eliminateSequential();
590
591 // Set up sampling
592 std::mt19937_64 gen(11);
593
594 // Initialize containers for computing the mean values.
595 vector<double> discrete_samples;
596 VectorValues average_continuous;
597
598 size_t num_samples = 1000;
599 for (size_t i = 0; i < num_samples; i++) {
600 // Sample
601 HybridValues sample = bn->sample(rng: &gen);
602
603 discrete_samples.push_back(x: sample.discrete().at(k: M(j: 0)));
604
605 if (i == 0) {
606 average_continuous.insert(values: sample.continuous());
607 } else {
608 average_continuous += sample.continuous();
609 }
610 }
611
612 EXPECT_LONGS_EQUAL(2, average_continuous.size());
613 EXPECT_LONGS_EQUAL(num_samples, discrete_samples.size());
614
615 // regression for specific RNG seed
616 double discrete_sum =
617 std::accumulate(first: discrete_samples.begin(), last: discrete_samples.end(),
618 init: decltype(discrete_samples)::value_type(0));
619 EXPECT_DOUBLES_EQUAL(0.519, discrete_sum / num_samples, 1e-9);
620
621 VectorValues expected;
622 // regression for specific RNG seed
623#if __APPLE__ || _WIN32
624 expected.insert({X(0), Vector1(0.0252479903896)});
625 expected.insert({X(1), Vector1(-0.513637101911)});
626#elif __linux__
627 expected.insert(key_value: {X(j: 0), Vector1(0.0165089744897)});
628 expected.insert(key_value: {X(j: 1), Vector1(-0.454323399979)});

630
631 EXPECT(assert_equal(expected, average_continuous.scale(1.0 / num_samples)));
632}
