162// Test API for the smallest switching network.
163// None of these are regression tests.
164TEST(HybridBayesNet, Switching) {
165 // Create switching network with two continuous variables and one discrete:
166 // Ï•(x0) Ï•(x0,x1,m0) Ï•(x1;z1) Ï•(m0)
167 const double betweenSigma = 0.3, priorSigma = 0.1;
168 Switching s(2, betweenSigma, priorSigma);
169
170 // Check size of linearized factor graph
171 const HybridGaussianFactorGraph &graph = s.linearizedFactorGraph();
172 EXPECT_LONGS_EQUAL(4, graph.size());
173
174 // Create some continuous and discrete values
175 const VectorValues continuousValues{{X(j: 0), Vector1(0.1)},
176 {X(j: 1), Vector1(1.2)}};
177 const DiscreteValues modeZero{{M(j: 0), 0}}, modeOne{{M(j: 0), 1}};
178
179 // Get the hybrid gaussian factor and check it is as expected
180 auto hgf = std::dynamic_pointer_cast<HybridGaussianFactor>(r: graph.at(i: 2));
181 CHECK(hgf);
182
183 // Get factors and scalars for both modes
184 auto [factor0, scalar0] = (*hgf)(modeZero);
185 auto [factor1, scalar1] = (*hgf)(modeOne);
186 CHECK(factor0);
187 CHECK(factor1);
188
189 // Check scalars against negLogConstant of noise model
190 auto betweenModel = noiseModel::Isotropic::Sigma(dim: 1, sigma: betweenSigma);
191 EXPECT_DOUBLES_EQUAL(betweenModel->negLogConstant(), scalar0, 1e-9);
192 EXPECT_DOUBLES_EQUAL(betweenModel->negLogConstant(), scalar1, 1e-9);
193
194 // Check error for M(0) = 0
195 const HybridValues values0{continuousValues, modeZero};
196 double expectedError0 = 0;
197 for (const auto &factor : graph) expectedError0 += factor->error(c: values0);
198 EXPECT_DOUBLES_EQUAL(expectedError0, graph.error(values0), 1e-5);
199
200 // Check error for M(0) = 1
201 const HybridValues values1{continuousValues, modeOne};
202 double expectedError1 = 0;
203 for (const auto &factor : graph) expectedError1 += factor->error(c: values1);
204 EXPECT_DOUBLES_EQUAL(expectedError1, graph.error(values1), 1e-5);
205
206 // Check errorTree
207 AlgebraicDecisionTree<Key> actualErrors = graph.errorTree(continuousValues);
208 // Create expected error tree
209 const AlgebraicDecisionTree<Key> expectedErrors(M(j: 0), expectedError0,
210 expectedError1);
211
212 // Check that the actual error tree matches the expected one
213 EXPECT(assert_equal(expectedErrors, actualErrors, 1e-5));
214
215 // Check probPrime
216 const double probPrime0 = graph.probPrime(values: values0);
217 EXPECT_DOUBLES_EQUAL(std::exp(-expectedError0), probPrime0, 1e-5);
218
219 const double probPrime1 = graph.probPrime(values: values1);
220 EXPECT_DOUBLES_EQUAL(std::exp(-expectedError1), probPrime1, 1e-5);
221
222 // Check discretePosterior
223 const AlgebraicDecisionTree<Key> graphPosterior =
224 graph.discretePosterior(continuousValues);
225 const double sum = probPrime0 + probPrime1;
226 const AlgebraicDecisionTree<Key> expectedPosterior(M(j: 0), probPrime0 / sum,
227 probPrime1 / sum);
228 EXPECT(assert_equal(expectedPosterior, graphPosterior, 1e-5));
229
230 // Make the clique of factors connected to x0:
231 HybridGaussianFactorGraph factors_x0;
232 factors_x0.push_back(factor: graph.at(i: 0));
233 factors_x0.push_back(factor: hgf);
234
235 // Test collectProductFactor
236 auto productFactor = factors_x0.collectProductFactor();
237
238 // For M(0) = 0
239 auto [gaussianFactor0, actualScalar0] = productFactor(modeZero);
240 EXPECT(gaussianFactor0.size() == 2);
241 EXPECT_DOUBLES_EQUAL((*hgf)(modeZero).second, actualScalar0, 1e-5);
242
243 // For M(0) = 1
244 auto [gaussianFactor1, actualScalar1] = productFactor(modeOne);
245 EXPECT(gaussianFactor1.size() == 2);
246 EXPECT_DOUBLES_EQUAL((*hgf)(modeOne).second, actualScalar1, 1e-5);
247
248 // Test eliminate x0
249 const Ordering ordering{X(j: 0)};
250 auto [conditional, factor] = factors_x0.eliminate(keys: ordering);
251
252 // Check the conditional
253 CHECK(conditional);
254 EXPECT(conditional->isHybrid());
255 auto p_x0_given_x1_m = conditional->asHybrid();
256 CHECK(p_x0_given_x1_m);
257 EXPECT(HybridGaussianConditional::CheckInvariants(*p_x0_given_x1_m, values1));
258 EXPECT_LONGS_EQUAL(1, p_x0_given_x1_m->nrFrontals()); // x0
259 EXPECT_LONGS_EQUAL(2, p_x0_given_x1_m->nrParents()); // x1, m0
260
261 // Check the remaining factor
262 EXPECT(factor);
263 EXPECT(std::dynamic_pointer_cast<HybridGaussianFactor>(factor));
264 auto phi_x1_m = std::dynamic_pointer_cast<HybridGaussianFactor>(r: factor);
265 EXPECT_LONGS_EQUAL(2, phi_x1_m->keys().size()); // x1, m0
266 // Check that the scalars incorporate the negative log constant of the
267 // conditional
268 EXPECT_DOUBLES_EQUAL(scalar0 - (*p_x0_given_x1_m)(modeZero)->negLogConstant(),
269 (*phi_x1_m)(modeZero).second, 1e-9);
270 EXPECT_DOUBLES_EQUAL(scalar1 - (*p_x0_given_x1_m)(modeOne)->negLogConstant(),
271 (*phi_x1_m)(modeOne).second, 1e-9);
272
273 // Check that the conditional and remaining factor are consistent for both
274 // modes
275 for (auto &&mode : {modeZero, modeOne}) {
276 const auto gc = (*p_x0_given_x1_m)(mode);
277 const auto [gf, scalar] = (*phi_x1_m)(mode);
278
279 // The error of the original factors should equal the sum of errors of the
280 // conditional and remaining factor, modulo the normalization constant of
281 // the conditional.
282 double originalError = factors_x0.error(values: {continuousValues, mode});
283 const double actualError = gc->negLogConstant() +
284 gc->error(c: continuousValues) +
285 gf->error(c: continuousValues) + scalar;
286 EXPECT_DOUBLES_EQUAL(originalError, actualError, 1e-9);
287 }
288
289 // Create a clique for x1
290 HybridGaussianFactorGraph factors_x1;
291 factors_x1.push_back(
292 factor); // Use the remaining factor from previous elimination
293 factors_x1.push_back(
294 factor: graph.at(i: 1)); // Add the factor for x1 from the original graph
295
296 // Test collectProductFactor for x1 clique
297 auto productFactor_x1 = factors_x1.collectProductFactor();
298
299 // For M(0) = 0
300 auto [gaussianFactor_x1_0, actualScalar_x1_0] = productFactor_x1(modeZero);
301 EXPECT_LONGS_EQUAL(2, gaussianFactor_x1_0.size());
302 // NOTE(Frank): prior on x1 does not contribute to the scalar
303 EXPECT_DOUBLES_EQUAL((*phi_x1_m)(modeZero).second, actualScalar_x1_0, 1e-5);
304
305 // For M(0) = 1
306 auto [gaussianFactor_x1_1, actualScalar_x1_1] = productFactor_x1(modeOne);
307 EXPECT_LONGS_EQUAL(2, gaussianFactor_x1_1.size());
308 // NOTE(Frank): prior on x1 does not contribute to the scalar
309 EXPECT_DOUBLES_EQUAL((*phi_x1_m)(modeOne).second, actualScalar_x1_1, 1e-5);
310
311 // Test eliminate for x1 clique
312 Ordering ordering_x1{X(j: 1)};
313 auto [conditional_x1, factor_x1] = factors_x1.eliminate(keys: ordering_x1);
314
315 // Check the conditional for x1
316 CHECK(conditional_x1);
317 EXPECT(conditional_x1->isHybrid());
318 auto p_x1_given_m = conditional_x1->asHybrid();
319 CHECK(p_x1_given_m);
320 EXPECT_LONGS_EQUAL(1, p_x1_given_m->nrFrontals()); // x1
321 EXPECT_LONGS_EQUAL(1, p_x1_given_m->nrParents()); // m0
322
323 // Check the remaining factor for x1
324 CHECK(factor_x1);
325 auto phi_x1 = std::dynamic_pointer_cast<TableFactor>(r: factor_x1);
326 CHECK(phi_x1);
327 EXPECT_LONGS_EQUAL(1, phi_x1->keys().size()); // m0
328 // We can't really check the error of the decision tree factor phi_x1, because
329 // the continuous factor whose error(kEmpty) we need is not available.
330
331 // Now test full elimination of the graph:
332 auto hybridBayesNet = graph.eliminateSequential();
333 CHECK(hybridBayesNet);
334
335 // Check that the posterior P(M|X=continuousValues) from the Bayes net is the
336 // same as the same posterior from the graph. This is a sanity check that the
337 // elimination is done correctly.
338 AlgebraicDecisionTree<Key> bnPosterior =
339 hybridBayesNet->discretePosterior(continuousValues);
340 EXPECT(assert_equal(graphPosterior, bnPosterior));
341}

347 // Create switching network with three continuous variables and two discrete:
348 // Ï•(x0) Ï•(x0,x1,m0) Ï•(x1,x2,m1) Ï•(x1;z1) Ï•(x2;z2) Ï•(m0) Ï•(m0,m1)

365 // same as the same posterior from the graph. This is a sanity check that the
366 // elimination is done correctly.
367 const AlgebraicDecisionTree<Key> graphPosterior =
368 graph.discretePosterior(continuousValues: delta.continuous());
369 const AlgebraicDecisionTree<Key> bnPosterior =
370 hybridBayesNet->discretePosterior(continuousValues: delta.continuous());
371 EXPECT(assert_equal(graphPosterior, bnPosterior));
372}

501// Test hybrid gaussian factor graph errorTree during incremental operation
502TEST(HybridGaussianFactorGraph, IncrementalErrorTree) {
503 Switching s(4);
504
505 HybridGaussianFactorGraph graph;
506 graph.push_back(factor: s.linearUnaryFactors.at(i: 0)); // f(X0)
507 graph.push_back(factor: s.linearBinaryFactors.at(i: 0)); // f(X0, X1, M0)
508 graph.push_back(factor: s.linearBinaryFactors.at(i: 1)); // f(X1, X2, M1)

579// Check that the factor graph unnormalized probability is proportional to the
580// Bayes net probability for the given measurements.
581bool ratioTest(const HybridBayesNet &bn, const VectorValues &measurements,
582 const HybridGaussianFactorGraph &fg, size_t num_samples = 100) {
583 auto compute_ratio = [&](HybridValues *sample) -> double {
584 sample->update(values: measurements); // update sample with given measurements:
585 return bn.evaluate(values: *sample) / fg.probPrime(values: *sample);
586 };
587
588 HybridValues sample = bn.sample(rng: &kRng);
589 double expected_ratio = compute_ratio(&sample);
590
591 // Test ratios for a number of independent samples:
592 for (size_t i = 0; i < num_samples; i++) {
593 HybridValues sample = bn.sample(rng: &kRng);
594 if (std::abs(x: expected_ratio - compute_ratio(&sample)) > 1e-6) return false;
595 }
596 return true;
597}

765// Test elimination of a switching network with one mode per measurement.
766TEST(HybridGaussianFactorGraph, EliminateSwitchingNetwork) {
767 // Create a switching network with one mode per measurement.
768 HybridBayesNet bn;
769
770 // NOTE: we add reverse topological so we can sample from the Bayes net.:
771
772 // Add measurements:
773 std::vector<std::pair<Vector, double>> measurementModels{{Z_1x1, 3},
774 {Z_1x1, 0.5}};
775 for (size_t t : {0, 1, 2}) {
776 // Create hybrid Gaussian factor on Z(t) conditioned on X(t) and mode N(t):
777 const auto noise_mode_t = DiscreteKey{N(j: t), 2};
778 bn.emplace_shared<HybridGaussianConditional>(args: noise_mode_t, args: Z(j: t), args: I_1x1,
779 args: X(j: t), args&: measurementModels);
780
781 // Create prior on discrete mode N(t):
782 bn.emplace_shared<DiscreteConditional>(args: noise_mode_t, args: "20/80");
783 }
784
785 // Add motion models. TODO(frank): why are they exactly the same?
786 std::vector<std::pair<Vector, double>> motionModels{{Z_1x1, 0.2},
787 {Z_1x1, 0.2}};
788 for (size_t t : {2, 1}) {
789 // Create hybrid Gaussian factor on X(t) conditioned on X(t-1)
790 // and mode M(t-1):
791 const auto motion_model_t = DiscreteKey{M(j: t), 2};
792 bn.emplace_shared<HybridGaussianConditional>(args: motion_model_t, args: X(j: t), args: I_1x1,
793 args: X(j: t - 1), args&: motionModels);
794
795 // Create prior on motion model M(t):
796 bn.emplace_shared<DiscreteConditional>(args: motion_model_t, args: "40/60");
797 }
798
799 // Create Gaussian prior on continuous X(0) using sharedMeanAndStddev:
800 bn.push_back(conditional: GaussianConditional::sharedMeanAndStddev(args: X(j: 0), args: Z_1x1, args: 0.1));
801
802 // Make sure we an sample from the Bayes net:
803 EXPECT_LONGS_EQUAL(6, bn.sample().continuous().size());

805 // Create measurements consistent with moving right every time:
806 const VectorValues measurements{
807 {Z(j: 0), Vector1(0.0)}, {Z(j: 1), Vector1(1.0)}, {Z(j: 2), Vector1(2.0)}};
808 const HybridGaussianFactorGraph fg = bn.toFactorGraph(measurements);

810 // Factor graph is:
811 // D D

813 // m1 m2

815 // C-x0-HC-x1-HC-x2
