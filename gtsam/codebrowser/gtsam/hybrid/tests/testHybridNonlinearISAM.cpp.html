49// Test if we can perform elimination incrementally.
50TEST(HybridNonlinearISAM, IncrementalElimination) {
51 Switching switching(3);
52 HybridNonlinearISAM isam;
53 HybridNonlinearFactorGraph graph;
54 Values initial;
55
56 // Create initial factor graph
57 // * * *
58 // | | |
59 // X0 -*- X1 -*- X2
60 // \*-M0-*/
61 graph.push_back(factor: switching.unaryFactors.at(i: 0)); // P(X0)
62 graph.push_back(factor: switching.binaryFactors.at(i: 0)); // P(X0, X1 | M0)
63 graph.push_back(factor: switching.binaryFactors.at(i: 1)); // P(X1, X2 | M1)
64 graph.push_back(factor: switching.modeChain.at(i: 0)); // P(M0)
65
66 initial.insert<double>(j: X(j: 0), val: 1);
67 initial.insert<double>(j: X(j: 1), val: 2);
68 initial.insert<double>(j: X(j: 2), val: 3);
69
70 // Run update step
71 isam.update(newFactors: graph, initialValues: initial);
72
73 // Check that after update we have 3 hybrid Bayes net nodes:
74 // P(X0 | X1, M0) and P(X1, X2 | M0, M1), P(M0, M1)
75 HybridGaussianISAM bayesTree = isam.bayesTree();
76 EXPECT_LONGS_EQUAL(3, bayesTree.size());
77 EXPECT(bayesTree[X(0)]->conditional()->frontals() == KeyVector{X(0)});
78 EXPECT(bayesTree[X(0)]->conditional()->parents() == KeyVector({X(1), M(0)}));
79 EXPECT(bayesTree[X(1)]->conditional()->frontals() == KeyVector({X(1), X(2)}));
80 EXPECT(bayesTree[X(1)]->conditional()->parents() == KeyVector({M(0), M(1)}));
81
82 /********************************************************/
83 // New factor graph for incremental update.
84 graph = HybridNonlinearFactorGraph();
85 initial = Values();
86
87 graph.push_back(factor: switching.unaryFactors.at(i: 1)); // P(X1)
88 graph.push_back(factor: switching.unaryFactors.at(i: 2)); // P(X2)
89 graph.push_back(factor: switching.modeChain.at(i: 1)); // P(M0, M1)
90
91 isam.update(newFactors: graph, initialValues: initial);
92
93 bayesTree = isam.bayesTree();
94 // Check that after the second update we have
95 // 1 additional hybrid Bayes net node:
96 // P(X1, X2 | M0, M1)
97 EXPECT_LONGS_EQUAL(3, bayesTree.size());
98 EXPECT(bayesTree[X(2)]->conditional()->frontals() == KeyVector({X(1), X(2)}));
99 EXPECT(bayesTree[X(2)]->conditional()->parents() == KeyVector({M(0), M(1)}));
100}

102/* ****************************************************************************/
103// Test if we can incrementally do the inference
104TEST(HybridNonlinearISAM, IncrementalInference) {
105 Switching switching(3);
106 HybridNonlinearISAM isam;
107 HybridNonlinearFactorGraph graph;
108 Values initial;
109
110 // Create initial factor graph
111 // * * *
112 // | | |
113 // X0 -*- X1 -*- X2
114 // | |
115 // *-M0 - * - M1
116 graph.push_back(factor: switching.unaryFactors.at(i: 0)); // P(X0)
117 graph.push_back(factor: switching.binaryFactors.at(i: 0)); // P(X0, X1 | M0)
118 graph.push_back(factor: switching.unaryFactors.at(i: 1)); // P(X1)
119 graph.push_back(factor: switching.modeChain.at(i: 0)); // P(M0)
120
121 initial.insert<double>(j: X(j: 0), val: 1);
122 initial.insert<double>(j: X(j: 1), val: 2);
123
124 // Run update step
125 isam.update(newFactors: graph, initialValues: initial);
126 HybridGaussianISAM bayesTree = isam.bayesTree();
127
128 auto discreteConditional_m0 = bayesTree[M(j: 0)]->conditional()->asDiscrete();
129 EXPECT(discreteConditional_m0->keys() == KeyVector({M(0)}));
130
131 /********************************************************/
132 // New factor graph for incremental update.
133 graph = HybridNonlinearFactorGraph();
134 initial = Values();
135
136 initial.insert<double>(j: X(j: 2), val: 3);
137
138 graph.push_back(factor: switching.binaryFactors.at(i: 1)); // P(X1, X2 | M1)
139 graph.push_back(factor: switching.unaryFactors.at(i: 2)); // P(X2)
140 graph.push_back(factor: switching.modeChain.at(i: 1)); // P(M0, M1)
141
142 isam.update(newFactors: graph, initialValues: initial);
143 bayesTree = isam.bayesTree();
144
145 /********************************************************/
146 // Run batch elimination so we can compare results.
147 const Ordering ordering{X(j: 0), X(j: 1), X(j: 2)};
148
149 // Now we calculate the actual factors using full elimination
150 const auto [expectedHybridBayesTree, expectedRemainingGraph] =
151 switching.linearizedFactorGraph()
152 .BaseEliminateable::eliminatePartialMultifrontal(ordering);
153
154 // The densities on X(1) should be the same
155 auto x0_conditional = dynamic_pointer_cast<HybridGaussianConditional>(
156 r: bayesTree[X(j: 0)]->conditional()->inner());
157 auto expected_x0_conditional =
158 dynamic_pointer_cast<HybridGaussianConditional>(
159 r: (*expectedHybridBayesTree)[X(j: 0)]->conditional()->inner());
160 EXPECT(assert_equal(*x0_conditional, *expected_x0_conditional));
161
162 // The densities on X(1) should be the same
163 auto x1_conditional = dynamic_pointer_cast<HybridGaussianConditional>(
164 r: bayesTree[X(j: 1)]->conditional()->inner());
165 auto expected_x1_conditional =
166 dynamic_pointer_cast<HybridGaussianConditional>(
167 r: (*expectedHybridBayesTree)[X(j: 1)]->conditional()->inner());
168 EXPECT(assert_equal(*x1_conditional, *expected_x1_conditional));
169
170 // The densities on X(2) should be the same
171 auto x2_conditional = dynamic_pointer_cast<HybridGaussianConditional>(
172 r: bayesTree[X(j: 2)]->conditional()->inner());
173 auto expected_x2_conditional =
174 dynamic_pointer_cast<HybridGaussianConditional>(
175 r: (*expectedHybridBayesTree)[X(j: 2)]->conditional()->inner());
176 EXPECT(assert_equal(*x2_conditional, *expected_x2_conditional));
177
178 // We only perform manual continuous elimination for 0,0.
179 // The other discrete probabilities on M(2) are calculated the same way
180 const Ordering discreteOrdering{M(j: 0), M(j: 1)};
181 HybridBayesTree::shared_ptr discreteBayesTree =
182 expectedRemainingGraph->eliminateMultifrontal(ordering: discreteOrdering);
183
184 // Test the probability values with regression tests.
185 auto discrete = bayesTree[M(j: 1)]->conditional()->asDiscrete();
186 EXPECT(assert_equal(0.095292, (*discrete)({{M(0), 0}, {M(1), 0}}), 1e-5));
187 EXPECT(assert_equal(0.282758, (*discrete)({{M(0), 1}, {M(1), 0}}), 1e-5));
188 EXPECT(assert_equal(0.314175, (*discrete)({{M(0), 0}, {M(1), 1}}), 1e-5));
189 EXPECT(assert_equal(0.307775, (*discrete)({{M(0), 1}, {M(1), 1}}), 1e-5));
190
191 // Check that the clique conditional generated from incremental elimination
192 // matches that of batch elimination.
193 auto expectedConditional = (*discreteBayesTree)[M(j: 1)]->conditional();
194 auto actualConditional = bayesTree[M(j: 1)]->conditional();
195 EXPECT(assert_equal(*expectedConditional, *actualConditional, 1e-6));
196}

199// Test if we can approximately do the inference (using pruning)
200TEST(HybridNonlinearISAM, ApproxInference) {
201 Switching switching(4);
202 HybridNonlinearISAM incrementalHybrid;
203 HybridNonlinearFactorGraph graph;
204 Values initial;
205
206 // Add the 3 hybrid factors, x0-x1, x1-x2, x2-x3
207 for (size_t i = 0; i < 3; i++) {
208 graph.push_back(factor: switching.binaryFactors.at(i));
209 }
210
211 // Add the Gaussian factors, 1 prior on X(0),
212 // 3 measurements on X(1), X(2), X(3)
213 for (size_t i = 0; i < 4; i++) {
214 graph.push_back(factor: switching.unaryFactors.at(i));
215 initial.insert<double>(j: X(j: i), val: i + 1);
216 }
217
218 // Create ordering.
219 Ordering ordering;
220 for (size_t j = 0; j < 4; j++) {
221 ordering.push_back(x: X(j));
222 }
223
224 // Now we calculate the actual factors using full elimination
225 const auto [unPrunedHybridBayesTree, unPrunedRemainingGraph] =
226 switching.linearizedFactorGraph()
227 .BaseEliminateable::eliminatePartialMultifrontal(ordering);
228
229 size_t maxNrLeaves = 5;
230 incrementalHybrid.update(newFactors: graph, initialValues: initial);
231 HybridGaussianISAM bayesTree = incrementalHybrid.bayesTree();
232
233 bayesTree.prune(maxNumberLeaves: maxNrLeaves);
234
235 /*
236 unPruned factor is:
237 Choice(m3)
238 0 Choice(m2)
239 0 0 Choice(m1)
240 0 0 0 Leaf 0.11267528
241 0 0 1 Leaf 0.18576102
242 0 1 Choice(m1)
243 0 1 0 Leaf 0.18754662
244 0 1 1 Leaf 0.30623871
245 1 Choice(m2)
246 1 0 Choice(m1)
247 1 0 0 Leaf 0.18576102
248 1 0 1 Leaf 0.30622428
249 1 1 Choice(m1)
250 1 1 0 Leaf 0.30623871
251 1 1 1 Leaf 0.5
252
253 pruned factors is:
254 Choice(m3)
255 0 Choice(m2)
256 0 0 Leaf 0
257 0 1 Choice(m1)
258 0 1 0 Leaf 0.18754662
259 0 1 1 Leaf 0.30623871
260 1 Choice(m2)
261 1 0 Choice(m1)
262 1 0 0 Leaf 0
263 1 0 1 Leaf 0.30622428
264 1 1 Choice(m1)
265 1 1 0 Leaf 0.30623871
266 1 1 1 Leaf 0.5
267 */
268
269 auto discreteConditional_m0 = *dynamic_pointer_cast<TableDistribution>(
270 r: bayesTree[M(j: 0)]->conditional()->inner());
271 EXPECT(discreteConditional_m0.keys() == KeyVector({M(0), M(1), M(2)}));
272
273 // Check that the number of leaves after pruning is 5.
274 EXPECT_LONGS_EQUAL(5, discreteConditional_m0.nrValues());
275
276 // Check that the hybrid nodes of the bayes net match those of the pre-pruning
277 // bayes net, at the same positions.
278 auto &unPrunedLastDensity = *dynamic_pointer_cast<HybridGaussianConditional>(
279 r: unPrunedHybridBayesTree->clique(j: X(j: 3))->conditional()->inner());
280 auto &lastDensity = *dynamic_pointer_cast<HybridGaussianConditional>(
281 r: bayesTree[X(j: 3)]->conditional()->inner());
282
283 std::vector<std::pair<DiscreteValues, double>> assignments =
284 discreteConditional_m0.enumerate();
285 // Loop over all assignments and check the pruned components
286 for (auto &&av : assignments) {
287 const DiscreteValues &assignment = av.first;
288 const double value = av.second;
289
290 if (value == 0.0) {
291 EXPECT(lastDensity(assignment) == nullptr);
292 } else {
293 CHECK(lastDensity(assignment));
294 EXPECT(assert_equal(*unPrunedLastDensity(assignment),
295 *lastDensity(assignment)));
296 }
297 }
298}

302TEST(HybridNonlinearISAM, IncrementalApproximate) {
303 Switching switching(5);
304 HybridNonlinearISAM incrementalHybrid;
305 HybridNonlinearFactorGraph graph;
306 Values initial;
307
308 /***** Run Round 1 *****/
309 // Add the 3 hybrid factors, x0-x1, x1-x2, x2-x3
310 for (size_t i = 0; i < 3; i++) {
311 graph.push_back(factor: switching.binaryFactors.at(i));
312 }
313
314 // Add the Gaussian factors, 1 prior on X(0),
315 // 3 measurements on X(1), X(2), X(3)
316 for (size_t i = 0; i < 4; i++) {
317 graph.push_back(factor: switching.unaryFactors.at(i));
318 initial.insert<double>(j: X(j: i), val: i + 1);
319 }
320
321 // TODO(Frank): no mode chain?
322
323 // Run update with pruning
324 size_t maxComponents = 5;
325 incrementalHybrid.update(newFactors: graph, initialValues: initial);
326 incrementalHybrid.prune(maxNumberLeaves: maxComponents);
327 HybridGaussianISAM bayesTree = incrementalHybrid.bayesTree();
328

330 // each with 2, 4, 8, and 5 (pruned) leaves respetively.

332 EXPECT_LONGS_EQUAL(
333 2, bayesTree[X(0)]->conditional()->asHybrid()->nrComponents());
334 EXPECT_LONGS_EQUAL(
335 3, bayesTree[X(1)]->conditional()->asHybrid()->nrComponents());
336 EXPECT_LONGS_EQUAL(
337 5, bayesTree[X(2)]->conditional()->asHybrid()->nrComponents());
338 EXPECT_LONGS_EQUAL(
339 5, bayesTree[X(3)]->conditional()->asHybrid()->nrComponents());
340
341 /***** Run Round 2 *****/
342 graph = HybridGaussianFactorGraph();
343 graph.push_back(factor: switching.binaryFactors.at(i: 3)); // x3-x4
344 graph.push_back(factor: switching.unaryFactors.at(i: 4)); // x4 measurement
345 initial = Values();
346 initial.insert<double>(j: X(j: 4), val: 5);
347
348 // Run update with pruning a second time.
349 incrementalHybrid.update(newFactors: graph, initialValues: initial);
350 incrementalHybrid.prune(maxNumberLeaves: maxComponents);
351 bayesTree = incrementalHybrid.bayesTree();
352
353 // Check if we have a bayes tree with pruned hybrid nodes,
354 // with 5 (pruned) leaves.
355 CHECK_EQUAL(5, bayesTree.size());
356 EXPECT_LONGS_EQUAL(
357 5, bayesTree[X(3)]->conditional()->asHybrid()->nrComponents());
358 EXPECT_LONGS_EQUAL(
359 5, bayesTree[X(4)]->conditional()->asHybrid()->nrComponents());
360}

362/* ************************************************************************/
363// A GTSAM-only test for running inference on a single-legged robot.
364// The leg links are represented by the chain X-Y-Z-W, where X is the base and
365// W is the foot.
366// We use BetweenFactor<Pose2> as constraints between each of the poses.
367TEST(HybridNonlinearISAM, NonTrivial) {
368 /*************** Run Round 1 ***************/
369 HybridNonlinearFactorGraph fg;
370 HybridNonlinearISAM inc;
371
372 // Add a prior on pose x0 at the origin.
373 // A prior factor consists of a mean and
374 // a noise model (covariance matrix)
375 Pose2 prior(0.0, 0.0, 0.0); // prior mean is at origin
376 auto priorNoise = noiseModel::Diagonal::Sigmas(
377 sigmas: Vector3(0.3, 0.3, 0.1)); // 30cm std on x,y, 0.1 rad on theta
378 fg.emplace_shared<PriorFactor<Pose2>>(args: X(j: 0), args&: prior, args&: priorNoise);
379
380 // create a noise model for the landmark measurements
381 auto poseNoise = noiseModel::Isotropic::Sigma(dim: 3, sigma: 0.1);
382
383 // We model a robot's single leg as X - Y - Z - W
384 // where X is the base link and W is the foot link.
385
386 // Add connecting poses similar to PoseFactors in GTD
387 fg.emplace_shared<BetweenFactor<Pose2>>(args: X(j: 0), args: Y(j: 0), args: Pose2(0, 1.0, 0),
388 args&: poseNoise);
389 fg.emplace_shared<BetweenFactor<Pose2>>(args: Y(j: 0), args: Z(j: 0), args: Pose2(0, 1.0, 0),
390 args&: poseNoise);
391 fg.emplace_shared<BetweenFactor<Pose2>>(args: Z(j: 0), args: W(j: 0), args: Pose2(0, 1.0, 0),
392 args&: poseNoise);
393
394 // Create initial estimate
395 Values initial;
396 initial.insert(j: X(j: 0), val: Pose2(0.0, 0.0, 0.0));
397 initial.insert(j: Y(j: 0), val: Pose2(0.0, 1.0, 0.0));
398 initial.insert(j: Z(j: 0), val: Pose2(0.0, 2.0, 0.0));
399 initial.insert(j: W(j: 0), val: Pose2(0.0, 3.0, 0.0));
400
401 // Don't run update now since we don't have discrete variables involved.
402
403 using PlanarMotionModel = BetweenFactor<Pose2>;
404
405 /*************** Run Round 2 ***************/
406 // Add odometry factor with discrete modes.
407 Pose2 odometry(1.0, 0.0, 0.0);
408 auto noise_model = noiseModel::Isotropic::Sigma(dim: 3, sigma: 1.0);
409 auto still = std::make_shared<PlanarMotionModel>(args: W(j: 0), args: W(j: 1), args: Pose2(0, 0, 0),
410 args&: noise_model),
411 moving = std::make_shared<PlanarMotionModel>(args: W(j: 0), args: W(j: 1), args&: odometry,
412 args&: noise_model);
413 std::vector<NoiseModelFactor::shared_ptr> components{moving, still};
414 fg.emplace_shared<HybridNonlinearFactor>(args: DiscreteKey(M(j: 1), 2), args&: components);
415
416 // Add equivalent of ImuFactor
417 fg.emplace_shared<BetweenFactor<Pose2>>(args: X(j: 0), args: X(j: 1), args: Pose2(1.0, 0.0, 0),
418 args&: poseNoise);
419 // PoseFactors-like at k=1
420 fg.emplace_shared<BetweenFactor<Pose2>>(args: X(j: 1), args: Y(j: 1), args: Pose2(0, 1, 0),
421 args&: poseNoise);
422 fg.emplace_shared<BetweenFactor<Pose2>>(args: Y(j: 1), args: Z(j: 1), args: Pose2(0, 1, 0),
423 args&: poseNoise);
424 fg.emplace_shared<BetweenFactor<Pose2>>(args: Z(j: 1), args: W(j: 1), args: Pose2(-1, 1, 0),
425 args&: poseNoise);
426
427 initial.insert(j: X(j: 1), val: Pose2(1.0, 0.0, 0.0));
428 initial.insert(j: Y(j: 1), val: Pose2(1.0, 1.0, 0.0));
429 initial.insert(j: Z(j: 1), val: Pose2(1.0, 2.0, 0.0));
430 // The leg link did not move so we set the expected pose accordingly.
431 initial.insert(j: W(j: 1), val: Pose2(0.0, 3.0, 0.0));
432
433 // Update without pruning
434 // The result is a HybridBayesNet with 1 discrete variable M(1).
435 // P(X | measurements) = P(W0|Z0, W1, M1) P(Z0|Y0, W1, M1) P(Y0|X0, W1, M1)
436 // P(X0 | X1, W1, M1) P(W1|Z1, X1, M1) P(Z1|Y1, X1, M1)
437 // P(Y1 | X1, M1)P(X1 | M1)P(M1)
438 // The MHS tree is a 1 level tree for time indices (1,) with 2 leaves.
439 inc.update(newFactors: fg, initialValues: initial);
440
441 fg = HybridNonlinearFactorGraph();
442 initial = Values();
443
444 /*************** Run Round 3 ***************/
445 // Add odometry factor with discrete modes.
446 still = std::make_shared<PlanarMotionModel>(args: W(j: 1), args: W(j: 2), args: Pose2(0, 0, 0),
447 args&: noise_model);
448 moving =
449 std::make_shared<PlanarMotionModel>(args: W(j: 1), args: W(j: 2), args&: odometry, args&: noise_model);
450 components = {moving, still};
451 fg.emplace_shared<HybridNonlinearFactor>(args: DiscreteKey(M(j: 2), 2), args&: components);
452
453 // Add equivalent of ImuFactor
454 fg.emplace_shared<BetweenFactor<Pose2>>(args: X(j: 1), args: X(j: 2), args: Pose2(1.0, 0.0, 0),
455 args&: poseNoise);
456 // PoseFactors-like at k=1
457 fg.emplace_shared<BetweenFactor<Pose2>>(args: X(j: 2), args: Y(j: 2), args: Pose2(0, 1, 0),
458 args&: poseNoise);
459 fg.emplace_shared<BetweenFactor<Pose2>>(args: Y(j: 2), args: Z(j: 2), args: Pose2(0, 1, 0),
460 args&: poseNoise);
461 fg.emplace_shared<BetweenFactor<Pose2>>(args: Z(j: 2), args: W(j: 2), args: Pose2(-2, 1, 0),
462 args&: poseNoise);
463
464 initial.insert(j: X(j: 2), val: Pose2(2.0, 0.0, 0.0));
465 initial.insert(j: Y(j: 2), val: Pose2(2.0, 1.0, 0.0));
466 initial.insert(j: Z(j: 2), val: Pose2(2.0, 2.0, 0.0));
467 initial.insert(j: W(j: 2), val: Pose2(0.0, 3.0, 0.0));
468
469 // Now we prune!
470 // P(X | measurements) = P(W0|Z0, W1, M1) P(Z0|Y0, W1, M1) P(Y0|X0, W1, M1)
471 // P(X0 | X1, W1, M1) P(W1|W2, Z1, X1, M1, M2)
472 // P(Z1| W2, Y1, X1, M1, M2) P(Y1 | W2, X1, M1, M2)
473 // P(X1 | W2, X2, M1, M2) P(W2|Z2, X2, M1, M2)
474 // P(Z2|Y2, X2, M1, M2) P(Y2 | X2, M1, M2)
475 // P(X2 | M1, M2) P(M1, M2)
476 // The MHS at this point should be a 2 level tree on (1, 2).
477 // 1 has 2 choices, and 2 has 4 choices.
478 inc.update(newFactors: fg, initialValues: initial);
479 inc.prune(maxNumberLeaves: 2);
480
481 fg = HybridNonlinearFactorGraph();
482 initial = Values();
483
484 /*************** Run Round 4 ***************/
485 // Add odometry factor with discrete modes.
486 still = std::make_shared<PlanarMotionModel>(args: W(j: 2), args: W(j: 3), args: Pose2(0, 0, 0),
487 args&: noise_model);
488 moving =
489 std::make_shared<PlanarMotionModel>(args: W(j: 2), args: W(j: 3), args&: odometry, args&: noise_model);
490 components = {moving, still};
491 fg.emplace_shared<HybridNonlinearFactor>(args: DiscreteKey(M(j: 3), 2), args&: components);
492
493 // Add equivalent of ImuFactor
494 fg.emplace_shared<BetweenFactor<Pose2>>(args: X(j: 2), args: X(j: 3), args: Pose2(1.0, 0.0, 0),
495 args&: poseNoise);
496 // PoseFactors-like at k=3
497 fg.emplace_shared<BetweenFactor<Pose2>>(args: X(j: 3), args: Y(j: 3), args: Pose2(0, 1, 0),
498 args&: poseNoise);
