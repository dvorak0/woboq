18#include <gtsam/hybrid/HybridBayesNet.h>

23namespace gtsam {
24namespace tiny {
25
26using symbol_shorthand::M;
27using symbol_shorthand::X;
28using symbol_shorthand::Z;
29
30// Create mode key: 0 is low-noise, 1 is high-noise.
31[[maybe_unused]] static const DiscreteKey mode{M(j: 0), 2};
32
33/**
34 * Create a tiny two variable hybrid model which represents
35 * the generative probability P(z,x,mode) = P(z|x,mode)P(x)P(mode).
36 * num_measurements is the number of measurements of the continuous variable x0.
37 * If manyModes is true, then we introduce one mode per measurement.
38 */
39[[maybe_unused]] static HybridBayesNet createHybridBayesNet(
40 size_t num_measurements = 1, bool manyModes = false) {
41 HybridBayesNet bayesNet;
42
43 // Create hybrid Gaussian factor z_i = x0 + noise for each measurement.
44 std::vector<std::pair<Vector, double>> measurementModels{{Z_1x1, 0.5},
45 {Z_1x1, 3.0}};
46 for (size_t i = 0; i < num_measurements; i++) {
47 const auto mode_i = manyModes ? DiscreteKey{M(j: i), 2} : mode;
48 bayesNet.emplace_shared<HybridGaussianConditional>(args: mode_i, args: Z(j: i), args: I_1x1,
49 args: X(j: 0), args&: measurementModels);
50 }
51
52 // Create prior on X(0).
53 bayesNet.push_back(
54 conditional: GaussianConditional::sharedMeanAndStddev(args: X(j: 0), args: Vector1(5.0), args: 0.5));
55
56 // Add prior on mode.
57 const size_t nrModes = manyModes ? num_measurements : 1;
58 for (size_t i = 0; i < nrModes; i++) {
59 bayesNet.emplace_shared<DiscreteConditional>(args: DiscreteKey{M(j: i), 2}, args: "4/6");
60 }
61 return bayesNet;
62}
63
64/**
65 * Create a tiny two variable hybrid factor graph which represents a discrete
66 * mode and a continuous variable x0, given a number of measurements of the
67 * continuous variable x0. If no measurements are given, they are sampled from
68 * the generative Bayes net model HybridBayesNet::Example(num_measurements)
69 */
70[[maybe_unused]] static HybridGaussianFactorGraph
71createHybridGaussianFactorGraph(size_t num_measurements = 1,
72 std::optional<VectorValues> measurements = {},
73 bool manyModes = false) {
74 auto bayesNet = createHybridBayesNet(num_measurements, manyModes);
75 if (measurements) {
76 // Use the measurements to create a hybrid factor graph.
77 return bayesNet.toFactorGraph(measurements: *measurements);
78 } else {
79 // Sample from the generative model to create a hybrid factor graph.
80 return bayesNet.toFactorGraph(measurements: bayesNet.sample().continuous());
81 }
82}
