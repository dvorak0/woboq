42/* ************************************************************************* */
43// Test if methods to get keys work as expected.
44TEST(HybridFactorGraph, Keys) {
45 HybridGaussianFactorGraph hfg;
46
47 // Add prior on x0
48 hfg.add(factorOrContainer: JacobianFactor(X(j: 0), I_3x3, Z_3x1));
49
50 // Add factor between x0 and x1
51 hfg.add(factorOrContainer: JacobianFactor(X(j: 0), I_3x3, X(j: 1), -I_3x3, Z_3x1));
52
53 // Add a hybrid Gaussian factor Ï•(x1, c1)
54 DiscreteKey m1(M(j: 1), 2);
55 std::vector<GaussianFactor::shared_ptr> components{
56 std::make_shared<JacobianFactor>(args: X(j: 1), args: I_3x3, args: Z_3x1),
57 std::make_shared<JacobianFactor>(args: X(j: 1), args: I_3x3, args: Vector3::Ones())};
58 hfg.add(factorOrContainer: HybridGaussianFactor(m1, components));
59
60 KeySet expected_continuous{X(j: 0), X(j: 1)};
61 EXPECT(
62 assert_container_equality(expected_continuous, hfg.continuousKeySet()));
63
64 KeySet expected_discrete{M(j: 1)};
65 EXPECT(assert_container_equality(expected_discrete, hfg.discreteKeySet()));
66}

71 return TestRegistry::runAllTests(result&: tr);
72}
