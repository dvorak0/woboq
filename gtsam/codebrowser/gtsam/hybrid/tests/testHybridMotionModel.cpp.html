50void addMeasurement(HybridBayesNet &hbn, Key z_key, Key x_key, double sigma) {
51 auto measurement_model = noiseModel::Isotropic::Sigma(dim: 1, sigma);
52 hbn.emplace_shared<GaussianConditional>(args&: z_key, args: Vector1(0.0), args: I_1x1, args&: x_key,
53 args: -I_1x1, args&: measurement_model);
54}

56/// Create hybrid motion model p(x1 | x0, m1)
57static HybridGaussianConditional::shared_ptr CreateHybridMotionModel(
58 double mu0, double mu1, double sigma0, double sigma1) {
59 std::vector<std::pair<Vector, double>> motionModels{{Vector1(mu0), sigma0},
60 {Vector1(mu1), sigma1}};
61 return std::make_shared<HybridGaussianConditional>(args&: m1, args: X(j: 1), args: I_1x1, args: X(j: 0),
62 args&: motionModels);
63}

65/// Create two state Bayes network with 1 or two measurement models
66HybridBayesNet CreateBayesNet(
67 const HybridGaussianConditional::shared_ptr &hybridMotionModel,
68 bool add_second_measurement = false) {
69 HybridBayesNet hbn;
70
71 // Add measurement model p(z0 | x0)
72 addMeasurement(hbn, z_key: Z(j: 0), x_key: X(j: 0), sigma: 3.0);
73
74 // Optionally add second measurement model p(z1 | x1)
75 if (add_second_measurement) {
76 addMeasurement(hbn, z_key: Z(j: 1), x_key: X(j: 1), sigma: 3.0);
77 }
78
79 // Add hybrid motion model
80 hbn.push_back(conditional: hybridMotionModel);
81
82 // Discrete uniform prior.
83 hbn.emplace_shared<DiscreteConditional>(args&: m1, args: "50/50");
84
85 return hbn;
86}

88/// Approximate the discrete marginal P(m1) using importance sampling
89std::pair<double, double> approximateDiscreteMarginal(
90 const HybridBayesNet &hbn,
91 const HybridGaussianConditional::shared_ptr &hybridMotionModel,
92 const VectorValues &given, size_t N = 100000) {
93 /// Create importance sampling network q(x0,x1,m) = p(x1|x0,m1) q(x0) P(m1),
94 /// using q(x0) = N(z0, sigmaQ) to sample x0.
95 HybridBayesNet q;
96 q.push_back(conditional: hybridMotionModel); // Add hybrid motion model
97 q.emplace_shared<GaussianConditional>(args: GaussianConditional::FromMeanAndStddev(
98 key: X(j: 0), mu: given.at(j: Z(j: 0)), /* sigmaQ = */ sigma: 3.0)); // Add proposal q(x0) for x0
99 q.emplace_shared<DiscreteConditional>(args&: m1, args: "50/50"); // Discrete prior.
100
101 // Do importance sampling
102 double w0 = 0.0, w1 = 0.0;
103 std::mt19937_64 rng(42);
104 for (size_t i = 0; i < N; i++) {
105 HybridValues sample = q.sample(rng: &rng);
106 sample.insert(values: given);
107 double weight = hbn.evaluate(values: sample) / q.evaluate(values: sample);
108 (sample.atDiscrete(j: M(j: 1)) == 0) ? w0 += weight : w1 += weight;
109 }
110 double pm1 = w1 / (w0 + w1);
111 std::cout << "p(m0) = " << 100 * (1.0 - pm1) << std::endl;
112 std::cout << "p(m1) = " << 100 * pm1 << std::endl;
113 return {1.0 - pm1, pm1};
114}

117/**
118 * Test a model p(z0|x0)p(z1|x1)p(x1|x0,m1)P(m1).
119 *
120 * p(x1|x0,m1) has mode-dependent mean but same covariance.
121 *
122 * Converting to a factor graph gives us ϕ(x0;z0)ϕ(x1;z1)ϕ(x1,x0,m1)P(m1)
123 *
124 * If we only have a measurement on x0, then
125 * the posterior probability of m1 should be 0.5/0.5.
126 * Getting a measurement on z1 gives use more information.
127 */
128TEST(HybridGaussianFactorGraph, TwoStateModel) {
129 double mu0 = 1.0, mu1 = 3.0;
130 double sigma = 0.5;
131 auto hybridMotionModel = CreateHybridMotionModel(mu0, mu1, sigma0: sigma, sigma1: sigma);
132
133 // Start with no measurement on x1, only on x0
134 const Vector1 z0(0.5);
135
136 VectorValues given;
137 given.insert(j: Z(j: 0), value: z0);
138
139 {
140 HybridBayesNet hbn = CreateBayesNet(hybridMotionModel);
141 HybridGaussianFactorGraph gfg = hbn.toFactorGraph(measurements: given);
142 HybridBayesNet::shared_ptr bn = gfg.eliminateSequential();
143
144 // Since no measurement on x1, we hedge our bets
145 // Importance sampling run with 100k samples gives 50.051/49.949
146 // approximateDiscreteMarginal(hbn, hybridMotionModel, given);
147 TableDistribution expected(m1, "50 50");
148 EXPECT(
149 assert_equal(expected, *(bn->at(2)->asDiscrete<TableDistribution>())));
150 }
151
152 {
153 // If we set z1=4.5 (>> 2.5 which is the halfway point),
154 // probability of discrete mode should be leaning to m1==1.
155 const Vector1 z1(4.5);
156 given.insert(j: Z(j: 1), value: z1);
157
158 HybridBayesNet hbn = CreateBayesNet(hybridMotionModel, add_second_measurement: true);
159 HybridGaussianFactorGraph gfg = hbn.toFactorGraph(measurements: given);
160 HybridBayesNet::shared_ptr bn = gfg.eliminateSequential();
161
162 // Since we have a measurement on x1, we get a definite result
163 // Values taken from an importance sampling run with 100k samples:
164 // approximateDiscreteMarginal(hbn, hybridMotionModel, given);
165 TableDistribution expected(m1, "44.3854 55.6146");
166 EXPECT(assert_equal(expected, *(bn->at(2)->asDiscrete<TableDistribution>()),
167 0.02));
168 }
169}

174 *
175 * P(x1|x0,m1) has different means and different covariances.
176 *
177 * Converting to a factor graph gives us

184TEST(HybridGaussianFactorGraph, TwoStateModel2) {
185 double mu0 = 1.0, mu1 = 3.0;
186 double sigma0 = 0.5, sigma1 = 2.0;
187 auto hybridMotionModel = CreateHybridMotionModel(mu0, mu1, sigma0, sigma1);
188
189 // Start with no measurement on x1, only on x0
190 const Vector1 z0(0.5);
191 VectorValues given;
192 given.insert(j: Z(j: 0), value: z0);
193
194 {
195 HybridBayesNet hbn = CreateBayesNet(hybridMotionModel);
196 HybridGaussianFactorGraph gfg = hbn.toFactorGraph(measurements: given);
197
198 HybridBayesNet::shared_ptr eliminated = gfg.eliminateSequential();
199
200 for (VectorValues vv :
201 {VectorValues{{X(j: 0), Vector1(0.0)}, {X(j: 1), Vector1(1.0)}},
202 VectorValues{{X(j: 0), Vector1(0.5)}, {X(j: 1), Vector1(3.0)}}}) {
203 vv.insert(values: given); // add measurements for HBN
204 const auto &expectedDiscretePosterior = hbn.discretePosterior(continuousValues: vv);
205
206 // Equality of posteriors asserts that the factor graph is correct (same
207 // ratios for all modes)
208 EXPECT(
209 assert_equal(expectedDiscretePosterior, gfg.discretePosterior(vv)));
210
211 // This one asserts that HBN resulting from elimination is correct.
212 EXPECT(assert_equal(expectedDiscretePosterior,
213 eliminated->discretePosterior(vv)));
214 }
215
216 // Importance sampling run with 100k samples gives 50.095/49.905
217 // approximateDiscreteMarginal(hbn, hybridMotionModel, given);
218
219 // Since no measurement on x1, we a 50/50 probability
220 auto p_m = eliminated->at(i: 2)->asDiscrete();
221 EXPECT_DOUBLES_EQUAL(0.5, p_m->operator()({{M(1), 0}}), 1e-9);
222 EXPECT_DOUBLES_EQUAL(0.5, p_m->operator()({{M(1), 1}}), 1e-9);
223 }
224
225 {
226 // Now we add a measurement z1 on x1
227 const Vector1 z1(4.0); // favors m==1
228 given.insert(j: Z(j: 1), value: z1);
229
230 HybridBayesNet hbn = CreateBayesNet(hybridMotionModel, add_second_measurement: true);
231 HybridGaussianFactorGraph gfg = hbn.toFactorGraph(measurements: given);
232 HybridBayesNet::shared_ptr eliminated = gfg.eliminateSequential();
233
234 // Check that ratio of Bayes net and factor graph for different modes is
235 // equal for several values of {x0,x1}.
236 for (VectorValues vv :
237 {VectorValues{{X(j: 0), Vector1(0.0)}, {X(j: 1), Vector1(1.0)}},
238 VectorValues{{X(j: 0), Vector1(0.5)}, {X(j: 1), Vector1(3.0)}}}) {
239 vv.insert(values: given); // add measurements for HBN
240 const auto &expectedDiscretePosterior = hbn.discretePosterior(continuousValues: vv);
241
242 // Equality of posteriors asserts that the factor graph is correct (same
243 // ratios for all modes)
244 EXPECT(
245 assert_equal(expectedDiscretePosterior, gfg.discretePosterior(vv)));
246
247 // This one asserts that HBN resulting from elimination is correct.
248 EXPECT(assert_equal(expectedDiscretePosterior,
249 eliminated->discretePosterior(vv)));
250 }
251
252 // Values taken from an importance sampling run with 100k samples:
253 // approximateDiscreteMarginal(hbn, hybridMotionModel, given);
254 TableDistribution expected(m1, "48.3158 51.6842");
255 EXPECT(assert_equal(
256 expected, *(eliminated->at(2)->asDiscrete<TableDistribution>()), 0.02));
257 }
258
259 {
260 // Add a different measurement z1 on x1 that favors m==0
261 const Vector1 z1(1.1);
262 given.insert_or_assign(j: Z(j: 1), value: z1);

264 HybridBayesNet hbn = CreateBayesNet(hybridMotionModel, add_second_measurement: true);
265 HybridGaussianFactorGraph gfg = hbn.toFactorGraph(measurements: given);
266 HybridBayesNet::shared_ptr bn = gfg.eliminateSequential();

268 // Values taken from an importance sampling run with 100k samples:
269 // approximateDiscreteMarginal(hbn, hybridMotionModel, given);
270 TableDistribution expected(m1, "55.396 44.604");
271 EXPECT(assert_equal(expected, *(bn->at(2)->asDiscrete<TableDistribution>()),
272 0.02));
273 }

397namespace test_direct_factor_graph {
398/**
399 * @brief Create a Factor Graph by directly specifying all
400 * the factors instead of creating conditionals first.
401 * This way we can directly provide the likelihoods and
402 * then perform linearization.
403 *
404 * @param values Initial values to linearize around.
405 * @param means The means of the HybridGaussianFactor components.
406 * @param sigmas The covariances of the HybridGaussianFactor components.
407 * @param m1 The discrete key.
408 * @return HybridGaussianFactorGraph
409 */
410static HybridGaussianFactorGraph CreateFactorGraph(
411 const gtsam::Values &values, const std::vector<double> &means,
412 const std::vector<double> &sigmas, DiscreteKey &m1,
413 double measurement_noise = 1e-3) {
414 auto model0 = noiseModel::Isotropic::Sigma(dim: 1, sigma: sigmas[0]);
415 auto model1 = noiseModel::Isotropic::Sigma(dim: 1, sigma: sigmas[1]);
416 auto prior_noise = noiseModel::Isotropic::Sigma(dim: 1, sigma: measurement_noise);
417
418 auto f0 =
419 std::make_shared<BetweenFactor<double>>(args: X(j: 0), args: X(j: 1), args: means[0], args&: model0)
420 ->linearize(x: values);
421 auto f1 =
422 std::make_shared<BetweenFactor<double>>(args: X(j: 0), args: X(j: 1), args: means[1], args&: model1)
423 ->linearize(x: values);
424
425 // Create HybridGaussianFactor
426 // We take negative since we want
427 // the underlying scalar to be log(\sqrt(|2πΣ|))
428 std::vector<GaussianFactorValuePair> factors{{f0, model0->negLogConstant()},
429 {f1, model1->negLogConstant()}};
430 HybridGaussianFactor motionFactor(m1, factors);
431
432 HybridGaussianFactorGraph hfg;
433 hfg.push_back(factor: motionFactor);
434
435 hfg.push_back(factor: PriorFactor<double>(X(j: 0), values.at<double>(j: X(j: 0)), prior_noise)
436 .linearize(x: values));
437
438 return hfg;
439}
440} // namespace test_direct_factor_graph

444 * @brief Test components with differing means but the same covariances.
445 * The factor graph is
446 * *-X1-*-X2
447 * |
448 * M1
449 */
450TEST(HybridGaussianFactorGraph, DifferentMeans) {
451 using namespace test_direct_factor_graph;
452
453 DiscreteKey m1(M(j: 1), 2);
454
455 Values values;

457 values.insert(j: X(j: 0), val: x1);

461
462 HybridGaussianFactorGraph hfg = CreateFactorGraph(values, means, sigmas, m1);

470 DiscreteValues{{M(j: 1), 0}});

472 EXPECT(assert_equal(expected, actual));

475 VectorValues cont0 = bn->optimize(assignment: dv0);
476 double error0 = bn->error(values: HybridValues(cont0, dv0));
477 // regression
478 EXPECT_DOUBLES_EQUAL(0.69314718056, error0, 1e-9);
479
480 DiscreteValues dv1{{M(j: 1), 1}};
481 VectorValues cont1 = bn->optimize(assignment: dv1);

492 HybridValues actual = bn->optimize();
493
494 HybridValues expected(
495 VectorValues{{X(j: 0), Vector1(0.0)}, {X(j: 1), Vector1(0.25)}},
496 DiscreteValues{{M(j: 1), 1}});
497
498 EXPECT(assert_equal(expected, actual));
499
500 {
501 DiscreteValues dv{{M(j: 1), 0}};
502 VectorValues cont = bn->optimize(assignment: dv);
503 double error = bn->error(values: HybridValues(cont, dv));
504 // regression
505 EXPECT_DOUBLES_EQUAL(2.12692448787, error, 1e-9);
506 }
507 {
508 DiscreteValues dv{{M(j: 1), 1}};
509 VectorValues cont = bn->optimize(assignment: dv);
510 double error = bn->error(values: HybridValues(cont, dv));
511 // regression
512 EXPECT_DOUBLES_EQUAL(0.126928487854, error, 1e-9);
