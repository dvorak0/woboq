67[[maybe_unused]] static Switching InitializeEstimationProblem(
68 const size_t K, const double between_sigma, const double measurement_sigma,
69 const std::vector<double>& measurements,
70 const std::string& transitionProbabilityTable,
71 HybridNonlinearFactorGraph* graph, Values* initial) {
72 Switching switching(K, between_sigma, measurement_sigma, measurements,
73 transitionProbabilityTable);
74
75 // Add prior on M(0)
76 graph->push_back(factor: switching.modeChain.at(i: 0));
77
78 // Add the X(0) prior
79 graph->push_back(factor: switching.unaryFactors.at(i: 0));
80 initial->insert(j: X(j: 0), val: switching.linearizationPoint.at<double>(j: X(j: 0)));
81
82 return switching;
83}
