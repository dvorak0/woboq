61 // with given measurements and equal mode priors.
62 Switching switching(K, 1.0, 0.1, measurements, "1/1 1/1");
63 HybridGaussianFactorGraph graph = switching.linearizedFactorGraph();
64
65 Ordering hybridOrdering;
66 for (size_t k = 0; k < K; k++) {
67 hybridOrdering.push_back(x: X(j: k));

144/**
145 * @brief Helper method to get the tree of
146 * unnormalized probabilities as per the elimination scheme.
147 *
148 * Used as a helper to compute q(\mu | M, Z) which is used by
149 * both P(X | M, Z) and P(M | Z).
150 *
151 * @param graph The HybridGaussianFactorGraph to eliminate.
152 * @return AlgebraicDecisionTree<Key>
153 */
154AlgebraicDecisionTree<Key> GetProbPrimeTree(
155 const HybridGaussianFactorGraph& graph) {
156 Ordering continuous(graph.continuousKeySet());
157 const auto [bayesNet, remainingGraph] =
158 graph.eliminatePartialSequential(ordering: continuous);
159
160 auto last_conditional = bayesNet->at(i: bayesNet->size() - 1);
161 DiscreteKeys discrete_keys = last_conditional->discreteKeys();
162
163 const std::vector<DiscreteValues> assignments =
164 DiscreteValues::CartesianProduct(keys: discrete_keys);
165
166 std::reverse(first: discrete_keys.begin(), last: discrete_keys.end());
167
168 vector<VectorValues::shared_ptr> vector_values;
169 for (const DiscreteValues& assignment : assignments) {
170 VectorValues values = bayesNet->optimize(assignment);
171 vector_values.push_back(x: std::make_shared<VectorValues>(args&: values));
172 }
173 DecisionTree<Key, VectorValues::shared_ptr> delta_tree(discrete_keys,
174 vector_values);
175
176 // Get the probPrime tree with the correct leaf probabilities
177 std::vector<double> probPrimes;
178 for (const DiscreteValues& assignment : assignments) {
179 VectorValues delta = *delta_tree(assignment);
180
181 // If VectorValues is empty, it means this is a pruned branch.
182 // Set the probPrime to 0.0.
183 if (delta.size() == 0) {
184 probPrimes.push_back(x: 0.0);
185 continue;
186 }
187
188 double error = graph.error(values: {delta, assignment});
189 probPrimes.push_back(x: exp(x: -error));
190 }
191 AlgebraicDecisionTree<Key> probPrimeTree(discrete_keys, probPrimes);
192 return probPrimeTree;
193}

233/**
234 * Test for correctness of different branches of the P'(Continuous | Discrete)
235 * in the multi-frontal setting. The values should match those of P'(Continuous)
236 * for each discrete mode.
237 */
238TEST(HybridEstimation, ProbabilityMultifrontal) {
239 using namespace estimation_fixture;
240
241 constexpr size_t K = 4;
242
243 double between_sigma = 1.0, measurement_sigma = 0.1;
244
245 // Switching example of robot moving in 1D with given measurements and equal
246 // mode priors.
247 Switching switching(K, between_sigma, measurement_sigma, measurements,
248 "1/1 1/1");
249 auto graph = switching.linearizedFactorGraph();
250
251 // Get the tree of unnormalized probabilities for each mode sequence.
252 AlgebraicDecisionTree<Key> expected_probPrimeTree = GetProbPrimeTree(graph);
253
254 // Eliminate continuous
255 Ordering continuous_ordering(graph.continuousKeySet());
256 const auto [bayesTree, discreteGraph] =
257 graph.eliminatePartialMultifrontal(ordering: continuous_ordering);
258
259 // Get the last continuous conditional which will have all the discrete keys
260 Key last_continuous_key =
261 continuous_ordering.at(n: continuous_ordering.size() - 1);
262 auto last_conditional = (*bayesTree)[last_continuous_key]->conditional();
263 DiscreteKeys discrete_keys = last_conditional->discreteKeys();
264
265 Ordering discrete(graph.discreteKeySet());
266 auto discreteBayesTree = discreteGraph->eliminateMultifrontal(ordering: discrete);
267
268 EXPECT_LONGS_EQUAL(1, discreteBayesTree->size());
269 // DiscreteBayesTree should have only 1 clique
270 auto discrete_clique = (*discreteBayesTree)[discrete.at(n: 0)];
271
272 std::set<HybridBayesTreeClique::shared_ptr> clique_set;
273 for (auto node : bayesTree->nodes()) {
274 clique_set.insert(x: node.second);
275 }
276
277 // Set the root of the bayes tree as the discrete clique
278 for (auto clique : clique_set) {
279 if (clique->conditional()->parents() ==
280 discrete_clique->conditional()->frontals()) {
281 discreteBayesTree->addClique(clique, parent_clique: discrete_clique);
282
283 } else {
284 // Remove the clique from the children of the parents since
285 // it will get added again in addClique.
286 auto clique_it = std::find(first: clique->parent()->children.begin(),
287 last: clique->parent()->children.end(), val: clique);
288 clique->parent()->children.erase(position: clique_it);
289 discreteBayesTree->addClique(clique, parent_clique: clique->parent());
290 }
291 }
292
293 HybridValues hybrid_values = discreteBayesTree->optimize();
294
295 // This is the correct sequence as designed
296 DiscreteValues expectedSequence{{M(j: 0), 1}, {M(j: 1), 1}, {M(j: 2), 0}};
297 EXPECT(assert_equal(expectedSequence, hybrid_values.discrete()));
298}

300/*********************************************************************************
301 // Create a hybrid nonlinear factor graph f(x0, x1, m0; z0, z1)
302 ********************************************************************************/
303static HybridNonlinearFactorGraph CreateHybridNonlinearFactorGraph() {
304 HybridNonlinearFactorGraph nfg;
305
306 constexpr double sigma = 0.5; // measurement noise
307 const auto noise_model = noiseModel::Isotropic::Sigma(dim: 1, sigma);
308
309 // Add "measurement" factors:
310 nfg.emplace_shared<PriorFactor<double>>(args: X(j: 0), args: 0.0, args: noise_model);
311 nfg.emplace_shared<PriorFactor<double>>(args: X(j: 1), args: 1.0, args: noise_model);
312
313 // Add hybrid nonlinear factor:
314 DiscreteKey m(M(j: 0), 2);
315 const auto zero_motion =
316 std::make_shared<BetweenFactor<double>>(args: X(j: 0), args: X(j: 1), args: 0, args: noise_model);
317 const auto one_motion =
318 std::make_shared<BetweenFactor<double>>(args: X(j: 0), args: X(j: 1), args: 1, args: noise_model);
319 std::vector<NoiseModelFactor::shared_ptr> components = {zero_motion,
320 one_motion};
321 nfg.emplace_shared<HybridNonlinearFactor>(args&: m, args&: components);
322
323 return nfg;
324}

367/*********************************************************************************
368 * Test for correctness via sampling.
369 *
370 * Compute the conditional P(x0, m0, x1| z0, z1)
371 * with measurements z0, z1. To do so, we:
372 * 1. Start with the corresponding Factor Graph `FG`.
373 * 2. Eliminate the factor graph into a Bayes Net `BN`.
374 * 3. Sample from the Bayes Net.
375 * 4. Check that the ratio `BN(x)/FG(x) = constant` for all samples `x`.
376 ********************************************************************************/
377TEST(HybridEstimation, CorrectnessViaSampling) {
378 // 1. Create the factor graph from the nonlinear factor graph.
379 const auto fg = CreateHybridGaussianFactorGraph();
380
381 // 2. Eliminate into BN
382 const HybridBayesNet::shared_ptr bn = fg->eliminateSequential();
383
384 // Set up sampling
385 std::mt19937_64 rng(11);
386
387 // Compute the log-ratio between the Bayes net and the factor graph.
388 auto compute_ratio = [&](const HybridValues& sample) -> double {
389 return bn->evaluate(values: sample) / fg->probPrime(values: sample);
390 };
391
392 // The error evaluated by the factor graph and the Bayes net should differ by
393 // the normalizing term computed via the Bayes net determinant.
394 const HybridValues sample = bn->sample(rng: &rng);
395 double expected_ratio = compute_ratio(sample);
396
397 // 3. Do sampling
398 constexpr int num_samples = 10;
399 for (size_t i = 0; i < num_samples; i++) {
400 // Sample from the bayes net
401 const HybridValues sample = bn->sample(rng: &rng);
402
403 // 4. Check that the ratio is constant.
404 EXPECT_DOUBLES_EQUAL(expected_ratio, compute_ratio(sample), 1e-6);
405 }
406}

528/*
529 * Test discrete optimization using a simple mixture.
530 *
531 * Construct a single factor (for a single variable) consisting of a
532 * discrete-conditional mixture.
533 * Here we have a "null hypothesis" consisting of
534 * a Gaussian with large variance and an "alternative hypothesis"
535 * consisting of a Gaussian with smaller variance.
536 * After initializing the continuous variable far away from
537 * the ground-truth solution (x1 = 0), the discrete hypothesis selector
538 * will initially choose the null hypothesis.
539 */
540TEST(HybridEstimation, DiscreteMixture) {
541 using namespace discrete_mixture_fixture;
542
543 // Make an empty hybrid factor graph
544 HybridNonlinearFactorGraph dcfg;
545
546 // Make a symbol for a single continuous variable and add to KeyVector
547 KeyVector keys;
548 keys.push_back(x: x1);
549
550 std::vector<NoiseModelFactor::shared_ptr> factorComponents{f1(), fNullHypo()};
551
552 HybridNonlinearFactor dcMixture(dk, factorComponents);
553 dcfg.push_back(factor: dcMixture);
554
555 DiscreteKey dkTest = dcMixture.discreteKeys()[0];
556 std::cout << "DK 1st: " << DefaultKeyFormatter(dkTest.first) << std::endl;
557 std::cout << "DK 2nd: " << dkTest.second << std::endl;
558
559 // Let's make an initial guess for the continuous values
560 Values initialGuess;
561 double initVal = -2.5;
562 initialGuess.insert(j: x1, val: initVal);
563
564 // We also need an initial guess for the discrete variables (this will only be
565 // used if it is needed by your factors), here it is ignored.
566 DiscreteValues initialGuessDiscrete;
567 initialGuessDiscrete[dk.first] = 0;
568
569 // Let's make a discrete factor graph
570 DiscreteFactorGraph dfg;
571
572 auto calculateError = [&](const NonlinearFactorValuePair& pair) -> double {
573 if (pair.first) {
574 auto gaussianNoiseModel = std::dynamic_pointer_cast<noiseModel::Gaussian>(
575 r: pair.first->noiseModel());
576 // `error` has the following contributions:
577 // - the scalar is the sum of all mode-dependent constants
578 // - factor->error(initial) is the error on the initial values
579 // - negLogK is log normalization constant from the noise model
580 return pair.second + pair.first->error(c: initialGuess) +
581 gaussianNoiseModel->negLogConstant();
582 } else {
583 // If the factor has been pruned, return infinite error
584 return std::numeric_limits<double>::infinity();
585 }
586 };

608/*
609 * Test continuous optimization using a simple mixture
610 *
611 * Construct a single factor (for a single variable) consisting of a
612 * discrete-conditional mixture. Here we have a "null hypothesis" consisting
613 of
614 * a Gaussian with large variance and an "alternative hypothesis" consisting
615 of
616 * a Gaussian with smaller variance. After initializing the continuous
617 variable
618 * far away from the ground-truth solution (x1 = 0), the discrete hypothesis
619 * selector will initially choose the null hypothesis.
620 *
621 * After a step of continuous optimization, the continuous solution will move
622 to
623 * x1 = 0 (since the problem is linear, we have convergence in one step).
624 * Finally, updating the discrete value will show that the correct discrete
625 * hypothesis will shift to the "alternative hypothesis."
626 */
627TEST(HybridEstimation, ContinuousMixture) {
628 using namespace discrete_mixture_fixture;
629
630 // Make an empty hybrid factor graph
631 HybridNonlinearFactorGraph dcfg;
632
633 // Make a symbol for a single continuous variable and add to KeyVector
634 KeyVector keys;
635 keys.push_back(x: x1);
636
637 std::vector<NonlinearFactorValuePair> factorComponents{
638 {f1(), priorNoise1()->negLogConstant()},
639 {fNullHypo(), priorNoiseNullHypo()->negLogConstant()}};
640
641 HybridNonlinearFactor dcMixture(dk, factorComponents);
642 dcfg.push_back(factor: dcMixture);
643
644 // Test calculation of negative log probability
645 DiscreteValues dv1, dvNH;
646 dv1[dk.first] = 0;
647 dvNH[dk.first] = 1;
648 Values xvals;
649 xvals.insert(j: x1, val: 0.0);
650 const double negLogProb_1 = dcMixture.error(continuousValues: xvals, assignment: dv1);
651 const double negLogProb_NH = dcMixture.error(continuousValues: xvals, assignment: dvNH);
652
653 // As calculated in MATLAB using -log(normpdf(0,0,1)), -log(normpdf(0,0,8))
654 EXPECT_DOUBLES_EQUAL(0.9189, negLogProb_1, 1e-3);
655 EXPECT_DOUBLES_EQUAL(2.9984, negLogProb_NH, 1e-3);
656
657 // Let's make an initial guess for the continuous values
658 Values values;
659 double initVal = -2.5;
660 values.insert(j: x1, val: initVal);
661
662 // We also need an initial guess for the discrete variables (this will only be
663 // used if it is needed by your factors), here it is ignored.
664 DiscreteValues initialGuessDiscrete;
665 initialGuessDiscrete[dk.first] = 0;
666
667 // Let's make some factor graphs
668 DiscreteFactorGraph dfg;
669 NonlinearFactorGraph nfg;
670
671 // Create DiscreteFactor from initial guess
672 DiscreteBoundaryFactor dtf(dcMixture.discreteKeys(), dcMixture.factors(),
673 values);
674
675 dfg.push_back(factor: dtf);
676
677 // Solve for discrete given continuous
678 DiscreteValues mostProbableEstimate = dfg.optimize();
679
680 // Get the most probable estimate
681 size_t mpeD = mostProbableEstimate.at(k: dk.first);
682
683 // From previous test we know this is == 1
684 EXPECT_LONGS_EQUAL(1, mpeD);
685
686 // Get the marginals
687 DiscreteMarginals newDiscreteMarginals(dfg);
688 Vector newMargProbs = newDiscreteMarginals.marginalProbabilities(key: dk);
689
690 // Use discrete info to get nonlinear factors
691 nfg.push_back(factor: dcMixture.factors()(mostProbableEstimate).first);

693 // Setup isam
694 ISAM2Params isam_params;
695 isam_params.relinearizeThreshold = 0.01;
696 isam_params.relinearizeSkip = 1;
697 isam_params.setOptimizationParams(ISAM2DoglegParams());
698 ISAM2 isam(isam_params);
699 isam.update(newFactors: nfg, newTheta: values);

702 values = isam.calculateEstimate();

704 // Now update the continuous info in the discrete solver
705 dtf = DiscreteBoundaryFactor(dcMixture.discreteKeys(), dcMixture.factors(),
706 values);

709 dfg.push_back(factor: dtf);

712 mostProbableEstimate = dfg.optimize();
713 mpeD = mostProbableEstimate.at(k: dk.first);

716 EXPECT_LONGS_EQUAL(0, mpeD);
