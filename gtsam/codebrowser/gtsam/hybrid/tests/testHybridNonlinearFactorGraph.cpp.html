53TEST(HybridNonlinearFactorGraph, GaussianFactorGraph) {
54 HybridNonlinearFactorGraph fg;

59 // Linearization point
60 Values linearizationPoint;

70 EXPECT_DOUBLES_EQUAL(fg.error(hybridValues), ghfg.error(hybridValues), 1e-9);
71}

216// Test hybrid nonlinear factor graph errorTree
217TEST(HybridNonlinearFactorGraph, ErrorTree) {
218 Switching s(3);
219
220 const HybridNonlinearFactorGraph &graph = s.nonlinearFactorGraph();
221 const Values &values = s.linearizationPoint;
222
223 auto error_tree = graph.errorTree(values: s.linearizationPoint);
224
225 auto dkeys = graph.discreteKeys();
226 DiscreteKeys discrete_keys(dkeys.begin(), dkeys.end());
227
228 // Compute the sum of errors for each factor.
229 auto assignments = DiscreteValues::CartesianProduct(keys: discrete_keys);
230 std::vector<double> leaves(assignments.size());
231 for (auto &&factor : graph) {
232 for (size_t i = 0; i < assignments.size(); ++i) {
233 leaves[i] +=
234 factor->error(c: HybridValues(VectorValues(), assignments[i], values));
235 }
236 }
237 // Swap i=1 and i=2 to give correct ordering.
238 double temp = leaves[1];
239 leaves[1] = leaves[2];
240 leaves[2] = temp;
241 AlgebraicDecisionTree<Key> expected_error(discrete_keys, leaves);
242
243 EXPECT(assert_equal(expected_error, error_tree, 1e-7));
244}

331/****************************************************************************
332 * Helper method to generate gaussian factor graphs with a specific mode.
333 */
334GaussianFactorGraph::shared_ptr batchGFG(double between,
335 Values linearizationPoint) {
336 NonlinearFactorGraph graph;
337 graph.addPrior<double>(key: X(j: 0), prior: 0, model: Isotropic::Sigma(dim: 1, sigma: 0.1));
338
339 auto between_x0_x1 = std::make_shared<MotionModel>(args: X(j: 0), args: X(j: 1), args&: between,
340 args: Isotropic::Sigma(dim: 1, sigma: 1.0));
341
342 graph.push_back(factor: between_x0_x1);
343
344 return graph.linearize(linearizationPoint);
345}

460/****************************************************************************
461 * Test full elimination
462 */
463TEST(HybridNonlinearFactorGraph, Full_Elimination) {
464 Switching self(3);
465
466 // We first do a partial elimination
467 DiscreteBayesNet discreteBayesNet;
468
469 {
470 // Create ordering.
471 Ordering ordering;
472 for (size_t k = 0; k < self.K; k++) ordering.push_back(x: X(j: k));
473
474 // Eliminate partially.
475 const auto [hybridBayesNet_partial, remainingFactorGraph_partial] =
476 self.linearizedFactorGraph().eliminatePartialSequential(ordering);
477
478 DiscreteFactorGraph discrete_fg =
479 remainingFactorGraph_partial->discreteFactors();
480
481 ordering.clear();
482 for (size_t k = 0; k < self.K - 1; k++) ordering.push_back(x: M(j: k));
483 discreteBayesNet =
484 *discrete_fg.eliminateSequential(ordering, function: EliminateDiscrete);
485 }
486
487 // Create ordering.
488 Ordering ordering;
489 for (size_t k = 0; k < self.K; k++) ordering.push_back(x: X(j: k));
490 for (size_t k = 0; k < self.K - 1; k++) ordering.push_back(x: M(j: k));
491
492 // Eliminate partially.
493 HybridBayesNet::shared_ptr hybridBayesNet =
494 self.linearizedFactorGraph().eliminateSequential(ordering);
495
496 CHECK(hybridBayesNet);
497 EXPECT_LONGS_EQUAL(5, hybridBayesNet->size());
498 // p(x0 | x1, m0)
499 EXPECT(hybridBayesNet->at(0)->frontals() == KeyVector{X(0)});
500 EXPECT(hybridBayesNet->at(0)->parents() == KeyVector({X(1), M(0)}));
501 // p(x1 | x2, m0, m1)
502 EXPECT(hybridBayesNet->at(1)->frontals() == KeyVector{X(1)});
503 EXPECT(hybridBayesNet->at(1)->parents() == KeyVector({X(2), M(0), M(1)}));
504 // p(x2 | m0, m1)
505 EXPECT(hybridBayesNet->at(2)->frontals() == KeyVector{X(2)});
506 EXPECT(hybridBayesNet->at(2)->parents() == KeyVector({M(0), M(1)}));
507 // P(m0 | m1)
508 EXPECT(hybridBayesNet->at(3)->frontals() == KeyVector{M(0)});
509 EXPECT(hybridBayesNet->at(3)->parents() == KeyVector({M(1)}));
510 EXPECT(
511 dynamic_pointer_cast<DiscreteConditional>(hybridBayesNet->at(3)->inner())
512 ->equals(*discreteBayesNet.at(0)));
513 // P(m1)
514 EXPECT(hybridBayesNet->at(4)->frontals() == KeyVector{M(1)});
515 EXPECT_LONGS_EQUAL(0, hybridBayesNet->at(4)->nrParents());
516 TableDistribution dtc =
517 *hybridBayesNet->at(i: 4)->asDiscrete<TableDistribution>();
518 EXPECT(DiscreteConditional(dtc.nrFrontals(), dtc.toDecisionTreeFactor())
519 .equals(*discreteBayesNet.at(1)));
520}

857/****************************************************************************
858 * Simple PlanarSLAM example test with 2 poses and 2 landmarks (each pose
859 * connects to 1 landmark) to expose issue with default decision tree creation
860 * in hybrid elimination. The hybrid factor is between the poses X0 and X1.
861 * The issue arises if we eliminate a landmark variable first since it is not
862 * connected to a HybridFactor.
863 */
864TEST(HybridNonlinearFactorGraph, DefaultDecisionTree) {
865 HybridNonlinearFactorGraph fg;
866
867 // Add a prior on pose x0 at the origin.
868 // A prior factor consists of a mean and a noise model (covariance matrix)
869 Pose2 prior(0.0, 0.0, 0.0); // prior mean is at origin
870 auto priorNoise = noiseModel::Diagonal::Sigmas(
871 sigmas: Vector3(0.3, 0.3, 0.1)); // 30cm std on x,y, 0.1 rad on theta
872 fg.emplace_shared<PriorFactor<Pose2>>(args: X(j: 0), args&: prior, args&: priorNoise);
873
874 using PlanarMotionModel = BetweenFactor<Pose2>;
875
876 // Add odometry factor
877 Pose2 odometry(2.0, 0.0, 0.0);
878 auto noise_model = noiseModel::Isotropic::Sigma(dim: 3, sigma: 1.0);
879 std::vector<NoiseModelFactor::shared_ptr> motion_models = {
880 std::make_shared<PlanarMotionModel>(args: X(j: 0), args: X(j: 1), args: Pose2(0, 0, 0),
881 args&: noise_model),
882 std::make_shared<PlanarMotionModel>(args: X(j: 0), args: X(j: 1), args&: odometry, args&: noise_model)};
883 fg.emplace_shared<HybridNonlinearFactor>(args: DiscreteKey{M(j: 1), 2}, args&: motion_models);
884
885 // Add Range-Bearing measurements to from X0 to L0 and X1 to L1.
886 // create a noise model for the landmark measurements
887 auto measurementNoise = noiseModel::Diagonal::Sigmas(
888 sigmas: Vector2(0.1, 0.2)); // 0.1 rad std on bearing, 20cm on range
889 // create the measurement values - indices are (pose id, landmark id)
890 Rot2 bearing11 = Rot2::fromDegrees(theta: 45), bearing22 = Rot2::fromDegrees(theta: 90);
891 double range11 = std::sqrt(x: 4.0 + 4.0), range22 = 2.0;
892
893 // Add Bearing-Range factors
894 fg.emplace_shared<BearingRangeFactor<Pose2, Point2>>(
895 args: X(j: 0), args: L(j: 0), args&: bearing11, args&: range11, args&: measurementNoise);
896 fg.emplace_shared<BearingRangeFactor<Pose2, Point2>>(
897 args: X(j: 1), args: L(j: 1), args&: bearing22, args&: range22, args&: measurementNoise);
898
899 // Create (deliberately inaccurate) initial estimate
900 Values initialEstimate;
901 initialEstimate.insert(j: X(j: 0), val: Pose2(0.5, 0.0, 0.2));
902 initialEstimate.insert(j: X(j: 1), val: Pose2(2.3, 0.1, -0.2));
903 initialEstimate.insert(j: L(j: 0), val: Point2(1.8, 2.1));
904 initialEstimate.insert(j: L(j: 1), val: Point2(4.1, 1.8));
905
906 // We want to eliminate variables not connected to DCFactors first.
907 const Ordering ordering{L(j: 0), L(j: 1), X(j: 0), X(j: 1)};
908
909 HybridGaussianFactorGraph linearized = *fg.linearize(continuousValues: initialEstimate);
910
911 // This should NOT fail
912 const auto [hybridBayesNet, remainingFactorGraph] =
913 linearized.eliminatePartialSequential(ordering);
914 EXPECT_LONGS_EQUAL(4, hybridBayesNet->size());
915 EXPECT_LONGS_EQUAL(1, remainingFactorGraph->size());
916}

918namespace test_relinearization {
919/**
920 * @brief Create a Factor Graph by directly specifying all
921 * the factors instead of creating conditionals first.
922 * This way we can directly provide the likelihoods and
923 * then perform (re-)linearization.
924 *
925 * @param means The means of the HybridGaussianFactor components.
926 * @param sigmas The covariances of the HybridGaussianFactor components.
927 * @param m1 The discrete key.
928 * @param x0_measurement A measurement on X0
929 * @return HybridGaussianFactorGraph
930 */
931static HybridNonlinearFactorGraph CreateFactorGraph(
932 const std::vector<double> &means, const std::vector<double> &sigmas,
933 DiscreteKey &m1, double x0_measurement, double measurement_noise = 1e-3) {
934 auto model0 = noiseModel::Isotropic::Sigma(dim: 1, sigma: sigmas[0]);
935 auto model1 = noiseModel::Isotropic::Sigma(dim: 1, sigma: sigmas[1]);
936 auto prior_noise = noiseModel::Isotropic::Sigma(dim: 1, sigma: measurement_noise);
937
938 auto f0 =
939 std::make_shared<BetweenFactor<double>>(args: X(j: 0), args: X(j: 1), args: means[0], args&: model0);
940 auto f1 =
941 std::make_shared<BetweenFactor<double>>(args: X(j: 0), args: X(j: 1), args: means[1], args&: model1);
942
943 // Create HybridNonlinearFactor
944 // We take negative since we want
945 // the underlying scalar to be log(\sqrt(|2|))
946 std::vector<NonlinearFactorValuePair> factors{{f0, 0.0}, {f1, 0.0}};
947
948 HybridNonlinearFactor mixtureFactor(m1, factors);
949
950 HybridNonlinearFactorGraph hfg;
951 hfg.push_back(factor: mixtureFactor);
952
953 hfg.push_back(factor: PriorFactor<double>(X(j: 0), x0_measurement, prior_noise));
954
955 return hfg;
956}
957} // namespace test_relinearization
