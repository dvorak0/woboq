19#pragma once

21#include <gtsam/discrete/DecisionTree.h>
22#include <gtsam/inference/Key.h>
23#include <gtsam/linear/GaussianFactorGraph.h>
24
25#include <iostream>
26
27namespace gtsam {
28
29class HybridGaussianFactor;
30
31using GaussianFactorGraphValuePair = std::pair<GaussianFactorGraph, double>;
32
33/// Alias for DecisionTree of GaussianFactorGraphs and their scalar sums
34class GTSAM_EXPORT HybridGaussianProductFactor
35 : public DecisionTree<Key, GaussianFactorGraphValuePair> {
36 public:
37 using Base = DecisionTree<Key, GaussianFactorGraphValuePair>;
38
39 /// @name Constructors
40 /// @{
41
42 /// Default constructor
43 HybridGaussianProductFactor() = default;
44
45 /**
46 * @brief Construct from a single factor
47 * @tparam FACTOR Factor type
48 * @param factor Shared pointer to the factor
49 */
50 template <class FACTOR>
51 HybridGaussianProductFactor(const std::shared_ptr<FACTOR>& factor)
52 : Base(GaussianFactorGraphValuePair{GaussianFactorGraph{factor}, 0.0}) {}
53
54 /**
55 * @brief Construct from DecisionTree
56 * @param tree Decision tree to construct from
57 */
58 HybridGaussianProductFactor(Base&& tree) : Base(std::move(tree)) {}
59
60 ///@}
61
62 /// @name Operators
63 ///@{
64
65 /// Add GaussianFactor into HybridGaussianProductFactor
66 HybridGaussianProductFactor operator+(
67 const GaussianFactor::shared_ptr& factor) const;
68
69 /// Add HybridGaussianFactor into HybridGaussianProductFactor
70 HybridGaussianProductFactor operator+(
71 const HybridGaussianFactor& factor) const;
72
73 /// Add-assign operator for GaussianFactor
74 HybridGaussianProductFactor& operator+=(
75 const GaussianFactor::shared_ptr& factor);
76
77 /// Add-assign operator for HybridGaussianFactor
78 HybridGaussianProductFactor& operator+=(const HybridGaussianFactor& factor);
79
80 ///@}
81
82 /// @name Testable
83 /// @{
84
85 /**
86 * @brief Print the HybridGaussianProductFactor
87 * @param s Optional string to prepend
88 * @param formatter Optional key formatter
89 */
90 void print(const std::string& s = "",
91 const KeyFormatter& formatter = DefaultKeyFormatter) const;
92
93 /**
94 * @brief Check if this HybridGaussianProductFactor is equal to another
95 * @param other The other HybridGaussianProductFactor to compare with
96 * @param tol Tolerance for floating point comparisons
97 * @return true if equal, false otherwise
98 */
99 bool equals(const HybridGaussianProductFactor& other,
100 double tol = 1e-9) const;
101
102 /// @}
103
104 /// @name Other methods
105 ///@{
106
107 /**
108 * @brief Remove empty GaussianFactorGraphs from the decision tree
109 * @return A new HybridGaussianProductFactor with empty GaussianFactorGraphs
110 * removed
111 *
112 * If any GaussianFactorGraph in the decision tree contains a nullptr, convert
113 * that leaf to an empty GaussianFactorGraph with zero scalar sum. This is
114 * needed because the DecisionTree will otherwise create a GaussianFactorGraph
115 * with a single (null) factor, which doesn't register as null.
116 */
117 HybridGaussianProductFactor removeEmpty() const;
118
119 ///@}
120
121 private:
122#if GTSAM_ENABLE_BOOST_SERIALIZATION
123 /** Serialization function */
124 friend class boost::serialization::access;
125 template <class Archive>
126 void serialize(Archive& ar, const unsigned int /*version*/) {
127 ar& BOOST_SERIALIZATION_BASE_OBJECT_NVP(Base);
128 }
129#endif
130};
131
132// Testable traits
133template <>
134struct traits<HybridGaussianProductFactor>
135 : public Testable<HybridGaussianProductFactor> {};
136
137/**
138 * Create a dummy overload of >> for GaussianFactorGraphValuePair
139 * so that HybridGaussianProductFactor compiles
140 * with the constructor
141 * `DecisionTree(const std::vector<LabelC>& labelCs, const std::string& table)`.
142 *
143 * Needed to compile on Windows.
144 */
145std::istream& operator>>(std::istream& is, GaussianFactorGraphValuePair& pair);
146
147} // namespace gtsam
