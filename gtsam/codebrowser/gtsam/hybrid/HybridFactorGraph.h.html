27namespace gtsam {

29class DiscreteFactor;
30class Ordering;
31
32using SharedFactor = std::shared_ptr<Factor>;
33
34/**
35 * Hybrid Factor Graph
36 * Factor graph with utilities for hybrid factors.
37 */
38class GTSAM_EXPORT HybridFactorGraph : public FactorGraph<Factor> {
39 public:
40 using Base = FactorGraph<Factor>;
41 using This = HybridFactorGraph; ///< this class
42 using shared_ptr = std::shared_ptr<This>; ///< shared_ptr to This
43
44 using Values = gtsam::Values; ///< backwards compatibility
45 using Indices = KeyVector; ///> map from keys to values
46
47 public:
48 /// @name Constructors
49 /// @{
50
51 /// Default constructor
52 HybridFactorGraph() = default;
53
54 /**
55 * Implicit copy/downcast constructor to override explicit template container
56 * constructor. In BayesTree this is used for:
57 * `cachedSeparatorMarginal_.reset(*separatorMarginal)`
58 * */
59 template <class DERIVEDFACTOR>
60 HybridFactorGraph(const FactorGraph<DERIVEDFACTOR>& graph) : Base(graph) {}
61
62 /** Construct from container of factors (shared_ptr or plain objects) */
63 template <class CONTAINER>
64 explicit HybridFactorGraph(const CONTAINER& factors) : Base(factors) {}
65
66 /// @}
67 /// @name Extra methods to inspect discrete/continuous keys.
68 /// @{
69
70 /// Get all the discrete keys in the factor graph.
71 std::set<DiscreteKey> discreteKeys() const;
72
73 /// Get all the discrete keys in the factor graph, as a set of Keys.
74 KeySet discreteKeySet() const;
75
76 /// Get all the continuous keys in the factor graph.
77 const KeySet continuousKeySet() const;
78
79 /// @}
80};
81
82} // namespace gtsam
