41/* ************************************************************************* */
42HybridBayesNet HybridBayesNet::prune(
43 size_t maxNrLeaves, const std::optional<double> &marginalThreshold,
44 DiscreteValues *fixedValues) const {
45#if GTSAM_HYBRID_TIMING
46 gttic_(HybridPruning);
47#endif
48 // Collect all the discrete conditionals. Could be small if already pruned.
49 const DiscreteBayesNet marginal = discreteMarginal();
50
51 // We use a separate value here since we need this to perform `restrict` on
52 // the HybridConditionals later.
53 DiscreteValues fixed;
54 // Prune discrete Bayes net
55 DiscreteBayesNet prunedBN =
56 marginal.prune(maxNrLeaves, marginalThreshold, fixedValues: &fixed);
57
58 // Multiply into one big conditional.
59 // NOTE: This is cheap since marginal.prune above creates a
60 // DBN with a single joint conditional.
61 DiscreteConditional pruned = prunedBN.joint();
62
63 // Set the fixed values if requested.
64 if (marginalThreshold && fixedValues) {
65 if (!fixedValues->empty()) {
66 throw std::invalid_argument(
67 "HybridBayesNet::prune: fixedValues should be empty since it is "
68 "a purely output argument.");
69 }
70 *fixedValues = fixed;
71 }
72
73 HybridBayesNet result;
74 result.reserve(size: size());
75
76 // Go through all the Gaussian conditionals, restrict them according to
77 // fixed values, and then prune further.
78 for (std::shared_ptr<HybridConditional> conditional : *this) {
79 if (conditional->isDiscrete()) continue;
80
81 // No-op if not a HybridGaussianConditional.
82 if (marginalThreshold) {
83 conditional = std::static_pointer_cast<HybridConditional>(
84 r: conditional->restrict(assignment: fixed));
85 }
86
87 // Now decide on type what to do:
88 if (auto hgc = conditional->asHybrid()) {
89 // Prune the hybrid Gaussian conditional!
90 auto prunedHybridGaussianConditional = hgc->prune(discreteProbs: pruned);
91 if (!prunedHybridGaussianConditional) {
92 throw std::runtime_error(
93 "A HybridGaussianConditional had all its conditionals pruned");
94 }
95 // Type-erase and add to the pruned Bayes Net fragment.
96 result.push_back(conditional: prunedHybridGaussianConditional);
97 } else if (conditional->isContinuous()) {
98 // Add the non-Hybrid GaussianConditional conditional
99 result.push_back(conditional);
100 } else
101 throw std::runtime_error(
102 "HybrdiBayesNet::prune: Unknown HybridConditional type.");
103 }
104
105#if GTSAM_HYBRID_TIMING
106 gttoc_(HybridPruning);
107#endif
108
109 // Add the pruned discrete conditionals to the result.
110 for (const DiscreteConditional::shared_ptr &discrete : prunedBN)
111 result.push_back(conditional: discrete);
112
113 return result;
114}

116/* ************************************************************************* */
117DiscreteBayesNet HybridBayesNet::discreteMarginal() const {
118 DiscreteBayesNet result;
119 for (auto &&conditional : *this) {
120 if (auto dc = conditional->asDiscrete()) {
121 result.push_back(factor: dc);
122 }
123 }
124 return result;
125}

127/* ************************************************************************* */
128GaussianBayesNet HybridBayesNet::choose(
129 const DiscreteValues &assignment) const {
130 GaussianBayesNet gbn;
131 for (auto &&conditional : *this) {
132 if (auto gm = conditional->asHybrid()) {
133 // If conditional is hybrid, select based on assignment.
134 gbn.push_back(factor: gm->choose(discreteValues: assignment));
135 } else if (auto gc = conditional->asGaussian()) {
136 // If continuous only, add Gaussian conditional.
137 gbn.push_back(factor: gc);
138 } else if (auto dc = conditional->asDiscrete()) {
139 // If conditional is discrete-only, we simply continue.
140 continue;
141 }
142 }
143
144 return gbn;
145}

147/* ************************************************************************* */
148DiscreteValues HybridBayesNet::mpe() const {
149 // Collect all the discrete factors to compute MPE
150 DiscreteFactorGraph discrete_fg;
151
152 for (auto &&conditional : *this) {
153 if (conditional->isDiscrete()) {
154 if (auto dtc = conditional->asDiscrete<TableDistribution>()) {
155 // The number of keys should be small so should not
156 // be expensive to convert to DiscreteConditional.
157 discrete_fg.push_back(factor: DiscreteConditional(dtc->nrFrontals(),
158 dtc->toDecisionTreeFactor()));
159 } else {
160 discrete_fg.push_back(factor: conditional->asDiscrete());
161 }
162 }
163 }
164
165 return discrete_fg.optimize();
166}

168/* ************************************************************************* */
169HybridValues HybridBayesNet::optimize() const {
170 // Solve for the MPE
171 DiscreteValues mpe = this->mpe();
172
173 // Given the MPE, compute the optimal continuous values.
174 return HybridValues(optimize(assignment: mpe), mpe);
175}

177/* ************************************************************************* */
178VectorValues HybridBayesNet::optimize(const DiscreteValues &assignment) const {
179 GaussianBayesNet gbn = choose(assignment);
180
181 // Check if there exists a nullptr in the GaussianBayesNet
182 // If yes, return an empty VectorValues
183 if (std::find(first: gbn.begin(), last: gbn.end(), val: nullptr) != gbn.end()) {
184 return VectorValues();
185 }
186 return gbn.optimize();
187}

189/* ************************************************************************* */
190HybridValues HybridBayesNet::sample(const HybridValues &given,
191 std::mt19937_64 *rng) const {
192 DiscreteBayesNet dbn;
193 for (auto &&conditional : *this) {
194 if (conditional->isDiscrete()) {
195 // If conditional is discrete-only, we add to the discrete Bayes net.
196 dbn.push_back(factor: conditional->asDiscrete());
197 }
198 }
199 // Sample a discrete assignment.
200 const DiscreteValues assignment = dbn.sample(given: given.discrete(), rng);
201 // Select the continuous Bayes net corresponding to the assignment.
202 GaussianBayesNet gbn = choose(assignment);
203 // Sample from the Gaussian Bayes net.
204 VectorValues sample = gbn.sample(given: given.continuous(), rng);
205 return {sample, assignment};
206}

209HybridValues HybridBayesNet::sample(std::mt19937_64 *rng) const {
210 HybridValues given;
211 return sample(given, rng);

214/* ************************************************************************* */
215AlgebraicDecisionTree<Key> HybridBayesNet::errorTree(
216 const VectorValues &continuousValues) const {
217 AlgebraicDecisionTree<Key> result(0.0);
218
219 // Iterate over each conditional.
220 for (auto &&conditional : *this) {
221 result = result + conditional->errorTree(continuousValues);
222 }
223
224 return result;
225}

227/* ************************************************************************* */
228double HybridBayesNet::negLogConstant(
229 const std::optional<DiscreteValues> &discrete) const {
230 double negLogNormConst = 0.0;
231 // Iterate over each conditional.
232 for (auto &&conditional : *this) {
233 if (discrete.has_value()) {
234 if (auto gm = conditional->asHybrid()) {
235 negLogNormConst += gm->choose(discreteValues: *discrete)->negLogConstant();
236 } else if (auto gc = conditional->asGaussian()) {
237 negLogNormConst += gc->negLogConstant();
238 } else if (auto dc = conditional->asDiscrete()) {
239 negLogNormConst += dc->choose(given: *discrete)->negLogConstant();
240 } else {
241 throw std::runtime_error(
242 "Unknown conditional type when computing negLogConstant");
243 }
244 } else {
245 negLogNormConst += conditional->negLogConstant();
246 }
247 }
248 return negLogNormConst;
249}

251/* ************************************************************************* */
252AlgebraicDecisionTree<Key> HybridBayesNet::discretePosterior(
253 const VectorValues &continuousValues) const {
254 AlgebraicDecisionTree<Key> errors = this->errorTree(continuousValues);
255 AlgebraicDecisionTree<Key> p =
256 errors.apply(op: [](double error) { return exp(x: -error); });
257 return p / p.sum();
258}

260/* ************************************************************************* */
261double HybridBayesNet::evaluate(const HybridValues &values) const {
262 return exp(x: logProbability(x: values));
263}

265/* ************************************************************************* */
266HybridGaussianFactorGraph HybridBayesNet::toFactorGraph(
267 const VectorValues &measurements) const {
268 HybridGaussianFactorGraph fg;
269
270 // For all nodes in the Bayes net, if its frontal variable is in measurements,
271 // replace it by a likelihood factor:
272 for (auto &&conditional : *this) {
273 if (conditional->frontalsIn(measurements)) {
274 if (auto gc = conditional->asGaussian()) {
275 fg.push_back(factor: gc->likelihood(frontalValues: measurements));
276 } else if (auto gm = conditional->asHybrid()) {
277 fg.push_back(factor: gm->likelihood(given: measurements));
278 } else {
279 throw std::runtime_error("Unknown conditional type");
280 }
281 } else {
282 fg.push_back(factor: conditional);
283 }
284 }
285 return fg;
286}
