38/**
39 * @brief A clique in a HybridBayesTree
40 * which is a HybridConditional internally.
41 *
42 * @ingroup hybrid
43 */
44class GTSAM_EXPORT HybridBayesTreeClique
45 : public BayesTreeCliqueBase<HybridBayesTreeClique,
46 HybridGaussianFactorGraph> {
47 public:
48 typedef HybridBayesTreeClique This;
49 typedef BayesTreeCliqueBase<HybridBayesTreeClique, HybridGaussianFactorGraph>
50 Base;
51 typedef std::shared_ptr<This> shared_ptr;
52 typedef std::weak_ptr<This> weak_ptr;
53 HybridBayesTreeClique() {}
54 HybridBayesTreeClique(const std::shared_ptr<HybridConditional>& conditional)
55 : Base(conditional) {}
56 ///< Copy constructor
57 HybridBayesTreeClique(const HybridBayesTreeClique& clique) : Base(clique) {}
58};

60/* ************************************************************************* */
61/** A Bayes tree representing a Hybrid density */
62class GTSAM_EXPORT HybridBayesTree : public BayesTree<HybridBayesTreeClique> {
63 private:
64 typedef BayesTree<HybridBayesTreeClique> Base;
65
66 public:
67 typedef HybridBayesTree This;
68 typedef std::shared_ptr<This> shared_ptr;
69
70 /// @name Standard interface
71 /// @{
72 /** Default constructor, creates an empty Bayes tree */
73 HybridBayesTree() = default;
74
75 /** Check equality */
76 bool equals(const This& other, double tol = 1e-9) const;
77
78 /**
79 * @brief Get the Gaussian Bayes Tree which corresponds to a specific discrete
80 * value assignment.
81 *
82 * @param assignment The discrete value assignment for the discrete keys.
83 * @return GaussianBayesTree
84 */
85 GaussianBayesTree choose(const DiscreteValues& assignment) const;
86
87 /** Error for all conditionals. */
88 double error(const HybridValues& values) const;
89
90 /**
91 * @brief Optimize the hybrid Bayes tree by computing the MPE for the current
92 * set of discrete variables and using it to compute the best continuous
93 * update delta.
94 *
95 * @return HybridValues
96 */
97 HybridValues optimize() const;
98
99 /**
100 * @brief Recursively optimize the BayesTree to produce a vector solution.
101 *
102 * @param assignment The discrete values assignment to select
103 * the hybrid conditional.
104 * @return VectorValues
105 */
106 VectorValues optimize(const DiscreteValues& assignment) const;
107
108 /**
109 * @brief Compute the Most Probable Explanation (MPE)
110 * of the discrete variables.
111 *
112 * @return DiscreteValues
113 */
114 DiscreteValues mpe() const;
115
116 /**
117 * @brief Prune the underlying Bayes tree.
118 *
119 * @param maxNumberLeaves The max number of leaf nodes to keep.
120 */
121 void prune(const size_t maxNumberLeaves);
122
123 /// @}
124
125 private:
126 /// Helper method to compute the max product assignment
127 /// given a DiscreteFactorGraph
128 DiscreteValues discreteMaxProduct(const DiscreteFactorGraph& dfg) const;
129
130#if GTSAM_ENABLE_BOOST_SERIALIZATION
131 /** Serialization function */
132 friend class boost::serialization::access;
133 template <class ARCHIVE>
134 void serialize(ARCHIVE& ar, const unsigned int /*version*/) {
135 ar& BOOST_SERIALIZATION_BASE_OBJECT_NVP(Base);
136 }
137#endif
138};

148/**
149 * @brief Class for Hybrid Bayes tree orphan subtrees.
150 *
151 * This object stores parent keys in our base type factor so that
152 * eliminating those parent keys will pull this subtree into the
153 * elimination.
154 *
155 * This is a template instantiation for hybrid Bayes tree cliques, storing both
156 * the regular keys *and* discrete keys in the HybridConditional.
157 */
158template <>
159class BayesTreeOrphanWrapper<HybridBayesTreeClique> : public HybridConditional {
160 public:
161 typedef HybridBayesTreeClique CliqueType;
162 typedef HybridConditional Base;
163
164 std::shared_ptr<CliqueType> clique;
165
166 /**
167 * @brief Construct a new Bayes Tree Orphan Wrapper object.
168 *
169 * @param clique Bayes tree clique.
170 */
171 BayesTreeOrphanWrapper(const std::shared_ptr<CliqueType>& clique)
172 : clique(clique) {
173 // Store parent keys in our base type factor so that eliminating those
174 // parent keys will pull this subtree into the elimination.
175 this->keys_.assign(first: clique->conditional()->beginParents(),
176 last: clique->conditional()->endParents());
177 this->discreteKeys_.assign(first: clique->conditional()->discreteKeys().begin(),
178 last: clique->conditional()->discreteKeys().end());
179 }
180
181 /// print utility
182 void print(
183 const std::string& s = "",
184 const KeyFormatter& formatter = DefaultKeyFormatter) const override {
185 clique->print(s: s + " stored clique ", keyFormatter: formatter);
186 }
187};
