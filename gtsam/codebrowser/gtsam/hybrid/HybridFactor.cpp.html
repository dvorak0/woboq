23KeyVector CollectKeys(const KeyVector &continuousKeys,
24 const DiscreteKeys &discreteKeys) {
25 KeyVector allKeys;
26 allKeys.reserve(n: continuousKeys.size() + discreteKeys.size());
27 std::copy(first: continuousKeys.begin(), last: continuousKeys.end(),
28 result: std::back_inserter(x&: allKeys));
29 std::transform(first: discreteKeys.begin(), last: discreteKeys.end(),
30 result: std::back_inserter(x&: allKeys),
31 unary_op: [](const DiscreteKey &k) { return k.first; });
32 return allKeys;
33}

57/* ************************************************************************ */
58HybridFactor::Category GetCategory(const KeyVector &continuousKeys,
59 const DiscreteKeys &discreteKeys) {
60 if ((continuousKeys.size() == 0) && (discreteKeys.size() != 0)) {
61 return HybridFactor::Category::Discrete;
62 } else if ((continuousKeys.size() != 0) && (discreteKeys.size() == 0)) {
63 return HybridFactor::Category::Continuous;
64 } else if ((continuousKeys.size() != 0) && (discreteKeys.size() != 0)) {
65 return HybridFactor::Category::Hybrid;
66 } else {
67 // Case where we have no keys. Should never happen.
68 return HybridFactor::Category::None;
69 }
70}

95/* ************************************************************************ */
96void HybridFactor::print(const std::string &s,
97 const KeyFormatter &formatter) const {
98 std::cout << (s.empty() ? "" : s + "\n");
99 switch (category_) {
100 case Category::Continuous:
101 std::cout << "Continuous ";
102 break;
103 case Category::Discrete:
104 std::cout << "Discrete ";
105 break;
106 case Category::Hybrid:
107 std::cout << "Hybrid ";
108 break;
109 case Category::None:
110 std::cout << "None ";
111 break;
112 }
113
114 std::cout << "[";
115 for (size_t c = 0; c < continuousKeys_.size(); c++) {
116 std::cout << formatter(continuousKeys_.at(n: c));
117 if (c < continuousKeys_.size() - 1) {
118 std::cout << " ";
119 } else {
120 std::cout << (discreteKeys_.size() > 0 ? "; " : "");
121 }
122 }
123 for (size_t d = 0; d < discreteKeys_.size(); d++) {
124 std::cout << formatter(discreteKeys_.at(n: d).first);
125 if (d < discreteKeys_.size() - 1) {
126 std::cout << " ";
127 }
128 }
129 std::cout << "]";
130}
