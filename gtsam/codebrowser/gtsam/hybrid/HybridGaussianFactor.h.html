31namespace gtsam {
32
33class HybridValues;
34class DiscreteValues;
35class VectorValues;
36
37/// Alias for pair of GaussianFactor::shared_pointer and a double value.
38using GaussianFactorValuePair = std::pair<GaussianFactor::shared_ptr, double>;
39
40/**
41 * @brief Implementation of a discrete-conditioned hybrid factor.
42 * Implements a joint discrete-continuous factor where the discrete variable
43 * serves to "select" a component corresponding to a GaussianFactor.
44 *
45 * Represents the underlying hybrid Gaussian components as a Decision Tree,
46 * where the set of discrete variables indexes to
47 * the continuous gaussian distribution.
48 *
49 * In factor graphs the error function typically returns 0.5*|A*x - b|^2, i.e.,
50 * the negative log-likelihood for a Gaussian noise model.
51 * In hybrid factor graphs we allow *adding* an arbitrary scalar dependent on
52 * the discrete assignment.
53 * For example, adding a 70/30 mode probability is supported by providing the
54 * scalars $-log(.7)$ and $-log(.3)$.
55 * Note that adding a common constant will not make any difference in the
56 * optimization, so $-log(70)$ and $-log(30)$ work just as well.
57 *
58 * @ingroup hybrid
59 */
60class GTSAM_EXPORT HybridGaussianFactor : public HybridFactor {
61 public:
62 using Base = HybridFactor;
63 using This = HybridGaussianFactor;
64 using shared_ptr = std::shared_ptr<This>;
65
66 using sharedFactor = std::shared_ptr<GaussianFactor>;
67
68 /// typedef for Decision Tree of Gaussian factors and arbitrary value.
69 using FactorValuePairs = DecisionTree<Key, GaussianFactorValuePair>;
70
71 private:
72 /// Decision tree of Gaussian factors indexed by discrete keys.
73 FactorValuePairs factors_;
74
75 public:
76 /// @name Constructors
77 /// @{
78
79 /// Default constructor, mainly for serialization.
80 HybridGaussianFactor() = default;
81
82 /**
83 * @brief Construct a new HybridGaussianFactor on a single discrete key,
84 * providing the factors for each mode m as a vector of factors ϕ_m(x).
85 * The value ϕ(x,m) for the factor is simply ϕ_m(x).
86 *
87 * @param discreteKey The discrete key for the "mode", indexing components.
88 * @param factors Vector of gaussian factors, one for each mode.
89 */
90 HybridGaussianFactor(const DiscreteKey &discreteKey,
91 const std::vector<GaussianFactor::shared_ptr> &factors);
92
93 /**
94 * @brief Construct a new HybridGaussianFactor on a single discrete key,
95 * including a scalar error value for each mode m. The factors and scalars are
96 * provided as a vector of pairs (ϕ_m(x), E_m).
97 * The value ϕ(x,m) for the factor is now ϕ_m(x) + E_m.
98 *
99 * @param discreteKey The discrete key for the "mode", indexing components.
100 * @param factorPairs Vector of gaussian factor-scalar pairs, one per mode.
101 */
102 HybridGaussianFactor(const DiscreteKey &discreteKey,
103 const std::vector<GaussianFactorValuePair> &factorPairs);
104
105 /**
106 * @brief Construct a new HybridGaussianFactor on a several discrete keys M,
107 * including a scalar error value for each assignment m. The factors and
108 * scalars are provided as a DecisionTree<Key> of pairs (ϕ_M(x), E_M).
109 * The value ϕ(x,M) for the factor is again ϕ_m(x) + E_m.
110 *
111 * @param discreteKeys Discrete variables and their cardinalities.
112 * @param factorPairs The decision tree of Gaussian factor/scalar pairs.
113 */
114 HybridGaussianFactor(const DiscreteKeys &discreteKeys,
115 const FactorValuePairs &factorPairs);
116
117 /// @}
118 /// @name Testable
119 /// @{
120
121 bool equals(const HybridFactor &lf, double tol = 1e-9) const override;
122
123 void print(const std::string &s = "", const KeyFormatter &formatter =
124 DefaultKeyFormatter) const override;
125
126 /// @}
127 /// @name Standard API
128 /// @{
129
130 /// Get factor at a given discrete assignment.
131 GaussianFactorValuePair operator()(const DiscreteValues &assignment) const;
132
133 /**
134 * @brief Compute error of the HybridGaussianFactor as a tree.
135 *
136 * @param continuousValues The continuous VectorValues.
137 * @return AlgebraicDecisionTree<Key> A decision tree with the same keys
138 * as the factors involved, and leaf values as the error.
139 */
140 AlgebraicDecisionTree<Key> errorTree(
141 const VectorValues &continuousValues) const override;
142
143 /**
144 * @brief Compute the log-likelihood, including the log-normalizing constant.
145 * @return double
146 */
147 double error(const HybridValues &values) const override;
148
149 /// Getter for GaussianFactor decision tree
150 const FactorValuePairs &factors() const { return factors_; }
151
152 /**
153 * @brief Helper function to return factors and functional to create a
154 * DecisionTree of Gaussian Factor Graphs.
155 *
156 * @return HybridGaussianProductFactor
157 */
158 virtual HybridGaussianProductFactor asProductFactor() const;
159
160 /// Restrict the factor to the given discrete values.
161 std::shared_ptr<Factor> restrict(
162 const DiscreteValues &discreteValues) const override;
163
164 /// @}
165
166 private:
167 /**
168 * @brief Helper function to augment the [A|b] matrices in the factor
169 * components with the additional scalar values. This is done by storing the
170 * value in the `b` vector as an additional row.
171 *
172 * @param factors DecisionTree of GaussianFactors and arbitrary scalars.
173 * @return FactorValuePairs
174 */
175 static FactorValuePairs augment(const FactorValuePairs &factors);
176
177 /// Helper struct to assist private constructor below.
178 struct ConstructorHelper;
179
180 // Private constructor using ConstructorHelper above.
181 HybridGaussianFactor(const ConstructorHelper &helper);
182
183#if GTSAM_ENABLE_BOOST_SERIALIZATION
184 /** Serialization function */
185 friend class boost::serialization::access;
186 template <class ARCHIVE>
187 void serialize(ARCHIVE &ar, const unsigned int /*version*/) {
188 ar &BOOST_SERIALIZATION_BASE_OBJECT_NVP(Base);
189 ar &BOOST_SERIALIZATION_NVP(factors_);
190 }
191#endif
192};
193
194// traits
195template <>
196struct traits<HybridGaussianFactor> : public Testable<HybridGaussianFactor> {};
