30/**
31 * A hybrid Bayes net is a collection of HybridConditionals, which can have
32 * discrete conditionals, hybrid Gaussian conditionals,
33 * or pure Gaussian conditionals.
34 *
35 * @ingroup hybrid
36 */
37class GTSAM_EXPORT HybridBayesNet : public BayesNet<HybridConditional> {
38 public:
39 using Base = BayesNet<HybridConditional>;
40 using This = HybridBayesNet;
41 using ConditionalType = HybridConditional;
42 using shared_ptr = std::shared_ptr<HybridBayesNet>;
43 using sharedConditional = std::shared_ptr<ConditionalType>;
44
45 /// @name Standard Constructors
46 /// @{
47
48 /// Construct empty Bayes net.
49 HybridBayesNet() = default;
50
51 /// Constructor that takes an initializer list of shared pointers.
52 HybridBayesNet(
53 std::initializer_list<HybridConditional::shared_ptr> conditionals)
54 : Base(conditionals) {}
55
56 /// @}
57 /// @name Testable
58 /// @{
59
60 /// GTSAM-style printing
61 void print(const std::string &s = "", const KeyFormatter &formatter =
62 DefaultKeyFormatter) const override;
63
64 /// GTSAM-style equals
65 bool equals(const This &fg, double tol = 1e-9) const;
66
67 /// @}
68 /// @name Standard Interface
69 /// @{
70
71 /**
72 * @brief Add a hybrid conditional using a shared_ptr.
73 *
74 * This is the "native" push back, as this class stores hybrid conditionals.
75 */
76 void push_back(std::shared_ptr<HybridConditional> conditional) {
77 factors_.push_back(x: conditional);
78 }
79
80 /**
81 * Move a HybridConditional into a shared pointer and add.
82
83 * Example:
84 * HybridGaussianConditional conditional(...);
85 * hbn.push_back(conditional); // loses the original conditional
86 */
87 void push_back(HybridConditional &&conditional) {
88 factors_.push_back(
89 x: std::make_shared<HybridConditional>(args: std::move(conditional)));
90 }
91
92 /**
93 * @brief Add a conditional to the Bayes net.
94 * Implicitly convert to a HybridConditional.
95 *
96 * E.g.
97 * hbn.push_back(std::make_shared<DiscreteConditional>(m, "1/1"));
98 *
99 * @tparam CONDITIONAL Type of conditional. This is shared_ptr version.
100 * @param conditional The conditional as a shared pointer.
101 */
102 template <class CONDITIONAL>
103 void push_back(const std::shared_ptr<CONDITIONAL> &conditional) {
104 factors_.push_back(std::make_shared<HybridConditional>(conditional));
105 }
106
107 /**
108 * Preferred: Emplace a conditional directly using arguments.
109 *
110 * Examples:
111 * hbn.emplace_shared<HybridGaussianConditional>(...)));
112 * hbn.emplace_shared<GaussianConditional>(...)));
113 * hbn.emplace_shared<DiscreteConditional>(...)));
114 */
115 template <class CONDITIONAL, class... Args>
116 void emplace_shared(Args &&...args) {
117 auto cond = std::allocate_shared<CONDITIONAL>(
118 Eigen::aligned_allocator<CONDITIONAL>(), std::forward<Args>(args)...);
119 factors_.push_back(std::make_shared<HybridConditional>(std::move(cond)));
120 }
121
122 /**
123 * @brief Get the discrete Bayes Net P(M). As the hybrid Bayes net defines
124 * P(X,M) = P(X|M) P(M), this method returns the marginal distribution on the
125 * discrete variables.
126 *
127 * @return discrete marginal as a DiscreteBayesNet.
128 */
129 DiscreteBayesNet discreteMarginal() const;
130
131 /**
132 * @brief Get the Gaussian Bayes net P(X|M=m) corresponding to a specific
133 * assignment m for the discrete variables M. As the hybrid Bayes net defines
134 * P(X,M) = P(X|M) P(M), this method returns the **posterior** p(X|M=m).
135 *
136 * @param assignment The discrete value assignment for the discrete keys.
137 * @return Gaussian posterior P(X|M=m) as a GaussianBayesNet.
138 */
139 GaussianBayesNet choose(const DiscreteValues &assignment) const;
140
141 /// Evaluate hybrid probability density for given HybridValues.
142 double evaluate(const HybridValues &values) const;
143
144 /// Evaluate hybrid probability density for given HybridValues, sugar.
145 double operator()(const HybridValues &values) const {
146 return evaluate(values);
147 }
148
149 /**
150 * @brief Compute the Most Probable Explanation (MPE)
151 * of the discrete variables.
152 *
153 * @return DiscreteValues
154 */
155 DiscreteValues mpe() const;
156
157 /**
158 * @brief Solve the HybridBayesNet by first computing the MPE of all the
159 * discrete variables and then optimizing the continuous variables based on
160 * the MPE assignment.
161 *
162 * @return HybridValues
163 */
164 HybridValues optimize() const;
165
166 /**
167 * @brief Given the discrete assignment, return the optimized estimate for the
168 * selected Gaussian BayesNet.
169 *
170 * @param assignment An assignment of discrete values.
171 * @return Values
172 */
173 VectorValues optimize(const DiscreteValues &assignment) const;
174
175 /**
176 * @brief Sample from an incomplete BayesNet, given missing variables.
177 *
178 * Example:
179 * std::mt19937_64 rng(42);
180 * VectorValues given = ...;
181 * auto sample = bn.sample(given, &rng);
182 *
183 * @param given Values of missing variables.
184 * @param rng The optional pseudo-random number generator.
185 * @return HybridValues
186 */
187 HybridValues sample(const HybridValues &given,
188 std::mt19937_64 *rng = nullptr) const;
189
190 /**
191 * @brief Sample using ancestral sampling.
192 *
193 * Example:
194 * std::mt19937_64 rng(42);
195 * auto sample = bn.sample(&rng);
196 *
197 * @param rng The optional pseudo-random number generator.
198 * @return HybridValues
199 */
200 HybridValues sample(std::mt19937_64 *rng = nullptr) const;
201
202 /**
203 * @brief Prune the Bayes Net such that we have at most maxNrLeaves leaves.
204 *
205 * @param maxNrLeaves Continuous values at which to compute the error.
206 * @param marginalThreshold The threshold to check the mode marginals against.
207 * @param fixedValues The fixed values resulting from dead mode removal.
208 *
209 * @note If marginal greater than this threshold, the mode gets assigned that
210 * value and is considered "dead" for hybrid elimination. The mode can then be
211 * removed since it only has a single possible assignment.
212 *
213 * @return A pruned HybridBayesNet
214 */
215 HybridBayesNet prune(size_t maxNrLeaves,
216 const std::optional<double> &marginalThreshold = {},
217 DiscreteValues *fixedValues = nullptr) const;
218
219 /**
220 * @brief Error method using HybridValues which returns specific error for
221 * assignment.
222 */
223 using Base::error;
224
225 /**
226 * @brief Compute the negative log posterior log P'(M|x) of all assignments up
227 * to a constant, returning the result as an algebraic decision tree.
228 *
229 * @note The joint P(X,M) is p(X|M) P(M)
230 * Then the posterior on M given X=x is is P(M|x) = p(x|M) P(M) / p(x).
231 * Ideally we want log P(M|x) = log p(x|M) + log P(M) - log p(x), but
232 * unfortunately log p(x) is expensive, so we compute the log of the
233 * unnormalized posterior log P'(M|x) = log p(x|M) + log P(M)
234 *
235 * @param continuousValues Continuous values x at which to compute log P'(M|x)
236 * @return AlgebraicDecisionTree<Key>
237 */
238 AlgebraicDecisionTree<Key> errorTree(
239 const VectorValues &continuousValues) const;
240
241 using BayesNet::logProbability; // expose HybridValues version
242
243 /**
244 * @brief Get the negative log of the normalization constant
245 * corresponding to the joint density represented by this Bayes net.
246 * Optionally index by `discrete`.
247 *
248 * @param discrete Optional DiscreteValues
249 * @return double
250 */
251 double negLogConstant(const std::optional<DiscreteValues>& discrete = {}) const;
252
253 /**
254 * @brief Compute normalized posterior P(M|X=x) and return as a tree.
255 *
256 * @note Not a DiscreteConditional as the cardinalities of the DiscreteKeys,
257 * which we would need, are hard to recover.
258 *
259 * @param continuousValues Continuous values x to condition P(M|X=x) on.
260 * @return AlgebraicDecisionTree<Key>
261 */
262 AlgebraicDecisionTree<Key> discretePosterior(
263 const VectorValues &continuousValues) const;
264
265 /**
266 * Convert a hybrid Bayes net to a hybrid Gaussian factor graph by converting
267 * all conditionals with instantiated measurements into likelihood factors.
268 */
269 HybridGaussianFactorGraph toFactorGraph(
270 const VectorValues &measurements) const;
271 /// @}
272
273 private:
274#if GTSAM_ENABLE_BOOST_SERIALIZATION
275 /** Serialization function */
276 friend class boost::serialization::access;
277 template <class ARCHIVE>
278 void serialize(ARCHIVE &ar, const unsigned int /*version*/) {
279 ar &BOOST_SERIALIZATION_BASE_OBJECT_NVP(Base);
280 }
281#endif
282};
