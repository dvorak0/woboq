18#pragma once
19
20#include <gtsam/hybrid/HybridGaussianISAM.h>
21#include <gtsam/hybrid/HybridNonlinearFactorGraph.h>
22
23#include <optional>
24
25namespace gtsam {
26/**
27 * Wrapper class to manage ISAM in a nonlinear context
28 */
29class GTSAM_EXPORT HybridNonlinearISAM {
30 protected:
31 /** The internal iSAM object */
32 gtsam::HybridGaussianISAM isam_;
33
34 /** The current linearization point */
35 Values linPoint_;
36
37 /// The discrete assignment
38 DiscreteValues assignment_;
39
40 /** The original factors, used when re-linearizing */
41 HybridNonlinearFactorGraph factors_;
42
43 /** The reordering interval and counter */
44 int reorderInterval_;
45 int reorderCounter_;
46
47 /** The elimination function */
48 HybridGaussianFactorGraph::Eliminate eliminationFunction_;
49
50 public:
51 /// @name Standard Constructors
52 /// @{
53
54 /**
55 * Periodically reorder and re-linearize
56 * @param reorderInterval is the number of updates between re-orderings,
57 * 0 never reorders (and is dangerous for memory consumption)
58 * 1 (default) reorders every time, in worse case is batch every update
59 * typical values are 50 or 100
60 */
61 HybridNonlinearISAM(
62 int reorderInterval = 1,
63 const HybridGaussianFactorGraph::Eliminate& eliminationFunction =
64 HybridGaussianFactorGraph::EliminationTraitsType::DefaultEliminate)
65 : reorderInterval_(reorderInterval),
66 reorderCounter_(0),
67 eliminationFunction_(eliminationFunction) {}
68
69 /// @}
70 /// @name Standard Interface
71 /// @{
72
73 /** Return the current solution estimate */
74 Values estimate();
75
76 // /** find the marginal covariance for a single variable */
77 // Matrix marginalCovariance(Key key) const;
78
79 // access
80
81 /** access the underlying bayes tree */
82 const HybridGaussianISAM& bayesTree() const { return isam_; }
83
84 /**
85 * @brief Prune the underlying Bayes tree.
86 *
87 * @param maxNumberLeaves The max number of leaf nodes to keep.
88 */
89 void prune(const size_t maxNumberLeaves) { isam_.prune(maxNumberLeaves); }
90
91 /** Return the current linearization point */
92 const Values& getLinearizationPoint() const { return linPoint_; }
93
94 /** Return the current discrete assignment */
95 const DiscreteValues& assignment() const { return assignment_; }
96
97 /** get underlying nonlinear graph */
98 const HybridNonlinearFactorGraph& getFactorsUnsafe() const {
99 return factors_;
100 }
101
102 /** get counters */
103 int reorderInterval() const { return reorderInterval_; } ///< TODO: comment
104 int reorderCounter() const { return reorderCounter_; } ///< TODO: comment
105
106 /** prints out all contents of the system */
107 void print(const std::string& s = "",
108 const KeyFormatter& keyFormatter = DefaultKeyFormatter) const;
109
110 /** prints out clique statistics */
111 void printStats() const;
112
113 /** saves the Tree to a text file in GraphViz format */
114 void saveGraph(const std::string& s,
115 const KeyFormatter& keyFormatter = DefaultKeyFormatter) const;
116
117 /// @}
118 /// @name Advanced Interface
119 /// @{
120
121 /** Add new factors along with their initial linearization points */
122 void update(const HybridNonlinearFactorGraph& newFactors,
123 const Values& initialValues,
124 const std::optional<size_t>& maxNrLeaves = {},
125 const std::optional<Ordering>& ordering = {});
126
127 /** Re-linearization and reordering of variables */
128 void reorderRelinearize();
129
130 /// @}
131};
132
133} // namespace gtsam
