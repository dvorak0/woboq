19#pragma once
20
21#include <gtsam/hybrid/HybridFactorGraph.h>
22
23namespace gtsam {
24
25class HybridGaussianFactorGraph;
26
27/**
28 * Nonlinear Hybrid Factor Graph
29 * -----------------------
30 * This is the non-linear version of a hybrid factor graph.
31 * Everything inside needs to be hybrid factor or hybrid conditional.
32 */
33class GTSAM_EXPORT HybridNonlinearFactorGraph : public HybridFactorGraph {
34 protected:
35 public:
36 using Base = HybridFactorGraph;
37 using This = HybridNonlinearFactorGraph; ///< this class
38 using shared_ptr = std::shared_ptr<This>; ///< shared_ptr to This
39
40 using Values = gtsam::Values; ///< backwards compatibility
41 using Indices = KeyVector; ///> map from keys to values
42
43 /// @name Constructors
44 /// @{
45
46 HybridNonlinearFactorGraph() = default;
47
48 /**
49 * Implicit copy/downcast constructor to override explicit template container
50 * constructor. In BayesTree this is used for:
51 * `cachedSeparatorMarginal_.reset(*separatorMarginal)`
52 * */
53 template <class DERIVEDFACTOR>
54 HybridNonlinearFactorGraph(const FactorGraph<DERIVEDFACTOR>& graph)
55 : Base(graph) {}
56
57 /// @}
58 /// @name Constructors
59 /// @{
60
61 /// Print the factor graph.
62 void print(
63 const std::string& s = "HybridNonlinearFactorGraph",
64 const KeyFormatter& keyFormatter = DefaultKeyFormatter) const override;
65
66 /** print errors along with factors*/
67 void printErrors(
68 const HybridValues& values,
69 const std::string& str = "HybridNonlinearFactorGraph: ",
70 const KeyFormatter& keyFormatter = DefaultKeyFormatter,
71 const std::function<bool(const Factor* /*factor*/,
72 double /*whitenedError*/, size_t /*index*/)>&
73 printCondition =
74 [](const Factor*, double, size_t) { return true; }) const;
75
76 /// @}
77 /// @name Standard Interface
78 /// @{
79
80 /**
81 * @brief Linearize all the continuous factors in the
82 * HybridNonlinearFactorGraph.
83 *
84 * @param continuousValues: Dictionary of continuous values.
85 * @return HybridGaussianFactorGraph::shared_ptr
86 */
87 std::shared_ptr<HybridGaussianFactorGraph> linearize(
88 const Values& continuousValues) const;
89
90 /// Expose error(const HybridValues&) method.
91 using Base::error;
92
93 /**
94 * @brief Compute error of (hybrid) nonlinear factors and discrete factors
95 * over each discrete assignment, and return as a tree.
96 *
97 * Error \f$ e = \Vert f(x) - \mu \Vert_{\Sigma} \f$.
98 *
99 * @note: Gaussian and hybrid Gaussian factors are not considered!
100 *
101 * @param values Manifold values at which to compute the error.
102 * @return AlgebraicDecisionTree<Key>
103 */
104 AlgebraicDecisionTree<Key> errorTree(const Values& values) const;
105
106 /**
107 * @brief Computer posterior P(M|X=x) when all continuous values X are given.
108 * This is efficient as this simply takes -exp(.) of errorTree and normalizes.
109 *
110 * @note Not a DiscreteConditional as the cardinalities of the DiscreteKeys,
111 * which we would need, are hard to recover.
112 *
113 * @param continuousValues Continuous values x to condition on.
114 * @return DecisionTreeFactor
115 */
116 AlgebraicDecisionTree<Key> discretePosterior(
117 const Values& continuousValues) const;
118
119 /// Restrict all factors in the graph to the given discrete values.
120 HybridNonlinearFactorGraph restrict(
121 const DiscreteValues& assignment) const;
122
123 /// @}
124};
125
126template <>
127struct traits<HybridNonlinearFactorGraph>
128 : public Testable<HybridNonlinearFactorGraph> {};
129
130} // namespace gtsam
