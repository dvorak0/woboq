26/* ************************************************************************* */
27HybridValues::HybridValues(const VectorValues& cv, const DiscreteValues& dv)
28 : continuous_(cv), discrete_(dv) {}
29
30HybridValues::HybridValues(const VectorValues& cv, const DiscreteValues& dv,
31 const Values& v)
32 : continuous_(cv), discrete_(dv), nonlinear_(v) {}
33
34/* ************************************************************************* */
35HybridValues::HybridValues(const DiscreteValues& dv, const Values& v)
36 : discrete_(dv), nonlinear_(v) {}

38/* ************************************************************************* */
39void HybridValues::print(const std::string& s,
40 const KeyFormatter& keyFormatter) const {
41 std::cout << s << ": \n";
42 // print continuous components
43 continuous_.print(str: " Continuous", formatter: keyFormatter);
44 // print discrete components
45 discrete_.print(s: " Discrete", keyFormatter);
46 // print nonlinear components
47 nonlinear_.print(str: " Nonlinear", keyFormatter);
48}
49
50bool HybridValues::equals(const HybridValues& other, double tol) const {
51 return continuous_.equals(x: other.continuous_, tol) &&
52 discrete_.equals(x: other.discrete_, tol);
53}

55/* ************************************************************************* */
56const VectorValues& HybridValues::continuous() const { return continuous_; }
57
58const DiscreteValues& HybridValues::discrete() const { return discrete_; }
59
60const Values& HybridValues::nonlinear() const { return nonlinear_; }

62/* ************************************************************************* */
63bool HybridValues::existsVector(Key j) { return continuous_.exists(j); }
64
65bool HybridValues::existsDiscrete(Key j) {
66 return (discrete_.find(x: j) != discrete_.end());
67}
68
69bool HybridValues::existsNonlinear(Key j) { return nonlinear_.exists(j); }
70
71bool HybridValues::exists(Key j) {
72 return existsVector(j) || existsDiscrete(j) || existsNonlinear(j);
73}

75/* ************************************************************************* */
76HybridValues HybridValues::retract(const VectorValues& delta) const {
77 HybridValues updated(continuous_, discrete_, nonlinear_.retract(delta));
78 return updated;
79}

150/* ************************************************************************* */
151VectorValues HybridValues::continuousSubset(const KeyVector& keys) const {
152 VectorValues measurements;
153 for (const auto& key : keys) {
154 measurements.insert(j: key, value: continuous_.at(j: key));
155 }
156 return measurements;
157}

160std::string HybridValues::html(const KeyFormatter& keyFormatter) const {
161 std::stringstream ss;
162 ss << this->continuous_.html(keyFormatter);
163 ss << this->discrete_.html(keyFormatter);
164 return ss.str();
