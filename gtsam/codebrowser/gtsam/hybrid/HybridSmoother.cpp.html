37/* ************************************************************************* */
38Ordering HybridSmoother::getOrdering(const HybridGaussianFactorGraph &factors,
39 const KeySet &lastKeysToEliminate) {
40 // Get all the discrete keys from the factors
41 KeySet allDiscrete = factors.discreteKeySet();
42
43 // Create KeyVector with continuous keys followed by discrete keys.
44 KeyVector lastKeys;
45
46 // Insert continuous keys first.
47 for (auto &k : lastKeysToEliminate) {
48 if (!allDiscrete.exists(e: k)) {
49 lastKeys.push_back(x: k);
50 }
51 }
52
53 // Insert discrete keys at the end
54 std::copy(first: allDiscrete.begin(), last: allDiscrete.end(),
55 result: std::back_inserter(x&: lastKeys));
56
57 // Get an ordering where the new keys are eliminated last
58 Ordering ordering = Ordering::ColamdConstrainedLast(
59 graph: factors, constrainLast: KeyVector(lastKeys.begin(), lastKeys.end()), forceOrder: true);
60
61 return ordering;
62}

64/* ************************************************************************* */
65Ordering HybridSmoother::maybeComputeOrdering(
66 const HybridGaussianFactorGraph &updatedGraph,
67 const std::optional<Ordering> givenOrdering) {
68 Ordering ordering;
69 // If no ordering provided, then we compute one
70 if (!givenOrdering.has_value()) {
71 // Get the keys from the new factors
72 KeySet continuousKeysToInclude; // Scheme 1: empty, 15sec/2000, 64sec/3000
73 // (69s without TF)
74 // continuousKeysToInclude = newFactors.keys(); // Scheme 2: all,
75 // 8sec/2000, 160sec/3000 continuousKeysToInclude = updatedGraph.keys(); //
76 // Scheme 3: all, stopped after 80sec/2000
77
78 // Since updatedGraph now has all the connected conditionals,
79 // we can get the correct ordering.
80 ordering = this->getOrdering(factors: updatedGraph, lastKeysToEliminate: continuousKeysToInclude);
81 } else {
82 ordering = *givenOrdering;
83 }
84
85 return ordering;
86}

88/* ************************************************************************* */
89HybridGaussianFactorGraph HybridSmoother::removeFixedValues(
90 const HybridGaussianFactorGraph &graph,
91 const HybridGaussianFactorGraph &newFactors) {
92 // Initialize graph
93 HybridGaussianFactorGraph updatedGraph(graph);
94
95 for (DiscreteKey dkey : newFactors.discreteKeys()) {
96 Key key = dkey.first;
97 if (fixedValues_.find(x: key) != fixedValues_.end()) {
98 // Add corresponding discrete factor to reintroduce the information
99 std::vector<double> probabilities(
100 dkey.second, (1 - *marginalThreshold_) / dkey.second);
101 probabilities[fixedValues_[key]] = *marginalThreshold_;
102 DiscreteConditional dc(dkey, DiscreteKeys{}, probabilities);
103 updatedGraph.push_back(factor: dc);
104
105 // Remove fixed value
106 fixedValues_.erase(x: key);
107 }
108 }
109
110 return updatedGraph;
111}

113/* ************************************************************************* */
114void HybridSmoother::update(const HybridNonlinearFactorGraph &newFactors,
115 const Values &initial,
116 std::optional<size_t> maxNrLeaves,
117 const std::optional<Ordering> given_ordering) {
118 HybridGaussianFactorGraph linearizedFactors = *newFactors.linearize(continuousValues: initial);
119
120 // Record the new nonlinear factors and
121 // linearization point for relinearization
122 allFactors_.push_back(container: newFactors);
123 linearizationPoint_.insert_or_assign(values: initial);
124
125 const KeySet originalNewFactorKeys = newFactors.keys();
126#ifdef DEBUG_SMOOTHER
127 std::cout << "hybridBayesNet_ size before: " << hybridBayesNet_.size()
128 << std::endl;
129 std::cout << "newFactors size: " << newFactors.size() << std::endl;
130#endif
131 HybridGaussianFactorGraph updatedGraph;
132 // Add the necessary conditionals from the previous timestep(s).
133 std::tie(args&: updatedGraph, args&: hybridBayesNet_) =
134 addConditionals(graph: linearizedFactors, hybridBayesNet: hybridBayesNet_);
135#ifdef DEBUG_SMOOTHER
136 // print size of newFactors, updatedGraph, hybridBayesNet_
137 std::cout << "updatedGraph size: " << updatedGraph.size() << std::endl;
138 std::cout << "hybridBayesNet_ size after: " << hybridBayesNet_.size()
139 << std::endl;
140 std::cout << "total size: " << updatedGraph.size() + hybridBayesNet_.size()
141 << std::endl;
142#endif
143
144 if (marginalThreshold_) {
145 // Remove fixed values for discrete keys which are introduced in newFactors
146 updatedGraph = removeFixedValues(graph: updatedGraph, newFactors);
147 }
148
149 Ordering ordering = this->maybeComputeOrdering(updatedGraph, givenOrdering: given_ordering);
150
151#if GTSAM_HYBRID_TIMING
152 gttic_(HybridSmootherEliminate);
153#endif
154 // Eliminate.
155 HybridBayesNet bayesNetFragment = *updatedGraph.eliminateSequential(ordering);
156#if GTSAM_HYBRID_TIMING
157 gttoc_(HybridSmootherEliminate);
158#endif
159
160#ifdef DEBUG_SMOOTHER_DETAIL
161 for (auto conditional : bayesNetFragment) {
162 auto e = std::dynamic_pointer_cast<HybridConditional::BaseConditional>(
163 conditional);
164 GTSAM_PRINT(*e);
165 }
166#endif
167
168#ifdef DEBUG_SMOOTHER
169 // Print discrete keys in the bayesNetFragment:
170 std::cout << "Discrete keys in bayesNetFragment: ";
171 for (auto &key : HybridFactorGraph(bayesNetFragment).discreteKeySet()) {
172 std::cout << DefaultKeyFormatter(key) << " ";
173 }
174#endif
175
176 /// Prune
177 if (maxNrLeaves) {
178#if GTSAM_HYBRID_TIMING
179 gttic_(HybridSmootherPrune);
180#endif
181 // `pruneBayesNet` sets the leaves with 0 in discreteFactor to nullptr in
182 // all the conditionals with the same keys in bayesNetFragment.
183 DiscreteValues newlyFixedValues;
184 bayesNetFragment = bayesNetFragment.prune(maxNrLeaves: *maxNrLeaves, marginalThreshold: marginalThreshold_,
185 fixedValues: &newlyFixedValues);
186 // Insert or assign so it updates existing fixed values
187 // instead of replacing them.
188 fixedValues_.insert_or_assign(values: newlyFixedValues);
189#if GTSAM_HYBRID_TIMING
190 gttoc_(HybridSmootherPrune);
191#endif
192 }
193
194#ifdef DEBUG_SMOOTHER
195 // Print discrete keys in the bayesNetFragment:
196 std::cout << "\nAfter pruning: ";
197 for (auto &key : HybridFactorGraph(bayesNetFragment).discreteKeySet()) {
198 std::cout << DefaultKeyFormatter(key) << " ";
199 }
200 std::cout << std::endl << std::endl;
201#endif
202
203#ifdef DEBUG_SMOOTHER_DETAIL
204 for (auto conditional : bayesNetFragment) {
205 auto c = std::dynamic_pointer_cast<HybridConditional::BaseConditional>(
206 conditional);
207 GTSAM_PRINT(*c);
208 }
209#endif
210
211 // Add the partial bayes net to the posterior bayes net.
212 hybridBayesNet_.add(factorOrContainer: bayesNetFragment);
213}

215/* ************************************************************************* */
216std::pair<HybridGaussianFactorGraph, HybridBayesNet>
217HybridSmoother::addConditionals(const HybridGaussianFactorGraph &newFactors,
218 const HybridBayesNet &hybridBayesNet) const {
219 HybridGaussianFactorGraph graph(newFactors);
220 HybridBayesNet updatedHybridBayesNet(hybridBayesNet);
221
222 KeySet involvedKeys = newFactors.keys();
223 auto involved = [](const KeySet &involvedKeys, const Key &key) {
224 return involvedKeys.find(x: key) != involvedKeys.end();
225 };
226
227 // If hybridBayesNet is not empty,
228 // it means we have conditionals to add to the factor graph.
229 if (!hybridBayesNet.empty()) {
230 // We add all relevant hybrid conditionals on the last continuous variable
231 // in the previous `hybridBayesNet` to the graph
232
233 // New conditionals to add to the graph
234 HybridBayesNet newConditionals;
235
236 // NOTE(Varun) Using a for-range loop doesn't work since some of the
237 // conditionals are invalid pointers
238
239 // First get all the keys involved.
240 // We do this by iterating over all conditionals, and checking if their
241 // frontals are involved in the factor graph. If yes, then also make the
242 // parent keys involved in the factor graph.
243 for (size_t i = 0; i < hybridBayesNet.size(); i++) {
244 auto conditional = hybridBayesNet.at(i);
245
246 for (auto &key : conditional->frontals()) {
247 if (involved(involvedKeys, key)) {
248 // Add the conditional parents to involvedKeys
249 // so we add those conditionals too.
250 for (auto &&parentKey : conditional->parents()) {
251 involvedKeys.insert(x: parentKey);
252 }
253 // Break so we don't add parents twice.
254 break;
255 }
256 }
257 }
258#ifdef DEBUG_SMOOTHER
259 PrintKeySet(involvedKeys);
260#endif
261
262 for (size_t i = 0; i < hybridBayesNet.size(); i++) {
263 auto conditional = hybridBayesNet.at(i);
264
265 for (auto &key : conditional->frontals()) {
266 if (involved(involvedKeys, key)) {
267 newConditionals.push_back(conditional);
268
269 // Remove the conditional from the updated Bayes net
270 auto it = find(first: updatedHybridBayesNet.begin(),
271 last: updatedHybridBayesNet.end(), val: conditional);
272 updatedHybridBayesNet.erase(item: it);
273
274 break;
275 }
276 }
277 }
278
279 graph.push_back(container: newConditionals);
280 }
281
282 return {graph, updatedHybridBayesNet};
283}

296/* ************************************************************************* */
297HybridValues HybridSmoother::optimize() const {
298 // Solve for the MPE
299 DiscreteValues mpe = hybridBayesNet_.mpe();
300
301 // Add fixed values to the MPE.
302 mpe.insert(values: fixedValues_);
303
304 // Given the MPE, compute the optimal continuous values.
305 GaussianBayesNet gbn = hybridBayesNet_.choose(assignment: mpe);
306 const VectorValues continuous = gbn.optimize();
307 if (std::find(first: gbn.begin(), last: gbn.end(), val: nullptr) != gbn.end()) {
308 throw std::runtime_error("At least one nullptr factor in hybridBayesNet_");
309 }
310 return HybridValues(continuous, mpe);
311}

313/* ************************************************************************* */
314void HybridSmoother::relinearize(const std::optional<Ordering> givenOrdering) {
315 allFactors_ = allFactors_.restrict(assignment: fixedValues_);
316 HybridGaussianFactorGraph::shared_ptr linearized =
317 allFactors_.linearize(continuousValues: linearizationPoint_);
318
319 // Compute ordering if not given
320 Ordering ordering = this->maybeComputeOrdering(updatedGraph: *linearized, givenOrdering);
321
322 HybridBayesNet::shared_ptr bayesNet =
323 linearized->eliminateSequential(ordering);
324 HybridValues delta = bayesNet->optimize();
325 linearizationPoint_ = linearizationPoint_.retract(delta: delta.continuous());
326
327 reInitialize(hybridBayesNet&: *bayesNet);
328}
