20#include <gtsam/hybrid/HybridBayesNet.h>
21#include <gtsam/hybrid/HybridGaussianFactorGraph.h>
22#include <gtsam/inference/EliminationTree.h>

24namespace gtsam {
25
26/**
27 * Elimination Tree type for Hybrid Factor Graphs.
28 *
29 * @ingroup hybrid
30 */
31class GTSAM_EXPORT HybridEliminationTree
32 : public EliminationTree<HybridBayesNet, HybridGaussianFactorGraph> {
33 private:
34 friend class ::EliminationTreeTester;
35
36 public:
37 typedef EliminationTree<HybridBayesNet, HybridGaussianFactorGraph>
38 Base; ///< Base class
39 typedef HybridEliminationTree This; ///< This class
40 typedef std::shared_ptr<This> shared_ptr; ///< Shared pointer to this class
41
42 /// @name Constructors
43 /// @{
44
45 /**
46 * Build the elimination tree of a factor graph using pre-computed column
47 * structure.
48 * @param factorGraph The factor graph for which to build the elimination tree
49 * @param structure The set of factors involving each variable. If this is
50 * not precomputed, you can call the Create(const FactorGraph<DERIVEDFACTOR>&)
51 * named constructor instead.
52 * @return The elimination tree
53 */
54 HybridEliminationTree(const HybridGaussianFactorGraph& factorGraph,
55 const VariableIndex& structure, const Ordering& order);
56
57 /** Build the elimination tree of a factor graph. Note that this has to
58 * compute the column structure as a VariableIndex, so if you already have
59 * this precomputed, use the other constructor instead.
60 * @param factorGraph The factor graph for which to build the elimination tree
61 */
62 HybridEliminationTree(const HybridGaussianFactorGraph& factorGraph,
63 const Ordering& order);
64
65 /// @}
66
67 /** Test whether the tree is equal to another */
68 bool equals(const This& other, double tol = 1e-9) const;
69};
70
71} // namespace gtsam
