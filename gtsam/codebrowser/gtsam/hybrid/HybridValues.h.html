19#pragma once

21#include <gtsam/discrete/DiscreteValues.h>
22#include <gtsam/inference/Key.h>
23#include <gtsam/linear/VectorValues.h>
24#include <gtsam/nonlinear/Values.h>
25
26#include <map>
27#include <string>
28#include <vector>
29
30namespace gtsam {
31
32/**
33 * HybridValues represents a collection of DiscreteValues and VectorValues.
34 * It is typically used to store the variables of a HybridGaussianFactorGraph.
35 * Optimizing a HybridGaussianBayesNet returns this class.
36 */
37class GTSAM_EXPORT HybridValues {
38 private:
39 /// Continuous multi-dimensional vectors for \class GaussianFactor.
40 VectorValues continuous_;
41
42 /// Discrete values for \class DiscreteFactor.
43 DiscreteValues discrete_;
44
45 /// Continuous, differentiable manifold values for \class NonlinearFactor.
46 Values nonlinear_;
47
48 public:
49 /// @name Standard Constructors
50 /// @{
51
52 /// Default constructor creates an empty HybridValues.
53 HybridValues() = default;
54
55 /// Construct from DiscreteValues and VectorValues.
56 HybridValues(const VectorValues& cv, const DiscreteValues& dv);
57
58 /// Construct from all values types.
59 HybridValues(const VectorValues& cv, const DiscreteValues& dv,
60 const Values& v);
61
62 /// Construct from DiscreteValues and Values.
63 HybridValues(const DiscreteValues& dv, const Values& v);
64
65 /// @}
66 /// @name Testable
67 /// @{
68
69 /// print required by Testable for unit testing
70 void print(const std::string& s = "HybridValues",
71 const KeyFormatter& keyFormatter = DefaultKeyFormatter) const;
72
73 /// equals required by Testable for unit testing
74 bool equals(const HybridValues& other, double tol = 1e-9) const;
75
76 /// @}
77 /// @name Interface
78 /// @{
79
80 /// Return the multi-dimensional vector values.
81 const VectorValues& continuous() const;
82
83 /// Return the discrete values.
84 const DiscreteValues& discrete() const;
85
86 /// Return the nonlinear values.
87 const Values& nonlinear() const;
88
89 /// Check whether a variable with key \c j exists in VectorValues.
90 bool existsVector(Key j);
91
92 /// Check whether a variable with key \c j exists in DiscreteValues.
93 bool existsDiscrete(Key j);
94
95 /// Check whether a variable with key \c j exists in values.
96 bool existsNonlinear(Key j);
97
98 /// Check whether a variable with key \c j exists.
99 bool exists(Key j);
100
101 /**
102 * Add a delta config to current config and returns a new config.
103 * @param delta The delta to be added.
104 * @note This function applies the delta to the nonlinear values, while keeping
105 * the continuous and discrete values unchanged.
106 */
107 HybridValues retract(const VectorValues& delta) const;
108
109 /** Insert a vector \c value with key \c j. Throws an invalid_argument
110 * exception if the key \c j is already used.
111 * @param value The vector to be inserted.
112 * @param j The index with which the value will be associated. */
113 HybridValues& insert(Key j, const Vector& value);
114
115 /** Insert a discrete \c value with key \c j. Replaces the existing value if
116 * the key \c j is already used.
117 * @param value The vector to be inserted.
118 * @param j The index with which the value will be associated. */
119 HybridValues& insert(Key j, size_t value);
120
121 /**
122 * Insert a nonlinear value with key \c j. Throws an invalid_argument
123 * exception if the key \c j is already used.
124 * @tparam T The type of value to insert.
125 * @param j The index with which the value will be associated.
126 * @param value The value to be inserted.
127 */
128 template <typename T>
129 HybridValues& insertNonlinear(Key j, const T& value) {
130 nonlinear_.insert<T>(j, value);
131 return *this;
132 }
133
134 /// insert_or_assign() , similar to Values.h
135 void insert_or_assign(Key j, const Vector& value);
136
137 /// insert_or_assign() , similar to Values.h
138 void insert_or_assign(Key j, size_t value);
139
140 /** Insert all continuous values from \c values. Throws an invalid_argument
141 * exception if any keys to be inserted are already used. */
142 HybridValues& insert(const VectorValues& values);
143
144 /** Insert all discrete values from \c values. Throws an invalid_argument
145 * exception if any keys to be inserted are already used. */
146 HybridValues& insert(const DiscreteValues& values);
147
148 /** Insert all values from \c values. Throws an invalid_argument
149 * exception if any keys to be inserted are already used. */
150 HybridValues& insert(const Values& values);
151
152 /** Insert all values from \c values. Throws an invalid_argument exception if
153 * any keys to be inserted are already used. */
154 HybridValues& insert(const HybridValues& values);
155
156 /**
157 * Read/write access to the vector value with key \c j, throws
158 * std::out_of_range if \c j does not exist.
159 */
160 Vector& at(Key j);
161
162 /**
163 * Read/write access to the discrete value with key \c j, throws
164 * std::out_of_range if \c j does not exist.
165 */
166 size_t& atDiscrete(Key j);
167
168 /** For all key/value pairs in \c values, replace continuous values with
169 * corresponding keys in this object with those in \c values. Throws
170 * std::out_of_range if any keys in \c values are not present in this object.
171 */
172 HybridValues& update(const VectorValues& values);
173
174 /** For all key/value pairs in \c values, replace discrete values with
175 * corresponding keys in this object with those in \c values. Throws
176 * std::out_of_range if any keys in \c values are not present in this object.
177 */
178 HybridValues& update(const DiscreteValues& values);
179
180 /** For all key/value pairs in \c values, replace nonlinear values with
181 * corresponding keys in this object with those in \c values. Throws
182 * std::out_of_range if any keys in \c values are not present in this object.
183 */
184 HybridValues& update(const Values& values);
185
186 /** For all key/value pairs in \c values, replace all values with
187 * corresponding keys in this object with those in \c values. Throws
188 * std::out_of_range if any keys in \c values are not present in this object.
189 */
190 HybridValues& update(const HybridValues& values);
191
192 /// Extract continuous values with given keys.
193 VectorValues continuousSubset(const KeyVector& keys) const;
194
195 /// @}
196 /// @name Wrapper support
197 /// @{
198
199 /**
200 * @brief Output as a html table.
201 *
202 * @param keyFormatter function that formats keys.
203 * @return string html output.
204 */
205 std::string html(
206 const KeyFormatter& keyFormatter = DefaultKeyFormatter) const;
207
208 /// @}
209};
210
211// traits
212template <>
213struct traits<HybridValues> : public Testable<HybridValues> {};

215} // namespace gtsam
