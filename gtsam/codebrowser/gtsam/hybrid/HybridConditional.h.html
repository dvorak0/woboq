37/**
38 * Hybrid Conditional Density
39 *
40 * As a type-erased variant of:
41 * - DiscreteConditional
42 * - GaussianConditional
43 * - HybridGaussianConditional
44 *
45 * The reason why this is important is that `Conditional<T>` is a CRTP class.
46 * CRTP is static polymorphism such that all CRTP classes, while bearing the
47 * same name, are different classes not sharing a vtable. This prevents them
48 * from being contained in any container, and thus it is impossible to
49 * dynamically cast between them. A better option, as illustrated here, is
50 * treating them as an implementation detail - such that the hybrid mechanism
51 * does not know what is inside the HybridConditional. This prevents us from
52 * having diamond inheritances, and neutralized the need to change other
53 * components of GTSAM to make hybrid elimination work.
54 *
55 * A great reference to the type-erasure pattern is Eduardo Madrid's CppCon
56 * talk (https://www.youtube.com/watch?v=s082Qmd_nHs).
57 *
58 * @ingroup hybrid
59 */
60class GTSAM_EXPORT HybridConditional
61 : public HybridFactor,
62 public Conditional<HybridFactor, HybridConditional> {
63 public:
64 // typedefs needed to play nice with gtsam
65 typedef HybridConditional This; ///< Typedef to this class
66 typedef std::shared_ptr<This> shared_ptr; ///< shared_ptr to this class
67 typedef HybridFactor BaseFactor; ///< Typedef to our factor base class
68 typedef Conditional<BaseFactor, This>
69 BaseConditional; ///< Typedef to our conditional base class
70
71 protected:
72 /// Type-erased pointer to the inner type
73 std::shared_ptr<Factor> inner_;
74
75 public:
76 /// @name Standard Constructors
77 /// @{
78
79 /// Default constructor needed for serialization.
80 HybridConditional() = default;
81
82 /**
83 * @brief Construct a new Hybrid Conditional object
84 *
85 * @param continuousKeys Vector of keys for continuous variables.
86 * @param discreteKeys Keys and cardinalities for discrete variables.
87 * @param nFrontals The number of frontal variables in the conditional.
88 */
89 HybridConditional(const KeyVector& continuousKeys,
90 const DiscreteKeys& discreteKeys, size_t nFrontals)
91 : BaseFactor(continuousKeys, discreteKeys), BaseConditional(nFrontals) {}
92
93 /**
94 * @brief Construct a new Hybrid Conditional object
95 *
96 * @param continuousFrontals Vector of keys for continuous variables.
97 * @param discreteFrontals Keys and cardinalities for discrete variables.
98 * @param continuousParents Vector of keys for parent continuous variables.
99 * @param discreteParents Keys and cardinalities for parent discrete
100 * variables.
101 */
102 HybridConditional(const KeyVector& continuousFrontals,
103 const DiscreteKeys& discreteFrontals,
104 const KeyVector& continuousParents,
105 const DiscreteKeys& discreteParents);
106
107 /**
108 * @brief Construct a new Hybrid Conditional object
109 *
110 * @param continuousConditional Conditional used to create the
111 * HybridConditional.
112 */
113 HybridConditional(
114 const std::shared_ptr<GaussianConditional>& continuousConditional);
115
116 /**
117 * @brief Construct a new Hybrid Conditional object
118 *
119 * @param discreteConditional Conditional used to create the
120 * HybridConditional.
121 */
122 HybridConditional(
123 const std::shared_ptr<DiscreteConditional>& discreteConditional);
124
125 /**
126 * @brief Construct a new Hybrid Conditional object
127 *
128 * @param hybridGaussianCond Hybrid Gaussian Conditional used to create the
129 * HybridConditional.
130 */
131 HybridConditional(
132 const std::shared_ptr<HybridGaussianConditional>& hybridGaussianCond);
133
134 /// @}
135 /// @name Testable
136 /// @{
137
138 /// GTSAM-style print
139 void print(
140 const std::string& s = "Hybrid Conditional: ",
141 const KeyFormatter& formatter = DefaultKeyFormatter) const override;
142
143 /// GTSAM-style equals
144 bool equals(const HybridFactor& other, double tol = 1e-9) const override;
145
146 /// @}
147 /// @name Standard Interface
148 /// @{
149
150 /**
151 * @brief Return HybridConditional as a HybridGaussianConditional
152 * @return nullptr if not a conditional
153 * @return HybridGaussianConditional::shared_ptr otherwise
154 */
155 HybridGaussianConditional::shared_ptr asHybrid() const {
156 if (!isHybrid()) return nullptr;
157 return std::static_pointer_cast<HybridGaussianConditional>(r: inner_);
158 }
159
160 /**
161 * @brief Return HybridConditional as a GaussianConditional
162 * @return nullptr if not a GaussianConditional
163 * @return GaussianConditional::shared_ptr otherwise
164 */
165 GaussianConditional::shared_ptr asGaussian() const {
166 if (!isContinuous()) return nullptr;
167 return std::static_pointer_cast<GaussianConditional>(r: inner_);
168 }
169
170 /**
171 * @brief Return conditional as a DiscreteConditional or specified type T.
172 * @return nullptr if not a DiscreteConditional
173 * @return DiscreteConditional::shared_ptr
174 */
175 template <typename T = DiscreteConditional>
176 typename T::shared_ptr asDiscrete() const {
177 if (!isDiscrete()) return nullptr;
178 return std::static_pointer_cast<T>(inner_);
179 }
180
181 /// Get the type-erased pointer to the inner type
182 std::shared_ptr<Factor> inner() const { return inner_; }
183
184 /// Return the error of the underlying conditional.
185 double error(const HybridValues& hybridValues) const override;
186
187 /**
188 * @brief Compute error of the HybridConditional as a tree.
189 *
190 * @param continuousValues The continuous VectorValues.
191 * @return AlgebraicDecisionTree<Key> A decision tree with the same keys
192 * as the conditionals involved, and leaf values as the error.
193 */
194 AlgebraicDecisionTree<Key> errorTree(
195 const VectorValues& continuousValues) const override;
196
197 /// Return the log-probability (or density) of the underlying conditional.
198 double logProbability(const HybridValues& values) const override;
199
200 /**
201 * Return the negative log of the normalization constant.
202 * This shows up in the error as -(error(x) + negLogConstant)
203 * Note this is 0.0 for discrete and hybrid conditionals, but depends
204 * on the continuous parameters for Gaussian conditionals.
205 */
206 double negLogConstant() const override;
207
208 /// Return the probability (or density) of the underlying conditional.
209 double evaluate(const HybridValues& values) const override;
210
211 /// Check if VectorValues `measurements` contains all frontal keys.
212 bool frontalsIn(const VectorValues& measurements) const {
213 for (Key key : frontals()) {
214 if (!measurements.exists(j: key)) {
215 return false;
216 }
217 }
218 return true;
219 }
220
221 /**
222 * Return a HybridConditional by choosing branches based on the given discrete
223 * values. If all discrete parents are specified, return a HybridConditional
224 * which is just a GaussianConditional. If this conditional is *not* a hybrid
225 * conditional, just return that.
226 */
227 std::shared_ptr<Factor> restrict(
228 const DiscreteValues& assignment) const override;
229
230 /// @}
231
232 private:
233#if GTSAM_ENABLE_BOOST_SERIALIZATION
234 /** Serialization function */
235 friend class boost::serialization::access;
236 template <class Archive>
237 void serialize(Archive& ar, const unsigned int /*version*/) {
238 ar& BOOST_SERIALIZATION_BASE_OBJECT_NVP(BaseFactor);
239 ar& BOOST_SERIALIZATION_BASE_OBJECT_NVP(BaseConditional);
240 ar& BOOST_SERIALIZATION_NVP(inner_);
241
242 // register the various casts based on the type of inner_
243 // https://www.boost.org/doc/libs/1_80_0/libs/serialization/doc/serialization.html#runtimecasting
244 if (isDiscrete()) {
245 boost::serialization::void_cast_register<DiscreteConditional, Factor>(
246 dnull: static_cast<DiscreteConditional*>(NULL), bnull: static_cast<Factor*>(NULL));
247 } else if (isContinuous()) {
248 boost::serialization::void_cast_register<GaussianConditional, Factor>(
249 dnull: static_cast<GaussianConditional*>(NULL), bnull: static_cast<Factor*>(NULL));
250 } else {
251 boost::serialization::void_cast_register<HybridGaussianConditional,
252 Factor>(
253 dnull: static_cast<HybridGaussianConditional*>(NULL),
254 bnull: static_cast<Factor*>(NULL));
255 }
256 }
257#endif
258
259}; // HybridConditional
