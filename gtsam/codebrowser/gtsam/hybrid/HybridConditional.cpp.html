61/* ************************************************************************ */
62void HybridConditional::print(const std::string &s,
63 const KeyFormatter &formatter) const {
64 std::cout << s;
65
66 if (inner_) {
67 inner_->print(s: "", formatter);
68 } else {
69 if (isContinuous()) std::cout << "Continuous ";
70 if (isDiscrete()) std::cout << "Discrete ";
71 if (isHybrid()) std::cout << "Hybrid ";
72 BaseConditional::print(s: "", formatter);
73
74 std::cout << "P(";
75 size_t index = 0;
76 const size_t N = keys().size();
77 const size_t contN = N - discreteKeys_.size();
78 while (index < N) {
79 if (index > 0) {
80 if (index == nrFrontals_)
81 std::cout << " | ";
82 else
83 std::cout << ", ";
84 }
85 if (index < contN) {
86 std::cout << formatter(keys()[index]);
87 } else {
88 auto &dk = discreteKeys_[index - contN];
89 std::cout << "(" << formatter(dk.first) << ", " << dk.second << ")";
90 }
91 index++;
92 }
93 }
94}

96/* ************************************************************************ */
97bool HybridConditional::equals(const HybridFactor &other, double tol) const {
98 const This *e = dynamic_cast<const This *>(&other);
99 if (e == nullptr) return false;
100 if (auto gm = asHybrid()) {
101 auto other = e->asHybrid();
102 return other != nullptr && gm->equals(lf: *other, tol);
103 } else if (auto gc = asGaussian()) {
104 auto other = e->asGaussian();
105 return other != nullptr && gc->equals(cg: *other, tol);
106 } else if (auto dc = asDiscrete()) {
107 auto other = e->asDiscrete();
108 return other != nullptr && dc->equals(other: *other, tol);
109 } else
110 return inner_ ? (e->inner_ ? inner_->equals(other: *(e->inner_), tol) : false)
111 : !(e->inner_);
112}

115double HybridConditional::error(const HybridValues &hybridValues) const {
116 if (auto gc = asGaussian()) {
117 return gc->error(c: hybridValues.continuous());
118 } else if (auto gm = asHybrid()) {
119 return gm->error(values: hybridValues);
120 } else if (auto dc = asDiscrete()) {
121 return dc->error(values: hybridValues.discrete());
122 } else
123 throw std::runtime_error(
124 "HybridConditional::error: conditional type not handled");
125}

172/* ************************************************************************ */
173std::shared_ptr<Factor> HybridConditional::restrict(
174 const DiscreteValues &assignment) const {
175 if (auto gc = asGaussian()) {
176 return std::make_shared<HybridConditional>(args&: gc);
177 } else if (auto dc = asDiscrete()) {
178 return std::make_shared<HybridConditional>(args&: dc);
179 };
180
181 auto hgc = asHybrid();
182 if (!hgc)
183 throw std::runtime_error(
184 "HybridConditional::restrict: conditional type not handled");
185
186 // Case 1: Fully determined, return corresponding Gaussian conditional
187 auto parentValues = assignment.filter(keys: discreteKeys_);
188 if (parentValues.size() == discreteKeys_.size()) {
189 return std::make_shared<HybridConditional>(args: hgc->choose(discreteValues: parentValues));
190 }
191
192 // Case 2: Some live parents remain, build a new tree
193 auto remainingKeys = assignment.missingKeys(keys: discreteKeys_);
194 if (!remainingKeys.empty()) {
195 auto newTree = hgc->factors();
196 for (const auto &[key, value] : parentValues) {
197 newTree = newTree.choose(label: key, index: value);
198 }
199 return std::make_shared<HybridConditional>(
200 args: std::make_shared<HybridGaussianConditional>(args&: remainingKeys, args&: newTree));
201 }
202
203 // Case 3: No changes needed, return original
204 return std::make_shared<HybridConditional>(args&: hgc);
205}
