29void HybridNonlinearFactorGraph::print(const std::string& s,
30 const KeyFormatter& keyFormatter) const {
31 // Base::print(str, keyFormatter);
32 std::cout << (s.empty() ? "" : s + " ") << std::endl;
33 std::cout << "size: " << size() << std::endl;
34 for (size_t i = 0; i < factors_.size(); i++) {
35 std::stringstream ss;
36 ss << "factor " << i << ": ";
37 if (factors_[i]) {
38 factors_[i]->print(s: ss.str(), formatter: keyFormatter);
39 std::cout << std::endl;
40 }
41 }
42}

44/* ************************************************************************* */
45void HybridNonlinearFactorGraph::printErrors(
46 const HybridValues& values, const std::string& str,
47 const KeyFormatter& keyFormatter,
48 const std::function<bool(const Factor* /*factor*/, double /*whitenedError*/,
49 size_t /*index*/)>& printCondition) const {
50 std::cout << str << "size: " << size() << std::endl << std::endl;
51
52 std::stringstream ss;
53
54 for (size_t i = 0; i < factors_.size(); i++) {
55 auto&& factor = factors_[i];
56 std::cout << "Factor " << i << ": ";
57
58 // Clear the stringstream
59 ss.str(s: std::string());
60
61 if (auto mf = std::dynamic_pointer_cast<HybridNonlinearFactor>(r: factor)) {
62 if (factor == nullptr) {
63 std::cout << "nullptr"
64 << "\n";
65 } else {
66 factor->print(s: ss.str(), formatter: keyFormatter);
67 std::cout << "error = ";
68 mf->errorTree(continuousValues: values.nonlinear()).print(s: "", labelFormatter: keyFormatter);
69 std::cout << std::endl;
70 }
71 } else if (auto gmf =
72 std::dynamic_pointer_cast<HybridGaussianFactor>(r: factor)) {
73 if (factor == nullptr) {
74 std::cout << "nullptr"
75 << "\n";
76 } else {
77 factor->print(s: ss.str(), formatter: keyFormatter);
78 std::cout << "error = ";
79 gmf->errorTree(continuousValues: values.continuous()).print(s: "", labelFormatter: keyFormatter);
80 std::cout << std::endl;
81 }
82 } else if (auto gm = std::dynamic_pointer_cast<HybridGaussianConditional>(
83 r: factor)) {
84 if (factor == nullptr) {
85 std::cout << "nullptr"
86 << "\n";
87 } else {
88 factor->print(s: ss.str(), formatter: keyFormatter);
89 std::cout << "error = ";
90 gm->errorTree(continuousValues: values.continuous()).print(s: "", labelFormatter: keyFormatter);
91 std::cout << std::endl;
92 }
93 } else if (auto nf = std::dynamic_pointer_cast<NonlinearFactor>(r: factor)) {
94 const double errorValue = (factor != nullptr ? nf->error(c: values) : .0);
95 if (!printCondition(factor.get(), errorValue, i))
96 continue; // User-provided filter did not pass
97
98 if (factor == nullptr) {
99 std::cout << "nullptr"
100 << "\n";
101 } else {
102 factor->print(s: ss.str(), formatter: keyFormatter);
103 std::cout << "error = " << errorValue << "\n";
104 }
105 } else if (auto gf = std::dynamic_pointer_cast<GaussianFactor>(r: factor)) {
106 const double errorValue = (factor != nullptr ? gf->error(c: values) : .0);
107 if (!printCondition(factor.get(), errorValue, i))
108 continue; // User-provided filter did not pass
109
110 if (factor == nullptr) {
111 std::cout << "nullptr"
112 << "\n";
113 } else {
114 factor->print(s: ss.str(), formatter: keyFormatter);
115 std::cout << "error = " << errorValue << "\n";
116 }
117 } else if (auto df = std::dynamic_pointer_cast<DiscreteFactor>(r: factor)) {
118 if (factor == nullptr) {
119 std::cout << "nullptr"
120 << "\n";
121 } else {
122 factor->print(s: ss.str(), formatter: keyFormatter);
123 std::cout << "error = ";
124 df->errorTree().print(s: "", labelFormatter: keyFormatter);
125 std::cout << std::endl;
126 }
127
128 } else {
129 continue;
130 }
131
132 std::cout << "\n";
133 }
134 std::cout.flush();
135}

137/* ************************************************************************* */
138HybridGaussianFactorGraph::shared_ptr HybridNonlinearFactorGraph::linearize(
139 const Values& continuousValues) const {
140 using std::dynamic_pointer_cast;
141
142 // create an empty linear FG
143 auto linearFG = std::make_shared<HybridGaussianFactorGraph>();
144
145 linearFG->reserve(size: size());
146
147 // linearize all hybrid factors
148 for (auto& f : factors_) {
149 // First check if it is a valid factor
150 if (!f) {
151 continue;
152 }
153 // Check if it is a hybrid nonlinear factor
154 if (auto mf = dynamic_pointer_cast<HybridNonlinearFactor>(r: f)) {
155 const HybridGaussianFactor::shared_ptr& gmf =
156 mf->linearize(continuousValues);
157 linearFG->push_back(factor: gmf);
158 } else if (auto nlf = dynamic_pointer_cast<NonlinearFactor>(r: f)) {
159 const GaussianFactor::shared_ptr& gf = nlf->linearize(c: continuousValues);
160 linearFG->push_back(factor: gf);
161 } else if (dynamic_pointer_cast<DiscreteFactor>(r: f)) {
162 // If discrete-only: doesn't need linearization.
163 linearFG->push_back(factor: f);
164 } else if (auto gmf = dynamic_pointer_cast<HybridGaussianFactor>(r: f)) {
165 linearFG->push_back(factor: gmf);
166 } else if (auto gm = dynamic_pointer_cast<HybridGaussianConditional>(r: f)) {
167 linearFG->push_back(factor: gm);
168 } else if (dynamic_pointer_cast<GaussianFactor>(r: f)) {
169 linearFG->push_back(factor: f);
170 } else {
171 auto& fr = *f;
172 throw std::invalid_argument(
173 std::string("HybridNonlinearFactorGraph::linearize: factor type "
174 "not handled: ") +
175 demangle(name: typeid(fr).name()));
176 }
177 }
178 return linearFG;
179}

181/* ************************************************************************* */
182AlgebraicDecisionTree<Key> HybridNonlinearFactorGraph::errorTree(
183 const Values& values) const {
184 AlgebraicDecisionTree<Key> result(0.0);
185
186 // Iterate over each factor.
187 for (auto& factor : factors_) {
188 if (auto hnf = std::dynamic_pointer_cast<HybridNonlinearFactor>(r: factor)) {
189 // Compute factor error and add it.
190 result = result + hnf->errorTree(continuousValues: values);
191
192 } else if (auto nf = std::dynamic_pointer_cast<NonlinearFactor>(r: factor)) {
193 // If continuous only, get the (double) error
194 // and add it to every leaf of the result
195 result = result + nf->error(c: values);
196
197 } else if (auto df = std::dynamic_pointer_cast<DiscreteFactor>(r: factor)) {
198 // If discrete, just add its errorTree as well
199 result = result + df->errorTree();
200
201 } else {
202 throw std::runtime_error(
203 "HybridNonlinearFactorGraph::errorTree(Values) not implemented for "
204 "factor type " +
205 demangle(name: typeid(factor).name()) + ".");
206 }
207 }
208
209 return result;
210}

212/* ************************************************************************ */
213AlgebraicDecisionTree<Key> HybridNonlinearFactorGraph::discretePosterior(
214 const Values& continuousValues) const {
215 AlgebraicDecisionTree<Key> errors = this->errorTree(values: continuousValues);
216 AlgebraicDecisionTree<Key> p = errors.apply(op: [](double error) {
217 // NOTE: The 0.5 term is handled by each factor
218 return exp(x: -error);
219 });
220 return p / p.sum();
221}

224HybridNonlinearFactorGraph HybridNonlinearFactorGraph::restrict(
225 const DiscreteValues& discreteValues) const {
226 using std::dynamic_pointer_cast;
227
228 HybridNonlinearFactorGraph result;
229 result.reserve(size: size());
230 for (auto& f : factors_) {
231 // First check if it is a valid factor
232 if (!f) {
233 continue;
234 }
235 // Check if it is a hybrid factor
236 if (auto hf = dynamic_pointer_cast<HybridFactor>(r: f)) {
237 result.push_back(factor: hf->restrict(discreteValues));
238 } else if (auto df = dynamic_pointer_cast<DiscreteFactor>(r: f)) {
239 auto restricted_df = df->restrict(assignment: discreteValues);
240 // In the case where all the discrete values in the factor
241 // have been selected, we get a factor without any keys,
242 // and default values of 0.5.
243 // Since this factor no longer adds any information, we ignore it to make
244 // inference faster.
245 if (restricted_df->discreteKeys().size() > 0) {
246 result.push_back(factor: restricted_df);
247 }
248 } else {
249 result.push_back(factor: f); // Everything else is just added as is
250 }
251 }
252
253 return result;
254}
