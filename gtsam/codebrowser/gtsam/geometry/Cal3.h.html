28/**
29 * Function which makes use of the Implicit Function Theorem to compute the
30 * Jacobians of `calibrate` using `uncalibrate`.
31 * This is useful when there are iterative operations in the `calibrate`
32 * function which make computing jacobians difficult.
33 *
34 * Given f(pi, pn) = uncalibrate(pn) - pi, and g(pi) = calibrate, we can
35 * easily compute the Jacobians:
36 * df/pi = -I (pn and pi are independent args)
37 * Dp = -inv(H_uncal_pn) * df/pi = -inv(H_uncal_pn) * (-I) = inv(H_uncal_pn)
38 * Dcal = -inv(H_uncal_pn) * df/K = -inv(H_uncal_pn) * H_uncal_K
39 *
40 * @tparam Cal Calibration model.
41 * @tparam Dim The number of parameters in the calibration model.
42 * @param p Calibrated point.
43 * @param Dcal optional 2*p Jacobian wrpt `p` Cal3DS2 parameters.
44 * @param Dp optional 2*2 Jacobian wrpt intrinsic coordinates.
45 */
46template <typename Cal, size_t Dim>
47void calibrateJacobians(const Cal& calibration, const Point2& pn,
48 OptionalJacobian<2, Dim> Dcal = {},
49 OptionalJacobian<2, 2> Dp = {}) {
50 if (Dcal || Dp) {
51 Eigen::Matrix<double, 2, Dim> H_uncal_K;
52 Matrix22 H_uncal_pn, H_uncal_pn_inv;
53
54 // Compute uncalibrate Jacobians
55 calibration.uncalibrate(pn, Dcal ? &H_uncal_K : nullptr, H_uncal_pn);
56
57 H_uncal_pn_inv = H_uncal_pn.inverse();
58
59 if (Dp) *Dp = H_uncal_pn_inv;
60 if (Dcal) *Dcal = -H_uncal_pn_inv * H_uncal_K;
61 }
62}

64/**
65 * @brief Common base class for all calibration models.
66 * @ingroup geometry
67 * \nosubgrouping
68 */
69class GTSAM_EXPORT Cal3 {
70 protected:
71 double fx_ = 1.0f, fy_ = 1.0f; ///< focal length
72 double s_ = 0.0f; ///< skew
73 double u0_ = 0.0f, v0_ = 0.0f; ///< principal point
74
75 public:
76 ///< shared pointer to calibration object
77 using shared_ptr = std::shared_ptr<Cal3>;
78
79 /// @name Standard Constructors
80 /// @{
81
82 /// Create a default calibration that leaves coordinates unchanged
83 Cal3() = default;
84
85 /// constructor from doubles
86 Cal3(double fx, double fy, double s, double u0, double v0)
87 : fx_(fx), fy_(fy), s_(s), u0_(u0), v0_(v0) {}
88
89 /// constructor from vector
90 Cal3(const Vector5& d)
91 : fx_(d(0)), fy_(d(1)), s_(d(2)), u0_(d(3)), v0_(d(4)) {}
92
93 /**
94 * Easy constructor, takes fov in degrees, asssumes zero skew, unit aspect
95 * @param fov field of view in degrees
96 * @param w image width
97 * @param h image height
98 */
99 Cal3(double fov, int w, int h);
100
101 /// Virtual destructor
102 virtual ~Cal3() {}
103
104 /// @}
105 /// @name Advanced Constructors
106 /// @{
107
108 /**
109 * Load calibration parameters from `calibration_info.txt` file located in
110 * `path` directory.
111 *
112 * The contents of calibration file should be the 5 parameters in order:
113 * `fx, fy, s, u0, v0`
114 *
115 * @param path path to directory containing `calibration_info.txt`.
116 */
117 Cal3(const std::string& path);
118
119 /// @}
120 /// @name Testable
121 /// @{
122
123 /// Output stream operator
124 GTSAM_EXPORT friend std::ostream& operator<<(std::ostream& os,
125 const Cal3& cal);
126
127 /// print with optional string
128 virtual void print(const std::string& s = "") const;
129
130 /// Check if equal up to specified tolerance
131 bool equals(const Cal3& K, double tol = 10e-9) const;
132
133 /// @}
134 /// @name Standard Interface
135 /// @{
136
137 /// focal length x
138 double fx() const { return fx_; }
139
140 /// focal length y
141 double fy() const { return fy_; }
142
143 /// aspect ratio
144 double aspectRatio() const { return fx_ / fy_; }
145
146 /// skew
147 double skew() const { return s_; }
148
149 /// image center in x
150 double px() const { return u0_; }
151
152 /// image center in y
153 double py() const { return v0_; }
154
155 /// return the principal point
156 Point2 principalPoint() const { return Point2(u0_, v0_); }
157
158 /// vectorized form (column-wise)
159 Vector5 vector() const {
160 Vector5 v;
161 v << fx_, fy_, s_, u0_, v0_;
162 return v;
163 }
164
165 /// return calibration matrix K
166 virtual Matrix3 K() const {
167 Matrix3 K;
168 K << fx_, s_, u0_, 0.0, fy_, v0_, 0.0, 0.0, 1.0;
169 return K;
170 }
171
172 /// Return inverted calibration matrix inv(K)
173 Matrix3 inverse() const;
174
175 /// @}
176 /// @name Advanced Interface
177 /// @{
178
179 private:
180#if GTSAM_ENABLE_BOOST_SERIALIZATION ///
181 /// Serialization function
182 friend class boost::serialization::access;
183 template <class Archive>
184 void serialize(Archive& ar, const unsigned int /*version*/) {
185 ar& BOOST_SERIALIZATION_NVP(fx_);
186 ar& BOOST_SERIALIZATION_NVP(fy_);
187 ar& BOOST_SERIALIZATION_NVP(s_);
188 ar& BOOST_SERIALIZATION_NVP(u0_);
189 ar& BOOST_SERIALIZATION_NVP(v0_);
190 }
191#endif
192
193 /// @}
194};
195
196} // \ namespace gtsam
