65// Static Constructor/Create functions
66//------------------------------------------------------------------------------
67Gal3 Gal3::Create(const Rot3& R, const Point3& r, const Velocity3& v, double t,
68 OptionalJacobian<10, 3> H1, OptionalJacobian<10, 3> H2,
69 OptionalJacobian<10, 3> H3, OptionalJacobian<10, 1> H4) {
70 Matrix3 Rt;
71 if (H2 || H3 || H4) Rt = R.transpose();
72 if (H1) {
73 H1->setZero();
74 H1->block<3, 3>(startRow: 0, startCol: 0) = I_3x3;
75 }
76 if (H2) {
77 H2->setZero();
78 H2->block<3, 3>(startRow: 6, startCol: 0) = Rt;
79 }
80 if (H3) {
81 H3->setZero();
82 H3->block<3, 3>(startRow: 3, startCol: 0) = Rt;
83 }
84 if (H4) {
85 H4->setZero();
86 H4->block<3, 1>(startRow: 6, startCol: 0) = -Rt * v; // drho_dt;
87 (*H4)(9, 0) = 1.0;
88 }
89 return Gal3(R, r, v, t);
90}

92//------------------------------------------------------------------------------
93Gal3 Gal3::FromPoseVelocityTime(const Pose3& pose, const Velocity3& v, double t,
94 OptionalJacobian<10, 6> H1,
95 OptionalJacobian<10, 3> H2,
96 OptionalJacobian<10, 1> H3) {
97 const Rot3& R = pose.rotation();
98 const Point3& r = pose.translation();
99 Matrix3 Rt;
100 if (H2 || H3) Rt = R.transpose();
101 if (H1) {
102 H1->setZero();
103 H1->block<3, 3>(startRow: 0, startCol: 0) = I_3x3;
104 H1->block<3, 3>(startRow: 6, startCol: 3) = I_3x3;
105 }
106 if (H2) {
107 H2->setZero();
108 H2->block<3, 3>(startRow: 3, startCol: 0) = Rt;
109 }
110 if (H3) {
111 H3->setZero();
112 H3->block<3, 1>(startRow: 6, startCol: 0) = -Rt * v;
113 (*H3)(9, 0) = 1.0;
114 }
115 return Gal3(R, r, v, t);
116}

118//------------------------------------------------------------------------------
119// Constructors
120//------------------------------------------------------------------------------
121Gal3::Gal3(const Matrix5& M) {
122 // Constructor from 5x5 matrix representation (Equation 9, Page 5)
123 if (std::abs(x: M(3, 3) - 1.0) > 1e-9 || std::abs(x: M(4, 4) - 1.0) > 1e-9 ||
124 M.row(i: 4).head(n: 4).norm() > 1e-9 || M.row(i: 3).head(n: 3).norm() > 1e-9) {
125 throw std::invalid_argument("Invalid Gal3 matrix structure: Check zero blocks and diagonal ones.");
126 }
127 R_ = Rot3(M.block<3, 3>(startRow: 0, startCol: 0));
128 v_ = M.block<3, 1>(startRow: 0, startCol: 3);
129 r_ = Point3(M.block<3, 1>(startRow: 0, startCol: 4));
130 t_ = M(3, 4);
131}

172//------------------------------------------------------------------------------
173Matrix5 Gal3::matrix() const {
174 // Returns 5x5 matrix representation as in Equation 9, Page 5
175 Matrix5 M = Matrix5::Identity();
176 M.block<3, 3>(startRow: 0, startCol: 0) = R_.matrix();
177 M.block<3, 1>(startRow: 0, startCol: 3) = v_;
178 M.block<3, 1>(startRow: 0, startCol: 4) = Vector3(r_);
179 M(3, 4) = t_;
180 M.block<1,3>(startRow: 3,startCol: 0).setZero();
181 M.block<1,4>(startRow: 4,startCol: 0).setZero();
182 return M;
183}

185//------------------------------------------------------------------------------
186Gal3::Vector25 Gal3::vec(OptionalJacobian<25, 10> H) const {
187 const Matrix5 T = this->matrix();
188 if (H) {
189 H->setZero();
190 auto R = T.block<3, 3>(startRow: 0, startCol: 0);
191 H->block<3, 1>(startRow: 0, startCol: 1) = -R.col(i: 2);
192 H->block<3, 1>(startRow: 0, startCol: 2) = R.col(i: 1);
193 H->block<3, 1>(startRow: 5, startCol: 0) = R.col(i: 2);
194 H->block<3, 1>(startRow: 5, startCol: 2) = -R.col(i: 0);
195 H->block<3, 1>(startRow: 10, startCol: 0) = -R.col(i: 1);
196 H->block<3, 1>(startRow: 10, startCol: 1) = R.col(i: 0);
197 H->block<3, 3>(startRow: 15, startCol: 3) = R;
198 H->block<3, 3>(startRow: 20, startCol: 6) = R;
199 H->block<3, 1>(startRow: 20, startCol: 9) = T.block<3, 1>(startRow: 0, startCol: 3);
200 (*H)(23, 9) = 1;
201 }
202 return Eigen::Map<const Vector25>(T.data());
203}

224//------------------------------------------------------------------------------
225bool Gal3::equals(const Gal3& other, double tol) const {
226 return R_.equals(p: other.R_, tol) &&
227 traits<Point3>::Equals(v1: r_, v2: other.r_, tol) &&
228 traits<Velocity3>::Equals(v1: v_, v2: other.v_, tol) &&
229 std::abs(x: t_ - other.t_) < tol;

232//------------------------------------------------------------------------------
233// Group Operations
234//------------------------------------------------------------------------------
235Gal3 Gal3::inverse() const {
236 // Implements inverse formula from Equation 10, Page 5
237 const Rot3 Rinv = R_.inverse();
238 const Velocity3 v_inv = -(Rinv.rotate(p: v_));
239 const Point3 r_inv = -(Rinv.rotate(p: Vector3(r_) - t_ * v_));
240 const double t_inv = -t_;
241 return Gal3(Rinv, r_inv, v_inv, t_inv);
242}

244//------------------------------------------------------------------------------
245Gal3 Gal3::operator*(const Gal3& other) const {
246 // Implements group composition through matrix multiplication
247 const Gal3& g1 = *this;
248 const Gal3& g2 = other;
249
250 const Rot3 R_comp = g1.R_.compose(g: g2.R_);
251 const Vector3 r1_vec(g1.r_);
252 const Vector3 r2_vec(g2.r_);
253 const Vector3 r_comp_vec = g1.R_.rotate(p: r2_vec) + g2.t_ * g1.v_ + r1_vec;
254 const Velocity3 v_comp = g1.R_.rotate(p: g2.v_) + g1.v_;
255 const double t_comp = g1.t_ + g2.t_;
256
257 return Gal3(R_comp, Point3(r_comp_vec), v_comp, t_comp);
258}

260//------------------------------------------------------------------------------
261// Lie Group Static Functions
262//------------------------------------------------------------------------------
263Gal3 Gal3::Expmap(const TangentVector& xi, OptionalJacobian<10, 10> Hxi) {
264 // Implements exponential map from Equations 16-19, Pages 7-8
265 const Vector3 w = xi_w(v: xi);
266 const Vector3 nu = xi_nu(v: xi);
267 const Vector3 rho = xi_rho(v: xi);
268 const double alpha = xi_t(v: xi)(0);
269
270 // Create functor for SO3::Expmap and derivatives, plus Gamma matrix we need
271 const so3::DexpFunctor local(w);
272
273 // Compute rotation using Expmap
274#ifdef GTSAM_USE_QUATERNIONS
275 const Rot3 R = traits<Quaternion>::Expmap(w);
276#else
277 const Rot3 R(local.expmap());
278#endif
279 Matrix3 Rt;
280 if (Hxi) Rt = R.transpose();
281
282 // Compute velocity: just apply left SO(3) Jacobian,
283 Matrix3 H_v_w;
284 const Velocity3 v = local.Jacobian().applyLeft(v: nu, Hw: Hxi ? &H_v_w : nullptr);
285
286 // Compute position: apply left Jacobian and compute time-dependent part
287 Matrix3 H_p_w, H_delta_w;
288 Point3 p = local.Jacobian().applyLeft(v: rho, Hw: Hxi ? &H_p_w : nullptr);
289 const Point3 delta = local.Gamma().applyLeft(v: nu, Hw: Hxi ? &H_delta_w : nullptr);
290
291 // (*) means: different from Arxiv paper!
292 if (Hxi) {
293 const Matrix3 Jr = local.Jacobian().right();
294 const Matrix3 Gr = local.Gamma().right();
295 *Hxi << Jr, Z_3x3, Z_3x3, Z_3x1, //
296 Rt * H_v_w, Jr, Z_3x3, Z_3x1, //
297 Rt * H_p_w, Z_3x3, Jr, -Gr * nu, // (*)
298 Z_9x1.transpose(), 1.0;
299 }
300
301 // if alpha!=0, augment position with time-dependent bit.
302 if (std::abs(x: alpha) > kSmallTimeThreshold) {
303 p += alpha * delta;
304 if (Hxi) {
305 // Derivative of time-dependent part
306 Hxi->block<3, 3>(startRow: 6, startCol: 0) += alpha * Rt * H_delta_w;
307 Hxi->block<3, 3>(startRow: 6, startCol: 3) = alpha * Rt * local.Gamma().left(); // (*)
308 }
309 }
310 return Gal3(R, p, v, alpha);
311}

313//------------------------------------------------------------------------------
314Gal3::TangentVector Gal3::Logmap(const Gal3& g, OptionalJacobian<10, 10> Hg) {
315 // Implements logarithmic map from Equations 20-23, Page 8
316 const Vector3 w = Rot3::Logmap(R: g.R_);
317 const so3::DexpFunctor local(w);
318
319 // Implementation of Equation 23, Page 8
320 const Matrix3 Jl_inv = local.InvJacobian().left();
321 Matrix3 Gl = local.Gamma().left();
322 const Vector3 nu = Jl_inv * g.v_;
323 const Vector3 rho = Jl_inv * (g.r_ - Gl * (g.t_ * nu));
324
325 TangentVector xi;
326 xi_w(v&: xi) = w;
327 xi_nu(v&: xi) = nu;
328 xi_rho(v&: xi) = rho;
329 xi_t(v&: xi)(0) = g.t_;
330
331 if (Hg) {
332 *Hg = Gal3::LogmapDerivative(g);
333 }
334
335 return xi;
336}

338//------------------------------------------------------------------------------
339Gal3::Jacobian Gal3::AdjointMap() const {
340 // Implements Adjoint map as in Equation 26, Page 9
341 const Matrix3 R = R_.matrix();
342 Matrix3 A = skewSymmetric(w: r_ - t_ * v_) * R;
343 Matrix3 B = skewSymmetric(w: v_) * R;
344
345 Jacobian Ad;
346 Ad << R, Z_3x3, Z_3x3, Z_3x1, //
347 B, R, Z_3x3, Z_3x1, //
348 A, -t_ * R, R, v_, //
349 Z_9x1.transpose(), 1.0;
350 return Ad;
351}

353//------------------------------------------------------------------------------
354Gal3::TangentVector Gal3::Adjoint(const TangentVector& xi, OptionalJacobian<10, 10> H_g, OptionalJacobian<10, 10> H_xi) const {
355 Jacobian Ad = AdjointMap();
356 TangentVector y = Ad * xi;
357
358 if (H_xi) {
359 *H_xi = Ad;
360 }
361
362 if (H_g) {
363 // NOTE: Using numerical derivative for the Jacobian with respect to
364 // the group element instead of deriving the analytical expression.
365 // Future work to use analytical instead.
366 std::function<TangentVector(const Gal3&, const TangentVector&)> adjoint_action_wrt_g =
367 [&](const Gal3& g_in, const TangentVector& xi_in) {
368 return g_in.Adjoint(xi: xi_in);
369 };
370 *H_g = numericalDerivative21(h: adjoint_action_wrt_g, x1: *this, x2: xi, delta: 1e-7);
371 }
372 return y;
373}

375//------------------------------------------------------------------------------
376Gal3::Jacobian Gal3::adjointMap(const TangentVector& xi) {
377 // Implements adjoint representation as in Equation 28, Page 10
378 const Matrix3 Omega = skewSymmetric(w: xi_w(v: xi));
379 const Matrix3 nuHat = skewSymmetric(w: xi_nu(v: xi));
380 const Matrix3 rhoHat = skewSymmetric(w: xi_rho(v: xi));
381 const double alpha = xi_t(v: xi)(0);
382 const Vector3 nu = xi_nu(v: xi);
383
384 Jacobian ad;
385 ad << Omega, Z_3x3, Z_3x3, Z_3x1, //
386 nuHat, Omega, Z_3x3, Z_3x1, //
387 rhoHat, -alpha * I_3x3, Omega, nu, //
388 Z_9x1.transpose(), 0.0;
389 return ad;
390}

393Gal3::TangentVector Gal3::adjoint(const TangentVector& xi, const TangentVector& y, OptionalJacobian<10, 10> Hxi, OptionalJacobian<10, 10> Hy) {
394 Jacobian ad_xi = adjointMap(xi);
395 if (Hy) *Hy = ad_xi;
396 if (Hxi) {
397 *Hxi = -adjointMap(xi: y);
398 }
399 return ad_xi * y;

409//------------------------------------------------------------------------------
410Gal3::Jacobian Gal3::LogmapDerivative(const Gal3& g) {
411 // Related to the inverse of left Jacobian in Equations 31-36, Pages 10-11
412 // NOTE: Using numerical approximation instead of implementing the analytical
413 // expression for the inverse Jacobian. Future work to replace this
414 // with analytical derivative.
415 TangentVector xi = Gal3::Logmap(g);
416 if (xi.norm() < kSmallAngleThreshold) return Jacobian::Identity();
417 std::function<TangentVector(const Gal3&)> fn =
418 [](const Gal3& g_in) { return Gal3::Logmap(g: g_in); };
419 return numericalDerivative11<TangentVector, Gal3>(h: fn, x: g, delta: 1e-5);
420}

422//------------------------------------------------------------------------------
423// Lie Algebra (Hat/Vee maps)
424//------------------------------------------------------------------------------
425Gal3::LieAlgebra Gal3::Hat(const TangentVector& xi) {
426 // Implements hat operator as in Equation 13, Page 6
427 const Vector3 rho = xi_rho(v: xi);
428 const Vector3 nu = xi_nu(v: xi);
429 const Vector3 w = xi_w(v: xi);
430 const double t = xi_t(v: xi)(0);
431
432 Matrix5 X = Matrix5::Zero();
433 X.block<3, 3>(startRow: 0, startCol: 0) = skewSymmetric(w);
434 X.block<3, 1>(startRow: 0, startCol: 3) = nu;
435 X.block<3, 1>(startRow: 0, startCol: 4) = rho;
436 X(3, 4) = t;
437 return X;
438}

440//------------------------------------------------------------------------------
441Gal3::TangentVector Gal3::Vee(const LieAlgebra& X) {
442 // Implements vee operator (inverse of hat operator in Equation 13, Page 6)
443 if (X.row(i: 4).norm() > 1e-9 || X.row(i: 3).head(n: 3).norm() > 1e-9 || std::abs(x: X(3,3)) > 1e-9) {
444 throw std::invalid_argument("Matrix is not in sgal(3)");
445 }
446
447 TangentVector xi;
448 xi_rho(v&: xi) = X.block<3, 1>(startRow: 0, startCol: 4);
449 xi_nu(v&: xi) = X.block<3, 1>(startRow: 0, startCol: 3);
450 const Matrix3& S = X.block<3, 3>(startRow: 0, startCol: 0);
451 xi_w(v&: xi) << S(2, 1), S(0, 2), S(1, 0);
452 xi_t(v&: xi)(0) = X(3, 4);
453 return xi;
454}

468//------------------------------------------------------------------------------
469Event Gal3::act(const Event& e, OptionalJacobian<4, 10> Hself,
470 OptionalJacobian<4, 4> He) const {
471 // Implements group action on events (spacetime points) as described in Section 4.1, Page 3-4
472 const double& t_in = e.time();
473 const Point3& p_in = e.location();
474
475 const double t_out = t_in + t_;
476 const Point3 p_out = R_.rotate(p: p_in) + v_ * t_in + r_;
477
478 if (Hself) {
479 Hself->setZero();
480 const Matrix3 R = R_.matrix();
481
482 (*Hself)(0, 9) = 1.0;
483 Hself->block<3, 3>(startRow: 1, startCol: 6) = R;
484 Hself->block<3, 3>(startRow: 1, startCol: 3) = R * t_in;
485 Hself->block<3, 3>(startRow: 1, startCol: 0) = -R * skewSymmetric(w: p_in);
486 Hself->block<3, 1>(startRow: 1, startCol: 9) = v_;
487 }
488
489 if (He) {
490 He->setZero();
491 (*He)(0, 0) = 1.0;
492 He->block<3, 1>(startRow: 1, startCol: 0) = v_;
493 He->block<3, 3>(startRow: 1, startCol: 1) = R_.matrix();
494 }
495
496 return Event(t_out, p_out);
497}
