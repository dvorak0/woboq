36static constexpr double one_6th = 1.0 / 6.0;
37static constexpr double one_12th = 1.0 / 12.0;
38static constexpr double one_24th = 1.0 / 24.0;
39static constexpr double one_60th = 1.0 / 60.0;
40static constexpr double one_120th = 1.0 / 120.0;
41static constexpr double one_180th = 1.0 / 180.0;
42static constexpr double one_360th = 1.0 / 360.0;
43static constexpr double one_720th = 1.0 / 720.0;
44static constexpr double one_1260th = 1.0 / 1260.0;
45
46static constexpr double kPi_inv = 1.0 / M_PI;
47static constexpr double kPi2 = M_PI * M_PI;
48static constexpr double k1_Pi2 = 1.0 / kPi2;
49static constexpr double kPi3 = M_PI * kPi2;
50static constexpr double k1_Pi3 = 1.0 / kPi3;
51static constexpr double k2_Pi3 = 2.0 * k1_Pi3;
52static constexpr double k1_4Pi = 0.25 * kPi_inv; // 1/(4*pi)
53
54// --- Thresholds ---
55// Tolerance for near zero (theta^2)
56static constexpr double kNearZeroThresholdSq = 1e-6;
57// Tolerance for near pi (delta^2 = (pi - theta)^2)
58static constexpr double kNearPiThresholdSq = 1e-6;
59
60GTSAM_EXPORT Matrix99 Dcompose(const SO3& Q) {
61 Matrix99 H;
62 auto R = Q.matrix();
63 H << I_3x3 * R(0, 0), I_3x3 * R(1, 0), I_3x3 * R(2, 0), //
64 I_3x3 * R(0, 1), I_3x3 * R(1, 1), I_3x3 * R(2, 1), //
65 I_3x3 * R(0, 2), I_3x3 * R(1, 2), I_3x3 * R(2, 2);
66 return H;
67}

69GTSAM_EXPORT Matrix3 compose(const Matrix3& M, const SO3& R,
70 OptionalJacobian<9, 9> H) {
71 Matrix3 MR = M * R.matrix();
72 if (H) *H = Dcompose(Q: R);
73 return MR;
74}

76void ExpmapFunctor::init(double nearZeroThresholdSq) {
77 nearZero = (theta2 <= nearZeroThresholdSq);
78
79 if (!nearZero) {
80 // General case: Use standard stable formulas for A and B
81 const double sin_theta = std::sin(x: theta);
82 A = sin_theta / theta;
83 const double s2 = std::sin(x: theta / 2.0);
84 const double one_minus_cos =
85 2.0 * s2 * s2; // numerically better than [1 - cos(theta)]
86 B = one_minus_cos / theta2;
87 } else {
88 // Taylor expansion at 0 for A, B (Order theta^2)
89 A = 1.0 - theta2 * one_6th;
90 B = 0.5 - theta2 * one_24th;
91 }
92}
93
94ExpmapFunctor::ExpmapFunctor(const Vector3& omega) :ExpmapFunctor(kNearZeroThresholdSq, omega) {}
95
96ExpmapFunctor::ExpmapFunctor(double nearZeroThresholdSq, const Vector3& omega)
97 : theta2(omega.dot(other: omega)),
98 theta(std::sqrt(x: theta2)),
99 W(skewSymmetric(w: omega)),
100 WW(W * W) {
101 init(nearZeroThresholdSq);
102}
103
104ExpmapFunctor::ExpmapFunctor(const Vector3& axis, double angle)
105 : theta2(angle * angle),
106 theta(angle),
107 W(skewSymmetric(w: axis * angle)),
108 WW(W * W) {
109 init(nearZeroThresholdSq: kNearZeroThresholdSq);
110}
111
112
113Matrix3 ExpmapFunctor::expmap() const { return I_3x3 + A * W + B * WW; }
114
115DexpFunctor::DexpFunctor(const Vector3& omega, double nearZeroThresholdSq, double nearPiThresholdSq)
116 : ExpmapFunctor(nearZeroThresholdSq, omega), omega(omega) {
117 // General case or nearPi: Use standard stable formulas first
118 const double delta = M_PI > theta ? M_PI - theta : 0.0;
119 const double delta2 = delta * delta;
120 nearPi = (delta2 < nearPiThresholdSq);
121}
122
123DexpFunctor::DexpFunctor(const Vector3& omega)
124 : DexpFunctor(omega, kNearZeroThresholdSq, kNearPiThresholdSq) {}
125
126double DexpFunctor::C() const {
127 if (!C_.has_value()) {
128 // Usually stable, even near pi (1-0)/pi^2
129 C_ = !nearZero ? (1.0 - A) / theta2 : (one_6th - theta2 * one_120th);
130 }
131 return C_.value();
132}
133
134double DexpFunctor::D() const {
135 if (!D_.has_value()) {
136 D_ = !nearZero ? (nearPi ? (k1_Pi2 + (k2_Pi3 - k1_4Pi) * (M_PI - theta))
137 : ((1.0 - A / (2.0 * B)) / theta2))
138 : (one_12th + theta2 * one_720th);
139 }
140 return D_.value();
141}
142
143double DexpFunctor::E() const {
144 if (!E_.has_value()) {
145 E_ = !nearZero ? ((1.0 - 2.0 * B) / (2.0 * theta2))
146 : (one_24th - theta2 * one_720th);
147 }
148 return E_.value();
149}
150
151double DexpFunctor::dA() const {
152 if (!dA_.has_value()) {
153 // Identity: dA = A/ = C B (valid for all , with our near-zero series)
154 dA_ = C() - B;
155 }
156 return dA_.value();
157}
158
159double DexpFunctor::dB() const {
160 if (!dB_.has_value()) {
161 dB_ =
162 !nearZero ? ((A - 2.0 * B) / theta2) : (-one_12th + theta2 * one_180th);
163 }
164 return dB_.value();
165}
166
167double DexpFunctor::dC() const {
168 if (!dC_.has_value()) {
169 dC_ = !nearZero ? ((B - 3.0 * C()) / theta2)
170 : (-one_60th + theta2 * one_1260th);
171 }
172 return dC_.value();
173}
174
175double DexpFunctor::dE() const {
176 if (!dE_.has_value()) {
177 dE_ = !nearZero ? (-(dB() + 2.0 * E()) / theta2) : (-one_360th);
178 }
179 return dE_.value();
180}

231template <>
232GTSAM_EXPORT
233SO3 SO3::ClosestTo(const Matrix3& M) {
234 Eigen::JacobiSVD<Matrix3> svd(M, Eigen::ComputeFullU | Eigen::ComputeFullV);
235 const auto& U = svd.matrixU();
236 const auto& V = svd.matrixV();
237 const double det = (U * V.transpose()).determinant();
238 return SO3(U * Vector3(1, 1, det).asDiagonal() * V.transpose());
239}

242template <>
243GTSAM_EXPORT
244SO3 SO3::ChordalMean(const std::vector<SO3>& rotations) {
245 // See Hartley13ijcv:
246 // Cost function C(R) = \sum sqr(|R-R_i|_F)
247 // Closed form solution = ClosestTo(C_e), where C_e = \sum R_i !!!!
248 Matrix3 C_e{Z_3x3};
249 for (const auto& R_i : rotations) {
250 C_e += R_i.matrix();
251 }
252 return ClosestTo(M: C_e);
253}

295template <>
296GTSAM_EXPORT
297Vector3 SO3::Logmap(const SO3& Q, ChartJacobian H) {
298 using std::sin;
299 using std::sqrt;
300
301 // note switch to base 1
302 const Matrix3& R = Q.matrix();
303 const double &R11 = R(0, 0), R12 = R(0, 1), R13 = R(0, 2);
304 const double &R21 = R(1, 0), R22 = R(1, 1), R23 = R(1, 2);
305 const double &R31 = R(2, 0), R32 = R(2, 1), R33 = R(2, 2);
306
307 const double tr = R.trace();
308 Vector3 omega;
309
310 // when trace == -1, i.e., when theta = +-pi, +-3pi, +-5pi, etc.
311 // we do something special
312 if (tr + 1.0 < 1e-3) {
313 if (R33 > R22 && R33 > R11) {
314 // R33 is the largest diagonal, a=3, b=1, c=2
315 const double W = R21 - R12;
316 const double Q1 = 2.0 + 2.0 * R33;
317 const double Q2 = R31 + R13;
318 const double Q3 = R23 + R32;
319 const double r = sqrt(x: Q1);
320 const double one_over_r = 1 / r;
321 const double norm = sqrt(x: Q1*Q1 + Q2*Q2 + Q3*Q3 + W*W);
322 const double sgn_w = W < 0 ? -1.0 : 1.0;
323 const double mag = M_PI - (2 * sgn_w * W) / norm;
324 const double scale = 0.5 * one_over_r * mag;
325 omega = sgn_w * scale * Vector3(Q2, Q3, Q1);
326 } else if (R22 > R11) {
327 // R22 is the largest diagonal, a=2, b=3, c=1
328 const double W = R13 - R31;
329 const double Q1 = 2.0 + 2.0 * R22;
330 const double Q2 = R23 + R32;
331 const double Q3 = R12 + R21;
332 const double r = sqrt(x: Q1);
333 const double one_over_r = 1 / r;
334 const double norm = sqrt(x: Q1*Q1 + Q2*Q2 + Q3*Q3 + W*W);
335 const double sgn_w = W < 0 ? -1.0 : 1.0;
336 const double mag = M_PI - (2 * sgn_w * W) / norm;
337 const double scale = 0.5 * one_over_r * mag;
338 omega = sgn_w * scale * Vector3(Q3, Q1, Q2);
339 } else {
340 // R11 is the largest diagonal, a=1, b=2, c=3
341 const double W = R32 - R23;
342 const double Q1 = 2.0 + 2.0 * R11;
343 const double Q2 = R12 + R21;
344 const double Q3 = R31 + R13;
345 const double r = sqrt(x: Q1);
346 const double one_over_r = 1 / r;
347 const double norm = sqrt(x: Q1*Q1 + Q2*Q2 + Q3*Q3 + W*W);
348 const double sgn_w = W < 0 ? -1.0 : 1.0;
349 const double mag = M_PI - (2 * sgn_w * W) / norm;
350 const double scale = 0.5 * one_over_r * mag;
351 omega = sgn_w * scale * Vector3(Q1, Q2, Q3);
352 }
353 } else {
354 double magnitude;
355 const double tr_3 = tr - 3.0; // could be non-negative if the matrix is off orthogonal
356 if (tr_3 < -so3::kNearZeroThresholdSq) {
357 // this is the normal case -1 < trace < 3
358 double theta = acos(x: (tr - 1.0) / 2.0);
359 magnitude = theta / (2.0 * sin(x: theta));
360 } else {
361 // when theta near 0, +-2pi, +-4pi, etc. (trace near 3.0)
362 // use Taylor expansion: theta \approx 1/2-(t-3)/12 + O((t-3)^2)
363 // see https://github.com/borglab/gtsam/issues/746 for details
364 magnitude = 0.5 - tr_3 * (1.0 / 12.0) + tr_3 * tr_3 * (1.0 / 60.0);
365 }
366 omega = magnitude * Vector3(R32 - R23, R13 - R31, R21 - R12);
367 }
368
369 if (H) *H = LogmapDerivative(omega);
370 return omega;
371}

389template <>
390GTSAM_EXPORT
391Vector9 SO3::vec(OptionalJacobian<9, 3> H) const {
392 const Matrix3& R = matrix_;
393 if (H) {
394 H->setZero();
395 H->block<3, 1>(startRow: 0, startCol: 1) = -R.col(i: 2);
396 H->block<3, 1>(startRow: 0, startCol: 2) = R.col(i: 1);
397 H->block<3, 1>(startRow: 3, startCol: 0) = R.col(i: 2);
398 H->block<3, 1>(startRow: 3, startCol: 2) = -R.col(i: 0);
399 H->block<3, 1>(startRow: 6, startCol: 0) = -R.col(i: 1);
400 H->block<3, 1>(startRow: 6, startCol: 1) = R.col(i: 0);
401 }
402 return Eigen::Map<const Vector9>(R.data());
403}
