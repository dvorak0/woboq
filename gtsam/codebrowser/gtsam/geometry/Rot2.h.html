20#pragma once
21
22#include <gtsam/geometry/Point2.h>
23#include <gtsam/base/MatrixLieGroup.h>
24
25#include <random>
26
27namespace gtsam {
28
29 /**
30 * Rotation matrix
31 * NOTE: the angle theta is in radians unless explicitly stated
32 * @ingroup geometry
33 * \nosubgrouping
34 */
35 class GTSAM_EXPORT Rot2 : public MatrixLieGroup<Rot2, 1, 2> {
36 /** we store cos(theta) and sin(theta) */
37 double c_, s_;
38
39 /** normalize to make sure cos and sin form unit vector */
40 Rot2& normalize();
41
42 /** private constructor from cos/sin */
43 inline Rot2(double c, double s) : c_(c), s_(s) {}
44
45 public:
46
47 /// @name Constructors and named constructors
48 /// @{
49
50 /** default constructor, zero rotation */
51 Rot2() : c_(1.0), s_(0.0) {}
52
53 /** copy constructor */
54 Rot2(const Rot2& r) = default;
55
56 Rot2& operator=(const Rot2& other) = default;
57
58 /// Constructor from angle in radians == exponential map at identity
59 Rot2(double theta) : c_(cos(x: theta)), s_(sin(x: theta)) {}
60
61 // Rot2& operator=(const gtsam::Rot2& other) = default;
62
63 /// Named constructor from angle in radians
64 static Rot2 fromAngle(double theta) {
65 return Rot2(theta);
66 }
67
68 /// Named constructor from angle in degrees
69 static Rot2 fromDegrees(double theta) {
70 static const double degree = M_PI / 180;
71 return fromAngle(theta: theta * degree);
72 }
73
74 /// Named constructor from cos(theta),sin(theta) pair
75 static Rot2 fromCosSin(double c, double s);
76
77 /**
78 * Named constructor with derivative
79 * Calculate relative bearing to a landmark in local coordinate frame
80 * @param d 2D location of landmark
81 * @param H optional reference for Jacobian
82 * @return 2D rotation \f$ \in SO(2) \f$
83 */
84 static Rot2 relativeBearing(const Point2& d, OptionalJacobian<1,2> H =
85 {});
86
87 /** Named constructor that behaves as atan2, i.e., y,x order (!) and normalizes */
88 static Rot2 atan2(double y, double x);
89
90 /**
91 * Random, generates random angle \f$\in\f$ [-pi,pi]
92 * Example:
93 * std::mt19937 engine(42);
94 * Unit3 unit = Unit3::Random(engine);
95 */
96 static Rot2 Random(std::mt19937 & rng);
97
98 /// @}
99 /// @name Testable
100 /// @{
101
102 /** print */
103 void print(const std::string& s = "theta") const;
104
105 /** equals with an tolerance */
106 bool equals(const Rot2& R, double tol = 1e-9) const;
107
108 /// @}
109 /// @name Group
110 /// @{
111
112 /** Identity */
113 inline static Rot2 Identity() { return Rot2(); }
114
115 /** The inverse rotation - negative angle */
116 Rot2 inverse() const { return Rot2(c_, -s_);}
117
118 /** Compose - make a new rotation by adding angles */
119 Rot2 operator*(const Rot2& R) const {
120 return fromCosSin(c: c_ * R.c_ - s_ * R.s_, s: s_ * R.c_ + c_ * R.s_);
121 }
122
123 /// @}
124 /// @name Lie Group
125 /// @{
126
127 using LieAlgebra = Matrix2;
128
129 /// Exponential map at identity - create a rotation from canonical coordinates
130 static Rot2 Expmap(const Vector1& v, ChartJacobian H = {});
131
132 /// Log map at identity - return the canonical coordinates of this rotation
133 static Vector1 Logmap(const Rot2& r, ChartJacobian H = {});
134
135 /** Calculate Adjoint map */
136 Matrix1 AdjointMap() const { return I_1x1; }
137
138 /// Left-trivialized derivative of the exponential map
139 static Matrix ExpmapDerivative(const Vector& /*v*/) {
140 return I_1x1;
141 }
142
143 /// Left-trivialized derivative inverse of the exponential map
144 static Matrix LogmapDerivative(const Vector& /*v*/) {
145 return I_1x1;
146 }
147
148 // Chart at origin simply uses exponential map and its inverse
149 struct ChartAtOrigin {
150 static Rot2 Retract(const Vector1& v, ChartJacobian H = {}) {
151 return Expmap(v, H);
152 }
153 static Vector1 Local(const Rot2& r, ChartJacobian H = {}) {
154 return Logmap(r, H);
155 }
156 };
157
158 using LieGroup<Rot2, 1>::inverse; // version with derivative
159
160 /// Hat maps from tangent vector to Lie algebra
161 static Matrix2 Hat(const Vector1& xi);
162
163 /// Vee maps from Lie algebra to tangent vector
164 static Vector1 Vee(const Matrix2& X);
165
166 /// @}
167 /// @name Group Action on Point2
168 /// @{
169
170 /**
171 * rotate point from rotated coordinate frame to world \f$ p^w = R_c^w p^c \f$
172 */
173 Point2 rotate(const Point2& p, OptionalJacobian<2, 1> H1 = {},
174 OptionalJacobian<2, 2> H2 = {}) const;
175
176 /** syntactic sugar for rotate */
177 inline Point2 operator*(const Point2& p) const {
178 return rotate(p);
179 }
180
181 /**
182 * rotate point from world to rotated frame \f$ p^c = (R_c^w)^T p^w \f$
183 */
184 Point2 unrotate(const Point2& p, OptionalJacobian<2, 1> H1 = {},
185 OptionalJacobian<2, 2> H2 = {}) const;
186
187 /// @}
188 /// @name Standard Interface
189 /// @{
190
191 /// Creates a unit vector as a Point2
192 inline Point2 unit() const {
193 return Point2(c_, s_);
194 }
195
196 /** return angle (RADIANS) */
197 double theta() const {
198 return ::atan2(y: s_, x: c_);
199 }
200
201 /** return angle (DEGREES) */
202 double degrees() const {
203 const double degree = M_PI / 180;
204 return theta() / degree;
205 }
206
207 /** return cos */
208 inline double c() const {
209 return c_;
210 }
211
212 /** return sin */
213 inline double s() const {
214 return s_;
215 }
216
217 /** return 2*2 rotation matrix */
218 Matrix2 matrix() const;
219
220 /** return 2*2 transpose (inverse) rotation matrix */
221 Matrix2 transpose() const;
222
223 /** Find closest valid rotation matrix, given a 2x2 matrix */
224 static Rot2 ClosestTo(const Matrix2& M);
225
226 /** Vectorize the rotation matrix into a 4D vector */
227 Vector4 vec(OptionalJacobian<4, 1> H = {}) const;
228 /// @}
229
230 private:
231#if GTSAM_ENABLE_BOOST_SERIALIZATION
232 /** Serialization function */
233 friend class boost::serialization::access;
234 template<class ARCHIVE>
235 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
236 ar & BOOST_SERIALIZATION_NVP(c_);
237 ar & BOOST_SERIALIZATION_NVP(s_);
238 }
239#endif
240
241 };
242
243 template <>
244struct traits<Rot2> : public internal::MatrixLieGroup<Rot2, 2> {};
245
246template <>
247struct traits<const Rot2> : public internal::MatrixLieGroup<Rot2, 2> {};

249} // gtsam
