26/**
27 * A pinhole camera class that has a Pose3 and a Calibration.
28 * Use PinholePose if you will not be optimizing for Calibration
29 * @ingroup geometry
30 * \nosubgrouping
31 */
32template<typename Calibration>
33class PinholeCamera: public PinholeBaseK<Calibration> {
34
35public:
36
37 /**
38 * Some classes template on either PinholeCamera or StereoCamera,
39 * and this typedef informs those classes what "project" returns.
40 */
41 typedef Point2 Measurement;
42 typedef Point2Vector MeasurementVector;
43
44private:
45
46 typedef PinholeBaseK<Calibration> Base; ///< base class has 3D pose as private member
47 Calibration K_; ///< Calibration, part of class now
48
49 // Get dimensions of calibration type at compile time
50 static const int DimK = FixedDimension<Calibration>::value;
51
52public:
53
54 inline constexpr static auto dimension = 6 + DimK; ///< Dimension depends on calibration
55
56 /// @name Standard Constructors
57 /// @{
58
59 /** default constructor */
60 PinholeCamera() {
61 }
62
63 /** constructor with pose */
64 explicit PinholeCamera(const Pose3& pose) :
65 Base(pose) {
66 }
67
68 /** constructor with pose and calibration */
69 PinholeCamera(const Pose3& pose, const Calibration& K) :
70 Base(pose), K_(K) {
71 }
72
73 /// @}
74 /// @name Named Constructors
75 /// @{
76
77 /**
78 * Create a level camera at the given 2D pose and height
79 * @param K the calibration
80 * @param pose2 specifies the location and viewing direction
81 * (theta 0 = looking in direction of positive X axis)
82 * @param height camera height
83 */
84 static PinholeCamera Level(const Calibration &K, const Pose2& pose2,
85 double height) {
86 return PinholeCamera(Base::LevelPose(pose2, height), K);
87 }
88
89 /// PinholeCamera::level with default calibration
90 static PinholeCamera Level(const Pose2& pose2, double height) {
91 return PinholeCamera::Level(Calibration(), pose2, height);
92 }
93
94 /**
95 * Create a camera at the given eye position looking at a target point in the scene
96 * with the specified up direction vector.
97 * @param eye specifies the camera position
98 * @param target the point to look at
99 * @param upVector specifies the camera up direction vector,
100 * doesn't need to be on the image plane nor orthogonal to the viewing axis
101 * @param K optional calibration parameter
102 */
103 static PinholeCamera Lookat(const Point3& eye, const Point3& target,
104 const Point3& upVector, const Calibration& K = Calibration()) {
105 return PinholeCamera(Base::LookatPose(eye, target, upVector), K);
106 }
107
108 // Create PinholeCamera, with derivatives
109 static PinholeCamera Create(const Pose3& pose, const Calibration &K,
110 OptionalJacobian<dimension, 6> H1 = {}, //
111 OptionalJacobian<dimension, DimK> H2 = {}) {
112 typedef Eigen::Matrix<double, DimK, 6> MatrixK6;
113 if (H1)
114 *H1 << I_6x6, MatrixK6::Zero();
115 typedef Eigen::Matrix<double, 6, DimK> Matrix6K;
116 typedef Eigen::Matrix<double, DimK, DimK> MatrixK;
117 if (H2)
118 *H2 << Matrix6K::Zero(), MatrixK::Identity();
119 return PinholeCamera(pose,K);
120 }
121
122 /// @}
123 /// @name Advanced Constructors
124 /// @{
125
126 /// Init from vector, can be 6D (default calibration) or dim
127 explicit PinholeCamera(const Vector &v) :
128 Base(v.head<6>()) {
129 if (v.size() > 6)
130 K_ = Calibration(v.tail<DimK>());
131 }
132
133 /// Init from Vector and calibration
134 PinholeCamera(const Vector &v, const Vector &K) :
135 Base(v), K_(K) {
136 }
137
138 /// @}
139 /// @name Testable
140 /// @{
141
142 /// assert equality up to a tolerance
143 bool equals(const Base &camera, double tol = 1e-9) const {
144 const PinholeCamera* e = dynamic_cast<const PinholeCamera*>(&camera);
145 return Base::equals(camera, tol) && K_.equals(e->calibration(), tol);
146 }
147
148 /// print
149 void print(const std::string& s = "PinholeCamera") const override {
150 Base::print(s);
151 K_.print(s + ".calibration");
152 }
153
154 /// @}
155 /// @name Standard Interface
156 /// @{
157
158 ~PinholeCamera() override {
159 }
160
161 /// return pose
162 const Pose3& pose() const {
163 return Base::pose();
164 }
165
166 /// return pose, with derivative
167 const Pose3& getPose(OptionalJacobian<6, dimension> H) const {
168 if (H) {
169 H->setZero();
170 H->template block<6, 6>(0, 0) = I_6x6;
171 }
172 return Base::pose();
173 }
174
175 /// return calibration
176 const Calibration& calibration() const override {
177 return K_;
178 }
179
180 /// @}
181 /// @name Manifold
182 /// @{
183
184 size_t dim() const {
185 return dimension;
186 }
187
188 static size_t Dim() {
189 return dimension;
190 }
191
192 typedef Eigen::Matrix<double, dimension, 1> VectorK6;
193
194 /// move a cameras according to d
195 PinholeCamera retract(const Vector& d) const {
196 if ((size_t) d.size() == 6)
197 return PinholeCamera(this->pose().retract(d), calibration());
198 else
199 return PinholeCamera(this->pose().retract(d.head<6>()),
200 calibration().retract(d.tail<DimK>()));
201 }
202
203 /// return canonical coordinate
204 VectorK6 localCoordinates(const PinholeCamera& T2) const {
205 VectorK6 d;
206 d.template head<6>() = this->pose().localCoordinates(T2.pose());
207 d.template tail<DimK>() = calibration().localCoordinates(T2.calibration());
208 return d;
209 }
210
211 /// for Canonical
212 static PinholeCamera Identity() {
213 return PinholeCamera(); // assumes that the default constructor is valid
214 }
215
216 /// @}
217 /// @name Transformations and measurement functions
218 /// @{
219
220 typedef Eigen::Matrix<double, 2, DimK> Matrix2K;
221
222 /** Templated projection of a 3D point or a point at infinity into the image
223 * @param pw either a Point3 or a Unit3, in world coordinates
224 */
225 template<class POINT>
226 Point2 _project2(const POINT& pw, OptionalJacobian<2, dimension> Dcamera,
227 OptionalJacobian<2, FixedDimension<POINT>::value> Dpoint) const {
228 // We just call 3-derivative version in Base
229 if (Dcamera){
230 Matrix26 Dpose;
231 Eigen::Matrix<double, 2, DimK> Dcal;
232 const Point2 pi = Base::project(pw, Dpose, Dpoint, Dcal);
233 *Dcamera << Dpose, Dcal;
234 return pi;
235 } else {
236 return Base::project(pw, {}, Dpoint, {});
237 }
238 }
239
240 /// project a 3D point from world coordinates into the image
241 Point2 project2(const Point3& pw, OptionalJacobian<2, dimension> Dcamera =
242 {}, OptionalJacobian<2, 3> Dpoint = {}) const {
243 return _project2(pw, Dcamera, Dpoint);
244 }
245
246 /// project a point at infinity from world coordinates into the image
247 Point2 project2(const Unit3& pw, OptionalJacobian<2, dimension> Dcamera =
248 {}, OptionalJacobian<2, 2> Dpoint = {}) const {
249 return _project2(pw, Dcamera, Dpoint);
250 }
251
252 /**
253 * Calculate range to a landmark
254 * @param point 3D location of landmark
255 * @return range (double)
256 */
257 double range(const Point3& point, OptionalJacobian<1, dimension> Dcamera =
258 {}, OptionalJacobian<1, 3> Dpoint = {}) const {
259 Matrix16 Dpose_;
260 double result = this->pose().range(point, Dcamera ? &Dpose_ : 0, Dpoint);
261 if (Dcamera)
262 *Dcamera << Dpose_, Eigen::Matrix<double, 1, DimK>::Zero();
263 return result;
264 }
265
266 /**
267 * Calculate range to another pose
268 * @param pose Other SO(3) pose
269 * @return range (double)
270 */
271 double range(const Pose3& pose, OptionalJacobian<1, dimension> Dcamera =
272 {}, OptionalJacobian<1, 6> Dpose = {}) const {
273 Matrix16 Dpose_;
274 double result = this->pose().range(pose, Dcamera ? &Dpose_ : 0, Dpose);
275 if (Dcamera)
276 *Dcamera << Dpose_, Eigen::Matrix<double, 1, DimK>::Zero();
277 return result;
278 }
279
280 /**
281 * Calculate range to another camera
282 * @param camera Other camera
283 * @return range (double)
284 */
285 template<class CalibrationB>
286 double range(const PinholeCamera<CalibrationB>& camera,
287 OptionalJacobian<1, dimension> Dcamera = {},
288 OptionalJacobian<1, 6 + CalibrationB::dimension> Dother = {}) const {
289 Matrix16 Dcamera_, Dother_;
290 double result = this->pose().range(camera.pose(), Dcamera ? &Dcamera_ : 0,
291 Dother ? &Dother_ : 0);
292 if (Dcamera) {
293 *Dcamera << Dcamera_, Eigen::Matrix<double, 1, DimK>::Zero();
294 }
295 if (Dother) {
296 Dother->setZero();
297 Dother->template block<1, 6>(0, 0) = Dother_;
298 }
299 return result;
300 }
301
302 /**
303 * Calculate range to a calibrated camera
304 * @param camera Other camera
305 * @return range (double)
306 */
307 double range(const CalibratedCamera& camera,
308 OptionalJacobian<1, dimension> Dcamera = {},
309 OptionalJacobian<1, 6> Dother = {}) const {
310 return range(camera.pose(), Dcamera, Dother);
311 }
312
313 /// for Linear Triangulation
314 Matrix34 cameraProjectionMatrix() const {
315 return K_.K() * PinholeBase::pose().inverse().matrix().block(0, 0, 3, 4);
316 }
317
318 /// for Nonlinear Triangulation
319 Vector defaultErrorWhenTriangulatingBehindCamera() const {
320 return Eigen::Matrix<double,traits<Point2>::dimension,1>::Constant(2.0 * K_.fx());
321 }
322
323private:
324

332 boost::serialization::base_object<Base>(*this));
333 ar & BOOST_SERIALIZATION_NVP(K_);
334 }
335#endif
