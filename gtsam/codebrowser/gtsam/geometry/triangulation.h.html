116/**
117 * Create a factor graph with projection factors from poses and one calibration
118 * @param poses Camera poses
119 * @param sharedCal shared pointer to single calibration object (monocular only!)
120 * @param measurements 2D measurements
121 * @param landmarkKey to refer to landmark
122 * @param initialEstimate
123 * @return graph and initial values
124 */
125template<class CALIBRATION>
126std::pair<NonlinearFactorGraph, Values> triangulationGraph(
127 const std::vector<Pose3>& poses, std::shared_ptr<CALIBRATION> sharedCal,
128 const Point2Vector& measurements, Key landmarkKey,
129 const Point3& initialEstimate,
130 const SharedNoiseModel& model = noiseModel::Unit::Create(dim: 2)) {
131 Values values;
132 values.insert(j: landmarkKey, val: initialEstimate); // Initial landmark value
133 NonlinearFactorGraph graph;
134 for (size_t i = 0; i < measurements.size(); i++) {
135 const Pose3& pose_i = poses[i];
136 typedef PinholePose<CALIBRATION> Camera;
137 Camera camera_i(pose_i, sharedCal);
138 graph.emplace_shared<TriangulationFactor<Camera> > //
139 (camera_i, measurements[i], model, landmarkKey);
140 }
141 return {graph, values};
142}

144/**
145 * Create a factor graph with projection factors from pinhole cameras
146 * (each camera has a pose and calibration)
147 * @param cameras pinhole cameras (monocular or stereo)
148 * @param measurements 2D measurements
149 * @param landmarkKey to refer to landmark
150 * @param initialEstimate
151 * @return graph and initial values
152 */
153template<class CAMERA>
154std::pair<NonlinearFactorGraph, Values> triangulationGraph(
155 const CameraSet<CAMERA>& cameras,
156 const typename CAMERA::MeasurementVector& measurements, Key landmarkKey,
157 const Point3& initialEstimate,
158 const SharedNoiseModel& model = nullptr) {
159 Values values;
160 values.insert(j: landmarkKey, val: initialEstimate); // Initial landmark value
161 NonlinearFactorGraph graph;
162 static SharedNoiseModel unit(noiseModel::Unit::Create(
163 dim: traits<typename CAMERA::Measurement>::dimension));
164 for (size_t i = 0; i < measurements.size(); i++) {
165 const CAMERA& camera_i = cameras[i];
166 graph.emplace_shared<TriangulationFactor<CAMERA> > //
167 (camera_i, measurements[i], model? model : unit, landmarkKey);
168 }
169 return {graph, values};
170}

182/**
183 * Given an initial estimate , refine a point using measurements in several cameras
184 * @param poses Camera poses
185 * @param sharedCal shared pointer to single calibration object
186 * @param measurements 2D measurements
187 * @param initialEstimate
188 * @return refined Point3
189 */
190template<class CALIBRATION>
191Point3 triangulateNonlinear(const std::vector<Pose3>& poses,
192 std::shared_ptr<CALIBRATION> sharedCal,
193 const Point2Vector& measurements, const Point3& initialEstimate,
194 const SharedNoiseModel& model = nullptr) {
195
196 // Create a factor graph and initial values
197 const auto [graph, values] = triangulationGraph<CALIBRATION> //
198 (poses, sharedCal, measurements, Symbol('p', 0), initialEstimate, model);
199
200 return optimize(graph, values, Symbol('p', 0));
201}

223template<class CAMERA>
224std::vector<Matrix34, Eigen::aligned_allocator<Matrix34>>
225projectionMatricesFromCameras(const CameraSet<CAMERA> &cameras) {
226 std::vector<Matrix34, Eigen::aligned_allocator<Matrix34>> projection_matrices;
227 for (const CAMERA &camera: cameras) {
228 projection_matrices.push_back(camera.cameraProjectionMatrix());
229 }
230 return projection_matrices;
231}

233// overload, assuming pinholePose
234template<class CALIBRATION>
235std::vector<Matrix34, Eigen::aligned_allocator<Matrix34>> projectionMatricesFromPoses(
236 const std::vector<Pose3> &poses, std::shared_ptr<CALIBRATION> sharedCal) {
237 std::vector<Matrix34, Eigen::aligned_allocator<Matrix34>> projection_matrices;
238 for (size_t i = 0; i < poses.size(); i++) {
239 PinholePose<CALIBRATION> camera(poses.at(n: i), sharedCal);
240 projection_matrices.push_back(camera.cameraProjectionMatrix());
241 }
242 return projection_matrices;
243}

245/** Create a pinhole calibration from a different Cal3 object, removing
246 * distortion.
247 *
248 * @tparam CALIBRATION Original calibration object.
249 * @param cal Input calibration object.
250 * @return Cal3_S2 with only the pinhole elements of cal.
251 */
252template <class CALIBRATION>
253Cal3_S2 createPinholeCalibration(const CALIBRATION& cal) {
254 const auto& K = cal.K();
255 return Cal3_S2(K(0, 0), K(1, 1), K(0, 1), K(0, 2), K(1, 2));
256}

258/** Internal undistortMeasurement to be used by undistortMeasurement and
259 * undistortMeasurements */
260template <class CALIBRATION, class MEASUREMENT>
261MEASUREMENT undistortMeasurementInternal(
262 const CALIBRATION& cal, const MEASUREMENT& measurement,
263 std::optional<Cal3_S2> pinholeCal = {}) {
264 if (!pinholeCal) {
265 pinholeCal = createPinholeCalibration(cal);
266 }
267 return pinholeCal->uncalibrate(p: cal.calibrate(measurement));
268}

270/** Remove distortion for measurements so as if the measurements came from a
271 * pinhole camera.
272 *
273 * Removes distortion but maintains the K matrix of the initial cal. Operates by
274 * calibrating using full calibration and uncalibrating with only the pinhole
275 * component of the calibration.
276 * @tparam CALIBRATION Calibration type to use.
277 * @param cal Calibration with which measurements were taken.
278 * @param measurements Vector of measurements to undistort.
279 * @return measurements with the effect of the distortion of sharedCal removed.
280 */
281template <class CALIBRATION>
282Point2Vector undistortMeasurements(const CALIBRATION& cal,
283 const Point2Vector& measurements) {
284 Cal3_S2 pinholeCalibration = createPinholeCalibration(cal);
285 Point2Vector undistortedMeasurements;
286 // Calibrate with cal and uncalibrate with pinhole version of cal so that
287 // measurements are undistorted.
288 std::transform(measurements.begin(), measurements.end(),
289 std::back_inserter(x&: undistortedMeasurements),
290 [&cal, &pinholeCalibration](const Point2& measurement) {
291 return undistortMeasurementInternal<CALIBRATION>(
292 cal, measurement, pinholeCalibration);
293 });
294 return undistortedMeasurements;
295}

304/** Remove distortion for measurements so as if the measurements came from a
305 * pinhole camera.
306 *
307 * Removes distortion but maintains the K matrix of the initial calibrations.
308 * Operates by calibrating using full calibration and uncalibrating with only
309 * the pinhole component of the calibration.
310 * @tparam CAMERA Camera type to use.
311 * @param cameras Cameras corresponding to each measurement.
312 * @param measurements Vector of measurements to undistort.
313 * @return measurements with the effect of the distortion of the camera removed.
314 */
315template <class CAMERA>
316typename CAMERA::MeasurementVector undistortMeasurements(
317 const CameraSet<CAMERA>& cameras,
318 const typename CAMERA::MeasurementVector& measurements) {
319 const size_t nrMeasurements = measurements.size();
320#ifndef NDEBUG
321 if (nrMeasurements != cameras.size()) {
322 throw;
323 }
324#endif
325 typename CAMERA::MeasurementVector undistortedMeasurements(nrMeasurements);
326 for (size_t ii = 0; ii < nrMeasurements; ++ii) {
327 // Calibrate with cal and uncalibrate with pinhole version of cal so that
328 // measurements are undistorted.
329 undistortedMeasurements[ii] =
330 undistortMeasurementInternal<typename CAMERA::CalibrationType>(
331 cameras[ii].calibration(), measurements[ii]);
332 }
333 return undistortedMeasurements;
334}

352/** Convert pixel measurements in image to homogeneous measurements in the image
353 * plane using shared camera intrinsics.
354 *
355 * @tparam CALIBRATION Calibration type to use.
356 * @param cal Calibration with which measurements were taken.
357 * @param measurements Vector of measurements to undistort.
358 * @return homogeneous measurements in image plane
359 */
360template <class CALIBRATION>
361inline Point3Vector calibrateMeasurementsShared(
362 const CALIBRATION& cal, const Point2Vector& measurements) {
363 Point3Vector calibratedMeasurements;
364 // Calibrate with cal and uncalibrate with pinhole version of cal so that
365 // measurements are undistorted.
366 std::transform(measurements.begin(), measurements.end(),
367 std::back_inserter(x&: calibratedMeasurements),
368 [&cal](const Point2& measurement) {
369 Point3 p;
370 p << cal.calibrate(measurement), 1.0;
371 return p;
372 });
373 return calibratedMeasurements;
374}
375
376/** Convert pixel measurements in image to homogeneous measurements in the image
377 * plane using camera intrinsics of each measurement.
378 *
379 * @tparam CAMERA Camera type to use.
380 * @param cameras Cameras corresponding to each measurement.
381 * @param measurements Vector of measurements to undistort.
382 * @return homogeneous measurements in image plane
383 */
384template <class CAMERA>
385inline Point3Vector calibrateMeasurements(
386 const CameraSet<CAMERA>& cameras,
387 const typename CAMERA::MeasurementVector& measurements) {
388 const size_t nrMeasurements = measurements.size();
389 assert(nrMeasurements == cameras.size());
390 Point3Vector calibratedMeasurements(nrMeasurements);
391 for (size_t ii = 0; ii < nrMeasurements; ++ii) {
392 calibratedMeasurements[ii]
393 << cameras[ii].calibration().calibrate(measurements[ii]),
394 1.0;
395 }
396 return calibratedMeasurements;
397}

399/** Specialize for SphericalCamera to do nothing. */
400template <class CAMERA = SphericalCamera>
401inline Point3Vector calibrateMeasurements(
402 const CameraSet<SphericalCamera>& cameras,
403 const SphericalCamera::MeasurementVector& measurements) {
404 Point3Vector calibratedMeasurements(measurements.size());
405 for (size_t ii = 0; ii < measurements.size(); ++ii) {
406 calibratedMeasurements[ii] << measurements[ii].point3();
407 }
408 return calibratedMeasurements;

411/**
412 * Function to triangulate 3D landmark point from an arbitrary number
413 * of poses (at least 2) using the DLT. The function checks that the
414 * resulting point lies in front of all cameras, but has no other checks
415 * to verify the quality of the triangulation.
416 * @param poses A vector of camera poses
417 * @param sharedCal shared pointer to single calibration object
418 * @param measurements A vector of camera measurements
419 * @param rank_tol rank tolerance, default 1e-9
420 * @param optimize Flag to turn on nonlinear refinement of triangulation
421 * @param useLOST whether to use the LOST algorithm instead of DLT
422 * @return Returns a Point3
423 */
424template <class CALIBRATION>
425Point3 triangulatePoint3(const std::vector<Pose3>& poses,
426 std::shared_ptr<CALIBRATION> sharedCal,
427 const Point2Vector& measurements,
428 double rank_tol = 1e-9, bool optimize = false,
429 const SharedNoiseModel& model = nullptr,
430 const bool useLOST = false) {
431 assert(poses.size() == measurements.size());
432 if (poses.size() < 2) throw(TriangulationUnderconstrainedException());
433
434 // Triangulate linearly
435 Point3 point;
436 if (useLOST) {
437 // Reduce input noise model to an isotropic noise model using the mean of
438 // the diagonal.
439 const double measurementSigma = model ? model->sigmas().mean() : 1e-4;
440 SharedIsotropic measurementNoise =
441 noiseModel::Isotropic::Sigma(dim: 2, sigma: measurementSigma);
442 // calibrate the measurements to obtain homogenous coordinates in image
443 // plane.
444 auto calibratedMeasurements =
445 calibrateMeasurementsShared<CALIBRATION>(*sharedCal, measurements);
446
447 point = triangulateLOST(poses, calibratedMeasurements, measurementNoise,
448 rank_tol);
449 } else {
450 // construct projection matrices from poses & calibration
451 auto projection_matrices = projectionMatricesFromPoses(poses, sharedCal);
452
453 // Undistort the measurements, leaving only the pinhole elements in effect.
454 auto undistortedMeasurements =
455 undistortMeasurements<CALIBRATION>(*sharedCal, measurements);
456
457 point =
458 triangulateDLT(projection_matrices, undistortedMeasurements, rank_tol);
459 }
460
461 // Then refine using non-linear optimization
462 if (optimize)
463 point = triangulateNonlinear<CALIBRATION> //
464 (poses, sharedCal, measurements, point, model);
465
466#ifdef GTSAM_THROW_CHEIRALITY_EXCEPTION
467 // verify that the triangulated point lies in front of all cameras
468 for (const Pose3& pose : poses) {
469 const Point3& p_local = pose.transformTo(point);
470 if (p_local.z() <= 0) throw(TriangulationCheiralityException());
471 }
472#endif
473
474 return point;
475}

477/**
478 * Function to triangulate 3D landmark point from an arbitrary number
479 * of poses (at least 2) using the DLT. This function is similar to the one
480 * above, except that each camera has its own calibration. The function
481 * checks that the resulting point lies in front of all cameras, but has
482 * no other checks to verify the quality of the triangulation.
483 * @param cameras pinhole cameras
484 * @param measurements A vector of camera measurements
485 * @param rank_tol rank tolerance, default 1e-9
486 * @param optimize Flag to turn on nonlinear refinement of triangulation
487 * @param useLOST whether to use the LOST algorithm instead of
488 * DLT
489 * @return Returns a Point3
490 */
491template <class CAMERA>
492Point3 triangulatePoint3(const CameraSet<CAMERA>& cameras,
493 const typename CAMERA::MeasurementVector& measurements,
494 double rank_tol = 1e-9, bool optimize = false,
495 const SharedNoiseModel& model = nullptr,
496 const bool useLOST = false) {
497 size_t m = cameras.size();
498 assert(measurements.size() == m);
499
500 if (m < 2) throw(TriangulationUnderconstrainedException());
501
502 // Triangulate linearly
503 Point3 point;
504 if (useLOST) {
505 // Reduce input noise model to an isotropic noise model using the mean of
506 // the diagonal.
507 const double measurementSigma = model ? model->sigmas().mean() : 1e-4;
508 SharedIsotropic measurementNoise =
509 noiseModel::Isotropic::Sigma(dim: 2, sigma: measurementSigma);
510
511 // construct poses from cameras.
512 std::vector<Pose3> poses;
513 poses.reserve(n: cameras.size());
514 for (const auto& camera : cameras) poses.push_back(camera.pose());
515
516 // calibrate the measurements to obtain homogenous coordinates in image
517 // plane.
518 auto calibratedMeasurements =
519 calibrateMeasurements<CAMERA>(cameras, measurements);
520
521 point = triangulateLOST(poses, calibratedMeasurements, measurementNoise,
522 rank_tol);
523 } else {
524 // construct projection matrices from poses & calibration
525 auto projection_matrices = projectionMatricesFromCameras(cameras);
526
527 // Undistort the measurements, leaving only the pinhole elements in effect.
528 auto undistortedMeasurements =
529 undistortMeasurements<CAMERA>(cameras, measurements);
530
531 point =
532 triangulateDLT(projection_matrices, undistortedMeasurements, rank_tol);
533 }
534
535 // Then refine using non-linear optimization
536 if (optimize) {
537 point = triangulateNonlinear<CAMERA>(cameras, measurements, point, model);
538 }
539
540#ifdef GTSAM_THROW_CHEIRALITY_EXCEPTION
541 // verify that the triangulated point lies in front of all cameras
542 for (const CAMERA& camera : cameras) {
543 const Point3& p_local = camera.pose().transformTo(point);
544 if (p_local.z() <= 0) throw(TriangulationCheiralityException());
545 }
546#endif
547
548 return point;
549}

562struct GTSAM_EXPORT TriangulationParameters {
563
564 double rankTolerance; ///< threshold to decide whether triangulation is result.degenerate
565 ///< (the rank is the number of singular values of the triangulation matrix which are larger than rankTolerance)
566 bool enableEPI; ///< if set to true, will refine triangulation using LM
567
568 /**
569 * if the landmark is triangulated at distance larger than this,
570 * result is flagged as degenerate.
571 */
572 double landmarkDistanceThreshold; //
573
574 /**
575 * If this is nonnegative the we will check if the average reprojection error
576 * is smaller than this threshold after triangulation, otherwise result is
577 * flagged as degenerate.
578 */
579 double dynamicOutlierRejectionThreshold;
580
581 /**
582 * if true, will use the LOST algorithm instead of DLT
583 */
584 bool useLOST;
585
586 SharedNoiseModel noiseModel; ///< used in the nonlinear triangulation
587
588 /**
589 * Constructor
590 * @param rankTol tolerance used to check if point triangulation is degenerate
591 * @param enableEPI if true refine triangulation with embedded LM iterations
592 * @param landmarkDistanceThreshold flag as degenerate if point further than this
593 * @param dynamicOutlierRejectionThreshold or if average error larger than this
594 * @param noiseModel noise model to use during nonlinear triangulation
595 *
596 */
597 TriangulationParameters(const double _rankTolerance = 1.0,
598 const bool _enableEPI = false, double _landmarkDistanceThreshold = -1,
599 double _dynamicOutlierRejectionThreshold = -1,
600 const bool _useLOST = false,
601 const SharedNoiseModel& _noiseModel = nullptr) :
602 rankTolerance(_rankTolerance), enableEPI(_enableEPI), //
603 landmarkDistanceThreshold(_landmarkDistanceThreshold), //
604 dynamicOutlierRejectionThreshold(_dynamicOutlierRejectionThreshold),
605 useLOST(_useLOST),
606 noiseModel(_noiseModel){
607 }
608
609 // stream to output
610 friend std::ostream &operator<<(std::ostream &os,
611 const TriangulationParameters& p) {
612 os << "rankTolerance = " << p.rankTolerance << std::endl;
613 os << "enableEPI = " << p.enableEPI << std::endl;
614 os << "landmarkDistanceThreshold = " << p.landmarkDistanceThreshold
615 << std::endl;
616 os << "dynamicOutlierRejectionThreshold = "
617 << p.dynamicOutlierRejectionThreshold << std::endl;
618 os << "useLOST = " << p.useLOST << std::endl;
619 os << "noise model" << std::endl;
620 return os;
621 }

638/**
639 * TriangulationResult is an optional point, along with the reasons why it is
640 * invalid.
641 */
642class TriangulationResult : public std::optional<Point3> {
643 public:
644 enum Status { VALID, DEGENERATE, BEHIND_CAMERA, OUTLIER, FAR_POINT };
645 Status status;
646
647 private:
648 TriangulationResult(Status s) : status(s) {}
649
650 public:
651 /**
652 * Default constructor, only for serialization
653 */
654 TriangulationResult() {}
655
656 /**
657 * Constructor
658 */
659 TriangulationResult(const Point3& p) : status(VALID) { emplace(args: p); }
660 static TriangulationResult Degenerate() {
661 return TriangulationResult(DEGENERATE);
662 }
663 static TriangulationResult Outlier() { return TriangulationResult(OUTLIER); }
664 static TriangulationResult FarPoint() {
665 return TriangulationResult(FAR_POINT);
666 }
667 static TriangulationResult BehindCamera() {
668 return TriangulationResult(BEHIND_CAMERA);
669 }
670 bool valid() const { return status == VALID; }
671 bool degenerate() const { return status == DEGENERATE; }
672 bool outlier() const { return status == OUTLIER; }
673 bool farPoint() const { return status == FAR_POINT; }
674 bool behindCamera() const { return status == BEHIND_CAMERA; }
675 const gtsam::Point3& get() const {
676 if (!has_value()) throw std::runtime_error("TriangulationResult has no value");
677 return value();
678 }
679 // stream to output
680 GTSAM_EXPORT friend std::ostream& operator<<(
681 std::ostream& os, const TriangulationResult& result) {
682 if (result)
683 os << "point = " << *result << std::endl;
684 else
685 os << "no point, status = " << result.status << std::endl;
686 return os;
687 }

700/// triangulateSafe: extensive checking of the outcome
701template<class CAMERA>
702TriangulationResult triangulateSafe(const CameraSet<CAMERA>& cameras,
703 const typename CAMERA::MeasurementVector& measured,
704 const TriangulationParameters& params) {
705
706 size_t m = cameras.size();
707
708 // if we have a single pose the corresponding factor is uninformative
709 if (m < 2)
710 return TriangulationResult::Degenerate();
711 else
712 // We triangulate the 3D position of the landmark
713 try {
714 Point3 point =
715 triangulatePoint3<CAMERA>(cameras, measured, params.rankTolerance,
716 params.enableEPI, params.noiseModel, params.useLOST);
717
718 // Check landmark distance and re-projection errors to avoid outliers
719 size_t i = 0;
720 double maxReprojError = 0.0;
721 for(const CAMERA& camera: cameras) {
722 const Pose3& pose = camera.pose();
723 if (params.landmarkDistanceThreshold > 0
724 && distance3(p1: pose.translation(), q: point)
725 > params.landmarkDistanceThreshold)
726 return TriangulationResult::FarPoint();
727#ifdef GTSAM_THROW_CHEIRALITY_EXCEPTION
728 // verify that the triangulated point lies in front of all cameras
729 // Only needed if this was not yet handled by exception
730 const Point3& p_local = pose.transformTo(point);
731 if (p_local.z() <= 0)
732 return TriangulationResult::BehindCamera();
733#endif
734 // Check reprojection error
735 if (params.dynamicOutlierRejectionThreshold > 0) {
736 const typename CAMERA::Measurement& zi = measured.at(i);
737 Point2 reprojectionError = camera.reprojectionError(point, zi);
738 maxReprojError = std::max(a: maxReprojError, b: reprojectionError.norm());
739 }
740 i += 1;
741 }
742 // Flag as degenerate if average reprojection error is too large
743 if (params.dynamicOutlierRejectionThreshold > 0
744 && maxReprojError > params.dynamicOutlierRejectionThreshold)
745 return TriangulationResult::Outlier();
746
747 // all good!
748 return TriangulationResult(point);
749 } catch (TriangulationUnderconstrainedException&) {
750 // This exception is thrown if
751 // 1) There is a single pose for triangulation - this should not happen because we checked the number of poses before
752 // 2) The rank of the matrix used for triangulation is < 3: rotation-only, parallel cameras (or motion towards the landmark)
753 return TriangulationResult::Degenerate();
754 } catch (TriangulationCheiralityException&) {
755 // point is behind one of the cameras: can be the case of close-to-parallel cameras or may depend on outliers
756 return TriangulationResult::BehindCamera();
757 }
758}
