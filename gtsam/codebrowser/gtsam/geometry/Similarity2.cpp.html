30/// Subtract centroids from point pairs.
31static Point2Pairs SubtractCentroids(const Point2Pairs& abPointPairs,
32 const Point2Pair& centroids) {
33 Point2Pairs d_abPointPairs;
34 for (const auto& [a, b] : abPointPairs) {
35 Point2 da = a - centroids.first;
36 Point2 db = b - centroids.second;
37 d_abPointPairs.emplace_back(args&: da, args&: db);
38 }
39 return d_abPointPairs;
40}
41
42/// Form inner products x and y and calculate scale.
43static double CalculateScale(const Point2Pairs& d_abPointPairs,
44 const Rot2& aRb) {
45 double x = 0, y = 0;
46
47 for (const auto& [da, db] : d_abPointPairs) {
48 const Vector2 da_prime = aRb * db;
49 y += da.transpose() * da_prime;
50 x += da_prime.transpose() * da_prime;
51 }
52 const double s = y / x;
53 return s;
54}
55
56/// Form outer product H.
57static Matrix2 CalculateH(const Point2Pairs& d_abPointPairs) {
58 Matrix2 H = Z_2x2;
59 for (const auto& [da, db] : d_abPointPairs) {
60 H += da * db.transpose();
61 }
62 return H;
63}

65/**
66 * @brief This method estimates the similarity transform from differences point
67 * pairs, given a known or estimated rotation and point centroids.
68 *
69 * @param d_abPointPairs
70 * @param aRb
71 * @param centroids
72 * @return Similarity2
73 */
74static Similarity2 Align(const Point2Pairs& d_abPointPairs, const Rot2& aRb,
75 const Point2Pair& centroids) {
76 const double s = CalculateScale(d_abPointPairs, aRb);
77 // dividing aTb by s is required because the registration cost function
78 // minimizes ||a - sRb - t||, whereas Sim(2) computes s(Rb + t)
79 const Point2 aTb = (centroids.first - s * (aRb * centroids.second)) / s;
80 return Similarity2(aRb, aTb, s);
81}

83/**
84 * @brief This method estimates the similarity transform from point pairs,
85 * given a known or estimated rotation.
86 * Refer to:
87 * http://www5.informatik.uni-erlangen.de/Forschung/Publikationen/2005/Zinsser05-PSR.pdf
88 * Chapter 3
89 *
90 * @param abPointPairs
91 * @param aRb
92 * @return Similarity2
93 */
94static Similarity2 AlignGivenR(const Point2Pairs& abPointPairs,
95 const Rot2& aRb) {
96 auto centroids = means(abPointPairs);
97 auto d_abPointPairs = internal::SubtractCentroids(abPointPairs, centroids);
98 return internal::Align(d_abPointPairs, aRb, centroids);
99}

117bool Similarity2::equals(const Similarity2& other, double tol) const {
118 return R_.equals(R: other.R_, tol) &&
119 traits<Point2>::Equals(v1: t_, v2: other.t_, tol) && s_ < (other.s_ + tol) &&
120 s_ > (other.s_ - tol);
121}
122
123bool Similarity2::operator==(const Similarity2& other) const {
124 return R_.matrix() == other.R_.matrix() && t_ == other.t_ && s_ == other.s_;
125}

137Similarity2 Similarity2::operator*(const Similarity2& S) const {
138 return Similarity2(R_ * S.R_, ((1.0 / S.s_) * t_) + R_ * S.t_, s_ * S.s_);

141Similarity2 Similarity2::inverse() const {
142 const Rot2 Rt = R_.inverse();
143 const Point2 sRt = Rt * (-s_ * t_);
144 return Similarity2(Rt, sRt, 1.0 / s_);

162Similarity2 Similarity2::Align(const Point2Pairs& abPointPairs) {
163 // Refer to Chapter 3 of
164 // http://www5.informatik.uni-erlangen.de/Forschung/Publikationen/2005/Zinsser05-PSR.pdf
165 if (abPointPairs.size() < 2)
166 throw std::runtime_error("input should have at least 2 pairs of points");
167 auto centroids = means(abPointPairs);
168 auto d_abPointPairs = internal::SubtractCentroids(abPointPairs, centroids);
169 Matrix2 H = internal::CalculateH(d_abPointPairs);
170 // ClosestTo finds rotation matrix closest to H in Frobenius sense
171 Rot2 aRb = Rot2::ClosestTo(M: H);
172 return internal::Align(d_abPointPairs, aRb, centroids);
173}

175Similarity2 Similarity2::Align(const Pose2Pairs& abPosePairs) {
176 const size_t n = abPosePairs.size();
177 if (n < 2)
178 throw std::runtime_error("input should have at least 2 pairs of poses");
179
180 // calculate rotation
181 vector<Rot2> rotations;
182 Point2Pairs abPointPairs;
183 rotations.reserve(n: n);
184 abPointPairs.reserve(n: n);
185 // Below denotes the pose of the i'th object/camera/etc
186 // in frame "a" or frame "b".
187 for (const auto& [aTi, bTi] : abPosePairs) {
188 const Rot2 aRb = aTi.rotation().compose(g: bTi.rotation().inverse());
189 rotations.emplace_back(args: aRb);
190 abPointPairs.emplace_back(args: aTi.translation(), args: bTi.translation());
191 }
192 const Rot2 aRb_estimate = FindKarcherMean<Rot2>(rotations);
193
194 return internal::AlignGivenR(abPointPairs, aRb: aRb_estimate);
195}

197Matrix2 Similarity2::GetV(double theta, double lambda) {
198 // Derivation from https://ethaneade.com/lie_groups.pdf page 6
199 const double lambda2 = lambda * lambda, theta2 = theta * theta;
200
201 // SE(2) or near-SE(2) case (|| tiny)
202 if (std::abs(x: lambda) < 1e-9) {
203 double A, B;
204 if (theta2 > 1e-9) {
205 A = sin(x: theta) / theta;
206 B = (1 - cos(x: theta)) / theta2;
207 }
208 else { // 0 series
209 A = 1.0 - theta2 / 6.0;
210 B = 0.5 - theta2 / 24.0;
211 }
212 Matrix2 V;
213 V << A, -theta * B,
214 theta* B, A;
215 return V;
216 }
217
218 // general Sim(2) case
219 const double d2 = lambda2 + theta2;
220 if (d2 < 1e-15) // both tiny identity
221 return Matrix2::Identity();
222
223 // rotation scalars (unchanged)
224 double A, B, C;
225 if (theta2 > 1e-9) {
226 A = sin(x: theta) / theta;
227 B = (1 - cos(x: theta)) / theta2;
228 C = (1 - A) / theta2;
229 } else { // series
230 A = 1.0 - theta2 / 6.0;
231 B = 0.5 - theta2 / 24.0;
232 C = 1.0 / 6.0 - theta2 / 120.0;
233 }
234
235 const double alpha = lambda2 / (lambda2 + theta2);
236 const double s_inv = exp(x: -lambda);
237 const double X = alpha * (1 - s_inv) / lambda + (1 - alpha) * (A - lambda * B);
238 const double Y = alpha * (s_inv - 1 + lambda) / lambda2 + (1 - alpha) * (B - lambda * C);
239
240 Matrix2 V;
241 V << X, -theta * Y, theta* Y, X;
242 return V;
243}

245Vector4 Similarity2::Logmap(const Similarity2& S, //
246 OptionalJacobian<4, 4> Hm) {
247 const Vector1 w = Rot2::Logmap(r: S.R_);
248 const double lambda = log(x: S.s_);
249 // In Expmap, t = V * u -> in Logmap, u = V^{-1} * t
250 Vector4 result;
251 result << GetV(theta: w[0], lambda).inverse() * S.t_, w, lambda;
252 if (Hm) {
253 throw std::runtime_error("Similarity2::Logmap: derivative not implemented");
254 }
255 return result;
256}

258Similarity2 Similarity2::Expmap(const Vector4& v, //
259 OptionalJacobian<4, 4> Hm) {
260 const Vector2 u = v.head<2>();
261 const double theta = v[2];
262 const double lambda = v[3];
263 if (Hm) {
264 throw std::runtime_error("Similarity2::Expmap: derivative not implemented");
265 }
266 const Matrix2 V = GetV(theta, lambda);
267 return Similarity2(Rot2::Expmap(v: v.segment<1>(start: 2)), V * u, exp(x: lambda));
268}

270Matrix4 Similarity2::AdjointMap() const {
271 const Matrix2& R = R_.matrix();
272 const Point2& t = t_;
273 const double& s = s_;
274
275 Matrix4 Adj = Matrix4::Identity(); // Start with Identity
276
277 // Top-left 2x2 block: s * R
278 Adj.block<2, 2>(startRow: 0, startCol: 0) = s * R;
279
280 // Top-right coupling terms, derived from T*Hat(xi)*T_inv
281 // Column w.r.t 'w': maps to [-s*J*t]
282 Adj(0, 2) = s * t.y();
283 Adj(1, 2) = -s * t.x();
284
285 // Column w.r.t 'lambda': maps to [-s*t]
286 Adj(0, 3) = -s * t.x();
287 Adj(1, 3) = -s * t.y();
288
289 return Adj;
290}

292Matrix3 Similarity2::Hat(const Vector4 &xi) {
293 const auto w = xi[2];
294 const auto u = xi.head<2>();
295 const double lambda = xi[3];
296 Matrix3 W;
297 W << 0, -w, u[0],
298 w, 0, u[1],
299 0, 0, -lambda;
300 return W;
301}

303Vector4 Similarity2::Vee(const Matrix3 &Xi) {
304 Vector4 xi;
305 xi[2] = Xi(1, 0);
306 xi.head<2>() = Xi.topRightCorner<2, 1>();
307 xi[3] = -Xi(2, 2);
308 return xi;
309}

317Matrix3 Similarity2::matrix() const {
318 Matrix3 T;
319 T.topRows<2>() << R_.matrix(), t_;
320 T.bottomRows<1>() << 0, 0, 1.0 / s_;
321 return T;
