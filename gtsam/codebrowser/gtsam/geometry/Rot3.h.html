155 /// Rotations around Z, Y, then X axes as in http://en.wikipedia.org/wiki/Rotation_matrix, counterclockwise when looking from unchanging axis.
156 static Rot3 RzRyRx(double x, double y, double z,
157 OptionalJacobian<3, 1> Hx = {},
158 OptionalJacobian<3, 1> Hy = {},
159 OptionalJacobian<3, 1> Hz = {});
160
161 /// Rotations around Z, Y, then X axes as in http://en.wikipedia.org/wiki/Rotation_matrix, counterclockwise when looking from unchanging axis.
162 inline static Rot3 RzRyRx(const Vector& xyz,
163 OptionalJacobian<3, 3> H = {}) {
164#ifndef NDEBUG
165 if (xyz.size() != 3) {
166 throw;
167 }
168#endif
169 Rot3 out;
170 if (H) {
171 Vector3 Hx, Hy, Hz;
172 out = RzRyRx(x: xyz(0), y: xyz(1), z: xyz(2), Hx, Hy, Hz);
173 (*H) << Hx, Hy, Hz;
174 } else
175 out = RzRyRx(x: xyz(0), y: xyz(1), z: xyz(2));
176 return out;
177 }

179 /// Positive yaw is to right (as in aircraft heading). See ypr
180 static Rot3 Yaw (double t) { return Rz(t); }
181
182 /// Positive pitch is up (increasing aircraft altitude).See ypr
183 static Rot3 Pitch(double t) { return Ry(t); }
184
185 //// Positive roll is to right (increasing yaw in aircraft).
186 static Rot3 Roll (double t) { return Rx(t); }

188 /**
189 * Returns rotation nRb from body to nav frame.
190 * For vehicle coordinate frame X forward, Y right, Z down:
191 * Positive yaw is to right (as in aircraft heading).
192 * Positive pitch is up (increasing aircraft altitude).
193 * Positive roll is to right (increasing yaw in aircraft).
194 * Tait-Bryan system from Spatial Reference Model (SRM) (x,y,z) = (roll,pitch,yaw)
195 * as described in http://www.sedris.org/wg8home/Documents/WG80462.pdf.
196 *
197 * For vehicle coordinate frame X forward, Y left, Z up:
198 * Positive yaw is to left (as in aircraft heading).
199 * Positive pitch is down (decreasing aircraft altitude).
200 * Positive roll is to right (decreasing yaw in aircraft).
201 */
202 static Rot3 Ypr(double y, double p, double r,
203 OptionalJacobian<3, 1> Hy = {},
204 OptionalJacobian<3, 1> Hp = {},
205 OptionalJacobian<3, 1> Hr = {}) {
206 return RzRyRx(x: r, y: p, z: y, Hx: Hr, Hy: Hp, Hz: Hy);
207 }

215 /**
216 * Convert from axis/angle representation
217 * @param axis is the rotation axis, unit length
218 * @param angle rotation angle
219 * @return incremental rotation
220 */
221 static Rot3 AxisAngle(const Point3& axis, double angle) {
222 // Convert to unit vector.
223 Vector3 unitAxis = Unit3(axis).unitVector();
224#ifdef GTSAM_USE_QUATERNIONS
225 return gtsam::Quaternion(Eigen::AngleAxis<double>(angle, unitAxis));
226#else

268 /**
269 * Static, named constructor that finds Rot3 element closest to M in Frobenius norm.
270 *
271 * Uses Full SVD to compute the orthogonal matrix, thus is highly accurate and robust.
272 *
273 * N. J. Higham. Matrix nearness problems and applications.
274 * In M. J. C. Gover and S. Barnett, editors, Applications of Matrix Theory, pages 127.
275 * Oxford University Press, 1989.
276 */
277 static Rot3 ClosestTo(const Matrix3& M) { return Rot3(SO3::ClosestTo(M)); }

322 /**
323 * Conjugation: given a rotation acting in frame B, compute rotation c1Rc2 acting in a frame C
324 * @param cRb rotation from B frame to C frame
325 * @return c1Rc2 = cRb * b1Rb2 * cRb'
326 */
327 Rot3 conjugate(const Rot3& cRb) const {
328 // TODO: do more efficiently by using Eigen or quaternion properties
329 return cRb * (*this) * cRb.inverse();

336 /**
337 * The method retract() is used to map from the tangent space back to the manifold.
338 * Its inverse, is localCoordinates(). For Lie groups, an obvious retraction is the
339 * exponential map, but this can be expensive to compute. The following Enum is used
340 * to indicate which method should be used. The default
341 * is determined by ROT3_DEFAULT_COORDINATES_MODE, which may be set at compile time,
342 * and itself defaults to Rot3::CAYLEY, or if GTSAM_USE_QUATERNIONS is defined,
343 * to Rot3::EXPMAP.
344 */
345 enum CoordinatesMode {
346 EXPMAP, ///< Use the Lie group exponential map to retract
347#ifndef GTSAM_USE_QUATERNIONS
348 CAYLEY, ///< Retract and localCoordinates using the Cayley transform.
349#endif
350 };

354 // Cayley chart around origin
355 struct CayleyChart {
356 static Rot3 Retract(const Vector3& v, OptionalJacobian<3, 3> H = {});
357 static Vector3 Local(const Rot3& r, OptionalJacobian<3, 3> H = {});
358 };
359
360 /// Retraction from R^3 to Rot3 manifold using the Cayley transform
361 Rot3 retractCayley(const Vector& omega) const {
362 return compose(g: CayleyChart::Retract(v: omega));
363 }
364
365 /// Inverse of retractCayley
366 Vector3 localCayley(const Rot3& other) const {
367 return CayleyChart::Local(r: between(g: other));
368 }

582 /**
583 * [RQ] receives a 3 by 3 matrix and returns an upper triangular matrix R
584 * and 3 rotation angles corresponding to the rotation matrix Q=Qz'*Qy'*Qx'
585 * such that A = R*Q = R*Qz'*Qy'*Qx'. When A is a rotation matrix, R will
586 * be the identity and Q is a yaw-pitch-roll decomposition of A.
587 * The implementation uses Givens rotations and is based on Hartley-Zisserman.
588 * @param A 3 by 3 matrix A=RQ
589 * @return an upper triangular matrix R
590 * @return a vector [thetax, thetay, thetaz] in radians.
591 */

593 const Matrix3& A, OptionalJacobian<3, 9> H = {});
