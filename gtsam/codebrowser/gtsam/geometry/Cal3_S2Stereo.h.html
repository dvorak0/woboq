17
18#pragma once
19
20#include <gtsam/geometry/Cal3_S2.h>
21#include <iosfwd>
22
23namespace gtsam {
24
25/**
26 * @brief The most common 5DOF 3D->2D calibration, stereo version
27 * @ingroup geometry
28 * \nosubgrouping
29 */
30class GTSAM_EXPORT Cal3_S2Stereo : public Cal3_S2 {
31 private:
32 double b_ = 1.0f; ///< Stereo baseline.
33
34 public:
35 constexpr static auto dimension = 6;
36
37 ///< shared pointer to stereo calibration object
38 using shared_ptr = std::shared_ptr<Cal3_S2Stereo>;
39
40 /// @name Standard Constructors
41 /// @{
42
43 /// default calibration leaves coordinates unchanged
44 Cal3_S2Stereo() = default;
45
46 /// constructor from doubles
47 Cal3_S2Stereo(double fx, double fy, double s, double u0, double v0, double b)
48 : Cal3_S2(fx, fy, s, u0, v0), b_(b) {}
49
50 /// constructor from vector
51 Cal3_S2Stereo(const Vector6& d)
52 : Cal3_S2(d(0), d(1), d(2), d(3), d(4)), b_(d(5)) {}
53
54 /// easy constructor; field-of-view in degrees, assumes zero skew
55 Cal3_S2Stereo(double fov, int w, int h, double b)
56 : Cal3_S2(fov, w, h), b_(b) {}
57
58 /// @}
59
60 /**
61 * Convert intrinsic coordinates xy to image coordinates uv, fixed derivaitves
62 * @param p point in intrinsic coordinates
63 * @param Dcal optional 2*6 Jacobian wrpt Cal3_S2Stereo parameters
64 * @param Dp optional 2*2 Jacobian wrpt intrinsic coordinates
65 * @return point in image coordinates
66 */
67 Point2 uncalibrate(const Point2& p, OptionalJacobian<2, 6> Dcal = {},
68 OptionalJacobian<2, 2> Dp = {}) const;
69
70 /**
71 * Convert image coordinates uv to intrinsic coordinates xy
72 * @param p point in image coordinates
73 * @param Dcal optional 2*6 Jacobian wrpt Cal3_S2Stereo parameters
74 * @param Dp optional 2*2 Jacobian wrpt intrinsic coordinates
75 * @return point in intrinsic coordinates
76 */
77 Point2 calibrate(const Point2& p, OptionalJacobian<2, 6> Dcal = {},
78 OptionalJacobian<2, 2> Dp = {}) const;
79
80 /**
81 * Convert homogeneous image coordinates to intrinsic coordinates
82 * @param p point in image coordinates
83 * @return point in intrinsic coordinates
84 */
85 Vector3 calibrate(const Vector3& p) const { return Cal3_S2::calibrate(p); }
86
87 /// @name Testable
88 /// @{
89
90 /// Output stream operator
91 GTSAM_EXPORT friend std::ostream& operator<<(std::ostream& os,
92 const Cal3_S2Stereo& cal);
93
94 /// print with optional string
95 void print(const std::string& s = "") const override;
96
97 /// Check if equal up to specified tolerance
98 bool equals(const Cal3_S2Stereo& other, double tol = 10e-9) const;
99
100 /// @}
101 /// @name Standard Interface
102 /// @{
103
104 /// return calibration, same for left and right
105 const Cal3_S2& calibration() const { return *this; }
106
107 /// return calibration matrix K, same for left and right
108 Matrix3 K() const override { return Cal3_S2::K(); }
109
110 /// return baseline
111 double baseline() const { return b_; }
112
113 /// vectorized form (column-wise)
114 Vector6 vector() const {
115 Vector6 v;
116 v << Cal3_S2::vector(), b_;
117 return v;
118 }
119
120 /// @}
121 /// @name Manifold
122 /// @{
123
124 /// return DOF, dimensionality of tangent space
125 size_t dim() const override { return Dim(); }
126
127 /// return DOF, dimensionality of tangent space
128 static size_t Dim() { return dimension; }
129
130 /// Given 6-dim tangent vector, create new calibration
131 Cal3_S2Stereo retract(const Vector& d) const {
132 return Cal3_S2Stereo(fx() + d(0), fy() + d(1), skew() + d(2), px() + d(3),
133 py() + d(4), b_ + d(5));
134 }
135
136 /// Unretraction for the calibration
137 Vector6 localCoordinates(const Cal3_S2Stereo& T2) const {
138 return T2.vector() - vector();
139 }
140
141 /// @}
142 /// @name Advanced Interface
143 /// @{
144
145 private:
146#if GTSAM_ENABLE_BOOST_SERIALIZATION
147 /** Serialization function */
148 friend class boost::serialization::access;
149 template <class Archive>
150 void serialize(Archive& ar, const unsigned int /*version*/) {
151 ar& boost::serialization::make_nvp(
152 n: "Cal3_S2", v&: boost::serialization::base_object<Cal3_S2>(d&: *this));
153 ar& BOOST_SERIALIZATION_NVP(b_);
154 }
155#endif
156 /// @}
157};
158
159// Define GTSAM traits
160template <>
161struct traits<Cal3_S2Stereo> : public internal::Manifold<Cal3_S2Stereo> {};
162
163template <>
164struct traits<const Cal3_S2Stereo> : public internal::Manifold<Cal3_S2Stereo> {
165};
