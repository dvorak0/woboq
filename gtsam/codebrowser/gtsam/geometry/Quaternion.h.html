26#define QUATERNION_TYPE Eigen::Quaternion<_Scalar,_Options>
27
28namespace gtsam {
29
30// Define traits
31template<typename _Scalar, int _Options>
32struct traits<QUATERNION_TYPE> {
33 typedef QUATERNION_TYPE ManifoldType;
34 typedef QUATERNION_TYPE Q;
35
36 typedef lie_group_tag structure_category;
37 typedef multiplicative_group_tag group_flavor;
38
39 /// @name Group traits
40 /// @{
41 static Q Identity() {
42 return Q::Identity();
43 }
44
45 /// @}
46 /// @name Basic manifold traits
47 /// @{
48 inline constexpr static auto dimension = 3;
49 typedef OptionalJacobian<3, 3> ChartJacobian;
50 typedef Eigen::Matrix<_Scalar, 3, 1, _Options, 3, 1> TangentVector;
51
52 /// @}
53 /// @name Lie group traits
54 /// @{
55 static Q Compose(const Q &g, const Q & h,
56 ChartJacobian Hg = {}, ChartJacobian Hh = {}) {
57 if (Hg) *Hg = h.toRotationMatrix().transpose();
58 if (Hh) *Hh = I_3x3;
59 return g * h;
60 }
61
62 static Q Between(const Q &g, const Q & h,
63 ChartJacobian Hg = {}, ChartJacobian Hh = {}) {
64 Q d = g.inverse() * h;
65 if (Hg) *Hg = -d.toRotationMatrix().transpose();
66 if (Hh) *Hh = I_3x3;
67 return d;
68 }
69
70 static Q Inverse(const Q &g,
71 ChartJacobian H = {}) {
72 if (H) *H = -g.toRotationMatrix();
73 return g.inverse();
74 }
75
76 /// Exponential map, using the inlined code from Eigen's conversion from axis/angle
77 static Q Expmap(const Eigen::Ref<const TangentVector>& omega,
78 ChartJacobian H = {}) {
79 using std::cos;
80 using std::sin;
81 if (H) *H = SO3::ExpmapDerivative(omega: omega.template cast<double>());
82 _Scalar theta2 = omega.dot(omega);
83 if (theta2 > std::numeric_limits<_Scalar>::epsilon()) {
84 _Scalar theta = std::sqrt(theta2);
85 _Scalar ha = _Scalar(0.5) * theta;
86 Vector3 vec = (sin(ha) / theta) * omega;
87 return Q(cos(ha), vec.x(), vec.y(), vec.z());
88 } else {
89 // first order approximation sin(theta/2)/theta = 0.5
90 Vector3 vec = _Scalar(0.5) * omega;
91 return Q(1.0, vec.x(), vec.y(), vec.z());
92 }
93 }
94
95 /// We use our own Logmap, as there is a slight bug in Eigen
96 static TangentVector Logmap(const Q& q, ChartJacobian H = {}) {
97 using std::acos;
98 using std::sqrt;
99
100 // define these compile time constants to avoid std::abs:
101 static const double twoPi = 2.0 * M_PI, NearlyOne = 1.0 - 1e-10,
102 NearlyNegativeOne = -1.0 + 1e-10;
103
104 TangentVector omega;
105
106 const _Scalar qw = q.w();
107 // See Quaternion-Logmap.nb in doc for Taylor expansions
108 if (qw > NearlyOne) {
109 // Taylor expansion of (angle / s) at 1
110 // (2 + 2 * (1-qw) / 3) * q.vec();
111 omega = ( 8. / 3. - 2. / 3. * qw) * q.vec();
112 } else if (qw < NearlyNegativeOne) {
113 // Taylor expansion of (angle / s) at -1
114 // (-2 - 2 * (1 + qw) / 3) * q.vec();
115 omega = (-8. / 3. - 2. / 3. * qw) * q.vec();
116 } else {
117 // Normal, away from zero case
118 if (qw > 0) {
119 _Scalar angle = 2 * acos(qw), s = sqrt(1 - qw * qw);
120 // Important: convert to [-pi,pi] to keep error continuous
121 if (angle > M_PI)
122 angle -= twoPi;
123 else if (angle < -M_PI)
124 angle += twoPi;
125 omega = (angle / s) * q.vec();
126 } else {
127 // Make sure that we are using a canonical quaternion with w > 0
128 _Scalar angle = 2 * acos(-qw), s = sqrt(1 - qw * qw);
129 if (angle > M_PI)
130 angle -= twoPi;
131 else if (angle < -M_PI)
132 angle += twoPi;
133 omega = (angle / s) * -q.vec();
134 }
135 }
136
137 if(H) *H = SO3::LogmapDerivative(omega: omega.template cast<double>());
138 return omega;
139 }
140
141 static Matrix3 AdjointMap(const Q &g) {
142 return g.toRotationMatrix();
143 }
144
145 using LieAlgebra = Matrix3;
146
147 static Matrix3 Hat(const Vector3& v) {
148 return SO3::Hat(xi: v);
149 }
150
151 static Vector3 Vee(const Matrix3& X) {
152 return SO3::Vee(X);
153 }
154
155 static Vector9 Vec(const Q& q, OptionalJacobian<9, 3> H = {}) {
156 return SO3(q.toRotationMatrix()).SO3::vec(H);
157 }
158
159 /// @}
160 /// @name Manifold traits
161 /// @{
162
163 static TangentVector Local(const Q& g, const Q& h,
164 ChartJacobian H1 = {}, ChartJacobian H2 = {}) {
165 Q b = Between(g, h, Hg: H1, Hh: H2);
166 Matrix3 D_v_b;
167 TangentVector v = Logmap(q: b, H: (H1 || H2) ? &D_v_b : 0);
168 if (H1) *H1 = D_v_b * (*H1);
169 if (H2) *H2 = D_v_b * (*H2);
170 return v;
171 }
172
173 static Q Retract(const Q& g, const TangentVector& v,
174 ChartJacobian H1 = {}, ChartJacobian H2 = {}) {
175 Matrix3 D_h_v;
176 Q b = Expmap(omega: v,H: H2 ? &D_h_v : 0);
177 Q h = Compose(g, h: b, Hg: H1, Hh: H2);
178 if (H2) *H2 = (*H2) * D_h_v;
179 return h;
180 }
181
182 /// @}
183 /// @name Testable
184 /// @{
185 static void Print(const Q& q, const std::string& str = "") {
186 if (str.size() == 0)
187 std::cout << "Eigen::Quaternion: ";
188 else
189 std::cout << str << " ";
190 std::cout << q.vec().transpose() << std::endl;
191 }
192 static bool Equals(const Q& q1, const Q& q2, double tol = 1e-8) {
193 return Between(g: q1, h: q2).vec().array().abs().maxCoeff() < tol;
194 }
195 /// @}
196};
