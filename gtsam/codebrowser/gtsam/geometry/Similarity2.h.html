26namespace gtsam {
27
28// Forward declarations
29class Pose2;
30
31/**
32 * 2D similarity transform
33 */
34class GTSAM_EXPORT Similarity2 : public MatrixLieGroup<Similarity2, 4, 3> {
35 /// @name Pose Concept
36 /// @{
37 typedef Rot2 Rotation;
38 typedef Point2 Translation;
39 /// @}
40
41private:
42 Rot2 R_;
43 Point2 t_;
44 double s_;
45
46public:
47 /// @name Constructors
48 /// @{
49
50 /// Default constructor
51 Similarity2();
52
53 /// Construct pure scaling
54 Similarity2(double s);
55
56 /// Construct from GTSAM types
57 Similarity2(const Rot2& R, const Point2& t, double s);
58
59 /// Construct from Eigen types
60 Similarity2(const Matrix2& R, const Vector2& t, double s);
61
62 /// Construct from matrix [R t; 0 s^-1]
63 Similarity2(const Matrix3& T);
64
65 /// @}
66 /// @name Testable
67 /// @{
68
69 /// Compare with tolerance
70 bool equals(const Similarity2& sim, double tol) const;
71
72 /// Exact equality
73 bool operator==(const Similarity2& other) const;
74
75 /// Print with optional string
76 void print(const std::string& s = "") const;
77
78 GTSAM_EXPORT friend std::ostream& operator<<(std::ostream& os,
79 const Similarity2& p);
80
81 /// @}
82 /// @name Group
83 /// @{
84
85 /// Return an identity transform
86 static Similarity2 Identity();
87
88 /// Composition
89 Similarity2 operator*(const Similarity2& S) const;
90
91 /// Return the inverse
92 Similarity2 inverse() const;
93
94 /// @}
95 /// @name Group action on Point2
96 /// @{
97
98 /// Action on a point p is s*(R*p+t)
99 Point2 transformFrom(const Point2& p) const;
100
101 /**
102 * Action on a pose T.
103 * |Rs ts| |R t| |Rs*R Rs*t+ts|
104 * |0 1/s| * |0 1| = | 0 1/s |, the result is still a Sim2 object.
105 * To retrieve a Pose2, we normalized the scale value into 1.
106 * |Rs*R Rs*t+ts| |Rs*R s(Rs*t+ts)|
107 * | 0 1/s | = | 0 1 |
108 *
109 * This group action satisfies the compatibility condition.
110 * For more details, refer to: https://en.wikipedia.org/wiki/Group_action
111 */
112 Pose2 transformFrom(const Pose2& T) const;
113
114 /* syntactic sugar for transformFrom */
115 Point2 operator*(const Point2& p) const;
116
117 /**
118 * Create Similarity2 by aligning at least two point pairs
119 */
120 static Similarity2 Align(const Point2Pairs& abPointPairs);
121
122 /**
123 * Create the Similarity2 object that aligns at least two pose pairs.
124 * Each pair is of the form (aTi, bTi).
125 * Given a list of pairs in frame a, and a list of pairs in frame b,
126 Align()
127 * will compute the best-fit Similarity2 aSb transformation to align them.
128 * First, the rotation aRb will be computed as the average (Karcher mean)
129 of
130 * many estimates aRb (from each pair). Afterwards, the scale factor will
131 be computed
132 * using the algorithm described here:
133 * http://www5.informatik.uni-erlangen.de/Forschung/Publikationen/2005/Zinsser05-PSR.pdf
134 */
135 static Similarity2 Align(const Pose2Pairs& abPosePairs);
136
137 /// @}
138 /// @name Lie Group
139 /// @{
140
141 using LieAlgebra = Matrix3;
142
143 /// Calculate expmap and logmap coefficients.
144 static Matrix2 GetV(double theta, double lambda);
145
146 /**
147 * Log map at the identity
148 * \f$ [t_x, t_y, \delta, \lambda] \f$
149 */
150 static Vector4 Logmap(const Similarity2& S, //
151 OptionalJacobian<4, 4> Hm = {});
152
153 /// Exponential map at the identity
154 static Similarity2 Expmap(const Vector4& v, //
155 OptionalJacobian<4, 4> Hm = {});
156
157 /// Chart at the origin
158 struct ChartAtOrigin {
159 static Similarity2 Retract(const Vector4& v,
160 ChartJacobian H = {}) {
161 return Similarity2::Expmap(v, Hm: H);
162 }
163 static Vector4 Local(const Similarity2& other,
164 ChartJacobian H = {}) {
165 return Similarity2::Logmap(S: other, Hm: H);
166 }
167 };
168
169 /// Project from one tangent space to another
170 Matrix4 AdjointMap() const;
171
172 using LieGroup<Similarity2, 4>::inverse;
173
174 /// Hat maps from tangent vector to Lie algebra
175 static Matrix3 Hat(const Vector4& xi);
176
177 /// Vee maps from Lie algebra to tangent vector
178 static Vector4 Vee(const Matrix3& X);
179
180 /// @}
181 /// @name Standard interface
182 /// @{
183
184 /// Calculate 3*3 matrix group equivalent
185 Matrix3 matrix() const;
186
187 /// Return a GTSAM rotation
188 Rot2 rotation() const { return R_; }
189
190 /// Return a GTSAM translation
191 Point2 translation() const { return t_; }
192
193 /// Return the scale
194 double scale() const { return s_; }
195
196 private:
197
198 #if GTSAM_ENABLE_BOOST_SERIALIZATION
199 /** Serialization function */
200 friend class boost::serialization::access;
201 template<class Archive>
202 void serialize(Archive & ar, const unsigned int /*version*/) {
203 ar & BOOST_SERIALIZATION_NVP(R_);
204 ar & BOOST_SERIALIZATION_NVP(t_);
205 ar & BOOST_SERIALIZATION_NVP(s_);
206 }
207 #endif
208
209 /// @}
210};
211
