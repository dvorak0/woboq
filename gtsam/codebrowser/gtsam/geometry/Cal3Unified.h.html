30/**
31 * @brief Calibration of a omni-directional camera with mirror + lens radial
32 * distortion
33 * @ingroup geometry
34 * \nosubgrouping
35 *
36 * Similar to Cal3DS2, does distortion but has additional mirror parameter xi
37 * K = [ fx s u0 ; 0 fy v0 ; 0 0 1 ]
38 * Pn = [ P.x / (1 + xi * \sqrt{P.x² + P.y² + 1}), P.y / (1 + xi * \sqrt{P.x² +
39 * P.y² + 1})]
40 * r² = Pn.x² + Pn.y²
41 * \hat{pn} = (1 + k1*r² + k2*r⁴ ) pn + [ 2*k3 pn.x pn.y + k4 (r² + 2 Pn.x²) ;
42 * k3 (rr + 2 Pn.y²) + 2*k4 pn.x pn.y ]
43 * pi = K*pn
44 */
45class GTSAM_EXPORT Cal3Unified : public Cal3DS2_Base {
46 using This = Cal3Unified;
47 using Base = Cal3DS2_Base;
48
49 private:
50 double xi_ = 0.0f; ///< mirror parameter
51
52 public:
53 constexpr static auto dimension = 10;
54
55 ///< shared pointer to stereo calibration object
56 using shared_ptr = std::shared_ptr<Cal3Unified>;
57
58 /// @name Standard Constructors
59 /// @{
60
61 /// Default Constructor with only unit focal length
62 Cal3Unified() = default;
63
64 Cal3Unified(double fx, double fy, double s, double u0, double v0, double k1,
65 double k2, double p1 = 0.0, double p2 = 0.0, double xi = 0.0)
66 : Base(fx, fy, s, u0, v0, k1, k2, p1, p2), xi_(xi) {}
67
68 ~Cal3Unified() override {}
69
70 /// @}
71 /// @name Advanced Constructors
72 /// @{
73
74 Cal3Unified(const Vector10& v)
75 : Base(v(0), v(1), v(2), v(3), v(4), v(5), v(6), v(7), v(8)), xi_(v(9)) {}
76
77 /// @}
78 /// @name Testable
79 /// @{
80
81 /// Output stream operator
82 GTSAM_EXPORT friend std::ostream& operator<<(std::ostream& os,
83 const Cal3Unified& cal);
84
85 /// print with optional string
86 void print(const std::string& s = "") const override;
87
88 /// assert equality up to a tolerance
89 bool equals(const Cal3Unified& K, double tol = 10e-9) const;
90
91 /// @}
92 /// @name Standard Interface
93 /// @{
94
95 /// mirror parameter
96 double xi() const { return xi_; }
97
98 /// Return all parameters as a vector
99 Vector10 vector() const;
100
101 /**
102 * convert intrinsic coordinates xy to image coordinates uv
103 * @param p point in intrinsic coordinates
104 * @param Dcal optional 2*10 Jacobian wrpt Cal3Unified parameters
105 * @param Dp optional 2*2 Jacobian wrpt intrinsic coordinates
106 * @return point in image coordinates
107 */
108 Point2 uncalibrate(const Point2& p,
109 OptionalJacobian<2, 10> Dcal = {},
110 OptionalJacobian<2, 2> Dp = {}) const;
111
112 /// Conver a pixel coordinate to ideal coordinate
113 Point2 calibrate(const Point2& p, OptionalJacobian<2, 10> Dcal = {},
114 OptionalJacobian<2, 2> Dp = {}) const;
115
116 /// Convert a 3D point to normalized unit plane
117 Point2 spaceToNPlane(const Point2& p) const;
118
119 /// Convert a normalized unit plane point to 3D space
120 Point2 nPlaneToSpace(const Point2& p) const;
121
122 /// @}
123 /// @name Manifold
124 /// @{
125
126 /// Given delta vector, update calibration
127 Cal3Unified retract(const Vector& d) const;
128
129 /// Given a different calibration, calculate update to obtain it
130 Vector localCoordinates(const Cal3Unified& T2) const;
131
132 /// Return dimensions of calibration manifold object
133 size_t dim() const { return Dim(); }
134
135 /// Return dimensions of calibration manifold object
136 static size_t Dim() { return dimension; }
137
138 /// @}
139
140 private:
141#if GTSAM_ENABLE_BOOST_SERIALIZATION
142 /** Serialization function */
143 friend class boost::serialization::access;
144 template <class Archive>
145 void serialize(Archive& ar, const unsigned int /*version*/) {
146 ar& boost::serialization::make_nvp(
147 n: "Cal3Unified", v&: boost::serialization::base_object<Cal3DS2_Base>(d&: *this));
148 ar& BOOST_SERIALIZATION_NVP(xi_);
149 }
150#endif
151};
152

154struct traits<Cal3Unified> : public internal::Manifold<Cal3Unified> {};
155
156template <>
157struct traits<const Cal3Unified> : public internal::Manifold<Cal3Unified> {};
158}
