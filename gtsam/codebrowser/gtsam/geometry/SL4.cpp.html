18Eigen::Matrix<double, 16, 15> setVecToAlgMatrix() {
19 Eigen::Matrix<double, 16, 15> alg = Eigen::Matrix<double, 16, 15>::Zero();
20
21 // 12 Off-diagonal E_ij generators
22 int k = 0;
23 for (int i = 0; i < 4; ++i) {
24 for (int j = 0; j < 4; ++j) {
25 if (i != j) {
26 alg(i * 4 + j, k++) = 1.0;
27 }
28 }
29 }
30
31 // For Diagonal generators B1 = diag(1, -1, 0, 0)
32 alg(0, 12) = 1.0;
33 alg(5, 12) = -1.0;
34
35 // For B2 = diag(0, 1, -1, 0)
36 alg(5, 13) = 1.0;
37 alg(10, 13) = -1.0;
38
39 // For B3 = diag(0, 0, 1, -1)
40 alg(10, 14) = 1.0;
41 alg(15, 14) = -1.0;
42
43 return alg;
44}

71SL4::SL4(const Matrix44& pose) {
72 double det = pose.determinant();
73 if (det <= 0.0) {
74 throw std::runtime_error(
75 "Matrix determinant must be positive for SL(4) normalization. Got det "
76 "= " +
77 std::to_string(val: det));
78 }
79
80 T_ = pose / std::pow(x: det, y: 1.0 / 4.0);
81}

105SL4 SL4::Expmap(const Vector& xi, SL4Jacobian H) {
106 if (xi.size() != 15) {
107 throw std::runtime_error(
108 "SL4::Expmap: xi must be a vector of size 15. Got size " +
109 std::to_string(val: xi.size()));
110 }
111 const auto& A = Hat(xi);
112
113 if (H) throw std::runtime_error("SL4::Expmap: Jacobian not implemented.");
114
115 // NOTE(hlim):
116 // The cost of the computation is approximately 20n^3 for matrices of size n.
117 // The number 20 depends weakly on the norm of the matrix.
118 // See
119 // https://eigen.tuxfamily.org/dox/unsupported/group__MatrixFunctions__Module.html
120
121 return SL4(A.exp());
122}

130/* ************************************************************************* */
131Matrix15x15 SL4::AdjointMap() const {
132 Matrix44 H_inv_T = T_.inverse().transpose();
133 Matrix16x16 C_H;
134
135 // Kronecker product H H^{-T}
136 for (int i = 0; i < 4; ++i)
137 for (int j = 0; j < 4; ++j)
138 C_H.block<4, 4>(startRow: i * 4, startCol: j * 4) = T_(i, j) * H_inv_T;
139
140 return ALG_TO_VEC * C_H * VEC_TO_ALG;
141}

143/* ************************************************************************* */
144Matrix44 SL4::Hat(const Vector& xi) {
145 if (xi.size() != 15) {
146 throw std::runtime_error(
147 "SL4::Hat: xi must be a vector of size 15. Got size " +
148 std::to_string(val: xi.size()));
149 }
150 Matrix44 A;
151 const double d11 = xi(12);
152 const double d22 = -xi(12) + xi(13);
153 const double d33 = -xi(13) + xi(14);
154 const double d44 = -xi(14);
155
156 A << d11, xi(0), xi(1), xi(2), xi(3), d22, xi(4), xi(5), xi(6), xi(7), d33,
157 xi(8), xi(9), xi(10), xi(11), d44;
158
159 return A;
160}

163// NOTE(hlim): Why 'X'? - I just follow the convention of GTSAM
164Vector SL4::Vee(const Matrix44& X) {
165 Vector vec(15);
166 const double x12 = X(0, 0);
167 const double x13 = X(1, 1) + x12;
168 const double x14 = -X(3, 3);
169 vec << X(0, 1), X(0, 2), X(0, 3), X(1, 0), X(1, 2), X(1, 3), X(2, 0), X(2, 1),
170 X(2, 3), X(3, 0), X(3, 1), X(3, 2), x12, x13, x14;
171 return vec;
172}
