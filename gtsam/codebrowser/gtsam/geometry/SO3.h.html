131/**
132 * Opaque evaluation context at ω: caches Ω, Ω², θ, θ², nearZero/nearPi,
133 * Lazily computes C, D, E, dA, dB, dC, dE on demand.
134 * Math is based on Ethan Eade's elegant Lie group document, at
135 * https://www.ethaneade.org/lie.pdf, and the Kernel idea in doc/Jacobians.md
136 */
137struct GTSAM_EXPORT ExpmapFunctor {
138 double theta2; ///< The squared norm of the rotation vector (θ²).
139 double theta; ///< The norm of the rotation vector (θ).
140 Matrix3 W; ///< The skew-symmetric matrix Ω for the rotation vector.
141 Matrix3 WW; ///< The square of the skew-symmetric matrix (Ω²).
142 bool nearZero{false}; ///< Flag indicating if theta is near zero.
143
144 // Ethan Eade's constants:
145 double A; // A = sin(theta) / theta
146 double B; // B = (1 - cos(theta))
147
148 /// Constructor with element of Lie algebra so(3)
149 explicit ExpmapFunctor(const Vector3& omega);
150
151 /// Constructor with threshold (advanced)
152 ExpmapFunctor(double nearZeroThresholdSq, const Vector3& axis);
153
154 /// Constructor with axis-angle
155 ExpmapFunctor(const Vector3& axis, double angle);
156
157 /// Rodrigues formula
158 Matrix3 expmap() const;
159
160protected:
161 void init(double nearZeroThresholdSq);
162};
163
164/// Functor that implements Exponential map *and* its derivatives
165/// Math extends Ethan theme of elegant I + aW + bWW expressions.
166/// See https://www.ethaneade.org/lie.pdf expmap (82) and left Jacobian (83).
167struct GTSAM_EXPORT DexpFunctor : public ExpmapFunctor {
168 const Vector3 omega; ///< The rotation vector.
169 bool nearPi{false}; ///< Flag indicating if theta is near pi.
170
171 /// Constructor with element of Lie algebra so(3)
172 explicit DexpFunctor(const Vector3& omega);
173
174 /// Constructor with custom thresholds (advanced)
175 explicit DexpFunctor(const Vector3& omega, double nearZeroThresholdSq, double nearPiThresholdSq);
176
177 // Rodrigues kernel: R_[l/r](ω) = I + A(θ) Ω + B(θ) Ω² (left).
178 Kernel Rodrigues() const&;
179
180 // Jacobian kernel J_[l/r](ω) = I +/0 B Ω + C Ω² (left/right).
181 Kernel Jacobian() const&;
182
183 // Specialized kernel for inverse Jacobian, stable even for |ω| > π
184 InvJKernel InvJacobian() const&; // I +/- 1/2 Ω + D Ω²
185
186 // Gamma kernel: Γ_[l/r](ω) = 0.5 I ± C Ω + E Ω² (left/right).
187 Kernel Gamma() const&;
188
189 // NOTE(luca): Right Jacobian for Exponential map in SO(3)
190 // This maps a perturbation dxi=(w,v) in the tangent space to
191 // a perturbation on the manifold Expmap(dexp * xi)
192 Matrix3 rightJacobian() const;
193
194 // Compute the left Jacobian for Exponential map in SO(3)
195 Matrix3 leftJacobian() const;
196
197#ifdef GTSAM_ALLOW_DEPRECATED_SINCE_V43
198 /// @deprecated: use InvJacobian().right()
199 Matrix3 rightJacobianInverse() const;
200
201 /// @deprecated: use InvJacobian().left()
202 Matrix3 leftJacobianInverse() const;
203
204 /// @deprecated: use Jacobian().applyRight()
205 Vector3 applyRightJacobian(const Vector3& v,
206 OptionalJacobian<3, 3> H1 = {}, OptionalJacobian<3, 3> H2 = {}) const;
207
208 /// @deprecated: use InvJacobian().applyRight()
209 Vector3 applyRightJacobianInverse(const Vector3& v,
210 OptionalJacobian<3, 3> H1 = {}, OptionalJacobian<3, 3> H2 = {}) const;
211
212 /// @deprecated: use Jacobian().applyLeft()
213 Vector3 applyLeftJacobian(const Vector3& v,
214 OptionalJacobian<3, 3> H1 = {}, OptionalJacobian<3, 3> H2 = {}) const;
215
216 /// @deprecated: use InvJacobian().applyLeft()
217 Vector3 applyLeftJacobianInverse(const Vector3& v,
218 OptionalJacobian<3, 3> H1 = {}, OptionalJacobian<3, 3> H2 = {}) const;
219
220 /// @deprecated: use rightJacobian()
221 inline Matrix3 dexp() const { return rightJacobian(); }
222
223 /// @deprecated: use InvJacobian().right()
224 inline Matrix3 invDexp() const { return rightJacobianInverse(); }
225#endif
226
227 // access to (lazily evaluated) coefficients
228 double C() const;
229 double D() const;
230 double E() const;
231
232 // access to (lazily evaluated) radial derivatives c'(θ)/θ
233 double dA() const;
234 double dB() const;
235 double dC() const;
236 double dE() const;
237
238 protected:
239 // Lazy caches stored as std::optional
240 mutable std::optional<double> C_, D_, E_; ///< C, D and E lazily computed.
241 mutable std::optional<double> dA_, dB_, dC_, dE_; ///< lazy c(θ)′/θ
242};
