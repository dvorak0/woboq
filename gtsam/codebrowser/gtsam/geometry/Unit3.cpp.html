43/* ************************************************************************* */
44Unit3 Unit3::FromPoint3(const Point3& point, OptionalJacobian<2, 3> H) {
45 // 3*3 Derivative of representation with respect to point is 3*3:
46 Matrix3 D_p_point;
47 Unit3 direction;
48 direction.p_ = normalize(p: point, H: H ? &D_p_point : 0);
49 if (H)
50 *H << direction.basis().transpose() * D_p_point;
51 return direction;
52}

54/* ************************************************************************* */
55Unit3 Unit3::Random(std::mt19937& rng) {
56 // http://mathworld.wolfram.com/SpherePointPicking.html
57 // Adapted from implementation in boost, but using std <random>
58 std::uniform_real_distribution<double> uniform(-1.0, 1.0);
59 double sqsum;
60 double x, y;
61 do {
62 x = uniform(rng);
63 y = uniform(rng);
64 sqsum = x * x + y * y;
65 } while (sqsum > 1);
66 const double mult = 2 * sqrt(x: 1 - sqsum);
67 return Unit3(x * mult, y * mult, 2 * sqsum - 1);
68}

71// Get the axis of rotation with the minimum projected length of the point
72static Point3 CalculateBestAxis(const Point3& n) {
73 double mx = std::abs(x: n.x()), my = std::abs(x: n.y()), mz = std::abs(x: n.z());
74 if ((mx <= my) && (mx <= mz)) {
75 return Point3(1.0, 0.0, 0.0);
76 } else if ((my <= mx) && (my <= mz)) {
77 return Point3(0.0, 1.0, 0.0);
78 } else {
79 return Point3(0, 0, 1);
80 }
81}

84const Matrix32& Unit3::basis(OptionalJacobian<6, 2> H) const {
85#ifdef GTSAM_USE_TBB
86 // NOTE(hayk): At some point it seemed like this reproducably resulted in
87 // deadlock. However, I don't know why and I can no longer reproduce it.
88 // It either was a red herring or there is still a latent bug left to debug.
89 std::unique_lock<std::mutex> lock(B_mutex_);
90#endif
91
92 const bool cachedBasis = static_cast<bool>(B_);
93 const bool cachedJacobian = static_cast<bool>(H_B_);
94
95 if (H) {
96 if (!cachedJacobian) {
97 // Compute Jacobian. Recomputes B_
98 Matrix32 B;
99 Matrix62 jacobian;
100 Matrix33 H_B1_n, H_b1_B1, H_b2_n, H_b2_b1;
101
102 // Choose the direction of the first basis vector b1 in the tangent plane
103 // by crossing n with the chosen axis.
104 const Point3 n(p_), axis = CalculateBestAxis(n);
105 const Point3 B1 = gtsam::cross(p: n, q: axis, H_p: &H_B1_n);
106
107 // Normalize result to get a unit vector: b1 = B1 / |B1|.
108 B.col(i: 0) = normalize(p: B1, H: &H_b1_B1);
109
110 // Get the second basis vector b2, which is orthogonal to n and b1.
111 B.col(i: 1) = gtsam::cross(p: n, q: B.col(i: 0), H_p: &H_b2_n, H_q: &H_b2_b1);
112
113 // Chain rule tomfoolery to compute the jacobian.
114 const Matrix32& H_n_p = B;
115 jacobian.block<3, 2>(startRow: 0, startCol: 0) = H_b1_B1 * H_B1_n * H_n_p;
116 auto H_b1_p = jacobian.block<3, 2>(startRow: 0, startCol: 0);
117 jacobian.block<3, 2>(startRow: 3, startCol: 0) = H_b2_n * H_n_p + H_b2_b1 * H_b1_p;
118
119 // Cache the result and jacobian
120 H_B_ = (jacobian);
121 B_ = (B);
122 }
123
124 // Return cached jacobian, possibly computed just above
125 *H = *H_B_;
126 }
127
128 if (!cachedBasis) {
129 // Same calculation as above, without derivatives.
130 // Done after H block, as that possibly computes B_ for the first time
131 Matrix32 B;
132
133 const Point3 n(p_), axis = CalculateBestAxis(n);
134 const Point3 B1 = gtsam::cross(p: n, q: axis);
135 B.col(i: 0) = normalize(p: B1);
136 B.col(i: 1) = gtsam::cross(p: n, q: B.col(i: 0));
137 B_ = (B);
138 }
139
140 return *B_;
141}

173/* ************************************************************************* */
174double Unit3::dot(const Unit3& q, OptionalJacobian<1, 2> H_p,
175 OptionalJacobian<1, 2> H_q) const {
176 // Get the unit vectors of each, and the derivative.
177 Matrix32 H_pn_p;
178 Point3 pn = point3(H: H_p ? &H_pn_p : nullptr);
179
180 Matrix32 H_qn_q;
181 const Point3 qn = q.point3(H: H_q ? &H_qn_q : nullptr);
182
183 // Compute the dot product of the Point3s.
184 Matrix13 H_dot_pn, H_dot_qn;
185 double d = gtsam::dot(p: pn, q: qn, H_p: H_p ? &H_dot_pn : nullptr, H_q: H_q ? &H_dot_qn : nullptr);
186
187 if (H_p) {
188 (*H_p) << H_dot_pn * H_pn_p;
189 }
190
191 if (H_q) {
192 (*H_q) = H_dot_qn * H_qn_q;
193 }
194
195 return d;
196}

198/* ************************************************************************* */
199Vector2 Unit3::errorVector(const Unit3& q, OptionalJacobian<2, 2> H_p,
200 OptionalJacobian<2, 2> H_q) const {
201 // Get the point3 of this, and the derivative.
202 Matrix32 H_qn_q;
203 const Point3 qn = q.point3(H: H_q ? &H_qn_q : nullptr);
204
205 // 2D error here is projecting q into the tangent plane of this (p).
206 Matrix62 H_B_p;
207 Matrix23 Bt = basis(H: H_p ? &H_B_p : nullptr).transpose();
208 Vector2 xi = Bt * qn;
209
210 if (H_p) {
211 // Derivatives of each basis vector.
212 const Matrix32& H_b1_p = H_B_p.block<3, 2>(startRow: 0, startCol: 0);
213 const Matrix32& H_b2_p = H_B_p.block<3, 2>(startRow: 3, startCol: 0);
214
215 // dxi/dB = qn.transpose();
216 // Assemble dxi/dp = dxi/dB * dB/dp.
217 *H_p << qn.transpose() * H_b1_p, qn.transpose() * H_b2_p;
218 }
219
220 if (H_q) {
221 // dxi/dq is given by dxi/dqu * dqu/dq, where qu is the unit vector of q.
222 *H_q = Bt * H_qn_q;
223 }
224
225 return xi;
226}

229double Unit3::distance(const Unit3& q, OptionalJacobian<1, 2> H) const {
230 Matrix2 H_xi_q;
231 const Vector2 xi = errorVector(q, H_p: {}, H_q: H ? &H_xi_q : nullptr);
232 const double theta = xi.norm();
233 if (H)
234 *H = (xi.transpose() / theta) * H_xi_q;
235 return theta;
236}

239Unit3 Unit3::retract(const Vector2& v, OptionalJacobian<2,2> H) const {
240 // Compute the 3D xi_hat vector
241 const Vector3 xi_hat = basis() * v;
242 const double theta = xi_hat.norm();
243 const double c = std::cos(x: theta);
244
245 // Treat case of very small v differently.
246 Matrix23 H_from_point;
247 if (theta < std::numeric_limits<double>::epsilon()) {
248 const Unit3 exp_p_xi_hat = Unit3::FromPoint3(point: c * p_ + xi_hat,
249 H: H? &H_from_point : nullptr);
250 if (H) { // Jacobian
251 *H = H_from_point *
252 (-p_ * xi_hat.transpose() + Matrix33::Identity()) * basis();
253 }
254 return exp_p_xi_hat;
255 }
256
257 const double st = std::sin(x: theta) / theta;
258 const Unit3 exp_p_xi_hat = Unit3::FromPoint3(point: c * p_ + xi_hat * st,
259 H: H? &H_from_point : nullptr);
260 if (H) { // Jacobian
261 *H = H_from_point *
262 (p_ * -st * xi_hat.transpose() + st * Matrix33::Identity() +
263 xi_hat * ((c - st) / std::pow(x: theta, y: 2)) * xi_hat.transpose()) * basis();
264 }
265 return exp_p_xi_hat;
266}

268/* ************************************************************************* */
269Vector2 Unit3::localCoordinates(const Unit3& other) const {
270 const double x = p_.dot(other: other.p_);
271 // Crucial quantity here is y = theta/sin(theta) with theta=acos(x)
272 // Now, y = acos(x) / sin(acos(x)) = acos(x)/sqrt(1-x^2)
273 // We treat the special case 1 and -1 below
274 const double x2 = x * x;
275 const double z = 1 - x2;
276 double y;
277 if (z < std::numeric_limits<double>::epsilon()) {
278 if (x > 0) // first order expansion at x=1
279 y = 1.0 - (x - 1.0) / 3.0;
280 else // cop out
281 return Vector2(M_PI, 0.0);
282 } else {
283 // no special case
284 y = acos(x: x) / sqrt(x: z);
285 }
286 return basis().transpose() * y * (other.p_ - x * p_);
287}
