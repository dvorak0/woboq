20#pragma once
21
22#include <gtsam/geometry/Cal3.h>
23#include <gtsam/geometry/Point2.h>
24#include <memory>
25
26namespace gtsam {
27
28/**
29 * @brief Calibration of a camera with radial distortion
30 * @ingroup geometry
31 * \nosubgrouping
32 *
33 * Uses same distortionmodel as OpenCV, with
34 * http://docs.opencv.org/modules/calib3d/doc/camera_calibration_and_3d_reconstruction.html
35 * but using only k1,k2,p1, and p2 coefficients.
36 * K = [ fx s u0 ; 0 fy v0 ; 0 0 1 ]
37 * r² = P.x² + P.y²
38 * P̂ = (1 + k1*r² + k2*r⁴) P + [ (2*p1 P.x P.y) + p2 (r² + 2 Pn.x²)
39 * p1 (r² + 2 Pn.y²) + (2*p2 Pn.x Pn.y) ]
40 * pi = K*P̂
41 */
42class GTSAM_EXPORT Cal3DS2_Base : public Cal3 {
43 protected:
44 double k1_ = 0.0f, k2_ = 0.0f; ///< radial 2nd-order and 4th-order
45 double p1_ = 0.0f, p2_ = 0.0f; ///< tangential distortion
46 double tol_ = 1e-5; ///< tolerance value when calibrating
47
48 public:
49 constexpr static auto dimension = 9;
50
51 ///< shared pointer to stereo calibration object
52 using shared_ptr = std::shared_ptr<Cal3DS2_Base>;
53
54 /// @name Standard Constructors
55 /// @{
56
57 /// Default Constructor with only unit focal length
58 Cal3DS2_Base() = default;
59
60 Cal3DS2_Base(double fx, double fy, double s, double u0, double v0, double k1,
61 double k2, double p1 = 0.0, double p2 = 0.0, double tol = 1e-5)
62 : Cal3(fx, fy, s, u0, v0),
63 k1_(k1),
64 k2_(k2),
65 p1_(p1),
66 p2_(p2),
67 tol_(tol) {}
68
69 ~Cal3DS2_Base() override {}
70
71 /// @}
72 /// @name Advanced Constructors
73 /// @{
74
75 Cal3DS2_Base(const Vector9& v)
76 : Cal3(v(0), v(1), v(2), v(3), v(4)),
77 k1_(v(5)),
78 k2_(v(6)),
79 p1_(v(7)),
80 p2_(v(8)) {}
81
82 /// @}
83 /// @name Testable
84 /// @{
85
86 /// Output stream operator
87 GTSAM_EXPORT friend std::ostream& operator<<(std::ostream& os,
88 const Cal3DS2_Base& cal);
89
90 /// print with optional string
91 void print(const std::string& s = "") const override;
92
93 /// assert equality up to a tolerance
94 bool equals(const Cal3DS2_Base& K, double tol = 1e-8) const;
95
96 /// @}
97 /// @name Standard Interface
98 /// @{
99
100 /// First distortion coefficient
101 double k1() const { return k1_; }
102
103 /// Second distortion coefficient
104 double k2() const { return k2_; }
105
106 /// First tangential distortion coefficient
107 double p1() const { return p1_; }
108
109 /// Second tangential distortion coefficient
110 double p2() const { return p2_; }
111
112 /// return distortion parameter vector
113 Vector4 k() const { return Vector4(k1_, k2_, p1_, p2_); }
114
115 /// Return all parameters as a vector
116 Vector9 vector() const;
117
118 /**
119 * convert intrinsic coordinates xy to (distorted) image coordinates uv
120 * @param p point in intrinsic coordinates
121 * @param Dcal optional 2*9 Jacobian wrpt Cal3DS2 parameters
122 * @param Dp optional 2*2 Jacobian wrpt intrinsic coordinates
123 * @return point in (distorted) image coordinates
124 */
125 Point2 uncalibrate(const Point2& p, OptionalJacobian<2, 9> Dcal = {},
126 OptionalJacobian<2, 2> Dp = {}) const;
127
128 /// Convert (distorted) image coordinates uv to intrinsic coordinates xy
129 Point2 calibrate(const Point2& p, OptionalJacobian<2, 9> Dcal = {},
130 OptionalJacobian<2, 2> Dp = {}) const;
131
132 /// Derivative of uncalibrate wrpt intrinsic coordinates
133 Matrix2 D2d_intrinsic(const Point2& p) const;
134
135 /// Derivative of uncalibrate wrpt the calibration parameters
136 Matrix29 D2d_calibration(const Point2& p) const;
137
138 /// return DOF, dimensionality of tangent space
139 size_t dim() const { return Dim(); }
140
141 /// return DOF, dimensionality of tangent space
142 static size_t Dim() { return dimension; }
143
144 /// @}
145 /// @name Clone
146 /// @{
147
148 /// @return a deep copy of this object
149 virtual std::shared_ptr<Cal3DS2_Base> clone() const {
150 return std::shared_ptr<Cal3DS2_Base>(new Cal3DS2_Base(*this));
151 }
152
153 /// @}
154
155 private:
156 /// @name Advanced Interface
157 /// @{
158
159#if GTSAM_ENABLE_BOOST_SERIALIZATION
160 /** Serialization function */
161 friend class boost::serialization::access;
162 template <class Archive>
163 void serialize(Archive& ar, const unsigned int /*version*/) {
164 ar& boost::serialization::make_nvp(
165 n: "Cal3DS2_Base", v&: boost::serialization::base_object<Cal3>(d&: *this));
166 ar& BOOST_SERIALIZATION_NVP(k1_);
167 ar& BOOST_SERIALIZATION_NVP(k2_);
168 ar& BOOST_SERIALIZATION_NVP(p1_);
169 ar& BOOST_SERIALIZATION_NVP(p2_);
170 ar& BOOST_SERIALIZATION_NVP(tol_);
171 }
172#endif
173
174 /// @}
175};
176}
