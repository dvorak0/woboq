30struct DexpFunctor;
31
32/**
33 * Kernel: M(ω) = a I + b Ω + c Ω² with radial derivatives db,dc for Fréchet.
34 * Right variants flip b→-b, db→-db (no recompute of Ω/Ω²).
35 * Keep a pointer to Local: Kernel methods above return a const & to prevent
36 * having a pointer to a temporary.
37 */
38struct GTSAM_EXPORT Kernel {
39 const DexpFunctor* S;
40 double a{0}, b{0}, c{0}, db{0}, dc{0}; // left-specialization form
41
42 Matrix3 left() const; // a I + b Ω + c Ω²
43 Matrix3 right() const; // a I - b Ω + c Ω²
44
45 Vector3 applyLeft(const Vector3& v, OptionalJacobian<3, 3> Hw = {},
46 OptionalJacobian<3, 3> Hv = {}) const;
47 Vector3 applyRight(const Vector3& v, OptionalJacobian<3, 3> Hw = {},
48 OptionalJacobian<3, 3> Hv = {}) const;
49
50 /// Fréchet derivative of left-kernel M(ω) in the direction X ∈ so(3)
51 /// L_M(Ω)[X] = b X + c (Ω X + X Ω) + s (db Ω + dc Ω²), with s = -½ tr(Ω X)
52 Matrix3 frechet(const Matrix3& X) const;
53 /// Apply Fréchet derivative to vector (left specialization)
54 Matrix3 applyFrechet(const Vector3& v) const;
55};
56
57// Stable inverse Jacobian kernel
58struct GTSAM_EXPORT InvJKernel {
59 const DexpFunctor* S;
60 Kernel J; // holds the forward kernel
61
62 Matrix3 left() const;
63 Matrix3 right() const;
64
65 Vector3 applyLeft(const Vector3& v, OptionalJacobian<3, 3> Hw = {},
66 OptionalJacobian<3, 3> Hv = {}) const;
67 Vector3 applyRight(const Vector3& v, OptionalJacobian<3, 3> Hw = {},
68 OptionalJacobian<3, 3> Hv = {}) const;
69};
70
71/// y + alpha * x (functional)
72GTSAM_EXPORT Kernel axpy(double alpha, const Kernel& X, const Kernel& Y);
73
74// Blend K = α X + (1-α) Y with radial derivative (·)'/θ via dalpha
75GTSAM_EXPORT Kernel blend(double alpha, double dalpha, const Kernel& X,
76 const Kernel& Y);
