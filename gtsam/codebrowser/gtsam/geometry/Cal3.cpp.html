26/* ************************************************************************* */
27Cal3::Cal3(double fov, int w, int h)
28 : s_(0), u0_((double)w / 2.0), v0_((double)h / 2.0) {
29 double a = fov * M_PI / 360.0; // fov/2 in radians
30 fx_ = double(w) / (2.0 * tan(x: a));
31 fy_ = fx_;
32}
33
34/* ************************************************************************* */
35Cal3::Cal3(const std::string& path) {
36 const auto buffer = path + std::string("/calibration_info.txt");
37 std::ifstream infile(buffer, std::ios::in);
38
39 if (infile && !infile.eof()) {
40 infile >> fx_ >> fy_ >> s_ >> u0_ >> v0_;
41 } else {
42 throw std::runtime_error("Cal3: Unable to load the calibration");
43 }
44
45 infile.close();
46}

48/* ************************************************************************* */
49std::ostream& operator<<(std::ostream& os, const Cal3& cal) {
50 os << "fx: " << cal.fx() << ", fy: " << cal.fy() << ", s: " << cal.skew()
51 << ", px: " << cal.px() << ", py: " << cal.py();
52 return os;
53}

55/* ************************************************************************* */
56void Cal3::print(const std::string& s) const { gtsam::print(A: (Matrix)K(), s); }
57
58/* ************************************************************************* */
59bool Cal3::equals(const Cal3& K, double tol) const {
60 return (std::fabs(x: fx_ - K.fx_) < tol && std::fabs(x: fy_ - K.fy_) < tol &&
61 std::fabs(x: s_ - K.s_) < tol && std::fabs(x: u0_ - K.u0_) < tol &&
62 std::fabs(x: v0_ - K.v0_) < tol);
63}
64
65Matrix3 Cal3::inverse() const {
66 const double fxy = fx_ * fy_, sv0 = s_ * v0_, fyu0 = fy_ * u0_;
67 Matrix3 K_inverse;
68 K_inverse << 1.0 / fx_, -s_ / fxy, (sv0 - fyu0) / fxy, 0.0, 1.0 / fy_,
69 -v0_ / fy_, 0.0, 0.0, 1.0;
70 return K_inverse;
71}
