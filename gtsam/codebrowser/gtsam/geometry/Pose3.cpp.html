39/* ************************************************************************* */
40Pose3 Pose3::Create(const Rot3& R, const Point3& t, OptionalJacobian<6, 3> HR,
41 OptionalJacobian<6, 3> Ht) {
42 if (HR) *HR << I_3x3, Z_3x3;
43 if (Ht) *Ht << Z_3x3, R.transpose();
44 return Pose3(R, t);

68// Calculate Adjoint map
69// Ad_pose is 6*6 matrix that when applied to twist xi, returns Ad_pose(xi)
70Matrix6 Pose3::AdjointMap() const {
71 const Matrix3 R = R_.matrix();
72 Matrix3 A = skewSymmetric(wx: t_.x(), wy: t_.y(), wz: t_.z()) * R;
73 Matrix6 adj;
74 adj << R, Z_3x3, A, R; // Gives [R 0; A R]
75 return adj;
76}

78/* ************************************************************************* */
79// Calculate AdjointMap applied to xi_b, with Jacobians
80Vector6 Pose3::Adjoint(const Vector6& xi_b, OptionalJacobian<6, 6> H_pose,
81 OptionalJacobian<6, 6> H_xib) const {
82 const Matrix6 Ad = AdjointMap();
83
84 // Jacobians
85 // D1 Ad_T(xi_b) = D1 Ad_T Ad_I(xi_b) = Ad_T * D1 Ad_I(xi_b) = Ad_T * ad_xi_b
86 // D2 Ad_T(xi_b) = Ad_T
87 // See docs/math.pdf for more details.
88 // In D1 calculation, we could be more efficient by writing it out, but do not
89 // for readability
90 if (H_pose) *H_pose = -Ad * adjointMap(xi: xi_b);
91 if (H_xib) *H_xib = Ad;
92
93 return Ad * xi_b;
94}

96/* ************************************************************************* */
97/// The dual version of Adjoint
98Vector6 Pose3::AdjointTranspose(const Vector6& x, OptionalJacobian<6, 6> H_pose,
99 OptionalJacobian<6, 6> H_x) const {
100 const Matrix6 Ad = AdjointMap();
101 const Vector6 AdTx = Ad.transpose() * x;
102
103 // Jacobians
104 // See docs/math.pdf for more details.
105 if (H_pose) {
106 const auto w_T_hat = skewSymmetric(w: AdTx.head<3>()),
107 v_T_hat = skewSymmetric(w: AdTx.tail<3>());
108 *H_pose << w_T_hat, v_T_hat, //
109 /* */ v_T_hat, Z_3x3;
110 }
111 if (H_x) {
112 *H_x = Ad.transpose();
113 }
114
115 return AdTx;
116}

118/* ************************************************************************* */
119Matrix6 Pose3::adjointMap(const Vector6& xi) {
120 Matrix3 w_hat = skewSymmetric(wx: xi(0), wy: xi(1), wz: xi(2));
121 Matrix3 v_hat = skewSymmetric(wx: xi(3), wy: xi(4), wz: xi(5));
122 Matrix6 adj;
123 adj << w_hat, Z_3x3, v_hat, w_hat;
124
125 return adj;
126}

128/* ************************************************************************* */
129Vector6 Pose3::adjoint(const Vector6& xi, const Vector6& y,
130 OptionalJacobian<6, 6> Hxi, OptionalJacobian<6, 6> H_y) {
131 if (Hxi) {
132 Hxi->setZero();
133 for (int i = 0; i < 6; ++i) {
134 Vector6 dxi;
135 dxi.setZero();
136 dxi(i) = 1.0;
137 Matrix6 Gi = adjointMap(xi: dxi);
138 Hxi->col(i) = Gi * y;
139 }
140 }
141 const Matrix6& ad_xi = adjointMap(xi);
142 if (H_y) *H_y = ad_xi;
143 return ad_xi * y;
144}

146/* ************************************************************************* */
147Vector6 Pose3::adjointTranspose(const Vector6& xi, const Vector6& y,
148 OptionalJacobian<6, 6> Hxi, OptionalJacobian<6, 6> H_y) {
149 if (Hxi) {
150 Hxi->setZero();
151 for (int i = 0; i < 6; ++i) {
152 Vector6 dxi;
153 dxi.setZero();
154 dxi(i) = 1.0;
155 Matrix6 GTi = adjointMap(xi: dxi).transpose();
156 Hxi->col(i) = GTi * y;
157 }
158 }
159 const Matrix6& adT_xi = adjointMap(xi).transpose();
160 if (H_y) *H_y = adT_xi;
161 return adT_xi * y;
162}

189/* ************************************************************************* */
190Pose3 Pose3::interpolateRt(const Pose3& T, double t,
191 OptionalJacobian<6, 6> Hself,
192 OptionalJacobian<6, 6> Harg,
193 OptionalJacobian<6, 1> Ht) const {
194 if(Hself || Harg || Ht){
195 typename MakeJacobian<Rot3, Rot3>::type HselfRot, HargRot;
196 typename MakeJacobian<Rot3, double>::type HtRot;
197 typename MakeJacobian<Point3, Point3>::type HselfPoint, HargPoint;
198 typename MakeJacobian<Point3, double>::type HtPoint;
199
200 Rot3 Rint = interpolate<Rot3>(X: R_, Y: T.R_, t, Hx: HselfRot, Hy: HargRot, Ht: HtRot);
201 Point3 Pint = interpolate<Point3>(X: t_, Y: T.t_, t, Hx: HselfPoint, Hy: HargPoint, Ht: HtPoint);
202 Pose3 result = Pose3(Rint, Pint);
203
204 if(Hself) *Hself << HselfRot, Z_3x3, Z_3x3, Rint.transpose() * R_.matrix() * HselfPoint;
205 if(Harg) *Harg << HargRot, Z_3x3, Z_3x3, Rint.transpose() * T.R_.matrix() * HargPoint;
206 if(Ht) *Ht << HtRot, Rint.transpose() * HtPoint;
207
208 return result;
209 }
210 return Pose3(interpolate<Rot3>(X: R_, Y: T.R_, t),
211 interpolate<Point3>(X: t_, Y: T.t_, t));
212
213}

216// Expmap is implemented in so3::ExpmapFunctor::expmap, based on Ethan Eade's
217// elegant Lie group document, at https://www.ethaneade.org/lie.pdf.
218// See also [this document](doc/Jacobians.md)
219Pose3 Pose3::Expmap(const Vector6& xi, OptionalJacobian<6, 6> Hxi) {
220 // Get angular velocity omega and translational velocity v from twist xi
221 const Vector3 w = xi.head<3>(), v = xi.tail<3>();
222
223 // Instantiate functor for Dexp-related operations:
224 const so3::DexpFunctor local(w);
225
226 // Compute rotation using Expmap
227#ifdef GTSAM_USE_QUATERNIONS
228 const Rot3 R = traits<gtsam::Quaternion>::Expmap(w);
229#else
230 const Rot3 R(local.expmap());
231#endif
232
233 // The translation t = local.Jacobian().left() * v.
234 // Here we call local.Jacobian().applyLeft, which is faster if you don't need
235 // Jacobians, and returns Jacobian of t with respect to w if asked.
236 // NOTE(Frank): this does the same as the intuitive formulas:
237 // t_parallel = w * w.dot(v); // translation parallel to axis
238 // w_cross_v = w.cross(v); // translation orthogonal to axis
239 // t = (w_cross_v - Rot3::Expmap(w) * w_cross_v + t_parallel) / theta2;
240 // but Local does not need R, deals automatically with the case where theta2
241 // is near zero, and also gives us the machinery for the Jacobians.
242 Matrix3 H;
243 const Vector3 t = local.Jacobian().applyLeft(v, Hw: Hxi ? &H : nullptr);
244
245 if (Hxi) {
246 // The Jacobian of expmap is given by the right Jacobian of SO(3):
247 const Matrix3 Jr = local.Jacobian().right();
248 // We are creating a Pose3, so we still need to chain H with R^T, the
249 // Jacobian of Pose3::Create with respect to t.
250 const Matrix3 Rt = R.transpose();
251 *Hxi << Jr, Z_3x3, // Jr here *is* the Jacobian of expmap
252 Rt * H, Jr; // Here Jr = R^T * Jl, with Jl the Jacobian of t in v.
253 }
254
255 return Pose3(R, t);
256}

258/* ************************************************************************* */
259Vector6 Pose3::Logmap(const Pose3& pose, OptionalJacobian<6, 6> Hpose) {
260 const Vector3 w = Rot3::Logmap(R: pose.rotation());
261
262 // Instantiate functor for Dexp-related operations:
263 const so3::DexpFunctor local(w);
264
265 const Vector3 t = pose.translation();
266 const Vector3 u = local.InvJacobian().applyLeft(v: t);
267 Vector6 xi;
268 xi << w, u;
269 if (Hpose) *Hpose = LogmapDerivative(xi);
270 return xi;
271}

274Pose3 Pose3::ChartAtOrigin::Retract(const Vector6& xi, ChartJacobian Hxi) {

289Vector6 Pose3::ChartAtOrigin::Local(const Pose3& pose, ChartJacobian Hpose) {

302#endif

312/* ************************************************************************* */
313Matrix6 Pose3::LogmapDerivative(const Vector6& xi) {
314 const Vector3 w = xi.head<3>();
315 Vector3 v = xi.segment<3>(start: 3);
316
317 // Instantiate functor for Dexp-related operations:
318 const so3::DexpFunctor local(w);
319
320 // Call applyLeftJacobian to get its Jacobians
321 Matrix3 H_t_w;
322 local.Jacobian().applyLeft(v, Hw: H_t_w);
323
324 // Multiply with R^T to account for NavState::Create Jacobian.
325 const Matrix3 R = local.expmap();
326 const Matrix3 Qt = R.transpose() * H_t_w;
327
328 // Now compute the blocks of the LogmapDerivative Jacobian
329 const Matrix3 Jw = Rot3::LogmapDerivative(x: w);
330 const Matrix3 Qt2 = -Jw * Qt * Jw;
331
332 Matrix6 J;
333 J << Jw, Z_3x3, Qt2, Jw;
334 return J;
335}

365/* ************************************************************************* */
366Pose3::Vector16 Pose3::vec(OptionalJacobian<16, 6> H) const {
367 // Vectorize
368 const Matrix4 M = matrix();
369 const Vector16 v = Eigen::Map<const Vector16>(M.data());
370
371 // If requested, calculate H
372 if (H) {
373 H->setZero();
374 auto R = M.block<3, 3>(startRow: 0, startCol: 0);
375 H->block<3, 1>(startRow: 0, startCol: 1) = -R.col(i: 2);
376 H->block<3, 1>(startRow: 0, startCol: 2) = R.col(i: 1);
377 H->block<3, 1>(startRow: 4, startCol: 0) = R.col(i: 2);
378 H->block<3, 1>(startRow: 4, startCol: 2) = -R.col(i: 0);
379 H->block<3, 1>(startRow: 8, startCol: 0) = -R.col(i: 1);
380 H->block<3, 1>(startRow: 8, startCol: 1) = R.col(i: 0);
381 H->block<3,3>(startRow: 12,startCol: 3) = R;
382 }
383
384 return v;
385}

403/* ************************************************************************* */
404Point3 Pose3::transformFrom(const Point3& point, OptionalJacobian<3, 6> Hself,
405 OptionalJacobian<3, 3> Hpoint) const {
406 // Only get matrix once, to avoid multiple allocations,
407 // as well as multiple conversions in the Quaternion case
408 const Matrix3 R = R_.matrix();
409 if (Hself) {
410 Hself->leftCols<3>() = R * skewSymmetric(wx: -point.x(), wy: -point.y(), wz: -point.z());
411 Hself->rightCols<3>() = R;
412 }
413 if (Hpoint) {
414 *Hpoint = R;
415 }
416 return R_ * point + t_;
417}

419Matrix Pose3::transformFrom(const Matrix& points) const {
420 if (points.rows() != 3) {
421 throw std::invalid_argument("Pose3:transformFrom expects 3*N matrix.");
422 }
423 const Matrix3 R = R_.matrix();
424 return (R * points).colwise() + t_; // Eigen broadcasting!
425}

427/* ************************************************************************* */
428Point3 Pose3::transformTo(const Point3& point, OptionalJacobian<3, 6> Hself,
429 OptionalJacobian<3, 3> Hpoint) const {
430 // Only get transpose once, to avoid multiple allocations,
431 // as well as multiple conversions in the Quaternion case
432 const Matrix3 Rt = R_.transpose();
433 const Point3 q(Rt*(point - t_));
434 if (Hself) {
435 const double wx = q.x(), wy = q.y(), wz = q.z();
436 (*Hself) <<
437 0.0, -wz, +wy,-1.0, 0.0, 0.0,
438 +wz, 0.0, -wx, 0.0,-1.0, 0.0,
439 -wy, +wx, 0.0, 0.0, 0.0,-1.0;
440 }
441 if (Hpoint) {
442 *Hpoint = Rt;
443 }
444 return q;
445}

447Matrix Pose3::transformTo(const Matrix& points) const {
448 if (points.rows() != 3) {
449 throw std::invalid_argument("Pose3:transformTo expects 3*N matrix.");
450 }
451 const Matrix3 Rt = R_.transpose();
452 return Rt * (points.colwise() - t_); // Eigen broadcasting!

455/* ************************************************************************* */
456double Pose3::range(const Point3& point, OptionalJacobian<1, 6> Hself,
457 OptionalJacobian<1, 3> Hpoint) const {
458 Matrix36 D_local_pose;
459 Matrix3 D_local_point;
460 Point3 local = transformTo(point, Hself: Hself ? &D_local_pose : 0, Hpoint: Hpoint ? &D_local_point : 0);
461 if (!Hself && !Hpoint) {
462 return local.norm();
463 } else {
464 Matrix13 D_r_local;
465 const double r = norm3(p: local, H: D_r_local);
466 if (Hself) *Hself = D_r_local * D_local_pose;
467 if (Hpoint) *Hpoint = D_r_local * D_local_point;
468 return r;
469 }
470}

484Unit3 Pose3::bearing(const Point3& point, OptionalJacobian<2, 6> Hself,
485 OptionalJacobian<2, 3> Hpoint) const {

496 return b;

511/* ************************************************************************* */
512std::optional<Pose3> Pose3::Align(const Point3Pairs &abPointPairs) {
513 const size_t n = abPointPairs.size();
514 if (n < 3) {
515 return {}; // we need at least three pairs
516 }
517
518 // calculate centroids
519 const auto centroids = means(abPointPairs);
520
521 // Add to form H matrix
522 Matrix3 H = Z_3x3;
523 for (const Point3Pair &abPair : abPointPairs) {
524 const Point3 da = abPair.first - centroids.first;
525 const Point3 db = abPair.second - centroids.second;
526 H += da * db.transpose();
527 }
528
529 // ClosestTo finds rotation matrix closest to H in Frobenius sense
530 const Rot3 aRb = Rot3::ClosestTo(M: H);
531 const Point3 aTb = centroids.first - aRb * centroids.second;
532 return Pose3(aRb, aTb);
533}

535std::optional<Pose3> Pose3::Align(const Matrix& a, const Matrix& b) {
536 if (a.rows() != 3 || b.rows() != 3 || a.cols() != b.cols()) {
537 throw std::invalid_argument(
538 "Pose3:Align expects 3*N matrices of equal shape.");
539 }
540 Point3Pairs abPointPairs;
541 for (Eigen::Index j = 0; j < a.cols(); j++) {
542 abPointPairs.emplace_back(args: a.col(i: j), args: b.col(i: j));
543 }
544 return Pose3::Align(abPointPairs);
545}
