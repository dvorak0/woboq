27/**
28 * A pinhole camera class that has a Pose3 and a *fixed* Calibration.
29 * @ingroup geometry
30 * \nosubgrouping
31 */
32template<typename CALIBRATION>
33class PinholeBaseK: public PinholeBase {
34
35private:
36
37 GTSAM_CONCEPT_MANIFOLD_TYPE(CALIBRATION)
38
39 // Get dimensions of calibration type at compile time
40 static const int DimK = FixedDimension<CALIBRATION>::value;
41
42public:
43
44 typedef CALIBRATION CalibrationType;
45
46 /// @name Standard Constructors
47 /// @{
48
49 /** default constructor */
50 PinholeBaseK() {
51 }
52
53 /** constructor with pose */
54 explicit PinholeBaseK(const Pose3& pose) :
55 PinholeBase(pose) {
56 }
57
58 /// @}
59 /// @name Advanced Constructors
60 /// @{
61
62 explicit PinholeBaseK(const Vector& v) : PinholeBase(v) {}
63
64 /// @}
65 /// @name Standard Interface
66 /// @{
67
68 /// return calibration
69 virtual const CALIBRATION& calibration() const = 0;
70
71 /// @}
72 /// @name Transformations and measurement functions
73 /// @{
74
75 /// Project a point into the image and check depth
76 std::pair<Point2, bool> projectSafe(const Point3& pw) const {
77 std::pair<Point2, bool> pn = PinholeBase::projectSafe(pw);
78 pn.first = calibration().uncalibrate(pn.first);
79 return pn;
80 }
81
82
83 /** Templated projection of a point (possibly at infinity) from world coordinate to the image
84 * @param pw is a 3D point or a Unit3 (point at infinity) in world coordinates
85 * @param Dpose is the Jacobian w.r.t. pose3
86 * @param Dpoint is the Jacobian w.r.t. point3
87 * @param Dcal is the Jacobian w.r.t. calibration
88 */
89 template <class POINT>
90 Point2 _project(const POINT& pw, OptionalJacobian<2, 6> Dpose,
91 OptionalJacobian<2, FixedDimension<POINT>::value> Dpoint,
92 OptionalJacobian<2, DimK> Dcal) const {
93
94 // project to normalized coordinates
95 const Point2 pn = PinholeBase::project2(pw, Dpose, Dpoint);
96
97 // uncalibrate to pixel coordinates
98 Matrix2 Dpi_pn;
99 const Point2 pi = calibration().uncalibrate(pn, Dcal,
100 Dpose || Dpoint ? &Dpi_pn : 0);
101
102 // If needed, apply chain rule
103 if (Dpose)
104 *Dpose = Dpi_pn * *Dpose;
105 if (Dpoint)
106 *Dpoint = Dpi_pn * *Dpoint;
107
108 return pi;
109 }
110
111 /// project a 3D point from world coordinates into the image
112 Point2 project(const Point3& pw, OptionalJacobian<2, 6> Dpose = {},
113 OptionalJacobian<2, 3> Dpoint = {},
114 OptionalJacobian<2, DimK> Dcal = {}) const {
115 return _project(pw, Dpose, Dpoint, Dcal);
116 }
117
118 /// project a 3D point from world coordinates into the image
119 Point2 reprojectionError(const Point3& pw, const Point2& measured, OptionalJacobian<2, 6> Dpose = {},
120 OptionalJacobian<2, 3> Dpoint = {},
121 OptionalJacobian<2, DimK> Dcal = {}) const {
122 return Point2(_project(pw, Dpose, Dpoint, Dcal) - measured);
123 }
124
125 /// project a point at infinity from world coordinates into the image
126 Point2 project(const Unit3& pw, OptionalJacobian<2, 6> Dpose = {},
127 OptionalJacobian<2, 2> Dpoint = {},
128 OptionalJacobian<2, DimK> Dcal = {}) const {
129 return _project(pw, Dpose, Dpoint, Dcal);
130 }
131
132 /// backproject a 2-dimensional point to a 3-dimensional point at given depth
133 Point3 backproject(const Point2& p, double depth,
134 OptionalJacobian<3, 6> Dresult_dpose = {},
135 OptionalJacobian<3, 2> Dresult_dp = {},
136 OptionalJacobian<3, 1> Dresult_ddepth = {},
137 OptionalJacobian<3, DimK> Dresult_dcal = {}) const {
138 typedef Eigen::Matrix<double, 2, DimK> Matrix2K;
139 Matrix2K Dpn_dcal;
140 Matrix22 Dpn_dp;
141 const Point2 pn = calibration().calibrate(p, Dresult_dcal ? &Dpn_dcal : 0,
142 Dresult_dp ? &Dpn_dp : 0);
143 Matrix32 Dpoint_dpn;
144 Matrix31 Dpoint_ddepth;
145 const Point3 point = BackprojectFromCamera(p: pn, depth,
146 Dpoint: (Dresult_dp || Dresult_dcal) ? &Dpoint_dpn : 0,
147 Ddepth: Dresult_ddepth ? &Dpoint_ddepth : 0);
148 Matrix33 Dresult_dpoint;
149 const Point3 result = pose().transformFrom(point, Dresult_dpose,
150 (Dresult_ddepth ||
151 Dresult_dp ||
152 Dresult_dcal) ? &Dresult_dpoint : 0);
153 if (Dresult_dcal)
154 *Dresult_dcal = Dresult_dpoint * Dpoint_dpn * Dpn_dcal; // (3x3)*(3x2)*(2xDimK)
155 if (Dresult_dp)
156 *Dresult_dp = Dresult_dpoint * Dpoint_dpn * Dpn_dp; // (3x3)*(3x2)*(2x2)
157 if (Dresult_ddepth)
158 *Dresult_ddepth = Dresult_dpoint * Dpoint_ddepth; // (3x3)*(3x1)
159
160 return result;
161 }
162
163 /// backproject a 2-dimensional point to a 3-dimensional point at infinity
164 Unit3 backprojectPointAtInfinity(const Point2& p) const {
165 const Point2 pn = calibration().calibrate(p);
166 const Unit3 pc(pn.x(), pn.y(), 1.0); //by convention the last element is 1
167 return pose().rotation().rotate(pc);
168 }
169
170 /**
171 * Calculate range to a landmark
172 * @param point 3D location of landmark
173 * @return range (double)
174 */
175 double range(const Point3& point,
176 OptionalJacobian<1, 6> Dcamera = {},
177 OptionalJacobian<1, 3> Dpoint = {}) const {
178 return pose().range(point, Dcamera, Dpoint);
179 }
180
181 /**
182 * Calculate range to another pose
183 * @param pose Other SO(3) pose
184 * @return range (double)
185 */
186 double range(const Pose3& pose, OptionalJacobian<1, 6> Dcamera = {},
187 OptionalJacobian<1, 6> Dpose = {}) const {
188 return this->pose().range(pose, Dcamera, Dpose);
189 }
190
191 /**
192 * Calculate range to a CalibratedCamera
193 * @param camera Other camera
194 * @return range (double)
195 */
196 double range(const CalibratedCamera& camera, OptionalJacobian<1, 6> Dcamera =
197 {}, OptionalJacobian<1, 6> Dother = {}) const {
198 return pose().range(camera.pose(), Dcamera, Dother);
199 }
200
201 /**
202 * Calculate range to a PinholePoseK derived class
203 * @param camera Other camera
204 * @return range (double)
205 */
206 template<class CalibrationB>
207 double range(const PinholeBaseK<CalibrationB>& camera,
208 OptionalJacobian<1, 6> Dcamera = {},
209 OptionalJacobian<1, 6> Dother = {}) const {
210 return pose().range(camera.pose(), Dcamera, Dother);
211 }
212
213private:
214
215#if GTSAM_ENABLE_BOOST_SERIALIZATION
216 /** Serialization function */
217 friend class boost::serialization::access;

219 void serialize(Archive & ar, const unsigned int /*version*/) {
220 ar
221 & boost::serialization::make_nvp("PinholeBase",
222 boost::serialization::base_object<PinholeBase>(*this));
223 }
224#endif
225
226public:
227 GTSAM_MAKE_ALIGNED_OPERATOR_NEW
228};

231/**
232 * A pinhole camera class that has a Pose3 and a *fixed* Calibration.
233 * Instead of using this class, one might consider calibrating the measurements
234 * and using CalibratedCamera, which would then be faster.
235 * @ingroup geometry
236 * \nosubgrouping
237 */
238template<typename CALIBRATION>
239class PinholePose: public PinholeBaseK<CALIBRATION> {
240
241private:
242
243 typedef PinholeBaseK<CALIBRATION> Base; ///< base class has 3D pose as private member
244 std::shared_ptr<CALIBRATION> K_; ///< shared pointer to fixed calibration
245
246public:
247
248 inline constexpr static auto dimension = 6; ///< There are 6 DOF to optimize for
249
250 /// @name Standard Constructors
251 /// @{
252
253 /** default constructor */
254 PinholePose() {
255 }
256
257 /** constructor with pose, uses default calibration */
258 explicit PinholePose(const Pose3& pose) :
259 Base(pose), K_(new CALIBRATION()) {
260 }
261
262 /** constructor with pose and calibration */
263 PinholePose(const Pose3& pose, const std::shared_ptr<CALIBRATION>& K) :
264 Base(pose), K_(K) {
265 }
266
267 /// @}
268 /// @name Named Constructors
269 /// @{
270
271 /**
272 * Create a level camera at the given 2D pose and height
273 * @param K the calibration
274 * @param pose2 specifies the location and viewing direction
275 * (theta 0 = looking in direction of positive X axis)
276 * @param height camera height
277 */
278 static PinholePose Level(const std::shared_ptr<CALIBRATION>& K,
279 const Pose2& pose2, double height) {
280 return PinholePose(Base::LevelPose(pose2, height), K);
281 }
282
283 /// PinholePose::level with default calibration
284 static PinholePose Level(const Pose2& pose2, double height) {
285 return PinholePose::Level(std::make_shared<CALIBRATION>(), pose2, height);
286 }
287
288 /**
289 * Create a camera at the given eye position looking at a target point in the scene
290 * with the specified up direction vector.
291 * @param eye specifies the camera position
292 * @param target the point to look at
293 * @param upVector specifies the camera up direction vector,
294 * doesn't need to be on the image plane nor orthogonal to the viewing axis
295 * @param K optional calibration parameter
296 */
297 static PinholePose Lookat(const Point3& eye, const Point3& target,
298 const Point3& upVector, const std::shared_ptr<CALIBRATION>& K =
299 std::make_shared<CALIBRATION>()) {
300 return PinholePose(Base::LookatPose(eye, target, upVector), K);
301 }
302
303 /// @}
304 /// @name Advanced Constructors
305 /// @{
306
307 /// Init from 6D vector
308 explicit PinholePose(const Vector &v) :
309 Base(v), K_(new CALIBRATION()) {
310 }
311
312 /// Init from Vector and calibration
313 PinholePose(const Vector &v, const Vector &K) :
314 Base(v), K_(new CALIBRATION(K)) {
315 }
316
317 // Init from Pose3 and calibration
318 PinholePose(const Pose3 &pose, const Vector &K) :
319 Base(pose), K_(new CALIBRATION(K)) {
320 }
321
322 /// @}
323 /// @name Testable
324 /// @{
325
326 /// assert equality up to a tolerance
327 bool equals(const Base &camera, double tol = 1e-9) const {
328 const PinholePose* e = dynamic_cast<const PinholePose*>(&camera);
329 return Base::equals(camera, tol) && K_->equals(e->calibration(), tol);
330 }
331
332 /// stream operator
333 GTSAM_EXPORT friend std::ostream& operator<<(std::ostream& os,
334 const PinholePose& camera) {
335 os << "{R: " << camera.pose().rotation().rpy().transpose();
336 os << ", t: " << camera.pose().translation().transpose();
337 if (!camera.K_) os << ", K: none";
338 else os << ", K: " << *camera.K_;
339 os << "}";
340 return os;
341 }
342
343 /// print
344 void print(const std::string& s = "PinholePose") const override {
345 Base::print(s);
346 if (!K_)
347 std::cout << "s No calibration given" << std::endl;
348 else
349 K_->print(s + ".calibration");
350 }
351
352 /// @}
353 /// @name Standard Interface
354 /// @{
355
356 ~PinholePose() override {
357 }
358
359 /// return shared pointer to calibration
360 const std::shared_ptr<CALIBRATION>& sharedCalibration() const {
361 return K_;
362 }
363
364 /// return calibration
365 const CALIBRATION& calibration() const override {
366 return *K_;
367 }
368
369 /** project a point from world coordinate to the image, 2 derivatives only
370 * @param pw is a point in world coordinates
371 * @param Dpose is the Jacobian w.r.t. the whole camera (really only the pose)
372 * @param Dpoint is the Jacobian w.r.t. point3
373 */
374 Point2 project2(const Point3& pw, OptionalJacobian<2, 6> Dpose = {},
375 OptionalJacobian<2, 3> Dpoint = {}) const {
376 return Base::project(pw, Dpose, Dpoint);
377 }
378
379 /// project2 version for point at infinity
380 Point2 project2(const Unit3& pw, OptionalJacobian<2, 6> Dpose = {},
381 OptionalJacobian<2, 2> Dpoint = {}) const {
382 return Base::project(pw, Dpose, Dpoint);
383 }
384
385 /// @}
386 /// @name Manifold
387 /// @{
388
389 size_t dim() const {
390 return 6;
391 }
392
393 static size_t Dim() {
394 return 6;
395 }
396
397 /// move a cameras according to d
398 PinholePose retract(const Vector6& d) const {
399 return PinholePose(Base::pose().retract(d), K_);
400 }
401
402 /// return canonical coordinate
403 Vector6 localCoordinates(const PinholePose& p) const {
404 return Base::pose().localCoordinates(p.Base::pose());
405 }
406
407 /// for Canonical
408 static PinholePose Identity() {
409 return PinholePose(); // assumes that the default constructor is valid
410 }
411
412 /// for Linear Triangulation
413 Matrix34 cameraProjectionMatrix() const {
414 Matrix34 P = Matrix34(PinholeBase::pose().inverse().matrix().block(0, 0, 3, 4));
415 return K_->K() * P;
416 }
417
418 /// for Nonlinear Triangulation
419 Vector defaultErrorWhenTriangulatingBehindCamera() const {
420 return Eigen::Matrix<double,traits<Point2>::dimension,1>::Constant(2.0 * K_->fx());
421 }
422 /// @}
423
424private:
425
426#if GTSAM_ENABLE_BOOST_SERIALIZATION
427 /** Serialization function */
428 friend class boost::serialization::access;
429 template<class Archive>
430 void serialize(Archive & ar, const unsigned int /*version*/) {
431 ar
432 & boost::serialization::make_nvp("PinholeBaseK",
433 boost::serialization::base_object<Base>(*this));
434 ar & BOOST_SERIALIZATION_NVP(K_);
435 }
436#endif
437
438public:
439 GTSAM_MAKE_ALIGNED_OPERATOR_NEW
440};
441// end of class PinholePose
