26namespace gtsam {
27
28/// Cyclic group of order N
29template<size_t N>
30class Cyclic {
31 size_t i_; ///< we just use an unsigned int
32public:
33 /// Constructor
34 Cyclic(size_t i) :
35 i_(i) {
36 assert(i < N);
37 }
38 /// Default constructor yields identity
39 Cyclic():i_(0) {
40 }
41 static Cyclic Identity() { return Cyclic();}
42
43 /// Cast to size_t
44 operator size_t() const {
45 return i_;
46 }
47 /// Addition modulo N
48 Cyclic operator+(const Cyclic& h) const {
49 return (i_ + h.i_) % N;
50 }
51 /// Subtraction modulo N
52 Cyclic operator-(const Cyclic& h) const {
53 return (N + i_ - h.i_) % N;
54 }
55 /// Inverse
56 Cyclic operator-() const {
57 return (N - i_) % N;
58 }
59 /// print with optional string
60 void print(const std::string& s = "") const {
61 std::cout << s << i_ << std::endl;
62 }
63 /// equals with an tolerance, prints out message if unequal
64 bool equals(const Cyclic& other, double tol = 1e-9) const {
65 return other.i_ == i_;
66 }
67};
68
69/// Define cyclic group to be a model of the Additive Group concept
70template<size_t N>
71struct traits<Cyclic<N> > : internal::AdditiveGroupTraits<Cyclic<N> >, //
72 Testable<Cyclic<N> > {
73};

75} // \namespace gtsam
