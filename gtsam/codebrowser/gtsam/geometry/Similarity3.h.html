28namespace gtsam {
29
30// Forward declarations
31class Pose3;
32
33/**
34 * 3D similarity transform
35 */
36class GTSAM_EXPORT Similarity3 : public MatrixLieGroup<Similarity3, 7, 4> {
37 public:
38 /// @name Pose Concept
39 /// @{
40 typedef Rot3 Rotation;
41 typedef Point3 Translation;
42 /// @}
43
44 using Vector16 = Eigen::Matrix<double, 16, 1>;
45
46 private:
47 Rot3 R_;
48 Point3 t_;
49 double s_;
50
51 public:
52 /// @name Constructors
53 /// @{
54
55 /// Default constructor
56 Similarity3();
57
58 /// Construct pure scaling
59 Similarity3(double s);
60
61 /// Construct from GTSAM types
62 Similarity3(const Rot3& R, const Point3& t, double s);
63
64 /// Construct from Eigen types
65 Similarity3(const Matrix3& R, const Vector3& t, double s);
66
67 /// Construct from matrix [R t; 0 s^-1]
68 Similarity3(const Matrix4& T);
69
70
71
72 /// @}
73 /// @name Testable
74 /// @{
75
76 /// Compare with tolerance
77 bool equals(const Similarity3& sim, double tol) const;
78
79 /// Exact equality
80 bool operator==(const Similarity3& other) const;
81
82 /// Print with optional string
83 void print(const std::string& s = "") const;
84
85 GTSAM_EXPORT friend std::ostream& operator<<(std::ostream& os,
86 const Similarity3& p);
87
88 /// @}
89 /// @name Group
90 /// @{
91
92 /// Return an identity transform
93 static Similarity3 Identity();
94
95 /// Composition
96 Similarity3 operator*(const Similarity3& S) const;
97
98 /// Return the inverse
99 Similarity3 inverse() const;
100
101 /// @}
102 /// @name Group action on Point3
103 /// @{
104
105 /// Action on a point p is s*(R*p+t)
106 Point3 transformFrom(const Point3& p, //
107 OptionalJacobian<3, 7> H1 = {}, //
108 OptionalJacobian<3, 3> H2 = {}) const;
109
110 /**
111 * Action on a pose T.
112 * |Rs ts| |R t| |Rs*R Rs*t+ts|
113 * |0 1/s| * |0 1| = | 0 1/s |, the result is still a Sim3 object.
114 * To retrieve a Pose3, we normalized the scale value into 1.
115 * |Rs*R Rs*t+ts| |Rs*R s(Rs*t+ts)|
116 * | 0 1/s | = | 0 1 |
117 *
118 * This group action satisfies the compatibility condition.
119 * For more details, refer to: https://en.wikipedia.org/wiki/Group_action
120 */
121 Pose3 transformFrom(const Pose3& T) const;
122
123 /** syntactic sugar for transformFrom */
124 Point3 operator*(const Point3& p) const;
125
126 /**
127 * Create Similarity3 by aligning at least three point pairs
128 */
129 static Similarity3 Align(const Point3Pairs& abPointPairs);
130
131 /**
132 * Create the Similarity3 object that aligns at least two pose pairs.
133 * Each pair is of the form (aTi, bTi).
134 * Given a list of pairs in frame a, and a list of pairs in frame b, Align()
135 * will compute the best-fit Similarity3 aSb transformation to align them.
136 * First, the rotation aRb will be computed as the average (Karcher mean) of
137 * many estimates aRb (from each pair). Afterwards, the scale factor will be
138 * computed using the algorithm described here:
139 * http://www5.informatik.uni-erlangen.de/Forschung/Publikationen/2005/Zinsser05-PSR.pdf
140 */
141 static Similarity3 Align(const Pose3Pairs& abPosePairs);
142
143 /// @}
144 /// @name Lie Group
145 /// @{
146
147 using LieAlgebra = Matrix4;
148
149 /** Log map at the identity
150 * \f$ [R_x,R_y,R_z, t_x, t_y, t_z, \lambda] \f$
151 */
152 static Vector7 Logmap(const Similarity3& s, //
153 OptionalJacobian<7, 7> Hm = {});
154
155 /** Exponential map at the identity
156 */
157 static Similarity3 Expmap(const Vector7& v, //
158 OptionalJacobian<7, 7> Hm = {});
159
160 /// Chart at the origin
161 struct ChartAtOrigin {
162 static Similarity3 Retract(const Vector7& v,
163 ChartJacobian H = {}) {
164 return Similarity3::Expmap(v, Hm: H);
165 }
166 static Vector7 Local(const Similarity3& other,
167 ChartJacobian H = {}) {
168 return Similarity3::Logmap(s: other, Hm: H);
169 }
170 };
171
172 using LieGroup<Similarity3, 7>::inverse;
173
174 /// Project from one tangent space to another
175 Matrix7 AdjointMap() const;
176
177 /**
178 * Hat for Similarity3:
179 * @param xi 7-dim twist (w,u,lambda) where
180 * @return 4*4 element of Lie algebra that can be exponentiated
181 */
182 static Matrix4 Hat(const Vector7& xi);
183
184 /// Vee maps from Lie algebra to tangent vector
185 static Vector7 Vee(const Matrix4& X);
186
187 /// @}
188 /// @name Standard interface
189 /// @{
190
191 /// Calculate 4*4 matrix group equivalent
192 Matrix4 matrix() const;
193
194 /// Return a GTSAM rotation
195 Rot3 rotation() const { return R_; }
196
197 /// Return a GTSAM translation
198 Point3 translation() const { return t_; }
199
200 /// Return the scale
201 double scale() const { return s_; }
202
203 /// @}
204 /// @name Deprecated
205 /// @{
206
207#ifdef GTSAM_ALLOW_DEPRECATED_SINCE_V43
208 /// @deprecated: use Similarity3::Hat
209 static Matrix4 wedge(const Vector7& xi) {
210 return Similarity3::Hat(xi);
211 }
212#endif
213
214 /// @}
215 /// @name Helper functions
216 /// @{
217
218 private:
219
220 #if GTSAM_ENABLE_BOOST_SERIALIZATION
221 /** Serialization function */
222 friend class boost::serialization::access;
223 template<class Archive>
224 void serialize(Archive & ar, const unsigned int /*version*/) {
225 ar & BOOST_SERIALIZATION_NVP(R_);
226 ar & BOOST_SERIALIZATION_NVP(t_);
227 ar & BOOST_SERIALIZATION_NVP(s_);
228 }
229 #endif
230
231 /// Calculate expmap and logmap coefficients.
232 static Matrix3 GetV(Vector3 w, double lambda);
233
234 /// @}
235};

242}
243#endif
