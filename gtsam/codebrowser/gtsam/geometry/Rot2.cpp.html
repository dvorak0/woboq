55/* ************************************************************************* */
56Rot2& Rot2::normalize() {
57 double scale = c_*c_ + s_*s_;
58 if(std::abs(x: scale-1.0) > 1e-10) {
59 scale = 1 / sqrt(x: scale);
60 c_ *= scale;
61 s_ *= scale;
62 }
63 return *this;
64}

114// see doc/math.lyx, SO(2) section
115Point2 Rot2::rotate(const Point2& p, OptionalJacobian<2, 1> H1,
116 OptionalJacobian<2, 2> H2) const {
117 const Point2 q = Point2(c_ * p.x() + -s_ * p.y(), s_ * p.x() + c_ * p.y());
118 if (H1) *H1 << -q.y(), q.x();
119 if (H2) *H2 = matrix();
120 return q;
121}

123/* ************************************************************************* */
124// see doc/math.lyx, SO(2) section
125Point2 Rot2::unrotate(const Point2& p,
126 OptionalJacobian<2, 1> H1, OptionalJacobian<2, 2> H2) const {
127 const Point2 q = Point2(c_ * p.x() + s_ * p.y(), -s_ * p.x() + c_ * p.y());
128 if (H1) *H1 << q.y(), -q.x();
129 if (H2) *H2 = transpose();
130 return q;
131}

133/* ************************************************************************* */
134Rot2 Rot2::relativeBearing(const Point2& d, OptionalJacobian<1, 2> H) {
135 double x = d.x(), y = d.y(), d2 = x * x + y * y, n = sqrt(x: d2);
136 if(std::abs(x: n) > 1e-5) {
137 if (H) {
138 *H << -y / d2, x / d2;
139 }
140 return Rot2::fromCosSin(c: x / n, s: y / n);
141 } else {
142 if (H) *H << 0.0, 0.0;
143 return Rot2();
144 }
145}

147/* ************************************************************************* */
148Rot2 Rot2::ClosestTo(const Matrix2& M) {
149 Eigen::JacobiSVD<Matrix2> svd(M, Eigen::ComputeFullU | Eigen::ComputeFullV);
150 const Matrix2& U = svd.matrixU();
151 const Matrix2& V = svd.matrixV();
152 const double det = (U * V.transpose()).determinant();
153 Matrix2 M_prime = (U * Vector2(1, det).asDiagonal() * V.transpose());
154
155 double c = M_prime(0, 0);
156 double s = M_prime(1, 0);
157 return Rot2::fromCosSin(c, s);
158}

161Vector4 Rot2::vec(OptionalJacobian<4, 1> H) const {
162 const Matrix2 R = matrix();
163 if (H) {
164 H->block<2, 1>(startRow: 0, startCol: 0) = R.col(i: 1);
165 H->block<2, 1>(startRow: 2, startCol: 0) = -R.col(i: 0);
166 }
167 return Eigen::Map<const Vector4>(R.data());
168}
