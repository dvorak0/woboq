117 /// compose syntactic sugar

122 /**
123 * Interpolate between two poses via individual rotation and translation
124 * interpolation.
125 *
126 * The default "interpolate" method defined in Lie.h minimizes the geodesic
127 * distance on the manifold, leading to a screw motion interpolation in
128 * Cartesian space, which might not be what is expected.
129 * In contrast, this method executes a straight line interpolation for the
130 * translation, while still using interpolate (aka "slerp") for the rotational
131 * component. This might be more intuitive in many applications.
132 *
133 * @param T End point of interpolation.
134 * @param t A value in [0, 1].
135 */
136 Pose3 interpolateRt(const Pose3& T, double t,
137 OptionalJacobian<6, 6> Hself = {},
138 OptionalJacobian<6, 6> Harg = {},
139 OptionalJacobian<6, 1> Ht = {}) const;

141 /// @}
142 /// @name Lie Group
143 /// @{
144
145 using LieAlgebra = Matrix4;
146
147 /// Exponential map at identity - create a rotation from canonical coordinates \f$ [R_x,R_y,R_z,T_x,T_y,T_z] \f$
148 static Pose3 Expmap(const Vector6& xi, OptionalJacobian<6, 6> Hxi = {});
149
150 /// Log map at identity - return the canonical coordinates \f$ [R_x,R_y,R_z,T_x,T_y,T_z] \f$ of this rotation
151 static Vector6 Logmap(const Pose3& pose, OptionalJacobian<6, 6> Hpose = {});
152
153 /**
154 * Calculate Adjoint map, transforming a twist in this pose's (i.e, body) frame to the world spatial frame
155 * Ad_pose is 6*6 matrix that when applied to twist xi \f$ [R_x,R_y,R_z,T_x,T_y,T_z] \f$, returns Ad_pose(xi)
156 */
157 Matrix6 AdjointMap() const;
158
159 /**
160 * Apply this pose's AdjointMap Ad_g to a twist \f$ \xi_b \f$, i.e. a
161 * body-fixed velocity, transforming it to the spatial frame
162 * \f$ \xi^s = g*\xi^b*g^{-1} = Ad_g * \xi^b \f$
163 * Note that H_xib = AdjointMap()
164 */
165 Vector6 Adjoint(const Vector6& xi_b,
166 OptionalJacobian<6, 6> H_this = {},
167 OptionalJacobian<6, 6> H_xib = {}) const;
168
169 /// The dual version of Adjoint
170 Vector6 AdjointTranspose(const Vector6& x,
171 OptionalJacobian<6, 6> H_this = {},
172 OptionalJacobian<6, 6> H_x = {}) const;
173
174 /**
175 * Compute the [ad(w,v)] operator as defined in [Kobilarov09siggraph], pg 11
176 * [ad(w,v)] = [w^, zero3; v^, w^]
177 * Note that this is the matrix representation of the adjoint operator for se3 Lie algebra,
178 * aka the Lie bracket, and also the derivative of Adjoint map for the Lie group SE3.
179 *
180 * Let \f$ \hat{\xi}_i \f$ be the se3 Lie algebra, and \f$ \hat{\xi}_i^\vee = \xi_i = [\omega_i,v_i] \in \mathbb{R}^6\f$ be its
181 * vector representation.
182 * We have the following relationship:
183 * \f$ [\hat{\xi}_1,\hat{\xi}_2]^\vee = ad_{\xi_1}(\xi_2) = [ad_{(\omega_1,v_1)}]*\xi_2 \f$
184 *
185 * We use this to compute the discrete version of the inverse right-trivialized tangent map,
186 * and its inverse transpose in the discrete Euler Poincare' (DEP) operator.
187 *
188 */
189 static Matrix6 adjointMap(const Vector6& xi);
190
191 /**
192 * Action of the adjointMap on a Lie-algebra vector y, with optional derivatives
193 */
194 static Vector6 adjoint(const Vector6& xi, const Vector6& y,
195 OptionalJacobian<6, 6> Hxi = {},
196 OptionalJacobian<6, 6> H_y = {});
197
198 // temporary fix for wrappers until case issue is resolved
199 static Matrix6 adjointMap_(const Vector6 &xi) { return adjointMap(xi);}
200 static Vector6 adjoint_(const Vector6 &xi, const Vector6 &y) { return adjoint(xi, y);}
201
202 /**
203 * The dual version of adjoint action, acting on the dual space of the Lie-algebra vector space.
204 */
205 static Vector6 adjointTranspose(const Vector6& xi, const Vector6& y,
206 OptionalJacobian<6, 6> Hxi = {},
207 OptionalJacobian<6, 6> H_y = {});
