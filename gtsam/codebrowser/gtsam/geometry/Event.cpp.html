26void Event::print(const std::string& s) const {
27 std::cout << s << "{'time':" << time_
28 << ", 'location': " << location_.transpose() << "}";
29}
30
31/* ************************************************************************* */
32bool Event::equals(const Event& other, double tol) const {
33 return std::abs(x: time_ - other.time_) < tol &&
34 traits<Point3>::Equals(v1: location_, v2: other.location_, tol);
35}
