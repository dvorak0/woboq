20#include <gtsam/base/numericalDerivative.h>
21#include <gtsam/nonlinear/Expression.h>
22#include <gtsam/geometry/Event.h>

24#include <CppUnitLite/TestHarness.h>

26using namespace std::placeholders;
27using namespace std;
28using namespace gtsam;
29
30// Create a noise model for the TOA error
31static const double ms = 1e-3;
32static const double cm = 1e-2;
33typedef Eigen::Matrix<double, 1, 1> Vector1;
34static SharedNoiseModel model(noiseModel::Isotropic::Sigma(dim: 1, sigma: 0.5 * ms));
35
36static const double timeOfEvent = 25;
37static const Event exampleEvent(timeOfEvent, 1, 0, 0);
38static const Point3 microphoneAt0(0, 0, 0);
39
40static const double kSpeedOfSound = 340;
41static const TimeOfArrival kToa(kSpeedOfSound);
42
43//*****************************************************************************
44TEST(Event, Constructor) {
45 const double t = 0;
46 Event actual(t, 201.5 * cm, 201.5 * cm, (212 - 45) * cm);
47}
48
49//*****************************************************************************
50TEST(Event, Toa1) {
51 Event event(0, 1, 0, 0);
52 double expected = 1. / kSpeedOfSound;
53 EXPECT_DOUBLES_EQUAL(expected, kToa(event, microphoneAt0), 1e-9);
54}
55
56//*****************************************************************************
57TEST(Event, Toa2) {
58 double expectedTOA = timeOfEvent + 1. / kSpeedOfSound;
59 EXPECT_DOUBLES_EQUAL(expectedTOA, kToa(exampleEvent, microphoneAt0), 1e-9);
60}
61
62//*************************************************************************
63TEST(Event, Derivatives) {
64 Matrix14 actualH1;
65 Matrix13 actualH2;
66 kToa(exampleEvent, microphoneAt0, actualH1, actualH2);
67 Matrix expectedH1 = numericalDerivative11<double, Event>(
68 h: std::bind(f: kToa, args: std::placeholders::_1, args: microphoneAt0, args: nullptr, args: nullptr),
69 x: exampleEvent);
70 EXPECT(assert_equal(expectedH1, actualH1, 1e-8));
71 Matrix expectedH2 = numericalDerivative11<double, Point3>(
72 h: std::bind(f: kToa, args: exampleEvent, args: std::placeholders::_1, args: nullptr, args: nullptr),
73 x: microphoneAt0);
74 EXPECT(assert_equal(expectedH2, actualH2, 1e-8));
75}
76
77//*****************************************************************************
78TEST(Event, Expression) {
79 Key key = 12;
80 Expression<Event> event_(key);
81 Expression<Point3> knownMicrophone_(microphoneAt0); // constant expression
82 Expression<double> expression(kToa, event_, knownMicrophone_);
83
84 Values values;
85 values.insert(j: key, val: exampleEvent);
86 double expectedTOA = timeOfEvent + 1. / kSpeedOfSound;
87 EXPECT_DOUBLES_EQUAL(expectedTOA, expression.value(values), 1e-9);
88}
89
90//*****************************************************************************
91TEST(Event, Retract) {
92 Event event, expected(1, 2, 3, 4);
93 Vector4 v;
94 v << 1, 2, 3, 4;
95 EXPECT(assert_equal(expected, event.retract(v)));
96}
