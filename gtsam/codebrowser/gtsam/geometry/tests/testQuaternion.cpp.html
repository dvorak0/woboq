24using namespace std;
25using namespace gtsam;
26
27typedef Quaternion Q; // Typedef
28typedef traits<Q>::ChartJacobian QuaternionJacobian;
29
30GTSAM_CONCEPT_TESTABLE_INST(Quaternion)
31GTSAM_CONCEPT_MATRIX_LIE_GROUP_INST(Quaternion)
32
33//******************************************************************************
34TEST(Quaternion , Concept) {
35 GTSAM_CONCEPT_ASSERT(IsGroup<Quaternion >);
36 GTSAM_CONCEPT_ASSERT(IsManifold<Quaternion >);
37 GTSAM_CONCEPT_ASSERT(IsMatrixLieGroup<Quaternion >);
38}
39
40//******************************************************************************
41TEST(Quaternion , Constructor) {
42 Q q(Eigen::AngleAxisd(1, Vector3(0, 0, 1)));
43}
44
45//******************************************************************************
46TEST(Quaternion , Logmap) {
47 Q q1(5e-06, 0, 0, 1), q2(-5e-06, 0, 0, -1);
48 Vector3 v1 = traits<Q>::Logmap(q: q1);
49 Vector3 v2 = traits<Q>::Logmap(q: q2);
50 EXPECT(assert_equal(v1, v2));
51}
52
53//******************************************************************************
54TEST(Quaternion , Local) {
55 Vector3 z_axis(0, 0, 1);
56 Q q1(Eigen::AngleAxisd(0, z_axis));
57 Q q2(Eigen::AngleAxisd(0.1, z_axis));
58 QuaternionJacobian H1, H2;
59 Vector3 expected(0, 0, 0.1);
60 Vector3 actual = traits<Q>::Local(g: q1, h: q2, H1, H2);
61 EXPECT(assert_equal((Vector )expected, actual));
62}
63
64//******************************************************************************
65TEST(Quaternion , Retract) {
66 Vector3 z_axis(0, 0, 1);
67 Q q(Eigen::AngleAxisd(0, z_axis));
68 Q expected(Eigen::AngleAxisd(0.1, z_axis));
69 Vector3 v(0, 0, 0.1);
70 QuaternionJacobian Hq, Hv;
71 Q actual = traits<Q>::Retract(g: q, v, H1: Hq, H2: Hv);
72 EXPECT(actual.isApprox(expected));
73}
74
75//******************************************************************************
76TEST(Quaternion , Compose) {
77 Vector3 z_axis(0, 0, 1);
78 Q q1(Eigen::AngleAxisd(0.2, z_axis));
79 Q q2(Eigen::AngleAxisd(0.1, z_axis));
80
81 Q expected = q1 * q2;
82 Q actual = traits<Q>::Compose(g: q1, h: q2);
83 EXPECT(traits<Q>::Equals(expected, actual));
84}
85
86//******************************************************************************
87TEST(Quaternion , Between) {
88 Vector3 z_axis(0, 0, 1);
89 Q q1(Eigen::AngleAxisd(0.2, z_axis));
90 Q q2(Eigen::AngleAxisd(0.1, z_axis));
91
92 Q expected = q1.inverse() * q2;
93 Q actual = traits<Q>::Between(g: q1, h: q2);
94 EXPECT(traits<Q>::Equals(expected, actual));
95}
96
97//******************************************************************************
98TEST(Quaternion , Inverse) {
99 Vector3 z_axis(0, 0, 1);
100 Q q1(Eigen::AngleAxisd(0.1, z_axis));
101 Q expected(Eigen::AngleAxisd(-0.1, z_axis));
102
103 Q actual = traits<Q>::Inverse(g: q1);
104 EXPECT(traits<Q>::Equals(expected, actual));
105}
