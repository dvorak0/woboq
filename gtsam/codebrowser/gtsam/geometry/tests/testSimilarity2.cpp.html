64TEST(Similarity2, HatAndVee) {
65 // Create a few test vectors
66 Vector4 v1(1, 2, 3, 4);
67 Vector4 v2(0.1, -0.5, 1.0, -1.0);
68 Vector4 v3(0.0, 0.0, 0.0, 0.0);
69
70 // Test that Vee(Hat(v)) == v for various inputs
71 EXPECT(assert_equal(v1, Similarity2::Vee(Similarity2::Hat(v1))));
72 EXPECT(assert_equal(v2, Similarity2::Vee(Similarity2::Hat(v2))));
73 EXPECT(assert_equal(v3, Similarity2::Vee(Similarity2::Hat(v3))));
74
75 // Check the structure of the Lie Algebra element
76 Matrix3 expected;
77 expected << 0, -3, 1,
78 3, 0, 2,
79 0, 0, -4;
80
81 EXPECT(assert_equal(expected, Similarity2::Hat(v1)));
82}

92TEST(Similarity2, Compose) {
93 // Test group operation: compose two Similarity2 elements.
94 Rot2 R1 = Rot2::fromDegrees(theta: 30);
95 Point2 t1(1, 2);
96 double s1 = 2.0;
97 Similarity2 S1(R1, t1, s1);
98
99 Rot2 R2 = Rot2::fromDegrees(theta: 45);
100 Point2 t2(-1, 1);
101 double s2 = 4.0;
102 Similarity2 S2(R2, t2, s2);
103
104 Similarity2 S3 = S1.compose(g: S2);
105
106 // Compose manually
107 Rot2 expected_R = R1 * R2;
108 double expected_s = s1 * s2;
109 Point2 expected_t = t1 / s2 + R1.matrix() * t2;
110 Similarity2 expected_S3(expected_R, expected_t, expected_s);
111
112 EXPECT(assert_equal(expected_S3, S3));
113 EXPECT(assert_equal(expected_S3, S1 * S2));
114 EXPECT(assert_equal<Matrix3>(S3.matrix(), S1.matrix() * S2.matrix()));
115}

178TEST(Similarity2, TransformFrom_Pose2) {
179 // Setup
180 Rot2 R_sim = Rot2::fromAngle(M_PI / 6); // 30 degrees
181 Point2 t_sim(1.0, -1.0);
182 double s_sim = 2.0;
183 Similarity2 sim(R_sim, t_sim, s_sim);
184
185 Rot2 R_pose = Rot2::fromAngle(theta: -M_PI / 4); // -45 degrees
186 Point2 t_pose(3.0, 4.0);
187 Pose2 pose(R_pose, t_pose);
188
189 Rot2 expected_R = R_sim * R_pose;
190 Point2 expected_t = s_sim * (R_sim * t_pose + t_sim);
191 Pose2 expected(expected_R, expected_t);
192
193 Pose2 actual = sim.transformFrom(T: pose);
194
195 EXPECT(assert_equal(expected, actual, 1e-9));
196}

199TEST(Similarity2, Vec) {
200 const double theta = 0.3;
201 const Rot2 R_test = Rot2::fromAngle(theta);
202 const Point2 t_test(0.2, 0.7);
203 const double s_test = 4.0;
204 const Similarity2 sim(R_test, t_test, s_test);
205
206 // 1. Test the Value
207 Vector9 expected_vec;
208 const double c = cos(x: theta), si = sin(x: theta);
209 expected_vec << c, si, 0, // First column
210 -si, c, 0, // Second column
211 t_test.x(), t_test.y(), 1.0 / s_test; // Third column
212 Vector9 actual_vec = sim.vec();
213 EXPECT(assert_equal(expected_vec, actual_vec, 1e-9));
214
215 // 2. Test the Jacobian
216 Matrix94 H_actual;
217 sim.vec(H: H_actual);
218 auto vec_fun = [](const Similarity2& sim_arg) -> Vector9 {
219 return sim_arg.vec();
220 };
221 Matrix H_numerical = numericalDerivative11<Vector9, Similarity2, 4>(h: vec_fun, x: sim);
222 EXPECT(assert_equal(H_numerical, H_actual, 1e-7));
223}

226TEST(Similarity2, AdjointMap) {
227 // Create a non-trivial Similarity2 object
228 const Rot2 R = Rot2::fromAngle(theta: -0.7);
229 const Point2 t(1.5, -2.3);
230 const double s = 0.5;
231 const Similarity2 sim(R, t, s);
232
233 // Call the specialized AdjointMap
234 Matrix4 specialized_Adj = sim.AdjointMap();
235
236 // Call the generic AdjointMap from the base class
237 Matrix4 generic_Adj = static_cast<const MatrixLieGroup<Similarity2, 4, 3>*>(&sim)->AdjointMap();
238
239 // Assert that they are equal
240 EXPECT(assert_equal(specialized_Adj, generic_Adj, 1e-9));
241}
