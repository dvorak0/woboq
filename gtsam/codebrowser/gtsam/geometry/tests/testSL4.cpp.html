97TEST(SL4, ComposeAndInverse) {
98 SL4 sl4_1 = SL4::Expmap(xi: xi1);
99 SL4 sl4_2 = SL4::Expmap(xi: xi2);
100
101 SL4 composed = sl4_1.compose(g: sl4_2);
102 SL4 expected(composed.matrix());
103 EXPECT(assert_equal(expected, composed, 1e-8));
104
105 SL4 inv = sl4_1.inverse();
106 SL4 identity = sl4_1.compose(g: inv);
107 EXPECT(assert_equal(SL4(), identity, 1e-8));

113 SL4 sl4_2 = SL4::Expmap(xi: xi2);
114
115 Matrix H1, H2;

117 SL4 actual = sl4_1.between(g: sl4_2, H1, H2);
118 EXPECT(assert_equal(expected, actual, 1e-8));
119
120 Matrix numericalH1 =
121 numericalDerivative21(h: testing::between<SL4>, x1: sl4_1, x2: sl4_2);

125 numericalDerivative22(h: testing::between<SL4>, x1: sl4_1, x2: sl4_2);
