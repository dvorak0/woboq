69// Cal3Bundler test
70#include <gtsam/geometry/PinholeCamera.h>
71#include <gtsam/geometry/Cal3Bundler.h>
72TEST(PinholeSet, Pinhole) {
73 typedef PinholeCamera<Cal3Bundler> Camera;
74 typedef Point2Vector ZZ;
75 PinholeSet<Camera> set;
76 Camera camera;
77 set.push_back(x: camera);
78 set.push_back(x: camera);
79 Point3 p(0, 0, 1);
80 EXPECT(assert_equal(set, set));
81 PinholeSet<Camera> set2 = set;
82 set2.push_back(x: camera);
83 EXPECT(!set.equals(set2));
84
85 // Check measurements
86 Point2 expected(0,0);
87 ZZ z = set.project2(point: p);
88 EXPECT(assert_equal(expected, z[0]));
89 EXPECT(assert_equal(expected, z[1]));
90
91 // Calculate expected derivatives using Pinhole
92 Matrix actualE;
93 Matrix F1;
94 {
95 Matrix23 E1;
96 camera.project2(pw: p, Dcamera: F1, Dpoint: E1);
97 actualE.resize(rows: 4, cols: 3);
98 actualE << E1, E1;
99 }
100
101 // Check computed derivatives
102 {
103 PinholeSet<Camera>::FBlocks Fs;
104 Matrix E;
105 set.project2(point: p, args&: Fs, args&: E);
106 EXPECT(assert_equal(actualE, E));
107 LONGS_EQUAL(2, Fs.size());
108 EXPECT(assert_equal(F1, Fs[0]));
109 EXPECT(assert_equal(F1, Fs[1]));
110 }
111
112 // Check errors
113 ZZ measured;
114 measured.push_back(x: Point2(1, 2));
115 measured.push_back(x: Point2(3, 4));
116 Vector4 expectedV;
117
118 // reprojectionError
119 expectedV << -1, -2, -3, -4;
120 Vector actualV = set.reprojectionError(point: p, measured);
121 EXPECT(assert_equal(expectedV, actualV));
122
123 // reprojectionErrorAtInfinity
124 Unit3 pointAtInfinity(0, 0, 1000);
125 {
126 Matrix22 E1;
127 camera.project2(pw: pointAtInfinity, Dcamera: F1, Dpoint: E1);
128 actualE.resize(rows: 4, cols: 2);
129 actualE << E1, E1;
130 }
131 EXPECT(
132 assert_equal(pointAtInfinity,
133 camera.backprojectPointAtInfinity(Point2(0,0))));
134 {
135 PinholeSet<Camera>::FBlocks Fs;
136 Matrix E;
137 actualV = set.reprojectionError(point: pointAtInfinity, measured, args&: Fs, args&: E);
138 EXPECT(assert_equal(actualE, E));
139 LONGS_EQUAL(2, Fs.size());
140 EXPECT(assert_equal(F1, Fs[0]));
141 EXPECT(assert_equal(F1, Fs[1]));
142 }
143 EXPECT(assert_equal(expectedV, actualV));
144
145 // Instantiate triangulateSafe
146 TriangulationParameters params;
147 TriangulationResult actual = set.triangulateSafe(measured: z, params);
148 CHECK(actual.degenerate());
149}
