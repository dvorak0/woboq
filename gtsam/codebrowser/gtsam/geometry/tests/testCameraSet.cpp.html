29// Cal3Bundler test
30#include <gtsam/geometry/PinholeCamera.h>
31#include <gtsam/geometry/Cal3Bundler.h>
32TEST(CameraSet, Pinhole) {
33 typedef PinholeCamera<Cal3Bundler> Camera;
34 typedef CameraSet<Camera> Set;
35 typedef Point2Vector ZZ;
36 Set set;
37 Camera camera;
38 set.push_back(x: camera);
39 set.push_back(x: camera);
40 Point3 p(0, 0, 1);
41 EXPECT(assert_equal(set, set));
42 Set set2 = set;
43 set2.push_back(x: camera);
44 EXPECT(!set.equals(set2));
45
46 // Check measurements
47 Point2 expected(0,0);
48 ZZ z = set.project2(point: p);
49 EXPECT(assert_equal(expected, z[0]));
50 EXPECT(assert_equal(expected, z[1]));
51
52 // Calculate expected derivatives using Pinhole
53 Matrix actualE;
54 Matrix29 F1;
55 {
56 Matrix23 E1;
57 camera.project2(pw: p, Dcamera: F1, Dpoint: E1);
58 actualE.resize(rows: 4,cols: 3);
59 actualE << E1, E1;
60 }
61
62 // Check computed derivatives
63 Set::FBlocks Fs;
64 Matrix E;
65 set.project2(point: p, args&: Fs, args&: E);
66 LONGS_EQUAL(2, Fs.size());
67 EXPECT(assert_equal(F1, Fs[0]));
68 EXPECT(assert_equal(F1, Fs[1]));
69 EXPECT(assert_equal(actualE, E));
70
71 // Check errors
72 ZZ measured;
73 measured.push_back(x: Point2(1, 2));
74 measured.push_back(x: Point2(3, 4));
75 Vector4 expectedV;
76
77 // reprojectionError
78 expectedV << -1, -2, -3, -4;
79 Vector actualV = set.reprojectionError(point: p, measured);
80 EXPECT(assert_equal(expectedV, actualV));
81
82 // Check Schur complement
83 Matrix F(4, 18);
84 F << F1, Matrix29::Zero(), Matrix29::Zero(), F1;
85 Matrix Ft = F.transpose();
86 Matrix34 Et = E.transpose();
87 Matrix3 P = Et * E;
88 Matrix schur(19, 19);
89 Vector4 b = actualV;
90 Vector v = Ft * (b - E * P * Et * b);
91 schur << Ft * F - Ft * E * P * Et * F, v, v.transpose(), 30;
92 SymmetricBlockMatrix actualReduced = Set::SchurComplement<3>(Fs, E, P, b);
93 EXPECT(assert_equal(schur, actualReduced.selfadjointView()));
94
95 // Check Schur complement update, same order, should just double
96 KeyVector allKeys {1, 2}, keys {1, 2};
97 Set::UpdateSchurComplement<3>(Fs, E, P, b, allKeys, keys, augmentedHessian&: actualReduced);
98 EXPECT(assert_equal((Matrix )(2.0 * schur), actualReduced.selfadjointView()));
99
100 // Check Schur complement update, keys reversed
101 KeyVector keys2 {2, 1};
102 Set::UpdateSchurComplement<3>(Fs, E, P, b, allKeys, keys: keys2, augmentedHessian&: actualReduced);
103 Vector4 reverse_b;
104 reverse_b << b.tail<2>(), b.head<2>();
105 Vector reverse_v = Ft * (reverse_b - E * P * Et * reverse_b);
106 Matrix A(19, 19);
107 A << Ft * F - Ft * E * P * Et * F, reverse_v, reverse_v.transpose(), 30;
108 EXPECT(assert_equal((Matrix )(2.0 * schur + A), actualReduced.selfadjointView()));
109
110 // reprojectionErrorAtInfinity
111 Unit3 pointAtInfinity(0, 0, 1000);
112 EXPECT(
113 assert_equal(pointAtInfinity,
114 camera.backprojectPointAtInfinity(Point2(0,0))));
115 actualV = set.reprojectionError(point: pointAtInfinity, measured, args&: Fs, args&: E);
116 EXPECT(assert_equal(expectedV, actualV));
117 LONGS_EQUAL(2, Fs.size());
118 {
119 Matrix22 E1;
120 camera.project2(pw: pointAtInfinity, Dcamera: F1, Dpoint: E1);
121 actualE.resize(rows: 4,cols: 2);
122 actualE << E1, E1;
123 }
124 EXPECT(assert_equal(F1, Fs[0]));
125 EXPECT(assert_equal(F1, Fs[1]));
126 EXPECT(assert_equal(actualE, E));
127}

130TEST(CameraSet, SchurComplementAndRearrangeBlocks) {
131 typedef PinholePose<Cal3Bundler> Camera;
132 typedef CameraSet<Camera> Set;
133
134 // this is the (block) Jacobian with respect to the nonuniqueKeys
135 std::vector<Eigen::Matrix<double, 2, 12>,
136 Eigen::aligned_allocator<Eigen::Matrix<double, 2, 12> > > Fs;
137 Fs.push_back(x: 1 * Matrix::Ones(rows: 2, cols: 12)); // corresponding to key pair (0,1)
138 Fs.push_back(x: 2 * Matrix::Ones(rows: 2, cols: 12)); // corresponding to key pair (1,2)
139 Fs.push_back(x: 3 * Matrix::Ones(rows: 2, cols: 12)); // corresponding to key pair (2,0)
140 Matrix E = 4 * Matrix::Identity(rows: 6, cols: 3) + Matrix::Ones(rows: 6, cols: 3);
141 E(0, 0) = 3;
142 E(1, 1) = 2;
143 E(2, 2) = 5;
144 Matrix Et = E.transpose();
145 Matrix P = (Et * E).inverse();
146 Vector b = 5 * Vector::Ones(newSize: 6);
147
148 { // SchurComplement
149 // Actual
150 SymmetricBlockMatrix augmentedHessianBM = Set::SchurComplement<3, 12>(Fs, E,
151 P, b);
152 Matrix actualAugmentedHessian = augmentedHessianBM.selfadjointView();
153
154 // Expected
155 Matrix F = Matrix::Zero(rows: 6, cols: 3 * 12);
156 F.block<2, 12>(startRow: 0, startCol: 0) = 1 * Matrix::Ones(rows: 2, cols: 12);
157 F.block<2, 12>(startRow: 2, startCol: 12) = 2 * Matrix::Ones(rows: 2, cols: 12);
158 F.block<2, 12>(startRow: 4, startCol: 24) = 3 * Matrix::Ones(rows: 2, cols: 12);
159
160 Matrix Ft = F.transpose();
161 Matrix H = Ft * F - Ft * E * P * Et * F;
162 Vector v = Ft * (b - E * P * Et * b);
163 Matrix expectedAugmentedHessian = Matrix::Zero(rows: 3 * 12 + 1, cols: 3 * 12 + 1);
164 expectedAugmentedHessian.block<36, 36>(startRow: 0, startCol: 0) = H;
165 expectedAugmentedHessian.block<36, 1>(startRow: 0, startCol: 36) = v;
166 expectedAugmentedHessian.block<1, 36>(startRow: 36, startCol: 0) = v.transpose();
167 expectedAugmentedHessian(36, 36) = b.squaredNorm();
168
169 EXPECT(assert_equal(expectedAugmentedHessian, actualAugmentedHessian));
170 }
171
172 { // SchurComplementAndRearrangeBlocks
173 KeyVector nonuniqueKeys;
174 nonuniqueKeys.push_back(x: 0);
175 nonuniqueKeys.push_back(x: 1);
176 nonuniqueKeys.push_back(x: 1);
177 nonuniqueKeys.push_back(x: 2);
178 nonuniqueKeys.push_back(x: 2);
179 nonuniqueKeys.push_back(x: 0);
180
181 KeyVector uniqueKeys;
182 uniqueKeys.push_back(x: 0);
183 uniqueKeys.push_back(x: 1);
184 uniqueKeys.push_back(x: 2);
185
186 // Actual
187 SymmetricBlockMatrix augmentedHessianBM =
188 Set::SchurComplementAndRearrangeBlocks<3, 12, 6>(
189 Fs, E, P, b, jacobianKeys: nonuniqueKeys, hessianKeys: uniqueKeys);
190 Matrix actualAugmentedHessian = augmentedHessianBM.selfadjointView();
191
192 // Expected
193 // we first need to build the Jacobian F according to unique keys
194 Matrix F = Matrix::Zero(rows: 6, cols: 18);
195 F.block<2, 6>(startRow: 0, startCol: 0) = Fs[0].block<2, 6>(startRow: 0, startCol: 0);
196 F.block<2, 6>(startRow: 0, startCol: 6) = Fs[0].block<2, 6>(startRow: 0, startCol: 6);
197 F.block<2, 6>(startRow: 2, startCol: 6) = Fs[1].block<2, 6>(startRow: 0, startCol: 0);
198 F.block<2, 6>(startRow: 2, startCol: 12) = Fs[1].block<2, 6>(startRow: 0, startCol: 6);
199 F.block<2, 6>(startRow: 4, startCol: 12) = Fs[2].block<2, 6>(startRow: 0, startCol: 0);
200 F.block<2, 6>(startRow: 4, startCol: 0) = Fs[2].block<2, 6>(startRow: 0, startCol: 6);
201
202 Matrix Ft = F.transpose();
203 Vector v = Ft * (b - E * P * Et * b);
204 Matrix H = Ft * F - Ft * E * P * Et * F;
205 Matrix expectedAugmentedHessian(19, 19);
206 expectedAugmentedHessian << H, v, v.transpose(), b.squaredNorm();
207
208 EXPECT(assert_equal(expectedAugmentedHessian, actualAugmentedHessian));
209 }
210}
