33TEST(StereoPoint2 , Concept) {
34 GTSAM_CONCEPT_ASSERT(IsGroup<StereoPoint2>);
35 GTSAM_CONCEPT_ASSERT(IsManifold<StereoPoint2 >);
36 GTSAM_CONCEPT_ASSERT(IsVectorSpace<StereoPoint2>);
37}
38
39/* ************************************************************************* */
40TEST(StereoPoint2, constructor) {
41 StereoPoint2 p1(1, 2, 3), p2 = p1;
42 EXPECT(assert_equal(p1, p2));
43}
44
45/* ************************************************************************* */
46TEST( StereoPoint2, arithmetic) {
47 EXPECT(assert_equal( StereoPoint2(5,6,7), StereoPoint2(4,5,6)+StereoPoint2(1,1,1)));
48 EXPECT(assert_equal( StereoPoint2(3,4,5), StereoPoint2(4,5,6)-StereoPoint2(1,1,1)));
49}
50
51/* ************************************************************************* */
52TEST(testStereoPoint2, left_right) {
53 StereoPoint2 p1(1,2,3);
54
55 EXPECT(assert_equal( Point2(1,3), p1.point2()));
56 EXPECT(assert_equal( Point2(2,3), p1.right()));
57}
