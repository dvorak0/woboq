218 return Camera(pose,cal).project(pw: point3D);
219}

221TEST( PinholeCamera, Dproject_Infinity)
222{
223 Matrix Dpose, Dpoint, Dcal;
224 Unit3 point3D(point1.x(), point1.y(), -10.0); // a point in front of the camera1
225
226 // test Projection
227 Point2 actual = camera.project(pw: point3D, Dpose, Dpoint, Dcal);
228 Point2 expected(-5.0, 5.0);
229 EXPECT(assert_equal(actual, expected, 1e-7));
230
231 // test Jacobians
232 Matrix numerical_pose = numericalDerivative31(h: projectInfinity3, x1: pose, x2: point3D, x3: K);
233 Matrix Hexpected2 = numericalDerivative32(h: projectInfinity3, x1: pose, x2: point3D, x3: K);
234 Matrix numerical_point2x2 = Hexpected2.block(startRow: 0,startCol: 0,blockRows: 2,blockCols: 2); // only the direction to the point matters
235 Matrix numerical_cal = numericalDerivative33(h: projectInfinity3, x1: pose, x2: point3D, x3: K);
236 EXPECT(assert_equal(numerical_pose, Dpose, 1e-7));
237 EXPECT(assert_equal(numerical_point2x2, Dpoint, 1e-7));
238 EXPECT(assert_equal(numerical_cal, Dcal, 1e-7));
239}
