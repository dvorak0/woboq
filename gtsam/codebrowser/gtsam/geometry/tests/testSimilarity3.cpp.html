246// Group action on Point3 (with simpler transform)
247TEST(Similarity3, GroupAction) {
248 EXPECT(assert_equal(Point3(2, 2, 0), T6 * Point3(0, 0, 0)));
249 EXPECT(assert_equal(Point3(4, 2, 0), T6 * Point3(1, 0, 0)));
250
251 // Test group action on R^4 via matrix representation
252 Vector4 qh;
253 qh << 1, 0, 0, 1;
254 Vector4 ph;
255 ph << 2, 1, 0, 0.5; // equivalent to Point3(4, 2, 0)
256 EXPECT(assert_equal((Vector )ph, T6.matrix() * qh));
257
258 // Test some more...
259 Point3 pa = Point3(1, 0, 0);
260 Similarity3 Ta(Rot3(), Point3(1, 2, 3), 1.0);
261 Similarity3 Tb(Rot3(), Point3(1, 2, 3), 2.0);
262 EXPECT(assert_equal(Point3(2, 2, 3), Ta.transformFrom(pa)));
263 EXPECT(assert_equal(Point3(4, 4, 6), Tb.transformFrom(pa)));
264
265 Similarity3 Tc(Rot3::Rz(M_PI / 2.0), Point3(1, 2, 3), 1.0);
266 Similarity3 Td(Rot3::Rz(M_PI / 2.0), Point3(1, 2, 3), 2.0);
267 EXPECT(assert_equal(Point3(1, 3, 3), Tc.transformFrom(pa)));
268 EXPECT(assert_equal(Point3(2, 6, 6), Td.transformFrom(pa)));
269
270 // Test derivative
271 // Use lambda to resolve overloaded method
272 std::function<Point3(const Similarity3&, const Point3&)>
273 f = [](const Similarity3& S, const Point3& p){ return S.transformFrom(p); };
274
275 Point3 q(1, 2, 3);
276 for (const auto& T : { T1, T2, T3, T4, T5, T6 }) {
277 Point3 q(1, 0, 0);
278 Matrix H1 = numericalDerivative21<Point3, Similarity3, Point3>(h: f, x1: T, x2: q);
279 Matrix H2 = numericalDerivative22<Point3, Similarity3, Point3>(h: f, x1: T, x2: q);
280 Matrix actualH1, actualH2;
281 T.transformFrom(p: q, H1: actualH1, H2: actualH2);
282 EXPECT(assert_equal(H1, actualH1));
283 EXPECT(assert_equal(H2, actualH2));
284 }
285}

288// Group action on Pose3
289// Estimate Sim(3) object "aSb" from pose pairs {(aTi, bTi)}
290// In the example below, let the "a" frame be the "world" frame below,
291// and let the "b" frame be the "egovehicle" frame.
292// Suppose within the egovehicle frame, we know the poses of two objects "o1" and "o2"
293TEST(Similarity3, GroupActionPose3) {
294 // Suppose we know the pose of the egovehicle in the world frame
295 Similarity3 wSe(Rot3::Ry(t: 180 * degree), Point3(2, 3, 5), 2.0);
296
297 // Create source poses (two objects o1 and o2 living in the egovehicle "e" frame)
298 // Suppose they are 3d cuboids detected by onboard sensor, in the egovehicle frame
299 Pose3 eTo1(Rot3(), Point3(0, 0, 0));
300 Pose3 eTo2(Rot3(-1, 0, 0, 0, -1, 0, 0, 0, 1), Point3(4, 0, 0));
301
302 // Create destination poses (two objects now living in the world/city "w" frame)
303 // Desired to place the objects into the world frame for tracking
304 Pose3 expected_wTo1(Rot3(-1, 0, 0, 0, 1, 0, 0, 0, -1), Point3(4, 6, 10));
305 Pose3 expected_wTo2(Rot3(1, 0, 0, 0, -1, 0, 0, 0, -1), Point3(-4, 6, 10));
306
307 // objects now live in the world frame, instead of in the egovehicle frame
308 EXPECT(assert_equal(expected_wTo1, wSe.transformFrom(eTo1)));
309 EXPECT(assert_equal(expected_wTo2, wSe.transformFrom(eTo2)));
310}

412 // Create a PriorFactor with a Sim3 prior
413 Similarity3 prior = Similarity3(Rot3::Ypr(y: 0.1, p: 0.2, r: 0.3), Point3(1, 2, 3), 4);
414 noiseModel::Isotropic::shared_ptr model = noiseModel::Isotropic::Sigma(dim: 7, sigma: 1);
415 Symbol key('x', 1);
416
417 // Create graph
418 NonlinearFactorGraph graph;
419 graph.addPrior(key, prior, model);
420
421 // Create initial estimate with Identity transform
422 Values initial;
423 initial.insert<Similarity3>(j: key, val: Similarity3());
424
425 // Optimize
426 Values result;
427 LevenbergMarquardtParams params;
428 params.setVerbosityLM("TRYCONFIG");

432 EXPECT(assert_equal(prior, result.at<Similarity3>(key), 1e-4));
