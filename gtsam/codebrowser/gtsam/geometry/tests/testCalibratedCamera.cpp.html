63TEST( CalibratedCamera, level1)
64{
65 // Create a level camera, looking in X-direction
66 Pose2 pose2(0.1,0.2,0);
67 CalibratedCamera camera = CalibratedCamera::Level(pose2, height: 0.3);
68

127static Point2 project2(const CalibratedCamera& camera, const Point3& point) {
128 return camera.project(point);
129}

132TEST( CalibratedCamera, Dproject_point_pose)
133{
134 Matrix Dpose, Dpoint;
135 Point2 result = camera.project(point: point1, Dcamera: Dpose, Dpoint);
136 Matrix numerical_pose = numericalDerivative21(h: project2, x1: camera, x2: point1);
137 Matrix numerical_point = numericalDerivative22(h: project2, x1: camera, x2: point1);
138 CHECK(assert_equal(Point3(-0.08, 0.08, 0.5), camera.pose().transformTo(point1)));
139 CHECK(assert_equal(Point2(-.16, .16), result));
140 CHECK(assert_equal(numerical_pose, Dpose, 1e-7));
141 CHECK(assert_equal(numerical_point, Dpoint, 1e-7));
142}

210TEST( PinholePose, DbackprojectCalibCamera)
211{
212 Matrix36 Dpose;
213 Matrix31 Ddepth;
214 Matrix32 Dpoint;
215 const Point2 point(-100, 100);
216 const double depth(10);
217 static const Pose3 pose(Rot3::Ypr(y: -M_PI/4.0, M_PI + M_PI/9.5, M_PI/8.0), Point3(0, 0, -10));
218 static const CalibratedCamera camera(pose);
219 camera.backproject(pn: point, depth, Dresult_dpose: Dpose, Dresult_dp: Dpoint, Dresult_ddepth: Ddepth);
220 Matrix expectedDpose = numericalDerivative31(h: backproject, x1: pose, x2: point, x3: depth);
221 Matrix expectedDpoint = numericalDerivative32(h: backproject, x1: pose, x2: point, x3: depth);
222 Matrix expectedDdepth = numericalDerivative33(h: backproject, x1: pose,x2: point, x3: depth);
223
224 EXPECT(assert_equal(expectedDpose, Dpose, 1e-7));
225 EXPECT(assert_equal(expectedDpoint, Dpoint, 1e-7));
226 EXPECT(assert_equal(expectedDdepth, Ddepth, 1e-7));
227}
