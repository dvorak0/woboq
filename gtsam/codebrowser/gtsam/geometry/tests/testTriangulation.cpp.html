38static const std::shared_ptr<Cal3_S2> kSharedCal = //
39 std::make_shared<Cal3_S2>(args: 1500, args: 1200, args: 0.1, args: 640, args: 480);

41// Looking along X-axis, 1 meter above ground plane (x-y)
42static const Rot3 upright = Rot3::Ypr(y: -M_PI / 2, p: 0., r: -M_PI / 2);
43static const Pose3 kPose1 = Pose3(upright, gtsam::Point3(0, 0, 1));
44static const PinholeCamera<Cal3_S2> kCamera1(kPose1, *kSharedCal);
45
46// create second camera 1 meter to the right of first camera

48static const PinholeCamera<Cal3_S2> kCamera2(kPose2, *kSharedCal);
49
50static const std::vector<Pose3> kPoses = {kPose1, kPose2};
51
52
53// landmark ~5 meters in front of camera
54static const Point3 kLandmark(5, 0.5, 1.2);
55
56// 1. Project two landmarks into two cameras and triangulate
57static const Point2 kZ1 = kCamera1.project(pw: kLandmark);
58static const Point2 kZ2 = kCamera2.project(pw: kLandmark);
59static const Point2Vector kMeasurements{kZ1, kZ2};

125TEST(triangulation, twoCamerasLOSTvsDLT) {
126 // LOST has been shown to preform better when the point is much closer to one
127 // camera compared to another. This unit test checks this configuration.
128 const Cal3_S2 identityK;
129 const Pose3 pose1;
130 const Pose3 pose2(Rot3(), Point3(5., 0., -5.));
131 CameraSet<PinholeCamera<Cal3_S2>> cameras;
132 cameras.emplace_back(args: pose1, args: identityK);
133 cameras.emplace_back(args: pose2, args: identityK);
134
135 Point3 landmark(0, 0, 1);
136 Point2 x1_noisy = cameras[0].project(pw: landmark) + Point2(0.00817, 0.00977);
137 Point2 x2_noisy = cameras[1].project(pw: landmark) + Point2(-0.00610, 0.01969);
138 Point2Vector measurements = {x1_noisy, x2_noisy};
139
140 const double rank_tol = 1e-9;
141 SharedNoiseModel measurementNoise = noiseModel::Isotropic::Sigma(dim: 2, sigma: 1e-2);
142
143 std::optional<Point3> estimateLOST = //
144 triangulatePoint3<PinholeCamera<Cal3_S2>>(cameras, measurements, rank_tol,
145 /*optimize=*/false,
146 model: measurementNoise,
147 /*useLOST=*/true);
148
149 // These values are from a MATLAB implementation.
150 EXPECT(assert_equal(Point3(0.007, 0.011, 0.945), *estimateLOST, 1e-3));
151
152 std::optional<Point3> estimateDLT =
153 triangulatePoint3<Cal3_S2>(cameras, measurements, rank_tol, optimize: false);
154
155 // The LOST estimate should have a smaller error.
156 EXPECT((landmark - *estimateLOST).norm() <= (landmark - *estimateDLT).norm());
157}

345TEST(triangulation, threePoses_robustNoiseModel) {
346
347 Pose3 pose3 = kPose1 * Pose3(Rot3::Ypr(y: 0.1, p: 0.2, r: 0.1), Point3(0.1, -2, -.1));
348 PinholeCamera<Cal3_S2> camera3(pose3, *kSharedCal);
349 Point2 z3 = camera3.project(pw: kLandmark);
350
351 const vector<Pose3> poses{kPose1, kPose2, pose3};
352 Point2Vector measurements{kZ1, kZ2, z3};
353
354 // noise free, so should give exactly the landmark
355 std::optional<Point3> actual =
356 triangulatePoint3<Cal3_S2>(poses, sharedCal: kSharedCal, measurements);
357 EXPECT(assert_equal(kLandmark, *actual, 1e-2));
358
359 // Add outlier
360 measurements.at(n: 0) += Point2(100, 120); // very large pixel noise!
361
362 // now estimate does not match landmark
363 std::optional<Point3> actual2 = //
364 triangulatePoint3<Cal3_S2>(poses, sharedCal: kSharedCal, measurements);
365 // DLT is surprisingly robust, but still off (actual error is around 0.26m):
366 EXPECT( (kLandmark - *actual2).norm() >= 0.2);
367 EXPECT( (kLandmark - *actual2).norm() <= 0.5);
368
369 // Again with nonlinear optimization
370 std::optional<Point3> actual3 =
371 triangulatePoint3<Cal3_S2>(poses, sharedCal: kSharedCal, measurements, rank_tol: 1e-9, optimize: true);
372 // result from nonlinear (but non-robust optimization) is close to DLT and still off
373 EXPECT(assert_equal(*actual2, *actual3, 0.1));
374
375 // Again with nonlinear optimization, this time with robust loss
376 auto model = noiseModel::Robust::Create(
377 robust: noiseModel::mEstimator::Huber::Create(k: 1.345), noise: noiseModel::Unit::Create(dim: 2));
378 std::optional<Point3> actual4 = triangulatePoint3<Cal3_S2>(
379 poses, sharedCal: kSharedCal, measurements, rank_tol: 1e-9, optimize: true, model);
380 // using the Huber loss we now have a quite small error!! nice!
381 EXPECT(assert_equal(kLandmark, *actual4, 0.05));
382}
