179namespace exmap_derivative {
180 static const Vector3 w(0.1, 0.27, -0.2);
181}
182// Left trivialized Derivative of exp(w) wrpt w:
183// How does exp(w) change when w changes?
184// We find a y such that: exp(w) exp(y) = exp(w + dw) for dw --> 0
185// => y = log (exp(-w) * exp(w+dw))
186Vector3 testDexpL(const Vector3& dw) {
187 using exmap_derivative::w;
188 return SO3::Logmap(R: SO3::Expmap(omega: -w) * SO3::Expmap(omega: w + dw));
189}

196 EXPECT(assert_equal(expectedDexpL, actualDexpL, 1e-7));
197
198 const Matrix actualDexpInvL = SO3::LogmapDerivative(omega: w);

225TEST(SO3, ExpmapDerivative4) {
226 // Iserles05an (Lie-group Methods) says:
227 // scalar is easy: d exp(a(t)) / dt = exp(a(t)) a'(t)
228 // matrix is hard: d exp(A(t)) / dt = exp(A(t)) dexp[-A(t)] A'(t)
229 // where A(t): R -> so(3) is a trajectory in the tangent space of SO(3)
230 // and dexp[A] is a linear map from 3*3 to 3*3 derivatives of se(3)
231 // Hence, the above matrix equation is typed: 3*3 = SO(3) * linear_map(3*3)
232
233 // In GTSAM, we don't work with the skew-symmetric matrices A directly, but
234 // with 3-vectors.
235 // omega is easy: d Expmap(w(t)) / dt = ExmapDerivative[w(t)] * w'(t)
236
237 // Let's verify the above formula.
238
239 auto w = [](double t) { return Vector3(2 * t, sin(x: t), 4 * t * t); };
240 auto w_dot = [](double t) { return Vector3(2, cos(x: t), 8 * t); };
241
242 // We define a function R
243 auto R = [w](double t) { return SO3::Expmap(omega: w(t)); };
244
245 for (double t = -2.0; t < 2.0; t += 0.3) {
246 const Matrix expected = numericalDerivative11<SO3, double>(h: R, x: t);
247 const Matrix actual = SO3::ExpmapDerivative(omega: w(t)) * w_dot(t);
248 CHECK(assert_equal(expected, actual, 1e-7));
249 }
250}

279TEST(SO3, LogmapDerivative) {
280 const SO3 R0; // Identity
281 const Vector3 omega1(0.1, 0, 0.1);
282 const SO3 R1 = SO3::Expmap(omega: omega1); // Small rotation
283 const SO3 R2((Matrix3() << // Near pi
284 -0.750767, -0.0285082, -0.659952,
285 -0.0102558, -0.998445, 0.0547974,
286 -0.660487, 0.0479084, 0.749307).finished());
287 const SO3 R3((Matrix3() << // Near pi
288 -0.747473, -0.00190019, -0.664289,
289 -0.0385114, -0.99819, 0.0461892,
290 -0.663175, 0.060108, 0.746047).finished());
291 const SO3 R4((Matrix3() << // Final pose in a drone experiment
292 0.324237, 0.902975, 0.281968,
293 -0.674322, 0.429668, -0.600562,
294 -0.663445, 0.00458662, 0.748211).finished());
295 size_t i = 0;
296 for (const SO3& R : { R0, R1, R2, R3, R4 }) {
297 const bool nearPi = (i == 2 || i == 3); // Flag cases near pi
298
299 Matrix3 actualH; // H computed by Logmap(R, H) using LogmapDerivative(omega)
300 const Vector3 omega = SO3::Logmap(R, H: actualH);
301
302 // 1. Check self-consistency of analytical derivative calculation:
303 // Does the H returned by Logmap match an independent calculation
304 // of J_r^{-1} using Local with the computed omega?
305 so3::DexpFunctor local(omega);
306 Matrix3 J_r_inv = local.InvJacobian().right(); // J_r^{-1} via Local
307 EXPECT(assert_equal(J_r_inv, actualH)); // This test is crucial and should pass
308
309 // 2. Check analytical derivative against numerical derivative:
310 // Only perform this check AWAY from the pi singularity, where
311 // numerical differentiation of Logmap is expected to be reliable
312 // and should match the analytical derivative.
313 if (!nearPi) {
314 const Matrix expectedH = numericalDerivative11<Vector3, SO3>(
315 h: std::bind(f: &SO3::Logmap, args: std::placeholders::_1, args: nullptr), x: R, delta: 1e-7);
316 EXPECT(assert_equal(expectedH, actualH, 1e-6)); // 1e-6 needed to pass R4
317 }
318 else {
319 // We accept that the numerical derivative of this specific Logmap implementation
320 // near pi will not match the standard analytical derivative J_r^{-1}.
321 }
322 i++;
323 }
324}
