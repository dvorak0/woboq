17/**
18 * @class FundamentalMatrix
19 * @brief Represents a fundamental matrix in computer vision, which encodes the
20 * epipolar geometry between two views.
21 *
22 * The FundamentalMatrix class encapsulates the fundamental matrix, which
23 * relates corresponding points in stereo images. It is parameterized by two
24 * rotation matrices (U and V) and a scalar parameter (s).
25 * Using these values, the fundamental matrix is represented as
26 *
27 * F = U * diag(1, s, 0) * V^T
28 */
29class GTSAM_EXPORT FundamentalMatrix {
30 private:
31 Rot3 U_; ///< Left rotation
32 double s_; ///< Scalar parameter for S
33 Rot3 V_; ///< Right rotation
34
35 public:
36 /// Default constructor
37 FundamentalMatrix() : U_(Rot3()), s_(1.0), V_(Rot3()) {}
38
39 /**
40 * @brief Construct from U, V, and scalar s
41 *
42 * Initializes the FundamentalMatrix From the SVD representation
43 * U*diag(1,s,0)*V^T. It will internally convert to using SO(3).
44 */
45 FundamentalMatrix(const Matrix3& U, double s, const Matrix3& V);
46
47 /**
48 * @brief Construct from a 3x3 matrix using SVD
49 *
50 * Initializes the FundamentalMatrix by performing SVD on the given
51 * matrix and ensuring U and V are not reflections.
52 *
53 * @param F A 3x3 matrix representing the fundamental matrix
54 */
55 FundamentalMatrix(const Matrix3& F);
56
57 /**
58 * @brief Construct from essential matrix and calibration matrices
59 *
60 * Initializes the FundamentalMatrix from the given essential matrix E
61 * and calibration matrices Ka and Kb, using
62 * F = Ka^(-T) * E * Kb^(-1)
63 * and then calls constructor that decomposes F via SVD.
64 *
65 * @param E Essential matrix
66 * @param Ka Calibration matrix for the left camera
67 * @param Kb Calibration matrix for the right camera
68 */
69 FundamentalMatrix(const Matrix3& Ka, const EssentialMatrix& E,
70 const Matrix3& Kb)
71 : FundamentalMatrix(Ka.transpose().inverse() * E.matrix() *
72 Kb.inverse()) {}
73
74 /**
75 * @brief Construct from calibration matrices Ka, Kb, and pose aPb
76 *
77 * Initializes the FundamentalMatrix from the given calibration
78 * matrices Ka and Kb, and the pose aPb.
79 *
80 * @param Ka Calibration matrix for the left camera
81 * @param aPb Pose from the left to the right camera
82 * @param Kb Calibration matrix for the right camera
83 */
84 FundamentalMatrix(const Matrix3& Ka, const Pose3& aPb, const Matrix3& Kb)
85 : FundamentalMatrix(Ka, EssentialMatrix::FromPose3(1P2_: aPb), Kb) {}
86
87 /// Return the fundamental matrix representation
88 Matrix3 matrix() const;
89
90 /// Computes the epipolar line in a (left) for a given point in b (right)
91 Vector3 epipolarLine(const Point2& p, OptionalJacobian<3, 7> H = {});
92
93 /// @name Testable
94 /// @{
95 /// Print the FundamentalMatrix
96 void print(const std::string& s = "") const;
97
98 /// Check if the FundamentalMatrix is equal to another within a
99 /// tolerance
100 bool equals(const FundamentalMatrix& other, double tol = 1e-9) const;
101 /// @}
102
103 /// @name Manifold
104 /// @{
105 inline constexpr static auto dimension = 7; // 3 for U, 1 for s, 3 for V
106 inline static size_t Dim() { return dimension; }
107 inline size_t dim() const { return dimension; }
108
109 /// Return local coordinates with respect to another FundamentalMatrix
110 Vector localCoordinates(const FundamentalMatrix& F) const;
111
112 /// Retract the given vector to get a new FundamentalMatrix
113 FundamentalMatrix retract(const Vector& delta) const;
114 /// @}
115 private:
116 /// Private constructor for internal use
117 FundamentalMatrix(const Rot3& U, double s, const Rot3& V)
118 : U_(U), s_(s), V_(V) {}
119
120 /// Initialize SO(3) matrices from general O(3) matrices
121 void initialize(Matrix3 U, double s, Matrix3 V);
122};

124/**
125 * @class SimpleFundamentalMatrix
126 * @brief Class for representing a simple fundamental matrix.
127 *
128 * This class represents a simple fundamental matrix, which is a
129 * parameterization of the essential matrix and focal lengths for left and right
130 * cameras. Principal points are not part of the manifold but a convenience.
131 */
132class GTSAM_EXPORT SimpleFundamentalMatrix {
133 private:
134 EssentialMatrix E_; ///< Essential matrix
135 double fa_; ///< Focal length for left camera
136 double fb_; ///< Focal length for right camera
137 Point2 ca_; ///< Principal point for left camera
138 Point2 cb_; ///< Principal point for right camera
139
140 /// Return the left calibration matrix
141 Matrix3 Ka() const;
142
143 /// Return the right calibration matrix
144 Matrix3 Kb() const;
145
146 public:
147 /// Default constructor
148 SimpleFundamentalMatrix()
149 : E_(), fa_(1.0), fb_(1.0), ca_(0.0, 0.0), cb_(0.0, 0.0) {}
150
151 /**
152 * @brief Construct from essential matrix and focal lengths
153 * @param E Essential matrix
154 * @param fa Focal length for left camera
155 * @param fb Focal length for right camera
156 * @param ca Principal point for left camera
157 * @param cb Principal point for right camera
158 */
159 SimpleFundamentalMatrix(const EssentialMatrix& E, //
160 double fa, double fb, const Point2& ca,
161 const Point2& cb)
162 : E_(E), fa_(fa), fb_(fb), ca_(ca), cb_(cb) {}
163
164 /// Return the fundamental matrix representation
165 /// F = Ka^(-T) * E * Kb^(-1)
166 Matrix3 matrix() const;
167
168 /// Computes the epipolar line in a (left) for a given point in b (right)
169 Vector3 epipolarLine(const Point2& p, OptionalJacobian<3, 7> H = {});
170
171 /// @name Testable
172 /// @{
173 /// Print the SimpleFundamentalMatrix
174 void print(const std::string& s = "") const;
175
176 /// Check equality within a tolerance
177 bool equals(const SimpleFundamentalMatrix& other, double tol = 1e-9) const;
178 /// @}
179
180 /// @name Manifold
181 /// @{
182 inline constexpr static auto dimension = 7; // 5 for E, 1 for fa, 1 for fb
183 inline static size_t Dim() { return dimension; }
184 inline size_t dim() const { return dimension; }
185
186 /// Return local coordinates with respect to another SimpleFundamentalMatrix
187 Vector localCoordinates(const SimpleFundamentalMatrix& F) const;
188
189 /// Retract the given vector to get a new SimpleFundamentalMatrix
190 SimpleFundamentalMatrix retract(const Vector& delta) const;
191 /// @}
192};

194/**
195 * @brief Transfer projections from cameras a and b to camera c
196 *
197 * Take two fundamental matrices Fca and Fcb, and two points pa and pb, and
198 * returns the 2D point in view (c) where the epipolar lines intersect.
199 */
200GTSAM_EXPORT Point2 EpipolarTransfer(const Matrix3& Fca, const Point2& pa,
201 const Matrix3& Fcb, const Point2& pb);
202
203/// Represents a set of three fundamental matrices for transferring points
204/// between three cameras.
205template <typename F>
206struct TripleF {
207 F Fab, Fbc, Fca;
208
209 /// Transfers a point from cameras b,c to camera a.
210 Point2 transferToA(const Point2& pb, const Point2& pc) {
211 return EpipolarTransfer(Fab.matrix(), pb, Fca.matrix().transpose(), pc);
212 }
213
214 /// Transfers a point from camera a,c to camera b.
215 Point2 transferToB(const Point2& pa, const Point2& pc) {
216 return EpipolarTransfer(Fab.matrix().transpose(), pa, Fbc.matrix(), pc);
217 }
218
219 /// Transfers a point from cameras a,b to camera c.
220 Point2 transferToC(const Point2& pa, const Point2& pb) {
221 return EpipolarTransfer(Fca.matrix(), pa, Fbc.matrix().transpose(), pb);
222 }
223};
