40/// Calculate dimensionality of SO<N> manifold, or return Dynamic if so
41constexpr int DimensionSO(int N) {
42 return (N < 0) ? Eigen::Dynamic : N * (N - 1) / 2;
43}
44
45// Calculate N^2 at compile time, or return Dynamic if so
46constexpr int NSquaredSO(int N) { return (N < 0) ? Eigen::Dynamic : N * N; }
47} // namespace internal

49/**
50 * Manifold of special orthogonal rotation matrices SO<N>.
51 * Template paramater N can be a fixed integer or can be Eigen::Dynamic
52 */
53template <int N>
54class SO : public MatrixLieGroup<SO<N>, internal::DimensionSO(N), N> {
55 public:
56 inline constexpr static auto dimension = internal::DimensionSO(N);
57 using MatrixNN = Eigen::Matrix<double, N, N>;
58 using VectorN2 = Eigen::Matrix<double, internal::NSquaredSO(N), 1>;
59 using MatrixDD = Eigen::Matrix<double, dimension, dimension>;
60
61 /// LieGroup Concept requirements
62 using LieAlgebra = MatrixNN;
63
64 GTSAM_MAKE_ALIGNED_OPERATOR_NEW_IF(true)
65
66 protected:
67 MatrixNN matrix_; ///< Rotation matrix
68
69 // enable_if_t aliases, used to specialize constructors/methods, see
70 // https://www.fluentcpp.com/2018/05/18/make-sfinae-pretty-2-hidden-beauty-sfinae/
71 template <int N_>
72 using IsDynamic = typename std::enable_if<N_ == Eigen::Dynamic, void>::type;
73 template <int N_>
74 using IsFixed = typename std::enable_if<N_ >= 2, void>::type;
75 template <int N_>
76 using IsSO3 = typename std::enable_if<N_ == 3, void>::type;
77
78 public:
79 /// @name Constructors
80 /// @{
81
82 /// Construct SO<N> identity for N >= 2
83 template <int N_ = N, typename = IsFixed<N_>>
84 SO() : matrix_(MatrixNN::Identity()) {}
85
86 /// Construct SO<N> identity for N == Eigen::Dynamic
87 template <int N_ = N, typename = IsDynamic<N_>>
88 explicit SO(size_t n = 0) {
89 // We allow for n=0 as the default constructor, needed for serialization,
90 // wrappers etc.
91 matrix_ = Eigen::MatrixXd::Identity(rows: n, cols: n);
92 }
93
94 /// Constructor from Eigen Matrix, dynamic version
95 template <typename Derived>
96 explicit SO(const Eigen::MatrixBase<Derived>& R) : matrix_(R.eval()) {}
97
98 /// Named constructor from Eigen Matrix
99 template <typename Derived>
100 static SO FromMatrix(const Eigen::MatrixBase<Derived>& R) {
101 return SO(R);
102 }
103
104 /// Named constructor from lower dimensional matrix
105 template <typename Derived, int N_ = N, typename = IsDynamic<N_>>
106 static SO Lift(size_t n, const Eigen::MatrixBase<Derived> &R) {
107 Matrix Q = Matrix::Identity(rows: n, cols: n);
108 const int p = R.rows();
109 assert(p >= 0 && p <= static_cast<int>(n) && R.cols() == p);
110 Q.topLeftCorner(cRows: p, cCols: p) = R;
111 return SO(Q);
112 }
113
114 /// Construct dynamic SO(n) from Fixed SO<M>
115 template <int M, int N_ = N, typename = IsDynamic<N_>>
116 explicit SO(const SO<M>& R) : matrix_(R.matrix()) {}
117
118 /// Constructor from AngleAxisd
119 template <int N_ = N, typename = IsSO3<N_>>
120 explicit SO(const Eigen::AngleAxisd& angleAxis) : matrix_(angleAxis) {}
121
122 /// Constructor from axis and angle. Only defined for SO3
123 static SO AxisAngle(const Vector3& axis, double theta);
124
125 /// Named constructor that finds SO(n) matrix closest to M in Frobenius norm,
126 /// currently only defined for SO3.
127 static SO ClosestTo(const MatrixNN& M);
128
129 /// Named constructor that finds chordal mean
130 /// \f$ mu = argmin_R \sum sqr(|R-R_i|_F) \f$,
131 /// currently only defined for SO3.
132 static SO ChordalMean(const std::vector<SO>& rotations);
133
134 /// Random SO(n) element (no big claims about uniformity). SO(3) is specialized in SO3.cpp
135 template <int N_ = N, typename = IsDynamic<N_>>
136 static SO Random(std::mt19937& rng, size_t n = 0) {
137 if (n == 0) throw std::runtime_error("SO: Dimensionality not known.");
138 // TODO(frank): this might need to be re-thought
139 static std::uniform_real_distribution<double> randomAngle(-M_PI, M_PI);
140 const size_t d = SO::Dimension(n);
141 Vector xi(d);
142 for (size_t j = 0; j < d; j++) {
143 xi(j) = randomAngle(rng);
144 }
145 return SO::Retract(xi);
146 }
147
148 /// Random SO(N) element (no big claims about uniformity)
149 template <int N_ = N, typename = IsFixed<N_>>
150 static SO Random(std::mt19937& rng) {
151 // By default, use dynamic implementation above. Specialized for SO(3).
152 return SO(SO<Eigen::Dynamic>::Random(rng, n: N).matrix());
153 }
154
155 /// @}
156 /// @name Standard methods
157 /// @{
158
159 /// Return matrix
160 const MatrixNN& matrix() const { return matrix_; }
161
162 size_t rows() const { return matrix_.rows(); }
163 size_t cols() const { return matrix_.cols(); }
164
165 /// @}
166 /// @name Testable
167 /// @{
168
169 void print(const std::string& s = std::string()) const;
170
171 bool equals(const SO& other, double tol) const {
172 return equal_with_abs_tol(matrix_, other.matrix_, tol);
173 }
174
175 /// @}
176 /// @name Group
177 /// @{
178
179 /// Multiplication
180 SO operator*(const SO& other) const {
181 assert(dim() == other.dim());
182 return SO(matrix_ * other.matrix_);
183 }
184
185 /// SO<N> identity for N >= 2
186 template <int N_ = N, typename = IsFixed<N_>>
187 static SO Identity() {
188 return SO();
189 }
190
191 /// SO<N> identity for N == Eigen::Dynamic
192 template <int N_ = N, typename = IsDynamic<N_>>
193 static SO Identity(size_t n = 0) {
194 return SO(n);
195 }
196
197 /// inverse of a rotation = transpose
198 SO inverse() const { return SO(matrix_.transpose()); }
199
200 /// @}
201 /// @name Manifold
202 /// @{
203
204 using TangentVector = Eigen::Matrix<double, dimension, 1>;
205 using ChartJacobian = OptionalJacobian<dimension, dimension>;
206
207 // Calculate manifold dimensionality for SO(n).
208 // Available as dimension or Dim() for fixed N.
209 static size_t Dimension(size_t n) { return n * (n - 1) / 2; }
210
211 // Calculate ambient dimension n from manifold dimensionality d.
212 static size_t AmbientDim(size_t d) { return (1 + std::sqrt(x: 1 + 8 * d)) / 2; }
213
214 // Calculate run-time dimensionality of manifold.
215 size_t dim() const { return Dimension(n: static_cast<size_t>(matrix_.rows())); }
216
217 /**
218 * Hat operator creates Lie algebra element corresponding to d-vector, where d
219 * is the dimensionality of the manifold. This function is implemented
220 * recursively, and the d-vector is assumed to laid out such that the last
221 * element corresponds to so(2), the last 3 to so(3), the last 6 to so(4)
222 * etc... For example, the vector-space isomorphic to so(5) is laid out as:
223 * a b c d | u v w | x y | z
224 * where the latter elements correspond to "telescoping" sub-algebras:
225 * 0 -z y w -d
226 * z 0 -x -v c
227 * -y x 0 u -b
228 * -w v -u 0 a
229 * d -c b -a 0
230 * This scheme behaves exactly as expected for SO(2) and SO(3).
231 */
232 static MatrixNN Hat(const TangentVector& xi);
233
234 /// In-place version of Hat (see details there), implements recursion.
235 static void Hat(const Vector &xi, Eigen::Ref<MatrixNN> X);
236
237 /// Inverse of Hat. See note about xi element order in Hat.
238 static TangentVector Vee(const MatrixNN& X);
239
240 // Chart at origin
241 struct ChartAtOrigin {
242 /**
243 * Retract uses Cayley map. See note about xi element order in Hat.
244 * Deafault implementation has no Jacobian implemented
245 */
246 static SO Retract(const TangentVector& xi, ChartJacobian H = {});
247
248 /**
249 * Inverse of Retract. See note about xi element order in Hat.
250 */
251 static TangentVector Local(const SO& R, ChartJacobian H = {});
252 };
253
254 // Return dynamic identity DxD Jacobian for given SO(n)
255 template <int N_ = N, typename = IsDynamic<N_>>
256 static MatrixDD IdentityJacobian(size_t n) {
257 const size_t d = Dimension(n);
258 return MatrixDD::Identity(d, d);
259 }
260
261 /// @}
262 /// @name Lie Group
263 /// @{
264
265 /// Adjoint map
266 MatrixDD AdjointMap() const {
267 return MatrixLieGroup<SO<N>, internal::DimensionSO(N), N>::AdjointMap();
268 }
269
270 /**
271 * Exponential map at identity - create a rotation from canonical coordinates
272 */
273 static SO Expmap(const TangentVector& omega, ChartJacobian H = {});
274
275 /// Derivative of Expmap, currently only defined for SO3
276 static MatrixDD ExpmapDerivative(const TangentVector& omega);
277
278 /**
279 * Log map at identity - returns the canonical coordinates of this rotation
280 */
281 static TangentVector Logmap(const SO& R, ChartJacobian H = {});
282
283 /// Derivative of Logmap, currently only defined for SO3
284 static MatrixDD LogmapDerivative(const TangentVector& omega);
285
286 // inverse with optional derivative
287 using LieGroup<SO<N>, internal::DimensionSO(N)>::inverse;
288
289 /// @}
290 /// @name Other methods
291 /// @{
292
293 /// Return vectorized rotation matrix in column order.
294 VectorN2 vec(OptionalJacobian<internal::NSquaredSO(N), dimension> H =
295 {}) const {
296 return MatrixLieGroup<SO<N>, internal::DimensionSO(N), N>::vec(H);
297 }
298
299 /// Calculate N^2 x dim matrix of vectorized Lie algebra generators for SO(N)
300 template <int N_ = N, typename = IsFixed<N_>>
301 static Matrix VectorizedGenerators() {
302 constexpr size_t N2 = static_cast<size_t>(N * N);
303 Eigen::Matrix<double, N2, dimension> G;
304 for (size_t j = 0; j < dimension; j++) {
305 const auto X = Hat(Vector::Unit(size: dimension, i: j));
306 G.col(j) = Eigen::Map<const VectorN2>(X.data());
307 }
308 return G;
309 }
310
311 /// Calculate n^2 x dim matrix of vectorized Lie algebra generators for SO(n)
312 template <int N_ = N, typename = IsDynamic<N_>>
313 static Matrix VectorizedGenerators(size_t n = 0) {
314 const size_t n2 = n * n, dim = Dimension(n);
315 Matrix G(n2, dim);
316 for (size_t j = 0; j < dim; j++) {
317 const auto X = Hat(Vector::Unit(size: dim, i: j));
318 G.col(i: j) = Eigen::Map<const Matrix>(X.data(), n2, 1);
319 }
320 return G;
321 }
