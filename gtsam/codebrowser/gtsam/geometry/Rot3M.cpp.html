55/* ************************************************************************* */
56Rot3 Rot3::Rx(double t) {
57 double st = sin(x: t), ct = cos(x: t);
58 return Rot3(
59 1, 0, 0,
60 0, ct,-st,
61 0, st, ct);
62}
63
64/* ************************************************************************* */
65Rot3 Rot3::Ry(double t) {
66 double st = sin(x: t), ct = cos(x: t);
67 return Rot3(
68 ct, 0, st,
69 0, 1, 0,
70 -st, 0, ct);
71}
72
73/* ************************************************************************* */
74Rot3 Rot3::Rz(double t) {
75 double st = sin(x: t), ct = cos(x: t);
76 return Rot3(
77 ct,-st, 0,
78 st, ct, 0,
79 0, 0, 1);
80}

83// Considerably faster than composing matrices above !
84Rot3 Rot3::RzRyRx(double x, double y, double z, OptionalJacobian<3, 1> Hx,
85 OptionalJacobian<3, 1> Hy, OptionalJacobian<3, 1> Hz) {
86 double cx=cos(x: x),sx=sin(x: x);
87 double cy=cos(x: y),sy=sin(x: y);
88 double cz=cos(x: z),sz=sin(x: z);
89 double ss_ = sx * sy;
90 double cs_ = cx * sy;
91 double sc_ = sx * cy;
92 double cc_ = cx * cy;
93 double c_s = cx * sz;
94 double s_s = sx * sz;
95 double _cs = cy * sz;
96 double _cc = cy * cz;
97 double s_c = sx * cz;
98 double c_c = cx * cz;
99 double ssc = ss_ * cz, csc = cs_ * cz, sss = ss_ * sz, css = cs_ * sz;
100 if (Hx) (*Hx) << 1, 0, 0;
101 if (Hy) (*Hy) << 0, cx, -sx;
102 if (Hz) (*Hz) << -sy, sc_, cc_;
103 return Rot3(
104 _cc,- c_s + ssc, s_s + csc,
105 _cs, c_c + sss, -s_c + css,
106 -sy, sc_, cc_
107 );
108}

110/* ************************************************************************* */
111Rot3 Rot3::normalized() const {
112 /// Implementation from here: https://stackoverflow.com/a/23082112/1236990
113
114 /// Essentially, this computes the orthogonalization error, distributes the
115 /// error to the x and y rows, and then performs a Taylor expansion to
116 /// orthogonalize.
117
118 Matrix3 rot = rot_.matrix(), rot_orth;
119
120 // Check if determinant is already 1.
121 // If yes, then return the current Rot3.
122 if (std::fabs(x: rot.determinant()-1) < 1e-12) return Rot3(rot_);
123
124 Vector3 x = rot.block<1, 3>(startRow: 0, startCol: 0), y = rot.block<1, 3>(startRow: 1, startCol: 0);
125 double error = x.dot(other: y);
126
127 Vector3 x_ort = x - (error / 2) * y, y_ort = y - (error / 2) * x;
128 Vector3 z_ort = x_ort.cross(other: y_ort);
129
130 rot_orth.block<1, 3>(startRow: 0, startCol: 0) = 0.5 * (3 - x_ort.dot(other: x_ort)) * x_ort;
131 rot_orth.block<1, 3>(startRow: 1, startCol: 0) = 0.5 * (3 - y_ort.dot(other: y_ort)) * y_ort;
132 rot_orth.block<1, 3>(startRow: 2, startCol: 0) = 0.5 * (3 - z_ort.dot(other: z_ort)) * z_ort;
133
134 return Rot3(rot_orth);
135}

147/* ************************************************************************* */
148Point3 Rot3::rotate(const Point3& p,
149 OptionalJacobian<3,3> H1, OptionalJacobian<3,3> H2) const {
150 if (H1) *H1 = rot_.matrix() * skewSymmetric(wx: -p.x(), wy: -p.y(), wz: -p.z());
151 if (H2) *H2 = rot_.matrix();
152 return rot_.matrix() * p;
153}

156Rot3 Rot3::Expmap(const Vector3& omega, OptionalJacobian<3, 3> H) {
157 so3::DexpFunctor local(omega);
158 if (H) *H = local.rightJacobian();
159 const Matrix3 M = local.expmap();
160 return Rot3(M);
161}

168/* ************************************************************************* */
169Rot3 Rot3::CayleyChart::Retract(const Vector3& omega, OptionalJacobian<3,3> H) {
170 if (H) throw std::runtime_error("Rot3::CayleyChart::Retract Derivative");
171 const double x = omega(0), y = omega(1), z = omega(2);
172 const double x2 = x * x, y2 = y * y, z2 = z * z;
173 const double xy = x * y, xz = x * z, yz = y * z;
174 const double f = 1.0 / (4.0 + x2 + y2 + z2), _2f = 2.0 * f;
175 return Rot3((4 + x2 - y2 - z2) * f, (xy - 2 * z) * _2f, (xz + 2 * y) * _2f,
176 (xy + 2 * z) * _2f, (4 - x2 + y2 - z2) * f, (yz - 2 * x) * _2f,
177 (xz - 2 * y) * _2f, (yz + 2 * x) * _2f, (4 - x2 - y2 + z2) * f);
178}

180/* ************************************************************************* */
181Vector3 Rot3::CayleyChart::Local(const Rot3& R, OptionalJacobian<3,3> H) {
182 if (H) throw std::runtime_error("Rot3::CayleyChart::Local Derivative");
183 // Create a fixed-size matrix
184 Matrix3 A = R.matrix();
185
186 // Check if (A+I) is invertible. Same as checking for -1 eigenvalue.
187 if ((A + I_3x3).determinant() == 0.0) {
188 throw std::runtime_error("Rot3::CayleyChart::Local Invalid Rotation");
189 }
190
191 // Mathematica closed form optimization.
192 // The following are the essential computations for the following algorithm
193 // 1. Compute the inverse of P = (A+I), using a closed-form formula since P is 3x3
194 // 2. Compute the Cayley transform C = 2 * P^{-1} * (A-I)
195 // 3. C is skew-symmetric, so we pick out the computations corresponding only to x, y, and z.
196 const double a = A(0, 0), b = A(0, 1), c = A(0, 2);
197 const double d = A(1, 0), e = A(1, 1), f = A(1, 2);
198 const double g = A(2, 0), h = A(2, 1), i = A(2, 2);
199 const double di = d * i, ce = c * e, cd = c * d, fg = f * g;
200 const double M = 1 + e - f * h + i + e * i;
201 const double K = -4.0 / (cd * h + M + a * M - g * (c + ce) - b * (d + di - fg));
202 const double x = a * f - cd + f;
203 const double y = b * f - ce - c;
204 const double z = fg - di - d;
205 return K * Vector3(x, y, z);
206}

208/* ************************************************************************* */
209Rot3 Rot3::ChartAtOrigin::Retract(const Vector3& omega, ChartJacobian H) {
210 static const CoordinatesMode mode = ROT3_DEFAULT_COORDINATES_MODE;
211 if (mode == Rot3::EXPMAP) return Expmap(omega, H);
212 if (mode == Rot3::CAYLEY) return CayleyChart::Retract(omega, H);
213 else throw std::runtime_error("Rot3::Retract: unknown mode");
214}

216/* ************************************************************************* */
217Vector3 Rot3::ChartAtOrigin::Local(const Rot3& R, ChartJacobian H) {
218 static const CoordinatesMode mode = ROT3_DEFAULT_COORDINATES_MODE;
219 if (mode == Rot3::EXPMAP) return Logmap(R, H);
220 if (mode == Rot3::CAYLEY) return CayleyChart::Local(R, H);
221 else throw std::runtime_error("Rot3::Local: unknown mode");
222}
