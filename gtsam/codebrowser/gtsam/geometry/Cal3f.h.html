18/**
19 * @file Cal3f.h
20 * @brief Calibration model with a single focal length and zero skew.
21 * @author Frank Dellaert
22 */
23
24#pragma once
25
26#include <gtsam/geometry/Cal3.h>
27
28namespace gtsam {
29
30/**
31 * @brief Calibration model with a single focal length and zero skew.
32 * @ingroup geometry
33 * \nosubgrouping
34 */
35class GTSAM_EXPORT Cal3f : public Cal3 {
36 public:
37 constexpr static auto dimension = 1;
38 using shared_ptr = std::shared_ptr<Cal3f>;
39
40 /// @name Constructors
41 /// @{
42
43 /// Default constructor
44 Cal3f() = default;
45
46 /**
47 * Constructor
48 * @param f focal length
49 * @param u0 image center x-coordinate (considered a constant)
50 * @param v0 image center y-coordinate (considered a constant)
51 */
52 Cal3f(double f, double u0, double v0);
53
54 ~Cal3f() override = default;
55
56 /// @}
57 /// @name Testable
58 /// @{
59
60 /// Output stream operator
61 GTSAM_EXPORT friend std::ostream& operator<<(std::ostream& os, const Cal3f& cal);
62
63 /// print with optional string
64 void print(const std::string& s = "") const override;
65
66 /// assert equality up to a tolerance
67 bool equals(const Cal3f& K, double tol = 1e-9) const;
68
69 /// @}
70 /// @name Standard Interface
71 /// @{
72
73 /// focal length
74 double f() const { return fx_; }
75
76 /// vectorized form (column-wise)
77 Vector1 vector() const;
78
79 /**
80 * @brief: convert intrinsic coordinates xy to image coordinates uv
81 * Version of uncalibrate with derivatives
82 * @param p point in intrinsic coordinates
83 * @param Dcal optional 2*1 Jacobian wrpt focal length
84 * @param Dp optional 2*2 Jacobian wrpt intrinsic coordinates
85 * @return point in image coordinates
86 */
87 Point2 uncalibrate(const Point2& p, OptionalJacobian<2, 1> Dcal = {},
88 OptionalJacobian<2, 2> Dp = {}) const;
89
90 /**
91 * Convert a pixel coordinate to ideal coordinate xy
92 * @param pi point in image coordinates
93 * @param Dcal optional 2*1 Jacobian wrpt focal length
94 * @param Dp optional 2*2 Jacobian wrpt intrinsic coordinates
95 * @return point in intrinsic coordinates
96 */
97 Point2 calibrate(const Point2& pi, OptionalJacobian<2, 1> Dcal = {},
98 OptionalJacobian<2, 2> Dp = {}) const;
99
100 /// @}
101 /// @name Manifold
102 /// @{
103
104 /// Return DOF, dimensionality of tangent space
105 size_t dim() const { return Dim(); }
106
107 /// Return DOF, dimensionality of tangent space
108 static size_t Dim() { return dimension; }
109
110 /// Update calibration with tangent space delta
111 Cal3f retract(const Vector& d) const { return Cal3f(fx_ + d(0), u0_, v0_); }
112
113 /// Calculate local coordinates to another calibration
114 Vector1 localCoordinates(const Cal3f& T2) const {
115 return Vector1(T2.fx_ - fx_);
116 }
117
118 /// @}
119
120 private:
121#if GTSAM_ENABLE_BOOST_SERIALIZATION
122 /** Serialization function */
123 friend class boost::serialization::access;
124 template <class Archive>
125 void serialize(Archive& ar, const unsigned int /*version*/) {
126 ar& BOOST_SERIALIZATION_NVP(fx_);
127 ar& BOOST_SERIALIZATION_NVP(u0_);
128 ar& BOOST_SERIALIZATION_NVP(v0_);
129 }
130#endif
131
132 /// @}
133};
134
135template <>
136struct traits<Cal3f> : public internal::Manifold<Cal3f> {};
137
138template <>
139struct traits<const Cal3f> : public internal::Manifold<Cal3f> {};
140
141} // namespace gtsam
