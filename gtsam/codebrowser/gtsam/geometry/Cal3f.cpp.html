62/* ************************************************************************* */
63Point2 Cal3f::uncalibrate(const Point2& p, OptionalJacobian<2, 1> Dcal,
64 OptionalJacobian<2, 2> Dp) const {
65 assert(fx_ == fy_ && s_ == 0.0);
66 const double x = p.x(), y = p.y();
67 const double u = fx_ * x + u0_;
68 const double v = fx_ * y + v0_;
69
70 if (Dcal) {
71 Dcal->resize(rows: 2, cols: 1);
72 (*Dcal) << x, y;
73 }
74
75 if (Dp) {
76 Dp->resize(rows: 2, cols: 2);
77 (*Dp) << fx_, 0.0, //
78 0.0, fx_;
79 }
80
81 return Point2(u, v);
82}
83
84/* ************************************************************************* */
85Point2 Cal3f::calibrate(const Point2& pi, OptionalJacobian<2, 1> Dcal,
86 OptionalJacobian<2, 2> Dp) const {
87 assert(fx_ == fy_ && s_ == 0.0);
88 const double u = pi.x(), v = pi.y();
89 double delta_u = u - u0_, delta_v = v - v0_;
90 double inv_f = 1.0 / fx_;
91 Point2 point(inv_f * delta_u, inv_f * delta_v);
92
93 if (Dcal) {
94 Dcal->resize(rows: 2, cols: 1);
95 (*Dcal) << -inv_f * point.x(), -inv_f * point.y();
96 }
97
98 if (Dp) {
99 Dp->resize(rows: 2, cols: 2);
100 (*Dp) << inv_f, 0.0, //
101 0.0, inv_f;
102 }
103
104 return point;
105}
