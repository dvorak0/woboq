36/* ************************************************************************* */
37Matrix3 Pose2::matrix() const {
38 Matrix2 R = r_.matrix();
39 Matrix32 R0;
40 R0.block<2,2>(startRow: 0,startCol: 0) = R;
41 R0.block<1,2>(startRow: 2,startCol: 0).setZero();
42 Matrix31 T;
43 T << t_.x(), t_.y(), 1.0;
44 Matrix3 RT_;
45 RT_.block<3,2>(startRow: 0,startCol: 0) = R0;
46 RT_.block<3,1>(startRow: 0,startCol: 2) = T;
47 return RT_;
48}

50/* ************************************************************************* */
51Vector9 Pose2::vec(OptionalJacobian<9, 3> H) const {
52 // Vectorize
53 const Matrix3 M = matrix();
54 const Vector9 v = Eigen::Map<const Vector9>(M.data());
55
56 // If requested, calculate H
57 if (H) {
58 H->setZero();
59 auto R = M.block<2, 2>(startRow: 0, startCol: 0);
60 H->block<2, 1>(startRow: 0, startCol: 2) = R.col(i: 1);
61 H->block<2, 1>(startRow: 3, startCol: 2) = -R.col(i: 0);
62 H->block<2, 2>(startRow: 6, startCol: 0) = R;
63 }
64
65 return v;
66}

84/* ************************************************************************* */
85Pose2 Pose2::Expmap(const Vector3& xi, OptionalJacobian<3, 3> H) {
86 assert(xi.size() == 3);
87 if (H) *H = Pose2::ExpmapDerivative(v: xi);
88 const Point2 v(xi(0),xi(1));
89 const double w = xi(2);
90 if (std::abs(x: w) < 1e-10)
91 return Pose2(xi[0], xi[1], xi[2]);
92 else {
93 const Rot2 R(Rot2::fromAngle(theta: w));
94 const Point2 v_ortho = R_PI_2 * v; // points towards rot center
95 const Point2 t = (v_ortho - R.rotate(p: v_ortho)) / w;
96 return Pose2(R, t);
97 }
98}

100/* ************************************************************************* */
101Vector3 Pose2::Logmap(const Pose2& p, OptionalJacobian<3, 3> H) {
102 if (H) *H = Pose2::LogmapDerivative(v: p);
103 const Rot2& R = p.r();
104 const Point2& t = p.t();
105 double w = R.theta();
106 if (std::abs(x: w) < 1e-10)
107 return Vector3(t.x(), t.y(), w);
108 else {
109 double c_1 = R.c()-1.0, s = R.s();
110 double det = c_1*c_1 + s*s;
111 Point2 p = R_PI_2 * (R.unrotate(p: t) - t);
112 Point2 v = (w/det) * p;
113 return Vector3(v.x(), v.y(), w);
114 }
115}

143// Calculate Adjoint map
144// Ad_pose is 3*3 matrix that when applied to twist xi, returns Ad_pose(xi)
145Matrix3 Pose2::AdjointMap() const {
146 double c = r_.c(), s = r_.s(), x = t_.x(), y = t_.y();
147 Matrix3 rvalue;
148 rvalue <<
149 c, -s, y,
150 s, c, -x,
151 0.0, 0.0, 1.0;
152 return rvalue;
153}

155/* ************************************************************************* */
156Matrix3 Pose2::adjointMap(const Vector3& v) {
157 // See Chirikjian12book2, vol.2, pg. 36
158 Matrix3 ad = Z_3x3;
159 ad(0,1) = -v[2];
160 ad(1,0) = v[2];
161 ad(0,2) = v[1];
162 ad(1,2) = -v[0];
163 return ad;
164}

166/* ************************************************************************* */
167Matrix3 Pose2::ExpmapDerivative(const Vector3& v) {
168 double alpha = v[2];
169 Matrix3 J;
170 if (std::abs(x: alpha) > 1e-5) {
171 // Chirikjian11book2, pg. 36
172 /* !!!Warning!!! Compare Iserles05an, formula 2.42 and Chirikjian11book2 pg.26
173 * Iserles' right-trivialization dexpR is actually the left Jacobian J_l in Chirikjian's notation
174 * In fact, Iserles 2.42 can be written as:
175 * \dot{g} g^{-1} = dexpR_{q}\dot{q}
176 * where q = A, and g = exp(A)
177 * and the LHS is in the definition of J_l in Chirikjian11book2, pg. 26.
178 * Hence, to compute ExpmapDerivative, we have to use the formula of J_r Chirikjian11book2, pg.36
179 */
180 double sZalpha = sin(x: alpha)/alpha, c_1Zalpha = (cos(x: alpha)-1)/alpha;
181 double v1Zalpha = v[0]/alpha, v2Zalpha = v[1]/alpha;
182 J << sZalpha, -c_1Zalpha, v1Zalpha + v2Zalpha*c_1Zalpha - v1Zalpha*sZalpha,
183 c_1Zalpha, sZalpha, -v1Zalpha*c_1Zalpha + v2Zalpha - v2Zalpha*sZalpha,
184 0, 0, 1;
185 }
186 else {
187 // Thanks to Krunal: Apply L'Hospital rule to several times to
188 // compute the limits when alpha -> 0
189 J << 1,0,-0.5*v[1],
190 0,1, 0.5*v[0],
191 0,0, 1;
192 }
193
194 return J;
195}

197/* ************************************************************************* */
198Matrix3 Pose2::LogmapDerivative(const Pose2& p) {
199 Vector3 v = Logmap(p);
200 double alpha = v[2];
201 Matrix3 J;
202 if (std::abs(x: alpha) > 1e-5) {
203 double alphaInv = 1/alpha;
204 double halfCotHalfAlpha = 0.5*sin(x: alpha)/(1-cos(x: alpha));
205 double v1 = v[0], v2 = v[1];
206
207 J << alpha*halfCotHalfAlpha, -0.5*alpha, v1*alphaInv - v1*halfCotHalfAlpha + 0.5*v2,
208 0.5*alpha, alpha*halfCotHalfAlpha, v2*alphaInv - 0.5*v1 - v2*halfCotHalfAlpha,
209 0, 0, 1;
210 }
211 else {
212 J << 1,0, 0.5*v[1],
213 0,1, -0.5*v[0],
214 0,0, 1;
215 }
216 return J;
217}

239// see doc/math.lyx, SE(2) section
240Point2 Pose2::transformTo(const Point2& point,
241 OptionalJacobian<2, 3> Hpose, OptionalJacobian<2, 2> Hpoint) const {
242 OptionalJacobian<2, 2> Htranslation = Hpose.cols<2>(startCol: 0);
243 OptionalJacobian<2, 1> Hrotation = Hpose.cols<1>(startCol: 2);
244 const Point2 q = r_.unrotate(p: point - t_, H1: Hrotation, H2: Hpoint);
245 if (Htranslation) *Htranslation << -1.0, 0.0, 0.0, -1.0;
246 return q;
247}

249Matrix Pose2::transformTo(const Matrix& points) const {
250 if (points.rows() != 2) {
251 throw std::invalid_argument("Pose2:transformTo expects 2*N matrix.");
252 }
253 const Matrix2 Rt = rotation().transpose();
254 return Rt * (points.colwise() - t_); // Eigen broadcasting!
255}

258// see doc/math.lyx, SE(2) section
259Point2 Pose2::transformFrom(const Point2& point,
260 OptionalJacobian<2, 3> Hpose, OptionalJacobian<2, 2> Hpoint) const {
261 OptionalJacobian<2, 2> Htranslation = Hpose.cols<2>(startCol: 0);
262 OptionalJacobian<2, 1> Hrotation = Hpose.cols<1>(startCol: 2);
263 const Point2 q = r_.rotate(p: point, H1: Hrotation, H2: Hpoint);
264 if (Htranslation) *Htranslation = (Hpoint ? *Hpoint : r_.matrix());
265 return q + t_;
266}

278Rot2 Pose2::bearing(const Point2& point,
279 OptionalJacobian<1, 3> Hpose, OptionalJacobian<1, 2> Hpoint) const {
280 // make temporary matrices
281 Matrix23 D_d_pose; Matrix2 D_d_point;
282 Point2 d = transformTo(point, Hpose: Hpose ? &D_d_pose : 0, Hpoint: Hpoint ? &D_d_point : 0);
283 if (!Hpose && !Hpoint) return Rot2::relativeBearing(d);
284 Matrix12 D_result_d;
285 Rot2 result = Rot2::relativeBearing(d, H: Hpose || Hpoint ? &D_result_d : 0);
286 if (Hpose) *Hpose = D_result_d * D_d_pose;
287 if (Hpoint) *Hpoint = D_result_d * D_d_point;
288 return result;
289}

302/* ************************************************************************* */
303double Pose2::range(const Point2& point,
304 OptionalJacobian<1,3> Hpose, OptionalJacobian<1,2> Hpoint) const {
305 Point2 d = point - t_;
306 if (!Hpose && !Hpoint) return d.norm();
307 Matrix12 D_r_d;
308 double r = norm2(p: d, H: D_r_d);
309 if (Hpose) {
310 Matrix23 D_d_pose;
311 D_d_pose << -r_.c(), r_.s(), 0.0,
312 -r_.s(), -r_.c(), 0.0;
313 *Hpose = D_r_d * D_d_pose;
314 }
315 if (Hpoint) *Hpoint = D_r_d;
316 return r;
317}

319/* ************************************************************************* */
320double Pose2::range(const Pose2& pose,
321 OptionalJacobian<1,3> Hpose,
322 OptionalJacobian<1,3> Hother) const {
323 Point2 d = pose.t() - t_;
324 if (!Hpose && !Hother) return d.norm();
325 Matrix12 D_r_d;
326 double r = norm2(p: d, H: D_r_d);
327 if (Hpose) {
328 Matrix23 D_d_pose;
329 D_d_pose <<
330 -r_.c(), r_.s(), 0.0,
331 -r_.s(), -r_.c(), 0.0;
332 *Hpose = D_r_d * D_d_pose;
333 }
334 if (Hother) {
335 Matrix23 D_d_other;
336 D_d_other <<
337 pose.r_.c(), -pose.r_.s(), 0.0,
338 pose.r_.s(), pose.r_.c(), 0.0;
339 *Hother = D_r_d * D_d_other;
340 }
341 return r;
342}

348/* *************************************************************************
349 * Align finds the angle using a linear method:
350 * a = Pose2::transformFrom(b) = t + R*b
351 * We need to remove the centroids from the data to find the rotation
352 * using db=[dbx;dby] and a=[dax;day] we have
353 * |dax| |c -s| |dbx| |dbx -dby| |c|
354 * | | = | | * | | = | | * | | = H_i*cs
355 * |day| |s c| |dby| |dby dbx| |s|
356 * where the Hi are the 2*2 matrices. Then we will minimize the criterion
357 * J = \sum_i norm(a_i - H_i * cs)
358 * Taking the derivative with respect to cs and setting to zero we have
359 * cs = (\sum_i H_i' * a_i)/(\sum H_i'*H_i)
360 * The hessian is diagonal and just divides by a constant, but this
361 * normalization constant is irrelevant, since we take atan2.
362 * i.e., cos ~ sum(dbx*dax + dby*day) and sin ~ sum(-dby*dax + dbx*day)
363 * The translation is then found from the centroids
364 * as they also satisfy ca = t + R*cb, hence t = ca - R*cb
365 */
366
367std::optional<Pose2> Pose2::Align(const Point2Pairs &ab_pairs) {
368 const size_t n = ab_pairs.size();
369 if (n < 2) {
370 return {}; // we need at least 2 pairs
371 }
372
373 // calculate centroids
374 Point2 ca(0, 0), cb(0, 0);
375 for (const Point2Pair& pair : ab_pairs) {
376 ca += pair.first;
377 cb += pair.second;
378 }
379 const double f = 1.0/n;
380 ca *= f;
381 cb *= f;
382
383 // calculate cos and sin
384 double c = 0, s = 0;
385 for (const Point2Pair& pair : ab_pairs) {
386 Point2 da = pair.first - ca;
387 Point2 db = pair.second - cb;
388 c += db.x() * da.x() + db.y() * da.y();
389 s += -db.y() * da.x() + db.x() * da.y();
390 }
391
392 // calculate angle and translation
393 const double theta = atan2(y: s, x: c);
394 const Rot2 R = Rot2::fromAngle(theta);
395 const Point2 t = ca - R*cb;
396 return Pose2(R, t);
397}

399std::optional<Pose2> Pose2::Align(const Matrix& a, const Matrix& b) {
400 if (a.rows() != 2 || b.rows() != 2 || a.cols() != b.cols()) {
401 throw std::invalid_argument(
402 "Pose2:Align expects 2*N matrices of equal shape.");
403 }
404 Point2Pairs ab_pairs;
405 for (Eigen::Index j = 0; j < a.cols(); j++) {
406 ab_pairs.emplace_back(args: a.col(i: j), args: b.col(i: j));
407 }
408 return Pose2::Align(ab_pairs);
409}
