22#pragma once
23
24#include <gtsam/geometry/Cal3.h>
25#include <gtsam/geometry/Point2.h>
26
27namespace gtsam {
28
29/**
30 * @brief The most common 5DOF 3D->2D calibration
31 * @ingroup geometry
32 * \nosubgrouping
33 */
34class GTSAM_EXPORT Cal3_S2 : public Cal3 {
35 public:
36 constexpr static auto dimension = 5;
37
38 ///< shared pointer to calibration object
39 using shared_ptr = std::shared_ptr<Cal3_S2>;
40
41 /// @name Standard Constructors
42 /// @{
43
44 /// Create a default calibration that leaves coordinates unchanged
45 Cal3_S2() = default;
46
47 /// constructor from doubles
48 Cal3_S2(double fx, double fy, double s, double u0, double v0)
49 : Cal3(fx, fy, s, u0, v0) {}
50
51 /// constructor from vector
52 Cal3_S2(const Vector5& d) : Cal3(d) {}
53
54 /**
55 * Easy constructor, takes fov in degrees, asssumes zero skew, unit aspect
56 * @param fov field of view in degrees
57 * @param w image width
58 * @param h image height
59 */
60 Cal3_S2(double fov, int w, int h) : Cal3(fov, w, h) {}
61
62 /**
63 * Convert intrinsic coordinates xy to image coordinates uv, fixed derivatives
64 * @param p point in intrinsic coordinates
65 * @param Dcal optional 2*5 Jacobian wrpt Cal3 parameters
66 * @param Dp optional 2*2 Jacobian wrpt intrinsic coordinates
67 * @return point in image coordinates
68 */
69 Point2 uncalibrate(const Point2& p, OptionalJacobian<2, 5> Dcal = {},
70 OptionalJacobian<2, 2> Dp = {}) const;
71
72 /**
73 * Convert image coordinates uv to intrinsic coordinates xy
74 * @param p point in image coordinates
75 * @param Dcal optional 2*5 Jacobian wrpt Cal3 parameters
76 * @param Dp optional 2*2 Jacobian wrpt intrinsic coordinates
77 * @return point in intrinsic coordinates
78 */
79 Point2 calibrate(const Point2& p, OptionalJacobian<2, 5> Dcal = {},
80 OptionalJacobian<2, 2> Dp = {}) const;
81
82 /**
83 * Convert homogeneous image coordinates to intrinsic coordinates
84 * @param p point in image coordinates
85 * @return point in intrinsic coordinates
86 */
87 Vector3 calibrate(const Vector3& p) const;
88
89 /// @}
90 /// @name Testable
91 /// @{
92
93 /// Output stream operator
94 GTSAM_EXPORT friend std::ostream& operator<<(std::ostream& os,
95 const Cal3_S2& cal);
96
97 /// print with optional string
98 void print(const std::string& s = "Cal3_S2") const override;
99
100 /// Check if equal up to specified tolerance
101 bool equals(const Cal3_S2& K, double tol = 10e-9) const;
102
103 /// "Between", subtracts calibrations. between(p,q) == compose(inverse(p),q)
104 Cal3_S2 between(const Cal3_S2& q,
105 OptionalJacobian<5, 5> H1 = {},
106 OptionalJacobian<5, 5> H2 = {}) const {
107 if (H1) *H1 = -I_5x5;
108 if (H2) *H2 = I_5x5;
109 return Cal3_S2(q.fx_ - fx_, q.fy_ - fy_, q.s_ - s_, q.u0_ - u0_,
110 q.v0_ - v0_);
111 }
112
113 /// @}
114 /// @name Manifold
115 /// @{
116
117 /// return DOF, dimensionality of tangent space
118 virtual size_t dim() const { return Dim(); };
119
120 /// return DOF, dimensionality of tangent space
121 static size_t Dim() { return dimension; }
122
123 /// Given 5-dim tangent vector, create new calibration
124 Cal3_S2 retract(const Vector& d) const {
125 return Cal3_S2(fx_ + d(0), fy_ + d(1), s_ + d(2), u0_ + d(3), v0_ + d(4));
126 }
127
128 /// Unretraction for the calibration
129 Vector5 localCoordinates(const Cal3_S2& T2) const {
130 return T2.vector() - vector();
131 }
132
133 /// @}
134 /// @name Advanced Interface
135 /// @{
136
137 private:
138#if GTSAM_ENABLE_BOOST_SERIALIZATION ///
139 /// Serialization function
140 friend class boost::serialization::access;
141 template <class Archive>
142 void serialize(Archive& ar, const unsigned int /*version*/) {
143 ar& boost::serialization::make_nvp(
144 n: "Cal3_S2", v&: boost::serialization::base_object<Cal3>(d&: *this));
145 }
146#endif
147
148 /// @}
149};
150
151template <>
152struct traits<Cal3_S2> : public internal::Manifold<Cal3_S2> {};
153
154template <>
155struct traits<const Cal3_S2> : public internal::Manifold<Cal3_S2> {};
156
157} // \ namespace gtsam
