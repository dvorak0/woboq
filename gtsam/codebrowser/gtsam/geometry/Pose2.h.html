34/**
35 * A 2D pose (Point2,Rot2)
36 * @ingroup geometry
37 * \nosubgrouping
38 */
39class GTSAM_EXPORT Pose2: public MatrixLieGroup<Pose2, 3, 3> {
40
41public:
42
43 /// Pose Concept requirements
44 using Rotation = Rot2;
45 using Translation = Point2;
46
47 /// LieGroup Concept requirements
48 using LieAlgebra = Matrix3;
49
50private:
51
52 Rot2 r_;
53 Point2 t_;
54
55public:
56
57 /// @name Standard Constructors
58 /// @{
59
60 /** default constructor = origin */
61 Pose2() :
62 r_(traits<Rot2>::Identity()), t_(traits<Point2>::Identity()) {
63 }
64
65 /** copy constructor */
66 Pose2(const Pose2& pose) = default;
67 // : r_(pose.r_), t_(pose.t_) {}
68
69 Pose2& operator=(const Pose2& other) = default;
70
71 /**
72 * construct from (x,y,theta)
73 * @param x x coordinate
74 * @param y y coordinate
75 * @param theta angle with positive X-axis
76 */
77 Pose2(double x, double y, double theta) :
78 r_(Rot2::fromAngle(theta)), t_(x, y) {
79 }
80
81 /** construct from rotation and translation */
82 Pose2(double theta, const Point2& t) :
83 r_(Rot2::fromAngle(theta)), t_(t) {
84 }
85
86 /** construct from r,t */
87 Pose2(const Rot2& r, const Point2& t) : r_(r), t_(t) {}
88
89 /** Constructor from 3*3 matrix */
90 Pose2(const Matrix &T)
91 : r_(Rot2::atan2(y: T(1, 0), x: T(0, 0))), t_(T(0, 2), T(1, 2)) {
92#ifndef NDEBUG
93 if (T.rows() != 3 || T.cols() != 3) {
94 throw;
95 }
96#endif
97 }
98
99 /// @}
100 /// @name Advanced Constructors
101 /// @{
102
103 /** Construct from canonical coordinates \f$ [T_x,T_y,\theta] \f$ (Lie algebra) */
104 Pose2(const Vector& v) : Pose2() {
105 *this = Expmap(xi: v);
106 }
107
108 /**
109 * Create Pose2 by aligning two point pairs
110 * A pose aTb is estimated between pairs (a_point, b_point) such that
111 * a_point = aTb * b_point
112 * Note this allows for noise on the points but in that case the mapping
113 * will not be exact.
114 */
115 static std::optional<Pose2> Align(const Point2Pairs& abPointPairs);
116
117 // Version of Pose2::Align that takes 2 matrices.
118 static std::optional<Pose2> Align(const Matrix& a, const Matrix& b);
119
120 /// @}
121 /// @name Testable
122 /// @{
123
124 /** print with optional string */
125 void print(const std::string& s = "") const;
126
127 /** assert equality up to a tolerance */
128 bool equals(const Pose2& pose, double tol = 1e-9) const;
129
130 /// @}
131 /// @name Group
132 /// @{
133
134 /// identity for group operation
135 inline static Pose2 Identity() { return Pose2(); }
136
137 /// inverse
138 Pose2 inverse() const;
139
140 /// compose syntactic sugar
141 inline Pose2 operator*(const Pose2& p2) const {
142 return Pose2(r_*p2.r(), t_ + r_*p2.t());
143 }
144
145 /// @}
146 /// @name Lie Group
147 /// @{
148
149 ///Exponential map at identity - create a rotation from canonical coordinates \f$ [T_x,T_y,\theta] \f$
150 static Pose2 Expmap(const Vector3& xi, ChartJacobian H = {});
151
152 ///Log map at identity - return the canonical coordinates \f$ [T_x,T_y,\theta] \f$ of this rotation
153 static Vector3 Logmap(const Pose2& p, ChartJacobian H = {});
154
155 /**
156 * Calculate Adjoint map
157 * Ad_pose is 3*3 matrix that when applied to twist xi \f$ [T_x,T_y,\theta] \f$, returns Ad_pose(xi)
158 */
159 Matrix3 AdjointMap() const;
160
161 /// Apply AdjointMap to twist xi
162 inline Vector3 Adjoint(const Vector3& xi) const {
163 return AdjointMap()*xi;
164 }
165
166 /**
167 * Compute the [ad(w,v)] operator for SE2 as in [Kobilarov09siggraph], pg 19
168 */
169 static Matrix3 adjointMap(const Vector3& v);
170
171 /**
172 * Action of the adjointMap on a Lie-algebra vector y, with optional derivatives
173 */
174 static Vector3 adjoint(const Vector3& xi, const Vector3& y) {
175 return adjointMap(v: xi) * y;
176 }
177
178 /**
179 * The dual version of adjoint action, acting on the dual space of the Lie-algebra vector space.
180 */
181 static Vector3 adjointTranspose(const Vector3& xi, const Vector3& y) {
182 return adjointMap(v: xi).transpose() * y;
183 }
184
185 // temporary fix for wrappers until case issue is resolved
186 static Matrix3 adjointMap_(const Vector3 &xi) { return adjointMap(v: xi);}
187 static Vector3 adjoint_(const Vector3 &xi, const Vector3 &y) { return adjoint(xi, y);}
188
189 /// Derivative of Expmap
190 static Matrix3 ExpmapDerivative(const Vector3& v);
191
192 /// Derivative of Logmap
193 static Matrix3 LogmapDerivative(const Pose2& v);
194
195 // Chart at origin, depends on compile-time flag SLOW_BUT_CORRECT_EXPMAP
196 struct GTSAM_EXPORT ChartAtOrigin {
197 static Pose2 Retract(const Vector3& v, ChartJacobian H = {});
198 static Vector3 Local(const Pose2& r, ChartJacobian H = {});
199 };

201 using LieGroup<Pose2, 3>::inverse; // version with derivative
202
203 /// Hat maps from tangent vector to Lie algebra
204 static Matrix3 Hat(const Vector3& xi);
205
206 /// Vee maps from Lie algebra to tangent vector
207 static Vector3 Vee(const Matrix3& X);
208
209 /// @}
210 /// @name Group Action on Point2
211 /// @{
212
213 /** Return point coordinates in pose coordinate frame */
214 Point2 transformTo(const Point2& point,
215 OptionalJacobian<2, 3> Dpose = {},
216 OptionalJacobian<2, 2> Dpoint = {}) const;
217
218 /**
219 * @brief transform many points in world coordinates and transform to Pose.
220 * @param points 2*N matrix in world coordinates
221 * @return points in Pose coordinates, as 2*N Matrix
222 */
223 Matrix transformTo(const Matrix& points) const;
224
225 /** Return point coordinates in global frame */
226 Point2 transformFrom(const Point2& point,
227 OptionalJacobian<2, 3> Dpose = {},
228 OptionalJacobian<2, 2> Dpoint = {}) const;
229
230 /**
231 * @brief transform many points in Pose coordinates and transform to world.
232 * @param points 2*N matrix in Pose coordinates
233 * @return points in world coordinates, as 2*N Matrix
234 */
235 Matrix transformFrom(const Matrix& points) const;
236
237 /** syntactic sugar for transformFrom */
238 inline Point2 operator*(const Point2& point) const {
239 return transformFrom(point);
240 }
241
242 /// @}
243 /// @name Standard Interface
244 /// @{
245
246 /// get x
247 inline double x() const { return t_.x(); }
248
249 /// get y
250 inline double y() const { return t_.y(); }
251
252 /// get theta
253 inline double theta() const { return r_.theta(); }
254
255 /// translation
256 inline const Point2& t() const { return t_; }
257
258 /// rotation
259 inline const Rot2& r() const { return r_; }
260
261 /// translation
262 inline const Point2& translation(OptionalJacobian<2, 3> Hself={}) const {
263 if (Hself) {
264 *Hself = Matrix::Zero(rows: 2, cols: 3);
265 (*Hself).block<2, 2>(startRow: 0, startCol: 0) = rotation().matrix();
266 }
267 return t_;
268 }
269
270 /// rotation
271 inline const Rot2& rotation(OptionalJacobian<1, 3> Hself={}) const {
272 if (Hself) *Hself << 0, 0, 1;
273 return r_;
274 }
275
276 /// return transformation matrix
277 Matrix3 matrix() const;
278
279 /// Vectorize the rotation matrix into a 9D vector.
280 Vector9 vec(OptionalJacobian<9, 3> H = {}) const;
281

283 * Calculate bearing to a landmark
284 * @param point 2D location of landmark

287 Rot2 bearing(const Point2& point,

294 */
295 Rot2 bearing(const Pose2& pose,
296 OptionalJacobian<1, 3> H1={}, OptionalJacobian<1, 3> H2={}) const;
297

299 * Calculate range to a landmark
300 * @param point 2D location of landmark
301 * @return range (double)
302 */
303 double range(const Point2& point,
304 OptionalJacobian<1, 3> H1={},
305 OptionalJacobian<1, 2> H2={}) const;
306
307 /**
308 * Calculate range to another pose
309 * @param point 2D location of other pose
310 * @return range (double)
311 */
312 double range(const Pose2& point,
313 OptionalJacobian<1, 3> H1={},
314 OptionalJacobian<1, 3> H2={}) const;
315
316 /// @}
317 /// @name Advanced Interface
318 /// @{
319
320 /**
321 * Return the start and end indices (inclusive) of the translation component of the
322 * exponential map parameterization
323 * @return a pair of [start, end] indices into the tangent space vector
324 */
325 inline static std::pair<size_t, size_t> translationInterval() { return {0, 1}; }
326
327 /**
328 * Return the start and end indices (inclusive) of the rotation component of the
329 * exponential map parameterization
330 * @return a pair of [start, end] indices into the tangent space vector
331 */
332 static std::pair<size_t, size_t> rotationInterval() { return {2, 2}; }
