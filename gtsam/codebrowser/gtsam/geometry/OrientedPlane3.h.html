27namespace gtsam {
28
29/**
30 * @brief Represents an infinite plane in 3D, which is composed of a planar
31 * normal and its perpendicular distance to the origin.
32 * Currently it provides a transform of the plane, and a norm 1 differencing of
33 * two planes.
34 * Refer to Trevor12iros for more math details.
35 */
36class GTSAM_EXPORT OrientedPlane3 {
37private:
38 Unit3 n_; ///< The direction of the planar normal
39 double d_; ///< The perpendicular distance to this plane
40
41public:
42 inline constexpr static auto dimension = 3;
43
44 /// @name Constructors
45 /// @{
46
47 /// Default constructor
48 OrientedPlane3() :
49 n_(), d_(0.0) {
50 }
51
52 /// Construct from a Unit3 and a distance
53 OrientedPlane3(const Unit3& n, double d) :
54 n_(n), d_(d) {
55 }
56
57 /// Construct from a vector of plane coefficients
58 explicit OrientedPlane3(const Vector4& vec)
59 : n_(vec(0), vec(1), vec(2)), d_(vec(3)) {}
60
61 /// Construct from four numbers of plane coeffcients (a, b, c, d)
62 OrientedPlane3(double a, double b, double c, double d) {
63 n_ = Unit3(a, b, c);
64 d_ = d;
65 }
66
67 /// @}
68 /// @name Testable
69 /// @{
70
71 /// The print function
72 void print(const std::string& s = std::string()) const;
73
74 /// The equals function with tolerance
75 bool equals(const OrientedPlane3& s, double tol = 1e-9) const {
76 return (n_.equals(s: s.n_, tol) && (std::abs(x: d_ - s.d_) < tol));
77 }
78
79 /// @}
80
81 /** Transforms a plane to the specified pose
82 * @param xr a transformation in current coordiante
83 * @param Hp optional Jacobian wrpt the destination plane
84 * @param Hr optional jacobian wrpt the pose transformation
85 * @return the transformed plane
86 */
87 OrientedPlane3 transform(const Pose3& xr,
88 OptionalJacobian<3, 3> Hp = {},
89 OptionalJacobian<3, 6> Hr = {}) const;
90
91 /** Computes the error between the two planes, with derivatives.
92 * This uses Unit3::errorVector, as opposed to the other .error() in this
93 * class, which uses Unit3::localCoordinates. This one has correct
94 * derivatives.
95 * NOTE(hayk): The derivatives are zero when normals are exactly orthogonal.
96 * @param other the other plane
97 */
98 Vector3 errorVector(const OrientedPlane3& other,
99 OptionalJacobian<3, 3> H1 = {},
100 OptionalJacobian<3, 3> H2 = {}) const;
101
102 /// Dimensionality of tangent space = 3 DOF
103 inline static size_t Dim() {
104 return 3;
105 }
106
107 /// Dimensionality of tangent space = 3 DOF
108 inline size_t dim() const {
109 return 3;
110 }
111
112 /// The retract function
113 OrientedPlane3 retract(const Vector3& v,
114 OptionalJacobian<3, 3> H = {}) const;
115
116 /// The local coordinates function
117 Vector3 localCoordinates(const OrientedPlane3& s) const;
118
119 /// Returns the plane coefficients
120 inline Vector4 planeCoefficients() const {
121 Vector3 unit_vec = n_.unitVector();
122 return Vector4(unit_vec[0], unit_vec[1], unit_vec[2], d_);
123 }
124
125 /// Return the normal
126 inline Unit3 normal(OptionalJacobian<2, 3> H = {}) const {
127 if (H) *H << I_2x2, Z_2x1;
128 return n_;
129 }
130
131 /// Return the perpendicular distance to the origin
132 inline double distance(OptionalJacobian<1, 3> H = {}) const {
133 if (H) *H << 0,0,1;
134 return d_;
135 }
136};

140};

143OrientedPlane3> {
144};
