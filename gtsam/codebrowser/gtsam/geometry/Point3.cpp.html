27double distance3(const Point3 &p1, const Point3 &q, OptionalJacobian<1, 3> H1,
28 OptionalJacobian<1, 3> H2) {
29 double d = (q - p1).norm();
30 if (H1) {
31 *H1 << p1.x() - q.x(), p1.y() - q.y(), p1.z() - q.z();
32 *H1 = *H1 *(1. / d);
33 }
34 if (H2) {
35 *H2 << -p1.x() + q.x(), -p1.y() + q.y(), -p1.z() + q.z();
36 *H2 = *H2 *(1. / d);
37 }
38 return d;
39}

41double norm3(const Point3 &p, OptionalJacobian<1, 3> H) {
42 double r = sqrt(x: p.x() * p.x() + p.y() * p.y() + p.z() * p.z());
43 if (H) {
44 if (std::abs(x: r) > 1e-10)
45 *H << p.x() / r, p.y() / r, p.z() / r;
46 else
47 *H << 1, 1, 1; // really infinity, why 1 ?
48 }
49 return r;
50}

52Point3 normalize(const Point3 &p, OptionalJacobian<3, 3> H) {
53 Point3 normalized = p / p.norm();
54 if (H) {
55 // 3*3 Derivative
56 double x2 = p.x() * p.x(), y2 = p.y() * p.y(), z2 = p.z() * p.z();
57 double xy = p.x() * p.y(), xz = p.x() * p.z(), yz = p.y() * p.z();
58 *H << y2 + z2, -xy, -xz, /**/ -xy, x2 + z2, -yz, /**/ -xz, -yz, x2 + y2;
59 *H /= pow(x: x2 + y2 + z2, y: 1.5);
60 }
61 return normalized;
62}

64Point3 cross(const Point3 &p, const Point3 &q, OptionalJacobian<3, 3> H1,
65 OptionalJacobian<3, 3> H2) {
66 if (H1) *H1 << skewSymmetric(wx: -q.x(), wy: -q.y(), wz: -q.z());
67 if (H2) *H2 << skewSymmetric(wx: p.x(), wy: p.y(), wz: p.z());
68 return Point3(p.y() * q.z() - p.z() * q.y(), p.z() * q.x() - p.x() * q.z(),
69 p.x() * q.y() - p.y() * q.x());
70}

72Point3 doubleCross(const Point3 &p, const Point3 &q, //
73 OptionalJacobian<3, 3> H1, OptionalJacobian<3, 3> H2) {
74 if (H1) *H1 = q.dot(other: p) * I_3x3 + p * q.transpose() - 2 * q * p.transpose();
75 if (H2) {
76 const Matrix3 W = skewSymmetric(w: p);
77 *H2 = W * W;
78 }
79 return gtsam::cross(p, q: gtsam::cross(p, q));
80}

82double dot(const Point3 &p, const Point3 &q, OptionalJacobian<1, 3> H1,
83 OptionalJacobian<1, 3> H2) {
84 if (H1) *H1 << q.x(), q.y(), q.z();
85 if (H2) *H2 << p.x(), p.y(), p.z();
86 return p.x() * q.x() + p.y() * q.y() + p.z() * q.z();
87}

89Point3Pair means(const std::vector<Point3Pair> &abPointPairs) {
90 const size_t n = abPointPairs.size();
91 if (n == 0) throw std::invalid_argument("Point3::mean input Point3Pair vector is empty");
92 Point3 aSum(0, 0, 0), bSum(0, 0, 0);
93 for (const Point3Pair &abPair : abPointPairs) {
94 aSum += abPair.first;
95 bSum += abPair.second;
96 }
97 const double f = 1.0 / n;
98 return {aSum * f, bSum * f};
99}
