53template <>
54GTSAM_EXPORT
55Matrix4 SO4::Hat(const Vector6& xi) {
56 // skew symmetric matrix X = xi^
57 // Unlike Luca, makes upper-left the SO(3) subgroup.
58 Matrix4 Y = Z_4x4;
59 Y(0, 1) = -xi(5);
60 Y(0, 2) = +xi(4);
61 Y(1, 2) = -xi(3);
62 Y(0, 3) = +xi(2);
63 Y(1, 3) = -xi(1);
64 Y(2, 3) = +xi(0);
65 return Y - Y.transpose();
66}

69template <>
70GTSAM_EXPORT
71Vector6 SO4::Vee(const Matrix4& X) {
72 Vector6 xi;
73 xi(5) = -X(0, 1);
74 xi(4) = +X(0, 2);
75 xi(3) = -X(1, 2);
76 xi(2) = +X(0, 3);
77 xi(1) = -X(1, 3);
78 xi(0) = +X(2, 3);
79 return xi;
80}

83/* Exponential map, porting MATLAB implementation by Luca, which follows
84 * "SOME REMARKS ON THE EXPONENTIAL MAP ON THE GROUPS SO(n) AND SE(n)" by
85 * Ramona-Andreaa Rohan */
86template <>
87GTSAM_EXPORT
88SO4 SO4::Expmap(const Vector6& xi, ChartJacobian H) {
89 using namespace std;
90 if (H) throw std::runtime_error("SO4::Expmap Jacobian");
91
92 // skew symmetric matrix X = xi^
93 const Matrix4 X = Hat(xi);
94
95 // do eigen-decomposition
96 auto eig = Eigen::EigenSolver<Matrix4>(X);
97 Eigen::Vector4cd e = eig.eigenvalues();
98 using std::abs;
99 sort(first: e.data(), last: e.data() + 4, comp: [](complex<double> a, complex<double> b) {
100 return abs(x: a.imag()) > abs(x: b.imag());
101 });
102
103 // Get a and b from eigenvalues +/i ai and +/- bi
104 double a = e[0].imag(), b = e[2].imag();
105 if (!e.real().isZero() || e[1].imag() != -a || e[3].imag() != -b) {
106 throw runtime_error("SO4::Expmap: wrong eigenvalues.");
107 }
108
109 // Build expX = exp(xi^)
110 using std::cos;
111 using std::sin;
112 const auto X2 = X * X;
113 const auto X3 = X2 * X;
114 double a2 = a * a, a3 = a2 * a, b2 = b * b, b3 = b2 * b;
115 if (a != 0 && b == 0) {
116 double c2 = (1 - cos(x: a)) / a2, c3 = (a - sin(x: a)) / a3;
117 return SO4(I_4x4 + X + c2 * X2 + c3 * X3);
118 } else if (a == b && b != 0) {
119 double sin_a = sin(x: a), cos_a = cos(x: a);
120 double c0 = (a * sin_a + 2 * cos_a) / 2,
121 c1 = (3 * sin_a - a * cos_a) / (2 * a), c2 = sin_a / (2 * a),
122 c3 = (sin_a - a * cos_a) / (2 * a3);
123 return SO4(c0 * I_4x4 + c1 * X + c2 * X2 + c3 * X3);
124 } else if (a != b) {
125 double sin_a = sin(x: a), cos_a = cos(x: a);
126 double sin_b = sin(x: b), cos_b = cos(x: b);
127 double c0 = (b2 * cos_a - a2 * cos_b) / (b2 - a2),
128 c1 = (b3 * sin_a - a3 * sin_b) / (a * b * (b2 - a2)),
129 c2 = (cos_a - cos_b) / (b2 - a2),
130 c3 = (b * sin_a - a * sin_b) / (a * b * (b2 - a2));
131 return SO4(c0 * I_4x4 + c1 * X + c2 * X2 + c3 * X3);
132 } else {
133 return SO4();
134 }
135}

138template <>
139GTSAM_EXPORT
140SO4::VectorN2 SO4::vec(OptionalJacobian<16, 6> H) const {
141 const Matrix& Q = matrix_;
142 if (H) {
143 H->setZero();
144 H->block<4, 1>(startRow: 0, startCol: 2) = -Q.col(i: 3);
145 H->block<4, 1>(startRow: 0, startCol: 4) = -Q.col(i: 2);
146 H->block<4, 1>(startRow: 0, startCol: 5) = Q.col(i: 1);
147 H->block<4, 1>(startRow: 4, startCol: 1) = Q.col(i: 3);
148 H->block<4, 1>(startRow: 4, startCol: 3) = Q.col(i: 2);
149 H->block<4, 1>(startRow: 4, startCol: 5) = -Q.col(i: 0);
150 H->block<4, 1>(startRow: 8, startCol: 0) = -Q.col(i: 3);
151 H->block<4, 1>(startRow: 8, startCol: 3) = -Q.col(i: 1);
152 H->block<4, 1>(startRow: 8, startCol: 4) = Q.col(i: 0);
153 H->block<4, 1>(startRow: 12, startCol: 0) = Q.col(i: 2);
154 H->block<4, 1>(startRow: 12, startCol: 1) = -Q.col(i: 1);
155 H->block<4, 1>(startRow: 12, startCol: 2) = Q.col(i: 0);
156 }
157 return Eigen::Map<const SO4::VectorN2>(Q.data());
158}

161template <>
162GTSAM_EXPORT
163SO4 SO4::ChartAtOrigin::Retract(const Vector6& xi, ChartJacobian H) {
164 if (H) throw std::runtime_error("SO4::ChartAtOrigin::Retract Jacobian");
165 gttic(SO4_Retract);
166 const Matrix4 X = Hat(xi: xi / 2);
167 return SO4((I_4x4 + X) * (I_4x4 - X).inverse());
168}

170//******************************************************************************
171template <>
172GTSAM_EXPORT
173Vector6 SO4::ChartAtOrigin::Local(const SO4& Q, ChartJacobian H) {
174 if (H) throw std::runtime_error("SO4::ChartAtOrigin::Retract Jacobian");
175 const Matrix4& R = Q.matrix();
176 const Matrix4 X = (I_4x4 - R) * (I_4x4 + R).inverse();
177 return -2 * Vee(X);
178}

181GTSAM_EXPORT Matrix3 topLeft(const SO4& Q, OptionalJacobian<9, 6> H) {
182 const Matrix4& R = Q.matrix();
183 const Matrix3 M = R.topLeftCorner<3, 3>();
184 if (H) {
185 const Vector3 m1 = M.col(i: 0), m2 = M.col(i: 1), m3 = M.col(i: 2),
186 q = R.topRightCorner<3, 1>();
187 *H << Z_3x1, Z_3x1, -q, Z_3x1, -m3, m2, //
188 Z_3x1, q, Z_3x1, m3, Z_3x1, -m1, //
189 -q, Z_3x1, Z_3x1, -m2, m1, Z_3x1;
190 }
191 return M;
192}

195GTSAM_EXPORT Matrix43 stiefel(const SO4& Q, OptionalJacobian<12, 6> H) {
196 const Matrix4& R = Q.matrix();
197 const Matrix43 M = R.leftCols<3>();
198 if (H) {
199 const auto &m1 = R.col(i: 0), m2 = R.col(i: 1), m3 = R.col(i: 2), q = R.col(i: 3);
200 *H << Z_4x1, Z_4x1, -q, Z_4x1, -m3, m2, //
201 Z_4x1, q, Z_4x1, m3, Z_4x1, -m1, //
202 -q, Z_4x1, Z_4x1, -m2, m1, Z_4x1;
203 }
204 return M;
205}
