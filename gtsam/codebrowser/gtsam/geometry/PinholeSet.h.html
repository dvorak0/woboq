17
18#pragma once
19
20#include <gtsam/geometry/CameraSet.h>
21#include <gtsam/geometry/triangulation.h>
22
23namespace gtsam {
24
25/**
26 * PinholeSet: triangulates point and keeps an estimate of it around.
27 */
28template<class CAMERA>
29class PinholeSet: public CameraSet<CAMERA> {
30
31private:
32 typedef CameraSet<CAMERA> Base;
33 typedef PinholeSet<CAMERA> This;
34
35protected:
36
37public:
38
39 /** Virtual destructor */
40 virtual ~PinholeSet() {
41 }
42
43 /// @name Testable
44 /// @{
45
46 /// print
47 void print(const std::string& s = "") const override {
48 Base::print(s);
49 }
50
51 /// equals
52 bool equals(const PinholeSet& p, double tol = 1e-9) const {
53 return Base::equals(p, tol); // TODO all flags
54 }
55
56 /// @}
57
58 /// triangulateSafe
59 TriangulationResult triangulateSafe(
60 const typename CAMERA::MeasurementVector& measured,
61 const TriangulationParameters& params) const {
62 return gtsam::triangulateSafe(*this, measured, params);
63 }
64
65private:
66
67#if GTSAM_ENABLE_BOOST_SERIALIZATION ///
68 /// Serialization function
69 friend class boost::serialization::access;
70 template<class ARCHIVE>
71 void serialize(ARCHIVE & ar, const unsigned int version) {
72 ar & BOOST_SERIALIZATION_BASE_OBJECT_NVP(Base);
73 }
74#endif
75};
76
77template<class CAMERA>
78struct traits<PinholeSet<CAMERA> > : public Testable<PinholeSet<CAMERA> > {
79};
80
81template<class CAMERA>
82struct traits<const PinholeSet<CAMERA> > : public Testable<PinholeSet<CAMERA> > {
83};
84
85} // \ namespace gtsam
