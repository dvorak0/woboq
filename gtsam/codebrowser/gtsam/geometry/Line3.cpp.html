5Line3 Line3::retract(const Vector4 &v, OptionalJacobian<4, 4> Dp, OptionalJacobian<4, 4> Dv) const {
6 Vector3 w;
7 w << v[0], v[1], 0;
8 Rot3 incR;
9
10 if (Dp) {
11 Dp->setIdentity();
12 incR = Rot3::Expmap(v: w);
13 Dp->block<2, 2>(startRow: 0, startCol: 0) = ((incR.matrix()).transpose()).block<2, 2>(startRow: 0, startCol: 0);
14 }
15 if (Dv) {
16 Matrix3 Dw;
17 incR = Rot3::Expmap(v: w, H: Dw);
18 Dv->setIdentity();
19 Dv->block<2, 2>(startRow: 0, startCol: 0) = Dw.block<2, 2>(startRow: 0, startCol: 0);
20 } else {
21 incR = Rot3::Expmap(v: w);
22 }
23 Rot3 Rt = R_ * incR;
24 return Line3(Rt, a_ + v[2], b_ + v[3]);
25}

27Vector4 Line3::localCoordinates(const Line3 &q, OptionalJacobian<4, 4> Dp,
28 OptionalJacobian<4, 4> Dq) const {
29 Vector3 omega;
30 Matrix3 D_log;
31 omega = Rot3::Logmap(R: R_.inverse() * q.R_, H: D_log);
32 if (Dp) {
33 Matrix3 D_log_wp = -((q.R_).matrix()).transpose() * R_.matrix();
34 Matrix3 Dwp = D_log * D_log_wp;
35 Dp->setIdentity();
36 Dp->block<2, 2>(startRow: 0, startCol: 0) = Dwp.block<2, 2>(startRow: 0, startCol: 0);
37 (*Dp)(2, 2) = -1;
38 (*Dp)(3, 3) = -1;
39 }
40 if (Dq) {
41 Dq->setIdentity();
42 Dq->block<2, 2>(startRow: 0, startCol: 0) = D_log.block<2, 2>(startRow: 0, startCol: 0);
43 }
44 Vector4 local;
45 local << omega[0], omega[1], q.a_ - a_, q.b_ - b_;
46 return local;
47}

61Unit3 Line3::project(OptionalJacobian<2, 4> Dline) const {
62 Vector3 V_0;
63 V_0 << -b_, a_, 0.0;
64
65 Unit3 l;
66 if (Dline) {
67 // Jacobian of the normalized Unit3 projected line with respect to
68 // un-normalized Vector3 projected line in homogeneous coordinates.
69 Matrix23 D_unit_line;
70 l = Unit3::FromPoint3(point: Point3(R_ * V_0), H: D_unit_line);
71 // Jacobian of the un-normalized Vector3 line with respect to
72 // input 3D line
73 Matrix34 D_vec_line = Matrix34::Zero();
74 D_vec_line.col(i: 0) = a_ * R_.r3();
75 D_vec_line.col(i: 1) = b_ * R_.r3();
76 D_vec_line.col(i: 2) = R_.r2();
77 D_vec_line.col(i: 3) = -R_.r1();
78 // Jacobian of output wrt input is the product of the two.
79 *Dline = D_unit_line * D_vec_line;
80 } else {
81 l = Unit3::FromPoint3(point: Point3(R_ * V_0));
82 }
83 return l;
84}

86Point3 Line3::point(double distance) const {
87 // defining "center" of the line to be the point where it
88 // intersects rotated XY axis
89 Point3 center(a_, b_, 0);
90 Point3 rotated_center = R_ * center;
91 return rotated_center + distance * R_.r3();
92}

94Line3 transformTo(const Pose3 &wTc, const Line3 &wL,
95 OptionalJacobian<4, 6> Dpose, OptionalJacobian<4, 4> Dline) {
96 Rot3 wRc = wTc.rotation();
97 Rot3 cRw = wRc.inverse();
98 Rot3 cRl = cRw * wL.R_;
99
100 Vector3 t = ((wL.R_).transpose() * wTc.translation());
101 Vector2 c_ab(wL.a_ - t[0], wL.b_ - t[1]);
102
103 if (Dpose) {
104 Matrix3 lRc = (cRl.matrix()).transpose();
105 Dpose->setZero();
106 // rotation
107 Dpose->block<2, 3>(startRow: 0, startCol: 0) = -lRc.block<2, 3>(startRow: 0, startCol: 0);
108 // translation
109 Dpose->block<2, 3>(startRow: 2, startCol: 3) = -lRc.block<2, 3>(startRow: 0, startCol: 0);
110 }
111 if (Dline) {
112 Dline->setIdentity();
113 (*Dline)(3, 0) = -t[2];
114 (*Dline)(2, 1) = t[2];
115 }
116 return Line3(cRl, c_ab[0], c_ab[1]);
117}
