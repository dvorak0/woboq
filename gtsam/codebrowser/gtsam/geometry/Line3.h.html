26/**
27 * Transform a line from world to camera frame
28 * @param wTc - Pose3 of camera in world frame
29 * @param wL - Line3 in world frame
30 * @param Dpose - OptionalJacobian of transformed line with respect to p
31 * @param Dline - OptionalJacobian of transformed line with respect to l
32 * @return Transformed line in camera frame
33 */
34GTSAM_EXPORT Line3 transformTo(const Pose3 &wTc, const Line3 &wL,
35 OptionalJacobian<4, 6> Dpose = {},
36 OptionalJacobian<4, 4> Dline = {});
37
38
39/**
40 * A 3D line (R,a,b) : (Rot3,Scalar,Scalar)
41 * @ingroup geometry
42 * \nosubgrouping
43 */
44class GTSAM_EXPORT Line3 {
45 private:
46 Rot3 R_; // Rotation of line about x and y in world frame
47 double a_, b_; // Intersection of line with the world x-y plane rotated by R_
48 // Also the closest point on line to origin
49 public:
50 inline constexpr static auto dimension = 4;
51
52 /** Default constructor is the Z axis **/
53 Line3() :
54 a_(0), b_(0) {}
55
56 /** Constructor for general line from (R, a, b) **/
57 Line3(const Rot3 &R, const double a, const double b) :
58 R_(R), a_(a), b_(b) {}
59
60 /**
61 * The retract method maps from the tangent space back to the manifold.
62 * The method q = p + v, where p is this line, v is an increment along
63 * the tangent space and q is the resulting line.
64 * The tangent space for the rotation of a line is only two dimensional -
65 * rotation about x and y
66 * @param v: increment in tangent space
67 * @param Dp: increment of retraction with respect to this line
68 * @param Dv: Jacobian of retraction with respect to the increment
69 * @return q: resulting line after adding the increment and mapping to the manifold
70 */
71 Line3 retract(const Vector4 &v,
72 OptionalJacobian<4, 4> Dp = {},
73 OptionalJacobian<4, 4> Dv = {}) const;
74
75 /**
76 * The localCoordinates method is the inverse of retract and finds the difference
77 * between two lines in the tangent space. It computes v = q - p where q is an input
78 * line, p is this line and v is their difference in the tangent space.
79 * @param q: Line3 on manifold
80 * @param Dp: OptionalJacobian of localCoordinates with respect to this line
81 * @param Dq: OptionalJacobian of localCoordinates with respect to this line
82 * @return v: difference in the tangent space
83 */
84 Vector4 localCoordinates(const Line3 &q,
85 OptionalJacobian<4, 4> Dp = {},
86 OptionalJacobian<4, 4> Dq = {}) const;
87
88 /**
89 * Print R, a, b
90 * @param s: optional starting string
91 */
92 void print(const std::string &s = "") const;
93
94 /**
95 * Check if two lines are equal
96 * @param l2 - line to be compared
97 * @param tol : optional tolerance
98 * @return boolean - true if lines are equal
99 */
100 bool equals(const Line3 &l2, double tol = 10e-9) const;
101
102 /**
103 * Projecting a line to the image plane. Assumes this line is in camera frame.
104 * @param Dline: OptionalJacobian of projected line with respect to this line
105 * @return Unit3 - projected line in image plane, in homogenous coordinates.
106 * We use Unit3 since it is a manifold with the right dimension.
107 */
108 Unit3 project(OptionalJacobian<2, 4> Dline = {}) const;
109
110 /**
111 * Returns point on the line that is at a certain distance starting from the
112 * point where the rotated XY axis intersects the line.
113 * @param distance (can be positive or negative) - positive is the positive
114 * direction of the rotated z axis that forms the line. The default value of zero
115 * returns the point where the rotated XY axis intersects the line.
116 * @return Point on the line
117 */
118 Point3 point(double distance = 0) const;
119
120 /**
121 * Return the rotation of the line.
122 */
123 inline Rot3 R() const {
124 return R_;
125 }
126
127 /**
128 * Return the x-coordinate of the intersection of the line with the xy plane.
129 */
130 inline double a() const {
131 return a_;
132 }
133
134 /**
135 * Return the y-coordinate of the intersection of the line with the xy plane.
136 */
137 inline double b() const {
138 return b_;
139 }
140
141 /**
142 * Transform a line from world to camera frame
143 * @param wTc - Pose3 of camera in world frame
144 * @param wL - Line3 in world frame
145 * @param Dpose - OptionalJacobian of transformed line with respect to p
146 * @param Dline - OptionalJacobian of transformed line with respect to l
147 * @return Transformed line in camera frame
148 */
149 GTSAM_EXPORT friend Line3 transformTo(const Pose3 &wTc, const Line3 &wL,
150 OptionalJacobian<4, 6> Dpose,
151 OptionalJacobian<4, 4> Dline);
152};
153

159}
