27// Lightweight 3x3 inverse (adjugate / determinant), avoids Eigen's LU
28// machinery. Assumes M is invertible. Used to keep link-time/code-size small on
29// GCC/Ubuntu.
30static inline Matrix3 inverse3x3(const Matrix3& M) {
31 const double a = M(0, 0), b = M(0, 1), c = M(0, 2);
32 const double d = M(1, 0), e = M(1, 1), f = M(1, 2);
33 const double g = M(2, 0), h = M(2, 1), i = M(2, 2);
34
35 const double A = (e * i - f * h);
36 const double B = -(d * i - f * g);
37 const double C = (d * h - e * g);
38 const double D = -(b * i - c * h);
39 const double E = (a * i - c * g);
40 const double F = -(a * h - b * g);
41 const double G = (b * f - c * e);
42 const double H = -(a * f - c * d);
43 const double I = (a * e - b * d);
44
45 const double det = a * A + b * B + c * C;
46 const double inv_det = 1.0 / det;
47
48 Matrix3 adjT; // transpose of cofactor matrix = adjugate^T
49 adjT << A, D, G, B, E, H, C, F, I;
50 return inv_det * adjT;
51}

53// --- Kernel matrices (out-of-line to keep header tight) ---
54Matrix3 Kernel::left() const {
55 // left: a I + b Ω + c Ω²
56 return a * I_3x3 + b * S->W + c * S->WW;
57}
58Matrix3 Kernel::right() const {
59 // right: a I - b Ω + c Ω²
60 return a * I_3x3 - b * S->W + c * S->WW;
61}

63Vector3 Kernel::applyLeft(const Vector3& v, OptionalJacobian<3, 3> Hw,
64 OptionalJacobian<3, 3> Hv) const {
65 const Vector3 Wv = S->omega.cross(other: v);
66 const Vector3 WWv = S->omega.cross(other: Wv);
67 if (Hw) {
68 // Closed-form ∂/∂ω without doubleCross Jacobian:
69 const auto& w = S->omega;
70 const auto wt = w.transpose();
71 const auto vt = v.transpose();
72 const auto sI = w.dot(other: v) * I_3x3;
73 *Hw = -b * skewSymmetric(w: v) + // d(Ω v)/dω = -[v]×
74 c * (w * vt + sI - 2 * v * wt) + // d(Ω² v)/dω = ωvᵀ + (ω·v)I - 2vωᵀ
75 (db * Wv + dc * WWv) * wt; // radial derivative terms
76 }
77 if (Hv) *Hv = left(); // ∂y/∂v = a I + b Ω + c Ω²
78 return a * v + b * Wv + c * WWv;
79}

81Vector3 Kernel::applyRight(const Vector3& v, OptionalJacobian<3, 3> Hw,
82 OptionalJacobian<3, 3> Hv) const {
83 // Implement by flipping b and db and reusing left machinery
84 Kernel tmp{.S: S, .a: a, .b: -b, .c: c, .db: -db, .dc: dc};
85 return tmp.applyLeft(v, Hw, Hv);
86}

88Matrix3 Kernel::frechet(const Matrix3& X) const {
89 const Matrix3& W = S->W;
90 const Matrix3& WW = S->WW;
91 const double s = -0.5 * (W * X).trace();
92 return b * X + c * (W * X + X * W) + s * (db * W + dc * WW);
93}
94
95Matrix3 Kernel::applyFrechet(const Vector3& v) const {
96 Matrix3 H;
97 H.col(i: 0) = frechet(X: skewSymmetric(w: Vector3::UnitX())) * v; // δω = e_x
98 H.col(i: 1) = frechet(X: skewSymmetric(w: Vector3::UnitY())) * v; // δω = e_y
99 H.col(i: 2) = frechet(X: skewSymmetric(w: Vector3::UnitZ())) * v; // δω = e_z
100 return H;
101}

103// --- InvJKernel matrices (closed form; π-stable via Local::InvJacobian()
104// selection) ---
105Matrix3 InvJKernel::left() const {
106 if (S->theta > M_PI) {
107 // Above π the closed-form coefficients can suffer cancellation; invert
108 // numerically but with a tiny, fixed-cost 3x3 adjugate instead of Eigen's
109 // LU to avoid link-time bloat.
110 return inverse3x3(M: J.left());
111 }
112 // Closed-form inverse in the {I, Ω, Ω²} basis
113 return I_3x3 - 0.5 * S->W + S->D() * S->WW;
114}
115Matrix3 InvJKernel::right() const {
116 if (S->theta > M_PI) {
117 return inverse3x3(M: J.right());
118 }
119 return I_3x3 + 0.5 * S->W + S->D() * S->WW;
120}

122Vector3 InvJKernel::applyLeft(const Vector3& v, OptionalJacobian<3, 3> Hw,
123 OptionalJacobian<3, 3> Hv) const {
124 const Matrix3 Linv = left();
125 const Vector3 c = Linv * v;
126 if (Hw) {
127 Matrix3 Hf;
128 J.applyLeft(v: c, Hw: Hf); // derivative of forward mapping at c
129 *Hw = -Linv * Hf; // chain rule for inverse
130 }
131 if (Hv) *Hv = Linv;
132 return c;
133}
134
135Vector3 InvJKernel::applyRight(const Vector3& v, OptionalJacobian<3, 3> Hw,
136 OptionalJacobian<3, 3> Hv) const {
137 const Matrix3 Rinv = right();
138 const Vector3 c = Rinv * v;
139 if (Hw) {
140 Matrix3 Hf;
141 J.applyRight(v: c, Hw: Hf); // derivative of forward right-mapping at c
142 *Hw = -Rinv * Hf; // chain rule for inverse
143 }
144 if (Hv) *Hv = Rinv;
145 return c;
146}

148Kernel axpy(double alpha, const Kernel& X, const Kernel& Y) {
149 return Kernel{.S: Y.S,
150 .a: Y.a + alpha * X.a,
151 .b: Y.b + alpha * X.b,
152 .c: Y.c + alpha * X.c,
153 .db: Y.db + alpha * X.db,
154 .dc: Y.dc + alpha * X.dc};
155}

157Kernel blend(double alpha, double dalpha, const Kernel& X, const Kernel& Y) {
158 const double beta = 1.0 - alpha;
159 // L has db=dc=0; derivative comes from α only
160 return Kernel{
161 .S: Y.S,
162 .a: alpha * X.a + beta * Y.a,
163 .b: alpha * X.b + beta * Y.b,
164 .c: alpha * X.c + beta * Y.c,
165 .db: dalpha * (X.b - Y.b) + beta * Y.db, // db
166 .dc: dalpha * (X.c - Y.c) + beta * Y.dc // dc
167 };
