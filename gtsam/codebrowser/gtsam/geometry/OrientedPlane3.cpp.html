34/* ************************************************************************* */
35OrientedPlane3 OrientedPlane3::transform(const Pose3& xr,
36 OptionalJacobian<3, 3> Hp,
37 OptionalJacobian<3, 6> Hr) const {
38 Matrix23 D_rotated_plane;
39 Matrix22 D_rotated_pose;
40 Unit3 n_rotated = xr.rotation().unrotate(p: n_, HR: D_rotated_plane, Hp: D_rotated_pose);
41
42 Vector3 unit_vec = n_rotated.unitVector();
43 double pred_d = n_.unitVector().dot(other: xr.translation()) + d_;
44
45 if (Hr) {
46 Hr->setZero();
47 Hr->block<2, 3>(startRow: 0, startCol: 0) = D_rotated_plane;
48 Hr->block<1, 3>(startRow: 2, startCol: 3) = unit_vec;
49 }
50 if (Hp) {
51 Hp->setZero();
52 Hp->block<2, 2>(startRow: 0, startCol: 0) = D_rotated_pose;
53 Hp->block<1, 2>(startRow: 2, startCol: 0) = n_.basis().transpose() * xr.translation();
54 (*Hp)(2, 2) = 1;
55 }
56
57 return OrientedPlane3(unit_vec(0), unit_vec(1), unit_vec(2), pred_d);
58}

60/* ************************************************************************* */
61Vector3 OrientedPlane3::errorVector(const OrientedPlane3& other,
62 OptionalJacobian<3, 3> H1,
63 OptionalJacobian<3, 3> H2) const {
64 Matrix22 H_n_error_this, H_n_error_other;
65 Vector2 n_error = n_.errorVector(q: other.n_, H_p: H1 ? &H_n_error_this : 0,
66 H_q: H2 ? &H_n_error_other : 0);
67
68 double d_error = d_ - other.d_;
69
70 if (H1) {
71 *H1 << H_n_error_this, Z_2x1, 0, 0, 1;
72 }
73 if (H2) {
74 *H2 << H_n_error_other, Z_2x1, 0, 0, -1;
75 }
76
77 return Vector3(n_error(0), n_error(1), d_error);
78}

80/* ************************************************************************* */
81OrientedPlane3 OrientedPlane3::retract(const Vector3& v,
82 OptionalJacobian<3, 3> H) const {
83 Matrix22 H_n;
84 Unit3 n_retract(n_.retract(v: Vector2(v(0), v(1)), H: H? &H_n : nullptr));
85 if (H) {
86 *H << H_n, Z_2x1, 0, 0, 1;
87 }
88 return OrientedPlane3(n_retract, d_ + v(2));
89}

91/* ************************************************************************* */
92Vector3 OrientedPlane3::localCoordinates(const OrientedPlane3& y) const {
93 Vector2 n_local = n_.localCoordinates(s: y.n_);
94 double d_local = d_ - y.d_;
95 return Vector3(n_local(0), n_local(1), -d_local);
96}
