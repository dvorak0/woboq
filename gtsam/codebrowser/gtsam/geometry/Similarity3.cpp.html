28/// Subtract centroids from point pairs.
29static Point3Pairs subtractCentroids(const Point3Pairs &abPointPairs,
30 const Point3Pair &centroids) {
31 Point3Pairs d_abPointPairs;
32 for (const auto& [a, b] : abPointPairs) {
33 Point3 da = a - centroids.first;
34 Point3 db = b - centroids.second;
35 d_abPointPairs.emplace_back(args&: da, args&: db);
36 }
37 return d_abPointPairs;
38}

40/// Form inner products x and y and calculate scale.
41// We force the scale to be a non-negative quantity
42// (see Section 10.1 of https://ethaneade.com/lie_groups.pdf)
43static double calculateScale(const Point3Pairs &d_abPointPairs,
44 const Rot3 &aRb) {
45 double x = 0, y = 0;
46 for (const auto& [da, db] : d_abPointPairs) {
47 const Vector3 da_prime = aRb * db;
48 y += da.transpose() * da_prime;
49 x += da_prime.transpose() * da_prime;
50 }
51 const double s = std::fabs(x: y / x);
52 return s;
53}
54
55/// Form outer product H.
56static Matrix3 calculateH(const Point3Pairs &d_abPointPairs) {
57 Matrix3 H = Z_3x3;
58 for (const auto& [da, db] : d_abPointPairs) {
59 H += da * db.transpose();
60 }
61 return H;
62}

64/// This method estimates the similarity transform from differences point pairs,
65// given a known or estimated rotation and point centroids.
66static Similarity3 align(const Point3Pairs &d_abPointPairs, const Rot3 &aRb,
67 const Point3Pair &centroids) {
68 const double s = calculateScale(d_abPointPairs, aRb);
69 // dividing aTb by s is required because the registration cost function
70 // minimizes ||a - sRb - t||, whereas Sim(3) computes s(Rb + t)
71 const Point3 aTb = (centroids.first - s * (aRb * centroids.second)) / s;
72 return Similarity3(aRb, aTb, s);
73}
74
75/// This method estimates the similarity transform from point pairs, given a known or estimated rotation.
76// Refer to: http://www5.informatik.uni-erlangen.de/Forschung/Publikationen/2005/Zinsser05-PSR.pdf Chapter 3
77static Similarity3 alignGivenR(const Point3Pairs &abPointPairs,
78 const Rot3 &aRb) {
79 auto centroids = means(abPointPairs);
80 auto d_abPointPairs = internal::subtractCentroids(abPointPairs, centroids);
81 return align(d_abPointPairs, aRb, centroids);
82}

105bool Similarity3::equals(const Similarity3& other, double tol) const {
106 return R_.equals(p: other.R_, tol) && traits<Point3>::Equals(v1: t_, v2: other.t_, tol)
107 && s_ < (other.s_ + tol) && s_ > (other.s_ - tol);
108}

110bool Similarity3::operator==(const Similarity3& other) const {
111 return R_.matrix() == other.R_.matrix() && t_ == other.t_ && s_ == other.s_;
112}

124Similarity3 Similarity3::operator*(const Similarity3& S) const {
125 return Similarity3(R_ * S.R_, ((1.0 / S.s_) * t_) + R_ * S.t_, s_ * S.s_);

128Similarity3 Similarity3::inverse() const {
129 const Rot3 Rt = R_.inverse();
130 const Point3 sRt = Rt * (-s_ * t_);
131 return Similarity3(Rt, sRt, 1.0 / s_);

134Point3 Similarity3::transformFrom(const Point3& p, //
135 OptionalJacobian<3, 7> H1, OptionalJacobian<3, 3> H2) const {
136 const Point3 q = R_ * p + t_;
137 if (H1) {
138 // For this derivative, see LieGroups.pdf
139 const Matrix3 sR = s_ * R_.matrix();
140 const Matrix3 DR = sR * skewSymmetric(wx: -p.x(), wy: -p.y(), wz: -p.z());
141 *H1 << DR, sR, sR * p;
142 }
143 if (H2)
144 *H2 = s_ * R_.matrix(); // just 3*3 sub-block of matrix()
145 return s_ * q;
146}

158Similarity3 Similarity3::Align(const Point3Pairs &abPointPairs) {
159 // Refer to Chapter 3 of
160 // http://www5.informatik.uni-erlangen.de/Forschung/Publikationen/2005/Zinsser05-PSR.pdf
161 if (abPointPairs.size() < 3)
162 throw std::runtime_error("input should have at least 3 pairs of points");
163 auto centroids = means(abPointPairs);
164 auto d_abPointPairs = internal::subtractCentroids(abPointPairs, centroids);
165 Matrix3 H = internal::calculateH(d_abPointPairs);
166 // ClosestTo finds rotation matrix closest to H in Frobenius sense
167 Rot3 aRb = Rot3::ClosestTo(M: H);
168 return internal::align(d_abPointPairs, aRb, centroids);
169}

171Similarity3 Similarity3::Align(const Pose3Pairs &abPosePairs) {
172 const size_t n = abPosePairs.size();
173 if (n < 2)
174 throw std::runtime_error("input should have at least 2 pairs of poses");
175
176 // calculate rotation
177 std::vector<Rot3> rotations;
178 Point3Pairs abPointPairs;
179 rotations.reserve(n: n);
180 abPointPairs.reserve(n: n);
181 // Below denotes the pose of the i'th object/camera/etc in frame "a" or frame "b"
182 Pose3 aTi, bTi;
183 for (const auto &[aTi, bTi] : abPosePairs) {
184 const Rot3 aRb = aTi.rotation().compose(g: bTi.rotation().inverse());
185 rotations.emplace_back(args: aRb);
186 abPointPairs.emplace_back(args: aTi.translation(), args: bTi.translation());
187 }
188 const Rot3 aRb_estimate = FindKarcherMean<Rot3>(rotations);
189
190 return internal::alignGivenR(abPointPairs, aRb: aRb_estimate);
191}

193Matrix4 Similarity3::Hat(const Vector7 &xi) {
194 // http://www.ethaneade.org/latex2html/lie/node29.html
195 const auto w = xi.head<3>();
196 const auto u = xi.segment<3>(start: 3);
197 const double lambda = xi[6];
198 Matrix4 W;
199 W << skewSymmetric(w), u, 0, 0, 0, -lambda;
200 return W;
201}

211Matrix7 Similarity3::AdjointMap() const {
212 // http://www.ethaneade.org/latex2html/lie/node30.html
213 const Matrix3 R = R_.matrix();
214 const Vector3 t = t_;
215 const Matrix3 A = s_ * skewSymmetric(w: t) * R;
216 Matrix7 adj;
217 adj << R, Z_3x3, Matrix31::Zero(), // 3*7
218 A, s_ * R, -s_ * t, // 3*7
219 Matrix16::Zero(), 1; // 1*7
220 return adj;
221}

224// Functor that implements the Similarity3 V(, ) kernel:
225// See http://www.ethaneade.org/latex2html/lie/node29.html
226struct LocalV : public so3::DexpFunctor {
227 double lambda; ///< scale log parameter
228 double alpha{0}, beta{0}, mu{0};
229 double P{0}, Q{0}, R{0};
230
231 explicit LocalV(const Vector3& omega, double lambda,
232 double nearZeroThresholdSq, double nearPiThresholdSq)
233 : so3::DexpFunctor(omega, nearZeroThresholdSq, nearPiThresholdSq),
234 lambda(lambda) {
235 compute_();
236 }
237
238 explicit LocalV(const Vector3& omega, double lambda)
239 : so3::DexpFunctor(omega), lambda(lambda) {
240 compute_();
241 }
242
243 void compute_() {
244 const double lambda2 = lambda * lambda, lambda3 = lambda2 * lambda;
245 if (lambda2 > 1e-9) {
246 const double e = std::exp(x: -lambda);
247 P = (1.0 - e) / lambda;
248 alpha = 1.0 / (1.0 + (theta2 / lambda2)); // = /(+)
249 beta = (e - 1.0 + lambda) / lambda2;
250 mu = (1.0 - lambda + 0.5 * lambda2 - e) / lambda3;
251 } else {
252 P = 1.0 - lambda / 2.0 + lambda2 / 6.0;
253 alpha = 0.0;
254 beta = 0.5 - lambda / 6.0 + lambda2 / 24.0 - lambda3 / 120.0;
255 mu = 1.0 / 6.0 - lambda / 24.0 + lambda2 / 120.0 - lambda3 / 720.0;
256 }
257 const double one_minus_alpha = 1.0 - alpha;
258 Q = alpha * beta + one_minus_alpha * (B - lambda * C());
259 R = alpha * mu + one_minus_alpha * (C() - lambda * E());
260 }
261
262 Matrix3 V() const { return P * I_3x3 + Q * W + R * WW; }
263
264 so3::Kernel kernel() const {
265 const double lambda2 = lambda * lambda;
266 const double dalpha =
267 (lambda2 > 1e-9) ? (-2.0 * alpha * alpha / lambda2) : 0.0;
268 const double dQ = (beta - (B - lambda * C())) * dalpha +
269 (1.0 - alpha) * (dB() - lambda * dC());
270 const double dR = (mu - (C() - lambda * E())) * dalpha +
271 (1.0 - alpha) * (dC() - lambda * dE());
272 return so3::Kernel{.S: this, .a: P, .b: Q, .c: R, .db: dQ, .dc: dR};
273 }
274};
275} // namespace

277Matrix3 Similarity3::GetV(Vector3 w, double lambda) {

280Vector7 Similarity3::Logmap(const Similarity3& T, OptionalJacobian<7, 7> Hm) {
281 // To get the logmap, calculate w and lambda, then solve for u as shown by Ethan at
282 // www.ethaneade.org/latex2html/lie/node29.html
283 const Vector3 w = Rot3::Logmap(R: T.R_);
284 const double lambda = log(x: T.s_);
285 Vector7 result;
286 result << w, GetV(w, lambda).inverse() * T.t_, lambda;
287 if (Hm) {
288 throw std::runtime_error("Similarity3::Logmap: derivative not implemented");
289 }
290 return result;
291}

293Similarity3 Similarity3::Expmap(const Vector7& v, OptionalJacobian<7, 7> Hm) {
294 const auto w = v.head<3>();
295 const auto rho = v.segment<3>(start: 3);
296 const double lambda = v[6];
297 if (Hm) {
298 throw std::runtime_error("Similarity3::Expmap: derivative not implemented");
299 }
300 const Matrix3 V = GetV(w, lambda);
301 return Similarity3(Rot3::Expmap(v: w), Point3(V * rho), exp(x: lambda));
302}

310Matrix4 Similarity3::matrix() const {
311 Matrix4 T;
312 T.topRows<3>() << R_.matrix(), t_;
313 T.bottomRows<1>() << 0, 0, 0, 1.0 / s_;
314 return T;
