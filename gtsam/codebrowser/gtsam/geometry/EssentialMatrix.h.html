20/**
21 * An essential matrix is like a Pose3, except with translation up to scale
22 * It is named after the 3*3 matrix aEb = [aTb]x aRb from computer vision,
23 * but here we choose instead to parameterize it as a (Rot3,Unit3) pair.
24 * We can then non-linearly optimize immediately on this 5-dimensional manifold.
25 */
26class EssentialMatrix {
27 private:
28 Rot3 R_; ///< Rotation
29 Unit3 t_; ///< Translation
30 Matrix3 E_; ///< Essential matrix
31
32 public:
33 /// Static function to convert Point2 to homogeneous coordinates
34 static Vector3 Homogeneous(const Point2& p) {
35 return Vector3(p.x(), p.y(), 1);
36 }
37
38 /// @name Constructors and named constructors
39 /// @{
40
41 /// Default constructor
42 EssentialMatrix() :E_(t_.skew()) {
43 }
44
45 /// Construct from rotation and translation
46 EssentialMatrix(const Rot3& aRb, const Unit3& aTb) :
47 R_(aRb), t_(aTb), E_(t_.skew() * R_.matrix()) {
48 }
49
50 /// Named constructor with derivatives
51 GTSAM_EXPORT static EssentialMatrix FromRotationAndDirection(const Rot3& aRb, const Unit3& aTb,
52 OptionalJacobian<5, 3> H1 = {},
53 OptionalJacobian<5, 2> H2 = {});
54
55 /// Named constructor converting a Pose3 with scale to EssentialMatrix (no scale)
56 GTSAM_EXPORT static EssentialMatrix FromPose3(const Pose3& _1P2_,
57 OptionalJacobian<5, 6> H = {});
58
59 /// Random, using Rot3::Random and Unit3::Random
60 template<typename Engine>
61 static EssentialMatrix Random(Engine & rng) {
62 return EssentialMatrix(Rot3::Random(rng), Unit3::Random(rng));
63 }
64
65 virtual ~EssentialMatrix() {}
66
67 /// @}
68
69 /// @name Testable
70 /// @{
71
72 /// print with optional string
73 GTSAM_EXPORT void print(const std::string& s = "") const;
74
75 /// assert equality up to a tolerance
76 bool equals(const EssentialMatrix& other, double tol = 1e-8) const {
77 return R_.equals(p: other.R_, tol)
78 && t_.equals(s: other.t_, tol);
79 }
80
81 /// @}
82
83 /// @name Manifold
84 /// @{
85 inline constexpr static auto dimension = 5;
86 inline static size_t Dim() { return dimension;}
87 inline size_t dim() const { return dimension;}
88
89 typedef OptionalJacobian<dimension, dimension> ChartJacobian;
90
91 /// Retract delta to manifold
92 EssentialMatrix retract(const Vector5& xi) const {
93 return EssentialMatrix(R_.retract(v: xi.head<3>()), t_.retract(v: xi.tail<2>()));
94 }
95
96 /// Compute the coordinates in the tangent space
97 Vector5 localCoordinates(const EssentialMatrix& other) const {
98 auto v1 = R_.localCoordinates(g: other.R_);
99 auto v2 = t_.localCoordinates(s: other.t_);
100 Vector5 v;
101 v << v1, v2;
102 return v;
103 }
104 /// @}
105
106 /// @name Essential matrix methods
107 /// @{
108
109 /// Rotation
110 inline const Rot3& rotation() const {
111 return R_;
112 }
113
114 /// Direction
115 inline const Unit3& direction() const {
116 return t_;
117 }
118
119 /// Return 3*3 matrix representation
120 inline const Matrix3& matrix() const {
121 return E_;
122 }
123
124 /// Return epipole in image_a , as Unit3 to allow for infinity
125 inline const Unit3& epipole_a() const {
126 return t_;
127 }
128
129 /// Return epipole in image_b, as Unit3 to allow for infinity
130 inline Unit3 epipole_b() const {
131 return R_.unrotate(p: t_);
132 }
133
134 /**
135 * @brief takes point in world coordinates and transforms it to pose with |t|==1
136 * @param p point in world coordinates
137 * @param DE optional 3*5 Jacobian wrpt to E
138 * @param Dpoint optional 3*3 Jacobian wrpt point
139 * @return point in pose coordinates
140 */
141 GTSAM_EXPORT Point3 transformTo(const Point3& p,
142 OptionalJacobian<3, 5> DE = {},
143 OptionalJacobian<3, 3> Dpoint = {}) const;
144
145 /**
146 * Given essential matrix E in camera frame B, convert to body frame C
147 * @param cRb rotation from body frame to camera frame
148 * @param E essential matrix E in camera frame C
149 */
150 GTSAM_EXPORT EssentialMatrix rotate(const Rot3& cRb, OptionalJacobian<5, 5> HE =
151 {}, OptionalJacobian<5, 3> HR = {}) const;
152
153 /**
154 * Given essential matrix E in camera frame B, convert to body frame C
155 * @param cRb rotation from body frame to camera frame
156 * @param E essential matrix E in camera frame C
157 */
158 friend EssentialMatrix operator*(const Rot3& cRb, const EssentialMatrix& E) {
159 return E.rotate(cRb);
160 }
161
162 /// epipolar error, algebraic
163 GTSAM_EXPORT double error(const Vector3& vA, const Vector3& vB,
164 OptionalJacobian<1, 5> H = {}) const;
165
166 /// @}
167
168 /// @name Streaming operators
169 /// @{
170
171 /// stream to stream
172 GTSAM_EXPORT friend std::ostream& operator <<(std::ostream& os, const EssentialMatrix& E);
173
174 /// stream from stream
175 GTSAM_EXPORT friend std::istream& operator >>(std::istream& is, EssentialMatrix& E);
176
177 /// @}
178
179 private:
180 /// @name Advanced Interface
181 /// @{
182
183#if GTSAM_ENABLE_BOOST_SERIALIZATION
184 /** Serialization function */
185 friend class boost::serialization::access;
186 template<class ARCHIVE>
187 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
188 ar & BOOST_SERIALIZATION_NVP(R_);
189 ar & BOOST_SERIALIZATION_NVP(t_);
190
191 ar & boost::serialization::make_nvp(n: "E11", v&: E_(0, 0));
192 ar & boost::serialization::make_nvp(n: "E12", v&: E_(0, 1));
193 ar & boost::serialization::make_nvp(n: "E13", v&: E_(0, 2));
194 ar & boost::serialization::make_nvp(n: "E21", v&: E_(1, 0));
195 ar & boost::serialization::make_nvp(n: "E22", v&: E_(1, 1));
196 ar & boost::serialization::make_nvp(n: "E23", v&: E_(1, 2));
197 ar & boost::serialization::make_nvp(n: "E31", v&: E_(2, 0));
198 ar & boost::serialization::make_nvp(n: "E32", v&: E_(2, 1));
199 ar & boost::serialization::make_nvp(n: "E33", v&: E_(2, 2));
200 }
201#endif
202
203 /// @}
204
205 public:
206 GTSAM_MAKE_ALIGNED_OPERATOR_NEW
207};
