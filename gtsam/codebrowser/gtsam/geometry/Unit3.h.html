39namespace gtsam {

41/// Represents a 3D point on a unit sphere.
42class GTSAM_EXPORT Unit3 {
43
44private:
45
46 Vector3 p_; ///< The location of the point on the unit sphere
47 mutable std::optional<Matrix32> B_; ///< Cached basis
48 mutable std::optional<Matrix62> H_B_; ///< Cached basis derivative
49
50#ifdef GTSAM_USE_TBB
51 mutable std::mutex B_mutex_; ///< Mutex to protect the cached basis.
52#endif
53
54public:
55
56 inline constexpr static auto dimension = 2;
57
58 /// @name Constructors
59 /// @{
60
61 /// Default constructor
62 Unit3() :
63 p_(1.0, 0.0, 0.0) {
64 }
65
66 /// Construct from point
67 explicit Unit3(const Vector3& p);
68
69 /// Construct from x,y,z
70 Unit3(double x, double y, double z);
71
72 /// Construct from 2D point in plane at focal length f
73 /// Unit3(p,1) can be viewed as normalized homogeneous coordinates of 2D point
74 explicit Unit3(const Point2& p, double f);
75
76 /// Copy constructor: copies essential data and discards caches.
77 Unit3(const Unit3& u) : p_(u.p_) {}
78
79 /// Copy assignment: copies essential data and invalidates local caches.
80 Unit3& operator=(const Unit3& u) {
81 if (this == &u) return *this;
82 p_ = u.p_;
83
84 // Since p_ has changed, the old cached basis is no longer valid.
85 B_.reset();
86 H_B_.reset();
87 return *this;
88 }
89
90 /// Named constructor from Point3 with optional Jacobian
91 static Unit3 FromPoint3(const Point3& point, //
92 OptionalJacobian<2, 3> H = {});
93
94 /**
95 * Random direction, using boost::uniform_on_sphere
96 * Example:
97 * std::mt19937 engine(42);
98 * Unit3 unit = Unit3::Random(engine);
99 */
100 static Unit3 Random(std::mt19937 & rng);
101
102 /// @}
103
104 /// @name Testable
105 /// @{
106
107 GTSAM_EXPORT friend std::ostream& operator<<(std::ostream& os,
108 const Unit3& pair);
109
110 /// The print fuction
111 void print(const std::string& s = std::string()) const;
112
113 /// The equals function with tolerance
114 bool equals(const Unit3& s, double tol = 1e-9) const {
115 return equal_with_abs_tol(A: p_, B: s.p_, tol);
116 }
117 /// @}
118
119 /// @name Other functionality
120 /// @{
121
122 /**
123 * Returns the local coordinate frame to tangent plane
124 * It is a 3*2 matrix [b1 b2] composed of two orthogonal directions
125 * tangent to the sphere at the current direction.
126 * Provides derivatives of the basis with the two basis vectors stacked up as a 6x1.
127 */
128 const Matrix32& basis(OptionalJacobian<6, 2> H = {}) const;
129
130 /// Return skew-symmetric associated with 3D point on unit sphere
131 Matrix3 skew() const;
132
133 /// Return unit-norm Point3
134 Point3 point3(OptionalJacobian<3, 2> H = {}) const;
135
136 /// Return unit-norm Vector
137 Vector3 unitVector(OptionalJacobian<3, 2> H = {}) const;
138
139 /// Return scaled direction as Point3
140 friend Point3 operator*(double s, const Unit3& d) {
141 return Point3(s * d.p_);
142 }
143
144 /// Return dot product with q
145 double dot(const Unit3& q, OptionalJacobian<1,2> H1 = {}, //
146 OptionalJacobian<1,2> H2 = {}) const;
147
148 /// Signed, vector-valued error between two directions
149 /// NOTE(hayk): This method has zero derivatives if this (p) and q are orthogonal.
150 Vector2 errorVector(const Unit3& q, OptionalJacobian<2, 2> H_p = {}, //
151 OptionalJacobian<2, 2> H_q = {}) const;
152
153 /// Distance between two directions
154 double distance(const Unit3& q, OptionalJacobian<1, 2> H = {}) const;
155
156 /// Cross-product between two Unit3s
157 Unit3 cross(const Unit3& q) const {
158 return Unit3(p_.cross(other: q.p_));
159 }
160
161 /// Cross-product w Point3
162 Point3 cross(const Point3& q) const {
163 return point3().cross(other: q);
164 }
165
166 /// @}
167 /// @name Manifold
168 /// @{
169
170 /// Dimensionality of tangent space = 2 DOF
171 inline static size_t Dim() {
172 return 2;
173 }
174
175 /// Dimensionality of tangent space = 2 DOF
176 inline size_t dim() const {
177 return 2;
178 }
179
180 enum CoordinatesMode {
181 EXPMAP, ///< Use the exponential map to retract
182 RENORM ///< Retract with vector addition and renormalize.
183 };
184
185 /// The retract function
186 Unit3 retract(const Vector2& v, OptionalJacobian<2,2> H = {}) const;
187
188 /// The local coordinates function
189 Vector2 localCoordinates(const Unit3& s) const;
190
191 /// @}
192
193#ifdef GTSAM_ALLOW_DEPRECATED_SINCE_V43
194 /// @deprecated, errorVector has the proper derivatives, this confusingly has only the second.
195 Vector2 error(const Unit3& q, OptionalJacobian<2, 2> H_q = {}) const {
196 return errorVector(q, H_p: {}, H_q);
197 }
198#endif
199
200private:
201
202 /// @name Advanced Interface
203 /// @{
204#if GTSAM_ENABLE_BOOST_SERIALIZATION
205 /** Serialization function */
206 friend class boost::serialization::access;
207 template<class ARCHIVE>
208 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
209 ar & BOOST_SERIALIZATION_NVP(p_);
210 }
211#endif
212
213 /// @}
214
215public:
216 GTSAM_MAKE_ALIGNED_OPERATOR_NEW
217};
