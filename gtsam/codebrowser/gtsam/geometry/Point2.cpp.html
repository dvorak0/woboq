27double norm2(const Point2& p, OptionalJacobian<1,2> H) {
28 double r = std::sqrt(x: p.x() * p.x() + p.y() * p.y());
29 if (H) {
30 if (std::abs(x: r) > 1e-10)
31 *H << p.x() / r, p.y() / r;
32 else
33 *H << 1, 1; // really infinity, why 1 ?
34 }
35 return r;
36}

38/* ************************************************************************* */
39double distance2(const Point2& p, const Point2& q, OptionalJacobian<1, 2> H1,
40 OptionalJacobian<1, 2> H2) {
41 Point2 d = q - p;
42 if (H1 || H2) {
43 Matrix12 H;
44 double r = norm2(p: d, H);
45 if (H1) *H1 = -H;
46 if (H2) *H2 = H;
47 return r;
48 } else {
49 return d.norm();
50 }
51}

54// Math inspired by http://paulbourke.net/geometry/circlesphere/
55std::optional<Point2> circleCircleIntersection(double R_d, double r_d,
56 double tol) {
57
58 double R2_d2 = R_d*R_d; // Yes, RD-D2 !
59 double f = 0.5 + 0.5*(R2_d2 - r_d*r_d);
60 double h2 = R2_d2 - f*f; // just right triangle rule
61
62 // h^2<0 is equivalent to (d > (R + r) || d < (R - r))
63 // Hence, there are only solutions if >=0
64 if (h2<-tol) return {}; // allow *slightly* negative
65 else if (h2<tol) return Point2(f,0.0); // one solution
66 else return Point2(f,std::sqrt(x: h2)); // two solutions
67}

69/* ************************************************************************* */
70list<Point2> circleCircleIntersection(Point2 c1, Point2 c2,
71 std::optional<Point2> fh) {
72
73 list<Point2> solutions;
74 // If fh==std::nullopt, there are no solutions, i.e., d > (R + r) || d < (R - r)
75 if (fh) {
76 // vector between circle centers
77 Point2 c12 = c2-c1;
78
79 // Determine p2, the point where the line through the circle
80 // intersection points crosses the line between the circle centers.
81 Point2 p2 = c1 + fh->x() * c12;
82
83 // If h == 0, the circles are touching, so just return one point
84 if (fh->y()==0.0)
85 solutions.push_back(x: p2);
86 else {
87 // determine the offsets of the intersection points from p
88 Point2 offset = fh->y() * Point2(-c12.y(), c12.x());
89
90 // Determine the absolute intersection points.
91 solutions.push_back(x: p2 + offset);
92 solutions.push_back(x: p2 - offset);
93 }
94 }
95 return solutions;
96}

98/* ************************************************************************* */
99list<Point2> circleCircleIntersection(Point2 c1, double r1, Point2 c2,
100 double r2, double tol) {
101
102 // distance between circle centers.
103 double d = distance2(p: c1, q: c2);
104
105 // centers coincide, either no solution or infinite number of solutions.
106 if (d<1e-9) return list<Point2>();
107
108 // Calculate f and h given normalized radii
109 double _d = 1.0/d, R_d = r1*_d, r_d=r2*_d;
110 std::optional<Point2> fh = circleCircleIntersection(R_d,r_d);
111
112 // Call version that takes fh
113 return circleCircleIntersection(c1, c2, fh);
114}

116Point2Pair means(const std::vector<Point2Pair> &abPointPairs) {
117 const size_t n = abPointPairs.size();
118 if (n == 0) throw std::invalid_argument("Point2::mean input Point2Pair vector is empty");
119 Point2 aSum(0, 0), bSum(0, 0);
120 for (const Point2Pair &abPair : abPointPairs) {
121 aSum += abPair.first;
122 bSum += abPair.second;
123 }
124 const double f = 1.0 / n;
125 return {aSum * f, bSum * f};
126}
