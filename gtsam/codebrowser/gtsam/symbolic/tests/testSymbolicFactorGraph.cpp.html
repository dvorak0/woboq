92TEST(SymbolicFactorGraph, eliminatePartialMultifrontal) {
93 SymbolicBayesTree expectedBayesTree;
94 SymbolicConditional::shared_ptr root =
95 std::make_shared<SymbolicConditional>(
96 args: SymbolicConditional::FromKeys(keys: KeyVector{4, 5, 1}, nrFrontals: 2));
97 expectedBayesTree.insertRoot(
98 subtree: std::make_shared<SymbolicBayesTreeClique>(args&: root));
99
100 const auto expectedFactorGraph =
101 SymbolicFactorGraph(SymbolicFactor(0, 1))(SymbolicFactor(0, 2))(
102 SymbolicFactor(1, 3))(SymbolicFactor(2, 3))(SymbolicFactor(1));
103
104 const auto [actualBayesTree, actualFactorGraph] =
105 simpleTestGraph2.eliminatePartialMultifrontal(ordering: Ordering{4, 5});
106
107 EXPECT(assert_equal(expectedFactorGraph, *actualFactorGraph));
108 EXPECT(assert_equal(expectedBayesTree, *actualBayesTree));
109
110 SymbolicBayesTree expectedBayesTree2;
111 SymbolicBayesTreeClique::shared_ptr root2 =
112 std::make_shared<SymbolicBayesTreeClique>(
113 args: std::make_shared<SymbolicConditional>(args: 4, args: 1));
114 root2->children.push_back(x: std::make_shared<SymbolicBayesTreeClique>(
115 args: std::make_shared<SymbolicConditional>(args: 5, args: 4)));
116 expectedBayesTree2.insertRoot(subtree: root2);
117
118 const auto [actualBayesTree2, actualFactorGraph2] =
119 simpleTestGraph2.eliminatePartialMultifrontal(variables: KeyVector{4, 5});
120
121 EXPECT(assert_equal(expectedFactorGraph, *actualFactorGraph2));
122 EXPECT(assert_equal(expectedBayesTree2, *actualBayesTree2));
123}

232TEST(SymbolicFactorGraph, marginals) {
233 // Create factor graph
234 SymbolicFactorGraph fg;
235 fg.push_factor(key1: 0, key2: 1);
236 fg.push_factor(key1: 0, key2: 2);
237 fg.push_factor(key1: 1, key2: 4);
238 fg.push_factor(key1: 2, key2: 4);
239 fg.push_factor(key1: 3, key2: 4);
240
241 // eliminate
242 Ordering ord{3, 4, 2, 1, 0};
243 auto actual = fg.eliminateSequential(ordering: ord);
244 SymbolicBayesNet expected;
245 expected.emplace_shared<SymbolicConditional>(args: 3, args: 4);
246 expected.emplace_shared<SymbolicConditional>(args: 4, args: 1, args: 2);
247 expected.emplace_shared<SymbolicConditional>(args: 2, args: 0, args: 1);
248 expected.emplace_shared<SymbolicConditional>(args: 1, args: 0);
249 expected.emplace_shared<SymbolicConditional>(args: 0);
250 EXPECT(assert_equal(expected, *actual));
251
252 {
253 // jointBayesNet
254 Ordering ord{0, 4, 3};
255 auto actual = fg.eliminatePartialSequential(ordering: ord);
256 SymbolicBayesNet expectedBN;
257 expectedBN.emplace_shared<SymbolicConditional>(args: 0, args: 1, args: 2);
258 expectedBN.emplace_shared<SymbolicConditional>(args: 4, args: 1, args: 2, args: 3);
259 expectedBN.emplace_shared<SymbolicConditional>(args: 3, args: 1, args: 2);
260 EXPECT(assert_equal(expectedBN, *(actual.first)));
261 }
262
263 {
264 // jointBayesNet
265 Ordering ord{0, 2, 3};
266 auto actual = fg.eliminatePartialSequential(ordering: ord);
267 SymbolicBayesNet expectedBN;
268 expectedBN.emplace_shared<SymbolicConditional>(args: 0, args: 1, args: 2);
269 expectedBN.emplace_shared<SymbolicConditional>(args: 2, args: 1, args: 4);
270 expectedBN.emplace_shared<SymbolicConditional>(args: 3, args: 4);
271 EXPECT(assert_equal(expectedBN, *(actual.first)));
272 }
273
274 {
275 // conditionalBayesNet
276 Ordering ord{0, 2};
277 auto actual = fg.eliminatePartialSequential(ordering: ord);
278 SymbolicBayesNet expectedBN;
279 expectedBN.emplace_shared<SymbolicConditional>(args: 0, args: 1, args: 2);
280 expectedBN.emplace_shared<SymbolicConditional>(args: 2, args: 1, args: 4);
281 EXPECT(assert_equal(expectedBN, *(actual.first)));
282 }
283}

321TEST(SymbolicFactorGraph, push_back) {
322 // Create two factor graphs and expected combined graph
323 SymbolicFactorGraph fg1, fg2, expected;
324
325 fg1.push_factor(key: 1);
326 fg1.push_factor(key1: 0, key2: 1);
327
328 fg2.push_factor(key1: 1, key2: 2);
329 fg2.push_factor(key1: 0, key2: 2);
330
331 expected.push_factor(key: 1);
332 expected.push_factor(key1: 0, key2: 1);
333 expected.push_factor(key1: 1, key2: 2);
334 expected.push_factor(key1: 0, key2: 2);
335
336 // combine
337 SymbolicFactorGraph actual;
338 actual.push_back(container: fg1);
339 actual.push_back(container: fg2);
340 CHECK(assert_equal(expected, actual));
341
342 // combine in second way
343 SymbolicFactorGraph actual2 = fg1;
344 actual2.push_back(container: fg2);
345 CHECK(assert_equal(expected, actual2));

349TEST(SymbolicFactorGraph, add_factors) {
350 SymbolicFactorGraph fg1;
351 fg1.push_factor(key: 10);
352 fg1.push_back(factor: SymbolicFactor::shared_ptr()); // empty slot!
353 fg1.push_factor(key: 11);
354
355 SymbolicFactorGraph fg2;
356 fg2.push_factor(key: 1);
357 fg2.push_factor(key: 2);
358
359 SymbolicFactorGraph expected;
360 expected.push_factor(key: 10);
361 expected.push_factor(key: 1);
362 expected.push_factor(key: 11);
363 expected.push_factor(key: 2);
364 const FactorIndices expectedIndices{1, 3};
365 const FactorIndices actualIndices = fg1.add_factors(factors: fg2, useEmptySlots: true);
366
367 EXPECT(assert_equal(expected, fg1));
368 EXPECT(assert_container_equality(expectedIndices, actualIndices));
369
370 expected.push_factor(key: 1);
371 expected.push_factor(key: 2);
372 const FactorIndices expectedIndices2{4, 5};
373 const FactorIndices actualIndices2 = fg1.add_factors(factors: fg2, useEmptySlots: false);
374
375 EXPECT(assert_equal(expected, fg1));
376 EXPECT(assert_container_equality(expectedIndices2, actualIndices2));
377}
