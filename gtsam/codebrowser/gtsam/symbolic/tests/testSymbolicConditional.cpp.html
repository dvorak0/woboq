26/* ************************************************************************* */
27TEST( SymbolicConditional, empty )
28{
29 SymbolicConditional c0;
30 LONGS_EQUAL(0, (long)c0.nrFrontals());
31 LONGS_EQUAL(0, (long)c0.nrParents());
32}
33
34/* ************************************************************************* */
35TEST( SymbolicConditional, noParents )
36{
37 SymbolicConditional c0(0);
38 LONGS_EQUAL(1, (long)c0.nrFrontals());
39 LONGS_EQUAL(0, (long)c0.nrParents());
40}

67TEST( SymbolicConditional, fourParents )
68{

70 LONGS_EQUAL(1, (long)c0.nrFrontals());
71 LONGS_EQUAL(4, (long)c0.nrParents());
72}

74/* ************************************************************************* */
75TEST( SymbolicConditional, FromRange )
76{
77 auto c0 = std::make_shared<SymbolicConditional>(
78 args: SymbolicConditional::FromKeys(keys: KeyVector{1, 2, 3, 4, 5}, nrFrontals: 2));
79 LONGS_EQUAL(2, (long)c0->nrFrontals());
80 LONGS_EQUAL(3, (long)c0->nrParents());
81}

83/* ************************************************************************* */
84TEST(SymbolicConditional, Constructors)
85{
86 SymbolicConditional expected(3, 4);
87
88 SymbolicConditional actual1 = SymbolicConditional::FromKeys(keys: expected.keys(), nrFrontals: 1);
89 SymbolicConditional actual2 = SymbolicConditional::FromIterators(firstKey: expected.begin(), lastKey: expected.end(), nrFrontals: 1);
90 SymbolicConditional actual3 = *SymbolicConditional::FromKeysShared(keys: expected.keys(), nrFrontals: 1);
91 SymbolicConditional actual4 = *SymbolicConditional::FromIteratorsShared(firstKey: expected.begin(), lastKey: expected.end(), nrFrontals: 1);
92
93 EXPECT(assert_equal(expected, actual1));
94 EXPECT(assert_equal(expected, actual2));
95 EXPECT(assert_equal(expected, actual3));
96 EXPECT(assert_equal(expected, actual4));
97}

99/* ************************************************************************* */
100TEST( SymbolicConditional, equals )
101{
102 SymbolicConditional c0(0, 1, 2), c1(0, 1, 2), c2(1, 2, 3), c3(3,4);
103 CHECK(c0.equals(c1));
104 CHECK(c1.equals(c0));
105 CHECK(!c0.equals(c2));
106 CHECK(!c2.equals(c0));
107 CHECK(!c0.equals(c3));
108 CHECK(!c3.equals(c0));
109}
