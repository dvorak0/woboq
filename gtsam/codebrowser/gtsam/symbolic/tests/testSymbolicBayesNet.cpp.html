31static const Key _L_ = 0;
32static const Key _A_ = 1;
33static const Key _B_ = 2;
34static const Key _C_ = 3;
35
36static SymbolicConditional::shared_ptr
37 B(new SymbolicConditional(_B_)),
38 L(new SymbolicConditional(_L_, _B_));
39
40/* ************************************************************************* */
41TEST( SymbolicBayesNet, equals )
42{
43 SymbolicBayesNet f1;
44 f1.push_back(factor: B);
45 f1.push_back(factor: L);
46 SymbolicBayesNet f2;
47 f2.push_back(factor: L);
48 f2.push_back(factor: B);
49 CHECK(f1.equals(f1));
50 CHECK(!f1.equals(f2));
51}

53/* ************************************************************************* */
54TEST( SymbolicBayesNet, combine )
55{
56 SymbolicConditional::shared_ptr
57 A(new SymbolicConditional(_A_,_B_,_C_)),
58 B(new SymbolicConditional(_B_,_C_)),
59 C(new SymbolicConditional(_C_));
60
61 // p(A|BC)
62 SymbolicBayesNet p_ABC;
63 p_ABC.push_back(factor: A);
64
65 // P(BC)=P(B|C)P(C)
66 SymbolicBayesNet p_BC;
67 p_BC.push_back(factor: B);
68 p_BC.push_back(factor: C);
69
70 // P(ABC) = P(A|BC) P(BC)
71 p_ABC.push_back(container: p_BC);
72
73 SymbolicBayesNet expected;
74 expected.push_back(factor: A);
75 expected.push_back(factor: B);
76 expected.push_back(factor: C);
77
78 CHECK(assert_equal(expected,p_ABC));
79}

82TEST(SymbolicBayesNet, Dot) {
83 using symbol_shorthand::A;
84 using symbol_shorthand::X;
85 SymbolicBayesNet bn;
86 bn.emplace_shared<SymbolicConditional>(args: X(j: 3), args: X(j: 2), args: A(j: 2));
87 bn.emplace_shared<SymbolicConditional>(args: X(j: 2), args: X(j: 1), args: A(j: 1));
88 bn.emplace_shared<SymbolicConditional>(args: X(j: 1));
89
90 DotWriter writer;
91 writer.positionHints.emplace(args: 'a', args: 2);
92 writer.positionHints.emplace(args: 'x', args: 1);
93 writer.boxes.emplace(args: A(j: 1));
94 writer.boxes.emplace(args: A(j: 2));
95
96 auto position = writer.variablePos(key: A(j: 1));
97 CHECK(position);
98 EXPECT(assert_equal(Vector2(1, 2), *position, 1e-5));
99
100 string actual = bn.dot(keyFormatter: DefaultKeyFormatter, writer);
101 bn.saveGraph(filename: "bn.dot", keyFormatter: DefaultKeyFormatter, writer);
102 EXPECT(actual ==
103 "digraph {\n"
104 " size=\"5,5\";\n"
105 "\n"
106 " var6989586621679009793[label=\"a1\", pos=\"1,2!\", shape=box];\n"

117}
