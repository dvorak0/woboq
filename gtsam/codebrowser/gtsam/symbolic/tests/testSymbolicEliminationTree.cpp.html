31using sharedNode = SymbolicEliminationTree::sharedNode;
32
33// Use list_of replacement defined in symbolicExampleGraphs.h
34using ChildNodes = ChainedVector<sharedNode>;
35
36class EliminationTreeTester {
37 public:
38 // build hardcoded tree
39 static SymbolicEliminationTree buildHardcodedTree(
40 const SymbolicFactorGraph& fg) {
41 sharedNode leaf0(new SymbolicEliminationTree::Node);
42 leaf0->key = 0;
43 leaf0->factors.push_back(x: fg[0]);
44 leaf0->factors.push_back(x: fg[1]);
45
46 sharedNode node1(new SymbolicEliminationTree::Node);
47 node1->key = 1;
48 node1->factors.push_back(x: fg[2]);
49 node1->children.push_back(x: leaf0);
50
51 sharedNode node2(new SymbolicEliminationTree::Node);
52 node2->key = 2;
53 node2->factors.push_back(x: fg[3]);
54 node2->children.push_back(x: node1);
55
56 sharedNode leaf3(new SymbolicEliminationTree::Node);
57 leaf3->key = 3;
58 leaf3->factors.push_back(x: fg[4]);
59
60 sharedNode root(new SymbolicEliminationTree::Node);
61 root->key = 4;
62 root->children.push_back(x: leaf3);
63 root->children.push_back(x: node2);
64
65 SymbolicEliminationTree tree;
66 tree.roots_.push_back(x: root);
67 return tree;
68 }
69
70 static SymbolicEliminationTree MakeTree(const ChildNodes::Result& roots) {
71 SymbolicEliminationTree et;
72 et.roots_.assign(first: roots.begin(), last: roots.end());
73 return et;
74 }
75};
76
77// Create a leaf node.
78static sharedNode Leaf(Key key, const SymbolicFactorGraph& factors) {
79 sharedNode node(new SymbolicEliminationTree::Node());
80 node->key = key;
81 node->factors.assign(first: factors.begin(), last: factors.end());
82 return node;
83}
84
85// Create a node with children.
86static sharedNode Node(Key key, const SymbolicFactorGraph& factors,
87 const ChildNodes::Result& children) {
88 sharedNode node = Leaf(key, factors);
89 node->children.assign(first: children.begin(), last: children.end());
90 return node;
91}

93/* ************************************************************************* */
94TEST(EliminationTree, Create) {
95 SymbolicEliminationTree expected =
96 EliminationTreeTester::buildHardcodedTree(fg: simpleTestGraph1);
97
98 // Build from factor graph
99 const Ordering order{0, 1, 2, 3, 4};
100 SymbolicEliminationTree actual(simpleTestGraph1, order);
101
102 CHECK(assert_equal(expected, actual));
103}

107 // l1 l2
108 // / | / |
109 // x1 --- x2 --- x3 --- x4 --- x5
110 // \ |
111 // l3
112 SymbolicFactorGraph graph;
113 graph.emplace_shared<SymbolicFactor>(args: X(j: 1), args: L(j: 1));
114 graph.emplace_shared<SymbolicFactor>(args: X(j: 1), args: X(j: 2));
115 graph.emplace_shared<SymbolicFactor>(args: X(j: 2), args: L(j: 1));
116 graph.emplace_shared<SymbolicFactor>(args: X(j: 2), args: X(j: 3));
117 graph.emplace_shared<SymbolicFactor>(args: X(j: 3), args: X(j: 4));
118 graph.emplace_shared<SymbolicFactor>(args: X(j: 4), args: L(j: 2));
119 graph.emplace_shared<SymbolicFactor>(args: X(j: 4), args: X(j: 5));

122 graph.emplace_shared<SymbolicFactor>(args: X(j: 5), args: L(j: 3));

158 ) //
