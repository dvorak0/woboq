36 template<> struct EliminationTraits<SymbolicFactorGraph>
37 {
38 typedef SymbolicFactor FactorType; ///< Type of factors in factor graph
39 typedef SymbolicFactorGraph FactorGraphType; ///< Type of the factor graph (e.g. GaussianFactorGraph)
40 typedef SymbolicConditional ConditionalType; ///< Type of conditionals from elimination
41 typedef SymbolicBayesNet BayesNetType; ///< Type of Bayes net from sequential elimination
42 typedef SymbolicEliminationTree EliminationTreeType; ///< Type of elimination tree
43 typedef SymbolicBayesTree BayesTreeType; ///< Type of Bayes tree
44 typedef SymbolicJunctionTree JunctionTreeType; ///< Type of Junction tree
45 /// The default dense elimination function
46 static std::pair<std::shared_ptr<ConditionalType>, std::shared_ptr<FactorType> >
47 DefaultEliminate(const FactorGraphType& factors, const Ordering& keys) {
48 return EliminateSymbolic(factors, keys); }
49 /// The default ordering generation function
50 static Ordering DefaultOrderingFunc(
51 const FactorGraphType& graph,
52 std::optional<std::reference_wrapper<const VariableIndex>> variableIndex) {
53 return Ordering::Colamd(variableIndex: (*variableIndex).get());
54 }
55 };

57 /* ************************************************************************* */
58 /** Symbolic Factor Graph
59 * \nosubgrouping
60 */
61 class GTSAM_EXPORT SymbolicFactorGraph :
62 public FactorGraph<SymbolicFactor>,
63 public EliminateableFactorGraph<SymbolicFactorGraph>
64 {
65 public:
66
67 typedef SymbolicFactorGraph This; ///< Typedef to this class
68 typedef FactorGraph<SymbolicFactor> Base; ///< Typedef to base factor graph type
69 typedef EliminateableFactorGraph<This> BaseEliminateable; ///< Typedef to base elimination class
70 typedef std::shared_ptr<This> shared_ptr; ///< shared_ptr to this class
71
72 /// @name Standard Constructors
73 /// @{
74
75 /** Construct empty factor graph */
76 SymbolicFactorGraph() {}
77
78 /** Construct from iterator over factors */
79 template<typename ITERATOR>
80 SymbolicFactorGraph(ITERATOR firstFactor, ITERATOR lastFactor) : Base(firstFactor, lastFactor) {}
81
82 /** Construct from container of factors (shared_ptr or plain objects) */
83 template<class CONTAINER>
84 explicit SymbolicFactorGraph(const CONTAINER& factors) : Base(factors) {}
85
86 /** Implicit copy/downcast constructor to override explicit template container constructor */
87 template<class DERIVEDFACTOR>
88 SymbolicFactorGraph(const FactorGraph<DERIVEDFACTOR>& graph) : Base(graph) {}
89
90 /**
91 * Constructor that takes an initializer list of shared pointers.
92 * FactorGraph fg = {make_shared<MyFactor>(), ...};
93 */
94 SymbolicFactorGraph(
95 std::initializer_list<std::shared_ptr<SymbolicFactor>> sharedFactors)
96 : Base(sharedFactors) {}
97
98 /// Construct from a single factor
99 SymbolicFactorGraph(SymbolicFactor&& c) {
100 emplace_shared<SymbolicFactor>(args&: c);
101 }
102
103 /**
104 * @brief Add a single factor and return a reference.
105 * This allows for chaining, e.g.,
106 * SymbolicFactorGraph bn =
107 * SymbolicFactorGraph(SymbolicFactor(...))(SymbolicFactor(...));
108 */
109 SymbolicFactorGraph& operator()(SymbolicFactor&& c) {
110 emplace_shared<SymbolicFactor>(args&: c);
111 return *this;
112 }
113
114 /// @}
115
116 /// @name Testable
117 /// @{
118
119 bool equals(const This& fg, double tol = 1e-9) const;
120
121 /// print
122 void print(
123 const std::string& s = "SymbolicFactorGraph",
124 const KeyFormatter& formatter = DefaultKeyFormatter) const override {
125 Base::print(s, formatter);
126 }
127
128 /// @}
129
130 /// @name Standard Interface
131 /// @{
132
133 /** Push back unary factor */
134 void push_factor(Key key);
135
136 /** Push back binary factor */
137 void push_factor(Key key1, Key key2);
138
139 /** Push back ternary factor */
140 void push_factor(Key key1, Key key2, Key key3);
141
142 /** Push back 4-way factor */
143 void push_factor(Key key1, Key key2, Key key3, Key key4);
144
145 /// @}
146
147 private:
148#if GTSAM_ENABLE_BOOST_SERIALIZATION
149 /** Serialization function */
150 friend class boost::serialization::access;
151 template<class ARCHIVE>
152 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
153 ar & BOOST_SERIALIZATION_BASE_OBJECT_NVP(Base);
154 }
155#endif
156 };
