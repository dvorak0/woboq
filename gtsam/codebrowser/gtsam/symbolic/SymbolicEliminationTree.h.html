21#include <gtsam/symbolic/SymbolicBayesNet.h>

25namespace gtsam {
26
27 class GTSAM_EXPORT SymbolicEliminationTree :
28 public EliminationTree<SymbolicBayesNet, SymbolicFactorGraph>
29 {
30 public:
31 typedef EliminationTree<SymbolicBayesNet, SymbolicFactorGraph> Base; ///< Base class
32 typedef SymbolicEliminationTree This; ///< This class
33 typedef std::shared_ptr<This> shared_ptr; ///< Shared pointer to this class
34
35 /** Build the elimination tree of a factor graph using pre-computed column structure.
36 * @param factorGraph The factor graph for which to build the elimination tree
37 * @param structure The set of factors involving each variable. If this is not precomputed,
38 * you can call the Create(const FactorGraph<DERIVEDFACTOR>&) named constructor instead.
39 * @return The elimination tree */
40 SymbolicEliminationTree(const SymbolicFactorGraph& factorGraph,
41 const VariableIndex& structure, const Ordering& order);
42
43 /** Build the elimination tree of a factor graph. Note that this has to compute the column
44 * structure as a VariableIndex, so if you already have this precomputed, use the other
45 * constructor instead.
46 * @param factorGraph The factor graph for which to build the elimination tree */
47 SymbolicEliminationTree(const SymbolicFactorGraph& factorGraph,
48 const Ordering& order);
49
50 /** Test whether the tree is equal to another */
51 bool equals(const This& other, double tol = 1e-9) const;
52
53 private:
54
55 /** Private default constructor for testing */
56 SymbolicEliminationTree() {}
57
58 friend class ::EliminationTreeTester;
59
60 };

65};
