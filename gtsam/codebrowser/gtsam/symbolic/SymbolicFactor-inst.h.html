34 /** Implementation of dense elimination function for symbolic factors. This is a templated
35 * version for internally doing symbolic elimination on any factor. */
36 template<class FACTOR>
37 std::pair<std::shared_ptr<SymbolicConditional>, std::shared_ptr<SymbolicFactor> >
38 EliminateSymbolic(const FactorGraph<FACTOR>& factors, const Ordering& keys)
39 {
40 gttic(EliminateSymbolic);
41
42 // Gather all keys
43 KeySet allKeys;
44 for(const std::shared_ptr<FACTOR>& factor: factors) {
45 // Non-active factors are nullptr
46 if (factor)
47 allKeys.insert(factor->begin(), factor->end());
48 }
49
50 // Check keys
51 for(Key key: keys) {
52 if(allKeys.find(x: key) == allKeys.end())
53 throw std::runtime_error("Requested to eliminate a key that is not in the factors");
54 }
55
56 // Sort frontal keys
57 KeySet frontals(keys);
58 const size_t nFrontals = keys.size();
59
60 // Build a key vector with the frontals followed by the separator
61 KeyVector orderedKeys(allKeys.size());
62 std::copy(first: keys.begin(), last: keys.end(), result: orderedKeys.begin());
63 std::set_difference(first1: allKeys.begin(), last1: allKeys.end(), first2: frontals.begin(), last2: frontals.end(), result: orderedKeys.begin() + nFrontals);
64
65 // Return resulting conditional and factor
66 return {
67 SymbolicConditional::FromKeysShared(keys: orderedKeys, nrFrontals: nFrontals),
68 SymbolicFactor::FromIteratorsShared(beginKey: orderedKeys.begin() + nFrontals, endKey: orderedKeys.end())
69 };
70 }
71 }
