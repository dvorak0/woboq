21namespace gtsam {
22
23/**
24 * @brief Scale x from [a, b] to [t1, t2]
25 *
26 * ((b'-a') * (x - a) / (b - a)) + a'
27 *
28 * @param x Value to scale to new range.
29 * @param a Original lower limit.
30 * @param b Original upper limit.
31 * @param t1 New lower limit.
32 * @param t2 New upper limit.
33 * @return double
34 */
35static double scale(double x, double a, double b, double t1, double t2) {
36 return ((t2 - t1) * (x - a) / (b - a)) + t1;
37}
38
39Weights Chebyshev1Basis::CalculateWeights(size_t N, double x, double a,
40 double b) {
41 Weights Tx(1, N);
42
43 x = scale(x, a, b, t1: -1, t2: 1);
44
45 Tx(0) = 1;
46 Tx(1) = x;
47 for (size_t i = 2; i < N; i++) {
48 // instead of cos(i*acos(x)), this recurrence is much faster
49 Tx(i) = 2 * x * Tx(i - 1) - Tx(i - 2);
50 }
51 return Tx;
52}
53
54Weights Chebyshev1Basis::DerivativeWeights(size_t N, double x, double a,
55 double b) {
56 Weights Ux = Chebyshev2Basis::CalculateWeights(N, x, a, b);
57 Weights weights = Weights::Zero(size: N);
58 for (size_t n = 1; n < N; n++) {
59 weights(n) = n * Ux(n - 1);
60 }
61 return weights;
62}
63
64Weights Chebyshev2Basis::CalculateWeights(size_t N, double x, double a,
65 double b) {
66 Weights Ux(N);
67
68 x = scale(x, a, b, t1: -1, t2: 1);
69
70 Ux(0) = 1;
71 Ux(1) = 2 * x;
72 for (size_t i = 2; i < N; i++) {
73 // instead of cos(i*acos(x)), this recurrence is much faster
74 Ux(i) = 2 * x * Ux(i - 1) - Ux(i - 2);
75 }
76 return Ux;
77}
78
79Weights Chebyshev2Basis::DerivativeWeights(size_t N, double x, double a,
80 double b) {
81 Weights Tx = Chebyshev1Basis::CalculateWeights(N: N + 1, x, a, b);
82 Weights Ux = Chebyshev2Basis::CalculateWeights(N, x, a, b);
83
84 Weights weights(N);
85
86 x = scale(x, a, b, t1: -1, t2: 1);
87 if (x == -1 || x == 1) {
88 throw std::runtime_error(
89 "Derivative of Chebyshev2 Basis does not exist at range limits.");
90 }
91
92 for (size_t n = 0; n < N; n++) {
93 weights(n) = ((n + 1) * Tx(n + 1) - x * Ux(n)) / (x * x - 1);
94 }
95 return weights;
96}
