23Matrix kroneckerProductIdentity(size_t M, const Weights& w) {
24 Matrix result(M, w.cols() * M);
25 result.setZero();
26
27 for (int i = 0; i < w.cols(); i++) {
28 result.block(startRow: 0, startCol: i * M, blockRows: M, blockCols: M).diagonal().array() = w(i);
29 }
30 return result;
31}
