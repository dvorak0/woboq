12/**
13 * @file Chebyshev2.h
14 * @brief Pseudo-spectral parameterization for Chebyshev polynomials of the
15 * second kind.
16 *
17 * In a pseudo-spectral case, rather than the parameters acting as
18 * weights for the bases polynomials (as in Chebyshev2Basis), here the
19 * parameters are the *values* at a specific set of points in the interval, the
20 * "Chebyshev points". These values uniquely determine the polynomial that
21 * interpolates them at the Chebyshev points.
22 *
23 * This is different from Chebyshev.h since it leverage ideas from
24 * pseudo-spectral optimization, i.e. we don't decompose into basis functions,
25 * rather estimate function values at the Chebyshev points.
26 *
27 * Please refer to Agrawal21icra for more details.
28 *
29 * @author Varun Agrawal, Jing Dong, Frank Dellaert
30 * @date July 4, 2020
31 */
32
33#pragma once
34
35#include <gtsam/base/Manifold.h>
36#include <gtsam/base/OptionalJacobian.h>
37#include <gtsam/basis/Basis.h>
38
39namespace gtsam {
40
41/**
42 * Chebyshev Interpolation on Chebyshev points of the second kind
43 * Note that N here, the number of points, is one less than N from
44 * 'Approximation Theory and Approximation Practice by L. N. Trefethen (pg.42)'.
45 */
46class GTSAM_EXPORT Chebyshev2 : public Basis<Chebyshev2> {
47 public:
48 EIGEN_MAKE_ALIGNED_OPERATOR_NEW
49
50 using Base = Basis<Chebyshev2>;
51 using Parameters = Eigen::Matrix<double, /*Nx1*/ -1, 1>;
52 using DiffMatrix = Eigen::Matrix<double, /*NxN*/ -1, -1>;
53
54 /**
55 * @brief Specific Chebyshev point, within [-1,1] interval.
56 *
57 * @param N The degree of the polynomial
58 * @param j The index of the Chebyshev point
59 * @return double
60 */
61 static double Point(size_t N, int j);
62
63 /**
64 * @brief Specific Chebyshev point, within [a,b] interval.
65 *
66 * @param N The degree of the polynomial
67 * @param j The index of the Chebyshev point
68 * @param a Lower bound of interval (default: -1)
69 * @param b Upper bound of interval (default: 1)
70 * @return double
71 */
72 static double Point(size_t N, int j, double a, double b);
73
74 /// All Chebyshev points
75 static Vector Points(size_t N);
76
77 /// All Chebyshev points, within [a,b] interval
78 static Vector Points(size_t N, double a, double b);
79
80 /**
81 * Evaluate Chebyshev Weights on [-1,1] at any x up to order N-1 (N values)
82 * These weights implement barycentric interpolation at a specific x.
83 * More precisely, f(x) ~ [w0;...;wN] * [f0;...;fN], where the fj are the
84 * values of the function f at the Chebyshev points. As such, for a given x we
85 * obtain a linear map from parameter vectors f to interpolated values f(x).
86 * Optional [a,b] interval can be specified as well.
87 */
88 static Weights CalculateWeights(size_t N, double x, double a = -1, double b = 1);
89
90 /**
91 * Evaluate derivative of barycentric weights.
92 * This is easy and efficient via the DifferentiationMatrix.
93 */
94 static Weights DerivativeWeights(size_t N, double x, double a = -1, double b = 1);
95
96 /// Compute D = differentiation matrix, Trefethen00book p.53
97 /// When given a parameter vector f of function values at the Chebyshev
98 /// points, D*f are the values of f'.
99 /// https://people.maths.ox.ac.uk/trefethen/8all.pdf Theorem 8.4
100 static DiffMatrix DifferentiationMatrix(size_t N);
101
102 /// Compute D = differentiation matrix, for interval [a,b]
103 static DiffMatrix DifferentiationMatrix(size_t N, double a, double b);
104
105 /// IntegrationMatrix returns the (N+1)×(N+1) matrix P such that for any f,
106 /// F = P * f recovers F (the antiderivative) satisfying f = D * F and F(0)=0.
107 static Matrix IntegrationMatrix(size_t N);
108
109 /// IntegrationMatrix returns the (N+1)×(N+1) matrix P for interval [a,b]
110 static Matrix IntegrationMatrix(size_t N, double a, double b);
111
112 /**
113 * Calculate Clenshaw-Curtis integration weights.
114 * Trefethen00book, pg 128, clencurt.m
115 * Note that N in clencurt.m is 1 less than our N
116 */
117 static Weights IntegrationWeights(size_t N);
118
119 /// Calculate Clenshaw-Curtis integration weights, for interval [a,b]
120 static Weights IntegrationWeights(size_t N, double a, double b);
121
122 /**
123 * Calculate Double Clenshaw-Curtis integration weights
124 * We compute them as W * P, where W are the Clenshaw-Curtis weights and P is
125 * the integration matrix.
126 */
127 static Weights DoubleIntegrationWeights(size_t N);
128
129 /// Calculate Double Clenshaw-Curtis integration weights, for interval [a,b]
130 static Weights DoubleIntegrationWeights(size_t N, double a, double b);
131
132 /// Create matrix of values at Chebyshev points given vector-valued function.
133 static Vector vector(std::function<double(double)> f,
134 size_t N, double a = -1, double b = 1);
135
136 /// Create matrix of values at Chebyshev points given vector-valued function.
137 template <size_t M>
138 static Matrix matrix(std::function<Eigen::Matrix<double, M, 1>(double)> f,
139 size_t N, double a = -1, double b = 1) {
140 Matrix Xmat(M, N);
141 const Vector points = Points(N, a, b);
142 for (size_t j = 0; j < N; j++) Xmat.col(i: j) = f(points(j));
143 return Xmat;
144 }
145}; // \ Chebyshev2
146
147} // namespace gtsam
