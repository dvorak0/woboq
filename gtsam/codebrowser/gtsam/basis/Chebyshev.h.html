26/**
27 * Basis of Chebyshev polynomials of the first kind
28 * https://en.wikipedia.org/wiki/Chebyshev_polynomials#First_kind
29 * These are typically denoted with the symbol T_n, where n is the degree.
30 * The parameter N is the number of coefficients, i.e., N = n+1.
31 */
32struct GTSAM_EXPORT Chebyshev1Basis : Basis<Chebyshev1Basis> {
33 using Parameters = Eigen::Matrix<double, -1, 1 /*Nx1*/>;
34
35 Parameters parameters_;
36
37 /**
38 * @brief Evaluate Chebyshev Weights on [-1,1] at x up to order N-1 (N values)
39 *
40 * @param N Degree of the polynomial.
41 * @param x Point to evaluate polynomial at.
42 * @param a Lower limit of polynomial (default=-1).
43 * @param b Upper limit of polynomial (default=1).
44 */
45 static Weights CalculateWeights(size_t N, double x, double a = -1,
46 double b = 1);
47
48 /**
49 * @brief Evaluate Chebyshev derivative at x.
50 * The derivative weights are pre-multiplied to the polynomial Parameters.
51 *
52 * From Wikipedia we have D[T_n(x),x] = n*U_{n-1}(x)
53 * I.e. the derivative fo a first kind cheb is just a second kind cheb
54 * So, we define a second kind basis here of order N-1
55 * Note that it has one less weight.
56 *
57 * The Parameters pertain to 1st kind chebs up to order N-1
58 * But of course the first one (order 0) is constant, so omit that weight.
59 *
60 * @param N Degree of the polynomial.
61 * @param x Point to evaluate polynomial at.
62 * @param a Lower limit of polynomial (default=-1).
63 * @param b Upper limit of polynomial (default=1).
64 * @return Weights
65 */
66 static Weights DerivativeWeights(size_t N, double x, double a = -1,
67 double b = 1);
68}; // Chebyshev1Basis
69
70/**
71 * Basis of Chebyshev polynomials of the second kind.
72 * https://en.wikipedia.org/wiki/Chebyshev_polynomials#Second_kind
73 * These are typically denoted with the symbol U_n, where n is the degree.
74 * The parameter N is the number of coefficients, i.e., N = n+1.
75 * In contrast to the templates in Chebyshev2, the classes below specify
76 * basis functions, weighted combinations of which are used to approximate
77 * functions. In this sense, they are like the sines and cosines of the Fourier
78 * basis.
79 */
80struct GTSAM_EXPORT Chebyshev2Basis : Basis<Chebyshev2Basis> {
81 using Parameters = Eigen::Matrix<double, -1, 1 /*Nx1*/>;
82
83 /**
84 * Evaluate Chebyshev Weights on [-1,1] at any x up to order N-1 (N values).
85 *
86 * @param N Degree of the polynomial.
87 * @param x Point to evaluate polynomial at.
88 * @param a Lower limit of polynomial (default=-1).
89 * @param b Upper limit of polynomial (default=1).
90 */
91 static Weights CalculateWeights(size_t N, double x, double a = -1,
92 double b = 1);
93
94 /**
95 * @brief Evaluate Chebyshev derivative at x.
96 *
97 * @param N Degree of the polynomial.
98 * @param x Point to evaluate polynomial at.
99 * @param a Lower limit of polynomial (default=-1).
100 * @param b Upper limit of polynomial (default=1).
101 * @return Weights
102 */
103 static Weights DerivativeWeights(size_t N, double x, double a = -1,
104 double b = 1);
105}; // Chebyshev2Basis
