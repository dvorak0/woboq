20#pragma once
21
22#include <gtsam/basis/Basis.h>
23
24namespace gtsam {
25
26/// Fourier basis
27class FourierBasis : public Basis<FourierBasis> {
28 public:
29 using Parameters = Eigen::Matrix<double, /*Nx1*/ -1, 1>;
30 using DiffMatrix = Eigen::Matrix<double, /*NxN*/ -1, -1>;
31
32 /**
33 * @brief Evaluate Real Fourier Weights of size N in interval [a, b],
34 * e.g. N=5 yields bases: 1, cos(x), sin(x), cos(2*x), sin(2*x)
35 *
36 * @param N The degree of the polynomial to use.
37 * @param x The point at which to compute the derivaive weights.
38 * @return Weights
39 */
40 static Weights CalculateWeights(size_t N, double x) {
41 Weights b(N);
42 b[0] = 1;
43 for (size_t i = 1, n = 1; i < N; i++) {
44 if (i % 2 == 1) {
45 b[i] = cos(x: n * x);
46 } else {
47 b[i] = sin(x: n * x);
48 n++;
49 }
50 }
51 return b;
52 }
53
54 /**
55 * @brief Evaluate Real Fourier Weights of size N in interval [a, b],
56 * e.g. N=5 yields bases: 1, cos(x), sin(x), cos(2*x), sin(2*x)
57 *
58 * @param N The degree of the polynomial to use.
59 * @param x The point at which to compute the weights.
60 * @param a Lower bound of interval.
61 * @param b Upper bound of interval.
62 * @return Weights
63 */
64 static Weights CalculateWeights(size_t N, double x, double a, double b) {
65 // TODO(Varun) How do we enforce an interval for Fourier series?
66 return CalculateWeights(N, x);
67 }
68
69 /**
70 * Compute D = differentiation matrix.
71 * Given coefficients c of a Fourier series c, D*c are the values of c'.
72 */
73 static DiffMatrix DifferentiationMatrix(size_t N) {
74 DiffMatrix D = DiffMatrix::Zero(rows: N, cols: N);
75 double k = 1;
76 for (size_t i = 1; i < N; i += 2) {
77 D(i, i + 1) = k; // sin'(k*x) = k*cos(k*x)
78 D(i + 1, i) = -k; // cos'(k*x) = -k*sin(k*x)
79 k += 1;
80 }
81
82 return D;
83 }
84
85 /**
86 * @brief Get weights at a given x that calculate the derivative.
87 *
88 * @param N The degree of the polynomial to use.
89 * @param x The point at which to compute the derivaive weights.
90 * @return Weights
91 */
92 static Weights DerivativeWeights(size_t N, double x) {
93 return CalculateWeights(N, x) * DifferentiationMatrix(N);
94 }
95
96 /**
97 * @brief Get derivative weights at a given x that calculate the derivative,
98 in the interval [a, b].
99 *
100 * @param N The degree of the polynomial to use.
101 * @param x The point at which to compute the derivaive weights.
102 * @param a Lower bound of interval.
103 * @param b Upper bound of interval.
104 * @return Weights
105 */
106 static Weights DerivativeWeights(size_t N, double x, double a, double b) {
107 return CalculateWeights(N, x, a, b) * DifferentiationMatrix(N);
108 }
109
110}; // FourierBasis
111
112} // namespace gtsam
