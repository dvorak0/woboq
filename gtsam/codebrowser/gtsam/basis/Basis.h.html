26/**
27 * This file supports creating continuous functions `f(x;p)` as a linear
28 * combination of `basis functions` such as the Fourier basis on SO(2) or a set
29 * of Chebyshev polynomials on [-1,1].
30 *
31 * In the expression `f(x;p)` the variable `x` is
32 * the continuous argument at which the function is evaluated, and `p` are
33 * the parameters which are coefficients of the different basis functions,
34 * e.g. p = [4; 3; 2] => 4 + 3x + 2x^2 for a polynomial.
35 * However, different parameterizations are also possible.
36
37 * The `Basis` class below defines a number of functors that can be used to
38 * evaluate `f(x;p)` at a given `x`, and these functors also calculate
39 * the Jacobian of `f(x;p)` with respect to the parameters `p`.
40 * This is actually the most important calculation, as it will allow GTSAM
41 * to optimize over the parameters `p`.
42
43 * This functionality is implemented using the `CRTP` or "Curiously recurring
44 * template pattern" C++ idiom, which is a meta-programming technique in which
45 * the derived class is passed as a template argument to `Basis<DERIVED>`.
46 * The DERIVED class is assumed to satisfy a C++ concept,
47 * i.e., we expect it to define the following types and methods:
48
49 - type `Parameters`: the parameters `p` in f(x;p)
50 - `CalculateWeights(size_t N, double x, double a=default, double b=default)`
51 - `DerivativeWeights(size_t N, double x, double a=default, double b=default)`
52
53 where `Weights` is an N*1 row vector which defines the basis values for the
54 polynomial at the specified point `x`.
55
56 E.g. A Fourier series would give the following:
57 - `CalculateWeights` -> For N=5, the values for the bases:
58 [1, cos(x), sin(x), cos(2x), sin(2x)]
59 - `DerivativeWeights` -> For N=5, these are:
60 [0, -sin(x), cos(x), -2sin(2x), 2cos(x)]
61
62 Note that for a pseudo-spectral basis (as in Chebyshev2), the weights are
63 instead the values for the Barycentric interpolation formula, since the values
64 at the polynomial points (e.g. Chebyshev points) define the bases.
65 */
66

69using Weights = Eigen::Matrix<double, 1, -1>; /* 1xN vector */

71/**
72 * @brief Function for computing the kronecker product of the 1*N Weight vector
73 * `w` with the MxM identity matrix `I` efficiently.
74 * The main reason for this is so we don't need to use Eigen's Unsupported
75 * library.
76 *
77 * @tparam M Size of the identity matrix.
78 * @param w The weights of the polynomial.
79 * @return Mx(M*N) kronecker product [w(0)*I, w(1)*I, ..., w(N-1)*I]
80 *
81 * @ingroup basis
82 */
83Matrix GTSAM_EXPORT kroneckerProductIdentity(size_t M, const Weights& w);
84
85/**
86 * CRTP Base class for function bases
87 * @ingroup basis
88 */
89template <typename DERIVED>
90class Basis {
91 public:
92 /**
93 * Calculate weights for all x in vector X.
94 * Returns M*N matrix where M is the size of the vector X,
95 * and N is the number of basis functions.
96 */
97 static Matrix WeightMatrix(size_t N, const Vector& X) {
98 Matrix W(X.size(), N);
99 for (int i = 0; i < X.size(); i++)
100 W.row(i) = DERIVED::CalculateWeights(N, X(i));
101 return W;
102 }
103
104 /**
105 * @brief Calculate weights for all x in vector X, with interval [a,b].
106 *
107 * @param N The number of basis functions.
108 * @param X The vector for which to compute the weights.
109 * @param a The lower bound for the interval range.
110 * @param b The upper bound for the interval range.
111 * @return Returns M*N matrix where M is the size of the vector X.
112 */
113 static Matrix WeightMatrix(size_t N, const Vector& X, double a, double b) {
114 Matrix W(X.size(), N);
115 for (int i = 0; i < X.size(); i++)
116 W.row(i) = DERIVED::CalculateWeights(N, X(i), a, b);
117 return W;
118 }

120 /**
121 * An instance of an EvaluationFunctor calculates f(x;p) at a given `x`,
122 * applied to Parameters `p`.
123 * This functor is used to evaluate a parameterized function at a given scalar
124 * value x. When given a specific N*1 vector of Parameters, returns the scalar
125 * value of the function at x, possibly with Jacobians wrpt the parameters.
126 */
127 class EvaluationFunctor {
128 protected:
129 Weights weights_;
130
131 public:
132 /// For serialization
133 EvaluationFunctor() {}
134
135 /// Constructor with interval [a,b]
136 EvaluationFunctor(size_t N, double x)
137 : weights_(DERIVED::CalculateWeights(N, x)) {}
138
139 /// Constructor with interval [a,b]
140 EvaluationFunctor(size_t N, double x, double a, double b)
141 : weights_(DERIVED::CalculateWeights(N, x, a, b)) {}
142
143 /// Regular 1D evaluation
144 double apply(const typename DERIVED::Parameters& p,
145 OptionalJacobian<-1, -1> H = {}) const {
146 if (H) *H = weights_;
147 return (weights_ * p)(0);
148 }
149
150 /// c++ sugar
151 double operator()(const typename DERIVED::Parameters& p,
152 OptionalJacobian<-1, -1> H = {}) const {
153 return apply(p, H); // might call apply in derived
154 }
155
156 void print(const std::string& s = "") const {
157 std::cout << s << (s != "" ? " " : "") << weights_ << std::endl;
158 }
159 };

161 /**
162 * VectorEvaluationFunctor at a given x, applied to a parameter Matrix.
163 * This functor is used to evaluate a parameterized function at a given scalar
164 * value x. When given a specific M*N parameters, returns an M-vector the M
165 * corresponding functions at x, possibly with Jacobians wrpt the parameters.
166 */
167 class VectorEvaluationFunctor : protected EvaluationFunctor {
168 protected:
169 using Jacobian = Eigen::Matrix<double, /*MxMN*/ -1, -1>;
170 Jacobian H_;
171
172 size_t M_;
173
174 /**
175 * Calculate the `M*(M*N)` Jacobian of this functor with respect to
176 * the M*N parameter matrix `P`.
177 * We flatten assuming column-major order, e.g., if N=3 and M=2, we have
178 * H =[ w(0) 0 w(1) 0 w(2) 0
179 * 0 w(0) 0 w(1) 0 w(2) ]
180 * i.e., the Kronecker product of weights_ with the MxM identity matrix.
181 */
182 void calculateJacobian() {
183 H_ = kroneckerProductIdentity(M_, this->weights_);
184 }
185
186 public:
187 EIGEN_MAKE_ALIGNED_OPERATOR_NEW
188
189 /// For serialization
190 VectorEvaluationFunctor() {}
191
192 /// Default Constructor
193 VectorEvaluationFunctor(size_t M, size_t N, double x)
194 : EvaluationFunctor(N, x), M_(M) {
195 calculateJacobian();
196 }
197
198 /// Constructor, with interval [a,b]
199 VectorEvaluationFunctor(size_t M, size_t N, double x, double a, double b)
200 : EvaluationFunctor(N, x, a, b), M_(M) {
201 calculateJacobian();
202 }
203
204 /// M-dimensional evaluation
205 Vector apply(const Matrix& P,
206 OptionalJacobian</*MxN*/ -1, -1> H = {}) const {
207 if (H) *H = H_;
208 return P.matrix() * this->weights_.transpose();
209 }
210
211 /// c++ sugar
212 Vector operator()(const Matrix& P,
213 OptionalJacobian</*MxN*/ -1, -1> H = {}) const {
214 return apply(P, H);
215 }
216 };

218 /**
219 * Given a M*N Matrix of M-vectors at N polynomial points, an instance of
220 * VectorComponentFunctor computes the N-vector value for a specific row
221 * component of the M-vectors at all the polynomial points.
222 *
223 * This component is specified by the row index i, with 0<i<M.
224 */
225 class VectorComponentFunctor : public EvaluationFunctor {
226 protected:
227 using Jacobian = Eigen::Matrix<double, /*1xMN*/ 1, -1>;
228 Jacobian H_;
229
230 size_t M_;
231 size_t rowIndex_;
232
233 /*
234 * Calculate the `1*(M*N)` Jacobian of this functor with respect to
235 * the M*N parameter matrix `P`.
236 * We flatten assuming column-major order, e.g., if N=3 and M=2, we have
237 * H=[w(0) 0 w(1) 0 w(2) 0] for rowIndex==0
238 * H=[0 w(0) 0 w(1) 0 w(2)] for rowIndex==1
239 * i.e., one row of the Kronecker product of weights_ with the
240 * MxM identity matrix. See also VectorEvaluationFunctor.
241 */
242 void calculateJacobian() {
243 H_.setZero(1, M_ * EvaluationFunctor::weights_.size());
244 for (int j = 0; j < EvaluationFunctor::weights_.size(); j++)
245 H_(0, rowIndex_ + j * M_) = EvaluationFunctor::weights_(j);
246 }
247
248 public:
249 /// For serialization
250 VectorComponentFunctor() {}
251
252 /// Construct with row index
253 VectorComponentFunctor(size_t M, size_t N, size_t i, double x)
254 : EvaluationFunctor(N, x), M_(M), rowIndex_(i) {
255 calculateJacobian();
256 }
257
258 /// Construct with row index and interval
259 VectorComponentFunctor(size_t M, size_t N, size_t i, double x, double a,
260 double b)
261 : EvaluationFunctor(N, x, a, b), M_(M), rowIndex_(i) {
262 calculateJacobian();
263 }
264
265 /// Calculate component of component rowIndex_ of P
266 double apply(const Matrix& P,
267 OptionalJacobian</*1xMN*/ -1, -1> H = {}) const {
268 if (H) *H = H_;
269 return P.row(i: rowIndex_) * EvaluationFunctor::weights_.transpose();
270 }
271
272 /// c++ sugar
273 double operator()(const Matrix& P,
274 OptionalJacobian</*1xMN*/ -1, -1> H = {}) const {
275 return apply(P, H);
276 }
277 };

279 /**
280 * Manifold EvaluationFunctor at a given x, applied to a parameter Matrix.
281 * This functor is used to evaluate a parameterized function at a given scalar
282 * value x. When given a specific M*N parameters, returns an M-vector the M
283 * corresponding functions at x, possibly with Jacobians wrpt the parameters.
284 *
285 * The difference with the VectorEvaluationFunctor is that after computing the
286 * M*1 vector xi=F(x;P), with x a scalar and P the M*N parameter vector, we
287 * also retract xi back to the T manifold.
288 * For example, if T==Rot3, then we first compute a 3-vector xi using x and P,
289 * and then map that 3-vector xi back to the Rot3 manifold, yielding a valid
290 * 3D rotation.
291 */
292 template <class T>
293 class ManifoldEvaluationFunctor : public VectorEvaluationFunctor {
294 inline constexpr static auto M = traits<T>::dimension;
295 using Base = VectorEvaluationFunctor;
296
297 public:
298 /// For serialization
299 ManifoldEvaluationFunctor() {}
300
301 /// Default Constructor
302 ManifoldEvaluationFunctor(size_t N, double x) : Base(M, N, x) {}
303
304 /// Constructor, with interval [a,b]
305 ManifoldEvaluationFunctor(size_t N, double x, double a, double b)
306 : Base(M, N, x, a, b) {}
307
308 /// Manifold evaluation
309 T apply(const Matrix& P, OptionalJacobian</*MxMN*/ -1, -1> H = {}) const {
310 // Interpolate the M-dimensional vector to yield a vector in tangent space
311 Eigen::Matrix<double, M, 1> xi = Base::operator()(P, H);
312
313 // Now call retract with this M-vector, possibly with derivatives
314 Eigen::Matrix<double, M, M> D_result_xi;
315 T result = T::ChartAtOrigin::Retract(xi, H ? &D_result_xi : 0);
316
317 // Finally, if derivatives are asked, apply chain rule where H is Mx(M*N)
318 // derivative of interpolation and D_result_xi is MxM derivative of
319 // retract.
320 if (H) *H = D_result_xi * (*H);
321
322 // and return a T
323 return result;
324 }
325
326 /// c++ sugar
327 T operator()(const Matrix& P,
328 OptionalJacobian</*MxN*/ -1, -1> H = {}) const {
329 return apply(P, H); // might call apply in derived
330 }
331 };

353 /**
354 * An instance of a DerivativeFunctor calculates f'(x;p) at a given `x`,
355 * applied to Parameters `p`.

357 * this functor returns the scalar derivative value of the function at x,
358 * possibly with Jacobians wrpt the parameters.
359 */
360 class DerivativeFunctor : protected DerivativeFunctorBase {
361 public:
362 /// For serialization
363 DerivativeFunctor() {}

365 DerivativeFunctor(size_t N, double x) : DerivativeFunctorBase(N, x) {}
366
367 DerivativeFunctor(size_t N, double x, double a, double b)
368 : DerivativeFunctorBase(N, x, a, b) {}
369
370 double apply(const typename DERIVED::Parameters& p,
371 OptionalJacobian</*1xN*/ -1, -1> H = {}) const {
372 if (H) *H = this->weights_;

459 * H=[0 w(0) 0 w(1) 0 w(2)] for rowIndex==1
