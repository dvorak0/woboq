38// Coefficients for testing, respectively 3 and 7 parameter Fourier basis.
39// They correspond to best approximation of TestFunction(x)
40const Vector k3Coefficients = (Vector3() << 1.5661, 1.2717, 1.2717).finished();
41const Vector7 k7Coefficients =
42 (Vector7() << 1.5661, 1.2717, 1.2717, -0.0000, 0.5887, -0.0943, 0.0943)

45// The test-function used below
46static double TestFunction(double x) { return exp(x: sin(x: x) + cos(x: x)); }

70TEST(Basis, Manual) {
71 const size_t N = 3;
72
73 // We will create a linear factor graph
74 GaussianFactorGraph graph;
75
76 // We create an unknown Vector expression for the coefficients
77 Key key(1);
78
79 // We will need values below to test the Jacobians
80 Values values;
81 values.insert<Vector>(j: key, val: Vector::Zero(size: N)); // value does not really matter
82
83 // At 16 different samples points x, check Predict_ expression
84 for (size_t i = 0; i < 16; i++) {
85 const double x = i * M_PI / 8;
86 const double desiredValue = TestFunction(x);
87
88 // Manual JacobianFactor
89 Matrix A(1, N);
90 A << 1, cos(x: x), sin(x: x);
91 Vector b(1);
92 b << desiredValue;
93 JacobianFactor linearFactor(key, A, b);
94 graph.add(factor: linearFactor);
95
96 // Create factor to predict value at x
97 EvaluationFactor<FourierBasis> predictFactor(key, desiredValue, model, N,
98 x);
99
100 // Check expression Jacobians
101 EXPECT_CORRECT_FACTOR_JACOBIANS(predictFactor, values, 1e-5, 1e-9);
102
103 auto linearizedFactor = predictFactor.linearize(x: values);
104 auto linearizedJacobianFactor =
105 std::dynamic_pointer_cast<JacobianFactor>(r: linearizedFactor);
106 CHECK(linearizedJacobianFactor); // makes sure it's indeed a JacobianFactor
107 EXPECT(assert_equal(linearFactor, *linearizedJacobianFactor, 1e-9));
108 }
109
110 // Solve linear graph
111 VectorValues actual = graph.optimize();
112 EXPECT(assert_equal((Vector)k3Coefficients, actual.at(key), 1e-4));
113}

204//******************************************************************************
205// Suppose we want to parameterize a periodic function with function values at
206// specific times, rather than coefficients. Can we do it? This would be a
207// generalization of the Fourier transform, and constitute a "pseudo-spectral"
208// parameterization. One way to do this is to establish hard constraints that
209// express the relationship between the new parameters and the coefficients.
210// For example, below I'd like the parameters to be the function values at
211// X = {0.1,0.2,0.3}, rather than a 3-vector of coefficients.
212// Because the values f(X) = at these points can be written as f(X) = W(X)*c,
213// we can simply express the coefficents c as c=inv(W(X))*f, which is a
214// generalized Fourier transform. That also means we can create factors with the
215// unknown f-values, as done manually below.
216TEST(Basis, PseudoSpectral) {
217 // We will create a linear factor graph
218 GaussianFactorGraph graph;
219
220 const size_t N = 3;
221 const Key key(1);
222
223 // The correct values at X = {0.1,0.2,0.3} are simply W*c
224 const Vector X = (Vector3() << 0.1, 0.2, 0.3).finished();
225 const Matrix W = FourierBasis::WeightMatrix(N, X);
226 const Vector expected = W * k3Coefficients;
227
228 // Check those values are indeed correct values of Fourier approximation
229 using Eval = FourierBasis::EvaluationFunctor;
230 EXPECT_DOUBLES_EQUAL(Eval(N, 0.1)(k3Coefficients), expected(0), 1e-9);
231 EXPECT_DOUBLES_EQUAL(Eval(N, 0.2)(k3Coefficients), expected(1), 1e-9);
232 EXPECT_DOUBLES_EQUAL(Eval(N, 0.3)(k3Coefficients), expected(2), 1e-9);
233
234 // Calculate "inverse Fourier transform" matrix
235 const Matrix invW = W.inverse();
236
237 // At 16 different samples points x, add a factor on fExpr
238 for (size_t i = 0; i < 16; i++) {
239 const double x = i * M_PI / 8;
240 const double desiredValue = TestFunction(x);
241
242 // Manual JacobianFactor
243 Matrix A(1, 3);
244 A << 1, cos(x: x), sin(x: x);
245 Vector b(1);
246 b << desiredValue;
247 JacobianFactor linearFactor(key, A * invW, b);
248 graph.add(factor: linearFactor);
249 }
250
251 // Solve linear graph
252 VectorValues actual = graph.optimize();
253 EXPECT(assert_equal((Vector)expected, actual.at(key), 1e-4));
254}
