26/**
27 * @brief Factor for enforcing the scalar value of the polynomial BASIS
28 * representation at `x` is the same as the measurement `z` when using a
29 * pseudo-spectral parameterization.
30 *
31 * @tparam BASIS The basis class to use e.g. Chebyshev2
32 *
33 * Example, degree 8 Chebyshev polynomial measured at x=0.5:
34 * EvaluationFactor<Chebyshev2> factor(key, measured, model, 8, 0.5);
35 *
36 * @ingroup basis
37 */
38template <class BASIS>
39class EvaluationFactor : public FunctorizedFactor<double, Vector> {
40 private:
41 using Base = FunctorizedFactor<double, Vector>;
42
43 public:
44 EvaluationFactor() {}
45
46 /**
47 * @brief Construct a new EvaluationFactor object
48 *
49 * @param key Symbol for value to optimize.
50 * @param z The measurement value.
51 * @param model Noise model
52 * @param N The degree of the polynomial.
53 * @param x The point at which to evaluate the polynomial.
54 */
55 EvaluationFactor(Key key, double z, const SharedNoiseModel &model,
56 const size_t N, double x)
57 : Base(key, z, model, typename BASIS::EvaluationFunctor(N, x)) {}
58
59 /**
60 * @brief Construct a new EvaluationFactor object
61 *
62 * @param key Symbol for value to optimize.
63 * @param z The measurement value.
64 * @param model Noise model
65 * @param N The degree of the polynomial.
66 * @param x The point at which to evaluate the polynomial.
67 * @param a Lower bound for the polynomial.
68 * @param b Upper bound for the polynomial.
69 */
70 EvaluationFactor(Key key, double z, const SharedNoiseModel &model,
71 const size_t N, double x, double a, double b)
72 : Base(key, z, model, typename BASIS::EvaluationFunctor(N, x, a, b)) {}
73
74 virtual ~EvaluationFactor() {}
75};

77/**
78 * Unary factor for enforcing BASIS polynomial evaluation on a parameter Matrix
79 * of size (M, N) is equal to a vector-valued measurement at the same point,
80 when
81 * using a pseudo-spectral parameterization.
82 *
83 * This factors tries to enforce the basis function evaluation `f(x;p)` to take
84 * on the value `z` at location `x`, providing a gradient on the parameters p.
85 * In a probabilistic estimation context, `z` is known as a measurement, and the
86 * parameterized basis function can be seen as a
87 * measurement prediction function.
88 *
89 * @param BASIS: The basis class to use e.g. Chebyshev2
90 *
91 * @ingroup basis
92 */
93template <class BASIS>
94class VectorEvaluationFactor : public FunctorizedFactor<Vector, Matrix> {
95 private:
96 using Base = FunctorizedFactor<Vector, Matrix>;
97
98 public:
99 VectorEvaluationFactor() {}
100
101 /**
102 * @brief Construct a new VectorEvaluationFactor object.
103 *
104 * @param key The key to the parameter Matrix object used to represent the
105 * polynomial.
106 * @param z The measurement value.
107 * @param model The noise model.
108 * @param M Size of the evaluated state vector.
109 * @param N The degree of the polynomial.
110 * @param x The point at which to evaluate the basis polynomial.
111 */
112 VectorEvaluationFactor(Key key, const Vector &z,
113 const SharedNoiseModel &model, const size_t M,
114 const size_t N, double x)
115 : Base(key, z, model, typename BASIS::VectorEvaluationFunctor(M, N, x)) {}
116
117 /**
118 * @brief Construct a new VectorEvaluationFactor object.
119 *
120 * @param key The key to the parameter Matrix object used to represent the
121 * polynomial.
122 * @param z The measurement value.
123 * @param model The noise model.
124 * @param M Size of the evaluated state vector.
125 * @param N The degree of the polynomial.
126 * @param x The point at which to evaluate the basis polynomial.
127 * @param a Lower bound for the polynomial.
128 * @param b Upper bound for the polynomial.
129 */
130 VectorEvaluationFactor(Key key, const Vector &z,
131 const SharedNoiseModel &model, const size_t M,
132 const size_t N, double x, double a, double b)
133 : Base(key, z, model,
134 typename BASIS::VectorEvaluationFunctor(M, N, x, a, b)) {}
135
136 virtual ~VectorEvaluationFactor() {}
137};

139/**
140 * Unary factor for enforcing BASIS polynomial evaluation on a parameter Matrix
141 * of size (P, N) is equal to specified measurement at the same point, when
142 * using a pseudo-spectral parameterization.
143 *
144 * This factor is similar to `VectorEvaluationFactor` with the key difference
145 * being that it only enforces the constraint for a single scalar in the vector,
146 * indexed by `i`.
147 *
148 * @param BASIS: The basis class to use e.g. Chebyshev2
149 *
150 * Example:
151 * VectorComponentFactor<BASIS> controlPrior(key, measured, model,
152 * N, i, t, a, b);
153 * where N is the degree and i is the component index.
154 *
155 * @ingroup basis
156 */
157template <class BASIS>
158class VectorComponentFactor : public FunctorizedFactor<double, Matrix> {
159 private:
160 using Base = FunctorizedFactor<double, Matrix>;
161
162 public:
163 VectorComponentFactor() {}
164
165 /**
166 * @brief Construct a new VectorComponentFactor object.
167 *
168 * @param key The key to the parameter Matrix object used to represent the
169 * polynomial.
170 * @param z The scalar value at a specified index `i` of the full measurement
171 * vector.
172 * @param model The noise model.
173 * @param P Size of the fixed-size vector.
174 * @param N The degree of the polynomial.
175 * @param i The index for the evaluated vector to give us the desired scalar
176 * value.
177 * @param x The point at which to evaluate the basis polynomial.
178 */
179 VectorComponentFactor(Key key, const double &z, const SharedNoiseModel &model,
180 const size_t P, const size_t N, size_t i, double x)
181 : Base(key, z, model,
182 typename BASIS::VectorComponentFunctor(P, N, i, x)) {}
183
184 /**
185 * @brief Construct a new VectorComponentFactor object.
186 *
187 * @param key The key to the parameter Matrix object used to represent the
188 * polynomial.
189 * @param z The scalar value at a specified index `i` of the full measurement
190 * vector.
191 * @param model The noise model.
192 * @param P Size of the fixed-size vector.
193 * @param N The degree of the polynomial.
194 * @param i The index for the evaluated vector to give us the desired scalar
195 * value.
196 * @param x The point at which to evaluate 0the basis polynomial.
197 * @param a Lower bound for the polynomial.
198 * @param b Upper bound for the polynomial.
199 */
200 VectorComponentFactor(Key key, const double &z, const SharedNoiseModel &model,
201 const size_t P, const size_t N, size_t i, double x,
202 double a, double b)
203 : Base(key, z, model,
204 typename BASIS::VectorComponentFunctor(P, N, i, x, a, b)) {}
205
206 virtual ~VectorComponentFactor() {}
207};

209/**
210 * For a measurement value of type T i.e. `T z = g(x)`, this unary
211 * factor enforces that the polynomial basis, when evaluated at `x`, gives us
212 * the same `z`, i.e. `T z = g(x) = f(x;p)`.
213 *
214 * This is done via computations on the tangent space of the
215 * manifold of T.
216 *
217 * @param BASIS: The basis class to use e.g. Chebyshev2
218 * @param T: Object type which is synthesized by the provided functor.
219 *
220 * Example:
221 * ManifoldEvaluationFactor<Chebyshev2, Rot3> rotationFactor(key, measurement,
222 * model, N, x, a, b);
223 *
224 * where `x` is the value (e.g. timestep) at which the rotation was evaluated.
225 */
226template <class BASIS, typename T>
227class ManifoldEvaluationFactor : public FunctorizedFactor<T, Matrix> {
228 private:
229 using Base = FunctorizedFactor<T, Matrix>;
230
231 public:
232 ManifoldEvaluationFactor() {}
233
234 /**
235 * @brief Construct a new ManifoldEvaluationFactor object.
236 *
237 * @param key Key for the state matrix parameterizing the function to estimate
238 * via the BASIS.
239 * @param z The measurement value.
240 * @param model The noise model.
241 * @param N The degree of the polynomial.
242 * @param x The point at which the estimated function is evaluated.
243 */
244 ManifoldEvaluationFactor(Key key, const T &z, const SharedNoiseModel &model,
245 const size_t N, double x)
246 : Base(key, z, model,
247 typename BASIS::template ManifoldEvaluationFunctor<T>(N, x)) {}
248
249 /**
250 * @brief Construct a new ManifoldEvaluationFactor object.
251 *
252 * @param key Key for the state matrix parameterizing the function to estimate
253 * via the BASIS.
254 * @param z The measurement value.
255 * @param model The noise model.
256 * @param N The degree of the polynomial.
257 * @param x The point at which the estimated function is evaluated.
258 * @param a Lower bound for the polynomial.
259 * @param b Upper bound for the polynomial.
260 */
261 ManifoldEvaluationFactor(Key key, const T &z, const SharedNoiseModel &model,
262 const size_t N, double x, double a, double b)
263 : Base(
264 key, z, model,
265 typename BASIS::template ManifoldEvaluationFunctor<T>(N, x, a, b)) {
266 }
267
268 virtual ~ManifoldEvaluationFactor() {}
269};
