33namespace gtsam {
34
35/// Our sequence representation is a map of {x: y} values where y = f(x)
36using Sequence = std::map<double, double>;
37/// A sample is a key-value pair from a sequence.
38using Sample = std::pair<double, double>;
39
40/**
41 * Class that does regression via least squares
42 * Example usage:
43 * size_t N = 3;
44 * auto fit = FitBasis<Chebyshev2>(data_points, noise_model, N);
45 * Vector coefficients = fit.parameters();
46 *
47 * where `data_points` are a map from `x` to `y` values indicating a function
48 * mapping at specific points, `noise_model` is the gaussian noise model, and
49 * `N` is the degree of the polynomial basis used to fit the function.
50 */
51template <class Basis>
52class FitBasis {
53 public:
54 using Parameters = typename Basis::Parameters;
55
56 private:
57 Parameters parameters_;
58
59 public:
60 /// Create nonlinear FG from Sequence
61 static NonlinearFactorGraph NonlinearGraph(const Sequence& sequence,
62 const SharedNoiseModel& model,
63 size_t N) {
64 NonlinearFactorGraph graph;
65 for (const Sample sample : sequence) {
66 graph.emplace_shared<EvaluationFactor<Basis>>(0, sample.second, model, N,
67 sample.first);
68 }
69 return graph;
70 }
71
72 /// Create linear FG from Sequence
73 static GaussianFactorGraph::shared_ptr LinearGraph(
74 const Sequence& sequence, const SharedNoiseModel& model, size_t N) {
75 NonlinearFactorGraph graph = NonlinearGraph(sequence, model, N);
76 Values values;
77 values.insert<Parameters>(0, Parameters::Zero(N));
78 GaussianFactorGraph::shared_ptr gfg = graph.linearize(linearizationPoint: values);
79 return gfg;
80 }
81
82 /**
83 * @brief Construct a new FitBasis object.
84 *
85 * @param sequence map of x->y values for a function, a.k.a. y = f(x).
86 * @param model The noise model to use.
87 * @param N The degree of the polynomial to fit.
88 */
89 FitBasis(const Sequence& sequence, const SharedNoiseModel& model, size_t N) {
90 GaussianFactorGraph::shared_ptr gfg = LinearGraph(sequence, model, N);
91 VectorValues solution = gfg->optimize();
92 parameters_ = solution.at(j: 0);
93 }
94
95 /// Return Fourier coefficients
96 Parameters parameters() const { return parameters_; }
97};

99} // namespace gtsam
