26/* ************************************************************************* */
27PenaltyOptimizer::State PenaltyOptimizer::iterate(const State& state) const {
28 State newState(state.iteration + 1);
29
30 // Update penalty parameter.
31 newState.muEq = state.iteration == 0 ? p_->initialMuEq
32 : state.muEq * p_->muEqIncreaseRate;
33 newState.muIneq = state.iteration == 0
34 ? p_->initialMuIneq
35 : state.muIneq * p_->muIneqIncreaseRate;
36
37 // Construct merit function.
38 NonlinearFactorGraph meritGraph =
39 meritFunction(muEq: newState.muEq, muIneq: newState.muIneq);
40
41 // Run unconstrained optimization.
42 auto optimizer = createUnconstrainedOptimizer(graph: meritGraph, values: state.values);
43 newState.setValues(values: optimizer->optimize(), problem: problem_);
44 newState.unconstrainedIterationss = optimizer->iterations();
45
46 return newState;
47}
48
49/* ************************************************************************* */
50Values PenaltyOptimizer::optimize() const {
51 /// Construct initial state
52 State previousState;
53 State state(0, initialValues_, problem_);
54 logInitialState(state);
55
56 do {
57 previousState = std::move(state);
58 state = iterate(state: previousState);
59 logIteration(state);
60 } while (!checkConvergence(state, previousState, params: *p_));
61
62 return state.values;
63}
64
65/* ************************************************************************* */
66NonlinearFactorGraph PenaltyOptimizer::meritFunction(
67 const double muEq, const double muIneq) const {
68 NonlinearFactorGraph graph = problem_.costs();
69 graph.add(factorOrContainer: problem_.eConstraints().penaltyGraph(mu: muEq));
70 graph.add(factorOrContainer: problem_.iConstraints().penaltyGraphCustom(
71 func: p_->ineqConstraintPenaltyFunction, mu: muIneq));
72 return graph;
73}

75/* ************************************************************************* */
76PenaltyOptimizer::SharedOptimizer
77PenaltyOptimizer::createUnconstrainedOptimizer(
78 const NonlinearFactorGraph& graph, const Values& values) const {
79 // TODO(yetong): make compatible with all NonlinearOptimizers.
80 return std::make_shared<LevenbergMarquardtOptimizer>(args: graph, args: values,
81 args&: p_->lm_params);
82}

84/* ************************************************************************* */
85void PenaltyOptimizer::logInitialState(const State& state) const {
86 if (p_->verbose) {
87 // Log title line.
88 cout << setw(10) << "Iter"
89 << "|" << setw(10) << "muEq"
90 << "|" << setw(10) << "muIneq"
91 << "|" << setw(10) << "cost"
92 << "|" << setw(10) << "vio_e"
93 << "|" << setw(10) << "vio_i"
94 << "|" << setw(10) << "uopt_iters"
95 << "|" << setw(10) << "time"
96 << "|" << endl;
97
98 // Log initial value line.
99 cout << setw(10) << state.iteration;
100 cout << "|" << setw(10) << "-";
101 cout << "|" << setw(10) << "-";
102 cout << "|" << setw(10) << setprecision(4) << state.cost;
103 cout << "|" << setw(10) << setprecision(4) << state.eqConstraintViolation;
104 cout << "|" << setw(10) << setprecision(4) << state.ineqConstraintViolation;
105 cout << "|" << setw(10) << "-";
106 cout << "|" << setw(10) << "-";
107 cout << "|" << endl;
108 }
109
110 // Store state
111 if (p_->storeOptProgress) {
112 progress_.emplace_back(args: state);
113 }
114}

116/* ************************************************************************* */
117void PenaltyOptimizer::logIteration(const State& state) const {
118 if (p_->verbose) {
119 cout << setw(10) << state.iteration;
120 cout << "|" << setw(10) << state.muEq;
121 cout << "|" << setw(10) << state.muIneq;
122 cout << "|" << setw(10) << setprecision(4) << state.cost;
123 cout << "|" << setw(10) << setprecision(4) << state.eqConstraintViolation;
124 cout << "|" << setw(10) << setprecision(4) << state.ineqConstraintViolation;
125 cout << "|" << setw(10) << state.unconstrainedIterationss;
126 cout << "|" << setw(10) << state.time;
127 cout << "|" << endl;
128 }
129
130 // Store state
131 if (p_->storeOptProgress) {
132 progress_.emplace_back(args: state);
133 }
134}
