53/** Equality constraint that force g(x) = M. */
54template <typename T>
55class ExpressionEqualityConstraint : public NonlinearEqualityConstraint {
56 public:
57 using Base = NonlinearEqualityConstraint;
58 using This = ExpressionEqualityConstraint;
59 using shared_ptr = std::shared_ptr<This>;
60
61 protected:
62 Expression<T> expression_;
63 T rhs_;
64 FastVector<int> dims_;
65
66 public:
67 /**
68 * @brief Constructor.
69 *
70 * @param expression expression representing g(x).
71 * @param tolerance vector representing tolerance in each dimension.
72 */
73 ExpressionEqualityConstraint(const Expression<T>& expression, const T& rhs, const Vector& sigmas);
74
75 virtual Vector unwhitenedError(const Values& x, OptionalMatrixVecType H = nullptr) const override;
76
77 virtual NoiseModelFactor::shared_ptr penaltyFactor(const double mu = 1.0) const override;
78
79 const Expression<T>& expression() const { return expression_; }
80
81 /// @return a deep copy of this factor
82 gtsam::NonlinearFactor::shared_ptr clone() const override {
83 return std::static_pointer_cast<gtsam::NonlinearFactor>(
84 r: gtsam::NonlinearFactor::shared_ptr(new This(*this)));
85 }
86
87 private:
88#if GTSAM_ENABLE_BOOST_SERIALIZATION
89 /** Serialization function */
90 friend class boost::serialization::access;
91 template <class ARCHIVE>
92 void serialize(ARCHIVE& ar, const unsigned int /*version*/) {
93 ar& boost::serialization::make_nvp("ExpressionEqualityConstraint",
94 boost::serialization::base_object<Base>(*this));
95 ar& BOOST_SERIALIZATION_NVP(expression_);
96 ar& BOOST_SERIALIZATION_NVP(rhs_);
97 ar& BOOST_SERIALIZATION_NVP(dims_);
98 }
99#endif
100};

102/** Equality constraint that enforce the cost factor with zero error.
103 * e.g., for a factor with unwhitened cost 2x-1, the constraint enforces the
104 * equlity 2x-1=0.
105 */
106class GTSAM_EXPORT ZeroCostConstraint : public NonlinearEqualityConstraint {
107 public:
108 using Base = NonlinearEqualityConstraint;
109 using This = ZeroCostConstraint;
110 using shared_ptr = std::shared_ptr<This>;
111
112 protected:
113 NoiseModelFactor::shared_ptr factor_;
114
115 public:
116 /**
117 * @brief Constructor.
118 *
119 * @param factor NoiseModel factor.
120 */
121 ZeroCostConstraint(const NoiseModelFactor::shared_ptr& factor);
122
123 virtual Vector unwhitenedError(const Values& x, OptionalMatrixVecType H = nullptr) const override;
124
125 virtual NoiseModelFactor::shared_ptr penaltyFactor(const double mu = 1.0) const override;
126
127 /// @return a deep copy of this factor
128 gtsam::NonlinearFactor::shared_ptr clone() const override {
129 return std::static_pointer_cast<gtsam::NonlinearFactor>(
130 r: gtsam::NonlinearFactor::shared_ptr(new This(*this)));
131 }
132
133 private:
134#if GTSAM_ENABLE_BOOST_SERIALIZATION
135 /** Serialization function */
136 friend class boost::serialization::access;
137 template <class ARCHIVE>
138 void serialize(ARCHIVE& ar, const unsigned int /*version*/) {
139 ar& boost::serialization::make_nvp(n: "ZeroCostConstraint",
140 v&: boost::serialization::base_object<Base>(d&: *this));
141 ar& BOOST_SERIALIZATION_NVP(factor_);
142 }
143#endif
144};
