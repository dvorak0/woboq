26/** Constrained optimization problem, in the form of
27 * argmin_x 0.5||f(X)||^2
28 * s.t. h(X) = 0
29 * g(X) <= 0
30 * where X represents the variables, 0.5||f(X)||^2 represents the quadratic cost
31 * functions, h(X)=0 represents the nonlinear equality constraints, g(x)<=0 represents the
32 * inequality constraints.
33 */
34class GTSAM_EXPORT ConstrainedOptProblem {
35 public:
36 typedef ConstrainedOptProblem This;
37 typedef std::shared_ptr<This> shared_ptr;
38
39 protected:
40 NonlinearFactorGraph costs_; // cost function, ||f(X)||^2
41 NonlinearEqualityConstraints eqConstraints_; // equality constraints, h(X)=0
42 NonlinearInequalityConstraints ineqConstraints_; // inequality constraints, g(X)<=0
43
44 public:
45 /** Default constructor. */
46 ConstrainedOptProblem() {}
47
48 /** Constructor that packs cost and constraints into a single factor graph. */
49 ConstrainedOptProblem(const NonlinearFactorGraph& graph);
50
51 /** Constructor with both equality and inequality constraints. */
52 ConstrainedOptProblem(const NonlinearFactorGraph& costs,
53 const NonlinearEqualityConstraints& eqConstraints,
54 const NonlinearInequalityConstraints& ineqConstraints);
55
56 /** Constructor with equality constraints only. */
57 static ConstrainedOptProblem EqConstrainedOptProblem(
58 const NonlinearFactorGraph& costs,
59 const NonlinearEqualityConstraints& eqConstraints) {
60 return ConstrainedOptProblem(costs, eqConstraints, NonlinearInequalityConstraints());
61 }
62
63 /** Member variable access functions. */
64 const NonlinearFactorGraph& costs() const { return costs_; }
65 const NonlinearEqualityConstraints& eConstraints() const { return eqConstraints_; }
66 const NonlinearInequalityConstraints& iConstraints() const { return ineqConstraints_; }
67
68 /** Evaluate cost and constraint violations.
69 * Return a tuple representing (cost, e-constraint violation, i-constraint violation).
70 */
71 std::tuple<double, double, double> evaluate(const Values& values) const;
72
73 /** Return the dimension of the problem, as a tuple of
74 * total dimension of cost factors,
75 * total dimension of equality constraints,
76 * total dimension of inequality constraints,
77 * total dimension of variables.
78 */
79 std::tuple<size_t, size_t, size_t> dim() const;
80
81 /** Base class to generate keys for auxiliary variables. */
82 class GTSAM_EXPORT AuxiliaryKeyGenerator {
83 public:
84 AuxiliaryKeyGenerator() {}
85 virtual ~AuxiliaryKeyGenerator() {}
86
87 virtual Key generate(const size_t k) const { return Symbol('u', k); }
88
89 /** generate the next available auxiliary key that is not in values. */
90 Key generate(size_t& k, const Values& values) const {
91 Key key = generate(k: k++);
92 while (values.exists(j: key)) {
93 key = generate(k: k++);
94 }
95 return key;
96 }
97 };
98
99 /// Equivalent equality-constrained optimization problem with auxiliary
100 /// variables z. Inequality constraints g(x)<=0 are transformed into equality
101 /// constraints g(x)+z^2=0.
102 std::pair<ConstrainedOptProblem, Values> auxiliaryProblem(
103 const Values& values, const AuxiliaryKeyGenerator& generator) const;
104};
