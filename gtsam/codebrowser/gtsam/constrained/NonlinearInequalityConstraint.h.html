27/**
28 * Inequality constraint base class, enforcing g(x) <= 0.
29 */
30class GTSAM_EXPORT NonlinearInequalityConstraint : public NonlinearConstraint {
31 public:
32 using Base = NonlinearConstraint;
33 using This = NonlinearInequalityConstraint;
34 using shared_ptr = std::shared_ptr<This>;
35
36 /** Default constructor. */
37 using Base::Base;
38
39 /** Destructor. */
40 virtual ~NonlinearInequalityConstraint() {}
41
42 /** Return g(x). */
43 virtual Vector unwhitenedExpr(const Values& x, OptionalMatrixVecType H = nullptr) const = 0;
44
45 virtual Vector whitenedExpr(const Values& x) const;
46
47 /** Return ramp(g(x)). */
48 virtual Vector unwhitenedError(const Values& x, OptionalMatrixVecType H = nullptr) const override;
49
50 /** Return true if g(x)>=0 in any dimension. */
51 virtual bool active(const Values& x) const override;
52
53 /** Return an equality constraint corresponding to g(x)=0. */
54 virtual NonlinearEqualityConstraint::shared_ptr createEqualityConstraint() const;
55
56 /** Cost factor using a customized penalty function. */
57 virtual NoiseModelFactor::shared_ptr penaltyFactorCustom(
58 InequalityPenaltyFunction::shared_ptr func, const double mu = 1.0) const;
59
60 /** penalty function as if the constraint is equality, 0.5 * mu * ||g(x)||^2 */
61 virtual NoiseModelFactor::shared_ptr penaltyFactorEquality(const double mu = 1.0) const;
62
63 private:
64#if GTSAM_ENABLE_BOOST_SERIALIZATION
65 /** Serialization function */
66 friend class boost::serialization::access;
67 template <class ARCHIVE>
68 void serialize(ARCHIVE& ar, const unsigned int /*version*/) {
69 ar& boost::serialization::make_nvp(n: "NonlinearInequalityConstraint",
70 v&: boost::serialization::base_object<Base>(d&: *this));
71 }
72#endif
73};

75/** Inequality constraint that force g(x) <= 0, where g(x) is a scalar-valued
76 * nonlinear function.
77 */
78class GTSAM_EXPORT ScalarExpressionInequalityConstraint : public NonlinearInequalityConstraint {
79 public:
80 using Base = NonlinearInequalityConstraint;
81 using This = ScalarExpressionInequalityConstraint;
82 using shared_ptr = std::shared_ptr<This>;
83
84 protected:
85 Double_ expression_;
86 FastVector<int> dims_;
87
88 public:
89 /**
90 * @brief Constructor.
91 *
92 * @param expression expression representing g(x) (or -g(x) for GeqZero).
93 * @param sigma scalar representing sigma.
94 */
95 ScalarExpressionInequalityConstraint(const Double_& expression, const double& sigma);
96
97 /** Create an inequality constraint g(x)/sigma >= 0, internally represented as -g(x)/sigma <= 0.
98 */
99 static ScalarExpressionInequalityConstraint::shared_ptr GeqZero(const Double_& expression,
100 const double& sigma);
101
102 /** Create an inequality constraint g(x)/sigma <= 0. */
103 static ScalarExpressionInequalityConstraint::shared_ptr LeqZero(const Double_& expression,
104 const double& sigma);
105
106 /** Compute g(x), or -g(x) for objects constructed from GeqZero. */
107 virtual Vector unwhitenedExpr(const Values& x, OptionalMatrixVecType H = nullptr) const override;
108
109 /** Equality constraint representing g(x)/sigma = 0. */
110 NonlinearEqualityConstraint::shared_ptr createEqualityConstraint() const override;
111
112 /** Penalty function 0.5*mu*||ramp(g(x)/sigma||^2. */
113 NoiseModelFactor::shared_ptr penaltyFactor(const double mu = 1.0) const override;
114
115 /** Penalty function using a smooth approxiamtion of the ramp funciton. */
116 NoiseModelFactor::shared_ptr penaltyFactorCustom(InequalityPenaltyFunction::shared_ptr func,
117 const double mu = 1.0) const override;
118
119 /** Penalty function as if the constraint is equality, 0.5 * mu * ||g(x)/sigma||^2. */
120 virtual NoiseModelFactor::shared_ptr penaltyFactorEquality(const double mu = 1.0) const override;
121
122 /** Return expression g(x), or -g(x) for objects constructed from GeqZero. */
123 const Double_& expression() const { return expression_; }
124
125 /** return a deep copy of this factor. */
126 gtsam::NonlinearFactor::shared_ptr clone() const override {
127 return std::static_pointer_cast<gtsam::NonlinearFactor>(
128 r: gtsam::NonlinearFactor::shared_ptr(new This(*this)));
129 }
130
131 private:
132#if GTSAM_ENABLE_BOOST_SERIALIZATION
133 /** Serialization function */
134 friend class boost::serialization::access;
135 template <class ARCHIVE>
136 void serialize(ARCHIVE& ar, const unsigned int /*version*/) {
137 ar& boost::serialization::make_nvp(n: "ExpressionEqualityConstraint",
138 v&: boost::serialization::base_object<Base>(d&: *this));
139 ar& BOOST_SERIALIZATION_NVP(expression_);
140 ar& BOOST_SERIALIZATION_NVP(dims_);
141 }
142#endif
143};

145/// Container of NonlinearInequalityConstraint.
146class GTSAM_EXPORT NonlinearInequalityConstraints
147 : public FactorGraph<NonlinearInequalityConstraint> {
148 public:
149 using Base = FactorGraph<NonlinearInequalityConstraint>;
150 using This = NonlinearInequalityConstraints;
151 using shared_ptr = std::shared_ptr<This>;
152
153 using Base::Base;
154
155 /** Return the total dimension of constraints. */
156 size_t dim() const;
157
158 /** Evaluate the constraint violation as a vector. */
159 Vector violationVector(const Values& values, bool whiten = true) const;
160
161 /** Evaluate the constraint violation (as L2 norm). */
162 double violationNorm(const Values& values) const;
163
164 /** Return the penalty function corresponding to \sum_i||ramp(g_i(x))||^2. */
165 NonlinearFactorGraph penaltyGraph(const double mu = 1.0) const;
166
167 /** Return the cost graph constructed using a customized penalty function. */
168 NonlinearFactorGraph penaltyGraphCustom(InequalityPenaltyFunction::shared_ptr func,
169 const double mu = 1.0) const;
170
171 private:
172#if GTSAM_ENABLE_BOOST_SERIALIZATION
173 /** Serialization function */
174 friend class boost::serialization::access;
175 template <class ARCHIVE>
176 void serialize(ARCHIVE& ar, const unsigned int /*version*/) {
177 ar& boost::serialization::make_nvp(n: "NonlinearInequalityConstraints",
178 v&: boost::serialization::base_object<Base>(d&: *this));
179 }
180#endif
181};
