25/** Base class for constrained optimization parameters. */
26class GTSAM_EXPORT ConstrainedOptimizerParams {
27 public:
28 using shared_ptr = std::shared_ptr<ConstrainedOptimizerParams>;
29
30 size_t maxIterations = 20; // maximum number of iterations
31 double absoluteViolationTolerance = 1e-5;
32 double relativeViolationTolerance = 1e-5;
33 double absoluteCostTolerance = 1e-5;
34 double relativeCostTolerance = 1e-5;
35 bool verbose = false;
36 bool storeOptProgress = false; // Store detailed info of each iter
37
38 /** Constructor. */
39 ConstrainedOptimizerParams() {}
40};

42/** Base class for constrained optimizer state. */
43class GTSAM_EXPORT ConstrainedOptimizerState {
44 public:
45 size_t iteration = 0;
46 Values values;
47 double cost = 0;
48 double eqConstraintViolation = 0;
49 double ineqConstraintViolation = 0;
50 double time = 0.0;
51
52 ConstrainedOptimizerState() {}
53
54 ConstrainedOptimizerState(const size_t _iteration) : iteration(_iteration) {}
55
56 ConstrainedOptimizerState(const size_t _iteration, const Values& _values,
57 const ConstrainedOptProblem& problem)
58 : iteration(_iteration), values(_values) {
59 std::tie(args&: cost, args&: eqConstraintViolation, args&: ineqConstraintViolation) =
60 problem.evaluate(values: _values);
61 }
62
63 void setValues(const Values& _values, const ConstrainedOptProblem& problem) {
64 values = _values;
65 std::tie(args&: cost, args&: eqConstraintViolation, args&: ineqConstraintViolation) =
66 problem.evaluate(values: _values);
67 }
68
69 double violation() const {
70 return sqrt(x: pow(x: eqConstraintViolation, y: 2) +
71 pow(x: ineqConstraintViolation, y: 2));
72 }
73};

75/** Base class for constrained optimizer. This class is different from
76 * NonlinearOptimizer because it also includes constraints in the optimization
77 * problem. Although NonlinearOptimizer and ConstrainedOptimizer both solve the
78 * optimization problem iteratively, ConstrainedOptimizer does not inherit
79 * NonlinearOptimizer because the sub-problems solved in each iteration is
80 * different. For NonlinearOptimizer, each iteration solves an unconstrained
81 * linear least-squares problem (i.e., GaussianFactorGraph). For certain
82 * ConstrainedOptimizer (PenaltyOptimizer and AugmentedLagrangianOptimizer),
83 * each iteration solves an unconstrained nonlinear least-squares problem (i.e.,
84 * NonlinearFactorGraph).*/
85class GTSAM_EXPORT ConstrainedOptimizer {
86 public:
87 typedef std::shared_ptr<NonlinearOptimizer> SharedOptimizer;
88 typedef ConstrainedOptimizerParams Params;
89 typedef ConstrainedOptimizerState State;
90
91 protected:
92 ConstrainedOptProblem problem_;
93 Values initialValues_;
94
95 public:
96 /** Default constructor. */
97 ConstrainedOptimizer() {}
98
99 /** Constructor. */
100 ConstrainedOptimizer(const ConstrainedOptProblem& problem,
101 const Values& initialValues)
102 : problem_(problem), initialValues_(initialValues) {}
103
104 /** Constructor that packs the cost and constraints into a single factor
105 * graph. */
106 ConstrainedOptimizer(const NonlinearFactorGraph& graph,
107 const Values& initialValues)
108 : problem_(graph), initialValues_(initialValues) {}
109
110 virtual ~ConstrainedOptimizer() {}
111
112 /// Solve a constrained optimization problem.
113 virtual Values optimize() const = 0;
114
115 protected:
116 virtual bool checkConvergence(const State& state, const State& previousState,
117 const Params& params) const {
118 if (state.iteration >= params.maxIterations) {
119 return true;
120 }
121
122 if (state.violation() < params.absoluteViolationTolerance &&
123 state.cost < params.absoluteCostTolerance) {
124 return true;
125 }
126
127 if (abs(x: state.violation() - previousState.violation()) <
128 params.relativeViolationTolerance &&
129 abs(x: state.cost - previousState.cost) < params.relativeCostTolerance) {
130 return true;
131 }
132
133 return false;
134 }
135};
