24namespace gtsam {

26/** Base class for smooth approximation of the ramp function. */
27class GTSAM_EXPORT InequalityPenaltyFunction {
28 public:
29 using shared_ptr = std::shared_ptr<InequalityPenaltyFunction>;
30 using UnaryScalarFunc =
31 std::function<double(const double& x, OptionalJacobian<1, 1> H)>;
32
33 /** Constructor. */
34 InequalityPenaltyFunction() {}
35
36 /** Destructor. */
37 virtual ~InequalityPenaltyFunction() {}
38
39 virtual double operator()(const double& x,
40 OptionalJacobian<1, 1> H = {}) const = 0;
41
42 virtual UnaryScalarFunc function() const;
43};
44
45/** Ramp function f : R -> R.
46 * f(x) = 0 for x <= 0
47 * x for x > 0
48 */
49class GTSAM_EXPORT RampFunction : public InequalityPenaltyFunction {
50 public:
51 using Base = InequalityPenaltyFunction;
52 using This = RampFunction;
53 using shared_ptr = std::shared_ptr<This>;
54
55 public:
56 RampFunction() : Base() {}
57
58 virtual double operator()(const double& x,
59 OptionalJacobian<1, 1> H = {}) const override {
60 return Ramp(x, H);
61 }
62
63 virtual UnaryScalarFunc function() const override { return Ramp; }
64
65 static double Ramp(const double x, OptionalJacobian<1, 1> H = {});
66};
67
68/** Ramp function approximated with a polynomial of degree 2 in [0, epsilon].
69 * The coefficients are computed as
70 * a = 1 / (2 * epsilon)
71 * Function f(x) = 0 for x <= 0
72 * a * x^2 for 0 < x < epsilon
73 * x - epsilon/2 for x >= epsilon
74 */
75class GTSAM_EXPORT SmoothRampPoly2 : public InequalityPenaltyFunction {
76 public:
77 using Base = InequalityPenaltyFunction;
78 using This = SmoothRampPoly2;
79 using shared_ptr = std::shared_ptr<This>;
80
81 protected:
82 double epsilon_;
83 double a_;
84
85 public:
86 /** Constructor.
87 * @param epsilon parameter for adjusting the smoothness of the function.
88 */
89 SmoothRampPoly2(const double epsilon = 1)
90 : Base(), epsilon_(epsilon), a_(0.5 / epsilon) {}
91
92 virtual double operator()(const double& x,
93 OptionalJacobian<1, 1> H = {}) const override;
94};
95
96/** Ramp function approximated with a polynomial of degree 3 in [0, epsilon].
97 * The coefficients are computed as
98 * a = -1 / epsilon^2
99 * b = 2 / epsilon
100 * Function f(x) = 0 for x <= 0
101 * a * x^3 + b * x^2 for 0 < x < epsilon
102 * x for x >= epsilon
103 */
104class GTSAM_EXPORT SmoothRampPoly3 : public InequalityPenaltyFunction {
105 public:
106 using Base = InequalityPenaltyFunction;
107 using This = SmoothRampPoly3;
108 using shared_ptr = std::shared_ptr<This>;
109
110 protected:
111 double epsilon_;
112 double a_;
113 double b_;
114
115 public:
116 /** Constructor.
117 * @param epsilon parameter for adjusting the smoothness of the function.
118 */
119 SmoothRampPoly3(const double epsilon = 1)
120 : Base(),
121 epsilon_(epsilon),
122 a_(-1 / (epsilon * epsilon)),
123 b_(2 / epsilon) {}
124
125 virtual double operator()(const double& x,
126 OptionalJacobian<1, 1> H = {}) const override;
127};

129/** Softplus function that implements f(x) = log(1 + exp(k*x)) / k. */
130class GTSAM_EXPORT SoftPlusFunction : public InequalityPenaltyFunction {

132 using Base = InequalityPenaltyFunction;
133 using This = SoftPlusFunction;
134 using shared_ptr = std::shared_ptr<This>;
135
136 protected:
137 double k_;
138
139 public:
140 /** Constructor.
141 * @param k parameter for adjusting the smoothness of the function.
142 */
143 SoftPlusFunction(const double k = 1) : Base(), k_(k) {}
144
145 virtual double operator()(const double& x,
146 OptionalJacobian<1, 1> H = {}) const override;
147};
