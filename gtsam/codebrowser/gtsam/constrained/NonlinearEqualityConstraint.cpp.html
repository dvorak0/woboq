23/* ********************************************************************************************* */
24ZeroCostConstraint::ZeroCostConstraint(const NoiseModelFactor::shared_ptr& factor)
25 : Base(constrainedNoise(sigmas: factor->noiseModel()->sigmas()), factor->keys()), factor_(factor) {}

27/* ********************************************************************************************* */
28Vector ZeroCostConstraint::unwhitenedError(const Values& x, OptionalMatrixVecType H) const {
29 return factor_->unwhitenedError(x, H);
30}
31
32/* ********************************************************************************************* */
33NoiseModelFactor::shared_ptr ZeroCostConstraint::penaltyFactor(const double mu) const {
34 return factor_->cloneWithNewNoiseModel(newNoise: penaltyNoise(mu));
35}

37/* ********************************************************************************************* */
38NonlinearEqualityConstraints NonlinearEqualityConstraints::FromCostGraph(
39 const NonlinearFactorGraph& graph) {
40 NonlinearEqualityConstraints constraints;
41 for (const auto& factor : graph) {
42 auto noise_factor = std::dynamic_pointer_cast<NoiseModelFactor>(r: factor);
43 constraints.emplace_shared<ZeroCostConstraint>(args&: noise_factor);
44 }
45 return constraints;
46}

48/* ********************************************************************************************* */
49size_t NonlinearEqualityConstraints::dim() const {
50 size_t dimension = 0;
51 for (const auto& constraint : *this) {
52 dimension += constraint->dim();
53 }
54 return dimension;
55}
56
57/* ********************************************************************************************* */
58Vector NonlinearEqualityConstraints::violationVector(const Values& values, bool whiten) const {
59 Vector violation(dim());
60 size_t start_idx = 0;
61 for (const auto& constraint : *this) {
62 size_t dim = constraint->dim();
63 violation.middleCols(startCol: start_idx, numCols: dim) =
64 whiten ? constraint->whitenedError(c: values) : constraint->unwhitenedError(x: values);
65 start_idx += dim;
66 }
67 return violation;
68}
69
70/* ********************************************************************************************* */
71double NonlinearEqualityConstraints::violationNorm(const Values& values) const {
72 return violationVector(values, whiten: true).norm();
73}

75/* ********************************************************************************************* */
76NonlinearFactorGraph NonlinearEqualityConstraints::penaltyGraph(const double mu) const {
77 NonlinearFactorGraph graph;
78 for (const auto& constraint : *this) {
79 graph.add(factor: constraint->penaltyFactor(mu));
80 }
81 return graph;
82}
