28namespace gtsam {
29
30/**
31 * Base class for nonlinear constraint.
32 * The constraint is represented as a NoiseModelFactor with Constrained noise model.
33 * whitenedError() returns The constraint violation vector.
34 * unwhitenedError() returns the sigma-scaled constraint violation vector.
35 */
36class GTSAM_EXPORT NonlinearConstraint : public NoiseModelFactor {
37 public:
38 using Base = NoiseModelFactor;
39
40 /** Use constructors of NoiseModelFactor. */
41 using Base::Base;
42
43 using NonlinearFactor::error;
44
45 /** Destructor. */
46 virtual ~NonlinearConstraint() {}
47
48 /** Create a cost factor representing the L2 penalty function with scaling coefficient mu. */
49 virtual NoiseModelFactor::shared_ptr penaltyFactor(const double mu = 1.0) const {
50 return cloneWithNewNoiseModel(newNoise: penaltyNoise(mu));
51 }
52
53 /** Return the norm of the constraint violation vector. */
54 virtual double violation(const Values& x) const { return sqrt(x: 2 * error(c: x)); }
55
56 /** Check if constraint violation is within tolerance. */
57 virtual bool feasible(const Values& x, const double tolerance = 1e-5) const {
58 return violation(x) <= tolerance;
59 }
60
61 const Vector sigmas() const { return noiseModel()->sigmas(); }
62
63 // return the hessian of unwhitened error function in each dimension.
64 virtual std::vector<Matrix> unwhitenedHessian(const Values& x) const {
65 throw std::runtime_error("hessian not implemented");
66 }
67
68 protected:
69 /** Noise model used for the penalty function. */
70 SharedDiagonal penaltyNoise(const double mu) const {
71 return noiseModel::Diagonal::Sigmas(sigmas: noiseModel()->sigmas() / sqrt(x: mu));
72 }
73
74 /** Default constrained noisemodel used for construction of constraint. */
75 static SharedNoiseModel constrainedNoise(const Vector& sigmas) {
76 return noiseModel::Constrained::MixedSigmas(m: 1.0, sigmas);
77 }
78
79 private:
80#if GTSAM_ENABLE_BOOST_SERIALIZATION
81 /** Serialization function */
82 friend class boost::serialization::access;
83 template <class ARCHIVE>
84 void serialize(ARCHIVE& ar, const unsigned int /*version*/) {
85 ar& boost::serialization::make_nvp(n: "NonlinearConstraint",
86 v&: boost::serialization::base_object<Base>(d&: *this));
87 }
88#endif
89};
