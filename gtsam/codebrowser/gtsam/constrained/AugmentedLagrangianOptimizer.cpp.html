28/** A factor that adds a constant bias term to the original factor.
29 * This factor is used in augmented Lagrangian optimizer to create biased cost
30 * functions.
31 * Note that the noise model is stored twice (both in original factor and the
32 * noisemodel of substitute factor. The noisemodel in the original factor will
33 * be ignored. */
34class GTSAM_EXPORT BiasedFactor : public NoiseModelFactor {
35 protected:
36 typedef NoiseModelFactor Base;
37 typedef BiasedFactor This;
38
39 // original factor
40 Base::shared_ptr originalFactor_;
41 Vector bias_;
42
43 public:
44 typedef std::shared_ptr<This> shared_ptr;
45
46 /** Default constructor for I/O only */
47 BiasedFactor() {}
48
49 /** Destructor */
50 ~BiasedFactor() override {}
51
52 /**
53 * Constructor
54 * @param originalFactor original factor on X
55 * @param bias the bias term
56 */
57 BiasedFactor(const Base::shared_ptr& originalFactor, const Vector& bias)
58 : Base(originalFactor->noiseModel(), originalFactor->keys()),
59 originalFactor_(originalFactor),
60 bias_(bias) {}
61
62 /**
63 * Error function *without* the NoiseModel, \f$ z-h(x) \f$.
64 * Override this method to finish implementing an N-way factor.
65 * If the optional arguments is specified, it should compute
66 * both the function evaluation and its derivative(s) in H.
67 */
68 virtual Vector unwhitenedError(
69 const Values& x,
70 gtsam::OptionalMatrixVecType H = nullptr) const override {
71 return originalFactor_->unwhitenedError(x, H) + bias_;
72 }
73
74 /** print */
75 void print(const std::string& s, const KeyFormatter& keyFormatter =
76 DefaultKeyFormatter) const override {
77 std::cout << s << "BiasedFactor " << bias_.transpose()
78 << " version of:" << std::endl;
79 originalFactor_->print(s, keyFormatter);
80 }
81
82 /** Return a deep copy of this factor. */
83 gtsam::NonlinearFactor::shared_ptr clone() const override {
84 return std::static_pointer_cast<gtsam::NonlinearFactor>(
85 r: gtsam::NonlinearFactor::shared_ptr(new This(*this)));
86 }
87
88}; // \class BiasedFactor

90/* ************************************************************************* */
91void AugmentedLagrangianState::initializeLagrangeMultipliers(
92 const ConstrainedOptProblem& problem) {
93 lambdaEq = std::vector<Vector>();
94 lambdaEq.reserve(n: problem.eConstraints().size());
95 for (const auto& constraint : problem.eConstraints()) {
96 lambdaEq.push_back(x: Vector::Zero(size: constraint->dim()));
97 }
98 lambdaIneq = std::vector<double>(problem.iConstraints().size(), 0);
99}

101/* ************************************************************************* */
102std::tuple<AugmentedLagrangianOptimizer::State, double, double>
103AugmentedLagrangianOptimizer::iterate(const State& state, const double muEq,
104 const double muIneq) const {
105 State newState(state.iteration + 1);
106 newState.muEq = muEq;
107 newState.muIneq = muIneq;
108
109 // Update Lagrangian multipliers.
110 updateLagrangeMultiplier(prev_state: state, state: &newState);
111
112 // Construct merit function.
113 const NonlinearFactorGraph augmentedLagrangian =
114 augmentedLagrangianFunction(state: newState);
115
116 // Run unconstrained optimization.
117 auto optimizer =
118 createUnconstrainedOptimizer(graph: augmentedLagrangian, values: state.values);
119 newState.setValues(values: optimizer->optimize(), problem: problem_);
120 newState.unconstrainedIterationss = optimizer->iterations();
121
122 // Update penalty parameters for next iteration.
123 double next_muEq, next_muIneq;
124 std::tie(args&: next_muEq, args&: next_muIneq) = updatePenaltyParameter(prev_state: state, state: newState);
125
126 return {newState, next_muEq, next_muIneq};
127}

129/* ************************************************************************* */
130Values AugmentedLagrangianOptimizer::optimize() const {
131 /// Construct initial state
132 State previousState;
133 State state(0, initialValues_, problem_);
134 state.initializeLagrangeMultipliers(problem: problem_);
135 logInitialState(state);
136
137 /// Set penalty parameters for the first iteration.
138 double muEq = p_->initialMuEq;
139 double muIneq = p_->initialMuIneq;
140
141 /// iterates
142 do {
143 previousState = std::move(state);
144 std::tie(args&: state, args&: muEq, args&: muIneq) = iterate(state: previousState, muEq, muIneq);
145 logIteration(state);
146 } while (!checkConvergence(state, previousState, params: *p_));
147
148 return state.values;
149}

151/* ************************************************************************* */
152NonlinearFactorGraph AugmentedLagrangianOptimizer::augmentedLagrangianFunction(
153 const State& state, const double epsilon) const {
154 // Initialize by adding in cost factors.
155 NonlinearFactorGraph graph = problem_.costs();
156
157 // Create factors corresponding to equality constraints.
158 const NonlinearEqualityConstraints& eqConstraints = problem_.eConstraints();
159 const double& muEq = state.muEq;
160 for (size_t i = 0; i < eqConstraints.size(); i++) {
161 const auto& constraint = eqConstraints.at(i);
162 Vector bias = state.lambdaEq[i] / muEq * constraint->sigmas();
163 auto penalty_l2 = constraint->penaltyFactor(mu: muEq);
164 graph.emplace_shared<BiasedFactor>(args&: penalty_l2, args&: bias);
165 }
166
167 // Create factors corresponding to penalty terms of inequality constraints.
168 const NonlinearInequalityConstraints& ineqConstraints =
169 problem_.iConstraints();
170 const double& muIneq = state.muIneq;
171 graph.add(factorOrContainer: ineqConstraints.penaltyGraphCustom(
172 func: p_->ineqConstraintPenaltyFunction, mu: muIneq));
173
174 // Create factors corresponding to Lagrange multiplier terms of i-constraints.
175 for (size_t i = 0; i < ineqConstraints.size(); i++) {
176 const auto& constraint = ineqConstraints.at(i);
177 Vector bias = state.lambdaIneq[i] / epsilon * constraint->sigmas();
178 auto penalty_l2 = constraint->penaltyFactorEquality(mu: epsilon);
179 graph.emplace_shared<BiasedFactor>(args&: penalty_l2, args&: bias);
180 graph.emplace_shared<AntiFactor>(args&: penalty_l2);
181 }
182
183 return graph;
184}

186/* ************************************************************************* */
187void AugmentedLagrangianOptimizer::updateLagrangeMultiplier(
188 const State& previousState, State* state) const {
189 // Perform dual ascent on Lagrange multipliers for e-constriants.
190 const NonlinearEqualityConstraints& eqConstraints = problem_.eConstraints();
191 state->lambdaEq.resize(new_size: eqConstraints.size());
192 for (size_t i = 0; i < eqConstraints.size(); i++) {
193 const auto& constraint = eqConstraints.at(i);
194 // Compute constraint violation as the gradient of the dual function.
195 Vector violation = constraint->whitenedError(c: previousState.values);
196 double step_size = std::min(a: p_->maxDualStepSizeEq,
197 b: previousState.muEq * p_->dualStepSizeFactorEq);
198 state->lambdaEq[i] = previousState.lambdaEq[i] + step_size * violation;
199 }
200
201 // Perform dual ascent on Lagrange multipliers for i-constriants.
202 const NonlinearInequalityConstraints& ineqConstraints =
203 problem_.iConstraints();
204 state->lambdaIneq.resize(new_size: ineqConstraints.size());
205 // Update Lagrangian multipliers.
206 for (size_t i = 0; i < ineqConstraints.size(); i++) {
207 const auto& constraint = ineqConstraints.at(i);
208 double violation = constraint->whitenedExpr(x: previousState.values)(0);
209 double step_size =
210 std::min(a: p_->maxDualStepSizeIneq,
211 b: previousState.muIneq * p_->dualStepSizeFactorIneq);
212 state->lambdaIneq[i] =
213 std::max(a: 0.0, b: previousState.lambdaIneq[i] + step_size * violation);
214 }
215}

217/* ************************************************************************* */
218std::pair<double, double> AugmentedLagrangianOptimizer::updatePenaltyParameter(
219 const State& previousState, const State& state) const {
220 double muEq = state.muEq;
221 if (problem_.eConstraints().size() > 0 &&
222 state.eqConstraintViolation >=
223 p_->muIncreaseThreshold * previousState.eqConstraintViolation) {
224 muEq *= p_->muEqIncreaseRate;
225 }
226
227 double muIneq = state.muIneq;
228 if (problem_.iConstraints().size() > 0 &&
229 state.ineqConstraintViolation >=
230 p_->muIncreaseThreshold * previousState.ineqConstraintViolation) {
231 muIneq *= p_->muIneqIncreaseRate;
232 }
233 return {muEq, muIneq};
234}

238AugmentedLagrangianOptimizer::createUnconstrainedOptimizer(
239 const NonlinearFactorGraph& graph, const Values& values) const {
240 // TODO(yetong): make compatible with all NonlinearOptimizers.

247 if (p_->verbose) {
248 // Log title line.
249 cout << setw(10) << "Iter"
250 << "|" << setw(10) << "muEq"
251 << "|" << setw(10) << "muIneq"
252 << "|" << setw(10) << "cost"
253 << "|" << setw(10) << "vio_e"
254 << "|" << setw(10) << "vio_i"
255 << "|" << setw(10) << "uopt_iters"
256 << "|" << setw(10) << "time"
257 << "|" << endl;
258
259 // Log initial value line.
260 cout << setw(10) << state.iteration;
261 cout << "|" << setw(10) << "-";
262 cout << "|" << setw(10) << "-";
263 cout << "|" << setw(10) << setprecision(4) << state.cost;
264 cout << "|" << setw(10) << setprecision(4) << state.eqConstraintViolation;
265 cout << "|" << setw(10) << setprecision(4) << state.ineqConstraintViolation;
266 cout << "|" << setw(10) << "-";
267 cout << "|" << setw(10) << "-";
268 cout << "|" << endl;
269 }
270
271 // Store state
272 if (p_->storeOptProgress) {
273 progress_.emplace_back(args: state);
274 }
