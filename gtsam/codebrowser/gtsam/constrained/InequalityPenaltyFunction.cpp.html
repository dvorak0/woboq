23/* ************************************************************************* */
24InequalityPenaltyFunction::UnaryScalarFunc InequalityPenaltyFunction::function()
25 const {
26 return [this](const double& x, OptionalJacobian<1, 1> H = {}) {
27 return this->operator()(x, H);
28 };
29}

31/* ************************************************************************* */
32double RampFunction::Ramp(const double x, OptionalJacobian<1, 1> H) {
33 if (x < 0) {
34 if (H) {
35 H->setConstant(0);
36 }
37 return 0;
38 } else {
39 if (H) {
40 H->setConstant(1);
41 }
42 return x;
43 }
44}

46/* ************************************************************************* */
47double SmoothRampPoly2::operator()(const double& x,
48 OptionalJacobian<1, 1> H) const {
49 if (x <= 0) {
50 if (H) {
51 H->setZero();
52 }
53 return 0;
54 } else if (x < epsilon_) {
55 if (H) {
56 H->setConstant(2 * a_ * x);
57 }
58 return a_ * pow(x: x, y: 2);
59 } else {
60 if (H) {
61 H->setOnes();
62 }
63 return x - epsilon_ / 2;
64 }
65}
66
67/* ************************************************************************* */
68double SmoothRampPoly3::operator()(const double& x,
69 OptionalJacobian<1, 1> H) const {
70 if (x <= 0) {
71 if (H) {
72 H->setZero();
73 }
74 return 0;
75 } else if (x < epsilon_) {
76 if (H) {
77 H->setConstant(3 * a_ * pow(x: x, y: 2) + 2 * b_ * x);
78 }
79 return a_ * pow(x: x, y: 3) + b_ * pow(x: x, y: 2);
80 } else {
81 if (H) {
82 H->setOnes();
83 }
84 return x;
85 }
86}

88/* ************************************************************************* */
89double SoftPlusFunction::operator()(const double& x,
90 OptionalJacobian<1, 1> H) const {
91 if (H) {
92 H->setConstant(1 / (1 + std::exp(x: -k_ * x)));
93 }
94 return std::log(x: 1 + std::exp(x: k_ * x)) / k_;
95}
