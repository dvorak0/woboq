24/* ********************************************************************************************* */
25Vector NonlinearInequalityConstraint::whitenedExpr(const Values& x) const {
26 return noiseModel()->whiten(v: unwhitenedExpr(x));
27}
28
29/* ********************************************************************************************* */
30Vector NonlinearInequalityConstraint::unwhitenedError(const Values& x,
31 OptionalMatrixVecType H) const {
32 Vector error = unwhitenedExpr(x, H);
33 for (size_t i = 0; i < dim(); i++) {
34 if (error(i) < 0) {
35 error(i) = 0;
36 if (H) {
37 for (Matrix& m : *H) {
38 m.row(i).setZero();
39 }
40 }
41 }
42 }
43 return error;
44}

46/* ********************************************************************************************* */
47bool NonlinearInequalityConstraint::active(const Values& x) const {
48 return (unwhitenedExpr(x).array() >= 0).any();
49}

91/* ********************************************************************************************* */
92Vector ScalarExpressionInequalityConstraint::unwhitenedExpr(const Values& x,
93 OptionalMatrixVecType H) const {
94 // Copy-paste from ExpressionFactor.
95 if (H) {
96 return Vector1(expression_.valueAndDerivatives(values: x, keys: keys_, dims: dims_, H&: *H));
97 } else {
98 return Vector1(expression_.value(values: x));
99 }
100}

109/* ********************************************************************************************* */
110NoiseModelFactor::shared_ptr ScalarExpressionInequalityConstraint::penaltyFactor(
111 const double mu) const {
112 Double_ penalty_expression(RampFunction::Ramp, expression_);
113 return std::make_shared<ExpressionFactor<double>>(args: penaltyNoise(mu), args: 0.0, args&: penalty_expression);
114}

116/* ********************************************************************************************* */
117NoiseModelFactor::shared_ptr ScalarExpressionInequalityConstraint::penaltyFactorCustom(
118 InequalityPenaltyFunction::shared_ptr func, const double mu) const {
119 if (!func) {
120 return penaltyFactor(mu);
121 }
122 // TODO(yetong): can we pass the functor directly to construct the expression?
123 Double_ error(func->function(), expression_);
124 return std::make_shared<ExpressionFactor<double>>(args: penaltyNoise(mu), args: 0.0, args&: error);
125}

133/* ********************************************************************************************* */
134size_t NonlinearInequalityConstraints::dim() const {
135 size_t dimension = 0;
136 for (const auto& constraint : *this) {
137 dimension += constraint->dim();
138 }
139 return dimension;
140}
141
142/* ********************************************************************************************* */
143Vector NonlinearInequalityConstraints::violationVector(const Values& values, bool whiten) const {
144 Vector violation(dim());
145 size_t start_idx = 0;
146 for (const auto& constraint : *this) {
147 size_t dim = constraint->dim();
148 violation.middleCols(startCol: start_idx, numCols: dim) =
149 whiten ? constraint->whitenedError(c: values) : constraint->unwhitenedError(x: values);
150 start_idx += dim;
151 }
152 return violation;
153}
154
155/* ********************************************************************************************* */
156double NonlinearInequalityConstraints::violationNorm(const Values& values) const {
157 return violationVector(values, whiten: true).norm();
158}

162 NonlinearFactorGraph graph;
163 for (const auto& constraint : *this) {

165 }
166 return graph;
