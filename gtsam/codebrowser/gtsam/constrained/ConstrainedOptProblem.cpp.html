24size_t GraphDimension(const NonlinearFactorGraph& graph) {
25 size_t total_dim = 0;
26 for (const auto& factor : graph) {
27 total_dim += factor->dim();
28 }
29 return total_dim;
30}
31
32/* ************************************************************************* */
33bool CheckPureCost(const NonlinearFactorGraph& graph) {
34 for (const auto& factor : graph) {
35 if (NoiseModelFactor::shared_ptr f =
36 std::dynamic_pointer_cast<NoiseModelFactor>(r: factor)) {
37 if (f->noiseModel()->isConstrained()) {
38 return false;
39 }
40 }
41 }
42 return true;
43}

45/* ************************************************************************* */
46ConstrainedOptProblem::ConstrainedOptProblem(
47 const NonlinearFactorGraph& costs,
48 const NonlinearEqualityConstraints& eqConstraints,
49 const NonlinearInequalityConstraints& ineqConstraints)
50 : costs_(costs),
51 eqConstraints_(eqConstraints),
52 ineqConstraints_(ineqConstraints) {
53 if (!CheckPureCost(graph: costs)) {
54 throw std::runtime_error(
55 "Cost contains factors with constrained noise model. They should be "
56 "moved to constraints.");
57 }
58}
59
60ConstrainedOptProblem::ConstrainedOptProblem(
61 const NonlinearFactorGraph& graph) {
62 for (const auto& factor : graph) {
63 if (NonlinearEqualityConstraint::shared_ptr f =
64 std::dynamic_pointer_cast<NonlinearEqualityConstraint>(r: factor)) {
65 eqConstraints_.push_back(factor: f);
66 } else if (NonlinearInequalityConstraint::shared_ptr f =
67 std::dynamic_pointer_cast<NonlinearInequalityConstraint>(
68 r: factor)) {
69 ineqConstraints_.push_back(factor: f);
70 } else {
71 costs_.push_back(factor);
72 }
73 }
74}

76/* ************************************************************************* */
77std::tuple<size_t, size_t, size_t> ConstrainedOptProblem::dim() const {
78 return {GraphDimension(graph: costs()), eConstraints().dim(), iConstraints().dim()};
79}

81/* ************************************************************************* */
82std::tuple<double, double, double> ConstrainedOptProblem::evaluate(
83 const Values& values) const {
84 return {costs().error(values), eConstraints().violationNorm(values),
85 iConstraints().violationNorm(values)};
86}

88/* ************************************************************************* */
89std::pair<ConstrainedOptProblem, Values>
90ConstrainedOptProblem::auxiliaryProblem(
91 const Values& values, const AuxiliaryKeyGenerator& generator) const {
92 if (iConstraints().size() == 0) {
93 return {*this, values};
94 }
95
96 NonlinearEqualityConstraints newEqConstraints = eConstraints();
97 Values new_values = values;
98
99 size_t k = 0;
100 for (const auto& i_constraint : iConstraints()) {
101 if (ScalarExpressionInequalityConstraint::shared_ptr p =
102 std::dynamic_pointer_cast<ScalarExpressionInequalityConstraint>(
103 r: i_constraint)) {
104 // Generate next available auxiliary key.
105 Key aux_key = generator.generate(k, values: new_values);
106
107 // Construct auxiliary equality constraint.
108 Double_ aux_expr(aux_key);
109 Double_ equality_expr = p->expression() + aux_expr * aux_expr;
110 newEqConstraints.emplace_shared<ExpressionEqualityConstraint<double>>(
111 args&: equality_expr, args: 0.0, args: p->noiseModel()->sigmas());
112
113 // Compute initial value for auxiliary key.
114 if (!i_constraint->feasible(x: values)) {
115 new_values.insert(j: aux_key, val: 0.0);
116 } else {
117 Vector gap = i_constraint->unwhitenedExpr(x: values);
118 new_values.insert(j: aux_key, val: sqrt(x: -gap(0)));
119 }
120 }
121 }
122 return {
123 ConstrainedOptProblem::EqConstrainedOptProblem(costs: costs_, eqConstraints: newEqConstraints),
124 new_values};
125}
