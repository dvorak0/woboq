58/** Augmented Lagrangian method to solve constrained nonlinear least squares
59 * problems. The implementation follows
60 * https://www.seas.ucla.edu/~vandenbe/133B/lectures/nllseq.pdf for problems
61 * with equality constraints only. We further generalize the implementation to
62 * incorporate inequality constraints with reference to
63 * https://www.stat.cmu.edu/~ryantibs/convexopt/scribes/dual-decomp-scribed.pdf.
64 * Example problem:
65 * argmin_x 0.5 * ||x1-1||^2 + 0.5 * ||x2-1||^2
66 * s.t. x1^2 + x2^2 - 1 = 0
67 * s.t. 4*x1^2 + 0.25*x2^2 - 1 <= 0
68 */
69class GTSAM_EXPORT AugmentedLagrangianOptimizer : public ConstrainedOptimizer {
70 public:
71 using Base = ConstrainedOptimizer;
72 using This = PenaltyOptimizer;
73 using shared_ptr = std::shared_ptr<This>;
74
75 using Params = AugmentedLagrangianParams;
76 using State = AugmentedLagrangianState;
77 using Progress = std::vector<AugmentedLagrangianState>;
78
79 protected:
80 Params::shared_ptr p_;
81 mutable Progress progress_;
82
83 public:
84 /// Constructor.
85 AugmentedLagrangianOptimizer(
86 const ConstrainedOptProblem& problem, const Values& initialValues,
87 Params::shared_ptr p = std::make_shared<Params>())
88 : Base(problem, initialValues), p_(p), progress_() {}
89
90 /// Constructor that packs costs and constraints into a single factor graph.
91 AugmentedLagrangianOptimizer(
92 const NonlinearFactorGraph& graph, const Values& initialValues,
93 Params::shared_ptr p = std::make_shared<Params>())
94 : Base(graph, initialValues), p_(p), progress_() {}
95
96 /// Run optimization with equality constraints only.
97 Values optimize() const override;
98
99 std::tuple<State, double, double> iterate(const State& state,
100 const double muEq,
101 const double muIneq) const;
102
103 /// Return progress of iterations.
104 const Progress& progress() const { return progress_; }
105
106 /**
107 * Lagrange dual function for equality constraints and inequality constraints
108 * m(x) = 0.5 * ||f(x)||^2 - lambdaEq * h(x) + 0.5 * muEq * ||h(x)||^2
109 * - lambdaIneq * g(x) + 0.5 * muIneq * ||g(x)_-||^2
110 * To express in nonlinear least squares form, it is rewritten as
111 * m(x)
112 * = m(x) + 0.5 * epsilon * ||g(x)||^2
113 * = 0.5 * ||f(x)||^2
114 * + (0.5 * muEq * ||h(x)||^2 - lambdaEq * h(x))
115 * + (0.5 * epsilon * ||g(x)||^2 - lambdaIneq * g(x))
116 * + 0.5 * muIneq * ||g(x)_-||^2
117 * - 0.5 * epsilon * ||g(x)||^2
118 * = 0.5 * ||f(x)||^2
119 * + 0.5 * muEq * ||h(x)- lambdaEq/muEq||^2
120 * + 0.5 * epsilon * ||g(x)-lambdaIneq/muIneq||^2
121 * + 0.5 * muIneq * ||g(x)_-||^2
122 * - 0.5 * epsilon * ||g(x)||^2
123 * - c
124 * where
125 * c = ||lambdaEq||^2 / (2 * muEq) + ||lambdaIneq||^2 / (2 * epsilon)
126 * is a constant term,
127 * and epsilon can be any positive scalar value.
128 *
129 * Notice: the purpose of epsilon is to incorporate (-lambdaIneq * g(x)) in
130 * nonlinear least squares form. To do so, we manually create an additional
131 * term (0.5 * epsilon * ||g(x)||^2), which is added and then subtracted in
132 * the merit function. The term (-lambdaIneq * g(x)) and (0.5 * epsilon *
133 * ||g(x)||^2) can be combined as a least-square term, and the subtraction of
134 * (0.5 * epsilon * ||g(x)||^2) can be performed with anti-factor.
135 * @return: factor graph representing m(x) + 0.5d * ||g(x)||^2 + c
136 */
137 NonlinearFactorGraph augmentedLagrangianFunction(
138 const State& state, const double epsilon = 1.0) const;
139
140 protected:
141 /// Create an unconstrained optimizer that solves the augmented Lagrangian.
142 SharedOptimizer createUnconstrainedOptimizer(
143 const NonlinearFactorGraph& graph, const Values& values) const;
144
145 /** Update the Lagrange multipliers using dual ascent. */
146 void updateLagrangeMultiplier(const State& prev_state, State* state) const;
147
148 /// Set the penalty parameters for the state using results from prev state.
149 std::pair<double, double> updatePenaltyParameter(const State& prev_state,
150 const State& state) const;
151
152 /// Store and log the initial state of the optimization.
153 void logInitialState(const State& state) const;
154
155 /// Store and log the state after any iteration of the optimization.
156 void logIteration(const State& state) const;
157};
