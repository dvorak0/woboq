26/// Parameters for penalty method
27class GTSAM_EXPORT PenaltyOptimizerParams : public ConstrainedOptimizerParams {
28 public:
29 using Base = ConstrainedOptimizerParams;
30 using This = PenaltyOptimizerParams;
31 using shared_ptr = std::shared_ptr<This>;
32
33 double initialMuEq = 1.0; // initial penalty parameter
34 double initialMuIneq = 1.0; // initial penalty parameter

36 double muIneqIncreaseRate = 2;

44/// Details for each iteration.

47 using Base = ConstrainedOptimizerState;

58/** Penalty method optimizer for solving constrained nonlinear optimization
59 * problems. In each iteration, it solves an unconstrained nonlinear
60 * optimization problem that minimize a merit function. The merit function is
61 * constructed as the sum of the cost function and penalty functions for
62 * constraints.
63 * Example problem:
64 * argmin_x 0.5 * ||x1-1||^2 + 0.5 * ||x2-1||^2
65 * s.t. x1^2 + x2^2 - 1 = 0
66 * s.t. 4*x1^2 + 0.25*x2^2 - 1 <= 0
67 */
68class GTSAM_EXPORT PenaltyOptimizer : public ConstrainedOptimizer {
69 public:
70 using Base = ConstrainedOptimizer;
71 using This = PenaltyOptimizer;
72 using shared_ptr = std::shared_ptr<This>;
73
74 using Params = PenaltyOptimizerParams;
75 using State = PenaltyOptimizerState;
76 using Progress = std::vector<PenaltyOptimizerState>;
77
78 protected:
79 Params::shared_ptr p_;
80 mutable Progress progress_;
81
82 public:
83 /// Constructor.
84 PenaltyOptimizer(const ConstrainedOptProblem& problem,
85 const Values& initialValues,
86 Params::shared_ptr p = std::make_shared<Params>())
87 : Base(problem, initialValues), p_(p), progress_() {}
88
89 /// Constructor that packs costs and constraints into a single factor graph.
90 PenaltyOptimizer(const NonlinearFactorGraph& graph,
91 const Values& initialValues,
92 Params::shared_ptr p = std::make_shared<Params>())
93 : Base(graph, initialValues), p_(p), progress_() {}
94
95 /// Run optimization with equality constraints only.
96 Values optimize() const override;
97
98 State iterate(const State& state) const;
99
100 /// Return progress of iterations.
101 const Progress& progress() const { return progress_; }
102
103 protected:
104 /// Merit function in the form
105 /// m(x) = f(x) + 0.5 muEq * ||h(x)||^2 + 0.5 * muIneq * ||g(x)_+||^2
106 NonlinearFactorGraph meritFunction(const double muEq, const double muIneq) const;
107
108 /// Create an unconstrained optimizer that solves the merit function (cost +
109 /// penalty functions).
110 SharedOptimizer createUnconstrainedOptimizer(
111 const NonlinearFactorGraph& graph, const Values& values) const;
112
113 /// Store and log the initial state of the optimization.
114 void logInitialState(const State& state) const;
115
116 /// Store and log the state after any iteration of the optimization.
117 void logIteration(const State& state) const;
118};
