29/// Exponential function e^x.
30[[maybe_unused]] static double exp_func(const double& x,
31 gtsam::OptionalJacobian<1, 1> H1 = {}) {
32 double result = exp(x: x);
33 if (H1) H1->setConstant(result);
34 return result;
35}
36
37/// Exponential expression e^x.
38[[maybe_unused]] static Double_ exp(const Double_& x) {
39 return Double_(exp_func, x);
40}
41
42/// Pow functor used for pow function.
43class PowFunctor {
44 private:
45 double c_;
46
47 public:
48 PowFunctor(const double& c) : c_(c) {}
49
50 double operator()(const double& x,
51 gtsam::OptionalJacobian<1, 1> H1 = {}) const {
52 if (H1) H1->setConstant(c_ * pow(x: x, y: c_ - 1));
53 return pow(x: x, y: c_);
54 }
55};
56
57/// Pow function.
58[[maybe_unused]] static Double_ pow(const Double_& x, const double& c) {
59 PowFunctor pow_functor(c);
60 return Double_(pow_functor, x);
61}
62
63/// Plus between Double expression and double.
64[[maybe_unused]] static Double_ operator+(const Double_& x, const double& d) {
65 return x + Double_(d);
66}
67
68/// Negative sign operator.
69[[maybe_unused]] static Double_ operator-(const Double_& x) {
70 return Double_(0.0) - x;
71}
72
73/// Keys for creating expressions.
74[[maybe_unused]] static Symbol x1_key('x', 1);
75[[maybe_unused]] static Symbol x2_key('x', 2);
76[[maybe_unused]] static Double_ x1(x1_key), x2(x2_key);

81/**
82 * Constrained optimization example in L. Vandenberghe slides:
83 * https://www.seas.ucla.edu/~vandenbe/133B/lectures/nllseq.pdf
84 * f(x) = 0.5*||x1 + e^(-x2)||^2 + 0.5*||x1^2 + 2*x2 + 1||^2
85 * h(x) = x1 + x1^3 + x2 + x2^2 = 0
86 */
87namespace constrained_example1 {
88using namespace constrained_example;
89[[maybe_unused]] static NonlinearFactorGraph Cost() {
90 NonlinearFactorGraph graph;
91 auto f1 = x1 + exp(x: -x2);
92 auto f2 = pow(x: x1, c: 2.0) + 2.0 * x2 + 1.0;
93 auto cost_noise = gtsam::noiseModel::Isotropic::Sigma(dim: 1, sigma: 1.0);
94 graph.add(factorOrContainer: ExpressionFactor<double>(cost_noise, 0., f1));
95 graph.add(factorOrContainer: ExpressionFactor<double>(cost_noise, 0., f2));
96 return graph;
97}
98
99[[maybe_unused]] static NonlinearEqualityConstraints EqConstraints() {
100 NonlinearEqualityConstraints constraints;
101 Vector sigmas = Vector1(1.0);
102 auto h1 = x1 + pow(x: x1, c: 3) + x2 + pow(x: x2, c: 2);
103 constraints.push_back(factor: NonlinearEqualityConstraint::shared_ptr(
104 new ExpressionEqualityConstraint<double>(h1, 0.0, sigmas)));
105 return constraints;
106}
107
108[[maybe_unused]] static Values InitValues() {
109 Values values;
110 values.insert(j: x1_key, val: -0.2);
111 values.insert(j: x2_key, val: -0.2);
112 return values;
113}
114
115[[maybe_unused]] static Values OptimalValues() {
116 Values values;
117 values.insert(j: x1_key, val: 0.0);
118 values.insert(j: x2_key, val: 0.0);
119 return values;
120}
121
122[[maybe_unused]] static NonlinearFactorGraph costs = Cost();
123[[maybe_unused]] static NonlinearEqualityConstraints eqConstraints =
124 EqConstraints();
125[[maybe_unused]] static NonlinearInequalityConstraints ineqConstraints;
126[[maybe_unused]] static Values init_values = InitValues();
127[[maybe_unused]] static ConstrainedOptProblem problem(costs, eqConstraints,
128 ineqConstraints);
129[[maybe_unused]] static Values optimal_values = OptimalValues();
130} // namespace constrained_example1

132/* ************************************************************************* */
133/**
134 * Constrained optimization example with inequality constraints
135 * Approach a point while staying on unit circle, and within an ellipse.
136 * f(x) = 0.5 * ||x1-1||^2 + 0.5 * ||x2-1||^2
137 * h(x) = x1^2 + x2^2 - 1 = 0
138 * g(x) = 4*x1^2 + 0.25*x2^2 - 1 <= 0
139 */
140namespace constrained_example2 {
141using namespace constrained_example;
142[[maybe_unused]] static NonlinearFactorGraph Cost() {
143 NonlinearFactorGraph graph;
144 auto cost_noise = gtsam::noiseModel::Isotropic::Sigma(dim: 1, sigma: 1.0);
145 graph.addPrior(key: x1_key, prior: 1.0, model: cost_noise);
146 graph.addPrior(key: x2_key, prior: 1.0, model: cost_noise);
147 return graph;
148}
149
150[[maybe_unused]] static NonlinearEqualityConstraints EqConstraints() {
151 NonlinearEqualityConstraints constraints;
152 Vector1 sigmas(1.0);
153 Double_ h1 = x1 * x1 + x2 * x2;
154 constraints.emplace_shared<ExpressionEqualityConstraint<double>>(args&: h1, args: 1.0,
155 args&: sigmas);
156 return constraints;
157}
158
159[[maybe_unused]] static NonlinearInequalityConstraints IneqConstraints() {
160 NonlinearInequalityConstraints constraints;
161 Double_ g1 = 4 * x1 * x1 + 0.25 * x2 * x2 - Double_(1.0);
162 double sigma = 1;
163 constraints.emplace_shared<ScalarExpressionInequalityConstraint>(args&: g1, args&: sigma);
164 return constraints;
165}
166
167[[maybe_unused]] static Values InitValues() {
168 Values values;
169 values.insert(j: x1_key, val: -1.0);
170 values.insert(j: x2_key, val: 2.0);
171 return values;
172}
173
174[[maybe_unused]] static Values OptimalValues() {
175 Values values;
176 values.insert(j: x1_key, val: 1.0 / sqrt(x: 5));
177 values.insert(j: x2_key, val: 2.0 / sqrt(x: 5));
178 return values;
179}
180
181[[maybe_unused]] static NonlinearFactorGraph costs = Cost();
182[[maybe_unused]] static NonlinearEqualityConstraints eqConstraints =
183 EqConstraints();
184[[maybe_unused]] static NonlinearInequalityConstraints ineqConstraints =
185 IneqConstraints();
186[[maybe_unused]] static Values init_values = InitValues();
187[[maybe_unused]] static ConstrainedOptProblem problem(costs, eqConstraints,
188 ineqConstraints);
189[[maybe_unused]] static Values optimal_values = OptimalValues();
190
191} // namespace constrained_example2
