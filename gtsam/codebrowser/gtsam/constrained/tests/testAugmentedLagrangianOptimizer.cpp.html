27/* ********************************************************************************************* */
28double EvaluateLagrangeTerm(const std::vector<Vector>& lambdas,
29 const NonlinearEqualityConstraints& constraints,
30 const Values& values) {
31 double s = 0;
32 for (size_t i = 0; i < constraints.size(); i++) {
33 const auto& constraint = constraints.at(i);
34 s += lambdas.at(n: i).dot(other: constraint->whitenedError(c: values));
35 }
36 return s;
37}
38
39/* ********************************************************************************************* */
40double EvaluateLagrangeTerm(const std::vector<double>& lambdas,
41 const NonlinearInequalityConstraints& constraints,
42 const Values& values) {
43 double s = 0;
44 for (size_t i = 0; i < constraints.size(); i++) {
45 const auto& constraint = constraints.at(i);
46 s += lambdas.at(n: i) * constraint->whitenedExpr(x: values)(0);
47 }
48 return s;
49}

51/* ********************************************************************************************* */
52double ComputeBias(const std::vector<Vector>& lambdas, double mu) {
53 double norm_squared = 0;
54 for (const auto& lambda : lambdas) {
55 norm_squared += pow(x: lambda.norm(), y: 2);
56 }
57 return 0.5 / mu * norm_squared;
58}
59
60/* ********************************************************************************************* */
61double ComputeBias(const std::vector<double>& lambdas, double epsilon) {
62 double norm_squared = 0;
63 for (const auto& lambda : lambdas) {
64 norm_squared += pow(x: lambda, y: 2);
65 }
66 return 0.5 / epsilon * norm_squared;
67}

70TEST(AugmentedLagrangian, constrained_example1) {
71 using namespace constrained_example1;
72
73 // Construct optimizer
74 auto params = std::make_shared<AugmentedLagrangianParams>();
75 AugmentedLagrangianOptimizer optimizer(problem, init_values, params);
76
77 AugmentedLagrangianState state;
78 state.lambdaEq.emplace_back(args: Vector1(0.3));
79 state.muEq = 0.2;
80 NonlinearFactorGraph augmentedLagrangian = optimizer.augmentedLagrangianFunction(state);
81
82 const Values& values = init_values;
83 double expected_cost = costs.error(values);
84 double expected_l2_penalty = eqConstraints.penaltyGraph(mu: state.muEq).error(values);
85 double expected_lagrange_term = EvaluateLagrangeTerm(lambdas: state.lambdaEq, constraints: eqConstraints, values);
86 double bias = ComputeBias(lambdas: state.lambdaEq, mu: state.muEq);
87 double expected_error = expected_cost + expected_l2_penalty + expected_lagrange_term + bias;
88 EXPECT_DOUBLES_EQUAL(expected_error, augmentedLagrangian.error(values), 1e-6);
89}

91/* ********************************************************************************************* */
92TEST(AugmentedLagrangian, constrained_example2) {
93 using namespace constrained_example2;
94
95 // Construct optimizer
96 auto params = std::make_shared<AugmentedLagrangianParams>();
97 AugmentedLagrangianOptimizer optimizer(problem, init_values, params);
98
99 AugmentedLagrangianState state;
100 state.lambdaEq.emplace_back(args: Vector1(0.3));
101 state.lambdaIneq.emplace_back(args: -2.0);
102 state.muEq = 0.2;
103 state.muIneq = 0.3;
104 double epsilon = 1.0;
105 NonlinearFactorGraph augmentedLagrangian = optimizer.augmentedLagrangianFunction(state, epsilon);
106
107 const Values& values = init_values;
108 double expected_cost = costs.error(values);
109 double expected_l2_penalty_e = eqConstraints.penaltyGraph(mu: state.muEq).error(values);
110 double expected_lagrange_term_e = EvaluateLagrangeTerm(lambdas: state.lambdaEq, constraints: eqConstraints, values);
111 double bias_e = ComputeBias(lambdas: state.lambdaEq, mu: state.muEq);
112 double expected_penalty_i = ineqConstraints.penaltyGraph(mu: state.muIneq).error(values);
113 double expected_lagrange_term_i = EvaluateLagrangeTerm(lambdas: state.lambdaIneq, constraints: ineqConstraints, values);
114 double bias_i = ComputeBias(lambdas: state.lambdaIneq, epsilon);
115 double expected_error = expected_cost + expected_l2_penalty_e + expected_penalty_i +
116 expected_lagrange_term_e + expected_lagrange_term_i + bias_e + bias_i;
117
118 EXPECT_DOUBLES_EQUAL(expected_error, augmentedLagrangian.error(values), 1e-6);
119}
