28TEST(RampFunction, error_and_jacobian) {
29 /// Helper function for numerical Jacobian computation.
30 auto ramp_helper = [&](const double& x) { return RampFunction::Ramp(x); };
31
32 /// Create a set of values to test the function.
33 static std::vector<double> x_vec{-3.0, 0.0, 1.0, 2.0, 3.0};
34 static std::vector<double> expected_r_vec{0.0, 0.0, 1.0, 2.0, 3.0};
35
36 for (size_t i = 0; i < x_vec.size(); i++) {
37 double x = x_vec.at(n: i);
38 Matrix H;
39 double r = RampFunction::Ramp(x, H);
40
41 /// Check function evaluation.
42 EXPECT_DOUBLES_EQUAL(expected_r_vec.at(i), r, 1e-9);
43
44 /// Check derivative.
45 if (abs(x: x) > 1e-6) { // function is not smooth at 0, so Jacobian is undefined.
46 Matrix expected_H = gtsam::numericalDerivative11<double, double, 1>(h: ramp_helper, x, delta: 1e-6);
47 EXPECT(assert_equal(expected_H, H));
48 }
49 }
50}
