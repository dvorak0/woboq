42/** These typedefs and aliases will help with making the evaluateError interface
43 * independent of boost
44 * TODO(kartikarcot): Change this to OptionalMatrixNone
45 * This typedef is used to indicate that the Jacobian is not required
46 * and the default value used for optional matrix pointer arguments in evaluateError.
47 * Had to use the static_cast of a nullptr, because the compiler is not able to
48 * deduce the type of the nullptr when expanding the evaluateError templates.
49 */
50#define OptionalNone static_cast<gtsam::Matrix*>(nullptr)
51
52/** This typedef will be used everywhere boost::optional<Matrix&> reference was used
53 * previously. This is used to indicate that the Jacobian is optional. In the future
54 * we will change this to OptionalJacobian
55 */
56using OptionalMatrixType = Matrix*;
57
58/** The OptionalMatrixVecType is a pointer to a vector of matrices. It will
59 * be used in situations where a vector of matrices is optional, like in
60 * unwhitenedError.

62using OptionalMatrixVecType = std::vector<Matrix>*;

64/**
65 * Nonlinear factor base class

67 * \nosubgrouping

69class GTSAM_EXPORT NonlinearFactor: public Factor {
70
71protected:
72

75 typedef NonlinearFactor This;

77public:

79 typedef std::shared_ptr<This> shared_ptr;

81 /// @name Standard Constructors
82 /// @{
83
84 /** Default constructor for I/O only */
85 NonlinearFactor() {}
86
87 /**
88 * Constructor from a collection of the keys involved in this factor
89 */
90 template<typename CONTAINER>
91 NonlinearFactor(const CONTAINER& keys) :
92 Base(keys) {}
93
94 /// @}
95 /// @name Testable
96 /// @{
97
98 /** print */
99 void print(const std::string& s = "", const KeyFormatter& keyFormatter =
100 DefaultKeyFormatter) const override;
101
102 /** Check if two factors are equal */
103 virtual bool equals(const NonlinearFactor& f, double tol = 1e-9) const;
104
105 /// @}
106 /// @name Standard Interface
107 /// @{
108
109 /**
110 * In nonlinear factors, the error function returns the negative log-likelihood
111 * as a non-linear function of the values in a \class Values object.
112 *
113 * The idea is that Gaussian factors have a quadratic error function that locally
114 * approximates the negative log-likelihood, and are obtained by \b linearizing
115 * the nonlinear error function at a given linearization.
116 *
117 * The derived class, \class NoiseModelFactor, adds a noise model to the factor,
118 * and calculates the error by asking the user to implement the method
119 * \code double evaluateError(const Values& c) const \endcode.
120 */
121 virtual double error(const Values& c) const;
122
123 /**
124 * The Factor::error simply extracts the \class Values from the
125 * \class HybridValues and calculates the error.
126 */
127 double error(const HybridValues& c) const override;
128
129 /** get the dimension of the factor (number of rows on linearization) */
130 virtual size_t dim() const = 0;
131
132 /**
133 * Checks whether a factor should be used based on a set of values.
134 * This is primarily used to implement inequality constraints that
135 * require a variable active set. For all others, the default implementation
136 * returning true solves this problem.
137 *
138 * In an inequality/bounding constraint, this active() returns true
139 * when the constraint is *NOT* fulfilled.
140 * @return true if the constraint is active
141 */
142 virtual bool active(const Values& c) const { return true; }
143
144 /** linearize to a GaussianFactor */
145 virtual std::shared_ptr<GaussianFactor>
146 linearize(const Values& c) const = 0;
147
148 /**
149 * Creates a shared_ptr clone of the factor - needs to be specialized to allow
150 * for subclasses
151 *
152 * By default, throws exception if subclass does not implement the function.
153 */
154 virtual shared_ptr clone() const {
155 // TODO: choose better exception to throw here
156 throw std::runtime_error("NonlinearFactor::clone(): Attempting to clone factor with no clone() implemented!");
157 return shared_ptr();
158 }
159
160 /**
161 * Creates a shared_ptr clone of the
162 * factor with different keys using
163 * a map from old->new keys
164 */
165 virtual shared_ptr rekey(const std::map<Key,Key>& rekey_mapping) const;
166
167 /**
168 * Clones a factor and fully replaces its keys
169 * @param new_keys is the full replacement set of keys
170 */
171 virtual shared_ptr rekey(const KeyVector& new_keys) const;
172
173 /**
174 * Should the factor be evaluated in the same thread as the caller
175 * This is to enable factors that has shared states (like the Python GIL lock)
176 */
177 virtual bool sendable() const {
178 return true;
179 }
180
181}; // \class NonlinearFactor

188/**
189 * A nonlinear sum-of-squares factor with a zero-mean noise model
190 * implementing the density \f$ P(z|x) \propto exp -0.5*|z-h(x)|^2_C \f$
191 * Templated on the parameter type X and the values structure Values
192 * There is no return type specified for h(x). Instead, we require
193 * the derived class implements \f$ \mathtt{error\_vector}(x) = h(x)-z \approx A \delta x - b \f$
194 * This allows a graph to have factors with measurements of mixed type.
195
196 * The noise model is typically Gaussian, but robust and constrained error models are also supported.
197 */
198class GTSAM_EXPORT NoiseModelFactor: public NonlinearFactor {
199
200protected:
201
202 // handy typedefs
203 typedef NonlinearFactor Base;
204 typedef NoiseModelFactor This;
205
206 SharedNoiseModel noiseModel_; /** Noise model */
207
208public:
209
210 typedef std::shared_ptr<This> shared_ptr;
211
212 /** Default constructor for I/O only */
213 NoiseModelFactor() {}
214
215 /** Destructor */
216 ~NoiseModelFactor() override {}
217
218 /**
219 * Constructor
220 */
221 template<typename CONTAINER>
222 NoiseModelFactor(const SharedNoiseModel& noiseModel, const CONTAINER& keys) :
223 Base(keys), noiseModel_(noiseModel) {}
224
225protected:
226
227 /**
228 * Constructor - only for subclasses, as this does not set keys.
229 */
230 NoiseModelFactor(const SharedNoiseModel& noiseModel) : noiseModel_(noiseModel) {}
231
232public:
233 /** Print */
234 void print(const std::string& s = "",
235 const KeyFormatter& keyFormatter = DefaultKeyFormatter) const override;
236
237 /** Check if two factors are equal */
238 bool equals(const NonlinearFactor& f, double tol = 1e-9) const override;
239
240 /** get the dimension of the factor (number of rows on linearization) */
241 size_t dim() const override {
242 return noiseModel_->dim();
243 }
244
245 /// access to the noise model
246 const SharedNoiseModel& noiseModel() const {
247 return noiseModel_;
248 }
249
250 /**
251 * Error function *without* the NoiseModel, \f$ z-h(x) \f$.
252 * Override this method to finish implementing an N-way factor.
253 * If the optional arguments is specified, it should compute
254 * both the function evaluation and its derivative(s) in H.
255 */
256 virtual Vector unwhitenedError(const Values& x, OptionalMatrixVecType H = nullptr) const = 0;
257
258 /** support taking in the actual vector instead of the pointer as well
259 * to get access to this version of the function from derived classes
260 * one will need to use the "using" keyword and specify that like this:
261 * public:
262 * using NoiseModelFactor::unwhitenedError;
263 */
264 Vector unwhitenedError(const Values& x, std::vector<Matrix>& H) const {
265 return unwhitenedError(x, H: &H);
266 }
267
268 /**
269 * Vector of errors, whitened
270 * This is the raw error, i.e., i.e. \f$ (h(x)-z)/\sigma \f$ in case of a Gaussian
271 */
272 Vector whitenedError(const Values& c) const;
273
274 /**
275 * Vector of errors, whitened, but unweighted by any loss function
276 */
277 Vector unweightedWhitenedError(const Values& c) const;
278
279 /**
280 * Compute the effective weight of the factor from the noise model.
281 */
282 double weight(const Values& c) const;
283
284 using NonlinearFactor::error;
285
286 /**
287 * Calculate the error of the factor.
288 * This is the log-likelihood, e.g. \f$ 0.5(h(x)-z)^2/\sigma^2 \f$ in case of Gaussian.
289 * In this class, we take the raw prediction error \f$ h(x)-z \f$, ask the noise model
290 * to transform it to \f$ (h(x)-z)^2/\sigma^2 \f$, and then multiply by 0.5.
291 */
292 double error(const Values& c) const override;
293
294 /**
295 * Linearize a non-linearFactorN to get a GaussianFactor,
296 * \f$ Ax-b \approx h(x+\delta x)-z = h(x) + A \delta x - z \f$
297 * Hence \f$ b = z - h(x) = - \mathtt{error\_vector}(x) \f$
298 */
299 std::shared_ptr<GaussianFactor> linearize(const Values& x) const override;
300
301 /**
302 * Creates a shared_ptr clone of the
303 * factor with a new noise model
304 */
305 shared_ptr cloneWithNewNoiseModel(const SharedNoiseModel newNoise) const;
306
307 private:
308#if GTSAM_ENABLE_BOOST_SERIALIZATION
309 /** Serialization function */
310 friend class boost::serialization::access;
311 template<class ARCHIVE>
312 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
313 ar & boost::serialization::make_nvp(n: "NonlinearFactor",
314 v&: boost::serialization::base_object<Base>(d&: *this));
315 ar & BOOST_SERIALIZATION_NVP(noiseModel_);
316 }
317#endif
318
319}; // \class NoiseModelFactor

381/**
382 * A convenient base class for creating your own NoiseModelFactor
383 * with n variables. To derive from this class, implement evaluateError().
384 *
385 * For example, a 2-way factor that computes the difference in x-translation
386 * between a Pose3 and Point3 could be implemented like so:
387 *
388 * ~~~~~~~~~~~~~~~~~~~~{.cpp}
389 * class MyFactor : public NoiseModelFactorN<Pose3, Point3> {
390 * public:
391 * using Base = NoiseModelFactorN<Pose3, Point3>;
392 *
393 * MyFactor(Key pose_key, Key point_key, const SharedNoiseModel& noiseModel)
394 * : Base(noiseModel, pose_key, point_key) {}
395 *
396 * Vector evaluateError(
397 * const Pose3& T, const Point3& p,
398 * OptionalMatrixType H_T = OptionalNone,
399 * OptionalMatrixType H_p = OptionalNone) const override {
400 * Matrix36 t_H_T; // partial derivative of translation w.r.t. pose T
401 *
402 * // Only compute t_H_T if needed:
403 * Point3 t = T.translation(H_T ? &t_H_T : 0);
404 * double a = t(0); // a_H_t = [1, 0, 0]
405 * double b = p(0); // b_H_p = [1, 0, 0]
406 * double error = a - b; // H_a = 1, H_b = -1
407 *
408 * // H_T = H_a * a_H_t * t_H_T = the first row of t_H_T
409 * if (H_T) *H_T = (Matrix(1, 6) << t_H_T.row(0)).finished();
410 * // H_p = H_b * b_H_p = -1 * [1, 0, 0]
411 * if (H_p) *H_p = (Matrix(1, 3) << -1., 0., 0.).finished();
412 *
413 * return Vector1(error);
414 * }
415 * };
416 *
417 * // Unit Test
418 * TEST(NonlinearFactor, MyFactor) {
419 * MyFactor f(X(1), X(2), noiseModel::Unit::Create(1));
420 * EXPECT_DOUBLES_EQUAL(-8., f.evaluateError(Pose3(), Point3(8., 7., 6.))(0),
421 * 1e-9);
422 * Values values;
423 * values.insert(X(1), Pose3(Rot3::RzRyRx(0.1, 0.2, 0.3), Point3(1, 2, 3)));
424 * values.insert(X(2), Point3(1, 2, 3));
425 * EXPECT_CORRECT_FACTOR_JACOBIANS(f, values, 1e-5, 1e-5);
426 * }
427 * ~~~~~~~~~~~~~~~~~~~~
428 *
429 * These factors are templated on a values structure type. The values structures
430 * are typically more general than just vectors, e.g., Rot3 or Pose3, which are
431 * objects in non-linear manifolds (Lie groups).
432 */
433template <class... ValueTypes>
434class NoiseModelFactorN
435 : public NoiseModelFactor,
436 public detail::NoiseModelFactorAliases<ValueTypes...> {
437 public:
438 /// N is the number of variables (N-way factor)
439 inline constexpr static auto N = sizeof...(ValueTypes);
440
441 using NoiseModelFactor::unwhitenedError;
442
443protected:
444 using Base = NoiseModelFactor;
445 using This = NoiseModelFactorN<ValueTypes...>;
446
447 /// @name SFINAE aliases
448 /// @{
449
450 template <typename From, typename To>
451 using IsConvertible =
452 typename std::enable_if<std::is_convertible<From, To>::value, void>::type;
453
454 template <int I>
455 using IndexIsValid = typename std::enable_if<(I >= 1) && (I <= N),
456 void>::type; // 1-indexed!
457
458 template <typename Container>
459 using ContainerElementType =
460 typename std::decay<decltype(*std::declval<Container>().begin())>::type;
461 template <typename Container>
462 using IsContainerOfKeys = IsConvertible<ContainerElementType<Container>, Key>;
463
464 /** A helper alias to check if a list of args
465 * are all references to a matrix or not. It will be used
466 * to choose the right overload of evaluateError.
467 */
468 template <typename Ret, typename ...Args>
469 using AreAllMatrixRefs = std::enable_if_t<(... &&
470 std::is_convertible<Args, Matrix&>::value), Ret>;
471
472 template<typename Arg>
473 using IsMatrixPointer = std::is_same<typename std::decay_t<Arg>, Matrix*>;
474
475 template<typename Arg>
476 using IsNullpointer = std::is_same<typename std::decay_t<Arg>, std::nullptr_t>;
477
478 /** A helper alias to check if a list of args
479 * are all pointers to a matrix or not. It will be used
480 * to choose the right overload of evaluateError.
481 */
482 template <typename Ret, typename ...Args>
483 using AreAllMatrixPtrs = std::enable_if_t<(... &&
484 (IsMatrixPointer<Args>::value || IsNullpointer<Args>::value)), Ret>;
485
486 /// @}
487
488 /* Like std::void_t, except produces `OptionalMatrixType` instead of
489 * `void`. Used to expand fixed-type parameter-packs with same length as
490 * ValueTypes. */
491 template <typename T = void>
492 using OptionalMatrixTypeT = Matrix*;
493
494 /* Like std::void_t, except produces `Key` instead of `void`. Used to expand
495 * fixed-type parameter-packs with same length as ValueTypes. */
496 template <typename T>
497 using KeyType = Key;
498
499 /* Like std::void_t, except produces `Matrix` instead of
500 * `void`. Used to expand fixed-type parameter-packs with same length as
501 * ValueTypes. This helps in creating an evaluateError overload that accepts
502 * Matrices instead of pointers to matrices */
503 template <typename T = void>
504 using MatrixTypeT = Matrix;
505
506 public:
507 /**
508 * The type of the I'th template param can be obtained as ValueType<I>.
509 * I is 1-indexed for backwards compatibility/consistency! So for example,
510 * ```
511 * using Factor = NoiseModelFactorN<Pose3, Point3>;
512 * Factor::ValueType<1> // Pose3
513 * Factor::ValueType<2> // Point3
514 * // Factor::ValueType<0> // ERROR! Will not compile.
515 * // Factor::ValueType<3> // ERROR! Will not compile.
516 * ```
517 *
518 * You can also use the shortcuts `X1`, ..., `X6` which are the same as
519 * `ValueType<1>`, ..., `ValueType<6>` respectively (see
520 * detail::NoiseModelFactorAliases).
521 *
522 * Note that, if your class is templated AND you want to use `ValueType<1>`
523 * inside your class, due to dependent types you need the `template` keyword:
524 * `typename MyFactor<T>::template ValueType<1>`.
525 */
526 template <int I, typename = IndexIsValid<I>>
527 using ValueType =
528 typename std::tuple_element<I - 1, std::tuple<ValueTypes...>>::type;
529
530 public:
531
532 /// @name Constructors
533 /// @{
534
535 /// Default Constructor for I/O
536 NoiseModelFactorN() {}
537
538 /**
539 * Constructor.
540 * Example usage: NoiseModelFactorN(noise, key1, key2, ..., keyN)
541 * @param noiseModel Shared pointer to noise model.
542 * @param keys Keys for the variables in this factor, passed in as separate
543 * arguments.
544 */
545 NoiseModelFactorN(const SharedNoiseModel& noiseModel,
546 KeyType<ValueTypes>... keys)
547 : Base(noiseModel, std::array<Key, N>{keys...}) {}
548
549 /**
550 * Constructor.
551 * Example usage: `NoiseModelFactorN(noise, {key1, key2, ..., keyN})`
552 * Example usage: `NoiseModelFactorN(noise, keys)` where keys is a vector<Key>
553 * @param noiseModel Shared pointer to noise model.
554 * @param keys A container of keys for the variables in this factor.
555 */
556 template <typename CONTAINER = std::initializer_list<Key>,
557 typename = IsContainerOfKeys<CONTAINER>>
558 NoiseModelFactorN(const SharedNoiseModel& noiseModel, CONTAINER keys)
559 : Base(noiseModel, keys) {
560 if (keys.size() != N) {
561 throw std::invalid_argument(
562 "NoiseModelFactorN: wrong number of keys given");
563 }
564 }
565
566 /// @}
567
568 ~NoiseModelFactorN() override {}

570 /** Returns a key. Usage: `key<I>()` returns the I'th key.
571 * I is 1-indexed for backwards compatibility/consistency! So for example,
572 * ```
573 * NoiseModelFactorN<Pose3, Point3> factor(noise, key1, key2);
574 * key<1>() // = key1
575 * key<2>() // = key2
576 * // key<0>() // ERROR! Will not compile
577 * // key<3>() // ERROR! Will not compile
578 * ```
579 *
580 * Note that, if your class is templated AND you are trying to call `key<1>`
581 * inside your class, due to dependent types you need the `template` keyword:
582 * `this->key1()`.
583 */
584 template <int I = 1>
585 inline Key key() const {
586 static_assert(I <= N, "Index out of bounds");
587 return keys_[I - 1];
588 }

590 /// @name NoiseModelFactor methods
591 /// @{
592
593 /** This implements the `unwhitenedError` virtual function by calling the
594 * n-key specific version of evaluateError, which is pure virtual so must be
595 * implemented in the derived class.
596 *
597 * Example usage:
598 * ```
599 * gtsam::Values values;
600 * values.insert(...) // populate values
601 * std::vector<Matrix> Hs(2); // this will be an optional output argument
602 * const Vector error = factor.unwhitenedError(values, Hs);
603 * ```
604 * @param[in] x A Values object containing the values of all the variables
605 * used in this factor
606 * @param[out] H A vector of (dynamic) matrices whose size should be equal to
607 * n. The Jacobians w.r.t. each variable will be output in this parameter.
608 */
609 Vector unwhitenedError(
610 const Values& x,
611 OptionalMatrixVecType H = nullptr) const override {
612 return unwhitenedError(gtsam::index_sequence_for<ValueTypes...>{}, x,
613 H);
614 }
615
616 /// @}
617 /// @name Virtual methods
618 /// @{
619
620 /**
621 * Override `evaluateError` to finish implementing an n-way factor.
622 *
623 * Both the `x` and `H` arguments are written here as parameter packs, but
624 * when overriding this method, you probably want to explicitly write them
625 * out. For example, for a 2-way factor with variable types Pose3 and Point3,
626 * you should implement:
627 * ```
628 * Vector evaluateError(
629 * const Pose3& x1, const Point3& x2,
630 * OptionalMatrixType H1 = OptionalNone,
631 * OptionalMatrixType H2 = OptionalNone) const override { ... }
632 * ```
633 *
634 * If any of the optional Matrix reference arguments are specified, it should
635 * compute both the function evaluation and its derivative(s) in the requested
636 * variables.
637 *
638 * @param x The values of the variables to evaluate the error for. Passed in
639 * as separate arguments.
640 * @param[out] H The Jacobian with respect to each variable (optional).
641 */
642 virtual Vector evaluateError(const ValueTypes&... x,
643 OptionalMatrixTypeT<ValueTypes>... H) const = 0;
644
645 /** If all the optional arguments are matrices then redirect the call to
646 * the one which takes pointers.
647 * To get access to this version of the function from derived classes
648 * one will need to use the "using" keyword and specify that like this:
649 * public:
650 * using NoiseModelFactorN<list the value types here>::evaluateError;
651 */
652 Vector evaluateError(const ValueTypes&... x, MatrixTypeT<ValueTypes>&... H) const {
653 return evaluateError(x..., (&H)...);
654 }
655
656 /// @}
657 /// @name Convenience method overloads
658 /// @{
659
660 /** No-Jacobians requested function overload.
661 * This specializes the version below to avoid recursive calls since this is
662 * commonly used.
663 *
664 * e.g. `const Vector error = factor.evaluateError(pose, point);`
665 */
666 inline Vector evaluateError(const ValueTypes&... x) const {
667 return evaluateError(x..., OptionalMatrixTypeT<ValueTypes>()...);
668 }
669
670 /** Some (but not all) optional Jacobians are omitted (function overload)
671 * and the jacobians are l-value references to matrices.
672 * e.g. `const Vector error = factor.evaluateError(pose, point, Hpose);`
673 */
674 template <typename... OptionalJacArgs, typename = IndexIsValid<sizeof...(OptionalJacArgs) + 1>>
675 inline AreAllMatrixRefs<Vector, OptionalJacArgs...> evaluateError(const ValueTypes&... x,
676 OptionalJacArgs&&... H) const {
677 return evaluateError(x..., (&H)...);
678 }
679
680 /** Some (but not all) optional Jacobians are omitted (function overload)
681 * and the jacobians are pointers to matrices.
682 * e.g. `const Vector error = factor.evaluateError(pose, point, &Hpose);`
683 */
684 template <typename... OptionalJacArgs, typename = IndexIsValid<sizeof...(OptionalJacArgs) + 1>>
685 inline AreAllMatrixPtrs<Vector, OptionalJacArgs...> evaluateError(const ValueTypes&... x,
686 OptionalJacArgs&&... H) const {
687 // If they are pointer version, ensure to cast them all to be Matrix* types
688 // This will ensure any arguments inferred as std::nonetype_t are cast to (Matrix*) nullptr
689 // This guides the compiler to the correct overload which is the one that takes pointers
690 return evaluateError(x...,
691 std::forward<OptionalJacArgs>(H)..., static_cast<OptionalMatrixType>(OptionalNone));
692 }
693
694 /// @}
695
696 private:
697 /** Pack expansion with index_sequence template pattern, used to index into
698 * `keys_` and `H`.
