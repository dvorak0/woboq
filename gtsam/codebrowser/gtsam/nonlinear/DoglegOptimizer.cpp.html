29/* ************************************************************************* */
30DoglegParams::VerbosityDL DoglegParams::verbosityDLTranslator(const std::string &verbosityDL) const {
31 std::string s = verbosityDL;
32 // convert to upper case
33 std::transform(first: s.begin(), last: s.end(), result: s.begin(), unary_op: ::toupper);
34 if (s == "SILENT") return DoglegParams::SILENT;
35 if (s == "VERBOSE") return DoglegParams::VERBOSE;
36
37 /* default is silent */
38 return DoglegParams::SILENT;
39}
40
41/* ************************************************************************* */
42std::string DoglegParams::verbosityDLTranslator(VerbosityDL verbosityDL) const {
43 std::string s;
44 switch (verbosityDL) {
45 case DoglegParams::SILENT: s = "SILENT"; break;
46 case DoglegParams::VERBOSE: s = "VERBOSE"; break;
47 default: s = "UNDEFINED"; break;
48 }
49 return s;
50}

54struct DoglegState : public NonlinearOptimizerState {
55 const double delta;
56
57 DoglegState(const Values& _values, double _error, double _delta, unsigned int _iterations = 0)
58 : NonlinearOptimizerState(_values, _error, _iterations), delta(_delta) {}
59};
60}

72DoglegOptimizer::DoglegOptimizer(const NonlinearFactorGraph& graph, const Values& initialValues,
73 const Ordering& ordering)
74 : NonlinearOptimizer(graph, std::unique_ptr<State>(

79double DoglegOptimizer::getDelta() const {
80 return static_cast<const State*>(state_.get())->delta;
81}

83/* ************************************************************************* */
84GaussianFactorGraph::shared_ptr DoglegOptimizer::iterate(void) {
85
86 // Linearize graph
87 GaussianFactorGraph::shared_ptr linear = graph_.linearize(linearizationPoint: state_->values);
88
89 // Pull out parameters we'll use
90 const bool dlVerbose = (params_.verbosityDL > DoglegParams::SILENT);
91
92 // Do Dogleg iteration with either Multifrontal or Sequential elimination
93 DoglegOptimizerImpl::IterationResult result;
94
95 if ( params_.isMultifrontal() ) {
96 GaussianBayesTree bt = *linear->eliminateMultifrontal(ordering: *params_.ordering, function: params_.getEliminationFunction());
97 VectorValues dx_u = bt.optimizeGradientSearch();
98 VectorValues dx_n = bt.optimize();
99 result = DoglegOptimizerImpl::Iterate(delta: getDelta(), mode: DoglegOptimizerImpl::ONE_STEP_PER_ITERATION,
100 dx_u, dx_n, Rd: bt, f: graph_, x0: state_->values, f_error: state_->error, verbose: dlVerbose);
101 }
102 else if ( params_.isSequential() ) {
103 GaussianBayesNet bn = *linear->eliminateSequential(ordering: *params_.ordering, function: params_.getEliminationFunction());
104 VectorValues dx_u = bn.optimizeGradientSearch();
105 VectorValues dx_n = bn.optimize();
106 result = DoglegOptimizerImpl::Iterate(delta: getDelta(), mode: DoglegOptimizerImpl::ONE_STEP_PER_ITERATION,
107 dx_u, dx_n, Rd: bn, f: graph_, x0: state_->values, f_error: state_->error, verbose: dlVerbose);
108 }
109 else if ( params_.isIterative() ) {
110 throw std::runtime_error("Dogleg is not currently compatible with the linear conjugate gradient solver");
111 }
112 else {
113 throw std::runtime_error("Optimization parameter is invalid: DoglegParams::elimination");
114 }
115
116 // Maybe show output
117 if(params_.verbosity >= NonlinearOptimizerParams::DELTA) result.dx_d.print(str: "delta");
118
119 // Create new state with new values and new error
120 state_.reset(p: new State(state_->values.retract(delta: result.dx_d), result.f_error, result.delta,
121 state_->iterations + 1));
122 return linear;
123}

125/* ************************************************************************* */
126DoglegParams DoglegOptimizer::ensureHasOrdering(DoglegParams params, const NonlinearFactorGraph& graph) const {
127 if (!params.ordering)
128 params.ordering = Ordering::Create(orderingType: params.orderingType, graph);
129 return params;
130}
