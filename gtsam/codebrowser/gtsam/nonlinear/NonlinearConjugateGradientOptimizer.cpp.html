29typedef internal::NonlinearOptimizerState State;
30
31/**
32 * @brief Return the gradient vector of a nonlinear factor graph
33 * @param nfg the graph
34 * @param values a linearization point
35 * Can be moved to NonlinearFactorGraph.h if desired
36 */
37static VectorValues gradientInPlace(const NonlinearFactorGraph& nfg,
38 const Values& values) {
39 // Linearize graph
40 GaussianFactorGraph::shared_ptr linear = nfg.linearize(linearizationPoint: values);
41 return linear->gradientAtZero();
42}
43
44NonlinearConjugateGradientOptimizer::NonlinearConjugateGradientOptimizer(
45 const NonlinearFactorGraph& graph, const Values& initialValues,
46 const Parameters& params, const DirectionMethod& directionMethod)
47 : Base(graph, std::unique_ptr<State>(
48 new State(initialValues, graph.error(values: initialValues)))),
49 params_(params) {}
50
51double NonlinearConjugateGradientOptimizer::System::error(
52 const State& state) const {
53 return graph_.error(values: state);
54}
55
56NonlinearConjugateGradientOptimizer::System::Gradient
57NonlinearConjugateGradientOptimizer::System::gradient(
58 const State& state) const {
59 return gradientInPlace(nfg: graph_, values: state);
60}
61
62NonlinearConjugateGradientOptimizer::System::State
63NonlinearConjugateGradientOptimizer::System::advance(const State& current,
64 const double alpha,
65 const Gradient& g) const {
66 Gradient step = g;
67 step *= alpha;
68 return current.retract(delta: step);
69}
70
71GaussianFactorGraph::shared_ptr NonlinearConjugateGradientOptimizer::iterate() {
72 const auto [newValues, dummy] = nonlinearConjugateGradient<System, Values>(
73 system: System(graph_), initial: state_->values, params: params_, singleIteration: true /* single iteration */,
74 directionMethod: directionMethod_);
75 state_.reset(
76 p: new State(newValues, graph_.error(values: newValues), state_->iterations + 1));
77
78 // NOTE(frank): We don't linearize this system, so we must return null here.
79 return nullptr;
80}
81
82const Values& NonlinearConjugateGradientOptimizer::optimize() {
83 // Optimize until convergence
84 System system(graph_);
85 const auto [newValues, iterations] = nonlinearConjugateGradient(
86 system, initial: state_->values, params: params_, singleIteration: false, directionMethod: directionMethod_);
87 state_.reset(
88 p: new State(std::move(newValues), graph_.error(values: newValues), iterations));
89 return state_->values;
90}
