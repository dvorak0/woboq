12/**
13 * @file FixedLagSmoother.h
14 * @brief Base class for a fixed-lag smoother. This mimics the basic interface to iSAM2.
15 *
16 * @author Stephen Williams
17 * @date Feb 27, 2013
18 */
19
20// \callgraph
21#pragma once
22
23#include <gtsam/dllexport.h>
24#include <gtsam/inference/Key.h>
25#include <gtsam/nonlinear/NonlinearFactorGraph.h>
26#include <gtsam/nonlinear/Values.h>
27
28#include <map>
29#include <vector>
30
31namespace gtsam {
32
33class GTSAM_EXPORT FixedLagSmoother {
34
35public:
36
37 /// Typedef for a shared pointer to an Incremental Fixed-Lag Smoother
38 typedef std::shared_ptr<FixedLagSmoother> shared_ptr;
39
40 /// Typedef for a Key-Timestamp map/database
41 typedef std::map<Key, double> KeyTimestampMap;
42 typedef std::multimap<double, Key> TimestampKeyMap;
43
44 /**
45 * Meta information returned about the update
46 */
47 // TODO: Think of some more things to put here
48 struct Result {
49 size_t iterations; ///< The number of optimizer iterations performed
50 size_t intermediateSteps; ///< The number of intermediate steps performed within the optimization. For L-M, this is the number of lambdas tried.
51 size_t nonlinearVariables; ///< The number of variables that can be relinearized
52 size_t linearVariables; ///< The number of variables that must keep a constant linearization point
53 double error; ///< The final factor graph error
54 Result() : iterations(0), intermediateSteps(0), nonlinearVariables(0), linearVariables(0), error(0) {}
55
56 /// Getter methods
57 size_t getIterations() const { return iterations; }
58 size_t getIntermediateSteps() const { return intermediateSteps; }
59 size_t getNonlinearVariables() const { return nonlinearVariables; }
60 size_t getLinearVariables() const { return linearVariables; }
61 double getError() const { return error; }
62 void print() const;
63 };
64
65 /** default constructor */
66 FixedLagSmoother(double smootherLag = 0.0) : smootherLag_(smootherLag) { }
67
68 /** destructor */
69 virtual ~FixedLagSmoother() { }
70
71 /** Print the factor for debugging and testing (implementing Testable) */
72 virtual void print(
73 const std::string& s = "FixedLagSmoother:\n",
74 const KeyFormatter& keyFormatter = DefaultKeyFormatter) const;
75
76 /** Check if two IncrementalFixedLagSmoother Objects are equal */
77 virtual bool equals(const FixedLagSmoother& rhs, double tol = 1e-9) const;
78
79 /** read the current smoother lag */
80 double smootherLag() const {
81 return smootherLag_;
82 }
83
84 /** write to the current smoother lag */
85 double& smootherLag() {
86 return smootherLag_;
87 }
88
89 /** Access the current set of timestamps associated with each variable */
90 const KeyTimestampMap& timestamps() const {
91 return keyTimestampMap_;
92 }
93
94 /** Add new factors, updating the solution and relinearizing as needed. */
95 virtual Result update(const NonlinearFactorGraph& newFactors = NonlinearFactorGraph(),
96 const Values& newTheta = Values(),
97 const KeyTimestampMap& timestamps = KeyTimestampMap(),
98 const FactorIndices& factorsToRemove = FactorIndices()) = 0;
99
100 /** Compute an estimate from the incomplete linear delta computed during the last update.
101 * This delta is incomplete because it was not updated below wildfire_threshold. If only
102 * a single variable is needed, it is faster to call calculateEstimate(const KEY&).
103 */
104 virtual Values calculateEstimate() const = 0;
105
106
107protected:
108
109 /** The length of the smoother lag. Any variable older than this amount will be marginalized out. */
110 double smootherLag_;
111
112 /** The current timestamp associated with each tracked key */
113 TimestampKeyMap timestampKeyMap_;
114 KeyTimestampMap keyTimestampMap_;
115
116 /** Update the Timestamps associated with the keys */
117 void updateKeyTimestampMap(const KeyTimestampMap& newTimestamps);
118
119 /** Erase keys from the Key-Timestamps database */
120 void eraseKeyTimestampMap(const KeyVector& keys);
121
122 /** Find the most recent timestamp of the system */
123 double getCurrentTimestamp() const;
124
125 /** Find all of the keys associated with timestamps before the provided time */
126 KeyVector findKeysBefore(double timestamp) const;
127
128 /** Find all of the keys associated with timestamps before the provided time */
129 KeyVector findKeysAfter(double timestamp) const;
130
131}; // FixedLagSmoother
132
133/// Typedef for matlab wrapping
134typedef FixedLagSmoother::KeyTimestampMap FixedLagSmootherKeyTimestampMap;
135typedef FixedLagSmootherKeyTimestampMap::value_type FixedLagSmootherKeyTimestampMapValue;
136typedef FixedLagSmoother::Result FixedLagSmootherResult;
137
138} /// namespace gtsam
