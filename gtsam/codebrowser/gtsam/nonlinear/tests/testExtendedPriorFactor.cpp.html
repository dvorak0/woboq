45TEST(ExtendedPriorFactor, Error) {
46 Key key(1);
47 Pose2 origin(1, 2, 0.3);
48 auto model = noiseModel::Isotropic::Sigma(dim: 3, sigma: 0.5);
49 ExtendedPriorFactor<Pose2> factor(key, origin, model);
50
51 Pose2 x(1.1, 2.2, 0.3);
52 Vector expected_error =
53 (Vector(3) << 0.15463769, 0.161515277, 0.0).finished();
54 Vector actual_error = factor.evaluateError(x);
55 EXPECT(assert_equal(expected_error, actual_error, 1e-8));
56}

58//******************************************************************************
59TEST(ExtendedPriorFactor, ErrorWithMean) {
60 Key key(1);

65
66 Pose2 x(1.1, 2.2, 0.3);

74TEST(ExtendedPriorFactor, Likelihood) {
75 Key key(1);
76 Pose2 origin(1, 2, 0.3);
77 auto model = noiseModel::Isotropic::Sigma(dim: 3, sigma: 1.0);
78 Vector mean = (Vector(3) << 0.1, 0.2, 0.05).finished();
79 ExtendedPriorFactor<Pose2> factor(key, origin, mean, model);
80
81 Pose2 x = origin.retract(v: mean);
82 double expected_likelihood = 1.0;
83 double actual_likelihood = factor.likelihood(x);
84 EXPECT_DOUBLES_EQUAL(expected_likelihood, actual_likelihood, 1e-4);
85}
