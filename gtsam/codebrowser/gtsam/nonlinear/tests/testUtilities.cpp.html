32TEST(Utilities, ExtractPoint2) {
33 Point2 p0(0, 0), p1(1, 0);
34 Values values;
35 values.insert<Point2>(j: L(j: 0), val: p0);
36 values.insert<Point2>(j: L(j: 1), val: p1);
37 values.insert<Rot3>(j: R(j: 0), val: Rot3());
38 values.insert<Pose3>(j: X(j: 0), val: Pose3());
39
40 Matrix all_points = utilities::extractPoint2(values);
41 EXPECT_LONGS_EQUAL(2, all_points.rows());
42}

58TEST(Utilities, ExtractVector) {
59 // Test normal case with 3 vectors and 1 non-vector (ignore non-vector)
60 auto values = Values();
61 values.insert(j: X(j: 0), val: (Vector(4) << 1, 2, 3, 4).finished());
62 values.insert(j: X(j: 2), val: (Vector(4) << 13, 14, 15, 16).finished());
63 values.insert(j: X(j: 1), val: (Vector(4) << 6, 7, 8, 9).finished());
64 values.insert(j: X(j: 3), val: Pose3());
65 auto actual = utilities::extractVectors(values, c: 'x');
66 auto expected =
67 (Matrix(3, 4) << 1, 2, 3, 4, 6, 7, 8, 9, 13, 14, 15, 16).finished();
68 EXPECT(assert_equal(expected, actual));
69
70 // Check that mis-sized vectors fail
71 values.insert(j: X(j: 4), val: (Vector(2) << 1, 2).finished());
72 THROWS_EXCEPTION(utilities::extractVectors(values, 'x'));
73 values.update(j: X(j: 4), val: (Vector(6) << 1, 2, 3, 4, 5, 6).finished());
74 THROWS_EXCEPTION(utilities::extractVectors(values, 'x'));
75}
