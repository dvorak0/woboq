45/* ************************************************************************* */
46bool check_smoother(const NonlinearFactorGraph& fullgraph,
47 const Values& fullinit,
48 const IncrementalFixedLagSmoother& smoother,
49 const Key& key) {
50 GaussianFactorGraph linearized = *fullgraph.linearize(linearizationPoint: fullinit);
51 VectorValues delta = linearized.optimize();
52 Values fullfinal = fullinit.retract(delta);
53
54 Point2 expected = fullfinal.at<Point2>(j: key);
55 Point2 actual = smoother.calculateEstimate<Point2>(key);
56
57 return assert_equal(expected, actual);
58}

60/* ************************************************************************* */
61void PrintSymbolicTreeHelper(const ISAM2Clique::shared_ptr& clique,
62 const std::string indent = "") {
63 // Print the current clique
64 std::cout << indent << "P( ";
65 for (Key key : clique->conditional()->frontals()) {
66 std::cout << DefaultKeyFormatter(key) << " ";
67 }
68 if (clique->conditional()->nrParents() > 0) std::cout << "| ";
69 for (Key key : clique->conditional()->parents()) {
70 std::cout << DefaultKeyFormatter(key) << " ";
71 }
72 std::cout << ")" << std::endl;
73
74 // Recursively print all of the children
75 for (const ISAM2Clique::shared_ptr& child : clique->children) {
76 PrintSymbolicTreeHelper(clique: child, indent: indent + " ");
77 }
78}

80/* ************************************************************************* */
81void PrintSymbolicTree(const ISAM2& isam, const std::string& label) {
82 std::cout << label << std::endl;
83 if (!isam.roots().empty()) {
84 for (const ISAM2::sharedClique& root : isam.roots()) {
85 PrintSymbolicTreeHelper(clique: root);
86 }
87 } else
88 std::cout << "{Empty Tree}" << std::endl;
89}

92TEST(IncrementalFixedLagSmoother, Example) {
93 // Test the IncrementalFixedLagSmoother in a pure linear environment. Thus,
94 // full optimization and the IncrementalFixedLagSmoother should be identical
95 // (even with the linearized approximations at the end of the smoothing lag)
96
97 SETDEBUG("IncrementalFixedLagSmoother update", true);
98
99 // Set up parameters
100 SharedDiagonal odoNoise = noiseModel::Diagonal::Sigmas(sigmas: Vector2(0.1, 0.1));
101 SharedDiagonal loopNoise = noiseModel::Diagonal::Sigmas(sigmas: Vector2(0.1, 0.1));
102
103 // Create a Fixed-Lag Smoother
104 typedef IncrementalFixedLagSmoother::KeyTimestampMap Timestamps;
105 IncrementalFixedLagSmoother smoother(12.0, ISAM2Params());
106
107 // Create containers to keep the full graph
108 Values fullinit;
109 NonlinearFactorGraph fullgraph;
110
111 // i keeps track of the time step
112 size_t i = 0;
113
114 // Add a prior at time 0 and update the HMF
115 {
116 Key key0 = X(j: 0);
117
118 NonlinearFactorGraph newFactors;
119 Values newValues;
120 Timestamps newTimestamps;
121
122 newFactors.addPrior(key: key0, prior: Point2(0.0, 0.0), model: odoNoise);
123 newValues.insert(j: key0, val: Point2(0.01, 0.01));
124 newTimestamps[key0] = 0.0;
125
126 fullgraph.push_back(container: newFactors);
127 fullinit.insert(values: newValues);
128
129 // Update the smoother
130 smoother.update(newFactors, newTheta: newValues, timestamps: newTimestamps);
131
132 // Check
133 CHECK(check_smoother(fullgraph, fullinit, smoother, key0));
134
135 ++i;
136 }
137
138 // Add odometry from time 0 to time 5
139 while (i <= 5) {
140 Key key1 = X(j: i - 1);
141 Key key2 = X(j: i);
142
143 NonlinearFactorGraph newFactors;
144 Values newValues;
145 Timestamps newTimestamps;
146
147 newFactors.emplace_shared<BetweenPoint2>(args&: key1, args&: key2, args: Point2(1.0, 0.0),
148 args&: odoNoise);
149 newValues.insert(j: key2, val: Point2(double(i) + 0.1, -0.1));
150 newTimestamps[key2] = double(i);
151
152 fullgraph.push_back(container: newFactors);
153 fullinit.insert(values: newValues);
154
155 // Update the smoother
156 smoother.update(newFactors, newTheta: newValues, timestamps: newTimestamps);
157
158 // Check
159 CHECK(check_smoother(fullgraph, fullinit, smoother, key2));
160
161 ++i;
162 }
163
164 // Add odometry from time 5 to 6 to the HMF and a loop closure at time 5 to
165 // the TSM
166 {
167 // Add the odometry factor to the HMF
168 Key key1 = X(j: i - 1);
169 Key key2 = X(j: i);
170
171 NonlinearFactorGraph newFactors;
172 Values newValues;
173 Timestamps newTimestamps;
174
175 newFactors.emplace_shared<BetweenPoint2>(args&: key1, args&: key2, args: Point2(1.0, 0.0),
176 args&: odoNoise);
177 newFactors.emplace_shared<BetweenPoint2>(args: X(j: 2), args: X(j: 5), args: Point2(3.5, 0.0),
178 args&: loopNoise);
179 newValues.insert(j: key2, val: Point2(double(i) + 0.1, -0.1));
180 newTimestamps[key2] = double(i);
181
182 fullgraph.push_back(container: newFactors);
183 fullinit.insert(values: newValues);
184
185 // Update the smoother
186 smoother.update(newFactors, newTheta: newValues, timestamps: newTimestamps);
187
188 // Check
189 CHECK(check_smoother(fullgraph, fullinit, smoother, key2));
190
191 ++i;
192 }
193
194 // Add odometry from time 6 to time 15
195 while (i <= 15) {
196 Key key1 = X(j: i - 1);
197 Key key2 = X(j: i);
198
199 NonlinearFactorGraph newFactors;
200 Values newValues;
201 Timestamps newTimestamps;
202
203 // Add the odometry factor twice to ensure the removeFactor test below
204 // works, where we need to keep the connectivity of the graph.
205 newFactors.emplace_shared<BetweenPoint2>(args&: key1, args&: key2, args: Point2(1.0, 0.0),
206 args&: odoNoise);
207 newFactors.emplace_shared<BetweenPoint2>(args&: key1, args&: key2, args: Point2(1.0, 0.0),
208 args&: odoNoise);
209 newValues.insert(j: key2, val: Point2(double(i) + 0.1, -0.1));
210 newTimestamps[key2] = double(i);
211
212 fullgraph.push_back(container: newFactors);
213 fullinit.insert(values: newValues);
214
215 // Update the smoother
216 smoother.update(newFactors, newTheta: newValues, timestamps: newTimestamps);
217
218 // Check
219 CHECK(check_smoother(fullgraph, fullinit, smoother, key2));
220
221 ++i;
222 }
223
224 // add/remove an extra factor
225 {
226 Key key1 = X(j: i - 1);
227 Key key2 = X(j: i);
228
229 NonlinearFactorGraph newFactors;
230 Values newValues;
231 Timestamps newTimestamps;
232
233 // add 2 odometry factors
234 newFactors.emplace_shared<BetweenPoint2>(args&: key1, args&: key2, args: Point2(1.0, 0.0),
235 args&: odoNoise);
236 newFactors.emplace_shared<BetweenPoint2>(args&: key1, args&: key2, args: Point2(1.0, 0.0),
237 args&: odoNoise);
238 newValues.insert(j: key2, val: Point2(double(i) + 0.1, -0.1));
239 newTimestamps[key2] = double(i);
240 ++i;
241
242 fullgraph.push_back(container: newFactors);
243 fullinit.insert(values: newValues);
244
245 // Update the smoother
246 smoother.update(newFactors, newTheta: newValues, timestamps: newTimestamps);
247
248 // Check
249 CHECK(check_smoother(fullgraph, fullinit, smoother, key2));
250
251 // now remove one of the two and try again
252 // empty values and new factors for fake update in which we only remove
253 // factors
254 NonlinearFactorGraph emptyNewFactors;
255 Values emptyNewValues;
256 Timestamps emptyNewTimestamps;
257
258 size_t factorIndex =
259 25; // any index that does not break connectivity of the graph
260 FactorIndices factorToRemove;
261 factorToRemove.push_back(x: factorIndex);
262
263 const NonlinearFactorGraph smootherFactorsBeforeRemove =
264 smoother.getFactors();
265
266 std::cout << "fullgraph.size() = " << fullgraph.size() << std::endl;
267 std::cout << "smootherFactorsBeforeRemove.size() = "
268 << smootherFactorsBeforeRemove.size() << std::endl;
269
270 // remove factor
271 smoother.update(newFactors: emptyNewFactors, newTheta: emptyNewValues, timestamps: emptyNewTimestamps,
272 factorsToRemove: factorToRemove);
273
274 // Note: the following test (checking that the number of factor is reduced
275 // by 1) fails since we are not reusing slots, hence also when removing a
276 // factor we do not change the size of the factor graph size_t
277 // nrFactorsAfterRemoval = smoother.getFactors().size();
278 // DOUBLES_EQUAL(nrFactorsBeforeRemoval-1, nrFactorsAfterRemoval, 1e-5);
279
280 // check that the factors in the smoother are right
281 NonlinearFactorGraph actual = smoother.getFactors();
282 for (size_t i = 0; i < smootherFactorsBeforeRemove.size(); i++) {
283 // check that the factors that were not removed are there
284 if (smootherFactorsBeforeRemove[i] && i != factorIndex) {
285 EXPECT(smootherFactorsBeforeRemove[i]->equals(*actual[i]));
286 } else { // while the factors that were not there or were removed are no
287 // longer there
288 EXPECT(!actual[i]);
289 }
290 }
291 }
292
293 {
294 SETDEBUG("BayesTreeMarginalizationHelper", true);
295 PrintSymbolicTree(isam: smoother.getISAM2(),
296 label: "Bayes Tree Before marginalization test:");
297
298 // Do pressure test on marginalization. Enlarge max_i to enhance the test.
299 const int max_i = 500;
300 while (i <= max_i) {
301 Key key_0 = X(j: i);
302 Key key_1 = X(j: i - 1);
303 Key key_2 = X(j: i - 2);
304 Key key_3 = X(j: i - 3);
305 Key key_4 = X(j: i - 4);
306 Key key_5 = X(j: i - 5);
307 Key key_6 = X(j: i - 6);
308 Key key_7 = X(j: i - 7);
309 Key key_8 = X(j: i - 8);
310 Key key_9 = X(j: i - 9);
311 Key key_10 = X(j: i - 10);
312
313 NonlinearFactorGraph newFactors;
314 Values newValues;
315 Timestamps newTimestamps;
316
317 // To make a complex graph
318 const Point2 z(1.0, 0.0);
319 newFactors.emplace_shared<BetweenPoint2>(args&: key_1, args&: key_0, args: z, args&: odoNoise);
320 if (i % 2 == 0)
321 newFactors.emplace_shared<BetweenPoint2>(args&: key_2, args&: key_1, args: z, args&: odoNoise);
322 if (i % 3 == 0)
323 newFactors.emplace_shared<BetweenPoint2>(args&: key_3, args&: key_2, args: z, args&: odoNoise);
324 if (i % 4 == 0)
325 newFactors.emplace_shared<BetweenPoint2>(args&: key_4, args&: key_3, args: z, args&: odoNoise);
326 if (i % 5 == 0)
327 newFactors.emplace_shared<BetweenPoint2>(args&: key_5, args&: key_4, args: z, args&: odoNoise);
328 if (i % 6 == 0)
329 newFactors.emplace_shared<BetweenPoint2>(args&: key_6, args&: key_5, args: z, args&: odoNoise);
330 if (i % 7 == 0)
331 newFactors.emplace_shared<BetweenPoint2>(args&: key_7, args&: key_6, args: z, args&: odoNoise);
332 if (i % 8 == 0)
333 newFactors.emplace_shared<BetweenPoint2>(args&: key_8, args&: key_7, args: z, args&: odoNoise);
334 if (i % 9 == 0)
335 newFactors.emplace_shared<BetweenPoint2>(args&: key_9, args&: key_8, args: z, args&: odoNoise);
336 if (i % 10 == 0)
337 newFactors.emplace_shared<BetweenPoint2>(args&: key_10, args&: key_9, args: z, args&: odoNoise);
338
339 newValues.insert(j: key_0, val: Point2(double(i) + 0.1, -0.1));
340 newTimestamps[key_0] = double(i);
341
342 fullgraph.push_back(container: newFactors);
343 fullinit.insert(values: newValues);
344
345 // Update the smoother
346 smoother.update(newFactors, newTheta: newValues, timestamps: newTimestamps);
347
348 // Check
349 CHECK(check_smoother(fullgraph, fullinit, smoother, key_0));
350 PrintSymbolicTree(
351 isam: smoother.getISAM2(),
352 label: "Bayes Tree marginalization test: i = " + std::to_string(val: i));
353
354 ++i;
355 }
356 }
