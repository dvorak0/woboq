94 EXPECT(equalsObj(pc));

109/**
110 * Test deserializing from a known serialization generated by code from commit
111 * 0af17f438f62f4788f3a572ecd36d06d224fd1e1 (>4.2a7)
112 * We only test that deserialization matches since
113 * (1) that's the main backward compatibility requirement and
114 * (2) serialized string depends on boost version
115 * Also note: we don't run this test when quaternions or TBB are enabled since
116 * serialization structures are different and the serialized strings/xml are
117 * hard-coded in this test.
118 */
119TEST(Serialization, NoiseModelFactor1_backwards_compatibility) {
120 PriorFactor<Pose3> factor(
121 12345, Pose3(Rot3::RzRyRx(xyz: Vector3(1., 2., 3.)), Point3(4., 5., 6.)),
122 noiseModel::Unit::Create(dim: 6));
123
124 // roundtrip (sanity check)
125 PriorFactor<Pose3> factor_deserialized_str_2 = PriorFactor<Pose3>();
126 roundtrip(input: factor, output&: factor_deserialized_str_2);
127 EXPECT(assert_equal(factor, factor_deserialized_str_2));
128
129#if !defined(GTSAM_USE_QUATERNIONS) && !defined(GTSAM_USE_TBB)

136 "3 0 0 0 0 6 0 1 0 0 0 6 1.00000000000000000e+00 1.00000000000000000e+00 "

138 "1.00000000000000000e+00 6 1.00000000000000000e+00 "
139 "1.00000000000000000e+00 1.00000000000000000e+00 1.00000000000000000e+00 "

173TEST(Serialization, ISAM2) {
174 gtsam::ISAM2Params parameters;
175 gtsam::ISAM2 solver(parameters);
176 gtsam::NonlinearFactorGraph graph;
177 gtsam::Values initialValues;
178 initialValues.clear();
179
180 gtsam::Vector6 temp6;
181 for (int i = 0; i < 6; ++i) {
182 temp6[i] = 0.0001;
183 }
184 gtsam::noiseModel::Diagonal::shared_ptr noiseModel = gtsam::noiseModel::Diagonal::Sigmas(sigmas: temp6);
185
186 gtsam::Pose3 pose0(gtsam::Rot3(), gtsam::Point3(0, 0, 0));
187 gtsam::Symbol symbol0('x', 0);
188 graph.add(factorOrContainer: gtsam::PriorFactor<gtsam::Pose3>(symbol0, pose0, noiseModel));
189 initialValues.insert(j: symbol0, val: pose0);
190
191 solver.update(newFactors: graph, newTheta: initialValues,
192 removeFactorIndices: gtsam::FastVector<gtsam::FactorIndex>());
193
194 std::string binaryPath = "saved_solver.dat";
195 try {
196 std::ofstream outputStream(binaryPath, std::ios::out | std::ios::binary);
197 boost::archive::binary_oarchive outputArchive(outputStream);
198 outputArchive << solver;
199 } catch(...) {
200 EXPECT(false);
201 }
202
203 gtsam::ISAM2 solverFromDisk;
204 try {
205 std::ifstream ifs(binaryPath, std::ios::in | std::ios::binary);
206 boost::archive::binary_iarchive inputArchive(ifs);
207 inputArchive >> solverFromDisk;
208 } catch(...) {
209 EXPECT(false);
210 }
211
212 gtsam::Pose3 p1, p2;
213 try {
214 p1 = solver.calculateEstimate<gtsam::Pose3>(key: symbol0);
215 } catch(std::exception &e) {
216 EXPECT(false);
217 }
218
219 try {
220 p2 = solverFromDisk.calculateEstimate<gtsam::Pose3>(key: symbol0);
221 } catch(std::exception &e) {
222 EXPECT(false);
223 }
224 EXPECT(assert_equal(p1, p2));
225}
