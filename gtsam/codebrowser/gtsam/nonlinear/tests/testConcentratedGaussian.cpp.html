29TEST(ConcentratedGaussian, Pose2) {
30 Key key(1);
31 Pose2 origin(1, 2, 0.3);
32 sharedGaussianNoiseModel model =
33 noiseModel::Diagonal::Sigmas(sigmas: Vector3(0.1, 0.2, 0.3));
34 ConcentratedGaussian<Pose2> factor(key, origin, model);
35
36 // Test error
37 Pose2 x = origin;
38 Values values;
39 values.insert(j: key, val: x);
40 EXPECT_DOUBLES_EQUAL(0.0, factor.error(values), 1e-9);
41
42 // Test logProbability and evaluate
43 double expected_log_prob =
44 -0.5 * 3 * log(x: 2 * M_PI) + log(x: 1.0 / (0.1 * 0.2 * 0.3));
45 EXPECT_DOUBLES_EQUAL(expected_log_prob, factor.logProbability(x), 1e-9);
46 EXPECT_DOUBLES_EQUAL(expected_log_prob, factor.logProbability(values), 1e-9);
47 EXPECT_DOUBLES_EQUAL(exp(expected_log_prob), factor.evaluate(x), 1e-9);
48 EXPECT_DOUBLES_EQUAL(exp(expected_log_prob), factor.evaluate(values), 1e-9);
49}

52TEST(ConcentratedGaussian, Pose2WithMean) {
53 Key key(1);
54 Pose2 origin(1, 2, 0.3);
55 sharedGaussianNoiseModel model =
56 noiseModel::Diagonal::Sigmas(sigmas: Vector3(0.1, 0.2, 0.3));
57
58 Pose2 x = origin;
59 Values values;
60 values.insert(j: key, val: x);
61
62 // Non-zero mean in tangent space (dx, dy, dtheta)
63 Vector3 mean(0.05, -0.02, 0.1);
64 ConcentratedGaussian<Pose2> factor(key, origin, mean, model);
65
66 // Compute expected error = 0.5 * ||-mean||^2_{Sigma^{-1}}
67 Vector sigmas(3);
68 sigmas << 0.1, 0.2, 0.3;
69 Vector invSigmas = sigmas.cwiseInverse();
70 double mahalanobis =
71 (-mean.cwiseProduct(other: invSigmas)).squaredNorm(); // (r .* invSigmas)^2 sum
72 double expected_error = 0.5 * mahalanobis;
73 EXPECT_DOUBLES_EQUAL(expected_error, factor.error(values), 1e-9);
74
75 // Normalization constant (same as zero-mean case)
76 double logDetSigma =
77 2 * log(x: 0.1) + 2 * log(x: 0.2) + 2 * log(x: 0.3); // log(Ïƒ^2) per component sum
78 double expected_log_k = -0.5 * 3 * log(x: 2 * M_PI) - 0.5 * logDetSigma;
79 double expected_log_prob = expected_log_k - expected_error;
80 EXPECT_DOUBLES_EQUAL(expected_log_prob, factor.logProbability(x), 1e-9);
81 EXPECT_DOUBLES_EQUAL(expected_log_prob, factor.logProbability(values), 1e-9);
82 EXPECT_DOUBLES_EQUAL(exp(expected_log_prob), factor.evaluate(x), 1e-9);
83 EXPECT_DOUBLES_EQUAL(exp(expected_log_prob), factor.evaluate(values), 1e-9);
84
85 // Compare value vs values interface consistency
86 EXPECT_DOUBLES_EQUAL(factor.logProbability(x), factor.logProbability(values),
87 1e-9);
88 EXPECT_DOUBLES_EQUAL(factor.evaluate(x), factor.evaluate(values), 1e-9);
89}

92TEST(ConcentratedGaussian, TransportToSameOrigin) {
93 Key key(1);
94 Pose2 origin(1.0, 2.0, 0.3);
95 Matrix3 Sigma;
96 Sigma << 0.04, 0.0, 0.0, 0.0, 0.09, 0.0, 0.0, 0.0, 0.16;
97
98 // Nonzero mean to verify it's preserved when mapping to same reference
99 Vector3 mean;
100 mean << 0.01, -0.02, 0.05;
101 ConcentratedGaussian<Pose2> d(key, origin, mean, Sigma);
102
103 ConcentratedGaussian<Pose2> mapped = d.transportTo(x_hat: origin);
104
105 // Origin preserved
106 EXPECT(assert_equal(origin, mapped.origin()));
107
108 // Mean and covariance preserved
109 auto g = mapped.gaussianModel(method: "TransportToSameOrigin");
110 CHECK(g);
111 EXPECT(assert_equal(Sigma, g->covariance(), 1e-9));
112 CHECK(mapped.mean().has_value());
113 EXPECT(assert_equal(mean, *mapped.mean(), 1e-9));
114}

117TEST(ConcentratedGaussian, ResetMatchesTransport) {
118 Key key(1);
119 Pose2 origin(1.0, 2.0, 0.3);
120 Matrix3 Sigma;
121 Sigma << 0.04, 0.0, 0.0, 0.0, 0.09, 0.0, 0.0, 0.0, 0.16;
122
123 Vector3 mean;
124 mean << 0.05, -0.02, 0.1;
125 ConcentratedGaussian<Pose2> d(key, origin, mean, Sigma);
126
127 // Compute expected new origin xplus = Retract(origin, mean)
128 Pose2 xplus = traits<Pose2>::Retract(origin, v: mean);
129
130 // Our reset method should move origin to xplus and set zero mean
131 ConcentratedGaussian<Pose2> r = d.reset();
132 EXPECT(assert_equal(xplus, r.origin()));
133 CHECK(!r.mean().has_value());
134
135 // Transport-to(xplus) should yield the same covariance, and near-zero mean
136 ConcentratedGaussian<Pose2> mapped = d.transportTo(x_hat: xplus);
137 auto g_r = r.gaussianModel(method: "ResetMatchesTransport");
138 auto g_m = mapped.gaussianModel(method: "ResetMatchesTransport");
139 CHECK(g_r && g_m);
140 EXPECT(assert_equal(g_r->covariance(), g_m->covariance(), 1e-9));
141
142 // Mapped mean should be (approximately) zero in the chart at xplus
143 Vector3 zero = Vector3::Zero();
144 Vector3 mapped_mean = mapped.mean().value_or(u: Vector3::Zero());
145 EXPECT(assert_equal(zero, mapped_mean, 1e-7));
146}

149TEST(ConcentratedGaussian, FusionPose2Identical) {
150 Key key(1);
151 Pose2 origin(1.0, 2.0, 0.3);
152 Matrix3 Sigma;
153 Sigma << 0.04, 0.0, 0.0, 0.0, 0.09, 0.0, 0.0, 0.0, 0.16;
154 sharedGaussianNoiseModel model = noiseModel::Gaussian::Covariance(covariance: Sigma);
155
156 ConcentratedGaussian<Pose2> a(key, origin, model);
157 ConcentratedGaussian<Pose2> b(key, origin, model);
158
159 ConcentratedGaussian<Pose2> fused = a * b;
160
161 // Expect origin unchanged for identical inputs
162 EXPECT(assert_equal(origin, fused.origin()));
163
164 // Covariance should be halved: (Î£^{-1}+Î£^{-1})^{-1} = Î£/2
165 auto gf = fused.gaussianModel(method: "FusionPose2Identical");
166 CHECK(gf);
167 Matrix3 Sigma_f = gf->covariance();
168 EXPECT(assert_equal<Matrix3>(0.5 * Sigma, Sigma_f, 1e-9));
169}
