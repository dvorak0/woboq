42/// Functor that takes a matrix and multiplies every element by m
43class MultiplyFunctor {
44 double m_; ///< simple multiplier
45
46 public:
47 MultiplyFunctor(double m) : m_(m) {}
48
49 Matrix operator()(const Matrix &X,
50 OptionalJacobian<-1, -1> H = {}) const {
51 if (H) *H = m_ * Matrix::Identity(rows: X.rows() * X.cols(), cols: X.rows() * X.cols());
52 return m_ * X;
53 }
54};
55
56/// Functor that performs Ax where A is a matrix and x is a vector.
57class ProjectionFunctor {
58 public:
59 Vector operator()(const Matrix &A, const Vector &x,
60 OptionalJacobian<-1, -1> H1 = {},
61 OptionalJacobian<-1, -1> H2 = {}) const {
62 if (H1) {
63 H1->resize(rows: x.size(), cols: A.size());
64 *H1 << I_3x3, I_3x3, I_3x3;
65 }
66 if (H2) *H2 = A;
67 return A * x;
68 }
69};

74 Matrix X = Matrix::Identity(rows: 3, cols: 3), measurement = Matrix::Identity(rows: 3, cols: 3);

76 double multiplier = 1.0;
77 auto functor = MultiplyFunctor(multiplier);
78 auto factor = MakeFunctorizedFactor<Matrix>(key, z: measurement, model, func: functor);
79
80 Vector error = factor.evaluateError(x: X);
81
82 EXPECT(assert_equal(Vector::Zero(9), error, 1e-9));
83}
