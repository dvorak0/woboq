25// Generate a small PoseSLAM problem
26std::tuple<NonlinearFactorGraph, Values> generateProblem() {
27 // 1. Create graph container and add factors to it
28 NonlinearFactorGraph graph;
29
30 // 2a. Add Gaussian prior
31 Pose2 priorMean(0.0, 0.0, 0.0); // prior at origin
32 SharedDiagonal priorNoise =
33 noiseModel::Diagonal::Sigmas(sigmas: Vector3(0.3, 0.3, 0.1));
34 graph.addPrior(key: 1, prior: priorMean, model: priorNoise);
35
36 // 2b. Add odometry factors
37 SharedDiagonal odometryNoise =
38 noiseModel::Diagonal::Sigmas(sigmas: Vector3(0.2, 0.2, 0.1));
39 graph.emplace_shared<BetweenFactor<Pose2>>(args: 1, args: 2, args: Pose2(2.0, 0.0, 0.0),
40 args&: odometryNoise);
41 graph.emplace_shared<BetweenFactor<Pose2>>(args: 2, args: 3, args: Pose2(2.0, 0.0, M_PI_2),
42 args&: odometryNoise);
43 graph.emplace_shared<BetweenFactor<Pose2>>(args: 3, args: 4, args: Pose2(2.0, 0.0, M_PI_2),
44 args&: odometryNoise);
45 graph.emplace_shared<BetweenFactor<Pose2>>(args: 4, args: 5, args: Pose2(2.0, 0.0, M_PI_2),
46 args&: odometryNoise);
47
48 // 2c. Add pose constraint
49 SharedDiagonal constraintUncertainty =
50 noiseModel::Diagonal::Sigmas(sigmas: Vector3(0.2, 0.2, 0.1));
51 graph.emplace_shared<BetweenFactor<Pose2>>(args: 5, args: 2, args: Pose2(2.0, 0.0, M_PI_2),
52 args&: constraintUncertainty);
53
54 // 3. Create the data structure to hold the initialEstimate estimate to the
55 // solution
56 Values initialEstimate;
57 Pose2 x1(0.5, 0.0, 0.2);
58 initialEstimate.insert(j: 1, val: x1);
59 Pose2 x2(2.3, 0.1, -0.2);
60 initialEstimate.insert(j: 2, val: x2);
61 Pose2 x3(4.1, 0.1, M_PI_2);
62 initialEstimate.insert(j: 3, val: x3);
63 Pose2 x4(4.0, 2.0, M_PI);
64 initialEstimate.insert(j: 4, val: x4);
65 Pose2 x5(2.1, 2.1, -M_PI_2);
66 initialEstimate.insert(j: 5, val: x5);
67
68 return {graph, initialEstimate};
69}

72TEST(NonlinearConjugateGradientOptimizer, Optimize) {
73 const auto [graph, initialEstimate] = generateProblem();
74 // cout << "initial error = " << graph.error(initialEstimate) << endl;
75
76 NonlinearOptimizerParams param;
77 param.maxIterations =

79 param.verbosity = NonlinearOptimizerParams::SILENT;

84

88namespace rosenbrock {
89
90class Rosenbrock1Factor : public NoiseModelFactorN<double> {
91 private:
92 typedef Rosenbrock1Factor This;
93 typedef NoiseModelFactorN<double> Base;
94
95 double a_;
96
97 public:
98 /** Constructor: key is x */
99 Rosenbrock1Factor(Key key, double a, const SharedNoiseModel& model = nullptr)
100 : Base(model, key), a_(a) {}
101
102 /// evaluate error
103 Vector evaluateError(const double& x, OptionalMatrixType H) const override {
104 double d = x - a_;
105 // Because linearized gradient is -A'b/sigma, it will multiply by d
106 if (H) (*H) = Vector1(1).transpose();
107 return Vector1(d);
108 }
109};
110
111/**
112 * @brief Factor for the second term of the Rosenbrock function.
113 * f2 = (y - x*x)
114 *
115 * We use the noise model to premultiply with `b`.
116 */
117class Rosenbrock2Factor : public NoiseModelFactorN<double, double> {
118 private:
119 typedef Rosenbrock2Factor This;
120 typedef NoiseModelFactorN<double, double> Base;
121
122 public:
123 /** Constructor: key1 is x, key2 is y */
124 Rosenbrock2Factor(Key key1, Key key2, const SharedNoiseModel& model = nullptr)
125 : Base(model, key1, key2) {}
126
127 /// evaluate error
128 Vector evaluateError(const double& x, const double& y, OptionalMatrixType H1,
129 OptionalMatrixType H2) const override {
130 double x2 = x * x, d = x2 - y;
131 // Because linearized gradient is -A'b/sigma, it will multiply by d
132 if (H1) (*H1) = Vector1(2 * x).transpose();
133 if (H2) (*H2) = Vector1(-1).transpose();
134 return Vector1(d);
135 }
136};

138/**
139 * @brief Get a nonlinear factor graph representing
140 * the Rosenbrock Banana function.
141 *
142 * More details: https://en.wikipedia.org/wiki/Rosenbrock_function
143 *
144 * @param a
145 * @param b
146 * @return NonlinearFactorGraph
147 */
148static NonlinearFactorGraph GetRosenbrockGraph(double a = 1.0,
149 double b = 100.0) {
150 NonlinearFactorGraph graph;
151 graph.emplace_shared<Rosenbrock1Factor>(
152 args: X(j: 0), args&: a, args: noiseModel::Isotropic::Precision(dim: 1, precision: 2));
153 graph.emplace_shared<Rosenbrock2Factor>(
154 args: X(j: 0), args: Y(j: 0), args: noiseModel::Isotropic::Precision(dim: 1, precision: 2 * b));
155
156 return graph;
157}
158
159/// Compute the Rosenbrock function error given the nonlinear factor graph
160/// and input values.
161double f(const NonlinearFactorGraph& graph, double x, double y) {
162 Values initial;
163 initial.insert<double>(j: X(j: 0), val: x);
164 initial.insert<double>(j: Y(j: 0), val: y);
165
166 return graph.error(values: initial);
167}
168
169/// True Rosenbrock Banana function.
170double rosenbrock_func(double x, double y, double a = 1.0, double b = 100.0) {
171 double m = (a - x) * (a - x);
172 double n = b * (y - x * x) * (y - x * x);
173 return m + n;
174}
175} // namespace rosenbrock

177/* ************************************************************************* */
178// Test whether the 2 factors are properly implemented.
179TEST(NonlinearConjugateGradientOptimizer, Rosenbrock) {
180 using namespace rosenbrock;
181 double a = 1.0, b = 100.0;
182 auto graph = GetRosenbrockGraph(a, b);
183 Rosenbrock1Factor f1 =
184 *std::static_pointer_cast<Rosenbrock1Factor>(r: graph.at(i: 0));
185 Rosenbrock2Factor f2 =
186 *std::static_pointer_cast<Rosenbrock2Factor>(r: graph.at(i: 1));
187 Values values;
188 values.insert<double>(j: X(j: 0), val: 3.0);
189 values.insert<double>(j: Y(j: 0), val: 5.0);
190 EXPECT_CORRECT_FACTOR_JACOBIANS(f1, values, 1e-7, 1e-5);
191 EXPECT_CORRECT_FACTOR_JACOBIANS(f2, values, 1e-7, 1e-5);
192
193 std::mt19937 rng(42);
194 std::uniform_real_distribution<double> dist(0.0, 100.0);
195 for (size_t i = 0; i < 50; ++i) {
196 double x = dist(rng);
197 double y = dist(rng);
198
199 auto graph = GetRosenbrockGraph(a, b);
200 EXPECT_DOUBLES_EQUAL(rosenbrock_func(x, y, a, b), f(graph, x, y), 1e-5);
201 }
202}

205// Optimize the Rosenbrock function to verify optimizer works
206TEST(NonlinearConjugateGradientOptimizer, Optimization) {
207 using namespace rosenbrock;
208
209 double a = 12;
210 auto graph = GetRosenbrockGraph(a);
211
212 // Assert that error at global minimum is 0.
213 double error = f(graph, x: a, y: a * a);
214 EXPECT_DOUBLES_EQUAL(0.0, error, 1e-12);
215
216 NonlinearOptimizerParams param;
217 param.maxIterations = 350;
218 // param.verbosity = NonlinearOptimizerParams::LINEAR;
219 param.verbosity = NonlinearOptimizerParams::SILENT;
220
221 double x = 3.0, y = 5.0;
222 Values initialEstimate;
223 initialEstimate.insert<double>(j: X(j: 0), val: x);
224 initialEstimate.insert<double>(j: Y(j: 0), val: y);
225
226 GaussianFactorGraph::shared_ptr linear = graph.linearize(linearizationPoint: initialEstimate);
227 // std::cout << "error: " << f(graph, x, y) << std::endl;
228 // linear->print();
229 // linear->gradientAtZero().print("Gradient: ");
230
231 NonlinearConjugateGradientOptimizer optimizer(graph, initialEstimate, param);
232 Values result = optimizer.optimize();
233 // result.print();
234 // cout << "cg final error = " << graph.error(result) << endl;
235
236 Values expected;
237 expected.insert<double>(j: X(j: 0), val: a);
238 expected.insert<double>(j: Y(j: 0), val: a * a);
239 EXPECT(assert_equal(expected, result, 1e-1));
240}

244TEST(NonlinearConjugateGradientOptimizer, DirectionMethods) {
245 const auto [graph, initialEstimate] = generateProblem();
246
247 NonlinearOptimizerParams param;
248 param.maxIterations =

250 param.verbosity = NonlinearOptimizerParams::SILENT;

259 }
260 // Polak-Ribiere
261 {
262 NonlinearConjugateGradientOptimizer optimizer(
263 graph, initialEstimate, param, DirectionMethod::PolakRibiere);

266 EXPECT_DOUBLES_EQUAL(0.0, graph.error(result), 1e-4);
267 }

270 NonlinearConjugateGradientOptimizer optimizer(
