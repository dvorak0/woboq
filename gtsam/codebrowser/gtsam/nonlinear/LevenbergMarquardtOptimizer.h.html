21#pragma once
22
23#include <gtsam/nonlinear/NonlinearOptimizer.h>
24#include <gtsam/nonlinear/LevenbergMarquardtParams.h>
25#include <gtsam/linear/VectorValues.h>
26#include <chrono>
27
28class NonlinearOptimizerMoreOptimizationTest;
29
30namespace gtsam {
31
32/**
33 * This class performs Levenberg-Marquardt nonlinear optimization
34 */
35class GTSAM_EXPORT LevenbergMarquardtOptimizer: public NonlinearOptimizer {
36
37protected:
38 const LevenbergMarquardtParams params_; ///< LM parameters
39
40 // startTime_ is a chrono time point
41 std::chrono::time_point<std::chrono::high_resolution_clock> startTime_; ///< time when optimization started
42
43 void initTime();
44
45public:
46 typedef std::shared_ptr<LevenbergMarquardtOptimizer> shared_ptr;
47
48 /// @name Constructors/Destructor
49 /// @{
50
51 /** Standard constructor, requires a nonlinear factor graph, initial
52 * variable assignments, and optimization parameters. For convenience this
53 * version takes plain objects instead of shared pointers, but internally
54 * copies the objects.
55 * @param graph The nonlinear factor graph to optimize
56 * @param initialValues The initial variable assignments
57 * @param params The optimization parameters
58 */
59 LevenbergMarquardtOptimizer(const NonlinearFactorGraph& graph, const Values& initialValues,
60 const LevenbergMarquardtParams& params = LevenbergMarquardtParams());
61
62 /** Standard constructor, requires a nonlinear factor graph, initial
63 * variable assignments, and optimization parameters. For convenience this
64 * version takes plain objects instead of shared pointers, but internally
65 * copies the objects.
66 * @param graph The nonlinear factor graph to optimize
67 * @param initialValues The initial variable assignments
68 */
69 LevenbergMarquardtOptimizer(const NonlinearFactorGraph& graph, const Values& initialValues,
70 const Ordering& ordering,
71 const LevenbergMarquardtParams& params = LevenbergMarquardtParams());
72
73 /** Virtual destructor */
74 ~LevenbergMarquardtOptimizer() override {
75 }
76
77 /// @}
78
79 /// @name Standard interface
80 /// @{
81
82 /// Access the current damping value
83 double lambda() const;
84
85 /// Access the current number of inner iterations
86 int getInnerIterations() const;
87
88 /// print
89 void print(const std::string& str = "") const {
90 std::cout << str << "LevenbergMarquardtOptimizer" << std::endl;
91 this->params_.print(str: " parameters:\n");
92 }
93
94 /// @}
95
96 /// @name Advanced interface
97 /// @{
98
99 /**
100 * Perform a single iteration, returning GaussianFactorGraph corresponding to
101 * the linearized factor graph.
102 */
103 GaussianFactorGraph::shared_ptr iterate() override;
104
105 /** Read-only access the parameters */
106 const LevenbergMarquardtParams& params() const {
107 return params_;
108 }
109
110 void writeLogFile(double currentError);
111
112 /** linearize, can be overwritten */
113 virtual GaussianFactorGraph::shared_ptr linearize() const;
114
115 /** Build a damped system for a specific lambda -- for testing only */
116 GaussianFactorGraph buildDampedSystem(const GaussianFactorGraph& linear,
117 const VectorValues& sqrtHessianDiagonal) const;
118
119 /** Inner loop, changes state, returns true if successful or giving up */
120 bool tryLambda(const GaussianFactorGraph& linear, const VectorValues& sqrtHessianDiagonal);
121
122 /// @}
123
124protected:
125
126 /** Access the parameters (base class version) */
127 const NonlinearOptimizerParams& _params() const override {
128 return params_;
129 }
130};
131
132}
