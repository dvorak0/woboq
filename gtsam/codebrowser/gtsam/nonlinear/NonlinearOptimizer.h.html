28/**
29 * This is the abstract interface for classes that can optimize for the
30 * maximum-likelihood estimate of a NonlinearFactorGraph.
31 *
32 * To use a class derived from this interface, construct the class with a
33 * NonlinearFactorGraph and an initial Values variable assignment. Next, call the
34 * optimize() method which returns the optimized variable assignment.
35 *
36 * Simple and compact example:
37 * \code
38// One-liner to do full optimization and use the result.
39Values result = DoglegOptimizer(graph, initialValues).optimize();
40\endcode
41 *
42 * Example exposing more functionality and details:
43 * \code
44// Create initial optimizer
45DoglegOptimizer optimizer(graph, initialValues);
46
47// Run full optimization until convergence.
48Values result = optimizer->optimize();
49
50// The new optimizer has results and statistics
51cout << "Converged in " << optimizer.iterations() << " iterations "
52 "with final error " << optimizer.error() << endl;
53\endcode
54 *
55 * Example of setting parameters before optimization:
56 * \code
57// Each derived optimizer type has its own parameters class, which inherits from NonlinearOptimizerParams
58DoglegParams params;
59params.factorization = DoglegParams::QR;
60params.relativeErrorTol = 1e-3;
61params.absoluteErrorTol = 1e-3;
62
63// Optimize
64Values result = DoglegOptimizer(graph, initialValues, params).optimize();
65\endcode
66 *
67 * This interface also exposes an iterate() method, which performs one
68 * iteration. The optimize() method simply calls iterate() multiple times,
69 * until the error changes less than a threshold. We expose iterate() so that
70 * you can easily control what happens between iterations, such as drawing or
71 * printing, moving points from behind the camera to in front, etc.
72 *
73 * For more flexibility you may override virtual methods in your own derived class.
74 */
75class GTSAM_EXPORT NonlinearOptimizer {
76
77protected:
78 NonlinearFactorGraph graph_; ///< The graph with nonlinear factors
79
80 std::unique_ptr<internal::NonlinearOptimizerState> state_; ///< PIMPL'd state
81
82public:
83 /** A shared pointer to this class */
84 using shared_ptr = std::shared_ptr<const NonlinearOptimizer>;
85
86 /// @name Standard interface
87 /// @{
88
89 /**
90 * Optimize for the maximum-likelihood estimate, returning a the optimized
91 * variable assignments.
92 *
93 * This function simply calls iterate() in a loop, checking for convergence
94 * with check_convergence(). For fine-grain control over the optimization
95 * process, you may call iterate() and check_convergence() yourself, and if
96 * needed modify the optimization state between iterations.
97 */
98 virtual const Values& optimize() { defaultOptimize(); return values(); }
99
100 /**
101 * Optimize, but return empty result if any uncaught exception is thrown
102 * Intended for MATLAB. In C++, use above and catch exceptions.
103 * No message is printed: it is up to the caller to check the result
104 * @param optimizer a non-linear optimizer
105 */
106 const Values& optimizeSafely();
107
108 /// return error in current optimizer state
109 double error() const;
110
111 /// return number of iterations in current optimizer state
112 size_t iterations() const;
113
114 /// return values in current optimizer state
115 const Values &values() const;
116
117 /// return the graph with nonlinear factors
118 const NonlinearFactorGraph &graph() const { return graph_; }
119
120 /// @}
121
122 /// @name Advanced interface
123 /// @{
124
125 /** Virtual destructor */
126 virtual ~NonlinearOptimizer();
127
128 /** Default function to do linear solve, i.e. optimize a GaussianFactorGraph */
129 virtual VectorValues solve(const GaussianFactorGraph &gfg,
130 const NonlinearOptimizerParams& params) const;
131
132 /**
133 * Perform a single iteration, returning GaussianFactorGraph corresponding to
134 * the linearized factor graph.
135 */
136 virtual GaussianFactorGraph::shared_ptr iterate() = 0;
137
138 /// @}
139
140protected:
141 /** A default implementation of the optimization loop, which calls iterate()
142 * until checkConvergence returns true.
143 */
144 void defaultOptimize();
145
146 virtual const NonlinearOptimizerParams& _params() const = 0;
147
148 /** Constructor for initial construction of base classes. Takes ownership of state. */
149 NonlinearOptimizer(const NonlinearFactorGraph& graph,
150 std::unique_ptr<internal::NonlinearOptimizerState> state);
151};
152
153/** Check whether the relative error decrease is less than relativeErrorTreshold,
154 * the absolute error decrease is less than absoluteErrorTreshold, <em>or</em>
155 * the error itself is less than errorThreshold.
156 */
157GTSAM_EXPORT bool checkConvergence(double relativeErrorTreshold,
158 double absoluteErrorTreshold, double errorThreshold,
159 double currentError, double newError, NonlinearOptimizerParams::Verbosity verbosity = NonlinearOptimizerParams::SILENT);
160
161GTSAM_EXPORT bool checkConvergence(const NonlinearOptimizerParams& params, double currentError,
162 double newError);
163
164} // gtsam
