51/* ************************************************************************* */
52NonlinearFactor::shared_ptr NonlinearFactor::rekey(
53 const std::map<Key, Key>& rekey_mapping) const {
54 shared_ptr new_factor = clone();
55 for (size_t i = 0; i < new_factor->size(); ++i) {
56 Key& cur_key = new_factor->keys()[i];
57 std::map<Key, Key>::const_iterator mapping = rekey_mapping.find(x: cur_key);
58 if (mapping != rekey_mapping.end())
59 cur_key = mapping->second;
60 }
61 return new_factor;
62}

122/* ************************************************************************* */
123double NoiseModelFactor::weight(const Values& c) const {
124 if (active(c)) {
125 if (noiseModel_) {
126 const Vector b = unwhitenedError(x: c);
127 check(noiseModel: noiseModel_, m: b.size());
128 return noiseModel_->weight(v: b);
129 }
130 else
131 return 1.0;
132 } else {
133 return 0.0;
134 }
135}

137/* ************************************************************************* */
138double NoiseModelFactor::error(const Values& c) const {
139 if (active(c)) {
140 const Vector b = unwhitenedError(x: c);
141 check(noiseModel: noiseModel_, m: b.size());
142 if (noiseModel_)
143 return noiseModel_->loss(squared_distance: noiseModel_->squaredMahalanobisDistance(v: b));
144 else
145 return 0.5 * b.squaredNorm();
146 } else {
147 return 0.0;
148 }
149}

151/* ************************************************************************* */
152std::shared_ptr<GaussianFactor> NoiseModelFactor::linearize(
153 const Values& x) const {
154
155 // Only linearize if the factor is active
156 if (!active(c: x))
157 return std::shared_ptr<JacobianFactor>();
158
159 // Call evaluate error to get Jacobians and RHS vector b
160 std::vector<Matrix> A(size());
161 Vector b = -unwhitenedError(x, H&: A);
162 check(noiseModel: noiseModel_, m: b.size());
163
164 // Whiten the corresponding system now
165 if (noiseModel_)
166 noiseModel_->WhitenSystem(A, b);
167
168 // Fill in terms, needed to create JacobianFactor below
169 std::vector<std::pair<Key, Matrix> > terms(size());
170 for (size_t j = 0; j < size(); ++j) {
171 terms[j].first = keys()[j];
172 terms[j].second.swap(other&: A[j]);
173 }
174
175 // TODO pass unwhitened + noise model to Gaussian factor
176 using noiseModel::Constrained;
177 if (noiseModel_ && noiseModel_->isConstrained())
178 return GaussianFactor::shared_ptr(
179 new JacobianFactor(terms, b,
180 std::static_pointer_cast<Constrained>(r: noiseModel_)->unit()));
181 else {
182 return GaussianFactor::shared_ptr(new JacobianFactor(terms, b));
183 }
184}
