29/**
30 * @brief Compute the quantile function of the Chi-Squared distribution.
31 *
32 * The quantile function of the Chi-squared distribution is the quantile of
33 * the specific (inverse) incomplete Gamma distribution.
34 *
35 * @param dofs Degrees of freedom
36 * @param alpha Quantile value
37 * @return double
38 */
39inline double chi_squared_quantile(const double dofs, const double alpha) {
40 return 2 * igami(a: dofs / 2, p: alpha);
41}
