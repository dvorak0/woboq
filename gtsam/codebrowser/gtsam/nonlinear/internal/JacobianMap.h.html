26namespace internal {

28// A JacobianMap is the primary mechanism by which derivatives are returned.
29// Expressions are designed to write their derivatives into an already allocated
30// Jacobian of the correct size, of type VerticalBlockMatrix.
31// The JacobianMap provides a mapping from keys to the underlying blocks.
32class JacobianMap {
33private:
34 const KeyVector& keys_;
35 VerticalBlockMatrix& Ab_;
36
37public:
38 /// Construct a JacobianMap for writing into a VerticalBlockMatrix Ab
39 JacobianMap(const KeyVector& keys, VerticalBlockMatrix& Ab) :
40 keys_(keys), Ab_(Ab) {
41 }
42
43 /// Access blocks of via key
44 VerticalBlockMatrix::Block operator()(Key key) {
45 KeyVector::const_iterator it = std::find(first: keys_.begin(), last: keys_.end(), val: key);
46 DenseIndex block = it - keys_.begin();
47 return Ab_(block);
48 }
49};

51} // namespace internal
