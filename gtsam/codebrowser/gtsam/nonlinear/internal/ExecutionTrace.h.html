38/// Storage type for the execution trace.
39/// It enforces the proper alignment in a portable way.
40/// Provide a traceSize() sized array of this type to traceExecution as traceStorage.
41#ifdef _MSC_VER
42// TODO(dellaert): this might lead to trouble if Eigen decides to use 32 on Windows.
43static const unsigned TraceAlignment = 16; // 16 bytes max_align on Windows
44#else
45static const unsigned TraceAlignment = 32; // Alignment used by Eigen on some platforms.
46#endif
47// TODO(dellaert): we *should* be able to simplify the code using the pointer arithmetic from ExecutionTraceStorage.
48typedef std::aligned_storage<1, TraceAlignment>::type ExecutionTraceStorage;

50template<bool UseBlock, typename Derived>
51struct UseBlockIf {
52 static void addToJacobian(const Eigen::MatrixBase<Derived>& dTdA,
53 JacobianMap& jacobians, Key key) {
54 // block makes HUGE difference
55 jacobians(key).block<Derived::RowsAtCompileTime, Derived::ColsAtCompileTime>(
56 0, 0) += dTdA;
57 }
58};
59
60/// Handle Leaf Case for Dynamic Matrix type (slower)
61template<typename Derived>
62struct UseBlockIf<false, Derived> {
63 static void addToJacobian(const Eigen::MatrixBase<Derived>& dTdA,
64 JacobianMap& jacobians, Key key) {
65 jacobians(key) += dTdA;
66 }
67};
68
69/// Handle Leaf Case: reverse AD ends here, by writing a matrix into Jacobians
70template<typename Derived>
71void handleLeafCase(const Eigen::MatrixBase<Derived>& dTdA,
72 JacobianMap& jacobians, Key key) {
73 UseBlockIf<
74 Derived::RowsAtCompileTime != Eigen::Dynamic
75 && Derived::ColsAtCompileTime != Eigen::Dynamic, Derived>::addToJacobian(
76 dTdA, jacobians, key);
77}

79/**
80 * The ExecutionTrace class records a tree-structured expression's execution.
81 *
82 * The class looks a bit complicated but it is so for performance.
83 * It is a tagged union that obviates the need to create
84 * a ExecutionTrace subclass for Constants and Leaf Expressions. Instead
85 * the key for the leaf is stored in the space normally used to store a
86 * CallRecord*. Nothing is stored for a Constant.
87 *
88 * A full execution trace of a Binary(Unary(Binary(Leaf,Constant)),Leaf) would be:
89 * Trace(Function) ->
90 * BinaryRecord with two traces in it
91 * trace1(Function) ->
92 * UnaryRecord with one trace in it
93 * trace1(Function) ->
94 * BinaryRecord with two traces in it
95 * trace1(Leaf)
96 * trace2(Constant)
97 * trace2(Leaf)
98 * Hence, there are three Record structs, written to memory by traceExecution
99 */
100template<class T>
101class ExecutionTrace {
102 static const int Dim = traits<T>::dimension;
103 typedef Eigen::Matrix<double, Dim, Dim> JacobianTT;
104 enum {
105 Constant, Leaf, Function
106 } kind;
107 union {
108 Key key;
109 CallRecord<Dim>* ptr;
110 } content;
111
112 public:
113
114 /// Pointer always starts out as a Constant
115 ExecutionTrace() :
116 kind(Constant) {
117 }
118
119 /// Change pointer to a Leaf Record
120 void setLeaf(Key key) {
121 kind = Leaf;
122 content.key = key;
123 }
124
125 /// Take ownership of pointer to a Function Record
126 void setFunction(CallRecord<Dim>* record) {
127 kind = Function;
128 content.ptr = record;
129 }
130
131 /// Print
132 void print(const std::string& indent = "") const {
133 if (kind == Constant) {
134 std::cout << indent << "Constant" << std::endl;
135 } else if (kind == Leaf) {
136 std::cout << indent << "Leaf, key = " << content.key << std::endl;
137 } else if (kind == Function) {
138 content.ptr->print(indent + " ");
139 }
140 }
141
142 /// Return record pointer, quite unsafe, used only for testing
143 template<class Record>
144 std::optional<Record*> record() {
145 if (kind != Function) {
146 return {};
147 } else {
148 Record* p = dynamic_cast<Record*>(content.ptr);
149 return p ? std::optional<Record*>(p) : std::nullopt;
150 }
151 }
152
153 /**
154 * *** This is the main entry point for reverse AD, called from Expression ***
155 * Called only once, either inserts I into Jacobians (Leaf) or starts AD (Function)
156 */
157 void startReverseAD1(JacobianMap& jacobians) const {
158 if (kind == Leaf) {
159 // This branch will only be called on trivial Leaf expressions, i.e. Priors
160 static const JacobianTT I = JacobianTT::Identity();
161 handleLeafCase(I, jacobians, content.key);
162 } else if (kind == Function) {
163 // This is the more typical entry point, starting the AD pipeline
164 // Inside startReverseAD2 the correctly dimensioned pipeline is chosen.
165 content.ptr->startReverseAD2(jacobians);
166 }
167 }
168
169 /// Either add to Jacobians (Leaf) or propagate (Function)
170 // This is a crucial method in selecting the pipeline dimension: since it is templated method
171 // it will call the templated Record method reverseAD2. Hence, at this point in the pipeline
172 // there are as many reverseAD1 and reverseAD2 versions as there are different Jacobian sizes.
173 template<typename DerivedMatrix>
174 void reverseAD1(const Eigen::MatrixBase<DerivedMatrix> & dTdA,
175 JacobianMap& jacobians) const {
176 if (kind == Leaf)
177 handleLeafCase(dTdA, jacobians, content.key);
178 else if (kind == Function)
179 content.ptr->reverseAD2(dTdA, jacobians);
180 }
181
182 ~ExecutionTrace() {
183 if (kind == Function) {
184 content.ptr->~CallRecord<Dim>();
185 }
186 }
187
188 /// Define type so we can apply it as a meta-function
189 typedef ExecutionTrace<T> type;
190};
