12/**
13 * @file CustomFactor.h
14 * @brief Class to enable arbitrary factors with runtime swappable error function.
15 * @author Fan Jiang
16 */
17
18#pragma once
19
20#include <gtsam/nonlinear/NonlinearFactor.h>
21
22namespace gtsam {
23
24using JacobianVector = std::vector<Matrix>;
25
26class CustomFactor;
27
28/*
29 * NOTE
30 * ==========
31 * pybind11 will invoke a copy if this is `JacobianVector &`, and modifications in Python will not be reflected.
32 *
33 * This is safe because this is passing a const pointer, and pybind11 will maintain the `std::vector` memory layout.
34 * Thus the pointer will never be invalidated.
35 */
36using CustomErrorFunction = std::function<Vector(const CustomFactor &, const Values &, const JacobianVector *)>;
37
38/**
39 * @brief Custom factor that takes a std::function as the error
40 * @addtogroup nonlinear
41 * \nosubgrouping
42 *
43 * This factor is mainly for creating a custom factor in Python.
44 */
45class GTSAM_EXPORT CustomFactor: public NoiseModelFactor {
46protected:
47 CustomErrorFunction error_function_;
48
49protected:
50
51 using Base = NoiseModelFactor;
52 using This = CustomFactor;
53
54public:
55
56 /**
57 * Default Constructor for I/O
58 */
59 CustomFactor() = default;
60
61 /**
62 * Constructor
63 * @param noiseModel shared pointer to noise model
64 * @param keys keys of the variables
65 * @param errorFunction the error functional
66 */
67 CustomFactor(const SharedNoiseModel &noiseModel, const KeyVector &keys, const CustomErrorFunction &errorFunction) :
68 Base(noiseModel, keys) {
69 this->error_function_ = errorFunction;
70 }
71
72 /**
73 * Calls the errorFunction closure, which is a std::function object
74 * One can check if a derivative is needed in the errorFunction by checking the length of Jacobian array
75 */
76 Vector unwhitenedError(const Values &x, OptionalMatrixVecType H = nullptr) const override;
77
78 /** print */
79 void print(const std::string &s,
80 const KeyFormatter &keyFormatter = DefaultKeyFormatter) const override;
81
82 /**
83 * Mark not sendable
84 */
85 bool sendable() const override {
86 return false;
87 }
88
89private:
90
91#if GTSAM_ENABLE_BOOST_SERIALIZATION
92 /** Serialization function */
93 friend class boost::serialization::access;
94 template<class ARCHIVE>
95 void serialize(ARCHIVE &ar, const unsigned int /*version*/) {
96 ar & boost::serialization::make_nvp(n: "CustomFactor",
97 v&: boost::serialization::base_object<Base>(d&: *this));
98 }
99#endif
100};
101
102}
