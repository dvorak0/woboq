18#pragma once
19
20#include <gtsam/nonlinear/NonlinearFactorGraph.h>
21#include <gtsam/linear/GaussianISAM.h>
22
23namespace gtsam {
24/**
25 * Wrapper class to manage ISAM in a nonlinear context
26 */
27class GTSAM_EXPORT NonlinearISAM {
28protected:
29
30 /** The internal iSAM object */
31 gtsam::GaussianISAM isam_;
32
33 /** The current linearization point */
34 Values linPoint_;
35
36 /** The original factors, used when relinearizing */
37 NonlinearFactorGraph factors_;
38
39 /** The reordering interval and counter */
40 int reorderInterval_;
41 int reorderCounter_;
42
43 /** The elimination function */
44 GaussianFactorGraph::Eliminate eliminationFunction_;
45
46public:
47
48 /// @name Standard Constructors
49 /// @{
50
51 /**
52 * Periodically reorder and relinearize
53 * @param reorderInterval is the number of updates between reorderings,
54 * 0 never reorders (and is dangerous for memory consumption)
55 * 1 (default) reorders every time, in worse case is batch every update
56 * typical values are 50 or 100
57 */
58 NonlinearISAM(int reorderInterval = 1,
59 const GaussianFactorGraph::Eliminate& eliminationFunction = GaussianFactorGraph::EliminationTraitsType::DefaultEliminate) :
60 reorderInterval_(reorderInterval), reorderCounter_(0), eliminationFunction_(eliminationFunction) {}
61
62 /// @}
63 /// @name Standard Interface
64 /// @{
65
66 /** Return the current solution estimate */
67 Values estimate() const;
68
69 /** find the marginal covariance for a single variable */
70 Matrix marginalCovariance(Key key) const;
71
72 // access
73
74 /** access the underlying bayes tree */
75 const GaussianISAM& bayesTree() const { return isam_; }
76
77 /** Return the current linearization point */
78 const Values& getLinearizationPoint() const { return linPoint_; }
79
80 /** get underlying nonlinear graph */
81 const NonlinearFactorGraph& getFactorsUnsafe() const { return factors_; }
82
83 /** get counters */
84 int reorderInterval() const { return reorderInterval_; } ///<TODO: comment
85 int reorderCounter() const { return reorderCounter_; } ///<TODO: comment
86
87 /** prints out all contents of the system */
88 void print(const std::string& s="", const KeyFormatter& keyFormatter = DefaultKeyFormatter) const;
89
90 /** prints out clique statistics */
91 void printStats() const;
92
93 /** saves the Tree to a text file in GraphViz format */
94 void saveGraph(const std::string& s, const KeyFormatter& keyFormatter = DefaultKeyFormatter) const;
95
96 /// @}
97 /// @name Advanced Interface
98 /// @{
99
100 /** Add new factors along with their initial linearization points */
101 void update(const NonlinearFactorGraph& newFactors, const Values& initialValues);
102
103 /** Relinearization and reordering of variables */
104 void reorder_relinearize();
105
106 /// @}
107
108};
109
110} // \namespace gtsam
