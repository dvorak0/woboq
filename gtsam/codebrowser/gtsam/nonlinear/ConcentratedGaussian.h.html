27/**
28 * A nonlinear density, inherits from ExtendedPriorFactor. This class models
29 * a (left) extended concentrated Gaussian (L-ECG) with Gaussian noise models.
30 * @ingroup nonlinear
31 */
32template <class T>
33class ConcentratedGaussian : public ExtendedPriorFactor<T> {
34 public:
35 using Base = ExtendedPriorFactor<T>;
36 using Gaussian = typename Base::Gaussian;
37 using sharedGaussianNoiseModel = noiseModel::Gaussian::shared_ptr;
38
39 /// @name Standard Constructors
40 /// @{
41
42 /// Default constructor for serialization.
43 ConcentratedGaussian() {}
44
45 /// Constructor with noise model and optional mean in tangent space
46 ConcentratedGaussian(Key key, const T& origin,
47 const sharedGaussianNoiseModel& model)
48 : Base(key, origin, model) {}
49
50 /// Constructor with noise model and optional mean in tangent space
51 ConcentratedGaussian(Key key, const T& origin, const Vector& mean,
52 const sharedGaussianNoiseModel& model)
53 : Base(key, origin, mean, model) {
54 if (mean.size() != static_cast<Eigen::Index>(model->dim()))
55 throw std::invalid_argument(
56 "ConcentratedGaussian: mean dimension does not match noise model");
57 }
58
59 /// Constructor with covariance matrix (zero mean in tangent space)
60 ConcentratedGaussian(Key key, const T& origin, const Matrix& covariance)
61 : Base(key, origin, covariance) {}
62
63 /// Constructor with mean (in tangent space) and covariance matrix
64 ConcentratedGaussian(Key key, const T& origin, const Vector& mean,
65 const Matrix& covariance)
66 : Base(key, origin, mean, covariance) {
67 if (mean.size() != covariance.rows())
68 throw std::invalid_argument(
69 "ConcentratedGaussian: mean dimension does not match covariance");
70 if (covariance.rows() != covariance.cols())
71 throw std::invalid_argument(
72 "ConcentratedGaussian: covariance matrix is not square");
73 }
74
75 /// @}
76 /// @name Standard Destructor
77 /// @{
78
79 ~ConcentratedGaussian() override {}
80
81 /// @}
82 /// @name Testable
83 /// @{
84
85 /// print
86 void print(const std::string& s, const KeyFormatter& keyFormatter =
87 DefaultKeyFormatter) const override {
88 std::cout << s << "ConcentratedGaussian on " << keyFormatter(this->key())
89 << "\n";
90 traits<T>::Print(this->origin_, " origin: ");
91 if (this->mean_) gtsam::print(*this->mean_, " tangent space mean: ");
92 if (this->noiseModel_)
93 this->noiseModel_->print(" noise model: ");
94 else
95 std::cout << "no noise model\n";
96 }
97
98 /// equals
99 bool equals(const NonlinearFactor& expected,
100 double tol = 1e-9) const override {
101 const auto* e = dynamic_cast<const ConcentratedGaussian*>(&expected);
102 return e && Base::equals(*e, tol);
103 }
104
105 /// @}
106 /// @name Standard API
107 /// @{
108
109 /// Return T element corresponding to the mean, with optional Jacobian
110 T retractMean(Matrix* xHm = nullptr) const {
111 const size_t n = this->dim();
112 const bool zeroMean = !this->mean_;
113 if (xHm && zeroMean) xHm->setIdentity(rows: n, cols: n);
114 return zeroMean
115 ? this->origin_
116 : traits<T>::Retract(this->origin_, *(this->mean_), {}, xHm);
117 }
118
119 /**
120 * Calculate the normalization constant for the density.
121 * For a Gaussian noise model with covariance Σ, we return
122 * - log k = 0.5 * n * log(2*pi) + 0.5 * log |Σ|
123 * where n = dim(). Note: gaussian->logDeterminant() returns log|Σ|.
124 */
125 double negLogConstant() const {
126 const size_t n = this->dim();
127 auto gaussian = this->gaussianModel("ConcentratedGaussian::negLogConstant",
128 /* throw */ true);
129 constexpr double log2pi = 1.8378770664093454835606594728112; // log(2*pi)
130 const double logDetSigma = gaussian->logDeterminant(); // log |Σ|
131 return 0.5 * n * log2pi + 0.5 * logDetSigma;
132 }
133
134 /**
135 * Calculate the log-probability of the given value.
136 * error(x) as defined for a GTSAM factor already equals 0.5 * ||r(x)||^2_Σ
137 * (i.e. the negative log-likelihood without the normalization constant).
138 * Hence: log P(x) = log k - error(x).
139 */
140 double logProbability(const T& x) const {
141 return -(negLogConstant() + this->error(x));
142 }
143
144 /**
145 * Log-probability overload taking a Values container. This mirrors the
146 * linear GaussianConditional interface so densities can be queried in a
147 * uniform way when only a Values is available.
148 */
149 double logProbability(const Values& values) const {
150 const T& x = values.at<T>(this->key());
151 return logProbability(x);
152 }
153
154 /**
155 * Evaluate the probability density at the given value.
156 * P(x) = exp(logProbability(x)).
157 */
158 double evaluate(const T& x) const { return exp(logProbability(x)); }
159
160 /// Evaluate density P(x) using a Values container.
161 double evaluate(const Values& values) const {
162 return exp(logProbability(values));
163 }
164
165 /// @}
166 /// @name Transport and Fusion
167 /// @{
168
169 /**
170 * Create a new ConcentratedGaussian with zero mean by moving the origin to x̂
171 * = Retract(origin, mean). Returns an ECG with origin=x̂, zero mean, and
172 * covariance transported to x̂.
173 */
174 ConcentratedGaussian reset() const {
175 if (!this->mean_) return *this; // already zero-mean
176 auto g =
177 this->gaussianModel("ConcentratedGaussian::reset", /* throw */ true);
178
179 Matrix hatJm;
180 const T x_hat = retractMean(xHm: &hatJm);
181 const Matrix covHat = hatJm * g->covariance() * hatJm.transpose();
182
183 return ConcentratedGaussian(this->key(), x_hat, Symmetrize(matrix: covHat));
184 }
185
186 /**
187 * Transport this density to a new origin x̂, returning a density at x̂
188 * with nonzero mean in that chart. Uses a full first-order Jacobian for the
189 * change of coordinates between charts via the chain rule:
190 * J = ∂Local(x̂,x)/∂x · ∂Retract(origin,m)/∂m
191 */
192 ConcentratedGaussian transportTo(const T& x_hat) const {
193 auto g = this->gaussianModel("ConcentratedGaussian::transportTo",
194 /* throw */ true);
195
196 Matrix xHm; // ∂Retract(origin,m)/∂m
197 const T x = retractMean(xHm: &xHm);
198
199 Matrix hatHx; // ∂Local(x̂,x)/∂x
200 Vector muHat = traits<T>::Local(x_hat, x, {}, hatHx);
201 const Matrix hatJm = hatHx * xHm; // chain rule
202 const Matrix covHat = hatJm * g->covariance() * hatJm.transpose();
203
204 return ConcentratedGaussian(this->key(), x_hat, muHat, Symmetrize(matrix: covHat));
205 }
206
207 /**
208 * Fusion operator implementing the (approximate) three-step Fusion
209 * method in:
210 * Y. Ge, P. van Goor and R. Mahony, "A Geometric Perspective on Fusing
211 * Gaussian Distributions on Lie Groups," in IEEE Control Systems Letters,
212 * vol. 8, pp. 844-849, 2024, https://ieeexplore.ieee.org/document/10539262
213 *
214 * We choose this->origin_ as the reference, express other density in our
215 * chart, fuse the Gaussians, then reset to a zero-mean concentrated Gaussian.
216 */
217 ConcentratedGaussian operator*(const ConcentratedGaussian& other) const {
218 // 0) Sanity checks
219 if (this->key() != other.key())
220 throw std::invalid_argument(
221 "ConcentratedGaussian::operator*: keys differ");
222 if (this->dim() != other.dim())
223 throw std::invalid_argument(
224 "ConcentratedGaussian::operator*: dimension mismatch");
225
226 // 1) Transport other to our chart
227 ConcentratedGaussian o = other.transportTo(this->origin_);
228
229 // 2) Fuse the Gaussians in our tangent space
230 const auto g1 = this->gaussian("ConcentratedGaussian::operator*", true);
231 const auto g2 = o.gaussian("ConcentratedGaussian::operator*", true);
232 auto [m, P] = Fuse(g1: *g1, g2: *g2);
233
234 // 3) Create a fused ConcentratedGaussian at our origin with fused mean
235 ConcentratedGaussian ecg(this->key(), this->origin_, m, P);
236
237 // 4) Reset to zero mean
238 return ecg.reset();
239 }
240
241 /// @}
242
243 private:
244 /// Helper function to symmetrize a covariance matrix.
245 static Matrix Symmetrize(const Matrix& matrix) {
246 return 0.5 * (matrix + matrix.transpose());
247 }
248
249 /// Fuse two Gaussian distributions into a single Gaussian.
250 static Gaussian Fuse(const Gaussian& g1, const Gaussian& g2) {
251 const Matrix& P1 = g1.second;
252 const Matrix& P2 = g2.second;
253 const Vector& m1 = g1.first;
254 const Vector& m2 = g2.first;
255
256 const Matrix W1 = P1.inverse();
257 const Matrix W2 = P2.inverse();
258 const Matrix P = (W1 + W2).inverse();
259 const Vector m = P * (W1 * m1 + W2 * m2);
260 return {m, Symmetrize(matrix: P)};
261 }

271#endif
