91/// Extract all Point2 values into a single matrix [x y]
92Matrix extractPoint2(const Values& values) {
93 const auto points = values.extract<gtsam::Point2>();
94 // Point2 is aliased as a gtsam::Vector in the wrapper
95 const auto points2 = values.extract<gtsam::Vector>();
96
97 Matrix result(points.size() + points2.size(), 2);
98
99 size_t j = 0;
100 for (const auto& key_value : points) {
101 result.row(i: j++) = key_value.second;
102 }
103 for (const auto& key_value : points2) {
104 if (key_value.second.rows() == 2) {
105 result.row(i: j++) = key_value.second;
106 }
107 }
108 return result;
109}

157/// Extract all Pose3 values into a single matrix [r11 r12 r13 r21 r22 r23 r31 r32 r33 x y z]
158Matrix extractPose3(const Values& values) {
159 const auto poses = values.extract<Pose3>();
160 Matrix result(poses.size(), 12);
161 size_t j = 0;
162 for(const auto& key_value: poses) {
163 result.row(i: j).segment(start: 0, n: 3) << key_value.second.rotation().matrix().row(i: 0);
164 result.row(i: j).segment(start: 3, n: 3) << key_value.second.rotation().matrix().row(i: 1);
165 result.row(i: j).segment(start: 6, n: 3) << key_value.second.rotation().matrix().row(i: 2);
166 result.row(i: j).tail(n: 3) = key_value.second.translation();
167 j++;
168 }
169 return result;
170}

172/// Extract all Vector values with a given symbol character into an mxn matrix,
173/// where m is the number of symbols that match the character and n is the
174/// dimension of the variables. If not all variables have dimension n, then a
175/// runtime error will be thrown. The order of returned values are sorted by
176/// the symbol.
177/// For example, calling extractVector(values, 'x'), where values contains 200
178/// variables x1, x2, ..., x200 of type Vector each 5-dimensional, will return a
179/// 200x5 matrix with row i containing xi.
180Matrix extractVectors(const Values& values, char c) {
181 const auto vectors = values.extract<Vector>(filterFcn: Symbol::ChrTest(c));
182 if (vectors.size() == 0) {
183 return Matrix();
184 }
185 auto dim = vectors.begin()->second.size();
186 Matrix result(vectors.size(), dim);
187 Eigen::Index rowi = 0;
188 for (const auto& kv : vectors) {
189 if (kv.second.size() != dim) {
190 throw std::runtime_error(
191 "Tried to extract different-sized vectors into a single matrix");
192 }
193 result.row(i: rowi) = kv.second;
194 ++rowi;
195 }
196 return result;
197}

199/// Perturb all Point2 values using normally distributed noise
200void perturbPoint2(Values& values, double sigma, int32_t seed = 42u) {
201 noiseModel::Isotropic::shared_ptr model =
202 noiseModel::Isotropic::Sigma(dim: 2, sigma);
203 Sampler sampler(model, seed);
204 for (const auto& key_value : values.extract<Point2>()) {
205 values.update<Point2>(j: key_value.first,
206 val: key_value.second + Point2(sampler.sample()));
207 }
208 for (const auto& key_value : values.extract<gtsam::Vector>()) {
209 if (key_value.second.rows() == 2) {
210 values.update<gtsam::Vector>(j: key_value.first,
211 val: key_value.second + Point2(sampler.sample()));
212 }
213 }
214}

244/**
245 * @brief Insert a number of initial point values by backprojecting
246 *
247 * @param values The values dict to insert the backprojections to.
248 * @param camera The camera model.
249 * @param J Vector of key indices.
250 * @param Z 2*J matrix of pixel values.
251 * @param depth Initial depth value.
252 */
253void insertBackprojections(Values& values, const PinholeCamera<Cal3_S2>& camera,
254 const Vector& J, const Matrix& Z, double depth) {
255 if (Z.rows() != 2)
256 throw std::invalid_argument("insertBackProjections: Z must be 2*J");
257 if (Z.cols() != J.size())
258 throw std::invalid_argument(
259 "insertBackProjections: J and Z must have same number of entries");
260 for (int k = 0; k < Z.cols(); k++) {
261 Point2 p(Z(0, k), Z(1, k));
262 Point3 P = camera.backproject(p, depth);
263 values.insert(j: J(k), val: P);
264 }
265}

267/**
268 * @brief Insert multiple projection factors for a single pose key
269 *
270 * @param graph The nonlinear factor graph to add the factors to.
271 * @param i Camera key.
272 * @param J Vector of key indices.
273 * @param Z 2*J matrix of pixel values.
274 * @param model Factor noise model.
275 * @param K Calibration matrix.
276 * @param body_P_sensor Pose of the camera sensor in the body frame.
277 */
278void insertProjectionFactors(NonlinearFactorGraph& graph, Key i,
279 const Vector& J, const Matrix& Z, const SharedNoiseModel& model,
280 const Cal3_S2::shared_ptr K, const Pose3& body_P_sensor = Pose3()) {
281 if (Z.rows() != 2)
282 throw std::invalid_argument("addMeasurements: Z must be 2*K");
283 if (Z.cols() != J.size())
284 throw std::invalid_argument(
285 "addMeasurements: J and Z must have same number of entries");
286 for (int k = 0; k < Z.cols(); k++) {
287 graph.push_back(
288 factor: std::make_shared<GenericProjectionFactor<Pose3, Point3> >(
289 args: Point2(Z(0, k), Z(1, k)), args: model, args&: i, args: Key(J(k)), args: K, args: body_P_sensor));
290 }
291}

293/// Calculate the errors of all projection factors in a graph
294Matrix reprojectionErrors(const NonlinearFactorGraph& graph,
295 const Values& values) {
296 // first count
297 size_t K = 0, k = 0;
298 for(const NonlinearFactor::shared_ptr& f: graph)
299 if (std::dynamic_pointer_cast<const GenericProjectionFactor<Pose3, Point3> >(
300 r: f))
301 ++K;
302 // now fill
303 Matrix errors(2, K);
304 for(const NonlinearFactor::shared_ptr& f: graph) {
305 std::shared_ptr<const GenericProjectionFactor<Pose3, Point3> > p =
306 std::dynamic_pointer_cast<const GenericProjectionFactor<Pose3, Point3> >(
307 r: f);
308 if (p)
309 errors.col(i: k++) = p->unwhitenedError(x: values);
310 }
311 return errors;
312}

314/// Convert from local to world coordinates
315Values localToWorld(const Values& local, const Pose2& base,
316 const KeyVector user_keys = KeyVector()) {
317
318 Values world;
319
320 // if no keys given, get all keys from local values
321 KeyVector keys(user_keys);
322 if (keys.size()==0)
323 keys = local.keys();
324
325 // Loop over all keys
326 for(Key key: keys) {
327 try {
328 // if value is a Pose2, compose it with base pose
329 Pose2 pose = local.at<Pose2>(j: key);
330 world.insert(j: key, val: base.compose(g: pose));
331 } catch ([[maybe_unused]] const std::exception& e1) {
332 try {
333 // if value is a Point2, transform it from base pose
334 Point2 point = local.at<Point2>(j: key);
335 world.insert(j: key, val: base.transformFrom(point));
336 } catch ([[maybe_unused]] const std::exception& e2) {
337 // if not Pose2 or Point2, do nothing
338 #ifndef NDEBUG
339 std::cerr << "Values[key] is neither Pose2 nor Point2, so skip" << std::endl;
340 #endif
341 }
342 }
343 }
344 return world;
345}
