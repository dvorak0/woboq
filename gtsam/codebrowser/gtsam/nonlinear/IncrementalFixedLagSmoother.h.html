21#pragma once

23#include <gtsam/nonlinear/FixedLagSmoother.h>
24#include <gtsam/nonlinear/ISAM2.h>
25#include "gtsam/dllexport.h"

27namespace gtsam {
28
29/**
30 * This is a base class for the various HMF2 implementations. The HMF2 eliminates the factor graph
31 * such that the active states are placed in/near the root. This base class implements a function
32 * to calculate the ordering, and an update function to incorporate new factors into the HMF.
33 */
34class GTSAM_EXPORT IncrementalFixedLagSmoother: public FixedLagSmoother {
35
36public:
37
38 /// Typedef for a shared pointer to an Incremental Fixed-Lag Smoother
39 typedef std::shared_ptr<IncrementalFixedLagSmoother> shared_ptr;
40
41 /** default constructor */
42 IncrementalFixedLagSmoother(double smootherLag = 0.0,
43 const ISAM2Params& parameters = DefaultISAM2Params()) :
44 FixedLagSmoother(smootherLag), isam_(parameters) {
45 }
46
47 /** destructor */
48 ~IncrementalFixedLagSmoother() override {
49 }
50
51 /** Print the factor for debugging and testing (implementing Testable) */
52 void print(const std::string& s = "IncrementalFixedLagSmoother:\n",
53 const KeyFormatter& keyFormatter = DefaultKeyFormatter) const override;
54
55 /** Check if two IncrementalFixedLagSmoother Objects are equal */
56 bool equals(const FixedLagSmoother& rhs, double tol = 1e-9) const override;
57
58 /**
59 * Add new factors, updating the solution and re-linearizing as needed.
60 * @param newFactors new factors on old and/or new variables
61 * @param newTheta new values for new variables only
62 * @param timestamps an (optional) map from keys to real time stamps
63 * @param factorsToRemove an (optional) list of factors to remove.
64 */
65 Result update(const NonlinearFactorGraph& newFactors = NonlinearFactorGraph(),
66 const Values& newTheta = Values(), //
67 const KeyTimestampMap& timestamps = KeyTimestampMap(),
68 const FactorIndices& factorsToRemove = FactorIndices()) override;
69
70 /** Compute an estimate from the incomplete linear delta computed during the last update.
71 * This delta is incomplete because it was not updated below wildfire_threshold. If only
72 * a single variable is needed, it is faster to call calculateEstimate(const KEY&).
73 */
74 Values calculateEstimate() const override {
75 return isam_.calculateEstimate();
76 }
77
78 /** Compute an estimate for a single variable using its incomplete linear delta computed
79 * during the last update. This is faster than calling the no-argument version of
80 * calculateEstimate, which operates on all variables.
81 * @param key
82 * @return
83 */
84 template<class VALUE>
85 VALUE calculateEstimate(Key key) const {
86 return isam_.calculateEstimate<VALUE>(key);
87 }
88
89 /** return the current set of iSAM2 parameters */
90 const ISAM2Params& params() const {
91 return isam_.params();
92 }
93
94 /** Access the current set of factors */
95 const NonlinearFactorGraph& getFactors() const {
96 return isam_.getFactorsUnsafe();
97 }
98
99 /** Access the current linearization point */
100 const Values& getLinearizationPoint() const {
101 return isam_.getLinearizationPoint();
102 }
103
104 /** Access the current set of deltas to the linearization point */
105 const VectorValues& getDelta() const {
106 return isam_.getDelta();
107 }
108
109 /// Calculate marginal covariance on given variable
110 Matrix marginalCovariance(Key key) const {
111 return isam_.marginalCovariance(key);
112 }
113
114 /// Get results of latest isam2 update
115 const ISAM2Result& getISAM2Result() const{ return isamResult_; }
116
117 /// Get the iSAM2 object which is used for the inference internally
118 const ISAM2& getISAM2() const { return isam_; }
119
120protected:
121
122 /** Create default parameters */
123 static ISAM2Params DefaultISAM2Params() {
124 ISAM2Params params;
125 params.findUnusedFactorSlots = true;
126 return params;
127 }
128
129 /** An iSAM2 object used to perform inference. The smoother lag is controlled
130 * by what factors are removed each iteration */
131 ISAM2 isam_;
132
133 /** Store results of latest isam2 update */
134 ISAM2Result isamResult_;
135
136 /** Erase any keys associated with timestamps before the provided time */
137 void eraseKeysBefore(double timestamp);
138
139 /** Fill in an iSAM2 ConstrainedKeys structure such that the provided keys are eliminated before all others */
140 void createOrderingConstraints(const KeyVector& marginalizableKeys,
141 std::optional<FastMap<Key, int> >& constrainedKeys) const;
142
143private:
144 /** Private methods for printing debug information */
145 static void PrintKeySet(const std::set<Key>& keys, const std::string& label =
146 "Keys:");
147 static void PrintSymbolicFactor(const GaussianFactor::shared_ptr& factor);
148 static void PrintSymbolicGraph(const GaussianFactorGraph& graph,
149 const std::string& label = "Factor Graph:");
150 static void PrintSymbolicTree(const gtsam::ISAM2& isam,
151 const std::string& label = "Bayes Tree:");
152 static void PrintSymbolicTreeHelper(
153 const gtsam::ISAM2Clique::shared_ptr& clique, const std::string indent =
154 "");
155
156};
157// IncrementalFixedLagSmoother
158
159}/// namespace gtsam
