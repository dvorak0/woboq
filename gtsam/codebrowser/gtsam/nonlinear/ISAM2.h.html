34/**
35 * @ingroup isam2
36 * Implementation of the full ISAM2 algorithm for incremental nonlinear
37 * optimization.
38 *
39 * The typical cycle of using this class to create an instance by providing
40 * ISAM2Params to the constructor, then add measurements and variables as they
41 * arrive using the update() method. At any time, calculateEstimate() may be
42 * called to obtain the current estimate of all variables.
43 *
44 */
45class GTSAM_EXPORT ISAM2 : public BayesTree<ISAM2Clique> {
46 protected:
47 /** The current linearization point */
48 Values theta_;
49
50 /** VariableIndex lets us look up factors by involved variable and keeps track
51 * of dimensions */
52 VariableIndex variableIndex_;
53
54 /** The linear delta from the last linear solution, an update to the estimate
55 * in theta
56 *
57 * This is \c mutable because it is a "cached" variable - it is not updated
58 * until either requested with getDelta() or calculateEstimate(), or needed
59 * during update() to evaluate whether to relinearize variables.
60 */
61 mutable VectorValues delta_;
62
63 mutable VectorValues deltaNewton_; // Only used when using Dogleg - stores
64 // the Gauss-Newton update
65 mutable VectorValues RgProd_; // Only used when using Dogleg - stores R*g and
66 // is updated incrementally
67
68 /** A cumulative mask for the variables that were replaced and have not yet
69 * been updated in the linear solution delta_, this is only used internally,
70 * delta will always be updated if necessary when requested with getDelta()
71 * or calculateEstimate().
72 *
73 * This is \c mutable because it is used internally to not update delta_
74 * until it is needed.
75 */
76 mutable KeySet deltaReplacedMask_; // TODO(dellaert): Make sure accessed in
77 // the right way
78
79 /** All original nonlinear factors are stored here to use during
80 * relinearization */
81 NonlinearFactorGraph nonlinearFactors_;
82
83 /** The current linear factors, which are only updated as needed */
84 mutable GaussianFactorGraph linearFactors_;
85
86 /** The current parameters */
87 ISAM2Params params_;
88
89 /** The current Dogleg Delta (trust region radius) */
90 mutable std::optional<double> doglegDelta_;
91
92 /** Set of variables that are involved with linear factors from marginalized
93 * variables and thus cannot have their linearization points changed. */
94 KeySet fixedVariables_;
95
96 int update_count_; ///< Counter incremented every update(), used to determine
97 ///< periodic relinearization
98
99 public:
100 using This = ISAM2; ///< This class
101 using Base = BayesTree<ISAM2Clique>; ///< The BayesTree base class
102 using Clique = Base::Clique; ///< A clique
103 using sharedClique = Base::sharedClique; ///< Shared pointer to a clique
104 using Cliques = Base::Cliques; ///< List of Cliques

109 /** Create an empty ISAM2 instance using the default set of parameters (see

119 /**
120 * Add new factors, updating the solution and relinearizing as needed.
121 *
122 * Optionally, this function remove existing factors from the system to enable
123 * behaviors such as swapping existing factors with new ones.
124 *
125 * Add new measurements, and optionally new variables, to the current system.
126 * This runs a full step of the ISAM2 algorithm, relinearizing and updating
127 * the solution as needed, according to the wildfire and relinearize
128 * thresholds.
129 *
130 * @param newFactors The new factors to be added to the system
131 * @param newTheta Initialization points for new variables to be added to the
132 * system. You must include here all new variables occuring in newFactors
133 * (which were not already in the system). There must not be any variables
134 * here that do not occur in newFactors, and additionally, variables that were
135 * already in the system must not be included here.
136 * @param removeFactorIndices Indices of factors to remove from system
137 * @param force_relinearize Relinearize any variables whose delta magnitude is
138 * sufficiently large (Params::relinearizeThreshold), regardless of the
139 * relinearization interval (Params::relinearizeSkip).
140 * @param constrainedKeys is an optional map of keys to group labels, such
141 * that a variable can be constrained to a particular grouping in the
142 * BayesTree
143 * @param noRelinKeys is an optional set of nonlinear keys that iSAM2 will
144 * hold at a constant linearization point, regardless of the size of the
145 * linear delta
146 * @param extraReelimKeys is an optional set of nonlinear keys that iSAM2 will
147 * re-eliminate, regardless of the size of the linear delta. This allows the
148 * provided keys to be reordered.
149 * @return An ISAM2Result struct containing information about the update
150 */
151 virtual ISAM2Result update(
152 const NonlinearFactorGraph& newFactors = NonlinearFactorGraph(),
153 const Values& newTheta = Values(),
154 const FactorIndices& removeFactorIndices = FactorIndices(),
155 const std::optional<FastMap<Key, int> >& constrainedKeys = {},
156 const std::optional<FastList<Key> >& noRelinKeys = {},
157 const std::optional<FastList<Key> >& extraReelimKeys = {},
158 bool force_relinearize = false);
159
160 /**
161 * Add new factors, updating the solution and relinearizing as needed.
162 *
163 * Alternative signature of update() (see its documentation above), with all
164 * additional parameters in one structure. This form makes easier to keep
165 * future API/ABI compatibility if parameters change.
166 *
167 * @param newFactors The new factors to be added to the system
168 * @param newTheta Initialization points for new variables to be added to the
169 * system. You must include here all new variables occuring in newFactors
170 * (which were not already in the system). There must not be any variables
171 * here that do not occur in newFactors, and additionally, variables that were
172 * already in the system must not be included here.
173 * @param updateParams Additional parameters to control relinearization,
174 * constrained keys, etc.
175 * @return An ISAM2Result struct containing information about the update
176 * @note No default parameters to avoid ambiguous call errors.
177 */
178 virtual ISAM2Result update(const NonlinearFactorGraph& newFactors,
179 const Values& newTheta,
180 const ISAM2UpdateParams& updateParams);
181
182 /** Marginalize out variables listed in leafKeys. These keys must be leaves
183 * in the BayesTree. Throws MarginalizeNonleafException if non-leaves are
184 * requested to be marginalized. Marginalization leaves a linear
185 * approximation of the marginal in the system, and the linearization points
186 * of any variables involved in this linear marginal become fixed. The set
187 * fixed variables will include any key involved with the marginalized
188 * variables in the original factors, and possibly additional ones due to
189 * fill-in.
190 *
191 * If provided, 'marginalFactorsIndices' will be augmented with the factor
192 * graph indices of the marginal factors added during the 'marginalizeLeaves'
193 * call
194 *
195 * If provided, 'deletedFactorsIndices' will be augmented with the factor
196 * graph indices of any factor that was removed during the 'marginalizeLeaves'
197 * call
198 */
199 void marginalizeLeaves(
200 const FastList<Key>& leafKeys,
201 FactorIndices* marginalFactorsIndices = nullptr,
202 FactorIndices* deletedFactorsIndices = nullptr);
203
204 /** An overload of marginalizeLeaves that takes references
205 * to vectors instead of pointers to vectors and passes
206 * it to the pointer version of the function.
207 */
208 template <class... OptArgs>
209 void marginalizeLeaves(const FastList<Key>& leafKeys,
210 OptArgs&&... optArgs) {
211 // dereference the optional arguments and pass
212 // it to the pointer version
213 marginalizeLeaves(leafKeys, (&optArgs)...);
214 }

216 /// Access the current linearization point
217 const Values& getLinearizationPoint() const { return theta_; }
218
219 /// Check whether variable with given key exists in linearization point
220 bool valueExists(Key key) const { return theta_.exists(j: key); }
221
222 /** Compute an estimate from the incomplete linear delta computed during the
223 * last update. This delta is incomplete because it was not updated below
224 * wildfire_threshold. If only a single variable is needed, it is faster to
225 * call calculateEstimate(const KEY&).
226 */
227 Values calculateEstimate() const;
228
229 /** Compute an estimate for a single variable using its incomplete linear
230 * delta computed during the last update. This is faster than calling the
231 * no-argument version of calculateEstimate, which operates on all variables.
232 * @param key
233 * @return
234 */
235 template <class VALUE>
236 VALUE calculateEstimate(Key key) const {
237 const Vector& delta = getDelta()[key];
238 return traits<VALUE>::Retract(theta_.at<VALUE>(key), delta);
239 }
240
241 /** Compute an estimate for a single variable using its incomplete linear
242 * delta computed during the last update. This is faster than calling the
243 * no-argument version of calculateEstimate, which operates on all variables.
244 * This is a non-templated version that returns a Value base class for use
245 * with the MATLAB wrapper.
246 * @param key
247 * @return
248 */
249 const Value& calculateEstimate(Key key) const;
250
251 /** Return marginal on any variable as a covariance matrix */
252 Matrix marginalCovariance(Key key) const;

284 /** Compute the gradient of the energy function, \f$ \nabla_{x=0} \left\Vert
285 * \Sigma^{-1} R x - d \right\Vert^2 \f$, centered around zero. The gradient
286 * about zero is \f$ -R^T d \f$. See also gradient(const GaussianBayesNet&,
287 * const VectorValues&).
288 *
289 * @return A VectorValues storing the gradient.
290 */
291 VectorValues gradientAtZero() const;

297 void recalculate(const ISAM2UpdateParams& updateParams,
298 const KeySet& relinKeys, ISAM2Result* result);

310 /**
311 * @brief Perform an incremental update of the factor graph to return a new
312 * Bayes Tree with affected keys.
313 *
314 * @param updateParams Parameters for the ISAM2 update.
315 * @param relinKeys Keys of variables to relinearize.
316 * @param affectedKeys The set of keys which are affected in the update.
317 * @param affectedKeysSet [output] Affected and contaminated keys.
318 * @param orphans [output] List of orphanes cliques after elimination.
319 * @param result [output] The result of the incremental update step.
320 */

322 const KeySet& relinKeys,
