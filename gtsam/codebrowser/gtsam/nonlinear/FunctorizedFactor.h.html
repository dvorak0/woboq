27/**
28 * Factor which evaluates provided unary functor and uses the result to compute
29 * error with respect to the provided measurement.
30 *
31 * Template parameters are
32 * @param R: The return type of the functor after evaluation.
33 * @param T: The argument type for the functor.
34 *
35 * Example:
36 * Key key = Symbol('X', 0);
37 * auto model = noiseModel::Isotropic::Sigma(9, 1);
38 *
39 * /// Functor that takes a matrix and multiplies every element by m
40 * class MultiplyFunctor {
41 * double m_; ///< simple multiplier
42 * public:
43 * MultiplyFunctor(double m) : m_(m) {}
44 * Matrix operator()(const Matrix &X,
45 * OptionalJacobian<-1, -1> H = {}) const {
46 * if (H)
47 * *H = m_ * Matrix::Identity(X.rows()*X.cols(), X.rows()*X.cols());
48 * return m_ * X;
49 * }
50 * };
51 *
52 * Matrix measurement = Matrix::Identity(3, 3);
53 * double multiplier = 2.0;
54 *
55 * FunctorizedFactor<Matrix, Matrix> factor(keyX, measurement, model,
56 * MultiplyFunctor(multiplier));
57 */
58template <typename R, typename T>
59class FunctorizedFactor : public NoiseModelFactorN<T> {
60 private:
61 using Base = NoiseModelFactorN<T>;
62
63 R measured_; ///< value that is compared with functor return value
64 SharedNoiseModel noiseModel_; ///< noise model
65 std::function<R(T, OptionalMatrixType)> func_; ///< functor instance
66
67 public:
68
69 // Provide access to the Matrix& version of evaluateError:
70 using Base::evaluateError;
71
72 /** default constructor - only use for serialization */
73 FunctorizedFactor() {}
74
75 /** Construct with given x and the parameters of the basis
76 *
77 * @param key: Factor key
78 * @param z: Measurement object of same type as that returned by functor
79 * @param model: Noise model
80 * @param func: The instance of the functor object
81 */
82 FunctorizedFactor(Key key, const R &z, const SharedNoiseModel &model,
83 const std::function<R(T, OptionalMatrixType)> func)
84 : Base(model, key), measured_(z), noiseModel_(model), func_(func) {}
85
86 ~FunctorizedFactor() override {}
87
88 /// @return a deep copy of this factor
89 NonlinearFactor::shared_ptr clone() const override {
90 return std::static_pointer_cast<NonlinearFactor>(
91 r: NonlinearFactor::shared_ptr(new FunctorizedFactor<R, T>(*this)));
92 }
93
94 Vector evaluateError(const T &params, OptionalMatrixType H) const override {
95 R x = func_(params, H);
96 Vector error = traits<R>::Local(measured_, x);
97 return error;
98 }
99
100 /// @name Testable
101 /// @{
102 void print(
103 const std::string &s = "",
104 const KeyFormatter &keyFormatter = DefaultKeyFormatter) const override {
105 Base::print(s, keyFormatter);
106 std::cout << s << (s != "" ? " " : "") << "FunctorizedFactor("
107 << keyFormatter(this->key1()) << ")" << std::endl;
108 traits<R>::Print(measured_, " measurement: ");
109 std::cout << " noise model sigmas: " << noiseModel_->sigmas().transpose()
110 << std::endl;
111 }
112
113 bool equals(const NonlinearFactor &other, double tol = 1e-9) const override {
114 const FunctorizedFactor<R, T> *e =
115 dynamic_cast<const FunctorizedFactor<R, T> *>(&other);
116 return e != nullptr && Base::equals(other, tol) &&
117 traits<R>::Equals(this->measured_, e->measured_, tol);
118 }
119 /// @}
120
121 private:
122#if GTSAM_ENABLE_BOOST_SERIALIZATION
123 /** Serialization function */
124 friend class boost::serialization::access;
125 template <class ARCHIVE>
126 void serialize(ARCHIVE &ar, const unsigned int /*version*/) {
127 // NoiseModelFactor1 instead of NoiseModelFactorN for backward compatibility
128 ar &boost::serialization::make_nvp(
129 "NoiseModelFactor1", boost::serialization::base_object<Base>(*this));
130 ar &BOOST_SERIALIZATION_NVP(measured_);
131 ar &BOOST_SERIALIZATION_NVP(func_);
132 }
133#endif
134};

136/// traits
137template <typename R, typename T>
138struct traits<FunctorizedFactor<R, T>>
139 : public Testable<FunctorizedFactor<R, T>> {};

141/**
142 * Helper function to create a functorized factor.
143 *
144 * Uses function template deduction to identify return type and functor type, so
145 * template list only needs the functor argument type.
146 */
147template <typename T, typename R, typename FUNC>
148FunctorizedFactor<R, T> MakeFunctorizedFactor(Key key, const R &z,
149 const SharedNoiseModel &model,
150 const FUNC func) {
151 return FunctorizedFactor<R, T>(key, z, model, func);
152}

154/**
155 * Factor which evaluates provided binary functor and uses the result to compute
156 * error with respect to the provided measurement.
157 *
158 * Template parameters are
159 * @param R: The return type of the functor after evaluation.
160 * @param T1: The first argument type for the functor.
161 * @param T2: The second argument type for the functor.
162 */
163template <typename R, typename T1, typename T2>
164class FunctorizedFactor2 : public NoiseModelFactorN<T1, T2> {
165 private:
166 using Base = NoiseModelFactorN<T1, T2>;
167
168 R measured_; ///< value that is compared with functor return value
169 SharedNoiseModel noiseModel_; ///< noise model
170 using FunctionType = std::function<R(T1, T2, OptionalMatrixType, OptionalMatrixType)>;
171 FunctionType func_; ///< functor instance
172
173 public:
174
175 // Provide access to the Matrix& version of evaluateError:
176 using Base::evaluateError;
177
178 /** default constructor - only use for serialization */
179 FunctorizedFactor2() {}
180
181 /** Construct with given x and the parameters of the basis
182 *
183 * @param key: Factor key
184 * @param z: Measurement object of same type as that returned by functor
185 * @param model: Noise model
186 * @param func: The instance of the functor object
187 */
188 FunctorizedFactor2(Key key1, Key key2, const R &z,
189 const SharedNoiseModel &model, const FunctionType func)
190 : Base(model, key1, key2),
191 measured_(z),
192 noiseModel_(model),
193 func_(func) {}
194
195 ~FunctorizedFactor2() override {}
196
197 /// @return a deep copy of this factor
198 NonlinearFactor::shared_ptr clone() const override {
199 return std::static_pointer_cast<NonlinearFactor>(
200 r: NonlinearFactor::shared_ptr(new FunctorizedFactor2<R, T1, T2>(*this)));
201 }
202
203 Vector evaluateError(
204 const T1 &params1, const T2 &params2,
205 OptionalMatrixType H1, OptionalMatrixType H2) const override {
206 R x = func_(params1, params2, H1, H2);
207 Vector error = traits<R>::Local(measured_, x);
208 return error;
209 }
210
211 /// @name Testable
212 /// @{
213 void print(
214 const std::string &s = "",
215 const KeyFormatter &keyFormatter = DefaultKeyFormatter) const override {
216 Base::print(s, keyFormatter);
217 std::cout << s << (s != "" ? " " : "") << "FunctorizedFactor2("
218 << keyFormatter(this->key1()) << ", "
219 << keyFormatter(this->key2()) << ")" << std::endl;
220 traits<R>::Print(measured_, " measurement: ");
221 std::cout << " noise model sigmas: " << noiseModel_->sigmas().transpose()
222 << std::endl;
223 }
224
225 bool equals(const NonlinearFactor &other, double tol = 1e-9) const override {
226 const FunctorizedFactor2<R, T1, T2> *e =
227 dynamic_cast<const FunctorizedFactor2<R, T1, T2> *>(&other);
228 return e && Base::equals(other, tol) &&
229 traits<R>::Equals(this->measured_, e->measured_, tol);
230 }
231 /// @}
232
233 private:
234#if GTSAM_ENABLE_BOOST_SERIALIZATION
235 /** Serialization function */
236 friend class boost::serialization::access;
237 template <class ARCHIVE>
238 void serialize(ARCHIVE &ar, const unsigned int /*version*/) {
239 // NoiseModelFactor2 instead of NoiseModelFactorN for backward compatibility
240 ar &boost::serialization::make_nvp(
241 "NoiseModelFactor2", boost::serialization::base_object<Base>(*this));
242 ar &BOOST_SERIALIZATION_NVP(measured_);
243 ar &BOOST_SERIALIZATION_NVP(func_);
244 }
245#endif
246};
