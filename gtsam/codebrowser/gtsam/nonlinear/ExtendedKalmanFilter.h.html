25namespace gtsam {

27/**
28 * This is a generic Extended Kalman Filter class implemented using nonlinear factors. GTSAM
29 * basically does SRIF with Cholesky to solve the filter problem, making this an efficient,
30 *numerically
31 * stable Kalman Filter implementation.
32 *
33 * The Kalman Filter relies on two models: a motion model that predicts the next state using
34 * the current state, and a measurement model that predicts the measurement value at a given
35 * state. Because these two models are situation-dependent, base classes for each have been
36 * provided above, from which the user must derive a class and incorporate the actual modeling
37 * equations.
38 *
39 * The class provides a "predict" and "update" function to perform these steps independently.
40 * TODO: a "predictAndUpdate" that combines both steps for some computational savings.
41 * \nosubgrouping
42 */
43
44template <class VALUE>
45class ExtendedKalmanFilter {
46 // Check that VALUE type is a testable Manifold
47 GTSAM_CONCEPT_ASSERT(IsTestable<VALUE>);
48 GTSAM_CONCEPT_ASSERT(IsManifold<VALUE>);
49
50 public:
51 typedef std::shared_ptr<ExtendedKalmanFilter<VALUE> > shared_ptr;
52 typedef VALUE T;
53
54 protected:
55 T x_; // linearization point
56 JacobianFactor::shared_ptr priorFactor_; // Gaussian density on x_
57
58 static T solve_(const GaussianFactorGraph& linearFactorGraph, const Values& linearizationPoints,
59 Key x, JacobianFactor::shared_ptr* newPrior);
60
61 public:
62 /// @name Standard Constructors
63 /// @{
64
65 ExtendedKalmanFilter(Key key_initial, T x_initial, noiseModel::Gaussian::shared_ptr P_initial);
66
67 /// @}
68 /// @name Testable
69 /// @{
70
71 /// print
72 void print(const std::string& s = "") const {
73 std::cout << s << "\n";
74 x_.print(s + "x");
75 priorFactor_->print(s: s + "density");
76 }
77
78 /// @}
79 /// @name Interface
80 /// @{
81
82 /**
83 * Calculate predictive density
84 * \f$ P(x_) ~ \int P(x_min) P(x_min, x_)\f$
85 * The motion model should be given as a factor with key1 for \f$x_min\f$ and key2 for \f$x_\f$
86 */
87 T predict(const NoiseModelFactor& motionFactor);
88
89 /**
90 * Calculate posterior density P(x_) ~ L(z|x) P(x)
91 * The likelihood L(z|x) should be given as a unary factor on x
92 */
93 T update(const NoiseModelFactor& measurementFactor);
94
95 /// Return current predictive (if called after predict)/posterior (if called after update)
96 const JacobianFactor::shared_ptr Density() const {
97 return priorFactor_;
98 }
99
100 /// @}
101};

103} // namespace
