65bool LinearContainerFactor::equals(const NonlinearFactor& f, double tol) const {
66 const LinearContainerFactor* jcf = dynamic_cast<const LinearContainerFactor*>(&f);
67 if (!jcf || !factor_->equals(lf: *jcf->factor_, tol) || !NonlinearFactor::equals(f))
68 return false;
69 if (!linearizationPoint_ && !jcf->linearizationPoint_)
70 return true;
71 if (linearizationPoint_ && jcf->linearizationPoint_)
72 return linearizationPoint_->equals(other: *jcf->linearizationPoint_, tol);
73 return false;
74}

76/* ************************************************************************* */
77double LinearContainerFactor::error(const Values& c) const {
78 if (!linearizationPoint_)
79 return 0;
80
81 // Extract subset of values for comparison
82 Values csub;
83 for (Key key : keys())
84 csub.insert(j: key, val: c.at(j: key));
85
86 // create dummy ordering for evaluation
87 VectorValues delta = linearizationPoint_->localCoordinates(cp: csub);
88
89 // compute error
90 double error = factor_->error(c: delta);
91
92 return error;
93}

103/* ************************************************************************* */
104GaussianFactor::shared_ptr LinearContainerFactor::linearize(const Values& c) const {
105 // Clone factor and update as necessary
106 GaussianFactor::shared_ptr linFactor = factor_->clone();
107 if (!hasLinearizationPoint())
108 return linFactor;
109
110 // Extract subset of values
111 Values subsetC;
112 for (Key key : keys())
113 subsetC.insert(j: key, val: c.at(j: key));
114
115 // Determine delta between linearization points using new ordering
116 VectorValues delta = linearizationPoint_->localCoordinates(cp: subsetC);
117
118 // Apply changes due to relinearization
119 if (isJacobian()) {
120 JacobianFactor::shared_ptr jacFactor = std::dynamic_pointer_cast<JacobianFactor>(r: linFactor);
121 jacFactor->getb() = -jacFactor->unweighted_error(c: delta);
122 } else {
123 HessianFactor::shared_ptr hesFactor = std::dynamic_pointer_cast<HessianFactor>(r: linFactor);
124
125 const auto view = hesFactor->informationView();
126 Vector deltaVector = delta.vector(keys: keys());
127 Vector G_delta = view * deltaVector;
128 hesFactor->constantTerm() += deltaVector.dot(other: G_delta) - 2.0 * deltaVector.dot(other: hesFactor->linearTerm().col(i: 0));
129 hesFactor->linearTerm() -= G_delta;
130 }
131
132 return linFactor;
133}

168NonlinearFactor::shared_ptr LinearContainerFactor::rekey(
169 const std::map<Key, Key>& rekey_mapping) const {
170 auto rekeyed_base_factor = Base::rekey(rekey_mapping);
171 // Update the keys to the properties as well
172 // Downncast so we have access to members
173 auto new_factor = std::static_pointer_cast<LinearContainerFactor>(r: rekeyed_base_factor);
174 // Create a new Values to assign later
175 Values newLinearizationPoint;
176 for (size_t i = 0; i < factor_->size(); ++i) {
177 auto mapping = rekey_mapping.find(x: factor_->keys()[i]);
178 if (mapping != rekey_mapping.end()) {
179 new_factor->factor_->keys()[i] = mapping->second;
180 newLinearizationPoint.insert(j: mapping->second, val: linearizationPoint_->at(j: mapping->first));
181 }
182 }
183 new_factor->linearizationPoint_ = newLinearizationPoint;
184
185 // upcast back and return
186 return std::static_pointer_cast<NonlinearFactor>(r: new_factor);
187}
