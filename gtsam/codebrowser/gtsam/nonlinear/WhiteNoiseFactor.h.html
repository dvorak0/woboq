28 const double logSqrt2PI = log(x: std::sqrt(x: 2.0 * M_PI)); ///< constant needed below
29
30 /**
31 * @brief Binary factor to estimate parameters of zero-mean Gaussian white noise
32 *
33 * This factor uses the mean-precision parameterization.
34 *
35 * Takes three template arguments:
36 * Key: key type to use for mean
37 * Key: key type to use for precision
38 * Values: Values type for optimization
39 * \nosubgrouping
40 */
41 class WhiteNoiseFactor: public NonlinearFactor {
42
43 private:
44
45 double z_; ///< Measurement
46
47 Key meanKey_; ///< key by which to access mean variable
48 Key precisionKey_; ///< key by which to access precision variable
49
50 typedef NonlinearFactor Base;
51
52 public:
53
54 /** @brief negative log likelihood as a function of mean \f$ \mu \f$ and precision \f$ \tau \f$
55 * @f[
56 * f(z, \tau, \mu)
57 * = -\log \left( \frac{\sqrt{\tau}}{\sqrt{2\pi}} \exp(-0.5\tau(z-\mu)^2) \right)
58 * = \log(\sqrt{2\pi}) - 0.5* \log(\tau) + 0.5\tau(z-\mu)^2
59 * @f]
60 */
61 static double f(double z, double u, double p) {
62 return logSqrt2PI - 0.5 * log(x: p) + 0.5 * (z - u) * (z - u) * p;
63 }
64
65 /**
66 * @brief linearize returns a Hessianfactor that approximates error
67 * Hessian is @f[
68 * 0.5f - x^T g + 0.5*x^T G x
69 * @f]
70 * Taylor expansion is @f[
71 * f(x+dx) = f(x) + df(x) dx + 0.5 ddf(x) dx^2
72 * @f]
73 * So f = 2 f(x), g = -df(x), G = ddf(x)
74 */
75 static HessianFactor::shared_ptr linearize(double z, double u, double p,
76 Key j1, Key j2) {
77 double e = u - z, e2 = e * e;
78 double c = 2 * logSqrt2PI - log(x: p) + e2 * p;
79 Vector g1 = (Vector(1) << -e * p).finished();
80 Vector g2 = (Vector(1) << 0.5 / p - 0.5 * e2).finished();
81 Matrix G11 = (Matrix(1, 1) << p).finished();
82 Matrix G12 = (Matrix(1, 1) << e).finished();
83 Matrix G22 = (Matrix(1, 1) << 0.5 / (p * p)).finished();
84 return HessianFactor::shared_ptr(
85 new HessianFactor(j1, j2, G11, G12, g1, G22, g2, c));
86 }
87

89 /// @{
90
91 /** Construct from measurement
92 * @param z Measurment value
93 * @param meanKey Key for mean variable
94 * @param precisionKey Key for precision variable
95 */
96 WhiteNoiseFactor(double z, Key meanKey, Key precisionKey) :
97 Base(), z_(z), meanKey_(meanKey), precisionKey_(precisionKey) {
98 }
99
100 /// @}

116 std::cout << p + ".z: " << z_ << std::endl;
117 }
118
119 /// @}
120 /// @name Standard Interface
121 /// @{
122
123 /// get the dimension of the factor (number of rows on linearization)
124 size_t dim() const override {
125 return 2;
126 }
127
128 /// Calculate the error of the factor, typically equal to log-likelihood
129 double error(const Values& x) const override {
130 return f(z: z_, u: x.at<double>(j: meanKey_), p: x.at<double>(j: precisionKey_));
131 }
132
133 /**
134 * Vector of errors
135 * "unwhitened" does not make sense for this factor
136 * What is meant typically is only "e" above
137 * Here we shoehorn sqrt(2*error(p))
138 * TODO: Where is this used? should disappear.
139 */
140 virtual Vector unwhitenedError(const Values& x) const {
141 return (Vector(1) << std::sqrt(x: 2 * error(x))).finished();
142 }

153 /// @}
154 /// @name Advanced Interface
155 /// @{
156
157 /// linearize returns a Hessianfactor that is an approximation of error(p)
158 std::shared_ptr<GaussianFactor> linearize(const Values& x) const override {
159 double u = x.at<double>(j: meanKey_);
160 double p = x.at<double>(j: precisionKey_);
161 Key j1 = meanKey_;
162 Key j2 = precisionKey_;
163 return linearize(z: z_, u, p, j1, j2);
164 }
