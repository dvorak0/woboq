16NonlinearOptimizerParams::Verbosity NonlinearOptimizerParams::verbosityTranslator(
17 const std::string &src) {
18 std::string s = src;
19 // Convert to upper case
20 std::transform(first: s.begin(), last: s.end(), result: s.begin(), unary_op: ::toupper);
21 if (s == "SILENT")
22 return NonlinearOptimizerParams::SILENT;
23 if (s == "ERROR")
24 return NonlinearOptimizerParams::ERROR;
25 if (s == "VALUES")
26 return NonlinearOptimizerParams::VALUES;
27 if (s == "DELTA")
28 return NonlinearOptimizerParams::DELTA;
29 if (s == "LINEAR")
30 return NonlinearOptimizerParams::LINEAR;
31 if (s == "TERMINATION")
32 return NonlinearOptimizerParams::TERMINATION;
33
34 /* default is silent */
35 return NonlinearOptimizerParams::SILENT;
36}

76
77 //NonlinearOptimizerParams::print(str);
78 std::cout << str << "\n";
79 std::cout << "relative decrease threshold: " << relativeErrorTol << "\n";
80 std::cout << "absolute decrease threshold: " << absoluteErrorTol << "\n";
81 std::cout << " total error threshold: " << errorTol << "\n";
82 std::cout << " maximum iterations: " << maxIterations << "\n";
83 std::cout << " verbosity: " << verbosityTranslator(value: verbosity)

93 break;

96 break;
97 case SEQUENTIAL_QR:
98 std::cout << " linear solver type: SEQUENTIAL QR\n";
99 break;
100 case CHOLMOD:
101 std::cout << " linear solver type: CHOLMOD\n";
102 break;
103 case Iterative:
104 std::cout << " linear solver type: ITERATIVE\n";

126/* ************************************************************************* */
127bool NonlinearOptimizerParams::equals(const NonlinearOptimizerParams& other,
128 double tol) const {
129 // Check for equality of shared ptrs
130 bool iterative_params_equal = iterativeParams == other.iterativeParams;
131 // Check equality of components
132 if (iterativeParams && other.iterativeParams) {
133 iterative_params_equal = iterativeParams->equals(other: *other.iterativeParams);
134 } else {
135 // Check if either is null. If both are null, then true
136 iterative_params_equal = !iterativeParams && !other.iterativeParams;
137 }
138
139 return maxIterations == other.getMaxIterations() &&
140 std::abs(x: relativeErrorTol - other.getRelativeErrorTol()) <= tol &&
141 std::abs(x: absoluteErrorTol - other.getAbsoluteErrorTol()) <= tol &&
142 std::abs(x: errorTol - other.getErrorTol()) <= tol &&
143 verbosityTranslator(value: verbosity) == other.getVerbosity() &&
144 orderingType == other.orderingType && ordering == other.ordering &&
145 linearSolverType == other.linearSolverType && iterative_params_equal;
146}
