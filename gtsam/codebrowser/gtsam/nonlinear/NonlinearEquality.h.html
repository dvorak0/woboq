31/**
32 * An equality factor that forces either one variable to a constant,
33 * or a set of variables to be equal to each other.
34 *
35 * Depending on flag, throws an error at linearization if the constraints are not met.
36 *
37 * Switchable implementation:
38 * - ALLLOW_ERROR : if we allow that there can be nonzero error, does not throw, and uses gain
39 * - ONLY_EXACT : throws error at linearization if not at exact feasible point, and infinite error
40 *
41 * \nosubgrouping
42 */
43template<class VALUE>
44class NonlinearEquality: public NonlinearEqualityConstraint {
45
46public:
47 typedef VALUE T;
48
49private:
50
51 // feasible value
52 T feasible_;
53
54 // error handling flag
55 bool allow_error_;
56
57 // error gain in allow error case
58 double error_gain_;
59
60 // typedef to this class
61 using This = NonlinearEquality<VALUE>;
62
63 // typedef to base class
64 using Base = NonlinearEqualityConstraint;
65
66public:
67
68 /// Function that compares two values.
69 using CompareFunction = std::function<bool(const T&, const T&)>;
70 CompareFunction compare_;
71
72 /// Default constructor - only for serialization
73 NonlinearEquality() {
74 }
75
76 ~NonlinearEquality() override {
77 }
78
79 /// @name Standard Constructors
80 /// @{
81
82 /**
83 * Constructor - forces exact evaluation
84 */
85 NonlinearEquality(Key j, const T& feasible,
86 const CompareFunction &_compare = std::bind(traits<T>::Equals,
87 std::placeholders::_1, std::placeholders::_2, 1e-9)) :
88 Base(noiseModel::Constrained::All(traits<T>::GetDimension(feasible)),
89 KeyVector{j}), feasible_(feasible), allow_error_(false), error_gain_(0.0), //
90 compare_(_compare) {
91 }
92
93 /**
94 * Constructor - allows inexact evaluation
95 */
96 NonlinearEquality(Key j, const T& feasible, double error_gain,
97 const CompareFunction &_compare = std::bind(traits<T>::Equals,
98 std::placeholders::_1, std::placeholders::_2, 1e-9)) :
99 Base(noiseModel::Constrained::All(traits<T>::GetDimension(feasible)),
100 KeyVector{j}), feasible_(feasible), allow_error_(true), error_gain_(error_gain), //
101 compare_(_compare) {
102 }
103
104 Key key() const {
105 return keys().front();
106 }
107
108 /// @}
109 /// @name Testable
110 /// @{
111
112 void print(const std::string& s = "",
113 const KeyFormatter& keyFormatter = DefaultKeyFormatter) const override {
114 std::cout << (s.empty() ? s : s + " ") << "Constraint: on ["
115 << keyFormatter(this->key()) << "]\n";
116 traits<VALUE>::Print(feasible_, "Feasible Point:\n");
117 std::cout << "Variable Dimension: " << traits<T>::GetDimension(feasible_)
118 << std::endl;
119 }
120
121 /** Check if two factors are equal */
122 bool equals(const NonlinearFactor& f, double tol = 1e-9) const override {
123 const This* e = dynamic_cast<const This*>(&f);
124 return e && Base::equals(f) && traits<T>::Equals(feasible_,e->feasible_, tol)
125 && std::abs(error_gain_ - e->error_gain_) < tol;
126 }
127
128 /// @}
129 /// @name Standard Interface
130 /// @{
131
132 /// Actual error function calculation
133 double error(const Values& c) const override {
134 const T& xj = c.at<T>(this->key());
135 Vector e = this->unwhitenedError(c);
136 if (allow_error_ || !compare_(xj, feasible_)) {
137 return error_gain_ * dot(a: e, b: e);
138 } else {
139 return 0.0;
140 }
141 }
142
143 /// Error function
144 Vector evaluateError(const T& xj, OptionalMatrixType H = nullptr) const {
145 const size_t nj = traits<T>::GetDimension(feasible_);
146 if (allow_error_) {
147 if (H)
148 *H = Matrix::Identity(rows: nj,cols: nj); // FIXME: this is not the right linearization for nonlinear compare
149 return traits<T>::Local(xj,feasible_);
150 } else if (compare_(feasible_, xj)) {
151 if (H)
152 *H = Matrix::Identity(rows: nj,cols: nj);
153 return Vector::Zero(size: nj); // set error to zero if equal
154 } else {
155 if (H)
156 throw std::invalid_argument(
157 "Linearization point not feasible for "
158 + DefaultKeyFormatter(this->key()) + "!");
159 return Vector::Constant(size: nj, value: std::numeric_limits<double>::infinity()); // set error to infinity if not equal
160 }
161 }
162
163 Vector unwhitenedError(const Values& x, OptionalMatrixVecType H = nullptr) const override {
164 VALUE x1 = x.at<VALUE>(key());
165 if (H) {
166 return evaluateError(xj: x1, H: &(H->front()));
167 } else {
168 return evaluateError(xj: x1);
169 }
170 }
171
172 /// Linearize is over-written, because base linearization tries to whiten
173 GaussianFactor::shared_ptr linearize(const Values& x) const override {
174 const T& xj = x.at<T>(this->key());
175 Matrix A;
176 Vector b = evaluateError(xj, H: &A);
177 SharedDiagonal model = noiseModel::Constrained::All(dim: b.size());
178 return GaussianFactor::shared_ptr(
179 new JacobianFactor(this->key(), A, b, model));
180 }
181
182 /// @return a deep copy of this factor
183 gtsam::NonlinearFactor::shared_ptr clone() const override {
184 return std::static_pointer_cast<gtsam::NonlinearFactor>(
185 r: gtsam::NonlinearFactor::shared_ptr(new This(*this)));
186 }
187
188 /// @}
189
190 GTSAM_MAKE_ALIGNED_OPERATOR_NEW
191
192private:
193
194#if GTSAM_ENABLE_BOOST_SERIALIZATION ///
195 /// Serialization function
196 friend class boost::serialization::access;
197 template<class ARCHIVE>
198 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
199 // NoiseModelFactor1 instead of NoiseModelFactorN for backward compatibility
200 ar
201 & boost::serialization::make_nvp("NoiseModelFactor1",
202 boost::serialization::base_object<Base>(*this));
203 ar & BOOST_SERIALIZATION_NVP(feasible_);
204 ar & BOOST_SERIALIZATION_NVP(allow_error_);
205 ar & BOOST_SERIALIZATION_NVP(error_gain_);
206 }
207#endif
208
209};
210// \class NonlinearEquality
211
212template <typename VALUE>
213struct traits<NonlinearEquality<VALUE>> : Testable<NonlinearEquality<VALUE>> {};

215/* ************************************************************************* */
216/**
217 * Simple unary equality constraint - fixes a value for a variable
218 */
219template<class VALUE>
220class NonlinearEquality1: public NonlinearEqualityConstraint {
221
222public:
223 typedef VALUE X;
224
225protected:
226 typedef NonlinearEqualityConstraint Base;
227 typedef NonlinearEquality1<VALUE> This;
228
229 /// Default constructor to allow for serialization
230 NonlinearEquality1() {
231 }
232
233 X value_; /// fixed value for variable
234
235 GTSAM_CONCEPT_MANIFOLD_TYPE(X)
236 GTSAM_CONCEPT_TESTABLE_TYPE(X)
237
238public:
239
240 typedef std::shared_ptr<NonlinearEquality1<VALUE> > shared_ptr;
241
242 /**
243 * Constructor
244 * @param value feasible value that values(key) shouild be equal to
245 * @param key the key for the unknown variable to be constrained
246 * @param mu a parameter which really turns this into a strong prior
247 */
248 NonlinearEquality1(const X& value, Key key, double mu = 1000.0)
249 : Base(noiseModel::Constrained::All(traits<X>::GetDimension(value),
250 std::abs(x: mu)),
251 KeyVector{key}),
252 value_(value) {}
253
254 ~NonlinearEquality1() override {
255 }
256
257 /// @return a deep copy of this factor
258 gtsam::NonlinearFactor::shared_ptr clone() const override {
259 return std::static_pointer_cast<gtsam::NonlinearFactor>(
260 r: gtsam::NonlinearFactor::shared_ptr(new This(*this)));
261 }
262
263 Key key() const { return keys().front(); }
264
265 /// g(x) with optional derivative
266 Vector evaluateError(const X& x1, OptionalMatrixType H = nullptr) const {
267 if (H)
268 (*H) = Matrix::Identity(traits<X>::GetDimension(x1),traits<X>::GetDimension(x1));
269 // manifold equivalent of h(x)-z -> log(z,h(x))
270 return traits<X>::Local(value_,x1);
271 }
272
273 Vector unwhitenedError(const Values& x, OptionalMatrixVecType H = nullptr) const override {
274 X x1 = x.at<X>(key());
275 if (H) {
276 return evaluateError(x1, H: &(H->front()));
277 } else {
278 return evaluateError(x1);
279 }
280 }
281
282 /// Print
283 void print(const std::string& s = "",
284 const KeyFormatter& keyFormatter = DefaultKeyFormatter) const override {
285 std::cout << s << ": NonlinearEquality1(" << keyFormatter(this->key())
286 << ")," << "\n";
287 this->noiseModel_->print();
288 traits<X>::Print(value_, "Value");
289 }
290
291 GTSAM_MAKE_ALIGNED_OPERATOR_NEW
292
293private:
294
295#if GTSAM_ENABLE_BOOST_SERIALIZATION ///
296 /// Serialization function
297 friend class boost::serialization::access;
298 template<class ARCHIVE>
299 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
300 // NoiseModelFactor1 instead of NoiseModelFactorN for backward compatibility
301 ar
302 & boost::serialization::make_nvp("NoiseModelFactor1",
303 boost::serialization::base_object<Base>(*this));
304 ar & BOOST_SERIALIZATION_NVP(value_);
305 }
306#endif
307};
308// \NonlinearEquality1

310template <typename VALUE>
311struct traits<NonlinearEquality1<VALUE> >
312 : Testable<NonlinearEquality1<VALUE> > {};

314/* ************************************************************************* */
315/**
316 * Simple binary equality constraint - this constraint forces two variables to
317 * be the same.
318 */
319template <class T>
320class NonlinearEquality2 : public NonlinearEqualityConstraint {
321 protected:
322 typedef NonlinearEqualityConstraint Base;
323 typedef NonlinearEquality2<T> This;
324
325 GTSAM_CONCEPT_MANIFOLD_TYPE(T)
326
327 /// Default constructor to allow for serialization
328 NonlinearEquality2() {}
329
330 public:
331 typedef std::shared_ptr<NonlinearEquality2<T>> shared_ptr;
332
333
334 /**
335 * Constructor
336 * @param key1 the key for the first unknown variable to be constrained
337 * @param key2 the key for the second unknown variable to be constrained
338 * @param mu a parameter which really turns this into a strong prior
339 */
340 NonlinearEquality2(Key key1, Key key2, double mu = 1e4)
341 : Base(noiseModel::Constrained::All(traits<T>::dimension, std::abs(x: mu)),
342 KeyVector{key1, key2}) {}
343 ~NonlinearEquality2() override {}
344
345 /// @return a deep copy of this factor
346 gtsam::NonlinearFactor::shared_ptr clone() const override {
347 return std::static_pointer_cast<gtsam::NonlinearFactor>(
348 r: gtsam::NonlinearFactor::shared_ptr(new This(*this)));
349 }
350
351 /// g(x) with optional derivative2
352 Vector evaluateError(
353 const T& x1, const T& x2, OptionalMatrixType H1 = nullptr, OptionalMatrixType H2 = nullptr) const {
354 static const size_t p = traits<T>::dimension;
355 if (H1) *H1 = -Matrix::Identity(rows: p, cols: p);
356 if (H2) *H2 = Matrix::Identity(rows: p, cols: p);
357 return traits<T>::Local(x1, x2);
358 }
359
360 Vector unwhitenedError(const Values& x, OptionalMatrixVecType H = nullptr) const override {
361 T x1 = x.at<T>(keys().front());
362 T x2 = x.at<T>(keys().back());
363 if (H) {
364 return evaluateError(x1, x2, H1: &(H->front()), H2: &(H->back()));
365 } else {
366 return evaluateError(x1, x2);
367 }
368 }
369
370 GTSAM_MAKE_ALIGNED_OPERATOR_NEW
371
372 private:
373#if GTSAM_ENABLE_BOOST_SERIALIZATION
374 /// Serialization function
375 friend class boost::serialization::access;
376 template <class ARCHIVE>
377 void serialize(ARCHIVE& ar, const unsigned int /*version*/) {
378 // NoiseModelFactor2 instead of NoiseModelFactorN for backward compatibility
379 ar& boost::serialization::make_nvp(
380 "NoiseModelFactor2", boost::serialization::base_object<Base>(*this));
381 }
382#endif
383};

387struct traits<NonlinearEquality2<VALUE>> : Testable<NonlinearEquality2<VALUE>> {
388};
