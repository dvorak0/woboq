56 /**
57 * A non-templated config holding any types of Manifold-group elements. A
58 * values structure is a map from keys to values. It is used to specify the
59 * value of a bunch of variables in a factor graph. A Values is a values
60 * structure which can hold variables that are elements on manifolds, not just
61 * vectors. It then, as a whole, implements a aggregate type which is also a
62 * manifold element, and hence supports operations dim, retract, and
63 * localCoordinates.
64 */
65 class GTSAM_EXPORT Values {
66
67 private:
68 // Internally we store a boost ptr_map, with a ValueCloneAllocator (defined
69 // below) to clone and deallocate the Value objects, and our compile-flag-
70 // dependent FastDefaultAllocator to allocate map nodes. In this way, the
71 // user defines the allocation details (i.e. optimize for memory pool/arenas
72 // concurrency).
73 typedef internal::FastDefaultAllocator<typename std::pair<const Key, void*>>::type KeyValuePtrPairAllocator;
74 using KeyValueMap =
75 std::map<Key, std::unique_ptr<Value>, std::less<Key>,
76 std::allocator<std::pair<const Key, std::unique_ptr<Value>>>>;

78 // The member to store the values, see just above
79 KeyValueMap values_;

187 struct deref_iterator {
188 using const_iterator_type = typename KeyValueMap::const_iterator;
189 const_iterator_type it_;
190 deref_iterator(const_iterator_type it) : it_(it) {}
191 ConstKeyValuePair operator*() const { return {it_->first, *(it_->second)}; }
192 std::unique_ptr<ConstKeyValuePair> operator->() const {
193 return std::make_unique<ConstKeyValuePair>(args: it_->first, args&: *(it_->second));
194 }
195 bool operator==(const deref_iterator& other) const {
196 return it_ == other.it_;
197 }
198 bool operator!=(const deref_iterator& other) const { return it_ != other.it_; }
199 deref_iterator& operator++() {
200 ++it_;
201 return *this;
202 }
203 };
204
205 deref_iterator begin() const { return deref_iterator(values_.begin()); }
206 deref_iterator end() const { return deref_iterator(values_.end()); }

245 template <typename ValueType>

247
248 /** Partial specialization that allows passing a unary Eigen expression for val.
249 *
250 * A unary expression is an expression such as 2*a or -a, where a is a valid Vector or Matrix type.
251 * The typical usage is for types Point2 (i.e. Eigen::Vector2d) or Point3 (i.e. Eigen::Vector3d).
252 * For example, together with the partial specialization for binary operators, a user may call insert(j, 2*a + M*b - c),
253 * where M is an appropriately sized matrix (such as a rotation matrix).
254 * Thus, it isn't necessary to explicitly evaluate the Eigen expression, as in insert(j, (2*a + M*b - c).eval()),
255 * nor is it necessary to first assign the expression to a separate variable.
256 */
257 template <typename UnaryOp, typename ValueType>
258 void insert(Key j, const Eigen::CwiseUnaryOp<UnaryOp, const ValueType>& val);
259
260 /** Partial specialization that allows passing a binary Eigen expression for val.
261 *
262 * A binary expression is an expression such as a + b, where a and b are valid Vector or Matrix
263 * types of compatible size.
264 * The typical usage is for types Point2 (i.e. Eigen::Vector2d) or Point3 (i.e. Eigen::Vector3d).
265 * For example, together with the partial specialization for binary operators, a user may call insert(j, 2*a + M*b - c),
266 * where M is an appropriately sized matrix (such as a rotation matrix).
267 * Thus, it isn't necessary to explicitly evaluate the Eigen expression, as in insert(j, (2*a + M*b - c).eval()),
268 * nor is it necessary to first assign the expression to a separate variable.
269 */
270 template <typename BinaryOp, typename ValueType1, typename ValueType2>
271 void insert(Key j, const Eigen::CwiseBinaryOp<BinaryOp, const ValueType1, const ValueType2>& val);

362 /**
363 * Extract a subset of values of the given type \c ValueType.
364 *
365 * In this templated version, only key-value pairs whose value matches the
366 * template argument \c ValueType and whose key causes the function argument
367 * \c filterFcn to return true are visited when iterating over the filtered
368 * view. This replaces the fancier but very boost-dependent \c filter methods
369 * that were previously available up to GTSAM 4.2.
370 *
371 * @tparam ValueType The type that the value in a key-value pair must match
372 * to be included in the filtered view. Currently, base classes are not
373 * resolved so the type must match exactly, except if ValueType = Value, in
374 * which case no type filtering is done.
375 * @param filterFcn The function that determines which key-value pairs are
376 * included in the filtered view, for which this function returns \c true
377 * on their keys (default: always return true so that filter() only
378 * filters by type, matching \c ValueType).
379 * @return An Eigen aligned map on Key with the filtered values.
380 */
381 template <class ValueType>
382 std::map<Key, ValueType> // , std::less<Key>, Eigen::aligned_allocator<ValueType>
383 extract(const std::function<bool(Key)>& filterFcn = &_truePredicate<Key>) const;
384
385 private:
386 // Filters based on ValueType (if not Value) and also based on the user-
387 // supplied \c filter function.
388 template<class ValueType>
389 static bool filterHelper(const std::function<bool(Key)> filter, const ConstKeyValuePair& key_value) {
390 // static_assert if ValueType is type: Value
391 static_assert(!std::is_same<Value, ValueType>::value, "ValueType must not be type: Value to use this filter");
392 // Filter and check the type
393 return filter(key_value.key) && (dynamic_cast<const GenericValue<ValueType>*>(&key_value.value));
394 }
