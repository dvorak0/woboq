19#pragma once

21#include <gtsam/nonlinear/ExpressionFactor.h>
22#include <gtsam/nonlinear/NonlinearFactorGraph.h>
23
24namespace gtsam {
25
26/**
27 * Factor graph that supports adding ExpressionFactors directly
28 */
29class ExpressionFactorGraph: public NonlinearFactorGraph {
30
31public:
32
33 /// @name Adding Factors
34 /// @{
35
36 /**
37 * Directly add ExpressionFactor that implements |h(x)-z|^2_R
38 * @param h expression that implements measurement function
39 * @param z measurement
40 * @param R model
41 */
42 template<typename T>
43 void addExpressionFactor(const Expression<T>& h, const T& z,
44 const SharedNoiseModel& R) {
45 using F = ExpressionFactor<T>;
46 push_back(std::allocate_shared<F>(Eigen::aligned_allocator<F>(), R, z, h));
47 }
48
49 /// @}
50};
51
52}
