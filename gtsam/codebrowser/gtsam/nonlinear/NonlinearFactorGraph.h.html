45 /**
46 * A NonlinearFactorGraph is a graph of non-Gaussian, i.e. non-linear factors,
47 * which derive from NonlinearFactor. The values structures are typically (in
48 * SAM) more general than just vectors, e.g., Rot3 or Pose3, which are objects
49 * in non-linear manifolds. Linearizing the non-linear factor graph creates a
50 * linear factor graph on the tangent vector space at the linearization point.
51 * Because the tangent space is a true vector space, the config type will be
52 * an VectorValues in that linearized factor graph.
53 * @addtogroup nonlinear
54 */
55 class GTSAM_EXPORT NonlinearFactorGraph: public FactorGraph<NonlinearFactor> {
56
57 public:
58
59 typedef FactorGraph<NonlinearFactor> Base;
60 typedef NonlinearFactorGraph This;
61 typedef std::shared_ptr<This> shared_ptr;
62
63 /// @name Standard Constructors
64 /// @{
65
66 /** Default constructor */
67 NonlinearFactorGraph() {}
68
69 /** Construct from iterator over factors */
70 template<typename ITERATOR>
71 NonlinearFactorGraph(ITERATOR firstFactor, ITERATOR lastFactor) : Base(firstFactor, lastFactor) {}
72
73 /** Construct from container of factors (shared_ptr or plain objects) */
74 template<class CONTAINER>
75 explicit NonlinearFactorGraph(const CONTAINER& factors) : Base(factors) {}
76
77 /** Implicit copy/downcast constructor to override explicit template container constructor */
78 template<class DERIVEDFACTOR>
79 NonlinearFactorGraph(const FactorGraph<DERIVEDFACTOR>& graph) : Base(graph) {}
80
81 /// @}
82 /// @name Testable
83 /// @{
84
85 /** print */
86 void print(
87 const std::string& str = "NonlinearFactorGraph: ",
88 const KeyFormatter& keyFormatter = DefaultKeyFormatter) const override;
89
90 /** print errors along with factors*/
91 void printErrors(const Values& values, const std::string& str = "NonlinearFactorGraph: ",
92 const KeyFormatter& keyFormatter = DefaultKeyFormatter,
93 const std::function<bool(const Factor* /*factor*/, double /*whitenedError*/, size_t /*index*/)>&
94 printCondition = [](const Factor *,double, size_t) {return true;}) const;
95
96 /** Test equality */
97 bool equals(const NonlinearFactorGraph& other, double tol = 1e-9) const;
98
99 /// @}
100 /// @name Standard Interface
101 /// @{
102
103 /** unnormalized error, \f$ \sum_i 0.5 (h_i(X_i)-z)^2 / \sigma^2 \f$ in the most common case */
104 double error(const Values& values) const;
105
106 /** Unnormalized probability. O(n) */
107 double probPrime(const Values& values) const;
108
109 /**
110 * Create a symbolic factor graph
111 */
112 std::shared_ptr<SymbolicFactorGraph> symbolic() const;
113
114 /**
115 * Compute a fill-reducing ordering using COLAMD.
116 */
117 Ordering orderingCOLAMD() const;
118
119 /**
120 * Compute a fill-reducing ordering with constraints using CCOLAMD
121 *
122 * @param constraints is a map of Key->group, where 0 is unconstrained, and higher
123 * group numbers are further back in the ordering. Only keys with nonzero group
124 * indices need to appear in the constraints, unconstrained is assumed for all
125 * other variables
126 */
127 Ordering orderingCOLAMDConstrained(const FastMap<Key, int>& constraints) const;
128
129 /// Linearize a nonlinear factor graph
130 std::shared_ptr<GaussianFactorGraph> linearize(const Values& linearizationPoint) const;
131
132 /// typdef for dampen functions used below
133 typedef std::function<void(const std::shared_ptr<HessianFactor>& hessianFactor)> Dampen;
134
135 /**
136 * Instead of producing a GaussianFactorGraph, pre-allocate and linearize directly
137 * into a HessianFactor. Avoids the many mallocs and pointer indirection in constructing
138 * a new graph, and hence useful in case a dense solve is appropriate for your problem.
139 * An optional lambda function can be used to apply damping on the filled Hessian.
140 * No parallelism is exploited, because all the factors write in the same memory.
141 */
142 std::shared_ptr<HessianFactor> linearizeToHessianFactor(
143 const Values& values, const Dampen& dampen = nullptr) const;
144
145 /**
146 * Instead of producing a GaussianFactorGraph, pre-allocate and linearize directly
147 * into a HessianFactor. Avoids the many mallocs and pointer indirection in constructing
148 * a new graph, and hence useful in case a dense solve is appropriate for your problem.
149 * An ordering is given that still decides how the Hessian is laid out.
150 * An optional lambda function can be used to apply damping on the filled Hessian.
151 * No parallelism is exploited, because all the factors write in the same memory.
152 */
153 std::shared_ptr<HessianFactor> linearizeToHessianFactor(
154 const Values& values, const Ordering& ordering, const Dampen& dampen = nullptr) const;
155
156 /// Linearize and solve in one pass.
157 /// Calls linearizeToHessianFactor, densely solves the normal equations, and updates the values.
158 Values updateCholesky(const Values& values,
159 const Dampen& dampen = nullptr) const;
160
161 /// Linearize and solve in one pass.
162 /// Calls linearizeToHessianFactor, densely solves the normal equations, and updates the values.
163 Values updateCholesky(const Values& values, const Ordering& ordering,
164 const Dampen& dampen = nullptr) const;
165
166 /// Clone() performs a deep-copy of the graph, including all of the factors
167 NonlinearFactorGraph clone() const;
168
169 /**
170 * Rekey() performs a deep-copy of all of the factors, and changes
171 * keys according to a mapping.
172 *
173 * Keys not specified in the mapping will remain unchanged.
174 *
175 * @param rekey_mapping is a map of old->new keys
176 * @result a cloned graph with updated keys
177 */
178 NonlinearFactorGraph rekey(const std::map<Key,Key>& rekey_mapping) const;
179
180 /**
181 * Directly add ExpressionFactor that implements |h(x)-z|^2_R
182 * @param h expression that implements measurement function
183 * @param z measurement
184 * @param R model
185 */
186 template<typename T>
187 void addExpressionFactor(const SharedNoiseModel& R, const T& z,
188 const Expression<T>& h) {
189 this->emplace_shared<ExpressionFactor<T>>(R, z, h);
190 }
191
192 /**
193 * Convenience method which adds a PriorFactor to the factor graph.
194 * @param key Variable key
195 * @param prior The variable's prior value
196 * @param model Noise model for prior factor
197 */
198 template<typename T>
199 void addPrior(Key key, const T& prior,
200 const SharedNoiseModel& model = nullptr) {
201 emplace_shared<PriorFactor<T>>(key, prior, model);
202 }
203
204 /**
205 * Convenience method which adds a PriorFactor to the factor graph.
206 * @param key Variable key
207 * @param prior The variable's prior value
208 * @param covariance Covariance matrix.
209 *
210 * Note that the smart noise model associated with the prior factor
211 * automatically picks the right noise model (e.g. a diagonal noise model
212 * if the provided covariance matrix is diagonal).
213 */
214 template<typename T>
215 void addPrior(Key key, const T& prior, const Matrix& covariance) {
216 emplace_shared<PriorFactor<T>>(key, prior, covariance);
217 }
