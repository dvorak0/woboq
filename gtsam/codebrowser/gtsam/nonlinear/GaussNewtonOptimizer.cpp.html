28/* ************************************************************************* */
29GaussNewtonOptimizer::GaussNewtonOptimizer(const NonlinearFactorGraph& graph,
30 const Values& initialValues,
31 const GaussNewtonParams& params)
32 : NonlinearOptimizer(
33 graph, std::unique_ptr<State>(new State(initialValues, graph.error(values: initialValues)))),
34 params_(ensureHasOrdering(params, graph)) {}

36GaussNewtonOptimizer::GaussNewtonOptimizer(const NonlinearFactorGraph& graph,
37 const Values& initialValues, const Ordering& ordering)
38 : NonlinearOptimizer(
39 graph, std::unique_ptr<State>(new State(initialValues, graph.error(values: initialValues)))) {
40 params_.ordering = ordering;
41}

43/* ************************************************************************* */
44GaussianFactorGraph::shared_ptr GaussNewtonOptimizer::iterate() {
45 gttic(GaussNewtonOptimizer_Iterate);
46
47 // Linearize graph
48 gttic(GaussNewtonOptimizer_Linearize);
49 GaussianFactorGraph::shared_ptr linear = graph_.linearize(linearizationPoint: state_->values);
50 gttoc(GaussNewtonOptimizer_Linearize);
51
52 // Solve Factor Graph
53 gttic(GaussNewtonOptimizer_Solve);
54 const VectorValues delta = solve(gfg: *linear, params: params_);
55 gttoc(GaussNewtonOptimizer_Solve);
56
57 // Maybe show output
58 if (params_.verbosity >= NonlinearOptimizerParams::DELTA)
59 delta.print(str: "delta");
60
61 // Create new state with new values and new error
62 Values newValues = state_->values.retract(delta);
63 state_.reset(p: new State(std::move(newValues), graph_.error(values: newValues), state_->iterations + 1));
64
65 return linear;
66}

68/* ************************************************************************* */
69GaussNewtonParams GaussNewtonOptimizer::ensureHasOrdering(
70 GaussNewtonParams params, const NonlinearFactorGraph& graph) const {
71 if (!params.ordering)
72 params.ordering = Ordering::Create(orderingType: params.orderingType, graph);
73 return params;
74}
