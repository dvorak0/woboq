17namespace gtsam {
18
19 // Forward declarations
20 class JacobianFactor;
21 class HessianFactor;
22
23/**
24 * Dummy version of a generic linear factor to be injected into a nonlinear factor graph
25 *
26 * This factor does have the ability to perform relinearization under small-angle and
27 * linearity assumptions if a linearization point is added.
28 */
29class GTSAM_EXPORT LinearContainerFactor : public NonlinearFactor {
30protected:
31
32 GaussianFactor::shared_ptr factor_;
33 std::optional<Values> linearizationPoint_;
34
35 /** direct copy constructor */
36 LinearContainerFactor(const GaussianFactor::shared_ptr& factor, const std::optional<Values>& linearizationPoint);
37
38 // Some handy typedefs
39 typedef NonlinearFactor Base;
40 typedef LinearContainerFactor This;
41
42public:
43
44 typedef std::shared_ptr<This> shared_ptr;
45
46 /** Default constructor - necessary for serialization */
47 LinearContainerFactor() {}
48
49 /** Primary constructor: store a linear factor with optional linearization point */
50 LinearContainerFactor(const JacobianFactor& factor, const Values& linearizationPoint = Values());
51
52 /** Primary constructor: store a linear factor with optional linearization point */
53 LinearContainerFactor(const HessianFactor& factor, const Values& linearizationPoint = Values());
54
55 /** Constructor from shared_ptr */
56 LinearContainerFactor(const GaussianFactor::shared_ptr& factor, const Values& linearizationPoint = Values());
57
58 // Access
59
60 const GaussianFactor::shared_ptr& factor() const { return factor_; }
61
62 // Testable
63
64 /** print */
65 void print(const std::string& s = "", const KeyFormatter& keyFormatter = gtsam::DefaultKeyFormatter) const override;
66
67 /** Check if two factors are equal */
68 bool equals(const NonlinearFactor& f, double tol = 1e-9) const override;
69
70 // NonlinearFactor
71
72 /**
73 * Calculate the nonlinear error for the factor, where the error is computed
74 * by passing the delta between linearization point and c, where
75 * delta = linearizationPoint_.localCoordinates(c), into the error function
76 * of the stored linear factor.
77 *
78 * @return nonlinear error if linearizationPoint present, zero otherwise
79 */
80 double error(const Values& c) const override;
81
82 /** get the dimension of the factor: rows of linear factor */
83 size_t dim() const override;
84
85 /** Extract the linearization point used in recalculating error */
86 const std::optional<Values>& linearizationPoint() const { return linearizationPoint_; }
87
88 /**
89 * Linearize to a GaussianFactor, with method depending on the presence of a linearizationPoint
90 * - With no linearization point, returns a cloned version of the stored linear factor.
91 * - With a linearization point provided, returns a relinearized version of
92 * the linearized factor.
93 *
94 * The relinearization approach used computes a linear delta between the original linearization
95 * point and the new values c, where delta = linearizationPoint_.localCoordinates(c), and
96 * substitutes this change into the system. This substitution is only really valid for
97 * linear variable manifolds, and for any variables based on a non-commutative
98 * manifold (such as Pose2, Pose3), the relinearized version will be effective
99 * for only small angles.
100 *
101 * TODO: better approximation of relinearization
102 * TODO: switchable modes for approximation technique
103 */
104 GaussianFactor::shared_ptr linearize(const Values& c) const override;
105
106 /**
107 * Creates an anti-factor directly
108 */
109 GaussianFactor::shared_ptr negateToGaussian() const;
110
111 /**
112 * Creates the equivalent anti-factor as another LinearContainerFactor.
113 */
114 NonlinearFactor::shared_ptr negateToNonlinear() const;
115
116 /**
117 * Creates a shared_ptr clone of the factor - needs to be specialized to allow
118 * for subclasses
119 *
120 * Clones the underlying linear factor
121 */
122 NonlinearFactor::shared_ptr clone() const override {
123 return NonlinearFactor::shared_ptr(new LinearContainerFactor(factor_,linearizationPoint_));
124 }
125
126 /**
127 * Creates a shared_ptr clone of the
128 * factor with different keys using
129 * a map from old->new keys
130 */
131 NonlinearFactor::shared_ptr rekey(
132 const std::map<Key, Key>& rekey_mapping) const override;
133
134 /**
135 * Clones a factor and fully replaces its keys
136 * @param new_keys is the full replacement set of keys
137 */
138 NonlinearFactor::shared_ptr rekey(const KeyVector& new_keys) const override;
139
140 /// Casting syntactic sugar
141 inline bool hasLinearizationPoint() const { return linearizationPoint_.has_value(); }
142
143 /**
144 * Simple checks whether this is a Jacobian or Hessian factor
145 */
146 bool isJacobian() const;
147 bool isHessian() const;
148
149 /** Casts to JacobianFactor */
150 std::shared_ptr<JacobianFactor> toJacobian() const;
151
152 /** Casts to HessianFactor */
153 std::shared_ptr<HessianFactor> toHessian() const;
154
155 /**
156 * Utility function for converting linear graphs to nonlinear graphs
157 * consisting of LinearContainerFactors.
158 */
159 static NonlinearFactorGraph ConvertLinearGraph(const GaussianFactorGraph& linear_graph,
160 const Values& linearizationPoint = Values());
161
162 protected:
163 void initializeLinearizationPoint(const Values& linearizationPoint);
164
165 private:
166#if GTSAM_ENABLE_BOOST_SERIALIZATION
167 /** Serialization function */
168 friend class boost::serialization::access;
169 template<class ARCHIVE>
170 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
171 ar & boost::serialization::make_nvp(n: "NonlinearFactor",
172 v&: boost::serialization::base_object<Base>(d&: *this));
173 ar & BOOST_SERIALIZATION_NVP(factor_);
174 ar & BOOST_SERIALIZATION_NVP(linearizationPoint_);
175 }
176#endif
177
178}; // \class LinearContainerFactor
179
180template<> struct traits<LinearContainerFactor> : public Testable<LinearContainerFactor> {};

182} // \namespace gtsam
