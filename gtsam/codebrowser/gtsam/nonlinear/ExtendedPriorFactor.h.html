25/**
26 * A class for a soft prior on any Value type, but with a non-zero mean
27 * in the tangent space.
28 * The error is `e(x) = -Local(x, origin) - mean`.
29 * The loss is `0.5 * ||e(x)||^2_Sigma` for a Gaussian noise model,
30 * and `rho(e(x))` for a robust noise model `rho`.
31 * The likelihood is `exp(-loss)`.
32 *
33 * Note: The "extended concentrated Gaussian" on Lie groups is defined as
34 * `~ exp{-0.5*|log(origin_^{-1} x) - mean|^2_Sigma}`. Our implementation,
35 * with a Gaussian noise model, is identical (up to a constant) to this
36 * for all Lie Groups.
37 *
38 * @ingroup nonlinear
39 */
40template <class VALUE>
41class ExtendedPriorFactor : public NoiseModelFactorN<VALUE> {
42 public:
43 typedef VALUE T;
44
45 /// Provide access to the Matrix& version of evaluateError:
46 using NoiseModelFactor1<VALUE>::evaluateError;
47
48 protected:
49 typedef NoiseModelFactorN<VALUE> Base;
50
51 VALUE origin_; /** The point in manifold at which tangent space is rooted. */
52 std::optional<Vector>
53 mean_; /** Mean in the tangent space, default nullopt. */
54
55 /// concept check by type
56 GTSAM_CONCEPT_TESTABLE_TYPE(T)
57
58 public:
59 /// Typedef to this class
60 typedef ExtendedPriorFactor<VALUE> This;
61
62 /// @name Standard Constructors
63 /// @{
64
65 /// default constructor - only use for serialization
66 ExtendedPriorFactor() {}
67
68 /// Constructor with noise model and optional mean in tangent space
69 ExtendedPriorFactor(Key key, const T& origin, const SharedNoiseModel& model)
70 : Base(model, key), origin_(origin) {}
71
72 /// Constructor with noise model and optional mean in tangent space
73 ExtendedPriorFactor(Key key, const T& origin, const Vector& mean,
74 const SharedNoiseModel& model)
75 : Base(model, key), origin_(origin), mean_(mean) {
76 if (mean.size() != static_cast<Eigen::Index>(model->dim()))
77 throw std::invalid_argument(
78 "ConcentratedGaussian: mean dimension does not match noise model");
79 }
80
81 /// Constructor with covariance matrix (zero mean in tangent space)
82 ExtendedPriorFactor(Key key, const T& origin, const Matrix& covariance)
83 : Base(noiseModel::Gaussian::Covariance(covariance), key),
84 origin_(origin) {}
85
86 /// Constructor with mean (in tangent space) and covariance matrix
87 ExtendedPriorFactor(Key key, const T& origin, const Vector& mean,
88 const Matrix& covariance)
89 : Base(noiseModel::Gaussian::Covariance(covariance), key),
90 origin_(origin),
91 mean_(mean) {
92 if (mean.size() != covariance.rows() ||
93 covariance.rows() != covariance.cols())
94 throw std::invalid_argument(
95 "ConcentratedGaussian: mean and covariance dimensions do not match");
96 }
97
98 /// @}
99 /// @name Standard Destructor
100 /// @{
101
102 ~ExtendedPriorFactor() override {}
103
104 /// @}
105 /// @name Testable
106 /// @{
107
108 /// print
109 void print(const std::string& s, const KeyFormatter& keyFormatter =
110 DefaultKeyFormatter) const override {
111 std::cout << s << "ExtendedPriorFactor on " << keyFormatter(this->key())
112 << "\n";
113 traits<T>::Print(origin_, " origin: ");
114 if (mean_) {
115 gtsam::print(v: *mean_, s: " tangent space mean: ");
116 }
117 if (this->noiseModel_)
118 this->noiseModel_->print(" noise model: ");
119 else
120 std::cout << "no noise model" << std::endl;
121 }
122
123 /// equals
124 bool equals(const NonlinearFactor& expected,
125 double tol = 1e-9) const override {
126 const This* e = dynamic_cast<const This*>(&expected);
127 if (!e) return false;
128 bool mean_equals =
129 (!mean_ && !e->mean_) ||
130 (mean_ && e->mean_ && equal_with_abs_tol(*mean_, *e->mean_, tol));
131 return Base::equals(*e, tol) &&
132 traits<T>::Equals(origin_, e->origin_, tol) && mean_equals;
133 }
134
135 /// @}
136 /// @name Factor
137 /// @{
138
139 /// @return a deep copy of this factor
140 gtsam::NonlinearFactor::shared_ptr clone() const override {
141 return std::static_pointer_cast<gtsam::NonlinearFactor>(
142 r: gtsam::NonlinearFactor::shared_ptr(new This(*this)));
143 }
144
145 /// vector of errors
146 Vector evaluateError(const T& x, OptionalMatrixType H) const override {
147 if (H) {
148 (*H) = Matrix::Identity(traits<T>::GetDimension(x),
149 traits<T>::GetDimension(x));
150 }
151 // manifold equivalent of z-x -> Local(x,z)
152 Vector error = -traits<T>::Local(x, origin_);
153 if (mean_) {
154 return error - *mean_;
155 }
156 return error;
157 }
158
159 /// Bring the error(const Values&) method into scope.
160 using Base::error;
161
162 /// Compute the negative log-likelihood of a given value
163 double error(const T& x) const {
164 Vector e = evaluateError(x);
165 double squared_mahalanobis_distance =
166 this->noiseModel_->squaredMahalanobisDistance(e);
167 return this->noiseModel_->loss(squared_mahalanobis_distance);
168 }
169
170 /// Compute the likelihood of a given value
171 double likelihood(const T& x) const { return exp(-error(x)); }
172
173 /// @}
174 /// @name Access
175 /// @{
176
177 const VALUE& origin() const { return origin_; }
178 const std::optional<Vector>& mean() const { return mean_; }
179
180 /// Get the Gaussian noise model, or return nullopt/throw if not Gaussian.
181 noiseModel::Gaussian::shared_ptr gaussianModel(
182 const std::string& method = "<unknown>",
183 bool throwOnFailure = false) const {
184 using noiseModel::Gaussian;
185 const auto& model = this->noiseModel();
186 if (!model) {
187 if (throwOnFailure) {
188 throw std::runtime_error(method + " requires a noise model");
189 }
190 return nullptr;
191 }
192 auto g = std::dynamic_pointer_cast<Gaussian>(model);
193 if (!g) {
194 if (throwOnFailure) {
195 throw std::runtime_error(method +
196 " is only implemented for Gaussian noise "
197 "models. The noise model used is of type " +
198 std::string(typeid(*model).name()));
199 }
200 return nullptr;
201 }
202 return g;
203 }
204
205 /// Get Gaussian covariance, or return nullopt/throw if not Gaussian.
206 std::optional<Matrix> covariance(const std::string& method = "<unknown>",
207 bool throwOnFailure = false) const {
208 auto gaussian = gaussianModel(method, throwOnFailure);
209 if (!gaussian) return std::nullopt;
210 return gaussian->covariance();
211 }
212
213 /// Simple, non-templated Gaussian fusion in a common tangent space
214 using Gaussian = std::pair<Vector, Matrix>; // mean, covariance
215
216 /// Get a Gaussian, or return nullopt if not Gaussian and throw is false
217 std::optional<Gaussian> gaussian(const std::string& method = "<unknown>",
218 bool throwOnFailure = false) const {
219 auto cov = covariance(method, throwOnFailure);
220 if (!cov) return std::nullopt;
221 return Gaussian{this->mean_.value_or(Vector::Zero(this->dim())), *cov};
222 }
223
224 /// @}
225
226 private:
227#if GTSAM_ENABLE_BOOST_SERIALIZATION
228 /// Serialization function
229 friend class boost::serialization::access;
230 template <class ARCHIVE>
231 void serialize(ARCHIVE& ar, const unsigned int /*version*/) {
232 // NoiseModelFactor1 instead of NoiseModelFactorN for backward compatibility
233 ar& boost::serialization::make_nvp(
234 "NoiseModelFactor1", boost::serialization::base_object<Base>(*this));
235 ar& BOOST_SERIALIZATION_NVP(origin_);
236 ar& BOOST_SERIALIZATION_NVP(mean_);
237 }
238#endif
239
240 /// Alignment, see
241 /// https://eigen.tuxfamily.org/dox/group__TopicStructHavingEigenMembers.html
242 inline constexpr static auto NeedsToAlign = (sizeof(T) % 16) == 0;
243
244 public:
245 GTSAM_MAKE_ALIGNED_OPERATOR_NEW_IF(NeedsToAlign)
246};
