29/**
30 * Linearize a nonlinear factor using numerical differentiation
31 * The benefit of this method is that it does not need to know what types are
32 * involved to evaluate the factor. If all the machinery of gtsam is working
33 * correctly, we should get the correct numerical derivatives out the other side.
34 * NOTE(frank): factors that have non vector-space measurements use between or LocalCoordinates
35 * to evaluate the error, and their derivatives will only be correct for near-zero errors.
36 * This is fixable but expensive, and does not matter in practice as most factors will sit near
37 * zero errors anyway. However, it means that below will only be exact for the correct measurement.
38 */
39inline JacobianFactor linearizeNumerically(const NoiseModelFactor& factor,
40 const Values& values,
41 double delta = 1e-5) {
42 // We will fill a vector of key/Jacobians pairs (a map would sort)
43 std::vector<std::pair<Key, Matrix> > jacobians;
44
45 // Get size
46 const Vector e = factor.whitenedError(c: values);
47 const size_t rows = e.size();
48
49 // Loop over all variables
50 const double one_over_2delta = 1.0 / (2.0 * delta);
51 for (Key key : factor) {
52 // Compute central differences using the values struct.
53 VectorValues dX = values.zeroVectors();
54 const size_t cols = dX.dim(j: key);
55 Matrix J = Matrix::Zero(rows, cols);
56 for (size_t col = 0; col < cols; ++col) {
57 Vector dx = Vector::Zero(size: cols);
58 dx(col) = delta;
59 dX[key] = dx;
60 Values eval_values = values.retract(delta: dX);
61 const Vector left = factor.whitenedError(c: eval_values);
62 dx(col) = -delta;
63 dX[key] = dx;
64 eval_values = values.retract(delta: dX);
65 const Vector right = factor.whitenedError(c: eval_values);
66 J.col(i: col) = (left - right) * one_over_2delta;
67 }
68 jacobians.emplace_back(args&: key, args&: J);
69 }
70
71 // Next step...return JacobianFactor
72 return JacobianFactor(jacobians, -e);
73}

75namespace internal {
76// CPPUnitLite-style test for linearization of a factor
77inline bool testFactorJacobians(const std::string& name_,
78 const NoiseModelFactor& factor,
79 const gtsam::Values& values, double delta,
80 double tolerance) {
81 // Create expected value by numerical differentiation
82 JacobianFactor expected = linearizeNumerically(factor, values, delta);
83
84 // Create actual value by linearize
85 auto actual =
86 std::dynamic_pointer_cast<JacobianFactor>(r: factor.linearize(x: values));
87 if (!actual) return false;
88
89 // Check cast result and then equality
90 bool equal = assert_equal(expected, actual: *actual, tol: tolerance);
91
92 // if not equal, test individual jacobians:
93 if (!equal) {
94 for (size_t i = 0; i < actual->size(); i++) {
95 bool i_good =
96 assert_equal(A: (Matrix)(expected.getA(variable: expected.begin() + i)),
97 B: (Matrix)(actual->getA(variable: actual->begin() + i)), tol: tolerance);
98 if (!i_good) {
99 std::cout << "Mismatch in Jacobian " << i + 1
100 << " (base 1), as shown above" << std::endl;
101 }
102 }
103 }
104
105 return equal;
106}
107} // namespace internal

109/// \brief Check the Jacobians produced by a factor against finite differences.
110/// \param factor The factor to test.
111/// \param values Values filled in for testing the Jacobians.
112/// \param numerical_derivative_step The step to use when computing the numerical derivative Jacobians
113/// \param tolerance The numerical tolerance to use when comparing Jacobians.
114#define EXPECT_CORRECT_FACTOR_JACOBIANS(factor, values, numerical_derivative_step, tolerance) \
115 { EXPECT(gtsam::internal::testFactorJacobians(name_, factor, values, numerical_derivative_step, tolerance)); }
