45/**
46 * Expression class that supports automatic differentiation
47 */
48template<typename T>
49class Expression {
50
51public:
52
53 /// Define type so we can apply it as a meta-function
54 typedef Expression<T> type;
55
56protected:
57
58 // Paul's trick shared pointer, polymorphic root of entire expression tree
59 std::shared_ptr<internal::ExpressionNode<T> > root_;
60
61 /// Construct with a custom root
62 Expression(const std::shared_ptr<internal::ExpressionNode<T> >& root) : root_(root) {}
63
64public:
65
66 // Expressions wrap trees of functions that can evaluate their own derivatives.
67 // The meta-functions below are useful to specify the type of those functions.
68 // Example, a function taking a camera and a 3D point and yielding a 2D point:
69 // Expression<Point2>::BinaryFunction<PinholeCamera<Cal3_S2>,Point3>::type
70 template<class A1>
71 struct UnaryFunction {
72 typedef std::function<
73 T(const A1&, typename MakeOptionalJacobian<T, A1>::type)> type;
74 };
75
76 template<class A1, class A2>
77 struct BinaryFunction {
78 typedef std::function<
79 T(const A1&, const A2&, typename MakeOptionalJacobian<T, A1>::type,
80 typename MakeOptionalJacobian<T, A2>::type)> type;
81 };
82
83 template<class A1, class A2, class A3>
84 struct TernaryFunction {
85 typedef std::function<
86 T(const A1&, const A2&, const A3&,
87 typename MakeOptionalJacobian<T, A1>::type,
88 typename MakeOptionalJacobian<T, A2>::type,
89 typename MakeOptionalJacobian<T, A3>::type)> type;
90 };
91
92 /// Construct a constant expression
93 Expression(const T& value);
94
95 /// Construct a leaf expression, with Key
96 Expression(const Key& key);
97
98 /// Construct a leaf expression, with Symbol
99 Expression(const Symbol& symbol);
100
101 /// Construct a leaf expression, creating Symbol
102 Expression(unsigned char c, std::uint64_t j);
103
104 /// Construct a unary function expression
105 template<typename A>
106 Expression(typename UnaryFunction<A>::type function,
107 const Expression<A>& expression);
108
109 /// Construct a binary function expression
110 template<typename A1, typename A2>
111 Expression(typename BinaryFunction<A1, A2>::type function,
112 const Expression<A1>& expression1, const Expression<A2>& expression2);
113
114 /// Construct a ternary function expression
115 template<typename A1, typename A2, typename A3>
116 Expression(typename TernaryFunction<A1, A2, A3>::type function,
117 const Expression<A1>& expression1, const Expression<A2>& expression2,
118 const Expression<A3>& expression3);
119
120 /// Construct a nullary method expression
121 template<typename A>
122 Expression(const Expression<A>& expression,
123 T (A::*method)(typename MakeOptionalJacobian<T, A>::type) const);
124
125 /// Construct a unary method expression
126 template<typename A1, typename A2>
127 Expression(const Expression<A1>& expression1,
128 T (A1::*method)(const A2&, typename MakeOptionalJacobian<T, A1>::type,
129 typename MakeOptionalJacobian<T, A2>::type) const,
130 const Expression<A2>& expression2);
131
132 /// Construct a binary method expression
133 template<typename A1, typename A2, typename A3>
134 Expression(const Expression<A1>& expression1,
135 T (A1::*method)(const A2&, const A3&,
136 typename MakeOptionalJacobian<T, A1>::type,
137 typename MakeOptionalJacobian<T, A2>::type,
138 typename MakeOptionalJacobian<T, A3>::type) const,
139 const Expression<A2>& expression2, const Expression<A3>& expression3);
140
141 /// Destructor
142 virtual ~Expression() {
143 }
144
145 /// Return keys that play in this expression
146 std::set<Key> keys() const;
147
148 /// Return dimensions for each argument, as a map
149 void dims(std::map<Key, int>& map) const;
150
151 /// Print
152 void print(const std::string& s) const;
153
154 /**
155 * @brief Return value and optional derivatives, reverse AD version
156 * Notes: this is not terribly efficient, and H should have correct size.
157 * The order of the Jacobians is same as keys in either keys() or dims()
158 */
159 T value(const Values& values, std::vector<Matrix>* H = nullptr) const;
160
161 /**
162 * An overload of the value function to accept reference to vector of matrices instead of
163 * a pointer to vector of matrices.
164 */
165 T value(const Values& values, std::vector<Matrix>& H) const {
166 return value(values, &H);
167 }
168
169 /**
170 * @return a "deep" copy of this Expression
171 * "deep" is in quotes because the ExpressionNode hierarchy is *not* cloned.
172 * The intent is for derived classes to be copied using only a Base pointer.
173 */
174 virtual std::shared_ptr<Expression> clone() const {
175 return std::make_shared<Expression>(*this);
176 }
177
178 /// Return root
179 const std::shared_ptr<internal::ExpressionNode<T> >& root() const;
180
181 /// Return size needed for memory buffer in traceExecution
182 size_t traceSize() const;
183
184 /// Add another expression to this expression
185 Expression<T>& operator+=(const Expression<T>& e);
186
187protected:
188
189 /// Default constructor, for serialization
190 Expression() {}
191
192 /// Keys and dimensions in same order
193 typedef std::pair<KeyVector, FastVector<int> > KeysAndDims;
194 KeysAndDims keysAndDims() const;
195
196 /// private version that takes keys and dimensions, returns derivatives
197 T valueAndDerivatives(const Values& values, const KeyVector& keys,
198 const FastVector<int>& dims, std::vector<Matrix>& H) const;
199
200 /// trace execution, very unsafe
201 T traceExecution(const Values& values, internal::ExecutionTrace<T>& trace,
202 char* traceStorage) const;
203
204 /// brief Return value and derivatives, reverse AD version
205 T valueAndJacobianMap(const Values& values,
206 internal::JacobianMap& jacobians) const;
207
208 // be very selective on who can access these private methods:
209 friend class ExpressionFactor<T> ;
210 friend class internal::ExpressionNode<T>;
211 friend class ExpressionEqualityConstraint<T>;
212 friend class ScalarExpressionInequalityConstraint;
213
214 // and add tests
215 friend class ::ExpressionFactorShallowTest;
216};

218/**
219 * A ScalarMultiplyExpression is a specialization of Expression that multiplies with a scalar
220 * It optimizes the Jacobian calculation for this specific case
221 */
222template <typename T>
223class ScalarMultiplyExpression : public Expression<T> {
224 // Check that T is a vector space
225 GTSAM_CONCEPT_ASSERT(IsVectorSpace<T>);
226
227 public:
228 explicit ScalarMultiplyExpression(double s, const Expression<T>& e);
229};

231/**
232 * A BinarySumExpression is a specialization of Expression that adds two expressions together
233 * It optimizes the Jacobian calculation for this specific case
234 */
235template <typename T>
236class BinarySumExpression : public Expression<T> {
237 // Check that T is a vector space
238 GTSAM_CONCEPT_ASSERT(IsVectorSpace<T>);
239
240 public:
241 explicit BinarySumExpression(const Expression<T>& e1, const Expression<T>& e2);
242};

245/**
246 * Create an expression out of a linear function f:T->A with (constant) Jacobian dTdA
247 * TODO(frank): create a more efficient version like ScalarMultiplyExpression. This version still
248 * does a malloc every linearize.
249 */
250template <typename T, typename A>
251Expression<T> linearExpression(
252 const std::function<T(A)>& f, const Expression<A>& expression,
253 const Eigen::Matrix<double, traits<T>::dimension, traits<A>::dimension>& dTdA) {
254 // Use lambda to endow f with a linear Jacobian
255 typename Expression<T>::template UnaryFunction<A>::type g =
256 [=](const A& value, typename MakeOptionalJacobian<T, A>::type H) {
257 if (H)
258 *H << dTdA;
259 return f(value);
260 };
261 return Expression<T>(g, expression);
262}
263
264/**
265 * Construct an expression that executes the scalar multiplication with an input expression
266 * The type T must be a vector space
267 * Example:
268 * Expression<Point2> a(0), b = 12 * a;
269 */
270template <typename T>
271ScalarMultiplyExpression<T> operator*(double s, const Expression<T>& e) {
272 return ScalarMultiplyExpression<T>(s, e);
273}
274
275/**
276 * Construct an expression that sums two input expressions of the same type T
277 * The type T must be a vector space
278 * Example:
279 * Expression<Point2> a(0), b(1), c = a + b;
280 */
281template <typename T>
282BinarySumExpression<T> operator+(const Expression<T>& e1, const Expression<T>& e2) {
283 return BinarySumExpression<T>(e1, e2);
284}
285
286/// Construct an expression that subtracts one expression from another
287template <typename T>
288BinarySumExpression<T> operator-(const Expression<T>& e1, const Expression<T>& e2) {
289 // TODO(frank, abe): Implement an actual negate operator instead of multiplying by -1
290 return e1 + (-1.0) * e2;
291}

298template<typename T>
