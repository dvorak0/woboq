19#pragma once
20
21#include <gtsam/inference/DotWriter.h>
22
23namespace gtsam {
24
25class Values;
26class Value;
27
28/**
29 * Formatting options and functions for saving a NonlinearFactorGraph instance
30 * in GraphViz format.
31 */
32struct GTSAM_EXPORT GraphvizFormatting : public DotWriter {
33 /// World axes to be assigned to paper axes
34 enum Axis { X, Y, Z, NEGX, NEGY, NEGZ };
35
36 Axis paperHorizontalAxis; ///< The world axis assigned to the horizontal
37 ///< paper axis
38 Axis paperVerticalAxis; ///< The world axis assigned to the vertical paper
39 ///< axis
40 double scale; ///< Scale all positions to reduce / increase density
41 bool mergeSimilarFactors; ///< Merge multiple factors that have the same
42 ///< connectivity
43
44 /// Default constructor sets up robot coordinates. Paper horizontal is robot
45 /// Y, paper vertical is robot X. Default figure size of 5x5 in.
46 GraphvizFormatting()
47 : paperHorizontalAxis(Y),
48 paperVerticalAxis(X),
49 scale(1),
50 mergeSimilarFactors(false) {}
51
52 // Find bounds
53 Vector2 findBounds(const Values& values, const KeySet& keys) const;
54
55 /// Extract a Vector2 from either Vector2, Pose2, Pose3, or Point3
56 std::optional<Vector2> extractPosition(const Value& value) const;
57
58 /// Return affinely transformed variable position if it exists.
59 std::optional<Vector2> variablePos(const Values& values, const Vector2& min,
60 Key key) const;
61
62 /// Return affinely transformed factor position if it exists.
63 std::optional<Vector2> factorPos(const Vector2& min, size_t i) const;
64};
65
66} // namespace gtsam
