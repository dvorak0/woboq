21#pragma once
22
23#include <gtsam/nonlinear/FixedLagSmoother.h>
24#include <gtsam/nonlinear/LevenbergMarquardtOptimizer.h>
25#include <queue>
26
27namespace gtsam {
28
29class GTSAM_EXPORT BatchFixedLagSmoother : public FixedLagSmoother {
30
31public:
32
33 /// Typedef for a shared pointer to an Incremental Fixed-Lag Smoother
34 typedef std::shared_ptr<BatchFixedLagSmoother> shared_ptr;
35
36 /**
37 * Construct with parameters
38 *
39 * @param smootherLag The length of the smoother lag. Any variable older than this amount will be marginalized out.
40 * @param parameters The L-M optimization parameters
41 * @param enforceConsistency A flag indicating if the optimizer should enforce probabilistic consistency by maintaining the
42 * linearization point of all variables involved in linearized/marginal factors at the edge of the
43 * smoothing window.
44 */
45 BatchFixedLagSmoother(double smootherLag = 0.0, const LevenbergMarquardtParams& parameters = LevenbergMarquardtParams(), bool enforceConsistency = true) :
46 FixedLagSmoother(smootherLag), parameters_(parameters), enforceConsistency_(enforceConsistency) {
47 }
48
49 /** destructor */
50 ~BatchFixedLagSmoother() override {}
51
52 /** Print the factor for debugging and testing (implementing Testable) */
53 void print(const std::string& s = "BatchFixedLagSmoother:\n", const KeyFormatter& keyFormatter = DefaultKeyFormatter) const override;
54
55 /** Check if two IncrementalFixedLagSmoother Objects are equal */
56 bool equals(const FixedLagSmoother& rhs, double tol = 1e-9) const override;
57
58 /** Add new factors, updating the solution and relinearizing as needed. */
59 Result update(const NonlinearFactorGraph& newFactors = NonlinearFactorGraph(),
60 const Values& newTheta = Values(),
61 const KeyTimestampMap& timestamps = KeyTimestampMap(),
62 const FactorIndices& factorsToRemove = FactorIndices()) override;
63
64 /** Compute an estimate from the incomplete linear delta computed during the last update.
65 * This delta is incomplete because it was not updated below wildfire_threshold. If only
66 * a single variable is needed, it is faster to call calculateEstimate(const KEY&).
67 */
68 Values calculateEstimate() const override {
69 return theta_.retract(delta: delta_);
70 }
71
72 /** Compute an estimate for a single variable using its incomplete linear delta computed
73 * during the last update. This is faster than calling the no-argument version of
74 * calculateEstimate, which operates on all variables.
75 * @param key
76 * @return
77 */
78 template<class VALUE>
79 VALUE calculateEstimate(Key key) const {
80 const Vector delta = delta_.at(j: key);
81 return traits<VALUE>::Retract(theta_.at<VALUE>(key), delta);
82 }
83
84 /** read the current set of optimizer parameters */
85 const LevenbergMarquardtParams& params() const {
86 return parameters_;
87 }
88
89 /** update the current set of optimizer parameters */
90 LevenbergMarquardtParams& params() {
91 return parameters_;
92 }
93
94 /** Access the current set of factors */
95 const NonlinearFactorGraph& getFactors() const {
96 return factors_;
97 }
98
99 /** Access the current linearization point */
100 const Values& getLinearizationPoint() const {
101 return theta_;
102 }
103
104 /** Access the current ordering */
105 const Ordering& getOrdering() const {
106 return ordering_;
107 }
108
109 /** Access the current set of deltas to the linearization point */
110 const VectorValues& getDelta() const {
111 return delta_;
112 }
113
114 /// Calculate marginal covariance on given variable
115 Matrix marginalCovariance(Key key) const;
116
117 /// Marginalize specific keys from a linear graph.
118 /// Does not check whether keys actually exist in graph.
119 /// In that case will fail somewhere deep within elimination
120 static GaussianFactorGraph CalculateMarginalFactors(
121 const GaussianFactorGraph& graph, const KeyVector& keys,
122 const GaussianFactorGraph::Eliminate& eliminateFunction = EliminatePreferCholesky);
123
124 /// Marginalize specific keys from a nonlinear graph, wrap in LinearContainers
125 static NonlinearFactorGraph CalculateMarginalFactors(
126 const NonlinearFactorGraph& graph, const Values& theta, const KeyVector& keys,
127 const GaussianFactorGraph::Eliminate& eliminateFunction = EliminatePreferCholesky);
128
129protected:
130
131 /** A typedef defining an Key-Factor mapping **/
132 typedef std::map<Key, std::set<Key> > FactorIndex;
133
134 /** The L-M optimization parameters **/
135 LevenbergMarquardtParams parameters_;
136
137 /** A flag indicating if the optimizer should enforce probabilistic consistency by maintaining the
138 * linearization point of all variables involved in linearized/marginal factors at the edge of the
139 * smoothing window. This idea is from ??? TODO: Look up paper reference **/
140 bool enforceConsistency_;
141
142 /** The nonlinear factors **/
143 NonlinearFactorGraph factors_;
144
145 /** The current linearization point **/
146 Values theta_;
147
148 /** The set of values involved in current linear factors. **/
149 Values linearValues_;
150
151 /** The current ordering */
152 Ordering ordering_;
153
154 /** The current set of linear deltas */
155 VectorValues delta_;
156
157 /** The set of available factor graph slots. These occur because we are constantly deleting factors, leaving holes. **/
158 std::queue<size_t> availableSlots_;
159
160 /** A cross-reference structure to allow efficient factor lookups by key **/
161 FactorIndex factorIndex_;
162
163 /** Augment the list of factors with a set of new factors */
164 void insertFactors(const NonlinearFactorGraph& newFactors);
165
166 /** Remove factors from the list of factors by slot index */
167 void removeFactors(const std::set<size_t>& deleteFactors);
168
169 /** Erase any keys associated with timestamps before the provided time */
170 void eraseKeys(const KeyVector& keys);
171
172 /** Use colamd to update into an efficient ordering */
173 void reorder(const KeyVector& marginalizeKeys = KeyVector());
174
175 /** Optimize the current graph using a modified version of L-M */
176 Result optimize();
177
178 /** Marginalize out selected variables */
179 void marginalize(const KeyVector& marginalizableKeys);
180
181private:
182 /** Private methods for printing debug information */
183 static void PrintKeySet(const std::set<Key>& keys, const std::string& label);
184 static void PrintKeySet(const gtsam::KeySet& keys, const std::string& label);
185 static void PrintSymbolicFactor(const NonlinearFactor::shared_ptr& factor);
186 static void PrintSymbolicFactor(const GaussianFactor::shared_ptr& factor);
187 static void PrintSymbolicGraph(const NonlinearFactorGraph& graph, const std::string& label);
188 static void PrintSymbolicGraph(const GaussianFactorGraph& graph, const std::string& label);
189}; // BatchFixedLagSmoother
190
191} /// namespace gtsam
