29namespace gtsam {
30
31/**
32 * Specialized Clique structure for ISAM2, incorporating caching and gradient
33 * contribution
34 * TODO: more documentation
35 */
36class GTSAM_EXPORT ISAM2Clique
37 : public BayesTreeCliqueBase<ISAM2Clique, GaussianFactorGraph> {
38 public:
39 typedef ISAM2Clique This;
40 typedef BayesTreeCliqueBase<This, GaussianFactorGraph> Base;
41 typedef std::shared_ptr<This> shared_ptr;
42 typedef std::weak_ptr<This> weak_ptr;
43 typedef GaussianConditional ConditionalType;
44 typedef ConditionalType::shared_ptr sharedConditional;
45
46 Base::FactorType::shared_ptr cachedFactor_;
47 Vector gradientContribution_;
48#ifdef USE_BROKEN_FAST_BACKSUBSTITUTE
49 mutable FastMap<Key, VectorValues::iterator> solnPointers_;
50#endif
51
52 /// Default constructor
53 ISAM2Clique() : Base() {}
54
55 /// Copy constructor, does *not* copy solution pointers as these are invalid
56 /// in different trees.
57 ISAM2Clique(const ISAM2Clique& other)
58 : Base(other),
59 cachedFactor_(other.cachedFactor_),
60 gradientContribution_(other.gradientContribution_) {}
61
62 /// Assignment operator, does *not* copy solution pointers as these are
63 /// invalid in different trees.
64 ISAM2Clique& operator=(const ISAM2Clique& other) {
65 Base::operator=(c: other);
66 cachedFactor_ = other.cachedFactor_;
67 gradientContribution_ = other.gradientContribution_;
68 return *this;
69 }
70
71 /// Overridden to also store the remaining factor and gradient contribution
72 void setEliminationResult(
73 const FactorGraphType::EliminationResult& eliminationResult);
74
75 /** Access the cached factor */
76 Base::FactorType::shared_ptr& cachedFactor() { return cachedFactor_; }
77
78 /// Access the gradient contribution
79 const Vector& gradientContribution() const { return gradientContribution_; }
80
81 /// Recursively add gradient at zero to g
82 void addGradientAtZero(VectorValues* g) const;
83
84 bool equals(const This& other, double tol = 1e-9) const;
85
86 /** print this node */
87 void print(const std::string& s = "",
88 const KeyFormatter& formatter = DefaultKeyFormatter) const override;
89
90 void optimizeWildfire(const KeySet& replaced, double threshold,
91 KeySet* changed, VectorValues* delta,
92 size_t* count) const;
93
94 bool optimizeWildfireNode(const KeySet& replaced, double threshold,
95 KeySet* changed, VectorValues* delta,
96 size_t* count) const;
97
98 /**
99 * Starting from the root, add up entries of frontal and conditional matrices
100 * of each conditional
101 */
102 void nnz_internal(size_t* result) const;
103 size_t calculate_nnz() const;
104
105 /**
106 * Recursively search this clique and its children for marked keys appearing
107 * in the separator, and add the *frontal* keys of any cliques whose
108 * separator contains any marked keys to the set \c keys. The purpose of
109 * this is to discover the cliques that need to be redone due to information
110 * propagating to them from cliques that directly contain factors being
111 * relinearized.
112 *
113 * The original comment says this finds all variables directly connected to
114 * the marked ones by measurements. Is this true, because it seems like this
115 * would also pull in variables indirectly connected through other frontal or
116 * separator variables?
117 *
118 * Alternatively could we trace up towards the root for each variable here?
119 */
120 void findAll(const KeySet& markedMask, KeySet* keys) const;
121
122 private:
123 /**
124 * Check if clique was replaced, or if any parents were changed above the
125 * threshold or themselves replaced.
126 */
127 bool isDirty(const KeySet& replaced, const KeySet& changed) const;
128
129 /**
130 * Back-substitute - special version stores solution pointers in cliques for
131 * fast access.
132 */
133 void fastBackSubstitute(VectorValues* delta) const;
134
135 /*
136 * Check whether the values changed above a threshold, or always true if the
137 * clique was replaced.
138 */
139 bool valuesChanged(const KeySet& replaced, const Vector& originalValues,
140 const VectorValues& delta, double threshold) const;
141
142 /// Set changed flag for each frontal variable
143 void markFrontalsAsChanged(KeySet* changed) const;
144
145 /// Restore delta to original values, guided by frontal keys.
146 void restoreFromOriginals(const Vector& originalValues,
147 VectorValues* delta) const;
148
149#if GTSAM_ENABLE_BOOST_SERIALIZATION
150 /** Serialization function */
151 friend class boost::serialization::access;
152 template <class ARCHIVE>
153 void serialize(ARCHIVE& ar, const unsigned int /*version*/) {
154 ar& BOOST_SERIALIZATION_BASE_OBJECT_NVP(Base);
155 ar& BOOST_SERIALIZATION_NVP(cachedFactor_);
156 ar& BOOST_SERIALIZATION_NVP(gradientContribution_);
157 }
158#endif
159}; // \struct ISAM2Clique
160
161/**
162 * Optimize the BayesTree, starting from the root.
163 * @param threshold The maximum change against the PREVIOUS delta for
164 * non-replaced variables that can be ignored, ie. the old delta entry is kept
165 * and recursive backsubstitution might eventually stop if none of the changed
166 * variables are contained in the subtree.
167 * @param replaced Needs to contain all variables that are contained in the top
168 * of the Bayes tree that has been redone.
169 * @return The number of variables that were solved for.
170 * @param delta The current solution, an offset from the linearization point.
171 */
172size_t optimizeWildfire(const ISAM2Clique::shared_ptr& root, double threshold,
173 const KeySet& replaced, VectorValues* delta);
174
175size_t optimizeWildfireNonRecursive(const ISAM2Clique::shared_ptr& root,
176 double threshold, const KeySet& replaced,
177 VectorValues* delta);
178
179} // namespace gtsam
