32/**
33 * @ingroup isam2
34 * This struct is returned from ISAM2::update() and contains information about
35 * the update that is useful for determining whether the solution is
36 * converging, and about how much work was required for the update. See member
37 * variables for details and information about each entry.
38 */
39struct ISAM2Result {
40 /** The nonlinear error of all of the factors, \a including new factors and
41 * variables added during the current call to ISAM2::update(). This error is
42 * calculated using the following variable values:
43 * \li Pre-existing variables will be evaluated by combining their
44 * linearization point before this call to update, with their partial linear
45 * delta, as computed by ISAM2::calculateEstimate().
46 * \li New variables will be evaluated at their initialization points passed
47 * into the current call to update.
48 * \par Note: This will only be computed if
49 * ISAM2Params::evaluateNonlinearError is set to \c true, because there is
50 * some cost to this computation.
51 */
52 std::optional<double> errorBefore;
53
54 /** The nonlinear error of all of the factors computed after the current
55 * update, meaning that variables above the relinearization threshold
56 * (ISAM2Params::relinearizeThreshold) have been relinearized and new
57 * variables have undergone one linear update. Variable values are
58 * again computed by combining their linearization points with their
59 * partial linear deltas, by ISAM2::calculateEstimate().
60 * \par Note: This will only be computed if
61 * ISAM2Params::evaluateNonlinearError is set to \c true, because there is
62 * some cost to this computation.
63 */
64 std::optional<double> errorAfter;
65
66 /** The number of variables that were relinearized because their linear
67 * deltas exceeded the reslinearization threshold
68 * (ISAM2Params::relinearizeThreshold), combined with any additional
69 * variables that had to be relinearized because they were involved in
70 * the same factor as a variable above the relinearization threshold.
71 * On steps where no relinearization is considered
72 * (see ISAM2Params::relinearizeSkip), this count will be zero.
73 */
74 size_t variablesRelinearized;
75
76 /** The number of variables that were reeliminated as parts of the Bayes'
77 * Tree were recalculated, due to new factors. When loop closures occur,
78 * this count will be large as the new loop-closing factors will tend to
79 * involve variables far away from the root, and everything up to the root
80 * will be reeliminated.
81 */
82 size_t variablesReeliminated;
83
84 /** The number of factors that were included in reelimination of the Bayes'
85 * tree. */
86 size_t factorsRecalculated;
87
88 /** The number of cliques in the Bayes' Tree */
89 size_t cliques;
90
91 /** The indices of the newly-added factors, in 1-to-1 correspondence with the
92 * factors passed as \c newFactors to ISAM2::update(). These indices may be
93 * used later to refer to the factors in order to remove them.
94 */
95 FactorIndices newFactorsIndices;
96
97 /** Unused keys, and indices for unused keys,
98 * i.e., keys that are empty now and do not appear in the new factors.
99 */
100 KeySet unusedKeys;
101
102 /** keys for variables that were observed, i.e., not unused. */
103 KeyVector observedKeys;
104
105 /** Keys of variables that had factors removed. */
106 KeySet keysWithRemovedFactors;
107
108 /** All keys that were marked during the update process. */
109 KeySet markedKeys;
110
111 /**
112 * A struct holding detailed results, which must be enabled with
113 * ISAM2Params::enableDetailedResults.
114 */
115 struct DetailedResults {
116 /** The status of a single variable, this struct is stored in
117 * DetailedResults::variableStatus */
118 struct VariableStatus {
119 /** Whether the variable was just reeliminated, due to being relinearized,
120 * observed, new, or on the path up to the root clique from another
121 * reeliminated variable. */
122 bool isReeliminated;
123 bool isAboveRelinThreshold; ///< Whether the variable was just
124 ///< relinearized due to being above the
125 ///< relinearization threshold
126 bool isRelinearizeInvolved; ///< Whether the variable was below the
127 ///< relinearization threshold but was
128 ///< relinearized by being involved in a
129 ///< factor with a variable above the
130 ///< relinearization threshold
131 bool isRelinearized; /// Whether the variable was relinearized, either by
132 /// being above the relinearization threshold or by
133 /// involvement.
134 bool isObserved; ///< Whether the variable was just involved in new
135 ///< factors
136 bool isNew; ///< Whether the variable itself was just added
137 bool inRootClique; ///< Whether the variable is in the root clique
138 VariableStatus()
139 : isReeliminated(false),
140 isAboveRelinThreshold(false),
141 isRelinearizeInvolved(false),
142 isRelinearized(false),
143 isObserved(false),
144 isNew(false),
145 inRootClique(false) {}
146 };
147
148 using StatusMap = FastMap<Key, VariableStatus>;
149
150 /// The status of each variable during this update, see VariableStatus.
151 StatusMap variableStatus;
152 };
153
154 /** Detailed results, if enabled by ISAM2Params::enableDetailedResults. See
155 * Detail for information about the results data stored here. */
156 std::optional<DetailedResults> detail;
157
158 explicit ISAM2Result(bool enableDetailedResults = false) {
159 if (enableDetailedResults) detail = DetailedResults();
160 }
161
162 /// Return pointer to detail, 0 if no detail requested
163 DetailedResults* details() {
164 if (detail.has_value()) {
165 return &(*detail);
166 } else {
167 return nullptr;
168 }
169 }
170
171 /// Print results
172 void print(const std::string str = "") const {
173 using std::cout;
174 cout << str << " Reelimintated: " << variablesReeliminated
175 << " Relinearized: " << variablesRelinearized
176 << " Cliques: " << cliques << std::endl;
177 }
178
179 /** Getters and Setters */
180 size_t getVariablesRelinearized() const { return variablesRelinearized; }
181 size_t getVariablesReeliminated() const { return variablesReeliminated; }
182 FactorIndices getNewFactorsIndices() const { return newFactorsIndices; }
183 size_t getCliques() const { return cliques; }
184 double getErrorBefore() const { return errorBefore ? *errorBefore : std::nan(tagb: ""); }
185 double getErrorAfter() const { return errorAfter ? *errorAfter : std::nan(tagb: ""); }
186};
