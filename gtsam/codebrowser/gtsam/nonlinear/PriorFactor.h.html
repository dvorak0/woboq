12/**
13 * @file PriorFactor.h
14 * @author Frank Dellaert
15 **/
16#pragma once
17
18#include <gtsam/nonlinear/ExtendedPriorFactor.h>
19
20namespace gtsam {
21
22/**
23 * A class for a soft prior on any Value type.
24 * @ingroup nonlinear
25 **/
26template <class VALUE>
27class PriorFactor : public ExtendedPriorFactor<VALUE> {
28 public:
29 typedef VALUE T;
30 typedef ExtendedPriorFactor<VALUE> Base;
31
32 private:
33 /// concept check by type
34 GTSAM_CONCEPT_TESTABLE_TYPE(T)
35
36 public:
37 /// @name Standard Constructors
38 /// @{
39
40 /// shorthand for a smart pointer to a factor
41 typedef typename std::shared_ptr<PriorFactor<VALUE>> shared_ptr;
42
43 /// Typedef to this class
44 typedef PriorFactor<VALUE> This;
45
46 /// default constructor - only use for serialization
47 PriorFactor() {}
48
49 /// Constructor
50 PriorFactor(Key key, const VALUE& prior,
51 const SharedNoiseModel& model = nullptr)
52 : Base(key, prior, model) {}
53
54 /// Convenience constructor that takes a full covariance argument
55 PriorFactor(Key key, const VALUE& prior, const Matrix& covariance)
56 : Base(key, prior, noiseModel::Gaussian::Covariance(covariance)) {}
57
58 /// @}
59 /// @name Standard Destructor
60 /// @{
61
62 ~PriorFactor() override {}
63
64 /// @}
65 /// @name Testable
66 /// @{
67
68 /// @return a deep copy of this factor
69 gtsam::NonlinearFactor::shared_ptr clone() const override {
70 return std::static_pointer_cast<gtsam::NonlinearFactor>(
71 r: gtsam::NonlinearFactor::shared_ptr(new This(*this)));
72 }
73
74 /// print
75 void print(const std::string& s, const KeyFormatter& keyFormatter =
76 DefaultKeyFormatter) const override {
77 std::cout << s << "PriorFactor on " << keyFormatter(this->key()) << "\n";
78 traits<T>::Print(this->origin(), " prior mean: ");
79 if (this->noiseModel_)
80 this->noiseModel_->print(" noise model: ");
81 else
82 std::cout << "no noise model" << std::endl;
83 }
84
85 /// equals
86 bool equals(const NonlinearFactor& expected,
87 double tol = 1e-9) const override {
88 const auto* e = dynamic_cast<const This*>(&expected);
89 return e && Base::equals(*e, tol);
90 }
91
92 /// @}
93 /// @name Access
94 /// @{
95
96 const VALUE& prior() const { return this->origin(); }
97
98 /// @}
99
100 private:
101#if GTSAM_ENABLE_BOOST_SERIALIZATION
102 /// Serialization function
103 friend class boost::serialization::access;
104 template <class ARCHIVE>
105 void serialize(ARCHIVE& ar, const unsigned int /*version*/) {
106 // For backwards compatibility, we manually serialize the base class members
107 // as if we were a PriorFactor from before the refactor.
108 ar& boost::serialization::make_nvp(
109 "NoiseModelFactor1",
110 boost::serialization::base_object<NoiseModelFactorN<VALUE>>(*this));
111 ar& boost::serialization::make_nvp("prior_", this->origin_);
112 }
113#endif
114
115 // Alignment, see
116 // https://eigen.tuxfamily.org/dox/group__TopicStructHavingEigenMembers.html
117 inline constexpr static auto NeedsToAlign = (sizeof(T) % 16) == 0;
118
119 public:
120 GTSAM_MAKE_ALIGNED_OPERATOR_NEW_IF(NeedsToAlign)
121};
122
123/// traits
124template <class VALUE>
125struct traits<PriorFactor<VALUE>> : public Testable<PriorFactor<VALUE>> {};
126
127} // namespace gtsam
