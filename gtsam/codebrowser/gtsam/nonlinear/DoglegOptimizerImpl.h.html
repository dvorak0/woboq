27/** This class contains the implementation of the Dogleg algorithm. It is used
28 * by DoglegOptimizer and can be used to easily put together custom versions of
29 * Dogleg. Each function is well-documented and unit-tested. The notation
30 * here matches that in "trustregion.pdf" in doc, see this file for further
31 * explanation of the computations performed by this class.
32 */
33struct GTSAM_EXPORT DoglegOptimizerImpl {
34
35 struct GTSAM_EXPORT IterationResult {
36 double delta;
37 VectorValues dx_d;
38 double f_error;
39 };
40
41 /** Specifies how the trust region is adapted at each Dogleg iteration. If
42 * this is SEARCH_EACH_ITERATION, then the trust region radius will be
43 * increased potentially multiple times during one iteration until increasing
44 * it further no longer decreases the error. If this is
45 * ONE_STEP_PER_ITERATION, then the step in one iteration will not exceed the
46 * current trust region radius, but the radius will be increased for the next
47 * iteration if the error decrease is good. The former will generally result
48 * in slower iterations, but sometimes larger steps in early iterations. The
49 * latter generally results in faster iterations but it may take several
50 * iterations before the trust region radius is increased to the optimal
51 * value. Generally ONE_STEP_PER_ITERATION should be used, corresponding to
52 * most published descriptions of the algorithm.
53 */
54 enum TrustRegionAdaptationMode {
55 SEARCH_EACH_ITERATION,
56 SEARCH_REDUCE_ONLY,
57 ONE_STEP_PER_ITERATION
58 };
59
60 /**
61 * Compute the update point for one iteration of the Dogleg algorithm, given
62 * an initial trust region radius \f$ \delta \f$. The trust region radius is
63 * adapted based on the error of a NonlinearFactorGraph \f$ f(x) \f$, and
64 * depending on the update mode \c mode.
65 *
66 * The update is computed using a quadratic approximation \f$ M(\delta x) \f$
67 * of an original nonlinear error function (a NonlinearFactorGraph) \f$ f(x) \f$.
68 * The quadratic approximation is represented as a GaussianBayesNet \f$ \bayesNet \f$, which is
69 * obtained by eliminating a GaussianFactorGraph resulting from linearizing
70 * the nonlinear factor graph \f$ f(x) \f$. Thus, \f$ M(\delta x) \f$ is
71 * \f[
72 * M(\delta x) = (R \delta x - d)^T (R \delta x - d)
73 * \f]
74 * where \f$ R \f$ and \f$ d \f$ together are a Bayes' net or Bayes' tree.
75 * \f$ R \f$ is upper-triangular and \f$ d \f$ is a vector, in GTSAM represented
76 * as a BayesNet<GaussianConditional> (GaussianBayesNet) or
77 * BayesTree<GaussianConditional>, containing GaussianConditional s.
78 *
79 * @tparam M The type of the Bayes' net or tree, currently
80 * either BayesNet<GaussianConditional> (or GaussianBayesNet) or BayesTree<GaussianConditional>.
81 * @tparam F For normal usage this will be NonlinearFactorGraph<VALUES>.
82 * @tparam VALUES The Values or TupleValues to pass to F::error() to evaluate
83 * the error function.
84 * @param delta The initial trust region radius.
85 * @param mode See DoglegOptimizerImpl::TrustRegionAdaptationMode
86 * @param Rd The Bayes' net or tree as described above.
87 * @param f The original nonlinear factor graph with which to evaluate the
88 * accuracy of \f$ M(\delta x) \f$ to adjust \f$ \delta \f$.
89 * @param x0 The linearization point about which \f$ \bayesNet \f$ was created
90 * @param ordering The variable ordering used to create\f$ \bayesNet \f$
91 * @param f_error The result of <tt>f.error(x0)</tt>.
92 * @return A DoglegIterationResult containing the new \c delta, the linear
93 * update \c dx_d, and the resulting nonlinear error \c f_error.
94 */
95 template<class M, class F, class VALUES>
96 static IterationResult Iterate(
97 double delta, TrustRegionAdaptationMode mode, const VectorValues& dx_u, const VectorValues& dx_n,
98 const M& Rd, const F& f, const VALUES& x0, const double f_error, const bool verbose=false);
99
100 /**
101 * Compute the dogleg point given a trust region radius \f$ \delta \f$. The
102 * dogleg point is the intersection between the dogleg path and the trust
103 * region boundary, see doc/trustregion.pdf for more details.
104 *
105 * The update is computed using a quadratic approximation \f$ M(\delta x) \f$
106 * of an original nonlinear error function (a NonlinearFactorGraph) \f$ f(x) \f$.
107 * The quadratic approximation is represented as a GaussianBayesNet \f$ \bayesNet \f$, which is
108 * obtained by eliminating a GaussianFactorGraph resulting from linearizing
109 * the nonlinear factor graph \f$ f(x) \f$. Thus, \f$ M(\delta x) \f$ is
110 * \f[
111 * M(\delta x) = (R \delta x - d)^T (R \delta x - d)
112 * \f]
113 * where \f$ R \f$ and \f$ d \f$ together are a Bayes' net. \f$ R \f$ is
114 * upper-triangular and \f$ d \f$ is a vector, in GTSAM represented as a
115 * GaussianBayesNet, containing GaussianConditional s.
116 *
117 * @param delta The trust region radius
118 * @param dx_u The steepest descent point, i.e. the Cauchy point
119 * @param dx_n The Gauss-Newton point
120 * @return The dogleg point \f$ \delta x_d \f$
121 */
122 static VectorValues ComputeDoglegPoint(double delta, const VectorValues& dx_u, const VectorValues& dx_n, const bool verbose=false);
123
124 /** Compute the point on the line between the steepest descent point and the
125 * Newton's method point intersecting the trust region boundary.
126 * Mathematically, computes \f$ \tau \f$ such that \f$ 0<\tau<1 \f$ and
127 * \f$ \| (1-\tau)\delta x_u + \tau\delta x_n \| = \delta \f$, where \f$ \delta \f$
128 * is the trust region radius.
129 * @param delta Trust region radius
130 * @param x_u Steepest descent minimizer
131 * @param x_n Newton's method minimizer
132 */
133 static VectorValues ComputeBlend(double delta, const VectorValues& x_u, const VectorValues& x_n, const bool verbose=false);
134};

137/* ************************************************************************* */
138template<class M, class F, class VALUES>
139typename DoglegOptimizerImpl::IterationResult DoglegOptimizerImpl::Iterate(
140 double delta, TrustRegionAdaptationMode mode, const VectorValues& dx_u, const VectorValues& dx_n,
141 const M& Rd, const F& f, const VALUES& x0, const double f_error, const bool verbose)
142{
143 gttic(M_error);
144 const double M_error = Rd.error(VectorValues::Zero(other: dx_u));
145 gttoc(M_error);
146
147 // Result to return
148 IterationResult result;
149
150 bool stay = true;
151 enum { NONE, INCREASED_DELTA, DECREASED_DELTA } lastAction = NONE; // Used to prevent alternating between increasing and decreasing in one iteration
152 while(stay) {
153 gttic(Dog_leg_point);
154 // Compute dog leg point
155 result.dx_d = ComputeDoglegPoint(delta, dx_u, dx_n, verbose);
156 gttoc(Dog_leg_point);
157
158 if(verbose) std::cout << "delta = " << delta << ", dx_d_norm = " << result.dx_d.norm() << std::endl;
159
160 gttic(retract);
161 // Compute expmapped solution
162 const VALUES x_d(x0.retract(result.dx_d));
163 gttoc(retract);
164
165 gttic(decrease_in_f);
166 // Compute decrease in f
167 result.f_error = f.error(x_d);
168 gttoc(decrease_in_f);
169
170 gttic(new_M_error);
171 // Compute decrease in M
172 const double new_M_error = Rd.error(result.dx_d);
173 gttoc(new_M_error);
174
175 if(verbose) std::cout << std::setprecision(15) << "f error: " << f_error << " -> " << result.f_error << std::endl;
176 if(verbose) std::cout << std::setprecision(15) << "M error: " << M_error << " -> " << new_M_error << std::endl;
177
178 gttic(adjust_delta);
179 // Compute gain ratio. Here we take advantage of the invariant that the
180 // Bayes' net error at zero is equal to the nonlinear error
181 const double rho = std::abs(x: f_error - result.f_error) < 1e-15 || std::abs(x: M_error - new_M_error) < 1e-15 ?
182 0.5 :
183 (f_error - result.f_error) / (M_error - new_M_error);
184
185 if(verbose) std::cout << std::setprecision(15) << "rho = " << rho << std::endl;
186
187 if(rho >= 0.75) {
188 // M agrees very well with f, so try to increase lambda
189 const double dx_d_norm = result.dx_d.norm();
190 const double newDelta = std::max(a: delta, b: 3.0 * dx_d_norm); // Compute new delta
191
192 if(mode == ONE_STEP_PER_ITERATION || mode == SEARCH_REDUCE_ONLY)
193 stay = false; // If not searching, just return with the new delta
194 else if(mode == SEARCH_EACH_ITERATION) {
195 if(std::abs(x: newDelta - delta) < 1e-15 || lastAction == DECREASED_DELTA)
196 stay = false; // Searching, but Newton's solution is within trust region so keep the same trust region
197 else {
198 stay = true; // Searching and increased delta, so try again to increase delta
199 lastAction = INCREASED_DELTA;
200 }
201 } else {
202 assert(false); }
203
204 delta = newDelta; // Update delta from new delta
205
206 } else if(0.75 > rho && rho >= 0.25) {
207 // M agrees so-so with f, keep the same delta
208 stay = false;
209
210 } else if(0.25 > rho && rho >= 0.0) {
211 // M does not agree well with f, decrease delta until it does
212 double newDelta;
213 bool hitMinimumDelta;
214 if(delta > 1e-5) {
215 newDelta = 0.5 * delta;
216 hitMinimumDelta = false;
217 } else {
218 newDelta = delta;
219 hitMinimumDelta = true;
220 }
221 if(mode == ONE_STEP_PER_ITERATION || /* mode == SEARCH_EACH_ITERATION && */ lastAction == INCREASED_DELTA || hitMinimumDelta)
222 stay = false; // If not searching, just return with the new smaller delta
223 else if(mode == SEARCH_EACH_ITERATION || mode == SEARCH_REDUCE_ONLY) {
224 stay = true;
225 lastAction = DECREASED_DELTA;
226 } else {
227 assert(false); }
228
229 delta = newDelta; // Update delta from new delta
230
231 } else {
232 // f actually increased, so keep decreasing delta until f does not decrease.
233 // NOTE: NaN and Inf solutions also will fall into this case, so that we
234 // decrease delta if the solution becomes undetermined.
235 assert(0.0 > rho);
236 if(delta > 1e-5) {
237 delta *= 0.5;
238 stay = true;
239 lastAction = DECREASED_DELTA;
240 } else {
241 if(verbose) std::cout << "Warning: Dog leg stopping because cannot decrease error with minimum delta" << std::endl;
242 result.dx_d.setZero(); // Set delta to zero - don't allow error to increase
243 result.f_error = f_error;
244 stay = false;
245 }
246 }
247 gttoc(adjust_delta);
248 }
249
250 // dx_d and f_error have already been filled in during the loop
251 result.delta = delta;
252 return result;
253}
