29/**
30 * A class for computing Gaussian marginals of variables in a NonlinearFactorGraph
31 */
32class GTSAM_EXPORT Marginals {
33
34public:
35
36 /** The linear factorization mode - either CHOLESKY (faster and suitable for most problems) or QR (slower but more numerically stable for poorly-conditioned problems). */
37 enum Factorization {
38 CHOLESKY,
39 QR
40 };
41
42protected:
43
44 GaussianFactorGraph graph_;
45 Values values_;
46 Factorization factorization_;
47 GaussianBayesTree bayesTree_;
48
49public:
50
51 /// Default constructor only for wrappers
52 Marginals(){}
53
54 /** Construct a marginals class from a nonlinear factor graph.
55 * @param graph The factor graph defining the full joint density on all variables.
56 * @param solution The linearization point about which to compute Gaussian marginals (usually the MLE as obtained from a NonlinearOptimizer).
57 * @param factorization The linear decomposition mode - either Marginals::CHOLESKY (faster and suitable for most problems) or Marginals::QR (slower but more numerically stable for poorly-conditioned problems).
58 */
59 Marginals(const NonlinearFactorGraph& graph, const Values& solution, Factorization factorization = CHOLESKY);
60
61 /** Construct a marginals class from a nonlinear factor graph.
62 * @param graph The factor graph defining the full joint density on all variables.
63 * @param solution The linearization point about which to compute Gaussian marginals (usually the MLE as obtained from a NonlinearOptimizer).
64 * @param factorization The linear decomposition mode - either Marginals::CHOLESKY (faster and suitable for most problems) or Marginals::QR (slower but more numerically stable for poorly-conditioned problems).
65 * @param ordering The ordering for elimination.
66 */
67 Marginals(const NonlinearFactorGraph& graph, const Values& solution, const Ordering& ordering,
68 Factorization factorization = CHOLESKY);
69
70 /** Construct a marginals class from a linear factor graph.
71 * @param graph The factor graph defining the full joint density on all variables.
72 * @param solution The solution point to compute Gaussian marginals.
73 * @param factorization The linear decomposition mode - either Marginals::CHOLESKY (faster and suitable for most problems) or Marginals::QR (slower but more numerically stable for poorly-conditioned problems).
74 */
75 Marginals(const GaussianFactorGraph& graph, const Values& solution, Factorization factorization = CHOLESKY);
76
77 /** Construct a marginals class from a linear factor graph.
78 * @param graph The factor graph defining the full joint density on all variables.
79 * @param solution The solution point to compute Gaussian marginals.
80 * @param factorization The linear decomposition mode - either Marginals::CHOLESKY (faster and suitable for most problems) or Marginals::QR (slower but more numerically stable for poorly-conditioned problems).
81 * @param ordering The ordering for elimination.
82 */
83 Marginals(const GaussianFactorGraph& graph, const Values& solution, const Ordering& ordering,
84 Factorization factorization = CHOLESKY);
85
86 /** Construct a marginals class from a linear factor graph.
87 * @param graph The factor graph defining the full joint density on all variables.
88 * @param solution The solution point to compute Gaussian marginals.
89 * @param factorization The linear decomposition mode - either Marginals::CHOLESKY (faster and suitable for most problems) or Marginals::QR (slower but more numerically stable for poorly-conditioned problems).
90 * @param ordering An optional variable ordering for elimination.
91 */
92 Marginals(const GaussianFactorGraph& graph, const VectorValues& solution, Factorization factorization = CHOLESKY);
93
94 /** Construct a marginals class from a linear factor graph.
95 * @param graph The factor graph defining the full joint density on all variables.
96 * @param solution The solution point to compute Gaussian marginals.
97 * @param factorization The linear decomposition mode - either Marginals::CHOLESKY (faster and suitable for most problems) or Marginals::QR (slower but more numerically stable for poorly-conditioned problems).
98 * @param ordering An optional variable ordering for elimination.
99 */
100 Marginals(const GaussianFactorGraph& graph, const VectorValues& solution, const Ordering& ordering,
101 Factorization factorization = CHOLESKY);
102
103 /** print */
104 void print(const std::string& str = "Marginals: ", const KeyFormatter& keyFormatter = DefaultKeyFormatter) const;
105
106 /** Compute the marginal factor of a single variable */
107 GaussianFactor::shared_ptr marginalFactor(Key variable) const;
108
109 /** Compute the marginal information matrix of a single variable.
110 * Use LLt(const Matrix&) or RtR(const Matrix&) to obtain the square-root information matrix. */
111 Matrix marginalInformation(Key variable) const;
112
113 /** Compute the marginal covariance of a single variable */
114 Matrix marginalCovariance(Key variable) const;
115
116 /** Compute the joint marginal covariance of several variables */
117 JointMarginal jointMarginalCovariance(const KeyVector& variables) const;
118
119 /** Compute the joint marginal information of several variables */
120 JointMarginal jointMarginalInformation(const KeyVector& variables) const;
121
122 /** Optimize the bayes tree */
123 VectorValues optimize() const;
124
125protected:
126
127 /** Compute the Bayes Tree as a helper function to the constructor */
128 void computeBayesTree();
129
130 /** Compute the Bayes Tree as a helper function to the constructor */
131 void computeBayesTree(const Ordering& ordering);
132};

134/**
135 * A class to store and access a joint marginal, returned from Marginals::jointMarginalCovariance and Marginals::jointMarginalInformation
136 */
137class GTSAM_EXPORT JointMarginal {
138
139protected:
140 SymmetricBlockMatrix blockMatrix_;
141 KeyVector keys_;
142 FastMap<Key, size_t> indices_;
143
144public:
145 /// Default constructor only for wrappers
146 JointMarginal() {}
147
148 /** Access a block, corresponding to a pair of variables, of the joint
149 * marginal. Each block is accessed by its "vertical position",
150 * corresponding to the variable with nonlinear Key \c iVariable and
151 * "horizontal position", corresponding to the variable with nonlinear Key
152 * \c jVariable.
153 *
154 * For example, if we have the joint marginal on a 2D pose "x3" and a 2D
155 * landmark "l2", then jointMarginal(Symbol('x',3), Symbol('l',2)) will
156 * return the 3x2 block of the joint covariance matrix corresponding to x3
157 * and l2.
158 * @param iVariable The nonlinear Key specifying the "vertical position" of the requested block
159 * @param jVariable The nonlinear Key specifying the "horizontal position" of the requested block
160 */
161 Matrix operator()(Key iVariable, Key jVariable) const {
162 const auto indexI = indices_.at(k: iVariable);
163 const auto indexJ = indices_.at(k: jVariable);
164 return blockMatrix_.block(I: indexI, J: indexJ);
165 }
166
167 /** Synonym for operator() */
168 Matrix at(Key iVariable, Key jVariable) const {
169 return (*this)(iVariable, jVariable);
170 }
171
172 /** The full, dense covariance/information matrix of the joint marginal. */
173 Matrix fullMatrix() const {
174 return blockMatrix_.selfadjointView();
175 }
176
177 /** Print */
178 void print(const std::string& s = "", const KeyFormatter& formatter = DefaultKeyFormatter) const;
179
180protected:
181 JointMarginal(const Matrix& fullMatrix, const std::vector<size_t>& dims, const KeyVector& keys) :
182 blockMatrix_(dims, fullMatrix), keys_(keys), indices_(Ordering(keys).invert()) {}
183
184 friend class Marginals;
185
186};
