81/* ************************************************************************* */
82void Marginals::computeBayesTree() {
83 // The default ordering to use.
84 const Ordering::OrderingType defaultOrderingType = Ordering::COLAMD;
85 // Compute BayesTree
86 if (factorization_ == CHOLESKY)
87 bayesTree_ = *graph_.eliminateMultifrontal(orderingType: defaultOrderingType,
88 function: EliminatePreferCholesky);
89 else if (factorization_ == QR)
90 bayesTree_ =
91 *graph_.eliminateMultifrontal(orderingType: defaultOrderingType, function: EliminateQR);
92}
93
94/* ************************************************************************* */
95void Marginals::computeBayesTree(const Ordering& ordering) {
96 // Compute BayesTree
97 if(factorization_ == CHOLESKY)
98 bayesTree_ = *graph_.eliminateMultifrontal(ordering, function: EliminatePreferCholesky);
99 else if(factorization_ == QR)
100 bayesTree_ = *graph_.eliminateMultifrontal(ordering, function: EliminateQR);
101}

111/* ************************************************************************* */
112GaussianFactor::shared_ptr Marginals::marginalFactor(Key variable) const {
113 gttic(marginalFactor);
114
115 // Compute marginal factor
116 if(factorization_ == CHOLESKY)
117 return bayesTree_.marginalFactor(j: variable, function: EliminatePreferCholesky);
118 else if(factorization_ == QR)
119 return bayesTree_.marginalFactor(j: variable, function: EliminateQR);
120 else
121 throw std::runtime_error("Marginals::marginalFactor: Unknown factorization");
122}

125Matrix Marginals::marginalInformation(Key variable) const {
126
127 // Get information matrix (only store upper-right triangle)

144/* ************************************************************************* */
145JointMarginal Marginals::jointMarginalInformation(const KeyVector& variables) const {
146
147 // If 2 variables, we can use the BayesTree::joint function, otherwise we
148 // have to use sequential elimination.
149 if(variables.size() == 1)
150 {
151 Matrix info = marginalInformation(variable: variables.front());
152 std::vector<size_t> dims;
153 dims.push_back(x: info.rows());
154 return JointMarginal(info, dims, variables);
155 }
156 else
157 {
158 // Compute joint marginal factor graph.
159 GaussianFactorGraph jointFG;
160 if(variables.size() == 2) {
161 if(factorization_ == CHOLESKY)
162 jointFG = *bayesTree_.joint(j1: variables[0], j2: variables[1], function: EliminatePreferCholesky);
163 else if(factorization_ == QR)
164 jointFG = *bayesTree_.joint(j1: variables[0], j2: variables[1], function: EliminateQR);
165 } else {
166 if(factorization_ == CHOLESKY)
167 jointFG = GaussianFactorGraph(*graph_.marginalMultifrontalBayesTree(variables, function: EliminatePreferCholesky));
168 else if(factorization_ == QR)
169 jointFG = GaussianFactorGraph(*graph_.marginalMultifrontalBayesTree(variables, function: EliminateQR));
170 }
171
172 // Get information matrix
173 Matrix augmentedInfo = jointFG.augmentedHessian();
174 Matrix info = augmentedInfo.topLeftCorner(cRows: augmentedInfo.rows()-1, cCols: augmentedInfo.cols()-1);
175
176 // Information matrix will be returned with sorted keys
177 KeyVector variablesSorted = variables;
178 std::sort(first: variablesSorted.begin(), last: variablesSorted.end());
179
180 // Get dimensions from factor graph
181 std::vector<size_t> dims;
182 dims.reserve(n: variablesSorted.size());
183 for(const auto& key: variablesSorted) {
184 dims.push_back(x: values_.at(j: key).dim());
185 }
186
187 return JointMarginal(info, dims, variablesSorted);
188 }
189}
