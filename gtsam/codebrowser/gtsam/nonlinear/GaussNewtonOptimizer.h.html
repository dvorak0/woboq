19#pragma once
20
21#include <gtsam/nonlinear/NonlinearOptimizer.h>
22
23namespace gtsam {
24
25class GaussNewtonOptimizer;
26
27/** Parameters for Gauss-Newton optimization, inherits from
28 * NonlinearOptimizationParams.
29 */
30class GTSAM_EXPORT GaussNewtonParams : public NonlinearOptimizerParams {
31public:
32 using OptimizerType = GaussNewtonOptimizer;
33};
34
35/**
36 * This class performs Gauss-Newton nonlinear optimization
37 */
38class GTSAM_EXPORT GaussNewtonOptimizer : public NonlinearOptimizer {
39
40protected:
41 GaussNewtonParams params_;
42
43public:
44 /// @name Standard interface
45 /// @{
46
47 /** Standard constructor, requires a nonlinear factor graph, initial
48 * variable assignments, and optimization parameters. For convenience this
49 * version takes plain objects instead of shared pointers, but internally
50 * copies the objects.
51 * @param graph The nonlinear factor graph to optimize
52 * @param initialValues The initial variable assignments
53 * @param params The optimization parameters
54 */
55 GaussNewtonOptimizer(const NonlinearFactorGraph& graph, const Values& initialValues,
56 const GaussNewtonParams& params = GaussNewtonParams());
57
58 /** Standard constructor, requires a nonlinear factor graph, initial
59 * variable assignments, and optimization parameters. For convenience this
60 * version takes plain objects instead of shared pointers, but internally
61 * copies the objects.
62 * @param graph The nonlinear factor graph to optimize
63 * @param initialValues The initial variable assignments
64 */
65 GaussNewtonOptimizer(const NonlinearFactorGraph& graph, const Values& initialValues,
66 const Ordering& ordering);
67 /// @}
68
69 /// @name Advanced interface
70 /// @{
71
72 /** Virtual destructor */
73 ~GaussNewtonOptimizer() override {}
74
75 /**
76 * Perform a single iteration, returning GaussianFactorGraph corresponding to
77 * the linearized factor graph.
78 */
79 GaussianFactorGraph::shared_ptr iterate() override;
80
81 /** Read-only access the parameters */
82 const GaussNewtonParams& params() const { return params_; }
83
84 /// @}
85
86protected:
87 /** Access the parameters (base class version) */
88 const NonlinearOptimizerParams& _params() const override { return params_; }
89
90 /** Internal function for computing a COLAMD ordering if no ordering is specified */
91 GaussNewtonParams ensureHasOrdering(GaussNewtonParams params, const NonlinearFactorGraph& graph) const;
92
93};
94
95}
