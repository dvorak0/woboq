23namespace gtsam {

25class DoglegOptimizer;
26
27/** Parameters for Levenberg-Marquardt optimization. Note that this parameters
28 * class inherits from NonlinearOptimizerParams, which specifies the parameters
29 * common to all nonlinear optimization algorithms. This class also contains
30 * all of those parameters.
31 */
32class GTSAM_EXPORT DoglegParams : public NonlinearOptimizerParams {
33public:
34 /** See DoglegParams::dlVerbosity */
35 enum VerbosityDL {
36 SILENT,
37 VERBOSE
38 };
39
40 double deltaInitial; ///< The initial trust region radius (default: 10.0)
41 VerbosityDL verbosityDL; ///< The verbosity level for Dogleg (default: SILENT), see also NonlinearOptimizerParams::verbosity
42
43 DoglegParams() :
44 deltaInitial(1.0), verbosityDL(SILENT) {}
45
46 ~DoglegParams() override {}
47
48 void print(const std::string& str = "") const override {
49 NonlinearOptimizerParams::print(str);
50 std::cout << " deltaInitial: " << deltaInitial << "\n";
51 std::cout.flush();
52 }
53
54 double getDeltaInitial() const { return deltaInitial; }
55 std::string getVerbosityDL() const { return verbosityDLTranslator(verbosityDL); }
56
57 void setDeltaInitial(double deltaInitial) { this->deltaInitial = deltaInitial; }
58 void setVerbosityDL(const std::string& verbosityDL) { this->verbosityDL = verbosityDLTranslator(verbosityDL); }
59
60private:
61 VerbosityDL verbosityDLTranslator(const std::string& verbosityDL) const;
62 std::string verbosityDLTranslator(VerbosityDL verbosityDL) const;
63};
64
65/**
66 * This class performs Dogleg nonlinear optimization
67 */
68class GTSAM_EXPORT DoglegOptimizer : public NonlinearOptimizer {
69
70protected:
71 DoglegParams params_;
72
73public:
74 typedef std::shared_ptr<DoglegOptimizer> shared_ptr;
75
76 /// @name Standard interface
77 /// @{
78
79 /** Standard constructor, requires a nonlinear factor graph, initial
80 * variable assignments, and optimization parameters. For convenience this
81 * version takes plain objects instead of shared pointers, but internally
82 * copies the objects.
83 * @param graph The nonlinear factor graph to optimize
84 * @param initialValues The initial variable assignments
85 * @param params The optimization parameters
86 */
87 DoglegOptimizer(const NonlinearFactorGraph& graph, const Values& initialValues,
88 const DoglegParams& params = DoglegParams());
89
90 /** Standard constructor, requires a nonlinear factor graph, initial
91 * variable assignments, and optimization parameters. For convenience this
92 * version takes plain objects instead of shared pointers, but internally
93 * copies the objects.
94 * @param graph The nonlinear factor graph to optimize
95 * @param initialValues The initial variable assignments
96 */
97 DoglegOptimizer(const NonlinearFactorGraph& graph, const Values& initialValues,
98 const Ordering& ordering);
99
100 /// @}
101
102 /// @name Advanced interface
103 /// @{
104
105 /** Virtual destructor */
106 ~DoglegOptimizer() override {}
107
108 /**
109 * Perform a single iteration, returning GaussianFactorGraph corresponding to
110 * the linearized factor graph.
111 */
112 GaussianFactorGraph::shared_ptr iterate() override;
113
114 /** Read-only access the parameters */
115 const DoglegParams& params() const { return params_; }
116
117 /** Access the current trust region radius delta */
118 double getDelta() const;
119
120 /// @}
121
122protected:
123 /** Access the parameters (base class version) */
124 const NonlinearOptimizerParams& _params() const override { return params_; }
125
126 /** Internal function for computing a COLAMD ordering if no ordering is specified */
127 DoglegParams ensureHasOrdering(DoglegParams params, const NonlinearFactorGraph& graph) const;
128};
