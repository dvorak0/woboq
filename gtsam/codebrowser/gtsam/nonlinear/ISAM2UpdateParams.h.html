25namespace gtsam {
26
27/**
28 * @ingroup isam2
29 * This struct is used by ISAM2::update() to pass additional parameters to
30 * give the user a fine-grained control on how factors and relinearized, etc.
31 */
32struct ISAM2UpdateParams {
33 ISAM2UpdateParams() = default;
34
35 /** Indices of factors to remove from system (default: empty) */
36 FactorIndices removeFactorIndices;
37
38 /** An optional map of keys to group labels, such that a variable can be
39 * constrained to a particular grouping in the BayesTree */
40 std::optional<FastMap<Key, int>> constrainedKeys;
41
42 /** An optional set of nonlinear keys that iSAM2 will hold at a constant
43 * linearization point, regardless of the size of the linear delta */
44 std::optional<FastList<Key>> noRelinKeys;
45
46 /** An optional set of nonlinear keys that iSAM2 will re-eliminate, regardless
47 * of the size of the linear delta. This allows the provided keys to be
48 * reordered. */
49 std::optional<FastList<Key>> extraReelimKeys;
50
51 /** Relinearize any variables whose delta magnitude is sufficiently large
52 * (Params::relinearizeThreshold), regardless of the relinearization
53 * interval (Params::relinearizeSkip). */
54 bool force_relinearize{false};
55
56 /** An optional set of new Keys that are now affected by factors,
57 * indexed by factor indices (as returned by ISAM2::update()).
58 * Use when working with smart factors. For example:
59 * - Timestamp `i`: ISAM2::update() called with a new smart factor depending
60 * on Keys `X(0)` and `X(1)`. It returns that the factor index for the new
61 * smart factor (inside ISAM2) is `13`.
62 * - Timestamp `i+1`: The same smart factor has been augmented to now also
63 * depend on Keys `X(2)`, `X(3)`. Next call to ISAM2::update() must include
64 * its `newAffectedKeys` field with the map `13 -> {X(2), X(3)}`.
65 */
66 std::optional<FastMap<FactorIndex, KeySet>> newAffectedKeys;
67
68 /** By default, iSAM2 uses a wildfire update scheme that stops updating when
69 * the deltas become too small down in the tree. This flagg forces a full
70 * solve instead. */
71 bool forceFullSolve{false};
72};
73
74} // namespace gtsam
