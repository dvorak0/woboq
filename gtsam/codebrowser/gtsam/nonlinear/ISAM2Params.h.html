30/**
31 * @ingroup isam2
32 * Parameters for ISAM2 using Gauss-Newton optimization. Either this class or
33 * ISAM2DoglegParams should be specified as the optimizationParams in
34 * ISAM2Params, which should in turn be passed to ISAM2(const ISAM2Params&).
35 */
36struct GTSAM_EXPORT ISAM2GaussNewtonParams {
37 double
38 wildfireThreshold; ///< Continue updating the linear delta only when
39 ///< changes are above this threshold (default: 0.001)
40
41 /** Specify parameters as constructor arguments */
42 ISAM2GaussNewtonParams(
43 double _wildfireThreshold =
44 0.001 ///< see ISAM2GaussNewtonParams public variables,
45 ///< ISAM2GaussNewtonParams::wildfireThreshold
46 )
47 : wildfireThreshold(_wildfireThreshold) {}
48
49 void print(const std::string str = "") const {
50 using std::cout;
51 cout << str << "type: ISAM2GaussNewtonParams\n";
52 cout << str << "wildfireThreshold: " << wildfireThreshold << "\n";
53 cout.flush();
54 }
55
56 double getWildfireThreshold() const { return wildfireThreshold; }
57 void setWildfireThreshold(double wildfireThreshold) {
58 this->wildfireThreshold = wildfireThreshold;
59 }
60};

62/**
63 * @ingroup isam2
64 * Parameters for ISAM2 using Dogleg optimization. Either this class or
65 * ISAM2GaussNewtonParams should be specified as the optimizationParams in
66 * ISAM2Params, which should in turn be passed to ISAM2(const ISAM2Params&).
67 */
68struct GTSAM_EXPORT ISAM2DoglegParams {
69 double initialDelta; ///< The initial trust region radius for Dogleg
70 double
71 wildfireThreshold; ///< Continue updating the linear delta only when
72 ///< changes are above this threshold (default: 1e-5)
73 DoglegOptimizerImpl::TrustRegionAdaptationMode
74 adaptationMode; ///< See description in
75 ///< DoglegOptimizerImpl::TrustRegionAdaptationMode
76 bool
77 verbose; ///< Whether Dogleg prints iteration and convergence information
78
79 /** Specify parameters as constructor arguments */
80 ISAM2DoglegParams(
81 double _initialDelta = 1.0, ///< see ISAM2DoglegParams::initialDelta
82 double _wildfireThreshold =
83 1e-5, ///< see ISAM2DoglegParams::wildfireThreshold
84 DoglegOptimizerImpl::TrustRegionAdaptationMode _adaptationMode =
85 DoglegOptimizerImpl::
86 SEARCH_EACH_ITERATION, ///< see ISAM2DoglegParams::adaptationMode
87 bool _verbose = false ///< see ISAM2DoglegParams::verbose
88 )
89 : initialDelta(_initialDelta),
90 wildfireThreshold(_wildfireThreshold),
91 adaptationMode(_adaptationMode),
92 verbose(_verbose) {}
93
94 void print(const std::string str = "") const {
95 using std::cout;
96 cout << str << "type: ISAM2DoglegParams\n";
97 cout << str << "initialDelta: " << initialDelta << "\n";
98 cout << str << "wildfireThreshold: " << wildfireThreshold << "\n";
99 cout << str
100 << "adaptationMode: " << adaptationModeTranslator(adaptationMode)
101 << "\n";
102 cout.flush();
103 }
104
105 double getInitialDelta() const { return initialDelta; }
106 double getWildfireThreshold() const { return wildfireThreshold; }
107 std::string getAdaptationMode() const {
108 return adaptationModeTranslator(adaptationMode);
109 }
110 bool isVerbose() const { return verbose; }
111 void setInitialDelta(double initialDelta) {
112 this->initialDelta = initialDelta;
113 }
114 void setWildfireThreshold(double wildfireThreshold) {
115 this->wildfireThreshold = wildfireThreshold;
116 }
117 void setAdaptationMode(const std::string& adaptationMode) {
118 this->adaptationMode = adaptationModeTranslator(adaptationMode);
119 }
120 void setVerbose(bool verbose) { this->verbose = verbose; }
121
122 std::string adaptationModeTranslator(
123 const DoglegOptimizerImpl::TrustRegionAdaptationMode& adaptationMode)
124 const;
125 DoglegOptimizerImpl::TrustRegionAdaptationMode adaptationModeTranslator(
126 const std::string& adaptationMode) const;
127};

129/**
130 * @ingroup isam2
131 * Parameters for the ISAM2 algorithm. Default parameter values are listed
132 * below.
133 */
134typedef FastMap<char, Vector> ISAM2ThresholdMap;
135typedef ISAM2ThresholdMap::value_type ISAM2ThresholdMapValue;
136struct GTSAM_EXPORT ISAM2Params {
137 typedef std::variant<ISAM2GaussNewtonParams, ISAM2DoglegParams>
138 OptimizationParams; ///< Either ISAM2GaussNewtonParams or
139 ///< ISAM2DoglegParams
140 typedef std::variant<double, FastMap<char, Vector> >
141 RelinearizationThreshold; ///< Either a constant relinearization
142 ///< threshold or a per-variable-type set of
143 ///< thresholds
144
145 /** Optimization parameters, this both selects the nonlinear optimization
146 * method and specifies its parameters, either ISAM2GaussNewtonParams or
147 * ISAM2DoglegParams. In the former, Gauss-Newton optimization will be used
148 * with the specified parameters, and in the latter Powell's dog-leg
149 * algorithm will be used with the specified parameters.
150 */
151 OptimizationParams optimizationParams;
152
153 /** Only relinearize variables whose linear delta magnitude is greater than
154 * this threshold (default: 0.1). If this is a FastMap<char,Vector> instead
155 * of a double, then the threshold is specified for each dimension of each
156 * variable type. This parameter then maps from a character indicating the
157 * variable type to a Vector of thresholds for each dimension of that
158 * variable. For example, if Pose keys are of type TypedSymbol<'x',Pose3>,
159 * and landmark keys are of type TypedSymbol<'l',Point3>, then appropriate
160 * entries would be added with:
161 * \code
162 FastMap<char,Vector> thresholds;
163 thresholds['x'] = (Vector(6) << 0.1, 0.1, 0.1, 0.5, 0.5, 0.5).finished();
164 // 0.1 rad rotation threshold, 0.5 m translation threshold thresholds['l'] =
165 Vector3(1.0, 1.0, 1.0); // 1.0 m landmark position threshold
166 params.relinearizeThreshold = thresholds;
167 \endcode
168 */
169 RelinearizationThreshold relinearizeThreshold;
170
171 int relinearizeSkip; ///< Only relinearize any variables every
172 ///< relinearizeSkip calls to ISAM2::update (default:
173 ///< 10)
174
175 bool enableRelinearization; ///< Controls whether ISAM2 will ever relinearize
176 ///< any variables (default: true)
177
178 bool evaluateNonlinearError; ///< Whether to evaluate the nonlinear error
179 ///< before and after the update, to return in
180 ///< ISAM2Result from update()
181
182 enum Factorization { CHOLESKY, QR };
183 /** Specifies whether to use QR or CHOESKY numerical factorization (default:
184 * CHOLESKY). Cholesky is faster but potentially numerically unstable for
185 * poorly-conditioned problems, which can occur when uncertainty is very low
186 * in some variables (or dimensions of variables) and very high in others. QR
187 * is slower but more numerically stable in poorly-conditioned problems. We
188 * suggest using the default of Cholesky unless gtsam sometimes throws
189 * IndefiniteLinearSystemException when your problem's Hessian is actually
190 * positive definite. For positive definite problems, numerical error
191 * accumulation can cause the problem to become numerically negative or
192 * indefinite as solving proceeds, especially when using Cholesky.
193 */
194 Factorization factorization;
195
196 /** Whether to cache linear factors (default: true).
197 * This can improve performance if linearization is expensive, but can hurt
198 * performance if linearization is very cleap due to computation to look up
199 * additional keys.
200 */
201 bool cacheLinearizedFactors;
202
203 KeyFormatter
204 keyFormatter; ///< A KeyFormatter for when keys are printed during
205 ///< debugging (default: DefaultKeyFormatter)
206
207 bool enableDetailedResults; ///< Whether to compute and return
208 ///< ISAM2Result::detailedResults, this can
209 ///< increase running time (default: false)
210
211 /** Check variables for relinearization in tree-order, stopping the check once
212 * a variable does not need to be relinearized (default: false). This can
213 * improve speed by only checking a small part of the top of the tree.
214 * However, variables below the check cut-off can accumulate significant
215 * deltas without triggering relinearization. This is particularly useful in
216 * exploration scenarios where real-time performance is desired over
217 * correctness. Use with caution.
218 */
219 bool enablePartialRelinearizationCheck;
220
221 /// When you will be removing many factors, e.g. when using ISAM2 as a
222 /// fixed-lag smoother, enable this option to add factors in the first
223 /// available factor slots, to avoid accumulating nullptr factor slots, at the
224 /// cost of having to search for slots every time a factor is added.
225 bool findUnusedFactorSlots;
226
227 /**
228 * Specify parameters as constructor arguments
229 * See the documentation of member variables above.
230 */
231 ISAM2Params(OptimizationParams _optimizationParams = ISAM2GaussNewtonParams(),
232 RelinearizationThreshold _relinearizeThreshold = 0.1,
233 int _relinearizeSkip = 10, bool _enableRelinearization = true,
234 bool _evaluateNonlinearError = false,
235 Factorization _factorization = ISAM2Params::CHOLESKY,
236 bool _cacheLinearizedFactors = true,
237 const KeyFormatter& _keyFormatter =
238 DefaultKeyFormatter, ///< see ISAM2::Params::keyFormatter,
239 bool _enableDetailedResults = false)
240 : optimizationParams(_optimizationParams),
241 relinearizeThreshold(_relinearizeThreshold),
242 relinearizeSkip(_relinearizeSkip),
243 enableRelinearization(_enableRelinearization),
244 evaluateNonlinearError(_evaluateNonlinearError),
245 factorization(_factorization),
246 cacheLinearizedFactors(_cacheLinearizedFactors),
247 keyFormatter(_keyFormatter),
248 enableDetailedResults(_enableDetailedResults),
249 enablePartialRelinearizationCheck(false),
250 findUnusedFactorSlots(false) {}
251
252 /// print iSAM2 parameters
253 void print(const std::string& str = "") const {
254 using std::cout;
255 cout << str << "\n";
256
257 static const std::string kStr("optimizationParams: ");
258 if (std::holds_alternative<ISAM2GaussNewtonParams>(v: optimizationParams)) {
259 std::get<ISAM2GaussNewtonParams>(v: optimizationParams).print();
260 } else if (std::holds_alternative<ISAM2DoglegParams>(v: optimizationParams)) {
261 std::get<ISAM2DoglegParams>(v: optimizationParams).print(str: kStr);
262 } else {
263 cout << kStr << "{unknown type}\n";
264 }
265
266 cout << "relinearizeThreshold: ";
267 if (std::holds_alternative<double>(v: relinearizeThreshold)) {
268 cout << std::get<double>(v: relinearizeThreshold) << "\n";
269 } else {
270 cout << "{mapped}\n";
271 for (const ISAM2ThresholdMapValue& value :
272 std::get<ISAM2ThresholdMap>(v: relinearizeThreshold)) {
273 cout << " '" << value.first
274 << "' -> [" << value.second.transpose() << " ]\n";
275 }
276 }
277
278 cout << "relinearizeSkip: " << relinearizeSkip << "\n";
279 cout << "enableRelinearization: " << enableRelinearization
280 << "\n";
281 cout << "evaluateNonlinearError: " << evaluateNonlinearError
282 << "\n";
283 cout << "factorization: "
284 << factorizationTranslator(value: factorization) << "\n";
285 cout << "cacheLinearizedFactors: " << cacheLinearizedFactors
286 << "\n";
287 cout << "enableDetailedResults: " << enableDetailedResults
288 << "\n";
289 cout << "enablePartialRelinearizationCheck: "
290 << enablePartialRelinearizationCheck << "\n";
291 cout << "findUnusedFactorSlots: " << findUnusedFactorSlots
292 << "\n";
293 cout.flush();
294 }

323 GaussianFactorGraph::Eliminate getEliminationFunction() const {
324 return factorization == CHOLESKY
325 ? (GaussianFactorGraph::Eliminate)EliminatePreferCholesky
326 : (GaussianFactorGraph::Eliminate)EliminateQR;
327 }
