39/* ************************************************************************* */
40bool FixedLagSmoother::equals(const FixedLagSmoother& rhs, double tol) const {
41 return std::abs(x: smootherLag_ - rhs.smootherLag_) < tol
42 && std::equal(first1: timestampKeyMap_.begin(), last1: timestampKeyMap_.end(), first2: rhs.timestampKeyMap_.begin());
43}

45/* ************************************************************************* */
46void FixedLagSmoother::updateKeyTimestampMap(const KeyTimestampMap& timestamps) {
47 // Loop through each key and add/update it in the map
48 for(const auto& key_timestamp: timestamps) {
49 // Check to see if this key already exists in the database
50 KeyTimestampMap::iterator keyIter = keyTimestampMap_.find(x: key_timestamp.first);
51
52 // If the key already exists
53 if(keyIter != keyTimestampMap_.end()) {
54 // Find the entry in the Timestamp-Key database
55 std::pair<TimestampKeyMap::iterator,TimestampKeyMap::iterator> range = timestampKeyMap_.equal_range(x: keyIter->second);
56 TimestampKeyMap::iterator timeIter = range.first;
57 while(timeIter->second != key_timestamp.first) {
58 ++timeIter;
59 }
60 // remove the entry in the Timestamp-Key database
61 timestampKeyMap_.erase(position: timeIter);
62 // insert an entry at the new time
63 timestampKeyMap_.insert(x: TimestampKeyMap::value_type(key_timestamp.second, key_timestamp.first));
64 // update the Key-Timestamp database
65 keyIter->second = key_timestamp.second;
66 } else {
67 // Add the Key-Timestamp database
68 keyTimestampMap_.insert(x: key_timestamp);
69 // Add the key to the Timestamp-Key database
70 timestampKeyMap_.insert(x: TimestampKeyMap::value_type(key_timestamp.second, key_timestamp.first));
71 }
72 }
73}

75/* ************************************************************************* */
76void FixedLagSmoother::eraseKeyTimestampMap(const KeyVector& keys) {
77 for(Key key: keys) {
78 // Erase the key from the Timestamp->Key map
79 double timestamp = keyTimestampMap_.at(k: key);
80
81 TimestampKeyMap::iterator iter = timestampKeyMap_.lower_bound(x: timestamp);
82 while(iter != timestampKeyMap_.end() && iter->first == timestamp) {
83 if(iter->second == key) {
84 timestampKeyMap_.erase(position: iter++);
85 } else {
86 ++iter;
87 }
88 }
89 // Erase the key from the Key->Timestamp map
90 keyTimestampMap_.erase(x: key);
91 }
92}

94/* ************************************************************************* */
95double FixedLagSmoother::getCurrentTimestamp() const {
96 if(timestampKeyMap_.size() > 0) {
97 return timestampKeyMap_.rbegin()->first;
98 } else {
99 return -std::numeric_limits<double>::max();
100 }
101}

103/* ************************************************************************* */
104KeyVector FixedLagSmoother::findKeysBefore(double timestamp) const {
105 KeyVector keys;
106 TimestampKeyMap::const_iterator end = timestampKeyMap_.lower_bound(x: timestamp);
107 for(TimestampKeyMap::const_iterator iter = timestampKeyMap_.begin(); iter != end; ++iter) {
108 keys.push_back(x: iter->second);
109 }
110 return keys;
111}
