34/* ************************************************************************* */
35void NonlinearISAM::update(const NonlinearFactorGraph& newFactors, const Values& initialValues) {
36
37 if(newFactors.size() > 0) {
38
39 // Reorder and relinearize every reorderInterval updates
40 if(reorderInterval_ > 0 && ++reorderCounter_ >= reorderInterval_) {
41 reorder_relinearize();
42 reorderCounter_ = 0;
43 }
44
45 factors_.push_back(container: newFactors);
46
47 // Linearize new factors and insert them
48 // TODO: optimize for whole config?
49 linPoint_.insert(values: initialValues);
50
51 std::shared_ptr<GaussianFactorGraph> linearizedNewFactors = newFactors.linearize(linearizationPoint: linPoint_);
52
53 // Update ISAM
54 isam_.update(newFactors: *linearizedNewFactors, function: eliminationFunction_);
55 }
56}
57
58/* ************************************************************************* */
59void NonlinearISAM::reorder_relinearize() {
60
61// cout << "Reordering, relinearizing..." << endl;
62
63 if(factors_.size() > 0) {
64 // Obtain the new linearization point
65 const Values newLinPoint = estimate();
66
67 isam_.clear();
68
69 // Just recreate the whole BayesTree
70 // TODO: allow for constrained ordering here
71 // TODO: decouple relinearization and reordering to avoid
72 isam_.update(newFactors: *factors_.linearize(linearizationPoint: newLinPoint), function: eliminationFunction_);
73
74 // Update linearization point
75 linPoint_ = newLinPoint;
76 }
77}

79/* ************************************************************************* */
80Values NonlinearISAM::estimate() const {
81 if(isam_.size() > 0)
82 return linPoint_.retract(delta: isam_.optimize());
83 else
84 return linPoint_;
85}
