30/**
31 * A class for computing marginals of variables in a DiscreteFactorGraph
32 * @ingroup discrete
33 */
34class GTSAM_EXPORT DiscreteMarginals {
35 protected:
36 DiscreteBayesTree::shared_ptr bayesTree_;
37
38 public:
39 DiscreteMarginals() {}
40
41 /** Construct a marginals class.
42 * @param graph The factor graph defining the full joint
43 * distribution on all variables.
44 */
45 DiscreteMarginals(const DiscreteFactorGraph& graph);
46
47 /** Compute the marginal of a single variable */
48 DiscreteFactor::shared_ptr operator()(Key variable) const;
49
50 /** Compute the marginal of a single variable
51 * @param key DiscreteKey of the Variable
52 * @return Vector of marginal probabilities
53 */
54 Vector marginalProbabilities(const DiscreteKey& key) const;
55
56 /// Print details
57 void print(const std::string& s = "",
58 const KeyFormatter formatter = DefaultKeyFormatter) const;
59};
