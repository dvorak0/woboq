32/**
33 * Discrete Conditional Density
34 * Derives from DecisionTreeFactor
35 *
36 * @ingroup discrete
37 */
38class GTSAM_EXPORT DiscreteConditional
39 : public DecisionTreeFactor,
40 public Conditional<DecisionTreeFactor, DiscreteConditional> {
41 public:
42 // typedefs needed to play nice with gtsam
43 typedef DiscreteConditional This; ///< Typedef to this class
44 typedef std::shared_ptr<This> shared_ptr; ///< shared_ptr to this class
45 typedef DecisionTreeFactor BaseFactor; ///< Typedef to our factor base class
46 typedef Conditional<BaseFactor, This>
47 BaseConditional; ///< Typedef to our conditional base class
48
49 using Values = DiscreteValues; ///< backwards compatibility
50
51 /// @name Standard Constructors
52 /// @{
53
54 /// Default constructor needed for serialization.
55 DiscreteConditional() {}
56
57 /// Construct from factor, taking the first `nFrontals` keys as frontals.
58 DiscreteConditional(size_t nFrontals, const DiscreteFactor& f);
59
60 /**
61 * Construct from DiscreteKeys and AlgebraicDecisionTree, taking the first
62 * `nFrontals` keys as frontals, in the order given.
63 */
64 DiscreteConditional(size_t nFrontals, const DiscreteKeys& keys,
65 const ADT& potentials);
66
67 /** Construct from signature */
68 explicit DiscreteConditional(const Signature& signature);
69
70 /**
71 * Construct from key, parents, and a Signature::Table specifying the
72 * conditional probability table (CPT) in 00 01 10 11 order. For
73 * three-valued, it would be 00 01 02 10 11 12 20 21 22, etc....
74 *
75 * Example: DiscreteConditional P(D, {B,E}, table);
76 */
77 DiscreteConditional(const DiscreteKey& key, const DiscreteKeys& parents,
78 const Signature::Table& table)
79 : DiscreteConditional(Signature(key, parents, table)) {}
80
81 /**
82 * Construct from key, parents, and a vector<double> specifying the
83 * conditional probability table (CPT) in 00 01 10 11 order. For
84 * three-valued, it would be 00 01 02 10 11 12 20 21 22, etc....
85 *
86 * Example: DiscreteConditional P(D, {B,E}, table);
87 */
88 DiscreteConditional(const DiscreteKey& key, const DiscreteKeys& parents,
89 const std::vector<double>& table)
90 : DiscreteConditional(1, DiscreteKeys{key} & parents,
91 ADT(DiscreteKeys{key} & parents, table)) {}
92
93 /**
94 * Construct from key, parents, and a string specifying the conditional
95 * probability table (CPT) in 00 01 10 11 order. For three-valued, it would
96 * be 00 01 02 10 11 12 20 21 22, etc....
97 *
98 * The string is parsed into a Signature::Table.
99 *
100 * Example: DiscreteConditional P(D, {B,E}, "9/1 2/8 3/7 1/9");
101 */
102 DiscreteConditional(const DiscreteKey& key, const DiscreteKeys& parents,
103 const std::string& spec)
104 : DiscreteConditional(Signature(key, parents, spec)) {}
105
106 /// No-parent specialization; can also use DiscreteDistribution.
107 DiscreteConditional(const DiscreteKey& key, const std::string& spec)
108 : DiscreteConditional(Signature(key, {}, spec)) {}
109
110 /**
111 * @brief construct P(X|Y) = f(X,Y)/f(Y) from f(X,Y) and f(Y)
112 * Assumes but *does not check* that f(Y)=sum_X f(X,Y).
113 */
114 DiscreteConditional(const DecisionTreeFactor& joint,
115 const DecisionTreeFactor& marginal);
116
117 /**
118 * @brief construct P(X|Y) = f(X,Y)/f(Y) from f(X,Y) and f(Y)
119 * Assumes but *does not check* that f(Y)=sum_X f(X,Y).
120 * Makes sure the keys are ordered as given. Does not check orderedKeys.
121 */
122 DiscreteConditional(const DecisionTreeFactor& joint,
123 const DecisionTreeFactor& marginal,
124 const Ordering& orderedKeys);
125
126 using DecisionTreeFactor::operator*;
127
128 /**
129 * @brief Combine two conditionals, yielding a new conditional with the union
130 * of the frontal keys, ordered by gtsam::Key.
131 *
132 * The two conditionals must make a valid Bayes net fragment, i.e.,
133 * the frontal variables cannot overlap, and must be acyclic:
134 * Example of correct use:
135 * P(A,B) = P(A|B) * P(B)
136 * P(A,B|C) = P(A|B) * P(B|C)
137 * P(A,B,C) = P(A,B|C) * P(C)
138 * Example of incorrect use:
139 * P(A|B) * P(A|C) = ?
140 * P(A|B) * P(B|A) = ?
141 * We check for overlapping frontals, but do *not* check for cyclic.
142 */
143 DiscreteConditional operator*(const DiscreteConditional& other) const;
144
145 /** Calculate marginal on given key, no parent case. */
146 DiscreteConditional marginal(Key key) const;
147
148 /// @}
149 /// @name Testable
150 /// @{
151
152 /// GTSAM-style print
153 void print(
154 const std::string& s = "Discrete Conditional: ",
155 const KeyFormatter& formatter = DefaultKeyFormatter) const override;
156
157 /// GTSAM-style equals
158 bool equals(const DiscreteFactor& other, double tol = 1e-9) const override;
159
160 /// @}
161 /// @name Standard Interface
162 /// @{
163
164 /// Log-probability is just -error(x).
165 double logProbability(const DiscreteValues& x) const { return -error(values: x); }
166
167 /// print index signature only
168 void printSignature(
169 const std::string& s = "Discrete Conditional: ",
170 const KeyFormatter& formatter = DefaultKeyFormatter) const {
171 static_cast<const BaseConditional*>(this)->print(s, formatter);
172 }
173
174 using BaseFactor::error; ///< DiscreteValues version
175 using BaseFactor::evaluate; ///< DiscreteValues version
176 using BaseFactor::operator(); ///< DiscreteValues version
177
178 /**
179 * @brief restrict to given *parent* values.
180 *
181 * Note: does not need be complete set. Examples:
182 *
183 * P(C|D,E) + . -> P(C|D,E)
184 * P(C|D,E) + E -> P(C|D)
185 * P(C|D,E) + D -> P(C|E)
186 * P(C|D,E) + D,E -> P(C)
187 * P(C|D,E) + C -> error!
188 *
189 * @return a shared_ptr to a new DiscreteConditional
190 */
191 shared_ptr choose(const DiscreteValues& given) const;
192
193 /** Convert to a likelihood factor by providing value before bar. */
194 DecisionTreeFactor::shared_ptr likelihood(
195 const DiscreteValues& frontalValues) const;
196
197 /** Single variable version of likelihood. */
198 DecisionTreeFactor::shared_ptr likelihood(size_t frontal) const;
199
200 /**
201 * Sample from conditional, given missing variables
202 * Example:
203 * std::mt19937_64 rng(42);
204 * DiscreteValues given = ...;
205 * size_t sample = dc.sample(given, &rng);
206 *
207 * @param parentsValues Known values of the parents
208 * @param rng Pseudo-Random Number Generator.
209 * @return sample from conditional
210 */
211 virtual size_t sample(const DiscreteValues& parentsValues,
212 std::mt19937_64* rng = nullptr) const;
213
214 /// Single parent version.
215 size_t sample(size_t parent_value, std::mt19937_64* rng = nullptr) const;
216
217 /**
218 * Sample from conditional, zero parent version
219 * Example:
220 * std::mt19937_64 rng(42);
221 * auto sample = dc.sample(&rng);
222 */
223 size_t sample(std::mt19937_64* rng = nullptr) const;
224
225 /**
226 * @brief Return assignment for single frontal variable that maximizes value.
227 * @param parentsValues Known assignments for the parents.
228 * @return maximizing assignment for the frontal variable.
229 */
230 size_t argmax(const DiscreteValues& parentsValues = DiscreteValues()) const;
231
232 /**
233 * @brief Create new factor by maximizing over all
234 * values with the same separator.
235 *
236 * @param keys The keys to sum over.
237 * @return DiscreteFactor::shared_ptr
238 */
239 virtual DiscreteFactor::shared_ptr max(const Ordering& keys) const override;
240
241 /// @}
242 /// @name Advanced Interface
243 /// @{
244
245 /// Sample in place with optional PRNG, stores result in partial solution
246 void sampleInPlace(DiscreteValues* parentsValues,
247 std::mt19937_64* rng = nullptr) const;
248
249 /// Return all assignments for frontal variables.
250 std::vector<DiscreteValues> frontalAssignments() const;
251
252 /// Return all assignments for frontal *and* parent variables.
253 std::vector<DiscreteValues> allAssignments() const;
254
255 /// @}
256 /// @name Wrapper support
257 /// @{
258
259 /// Render as markdown table.
260 std::string markdown(const KeyFormatter& keyFormatter = DefaultKeyFormatter,
261 const Names& names = {}) const override;
262
263 /// Render as html table.
264 std::string html(const KeyFormatter& keyFormatter = DefaultKeyFormatter,
265 const Names& names = {}) const override;
266
267 /// @}
268 /// @name HybridValues methods.
269 /// @{
270
271 /**
272 * Calculate probability for HybridValues `x`.
273 * Dispatches to DiscreteValues version.
274 */
275 double evaluate(const HybridValues& x) const override;
276
277 using BaseConditional::operator(); ///< HybridValues version
278
279 /**
280 * Calculate log-probability log(evaluate(x)) for HybridValues `x`.
281 * This is actually just -error(x).
282 */
283 double logProbability(const HybridValues& x) const override {
284 return -error(values: x);
285 }
286
287 /**
288 * negLogConstant is just zero, such that
289 * -logProbability(x) = -log(evaluate(x)) = error(x)
290 * and hence error(x) > 0 for all x.
291 * Thus -log(K) for the normalization constant k is 0.
292 */
293 double negLogConstant() const override;
294
295 /// Prune the conditional
296 virtual void prune(size_t maxNrAssignments);
297
298 /**
299 * @brief Remove the discrete modes whose assignments are given to us.
300 * Only applies to discrete conditionals.
301 *
302 * Imperative method so we can update nodes in the Bayes net or Bayes tree.
303 *
304 * @param given The discrete modes whose assignments we know.
305 */
306 void removeDiscreteModes(const DiscreteValues& given);
307
308 /// @}
309
310 protected:
311 /// Internal version of choose
312 DiscreteConditional::ADT choose(const DiscreteValues& given,
313 bool forceComplete) const;
314
315 private:
316#if GTSAM_ENABLE_BOOST_SERIALIZATION
317 /** Serialization function */
318 friend class boost::serialization::access;
319 template <class Archive>
320 void serialize(Archive& ar, const unsigned int /*version*/) {
321 ar& BOOST_SERIALIZATION_BASE_OBJECT_NVP(BaseFactor);
322 ar& BOOST_SERIALIZATION_BASE_OBJECT_NVP(BaseConditional);
323 }
324#endif
325};
