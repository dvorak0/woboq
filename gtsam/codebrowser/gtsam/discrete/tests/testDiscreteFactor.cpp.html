44/*
45 * Test simple DiscretePriorFactor. We construct a DiscreteFactor factor
46 * graph with a single binary variable d1 and a single custom
47 * DiscretePriorFactor p(d1) specifying the distribution:
48 * p(d1 = 0) = 0.1, p(d1 = 1) = 0.9.
49 *
50 * After constructing the factor graph, we validate that the marginals are
51 * correct (i.e. they match the input p(d1)) and the most probable estimate
52 * (computed by maximizing p(d1) over possible values of d1) is 1.
53 */
54TEST(DCSAM, DiscretePriorFactor) {
55 // Make an empty discrete factor graph
56 DiscreteFactorGraph dfg;
57
58 // We'll make a variable with 2 possible assignments
59 const size_t cardinality = 2;
60 DiscreteKey dk(D(j: 1), cardinality);
61 const std::vector<double> probs{0.1, 0.9};
62
63 // Make a discrete prior factor and add it to the graph
64 DecisionTreeFactor dpf(dk, probs);
65 dfg.push_back(factor: dpf);
66
67 // Solve
68 DiscreteValues mostProbableEstimate = dfg.optimize();
69
70 // Get the most probable estimate
71 const size_t mpeD = mostProbableEstimate.at(k: dk.first);
72
73 // Get the marginals
74 DiscreteMarginals discreteMarginals(dfg);
75 Vector margProbs = discreteMarginals.marginalProbabilities(key: dk);
76
77 // Verify that each marginal probability is within `tol` of the true marginal
78 for (size_t i = 0; i < dk.second; i++) {
79 bool margWithinTol = (abs(x: margProbs[i] - probs[i]) < 1e-7);
80 EXPECT(margWithinTol);
81 }
82
83 // Ensure that the most probable estimate is correct
84 EXPECT_LONGS_EQUAL(mpeD, 1);
85}
