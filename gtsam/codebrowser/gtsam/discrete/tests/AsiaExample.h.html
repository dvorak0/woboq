19#include <gtsam/discrete/DiscreteBayesNet.h>
20#include <gtsam/inference/Symbol.h>
21
22namespace gtsam {
23namespace asia_example {
24
25[[maybe_unused]] static const Key D = Symbol('D', 1), X = Symbol('X', 2),
26 E = Symbol('E', 3), B = Symbol('B', 4),
27 L = Symbol('L', 5), T = Symbol('T', 6),
28 S = Symbol('S', 7), A = Symbol('A', 8);
29
30[[maybe_unused]] static const DiscreteKey Dyspnea(D, 2), XRay(X, 2),
31 Either(E, 2), Bronchitis(B, 2), LungCancer(L, 2), Tuberculosis(T, 2),
32 Smoking(S, 2), Asia(A, 2);
33
34// Function to construct the Asia priors
35[[maybe_unused]] static DiscreteBayesNet createPriors() {
36 DiscreteBayesNet priors;
37 priors.add(args: Smoking % "50/50");
38 priors.add(args: Asia, args: "99/1");
39 return priors;
40}
41
42// Function to construct the incomplete Asia example
43[[maybe_unused]] static DiscreteBayesNet createFragment() {
44 DiscreteBayesNet fragment;
45 fragment.add(args&: (Either | Tuberculosis, LungCancer) = "F T T T");
46 fragment.add(args&: LungCancer | Smoking = "99/1 90/10");
47 fragment.add(args&: Tuberculosis | Asia = "99/1 95/5");
48 for (const auto& factor : createPriors()) fragment.push_back(factor);
49 return fragment;
50}
51
52// Function to construct the Asia example
53[[maybe_unused]] static DiscreteBayesNet createAsiaExample() {
54 DiscreteBayesNet asia;
55 asia.add(args&: (Dyspnea | Either, Bronchitis) = "9/1 2/8 3/7 1/9");
56 asia.add(args&: XRay | Either = "95/5 2/98");
57 asia.add(args&: Bronchitis | Smoking = "70/30 40/60");
58 for (const auto& factor : createFragment()) asia.push_back(factor);
59 return asia;
60}
61} // namespace asia_example
62} // namespace gtsam
