34struct TestFixture {
35 DiscreteKeys keys;
36 std::vector<DiscreteValues> assignments;
37 DiscreteBayesNet bayesNet;
38 std::shared_ptr<DiscreteBayesTree> bayesTree;
39
40 /**
41 * Create a thin-tree Bayesnet, a la Jean-Guillaume Durand (former student),
42 * and then create the Bayes tree from it.
43 */
44 TestFixture() {
45 // Define variables.
46 for (int i = 0; i < 15; i++) {
47 DiscreteKey key_i(i, 2);
48 keys.push_back(x: key_i);
49 }
50
51 // Enumerate all assignments.
52 assignments = DiscreteValues::CartesianProduct(keys);
53
54 // Create thin-tree Bayesnet.
55 bayesNet.add(args: keys[14] % "1/3");
56
57 bayesNet.add(args&: keys[13] | keys[14] = "1/3 3/1");
58 bayesNet.add(args&: keys[12] | keys[14] = "3/1 3/1");
59
60 bayesNet.add(args&: (keys[11] | keys[13], keys[14]) = "1/4 2/3 3/2 4/1");
61 bayesNet.add(args&: (keys[10] | keys[13], keys[14]) = "1/4 3/2 2/3 4/1");
62 bayesNet.add(args&: (keys[9] | keys[12], keys[14]) = "4/1 2/3 F 1/4");
63 bayesNet.add(args&: (keys[8] | keys[12], keys[14]) = "T 1/4 3/2 4/1");
64
65 bayesNet.add(args&: (keys[7] | keys[11], keys[13]) = "1/4 2/3 3/2 4/1");
66 bayesNet.add(args&: (keys[6] | keys[11], keys[13]) = "1/4 3/2 2/3 4/1");
67 bayesNet.add(args&: (keys[5] | keys[10], keys[13]) = "4/1 2/3 3/2 1/4");
68 bayesNet.add(args&: (keys[4] | keys[10], keys[13]) = "2/3 1/4 3/2 4/1");
69
70 bayesNet.add(args&: (keys[3] | keys[9], keys[12]) = "1/4 2/3 3/2 4/1");
71 bayesNet.add(args&: (keys[2] | keys[9], keys[12]) = "1/4 8/2 2/3 4/1");
72 bayesNet.add(args&: (keys[1] | keys[8], keys[12]) = "4/1 2/3 3/2 1/4");
73 bayesNet.add(args&: (keys[0] | keys[8], keys[12]) = "2/3 1/4 3/2 4/1");
74
75 // Create a BayesTree out of the Bayes net.
76 bayesTree = DiscreteFactorGraph(bayesNet).eliminateMultifrontal();
77 }
78};

81// Check that BN and BT give the same answer on all configurations
82TEST(DiscreteBayesTree, ThinTree) {
83 TestFixture self;
84
85 if (debug) {
86 GTSAM_PRINT(self.bayesNet);
87 self.bayesNet.saveGraph(filename: "/tmp/discreteBayesNet.dot");
88 }
89
90 // create a BayesTree out of a Bayes net
91 if (debug) {
92 GTSAM_PRINT(*self.bayesTree);
93 self.bayesTree->saveGraph(filename: "/tmp/discreteBayesTree.dot");
94 }
95
96 // Check frontals and parents
97 for (size_t i : {13, 14, 9, 3, 2, 8, 1, 0, 10, 5, 4}) {
98 auto clique_i = (*self.bayesTree)[i];
99 EXPECT_LONGS_EQUAL(i, *(clique_i->conditional_->beginFrontals()));
100 }
101
102 for (const auto& x : self.assignments) {
103 double expected = self.bayesNet.evaluate(values: x);
104 double actual = self.bayesTree->evaluate(values: x);
105 DOUBLES_EQUAL(expected, actual, 1e-9);
106 }
107}

110// Check calculation of separator marginals
111TEST(DiscreteBayesTree, SeparatorMarginals) {
112 TestFixture self;
113
114 // Calculate some marginals for DiscreteValues==all1
115 double marginal_14 = 0, joint_8_12 = 0;
116 for (auto& x : self.assignments) {
117 double px = self.bayesTree->evaluate(values: x);
118 if (x[8] && x[12]) joint_8_12 += px;
119 if (x[14]) marginal_14 += px;
120 }
121 DiscreteValues all1 = self.assignments.back();
122
123 // check separator marginal P(S0)
124 auto clique = (*self.bayesTree)[0];
125 DiscreteFactorGraph separatorMarginal0 =
126 clique->separatorMarginal(function: EliminateDiscrete);
127 DOUBLES_EQUAL(joint_8_12, separatorMarginal0(all1), 1e-9);
128
129 // check separator marginal P(S9), should be P(14)
130 clique = (*self.bayesTree)[9];
131 DiscreteFactorGraph separatorMarginal9 =
132 clique->separatorMarginal(function: EliminateDiscrete);
133 DOUBLES_EQUAL(marginal_14, separatorMarginal9(all1), 1e-9);
134
135 // check separator marginal of root, should be empty
136 clique = (*self.bayesTree)[11];
137 DiscreteFactorGraph separatorMarginal11 =
138 clique->separatorMarginal(function: EliminateDiscrete);
139 LONGS_EQUAL(0, separatorMarginal11.size());
140}

143// Check shortcuts in the tree
144TEST(DiscreteBayesTree, Shortcuts) {
145 TestFixture self;
146
147 // Calculate some marginals for DiscreteValues==all1
148 double joint_11_13 = 0, joint_11_13_14 = 0, joint_11_12_13_14 = 0,
149 joint_9_11_12_13 = 0, joint_8_11_12_13 = 0;
150 for (auto& x : self.assignments) {
151 double px = self.bayesTree->evaluate(values: x);
152 if (x[11] && x[13]) {
153 joint_11_13 += px;
154 if (x[8] && x[12]) joint_8_11_12_13 += px;
155 if (x[9] && x[12]) joint_9_11_12_13 += px;
156 if (x[14]) {
157 joint_11_13_14 += px;
158 if (x[12]) {
159 joint_11_12_13_14 += px;
160 }
161 }
162 }
163 }
164 DiscreteValues all1 = self.assignments.back();
165
166 auto R = self.bayesTree->roots().front();
167
168 // check shortcut P(S9||R) to root
169 auto clique = (*self.bayesTree)[9];
170 DiscreteBayesNet shortcut = clique->shortcut(root: R, function: EliminateDiscrete);
171 LONGS_EQUAL(1, shortcut.size());
172 DOUBLES_EQUAL(joint_11_13_14 / joint_11_13, shortcut.evaluate(all1), 1e-9);
173
174 // check shortcut P(S8||R) to root
175 clique = (*self.bayesTree)[8];
176 shortcut = clique->shortcut(root: R, function: EliminateDiscrete);
177 DOUBLES_EQUAL(joint_11_12_13_14 / joint_11_13, shortcut.evaluate(all1), 1e-9);
178
179 // check shortcut P(S2||R) to root
180 clique = (*self.bayesTree)[2];
181 shortcut = clique->shortcut(root: R, function: EliminateDiscrete);
182 DOUBLES_EQUAL(joint_9_11_12_13 / joint_11_13, shortcut.evaluate(all1), 1e-9);
183
184 // check shortcut P(S0||R) to root
185 clique = (*self.bayesTree)[0];
186 shortcut = clique->shortcut(root: R, function: EliminateDiscrete);
187 DOUBLES_EQUAL(joint_8_11_12_13 / joint_11_13, shortcut.evaluate(all1), 1e-9);
188
189 if (debug) {
190 // print all shortcuts to root
191 DiscreteBayesTree::Nodes cliques = self.bayesTree->nodes();
192 for (auto clique : cliques) {
193 DiscreteBayesNet shortcut = clique.second->shortcut(root: R, function: EliminateDiscrete);
194 clique.second->conditional_->printSignature();
195 shortcut.print(s: "shortcut:");
196 }
197 }

219 DOUBLES_EQUAL(marginals[i], actual, 1e-9);
220 }

224// Check a number of joint marginals.
225TEST(DiscreteBayesTree, Joints) {
226 TestFixture self;
227
228 // Calculate some marginals for DiscreteValues==all1
229 Vector marginals = Vector::Zero(size: 15);
230 double joint_12_14 = 0, joint_9_12_14 = 0, joint_8_12_14 = 0, joint82 = 0,
231 joint12 = 0, joint24 = 0, joint45 = 0, joint46 = 0, joint_4_11 = 0;
232 for (size_t i = 0; i < self.assignments.size(); ++i) {
233 DiscreteValues& x = self.assignments[i];
234 double px = self.bayesTree->evaluate(values: x);
235 for (size_t i = 0; i < 15; i++)
236 if (x[i]) marginals[i] += px;
237 if (x[12] && x[14]) {
238 joint_12_14 += px;
239 if (x[9]) joint_9_12_14 += px;
240 if (x[8]) joint_8_12_14 += px;
241 }
242 if (x[2]) {
243 if (x[8]) joint82 += px;
244 if (x[1]) joint12 += px;
245 }
246 if (x[4]) {
247 if (x[2]) joint24 += px;
248 if (x[5]) joint45 += px;
249 if (x[6]) joint46 += px;
250 if (x[11]) joint_4_11 += px;
251 }
252 }
253
254 // regression tests:
255 DOUBLES_EQUAL(joint_12_14, 0.1875, 1e-9);
256 DOUBLES_EQUAL(joint_8_12_14, 0.0375, 1e-9);
257 DOUBLES_EQUAL(joint_9_12_14, 0.15, 1e-9);
258
259 DiscreteValues all1 = self.assignments.back();
260 DiscreteBayesNet::shared_ptr actualJoint;
261
262 // Check joint P(8, 2)
263 actualJoint = self.bayesTree->jointBayesNet(j1: 8, j2: 2, function: EliminateDiscrete);
264 DOUBLES_EQUAL(joint82, actualJoint->evaluate(all1), 1e-9);
265
266 // Check joint P(1, 2)
267 actualJoint = self.bayesTree->jointBayesNet(j1: 1, j2: 2, function: EliminateDiscrete);
268 DOUBLES_EQUAL(joint12, actualJoint->evaluate(all1), 1e-9);
269
270 // Check joint P(2, 4)
271 actualJoint = self.bayesTree->jointBayesNet(j1: 2, j2: 4, function: EliminateDiscrete);
272 DOUBLES_EQUAL(joint24, actualJoint->evaluate(all1), 1e-9);
273
274 // Check joint P(4, 5)
275 actualJoint = self.bayesTree->jointBayesNet(j1: 4, j2: 5, function: EliminateDiscrete);
276 DOUBLES_EQUAL(joint45, actualJoint->evaluate(all1), 1e-9);
277
278 // Check joint P(4, 6)
279 actualJoint = self.bayesTree->jointBayesNet(j1: 4, j2: 6, function: EliminateDiscrete);
280 DOUBLES_EQUAL(joint46, actualJoint->evaluate(all1), 1e-9);
281
282 // Check joint P(4, 11)
283 actualJoint = self.bayesTree->jointBayesNet(j1: 4, j2: 11, function: EliminateDiscrete);
284 DOUBLES_EQUAL(joint_4_11, actualJoint->evaluate(all1), 1e-9);
285}

320// Check that we can have a multi-frontal lookup table
321TEST(DiscreteBayesTree, Lookup) {
322 using gtsam::symbol_shorthand::A;
323 using gtsam::symbol_shorthand::X;
324
325 // Make a small planning-like graph: 3 states, 2 actions
326 DiscreteFactorGraph graph;
327 const DiscreteKey x1{X(j: 1), 3}, x2{X(j: 2), 3}, x3{X(j: 3), 3};
328 const DiscreteKey a1{A(j: 1), 2}, a2{A(j: 2), 2};
329
330 // Constraint on start and goal
331 graph.add(args: DiscreteKeys{x1}, args: std::vector<double>{1, 0, 0});
332 graph.add(args: DiscreteKeys{x3}, args: std::vector<double>{0, 0, 1});
333
334 // Should I stay or should I go?
335 // "Reward" (exp(-cost)) for an action is 10, and rewards multiply:
336 const double r = 10;
337 std::vector<double> table{
338 r, 0, 0, 0, r, 0, // x1 = 0
339 0, r, 0, 0, 0, r, // x1 = 1
340 0, 0, r, 0, 0, r // x1 = 2
341 };
342 graph.add(args: DiscreteKeys{x1, a1, x2}, args&: table);
343 graph.add(args: DiscreteKeys{x2, a2, x3}, args&: table);
344
345 // eliminate for MPE (maximum probable explanation).
346 Ordering ordering{A(j: 2), X(j: 3), X(j: 1), A(j: 1), X(j: 2)};
347 auto lookup = graph.eliminateMultifrontal(ordering, function: EliminateForMPE);
348
349 // Check that the lookup table is correct
350 EXPECT_LONGS_EQUAL(2, lookup->size());
351 auto lookup_x1_a1_x2 = (*lookup)[X(j: 1)]->conditional();
352 EXPECT_LONGS_EQUAL(3, lookup_x1_a1_x2->frontals().size());
353 // check that sum is 1.0 (not 100, as we now normalize)
354 DiscreteValues empty;
355 EXPECT_DOUBLES_EQUAL(1.0, (*lookup_x1_a1_x2->sum(3))(empty), 1e-9);
356 // And that only non-zero reward is for x1 a1 x2 == 0 1 1
357 EXPECT_DOUBLES_EQUAL(1.0, (*lookup_x1_a1_x2)({{X(1),0},{A(1),1},{X(2),1}}), 1e-9);
358
359 auto lookup_a2_x3 = (*lookup)[X(j: 3)]->conditional();
360 // check that the sum depends on x2 and is non-zero only for x2 \in {1,2}
361 auto sum_x2 = lookup_a2_x3->sum(nrFrontals: 2);
362 EXPECT_DOUBLES_EQUAL(0, (*sum_x2)({{X(2),0}}), 1e-9);
363 EXPECT_DOUBLES_EQUAL(1.0, (*sum_x2)({{X(2),1}}), 1e-9);
364 EXPECT_DOUBLES_EQUAL(2.0, (*sum_x2)({{X(2),2}}), 1e-9);
365 EXPECT_LONGS_EQUAL(2, lookup_a2_x3->frontals().size());
366 // And that the non-zero rewards are for
367 // x2 a2 x3 == 1 1 2
368 EXPECT_DOUBLES_EQUAL(1.0, (*lookup_a2_x3)({{X(2),1},{A(2),1},{X(3),2}}), 1e-9);
369 // x2 a2 x3 == 2 0 2
370 EXPECT_DOUBLES_EQUAL(1.0, (*lookup_a2_x3)({{X(2),2},{A(2),0},{X(3),2}}), 1e-9);
371 // x2 a2 x3 == 2 1 2
372 EXPECT_DOUBLES_EQUAL(1.0, (*lookup_a2_x3)({{X(2),2},{A(2),1},{X(3),2}}), 1e-9);
373}

376// Test creating a Bayes tree directly from cliques
377TEST(DiscreteBayesTree, DirectFromCliques) {
378 // Create a BayesNet
379 DiscreteBayesNet bayesNet;
380 DiscreteKey A(0, 2), B(1, 2), C(2, 2);
381 bayesNet.add(args: A % "1/3");
382 bayesNet.add(args&: B | A = "1/3 3/1");
383 bayesNet.add(args&: C | B = "3/1 3/1");
384
385 // Create cliques directly
386 auto clique2 = std::make_shared<DiscreteBayesTree::Clique>(
387 args: std::make_shared<DiscreteConditional>(args&: C | B = "3/1 3/1"));
388 auto clique1 = std::make_shared<DiscreteBayesTree::Clique>(
389 args: std::make_shared<DiscreteConditional>(args&: B | A = "1/3 3/1"));
390 auto clique0 = std::make_shared<DiscreteBayesTree::Clique>(
391 args: std::make_shared<DiscreteConditional>(args: A % "1/3"));
392
393 // Create a BayesTree
394 DiscreteBayesTree bayesTree;
395 bayesTree.insertRoot(subtree: clique2);
396 bayesTree.addClique(clique: clique1, parent_clique: clique2);
397 bayesTree.addClique(clique: clique0, parent_clique: clique1);
398
399 // Check that the BayesTree is correct
400 DiscreteValues values;
401 values[A.first] = 1;
402 values[B.first] = 1;
403 values[C.first] = 1;
404
405 // Regression
406 double expected = .046875;
407 DOUBLES_EQUAL(expected, bayesTree.evaluate(values), 1e-9);
408}
