37TEST(DiscreteBayesNet, bayesNet) {
38 using ADT = AlgebraicDecisionTree<Key>;
39 DiscreteBayesNet bayesNet;
40 DiscreteKey Parent(0, 2), Child(1, 2);
41
42 auto prior = std::make_shared<DiscreteConditional>(args: Parent % "6/4");
43 CHECK(assert_equal(ADT({Parent}, "0.6 0.4"), (ADT)*prior));
44 bayesNet.push_back(factor: prior);
45
46 auto conditional =
47 std::make_shared<DiscreteConditional>(args&: Child | Parent = "7/3 8/2");
48 EXPECT_LONGS_EQUAL(1, *(conditional->beginFrontals()));
49 ADT expected(Child & Parent, "0.7 0.8 0.3 0.2");
50 CHECK(assert_equal(expected, (ADT)*conditional));
51 bayesNet.push_back(factor: conditional);
52
53 DiscreteFactorGraph fg(bayesNet);
54 LONGS_EQUAL(2, fg.back()->size());
55
56 // Check the marginals
57 const double expectedMarginal[2]{0.4, 0.6 * 0.3 + 0.4 * 0.2};
58 DiscreteMarginals marginals(fg);
59 for (size_t j = 0; j < 2; j++) {
60 Vector FT = marginals.marginalProbabilities(key: DiscreteKey(j, 2));
61 EXPECT_DOUBLES_EQUAL(expectedMarginal[j], FT[1], 1e-3);
62 EXPECT_DOUBLES_EQUAL(FT[0], 1.0 - FT[1], 1e-9);
63 }
64}

67TEST(DiscreteBayesNet, Asia) {
68 using namespace asia_example;
69 const DiscreteBayesNet asia = createAsiaExample();
70
71 // Convert to factor graph
72 DiscreteFactorGraph fg(asia);
73 LONGS_EQUAL(1, fg.back()->size());
74
75 // Check the marginals we know (of the parent-less nodes)
76 DiscreteMarginals marginals(fg);
77 Vector2 va(0.99, 0.01), vs(0.5, 0.5);
78 EXPECT(assert_equal(va, marginals.marginalProbabilities(Asia)));
79 EXPECT(assert_equal(vs, marginals.marginalProbabilities(Smoking)));
80
81 // Create solver and eliminate
82 const Ordering ordering{A, D, T, X, S, E, L, B};
83 DiscreteBayesNet::shared_ptr chordal = fg.eliminateSequential(ordering);
84 DiscreteConditional expected2(Bronchitis % "11/9");
85 EXPECT(assert_equal(expected2, *chordal->back()));
86
87 // Check evaluate and logProbability
88 auto result = fg.optimize();
89 EXPECT_DOUBLES_EQUAL(asia.logProbability(result),
90 std::log(asia.evaluate(result)), 1e-9);
91
92 // add evidence, we were in Asia and we have dyspnea
93 fg.add(args: Asia, args: "0 1");
94 fg.add(args: Dyspnea, args: "0 1");
95
96 // solve again, now with evidence
97 DiscreteBayesNet::shared_ptr chordal2 = fg.eliminateSequential(ordering);
98 EXPECT(assert_equal(expected2, *chordal->back()));
99
100 // now sample from it
101 DiscreteValues expectedSample{{Asia.first, 1}, {Dyspnea.first, 1},
102 {XRay.first, 0}, {Tuberculosis.first, 0},
103 {Smoking.first, 1}, {Either.first, 0},
104 {LungCancer.first, 0}, {Bronchitis.first, 1}};
105 SETDEBUG("DiscreteConditional::sample", false);
106 auto actualSample = chordal2->sample();
107 EXPECT(assert_equal(expectedSample, actualSample));
108}

126// Test that pruning a DiscreteBayesNet results in a conditional whose leaves sum to 1.
127TEST(DiscreteBayesNet, PruneSumToOne) {
128 using namespace asia_example;
129 const DiscreteBayesNet asiaBn = createAsiaExample();
130
131 // Test with various numbers of max leaves.
132 // Asia network has 8 binary variables, so 2^8 = 256 possible assignments in the full joint.
133 std::vector<size_t> maxLeavesToTest = { 1, 2, 5, 10, 50, 100, 256, 500 };
134
135 for (size_t maxLeaves : maxLeavesToTest) {
136 // We expect maxLeaves >= 1 for the sum-to-one property to hold meaningfully.
137
138 DiscreteBayesNet prunedBn = asiaBn.prune(maxNrLeaves: maxLeaves);
139
140 // If the original BN was not empty and maxLeaves >= 1,
141 // the prunedBN should contain one conditional (the pruned joint).
142 EXPECT(prunedBn.size() > 0);
143 if (prunedBn.size() > 0) {
144 EXPECT_LONGS_EQUAL(1, prunedBn.size());
145
146 DiscreteConditional::shared_ptr prunedConditional = prunedBn.front();
147 CHECK(prunedConditional); // Ensure it's not null
148
149 double sumOfProbs = prunedConditional->sum();
150 EXPECT_DOUBLES_EQUAL(1.0, sumOfProbs, 1e-8);
151 }
152 }
153}

180// Check markdown representation looks as expected.
181TEST(DiscreteBayesNet, markdown) {
182 using namespace asia_example;
183 DiscreteBayesNet priors = createPriors();
184
185 std::string expected =
186 "`DiscreteBayesNet` of size 2\n"
187 "\n"
188 " *P(Smoking):*\n\n"
189 "|Smoking|value|\n"
190 "|:-:|:-:|\n"
191 "|0|0.5|\n"
192 "|1|0.5|\n"
193 "\n"
194 " *P(Asia):*\n\n"
195 "|Asia|value|\n"
196 "|:-:|:-:|\n"
197 "|0|0.99|\n"
198 "|1|0.01|\n\n";
199 auto formatter = [](Key key) { return key == A ? "Asia" : "Smoking"; };
200 std::string actual = priors.markdown(keyFormatter: formatter);
201 EXPECT(actual == expected);
202}
