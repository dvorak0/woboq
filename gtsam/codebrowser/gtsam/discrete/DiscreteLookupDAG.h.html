34/**
35 * @brief DiscreteLookupTable table for max-product
36 * @ingroup discrete
37 *
38 * Inherits from discrete conditional for convenience, but is not normalized.
39 * Is used in the max-product algorithm.
40 */
41class GTSAM_EXPORT DiscreteLookupTable : public DiscreteConditional {
42 public:
43 using This = DiscreteLookupTable;
44 using shared_ptr = std::shared_ptr<This>;
45 using BaseConditional = Conditional<DecisionTreeFactor, This>;
46
47 /**
48 * @brief Construct a new Discrete Lookup Table object
49 *
50 * @param nFrontals number of frontal variables
51 * @param keys a sorted list of gtsam::Keys
52 * @param potentials the algebraic decision tree with lookup values
53 */
54 DiscreteLookupTable(size_t nFrontals, const DiscreteKeys& keys,
55 const ADT& potentials)
56 : DiscreteConditional(nFrontals, keys, potentials) {}
57
58 /**
59 * @brief Construct a new Discrete Lookup Table object
60 *
61 * @param nFrontals number of frontal variables
62 * @param keys a sorted list of gtsam::Keys
63 * @param potentials Discrete potentials as a TableFactor.
64 */
65 DiscreteLookupTable(size_t nFrontals, const DiscreteKeys& keys,
66 const TableFactor& potentials)
67 : DiscreteConditional(nFrontals, keys,
68 potentials.toDecisionTreeFactor()) {}
69
70 /// GTSAM-style print
71 void print(
72 const std::string& s = "Discrete Lookup Table: ",
73 const KeyFormatter& formatter = DefaultKeyFormatter) const override;
74
75 /**
76 * @brief return assignment for single frontal variable that maximizes value.
77 * @param parentsValues Known assignments for the parents.
78 * @return maximizing assignment for the frontal variable.
79 */
80 size_t argmax(const DiscreteValues& parentsValues) const;
81
82 /**
83 * @brief Calculate assignment for frontal variables that maximizes value.
84 * @param (in/out) parentsValues Known assignments for the parents.
85 */
86 void argmaxInPlace(DiscreteValues* parentsValues) const;
87};
88
89/** A DAG made from lookup tables, as defined above. */
90class GTSAM_EXPORT DiscreteLookupDAG : public BayesNet<DiscreteLookupTable> {
91 public:
92 using Base = BayesNet<DiscreteLookupTable>;
93 using This = DiscreteLookupDAG;
94 using shared_ptr = std::shared_ptr<This>;
95
96 /// @name Standard Constructors
97 /// @{
98
99 /// Construct empty DAG.
100 DiscreteLookupDAG() {}
101
102 /// Create from BayesNet with LookupTables
103 static DiscreteLookupDAG FromBayesNet(const DiscreteBayesNet& bayesNet);
104
105 /// @}
106
107 /// @name Testable
108 /// @{
109
110 /** Check equality */
111 bool equals(const This& bn, double tol = 1e-9) const;
112
113 /// @}
114
115 /// @name Standard Interface
116 /// @{
117
118 /** Add a DiscreteLookupTable */
119 template <typename... Args>
120 void add(Args&&... args) {
121 emplace_shared<DiscreteLookupTable>(std::forward<Args>(args)...);
122 }
123
124 /**
125 * @brief argmax by back-substitution, optionally given certain variables.
126 *
127 * Assumes the DAG is reverse topologically sorted, i.e. last
128 * conditional will be optimized first *and* that the
129 * DAG does not contain any conditionals for the given variables. If the DAG
130 * resulted from eliminating a factor graph, this is true for the elimination
131 * ordering.
132 *
133 * @return given assignment extended w. optimal assignment for all variables.
134 */
135 DiscreteValues argmax(DiscreteValues given = DiscreteValues()) const;
136 /// @}
137
138 private:
139#if GTSAM_ENABLE_BOOST_SERIALIZATION
140 /** Serialization function */
141 friend class boost::serialization::access;
142 template <class ARCHIVE>
143 void serialize(ARCHIVE& ar, const unsigned int /*version*/) {
144 ar& BOOST_SERIALIZATION_BASE_OBJECT_NVP(Base);
145 }
146#endif
147};
