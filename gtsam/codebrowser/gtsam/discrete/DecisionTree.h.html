40 /**
41 * @brief a decision tree is a function from assignments to values.
42 * @tparam L label for variables
43 * @tparam Y function range (any algebra), e.g., bool, int, double
44 *
45 * After creating a decision tree on some variables, the tree can be evaluated
46 * on an assignment to those variables. Example:
47 *
48 * @code{.cpp}
49 * // Create a decision stump one one variable 'a' with values 10 and 20.
50 * DecisionTree<char, int> tree('a', 10, 20);
51 *
52 * // Evaluate the tree on an assignment to the variable.
53 * int value0 = tree({{'a', 0}}); // value0 = 10
54 * int value1 = tree({{'a', 1}}); // value1 = 20
55 * @endcode
56 *
57 * More examples can be found in testDecisionTree.cpp
58 *
59 * @ingroup discrete
60 */
61 template<typename L, typename Y>
62 class DecisionTree {
63 protected:
64 /// Default method for comparison of two objects of type Y.
65 static bool DefaultCompare(const Y& a, const Y& b) {
66 return a == b;
67 }
68
69 public:
70 using LabelFormatter = std::function<std::string(L)>;
71 using ValueFormatter = std::function<std::string(Y)>;
72 using CompareFunc = std::function<bool(const Y&, const Y&)>;
73
74 /** Handy typedefs for unary and binary function types */
75 using Unary = std::function<Y(const Y&)>;
76 using UnaryAssignment = std::function<Y(const Assignment<L>&, const Y&)>;
77 using Binary = std::function<Y(const Y&, const Y&)>;
78
79 /** A label annotated with cardinality */
80 using LabelC = std::pair<L, size_t>;
81
82 /** DTs consist of Leaf and Choice nodes, both subclasses of Node */
83 struct Leaf;
84 struct Choice;
85
86 /** ------------------------ Node base class --------------------------- */
87 struct Node {
88 using Ptr = std::shared_ptr<Node>;
89
90#ifdef DT_DEBUG_MEMORY
91 static int nrNodes;
92#endif
93
94 // Constructor
95 Node() {
96#ifdef DT_DEBUG_MEMORY
97 std::cout << ++nrNodes << " constructed " << id() << std::endl;
98 std::cout.flush();
99#endif
100 }
101
102 // Destructor
103 virtual ~Node() {
104#ifdef DT_DEBUG_MEMORY
105 std::cout << --nrNodes << " destructed " << id() << std::endl;
106 std::cout.flush();
107#endif
108 }
109
110 // Unique ID for dot files
111 const void* id() const { return this; }
112
113 // everything else is virtual, no documentation here as internal
114 virtual void print(const std::string& s,
115 const LabelFormatter& labelFormatter,
116 const ValueFormatter& valueFormatter) const = 0;
117 virtual void dot(std::ostream& os, const LabelFormatter& labelFormatter,
118 const ValueFormatter& valueFormatter,
119 bool showZero) const = 0;
120 virtual bool sameLeaf(const Leaf& q) const = 0;
121 virtual bool sameLeaf(const Node& q) const = 0;
122 virtual bool equals(const Node& other, const CompareFunc& compare =
123 &DefaultCompare) const = 0;
124 virtual const Y& operator()(const Assignment<L>& x) const = 0;
125 virtual Ptr apply(const Unary& op) const = 0;
126 virtual Ptr apply(const UnaryAssignment& op,
127 const Assignment<L>& assignment) const = 0;
128 virtual Ptr apply_f_op_g(const Node&, const Binary&) const = 0;
129 virtual Ptr apply_g_op_fL(const Leaf&, const Binary&) const = 0;
130 virtual Ptr apply_g_op_fC(const Choice&, const Binary&) const = 0;
131 virtual Ptr choose(const L& label, size_t index) const = 0;
132 virtual bool isLeaf() const = 0;
133
134 private:
135#if GTSAM_ENABLE_BOOST_SERIALIZATION
136 /** Serialization function */
137 friend class boost::serialization::access;
138 template <class ARCHIVE>
139 void serialize(ARCHIVE& ar, const unsigned int /*version*/) {}
140#endif
141 };
142 /** ------------------------ Node base class --------------------------- */
143
144 public:
145 /** A function is a shared pointer to the root of a DT */
146 using NodePtr = typename Node::Ptr;
147
148 /// A DecisionTree just contains the root. TODO(dellaert): make protected.
149 NodePtr root_;
150
151 protected:
152 /**
153 * Internal recursive function to create from keys, cardinalities,
154 * and Y values
155 */
156 template <typename It, typename ValueIt>
157 static NodePtr build(It begin, It end, ValueIt beginY, ValueIt endY);
158
159 /**
160 * Internal helper function to create a tree from keys, cardinalities, and Y
161 * values. Calls `build` which builds the tree bottom-up, before we prune in
162 * a top-down fashion.
163 */
164 template <typename It, typename ValueIt>
165 static NodePtr create(It begin, It end, ValueIt beginY, ValueIt endY);
166
167 /**
168 * @brief Convert from a DecisionTree<L, X> to DecisionTree<L, Y>.
169 *
170 * @tparam M The previous label type.
171 * @tparam X The previous value type.
172 * @param f The node pointer to the root of the previous DecisionTree.
173 * @param Y_of_X Functor to convert from value type X to type Y.
174 * @return NodePtr
175 */
176 template <typename X>
177 static NodePtr convertFrom(const typename DecisionTree<L, X>::NodePtr& f,
178 std::function<Y(const X&)> Y_of_X);
179
180 /**
181 * @brief Convert from a DecisionTree<M, X> to DecisionTree<L, Y>.
182 *
183 * @tparam M The previous label type.
184 * @tparam X The previous value type.
185 * @param f The node pointer to the root of the previous DecisionTree.
186 * @param L_of_M Functor to convert from label type M to type L.
187 * @param Y_of_X Functor to convert from value type X to type Y.
188 * @return NodePtr
189 */
190 template <typename M, typename X>
191 static NodePtr convertFrom(const typename DecisionTree<M, X>::NodePtr& f,
192 std::function<L(const M&)> L_of_M,
193 std::function<Y(const X&)> Y_of_X);

195 public:
196 /// @name Standard Constructors
197 /// @{
198
199 /** Default constructor (for serialization) */
200 DecisionTree();
201
202 /** Create a constant */
203 explicit DecisionTree(const Y& y);
204
205 /**
206 * @brief Create tree with 2 assignments `y1`, `y2`, splitting on variable `label`
207 *
208 * @param label The variable to split on.
209 * @param y1 The value for the first assignment.
210 * @param y2 The value for the second assignment.
211 */
212 DecisionTree(const L& label, const Y& y1, const Y& y2);
213
214 /** Allow Label+Cardinality for convenience */
215 DecisionTree(const LabelC& label, const Y& y1, const Y& y2);
216
217 /** Create from keys and a corresponding vector of values */
218 DecisionTree(const std::vector<LabelC>& labelCs, const std::vector<Y>& ys);
219
220 /** Create from keys and string table */
221 DecisionTree(const std::vector<LabelC>& labelCs, const std::string& table);
222
223 /** Create DecisionTree from others */
224 template<typename Iterator>
225 DecisionTree(Iterator begin, Iterator end, const L& label);
226
227 /** Create DecisionTree from two others */
228 DecisionTree(const L& label, const DecisionTree& f0,
229 const DecisionTree& f1);
230
231 /**
232 * @brief Move constructor for DecisionTree. Very efficient as does not
233 * allocate anything, just changes in-place. But `other` is consumed.
234 *
235 * @param op The unary operation to apply to the moved DecisionTree.
236 * @param other The DecisionTree to move from, will be empty afterwards.
237 */
238 DecisionTree(const Unary& op, DecisionTree&& other) noexcept;
239
240 /**
241 * @brief Convert from a different value type.
242 *
243 * @tparam X The previous value type.
244 * @param other The DecisionTree to convert from.
245 * @param Y_of_X Functor to convert from value type X to type Y.
246 */
247 template <typename X, typename Func>
248 DecisionTree(const DecisionTree<L, X>& other, Func Y_of_X);
249
250 /**
251 * @brief Convert from a different value type X to value type Y, also translate
252 * labels via map from type M to L.
253 *
254 * @tparam M Previous label type.
255 * @tparam X Previous value type.
256 * @param other The decision tree to convert.
257 * @param L_of_M Map from label type M to type L.
258 * @param Y_of_X Functor to convert from type X to type Y.
259 */
260 template <typename M, typename X, typename Func>
261 DecisionTree(const DecisionTree<M, X>& other, const std::map<M, L>& map,
262 Func Y_of_X);

289 /// Check if tree is empty.
290 bool empty() const { return !root_; }

292 /** equality */
293 bool operator==(const DecisionTree& q) const;
294
295 /** evaluate */
296 const Y& operator()(const Assignment<L>& x) const;

298 /**
299 * @brief Visit all leaves in depth-first fashion.
300 *
301 * @param f (side-effect) Function taking the value of the leaf node.
302 *
303 * @note Due to pruning, the number of leaves may not be the same as the
304 * number of assignments. E.g. if we have a tree on 2 binary variables with
305 * all values being 1, then there are 2^2=4 assignments, but only 1 leaf.
306 *
307 * Example:
308 * int sum = 0;
309 * auto visitor = [&](int y) { sum += y; };
310 * tree.visit(visitor);
311 */
312 template <typename Func>
313 void visit(Func f) const;
314
315 /**
316 * @brief Visit all leaves in depth-first fashion.
317 *
318 * @param f (side-effect) Function taking the leaf node pointer.
319 *
320 * @note Due to pruning, the number of leaves may not be the same as the
321 * number of assignments. E.g. if we have a tree on 2 binary variables with
322 * all values being 1, then there are 2^2=4 assignments, but only 1 leaf.
323 *
324 * Example:
325 * int sum = 0;
326 * auto visitor = [&](const Leaf& leaf) { sum += leaf.constant(); };
327 * tree.visitLeaf(visitor);
328 */
329 template <typename Func>
330 void visitLeaf(Func f) const;
331
332 /**
333 * @brief Visit all leaves in depth-first fashion.
334 *
335 * @param f (side-effect) Function taking an assignment and a value.
336 *
337 * @note Due to pruning, the number of leaves may not be the same as the
338 * number of assignments. E.g. if we have a tree on 2 binary variables with
339 * all values being 1, then there are 2^2=4 assignments, but only 1 leaf.
340 *
341 * Example:
342 * int sum = 0;
343 * auto visitor = [&](const Assignment<L>& assignment, int y) { sum += y; };
344 * tree.visitWith(visitor);
345 */
346 template <typename Func>
347 void visitWith(Func f) const;
348
349 /// Return the number of leaves in the tree.
350 size_t nrLeaves() const;
351
352 /**
353 * @brief Fold a binary function over the tree, returning accumulator.
354 *
355 * @tparam X type for accumulator.
356 * @param f binary function: Y * X -> X returning an updated accumulator.
357 * @param x0 initial value for accumulator.
358 * @return X final value for accumulator.
359 *
360 * @note X is always passed by value.
361 * @note Due to pruning, leaves might not exhaust choices.
362 *
363 * Example:
364 * auto add = [](const double& y, double x) { return y + x; };
365 * double sum = tree.fold(add, 0.0);
366 */
367 template <typename Func, typename X>
368 X fold(Func f, X x0) const;
369
370 /** Retrieve all unique labels as a set. */
371 std::set<L> labels() const;
372
373 /** apply Unary operation "op" to f */
374 DecisionTree apply(const Unary& op) const;
375
376 /**
377 * @brief Apply Unary operation "op" to f while also providing the
378 * corresponding assignment.
379 *
380 * @param op Function which takes Assignment<L> and Y as input and returns
381 * object of type Y.
382 * @return DecisionTree
383 */
384 DecisionTree apply(const UnaryAssignment& op) const;
385
386 /** apply binary operation "op" to f and g */
387 DecisionTree apply(const DecisionTree& g, const Binary& op) const;
388
389 /** create a new function where value(label)==index
390 * It's like "restrict" in Darwiche09book pg329, 330? */
391 DecisionTree choose(const L& label, size_t index) const {
392 NodePtr newRoot = root_->choose(label, index);
393 return DecisionTree(newRoot);
394 }
395
396 /** Choose multiple values. */
397 DecisionTree restrict(const Assignment<L>& assignment) const {
398 NodePtr newRoot = root_;
399 for (const auto& [l, v] : assignment) newRoot = newRoot->choose(l, v);
400 return DecisionTree(newRoot);
401 }
402
403 /** combine subtrees on key with binary operation "op" */
404 DecisionTree combine(const L& label, size_t cardinality,
405 const Binary& op) const;
406
407 /** combine with LabelC for convenience */
408 DecisionTree combine(const LabelC& labelC, const Binary& op) const {
409 return combine(labelC.first, labelC.second, op);
410 }

412 /** output to graphviz format, stream version */
413 void dot(std::ostream& os, const LabelFormatter& labelFormatter,
414 const ValueFormatter& valueFormatter, bool showZero = true) const;
415
416 /** output to graphviz format, open a file */
417 void dot(const std::string& name, const LabelFormatter& labelFormatter,
418 const ValueFormatter& valueFormatter, bool showZero = true) const;
419
420 /** output to graphviz format string */
421 std::string dot(const LabelFormatter& labelFormatter,
422 const ValueFormatter& valueFormatter,
423 bool showZero = true) const;
424
425 /**
426 * @brief Convert into two trees with value types A and B.
427 *
428 * @tparam A First new value type.
429 * @tparam B Second new value type.
430 * @param AB_of_Y Functor to convert from type X to std::pair<A, B>.
431 * @return A pair of DecisionTrees with value types A and B respectively.
432 */
433 template <typename A, typename B>
434 std::pair<DecisionTree<L, A>, DecisionTree<L, B>> split(
435 std::function<std::pair<A, B>(const Y&)> AB_of_Y) const;

463 /** free versions of apply */
464
465 /// Apply unary operator `op` to DecisionTree `f`.
466 template<typename L, typename Y>
467 DecisionTree<L, Y> apply(const DecisionTree<L, Y>& f,
468 const typename DecisionTree<L, Y>::Unary& op) {
469 return f.apply(op);
470 }
471
472 /// Apply unary operator `op` with Assignment to DecisionTree `f`.
473 template<typename L, typename Y>
474 DecisionTree<L, Y> apply(const DecisionTree<L, Y>& f,
475 const typename DecisionTree<L, Y>::UnaryAssignment& op) {
476 return f.apply(op);
477 }
478
479 /// Apply binary operator `op` to DecisionTree `f`.
480 template<typename L, typename Y>
481 DecisionTree<L, Y> apply(const DecisionTree<L, Y>& f,
482 const DecisionTree<L, Y>& g,
483 const typename DecisionTree<L, Y>::Binary& op) {
484 return f.apply(g, op);
485 }
486
487 /**
488 * @brief unzip a DecisionTree with `std::pair` values.
489 *
490 * @param input the DecisionTree with `(T1,T2)` values.
491 * @return a pair of DecisionTree on T1 and T2, respectively.
492 */
493 template <typename L, typename T1, typename T2>
494 std::pair<DecisionTree<L, T1>, DecisionTree<L, T2> > unzip(
495 const DecisionTree<L, std::pair<T1, T2> >& input) {
496 return {
497 DecisionTree<L, T1>(input, [](std::pair<T1, T2> i) { return i.first; }),
498 DecisionTree<L, T2>(input, [](std::pair<T1, T2> i) { return i.second; })
499 };
