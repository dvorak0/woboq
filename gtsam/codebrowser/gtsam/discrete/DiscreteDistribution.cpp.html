29double DiscreteDistribution::operator()(size_t value) const {
30 if (nrFrontals() != 1)
31 throw std::invalid_argument(
32 "Single value operator can only be invoked on single-variable "
33 "priors");
34 DiscreteValues values;
35 values.emplace(args: keys_[0], args&: value);
36 return Base::operator()(values);
37}
38
39std::vector<double> DiscreteDistribution::pmf() const {
40 if (nrFrontals() != 1)
41 throw std::invalid_argument(
42 "DiscreteDistribution::pmf only defined for single-variable priors");
43 const size_t nrValues = cardinalities_.at(k: keys_[0]);
44 std::vector<double> array;
45 array.reserve(n: nrValues);
46 for (size_t v = 0; v < nrValues; v++) {
47 array.push_back(x: operator()(value: v));
48 }
49 return array;
50}
