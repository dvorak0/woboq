25namespace gtsam {
26
27/**
28 * A prior probability on a set of discrete variables.
29 * Derives from DiscreteConditional
30 *
31 * @ingroup discrete
32 */
33class GTSAM_EXPORT DiscreteDistribution : public DiscreteConditional {
34 public:
35 using Base = DiscreteConditional;
36
37 /// @name Standard Constructors
38 /// @{
39
40 /// Default constructor needed for serialization.
41 DiscreteDistribution() {}
42
43 /// Constructor from DecisionTreeFactor.
44 explicit DiscreteDistribution(const DecisionTreeFactor& f)
45 : Base(f.size(), f) {}
46
47 /**
48 * Construct from a Signature.
49 *
50 * Example: DiscreteDistribution P(D % "3/2");
51 */
52 explicit DiscreteDistribution(const Signature& s) : Base(s) {}
53
54 /**
55 * Construct from key and a vector of floats specifying the probability mass
56 * function (PMF).
57 *
58 * Example: DiscreteDistribution P(D, {0.4, 0.6});
59 */
60 DiscreteDistribution(const DiscreteKey& key, const std::vector<double>& spec)
61 : DiscreteDistribution(Signature(key, {}, Signature::Table{spec})) {}
62
63 /**
64 * Construct from key and a string specifying the probability mass function
65 * (PMF).
66 *
67 * Example: DiscreteDistribution P(D, "9/1 2/8 3/7 1/9");
68 */
69 DiscreteDistribution(const DiscreteKey& key, const std::string& spec)
70 : DiscreteDistribution(Signature(key, {}, spec)) {}
71
72 /// @}
73 /// @name Testable
74 /// @{
75
76 /// GTSAM-style print
77 void print(
78 const std::string& s = "Discrete Prior: ",
79 const KeyFormatter& formatter = DefaultKeyFormatter) const override;
80
81 /// @}
82 /// @name Standard interface
83 /// @{
84
85 /// Evaluate given a single value.
86 double operator()(size_t value) const;
87
88 /// We also want to keep the Base version, taking DiscreteValues:
89 // TODO(dellaert): does not play well with wrapper!
90 // using Base::operator();
91
92 /// Return entire probability mass function.
93 std::vector<double> pmf() const;
94
95 /// @}
96};
97// DiscreteDistribution
98
99// traits
100template <>
101struct traits<DiscreteDistribution> : public Testable<DiscreteDistribution> {};
102
103} // namespace gtsam
