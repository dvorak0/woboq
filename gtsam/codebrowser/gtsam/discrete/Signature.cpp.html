69vector<double> Signature::cpt() const {
70 vector<double> cpt;
71 if (table_) {
72 const size_t nrStates = table_->at(n: 0).size();
73 for (size_t j = 0; j < nrStates; j++) {
74 for (const Row& row : *table_) {
75 assert(row.size() == nrStates);
76 cpt.push_back(x: row[j]);
77 }
78 }
79 }
80 return cpt;
81}

88static void normalize(Signature::Row& row) {
89 double sum = 0;
90 for (size_t i = 0; i < row.size(); i++) sum += row[i];
91 for (size_t i = 0; i < row.size(); i++) row[i] /= sum;
92}

94Signature& Signature::operator=(const string& spec) {
95 spec_ = spec;
96 auto table = SignatureParser::Parse(str: spec);
97 if (table) {
98 for (Row& row : *table) normalize(row);
99 table_ = *table;
100 }
101 return *this;
102}
103
104Signature& Signature::operator=(const Table& t) {
105 Table table = t;
106 for (Row& row : table) normalize(row);
107 table_ = table;
108 return *this;
109}

111ostream& operator<<(ostream& os, const Signature& s) {
112 os << s.key_.first;
113 if (s.parents_.empty()) {
114 os << " % ";
115 } else {
116 os << " | " << s.parents_[0].first;
117 for (size_t i = 1; i < s.parents_.size(); i++)
118 os << " && " << s.parents_[i].first;
119 os << " = ";
120 }
121 os << (s.spec_ ? *s.spec_ : "no spec") << endl;
122 if (s.table_)
123 os << (*s.table_);
124 else
125 os << "spec could not be parsed" << endl;
126 return os;
127}
