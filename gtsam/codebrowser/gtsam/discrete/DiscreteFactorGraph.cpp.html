53 /* ************************************************************************ */
54 DiscreteKeys DiscreteFactorGraph::discreteKeys() const {
55 DiscreteKeys result;
56 for (auto&& factor : *this) {
57 if (auto p = std::dynamic_pointer_cast<DiscreteFactor>(r: factor)) {
58 DiscreteKeys factor_keys = p->discreteKeys();
59 result.insert(position: result.end(), first: factor_keys.begin(), last: factor_keys.end());
60 }
61 }
62
63 return result;
64 }

66 /* ************************************************************************ */
67 DiscreteFactor::shared_ptr DiscreteFactorGraph::product() const {
68 DiscreteFactor::shared_ptr result = nullptr;
69 for (const auto& factor : *this) {
70 if (factor) {
71 result = result ? result->multiply(df: factor) : factor;
72 }
73 }
74 if (result->max() == 0.0 && result->nrValues() == 1) {
75 throw std::runtime_error(
76 "Product factor is 0.0, possibly due to disjointed discrete "
77 "factors.");
78 }
79 return result;
80 }

82 /* ************************************************************************ */
83 double DiscreteFactorGraph::operator()(const DiscreteValues& values) const {
84 double product = 1.0;
85 for (const sharedFactor& factor : factors_) {
86 if (factor) product *= (*factor)(values);
87 }
88 return product;
89 }

126 /* ************************************************************************ */
127 // Alternate eliminate function for MPE
128 std::pair<DiscreteConditional::shared_ptr, DiscreteFactor::shared_ptr> //
129 EliminateForMPE(const DiscreteFactorGraph& factors,
130 const Ordering& frontalKeys) {
131 DiscreteFactor::shared_ptr product = factors.scaledProduct();
132
133 // max out frontals, this is the factor on the separator
134 gttic(max);
135 DiscreteFactor::shared_ptr max = product->max(keys: frontalKeys);
136 gttoc(max);
137
138 // Ordering keys for the conditional so that frontalKeys are really in front
139 DiscreteKeys orderedKeys;
140 for (auto&& key : frontalKeys)
141 orderedKeys.emplace_back(args: key, args: product->cardinality(j: key));
142 for (auto&& key : max->keys())
143 orderedKeys.emplace_back(args&: key, args: product->cardinality(j: key));
144
145 // Make lookup with product
146 gttic(lookup);
147 size_t nrFrontals = frontalKeys.size();
148 auto lookup = std::make_shared<DiscreteLookupTable>(
149 args&: nrFrontals, args&: orderedKeys, args: product->toDecisionTreeFactor());
150 gttoc(lookup);
151
152 return {std::dynamic_pointer_cast<DiscreteConditional>(r: lookup), max};
153 }

171 /* ************************************************************************ */
172 // The max-product solution below is a bit clunky: the elimination machinery
173 // does not allow for differently *typed* versions of elimination, so we
174 // eliminate into a Bayes Net using the special eliminate function above, and
175 // then create the DiscreteLookupDAG after the fact, in linear time.
176
177 DiscreteLookupDAG DiscreteFactorGraph::maxProduct(
178 OptionalOrderingType orderingType) const {
179 gttic(DiscreteFactorGraph_maxProduct);
180 auto bayesNet = eliminateSequential(orderingType, function: EliminateForMPE);
181 return DiscreteLookupDAG::FromBayesNet(bayesNet: *bayesNet);
182 }

184 DiscreteLookupDAG DiscreteFactorGraph::maxProduct(
185 const Ordering& ordering) const {
186 gttic(DiscreteFactorGraph_maxProduct);
187 auto bayesNet = eliminateSequential(ordering, function: EliminateForMPE);
188 return DiscreteLookupDAG::FromBayesNet(bayesNet: *bayesNet);
189 }

191 /* ************************************************************************ */
192 DiscreteValues DiscreteFactorGraph::optimize(
193 OptionalOrderingType orderingType) const {
194 gttic(DiscreteFactorGraph_optimize);
195 DiscreteLookupDAG dag = maxProduct(orderingType);
196 return dag.argmax();
197 }
198
199 DiscreteValues DiscreteFactorGraph::optimize(const Ordering& ordering) const {
200 gttic(DiscreteFactorGraph_optimize);
201 DiscreteLookupDAG dag = maxProduct(ordering);
202 return dag.argmax();
203 }

205 /* ************************************************************************ */
206 std::pair<DiscreteConditional::shared_ptr, DiscreteFactor::shared_ptr> //
207 EliminateDiscrete(const DiscreteFactorGraph& factors,
208 const Ordering& frontalKeys) {
209 gttic(product);
210 // `product` is scaled later to prevent underflow.
211 DiscreteFactor::shared_ptr product = factors.scaledProduct();
212 gttoc(product);
213
214 // sum out frontals, this is the factor on the separator
215 gttic(sum);
216 DiscreteFactor::shared_ptr sum = product->sum(keys: frontalKeys);
217 gttoc(sum);
218
219 // Ordering keys for the conditional so that frontalKeys are really in front
220 Ordering orderedKeys;
221 orderedKeys.insert(position: orderedKeys.end(), first: frontalKeys.begin(),
222 last: frontalKeys.end());
223 orderedKeys.insert(position: orderedKeys.end(), first: sum->keys().begin(),
224 last: sum->keys().end());
225
226 // now divide product/sum to get conditional
227 gttic(divide);
228 auto conditional = std::make_shared<DiscreteConditional>(
229 args: product->toDecisionTreeFactor(), args: sum->toDecisionTreeFactor(),
230 args&: orderedKeys);
231 gttoc(divide);
232
233 return {conditional, sum};
234 }
