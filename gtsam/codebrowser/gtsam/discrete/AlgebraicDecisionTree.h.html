33 /**
34 * An algebraic decision tree fixes the range of a DecisionTree to double.
35 * Just has some nice constructors and some syntactic sugar.
36 * TODO(dellaert): consider eliminating this class altogether?
37 *
38 * @ingroup discrete
39 */
40 template <typename L>
41 class AlgebraicDecisionTree : public DecisionTree<L, double> {
42 /**
43 * @brief Default method used by `labelFormatter` or `valueFormatter` when
44 * printing.
45 *
46 * @param x The value passed to format.
47 * @return std::string
48 */
49 static std::string DefaultFormatter(const L& x) {
50 std::stringstream ss;
51 ss << x;
52 return ss.str();
53 }
54
55 public:
56 using Base = DecisionTree<L, double>;
57
58 AlgebraicDecisionTree(double leaf = 1.0) : Base(leaf) {}
59
60 /// Constructor which accepts root pointer
61 AlgebraicDecisionTree(const typename Base::NodePtr root) : Base(root) {}
62
63 // Explicitly non-explicit constructor
64 AlgebraicDecisionTree(const Base& add) : Base(add) {}
65
66 /** Create a new leaf function splitting on a variable */
67 AlgebraicDecisionTree(const L& label, double y1, double y2)
68 : Base(label, y1, y2) {}
69
70 /**
71 * @brief Create a new leaf function splitting on a variable
72 *
73 * @param labelC: The label with cardinality 2
74 * @param y1: The value for the first key
75 * @param y2: The value for the second key
76 *
77 * Example:
78 * @code{.cpp}
79 * std::pair<string, size_t> A {"a", 2};
80 * AlgebraicDecisionTree<string> a(A, 0.6, 0.4);
81 * @endcode
82 */
83 AlgebraicDecisionTree(const typename Base::LabelC& labelC, double y1,
84 double y2)
85 : Base(labelC, y1, y2) {}
86
87 /**
88 * @brief Create from keys with cardinalities and a vector table
89 *
90 * @param labelCs: The keys, with cardinalities, given as pairs
91 * @param ys: The vector table
92 *
93 * Example with three keys, A, B, and C, with cardinalities 2, 3, and 2,
94 * respectively, and a vector table of size 12:
95 * @code{.cpp}
96 * DiscreteKey A(0, 2), B(1, 3), C(2, 2);
97 * const vector<double> cpt{
98 * 1.0 / 3, 2.0 / 3, 3.0 / 7, 4.0 / 7, 5.0 / 11, 6.0 / 11, //
99 * 1.0 / 9, 8.0 / 9, 3.0 / 6, 3.0 / 6, 5.0 / 10, 5.0 / 10};
100 * AlgebraicDecisionTree<Key> expected(A & B & C, cpt);
101 * @endcode
102 * The table is given in the following order:
103 * A=0, B=0, C=0
104 * A=0, B=0, C=1
105 * ...
106 * A=1, B=1, C=1
107 * Hence, the first line in the table is for A==0, and the second for A==1.
108 * In each line, the first two entries are for B==0, the next two for B==1,
109 * and the last two for B==2. Each pair is for a C value of 0 and 1.
110 */
111 AlgebraicDecisionTree //
112 (const std::vector<typename Base::LabelC>& labelCs,
113 const std::vector<double>& ys) {
114 this->root_ =
115 Base::create(labelCs.begin(), labelCs.end(), ys.begin(), ys.end());
116 }
117
118 /**
119 * @brief Create from keys and string table
120 *
121 * @param labelCs: The keys, with cardinalities, given as pairs
122 * @param table: The string table, given as a string of doubles.
123 *
124 * @note Table needs to be in same order as the vector table in the other constructor.
125 */
126 AlgebraicDecisionTree //
127 (const std::vector<typename Base::LabelC>& labelCs,
128 const std::string& table) {
129 // Convert string to doubles
130 std::vector<double> ys;
131 std::istringstream iss(table);
132 std::copy(std::istream_iterator<double>(iss),
133 std::istream_iterator<double>(), std::back_inserter(x&: ys));
134
135 // now call recursive Create
136 this->root_ =
137 Base::create(labelCs.begin(), labelCs.end(), ys.begin(), ys.end());
138 }
139
140 /**
141 * @brief Create a range of decision trees, splitting on a single variable.
142 *
143 * @param begin: Iterator to beginning of a range of decision trees
144 * @param end: Iterator to end of a range of decision trees
145 * @param label: The label to split on
146 */
147 template <typename Iterator>
148 AlgebraicDecisionTree(Iterator begin, Iterator end, const L& label)
149 : Base(nullptr) {
150 this->root_ = compose(begin, end, label);
151 }
152
153 /**
154 * Convert labels from type M to type L.
155 *
156 * @param other: The AlgebraicDecisionTree with label type M to convert.
157 * @param map: Map from label type M to label type L.
158 */
159 template <typename M>
160 AlgebraicDecisionTree(const AlgebraicDecisionTree<M>& other,
161 const std::map<M, L>& map) {
162 // Functor for label conversion so we can use `convertFrom`.
163 std::function<L(const M&)> L_of_M = [&map](const M& label) -> L {
164 return map.at(label);
165 };
166 std::function<double(const double&)> op = Ring::id;
167 this->root_ = DecisionTree<L, double>::convertFrom(other.root_, L_of_M, op);
168 }
169
170 /**
171 * @brief Create from an arbitrary DecisionTree<L, X> by operating on it
172 * with a functional `f`.
173 *
174 * @tparam X The type of the leaf of the original DecisionTree
175 * @tparam Func Type signature of functional `f`.
176 * @param other The original DecisionTree from which the
177 * AlgbraicDecisionTree is constructed.
178 * @param f Functional used to operate on
179 * the leaves of the input DecisionTree.
180 */
181 template <typename X, typename Func>
182 AlgebraicDecisionTree(const DecisionTree<L, X>& other, Func f)
183 : Base(other, f) {}
184
185 /** sum */
186 AlgebraicDecisionTree operator+(const AlgebraicDecisionTree& g) const {
187 return this->apply(g, &Ring::add);
188 }
189
190 /** negation */
191 AlgebraicDecisionTree operator-() const {
192 return this->apply(&Ring::negate);
193 }
194
195 /** subtract */
196 AlgebraicDecisionTree operator-(const AlgebraicDecisionTree& g) const {
197 return *this + (-g);
198 }
199
200 /** product */
201 AlgebraicDecisionTree operator*(const AlgebraicDecisionTree& g) const {
202 return this->apply(g, &Ring::mul);
203 }
204
205 /** division */
206 AlgebraicDecisionTree operator/(const AlgebraicDecisionTree& g) const {
207 return this->apply(g, &Ring::div);
208 }
209
210 /// Compute sum of all values
211 double sum() const {
212 double sum = 0;
213 auto visitor = [&](double y) { sum += y; };
214 this->visit(visitor);
215 return sum;
216 }
217
218 /**
219 * @brief Helper method to perform normalization such that all leaves in the
220 * tree sum to 1
221 *
222 * @return AlgebraicDecisionTree
223 */
224 AlgebraicDecisionTree normalize() const { return (*this) / this->sum(); }
225
226 /// Find the minimum values amongst all leaves
227 double min() const {
228 double min = std::numeric_limits<double>::max();
229 auto visitor = [&](double x) { min = x < min ? x : min; };
230 this->visit(visitor);
231 return min;
232 }
233
234 /// Find the maximum values amongst all leaves
235 double max() const {
236 // Get the most negative value
237 double max = -std::numeric_limits<double>::max();
238 auto visitor = [&](double x) { max = x > max ? x : max; };
239 this->visit(visitor);
240 return max;
241 }
242
243 /** sum out variable */
244 AlgebraicDecisionTree sum(const L& label, size_t cardinality) const {
245 return this->combine(label, cardinality, &Ring::add);
246 }
247
248 /** sum out variable */
249 AlgebraicDecisionTree sum(const typename Base::LabelC& labelC) const {
250 return this->combine(labelC, &Ring::add);
251 }
252
253 /// print method customized to value type `double`.
254 void print(const std::string& s = "",
255 const typename Base::LabelFormatter& labelFormatter =
256 &DefaultFormatter) const {
257 auto valueFormatter = [](const double& v) {
258 std::stringstream ss;
259 ss << std::setw(4) << std::setprecision(8) << v;
260 return ss.str();
261 };
262 Base::print(s, labelFormatter, valueFormatter);
263 }
264
265 /// Equality method customized to value type `double`.
266 bool equals(const AlgebraicDecisionTree& other, double tol = 1e-9) const {
267 // lambda for comparison of two doubles upto some tolerance.
268 auto compare = [tol](double a, double b) {
269 return std::abs(x: a - b) < tol;
270 };
271 return Base::equals(other, compare);
272 }
273 };

276struct traits<AlgebraicDecisionTree<T>>
277 : public Testable<AlgebraicDecisionTree<T>> {};
