43/**
44 * @brief Main elimination function for DiscreteFactorGraph.
45 *
46 * @param factors The factor graph to eliminate.
47 * @param frontalKeys An ordering for which variables to eliminate.
48 * @return A pair of the resulting conditional and the separator factor.
49 * @ingroup discrete
50 */
51GTSAM_EXPORT
52std::pair<DiscreteConditional::shared_ptr, DiscreteFactor::shared_ptr>
53EliminateDiscrete(const DiscreteFactorGraph& factors,
54 const Ordering& frontalKeys);
55
56/**
57 * @brief Alternate elimination function for that creates non-normalized lookup tables.
58 *
59 * @param factors The factor graph to eliminate.

62 * @ingroup discrete

64GTSAM_EXPORT

69template<> struct EliminationTraits<DiscreteFactorGraph>
70{
71 typedef DiscreteFactor FactorType; ///< Type of factors in factor graph
72 typedef DiscreteFactorGraph FactorGraphType; ///< Type of the factor graph (e.g. DiscreteFactorGraph)
73 typedef DiscreteConditional ConditionalType; ///< Type of conditionals from elimination
74 typedef DiscreteBayesNet BayesNetType; ///< Type of Bayes net from sequential elimination
75 typedef DiscreteEliminationTree EliminationTreeType; ///< Type of elimination tree
76 typedef DiscreteBayesTree BayesTreeType; ///< Type of Bayes tree
77 typedef DiscreteJunctionTree JunctionTreeType; ///< Type of Junction tree
78
79 /// The default dense elimination function
80 static std::pair<std::shared_ptr<ConditionalType>,
81 std::shared_ptr<FactorType> >
82 DefaultEliminate(const FactorGraphType& factors, const Ordering& keys) {
83 return EliminateDiscrete(factors, frontalKeys: keys);
84 }
85
86 /// The default ordering generation function
87 static Ordering DefaultOrderingFunc(
88 const FactorGraphType& graph,
89 std::optional<std::reference_wrapper<const VariableIndex>> variableIndex) {
90 return Ordering::Colamd(variableIndex: (*variableIndex).get());
91 }
92};

94/**
95 * A Discrete Factor Graph is a factor graph where all factors are Discrete, i.e.
96 * Factor == DiscreteFactor
97 * @ingroup discrete
98 */
99class GTSAM_EXPORT DiscreteFactorGraph
100 : public FactorGraph<DiscreteFactor>,
101 public EliminateableFactorGraph<DiscreteFactorGraph> {
102 public:
103 using This = DiscreteFactorGraph; ///< this class
104 using Base = FactorGraph<DiscreteFactor>; ///< base factor graph type
105 using BaseEliminateable =
106 EliminateableFactorGraph<This>; ///< for elimination
107 using shared_ptr = std::shared_ptr<This>; ///< shared_ptr to This
108
109 using Values = DiscreteValues; ///< backwards compatibility
110
111 using Indices = KeyVector; ///> map from keys to values
112
113 /** Default constructor */
114 DiscreteFactorGraph() {}
115
116 /** Construct from iterator over factors */
117 template <typename ITERATOR>
118 DiscreteFactorGraph(ITERATOR firstFactor, ITERATOR lastFactor)
119 : Base(firstFactor, lastFactor) {}
120
121 /** Construct from container of factors (shared_ptr or plain objects) */
122 template <class CONTAINER>
123 explicit DiscreteFactorGraph(const CONTAINER& factors) : Base(factors) {}
124
125 /** Implicit copy/downcast constructor to override explicit template container
126 * constructor */
127 template <class DERIVED_FACTOR>
128 DiscreteFactorGraph(const FactorGraph<DERIVED_FACTOR>& graph) : Base(graph) {}
129
130 /// Destructor
131 virtual ~DiscreteFactorGraph() {}
132
133 /// @name Testable
134 /// @{
135
136 bool equals(const This& fg, double tol = 1e-9) const;
137
138 /// @}
139
140 //TODO(Varun): Make compatible with TableFactor
141 /** Add a decision-tree factor */
142 template <typename... Args>
143 void add(Args&&... args) {
144 emplace_shared<DecisionTreeFactor>(std::forward<Args>(args)...);
145 }
146
147 /** Return the set of variables involved in the factors (set union) */
148 KeySet keys() const;
149
150 /// Return the DiscreteKeys in this factor graph.
151 DiscreteKeys discreteKeys() const;
152
153 /** return product of all factors as a single factor */
154 DiscreteFactor::shared_ptr product() const;
155
156 /**
157 * @brief Return product of all `factors` as a single factor,
158 * which is scaled by the max value to prevent underflow
159 *
160 * @param factors The factors to multiply as a DiscreteFactorGraph.
161 * @return DiscreteFactor::shared_ptr
162 */
163 DiscreteFactor::shared_ptr scaledProduct() const;
164
165 /**
166 * Evaluates the factor graph given values, returns the joint probability of
167 * the factor graph given specific instantiation of values
168 */
169 double operator()(const DiscreteValues& values) const;
170
171 /// print
172 void print(
173 const std::string& s = "DiscreteFactorGraph",
174 const KeyFormatter& formatter = DefaultKeyFormatter) const override;
175
176 /**
177 * @brief Implement the sum-product algorithm
178 *
179 * @param orderingType : one of COLAMD, METIS, NATURAL, CUSTOM
180 * @return DiscreteBayesNet encoding posterior P(X|Z)
181 */
182 DiscreteBayesNet sumProduct(
183 OptionalOrderingType orderingType = {}) const;
184
185 /**
186 * @brief Implement the sum-product algorithm
187 *
188 * @param ordering
189 * @return DiscreteBayesNet encoding posterior P(X|Z)
190 */
191 DiscreteBayesNet sumProduct(const Ordering& ordering) const;
192
193 /**
194 * @brief Implement the max-product algorithm
195 *
196 * @param orderingType : one of COLAMD, METIS, NATURAL, CUSTOM
197 * @return DiscreteLookupDAG DAG with lookup tables
198 */
199 DiscreteLookupDAG maxProduct(
200 OptionalOrderingType orderingType = {}) const;
201
202 /**
203 * @brief Implement the max-product algorithm
204 *
205 * @param ordering
206 * @return DiscreteLookupDAG `DAG with lookup tables
207 */
208 DiscreteLookupDAG maxProduct(const Ordering& ordering) const;
209
210 /**
211 * @brief Find the maximum probable explanation (MPE) by doing max-product.
212 *
213 * @param orderingType
214 * @return DiscreteValues : MPE
215 */
216 DiscreteValues optimize(
217 OptionalOrderingType orderingType = {}) const;
218
219 /**
220 * @brief Find the maximum probable explanation (MPE) by doing max-product.
221 *
222 * @param ordering
223 * @return DiscreteValues : MPE
224 */
225 DiscreteValues optimize(const Ordering& ordering) const;
226
227 /// @name Wrapper support
228 /// @{
229
230 /**
231 * @brief Render as markdown tables
232 *
233 * @param keyFormatter GTSAM-style Key formatter.
234 * @param names optional, a map from Key to category names.
235 * @return std::string a (potentially long) markdown string.
236 */
237 std::string markdown(const KeyFormatter& keyFormatter = DefaultKeyFormatter,
238 const DiscreteFactor::Names& names = {}) const;
239
240 /**
241 * @brief Render as html tables
242 *
243 * @param keyFormatter GTSAM-style Key formatter.
244 * @param names optional, a map from Key to category names.
245 * @return std::string a (potentially long) html string.
246 */
247 std::string html(const KeyFormatter& keyFormatter = DefaultKeyFormatter,
248 const DiscreteFactor::Names& names = {}) const;
249
250 /// @}
251 /// @name HybridValues methods.
252 /// @{
253
254 using Base::error; // Expose error(const HybridValues&) method..
255
256 /// @}
257}; // \ DiscreteFactorGraph
