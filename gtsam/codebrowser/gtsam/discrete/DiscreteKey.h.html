31namespace gtsam {

33 /**
34 * Key type for discrete variables.
35 * Includes Key and cardinality.
36 * @ingroup discrete
37 */
38 using DiscreteKey = std::pair<Key,size_t>;
39
40 /// DiscreteKeys is a set of keys that can be assembled using the & operator
41 struct GTSAM_EXPORT DiscreteKeys: public std::vector<DiscreteKey> {
42
43 // Forward all constructors.
44 using std::vector<DiscreteKey>::vector;
45
46 /// Constructor for serialization
47 DiscreteKeys() : std::vector<DiscreteKey>::vector() {}
48
49 /// Construct from a key
50 explicit DiscreteKeys(const DiscreteKey& key) { push_back(x: key); }
51
52 /// Construct from cardinalities.
53 explicit DiscreteKeys(std::map<Key, size_t> cardinalities) {
54 for (auto&& kv : cardinalities) emplace_back(args&: kv);
55 }
56
57 /// Construct from a vector of keys
58 DiscreteKeys(const std::vector<DiscreteKey>& keys) :
59 std::vector<DiscreteKey>(keys) {
60 }
61
62 /// Construct from cardinalities with default names
63 DiscreteKeys(const std::vector<int>& cs);
64
65 /// Return a vector of indices
66 KeyVector indices() const;
67
68 /// Return a map from index to cardinality
69 std::map<Key,size_t> cardinalities() const;
70
71 /// Add a key (non-const!)
72 DiscreteKeys& operator&(const DiscreteKey& key) {
73 push_back(x: key);
74 return *this;
75 }
76
77 /// Add multiple keys (non-const!)
78 DiscreteKeys& operator&(const DiscreteKeys& keys) {
79 this->insert(position: this->end(), first: keys.begin(), last: keys.end());
80 return *this;
81 }
82
83 /// Print the keys and cardinalities.
84 void print(const std::string& s = "",
85 const KeyFormatter& keyFormatter = DefaultKeyFormatter) const;
86
87 /// Check equality to another DiscreteKeys object.
88 bool equals(const DiscreteKeys& other, double tol = 0) const;
89
90#if GTSAM_ENABLE_BOOST_SERIALIZATION
91 /** Serialization function */
92 friend class boost::serialization::access;
93 template <class ARCHIVE>
94 void serialize(ARCHIVE& ar, const unsigned int /*version*/) {
95 ar& boost::serialization::make_nvp(
96 n: "DiscreteKeys",
97 v&: boost::serialization::base_object<std::vector<DiscreteKey>>(d&: *this));
98 }
99#endif
100
101 }; // DiscreteKeys
102
103 /// Create a list from two keys
104 GTSAM_EXPORT DiscreteKeys operator&(const DiscreteKey& key1, const DiscreteKey& key2);
105
106 // traits
107 template <>
108 struct traits<DiscreteKeys> : public Testable<DiscreteKeys> {};
109
