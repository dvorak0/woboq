66 /* ************************************************************************ */
67 DiscreteFactor::shared_ptr DecisionTreeFactor::multiply(
68 const DiscreteFactor::shared_ptr& f) const {
69 DiscreteFactor::shared_ptr result;
70 if (auto tf = std::dynamic_pointer_cast<TableFactor>(r: f)) {
71 // If f is a TableFactor, we convert `this` to a TableFactor since this
72 // conversion is cheaper than converting `f` to a DecisionTreeFactor. We
73 // then return a TableFactor.
74 result = std::make_shared<TableFactor>(args: (*tf) * TableFactor(*this));
75
76 } else if (auto dtf = std::dynamic_pointer_cast<DecisionTreeFactor>(r: f)) {
77 // If `f` is a DecisionTreeFactor, simply call operator*.
78 result = std::make_shared<DecisionTreeFactor>(args: this->operator*(f: *dtf));
79
80 } else {
81 // Simulate double dispatch in C++
82 // Useful for other classes which inherit from DiscreteFactor and have
83 // only `operator*(DecisionTreeFactor)` defined. Thus, other classes don't
84 // need to be updated.
85 result = std::make_shared<DecisionTreeFactor>(args: f->operator*(*this));
86 }
87 return result;
88 }

90 /* ************************************************************************ */
91 DiscreteFactor::shared_ptr DecisionTreeFactor::operator/(
92 const DiscreteFactor::shared_ptr& f) const {
93 if (auto tf = std::dynamic_pointer_cast<TableFactor>(r: f)) {
94 // Check if `f` is a TableFactor. If yes, then
95 // convert `this` to a TableFactor which is cheaper.
96 return std::make_shared<TableFactor>(args: tf->operator/(f: TableFactor(*this)));
97
98 } else if (auto dtf = std::dynamic_pointer_cast<DecisionTreeFactor>(r: f)) {
99 // If `f` is a DecisionTreeFactor, divide normally.
100 return std::make_shared<DecisionTreeFactor>(args: this->operator/(f: *dtf));
101
102 } else {
103 // Else, convert `f` to a DecisionTreeFactor so we can divide
104 return std::make_shared<DecisionTreeFactor>(
105 args: this->operator/(f: f->toDecisionTreeFactor()));
106 }
107 }

110 double DecisionTreeFactor::safe_div(const double& a, const double& b) {
111 // The use for safe_div is when we divide the product factor by the sum
112 // factor. If the product or sum is zero, we accord zero probability to the
113 // event.
114 return (a == 0 || b == 0) ? 0 : (a / b);
115 }

145 /* ************************************************************************ */
146 DecisionTreeFactor DecisionTreeFactor::apply(const DecisionTreeFactor& f,
147 Binary op) const {
148 map<Key, size_t> cs; // new cardinalities
149 // make unique key-cardinality map
150 for (Key j : keys()) cs[j] = cardinality(j);
151 for (Key j : f.keys()) cs[j] = f.cardinality(j);
152 // Convert map into keys
153 DiscreteKeys keys;
154 keys.reserve(n: cs.size());
155 for (const auto& key : cs) {
156 keys.emplace_back(args: key);
157 }
158 // apply operand
159 ADT result = ADT::apply(g: f, op);
160 // Make a new factor
161 return DecisionTreeFactor(keys, result);
162 }

164 /* ************************************************************************ */
165 DecisionTreeFactor::shared_ptr DecisionTreeFactor::combine(size_t nrFrontals,
166 Binary op) const {
167 if (nrFrontals > size()) {
168 throw invalid_argument(
169 "DecisionTreeFactor::combine: invalid number of frontal "
170 "keys " +
171 std::to_string(val: nrFrontals) + ", nr.keys=" + std::to_string(val: size()));
172 }
173
174 // sum over nrFrontals keys
175 size_t i;
176 ADT result(*this);
177 for (i = 0; i < nrFrontals; i++) {
178 Key j = keys_[i];
179 result = result.combine(label: j, cardinality: cardinality(j), op);
180 }
181
182 // Create new factor, note we start with keys after nrFrontals:
183 DiscreteKeys dkeys;
184 for (; i < keys_.size(); i++) {
185 Key j = keys_[i];
186 dkeys.push_back(x: DiscreteKey(j, cardinality(j)));
187 }
188 return std::make_shared<DecisionTreeFactor>(args&: dkeys, args&: result);
189 }

191 /* ************************************************************************ */
192 DecisionTreeFactor::shared_ptr DecisionTreeFactor::combine(
193 const Ordering& frontalKeys, Binary op) const {
194 if (frontalKeys.size() > size()) {
195 throw invalid_argument(
196 "DecisionTreeFactor::combine: invalid number of frontal "
197 "keys " +
198 std::to_string(val: frontalKeys.size()) + ", nr.keys=" +
199 std::to_string(val: size()));
200 }
201
202 // sum over nrFrontals keys
203 size_t i;
204 ADT result(*this);
205 for (i = 0; i < frontalKeys.size(); i++) {
206 Key j = frontalKeys[i];
207 result = result.combine(label: j, cardinality: cardinality(j), op);
208 }
209
210 /*
211 Create new factor, note we collect keys that are not in frontalKeys.
212
213 Due to branch merging, the labels in `result` may be missing some keys.
214 E.g. After branch merging, we may get a ADT like:
215 Leaf [2] 1.0204082
216
217 Hence, code below traverses the original keys and omits those in
218 frontalKeys. We loop over cardinalities, which is O(n) even for a map, and
219 then "contains" is a binary search on a small vector.
220 */
221 DiscreteKeys dkeys;
222 for (auto&& [key, cardinality] : cardinalities_) {
223 if (!frontalKeys.contains(key)) {
224 dkeys.push_back(x: DiscreteKey(key, cardinality));
225 }
226 }
227 return std::make_shared<DecisionTreeFactor>(args&: dkeys, args&: result);
228 }

231 std::vector<std::pair<DiscreteValues, double>> DecisionTreeFactor::enumerate()
232 const {
233 // Get all possible assignments
234 DiscreteKeys pairs = discreteKeys();
235 // Reverse to make cartesian product output a more natural ordering.
236 DiscreteKeys rpairs(pairs.rbegin(), pairs.rend());
237 const auto assignments = DiscreteValues::CartesianProduct(keys: rpairs);
238
239 // Construct unordered_map with values
240 std::vector<std::pair<DiscreteValues, double>> result;
241 for (const auto& assignment : assignments) {
242 result.emplace_back(args: assignment, args: evaluate(values: assignment));
243 }
244 return result;
245 }

247 /* ************************************************************************ */
248 std::vector<double> DecisionTreeFactor::probabilities() const {
249 // Set of all keys
250 KeySet allKeys(keys().begin(), keys().end());
251
252 std::vector<double> probs;
253
254 /* An operation that takes each leaf probability, and computes the
255 * nrAssignments by checking the difference between the keys in the factor
256 * and the keys in the assignment.
257 * The nrAssignments is then used to append
258 * the correct number of leaf probability values to the `probs` vector
259 * defined above.
260 */
261 auto op = [&](const Assignment<Key>& a, double p) {
262 // Get all the keys in the current assignment
263 KeySet assignment_keys;
264 for (auto&& [k, _] : a) {
265 assignment_keys.insert(x: k);
266 }
267
268 // Find the keys missing in the assignment
269 std::vector<Key> diff;
270 std::set_difference(first1: allKeys.begin(), last1: allKeys.end(),
271 first2: assignment_keys.begin(), last2: assignment_keys.end(),
272 result: std::back_inserter(x&: diff));
273
274 // Compute the total number of assignments in the (pruned) subtree
275 size_t nrAssignments = 1;
276 for (auto&& k : diff) {
277 nrAssignments *= cardinalities_.at(k: k);
278 }
279 // Add p `nrAssignments` times to the probs vector.
280 probs.insert(position: probs.end(), n: nrAssignments, x: p);
281
282 return p;
283 };
284
285 // Go through the tree
286 this->visitWith(f: op);
287
288 return probs;
289 }

396 /**
397 * @brief Min-Heap class to help with pruning.
398 * The `top` element is always the smallest value.
399 */
400 class MinHeap {
401 std::vector<double> v_;
402
403 public:
404 /// Default constructor
405 MinHeap() {}
406
407 /// Push value onto the heap
408 void push(double x) {
409 v_.push_back(x: x);
410 std::push_heap(first: v_.begin(), last: v_.end(), comp: std::greater<double>{});
411 }
412
413 /// Push value `x`, `n` number of times.
414 void push(double x, size_t n) {
415 for (size_t i = 0; i < n; ++i) {
416 v_.push_back(x: x);
417 std::push_heap(first: v_.begin(), last: v_.end(), comp: std::greater<double>{});
418 }
419 }
420
421 /// Pop the top value of the heap.
422 double pop() {
423 std::pop_heap(first: v_.begin(), last: v_.end(), comp: std::greater<double>{});
424 double x = v_.back();
425 v_.pop_back();
426 return x;
427 }
428
429 /// Return the top value of the heap without popping it.
430 double top() { return v_.at(n: 0); }
431
432 /**
433 * @brief Print the heap as a sequence.
434 *
435 * @param s A string to prologue the output.
436 */
437 void print(const std::string& s = "") {
438 std::cout << (s.empty() ? "" : s + " ");
439 for (size_t i = 0; i < v_.size(); i++) {
440 std::cout << v_.at(n: i);
441 if (v_.size() > 1 && i < v_.size() - 1) std::cout << ", ";
442 }
443 std::cout << std::endl;
444 }
445
446 /// Return true if heap is empty.
447 bool empty() const { return v_.empty(); }
448
449 /// Return the size of the heap.
450 size_t size() const { return v_.size(); }
451 };

453 /* ************************************************************************ */
454 double DecisionTreeFactor::computeThreshold(const size_t N) const {
455 // Set of all keys
456 std::set<Key> allKeys = this->labels();
457 MinHeap min_heap;
458
459 auto op = [&](const Assignment<Key>& a, double p) {
460 // Get all the keys in the current assignment
461 KeySet assignment_keys;
462 for (auto&& [k, _] : a) {
463 assignment_keys.insert(x: k);
464 }
465
466 // Find the keys missing in the assignment
467 KeyVector diff;
468 std::set_difference(first1: allKeys.begin(), last1: allKeys.end(),
469 first2: assignment_keys.begin(), last2: assignment_keys.end(),
470 result: std::back_inserter(x&: diff));
471
472 // Compute the total number of assignments in the (pruned) subtree
473 size_t nrAssignments = 1;
474 for (auto&& k : diff) {
475 nrAssignments *= cardinalities_.at(k: k);
476 }
477
478 // If min-heap is empty, fill it initially.
479 // This is because there is nothing at the top.
480 if (min_heap.empty()) {
481 min_heap.push(x: p, n: std::min(a: nrAssignments, b: N));
482
483 } else {
484 for (size_t i = 0; i < std::min(a: nrAssignments, b: N); ++i) {
485 // If p is larger than the smallest element,
486 // then we insert into the min heap.
487 // We check against the top each time because the
488 // heap maintains the smallest element at the top.
489 if (p > min_heap.top()) {
490 if (min_heap.size() == N) {
491 min_heap.pop();
492 }
493 min_heap.push(x: p);
494 } else {
495 // p is <= min value so move to the next one
496 break;
497 }
498 }
499 }
500 return p;
501 };
502 this->visitWith(f: op);
503
504 // If total number of hypotheses is less than N, return 0.0
505 if (min_heap.size() < N) {
506 return 0.0;
507 }
508 return min_heap.top();
509 }

511 /* ************************************************************************ */
512 DecisionTreeFactor DecisionTreeFactor::prune(size_t maxNrAssignments) const {
513 const size_t N = maxNrAssignments;
514
515 double threshold = computeThreshold(N);
516
517 // Now threshold the decision tree
518 size_t total = 0;
519 auto thresholdFunc = [threshold, &total, N](const double& value) {
520 // There is a possible case where the `threshold` is equal to 0.0
521 // In that case `(value < threshold) == false`
522 // which increases the leaf total erroneously.
523 // Hence we check for 0.0 explicitly.
524 if (fpEqual(a: value, b: 0.0, tol: 1e-12)) {
525 return 0.0;
526 }
527
528 // Check if value is less than the threshold and
529 // we haven't exceeded the maximum number of leaves.
530 // TODO(Varun): Bug since we can have a case where we need to prune higher
531 // probabilities after we have reached N.
532 // E.g. N=3 for [0.2, 0.2, 0.1, 0.2, 0.3]
533 // will give [0.2, 0.2, 0.0, 0.2, 0.0]
534 if (value < threshold || total >= N) {
535 return 0.0;
536 } else {
537 total += 1;
538 return value;
539 }
540 };
541 DecisionTree<Key, double> thresholded(*this, thresholdFunc);
542
543 // Create pruned decision tree factor and return.
544 return DecisionTreeFactor(this->discreteKeys(), thresholded);
545 }

548DiscreteFactor::shared_ptr DecisionTreeFactor::restrict(
549 const DiscreteValues& assignment) const {
550 ADT restricted_tree = ADT::restrict(assignment);
551 // Get all the keys that are not restricted by the assignment
552 // This ensures that the new restricted factor doesn't have keys
553 // for which the information has been removed.
554 DiscreteKeys restricted_keys = this->discreteKeys();
555 for (auto&& kv : assignment) {
556 Key key = kv.first;
557 // Remove the key from the keys list
558 restricted_keys.erase(
559 first: std::remove_if(first: restricted_keys.begin(), last: restricted_keys.end(),
560 pred: [key](const DiscreteKey& k) { return k.first == key; }),
561 last: restricted_keys.end());
562 }
563 // Create the restricted factor with the appropriate keys and tree.
564 return std::make_shared<DecisionTreeFactor>(args&: restricted_keys, args&: restricted_tree);
565}
