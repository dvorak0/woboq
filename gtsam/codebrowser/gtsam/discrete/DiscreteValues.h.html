28namespace gtsam {

30/**
31 * A map from keys to values
32 * @ingroup discrete
33 */
34class GTSAM_EXPORT DiscreteValues : public Assignment<Key> {
35 public:
36 using Base = Assignment<Key>; // base class
37
38 /// @name Standard Constructors
39 /// @{
40 using Assignment::Assignment; // all constructors
41
42 // Define the implicit default constructor.
43 DiscreteValues() = default;
44
45 // Construct from assignment.
46 explicit DiscreteValues(const Base& a) : Base(a) {}
47
48 // Construct from initializer list.
49 DiscreteValues(std::initializer_list<std::pair<const Key, size_t>> init)
50 : Assignment<Key>{init} {}
51
52 /// @}
53 /// @name Testable
54 /// @{
55
56 /// print required by Testable.
57 void print(const std::string& s = "",
58 const KeyFormatter& keyFormatter = DefaultKeyFormatter) const;
59
60 /// equals required by Testable for unit testing.
61 bool equals(const DiscreteValues& x, double tol = 1e-9) const;
62
63 /// @}
64 /// @name Standard Interface
65 /// @{
66
67 /// ostream operator:
68 friend std::ostream& operator<<(std::ostream& os, const DiscreteValues& x);
69
70 // insert in base class;
71 std::pair<iterator, bool> insert(const value_type& value) {
72 return Base::insert(x: value);
73 }
74
75 /**
76 * @brief Insert key-assignment pair.
77 *
78 * @param assignment The key-assignment pair to insert.
79 * @return DiscreteValues& Reference to the updated DiscreteValues object.
80 * @throws std::invalid_argument if any keys to be inserted are already used.
81 */
82 DiscreteValues& insert(const std::pair<Key, size_t>& assignment);
83
84 /**
85 * @brief Insert all values from another DiscreteValues object.
86 *
87 * @param values The DiscreteValues object containing values to insert.
88 * @return DiscreteValues& Reference to the updated DiscreteValues object.
89 * @throws std::invalid_argument if any keys to be inserted are already used.
90 */
91 DiscreteValues& insert(const DiscreteValues& values);
92
93 /**
94 * @brief Update values with corresponding keys from another DiscreteValues
95 * object.
96 *
97 * @param values The DiscreteValues object containing values to update.
98 * @return DiscreteValues& Reference to the updated DiscreteValues object.
99 * @throws std::out_of_range if any keys in values are not present in this
100 * object.
101 */
102 DiscreteValues& update(const DiscreteValues& values);
103
104 /**
105 * @brief Insert each element of `values` if it does not already exist,
106 * else assign it.
107 *
108 * @param values DiscreteValues to insert or assign to this.
109 * @return DiscreteValues&
110 */
111 DiscreteValues& insert_or_assign(const DiscreteValues& values);
112
113 /**
114 * @brief Check if the DiscreteValues contains the given key.
115 *
116 * @param key The key to check for.
117 * @return True if the key is present, false otherwise.
118 */
119 bool contains(Key key) const { return this->find(x: key) != this->end(); }
120
121 /**
122 * @brief Filter values by keys.
123 *
124 * @param keys The keys to filter by.
125 * @return DiscreteValues The filtered DiscreteValues object.
126 */
127 DiscreteValues filter(const DiscreteKeys& keys) const {
128 DiscreteValues result;
129 for (const auto& [key, _] : keys) {
130 if (auto it = this->find(x: key); it != this->end())
131 result[key] = it->second;
132 }
133 return result;
134 }
135
136 /**
137 * @brief Return the keys that are not present in the DiscreteValues object.
138 *
139 * @param keys The keys to check for.
140 * @return DiscreteKeys Keys not present in the DiscreteValues object.
141 */
142 DiscreteKeys missingKeys(const DiscreteKeys& keys) const {
143 DiscreteKeys result;
144 for (const auto& [key, cardinality] : keys) {
145 if (!this->contains(key)) result.emplace_back(args: key, args: cardinality);
146 }
147 return result;
148 }
149
150 /**
151 * @brief Return a vector of DiscreteValues, one for each possible combination
152 * of values.
153 *
154 * @param keys The keys to generate the Cartesian product for.
155 * @return std::vector<DiscreteValues> The vector of DiscreteValues.
156 */
157 static std::vector<DiscreteValues> CartesianProduct(
158 const DiscreteKeys& keys) {
159 return Base::CartesianProduct<DiscreteValues>(keys);
160 }
161
162 /// @}
163 /// @name Wrapper support
164 /// @{
165
166 /// Translation table from values to strings.
167 using Names = std::map<Key, std::vector<std::string>>;
168
169 /// Translate an integer index value for given key to a string.
170 static std::string Translate(const Names& names, Key key, size_t index);
171
172 /**
173 * @brief Output as a markdown table.
174 *
175 * @param keyFormatter function that formats keys.
176 * @param names translation table for values.
177 * @return string markdown output.
178 */
179 std::string markdown(const KeyFormatter& keyFormatter = DefaultKeyFormatter,
180 const Names& names = {}) const;
181
182 /**
183 * @brief Output as a html table.
184 *
185 * @param keyFormatter function that formats keys.
186 * @param names translation table for values.
187 * @return string html output.
188 */
189 std::string html(const KeyFormatter& keyFormatter = DefaultKeyFormatter,
190 const Names& names = {}) const;
191
192 /// @}
193};
194
195/// Free version of CartesianProduct.
196inline std::vector<DiscreteValues> cartesianProduct(const DiscreteKeys& keys) {
197 return DiscreteValues::CartesianProduct(keys);
198}
