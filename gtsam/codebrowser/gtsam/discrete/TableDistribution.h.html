28namespace gtsam {
29
30/**
31 * Distribution which uses a SparseVector as the internal
32 * representation, similar to the TableFactor.
33 *
34 * This is primarily used in the case when we have a clique in the BayesTree
35 * which consists of all the discrete variables, e.g. in hybrid elimination.
36 *
37 * @ingroup discrete
38 */
39class GTSAM_EXPORT TableDistribution : public DiscreteConditional {
40 private:
41 TableFactor table_;
42
43 typedef Eigen::SparseVector<double>::InnerIterator SparseIt;
44
45 public:
46 // typedefs needed to play nice with gtsam
47 typedef TableDistribution This; ///< Typedef to this class
48 typedef std::shared_ptr<This> shared_ptr; ///< shared_ptr to this class
49 typedef DiscreteConditional
50 BaseConditional; ///< Typedef to our conditional base class
51
52 using Values = DiscreteValues; ///< backwards compatibility
53
54 /// @name Standard Constructors
55 /// @{
56
57 /// Default constructor needed for serialization.
58 TableDistribution() {}
59
60 /// Construct from TableFactor.
61 TableDistribution(const TableFactor& f);
62
63 /**
64 * Construct from DiscreteKeys and std::vector.
65 */
66 TableDistribution(const DiscreteKeys& keys,
67 const std::vector<double>& potentials);
68
69 /**
70 * Construct from single DiscreteKey and std::vector.
71 */
72 TableDistribution(const DiscreteKey& key,
73 const std::vector<double>& potentials)
74 : TableDistribution(DiscreteKeys(key), potentials) {}
75
76 /**
77 * Construct from DiscreteKey and std::string.
78 */
79 TableDistribution(const DiscreteKeys& keys, const std::string& potentials);
80
81 /**
82 * Construct from single DiscreteKey and std::string.
83 */
84 TableDistribution(const DiscreteKey& key, const std::string& potentials)
85 : TableDistribution(DiscreteKeys(key), potentials) {}
86
87 /// @}
88 /// @name Testable
89 /// @{
90
91 /// GTSAM-style print
92 void print(
93 const std::string& s = "Table Distribution: ",
94 const KeyFormatter& formatter = DefaultKeyFormatter) const override;
95
96 /// GTSAM-style equals
97 bool equals(const DiscreteFactor& other, double tol = 1e-9) const override;
98
99 /// @}
100 /// @name Standard Interface
101 /// @{
102
103 /// Return the underlying TableFactor
104 TableFactor table() const { return table_; }
105
106 using BaseConditional::evaluate; // HybridValues version
107
108 /// Evaluate the conditional given the values.
109 virtual double evaluate(const Assignment<Key>& values) const override {
110 return table_.evaluate(values);
111 }
112
113 /// Create new factor by summing all values with the same separator values
114 DiscreteFactor::shared_ptr sum(size_t nrFrontals) const override;
115
116 /// Create new factor by summing all values with the same separator values
117 DiscreteFactor::shared_ptr sum(const Ordering& keys) const override;
118
119 /// Find the maximum value in the factor.
120 double max() const override { return table_.max(); }
121
122 /// Create new factor by maximizing over all values with the same separator.
123 DiscreteFactor::shared_ptr max(size_t nrFrontals) const override;
124
125 /// Create new factor by maximizing over all values with the same separator.
126 DiscreteFactor::shared_ptr max(const Ordering& keys) const override;
127
128 /// Multiply by scalar s
129 DiscreteFactor::shared_ptr operator*(double s) const override;
130
131 /// divide by DiscreteFactor::shared_ptr f (safely)
132 DiscreteFactor::shared_ptr operator/(
133 const DiscreteFactor::shared_ptr& f) const override;
134
135 /**
136 * @brief Return assignment that maximizes value.
137 *
138 * @return maximizing assignment for the variables.
139 */
140 DiscreteValues argmax() const;
141
142 /**
143 * sample
144 * @param parentsValues Known values of the parents
145 * @param rng Pseudo random number generator
146 * @return sample from conditional
147 */
148 virtual size_t sample(const DiscreteValues& parentsValues,
149 std::mt19937_64* rng = nullptr) const override;
150
151 /// @}
152 /// @name Advanced Interface
153 /// @{
154
155 /// Prune the conditional
156 virtual void prune(size_t maxNrAssignments) override;
157
158 /// Get a DecisionTreeFactor representation.
159 DecisionTreeFactor toDecisionTreeFactor() const override {
160 return table_.toDecisionTreeFactor();
161 }
162
163 /// Get the number of non-zero values.
164 uint64_t nrValues() const override { return table_.sparseTable().nonZeros(); }
165
166 /// @}
167
168 private:
169#if GTSAM_ENABLE_BOOST_SERIALIZATION
170 /** Serialization function */
171 friend class boost::serialization::access;
172 template <class Archive>
173 void serialize(Archive& ar, const unsigned int /*version*/) {
174 ar& BOOST_SERIALIZATION_BASE_OBJECT_NVP(BaseConditional);
175 ar& BOOST_SERIALIZATION_NVP(table_);
176 }
177#endif
178};
179// TableDistribution
180
181// traits
182template <>
183struct traits<TableDistribution> : public Testable<TableDistribution> {};
184
185} // namespace gtsam
