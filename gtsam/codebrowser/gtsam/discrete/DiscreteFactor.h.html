21#include <gtsam/base/Testable.h>
22#include <gtsam/discrete/AlgebraicDecisionTree.h>

27#include <string>
28namespace gtsam {
29
30class DecisionTreeFactor;
31class DiscreteConditional;
32class HybridValues;
33
34/**
35 * Base class for discrete probabilistic factors
36 * The most general one is the derived DecisionTreeFactor
37 *
38 * @ingroup discrete
39 */
40class GTSAM_EXPORT DiscreteFactor : public Factor {
41 public:
42 // typedefs needed to play nice with gtsam
43 typedef DiscreteFactor This; ///< This class
44 typedef std::shared_ptr<DiscreteFactor>
45 shared_ptr; ///< shared_ptr to this class
46 typedef Factor Base; ///< Our base class
47
48 using Values = DiscreteValues; ///< backwards compatibility
49
50 using Unary = std::function<double(const double&)>;
51 using UnaryAssignment =
52 std::function<double(const Assignment<Key>&, const double&)>;
53 using Binary = std::function<double(const double, const double)>;
54
55 protected:
56 /// Map of Keys and their cardinalities.
57 std::map<Key, size_t> cardinalities_;
58
59 public:
60 /// @name Standard Constructors
61 /// @{
62
63 /** Default constructor creates empty factor */
64 DiscreteFactor() {}
65
66 /**
67 * Construct from container of keys and map of cardinalities.
68 * This constructor is used internally from derived factor constructors,
69 * either from a container of keys or from a boost::assign::list_of.
70 */
71 template <typename CONTAINER>
72 DiscreteFactor(const CONTAINER& keys,
73 const std::map<Key, size_t> cardinalities = {})
74 : Base(keys), cardinalities_(cardinalities) {}
75
76 /// @}
77 /// @name Testable
78 /// @{
79
80 /// equals
81 virtual bool equals(const DiscreteFactor& lf, double tol = 1e-9) const;
82
83 /// print
84 void print(
85 const std::string& s = "DiscreteFactor\n",
86 const KeyFormatter& formatter = DefaultKeyFormatter) const override {
87 Base::print(s, formatter);
88 }
89
90 /// @}
91 /// @name Standard Interface
92 /// @{
93
94 /// Return all the discrete keys associated with this factor.
95 DiscreteKeys discreteKeys() const;
96
97 std::map<Key, size_t> cardinalities() const { return cardinalities_; }
98
99 size_t cardinality(Key j) const { return cardinalities_.at(k: j); }
100
101 /**
102 * @brief Calculate probability for given values.
103 * Calls specialized evaluation under the hood.
104 *
105 * Note: Uses Assignment<Key> as it is the base class of DiscreteValues.
106 *
107 * @param values Discrete assignment.
108 * @return double
109 */
110 virtual double evaluate(const Assignment<Key>& values) const = 0;
111
112 /// Find value for given assignment of values to variables
113 double operator()(const DiscreteValues& values) const {
114 return evaluate(values);
115 }
116
117 /// Error is just -log(value)
118 virtual double error(const DiscreteValues& values) const;
119
120 /**
121 * The Factor::error simply extracts the \class DiscreteValues from the
122 * \class HybridValues and calculates the error.
123 */
124 double error(const HybridValues& c) const override;
125
126 /// Compute error for each assignment and return as a tree
127 virtual AlgebraicDecisionTree<Key> errorTree() const;
128
129 /// Multiply with a scalar
130 virtual DiscreteFactor::shared_ptr operator*(double s) const = 0;
131
132 /// Multiply in a DecisionTreeFactor and return the result as
133 /// DecisionTreeFactor
134 virtual DecisionTreeFactor operator*(const DecisionTreeFactor&) const = 0;
135
136 /**
137 * @brief Multiply in a DiscreteFactor and return the result as
138 * DiscreteFactor, both via shared pointers.
139 *
140 * @param df DiscreteFactor shared_ptr
141 * @return DiscreteFactor::shared_ptr
142 */
143 virtual DiscreteFactor::shared_ptr multiply(
144 const DiscreteFactor::shared_ptr& df) const = 0;
145
146 /// divide by DiscreteFactor::shared_ptr f (safely)
147 virtual DiscreteFactor::shared_ptr operator/(
148 const DiscreteFactor::shared_ptr& df) const = 0;
149
150 virtual DecisionTreeFactor toDecisionTreeFactor() const = 0;
151
152 /// Create new factor by summing all values with the same separator values
153 virtual DiscreteFactor::shared_ptr sum(size_t nrFrontals) const = 0;
154
155 /// Create new factor by summing all values with the same separator values
156 virtual DiscreteFactor::shared_ptr sum(const Ordering& keys) const = 0;
157
158 /// Find the maximum value in the factor.
159 virtual double max() const = 0;
160
161 /// Create new factor by maximizing over all values with the same separator.
162 virtual DiscreteFactor::shared_ptr max(size_t nrFrontals) const = 0;
163
164 /// Create new factor by maximizing over all values with the same separator.
165 virtual DiscreteFactor::shared_ptr max(const Ordering& keys) const = 0;
166
167 /**
168 * @brief Scale the factor values by the maximum
169 * to prevent underflow/overflow.
170 *
171 * @return DiscreteFactor::shared_ptr
172 */
173 DiscreteFactor::shared_ptr scale() const;
174
175 /**
176 * Get the number of non-zero values contained in this factor.
177 * It could be much smaller than `prod_{key}(cardinality(key))`.
178 */
179 virtual uint64_t nrValues() const = 0;
180
181 /// Restrict the factor to the given assignment.
182 virtual DiscreteFactor::shared_ptr restrict(
183 const DiscreteValues& assignment) const = 0;
184
185 /// @}
186 /// @name Wrapper support
187 /// @{
188
189 /// Translation table from values to strings.
190 using Names = DiscreteValues::Names;
191
192 /**
193 * @brief Render as markdown table
194 *
195 * @param keyFormatter GTSAM-style Key formatter.
196 * @param names optional, category names corresponding to choices.
197 * @return std::string a markdown string.
198 */
199 virtual std::string markdown(
200 const KeyFormatter& keyFormatter = DefaultKeyFormatter,
201 const Names& names = {}) const = 0;
202
203 /**
204 * @brief Render as html table
205 *
206 * @param keyFormatter GTSAM-style Key formatter.
207 * @param names optional, category names corresponding to choices.
208 * @return std::string a html string.
209 */
210 virtual std::string html(
211 const KeyFormatter& keyFormatter = DefaultKeyFormatter,
212 const Names& names = {}) const = 0;
213
214 /// @}
215
216 private:
217#if GTSAM_ENABLE_BOOST_SERIALIZATION
218 /** Serialization function */
219 friend class boost::serialization::access;
220 template <class ARCHIVE>
221 void serialize(ARCHIVE& ar, const unsigned int /*version*/) {
222 ar& BOOST_SERIALIZATION_BASE_OBJECT_NVP(Base);
223 ar& BOOST_SERIALIZATION_NVP(cardinalities_);
224 }
225#endif
226};
227// DiscreteFactor
228
229// traits
230template <>
231struct traits<DiscreteFactor> : public Testable<DiscreteFactor> {};
