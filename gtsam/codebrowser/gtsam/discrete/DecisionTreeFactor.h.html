40 /**
41 * A discrete probabilistic factor.
42 *
43 * @ingroup discrete
44 */
45 class GTSAM_EXPORT DecisionTreeFactor : public DiscreteFactor,

47 public:

49 typedef DecisionTreeFactor This;
50 typedef DiscreteFactor Base; ///< Typedef to base class
51 typedef std::shared_ptr<DecisionTreeFactor> shared_ptr;
52 typedef AlgebraicDecisionTree<Key> ADT;

54 // Needed since we have definitions in both DiscreteFactor and DecisionTree
55 using Base::Binary;
56 using Base::Unary;

61
62 /** Default constructor for I/O */

65 /** Constructor from DiscreteKeys and AlgebraicDecisionTree */
66 DecisionTreeFactor(const DiscreteKeys& keys, const ADT& potentials);

68 /**
69 * @brief Constructor from doubles
70 *
71 * @param keys The discrete keys.
72 * @param table The table of values.
73 *
74 * @throw std::invalid_argument if the size of `table` does not match the
75 * number of assignments.
76 *
77 * Example:
78 * @code{.cpp}
79 * DiscreteKey X(0,2), Y(1,3);
80 * const std::vector<double> table {2, 5, 3, 6, 4, 7};
81 * DecisionTreeFactor f1({X, Y}, table);
82 * @endcode
83 *
84 * The values in the table should be laid out so that the first key varies
85 * the slowest, and the last key the fastest.
86 */
87 DecisionTreeFactor(const DiscreteKeys& keys,
88 const std::vector<double>& table);
89
90 /**
91 * @brief Constructor from string
92 *
93 * @param keys The discrete keys.
94 * @param table The table of values.
95 *
96 * @throw std::invalid_argument if the size of `table` does not match the
97 * number of assignments.
98 *
99 * Example:
100 * @code{.cpp}
101 * DiscreteKey X(0,2), Y(1,3);
102 * DecisionTreeFactor factor({X, Y}, "2 5 3 6 4 7");
103 * @endcode
104 *
105 * The values in the table should be laid out so that the first key varies
106 * the slowest, and the last key the fastest.
107 */
108 DecisionTreeFactor(const DiscreteKeys& keys, const std::string& table);

138 /// Calculate probability for given values,
139 /// is just look up in AlgebraicDecisionTree.
140 virtual double evaluate(const Assignment<Key>& values) const override {
141 return ADT::operator()(x: values);
142 }

147 /// Calculate error for DiscreteValues `x`, is -log(probability).
148 double error(const DiscreteValues& values) const override;

150 /**
151 * @brief Multiply factors, DiscreteFactor::shared_ptr edition.
152 *
153 * This method accepts `DiscreteFactor::shared_ptr` and uses dynamic
154 * dispatch and specializations to perform the most efficient
155 * multiplication.
156 *
157 * While converting a DecisionTreeFactor to a TableFactor is efficient, the
158 * reverse is not. Hence we specialize the code to return a TableFactor if
159 * `f` is a TableFactor, and DecisionTreeFactor otherwise.
160 *
161 * @param f The factor to multiply with.
162 * @return DiscreteFactor::shared_ptr
163 */
164 virtual DiscreteFactor::shared_ptr multiply(
165 const DiscreteFactor::shared_ptr& f) const override;
166
167 /// multiply with a scalar
168 DiscreteFactor::shared_ptr operator*(double s) const override {
169 return std::make_shared<DecisionTreeFactor>(
170 args: apply(op: [s](const double& a) { return Ring::mul(a, b: s); }));
171 }
172
173 /// multiply two factors
174 DecisionTreeFactor operator*(const DecisionTreeFactor& f) const override {
175 return apply(f, op: Ring::mul);
176 }
177
178 static double safe_div(const double& a, const double& b);
179
180 /**
181 * @brief Divide by factor f (safely).
182 * Division of a factor \f$f(x, y)\f$ by another factor \f$g(y, z)\f$
183 * results in a function which involves all keys
184 * \f$(\frac{f}{g})(x, y, z) = f(x, y) / g(y, z)\f$
185 *
186 * @param f The DecisinTreeFactor to divide by.
187 * @return DecisionTreeFactor
188 */
189 DecisionTreeFactor operator/(const DecisionTreeFactor& f) const {
190 return apply(f, op: safe_div);
191 }
192
193 /// divide by DiscreteFactor::shared_ptr f (safely)
194 DiscreteFactor::shared_ptr operator/(
195 const DiscreteFactor::shared_ptr& f) const override;
196
197 /// Convert into a decision tree
198 DecisionTreeFactor toDecisionTreeFactor() const override { return *this; }
199
200 /// Use sum() from AlgebraicDecisionTree
201 using ADT::sum;
202
203 /// Create new factor by summing all values with the same separator values
204 DiscreteFactor::shared_ptr sum(size_t nrFrontals) const override {
205 return combine(nrFrontals, op: Ring::add);
206 }
207
208 /// Create new factor by summing all values with the same separator values
209 DiscreteFactor::shared_ptr sum(const Ordering& keys) const override {
210 return combine(keys, op: Ring::add);
211 }
212
213 /// Find the maximum value in the factor.
214 double max() const override { return ADT::max(); };
215
216 /// Create new factor by maximizing over all values with the same separator.
217 DiscreteFactor::shared_ptr max(size_t nrFrontals) const override {
218 return combine(nrFrontals, op: Ring::max);
219 }
220
221 /// Create new factor by maximizing over all values with the same separator.
222 DiscreteFactor::shared_ptr max(const Ordering& keys) const override {
223 return combine(keys, op: Ring::max);
224 }

276 /**
277 * @brief Compute the probability value which is the threshold above which
278 * only `N` leaves are present.
279 *
280 * This is used for pruning out the smaller probabilities.
281 *
282 * @param N The number of leaves to keep post pruning.
283 * @return double
284 */
285 double computeThreshold(const size_t N) const;
286
287 /**
288 * @brief Prune the decision tree of discrete variables.
289 *
290 * Pruning will set the leaves to be "pruned" to 0 indicating a 0
291 * probability. An assignment is pruned if it is not in the top
292 * `maxNrAssignments` values.
293 *
294 * A violation can occur if there are more
295 * duplicate values than `maxNrAssignments`. A violation here is the need to
296 * un-prune the decision tree (e.g. all assignment values are 1.0). We could
297 * have another case where some subset of duplicates exist (e.g. for a tree
298 * with 8 assignments we have 1, 1, 1, 1, 0.8, 0.7, 0.6, 0.5), but this is
299 * not a violation since the for `maxNrAssignments=5` the top values are (1,
300 * 0.8).
301 *
302 * @param maxNrAssignments The maximum number of assignments to keep.
303 * @return DecisionTreeFactor
304 */
305 DecisionTreeFactor prune(size_t maxNrAssignments) const;
306
307 /**
308 * Get the number of non-zero values contained in this factor.
309 * It could be much smaller than `prod_{key}(cardinality(key))`.
310 */
311 uint64_t nrValues() const override { return nrLeaves(); }
