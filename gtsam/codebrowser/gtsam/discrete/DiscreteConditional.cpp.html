78/* ************************************************************************** */
79DiscreteConditional DiscreteConditional::operator*(
80 const DiscreteConditional& other) const {
81 // If the root is a nullptr, we have a TableDistribution
82 // TODO(Varun) Revisit this hack after RSS2025 submission
83 if (!other.root_) {
84 DiscreteConditional dc(other.nrFrontals(), other.toDecisionTreeFactor());
85 return dc * (*this);
86 }
87
88 // Take union of frontal keys
89 std::set<Key> newFrontals;
90 for (auto&& key : this->frontals()) newFrontals.insert(x: key);
91 for (auto&& key : other.frontals()) newFrontals.insert(x: key);
92
93 // Check if frontals overlapped
94 if (nrFrontals() + other.nrFrontals() > newFrontals.size())
95 throw std::invalid_argument(
96 "DiscreteConditional::operator* called with overlapping frontal keys.");
97
98 // Now, add cardinalities.
99 DiscreteKeys discreteKeys;
100 for (auto&& key : frontals())
101 discreteKeys.emplace_back(args: key, args: cardinality(j: key));
102 for (auto&& key : other.frontals())
103 discreteKeys.emplace_back(args: key, args: other.cardinality(j: key));
104
105 // Sort
106 std::sort(first: discreteKeys.begin(), last: discreteKeys.end());
107
108 // Add parents to set, to make them unique
109 std::set<DiscreteKey> parents;
110 for (auto&& key : this->parents())
111 if (!newFrontals.count(x: key)) parents.emplace(args: key, args: cardinality(j: key));
112 for (auto&& key : other.parents())
113 if (!newFrontals.count(x: key)) parents.emplace(args: key, args: other.cardinality(j: key));
114
115 // Finally, add parents to keys, in order
116 for (auto&& dk : parents) discreteKeys.push_back(x: dk);
117
118 ADT product = ADT::apply(g: other, op: Ring::mul);
119 return DiscreteConditional(newFrontals.size(), discreteKeys, product);
120}

122/* ************************************************************************** */
123DiscreteConditional DiscreteConditional::marginal(Key key) const {
124 if (nrParents() > 0)
125 throw std::invalid_argument(
126 "DiscreteConditional::marginal: single argument version only valid for "
127 "fully specified joint distributions (i.e., no parents).");
128
129 // Calculate the keys as the frontal keys without the given key.
130 DiscreteKeys discreteKeys{{key, cardinality(j: key)}};
131
132 // Calculate sum
133 ADT adt(*this);
134 for (auto&& k : frontals())
135 if (k != key) adt = adt.sum(label: k, cardinality: cardinality(j: k));
136
137 // Return new factor
138 return DiscreteConditional(1, discreteKeys, adt);
139}

170/* ************************************************************************** */
171DiscreteConditional::ADT DiscreteConditional::choose(
172 const DiscreteValues& given, bool forceComplete) const {
173 // Get the big decision tree with all the levels, and then go down the
174 // branches based on the value of the parent variables.
175 DiscreteConditional::ADT adt(*this);
176 size_t value;
177 for (Key j : parents()) {
178 try {
179 value = given.at(k: j);
180 adt = adt.choose(label: j, index: value); // ADT keeps getting smaller.
181 } catch (std::out_of_range&) {
182 if (forceComplete) {
183 given.print(s: "parentsValues: ");
184 throw runtime_error(
185 "DiscreteConditional::choose: parent value missing");
186 }
187 }
188 }
189 return adt;
190}

192/* ************************************************************************** */
193DiscreteConditional::shared_ptr DiscreteConditional::choose(
194 const DiscreteValues& given) const {
195 ADT adt = choose(given, forceComplete: false); // P(F|S=given)
196
197 // Collect all keys not in given.
198 DiscreteKeys dKeys;
199 for (Key j : frontals()) {
200 dKeys.emplace_back(args&: j, args: this->cardinality(j));
201 }
202 for (size_t i = nrFrontals(); i < size(); i++) {
203 Key j = keys_[i];
204 if (given.count(x: j) == 0) {
205 dKeys.emplace_back(args&: j, args: this->cardinality(j));
206 }
207 }
208 return std::make_shared<DiscreteConditional>(args: nrFrontals(), args&: dKeys, args&: adt);
209}

211/* ************************************************************************** */
212DecisionTreeFactor::shared_ptr DiscreteConditional::likelihood(
213 const DiscreteValues& frontalValues) const {
214 // Get the big decision tree with all the levels, and then go down the
215 // branches based on the value of the frontal variables.
216 ADT adt(*this);
217 size_t value;
218 for (Key j : frontals()) {
219 try {
220 value = frontalValues.at(k: j);
221 adt = adt.choose(label: j, index: value); // ADT keeps getting smaller.
222 } catch (exception&) {
223 frontalValues.print(s: "frontalValues: ");
224 throw runtime_error("DiscreteConditional::choose: frontal value missing");
225 }
226 }
227
228 // Convert ADT to factor.
229 DiscreteKeys discreteKeys;
230 for (Key j : parents()) {
231 discreteKeys.emplace_back(args&: j, args: this->cardinality(j));
232 }
233 return std::make_shared<DecisionTreeFactor>(args&: discreteKeys, args&: adt);
234}

248/* ************************************************************************** */
249size_t DiscreteConditional::argmax(const DiscreteValues& parentsValues) const {
250 ADT pFS = choose(given: parentsValues, forceComplete: true); // P(F|S=parentsValues)
251
252 // Initialize
253 size_t maxValue = 0;
254 double maxP = 0;
255 DiscreteValues values = parentsValues;
256
257 assert(nrFrontals() == 1);
258 Key j = firstFrontalKey();
259 for (size_t value = 0; value < cardinality(j); value++) {
260 values[j] = value;
261 double pValueS = (*this)(values);
262 // Update MPE solution if better
263 if (pValueS > maxP) {
264 maxP = pValueS;
265 maxValue = value;
266 }
267 }
268 return maxValue;
269}

271/* ************************************************************************** */
272void DiscreteConditional::sampleInPlace(DiscreteValues* values,
273 std::mt19937_64* rng) const {
274 // throw if more than one frontal:
275 if (nrFrontals() != 1) {
276 throw std::invalid_argument(
277 "DiscreteConditional::sampleInPlace can only be called on single "
278 "variable conditionals");
279 }
280 Key j = firstFrontalKey();
281 // throw if values already contains j:
282 if (values->count(x: j) > 0) {
283 throw std::invalid_argument(
284 "DiscreteConditional::sampleInPlace: values already contains j");
285 }
286 size_t sampled = sample(parentsValues: *values, rng); // Sample variable given parents
287 (*values)[j] = sampled; // store result in partial solution
288}

290/* ************************************************************************** */
291size_t DiscreteConditional::sample(const DiscreteValues& parentsValues,
292 std::mt19937_64* rng) const {
293 // Get the correct conditional distribution
294 ADT pFS = choose(given: parentsValues, forceComplete: true); // P(F|S=parentsValues)
295
296 // TODO(Duy): only works for one key now, seems horribly slow this way
297 if (nrFrontals() != 1) {
298 throw std::invalid_argument(
299 "DiscreteConditional::sample can only be called on single variable "
300 "conditionals");
301 }
302 Key key = firstFrontalKey();
303 size_t nj = cardinality(j: key);
304 vector<double> p(nj);
305 DiscreteValues frontals;
306 for (size_t value = 0; value < nj; value++) {
307 frontals[key] = value;
308 p[value] = pFS(frontals); // P(F=value|S=parentsValues)
309 if (p[value] == 1.0) {
310 return value; // shortcut exit
311 }
312 }
313
314 // Check if rng is nullptr, then assign default
315 rng = (rng == nullptr) ? &kRandomNumberGenerator : rng;
316
317 std::discrete_distribution<size_t> distribution(p.begin(), p.end());
318 return distribution(*rng);
319}

342/* ************************************************************************* */
343vector<DiscreteValues> DiscreteConditional::frontalAssignments() const {
344 vector<pair<Key, size_t>> pairs;
345 for (Key key : frontals()) pairs.emplace_back(args&: key, args: cardinalities_.at(k: key));
346 vector<pair<Key, size_t>> rpairs(pairs.rbegin(), pairs.rend());
347 return DiscreteValues::CartesianProduct(keys: rpairs);
348}

350/* ************************************************************************* */
351vector<DiscreteValues> DiscreteConditional::allAssignments() const {
352 vector<pair<Key, size_t>> pairs;
353 for (Key key : parents()) pairs.emplace_back(args&: key, args: cardinalities_.at(k: key));
354 for (Key key : frontals()) pairs.emplace_back(args&: key, args: cardinalities_.at(k: key));
355 vector<pair<Key, size_t>> rpairs(pairs.rbegin(), pairs.rend());
356 return DiscreteValues::CartesianProduct(keys: rpairs);
357}

360// Print out signature.
361static void streamSignature(const DiscreteConditional& conditional,
362 const KeyFormatter& keyFormatter,
363 stringstream* ss) {
364 *ss << "P(";
365 bool first = true;
366 for (Key key : conditional.frontals()) {
367 if (!first) *ss << ",";
368 *ss << keyFormatter(key);
369 first = false;
370 }
371 if (conditional.nrParents() > 0) {
372 *ss << "|";
373 bool first = true;
374 for (Key parent : conditional.parents()) {
375 if (!first) *ss << ",";
376 *ss << keyFormatter(parent);
377 first = false;
378 }
379 }
380 *ss << "):";
381}

383/* ************************************************************************* */
384std::string DiscreteConditional::markdown(const KeyFormatter& keyFormatter,
385 const Names& names) const {
386 stringstream ss;
387 ss << " *";
388 streamSignature(conditional: *this, keyFormatter, ss: &ss);
389 ss << "*\n" << std::endl;
390 if (nrParents() == 0) {
391 // We have no parents, call factor method.
392 ss << BaseFactor::markdown(keyFormatter, names);
393 return ss.str();
394 }
395
396 // Print out header.
397 ss << "|";
398 for (Key parent : parents()) {
399 ss << "*" << keyFormatter(parent) << "*|";
400 }
401
402 auto frontalAssignments = this->frontalAssignments();
403 for (const auto& a : frontalAssignments) {
404 for (auto&& it = beginFrontals(); it != endFrontals(); ++it) {
405 size_t index = a.at(k: *it);
406 ss << DiscreteValues::Translate(names, key: *it, index);
407 }
408 ss << "|";
409 }
410 ss << "\n";
411
412 // Print out separator with alignment hints.
413 ss << "|";
414 size_t n = frontalAssignments.size();
415 for (size_t j = 0; j < nrParents() + n; j++) ss << ":-:|";
416 ss << "\n";
417
418 // Print out all rows.
419 size_t count = 0;
420 for (const auto& a : allAssignments()) {
421 if (count == 0) {
422 ss << "|";
423 for (auto&& it = beginParents(); it != endParents(); ++it) {
424 size_t index = a.at(k: *it);
425 ss << DiscreteValues::Translate(names, key: *it, index) << "|";
426 }
427 }
428 ss << operator()(values: a) << "|";
429 count = (count + 1) % n;
430 if (count == 0) ss << "\n";
431 }
432 return ss.str();
433}

435/* ************************************************************************ */
436string DiscreteConditional::html(const KeyFormatter& keyFormatter,
437 const Names& names) const {
438 stringstream ss;
439 ss << "<div>\n<p> <i>";
440 streamSignature(conditional: *this, keyFormatter, ss: &ss);
441 ss << "</i></p>\n";
442 if (nrParents() == 0) {
443 // We have no parents, call factor method.
444 ss << BaseFactor::html(keyFormatter, names);
445 return ss.str();
446 }
447
448 // Print out preamble.
449 ss << "<table class='DiscreteConditional'>\n <thead>\n";
450
451 // Print out header row.
452 ss << " <tr>";
453 for (Key parent : parents()) {
454 ss << "<th><i>" << keyFormatter(parent) << "</i></th>";
455 }
456 auto frontalAssignments = this->frontalAssignments();
457 for (const auto& a : frontalAssignments) {
458 ss << "<th>";
459 for (auto&& it = beginFrontals(); it != endFrontals(); ++it) {
460 size_t index = a.at(k: *it);
461 ss << DiscreteValues::Translate(names, key: *it, index);
462 }
463 ss << "</th>";
464 }
465 ss << "</tr>\n";
466
467 // Finish header and start body.
468 ss << " </thead>\n <tbody>\n";
469
470 // Output all rows, one per assignment:
471 size_t count = 0, n = frontalAssignments.size();
472 for (const auto& a : allAssignments()) {
473 if (count == 0) {
474 ss << " <tr>";
475 for (auto&& it = beginParents(); it != endParents(); ++it) {
476 size_t index = a.at(k: *it);
477 ss << "<th>" << DiscreteValues::Translate(names, key: *it, index) << "</th>";
478 }
479 }
480 ss << "<td>" << operator()(values: a) << "</td>"; // value
481 count = (count + 1) % n;
482 if (count == 0) ss << "</tr>\n";
483 }
484
485 // Finish up
486 ss << " </tbody>\n</table>\n</div>";
487 return ss.str();
488}

508/* ************************************************************************ */
509void DiscreteConditional::removeDiscreteModes(const DiscreteValues& given) {
510 AlgebraicDecisionTree<Key> tree(*this);
511 for (auto [key, value] : given) {
512 tree = tree.choose(label: key, index: value);
513 }
514
515 // Get the leftover DiscreteKey frontals
516 DiscreteKeys frontals;
517 std::for_each(first: this->frontals().begin(), last: this->frontals().end(), f: [&](Key key) {
518 // Check if frontal key exists in given, if not add to new frontals
519 if (given.count(x: key) == 0) {
520 frontals.emplace_back(args&: key, args&: this->cardinalities_.at(k: key));
521 }
522 });
523 // Get the leftover DiscreteKey parents
524 DiscreteKeys parents;
525 std::for_each(first: this->parents().begin(), last: this->parents().end(), f: [&](Key key) {
526 // Check if parent key exists in given, if not add to new parents
527 if (given.count(x: key) == 0) {
528 parents.emplace_back(args&: key, args&: this->cardinalities_.at(k: key));
529 }
530 });
531
532 DiscreteKeys allDkeys(frontals);
533 allDkeys.insert(position: allDkeys.end(), first: parents.begin(), last: parents.end());
534
535 // Update the conditional
536 this->keys_ = allDkeys.indices();
537 this->cardinalities_ = allDkeys.cardinalities();
538 this->root_ = tree.root_;
539 this->nrFrontals_ = frontals.size();
540}
