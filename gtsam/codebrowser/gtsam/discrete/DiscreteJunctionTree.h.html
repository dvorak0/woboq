30/**
31 * An EliminatableClusterTree, i.e., a set of variable clusters with factors,
32 * arranged in a tree, with the additional property that it represents the
33 * clique tree associated with a Bayes net.
34 *
35 * In GTSAM a junction tree is an intermediate data structure in multifrontal
36 * variable elimination. Each node is a cluster of factors, along with a
37 * clique of variables that are eliminated all at once. In detail, every node k
38 * represents a clique (maximal fully connected subset) of an associated chordal
39 * graph, such as a chordal Bayes net resulting from elimination.
40 *
41 * The difference with the BayesTree is that a JunctionTree stores factors,
42 * whereas a BayesTree stores conditionals, that are the product of eliminating
43 * the factors in the corresponding JunctionTree cliques.
44 *
45 * The tree structure and elimination method are exactly analogous to the
46 * EliminationTree, except that in the JunctionTree, at each node multiple
47 * variables are eliminated at a time.
48 *
49 * \ingroup Multifrontal
50 * @ingroup discrete
51 * \nosubgrouping
52 */
53class GTSAM_EXPORT DiscreteJunctionTree
54 : public JunctionTree<DiscreteBayesTree, DiscreteFactorGraph> {
55 public:
56 typedef JunctionTree<DiscreteBayesTree, DiscreteFactorGraph>
57 Base; ///< Base class
58 typedef DiscreteJunctionTree This; ///< This class
59 typedef std::shared_ptr<This> shared_ptr; ///< Shared pointer to this class
60
61 /// @name Constructors
62 /// @{
63
64 /**
65 * Build the elimination tree of a factor graph using precomputed column
66 * structure.
67 * @param factorGraph The factor graph for which to build the elimination tree
68 * @param structure The set of factors involving each variable. If this is
69 * not precomputed, you can call the Create(const FactorGraph<DERIVEDFACTOR>&)
70 * named constructor instead.
71 * @return The elimination tree
72 */
73 DiscreteJunctionTree(const DiscreteEliminationTree& eliminationTree);
74
75 /// @}
76 /// @name Testable
77 /// @{
78
79 /** Print the tree to cout */
80 void print(const std::string& name = "DiscreteJunctionTree: ",
81 const KeyFormatter& formatter = DefaultKeyFormatter) const;
82
83 /// @}
84};
