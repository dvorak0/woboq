19#pragma once
20#include <string>
21#include <vector>

23#include <gtsam/discrete/DiscreteKey.h>

25namespace gtsam {

27 /**
28 * Signature for a discrete conditional distribution, used to construct conditionals.
29 *
30 * The format is (Key % string) for nodes with no parents,
31 * and (Key | Key, Key = string) for nodes with parents.
32 *
33 * The string specifies a conditional probability table in 00 01 10 11 order.
34 * For three-valued, it would be 00 01 02 10 11 12 20 21 22, etc...
35 *
36 * For example, given the following keys
37 *
38 * Key A("Asia"), S("Smoking"), T("Tuberculosis"), L("LungCancer"),
39 * B("Bronchitis"), E("Either"), X("XRay"), D("Dyspnoea");
40 *
41 * These are all valid signatures (Asia network example):
42 *
43 * A % "99/1"
44 * S % "50/50"
45 * T|A = "99/1 95/5"
46 * L|S = "99/1 90/10"
47 * B|S = "70/30 40/60"
48 * (E|T,L) = "F F F 1"
49 * X|E = "95/5 2/98"
50 * (D|E,B) = "9/1 2/8 3/7 1/9"
51 *
52 * @ingroup discrete
53 */
54 class GTSAM_EXPORT Signature {
55
56 public:
57
58 /** Data type for the CPT */
59 typedef std::vector<double> Row;
60 typedef std::vector<Row> Table;
61
62 private:
63
64 /** the variable key */
65 DiscreteKey key_;
66
67 /** the parent keys */
68 DiscreteKeys parents_;
69
70 // the given CPT specification string
71 std::optional<std::string> spec_;
72
73 // the CPT as parsed, if successful
74 std::optional<Table> table_;
75
76 public:
77 /**
78 * Construct from key, parents, and a Signature::Table specifying the
79 * conditional probability table (CPT) in 00 01 10 11 order. For
80 * three-valued, it would be 00 01 02 10 11 12 20 21 22, etc....
81 *
82 * The first string is parsed to add a key and parents.
83 *
84 * Example:
85 * Signature::Table table{{0.9, 0.1}, {0.2, 0.8}, {0.3, 0.7}, {0.1, 0.9}};
86 * Signature sig(D, {E, B}, table);
87 */
88 Signature(const DiscreteKey& key, const DiscreteKeys& parents,
89 const Table& table);
90
91 /**
92 * Construct from key, parents, and a string specifying the conditional
93 * probability table (CPT) in 00 01 10 11 order. For three-valued, it would
94 * be 00 01 02 10 11 12 20 21 22, etc....
95 *
96 * The first string is parsed to add a key and parents. The second string
97 * parses into a table.
98 *
99 * Example (same CPT as above):
100 * Signature sig(D, {B,E}, "9/1 2/8 3/7 1/9");
101 */
102 Signature(const DiscreteKey& key, const DiscreteKeys& parents,
103 const std::string& spec);
104
105 /**
106 * Construct from a single DiscreteKey.
107 *
108 * The resulting signature has no parents or CPT table. Typical use then
109 * either adds parents with | and , operators below, or assigns a table with
110 * operator=().
111 */
112 Signature(const DiscreteKey& key);
113
114 /** the variable key */
115 const DiscreteKey& key() const { return key_; }
116
117 /** the parent keys */
118 const DiscreteKeys& parents() const { return parents_; }
119
120 /** All keys, with variable key first */
121 DiscreteKeys discreteKeys() const;
122
123 /** All key indices, with variable key first */
124 KeyVector indices() const;
125
126 // the CPT as parsed, if successful
127 const std::optional<Table>& table() const { return table_; }
128
129 // the CPT as a vector of doubles, with key's values most rapidly changing
130 std::vector<double> cpt() const;
131
132 /** Add a parent */
133 Signature& operator,(const DiscreteKey& parent);
134
135 /** Add the CPT spec */
136 Signature& operator=(const std::string& spec);
137
138 /** Add the CPT spec directly as a table */
139 Signature& operator=(const Table& table);
140
141 /** provide streaming */
142 GTSAM_EXPORT friend std::ostream& operator<<(std::ostream& os,
143 const Signature& s);
144 };
145
146 /**
147 * Helper function to create Signature objects
148 * example: Signature s = D | E;
149 */
150 GTSAM_EXPORT Signature operator|(const DiscreteKey& key, const DiscreteKey& parent);
151
152 /**
153 * Helper function to create Signature objects
154 * example: Signature s(D % "99/1");
155 */
156 GTSAM_EXPORT Signature operator%(const DiscreteKey& key, const std::string& parent);
157
158 /**
159 * Helper function to create Signature objects, using table construction directly
160 * example: Signature s(D % table);
161 */
162 GTSAM_EXPORT Signature operator%(const DiscreteKey& key, const Signature::Table& parent);
163
164}
