23#include <gtsam/discrete/DiscreteBayesNet.h>
24#include <gtsam/discrete/DiscreteBayesTree.h>
25
26#include <queue>
27
28namespace gtsam {
29
30/**
31 * @brief DiscreteSearch: Search for the K best solutions.
32 *
33 * This class is used to search for the K best solutions in a DiscreteBayesNet.
34 * This is implemented with a modified A* search algorithm that uses a priority
35 * queue to manage the search nodes. That machinery is defined in the .cpp file.
36 * The heuristic we use is the sum of the log-probabilities of the
37 * maximum-probability assignments for each slot, for all slots to the right of
38 * the current slot.
39 *
40 * TODO: The heuristic could be refined by using the partial assignment in
41 * search node to refine the max-probability assignment for the remaining slots.
42 * This would incur more computation but will lead to fewer expansions.
43 */
44class GTSAM_EXPORT DiscreteSearch {
45 public:
46 /**
47 * We structure the search as a set of slots, each with a factor and
48 * a set of variable assignments that need to be chosen. In addition, each
49 * slot has a heuristic associated with it.
50 *
51 * Example:
52 * The factors in the search problem (always parents before descendents!):
53 * [P(A), P(B|A), P(C|A,B)]
54 * The assignments for each factor.
55 * [[A0,A1], [B0,B1], [C0,C1,C2]]
56 * A lower bound on the cost-to-go after each slot, e.g.,
57 * [-log(max_B P(B|A)) -log(max_C P(C|A,B)), -log(max_C P(C|A,B)), 0.0]
58 * Note that these decrease as we move from right to left.
59 * We keep the global lower bound as lowerBound_. In the example, it is:
60 * -log(max_B P(B|A)) -log(max_C P(C|A,B)) -log(max_C P(C|A,B))
61 */
62 struct Slot {
63 DiscreteFactor::shared_ptr factor;
64 std::vector<DiscreteValues> assignments;
65 double heuristic;
66
67 friend std::ostream& operator<<(std::ostream& os, const Slot& slot) {
68 os << "Slot with " << slot.assignments.size()
69 << " assignments, heuristic=" << slot.heuristic;
70 os << ", factor:\n" << slot.factor->markdown() << std::endl;
71 return os;
72 }
73 };
74
75 /**
76 * A solution is a set of assignments, covering all the slots.
77 * as well as an associated error = -log(probability)
78 */
79 struct Solution {
80 double error;
81 DiscreteValues assignment;
82 Solution(double err, const DiscreteValues& assign)
83 : error(err), assignment(assign) {}
84 friend std::ostream& operator<<(std::ostream& os, const Solution& sn) {
85 os << "[ error=" << sn.error << " assignment={" << sn.assignment << "}]";
86 return os;
87 }
88 };
89
90 public:
91 /// @name Standard Constructors
92 /// @{
93
94 /**
95 * Construct from a DiscreteFactorGraph.
96 *
97 * Internally creates either an elimination tree or a junction tree. The
98 * latter incurs more up-front computation but the search itself might be
99 * faster. Then again, for the elimination tree, the heuristic will be more
100 * fine-grained (more slots).
101 *
102 * @param factorGraph The factor graph to search over.
103 * @param ordering The ordering used to create etree (and maybe jtree).
104 * @param buildJunctionTree Whether to build a junction tree or not.
105 */
106 static DiscreteSearch FromFactorGraph(const DiscreteFactorGraph& factorGraph,
107 const Ordering& ordering,
108 bool buildJunctionTree = false);
109
110 /// Construct from a DiscreteEliminationTree.
111 DiscreteSearch(const DiscreteEliminationTree& etree);
112
113 /// Construct from a DiscreteJunctionTree.
114 DiscreteSearch(const DiscreteJunctionTree& junctionTree);
115
116 //// Construct from a DiscreteBayesNet.
117 DiscreteSearch(const DiscreteBayesNet& bayesNet);
118
119 /// Construct from a DiscreteBayesTree.
120 DiscreteSearch(const DiscreteBayesTree& bayesTree);
121
122 /// @}
123 /// @name Testable
124 /// @{
125
126 /** Print the tree to cout */
127 void print(const std::string& name = "DiscreteSearch: ",
128 const KeyFormatter& formatter = DefaultKeyFormatter) const;
129
130 /// @}
131 /// @name Standard API
132 /// @{
133
134 /// Return lower bound on the cost-to-go for the entire search
135 double lowerBound() const { return lowerBound_; }
136
137 /// Read access to the slots
138 const std::vector<Slot>& slots() const { return slots_; }
139
140 /**
141 * @brief Search for the K best solutions.
142 *
143 * This method performs a search to find the K best solutions for the given
144 * DiscreteBayesNet. It uses a priority queue to manage the search nodes,
145 * expanding nodes with the smallest bound first. The search continues until
146 * all possible nodes have been expanded or pruned.
147 *
148 * @return A vector of the K best solutions found during the search.
149 */
150 std::vector<Solution> run(size_t K = 1) const;
151
152 /// @}
153
154 private:
155 /**
156 * Compute the cumulative lower-bound cost-to-go after each slot is filled.
157 * @return the estimated lower bound of the cost for *all* slots.
158 */
159 double computeHeuristic();
160
161 double lowerBound_; ///< Lower bound on the cost-to-go for the entire search.
162 std::vector<Slot> slots_; ///< The slots to fill in the search.
163};
164
165using DiscreteSearchSolution = DiscreteSearch::Solution; // for wrapping
166} // namespace gtsam
