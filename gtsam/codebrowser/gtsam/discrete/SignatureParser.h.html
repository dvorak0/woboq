27namespace gtsam {
28/**
29 * @brief A simple parser that replaces the boost spirit parser.
30 *
31 * It is meant to parse strings like "1/1 2/3 1/4". Every word of the form
32 * "a/b/c/..." is parsed as a row, and the rows are stored in a table.
33 *
34 * A `Row` is a vector of doubles, and a `Table` is a vector of rows.
35 *
36 * Examples: the parser is able to parse the following strings:
37 * "1/1 2/3 1/4":
38 * {{1,1},{2,3},{1,4}}
39 * "1/1 2/3 1/4 1/1 2/3 1/4" :
40 * {{1,1},{2,3},{1,4},{1,1},{2,3},{1,4}}
41 * "1/2/3 2/3/4 1/2/3 2/3/4 1/2/3 2/3/4 1/2/3 2/3/4 1/2/3" :
42 * {{1,2,3},{2,3,4},{1,2,3},{2,3,4},{1,2,3},{2,3,4},{1,2,3},{2,3,4},{1,2,3}}
43 *
44 * If the string has un-parsable elements the parser will fail with nullopt:
45 * "1/2 sdf" : nullopt !
46 *
47 * It also fails if the string is empty:
48 * "": nullopt !
49 *
50 * Also fails if the rows are not of the same size.
51 */
52struct GTSAM_EXPORT SignatureParser {
53 using Row = std::vector<double>;
54 using Table = std::vector<Row>;
55
56 static std::optional<Table> Parse(const std::string& str);
57};
58} // namespace gtsam
