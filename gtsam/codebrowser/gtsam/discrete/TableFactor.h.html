46/**
47 * A discrete probabilistic factor optimized for sparsity.
48 * Uses sparse_table_ to store only the nonzero probabilities.
49 * Computes the assigned value for the key using the ordering which the
50 * nonzero probabilties are stored in. (lazy cartesian product)
51 *
52 * @ingroup discrete
53 */
54class GTSAM_EXPORT TableFactor : public DiscreteFactor {
55 protected:
56 /// SparseVector of nonzero probabilities.
57 Eigen::SparseVector<double> sparse_table_;
58
59 private:
60 /// Map of Keys and their denominators used in keyValueForIndex.
61 std::map<Key, size_t> denominators_;
62 /// Sorted DiscreteKeys to use internally.
63 DiscreteKeys sorted_dkeys_;
64
65 /**
66 * @brief Uses lazy cartesian product to find nth entry in the cartesian
67 * product of arrays in O(1)
68 * Example)
69 * v0 | v1 | val
70 * 0 | 0 | 10
71 * 0 | 1 | 21
72 * 1 | 0 | 32
73 * 1 | 1 | 43
74 * keyValueForIndex(v1, 2) = 0
75 * @param target_key nth entry's key to find out its assigned value
76 * @param index nth entry in the sparse vector
77 * @return TableFactor
78 */
79 size_t keyValueForIndex(Key target_key, uint64_t index) const;
80
81 /**
82 * @brief Return ith key in keys_ as a DiscreteKey
83 * @param i ith key in keys_
84 * @return DiscreteKey
85 */
86 DiscreteKey discreteKey(size_t i) const {
87 return DiscreteKey(keys_[i], cardinalities_.at(k: keys_[i]));
88 }
89
90 public:
91 /**
92 * Convert probability table given as doubles to SparseVector.
93 * Example: {0, 1, 1, 0, 0, 1, 0} -> values: {1, 1, 1}, indices: {1, 2, 5}
94 */
95 static Eigen::SparseVector<double> Convert(const DiscreteKeys& keys,
96 const std::vector<double>& table);
97
98 /// Convert probability table given as string to SparseVector.
99 static Eigen::SparseVector<double> Convert(const DiscreteKeys& keys,
100 const std::string& table);
101

106 typedef Eigen::SparseVector<double>::InnerIterator SparseIt;
107 typedef std::vector<std::pair<DiscreteValues, double>> AssignValList;

188 /**
189 * @brief Multiply factors, DiscreteFactor::shared_ptr edition.
190 *
191 * This method accepts `DiscreteFactor::shared_ptr` and uses dynamic
192 * dispatch and specializations to perform the most efficient
193 * multiplication.
194 *
195 * While converting a DecisionTreeFactor to a TableFactor is efficient, the
196 * reverse is not.
197 * Hence we specialize the code to return a TableFactor always.

199 * @param f The factor to multiply with.
200 * @return DiscreteFactor::shared_ptr

316 /// Enumerate all values into a map from values to double.
317 std::vector<std::pair<DiscreteValues, double>> enumerate() const;
318
319 /**
320 * @brief Prune the decision tree of discrete variables.
321 *
322 * Pruning will set the values to be "pruned" to 0 indicating a 0
323 * probability. An assignment is pruned if it is not in the top
324 * `maxNrAssignments` values.
325 *
326 * A violation can occur if there are more
327 * duplicate values than `maxNrAssignments`. A violation here is the need to
328 * un-prune the decision tree (e.g. all assignment values are 1.0). We could
329 * have another case where some subset of duplicates exist (e.g. for a tree
330 * with 8 assignments we have 1, 1, 1, 1, 0.8, 0.7, 0.6, 0.5), but this is
331 * not a violation since the for `maxNrAssignments=5` the top values are (1,
332 * 0.8).
333 *
334 * @param maxNrAssignments The maximum number of assignments to keep.
335 * @return TableFactor
336 */
337 TableFactor prune(size_t maxNrAssignments) const;
338
339 /**
340 * Get the number of non-zero values contained in this factor.
341 * It could be much smaller than `prod_{key}(cardinality(key))`.
342 */
343 uint64_t nrValues() const override { return sparse_table_.nonZeros(); }
