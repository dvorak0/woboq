30static void stream(std::ostream& os, const DiscreteValues& x,
31 const KeyFormatter& keyFormatter) {
32 for (const auto& kv : x)
33 os << "(" << keyFormatter(kv.first) << ", " << kv.second << ")";
34}

50/* ************************************************************************ */
51bool DiscreteValues::equals(const DiscreteValues& x, double tol) const {
52 if (this->size() != x.size()) return false;
53 auto it1 = x.begin();
54 auto it2 = this->begin();
55 for (; it1 != x.end(); ++it1, ++it2) {
56 if (it1->first != it2->first || it1->second != it2->second) return false;
57 }
58 return true;
59}

81/* ************************************************************************ */
82DiscreteValues& DiscreteValues::update(const DiscreteValues& values) {
83 for (const auto& kv : values) {
84 if (!count(x: kv.first)) {
85 throw std::out_of_range(
86 "Requested to update a DiscreteValues with another DiscreteValues "
87 "that contains keys not present in the first.");
88 } else {
89 (*this)[kv.first] = kv.second;
90 }
91 }
92 return *this;
93}

114string DiscreteValues::markdown(const KeyFormatter& keyFormatter,
115 const Names& names) const {
116 stringstream ss;
117
118 // Print out header and separator with alignment hints.
119 ss << "|Variable|value|\n|:-:|:-:|\n";
120
121 // Print out all rows.
122 for (const auto& kv : *this) {
123 ss << "|" << keyFormatter(kv.first) << "|"
124 << Translate(names, key: kv.first, index: kv.second) << "|\n";
125 }
126
127 return ss.str();
128}

130/* ************************************************************************ */
131string DiscreteValues::html(const KeyFormatter& keyFormatter,
132 const Names& names) const {
133 stringstream ss;
134
135 // Print out preamble.
136 ss << "<div>\n<table class='DiscreteValues'>\n <thead>\n";
137
138 // Print out header row.
139 ss << " <tr><th>Variable</th><th>value</th></tr>\n";
140
141 // Finish header and start body.
142 ss << " </thead>\n <tbody>\n";
143
144 // Print out all rows.
145 for (const auto& kv : *this) {
146 ss << " <tr>";
147 ss << "<th>" << keyFormatter(kv.first) << "</th><td>"
148 << Translate(names, key: kv.first, index: kv.second) << "</td>";
149 ss << "</tr>\n";
150 }
151 ss << " </tbody>\n</table>\n</div>";
152 return ss.str();
153}
