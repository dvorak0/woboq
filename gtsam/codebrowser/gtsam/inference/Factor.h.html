42 /**
43 * This is the base class for all factor types, as well as conditionals,
44 * which are implemented as specialized factors. This class does not store any
45 * data other than its keys. Derived classes store data such as matrices and
46 * probability tables.
47 *
48 * The `error` method is used to evaluate the factor, and is the only method
49 * that is required to be implemented in derived classes, although it has a
50 * default implementation that throws an exception.
51 *
52 * There are five broad classes of factors that derive from Factor:
53 *
54 * - \b Nonlinear factors, such as \class NonlinearFactor and \class NoiseModelFactor, which
55 * represent a nonlinear likelihood function over a set of variables.
56 * - \b Gaussian factors, such as \class JacobianFactor and \class HessianFactor, which
57 * represent a Gaussian likelihood over a set of variables.
58 * - \b Discrete factors, such as \class DiscreteFactor and \class DecisionTreeFactor, which
59 * represent a discrete distribution over a set of variables.
60 * - \b Hybrid factors, such as \class HybridFactor, which represent a mixture of
61 * Gaussian and discrete distributions over a set of variables.
62 * - \b Symbolic factors, used to represent a graph structure, such as
63 * \class SymbolicFactor, only used for symbolic elimination etc.
64 *
65 * Note that derived classes must also redefine the `This` and `shared_ptr`
66 * typedefs. See JacobianFactor, etc. for examples.
67 *
68 * \nosubgrouping
69 */
70 class GTSAM_EXPORT Factor
71 {
72
73 private:
74 // These typedefs are private because they must be overridden in derived classes.
75 typedef Factor This; ///< This class
76 typedef std::shared_ptr<Factor> shared_ptr; ///< A shared_ptr to this class.
77
78 public:
79 /// Iterator over keys
80 typedef KeyVector::iterator iterator;
81
82 /// Const iterator over keys
83 typedef KeyVector::const_iterator const_iterator;
84
85 protected:
86
87 /// The keys involved in this factor
88 KeyVector keys_;
89
90 /// @name Standard Constructors
91 /// @{
92
93 /** Default constructor for I/O */
94 Factor() {}
95
96 /** Construct factor from container of keys. This constructor is used internally from derived factor
97 * constructors, either from a container of keys or from a boost::assign::list_of. */
98 template<typename CONTAINER>
99 explicit Factor(const CONTAINER& keys) : keys_(keys.begin(), keys.end()) {}
100
101 /** Construct factor from iterator keys. This constructor may be used internally from derived
102 * factor constructors, although our code currently does not use this. */
103 template<typename ITERATOR>
104 Factor(ITERATOR first, ITERATOR last) : keys_(first, last) {}
105
106 /** Construct factor from container of keys. This is called internally from derived factor static
107 * factor methods, as a workaround for not being able to call the protected constructors above. */
108 template<typename CONTAINER>
109 static Factor FromKeys(const CONTAINER& keys) {
110 return Factor(keys.begin(), keys.end()); }
111
112 /** Construct factor from iterator keys. This is called internally from derived factor static
113 * factor methods, as a workaround for not being able to call the protected constructors above. */
114 template<typename ITERATOR>
115 static Factor FromIterators(ITERATOR first, ITERATOR last) {
116 return Factor(first, last); }
117
118 /// @}
119
120 public:
121 /// Default destructor
122 // public since it is required for boost serialization and static methods.
123 // virtual since it is public.
124 // http://isocpp.github.io/CppCoreGuidelines/CppCoreGuidelines#Rc-dtor-virtual
125 virtual ~Factor() = default;
126
127 /// @name Standard Interface
128 /// @{
129
130 /// Whether the factor is empty (involves zero variables).
131 bool empty() const { return keys_.empty(); }
132
133 /// First key
134 Key front() const { return keys_.front(); }
135
136 /// Last key
137 Key back() const { return keys_.back(); }
138
139 /// find
140 const_iterator find(Key key) const { return std::find(first: begin(), last: end(), val: key); }
141
142 /// Access the factor's involved variable keys
143 const KeyVector& keys() const { return keys_; }
144
145 /** Iterator at beginning of involved variable keys */
146 const_iterator begin() const { return keys_.begin(); }
147
148 /** Iterator at end of involved variable keys */
149 const_iterator end() const { return keys_.end(); }
150
151 /**
152 * All factor types need to implement an error function.
153 * In factor graphs, this is the negative log-likelihood.
154 */
155 virtual double error(const HybridValues& c) const;
156
157 /**
158 * @return the number of variables involved in this factor
159 */
160 size_t size() const { return keys_.size(); }
161
162 /// @}
163
164 /// @name Testable
165 /// @{
166
167 /// print
168 virtual void print(
169 const std::string& s = "Factor",
170 const KeyFormatter& formatter = DefaultKeyFormatter) const;
171
172 /// print only keys
173 virtual void printKeys(
174 const std::string& s = "Factor",
175 const KeyFormatter& formatter = DefaultKeyFormatter) const;
176
177 /// check equality
178 bool equals(const This& other, double tol = 1e-9) const;
179
180 /// @}
181 /// @name Advanced Interface
182 /// @{
183
184 /** @return keys involved in this factor */
185 KeyVector& keys() { return keys_; }
186
187 /** Iterator at beginning of involved variable keys */
188 iterator begin() { return keys_.begin(); }
189
190 /** Iterator at end of involved variable keys */
191 iterator end() { return keys_.end(); }
192
193 /// @}
194
195 private:
196#if GTSAM_ENABLE_BOOST_SERIALIZATION
197 /// @name Serialization
198 /// @{
199 /** Serialization function */
200 friend class boost::serialization::access;
201 template<class Archive>
202 void serialize(Archive & ar, const unsigned int /*version*/) {
203 ar & BOOST_SERIALIZATION_NVP(keys_);
204 }
205#endif
206
207 /// @}
208
209 };
