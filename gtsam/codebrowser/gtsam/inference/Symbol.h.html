32/**
33 * Character and index key used to refer to variables. Will simply cast to a Key,
34 * i.e., a large integer. Keys are used to retrieve values from Values,
35 * specify what variables factors depend on, etc.
36 */
37class GTSAM_EXPORT Symbol {
38protected:
39 unsigned char c_;
40 std::uint64_t j_;
41
42public:
43
44 /** Default constructor */
45 Symbol() :
46 c_(0), j_(0) {
47 }
48
49 /** Copy constructor */
50 Symbol(const Symbol& key) :
51 c_(key.c_), j_(key.j_) {
52 }
53
54 /** Constructor */
55 Symbol(unsigned char c, std::uint64_t j) :
56 c_(c), j_(j) {
57 }
58
59 /** Constructor that decodes an integer Key */
60 Symbol(Key key);
61
62 /** return Key (integer) representation */
63 Key key() const;
64
65 /** Cast to integer */
66 operator Key() const { return key(); }
67
68 /// Print
69 void print(const std::string& s = "") const;
70
71 /// Check equality
72 bool equals(const Symbol& expected, double tol = 0.0) const;
73
74 /** Retrieve key character */
75 unsigned char chr() const {
76 return c_;
77 }
78
79 /** Retrieve key index */
80 std::uint64_t index() const {
81 return j_;
82 }
83
84 /** Create a string from the key */
85 operator std::string() const;
86
87 /// Return string representation of the key
88 std::string string() const { return std::string(*this); }
89
90 /** Comparison for use in maps */
91 bool operator<(const Symbol& comp) const {
92 return c_ < comp.c_ || (comp.c_ == c_ && j_ < comp.j_);
93 }
94
95 /** Comparison for use in maps */
96 bool operator==(const Symbol& comp) const {
97 return comp.c_ == c_ && comp.j_ == j_;
98 }
99
100 /** Comparison for use in maps */
101 bool operator==(Key comp) const {
102 return comp == (Key)(*this);
103 }
104
105 /** Comparison for use in maps */
106 bool operator!=(const Symbol& comp) const {
107 return comp.c_ != c_ || comp.j_ != j_;
108 }
109
110 /** Comparison for use in maps */
111 bool operator!=(Key comp) const {
112 return comp != (Key)(*this);
113 }
114
115 /** Return a filter function that returns true when evaluated on a Key whose
116 * character (when converted to a Symbol) matches \c c. Use this with the
117 * Values::filter() function to retrieve all key-value pairs with the
118 * requested character.
119 */
120 static std::function<bool(Key)> ChrTest(unsigned char c);
121
122 /// Output stream operator that can be used with key_formatter (see Key.h).
123 GTSAM_EXPORT friend std::ostream &operator<<(std::ostream &, const Symbol &);
124
125private:
126
127#if GTSAM_ENABLE_BOOST_SERIALIZATION
128 /** Serialization function */
129 friend class boost::serialization::access;
130 template<class ARCHIVE>
131 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
132 ar & BOOST_SERIALIZATION_NVP(c_);
133 ar & BOOST_SERIALIZATION_NVP(j_);
134 }
135#endif
136};

138/** Create a symbol key from a character and index, i.e. x5. */
139inline Key symbol(unsigned char c, std::uint64_t j) { return (Key)Symbol(c,j); }
140
141/** Return the character portion of a symbol key. */
142inline unsigned char symbolChr(Key key) { return Symbol(key).chr(); }
143
144/** Return the index portion of a symbol key. */
145inline std::uint64_t symbolIndex(Key key) { return Symbol(key).index(); }

176/** Generates symbol shorthands with alternative names different than the
177 * one-letter predefined ones. */
178class SymbolGenerator {
179 const unsigned char c_;
180public:
181 constexpr SymbolGenerator(const unsigned char c) : c_(c) {}
182 Symbol operator()(const std::uint64_t j) const { return Symbol(c_, j); }
183 constexpr unsigned char chr() const { return c_; }
184};
