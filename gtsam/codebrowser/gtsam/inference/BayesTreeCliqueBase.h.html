14 * @brief Base class for cliques of a BayesTree

18#pragma once
19
20#include <gtsam/inference/Key.h>
21#include <gtsam/inference/Ordering.h>
22#include <gtsam/base/types.h>
23#include <gtsam/base/FastVector.h>
24
25#include <string>
26#include <mutex>
27#include <optional>

29namespace gtsam {
30
31 // Forward declarations
32 template<class CLIQUE> class BayesTree;
33 template<class GRAPH> struct EliminationTraits;
34
35 /**
36 * This is the base class for BayesTree cliques. The default and standard derived type is
37 * BayesTreeClique, but some algorithms, like iSAM2, use a different clique type in order to store
38 * extra data along with the clique.
39 *
40 * This class is templated on the derived class (i.e. the curiously recursive template pattern).
41 * The advantage of this over using virtual classes is that it avoids the need for casting to get
42 * the derived type. This is possible because all cliques in a BayesTree are the same type - if
43 * they were not then we'd need a virtual class.
44 *
45 * @tparam DERIVED The derived clique type.
46 * @tparam CONDITIONAL The conditional type.
47 * \nosubgrouping */
48 template<class DERIVED, class FACTORGRAPH>
49 class BayesTreeCliqueBase
50 {
51 private:
52 typedef BayesTreeCliqueBase<DERIVED, FACTORGRAPH> This;
53 typedef DERIVED DerivedType;
54 typedef EliminationTraits<FACTORGRAPH> EliminationTraitsType;
55 typedef std::shared_ptr<This> shared_ptr;
56 typedef std::weak_ptr<This> weak_ptr;
57 typedef std::shared_ptr<DerivedType> derived_ptr;
58 typedef std::weak_ptr<DerivedType> derived_weak_ptr;
59
60 public:
61 typedef FACTORGRAPH FactorGraphType;
62 typedef typename EliminationTraitsType::BayesNetType BayesNetType;
63 typedef typename BayesNetType::ConditionalType ConditionalType;
64 typedef std::shared_ptr<ConditionalType> sharedConditional;
65 typedef typename FactorGraphType::FactorType FactorType;
66 typedef typename FactorGraphType::Eliminate Eliminate;
67
68 protected:
69
70 /// @name Standard Constructors
71 /// @{
72
73 /// Default constructor
74 BayesTreeCliqueBase() : problemSize_(1) {}
75
76 /// Construct from a conditional, leaving parent and child pointers
77 /// uninitialized.
78 BayesTreeCliqueBase(const sharedConditional& conditional)
79 : conditional_(conditional), problemSize_(1) {}
80
81 /// Shallow copy constructor.
82 BayesTreeCliqueBase(const BayesTreeCliqueBase& c)
83 : conditional_(c.conditional_),
84 parent_(c.parent_),
85 children(c.children),
86 problemSize_(c.problemSize_),
87 is_root(c.is_root) {}
88
89 /// Shallow copy assignment constructor
90 BayesTreeCliqueBase& operator=(const BayesTreeCliqueBase& c) {
91 conditional_ = c.conditional_;
92 parent_ = c.parent_;
93 children = c.children;
94 problemSize_ = c.problemSize_;
95 is_root = c.is_root;
96 return *this;
97 }
98
99 // Virtual destructor.
100 virtual ~BayesTreeCliqueBase() {}
101
102 /// @}
103
104 /// This stores the Cached separator marginal P(S)
105 mutable std::optional<FactorGraphType> cachedSeparatorMarginal_;
106 /// This protects Cached seperator marginal P(S) from concurrent read/writes
107 /// as many the functions which access it are const (hence the mutable)
108 /// leading to the false impression that these const functions are thread-safe
109 /// which is not true due to these mutable values. This is fixed by applying this mutex.
110 mutable std::mutex cachedSeparatorMarginalMutex_;
111
112 public:
113 sharedConditional conditional_;
114 derived_weak_ptr parent_;
115 FastVector<derived_ptr> children;
116 int problemSize_;
117
118 bool is_root = false;
119
120 /// Fill the elimination result produced during elimination. Here this just stores the
121 /// conditional and ignores the remaining factor, but this is overridden in ISAM2Clique
122 /// to also cache the remaining factor.
123 void setEliminationResult(const typename FactorGraphType::EliminationResult& eliminationResult);
124
125 /// @name Testable
126 /// @{
127
128 /** check equality */
129 bool equals(const DERIVED& other, double tol = 1e-9) const;
130
131 /** print this node */
132 virtual void print(
133 const std::string& s = "",
134 const KeyFormatter& keyFormatter = DefaultKeyFormatter) const;
135
136 /// @}
137 /// @name Standard Interface
138 /// @{
139
140 /** Access the conditional */
141 const sharedConditional& conditional() const { return conditional_; }
142
143 /** Write access to the conditional */
144 sharedConditional& conditional() { return conditional_; }
145
146 /// Return true if this clique is the root of a Bayes tree.
147 inline bool isRoot() const { return parent_.expired(); }
148
149 /// Return the number of children.
150 size_t nrChildren() const { return children.size(); }
151
152 /// Return the child at index i.
153 const derived_ptr operator[](size_t i) const { return children.at(i); }
154
155 /** The size of subtree rooted at this clique, i.e., nr of Cliques */
156 size_t treeSize() const;
157
158 /** Collect number of cliques with cached separator marginals */
159 size_t numCachedSeparatorMarginals() const;
160
161 /** return a shared_ptr to the parent clique */
162 derived_ptr parent() const { return parent_.lock(); }
163
164 /** Problem size (used for parallel traversal) */
165 int problemSize() const { return problemSize_; }
166
167 /// @}
168 /// @name Advanced Interface
169 /// @{
170
171 /** return the conditional P(S|Root) on the separator given the root */
172 BayesNetType shortcut(const derived_ptr& root, Eliminate function = EliminationTraitsType::DefaultEliminate) const;
173
174 /** return the marginal P(S) on the separator */
175 FactorGraphType separatorMarginal(Eliminate function = EliminationTraitsType::DefaultEliminate) const;
176
177 /** return the marginal P(C) of the clique, using marginal caching */
178 FactorGraphType marginal2(Eliminate function = EliminationTraitsType::DefaultEliminate) const;
179
180 /**
181 * This deletes the cached shortcuts of all cliques (subtree) below this clique.
182 * This is performed when the bayes tree is modified.
183 */
184 void deleteCachedShortcuts();
185
186 const std::optional<FactorGraphType>& cachedSeparatorMarginal() const {
187 std::lock_guard<std::mutex> marginalLock(cachedSeparatorMarginalMutex_);
188 return cachedSeparatorMarginal_;
189 }
190
191 friend class BayesTree<DerivedType>;
192
193 /// Calculate set \f$ S \setminus B \f$ for shortcut calculations
194 KeyVector separator_setminus_B(const derived_ptr& B) const;
195
196 protected:
197
198 /** Determine variable indices to keep in recursive separator shortcut calculation The factor
199 * graph p_Cp_B has keys from the parent clique Cp and from B. But we only keep the variables
200 * not in S union B. */
201 KeyVector shortcut_indices(const derived_ptr& B, const FactorGraphType& p_Cp_B) const;
202
203 /** Non-recursive delete cached shortcuts and marginals - internal only. */
204 void deleteCachedShortcutsNonRecursive() {
205 std::lock_guard<std::mutex> marginalLock(cachedSeparatorMarginalMutex_);
206 cachedSeparatorMarginal_ = {};
207 }
208
209 private:
210
211#if GTSAM_ENABLE_BOOST_SERIALIZATION
212 /** Serialization function */
213 friend class boost::serialization::access;
214 template<class ARCHIVE>
215 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
216 if(!parent_.lock()) {
217 is_root = true;
218 }
219 ar & BOOST_SERIALIZATION_NVP(is_root);
220 ar & BOOST_SERIALIZATION_NVP(conditional_);
221 if (!is_root) { // TODO(fan): Workaround for boost/serialization #119
222 ar & BOOST_SERIALIZATION_NVP(parent_);
223 }
224 ar & BOOST_SERIALIZATION_NVP(children);
225 }
226#endif
227
228 /// @}
229
230 };
231
232}
