29namespace gtsam {
30/**
31 * The MetisIndex class converts a factor graph into the Compressed Sparse Row format for use in
32 * METIS algorithms. Specifically, two vectors store the adjacency structure of the graph. It is built
33 * from a factor graph prior to elimination, and stores the list of factors
34 * that involve each variable.
35 * \nosubgrouping
36 */
37class GTSAM_EXPORT MetisIndex {
38public:
39 typedef std::shared_ptr<MetisIndex> shared_ptr;
40
41private:
42 // Stores Key <-> integer value relationship
43 struct BiMap {
44 std::map<Key, int32_t> left;
45 std::unordered_map<int32_t, Key> right;
46 void insert(const Key& left_value, const int32_t& right_value) {
47 left[left_value] = right_value;
48 right[right_value] = left_value;
49 }
50 };
51
52 std::vector<int32_t> xadj_; // Index of node's adjacency list in adj
53 std::vector<int32_t> adj_; // Stores ajacency lists of all nodes, appended into a single vector
54 BiMap intKeyBMap_; // Stores Key <-> integer value relationship
55 size_t nKeys_;
56
57public:
58 /// @name Standard Constructors
59 /// @{
60
61 /** Default constructor, creates empty MetisIndex */
62 MetisIndex() :
63 nKeys_(0) {
64 }
65
66 template<class FACTORGRAPH>
67 MetisIndex(const FACTORGRAPH& factorGraph) :
68 nKeys_(0) {
69 augment(factorGraph);
70 }
71
72 ~MetisIndex() {
73 }
74 /// @}
75 /// @name Advanced Interface
76 /// @{
77
78 /**
79 * Augment the variable index with new factors. This can be used when
80 * solving problems incrementally.
81 */
82 template<class FACTORGRAPH>
83 void augment(const FACTORGRAPH& factors);
84
85 const std::vector<int32_t>& xadj() const {
86 return xadj_;
87 }
88 const std::vector<int32_t>& adj() const {
89 return adj_;
90 }
91 size_t nValues() const {
92 return nKeys_;
93 }
94 Key intToKey(int32_t value) const {
95#ifndef NDEBUG
96 if (value < 0) {
97 throw;
98 }
99#endif
100 return intKeyBMap_.right.find(x: value)->second;
101 }
102
103 /// @}
104};
