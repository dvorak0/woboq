33class GTSAM_EXPORT Ordering: public KeyVector {
34protected:
35 typedef KeyVector Base;
36
37public:
38
39 /// Type of ordering to use
40 enum OrderingType {
41 COLAMD, METIS, NATURAL, CUSTOM
42 };
43
44 typedef Ordering This; ///< Typedef to this class
45 typedef std::shared_ptr<This> shared_ptr; ///< shared_ptr to this class
46
47 /// Create an empty ordering
48 Ordering() {
49 }
50
51 using KeyVector::KeyVector; // Inherit the KeyVector's constructors
52
53 /// Create from a container
54 template<typename KEYS>
55 explicit Ordering(const KEYS& keys) :
56 Base(keys.begin(), keys.end()) {
57 }
58
59 /// Add new variables to the ordering as
60 /// `ordering += key1, key2, ...`.
61 This& operator+=(Key key);
62
63 /// Overloading the comma operator allows for chaining appends
64 // e.g. keys += key1, key2
65 This& operator,(Key key);
66
67 /**
68 * @brief Append new keys to the ordering as `ordering += keys`.
69 *
70 * @param keys The key vector to append to this ordering.
71 * @return The ordering variable with appended keys.
72 */
73 This& operator+=(KeyVector& keys);
74
75 /// Check if key exists in ordering.
76 bool contains(const Key& key) const;
77
78 /**
79 * @brief Invert (not reverse) the ordering - returns a map from key to order
80 * position.
81 *
82 * @return FastMap<Key, size_t>
83 */
84 FastMap<Key, size_t> invert() const;
85
86 /// @name Fill-reducing Orderings
87 /// @{
88
89 /// Compute a fill-reducing ordering using COLAMD from a factor graph (see details for note on
90 /// performance). This internally builds a VariableIndex so if you already have a VariableIndex,
91 /// it is faster to use COLAMD(const VariableIndex&)
92 template<class FACTOR_GRAPH>
93 static Ordering Colamd(const FACTOR_GRAPH& graph) {
94 if (graph.empty())
95 return Ordering();
96 else
97 return Colamd(variableIndex: VariableIndex(graph));
98 }
99
100 /// Compute a fill-reducing ordering using COLAMD from a VariableIndex.
101 static Ordering Colamd(const VariableIndex& variableIndex);
102
103 /// Compute a fill-reducing ordering using constrained COLAMD from a factor graph (see details
104 /// for note on performance). This internally builds a VariableIndex so if you already have a
105 /// VariableIndex, it is faster to use COLAMD(const VariableIndex&). This function constrains
106 /// the variables in \c constrainLast to the end of the ordering, and orders all other variables
107 /// before in a fill-reducing ordering. If \c forceOrder is true, the variables in \c
108 /// constrainLast will be ordered in the same order specified in the KeyVector \c
109 /// constrainLast. If \c forceOrder is false, the variables in \c constrainLast will be
110 /// ordered after all the others, but will be rearranged by CCOLAMD to reduce fill-in as well.
111 template<class FACTOR_GRAPH>
112 static Ordering ColamdConstrainedLast(const FACTOR_GRAPH& graph,
113 const KeyVector& constrainLast, bool forceOrder = false) {
114 if (graph.empty())
115 return Ordering();
116 else
117 return ColamdConstrainedLast(variableIndex: VariableIndex(graph), constrainLast, forceOrder);
118 }
119
120 /// Compute a fill-reducing ordering using constrained COLAMD from a VariableIndex. This
121 /// function constrains the variables in \c constrainLast to the end of the ordering, and orders
122 /// all other variables before in a fill-reducing ordering. If \c forceOrder is true, the
123 /// variables in \c constrainLast will be ordered in the same order specified in the KeyVector
124 /// \c constrainLast. If \c forceOrder is false, the variables in \c constrainLast will be
125 /// ordered after all the others, but will be rearranged by CCOLAMD to reduce fill-in as well.
126 static Ordering ColamdConstrainedLast(
127 const VariableIndex& variableIndex, const KeyVector& constrainLast,
128 bool forceOrder = false);
129
130 /// Compute a fill-reducing ordering using constrained COLAMD from a factor graph (see details
131 /// for note on performance). This internally builds a VariableIndex so if you already have a
132 /// VariableIndex, it is faster to use COLAMD(const VariableIndex&). This function constrains
133 /// the variables in \c constrainLast to the end of the ordering, and orders all other variables
134 /// before in a fill-reducing ordering. If \c forceOrder is true, the variables in \c
135 /// constrainFirst will be ordered in the same order specified in the KeyVector \c
136 /// constrainFirst. If \c forceOrder is false, the variables in \c constrainFirst will be
137 /// ordered before all the others, but will be rearranged by CCOLAMD to reduce fill-in as well.
138 template<class FACTOR_GRAPH>
139 static Ordering ColamdConstrainedFirst(const FACTOR_GRAPH& graph,
140 const KeyVector& constrainFirst, bool forceOrder = false) {
141 if (graph.empty())
142 return Ordering();
143 else
144 return ColamdConstrainedFirst(variableIndex: VariableIndex(graph), constrainFirst, forceOrder);
145 }
146
147 /// Compute a fill-reducing ordering using constrained COLAMD from a VariableIndex. This
148 /// function constrains the variables in \c constrainFirst to the front of the ordering, and
149 /// orders all other variables after in a fill-reducing ordering. If \c forceOrder is true, the
150 /// variables in \c constrainFirst will be ordered in the same order specified in the
151 /// KeyVector \c constrainFirst. If \c forceOrder is false, the variables in \c
152 /// constrainFirst will be ordered before all the others, but will be rearranged by CCOLAMD to
153 /// reduce fill-in as well.
154 static Ordering ColamdConstrainedFirst(
155 const VariableIndex& variableIndex,
156 const KeyVector& constrainFirst, bool forceOrder = false);
157
158 /// Compute a fill-reducing ordering using constrained COLAMD from a factor graph (see details
159 /// for note on performance). This internally builds a VariableIndex so if you already have a
160 /// VariableIndex, it is faster to use COLAMD(const VariableIndex&). In this function, a group
161 /// for each variable should be specified in \c groups, and each group of variables will appear
162 /// in the ordering in group index order. \c groups should be a map from Key to group index.
163 /// The group indices used should be consecutive starting at 0, but may appear in \c groups in
164 /// arbitrary order. Any variables not present in \c groups will be assigned to group 0. This
165 /// function simply fills the \c cmember argument to CCOLAMD with the supplied indices, see the
166 /// CCOLAMD documentation for more information.
167 template<class FACTOR_GRAPH>
168 static Ordering ColamdConstrained(const FACTOR_GRAPH& graph,
169 const FastMap<Key, int>& groups) {
170 if (graph.empty())
171 return Ordering();
172 else
173 return ColamdConstrained(variableIndex: VariableIndex(graph), groups);
174 }
175
176 /// Compute a fill-reducing ordering using constrained COLAMD from a VariableIndex. In this
177 /// function, a group for each variable should be specified in \c groups, and each group of
178 /// variables will appear in the ordering in group index order. \c groups should be a map from
179 /// Key to group index. The group indices used should be consecutive starting at 0, but may
180 /// appear in \c groups in arbitrary order. Any variables not present in \c groups will be
181 /// assigned to group 0. This function simply fills the \c cmember argument to CCOLAMD with the
182 /// supplied indices, see the CCOLAMD documentation for more information.
183 static Ordering ColamdConstrained(
184 const VariableIndex& variableIndex, const FastMap<Key, int>& groups);
185
186 /// Return a natural Ordering. Typically used by iterative solvers
187 template<class FACTOR_GRAPH>
188 static Ordering Natural(const FACTOR_GRAPH &fg) {
189 KeySet src = fg.keys();
190 KeyVector keys(src.begin(), src.end());
191 std::stable_sort(first: keys.begin(), last: keys.end());
192 return Ordering(keys.begin(), keys.end());
193 }

195 /// METIS Formatting function
196 template<class FACTOR_GRAPH>
197 static void CSRFormat(std::vector<int>& xadj,
198 std::vector<int>& adj, const FACTOR_GRAPH& graph);

200 /// Compute an ordering determined by METIS from a VariableIndex
201 static Ordering Metis(const MetisIndex& met);
202
203 template<class FACTOR_GRAPH>
204 static Ordering Metis(const FACTOR_GRAPH& graph) {
205 if (graph.empty())
206 return Ordering();
207 else
208 return Metis(met: MetisIndex(graph));
209 }

211 /// @}
212
213 /// @name Named Constructors
214 /// @{
215
216 template<class FACTOR_GRAPH>
217 static Ordering Create(OrderingType orderingType,
218 const FACTOR_GRAPH& graph) {
219 if (graph.empty())
220 return Ordering();
221
222 switch (orderingType) {
223 case COLAMD:
224 return Colamd(graph);
225 case METIS:
226 return Metis(graph);
227 case NATURAL:
228 return Natural(graph);
229 case CUSTOM:
230 throw std::runtime_error(
231 "Ordering::Create error: called with CUSTOM ordering type.");
232 default:
233 throw std::runtime_error(
234 "Ordering::Create error: called with unknown ordering type.");
235 }
236 }
