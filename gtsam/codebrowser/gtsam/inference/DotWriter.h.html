30namespace gtsam {
31
32/**
33 * @brief DotWriter is a helper class for writing graphviz .dot files.
34 * @ingroup inference
35 */
36struct GTSAM_EXPORT DotWriter {
37 double figureWidthInches; ///< The figure width on paper in inches
38 double figureHeightInches; ///< The figure height on paper in inches
39 bool plotFactorPoints; ///< Plots each factor as a dot between the variables
40 bool connectKeysToFactor; ///< Draw a line from each key within a factor to
41 ///< the dot of the factor
42 bool binaryEdges; ///< just use non-dotted edges for binary factors
43
44 /**
45 * Variable positions can be optionally specified and will be included in the
46 * dot file with a "!' sign, so "neato" can use it to render them.
47 */
48 std::map<Key, Vector2> variablePositions;
49
50 /**
51 * The position hints allow one to use symbol character and index to specify
52 * position. Unless variable positions are specified, if a hint is present for
53 * a given symbol, it will be used to calculate the positions as (index,hint).
54 */
55 std::map<char, double> positionHints;
56
57 /** A set of keys that will be displayed as a box */
58 std::set<Key> boxes;
59
60 /**
61 * Factor positions can be optionally specified and will be included in the
62 * dot file with a "!' sign, so "neato" can use it to render them.
63 */
64 std::map<size_t, Vector2> factorPositions;
65
66 explicit DotWriter(double figureWidthInches = 5,
67 double figureHeightInches = 5,
68 bool plotFactorPoints = true,
69 bool connectKeysToFactor = true, bool binaryEdges = false)
70 : figureWidthInches(figureWidthInches),
71 figureHeightInches(figureHeightInches),
72 plotFactorPoints(plotFactorPoints),
73 connectKeysToFactor(connectKeysToFactor),
74 binaryEdges(binaryEdges) {}
75
76 /// Write out preamble for graph, including size.
77 void graphPreamble(std::ostream* os) const;
78
79 /// Write out preamble for digraph, including size.
80 void digraphPreamble(std::ostream* os) const;
81
82 /// Create a variable dot fragment.
83 void drawVariable(Key key, const KeyFormatter& keyFormatter,
84 const std::optional<Vector2>& position,
85 std::ostream* os) const;
86
87 /// Create factor dot.
88 static void DrawFactor(size_t i, const std::optional<Vector2>& position,
89 std::ostream* os);
90
91 /// Return variable position or none
92 std::optional<Vector2> variablePos(Key key) const;
93
94 /// Draw a single factor, specified by its index i and its variable keys.
95 void processFactor(size_t i, const KeyVector& keys,
96 const KeyFormatter& keyFormatter,
97 const std::optional<Vector2>& position,
98 std::ostream* os) const;
99};
100
101} // namespace gtsam
