19#pragma once
20
21#include <gtsam/global_includes.h>
22
23namespace gtsam {
24
25/**
26 * A Bayes tree with an update methods that implements the iSAM algorithm.
27 * Given a set of new factors, it re-eliminates the invalidated part of the
28 * tree. \nosubgrouping
29 */
30template <class BAYESTREE>
31class ISAM : public BAYESTREE {
32 public:
33 typedef BAYESTREE Base;
34 typedef typename Base::BayesNetType BayesNetType;
35 typedef typename Base::FactorGraphType FactorGraphType;
36 typedef typename Base::Clique Clique;
37 typedef typename Base::sharedClique sharedClique;
38 typedef typename Base::Cliques Cliques;
39
40 private:
41 typedef typename Base::Eliminate Eliminate;
42 typedef typename Base::EliminationTraitsType EliminationTraitsType;
43
44 public:
45 /// @name Standard Constructors
46 /// @{
47
48 /** Create an empty Bayes Tree */
49 ISAM() {}
50
51 /** Copy constructor */
52 explicit ISAM(const Base& bayesTree) : Base(bayesTree) {}
53
54 /// @}
55 /// @name Advanced Interface Interface
56 /// @{
57
58 /**
59 * update the Bayes tree with a set of new factors, typically derived from
60 * measurements
61 * @param newFactors is a factor graph that contains the new factors
62 * @param function an elimination routine
63 */
64 void update(
65 const FactorGraphType& newFactors,
66 const Eliminate& function = EliminationTraitsType::DefaultEliminate);
67
68 /** updateInternal provides access to list of orphans for drawing purposes
69 */
70 void updateInternal(
71 const FactorGraphType& newFactors, Cliques* orphans,
72 const Eliminate& function = EliminationTraitsType::DefaultEliminate);
73
74 /// @}
75};
76
77} // namespace gtsam
