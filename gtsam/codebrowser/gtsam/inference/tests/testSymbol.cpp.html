27// A custom (nonsensical) formatter.
28string symbolMyFormatter(Key key) {
29 return "special";
30}
31
32TEST(Symbol, Formatting) {
33 Symbol symbol('c', 3);
34
35 // use key_formatter with a function pointer
36 stringstream ss2;
37 ss2 << key_formatter(symbolMyFormatter) << symbol;
38 EXPECT("special" == ss2.str());
39
40 // use key_formatter with a function object.
41 stringstream ss3;
42 ss3 << key_formatter(MultiRobotKeyFormatter) << symbol;
43 EXPECT("c3" == ss3.str());
44}
