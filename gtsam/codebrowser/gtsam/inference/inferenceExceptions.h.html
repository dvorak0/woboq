27/** An inference algorithm was called with inconsistent arguments. The factor
28 * graph, ordering, or variable index were inconsistent with each other, or a
29 * full elimination routine was called with an ordering that does not include
30 * all of the variables. */
31class InconsistentEliminationRequested : public std::exception {
32 KeyVector keys_;
33 const KeyFormatter& keyFormatter = DefaultKeyFormatter;
34
35 public:
36 InconsistentEliminationRequested() noexcept {}
37
38 InconsistentEliminationRequested(
39 const KeySet& keys,
40 const KeyFormatter& key_formatter = DefaultKeyFormatter);
41
42 ~InconsistentEliminationRequested() noexcept override {}
43
44 const char* what() const noexcept override;
45};
46} // namespace gtsam
