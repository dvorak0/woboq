33 /**
34 * An elimination tree is a data structure used intermediately during
35 * elimination. In future versions it will be used to save work between
36 * multiple eliminations.
37 *
38 * When a variable is eliminated, a new factor is created by combining that
39 * variable's neighboring factors. The new combined factor involves the combined
40 * factors' involved variables. When the lowest-ordered one of those variables
41 * is eliminated, it consumes that combined factor. In the elimination tree,
42 * that lowest-ordered variable is the parent of the variable that was eliminated to
43 * produce the combined factor. This yields a tree in general, and not a chain
44 * because of the implicit sparse structure of the resulting Bayes net.
45 *
46 * This structure is examined even more closely in a JunctionTree, which
47 * additionally identifies cliques in the chordal Bayes net.
48 * \nosubgrouping
49 */
50 template<class BAYESNET, class GRAPH>
51 class EliminationTree
52 {
53 protected:
54 typedef EliminationTree<BAYESNET, GRAPH> This; ///< This class
55 typedef std::shared_ptr<This> shared_ptr; ///< Shared pointer to this class
56
57 public:
58 typedef GRAPH FactorGraphType; ///< The factor graph type
59 typedef typename GRAPH::FactorType FactorType; ///< The type of factors
60 typedef typename std::shared_ptr<FactorType> sharedFactor; ///< Shared pointer to a factor
61 typedef BAYESNET BayesNetType; ///< The BayesNet corresponding to FACTOR
62 typedef typename BayesNetType::ConditionalType ConditionalType; ///< The type of conditionals
63 typedef typename std::shared_ptr<ConditionalType> sharedConditional; ///< Shared pointer to a conditional
64 typedef typename GRAPH::Eliminate Eliminate;
65
66 struct Node {
67 typedef FastVector<sharedFactor> Factors;
68 typedef FastVector<std::shared_ptr<Node> > Children;
69
70 Key key; ///< key associated with root
71 Factors factors; ///< factors associated with root
72 Children children; ///< sub-trees
73
74 sharedFactor eliminate(const std::shared_ptr<BayesNetType>& output,
75 const Eliminate& function, const FastVector<sharedFactor>& childrenFactors) const;
76
77 void print(const std::string& str, const KeyFormatter& keyFormatter) const;
78 };
79
80 typedef std::shared_ptr<Node> sharedNode; ///< Shared pointer to Node
81
82 protected:
83 /** concept check */
84 GTSAM_CONCEPT_TESTABLE_TYPE(FactorType)
85
86 FastVector<sharedNode> roots_;
87 FastVector<sharedFactor> remainingFactors_;
88
89 /// @name Standard Constructors
90 /// @{
91
92 /**
93 * Build the elimination tree of a factor graph using pre-computed column structure.
94 * @param factorGraph The factor graph for which to build the elimination tree
95 * @param structure The set of factors involving each variable. If this is not
96 * precomputed, you can call the Create(const FactorGraph<DERIVEDFACTOR>&)
97 * named constructor instead.
98 * @return The elimination tree
99 */
100 EliminationTree(const FactorGraphType& factorGraph,
101 const VariableIndex& structure, const Ordering& order);
102
103 /** Build the elimination tree of a factor graph. Note that this has to compute the column
104 * structure as a VariableIndex, so if you already have this precomputed, use the other
105 * constructor instead.
106 * @param factorGraph The factor graph for which to build the elimination tree
107 */
108 EliminationTree(const FactorGraphType& factorGraph, const Ordering& order);
109
110 /** Copy constructor - makes a deep copy of the tree structure, but only pointers to factors are
111 * copied, factors are not cloned. */
112 EliminationTree(const This& other) { *this = other; }
113
114 /** Assignment operator - makes a deep copy of the tree structure, but only pointers to factors
115 * are copied, factors are not cloned. */
116 This& operator=(const This& other);
117
118 /// @}
119
120 protected:
121 ~EliminationTree();
122
123 public:
124 /// @name Standard Interface
125 /// @{
126
127 /** Eliminate the factors to a Bayes net and remaining factor graph
128 * @param function The function to use to eliminate, see the namespace functions
129 * in GaussianFactorGraph.h
130 * @return The Bayes net and factor graph resulting from elimination
131 */
132 std::pair<std::shared_ptr<BayesNetType>, std::shared_ptr<FactorGraphType> >
133 eliminate(Eliminate function) const;
134
135 /// @}
136 /// @name Testable
137 /// @{
138
139 /** Print the tree to cout */
140 void print(const std::string& name = "EliminationTree: ",
141 const KeyFormatter& formatter = DefaultKeyFormatter) const;
142
143 protected:
144 /** Test whether the tree is equal to another */
145 bool equals(const This& other, double tol = 1e-9) const;
146
147 /// @}
148
149 public:
150 /// @name Advanced Interface
151 /// @{
152
153 /** Return the set of roots (one for a tree, multiple for a forest) */
154 const FastVector<sharedNode>& roots() const { return roots_; }
155
156 /** Return the remaining factors that are not pulled into elimination */
157 const FastVector<sharedFactor>& remainingFactors() const { return remainingFactors_; }
158
159 /** Swap the data of this tree with another one, this operation is very fast. */
160 void swap(This& other);
161
162 protected:
163 /// Protected default constructor
164 EliminationTree() {}
165
166 private:
167 /// Allow access to constructor and add methods for testing purposes
168 friend class ::EliminationTreeTester;
169 };
170
171}
