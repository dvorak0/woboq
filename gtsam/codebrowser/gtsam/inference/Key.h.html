34/// Typedef for a function to format a key, i.e. to convert it to a string
35using KeyFormatter = std::function<std::string(Key)>;
36
37// Helper function for DefaultKeyFormatter
38GTSAM_EXPORT std::string _defaultKeyFormatter(Key key);
39
40/**
41 * The default KeyFormatter, which is used if no KeyFormatter is passed
42 * to a 'print' function.
43 *
44 * Automatically detects plain integer keys and Symbol keys.
45 *
46 * Marked as `extern` so that it can be updated by external libraries.
47 *
48 */
49extern GTSAM_EXPORT KeyFormatter DefaultKeyFormatter;
50
51// Helper function for Multi-robot Key Formatter
52GTSAM_EXPORT std::string _multirobotKeyFormatter(gtsam::Key key);
53
54///
55/// A KeyFormatter that will check for LabeledSymbol keys, as well as Symbol and plain
56/// integer keys. This keyformatter will need to be passed in to override the default
57/// formatter in print functions.
58///
59/// Checks for LabeledSymbol, Symbol and then plain keys, in order.
60extern GTSAM_EXPORT KeyFormatter MultiRobotKeyFormatter;
61
62/// To use the key_formatter on Keys, they must be wrapped in a StreamedKey.
63struct StreamedKey {
64 const Key &key_;
65 explicit StreamedKey(const Key &key) : key_(key) {}
66 GTSAM_EXPORT friend std::ostream &operator<<(std::ostream &, const StreamedKey &);
67};
68
69/**
70 * Output stream manipulator that will format gtsam::Keys according to the given
71 * KeyFormatter, as long as Key values are wrapped in a gtsam::StreamedKey.
72 * LabeledSymbol and Symbol values do not have to be wrapped.
73 * usage:
74 * Key key = LabeledSymbol('x', 'A', 5); // cast to key type
75 * cout << key_formatter(MultiRobotKeyFormatter) << StreamedKey(key);
76 */
77class key_formatter {
78 public:
79 explicit key_formatter(KeyFormatter v) : formatter_(v) {}
80 GTSAM_EXPORT friend std::ostream &operator<<(std::ostream &, const key_formatter &);
81 GTSAM_EXPORT friend std::ostream &operator<<(std::ostream &, const StreamedKey &);
82
83 private:
84 KeyFormatter formatter_;
85 static void *&property(std::ios_base &s);
86 static void set_property(std::ios_base &s, const KeyFormatter &f);
87 static KeyFormatter *get_property(std::ios_base &s);
88};
