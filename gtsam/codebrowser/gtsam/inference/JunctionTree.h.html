30 /**
31 * A JunctionTree is a cluster tree, a set of variable clusters with factors, arranged in a tree,
32 * with the additional property that it represents the clique tree associated with a Bayes Net.
33 *
34 * In GTSAM a junction tree is an intermediate data structure in multifrontal variable
35 * elimination. Each node is a cluster of factors, along with a clique of variables that are
36 * eliminated all at once. In detail, every node k represents a clique (maximal fully connected
37 * subset) of an associated chordal graph, such as a chordal Bayes net resulting from elimination.
38 *
39 * The difference with the BayesTree is that a JunctionTree stores factors, whereas a
40 * BayesTree stores conditionals, that are the product of eliminating the factors in the
41 * corresponding JunctionTree cliques.
42 *
43 * The tree structure and elimination method are exactly analagous to the EliminationTree,
44 * except that in the JunctionTree, at each node multiple variables are eliminated at a time.
45 *
46 * \ingroup Multifrontal
47 * \nosubgrouping
48 */
49 template<class BAYESTREE, class GRAPH>
50 class JunctionTree : public EliminatableClusterTree<BAYESTREE, GRAPH> {
51
52 public:
53
54 typedef JunctionTree<BAYESTREE, GRAPH> This; ///< This class
55 typedef std::shared_ptr<This> shared_ptr; ///< Shared pointer to this class
56 typedef EliminatableClusterTree<BAYESTREE, GRAPH> Base; ///< Our base class
57
58 protected:
59
60 /// @name Standard Constructors
61 /// @{
62
63 /** Build the junction tree from an elimination tree. */
64 template<class ETREE>
65 static This FromEliminationTree(const ETREE& eliminationTree) { return This(eliminationTree); }
66
67 /** Build the junction tree from an elimination tree. */
68 template<class ETREE_BAYESNET, class ETREE_GRAPH>
69 JunctionTree(const EliminationTree<ETREE_BAYESNET, ETREE_GRAPH>& eliminationTree);
70
71 /// @}
72
73 protected:
74
75 // Private default constructor (used in static construction methods)
76 JunctionTree() {}
77
78 };
