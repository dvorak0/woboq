23namespace gtsam {

25 class HybridValues; // forward declaration.

27 /**
28 * This is the base class for all conditional distributions/densities,
29 * which are implemented as specialized factors. This class does not store any
30 * data other than its keys. Derived classes store data such as matrices and
31 * probability tables.
32 *
33 * The `evaluate` method is used to evaluate the factor, and together with
34 * `logProbability` is the main methods that need to be implemented in derived
35 * classes. These two methods relate to the `error` method in the factor by:
36 * probability(x) = k exp(-error(x))
37 * where k is a normalization constant making
38 * \int probability(x) = \int k exp(-error(x)) == 1.0, and
39 * logProbability(x) = -(K + error(x))
40 * i.e., K = -log(k). The normalization constant k is assumed to *not* depend
41 * on any argument, only (possibly) on the conditional parameters.
42 * This class provides a default negative log normalization constant
43 * negLogConstant() == 0.0.
44 *
45 * There are four broad classes of conditionals that derive from Conditional:
46 *
47 * - \b Gaussian conditionals, implemented in \class GaussianConditional, a
48 * Gaussian density over a set of continuous variables.
49 * - \b Discrete conditionals, implemented in \class DiscreteConditional, which
50 * represent a discrete conditional distribution over discrete variables.
51 * - \b Hybrid conditional densities, such as \class HybridGaussianConditional, which is
52 * a density over continuous variables given discrete/continuous parents.
53 * - \b Symbolic factors, used to represent a graph structure, implemented in
54 * \class SymbolicConditional. Only used for symbolic elimination etc.
55 *
56 * Derived classes *must* redefine the Factor and shared_ptr typedefs to refer
57 * to the associated factor type and shared_ptr type of the derived class. See
58 * SymbolicConditional and GaussianConditional for examples.
59 *
60 * \nosubgrouping
61 */
62 template<class FACTOR, class DERIVEDCONDITIONAL>
63 class Conditional
64 {
65 protected:
66 /** The first nrFrontal variables are frontal and the rest are parents. */
67 size_t nrFrontals_;
68
69 private:
70 /// Typedef to this class
71 typedef Conditional<FACTOR,DERIVEDCONDITIONAL> This;
72
73
74 public:
75 /** A mini implementation of an iterator range, to share const
76 * views of frontals and parents. */
77 typedef std::pair<typename FACTOR::const_iterator, typename FACTOR::const_iterator> ConstFactorRange;
78 struct ConstFactorRangeIterator {
79 ConstFactorRange range_;
80 // Delete default constructor
81 ConstFactorRangeIterator() = delete;
82 ConstFactorRangeIterator(ConstFactorRange const& x) : range_(x) {}
83 // Implement begin and end for iteration
84 typename FACTOR::const_iterator begin() const { return range_.first; }
85 typename FACTOR::const_iterator end() const { return range_.second; }
86 size_t size() const { return std::distance(range_.first, range_.second); }
87 const auto& front() const { return *begin(); }
88 // == operator overload for comparison with another iterator
89 template <class OTHER>
90 bool operator==(const OTHER& rhs) const {
91 if (this->size() != rhs.size()) return false;
92 if (this->size() == 0) return true;
93 return std::equal(begin(), end(), rhs.begin(), rhs.end());
94 }
95 };
96
97 /** View of the frontal keys (call frontals()) */
98 typedef ConstFactorRangeIterator Frontals;
99
100 /** View of the separator keys (call parents()) */
101 typedef ConstFactorRangeIterator Parents;
102
103 protected:
104 /// @name Standard Constructors
105 /// @{
106
107 /** Empty Constructor to make serialization possible */
108 Conditional() : nrFrontals_(0) {}
109
110 /** Constructor */
111 Conditional(size_t nrFrontals) : nrFrontals_(nrFrontals) {}
112
113 /// @}
114
115 public:
116 /// @name Testable
117 /// @{
118
119 /** print with optional formatter */
120 void print(const std::string& s = "Conditional", const KeyFormatter& formatter = DefaultKeyFormatter) const;
121
122 /** check equality */
123 bool equals(const This& c, double tol = 1e-9) const;
124
125 /// @}
126
127 /// @name Standard Interface
128 /// @{
129
130 virtual ~Conditional() {}
131
132 /** return the number of frontals */
133 size_t nrFrontals() const { return nrFrontals_; }
134
135 /** return the number of parents */
136 size_t nrParents() const { return asFactor().size() - nrFrontals_; }
137
138 /** Convenience function to get the first frontal key */
139 Key firstFrontalKey() const {
140 if(nrFrontals_ > 0)
141 return asFactor().front();
142 else
143 throw std::invalid_argument("Requested Conditional::firstFrontalKey from a conditional with zero frontal keys");
144 }
145
146 /** return a view of the frontal keys */
147 Frontals frontals() const { return ConstFactorRangeIterator({beginFrontals(), endFrontals()});}
148
149 /** return a view of the parent keys */
150 Parents parents() const { return ConstFactorRangeIterator({beginParents(), endParents()}); }
151
152 /**
153 * All conditional types need to implement a `logProbability` function, for which
154 * exp(logProbability(x)) = evaluate(x).
155 */
156 virtual double logProbability(const HybridValues& c) const;
157
158 /**
159 * All conditional types need to implement an `evaluate` function, that yields
160 * a true probability. The default implementation just exponentiates logProbability.
161 */
162 virtual double evaluate(const HybridValues& c) const;
163
164 /// Evaluate probability density, sugar.
165 double operator()(const HybridValues& x) const {
166 return evaluate(c: x);
167 }
168
169 /**
170 * @brief All conditional types need to implement this as the negative log
171 * of the normalization constant to make it such that error>=0.
172 *
173 * @return double
174 */
175 virtual double negLogConstant() const;
176
177 /// @}
178 /// @name Advanced Interface
179 /// @{
180
181 /** Iterator pointing to first frontal key. */
182 typename FACTOR::const_iterator beginFrontals() const { return asFactor().begin(); }
183
184 /** Iterator pointing past the last frontal key. */
185 typename FACTOR::const_iterator endFrontals() const { return asFactor().begin() + nrFrontals_; }
186
187 /** Iterator pointing to the first parent key. */
188 typename FACTOR::const_iterator beginParents() const { return endFrontals(); }
189
190 /** Iterator pointing past the last parent key. */
191 typename FACTOR::const_iterator endParents() const { return asFactor().end(); }
192
193 /** Mutable version of nrFrontals */
194 size_t& nrFrontals() { return nrFrontals_; }
195
196 /** Mutable iterator pointing to first frontal key. */
197 typename FACTOR::iterator beginFrontals() { return asFactor().begin(); }
198
199 /** Mutable iterator pointing past the last frontal key. */
200 typename FACTOR::iterator endFrontals() { return asFactor().begin() + nrFrontals_; }
201
202 /** Mutable iterator pointing to the first parent key. */
203 typename FACTOR::iterator beginParents() { return asFactor().begin() + nrFrontals_; }
204
205 /** Mutable iterator pointing past the last parent key. */
206 typename FACTOR::iterator endParents() { return asFactor().end(); }
207
208 /**
209 * Check invariants of this conditional, given the values `x`.
210 * It tests:
211 * - evaluate >= 0.0
212 * - evaluate(x) == conditional(x)
213 * - exp(logProbability(x)) == evaluate(x)
214 * - negLogConstant() = -log(normalizationConstant())
215 * - error >= 0.0
216 * - logProbability(x) == -(negLogConstant() + error(x))
217 *
218 * @param conditional The conditional to test, as a reference to the derived type.
219 * @tparam VALUES HybridValues, or a more narrow type like DiscreteValues.
220 */
221 template <class VALUES>
222 static bool CheckInvariants(const DERIVEDCONDITIONAL& conditional,
223 const VALUES& x);
224
225 /// @}
226
227 private:
228
229 /// @name Serialization
230 /// @{
231
232 // Cast to factor type (non-const) (casts down to derived conditional type, then up to factor type)
233 FACTOR& asFactor() { return static_cast<FACTOR&>(static_cast<DERIVEDCONDITIONAL&>(*this)); }
234
235 // Cast to derived type (const) (casts down to derived conditional type, then up to factor type)
236 const FACTOR& asFactor() const { return static_cast<const FACTOR&>(static_cast<const DERIVEDCONDITIONAL&>(*this)); }
237
238#if GTSAM_ENABLE_BOOST_SERIALIZATION
239 /** Serialization function */
240 friend class boost::serialization::access;
241 template<class ARCHIVE>
242 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
243 ar & BOOST_SERIALIZATION_NVP(nrFrontals_);
244 }
245#endif
246
247 /// @}
248
249 };
