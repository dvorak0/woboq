27namespace gtsam {

29/**
30 * Customized version of gtsam::Symbol for multi-robot use.
31 *
32 * This variation of Symbol stores two char values in addition to
33 * an integer key, which is useful for encoding a group for a
34 * variable. This was originally designed for multi-robot systems,
35 * which allows expressing "Pose 7 from robot B" as "xB7".
36 */
37class GTSAM_EXPORT LabeledSymbol {
38 protected:
39 unsigned char c_, label_;
40 std::uint64_t j_;
41
42 public:
43 /// @name Constructors
44 /// @{
45
46 /// Default constructor
47 LabeledSymbol();
48
49 /// Copy constructor
50 LabeledSymbol(const LabeledSymbol& key);
51
52 /// Constructor fro characters c and label, and integer j
53 LabeledSymbol(unsigned char c, unsigned char label, std::uint64_t j);
54
55 /// Constructor that decodes an integer Key
56 LabeledSymbol(Key key);
57
58 /// @}
59 /// @name Testable
60 /// @{
61
62 /// Prints the LabeledSymbol with an optional prefix string.
63 void print(const std::string& s = "") const;
64
65 /// Checks if this LabeledSymbol is equal to another, tolerance is ignored.
66 bool equals(const LabeledSymbol& expected, double tol = 0.0) const {
67 return (*this) == expected;
68 }
69
70 /// @}
71 /// @name Standard API
72 /// @{
73
74 /// Cast to Key
75 operator Key() const;
76
77 /// return the integer version
78 Key key() const { return (Key) * this; }
79
80 /// Retrieve label character
81 inline unsigned char label() const { return label_; }
82
83 /// Retrieve key character
84 inline unsigned char chr() const { return c_; }
85
86 /// Retrieve key index
87 inline size_t index() const { return j_; }
88
89 /// Create a string from the key
90 operator std::string() const;
91
92 /// Output stream operator that can be used with key_formatter (see Key.h).
93 friend GTSAM_EXPORT std::ostream& operator<<(std::ostream&,
94 const LabeledSymbol&);
95
96 /// @}
97 /// @name Comparison
98 /// @{
99
100 bool operator<(const LabeledSymbol& comp) const;
101 bool operator==(const LabeledSymbol& comp) const;
102 bool operator==(Key comp) const;
103 bool operator!=(const LabeledSymbol& comp) const;
104 bool operator!=(Key comp) const;
105
106 /// @}
107 /// @name Filtering
108 /// @{
109 /// Return a filter function that returns true when evaluated on a Key whose
110 /// character (when converted to a LabeledSymbol) matches \c c. Use this with
111 /// the Values::filter() function to retrieve all key-value pairs with the
112 /// requested character.
113
114 /// Checks only the type
115 static std::function<bool(Key)> TypeTest(unsigned char c);
116
117 /// Checks only the robot ID (label_)
118 static std::function<bool(Key)> LabelTest(unsigned char label);
119
120 /// Checks both type and the robot ID
121 static std::function<bool(Key)> TypeLabelTest(unsigned char c,
122 unsigned char label);
123
124 /// @}
125 /// @name Advanced API
126 /// @{
127
128 /// Converts to upper/lower versions of labels
129 LabeledSymbol upper() const { return LabeledSymbol(c_, toupper(c: label_), j_); }
130 LabeledSymbol lower() const { return LabeledSymbol(c_, tolower(c: label_), j_); }
131
132 /// Create a new symbol with a different character.
133 LabeledSymbol newChr(unsigned char c) const {
134 return LabeledSymbol(c, label_, j_);
135 }
136
137 /// Create a new symbol with a different label.
138 LabeledSymbol newLabel(unsigned char label) const {
139 return LabeledSymbol(c_, label, j_);
140 }
141
142 /// @}
143
144 private:
145#if GTSAM_ENABLE_BOOST_SERIALIZATION
146 /// Serialization function
147 friend class boost::serialization::access;
148 template <class ARCHIVE>
149 void serialize(ARCHIVE& ar, const unsigned int /*version*/) {
150 ar& BOOST_SERIALIZATION_NVP(c_);
151 ar& BOOST_SERIALIZATION_NVP(label_);
152 ar& BOOST_SERIALIZATION_NVP(j_);
153 }
154#endif
155}; // \class LabeledSymbol
156
157/// Create a symbol key from a character, label and index, i.e. xA5.
158inline Key mrsymbol(unsigned char c, unsigned char label, size_t j) {
159 return (Key)LabeledSymbol(c, label, j);
160}
161
162/// Return the character portion of a symbol key.
163inline unsigned char mrsymbolChr(Key key) { return LabeledSymbol(key).chr(); }
164
165/// Return the label portion of a symbol key.
166inline unsigned char mrsymbolLabel(Key key) {
167 return LabeledSymbol(key).label();
168}
169
170/// Return the index portion of a symbol key.
171inline size_t mrsymbolIndex(Key key) { return LabeledSymbol(key).index(); }
172
173/// traits
174template <>
175struct traits<LabeledSymbol> : public Testable<LabeledSymbol> {};
