55 /**
56 * Bayes tree
57 * @tparam CONDITIONAL The type of the conditional densities, i.e. the type of node in the underlying Bayes chain,
58 * which could be a ConditionalProbabilityTable, a GaussianConditional, or a SymbolicConditional.
59 * @tparam CLIQUE The type of the clique data structure, defaults to BayesTreeClique, normally do not change this
60 * as it is only used when developing special versions of BayesTree, e.g. for ISAM2.
61 *
62 * \ingroup Multifrontal
63 * \nosubgrouping
64 */
65 template<class CLIQUE>
66 class BayesTree
67 {
68 protected:
69 typedef BayesTree<CLIQUE> This;
70 typedef std::shared_ptr<This> shared_ptr;
71
72 public:
73 typedef CLIQUE Clique; ///< The clique type, normally BayesTreeClique
74 typedef std::shared_ptr<Clique> sharedClique; ///< Shared pointer to a clique
75 typedef Clique Node; ///< Synonym for Clique (TODO: remove)
76 typedef sharedClique sharedNode; ///< Synonym for sharedClique (TODO: remove)
77 typedef typename CLIQUE::ConditionalType ConditionalType;
78 typedef std::shared_ptr<ConditionalType> sharedConditional;
79 typedef typename CLIQUE::BayesNetType BayesNetType;
80 typedef std::shared_ptr<BayesNetType> sharedBayesNet;
81 typedef typename CLIQUE::FactorType FactorType;
82 typedef std::shared_ptr<FactorType> sharedFactor;
83 typedef typename CLIQUE::FactorGraphType FactorGraphType;
84 typedef std::shared_ptr<FactorGraphType> sharedFactorGraph;
85 typedef typename FactorGraphType::Eliminate Eliminate;
86 typedef typename CLIQUE::EliminationTraitsType EliminationTraitsType;
87
88 /** A convenience class for a list of shared cliques */
89 typedef FastList<sharedClique> Cliques;
90
91 /** Map from keys to Clique */
92 typedef ConcurrentMap<Key, sharedClique> Nodes;
93
94 /** Root cliques */
95 typedef FastVector<sharedClique> Roots;
96
97 protected:
98
99 /** Map from indices to Clique */
100 Nodes nodes_;
101
102 /** Root cliques */
103 Roots roots_;
104
105 /// @name Standard Constructors
106 /// @{
107
108 /** Create an empty Bayes Tree */
109 BayesTree() {}
110
111 /** Copy constructor */
112 BayesTree(const This& other);
113
114 /// @}
115
116 /** Destructor */
117 ~BayesTree();
118
119 /** Assignment operator */
120 This& operator=(const This& other);
121
122 public:
123
124 /// @name Testable
125 /// @{
126
127 /** check equality */
128 bool equals(const This& other, double tol = 1e-9) const;
129
130 /** print */
131 void print(const std::string& s = "",
132 const KeyFormatter& keyFormatter = DefaultKeyFormatter) const;
133 /// @}
134
135 /// @name Standard Interface
136 /// @{
137
138 /** number of cliques */
139 size_t size() const;
140
141 /** Check if there are any cliques in the tree */
142 inline bool empty() const {
143 return nodes_.empty();
144 }
145
146 /** Return nodes. Each node is a clique of variables obtained after elimination. */
147 const Nodes& nodes() const { return nodes_; }
148
149 /** Access node by variable */
150 sharedClique operator[](Key j) const { return nodes_.at(j); }
151
152 /** return root cliques */
153 const Roots& roots() const { return roots_; }
154
155 /** alternate syntax for matlab: find the clique that contains the variable with Key j */
156 const sharedClique& clique(Key j) const {
157 typename Nodes::const_iterator c = nodes_.find(j);
158 if(c == nodes_.end())
159 throw std::out_of_range("Requested the BayesTree clique for a key that is not in the BayesTree");
160 else
161 return c->second;
162 }
163
164 /** Gather data on all cliques */
165 BayesTreeCliqueData getCliqueData() const;
166
167 /** Collect number of cliques with cached separator marginals */
168 size_t numCachedSeparatorMarginals() const;
169
170 /** Return marginal on any variable. Note that this actually returns a conditional, for which a
171 * solution may be directly obtained by calling .solve() on the returned object.
172 * Alternatively, it may be directly used as its factor base class. For example, for Gaussian
173 * systems, this returns a GaussianConditional, which inherits from JacobianFactor and
174 * GaussianFactor. */
175 sharedConditional marginalFactor(Key j, const Eliminate& function = EliminationTraitsType::DefaultEliminate) const;
176
177 /**
178 * return joint on two variables
179 * Limitation: can only calculate joint if cliques are disjoint or one of them is root
180 */
181 sharedFactorGraph joint(Key j1, Key j2, const Eliminate& function = EliminationTraitsType::DefaultEliminate) const;
182
183 /**
184 * return joint on two variables as a BayesNet
185 * Limitation: can only calculate joint if cliques are disjoint or one of them is root
186 */
187 sharedBayesNet jointBayesNet(Key j1, Key j2, const Eliminate& function = EliminationTraitsType::DefaultEliminate) const;
188
189 /// @}
190 /// @name Graph Display
191 /// @{
192
193 /// Output to graphviz format, stream version.
194 void dot(std::ostream& os, const KeyFormatter& keyFormatter = DefaultKeyFormatter) const;
195
196 /// Output to graphviz format string.
197 std::string dot(
198 const KeyFormatter& keyFormatter = DefaultKeyFormatter) const;
199
200 /// output to file with graphviz format.
201 void saveGraph(const std::string& filename,
202 const KeyFormatter& keyFormatter = DefaultKeyFormatter) const;
203
204 /// @}
205 /// @name Advanced Interface
206 /// @{
207
208 /**
209 * Find parent clique of a conditional. It will look at all parents and
210 * return the one with the lowest index in the ordering.
211 */
212 template<class CONTAINER>
213 Key findParentClique(const CONTAINER& parents) const;
214
215 /** Remove all nodes */
216 void clear();
217
218 /** Clear all shortcut caches - use before timing on marginal calculation to avoid residual cache data */
219 void deleteCachedShortcuts();
220
221 /**
222 * Remove path from clique to root and return that path as factors
223 * plus a list of orphaned subtree roots. Used in removeTop below.
224 */
225 void removePath(sharedClique clique, BayesNetType* bn, Cliques* orphans);
226
227 /**
228 * Given a list of indices, turn "contaminated" part of the tree back into a factor graph.
229 * Factors and orphans are added to the in/out arguments.
230 */
231 void removeTop(const KeyVector& keys, BayesNetType* bn, Cliques* orphans);
232
233 /**
234 * Remove the requested subtree. */
235 Cliques removeSubtree(const sharedClique& subtree);
236
237 /** Insert a new subtree with known parent clique. This function does not check that the
238 * specified parent is the correct parent. This function updates all of the internal data
239 * structures associated with adding a subtree, such as populating the nodes index. */
240 void insertRoot(const sharedClique& subtree);
241
242 /** add a clique (top down) */
243 void addClique(const sharedClique& clique, const sharedClique& parent_clique = sharedClique());
244
245 /** Add all cliques in this BayesTree to the specified factor graph */
246 void addFactorsToGraph(FactorGraph<FactorType>* graph) const;
247
248 protected:
249
250 /** private helper method for saving the Tree to a text file in GraphViz format */
251 void dot(std::ostream &s, sharedClique clique, const KeyFormatter& keyFormatter,
252 int parentnum = 0) const;
253
254 /** Gather data on a single clique */
255 void getCliqueData(sharedClique clique, BayesTreeCliqueData* stats) const;
256
257 /** remove a clique: warning, can result in a forest */
258 void removeClique(sharedClique clique);
259
260 /** Fill the nodes index for a subtree */
261 void fillNodesIndex(const sharedClique& subtree);
262
263 // Friend JunctionTree because it directly fills roots and nodes index.
264 template<class BAYESTREE, class GRAPH> friend class EliminatableClusterTree;
265
266 private:
267#if GTSAM_ENABLE_BOOST_SERIALIZATION
268 /** Serialization function */
269 friend class boost::serialization::access;
270 template<class ARCHIVE>
271 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
272 ar & BOOST_SERIALIZATION_NVP(nodes_);
273 ar & BOOST_SERIALIZATION_NVP(roots_);
274 }
275#endif
276
277 /// @}
278
279 }; // BayesTree

281 /* ************************************************************************* */
282 template <class CLIQUE, typename = void>
283 class BayesTreeOrphanWrapper : public CLIQUE::ConditionalType {
284 public:
285 typedef CLIQUE CliqueType;
286 typedef typename CLIQUE::ConditionalType Base;
287
288 std::shared_ptr<CliqueType> clique;
289
290 /**
291 * @brief Construct a new Bayes Tree Orphan Wrapper object
292 *
293 * This object stores parent keys in our base type factor so that
294 * eliminating those parent keys will pull this subtree into the
295 * elimination.
296 *
297 * @param clique Orphan clique to add for further consideration in
298 * elimination.
299 */
300 BayesTreeOrphanWrapper(const std::shared_ptr<CliqueType>& clique)
301 : clique(clique) {
302 this->keys_.assign(clique->conditional()->beginParents(),
303 clique->conditional()->endParents());
304 }
305
306 void print(
307 const std::string& s = "",
308 const KeyFormatter& formatter = DefaultKeyFormatter) const override {
309 clique->print(s + "stored clique", formatter);
310 }
311 };
