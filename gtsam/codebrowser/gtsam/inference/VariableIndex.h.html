30namespace gtsam {
31
32/**
33 * The VariableIndex class computes and stores the block column structure of a
34 * factor graph. The factor graph stores a collection of factors, each of
35 * which involves a set of variables. In contrast, the VariableIndex is built
36 * from a factor graph prior to elimination, and stores the list of factors
37 * that involve each variable. This information is stored as a deque of
38 * lists of factor indices.
39 * \nosubgrouping
40 */
41class GTSAM_EXPORT VariableIndex {
42 public:
43 typedef std::shared_ptr<VariableIndex> shared_ptr;
44 typedef FactorIndices::iterator Factor_iterator;
45 typedef FactorIndices::const_iterator Factor_const_iterator;
46
47 protected:
48 typedef FastMap<Key, FactorIndices> KeyMap;
49 KeyMap index_;
50 size_t nFactors_; // Number of factors in the original factor graph.
51 size_t nEntries_; // Sum of involved variable counts of each factor.
52
53 public:
54 typedef KeyMap::const_iterator const_iterator;
55 typedef KeyMap::const_iterator iterator;
56 typedef KeyMap::value_type value_type;
57
58 /// @name Standard Constructors
59 /// @{
60
61 /// Default constructor, creates an empty VariableIndex
62 VariableIndex() : nFactors_(0), nEntries_(0) {}
63
64 /**
65 * Create a VariableIndex that computes and stores the block column structure
66 * of a factor graph.
67 */
68 template <class FG>
69 explicit VariableIndex(const FG& factorGraph) : nFactors_(0), nEntries_(0) {
70 augment(factorGraph);
71 }
72
73 /// @}
74 /// @name Standard Interface
75 /// @{
76
77 /// The number of variable entries. This is equal to the number of unique variable Keys.
78 size_t size() const { return index_.size(); }
79
80 /// The number of factors in the original factor graph
81 size_t nFactors() const { return nFactors_; }
82
83 /// The number of nonzero blocks, i.e. the number of variable-factor entries
84 size_t nEntries() const { return nEntries_; }
85
86 /// Access a list of factors by variable without checking for existence
87 const FactorIndices& operator[](Key variable) const;
88 /// Access a list of factors by variable
89 const FactorIndices& at(Key variable) const;
90
91 /// Return true if no factors associated with a variable
92 bool empty(Key variable) const;
93
94 /// @}
95 /// @name Testable
96 /// @{
97
98 /// Test for equality (for unit tests and debug assertions).
99 bool equals(const VariableIndex& other, double tol=0.0) const;
100
101 /// Print the variable index (for unit tests and debugging).
102 void print(const std::string& str = "VariableIndex: ",
103 const KeyFormatter& keyFormatter = DefaultKeyFormatter) const;
104
105 /**
106 * Output dual hypergraph to Metis file format for use with hmetis
107 * In the dual graph, variables are hyperedges, factors are nodes.
108 */
109 void outputMetisFormat(std::ostream& os) const;
110
111
112 /// @}
113 /// @name Advanced Interface
114 /// @{
115
116 /**
117 * Augment the variable index with new factors. This can be used when
118 * solving problems incrementally.
119 */
120 template<class FG>
121 void augment(const FG& factors, const FactorIndices* newFactorIndices = nullptr);
122
123 /**
124 * An overload of augment() that takes a single factor. and l-value
125 * reference to FactorIndeces.
126 */
127 template<class FG>
128 void augment(const FG& factor, const FactorIndices& newFactorIndices) {
129 augment(factor, &newFactorIndices);
130 }
131
132 /**
133 * Augment the variable index after an existing factor now affects to more
134 * variable Keys. This can be used when solving problems incrementally, with
135 * smart factors or in general with factors with a dynamic number of Keys.
136 */
137 void augmentExistingFactor(const FactorIndex factorIndex, const KeySet & newKeys);
138
139 /**
140 * Remove entries corresponding to the specified factors. NOTE: We intentionally do not decrement
141 * nFactors_ because the factor indices need to remain consistent. Removing factors from a factor
142 * graph does not shift the indices of other factors. Also, we keep nFactors_ one greater than
143 * the highest-numbered factor referenced in a VariableIndex.
144 *
145 * @param indices The indices of the factors to remove, which must match \c factors
146 * @param factors The factors being removed, which must symbolically correspond exactly to the
147 * factors with the specified \c indices that were added.
148 */
149 template<typename ITERATOR, class FG>
150 void remove(ITERATOR firstFactor, ITERATOR lastFactor, const FG& factors);
151
152 /// Remove unused empty variables (in debug mode verifies they are empty).
153 template<typename ITERATOR>
154 void removeUnusedVariables(ITERATOR firstKey, ITERATOR lastKey);
155
156 /// Iterator to the first variable entry
157 const_iterator begin() const { return index_.begin(); }
158
159 /// Iterator to the first variable entry
160 const_iterator end() const { return index_.end(); }
161
162 /// Find the iterator for the requested variable entry
163 const_iterator find(Key key) const { return index_.find(x: key); }
164
165protected:
166 Factor_iterator factorsBegin(Key variable) { return internalAt(variable).begin(); }
167 Factor_iterator factorsEnd(Key variable) { return internalAt(variable).end(); }
168
169 Factor_const_iterator factorsBegin(Key variable) const { return internalAt(variable).begin(); }
170 Factor_const_iterator factorsEnd(Key variable) const { return internalAt(variable).end(); }
171
172 /// Internal version of 'at' that asserts existence
173 const FactorIndices& internalAt(Key variable) const {
174 const KeyMap::const_iterator item = index_.find(x: variable);
175 assert(item != index_.end());
176 return item->second;
177 }
178
179 /// Internal version of 'at' that asserts existence
180 FactorIndices& internalAt(Key variable) {
181 const KeyMap::iterator item = index_.find(x: variable);
182 assert(item != index_.end());
183 return item->second;
184 }
185
186private:
187#if GTSAM_ENABLE_BOOST_SERIALIZATION
188 /** Serialization function */
189 friend class boost::serialization::access;
190 template<class ARCHIVE>
191 void serialize(ARCHIVE & ar, const unsigned int /*version*/) {
192 ar & BOOST_SERIALIZATION_NVP(index_);
193 ar & BOOST_SERIALIZATION_NVP(nFactors_);
194 ar & BOOST_SERIALIZATION_NVP(nEntries_);
195 }
196#endif
197
198 /// @}
199};
200
201/// traits
202template<>
203struct traits<VariableIndex> : public Testable<VariableIndex> {
204};
