18/**
19 * A cluster-tree is associated with a factor graph and is defined as in Koller-Friedman:
20 * each node k represents a subset \f$ C_k \sub X \f$, and the tree is family preserving, in that
21 * each factor \f$ f_i \f$ is associated with a single cluster and \f$ scope(f_i) \sub C_k \f$.
22 * \nosubgrouping
23 */
24template <class GRAPH>
25class ClusterTree {
26 public:
27 typedef GRAPH FactorGraphType; ///< The factor graph type
28 typedef ClusterTree<GRAPH> This; ///< This class
29 typedef std::shared_ptr<This> shared_ptr; ///< Shared pointer to this class
30
31 typedef typename GRAPH::FactorType FactorType; ///< The type of factors
32 typedef std::shared_ptr<FactorType> sharedFactor; ///< Shared pointer to a factor
33
34 /// A Cluster is just a collection of factors
35 // TODO(frank): re-factor JunctionTree so we can make members private
36 struct Cluster {
37 typedef FastVector<std::shared_ptr<Cluster> > Children;
38 Children children; ///< sub-trees
39
40 typedef Ordering Keys;
41 Keys orderedFrontalKeys; ///< Frontal keys of this node
42
43 FactorGraphType factors; ///< Factors associated with this node
44
45 int problemSize_;
46
47 Cluster() : problemSize_(0) {}
48
49 virtual ~Cluster() {}
50
51 const Cluster& operator[](size_t i) const {
52 return *(children.at(i));
53 }
54
55 /// Construct from factors associated with a single key
56 template <class CONTAINER>
57 Cluster(Key key, const CONTAINER& factorsToAdd)
58 : problemSize_(0) {
59 addFactors(key, factorsToAdd);
60 }
61
62 /// Add factors associated with a single key
63 template <class CONTAINER>
64 void addFactors(Key key, const CONTAINER& factorsToAdd) {
65 orderedFrontalKeys.push_back(x: key);
66 factors.push_back(factorsToAdd);
67 problemSize_ += factors.size();
68 }
69
70 /// Add a child cluster
71 void addChild(const std::shared_ptr<Cluster>& cluster) {
72 children.push_back(cluster);
73 problemSize_ = std::max(problemSize_, cluster->problemSize_);
74 }
75
76 size_t nrChildren() const {
77 return children.size();
78 }
79
80 size_t nrFactors() const {
81 return factors.size();
82 }
83
84 size_t nrFrontals() const {
85 return orderedFrontalKeys.size();
86 }
87
88 int problemSize() const {
89 return problemSize_;
90 }
91
92 /// print this node
93 virtual void print(const std::string& s = "",
94 const KeyFormatter& keyFormatter = DefaultKeyFormatter) const;
95
96 /// Return a vector with nrFrontal keys for each child
97 std::vector<size_t> nrFrontalsOfChildren() const;
98
99 /// Merge in given cluster
100 void merge(const std::shared_ptr<Cluster>& cluster);
101
102 /// Merge all children for which bit is set into this node
103 void mergeChildren(const std::vector<bool>& merge);
104 };
105
106 typedef std::shared_ptr<Cluster> sharedCluster; ///< Shared pointer to Cluster
107
108 // Define Node=Cluster for compatibility with tree traversal functions
109 typedef Cluster Node;
110 typedef sharedCluster sharedNode;
111
112 /** concept check */
113 GTSAM_CONCEPT_TESTABLE_TYPE(FactorType)
114
115 protected:
116 FastVector<sharedNode> roots_;
117
118 /// @name Standard Constructors
119 /// @{
120
121 /** Copy constructor - makes a deep copy of the tree structure, but only pointers to factors are
122 * copied, factors are not cloned. */
123 ClusterTree(const This& other) {
124 *this = other;
125 }
126
127 /// @}
128
129 public:
130
131 /// Default constructor
132 ClusterTree() {}
133
134 /// @name Testable
135 /// @{
136
137 /** Print the cluster tree */
138 void print(const std::string& s = "",
139 const KeyFormatter& keyFormatter = DefaultKeyFormatter) const;
140
141 /// @}
142 /// @name Advanced Interface
143 /// @{
144
145 void addRoot(const std::shared_ptr<Cluster>& cluster) {
146 roots_.push_back(cluster);
147 }
148
149 void addChildrenAsRoots(const std::shared_ptr<Cluster>& cluster) {
150 for (auto child : cluster->children)
151 this->addRoot(child);
152 }
153
154 size_t nrRoots() const {
155 return roots_.size();
156 }
157
158 /** Return the set of roots (one for a tree, multiple for a forest) */
159 const FastVector<sharedNode>& roots() const {
160 return roots_;
161 }
162
163 const Cluster& operator[](size_t i) const {
164 return *(roots_.at(i));
165 }
166
167 /// @}
168
169 protected:
170 ~ClusterTree();
171
172 /// @name Details
173 /// @{
174
175 /// Assignment operator - makes a deep copy of the tree structure, but only pointers to factors
176 /// are copied, factors are not cloned.
177 This& operator=(const This& other);
178
179 /// @}
180};

182/**
183 * A cluster-tree that eliminates to a Bayes tree.
184 */
185template <class BAYESTREE, class GRAPH>
186class EliminatableClusterTree : public ClusterTree<GRAPH> {
187 public:
188 typedef BAYESTREE BayesTreeType; ///< The BayesTree type produced by elimination
189 typedef GRAPH FactorGraphType; ///< The factor graph type
190 typedef EliminatableClusterTree<BAYESTREE, GRAPH> This; ///< This class
191 typedef std::shared_ptr<This> shared_ptr; ///< Shared pointer to this class
192
193 typedef typename BAYESTREE::ConditionalType ConditionalType; ///< The type of conditionals
194 typedef std::shared_ptr<ConditionalType>
195 sharedConditional; ///< Shared pointer to a conditional
196
197 typedef typename GRAPH::Eliminate Eliminate; ///< Typedef for an eliminate subroutine
198 typedef typename GRAPH::FactorType FactorType; ///< The type of factors
199 typedef std::shared_ptr<FactorType> sharedFactor; ///< Shared pointer to a factor
200
201 protected:
202 FastVector<sharedFactor> remainingFactors_;
203
204 /// @name Standard Constructors
205 /// @{
206
207 /** Copy constructor - makes a deep copy of the tree structure, but only pointers to factors are
208 * copied, factors are not cloned. */
209 EliminatableClusterTree(const This& other) : ClusterTree<GRAPH>(other) {
210 *this = other;
211 }
212
213 /// @}
214
215 public:
216 /// @name Standard Interface
217 /// @{
218
219 /** Eliminate the factors to a Bayes tree and remaining factor graph
220 * @param function The function to use to eliminate, see the namespace functions
221 * in GaussianFactorGraph.h
222 * @return The Bayes tree and factor graph resulting from elimination
223 */
224 std::pair<std::shared_ptr<BayesTreeType>, std::shared_ptr<FactorGraphType> > eliminate(
225 const Eliminate& function) const;
226
227 /// @}
228
229 /// @name Advanced Interface
230 /// @{
231
232 /** Return the remaining factors that are not pulled into elimination */
233 const FastVector<sharedFactor>& remainingFactors() const {
234 return remainingFactors_;
235 }
236
237 /// @}
238
239 protected:
240 /// @name Details
241 /// @{
242
243 /// Assignment operator - makes a deep copy of the tree structure, but only pointers to factors
244 /// are copied, factors are not cloned.
245 This& operator=(const This& other);
246
247 /// Default constructor to be used in derived classes
248 EliminatableClusterTree() {}
249
250 /// @}
251};
252}
