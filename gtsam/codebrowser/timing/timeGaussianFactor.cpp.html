100 // time eliminate
101 JacobianFactor combined(_x2_, Ax2, _l1_, Al1, _x1_, Ax1, b2, noiseModel::Isotropic::Sigma(dim: 8,sigma: 1));
102 long timeLog = clock();
103 int n = 1000000;
104 GaussianConditional::shared_ptr conditional;
105 JacobianFactor::shared_ptr factor;
106
107 for(int i = 0; i < n; i++)
108 std::tie(args&: conditional, args&: factor) =
109 JacobianFactor(combined).eliminate(keys: Ordering{_x2_});
110
111 long timeLog2 = clock();
112 double seconds = (double)(timeLog2-timeLog)/CLOCKS_PER_SEC;
113 cout << "Single Eliminate Timing:" << endl;

115 cout << ((double)n/seconds) << " calls/second" << endl;
