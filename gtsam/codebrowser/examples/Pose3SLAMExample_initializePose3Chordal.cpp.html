12/**
13 * @file Pose3SLAMExample_initializePose3Chordal.cpp
14 * @brief A 3D Pose SLAM example that reads input from g2o, and initializes the Pose3 using InitializePose3
15 * Syntax for the script is ./Pose3SLAMExample_initializePose3Chordal input.g2o output.g2o
16 * @date Aug 25, 2014
17 * @author Luca Carlone
18 */
19
20#include <gtsam/slam/InitializePose3.h>
21#include <gtsam/slam/dataset.h>
22#include <gtsam/slam/BetweenFactor.h>
23#include <fstream>
24
25using namespace std;
26using namespace gtsam;
27
28int main(const int argc, const char* argv[]) {
29 // Read graph from file
30 string g2oFile;
31 if (argc < 2)
32 g2oFile = findExampleDataFile(name: "pose3example.txt");
33 else
34 g2oFile = argv[1];
35
36 NonlinearFactorGraph::shared_ptr graph;
37 Values::shared_ptr initial;
38 bool is3D = true;
39 std::tie(args&: graph, args&: initial) = readG2o(g2oFile, is3D);
40
41 // Add prior on the first key
42 auto priorModel = noiseModel::Diagonal::Variances(
43 variances: (Vector(6) << 1e-6, 1e-6, 1e-6, 1e-4, 1e-4, 1e-4).finished());
44 Key firstKey = 0;
45 for (const auto key : initial->keys()) {
46 std::cout << "Adding prior to g2o file " << std::endl;
47 firstKey = key;
48 graph->addPrior(key: firstKey, prior: Pose3(), model: priorModel);
49 break;
50 }
51
52 std::cout << "Initializing Pose3 - chordal relaxation" << std::endl;
53 Values initialization = InitializePose3::initialize(graph: *graph);
54 std::cout << "done!" << std::endl;
55
56 if (argc < 3) {
57 initialization.print(str: "initialization");
58 } else {
59 const string outputFile = argv[2];
60 std::cout << "Writing results to file: " << outputFile << std::endl;
61 NonlinearFactorGraph::shared_ptr graphNoKernel;
62 Values::shared_ptr initial2;
63 std::tie(args&: graphNoKernel, args&: initial2) = readG2o(g2oFile);
64 writeG2o(graph: *graphNoKernel, estimate: initialization, filename: outputFile);
65 std::cout << "done! " << std::endl;
66 }
67 return 0;
68}
