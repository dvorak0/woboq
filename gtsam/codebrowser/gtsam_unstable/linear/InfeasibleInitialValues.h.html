23/** An exception indicating that the provided initial value is infeasible
24 * Also used to inzdicatethat the noise model dimension passed into a
25 * JacobianFactor has a different dimensionality than the factor. */
26class InfeasibleInitialValues: public ThreadsafeException<
27 InfeasibleInitialValues> {
28public:
29 InfeasibleInitialValues() {
30 }
31
32 ~InfeasibleInitialValues() noexcept override {
33 }
34
35 const char *what() const noexcept override {
36 if (description_->empty())
37 description_ =
38 "An infeasible initial value was provided for the solver.\n";
39 return description_->c_str();
40 }
41};
42}
