44struct space_info
45{

47 boost::uintmax_t capacity;
48 boost::uintmax_t free; // <= capacity
49 boost::uintmax_t available; // <= free
50};
51
52BOOST_SCOPED_ENUM_UT_DECLARE_BEGIN(copy_options, unsigned int)
53{
54 none = 0u, // Default. For copy_file: error if the target file exists. For copy: do not recurse, follow symlinks, copy file contents.
55
56 // copy_file options:
57 skip_existing = 1u, // Don't overwrite the existing target file, don't report an error
58 overwrite_existing = 1u << 1, // Overwrite existing file
59 update_existing = 1u << 2, // Overwrite existing file if its last write time is older than the replacement file
60 synchronize_data = 1u << 3, // Flush all buffered data written to the target file to permanent storage
61 synchronize = 1u << 4, // Flush all buffered data and attributes written to the target file to permanent storage
62
63 // copy options:
64 recursive = 1u << 8, // Recurse into sub-directories
65 copy_symlinks = 1u << 9, // Copy symlinks as symlinks instead of copying the referenced file
66 skip_symlinks = 1u << 10, // Don't copy symlinks
67 directories_only = 1u << 11, // Only copy directory structure, do not copy non-directory files
68 create_symlinks = 1u << 12, // Create symlinks instead of copying files
69 create_hard_links = 1u << 13, // Create hard links instead of copying files
70 _detail_recursing = 1u << 14 // Internal use only, do not use
71}
72BOOST_SCOPED_ENUM_DECLARE_END(copy_options)
73
74BOOST_BITMASK(BOOST_SCOPED_ENUM_NATIVE(copy_options))
75
76#if !defined(BOOST_FILESYSTEM_NO_DEPRECATED)
77BOOST_SCOPED_ENUM_DECLARE_BEGIN(copy_option)
78{
79 none = static_cast< unsigned int >(copy_options::none),
80 fail_if_exists = none,
81 overwrite_if_exists = static_cast< unsigned int >(copy_options::overwrite_existing)
82}
83BOOST_SCOPED_ENUM_DECLARE_END(copy_option)
84#endif
