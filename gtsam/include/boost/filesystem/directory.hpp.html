61// GCC has a problem with a member function named path within a namespace or
62// sub-namespace that also has a class named path. The workaround is to always
63// fully qualify the name path when it refers to the class name.

444// Dispatch function for integration with path class
445template< typename Callback >
446BOOST_FORCEINLINE typename Callback::result_type dispatch(directory_entry const& de, Callback cb, const codecvt_type* cvt, directory_entry_tag)
447{
448 boost::filesystem::path::string_type const& source = de.path().native();
449 return cb(source.data(), source.data() + source.size(), cvt);
450}

461// Without these functions, calling (for example) 'is_directory' with a 'directory_entry' results in:
462// - a conversion to 'path' using 'operator boost::filesystem::path const&()',
463// - then a call to 'is_directory(path const& p)' which recomputes the status with 'detail::status(p)'.
464//
465// These functions avoid a costly recomputation of the status if one calls 'is_directory(e)' instead of 'is_directory(e.status())'

621BOOST_SCOPED_ENUM_UT_DECLARE_BEGIN(directory_options, unsigned int)
622{
623 none = 0u,
624 skip_permission_denied = 1u, // if a directory cannot be opened because of insufficient permissions, pretend that the directory is empty
625 follow_directory_symlink = 1u << 1u, // recursive_directory_iterator: follow directory symlinks
626 skip_dangling_symlinks = 1u << 2u, // non-standard extension for recursive_directory_iterator: don't follow dangling directory symlinks,
627 pop_on_error = 1u << 3u, // non-standard extension for recursive_directory_iterator: instead of producing an end iterator on errors,
628 // repeatedly invoke pop() until it succeeds or the iterator becomes equal to end iterator
629 _detail_no_follow = 1u << 4u, // internal use only
630 _detail_no_push = 1u << 5u // internal use only
631}
