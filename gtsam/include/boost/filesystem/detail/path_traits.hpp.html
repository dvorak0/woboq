415template< typename Callback >
416BOOST_FORCEINLINE typename Callback::result_type dispatch(const char* source, Callback cb, const codecvt_type* cvt, ntcts_type_tag)
417{
418 return cb(source, source + std::strlen(s: source), cvt);
419}
420
421template< typename Callback >
422BOOST_FORCEINLINE typename Callback::result_type dispatch(const wchar_t* source, Callback cb, const codecvt_type* cvt, ntcts_type_tag)
423{
424 return cb(source, source + std::wcslen(s: source), cvt);
425}
426
427template< typename Source, typename Callback >
428BOOST_FORCEINLINE typename Callback::result_type dispatch(Source const& source, Callback cb, const codecvt_type* cvt, string_class_tag)
429{
430 return cb(source.data(), source.data() + source.size(), cvt);
431}
432
433template< typename Source, typename Callback >
434BOOST_FORCEINLINE typename Callback::result_type dispatch(Source const& source, Callback cb, const codecvt_type* cvt, range_type_tag)
435{
436 std::basic_string< typename Source::value_type > src(source.begin(), source.end());
437 return cb(src.data(), src.data() + src.size(), cvt);
438}

442template< typename Callback >
443BOOST_FORCEINLINE typename Callback::result_type dispatch(std::vector< char > const& source, Callback cb, const codecvt_type* cvt, range_type_tag)
444{
445 const char* data = NULL, *data_end = NULL;
446 if (!source.empty())
447 {
448 data = &source[0];
449 data_end = data + source.size();
450 }
451 return cb(data, data_end, cvt);
452}

518// Note: We use separate checks for convertibility to std::string_view and other types to avoid ambiguity with an implicit range constructor
519// of std::string_view in the early C++23 draft (N4892). If a user's type is convertible to e.g. std::string and also satisfies
520// ranges::contiguous_range and ranges::sized_range concepts then the conversion is ambiguous: the type is convertible to std::string
521// through the conversion operator in the user's class and is also convertible to std::string_view through the implicit conversion
522// constructor in std::string_view. The solution is to check convertibility to std::string_view separately first.
523
524namespace is_convertible_to_std_string_view_impl {
525
526yes_type check(std::string_view const&);
527yes_type check(std::wstring_view const&);
528#if !defined(BOOST_NO_CXX11_NULLPTR)
529no_type check(std::nullptr_t);
530#endif
531no_type check(...);
532
533} // namespace is_convertible_to_std_string_view_impl
