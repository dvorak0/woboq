14#ifndef BOOST_FILESYSTEM_CONVENIENCE_HPP
15#define BOOST_FILESYSTEM_CONVENIENCE_HPP
16
17#include <boost/filesystem/config.hpp>
18#include <boost/filesystem/operations.hpp>
19#include <boost/system/error_code.hpp>
20
21#include <boost/filesystem/detail/header.hpp> // must be the last #include
22
23namespace boost {
24namespace filesystem {
25
26#ifndef BOOST_FILESYSTEM_NO_DEPRECATED
27
28BOOST_FILESYSTEM_DETAIL_DEPRECATED("Use path::extension() instead")
29inline std::string extension(const path& p)
30{
31 return p.extension().string();
32}
33
34BOOST_FILESYSTEM_DETAIL_DEPRECATED("Use path::stem() instead")
35inline std::string basename(const path& p)
36{
37 return p.stem().string();
38}
39
40BOOST_FILESYSTEM_DETAIL_DEPRECATED("Use path::replace_extension() instead")
41inline path change_extension(const path& p, const path& new_extension)
42{
43 path new_p(p);
44 new_p.replace_extension(new_extension);
45 return new_p;
46}
47
48#endif
49
50} // namespace filesystem
51} // namespace boost
52
53#include <boost/filesystem/detail/footer.hpp>
54
55#endif // BOOST_FILESYSTEM_CONVENIENCE_HPP
