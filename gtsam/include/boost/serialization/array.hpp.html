35template <class Archive, class T, std::size_t N>
36void serialize(Archive& ar, std::array<T,N>& a, const unsigned int /* version */)
37{
38 ar & boost::serialization::make_nvp(
39 "elems",
40 *static_cast<T (*)[N]>(static_cast<void *>(a.data()))
41 );
42
43}
44} } // end namespace boost::serialization
