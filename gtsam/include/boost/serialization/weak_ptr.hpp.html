22namespace boost {
23namespace serialization{
24
25template<class Archive, class T>
26inline void save(
27 Archive & ar,
28 const boost::weak_ptr< T > &t,
29 const unsigned int /* file_version */
30){
31 const boost::shared_ptr< T > sp = t.lock();
32 ar << boost::serialization::make_nvp("weak_ptr", sp);
33}
34
35template<class Archive, class T>
36inline void load(
37 Archive & ar,
38 boost::weak_ptr< T > &t,
39 const unsigned int /* file_version */
40){
41 boost::shared_ptr< T > sp;
42 ar >> boost::serialization::make_nvp("weak_ptr", sp);
43 t = sp;
44}
45
46template<class Archive, class T>
47inline void serialize(
48 Archive & ar,
49 boost::weak_ptr< T > &t,
50 const unsigned int file_version
51){
52 boost::serialization::split_free(ar, t, file_version);
53}
