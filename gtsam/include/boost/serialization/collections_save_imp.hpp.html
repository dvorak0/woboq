34// implementation of serialization for STL containers

36
37template<class Archive, class Container>
38inline void save_collection(
39 Archive & ar,
40 const Container &s,
41 collection_size_type count)
42{
43 ar << BOOST_SERIALIZATION_NVP(count);
44 // record number of elements
45 const item_version_type item_version(
46 version<typename Container::value_type>::value
47 );
48
49 ar << BOOST_SERIALIZATION_NVP(item_version);
50
51 typename Container::const_iterator it = s.begin();
52 while(count-- > 0){
53 // note borland emits a no-op without the explicit namespace
54 boost::serialization::save_construct_data_adl(
55 ar,
56 boost::addressof(*it),
57 item_version
58 );
59 ar << boost::serialization::make_nvp("item", *it++);
60 }
61}
62
63template<class Archive, class Container>
64inline void save_collection(Archive & ar, const Container &s)
65{
66 // record number of elements
67 collection_size_type count(s.size());
68 save_collection(ar, s, count);
69}
