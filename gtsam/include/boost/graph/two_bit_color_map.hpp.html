25namespace boost
26{
27
28enum two_bit_color_type
29{
30 two_bit_white = 0,
31 two_bit_gray = 1,
32 two_bit_green = 2,
33 two_bit_black = 3
34};
35
36template <> struct color_traits< two_bit_color_type >
37{
38 static two_bit_color_type white() { return two_bit_white; }
39 static two_bit_color_type gray() { return two_bit_gray; }
40 static two_bit_color_type green() { return two_bit_green; }
41 static two_bit_color_type black() { return two_bit_black; }
42};
43
44template < typename IndexMap = identity_property_map > struct two_bit_color_map
45{
46 std::size_t n;
47 IndexMap index;
48 shared_array< unsigned char > data;
49
50 BOOST_STATIC_CONSTANT(
51 int, bits_per_char = std::numeric_limits< unsigned char >::digits);
52 BOOST_STATIC_CONSTANT(int, elements_per_char = bits_per_char / 2);
53 typedef typename property_traits< IndexMap >::key_type key_type;
54 typedef two_bit_color_type value_type;
55 typedef void reference;
56 typedef read_write_property_map_tag category;
57
58 explicit two_bit_color_map(
59 std::size_t n, const IndexMap& index = IndexMap())
60 : n(n)
61 , index(index)
62 , data(new unsigned char[(n + elements_per_char - 1) / elements_per_char]())
63 {
64 }
65};
66
67template < typename IndexMap >
68inline two_bit_color_type get(const two_bit_color_map< IndexMap >& pm,
69 typename property_traits< IndexMap >::key_type key)
70{
71 BOOST_STATIC_CONSTANT(int,
72 elements_per_char = two_bit_color_map< IndexMap >::elements_per_char);
73 typename property_traits< IndexMap >::value_type i = get(pm.index, key);
74 BOOST_ASSERT((std::size_t)i < pm.n);
75 std::size_t byte_num = i / elements_per_char;
76 std::size_t bit_position = ((i % elements_per_char) * 2);
77 return two_bit_color_type((pm.data.get()[byte_num] >> bit_position) & 3);
78}
79
80template < typename IndexMap >
81inline void put(const two_bit_color_map< IndexMap >& pm,
82 typename property_traits< IndexMap >::key_type key,
83 two_bit_color_type value)
84{
85 BOOST_STATIC_CONSTANT(int,
86 elements_per_char = two_bit_color_map< IndexMap >::elements_per_char);
87 typename property_traits< IndexMap >::value_type i = get(pm.index, key);
88 BOOST_ASSERT((std::size_t)i < pm.n);
89 BOOST_ASSERT(value >= 0 && value < 4);
90 std::size_t byte_num = i / elements_per_char;
91 std::size_t bit_position = ((i % elements_per_char) * 2);
92 pm.data.get()[byte_num]
93 = (unsigned char)((pm.data.get()[byte_num] & ~(3 << bit_position))
94 | (value << bit_position));
95}
96
97template < typename IndexMap >
98inline two_bit_color_map< IndexMap > make_two_bit_color_map(
99 std::size_t n, const IndexMap& index_map)
100{
101 return two_bit_color_map< IndexMap >(n, index_map);
102}
