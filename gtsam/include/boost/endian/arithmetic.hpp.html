179template <BOOST_SCOPED_ENUM(order) Order, class T, std::size_t n_bits,
180 BOOST_SCOPED_ENUM(align) Align>
181class endian_arithmetic
182{
183private:
184
185 typedef endian_buffer<Order, T, n_bits, Align> buffer_type;
186
187#ifdef BOOST_ENDIAN_NO_CTORS
188public:
189#else
190private:
191#endif
192
193 buffer_type buf_;
194
195public:
196
197 typedef T value_type;
198
199#ifndef BOOST_ENDIAN_NO_CTORS
200
201 endian_arithmetic() BOOST_ENDIAN_DEFAULT_CONSTRUCT
202
203 BOOST_ENDIAN_EXPLICIT_OPT endian_arithmetic( T val ) BOOST_NOEXCEPT: buf_( val )
204 {
205 }
206
207#endif
208
209 endian_arithmetic& operator=( T val ) BOOST_NOEXCEPT
210 {
211 buf_ = val;
212 return *this;
213 }
214
215 value_type value() const BOOST_NOEXCEPT
216 {
217 return buf_.value();
218 }
219
220 unsigned char const * data() const BOOST_NOEXCEPT
221 {
222 return buf_.data();
223 }
224
225 unsigned char * data() BOOST_NOEXCEPT
226 {
227 return buf_.data();
228 }
229
230 operator value_type() const BOOST_NOEXCEPT
231 {
232 return this->value();
233 }
234
235 operator buffer_type& () BOOST_NOEXCEPT
236 {
237 return buf_;
238 }
239
240 operator buffer_type const& () BOOST_NOEXCEPT
241 {
242 return buf_;
243 }
244
245 // operators
246
247 T operator+() const BOOST_NOEXCEPT
248 {
249 return this->value();
250 }
251
252 endian_arithmetic& operator+=( T y ) BOOST_NOEXCEPT
253 {
254 *this = static_cast<T>( this->value() + y );
255 return *this;
256 }
257
258 endian_arithmetic& operator-=( T y ) BOOST_NOEXCEPT
259 {
260 *this = static_cast<T>( this->value() - y );
261 return *this;
262 }
263
264 endian_arithmetic& operator*=( T y ) BOOST_NOEXCEPT
265 {
266 *this = static_cast<T>( this->value() * y );
267 return *this;
268 }
269
270 endian_arithmetic& operator/=( T y ) BOOST_NOEXCEPT
271 {
272 *this = static_cast<T>( this->value() / y );
273 return *this;
274 }
275
276 endian_arithmetic& operator%=( T y ) BOOST_NOEXCEPT
277 {
278 *this = static_cast<T>( this->value() % y );
279 return *this;
280 }
281
282 endian_arithmetic& operator&=( T y ) BOOST_NOEXCEPT
283 {
284 *this = static_cast<T>( this->value() & y );
285 return *this;
286 }
287
288 endian_arithmetic& operator|=( T y ) BOOST_NOEXCEPT
289 {
290 *this = static_cast<T>( this->value() | y );
291 return *this;
292 }
293
294 endian_arithmetic& operator^=( T y ) BOOST_NOEXCEPT
295 {
296 *this = static_cast<T>( this->value() ^ y );
297 return *this;
298 }
299
300 endian_arithmetic& operator<<=( T y ) BOOST_NOEXCEPT
301 {
302 *this = static_cast<T>( this->value() << y );
303 return *this;
304 }
305
306 endian_arithmetic& operator>>=( T y ) BOOST_NOEXCEPT
307 {
308 *this = static_cast<T>( this->value() >> y );
309 return *this;
310 }
311
312 endian_arithmetic& operator++() BOOST_NOEXCEPT
313 {
314 *this += 1;
315 return *this;
316 }
317
318 endian_arithmetic& operator--() BOOST_NOEXCEPT
319 {
320 *this -= 1;
321 return *this;
322 }
323
324 endian_arithmetic operator++(int) BOOST_NOEXCEPT
325 {
326 endian_arithmetic tmp( *this );
327 *this += 1;
328 return tmp;
329 }
330
331 endian_arithmetic operator--(int) BOOST_NOEXCEPT
332 {
333 endian_arithmetic tmp( *this );
334 *this -= 1;
335 return tmp;
336 }
337
338 template<class Ch, class Tr>
339 friend std::basic_ostream<Ch, Tr>&
340 operator<<( std::basic_ostream<Ch, Tr>& os, endian_arithmetic const& x )
341 {
342 return os << x.value();
343 }
344
345 template<class Ch, class Tr>
346 friend std::basic_istream<Ch, Tr>&
347 operator>>( std::basic_istream<Ch, Tr>& is, endian_arithmetic& x )
348 {
349 T i;
350
351 if( is >> i )
352 {
353 x = i;
354 }
355
356 return is;
357 }
358};
