20template<class T> struct negation: boost::integral_constant<bool, !T::value> {};
21
22template<class T> struct is_scoped_enum:
23 boost::conditional<
24 boost::is_enum<T>::value,
25 negation< boost::is_convertible<T, int> >,
26 boost::false_type
27 >::type
28{
29};
