180 // Stream inserter
181 template <class charT, class traits, BOOST_SCOPED_ENUM(order) Order, class T,
182 std::size_t n_bits, BOOST_SCOPED_ENUM(align) A>
183 std::basic_ostream<charT, traits>&
184 operator<<(std::basic_ostream<charT, traits>& os,
185 const endian_buffer<Order, T, n_bits, A>& x)
186 {
187 return os << x.value();
188 }
189
190 // Stream extractor
191 template <class charT, class traits, BOOST_SCOPED_ENUM(order) Order, class T,
192 std::size_t n_bits, BOOST_SCOPED_ENUM(align) A>
193 std::basic_istream<charT, traits>&
194 operator>>(std::basic_istream<charT, traits>& is,
195 endian_buffer<Order, T, n_bits, A>& x)
196 {
197 T i;
198 if (is >> i)
199 x = i;
200 return is;
201 }

205// endian_buffer class template specializations --------------------------------------//
206
207// Specializations that represent unaligned bytes.
208// Taking an integer type as a parameter provides a nice way to pass both
209// the size and signedness of the desired integer and get the appropriate
210// corresponding integer type for the interface.
211
212// Q: Should endian_buffer supply "value_type operator value_type() const noexcept"?
213// A: No. The rationale for endian_buffers is to prevent high-cost hidden
214// conversions. If an implicit conversion operator is supplied, hidden conversions
215// can occur.
216
217// unaligned endian_buffer specialization
218
219template< BOOST_SCOPED_ENUM(order) Order, class T, std::size_t n_bits >
220class endian_buffer<Order, T, n_bits, align::no>
221{
222#ifdef BOOST_ENDIAN_NO_CTORS
223public:
224#endif
225
226 BOOST_STATIC_ASSERT( (n_bits/8)*8 == n_bits );
227
228 unsigned char value_[ n_bits / 8 ];
229
230public:
231
232 typedef T value_type;
233
234#ifndef BOOST_ENDIAN_NO_CTORS
235
236 endian_buffer() BOOST_ENDIAN_DEFAULT_CONSTRUCT
237
238 explicit endian_buffer( T val ) BOOST_NOEXCEPT
239 {
240 boost::endian::endian_store<T, n_bits / 8, Order>( value_, val );
241 }
242
243#endif
244
245 endian_buffer& operator=( T val ) BOOST_NOEXCEPT
246 {
247 boost::endian::endian_store<T, n_bits / 8, Order>( value_, val );
248 return *this;
249 }
250
251 value_type value() const BOOST_NOEXCEPT
252 {
253 return boost::endian::endian_load<T, n_bits / 8, Order>( value_ );
254 }
255
256 unsigned char const * data() const BOOST_NOEXCEPT
257 {
258 return value_;
259 }
260
261 unsigned char * data() BOOST_NOEXCEPT
262 {
263 return value_;
264 }
265};

271template< BOOST_SCOPED_ENUM(order) Order, class T, std::size_t n_bits >

334public:

336 typedef T value_type;
337
338#ifndef BOOST_ENDIAN_NO_CTORS
339

341

343 {
344 }
345
346#endif
347
348 endian_buffer& operator=( T val ) BOOST_NOEXCEPT
349 {
350 value_ = val;
351 return *this;
352 }
353
354 value_type value() const BOOST_NOEXCEPT
355 {
356 return value_;
357 }
358
359 unsigned char const * data() const BOOST_NOEXCEPT
360 {
361 return reinterpret_cast< unsigned char const* >( &value_ );
362 }
363
364 unsigned char * data() BOOST_NOEXCEPT
365 {
366 return reinterpret_cast< unsigned char* >( &value_ );
367 }
