151 friend BOOST_SYSTEM_CONSTEXPR bool operator==( error_category const & lhs, error_category const & rhs ) BOOST_NOEXCEPT
152 {
153 return rhs.id_ == 0? &lhs == &rhs: lhs.id_ == rhs.id_;
154 }
155
156 friend BOOST_SYSTEM_CONSTEXPR bool operator!=( error_category const & lhs, error_category const & rhs ) BOOST_NOEXCEPT
157 {
158 return !( lhs == rhs );
159 }
160
161 friend BOOST_SYSTEM_CONSTEXPR bool operator<( error_category const & lhs, error_category const & rhs ) BOOST_NOEXCEPT
162 {
163 if( lhs.id_ < rhs.id_ )
164 {
165 return true;
166 }
167
168 if( lhs.id_ > rhs.id_ )
169 {
170 return false;
171 }
172
173 if( rhs.id_ != 0 )
174 {
175 return false; // equal
176 }
177
178 return std::less<error_category const *>()( &lhs, &rhs );
179 }

205static const boost::ulong_long_type generic_category_id = ( boost::ulong_long_type( 0xB2AB117A ) << 32 ) + 0x257EDFD0;
206static const boost::ulong_long_type system_category_id = generic_category_id + 1;
207static const boost::ulong_long_type interop_category_id = generic_category_id + 2;
208
209BOOST_SYSTEM_CONSTEXPR inline bool failed_impl( int ev, error_category const & cat )
210{
211 if( cat.id_ == system_category_id || cat.id_ == generic_category_id )
212 {
213 return ev != 0;
214 }
215 else
216 {
217 return cat.failed( ev );
218 }
219}
