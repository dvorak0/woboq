28inline void snprintf( char * buffer, std::size_t len, char const * format, ... )
29{
30# if defined( BOOST_MSVC )
31# pragma warning( push )
32# pragma warning( disable: 4996 )
33# endif
34
35 if( len == 0 ) return;
36
37 va_list args;
38 va_start( args, format );
39
40 _vsnprintf( buffer, len - 1, format, args );
41 buffer[ len - 1 ] = 0;
42
43 va_end( args );
44
45# if defined( BOOST_MSVC )
46# pragma warning( pop )
47# endif
48}
49
50#else
51
52#if defined(__GNUC__) && __GNUC__ >= 3
53__attribute__((__format__ (__printf__, 3, 4)))
54#endif
55inline void snprintf( char * buffer, std::size_t len, char const * format, ... )
56{
57 va_list args;
58 va_start( args, format );
59
60 std::vsnprintf( s: buffer, maxlen: len, format: format, arg: args );
61
62 va_end( args );
63}
