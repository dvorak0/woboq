27#include <boost/type_traits/conditional.hpp>

29namespace boost { namespace detail {
30
31 template <typename TargetChar, typename SourceChar>
32 struct widest_char {
33 typedef BOOST_DEDUCED_TYPENAME boost::conditional<
34 (sizeof(TargetChar) > sizeof(SourceChar))
35 , TargetChar
36 , SourceChar
37 >::type type;
38 };

40}} // namespace boost::detail
