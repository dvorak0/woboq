31 template <typename Expr
32 , typename CopyExpr = mpl::false_, typename CopyAttr = mpl::false_
33 , typename Skipper = unused_type, typename Attribute = unused_type const>
34 struct match_manip
35 {
36 // This assertion makes sure we don't hit the only code path which is
37 // not implemented (because it isn't needed), where both, the
38 // expression and the attribute need to be held as a copy.
39 BOOST_SPIRIT_ASSERT_MSG(!CopyExpr::value || !CopyAttr::value
40 , error_invalid_should_not_happen, ());
41
42 match_manip(Expr const& xpr, Skipper const& s, Attribute& a)
43 : expr(xpr), skipper(s), attr(a), post_skip(skip_flag::postskip) {}
44
45 match_manip(Expr const& xpr, Skipper const& s
46 , BOOST_SCOPED_ENUM(skip_flag) ps, Attribute& a)
47 : expr(xpr), skipper(s), attr(a), post_skip(ps) {}
48
49 Expr const& expr;
50 Skipper const& skipper;
51 Attribute& attr;
52 BOOST_SCOPED_ENUM(skip_flag) const post_skip;
53 };
54

91 template <typename Expr, typename Enable = void>
92 struct match
93 {
94 // Report invalid expression error as early as possible.
95 // If you got an error_invalid_expression error message here,
96 // then the expression (expr) is not a valid spirit qi expression.
97 // Did you intend to use the auto_ facilities while forgetting to
98 // #include <boost/spirit/include/qi_match_auto.hpp>?
99 BOOST_SPIRIT_ASSERT_MATCH(qi::domain, Expr);
100 };
101
102 template <typename Expr>
103 struct match<Expr
104 , typename enable_if<traits::matches<qi::domain, Expr> >::type>
105 {
106 typedef match_manip<Expr> type;
107
108 static type call(Expr const& expr)
109 {
110 return type(expr, unused, unused);
111 }
112 };

115 template <typename Expr, typename Skipper, typename Enable = void>
116 struct phrase_match
117 {
118 // Report invalid expression error as early as possible.
119 // If you got an error_invalid_expression error message here,
120 // then the expression (expr) is not a valid spirit qi expression.
121 // Did you intend to use the auto_ facilities while forgetting to
122 // #include <boost/spirit/include/qi_match_auto.hpp>?
123 BOOST_SPIRIT_ASSERT_MATCH(qi::domain, Expr);
124 };
125
126 template <typename Expr, typename Skipper>
127 struct phrase_match<Expr, Skipper
128 , typename enable_if<traits::matches<qi::domain, Expr> >::type>
129 {
130 typedef match_manip<Expr, mpl::false_, mpl::false_, Skipper> type;
131
132 static type call(
133 Expr const& expr
134 , Skipper const& skipper
135 , BOOST_SCOPED_ENUM(skip_flag) post_skip)
136 {
137 // Report invalid expression error as early as possible.
138 // If you got an error_invalid_expression error message here,
139 // then the delimiter is not a valid spirit karma expression.
140 BOOST_SPIRIT_ASSERT_MATCH(qi::domain, Skipper);
141 return type(expr, skipper, post_skip, unused);
142 }
143 };

146 template<typename Char, typename Traits, typename Expr
147 , typename CopyExpr, typename CopyAttr>
148 inline std::basic_istream<Char, Traits> &
149 operator>>(std::basic_istream<Char, Traits> &is,
150 match_manip<Expr, CopyExpr, CopyAttr> const& fm)
151 {
152 typedef spirit::basic_istream_iterator<Char, Traits> input_iterator;
153
154 input_iterator f(is);
155 input_iterator l;
156 if (!qi::parse(f, l, fm.expr))
157 {
158 is.setstate(std::ios_base::failbit);
159 }
160 return is;
161 }
