23 template <typename A, typename B>
24 inline alternative<A, B>
25 operator|(parser<A> const& a, parser<B> const& b)
26 {
27 return alternative<A, B>(a.derived(), b.derived());
28 }
29
30 template <typename A>
31 inline alternative<A, chlit<char> >
32 operator|(parser<A> const& a, char b)
33 {
34 return alternative<A, chlit<char> >(a.derived(), b);
35 }
36
37 template <typename B>
38 inline alternative<chlit<char>, B>
39 operator|(char a, parser<B> const& b)
40 {
41 return alternative<chlit<char>, B>(a, b.derived());
42 }
43
44 template <typename A>
45 inline alternative<A, strlit<char const*> >
46 operator|(parser<A> const& a, char const* b)
47 {
48 return alternative<A, strlit<char const*> >(a.derived(), b);
49 }
50
51 template <typename B>
52 inline alternative<strlit<char const*>, B>
53 operator|(char const* a, parser<B> const& b)
54 {
55 return alternative<strlit<char const*>, B>(a, b.derived());
56 }
