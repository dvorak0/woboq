74// Public parameters
75Mat image;
76int roi_x0 = 0, roi_y0 = 0, roi_x1 = 0, roi_y1 = 0, num_of_rec = 0;
77bool start_draw = false, stop = false;
78
79// Window name for visualisation purposes
80const string window_name = "OpenCV Based Annotation Tool";
81
82// FUNCTION : Mouse response for selecting objects in images
83// If left button is clicked, start drawing a rectangle as long as mouse moves
84// Stop drawing once a new left click is detected by the on_mouse function
85void on_mouse(int event, int x, int y, int , void * )
86{
87 // Action when left button is clicked
88 if(event == EVENT_LBUTTONDOWN)
89 {
90 if(!start_draw)
91 {
92 roi_x0 = x;
93 roi_y0 = y;
94 start_draw = true;
95 } else {
96 roi_x1 = x;
97 roi_y1 = y;
98 start_draw = false;
99 }
100 }
101
102 // Action when mouse is moving and drawing is enabled
103 if((event == EVENT_MOUSEMOVE) && start_draw)
104 {
105 // Redraw bounding box for annotation
106 Mat current_view;
107 image.copyTo(current_view);
108 rectangle(current_view, Point(roi_x0,roi_y0), Point(x,y), Scalar(0,0,255));
109 imshow(window_name, current_view);
110 }
111}
112
113// FUNCTION : returns a vector of Rect objects given an image containing positive object instances
114vector<Rect> get_annotations(Mat input_image)
115{
116 vector<Rect> current_annotations;
117
118 // Make it possible to exit the annotation process
119 stop = false;
120
121 // Init window interface and couple mouse actions
122 namedWindow(window_name, WINDOW_AUTOSIZE);
123 setMouseCallback(window_name, on_mouse);
124
125 image = input_image;
126 imshow(window_name, image);
127 int key_pressed = 0;
128
129 do
130 {
131 // Get a temporary image clone
132 Mat temp_image = input_image.clone();
133 Rect currentRect(0, 0, 0, 0);
134
135 // Keys for processing
136 // You need to select one for confirming a selection and one to continue to the next image
137 // Based on the universal ASCII code of the keystroke: http://www.asciitable.com/
138 // <c> = 99 add rectangle to current image
139 // <n> = 110 save added rectangles and show next image
140 // <d> = 100 delete the last annotation made
141 // <ESC> = 27 exit program
142 key_pressed = 0xFF & waitKey(0);
143 switch( key_pressed )
144 {
145 case 27:
146 destroyWindow(window_name);
147 stop = true;
148 break;
149 case 99:
150 // Draw initiated from top left corner
151 if(roi_x0<roi_x1 && roi_y0<roi_y1)
152 {
153 currentRect.x = roi_x0;
154 currentRect.y = roi_y0;
155 currentRect.width = roi_x1-roi_x0;
156 currentRect.height = roi_y1-roi_y0;
157 }
158 // Draw initiated from bottom right corner
159 if(roi_x0>roi_x1 && roi_y0>roi_y1)
160 {
161 currentRect.x = roi_x1;
162 currentRect.y = roi_y1;
163 currentRect.width = roi_x0-roi_x1;
164 currentRect.height = roi_y0-roi_y1;
165 }
166 // Draw initiated from top right corner
167 if(roi_x0>roi_x1 && roi_y0<roi_y1)
168 {
169 currentRect.x = roi_x1;
170 currentRect.y = roi_y0;
171 currentRect.width = roi_x0-roi_x1;
172 currentRect.height = roi_y1-roi_y0;
173 }
174 // Draw initiated from bottom left corner
175 if(roi_x0<roi_x1 && roi_y0>roi_y1)
176 {
177 currentRect.x = roi_x0;
178 currentRect.y = roi_y1;
179 currentRect.width = roi_x1-roi_x0;
180 currentRect.height = roi_y0-roi_y1;
181 }
182 // Draw the rectangle on the canvas
183 // Add the rectangle to the vector of annotations
184 current_annotations.push_back(currentRect);
185 break;
186 case 100:
187 // Remove the last annotation
188 if(current_annotations.size() > 0){
189 current_annotations.pop_back();
190 }
191 break;
192 default:
193 // Default case --> do nothing at all
194 // Other keystrokes can simply be ignored
195 break;
196 }
197
198 // Check if escape has been pressed
199 if(stop)
200 {
201 break;
202 }
203
204 // Draw all the current rectangles onto the top image and make sure that the global image is linked
205 for(int i=0; i < (int)current_annotations.size(); i++){
206 rectangle(temp_image, current_annotations[i], Scalar(0,255,0), 1);
207 }
208 image = temp_image;
209
210 // Force an explicit redraw of the canvas --> necessary to visualize delete correctly
211 imshow(window_name, image);
212 }
213 // Continue as long as the next image key has not been pressed
214 while(key_pressed != 110);
215
216 // Close down the window
217 destroyWindow(window_name);
218
219 // Return the data
220 return current_annotations;
221}
222
223int main( int argc, const char** argv )
224{
225 // If no arguments are given, then supply some information on how this tool works
226 if( argc == 1 ){
227 cout << "Usage: " << argv[0] << endl;
228 cout << " -images <folder_location> [example - /data/testimages/]" << endl;
229 cout << " -annotations <ouput_file> [example - /data/annotations.txt]" << endl;
230 cout << "TIP: Use absolute paths to avoid any problems with the software!" << endl;
231 return -1;
232 }
233
234 // Read in the input arguments
235 string image_folder;
236 string annotations_file;
237 for(int i = 1; i < argc; ++i )
238 {
239 if( !strcmp( argv[i], "-images" ) )
240 {
241 image_folder = argv[++i];
242 }
243 else if( !strcmp( argv[i], "-annotations" ) )
244 {
245 annotations_file = argv[++i];
246 }
247 }
248
249 // Check if the folder actually exists
250 // If -1 is returned then the folder actually exists, and thus you can continue
251 // In all other cases there was a folder creation and thus the folder did not exist
252 #if defined(_WIN32)
253 if(_mkdir(image_folder.c_str()) != -1){
254 // Generate an error message
255 cerr << "The image folder given does not exist. Please check again!" << endl;
256 // Remove the created folder again, to ensure a second run with same code fails again
257 _rmdir(image_folder.c_str());
258 return 0;
259 }
260 #else
261 if(mkdir(image_folder.c_str(), 0777) != -1){
262 // Generate an error message
263 cerr << "The image folder given does not exist. Please check again!" << endl;
264 // Remove the created folder again, to ensure a second run with same code fails again
265 remove(image_folder.c_str());
266 return 0;
267 }
268 #endif
269
270 // Start by processing the data
271 // Return the image filenames inside the image folder
272 vector< vector<Rect> > annotations;
273 vector<String> filenames;
274 String folder(image_folder);
275 glob(folder, filenames);
276
277 // Loop through each image stored in the images folder
278 // Create and temporarily store the annotations
279 // At the end write everything to the annotations file
280 for (size_t i = 0; i < filenames.size(); i++){
281 // Read in an image
282 Mat current_image = imread(filenames[i]);
283
284 // Check if the image is actually read - avoid other files in the folder, because glob() takes them all
285 // If not then simply skip this iteration
286 if(current_image.empty()){
287 continue;
288 }
289
290 // Perform annotations & store the result inside the vectorized structure
291 vector<Rect> current_annotations = get_annotations(current_image);
292 annotations.push_back(current_annotations);
293
294 // Check if the ESC key was hit, then exit earlier then expected
295 if(stop){
296 break;
297 }
298 }
299
300 // When all data is processed, store the data gathered inside the proper file
301 // This now even gets called when the ESC button was hit to store preliminary results
302 ofstream output(annotations_file.c_str());
303 if ( !output.is_open() ){
304 cerr << "The path for the output file contains an error and could not be opened. Please check again!" << endl;
305 return 0;
306 }
307
308 // Store the annotations, write to the output file
309 for(int i = 0; i < (int)annotations.size(); i++){
310 output << filenames[i] << " " << annotations[i].size();
311 for(int j=0; j < (int)annotations[i].size(); j++){
312 Rect temp = annotations[i][j];
313 output << " " << temp.x << " " << temp.y << " " << temp.width << " " << temp.height;
314 }
315 output << endl;
316 }
317
318 return 0;
319}
