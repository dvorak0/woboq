42/*
43 * createsamples.cpp
44 *
45 * Create test/training samples
46 */
47
48#include <cstdio>
49#include <cstring>
50#include <cstdlib>
51#include <cmath>
52#include <ctime>
53#include <memory>
54
55// std::auto_ptr
56#if defined(__GNUC__) && __GNUC__ >= 6
57#pragma GCC diagnostic ignored "-Wdeprecated-declarations"
58#endif
59
60using namespace std;
61
62#include "cvhaartraining.h"
63#include "ioutput.h"
64
65int main( int argc, char* argv[] )
66{
67 int i = 0;
68 char* nullname = (char*)"(NULL)";
69 char* vecname = NULL; /* .vec file name */
70 char* infoname = NULL; /* file name with marked up image descriptions */
71 char* imagename = NULL; /* single sample image */
72 char* bgfilename = NULL; /* background */
73 int num = 1000;
74 int bgcolor = 0;
75 int bgthreshold = 80;
76 int invert = 0;
77 int maxintensitydev = 40;
78 double maxxangle = 1.1;
79 double maxyangle = 1.1;
80 double maxzangle = 0.5;
81 bool showsamples = false;
82 /* the samples are adjusted to this scale in the sample preview window */
83 double scale = 4.0;
84 int width = 24;
85 int height = 24;
86 bool pngoutput = false; /* whether to make the samples in png or in jpg*/
87
88 srand((unsigned int)time(0));
89
90 if( argc == 1 )
91 {
92 printf( "Usage: %s\n [-info <collection_file_name>]\n"
93 " [-img <image_file_name>]\n"
94 " [-vec <vec_file_name>]\n"
95 " [-bg <background_file_name>]\n [-num <number_of_samples = %d>]\n"
96 " [-bgcolor <background_color = %d>]\n"
97 " [-inv] [-randinv] [-bgthresh <background_color_threshold = %d>]\n"
98 " [-maxidev <max_intensity_deviation = %d>]\n"
99 " [-maxxangle <max_x_rotation_angle = %f>]\n"
100 " [-maxyangle <max_y_rotation_angle = %f>]\n"
101 " [-maxzangle <max_z_rotation_angle = %f>]\n"
102 " [-show [<scale = %f>]]\n"
103 " [-w <sample_width = %d>]\n [-h <sample_height = %d>]\n"
104 " [-pngoutput]",
105 argv[0], num, bgcolor, bgthreshold, maxintensitydev,
106 maxxangle, maxyangle, maxzangle, scale, width, height );
107
108 return 0;
109 }
110
111 for( i = 1; i < argc; ++i )
112 {
113 if( !strcmp( argv[i], "-info" ) )
114 {
115 infoname = argv[++i];
116 }
117 else if( !strcmp( argv[i], "-img" ) )
118 {
119 imagename = argv[++i];
120 }
121 else if( !strcmp( argv[i], "-vec" ) )
122 {
123 vecname = argv[++i];
124 }
125 else if( !strcmp( argv[i], "-bg" ) )
126 {
127 bgfilename = argv[++i];
128 }
129 else if( !strcmp( argv[i], "-num" ) )
130 {
131 num = atoi( argv[++i] );
132 }
133 else if( !strcmp( argv[i], "-bgcolor" ) )
134 {
135 bgcolor = atoi( argv[++i] );
136 }
137 else if( !strcmp( argv[i], "-bgthresh" ) )
138 {
139 bgthreshold = atoi( argv[++i] );
140 }
141 else if( !strcmp( argv[i], "-inv" ) )
142 {
143 invert = 1;
144 }
145 else if( !strcmp( argv[i], "-randinv" ) )
146 {
147 invert = CV_RANDOM_INVERT;
148 }
149 else if( !strcmp( argv[i], "-maxidev" ) )
150 {
151 maxintensitydev = atoi( argv[++i] );
152 }
153 else if( !strcmp( argv[i], "-maxxangle" ) )
154 {
155 maxxangle = atof( argv[++i] );
156 }
157 else if( !strcmp( argv[i], "-maxyangle" ) )
158 {
159 maxyangle = atof( argv[++i] );
160 }
161 else if( !strcmp( argv[i], "-maxzangle" ) )
162 {
163 maxzangle = atof( argv[++i] );
164 }
165 else if( !strcmp( argv[i], "-show" ) )
166 {
167 showsamples = true;
168 if( i+1 < argc && strlen( argv[i+1] ) > 0 && argv[i+1][0] != '-' )
169 {
170 double d;
171 d = strtod( argv[i+1], 0 );
172 if( d != -HUGE_VAL && d != HUGE_VAL && d > 0 ) scale = d;
173 ++i;
174 }
175 }
176 else if( !strcmp( argv[i], "-w" ) )
177 {
178 width = atoi( argv[++i] );
179 }
180 else if( !strcmp( argv[i], "-h" ) )
181 {
182 height = atoi( argv[++i] );
183 }
184 else if( !strcmp( argv[i], "-pngoutput" ) )
185 {
186 pngoutput = true;
187 }
188 }
189
190 printf( "Info file name: %s\n", ((infoname == NULL) ? nullname : infoname ) );
191 printf( "Img file name: %s\n", ((imagename == NULL) ? nullname : imagename ) );
192 printf( "Vec file name: %s\n", ((vecname == NULL) ? nullname : vecname ) );
193 printf( "BG file name: %s\n", ((bgfilename == NULL) ? nullname : bgfilename ) );
194 printf( "Num: %d\n", num );
195 printf( "BG color: %d\n", bgcolor );
196 printf( "BG threshold: %d\n", bgthreshold );
197 printf( "Invert: %s\n", (invert == CV_RANDOM_INVERT) ? "RANDOM"
198 : ( (invert) ? "TRUE" : "FALSE" ) );
199 printf( "Max intensity deviation: %d\n", maxintensitydev );
200 printf( "Max x angle: %g\n", maxxangle );
201 printf( "Max y angle: %g\n", maxyangle );
202 printf( "Max z angle: %g\n", maxzangle );
203 printf( "Show samples: %s\n", (showsamples) ? "TRUE" : "FALSE" );
204 if( showsamples )
205 {
206 printf( "Scale applied to display : %g\n", scale );
207 }
208 if( !pngoutput)
209 {
210 printf( "Original image will be scaled to:\n");
211 printf( "\tWidth: $backgroundWidth / %d\n", width );
212 printf( "\tHeight: $backgroundHeight / %d\n", height );
213 }
214
215 /* determine action */
216 if( imagename && vecname )
217 {
218 printf( "Create training samples from single image applying distortions...\n" );
219
220 cvCreateTrainingSamples( vecname, imagename, bgcolor, bgthreshold, bgfilename,
221 num, invert, maxintensitydev,
222 maxxangle, maxyangle, maxzangle,
223 showsamples, width, height );
224
225 printf( "Done\n" );
226 }
227 else if( imagename && bgfilename && infoname)
228 {
229 printf( "Create data set from single image applying distortions...\n"
230 "Output format: %s\n",
231 (( pngoutput ) ? "PNG" : "JPG") );
232
233 std::auto_ptr<DatasetGenerator> creator;
234 if( pngoutput )
235 {
236 creator = std::auto_ptr<DatasetGenerator>( new PngDatasetGenerator( infoname ) );
237 }
238 else
239 {
240 creator = std::auto_ptr<DatasetGenerator>( new JpgDatasetGenerator( infoname ) );
241 }
242 creator->create( imagename, bgcolor, bgthreshold, bgfilename, num,
243 invert, maxintensitydev, maxxangle, maxyangle, maxzangle,
244 showsamples, width, height );
245
246 printf( "Done\n" );
247 }
248 else if( infoname && vecname )
249 {
250 int total;
251
252 printf( "Create training samples from images collection...\n" );
253
254 total = cvCreateTrainingSamplesFromInfo( infoname, vecname, num, showsamples,
255 width, height );
256
257 printf( "Done. Created %d samples\n", total );
258 }
259 else if( vecname )
260 {
261 printf( "View samples from vec file (press ESC to exit)...\n" );
262
263 cvShowVecSamples( vecname, width, height, scale );
264
265 printf( "Done\n" );
266 }
267 else
268 {
269 printf( "Nothing to do\n" );
270 }
271
272 return 0;
273}
