58int icvMkDir( const char* filename )
59{
60 char path[PATH_MAX];
61 char* p;
62 int pos;
63
64#ifdef _WIN32
65 struct _stat st;
66#else /* _WIN32 */
67 struct stat st;
68 mode_t mode;
69
70 mode = 0755;
71#endif /* _WIN32 */
72
73 strcpy( path, filename );
74
75 p = path;
76 for( ; ; )
77 {
78 pos = (int)strcspn( p, "/\\" );
79
80 if( pos == (int) strlen( p ) ) break;
81 if( pos != 0 )
82 {
83 p[pos] = '\0';
84
85#ifdef _WIN32
86 if( p[pos-1] != ':' )
87 {
88 if( _stat( path, &st ) != 0 )
89 {
90 if( _mkdir( path ) != 0 ) return 0;
91 }
92 }
93#else /* _WIN32 */
94 if( stat( path, &st ) != 0 )
95 {
96 if( mkdir( path, mode ) != 0 ) return 0;
97 }
98#endif /* _WIN32 */
99 }
100
101 p[pos] = '/';
102
103 p += pos + 1;
104 }
105
106 return 1;
107}
