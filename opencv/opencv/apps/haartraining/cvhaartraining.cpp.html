112/*
113 * get sum image offsets for <rect> corner points
114 * step - row step (measured in image pixels!) of sum image
115 */
116#define CV_SUM_OFFSETS( p0, p1, p2, p3, rect, step ) \
117 /* (x, y) */ \
118 (p0) = (rect).x + (step) * (rect).y; \
119 /* (x + w, y) */ \
120 (p1) = (rect).x + (rect).width + (step) * (rect).y; \
121 /* (x + w, y) */ \
122 (p2) = (rect).x + (step) * ((rect).y + (rect).height); \
123 /* (x + w, y + h) */ \
124 (p3) = (rect).x + (rect).width + (step) * ((rect).y + (rect).height);
125
126/*
127 * get tilted image offsets for <rect> corner points
128 * step - row step (measured in image pixels!) of tilted image
129 */
130#define CV_TILTED_OFFSETS( p0, p1, p2, p3, rect, step ) \
131 /* (x, y) */ \
132 (p0) = (rect).x + (step) * (rect).y; \
133 /* (x - h, y + h) */ \
134 (p1) = (rect).x - (rect).height + (step) * ((rect).y + (rect).height);\
135 /* (x + w, y + w) */ \
136 (p2) = (rect).x + (rect).width + (step) * ((rect).y + (rect).width); \
137 /* (x + w - h, y + w + h) */ \
138 (p3) = (rect).x + (rect).width - (rect).height \
139 + (step) * ((rect).y + (rect).width + (rect).height);

142/*
143 * icvCreateIntHaarFeatures
144 *
145 * Create internal representation of haar features
146 *
147 * mode:
148 * 0 - BASIC = Viola
149 * 1 - CORE = All upright

160 CvMemStorage* storage = NULL;
161 CvSeq* seq = NULL;
162 CvSeqWriter writer;
163
164 int s0 = 36; /* minimum total area size of basic haar feature */
165 int s1 = 12; /* minimum total area size of tilted haar features 2 */
166 int s2 = 18; /* minimum total area size of tilted haar features 3 */
167 int s3 = 24; /* minimum total area size of tilted haar features 4 */
168
169 int x = 0;
170 int y = 0;
171 int dx = 0;
172 int dy = 0;
173
174#if 0
175 float factor = 1.0F;
176
177 factor = ((float) winsize.width) * winsize.height / (24 * 24);
178
179 s0 = (int) (s0 * factor);
180 s1 = (int) (s1 * factor);
181 s2 = (int) (s2 * factor);
182 s3 = (int) (s3 * factor);
183#else
184 s0 = 1;
185 s1 = 1;
186 s2 = 1;
187 s3 = 1;
188#endif
189
190 /* CV_VECTOR_CREATE( vec, CvIntHaarFeature, size, maxsize ) */

196 for( y = 0; y < winsize.height; y++ )
197 {
198 for( dx = 1; dx <= winsize.width; dx++ )
199 {

202 // haar_x2

204 if (dx*2*dy < s0) continue;
205 if (!symmetric || (x+x+dx*2 <=winsize.width)) {
206 haarFeature = cvHaarFeature( "haar_x2",
207 x, y, dx*2, dy, -1,
208 x+dx, y, dx , dy, +2 );
209 /* CV_VECTOR_PUSH( vec, CvIntHaarFeature, haarFeature, size, maxsize, step ) */
210 CV_WRITE_SEQ_ELEM( haarFeature, writer );
211 }
212 }
213
214 // haar_y2
215 if ( (x+dx <= winsize.width) && (y+dy*2 <= winsize.height) ) {
216 if (dx*2*dy < s0) continue;
217 if (!symmetric || (x+x+dx <= winsize.width)) {
218 haarFeature = cvHaarFeature( "haar_y2",
219 x, y, dx, dy*2, -1,
220 x, y+dy, dx, dy, +2 );
221 CV_WRITE_SEQ_ELEM( haarFeature, writer );

223 }
224
225 // haar_x3
226 if ( (x+dx*3 <= winsize.width) && (y+dy <= winsize.height) ) {
227 if (dx*3*dy < s0) continue;
228 if (!symmetric || (x+x+dx*3 <=winsize.width)) {
229 haarFeature = cvHaarFeature( "haar_x3",
230 x, y, dx*3, dy, -1,
231 x+dx, y, dx, dy, +3 );
232 CV_WRITE_SEQ_ELEM( haarFeature, writer );
233 }
234 }
235
236 // haar_y3
237 if ( (x+dx <= winsize.width) && (y+dy*3 <= winsize.height) ) {
238 if (dx*3*dy < s0) continue;
239 if (!symmetric || (x+x+dx <= winsize.width)) {
240 haarFeature = cvHaarFeature( "haar_y3",
241 x, y, dx, dy*3, -1,
242 x, y+dy, dx, dy, +3 );
243 CV_WRITE_SEQ_ELEM( haarFeature, writer );
244 }
245 }
246
247 if( mode != 0 /*BASIC*/ ) {
248 // haar_x4
249 if ( (x+dx*4 <= winsize.width) && (y+dy <= winsize.height) ) {
250 if (dx*4*dy < s0) continue;
251 if (!symmetric || (x+x+dx*4 <=winsize.width)) {
252 haarFeature = cvHaarFeature( "haar_x4",
253 x, y, dx*4, dy, -1,
254 x+dx, y, dx*2, dy, +2 );
255 CV_WRITE_SEQ_ELEM( haarFeature, writer );
256 }
257 }
258
259 // haar_y4
260 if ( (x+dx <= winsize.width ) && (y+dy*4 <= winsize.height) ) {
261 if (dx*4*dy < s0) continue;
262 if (!symmetric || (x+x+dx <=winsize.width)) {
263 haarFeature = cvHaarFeature( "haar_y4",
264 x, y, dx, dy*4, -1,
265 x, y+dy, dx, dy*2, +2 );
266 CV_WRITE_SEQ_ELEM( haarFeature, writer );
267 }
268 }
269 }
270
271 // x2_y2
272 if ( (x+dx*2 <= winsize.width) && (y+dy*2 <= winsize.height) ) {
273 if (dx*4*dy < s0) continue;
274 if (!symmetric || (x+x+dx*2 <=winsize.width)) {
275 haarFeature = cvHaarFeature( "haar_x2_y2",
276 x , y, dx*2, dy*2, -1,
277 x , y , dx , dy, +2,
278 x+dx, y+dy, dx , dy, +2 );
279 CV_WRITE_SEQ_ELEM( haarFeature, writer );
280 }
281 }
282
283 if (mode != 0 /*BASIC*/) {
284 // point
285 if ( (x+dx*3 <= winsize.width) && (y+dy*3 <= winsize.height) ) {
286 if (dx*9*dy < s0) continue;
287 if (!symmetric || (x+x+dx*3 <=winsize.width)) {
288 haarFeature = cvHaarFeature( "haar_point",
289 x , y, dx*3, dy*3, -1,
290 x+dx, y+dy, dx , dy , +9);
291 CV_WRITE_SEQ_ELEM( haarFeature, writer );
292 }
293 }
294 }
295
296 if (mode == 2 /*ALL*/) {
297 // tilted haar_x2 (x, y, w, h, b, weight)
298 if ( (x+2*dx <= winsize.width) && (y+2*dx+dy <= winsize.height) && (x-dy>= 0) ) {
299 if (dx*2*dy < s1) continue;
300
301 if (!symmetric || (x <= (winsize.width / 2) )) {
302 haarFeature = cvHaarFeature( "tilted_haar_x2",
303 x, y, dx*2, dy, -1,
304 x, y, dx , dy, +2 );
305 CV_WRITE_SEQ_ELEM( haarFeature, writer );
306 }
307 }
308
309 // tilted haar_y2 (x, y, w, h, b, weight)
310 if ( (x+dx <= winsize.width) && (y+dx+2*dy <= winsize.height) && (x-2*dy>= 0) ) {
311 if (dx*2*dy < s1) continue;
312
313 if (!symmetric || (x <= (winsize.width / 2) )) {
314 haarFeature = cvHaarFeature( "tilted_haar_y2",
315 x, y, dx, 2*dy, -1,
316 x, y, dx, dy, +2 );
317 CV_WRITE_SEQ_ELEM( haarFeature, writer );
318 }
319 }
320
321 // tilted haar_x3 (x, y, w, h, b, weight)
322 if ( (x+3*dx <= winsize.width) && (y+3*dx+dy <= winsize.height) && (x-dy>= 0) ) {
323 if (dx*3*dy < s2) continue;
324
325 if (!symmetric || (x <= (winsize.width / 2) )) {
326 haarFeature = cvHaarFeature( "tilted_haar_x3",
327 x, y, dx*3, dy, -1,
328 x+dx, y+dx, dx , dy, +3 );
329 CV_WRITE_SEQ_ELEM( haarFeature, writer );
330 }
331 }
332
333 // tilted haar_y3 (x, y, w, h, b, weight)
334 if ( (x+dx <= winsize.width) && (y+dx+3*dy <= winsize.height) && (x-3*dy>= 0) ) {
335 if (dx*3*dy < s2) continue;
336
337 if (!symmetric || (x <= (winsize.width / 2) )) {
338 haarFeature = cvHaarFeature( "tilted_haar_y3",
339 x, y, dx, 3*dy, -1,
340 x-dy, y+dy, dx, dy, +3 );
341 CV_WRITE_SEQ_ELEM( haarFeature, writer );
342 }
343 }
344
345
346 // tilted haar_x4 (x, y, w, h, b, weight)
347 if ( (x+4*dx <= winsize.width) && (y+4*dx+dy <= winsize.height) && (x-dy>= 0) ) {
348 if (dx*4*dy < s3) continue;
349
350 if (!symmetric || (x <= (winsize.width / 2) )) {
351 haarFeature = cvHaarFeature( "tilted_haar_x4",
352
353
354 x, y, dx*4, dy, -1,
355 x+dx, y+dx, dx*2, dy, +2 );
356 CV_WRITE_SEQ_ELEM( haarFeature, writer );
357 }
358 }
359
360 // tilted haar_y4 (x, y, w, h, b, weight)
361 if ( (x+dx <= winsize.width) && (y+dx+4*dy <= winsize.height) && (x-4*dy>= 0) ) {
362 if (dx*4*dy < s3) continue;
363
364 if (!symmetric || (x <= (winsize.width / 2) )) {
365 haarFeature = cvHaarFeature( "tilted_haar_y4",
366 x, y, dx, 4*dy, -1,
367 x-dy, y+dy, dx, 2*dy, +2 );
368 CV_WRITE_SEQ_ELEM( haarFeature, writer );
369 }
370 }
371
372
373 /*
374
375 // tilted point
376 if ( (x+dx*3 <= winsize.width - 1) && (y+dy*3 <= winsize.height - 1) && (x-3*dy>= 0)) {
377 if (dx*9*dy < 36) continue;
378 if (!symmetric || (x <= (winsize.width / 2) )) {
379 haarFeature = cvHaarFeature( "tilted_haar_point",
380 x, y, dx*3, dy*3, -1,
381 x, y+dy, dx , dy, +9 );
382 CV_WRITE_SEQ_ELEM( haarFeature, writer );
383 }
384 }
385 */
386 }
387 }
388 }
389 }
390 }
391
392 seq = cvEndWriteSeq( &writer );
393 features = (CvIntHaarFeatures*) cvAlloc( sizeof( CvIntHaarFeatures ) +
394 ( sizeof( CvTHaarFeature ) + sizeof( CvFastHaarFeature ) ) * seq->total );
395 features->feature = (CvTHaarFeature*) (features + 1);
396 features->fastfeature = (CvFastHaarFeature*) ( features->feature + seq->total );
397 features->count = seq->total;
398 features->winsize = winsize;
399 cvCvtSeqToArray( seq, (CvArr*) features->feature );
400 cvReleaseMemStorage( &storage );
401
402 icvConvertToFastHaarFeature( features->feature, features->fastfeature,
403 features->count, (winsize.width + 1) );
404
405 return features;
406}

419void icvConvertToFastHaarFeature( CvTHaarFeature* haarFeature,
420 CvFastHaarFeature* fastHaarFeature,
421 int size, int step )
422{
423 int i = 0;
424 int j = 0;
425
426 for( i = 0; i < size; i++ )
427 {
428 fastHaarFeature[i].tilted = haarFeature[i].tilted;
429 if( !fastHaarFeature[i].tilted )
430 {
431 for( j = 0; j < CV_HAAR_FEATURE_MAX; j++ )
432 {
433 fastHaarFeature[i].rect[j].weight = haarFeature[i].rect[j].weight;
434 if( fastHaarFeature[i].rect[j].weight == 0.0F )
435 {
436 break;
437 }
438 CV_SUM_OFFSETS( fastHaarFeature[i].rect[j].p0,
439 fastHaarFeature[i].rect[j].p1,
440 fastHaarFeature[i].rect[j].p2,
441 fastHaarFeature[i].rect[j].p3,
442 haarFeature[i].rect[j].r, step )
443 }
444
445 }
446 else
447 {
448 for( j = 0; j < CV_HAAR_FEATURE_MAX; j++ )
449 {
450 fastHaarFeature[i].rect[j].weight = haarFeature[i].rect[j].weight;
451 if( fastHaarFeature[i].rect[j].weight == 0.0F )
452 {
453 break;
454 }
455 CV_TILTED_OFFSETS( fastHaarFeature[i].rect[j].p0,
456 fastHaarFeature[i].rect[j].p1,
457 fastHaarFeature[i].rect[j].p2,
458 fastHaarFeature[i].rect[j].p3,
459 haarFeature[i].rect[j].r, step )
460 }
461 }
462 }

477
478 __BEGIN__;
479
480 data = NULL;

485 /* sum and tilted */

489 sizeof( float ) /* weight */
490 ) * maxnumsamples;
491

493 memset( (void*)data, 0, datasize );
494 data->maxnum = maxnumsamples;
495 data->winsize = winsize;
496 ptr = (uchar*)(data + 1);
497 data->sum = cvMat( maxnumsamples, (winsize.width + 1) * (winsize.height + 1),
498 CV_SUM_MAT_TYPE, (void*) ptr );
499 ptr += sizeof( sum_type ) * maxnumsamples * (winsize.width+1) * (winsize.height+1);
500 data->tilted = cvMat( maxnumsamples, (winsize.width + 1) * (winsize.height + 1),
501 CV_SUM_MAT_TYPE, (void*) ptr );
502 ptr += sizeof( sum_type ) * maxnumsamples * (winsize.width+1) * (winsize.height+1);
503 data->normfactor = cvMat( 1, maxnumsamples, CV_32FC1, (void*) ptr );
504 ptr += sizeof( float ) * maxnumsamples;
505 data->cls = cvMat( 1, maxnumsamples, CV_32FC1, (void*) ptr );
506 ptr += sizeof( float ) * maxnumsamples;
507 data->weights = cvMat( 1, maxnumsamples, CV_32FC1, (void*) ptr );
508
509 data->valcache = NULL;
510 data->idxcache = NULL;
511
512 __END__;
513
514 return data;
515}

549{

551 int j = 0;

563

566 if( sampleIdx == NULL )
567 {
568 int num_samples;
569
570#ifdef CV_COL_ARRANGEMENT

572#else
573 num_samples = mat->rows;
574#endif
575 for( i = 0; i < num_samples; i++ )
576 {
577 for( j = 0; j < num; j++ )
578 {
579 val = cvEvalFastHaarFeature(
580 ( haar_features->fastfeature
581 + first + j ),
582 (sum_type*) (training_data->sum.data.ptr
583 + i * training_data->sum.step),
584 (sum_type*) (training_data->tilted.data.ptr
585 + i * training_data->tilted.step) );
586 normfactor = training_data->normfactor.data.fl[i];
587 val = ( normfactor == 0.0F ) ? 0.0F : (val / normfactor);
588
589#ifdef CV_COL_ARRANGEMENT
590 CV_MAT_ELEM( *mat, float, j, i ) = val;
591#else
592 CV_MAT_ELEM( *mat, float, i, j ) = val;
593#endif
594 }
595 }
596 }
597 else
598 {
599 uchar* idxdata = NULL;
600 size_t step = 0;
601 int numidx = 0;
602 int idx = 0;
603
604 assert( CV_MAT_TYPE( sampleIdx->type ) == CV_32FC1 );
605
606 idxdata = sampleIdx->data.ptr;
607 if( sampleIdx->rows == 1 )
608 {
609 step = sizeof( float );
610 numidx = sampleIdx->cols;
611 }
612 else
613 {
614 step = sampleIdx->step;
615 numidx = sampleIdx->rows;
616 }
617
618 for( i = 0; i < numidx; i++ )
619 {
620 for( j = 0; j < num; j++ )
621 {
622 idx = (int)( *((float*) (idxdata + i * step)) );
623 val = cvEvalFastHaarFeature(
624 ( haar_features->fastfeature
625 + first + j ),
626 (sum_type*) (training_data->sum.data.ptr
627 + idx * training_data->sum.step),
628 (sum_type*) (training_data->tilted.data.ptr
629 + idx * training_data->tilted.step) );
630 normfactor = training_data->normfactor.data.fl[idx];
631 val = ( normfactor == 0.0F ) ? 0.0F : (val / normfactor);
632
633#ifdef CV_COL_ARRANGEMENT
634 CV_MAT_ELEM( *mat, float, j, idx ) = val;
635#else
636 CV_MAT_ELEM( *mat, float, idx, j ) = val;
637#endif
638
639 }
640 }
641 }
642#if 0 /*def CV_VERBOSE*/
643 if( first % 5000 == 0 )
644 {
645 fprintf( stderr, "%3d%%\r", (int) (100.0 * first /
646 haar_features->count) );
647 fflush( stderr );
648 }
649#endif /* CV_VERBOSE */
650}

652static
653void icvPrecalculate( CvHaarTrainingData* data, CvIntHaarFeatures* haarFeatures,
654 int numprecalculated )
655{
656 CV_FUNCNAME( "icvPrecalculate" );
657
658 __BEGIN__;
659
660 icvReleaseHaarTrainingDataCache( &data );
661
662 numprecalculated -= numprecalculated % CV_STUMP_TRAIN_PORTION;
663 numprecalculated = MIN( numprecalculated, haarFeatures->count );
664
665 if( numprecalculated > 0 )
666 {
667 //size_t datasize;
668 int m;
669 CvUserdata userdata;
670
671 /* private variables */
672 #ifdef CV_OPENMP
673 CvMat t_data;
674 CvMat t_idx;
675 int first;
676 int t_portion;
677 int portion = CV_STUMP_TRAIN_PORTION;
678 #endif /* CV_OPENMP */
679
680 m = data->sum.rows;
681
682#ifdef CV_COL_ARRANGEMENT

684#else

811/*
812 * icvCreateCARTStageClassifier
813 *
814 * Create stage classifier with trees as weak classifiers
815 * data - haar training data. It must be created and filled before call
816 * minhitrate - desired min hit rate
817 * maxfalsealarm - desired max false alarm rate
818 * symmetric - if not 0 it is assumed that samples are vertically symmetric
819 * numprecalculated - number of features that will be precalculated. Each precalculated
820 * feature need (number_of_samples*(sizeof( float ) + sizeof( short ))) bytes of memory
821 * weightfraction - weight trimming parameter
822 * numsplits - number of binary splits in each tree
823 * boosttype - type of applied boosting algorithm
824 * stumperror - type of used error if Discrete AdaBoost algorithm is applied
825 * maxsplits - maximum total number of splits in all weak classifiers.
826 * If it is not 0 then NULL returned if total number of splits exceeds <maxsplits>.
827 */
828static
829CvIntHaarClassifier* icvCreateCARTStageClassifier( CvHaarTrainingData* data,
830 CvMat* sampleIdx,
831 CvIntHaarFeatures* haarFeatures,
832 float minhitrate,
833 float maxfalsealarm,
834 int symmetric,
835 float weightfraction,
836 int numsplits,
837 CvBoostType boosttype,
838 CvStumpError stumperror,
839 int maxsplits )
840{
841
842#ifdef CV_COL_ARRANGEMENT
843 int flags = CV_COL_SAMPLE;
844#else
845 int flags = CV_ROW_SAMPLE;
846#endif
847
848 CvStageHaarClassifier* stage = NULL;
849 CvBoostTrainer* trainer;
850 CvCARTClassifier* cart = NULL;
851 CvCARTTrainParams trainParams;
852 CvMTStumpTrainParams stumpTrainParams;
853 //CvMat* trainData = NULL;
854 //CvMat* sortedIdx = NULL;
855 CvMat eval;
856 int n = 0;
857 int m = 0;
858 int numpos = 0;
859 int numneg = 0;
860 int numfalse = 0;
861 float sum_stage = 0.0F;
862 float threshold = 0.0F;
863 float falsealarm = 0.0F;
864
865 //CvMat* sampleIdx = NULL;
866 CvMat* trimmedIdx;
867 //float* idxdata = NULL;
868 //float* tempweights = NULL;
869 //int idxcount = 0;
870 CvUserdata userdata;
871
872 int i = 0;
873 int j = 0;
874 int idx;
875 int numsamples;
876 int numtrimmed;
877
878 CvCARTHaarClassifier* classifier;
879 CvSeq* seq = NULL;
880 CvMemStorage* storage = NULL;
881 CvMat* weakTrainVals;
882 float alpha;
883 float sumalpha;
884 int num_splits; /* total number of splits in all weak classifiers */
885
886#ifdef CV_VERBOSE
887 printf( "+----+----+-+---------+---------+---------+---------+\n" );
888 printf( "| N |%%SMP|F| ST.THR | HR | FA | EXP. ERR|\n" );
889 printf( "+----+----+-+---------+---------+---------+---------+\n" );
890#endif /* CV_VERBOSE */
891
892 n = haarFeatures->count;
893 m = data->sum.rows;
894 numsamples = (sampleIdx) ? MAX( sampleIdx->rows, sampleIdx->cols ) : m;
895
896 userdata = cvUserdata( data, haarFeatures );
897
898 stumpTrainParams.type = ( boosttype == CV_DABCLASS )
899 ? CV_CLASSIFICATION_CLASS : CV_REGRESSION;
900 stumpTrainParams.error = ( boosttype == CV_LBCLASS || boosttype == CV_GABCLASS )
901 ? CV_SQUARE : stumperror;
902 stumpTrainParams.portion = CV_STUMP_TRAIN_PORTION;
903 stumpTrainParams.getTrainData = icvGetTrainingDataCallback;
904 stumpTrainParams.numcomp = n;
905 stumpTrainParams.userdata = &userdata;
906 stumpTrainParams.sortedIdx = data->idxcache;
907
908 trainParams.count = numsplits;
909 trainParams.stumpTrainParams = (CvClassifierTrainParams*) &stumpTrainParams;
910 trainParams.stumpConstructor = cvCreateMTStumpClassifier;
911 trainParams.splitIdx = icvSplitIndicesCallback;
912 trainParams.userdata = &userdata;
913
914 eval = cvMat( 1, m, CV_32FC1, cvAlloc( sizeof( float ) * m ) );
915
916 storage = cvCreateMemStorage();
917 seq = cvCreateSeq( 0, sizeof( *seq ), sizeof( classifier ), storage );
918
919 weakTrainVals = cvCreateMat( 1, m, CV_32FC1 );
920 trainer = cvBoostStartTraining( &data->cls, weakTrainVals, &data->weights,
921 sampleIdx, boosttype );
922 num_splits = 0;
923 sumalpha = 0.0F;
924 do
925 {
926
927#ifdef CV_VERBOSE
928 int v_wt = 0;
929 int v_flipped = 0;
930#endif /* CV_VERBOSE */
931
932 trimmedIdx = cvTrimWeights( &data->weights, sampleIdx, weightfraction );
933 numtrimmed = (trimmedIdx) ? MAX( trimmedIdx->rows, trimmedIdx->cols ) : m;
934
935#ifdef CV_VERBOSE
936 v_wt = 100 * numtrimmed / numsamples;
937 v_flipped = 0;
938
939#endif /* CV_VERBOSE */
940
941 cart = (CvCARTClassifier*) cvCreateCARTClassifier( data->valcache,
942 flags,
943 weakTrainVals, 0, 0, 0, trimmedIdx,
944 &(data->weights),
945 (CvClassifierTrainParams*) &trainParams );
946
947 classifier = (CvCARTHaarClassifier*) icvCreateCARTHaarClassifier( numsplits );
948 icvInitCARTHaarClassifier( classifier, cart, haarFeatures );
949
950 num_splits += classifier->count;
951
952 cart->release( (CvClassifier**) &cart );
953
954 if( symmetric && (seq->total % 2) )
955 {
956 float normfactor = 0.0F;
957 CvStumpClassifier* stump;
958
959 /* flip haar features */
960 for( i = 0; i < classifier->count; i++ )
961 {
962 if( classifier->feature[i].desc[0] == 'h' )
963 {
964 for( j = 0; j < CV_HAAR_FEATURE_MAX &&
965 classifier->feature[i].rect[j].weight != 0.0F; j++ )
966 {
967 classifier->feature[i].rect[j].r.x = data->winsize.width -
968 classifier->feature[i].rect[j].r.x -
969 classifier->feature[i].rect[j].r.width;
970 }
971 }
972 else
973 {
974 int tmp = 0;
975
976 /* (x,y) -> (24-x,y) */
977 /* w -> h; h -> w */
978 for( j = 0; j < CV_HAAR_FEATURE_MAX &&
979 classifier->feature[i].rect[j].weight != 0.0F; j++ )
980 {
981 classifier->feature[i].rect[j].r.x = data->winsize.width -
982 classifier->feature[i].rect[j].r.x;
983 CV_SWAP( classifier->feature[i].rect[j].r.width,
984 classifier->feature[i].rect[j].r.height, tmp );
985 }
986 }
987 }
988 icvConvertToFastHaarFeature( classifier->feature,
989 classifier->fastfeature,
990 classifier->count, data->winsize.width + 1 );
991
992 stumpTrainParams.getTrainData = NULL;
993 stumpTrainParams.numcomp = 1;
994 stumpTrainParams.userdata = NULL;
995 stumpTrainParams.sortedIdx = NULL;
996
997 for( i = 0; i < classifier->count; i++ )
998 {
999 for( j = 0; j < numtrimmed; j++ )
1000 {
1001 idx = icvGetIdxAt( trimmedIdx, j );
1002
1003 eval.data.fl[idx] = cvEvalFastHaarFeature( &classifier->fastfeature[i],
1004 (sum_type*) (data->sum.data.ptr + idx * data->sum.step),
1005 (sum_type*) (data->tilted.data.ptr + idx * data->tilted.step) );
1006 normfactor = data->normfactor.data.fl[idx];
1007 eval.data.fl[idx] = ( normfactor == 0.0F )
1008 ? 0.0F : (eval.data.fl[idx] / normfactor);
1009 }
1010
1011 stump = (CvStumpClassifier*) trainParams.stumpConstructor( &eval,
1012 CV_COL_SAMPLE,
1013 weakTrainVals, 0, 0, 0, trimmedIdx,
1014 &(data->weights),
1015 trainParams.stumpTrainParams );
1016
1017 classifier->threshold[i] = stump->threshold;
1018 if( classifier->left[i] <= 0 )
1019 {
1020 classifier->val[-classifier->left[i]] = stump->left;
1021 }
1022 if( classifier->right[i] <= 0 )
1023 {
1024 classifier->val[-classifier->right[i]] = stump->right;
1025 }
1026
1027 stump->release( (CvClassifier**) &stump );
1028
1029 }
1030
1031 stumpTrainParams.getTrainData = icvGetTrainingDataCallback;
1032 stumpTrainParams.numcomp = n;
1033 stumpTrainParams.userdata = &userdata;
1034 stumpTrainParams.sortedIdx = data->idxcache;
1035
1036#ifdef CV_VERBOSE
1037 v_flipped = 1;
1038#endif /* CV_VERBOSE */
1039
1040 } /* if symmetric */
1041 if( trimmedIdx != sampleIdx )
1042 {
1043 cvReleaseMat( &trimmedIdx );
1044 trimmedIdx = NULL;
1045 }
1046
1047 for( i = 0; i < numsamples; i++ )
1048 {
1049 idx = icvGetIdxAt( sampleIdx, i );
1050
1051 eval.data.fl[idx] = classifier->eval( (CvIntHaarClassifier*) classifier,
1052 (sum_type*) (data->sum.data.ptr + idx * data->sum.step),
1053 (sum_type*) (data->tilted.data.ptr + idx * data->tilted.step),
1054 data->normfactor.data.fl[idx] );
1055 }
1056
1057 alpha = cvBoostNextWeakClassifier( &eval, &data->cls, weakTrainVals,
1058 &data->weights, trainer );
1059 sumalpha += alpha;
1060
1061 for( i = 0; i <= classifier->count; i++ )
1062 {
1063 if( boosttype == CV_RABCLASS )
1064 {
1065 classifier->val[i] = cvLogRatio( classifier->val[i] );
1066 }
1067 classifier->val[i] *= alpha;
1068 }
1069
1070 cvSeqPush( seq, (void*) &classifier );
1071
1072 numpos = 0;
1073 for( i = 0; i < numsamples; i++ )
1074 {
1075 idx = icvGetIdxAt( sampleIdx, i );
1076
1077 if( data->cls.data.fl[idx] == 1.0F )
1078 {
1079 eval.data.fl[numpos] = 0.0F;
1080 for( j = 0; j < seq->total; j++ )
1081 {
1082 classifier = *((CvCARTHaarClassifier**) cvGetSeqElem( seq, j ));
1083 eval.data.fl[numpos] += classifier->eval(
1084 (CvIntHaarClassifier*) classifier,
1085 (sum_type*) (data->sum.data.ptr + idx * data->sum.step),
1086 (sum_type*) (data->tilted.data.ptr + idx * data->tilted.step),
1087 data->normfactor.data.fl[idx] );
1088 }
1089 /* eval.data.fl[numpos] = 2.0F * eval.data.fl[numpos] - seq->total; */
1090 numpos++;
1091 }
1092 }
1093 icvSort_32f( eval.data.fl, numpos, 0 );
1094 threshold = eval.data.fl[(int) ((1.0F - minhitrate) * numpos)];
1095
1096 numneg = 0;
1097 numfalse = 0;
1098 for( i = 0; i < numsamples; i++ )
1099 {
1100 idx = icvGetIdxAt( sampleIdx, i );
1101
1102 if( data->cls.data.fl[idx] == 0.0F )
1103 {
1104 numneg++;
1105 sum_stage = 0.0F;
1106 for( j = 0; j < seq->total; j++ )
1107 {
1108 classifier = *((CvCARTHaarClassifier**) cvGetSeqElem( seq, j ));
1109 sum_stage += classifier->eval( (CvIntHaarClassifier*) classifier,
1110 (sum_type*) (data->sum.data.ptr + idx * data->sum.step),
1111 (sum_type*) (data->tilted.data.ptr + idx * data->tilted.step),
1112 data->normfactor.data.fl[idx] );
1113 }
1114 /* sum_stage = 2.0F * sum_stage - seq->total; */
1115 if( sum_stage >= (threshold - CV_THRESHOLD_EPS) )
1116 {
1117 numfalse++;
1118 }
1119 }
1120 }
1121 falsealarm = ((float) numfalse) / ((float) numneg);
1122
1123#ifdef CV_VERBOSE
1124 {
1125 float v_hitrate = 0.0F;
1126 float v_falsealarm = 0.0F;
1127 /* expected error of stage classifier regardless threshold */
1128 float v_experr = 0.0F;
1129
1130 for( i = 0; i < numsamples; i++ )
1131 {
1132 idx = icvGetIdxAt( sampleIdx, i );
1133
1134 sum_stage = 0.0F;
1135 for( j = 0; j < seq->total; j++ )
1136 {
1137 classifier = *((CvCARTHaarClassifier**) cvGetSeqElem( seq, j ));
1138 sum_stage += classifier->eval( (CvIntHaarClassifier*) classifier,
1139 (sum_type*) (data->sum.data.ptr + idx * data->sum.step),
1140 (sum_type*) (data->tilted.data.ptr + idx * data->tilted.step),
1141 data->normfactor.data.fl[idx] );
1142 }
1143 /* sum_stage = 2.0F * sum_stage - seq->total; */
1144 if( sum_stage >= (threshold - CV_THRESHOLD_EPS) )
1145 {
1146 if( data->cls.data.fl[idx] == 1.0F )
1147 {
1148 v_hitrate += 1.0F;
1149 }
1150 else
1151 {
1152 v_falsealarm += 1.0F;
1153 }
1154 }
1155 if( ( sum_stage >= 0.0F ) != (data->cls.data.fl[idx] == 1.0F) )
1156 {
1157 v_experr += 1.0F;
1158 }
1159 }
1160 v_experr /= numsamples;
1161 printf( "|%4d|%3d%%|%c|%9f|%9f|%9f|%9f|\n",
1162 seq->total, v_wt, ( (v_flipped) ? '+' : '-' ),
1163 threshold, v_hitrate / numpos, v_falsealarm / numneg,
1164 v_experr );
1165 printf( "+----+----+-+---------+---------+---------+---------+\n" );
1166 fflush( stdout );
1167 }
1168#endif /* CV_VERBOSE */
1169
1170 } while( falsealarm > maxfalsealarm && (!maxsplits || (num_splits < maxsplits) ) );
1171 cvBoostEndTraining( &trainer );
1172
1173 if( falsealarm > maxfalsealarm )
1174 {
1175 stage = NULL;
1176 }
1177 else
1178 {
1179 stage = (CvStageHaarClassifier*) icvCreateStageHaarClassifier( seq->total,
1180 threshold );
1181 cvCvtSeqToArray( seq, (CvArr*) stage->classifier );
1182 }
1183
1184 /* CLEANUP */
1185 cvReleaseMemStorage( &storage );
1186 cvReleaseMat( &weakTrainVals );
1187 cvFree( &(eval.data.ptr) );
1188
1189 return (CvIntHaarClassifier*) stage;
1190}

1408 ((float) data->winsize.height + reader->point.y) / ((float) reader->src.rows) );
1409
1410 reader->img = cvMat( (int) (reader->scale * reader->src.rows + 0.5F),
1411 (int) (reader->scale * reader->src.cols + 0.5F),

1418 * icvGetBackgroundImage
1419 *
1420 * Get an image from background
1421 * <img> must be allocated and have size, previously passed to icvInitBackgroundReaders
1422 *
1423 * Usage example:
1424 * icvInitBackgroundReaders( "bg.txt", cvSize( 24, 24 ) );
1425 * ...
1426 * #pragma omp parallel
1427 * {
1428 * ...
1429 * icvGetBackgourndImage( cvbgdata, cvbgreader, img );
1430 * ...
1431 * }
1432 * ...
1433 * icvDestroyBackgroundReaders();
1434 */
1435static
1436void icvGetBackgroundImage( CvBackgroundData* data,
1437 CvBackgroundReader* reader,
1438 CvMat* img )
1439{
1440 CvMat mat;
1441
1442 assert( data != NULL && reader != NULL && img != NULL );
1443 assert( CV_MAT_TYPE( img->type ) == CV_8UC1 );
1444 assert( img->cols == data->winsize.width );
1445 assert( img->rows == data->winsize.height );
1446
1447 if( reader->img.data.ptr == NULL )
1448 {
1449 icvGetNextFromBackgroundData( data, reader );
1450 }
1451
1452 mat = cvMat( data->winsize.height, data->winsize.width, CV_8UC1 );
1453 cvSetData( &mat, (void*) (reader->img.data.ptr + reader->point.y * reader->img.step
1454 + reader->point.x * sizeof( uchar )), reader->img.step );
1455
1456 cvCopy( &mat, img, 0 );
1457 if( (int) ( reader->point.x + (1.0F + reader->stepfactor ) * data->winsize.width )
1458 < reader->img.cols )
1459 {
1460 reader->point.x += (int) (reader->stepfactor * data->winsize.width);
1461 }
1462 else
1463 {
1464 reader->point.x = reader->offset.x;
1465 if( (int) ( reader->point.y + (1.0F + reader->stepfactor ) * data->winsize.height )
1466 < reader->img.rows )
1467 {
1468 reader->point.y += (int) (reader->stepfactor * data->winsize.height);
1469 }
1470 else
1471 {
1472 reader->point.y = reader->offset.y;
1473 reader->scale *= reader->scalefactor;
1474 if( reader->scale <= 1.0F )
1475 {
1476 reader->img = cvMat( (int) (reader->scale * reader->src.rows),
1477 (int) (reader->scale * reader->src.cols),
1478 CV_8UC1, (void*) (reader->img.data.ptr) );
1479 cvResize( &(reader->src), &(reader->img) );
1480 }
1481 else

1485 }
1486 }

1567/*
1568 * icvGetAuxImages
1569 *
1570 * Get sum, tilted, sqsum images and calculate normalization factor
1571 * All images must be allocated.
1572 */
1573static
1574void icvGetAuxImages( CvMat* img, CvMat* sum, CvMat* tilted,
1575 CvMat* sqsum, float* normfactor )
1576{
1577 CvRect normrect;
1578 int p0, p1, p2, p3;
1579 sum_type valsum = 0;
1580 sqsum_type valsqsum = 0;
1581 double area = 0.0;
1582
1583 cvIntegral( img, sum, sqsum, tilted );
1584 normrect = cvRect( 1, 1, img->cols - 2, img->rows - 2 );
1585 CV_SUM_OFFSETS( p0, p1, p2, p3, normrect, img->cols + 1 )
1586
1587 area = normrect.width * normrect.height;
1588 valsum = ((sum_type*) (sum->data.ptr))[p0] - ((sum_type*) (sum->data.ptr))[p1]
1589 - ((sum_type*) (sum->data.ptr))[p2] + ((sum_type*) (sum->data.ptr))[p3];
1590 valsqsum = ((sqsum_type*) (sqsum->data.ptr))[p0]
1591 - ((sqsum_type*) (sqsum->data.ptr))[p1]
1592 - ((sqsum_type*) (sqsum->data.ptr))[p2]
1593 + ((sqsum_type*) (sqsum->data.ptr))[p3];
1594
1595 /* sqrt( valsqsum / area - ( valsum / are )^2 ) * area */
1596 (*normfactor) = (float) sqrt( (double) (area * valsqsum - (double)valsum * valsum) );
1597}

1687 if( !ok )

1702 }
1703 }

1705#ifdef CV_VERBOSE

1707 {
1708 fprintf( stderr, "%3d%%\r", (int) ( 100.0 * (i - first) / count ) );

1711#endif /* CV_VERBOSE */
1712 }

1715 cvFree( &(sqsum.data.ptr) );
1716
1717 #ifdef CV_OPENMP
1718 #pragma omp critical (c_consumed_count)
1719 #endif /* CV_OPENMP */
1720 {
1721 /* consumed_count += thread_consumed_count; */
1722 CCOUNTER_ADD(getcount, thread_getcount);
1723 CCOUNTER_ADD(consumed_count, thread_consumed_count);
1724 }
1725 } /* omp parallel */
1726
1727 if( consumed != NULL )
1728 {
1729 *consumed = (int)consumed_count;
1730 }
1731
1732 if( acceptance_ratio != NULL )
1733 {
1734 /* *acceptance_ratio = ((double) count) / consumed_count; */
1735 *acceptance_ratio = CCOUNTER_DIV(count, consumed_count);
1736 }
1737
1738 return static_cast<int>(getcount);
1739}
