71/* returns index at specified position from index matrix of any type.
72 if matrix is NULL, then specified position is returned */
73CV_INLINE
74int icvGetIdxAt( CvMat* idx, int pos );
75
76CV_INLINE
77int icvGetIdxAt( CvMat* idx, int pos )
78{
79 if( idx == NULL )
80 {
81 return pos;
82 }
83 else
84 {
85 CvScalar sc;
86 int type;
87
88 type = CV_MAT_TYPE( idx->type );
89 cvRawDataToScalar( idx->data.ptr + pos *
90 ( (idx->rows == 1) ? CV_ELEM_SIZE( type ) : idx->step ), type, &sc );
91
92 return (int) sc.val[0];
93 }
94}
