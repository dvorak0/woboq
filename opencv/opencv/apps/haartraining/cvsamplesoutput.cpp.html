15void IOutput::findFilePathPart(char **partOfPath, char *fullPath)
16{
17 *partOfPath = strrchr( fullPath, '\\' );
18 if( *partOfPath == NULL )
19 {
20 *partOfPath = strrchr( fullPath, '/' );
21 }
22 if( *partOfPath == NULL )
23 {
24 *partOfPath = fullPath;
25 }
26 else
27 {
28 *partOfPath += 1;

32IOutput* IOutput::createOutput(const char *filename,
33 IOutput::OutputType type)
34{
35 IOutput* output = 0;
36 switch (type) {
37 case IOutput::PNG_DATASET:
38 output = new PngDatasetOutput();
39 break;
40 case IOutput::JPG_DATASET:
41 output = new JpgDatasetOutput();
42 break;
43 default:
44#if CV_VERBOSE
45 fprintf( stderr, "Invalid output type, valid types are: PNG_TRAINING_SET, JPG_TEST_SET");

48 }
49
50 if ( output->init( filename ) )
51 return output;
52 else
53 return 0;

56bool PngDatasetOutput::init( const char* annotationsListFileName )
57{
58 IOutput::init( annotationsListFileName );
59
60 if(imgFileName == imgFullPath)
61 {
62 #if CV_VERBOSE
63 fprintf( stderr, "Invalid path to annotations file: %s\n"
64 "It should contain a parent directory name\n", imgFullPath );
65 #endif /* CV_VERBOSE */
66 return false;
67 }
68
69
70 const char* annotationsdirname = "/annotations/";
71 const char* positivesdirname = "/pos/";
72
73 imgFileName[-1] = '\0'; //erase slash at the end of the path
74 imgFileName -= 1;
75
76 //copy path to dataset top-level dir
77 strcpy(annotationFullPath, imgFullPath);
78 //find the name of annotation starting from the top-level dataset dir
79 findFilePathPart(&annotationRelativePath, annotationFullPath);
80 if( !strcmp( annotationRelativePath, ".." ) || !strcmp( annotationRelativePath, "." ) )
81 {
82 #if CV_VERBOSE
83 fprintf( stderr, "Invalid path to annotations file: %s\n"
84 "It should contain a parent directory name\n", annotationsListFileName );
85 #endif /* CV_VERBOSE */
86 return false;
87 }
88 //find the name of output image starting from the top-level dataset dir
89 findFilePathPart(&imgRelativePath, imgFullPath);
90 annotationFileName = annotationFullPath + strlen(annotationFullPath);
91
92 sprintf(annotationFileName, "%s", annotationsdirname);
93 annotationFileName += strlen(annotationFileName);
94 sprintf(imgFileName, "%s", positivesdirname);
95 imgFileName += strlen(imgFileName);
96
97 if( !icvMkDir( annotationFullPath ) )
98 {
99 #if CV_VERBOSE
100 fprintf( stderr, "Unable to create directory hierarchy: %s\n", annotationFullPath );
101 #endif /* CV_VERBOSE */
102 return false;
103 }
104 if( !icvMkDir( imgFullPath ) )
105 {
106 #if CV_VERBOSE
107 fprintf( stderr, "Unable to create directory hierarchy: %s\n", imgFullPath );
108 #endif /* CV_VERBOSE */
109 return false;
110 }
111
112 return true;
113}

115bool PngDatasetOutput::write( const CvMat& img,

136
137 sprintf( imgFileName + strlen(imgFileName), ".%s", extension );
138
139

156CvRect PngDatasetOutput::addBoundingboxBorder(const CvRect& bbox) const
157{
158 CvRect boundingBox = bbox;
159 int border = 5;
160
161 boundingBox.x -= border;
162 boundingBox.y -= border;
163 boundingBox.width += 2*border;
164 boundingBox.height += 2*border;
165
166 return boundingBox;
167}

177bool IOutput::init(const char *filename)
178{
179 assert( filename != NULL );
180
181 if( !icvMkDir( filename ) )
182 {
183
184#if CV_VERBOSE
185 fprintf( stderr, "Unable to create directory hierarchy: %s\n", filename );
186#endif /* CV_VERBOSE */
187
188 return false;
189 }
190
191 annotationsList = fopen( filename, "w" );
192 if( annotationsList == NULL )
193 {
194#if CV_VERBOSE
195 fprintf( stderr, "Unable to create info file: %s\n", filename );
196#endif /* CV_VERBOSE */
197 return false;
198 }
199 strcpy( imgFullPath, filename );
200
201 findFilePathPart( &imgFileName, imgFullPath );
202
203 return true;
204}
