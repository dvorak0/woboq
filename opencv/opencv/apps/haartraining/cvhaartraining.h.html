56/*
57 * cvCreateTrainingSamples
58 *
59 * Create training samples applying random distortions to sample image and
60 * store them in .vec file
61 *
62 * filename - .vec file name
63 * imgfilename - sample image file name
64 * bgcolor - background color for sample image
65 * bgthreshold - background color threshold. Pixels those colors are in range
66 * [bgcolor-bgthreshold, bgcolor+bgthreshold] are considered as transparent
67 * bgfilename - background description file name. If not NULL samples
68 * will be put on arbitrary background
69 * count - desired number of samples
70 * invert - if not 0 sample foreground pixels will be inverted
71 * if invert == CV_RANDOM_INVERT then samples will be inverted randomly
72 * maxintensitydev - desired max intensity deviation of foreground samples pixels
73 * maxxangle - max rotation angles
74 * maxyangle
75 * maxzangle
76 * showsamples - if not 0 samples will be shown
77 * winwidth - desired samples width
78 * winheight - desired samples height
79 */
80#define CV_RANDOM_INVERT 0x7FFFFFFF
81

141/*
142 * cvCreateCascadeClassifier
143 *
144 * Create cascade classifier
145 * dirname - directory name in which cascade classifier will be created.
146 * It must exist and contain subdirectories 0, 1, 2, ... (nstages-1).
147 * vecfilename - name of .vec file with object's images
148 * bgfilename - name of background description file
149 * bg_vecfile - true if bgfilename represents a vec file with discrete negatives
150 * npos - number of positive samples used in training of each stage
151 * nneg - number of negative samples used in training of each stage
152 * nstages - number of stages
153 * numprecalculated - number of features being precalculated. Each precalculated feature
154 * requires (number_of_samples*(sizeof( float ) + sizeof( short ))) bytes of memory
155 * numsplits - number of binary splits in each weak classifier
156 * 1 - stumps, 2 and more - trees.
157 * minhitrate - desired min hit rate of each stage
158 * maxfalsealarm - desired max false alarm of each stage
159 * weightfraction - weight trimming parameter
160 * mode - 0 - BASIC = Viola
161 * 1 - CORE = All upright
162 * 2 - ALL = All features
163 * symmetric - if not 0 vertical symmetry is assumed
164 * equalweights - if not 0 initial weights of all samples will be equal
165 * winwidth - sample width
166 * winheight - sample height
167 * boosttype - type of applied boosting algorithm
168 * 0 - Discrete AdaBoost
169 * 1 - Real AdaBoost
170 * 2 - LogitBoost
171 * 3 - Gentle AdaBoost
172 * stumperror - type of used error if Discrete AdaBoost algorithm is applied
173 * 0 - misclassification error
174 * 1 - gini error
175 * 2 - entropy error
176 */
177void cvCreateCascadeClassifier( const char* dirname,
178 const char* vecfilename,
179 const char* bgfilename,
180 int npos, int nneg, int nstages,
181 int numprecalculated,
182 int numsplits,
183 float minhitrate = 0.995F, float maxfalsealarm = 0.5F,
184 float weightfraction = 0.95F,
185 int mode = 0, int symmetric = 1,
186 int equalweights = 1,
187 int winwidth = 24, int winheight = 24,
188 int boosttype = 3, int stumperror = 0 );
