45 * Measure performance of classifier

47#include "opencv2/core/core.hpp"
48#include "opencv2/core/internal.hpp"
49
50#include "cv.h"
51#include "highgui.h"
52
53#include <cstdio>
54#include <cmath>
55#include <ctime>
56
57#ifdef _WIN32
58/* use clock() function insted of time() */
59#define time( arg ) (((double) clock()) / CLOCKS_PER_SEC)
60#endif /* _WIN32 */
61
62#ifndef PATH_MAX
63#define PATH_MAX 512
64#endif /* PATH_MAX */
65
66typedef struct HidCascade
67{
68 int size;
69 int count;
70} HidCascade;
71
72typedef struct ObjectPos
73{
74 float x;
75 float y;
76 float width;
77 int found; /* for reference */
78 int neghbors;
79} ObjectPos;
80
81int main( int argc, char* argv[] )
82{
83 int i, j;
84 char* classifierdir = NULL;
85 //char* samplesdir = NULL;
86
87 int saveDetected = 1;
88 double scale_factor = 1.2;
89 float maxSizeDiff = 1.5F;
90 float maxPosDiff = 0.3F;
91
92 /* number of stages. if <=0 all stages are used */
93 int nos = -1, nos0;
94
95 int width = 24;
96 int height = 24;
97
98 int rocsize;
99
100 FILE* info;
101 char* infoname;
102 char fullname[PATH_MAX];
103 char detfilename[PATH_MAX];
104 char* filename;
105 char detname[] = "det-";
106
107 CvHaarClassifierCascade* cascade;
108 CvMemStorage* storage;
109 CvSeq* objects;
110
111 double totaltime;
112
113 infoname = (char*)"";
114 rocsize = 40;
115 if( argc == 1 )
116 {
117 printf( "Usage: %s\n -data <classifier_directory_name>\n"
118 " -info <collection_file_name>\n"
119 " [-maxSizeDiff <max_size_difference = %f>]\n"
120 " [-maxPosDiff <max_position_difference = %f>]\n"
121 " [-sf <scale_factor = %f>]\n"
122 " [-ni]\n"
123 " [-nos <number_of_stages = %d>]\n"
124 " [-rs <roc_size = %d>]\n"
125 " [-w <sample_width = %d>]\n"
126 " [-h <sample_height = %d>]\n",
127 argv[0], maxSizeDiff, maxPosDiff, scale_factor, nos, rocsize,
128 width, height );
129
130 return 0;
131 }
132
133 for( i = 1; i < argc; i++ )
134 {
135 if( !strcmp( argv[i], "-data" ) )
136 {
137 classifierdir = argv[++i];
138 }
139 else if( !strcmp( argv[i], "-info" ) )
140 {
141 infoname = argv[++i];
142 }
143 else if( !strcmp( argv[i], "-maxSizeDiff" ) )
144 {
145 maxSizeDiff = (float) atof( argv[++i] );
146 }
147 else if( !strcmp( argv[i], "-maxPosDiff" ) )
148 {
149 maxPosDiff = (float) atof( argv[++i] );
150 }
151 else if( !strcmp( argv[i], "-sf" ) )
152 {
153 scale_factor = atof( argv[++i] );
154 }
155 else if( !strcmp( argv[i], "-ni" ) )
156 {
157 saveDetected = 0;
158 }
159 else if( !strcmp( argv[i], "-nos" ) )
160 {
161 nos = atoi( argv[++i] );
162 }
163 else if( !strcmp( argv[i], "-rs" ) )
164 {
165 rocsize = atoi( argv[++i] );
166 }
167 else if( !strcmp( argv[i], "-w" ) )
168 {
169 width = atoi( argv[++i] );
170 }
171 else if( !strcmp( argv[i], "-h" ) )
172 {
173 height = atoi( argv[++i] );
174 }
175 }
176
177 cascade = cvLoadHaarClassifierCascade( classifierdir, cvSize( width, height ) );
178 if( cascade == NULL )
179 {
180 printf( "Unable to load classifier from %s\n", classifierdir );
181
182 return 1;
183 }
184
185 int* numclassifiers = new int[cascade->count];
186 numclassifiers[0] = cascade->stage_classifier[0].count;
187 for( i = 1; i < cascade->count; i++ )
188 {
189 numclassifiers[i] = numclassifiers[i-1] + cascade->stage_classifier[i].count;
190 }
191
192 storage = cvCreateMemStorage();
193
194 nos0 = cascade->count;
195 if( nos <= 0 )
196 nos = nos0;
197
198 strcpy( fullname, infoname );
199 filename = strrchr( fullname, '\\' );
200 if( filename == NULL )
201 {
202 filename = strrchr( fullname, '/' );
203 }
204 if( filename == NULL )
205 {
206 filename = fullname;
207 }
208 else
209 {
210 filename++;
211 }
212
213 info = fopen( infoname, "r" );
214 totaltime = 0.0;
215 if( info != NULL )
216 {
217 int x, y;
218 IplImage* img;
219 int hits, missed, falseAlarms;
220 int totalHits, totalMissed, totalFalseAlarms;
221 int found;
222 float distance;
223
224 int refcount;
225 ObjectPos* ref;
226 int detcount;
227 ObjectPos* det;
228 int error=0;
229
230 int* pos;
231 int* neg;
232
233 pos = (int*) cvAlloc( rocsize * sizeof( *pos ) );
234 neg = (int*) cvAlloc( rocsize * sizeof( *neg ) );
235 for( i = 0; i < rocsize; i++ ) { pos[i] = neg[i] = 0; }
236
237 printf( "+================================+======+======+======+\n" );
238 printf( "| File Name | Hits |Missed| False|\n" );
239 printf( "+================================+======+======+======+\n" );
240
241 totalHits = totalMissed = totalFalseAlarms = 0;
242 while( !feof( info ) )
243 {
244 if( fscanf( info, "%s %d", filename, &refcount ) != 2 || refcount <= 0 ) break;
245
246 img = cvLoadImage( fullname );
247 if( !img ) continue;
248
249 ref = (ObjectPos*) cvAlloc( refcount * sizeof( *ref ) );
250 for( i = 0; i < refcount; i++ )
251 {
252 int w, h;
253 error = (fscanf( info, "%d %d %d %d", &x, &y, &w, &h ) != 4);
254 if( error ) break;
255 ref[i].x = 0.5F * w + x;
256 ref[i].y = 0.5F * h + y;
257 ref[i].width = sqrtf( 0.5F * (w * w + h * h) );
258 ref[i].found = 0;
259 ref[i].neghbors = 0;
260 }
261 if( !error )
262 {
263 cvClearMemStorage( storage );
264
265 cascade->count = nos;
266 totaltime -= time( 0 );
267 objects = cvHaarDetectObjects( img, cascade, storage, scale_factor, 1 );
268 totaltime += time( 0 );
269 cascade->count = nos0;
270
271 detcount = ( objects ? objects->total : 0);
272 det = (detcount > 0) ?
273 ( (ObjectPos*)cvAlloc( detcount * sizeof( *det )) ) : NULL;
274 hits = missed = falseAlarms = 0;
275 for( i = 0; i < detcount; i++ )
276 {
277 CvAvgComp r = *((CvAvgComp*) cvGetSeqElem( objects, i ));
278 det[i].x = 0.5F * r.rect.width + r.rect.x;
279 det[i].y = 0.5F * r.rect.height + r.rect.y;
280 det[i].width = sqrtf( 0.5F * (r.rect.width * r.rect.width +
281 r.rect.height * r.rect.height) );
282 det[i].neghbors = r.neighbors;
283
284 if( saveDetected )
285 {
286 cvRectangle( img, cvPoint( r.rect.x, r.rect.y ),
287 cvPoint( r.rect.x + r.rect.width, r.rect.y + r.rect.height ),
288 CV_RGB( 255, 0, 0 ), 3 );
289 }
290
291 found = 0;
292 for( j = 0; j < refcount; j++ )
293 {
294 distance = sqrtf( (det[i].x - ref[j].x) * (det[i].x - ref[j].x) +
295 (det[i].y - ref[j].y) * (det[i].y - ref[j].y) );
296 if( (distance < ref[j].width * maxPosDiff) &&
297 (det[i].width > ref[j].width / maxSizeDiff) &&
298 (det[i].width < ref[j].width * maxSizeDiff) )
299 {
300 ref[j].found = 1;
301 ref[j].neghbors = MAX( ref[j].neghbors, det[i].neghbors );
302 found = 1;
303 }
304 }
305 if( !found )
306 {
307 falseAlarms++;
308 neg[MIN(det[i].neghbors, rocsize - 1)]++;
309 }
310 }
311 for( j = 0; j < refcount; j++ )
312 {
313 if( ref[j].found )
314 {
315 hits++;
316 pos[MIN(ref[j].neghbors, rocsize - 1)]++;
317 }
318 else
319 {
320 missed++;
321 }
322 }
323
324 totalHits += hits;
325 totalMissed += missed;
326 totalFalseAlarms += falseAlarms;
327 printf( "|%32.32s|%6d|%6d|%6d|\n", filename, hits, missed, falseAlarms );
328 printf( "+--------------------------------+------+------+------+\n" );
329 fflush( stdout );
330
331 if( saveDetected )
332 {
333 strcpy( detfilename, detname );
334 strcat( detfilename, filename );
335 strcpy( filename, detfilename );
336 cvvSaveImage( fullname, img );
337 }
338
339 if( det ) { cvFree( &det ); det = NULL; }
340 } /* if( !error ) */
341
342 cvReleaseImage( &img );
343 cvFree( &ref );
344 }
345 fclose( info );
346
347 printf( "|%32.32s|%6d|%6d|%6d|\n", "Total",
348 totalHits, totalMissed, totalFalseAlarms );
349 printf( "+================================+======+======+======+\n" );
350 printf( "Number of stages: %d\n", nos );
351 printf( "Number of weak classifiers: %d\n", numclassifiers[nos - 1] );
352 printf( "Total time: %f\n", totaltime );
353
354 /* print ROC to stdout */
355 for( i = rocsize - 1; i > 0; i-- )
356 {
357 pos[i-1] += pos[i];
358 neg[i-1] += neg[i];
359 }
360 fprintf( stderr, "%d\n", nos );
361 for( i = 0; i < rocsize; i++ )
362 {
363 fprintf( stderr, "\t%d\t%d\t%f\t%f\n", pos[i], neg[i],
364 ((float)pos[i]) / (totalHits + totalMissed),
365 ((float)neg[i]) / (totalHits + totalMissed) );
366 }
367
368 cvFree( &pos );
369 cvFree( &neg );
370 }
371
372 delete[] numclassifiers;
373
374 cvReleaseHaarClassifierCascade( &cascade );
375 cvReleaseMemStorage( &storage );
376
377 return 0;
378}
