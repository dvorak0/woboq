28bool CvCascadeImageReader::NegReader::create( const string _filename, Size _winSize )
29{
30 string dirname, str;
31 std::ifstream file(_filename.c_str());
32 if ( !file.is_open() )
33 return false;
34
35 while( !file.eof() )
36 {
37 std::getline(file, str);
38 if (str.empty()) break;
39 if (str.at(0) == '#' ) continue; /* comment */
40 imgFilenames.push_back(str);
41 }
42 file.close();
43
44 winSize = _winSize;
45 last = round = 0;
46 return true;
47}

49bool CvCascadeImageReader::NegReader::nextImg()
50{
51 Point _offset = Point(0,0);
52 size_t count = imgFilenames.size();
53 for( size_t i = 0; i < count; i++ )
54 {
55 src = imread( imgFilenames[last++], 0 );
56 if( src.empty() )
57 continue;
58 round += last / count;
59 round = round % (winSize.width * winSize.height);
60 last %= count;
61
62 _offset.x = std::min( (int)round % winSize.width, src.cols - winSize.width );
63 _offset.y = std::min( (int)round / winSize.width, src.rows - winSize.height );
64 if( !src.empty() && src.type() == CV_8UC1
65 && _offset.x >= 0 && _offset.y >= 0 )
66 break;
67 }
68
69 if( src.empty() )
70 return false; // no appropriate image
71 point = offset = _offset;
72 scale = max( ((float)winSize.width + point.x) / ((float)src.cols),
73 ((float)winSize.height + point.y) / ((float)src.rows) );
74
75 Size sz( (int)(scale*src.cols + 0.5F), (int)(scale*src.rows + 0.5F) );
76 resize( src, img, sz );
77 return true;
78}

80bool CvCascadeImageReader::NegReader::get( Mat& _img )
81{
82 CV_Assert( !_img.empty() );
83 CV_Assert( _img.type() == CV_8UC1 );
84 CV_Assert( _img.cols == winSize.width );
85 CV_Assert( _img.rows == winSize.height );
86
87 if( img.empty() )
88 if ( !nextImg() )
89 return false;
90
91 Mat mat( winSize.height, winSize.width, CV_8UC1,
92 (void*)(img.data + point.y * img.step + point.x * img.elemSize()), img.step );
93 mat.copyTo(_img);
94
95 if( (int)( point.x + (1.0F + stepFactor ) * winSize.width ) < img.cols )
96 point.x += (int)(stepFactor * winSize.width);
97 else
98 {
99 point.x = offset.x;
100 if( (int)( point.y + (1.0F + stepFactor ) * winSize.height ) < img.rows )
101 point.y += (int)(stepFactor * winSize.height);
102 else
103 {
104 point.y = offset.y;
105 scale *= scaleFactor;
106 if( scale <= 1.0F )
107 resize( src, img, Size( (int)(scale*src.cols), (int)(scale*src.rows) ) );
108 else
109 {
110 if ( !nextImg() )
111 return false;
112 }
113 }
114 }
115 return true;
116}

124bool CvCascadeImageReader::PosReader::create( const string _filename )
125{
126 if ( file )
127 fclose( file );
128 file = fopen( _filename.c_str(), "rb" );
129
130 if( !file )
131 return false;
132 short tmp = 0;
133 if( fread( &count, sizeof( count ), 1, file ) != 1 ||
134 fread( &vecSize, sizeof( vecSize ), 1, file ) != 1 ||
135 fread( &tmp, sizeof( tmp ), 1, file ) != 1 ||
136 fread( &tmp, sizeof( tmp ), 1, file ) != 1 )
137 CV_Error_( CV_StsParseError, ("wrong file format for %s\n", _filename.c_str()) );
138 base = sizeof( count ) + sizeof( vecSize ) + 2*sizeof( tmp );
139 if( feof( file ) )
140 return false;
141 last = 0;
142 vec = (short*) cvAlloc( sizeof( *vec ) * vecSize );
143 CV_Assert( vec );
144 return true;
145}

147bool CvCascadeImageReader::PosReader::get( Mat &_img )
148{
149 CV_Assert( _img.rows * _img.cols == vecSize );
150 uchar tmp = 0;
151 size_t elements_read = fread( &tmp, sizeof( tmp ), 1, file );
152 if( elements_read != 1 )
153 CV_Error( CV_StsBadArg, "Can not get new positive sample. The most possible reason is "
154 "insufficient count of samples in given vec-file.\n");
155 elements_read = fread( vec, sizeof( vec[0] ), vecSize, file );
156 if( elements_read != (size_t)(vecSize) )
157 CV_Error( CV_StsBadArg, "Can not get new positive sample. Seems that vec-file has incorrect structure.\n");
158
159 if( feof( file ) || last++ >= count )
160 CV_Error( CV_StsBadArg, "Can not get new positive sample. vec-file is over.\n");
161
162 for( int r = 0; r < _img.rows; r++ )
163 {
164 for( int c = 0; c < _img.cols; c++ )
165 _img.ptr(r)[c] = (uchar)vec[r * _img.cols + c];
166 }
167 return true;
168}
