41bool CvCascadeParams::read( const FileNode &node )
42{
43 if ( node.empty() )
44 return false;
45 string stageTypeStr, featureTypeStr;
46 FileNode rnode = node[CC_STAGE_TYPE];
47 if ( !rnode.isString() )
48 return false;
49 rnode >> stageTypeStr;
50 stageType = !stageTypeStr.compare( CC_BOOST ) ? BOOST : -1;
51 if (stageType == -1)
52 return false;
53 rnode = node[CC_FEATURE_TYPE];
54 if ( !rnode.isString() )
55 return false;
56 rnode >> featureTypeStr;
57 featureType = !featureTypeStr.compare( CC_HAAR ) ? CvFeatureParams::HAAR :
58 !featureTypeStr.compare( CC_LBP ) ? CvFeatureParams::LBP :
59 !featureTypeStr.compare( CC_HOG ) ? CvFeatureParams::HOG :
60 -1;
61 if (featureType == -1)
62 return false;
63 node[CC_HEIGHT] >> winSize.height;
64 node[CC_WIDTH] >> winSize.width;
65 return winSize.height > 0 && winSize.width > 0;
66}

100bool CvCascadeParams::scanAttr( const string prmName, const string val )
101{
102 bool res = true;
103 if( !prmName.compare( "-stageType" ) )
104 {
105 for( int i = 0; i < (int)(sizeof(stageTypes)/sizeof(stageTypes[0])); i++ )
106 if( !val.compare( stageTypes[i] ) )
107 stageType = i;
108 }
109 else if( !prmName.compare( "-featureType" ) )
110 {
111 for( int i = 0; i < (int)(sizeof(featureTypes)/sizeof(featureTypes[0])); i++ )
112 if( !val.compare( featureTypes[i] ) )
113 featureType = i;
114 }
115 else if( !prmName.compare( "-w" ) )
116 {
117 winSize.width = atoi( val.c_str() );
118 }
119 else if( !prmName.compare( "-h" ) )
120 {
121 winSize.height = atoi( val.c_str() );
122 }
123 else
124 res = false;
125 return res;

130bool CvCascadeClassifier::train( const string _cascadeDirName,
131 const string _posFilename,
132 const string _negFilename,
133 int _numPos, int _numNeg,
134 int _precalcValBufSize, int _precalcIdxBufSize,
135 int _numStages,
136 const CvCascadeParams& _cascadeParams,
137 const CvFeatureParams& _featureParams,
138 const CvCascadeBoostParams& _stageParams,
139 bool baseFormatSave,
140 double acceptanceRatioBreakValue)
141{
142 // Start recording clock ticks for training time output
143 double time = (double)getTickCount();
144
145 if( _cascadeDirName.empty() || _posFilename.empty() || _negFilename.empty() )
146 CV_Error( CV_StsBadArg, "_cascadeDirName or _bgfileName or _vecFileName is NULL" );
147
148 string dirName;
149 if (_cascadeDirName.find_last_of("/\\") == (_cascadeDirName.length() - 1) )
150 dirName = _cascadeDirName;
151 else
152 dirName = _cascadeDirName + '/';
153
154 numPos = _numPos;
155 numNeg = _numNeg;
156 numStages = _numStages;
157 if ( !imgReader.create( _posFilename, _negFilename, _cascadeParams.winSize ) )
158 {
159 cout << "Image reader can not be created from -vec " << _posFilename
160 << " and -bg " << _negFilename << "." << endl;
161 return false;
162 }
163 if ( !load( dirName ) )
164 {
165 cascadeParams = _cascadeParams;
166 featureParams = CvFeatureParams::create(cascadeParams.featureType);
167 featureParams->init(_featureParams);
168 stageParams = new CvCascadeBoostParams;
169 *stageParams = _stageParams;
170 featureEvaluator = CvFeatureEvaluator::create(cascadeParams.featureType);
171 featureEvaluator->init( (CvFeatureParams*)featureParams, numPos + numNeg, cascadeParams.winSize );
172 stageClassifiers.reserve( numStages );
173 }else{
174 // Make sure that if model parameters are preloaded, that people are aware of this,
175 // even when passing other parameters to the training command
176 cout << "---------------------------------------------------------------------------------" << endl;
177 cout << "Training parameters are pre-loaded from the parameter file in data folder!" << endl;
178 cout << "Please empty this folder if you want to use a NEW set of training parameters." << endl;
179 cout << "---------------------------------------------------------------------------------" << endl;
180 }
181 cout << "PARAMETERS:" << endl;
182 cout << "cascadeDirName: " << _cascadeDirName << endl;
183 cout << "vecFileName: " << _posFilename << endl;
184 cout << "bgFileName: " << _negFilename << endl;
185 cout << "numPos: " << _numPos << endl;
186 cout << "numNeg: " << _numNeg << endl;
187 cout << "numStages: " << numStages << endl;
188 cout << "precalcValBufSize[Mb] : " << _precalcValBufSize << endl;
189 cout << "precalcIdxBufSize[Mb] : " << _precalcIdxBufSize << endl;
190 cout << "acceptanceRatioBreakValue : " << acceptanceRatioBreakValue << endl;
191 cascadeParams.printAttrs();
192 stageParams->printAttrs();
193 featureParams->printAttrs();
194 cout << "Number of unique features given windowSize [" << _cascadeParams.winSize.width << "," << _cascadeParams.winSize.height << "] : " << featureEvaluator->getNumFeatures() << "" << endl;
195
196 int startNumStages = (int)stageClassifiers.size();
197 if ( startNumStages > 1 )
198 cout << endl << "Stages 0-" << startNumStages-1 << " are loaded" << endl;
199 else if ( startNumStages == 1)
200 cout << endl << "Stage 0 is loaded" << endl;
201
202 double requiredLeafFARate = pow( (double) stageParams->maxFalseAlarm, (double) numStages ) /
203 (double)stageParams->max_depth;
204 double tempLeafFARate;
205
206 for( int i = startNumStages; i < numStages; i++ )
207 {
208 cout << endl << "===== TRAINING " << i << "-stage =====" << endl;
209 cout << "<BEGIN" << endl;
210
211 if ( !updateTrainingSet( requiredLeafFARate, tempLeafFARate ) )
212 {
213 cout << "Train dataset for temp stage can not be filled. "
214 "Branch training terminated." << endl;
215 break;
216 }
217 if( tempLeafFARate <= requiredLeafFARate )
218 {
219 cout << "Required leaf false alarm rate achieved. "
220 "Branch training terminated." << endl;
221 break;
222 }
223 if( (tempLeafFARate <= acceptanceRatioBreakValue) && (acceptanceRatioBreakValue >= 0) ){
224 cout << "The required acceptanceRatio for the model has been reached to avoid overfitting of trainingdata. "
225 "Branch training terminated." << endl;
226 break;
227}
228
229 CvCascadeBoost* tempStage = new CvCascadeBoost;
230 bool isStageTrained = tempStage->train( (CvFeatureEvaluator*)featureEvaluator,
231 curNumSamples, _precalcValBufSize, _precalcIdxBufSize,
232 *((CvCascadeBoostParams*)stageParams) );
233 cout << "END>" << endl;
234
235 if(!isStageTrained)
236 break;
237
238 stageClassifiers.push_back( tempStage );
239
240 // save params
241 if( i == 0)
242 {
243 std::string paramsFilename = dirName + CC_PARAMS_FILENAME;
244 FileStorage fs( paramsFilename, FileStorage::WRITE);
245 if ( !fs.isOpened() )
246 {
247 cout << "Parameters can not be written, because file " << paramsFilename
248 << " can not be opened." << endl;
249 return false;
250 }
251 fs << FileStorage::getDefaultObjectName(paramsFilename) << "{";
252 writeParams( fs );
253 fs << "}";
254 }
255 // save current stage
256 char buf[10];
257 sprintf(buf, "%s%d", "stage", i );
258 string stageFilename = dirName + buf + ".xml";
259 FileStorage fs( stageFilename, FileStorage::WRITE );
260 if ( !fs.isOpened() )
261 {
262 cout << "Current stage can not be written, because file " << stageFilename
263 << " can not be opened." << endl;
264 return false;
265 }
266 fs << FileStorage::getDefaultObjectName(stageFilename) << "{";
267 tempStage->write( fs, Mat() );
268 fs << "}";
269
270 // Output training time up till now
271 double seconds = ( (double)getTickCount() - time)/ getTickFrequency();
272 int days = int(seconds) / 60 / 60 / 24;
273 int hours = (int(seconds) / 60 / 60) % 24;
274 int minutes = (int(seconds) / 60) % 60;
275 int seconds_left = int(seconds) % 60;
276 cout << "Training until now has taken " << days << " days " << hours << " hours " << minutes << " minutes " << seconds_left <<" seconds." << endl;
277 }
278
279 if(stageClassifiers.size() == 0)
280 {
281 cout << "Cascade classifier can't be trained. Check the used training parameters." << endl;
282 return false;
283 }
284
285 save( dirName + CC_CASCADE_FILENAME, baseFormatSave );
286
287 return true;
288}

302bool CvCascadeClassifier::updateTrainingSet( double minimumAcceptanceRatio, double& acceptanceRatio)
303{
304 int64 posConsumed = 0, negConsumed = 0;
305 imgReader.restart();
306 int posCount = fillPassedSamples( 0, numPos, true, 0, posConsumed );
307 if( !posCount )
308 return false;
309 cout << "POS count : consumed " << posCount << " : " << (int)posConsumed << endl;
310
311 int proNumNeg = cvRound( ( ((double)numNeg) * ((double)posCount) ) / numPos ); // apply only a fraction of negative samples. double is required since overflow is possible
312 int negCount = fillPassedSamples( posCount, proNumNeg, false, minimumAcceptanceRatio, negConsumed );
313 if ( !negCount )
314 return false;
315
316 curNumSamples = posCount + negCount;
317 acceptanceRatio = negConsumed == 0 ? 0 : ( (double)negCount/(double)(int64)negConsumed );
318 cout << "NEG count : acceptanceRatio " << negCount << " : " << acceptanceRatio << endl;
319 return true;
320}

322int CvCascadeClassifier::fillPassedSamples( int first, int count, bool isPositive, double minimumAcceptanceRatio, int64& consumed )
323{
324 int getcount = 0;
325 Mat img(cascadeParams.winSize, CV_8UC1);
326 for( int i = first; i < first + count; i++ )
327 {
328 for( ; ; )
329 {
330 if( consumed != 0 && ((double)getcount+1)/(double)(int64)consumed <= minimumAcceptanceRatio )
331 return getcount;
332
333 bool isGetImg = isPositive ? imgReader.getPos( img ) :
334 imgReader.getNeg( img );
335 if( !isGetImg )
336 return getcount;
337 consumed++;
338
339 featureEvaluator->setImage( img, isPositive ? 1 : 0, i );
340 if( predict( i ) == 1 )
341 {
342 getcount++;
343 printf("%s current samples: %d\r", isPositive ? "POS":"NEG", getcount);
344 break;
345 }
346 }
347 }
348 return getcount;
349}

397bool CvCascadeClassifier::readStages( const FileNode &node)
398{
399 FileNode rnode = node[CC_STAGES];
400 if (!rnode.empty() || !rnode.isSeq())
401 return false;
402 stageClassifiers.reserve(numStages);
403 FileNodeIterator it = rnode.begin();
404 for( int i = 0; i < min( (int)rnode.size(), numStages ); i++, it++ )
405 {
406 CvCascadeBoost* tempStage = new CvCascadeBoost;
407 if ( !tempStage->read( *it, (CvFeatureEvaluator *)featureEvaluator, *((CvCascadeBoostParams*)stageParams) ) )
408 {
409 delete tempStage;
410 return false;
411 }
412 stageClassifiers.push_back(tempStage);
413 }
414 return true;
415}

433void CvCascadeClassifier::save( const string filename, bool baseFormat )
434{
435 FileStorage fs( filename, FileStorage::WRITE );
436
437 if ( !fs.isOpened() )
438 return;
439
440 fs << FileStorage::getDefaultObjectName(filename) << "{";
441 if ( !baseFormat )
442 {
443 Mat featureMap;
444 getUsedFeaturesIdxMap( featureMap );
445 writeParams( fs );
446 fs << CC_STAGE_NUM << (int)stageClassifiers.size();
447 writeStages( fs, featureMap );
448 writeFeatures( fs, featureMap );
449 }
450 else
451 {
452 //char buf[256];
453 CvSeq* weak;
454 if ( cascadeParams.featureType != CvFeatureParams::HAAR )
455 CV_Error( CV_StsBadFunc, "old file format is used for Haar-like features only");
456 fs << ICV_HAAR_SIZE_NAME << "[:" << cascadeParams.winSize.width <<
457 cascadeParams.winSize.height << "]";
458 fs << ICV_HAAR_STAGES_NAME << "[";
459 for( size_t si = 0; si < stageClassifiers.size(); si++ )
460 {
461 fs << "{"; //stage
462 /*sprintf( buf, "stage %d", si );
463 CV_CALL( cvWriteComment( fs, buf, 1 ) );*/
464 weak = stageClassifiers[si]->get_weak_predictors();
465 fs << ICV_HAAR_TREES_NAME << "[";
466 for( int wi = 0; wi < weak->total; wi++ )
467 {
468 int inner_node_idx = -1, total_inner_node_idx = -1;
469 queue<const CvDTreeNode*> inner_nodes_queue;
470 CvCascadeBoostTree* tree = *((CvCascadeBoostTree**) cvGetSeqElem( weak, wi ));
471
472 fs << "[";
473 /*sprintf( buf, "tree %d", wi );
474 CV_CALL( cvWriteComment( fs, buf, 1 ) );*/
475
476 const CvDTreeNode* tempNode;
477
478 inner_nodes_queue.push( tree->get_root() );
479 total_inner_node_idx++;
480
481 while (!inner_nodes_queue.empty())
482 {
483 tempNode = inner_nodes_queue.front();
484 inner_node_idx++;
485
486 fs << "{";
487 fs << ICV_HAAR_FEATURE_NAME << "{";
488 ((CvHaarEvaluator*)((CvFeatureEvaluator*)featureEvaluator))->writeFeature( fs, tempNode->split->var_idx );
489 fs << "}";
490
491 fs << ICV_HAAR_THRESHOLD_NAME << tempNode->split->ord.c;
492
493 if( tempNode->left->left || tempNode->left->right )
494 {
495 inner_nodes_queue.push( tempNode->left );
496 total_inner_node_idx++;
497 fs << ICV_HAAR_LEFT_NODE_NAME << total_inner_node_idx;
498 }
499 else
500 fs << ICV_HAAR_LEFT_VAL_NAME << tempNode->left->value;
501
502 if( tempNode->right->left || tempNode->right->right )
503 {
504 inner_nodes_queue.push( tempNode->right );
505 total_inner_node_idx++;
506 fs << ICV_HAAR_RIGHT_NODE_NAME << total_inner_node_idx;
507 }
508 else
509 fs << ICV_HAAR_RIGHT_VAL_NAME << tempNode->right->value;
510 fs << "}"; // ICV_HAAR_FEATURE_NAME
511 inner_nodes_queue.pop();
512 }
513 fs << "]";
514 }
515 fs << "]"; //ICV_HAAR_TREES_NAME
516 fs << ICV_HAAR_STAGE_THRESHOLD_NAME << stageClassifiers[si]->getThreshold();
517 fs << ICV_HAAR_PARENT_NAME << (int)si-1 << ICV_HAAR_NEXT_NAME << -1;
518 fs << "}"; //stage
519 } /* for each stage */
520 fs << "]"; //ICV_HAAR_STAGES_NAME
521 }
522 fs << "}";
523}

525bool CvCascadeClassifier::load( const string cascadeDirName )
526{
527 FileStorage fs( cascadeDirName + CC_PARAMS_FILENAME, FileStorage::READ );
528 if ( !fs.isOpened() )
529 return false;
530 FileNode node = fs.getFirstTopLevelNode();
531 if ( !readParams( node ) )
532 return false;
533 featureEvaluator = CvFeatureEvaluator::create(cascadeParams.featureType);
534 featureEvaluator->init( ((CvFeatureParams*)featureParams), numPos + numNeg, cascadeParams.winSize );
535 fs.release();
536
537 char buf[10];
538 for ( int si = 0; si < numStages; si++ )
539 {
540 sprintf( buf, "%s%d", "stage", si);
541 fs.open( cascadeDirName + buf + ".xml", FileStorage::READ );
542 node = fs.getFirstTopLevelNode();
543 if ( !fs.isOpened() )
544 break;
545 CvCascadeBoost *tempStage = new CvCascadeBoost;
546
547 if ( !tempStage->read( node, (CvFeatureEvaluator*)featureEvaluator, *((CvCascadeBoostParams*)stageParams )) )
548 {
549 delete tempStage;
550 fs.release();
551 break;
552 }
553 stageClassifiers.push_back(tempStage);
554 }
555 return true;
556}

558void CvCascadeClassifier::getUsedFeaturesIdxMap( Mat& featureMap )
559{
560 int varCount = featureEvaluator->getNumFeatures() * featureEvaluator->getFeatureSize();
561 featureMap.create( 1, varCount, CV_32SC1 );
562 featureMap.setTo(Scalar(-1));
563
564 for( vector< Ptr<CvCascadeBoost> >::const_iterator it = stageClassifiers.begin();
565 it != stageClassifiers.end(); it++ )
566 ((CvCascadeBoost*)((Ptr<CvCascadeBoost>)(*it)))->markUsedFeaturesInMap( featureMap );
567
568 for( int fi = 0, idx = 0; fi < varCount; fi++ )
569 if ( featureMap.at<int>(0, fi) >= 0 )
570 featureMap.ptr<int>(0)[fi] = idx++;
571}
