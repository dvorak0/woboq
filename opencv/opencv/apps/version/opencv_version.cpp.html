5#include <iostream>
6
7#include <opencv2/core.hpp>
8
9int main(int argc, const char** argv)
10{
11 cv::CommandLineParser parser(argc, argv,
12 "{ h|help | false | show this help message }"
13 "{ v|verbose | false | show build configuration log }"
14 );
15 if (parser.get<bool>("help"))
16 {
17 parser.printParams();
18 }
19 else if (parser.get<bool>("verbose"))
20 {
21 std::cout << cv::getBuildInformation().c_str() << std::endl;
22 }
23 else
24 {
25 std::cout << CV_VERSION << std::endl;
26 }
27 return 0;
28}
