44/*****************************************************************************************************
45
46Software for visualising cascade classifier models trained by OpenCV and to get a better
47understanding of the used features.
48
49USAGE:
50./visualise_models -model <model.xml> -image <ref.png> -data <output folder>
51
52LIMITS
53- Use an absolute path for the output folder to ensure the tool works
54- Only handles cascade classifier models
55- Handles stumps only for the moment
56- Needs a valid training/test sample window with the original model dimensions, passed as `ref.png`
57- Can handle HAAR and LBP features
58
59Created by: Puttemans Steven - April 2016
60*****************************************************************************************************/
61
62#include <opencv2/core/core.hpp>
63#include <opencv2/highgui/highgui.hpp>
64#include <opencv2/imgproc/imgproc.hpp>
65
66#include <fstream>
67#include <iostream>
68
69using namespace std;
70using namespace cv;
71
72struct rect_data{
73 int x;
74 int y;
75 int w;
76 int h;
77 float weight;
78};
79
80int main( int argc, const char** argv )
81{
82 // Read in the input arguments
83 string model = "";
84 string output_folder = "";
85 string image_ref = "";
86 for(int i = 1; i < argc; ++i )
87 {
88 if( !strcmp( argv[i], "-model" ) )
89 {
90 model = argv[++i];
91 }else if( !strcmp( argv[i], "-image" ) ){
92 image_ref = argv[++i];
93 }else if( !strcmp( argv[i], "-data" ) ){
94 output_folder = argv[++i];
95 }
96 }
97
98 // Value for timing
99 // You can increase this to have a better visualisation during the generation
100 int timing = 1;
101
102 // Value for cols of storing elements
103 int cols_prefered = 5;
104
105 // Open the XML model
106 FileStorage fs;
107 fs.open(model, FileStorage::READ);
108
109 // Get a the required information
110 // First decide which feature type we are using
111 FileNode cascade = fs["cascade"];
112 string feature_type = cascade["featureType"];
113 bool haar = false, lbp = false;
114 if (feature_type.compare("HAAR") == 0){
115 haar = true;
116 }
117 if (feature_type.compare("LBP") == 0){
118 lbp = true;
119 }
120 if ( feature_type.compare("HAAR") != 0 && feature_type.compare("LBP")){
121 cerr << "The model is not an HAAR or LBP feature based model!" << endl;
122 cerr << "Please select a model that can be visualized by the software." << endl;
123 return -1;
124 }
125
126 // We make a visualisation mask - which increases the window to make it at least a bit more visible
127 int resize_factor = 10;
128 int resize_storage_factor = 10;
129 Mat reference_image = imread(image_ref, IMREAD_GRAYSCALE );
130 Mat visualization;
131 resize(reference_image, visualization, Size(reference_image.cols * resize_factor, reference_image.rows * resize_factor));
132
133 // First recover for each stage the number of weak features and their index
134 // Important since it is NOT sequential when using LBP features
135 vector< vector<int> > stage_features;
136 FileNode stages = cascade["stages"];
137 FileNodeIterator it_stages = stages.begin(), it_stages_end = stages.end();
138 int idx = 0;
139 for( ; it_stages != it_stages_end; it_stages++, idx++ ){
140 vector<int> current_feature_indexes;
141 FileNode weak_classifiers = (*it_stages)["weakClassifiers"];
142 FileNodeIterator it_weak = weak_classifiers.begin(), it_weak_end = weak_classifiers.end();
143 vector<int> values;
144 for(int idy = 0; it_weak != it_weak_end; it_weak++, idy++ ){
145 (*it_weak)["internalNodes"] >> values;
146 current_feature_indexes.push_back( (int)values[2] );
147 }
148 stage_features.push_back(current_feature_indexes);
149 }
150
151 // If the output option has been chosen than we will store a combined image plane for
152 // each stage, containing all weak classifiers for that stage.
153 bool draw_planes = false;
154 stringstream output_video;
155 output_video << output_folder << "model_visualization.avi";
156 VideoWriter result_video;
157 if( output_folder.compare("") != 0 ){
158 draw_planes = true;
159 result_video.open(output_video.str(), CV_FOURCC('X','V','I','D'), 15, Size(reference_image.cols * resize_factor, reference_image.rows * resize_factor), false);
160 }
161
162 if(haar){
163 // Grab the corresponding features dimensions and weights
164 FileNode features = cascade["features"];
165 vector< vector< rect_data > > feature_data;
166 FileNodeIterator it_features = features.begin(), it_features_end = features.end();
167 for(int idf = 0; it_features != it_features_end; it_features++, idf++ ){
168 vector< rect_data > current_feature_rectangles;
169 FileNode rectangles = (*it_features)["rects"];
170 int nrects = (int)rectangles.size();
171 for(int k = 0; k < nrects; k++){
172 rect_data current_data;
173 FileNode single_rect = rectangles[k];
174 current_data.x = (int)single_rect[0];
175 current_data.y = (int)single_rect[1];
176 current_data.w = (int)single_rect[2];
177 current_data.h = (int)single_rect[3];
178 current_data.weight = (float)single_rect[4];
179 current_feature_rectangles.push_back(current_data);
180 }
181 feature_data.push_back(current_feature_rectangles);
182 }
183
184 // Loop over each possible feature on its index, visualise on the mask and wait a bit,
185 // then continue to the next feature.
186 // If visualisations should be stored then do the in between calculations
187 Mat image_plane;
188 Mat metadata = Mat::zeros(150, 1000, CV_8UC1);
189 vector< rect_data > current_rects;
190 for(int sid = 0; sid < (int)stage_features.size(); sid ++){
191 if(draw_planes){
192 int features_nmbr = (int)stage_features[sid].size();
193 int cols = cols_prefered;
194 int rows = features_nmbr / cols;
195 if( (features_nmbr % cols) > 0){
196 rows++;
197 }
198 image_plane = Mat::zeros(reference_image.rows * resize_storage_factor * rows, reference_image.cols * resize_storage_factor * cols, CV_8UC1);
199 }
200 for(int fid = 0; fid < (int)stage_features[sid].size(); fid++){
201 stringstream meta1, meta2;
202 meta1 << "Stage " << sid << " / Feature " << fid;
203 meta2 << "Rectangles: ";
204 Mat temp_window = visualization.clone();
205 Mat temp_metadata = metadata.clone();
206 int current_feature_index = stage_features[sid][fid];
207 current_rects = feature_data[current_feature_index];
208 Mat single_feature = reference_image.clone();
209 resize(single_feature, single_feature, Size(), resize_storage_factor, resize_storage_factor);
210 for(int i = 0; i < (int)current_rects.size(); i++){
211 rect_data local = current_rects[i];
212 if(draw_planes){
213 if(local.weight >= 0){
214 rectangle(single_feature, Rect(local.x * resize_storage_factor, local.y * resize_storage_factor, local.w * resize_storage_factor, local.h * resize_storage_factor), Scalar(0), CV_FILLED);
215 }else{
216 rectangle(single_feature, Rect(local.x * resize_storage_factor, local.y * resize_storage_factor, local.w * resize_storage_factor, local.h * resize_storage_factor), Scalar(255), CV_FILLED);
217 }
218 }
219 Rect part(local.x * resize_factor, local.y * resize_factor, local.w * resize_factor, local.h * resize_factor);
220 meta2 << part << " (w " << local.weight << ") ";
221 if(local.weight >= 0){
222 rectangle(temp_window, part, Scalar(0), CV_FILLED);
223 }else{
224 rectangle(temp_window, part, Scalar(255), CV_FILLED);
225 }
226 }
227 imshow("features", temp_window);
228 putText(temp_window, meta1.str(), Point(15,15), FONT_HERSHEY_SIMPLEX, 0.5, Scalar(255));
229 result_video.write(temp_window);
230 // Copy the feature image if needed
231 if(draw_planes){
232 single_feature.copyTo(image_plane(Rect(0 + (fid%cols_prefered)*single_feature.cols, 0 + (fid/cols_prefered) * single_feature.rows, single_feature.cols, single_feature.rows)));
233 }
234 putText(temp_metadata, meta1.str(), Point(15,15), FONT_HERSHEY_SIMPLEX, 0.5, Scalar(255));
235 putText(temp_metadata, meta2.str(), Point(15,40), FONT_HERSHEY_SIMPLEX, 0.5, Scalar(255));
236 imshow("metadata", temp_metadata);
237 waitKey(timing);
238 }
239 //Store the stage image if needed
240 if(draw_planes){
241 stringstream save_location;
242 save_location << output_folder << "stage_" << sid << ".png";
243 imwrite(save_location.str(), image_plane);
244 }
245 }
246 }
247
248 if(lbp){
249 // Grab the corresponding features dimensions and weights
250 FileNode features = cascade["features"];
251 vector<Rect> feature_data;
252 FileNodeIterator it_features = features.begin(), it_features_end = features.end();
253 for(int idf = 0; it_features != it_features_end; it_features++, idf++ ){
254 FileNode rectangle = (*it_features)["rect"];
255 Rect current_feature ((int)rectangle[0], (int)rectangle[1], (int)rectangle[2], (int)rectangle[3]);
256 feature_data.push_back(current_feature);
257 }
258
259 // Loop over each possible feature on its index, visualise on the mask and wait a bit,
260 // then continue to the next feature.
261 Mat image_plane;
262 Mat metadata = Mat::zeros(150, 1000, CV_8UC1);
263 for(int sid = 0; sid < (int)stage_features.size(); sid ++){
264 if(draw_planes){
265 int features_nmbr = (int)stage_features[sid].size();
266 int cols = cols_prefered;
267 int rows = features_nmbr / cols;
268 if( (features_nmbr % cols) > 0){
269 rows++;
270 }
271 image_plane = Mat::zeros(reference_image.rows * resize_storage_factor * rows, reference_image.cols * resize_storage_factor * cols, CV_8UC1);
272 }
273 for(int fid = 0; fid < (int)stage_features[sid].size(); fid++){
274 stringstream meta1, meta2;
275 meta1 << "Stage " << sid << " / Feature " << fid;
276 meta2 << "Rectangle: ";
277 Mat temp_window = visualization.clone();
278 Mat temp_metadata = metadata.clone();
279 int current_feature_index = stage_features[sid][fid];
280 Rect current_rect = feature_data[current_feature_index];
281 Mat single_feature = reference_image.clone();
282 resize(single_feature, single_feature, Size(), resize_storage_factor, resize_storage_factor);
283
284 // VISUALISATION
285 // The rectangle is the top left one of a 3x3 block LBP constructor
286 Rect resized(current_rect.x * resize_factor, current_rect.y * resize_factor, current_rect.width * resize_factor, current_rect.height * resize_factor);
287 meta2 << resized;
288 // Top left
289 rectangle(temp_window, resized, Scalar(255), 1);
290 // Top middle
291 rectangle(temp_window, Rect(resized.x + resized.width, resized.y, resized.width, resized.height), Scalar(255), 1);
292 // Top right
293 rectangle(temp_window, Rect(resized.x + 2*resized.width, resized.y, resized.width, resized.height), Scalar(255), 1);
294 // Middle left
295 rectangle(temp_window, Rect(resized.x, resized.y + resized.height, resized.width, resized.height), Scalar(255), 1);
296 // Middle middle
297 rectangle(temp_window, Rect(resized.x + resized.width, resized.y + resized.height, resized.width, resized.height), Scalar(255), CV_FILLED);
298 // Middle right
299 rectangle(temp_window, Rect(resized.x + 2*resized.width, resized.y + resized.height, resized.width, resized.height), Scalar(255), 1);
300 // Bottom left
301 rectangle(temp_window, Rect(resized.x, resized.y + 2*resized.height, resized.width, resized.height), Scalar(255), 1);
302 // Bottom middle
303 rectangle(temp_window, Rect(resized.x + resized.width, resized.y + 2*resized.height, resized.width, resized.height), Scalar(255), 1);
304 // Bottom right
305 rectangle(temp_window, Rect(resized.x + 2*resized.width, resized.y + 2*resized.height, resized.width, resized.height), Scalar(255), 1);
306
307 if(draw_planes){
308 Rect resized_inner(current_rect.x * resize_storage_factor, current_rect.y * resize_storage_factor, current_rect.width * resize_storage_factor, current_rect.height * resize_storage_factor);
309 // Top left
310 rectangle(single_feature, resized_inner, Scalar(255), 1);
311 // Top middle
312 rectangle(single_feature, Rect(resized_inner.x + resized_inner.width, resized_inner.y, resized_inner.width, resized_inner.height), Scalar(255), 1);
313 // Top right
314 rectangle(single_feature, Rect(resized_inner.x + 2*resized_inner.width, resized_inner.y, resized_inner.width, resized_inner.height), Scalar(255), 1);
315 // Middle left
316 rectangle(single_feature, Rect(resized_inner.x, resized_inner.y + resized_inner.height, resized_inner.width, resized_inner.height), Scalar(255), 1);
317 // Middle middle
318 rectangle(single_feature, Rect(resized_inner.x + resized_inner.width, resized_inner.y + resized_inner.height, resized_inner.width, resized_inner.height), Scalar(255), CV_FILLED);
319 // Middle right
320 rectangle(single_feature, Rect(resized_inner.x + 2*resized_inner.width, resized_inner.y + resized_inner.height, resized_inner.width, resized_inner.height), Scalar(255), 1);
321 // Bottom left
322 rectangle(single_feature, Rect(resized_inner.x, resized_inner.y + 2*resized_inner.height, resized_inner.width, resized_inner.height), Scalar(255), 1);
323 // Bottom middle
324 rectangle(single_feature, Rect(resized_inner.x + resized_inner.width, resized_inner.y + 2*resized_inner.height, resized_inner.width, resized_inner.height), Scalar(255), 1);
325 // Bottom right
326 rectangle(single_feature, Rect(resized_inner.x + 2*resized_inner.width, resized_inner.y + 2*resized_inner.height, resized_inner.width, resized_inner.height), Scalar(255), 1);
327
328 single_feature.copyTo(image_plane(Rect(0 + (fid%cols_prefered)*single_feature.cols, 0 + (fid/cols_prefered) * single_feature.rows, single_feature.cols, single_feature.rows)));
329 }
330
331 putText(temp_metadata, meta1.str(), Point(15,15), FONT_HERSHEY_SIMPLEX, 0.5, Scalar(255));
332 putText(temp_metadata, meta2.str(), Point(15,40), FONT_HERSHEY_SIMPLEX, 0.5, Scalar(255));
333 imshow("metadata", temp_metadata);
334 imshow("features", temp_window);
335 putText(temp_window, meta1.str(), Point(15,15), FONT_HERSHEY_SIMPLEX, 0.5, Scalar(255));
336 result_video.write(temp_window);
337
338 waitKey(timing);
339 }
340
341 //Store the stage image if needed
342 if(draw_planes){
343 stringstream save_location;
344 save_location << output_folder << "stage_" << sid << ".png";
345 imwrite(save_location.str(), image_plane);
346 }
347 }
348 }
349 return 0;
350}
