176 void allocate(int dims, const int* sizes, int type, int*& refcount,
177 uchar*& datastart, uchar*& data, size_t* step)
178 {
179 PyEnsureGIL gil;
180
181 int depth = CV_MAT_DEPTH(type);
182 int cn = CV_MAT_CN(type);
183 const int f = (int)(sizeof(size_t)/8);
184 int typenum = depth == CV_8U ? NPY_UBYTE : depth == CV_8S ? NPY_BYTE :
185 depth == CV_16U ? NPY_USHORT : depth == CV_16S ? NPY_SHORT :
186 depth == CV_32S ? NPY_INT : depth == CV_32F ? NPY_FLOAT :
187 depth == CV_64F ? NPY_DOUBLE : f*NPY_ULONGLONG + (f^1)*NPY_UINT;
188 int i;
189 npy_intp _sizes[CV_MAX_DIM+1];
190 for( i = 0; i < dims; i++ )
191 _sizes[i] = sizes[i];
192 if( cn > 1 )
193 {
194 /*if( _sizes[dims-1] == 1 )
195 _sizes[dims-1] = cn;
196 else*/
197 _sizes[dims++] = cn;
198 }
199 PyObject* o = PyArray_SimpleNew(dims, _sizes, typenum);
200 if(!o)
201 CV_Error_(CV_StsError, ("The numpy array of typenum=%d, ndims=%d can not be created", typenum, dims));
202 refcount = refcountFromPyObject(o);
203 npy_intp* _strides = PyArray_STRIDES((PyArrayObject*) o);
204 for( i = 0; i < dims - (cn > 1); i++ )
205 step[i] = (size_t)_strides[i];
206 datastart = data = (uchar*)PyArray_DATA((PyArrayObject*) o);
207 }
208
209 void deallocate(int* refcount, uchar*, uchar*)
210 {
211 PyEnsureGIL gil;
212 if( !refcount )
213 return;
214 PyObject* o = pyObjectFromRefcount(refcount);
215 Py_INCREF(o);
216 Py_DECREF(o);
217 }

224// special case, when the convertor needs full ArgInfo structure
225static int pyopencv_to(const PyObject* o, Mat& m, const ArgInfo info, bool allowND=true)
226{
227 if(!o || o == Py_None)
228 {
229 if( !m.data )
230 m.allocator = &g_numpyAllocator;
231 return true;
232 }
233
234 if( PyInt_Check(o) )
235 {
236 double v[] = {
237 static_cast<double>(PyInt_AsLong((PyObject*)o)),
238 0.,
239 0.,
240 0.,
241 };
242 m = Mat(4, 1, CV_64F, v).clone();
243 return true;
244 }
245 if( PyFloat_Check(o) )
246 {
247 double v[] = {PyFloat_AsDouble((PyObject*)o), 0., 0., 0.};
248 m = Mat(4, 1, CV_64F, v).clone();
249 return true;
250 }
251 if( PyTuple_Check(o) )
252 {
253 int i, sz = (int)PyTuple_Size((PyObject*)o);
254 m = Mat(sz, 1, CV_64F);
255 for( i = 0; i < sz; i++ )
256 {
257 PyObject* oi = PyTuple_GET_ITEM(o, i);
258 if( PyInt_Check(oi) )
259 m.at<double>(i) = (double)PyInt_AsLong(oi);
260 else if( PyFloat_Check(oi) )
261 m.at<double>(i) = (double)PyFloat_AsDouble(oi);
262 else
263 {
264 failmsg("%s is not a numerical tuple", info.name);
265 m.release();
266 return false;
267 }
268 }
269 return true;
270 }
271
272 if( !PyArray_Check(o) )
273 {
274 failmsg("%s is not a numpy array, neither a scalar", info.name);
275 return false;
276 }
277
278 PyArrayObject* oarr = (PyArrayObject*) o;
279
280 bool needcopy = false, needcast = false;
281 int typenum = PyArray_TYPE(oarr), new_typenum = typenum;
282 int type = typenum == NPY_UBYTE ? CV_8U :
283 typenum == NPY_BYTE ? CV_8S :
284 typenum == NPY_USHORT ? CV_16U :
285 typenum == NPY_SHORT ? CV_16S :
286 typenum == NPY_INT ? CV_32S :
287 typenum == NPY_INT32 ? CV_32S :
288 typenum == NPY_FLOAT ? CV_32F :
289 typenum == NPY_DOUBLE ? CV_64F : -1;
290
291 if( type < 0 )
292 {
293 if( typenum == NPY_INT64 || typenum == NPY_UINT64 || typenum == NPY_LONG )
294 {
295 needcopy = needcast = true;
296 new_typenum = NPY_INT;
297 type = CV_32S;
298 }
299 else
300 {
301 failmsg("%s data type = %d is not supported", info.name, typenum);
302 return false;
303 }
304 }
305
306 int ndims = PyArray_NDIM(oarr);
307 if(ndims >= CV_MAX_DIM)
308 {
309 failmsg("%s dimensionality (=%d) is too high", info.name, ndims);
310 return false;
311 }
312
313 int size[CV_MAX_DIM+1];
314 size_t step[CV_MAX_DIM+1], elemsize = CV_ELEM_SIZE1(type);
315 const npy_intp* _sizes = PyArray_DIMS(oarr);
316 const npy_intp* _strides = PyArray_STRIDES(oarr);
317 bool ismultichannel = ndims == 3 && _sizes[2] <= CV_CN_MAX;
318
319 for( int i = ndims-1; i >= 0 && !needcopy; i-- )
320 {
321 // these checks handle cases of
322 // a) multi-dimensional (ndims > 2) arrays, as well as simpler 1- and 2-dimensional cases
323 // b) transposed arrays, where _strides[] elements go in non-descending order
324 // c) flipped arrays, where some of _strides[] elements are negative
325 if( (i == ndims-1 && (size_t)_strides[i] != elemsize) ||
326 (i < ndims-1 && _strides[i] < _strides[i+1]) )
327 needcopy = true;
328 }
329
330 if( ismultichannel && _strides[1] != (npy_intp)elemsize*_sizes[2] )
331 needcopy = true;
332
333 if (needcopy)
334 {
335 if (info.outputarg)
336 {
337 failmsg("Layout of the output array %s is incompatible with cv::Mat (step[ndims-1] != elemsize or step[1] != elemsize*nchannels)", info.name);
338 return false;
339 }
340
341 if( needcast ) {
342 o = PyArray_Cast(oarr, new_typenum);
343 oarr = (PyArrayObject*) o;
344 }
345 else {
346 oarr = PyArray_GETCONTIGUOUS(oarr);
347 o = (PyObject*) oarr;
348 }
349
350 _strides = PyArray_STRIDES(oarr);
351 }
352
353 for(int i = 0; i < ndims; i++)
354 {
355 size[i] = (int)_sizes[i];
356 step[i] = (size_t)_strides[i];
357 }
358
359 // handle degenerate case
360 if( ndims == 0) {
361 size[ndims] = 1;
362 step[ndims] = elemsize;
363 ndims++;
364 }
365
366 if( ismultichannel )
367 {
368 ndims--;
369 type |= CV_MAKETYPE(0, size[2]);
370 }
371
372 if( ndims > 2 && !allowND )
373 {
374 failmsg("%s has more than 2 dimensions", info.name);
375 return false;
376 }
377
378 m = Mat(ndims, size, type, PyArray_DATA(oarr), step);
379
380 if( m.data )
381 {
382 m.refcount = refcountFromPyObject(o);
383 if (!needcopy)
384 {
385 m.addref(); // protect the original numpy array from deallocation
386 // (since Mat destructor will decrement the reference counter)
387 }
388 };
389 m.allocator = &g_numpyAllocator;
390
391 return true;
392}

394static PyObject* pyopencv_from(const Mat& m)
395{
396 if( !m.data )
397 Py_RETURN_NONE;
398 Mat temp, *p = (Mat*)&m;
399 if(!p->refcount || p->allocator != &g_numpyAllocator)
400 {
401 temp.allocator = &g_numpyAllocator;
402 ERRWRAP2(m.copyTo(temp));
403 p = &temp;
404 }
405 p->addref();
406 return pyObjectFromRefcount(p->refcount);
407}

409static bool pyopencv_to(PyObject *o, Scalar& s, const char *name = "<unknown>")
410{
411 if(!o || o == Py_None)
412 return true;
413 if (PySequence_Check(o)) {
414 PyObject *fi = PySequence_Fast(o, name);
415 if (fi == NULL)
416 return false;
417 if (4 < PySequence_Fast_GET_SIZE(fi))
418 {
419 failmsg("Scalar value for argument '%s' is longer than 4", name);
420 return false;
421 }
422 for (Py_ssize_t i = 0; i < PySequence_Fast_GET_SIZE(fi); i++) {
423 PyObject *item = PySequence_Fast_GET_ITEM(fi, i);
424 if (PyFloat_Check(item) || PyInt_Check(item)) {
425 s[(int)i] = PyFloat_AsDouble(item);
426 } else {
427 failmsg("Scalar value for argument '%s' is not numeric", name);
428 return false;
429 }
430 }
431 Py_DECREF(fi);
432 } else {
433 if (PyFloat_Check(o) || PyInt_Check(o)) {
434 s[0] = PyFloat_AsDouble(o);
435 } else {
436 failmsg("Scalar value for argument '%s' is not numeric", name);
437 return false;
438 }
439 }
440 return true;
441}

699template<typename _Tp> struct pyopencvVecConverter
700{
701 static bool to(PyObject* obj, vector<_Tp>& value, const ArgInfo info)
702 {
703 typedef typename DataType<_Tp>::channel_type _Cp;
704 if(!obj || obj == Py_None)
705 return true;
706 if (PyArray_Check(obj))
707 {
708 Mat m;
709 pyopencv_to(obj, m, info);
710 m.copyTo(value);
711 }
712 if (!PySequence_Check(obj))
713 return false;
714 PyObject *seq = PySequence_Fast(obj, info.name);
715 if (seq == NULL)
716 return false;
717 int i, j, n = (int)PySequence_Fast_GET_SIZE(seq);
718 value.resize(n);
719
720 int type = DataType<_Tp>::type;
721 int depth = CV_MAT_DEPTH(type), channels = CV_MAT_CN(type);
722 PyObject** items = PySequence_Fast_ITEMS(seq);
723
724 for( i = 0; i < n; i++ )
725 {
726 PyObject* item = items[i];
727 PyObject* seq_i = 0;
728 PyObject** items_i = &item;
729 _Cp* data = (_Cp*)&value[i];
730
731 if( channels == 2 && PyComplex_CheckExact(item) )
732 {
733 Py_complex c = PyComplex_AsCComplex(obj);
734 data[0] = saturate_cast<_Cp>(c.real);
735 data[1] = saturate_cast<_Cp>(c.imag);
736 continue;
737 }
738 if( channels > 1 )
739 {
740 if( PyArray_Check(item))
741 {
742 Mat src;
743 pyopencv_to(item, src, info);
744 if( src.dims != 2 || src.channels() != 1 ||
745 ((src.cols != 1 || src.rows != channels) &&
746 (src.cols != channels || src.rows != 1)))
747 break;
748 Mat dst(src.rows, src.cols, depth, data);
749 src.convertTo(dst, type);
750 if( dst.data != (uchar*)data )
751 break;
752 continue;
753 }
754
755 seq_i = PySequence_Fast(item, info.name);
756 if( !seq_i || (int)PySequence_Fast_GET_SIZE(seq_i) != channels )
757 {
758 Py_XDECREF(seq_i);
759 break;
760 }
761 items_i = PySequence_Fast_ITEMS(seq_i);
762 }
763
764 for( j = 0; j < channels; j++ )
765 {
766 PyObject* item_ij = items_i[j];
767 if( PyInt_Check(item_ij))
768 {
769 int v = (int)PyInt_AsLong(item_ij);
770 if( v == -1 && PyErr_Occurred() )
771 break;
772 data[j] = saturate_cast<_Cp>(v);
773 }
774 else if( PyLong_Check(item_ij))
775 {
776 int v = (int)PyLong_AsLong(item_ij);
777 if( v == -1 && PyErr_Occurred() )
778 break;
779 data[j] = saturate_cast<_Cp>(v);
780 }
781 else if( PyFloat_Check(item_ij))
782 {
783 double v = PyFloat_AsDouble(item_ij);
784 if( PyErr_Occurred() )
785 break;
786 data[j] = saturate_cast<_Cp>(v);
787 }
788 else
789 break;
790 }
791 Py_XDECREF(seq_i);
792 if( j < channels )
793 break;
794 }
795 Py_DECREF(seq);
796 return i == n;
797 }

802 return PyTuple_New(0);
803 Mat src((int)value.size(), DataType<_Tp>::channels, DataType<_Tp>::depth, (uchar*)&value[0]);
804 return pyopencv_from(src);

991{
992 (void)name;
993 bool ok = false;
994 PyObject* keys = PyObject_CallMethod(o,(char*)"keys",0);
995 PyObject* values = PyObject_CallMethod(o,(char*)"values",0);
996
997 if( keys && values )
998 {
999 int i, n = (int)PyList_GET_SIZE(keys);
1000 for( i = 0; i < n; i++ )
1001 {
1002 PyObject* key = PyList_GET_ITEM(keys, i);
1003 PyObject* item = PyList_GET_ITEM(values, i);
1004 if( !PyString_Check(key) )
1005 break;
1006 std::string k = PyString_AsString(key);
1007 if( PyString_Check(item) )
1008 {
1009 const char* value = PyString_AsString(item);
1010 p.setString(k, value);
1011 }
1012 else if( !!PyBool_Check(item) )
1013 p.setBool(k, item == Py_True);
1014 else if( PyInt_Check(item) )
1015 {
1016 int value = (int)PyInt_AsLong(item);
1017 if( strcmp(k.c_str(), "algorithm") == 0 )
1018 p.setAlgorithm(value);
1019 else
1020 p.setInt(k, value);
1021 }
1022 else if( PyFloat_Check(item) )
1023 {
1024 double value = PyFloat_AsDouble(item);
1025 p.setDouble(k, value);
1026 }
1027 else
1028 break;
1029 }
1030 ok = i == n && !PyErr_Occurred();
1031 }
1032
1033 Py_XDECREF(keys);
1034 Py_XDECREF(values);
