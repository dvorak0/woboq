202static
203inline LARGE_INTEGER get_filetime_offset()
204{
205 SYSTEMTIME s;
206 FILETIME f;
207 LARGE_INTEGER t;
208
209 s.wYear = 1970;
210 s.wMonth = 1;
211 s.wDay = 1;
212 s.wHour = 0;
213 s.wMinute = 0;
214 s.wSecond = 0;
215 s.wMilliseconds = 0;
216 SystemTimeToFileTime(&s, &f);
217 t.QuadPart = f.dwHighDateTime;
218 t.QuadPart <<= 32;
219 t.QuadPart |= f.dwLowDateTime;
220 return t;
221}

223static
224inline void get_monotonic_time(timespec *tv)
225{
226 LARGE_INTEGER t;
227 FILETIME f;
228 double microseconds;
229 static LARGE_INTEGER offset;
230 static double frequencyToMicroseconds;
231 static int initialized = 0;
232 static BOOL usePerformanceCounter = 0;
233
234 if (!initialized)
235 {
236 LARGE_INTEGER performanceFrequency;
237 initialized = 1;
238 usePerformanceCounter = QueryPerformanceFrequency(&performanceFrequency);
239 if (usePerformanceCounter)
240 {
241 QueryPerformanceCounter(&offset);
242 frequencyToMicroseconds = (double)performanceFrequency.QuadPart / 1000000.;
243 }
244 else
245 {
246 offset = get_filetime_offset();
247 frequencyToMicroseconds = 10.;
248 }
249 }
250
251 if (usePerformanceCounter)
252 {
253 QueryPerformanceCounter(&t);
254 } else {
255 GetSystemTimeAsFileTime(&f);
256 t.QuadPart = f.dwHighDateTime;
257 t.QuadPart <<= 32;
258 t.QuadPart |= f.dwLowDateTime;
259 }
260
261 t.QuadPart -= offset.QuadPart;
262 microseconds = (double)t.QuadPart / frequencyToMicroseconds;
263 t.QuadPart = microseconds;
264 tv->tv_sec = t.QuadPart / 1000000;
265 tv->tv_nsec = (t.QuadPart % 1000000) * 1000;
266}

285static
286inline timespec get_monotonic_time_diff(timespec start, timespec end)
287{
288 timespec temp;
289 if (end.tv_nsec - start.tv_nsec < 0)
290 {
291 temp.tv_sec = end.tv_sec - start.tv_sec - 1;
292 temp.tv_nsec = 1000000000 + end.tv_nsec - start.tv_nsec;
293 }
294 else
295 {
296 temp.tv_sec = end.tv_sec - start.tv_sec;
297 temp.tv_nsec = end.tv_nsec - start.tv_nsec;
298 }
299 return temp;
300}
301
302static
303inline double get_monotonic_time_diff_ms(timespec time1, timespec time2)
304{
305 timespec delta = get_monotonic_time_diff(time1, time2);
306 double milliseconds = delta.tv_sec * 1000 + (double)delta.tv_nsec / 1000000.0;
307
308 return milliseconds;
309}

312static int get_number_of_cpus(void)
313{
314#if LIBAVFORMAT_BUILD < CALC_FFMPEG_VERSION(52, 111, 0)
315 return 1;
316#elif defined WIN32 || defined _WIN32
317 SYSTEM_INFO sysinfo;
318 GetSystemInfo( &sysinfo );
319
320 return (int)sysinfo.dwNumberOfProcessors;
321#elif defined __linux__
322 return (int)sysconf( _SC_NPROCESSORS_ONLN );
323#elif defined __APPLE__
324 int numCPU=0;
325 int mib[4];
326 size_t len = sizeof(numCPU);
327
328 // set the mib for hw.ncpu
329 mib[0] = CTL_HW;
330 mib[1] = HW_AVAILCPU; // alternatively, try HW_NCPU;
331
332 // get the number of CPUs from the system
333 sysctl(mib, 2, &numCPU, &len, NULL, 0);
334
335 if( numCPU < 1 )
336 {
337 mib[1] = HW_NCPU;
338 sysctl( mib, 2, &numCPU, &len, NULL, 0 );
339
340 if( numCPU < 1 )
341 numCPU = 1;
342 }
343
344 return (int)numCPU;
345#else
346 return 1;
347#endif
348}

378{
379 AVInterruptCallbackMetadata* metadata = (AVInterruptCallbackMetadata*)ptr;

676static int LockCallBack(void **mutex, AVLockOp op)
677{
678 ImplMutex* localMutex = reinterpret_cast<ImplMutex*>(*mutex);
679 switch (op)
680 {
681 case AV_LOCK_CREATE:
682 localMutex = reinterpret_cast<ImplMutex*>(malloc(sizeof(ImplMutex)));
683 localMutex->init();
684 *mutex = localMutex;
685 if (!*mutex)
686 return 1;
687 break;
688
689 case AV_LOCK_OBTAIN:
690 localMutex->lock();
691 break;
692
693 case AV_LOCK_RELEASE:
694 localMutex->unlock();
695 break;
696
697 case AV_LOCK_DESTROY:
698 localMutex->destroy();
699 free(localMutex);
700 localMutex = NULL;
701 *mutex = NULL;
702 break;
703 }
704 return 0;
705}

744bool CvCapture_FFMPEG::open( const char* _filename )
745{
746 unsigned i;
747 bool valid = false;
748
749 close();
750
751#if USE_AV_INTERRUPT_CALLBACK
752 /* interrupt callback */
753 interrupt_metadata.timeout_after_ms = LIBAVFORMAT_INTERRUPT_OPEN_TIMEOUT_MS;
754 get_monotonic_time(&interrupt_metadata.value);
755
756 ic = avformat_alloc_context();
757 ic->interrupt_callback.callback = _opencv_ffmpeg_interrupt_callback;
758 ic->interrupt_callback.opaque = &interrupt_metadata;
759#endif
760
761#if LIBAVFORMAT_BUILD >= CALC_FFMPEG_VERSION(52, 111, 0)
762 av_dict_set(&dict, "rtsp_transport", "tcp", 0);
763 int err = avformat_open_input(&ic, _filename, NULL, &dict);
764#else
765 int err = av_open_input_file(&ic, _filename, NULL, 0, NULL);
766#endif
767
768 if (err < 0)
769 {
770 CV_WARN("Error opening file");
771 goto exit_func;
772 }
773 err =
774#if LIBAVFORMAT_BUILD >= CALC_FFMPEG_VERSION(53, 6, 0)
775 avformat_find_stream_info(ic, NULL);
776#else
777 av_find_stream_info(ic);
778#endif
779 if (err < 0)
780 {
781 CV_WARN("Could not find codec parameters");
782 goto exit_func;
783 }
784 for(i = 0; i < ic->nb_streams; i++)
785 {
786#if LIBAVFORMAT_BUILD > 4628
787 AVCodecContext *enc = ic->streams[i]->codec;
788#else
789 AVCodecContext *enc = &ic->streams[i]->codec;
790#endif
791
792//#ifdef FF_API_THREAD_INIT
793// avcodec_thread_init(enc, get_number_of_cpus());
794//#else
795 enc->thread_count = get_number_of_cpus();
796//#endif
797
798#if LIBAVFORMAT_BUILD < CALC_FFMPEG_VERSION(53, 2, 0)
799#define AVMEDIA_TYPE_VIDEO CODEC_TYPE_VIDEO
800#endif
801
802 if( AVMEDIA_TYPE_VIDEO == enc->codec_type && video_stream < 0)
803 {
804 // backup encoder' width/height
805 int enc_width = enc->width;
806 int enc_height = enc->height;
807
808 AVCodec *codec = avcodec_find_decoder(enc->codec_id);
809 if (!codec ||
810#if LIBAVCODEC_VERSION_INT >= ((53<<16)+(8<<8)+0)
811 avcodec_open2(enc, codec, NULL)
812#else
813 avcodec_open(enc, codec)
814#endif
815 < 0)
816 goto exit_func;
817
818 // checking width/height (since decoder can sometimes alter it, eg. vp6f)
819 if (enc_width && (enc->width != enc_width)) { enc->width = enc_width; }
820 if (enc_height && (enc->height != enc_height)) { enc->height = enc_height; }
821
822 video_stream = i;
823 video_st = ic->streams[i];
824#if LIBAVCODEC_BUILD >= (LIBAVCODEC_VERSION_MICRO >= 100 \
825 ? CALC_FFMPEG_VERSION(55, 45, 101) : CALC_FFMPEG_VERSION(55, 28, 1))
826 picture = av_frame_alloc();
827#else
828 picture = avcodec_alloc_frame();
829#endif
830
831 frame.width = enc->width;
832 frame.height = enc->height;
833 frame.cn = 3;
834 frame.step = 0;
835 frame.data = NULL;
836 break;
837 }
838 }
839
840 if(video_stream >= 0) valid = true;
841
842exit_func:
843
844#if USE_AV_INTERRUPT_CALLBACK
845 // deactivate interrupt callback
846 interrupt_metadata.timeout_after_ms = 0;
847#endif
848
849 if( !valid )
850 close();
851
852 return valid;
853}

856bool CvCapture_FFMPEG::grabFrame()
857{
858 bool valid = false;
859 int got_picture;
860
861 int count_errs = 0;
862 const int max_number_of_attempts = 1 << 9;
863
864 if( !ic || !video_st ) return false;
865
866 if( ic->streams[video_stream]->nb_frames > 0 &&
867 frame_number > ic->streams[video_stream]->nb_frames )
868 return false;
869
870 picture_pts = AV_NOPTS_VALUE_;
871
872#if USE_AV_INTERRUPT_CALLBACK
873 // activate interrupt callback
874 get_monotonic_time(&interrupt_metadata.value);
875 interrupt_metadata.timeout_after_ms = LIBAVFORMAT_INTERRUPT_READ_TIMEOUT_MS;
876#endif
877
878 // get the next frame
879 while (!valid)
880 {
881 av_free_packet (&packet);
882
883#if USE_AV_INTERRUPT_CALLBACK
884 if (interrupt_metadata.timeout)
885 {
886 valid = false;
887 break;
888 }
889#endif
890
891 int ret = av_read_frame(ic, &packet);
892 if (ret == AVERROR(EAGAIN)) continue;
893
894 /* else if (ret < 0) break; */
895
896 if( packet.stream_index != video_stream )
897 {
898 av_free_packet (&packet);
899 count_errs++;
900 if (count_errs > max_number_of_attempts)
901 break;
902 continue;
903 }
904
905 // Decode video frame
906 #if LIBAVFORMAT_BUILD >= CALC_FFMPEG_VERSION(53, 2, 0)
907 avcodec_decode_video2(video_st->codec, picture, &got_picture, &packet);
908 #elif LIBAVFORMAT_BUILD > 4628
909 avcodec_decode_video(video_st->codec,
910 picture, &got_picture,
911 packet.data, packet.size);
912 #else
913 avcodec_decode_video(&video_st->codec,
914 picture, &got_picture,
915 packet.data, packet.size);
916 #endif
917
918 // Did we get a video frame?
919 if(got_picture)
920 {
921 //picture_pts = picture->best_effort_timestamp;
922 if( picture_pts == AV_NOPTS_VALUE_ )
923 picture_pts = packet.pts != AV_NOPTS_VALUE_ && packet.pts != 0 ? packet.pts : packet.dts;
924 frame_number++;
925 valid = true;
926 }
927 else
928 {
929 count_errs++;
930 if (count_errs > max_number_of_attempts)
931 break;
932 }
933 }
934
935 if( valid && first_frame_number < 0 )
936 first_frame_number = dts_to_frame_number(picture_pts);
937
938#if USE_AV_INTERRUPT_CALLBACK
939 // deactivate interrupt callback
940 interrupt_metadata.timeout_after_ms = 0;
941#endif
942
943 // return if we have a new picture or not
944 return valid;
945}

949{

951 return false;
952
953 if( img_convert_ctx == NULL ||
954 frame.width != video_st->codec->width ||
955 frame.height != video_st->codec->height ||
956 frame.data == NULL )
957 {
958 // Some sws_scale optimizations have some assumptions about alignment of data/step/width/height
959 // Also we use coded_width/height to workaround problem with legacy ffmpeg versions (like n0.8)
960 int buffer_width = video_st->codec->coded_width, buffer_height = video_st->codec->coded_height;
961
962 img_convert_ctx = sws_getCachedContext(
963 img_convert_ctx,
964 buffer_width, buffer_height,
965 video_st->codec->pix_fmt,
966 buffer_width, buffer_height,
967 AV_PIX_FMT_BGR24,
968 SWS_BICUBIC,
969 NULL, NULL, NULL
970 );
971
972 if (img_convert_ctx == NULL)
973 return false;//CV_Error(0, "Cannot initialize the conversion context!");
974
975#if USE_AV_FRAME_GET_BUFFER
976 av_frame_unref(&rgb_picture);

983 return false;
984 }
985#else

987 avcodec_align_dimensions2(video_st->codec, &buffer_width, &buffer_height, aligns);
988 rgb_picture.data[0] = (uint8_t*)realloc(rgb_picture.data[0],
989 avpicture_get_size( AV_PIX_FMT_BGR24,
990 buffer_width, buffer_height ));
991 avpicture_fill( (AVPicture*)&rgb_picture, rgb_picture.data[0],

999 }
1000
1001 sws_scale(
1002 img_convert_ctx,
1003 picture->data,
1004 picture->linesize,
1005 0, video_st->codec->coded_height,
1006 rgb_picture.data,
1007 rgb_picture.linesize
1008 );
1009
1010 *data = frame.data;
1011 *step = frame.step;
1012 *width = frame.width;
1013 *height = frame.height;
1014 *cn = frame.cn;
1015
1016 return true;
1017}

1133void CvCapture_FFMPEG::seek(int64_t _frame_number)
1134{
1135 _frame_number = std::min(_frame_number, get_total_frames());
1136 int delta = 16;
1137
1138 // if we have not grabbed a single frame before first seek, let's read the first frame
1139 // and get some valuable information during the process
1140 if( first_frame_number < 0 && get_total_frames() > 1 )
1141 grabFrame();
1142
1143 for(;;)
1144 {
1145 int64_t _frame_number_temp = std::max(_frame_number-delta, (int64_t)0);
1146 double sec = (double)_frame_number_temp / get_fps();
1147 int64_t time_stamp = ic->streams[video_stream]->start_time;
1148 double time_base = r2d(ic->streams[video_stream]->time_base);
1149 time_stamp += (int64_t)(sec / time_base + 0.5);
1150 if (get_total_frames() > 1) av_seek_frame(ic, video_stream, time_stamp, AVSEEK_FLAG_BACKWARD);
1151 avcodec_flush_buffers(ic->streams[video_stream]->codec);
1152 if( _frame_number > 0 )
1153 {
1154 grabFrame();
1155
1156 if( _frame_number > 1 )
1157 {
1158 frame_number = dts_to_frame_number(picture_pts) - first_frame_number;
1159 //printf("_frame_number = %d, frame_number = %d, delta = %d\n",
1160 // (int)_frame_number, (int)frame_number, delta);
1161
1162 if( frame_number < 0 || frame_number > _frame_number-1 )
1163 {
1164 if( _frame_number_temp == 0 || delta >= INT_MAX/4 )
1165 break;
1166 delta = delta < 16 ? delta*2 : delta*3/2;
1167 continue;
1168 }
1169 while( frame_number < _frame_number-1 )
1170 {
1171 if(!grabFrame())
1172 break;
1173 }
1174 frame_number++;
1175 break;
1176 }
1177 else
1178 {
1179 frame_number = 1;
1180 break;
1181 }
1182 }
1183 else
1184 {
1185 frame_number = 0;
1186 break;
1187 }
1188 }
1189}

1375/* add a video output stream to the container */
1376static AVStream *icv_add_video_stream_FFMPEG(AVFormatContext *oc,
1377 CV_CODEC_ID codec_id,
1378 int w, int h, int bitrate,
1379 double fps, int pixel_format)
1380{
1381 AVCodecContext *c;
1382 AVStream *st;
1383 int frame_rate, frame_rate_base;
1384 AVCodec *codec;
1385
1386#if LIBAVFORMAT_BUILD >= CALC_FFMPEG_VERSION(53, 10, 0)
1387 st = avformat_new_stream(oc, 0);
1388#else
1389 st = av_new_stream(oc, 0);
1390#endif
1391
1392 if (!st) {
1393 CV_WARN("Could not allocate stream");
1394 return NULL;
1395 }
1396
1397#if LIBAVFORMAT_BUILD > 4628
1398 c = st->codec;
1399#else
1400 c = &(st->codec);
1401#endif
1402
1403#if LIBAVFORMAT_BUILD > 4621
1404 c->codec_id = av_guess_codec(oc->oformat, NULL, oc->filename, NULL, AVMEDIA_TYPE_VIDEO);
1405#else
1406 c->codec_id = oc->oformat->video_codec;
1407#endif
1408
1409 if(codec_id != CV_CODEC(CODEC_ID_NONE)){
1410 c->codec_id = codec_id;
1411 }
1412
1413 //if(codec_tag) c->codec_tag=codec_tag;
1414 codec = avcodec_find_encoder(c->codec_id);
1415
1416 c->codec_type = AVMEDIA_TYPE_VIDEO;
1417
1418 /* put sample parameters */
1419 int64_t lbit_rate = (int64_t)bitrate;
1420 lbit_rate += (bitrate / 2);
1421 lbit_rate = std::min(lbit_rate, (int64_t)INT_MAX);
1422 c->bit_rate = lbit_rate;
1423
1424 // took advice from
1425 // http://ffmpeg-users.933282.n4.nabble.com/warning-clipping-1-dct-coefficients-to-127-127-td934297.html
1426 c->qmin = 3;
1427
1428 /* resolution must be a multiple of two */
1429 c->width = w;
1430 c->height = h;
1431
1432 /* time base: this is the fundamental unit of time (in seconds) in terms
1433 of which frame timestamps are represented. for fixed-fps content,
1434 timebase should be 1/framerate and timestamp increments should be
1435 identically 1. */
1436 frame_rate=(int)(fps+0.5);
1437 frame_rate_base=1;
1438 while (fabs((double)frame_rate/frame_rate_base) - fps > 0.001){
1439 frame_rate_base*=10;
1440 frame_rate=(int)(fps*frame_rate_base + 0.5);
1441 }
1442#if LIBAVFORMAT_BUILD > 4752
1443 c->time_base.den = frame_rate;
1444 c->time_base.num = frame_rate_base;
1445 /* adjust time base for supported framerates */
1446 if(codec && codec->supported_framerates){
1447 const AVRational *p= codec->supported_framerates;
1448 AVRational req = {frame_rate, frame_rate_base};
1449 const AVRational *best=NULL;
1450 AVRational best_error= {INT_MAX, 1};
1451 for(; p->den!=0; p++){
1452 AVRational error= av_sub_q(req, *p);
1453 if(error.num <0) error.num *= -1;
1454 if(av_cmp_q(error, best_error) < 0){
1455 best_error= error;
1456 best= p;
1457 }
1458 }
1459 c->time_base.den= best->num;
1460 c->time_base.num= best->den;
1461 }
1462#else
1463 c->frame_rate = frame_rate;
1464 c->frame_rate_base = frame_rate_base;
1465#endif
1466
1467 c->gop_size = 12; /* emit one intra frame every twelve frames at most */
1468 c->pix_fmt = (AVPixelFormat) pixel_format;
1469
1470 if (c->codec_id == CV_CODEC(CODEC_ID_MPEG2VIDEO)) {
1471 c->max_b_frames = 2;
1472 }
1473 if (c->codec_id == CV_CODEC(CODEC_ID_MPEG1VIDEO) || c->codec_id == CV_CODEC(CODEC_ID_MSMPEG4V3)){
1474 /* needed to avoid using macroblocks in which some coeffs overflow
1475 this doesnt happen with normal video, it just happens here as the
1476 motion of the chroma plane doesnt match the luma plane */
1477 /* avoid FFMPEG warning 'clipping 1 dct coefficients...' */
1478 c->mb_decision=2;
1479 }
1480#if LIBAVCODEC_VERSION_INT>0x000409
1481 // some formats want stream headers to be seperate
1482 if(oc->oformat->flags & AVFMT_GLOBALHEADER)
1483 {
1484 c->flags |= CODEC_FLAG_GLOBAL_HEADER;
1485 }
1486#endif
1487
1488#if LIBAVCODEC_BUILD >= CALC_FFMPEG_VERSION(52, 42, 0)
1489 st->avg_frame_rate = (AVRational){frame_rate, frame_rate_base};
1490#endif
1491
1492 return st;
1493}

1501 uint8_t * outbuf, uint32_t outbuf_size,
1502#endif
1503 AVFrame * picture )
1504{
1505#if LIBAVFORMAT_BUILD > 4628
1506 AVCodecContext * c = video_st->codec;
1507#else
1508 AVCodecContext * c = &(video_st->codec);
1509#endif
1510 int ret = OPENCV_NO_FRAMES_WRITTEN_CODE;
1511
1512 if (oc->oformat->flags & AVFMT_RAWPICTURE) {
1513 /* raw video case. The API will change slightly in the near
1514 futur for that */
1515 AVPacket pkt;
1516 av_init_packet(&pkt);
1517
1518#ifndef PKT_FLAG_KEY
1519#define PKT_FLAG_KEY AV_PKT_FLAG_KEY
1520#endif
1521
1522 pkt.flags |= PKT_FLAG_KEY;
1523 pkt.stream_index= video_st->index;
1524 pkt.data= (uint8_t *)picture;
1525 pkt.size= sizeof(AVPicture);
1526
1527 ret = av_write_frame(oc, &pkt);
1528 } else {
1529 /* encode the image */
1530 AVPacket pkt;
1531 av_init_packet(&pkt);
1532#if LIBAVCODEC_BUILD >= CALC_FFMPEG_VERSION(54, 1, 0)
1533 int got_output = 0;
1534 pkt.data = NULL;
1535 pkt.size = 0;
1536 ret = avcodec_encode_video2(c, &pkt, picture, &got_output);
1537 if (ret < 0)
1538 ;
1539 else if (got_output) {
1540 if (pkt.pts != (int64_t)AV_NOPTS_VALUE)
1541 pkt.pts = av_rescale_q(pkt.pts, c->time_base, video_st->time_base);
1542 if (pkt.dts != (int64_t)AV_NOPTS_VALUE)
1543 pkt.dts = av_rescale_q(pkt.dts, c->time_base, video_st->time_base);
1544 if (pkt.duration)
1545 pkt.duration = av_rescale_q(pkt.duration, c->time_base, video_st->time_base);
1546 pkt.stream_index= video_st->index;
1547 ret = av_write_frame(oc, &pkt);
1548 av_free_packet(&pkt);
1549 }
1550 else
1551 ret = OPENCV_NO_FRAMES_WRITTEN_CODE;
1552#else
1553 int out_size = avcodec_encode_video(c, outbuf, outbuf_size, picture);
1554 /* if zero size, it means the image was buffered */
1555 if (out_size > 0) {
1556#if LIBAVFORMAT_BUILD > 4752
1557 if(c->coded_frame->pts != (int64_t)AV_NOPTS_VALUE)
1558 pkt.pts = av_rescale_q(c->coded_frame->pts, c->time_base, video_st->time_base);
1559#else
1560 pkt.pts = c->coded_frame->pts;
1561#endif
1562 if(c->coded_frame->key_frame)
1563 pkt.flags |= PKT_FLAG_KEY;
1564 pkt.stream_index= video_st->index;
1565 pkt.data= outbuf;
1566 pkt.size= out_size;
1567
1568 /* write the compressed frame in the media file */
1569 ret = av_write_frame(oc, &pkt);
1570 }
1571#endif
1572 }
1573 return ret;
1574}

1576/// write a frame with FFMPEG
1577bool CvVideoWriter_FFMPEG::writeFrame( const unsigned char* data, int step, int width, int height, int cn, int origin )
1578{
1579 // check parameters
1580 if (input_pix_fmt == AV_PIX_FMT_BGR24) {
1581 if (cn != 3) {
1582 return false;
1583 }
1584 }
1585 else if (input_pix_fmt == AV_PIX_FMT_GRAY8) {
1586 if (cn != 1) {
1587 return false;
1588 }
1589 }
1590 else {
1591 assert(false);
1592 }
1593
1594 if( (width & -2) != frame_width || (height & -2) != frame_height || !data )
1595 return false;
1596 width = frame_width;
1597 height = frame_height;
1598
1599 // typecast from opaque data type to implemented struct
1600#if LIBAVFORMAT_BUILD > 4628
1601 AVCodecContext *c = video_st->codec;
1602#else
1603 AVCodecContext *c = &(video_st->codec);
1604#endif
1605
1606 // FFmpeg contains SIMD optimizations which can sometimes read data past
1607 // the supplied input buffer. To ensure that doesn't happen, we pad the
1608 // step to a multiple of 32 (that's the minimal alignment for which Valgrind
1609 // doesn't raise any warnings).
1610 const int STEP_ALIGNMENT = 32;
1611 if( step % STEP_ALIGNMENT != 0 )
1612 {
1613 int aligned_step = (step + STEP_ALIGNMENT - 1) & -STEP_ALIGNMENT;
1614
1615 if( !aligned_input )
1616 {
1617 aligned_input = (unsigned char*)av_mallocz(aligned_step * height);
1618 }
1619
1620 if (origin == 1)
1621 for( int y = 0; y < height; y++ )
1622 memcpy(aligned_input + y*aligned_step, data + (height-1-y)*step, step);
1623 else
1624 for( int y = 0; y < height; y++ )
1625 memcpy(aligned_input + y*aligned_step, data + y*step, step);
1626
1627 data = aligned_input;
1628 step = aligned_step;
1629 }
1630
1631 if ( c->pix_fmt != input_pix_fmt ) {
1632 assert( input_picture );
1633 // let input_picture point to the raw data buffer of 'image'
1634 avpicture_fill((AVPicture *)input_picture, (uint8_t *) data,
1635 (AVPixelFormat)input_pix_fmt, width, height);
1636 input_picture->linesize[0] = step;
1637
1638 if( !img_convert_ctx )
1639 {
1640 img_convert_ctx = sws_getContext(width,
1641 height,
1642 (AVPixelFormat)input_pix_fmt,
1643 c->width,
1644 c->height,
1645 c->pix_fmt,
1646 SWS_BICUBIC,
1647 NULL, NULL, NULL);
1648 if( !img_convert_ctx )
1649 return false;
1650 }
1651
1652 if ( sws_scale(img_convert_ctx, input_picture->data,
1653 input_picture->linesize, 0,
1654 height,
1655 picture->data, picture->linesize) < 0 )
1656 return false;
1657 }
1658 else{
1659 avpicture_fill((AVPicture *)picture, (uint8_t *) data,
1660 (AVPixelFormat)input_pix_fmt, width, height);
1661 picture->linesize[0] = step;
1662 }
1663
1664 picture->pts = frame_idx;
1665 bool ret = icv_av_write_frame_FFMPEG( oc, video_st, outbuf, outbuf_size, picture) >= 0;
1666 frame_idx++;
1667
1668 return ret;
1669}

1671/// close video output stream and free associated memory
1672void CvVideoWriter_FFMPEG::close()
1673{
1674 // nothing to do if already released
1675 if ( !picture )
1676 return;
1677
1678 /* no more frame to compress. The codec has a latency of a few
1679 frames if using B frames, so we get the last frames by
1680 passing the same picture again */
1681 // TODO -- do we need to account for latency here?
1682
1683 /* write the trailer, if any */
1684 if(ok && oc)
1685 {
1686 if( (oc->oformat->flags & AVFMT_RAWPICTURE) == 0 )
1687 {
1688 for(;;)
1689 {
1690 int ret = icv_av_write_frame_FFMPEG( oc, video_st, outbuf, outbuf_size, NULL);
1691 if( ret == OPENCV_NO_FRAMES_WRITTEN_CODE || ret < 0 )
1692 break;
1693 }
1694 }
1695 av_write_trailer(oc);
1696 }
1697
1698 if( img_convert_ctx )
1699 {
1700 sws_freeContext(img_convert_ctx);
1701 img_convert_ctx = 0;
1702 }
1703
1704 // free pictures
1705#if LIBAVFORMAT_BUILD > 4628
1706 if( video_st->codec->pix_fmt != input_pix_fmt)
1707#else
1708 if( video_st->codec.pix_fmt != input_pix_fmt)
1709#endif
1710 {
1711 if(picture->data[0])
1712 free(picture->data[0]);
1713 picture->data[0] = 0;
1714 }
1715 av_free(picture);
1716
1717 if (input_picture)
1718 av_free(input_picture);
1719
1720 /* close codec */
1721#if LIBAVFORMAT_BUILD > 4628
1722 avcodec_close(video_st->codec);
1723#else
1724 avcodec_close(&(video_st->codec));
1725#endif
1726
1727 av_free(outbuf);
1728
1729 if (!(fmt->flags & AVFMT_NOFILE))
1730 {
1731 /* close the output file */
1732
1733#if LIBAVCODEC_VERSION_INT < ((52<<16)+(123<<8)+0)
1734#if LIBAVCODEC_VERSION_INT >= ((51<<16)+(49<<8)+0)
1735 url_fclose(oc->pb);
1736#else
1737 url_fclose(&oc->pb);
1738#endif
1739#else
1740 avio_close(oc->pb);
1741#endif
1742
1743 }
1744
1745 /* free the stream */
1746 avformat_free_context(oc);
1747
1748 av_freep(&aligned_input);
1749
1750 init();
1751}

1753/// Create a video writer object that uses FFMPEG
1754bool CvVideoWriter_FFMPEG::open( const char * filename, int fourcc,
1755 double fps, int width, int height, bool is_color )
1756{
1757 CV_CODEC_ID codec_id = CV_CODEC(CODEC_ID_NONE);
1758 int err, codec_pix_fmt;
1759 double bitrate_scale = 1;
1760
1761 close();
1762
1763 // check arguments
1764 if( !filename )
1765 return false;
1766 if(fps <= 0)
1767 return false;
1768
1769 // we allow frames of odd width or height, but in this case we truncate
1770 // the rightmost column/the bottom row. Probably, this should be handled more elegantly,
1771 // but some internal functions inside FFMPEG swscale require even width/height.
1772 width &= -2;
1773 height &= -2;
1774 if( width <= 0 || height <= 0 )
1775 return false;
1776
1777 /* auto detect the output format from the name and fourcc code. */
1778
1779#if LIBAVFORMAT_BUILD >= CALC_FFMPEG_VERSION(53, 2, 0)
1780 fmt = av_guess_format(NULL, filename, NULL);
1781#else
1782 fmt = guess_format(NULL, filename, NULL);
1783#endif
1784
1785 if (!fmt)
1786 return false;
1787
1788 /* determine optimal pixel format */
1789 if (is_color) {
1790 input_pix_fmt = AV_PIX_FMT_BGR24;
1791 }
1792 else {
1793 input_pix_fmt = AV_PIX_FMT_GRAY8;
1794 }
1795
1796 /* Lookup codec_id for given fourcc */
1797#if LIBAVCODEC_VERSION_INT<((51<<16)+(49<<8)+0)
1798 if( (codec_id = codec_get_bmp_id( fourcc )) == CV_CODEC(CODEC_ID_NONE) )
1799 return false;
1800#else
1801 const struct AVCodecTag * tags[] = { codec_bmp_tags, NULL};
1802 if( (codec_id = av_codec_get_id(tags, fourcc)) == CV_CODEC(CODEC_ID_NONE) )
1803 return false;
1804#endif
1805
1806 // alloc memory for context
1807#if LIBAVFORMAT_BUILD >= CALC_FFMPEG_VERSION(53, 2, 0)
1808 oc = avformat_alloc_context();
1809#else
1810 oc = av_alloc_format_context();
1811#endif
1812 assert (oc);
1813
1814 /* set file name */
1815 oc->oformat = fmt;
1816 snprintf(oc->filename, sizeof(oc->filename), "%s", filename);
1817
1818 /* set some options */
1819 oc->max_delay = (int)(0.7*AV_TIME_BASE); /* This reduces buffer underrun warnings with MPEG */
1820
1821 // set a few optimal pixel formats for lossless codecs of interest..
1822 switch (codec_id) {
1823#if LIBAVCODEC_VERSION_INT>((50<<16)+(1<<8)+0)
1824 case CV_CODEC(CODEC_ID_JPEGLS):
1825 // BGR24 or GRAY8 depending on is_color...
1826 codec_pix_fmt = input_pix_fmt;
1827 break;
1828#endif
1829 case CV_CODEC(CODEC_ID_HUFFYUV):
1830 codec_pix_fmt = AV_PIX_FMT_YUV422P;
1831 break;
1832 case CV_CODEC(CODEC_ID_MJPEG):
1833 case CV_CODEC(CODEC_ID_LJPEG):
1834 codec_pix_fmt = AV_PIX_FMT_YUVJ420P;
1835 bitrate_scale = 3;
1836 break;
1837 case CV_CODEC(CODEC_ID_RAWVIDEO):
1838 codec_pix_fmt = input_pix_fmt == AV_PIX_FMT_GRAY8 ||
1839 input_pix_fmt == AV_PIX_FMT_GRAY16LE ||
1840 input_pix_fmt == AV_PIX_FMT_GRAY16BE ? input_pix_fmt : AV_PIX_FMT_YUV420P;
1841 break;
1842 default:
1843 // good for lossy formats, MPEG, etc.
1844 codec_pix_fmt = AV_PIX_FMT_YUV420P;
1845 break;
1846 }
1847
1848 double bitrate = MIN(bitrate_scale*fps*width*height, (double)INT_MAX/2);
1849
1850 // TODO -- safe to ignore output audio stream?
1851 video_st = icv_add_video_stream_FFMPEG(oc, codec_id,
1852 width, height, (int)(bitrate + 0.5),
1853 fps, codec_pix_fmt);
1854
1855 /* set the output parameters (must be done even if no
1856 parameters). */
1857#if LIBAVFORMAT_BUILD < CALC_FFMPEG_VERSION(53, 2, 0)
1858 if (av_set_parameters(oc, NULL) < 0) {
1859 return false;
1860 }
1861#endif
1862
1863#if 0
1864#if FF_API_DUMP_FORMAT
1865 dump_format(oc, 0, filename, 1);
1866#else
1867 av_dump_format(oc, 0, filename, 1);
1868#endif
1869#endif
1870
1871 /* now that all the parameters are set, we can open the audio and
1872 video codecs and allocate the necessary encode buffers */
1873 if (!video_st){
1874 return false;
1875 }
1876
1877 AVCodec *codec;
1878 AVCodecContext *c;
1879
1880#if LIBAVFORMAT_BUILD > 4628
1881 c = (video_st->codec);
1882#else
1883 c = &(video_st->codec);
1884#endif
1885
1886 c->codec_tag = fourcc;
1887 /* find the video encoder */
1888 codec = avcodec_find_encoder(c->codec_id);
1889 if (!codec) {
1890 fprintf(stderr, "Could not find encoder for codec id %d: %s", c->codec_id, icvFFMPEGErrStr(
1891 #if LIBAVFORMAT_BUILD >= CALC_FFMPEG_VERSION(53, 2, 0)
1892 AVERROR_ENCODER_NOT_FOUND
1893 #else
1894 -1
1895 #endif
1896 ));
1897 return false;
1898 }
1899
1900 int64_t lbit_rate = (int64_t)c->bit_rate;
1901 lbit_rate += (bitrate / 2);
1902 lbit_rate = std::min(lbit_rate, (int64_t)INT_MAX);
1903 c->bit_rate_tolerance = (int)lbit_rate;
1904 c->bit_rate = (int)lbit_rate;
1905
1906 /* open the codec */
1907 if ((err=
1908#if LIBAVCODEC_VERSION_INT >= ((53<<16)+(8<<8)+0)
1909 avcodec_open2(c, codec, NULL)
1910#else
1911 avcodec_open(c, codec)
1912#endif
1913 ) < 0) {
1914 fprintf(stderr, "Could not open codec '%s': %s", codec->name, icvFFMPEGErrStr(err));
1915 return false;
1916 }
1917
1918 outbuf = NULL;
1919
1920 if (!(oc->oformat->flags & AVFMT_RAWPICTURE)) {
1921 /* allocate output buffer */
1922 /* assume we will never get codec output with more than 4 bytes per pixel... */
1923 outbuf_size = width*height*4;
1924 outbuf = (uint8_t *) av_malloc(outbuf_size);
1925 }
1926
1927 bool need_color_convert;
1928 need_color_convert = (c->pix_fmt != input_pix_fmt);
1929
1930 /* allocate the encoded raw picture */
1931 picture = icv_alloc_picture_FFMPEG(c->pix_fmt, c->width, c->height, need_color_convert);
1932 if (!picture) {
1933 return false;
1934 }
1935
1936 /* if the output format is not our input format, then a temporary
1937 picture of the input format is needed too. It is then converted
1938 to the required output format */
1939 input_picture = NULL;
1940 if ( need_color_convert ) {
1941 input_picture = icv_alloc_picture_FFMPEG(input_pix_fmt, c->width, c->height, false);
1942 if (!input_picture) {
1943 return false;
1944 }
1945 }
1946
1947 /* open the output file, if needed */
1948 if (!(fmt->flags & AVFMT_NOFILE)) {
1949#if LIBAVFORMAT_BUILD < CALC_FFMPEG_VERSION(53, 2, 0)
1950 if (url_fopen(&oc->pb, filename, URL_WRONLY) < 0)
1951#else
1952 if (avio_open(&oc->pb, filename, AVIO_FLAG_WRITE) < 0)
1953#endif
1954 {
1955 return false;
1956 }
1957 }
1958
1959#if LIBAVFORMAT_BUILD >= CALC_FFMPEG_VERSION(52, 111, 0)
1960 /* write the stream header, if any */
1961 err=avformat_write_header(oc, NULL);
1962#else
1963 err=av_write_header( oc );
1964#endif
1965
1966 if(err < 0)
1967 {
1968 close();
1969 remove(filename);
1970 return false;
1971 }
1972 frame_width = width;
1973 frame_height = height;
1974 frame_idx = 0;
1975 ok = true;
1976
1977 return true;
1978}

2116AVStream* OutputMediaStream_FFMPEG::addVideoStream(AVFormatContext *oc, CV_CODEC_ID codec_id, int w, int h, int bitrate, double fps, AVPixelFormat pixel_format)
2117{
2118 #if LIBAVFORMAT_BUILD >= CALC_FFMPEG_VERSION(53, 10, 0)
2119 AVStream* st = avformat_new_stream(oc, 0);
2120 #else
2121 AVStream* st = av_new_stream(oc, 0);
2122 #endif
2123 if (!st)
2124 return 0;
2125
2126 #if LIBAVFORMAT_BUILD > 4628
2127 AVCodecContext* c = st->codec;
2128 #else
2129 AVCodecContext* c = &(st->codec);
2130 #endif
2131
2132 c->codec_id = codec_id;
2133 c->codec_type = AVMEDIA_TYPE_VIDEO;
2134
2135 // put sample parameters
2136 unsigned long long lbit_rate = static_cast<unsigned long long>(bitrate);
2137 lbit_rate += (bitrate / 4);
2138 lbit_rate = std::min(lbit_rate, static_cast<unsigned long long>(std::numeric_limits<int>::max()));
2139 c->bit_rate = bitrate;
2140
2141 // took advice from
2142 // http://ffmpeg-users.933282.n4.nabble.com/warning-clipping-1-dct-coefficients-to-127-127-td934297.html
2143 c->qmin = 3;
2144
2145 // resolution must be a multiple of two
2146 c->width = w;
2147 c->height = h;
2148
2149 AVCodec* codec = avcodec_find_encoder(c->codec_id);
2150
2151 // time base: this is the fundamental unit of time (in seconds) in terms
2152 // of which frame timestamps are represented. for fixed-fps content,
2153 // timebase should be 1/framerate and timestamp increments should be
2154 // identically 1
2155
2156 int frame_rate = static_cast<int>(fps+0.5);
2157 int frame_rate_base = 1;
2158 while (fabs(static_cast<double>(frame_rate)/frame_rate_base) - fps > 0.001)
2159 {
2160 frame_rate_base *= 10;
2161 frame_rate = static_cast<int>(fps*frame_rate_base + 0.5);
2162 }
2163 c->time_base.den = frame_rate;
2164 c->time_base.num = frame_rate_base;
2165
2166 #if LIBAVFORMAT_BUILD > 4752
2167 // adjust time base for supported framerates
2168 if (codec && codec->supported_framerates)
2169 {
2170 AVRational req = {frame_rate, frame_rate_base};
2171 const AVRational* best = NULL;
2172 AVRational best_error = {INT_MAX, 1};
2173
2174 for (const AVRational* p = codec->supported_framerates; p->den!=0; ++p)
2175 {
2176 AVRational error = av_sub_q(req, *p);
2177
2178 if (error.num < 0)
2179 error.num *= -1;
2180
2181 if (av_cmp_q(error, best_error) < 0)
2182 {
2183 best_error= error;
2184 best= p;
2185 }
2186 }
2187
2188 c->time_base.den= best->num;
2189 c->time_base.num= best->den;
2190 }
2191 #endif
