94/***************************** CvCapture structure ******************************/
95
96struct CvCapture
97{
98 virtual ~CvCapture() {}
99 virtual double getProperty(int) { return 0; }
100 virtual bool setProperty(int, double) { return 0; }
101 virtual bool grabFrame() { return true; }
102 virtual IplImage* retrieveFrame(int) { return 0; }
103 virtual int getCaptureDomain() { return CV_CAP_ANY; } // Return the type of the capture object: CV_CAP_VFW, etc...
104};

112};
