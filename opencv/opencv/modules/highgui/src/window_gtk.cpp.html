105void cvImageWidgetSetImage(CvImageWidget * widget, const CvArr *arr){
106 CvMat * mat, stub;
107 int origin=0;
108
109 //printf("cvImageWidgetSetImage\n");
110
111 if( CV_IS_IMAGE_HDR( arr ))
112 origin = ((IplImage*)arr)->origin;
113
114 mat = cvGetMat(arr, &stub);
115
116 if(widget->original_image && !CV_ARE_SIZES_EQ(mat, widget->original_image)){
117 cvReleaseMat( &widget->original_image );
118 }
119 if(!widget->original_image){
120 widget->original_image = cvCreateMat( mat->rows, mat->cols, CV_8UC3 );
121 gtk_widget_queue_resize( GTK_WIDGET( widget ) );
122 }
123 cvConvertImage( mat, widget->original_image,
124 (origin != 0 ? CV_CVTIMG_FLIP : 0) + CV_CVTIMG_SWAP_RB );
125 if(widget->scaled_image){
126 cvResize( widget->original_image, widget->scaled_image, CV_INTER_AREA );
127 }
128
129 // window does not refresh without this
130 gtk_widget_queue_draw( GTK_WIDGET(widget) );
131}

146static void
147cvImageWidget_realize (GtkWidget *widget)
148{
149 GdkWindowAttr attributes;
150 gint attributes_mask;
151
152 //printf("cvImageWidget_realize\n");
153 g_return_if_fail (widget != NULL);
154 g_return_if_fail (CV_IS_IMAGE_WIDGET (widget));
155
156 GTK_WIDGET_SET_FLAGS (widget, GTK_REALIZED);
157
158 attributes.x = widget->allocation.x;
159 attributes.y = widget->allocation.y;
160 attributes.width = widget->allocation.width;
161 attributes.height = widget->allocation.height;
162 attributes.wclass = GDK_INPUT_OUTPUT;
163 attributes.window_type = GDK_WINDOW_CHILD;
164 attributes.event_mask = gtk_widget_get_events (widget) |
165 GDK_EXPOSURE_MASK | GDK_BUTTON_PRESS_MASK |
166 GDK_BUTTON_RELEASE_MASK | GDK_POINTER_MOTION_MASK;
167 attributes.visual = gtk_widget_get_visual (widget);
168 attributes.colormap = gtk_widget_get_colormap (widget);
169
170 attributes_mask = GDK_WA_X | GDK_WA_Y | GDK_WA_VISUAL | GDK_WA_COLORMAP;
171 widget->window = gdk_window_new (widget->parent->window, &attributes, attributes_mask);
172
173 widget->style = gtk_style_attach (widget->style, widget->window);
174
175 gdk_window_set_user_data (widget->window, widget);
176
177 gtk_style_set_background (widget->style, widget->window, GTK_STATE_ACTIVE);
178}

180static CvSize cvImageWidget_calc_size( int im_width, int im_height, int max_width, int max_height ){
181 float aspect = (float)im_width/(float)im_height;
182 float max_aspect = (float)max_width/(float)max_height;
183 if(aspect > max_aspect){
184 return cvSize( max_width, cvRound(max_width/aspect) );
185 }
186 return cvSize( cvRound(max_height*aspect), max_height );
187}

189static void
190cvImageWidget_size_request (GtkWidget *widget,
191 GtkRequisition *requisition)
192{
193 CvImageWidget * image_widget = CV_IMAGE_WIDGET( widget );
194
195 //printf("cvImageWidget_size_request ");
196 // the case the first time cvShowImage called or when AUTOSIZE
197 if( image_widget->original_image &&
198 ((image_widget->flags & CV_WINDOW_AUTOSIZE) ||
199 (image_widget->flags & CV_WINDOW_NO_IMAGE)))
200 {
201 //printf("original ");
202 requisition->width = image_widget->original_image->cols;
203 requisition->height = image_widget->original_image->rows;
204 }
205 // default case
206 else if(image_widget->scaled_image){
207 //printf("scaled ");
208 requisition->width = image_widget->scaled_image->cols;
209 requisition->height = image_widget->scaled_image->rows;
210 }
211 // the case before cvShowImage called
212 else{
213 //printf("default ");
214 requisition->width = 320;
215 requisition->height = 240;
216 }
217 //printf("%d %d\n",requisition->width, requisition->height);
218}

220static void cvImageWidget_set_size(GtkWidget * widget, int max_width, int max_height){
221 CvImageWidget * image_widget = CV_IMAGE_WIDGET( widget );
222
223 //printf("cvImageWidget_set_size %d %d\n", max_width, max_height);
224
225 // don't allow to set the size
226 if(image_widget->flags & CV_WINDOW_AUTOSIZE) return;
227 if(!image_widget->original_image) return;
228
229 CvSize scaled_image_size = cvImageWidget_calc_size( image_widget->original_image->cols,
230 image_widget->original_image->rows, max_width, max_height );
231
232 if( image_widget->scaled_image &&
233 ( image_widget->scaled_image->cols != scaled_image_size.width ||
234 image_widget->scaled_image->rows != scaled_image_size.height ))
235 {
236 cvReleaseMat( &image_widget->scaled_image );
237 }
238 if( !image_widget->scaled_image ){
239 image_widget->scaled_image = cvCreateMat( scaled_image_size.height, scaled_image_size.width, CV_8UC3 );
240
241
242 }
243 assert( image_widget->scaled_image );
244}

246static void
247cvImageWidget_size_allocate (GtkWidget *widget,
248 GtkAllocation *allocation)
249{
250 CvImageWidget *image_widget;
251
252 //printf("cvImageWidget_size_allocate\n");
253 g_return_if_fail (widget != NULL);
254 g_return_if_fail (CV_IS_IMAGE_WIDGET (widget));
255 g_return_if_fail (allocation != NULL);
256
257 widget->allocation = *allocation;
258 image_widget = CV_IMAGE_WIDGET (widget);
259
260
261 if( (image_widget->flags & CV_WINDOW_AUTOSIZE)==0 && image_widget->original_image ){
262 // (re) allocated scaled image
263 if( image_widget->flags & CV_WINDOW_NO_IMAGE ){
264 cvImageWidget_set_size( widget, image_widget->original_image->cols,
265 image_widget->original_image->rows);
266 }
267 else{
268 cvImageWidget_set_size( widget, allocation->width, allocation->height );
269 }
270 cvResize( image_widget->original_image, image_widget->scaled_image, CV_INTER_AREA );
271 }
272
273 if (GTK_WIDGET_REALIZED (widget))
274 {
275 image_widget = CV_IMAGE_WIDGET (widget);
276
277 if( image_widget->original_image &&
278 ((image_widget->flags & CV_WINDOW_AUTOSIZE) ||
279 (image_widget->flags & CV_WINDOW_NO_IMAGE)) )
280 {
281 widget->allocation.width = image_widget->original_image->cols;
282 widget->allocation.height = image_widget->original_image->rows;
283 gdk_window_move_resize( widget->window, allocation->x, allocation->y,
284 image_widget->original_image->cols, image_widget->original_image->rows );
285 if(image_widget->flags & CV_WINDOW_NO_IMAGE){
286 image_widget->flags &= ~CV_WINDOW_NO_IMAGE;
287 gtk_widget_queue_resize( GTK_WIDGET(widget) );
288 }
289 }
290 else{
291 gdk_window_move_resize (widget->window,
292 allocation->x, allocation->y,
293 allocation->width, allocation->height );
294
295 }
296 }
297}

299static void
300cvImageWidget_destroy (GtkObject *object)
301{
302 CvImageWidget *image_widget;
303
304 g_return_if_fail (object != NULL);
305 g_return_if_fail (CV_IS_IMAGE_WIDGET (object));
306
307 image_widget = CV_IMAGE_WIDGET (object);
308
309 cvReleaseMat( &image_widget->scaled_image );
310 cvReleaseMat( &image_widget->original_image );
311
312 if (GTK_OBJECT_CLASS (parent_class)->destroy)
313 (* GTK_OBJECT_CLASS (parent_class)->destroy) (object);
314}

316static void cvImageWidget_class_init (CvImageWidgetClass * klass)
317{
318 GtkObjectClass *object_class;
319 GtkWidgetClass *widget_class;
320
321 object_class = (GtkObjectClass*) klass;
322 widget_class = (GtkWidgetClass*) klass;
323
324 parent_class = GTK_WIDGET_CLASS( gtk_type_class (gtk_widget_get_type ()) );
325
326 object_class->destroy = cvImageWidget_destroy;
327
328 widget_class->realize = cvImageWidget_realize;
329 widget_class->size_request = cvImageWidget_size_request;
330 widget_class->size_allocate = cvImageWidget_size_allocate;
331 widget_class->button_press_event = NULL;
332 widget_class->button_release_event = NULL;
333 widget_class->motion_notify_event = NULL;
334}

465CV_IMPL int cvStartWindowThread(){
466#ifdef HAVE_GTHREAD
467 cvInitSystem(0,NULL);
468 if (!thread_started) {
469 if (!g_thread_supported ()) {
470 /* the GThread system wasn't inited, so init it */
471 g_thread_init(NULL);
472 }
473
474 // this mutex protects the window resources
475 window_mutex = g_mutex_new();
476
477 // protects the 'last key pressed' variable
478 last_key_mutex = g_mutex_new();
479
480 // conditional that indicates a key has been pressed
481 cond_have_key = g_cond_new();
482
483#if !GLIB_CHECK_VERSION(2, 32, 0)
484 // this is the window update thread
485 window_thread = g_thread_create((GThreadFunc) icvWindowThreadLoop,
486 NULL, TRUE, NULL);
487#else
488 window_thread = g_thread_new("OpenCV window update", (GThreadFunc)icvWindowThreadLoop, NULL);
489#endif
490 }
491 thread_started = window_thread!=NULL;
492 return thread_started;
493#else
494 return 0;
495#endif
496}

498#ifdef HAVE_GTHREAD
499gpointer icvWindowThreadLoop(){
500 while(1){
501 g_mutex_lock(window_mutex);
502 gtk_main_iteration_do(FALSE);
503 g_mutex_unlock(window_mutex);
504
505 // little sleep
506 g_usleep(500);
507
508 g_thread_yield();
509 }
510 return NULL;
511}
512
513#define CV_LOCK_MUTEX() \
514if(thread_started && g_thread_self()!=window_thread){ g_mutex_lock( window_mutex ); } else { }
515
516#define CV_UNLOCK_MUTEX() \
517if(thread_started && g_thread_self()!=window_thread){ g_mutex_unlock( window_mutex); } else { }
518
519#else
520#define CV_LOCK_MUTEX()
521#define CV_UNLOCK_MUTEX()
522#endif

570void cvSetModeWindow_GTK( const char* name, double prop_value)//Yannick Verdie
571{
572
573 CV_FUNCNAME( "cvSetModeWindow_GTK" );
574
575 __BEGIN__;
576
577 CvWindow* window;
578
579 if(!name)
580 CV_ERROR( CV_StsNullPtr, "NULL name string" );
581
582 window = icvFindWindowByName( name );
583 if( !window )
584 CV_ERROR( CV_StsNullPtr, "NULL window" );
585
586 if(window->flags & CV_WINDOW_AUTOSIZE)//if the flag CV_WINDOW_AUTOSIZE is set
587 EXIT;
588
589 //so easy to do fullscreen here, Linux rocks !
590
591 if (window->status==CV_WINDOW_FULLSCREEN && prop_value==CV_WINDOW_NORMAL)
592 {
593 CV_LOCK_MUTEX();
594 gtk_window_unfullscreen(GTK_WINDOW(window->frame));
595 window->status=CV_WINDOW_NORMAL;
596 CV_UNLOCK_MUTEX();
597 EXIT;
598 }
599
600 if (window->status==CV_WINDOW_NORMAL && prop_value==CV_WINDOW_FULLSCREEN)
601 {
602 CV_LOCK_MUTEX();
603 gtk_window_fullscreen(GTK_WINDOW(window->frame));
604 window->status=CV_WINDOW_FULLSCREEN;
605 CV_UNLOCK_MUTEX();
606 EXIT;
607 }
608
609 __END__;
610}

696 void createGlContext(CvWindow* window)
697 {
698 GdkGLConfig* glconfig;
699
700 CV_FUNCNAME( "createGlContext" );
701
702 __BEGIN__;
703
704 // Try double-buffered visual
705 glconfig = gdk_gl_config_new_by_mode((GdkGLConfigMode)(GDK_GL_MODE_RGB | GDK_GL_MODE_DEPTH | GDK_GL_MODE_DOUBLE));
706 if (!glconfig)
707 CV_ERROR( CV_OpenGlApiCallError, "Can't Create A GL Device Context" );
708
709 // Set OpenGL-capability to the widget
710 if (!gtk_widget_set_gl_capability(window->widget, glconfig, NULL, TRUE, GDK_GL_RGBA_TYPE))
711 CV_ERROR( CV_OpenGlApiCallError, "Can't Create A GL Device Context" );
712
713 window->useGl = true;
714
715 __END__;
716 }
717
718 void drawGl(CvWindow* window)
719 {
720 CV_FUNCNAME( "drawGl" );
721
722 __BEGIN__;
723
724 GdkGLContext* glcontext = gtk_widget_get_gl_context(window->widget);
725 GdkGLDrawable* gldrawable = gtk_widget_get_gl_drawable(window->widget);
726
727 if (!gdk_gl_drawable_gl_begin (gldrawable, glcontext))
728 CV_ERROR( CV_OpenGlApiCallError, "Can't Activate The GL Rendering Context" );
729
730 glViewport(0, 0, window->widget->allocation.width, window->widget->allocation.height);
731
732 glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
733
734 if (window->glDrawCallback)
735 window->glDrawCallback(window->glDrawData);
736
737 if (gdk_gl_drawable_is_double_buffered (gldrawable))
738 gdk_gl_drawable_swap_buffers(gldrawable);
739 else
740 glFlush();
741
742 gdk_gl_drawable_gl_end(gldrawable);
743
744 __END__;
745 }
746}

751static gboolean cvImageWidget_expose(GtkWidget* widget, GdkEventExpose* event, gpointer data)
752{
753#ifdef HAVE_OPENGL
754 CvWindow* window = (CvWindow*)data;
755
756 if (window->useGl)
757 {
758 drawGl(window);
759 return TRUE;
760 }
761#else
762 (void)data;
763#endif
764
765 CvImageWidget *image_widget;
766
767 g_return_val_if_fail (widget != NULL, FALSE);
768 g_return_val_if_fail (CV_IS_IMAGE_WIDGET (widget), FALSE);
769 g_return_val_if_fail (event != NULL, FALSE);
770
771 if (event->count > 0)
772 return FALSE;
773
774 image_widget = CV_IMAGE_WIDGET (widget);
775
776 gdk_window_clear_area (widget->window,
777 0, 0,

990static void icvDeleteWindow( CvWindow* window )
991{
992 CvTrackbar* trackbar;
993
994 if( window->prev )
995 window->prev->next = window->next;
996 else
997 hg_windows = window->next;
998
999 if( window->next )
1000 window->next->prev = window->prev;
1001
1002 window->prev = window->next = 0;
1003
1004 gtk_widget_destroy( window->frame );
1005
1006 for( trackbar = window->toolbar.first; trackbar != 0; )
1007 {
1008 CvTrackbar* next = trackbar->next;
1009 cvFree( &trackbar );
1010 trackbar = next;
1011 }
1012
1013 cvFree( &window );
1014
1015 // if last window...
1016 if( hg_windows == 0 )
1017 {
1018#ifdef HAVE_GTHREAD
1019 if( thread_started )
1020 {
1021 // send key press signal to jump out of any waiting cvWaitKey's
1022 g_cond_broadcast( cond_have_key );
1023 }
1024 else
1025 {
1026#endif
1027 // Some GTK+ modules (like the Unity module) use GDBusConnection,
1028 // which has a habit of postponing cleanup by performing it via
1029 // idle sources added to the main loop. Since this was the last window,
1030 // we can assume that no event processing is going to happen in the
1031 // nearest future, so we should force that cleanup (by handling all pending
1032 // events) while we still have the chance.
1033 // This is not needed if thread_started is true, because the background
1034 // thread will process events continuously.
1035 while( gtk_events_pending() )
1036 gtk_main_iteration();
1037#ifdef HAVE_GTHREAD
1038 }
1039#endif
1040 }
1041}

1490static gboolean icvOnKeyPress( GtkWidget * /*widget*/,
1491 GdkEventKey* event, gpointer /*user_data*/ )
1492{
1493 int code = 0;
1494
1495 switch( event->keyval )
1496 {
1497 case GDK_Escape:
1498 code = 27;
1499 break;
1500 case GDK_Return:
1501 case GDK_Linefeed:
1502 code = '\n';
1503 break;
1504 case GDK_Tab:
1505 code = '\t';
1506 break;
1507 default:
1508 code = event->keyval;
1509 }
1510
1511 code |= event->state << 16;
1512
1513#ifdef HAVE_GTHREAD
1514 if(thread_started) g_mutex_lock(last_key_mutex);
1515#endif
1516
1517 last_key = code;
1518
1519#ifdef HAVE_GTHREAD
1520 if(thread_started){
1521 // signal any waiting threads
1522 g_cond_broadcast(cond_have_key);
1523 g_mutex_unlock(last_key_mutex);
1524 }
1525#endif
1526
1527 return FALSE;
1528}

1561static gboolean icvOnMouse( GtkWidget *widget, GdkEvent *event, gpointer user_data )
1562{
1563 // TODO move this logic to CvImageWidget
1564 CvWindow* window = (CvWindow*)user_data;
1565 CvPoint2D32f pt32f = {-1., -1.};
1566 CvPoint pt = {-1,-1};
1567 int cv_event = -1, state = 0;
1568 CvImageWidget * image_widget = CV_IMAGE_WIDGET( widget );
1569
1570 if( window->signature != CV_WINDOW_MAGIC_VAL ||
1571 window->widget != widget || !window->widget ||
1572 !window->on_mouse /*|| !image_widget->original_image*/)
1573 return FALSE;
1574
1575 if( event->type == GDK_MOTION_NOTIFY )
1576 {
1577 GdkEventMotion* event_motion = (GdkEventMotion*)event;
1578
1579 cv_event = CV_EVENT_MOUSEMOVE;
1580 pt32f.x = cvRound(event_motion->x);
1581 pt32f.y = cvRound(event_motion->y);
1582 state = event_motion->state;
1583 }
1584 else if( event->type == GDK_BUTTON_PRESS ||
1585 event->type == GDK_BUTTON_RELEASE ||
1586 event->type == GDK_2BUTTON_PRESS )
1587 {
1588 GdkEventButton* event_button = (GdkEventButton*)event;
1589 pt32f.x = cvRound(event_button->x);
1590 pt32f.y = cvRound(event_button->y);
1591
1592
1593 if( event_button->type == GDK_BUTTON_PRESS )
1594 {
1595 cv_event = event_button->button == 1 ? CV_EVENT_LBUTTONDOWN :
1596 event_button->button == 2 ? CV_EVENT_MBUTTONDOWN :
1597 event_button->button == 3 ? CV_EVENT_RBUTTONDOWN : 0;
1598 }
1599 else if( event_button->type == GDK_BUTTON_RELEASE )
1600 {
1601 cv_event = event_button->button == 1 ? CV_EVENT_LBUTTONUP :
1602 event_button->button == 2 ? CV_EVENT_MBUTTONUP :
1603 event_button->button == 3 ? CV_EVENT_RBUTTONUP : 0;
1604 }
1605 else if( event_button->type == GDK_2BUTTON_PRESS )
1606 {
1607 cv_event = event_button->button == 1 ? CV_EVENT_LBUTTONDBLCLK :
1608 event_button->button == 2 ? CV_EVENT_MBUTTONDBLCLK :
1609 event_button->button == 3 ? CV_EVENT_RBUTTONDBLCLK : 0;
1610 }
1611 state = event_button->state;
1612 }
1613
1614 if( cv_event >= 0 ){
1615 // scale point if image is scaled
1616 if( (image_widget->flags & CV_WINDOW_AUTOSIZE)==0 &&
1617 image_widget->original_image &&
1618 image_widget->scaled_image ){
1619 // image origin is not necessarily at (0,0)
1620 int x0 = (widget->allocation.width - image_widget->scaled_image->cols)/2;
1621 int y0 = (widget->allocation.height - image_widget->scaled_image->rows)/2;
1622 pt.x = cvFloor( ((pt32f.x-x0)*image_widget->original_image->cols)/
1623 image_widget->scaled_image->cols );
1624 pt.y = cvFloor( ((pt32f.y-y0)*image_widget->original_image->rows)/
1625 image_widget->scaled_image->rows );
1626 }
1627 else{
1628 pt = cvPointFrom32f( pt32f );
1629 }
1630
1631// if((unsigned)pt.x < (unsigned)(image_widget->original_image->width) &&
1632// (unsigned)pt.y < (unsigned)(image_widget->original_image->height) )
1633 {
1634 int flags = (state & GDK_SHIFT_MASK ? CV_EVENT_FLAG_SHIFTKEY : 0) |
1635 (state & GDK_CONTROL_MASK ? CV_EVENT_FLAG_CTRLKEY : 0) |
1636 (state & (GDK_MOD1_MASK|GDK_MOD2_MASK) ? CV_EVENT_FLAG_ALTKEY : 0) |
1637 (state & GDK_BUTTON1_MASK ? CV_EVENT_FLAG_LBUTTON : 0) |
1638 (state & GDK_BUTTON2_MASK ? CV_EVENT_FLAG_MBUTTON : 0) |
1639 (state & GDK_BUTTON3_MASK ? CV_EVENT_FLAG_RBUTTON : 0);
1640 window->on_mouse( cv_event, pt.x, pt.y, flags, window->on_mouse_param );
1641 }
1642 }
1643
1644 return FALSE;
1645 }
