107/**
108 * Camera dispatching method: index is the camera number.
109 * If given an index from 0 to 99, it tries to find the first
110 * API that can access a given camera index.
111 * Add multiples of 100 to select an API.
112 */
113CV_IMPL CvCapture * cvCreateCameraCapture (int index)
114{
115 int domains[] =
116 {
117#ifdef HAVE_DSHOW
118 CV_CAP_DSHOW,
119#endif
120#ifdef HAVE_MSMF
121 CV_CAP_MSMF,
122#endif
123#if 1
124 CV_CAP_IEEE1394, // identical to CV_CAP_DC1394
125#endif
126#ifdef HAVE_TYZX
127 CV_CAP_STEREO,
128#endif
129#ifdef HAVE_PVAPI
130 CV_CAP_PVAPI,
131#endif
132#if 1
133 CV_CAP_VFW, // identical to CV_CAP_V4L
134#endif
135#ifdef HAVE_MIL
136 CV_CAP_MIL,
137#endif
138#if defined(HAVE_QUICKTIME) || defined(HAVE_QTKIT)
139 CV_CAP_QT,
140#endif
141#ifdef HAVE_UNICAP
142 CV_CAP_UNICAP,
143#endif
144#ifdef HAVE_OPENNI
145 CV_CAP_OPENNI,
146#endif
147#ifdef HAVE_ANDROID_NATIVE_CAMERA
148 CV_CAP_ANDROID,
149#endif
150#ifdef HAVE_XIMEA
151 CV_CAP_XIAPI,
152#endif
153#ifdef HAVE_AVFOUNDATION
154 CV_CAP_AVFOUNDATION,
155#endif
156#ifdef HAVE_GIGE_API
157 CV_CAP_GIGANETIX,
158#endif
159#ifdef HAVE_INTELPERC
160 CV_CAP_INTELPERC,
161#endif
162 -1
163 };
164
165 // interpret preferred interface (0 = autodetect)
166 int pref = (index / 100) * 100;
167 if (pref)
168 {
169 domains[0]=pref;
170 index %= 100;
171 domains[1]=-1;
172 }
173
174 // try every possibly installed camera API
175 for (int i = 0; domains[i] >= 0; i++)
176 {
177#if defined(HAVE_DSHOW) || \
178 defined(HAVE_MSMF) || \
179 defined(HAVE_TYZX) || \
180 defined(HAVE_VFW) || \
181 defined(HAVE_LIBV4L) || \
182 defined(HAVE_CAMV4L) || \
183 defined(HAVE_CAMV4L2) || \
184 defined(HAVE_VIDEOIO) || \
185 defined(HAVE_GSTREAMER) || \
186 defined(HAVE_DC1394_2) || \
187 defined(HAVE_DC1394) || \
188 defined(HAVE_CMU1394) || \
189 defined(HAVE_MIL) || \
190 defined(HAVE_QUICKTIME) || \
191 defined(HAVE_QTKIT) || \
192 defined(HAVE_UNICAP) || \
193 defined(HAVE_PVAPI) || \
194 defined(HAVE_OPENNI) || \
195 defined(HAVE_XIMEA) || \
196 defined(HAVE_AVFOUNDATION) || \
197 defined(HAVE_ANDROID_NATIVE_CAMERA) || \
198 defined(HAVE_GIGE_API) || \
199 defined(HAVE_INTELPERC) || \
200 (0)
201 // local variable to memorize the captured device
202 CvCapture *capture;
203#endif
204
205 switch (domains[i])
206 {
207#ifdef HAVE_DSHOW
208 case CV_CAP_DSHOW:
209 capture = cvCreateCameraCapture_DShow (index);
210 if (capture)
211 return capture;
212 break;
213#endif
214#ifdef HAVE_MSMF
215 case CV_CAP_MSMF:
216 capture = cvCreateCameraCapture_MSMF (index);
217 if (capture)
218 return capture;
219 break;
220#endif
221#ifdef HAVE_TYZX
222 case CV_CAP_STEREO:
223 capture = cvCreateCameraCapture_TYZX (index);
224 if (capture)
225 return capture;
226 break;
227#endif
228 case CV_CAP_VFW:
229#ifdef HAVE_VFW
230 capture = cvCreateCameraCapture_VFW (index);
231 if (capture)
232 return capture;
233#endif
234#if defined HAVE_LIBV4L || defined HAVE_CAMV4L || defined HAVE_CAMV4L2 || defined HAVE_VIDEOIO
235 capture = cvCreateCameraCapture_V4L (index);
236 if (capture)
237 return capture;
238#endif
239
240#ifdef HAVE_GSTREAMER
241 capture = cvCreateCapture_GStreamer(CV_CAP_GSTREAMER_V4L2,
242 reinterpret_cast<char *>(index));
243 if (capture)
244 return capture;
245 capture = cvCreateCapture_GStreamer(CV_CAP_GSTREAMER_V4L,
246 reinterpret_cast<char *>(index));
247 if (capture)
248 return capture;
249#endif
250 break; //CV_CAP_VFW
251
252 case CV_CAP_FIREWIRE:
253#ifdef HAVE_DC1394_2
254 capture = cvCreateCameraCapture_DC1394_2 (index);
255 if (capture)
256 return capture;
257#endif
258
259#ifdef HAVE_DC1394
260 capture = cvCreateCameraCapture_DC1394 (index);
261 if (capture)
262 return capture;
263#endif
264
265#ifdef HAVE_CMU1394
266 capture = cvCreateCameraCapture_CMU (index);
267 if (capture)
268 return capture;
269#endif
270
271#if defined(HAVE_GSTREAMER) && 0
272 //Re-enable again when gstreamer 1394 support will land in the backend code
273 capture = cvCreateCapture_GStreamer(CV_CAP_GSTREAMER_1394, 0);
274 if (capture)
275 return capture;
276#endif
277 break; //CV_CAP_FIREWIRE
278
279#ifdef HAVE_MIL
280 case CV_CAP_MIL:
281 capture = cvCreateCameraCapture_MIL (index);
282 if (capture)
283 return capture;
284 break;
285#endif
286
287#if defined(HAVE_QUICKTIME) || defined(HAVE_QTKIT)
288 case CV_CAP_QT:
289 capture = cvCreateCameraCapture_QT (index);
290 if (capture)
291 return capture;
292 break;
293#endif
294
295#ifdef HAVE_UNICAP
296 case CV_CAP_UNICAP:
297 capture = cvCreateCameraCapture_Unicap (index);
298 if (capture)
299 return capture;
300 break;
301#endif
302
303#ifdef HAVE_PVAPI
304 case CV_CAP_PVAPI:
305 capture = cvCreateCameraCapture_PvAPI (index);
306 if (capture)
307 return capture;
308 break;
309#endif
310
311#ifdef HAVE_OPENNI
312 case CV_CAP_OPENNI:
313 capture = cvCreateCameraCapture_OpenNI (index);
314 if (capture)
315 return capture;
316 break;
317#endif
318
319#ifdef HAVE_ANDROID_NATIVE_CAMERA
320 case CV_CAP_ANDROID:
321 capture = cvCreateCameraCapture_Android (index);
322 if (capture)
323 return capture;
324 break;
325#endif
326
327#ifdef HAVE_XIMEA
328 case CV_CAP_XIAPI:
329 capture = cvCreateCameraCapture_XIMEA (index);
330 if (capture)
331 return capture;
332 break;
333#endif
334
335#ifdef HAVE_AVFOUNDATION
336 case CV_CAP_AVFOUNDATION:
337 capture = cvCreateCameraCapture_AVFoundation (index);
338 if (capture)
339 return capture;
340 break;
341#endif
342
343#ifdef HAVE_GIGE_API
344 case CV_CAP_GIGANETIX:
345 capture = cvCreateCameraCapture_Giganetix (index);
346 if (capture)
347 return capture;
348 break; // CV_CAP_GIGANETIX
349#endif
350
351#ifdef HAVE_INTELPERC
352 case CV_CAP_INTELPERC:
353 capture = cvCreateCameraCapture_IntelPerC(index);
354 if (capture)
355 return capture;
356 break; // CV_CAP_INTEL_PERC
357#endif
358 }
359 }
360
361 // failed open a camera
362 return 0;
363}

365/**
366 * Videoreader dispatching method: it tries to find the first
367 * API that can access a given filename.
368 */
369CV_IMPL CvCapture * cvCreateFileCapture (const char * filename)
370{
371 CvCapture * result = 0;
372
373#ifdef HAVE_FFMPEG
374 if (! result)
375 result = cvCreateFileCapture_FFMPEG_proxy (filename);
376#endif
377
378#ifdef HAVE_VFW
379 if (! result)
380 result = cvCreateFileCapture_VFW (filename);
381#endif
382
383#ifdef HAVE_MSMF
384 if (! result)
385 result = cvCreateFileCapture_MSMF (filename);
386#endif
387
388#ifdef HAVE_XINE
389 if (! result)
390 result = cvCreateFileCapture_XINE (filename);
391#endif
392
393#ifdef HAVE_GSTREAMER
394 if (! result)
395 result = cvCreateCapture_GStreamer (CV_CAP_GSTREAMER_FILE, filename);
396#endif
397
398#if defined(HAVE_QUICKTIME) || defined(HAVE_QTKIT)
399 if (! result)
400 result = cvCreateFileCapture_QT (filename);
401#endif
402
403#ifdef HAVE_AVFOUNDATION
404 if (! result)
405 result = cvCreateFileCapture_AVFoundation (filename);
406#endif
407
408#ifdef HAVE_OPENNI
409 if (! result)
410 result = cvCreateFileCapture_OpenNI (filename);
411#endif
412
413 if (! result)
414 result = cvCreateFileCapture_Images (filename);
415
416 return result;
417}

419/**
420 * Videowriter dispatching method: it tries to find the first
421 * API that can write a given stream.
422 */
423CV_IMPL CvVideoWriter* cvCreateVideoWriter( const char* filename, int fourcc,
424 double fps, CvSize frameSize, int is_color )
425{
426 //CV_FUNCNAME( "cvCreateVideoWriter" );
427
428 CvVideoWriter *result = 0;
429
430 if(!fourcc || !fps)
431 result = cvCreateVideoWriter_Images(filename);
432
433#ifdef HAVE_FFMPEG
434 if(!result)
435 result = cvCreateVideoWriter_FFMPEG_proxy (filename, fourcc, fps, frameSize, is_color);
436#endif
437
438#ifdef HAVE_VFW
439 if(!result)
440 result = cvCreateVideoWriter_VFW(filename, fourcc, fps, frameSize, is_color);
441#endif
442
443#ifdef HAVE_MSMF
444 if (!result)
445 result = cvCreateVideoWriter_MSMF(filename, fourcc, fps, frameSize, is_color);
446#endif
447
448/* #ifdef HAVE_XINE
449 if(!result)
450 result = cvCreateVideoWriter_XINE(filename, fourcc, fps, frameSize, is_color);
451 #endif
452*/
453#ifdef HAVE_AVFOUNDATION
454 if (! result)
455 result = cvCreateVideoWriter_AVFoundation(filename, fourcc, fps, frameSize, is_color);
456#endif
457
458#if defined(HAVE_QUICKTIME) || defined(HAVE_QTKIT)
459 if(!result)
460 result = cvCreateVideoWriter_QT(filename, fourcc, fps, frameSize, is_color);
461#endif
462
463#ifdef HAVE_GSTREAMER
464 if (! result)
465 result = cvCreateVideoWriter_GStreamer(filename, fourcc, fps, frameSize, is_color);
466#endif
467
468#if !defined(HAVE_FFMPEG) && \
469 !defined(HAVE_VFW) && \
470 !defined(HAVE_MSMF) && \
471 !defined(HAVE_AVFOUNDATION) && \
472 !defined(HAVE_QUICKTIME) && \
473 !defined(HAVE_QTKIT) && \
474 !defined(HAVE_GSTREAMER)
475// If none of the writers is used
476// these statements suppress 'unused parameter' warnings.
477 (void)frameSize;
478 (void)is_color;
479#endif
480
481 if(!result)
482 result = cvCreateVideoWriter_Images(filename);
483
484 return result;
485}

548bool VideoCapture::retrieve(Mat& image, int channel)
549{
550 IplImage* _img = cvRetrieveFrame(cap, channel);
551 if( !_img )
552 {
553 image.release();
554 return false;
555 }
556 if(_img->origin == IPL_ORIGIN_TL)
557 Mat(_img).copyTo(image);
558 else
559 {
560 Mat temp(_img);
561 flip(temp, image, 0);
562 }
563 return true;
564}
565
566bool VideoCapture::read(Mat& image)
567{
568 if(grab())
569 retrieve(image);
570 else
571 image.release();
572 return !image.empty();
573}

576{
577 read(image);
578 return *this;
