109bool ExrDecoder::readHeader()
110{
111 bool result = false;
112
113 m_file = new InputFile( m_filename.c_str() );
114
115 if( !m_file ) // probably paranoid
116 return false;
117
118 m_datawindow = m_file->header().dataWindow();
119 m_width = m_datawindow.max.x - m_datawindow.min.x + 1;
120 m_height = m_datawindow.max.y - m_datawindow.min.y + 1;
121
122 // the type HALF is converted to 32 bit float
123 // and the other types supported by OpenEXR are 32 bit anyway
124 m_bit_depth = 32;
125
126 if( hasChromaticities( m_file->header() ))
127 m_chroma = chromaticities( m_file->header() );
128
129 const ChannelList &channels = m_file->header().channels();
130 m_red = channels.findChannel( "R" );
131 m_green = channels.findChannel( "G" );
132 m_blue = channels.findChannel( "B" );
133 if( m_red || m_green || m_blue )
134 {
135 m_iscolor = true;
136 m_ischroma = false;
137 result = true;
138 }
139 else
140 {
141 m_green = channels.findChannel( "Y" );
142 if( m_green )
143 {
144 m_ischroma = true;
145 m_red = channels.findChannel( "RY" );
146 m_blue = channels.findChannel( "BY" );
147 m_iscolor = (m_blue || m_red);
148 result = true;
149 }
150 else
151 result = false;
152 }
153
154 if( result )
155 {
156 int uintcnt = 0;
157 int chcnt = 0;
158 if( m_red )
159 {
160 chcnt++;
161 uintcnt += ( m_red->type == UINT );
162 }
163 if( m_green )
164 {
165 chcnt++;
166 uintcnt += ( m_green->type == UINT );
167 }
168 if( m_blue )
169 {
170 chcnt++;
171 uintcnt += ( m_blue->type == UINT );
172 }
173 m_type = (chcnt == uintcnt) ? UINT : FLOAT;
174
175 m_isfloat = (m_type == FLOAT);
176 }
177
178 if( !result )
179 close();
180
181 return result;
182}

185bool ExrDecoder::readData( Mat& img )
186{
187 m_native_depth = CV_MAT_DEPTH(type()) == img.depth();
188 bool color = img.channels() > 1;
189
190 uchar* data = img.data;
191 int step = img.step;
192 bool justcopy = m_native_depth;
193 bool chromatorgb = false;
194 bool rgbtogray = false;
195 bool result = true;
196 FrameBuffer frame;
197 int xsample[3] = {1, 1, 1};
198 char *buffer;
199 int xstep;
200 int ystep;
201
202 xstep = m_native_depth ? 4 : 1;
203
204 if( !m_native_depth || (!color && m_iscolor ))
205 {
206 buffer = (char *)new float[ m_width * 3 ];
207 ystep = 0;
208 }
209 else
210 {
211 buffer = (char *)data;
212 ystep = step;
213 }
214
215 if( m_ischroma )
216 {
217 if( color )
218 {
219 if( m_iscolor )
220 {
221 if( m_blue )
222 {
223 frame.insert( "BY", Slice( m_type,
224 buffer - m_datawindow.min.x * 12 - m_datawindow.min.y * ystep,
225 12, ystep, m_blue->xSampling, m_blue->ySampling, 0.0 ));
226 xsample[0] = m_blue->ySampling;
227 }
228 if( m_green )
229 {
230 frame.insert( "Y", Slice( m_type,
231 buffer - m_datawindow.min.x * 12 - m_datawindow.min.y * ystep + 4,
232 12, ystep, m_green->xSampling, m_green->ySampling, 0.0 ));
233 xsample[1] = m_green->ySampling;
234 }
235 if( m_red )
236 {
237 frame.insert( "RY", Slice( m_type,
238 buffer - m_datawindow.min.x * 12 - m_datawindow.min.y * ystep + 8,
239 12, ystep, m_red->xSampling, m_red->ySampling, 0.0 ));
240 xsample[2] = m_red->ySampling;
241 }
242 chromatorgb = true;
243 }
244 else
245 {
246 frame.insert( "Y", Slice( m_type,
247 buffer - m_datawindow.min.x * 12 - m_datawindow.min.y * ystep,
248 12, ystep, m_green->xSampling, m_green->ySampling, 0.0 ));
249 frame.insert( "Y", Slice( m_type,
250 buffer - m_datawindow.min.x * 12 - m_datawindow.min.y * ystep + 4,
251 12, ystep, m_green->xSampling, m_green->ySampling, 0.0 ));
252 frame.insert( "Y", Slice( m_type,
253 buffer - m_datawindow.min.x * 12 - m_datawindow.min.y * ystep + 8,
254 12, ystep, m_green->xSampling, m_green->ySampling, 0.0 ));
255 xsample[0] = m_green->ySampling;
256 xsample[1] = m_green->ySampling;
257 xsample[2] = m_green->ySampling;
258 }
259 }
260 else
261 {
262 frame.insert( "Y", Slice( m_type,
263 buffer - m_datawindow.min.x * 4 - m_datawindow.min.y * ystep,
264 4, ystep, m_green->xSampling, m_green->ySampling, 0.0 ));
265 xsample[0] = m_green->ySampling;
266 }
267 }
268 else
269 {
270 if( m_blue )
271 {
272 frame.insert( "B", Slice( m_type,
273 buffer - m_datawindow.min.x * 12 - m_datawindow.min.y * ystep,
274 12, ystep, m_blue->xSampling, m_blue->ySampling, 0.0 ));
275 xsample[0] = m_blue->ySampling;
276 }
277 if( m_green )
278 {
279 frame.insert( "G", Slice( m_type,
280 buffer - m_datawindow.min.x * 12 - m_datawindow.min.y * ystep + 4,
281 12, ystep, m_green->xSampling, m_green->ySampling, 0.0 ));
282 xsample[1] = m_green->ySampling;
283 }
284 if( m_red )
285 {
286 frame.insert( "R", Slice( m_type,
287 buffer - m_datawindow.min.x * 12 - m_datawindow.min.y * ystep + 8,
288 12, ystep, m_red->xSampling, m_red->ySampling, 0.0 ));
289 xsample[2] = m_red->ySampling;
290 }
291 if(color == 0)
292 {
293 rgbtogray = true;
294 justcopy = false;
295 }
296 }
297
298 m_file->setFrameBuffer( frame );
299 if( justcopy )
300 {
301 m_file->readPixels( m_datawindow.min.y, m_datawindow.max.y );
302
303 if( color )
304 {
305 if( m_blue && (m_blue->xSampling != 1 || m_blue->ySampling != 1) )
306 UpSample( data, 3, step / xstep, xsample[0], m_blue->ySampling );
307 if( m_green && (m_green->xSampling != 1 || m_green->ySampling != 1) )
308 UpSample( data + xstep, 3, step / xstep, xsample[1], m_green->ySampling );
309 if( m_red && (m_red->xSampling != 1 || m_red->ySampling != 1) )
310 UpSample( data + 2 * xstep, 3, step / xstep, xsample[2], m_red->ySampling );
311 }
312 else if( m_green && (m_green->xSampling != 1 || m_green->ySampling != 1) )
313 UpSample( data, 1, step / xstep, xsample[0], m_green->ySampling );
314 }
315 else
316 {
317 uchar *out = data;
318 int x, y;
319 for( y = m_datawindow.min.y; y <= m_datawindow.max.y; y++ )
320 {
321 m_file->readPixels( y, y );
322
323 if( rgbtogray )
324 {
325 if( xsample[0] != 1 )
326 UpSampleX( (float *)buffer, 3, xsample[0] );
327 if( xsample[1] != 1 )
328 UpSampleX( (float *)buffer + 4, 3, xsample[1] );
329 if( xsample[2] != 1 )
330 UpSampleX( (float *)buffer + 8, 3, xsample[2] );
331
332 RGBToGray( (float *)buffer, (float *)out );
333 }
334 else
335 {
336 if( xsample[0] != 1 )
337 UpSampleX( (float *)buffer, 3, xsample[0] );
338 if( xsample[1] != 1 )
339 UpSampleX( (float *)(buffer + 4), 3, xsample[1] );
340 if( xsample[2] != 1 )
341 UpSampleX( (float *)(buffer + 8), 3, xsample[2] );
342
343 if( chromatorgb )
344 ChromaToBGR( (float *)buffer, 1, step );
345
346 if( m_type == FLOAT )
347 {
348 float *fi = (float *)buffer;
349 for( x = 0; x < m_width * 3; x++)
350 {
351 int t = cvRound(fi[x]*5);
352 out[x] = CV_CAST_8U(t);
353 }
354 }
355 else
356 {
357 unsigned *ui = (unsigned *)buffer;
358 for( x = 0; x < m_width * 3; x++)
359 {
360 unsigned t = ui[x];
361 out[x] = CV_CAST_8U(t);
362 }
363 }
364 }
365
366 out += step;
367 }
368 if( color )
369 {
370 if( m_blue && (m_blue->xSampling != 1 || m_blue->ySampling != 1) )
371 UpSampleY( data, 3, step / xstep, m_blue->ySampling );
372 if( m_green && (m_green->xSampling != 1 || m_green->ySampling != 1) )
373 UpSampleY( data + xstep, 3, step / xstep, m_green->ySampling );
374 if( m_red && (m_red->xSampling != 1 || m_red->ySampling != 1) )
375 UpSampleY( data + 2 * xstep, 3, step / xstep, m_red->ySampling );
376 }
377 else if( m_green && (m_green->xSampling != 1 || m_green->ySampling != 1) )
378 UpSampleY( data, 1, step / xstep, m_green->ySampling );
379 }
380
381 if( chromatorgb )
382 ChromaToBGR( (float *)data, m_height, step / xstep );
383
384 close();
385
386 return result;
387}

389/**
390// on entry pixel values are stored packed in the upper left corner of the image
391// this functions expands them by duplication to cover the whole image
392 */
393void ExrDecoder::UpSample( uchar *data, int xstep, int ystep, int xsample, int ysample )
394{
395 for( int y = (m_height - 1) / ysample, yre = m_height - ysample; y >= 0; y--, yre -= ysample )
396 {
397 for( int x = (m_width - 1) / xsample, xre = m_width - xsample; x >= 0; x--, xre -= xsample )
398 {
399 for( int i = 0; i < ysample; i++ )
400 {
401 for( int n = 0; n < xsample; n++ )
402 {
403 if( !m_native_depth )
404 data[(yre + i) * ystep + (xre + n) * xstep] = data[y * ystep + x * xstep];
405 else if( m_type == FLOAT )
406 ((float *)data)[(yre + i) * ystep + (xre + n) * xstep] = ((float *)data)[y * ystep + x * xstep];
407 else
408 ((unsigned *)data)[(yre + i) * ystep + (xre + n) * xstep] = ((unsigned *)data)[y * ystep + x * xstep];
409 }
410 }
411 }
412 }
413}

415/**
416// on entry pixel values are stored packed in the upper left corner of the image
417// this functions expands them by duplication to cover the whole image
418 */
419void ExrDecoder::UpSampleX( float *data, int xstep, int xsample )
420{
421 for( int x = (m_width - 1) / xsample, xre = m_width - xsample; x >= 0; x--, xre -= xsample )
422 {
423 for( int n = 0; n < xsample; n++ )
424 {
425 if( m_type == FLOAT )
426 ((float *)data)[(xre + n) * xstep] = ((float *)data)[x * xstep];
427 else
428 ((unsigned *)data)[(xre + n) * xstep] = ((unsigned *)data)[x * xstep];
429 }
430 }

437void ExrDecoder::UpSampleY( uchar *data, int xstep, int ystep, int ysample )

445 if( !m_native_depth )
446 data[(yre + i) * ystep + x * xstep] = data[y * ystep + x * xstep];

449 else
450 ((unsigned *)data)[(yre + i) * ystep + x * xstep] = ((unsigned *)data)[y * ystep + x * xstep];
451 }

456/**
457// algorithm from ImfRgbaYca.cpp
458 */
459void ExrDecoder::ChromaToBGR( float *data, int numlines, int step )
460{
461 for( int y = 0; y < numlines; y++ )
462 {
463 for( int x = 0; x < m_width; x++ )
464 {
465 double b, Y, r;
466 if( !m_native_depth )
467 {
468 b = ((uchar *)data)[y * step + x * 3];
469 Y = ((uchar *)data)[y * step + x * 3 + 1];
470 r = ((uchar *)data)[y * step + x * 3 + 2];
471 }
472 else if( m_type == FLOAT )
473 {
474 b = data[y * step + x * 3];
475 Y = data[y * step + x * 3 + 1];
476 r = data[y * step + x * 3 + 2];
477 }
478 else
479 {
480 b = ((unsigned *)data)[y * step + x * 3];
481 Y = ((unsigned *)data)[y * step + x * 3 + 1];
482 r = ((unsigned *)data)[y * step + x * 3 + 2];
483 }
484 r = (r + 1) * Y;
485 b = (b + 1) * Y;
486 Y = (Y - b * m_chroma.blue[1] - r * m_chroma.red[1]) / m_chroma.green[1];
487
488 if( !m_native_depth )
489 {
490 int t = cvRound(b);
491 ((uchar *)data)[y * step + x * 3] = CV_CAST_8U(t);
492 t = cvRound(Y);
493 ((uchar *)data)[y * step + x * 3 + 1] = CV_CAST_8U(t);
494 t = cvRound(r);
495 ((uchar *)data)[y * step + x * 3 + 2] = CV_CAST_8U(t);
496 }
497 else if( m_type == FLOAT )
498 {
499 data[y * step + x * 3] = (float)b;
500 data[y * step + x * 3 + 1] = (float)Y;
501 data[y * step + x * 3 + 2] = (float)r;
502 }
503 else
504 {
505 int t = cvRound(b);
506 ((unsigned *)data)[y * step + x * 3] = (unsigned)MAX(t,0);
507 t = cvRound(Y);
508 ((unsigned *)data)[y * step + x * 3 + 1] = (unsigned)MAX(t,0);
509 t = cvRound(r);
510 ((unsigned *)data)[y * step + x * 3 + 2] = (unsigned)MAX(t,0);
511 }
512 }
513 }

517/**
518// convert one row to gray
519*/
520void ExrDecoder::RGBToGray( float *in, float *out )
521{
522 if( m_type == FLOAT )
523 {
524 if( m_native_depth )
525 {
526 for( int i = 0, n = 0; i < m_width; i++, n += 3 )
527 out[i] = in[n] * m_chroma.blue[0] + in[n + 1] * m_chroma.green[0] + in[n + 2] * m_chroma.red[0];
528 }
529 else
530 {
531 uchar *o = (uchar *)out;
532 for( int i = 0, n = 0; i < m_width; i++, n += 3 )
533 o[i] = (uchar) (in[n] * m_chroma.blue[0] + in[n + 1] * m_chroma.green[0] + in[n + 2] * m_chroma.red[0]);
534 }
535 }
536 else // UINT
537 {
538 if( m_native_depth )
539 {
540 unsigned *ui = (unsigned *)in;
541 for( int i = 0; i < m_width * 3; i++ )
542 ui[i] -= 0x80000000;
543 int *si = (int *)in;
544 for( int i = 0, n = 0; i < m_width; i++, n += 3 )
545 ((int *)out)[i] = int(si[n] * m_chroma.blue[0] + si[n + 1] * m_chroma.green[0] + si[n + 2] * m_chroma.red[0]);
546 }
547 else // how to best convert float to uchar?
548 {
549 unsigned *ui = (unsigned *)in;
550 for( int i = 0, n = 0; i < m_width; i++, n += 3 )
551 ((uchar *)out)[i] = uchar((ui[n] * m_chroma.blue[0] + ui[n + 1] * m_chroma.green[0] + ui[n + 2] * m_chroma.red[0]) * (256.0 / 4294967296.0));
552 }
553 }

582// TODO scale appropriately
583bool ExrEncoder::write( const Mat& img, const vector<int>& )
584{
585 int width = img.cols, height = img.rows;
586 int depth = img.depth(), channels = img.channels();
587 bool result = false;
588 bool issigned = depth == CV_8S || depth == CV_16S || depth == CV_32S;
589 bool isfloat = depth == CV_32F || depth == CV_64F;
590 depth = CV_ELEM_SIZE1(depth)*8;
591 uchar* data = img.data;
592 int step = img.step;
593
594 Header header( width, height );
595 Imf::PixelType type;
596
597 if(depth == 8)
598 type = HALF;
599 else if(isfloat)
600 type = FLOAT;
601 else
602 type = UINT;
603
604 if( channels == 3 )
605 {
606 header.channels().insert( "R", Channel( type ));
607 header.channels().insert( "G", Channel( type ));
608 header.channels().insert( "B", Channel( type ));
609 //printf("bunt\n");
610 }
611 else
612 {
613 header.channels().insert( "Y", Channel( type ));
614 //printf("gray\n");
615 }
616
617 OutputFile file( m_filename.c_str(), header );
618
619 FrameBuffer frame;
620
621 char *buffer;
622 int bufferstep;
623 int size;
624 if( type == FLOAT && depth == 32 )
625 {
626 buffer = (char *)const_cast<uchar *>(data);
627 bufferstep = step;
628 size = 4;
629 }
630 else if( depth > 16 || type == UINT )
631 {
632 buffer = (char *)new unsigned[width * channels];
633 bufferstep = 0;
634 size = 4;
635 }
636 else
637 {
638 buffer = (char *)new half[width * channels];
639 bufferstep = 0;
640 size = 2;
641 }
642
643 //printf("depth %d %s\n", depth, types[type]);
644
645 if( channels == 3 )
646 {
647 frame.insert( "B", Slice( type, buffer, size * 3, bufferstep ));
648 frame.insert( "G", Slice( type, buffer + size, size * 3, bufferstep ));
649 frame.insert( "R", Slice( type, buffer + size * 2, size * 3, bufferstep ));
650 }
651 else
652 frame.insert( "Y", Slice( type, buffer, size, bufferstep ));
653
654 file.setFrameBuffer( frame );
655
656 int offset = issigned ? 1 << (depth - 1) : 0;
657
658 result = true;
659 if( type == FLOAT && depth == 32 )
660 {
661 try
662 {
663 file.writePixels( height );
664 }
665 catch(...)
666 {
667 result = false;
668 }
669 }
670 else
671 {
672 // int scale = 1 << (32 - depth);
673 // printf("scale %d\n", scale);
674 for(int line = 0; line < height; line++)
675 {
676 if(type == UINT)
677 {
678 unsigned *buf = (unsigned*)buffer; // FIXME 64-bit problems
679
680 if( depth <= 8 )
681 {
682 for(int i = 0; i < width * channels; i++)
683 buf[i] = data[i] + offset;
684 }
685 else if( depth <= 16 )
686 {
687 unsigned short *sd = (unsigned short *)data;
688 for(int i = 0; i < width * channels; i++)
689 buf[i] = sd[i] + offset;
690 }
691 else
692 {
693 int *sd = (int *)data; // FIXME 64-bit problems
694 for(int i = 0; i < width * channels; i++)
695 buf[i] = (unsigned) sd[i] + offset;
696 }
697 }
698 else
699 {
700 half *buf = (half *)buffer;
701
702 if( depth <= 8 )
703 {
704 for(int i = 0; i < width * channels; i++)
705 buf[i] = data[i];
706 }
707 else if( depth <= 16 )
708 {
709 unsigned short *sd = (unsigned short *)data;
710 for(int i = 0; i < width * channels; i++)
711 buf[i] = sd[i];
712 }
713 }
714 try
715 {
716 file.writePixels( 1 );
717 }
718 catch(...)
719 {
720 result = false;
721 break;
722 }
