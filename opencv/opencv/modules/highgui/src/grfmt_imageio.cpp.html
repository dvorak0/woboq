54bool ImageIODecoder::readHeader()
55{
56 CFURLRef imageURLRef;
57 CGImageSourceRef sourceRef;
58 // diciu, if ReadHeader is called twice in a row make sure to release the previously allocated imageRef
59 if (imageRef != NULL)
60 CGImageRelease(imageRef);
61 imageRef = NULL;
62
63 imageURLRef = CFURLCreateFromFileSystemRepresentation( NULL,
64 (const UInt8*)m_filename.c_str(), m_filename.size(), false );
65
66 sourceRef = CGImageSourceCreateWithURL( imageURLRef, NULL );
67 CFRelease( imageURLRef );
68 if ( !sourceRef )
69 return false;
70
71 imageRef = CGImageSourceCreateImageAtIndex( sourceRef, 0, NULL );
72 CFRelease( sourceRef );
73 if( !imageRef )
74 return false;
75
76 m_width = CGImageGetWidth( imageRef );
77 m_height = CGImageGetHeight( imageRef );
78
79 CGColorSpaceRef colorSpace = CGImageGetColorSpace( imageRef );
80 if( !colorSpace )
81 return false;
82
83 m_type = CGColorSpaceGetNumberOfComponents( colorSpace ) > 1 ? CV_8UC3 : CV_8UC1;
84
85 return true;
86}

89bool ImageIODecoder::readData( Mat& img )
90{
91 uchar* data = img.data;
92 int step = img.step;
93 bool color = img.channels() > 1;
94 int bpp; // Bytes per pixel
95 int bit_depth = 8;
96
97 // Get Height, Width, and color information
98 if( !readHeader() )
99 return false;
100
101 CGContextRef context = NULL; // The bitmap context
102 CGColorSpaceRef colorSpace = NULL;
103 uchar* bitmap = NULL;
104 CGImageAlphaInfo alphaInfo;
105
106 // CoreGraphics will take care of converting to grayscale and back as long as the
107 // appropriate colorspace is set
108 if( color == CV_LOAD_IMAGE_GRAYSCALE )
109 {
110 colorSpace = CGColorSpaceCreateDeviceGray();
111 bpp = 1;
112 alphaInfo = kCGImageAlphaNone;
113 }
114 else if( color == CV_LOAD_IMAGE_COLOR )
115 {
116#if TARGET_OS_IPHONE || TARGET_IPHONE_SIMULATOR
117 colorSpace = CGColorSpaceCreateDeviceRGB();
118#else
119 colorSpace = CGColorSpaceCreateWithName( kCGColorSpaceGenericRGBLinear );
120#endif
121 bpp = 4; /* CG only has 8 and 32 bit color spaces, so we waste a byte */
122 alphaInfo = kCGImageAlphaNoneSkipLast;
123 }
124 if( !colorSpace )
125 return false;
126
127 bitmap = (uchar*)malloc( bpp * m_height * m_width );
128 if( !bitmap )
129 {
130 CGColorSpaceRelease( colorSpace );
131 return false;
132 }
133
134 context = CGBitmapContextCreate( (void *)bitmap,
135 m_width, /* width */
136 m_height, /* height */
137 bit_depth, /* bit depth */
138 bpp * m_width, /* bytes per row */
139 colorSpace, /* color space */
140 alphaInfo);
141
142 CGColorSpaceRelease( colorSpace );
143 if( !context )
144 {
145 free( bitmap );
146 return false;
147 }
148
149 // Copy the image data into the bitmap region
150 CGRect rect = {{0,0},{m_width,m_height}};
151 CGContextDrawImage( context, rect, imageRef );
152
153 uchar* bitdata = (uchar*)CGBitmapContextGetData( context );
154 if( !bitdata )
155 {
156 free( bitmap);
157 CGContextRelease( context );
158 return false;
159 }
160
161 // Move the bitmap (in RGB) into data (in BGR)
162 int bitmapIndex = 0;
163
164 if( color == CV_LOAD_IMAGE_COLOR )
165 {
166 uchar * base = data;
167
168 for (int y = 0; y < m_height; y++)
169 {
170 uchar * line = base + y * step;
171
172 for (int x = 0; x < m_width; x++)
173 {
174 // Blue channel
175 line[0] = bitdata[bitmapIndex + 2];
176 // Green channel
177 line[1] = bitdata[bitmapIndex + 1];
178 // Red channel
179 line[2] = bitdata[bitmapIndex + 0];
180
181 line += 3;
182 bitmapIndex += bpp;
183 }
184 }
185 }
186 else if( color == CV_LOAD_IMAGE_GRAYSCALE )
187 {
188 for (int y = 0; y < m_height; y++)
189 memcpy (data + y * step, bitmap + y * m_width, m_width);
190 }
191
192 free( bitmap );
193 CGContextRelease( context );
194 return true;
195}

216static
217CFStringRef FilenameToUTI( const char* filename )
218{
219 const char* ext = filename;
220 char* ext_buf;
221 int i;
222 CFStringRef imageUTI = NULL;
223
224 for(;;)
225 {
226 const char* temp = strchr( ext + 1, '.' );
227 if( !temp ) break;
228 ext = temp;
229 }
230
231 if(!ext)
232 return NULL;
233
234 ext_buf = (char*)malloc(strlen(ext)+1);
235 for(i = 0; ext[i] != '\0'; i++)
236 ext_buf[i] = (char)tolower(ext[i]);
237 ext_buf[i] = '\0';
238 ext = ext_buf;
239
240 if( !strcmp(ext, ".bmp") || !strcmp(ext, ".dib") )
241 imageUTI = CFSTR( "com.microsoft.bmp" );
242 else if( !strcmp(ext, ".exr") )
243 imageUTI = CFSTR( "com.ilm.openexr-image" );
244 else if( !strcmp(ext, ".jpeg") || !strcmp(ext, ".jpg") || !strcmp(ext, ".jpe") )
245 imageUTI = CFSTR( "public.jpeg" );
246 else if( !strcmp(ext, ".jp2") )
247 imageUTI = CFSTR( "public.jpeg-2000" );
248 else if( !strcmp(ext, ".pdf") )
249 imageUTI = CFSTR( "com.adobe.pdf" );
250 else if( !strcmp(ext, ".png") )
251 imageUTI = CFSTR( "public.png" );
252 else if( !strcmp(ext, ".tiff") || !strcmp(ext, ".tif") )
253 imageUTI = CFSTR( "public.tiff" );
254
255 free(ext_buf);
256
257 return imageUTI;
258}

261bool ImageIOEncoder::write( const Mat& img, const vector<int>& params )
262{
263 int width = img.cols, height = img.rows;
264 int _channels = img.channels();
265 const uchar* data = img.data;
266 int step = img.step;
267
268 // Determine the appropriate UTI based on the filename extension
269 CFStringRef imageUTI = FilenameToUTI( m_filename.c_str() );
270
271 // Determine the Bytes Per Pixel
272 int bpp = (_channels == 1) ? 1 : 4;
273
274 // Write the data into a bitmap context
275 CGContextRef context;
276 CGColorSpaceRef colorSpace;
277 uchar* bitmapData = NULL;
278
279 if( bpp == 1 ) {
280#if TARGET_OS_IPHONE || TARGET_IPHONE_SIMULATOR
281 colorSpace = CGColorSpaceCreateDeviceGray();
282#else
283 colorSpace = CGColorSpaceCreateWithName( kCGColorSpaceGenericGray );
284#endif
285 }
286 else if( bpp == 4 ) {
287#if TARGET_OS_IPHONE || TARGET_IPHONE_SIMULATOR
288 colorSpace = CGColorSpaceCreateDeviceRGB();
289#else
290 colorSpace = CGColorSpaceCreateWithName( kCGColorSpaceGenericRGBLinear );
291#endif
292 }
293 if( !colorSpace )
294 return false;
295
296 bitmapData = (uchar*)malloc( bpp * height * width );
297 if( !bitmapData )
298 {
299 CGColorSpaceRelease( colorSpace );
300 return false;
301 }
302
303 context = CGBitmapContextCreate( bitmapData,
304 width,
305 height,
306 8,
307 bpp * width,
308 colorSpace,
309 (bpp == 1) ? kCGImageAlphaNone :
310 kCGImageAlphaNoneSkipLast );
311 CGColorSpaceRelease( colorSpace );
312 if( !context )
313 {
314 free( bitmapData );
315 return false;
316 }
317
318 // Copy pixel information from data into bitmapData
319 if (bpp == 4)
320 {
321 int bitmapIndex = 0;
322 const uchar * base = data;
323
324 for (int y = 0; y < height; y++)
325 {
326 const uchar * line = base + y * step;
327
328 for (int x = 0; x < width; x++)
329 {
330 // Blue channel
331 bitmapData[bitmapIndex + 2] = line[0];
332 // Green channel
333 bitmapData[bitmapIndex + 1] = line[1];
334 // Red channel
335 bitmapData[bitmapIndex + 0] = line[2];
336
337 line += 3;
338 bitmapIndex += bpp;
339 }
340 }
341 }
342 else if (bpp == 1)
343 {
344 for (int y = 0; y < height; y++)
345 memcpy (bitmapData + y * width, data + y * step, width);
346 }
347
348 // Turn the bitmap context into an imageRef
349 CGImageRef imageRef = CGBitmapContextCreateImage( context );
350 CGContextRelease( context );
351 if( !imageRef )
352 {
353 free( bitmapData );
354 return false;
355 }
356
357 // Write the imageRef to a file based on the UTI
358 CFURLRef imageURLRef = CFURLCreateFromFileSystemRepresentation( NULL,
359 (const UInt8*)m_filename.c_str(), m_filename.size(), false );
360 if( !imageURLRef )
361 {
362 CGImageRelease( imageRef );
363 free( bitmapData );
364 return false;
365 }
366
367 CGImageDestinationRef destRef = CGImageDestinationCreateWithURL( imageURLRef,
368 imageUTI,
369 1,
370 NULL);
371 CFRelease( imageURLRef );
372 if( !destRef )
373 {
374 CGImageRelease( imageRef );
375 free( bitmapData );
376 fprintf(stderr, "!destRef\n");
377 return false;
378 }
379
380 CGImageDestinationAddImage(destRef, imageRef, NULL);
381 if( !CGImageDestinationFinalize(destRef) )
382 {
383 fprintf(stderr, "Finalize failed\n");
384 return false;
385 }
386
387 CFRelease( destRef );
388 CGImageRelease( imageRef );
389 free( bitmapData );
390
391 return true;
392}
