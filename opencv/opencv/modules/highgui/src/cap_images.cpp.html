65class CvCapture_Images : public CvCapture
66{
67public:
68 CvCapture_Images()
69 {
70 filename = NULL;
71 currentframe = firstframe = 0;
72 length = 0;
73 frame = NULL;
74 grabbedInOpen = false;
75 }
76
77 virtual ~CvCapture_Images()
78 {
79 close();
80 }
81
82 virtual bool open(const char* _filename);
83 virtual void close();
84 virtual double getProperty(int);
85 virtual bool setProperty(int, double);
86 virtual bool grabFrame();
87 virtual IplImage* retrieveFrame(int);
88
89protected:
90 char* filename; // actually a printf-pattern
91 unsigned currentframe;
92 unsigned firstframe; // number of first frame
93 unsigned length; // length of sequence
94
95 IplImage* frame;
96 bool grabbedInOpen;
97};

100void CvCapture_Images::close()
101{
102 if( filename )
103 {
104 free(filename);
105 filename = NULL;
106 }
107 currentframe = firstframe = 0;
108 length = 0;
109 cvReleaseImage( &frame );
110}

113bool CvCapture_Images::grabFrame()
114{
115 char str[_MAX_PATH];
116 sprintf(str, filename, firstframe + currentframe);
117
118 if (grabbedInOpen)
119 {
120 grabbedInOpen = false;
121 ++currentframe;
122
123 return frame != NULL;
124 }
125
126 cvReleaseImage(&frame);
127 frame = cvLoadImage(str, CV_LOAD_IMAGE_ANYDEPTH | CV_LOAD_IMAGE_ANYCOLOR);
128 if( frame )
129 currentframe++;
130
131 return frame != NULL;
132}
133
134IplImage* CvCapture_Images::retrieveFrame(int)
135{
136 return grabbedInOpen ? NULL : frame;
137}

139double CvCapture_Images::getProperty(int id)
140{
141 switch(id)
142 {
143 case CV_CAP_PROP_POS_MSEC:
144 CV_WARN("collections of images don't have framerates\n");
145 return 0;
146 case CV_CAP_PROP_POS_FRAMES:
147 return currentframe;
148 case CV_CAP_PROP_POS_AVI_RATIO:
149 return (double)currentframe / (double)(length - 1);
150 case CV_CAP_PROP_FRAME_WIDTH:
151 return frame ? frame->width : 0;
152 case CV_CAP_PROP_FRAME_HEIGHT:
153 return frame ? frame->height : 0;
154 case CV_CAP_PROP_FPS:
155 CV_WARN("collections of images don't have framerates\n");
156 return 1;
157 case CV_CAP_PROP_FOURCC:
158 CV_WARN("collections of images don't have 4-character codes\n");
159 return 0;
160 }
161 return 0;
162}
163
164bool CvCapture_Images::setProperty(int id, double value)
165{
166 switch(id)
167 {
168 case CV_CAP_PROP_POS_MSEC:
169 case CV_CAP_PROP_POS_FRAMES:
170 if(value < 0) {
171 CV_WARN("seeking to negative positions does not work - clamping\n");
172 value = 0;
173 }
174 if(value >= length) {
175 CV_WARN("seeking beyond end of sequence - clamping\n");
176 value = length - 1;
177 }
178 currentframe = cvRound(value);
179 if (currentframe != 0)
180 grabbedInOpen = false; // grabbed frame is not valid anymore
181 return true;
182 case CV_CAP_PROP_POS_AVI_RATIO:
183 if(value > 1) {
184 CV_WARN("seeking beyond end of sequence - clamping\n");
185 value = 1;
186 } else if(value < 0) {
187 CV_WARN("seeking to negative positions does not work - clamping\n");
188 value = 0;
189 }
190 currentframe = cvRound((length - 1) * value);
191 if (currentframe != 0)
192 grabbedInOpen = false; // grabbed frame is not valid anymore
193 return true;
194 }
195 CV_WARN("unknown/unhandled property\n");
196 return false;
197}

199static char* icvExtractPattern(const char *filename, unsigned *offset)
200{
201 char *name = (char *)filename;
202
203 if( !filename )
204 return 0;
205
206 // check whether this is a valid image sequence filename
207 char *at = strchr(name, '%');
208 if(at)
209 {
210 int dummy;
211 if(sscanf(at + 1, "%ud", &dummy) != 1)
212 return 0;
213 name = strdup(filename);
214 }
215 else // no pattern filename was given - extract the pattern
216 {
217 at = name;
218
219 // ignore directory names
220 char *slash = strrchr(at, '/');
221 if (slash) at = slash + 1;
222
223#ifdef _WIN32
224 slash = strrchr(at, '\\');
225 if (slash) at = slash + 1;
226#endif
227
228 while (*at && !isdigit(*at)) at++;
229
230 if(!*at)
231 return 0;
232
233 sscanf(at, "%u", offset);
234
235 int size = (int)strlen(filename) + 20;
236 name = (char *)malloc(size);
237 strncpy(name, filename, at - filename);
238 name[at - filename] = 0;
239
240 strcat(name, "%0");
241
242 int i;
243 char *extension;
244 for(i = 0, extension = at; isdigit(at[i]); i++, extension++)
245 ;
246 char places[10];
247 sprintf(places, "%dd", i);
248
249 strcat(name, places);
250 strcat(name, extension);
251 }
252
253 return name;
254}

257bool CvCapture_Images::open(const char * _filename)
258{
259 unsigned offset = 0;
260 close();
261
262 filename = icvExtractPattern(_filename, &offset);
263 if(!filename)
264 return false;
265
266 // determine the length of the sequence
267 length = 0;
268 char str[_MAX_PATH];
269 for(;;)
270 {
271 sprintf(str, filename, offset + length);
272 struct stat s;
273 if(stat(str, &s))
274 {
275 if(length == 0 && offset == 0) // allow starting with 0 or 1
276 {
277 offset++;
278 continue;
279 }
280 }
281
282 if(!cvHaveImageReader(str))
283 break;
284
285 length++;
286 }
287
288 if(length == 0)
289 {
290 close();
291 return false;
292 }
293
294 firstframe = offset;
295
296 // grab frame to enable properties retrieval
297 bool grabRes = grabFrame();
298 grabbedInOpen = true;
299 currentframe = 0;
300
301 return grabRes;
302}

307 CvCapture_Images* capture = new CvCapture_Images;
308

311
312 delete capture;
313 return NULL;
314}

340bool CvVideoWriter_Images::writeFrame( const IplImage* image )
341{
342 char str[_MAX_PATH];
343 sprintf(str, filename, currentframe);
344 int ret = cvSaveImage(str, image);
345
346 currentframe++;
347
348 return ret > 0;
349}
350
351void CvVideoWriter_Images::close()
352{
353 if( filename )
354 {
355 free( filename );
356 filename = 0;
357 }
358 currentframe = 0;
359}

361
362bool CvVideoWriter_Images::open( const char* _filename )
363{
364 unsigned offset = 0;
365
366 close();
367
368 filename = icvExtractPattern(_filename, &offset);
369 if(!filename)
370 return false;
371
372 char str[_MAX_PATH];
373 sprintf(str, filename, 0);
374 if(!cvHaveImageWriter(str))
375 {
376 close();
377 return false;
378 }
379
380 currentframe = offset;
381 return true;
382}

391
392 delete writer;
