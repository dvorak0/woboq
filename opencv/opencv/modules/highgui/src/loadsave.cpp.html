57struct ImageCodecInitializer
58{
59 ImageCodecInitializer()
60 {
61 decoders.push_back( new BmpDecoder );
62 encoders.push_back( new BmpEncoder );
63 #ifdef HAVE_JPEG
64 decoders.push_back( new JpegDecoder );
65 encoders.push_back( new JpegEncoder );
66 #endif
67 decoders.push_back( new SunRasterDecoder );
68 encoders.push_back( new SunRasterEncoder );
69 decoders.push_back( new PxMDecoder );
70 encoders.push_back( new PxMEncoder );
71 #ifdef HAVE_TIFF
72 decoders.push_back( new TiffDecoder );
73 #endif
74 encoders.push_back( new TiffEncoder );
75 #ifdef HAVE_PNG
76 decoders.push_back( new PngDecoder );
77 encoders.push_back( new PngEncoder );
78 #endif
79 #ifdef HAVE_JASPER
80 decoders.push_back( new Jpeg2KDecoder );
81 encoders.push_back( new Jpeg2KEncoder );
82 #endif
83 #ifdef HAVE_OPENEXR
84 decoders.push_back( new ExrDecoder );
85 encoders.push_back( new ExrEncoder );
86 #endif
87 // because it is a generic image I/O API, supporting many formats,
88 // it should be last in the list.
89 #ifdef HAVE_IMAGEIO
90 decoders.push_back( new ImageIODecoder );
91 encoders.push_back( new ImageIOEncoder );
92 #endif
93 }

95 vector<ImageDecoder> decoders;
96 vector<ImageEncoder> encoders;
97};

99static ImageCodecInitializer codecs;

101static ImageDecoder findDecoder( const string& filename )
102{
103 size_t i, maxlen = 0;
104 for( i = 0; i < codecs.decoders.size(); i++ )
105 {
106 size_t len = codecs.decoders[i]->signatureLength();
107 maxlen = std::max(maxlen, len);
108 }
109
110 FILE* f= fopen( filename.c_str(), "rb" );
111 if( !f )
112 return ImageDecoder();
113 string signature(maxlen, ' ');
114 maxlen = fread( &signature[0], 1, maxlen, f );
115 fclose(f);
116 signature = signature.substr(0, maxlen);
117
118 for( i = 0; i < codecs.decoders.size(); i++ )
119 {
120 if( codecs.decoders[i]->checkSignature(signature) )
121 return codecs.decoders[i]->newDecoder();
122 }
123
124 return ImageDecoder();
125}
126
127static ImageDecoder findDecoder( const Mat& buf )
128{
129 size_t i, maxlen = 0;
130
131 if( buf.rows*buf.cols < 1 || !buf.isContinuous() )
132 return ImageDecoder();
133
134 for( i = 0; i < codecs.decoders.size(); i++ )
135 {
136 size_t len = codecs.decoders[i]->signatureLength();
137 maxlen = std::max(maxlen, len);
138 }
139
140 string signature(maxlen, ' ');
141 size_t bufSize = buf.rows*buf.cols*buf.elemSize();
142 maxlen = std::min(maxlen, bufSize);
143 memcpy( &signature[0], buf.data, maxlen );
144
145 for( i = 0; i < codecs.decoders.size(); i++ )
146 {
147 if( codecs.decoders[i]->checkSignature(signature) )
148 return codecs.decoders[i]->newDecoder();
149 }
150
151 return ImageDecoder();
152}
153
154static ImageEncoder findEncoder( const string& _ext )
155{
156 if( _ext.size() <= 1 )
157 return ImageEncoder();
158
159 const char* ext = strrchr( _ext.c_str(), '.' );
160 if( !ext )
161 return ImageEncoder();
162 int len = 0;
163 for( ext++; isalnum(ext[len]) && len < 128; len++ )
164 ;
165
166 for( size_t i = 0; i < codecs.encoders.size(); i++ )
167 {
168 string description = codecs.encoders[i]->getDescription();
169 const char* descr = strchr( description.c_str(), '(' );
170
171 while( descr )
172 {
173 descr = strchr( descr + 1, '.' );
174 if( !descr )
175 break;
176 int j = 0;
177 for( descr++; isalnum(descr[j]) && j < len; j++ )
178 {
179 int c1 = tolower(ext[j]);
180 int c2 = tolower(descr[j]);
181 if( c1 != c2 )
182 break;
183 }
184 if( j == len && !isalnum(descr[j]))
185 return codecs.encoders[i]->newEncoder();
186 descr += j;
187 }
188 }
189
190 return ImageEncoder();
191}

193enum { LOAD_CVMAT=0, LOAD_IMAGE=1, LOAD_MAT=2 };
194
195static void*
196imread_( const string& filename, int flags, int hdrtype, Mat* mat=0 )
197{
198 IplImage* image = 0;
199 CvMat *matrix = 0;
200 Mat temp, *data = &temp;
201
202 ImageDecoder decoder = findDecoder(filename);
203 if( decoder.empty() )
204 return 0;
205 decoder->setSource(filename);
206 if( !decoder->readHeader() )
207 return 0;
208
209 CvSize size;
210 size.width = decoder->width();
211 size.height = decoder->height();
212
213 int type = decoder->type();
214 if( flags != -1 )
215 {
216 if( (flags & CV_LOAD_IMAGE_ANYDEPTH) == 0 )
217 type = CV_MAKETYPE(CV_8U, CV_MAT_CN(type));
218
219 if( (flags & CV_LOAD_IMAGE_COLOR) != 0 ||
220 ((flags & CV_LOAD_IMAGE_ANYCOLOR) != 0 && CV_MAT_CN(type) > 1) )
221 type = CV_MAKETYPE(CV_MAT_DEPTH(type), 3);
222 else
223 type = CV_MAKETYPE(CV_MAT_DEPTH(type), 1);
224 }
225
226 if( hdrtype == LOAD_CVMAT || hdrtype == LOAD_MAT )
227 {
228 if( hdrtype == LOAD_CVMAT )
229 {
230 matrix = cvCreateMat( size.height, size.width, type );
231 temp = cvarrToMat(matrix);
232 }
233 else
234 {
235 mat->create( size.height, size.width, type );
236 data = mat;
237 }
238 }
239 else
240 {
241 image = cvCreateImage( size, cvIplDepth(type), CV_MAT_CN(type) );
242 temp = cvarrToMat(image);
243 }
244
245 if( !decoder->readData( *data ))
246 {
247 cvReleaseImage( &image );
248 cvReleaseMat( &matrix );
249 if( mat )
250 mat->release();
251 return 0;
252 }
253
254 return hdrtype == LOAD_CVMAT ? (void*)matrix :
255 hdrtype == LOAD_IMAGE ? (void*)image : (void*)mat;
256}

265static bool imwrite_( const string& filename, const Mat& image,
266 const vector<int>& params, bool flipv )
267{
268 Mat temp;
269 const Mat* pimage = &image;
270
271 CV_Assert( image.channels() == 1 || image.channels() == 3 || image.channels() == 4 );
272
273 ImageEncoder encoder = findEncoder( filename );
274 if( encoder.empty() )
275 CV_Error( CV_StsError, "could not find a writer for the specified extension" );
276
277 if( !encoder->isFormatSupported(image.depth()) )
278 {
279 CV_Assert( encoder->isFormatSupported(CV_8U) );
280 image.convertTo( temp, CV_8U );
281 pimage = &temp;
282 }
283
284 if( flipv )
285 {
286 flip(*pimage, temp, 0);
287 pimage = &temp;
288 }
289
290 encoder->setDestination( filename );
291 bool code = encoder->write( *pimage, params );
292
293 // CV_Assert( code );
294 return code;
295}

297bool imwrite( const string& filename, InputArray _img,
298 const vector<int>& params )
299{
300 Mat img = _img.getMat();
301 return imwrite_(filename, img, params, false);

304static void*
305imdecode_( const Mat& buf, int flags, int hdrtype, Mat* mat=0 )
306{
307 CV_Assert(buf.data && buf.isContinuous());
308 IplImage* image = 0;
309 CvMat *matrix = 0;
310 Mat temp, *data = &temp;
311 string filename;
312
313 ImageDecoder decoder = findDecoder(buf);
314 if( decoder.empty() )
315 return 0;
316
317 if( !decoder->setSource(buf) )
318 {
319 filename = tempfile();
320 FILE* f = fopen( filename.c_str(), "wb" );
321 if( !f )
322 return 0;
323 size_t bufSize = buf.cols*buf.rows*buf.elemSize();
324 fwrite( &buf.data[0], 1, bufSize, f );
325 fclose(f);
326 decoder->setSource(filename);
327 }
328
329 if( !decoder->readHeader() )
330 {
331 if( !filename.empty() )
332 remove(filename.c_str());
333 return 0;
334 }
335
336 CvSize size;
337 size.width = decoder->width();
338 size.height = decoder->height();
339
340 int type = decoder->type();
341 if( flags != -1 )
342 {
343 if( (flags & CV_LOAD_IMAGE_ANYDEPTH) == 0 )
344 type = CV_MAKETYPE(CV_8U, CV_MAT_CN(type));
345
346 if( (flags & CV_LOAD_IMAGE_COLOR) != 0 ||
347 ((flags & CV_LOAD_IMAGE_ANYCOLOR) != 0 && CV_MAT_CN(type) > 1) )
348 type = CV_MAKETYPE(CV_MAT_DEPTH(type), 3);
349 else
350 type = CV_MAKETYPE(CV_MAT_DEPTH(type), 1);
351 }
352
353 if( hdrtype == LOAD_CVMAT || hdrtype == LOAD_MAT )
354 {
355 if( hdrtype == LOAD_CVMAT )
356 {
357 matrix = cvCreateMat( size.height, size.width, type );
358 temp = cvarrToMat(matrix);
359 }
360 else
361 {
362 mat->create( size.height, size.width, type );
363 data = mat;
364 }
365 }
366 else
367 {
368 image = cvCreateImage( size, cvIplDepth(type), CV_MAT_CN(type) );
369 temp = cvarrToMat(image);
370 }
371
372 bool code = decoder->readData( *data );
373 if( !filename.empty() )
374 remove(filename.c_str());
375
376 if( !code )
377 {
378 cvReleaseImage( &image );
379 cvReleaseMat( &matrix );
380 if( mat )
381 mat->release();
382 return 0;
383 }
384
385 return hdrtype == LOAD_CVMAT ? (void*)matrix :
386 hdrtype == LOAD_IMAGE ? (void*)image : (void*)mat;
387}

405bool imencode( const string& ext, InputArray _image,
406 vector<uchar>& buf, const vector<int>& params )
407{
408 Mat image = _image.getMat();
409
410 int channels = image.channels();
411 CV_Assert( channels == 1 || channels == 3 || channels == 4 );
412
413 ImageEncoder encoder = findEncoder( ext );
414 if( encoder.empty() )
415 CV_Error( CV_StsError, "could not find encoder for the specified extension" );
416
417 if( !encoder->isFormatSupported(image.depth()) )
418 {
419 CV_Assert( encoder->isFormatSupported(CV_8U) );
420 Mat temp;
421 image.convertTo(temp, CV_8U);
422 image = temp;
423 }
424
425 bool code;
426 if( encoder->setDestination(buf) )
427 {
428 code = encoder->write(image, params);
429 encoder->throwOnEror();
430 CV_Assert( code );
431 }
432 else
433 {
434 string filename = tempfile();
435 code = encoder->setDestination(filename);
436 CV_Assert( code );
437
438 code = encoder->write(image, params);
439 encoder->throwOnEror();
440 CV_Assert( code );
441
442 FILE* f = fopen( filename.c_str(), "rb" );
443 CV_Assert(f != 0);
444 fseek( f, 0, SEEK_END );
445 long pos = ftell(f);
446 buf.resize((size_t)pos);
447 fseek( f, 0, SEEK_SET );
448 buf.resize(fread( &buf[0], 1, buf.size(), f ));
449 fclose(f);
450 remove(filename.c_str());
451 }
452 return code;
453}

486CV_IMPL int
487cvSaveImage( const char* filename, const CvArr* arr, const int* _params )
488{
489 int i = 0;
490 if( _params )
491 {
492 for( ; _params[i] > 0; i += 2 )
493 ;
494 }
495 return cv::imwrite_(filename, cv::cvarrToMat(arr),
496 i > 0 ? cv::vector<int>(_params, _params+i) : cv::vector<int>(),
497 CV_IS_IMAGE(arr) && ((const IplImage*)arr)->origin == IPL_ORIGIN_BL );
498}

500/* decode image stored in the buffer */
501CV_IMPL IplImage*
502cvDecodeImage( const CvMat* _buf, int iscolor )
503{
504 CV_Assert( _buf && CV_IS_MAT_CONT(_buf->type) );
505 cv::Mat buf(1, _buf->rows*_buf->cols*CV_ELEM_SIZE(_buf->type), CV_8U, _buf->data.ptr);
506 return (IplImage*)cv::imdecode_(buf, iscolor, cv::LOAD_IMAGE );
507}

509CV_IMPL CvMat*
510cvDecodeImageM( const CvMat* _buf, int iscolor )

512 CV_Assert( _buf && CV_IS_MAT_CONT(_buf->type) );
513 cv::Mat buf(1, _buf->rows*_buf->cols*CV_ELEM_SIZE(_buf->type), CV_8U, _buf->data.ptr);
514 return (CvMat*)cv::imdecode_(buf, iscolor, cv::LOAD_CVMAT );
515}

517CV_IMPL CvMat*
518cvEncodeImage( const char* ext, const CvArr* arr, const int* _params )
519{
520 int i = 0;
521 if( _params )
522 {
523 for( ; _params[i] > 0; i += 2 )
524 ;
525 }
526 cv::Mat img = cv::cvarrToMat(arr);
527 if( CV_IS_IMAGE(arr) && ((const IplImage*)arr)->origin == IPL_ORIGIN_BL )
528 {
529 cv::Mat temp;
530 cv::flip(img, temp, 0);
531 img = temp;
532 }
533 cv::vector<uchar> buf;
534
535 bool code = cv::imencode(ext, img, buf,
536 i > 0 ? std::vector<int>(_params, _params+i) : std::vector<int>() );
537 if( !code )
538 return 0;
539 CvMat* _buf = cvCreateMat(1, (int)buf.size(), CV_8U);
540 memcpy( _buf->data.ptr, &buf[0], buf.size() );
541
542 return _buf;
543}
