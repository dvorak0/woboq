51class CV_HighGuiOnlyGuiTest : public cvtest::BaseTest
52{
53protected:
54 void run(int);

57void Foo(int /*k*/, void* /*z*/) {}
58
59void CV_HighGuiOnlyGuiTest::run( int /*start_from */)
60{
61 ts->printf(ts->LOG, "GUI 0\n");
62 cvDestroyAllWindows();
63
64 ts->printf(ts->LOG, "GUI 1\n");
65 namedWindow("Win");
66
67 ts->printf(ts->LOG, "GUI 2\n");
68 Mat m(256, 256, CV_8U);
69 m = Scalar(128);
70
71 ts->printf(ts->LOG, "GUI 3\n");
72 imshow("Win", m);
73
74 ts->printf(ts->LOG, "GUI 4\n");
75 int value = 50;
76
77 ts->printf(ts->LOG, "GUI 5\n");
78 createTrackbar( "trackbar", "Win", &value, 100, Foo, &value);
79
80 ts->printf(ts->LOG, "GUI 6\n");
81 getTrackbarPos( "trackbar", "Win" );
82
83 ts->printf(ts->LOG, "GUI 7\n");
84 waitKey(500);
85
86 ts->printf(ts->LOG, "GUI 8\n");
87 cvDestroyAllWindows();
88 ts->set_failed_test_info(cvtest::TS::OK);
89}
