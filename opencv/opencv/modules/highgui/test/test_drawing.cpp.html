50class CV_DrawingTest : public cvtest::BaseTest
51{
52public:
53 CV_DrawingTest(){}
54protected:
55 void run( int );
56 virtual void draw( Mat& img ) = 0;
57 virtual int checkLineIterator( Mat& img) = 0;
58};
59
60void CV_DrawingTest::run( int )
61{
62 Mat testImg, valImg;
63 const string fname = "drawing/image.png";
64 string path = ts->get_data_path(), filename;
65 filename = path + fname;
66
67 draw( testImg );
68
69 valImg = imread( filename );
70 if( valImg.empty() )
71 {
72 imwrite( filename, testImg );
73 //ts->printf( ts->LOG, "test image can not be read");
74 //ts->set_failed_test_info(cvtest::TS::FAIL_INVALID_TEST_DATA);
75 }
76 else
77 {
78 float err = (float)norm( testImg, valImg, NORM_L1 );
79 float Eps = 1;
80 if( err > Eps)
81 {
82 ts->printf( ts->LOG, "NORM_L1 between testImg and valImg is equal %f (larger than %f)\n", err, Eps );
83 ts->set_failed_test_info(cvtest::TS::FAIL_BAD_ACCURACY);
84 }
85 else
86 {
87 ts->set_failed_test_info(checkLineIterator( testImg ));
88 }
89 }
90 ts->set_failed_test_info(cvtest::TS::OK);
91}

225int CV_DrawingTest_CPP::checkLineIterator( Mat& img )
226{
227 LineIterator it( img, Point(0,300), Point(1000, 300) );
228 for(int i = 0; i < it.count; ++it, i++ )
229 {
230 Vec3b v = (Vec3b)(*(*it)) - img.at<Vec3b>(300,i);
231 float err = (float)norm( v );
232 if( err != 0 )
233 {
234 ts->printf( ts->LOG, "LineIterator works incorrect" );
235 ts->set_failed_test_info(cvtest::TS::FAIL_INVALID_OUTPUT);
236 }
237 }
238 ts->set_failed_test_info(cvtest::TS::OK);
239 return 0;
240}

392int CV_DrawingTest_C::checkLineIterator( Mat& _img )
393{
394 CvLineIterator it;
395 CvMat img = _img;
396 int count = cvInitLineIterator( &img, cvPoint(0,300), cvPoint(1000, 300), &it );
397 for(int i = 0; i < count; i++ )
398 {
399 Vec3b v = (Vec3b)(*(it.ptr)) - _img.at<Vec3b>(300,i);
400 float err = (float)norm( v );
401 if( err != 0 )
402 {
403 ts->printf( ts->LOG, "CvLineIterator works incorrect" );
404 ts->set_failed_test_info(cvtest::TS::FAIL_INVALID_OUTPUT);
405 }
406 CV_NEXT_LINE_POINT(it);
407 }
408 ts->set_failed_test_info(cvtest::TS::OK);
409 return 0;

599 "\xD1\x8A\xD1\x8B\xD1\x8C\xD1\x8D\xD1\x8E\xD1\x8F");
600 // cyrillic letters capital
601 lines.push_back("\xD0\x90\xD0\x91\xD0\x92\xD0\x93\xD0\x94\xD0\x95\xD0\x81\xD0\x96\xD0\x97"
602 "\xD0\x98\xD0\x99\xD0\x9A\xD0\x9B\xD0\x9C\xD0\x9D\xD0\x9E\xD0\x9F\xD0\xA0"
603 "\xD0\xA1\xD0\xA2\xD0\xA3\xD0\xA4\xD0\xA5\xD0\xA6\xD0\xA7\xD0\xA8\xD0\xA9"
604 "\xD0\xAA\xD0\xAB\xD0\xAC\xD0\xAD\xD0\xAE\xD0\xAF");
605 // bounds
606 lines.push_back("-\xD0\x80-\xD0\x8E-\xD0\x8F-");
607 lines.push_back("-\xD1\x90-\xD1\x91-\xD1\xBF-");
608 // bad utf8
609 lines.push_back("-\x81-\x82-\x83-");
610 lines.push_back("--\xF0--");
611 lines.push_back("-\xF0");
612
613 vector<int> fonts;
614 fonts.push_back(FONT_HERSHEY_SIMPLEX);
615 fonts.push_back(FONT_HERSHEY_PLAIN);
616 fonts.push_back(FONT_HERSHEY_DUPLEX);
617 fonts.push_back(FONT_HERSHEY_COMPLEX);
618 fonts.push_back(FONT_HERSHEY_TRIPLEX);
619 fonts.push_back(FONT_HERSHEY_COMPLEX_SMALL);
620 fonts.push_back(FONT_HERSHEY_SCRIPT_SIMPLEX);
621 fonts.push_back(FONT_HERSHEY_SCRIPT_COMPLEX);
622
623 vector<Mat> results;
624 Size bigSize(0, 0);
625 for (vector<int>::const_iterator font = fonts.begin(); font != fonts.end(); ++font)
626 {
627 for (int italic = 0; italic <= FONT_ITALIC; italic += FONT_ITALIC)
628 {
629 for (vector<string>::const_iterator line = lines.begin(); line != lines.end(); ++line)
630 {
631 const float fontScale = 1;
632 const int thickness = 1;
633 const Scalar color(20,20,20);

637 Point textOrg(0, textSize.height + 2);

639 putText(img, *line, textOrg, *font | italic, fontScale, color, thickness, CV_AA);
640
641 results.push_back(img);
642 bigSize.width = max(bigSize.width, img.size().width);
643 bigSize.height += img.size().height + 1;
644 }
645 }
646 }
647
648 int shift = 0;
649 Mat result(bigSize, CV_8UC3, Scalar(100, 100, 100));
650 for (vector<Mat>::const_iterator img = results.begin(); img != results.end(); ++img)
651 {
652 Rect roi(Point(0, shift), img->size());
653 Mat sub(result, roi);
654 img->copyTo(sub);
655 shift += img->size().height + 1;
656 }
657 // imwrite("all_fonts.png", result);
658 }
