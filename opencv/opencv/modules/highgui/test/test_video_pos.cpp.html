43#include "test_precomp.hpp"
44#include "opencv2/highgui/highgui.hpp"
45
46using namespace cv;
47using namespace std;
48
49class CV_PositioningTest : public cvtest::BaseTest
50{
51public:
52 CV_PositioningTest()
53 {
54 framesize = Size(640, 480);
55 }
56
57 Mat drawFrame(int i)
58 {
59 Mat mat = Mat::zeros(framesize, CV_8UC3);
60
61 mat = Scalar(fabs(cos(i*0.08)*255), fabs(sin(i*0.05)*255), i);
62 putText(mat, format("%03d", i), Point(10, 350), 0, 10, Scalar(128, 255, 255), 15);
63 return mat;
64 }
65
66 string getFilename(const cvtest::VideoFormat& fmt)
67 {
68 return cv::tempfile((cvtest::fourccToString(fmt.fourcc) + "." + fmt.ext).c_str());
69 }
70
71 bool CreateTestVideo(const cvtest::VideoFormat& fmt, int framecount, string filename)
72 {
73 VideoWriter writer(filename, fmt.fourcc, 25, framesize, true);
74 if( !writer.isOpened() )
75 return false;
76
77 for (int i = 0; i < framecount; ++i)
78 {
79 Mat img = drawFrame(i);
80 writer << img;
81 }
82 return true;
83 }
84
85 void run(int)
86 {
87 int n_frames = 100;
88
89 for( int testcase = 0; ; testcase++ )
90 {
91 const cvtest::VideoFormat& fmt = cvtest::g_specific_fmt_list[testcase];
92 if( fmt.empty() )
93 break;
94 string filename = getFilename(fmt);
95 ts->printf(ts->LOG, "\nFile: %s\n", filename.c_str());
96
97 if( !CreateTestVideo(fmt, n_frames, filename) )
98 {
99 ts->printf(ts->LOG, "\nError: cannot create video file");
100 ts->set_failed_test_info(ts->FAIL_INVALID_OUTPUT);
101 return;
102 }
103
104 VideoCapture cap(filename);
105
106 if (!cap.isOpened())
107 {
108 ts->printf(ts->LOG, "\nError: cannot read video file.");
109 ts->set_failed_test_info(ts->FAIL_INVALID_TEST_DATA);
110 return;
111 }
112
113 int N0 = (int)cap.get(CV_CAP_PROP_FRAME_COUNT);
114 cap.set(CV_CAP_PROP_POS_FRAMES, 0);
115 int N = (int)cap.get(CV_CAP_PROP_FRAME_COUNT);
116
117 // See the same hack in CV_HighGuiTest::SpecificVideoTest for explanation.
118 int allowed_extra_frames = 0;
119 if (fmt.fourcc == CV_FOURCC('M', 'P', 'E', 'G') && fmt.ext == "mkv")
120 allowed_extra_frames = 1;
121
122 if (N < n_frames || N > n_frames + allowed_extra_frames || N != N0)
123 {
124 ts->printf(ts->LOG, "\nError: returned frame count (N0=%d, N=%d) is different from the reference number %d\n", N0, N, n_frames);
125 ts->set_failed_test_info(ts->FAIL_INVALID_OUTPUT);
126 return;
127 }
128
129 for (int k = 0; k < n_frames; ++k)
130 {
131 int idx = theRNG().uniform(0, n_frames);
132
133 if( !cap.set(CV_CAP_PROP_POS_FRAMES, idx) )
134 {
135 ts->printf(ts->LOG, "\nError: cannot seek to frame %d.\n", idx);
136 ts->set_failed_test_info(ts->FAIL_INVALID_OUTPUT);
137 return;
138 }
139
140 int idx1 = (int)cap.get(CV_CAP_PROP_POS_FRAMES);
141
142 Mat img; cap >> img;
143 Mat img0 = drawFrame(idx);
144
145 if( idx != idx1 )
146 {
147 ts->printf(ts->LOG, "\nError: the current position (%d) after seek is different from specified (%d)\n",
148 idx1, idx);
149 ts->printf(ts->LOG, "Saving both frames ...\n");
150 ts->set_failed_test_info(ts->FAIL_INVALID_OUTPUT);
151 // imwrite("opencv_test_highgui_postest_actual.png", img);
152 // imwrite("opencv_test_highgui_postest_expected.png", img0);
153 return;
154 }
155
156 if (img.empty())
157 {
158 ts->printf(ts->LOG, "\nError: cannot read a frame at position %d.\n", idx);
159 ts->set_failed_test_info(ts->FAIL_INVALID_OUTPUT);
160 return;
161 }
162
163 double err = PSNR(img, img0);
164
165 if( err < 20 )
166 {
167 ts->printf(ts->LOG, "The frame read after positioning to %d is incorrect (PSNR=%g)\n", idx, err);
168 ts->printf(ts->LOG, "Saving both frames ...\n");
169 ts->set_failed_test_info(ts->FAIL_INVALID_OUTPUT);
170 // imwrite("opencv_test_highgui_postest_actual.png", img);
171 // imwrite("opencv_test_highgui_postest_expected.png", img0);
172 return;
173 }
174 }
175 }
176 }
177
178 Size framesize;
179};
