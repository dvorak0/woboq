75string ext_from_int(int ext)
76{
77#ifdef HAVE_PNG
78 if (ext == 0) return ".png";

80 if (ext == 1) return ".bmp";
81 if (ext == 2) return ".pgm";

99 {
100 for (int ext = 0; ext < 4; ++ext) // 0 - png, 1 - bmp, 2 - pgm, 3 - tiff

102 if(ext_from_int(ext).empty())
103 continue;
104 for (int num_channels = 1; num_channels <= 3; num_channels+=2)

106 ts->printf(ts->LOG, "image type depth:%d channels:%d ext: %s\n", CV_8U, num_channels, ext_from_int(ext).c_str());

108 circle(img, Point2i((img_c * k) / 2, (img_r * k) / 2), cv::min((img_r * k), (img_c * k)) / 4 , Scalar::all(255));
109
110 string img_path = cv::tempfile(ext_from_int(ext).c_str());
111 ts->printf(ts->LOG, "writing image : %s\n", img_path.c_str());
112 imwrite(img_path, img);
113
114 ts->printf(ts->LOG, "reading test image : %s\n", img_path.c_str());

260TEST(Highgui_ImreadVSCvtColor, regression)
261{
262 cvtest::TS& ts = *cvtest::TS::ptr();
263
264 const int MAX_MEAN_DIFF = 1;
265 const int MAX_ABS_DIFF = 10;
266
267 string imgName = string(ts.get_data_path()) + "/../cv/shared/lena.png";
268 Mat original_image = imread(imgName);
269 Mat gray_by_codec = imread(imgName, 0);
270 Mat gray_by_cvt;
271
272 cvtColor(original_image, gray_by_cvt, CV_BGR2GRAY);
273
274 Mat diff;
275 absdiff(gray_by_codec, gray_by_cvt, diff);
276
277 double actual_avg_diff = (double)mean(diff)[0];
278 double actual_maxval, actual_minval;
279 minMaxLoc(diff, &actual_minval, &actual_maxval);
280 //printf("actual avg = %g, actual maxdiff = %g, npixels = %d\n", actual_avg_diff, actual_maxval, (int)diff.total());
281
282 EXPECT_LT(actual_avg_diff, MAX_MEAN_DIFF);
283 EXPECT_LT(actual_maxval, MAX_ABS_DIFF);
284}

478class CV_GrfmtReadTifTiledWithNotFullTiles: public cvtest::BaseTest
479{
480public:
481 void run(int)
482 {
483 try
484 {
485 /* see issue #3472 - dealing with tiled images where the tile size is
486 * not a multiple of image size.
487 * The tiled images were created with 'convert' from ImageMagick,
488 * using the command 'convert <input> -define tiff:tile-geometry=128x128 -depth [8|16] <output>
489 * Note that the conversion to 16 bits expands the range from 0-255 to 0-255*255,
490 * so the test converts back but rounding errors cause small differences.
491 */
492 cv::Mat img = imread(string(ts->get_data_path()) + "readwrite/non_tiled.tif",-1);
493 if (img.empty()) ts->set_failed_test_info(cvtest::TS::FAIL_INVALID_TEST_DATA);
494 ASSERT_TRUE(img.channels() == 3);
495 cv::Mat tiled8 = imread(string(ts->get_data_path()) + "readwrite/tiled_8.tif", -1);
496 if (tiled8.empty()) ts->set_failed_test_info(cvtest::TS::FAIL_INVALID_TEST_DATA);
497 ASSERT_PRED_FORMAT2(cvtest::MatComparator(0, 0), img, tiled8);
498
499 cv::Mat tiled16 = imread(string(ts->get_data_path()) + "readwrite/tiled_16.tif", -1);
500 if (tiled16.empty()) ts->set_failed_test_info(cvtest::TS::FAIL_INVALID_TEST_DATA);
501 ASSERT_TRUE(tiled16.elemSize() == 6);
502 tiled16.convertTo(tiled8, CV_8UC3, 1./256.);
503 ASSERT_PRED_FORMAT2(cvtest::MatComparator(2, 0), img, tiled8);
504 // What about 32, 64 bit?
505 }
506 catch(...)
507 {
508 ts->set_failed_test_info(cvtest::TS::FAIL_EXCEPTION);
509 }
510 ts->set_failed_test_info(cvtest::TS::OK);
511 }
