134static CvHaarClassifierCascade*
135icvCreateHaarClassifierCascade( int stage_count )
136{
137 CvHaarClassifierCascade* cascade = 0;
138
139 int block_size = sizeof(*cascade) + stage_count*sizeof(*cascade->stage_classifier);
140
141 if( stage_count <= 0 )
142 CV_Error( CV_StsOutOfRange, "Number of stages should be positive" );
143
144 cascade = (CvHaarClassifierCascade*)cvAlloc( block_size );
145 memset( cascade, 0, block_size );
146
147 cascade->stage_classifier = (CvHaarStageClassifier*)(cascade + 1);
148 cascade->flags = CV_HAAR_MAGIC_VAL;
149 cascade->count = stage_count;
150
151 return cascade;
152}

176/* create more efficient internal representation of haar classifier cascade */
177static CvHidHaarClassifierCascade*
178icvCreateHidHaarClassifierCascade( CvHaarClassifierCascade* cascade )
179{
180 CvRect* ipp_features = 0;
181 float *ipp_weights = 0, *ipp_thresholds = 0, *ipp_val1 = 0, *ipp_val2 = 0;
182 int* ipp_counts = 0;
183
184 CvHidHaarClassifierCascade* out = 0;
185
186 int i, j, k, l;
187 int datasize;
188 int total_classifiers = 0;
189 int total_nodes = 0;
190 char errorstr[1000];
191 CvHidHaarClassifier* haar_classifier_ptr;
192 CvHidHaarTreeNode* haar_node_ptr;
193 CvSize orig_window_size;
194 int has_tilted_features = 0;
195 int max_count = 0;
196
197 if( !CV_IS_HAAR_CLASSIFIER(cascade) )
198 CV_Error( !cascade ? CV_StsNullPtr : CV_StsBadArg, "Invalid classifier pointer" );
199
200 if( cascade->hid_cascade )
201 CV_Error( CV_StsError, "hid_cascade has been already created" );
202
203 if( !cascade->stage_classifier )
204 CV_Error( CV_StsNullPtr, "" );
205
206 if( cascade->count <= 0 )
207 CV_Error( CV_StsOutOfRange, "Negative number of cascade stages" );
208
209 orig_window_size = cascade->orig_window_size;
210
211 /* check input structure correctness and calculate total memory size needed for
212 internal representation of the classifier cascade */
213 for( i = 0; i < cascade->count; i++ )
214 {
215 CvHaarStageClassifier* stage_classifier = cascade->stage_classifier + i;
216
217 if( !stage_classifier->classifier ||
218 stage_classifier->count <= 0 )
219 {
220 sprintf( errorstr, "header of the stage classifier #%d is invalid "
221 "(has null pointers or non-positive classfier count)", i );
222 CV_Error( CV_StsError, errorstr );
223 }
224
225 max_count = MAX( max_count, stage_classifier->count );
226 total_classifiers += stage_classifier->count;
227
228 for( j = 0; j < stage_classifier->count; j++ )
229 {
230 CvHaarClassifier* classifier = stage_classifier->classifier + j;
231
232 total_nodes += classifier->count;
233 for( l = 0; l < classifier->count; l++ )
234 {
235 for( k = 0; k < CV_HAAR_FEATURE_MAX; k++ )
236 {
237 if( classifier->haar_feature[l].rect[k].r.width )
238 {
239 CvRect r = classifier->haar_feature[l].rect[k].r;
240 int tilted = classifier->haar_feature[l].tilted;
241 has_tilted_features |= tilted != 0;
242 if( r.width < 0 || r.height < 0 || r.y < 0 ||
243 r.x + r.width > orig_window_size.width
244 ||
245 (!tilted &&
246 (r.x < 0 || r.y + r.height > orig_window_size.height))
247 ||
248 (tilted && (r.x - r.height < 0 ||
249 r.y + r.width + r.height > orig_window_size.height)))
250 {
251 sprintf( errorstr, "rectangle #%d of the classifier #%d of "
252 "the stage classifier #%d is not inside "
253 "the reference (original) cascade window", k, j, i );
254 CV_Error( CV_StsNullPtr, errorstr );
255 }
256 }
257 }
258 }
259 }
260 }
261
262 // this is an upper boundary for the whole hidden cascade size
263 datasize = sizeof(CvHidHaarClassifierCascade) +
264 sizeof(CvHidHaarStageClassifier)*cascade->count +
265 sizeof(CvHidHaarClassifier) * total_classifiers +
266 sizeof(CvHidHaarTreeNode) * total_nodes +
267 sizeof(void*)*(total_nodes + total_classifiers);
268
269 out = (CvHidHaarClassifierCascade*)cvAlloc( datasize );
270 memset( out, 0, sizeof(*out) );
271
272 /* init header */
273 out->count = cascade->count;
274 out->stage_classifier = (CvHidHaarStageClassifier*)(out + 1);
275 haar_classifier_ptr = (CvHidHaarClassifier*)(out->stage_classifier + cascade->count);
276 haar_node_ptr = (CvHidHaarTreeNode*)(haar_classifier_ptr + total_classifiers);
277
278 out->isStumpBased = 1;
279 out->has_tilted_features = has_tilted_features;
280 out->is_tree = 0;
281
282 /* initialize internal representation */
283 for( i = 0; i < cascade->count; i++ )
284 {
285 CvHaarStageClassifier* stage_classifier = cascade->stage_classifier + i;
286 CvHidHaarStageClassifier* hid_stage_classifier = out->stage_classifier + i;
287
288 hid_stage_classifier->count = stage_classifier->count;
289 hid_stage_classifier->threshold = stage_classifier->threshold - icv_stage_threshold_bias;
290 hid_stage_classifier->classifier = haar_classifier_ptr;
291 hid_stage_classifier->two_rects = 1;
292 haar_classifier_ptr += stage_classifier->count;
293
294 hid_stage_classifier->parent = (stage_classifier->parent == -1)
295 ? NULL : out->stage_classifier + stage_classifier->parent;
296 hid_stage_classifier->next = (stage_classifier->next == -1)
297 ? NULL : out->stage_classifier + stage_classifier->next;
298 hid_stage_classifier->child = (stage_classifier->child == -1)
299 ? NULL : out->stage_classifier + stage_classifier->child;
300
301 out->is_tree |= hid_stage_classifier->next != NULL;
302
303 for( j = 0; j < stage_classifier->count; j++ )
304 {
305 CvHaarClassifier* classifier = stage_classifier->classifier + j;
306 CvHidHaarClassifier* hid_classifier = hid_stage_classifier->classifier + j;
307 int node_count = classifier->count;
308 float* alpha_ptr = (float*)(haar_node_ptr + node_count);
309
310 hid_classifier->count = node_count;
311 hid_classifier->node = haar_node_ptr;
312 hid_classifier->alpha = alpha_ptr;
313
314 for( l = 0; l < node_count; l++ )
315 {
316 CvHidHaarTreeNode* node = hid_classifier->node + l;
317 CvHaarFeature* feature = classifier->haar_feature + l;
318 memset( node, -1, sizeof(*node) );
319 node->threshold = classifier->threshold[l];
320 node->left = classifier->left[l];
321 node->right = classifier->right[l];
322
323 if( fabs(feature->rect[2].weight) < DBL_EPSILON ||
324 feature->rect[2].r.width == 0 ||
325 feature->rect[2].r.height == 0 )
326 memset( &(node->feature.rect[2]), 0, sizeof(node->feature.rect[2]) );
327 else
328 hid_stage_classifier->two_rects = 0;
329 }
330
331 memcpy( alpha_ptr, classifier->alpha, (node_count+1)*sizeof(alpha_ptr[0]));
332 haar_node_ptr =
333 (CvHidHaarTreeNode*)cvAlignPtr(alpha_ptr+node_count+1, sizeof(void*));
334
335 out->isStumpBased &= node_count == 1;
336 }
337 }
338/*
339#ifdef HAVE_IPP
340 int can_use_ipp = !out->has_tilted_features && !out->is_tree && out->isStumpBased;
341
342 if( can_use_ipp )
343 {
344 int ipp_datasize = cascade->count*sizeof(out->ipp_stages[0]);
345 float ipp_weight_scale=(float)(1./((orig_window_size.width-icv_object_win_border*2)*
346 (orig_window_size.height-icv_object_win_border*2)));
347
348 out->ipp_stages = (void**)cvAlloc( ipp_datasize );
349 memset( out->ipp_stages, 0, ipp_datasize );
350
351 ipp_features = (CvRect*)cvAlloc( max_count*3*sizeof(ipp_features[0]) );
352 ipp_weights = (float*)cvAlloc( max_count*3*sizeof(ipp_weights[0]) );
353 ipp_thresholds = (float*)cvAlloc( max_count*sizeof(ipp_thresholds[0]) );
354 ipp_val1 = (float*)cvAlloc( max_count*sizeof(ipp_val1[0]) );
355 ipp_val2 = (float*)cvAlloc( max_count*sizeof(ipp_val2[0]) );
356 ipp_counts = (int*)cvAlloc( max_count*sizeof(ipp_counts[0]) );
357
358 for( i = 0; i < cascade->count; i++ )
359 {
360 CvHaarStageClassifier* stage_classifier = cascade->stage_classifier + i;
361 for( j = 0, k = 0; j < stage_classifier->count; j++ )
362 {
363 CvHaarClassifier* classifier = stage_classifier->classifier + j;
364 int rect_count = 2 + (classifier->haar_feature->rect[2].r.width != 0);
365
366 ipp_thresholds[j] = classifier->threshold[0];
367 ipp_val1[j] = classifier->alpha[0];
368 ipp_val2[j] = classifier->alpha[1];
369 ipp_counts[j] = rect_count;
370
371 for( l = 0; l < rect_count; l++, k++ )
372 {
373 ipp_features[k] = classifier->haar_feature->rect[l].r;
374 //ipp_features[k].y = orig_window_size.height - ipp_features[k].y - ipp_features[k].height;
375 ipp_weights[k] = classifier->haar_feature->rect[l].weight*ipp_weight_scale;
376 }
377 }
378
379 if( ippiHaarClassifierInitAlloc_32f( (IppiHaarClassifier_32f**)&out->ipp_stages[i],
380 (const IppiRect*)ipp_features, ipp_weights, ipp_thresholds,
381 ipp_val1, ipp_val2, ipp_counts, stage_classifier->count ) < 0 )
382 break;
383 }
384
385 if( i < cascade->count )
386 {
387 for( j = 0; j < i; j++ )
388 if( out->ipp_stages[i] )
389 ippiHaarClassifierFree_32f( (IppiHaarClassifier_32f*)out->ipp_stages[i] );
390 cvFree( &out->ipp_stages );
391 }
392 }
393#endif
394*/
395 cascade->hid_cascade = out;
396 assert( (char*)haar_node_ptr - (char*)out <= datasize );
397
398 cvFree( &ipp_features );
399 cvFree( &ipp_weights );
400 cvFree( &ipp_thresholds );
401 cvFree( &ipp_val1 );
402 cvFree( &ipp_val2 );
403 cvFree( &ipp_counts );
404
405 return out;
406}

422CV_IMPL void
423cvSetImagesForHaarClassifierCascade( CvHaarClassifierCascade* _cascade,
424 const CvArr* _sum,
425 const CvArr* _sqsum,
426 const CvArr* _tilted_sum,
427 double scale )
428{
429 CvMat sum_stub, *sum = (CvMat*)_sum;
430 CvMat sqsum_stub, *sqsum = (CvMat*)_sqsum;
431 CvMat tilted_stub, *tilted = (CvMat*)_tilted_sum;
432 CvHidHaarClassifierCascade* cascade;
433 int coi0 = 0, coi1 = 0;
434 int i;
435 CvRect equRect;
436 double weight_scale;
437
438 if( !CV_IS_HAAR_CLASSIFIER(_cascade) )
439 CV_Error( !_cascade ? CV_StsNullPtr : CV_StsBadArg, "Invalid classifier pointer" );
440
441 if( scale <= 0 )
442 CV_Error( CV_StsOutOfRange, "Scale must be positive" );
443
444 sum = cvGetMat( sum, &sum_stub, &coi0 );
445 sqsum = cvGetMat( sqsum, &sqsum_stub, &coi1 );
446
447 if( coi0 || coi1 )
448 CV_Error( CV_BadCOI, "COI is not supported" );
449
450 if( !CV_ARE_SIZES_EQ( sum, sqsum ))
451 CV_Error( CV_StsUnmatchedSizes, "All integral images must have the same size" );
452
453 if( CV_MAT_TYPE(sqsum->type) != CV_64FC1 ||
454 CV_MAT_TYPE(sum->type) != CV_32SC1 )
455 CV_Error( CV_StsUnsupportedFormat,
456 "Only (32s, 64f, 32s) combination of (sum,sqsum,tilted_sum) formats is allowed" );
457
458 if( !_cascade->hid_cascade )
459 icvCreateHidHaarClassifierCascade(_cascade);
460
461 cascade = _cascade->hid_cascade;
462
463 if( cascade->has_tilted_features )
464 {
465 tilted = cvGetMat( tilted, &tilted_stub, &coi1 );
466
467 if( CV_MAT_TYPE(tilted->type) != CV_32SC1 )
468 CV_Error( CV_StsUnsupportedFormat,
469 "Only (32s, 64f, 32s) combination of (sum,sqsum,tilted_sum) formats is allowed" );
470
471 if( sum->step != tilted->step )
472 CV_Error( CV_StsUnmatchedSizes,
473 "Sum and tilted_sum must have the same stride (step, widthStep)" );
474
475 if( !CV_ARE_SIZES_EQ( sum, tilted ))
476 CV_Error( CV_StsUnmatchedSizes, "All integral images must have the same size" );
477 cascade->tilted = *tilted;
478 }
479
480 _cascade->scale = scale;
481 _cascade->real_window_size.width = cvRound( _cascade->orig_window_size.width * scale );
482 _cascade->real_window_size.height = cvRound( _cascade->orig_window_size.height * scale );
483
484 cascade->sum = *sum;
485 cascade->sqsum = *sqsum;
486
487 equRect.x = equRect.y = cvRound(scale);
488 equRect.width = cvRound((_cascade->orig_window_size.width-2)*scale);
489 equRect.height = cvRound((_cascade->orig_window_size.height-2)*scale);
490 weight_scale = 1./(equRect.width*equRect.height);
491 cascade->inv_window_area = weight_scale;
492
493 cascade->p0 = sum_elem_ptr(*sum, equRect.y, equRect.x);
494 cascade->p1 = sum_elem_ptr(*sum, equRect.y, equRect.x + equRect.width );
495 cascade->p2 = sum_elem_ptr(*sum, equRect.y + equRect.height, equRect.x );
496 cascade->p3 = sum_elem_ptr(*sum, equRect.y + equRect.height,
497 equRect.x + equRect.width );
498
499 cascade->pq0 = sqsum_elem_ptr(*sqsum, equRect.y, equRect.x);
500 cascade->pq1 = sqsum_elem_ptr(*sqsum, equRect.y, equRect.x + equRect.width );
501 cascade->pq2 = sqsum_elem_ptr(*sqsum, equRect.y + equRect.height, equRect.x );
502 cascade->pq3 = sqsum_elem_ptr(*sqsum, equRect.y + equRect.height,
503 equRect.x + equRect.width );
504
505 /* init pointers in haar features according to real window size and
506 given image pointers */
507 for( i = 0; i < _cascade->count; i++ )
508 {
509 int j, k, l;
510 for( j = 0; j < cascade->stage_classifier[i].count; j++ )
511 {
512 for( l = 0; l < cascade->stage_classifier[i].classifier[j].count; l++ )
513 {
514 CvHaarFeature* feature =
515 &_cascade->stage_classifier[i].classifier[j].haar_feature[l];
516 /* CvHidHaarClassifier* classifier =
517 cascade->stage_classifier[i].classifier + j; */
518 CvHidHaarFeature* hidfeature =
519 &cascade->stage_classifier[i].classifier[j].node[l].feature;
520 double sum0 = 0, area0 = 0;
521 CvRect r[3];
522
523 int base_w = -1, base_h = -1;
524 int new_base_w = 0, new_base_h = 0;
525 int kx, ky;
526 int flagx = 0, flagy = 0;
527 int x0 = 0, y0 = 0;
528 int nr;
529
530 /* align blocks */
531 for( k = 0; k < CV_HAAR_FEATURE_MAX; k++ )
532 {
533 if( !hidfeature->rect[k].p0 )
534 break;
535 r[k] = feature->rect[k].r;
536 base_w = (int)CV_IMIN( (unsigned)base_w, (unsigned)(r[k].width-1) );
537 base_w = (int)CV_IMIN( (unsigned)base_w, (unsigned)(r[k].x - r[0].x-1) );
538 base_h = (int)CV_IMIN( (unsigned)base_h, (unsigned)(r[k].height-1) );
539 base_h = (int)CV_IMIN( (unsigned)base_h, (unsigned)(r[k].y - r[0].y-1) );
540 }
541
542 nr = k;
543
544 base_w += 1;
545 base_h += 1;
546 kx = r[0].width / base_w;
547 ky = r[0].height / base_h;
548
549 if( kx <= 0 )
550 {
551 flagx = 1;
552 new_base_w = cvRound( r[0].width * scale ) / kx;
553 x0 = cvRound( r[0].x * scale );
554 }
555
556 if( ky <= 0 )
557 {
558 flagy = 1;
559 new_base_h = cvRound( r[0].height * scale ) / ky;
560 y0 = cvRound( r[0].y * scale );
561 }
562
563 for( k = 0; k < nr; k++ )
564 {
565 CvRect tr;
566 double correction_ratio;
567
568 if( flagx )
569 {
570 tr.x = (r[k].x - r[0].x) * new_base_w / base_w + x0;
571 tr.width = r[k].width * new_base_w / base_w;
572 }
573 else
574 {
575 tr.x = cvRound( r[k].x * scale );
576 tr.width = cvRound( r[k].width * scale );
577 }
578
579 if( flagy )
580 {
581 tr.y = (r[k].y - r[0].y) * new_base_h / base_h + y0;
582 tr.height = r[k].height * new_base_h / base_h;
583 }
584 else
585 {
586 tr.y = cvRound( r[k].y * scale );
587 tr.height = cvRound( r[k].height * scale );
588 }
589
590#if CV_ADJUST_WEIGHTS
591 {
592 // RAINER START
593 const float orig_feature_size = (float)(feature->rect[k].r.width)*feature->rect[k].r.height;
594 const float orig_norm_size = (float)(_cascade->orig_window_size.width)*(_cascade->orig_window_size.height);
595 const float feature_size = float(tr.width*tr.height);
596 //const float normSize = float(equRect.width*equRect.height);
597 float target_ratio = orig_feature_size / orig_norm_size;
598 //float isRatio = featureSize / normSize;
599 //correctionRatio = targetRatio / isRatio / normSize;
600 correction_ratio = target_ratio / feature_size;
601 // RAINER END
602 }
603#else
604 correction_ratio = weight_scale * (!feature->tilted ? 1 : 0.5);
605#endif
606
607 if( !feature->tilted )
608 {
609 hidfeature->rect[k].p0 = sum_elem_ptr(*sum, tr.y, tr.x);
610 hidfeature->rect[k].p1 = sum_elem_ptr(*sum, tr.y, tr.x + tr.width);
611 hidfeature->rect[k].p2 = sum_elem_ptr(*sum, tr.y + tr.height, tr.x);
