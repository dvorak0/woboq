148int Sampler::overlap(Sampler &other)
149{
150 for (int i = 0; i < 4; i++) {
151 CvScalar p;
152 p = cvGet2D(other.perim, i, 0);
153 if (isinside((int)p.val[0], (int)p.val[1]))
154 return 1;
155 p = cvGet2D(perim, i, 0);
156 if (other.isinside((int)p.val[0], (int)p.val[1]))
157 return 1;
158 }
159 return 0;

180CvMat *Sampler::extract()
181{
182 // return a 10x10 CvMat for the current contents, 0 is black, 255 is white
183 // Sampler has (0,0) at bottom left, so invert Y
184 CvMat *r = cvCreateMat(10, 10, CV_8UC1);
185 for (int x = 0; x < 10; x++)
186 for (int y = 0; y < 10; y++)
187 *cvPtr2D(r, 9 - y, x) = (getpixel(x, y) < 128) ? 0 : 255;
188 return r;

210static void cfollow(CvMat *src, CvMat *dst)
211{
212 int sx, sy;
213 uchar *vpd = cvPtr2D(src, 0, 0);
214 for (sy = 0; sy < src->rows; sy++) {
215 short *wr = (short*)cvPtr2D(dst, sy, 0);
216 for (sx = 0; sx < src->cols; sx++) {
217 int x = sx;
218 int y = sy;
219 float e = 0;
220 int ontrack = true;
221 int dir;
222
223 while (ontrack) {
224 dir = vpd[y * src->step + x];
225 int xd = ((dir & 0xf) - 2);
226 int yd = ((dir >> 4) - 2);
227 e += (dir == 0x22) ? 999 : ((dir & 1) ? eincD : eincO);
228 x += xd;
229 y += yd;
230 if (e > 10.) {
231 float d = (float)(((x - sx) * (x - sx)) + ((y - sy) * (y - sy)));
232 ontrack = d > (e * e);
233 }
234 }
235 if ((24 <= e) && (e < 999)) {
236 // printf("sx=%d, sy=%d, x=%d, y=%d\n", sx, sy, x, y);
237 *wr++ = (short)(x - sx);
238 *wr++ = (short)(y - sy);
239 } else {
240 *wr++ = 0;
241 *wr++ = 0;
242 }
243 }
244 }

252static int decode(Sampler &sa, code &cc)
253{
254 uchar binary[8] = {0,0,0,0,0,0,0,0};
255 uchar b = 0;
256 int sum;
257
258 sum = 0;
259
260 for (int i = 0; i < 64; i++)
261 sum += sa.getpixel(1 + (i & 7), 1 + (i >> 3));
262 uchar mean = (uchar)(sum / 64);
263 for (int i = 0; i < 64; i++) {
264 b = (b << 1) + (sa.getpixel(pickup[i].x, pickup[i].y) <= mean);
265 if ((i & 7) == 7) {
266 binary[i >> 3] = b;
267 b = 0;
268 }
269 }
270
271 // Compute the 5 RS codewords for the 3 datawords
272
273 uchar c[5] = {0,0,0,0,0};
274 {
275 uchar a[5] = {228, 48, 15, 111, 62};
276 int k = 5;
277 for (int i = 0; i < 3; i++) {
278 uchar t = binary[i] ^ c[4];
279 for (int j = k - 1; j != -1; j--) {
280 if (t == 0)
281 c[j] = 0;
282 else
283 c[j] = gf256mul(t, a[j]);
284 if (j > 0)
285 c[j] = c[j - 1] ^ c[j];
286 }
287 }
288 }
289
290 if ((c[4] == binary[3]) &&
291 (c[3] == binary[4]) &&
292 (c[2] == binary[5]) &&
293 (c[1] == binary[6]) &&
294 (c[0] == binary[7])) {
295 uchar x = 0xff & (binary[0] - 1);
296 uchar y = 0xff & (binary[1] - 1);
297 uchar z = 0xff & (binary[2] - 1);
298 cc.msg[0] = x;
299 cc.msg[1] = y;
300 cc.msg[2] = z;
301 cc.msg[3] = 0;
302 cc.sa = sa;
303 cc.original = sa.extract();
304 return 1;
305 } else {
306 return 0;
307 }
308}

310static deque<CvPoint> trailto(CvMat *v, int x, int y, CvMat *terminal)
311{
312 CvPoint np;
313 /* Return the last 10th of the trail of points following v from (x,y)
314 * to terminal
315 */
316
317 int ex = x + ((short*)cvPtr2D(terminal, y, x))[0];
318 int ey = y + ((short*)cvPtr2D(terminal, y, x))[1];
319 deque<CvPoint> r;
320 while ((x != ex) || (y != ey)) {
321 np.x = x;
322 np.y = y;
323 r.push_back(np);
324 int dir = *cvPtr2D(v, y, x);
325 int xd = ((dir & 0xf) - 2);
326 int yd = ((dir >> 4) - 2);
327 x += xd;
328 y += yd;
329 }
330
331 int l = (int)(r.size() * 9 / 10);
332 while (l--)
333 r.pop_front();
334 return r;
335}

338deque <CvDataMatrixCode> cvFindDataMatrix(CvMat *im)
339{
340#if CV_SSE2
341 int r = im->rows;
342 int c = im->cols;
343
344#define SAMESIZE(nm, ty) CvMat *nm = cvCreateMat(r, c, ty);
345
346 SAMESIZE(thresh, CV_8UC1)
347 SAMESIZE(vecpic, CV_8UC1)
348 SAMESIZE(vc, CV_8UC1)
349 SAMESIZE(vcc, CV_8UC1)
350 SAMESIZE(cxy, CV_16SC2)
351 SAMESIZE(ccxy, CV_16SC2)
352
353 cvAdaptiveThreshold(im, thresh, 255.0, CV_ADAPTIVE_THRESH_MEAN_C, CV_THRESH_BINARY, 13);
354 {
355 int x, y;
356 int sstride = thresh->step;
357 int sw = thresh->cols; // source width
358 for (y = 2; y < thresh->rows - 2; y++) {
359 uchar *ps = cvPtr2D(thresh, y, 0);
360 uchar *pd = cvPtr2D(vecpic, y, 0);
361 uchar *pvc = cvPtr2D(vc, y, 0);
362 uchar *pvcc = cvPtr2D(vcc, y, 0);
363 for (x = 0; x < sw; x++) {
364 uchar v =
365 (0x01 & ps[-2 * sstride]) |
366 (0x02 & ps[-sstride + 1]) |
367 (0x04 & ps[2]) |
368 (0x08 & ps[sstride + 1]) |
369 (0x10 & ps[2 * sstride]) |
370 (0x20 & ps[sstride - 1]) |
371 (0x40 & ps[-2]) |
372 (0x80 & ps[-sstride -1]);
373 *pd++ = v;
374 *pvc++ = cblk[v];
375 *pvcc++ = ccblk[v];
376 ps++;
377 }
378 }
379 apron(vc);
380 apron(vcc);
381 }
382
383 cfollow(vc, cxy);
384 cfollow(vcc, ccxy);
385
386 deque <CvPoint> candidates;
387 {
388 int x, y;
389 int rows = cxy->rows;
390 int cols = cxy->cols;
391 for (y = 0; y < rows; y++) {
392 const short *cd = (const short*)cvPtr2D(cxy, y, 0);
393 const short *ccd = (const short*)cvPtr2D(ccxy, y, 0);
394 for (x = 0; x < cols; x += 4, cd += 8, ccd += 8) {
395 __m128i v = _mm_loadu_si128((const __m128i*)cd);
396 __m128 cyxyxA = _mm_cvtepi32_ps(_mm_srai_epi32(_mm_unpacklo_epi16(v, v), 16));
397 __m128 cyxyxB = _mm_cvtepi32_ps(_mm_srai_epi32(_mm_unpackhi_epi16(v, v), 16));
398 __m128 cx = _mm_shuffle_ps(cyxyxA, cyxyxB, _MM_SHUFFLE(2, 0, 2, 0));
399 __m128 cy = _mm_shuffle_ps(cyxyxA, cyxyxB, _MM_SHUFFLE(3, 1, 3, 1));
400 __m128 cmag = _mm_sqrt_ps(_mm_add_ps(_mm_mul_ps(cx, cx), _mm_mul_ps(cy, cy)));
401 __m128 crmag = _mm_rcp_ps(cmag);
402 __m128 ncx = _mm_mul_ps(cx, crmag);
403 __m128 ncy = _mm_mul_ps(cy, crmag);
404
405 v = _mm_loadu_si128((const __m128i*)ccd);
406 __m128 ccyxyxA = _mm_cvtepi32_ps(_mm_srai_epi32(_mm_unpacklo_epi16(v, v), 16));
407 __m128 ccyxyxB = _mm_cvtepi32_ps(_mm_srai_epi32(_mm_unpackhi_epi16(v, v), 16));
408 __m128 ccx = _mm_shuffle_ps(ccyxyxA, ccyxyxB, _MM_SHUFFLE(2, 0, 2, 0));
409 __m128 ccy = _mm_shuffle_ps(ccyxyxA, ccyxyxB, _MM_SHUFFLE(3, 1, 3, 1));
410 __m128 ccmag = _mm_sqrt_ps(_mm_add_ps(_mm_mul_ps(ccx, ccx), _mm_mul_ps(ccy, ccy)));
411 __m128 ccrmag = _mm_rcp_ps(ccmag);
412 __m128 nccx = _mm_mul_ps(ccx, ccrmag);
413 __m128 nccy = _mm_mul_ps(ccy, ccrmag);
414
415 __m128 dot = _mm_mul_ps(_mm_mul_ps(ncx, nccx), _mm_mul_ps(ncy, nccy));
416 // iscand = (cmag > 30) & (ccmag > 30) & (numpy.minimum(cmag, ccmag) * 1.1 > numpy.maximum(cmag, ccmag)) & (abs(dot) < 0.25)
417 __m128 iscand = _mm_and_ps(_mm_cmpgt_ps(cmag, Kf(30)), _mm_cmpgt_ps(ccmag, Kf(30)));
418
419 iscand = _mm_and_ps(iscand, _mm_cmpgt_ps(_mm_mul_ps(_mm_min_ps(cmag, ccmag), Kf(1.1f)), _mm_max_ps(cmag, ccmag)));
420 iscand = _mm_and_ps(iscand, _mm_cmplt_ps(_mm_abs_ps(dot), Kf(0.25f)));
421
422 unsigned int CV_DECL_ALIGNED(16) result[4];
423 _mm_store_ps((float*)result, iscand);
424 int ix;
425 CvPoint np;
426 for (ix = 0; ix < 4; ix++) {
427 if (result[ix]) {
428 np.x = x + ix;
429 np.y = y;
430 candidates.push_back(np);
431 }
432 }
433 }
434 }
435 }
436
437 deque <code> codes;
438 size_t i, j, k;
439 while (!candidates.empty()) {
440 CvPoint o = candidates.front();
441 candidates.pop_front();
442 deque<CvPoint> ptc = trailto(vc, o.x, o.y, cxy);
443 deque<CvPoint> ptcc = trailto(vcc, o.x, o.y, ccxy);
444 for (j = 0; j < ptc.size(); j++) {
445 for (k = 0; k < ptcc.size(); k++) {
446 code cc;
447 Sampler sa(im, o, ptc[j], ptcc[k]);
448 for (i = 0; i < codes.size(); i++) {
449 if (sa.overlap(codes[i].sa))
450 {
451 cvReleaseMat(&sa.perim);
452 goto endo;
453 }
454 }
455 if (codes.size() > 0) {
456 //printf("searching for more\n");
457 }
458 if (decode(sa, cc)) {
459 codes.push_back(cc);
460 goto endo;
461 }
462
463 cvReleaseMat(&sa.perim);
464 }
465 }
466endo: ; // end search for this o
467 }
468
469 cvReleaseMat(&thresh);
470 cvReleaseMat(&vecpic);
471 cvReleaseMat(&vc);
472 cvReleaseMat(&vcc);
473 cvReleaseMat(&cxy);

476 deque <CvDataMatrixCode> rc;

483 }
484 return rc;

497void findDataMatrix(InputArray _image,
498 vector<string>& codes,
499 OutputArray _corners,
500 OutputArrayOfArrays _dmtx)
501{
502 Mat image = _image.getMat();
503 CvMat m(image);
504 deque <CvDataMatrixCode> rc = cvFindDataMatrix(&m);
505 int i, n = (int)rc.size();
506 Mat corners;
507
508 if( _corners.needed() )
509 {
510 _corners.create(n, 4, CV_32SC2);
511 corners = _corners.getMat();
512 }
513
514 if( _dmtx.needed() )
515 _dmtx.create(n, 1, CV_8U);
516
517 codes.resize(n);
518
519 for( i = 0; i < n; i++ )
520 {
521 CvDataMatrixCode& rc_i = rc[i];
522 codes[i] = string(rc_i.msg);
523
524 if( corners.data )
525 {
526 const Point* srcpt = (Point*)rc_i.corners->data.ptr;
527 Point* dstpt = (Point*)corners.ptr(i);
528 for( int k = 0; k < 4; k++ )
529 dstpt[k] = srcpt[k];
530 }
531 cvReleaseMat(&rc_i.corners);
532
533 if( _dmtx.needed() )
534 {
535 _dmtx.create(rc_i.original->rows, rc_i.original->cols, rc_i.original->type, i);
536 Mat dst = _dmtx.getMat(i);
537 Mat(rc_i.original).copyTo(dst);
538 }
539 cvReleaseMat(&rc_i.original);
540 }
541}
