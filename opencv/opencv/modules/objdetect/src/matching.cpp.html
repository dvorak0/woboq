13/*
14// Function for convolution computation
15//
16// INPUT
17// Fi - filter object
18// map - feature map
19// OUTPUT
20// f - the convolution
21// RESULT
22// Error status
23*/
24int convolution(const CvLSVMFilterObject *Fi, const CvLSVMFeatureMap *map, float *f)
25{
26 int n1, m1, n2, m2, p, /*size,*/ diff1, diff2;
27 int i1, i2, j1, j2, k;
28 float tmp_f1, tmp_f2, tmp_f3, tmp_f4;
29 float *pMap = NULL;
30 float *pH = NULL;
31
32 n1 = map->sizeY;
33 m1 = map->sizeX;
34 n2 = Fi->sizeY;
35 m2 = Fi->sizeX;
36 p = map->numFeatures;
37
38 diff1 = n1 - n2 + 1;
39 diff2 = m1 - m2 + 1;
40 //size = diff1 * diff2;
41 for (j1 = diff2 - 1; j1 >= 0; j1--)
42 {
43
44 for (i1 = diff1 - 1; i1 >= 0; i1--)
45 {
46 tmp_f1 = 0.0f;
47 tmp_f2 = 0.0f;
48 tmp_f3 = 0.0f;
49 tmp_f4 = 0.0f;
50 for (i2 = 0; i2 < n2; i2++)
51 {
52 for (j2 = 0; j2 < m2; j2++)
53 {
54 pMap = map->map + (i1 + i2) * m1 * p + (j1 + j2) * p;//sm2
55 pH = Fi->H + (i2 * m2 + j2) * p;//sm2
56 for (k = 0; k < p/4; k++)
57 {
58
59 tmp_f1 += pMap[4*k]*pH[4*k];//sm2
60 tmp_f2 += pMap[4*k+1]*pH[4*k+1];
61 tmp_f3 += pMap[4*k+2]*pH[4*k+2];
62 tmp_f4 += pMap[4*k+3]*pH[4*k+3];
63 }
64
65 if (p%4==1)
66 {
67 tmp_f1 += pH[p-1]*pMap[p-1];
68 }
69 else
70 {
71 if (p%4==2)
72 {
73 tmp_f1 += pH[p-2]*pMap[p-2] + pH[p-1]*pMap[p-1];
74 }
75 else
76 {
77 if (p%4==3)
78 {
79 tmp_f1 += pH[p-3]*pMap[p-3] + pH[p-2]*pMap[p-2] + pH[p-1]*pMap[p-1];
80 }
81 }
82 }
83
84 }
85 }
86 f[i1 * diff2 + j1] = tmp_f1 + tmp_f2 + tmp_f3 + tmp_f4;//sm1
87 }
88 }
89 return LATENT_SVM_OK;
90}

92/*
93// Computation multiplication of FFT images
94//
95// API
96// int fftImagesMulti(float *fftImage1, float *fftImage2, int numRows, int numColls,
97 float *multi);
98// INPUT
99// fftImage1 - first fft image
100// fftImage2 - second fft image
101// (numRows, numColls) - image dimesions
102// OUTPUT
103// multi - multiplication
104// RESULT
105// Error status
106*/
107int fftImagesMulti(float *fftImage1, float *fftImage2, int numRows, int numColls,
108 float *multi)
109{
110 int i, index, size;
111 size = numRows * numColls;
112 for (i = 0; i < size; i++)
113 {
114 index = 2 * i;
115 multi[index] = fftImage1[index] * fftImage2[index] -
116 fftImage1[index + 1] * fftImage2[index + 1];
117 multi[index + 1] = fftImage1[index] * fftImage2[index + 1] +
118 fftImage1[index + 1] * fftImage2[index];
119 }
120 return LATENT_SVM_OK;
121}

124// Turnover filter matrix for the single feature
125//
126// API
127// int rot2PI(float *filter, int dimX, int dimY, float *rot2PIFilter,
128 int p, int shift);
129// INPUT
130// filter - filter weight matrix
131// (dimX, dimY) - dimension of filter matrix
132// p - number of features
133// shift - number of feature (or channel)
134// OUTPUT
135// rot2PIFilter - rotated matrix
136// RESULT
137// Error status
138*/
139int rot2PI(float *filter, int dimX, int dimY, float *rot2PIFilter,
140 int p, int shift)
141{
142 int i, size;
143 size = dimX * dimY;
144 for (i = 0; i < size; i++)
145 {
146 rot2PIFilter[i] = filter[(size - i - 1) * p + shift];
147 }
148 return LATENT_SVM_OK;

154// API
155// int addNullableBars(float *rot2PIFilter, int dimX, int dimY,

157// INPUT
158// rot2PIFilter - filter matrix for the single feature that was rotated

160// (newDimX, newDimY)- dimension of feature map for the single feature
161// OUTPUT
162// newFilter - filter matrix with nullable bars
163// RESULT
164// Error status
165*/
166int addNullableBars(float *rot2PIFilter, int dimX, int dimY,
167 float *newFilter, int newDimX, int newDimY)
168{
169 int size, i, j;
170 size = newDimX * newDimY;
171 for (i = 0; i < size; i++)
172 {
173 newFilter[2 * i] = 0.0;
174 newFilter[2 * i + 1] = 0.0;
175 }
176 for (i = 0; i < dimY; i++)
177 {
178 for (j = 0; j < dimX; j++)
179 {
180 newFilter[2 * (i * newDimX + j)] = rot2PIFilter[i * dimX + j];
181 }
182 }
183 return LATENT_SVM_OK;

188//

192 fftImage **image);

194// filter - filter object
195// (mapDimX, mapDimY)- dimension of feature map
196// OUTPUT

201int getFFTImageFilterObject(const CvLSVMFilterObject *filter,

205 int i, mapSize, filterSize, res;

207

213 return res;

223 }
224 free(newFilter);

259/*
260// Function for convolution computation using FFT
261//
262// API
263// int convFFTConv2d(const fftImage *featMapImage, const fftImage *filterImage,
264 int filterDimX, int filterDimY, float **conv);
265// INPUT
266// featMapImage - feature map image
267// filterImage - filter image
268// (filterDimX,filterDimY) - filter dimension
269// OUTPUT
270// conv - the convolution
271// RESULT
272// Error status
273*/
274int convFFTConv2d(const CvLSVMFftImage *featMapImage, const CvLSVMFftImage *filterImage,
275 int filterDimX, int filterDimY, float **conv)
276{
277 int i, j, size, diffX, diffY, index;
278 float *imagesMult, *imagesMultRes, *fconv;
279 size = 2 * featMapImage->dimX * featMapImage->dimY;
280 imagesMult = (float *)malloc(sizeof(float) * size);
281 imagesMultRes = (float *)malloc(sizeof(float) * size);
282 fftImagesMulti(featMapImage->channels[0], filterImage->channels[0],
283 featMapImage->dimY, featMapImage->dimX, imagesMultRes);
284 for (i = 1; (i < (int)featMapImage->numFeatures) && (i < (int)filterImage->numFeatures); i++)
285 {
286 fftImagesMulti(featMapImage->channels[i],filterImage->channels[i],
287 featMapImage->dimY, featMapImage->dimX, imagesMult);
288 for (j = 0; j < size; j++)
289 {
290 imagesMultRes[j] += imagesMult[j];
291 }
292 }
293 fconv = (float *)malloc(sizeof(float) * size);
294 fftInverse2d(imagesMultRes, fconv, featMapImage->dimY, featMapImage->dimX);
295 diffX = featMapImage->dimX - filterDimX + 1;
296 diffY = featMapImage->dimY - filterDimY + 1;
297 *conv = (float *)malloc(sizeof(float) * (diffX * diffY));
298 for (i = 0; i < diffY; i++)
299 {
300 for (j = 0; j < diffX; j++)
301 {
302 index = (i + filterDimY - 1) * featMapImage->dimX +
303 (j + filterDimX - 1);
304 (*conv)[i * diffX + j] = fconv[2 * index];
305 }
306 }
307 free(imagesMult);
308 free(imagesMultRes);
309 free(fconv);
310 return LATENT_SVM_OK;
311}

313/*
314// Computation objective function D according the original paper
315//
316// API
317// int filterDispositionLevel(const CvLSVMFilterObject *Fi, const featurePyramid *H,
318 int level, float **scoreFi,
319 int **pointsX, int **pointsY);
320// INPUT
321// Fi - filter object (weights and coefficients of penalty
322 function that are used in this routine)
323// H - feature pyramid
324// level - level number
325// OUTPUT
326// scoreFi - values of distance transform on the level at all positions
327// (pointsX, pointsY)- positions that correspond to the maximum value
328 of distance transform at all grid nodes
329// RESULT
330// Error status
331*/
332int filterDispositionLevel(const CvLSVMFilterObject *Fi, const CvLSVMFeatureMap *pyramid,
333 float **scoreFi,
334 int **pointsX, int **pointsY)
335{
336 int n1, m1, n2, m2, /*p,*/ size, diff1, diff2;
337 float *f;
338 int i1, j1;
339 int res;
340
341 n1 = pyramid->sizeY;
342 m1 = pyramid->sizeX;
343 n2 = Fi->sizeY;
344 m2 = Fi->sizeX;
345 //p = pyramid->numFeatures;
346 (*scoreFi) = NULL;
347 (*pointsX) = NULL;
348 (*pointsY) = NULL;
349
350 // Processing the situation when part filter goes
351 // beyond the boundaries of the block set
352 if (n1 < n2 || m1 < m2)
353 {
354 return FILTER_OUT_OF_BOUNDARIES;
355 } /* if (n1 < n2 || m1 < m2) */
356
357 // Computation number of positions for the filter
358 diff1 = n1 - n2 + 1;
359 diff2 = m1 - m2 + 1;
360 size = diff1 * diff2;
361
362 // Allocation memory for additional array (must be free in this function)
363 f = (float *)malloc(sizeof(float) * size);
364 // Allocation memory for arrays for saving decisions
365 (*scoreFi) = (float *)malloc(sizeof(float) * size);
366 (*pointsX) = (int *)malloc(sizeof(int) * size);
367 (*pointsY) = (int *)malloc(sizeof(int) * size);
368
369 // Consruction values of the array f
370 // (a dot product vectors of feature map and weights of the filter)
371 res = convolution(Fi, pyramid, f);
372 if (res != LATENT_SVM_OK)
373 {
374 free(f);
375 free(*scoreFi);
376 free(*pointsX);
377 free(*pointsY);
378 return res;
379 }
380
381 // TODO: necessary to change
382 for (i1 = 0; i1 < diff1; i1++)
383 {
384 for (j1 = 0; j1 < diff2; j1++)
385 {
386 f[i1 * diff2 + j1] *= (-1);
387 }
388 }
389
390 // Decision of the general distance transform task
391 DistanceTransformTwoDimensionalProblem(f, diff1, diff2, Fi->fineFunction,
392 (*scoreFi), (*pointsX), (*pointsY));
393
394 // Release allocated memory
395 free(f);
396 return LATENT_SVM_OK;
397}

581/*
582// Computation the maximum of the score function at the level
583//
584// API
585// int maxFunctionalScoreFixedLevel(const CvLSVMFilterObject **all_F, int n,
586 const featurePyramid *H,
587 int level, float b,
588 int maxXBorder, int maxYBorder,
589 float *score, CvPoint **points, int *kPoints,
590 CvPoint ***partsDisplacement);
591// INPUT
592// all_F - the set of filters (the first element is root filter,
593 the other - part filters)
594// n - the number of part filters
595// H - feature pyramid
596// level - feature pyramid level for computation maximum score
597// b - linear term of the score function
598// maxXBorder - the largest root filter size (X-direction)
599// maxYBorder - the largest root filter size (Y-direction)
600// OUTPUT
601// score - the maximum of the score function at the level
602// points - the set of root filter positions (in the block space)
603// levels - the set of levels
604// kPoints - number of root filter positions
605// partsDisplacement - displacement of part filters (in the block space)
606// RESULT
607// Error status
608*/
609int maxFunctionalScoreFixedLevel(const CvLSVMFilterObject **all_F, int n,
610 const CvLSVMFeaturePyramid *H,
611 int level, float b,
612 int maxXBorder, int maxYBorder,
613 float *score, CvPoint **points,
614 int *kPoints, CvPoint ***partsDisplacement)
615{
616 int i, j, k, dimX, dimY, nF0, mF0/*, p*/;
617 int diff1, diff2, index, last, partsLevel;
618 CvLSVMFilterDisposition **disposition;
619 float *f;
620 float *scores;
621 float sumScorePartDisposition, maxScore;
622 int res;
623 CvLSVMFeatureMap *map;
624#ifdef FFT_CONV
625 CvLSVMFftImage *rootFilterImage, *mapImage;
626#else
627#endif
628
629 /*
630 // DEBUG variables
631 FILE *file;
632 char *tmp;
633 char buf[40] = "..\\Data\\score\\score", buf1[10] = ".csv";
634 tmp = (char *)malloc(sizeof(char) * 80);
635 itoa(level, tmp, 10);
636 strcat(tmp, buf1);
637 //*/
638
639 // Feature map matrix dimension on the level
640 dimX = H->pyramid[level]->sizeX;
641 dimY = H->pyramid[level]->sizeY;
642
643 // Number of features
644 //p = H->pyramid[level]->numFeatures;
645
646 // Getting dimension of root filter
647 nF0 = all_F[0]->sizeY;
648 mF0 = all_F[0]->sizeX;
649 // Processing the situation when root filter goes
650 // beyond the boundaries of the block set
651 if (nF0 > dimY || mF0 > dimX)
652 {
653 return LATENT_SVM_FAILED_SUPERPOSITION;
654 }
655
656 diff1 = dimY - nF0 + 1;
657 diff2 = dimX - mF0 + 1;
658
659 // Allocation memory for saving values of function D
660 // on the level for each part filter
661 disposition = (CvLSVMFilterDisposition **)malloc(sizeof(CvLSVMFilterDisposition *) * n);
662 for (i = 0; i < n; i++)
663 {
664 disposition[i] = (CvLSVMFilterDisposition *)malloc(sizeof(CvLSVMFilterDisposition));
665 }
666
667 // Allocation memory for values of score function for each block on the level
668 scores = (float *)malloc(sizeof(float) * (diff1 * diff2));
669
670 // A dot product vectors of feature map and weights of root filter
671#ifdef FFT_CONV
672 getFFTImageFeatureMap(H->pyramid[level], &mapImage);
673 getFFTImageFilterObject(all_F[0], H->pyramid[level]->sizeX, H->pyramid[level]->sizeY, &rootFilterImage);
674 res = convFFTConv2d(mapImage, rootFilterImage, all_F[0]->sizeX, all_F[0]->sizeY, &f);
675 freeFFTImage(&mapImage);
676 freeFFTImage(&rootFilterImage);
677#else
678 // Allocation memory for saving a dot product vectors of feature map and
679 // weights of root filter
680 f = (float *)malloc(sizeof(float) * (diff1 * diff2));
681 // A dot product vectors of feature map and weights of root filter
682 res = convolution(all_F[0], H->pyramid[level], f);
683#endif
684 if (res != LATENT_SVM_OK)
685 {
686 free(f);
687 free(scores);
688 for (i = 0; i < n; i++)
689 {
690 free(disposition[i]);
691 }
692 free(disposition);
693 return res;
694 }
695
696 // Computation values of function D for each part filter
697 // on the level (level - LAMBDA)
698 partsLevel = level - LAMBDA;
699 // For feature map at the level 'partsLevel' add nullable border
700 map = featureMapBorderPartFilter(H->pyramid[partsLevel],
701 maxXBorder, maxYBorder);
702
703 // Computation the maximum of score function
704 sumScorePartDisposition = 0.0;
705#ifdef FFT_CONV
706 getFFTImageFeatureMap(map, &mapImage);
707 for (k = 1; k <= n; k++)

1376// API

1378 const int n, const int bx, const int by,

1380 int **processingLevels)
1381// INPUT
1382// H - feature pyramid
1383// all_F - the set of filters (the first element is root filter,
1384 the other - part filters)
1385// n - the number of part filters
1386// bx - size of nullable border (X direction)
1387// by - size of nullable border (Y direction)
1388// threadsNum - number of threads that will be created in TBB version
1389// OUTPUT
1390// kLevels - array that contains number of levels processed
1391 by each thread
1392// processingLevels - array that contains lists of levels processed
1393 by each thread
1394// RESULT
1395// Error status
1396*/
1397static int createSchedule(const CvLSVMFeaturePyramid *H, const CvLSVMFilterObject **all_F,
1398 const int n, const int bx, const int by,
1399 const int threadsNum, int *kLevels, int **processingLevels)
1400{
1401 int rootFilterDim, sumPartFiltersDim, i, numLevels, dbx, dby;
1402 int j, minValue, argMin, lambda, maxValue, k;
1403 int *dotProd, *weights, *disp;
1404 if (H == NULL || all_F == NULL)
1405 {
1406 return LATENT_SVM_TBB_SCHEDULE_CREATION_FAILED;
1407 }
1408 // Number of feature vectors in root filter
1409 rootFilterDim = all_F[0]->sizeX * all_F[0]->sizeY;
1410 // Number of feature vectors in all part filters
1411 sumPartFiltersDim = 0;
1412 for (i = 1; i <= n; i++)
1413 {
1414 sumPartFiltersDim += all_F[i]->sizeX * all_F[i]->sizeY;
1415 }
1416 // Number of levels which are used for computation of score function
1417 numLevels = H->numLevels - LAMBDA;
1418 // Allocation memory for saving number of dot products that will be
1419 // computed for each level of feature pyramid
1420 dotProd = (int *)malloc(sizeof(int) * numLevels);
1421 // Size of nullable border that's used in computing convolution
1422 // of feature map with part filter
1423 dbx = 2 * bx;
1424 dby = 2 * by;
1425 lambda = LAMBDA;
1426 for (i = 0; i < numLevels; i++)
1427 {
1428 dotProd[i] = H->pyramid[i + lambda]->sizeX *
1429 H->pyramid[i + lambda]->sizeY * rootFilterDim +
1430 (H->pyramid[i]->sizeX + dbx) *
1431 (H->pyramid[i]->sizeY + dby) * sumPartFiltersDim;
1432 }
1433 // Allocation memory for saving dot product number performed by each thread
1434 weights = (int *)malloc(sizeof(int) * threadsNum);
1435 // Allocation memory for saving dispertion
1436 disp = (int *)malloc(sizeof(int) * threadsNum);
1437 // At the first step we think of first threadsNum levels will be processed
1438 // by different threads
1439 for (i = 0; i < threadsNum; i++)
1440 {
1441 kLevels[i] = 1;
1442 weights[i] = dotProd[i];
1443 disp[i] = 0;
1444 }
1445 // Computation number of levels that will be processed by each thread
1446 for (i = threadsNum; i < numLevels; i++)
1447 {
1448 // Search number of thread that will process level number i
1449 for (j = 0; j < threadsNum; j++)
1450 {
1451 weights[j] += dotProd[i];
1452 minValue = weights[0];
1453 maxValue = weights[0];
1454 for (k = 1; k < threadsNum; k++)
1455 {
1456 minValue = min(minValue, weights[k]);
1457 maxValue = max(maxValue, weights[k]);
1458 }
1459 disp[j] = maxValue - minValue;
1460 weights[j] -= dotProd[i];
1461 }
1462 minValue = disp[0];
1463 argMin = 0;
1464 for (j = 1; j < threadsNum; j++)
1465 {
1466 if (disp[j] < minValue)
1467 {
1468 minValue = disp[j];
1469 argMin = j;
1470 }
1471 }
1472 // Addition new level
1473 kLevels[argMin]++;
1474 weights[argMin] += dotProd[i];
1475 }
1476 for (i = 0; i < threadsNum; i++)
1477 {
1478 // Allocation memory for saving list of levels for each level
1479 processingLevels[i] = (int *)malloc(sizeof(int) * kLevels[i]);
1480 // At the first step we think of first threadsNum levels will be processed
1481 // by different threads
1482 processingLevels[i][0] = lambda + i;
1483 kLevels[i] = 1;
1484 weights[i] = dotProd[i];
1485 }
1486 // Creating list of levels
1487 for (i = threadsNum; i < numLevels; i++)
1488 {
1489 for (j = 0; j < threadsNum; j++)
1490 {
1491 weights[j] += dotProd[i];
1492 minValue = weights[0];
1493 maxValue = weights[0];
1494 for (k = 1; k < threadsNum; k++)
1495 {
1496 minValue = min(minValue, weights[k]);
1497 maxValue = max(maxValue, weights[k]);
1498 }
1499 disp[j] = maxValue - minValue;
1500 weights[j] -= dotProd[i];
1501 }
1502 minValue = disp[0];
1503 argMin = 0;
1504 for (j = 1; j < threadsNum; j++)
1505 {
1506 if (disp[j] < minValue)
1507 {
1508 minValue = disp[j];
1509 argMin = j;
1510 }
1511 }
1512 processingLevels[argMin][kLevels[argMin]] = lambda + i;
1513 kLevels[argMin]++;
1514 weights[argMin] += dotProd[i];
1515 }
1516 // Release allocated memory
1517 free(weights);
1518 free(dotProd);
1519 free(disp);
1520 return LATENT_SVM_OK;
1521}

1698/*
1699// Perform non-maximum suppression algorithm (described in original paper)
1700// to remove "similar" bounding boxes
1701//
1702// API
1703// int nonMaximumSuppression(int numBoxes, const CvPoint *points,
1704 const CvPoint *oppositePoints, const float *score,
1705 float overlapThreshold,
1706 int *numBoxesOut, CvPoint **pointsOut,
1707 CvPoint **oppositePointsOut, float **scoreOut);
1708// INPUT
1709// numBoxes - number of bounding boxes
1710// points - array of left top corner coordinates
1711// oppositePoints - array of right bottom corner coordinates
1712// score - array of detection scores
1713// overlapThreshold - threshold: bounding box is removed if overlap part
1714 is greater than passed value
1715// OUTPUT
1716// numBoxesOut - the number of bounding boxes algorithm returns
1717// pointsOut - array of left top corner coordinates
1718// oppositePointsOut - array of right bottom corner coordinates
1719// scoreOut - array of detection scores
1720// RESULT
1721// Error status
1722*/
1723int nonMaximumSuppression(int numBoxes, const CvPoint *points,
1724 const CvPoint *oppositePoints, const float *score,
1725 float overlapThreshold,
1726 int *numBoxesOut, CvPoint **pointsOut,
1727 CvPoint **oppositePointsOut, float **scoreOut)
1728{
1729 int i, j, index;
1730 float* box_area = (float*)malloc(numBoxes * sizeof(float));
1731 int* indices = (int*)malloc(numBoxes * sizeof(int));
1732 int* is_suppressed = (int*)malloc(numBoxes * sizeof(int));
1733
1734 for (i = 0; i < numBoxes; i++)
1735 {
1736 indices[i] = i;
1737 is_suppressed[i] = 0;
1738 box_area[i] = (float)( (oppositePoints[i].x - points[i].x + 1) *
1739 (oppositePoints[i].y - points[i].y + 1));
1740 }
1741
1742 sort(numBoxes, score, indices);
1743 for (i = 0; i < numBoxes; i++)
1744 {
1745 if (!is_suppressed[indices[i]])
1746 {
1747 for (j = i + 1; j < numBoxes; j++)
1748 {
1749 if (!is_suppressed[indices[j]])
1750 {
1751 int x1max = max(points[indices[i]].x, points[indices[j]].x);
1752 int x2min = min(oppositePoints[indices[i]].x, oppositePoints[indices[j]].x);
1753 int y1max = max(points[indices[i]].y, points[indices[j]].y);
1754 int y2min = min(oppositePoints[indices[i]].y, oppositePoints[indices[j]].y);
1755 int overlapWidth = x2min - x1max + 1;
1756 int overlapHeight = y2min - y1max + 1;
1757 if (overlapWidth > 0 && overlapHeight > 0)
1758 {
1759 float overlapPart = (overlapWidth * overlapHeight) / box_area[indices[j]];
1760 if (overlapPart > overlapThreshold)
1761 {
1762 is_suppressed[indices[j]] = 1;
1763 }
1764 }
1765 }
1766 }
1767 }
1768 }
1769
1770 *numBoxesOut = 0;
1771 for (i = 0; i < numBoxes; i++)
1772 {
1773 if (!is_suppressed[i]) (*numBoxesOut)++;
1774 }
1775
1776 *pointsOut = (CvPoint *)malloc((*numBoxesOut) * sizeof(CvPoint));
1777 *oppositePointsOut = (CvPoint *)malloc((*numBoxesOut) * sizeof(CvPoint));
1778 *scoreOut = (float *)malloc((*numBoxesOut) * sizeof(float));
1779 index = 0;
1780 for (i = 0; i < numBoxes; i++)
1781 {
1782 if (!is_suppressed[indices[i]])
1783 {
1784 (*pointsOut)[index].x = points[indices[i]].x;
1785 (*pointsOut)[index].y = points[indices[i]].y;
1786 (*oppositePointsOut)[index].x = oppositePoints[indices[i]].x;
1787 (*oppositePointsOut)[index].y = oppositePoints[indices[i]].y;
1788 (*scoreOut)[index] = score[indices[i]];
1789 index++;
1790 }
1791
1792 }
1793
1794 free(indices);
1795 free(box_area);
1796 free(is_suppressed);
1797
1798 return LATENT_SVM_OK;
1799}
