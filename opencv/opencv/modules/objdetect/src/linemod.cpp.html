54 * \brief Get the label [0,8) of the single bit set in quantized.

57{
58 switch (quantized)
59 {
60 case 1: return 0;

64 case 16: return 4;
65 case 32: return 5;
66 case 64: return 6;
67 case 128: return 7;
68 default:
69 CV_Error(CV_StsBadArg, "Invalid value of quantized parameter");
70 return -1; //avoid warning
71 }

87/**
88 * \brief Crop a set of overlapping templates from different modalities.
89 *
90 * \param[in,out] templates Set of templates representing the same object view.
91 *
92 * \return The bounding box of all the templates in original image coordinates.
93 */
94static Rect cropTemplates(std::vector<Template>& templates)
95{
96 int min_x = std::numeric_limits<int>::max();
97 int min_y = std::numeric_limits<int>::max();
98 int max_x = std::numeric_limits<int>::min();
99 int max_y = std::numeric_limits<int>::min();
100
101 // First pass: find min/max feature x,y over all pyramid levels and modalities
102 for (int i = 0; i < (int)templates.size(); ++i)
103 {
104 Template& templ = templates[i];
105
106 for (int j = 0; j < (int)templ.features.size(); ++j)
107 {
108 int x = templ.features[j].x << templ.pyramid_level;
109 int y = templ.features[j].y << templ.pyramid_level;
110 min_x = std::min(min_x, x);
111 min_y = std::min(min_y, y);
112 max_x = std::max(max_x, x);
113 max_y = std::max(max_y, y);
114 }
115 }
116
117 /// @todo Why require even min_x, min_y?
118 if (min_x % 2 == 1) --min_x;
119 if (min_y % 2 == 1) --min_y;
120
121 // Second pass: set width/height and shift all feature positions
122 for (int i = 0; i < (int)templates.size(); ++i)
123 {
124 Template& templ = templates[i];
125 templ.width = (max_x - min_x) >> templ.pyramid_level;
126 templ.height = (max_y - min_y) >> templ.pyramid_level;
127 int offset_x = min_x >> templ.pyramid_level;
128 int offset_y = min_y >> templ.pyramid_level;
129
130 for (int j = 0; j < (int)templ.features.size(); ++j)
131 {
132 templ.features[j].x -= offset_x;
133 templ.features[j].y -= offset_y;
134 }
135 }
136
137 return Rect(min_x, min_y, max_x - min_x, max_y - min_y);
138}

173void QuantizedPyramid::selectScatteredFeatures(const std::vector<Candidate>& candidates,
174 std::vector<Feature>& features,
175 size_t num_features, float distance)
176{
177 features.clear();
178 float distance_sq = CV_SQR(distance);
179 int i = 0;
180 while (features.size() < num_features)
181 {
182 Candidate c = candidates[i];
183
184 // Add if sufficient distance away from any previously chosen feature
185 bool keep = true;
186 for (int j = 0; (j < (int)features.size()) && keep; ++j)
187 {
188 Feature f = features[j];
189 keep = CV_SQR(c.f.x - f.x) + CV_SQR(c.f.y - f.y) >= distance_sq;
190 }
191 if (keep)
192 features.push_back(c.f);
193
194 if (++i == (int)candidates.size())
195 {
196 // Start back at beginning, and relax required distance
197 i = 0;
198 distance -= 1.0f;
199 distance_sq = CV_SQR(distance);
200 }
201 }

222void colormap(const Mat& quantized, Mat& dst)
223{
224 std::vector<Vec3b> lut(8);
225 lut[0] = Vec3b( 0, 0, 255);
226 lut[1] = Vec3b( 0, 170, 255);
227 lut[2] = Vec3b( 0, 255, 170);
228 lut[3] = Vec3b( 0, 255, 0);
229 lut[4] = Vec3b(170, 255, 0);
230 lut[5] = Vec3b(255, 170, 0);
231 lut[6] = Vec3b(255, 0, 0);
232 lut[7] = Vec3b(255, 0, 170);
233
234 dst = Mat::zeros(quantized.size(), CV_8UC3);
235 for (int r = 0; r < dst.rows; ++r)
236 {
237 const uchar* quant_r = quantized.ptr(r);
238 Vec3b* dst_r = dst.ptr<Vec3b>(r);
239 for (int c = 0; c < dst.cols; ++c)
240 {
241 uchar q = quant_r[c];
242 if (q)
243 dst_r[c] = lut[getLabel(q)];
244 }
245 }

256/**
257 * \brief Compute quantized orientation image from color image.
258 *
259 * Implements section 2.2 "Computing the Gradient Orientations."
260 *
261 * \param[in] src The source 8-bit, 3-channel image.
262 * \param[out] magnitude Destination floating-point array of squared magnitudes.
263 * \param[out] angle Destination 8-bit array of orientations. Each bit
264 * represents one bin of the orientation space.
265 * \param threshold Magnitude threshold. Keep only gradients whose norms are
266 * larger than this.
267 */
268static void quantizedOrientations(const Mat& src, Mat& magnitude,
269 Mat& angle, float threshold)
270{
271 magnitude.create(src.size(), CV_32F);
272
273 // Allocate temporary buffers
274 Size size = src.size();
275 Mat sobel_3dx; // per-channel horizontal derivative
276 Mat sobel_3dy; // per-channel vertical derivative
277 Mat sobel_dx(size, CV_32F); // maximum horizontal derivative
278 Mat sobel_dy(size, CV_32F); // maximum vertical derivative
279 Mat sobel_ag; // final gradient orientation (unquantized)
280 Mat smoothed;
281
282 // Compute horizontal and vertical image derivatives on all color channels separately
283 static const int KERNEL_SIZE = 7;
284 // For some reason cvSmooth/cv::GaussianBlur, cvSobel/cv::Sobel have different defaults for border handling...
285 GaussianBlur(src, smoothed, Size(KERNEL_SIZE, KERNEL_SIZE), 0, 0, BORDER_REPLICATE);
286 Sobel(smoothed, sobel_3dx, CV_16S, 1, 0, 3, 1.0, 0.0, BORDER_REPLICATE);
287 Sobel(smoothed, sobel_3dy, CV_16S, 0, 1, 3, 1.0, 0.0, BORDER_REPLICATE);
288
289 short * ptrx = (short *)sobel_3dx.data;
290 short * ptry = (short *)sobel_3dy.data;
291 float * ptr0x = (float *)sobel_dx.data;
292 float * ptr0y = (float *)sobel_dy.data;
293 float * ptrmg = (float *)magnitude.data;
294
295 const int length1 = static_cast<const int>(sobel_3dx.step1());
296 const int length2 = static_cast<const int>(sobel_3dy.step1());
297 const int length3 = static_cast<const int>(sobel_dx.step1());
298 const int length4 = static_cast<const int>(sobel_dy.step1());
299 const int length5 = static_cast<const int>(magnitude.step1());
300 const int length0 = sobel_3dy.cols * 3;
301
302 for (int r = 0; r < sobel_3dy.rows; ++r)
303 {
304 int ind = 0;
305
306 for (int i = 0; i < length0; i += 3)
307 {
308 // Use the gradient orientation of the channel whose magnitude is largest
309 int mag1 = CV_SQR(ptrx[i]) + CV_SQR(ptry[i]);
310 int mag2 = CV_SQR(ptrx[i + 1]) + CV_SQR(ptry[i + 1]);
311 int mag3 = CV_SQR(ptrx[i + 2]) + CV_SQR(ptry[i + 2]);
312
313 if (mag1 >= mag2 && mag1 >= mag3)
314 {
315 ptr0x[ind] = ptrx[i];
316 ptr0y[ind] = ptry[i];
317 ptrmg[ind] = (float)mag1;
318 }
319 else if (mag2 >= mag1 && mag2 >= mag3)
320 {
321 ptr0x[ind] = ptrx[i + 1];
322 ptr0y[ind] = ptry[i + 1];
323 ptrmg[ind] = (float)mag2;
324 }
325 else
326 {
327 ptr0x[ind] = ptrx[i + 2];
328 ptr0y[ind] = ptry[i + 2];
329 ptrmg[ind] = (float)mag3;
330 }
331 ++ind;
332 }
333 ptrx += length1;
334 ptry += length2;
335 ptr0x += length3;
336 ptr0y += length4;
337 ptrmg += length5;
338 }
339
340 // Calculate the final gradient orientations
341 phase(sobel_dx, sobel_dy, sobel_ag, true);
342 hysteresisGradient(magnitude, angle, sobel_ag, CV_SQR(threshold));
343}

345void hysteresisGradient(Mat& magnitude, Mat& quantized_angle,
346 Mat& angle, float threshold)
347{
348 // Quantize 360 degree range of orientations into 16 buckets
349 // Note that [0, 11.25), [348.75, 360) both get mapped in the end to label 0,
350 // for stability of horizontal and vertical features.
351 Mat_<unsigned char> quantized_unfiltered;
352 angle.convertTo(quantized_unfiltered, CV_8U, 16.0 / 360.0);
353
354 // Zero out top and bottom rows
355 /// @todo is this necessary, or even correct?
356 memset(quantized_unfiltered.ptr(), 0, quantized_unfiltered.cols);
357 memset(quantized_unfiltered.ptr(quantized_unfiltered.rows - 1), 0, quantized_unfiltered.cols);
358 // Zero out first and last columns
359 for (int r = 0; r < quantized_unfiltered.rows; ++r)
360 {
361 quantized_unfiltered(r, 0) = 0;
362 quantized_unfiltered(r, quantized_unfiltered.cols - 1) = 0;
363 }
364
365 // Mask 16 buckets into 8 quantized orientations
366 for (int r = 1; r < angle.rows - 1; ++r)
367 {
368 uchar* quant_r = quantized_unfiltered.ptr<uchar>(r);
369 for (int c = 1; c < angle.cols - 1; ++c)
370 {
371 quant_r[c] &= 7;
372 }
373 }
374
375 // Filter the raw quantized image. Only accept pixels where the magnitude is above some
376 // threshold, and there is local agreement on the quantization.
377 quantized_angle = Mat::zeros(angle.size(), CV_8U);
378 for (int r = 1; r < angle.rows - 1; ++r)
379 {
380 float* mag_r = magnitude.ptr<float>(r);
381
382 for (int c = 1; c < angle.cols - 1; ++c)
383 {
384 if (mag_r[c] > threshold)
385 {
386 // Compute histogram of quantized bins in 3x3 patch around pixel
387 int histogram[8] = {0, 0, 0, 0, 0, 0, 0, 0};
388
389 uchar* patch3x3_row = &quantized_unfiltered(r-1, c-1);
390 histogram[patch3x3_row[0]]++;
391 histogram[patch3x3_row[1]]++;
392 histogram[patch3x3_row[2]]++;
393
394 patch3x3_row += quantized_unfiltered.step1();
395 histogram[patch3x3_row[0]]++;
396 histogram[patch3x3_row[1]]++;
397 histogram[patch3x3_row[2]]++;
398
399 patch3x3_row += quantized_unfiltered.step1();
400 histogram[patch3x3_row[0]]++;
401 histogram[patch3x3_row[1]]++;
402 histogram[patch3x3_row[2]]++;
403
404 // Find bin with the most votes from the patch
405 int max_votes = 0;
406 int index = -1;
407 for (int i = 0; i < 8; ++i)
408 {
409 if (max_votes < histogram[i])
410 {
411 index = i;
412 max_votes = histogram[i];
413 }
414 }
415
416 // Only accept the quantization if majority of pixels in the patch agree
417 static const int NEIGHBOR_THRESHOLD = 5;
418 if (max_votes >= NEIGHBOR_THRESHOLD)
419 quantized_angle.at<uchar>(r, c) = uchar(1 << index);
420 }
421 }
422 }

499bool ColorGradientPyramid::extractTemplate(Template& templ) const
500{
501 // Want features on the border to distinguish from background
502 Mat local_mask;
503 if (!mask.empty())
504 {
505 erode(mask, local_mask, Mat(), Point(-1,-1), 1, BORDER_REPLICATE);
506 subtract(mask, local_mask, local_mask);
507 }
508
509 // Create sorted list of all pixels with magnitude greater than a threshold
510 std::vector<Candidate> candidates;
511 bool no_mask = local_mask.empty();
512 float threshold_sq = CV_SQR(strong_threshold);
513 for (int r = 0; r < magnitude.rows; ++r)
514 {
515 const uchar* angle_r = angle.ptr<uchar>(r);
516 const float* magnitude_r = magnitude.ptr<float>(r);
517 const uchar* mask_r = no_mask ? NULL : local_mask.ptr<uchar>(r);
518
519 for (int c = 0; c < magnitude.cols; ++c)
520 {
521 if (no_mask || mask_r[c])
522 {
523 uchar quantized = angle_r[c];
524 if (quantized > 0)
525 {
526 float score = magnitude_r[c];
527 if (score > threshold_sq)
528 {
529 candidates.push_back(Candidate(c, r, getLabel(quantized), score));
530 }
531 }
532 }
533 }
534 }
535 // We require a certain number of features
536 if (candidates.size() < num_features)
537 return false;
538 // NOTE: Stable sort to agree with old code, which used std::list::sort()
539 std::stable_sort(candidates.begin(), candidates.end());
540
541 // Use heuristic based on surplus of candidates in narrow outline for initial distance threshold
542 float distance = static_cast<float>(candidates.size() / num_features + 1);
543 selectScatteredFeatures(candidates, templ.features, num_features, distance);
544
545 // Size determined externally, needs to match templates for other modalities
546 templ.width = -1;
547 templ.height = -1;
548 templ.pyramid_level = pyramid_level;
549
550 return true;
551}

605static void accumBilateral(long delta, long i, long j, long * A, long * b, int threshold)
606{
607#ifdef __QNX__
608 long absdelta = (delta > 0) ? delta : -delta;
609 long f = absdelta < threshold ? 1 : 0;
610#else
611 long f = std::abs(delta) < threshold ? 1 : 0;
612#endif
613
614 const long fi = f * i;
615 const long fj = f * j;
616
617 A[0] += fi * i;
618 A[1] += fi * j;
619 A[3] += fj * j;
620 b[0] += fi * delta;
621 b[1] += fj * delta;
622}

624/**
625 * \brief Compute quantized normal image from depth image.
626 *
627 * Implements section 2.6 "Extension to Dense Depth Sensors."
628 *
629 * \param[in] src The source 16-bit depth image (in mm).
630 * \param[out] dst The destination 8-bit image. Each bit represents one bin of
631 * the view cone.
632 * \param distance_threshold Ignore pixels beyond this distance.
633 * \param difference_threshold When computing normals, ignore contributions of pixels whose
634 * depth difference with the central pixel is above this threshold.
635 *
636 * \todo Should also need camera model, or at least focal lengths? Replace distance_threshold with mask?
637 */
638static void quantizedNormals(const Mat& src, Mat& dst, int distance_threshold,
639 int difference_threshold)
640{
641 dst = Mat::zeros(src.size(), CV_8U);
642
643 IplImage src_ipl = src;
644 IplImage* ap_depth_data = &src_ipl;
645 IplImage dst_ipl = dst;
646 IplImage* dst_ipl_ptr = &dst_ipl;
647 IplImage** m_dep = &dst_ipl_ptr;
648
649 unsigned short * lp_depth = (unsigned short *)ap_depth_data->imageData;
650 unsigned char * lp_normals = (unsigned char *)m_dep[0]->imageData;
651
652 const int l_W = ap_depth_data->width;
653 const int l_H = ap_depth_data->height;
654
655 const int l_r = 5; // used to be 7
656 const int l_offset0 = -l_r - l_r * l_W;
657 const int l_offset1 = 0 - l_r * l_W;
658 const int l_offset2 = +l_r - l_r * l_W;
659 const int l_offset3 = -l_r;
660 const int l_offset4 = +l_r;
661 const int l_offset5 = -l_r + l_r * l_W;
662 const int l_offset6 = 0 + l_r * l_W;
663 const int l_offset7 = +l_r + l_r * l_W;
664
665 const int l_offsetx = GRANULARITY / 2;
666 const int l_offsety = GRANULARITY / 2;
667
668 for (int l_y = l_r; l_y < l_H - l_r - 1; ++l_y)
669 {
670 unsigned short * lp_line = lp_depth + (l_y * l_W + l_r);
671 unsigned char * lp_norm = lp_normals + (l_y * l_W + l_r);
672
673 for (int l_x = l_r; l_x < l_W - l_r - 1; ++l_x)
674 {
675 long l_d = lp_line[0];
676
677 if (l_d < distance_threshold)
678 {
679 // accum
680 long l_A[4]; l_A[0] = l_A[1] = l_A[2] = l_A[3] = 0;
681 long l_b[2]; l_b[0] = l_b[1] = 0;
682 accumBilateral(lp_line[l_offset0] - l_d, -l_r, -l_r, l_A, l_b, difference_threshold);
683 accumBilateral(lp_line[l_offset1] - l_d, 0, -l_r, l_A, l_b, difference_threshold);
684 accumBilateral(lp_line[l_offset2] - l_d, +l_r, -l_r, l_A, l_b, difference_threshold);
685 accumBilateral(lp_line[l_offset3] - l_d, -l_r, 0, l_A, l_b, difference_threshold);
686 accumBilateral(lp_line[l_offset4] - l_d, +l_r, 0, l_A, l_b, difference_threshold);
687 accumBilateral(lp_line[l_offset5] - l_d, -l_r, +l_r, l_A, l_b, difference_threshold);
688 accumBilateral(lp_line[l_offset6] - l_d, 0, +l_r, l_A, l_b, difference_threshold);
689 accumBilateral(lp_line[l_offset7] - l_d, +l_r, +l_r, l_A, l_b, difference_threshold);
690
691 // solve
692 long l_det = l_A[0] * l_A[3] - l_A[1] * l_A[1];
693 long l_ddx = l_A[3] * l_b[0] - l_A[1] * l_b[1];
694 long l_ddy = -l_A[1] * l_b[0] + l_A[0] * l_b[1];
695
696 /// @todo Magic number 1150 is focal length? This is something like
697 /// f in SXGA mode, but in VGA is more like 530.
698 float l_nx = static_cast<float>(1150 * l_ddx);
699 float l_ny = static_cast<float>(1150 * l_ddy);
700 float l_nz = static_cast<float>(-l_det * l_d);
701
702 float l_sqrt = sqrtf(l_nx * l_nx + l_ny * l_ny + l_nz * l_nz);
703
704 if (l_sqrt > 0)
705 {
706 float l_norminv = 1.0f / (l_sqrt);
707
708 l_nx *= l_norminv;
709 l_ny *= l_norminv;
710 l_nz *= l_norminv;
711
712 //*lp_norm = fabs(l_nz)*255;
713
714 int l_val1 = static_cast<int>(l_nx * l_offsetx + l_offsetx);
715 int l_val2 = static_cast<int>(l_ny * l_offsety + l_offsety);
716 int l_val3 = static_cast<int>(l_nz * GRANULARITY + GRANULARITY);
717
718 *lp_norm = NORMAL_LUT[l_val3][l_val2][l_val1];
719 }
720 else
721 {
722 *lp_norm = 0; // Discard shadows from depth sensor
723 }
724 }
725 else
726 {
727 *lp_norm = 0; //out of depth
728 }
729 ++lp_line;
730 ++lp_norm;
731 }
732 }
733 cvSmooth(m_dep[0], m_dep[0], CV_MEDIAN, 5, 5);
734}

796bool DepthNormalPyramid::extractTemplate(Template& templ) const
797{
798 // Features right on the object border are unreliable
799 Mat local_mask;
800 if (!mask.empty())
801 {
802 erode(mask, local_mask, Mat(), Point(-1,-1), 2, BORDER_REPLICATE);
803 }
804
805 // Compute distance transform for each individual quantized orientation
806 Mat temp = Mat::zeros(normal.size(), CV_8U);
807 Mat distances[8];
808 for (int i = 0; i < 8; ++i)
809 {
810 temp.setTo(1 << i, local_mask);
811 bitwise_and(temp, normal, temp);
812 // temp is now non-zero at pixels in the mask with quantized orientation i
813 distanceTransform(temp, distances[i], CV_DIST_C, 3);
814 }
815
816 // Count how many features taken for each label
817 int label_counts[8] = {0, 0, 0, 0, 0, 0, 0, 0};
818
819 // Create sorted list of candidate features
820 std::vector<Candidate> candidates;
821 bool no_mask = local_mask.empty();
822 for (int r = 0; r < normal.rows; ++r)
823 {
824 const uchar* normal_r = normal.ptr<uchar>(r);
825 const uchar* mask_r = no_mask ? NULL : local_mask.ptr<uchar>(r);
826
827 for (int c = 0; c < normal.cols; ++c)
828 {
829 if (no_mask || mask_r[c])
830 {
831 uchar quantized = normal_r[c];
832
833 if (quantized != 0 && quantized != 255) // background and shadow
834 {
835 int label = getLabel(quantized);
836
837 // Accept if distance to a pixel belonging to a different label is greater than
838 // some threshold. IOW, ideal feature is in the center of a large homogeneous
839 // region.
840 float score = distances[label].at<float>(r, c);
841 if (score >= extract_threshold)
842 {
843 candidates.push_back( Candidate(c, r, label, score) );
844 ++label_counts[label];
845 }
846 }
847 }
848 }
849 }
850 // We require a certain number of features
851 if (candidates.size() < num_features)
852 return false;
853
854 // Prefer large distances, but also want to collect features over all 8 labels.
855 // So penalize labels with lots of candidates.
856 for (size_t i = 0; i < candidates.size(); ++i)
857 {
858 Candidate& c = candidates[i];
859 c.score /= (float)label_counts[c.f.label];
860 }
861 std::stable_sort(candidates.begin(), candidates.end());
862
863 // Use heuristic based on object area for initial distance threshold
864 float area = no_mask ? (float)normal.total() : (float)countNonZero(local_mask);
865 float distance = sqrtf(area) / sqrtf((float)num_features) + 1.5f;
866 selectScatteredFeatures(candidates, templ.features, num_features, distance);
867
868 // Size determined externally, needs to match templates for other modalities
869 templ.width = -1;
870 templ.height = -1;
871 templ.pyramid_level = pyramid_level;
872
873 return true;
874}

1143 const Feature& f, int T, int W)
1144{
1145 // Retrieve the TxT grid of linear memories associated with the feature label
1146 const Mat& memory_grid = linear_memories[f.label];
1147 CV_DbgAssert(memory_grid.rows == T*T);
1148 CV_DbgAssert(f.x >= 0);
1149 CV_DbgAssert(f.y >= 0);

1153 int grid_index = grid_y * T + grid_x;
1154 CV_DbgAssert(grid_index >= 0);

1158 // input image width decimated by T.

1160 int lm_y = f.y / T;
1161 int lm_index = lm_y * W + lm_x;
1162 CV_DbgAssert(lm_index >= 0);
1163 CV_DbgAssert(lm_index < memory_grid.cols);

1167/**
1168 * \brief Compute similarity measure for a given template at each sampled image location.
1169 *
1170 * Uses linear memories to compute the similarity measure as described in Fig. 7.
1171 *
1172 * \param[in] linear_memories Vector of 8 linear memories, one for each label.
1173 * \param[in] templ Template to match against.
1174 * \param[out] dst Destination 8-bit similarity image of size (W/T, H/T).
1175 * \param size Size (W, H) of the original input image.
1176 * \param T Sampling step.
1177 */
1178static void similarity(const std::vector<Mat>& linear_memories, const Template& templ,
1179 Mat& dst, Size size, int T)
1180{
1181 // 63 features or less is a special case because the max similarity per-feature is 4.
1182 // 255/4 = 63, so up to that many we can add up similarities in 8 bits without worrying
1183 // about overflow. Therefore here we use _mm_add_epi8 as the workhorse, whereas a more
1184 // general function would use _mm_add_epi16.
1185 CV_Assert(templ.features.size() <= 63);
1186 /// @todo Handle more than 255/MAX_RESPONSE features!!
1187
1188 // Decimate input image size by factor of T
1189 int W = size.width / T;
1190 int H = size.height / T;
1191
1192 // Feature dimensions, decimated by factor T and rounded up
1193 int wf = (templ.width - 1) / T + 1;
1194 int hf = (templ.height - 1) / T + 1;
1195
1196 // Span is the range over which we can shift the template around the input image
1197 int span_x = W - wf;
1198 int span_y = H - hf;
1199
1200 // Compute number of contiguous (in memory) pixels to check when sliding feature over
1201 // image. This allows template to wrap around left/right border incorrectly, so any
1202 // wrapped template matches must be filtered out!
1203 int template_positions = span_y * W + span_x + 1; // why add 1?
1204 //int template_positions = (span_y - 1) * W + span_x; // More correct?
1205
1206 /// @todo In old code, dst is buffer of size m_U. Could make it something like
1207 /// (span_x)x(span_y) instead?
1208 dst = Mat::zeros(H, W, CV_8U);
1209 uchar* dst_ptr = dst.ptr<uchar>();
1210
1211#if CV_SSE2
1212 volatile bool haveSSE2 = checkHardwareSupport(CV_CPU_SSE2);
1213#if CV_SSE3
1214 volatile bool haveSSE3 = checkHardwareSupport(CV_CPU_SSE3);
1215#endif
1216#endif
1217
1218 // Compute the similarity measure for this template by accumulating the contribution of
1219 // each feature
1220 for (int i = 0; i < (int)templ.features.size(); ++i)
1221 {
1222 // Add the linear memory at the appropriate offset computed from the location of
1223 // the feature in the template
1224 Feature f = templ.features[i];
1225 // Discard feature if out of bounds
1226 /// @todo Shouldn't actually see x or y < 0 here?
1227 if (f.x < 0 || f.x >= size.width || f.y < 0 || f.y >= size.height)
1228 continue;
1229 const uchar* lm_ptr = accessLinearMemory(linear_memories, f, T, W);
1230
1231 // Now we do an aligned/unaligned add of dst_ptr and lm_ptr with template_positions elements
1232 int j = 0;
1233 // Process responses 16 at a time if vectorization possible
1234#if CV_SSE2
1235#if CV_SSE3
1236 if (haveSSE3)
1237 {
1238 // LDDQU may be more efficient than MOVDQU for unaligned load of next 16 responses
1239 for ( ; j < template_positions - 15; j += 16)
1240 {
1241 __m128i responses = _mm_lddqu_si128(reinterpret_cast<const __m128i*>(lm_ptr + j));
1242 __m128i* dst_ptr_sse = reinterpret_cast<__m128i*>(dst_ptr + j);
1243 *dst_ptr_sse = _mm_add_epi8(*dst_ptr_sse, responses);
1244 }
1245 }
1246 else
1247#endif
1248 if (haveSSE2)
1249 {
1250 // Fall back to MOVDQU
1251 for ( ; j < template_positions - 15; j += 16)
1252 {
1253 __m128i responses = _mm_loadu_si128(reinterpret_cast<const __m128i*>(lm_ptr + j));
1254 __m128i* dst_ptr_sse = reinterpret_cast<__m128i*>(dst_ptr + j);
1255 *dst_ptr_sse = _mm_add_epi8(*dst_ptr_sse, responses);
1256 }
1257 }
1258#endif
1259 for ( ; j < template_positions; ++j)
1260 dst_ptr[j] = uchar(dst_ptr[j] + lm_ptr[j]);
1261 }
1262}

1272 * \param center Center of the local region.
1273 */
1274static void similarityLocal(const std::vector<Mat>& linear_memories, const Template& templ,
1275 Mat& dst, Size size, int T, Point center)
1276{
1277 // Similar to whole-image similarity() above. This version takes a position 'center'
1278 // and computes the energy in the 16x16 patch centered on it.
1279 CV_Assert(templ.features.size() <= 63);
1280
1281 // Compute the similarity map in a 16x16 patch around center
1282 int W = size.width / T;
1283 dst = Mat::zeros(16, 16, CV_8U);
1284
1285 // Offset each feature point by the requested center. Further adjust to (-8,-8) from the
1286 // center to get the top-left corner of the 16x16 patch.
1287 // NOTE: We make the offsets multiples of T to agree with results of the original code.
1288 int offset_x = (center.x / T - 8) * T;
1289 int offset_y = (center.y / T - 8) * T;
1290
1291#if CV_SSE2
1292 volatile bool haveSSE2 = checkHardwareSupport(CV_CPU_SSE2);
1293#if CV_SSE3
1294 volatile bool haveSSE3 = checkHardwareSupport(CV_CPU_SSE3);
1295#endif
1296 __m128i* dst_ptr_sse = dst.ptr<__m128i>();
1297#endif
1298
1299 for (int i = 0; i < (int)templ.features.size(); ++i)
1300 {
1301 Feature f = templ.features[i];
1302 f.x += offset_x;
1303 f.y += offset_y;
1304 // Discard feature if out of bounds, possibly due to applying the offset
1305 if (f.x < 0 || f.y < 0 || f.x >= size.width || f.y >= size.height)
1306 continue;
1307
1308 const uchar* lm_ptr = accessLinearMemory(linear_memories, f, T, W);
1309
1310 // Process whole row at a time if vectorization possible
1311#if CV_SSE2
1312#if CV_SSE3
1313 if (haveSSE3)
1314 {
1315 // LDDQU may be more efficient than MOVDQU for unaligned load of 16 responses from current row
1316 for (int row = 0; row < 16; ++row)
1317 {
1318 __m128i aligned = _mm_lddqu_si128(reinterpret_cast<const __m128i*>(lm_ptr));
1319 dst_ptr_sse[row] = _mm_add_epi8(dst_ptr_sse[row], aligned);
1320 lm_ptr += W; // Step to next row
1321 }
1322 }
1323 else
1324#endif
1325 if (haveSSE2)
1326 {
1327 // Fall back to MOVDQU
1328 for (int row = 0; row < 16; ++row)
1329 {
1330 __m128i aligned = _mm_loadu_si128(reinterpret_cast<const __m128i*>(lm_ptr));
1331 dst_ptr_sse[row] = _mm_add_epi8(dst_ptr_sse[row], aligned);
1332 lm_ptr += W; // Step to next row
1333 }
1334 }
1335 else
1336#endif
1337 {
1338 uchar* dst_ptr = dst.ptr<uchar>();
1339 for (int row = 0; row < 16; ++row)
1340 {
1341 for (int col = 0; col < 16; ++col)
1342 dst_ptr[col] = uchar(dst_ptr[col] + lm_ptr[col]);
1343 dst_ptr += 16;
1344 lm_ptr += W;
1345 }
1346 }
1347 }
1348}

1351{

1353
1354 while (src1 != end)
1355 {
1356 *res = *src1 + *src2;
1357

1360 ++res;
1361 }

1365 * \brief Accumulate one or more 8-bit similarity images.
1366 *
1367 * \param[in] similarities Source 8-bit similarity images.
1368 * \param[out] dst Destination 16-bit similarity image.
1369 */
1370static void addSimilarities(const std::vector<Mat>& similarities, Mat& dst)
1371{
1372 if (similarities.size() == 1)
1373 {
1374 similarities[0].convertTo(dst, CV_16U);
1375 }
1376 else
1377 {
1378 // NOTE: add() seems to be rather slow in the 8U + 8U -> 16U case

1381
1382 /// @todo Optimize 16u + 8u -> 16u when more than 2 modalities
1383 for (size_t i = 2; i < similarities.size(); ++i)
1384 add(dst, similarities[i], dst, noArray(), CV_16U);
1385 }

1404void Detector::match(const std::vector<Mat>& sources, float threshold, std::vector<Match>& matches,
1405 const std::vector<std::string>& class_ids, OutputArrayOfArrays quantized_images,
1406 const std::vector<Mat>& masks) const
1407{
1408 matches.clear();
1409 if (quantized_images.needed())
1410 quantized_images.create(1, static_cast<int>(pyramid_levels * modalities.size()), CV_8U);
1411
1412 assert(sources.size() == modalities.size());
1413 // Initialize each modality with our sources
1414 std::vector< Ptr<QuantizedPyramid> > quantizers;
1415 for (int i = 0; i < (int)modalities.size(); ++i){
1416 Mat mask, source;
1417 source = sources[i];
1418 if(!masks.empty()){
1419 assert(masks.size() == modalities.size());
1420 mask = masks[i];
1421 }
1422 assert(mask.empty() || mask.size() == source.size());
1423 quantizers.push_back(modalities[i]->process(source, mask));
1424 }
1425 // pyramid level -> modality -> quantization
1426 LinearMemoryPyramid lm_pyramid(pyramid_levels,
1427 std::vector<LinearMemories>(modalities.size(), LinearMemories(8)));
1428
1429 // For each pyramid level, precompute linear memories for each modality
1430 std::vector<Size> sizes;
1431 for (int l = 0; l < pyramid_levels; ++l)
1432 {
1433 int T = T_at_level[l];
1434 std::vector<LinearMemories>& lm_level = lm_pyramid[l];
1435
1436 if (l > 0)
1437 {
1438 for (int i = 0; i < (int)quantizers.size(); ++i)
1439 quantizers[i]->pyrDown();
1440 }
1441
1442 Mat quantized, spread_quantized;
1443 std::vector<Mat> response_maps;
1444 for (int i = 0; i < (int)quantizers.size(); ++i)
1445 {
1446 quantizers[i]->quantize(quantized);
1447 spread(quantized, spread_quantized, T);
1448 computeResponseMaps(spread_quantized, response_maps);
1449
1450 LinearMemories& memories = lm_level[i];
1451 for (int j = 0; j < 8; ++j)
1452 linearize(response_maps[j], memories[j], T);
1453
1454 if (quantized_images.needed()) //use copyTo here to side step reference semantics.
1455 quantized.copyTo(quantized_images.getMatRef(static_cast<int>(l*quantizers.size() + i)));
1456 }
1457
1458 sizes.push_back(quantized.size());
1459 }
1460
1461 if (class_ids.empty())
1462 {
1463 // Match all templates
1464 TemplatesMap::const_iterator it = class_templates.begin(), itend = class_templates.end();
1465 for ( ; it != itend; ++it)
1466 matchClass(lm_pyramid, sizes, threshold, matches, it->first, it->second);
1467 }
1468 else
1469 {
1470 // Match only templates for the requested class IDs
1471 for (int i = 0; i < (int)class_ids.size(); ++i)
1472 {
1473 TemplatesMap::const_iterator it = class_templates.find(class_ids[i]);
1474 if (it != class_templates.end())
1475 matchClass(lm_pyramid, sizes, threshold, matches, it->first, it->second);
1476 }
1477 }
1478
1479 // Sort matches by similarity, and prune any duplicates introduced by pyramid refinement
1480 std::sort(matches.begin(), matches.end());
1481 std::vector<Match>::iterator new_end = std::unique(matches.begin(), matches.end());
1482 matches.erase(new_end, matches.end());
1483}

1485// Used to filter out weak matches
1486struct MatchPredicate
1487{
1488 MatchPredicate(float _threshold) : threshold(_threshold) {}
1489 bool operator() (const Match& m) { return m.similarity < threshold; }
1490 float threshold;
1491};

1493void Detector::matchClass(const LinearMemoryPyramid& lm_pyramid,
1494 const std::vector<Size>& sizes,
1495 float threshold, std::vector<Match>& matches,
1496 const std::string& class_id,
1497 const std::vector<TemplatePyramid>& template_pyramids) const
1498{
1499 // For each template...
1500 for (size_t template_id = 0; template_id < template_pyramids.size(); ++template_id)
1501 {
1502 const TemplatePyramid& tp = template_pyramids[template_id];
1503
1504 // First match over the whole image at the lowest pyramid level
1505 /// @todo Factor this out into separate function
1506 const std::vector<LinearMemories>& lowest_lm = lm_pyramid.back();
1507
1508 // Compute similarity maps for each modality at lowest pyramid level
1509 std::vector<Mat> similarities(modalities.size());
1510 int lowest_start = static_cast<int>(tp.size() - modalities.size());
1511 int lowest_T = T_at_level.back();
1512 int num_features = 0;
1513 for (int i = 0; i < (int)modalities.size(); ++i)
1514 {
1515 const Template& templ = tp[lowest_start + i];
1516 num_features += static_cast<int>(templ.features.size());
1517 similarity(lowest_lm[i], templ, similarities[i], sizes.back(), lowest_T);
1518 }
1519
1520 // Combine into overall similarity
1521 /// @todo Support weighting the modalities
1522 Mat total_similarity;
1523 addSimilarities(similarities, total_similarity);
1524
1525 // Convert user-friendly percentage to raw similarity threshold. The percentage
1526 // threshold scales from half the max response (what you would expect from applying
1527 // the template to a completely random image) to the max response.
1528 // NOTE: This assumes max per-feature response is 4, so we scale between [2*nf, 4*nf].
1529 int raw_threshold = static_cast<int>(2*num_features + (threshold / 100.f) * (2*num_features) + 0.5f);
1530
1531 // Find initial matches
1532 std::vector<Match> candidates;
1533 for (int r = 0; r < total_similarity.rows; ++r)
1534 {
1535 ushort* row = total_similarity.ptr<ushort>(r);
1536 for (int c = 0; c < total_similarity.cols; ++c)
1537 {
1538 int raw_score = row[c];
1539 if (raw_score > raw_threshold)
1540 {
1541 int offset = lowest_T / 2 + (lowest_T % 2 - 1);
1542 int x = c * lowest_T + offset;
1543 int y = r * lowest_T + offset;
1544 float score =(raw_score * 100.f) / (4 * num_features) + 0.5f;
1545 candidates.push_back(Match(x, y, score, class_id, static_cast<int>(template_id)));
1546 }
1547 }
1548 }
1549
1550 // Locally refine each match by marching up the pyramid
1551 for (int l = pyramid_levels - 2; l >= 0; --l)
1552 {
1553 const std::vector<LinearMemories>& lms = lm_pyramid[l];
1554 int T = T_at_level[l];
1555 int start = static_cast<int>(l * modalities.size());
1556 Size size = sizes[l];
1557 int border = 8 * T;
1558 int offset = T / 2 + (T % 2 - 1);
1559 int max_x = size.width - tp[start].width - border;
1560 int max_y = size.height - tp[start].height - border;
1561
1562 std::vector<Mat> similarities2(modalities.size());
1563 Mat total_similarity2;
1564 for (int m = 0; m < (int)candidates.size(); ++m)
1565 {
1566 Match& match2 = candidates[m];
1567 int x = match2.x * 2 + 1; /// @todo Support other pyramid distance
1568 int y = match2.y * 2 + 1;
1569
1570 // Require 8 (reduced) row/cols to the up/left
1571 x = std::max(x, border);
1572 y = std::max(y, border);
1573
1574 // Require 8 (reduced) row/cols to the down/left, plus the template size
1575 x = std::min(x, max_x);
1576 y = std::min(y, max_y);
1577
1578 // Compute local similarity maps for each modality
1579 int numFeatures = 0;
1580 for (int i = 0; i < (int)modalities.size(); ++i)
1581 {
1582 const Template& templ = tp[start + i];
1583 numFeatures += static_cast<int>(templ.features.size());
1584 similarityLocal(lms[i], templ, similarities2[i], size, T, Point(x, y));
1585 }
1586 addSimilarities(similarities2, total_similarity2);
1587
1588 // Find best local adjustment
1589 int best_score = 0;
1590 int best_r = -1, best_c = -1;
1591 for (int r = 0; r < total_similarity2.rows; ++r)
1592 {
1593 ushort* row = total_similarity2.ptr<ushort>(r);
1594 for (int c = 0; c < total_similarity2.cols; ++c)
1595 {
1596 int score = row[c];
1597 if (score > best_score)
1598 {
1599 best_score = score;
1600 best_r = r;
1601 best_c = c;
1602 }
1603 }
1604 }
1605 // Update current match
1606 match2.x = (x / T - 8 + best_c) * T + offset;
1607 match2.y = (y / T - 8 + best_r) * T + offset;
1608 match2.similarity = (best_score * 100.f) / (4 * numFeatures);
1609 }
1610
1611 // Filter out any matches that drop below the similarity threshold
1612 std::vector<Match>::iterator new_end = std::remove_if(candidates.begin(), candidates.end(),
1613 MatchPredicate(threshold));
1614 candidates.erase(new_end, candidates.end());
1615 }
1616
1617 matches.insert(matches.end(), candidates.begin(), candidates.end());
1618 }

1621int Detector::addTemplate(const std::vector<Mat>& sources, const std::string& class_id,
1622 const Mat& object_mask, Rect* bounding_box)
1623{
1624 int num_modalities = static_cast<int>(modalities.size());
1625 std::vector<TemplatePyramid>& template_pyramids = class_templates[class_id];
1626 int template_id = static_cast<int>(template_pyramids.size());
1627
1628 TemplatePyramid tp;
1629 tp.resize(num_modalities * pyramid_levels);
1630
1631 // For each modality...
1632 for (int i = 0; i < num_modalities; ++i)
1633 {
1634 // Extract a template at each pyramid level
1635 Ptr<QuantizedPyramid> qp = modalities[i]->process(sources[i], object_mask);
1636 for (int l = 0; l < pyramid_levels; ++l)
1637 {
1638 /// @todo Could do mask subsampling here instead of in pyrDown()
1639 if (l > 0)
1640 qp->pyrDown();
1641
1642 bool success = qp->extractTemplate(tp[l*num_modalities + i]);
1643 if (!success)
1644 return -1;
1645 }
1646 }
1647
1648 Rect bb = cropTemplates(tp);
1649 if (bounding_box)
1650 *bounding_box = bb;
1651
1652 /// @todo Can probably avoid a copy of tp here with swap
1653 template_pyramids.push_back(tp);
1654 return template_id;
1655}
