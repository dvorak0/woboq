4/*
5// Computation the point of intersection functions
6// (parabolas on the variable y)
7// a(y - q1) + b(q1 - y)(q1 - y) + f[q1]
8// a(y - q2) + b(q2 - y)(q2 - y) + f[q2]
9//
10//
11// API
12// int GetPointOfIntersection(const float *f,
13 const float a, const float b,
14 int q1, int q2, float *point);
15// INPUT
16// f - function on the regular grid
17// a - coefficient of the function
18// b - coefficient of the function
19// q1 - parameter of the function
20// q2 - parameter of the function
21// OUTPUT
22// point - point of intersection
23// RESULT
24// Error status
25*/
26int GetPointOfIntersection(const float *f,
27 const float a, const float b,
28 int q1, int q2, float *point)
29{
30 if (q1 == q2)
31 {
32 return DISTANCE_TRANSFORM_EQUAL_POINTS;
33 } /* if (q1 == q2) */
34 (*point) = ( (f[q2] - a * q2 + b *q2 * q2) -
35 (f[q1] - a * q1 + b * q1 * q1) ) / (2 * b * (q2 - q1));
36 return DISTANCE_TRANSFORM_OK;
37}

39/*
40// Decision of one dimensional problem generalized distance transform
41// on the regular grid at all points
42// min (a(y' - y) + b(y' - y)(y' - y) + f(y')) (on y')
43//
44// API
45// int DistanceTransformOneDimensionalProblem(const float *f, const int n,
46 const float a, const float b,
47 float *distanceTransform,
48 int *points);
49// INPUT
50// f - function on the regular grid
51// n - grid dimension
52// a - coefficient of optimizable function
53// b - coefficient of optimizable function
54// OUTPUT
55// distanceTransform - values of generalized distance transform
56// points - arguments that corresponds to the optimal value of function
57// RESULT
58// Error status
59*/
60int DistanceTransformOneDimensionalProblem(const float *f, const int n,
61 const float a, const float b,
62 float *distanceTransform,
63 int *points)
64{
65 int i, k;
66 int tmp;
67 int diff;
68 float pointIntersection;
69 int *v;
70 float *z;
71 k = 0;
72
73 // Allocation memory (must be free in this function)
74 v = (int *)malloc (sizeof(int) * n);
75 z = (float *)malloc (sizeof(float) * (n + 1));
76
77 v[0] = 0;
78 z[0] = (float)F_MIN; // left border of envelope
79 z[1] = (float)F_MAX; // right border of envelope
80
81 for (i = 1; i < n; i++)
82 {
83 tmp = GetPointOfIntersection(f, a, b, v[k], i, &pointIntersection);
84 if (tmp != DISTANCE_TRANSFORM_OK)
85 {
86 free(v);
87 free(z);
88 return DISTANCE_TRANSFORM_GET_INTERSECTION_ERROR;
89 } /* if (tmp != DISTANCE_TRANSFORM_OK) */
90 if (pointIntersection <= z[k])
91 {
92 // Envelope doesn't contain current parabola
93 do
94 {
95 k--;
96 tmp = GetPointOfIntersection(f, a, b, v[k], i, &pointIntersection);
97 if (tmp != DISTANCE_TRANSFORM_OK)
98 {
99 free(v);
100 free(z);
101 return DISTANCE_TRANSFORM_GET_INTERSECTION_ERROR;
102 } /* if (tmp != DISTANCE_TRANSFORM_OK) */
103 }while (pointIntersection <= z[k]);
104 // Addition parabola to the envelope
105 k++;
106 v[k] = i;
107 z[k] = pointIntersection;
108 z[k + 1] = (float)F_MAX;
109 }
110 else
111 {
112 // Addition parabola to the envelope
113 k++;
114 v[k] = i;
115 z[k] = pointIntersection;
116 z[k + 1] = (float)F_MAX;
117 } /* if (pointIntersection <= z[k]) */
118 }
119
120 // Computation values of generalized distance transform at all grid points
121 k = 0;
122 for (i = 0; i < n; i++)
123 {
124 while (z[k + 1] < i)
125 {
126 k++;
127 }
128 points[i] = v[k];
129 diff = i - v[k];
130 distanceTransform[i] = a * diff + b * diff * diff + f[v[k]];
131 }
132
133 // Release allocated memory
134 free(v);
135 free(z);
136 return DISTANCE_TRANSFORM_OK;
137}

139/*
140// Computation next cycle element
141//
142// API
143// int GetNextCycleElement(int k, int n, int q);
144// INPUT
145// k - index of the previous cycle element
146// n - number of matrix rows
147// q - parameter that equal
148 (number_of_rows * number_of_columns - 1)
149// OUTPUT
150// None
151// RESULT
152// Next cycle element
153*/
154int GetNextCycleElement(int k, int n, int q)
155{
156 return ((k * n) % q);
157}

159/*
160// Transpose cycle elements
161//
162// API
163// void TransposeCycleElements(float *a, int *cycle, int cycle_len)
164// INPUT
165// a - initial matrix
166// cycle - indeces array of cycle
167// cycle_len - number of elements in the cycle
168// OUTPUT
169// a - matrix with transposed elements
170// RESULT
171// Error status
172*/
173void TransposeCycleElements(float *a, int *cycle, int cycle_len)
174{
175 int i;
176 float buf;
177 for (i = cycle_len - 1; i > 0 ; i--)
178 {
179 buf = a[ cycle[i] ];
180 a[ cycle[i] ] = a[ cycle[i - 1] ];
181 a[ cycle[i - 1] ] = buf;
182 }
183}

211/*
212// Getting transposed matrix
213//
214// API
215// void Transpose(float *a, int n, int m);
216// INPUT
217// a - initial matrix
218// n - number of rows
219// m - number of columns
220// OUTPUT
221// a - transposed matrix
222// RESULT
223// None
224*/
225void Transpose(float *a, int n, int m)
226{
227 int *cycle;
228 int i, k, q, cycle_len;
229 int max_cycle_len;
230
231 max_cycle_len = n * m;
232
233 // Allocation memory (must be free in this function)
234 cycle = (int *)malloc(sizeof(int) * max_cycle_len);
235
236 cycle_len = 0;
237 q = n * m - 1;
238 for (i = 1; i < q; i++)
239 {
240 k = GetNextCycleElement(i, n, q);
241 cycle[cycle_len] = i;
242 cycle_len++;
243
244 while (k > i)
245 {
246 cycle[cycle_len] = k;
247 cycle_len++;
248 k = GetNextCycleElement(k, n, q);
249 }
250 if (k == i)
251 {
252 TransposeCycleElements(a, cycle, cycle_len);
253 } /* if (k == i) */
254 cycle_len = 0;
255 }
256
257 // Release allocated memory
258 free(cycle);
259}

311/*
312// Decision of two dimensional problem generalized distance transform
313// on the regular grid at all points
314// min{d2(y' - y) + d4(y' - y)(y' - y) +
315 min(d1(x' - x) + d3(x' - x)(x' - x) + f(x',y'))} (on x', y')
316//
317// API
318// int DistanceTransformTwoDimensionalProblem(const float *f,
319 const int n, const int m,
320 const float coeff[4],
321 float *distanceTransform,
322 int *pointsX, int *pointsY);
323// INPUT
324// f - function on the regular grid
325// n - number of rows
326// m - number of columns
327// coeff - coefficients of optimizable function
328 coeff[0] = d1, coeff[1] = d2,
329 coeff[2] = d3, coeff[3] = d4
330// OUTPUT
331// distanceTransform - values of generalized distance transform
332// pointsX - arguments x' that correspond to the optimal value
333// pointsY - arguments y' that correspond to the optimal value
334// RESULT
335// Error status
336*/
337int DistanceTransformTwoDimensionalProblem(const float *f,
338 const int n, const int m,
339 const float coeff[4],
340 float *distanceTransform,
341 int *pointsX, int *pointsY)
342{
343 int i, j, tmp;
344 int resOneDimProblem;
345 int size = n * m;
346 std::vector<float> internalDistTrans(size);
347 std::vector<int> internalPointsX(size);
348
349 for (i = 0; i < n; i++)
350 {
351 resOneDimProblem = DistanceTransformOneDimensionalProblem(
352 f + i * m, m,
353 coeff[0], coeff[2],
354 &internalDistTrans[i * m],
355 &internalPointsX[i * m]);
356 if (resOneDimProblem != DISTANCE_TRANSFORM_OK)
357 return DISTANCE_TRANSFORM_ERROR;
358 }
359 Transpose(&internalDistTrans[0], n, m);
360 for (j = 0; j < m; j++)
361 {
362 resOneDimProblem = DistanceTransformOneDimensionalProblem(
363 &internalDistTrans[j * n], n,
364 coeff[1], coeff[3],
365 distanceTransform + j * n,
366 pointsY + j * n);
367 if (resOneDimProblem != DISTANCE_TRANSFORM_OK)
368 return DISTANCE_TRANSFORM_ERROR;
369 }
370 Transpose(distanceTransform, m, n);
371 Transpose_int(pointsY, m, n);
372
373 for (i = 0; i < n; i++)
374 {
375 for (j = 0; j < m; j++)
376 {
377 tmp = pointsY[i * m + j];
378 pointsX[i * m + j] = internalPointsX[tmp * m + j];
379 }
380 }
381
382 return DISTANCE_TRANSFORM_OK;
383}
