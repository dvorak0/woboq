55const int num_detections = 3;
56const float true_scores[3] = {-0.383931f, -0.825876f, -0.959934f};
57const float score_thr = 0.05f;
58const CvRect true_bounding_boxes[3] = {cvRect(0, 45, 362, 452), cvRect(304, 0, 64, 80), cvRect(236, 0, 108, 59)};

62protected:
63 void run(int);
64 bool isEqual(CvRect r1, CvRect r2, int eps);

67bool CV_LatentSVMDetectorTest::isEqual(CvRect r1, CvRect r2, int eps)
68{
69 return (std::abs(r1.x - r2.x) <= eps
70 && std::abs(r1.y - r2.y) <= eps
71 && std::abs(r1.width - r2.width) <= eps
72 && std::abs(r1.height - r2.height) <= eps);
73}

75void CV_LatentSVMDetectorTest::run( int /* start_from */)
76{
77 string img_path = string(ts->get_data_path()) + "latentsvmdetector/cat.png";
78 string model_path = string(ts->get_data_path()) + "latentsvmdetector/models_VOC2007/cat.xml";
79 int numThreads = -1;
80
81#ifdef HAVE_TBB
82 numThreads = 2;
83 tbb::task_scheduler_init init(tbb::task_scheduler_init::deferred);
84 init.initialize(numThreads);
85#endif
86
87 IplImage* image = cvLoadImage(img_path.c_str());
88 if (!image)
89 {
90 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_TEST_DATA );
91 return;
92 }
93
94 CvLatentSvmDetector* detector = cvLoadLatentSvmDetector(model_path.c_str());
95 if (!detector)
96 {
97 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_TEST_DATA );
98 cvReleaseImage(&image);
99 return;
100 }
101
102 CvMemStorage* storage = cvCreateMemStorage(0);
103 CvSeq* detections = 0;
104 detections = cvLatentSvmDetectObjects(image, detector, storage, 0.5f, numThreads);
105 if (detections->total != num_detections)
106 {
107 ts->set_failed_test_info( cvtest::TS::FAIL_MISMATCH );
108 }
109 else
110 {
111 ts->set_failed_test_info(cvtest::TS::OK);
112 for (int i = 0; i < detections->total; i++)
113 {
114 CvObjectDetection detection = *(CvObjectDetection*)cvGetSeqElem( detections, i );
115 CvRect bounding_box = detection.rect;
116 float score = detection.score;
117 if ((!isEqual(bounding_box, true_bounding_boxes[i], 1)) || (fabs(score - true_scores[i]) > score_thr))
118 {
119 ts->set_failed_test_info( cvtest::TS::FAIL_MISMATCH );
120 break;
121 }
122 }
123 }
124#ifdef HAVE_TBB
125 init.terminate();
126#endif
127 cvReleaseMemStorage( &storage );
128 cvReleaseLatentSvmDetector( &detector );
129 cvReleaseImage( &image );
130}

134class LatentSVMDetectorTest : public cvtest::BaseTest
135{
136protected:
137 void run(int);
138};
139
140static void writeDetections( FileStorage& fs, const string& nodeName, const vector<LatentSvmDetector::ObjectDetection>& detections )
141{
142 fs << nodeName << "[";
143 for( size_t i = 0; i < detections.size(); i++ )
144 {
145 const LatentSvmDetector::ObjectDetection& d = detections[i];
146 fs << d.rect.x << d.rect.y << d.rect.width << d.rect.height
147 << d.score << d.classID;
148 }
149 fs << "]";
150}
151
152static void readDetections( FileStorage fs, const string& nodeName, vector<LatentSvmDetector::ObjectDetection>& detections )
153{
154 detections.clear();
155
156 FileNode fn = fs.root()[nodeName];
157 FileNodeIterator fni = fn.begin();
158 while( fni != fn.end() )
159 {
160 LatentSvmDetector::ObjectDetection d;
161 fni >> d.rect.x >> d.rect.y >> d.rect.width >> d.rect.height
162 >> d.score >> d.classID;
163 detections.push_back( d );
164 }
165}
166
167static inline bool isEqual( const LatentSvmDetector::ObjectDetection& d1, const LatentSvmDetector::ObjectDetection& d2, int eps, float threshold)
168{
169 return (
170 std::abs(d1.rect.x - d2.rect.x) <= eps
171 && std::abs(d1.rect.y - d2.rect.y) <= eps
172 && std::abs(d1.rect.width - d2.rect.width) <= eps
173 && std::abs(d1.rect.height - d2.rect.height) <= eps
174 && (d1.classID == d2.classID)
175 && std::abs(d1.score - d2.score) <= threshold
176 );
177}
178
179std::ostream& operator << (std::ostream& os, const CvRect& r)
180{
181 return (os << "[x=" << r.x << ", y=" << r.y << ", w=" << r.width << ", h=" << r.height << "]");
182}
183
184bool compareResults( const vector<LatentSvmDetector::ObjectDetection>& calc, const vector<LatentSvmDetector::ObjectDetection>& valid, int eps, float threshold)
185{
186 if( calc.size() != valid.size() )
187 return false;
188
189 for( size_t i = 0; i < calc.size(); i++ )
190 {
191 const LatentSvmDetector::ObjectDetection& c = calc[i];
192 const LatentSvmDetector::ObjectDetection& v = valid[i];
193 if( !isEqual(c, v, eps, threshold) )
194 {
195 std::cerr << "Expected: " << v.rect << " class=" << v.classID << " score=" << v.score << std::endl;
196 std::cerr << "Actual: " << c.rect << " class=" << c.classID << " score=" << c.score << std::endl;
197 return false;
198 }
199 }
200 return true;
201}

203void LatentSVMDetectorTest::run( int /* start_from */)
204{
205 string img_path_cat = string(ts->get_data_path()) + "latentsvmdetector/cat.png";
206 string img_path_cars = string(ts->get_data_path()) + "latentsvmdetector/cars.png";
207
208 string model_path_cat = string(ts->get_data_path()) + "latentsvmdetector/models_VOC2007/cat.xml";
209 string model_path_car = string(ts->get_data_path()) + "latentsvmdetector/models_VOC2007/car.xml";
210
211 string true_res_path = string(ts->get_data_path()) + "latentsvmdetector/results.xml";
212
213 int numThreads = 1;
214
215#ifdef HAVE_TBB
216 numThreads = 2;
217#endif
218
219 Mat image_cat = imread( img_path_cat );
220 Mat image_cars = imread( img_path_cars );
221 if( image_cat.empty() || image_cars.empty() )
222 {
223 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_TEST_DATA );
224 return;
225 }
226
227 // We will test 2 cases:
228 // detector1 - to test case of one class 'cat'
229 // detector12 - to test case of two (several) classes 'cat' and car
230
231 // Load detectors
232 LatentSvmDetector detector1( vector<string>(1,model_path_cat) );
233
234 vector<string> models_pathes(2);
235 models_pathes[0] = model_path_cat;
236 models_pathes[1] = model_path_car;
237 LatentSvmDetector detector12( models_pathes );
238
239 if( detector1.empty() || detector12.empty() || detector12.getClassCount() != 2 )
240 {
241 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_TEST_DATA );
242 return;
243 }
244
245 // 1. Test method detect
246 // Run detectors
247 vector<LatentSvmDetector::ObjectDetection> detections1_cat, detections12_cat, detections12_cars;
248 detector1.detect( image_cat, detections1_cat, 0.5, numThreads );
249 detector12.detect( image_cat, detections12_cat, 0.5, numThreads );
250 detector12.detect( image_cars, detections12_cars, 0.5, numThreads );
251
252 // Load true results
253 FileStorage fs( true_res_path, FileStorage::READ );
254 if( fs.isOpened() )
255 {
256 vector<LatentSvmDetector::ObjectDetection> true_detections1_cat, true_detections12_cat, true_detections12_cars;
257 readDetections( fs, "detections1_cat", true_detections1_cat );
258 readDetections( fs, "detections12_cat", true_detections12_cat );
259 readDetections( fs, "detections12_cars", true_detections12_cars );
260
261
262 if( !compareResults(detections1_cat, true_detections1_cat, 1, score_thr) )
263 {
264 std::cerr << "Results of detector1 are invalid on image cat.png" << std::endl;
265 ts->set_failed_test_info( cvtest::TS::FAIL_MISMATCH );
266 }
267 if( !compareResults(detections12_cat, true_detections12_cat, 1, score_thr) )
268 {
269 std::cerr << "Results of detector12 are invalid on image cat.png" << std::endl;
270 ts->set_failed_test_info( cvtest::TS::FAIL_MISMATCH );
271 }
272 if( !compareResults(detections12_cars, true_detections12_cars, 1, score_thr) )
273 {
274 std::cerr << "Results of detector12 are invalid on image cars.png" << std::endl;
275 ts->set_failed_test_info( cvtest::TS::FAIL_MISMATCH );
276 }
277 }
278 else
279 {
280 fs.open( true_res_path, FileStorage::WRITE );
281 if( fs.isOpened() )
282 {
283 writeDetections( fs, "detections1_cat", detections1_cat );
284 writeDetections( fs, "detections12_cat", detections12_cat );
285 writeDetections( fs, "detections12_cars", detections12_cars );
286 }
287 else
288 std::cerr << "File " << true_res_path << " cann't be opened to save test results" << std::endl;
289 }
290
291 // 2. Simple tests of other methods
292 if( detector1.getClassCount() != 1 || detector1.getClassNames()[0] != "cat" )
293 {
294 std::cerr << "Incorrect result of method getClassNames() or getClassCount()" << std::endl;
295 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_OUTPUT);
296 }
297
298 detector1.clear();
299 if( !detector1.empty() )
300 {
301 std::cerr << "There is a bug in method clear() or empty()" << std::endl;
302 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_OUTPUT);
303 }
304
305 ts->set_failed_test_info( cvtest::TS::OK);
306}
