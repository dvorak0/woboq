57StabilizerBase::StabilizerBase()
58{
59 setLog(new NullLog());
60 setFrameSource(new NullFrameSource());
61 setMotionEstimator(new PyrLkRobustMotionEstimator());
62 setDeblurer(new NullDeblurer());
63 setInpainter(new NullInpainter());
64 setRadius(15);
65 setTrimRatio(0);
66 setCorrectionForInclusion(false);
67 setBorderMode(BORDER_REPLICATE);
68}

71void StabilizerBase::setUp(int cacheSize, const Mat &frame)
72{
73 InpainterBase *_inpainter = static_cast<InpainterBase*>(inpainter_);
74 doInpainting_ = dynamic_cast<NullInpainter*>(_inpainter) == 0;
75 if (doInpainting_)
76 {
77 inpainter_->setRadius(radius_);
78 inpainter_->setFrames(frames_);
79 inpainter_->setMotions(motions_);
80 inpainter_->setStabilizedFrames(stabilizedFrames_);
81 inpainter_->setStabilizationMotions(stabilizationMotions_);
82 inpainter_->update();
83 }
84
85 DeblurerBase *deblurer = static_cast<DeblurerBase*>(deblurer_);
86 doDeblurring_ = dynamic_cast<NullDeblurer*>(deblurer) == 0;
87 if (doDeblurring_)
88 {
89 blurrinessRates_.resize(cacheSize);
90 float blurriness = calcBlurriness(frame);
91 for (int i = -radius_; i <= 0; ++i)
92 at(i, blurrinessRates_) = blurriness;
93 deblurer_->setRadius(radius_);
94 deblurer_->setFrames(frames_);
95 deblurer_->setMotions(motions_);
96 deblurer_->setBlurrinessRates(blurrinessRates_);
97 deblurer_->update();
98 }
99
100 log_->print("processing frames");
101}

104Mat StabilizerBase::nextStabilizedFrame()
105{
106 if (curStabilizedPos_ == curPos_ && curStabilizedPos_ != -1)
107 return Mat(); // we've processed all frames already
108
109 bool processed;
110 do processed = doOneIteration();
111 while (processed && curStabilizedPos_ == -1);
112
113 if (curStabilizedPos_ == -1)
114 return Mat(); // frame source is empty
115
116 const Mat &stabilizedFrame = at(curStabilizedPos_, stabilizedFrames_);
117 int dx = static_cast<int>(floor(trimRatio_ * stabilizedFrame.cols));
118 int dy = static_cast<int>(floor(trimRatio_ * stabilizedFrame.rows));
119 return stabilizedFrame(Rect(dx, dy, stabilizedFrame.cols - 2*dx, stabilizedFrame.rows - 2*dy));
120}

122
123bool StabilizerBase::doOneIteration()
124{
125 Mat frame = frameSource_->nextFrame();
126 if (!frame.empty())
127 {
128 curPos_++;
129
130 if (curPos_ > 0)
131 {
132 at(curPos_, frames_) = frame;
133
134 if (doDeblurring_)
135 at(curPos_, blurrinessRates_) = calcBlurriness(frame);
136
137 estimateMotion();
138
139 if (curPos_ >= radius_)
140 {
141 curStabilizedPos_ = curPos_ - radius_;
142 stabilizeFrame();
143 }
144 }
145 else
146 setUp(frame);
147
148 log_->print(".");
149 return true;
150 }
151
152 if (curStabilizedPos_ < curPos_)
153 {
154 curStabilizedPos_++;
155 at(curStabilizedPos_ + radius_, frames_) = at(curPos_, frames_);
156 at(curStabilizedPos_ + radius_ - 1, motions_) = at(curPos_ - 1, motions_);
157 stabilizeFrame();
158
159 log_->print(".");
160 return true;
161 }
162
163 return false;
164}

167void StabilizerBase::stabilizeFrame(const Mat &stabilizationMotion)
168{
169 Mat stabilizationMotion_;
170 if (doCorrectionForInclusion_)
171 stabilizationMotion_ = ensureInclusionConstraint(stabilizationMotion, frameSize_, trimRatio_);
172 else
173 stabilizationMotion_ = stabilizationMotion.clone();
174
175 at(curStabilizedPos_, stabilizationMotions_) = stabilizationMotion_;
176
177 if (doDeblurring_)
178 {
179 at(curStabilizedPos_, frames_).copyTo(preProcessedFrame_);
180 deblurer_->deblur(curStabilizedPos_, preProcessedFrame_);
181 }
182 else
183 preProcessedFrame_ = at(curStabilizedPos_, frames_);
184
185 // apply stabilization transformation
186 warpAffine(
187 preProcessedFrame_, at(curStabilizedPos_, stabilizedFrames_),
188 stabilizationMotion_(Rect(0,0,3,2)), frameSize_, INTER_LINEAR, borderMode_);
189
190 if (doInpainting_)
191 {
192 warpAffine(
193 frameMask_, at(curStabilizedPos_, stabilizedMasks_),
194 stabilizationMotion_(Rect(0,0,3,2)), frameSize_, INTER_NEAREST);
195
196 erode(at(curStabilizedPos_, stabilizedMasks_), at(curStabilizedPos_, stabilizedMasks_),
197 Mat());
198
199 at(curStabilizedPos_, stabilizedMasks_).copyTo(inpaintingMask_);
200
201 inpainter_->inpaint(
202 curStabilizedPos_, at(curStabilizedPos_, stabilizedFrames_), inpaintingMask_);
203 }
204}

214void OnePassStabilizer::resetImpl()
215{
216 curPos_ = -1;
217 curStabilizedPos_ = -1;
218 frames_.clear();
219 motions_.clear();
220 stabilizedFrames_.clear();
221 stabilizationMotions_.clear();
222 doDeblurring_ = false;
223 doInpainting_ = false;

227void OnePassStabilizer::setUp(Mat &firstFrame)
228{
229 frameSize_ = firstFrame.size();
230 frameMask_.create(frameSize_, CV_8U);
231 frameMask_.setTo(255);
232
233 int cacheSize = 2*radius_ + 1;
234
235 frames_.resize(cacheSize);
236 stabilizedFrames_.resize(cacheSize);
237 stabilizedMasks_.resize(cacheSize);
238 motions_.resize(cacheSize);
239 stabilizationMotions_.resize(cacheSize);
240
241 for (int i = -radius_; i < 0; ++i)
242 {
243 at(i, motions_) = Mat::eye(3, 3, CV_32F);
244 at(i, frames_) = firstFrame;
245 }
246
247 at(0, frames_) = firstFrame;
248
249 motionFilter_->setRadius(radius_);
250 motionFilter_->update();
251
252 StabilizerBase::setUp(cacheSize, firstFrame);
253}

256void OnePassStabilizer::estimateMotion()
257{
258 at(curPos_ - 1, motions_) = motionEstimator_->estimate(
259 at(curPos_ - 1, frames_), at(curPos_, frames_));
260}

263void OnePassStabilizer::stabilizeFrame()
264{
265 Mat stabilizationMotion = motionFilter_->stabilize(curStabilizedPos_, &motions_[0], (int)motions_.size());
266 StabilizerBase::stabilizeFrame(stabilizationMotion);
267}

285vector<Mat> TwoPassStabilizer::motions() const
286{
287 if (frameCount_ == 0)
288 return vector<Mat>();
289 vector<Mat> res(frameCount_ - 1);
290 copy(motions_.begin(), motions_.begin() + frameCount_ - 1, res.begin());
291 return res;
292}

295void TwoPassStabilizer::resetImpl()
296{
297 isPrePassDone_ = false;
298 frameCount_ = 0;
299 curPos_ = -1;
300 curStabilizedPos_ = -1;
301 frames_.clear();
302 motions_.clear();
303 stabilizedFrames_.clear();
304 stabilizationMotions_.clear();
305 doDeblurring_ = false;
306 doInpainting_ = false;
307}

309
310void TwoPassStabilizer::runPrePassIfNecessary()
311{
312 if (!isPrePassDone_)
313 {
314 log_->print("first pass: estimating motions");
315
316 Mat prevFrame, frame;
317
318 while (!(frame = frameSource_->nextFrame()).empty())
319 {
320 if (frameCount_ > 0)
321 motions_.push_back(motionEstimator_->estimate(prevFrame, frame));
322 else
323 {
324 frameSize_ = frame.size();
325 frameMask_.create(frameSize_, CV_8U);
326 frameMask_.setTo(255);
327 }
328
329 prevFrame = frame;
330 frameCount_++;
331
332 log_->print(".");
333 }
334
335 for (int i = 0; i < radius_; ++i)
336 motions_.push_back(Mat::eye(3, 3, CV_32F));
337 log_->print("\n");
338
339 IMotionStabilizer *_motionStabilizer = static_cast<IMotionStabilizer*>(motionStabilizer_);
340 MotionFilterBase *motionFilterBase = dynamic_cast<MotionFilterBase*>(_motionStabilizer);
341 if (motionFilterBase)
342 {
343 motionFilterBase->setRadius(radius_);
344 motionFilterBase->update();
345 }
346
347 stabilizationMotions_.resize(frameCount_);
348 motionStabilizer_->stabilize(&motions_[0], frameCount_, &stabilizationMotions_[0]);
349
350 if (mustEstTrimRatio_)
351 {
352 trimRatio_ = 0;
353 for (int i = 0; i < frameCount_; ++i)
354 {
355 Mat S = stabilizationMotions_[i];
356 trimRatio_ = std::max(trimRatio_, estimateOptimalTrimRatio(S, frameSize_));
357 }
358 log_->print("estimated trim ratio: %f\n", static_cast<double>(trimRatio_));
359 }
360
361 isPrePassDone_ = true;
362 frameSource_->reset();
363 }
364}

367void TwoPassStabilizer::setUp(Mat &firstFrame)
368{
369 int cacheSize = 2*radius_ + 1;
370
371 frames_.resize(cacheSize);
372 stabilizedFrames_.resize(cacheSize);
373 stabilizedMasks_.resize(cacheSize);
374
375 for (int i = -radius_; i <= 0; ++i)
376 at(i, frames_) = firstFrame;
377
378 StabilizerBase::setUp(cacheSize, firstFrame);
