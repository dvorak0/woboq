57inline float intensity(const cv::Point3_<uchar> &bgr)
58{
59 return 0.3f*bgr.x + 0.59f*bgr.y + 0.11f*bgr.z;
60}

62template <typename T> inline T& at(int index, T *items, int size)
63{
64 return items[cv::borderInterpolate(index, size, cv::BORDER_WRAP)];
65}
66
67template <typename T> inline const T& at(int index, const T *items, int size)
68{
69 return items[cv::borderInterpolate(index, size, cv::BORDER_WRAP)];
70}
71
72template <typename T> inline T& at(int index, std::vector<T> &items)
73{
74 return at(index, &items[0], static_cast<int>(items.size()));
75}
76
77template <typename T> inline const T& at(int index, const std::vector<T> &items)
78{
79 return items[cv::borderInterpolate(index, static_cast<int>(items.size()), cv::BORDER_WRAP)];
80}
