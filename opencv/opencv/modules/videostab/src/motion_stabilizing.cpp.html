61void GaussianMotionFilter::update()
62{
63 float sigma = stdev_ > 0.f ? stdev_ : sqrt(static_cast<float>(radius_));
64 float sum = 0;
65 weight_.resize(2*radius_ + 1);
66 for (int i = -radius_; i <= radius_; ++i)
67 sum += weight_[radius_ + i] = std::exp(-i*i/(sigma*sigma));
68 for (int i = -radius_; i <= radius_; ++i)
69 weight_[radius_ + i] /= sum;
70}

73Mat GaussianMotionFilter::stabilize(int index, const Mat *motions, int size) const
74{
75 const Mat &cur = at(index, motions, size);
76 Mat res = Mat::zeros(cur.size(), cur.type());
77 float sum = 0.f;
78 for (int i = std::max(index - radius_, 0); i <= index + radius_; ++i)
79 {
80 res += weight_[radius_ + i - index] * getMotion(index, i, motions, size);
81 sum += weight_[radius_ + i - index];
82 }
83 return res / sum;
84}

87static inline int areaSign(Point2f a, Point2f b, Point2f c)
88{
89 double area = (b-a).cross(c-a);
90 if (area < -1e-5) return -1;
91 if (area > 1e-5) return 1;
92 return 0;
93}

96static inline bool segmentsIntersect(Point2f a, Point2f b, Point2f c, Point2f d)
97{
98 return areaSign(a,b,c) * areaSign(a,b,d) < 0 &&
99 areaSign(c,d,a) * areaSign(c,d,b) < 0;
100}

103// Checks if rect a (with sides parallel to axis) is inside rect b (arbitrary).
104// Rects must be passed in the [(0,0), (w,0), (w,h), (0,h)] order.
105static inline bool isRectInside(const Point2f a[4], const Point2f b[4])
106{
107 for (int i = 0; i < 4; ++i)
108 if (b[i].x > a[0].x && b[i].x < a[2].x && b[i].y > a[0].y && b[i].y < a[2].y)
109 return false;
110 for (int i = 0; i < 4; ++i)
111 for (int j = 0; j < 4; ++j)
112 if (segmentsIntersect(a[i], a[(i+1)%4], b[j], b[(j+1)%4]))
113 return false;
114 return true;
115}

118static inline bool isGoodMotion(const float M[], float w, float h, float dx, float dy)
119{
120 Point2f pt[4] = {Point2f(0,0), Point2f(w,0), Point2f(w,h), Point2f(0,h)};
121 Point2f Mpt[4];
122
123 for (int i = 0; i < 4; ++i)
124 {
125 Mpt[i].x = M[0]*pt[i].x + M[1]*pt[i].y + M[2];
126 Mpt[i].y = M[3]*pt[i].x + M[4]*pt[i].y + M[5];
127 }
128
129 pt[0] = Point2f(dx, dy);
130 pt[1] = Point2f(w - dx, dy);
131 pt[2] = Point2f(w - dx, h - dy);
132 pt[3] = Point2f(dx, h - dy);
133
134 return isRectInside(pt, Mpt);
135}

138static inline void relaxMotion(const float M[], float t, float res[])
139{
140 res[0] = M[0]*(1.f-t) + t;
141 res[1] = M[1]*(1.f-t);
142 res[2] = M[2]*(1.f-t);
143 res[3] = M[3]*(1.f-t);
144 res[4] = M[4]*(1.f-t) + t;
145 res[5] = M[5]*(1.f-t);
146}

149Mat ensureInclusionConstraint(const Mat &M, Size size, float trimRatio)
150{
151 CV_Assert(M.size() == Size(3,3) && M.type() == CV_32F);
152
153 const float w = static_cast<float>(size.width);
154 const float h = static_cast<float>(size.height);
155 const float dx = floor(w * trimRatio);
156 const float dy = floor(h * trimRatio);
157 const float srcM[6] =
158 {M.at<float>(0,0), M.at<float>(0,1), M.at<float>(0,2),
159 M.at<float>(1,0), M.at<float>(1,1), M.at<float>(1,2)};
160
161 float curM[6];
162 float t = 0;
163 relaxMotion(srcM, t, curM);
164 if (isGoodMotion(curM, w, h, dx, dy))
165 return M;
166
167 float l = 0, r = 1;
168 while (r - l > 1e-3f)
169 {
170 t = (l + r) * 0.5f;
171 relaxMotion(srcM, t, curM);
172 if (isGoodMotion(curM, w, h, dx, dy))
173 r = t;
174 else
175 l = t;
176 t = r;
177 relaxMotion(srcM, r, curM);
178 }
179
180 return (1 - r) * M + r * Mat::eye(3, 3, CV_32F);
181}

184// TODO can be estimated for O(1) time
185float estimateOptimalTrimRatio(const Mat &M, Size size)
186{
187 CV_Assert(M.size() == Size(3,3) && M.type() == CV_32F);
188
189 const float w = static_cast<float>(size.width);
190 const float h = static_cast<float>(size.height);
191 Mat_<float> M_(M);
192
193 Point2f pt[4] = {Point2f(0,0), Point2f(w,0), Point2f(w,h), Point2f(0,h)};
194 Point2f Mpt[4];
195
196 for (int i = 0; i < 4; ++i)
197 {
198 Mpt[i].x = M_(0,0)*pt[i].x + M_(0,1)*pt[i].y + M_(0,2);
199 Mpt[i].y = M_(1,0)*pt[i].x + M_(1,1)*pt[i].y + M_(1,2);
200 }
201
202 float l = 0, r = 0.5f;
203 while (r - l > 1e-3f)
204 {
205 float t = (l + r) * 0.5f;
206 float dx = floor(w * t);
207 float dy = floor(h * t);
208 pt[0] = Point2f(dx, dy);
209 pt[1] = Point2f(w - dx, dy);
210 pt[2] = Point2f(w - dx, h - dy);
211 pt[3] = Point2f(dx, h - dy);
212 if (isRectInside(pt, Mpt))
213 r = t;
214 else
215 l = t;
216 }
217
218 return r;
219}
