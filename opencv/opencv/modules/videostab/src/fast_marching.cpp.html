53float FastMarchingMethod::solve(int x1, int y1, int x2, int y2) const
54{
55 float sol = inf_;
56 if (y1 >=0 && y1 < flag_.rows && x1 >= 0 && x1 < flag_.cols && flag_(y1,x1) == KNOWN)
57 {
58 float t1 = dist_(y1,x1);
59 if (y2 >=0 && y2 < flag_.rows && x2 >= 0 && x2 < flag_.cols && flag_(y2,x2) == KNOWN)
60 {
61 float t2 = dist_(y2,x2);
62 float r = sqrt(2 - sqr(t1 - t2));
63 float s = (t1 + t2 - r) / 2;
64
65 if (s >= t1 && s >= t2)
66 sol = s;
67 else
68 {
69 s += r;
70 if (s >= t1 && s >= t2)
71 sol = s;
72 }
73 }
74 else
75 sol = 1 + t1;
76 }
77 else if (y2 >=0 && y2 < flag_.rows && x2 >= 0 && x2 < flag_.cols && flag_(y2,x2) == KNOWN)
78 sol = 1 + dist_(y2,x1);
79 return sol;
80}

83void FastMarchingMethod::heapUp(int idx)
84{
85 int p = (idx-1)/2;
86 while (idx > 0 && narrowBand_[idx] < narrowBand_[p])
87 {
88 std::swap(indexOf(narrowBand_[p]), indexOf(narrowBand_[idx]));
89 std::swap(narrowBand_[p], narrowBand_[idx]);
90 idx = p;
91 p = (idx-1)/2;
92 }
93}

96void FastMarchingMethod::heapDown(int idx)
97{
98 int l, r, smallest;
99 for(;;)
100 {
101 l = 2*idx+1;
102 r = 2*idx+2;
103 smallest = idx;
104
105 if (l < size_ && narrowBand_[l] < narrowBand_[smallest]) smallest = l;
106 if (r < size_ && narrowBand_[r] < narrowBand_[smallest]) smallest = r;
107
108 if (smallest == idx)
109 break;
110 else
111 {
112 std::swap(indexOf(narrowBand_[idx]), indexOf(narrowBand_[smallest]));
113 std::swap(narrowBand_[idx], narrowBand_[smallest]);
114 idx = smallest;
115 }
116 }
117}

120void FastMarchingMethod::heapAdd(const DXY &dxy)
121{
122 if (static_cast<int>(narrowBand_.size()) < size_ + 1)
123 narrowBand_.resize(size_*2 + 1);
124 narrowBand_[size_] = dxy;
125 indexOf(dxy) = size_++;
126 heapUp(size_-1);
127}

129
130void FastMarchingMethod::heapRemoveMin()
131{
132 if (size_ > 0)
133 {
134 size_--;
135 std::swap(indexOf(narrowBand_[0]), indexOf(narrowBand_[size_]));
136 std::swap(narrowBand_[0], narrowBand_[size_]);
137 heapDown(0);
138 }
139}
