55 InputArray frame0, InputArray frame1, InputArray points0, InputOutputArray points1,

69void DensePyrLkOptFlowEstimatorGpu::run(
70 InputArray frame0, InputArray frame1, InputOutputArray flowX, InputOutputArray flowY,
71 OutputArray errors)
72{
73 frame0_.upload(frame0.getMat());
74 frame1_.upload(frame1.getMat());
75
76 optFlowEstimator_.winSize = winSize_;
77 optFlowEstimator_.maxLevel = maxLevel_;
78 if (errors.needed())
79 {
80 optFlowEstimator_.dense(frame0_, frame1_, flowX_, flowY_, &errors_);
81 errors_.download(errors.getMatRef());
82 }
83 else
84 optFlowEstimator_.dense(frame0_, frame1_, flowX_, flowY_);
85
86 flowX_.download(flowX.getMatRef());
87 flowY_.download(flowY.getMatRef());
88}
