8int solve_deg2(double a, double b, double c, double & x1, double & x2)
9{
10 double delta = b * b - 4 * a * c;
11
12 if (delta < 0) return 0;
13
14 double inv_2a = 0.5 / a;
15
16 if (delta == 0) {
17 x1 = -b * inv_2a;
18 x2 = x1;
19 return 1;
20 }
21
22 double sqrt_delta = sqrt(delta);
23 x1 = (-b + sqrt_delta) * inv_2a;
24 x2 = (-b - sqrt_delta) * inv_2a;
25 return 2;
26}

29/// Reference : Eric W. Weisstein. "Cubic Equation." From MathWorld--A Wolfram Web Resource.
30/// http://mathworld.wolfram.com/CubicEquation.html
31/// \return Number of real roots found.
32int solve_deg3(double a, double b, double c, double d,
33 double & x0, double & x1, double & x2)
34{
35 if (a == 0) {
36 // Solve second order sytem
37 if (b == 0) {
38 // Solve first order system
39 if (c == 0)
40 return 0;
41
42 x0 = -d / c;
43 return 1;
44 }
45
46 x2 = 0;
47 return solve_deg2(b, c, d, x0, x1);
48 }
49
50 // Calculate the normalized form x^3 + a2 * x^2 + a1 * x + a0 = 0
51 double inv_a = 1. / a;
52 double b_a = inv_a * b, b_a2 = b_a * b_a;
53 double c_a = inv_a * c;
54 double d_a = inv_a * d;
55
56 // Solve the cubic equation
57 double Q = (3 * c_a - b_a2) / 9;
58 double R = (9 * b_a * c_a - 27 * d_a - 2 * b_a * b_a2) / 54;
59 double Q3 = Q * Q * Q;
60 double D = Q3 + R * R;
61 double b_a_3 = (1. / 3.) * b_a;
62
63 if (Q == 0) {
64 if(R == 0) {
65 x0 = x1 = x2 = - b_a_3;
66 return 3;
67 }
68 else {
69 x0 = pow(2 * R, 1 / 3.0) - b_a_3;
70 return 1;
71 }
72 }
73
74 if (D <= 0) {
75 // Three real roots
76 double theta = acos(R / sqrt(-Q3));
77 double sqrt_Q = sqrt(-Q);
78 x0 = 2 * sqrt_Q * cos(theta / 3.0) - b_a_3;
79 x1 = 2 * sqrt_Q * cos((theta + 2 * CV_PI)/ 3.0) - b_a_3;
80 x2 = 2 * sqrt_Q * cos((theta + 4 * CV_PI)/ 3.0) - b_a_3;
81
82 return 3;
83 }
84
85 // D > 0, only one real root
86 double AD = pow(fabs(R) + sqrt(D), 1.0 / 3.0) * (R > 0 ? 1 : (R < 0 ? -1 : 0));
87 double BD = (AD == 0) ? 0 : -Q / AD;
88
89 // Calculate the only real root
90 x0 = AD + BD - b_a_3;
91
92 return 1;
93}

95/// Reference : Eric W. Weisstein. "Quartic Equation." From MathWorld--A Wolfram Web Resource.
96/// http://mathworld.wolfram.com/QuarticEquation.html
97/// \return Number of real roots found.
98int solve_deg4(double a, double b, double c, double d, double e,
99 double & x0, double & x1, double & x2, double & x3)
100{
101 if (a == 0) {
102 x3 = 0;
103 return solve_deg3(b, c, d, e, x0, x1, x2);
104 }
105
106 // Normalize coefficients
107 double inv_a = 1. / a;
108 b *= inv_a; c *= inv_a; d *= inv_a; e *= inv_a;
109 double b2 = b * b, bc = b * c, b3 = b2 * b;
110
111 // Solve resultant cubic
112 double r0, r1, r2;
113 int n = solve_deg3(1, -c, d * b - 4 * e, 4 * c * e - d * d - b2 * e, r0, r1, r2);
114 if (n == 0) return 0;
115
116 // Calculate R^2
117 double R2 = 0.25 * b2 - c + r0, R;
118 if (R2 < 0)
119 return 0;
120
121 R = sqrt(R2);
122 double inv_R = 1. / R;
123
124 int nb_real_roots = 0;
125
126 // Calculate D^2 and E^2
127 double D2, E2;
128 if (R < 10E-12) {
129 double temp = r0 * r0 - 4 * e;
130 if (temp < 0)
131 D2 = E2 = -1;
132 else {
133 double sqrt_temp = sqrt(temp);
134 D2 = 0.75 * b2 - 2 * c + 2 * sqrt_temp;
135 E2 = D2 - 4 * sqrt_temp;
136 }
137 }
138 else {
139 double u = 0.75 * b2 - 2 * c - R2,
140 v = 0.25 * inv_R * (4 * bc - 8 * d - b3);
141 D2 = u + v;
142 E2 = u - v;
143 }
144
145 double b_4 = 0.25 * b, R_2 = 0.5 * R;
146 if (D2 >= 0) {
147 double D = sqrt(D2);
148 nb_real_roots = 2;
149 double D_2 = 0.5 * D;
150 x0 = R_2 + D_2 - b_4;
151 x1 = x0 - D;
152 }
153
154 // Calculate E^2
155 if (E2 >= 0) {
156 double E = sqrt(E2);
157 double E_2 = 0.5 * E;
158 if (nb_real_roots == 0) {
159 x0 = - R_2 + E_2 - b_4;
160 x1 = x0 - E;
161 nb_real_roots = 2;
162 }
163 else {
164 x2 = - R_2 + E_2 - b_4;
165 x3 = x2 - E;
166 nb_real_roots = 4;
167 }
168 }
169
170 return nb_real_roots;
171}
