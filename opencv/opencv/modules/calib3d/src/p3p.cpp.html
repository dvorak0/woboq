37{
38 double rotation_matrix[3][3], translation[3];
39 std::vector<double> points;
40 if (opoints.depth() == ipoints.depth())
41 {
42 if (opoints.depth() == CV_32F)

48 extract_points<cv::Point3f,cv::Point2d>(opoints, ipoints, points);
49 else
50 extract_points<cv::Point3d,cv::Point2f>(opoints, ipoints, points);
51
52 bool result = solve(rotation_matrix, translation, points[0], points[1], points[2], points[3], points[4], points[5],
53 points[6], points[7], points[8], points[9], points[10], points[11], points[12], points[13], points[14],
54 points[15], points[16], points[17], points[18], points[19]);
55 cv::Mat(3, 1, CV_64F, translation).copyTo(tvec);

60bool p3p::solve(double R[3][3], double t[3],
61 double mu0, double mv0, double X0, double Y0, double Z0,
62 double mu1, double mv1, double X1, double Y1, double Z1,
63 double mu2, double mv2, double X2, double Y2, double Z2,
64 double mu3, double mv3, double X3, double Y3, double Z3)
65{
66 double Rs[4][3][3], ts[4][3];
67
68 int n = solve(Rs, ts, mu0, mv0, X0, Y0, Z0, mu1, mv1, X1, Y1, Z1, mu2, mv2, X2, Y2, Z2);
69
70 if (n == 0)
71 return false;
72
73 int ns = 0;
74 double min_reproj = 0;
75 for(int i = 0; i < n; i++) {
76 double X3p = Rs[i][0][0] * X3 + Rs[i][0][1] * Y3 + Rs[i][0][2] * Z3 + ts[i][0];
77 double Y3p = Rs[i][1][0] * X3 + Rs[i][1][1] * Y3 + Rs[i][1][2] * Z3 + ts[i][1];
78 double Z3p = Rs[i][2][0] * X3 + Rs[i][2][1] * Y3 + Rs[i][2][2] * Z3 + ts[i][2];
79 double mu3p = cx + fx * X3p / Z3p;
80 double mv3p = cy + fy * Y3p / Z3p;
81 double reproj = (mu3p - mu3) * (mu3p - mu3) + (mv3p - mv3) * (mv3p - mv3);
82 if (i == 0 || min_reproj > reproj) {
83 ns = i;
84 min_reproj = reproj;
85 }
86 }
87
88 for(int i = 0; i < 3; i++) {
89 for(int j = 0; j < 3; j++)
90 R[i][j] = Rs[ns][i][j];
91 t[i] = ts[ns][i];
92 }
93
94 return true;
95}

97int p3p::solve(double R[4][3][3], double t[4][3],
98 double mu0, double mv0, double X0, double Y0, double Z0,
99 double mu1, double mv1, double X1, double Y1, double Z1,
100 double mu2, double mv2, double X2, double Y2, double Z2)
101{
102 double mk0, mk1, mk2;
103 double norm;
104
105 mu0 = inv_fx * mu0 - cx_fx;
106 mv0 = inv_fy * mv0 - cy_fy;
107 norm = sqrt(mu0 * mu0 + mv0 * mv0 + 1);
108 mk0 = 1. / norm; mu0 *= mk0; mv0 *= mk0;
109
110 mu1 = inv_fx * mu1 - cx_fx;
111 mv1 = inv_fy * mv1 - cy_fy;
112 norm = sqrt(mu1 * mu1 + mv1 * mv1 + 1);
113 mk1 = 1. / norm; mu1 *= mk1; mv1 *= mk1;
114
115 mu2 = inv_fx * mu2 - cx_fx;
116 mv2 = inv_fy * mv2 - cy_fy;
117 norm = sqrt(mu2 * mu2 + mv2 * mv2 + 1);
118 mk2 = 1. / norm; mu2 *= mk2; mv2 *= mk2;
119
120 double distances[3];
121 distances[0] = sqrt( (X1 - X2) * (X1 - X2) + (Y1 - Y2) * (Y1 - Y2) + (Z1 - Z2) * (Z1 - Z2) );
122 distances[1] = sqrt( (X0 - X2) * (X0 - X2) + (Y0 - Y2) * (Y0 - Y2) + (Z0 - Z2) * (Z0 - Z2) );
123 distances[2] = sqrt( (X0 - X1) * (X0 - X1) + (Y0 - Y1) * (Y0 - Y1) + (Z0 - Z1) * (Z0 - Z1) );
124
125 // Calculate angles
126 double cosines[3];
127 cosines[0] = mu1 * mu2 + mv1 * mv2 + mk1 * mk2;
128 cosines[1] = mu0 * mu2 + mv0 * mv2 + mk0 * mk2;
129 cosines[2] = mu0 * mu1 + mv0 * mv1 + mk0 * mk1;
130
131 double lengths[4][3];
132 int n = solve_for_lengths(lengths, distances, cosines);
133
134 int nb_solutions = 0;
135 for(int i = 0; i < n; i++) {
136 double M_orig[3][3];
137
138 M_orig[0][0] = lengths[i][0] * mu0;
139 M_orig[0][1] = lengths[i][0] * mv0;
140 M_orig[0][2] = lengths[i][0] * mk0;
141
142 M_orig[1][0] = lengths[i][1] * mu1;
143 M_orig[1][1] = lengths[i][1] * mv1;
144 M_orig[1][2] = lengths[i][1] * mk1;
145
146 M_orig[2][0] = lengths[i][2] * mu2;
147 M_orig[2][1] = lengths[i][2] * mv2;
148 M_orig[2][2] = lengths[i][2] * mk2;
149
150 if (!align(M_orig, X0, Y0, Z0, X1, Y1, Z1, X2, Y2, Z2, R[nb_solutions], t[nb_solutions]))
151 continue;
152
153 nb_solutions++;
154 }
155
156 return nb_solutions;
157}

159/// Given 3D distances between three points and cosines of 3 angles at the apex, calculates
160/// the lentghs of the line segments connecting projection center (P) and the three 3D points (A, B, C).
161/// Returned distances are for |PA|, |PB|, |PC| respectively.
162/// Only the solution to the main branch.
163/// Reference : X.S. Gao, X.-R. Hou, J. Tang, H.-F. Chang; "Complete Solution Classification for the Perspective-Three-Point Problem"
164/// IEEE Trans. on PAMI, vol. 25, No. 8, August 2003
165/// \param lengths3D Lengths of line segments up to four solutions.
166/// \param dist3D Distance between 3D points in pairs |BC|, |AC|, |AB|.
167/// \param cosines Cosine of the angles /_BPC, /_APC, /_APB.
168/// \returns Number of solutions.
169/// WARNING: NOT ALL THE DEGENERATE CASES ARE IMPLEMENTED
170
171int p3p::solve_for_lengths(double lengths[4][3], double distances[3], double cosines[3])
172{
173 double p = cosines[0] * 2;
174 double q = cosines[1] * 2;
175 double r = cosines[2] * 2;
176
177 double inv_d22 = 1. / (distances[2] * distances[2]);
178 double a = inv_d22 * (distances[0] * distances[0]);
179 double b = inv_d22 * (distances[1] * distances[1]);
180
181 double a2 = a * a, b2 = b * b, p2 = p * p, q2 = q * q, r2 = r * r;
182 double pr = p * r, pqr = q * pr;
183
184 // Check reality condition (the four points should not be coplanar)
185 if (p2 + q2 + r2 - pqr - 1 == 0)
186 return 0;
187
188 double ab = a * b, a_2 = 2*a;
189
190 double A = -2 * b + b2 + a2 + 1 + ab*(2 - r2) - a_2;
191
192 // Check reality condition
193 if (A == 0) return 0;
194
195 double a_4 = 4*a;
196
197 double B = q*(-2*(ab + a2 + 1 - b) + r2*ab + a_4) + pr*(b - b2 + ab);
198 double C = q2 + b2*(r2 + p2 - 2) - b*(p2 + pqr) - ab*(r2 + pqr) + (a2 - a_2)*(2 + q2) + 2;
199 double D = pr*(ab-b2+b) + q*((p2-2)*b + 2 * (ab - a2) + a_4 - 2);
200 double E = 1 + 2*(b - a - ab) + b2 - b*p2 + a2;
201
202 double temp = (p2*(a-1+b) + r2*(a-1-b) + pqr - a*pqr);
203 double b0 = b * temp * temp;
204 // Check reality condition
205 if (b0 == 0)
206 return 0;
207
208 double real_roots[4];
209 int n = solve_deg4(A, B, C, D, E, real_roots[0], real_roots[1], real_roots[2], real_roots[3]);
210
211 if (n == 0)
212 return 0;
213
214 int nb_solutions = 0;
215 double r3 = r2*r, pr2 = p*r2, r3q = r3 * q;
216 double inv_b0 = 1. / b0;
217
218 // For each solution of x
219 for(int i = 0; i < n; i++) {
220 double x = real_roots[i];
221
222 // Check reality condition
223 if (x <= 0)
224 continue;
225
226 double x2 = x*x;
227
228 double b1 =
229 ((1-a-b)*x2 + (q*a-q)*x + 1 - a + b) *
230 (((r3*(a2 + ab*(2 - r2) - a_2 + b2 - 2*b + 1)) * x +
231
232 (r3q*(2*(b-a2) + a_4 + ab*(r2 - 2) - 2) + pr2*(1 + a2 + 2*(ab-a-b) + r2*(b - b2) + b2))) * x2 +
233
234 (r3*(q2*(1-2*a+a2) + r2*(b2-ab) - a_4 + 2*(a2 - b2) + 2) + r*p2*(b2 + 2*(ab - b - a) + 1 + a2) + pr2*q*(a_4 + 2*(b - ab - a2) - 2 - r2*b)) * x +
235
236 2*r3q*(a_2 - b - a2 + ab - 1) + pr2*(q2 - a_4 + 2*(a2 - b2) + r2*b + q2*(a2 - a_2) + 2) +
237 p2*(p*(2*(ab - a - b) + a2 + b2 + 1) + 2*q*r*(b + a_2 - a2 - ab - 1)));
238
239 // Check reality condition
240 if (b1 <= 0)
241 continue;
242
243 double y = inv_b0 * b1;
244 double v = x2 + y*y - x*y*r;
245
246 if (v <= 0)
247 continue;
248
249 double Z = distances[2] / sqrt(v);
250 double X = x * Z;
251 double Y = y * Z;
252
253 lengths[nb_solutions][0] = X;
254 lengths[nb_solutions][1] = Y;
255 lengths[nb_solutions][2] = Z;
256
257 nb_solutions++;
258 }
259
260 return nb_solutions;
261}

263bool p3p::align(double M_end[3][3],
264 double X0, double Y0, double Z0,
265 double X1, double Y1, double Z1,
266 double X2, double Y2, double Z2,
267 double R[3][3], double T[3])
268{
269 // Centroids:
270 double C_start[3], C_end[3];
271 for(int i = 0; i < 3; i++) C_end[i] = (M_end[0][i] + M_end[1][i] + M_end[2][i]) / 3;
272 C_start[0] = (X0 + X1 + X2) / 3;
273 C_start[1] = (Y0 + Y1 + Y2) / 3;
274 C_start[2] = (Z0 + Z1 + Z2) / 3;
275
276 // Covariance matrix s:
277 double s[3 * 3];
278 for(int j = 0; j < 3; j++) {
279 s[0 * 3 + j] = (X0 * M_end[0][j] + X1 * M_end[1][j] + X2 * M_end[2][j]) / 3 - C_end[j] * C_start[0];
280 s[1 * 3 + j] = (Y0 * M_end[0][j] + Y1 * M_end[1][j] + Y2 * M_end[2][j]) / 3 - C_end[j] * C_start[1];
281 s[2 * 3 + j] = (Z0 * M_end[0][j] + Z1 * M_end[1][j] + Z2 * M_end[2][j]) / 3 - C_end[j] * C_start[2];
282 }
283
284 double Qs[16], evs[4], U[16];
285
286 Qs[0 * 4 + 0] = s[0 * 3 + 0] + s[1 * 3 + 1] + s[2 * 3 + 2];
287 Qs[1 * 4 + 1] = s[0 * 3 + 0] - s[1 * 3 + 1] - s[2 * 3 + 2];
288 Qs[2 * 4 + 2] = s[1 * 3 + 1] - s[2 * 3 + 2] - s[0 * 3 + 0];
289 Qs[3 * 4 + 3] = s[2 * 3 + 2] - s[0 * 3 + 0] - s[1 * 3 + 1];
290
291 Qs[1 * 4 + 0] = Qs[0 * 4 + 1] = s[1 * 3 + 2] - s[2 * 3 + 1];
292 Qs[2 * 4 + 0] = Qs[0 * 4 + 2] = s[2 * 3 + 0] - s[0 * 3 + 2];
293 Qs[3 * 4 + 0] = Qs[0 * 4 + 3] = s[0 * 3 + 1] - s[1 * 3 + 0];
294 Qs[2 * 4 + 1] = Qs[1 * 4 + 2] = s[1 * 3 + 0] + s[0 * 3 + 1];
295 Qs[3 * 4 + 1] = Qs[1 * 4 + 3] = s[2 * 3 + 0] + s[0 * 3 + 2];
296 Qs[3 * 4 + 2] = Qs[2 * 4 + 3] = s[2 * 3 + 1] + s[1 * 3 + 2];
297
298 jacobi_4x4(Qs, evs, U);
299
300 // Looking for the largest eigen value:
301 int i_ev = 0;
302 double ev_max = evs[i_ev];
303 for(int i = 1; i < 4; i++)
304 if (evs[i] > ev_max)
305 ev_max = evs[i_ev = i];
306
307 // Quaternion:
308 double q[4];
309 for(int i = 0; i < 4; i++)
310 q[i] = U[i * 4 + i_ev];
311
312 double q02 = q[0] * q[0], q12 = q[1] * q[1], q22 = q[2] * q[2], q32 = q[3] * q[3];
313 double q0_1 = q[0] * q[1], q0_2 = q[0] * q[2], q0_3 = q[0] * q[3];
314 double q1_2 = q[1] * q[2], q1_3 = q[1] * q[3];
315 double q2_3 = q[2] * q[3];
316
317 R[0][0] = q02 + q12 - q22 - q32;
318 R[0][1] = 2. * (q1_2 - q0_3);
319 R[0][2] = 2. * (q1_3 + q0_2);
320
321 R[1][0] = 2. * (q1_2 + q0_3);
322 R[1][1] = q02 + q22 - q12 - q32;
323 R[1][2] = 2. * (q2_3 - q0_1);
324
325 R[2][0] = 2. * (q1_3 - q0_2);
326 R[2][1] = 2. * (q2_3 + q0_1);
327 R[2][2] = q02 + q32 - q12 - q22;
328
329 for(int i = 0; i < 3; i++)
330 T[i] = C_end[i] - (R[i][0] * C_start[0] + R[i][1] * C_start[1] + R[i][2] * C_start[2]);
331
332 return true;
333}

335bool p3p::jacobi_4x4(double * A, double * D, double * U)
336{
337 double B[4], Z[4];
338 double Id[16] = {1., 0., 0., 0.,
339 0., 1., 0., 0.,
340 0., 0., 1., 0.,
341 0., 0., 0., 1.};
342
343 memcpy(U, Id, 16 * sizeof(double));
344
345 B[0] = A[0]; B[1] = A[5]; B[2] = A[10]; B[3] = A[15];
346 memcpy(D, B, 4 * sizeof(double));
347 memset(Z, 0, 4 * sizeof(double));
348
349 for(int iter = 0; iter < 50; iter++) {
350 double sum = fabs(A[1]) + fabs(A[2]) + fabs(A[3]) + fabs(A[6]) + fabs(A[7]) + fabs(A[11]);
351
352 if (sum == 0.0)
353 return true;
354
355 double tresh = (iter < 3) ? 0.2 * sum / 16. : 0.0;
356 for(int i = 0; i < 3; i++) {
357 double * pAij = A + 5 * i + 1;
358 for(int j = i + 1 ; j < 4; j++) {
359 double Aij = *pAij;
360 double eps_machine = 100.0 * fabs(Aij);
361
362 if ( iter > 3 && fabs(D[i]) + eps_machine == fabs(D[i]) && fabs(D[j]) + eps_machine == fabs(D[j]) )
363 *pAij = 0.0;
364 else if (fabs(Aij) > tresh) {
365 double hh = D[j] - D[i], t;
366 if (fabs(hh) + eps_machine == fabs(hh))
367 t = Aij / hh;
368 else {
369 double theta = 0.5 * hh / Aij;
370 t = 1.0 / (fabs(theta) + sqrt(1.0 + theta * theta));
371 if (theta < 0.0) t = -t;
372 }
373
374 hh = t * Aij;
375 Z[i] -= hh;
376 Z[j] += hh;
377 D[i] -= hh;
378 D[j] += hh;
379 *pAij = 0.0;
380
381 double c = 1.0 / sqrt(1 + t * t);
382 double s = t * c;
383 double tau = s / (1.0 + c);
384 for(int k = 0; k <= i - 1; k++) {
385 double g = A[k * 4 + i], h = A[k * 4 + j];
386 A[k * 4 + i] = g - s * (h + g * tau);
387 A[k * 4 + j] = h + s * (g - h * tau);
388 }
389 for(int k = i + 1; k <= j - 1; k++) {
390 double g = A[i * 4 + k], h = A[k * 4 + j];
391 A[i * 4 + k] = g - s * (h + g * tau);
392 A[k * 4 + j] = h + s * (g - h * tau);
393 }
394 for(int k = j + 1; k < 4; k++) {
395 double g = A[i * 4 + k], h = A[j * 4 + k];
396 A[i * 4 + k] = g - s * (h + g * tau);
397 A[j * 4 + k] = h + s * (g - h * tau);
398 }
399 for(int k = 0; k < 4; k++) {
400 double g = U[k * 4 + i], h = U[k * 4 + j];
401 U[k * 4 + i] = g - s * (h + g * tau);
402 U[k * 4 + j] = h + s * (g - h * tau);
403 }
404 }
405 pAij++;
406 }
407 }
408
409 for(int i = 0; i < 4; i++) B[i] += Z[i];
410 memcpy(D, B, 4 * sizeof(double));
411 memset(Z, 0, 4 * sizeof(double));
412 }
413
414 return false;
415}
