57CV_IMPL CvStereoBMState* cvCreateStereoBMState( int /*preset*/, int numberOfDisparities )
58{
59 CvStereoBMState* state = (CvStereoBMState*)cvAlloc( sizeof(*state) );
60 if( !state )
61 return 0;
62
63 state->preFilterType = CV_STEREO_BM_XSOBEL; //CV_STEREO_BM_NORMALIZED_RESPONSE;
64 state->preFilterSize = 9;
65 state->preFilterCap = 31;
66 state->SADWindowSize = 15;
67 state->minDisparity = 0;
68 state->numberOfDisparities = numberOfDisparities > 0 ? numberOfDisparities : 64;
69 state->textureThreshold = 10;
70 state->uniquenessRatio = 15;
71 state->speckleRange = state->speckleWindowSize = 0;
72 state->trySmallerWindows = 0;
73 state->roi1 = state->roi2 = cvRect(0,0,0,0);
74 state->disp12MaxDiff = -1;
75
76 state->preFilteredImg0 = state->preFilteredImg1 = state->slidingSumBuf =
77 state->disp = state->cost = 0;
78
79 return state;
80}
81
82CV_IMPL void cvReleaseStereoBMState( CvStereoBMState** state )
83{
84 if( !state )
85 CV_Error( CV_StsNullPtr, "" );
86
87 if( !*state )
88 return;
89
90 cvReleaseMat( &(*state)->preFilteredImg0 );
91 cvReleaseMat( &(*state)->preFilteredImg1 );
92 cvReleaseMat( &(*state)->slidingSumBuf );
93 cvReleaseMat( &(*state)->disp );
94 cvReleaseMat( &(*state)->cost );
95 cvFree( state );
96}

101static void prefilterNorm( const Mat& src, Mat& dst, int winsize, int ftzero, uchar* buf )
102{
103 int x, y, wsz2 = winsize/2;
104 int* vsum = (int*)alignPtr(buf + (wsz2 + 1)*sizeof(vsum[0]), 32);
105 int scale_g = winsize*winsize/8, scale_s = (1024 + scale_g)/(scale_g*2);
106 const int OFS = 256*5, TABSZ = OFS*2 + 256;
107 uchar tab[TABSZ];
108 const uchar* sptr = src.data;
109 int srcstep = (int)src.step;
110 Size size = src.size();
111
112 scale_g *= scale_s;
113
114 for( x = 0; x < TABSZ; x++ )
115 tab[x] = (uchar)(x - OFS < -ftzero ? 0 : x - OFS > ftzero ? ftzero*2 : x - OFS + ftzero);
116
117 for( x = 0; x < size.width; x++ )
118 vsum[x] = (ushort)(sptr[x]*(wsz2 + 2));
119
120 for( y = 1; y < wsz2; y++ )
121 {
122 for( x = 0; x < size.width; x++ )
123 vsum[x] = (ushort)(vsum[x] + sptr[srcstep*y + x]);
124 }
125
126 for( y = 0; y < size.height; y++ )
127 {
128 const uchar* top = sptr + srcstep*MAX(y-wsz2-1,0);
129 const uchar* bottom = sptr + srcstep*MIN(y+wsz2,size.height-1);
130 const uchar* prev = sptr + srcstep*MAX(y-1,0);
131 const uchar* curr = sptr + srcstep*y;
132 const uchar* next = sptr + srcstep*MIN(y+1,size.height-1);
133 uchar* dptr = dst.ptr<uchar>(y);
134 x = 0;
135
136 for( ; x < size.width; x++ )
137 vsum[x] = (ushort)(vsum[x] + bottom[x] - top[x]);
138
139 for( x = 0; x <= wsz2; x++ )
140 {
141 vsum[-x-1] = vsum[0];
142 vsum[size.width+x] = vsum[size.width-1];
143 }
144
145 int sum = vsum[0]*(wsz2 + 1);
146 for( x = 1; x <= wsz2; x++ )
147 sum += vsum[x];
148
149 int val = ((curr[0]*5 + curr[1] + prev[0] + next[0])*scale_g - sum*scale_s) >> 10;
150 dptr[0] = tab[val + OFS];
151
152 for( x = 1; x < size.width-1; x++ )
153 {
154 sum += vsum[x+wsz2] - vsum[x-wsz2-1];
155 val = ((curr[x]*4 + curr[x-1] + curr[x+1] + prev[x] + next[x])*scale_g - sum*scale_s) >> 10;
156 dptr[x] = tab[val + OFS];
157 }
158
159 sum += vsum[x+wsz2] - vsum[x-wsz2-1];
160 val = ((curr[x]*5 + curr[x-1] + prev[x] + next[x])*scale_g - sum*scale_s) >> 10;
161 dptr[x] = tab[val + OFS];
162 }
163}

166static void
167prefilterXSobel( const Mat& src, Mat& dst, int ftzero )
168{
169 int x, y;
170 const int OFS = 256*4, TABSZ = OFS*2 + 256;
171 uchar tab[TABSZ];
172 Size size = src.size();
173
174 for( x = 0; x < TABSZ; x++ )
175 tab[x] = (uchar)(x - OFS < -ftzero ? 0 : x - OFS > ftzero ? ftzero*2 : x - OFS + ftzero);
176 uchar val0 = tab[0 + OFS];
177
178#if CV_SSE2
179 volatile bool useSIMD = checkHardwareSupport(CV_CPU_SSE2);
180#endif
181
182 for( y = 0; y < size.height-1; y += 2 )
183 {
184 const uchar* srow1 = src.ptr<uchar>(y);
185 const uchar* srow0 = y > 0 ? srow1 - src.step : size.height > 1 ? srow1 + src.step : srow1;
186 const uchar* srow2 = y < size.height-1 ? srow1 + src.step : size.height > 1 ? srow1 - src.step : srow1;
187 const uchar* srow3 = y < size.height-2 ? srow1 + src.step*2 : srow1;
188 uchar* dptr0 = dst.ptr<uchar>(y);
189 uchar* dptr1 = dptr0 + dst.step;
190
191 dptr0[0] = dptr0[size.width-1] = dptr1[0] = dptr1[size.width-1] = val0;
192 x = 1;
193
194 #if CV_SSE2
195 if( useSIMD )
196 {
197 __m128i z = _mm_setzero_si128(), ftz = _mm_set1_epi16((short)ftzero),
198 ftz2 = _mm_set1_epi8(CV_CAST_8U(ftzero*2));
199 for( ; x <= size.width-9; x += 8 )
200 {
201 __m128i c0 = _mm_unpacklo_epi8(_mm_loadl_epi64((__m128i*)(srow0 + x - 1)), z);
202 __m128i c1 = _mm_unpacklo_epi8(_mm_loadl_epi64((__m128i*)(srow1 + x - 1)), z);
203 __m128i d0 = _mm_unpacklo_epi8(_mm_loadl_epi64((__m128i*)(srow0 + x + 1)), z);
204 __m128i d1 = _mm_unpacklo_epi8(_mm_loadl_epi64((__m128i*)(srow1 + x + 1)), z);
205
206 d0 = _mm_sub_epi16(d0, c0);
207 d1 = _mm_sub_epi16(d1, c1);
208
209 __m128i c2 = _mm_unpacklo_epi8(_mm_loadl_epi64((__m128i*)(srow2 + x - 1)), z);
210 __m128i c3 = _mm_unpacklo_epi8(_mm_loadl_epi64((__m128i*)(srow3 + x - 1)), z);
211 __m128i d2 = _mm_unpacklo_epi8(_mm_loadl_epi64((__m128i*)(srow2 + x + 1)), z);
212 __m128i d3 = _mm_unpacklo_epi8(_mm_loadl_epi64((__m128i*)(srow3 + x + 1)), z);
213
214 d2 = _mm_sub_epi16(d2, c2);
215 d3 = _mm_sub_epi16(d3, c3);
216
217 __m128i v0 = _mm_add_epi16(d0, _mm_add_epi16(d2, _mm_add_epi16(d1, d1)));
218 __m128i v1 = _mm_add_epi16(d1, _mm_add_epi16(d3, _mm_add_epi16(d2, d2)));
219 v0 = _mm_packus_epi16(_mm_add_epi16(v0, ftz), _mm_add_epi16(v1, ftz));
220 v0 = _mm_min_epu8(v0, ftz2);
221
222 _mm_storel_epi64((__m128i*)(dptr0 + x), v0);
223 _mm_storel_epi64((__m128i*)(dptr1 + x), _mm_unpackhi_epi64(v0, v0));
224 }
225 }
226 #endif
227 #if CV_NEON
228 int16x8_t ftz = vdupq_n_s16 ((short) ftzero);
229 uint8x8_t ftz2 = vdup_n_u8 (cv::saturate_cast<uchar>(ftzero*2));
230
231 for(; x <=size.width-9; x += 8 )
232 {
233 uint8x8_t c0 = vld1_u8 (srow0 + x - 1);
234 uint8x8_t c1 = vld1_u8 (srow1 + x - 1);
235 uint8x8_t d0 = vld1_u8 (srow0 + x + 1);
236 uint8x8_t d1 = vld1_u8 (srow1 + x + 1);
237
238 int16x8_t t0 = vreinterpretq_s16_u16 (vsubl_u8 (d0, c0));
239 int16x8_t t1 = vreinterpretq_s16_u16 (vsubl_u8 (d1, c1));
240
241 uint8x8_t c2 = vld1_u8 (srow2 + x - 1);
242 uint8x8_t c3 = vld1_u8 (srow3 + x - 1);
243 uint8x8_t d2 = vld1_u8 (srow2 + x + 1);
244 uint8x8_t d3 = vld1_u8 (srow3 + x + 1);
245
246 int16x8_t t2 = vreinterpretq_s16_u16 (vsubl_u8 (d2, c2));
247 int16x8_t t3 = vreinterpretq_s16_u16 (vsubl_u8 (d3, c3));
248
249 int16x8_t v0 = vaddq_s16 (vaddq_s16 (t2, t0), vaddq_s16 (t1, t1));
250 int16x8_t v1 = vaddq_s16 (vaddq_s16 (t3, t1), vaddq_s16 (t2, t2));
251
252
253 uint8x8_t v0_u8 = vqmovun_s16 (vaddq_s16 (v0, ftz));
254 uint8x8_t v1_u8 = vqmovun_s16 (vaddq_s16 (v1, ftz));
255 v0_u8 = vmin_u8 (v0_u8, ftz2);
256 v1_u8 = vmin_u8 (v1_u8, ftz2);
257 vqmovun_s16 (vaddq_s16 (v1, ftz));
258
259 vst1_u8 (dptr0 + x, v0_u8);
260 vst1_u8 (dptr1 + x, v1_u8);
261 }
262 #endif
263
264 for( ; x < size.width-1; x++ )
265 {
266 int d0 = srow0[x+1] - srow0[x-1], d1 = srow1[x+1] - srow1[x-1],
267 d2 = srow2[x+1] - srow2[x-1], d3 = srow3[x+1] - srow3[x-1];
268 int v0 = tab[d0 + d1*2 + d2 + OFS];
269 int v1 = tab[d1 + d2*2 + d3 + OFS];
270 dptr0[x] = (uchar)v0;
271 dptr1[x] = (uchar)v1;
272 }
273 }
274
275#if CV_NEON
276 uint8x16_t val0_16 = vdupq_n_u8 (val0);
277#endif
278
279 for( ; y < size.height; y++ )
280 {
281 uchar* dptr = dst.ptr<uchar>(y);
282 x = 0;
283 #if CV_NEON
284 for(; x <= size.width-16; x+=16 )
285 vst1q_u8 (dptr + x, val0_16);
286 #endif
287 for(; x < size.width; x++ )
288 dptr[x] = val0;
289 }
290}

293static const int DISPARITY_SHIFT = 4;
294
295#if CV_SSE2
296static void findStereoCorrespondenceBM_SSE2( const Mat& left, const Mat& right,
297 Mat& disp, Mat& cost, CvStereoBMState& state,
298 uchar* buf, int _dy0, int _dy1 )
299{
300 const int ALIGN = 16;
301 int x, y, d;
302 int wsz = state.SADWindowSize, wsz2 = wsz/2;
303 int dy0 = MIN(_dy0, wsz2+1), dy1 = MIN(_dy1, wsz2+1);
304 int ndisp = state.numberOfDisparities;
305 int mindisp = state.minDisparity;
306 int lofs = MAX(ndisp - 1 + mindisp, 0);
307 int rofs = -MIN(ndisp - 1 + mindisp, 0);
308 int width = left.cols, height = left.rows;
309 int width1 = width - rofs - ndisp + 1;
310 int ftzero = state.preFilterCap;
311 int textureThreshold = state.textureThreshold;
312 int uniquenessRatio = state.uniquenessRatio*256/100;
313 short FILTERED = (short)((mindisp - 1) << DISPARITY_SHIFT);
314
315 ushort *sad, *hsad0, *hsad, *hsad_sub;
316 int *htext;
317 uchar *cbuf0, *cbuf;
318 const uchar* lptr0 = left.data + lofs;
319 const uchar* rptr0 = right.data + rofs;
320 const uchar *lptr, *lptr_sub, *rptr;
321 short* dptr = (short*)disp.data;
322 int sstep = (int)left.step;
323 int dstep = (int)(disp.step/sizeof(dptr[0]));
324 int cstep = (height + dy0 + dy1)*ndisp;
325 short costbuf = 0;
326 int coststep = cost.data ? (int)(cost.step/sizeof(costbuf)) : 0;
327 const int TABSZ = 256;
328 uchar tab[TABSZ];
329 const __m128i d0_8 = _mm_setr_epi16(0,1,2,3,4,5,6,7), dd_8 = _mm_set1_epi16(8);
330
331 sad = (ushort*)alignPtr(buf + sizeof(sad[0]), ALIGN);
332 hsad0 = (ushort*)alignPtr(sad + ndisp + 1 + dy0*ndisp, ALIGN);
333 htext = (int*)alignPtr((int*)(hsad0 + (height+dy1)*ndisp) + wsz2 + 2, ALIGN);
334 cbuf0 = (uchar*)alignPtr(htext + height + wsz2 + 2 + dy0*ndisp, ALIGN);
335
336 for( x = 0; x < TABSZ; x++ )
337 tab[x] = (uchar)std::abs(x - ftzero);
338
339 // initialize buffers
340 memset( hsad0 - dy0*ndisp, 0, (height + dy0 + dy1)*ndisp*sizeof(hsad0[0]) );
341 memset( htext - wsz2 - 1, 0, (height + wsz + 1)*sizeof(htext[0]) );
342
343 for( x = -wsz2-1; x < wsz2; x++ )
344 {
345 hsad = hsad0 - dy0*ndisp; cbuf = cbuf0 + (x + wsz2 + 1)*cstep - dy0*ndisp;
346 lptr = lptr0 + MIN(MAX(x, -lofs), width-lofs-1) - dy0*sstep;
347 rptr = rptr0 + MIN(MAX(x, -rofs), width-rofs-ndisp) - dy0*sstep;
348
349 for( y = -dy0; y < height + dy1; y++, hsad += ndisp, cbuf += ndisp, lptr += sstep, rptr += sstep )
350 {
351 int lval = lptr[0];
352 __m128i lv = _mm_set1_epi8((char)lval), z = _mm_setzero_si128();
353 for( d = 0; d < ndisp; d += 16 )
354 {
355 __m128i rv = _mm_loadu_si128((const __m128i*)(rptr + d));
356 __m128i hsad_l = _mm_load_si128((__m128i*)(hsad + d));
357 __m128i hsad_h = _mm_load_si128((__m128i*)(hsad + d + 8));
358 __m128i diff = _mm_adds_epu8(_mm_subs_epu8(lv, rv), _mm_subs_epu8(rv, lv));
359 _mm_store_si128((__m128i*)(cbuf + d), diff);
360 hsad_l = _mm_add_epi16(hsad_l, _mm_unpacklo_epi8(diff,z));
361 hsad_h = _mm_add_epi16(hsad_h, _mm_unpackhi_epi8(diff,z));
362 _mm_store_si128((__m128i*)(hsad + d), hsad_l);
363 _mm_store_si128((__m128i*)(hsad + d + 8), hsad_h);
364 }
365 htext[y] += tab[lval];
366 }
367 }
368
369 // initialize the left and right borders of the disparity map
370 for( y = 0; y < height; y++ )
371 {
372 for( x = 0; x < lofs; x++ )
373 dptr[y*dstep + x] = FILTERED;
374 for( x = lofs + width1; x < width; x++ )
375 dptr[y*dstep + x] = FILTERED;
376 }
377 dptr += lofs;
378
379 for( x = 0; x < width1; x++, dptr++ )
380 {
381 short* costptr = cost.data ? (short*)cost.data + lofs + x : &costbuf;
382 int x0 = x - wsz2 - 1, x1 = x + wsz2;
383 const uchar* cbuf_sub = cbuf0 + ((x0 + wsz2 + 1) % (wsz + 1))*cstep - dy0*ndisp;
384 cbuf = cbuf0 + ((x1 + wsz2 + 1) % (wsz + 1))*cstep - dy0*ndisp;
385 hsad = hsad0 - dy0*ndisp;
386 lptr_sub = lptr0 + MIN(MAX(x0, -lofs), width-1-lofs) - dy0*sstep;
387 lptr = lptr0 + MIN(MAX(x1, -lofs), width-1-lofs) - dy0*sstep;
388 rptr = rptr0 + MIN(MAX(x1, -rofs), width-ndisp-rofs) - dy0*sstep;
389
390 for( y = -dy0; y < height + dy1; y++, cbuf += ndisp, cbuf_sub += ndisp,
391 hsad += ndisp, lptr += sstep, lptr_sub += sstep, rptr += sstep )
392 {
393 int lval = lptr[0];
394 __m128i lv = _mm_set1_epi8((char)lval), z = _mm_setzero_si128();
395 for( d = 0; d < ndisp; d += 16 )
396 {
397 __m128i rv = _mm_loadu_si128((const __m128i*)(rptr + d));
398 __m128i hsad_l = _mm_load_si128((__m128i*)(hsad + d));
399 __m128i hsad_h = _mm_load_si128((__m128i*)(hsad + d + 8));
400 __m128i cbs = _mm_load_si128((const __m128i*)(cbuf_sub + d));
401 __m128i diff = _mm_adds_epu8(_mm_subs_epu8(lv, rv), _mm_subs_epu8(rv, lv));
402 __m128i diff_h = _mm_sub_epi16(_mm_unpackhi_epi8(diff, z), _mm_unpackhi_epi8(cbs, z));
403 _mm_store_si128((__m128i*)(cbuf + d), diff);
404 diff = _mm_sub_epi16(_mm_unpacklo_epi8(diff, z), _mm_unpacklo_epi8(cbs, z));
405 hsad_h = _mm_add_epi16(hsad_h, diff_h);
406 hsad_l = _mm_add_epi16(hsad_l, diff);
407 _mm_store_si128((__m128i*)(hsad + d), hsad_l);
408 _mm_store_si128((__m128i*)(hsad + d + 8), hsad_h);
409 }
410 htext[y] += tab[lval] - tab[lptr_sub[0]];
411 }
412
413 // fill borders
414 for( y = dy1; y <= wsz2; y++ )
415 htext[height+y] = htext[height+dy1-1];
416 for( y = -wsz2-1; y < -dy0; y++ )
417 htext[y] = htext[-dy0];
418
419 // initialize sums
420 for( d = 0; d < ndisp; d++ )
421 sad[d] = (ushort)(hsad0[d-ndisp*dy0]*(wsz2 + 2 - dy0));
422
423 hsad = hsad0 + (1 - dy0)*ndisp;
424 for( y = 1 - dy0; y < wsz2; y++, hsad += ndisp )
425 for( d = 0; d < ndisp; d += 16 )
426 {
427 __m128i s0 = _mm_load_si128((__m128i*)(sad + d));
428 __m128i s1 = _mm_load_si128((__m128i*)(sad + d + 8));
429 __m128i t0 = _mm_load_si128((__m128i*)(hsad + d));
430 __m128i t1 = _mm_load_si128((__m128i*)(hsad + d + 8));
431 s0 = _mm_add_epi16(s0, t0);
432 s1 = _mm_add_epi16(s1, t1);
433 _mm_store_si128((__m128i*)(sad + d), s0);
434 _mm_store_si128((__m128i*)(sad + d + 8), s1);
435 }
436 int tsum = 0;
437 for( y = -wsz2-1; y < wsz2; y++ )
438 tsum += htext[y];
439
440 // finally, start the real processing
441 for( y = 0; y < height; y++ )
442 {
443 int minsad = INT_MAX, mind = -1;
444 hsad = hsad0 + MIN(y + wsz2, height+dy1-1)*ndisp;
445 hsad_sub = hsad0 + MAX(y - wsz2 - 1, -dy0)*ndisp;
446 __m128i minsad8 = _mm_set1_epi16(SHRT_MAX);
447 __m128i mind8 = _mm_set1_epi16(0), d8 = d0_8, mask;
448
449 for( d = 0; d < ndisp; d += 16 )
450 {
451 __m128i u0 = _mm_load_si128((__m128i*)(hsad_sub + d));
452 __m128i u1 = _mm_load_si128((__m128i*)(hsad + d));
453
454 __m128i v0 = _mm_load_si128((__m128i*)(hsad_sub + d + 8));
455 __m128i v1 = _mm_load_si128((__m128i*)(hsad + d + 8));
456
457 __m128i usad8 = _mm_load_si128((__m128i*)(sad + d));
458 __m128i vsad8 = _mm_load_si128((__m128i*)(sad + d + 8));
459
460 u1 = _mm_sub_epi16(u1, u0);
461 v1 = _mm_sub_epi16(v1, v0);
462 usad8 = _mm_add_epi16(usad8, u1);
463 vsad8 = _mm_add_epi16(vsad8, v1);
464
465 mask = _mm_cmpgt_epi16(minsad8, usad8);
466 minsad8 = _mm_min_epi16(minsad8, usad8);
467 mind8 = _mm_max_epi16(mind8, _mm_and_si128(mask, d8));
468
469 _mm_store_si128((__m128i*)(sad + d), usad8);
470 _mm_store_si128((__m128i*)(sad + d + 8), vsad8);
471
472 mask = _mm_cmpgt_epi16(minsad8, vsad8);
473 minsad8 = _mm_min_epi16(minsad8, vsad8);
474
475 d8 = _mm_add_epi16(d8, dd_8);
476 mind8 = _mm_max_epi16(mind8, _mm_and_si128(mask, d8));
477 d8 = _mm_add_epi16(d8, dd_8);
478 }
479
480 tsum += htext[y + wsz2] - htext[y - wsz2 - 1];
481 if( tsum < textureThreshold )
482 {
483 dptr[y*dstep] = FILTERED;
484 continue;
485 }
486
487 __m128i minsad82 = _mm_unpackhi_epi64(minsad8, minsad8);
488 __m128i mind82 = _mm_unpackhi_epi64(mind8, mind8);
489 mask = _mm_cmpgt_epi16(minsad8, minsad82);
490 mind8 = _mm_xor_si128(mind8,_mm_and_si128(_mm_xor_si128(mind82,mind8),mask));
491 minsad8 = _mm_min_epi16(minsad8, minsad82);
492
493 minsad82 = _mm_shufflelo_epi16(minsad8, _MM_SHUFFLE(3,2,3,2));
494 mind82 = _mm_shufflelo_epi16(mind8, _MM_SHUFFLE(3,2,3,2));
495 mask = _mm_cmpgt_epi16(minsad8, minsad82);
496 mind8 = _mm_xor_si128(mind8,_mm_and_si128(_mm_xor_si128(mind82,mind8),mask));
497 minsad8 = _mm_min_epi16(minsad8, minsad82);
498
499 minsad82 = _mm_shufflelo_epi16(minsad8, 1);
500 mind82 = _mm_shufflelo_epi16(mind8, 1);
501 mask = _mm_cmpgt_epi16(minsad8, minsad82);
502 mind8 = _mm_xor_si128(mind8,_mm_and_si128(_mm_xor_si128(mind82,mind8),mask));
503 mind = (short)_mm_cvtsi128_si32(mind8);
504 minsad = sad[mind];
505
506 if( uniquenessRatio > 0 )
507 {
508 int thresh = minsad + ((minsad * uniquenessRatio) >> 8);
509 __m128i thresh8 = _mm_set1_epi16((short)(thresh + 1));
510 __m128i d1 = _mm_set1_epi16((short)(mind-1)), d2 = _mm_set1_epi16((short)(mind+1));
511 __m128i dd_16 = _mm_add_epi16(dd_8, dd_8);
512 d8 = _mm_sub_epi16(d0_8, dd_16);
513
514 for( d = 0; d < ndisp; d += 16 )
515 {
516 __m128i usad8 = _mm_load_si128((__m128i*)(sad + d));
517 __m128i vsad8 = _mm_load_si128((__m128i*)(sad + d + 8));
518 mask = _mm_cmpgt_epi16( thresh8, _mm_min_epi16(usad8,vsad8));
519 d8 = _mm_add_epi16(d8, dd_16);
520 if( !_mm_movemask_epi8(mask) )
521 continue;
522 mask = _mm_cmpgt_epi16( thresh8, usad8);
523 mask = _mm_and_si128(mask, _mm_or_si128(_mm_cmpgt_epi16(d1,d8), _mm_cmpgt_epi16(d8,d2)));
524 if( _mm_movemask_epi8(mask) )
525 break;
526 __m128i t8 = _mm_add_epi16(d8, dd_8);
527 mask = _mm_cmpgt_epi16( thresh8, vsad8);
528 mask = _mm_and_si128(mask, _mm_or_si128(_mm_cmpgt_epi16(d1,t8), _mm_cmpgt_epi16(t8,d2)));
529 if( _mm_movemask_epi8(mask) )
530 break;
531 }
532 if( d < ndisp )
533 {
534 dptr[y*dstep] = FILTERED;
535 continue;
536 }
537 }
538
539 if( 0 < mind && mind < ndisp - 1 )
540 {
541 int p = sad[mind+1], n = sad[mind-1];
542 d = p + n - 2*sad[mind] + std::abs(p - n);
543 dptr[y*dstep] = (short)(((ndisp - mind - 1 + mindisp)*256 + (d != 0 ? (p-n)*256/d : 0) + 15) >> 4);
544 }
545 else
546 dptr[y*dstep] = (short)((ndisp - mind - 1 + mindisp)*16);
547 costptr[y*coststep] = sad[mind];
548 }
549 }
550}
551#endif

553static void
554findStereoCorrespondenceBM( const Mat& left, const Mat& right,
555 Mat& disp, Mat& cost, const CvStereoBMState& state,
556 uchar* buf, int _dy0, int _dy1 )
557{
558
559 const int ALIGN = 16;
560 int x, y, d;
561 int wsz = state.SADWindowSize, wsz2 = wsz/2;
562 int dy0 = MIN(_dy0, wsz2+1), dy1 = MIN(_dy1, wsz2+1);
563 int ndisp = state.numberOfDisparities;
564 int mindisp = state.minDisparity;
565 int lofs = MAX(ndisp - 1 + mindisp, 0);
566 int rofs = -MIN(ndisp - 1 + mindisp, 0);
567 int width = left.cols, height = left.rows;
568 int width1 = width - rofs - ndisp + 1;
569 int ftzero = state.preFilterCap;
570 int textureThreshold = state.textureThreshold;
571 int uniquenessRatio = state.uniquenessRatio;
572 short FILTERED = (short)((mindisp - 1) << DISPARITY_SHIFT);
573
574#if CV_NEON
575 CV_Assert (ndisp % 8 == 0);
576 int32_t d0_4_temp [4];
577 for (int i = 0; i < 4; i ++)
578 d0_4_temp[i] = i;
579 int32x4_t d0_4 = vld1q_s32 (d0_4_temp);
580 int32x4_t dd_4 = vdupq_n_s32 (4);
581#endif
582
583 int *sad, *hsad0, *hsad, *hsad_sub, *htext;
584 uchar *cbuf0, *cbuf;
585 const uchar* lptr0 = left.data + lofs;
586 const uchar* rptr0 = right.data + rofs;
587 const uchar *lptr, *lptr_sub, *rptr;
588 short* dptr = (short*)disp.data;
589 int sstep = (int)left.step;
590 int dstep = (int)(disp.step/sizeof(dptr[0]));
591 int cstep = (height+dy0+dy1)*ndisp;
592 int costbuf = 0;
593 int coststep = cost.data ? (int)(cost.step/sizeof(costbuf)) : 0;
594 const int TABSZ = 256;
595 uchar tab[TABSZ];
596
597 sad = (int*)alignPtr(buf + sizeof(sad[0]), ALIGN);
598 hsad0 = (int*)alignPtr(sad + ndisp + 1 + dy0*ndisp, ALIGN);
599 htext = (int*)alignPtr((int*)(hsad0 + (height+dy1)*ndisp) + wsz2 + 2, ALIGN);
600 cbuf0 = (uchar*)alignPtr((uchar*)(htext + height + wsz2 + 2) + dy0*ndisp, ALIGN);
601
602 for( x = 0; x < TABSZ; x++ )
603 tab[x] = (uchar)std::abs(x - ftzero);
604
605 // initialize buffers
606 memset( hsad0 - dy0*ndisp, 0, (height + dy0 + dy1)*ndisp*sizeof(hsad0[0]) );
607 memset( htext - wsz2 - 1, 0, (height + wsz + 1)*sizeof(htext[0]) );
608
609 for( x = -wsz2-1; x < wsz2; x++ )
610 {
611 hsad = hsad0 - dy0*ndisp; cbuf = cbuf0 + (x + wsz2 + 1)*cstep - dy0*ndisp;
612 lptr = lptr0 + std::min(std::max(x, -lofs), width-lofs-1) - dy0*sstep;
613 rptr = rptr0 + std::min(std::max(x, -rofs), width-rofs-ndisp) - dy0*sstep;
614
615 for( y = -dy0; y < height + dy1; y++, hsad += ndisp, cbuf += ndisp, lptr += sstep, rptr += sstep )
616 {
617 int lval = lptr[0];
618 #if CV_NEON
619 int16x8_t lv = vdupq_n_s16 ((int16_t)lval);
620
621 for( d = 0; d < ndisp; d += 8 )
622 {
623 int16x8_t rv = vreinterpretq_s16_u16 (vmovl_u8 (vld1_u8 (rptr + d)));
624 int32x4_t hsad_l = vld1q_s32 (hsad + d);
625 int32x4_t hsad_h = vld1q_s32 (hsad + d + 4);
626 int16x8_t diff = vabdq_s16 (lv, rv);
627 vst1_u8 (cbuf + d, vmovn_u16(vreinterpretq_u16_s16(diff)));
628 hsad_l = vaddq_s32 (hsad_l, vmovl_s16(vget_low_s16 (diff)));
629 hsad_h = vaddq_s32 (hsad_h, vmovl_s16(vget_high_s16 (diff)));
630 vst1q_s32 ((hsad + d), hsad_l);
631 vst1q_s32 ((hsad + d + 4), hsad_h);
632 }
633 #else
634 for( d = 0; d < ndisp; d++ )
635 {
636 int diff = std::abs(lval - rptr[d]);
637 cbuf[d] = (uchar)diff;
638 hsad[d] = (int)(hsad[d] + diff);
639 }
640 #endif
641 htext[y] += tab[lval];
642 }
643 }
644
645 // initialize the left and right borders of the disparity map
646 for( y = 0; y < height; y++ )
647 {
648 for( x = 0; x < lofs; x++ )
649 dptr[y*dstep + x] = FILTERED;
650 for( x = lofs + width1; x < width; x++ )
651 dptr[y*dstep + x] = FILTERED;
652 }
653 dptr += lofs;
654
655 for( x = 0; x < width1; x++, dptr++ )
656 {
657 int* costptr = cost.data ? (int*)cost.data + lofs + x : &costbuf;
658 int x0 = x - wsz2 - 1, x1 = x + wsz2;
659 const uchar* cbuf_sub = cbuf0 + ((x0 + wsz2 + 1) % (wsz + 1))*cstep - dy0*ndisp;
660 cbuf = cbuf0 + ((x1 + wsz2 + 1) % (wsz + 1))*cstep - dy0*ndisp;
661 hsad = hsad0 - dy0*ndisp;
662 lptr_sub = lptr0 + MIN(MAX(x0, -lofs), width-1-lofs) - dy0*sstep;
663 lptr = lptr0 + MIN(MAX(x1, -lofs), width-1-lofs) - dy0*sstep;
664 rptr = rptr0 + MIN(MAX(x1, -rofs), width-ndisp-rofs) - dy0*sstep;
665
666 for( y = -dy0; y < height + dy1; y++, cbuf += ndisp, cbuf_sub += ndisp,
667 hsad += ndisp, lptr += sstep, lptr_sub += sstep, rptr += sstep )
668 {
669 int lval = lptr[0];
670 #if CV_NEON
671 int16x8_t lv = vdupq_n_s16 ((int16_t)lval);
672 for( d = 0; d < ndisp; d += 8 )
673 {
674 int16x8_t rv = vreinterpretq_s16_u16 (vmovl_u8 (vld1_u8 (rptr + d)));
675 int32x4_t hsad_l = vld1q_s32 (hsad + d);
676 int32x4_t hsad_h = vld1q_s32 (hsad + d + 4);
677 int16x8_t cbs = vreinterpretq_s16_u16 (vmovl_u8 (vld1_u8 (cbuf_sub + d)));
678 int16x8_t diff = vabdq_s16 (lv, rv);
679 int32x4_t diff_h = vsubl_s16 (vget_high_s16 (diff), vget_high_s16 (cbs));
680 int32x4_t diff_l = vsubl_s16 (vget_low_s16 (diff), vget_low_s16 (cbs));
681 vst1_u8 (cbuf + d, vmovn_u16(vreinterpretq_u16_s16(diff)));
682 hsad_h = vaddq_s32 (hsad_h, diff_h);
683 hsad_l = vaddq_s32 (hsad_l, diff_l);
684 vst1q_s32 ((hsad + d), hsad_l);
685 vst1q_s32 ((hsad + d + 4), hsad_h);
686 }
687 #else
688 for( d = 0; d < ndisp; d++ )
689 {
690 int diff = std::abs(lval - rptr[d]);
691 cbuf[d] = (uchar)diff;
692 hsad[d] = hsad[d] + diff - cbuf_sub[d];
693 }
694 #endif
695 htext[y] += tab[lval] - tab[lptr_sub[0]];
696 }
697
698 // fill borders
699 for( y = dy1; y <= wsz2; y++ )
700 htext[height+y] = htext[height+dy1-1];
701 for( y = -wsz2-1; y < -dy0; y++ )
702 htext[y] = htext[-dy0];
703
704 // initialize sums
705 for( d = 0; d < ndisp; d++ )
706 sad[d] = (int)(hsad0[d-ndisp*dy0]*(wsz2 + 2 - dy0));
707
708 hsad = hsad0 + (1 - dy0)*ndisp;
709 for( y = 1 - dy0; y < wsz2; y++, hsad += ndisp )
710 {
711 #if CV_NEON
712 for( d = 0; d <= ndisp-8; d += 8 )
713 {
714 int32x4_t s0 = vld1q_s32 (sad + d);
715 int32x4_t s1 = vld1q_s32 (sad + d + 4);
716 int32x4_t t0 = vld1q_s32 (hsad + d);
717 int32x4_t t1 = vld1q_s32 (hsad + d + 4);
718 s0 = vaddq_s32 (s0, t0);
719 s1 = vaddq_s32 (s1, t1);
720 vst1q_s32 (sad + d, s0);
721 vst1q_s32 (sad + d + 4, s1);
722 }
723 #else
724 for( d = 0; d < ndisp; d++ )
725 sad[d] = (int)(sad[d] + hsad[d]);
726 #endif
727 }
728 int tsum = 0;
729 for( y = -wsz2-1; y < wsz2; y++ )
730 tsum += htext[y];
731
732 // finally, start the real processing
733 for( y = 0; y < height; y++ )
734 {
735 int minsad = INT_MAX, mind = -1;
736 hsad = hsad0 + MIN(y + wsz2, height+dy1-1)*ndisp;
737 hsad_sub = hsad0 + MAX(y - wsz2 - 1, -dy0)*ndisp;
738 #if CV_NEON
739 int32x4_t minsad4 = vdupq_n_s32 (INT_MAX);
740 int32x4_t mind4 = vdupq_n_s32(0), d4 = d0_4;
741
742 for( d = 0; d <= ndisp-8; d += 8 )
743 {
744 int32x4_t u0 = vld1q_s32 (hsad_sub + d);
745 int32x4_t u1 = vld1q_s32 (hsad + d);
746
747 int32x4_t v0 = vld1q_s32 (hsad_sub + d + 4);
748 int32x4_t v1 = vld1q_s32 (hsad + d + 4);
749
750 int32x4_t usad4 = vld1q_s32(sad + d);
751 int32x4_t vsad4 = vld1q_s32(sad + d + 4);
752
753 u1 = vsubq_s32 (u1, u0);
754 v1 = vsubq_s32 (v1, v0);
755 usad4 = vaddq_s32 (usad4, u1);
756 vsad4 = vaddq_s32 (vsad4, v1);
757
758 uint32x4_t mask = vcgtq_s32 (minsad4, usad4);
759 minsad4 = vminq_s32 (minsad4, usad4);
760 mind4 = vbslq_s32(mask, d4, mind4);
761
762 vst1q_s32 (sad + d, usad4);
763 vst1q_s32 (sad + d + 4, vsad4);
764 d4 = vaddq_s32 (d4, dd_4);
765
766 mask = vcgtq_s32 (minsad4, vsad4);
767 minsad4 = vminq_s32 (minsad4, vsad4);
768 mind4 = vbslq_s32(mask, d4, mind4);
769
770 d4 = vaddq_s32 (d4, dd_4);
771
772 }
773 int32x2_t mind4_h = vget_high_s32 (mind4);
774 int32x2_t mind4_l = vget_low_s32 (mind4);
775 int32x2_t minsad4_h = vget_high_s32 (minsad4);
776 int32x2_t minsad4_l = vget_low_s32 (minsad4);
777
778 uint32x2_t mask = vorr_u32 (vclt_s32 (minsad4_h, minsad4_l), vand_u32 (vceq_s32 (minsad4_h, minsad4_l), vclt_s32 (mind4_h, mind4_l)));
779 mind4_h = vbsl_s32 (mask, mind4_h, mind4_l);
780 minsad4_h = vbsl_s32 (mask, minsad4_h, minsad4_l);
781
782 mind4_l = vext_s32 (mind4_h,mind4_h,1);
783 minsad4_l = vext_s32 (minsad4_h,minsad4_h,1);
784
785 mask = vorr_u32 (vclt_s32 (minsad4_h, minsad4_l), vand_u32 (vceq_s32 (minsad4_h, minsad4_l), vclt_s32 (mind4_h, mind4_l)));
786 mind4_h = vbsl_s32 (mask, mind4_h, mind4_l);
787 minsad4_h = vbsl_s32 (mask, minsad4_h, minsad4_l);
788
789 mind = (int) vget_lane_s32 (mind4_h, 0);
790 minsad = sad[mind];
791
792 #else
793 for( d = 0; d < ndisp; d++ )
794 {
795 int currsad = sad[d] + hsad[d] - hsad_sub[d];
796 sad[d] = currsad;
797 if( currsad < minsad )
798 {
799 minsad = currsad;
800 mind = d;
801 }
802 }
803 #endif
804 tsum += htext[y + wsz2] - htext[y - wsz2 - 1];
805 if( tsum < textureThreshold )
806 {
807 dptr[y*dstep] = FILTERED;
808 continue;
809 }
810
811 if( uniquenessRatio > 0 )
812 {
813 int thresh = minsad + (minsad * uniquenessRatio/100);
814 for( d = 0; d < ndisp; d++ )
815 {
816 if( sad[d] <= thresh && (d < mind-1 || d > mind+1))
817 break;
818 }
819 if( d < ndisp )
820 {
821 dptr[y*dstep] = FILTERED;
822 continue;
823 }
824 }
825
826 {
827 sad[-1] = sad[1];
828 sad[ndisp] = sad[ndisp-2];
829 int p = sad[mind+1], n = sad[mind-1];
830 d = p + n - 2*sad[mind] + std::abs(p - n);
831 dptr[y*dstep] = (short)(((ndisp - mind - 1 + mindisp)*256 + (d != 0 ? (p-n)*256/d : 0) + 15) >> 4);
832 costptr[y*coststep] = sad[mind];
833 }
834 }
835 }

942static void findStereoCorrespondenceBM( const Mat& left0, const Mat& right0, Mat& disp0, CvStereoBMState* state)
943{
944 if (left0.size() != right0.size() || disp0.size() != left0.size())
945 CV_Error( CV_StsUnmatchedSizes, "All the images must have the same size" );
946
947 if (left0.type() != CV_8UC1 || right0.type() != CV_8UC1)
948 CV_Error( CV_StsUnsupportedFormat, "Both input images must have CV_8UC1" );
949
950 if (disp0.type() != CV_16SC1 && disp0.type() != CV_32FC1)
951 CV_Error( CV_StsUnsupportedFormat, "Disparity image must have CV_16SC1 or CV_32FC1 format" );
952
953 if( !state )
954 CV_Error( CV_StsNullPtr, "Stereo BM state is NULL." );
955
956 if( state->preFilterType != CV_STEREO_BM_NORMALIZED_RESPONSE && state->preFilterType != CV_STEREO_BM_XSOBEL )
957 CV_Error( CV_StsOutOfRange, "preFilterType must be = CV_STEREO_BM_NORMALIZED_RESPONSE" );
958
959 if( state->preFilterSize < 5 || state->preFilterSize > 255 || state->preFilterSize % 2 == 0 )
960 CV_Error( CV_StsOutOfRange, "preFilterSize must be odd and be within 5..255" );
961
962 if( state->preFilterCap < 1 || state->preFilterCap > 63 )
963 CV_Error( CV_StsOutOfRange, "preFilterCap must be within 1..63" );
964
965 if( state->SADWindowSize < 5 || state->SADWindowSize > 255 || state->SADWindowSize % 2 == 0 ||
966 state->SADWindowSize >= min(left0.cols, left0.rows) )
967 CV_Error( CV_StsOutOfRange, "SADWindowSize must be odd, be within 5..255 and be not larger than image width or height" );
968
969 if( state->numberOfDisparities <= 0 || state->numberOfDisparities % 16 != 0 )
970 CV_Error( CV_StsOutOfRange, "numberOfDisparities must be positive and divisble by 16" );
971
972 if( state->textureThreshold < 0 )
973 CV_Error( CV_StsOutOfRange, "texture threshold must be non-negative" );
974
975 if( state->uniquenessRatio < 0 )
976 CV_Error( CV_StsOutOfRange, "uniqueness ratio must be non-negative" );
977
978 if( !state->preFilteredImg0 || state->preFilteredImg0->cols * state->preFilteredImg0->rows < left0.cols * left0.rows )
979 {
980 cvReleaseMat( &state->preFilteredImg0 );
981 cvReleaseMat( &state->preFilteredImg1 );
982 cvReleaseMat( &state->cost );
983
984 state->preFilteredImg0 = cvCreateMat( left0.rows, left0.cols, CV_8U );
985 state->preFilteredImg1 = cvCreateMat( left0.rows, left0.cols, CV_8U );
986 state->cost = cvCreateMat( left0.rows, left0.cols, CV_16S );
987 }
988 Mat left(left0.size(), CV_8U, state->preFilteredImg0->data.ptr);
989 Mat right(right0.size(), CV_8U, state->preFilteredImg1->data.ptr);
990
991 int mindisp = state->minDisparity;
992 int ndisp = state->numberOfDisparities;
993
994 int width = left0.cols;
995 int height = left0.rows;
996 int lofs = max(ndisp - 1 + mindisp, 0);
997 int rofs = -min(ndisp - 1 + mindisp, 0);
998 int width1 = width - rofs - ndisp + 1;
999 int FILTERED = (state->minDisparity - 1) << DISPARITY_SHIFT;
1000
1001 if( lofs >= width || rofs >= width || width1 < 1 )
1002 {
1003 disp0 = Scalar::all( FILTERED * ( disp0.type() < CV_32F ? 1 : 1./(1 << DISPARITY_SHIFT) ) );
1004 return;
1005 }
1006
1007 Mat disp = disp0;
1008
1009 if( disp0.type() == CV_32F)
1010 {
1011 if( !state->disp || state->disp->rows != disp0.rows || state->disp->cols != disp0.cols )
1012 {
1013 cvReleaseMat( &state->disp );
1014 state->disp = cvCreateMat(disp0.rows, disp0.cols, CV_16S);
1015 }
1016 disp = cv::cvarrToMat(state->disp);
1017 }
1018
1019 int wsz = state->SADWindowSize;
1020 int bufSize0 = (int)((ndisp + 2)*sizeof(int));
1021 bufSize0 += (int)((height+wsz+2)*ndisp*sizeof(int));
1022 bufSize0 += (int)((height + wsz + 2)*sizeof(int));
1023 bufSize0 += (int)((height+wsz+2)*ndisp*(wsz+2)*sizeof(uchar) + 256);
1024
1025 int bufSize1 = (int)((width + state->preFilterSize + 2) * sizeof(int) + 256);
1026 int bufSize2 = 0;
1027 if( state->speckleRange >= 0 && state->speckleWindowSize > 0 )
1028 bufSize2 = width*height*(sizeof(cv::Point_<short>) + sizeof(int) + sizeof(uchar));
1029
1030#if CV_SSE2
1031 bool useShorts = state->preFilterCap <= 31 && state->SADWindowSize <= 21 && checkHardwareSupport(CV_CPU_SSE2);
1032#else
1033 const bool useShorts = false;
1034#endif
1035
1036 const double SAD_overhead_coeff = 10.0;
1037 double N0 = 8000000 / (useShorts ? 1 : 4); // approx tbb's min number instructions reasonable for one thread
1038 double maxStripeSize = min(max(N0 / (width * ndisp), (wsz-1) * SAD_overhead_coeff), (double)height);
1039 int nstripes = cvCeil(height / maxStripeSize);
1040
1041 int bufSize = max(bufSize0 * nstripes, max(bufSize1 * 2, bufSize2));
1042
1043 if( !state->slidingSumBuf || state->slidingSumBuf->cols < bufSize )
1044 {
1045 cvReleaseMat( &state->slidingSumBuf );
1046 state->slidingSumBuf = cvCreateMat( 1, bufSize, CV_8U );
1047 }
1048
1049 uchar *_buf = state->slidingSumBuf->data.ptr;
1050 int idx[] = {0,1};
1051 parallel_do(idx, idx+2, PrefilterInvoker(left0, right0, left, right, _buf, _buf + bufSize1, state));
1052
1053 Rect validDisparityRect(0, 0, width, height), R1 = state->roi1, R2 = state->roi2;
1054 validDisparityRect = getValidDisparityROI(R1.area() > 0 ? Rect(0, 0, width, height) : validDisparityRect,
1055 R2.area() > 0 ? Rect(0, 0, width, height) : validDisparityRect,
1056 state->minDisparity, state->numberOfDisparities,
1057 state->SADWindowSize);
1058
1059 parallel_for_(Range(0, nstripes),
1060 FindStereoCorrespInvoker(left, right, disp, state, nstripes,
1061 bufSize0, useShorts, validDisparityRect));
1062
1063 if( state->speckleRange >= 0 && state->speckleWindowSize > 0 )
1064 {
1065 Mat buf(state->slidingSumBuf);
1066 filterSpeckles(disp, FILTERED, state->speckleWindowSize, state->speckleRange, buf);
1067 }
1068
1069 if (disp0.data != disp.data)
1070 disp.convertTo(disp0, disp0.type(), 1./(1 << DISPARITY_SHIFT), 0);
1071}
