45 "Stereo Processing by Semiglobal Matching and Mutual Information"
46 by Heiko Hirschmuller.

48 We match blocks rather than individual pixels, thus the algorithm is called
49 SGBM (Semi-global block matching)
50 */

101/*
102 For each pixel row1[x], max(-maxD, 0) <= minX <= x < maxX <= width - max(0, minD),
103 and for each disparity minD<=d<maxD the function
104 computes the cost (cost[(x-minX)*(maxD - minD) + (d - minD)]), depending on the difference between
105 row1[x] and row2[x-d]. The subpixel algorithm from
106 "Depth Discontinuities by Pixel-to-Pixel Stereo" by Stan Birchfield and C. Tomasi
107 is used, hence the suffix BT.
108
109 the temporary buffer should contain width2*2 elements
110 */
111static void calcPixelCostBT( const Mat& img1, const Mat& img2, int y,
112 int minD, int maxD, CostType* cost,
113 PixType* buffer, const PixType* tab,
114 int tabOfs, int )
115{
116 int x, c, width = img1.cols, cn = img1.channels();
117 int minX1 = max(-maxD, 0), maxX1 = width + min(-minD, 0);
118 int minX2 = max(minX1 + minD, 0), maxX2 = min(maxX1 + maxD, width);
119 int D = maxD - minD, width1 = maxX1 - minX1, width2 = maxX2 - minX2;
120 const PixType *row1 = img1.ptr<PixType>(y), *row2 = img2.ptr<PixType>(y);
121 PixType *prow1 = buffer + width2*2, *prow2 = prow1 + width*cn*2;
122
123 tab += tabOfs;
124
125 for( c = 0; c < cn*2; c++ )
126 {
127 prow1[width*c] = prow1[width*c + width-1] =
128 prow2[width*c] = prow2[width*c + width-1] = tab[0];
129 }
130
131 int n1 = y > 0 ? -(int)img1.step : 0, s1 = y < img1.rows-1 ? (int)img1.step : 0;
132 int n2 = y > 0 ? -(int)img2.step : 0, s2 = y < img2.rows-1 ? (int)img2.step : 0;
133
134 if( cn == 1 )
135 {
136 for( x = 1; x < width-1; x++ )
137 {
138 prow1[x] = tab[(row1[x+1] - row1[x-1])*2 + row1[x+n1+1] - row1[x+n1-1] + row1[x+s1+1] - row1[x+s1-1]];
139 prow2[width-1-x] = tab[(row2[x+1] - row2[x-1])*2 + row2[x+n2+1] - row2[x+n2-1] + row2[x+s2+1] - row2[x+s2-1]];
140
141 prow1[x+width] = row1[x];
142 prow2[width-1-x+width] = row2[x];
143 }
144 }
145 else
146 {
147 for( x = 1; x < width-1; x++ )
148 {
149 prow1[x] = tab[(row1[x*3+3] - row1[x*3-3])*2 + row1[x*3+n1+3] - row1[x*3+n1-3] + row1[x*3+s1+3] - row1[x*3+s1-3]];
150 prow1[x+width] = tab[(row1[x*3+4] - row1[x*3-2])*2 + row1[x*3+n1+4] - row1[x*3+n1-2] + row1[x*3+s1+4] - row1[x*3+s1-2]];
151 prow1[x+width*2] = tab[(row1[x*3+5] - row1[x*3-1])*2 + row1[x*3+n1+5] - row1[x*3+n1-1] + row1[x*3+s1+5] - row1[x*3+s1-1]];
152
153 prow2[width-1-x] = tab[(row2[x*3+3] - row2[x*3-3])*2 + row2[x*3+n2+3] - row2[x*3+n2-3] + row2[x*3+s2+3] - row2[x*3+s2-3]];
154 prow2[width-1-x+width] = tab[(row2[x*3+4] - row2[x*3-2])*2 + row2[x*3+n2+4] - row2[x*3+n2-2] + row2[x*3+s2+4] - row2[x*3+s2-2]];
155 prow2[width-1-x+width*2] = tab[(row2[x*3+5] - row2[x*3-1])*2 + row2[x*3+n2+5] - row2[x*3+n2-1] + row2[x*3+s2+5] - row2[x*3+s2-1]];
156
157 prow1[x+width*3] = row1[x*3];
158 prow1[x+width*4] = row1[x*3+1];
159 prow1[x+width*5] = row1[x*3+2];
160
161 prow2[width-1-x+width*3] = row2[x*3];
162 prow2[width-1-x+width*4] = row2[x*3+1];
163 prow2[width-1-x+width*5] = row2[x*3+2];
164 }
165 }
166
167 memset( cost, 0, width1*D*sizeof(cost[0]) );
168
169 buffer -= minX2;
170 cost -= minX1*D + minD; // simplify the cost indices inside the loop
171
172#if CV_SSE2
173 volatile bool useSIMD = checkHardwareSupport(CV_CPU_SSE2);
174#endif
175
176#if 1
177 for( c = 0; c < cn*2; c++, prow1 += width, prow2 += width )
178 {
179 int diff_scale = c < cn ? 0 : 2;
180
181 // precompute
182 // v0 = min(row2[x-1/2], row2[x], row2[x+1/2]) and
183 // v1 = max(row2[x-1/2], row2[x], row2[x+1/2]) and
184 for( x = minX2; x < maxX2; x++ )
185 {
186 int v = prow2[x];
187 int vl = x > 0 ? (v + prow2[x-1])/2 : v;
188 int vr = x < width-1 ? (v + prow2[x+1])/2 : v;
189 int v0 = min(vl, vr); v0 = min(v0, v);
190 int v1 = max(vl, vr); v1 = max(v1, v);
191 buffer[x] = (PixType)v0;
192 buffer[x + width2] = (PixType)v1;
193 }
194
195 for( x = minX1; x < maxX1; x++ )
196 {
197 int u = prow1[x];
198 int ul = x > 0 ? (u + prow1[x-1])/2 : u;
199 int ur = x < width-1 ? (u + prow1[x+1])/2 : u;
200 int u0 = min(ul, ur); u0 = min(u0, u);
201 int u1 = max(ul, ur); u1 = max(u1, u);
202
203 int minDlocal = max(minD, x-width+1);
204 int maxDlocal = min(maxD, x);
205 int d;
206 for( d = minD; d < minDlocal; d++ )
207 {
208 int v = prow2[0];
209 int v0 = buffer[0];
210 int v1 = buffer[width2];
211 int c0 = max(0, u - v1); c0 = max(c0, v0 - u);
212 int c1 = max(0, v - u1); c1 = max(c1, u0 - v);
213
214 cost[x*D + d] = (CostType)(cost[x*D+d] + (min(c0, c1) >> diff_scale));
215 }
216 #if CV_SSE2
217 if( useSIMD )
218 {
219 __m128i _u = _mm_set1_epi8((char)u), _u0 = _mm_set1_epi8((char)u0);
220 __m128i _u1 = _mm_set1_epi8((char)u1), z = _mm_setzero_si128();
221 __m128i ds = _mm_cvtsi32_si128(diff_scale);
222
223 for( ; d < maxDlocal - 15; d += 16 )
224 {
225 __m128i _v = _mm_loadu_si128((const __m128i*)(prow2 + width-x-1 + d));
226 __m128i _v0 = _mm_loadu_si128((const __m128i*)(buffer + width-x-1 + d));
227 __m128i _v1 = _mm_loadu_si128((const __m128i*)(buffer + width-x-1 + d + width2));
228 __m128i c0 = _mm_max_epu8(_mm_subs_epu8(_u, _v1), _mm_subs_epu8(_v0, _u));
229 __m128i c1 = _mm_max_epu8(_mm_subs_epu8(_v, _u1), _mm_subs_epu8(_u0, _v));
230 __m128i diff = _mm_min_epu8(c0, c1);
231
232 c0 = _mm_load_si128((__m128i*)(cost + x*D + d));
233 c1 = _mm_load_si128((__m128i*)(cost + x*D + d + 8));
234
235 _mm_store_si128((__m128i*)(cost + x*D + d), _mm_adds_epi16(c0, _mm_srl_epi16(_mm_unpacklo_epi8(diff,z), ds)));
236 _mm_store_si128((__m128i*)(cost + x*D + d + 8), _mm_adds_epi16(c1, _mm_srl_epi16(_mm_unpackhi_epi8(diff,z), ds)));
237 }
238 }
239 #endif
240 for( ; d < maxDlocal; d++ )
241 {
242 int v = prow2[width-x-1 + d];
243 int v0 = buffer[width-x-1 + d];
244 int v1 = buffer[width-x-1 + d + width2];
245 int c0 = max(0, u - v1); c0 = max(c0, v0 - u);
246 int c1 = max(0, v - u1); c1 = max(c1, u0 - v);
247
248 cost[x*D + d] = (CostType)(cost[x*D+d] + (min(c0, c1) >> diff_scale));
249 }
250 for( ; d < maxD; d++ )
251 {
252 int v = prow2[width-1];
253 int v0 = buffer[width-1];
254 int v1 = buffer[width-1 + width2];
255 int c0 = max(0, u - v1); c0 = max(c0, v0 - u);
256 int c1 = max(0, v - u1); c1 = max(c1, u0 - v);
257
258 cost[x*D + d] = (CostType)(cost[x*D+d] + (min(c0, c1) >> diff_scale));
259 }
260 }
261 }
262#else
263 for( c = 0; c < cn*2; c++, prow1 += width, prow2 += width )
264 {
265 for( x = minX1; x < maxX1; x++ )
266 {
267 int u = prow1[x];
268 #if CV_SSE2
269 if( useSIMD )
270 {
271 __m128i _u = _mm_set1_epi8(u), z = _mm_setzero_si128();
272
273 for( int d = minD; d < maxD; d += 16 )
274 {
275 __m128i _v = _mm_loadu_si128((const __m128i*)(prow2 + width-1-x + d));
276 __m128i diff = _mm_adds_epu8(_mm_subs_epu8(_u,_v), _mm_subs_epu8(_v,_u));
277 __m128i c0 = _mm_load_si128((__m128i*)(cost + x*D + d));
278 __m128i c1 = _mm_load_si128((__m128i*)(cost + x*D + d + 8));
279
280 _mm_store_si128((__m128i*)(cost + x*D + d), _mm_adds_epi16(c0, _mm_unpacklo_epi8(diff,z)));
281 _mm_store_si128((__m128i*)(cost + x*D + d + 8), _mm_adds_epi16(c1, _mm_unpackhi_epi8(diff,z)));
282 }
283 }
284 else
285 #endif
286 {
287 for( int d = minD; d < maxD; d++ )
288 {
289 int v = prow2[width-1-x + d];
290 cost[x*D + d] = (CostType)(cost[x*D + d] + (CostType)std::abs(u - v));
291 }
292 }
293 }
294 }
295#endif
296}

299/*
300 computes disparity for "roi" in img1 w.r.t. img2 and write it to disp1buf.
301 that is, disp1buf(x, y)=d means that img1(x+roi.x, y+roi.y) ~ img2(x+roi.x-d, y+roi.y).
302 minD <= d < maxD.
303 disp2full is the reverse disparity map, that is:
304 disp2full(x+roi.x,y+roi.y)=d means that img2(x+roi.x, y+roi.y) ~ img1(x+roi.x+d, y+roi.y)
305
306 note that disp1buf will have the same size as the roi and
307 disp2full will have the same size as img1 (or img2).
308 On exit disp2buf is not the final disparity, it is an intermediate result that becomes
309 final after all the tiles are processed.
310
311 the disparity in disp1buf is written with sub-pixel accuracy
312 (4 fractional bits, see CvStereoSGBM::DISP_SCALE),
313 using quadratic interpolation, while the disparity in disp2buf
314 is written as is, without interpolation.
315
316 disp2cost also has the same size as img1 (or img2).
317 It contains the minimum current cost, used to find the best disparity, corresponding to the minimal cost.
318 */
319static void computeDisparitySGBM( const Mat& img1, const Mat& img2,
320 Mat& disp1, const StereoSGBM& params,
321 Mat& buffer )
322{
323#if CV_SSE2
324 static const uchar LSBTab[] =
325 {
326 0, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0, 4, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
327 5, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0, 4, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
328 6, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0, 4, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
329 5, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0, 4, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
330 7, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0, 4, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
331 5, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0, 4, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
332 6, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0, 4, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
333 5, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0, 4, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0
334 };
335
336 volatile bool useSIMD = checkHardwareSupport(CV_CPU_SSE2);
337#endif
338
339 const int ALIGN = 16;
340 const int DISP_SHIFT = StereoSGBM::DISP_SHIFT;
341 const int DISP_SCALE = StereoSGBM::DISP_SCALE;
342 const CostType MAX_COST = SHRT_MAX;
343
344 int minD = params.minDisparity, maxD = minD + params.numberOfDisparities;
345 Size SADWindowSize;
346 SADWindowSize.width = SADWindowSize.height = params.SADWindowSize > 0 ? params.SADWindowSize : 5;
347 int ftzero = max(params.preFilterCap, 15) | 1;
348 int uniquenessRatio = params.uniquenessRatio >= 0 ? params.uniquenessRatio : 10;
349 int disp12MaxDiff = params.disp12MaxDiff > 0 ? params.disp12MaxDiff : 1;
350 int P1 = params.P1 > 0 ? params.P1 : 2, P2 = max(params.P2 > 0 ? params.P2 : 5, P1+1);
351 int k, width = disp1.cols, height = disp1.rows;
352 int minX1 = max(-maxD, 0), maxX1 = width + min(-minD, 0);
353 int D = maxD - minD, width1 = maxX1 - minX1;
354 int INVALID_DISP = minD - 1, INVALID_DISP_SCALED = INVALID_DISP*DISP_SCALE;
355 int SW2 = SADWindowSize.width/2, SH2 = SADWindowSize.height/2;
356 int npasses = params.fullDP ? 2 : 1;
357 const int TAB_OFS = 256*4, TAB_SIZE = 256 + TAB_OFS*2;
358 PixType clipTab[TAB_SIZE];
359
360 for( k = 0; k < TAB_SIZE; k++ )
361 clipTab[k] = (PixType)(min(max(k - TAB_OFS, -ftzero), ftzero) + ftzero);
362
363 if( minX1 >= maxX1 )
364 {
365 disp1 = Scalar::all(INVALID_DISP_SCALED);
366 return;
367 }
368
369 CV_Assert( D % 16 == 0 );
370
371 // NR - the number of directions. the loop on x below that computes Lr assumes that NR == 8.
372 // if you change NR, please, modify the loop as well.
373 int D2 = D+16, NRD2 = NR2*D2;
374
375 // the number of L_r(.,.) and min_k L_r(.,.) lines in the buffer:
376 // for 8-way dynamic programming we need the current row and
377 // the previous row, i.e. 2 rows in total
378 const int NLR = 2;
379 const int LrBorder = NLR - 1;
380
381 // for each possible stereo match (img1(x,y) <=> img2(x-d,y))
382 // we keep pixel difference cost (C) and the summary cost over NR directions (S).
383 // we also keep all the partial costs for the previous line L_r(x,d) and also min_k L_r(x, k)
384 size_t costBufSize = width1*D;
385 size_t CSBufSize = costBufSize*(params.fullDP ? height : 1);
386 size_t minLrSize = (width1 + LrBorder*2)*NR2, LrSize = minLrSize*D2;
387 int hsumBufNRows = SH2*2 + 2;
388 size_t totalBufSize = (LrSize + minLrSize)*NLR*sizeof(CostType) + // minLr[] and Lr[]
389 costBufSize*(hsumBufNRows + 1)*sizeof(CostType) + // hsumBuf, pixdiff
390 CSBufSize*2*sizeof(CostType) + // C, S
391 width*16*img1.channels()*sizeof(PixType) + // temp buffer for computing per-pixel cost
392 width*(sizeof(CostType) + sizeof(DispType)) + 1024; // disp2cost + disp2
393
394 if( !buffer.data || !buffer.isContinuous() ||
395 buffer.cols*buffer.rows*buffer.elemSize() < totalBufSize )
396 buffer.create(1, (int)totalBufSize, CV_8U);
397
398 // summary cost over different (nDirs) directions
399 CostType* Cbuf = (CostType*)alignPtr(buffer.data, ALIGN);
400 memset(Cbuf, 0, CSBufSize*sizeof(CostType));
401 CostType* Sbuf = Cbuf + CSBufSize;
402 CostType* hsumBuf = Sbuf + CSBufSize;
403 CostType* pixDiff = hsumBuf + costBufSize*hsumBufNRows;
404
405 CostType* disp2cost = pixDiff + costBufSize + (LrSize + minLrSize)*NLR;
406 DispType* disp2ptr = (DispType*)(disp2cost + width);
407 PixType* tempBuf = (PixType*)(disp2ptr + width);
408
409 // add P2 to every C(x,y). it saves a few operations in the inner loops
410 for( k = 0; k < width1*D; k++ )
411 Cbuf[k] = (CostType)P2;
412
413 for( int pass = 1; pass <= npasses; pass++ )
414 {
415 int x1, y1, x2, y2, dx, dy;
416
417 if( pass == 1 )
418 {
419 y1 = 0; y2 = height; dy = 1;
420 x1 = 0; x2 = width1; dx = 1;
421 }
422 else
423 {
424 y1 = height-1; y2 = -1; dy = -1;
425 x1 = width1-1; x2 = -1; dx = -1;
426 }
427
428 CostType *Lr[NLR]={0}, *minLr[NLR]={0};
429
430 for( k = 0; k < NLR; k++ )
431 {
432 // shift Lr[k] and minLr[k] pointers, because we allocated them with the borders,
433 // and will occasionally use negative indices with the arrays
434 // we need to shift Lr[k] pointers by 1, to give the space for d=-1.
435 // however, then the alignment will be imperfect, i.e. bad for SSE,
436 // thus we shift the pointers by 8 (8*sizeof(short) == 16 - ideal alignment)
437 Lr[k] = pixDiff + costBufSize + LrSize*k + NRD2*LrBorder + 8;
438 memset( Lr[k] - LrBorder*NRD2 - 8, 0, LrSize*sizeof(CostType) );
439 minLr[k] = pixDiff + costBufSize + LrSize*NLR + minLrSize*k + NR2*LrBorder;
440 memset( minLr[k] - LrBorder*NR2, 0, minLrSize*sizeof(CostType) );
441 }
442
443 for( int y = y1; y != y2; y += dy )
444 {
445 int x, d;
446 DispType* disp1ptr = disp1.ptr<DispType>(y);
447 CostType* C = Cbuf + (!params.fullDP ? 0 : y*costBufSize);
448 CostType* S = Sbuf + (!params.fullDP ? 0 : y*costBufSize);
449
450 if( pass == 1 ) // compute C on the first pass, and reuse it on the second pass, if any.
451 {
452 int dy1 = y == 0 ? 0 : y + SH2, dy2 = y == 0 ? SH2 : dy1;
453
454 for( k = dy1; k <= dy2; k++ )
455 {
456 CostType* hsumAdd = hsumBuf + (min(k, height-1) % hsumBufNRows)*costBufSize;
457
458 if( k < height )
459 {
460 calcPixelCostBT( img1, img2, k, minD, maxD, pixDiff, tempBuf, clipTab, TAB_OFS, ftzero );
461
462 memset(hsumAdd, 0, D*sizeof(CostType));
463 for( x = 0; x <= SW2*D; x += D )
464 {
465 int scale = x == 0 ? SW2 + 1 : 1;
466 for( d = 0; d < D; d++ )
467 hsumAdd[d] = (CostType)(hsumAdd[d] + pixDiff[x + d]*scale);
468 }
469
470 if( y > 0 )
471 {
472 const CostType* hsumSub = hsumBuf + (max(y - SH2 - 1, 0) % hsumBufNRows)*costBufSize;
473 const CostType* Cprev = !params.fullDP || y == 0 ? C : C - costBufSize;
474
475 for( x = D; x < width1*D; x += D )
476 {
477 const CostType* pixAdd = pixDiff + min(x + SW2*D, (width1-1)*D);
478 const CostType* pixSub = pixDiff + max(x - (SW2+1)*D, 0);
479
480 #if CV_SSE2
481 if( useSIMD )
482 {
483 for( d = 0; d < D; d += 8 )
484 {
485 __m128i hv = _mm_load_si128((const __m128i*)(hsumAdd + x - D + d));
486 __m128i Cx = _mm_load_si128((__m128i*)(Cprev + x + d));
487 hv = _mm_adds_epi16(_mm_subs_epi16(hv,
488 _mm_load_si128((const __m128i*)(pixSub + d))),
489 _mm_load_si128((const __m128i*)(pixAdd + d)));
490 Cx = _mm_adds_epi16(_mm_subs_epi16(Cx,
491 _mm_load_si128((const __m128i*)(hsumSub + x + d))),
492 hv);
493 _mm_store_si128((__m128i*)(hsumAdd + x + d), hv);
494 _mm_store_si128((__m128i*)(C + x + d), Cx);
495 }
496 }
497 else
498 #endif
499 {
500 for( d = 0; d < D; d++ )
501 {
502 int hv = hsumAdd[x + d] = (CostType)(hsumAdd[x - D + d] + pixAdd[d] - pixSub[d]);
503 C[x + d] = (CostType)(Cprev[x + d] + hv - hsumSub[x + d]);
504 }
505 }
506 }
507 }
508 else
509 {
510 for( x = D; x < width1*D; x += D )
511 {
512 const CostType* pixAdd = pixDiff + min(x + SW2*D, (width1-1)*D);
513 const CostType* pixSub = pixDiff + max(x - (SW2+1)*D, 0);
514
515 for( d = 0; d < D; d++ )
516 hsumAdd[x + d] = (CostType)(hsumAdd[x - D + d] + pixAdd[d] - pixSub[d]);
517 }
518 }
519 }
520
521 if( y == 0 )
522 {
523 int scale = k == 0 ? SH2 + 1 : 1;
524 for( x = 0; x < width1*D; x++ )
525 C[x] = (CostType)(C[x] + hsumAdd[x]*scale);
526 }
527 }
528
529 // also, clear the S buffer
530 for( k = 0; k < width1*D; k++ )
531 S[k] = 0;
532 }
533
534 // clear the left and the right borders
535 memset( Lr[0] - NRD2*LrBorder - 8, 0, NRD2*LrBorder*sizeof(CostType) );
536 memset( Lr[0] + width1*NRD2 - 8, 0, NRD2*LrBorder*sizeof(CostType) );
537 memset( minLr[0] - NR2*LrBorder, 0, NR2*LrBorder*sizeof(CostType) );
538 memset( minLr[0] + width1*NR2, 0, NR2*LrBorder*sizeof(CostType) );
539
540 /*
541 [formula 13 in the paper]
542 compute L_r(p, d) = C(p, d) +
543 min(L_r(p-r, d),
544 L_r(p-r, d-1) + P1,
545 L_r(p-r, d+1) + P1,
546 min_k L_r(p-r, k) + P2) - min_k L_r(p-r, k)
547 where p = (x,y), r is one of the directions.
548 we process all the directions at once:
549 0: r=(-dx, 0)
550 1: r=(-1, -dy)
551 2: r=(0, -dy)
552 3: r=(1, -dy)
553 4: r=(-2, -dy)
554 5: r=(-1, -dy*2)
555 6: r=(1, -dy*2)
556 7: r=(2, -dy)
557 */
558 for( x = x1; x != x2; x += dx )
559 {
560 int xm = x*NR2, xd = xm*D2;
561
562 int delta0 = minLr[0][xm - dx*NR2] + P2, delta1 = minLr[1][xm - NR2 + 1] + P2;
563 int delta2 = minLr[1][xm + 2] + P2, delta3 = minLr[1][xm + NR2 + 3] + P2;
564
565 CostType* Lr_p0 = Lr[0] + xd - dx*NRD2;
566 CostType* Lr_p1 = Lr[1] + xd - NRD2 + D2;
567 CostType* Lr_p2 = Lr[1] + xd + D2*2;
568 CostType* Lr_p3 = Lr[1] + xd + NRD2 + D2*3;
569
570 Lr_p0[-1] = Lr_p0[D] = Lr_p1[-1] = Lr_p1[D] =
571 Lr_p2[-1] = Lr_p2[D] = Lr_p3[-1] = Lr_p3[D] = MAX_COST;
572
573 CostType* Lr_p = Lr[0] + xd;
574 const CostType* Cp = C + x*D;
575 CostType* Sp = S + x*D;
576
577 #if CV_SSE2
578 if( useSIMD )
579 {
580 __m128i _P1 = _mm_set1_epi16((short)P1);
581
582 __m128i _delta0 = _mm_set1_epi16((short)delta0);
583 __m128i _delta1 = _mm_set1_epi16((short)delta1);
584 __m128i _delta2 = _mm_set1_epi16((short)delta2);
585 __m128i _delta3 = _mm_set1_epi16((short)delta3);
586 __m128i _minL0 = _mm_set1_epi16((short)MAX_COST);
587
588 for( d = 0; d < D; d += 8 )
589 {
590 __m128i Cpd = _mm_load_si128((const __m128i*)(Cp + d));
591 __m128i L0, L1, L2, L3;
592
593 L0 = _mm_load_si128((const __m128i*)(Lr_p0 + d));
594 L1 = _mm_load_si128((const __m128i*)(Lr_p1 + d));
595 L2 = _mm_load_si128((const __m128i*)(Lr_p2 + d));
596 L3 = _mm_load_si128((const __m128i*)(Lr_p3 + d));
597
598 L0 = _mm_min_epi16(L0, _mm_adds_epi16(_mm_loadu_si128((const __m128i*)(Lr_p0 + d - 1)), _P1));
599 L0 = _mm_min_epi16(L0, _mm_adds_epi16(_mm_loadu_si128((const __m128i*)(Lr_p0 + d + 1)), _P1));
600
601 L1 = _mm_min_epi16(L1, _mm_adds_epi16(_mm_loadu_si128((const __m128i*)(Lr_p1 + d - 1)), _P1));
602 L1 = _mm_min_epi16(L1, _mm_adds_epi16(_mm_loadu_si128((const __m128i*)(Lr_p1 + d + 1)), _P1));
603
604 L2 = _mm_min_epi16(L2, _mm_adds_epi16(_mm_loadu_si128((const __m128i*)(Lr_p2 + d - 1)), _P1));
605 L2 = _mm_min_epi16(L2, _mm_adds_epi16(_mm_loadu_si128((const __m128i*)(Lr_p2 + d + 1)), _P1));
606
607 L3 = _mm_min_epi16(L3, _mm_adds_epi16(_mm_loadu_si128((const __m128i*)(Lr_p3 + d - 1)), _P1));
608 L3 = _mm_min_epi16(L3, _mm_adds_epi16(_mm_loadu_si128((const __m128i*)(Lr_p3 + d + 1)), _P1));
609
610 L0 = _mm_min_epi16(L0, _delta0);
611 L0 = _mm_adds_epi16(_mm_subs_epi16(L0, _delta0), Cpd);
612
613 L1 = _mm_min_epi16(L1, _delta1);
614 L1 = _mm_adds_epi16(_mm_subs_epi16(L1, _delta1), Cpd);
615
616 L2 = _mm_min_epi16(L2, _delta2);
617 L2 = _mm_adds_epi16(_mm_subs_epi16(L2, _delta2), Cpd);
618
619 L3 = _mm_min_epi16(L3, _delta3);
620 L3 = _mm_adds_epi16(_mm_subs_epi16(L3, _delta3), Cpd);
621
622 _mm_store_si128( (__m128i*)(Lr_p + d), L0);
623 _mm_store_si128( (__m128i*)(Lr_p + d + D2), L1);
624 _mm_store_si128( (__m128i*)(Lr_p + d + D2*2), L2);
625 _mm_store_si128( (__m128i*)(Lr_p + d + D2*3), L3);
626
627 __m128i t0 = _mm_min_epi16(_mm_unpacklo_epi16(L0, L2), _mm_unpackhi_epi16(L0, L2));
628 __m128i t1 = _mm_min_epi16(_mm_unpacklo_epi16(L1, L3), _mm_unpackhi_epi16(L1, L3));
629 t0 = _mm_min_epi16(_mm_unpacklo_epi16(t0, t1), _mm_unpackhi_epi16(t0, t1));
630 _minL0 = _mm_min_epi16(_minL0, t0);
631
632 __m128i Sval = _mm_load_si128((const __m128i*)(Sp + d));
633
634 L0 = _mm_adds_epi16(L0, L1);
635 L2 = _mm_adds_epi16(L2, L3);
636 Sval = _mm_adds_epi16(Sval, L0);
637 Sval = _mm_adds_epi16(Sval, L2);
638
639 _mm_store_si128((__m128i*)(Sp + d), Sval);
640 }
641
642 _minL0 = _mm_min_epi16(_minL0, _mm_srli_si128(_minL0, 8));
643 _mm_storel_epi64((__m128i*)&minLr[0][xm], _minL0);
644 }
645 else
646 #endif
647 {
648 int minL0 = MAX_COST, minL1 = MAX_COST, minL2 = MAX_COST, minL3 = MAX_COST;
649
650 for( d = 0; d < D; d++ )
651 {
652 int Cpd = Cp[d], L0, L1, L2, L3;
653
654 L0 = Cpd + min((int)Lr_p0[d], min(Lr_p0[d-1] + P1, min(Lr_p0[d+1] + P1, delta0))) - delta0;
655 L1 = Cpd + min((int)Lr_p1[d], min(Lr_p1[d-1] + P1, min(Lr_p1[d+1] + P1, delta1))) - delta1;
656 L2 = Cpd + min((int)Lr_p2[d], min(Lr_p2[d-1] + P1, min(Lr_p2[d+1] + P1, delta2))) - delta2;
657 L3 = Cpd + min((int)Lr_p3[d], min(Lr_p3[d-1] + P1, min(Lr_p3[d+1] + P1, delta3))) - delta3;
658
659 Lr_p[d] = (CostType)L0;
660 minL0 = min(minL0, L0);
661
662 Lr_p[d + D2] = (CostType)L1;
663 minL1 = min(minL1, L1);
664
665 Lr_p[d + D2*2] = (CostType)L2;
666 minL2 = min(minL2, L2);
667
668 Lr_p[d + D2*3] = (CostType)L3;
669 minL3 = min(minL3, L3);
670
671 Sp[d] = saturate_cast<CostType>(Sp[d] + L0 + L1 + L2 + L3);
672 }
673 minLr[0][xm] = (CostType)minL0;
674 minLr[0][xm+1] = (CostType)minL1;
675 minLr[0][xm+2] = (CostType)minL2;
676 minLr[0][xm+3] = (CostType)minL3;
677 }
678 }
679
680 if( pass == npasses )
681 {
682 for( x = 0; x < width; x++ )
683 {
684 disp1ptr[x] = disp2ptr[x] = (DispType)INVALID_DISP_SCALED;
685 disp2cost[x] = MAX_COST;
686 }
687
688 for( x = width1 - 1; x >= 0; x-- )
689 {
690 CostType* Sp = S + x*D;
691 int minS = MAX_COST, bestDisp = -1;
692
693 if( npasses == 1 )
694 {
695 int xm = x*NR2, xd = xm*D2;
696
697 int minL0 = MAX_COST;
698 int delta0 = minLr[0][xm + NR2] + P2;
699 CostType* Lr_p0 = Lr[0] + xd + NRD2;
700 Lr_p0[-1] = Lr_p0[D] = MAX_COST;
701 CostType* Lr_p = Lr[0] + xd;
702
703 const CostType* Cp = C + x*D;
704
705 #if CV_SSE2
706 if( useSIMD )
707 {
708 __m128i _P1 = _mm_set1_epi16((short)P1);
709 __m128i _delta0 = _mm_set1_epi16((short)delta0);
710
711 __m128i _minL0 = _mm_set1_epi16((short)minL0);
712 __m128i _minS = _mm_set1_epi16(MAX_COST), _bestDisp = _mm_set1_epi16(-1);
713 __m128i _d8 = _mm_setr_epi16(0, 1, 2, 3, 4, 5, 6, 7), _8 = _mm_set1_epi16(8);
714
715 for( d = 0; d < D; d += 8 )
716 {
717 __m128i Cpd = _mm_load_si128((const __m128i*)(Cp + d)), L0;
718
719 L0 = _mm_load_si128((const __m128i*)(Lr_p0 + d));
720 L0 = _mm_min_epi16(L0, _mm_adds_epi16(_mm_loadu_si128((const __m128i*)(Lr_p0 + d - 1)), _P1));
721 L0 = _mm_min_epi16(L0, _mm_adds_epi16(_mm_loadu_si128((const __m128i*)(Lr_p0 + d + 1)), _P1));
722 L0 = _mm_min_epi16(L0, _delta0);
723 L0 = _mm_adds_epi16(_mm_subs_epi16(L0, _delta0), Cpd);
724
725 _mm_store_si128((__m128i*)(Lr_p + d), L0);
726 _minL0 = _mm_min_epi16(_minL0, L0);
727 L0 = _mm_adds_epi16(L0, *(__m128i*)(Sp + d));
728 _mm_store_si128((__m128i*)(Sp + d), L0);
729
730 __m128i mask = _mm_cmpgt_epi16(_minS, L0);
731 _minS = _mm_min_epi16(_minS, L0);
732 _bestDisp = _mm_xor_si128(_bestDisp, _mm_and_si128(_mm_xor_si128(_bestDisp,_d8), mask));
733 _d8 = _mm_adds_epi16(_d8, _8);
734 }
735
736 short CV_DECL_ALIGNED(16) bestDispBuf[8];
737 _mm_store_si128((__m128i*)bestDispBuf, _bestDisp);
738
739 _minL0 = _mm_min_epi16(_minL0, _mm_srli_si128(_minL0, 8));
740 _minL0 = _mm_min_epi16(_minL0, _mm_srli_si128(_minL0, 4));
741 _minL0 = _mm_min_epi16(_minL0, _mm_srli_si128(_minL0, 2));
742
743 __m128i qS = _mm_min_epi16(_minS, _mm_srli_si128(_minS, 8));
744 qS = _mm_min_epi16(qS, _mm_srli_si128(qS, 4));
745 qS = _mm_min_epi16(qS, _mm_srli_si128(qS, 2));
746
747 minLr[0][xm] = (CostType)_mm_cvtsi128_si32(_minL0);
748 minS = (CostType)_mm_cvtsi128_si32(qS);
749
750 qS = _mm_shuffle_epi32(_mm_unpacklo_epi16(qS, qS), 0);
751 qS = _mm_cmpeq_epi16(_minS, qS);
752 int idx = _mm_movemask_epi8(_mm_packs_epi16(qS, qS)) & 255;
753
754 bestDisp = bestDispBuf[LSBTab[idx]];
755 }
756 else
757 #endif
758 {
759 for( d = 0; d < D; d++ )
760 {
761 int L0 = Cp[d] + min((int)Lr_p0[d], min(Lr_p0[d-1] + P1, min(Lr_p0[d+1] + P1, delta0))) - delta0;
762
763 Lr_p[d] = (CostType)L0;
764 minL0 = min(minL0, L0);
765
766 int Sval = Sp[d] = saturate_cast<CostType>(Sp[d] + L0);
767 if( Sval < minS )
768 {
769 minS = Sval;
770 bestDisp = d;
771 }
772 }
773 minLr[0][xm] = (CostType)minL0;
774 }
775 }
776 else
777 {
778 for( d = 0; d < D; d++ )
779 {
780 int Sval = Sp[d];
781 if( Sval < minS )
782 {
783 minS = Sval;
784 bestDisp = d;
785 }
786 }
787 }
788
789 for( d = 0; d < D; d++ )
790 {
791 if( Sp[d]*(100 - uniquenessRatio) < minS*100 && std::abs(bestDisp - d) > 1 )
792 break;
793 }
794 if( d < D )
795 continue;
796 d = bestDisp;
797 int _x2 = x + minX1 - d - minD;
798 if( disp2cost[_x2] > minS )
799 {
800 disp2cost[_x2] = (CostType)minS;
801 disp2ptr[_x2] = (DispType)(d + minD);
802 }
803
804 if( 0 < d && d < D-1 )
805 {
806 // do subpixel quadratic interpolation:
807 // fit parabola into (x1=d-1, y1=Sp[d-1]), (x2=d, y2=Sp[d]), (x3=d+1, y3=Sp[d+1])
808 // then find minimum of the parabola.
809 int denom2 = max(Sp[d-1] + Sp[d+1] - 2*Sp[d], 1);
810 d = d*DISP_SCALE + ((Sp[d-1] - Sp[d+1])*DISP_SCALE + denom2)/(denom2*2);
811 }
812 else
813 d *= DISP_SCALE;
814 disp1ptr[x + minX1] = (DispType)(d + minD*DISP_SCALE);
815 }
816
817 for( x = minX1; x < maxX1; x++ )
818 {
819 // we round the computed disparity both towards -inf and +inf and check
820 // if either of the corresponding disparities in disp2 is consistent.
821 // This is to give the computed disparity a chance to look valid if it is.
822 int d1 = disp1ptr[x];
823 if( d1 == INVALID_DISP_SCALED )
824 continue;
825 int _d = d1 >> DISP_SHIFT;
826 int d_ = (d1 + DISP_SCALE-1) >> DISP_SHIFT;
827 int _x = x - _d, x_ = x - d_;
828 if( 0 <= _x && _x < width && disp2ptr[_x] >= minD && std::abs(disp2ptr[_x] - _d) > disp12MaxDiff &&
829 0 <= x_ && x_ < width && disp2ptr[x_] >= minD && std::abs(disp2ptr[x_] - d_) > disp12MaxDiff )
830 disp1ptr[x] = (DispType)INVALID_DISP_SCALED;
831 }
832 }
833
834 // now shift the cyclic buffers
835 std::swap( Lr[0], Lr[1] );
836 std::swap( minLr[0], minLr[1] );
837 }
838 }
839}

844 OutputArray _disp )
845{
846 Mat left = _left.getMat(), right = _right.getMat();
847 CV_Assert( left.size() == right.size() && left.type() == right.type() &&

861Rect getValidDisparityROI( Rect roi1, Rect roi2,
862 int minDisparity,
863 int numberOfDisparities,
864 int SADWindowSize )
865{
866 int SW2 = SADWindowSize/2;
867 int minD = minDisparity, maxD = minDisparity + numberOfDisparities - 1;
868
869 int xmin = max(roi1.x, roi2.x + maxD) + SW2;
870 int xmax = min(roi1.x + roi1.width, roi2.x + roi2.width - minD) - SW2;
871 int ymin = max(roi1.y, roi2.y) + SW2;
872 int ymax = min(roi1.y + roi1.height, roi2.y + roi2.height) - SW2;
873
874 Rect r(xmin, ymin, xmax - xmin, ymax - ymin);
875
876 return r.width > 0 && r.height > 0 ? r : Rect();
877}

881namespace
882{
883 template <typename T>
884 void filterSpecklesImpl(cv::Mat& img, int newVal, int maxSpeckleSize, int maxDiff, cv::Mat& _buf)
885 {
886 using namespace cv;
887
888 int width = img.cols, height = img.rows, npixels = width*height;
889 size_t bufSize = npixels*(int)(sizeof(Point2s) + sizeof(int) + sizeof(uchar));
890 if( !_buf.isContinuous() || !_buf.data || _buf.cols*_buf.rows*_buf.elemSize() < bufSize )
891 _buf.create(1, (int)bufSize, CV_8U);
892
893 uchar* buf = _buf.data;
894 int i, j, dstep = (int)(img.step/sizeof(T));
895 int* labels = (int*)buf;
896 buf += npixels*sizeof(labels[0]);
897 Point2s* wbuf = (Point2s*)buf;
898 buf += npixels*sizeof(wbuf[0]);
899 uchar* rtype = (uchar*)buf;
900 int curlabel = 0;
901
902 // clear out label assignments
903 memset(labels, 0, npixels*sizeof(labels[0]));
904
905 for( i = 0; i < height; i++ )
906 {
907 T* ds = img.ptr<T>(i);
908 int* ls = labels + width*i;
909
910 for( j = 0; j < width; j++ )
911 {
912 if( ds[j] != newVal ) // not a bad disparity
913 {
914 if( ls[j] ) // has a label, check for bad label
915 {
916 if( rtype[ls[j]] ) // small region, zero out disparity
917 ds[j] = (T)newVal;
918 }
919 // no label, assign and propagate
920 else
921 {
922 Point2s* ws = wbuf; // initialize wavefront
923 Point2s p((short)j, (short)i); // current pixel
924 curlabel++; // next label
925 int count = 0; // current region size
926 ls[j] = curlabel;
927
928 // wavefront propagation
929 while( ws >= wbuf ) // wavefront not empty
930 {
931 count++;
932 // put neighbors onto wavefront
933 T* dpp = &img.at<T>(p.y, p.x);
934 T dp = *dpp;
935 int* lpp = labels + width*p.y + p.x;
936
937 if( p.y < height-1 && !lpp[+width] && dpp[+dstep] != newVal && std::abs(dp - dpp[+dstep]) <= maxDiff )
938 {
939 lpp[+width] = curlabel;
940 *ws++ = Point2s(p.x, p.y+1);
941 }
942
943 if( p.y > 0 && !lpp[-width] && dpp[-dstep] != newVal && std::abs(dp - dpp[-dstep]) <= maxDiff )
944 {
945 lpp[-width] = curlabel;
946 *ws++ = Point2s(p.x, p.y-1);
947 }
948
949 if( p.x < width-1 && !lpp[+1] && dpp[+1] != newVal && std::abs(dp - dpp[+1]) <= maxDiff )
950 {
951 lpp[+1] = curlabel;
952 *ws++ = Point2s(p.x+1, p.y);
953 }
954
955 if( p.x > 0 && !lpp[-1] && dpp[-1] != newVal && std::abs(dp - dpp[-1]) <= maxDiff )
956 {
957 lpp[-1] = curlabel;
958 *ws++ = Point2s(p.x-1, p.y);
959 }
960
961 // pop most recent and propagate
962 // NB: could try least recent, maybe better convergence
963 p = *--ws;
964 }
965
966 // assign label type
967 if( count <= maxSpeckleSize ) // speckle region
968 {
969 rtype[ls[j]] = 1; // small region label
970 ds[j] = (T)newVal;
971 }
972 else

997void cv::validateDisparity( InputOutputArray _disp, InputArray _cost, int minDisparity,
998 int numberOfDisparities, int disp12MaxDiff )
999{
1000 Mat disp = _disp.getMat(), cost = _cost.getMat();
1001 int cols = disp.cols, rows = disp.rows;
1002 int minD = minDisparity, maxD = minDisparity + numberOfDisparities;
1003 int x, minX1 = max(maxD, 0), maxX1 = cols + min(minD, 0);
1004 AutoBuffer<int> _disp2buf(cols*2);
1005 int* disp2buf = _disp2buf;
1006 int* disp2cost = disp2buf + cols;
1007 const int DISP_SHIFT = 4, DISP_SCALE = 1 << DISP_SHIFT;
1008 int INVALID_DISP = minD - 1, INVALID_DISP_SCALED = INVALID_DISP*DISP_SCALE;
1009 int costType = cost.type();
1010
1011 disp12MaxDiff *= DISP_SCALE;
1012
1013 CV_Assert( numberOfDisparities > 0 && disp.type() == CV_16S &&
1014 (costType == CV_16S || costType == CV_32S) &&
1015 disp.size() == cost.size() );
1016
1017 for( int y = 0; y < rows; y++ )
1018 {
1019 short* dptr = disp.ptr<short>(y);
1020
1021 for( x = 0; x < cols; x++ )
1022 {
1023 disp2buf[x] = INVALID_DISP_SCALED;
1024 disp2cost[x] = INT_MAX;
1025 }
1026
1027 if( costType == CV_16S )
1028 {
1029 const short* cptr = cost.ptr<short>(y);
1030
1031 for( x = minX1; x < maxX1; x++ )
1032 {
1033 int d = dptr[x], c = cptr[x];
1034 int x2 = x - ((d + DISP_SCALE/2) >> DISP_SHIFT);
1035
1036 if( disp2cost[x2] > c )
1037 {
1038 disp2cost[x2] = c;
1039 disp2buf[x2] = d;
1040 }
1041 }
1042 }
1043 else
1044 {
1045 const int* cptr = cost.ptr<int>(y);
1046
1047 for( x = minX1; x < maxX1; x++ )
1048 {
1049 int d = dptr[x], c = cptr[x];
1050 int x2 = x - ((d + DISP_SCALE/2) >> DISP_SHIFT);
1051
1052 if( disp2cost[x2] < c )
1053 {
1054 disp2cost[x2] = c;
1055 disp2buf[x2] = d;
1056 }
1057 }
1058 }
1059
1060 for( x = minX1; x < maxX1; x++ )
1061 {
1062 // we round the computed disparity both towards -inf and +inf and check
1063 // if either of the corresponding disparities in disp2 is consistent.
1064 // This is to give the computed disparity a chance to look valid if it is.
1065 int d = dptr[x];
1066 if( d == INVALID_DISP_SCALED )
1067 continue;
1068 int d0 = d >> DISP_SHIFT;
1069 int d1 = (d + DISP_SCALE-1) >> DISP_SHIFT;
1070 int x0 = x - d0, x1 = x - d1;
1071 if( (0 <= x0 && x0 < cols && disp2buf[x0] > INVALID_DISP_SCALED && std::abs(disp2buf[x0] - d) > disp12MaxDiff) &&
1072 (0 <= x1 && x1 < cols && disp2buf[x1] > INVALID_DISP_SCALED && std::abs(disp2buf[x1] - d) > disp12MaxDiff) )
1073 dptr[x] = (short)INVALID_DISP_SCALED;
1074 }
1075 }
1076}
