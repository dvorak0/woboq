58static void icvGetQuadrangleHypotheses(CvSeq* contours, std::vector<std::pair<float, int> >& quads, int class_id)
59{
60 const float min_aspect_ratio = 0.3f;
61 const float max_aspect_ratio = 3.0f;
62 const float min_box_size = 10.0f;
63
64 for(CvSeq* seq = contours; seq != NULL; seq = seq->h_next)
65 {
66 CvBox2D box = cvMinAreaRect2(seq);
67 float box_size = MAX(box.size.width, box.size.height);
68 if(box_size < min_box_size)
69 {
70 continue;
71 }
72
73 float aspect_ratio = box.size.width/MAX(box.size.height, 1);
74 if(aspect_ratio < min_aspect_ratio || aspect_ratio > max_aspect_ratio)
75 {
76 continue;
77 }
78
79 quads.push_back(std::pair<float, int>(box_size, class_id));
80 }
81}

83static void countClasses(const std::vector<std::pair<float, int> >& pairs, size_t idx1, size_t idx2, std::vector<int>& counts)
84{
85 counts.assign(2, 0);
86 for(size_t i = idx1; i != idx2; i++)
87 {
88 counts[pairs[i].second]++;
89 }
90}

92inline bool less_pred(const std::pair<float, int>& p1, const std::pair<float, int>& p2)
93{
94 return p1.first < p2.first;
95}

97// does a fast check if a chessboard is in the input image. This is a workaround to
98// a problem of cvFindChessboardCorners being slow on images with no chessboard
99// - src: input image
100// - size: chessboard size
101// Returns 1 if a chessboard can be in this image and findChessboardCorners should be called,
102// 0 if there is no chessboard, -1 in case of error
103int cvCheckChessboard(IplImage* src, CvSize size)
104{
105 if(src->nChannels > 1)
106 {
107 cvError(CV_BadNumChannels, "cvCheckChessboard", "supports single-channel images only",
108 __FILE__, __LINE__);
109 }
110
111 if(src->depth != 8)
112 {
113 cvError(CV_BadDepth, "cvCheckChessboard", "supports depth=8 images only",
114 __FILE__, __LINE__);
115 }
116
117 const int erosion_count = 1;
118 const float black_level = 20.f;
119 const float white_level = 130.f;
120 const float black_white_gap = 70.f;
121
122#if defined(DEBUG_WINDOWS)
123 cvNamedWindow("1", 1);
124 cvShowImage("1", src);
125 cvWaitKey(0);
126#endif //DEBUG_WINDOWS
127
128 CvMemStorage* storage = cvCreateMemStorage();
129
130 IplImage* white = cvCloneImage(src);
131 IplImage* black = cvCloneImage(src);
132
133 cvErode(white, white, NULL, erosion_count);
134 cvDilate(black, black, NULL, erosion_count);
135 IplImage* thresh = cvCreateImage(cvGetSize(src), IPL_DEPTH_8U, 1);
136
137 int result = 0;
138 for(float thresh_level = black_level; thresh_level < white_level && !result; thresh_level += 20.0f)
139 {
140 cvThreshold(white, thresh, thresh_level + black_white_gap, 255, CV_THRESH_BINARY);
141
142#if defined(DEBUG_WINDOWS)
143 cvShowImage("1", thresh);
144 cvWaitKey(0);
145#endif //DEBUG_WINDOWS
146
147 CvSeq* first = 0;
148 std::vector<std::pair<float, int> > quads;
149 cvFindContours(thresh, storage, &first, sizeof(CvContour), CV_RETR_CCOMP);
150 icvGetQuadrangleHypotheses(first, quads, 1);
151
152 cvThreshold(black, thresh, thresh_level, 255, CV_THRESH_BINARY_INV);
153
154#if defined(DEBUG_WINDOWS)
155 cvShowImage("1", thresh);
156 cvWaitKey(0);
157#endif //DEBUG_WINDOWS
158
159 cvFindContours(thresh, storage, &first, sizeof(CvContour), CV_RETR_CCOMP);
160 icvGetQuadrangleHypotheses(first, quads, 0);
161
162 const size_t min_quads_count = size.width*size.height/2;
163 std::sort(quads.begin(), quads.end(), less_pred);
164
165 // now check if there are many hypotheses with similar sizes
166 // do this by floodfill-style algorithm
167 const float size_rel_dev = 0.4f;
168
169 for(size_t i = 0; i < quads.size(); i++)
170 {
171 size_t j = i + 1;
172 for(; j < quads.size(); j++)
173 {
174 if(quads[j].first/quads[i].first > 1.0f + size_rel_dev)
175 {
176 break;
177 }
178 }
179
180 if(j + 1 > min_quads_count + i)
181 {
182 // check the number of black and white squares
183 std::vector<int> counts;
184 countClasses(quads, i, j, counts);
185 const int black_count = cvRound(ceil(size.width/2.0)*ceil(size.height/2.0));
186 const int white_count = cvRound(floor(size.width/2.0)*floor(size.height/2.0));
187 if(counts[0] < black_count*0.75 ||
188 counts[1] < white_count*0.75)
189 {
190 continue;
191 }
192 result = 1;
193 break;
194 }
195 }
196 }
197
198
199 cvReleaseImage(&thresh);
200 cvReleaseImage(&white);
201 cvReleaseImage(&black);
202 cvReleaseMemStorage(&storage);
203
204 return result;
205}
