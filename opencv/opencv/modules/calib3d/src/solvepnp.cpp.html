49bool cv::solvePnP( InputArray _opoints, InputArray _ipoints,
50 InputArray _cameraMatrix, InputArray _distCoeffs,
51 OutputArray _rvec, OutputArray _tvec, bool useExtrinsicGuess, int flags )
52{
53 Mat opoints = _opoints.getMat(), ipoints = _ipoints.getMat();
54 int npoints = std::max(opoints.checkVector(3, CV_32F), opoints.checkVector(3, CV_64F));
55 CV_Assert( npoints >= 0 && npoints == std::max(ipoints.checkVector(2, CV_32F), ipoints.checkVector(2, CV_64F)) );
56 Mat cameraMatrix = _cameraMatrix.getMat(), distCoeffs = _distCoeffs.getMat();
57
58 Mat rvec, tvec;
59 if( flags != CV_ITERATIVE )
60 useExtrinsicGuess = false;
61
62 if( useExtrinsicGuess )
63 {
64 int rtype = _rvec.type(), ttype = _tvec.type();
65 Size rsize = _rvec.size(), tsize = _tvec.size();
66 CV_Assert( (rtype == CV_32F || rtype == CV_64F) &&
67 (ttype == CV_32F || ttype == CV_64F) );
68 CV_Assert( (rsize == Size(1, 3) || rsize == Size(3, 1)) &&
69 (tsize == Size(1, 3) || tsize == Size(3, 1)) );
70 }
71 else
72 {
73 _rvec.create(3, 1, CV_64F);
74 _tvec.create(3, 1, CV_64F);
75 }
76 rvec = _rvec.getMat();
77 tvec = _tvec.getMat();
78
79 if (flags == CV_EPNP)
80 {
81 cv::Mat undistortedPoints;
82 cv::undistortPoints(ipoints, undistortedPoints, cameraMatrix, distCoeffs);
83 epnp PnP(cameraMatrix, opoints, undistortedPoints);
84
85 cv::Mat R;
86 PnP.compute_pose(R, tvec);
87 cv::Rodrigues(R, rvec);
88 return true;
89 }
90 else if (flags == CV_P3P)
91 {
92 CV_Assert( npoints == 4);
93 cv::Mat undistortedPoints;
94 cv::undistortPoints(ipoints, undistortedPoints, cameraMatrix, distCoeffs);
95 p3p P3Psolver(cameraMatrix);
96
97 cv::Mat R;
98 bool result = P3Psolver.solve(R, tvec, opoints, undistortedPoints);
99 if (result)
100 cv::Rodrigues(R, rvec);
101 return result;
102 }
103 else if (flags == CV_ITERATIVE)
104 {
105 CvMat c_objectPoints = opoints, c_imagePoints = ipoints;
106 CvMat c_cameraMatrix = cameraMatrix, c_distCoeffs = distCoeffs;
107 CvMat c_rvec = rvec, c_tvec = tvec;
108 cvFindExtrinsicCameraParams2(&c_objectPoints, &c_imagePoints, &c_cameraMatrix,
109 c_distCoeffs.rows*c_distCoeffs.cols ? &c_distCoeffs : 0,
110 &c_rvec, &c_tvec, useExtrinsicGuess );
111 return true;
112 }
113 else
114 CV_Error(CV_StsBadArg, "The flags argument must be one of CV_ITERATIVE or CV_EPNP");
115 return false;
116}

124 static void project3dPoints(const Mat& points, const Mat& rvec, const Mat& tvec, Mat& modif_points)
125 {
126 modif_points.create(1, points.cols, CV_32FC3);
127 Mat R(3, 3, CV_64FC1);
128 Rodrigues(rvec, R);
129 Mat transformation(3, 4, CV_64F);
130 Mat r = transformation.colRange(0, 3);
131 R.copyTo(r);
132 Mat t = transformation.colRange(3, 4);
133 tvec.copyTo(t);
134 transform(points, modif_points, transformation);
135 }

159 template <typename OpointType, typename IpointType>
160 static void pnpTask(const int curIndex, const vector<char>& pointsMask, const Mat& objectPoints, const Mat& imagePoints,
161 const Parameters& params, vector<int>& inliers, int& bestIndex, Mat& rvec, Mat& tvec,
162 const Mat& rvecInit, const Mat& tvecInit, Mutex& resultsMutex)
163 {
164 Mat modelObjectPoints(1, MIN_POINTS_COUNT, CV_MAKETYPE(DataDepth<OpointType>::value, 3));
165 Mat modelImagePoints(1, MIN_POINTS_COUNT, CV_MAKETYPE(DataDepth<IpointType>::value, 2));
166 for (int i = 0, colIndex = 0; i < (int)pointsMask.size(); i++)
167 {
168 if (pointsMask[i])
169 {
170 Mat colModelImagePoints = modelImagePoints(Rect(colIndex, 0, 1, 1));
171 imagePoints.col(i).copyTo(colModelImagePoints);
172 Mat colModelObjectPoints = modelObjectPoints(Rect(colIndex, 0, 1, 1));
173 objectPoints.col(i).copyTo(colModelObjectPoints);
174 colIndex = colIndex+1;
175 }
176 }
177
178 //filter same 3d points, hang in solvePnP
179 double eps = 1e-10;
180 int num_same_points = 0;
181 for (int i = 0; i < MIN_POINTS_COUNT; i++)
182 for (int j = i + 1; j < MIN_POINTS_COUNT; j++)
183 {
184 if (norm(modelObjectPoints.at<Vec<OpointType,3> >(0, i) - modelObjectPoints.at<Vec<OpointType,3> >(0, j)) < eps)
185 num_same_points++;
186 }
187 if (num_same_points > 0)
188 return;
189
190 Mat localRvec, localTvec;
191 rvecInit.copyTo(localRvec);
192 tvecInit.copyTo(localTvec);
193
194 solvePnP(modelObjectPoints, modelImagePoints, params.camera.intrinsics, params.camera.distortion, localRvec, localTvec,
195 params.useExtrinsicGuess, params.flags);
196
197
198 vector<Point_<OpointType> > projected_points;
199 projected_points.resize(objectPoints.cols);
200 projectPoints(objectPoints, localRvec, localTvec, params.camera.intrinsics, params.camera.distortion, projected_points);
201
202 Mat rotatedPoints;
203 project3dPoints(objectPoints, localRvec, localTvec, rotatedPoints);
204
205 vector<int> localInliers;
206 for (int i = 0; i < objectPoints.cols; i++)
207 {
208 //Although p is a 2D point it needs the same type as the object points to enable the norm calculation
209 Point_<OpointType> p((OpointType)imagePoints.at<Vec<IpointType,2> >(0, i)[0],
210 (OpointType)imagePoints.at<Vec<IpointType,2> >(0, i)[1]);
211 if ((norm(p - projected_points[i]) < params.reprojectionError)
212 && (rotatedPoints.at<Vec<OpointType,3> >(0, i)[2] > 0)) //hack
213 {
214 localInliers.push_back(i);
215 }
216 }
217
218 resultsMutex.lock();
219 if ( (localInliers.size() > inliers.size()) || (localInliers.size() == inliers.size() && inliers.size() > 0 && curIndex > bestIndex))
220 {
221 inliers.clear();
222 inliers.resize(localInliers.size());
223 memcpy(&inliers[0], &localInliers[0], sizeof(int) * localInliers.size());
224 localRvec.copyTo(rvec);
225 localTvec.copyTo(tvec);
226 bestIndex = curIndex;
227 }
228 resultsMutex.unlock();
229 }

231 static void pnpTask(const int curIndex, const vector<char>& pointsMask, const Mat& objectPoints, const Mat& imagePoints,
232 const Parameters& params, vector<int>& inliers, int& bestIndex, Mat& rvec, Mat& tvec,
233 const Mat& rvecInit, const Mat& tvecInit, Mutex& resultsMutex)
234 {
235 CV_Assert(objectPoints.depth() == CV_64F || objectPoints.depth() == CV_32F);
236 CV_Assert(imagePoints.depth() == CV_64F || imagePoints.depth() == CV_32F);
237 const bool objectDoublePrecision = objectPoints.depth() == CV_64F;
238 const bool imageDoublePrecision = imagePoints.depth() == CV_64F;
239 if(objectDoublePrecision)
240 {
241 if(imageDoublePrecision)
242 pnpTask<double, double>(curIndex, pointsMask, objectPoints, imagePoints, params, inliers, bestIndex, rvec, tvec, rvecInit, tvecInit, resultsMutex);
243 else
244 pnpTask<double, float>(curIndex, pointsMask, objectPoints, imagePoints, params, inliers, bestIndex, rvec, tvec, rvecInit, tvecInit, resultsMutex);
245 }
246 else
247 {
248 if(imageDoublePrecision)
249 pnpTask<float, double>(curIndex, pointsMask, objectPoints, imagePoints, params, inliers, bestIndex, rvec, tvec, rvecInit, tvecInit, resultsMutex);
250 else
251 pnpTask<float, float>(curIndex, pointsMask, objectPoints, imagePoints, params, inliers, bestIndex, rvec, tvec, rvecInit, tvecInit, resultsMutex);
252 }

255 class PnPSolver
256 {
257 public:
258 void operator()( const BlockedRange& r ) const
259 {
260 vector<char> pointsMask(objectPoints.cols, 0);
261 for( int i=r.begin(); i!=r.end(); ++i )
262 {
263 memset(&pointsMask[0], 0, objectPoints.cols );
264 memset(&pointsMask[0], 1, MIN_POINTS_COUNT );
265 generateVar(pointsMask, rng_base_seed + i);
266 pnpTask(i, pointsMask, objectPoints, imagePoints, parameters,
267 inliers, bestIndex, rvec, tvec, initRvec, initTvec, syncMutex);
268 if ((int)inliers.size() >= parameters.minInliersCount)
269 {
270#ifdef HAVE_TBB
271 tbb::task::self().cancel_group_execution();
272#else
273 break;
274#endif
275 }
276 }
277 }
278 PnPSolver(const Mat& _objectPoints, const Mat& _imagePoints, const Parameters& _parameters,
279 Mat& _rvec, Mat& _tvec, vector<int>& _inliers, int& _bestIndex, uint64 _rng_base_seed):
280 objectPoints(_objectPoints), imagePoints(_imagePoints), parameters(_parameters),
281 rvec(_rvec), tvec(_tvec), inliers(_inliers), bestIndex(_bestIndex), rng_base_seed(_rng_base_seed)
282 {
283 bestIndex = -1;
284 rvec.copyTo(initRvec);
285 tvec.copyTo(initTvec);
286 }
287 private:
288 PnPSolver& operator=(const PnPSolver&);
289
290 const Mat& objectPoints;
291 const Mat& imagePoints;
292 const Parameters& parameters;
293 Mat &rvec, &tvec;
294 vector<int>& inliers;
295 int& bestIndex;
296 const uint64 rng_base_seed;
297 Mat initRvec, initTvec;
298
299 static Mutex syncMutex;
300
301 void generateVar(vector<char>& mask, uint64 rng_seed) const
302 {
303 RNG generator(rng_seed);
304 int size = (int)mask.size();
305 for (int i = 0; i < size; i++)
306 {
307 int i1 = generator.uniform(0, size);
308 int i2 = generator.uniform(0, size);
309 char curr = mask[i1];
310 mask[i1] = mask[i2];
311 mask[i2] = curr;
312 }
313 }
314 };
315
316 Mutex PnPSolver::syncMutex;
317
318 }

321void cv::solvePnPRansac(InputArray _opoints, InputArray _ipoints,
322 InputArray _cameraMatrix, InputArray _distCoeffs,
323 OutputArray _rvec, OutputArray _tvec, bool useExtrinsicGuess,
324 int iterationsCount, float reprojectionError, int minInliersCount,
325 OutputArray _inliers, int flags)
326{
327 const int _rng_seed = 0;
328 Mat opoints = _opoints.getMat(), ipoints = _ipoints.getMat();
329 Mat cameraMatrix = _cameraMatrix.getMat(), distCoeffs = _distCoeffs.getMat();
330
331 CV_Assert(opoints.isContinuous());
332 CV_Assert(opoints.depth() == CV_32F || opoints.depth() == CV_64F);
333 CV_Assert((opoints.rows == 1 && opoints.channels() == 3) || opoints.cols*opoints.channels() == 3);
334 CV_Assert(ipoints.isContinuous());
335 CV_Assert(ipoints.depth() == CV_32F || ipoints.depth() == CV_64F);
336 CV_Assert((ipoints.rows == 1 && ipoints.channels() == 2) || ipoints.cols*ipoints.channels() == 2);
337
338 Mat rvec, tvec;
339 if( flags != CV_ITERATIVE )
340 useExtrinsicGuess = false;
341
342 if( useExtrinsicGuess )
343 {
344 int rtype = _rvec.type(), ttype = _tvec.type();
345 Size rsize = _rvec.size(), tsize = _tvec.size();
346 CV_Assert( (rtype == CV_32F || rtype == CV_64F) &&
347 (ttype == CV_32F || ttype == CV_64F) );
348 CV_Assert( (rsize == Size(1, 3) || rsize == Size(3, 1)) &&
349 (tsize == Size(1, 3) || tsize == Size(3, 1)) );
350 }
351 else
352 {
353 _rvec.create(3, 1, CV_64F);
354 _tvec.create(3, 1, CV_64F);
355 }
356 rvec = _rvec.getMat();
357 tvec = _tvec.getMat();
358
359 Mat objectPoints = opoints.reshape(3, 1), imagePoints = ipoints.reshape(2, 1);
360
361 if (minInliersCount <= 0)
362 minInliersCount = objectPoints.cols;
363 cv::pnpransac::Parameters params;
364 params.iterationsCount = iterationsCount;
365 params.minInliersCount = minInliersCount;
366 params.reprojectionError = reprojectionError;
367 params.useExtrinsicGuess = useExtrinsicGuess;
368 params.camera.init(cameraMatrix, distCoeffs);
369 params.flags = flags;
370
371 vector<int> localInliers;
372 Mat localRvec, localTvec;
373 rvec.copyTo(localRvec);
374 tvec.copyTo(localTvec);
375 int bestIndex;
376
377 if (objectPoints.cols >= pnpransac::MIN_POINTS_COUNT)
378 {
379 parallel_for(BlockedRange(0,iterationsCount), cv::pnpransac::PnPSolver(objectPoints, imagePoints, params,
380 localRvec, localTvec, localInliers, bestIndex,
381 _rng_seed));
382 }
383
384 if (localInliers.size() >= (size_t)pnpransac::MIN_POINTS_COUNT)
385 {
386 if (flags != CV_P3P)
387 {
388 int i, pointsCount = (int)localInliers.size();
389 Mat inlierObjectPoints(1, pointsCount, CV_MAKE_TYPE(opoints.depth(), 3)), inlierImagePoints(1, pointsCount, CV_MAKE_TYPE(ipoints.depth(), 2));
390 for (i = 0; i < pointsCount; i++)
391 {
392 int index = localInliers[i];
393 Mat colInlierImagePoints = inlierImagePoints(Rect(i, 0, 1, 1));
394 imagePoints.col(index).copyTo(colInlierImagePoints);
395 Mat colInlierObjectPoints = inlierObjectPoints(Rect(i, 0, 1, 1));
396 objectPoints.col(index).copyTo(colInlierObjectPoints);
397 }
398 solvePnP(inlierObjectPoints, inlierImagePoints, params.camera.intrinsics, params.camera.distortion,
399 localRvec, localTvec, params.useExtrinsicGuess, flags);
400 }
401 localRvec.copyTo(rvec);
402 localTvec.copyTo(tvec);
403 if (_inliers.needed())
404 Mat(localInliers).copyTo(_inliers);
405 }
406 else
407 {
408 tvec.setTo(Scalar(0));
409 Mat R = Mat::eye(3, 3, CV_64F);
410 Rodrigues(R, rvec);
411 if( _inliers.needed() )
412 _inliers.release();
413 }
414 return;
415}
