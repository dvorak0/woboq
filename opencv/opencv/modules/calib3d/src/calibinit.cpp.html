106/// Corner info structure
107/** This structure stores information about the chessboard corner.*/
108struct CvCBCorner
109{
110 CvPoint2D32f pt; // Coordinates of the corner
111 int row; // Board row index
112 int count; // Number of neighbor corners
113 struct CvCBCorner* neighbors[4]; // Neighbor corners
114
115 float meanDist(int *_n) const
116 {
117 float sum = 0;
118 int n = 0;
119 for( int i = 0; i < 4; i++ )
120 {
121 if( neighbors[i] )
122 {
123 float dx = neighbors[i]->pt.x - pt.x;
124 float dy = neighbors[i]->pt.y - pt.y;
125 sum += sqrt(dx*dx + dy*dy);
126 n++;
127 }
128 }
129 if(_n)
130 *_n = n;
131 return sum/MAX(n,1);
132 }
133};

571//
572// Checks that each board row and column is pretty much monotonous curve:
573// It analyzes each row and each column of the chessboard as following:
574// for each corner c lying between end points in the same row/column it checks that
575// the point projection to the line segment (a,b) is lying between projections
576// of the neighbor corners in the same row/column.
577//
578// This function has been created as temporary workaround for the bug in current implementation
579// of cvFindChessboardCornes that produces absolutely unordered sets of corners.
580//
581
582static int
583icvCheckBoardMonotony( CvPoint2D32f* corners, CvSize pattern_size )
584{
585 int i, j, k;
586
587 for( k = 0; k < 2; k++ )
588 {
589 for( i = 0; i < (k == 0 ? pattern_size.height : pattern_size.width); i++ )
590 {
591 CvPoint2D32f a = k == 0 ? corners[i*pattern_size.width] : corners[i];
592 CvPoint2D32f b = k == 0 ? corners[(i+1)*pattern_size.width-1] :
593 corners[(pattern_size.height-1)*pattern_size.width + i];
594 float prevt = 0, dx0 = b.x - a.x, dy0 = b.y - a.y;
595 if( fabs(dx0) + fabs(dy0) < FLT_EPSILON )
596 return 0;
597 for( j = 1; j < (k == 0 ? pattern_size.width : pattern_size.height) - 1; j++ )
598 {
599 CvPoint2D32f c = k == 0 ? corners[i*pattern_size.width + j] :
600 corners[j*pattern_size.width + i];
601 float t = ((c.x - a.x)*dx0 + (c.y - a.y)*dy0)/(dx0*dx0 + dy0*dy0);
602 if( t < prevt || t > 1 )
603 return 0;
604 prevt = t;
605 }
606 }
607 }
608
609 return 1;
610}

612//
613// order a group of connected quads
614// order of corners:
615// 0 is top left
616// clockwise from there
617// note: "top left" is nominal, depends on initial ordering of starting quad
618// but all other quads are ordered consistently
619//
620// can change the number of quads in the group
621// can add quads, so we need to have quad/corner arrays passed in
622//

625icvOrderFoundConnectedQuads( int quad_count, CvCBQuad **quads,

632 // first find an interior quad
633 CvCBQuad *start = NULL;
634 for (int i=0; i<quad_count; i++)
635 {
636 if (quads[i]->count == 4)
637 {
638 start = quads[i];
639 break;
640 }
641 }
642
643 if (start == NULL)
644 return 0; // no 4-connected quad
645
646 // start with first one, assign rows/cols
647 int row_min = 0, col_min = 0, row_max=0, col_max = 0;
648
649 std::map<int, int> col_hist;
650 std::map<int, int> row_hist;
651
652 cvSeqPush(stack, &start);
653 start->row = 0;
654 start->col = 0;
655 start->ordered = true;
656
657 // Recursively order the quads so that all position numbers (e.g.,
658 // 0,1,2,3) are in the at the same relative corner (e.g., lower right).
659
660 while( stack->total )
661 {
662 CvCBQuad* q;
663 cvSeqPop( stack, &q );
664 int col = q->col;
665 int row = q->row;
666 col_hist[col]++;
667 row_hist[row]++;
668
669 // check min/max
670 if (row > row_max) row_max = row;
671 if (row < row_min) row_min = row;
672 if (col > col_max) col_max = col;
673 if (col < col_min) col_min = col;
674
675 for(int i = 0; i < 4; i++ )
676 {
677 CvCBQuad *neighbor = q->neighbors[i];
678 switch(i) // adjust col, row for this quad
679 { // start at top left, go clockwise
680 case 0:
681 row--; col--; break;
682 case 1:
683 col += 2; break;
684 case 2:
685 row += 2; break;
686 case 3:
687 col -= 2; break;
688 }
689
690 // just do inside quads
691 if (neighbor && neighbor->ordered == false && neighbor->count == 4)
692 {
693 PRINTF("col: %d row: %d\n", col, row);
694 icvOrderQuad(neighbor, q->corners[i], (i+2)%4); // set in order
695 neighbor->ordered = true;
696 neighbor->row = row;
697 neighbor->col = col;
698 cvSeqPush( stack, &neighbor );
699 }
700 }
701 }
702
703 for (int i=col_min; i<=col_max; i++)
704 PRINTF("HIST[%d] = %d\n", i, col_hist[i]);
705
706 // analyze inner quad structure
707 int w = pattern_size.width - 1;
708 int h = pattern_size.height - 1;
709 int drow = row_max - row_min + 1;
710 int dcol = col_max - col_min + 1;
711
712 // normalize pattern and found quad indices
713 if ((w > h && dcol < drow) ||
714 (w < h && drow < dcol))
715 {
716 h = pattern_size.width - 1;
717 w = pattern_size.height - 1;
718 }
719
720 PRINTF("Size: %dx%d Pattern: %dx%d\n", dcol, drow, w, h);
721
722 // check if there are enough inner quads
723 if (dcol < w || drow < h) // found enough inner quads?
724 {
725 PRINTF("Too few inner quad rows/cols\n");
726 return 0; // no, return
727 }
728#ifdef ENABLE_TRIM_COL_ROW
729 // too many columns, not very common
730 if (dcol == w+1) // too many, trim
731 {
732 PRINTF("Trimming cols\n");
733 if (col_hist[col_max] > col_hist[col_min])
734 {
735 PRINTF("Trimming left col\n");
736 quad_count = icvTrimCol(quads,quad_count,col_min,-1);
737 }
738 else
739 {
740 PRINTF("Trimming right col\n");
741 quad_count = icvTrimCol(quads,quad_count,col_max,+1);
742 }
743 }
744
745 // too many rows, not very common
746 if (drow == h+1) // too many, trim
747 {
748 PRINTF("Trimming rows\n");
749 if (row_hist[row_max] > row_hist[row_min])
750 {
751 PRINTF("Trimming top row\n");
752 quad_count = icvTrimRow(quads,quad_count,row_min,-1);
753 }
754 else
755 {
756 PRINTF("Trimming bottom row\n");
757 quad_count = icvTrimRow(quads,quad_count,row_max,+1);
758 }
759 }
760#endif
761
762 // check edges of inner quads
763 // if there is an outer quad missing, fill it in
764 // first order all inner quads
765 int found = 0;
766 for (int i=0; i<quad_count; i++)
767 {
768 if (quads[i]->count == 4)
769 { // ok, look at neighbors
770 int col = quads[i]->col;
771 int row = quads[i]->row;
772 for (int j=0; j<4; j++)
773 {
774 switch(j) // adjust col, row for this quad
775 { // start at top left, go clockwise
776 case 0:
777 row--; col--; break;
778 case 1:
779 col += 2; break;
780 case 2:
781 row += 2; break;
782 case 3:
783 col -= 2; break;
784 }
785 CvCBQuad *neighbor = quads[i]->neighbors[j];
786 if (neighbor && !neighbor->ordered && // is it an inner quad?
787 col <= col_max && col >= col_min &&
788 row <= row_max && row >= row_min)
789 {
790 // if so, set in order
791 PRINTF("Adding inner: col: %d row: %d\n", col, row);
792 found++;
793 icvOrderQuad(neighbor, quads[i]->corners[j], (j+2)%4);
794 neighbor->ordered = true;
795 neighbor->row = row;
796 neighbor->col = col;
797 }
798 }
799 }
800 }
801
802 // if we have found inner quads, add corresponding outer quads,
803 // which are missing
804 if (found > 0)
805 {
806 PRINTF("Found %d inner quads not connected to outer quads, repairing\n", found);
807 for (int i=0; i<quad_count && *all_count < max_quad_buf_size; i++)
808 {
809 if (quads[i]->count < 4 && quads[i]->ordered)
810 {
811 int added = icvAddOuterQuad(quads[i],quads,quad_count,all_quads,*all_count,corners, max_quad_buf_size);
812 *all_count += added;
813 quad_count += added;
814 }
815 }
816
817 if (*all_count >= max_quad_buf_size)
818 return 0;
819 }
820
821
822 // final trimming of outer quads
823 if (dcol == w && drow == h) // found correct inner quads
824 {
825 PRINTF("Inner bounds ok, check outer quads\n");
826 int rcount = quad_count;
827 for (int i=quad_count-1; i>=0; i--) // eliminate any quad not connected to
828 // an ordered quad
829 {
830 if (quads[i]->ordered == false)
831 {
832 bool outer = false;
833 for (int j=0; j<4; j++) // any neighbors that are ordered?
834 {
835 if (quads[i]->neighbors[j] && quads[i]->neighbors[j]->ordered)
836 outer = true;
837 }
838 if (!outer) // not an outer quad, eliminate
839 {
840 PRINTF("Removing quad %d\n", i);
841 icvRemoveQuadFromGroup(quads,rcount,quads[i]);
842 rcount--;
843 }
844 }
845
846 }
847 return rcount;
848 }
849
850 return 0;
851}

854// add an outer quad
855// looks for the neighbor of <quad> that isn't present,
856// tries to add it in.
857// <quad> is ordered
858
859static int
860icvAddOuterQuad( CvCBQuad *quad, CvCBQuad **quads, int quad_count,
861 CvCBQuad **all_quads, int all_count, CvCBCorner **corners, int max_quad_buf_size )
862
863{
864 int added = 0;
865 for (int i=0; i<4 && all_count < max_quad_buf_size; i++) // find no-neighbor corners
866 {
867 if (!quad->neighbors[i]) // ok, create and add neighbor
868 {
869 int j = (i+2)%4;
870 PRINTF("Adding quad as neighbor 2\n");
871 CvCBQuad *q = &(*all_quads)[all_count];
872 memset( q, 0, sizeof(*q) );
873 added++;
874 quads[quad_count] = q;
875 quad_count++;
876
877 // set neighbor and group id
878 quad->neighbors[i] = q;
879 quad->count += 1;
880 q->neighbors[j] = quad;
881 q->group_idx = quad->group_idx;
882 q->count = 1; // number of neighbors
883 q->ordered = false;
884 q->edge_len = quad->edge_len;
885
886 // make corners of new quad
887 // same as neighbor quad, but offset
888 CvPoint2D32f pt = quad->corners[i]->pt;
889 CvCBCorner* corner;
890 float dx = pt.x - quad->corners[j]->pt.x;
891 float dy = pt.y - quad->corners[j]->pt.y;
892 for (int k=0; k<4; k++)
893 {
894 corner = &(*corners)[all_count*4+k];
895 pt = quad->corners[k]->pt;
896 memset( corner, 0, sizeof(*corner) );
897 corner->pt = pt;
898 q->corners[k] = corner;
899 corner->pt.x += dx;
900 corner->pt.y += dy;
901 }
902 // have to set exact corner
903 q->corners[j] = quad->corners[i];
904
905 // now find other neighbor and add it, if possible
906 if (quad->neighbors[(i+3)%4] &&
907 quad->neighbors[(i+3)%4]->ordered &&
908 quad->neighbors[(i+3)%4]->neighbors[i] &&
909 quad->neighbors[(i+3)%4]->neighbors[i]->ordered )
910 {
911 CvCBQuad *qn = quad->neighbors[(i+3)%4]->neighbors[i];
912 q->count = 2;
913 q->neighbors[(j+1)%4] = qn;
914 qn->neighbors[(i+1)%4] = q;
915 qn->count += 1;
916 // have to set exact corner
917 q->corners[(j+1)%4] = qn->corners[(i+1)%4];
918 }
919
920 all_count++;
921 }
922 }
923 return added;
924}

1064//
1065// put quad into correct order, where <corner> has value <common>
1066//
1067
1068static void
1069icvOrderQuad(CvCBQuad *quad, CvCBCorner *corner, int common)
1070{
1071 // find the corner
1072 int tc;
1073 for (tc=0; tc<4; tc++)
1074 if (quad->corners[tc]->pt.x == corner->pt.x &&
1075 quad->corners[tc]->pt.y == corner->pt.y)
1076 break;
1077
1078 // set corner order
1079 // shift
1080 while (tc != common)
1081 {
1082 // shift by one
1083 CvCBCorner *tempc;
1084 CvCBQuad *tempq;
1085 tempc = quad->corners[3];
1086 tempq = quad->neighbors[3];
1087 for (int i=3; i>0; i--)
1088 {
1089 quad->corners[i] = quad->corners[i-1];
1090 quad->neighbors[i] = quad->neighbors[i-1];
1091 }
1092 quad->corners[0] = tempc;
1093 quad->neighbors[0] = tempq;
1094 tc++;
1095 tc = tc%4;
1096 }
1097}

1100// if we found too many connect quads, remove those which probably do not belong.
1101static int
1102icvCleanFoundConnectedQuads( int quad_count, CvCBQuad **quad_group, CvSize pattern_size )
1103{
1104 CvPoint2D32f center = {0,0};
1105 int i, j, k;
1106 // number of quads this pattern should contain
1107 int count = ((pattern_size.width + 1)*(pattern_size.height + 1) + 1)/2;
1108
1109 // if we have more quadrangles than we should,
1110 // try to eliminate duplicates or ones which don't belong to the pattern rectangle...
1111 if( quad_count <= count )
1112 return quad_count;
1113
1114 // create an array of quadrangle centers
1115 cv::AutoBuffer<CvPoint2D32f> centers( quad_count );
1116 cv::Ptr<CvMemStorage> temp_storage = cvCreateMemStorage(0);
1117
1118 for( i = 0; i < quad_count; i++ )
1119 {
1120 CvPoint2D32f ci = {0,0};
1121 CvCBQuad* q = quad_group[i];
1122
1123 for( j = 0; j < 4; j++ )
1124 {
1125 CvPoint2D32f pt = q->corners[j]->pt;
1126 ci.x += pt.x;
1127 ci.y += pt.y;
1128 }
1129
1130 ci.x *= 0.25f;
1131 ci.y *= 0.25f;
1132
1133 centers[i] = ci;
1134 center.x += ci.x;
1135 center.y += ci.y;
1136 }
1137 center.x /= quad_count;
1138 center.y /= quad_count;
1139
1140 // If we still have more quadrangles than we should,
1141 // we try to eliminate bad ones based on minimizing the bounding box.
1142 // We iteratively remove the point which reduces the size of
1143 // the bounding box of the blobs the most
1144 // (since we want the rectangle to be as small as possible)
1145 // remove the quadrange that causes the biggest reduction
1146 // in pattern size until we have the correct number
1147 for( ; quad_count > count; quad_count-- )
1148 {
1149 double min_box_area = DBL_MAX;
1150 int skip, min_box_area_index = -1;
1151 CvCBQuad *q0, *q;
1152
1153 // For each point, calculate box area without that point
1154 for( skip = 0; skip < quad_count; skip++ )
1155 {
1156 // get bounding rectangle
1157 CvPoint2D32f temp = centers[skip]; // temporarily make index 'skip' the same as
1158 centers[skip] = center; // pattern center (so it is not counted for convex hull)
1159 CvMat pointMat = cvMat(1, quad_count, CV_32FC2, centers);
1160 CvSeq *hull = cvConvexHull2( &pointMat, temp_storage, CV_CLOCKWISE, 1 );
1161 centers[skip] = temp;
1162 double hull_area = fabs(cvContourArea(hull, CV_WHOLE_SEQ));
1163
1164 // remember smallest box area
1165 if( hull_area < min_box_area )
1166 {
1167 min_box_area = hull_area;
1168 min_box_area_index = skip;
1169 }
1170 cvClearMemStorage( temp_storage );
1171 }
1172
1173 q0 = quad_group[min_box_area_index];
1174
1175 // remove any references to this quad as a neighbor
1176 for( i = 0; i < quad_count; i++ )
1177 {
1178 q = quad_group[i];
1179 for( j = 0; j < 4; j++ )
1180 {
1181 if( q->neighbors[j] == q0 )
1182 {
1183 q->neighbors[j] = 0;
1184 q->count--;
1185 for( k = 0; k < 4; k++ )
1186 if( q0->neighbors[k] == q )
1187 {
1188 q0->neighbors[k] = 0;
1189 q0->count--;
1190 break;
1191 }
1192 break;
1193 }
1194 }
1195 }
1196
1197 // remove the quad
1198 quad_count--;
1199 quad_group[min_box_area_index] = quad_group[quad_count];
1200 centers[min_box_area_index] = centers[quad_count];
1201 }
1202
1203 return quad_count;
1204}

1207
1208static int
1209icvFindConnectedQuads( CvCBQuad *quad, int quad_count, CvCBQuad **out_group,
1210 int group_idx, CvMemStorage* storage )
1211{
1212 cv::Ptr<CvMemStorage> temp_storage = cvCreateChildMemStorage( storage );
1213 CvSeq* stack = cvCreateSeq( 0, sizeof(*stack), sizeof(void*), temp_storage );
1214 int i, count = 0;
1215
1216 // Scan the array for a first unlabeled quad
1217 for( i = 0; i < quad_count; i++ )
1218 {
1219 if( quad[i].count > 0 && quad[i].group_idx < 0)
1220 break;
1221 }
1222
1223 // Recursively find a group of connected quads starting from the seed quad[i]
1224 if( i < quad_count )
1225 {
1226 CvCBQuad* q = &quad[i];
1227 cvSeqPush( stack, &q );
1228 out_group[count++] = q;
1229 q->group_idx = group_idx;
1230 q->ordered = false;
1231
1232 while( stack->total )
1233 {
1234 cvSeqPop( stack, &q );
1235 for( i = 0; i < 4; i++ )
1236 {
1237 CvCBQuad *neighbor = q->neighbors[i];
1238 if( neighbor && neighbor->count > 0 && neighbor->group_idx < 0 )
1239 {
1240 cvSeqPush( stack, &neighbor );
1241 out_group[count++] = neighbor;
1242 neighbor->group_idx = group_idx;
1243 neighbor->ordered = false;
1244 }
1245 }
1246 }
1247 }
1248
1249 return count;
1250}

1254
1255static int
1256icvCheckQuadGroup( CvCBQuad **quad_group, int quad_count,
1257 CvCBCorner **out_corners, CvSize pattern_size )
1258{
1259 const int ROW1 = 1000000;
1260 const int ROW2 = 2000000;
1261 const int ROW_ = 3000000;
1262 int result = 0;
1263 int i, out_corner_count = 0, corner_count = 0;
1264 std::vector<CvCBCorner*> corners(quad_count*4);
1265
1266 int j, k, kk;
1267 int width = 0, height = 0;
1268 int hist[5] = {0,0,0,0,0};
1269 CvCBCorner* first = 0, *first2 = 0, *right, *cur, *below, *c;
1270
1271 // build dual graph, which vertices are internal quad corners
1272 // and two vertices are connected iff they lie on the same quad edge
1273 for( i = 0; i < quad_count; i++ )
1274 {
1275 CvCBQuad* q = quad_group[i];
1276 /*CvScalar color = q->count == 0 ? cvScalar(0,255,255) :
1277 q->count == 1 ? cvScalar(0,0,255) :
1278 q->count == 2 ? cvScalar(0,255,0) :
1279 q->count == 3 ? cvScalar(255,255,0) :
1280 cvScalar(255,0,0);*/
1281
1282 for( j = 0; j < 4; j++ )
1283 {
1284 //cvLine( debug_img, cvPointFrom32f(q->corners[j]->pt), cvPointFrom32f(q->corners[(j+1)&3]->pt), color, 1, CV_AA, 0 );
1285 if( q->neighbors[j] )
1286 {
1287 CvCBCorner *a = q->corners[j], *b = q->corners[(j+1)&3];
1288 // mark internal corners that belong to:
1289 // - a quad with a single neighbor - with ROW1,
1290 // - a quad with two neighbors - with ROW2
1291 // make the rest of internal corners with ROW_
1292 int row_flag = q->count == 1 ? ROW1 : q->count == 2 ? ROW2 : ROW_;
1293
1294 if( a->row == 0 )
1295 {
1296 corners[corner_count++] = a;
1297 a->row = row_flag;
1298 }
1299 else if( a->row > row_flag )
1300 a->row = row_flag;
1301
1302 if( q->neighbors[(j+1)&3] )
1303 {
1304 if( a->count >= 4 || b->count >= 4 )
1305 goto finalize;
1306 for( k = 0; k < 4; k++ )
1307 {
1308 if( a->neighbors[k] == b )
1309 goto finalize;
1310 if( b->neighbors[k] == a )
1311 goto finalize;
1312 }
1313 a->neighbors[a->count++] = b;
1314 b->neighbors[b->count++] = a;
1315 }
1316 }
1317 }
1318 }
1319
1320 if( corner_count != pattern_size.width*pattern_size.height )
1321 goto finalize;
1322
1323 for( i = 0; i < corner_count; i++ )
1324 {
1325 int n = corners[i]->count;
1326 assert( 0 <= n && n <= 4 );
1327 hist[n]++;
1328 if( !first && n == 2 )
1329 {
1330 if( corners[i]->row == ROW1 )
1331 first = corners[i];
1332 else if( !first2 && corners[i]->row == ROW2 )
1333 first2 = corners[i];
1334 }
1335 }
1336
1337 // start with a corner that belongs to a quad with a signle neighbor.
1338 // if we do not have such, start with a corner of a quad with two neighbors.
1339 if( !first )
1340 first = first2;
1341
1342 if( !first || hist[0] != 0 || hist[1] != 0 || hist[2] != 4 ||
1343 hist[3] != (pattern_size.width + pattern_size.height)*2 - 8 )
1344 goto finalize;
1345
1346 cur = first;
1347 right = below = 0;
1348 out_corners[out_corner_count++] = cur;
1349
1350 for( k = 0; k < 4; k++ )
1351 {
1352 c = cur->neighbors[k];
1353 if( c )
1354 {
1355 if( !right )
1356 right = c;
1357 else if( !below )
1358 below = c;
1359 }
1360 }
1361
1362 if( !right || (right->count != 2 && right->count != 3) ||
1363 !below || (below->count != 2 && below->count != 3) )
1364 goto finalize;
1365
1366 cur->row = 0;
1367 //cvCircle( debug_img, cvPointFrom32f(cur->pt), 3, cvScalar(0,255,0), -1, 8, 0 );
1368
1369 first = below; // remember the first corner in the next row
1370 // find and store the first row (or column)
1371 for(j=1;;j++)
1372 {
1373 right->row = 0;
1374 out_corners[out_corner_count++] = right;
1375 //cvCircle( debug_img, cvPointFrom32f(right->pt), 3, cvScalar(0,255-j*10,0), -1, 8, 0 );
1376 if( right->count == 2 )
1377 break;
1378 if( right->count != 3 || out_corner_count >= MAX(pattern_size.width,pattern_size.height) )
1379 goto finalize;
1380 cur = right;
1381 for( k = 0; k < 4; k++ )
1382 {
1383 c = cur->neighbors[k];
1384 if( c && c->row > 0 )
1385 {
1386 for( kk = 0; kk < 4; kk++ )
1387 {
1388 if( c->neighbors[kk] == below )
1389 break;
1390 }
1391 if( kk < 4 )
1392 below = c;
1393 else
1394 right = c;
1395 }
1396 }
1397 }
1398
1399 width = out_corner_count;
1400 if( width == pattern_size.width )
1401 height = pattern_size.height;
1402 else if( width == pattern_size.height )
1403 height = pattern_size.width;
1404 else
1405 goto finalize;
1406
1407 // find and store all the other rows
1408 for( i = 1; ; i++ )
1409 {
1410 if( !first )
1411 break;
1412 cur = first;
1413 first = 0;
1414 for( j = 0;; j++ )
1415 {
1416 cur->row = i;
1417 out_corners[out_corner_count++] = cur;
1418 //cvCircle( debug_img, cvPointFrom32f(cur->pt), 3, cvScalar(0,0,255-j*10), -1, 8, 0 );
1419 if( cur->count == 2 + (i < height-1) && j > 0 )
1420 break;
1421
1422 right = 0;
1423
1424 // find a neighbor that has not been processed yet
1425 // and that has a neighbor from the previous row
1426 for( k = 0; k < 4; k++ )
1427 {
1428 c = cur->neighbors[k];
1429 if( c && c->row > i )
1430 {
1431 for( kk = 0; kk < 4; kk++ )
1432 {
1433 if( c->neighbors[kk] && c->neighbors[kk]->row == i-1 )
1434 break;
1435 }
1436 if( kk < 4 )
1437 {
1438 right = c;
1439 if( j > 0 )
1440 break;
1441 }
1442 else if( j == 0 )
1443 first = c;
1444 }
1445 }
1446 if( !right )
1447 goto finalize;
1448 cur = right;
1449 }
1450
1451 if( j != width - 1 )
1452 goto finalize;
1453 }
1454
1455 if( out_corner_count != corner_count )
1456 goto finalize;
1457
1458 // check if we need to transpose the board
1459 if( width != pattern_size.width )
1460 {
1461 CV_SWAP( width, height, k );
1462
1463 memcpy( &corners[0], out_corners, corner_count*sizeof(corners[0]) );
1464 for( i = 0; i < height; i++ )
1465 for( j = 0; j < width; j++ )
1466 out_corners[i*width + j] = corners[j*height + i];
1467 }
1468
1469 // check if we need to revert the order in each row
1470 {
1471 CvPoint2D32f p0 = out_corners[0]->pt, p1 = out_corners[pattern_size.width-1]->pt,
1472 p2 = out_corners[pattern_size.width]->pt;
1473 if( (p1.x - p0.x)*(p2.y - p1.y) - (p1.y - p0.y)*(p2.x - p1.x) < 0 )
1474 {
1475 if( width % 2 == 0 )
1476 {
1477 for( i = 0; i < height; i++ )
1478 for( j = 0; j < width/2; j++ )
1479 CV_SWAP( out_corners[i*width+j], out_corners[i*width+width-j-1], c );
1480 }
1481 else
1482 {
1483 for( j = 0; j < width; j++ )
1484 for( i = 0; i < height/2; i++ )
1485 CV_SWAP( out_corners[i*width+j], out_corners[(height - i - 1)*width+j], c );
1486 }
1487 }
1488 }
1489
1490 result = corner_count;
1491
1492finalize:
1493
1494 if( result <= 0 )
1495 {
1496 corner_count = MIN( corner_count, pattern_size.width*pattern_size.height );
1497 for( i = 0; i < corner_count; i++ )
1498 out_corners[i] = corners[i];
1499 result = -corner_count;
1500
1501 if (result == -pattern_size.width*pattern_size.height)
1502 result = -result;
1503 }
1504
1505 return result;
1506}

1512
1513static void icvFindQuadNeighbors( CvCBQuad *quads, int quad_count )
1514{
1515 const float thresh_scale = 1.f;
1516 int idx, i, k, j;
1517 float dx, dy, dist;
1518
1519 // find quad neighbors
1520 for( idx = 0; idx < quad_count; idx++ )
1521 {
1522 CvCBQuad* cur_quad = &quads[idx];
1523
1524 // choose the points of the current quadrangle that are close to
1525 // some points of the other quadrangles
1526 // (it can happen for split corners (due to dilation) of the
1527 // checker board). Search only in other quadrangles!
1528
1529 // for each corner of this quadrangle
1530 for( i = 0; i < 4; i++ )
1531 {
1532 CvPoint2D32f pt;
1533 float min_dist = FLT_MAX;
1534 int closest_corner_idx = -1;
1535 CvCBQuad *closest_quad = 0;
1536 CvCBCorner *closest_corner = 0;
1537
1538 if( cur_quad->neighbors[i] )
1539 continue;
1540
1541 pt = cur_quad->corners[i]->pt;
1542
1543 // find the closest corner in all other quadrangles
1544 for( k = 0; k < quad_count; k++ )
1545 {
1546 if( k == idx )
1547 continue;
1548
1549 for( j = 0; j < 4; j++ )
1550 {
1551 if( quads[k].neighbors[j] )
1552 continue;
1553
1554 dx = pt.x - quads[k].corners[j]->pt.x;
1555 dy = pt.y - quads[k].corners[j]->pt.y;
1556 dist = dx * dx + dy * dy;
1557
1558 if( dist < min_dist &&
1559 dist <= cur_quad->edge_len*thresh_scale &&
1560 dist <= quads[k].edge_len*thresh_scale )
1561 {
1562 // check edge lengths, make sure they're compatible
1563 // edges that are different by more than 1:4 are rejected
1564 float ediff = cur_quad->edge_len - quads[k].edge_len;
1565 if (ediff > 32*cur_quad->edge_len ||
1566 ediff > 32*quads[k].edge_len)
1567 {
1568 PRINTF("Incompatible edge lengths\n");
1569 continue;
1570 }
1571 closest_corner_idx = j;
1572 closest_quad = &quads[k];
1573 min_dist = dist;
1574 }
1575 }
1576 }
1577
1578 // we found a matching corner point?
1579 if( closest_corner_idx >= 0 && min_dist < FLT_MAX )
1580 {
1581 // If another point from our current quad is closer to the found corner
1582 // than the current one, then we don't count this one after all.
1583 // This is necessary to support small squares where otherwise the wrong
1584 // corner will get matched to closest_quad;
1585 closest_corner = closest_quad->corners[closest_corner_idx];
1586
1587 for( j = 0; j < 4; j++ )
1588 {
1589 if( cur_quad->neighbors[j] == closest_quad )
1590 break;
1591
1592 dx = closest_corner->pt.x - cur_quad->corners[j]->pt.x;
1593 dy = closest_corner->pt.y - cur_quad->corners[j]->pt.y;
1594
1595 if( dx * dx + dy * dy < min_dist )
1596 break;
1597 }
1598
1599 if( j < 4 || cur_quad->count >= 4 || closest_quad->count >= 4 )
1600 continue;
1601
1602 // Check that each corner is a neighbor of different quads
1603 for( j = 0; j < closest_quad->count; j++ )
1604 {
1605 if( closest_quad->neighbors[j] == cur_quad )
1606 break;
1607 }
1608 if( j < closest_quad->count )
1609 continue;
1610
1611 // check whether the closest corner to closest_corner
1612 // is different from cur_quad->corners[i]->pt
1613 for( k = 0; k < quad_count; k++ )
1614 {
1615 CvCBQuad* q = &quads[k];
1616 if( k == idx || q == closest_quad )
1617 continue;
1618
1619 for( j = 0; j < 4; j++ )
1620 if( !q->neighbors[j] )
1621 {
1622 dx = closest_corner->pt.x - q->corners[j]->pt.x;
1623 dy = closest_corner->pt.y - q->corners[j]->pt.y;
1624 dist = dx*dx + dy*dy;
1625 if( dist < min_dist )
1626 break;
1627 }
1628 if( j < 4 )
1629 break;
1630 }
1631
1632 if( k < quad_count )
1633 continue;
1634
1635 closest_corner->pt.x = (pt.x + closest_corner->pt.x) * 0.5f;
1636 closest_corner->pt.y = (pt.y + closest_corner->pt.y) * 0.5f;
1637
1638 // We've found one more corner - remember it
1639 cur_quad->count++;
1640 cur_quad->neighbors[i] = closest_quad;
1641 cur_quad->corners[i] = closest_corner;
1642
1643 closest_quad->count++;
1644 closest_quad->neighbors[closest_corner_idx] = cur_quad;
1645 }
1646 }
1647 }
1648}

1652// returns corners in clockwise order
1653// corners don't necessarily start at same position on quad (e.g.,
1654// top left corner)
1655
1656static int
1657icvGenerateQuads( CvCBQuad **out_quads, CvCBCorner **out_corners,
1658 CvMemStorage *storage, CvMat *image, int flags, int *max_quad_buf_size )
1659{
1660 int quad_count = 0;
1661 cv::Ptr<CvMemStorage> temp_storage;
1662
1663 if( out_quads )
1664 *out_quads = 0;
1665
1666 if( out_corners )
1667 *out_corners = 0;
1668
1669 CvSeq *src_contour = 0;
1670 CvSeq *root;
1671 CvContourEx* board = 0;
1672 CvContourScanner scanner;
1673 int i, idx, min_size;
1674
1675 CV_Assert( out_corners && out_quads );
1676
1677 // empiric bound for minimal allowed perimeter for squares
1678 min_size = 25; //cvRound( image->cols * image->rows * .03 * 0.01 * 0.92 );
1679
1680 // create temporary storage for contours and the sequence of pointers to found quadrangles
1681 temp_storage = cvCreateChildMemStorage( storage );
1682 root = cvCreateSeq( 0, sizeof(CvSeq), sizeof(CvSeq*), temp_storage );
1683
1684 // initialize contour retrieving routine
1685 scanner = cvStartFindContours( image, temp_storage, sizeof(CvContourEx),
1686 CV_RETR_CCOMP, CV_CHAIN_APPROX_SIMPLE );
1687
1688 // get all the contours one by one
1689 while( (src_contour = cvFindNextContour( scanner )) != 0 )
1690 {
1691 CvSeq *dst_contour = 0;
1692 CvRect rect = ((CvContour*)src_contour)->rect;
1693
1694 // reject contours with too small perimeter
1695 if( CV_IS_SEQ_HOLE(src_contour) && rect.width*rect.height >= min_size )
1696 {
1697 const int min_approx_level = 1, max_approx_level = MAX_CONTOUR_APPROX;
1698 int approx_level;
1699 for( approx_level = min_approx_level; approx_level <= max_approx_level; approx_level++ )
1700 {
1701 dst_contour = cvApproxPoly( src_contour, sizeof(CvContour), temp_storage,
1702 CV_POLY_APPROX_DP, (float)approx_level );
1703 if( dst_contour->total == 4 )
1704 break;
1705
1706 // we call this again on its own output, because sometimes
1707 // cvApproxPoly() does not simplify as much as it should.
1708 dst_contour = cvApproxPoly( dst_contour, sizeof(CvContour), temp_storage,
1709 CV_POLY_APPROX_DP, (float)approx_level );
1710
1711 if( dst_contour->total == 4 )
1712 break;
1713 }
1714
1715 // reject non-quadrangles
1716 if( dst_contour->total == 4 && cvCheckContourConvexity(dst_contour) )
1717 {
1718 CvPoint pt[4];
1719 double d1, d2, p = cvContourPerimeter(dst_contour);
1720 double area = fabs(cvContourArea(dst_contour, CV_WHOLE_SEQ));
1721 double dx, dy;
1722
1723 for( i = 0; i < 4; i++ )
1724 pt[i] = *(CvPoint*)cvGetSeqElem(dst_contour, i);
1725
1726 dx = pt[0].x - pt[2].x;
1727 dy = pt[0].y - pt[2].y;
1728 d1 = sqrt(dx*dx + dy*dy);
1729
1730 dx = pt[1].x - pt[3].x;
1731 dy = pt[1].y - pt[3].y;
1732 d2 = sqrt(dx*dx + dy*dy);
1733
1734 // philipg. Only accept those quadrangles which are more square
1735 // than rectangular and which are big enough
1736 double d3, d4;
1737 dx = pt[0].x - pt[1].x;
1738 dy = pt[0].y - pt[1].y;
1739 d3 = sqrt(dx*dx + dy*dy);
1740 dx = pt[1].x - pt[2].x;
1741 dy = pt[1].y - pt[2].y;
1742 d4 = sqrt(dx*dx + dy*dy);
1743 if( !(flags & CV_CALIB_CB_FILTER_QUADS) ||
1744 (d3*4 > d4 && d4*4 > d3 && d3*d4 < area*1.5 && area > min_size &&
1745 d1 >= 0.15 * p && d2 >= 0.15 * p) )
1746 {
1747 CvContourEx* parent = (CvContourEx*)(src_contour->v_prev);
1748 parent->counter++;
1749 if( !board || board->counter < parent->counter )
1750 board = parent;
1751 dst_contour->v_prev = (CvSeq*)parent;
1752 //for( i = 0; i < 4; i++ ) cvLine( debug_img, pt[i], pt[(i+1)&3], cvScalar(200,255,255), 1, CV_AA, 0 );
1753 cvSeqPush( root, &dst_contour );
1754 }
1755 }
1756 }
1757 }
1758
1759 // finish contour retrieving
1760 cvEndFindContours( &scanner );
1761
1762 // allocate quad & corner buffers
1763 *max_quad_buf_size = MAX(1, (root->total+root->total / 2)) * 2;
1764 *out_quads = (CvCBQuad*)cvAlloc(*max_quad_buf_size * sizeof((*out_quads)[0]));
1765 *out_corners = (CvCBCorner*)cvAlloc(*max_quad_buf_size * 4 * sizeof((*out_corners)[0]));
1766
1767 // Create array of quads structures
1768 for( idx = 0; idx < root->total; idx++ )
1769 {
1770 CvCBQuad* q = &(*out_quads)[quad_count];
1771 src_contour = *(CvSeq**)cvGetSeqElem( root, idx );
1772 if( (flags & CV_CALIB_CB_FILTER_QUADS) && src_contour->v_prev != (CvSeq*)board )
1773 continue;
1774
1775 // reset group ID
1776 memset( q, 0, sizeof(*q) );
1777 q->group_idx = -1;
1778 assert( src_contour->total == 4 );
1779 for( i = 0; i < 4; i++ )
1780 {
1781 CvPoint2D32f pt = cvPointTo32f(*(CvPoint*)cvGetSeqElem(src_contour, i));
1782 CvCBCorner* corner = &(*out_corners)[quad_count*4 + i];
1783
1784 memset( corner, 0, sizeof(*corner) );
1785 corner->pt = pt;
1786 q->corners[i] = corner;
1787 }
1788 q->edge_len = FLT_MAX;
1789 for( i = 0; i < 4; i++ )
1790 {
1791 float dx = q->corners[i]->pt.x - q->corners[(i+1)&3]->pt.x;
1792 float dy = q->corners[i]->pt.y - q->corners[(i+1)&3]->pt.y;
1793 float d = dx*dx + dy*dy;
1794 if( q->edge_len > d )
1795 q->edge_len = d;
1796 }
1797 quad_count++;
1798 }
1799
1800 return quad_count;
1801}
