68void CirclesGridClusterFinder::hierarchicalClustering(const vector<Point2f> points, const Size &patternSz, vector<Point2f> &patternPoints)
69{
70#ifdef HAVE_TEGRA_OPTIMIZATION
71 if(tegra::hierarchicalClustering(points, patternSz, patternPoints))
72 return;
73#endif
74 int j, n = (int)points.size();
75 size_t pn = static_cast<size_t>(patternSz.area());
76
77 patternPoints.clear();
78 if (pn >= points.size())
79 {
80 if (pn == points.size())
81 patternPoints = points;
82 return;
83 }
84
85 Mat dists(n, n, CV_32FC1, Scalar(0));
86 Mat distsMask(dists.size(), CV_8UC1, Scalar(0));
87 for(int i = 0; i < n; i++)
88 {
89 for(j = i+1; j < n; j++)
90 {
91 dists.at<float>(i, j) = (float)norm(points[i] - points[j]);
92 distsMask.at<uchar>(i, j) = 255;
93 //TODO: use symmetry
94 distsMask.at<uchar>(j, i) = 255;//distsMask.at<uchar>(i, j);
95 dists.at<float>(j, i) = dists.at<float>(i, j);
96 }
97 }
98
99 vector<std::list<size_t> > clusters(points.size());
100 for(size_t i=0; i<points.size(); i++)
101 {
102 clusters[i].push_back(i);
103 }
104
105 int patternClusterIdx = 0;
106 while(clusters[patternClusterIdx].size() < pn)
107 {
108 Point minLoc;
109 minMaxLoc(dists, 0, 0, &minLoc, 0, distsMask);
110 int minIdx = std::min(minLoc.x, minLoc.y);
111 int maxIdx = std::max(minLoc.x, minLoc.y);
112
113 distsMask.row(maxIdx).setTo(0);
114 distsMask.col(maxIdx).setTo(0);
115 Mat tmpRow = dists.row(minIdx);
116 Mat tmpCol = dists.col(minIdx);
117 cv::min(dists.row(minLoc.x), dists.row(minLoc.y), tmpRow);
118 tmpRow.copyTo(tmpCol);
119
120 clusters[minIdx].splice(clusters[minIdx].end(), clusters[maxIdx]);
121 patternClusterIdx = minIdx;
122 }
123
124 //the largest cluster can have more than pn points -- we need to filter out such situations
125 if(clusters[patternClusterIdx].size() != static_cast<size_t>(patternSz.area()))
126 {
127 return;
128 }
129
130 patternPoints.reserve(clusters[patternClusterIdx].size());
131 for(std::list<size_t>::iterator it = clusters[patternClusterIdx].begin(); it != clusters[patternClusterIdx].end(); it++)
132 {
133 patternPoints.push_back(points[*it]);
134 }
135}

137void CirclesGridClusterFinder::findGrid(const std::vector<cv::Point2f> points, cv::Size _patternSize, vector<Point2f>& centers)
138{
139 patternSize = _patternSize;
140 centers.clear();
141 if(points.empty())
142 {
143 return;
144 }
145
146 vector<Point2f> patternPoints;
147 hierarchicalClustering(points, patternSize, patternPoints);
148 if(patternPoints.empty())
149 {
150 return;
151 }
152
153#ifdef DEBUG_CIRCLES
154 Mat patternPointsImage(1024, 1248, CV_8UC1, Scalar(0));
155 drawPoints(patternPoints, patternPointsImage);
156 imshow("pattern points", patternPointsImage);
157#endif
158
159 vector<Point2f> hull2f;
160 convexHull(Mat(patternPoints), hull2f, false);
161 const size_t cornersCount = isAsymmetricGrid ? 6 : 4;
162 if(hull2f.size() < cornersCount)
163 return;
164
165 vector<Point2f> corners;
166 findCorners(hull2f, corners);
167 if(corners.size() != cornersCount)
168 return;
169
170 vector<Point2f> outsideCorners, sortedCorners;
171 if(isAsymmetricGrid)
172 {
173 findOutsideCorners(corners, outsideCorners);
174 const size_t outsideCornersCount = 2;
175 if(outsideCorners.size() != outsideCornersCount)
176 return;
177 }
178 getSortedCorners(hull2f, corners, outsideCorners, sortedCorners);
179 if(sortedCorners.size() != cornersCount)
180 return;
181
182 vector<Point2f> rectifiedPatternPoints;
183 rectifyPatternPoints(patternPoints, sortedCorners, rectifiedPatternPoints);
184 if(patternPoints.size() != rectifiedPatternPoints.size())
185 return;
186
187 parsePatternPoints(patternPoints, rectifiedPatternPoints, centers);
188}

190void CirclesGridClusterFinder::findCorners(const std::vector<cv::Point2f> &hull2f, std::vector<cv::Point2f> &corners)
191{
192 //find angles (cosines) of vertices in convex hull
193 vector<float> angles;
194 for(size_t i=0; i<hull2f.size(); i++)
195 {
196 Point2f vec1 = hull2f[(i+1) % hull2f.size()] - hull2f[i % hull2f.size()];
197 Point2f vec2 = hull2f[(i-1 + static_cast<int>(hull2f.size())) % hull2f.size()] - hull2f[i % hull2f.size()];
198 float angle = (float)(vec1.ddot(vec2) / (norm(vec1) * norm(vec2)));
199 angles.push_back(angle);
200 }
201
202 //sort angles by cosine
203 //corners are the most sharp angles (6)
204 Mat anglesMat = Mat(angles);
205 Mat sortedIndices;
206 sortIdx(anglesMat, sortedIndices, CV_SORT_EVERY_COLUMN + CV_SORT_DESCENDING);
207 CV_Assert(sortedIndices.type() == CV_32SC1);
208 CV_Assert(sortedIndices.cols == 1);
209 const int cornersCount = isAsymmetricGrid ? 6 : 4;
210 Mat cornersIndices;
211 cv::sort(sortedIndices.rowRange(0, cornersCount), cornersIndices, CV_SORT_EVERY_COLUMN + CV_SORT_ASCENDING);
212 corners.clear();
213 for(int i=0; i<cornersCount; i++)
214 {
215 corners.push_back(hull2f[cornersIndices.at<int>(i, 0)]);
216 }
217}

219void CirclesGridClusterFinder::findOutsideCorners(const std::vector<cv::Point2f> &corners, std::vector<cv::Point2f> &outsideCorners)
220{
221 outsideCorners.clear();
222 //find two pairs of the most nearest corners
223 int i, j, n = (int)corners.size();
224
225#ifdef DEBUG_CIRCLES
226 Mat cornersImage(1024, 1248, CV_8UC1, Scalar(0));
227 drawPoints(corners, cornersImage);
228 imshow("corners", cornersImage);
229#endif
230
231 vector<Point2f> tangentVectors(corners.size());
232 for(size_t k=0; k<corners.size(); k++)
233 {
234 Point2f diff = corners[(k + 1) % corners.size()] - corners[k];
235 tangentVectors[k] = diff * (1.0f / norm(diff));
236 }
237
238 //compute angles between all sides
239 Mat cosAngles(n, n, CV_32FC1, 0.0f);
240 for(i = 0; i < n; i++)
241 {
242 for(j = i + 1; j < n; j++)
243 {
244 float val = fabs(tangentVectors[i].dot(tangentVectors[j]));
245 cosAngles.at<float>(i, j) = val;
246 cosAngles.at<float>(j, i) = val;
247 }
248 }
249
250 //find two parallel sides to which outside corners belong
251 Point maxLoc;
252 minMaxLoc(cosAngles, 0, 0, 0, &maxLoc);
253 const int diffBetweenFalseLines = 3;
254 if(abs(maxLoc.x - maxLoc.y) == diffBetweenFalseLines)
255 {
256 cosAngles.row(maxLoc.x).setTo(0.0f);
257 cosAngles.col(maxLoc.x).setTo(0.0f);
258 cosAngles.row(maxLoc.y).setTo(0.0f);
259 cosAngles.col(maxLoc.y).setTo(0.0f);
260 minMaxLoc(cosAngles, 0, 0, 0, &maxLoc);
261 }
262
263#ifdef DEBUG_CIRCLES
264 Mat linesImage(1024, 1248, CV_8UC1, Scalar(0));
265 line(linesImage, corners[maxLoc.y], corners[(maxLoc.y + 1) % n], Scalar(255));
266 line(linesImage, corners[maxLoc.x], corners[(maxLoc.x + 1) % n], Scalar(255));
267 imshow("lines", linesImage);
268#endif
269
270 int maxIdx = std::max(maxLoc.x, maxLoc.y);
271 int minIdx = std::min(maxLoc.x, maxLoc.y);
272 const int bigDiff = 4;
273 if(maxIdx - minIdx == bigDiff)
274 {
275 minIdx += n;
276 std::swap(maxIdx, minIdx);
277 }
278 if(maxIdx - minIdx != n - bigDiff)
279 {
280 return;
281 }
282
283 int outsidersSegmentIdx = (minIdx + maxIdx) / 2;
284
285 outsideCorners.push_back(corners[outsidersSegmentIdx % n]);
286 outsideCorners.push_back(corners[(outsidersSegmentIdx + 1) % n]);
287
288#ifdef DEBUG_CIRCLES
289 drawPoints(outsideCorners, cornersImage, 2, Scalar(128));
290 imshow("corners", outsideCornersImage);
291#endif
292}

294void CirclesGridClusterFinder::getSortedCorners(const std::vector<cv::Point2f> &hull2f, const std::vector<cv::Point2f> &corners, const std::vector<cv::Point2f> &outsideCorners, std::vector<cv::Point2f> &sortedCorners)
295{
296 Point2f firstCorner;
297 if(isAsymmetricGrid)
298 {
299 Point2f center = std::accumulate(corners.begin(), corners.end(), Point2f(0.0f, 0.0f));
300 center *= 1.0 / corners.size();
301
302 vector<Point2f> centerToCorners;
303 for(size_t i=0; i<outsideCorners.size(); i++)
304 {
305 centerToCorners.push_back(outsideCorners[i] - center);
306 }
307
308 //TODO: use CirclesGridFinder::getDirection
309 float crossProduct = centerToCorners[0].x * centerToCorners[1].y - centerToCorners[0].y * centerToCorners[1].x;
310 //y axis is inverted in computer vision so we check > 0
311 bool isClockwise = crossProduct > 0;
312 firstCorner = isClockwise ? outsideCorners[1] : outsideCorners[0];
313 }
314 else
315 {
316 firstCorner = corners[0];
317 }
318
319 std::vector<Point2f>::const_iterator firstCornerIterator = std::find(hull2f.begin(), hull2f.end(), firstCorner);
320 sortedCorners.clear();
321 for(vector<Point2f>::const_iterator it = firstCornerIterator; it != hull2f.end(); it++)
322 {
323 vector<Point2f>::const_iterator itCorners = std::find(corners.begin(), corners.end(), *it);
324 if(itCorners != corners.end())
325 {
326 sortedCorners.push_back(*it);
327 }
328 }
329 for(vector<Point2f>::const_iterator it = hull2f.begin(); it != firstCornerIterator; it++)
330 {
331 vector<Point2f>::const_iterator itCorners = std::find(corners.begin(), corners.end(), *it);
332 if(itCorners != corners.end())
333 {
334 sortedCorners.push_back(*it);
335 }
336 }
337
338 if(!isAsymmetricGrid)
339 {
340 double dist1 = norm(sortedCorners[0] - sortedCorners[1]);
341 double dist2 = norm(sortedCorners[1] - sortedCorners[2]);
342
343 if((dist1 > dist2 && patternSize.height > patternSize.width) || (dist1 < dist2 && patternSize.height < patternSize.width))
344 {
345 for(size_t i=0; i<sortedCorners.size()-1; i++)
346 {
347 sortedCorners[i] = sortedCorners[i+1];
348 }
349 sortedCorners[sortedCorners.size() - 1] = firstCorner;
350 }
351 }
352}

354void CirclesGridClusterFinder::rectifyPatternPoints(const std::vector<cv::Point2f> &patternPoints, const std::vector<cv::Point2f> &sortedCorners, std::vector<cv::Point2f> &rectifiedPatternPoints)
355{
356 //indices of corner points in pattern
357 vector<Point> trueIndices;
358 trueIndices.push_back(Point(0, 0));
359 trueIndices.push_back(Point(patternSize.width - 1, 0));
360 if(isAsymmetricGrid)
361 {
362 trueIndices.push_back(Point(patternSize.width - 1, 1));
363 trueIndices.push_back(Point(patternSize.width - 1, patternSize.height - 2));
364 }
365 trueIndices.push_back(Point(patternSize.width - 1, patternSize.height - 1));
366 trueIndices.push_back(Point(0, patternSize.height - 1));
367
368 vector<Point2f> idealPoints;
369 for(size_t idx=0; idx<trueIndices.size(); idx++)
370 {
371 int i = trueIndices[idx].y;
372 int j = trueIndices[idx].x;
373 if(isAsymmetricGrid)
374 {
375 idealPoints.push_back(Point2f((2*j + i % 2)*squareSize, i*squareSize));
376 }
377 else
378 {
379 idealPoints.push_back(Point2f(j*squareSize, i*squareSize));
380 }
381 }
382
383 Mat homography = findHomography(Mat(sortedCorners), Mat(idealPoints), 0);
384 Mat rectifiedPointsMat;
385 transform(patternPoints, rectifiedPointsMat, homography);
386 rectifiedPatternPoints.clear();
387 convertPointsFromHomogeneous(rectifiedPointsMat, rectifiedPatternPoints);
388}

390void CirclesGridClusterFinder::parsePatternPoints(const std::vector<cv::Point2f> &patternPoints, const std::vector<cv::Point2f> &rectifiedPatternPoints, std::vector<cv::Point2f> &centers)
391{
392 flann::LinearIndexParams flannIndexParams;
393 flann::Index flannIndex(Mat(rectifiedPatternPoints).reshape(1), flannIndexParams);
394
395 centers.clear();
396 for( int i = 0; i < patternSize.height; i++ )
397 {
398 for( int j = 0; j < patternSize.width; j++ )
399 {
400 Point2f idealPt;
401 if(isAsymmetricGrid)
402 idealPt = Point2f((2*j + i % 2)*squareSize, i*squareSize);
403 else
404 idealPt = Point2f(j*squareSize, i*squareSize);
405
406 vector<float> query = Mat(idealPt);
407 int knn = 1;
408 vector<int> indices(knn);
409 vector<float> dists(knn);
410 flannIndex.knnSearch(query, indices, dists, knn, flann::SearchParams());
411 centers.push_back(patternPoints.at(indices[0]));
412
413 if(dists[0] > maxRectifiedDistance)
414 {
415#ifdef DEBUG_CIRCLES
416 cout << "Pattern not detected: too large rectified distance" << endl;
417#endif
418 centers.clear();
419 return;
420 }
421 }
422 }

485void Graph::floydWarshall(cv::Mat &distanceMatrix, int infinity) const
486{
487 const int edgeWeight = 1;
488
489 const int n = (int)getVerticesCount();
490 distanceMatrix.create(n, n, CV_32SC1);
491 distanceMatrix.setTo(infinity);
492 for (Vertices::const_iterator it1 = vertices.begin(); it1 != vertices.end(); it1++)
493 {
494 distanceMatrix.at<int> ((int)it1->first, (int)it1->first) = 0;
495 for (Neighbors::const_iterator it2 = it1->second.neighbors.begin(); it2 != it1->second.neighbors.end(); it2++)
496 {
497 assert( it1->first != *it2 );
498 distanceMatrix.at<int> ((int)it1->first, (int)*it2) = edgeWeight;
499 }
500 }
501
502 for (Vertices::const_iterator it1 = vertices.begin(); it1 != vertices.end(); it1++)
503 {
504 for (Vertices::const_iterator it2 = vertices.begin(); it2 != vertices.end(); it2++)
505 {
506 for (Vertices::const_iterator it3 = vertices.begin(); it3 != vertices.end(); it3++)
507 {
508 int i1 = (int)it1->first, i2 = (int)it2->first, i3 = (int)it3->first;
509 int val1 = distanceMatrix.at<int> (i2, i3);
510 int val2;
511 if (distanceMatrix.at<int> (i2, i1) == infinity ||
512 distanceMatrix.at<int> (i1, i3) == infinity)
513 val2 = val1;
514 else
515 {
516 val2 = distanceMatrix.at<int> (i2, i1) + distanceMatrix.at<int> (i1, i3);
517 }
518 distanceMatrix.at<int> (i2, i3) = (val1 == infinity) ? val2 : std::min(val1, val2);
519 }
520 }
521 }

572bool CirclesGridFinder::findHoles()
573{
574 switch (parameters.gridType)
575 {
576 case CirclesGridFinderParameters::SYMMETRIC_GRID:
577 {
578 vector<Point2f> vectors, filteredVectors, basis;
579 Graph rng(0);
580 computeRNG(rng, vectors);
581 filterOutliersByDensity(vectors, filteredVectors);
582 vector<Graph> basisGraphs;
583 findBasis(filteredVectors, basis, basisGraphs);
584 findMCS(basis, basisGraphs);
585 break;
586 }
587
588 case CirclesGridFinderParameters::ASYMMETRIC_GRID:
589 {
590 vector<Point2f> vectors, tmpVectors, filteredVectors, basis;
591 Graph rng(0);
592 computeRNG(rng, tmpVectors);
593 rng2gridGraph(rng, vectors);
594 filterOutliersByDensity(vectors, filteredVectors);
595 vector<Graph> basisGraphs;
596 findBasis(filteredVectors, basis, basisGraphs);
597 findMCS(basis, basisGraphs);
598 eraseUsedGraph(basisGraphs);
599 holes2 = holes;
600 holes.clear();
601 findMCS(basis, basisGraphs);
602 break;
603 }
604
605 default:
606 CV_Error(CV_StsBadArg, "Unkown pattern type");
607 }
608 return (isDetectionCorrect());
609 //CV_Error( 0, "Detection is not correct" );
610}

612void CirclesGridFinder::rng2gridGraph(Graph &rng, std::vector<cv::Point2f> &vectors) const
613{
614 for (size_t i = 0; i < rng.getVerticesCount(); i++)
615 {
616 Graph::Neighbors neighbors1 = rng.getNeighbors(i);
617 for (Graph::Neighbors::iterator it1 = neighbors1.begin(); it1 != neighbors1.end(); it1++)
618 {
619 Graph::Neighbors neighbors2 = rng.getNeighbors(*it1);
620 for (Graph::Neighbors::iterator it2 = neighbors2.begin(); it2 != neighbors2.end(); it2++)
621 {
622 if (i < *it2)
623 {
624 Point2f vec1 = keypoints[i] - keypoints[*it1];
625 Point2f vec2 = keypoints[*it1] - keypoints[*it2];
626 if (norm(vec1 - vec2) < parameters.minRNGEdgeSwitchDist || norm(vec1 + vec2)
627 < parameters.minRNGEdgeSwitchDist)
628 continue;
629
630 vectors.push_back(keypoints[i] - keypoints[*it2]);
631 vectors.push_back(keypoints[*it2] - keypoints[i]);
632 }
633 }
634 }
635 }

638void CirclesGridFinder::eraseUsedGraph(vector<Graph> &basisGraphs) const
639{
640 for (size_t i = 0; i < holes.size(); i++)
641 {
642 for (size_t j = 0; j < holes[i].size(); j++)
643 {
644 for (size_t k = 0; k < basisGraphs.size(); k++)
645 {
646 if (i != holes.size() - 1 && basisGraphs[k].areVerticesAdjacent(holes[i][j], holes[i + 1][j]))
647 {
648 basisGraphs[k].removeEdge(holes[i][j], holes[i + 1][j]);
649 }
650
651 if (j != holes[i].size() - 1 && basisGraphs[k].areVerticesAdjacent(holes[i][j], holes[i][j + 1]))
652 {
653 basisGraphs[k].removeEdge(holes[i][j], holes[i][j + 1]);
654 }
655 }
656 }
657 }

660bool CirclesGridFinder::isDetectionCorrect()
661{
662 switch (parameters.gridType)
663 {
664 case CirclesGridFinderParameters::SYMMETRIC_GRID:
665 {
666 if (holes.size() != patternSize.height)
667 return false;
668
669 set<size_t> vertices;
670 for (size_t i = 0; i < holes.size(); i++)
671 {
672 if (holes[i].size() != patternSize.width)
673 return false;
674
675 for (size_t j = 0; j < holes[i].size(); j++)
676 {
677 vertices.insert(holes[i][j]);
678 }
679 }
680
681 return vertices.size() == patternSize.area();
682 }
683
684 case CirclesGridFinderParameters::ASYMMETRIC_GRID:
685 {
686 if (holes.size() < holes2.size() || holes[0].size() < holes2[0].size())
687 {
688 largeHoles = &holes2;
689 smallHoles = &holes;
690 }
691 else
692 {
693 largeHoles = &holes;
694 smallHoles = &holes2;
695 }
696
697 size_t largeWidth = patternSize.width;
698 size_t largeHeight = (size_t)ceil(patternSize.height / 2.);
699 size_t smallWidth = patternSize.width;
700 size_t smallHeight = (size_t)floor(patternSize.height / 2.);
701
702 size_t sw = smallWidth, sh = smallHeight, lw = largeWidth, lh = largeHeight;
703 if (largeHoles->size() != largeHeight)
704 {
705 std::swap(lh, lw);
706 }
707 if (smallHoles->size() != smallHeight)
708 {
709 std::swap(sh, sw);
710 }
711
712 if (largeHoles->size() != lh || smallHoles->size() != sh)
713 {
714 return false;
715 }
716
717 set<size_t> vertices;
718 for (size_t i = 0; i < largeHoles->size(); i++)
719 {
720 if (largeHoles->at(i).size() != lw)
721 {
722 return false;
723 }
724
725 for (size_t j = 0; j < largeHoles->at(i).size(); j++)
726 {
727 vertices.insert(largeHoles->at(i)[j]);
728 }
729
730 if (i < smallHoles->size())
731 {
732 if (smallHoles->at(i).size() != sw)
733 {
734 return false;
735 }
736
737 for (size_t j = 0; j < smallHoles->at(i).size(); j++)
738 {
739 vertices.insert(smallHoles->at(i)[j]);
740 }
741 }
742 }
743 return (vertices.size() == largeHeight * largeWidth + smallHeight * smallWidth);
744 }
745
746 default:
747 CV_Error(0, "Unknown pattern type");
748 }
749
750 return false;
751}

774 //parameters.minGraphConfidence = holes[0].size() * parameters.existingVertexGain + (holes[0].size() / 2) * parameters.edgeGain;

797

799 for (size_t i = w; i < patternSize.width; i++)

812Mat CirclesGridFinder::rectifyGrid(Size detectedGridSize, const vector<Point2f>& centers,
813 const vector<Point2f> &keypoints, vector<Point2f> &warpedKeypoints)
814{
815 assert( !centers.empty() );
816 const float edgeLength = 30;
817 const Point2f offset(150, 150);
818
819 vector<Point2f> dstPoints;
820 bool isClockwiseBefore =
821 getDirection(centers[0], centers[detectedGridSize.width - 1], centers[centers.size() - 1]) < 0;
822
823 int iStart = isClockwiseBefore ? 0 : detectedGridSize.height - 1;
824 int iEnd = isClockwiseBefore ? detectedGridSize.height : -1;
825 int iStep = isClockwiseBefore ? 1 : -1;
826 for (int i = iStart; i != iEnd; i += iStep)
827 {
828 for (int j = 0; j < detectedGridSize.width; j++)
829 {
830 dstPoints.push_back(offset + Point2f(edgeLength * j, edgeLength * i));
831 }
832 }
833
834 Mat H = findHomography(Mat(centers), Mat(dstPoints), CV_RANSAC);
835 //Mat H = findHomography( Mat( corners ), Mat( dstPoints ) );
836
837 vector<Point2f> srcKeypoints;
838 for (size_t i = 0; i < keypoints.size(); i++)
839 {
840 srcKeypoints.push_back(keypoints[i]);
841 }
842
843 Mat dstKeypointsMat;
844 transform(Mat(srcKeypoints), dstKeypointsMat, H);
845 vector<Point2f> dstKeypoints;
846 convertPointsFromHomogeneous(dstKeypointsMat, dstKeypoints);
847
848 warpedKeypoints.clear();
849 for (size_t i = 0; i < dstKeypoints.size(); i++)
850 {
851 Point2f pt = dstKeypoints[i];
852 warpedKeypoints.push_back(pt);
853 }
854
855 return H;
856}

950void CirclesGridFinder::insertWinner(float aboveConfidence, float belowConfidence, float minConfidence, bool addRow,
951 const vector<size_t> &above, const vector<size_t> &below,
952 vector<vector<size_t> > &holes)
953{
954 if (aboveConfidence < minConfidence && belowConfidence < minConfidence)
955 return;
956
957 if (addRow)
958 {
959 if (aboveConfidence >= belowConfidence)
960 {
961 if (!areCentersNew(above, holes))
962 CV_Error( 0, "Centers are not new" );
963
964 holes.insert(holes.begin(), above);
965 }
966 else
967 {
968 if (!areCentersNew(below, holes))
969 CV_Error( 0, "Centers are not new" );
970
971 holes.insert(holes.end(), below);
972 }
973 }
974 else
975 {
976 if (aboveConfidence >= belowConfidence)
977 {
978 if (!areCentersNew(above, holes))
979 CV_Error( 0, "Centers are not new" );
980
981 for (size_t i = 0; i < holes.size(); i++)
982 {
983 holes[i].insert(holes[i].begin(), above[i]);
984 }
985 }
986 else
987 {
988 if (!areCentersNew(below, holes))
989 CV_Error( 0, "Centers are not new" );
990
991 for (size_t i = 0; i < holes.size(); i++)
992 {
993 holes[i].insert(holes[i].end(), below[i]);
994 }
995 }
996 }

999float CirclesGridFinder::computeGraphConfidence(const vector<Graph> &basisGraphs, bool addRow,
1000 const vector<size_t> &points, const vector<size_t> &seeds)
1001{
1002 assert( points.size() == seeds.size() );
1003 float confidence = 0;
1004 const size_t vCount = basisGraphs[0].getVerticesCount();
1005 assert( basisGraphs[0].getVerticesCount() == basisGraphs[1].getVerticesCount() );
1006
1007 for (size_t i = 0; i < seeds.size(); i++)
1008 {
1009 if (seeds[i] < vCount && points[i] < vCount)
1010 {
1011 if (!basisGraphs[addRow].areVerticesAdjacent(seeds[i], points[i]))
1012 {
1013 confidence += parameters.vertexPenalty;
1014 }
1015 else
1016 {
1017 confidence += parameters.vertexGain;
1018 }
1019 }
1020
1021 if (points[i] < vCount)
1022 {
1023 confidence += parameters.existingVertexGain;
1024 }
1025 }
1026
1027 for (size_t i = 1; i < points.size(); i++)
1028 {
1029 if (points[i - 1] < vCount && points[i] < vCount)
1030 {
1031 if (!basisGraphs[!addRow].areVerticesAdjacent(points[i - 1], points[i]))
1032 {
1033 confidence += parameters.edgePenalty;
1034 }
1035 else
1036 {
1037 confidence += parameters.edgeGain;
1038 }
1039 }
1040 }
1041 return confidence;
1042
1043}

1080void CirclesGridFinder::findBasis(const vector<Point2f> &samples, vector<Point2f> &basis, vector<Graph> &basisGraphs)
1081{
1082 basis.clear();
1083 Mat bestLabels;
1084 TermCriteria termCriteria;
1085 Mat centers;
1086 const int clustersCount = 4;
1087 kmeans(Mat(samples).reshape(1, 0), clustersCount, bestLabels, termCriteria, parameters.kmeansAttempts,
1088 KMEANS_RANDOM_CENTERS, centers);
1089 assert( centers.type() == CV_32FC1 );
1090
1091 vector<int> basisIndices;
1092 //TODO: only remove duplicate
1093 for (int i = 0; i < clustersCount; i++)
1094 {
1095 int maxIdx = (fabs(centers.at<float> (i, 0)) < fabs(centers.at<float> (i, 1)));
1096 if (centers.at<float> (i, maxIdx) > 0)
1097 {
1098 Point2f vec(centers.at<float> (i, 0), centers.at<float> (i, 1));
1099 basis.push_back(vec);
1100 basisIndices.push_back(i);
1101 }
1102 }
1103 if (basis.size() != 2)
1104 CV_Error(0, "Basis size is not 2");
1105
1106 if (basis[1].x > basis[0].x)
1107 {
1108 std::swap(basis[0], basis[1]);
1109 std::swap(basisIndices[0], basisIndices[1]);
1110 }
1111
1112 const float minBasisDif = 2;
1113 if (norm(basis[0] - basis[1]) < minBasisDif)
1114 CV_Error(0, "degenerate basis" );
1115
1116 vector<vector<Point2f> > clusters(2), hulls(2);
1117 for (int k = 0; k < (int)samples.size(); k++)
1118 {
1119 int label = bestLabels.at<int> (k, 0);
1120 int idx = -1;
1121 if (label == basisIndices[0])
1122 idx = 0;
1123 if (label == basisIndices[1])
1124 idx = 1;
1125 if (idx >= 0)
1126 {
1127 clusters[idx].push_back(basis[idx] + parameters.convexHullFactor * (samples[k] - basis[idx]));
1128 }
1129 }
1130 for (size_t i = 0; i < basis.size(); i++)
1131 {
1132 convexHull(Mat(clusters[i]), hulls[i]);
1133 }
1134
1135 basisGraphs.resize(basis.size(), Graph(keypoints.size()));
1136 for (size_t i = 0; i < keypoints.size(); i++)
1137 {
1138 for (size_t j = 0; j < keypoints.size(); j++)
1139 {
1140 if (i == j)
1141 continue;
1142
1143 Point2f vec = keypoints[i] - keypoints[j];
1144
1145 for (size_t k = 0; k < hulls.size(); k++)
1146 {
1147 if (pointPolygonTest(Mat(hulls[k]), vec, false) >= 0)
1148 {
1149 basisGraphs[k].addEdge(i, j);
1150 }
1151 }
1152 }
1153 }
1154 if (basisGraphs.size() != 2)
1155 CV_Error(0, "Number of basis graphs is not 2");
1156}

1204void computePredecessorMatrix(const Mat &dm, int verticesCount, Mat &predecessorMatrix)
1205{
1206 assert( dm.type() == CV_32SC1 );
1207 predecessorMatrix.create(verticesCount, verticesCount, CV_32SC1);
1208 predecessorMatrix = -1;
1209 for (int i = 0; i < predecessorMatrix.rows; i++)
1210 {
1211 for (int j = 0; j < predecessorMatrix.cols; j++)
1212 {
1213 int dist = dm.at<int> (i, j);
1214 for (int k = 0; k < verticesCount; k++)
1215 {
1216 if (dm.at<int> (i, k) == dist - 1 && dm.at<int> (k, j) == 1)
1217 {
1218 predecessorMatrix.at<int> (i, j) = k;
1219 break;
1220 }
1221 }
1222 }
1223 }

1238size_t CirclesGridFinder::findLongestPath(vector<Graph> &basisGraphs, Path &bestPath)
1239{
1240 vector<Path> longestPaths(1);
1241 vector<int> confidences;
1242
1243 size_t bestGraphIdx = 0;
1244 const int infinity = -1;
1245 for (size_t graphIdx = 0; graphIdx < basisGraphs.size(); graphIdx++)
1246 {
1247 const Graph &g = basisGraphs[graphIdx];
1248 Mat distanceMatrix;
1249 g.floydWarshall(distanceMatrix, infinity);
1250 Mat predecessorMatrix;
1251 computePredecessorMatrix(distanceMatrix, (int)g.getVerticesCount(), predecessorMatrix);
1252
1253 double maxVal;
1254 Point maxLoc;
1255 assert (infinity < 0);
1256 minMaxLoc(distanceMatrix, 0, &maxVal, 0, &maxLoc);
1257
1258 if (maxVal > longestPaths[0].length)
1259 {
1260 longestPaths.clear();
1261 confidences.clear();
1262 bestGraphIdx = graphIdx;
1263 }
1264 if (longestPaths.empty() || (maxVal == longestPaths[0].length && graphIdx == bestGraphIdx))
1265 {
1266 Path path = Path(maxLoc.x, maxLoc.y, cvRound(maxVal));
1267 CV_Assert(maxLoc.x >= 0 && maxLoc.y >= 0)
1268 ;
1269 size_t id1 = static_cast<size_t> (maxLoc.x);
1270 size_t id2 = static_cast<size_t> (maxLoc.y);
1271 computeShortestPath(predecessorMatrix, id1, id2, path.vertices);
1272 longestPaths.push_back(path);
1273
1274 int conf = 0;
1275 for (int v2 = 0; v2 < (int)path.vertices.size(); v2++)
1276 {
1277 conf += (int)basisGraphs[1 - (int)graphIdx].getDegree(v2);
1278 }
1279 confidences.push_back(conf);
1280 }
1281 }
1282 //if( bestGraphIdx != 0 )
1283 //CV_Error( 0, "" );
1284
1285 int maxConf = -1;
1286 int bestPathIdx = -1;
1287 for (int i = 0; i < (int)confidences.size(); i++)
1288 {
1289 if (confidences[i] > maxConf)
1290 {
1291 maxConf = confidences[i];
1292 bestPathIdx = i;
1293 }
1294 }
1295
1296 //int bestPathIdx = rand() % longestPaths.size();
1297 bestPath = longestPaths.at(bestPathIdx);
1298 bool needReverse = (bestGraphIdx == 0 && keypoints[bestPath.lastVertex].x < keypoints[bestPath.firstVertex].x)
1299 || (bestGraphIdx == 1 && keypoints[bestPath.lastVertex].y < keypoints[bestPath.firstVertex].y);
1300 if (needReverse)
1301 {
1302 std::swap(bestPath.lastVertex, bestPath.firstVertex);
1303 std::reverse(bestPath.vertices.begin(), bestPath.vertices.end());
1304 }
1305 return bestGraphIdx;
1306}

1475void CirclesGridFinder::getCornerSegments(const vector<vector<size_t> > &points, vector<vector<Segment> > &segments,
1476 vector<Point> &cornerIndices, vector<Point> &firstSteps,
1477 vector<Point> &secondSteps) const
1478{
1479 segments.clear();
1480 cornerIndices.clear();
1481 firstSteps.clear();
1482 secondSteps.clear();
1483 int h = (int)points.size();
1484 int w = (int)points[0].size();
1485 CV_Assert(h >= 2 && w >= 2)
1486 ;
1487
1488 //all 8 segments with one end in a corner
1489 vector<Segment> corner;
1490 corner.push_back(Segment(keypoints[points[1][0]], keypoints[points[0][0]]));
1491 corner.push_back(Segment(keypoints[points[0][0]], keypoints[points[0][1]]));
1492 segments.push_back(corner);
1493 cornerIndices.push_back(Point(0, 0));
1494 firstSteps.push_back(Point(1, 0));
1495 secondSteps.push_back(Point(0, 1));
1496 corner.clear();
1497
1498 corner.push_back(Segment(keypoints[points[0][w - 2]], keypoints[points[0][w - 1]]));
1499 corner.push_back(Segment(keypoints[points[0][w - 1]], keypoints[points[1][w - 1]]));
1500 segments.push_back(corner);
1501 cornerIndices.push_back(Point(w - 1, 0));
1502 firstSteps.push_back(Point(0, 1));
1503 secondSteps.push_back(Point(-1, 0));
1504 corner.clear();
1505
1506 corner.push_back(Segment(keypoints[points[h - 2][w - 1]], keypoints[points[h - 1][w - 1]]));
1507 corner.push_back(Segment(keypoints[points[h - 1][w - 1]], keypoints[points[h - 1][w - 2]]));
1508 segments.push_back(corner);
1509 cornerIndices.push_back(Point(w - 1, h - 1));
1510 firstSteps.push_back(Point(-1, 0));
1511 secondSteps.push_back(Point(0, -1));
1512 corner.clear();
1513
1514 corner.push_back(Segment(keypoints[points[h - 1][1]], keypoints[points[h - 1][0]]));
1515 corner.push_back(Segment(keypoints[points[h - 1][0]], keypoints[points[h - 2][0]]));
1516 cornerIndices.push_back(Point(0, h - 1));
1517 firstSteps.push_back(Point(0, -1));
1518 secondSteps.push_back(Point(1, 0));
1519 segments.push_back(corner);
1520 corner.clear();
1521
1522 //y axis is inverted in computer vision so we check < 0
1523 bool isClockwise =
1524 getDirection(keypoints[points[0][0]], keypoints[points[0][w - 1]], keypoints[points[h - 1][w - 1]]) < 0;
1525 if (!isClockwise)
1526 {
1527#ifdef DEBUG_CIRCLES
1528 cout << "Corners are counterclockwise" << endl;
1529#endif
1530 std::reverse(segments.begin(), segments.end());
1531 std::reverse(cornerIndices.begin(), cornerIndices.end());
1532 std::reverse(firstSteps.begin(), firstSteps.end());
1533 std::reverse(secondSteps.begin(), secondSteps.end());
1534 std::swap(firstSteps, secondSteps);
1535 }
1536}

1538bool CirclesGridFinder::doesIntersectionExist(const vector<Segment> &corner, const vector<vector<Segment> > &segments)
1539{

1555size_t CirclesGridFinder::getFirstCorner(vector<Point> &largeCornerIndices, vector<Point> &smallCornerIndices, vector<
1556 Point> &firstSteps, vector<Point> &secondSteps) const
1557{
1558 vector<vector<Segment> > largeSegments;
1559 vector<vector<Segment> > smallSegments;
1560
1561 getCornerSegments(*largeHoles, largeSegments, largeCornerIndices, firstSteps, secondSteps);
1562 getCornerSegments(*smallHoles, smallSegments, smallCornerIndices, firstSteps, secondSteps);
1563
1564 const size_t cornersCount = 4;
1565 CV_Assert(largeSegments.size() == cornersCount)
1566 ;
1567
1568 bool isInsider[cornersCount];
1569
1570 for (size_t i = 0; i < cornersCount; i++)
1571 {
1572 isInsider[i] = doesIntersectionExist(largeSegments[i], smallSegments);
1573 }
1574
1575 int cornerIdx = 0;
1576 bool waitOutsider = true;
1577
1578 for(;;)
1579 {
1580 if (waitOutsider)
1581 {
1582 if (!isInsider[(cornerIdx + 1) % cornersCount])
1583 waitOutsider = false;
1584 }
1585 else
1586 {
1587 if (isInsider[(cornerIdx + 1) % cornersCount])
1588 break;
1589 }
1590
1591 cornerIdx = (cornerIdx + 1) % cornersCount;
1592 }
1593
1594 return cornerIdx;
1595}
