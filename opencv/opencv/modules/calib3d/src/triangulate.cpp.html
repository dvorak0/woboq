51// This method is the same as icvReconstructPointsFor3View, with only a few numbers adjusted for two-view geometry
52CV_IMPL void
53cvTriangulatePoints(CvMat* projMatr1, CvMat* projMatr2, CvMat* projPoints1, CvMat* projPoints2, CvMat* points4D)
54{
55 if( projMatr1 == 0 || projMatr2 == 0 ||
56 projPoints1 == 0 || projPoints2 == 0 ||
57 points4D == 0)
58 CV_Error( CV_StsNullPtr, "Some of parameters is a NULL pointer" );
59
60 if( !CV_IS_MAT(projMatr1) || !CV_IS_MAT(projMatr2) ||
61 !CV_IS_MAT(projPoints1) || !CV_IS_MAT(projPoints2) ||
62 !CV_IS_MAT(points4D) )
63 CV_Error( CV_StsUnsupportedFormat, "Input parameters must be matrices" );
64
65 int numPoints;
66 numPoints = projPoints1->cols;
67
68 if( numPoints < 1 )
69 CV_Error( CV_StsOutOfRange, "Number of points must be more than zero" );
70
71 if( projPoints2->cols != numPoints || points4D->cols != numPoints )
72 CV_Error( CV_StsUnmatchedSizes, "Number of points must be the same" );
73
74 if( projPoints1->rows != 2 || projPoints2->rows != 2)
75 CV_Error( CV_StsUnmatchedSizes, "Number of proj points coordinates must be == 2" );
76
77 if( points4D->rows != 4 )
78 CV_Error( CV_StsUnmatchedSizes, "Number of world points coordinates must be == 4" );
79
80 if( projMatr1->cols != 4 || projMatr1->rows != 3 ||
81 projMatr2->cols != 4 || projMatr2->rows != 3)
82 CV_Error( CV_StsUnmatchedSizes, "Size of projection matrices must be 3x4" );
83
84 CvMat matrA;
85 double matrA_dat[16];
86 matrA = cvMat(4,4,CV_64F,matrA_dat);
87
88 //CvMat matrU;
89 CvMat matrW;
90 CvMat matrV;
91 //double matrU_dat[9*9];
92 double matrW_dat[4*4];
93 double matrV_dat[4*4];
94
95 //matrU = cvMat(6,6,CV_64F,matrU_dat);
96 matrW = cvMat(4,4,CV_64F,matrW_dat);
97 matrV = cvMat(4,4,CV_64F,matrV_dat);
98
99 CvMat* projPoints[2];
100 CvMat* projMatrs[2];
101
102 projPoints[0] = projPoints1;
103 projPoints[1] = projPoints2;
104
105 projMatrs[0] = projMatr1;
106 projMatrs[1] = projMatr2;
107
108 /* Solve system for each point */
109 int i,j;
110 for( i = 0; i < numPoints; i++ )/* For each point */
111 {
112 /* Fill matrix for current point */
113 for( j = 0; j < 2; j++ )/* For each view */
114 {
115 double x,y;
116 x = cvmGet(projPoints[j],0,i);
117 y = cvmGet(projPoints[j],1,i);
118 for( int k = 0; k < 4; k++ )
119 {
120 cvmSet(&matrA, j*2+0, k, x * cvmGet(projMatrs[j],2,k) - cvmGet(projMatrs[j],0,k) );
121 cvmSet(&matrA, j*2+1, k, y * cvmGet(projMatrs[j],2,k) - cvmGet(projMatrs[j],1,k) );
122 }
123 }
124 /* Solve system for current point */
125 {
126 cvSVD(&matrA,&matrW,0,&matrV,CV_SVD_V_T);
127
128 /* Copy computed point */
129 cvmSet(points4D,0,i,cvmGet(&matrV,3,0));/* X */
130 cvmSet(points4D,1,i,cvmGet(&matrV,3,1));/* Y */
131 cvmSet(points4D,2,i,cvmGet(&matrV,3,2));/* Z */
132 cvmSet(points4D,3,i,cvmGet(&matrV,3,3));/* W */
133 }
134 }
135
136#if 0
137 double err = 0;
138 /* Points was reconstructed. Try to reproject points */
139 /* We can compute reprojection error if need */
140 {
141 int i;
142 CvMat point3D;
143 double point3D_dat[4];
144 point3D = cvMat(4,1,CV_64F,point3D_dat);
145
146 CvMat point2D;
147 double point2D_dat[3];
148 point2D = cvMat(3,1,CV_64F,point2D_dat);
149
150 for( i = 0; i < numPoints; i++ )
151 {
152 double W = cvmGet(points4D,3,i);
153
154 point3D_dat[0] = cvmGet(points4D,0,i)/W;
155 point3D_dat[1] = cvmGet(points4D,1,i)/W;
156 point3D_dat[2] = cvmGet(points4D,2,i)/W;
157 point3D_dat[3] = 1;
158
159 /* !!! Project this point for each camera */
160 for( int currCamera = 0; currCamera < 2; currCamera++ )
161 {
162 cvMatMul(projMatrs[currCamera], &point3D, &point2D);
163
164 float x,y;
165 float xr,yr,wr;
166 x = (float)cvmGet(projPoints[currCamera],0,i);
167 y = (float)cvmGet(projPoints[currCamera],1,i);
168
169 wr = (float)point2D_dat[2];
170 xr = (float)(point2D_dat[0]/wr);
171 yr = (float)(point2D_dat[1]/wr);
172
173 float deltaX,deltaY;
174 deltaX = (float)fabs(x-xr);
175 deltaY = (float)fabs(y-yr);
176 err += deltaX*deltaX + deltaY*deltaY;
177 }
178 }
179 }
180#endif
181}

184/*
185 * The Optimal Triangulation Method (see HZ for details)
186 * For each given point correspondence points1[i] <-> points2[i], and a fundamental matrix F,
187 * computes the corrected correspondences new_points1[i] <-> new_points2[i] that minimize the
188 * geometric error d(points1[i],new_points1[i])^2 + d(points2[i],new_points2[i])^2 (where d(a,b)
189 * is the geometric distance between points a and b) subject to the epipolar constraint
190 * new_points2' * F * new_points1 = 0.
191 *
192 * F_ : 3x3 fundamental matrix
193 * points1_ : 1xN matrix containing the first set of points
194 * points2_ : 1xN matrix containing the second set of points
195 * new_points1 : the optimized points1_. if this is NULL, the corrected points are placed back in points1_
196 * new_points2 : the optimized points2_. if this is NULL, the corrected points are placed back in points2_
197 */
198CV_IMPL void
199cvCorrectMatches(CvMat *F_, CvMat *points1_, CvMat *points2_, CvMat *new_points1, CvMat *new_points2)
200{
201 cv::Ptr<CvMat> tmp33;
202 cv::Ptr<CvMat> tmp31, tmp31_2;
203 cv::Ptr<CvMat> T1i, T2i;
204 cv::Ptr<CvMat> R1, R2;
205 cv::Ptr<CvMat> TFT, TFTt, RTFTR;
206 cv::Ptr<CvMat> U, S, V;
207 cv::Ptr<CvMat> e1, e2;
208 cv::Ptr<CvMat> polynomial;
209 cv::Ptr<CvMat> result;
210 cv::Ptr<CvMat> points1, points2;
211 cv::Ptr<CvMat> F;
212
213 if (!CV_IS_MAT(F_) || !CV_IS_MAT(points1_) || !CV_IS_MAT(points2_) )
214 CV_Error( CV_StsUnsupportedFormat, "Input parameters must be matrices" );
215 if (!( F_->cols == 3 && F_->rows == 3))
216 CV_Error( CV_StsUnmatchedSizes, "The fundamental matrix must be a 3x3 matrix");
217 if (!(((F_->type & CV_MAT_TYPE_MASK) >> 3) == 0 ))
218 CV_Error( CV_StsUnsupportedFormat, "The fundamental matrix must be a single-channel matrix" );
219 if (!(points1_->rows == 1 && points2_->rows == 1 && points1_->cols == points2_->cols))
220 CV_Error( CV_StsUnmatchedSizes, "The point-matrices must have one row, and an equal number of columns" );
221 if (((points1_->type & CV_MAT_TYPE_MASK) >> 3) != 1 )
222 CV_Error( CV_StsUnmatchedSizes, "The first set of points must contain two channels; one for x and one for y" );
223 if (((points2_->type & CV_MAT_TYPE_MASK) >> 3) != 1 )
224 CV_Error( CV_StsUnmatchedSizes, "The second set of points must contain two channels; one for x and one for y" );
225 if (new_points1 != NULL) {
226 CV_Assert(CV_IS_MAT(new_points1));
227 if (new_points1->cols != points1_->cols || new_points1->rows != 1)
228 CV_Error( CV_StsUnmatchedSizes, "The first output matrix must have the same dimensions as the input matrices" );
229 if (CV_MAT_CN(new_points1->type) != 2)
230 CV_Error( CV_StsUnsupportedFormat, "The first output matrix must have two channels; one for x and one for y" );
231 }
232 if (new_points2 != NULL) {
233 CV_Assert(CV_IS_MAT(new_points2));
234 if (new_points2->cols != points2_->cols || new_points2->rows != 1)
235 CV_Error( CV_StsUnmatchedSizes, "The second output matrix must have the same dimensions as the input matrices" );
236 if (CV_MAT_CN(new_points2->type) != 2)
237 CV_Error( CV_StsUnsupportedFormat, "The second output matrix must have two channels; one for x and one for y" );
238 }
239
240 // Make sure F uses double precision
241 F = cvCreateMat(3,3,CV_64FC1);
242 cvConvert(F_, F);
243
244 // Make sure points1 uses double precision
245 points1 = cvCreateMat(points1_->rows,points1_->cols,CV_64FC2);
246 cvConvert(points1_, points1);
247
248 // Make sure points2 uses double precision
249 points2 = cvCreateMat(points2_->rows,points2_->cols,CV_64FC2);
250 cvConvert(points2_, points2);
251
252 tmp33 = cvCreateMat(3,3,CV_64FC1);
253 tmp31 = cvCreateMat(3,1,CV_64FC1), tmp31_2 = cvCreateMat(3,1,CV_64FC1);
254 T1i = cvCreateMat(3,3,CV_64FC1), T2i = cvCreateMat(3,3,CV_64FC1);
255 R1 = cvCreateMat(3,3,CV_64FC1), R2 = cvCreateMat(3,3,CV_64FC1);
256 TFT = cvCreateMat(3,3,CV_64FC1), TFTt = cvCreateMat(3,3,CV_64FC1), RTFTR = cvCreateMat(3,3,CV_64FC1);
257 U = cvCreateMat(3,3,CV_64FC1);
258 S = cvCreateMat(3,3,CV_64FC1);
259 V = cvCreateMat(3,3,CV_64FC1);
260 e1 = cvCreateMat(3,1,CV_64FC1), e2 = cvCreateMat(3,1,CV_64FC1);
261
262 double x1, y1, x2, y2;
263 double scale;
264 double f1, f2, a, b, c, d;
265 polynomial = cvCreateMat(1,7,CV_64FC1);
266 result = cvCreateMat(1,6,CV_64FC2);
267 double t_min, s_val, t, s;
268 for (int p = 0; p < points1->cols; ++p) {
269 // Replace F by T2-t * F * T1-t
270 x1 = points1->data.db[p*2];
271 y1 = points1->data.db[p*2+1];
272 x2 = points2->data.db[p*2];
273 y2 = points2->data.db[p*2+1];
274
275 cvSetZero(T1i);
276 cvSetReal2D(T1i,0,0,1);
277 cvSetReal2D(T1i,1,1,1);
278 cvSetReal2D(T1i,2,2,1);
279 cvSetReal2D(T1i,0,2,x1);
280 cvSetReal2D(T1i,1,2,y1);
281 cvSetZero(T2i);
282 cvSetReal2D(T2i,0,0,1);
283 cvSetReal2D(T2i,1,1,1);
284 cvSetReal2D(T2i,2,2,1);
285 cvSetReal2D(T2i,0,2,x2);
286 cvSetReal2D(T2i,1,2,y2);
287 cvGEMM(T2i,F,1,0,0,tmp33,CV_GEMM_A_T);
288 cvSetZero(TFT);
289 cvGEMM(tmp33,T1i,1,0,0,TFT);
290
291 // Compute the right epipole e1 from F * e1 = 0
292 cvSetZero(U);
293 cvSetZero(S);
294 cvSetZero(V);
295 cvSVD(TFT,S,U,V);
296 scale = sqrt(cvGetReal2D(V,0,2)*cvGetReal2D(V,0,2) + cvGetReal2D(V,1,2)*cvGetReal2D(V,1,2));
297 cvSetReal2D(e1,0,0,cvGetReal2D(V,0,2)/scale);
298 cvSetReal2D(e1,1,0,cvGetReal2D(V,1,2)/scale);
299 cvSetReal2D(e1,2,0,cvGetReal2D(V,2,2)/scale);
300 if (cvGetReal2D(e1,2,0) < 0) {
301 cvSetReal2D(e1,0,0,-cvGetReal2D(e1,0,0));
302 cvSetReal2D(e1,1,0,-cvGetReal2D(e1,1,0));
303 cvSetReal2D(e1,2,0,-cvGetReal2D(e1,2,0));
304 }
305
306 // Compute the left epipole e2 from e2' * F = 0 => F' * e2 = 0
307 cvSetZero(TFTt);
308 cvTranspose(TFT, TFTt);
309 cvSetZero(U);
310 cvSetZero(S);
311 cvSetZero(V);
312 cvSVD(TFTt,S,U,V);
313 cvSetZero(e2);
314 scale = sqrt(cvGetReal2D(V,0,2)*cvGetReal2D(V,0,2) + cvGetReal2D(V,1,2)*cvGetReal2D(V,1,2));
315 cvSetReal2D(e2,0,0,cvGetReal2D(V,0,2)/scale);
316 cvSetReal2D(e2,1,0,cvGetReal2D(V,1,2)/scale);
317 cvSetReal2D(e2,2,0,cvGetReal2D(V,2,2)/scale);
318 if (cvGetReal2D(e2,2,0) < 0) {
319 cvSetReal2D(e2,0,0,-cvGetReal2D(e2,0,0));
320 cvSetReal2D(e2,1,0,-cvGetReal2D(e2,1,0));
321 cvSetReal2D(e2,2,0,-cvGetReal2D(e2,2,0));
322 }
323
324 // Replace F by R2 * F * R1'
325 cvSetZero(R1);
326 cvSetReal2D(R1,0,0,cvGetReal2D(e1,0,0));
327 cvSetReal2D(R1,0,1,cvGetReal2D(e1,1,0));
328 cvSetReal2D(R1,1,0,-cvGetReal2D(e1,1,0));
329 cvSetReal2D(R1,1,1,cvGetReal2D(e1,0,0));
330 cvSetReal2D(R1,2,2,1);
331 cvSetZero(R2);
332 cvSetReal2D(R2,0,0,cvGetReal2D(e2,0,0));
333 cvSetReal2D(R2,0,1,cvGetReal2D(e2,1,0));
334 cvSetReal2D(R2,1,0,-cvGetReal2D(e2,1,0));
335 cvSetReal2D(R2,1,1,cvGetReal2D(e2,0,0));
336 cvSetReal2D(R2,2,2,1);
337 cvGEMM(R2,TFT,1,0,0,tmp33);
338 cvGEMM(tmp33,R1,1,0,0,RTFTR,CV_GEMM_B_T);
339
340 // Set f1 = e1(3), f2 = e2(3), a = F22, b = F23, c = F32, d = F33
341 f1 = cvGetReal2D(e1,2,0);
342 f2 = cvGetReal2D(e2,2,0);
343 a = cvGetReal2D(RTFTR,1,1);
344 b = cvGetReal2D(RTFTR,1,2);
345 c = cvGetReal2D(RTFTR,2,1);
346 d = cvGetReal2D(RTFTR,2,2);
347
348 // Form the polynomial g(t) = k6*t⁶ + k5*t⁵ + k4*t⁴ + k3*t³ + k2*t² + k1*t + k0
349 // from f1, f2, a, b, c and d
350 cvSetReal2D(polynomial,0,6,( +b*c*c*f1*f1*f1*f1*a-a*a*d*f1*f1*f1*f1*c ));
351 cvSetReal2D(polynomial,0,5,( +f2*f2*f2*f2*c*c*c*c+2*a*a*f2*f2*c*c-a*a*d*d*f1*f1*f1*f1+b*b*c*c*f1*f1*f1*f1+a*a*a*a ));
352 cvSetReal2D(polynomial,0,4,( +4*a*a*a*b+2*b*c*c*f1*f1*a+4*f2*f2*f2*f2*c*c*c*d+4*a*b*f2*f2*c*c+4*a*a*f2*f2*c*d-2*a*a*d*f1*f1*c-a*d*d*f1*f1*f1*f1*b+b*b*c*f1*f1*f1*f1*d ));
353 cvSetReal2D(polynomial,0,3,( +6*a*a*b*b+6*f2*f2*f2*f2*c*c*d*d+2*b*b*f2*f2*c*c+2*a*a*f2*f2*d*d-2*a*a*d*d*f1*f1+2*b*b*c*c*f1*f1+8*a*b*f2*f2*c*d ));
354 cvSetReal2D(polynomial,0,2,( +4*a*b*b*b+4*b*b*f2*f2*c*d+4*f2*f2*f2*f2*c*d*d*d-a*a*d*c+b*c*c*a+4*a*b*f2*f2*d*d-2*a*d*d*f1*f1*b+2*b*b*c*f1*f1*d ));
355 cvSetReal2D(polynomial,0,1,( +f2*f2*f2*f2*d*d*d*d+b*b*b*b+2*b*b*f2*f2*d*d-a*a*d*d+b*b*c*c ));
356 cvSetReal2D(polynomial,0,0,( -a*d*d*b+b*b*c*d ));
357
358 // Solve g(t) for t to get 6 roots
359 cvSetZero(result);
360 cvSolvePoly(polynomial, result, 100, 20);
361
362 // Evaluate the cost function s(t) at the real part of the 6 roots
363 t_min = DBL_MAX;
364 s_val = 1./(f1*f1) + (c*c)/(a*a+f2*f2*c*c);
365 for (int ti = 0; ti < 6; ++ti) {
366 t = result->data.db[2*ti];
367 s = (t*t)/(1 + f1*f1*t*t) + ((c*t + d)*(c*t + d))/((a*t + b)*(a*t + b) + f2*f2*(c*t + d)*(c*t + d));
368 if (s < s_val) {
369 s_val = s;
370 t_min = t;
371 }
372 }
373
374 // find the optimal x1 and y1 as the points on l1 and l2 closest to the origin
375 tmp31->data.db[0] = t_min*t_min*f1;
376 tmp31->data.db[1] = t_min;
377 tmp31->data.db[2] = t_min*t_min*f1*f1+1;
378 tmp31->data.db[0] /= tmp31->data.db[2];
379 tmp31->data.db[1] /= tmp31->data.db[2];
380 tmp31->data.db[2] /= tmp31->data.db[2];
381 cvGEMM(T1i,R1,1,0,0,tmp33,CV_GEMM_B_T);
382 cvGEMM(tmp33,tmp31,1,0,0,tmp31_2);
383 x1 = tmp31_2->data.db[0];
384 y1 = tmp31_2->data.db[1];
385
386 tmp31->data.db[0] = f2*pow(c*t_min+d,2);
387 tmp31->data.db[1] = -(a*t_min+b)*(c*t_min+d);
388 tmp31->data.db[2] = f2*f2*pow(c*t_min+d,2) + pow(a*t_min+b,2);
389 tmp31->data.db[0] /= tmp31->data.db[2];
390 tmp31->data.db[1] /= tmp31->data.db[2];
391 tmp31->data.db[2] /= tmp31->data.db[2];
392 cvGEMM(T2i,R2,1,0,0,tmp33,CV_GEMM_B_T);
393 cvGEMM(tmp33,tmp31,1,0,0,tmp31_2);
394 x2 = tmp31_2->data.db[0];
395 y2 = tmp31_2->data.db[1];
396
397 // Return the points in the matrix format that the user wants
398 points1->data.db[p*2] = x1;
399 points1->data.db[p*2+1] = y1;
400 points2->data.db[p*2] = x2;
401 points2->data.db[p*2+1] = y2;
402 }
403
404 if( new_points1 )
405 cvConvert( points1, new_points1 );
406 if( new_points2 )
407 cvConvert( points2, new_points2 );
408}

410void cv::triangulatePoints( InputArray _projMatr1, InputArray _projMatr2,
411 InputArray _projPoints1, InputArray _projPoints2,
412 OutputArray _points4D )
413{
414 Mat matr1 = _projMatr1.getMat(), matr2 = _projMatr2.getMat();
415 Mat points1 = _projPoints1.getMat(), points2 = _projPoints2.getMat();
416
417 if((points1.rows == 1 || points1.cols == 1) && points1.channels() == 2)
418 points1 = points1.reshape(1, static_cast<int>(points1.total())).t();
419
420 if((points2.rows == 1 || points2.cols == 1) && points2.channels() == 2)
421 points2 = points2.reshape(1, static_cast<int>(points2.total())).t();
422
423 CvMat cvMatr1 = matr1, cvMatr2 = matr2;
424 CvMat cvPoints1 = points1, cvPoints2 = points2;
425
426 _points4D.create(4, points1.cols, points1.type());
427 CvMat cvPoints4D = _points4D.getMat();
428
429 cvTriangulatePoints(&cvMatr1, &cvMatr2, &cvPoints1, &cvPoints2, &cvPoints4D);
430}

432void cv::correctMatches( InputArray _F, InputArray _points1, InputArray _points2,
433 OutputArray _newPoints1, OutputArray _newPoints2 )
434{
435 Mat F = _F.getMat();
436 Mat points1 = _points1.getMat(), points2 = _points2.getMat();
437
438 CvMat cvPoints1 = points1, cvPoints2 = points2;
439 CvMat cvF = F;
440
441 _newPoints1.create(points1.size(), points1.type());
442 _newPoints2.create(points2.size(), points2.type());
443 CvMat cvNewPoints1 = _newPoints1.getMat(), cvNewPoints2 = _newPoints2.getMat();
444
445 cvCorrectMatches(&cvF, &cvPoints1, &cvPoints2, &cvNewPoints1, &cvNewPoints2);
446}
