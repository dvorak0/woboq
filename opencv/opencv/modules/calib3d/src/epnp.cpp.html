46void epnp::choose_control_points(void)
47{
48 // Take C0 as the reference points centroid:
49 cws[0][0] = cws[0][1] = cws[0][2] = 0;
50 for(int i = 0; i < number_of_correspondences; i++)
51 for(int j = 0; j < 3; j++)
52 cws[0][j] += pws[3 * i + j];
53
54 for(int j = 0; j < 3; j++)
55 cws[0][j] /= number_of_correspondences;
56
57
58 // Take C1, C2, and C3 from PCA on the reference points:
59 CvMat * PW0 = cvCreateMat(number_of_correspondences, 3, CV_64F);
60
61 double pw0tpw0[3 * 3], dc[3], uct[3 * 3];
62 CvMat PW0tPW0 = cvMat(3, 3, CV_64F, pw0tpw0);
63 CvMat DC = cvMat(3, 1, CV_64F, dc);
64 CvMat UCt = cvMat(3, 3, CV_64F, uct);
65
66 for(int i = 0; i < number_of_correspondences; i++)
67 for(int j = 0; j < 3; j++)
68 PW0->data.db[3 * i + j] = pws[3 * i + j] - cws[0][j];
69
70 cvMulTransposed(PW0, &PW0tPW0, 1);
71 cvSVD(&PW0tPW0, &DC, &UCt, 0, CV_SVD_MODIFY_A | CV_SVD_U_T);
72
73 cvReleaseMat(&PW0);
74
75 for(int i = 1; i < 4; i++) {
76 double k = sqrt(dc[i - 1] / number_of_correspondences);
77 for(int j = 0; j < 3; j++)
78 cws[i][j] = cws[0][j] + k * uct[3 * (i - 1) + j];
79 }
80}

82void epnp::compute_barycentric_coordinates(void)
83{
84 double cc[3 * 3], cc_inv[3 * 3];
85 CvMat CC = cvMat(3, 3, CV_64F, cc);
86 CvMat CC_inv = cvMat(3, 3, CV_64F, cc_inv);
87
88 for(int i = 0; i < 3; i++)
89 for(int j = 1; j < 4; j++)
90 cc[3 * i + j - 1] = cws[j][i] - cws[0][i];
91
92 cvInvert(&CC, &CC_inv, CV_SVD);
93 double * ci = cc_inv;
94 for(int i = 0; i < number_of_correspondences; i++) {
95 double * pi = &pws[0] + 3 * i;
96 double * a = &alphas[0] + 4 * i;
97
98 for(int j = 0; j < 3; j++)
99 a[1 + j] =
100 ci[3 * j ] * (pi[0] - cws[0][0]) +
101 ci[3 * j + 1] * (pi[1] - cws[0][1]) +
102 ci[3 * j + 2] * (pi[2] - cws[0][2]);
103 a[0] = 1.0f - a[1] - a[2] - a[3];
104 }
105}

107void epnp::fill_M(CvMat * M,
108 const int row, const double * as, const double u, const double v)
109{
110 double * M1 = M->data.db + row * 12;
111 double * M2 = M1 + 12;
112
113 for(int i = 0; i < 4; i++) {
114 M1[3 * i ] = as[i] * fu;
115 M1[3 * i + 1] = 0.0;
116 M1[3 * i + 2] = as[i] * (uc - u);
117
118 M2[3 * i ] = 0.0;
119 M2[3 * i + 1] = as[i] * fv;
120 M2[3 * i + 2] = as[i] * (vc - v);
121 }
122}

124void epnp::compute_ccs(const double * betas, const double * ut)
125{
126 for(int i = 0; i < 4; i++)
127 ccs[i][0] = ccs[i][1] = ccs[i][2] = 0.0f;
128
129 for(int i = 0; i < 4; i++) {
130 const double * v = ut + 12 * (11 - i);
131 for(int j = 0; j < 4; j++)
132 for(int k = 0; k < 3; k++)
133 ccs[j][k] += betas[i] * v[3 * j + k];
134 }

137void epnp::compute_pcs(void)
138{
139 for(int i = 0; i < number_of_correspondences; i++) {
140 double * a = &alphas[0] + 4 * i;
141 double * pc = &pcs[0] + 3 * i;
142
143 for(int j = 0; j < 3; j++)
144 pc[j] = a[0] * ccs[0][j] + a[1] * ccs[1][j] + a[2] * ccs[2][j] + a[3] * ccs[3][j];
145 }

148void epnp::compute_pose(cv::Mat& R, cv::Mat& t)
149{
150 choose_control_points();
151 compute_barycentric_coordinates();
152
153 CvMat * M = cvCreateMat(2 * number_of_correspondences, 12, CV_64F);
154
155 for(int i = 0; i < number_of_correspondences; i++)
156 fill_M(M, 2 * i, &alphas[0] + 4 * i, us[2 * i], us[2 * i + 1]);
157
158 double mtm[12 * 12], d[12], ut[12 * 12];
159 CvMat MtM = cvMat(12, 12, CV_64F, mtm);
160 CvMat D = cvMat(12, 1, CV_64F, d);
161 CvMat Ut = cvMat(12, 12, CV_64F, ut);
162
163 cvMulTransposed(M, &MtM, 1);
164 cvSVD(&MtM, &D, &Ut, 0, CV_SVD_MODIFY_A | CV_SVD_U_T);
165 cvReleaseMat(&M);
166
167 double l_6x10[6 * 10], rho[6];
168 CvMat L_6x10 = cvMat(6, 10, CV_64F, l_6x10);
169 CvMat Rho = cvMat(6, 1, CV_64F, rho);
170
171 compute_L_6x10(ut, l_6x10);
172 compute_rho(rho);
173
174 double Betas[4][4], rep_errors[4];
175 double Rs[4][3][3], ts[4][3];
176
177 find_betas_approx_1(&L_6x10, &Rho, Betas[1]);
178 gauss_newton(&L_6x10, &Rho, Betas[1]);
179 rep_errors[1] = compute_R_and_t(ut, Betas[1], Rs[1], ts[1]);
180
181 find_betas_approx_2(&L_6x10, &Rho, Betas[2]);
182 gauss_newton(&L_6x10, &Rho, Betas[2]);
183 rep_errors[2] = compute_R_and_t(ut, Betas[2], Rs[2], ts[2]);
184
185 find_betas_approx_3(&L_6x10, &Rho, Betas[3]);
186 gauss_newton(&L_6x10, &Rho, Betas[3]);
187 rep_errors[3] = compute_R_and_t(ut, Betas[3], Rs[3], ts[3]);
188
189 int N = 1;
190 if (rep_errors[2] < rep_errors[1]) N = 2;
191 if (rep_errors[3] < rep_errors[N]) N = 3;
192
193 cv::Mat(3, 1, CV_64F, ts[N]).copyTo(t);
194 cv::Mat(3, 3, CV_64F, Rs[N]).copyTo(R);
195}

220void epnp::estimate_R_and_t(double R[3][3], double t[3])
221{
222 double pc0[3], pw0[3];
223
224 pc0[0] = pc0[1] = pc0[2] = 0.0;
225 pw0[0] = pw0[1] = pw0[2] = 0.0;
226
227 for(int i = 0; i < number_of_correspondences; i++) {
228 const double * pc = &pcs[3 * i];
229 const double * pw = &pws[3 * i];
230
231 for(int j = 0; j < 3; j++) {
232 pc0[j] += pc[j];
233 pw0[j] += pw[j];
234 }
235 }
236 for(int j = 0; j < 3; j++) {
237 pc0[j] /= number_of_correspondences;
238 pw0[j] /= number_of_correspondences;
239 }
240
241 double abt[3 * 3], abt_d[3], abt_u[3 * 3], abt_v[3 * 3];
242 CvMat ABt = cvMat(3, 3, CV_64F, abt);
243 CvMat ABt_D = cvMat(3, 1, CV_64F, abt_d);
244 CvMat ABt_U = cvMat(3, 3, CV_64F, abt_u);
245 CvMat ABt_V = cvMat(3, 3, CV_64F, abt_v);
246
247 cvSetZero(&ABt);
248 for(int i = 0; i < number_of_correspondences; i++) {
249 double * pc = &pcs[3 * i];
250 double * pw = &pws[3 * i];
251
252 for(int j = 0; j < 3; j++) {
253 abt[3 * j ] += (pc[j] - pc0[j]) * (pw[0] - pw0[0]);
254 abt[3 * j + 1] += (pc[j] - pc0[j]) * (pw[1] - pw0[1]);
255 abt[3 * j + 2] += (pc[j] - pc0[j]) * (pw[2] - pw0[2]);
256 }
257 }
258
259 cvSVD(&ABt, &ABt_D, &ABt_U, &ABt_V, CV_SVD_MODIFY_A);
260
261 for(int i = 0; i < 3; i++)
262 for(int j = 0; j < 3; j++)
263 R[i][j] = dot(abt_u + 3 * i, abt_v + 3 * j);
264
265 const double det =
266 R[0][0] * R[1][1] * R[2][2] + R[0][1] * R[1][2] * R[2][0] + R[0][2] * R[1][0] * R[2][1] -
267 R[0][2] * R[1][1] * R[2][0] - R[0][1] * R[1][0] * R[2][2] - R[0][0] * R[1][2] * R[2][1];
268
269 if (det < 0) {
270 R[2][0] = -R[2][0];
271 R[2][1] = -R[2][1];
272 R[2][2] = -R[2][2];
273 }
274
275 t[0] = pc0[0] - dot(R[0], pw0);
276 t[1] = pc0[1] - dot(R[1], pw0);
277 t[2] = pc0[2] - dot(R[2], pw0);
278}

280void epnp::solve_for_sign(void)
281{
282 if (pcs[2] < 0.0) {
283 for(int i = 0; i < 4; i++)
284 for(int j = 0; j < 3; j++)
285 ccs[i][j] = -ccs[i][j];
286
287 for(int i = 0; i < number_of_correspondences; i++) {
288 pcs[3 * i ] = -pcs[3 * i];
289 pcs[3 * i + 1] = -pcs[3 * i + 1];
290 pcs[3 * i + 2] = -pcs[3 * i + 2];
291 }
292 }

295double epnp::compute_R_and_t(const double * ut, const double * betas,
296 double R[3][3], double t[3])
297{

308double epnp::reprojection_error(const double R[3][3], const double t[3])
309{
310 double sum2 = 0.0;
311
312 for(int i = 0; i < number_of_correspondences; i++) {
313 double * pw = &pws[3 * i];
314 double Xc = dot(R[0], pw) + t[0];
315 double Yc = dot(R[1], pw) + t[1];
316 double inv_Zc = 1.0 / (dot(R[2], pw) + t[2]);
317 double ue = uc + fu * Xc * inv_Zc;
318 double ve = vc + fv * Yc * inv_Zc;
319 double u = us[2 * i], v = us[2 * i + 1];
320
321 sum2 += sqrt( (u - ue) * (u - ue) + (v - ve) * (v - ve) );
322 }
323
324 return sum2 / number_of_correspondences;
325}

330void epnp::find_betas_approx_1(const CvMat * L_6x10, const CvMat * Rho,
331 double * betas)
332{
333 double l_6x4[6 * 4], b4[4];
334 CvMat L_6x4 = cvMat(6, 4, CV_64F, l_6x4);
335 CvMat B4 = cvMat(4, 1, CV_64F, b4);
336
337 for(int i = 0; i < 6; i++) {
338 cvmSet(&L_6x4, i, 0, cvmGet(L_6x10, i, 0));
339 cvmSet(&L_6x4, i, 1, cvmGet(L_6x10, i, 1));
340 cvmSet(&L_6x4, i, 2, cvmGet(L_6x10, i, 3));
341 cvmSet(&L_6x4, i, 3, cvmGet(L_6x10, i, 6));
342 }
343
344 cvSolve(&L_6x4, Rho, &B4, CV_SVD);
345
346 if (b4[0] < 0) {
347 betas[0] = sqrt(-b4[0]);
348 betas[1] = -b4[1] / betas[0];
349 betas[2] = -b4[2] / betas[0];
350 betas[3] = -b4[3] / betas[0];
351 } else {
352 betas[0] = sqrt(b4[0]);
353 betas[1] = b4[1] / betas[0];
354 betas[2] = b4[2] / betas[0];
355 betas[3] = b4[3] / betas[0];
356 }
357}

362void epnp::find_betas_approx_2(const CvMat * L_6x10, const CvMat * Rho,
363 double * betas)
364{
365 double l_6x3[6 * 3], b3[3];
366 CvMat L_6x3 = cvMat(6, 3, CV_64F, l_6x3);
367 CvMat B3 = cvMat(3, 1, CV_64F, b3);
368
369 for(int i = 0; i < 6; i++) {
370 cvmSet(&L_6x3, i, 0, cvmGet(L_6x10, i, 0));
371 cvmSet(&L_6x3, i, 1, cvmGet(L_6x10, i, 1));
372 cvmSet(&L_6x3, i, 2, cvmGet(L_6x10, i, 2));
373 }
374
375 cvSolve(&L_6x3, Rho, &B3, CV_SVD);
376
377 if (b3[0] < 0) {
378 betas[0] = sqrt(-b3[0]);
379 betas[1] = (b3[2] < 0) ? sqrt(-b3[2]) : 0.0;
380 } else {
381 betas[0] = sqrt(b3[0]);
382 betas[1] = (b3[2] > 0) ? sqrt(b3[2]) : 0.0;
383 }
384
385 if (b3[1] < 0) betas[0] = -betas[0];
386
387 betas[2] = 0.0;
388 betas[3] = 0.0;
389}

423void epnp::compute_L_6x10(const double * ut, double * l_6x10)
424{
425 const double * v[4];
426
427 v[0] = ut + 12 * 11;
428 v[1] = ut + 12 * 10;
429 v[2] = ut + 12 * 9;
430 v[3] = ut + 12 * 8;
431
432 double dv[4][6][3];
433
434 for(int i = 0; i < 4; i++) {
435 int a = 0, b = 1;
436 for(int j = 0; j < 6; j++) {
437 dv[i][j][0] = v[i][3 * a ] - v[i][3 * b];
438 dv[i][j][1] = v[i][3 * a + 1] - v[i][3 * b + 1];
439 dv[i][j][2] = v[i][3 * a + 2] - v[i][3 * b + 2];
440
441 b++;
442 if (b > 3) {
443 a++;
444 b = a + 1;
445 }
446 }
447 }
448
449 for(int i = 0; i < 6; i++) {
450 double * row = l_6x10 + 10 * i;
451
452 row[0] = dot(dv[0][i], dv[0][i]);
453 row[1] = 2.0f * dot(dv[0][i], dv[1][i]);
454 row[2] = dot(dv[1][i], dv[1][i]);
455 row[3] = 2.0f * dot(dv[0][i], dv[2][i]);
456 row[4] = 2.0f * dot(dv[1][i], dv[2][i]);
457 row[5] = dot(dv[2][i], dv[2][i]);
458 row[6] = 2.0f * dot(dv[0][i], dv[3][i]);
459 row[7] = 2.0f * dot(dv[1][i], dv[3][i]);
460 row[8] = 2.0f * dot(dv[2][i], dv[3][i]);
461 row[9] = dot(dv[3][i], dv[3][i]);
462 }
463}

475void epnp::compute_A_and_b_gauss_newton(const double * l_6x10, const double * rho,
476 const double betas[4], CvMat * A, CvMat * b)
477{
478 for(int i = 0; i < 6; i++) {
479 const double * rowL = l_6x10 + i * 10;
480 double * rowA = A->data.db + i * 4;
481
482 rowA[0] = 2 * rowL[0] * betas[0] + rowL[1] * betas[1] + rowL[3] * betas[2] + rowL[6] * betas[3];
483 rowA[1] = rowL[1] * betas[0] + 2 * rowL[2] * betas[1] + rowL[4] * betas[2] + rowL[7] * betas[3];
484 rowA[2] = rowL[3] * betas[0] + rowL[4] * betas[1] + 2 * rowL[5] * betas[2] + rowL[8] * betas[3];
485 rowA[3] = rowL[6] * betas[0] + rowL[7] * betas[1] + rowL[8] * betas[2] + 2 * rowL[9] * betas[3];
486
487 cvmSet(b, i, 0, rho[i] -
488 (
489 rowL[0] * betas[0] * betas[0] +
490 rowL[1] * betas[0] * betas[1] +
491 rowL[2] * betas[1] * betas[1] +
492 rowL[3] * betas[0] * betas[2] +
493 rowL[4] * betas[1] * betas[2] +
494 rowL[5] * betas[2] * betas[2] +
495 rowL[6] * betas[0] * betas[3] +
496 rowL[7] * betas[1] * betas[3] +
497 rowL[8] * betas[2] * betas[3] +
498 rowL[9] * betas[3] * betas[3]
499 ));
500 }
501}

503void epnp::gauss_newton(const CvMat * L_6x10, const CvMat * Rho, double betas[4])
504{
505 const int iterations_number = 5;
506
507 double a[6*4], b[6], x[4];
508 CvMat A = cvMat(6, 4, CV_64F, a);
509 CvMat B = cvMat(6, 1, CV_64F, b);
510 CvMat X = cvMat(4, 1, CV_64F, x);
511
512 for(int k = 0; k < iterations_number; k++)
513 {
514 compute_A_and_b_gauss_newton(L_6x10->data.db, Rho->data.db,
515 betas, &A, &B);
516 qr_solve(&A, &B, &X);
517 for(int i = 0; i < 4; i++)
518 betas[i] += x[i];
519 }
520}

522void epnp::qr_solve(CvMat * A, CvMat * b, CvMat * X)
523{
524 const int nr = A->rows;
525 const int nc = A->cols;
526
527 if (max_nr != 0 && max_nr < nr)
528 {
529 delete [] A1;
530 delete [] A2;
531 }
532 if (max_nr < nr)
533 {
534 max_nr = nr;
535 A1 = new double[nr];
536 A2 = new double[nr];
537 }
538
539 double * pA = A->data.db, * ppAkk = pA;
540 for(int k = 0; k < nc; k++)
541 {
542 double * ppAik1 = ppAkk, eta = fabs(*ppAik1);
543 for(int i = k + 1; i < nr; i++)
544 {
545 double elt = fabs(*ppAik1);
546 if (eta < elt) eta = elt;
547 ppAik1 += nc;
548 }
549 if (eta == 0)
550 {
551 A1[k] = A2[k] = 0.0;
552 //cerr << "God damnit, A is singular, this shouldn't happen." << endl;
553 return;
554 }
555 else
556 {
557 double * ppAik2 = ppAkk, sum2 = 0.0, inv_eta = 1. / eta;
558 for(int i = k; i < nr; i++)
559 {
560 *ppAik2 *= inv_eta;
561 sum2 += *ppAik2 * *ppAik2;
562 ppAik2 += nc;
563 }
564 double sigma = sqrt(sum2);
565 if (*ppAkk < 0)
566 sigma = -sigma;
567 *ppAkk += sigma;
568 A1[k] = sigma * *ppAkk;
569 A2[k] = -eta * sigma;
570 for(int j = k + 1; j < nc; j++)
571 {
572 double * ppAik = ppAkk, sum = 0;
573 for(int i = k; i < nr; i++)
574 {
575 sum += *ppAik * ppAik[j - k];
576 ppAik += nc;
577 }
578 double tau = sum / A1[k];
579 ppAik = ppAkk;
580 for(int i = k; i < nr; i++)
581 {
582 ppAik[j - k] -= tau * *ppAik;
583 ppAik += nc;
584 }
585 }
586 }
587 ppAkk += nc + 1;
588 }
589
590 // b <- Qt b
591 double * ppAjj = pA, * pb = b->data.db;
592 for(int j = 0; j < nc; j++)
593 {
594 double * ppAij = ppAjj, tau = 0;
595 for(int i = j; i < nr; i++)
596 {
597 tau += *ppAij * pb[i];
598 ppAij += nc;
599 }
600 tau /= A1[j];
601 ppAij = ppAjj;
602 for(int i = j; i < nr; i++)
603 {
604 pb[i] -= tau * *ppAij;
605 ppAij += nc;
606 }
607 ppAjj += nc + 1;
608 }
609
610 // X = R-1 b
611 double * pX = X->data.db;
612 pX[nc - 1] = pb[nc - 1] / A2[nc - 1];
613 for(int i = nc - 2; i >= 0; i--)
614 {
615 double * ppAij = pA + i * nc + (i + 1), sum = 0;
616
617 for(int j = i + 1; j < nc; j++)
618 {
619 sum += *ppAij * pX[j];
620 ppAij++;
621 }
622 pX[i] = (pb[i] - sum) / A2[i];
623 }
