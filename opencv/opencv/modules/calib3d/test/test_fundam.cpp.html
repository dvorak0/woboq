47int cvTsRodrigues( const CvMat* src, CvMat* dst, CvMat* jacobian )
48{
49 int depth;
50 int i;
51 float Jf[27];
52 double J[27];
53 CvMat _Jf, matJ = cvMat( 3, 9, CV_64F, J );
54
55 depth = CV_MAT_DEPTH(src->type);
56
57 if( jacobian )
58 {
59 assert( (jacobian->rows == 9 && jacobian->cols == 3) ||
60 (jacobian->rows == 3 && jacobian->cols == 9) );
61 }
62
63 if( src->cols == 1 || src->rows == 1 )
64 {
65 double r[3], theta;
66 CvMat _r = cvMat( src->rows, src->cols, CV_MAKETYPE(CV_64F,CV_MAT_CN(src->type)), r);
67
68 assert( dst->rows == 3 && dst->cols == 3 );
69
70 cvConvert( src, &_r );
71
72 theta = sqrt(r[0]*r[0] + r[1]*r[1] + r[2]*r[2]);
73 if( theta < DBL_EPSILON )
74 {
75 cvSetIdentity( dst );
76
77 if( jacobian )
78 {
79 memset( J, 0, sizeof(J) );
80 J[5] = J[15] = J[19] = 1;
81 J[7] = J[11] = J[21] = -1;
82 }
83 }
84 else
85 {
86 // omega = r/theta (~[w1, w2, w3])
87 double itheta = 1./theta;
88 double w1 = r[0]*itheta, w2 = r[1]*itheta, w3 = r[2]*itheta;
89 double alpha = cos(theta);
90 double beta = sin(theta);
91 double gamma = 1 - alpha;
92 double omegav[] =
93 {
94 0, -w3, w2,
95 w3, 0, -w1,
96 -w2, w1, 0
97 };
98 double A[] =
99 {
100 w1*w1, w1*w2, w1*w3,
101 w2*w1, w2*w2, w2*w3,
102 w3*w1, w3*w2, w3*w3
103 };
104 double R[9];
105 CvMat _omegav = cvMat(3, 3, CV_64F, omegav);
106 CvMat matA = cvMat(3, 3, CV_64F, A);
107 CvMat matR = cvMat(3, 3, CV_64F, R);
108
109 cvSetIdentity( &matR, cvRealScalar(alpha) );
110 cvScaleAdd( &_omegav, cvRealScalar(beta), &matR, &matR );
111 cvScaleAdd( &matA, cvRealScalar(gamma), &matR, &matR );
112 cvConvert( &matR, dst );
113
114 if( jacobian )
115 {
116 // m3 = [r, theta]
117 double dm3din[] =
118 {
119 1, 0, 0,
120 0, 1, 0,
121 0, 0, 1,
122 w1, w2, w3
123 };
124 // m2 = [omega, theta]
125 double dm2dm3[] =
126 {
127 itheta, 0, 0, -w1*itheta,
128 0, itheta, 0, -w2*itheta,
129 0, 0, itheta, -w3*itheta,
130 0, 0, 0, 1
131 };
132 double t0[9*4];
133 double dm1dm2[21*4];
134 double dRdm1[9*21];
135 CvMat _dm3din = cvMat( 4, 3, CV_64FC1, dm3din );
136 CvMat _dm2dm3 = cvMat( 4, 4, CV_64FC1, dm2dm3 );
137 CvMat _dm1dm2 = cvMat( 21, 4, CV_64FC1, dm1dm2 );
138 CvMat _dRdm1 = cvMat( 9, 21, CV_64FC1, dRdm1 );
139 CvMat _dRdm1_part;
140 CvMat _t0 = cvMat( 9, 4, CV_64FC1, t0 );
141 CvMat _t1 = cvMat( 9, 4, CV_64FC1, dRdm1 );
142
143 // m1 = [alpha, beta, gamma, omegav; A]
144 memset( dm1dm2, 0, sizeof(dm1dm2) );
145 dm1dm2[3] = -beta;
146 dm1dm2[7] = alpha;
147 dm1dm2[11] = beta;
148
149 // dm1dm2(4:12,1:3) = [0 0 0 0 0 1 0 -1 0;
150 // 0 0 -1 0 0 0 1 0 0;
151 // 0 1 0 -1 0 0 0 0 0]'
152 // -------------------
153 // 0 0 0 0 0 0 0 0 0
154 dm1dm2[12 + 6] = dm1dm2[12 + 20] = dm1dm2[12 + 25] = 1;
155 dm1dm2[12 + 9] = dm1dm2[12 + 14] = dm1dm2[12 + 28] = -1;
156
157 double dm1dw[] =
158 {
159 2*w1, w2, w3, w2, 0, 0, w3, 0, 0,
160 0, w1, 0, w1, 2*w2, w3, 0, w3, 0,
161 0, 0, w1, 0, 0, w2, w1, w2, 2*w3
162 };
163
164 CvMat _dm1dw = cvMat( 3, 9, CV_64FC1, dm1dw );
165 CvMat _dm1dm2_part;
166
167 cvGetSubRect( &_dm1dm2, &_dm1dm2_part, cvRect(0,12,3,9) );
168 cvTranspose( &_dm1dw, &_dm1dm2_part );
169
170 memset( dRdm1, 0, sizeof(dRdm1) );
171 dRdm1[0*21] = dRdm1[4*21] = dRdm1[8*21] = 1;
172
173 cvGetCol( &_dRdm1, &_dRdm1_part, 1 );
174 cvTranspose( &_omegav, &_omegav );
175 cvReshape( &_omegav, &_omegav, 1, 1 );
176 cvTranspose( &_omegav, &_dRdm1_part );
177
178 cvGetCol( &_dRdm1, &_dRdm1_part, 2 );
179 cvReshape( &matA, &matA, 1, 1 );
180 cvTranspose( &matA, &_dRdm1_part );
181
182 cvGetSubRect( &_dRdm1, &_dRdm1_part, cvRect(3,0,9,9) );
183 cvSetIdentity( &_dRdm1_part, cvScalarAll(beta) );
184
185 cvGetSubRect( &_dRdm1, &_dRdm1_part, cvRect(12,0,9,9) );
186 cvSetIdentity( &_dRdm1_part, cvScalarAll(gamma) );
187
188 matJ = cvMat( 9, 3, CV_64FC1, J );
189
190 cvMatMul( &_dRdm1, &_dm1dm2, &_t0 );
191 cvMatMul( &_t0, &_dm2dm3, &_t1 );
192 cvMatMul( &_t1, &_dm3din, &matJ );
193
194 _t0 = cvMat( 3, 9, CV_64FC1, t0 );
195 cvTranspose( &matJ, &_t0 );
196
197 for( i = 0; i < 3; i++ )
198 {
199 _t1 = cvMat( 3, 3, CV_64FC1, t0 + i*9 );
200 cvTranspose( &_t1, &_t1 );
201 }
202
203 cvTranspose( &_t0, &matJ );
204 }
205 }
206 }
207 else if( src->cols == 3 && src->rows == 3 )
208 {
209 double R[9], A[9], I[9], r[3], W[3], U[9], V[9];
210 double tr, alpha, beta, theta;
211 CvMat matR = cvMat( 3, 3, CV_64F, R );
212 CvMat matA = cvMat( 3, 3, CV_64F, A );
213 CvMat matI = cvMat( 3, 3, CV_64F, I );
214 CvMat _r = cvMat( dst->rows, dst->cols, CV_MAKETYPE(CV_64F, CV_MAT_CN(dst->type)), r );
215 CvMat matW = cvMat( 1, 3, CV_64F, W );
216 CvMat matU = cvMat( 3, 3, CV_64F, U );
217 CvMat matV = cvMat( 3, 3, CV_64F, V );
218
219 cvConvert( src, &matR );
220 cvSVD( &matR, &matW, &matU, &matV, CV_SVD_MODIFY_A + CV_SVD_U_T + CV_SVD_V_T );
221 cvGEMM( &matU, &matV, 1, 0, 0, &matR, CV_GEMM_A_T );
222
223 cvMulTransposed( &matR, &matA, 0 );
224 cvSetIdentity( &matI );
225
226 if( cvNorm( &matA, &matI, CV_C ) > 1e-3 ||
227 fabs( cvDet(&matR) - 1 ) > 1e-3 )
228 return 0;
229
230 tr = (cvTrace(&matR).val[0] - 1.)*0.5;
231 tr = tr > 1. ? 1. : tr < -1. ? -1. : tr;
232 theta = acos(tr);
233 alpha = cos(theta);
234 beta = sin(theta);
235
236 if( beta >= 1e-5 )
237 {
238 double dtheta_dtr = -1./sqrt(1 - tr*tr);
239 double vth = 1/(2*beta);
240
241 // om1 = [R(3,2) - R(2,3), R(1,3) - R(3,1), R(2,1) - R(1,2)]'
242 double om1[] = { R[7] - R[5], R[2] - R[6], R[3] - R[1] };
243 // om = om1*vth
244 // r = om*theta
245 double d3 = vth*theta;
246
247 r[0] = om1[0]*d3; r[1] = om1[1]*d3; r[2] = om1[2]*d3;
248 cvConvert( &_r, dst );
249
250 if( jacobian )
251 {
252 // var1 = [vth;theta]
253 // var = [om1;var1] = [om1;vth;theta]
254 double dvth_dtheta = -vth*alpha/beta;
255 double d1 = 0.5*dvth_dtheta*dtheta_dtr;
256 double d2 = 0.5*dtheta_dtr;
257 // dvar1/dR = dvar1/dtheta*dtheta/dR = [dvth/dtheta; 1] * dtheta/dtr * dtr/dR
258 double dvardR[5*9] =
259 {
260 0, 0, 0, 0, 0, 1, 0, -1, 0,
261 0, 0, -1, 0, 0, 0, 1, 0, 0,
262 0, 1, 0, -1, 0, 0, 0, 0, 0,
263 d1, 0, 0, 0, d1, 0, 0, 0, d1,
264 d2, 0, 0, 0, d2, 0, 0, 0, d2
265 };
266 // var2 = [om;theta]
267 double dvar2dvar[] =
268 {
269 vth, 0, 0, om1[0], 0,
270 0, vth, 0, om1[1], 0,
271 0, 0, vth, om1[2], 0,
272 0, 0, 0, 0, 1
273 };
274 double domegadvar2[] =
275 {
276 theta, 0, 0, om1[0]*vth,
277 0, theta, 0, om1[1]*vth,
278 0, 0, theta, om1[2]*vth
279 };
280
281 CvMat _dvardR = cvMat( 5, 9, CV_64FC1, dvardR );
282 CvMat _dvar2dvar = cvMat( 4, 5, CV_64FC1, dvar2dvar );
283 CvMat _domegadvar2 = cvMat( 3, 4, CV_64FC1, domegadvar2 );
284 double t0[3*5];
285 CvMat _t0 = cvMat( 3, 5, CV_64FC1, t0 );
286
287 cvMatMul( &_domegadvar2, &_dvar2dvar, &_t0 );
288 cvMatMul( &_t0, &_dvardR, &matJ );
289 }
290 }
291 else if( tr > 0 )
292 {
293 cvZero( dst );
294 if( jacobian )
295 {
296 memset( J, 0, sizeof(J) );
297 J[5] = J[15] = J[19] = 0.5;
298 J[7] = J[11] = J[21] = -0.5;
299 }
300 }
301 else
302 {
303 r[0] = theta*sqrt((R[0] + 1)*0.5);
304 r[1] = theta*sqrt((R[4] + 1)*0.5)*(R[1] >= 0 ? 1 : -1);
305 r[2] = theta*sqrt((R[8] + 1)*0.5)*(R[2] >= 0 ? 1 : -1);
306 cvConvert( &_r, dst );
307
308 if( jacobian )
309 memset( J, 0, sizeof(J) );
310 }
311
312 if( jacobian )
313 {
314 for( i = 0; i < 3; i++ )
315 {
316 CvMat t = cvMat( 3, 3, CV_64F, J + i*9 );
317 cvTranspose( &t, &t );
318 }
319 }
320 }
321 else
322 {
323 assert(0);
324 return 0;
325 }
326
327 if( jacobian )
328 {
329 if( depth == CV_32F )
330 {
331 if( jacobian->rows == matJ.rows )
332 cvConvert( &matJ, jacobian );
333 else
334 {
335 _Jf = cvMat( matJ.rows, matJ.cols, CV_32FC1, Jf );
336 cvConvert( &matJ, &_Jf );
337 cvTranspose( &_Jf, jacobian );
338 }
339 }
340 else if( jacobian->rows == matJ.rows )
341 cvCopy( &matJ, jacobian );
342 else
343 cvTranspose( &matJ, jacobian );
344 }
345
346 return 1;
347}

359static void test_convertHomogeneous( const Mat& _src, Mat& _dst )
360{
361 Mat src = _src, dst = _dst;
362 int i, count, sdims, ddims;
363 int sstep1, sstep2, dstep1, dstep2;
364
365 if( src.depth() != CV_64F )
366 _src.convertTo(src, CV_64F);
367
368 if( dst.depth() != CV_64F )
369 dst.create(dst.size(), CV_MAKETYPE(CV_64F, _dst.channels()));
370
371 if( src.rows > src.cols )
372 {
373 count = src.rows;
374 sdims = src.channels()*src.cols;
375 sstep1 = (int)(src.step/sizeof(double));
376 sstep2 = 1;
377 }
378 else
379 {
380 count = src.cols;
381 sdims = src.channels()*src.rows;
382 if( src.rows == 1 )
383 {
384 sstep1 = sdims;
385 sstep2 = 1;
386 }
387 else
388 {
389 sstep1 = 1;
390 sstep2 = (int)(src.step/sizeof(double));
391 }
392 }
393
394 if( dst.rows > dst.cols )
395 {
396 CV_Assert( count == dst.rows );
397 ddims = dst.channels()*dst.cols;
398 dstep1 = (int)(dst.step/sizeof(double));
399 dstep2 = 1;
400 }
401 else
402 {
403 assert( count == dst.cols );
404 ddims = dst.channels()*dst.rows;
405 if( dst.rows == 1 )
406 {
407 dstep1 = ddims;
408 dstep2 = 1;
409 }
410 else
411 {
412 dstep1 = 1;
413 dstep2 = (int)(dst.step/sizeof(double));
414 }
415 }
416
417 double* s = src.ptr<double>();
418 double* d = dst.ptr<double>();
419
420 if( sdims <= ddims )
421 {
422 int wstep = dstep2*(ddims - 1);
423
424 for( i = 0; i < count; i++, s += sstep1, d += dstep1 )
425 {
426 double x = s[0];
427 double y = s[sstep2];
428
429 d[wstep] = 1;
430 d[0] = x;
431 d[dstep2] = y;
432
433 if( sdims >= 3 )
434 {
435 d[dstep2*2] = s[sstep2*2];
436 if( sdims == 4 )
437 d[dstep2*3] = s[sstep2*3];
438 }
439 }
440 }
441 else
442 {
443 int wstep = sstep2*(sdims - 1);
444
445 for( i = 0; i < count; i++, s += sstep1, d += dstep1 )
446 {
447 double w = s[wstep];
448 double x = s[0];
449 double y = s[sstep2];
450
451 w = w ? 1./w : 1;
452
453 d[0] = x*w;
454 d[dstep2] = y*w;
455
456 if( ddims == 3 )
457 d[dstep2*2] = s[sstep2*2]*w;
458 }
459 }
460
461 if( dst.data != _dst.data )
462 dst.convertTo(_dst, _dst.depth());
463}

466void
467test_projectPoints( const Mat& _3d, const Mat& Rt, const Mat& A, Mat& _2d, RNG* rng, double sigma )
468{
469 CV_Assert( _3d.isContinuous() );
470
471 double p[12];
472 Mat P( 3, 4, CV_64F, p );
473 gemm(A, Rt, 1, Mat(), 0, P);
474
475 int i, count = _3d.cols;
476
477 Mat noise;
478 if( rng )
479 {
480 if( sigma == 0 )
481 rng = 0;
482 else
483 {
484 noise.create( 1, _3d.cols, CV_64FC2 );
485 rng->fill(noise, RNG::NORMAL, Scalar::all(0), Scalar::all(sigma) );
486 }
487 }
488
489 Mat temp( 1, count, CV_64FC3 );
490
491 for( i = 0; i < count; i++ )
492 {
493 const double* M = _3d.ptr<double>() + i*3;
494 double* m = temp.ptr<double>() + i*3;
495 double X = M[0], Y = M[1], Z = M[2];
496 double u = p[0]*X + p[1]*Y + p[2]*Z + p[3];
497 double v = p[4]*X + p[5]*Y + p[6]*Z + p[7];
498 double s = p[8]*X + p[9]*Y + p[10]*Z + p[11];
499
500 if( !noise.empty() )
501 {
502 u += noise.at<Point2d>(i).x*s;
503 v += noise.at<Point2d>(i).y*s;
504 }
505
506 m[0] = u;
507 m[1] = v;
508 m[2] = s;
509 }
510
511 test_convertHomogeneous( temp, _2d );
512}

621
622void CV_RodriguesTest::fill_array( int test_case_idx, int i, int j, Mat& arr )
623{
624 if( i == INPUT && j == 0 )
625 {
626 double r[3], theta0, theta1, f;
627 Mat _r( arr.rows, arr.cols, CV_MAKETYPE(CV_64F,arr.channels()), r );
628 RNG& rng = ts->get_rng();
629
630 r[0] = cvtest::randReal(rng)*CV_PI*2;
631 r[1] = cvtest::randReal(rng)*CV_PI*2;
632 r[2] = cvtest::randReal(rng)*CV_PI*2;
633
634 theta0 = sqrt(r[0]*r[0] + r[1]*r[1] + r[2]*r[2]);
635 theta1 = fmod(theta0, CV_PI*2);
636
637 if( theta1 > CV_PI )
638 theta1 = -(CV_PI*2 - theta1);
639
640 f = theta1/(theta0 ? theta0 : 1);
641 r[0] *= f;
642 r[1] *= f;
643 r[2] *= f;
644
645 cvtest::convert( _r, arr, arr.type() );
646 }
647 else

997void CV_FundamentalMatTest::prepare_to_validation( int test_case_idx )
998{
999 const Mat& Rt = test_mat[INPUT][3];
1000 const Mat& A1 = test_mat[INPUT][4];
1001 const Mat& A2 = test_mat[INPUT][5];
1002 double f0[9], f[9];
1003 Mat F0(3, 3, CV_64FC1, f0), F(3, 3, CV_64F, f);
1004
1005 Mat invA1, invA2, R=Rt.colRange(0, 3), T;
1006
1007 cv::invert(A1, invA1, CV_SVD);
1008 cv::invert(A2, invA2, CV_SVD);
1009
1010 double tx = Rt.at<double>(0, 3);
1011 double ty = Rt.at<double>(1, 3);
1012 double tz = Rt.at<double>(2, 3);
1013
1014 double _t_x[] = { 0, -tz, ty, tz, 0, -tx, -ty, tx, 0 };
1015
1016 // F = (A2^-T)*[t]_x*R*(A1^-1)
1017 cv::gemm( invA2, Mat( 3, 3, CV_64F, _t_x ), 1, Mat(), 0, T, CV_GEMM_A_T );
1018 cv::gemm( R, invA1, 1, Mat(), 0, invA2 );
1019 cv::gemm( T, invA2, 1, Mat(), 0, F0 );
1020 F0 *= 1./f0[8];
1021
1022 uchar* status = test_mat[TEMP][1].data;
1023 double err_level = get_success_error_level( test_case_idx, OUTPUT, 1 );
1024 uchar* mtfm1 = test_mat[REF_OUTPUT][1].data;
1025 uchar* mtfm2 = test_mat[OUTPUT][1].data;
1026 double* f_prop1 = (double*)test_mat[REF_OUTPUT][0].data;
1027 double* f_prop2 = (double*)test_mat[OUTPUT][0].data;
1028
1029 int i, pt_count = test_mat[INPUT][2].cols;
1030 Mat p1( 1, pt_count, CV_64FC2 );
1031 Mat p2( 1, pt_count, CV_64FC2 );
1032
1033 test_convertHomogeneous( test_mat[INPUT][0], p1 );
1034 test_convertHomogeneous( test_mat[INPUT][1], p2 );
1035
1036 cvtest::convert(test_mat[TEMP][0], F, F.type());
1037
1038 if( method <= CV_FM_8POINT )
1039 memset( status, 1, pt_count );
1040
1041 for( i = 0; i < pt_count; i++ )
1042 {
1043 double x1 = p1.at<Point2d>(i).x;
1044 double y1 = p1.at<Point2d>(i).y;
1045 double x2 = p2.at<Point2d>(i).x;
1046 double y2 = p2.at<Point2d>(i).y;
1047 double n1 = 1./sqrt(x1*x1 + y1*y1 + 1);
1048 double n2 = 1./sqrt(x2*x2 + y2*y2 + 1);
1049 double t0 = fabs(f0[0]*x2*x1 + f0[1]*x2*y1 + f0[2]*x2 +
1050 f0[3]*y2*x1 + f0[4]*y2*y1 + f0[5]*y2 +
1051 f0[6]*x1 + f0[7]*y1 + f0[8])*n1*n2;
1052 double t = fabs(f[0]*x2*x1 + f[1]*x2*y1 + f[2]*x2 +
1053 f[3]*y2*x1 + f[4]*y2*y1 + f[5]*y2 +
1054 f[6]*x1 + f[7]*y1 + f[8])*n1*n2;
1055 mtfm1[i] = 1;
1056 mtfm2[i] = !status[i] || t0 > err_level || t < err_level;
1057 }
1058
1059 f_prop1[0] = 1;
1060 f_prop1[1] = 1;
1061 f_prop1[2] = 0;
1062
1063 f_prop2[0] = f_result != 0;
1064 f_prop2[1] = f[8];
1065 f_prop2[2] = cv::determinant( F );
1066}
