42#include "test_precomp.hpp"
43#include "test_chessboardgenerator.hpp"

45#include <limits>

47using namespace std;
48using namespace cv;
49
50class CV_ChessboardDetectorBadArgTest : public cvtest::BadArgTest
51{
52public:
53 CV_ChessboardDetectorBadArgTest();
54protected:
55 void run(int);
56 bool checkByGenerator();
57
58 bool cpp;
59
60 /* cpp interface */
61 Mat img;
62 Size pattern_size;
63 int flags;
64 vector<Point2f> corners;
65
66 /* c interface */
67 CvMat arr;
68 CvPoint2D32f* out_corners;
69 int* out_corner_count;
70
71
72 /* c interface draw corners */
73 bool drawCorners;
74 CvMat drawCorImg;
75 bool was_found;
76
77 void run_func()
78 {
79 if (cpp)
80 findChessboardCorners(img, pattern_size, corners, flags);
81 else
82 if (!drawCorners)
83 cvFindChessboardCorners( &arr, pattern_size, out_corners, out_corner_count, flags );
84 else
85 cvDrawChessboardCorners( &drawCorImg, pattern_size,
86 (CvPoint2D32f*)(corners.empty() ? 0 : &corners[0]),
87 (int)corners.size(), was_found);
88 }
89};
90
91CV_ChessboardDetectorBadArgTest::CV_ChessboardDetectorBadArgTest() {}
92
93/* ///////////////////// chess_corner_test ///////////////////////// */
94void CV_ChessboardDetectorBadArgTest::run( int /*start_from */)
95{
96 Mat bg(800, 600, CV_8U, Scalar(0));
97 Mat_<float> camMat(3, 3);
98 camMat << 300.f, 0.f, bg.cols/2.f, 0, 300.f, bg.rows/2.f, 0.f, 0.f, 1.f;
99 Mat_<float> distCoeffs(1, 5);
100 distCoeffs << 1.2f, 0.2f, 0.f, 0.f, 0.f;
101
102 ChessBoardGenerator cbg(Size(8,6));
103 vector<Point2f> exp_corn;
104 Mat cb = cbg(bg, camMat, distCoeffs, exp_corn);
105
106 /* /*//*/ */
107 int errors = 0;
108 flags = CV_CALIB_CB_ADAPTIVE_THRESH | CV_CALIB_CB_NORMALIZE_IMAGE;
109 cpp = true;
110
111 img = cb.clone();
112 pattern_size = Size(2,2);
113 errors += run_test_case( CV_StsOutOfRange, "Invlid pattern size" );
114
115 pattern_size = cbg.cornersSize();
116 cb.convertTo(img, CV_32F);
117 errors += run_test_case( CV_StsUnsupportedFormat, "Not 8-bit image" );
118
119 cv::merge(vector<Mat>(2, cb), img);
120 errors += run_test_case( CV_StsUnsupportedFormat, "2 channel image" );
121
122 cpp = false;
123 drawCorners = false;
124
125 img = cb.clone();
126 arr = img;
127 out_corner_count = 0;
128 out_corners = 0;
129 errors += run_test_case( CV_StsNullPtr, "Null pointer to corners" );
130
131 drawCorners = true;
132 Mat cvdrawCornImg(img.size(), CV_8UC2);
133 drawCorImg = cvdrawCornImg;
134 was_found = true;
135 errors += run_test_case( CV_StsUnsupportedFormat, "2 channel image" );
136
137
138 if (errors)
139 ts->set_failed_test_info(cvtest::TS::FAIL_MISMATCH);
140 else
141 ts->set_failed_test_info(cvtest::TS::OK);
142}

144TEST(Calib3d_ChessboardDetector, badarg) { CV_ChessboardDetectorBadArgTest test; test.safe_run(); }
