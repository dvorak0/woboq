57/* ///////////////////// chess_corner_test ///////////////////////// */
58void CV_ChessboardDetectorTimingTest::run( int start_from )
59{
60 int code = cvtest::TS::OK;
61
62 /* test parameters */
63 char filepath[1000];
64 char filename[1000];
65
66 CvMat* _v = 0;
67 CvPoint2D32f* v;
68
69 IplImage* img = 0;
70 IplImage* gray = 0;
71 IplImage* thresh = 0;
72
73 int idx, max_idx;
74 int progress = 0;
75
76 sprintf( filepath, "%scameracalibration/", ts->get_data_path().c_str() );
77 sprintf( filename, "%schessboard_timing_list.dat", filepath );
78 CvFileStorage* fs = cvOpenFileStorage( filename, 0, CV_STORAGE_READ );
79 CvFileNode* board_list = fs ? cvGetFileNodeByName( fs, 0, "boards" ) : 0;
80
81 if( !fs || !board_list || !CV_NODE_IS_SEQ(board_list->tag) ||
82 board_list->data.seq->total % 4 != 0 )
83 {
84 ts->printf( cvtest::TS::LOG, "chessboard_timing_list.dat can not be readed or is not valid" );
85 code = cvtest::TS::FAIL_MISSING_TEST_DATA;
86 goto _exit_;
87 }
88
89 max_idx = board_list->data.seq->total/4;
90
91 for( idx = start_from; idx < max_idx; idx++ )
92 {
93 int count0 = -1;
94 int count = 0;
95 CvSize pattern_size;
96 int result, result1 = 0;
97
98 const char* imgname = cvReadString((CvFileNode*)cvGetSeqElem(board_list->data.seq,idx*4), "dummy.txt");
99 int is_chessboard = cvReadInt((CvFileNode*)cvGetSeqElem(board_list->data.seq,idx*4+1), 0);
100 pattern_size.width = cvReadInt((CvFileNode*)cvGetSeqElem(board_list->data.seq,idx*4 + 2), -1);
101 pattern_size.height = cvReadInt((CvFileNode*)cvGetSeqElem(board_list->data.seq,idx*4 + 3), -1);
102
103 ts->update_context( this, idx-1, true );
104
105 /* read the image */
106 sprintf( filename, "%s%s", filepath, imgname );
107
108 img = cvLoadImage( filename );
109
110 if( !img )
111 {
112 ts->printf( cvtest::TS::LOG, "one of chessboard images can't be read: %s\n", filename );
113 code = cvtest::TS::FAIL_MISSING_TEST_DATA;
114 continue;
115 }
116
117 ts->printf(cvtest::TS::LOG, "%s: chessboard %d:\n", imgname, is_chessboard);
118
119 gray = cvCreateImage( cvSize( img->width, img->height ), IPL_DEPTH_8U, 1 );
120 thresh = cvCreateImage( cvSize( img->width, img->height ), IPL_DEPTH_8U, 1 );
121 cvCvtColor( img, gray, CV_BGR2GRAY );
122
123
124 count0 = pattern_size.width*pattern_size.height;
125
126 /* allocate additional buffers */
127 _v = cvCreateMat(1, count0, CV_32FC2);
128 count = count0;
129
130 v = (CvPoint2D32f*)_v->data.fl;
131
132 int64 _time0 = cvGetTickCount();
133 result = cvCheckChessboard(gray, pattern_size);
134 int64 _time01 = cvGetTickCount();
135
136 OPENCV_CALL( result1 = cvFindChessboardCorners(
137 gray, pattern_size, v, &count, 15 ));
138 int64 _time1 = cvGetTickCount();
139
140 if( result != is_chessboard )
141 {
142 ts->printf( cvtest::TS::LOG, "Error: chessboard was %sdetected in the image %s\n",
143 result ? "" : "not ", imgname );
144 code = cvtest::TS::FAIL_INVALID_OUTPUT;
145 goto _exit_;
146 }
147 if(result != result1)
148 {
149 ts->printf( cvtest::TS::LOG, "Warning: results differ cvCheckChessboard %d, cvFindChessboardCorners %d\n",
150 result, result1);
151 }
152
153 int num_pixels = gray->width*gray->height;
154 float check_chessboard_time = float(_time01 - _time0)/(float)cvGetTickFrequency(); // in us
155 ts->printf(cvtest::TS::LOG, " cvCheckChessboard time s: %f, us per pixel: %f\n",
156 check_chessboard_time*1e-6, check_chessboard_time/num_pixels);
157
158 float find_chessboard_time = float(_time1 - _time01)/(float)cvGetTickFrequency();
159 ts->printf(cvtest::TS::LOG, " cvFindChessboard time s: %f, us per pixel: %f\n",
160 find_chessboard_time*1e-6, find_chessboard_time/num_pixels);
161
162 cvReleaseMat( &_v );
163 cvReleaseImage( &img );
164 cvReleaseImage( &gray );
165 cvReleaseImage( &thresh );
166 progress = update_progress( progress, idx-1, max_idx, 0 );
167 }
168
169_exit_:
170
171 /* release occupied memory */
172 cvReleaseMat( &_v );
173 cvReleaseFileStorage( &fs );
174 cvReleaseImage( &img );
175 cvReleaseImage( &gray );
176 cvReleaseImage( &thresh );
177
178 if( code < 0 )
179 ts->set_failed_test_info( code );
180}

182TEST(Calib3d_ChessboardDetector, timing) { CV_ChessboardDetectorTimingTest test; test.safe_run(); }
