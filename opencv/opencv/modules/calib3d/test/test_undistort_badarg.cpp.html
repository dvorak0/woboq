99//initializing
100 img_size.width = 800;

102 double cam[9] = {150.f, 0.f, img_size.width/2.f, 0, 300.f, img_size.height/2.f, 0.f, 0.f, 1.f};
103 double dist[4] = {0.01,0.02,0.001,0.0005};
104 double s_points[N_POINTS2] = {
105 static_cast<double>(img_size.width) / 4.0,
106 static_cast<double>(img_size.height) / 4.0,
107 };
108 double d_points[N_POINTS2];
109 double p[9] = {155.f, 0.f, img_size.width/2.f+img_size.width/50.f, 0, 310.f, img_size.height/2.f+img_size.height/50.f, 0.f, 0.f, 1.f};
110 double r[9] = {1,0,0,0,1,0,0,0,1};
111
112 CvMat _camera_mat_orig = cvMat(3,3,CV_64F,cam);
113 CvMat _distortion_coeffs_orig = cvMat(1,4,CV_64F,dist);
114 CvMat _P_orig = cvMat(3,3,CV_64F,p);
115 CvMat _R_orig = cvMat(3,3,CV_64F,r);
116 CvMat _src_points_orig = cvMat(1,4,CV_64FC2,s_points);
117 CvMat _dst_points_orig = cvMat(1,4,CV_64FC2,d_points);
118
119 _camera_mat = &_camera_mat_orig;
120 _distortion_coeffs = &_distortion_coeffs_orig;
121 matP = &_P_orig;
122 matR = &_R_orig;
123 _src_points = &_src_points_orig;
124 _dst_points = &_dst_points_orig;
125
126//tests
127 CvMat* temp1;
128 CvMat* temp;
129 IplImage* temp_img = cvCreateImage(cvSize(img_size.width,img_size.height),8,3);
130
131//-----------

133 _src_points = temp;
134 errcount += run_test_case( CV_StsAssert, "Input data is not CvMat*" );
135 _src_points = &_src_points_orig;
136
137 temp = (CvMat*)temp_img;

139 errcount += run_test_case( CV_StsAssert, "Output data is not CvMat*" );

147

174 cvReleaseMat(&temp);
175//------------
176 temp = cvCreateMat(2,3,CV_64F);
177 _camera_mat = temp;
178 errcount += run_test_case( CV_StsAssert, "Invalid camera data matrix size" );
179 _camera_mat = &_camera_mat_orig;
180 cvReleaseMat(&temp);
181

192//----------
193

195 _distortion_coeffs = temp;

198
199 temp = cvCreateMat(1,6,CV_64F);
200 _distortion_coeffs = temp;
201 errcount += run_test_case( CV_StsAssert, "Invalid distortion coefficients data matrix size" );
202 _distortion_coeffs = &_distortion_coeffs_orig;
203 cvReleaseMat(&temp);
204

241 matP = temp;

245//------------
246 //C++ tests
247 useCPlus = true;
248

254
255 temp = cvCreateMat(2,2,CV_32FC2);
256 src_points = cv::Mat(temp);
257 errcount += run_test_case( CV_StsAssert, "Invalid input data matrix size" );
258 src_points = cv::Mat(&_src_points_orig);
259 cvReleaseMat(&temp);
260
261 temp = cvCreateMat(1,4,CV_64FC2);
262 src_points = cv::Mat(temp);
263 errcount += run_test_case( CV_StsAssert, "Invalid input data matrix type" );
264 src_points = cv::Mat(&_src_points_orig);

266
267 src_points = cv::Mat();
268 errcount += run_test_case( CV_StsAssert, "Input data matrix is not continuous" );
269 src_points = cv::Mat(&_src_points_orig);
270 cvReleaseMat(&temp);
271
272
273
274//------------
275 cvReleaseImage(&temp_img);
