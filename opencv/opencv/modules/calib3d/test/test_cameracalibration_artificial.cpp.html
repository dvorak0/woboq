65Mat calcRvec(const vector<Point3f>& points, const Size& cornerSize)
66{
67 Point3f p00 = points[0];
68 Point3f p10 = points[1];
69 Point3f p01 = points[cornerSize.width];
70
71 Vec3d ex(p10.x - p00.x, p10.y - p00.y, p10.z - p00.z);
72 Vec3d ey(p01.x - p00.x, p01.y - p00.y, p01.z - p00.z);
73 Vec3d ez = ex.cross(ey);
74
75 Mat rot(3, 3, CV_64F);
76 *rot.ptr<Vec3d>(0) = ex;
77 *rot.ptr<Vec3d>(1) = ey;
78 *rot.ptr<Vec3d>(2) = ez * (1.0/norm(ez));
79
80 Mat res;
81 Rodrigues(rot.t(), res);
82 return res.reshape(1, 1);
83}

101 bool checkErr(double a, double a0, double eps, double delta)
102 {
103 return fabs(a - a0) > eps * (fabs(a0) + delta);

106 void compareCameraMatrs(const Mat_<double>& camMat, const Mat& camMat_est)
107 {
108 if ( camMat_est.at<double>(0, 1) != 0 || camMat_est.at<double>(1, 0) != 0 ||
109 camMat_est.at<double>(2, 0) != 0 || camMat_est.at<double>(2, 1) != 0 ||
110 camMat_est.at<double>(2, 2) != 1)
111 {
112 ts->printf( cvtest::TS::LOG, "Bad shape of camera matrix returned \n");
113 ts->set_failed_test_info(cvtest::TS::FAIL_MISMATCH);
114 }
115
116 double fx_e = camMat_est.at<double>(0, 0), fy_e = camMat_est.at<double>(1, 1);
117 double cx_e = camMat_est.at<double>(0, 2), cy_e = camMat_est.at<double>(1, 2);
118
119 double fx = camMat(0, 0), fy = camMat(1, 1), cx = camMat(0, 2), cy = camMat(1, 2);
120
121 const double eps = 1e-2;
122 const double dlt = 1e-5;
123
124 bool fail = checkErr(fx_e, fx, eps, dlt) || checkErr(fy_e, fy, eps, dlt) ||
125 checkErr(cx_e, cx, eps, dlt) || checkErr(cy_e, cy, eps, dlt);
126
127 if (fail)
128 {
129 ts->set_failed_test_info(cvtest::TS::FAIL_BAD_ACCURACY);
130 }
131 ts->printf( cvtest::TS::LOG, "%d) Expected [Fx Fy Cx Cy] = [%.3f %.3f %.3f %.3f]\n", r, fx, fy, cx, cy);
132 ts->printf( cvtest::TS::LOG, "%d) Estimated [Fx Fy Cx Cy] = [%.3f %.3f %.3f %.3f]\n", r, fx_e, fy_e, cx_e, cy_e);
133 }

164 void compareShiftVecs(const vector<Mat>& tvecs, const vector<Mat>& tvecs_est)
165 {
166 const double eps = 1e-2;
167 const double dlt = 1e-4;
168
169 int err_count = 0;
170 const int errMsgNum = 4;
171 for(size_t i = 0; i < tvecs.size(); ++i)
172 {
173 const Point3d& tvec = *tvecs[i].ptr<Point3d>();
174 const Point3d& tvec_est = *tvecs_est[i].ptr<Point3d>();
175
176 if (norm(tvec_est - tvec) > eps* (norm(tvec) + dlt))
177 {
178 if (err_count++ < errMsgNum)
179 {
180 if (err_count == errMsgNum)
181 ts->printf( cvtest::TS::LOG, "%d) ...\n", r);
182 else
183 {
184 ts->printf( cvtest::TS::LOG, "%d) Bad accuracy in returned tvecs. Index = %d\n", r, i);
185 ts->printf( cvtest::TS::LOG, "%d) norm(tvec_est - tvec) = %f, norm(tvec_exp) = %f \n", r, norm(tvec_est - tvec), norm(tvec));
186 }
187 }
188 ts->set_failed_test_info(cvtest::TS::FAIL_BAD_ACCURACY);
189 }

224 double reprojectErrorWithoutIntrinsics(const vector<Point3f>& cb3d, const vector<Mat>& _rvecs_exp, const vector<Mat>& _tvecs_exp,
225 const vector<Mat>& rvecs_est, const vector<Mat>& tvecs_est)
226 {
227 const static Mat eye33 = Mat::eye(3, 3, CV_64F);
228 const static Mat zero15 = Mat::zeros(1, 5, CV_64F);
229 Mat _chessboard3D(cb3d);
230 vector<Point2f> uv_exp, uv_est;
231 double res = 0;
232
233 for(size_t i = 0; i < rvecs_exp.size(); ++i)
234 {
235 projectPoints(_chessboard3D, _rvecs_exp[i], _tvecs_exp[i], eye33, zero15, uv_exp);
236 projectPoints(_chessboard3D, rvecs_est[i], tvecs_est[i], eye33, zero15, uv_est);
237 for(size_t j = 0; j < cb3d.size(); ++j)
238 res += norm(uv_exp[i] - uv_est[i]);
239 }
240 return res;
241 }

252 void prepareForTest(const Mat& bg, const Mat& camMat, const Mat& distCoeffs, size_t brdsNum, const ChessBoardGenerator& cbg)
253 {
254 sqSile = Size2f(1.f, 1.f);
255 Size cornersSize = cbg.cornersSize();
256
257 chessboard3D.clear();
258 for(int j = 0; j < cornersSize.height; ++j)
259 for(int i = 0; i < cornersSize.width; ++i)
260 chessboard3D.push_back(Point3f(sqSile.width * i, sqSile.height * j, 0));
261
262 boards.resize(brdsNum);
263 rvecs_exp.resize(brdsNum);
264 tvecs_exp.resize(brdsNum);
265 objectPoints.clear();
266 objectPoints.resize(brdsNum, chessboard3D);
267 imagePoints_art.clear();
268 imagePoints_findCb.clear();
269
270 vector<Point2f> corners_art, corners_fcb;
271 for(size_t i = 0; i < brdsNum; ++i)
272 {
273 for(;;)
274 {
275 boards[i] = cbg(bg, camMat, distCoeffs, sqSile, corners_art);
276 if(findChessboardCorners(boards[i], cornersSize, corners_fcb))
277 break;
278 }
279
280 //cv::namedWindow("CB"); imshow("CB", boards[i]); cv::waitKey();
281
282 imagePoints_art.push_back(corners_art);
283 imagePoints_findCb.push_back(corners_fcb);
284
285 tvecs_exp[i].create(1, 3, CV_64F);
286 *tvecs_exp[i].ptr<Point3d>() = cbg.corners3d[0];
287 rvecs_exp[i] = calcRvec(cbg.corners3d, cbg.cornersSize());
288 }

292 void runTest(const Size& imgSize, const Mat_<double>& camMat, const Mat_<double>& distCoeffs, size_t brdsNum, const Size& cornersSize, int flag = 0)
293 {
294 const TermCriteria tc(TermCriteria::EPS|TermCriteria::MAX_ITER, 30, 0.1);
295
296 vector< vector<Point2f> > imagePoints;
297
298 switch(flag)
299 {
300 case JUST_FIND_CORNERS: imagePoints = imagePoints_findCb; break;
301 case ARTIFICIAL_CORNERS: imagePoints = imagePoints_art; break;
302
303 case USE_CORNERS_SUBPIX:
304 for(size_t i = 0; i < brdsNum; ++i)
305 {
306 Mat gray;
307 cvtColor(boards[i], gray, CV_BGR2GRAY);
308 vector<Point2f> tmp = imagePoints_findCb[i];
309 cornerSubPix(gray, tmp, Size(5, 5), Size(-1,-1), tc);
310 imagePoints.push_back(tmp);
311 }
312 break;
313 case USE_4QUAD_CORNERS:
314 for(size_t i = 0; i < brdsNum; ++i)
315 {
316 Mat gray;
317 cvtColor(boards[i], gray, CV_BGR2GRAY);
318 vector<Point2f> tmp = imagePoints_findCb[i];
319 find4QuadCornerSubpix(gray, tmp, Size(5, 5));
320 imagePoints.push_back(tmp);
321 }
322 break;
323 default:
324 throw std::exception();
325 }
326
327 Mat camMat_est = Mat::eye(3, 3, CV_64F), distCoeffs_est = Mat::zeros(1, 5, CV_64F);
328 vector<Mat> rvecs_est, tvecs_est;
329
330 int flags = /*CV_CALIB_FIX_K3|*/CV_CALIB_FIX_K4|CV_CALIB_FIX_K5|CV_CALIB_FIX_K6; //CALIB_FIX_K3; //CALIB_FIX_ASPECT_RATIO | | CALIB_ZERO_TANGENT_DIST;
331 TermCriteria criteria = TermCriteria(TermCriteria::COUNT+TermCriteria::EPS, 100, DBL_EPSILON);
332 double rep_error = calibrateCamera(objectPoints, imagePoints, imgSize, camMat_est, distCoeffs_est, rvecs_est, tvecs_est, flags, criteria);
333 rep_error /= brdsNum * cornersSize.area();
334
335 const double thres = 1;
336 if (rep_error > thres)
337 {
338 ts->printf( cvtest::TS::LOG, "%d) Too big reproject error = %f\n", r, rep_error);
339 ts->set_failed_test_info(cvtest::TS::FAIL_BAD_ACCURACY);
340 }
341
342 compareCameraMatrs(camMat, camMat_est);
343 compareDistCoeffs(distCoeffs, distCoeffs_est);
344 compareShiftVecs(tvecs_exp, tvecs_est);
345 compareRotationVecs(rvecs_exp, rvecs_est);
346
347 double rep_errorWOI = reprojectErrorWithoutIntrinsics(chessboard3D, rvecs_exp, tvecs_exp, rvecs_est, tvecs_est);
348 rep_errorWOI /= brdsNum * cornersSize.area();
349
350 const double thres2 = 0.01;
351 if (rep_errorWOI > thres2)
352 {
353 ts->printf( cvtest::TS::LOG, "%d) Too big reproject error without intrinsics = %f\n", r, rep_errorWOI);
354 ts->set_failed_test_info(cvtest::TS::FAIL_BAD_ACCURACY);
355 }
356
357 ts->printf( cvtest::TS::LOG, "%d) Testing solvePnP...\n", r);
358 rvecs_spnp.resize(brdsNum);
359 tvecs_spnp.resize(brdsNum);
360 for(size_t i = 0; i < brdsNum; ++i)
361 solvePnP(Mat(objectPoints[i]), Mat(imagePoints[i]), camMat, distCoeffs, rvecs_spnp[i], tvecs_spnp[i]);
362
363 compareShiftVecs(tvecs_exp, tvecs_spnp);
364 compareRotationVecs(rvecs_exp, rvecs_spnp);
365 }

367 void run(int)
368 {
369
370 ts->set_failed_test_info(cvtest::TS::OK);
371 RNG& rng = theRNG();
372
373 int progress = 0;
374 int repeat_num = 3;
375 for(r = 0; r < repeat_num; ++r)
376 {
377 const int brds_num = 20;
378
379 Mat bg(Size(640, 480), CV_8UC3);
380 randu(bg, Scalar::all(32), Scalar::all(255));
381 GaussianBlur(bg, bg, Size(5, 5), 2);
382
383 double fx = 300 + (20 * (double)rng - 10);
384 double fy = 300 + (20 * (double)rng - 10);
385
386 double cx = bg.cols/2 + (40 * (double)rng - 20);
387 double cy = bg.rows/2 + (40 * (double)rng - 20);
388
389 Mat_<double> camMat(3, 3);
390 camMat << fx, 0., cx, 0, fy, cy, 0., 0., 1.;
391
392 double k1 = 0.5 + (double)rng/5;
393 double k2 = (double)rng/5;
394 double k3 = (double)rng/5;
395
396 double p1 = 0.001 + (double)rng/10;
397 double p2 = 0.001 + (double)rng/10;
398
399 Mat_<double> distCoeffs(1, 5, 0.0);
400 distCoeffs << k1, k2, p1, p2, k3;
401
402 ChessBoardGenerator cbg(Size(9, 8));
403 cbg.min_cos = 0.9;
404 cbg.cov = 0.8;
405
406 progress = update_progress(progress, r, repeat_num, 0);
407 ts->printf( cvtest::TS::LOG, "\n");
408 prepareForTest(bg, camMat, distCoeffs, brds_num, cbg);
409
410 ts->printf( cvtest::TS::LOG, "artificial corners\n");
411 runTest(bg.size(), camMat, distCoeffs, brds_num, cbg.cornersSize(), ARTIFICIAL_CORNERS);
412 progress = update_progress(progress, r, repeat_num, 0);
413
414 ts->printf( cvtest::TS::LOG, "findChessboard corners\n");
415 runTest(bg.size(), camMat, distCoeffs, brds_num, cbg.cornersSize(), JUST_FIND_CORNERS);
416 progress = update_progress(progress, r, repeat_num, 0);
417
418 ts->printf( cvtest::TS::LOG, "cornersSubPix corners\n");
419 runTest(bg.size(), camMat, distCoeffs, brds_num, cbg.cornersSize(), USE_CORNERS_SUBPIX);
420 progress = update_progress(progress, r, repeat_num, 0);
421
422 ts->printf( cvtest::TS::LOG, "4quad corners\n");
423 runTest(bg.size(), camMat, distCoeffs, brds_num, cbg.cornersSize(), USE_4QUAD_CORNERS);
424 progress = update_progress(progress, r, repeat_num, 0);
425 }
426 }
