87int CV_DefaultNewCameraMatrixTest::prepare_test_case(int test_case_idx)
88{
89 int code = cvtest::ArrayTest::prepare_test_case( test_case_idx );
90
91 if (code <= 0)
92 return code;
93
94 RNG& rng = ts->get_rng();
95
96 img_size.width = cvtest::randInt(rng) % MAX_X + 1;
97 img_size.height = cvtest::randInt(rng) % MAX_Y + 1;
98
99 center_principal_point = ((cvtest::randInt(rng) % 2)!=0);
100
101 // Generating camera_mat matrix
102 double sz = MAX(img_size.width, img_size.height);
103 double aspect_ratio = cvtest::randReal(rng)*0.6 + 0.7;
104 double a[9] = {0,0,0,0,0,0,0,0,1};
105 Mat _a(3,3,CV_64F,a);
106 a[2] = (img_size.width - 1)*0.5 + cvtest::randReal(rng)*10 - 5;
107 a[5] = (img_size.height - 1)*0.5 + cvtest::randReal(rng)*10 - 5;
108 a[0] = sz/(0.9 - cvtest::randReal(rng)*0.6);
109 a[4] = aspect_ratio*a[0];
110
111 Mat& _a0 = test_mat[INPUT][0];
112 cvtest::convert(_a, _a0, _a0.type());
113 camera_mat = _a0;
114
115 return code;
116
117}

121 new_camera_mat = cv::getDefaultNewCameraMatrix(camera_mat,img_size,center_principal_point);
122}

124void CV_DefaultNewCameraMatrixTest::prepare_to_validation( int /*test_case_idx*/ )
125{
126 const Mat& src = test_mat[INPUT][0];
127 Mat& dst = test_mat[REF_OUTPUT][0];
128 Mat& test_output = test_mat[OUTPUT][0];
129 Mat& output = new_camera_mat;
130 cvtest::convert( output, test_output, test_output.type() );
131 if (!center_principal_point)
132 {
133 cvtest::copy(src, dst);
134 }
135 else
136 {
137 double a[9] = {0,0,0,0,0,0,0,0,1};
138 Mat _a(3,3,CV_64F,a);
139 if (matrix_type == CV_64F)
140 {
141 a[0] = src.at<double>(0,0);
142 a[4] = src.at<double>(1,1);
143 }
144 else
145 {
146 a[0] = src.at<float>(0,0);
147 a[4] = src.at<float>(1,1);
148 }
149 a[2] = (img_size.width - 1)*0.5;
150 a[5] = (img_size.height - 1)*0.5;
151 cvtest::convert( _a, dst, dst.type() );
152 }
153}

251int CV_UndistortPointsTest::prepare_test_case(int test_case_idx)
252{
253 RNG& rng = ts->get_rng();
254 int code = cvtest::ArrayTest::prepare_test_case( test_case_idx );
255
256 if (code <= 0)
257 return code;
258
259 useDstMat = (cvtest::randInt(rng) % 2) == 0;
260
261 img_size.width = cvtest::randInt(rng) % MAX_X + 1;
262 img_size.height = cvtest::randInt(rng) % MAX_Y + 1;
263 int dist_size = test_mat[INPUT][2].cols > test_mat[INPUT][2].rows ? test_mat[INPUT][2].cols : test_mat[INPUT][2].rows;
264 double cam[9] = {0,0,0,0,0,0,0,0,1};
265 vector<double> dist(dist_size);
266 vector<double> proj(test_mat[INPUT][4].cols * test_mat[INPUT][4].rows);
267 vector<Point2d> points(N_POINTS);
268
269 Mat _camera(3,3,CV_64F,cam);
270 Mat _distort(test_mat[INPUT][2].rows,test_mat[INPUT][2].cols,CV_64F,&dist[0]);
271 Mat _proj(test_mat[INPUT][4].size(), CV_64F, &proj[0]);
272 Mat _points(test_mat[INPUT][0].size(), CV_64FC2, &points[0]);
273
274 _proj = Scalar::all(0);
275
276 //Generating points
277 for( int i = 0; i < N_POINTS; i++ )
278 {
279 points[i].x = cvtest::randReal(rng)*img_size.width;
280 points[i].y = cvtest::randReal(rng)*img_size.height;
281 }
282
283 //Generating camera matrix
284 double sz = MAX(img_size.width,img_size.height);
285 double aspect_ratio = cvtest::randReal(rng)*0.6 + 0.7;
286 cam[2] = (img_size.width - 1)*0.5 + cvtest::randReal(rng)*10 - 5;
287 cam[5] = (img_size.height - 1)*0.5 + cvtest::randReal(rng)*10 - 5;
288 cam[0] = sz/(0.9 - cvtest::randReal(rng)*0.6);
289 cam[4] = aspect_ratio*cam[0];
290
291 //Generating distortion coeffs
292 dist[0] = cvtest::randReal(rng)*0.06 - 0.03;
293 dist[1] = cvtest::randReal(rng)*0.06 - 0.03;
294 if( dist[0]*dist[1] > 0 )
295 dist[1] = -dist[1];
296 if( cvtest::randInt(rng)%4 != 0 )
297 {
298 dist[2] = cvtest::randReal(rng)*0.004 - 0.002;
299 dist[3] = cvtest::randReal(rng)*0.004 - 0.002;
300 if (dist_size > 4)
301 dist[4] = cvtest::randReal(rng)*0.004 - 0.002;
302 }
303 else
304 {
305 dist[2] = dist[3] = 0;
306 if (dist_size > 4)
307 dist[4] = 0;
308 }
309
310 //Generating P matrix (projection)
311 if( test_mat[INPUT][4].cols != 4 )
312 {
313 proj[8] = 1;
314 if (cvtest::randInt(rng)%2 == 0) // use identity new camera matrix
315 {
316 proj[0] = 1;
317 proj[4] = 1;
318 }
319 else
320 {
321 proj[0] = cam[0] + (cvtest::randReal(rng) - (double)0.5)*0.2*cam[0]; //10%
322 proj[4] = cam[4] + (cvtest::randReal(rng) - (double)0.5)*0.2*cam[4]; //10%
323 proj[2] = cam[2] + (cvtest::randReal(rng) - (double)0.5)*0.3*img_size.width; //15%
324 proj[5] = cam[5] + (cvtest::randReal(rng) - (double)0.5)*0.3*img_size.height; //15%
325 }
326 }
327 else
328 {
329 proj[10] = 1;
330 proj[0] = cam[0] + (cvtest::randReal(rng) - (double)0.5)*0.2*cam[0]; //10%
331 proj[5] = cam[4] + (cvtest::randReal(rng) - (double)0.5)*0.2*cam[4]; //10%
332 proj[2] = cam[2] + (cvtest::randReal(rng) - (double)0.5)*0.3*img_size.width; //15%
333 proj[6] = cam[5] + (cvtest::randReal(rng) - (double)0.5)*0.3*img_size.height; //15%
334
335 proj[3] = (img_size.height + img_size.width - 1)*0.5 + cvtest::randReal(rng)*10 - 5;
336 proj[7] = (img_size.height + img_size.width - 1)*0.5 + cvtest::randReal(rng)*10 - 5;
337 proj[11] = (img_size.height + img_size.width - 1)*0.5 + cvtest::randReal(rng)*10 - 5;
338 }
339
340 //Generating R matrix
341 Mat _rot(3,3,CV_64F);
342 Mat rotation(1,3,CV_64F);
343 rotation.at<double>(0) = CV_PI*(cvtest::randReal(rng) - (double)0.5); // phi
344 rotation.at<double>(1) = CV_PI*(cvtest::randReal(rng) - (double)0.5); // ksi
345 rotation.at<double>(2) = CV_PI*(cvtest::randReal(rng) - (double)0.5); //khi
346 cvtest::Rodrigues(rotation, _rot);
347
348 //copying data
349 //src_points = &_points;
350 _points.convertTo(test_mat[INPUT][0], test_mat[INPUT][0].type());
351 _camera.convertTo(test_mat[INPUT][1], test_mat[INPUT][1].type());
352 _distort.convertTo(test_mat[INPUT][2], test_mat[INPUT][2].type());
353 _rot.convertTo(test_mat[INPUT][3], test_mat[INPUT][3].type());
354 _proj.convertTo(test_mat[INPUT][4], test_mat[INPUT][4].type());
355
356 zero_distortion = (cvtest::randInt(rng)%2) == 0 ? false : true;
357 zero_new_cam = (cvtest::randInt(rng)%2) == 0 ? false : true;
358 zero_R = (cvtest::randInt(rng)%2) == 0 ? false : true;
359
360 if (useCPlus)
361 {
362 _points.convertTo(src_points, CV_32F);
363
364 camera_mat = test_mat[INPUT][1];
365 distortion_coeffs = test_mat[INPUT][2];
366 R = test_mat[INPUT][3];
367 P = test_mat[INPUT][4];
368 }
369
370 return code;
371}

373void CV_UndistortPointsTest::prepare_to_validation(int /*test_case_idx*/)
374{
375 int dist_size = test_mat[INPUT][2].cols > test_mat[INPUT][2].rows ? test_mat[INPUT][2].cols : test_mat[INPUT][2].rows;
376 double cam[9] = {0,0,0,0,0,0,0,0,1};
377 double rot[9] = {1,0,0,0,1,0,0,0,1};
378
379 double* dist = new double[dist_size ];
380 double* proj = new double[test_mat[INPUT][4].cols * test_mat[INPUT][4].rows];
381 double* points = new double[N_POINTS*2];
382 double* r_points = new double[N_POINTS*2];
383 //Run reference calculations
384 CvMat ref_points= cvMat(test_mat[INPUT][0].rows,test_mat[INPUT][0].cols,CV_64FC2,r_points);
385 CvMat _camera = cvMat(3,3,CV_64F,cam);
386 CvMat _rot = cvMat(3,3,CV_64F,rot);
387 CvMat _distort = cvMat(test_mat[INPUT][2].rows,test_mat[INPUT][2].cols,CV_64F,dist);
388 CvMat _proj = cvMat(test_mat[INPUT][4].rows,test_mat[INPUT][4].cols,CV_64F,proj);
389 CvMat _points= cvMat(test_mat[TEMP][0].rows,test_mat[TEMP][0].cols,CV_64FC2,points);
390
391 Mat __camera = cvarrToMat(&_camera);
392 Mat __distort = cvarrToMat(&_distort);
393 Mat __rot = cvarrToMat(&_rot);
394 Mat __proj = cvarrToMat(&_proj);
395 Mat __points = cvarrToMat(&_points);
396 Mat _ref_points = cvarrToMat(&ref_points);
397
398 cvtest::convert(test_mat[INPUT][1], __camera, __camera.type());
399 cvtest::convert(test_mat[INPUT][2], __distort, __distort.type());
400 cvtest::convert(test_mat[INPUT][3], __rot, __rot.type());
401 cvtest::convert(test_mat[INPUT][4], __proj, __proj.type());
402
403 if (useCPlus)
404 {
405 if (useDstMat)
406 {
407 CvMat temp = dst_points_mat;
408 for (int i=0;i<N_POINTS*2;i++)
409 {
410 points[i] = temp.data.fl[i];
411 }
412 }
413 else
414 {
415 for (int i=0;i<N_POINTS;i++)
416 {
417 points[2*i] = dst_points[i].x;
418 points[2*i+1] = dst_points[i].y;
419 }
420 }
421 }
422 else
423 {
424 cvtest::convert(test_mat[TEMP][0],__points, __points.type());
425 }
426
427 CvMat* input2 = zero_distortion ? 0 : &_distort;
428 CvMat* input3 = zero_R ? 0 : &_rot;
429 CvMat* input4 = zero_new_cam ? 0 : &_proj;
430 distortPoints(&_points,&ref_points,&_camera,input2,input3,input4);
431
432 Mat& dst = test_mat[REF_OUTPUT][0];
433 cvtest::convert(_ref_points, dst, dst.type());
434
435 cvtest::copy(test_mat[INPUT][0], test_mat[OUTPUT][0]);
436
437 delete[] dist;
438 delete[] proj;
439 delete[] points;
440 delete[] r_points;
441}

443void CV_UndistortPointsTest::run_func()
444{
445
446 if (useCPlus)
447 {
448 cv::Mat input2,input3,input4;
449 input2 = zero_distortion ? cv::Mat() : cv::Mat(test_mat[INPUT][2]);
450 input3 = zero_R ? cv::Mat() : cv::Mat(test_mat[INPUT][3]);
451 input4 = zero_new_cam ? cv::Mat() : cv::Mat(test_mat[INPUT][4]);
452
453 if (useDstMat)
454 {
455 //cv::undistortPoints(src_points,dst_points_mat,camera_mat,distortion_coeffs,R,P);
456 cv::undistortPoints(src_points,dst_points_mat,camera_mat,input2,input3,input4);
457 }
458 else
459 {
460 //cv::undistortPoints(src_points,dst_points,camera_mat,distortion_coeffs,R,P);
461 cv::undistortPoints(src_points,dst_points,camera_mat,input2,input3,input4);
462 }
463 }
464 else
465 {
466 CvMat _input0 = test_mat[INPUT][0], _input1 = test_mat[INPUT][1], _input2, _input3, _input4;
467 CvMat _output = test_mat[TEMP][0];
468 if(!zero_distortion)
469 _input2 = test_mat[INPUT][2];
470 if(!zero_R)
471 _input3 = test_mat[INPUT][3];
472 if(!zero_new_cam)
473 _input4 = test_mat[INPUT][4];
474 cvUndistortPoints(&_input0, &_output, &_input1,
475 zero_distortion ? 0 : &_input2,
476 zero_R ? 0 : &_input3,
477 zero_new_cam ? 0 : &_input4);
478 }
479}

481void CV_UndistortPointsTest::distortPoints(const CvMat* _src, CvMat* _dst, const CvMat* _cameraMatrix,
482 const CvMat* _distCoeffs,
483 const CvMat* matR, const CvMat* matP)
484{
485 double a[9];
486
487 CvMat* __P;
488 if ((!matP)||(matP->cols == 3))
489 __P = cvCreateMat(3,3,CV_64F);
490 else
491 __P = cvCreateMat(3,4,CV_64F);
492 if (matP)
493 {
494 cvTsConvert(matP,__P);
495 }
496 else
497 {
498 cvZero(__P);
499 __P->data.db[0] = 1;
500 __P->data.db[4] = 1;
501 __P->data.db[8] = 1;
502 }
503 CvMat* __R = cvCreateMat(3,3,CV_64F);;
504 if (matR)
505 {
506 cvCopy(matR,__R);
507 }
508 else
509 {
510 cvZero(__R);
511 __R->data.db[0] = 1;
512 __R->data.db[4] = 1;
513 __R->data.db[8] = 1;
514 }
515 for (int i=0;i<N_POINTS;i++)
516 {
517 int movement = __P->cols > 3 ? 1 : 0;
518 double x = (_src->data.db[2*i]-__P->data.db[2])/__P->data.db[0];
519 double y = (_src->data.db[2*i+1]-__P->data.db[5+movement])/__P->data.db[4+movement];
520 CvMat inverse = cvMat(3,3,CV_64F,a);
521 cvInvert(__R,&inverse);
522 double w1 = x*inverse.data.db[6]+y*inverse.data.db[7]+inverse.data.db[8];
523 double _x = (x*inverse.data.db[0]+y*inverse.data.db[1]+inverse.data.db[2])/w1;
524 double _y = (x*inverse.data.db[3]+y*inverse.data.db[4]+inverse.data.db[5])/w1;
525
526 //Distortions
527
528 double __x = _x;
529 double __y = _y;
530 if (_distCoeffs)
531 {
532 double r2 = _x*_x+_y*_y;
533
534 __x = _x*(1+_distCoeffs->data.db[0]*r2+_distCoeffs->data.db[1]*r2*r2)+
535 2*_distCoeffs->data.db[2]*_x*_y+_distCoeffs->data.db[3]*(r2+2*_x*_x);
536 __y = _y*(1+_distCoeffs->data.db[0]*r2+_distCoeffs->data.db[1]*r2*r2)+
537 2*_distCoeffs->data.db[3]*_x*_y+_distCoeffs->data.db[2]*(r2+2*_y*_y);
538 if ((_distCoeffs->cols > 4) || (_distCoeffs->rows > 4))
539 {
540 __x+=_x*_distCoeffs->data.db[4]*r2*r2*r2;
541 __y+=_y*_distCoeffs->data.db[4]*r2*r2*r2;
542 }
543 }
544
545
546 _dst->data.db[2*i] = __x*_cameraMatrix->data.db[0]+_cameraMatrix->data.db[2];
547 _dst->data.db[2*i+1] = __y*_cameraMatrix->data.db[4]+_cameraMatrix->data.db[5];
548
549 }
550
551 cvReleaseMat(&__R);
552 cvReleaseMat(&__P);
553
554}

651int CV_InitUndistortRectifyMapTest::prepare_test_case(int test_case_idx)
652{
653 RNG& rng = ts->get_rng();
654 int code = cvtest::ArrayTest::prepare_test_case( test_case_idx );
655
656 if (code <= 0)
657 return code;
658
659 img_size.width = cvtest::randInt(rng) % MAX_X + 1;
660 img_size.height = cvtest::randInt(rng) % MAX_Y + 1;
661
662 if (useCPlus)
663 {
664 mat_type = (cvtest::randInt(rng) % 2) == 0 ? CV_32FC1 : CV_16SC2;
665 if ((cvtest::randInt(rng) % 4) == 0)
666 mat_type = -1;
667 if ((cvtest::randInt(rng) % 4) == 0)
668 mat_type = CV_32FC2;
669 _mapx = 0;
670 _mapy = 0;
671 }
672 else
673 {
674 int typex = (cvtest::randInt(rng) % 2) == 0 ? CV_32FC1 : CV_16SC2;
675 //typex = CV_32FC1; ///!!!!!!!!!!!!!!!!
676 int typey = (typex == CV_32FC1) ? CV_32FC1 : CV_16UC1;
677
678 _mapx = cvCreateMat(img_size.height,img_size.width,typex);
679 _mapy = cvCreateMat(img_size.height,img_size.width,typey);
680
681
682 }
683
684 int dist_size = test_mat[INPUT][2].cols > test_mat[INPUT][2].rows ? test_mat[INPUT][2].cols : test_mat[INPUT][2].rows;
685 double cam[9] = {0,0,0,0,0,0,0,0,1};
686 vector<double> dist(dist_size);
687 vector<double> new_cam(test_mat[INPUT][4].cols * test_mat[INPUT][4].rows);
688 vector<Point2d> points(N_POINTS);
689
690 Mat _camera(3,3,CV_64F,cam);
691 Mat _distort(test_mat[INPUT][2].size(),CV_64F,&dist[0]);
692 Mat _new_cam(test_mat[INPUT][4].size(),CV_64F,&new_cam[0]);
693 Mat _points(test_mat[INPUT][0].size(),CV_64FC2, &points[0]);
694
695 //Generating points
696 for (int i=0;i<N_POINTS;i++)
697 {
698 points[i].x = cvtest::randReal(rng)*img_size.width;
699 points[i].y = cvtest::randReal(rng)*img_size.height;
700 }
701
702 //Generating camera matrix
703 double sz = MAX(img_size.width,img_size.height);
704 double aspect_ratio = cvtest::randReal(rng)*0.6 + 0.7;
705 cam[2] = (img_size.width - 1)*0.5 + cvtest::randReal(rng)*10 - 5;
706 cam[5] = (img_size.height - 1)*0.5 + cvtest::randReal(rng)*10 - 5;
707 cam[0] = sz/(0.9 - cvtest::randReal(rng)*0.6);
708 cam[4] = aspect_ratio*cam[0];
709
710 //Generating distortion coeffs
711 dist[0] = cvtest::randReal(rng)*0.06 - 0.03;
712 dist[1] = cvtest::randReal(rng)*0.06 - 0.03;
713 if( dist[0]*dist[1] > 0 )
714 dist[1] = -dist[1];
715 if( cvtest::randInt(rng)%4 != 0 )
716 {
717 dist[2] = cvtest::randReal(rng)*0.004 - 0.002;
718 dist[3] = cvtest::randReal(rng)*0.004 - 0.002;
719 if (dist_size > 4)
720 dist[4] = cvtest::randReal(rng)*0.004 - 0.002;
721 }
722 else
723 {
724 dist[2] = dist[3] = 0;
725 if (dist_size > 4)
726 dist[4] = 0;
727 }
728
729 //Generating new camera matrix
730 _new_cam = Scalar::all(0);
731 new_cam[8] = 1;
732
733 //new_cam[0] = cam[0];
734 //new_cam[4] = cam[4];
735 //new_cam[2] = cam[2];
736 //new_cam[5] = cam[5];
737
738 new_cam[0] = cam[0] + (cvtest::randReal(rng) - (double)0.5)*0.2*cam[0]; //10%
739 new_cam[4] = cam[4] + (cvtest::randReal(rng) - (double)0.5)*0.2*cam[4]; //10%
740 new_cam[2] = cam[2] + (cvtest::randReal(rng) - (double)0.5)*0.3*img_size.width; //15%
741 new_cam[5] = cam[5] + (cvtest::randReal(rng) - (double)0.5)*0.3*img_size.height; //15%
742
743
744 //Generating R matrix
745 Mat _rot(3,3,CV_64F);
746 Mat rotation(1,3,CV_64F);
747 rotation.at<double>(0) = CV_PI/8*(cvtest::randReal(rng) - (double)0.5); // phi
748 rotation.at<double>(1) = CV_PI/8*(cvtest::randReal(rng) - (double)0.5); // ksi
749 rotation.at<double>(2) = CV_PI/3*(cvtest::randReal(rng) - (double)0.5); //khi
750 cvtest::Rodrigues(rotation, _rot);
751
752 //cvSetIdentity(_rot);
753 //copying data
754 cvtest::convert( _points, test_mat[INPUT][0], test_mat[INPUT][0].type());
755 cvtest::convert( _camera, test_mat[INPUT][1], test_mat[INPUT][1].type());
756 cvtest::convert( _distort, test_mat[INPUT][2], test_mat[INPUT][2].type());
757 cvtest::convert( _rot, test_mat[INPUT][3], test_mat[INPUT][3].type());
758 cvtest::convert( _new_cam, test_mat[INPUT][4], test_mat[INPUT][4].type());
759
760 zero_distortion = (cvtest::randInt(rng)%2) == 0 ? false : true;
761 zero_new_cam = (cvtest::randInt(rng)%2) == 0 ? false : true;
762 zero_R = (cvtest::randInt(rng)%2) == 0 ? false : true;
763
764 if (useCPlus)
765 {
766 camera_mat = test_mat[INPUT][1];
767 distortion_coeffs = test_mat[INPUT][2];
768 R = test_mat[INPUT][3];
769 new_camera_mat = test_mat[INPUT][4];
770 }
771
772 return code;
