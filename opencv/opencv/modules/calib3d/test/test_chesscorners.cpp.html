54void show_points( const Mat& gray, const Mat& u, const vector<Point2f>& v, Size pattern_size, bool was_found )
55{
56 Mat rgb( gray.size(), CV_8U);
57 merge(vector<Mat>(3, gray), rgb);
58
59 for(size_t i = 0; i < v.size(); i++ )
60 circle( rgb, v[i], 3, CV_RGB(255, 0, 0), CV_FILLED);
61
62 if( !u.empty() )
63 {
64 const Point2f* u_data = u.ptr<Point2f>();
65 size_t count = u.cols * u.rows;
66 for(size_t i = 0; i < count; i++ )
67 circle( rgb, u_data[i], 3, CV_RGB(0, 255, 0), CV_FILLED);
68 }
69 if (!v.empty())
70 {
71 Mat corners((int)v.size(), 1, CV_32FC2, (void*)&v[0]);
72 drawChessboardCorners( rgb, pattern_size, corners, was_found );
73 }
74 //namedWindow( "test", 0 ); imshow( "test", rgb ); waitKey(0);
75}

99double calcError(const vector<Point2f>& v, const Mat& u)
100{
101 int count_exp = u.cols * u.rows;
102 const Point2f* u_data = u.ptr<Point2f>();
103
104 double err = numeric_limits<double>::max();
105 for( int k = 0; k < 2; ++k )
106 {
107 double err1 = 0;
108 for( int j = 0; j < count_exp; ++j )
109 {
110 int j1 = k == 0 ? j : count_exp - j - 1;
111 double dx = fabs( v[j].x - u_data[j1].x );
112 double dy = fabs( v[j].y - u_data[j1].y );
113
114#if defined(_L2_ERR)
115 err1 += dx*dx + dy*dy;
116#else
117 dx = MAX( dx, dy );
118 if( dx > err1 )
119 err1 = dx;
120#endif //_L2_ERR
121 //printf("dx = %f\n", dx);
122 }
123 //printf("\n");
124 err = min(err, err1);
125 }
126
127#if defined(_L2_ERR)
128 err = sqrt(err/count_exp);
129#endif //_L2_ERR
130
131 return err;
132}

177void CV_ChessboardDetectorTest::run_batch( const string& filename )
178{
179 ts->printf(cvtest::TS::LOG, "\nRunning batch %s\n", filename.c_str());
180//#define WRITE_POINTS 1
181#ifndef WRITE_POINTS
182 double max_rough_error = 0, max_precise_error = 0;
183#endif
184 string folder;
185 switch( pattern )
186 {
187 case CHESSBOARD:
188 folder = string(ts->get_data_path()) + "cameracalibration/";
189 break;
190 case CIRCLES_GRID:
191 folder = string(ts->get_data_path()) + "cameracalibration/circles/";
192 break;
193 case ASYMMETRIC_CIRCLES_GRID:
194 folder = string(ts->get_data_path()) + "cameracalibration/asymmetric_circles/";
195 break;
196 }
197
198 FileStorage fs( folder + filename, FileStorage::READ );
199 FileNode board_list = fs["boards"];
200
201 if( !fs.isOpened() || board_list.empty() || !board_list.isSeq() || board_list.size() % 2 != 0 )
202 {
203 ts->printf( cvtest::TS::LOG, "%s can not be readed or is not valid\n", (folder + filename).c_str() );
204 ts->printf( cvtest::TS::LOG, "fs.isOpened=%d, board_list.empty=%d, board_list.isSeq=%d,board_list.size()%2=%d\n",
205 fs.isOpened(), (int)board_list.empty(), board_list.isSeq(), board_list.size()%2);
206 ts->set_failed_test_info( cvtest::TS::FAIL_MISSING_TEST_DATA );
207 return;
208 }
209
210 int progress = 0;
211 int max_idx = board_list.node->data.seq->total/2;
212 double sum_error = 0.0;
213 int count = 0;
214
215 for(int idx = 0; idx < max_idx; ++idx )
216 {
217 ts->update_context( this, idx, true );
218
219 /* read the image */
220 string img_file = board_list[idx * 2];
221 Mat gray = imread( folder + img_file, 0);
222
223 if( gray.empty() )
224 {
225 ts->printf( cvtest::TS::LOG, "one of chessboard images can't be read: %s\n", img_file.c_str() );
226 ts->set_failed_test_info( cvtest::TS::FAIL_MISSING_TEST_DATA );
227 return;
228 }
229
230 string _filename = folder + (string)board_list[idx * 2 + 1];
231 bool doesContatinChessboard;
232 Mat expected;
233 {
234 FileStorage fs1(_filename, FileStorage::READ);
235 fs1["corners"] >> expected;
236 fs1["isFound"] >> doesContatinChessboard;
237 fs1.release();
238 }
239 size_t count_exp = static_cast<size_t>(expected.cols * expected.rows);
240 Size pattern_size = expected.size();
241
242 vector<Point2f> v;
243 bool result = false;
244 switch( pattern )
245 {
246 case CHESSBOARD:
247 result = findChessboardCorners(gray, pattern_size, v, CV_CALIB_CB_ADAPTIVE_THRESH | CV_CALIB_CB_NORMALIZE_IMAGE);
248 break;
249 case CIRCLES_GRID:
250 result = findCirclesGrid(gray, pattern_size, v);
251 break;
252 case ASYMMETRIC_CIRCLES_GRID:
253 result = findCirclesGrid(gray, pattern_size, v, CALIB_CB_ASYMMETRIC_GRID | algorithmFlags);
254 break;
255 }
256 show_points( gray, Mat(), v, pattern_size, result );
257
258 if( result ^ doesContatinChessboard || v.size() != count_exp )
259 {
260 ts->printf( cvtest::TS::LOG, "chessboard is detected incorrectly in %s\n", img_file.c_str() );
261 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_OUTPUT );
262 return;
263 }
264
265 if( result )
266 {
267
268#ifndef WRITE_POINTS
269 double err = calcError(v, expected);
270#if 0
271 if( err > rough_success_error_level )
272 {
273 ts.printf( cvtest::TS::LOG, "bad accuracy of corner guesses\n" );
274 ts.set_failed_test_info( cvtest::TS::FAIL_BAD_ACCURACY );
275 continue;
276 }
277#endif
278 max_rough_error = MAX( max_rough_error, err );
279#endif
280 if( pattern == CHESSBOARD )
281 cornerSubPix( gray, v, Size(5, 5), Size(-1,-1), TermCriteria(TermCriteria::EPS|TermCriteria::MAX_ITER, 30, 0.1));
282 //find4QuadCornerSubpix(gray, v, Size(5, 5));
283 show_points( gray, expected, v, pattern_size, result );
284#ifndef WRITE_POINTS
285 // printf("called find4QuadCornerSubpix\n");
286 err = calcError(v, expected);
287 sum_error += err;
288 count++;
289#if 1
290 if( err > precise_success_error_level )
291 {
292 ts->printf( cvtest::TS::LOG, "Image %s: bad accuracy of adjusted corners %f\n", img_file.c_str(), err );
293 ts->set_failed_test_info( cvtest::TS::FAIL_BAD_ACCURACY );
294 return;
295 }
296#endif
297 ts->printf(cvtest::TS::LOG, "Error on %s is %f\n", img_file.c_str(), err);
298 max_precise_error = MAX( max_precise_error, err );
299#endif
300 }
301
302#ifdef WRITE_POINTS
303 Mat mat_v(pattern_size, CV_32FC2, (void*)&v[0]);
304 FileStorage fs(_filename, FileStorage::WRITE);
305 fs << "isFound" << result;
306 fs << "corners" << mat_v;
307 fs.release();
308#endif
309 progress = update_progress( progress, idx, max_idx, 0 );
310 }
311
312 sum_error /= count;
313 ts->printf(cvtest::TS::LOG, "Average error is %f\n", sum_error);
314}

316double calcErrorMinError(const Size& cornSz, const vector<Point2f>& corners_found, const vector<Point2f>& corners_generated)
317{
318 Mat m1(cornSz, CV_32FC2, (Point2f*)&corners_generated[0]);
319 Mat m2; flip(m1, m2, 0);
320
321 Mat m3; flip(m1, m3, 1); m3 = m3.t(); flip(m3, m3, 1);
322
323 Mat m4 = m1.t(); flip(m4, m4, 1);
324
325 double min1 = min(calcError(corners_found, m1), calcError(corners_found, m2));
326 double min2 = min(calcError(corners_found, m3), calcError(corners_found, m4));
327 return min(min1, min2);
328}

330bool validateData(const ChessBoardGenerator& cbg, const Size& imgSz,
331 const vector<Point2f>& corners_generated)
332{
333 Size cornersSize = cbg.cornersSize();
334 Mat_<Point2f> mat(cornersSize.height, cornersSize.width, (Point2f*)&corners_generated[0]);
335
336 double minNeibDist = std::numeric_limits<double>::max();
337 double tmp = 0;
338 for(int i = 1; i < mat.rows - 2; ++i)
339 for(int j = 1; j < mat.cols - 2; ++j)
340 {
341 const Point2f& cur = mat(i, j);
342
343 tmp = norm( cur - mat(i + 1, j + 1) );
344 if (tmp < minNeibDist)
345 tmp = minNeibDist;
346
347 tmp = norm( cur - mat(i - 1, j + 1 ) );
348 if (tmp < minNeibDist)
349 tmp = minNeibDist;
350
351 tmp = norm( cur - mat(i + 1, j - 1) );
352 if (tmp < minNeibDist)
353 tmp = minNeibDist;
354
355 tmp = norm( cur - mat(i - 1, j - 1) );
356 if (tmp < minNeibDist)
357 tmp = minNeibDist;
358 }
359
360 const double threshold = 0.25;
361 double cbsize = (max(cornersSize.width, cornersSize.height) + 1) * minNeibDist;
362 int imgsize = min(imgSz.height, imgSz.width);
363 return imgsize * threshold < cbsize;
364}

366bool CV_ChessboardDetectorTest::checkByGenerator()
367{
368 bool res = true;
369
370// for some reason, this test sometimes fails on Ubuntu
371#if (defined __APPLE__ && defined __x86_64__) || defined _MSC_VER
372 //theRNG() = 0x58e6e895b9913160;
373 //cv::DefaultRngAuto dra;
374 //theRNG() = *ts->get_rng();
375
376 Mat bg(Size(800, 600), CV_8UC3, Scalar::all(255));
377 randu(bg, Scalar::all(0), Scalar::all(255));
378 GaussianBlur(bg, bg, Size(7,7), 3.0);
379
380 Mat_<float> camMat(3, 3);
381 camMat << 300.f, 0.f, bg.cols/2.f, 0, 300.f, bg.rows/2.f, 0.f, 0.f, 1.f;
382
383 Mat_<float> distCoeffs(1, 5);
384 distCoeffs << 1.2f, 0.2f, 0.f, 0.f, 0.f;
385
386 const Size sizes[] = { Size(6, 6), Size(8, 6), Size(11, 12), Size(5, 4) };
387 const size_t sizes_num = sizeof(sizes)/sizeof(sizes[0]);
388 const int test_num = 16;
389 int progress = 0;
390 for(int i = 0; i < test_num; ++i)
391 {
392 progress = update_progress( progress, i, test_num, 0 );
393 ChessBoardGenerator cbg(sizes[i % sizes_num]);
394
395 vector<Point2f> corners_generated;
396
397 Mat cb = cbg(bg, camMat, distCoeffs, corners_generated);
398
399 if(!validateData(cbg, cb.size(), corners_generated))
400 {
401 ts->printf( cvtest::TS::LOG, "Chess board skipped - too small" );
402 continue;
403 }
404
405 /*cb = cb * 0.8 + Scalar::all(30);
406 GaussianBlur(cb, cb, Size(3, 3), 0.8); */
407 //cv::addWeighted(cb, 0.8, bg, 0.2, 20, cb);
408 //cv::namedWindow("CB"); cv::imshow("CB", cb); cv::waitKey();
409
410 vector<Point2f> corners_found;
411 int flags = i % 8; // need to check branches for all flags
412 bool found = findChessboardCorners(cb, cbg.cornersSize(), corners_found, flags);
413 if (!found)
414 {
415 ts->printf( cvtest::TS::LOG, "Chess board corners not found\n" );
416 ts->set_failed_test_info( cvtest::TS::FAIL_BAD_ACCURACY );
417 res = false;
418 return res;
419 }
420
421 double err = calcErrorMinError(cbg.cornersSize(), corners_found, corners_generated);
422 if( err > rough_success_error_level )
423 {
424 ts->printf( cvtest::TS::LOG, "bad accuracy of corner guesses" );
425 ts->set_failed_test_info( cvtest::TS::FAIL_BAD_ACCURACY );
426 res = false;
427 return res;
428 }
429 }
430
431 /* ***** negative ***** */
432 {
433 vector<Point2f> corners_found;
434 bool found = findChessboardCorners(bg, Size(8, 7), corners_found);
435 if (found)
436 res = false;
437
438 ChessBoardGenerator cbg(Size(8, 7));
439
440 vector<Point2f> cg;
441 Mat cb = cbg(bg, camMat, distCoeffs, cg);
442
443 found = findChessboardCorners(cb, Size(3, 4), corners_found);
444 if (found)
445 res = false;
446
447 Point2f c = std::accumulate(cg.begin(), cg.end(), Point2f(), plus<Point2f>()) * (1.f/cg.size());
448
449 Mat_<double> aff(2, 3);
450 aff << 1.0, 0.0, -(double)c.x, 0.0, 1.0, 0.0;
451 Mat sh;
452 warpAffine(cb, sh, aff, cb.size());
453
454 found = findChessboardCorners(sh, cbg.cornersSize(), corners_found);
455 if (found)
456 res = false;
457
458 vector< vector<Point> > cnts(1);
459 vector<Point>& cnt = cnts[0];
460 cnt.push_back(cg[ 0]); cnt.push_back(cg[0+2]);
461 cnt.push_back(cg[7+0]); cnt.push_back(cg[7+2]);
462 cv::drawContours(cb, cnts, -1, Scalar::all(128), CV_FILLED);
463
464 found = findChessboardCorners(cb, cbg.cornersSize(), corners_found);
465 if (found)
466 res = false;
467
468 cv::drawChessboardCorners(cb, cbg.cornersSize(), Mat(corners_found), found);
469 }
470#endif
471
472 return res;
473}
