96 template<class U, class V> double error(const Vec<U, 3>& v1, const Vec<V, 3>& v2) const
97 {
98 double tmp, sum = 0;
99 double nsum = 0;
100 for(int i = 0; i < 3; ++i)
101 {
102 tmp = v1[i];
103 nsum += tmp * tmp;
104
105 tmp = tmp - v2[i];
106 sum += tmp * tmp;
107
108 }
109 return sqrt(sum)/(sqrt(nsum)+1.);
110 }

112 template<class InT, class OutT> void runCase(int caseId, InT min, InT max)
113 {
114 typedef Vec<OutT, 3> out3d_t;
115
116 bool handleMissingValues = (unsigned)theRNG() % 2 == 0;
117
118 Mat_<InT> disp(Size(320, 240));
119 randu(disp, Scalar(min), Scalar(max));
120
121 if (handleMissingValues)
122 disp(disp.rows/2, disp.cols/2) = min - 1;
123
124 Mat_<double> Q(4, 4);
125 randu(Q, Scalar(-5), Scalar(5));
126
127 Mat_<out3d_t> _3dImg(disp.size());
128
129 CvMat cvdisp = disp; CvMat cv_3dImg = _3dImg; CvMat cvQ = Q;
130 cvReprojectImageTo3D( &cvdisp, &cv_3dImg, &cvQ, handleMissingValues );
131
132 if (numeric_limits<OutT>::max() == numeric_limits<float>::max())
133 reprojectImageTo3D(disp, _3dImg, Q, handleMissingValues);
134
135 for(int y = 0; y < disp.rows; ++y)
136 for(int x = 0; x < disp.cols; ++x)
137 {
138 InT d = disp(y, x);
139
140 double from[4] = {
141 static_cast<double>(x),
142 static_cast<double>(y),
143 static_cast<double>(d),
144 1.0,
145 };
146 Mat_<double> res = Q * Mat_<double>(4, 1, from);
147 res /= res(3, 0);
148
149 out3d_t pixel_exp = *(Vec3d*)res.data;
150 out3d_t pixel_out = _3dImg(y, x);
151
152 const int largeZValue = 10000; /* see documentation */
153
154 if (handleMissingValues && y == disp.rows/2 && x == disp.cols/2)
155 {
156 if (pixel_out[2] == largeZValue)
157 continue;
158
159 ts->printf(cvtest::TS::LOG, "Missing values are handled improperly\n");
160 ts->set_failed_test_info( cvtest::TS::FAIL_BAD_ACCURACY );
161 return;
162 }
163 else
164 {
165 double err = error(pixel_out, pixel_exp), t = thres<OutT>();
166 if ( err > t )
167 {
168 ts->printf(cvtest::TS::LOG, "case %d. too big error at (%d, %d): %g vs expected %g: res = (%g, %g, %g, w=%g) vs pixel_out = (%g, %g, %g)\n",
169 caseId, x, y, err, t, res(0,0), res(1,0), res(2,0), res(3,0),
170 (double)pixel_out[0], (double)pixel_out[1], (double)pixel_out[2]);
171 ts->set_failed_test_info( cvtest::TS::FAIL_BAD_ACCURACY );
172 return;
173 }
174 }
175 }
176 }
177};
