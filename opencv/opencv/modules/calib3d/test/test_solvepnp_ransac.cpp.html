49class CV_solvePnPRansac_Test : public cvtest::BaseTest
50{
51public:
52 CV_solvePnPRansac_Test()
53 {
54 eps[CV_ITERATIVE] = 1.0e-2;
55 eps[CV_EPNP] = 1.0e-2;
56 eps[CV_P3P] = 1.0e-2;
57 totalTestsCount = 10;
58 }
59 ~CV_solvePnPRansac_Test() {}
60protected:
61 void generate3DPointCloud(vector<Point3f>& points, Point3f pmin = Point3f(-1,
62 -1, 5), Point3f pmax = Point3f(1, 1, 10))
63 {
64 const Point3f delta = pmax - pmin;
65 for (size_t i = 0; i < points.size(); i++)
66 {
67 Point3f p(float(rand()) / RAND_MAX, float(rand()) / RAND_MAX,
68 float(rand()) / RAND_MAX);
69 p.x *= delta.x;
70 p.y *= delta.y;
71 p.z *= delta.z;
72 p = p + pmin;
73 points[i] = p;
74 }
75 }
76
77 void generateCameraMatrix(Mat& cameraMatrix, RNG& rng)
78 {
79 const double fcMinVal = 1e-3;
80 const double fcMaxVal = 100;
81 cameraMatrix.create(3, 3, CV_64FC1);
82 cameraMatrix.setTo(Scalar(0));
83 cameraMatrix.at<double>(0,0) = rng.uniform(fcMinVal, fcMaxVal);
84 cameraMatrix.at<double>(1,1) = rng.uniform(fcMinVal, fcMaxVal);
85 cameraMatrix.at<double>(0,2) = rng.uniform(fcMinVal, fcMaxVal);
86 cameraMatrix.at<double>(1,2) = rng.uniform(fcMinVal, fcMaxVal);
87 cameraMatrix.at<double>(2,2) = 1;
88 }
89
90 void generateDistCoeffs(Mat& distCoeffs, RNG& rng)
91 {
92 distCoeffs = Mat::zeros(4, 1, CV_64FC1);
93 for (int i = 0; i < 3; i++)
94 distCoeffs.at<double>(i,0) = rng.uniform(0.0, 1.0e-6);
95 }
96
97 void generatePose(Mat& rvec, Mat& tvec, RNG& rng)
98 {
99 const double minVal = 1.0e-3;
100 const double maxVal = 1.0;
101 rvec.create(3, 1, CV_64FC1);
102 tvec.create(3, 1, CV_64FC1);
103 for (int i = 0; i < 3; i++)
104 {
105 rvec.at<double>(i,0) = rng.uniform(minVal, maxVal);
106 tvec.at<double>(i,0) = rng.uniform(minVal, maxVal/10);
107 }
108 }
109
110 virtual bool runTest(RNG& rng, int mode, int method, const vector<Point3f>& points, const double* epsilon, double& maxError)
111 {
112 Mat rvec, tvec;
113 vector<int> inliers;
114 Mat trueRvec, trueTvec;
115 Mat intrinsics, distCoeffs;
116 generateCameraMatrix(intrinsics, rng);
117 if (mode == 0)
118 distCoeffs = Mat::zeros(4, 1, CV_64FC1);
119 else
120 generateDistCoeffs(distCoeffs, rng);
121 generatePose(trueRvec, trueTvec, rng);
122
123 vector<Point2f> projectedPoints;
124 projectedPoints.resize(points.size());
125 projectPoints(Mat(points), trueRvec, trueTvec, intrinsics, distCoeffs, projectedPoints);
126 for (size_t i = 0; i < projectedPoints.size(); i++)
127 {
128 if (i % 20 == 0)
129 {
130 projectedPoints[i] = projectedPoints[rng.uniform(0,(int)points.size()-1)];
131 }
132 }
133
134 solvePnPRansac(points, projectedPoints, intrinsics, distCoeffs, rvec, tvec,
135 false, 500, 0.5, -1, inliers, method);
136
137 bool isTestSuccess = inliers.size() >= points.size()*0.95;
138
139 double rvecDiff = norm(rvec-trueRvec), tvecDiff = norm(tvec-trueTvec);
140 isTestSuccess = isTestSuccess && rvecDiff < epsilon[method] && tvecDiff < epsilon[method];
141 double error = rvecDiff > tvecDiff ? rvecDiff : tvecDiff;
142 //cout << error << " " << inliers.size() << " " << eps[method] << endl;
143 if (error > maxError)
144 maxError = error;
145
146 return isTestSuccess;
147 }
148
149 void run(int)
150 {
151 ts->set_failed_test_info(cvtest::TS::OK);
152
153 vector<Point3f> points;
154 const int pointsCount = 500;
155 points.resize(pointsCount);
156 generate3DPointCloud(points);
157
158
159 const int methodsCount = 3;
160 RNG rng = ts->get_rng();
161
162
163 for (int mode = 0; mode < 2; mode++)
164 {
165 for (int method = 0; method < methodsCount; method++)
166 {
167 double maxError = 0;
168 int successfulTestsCount = 0;
169 for (int testIndex = 0; testIndex < totalTestsCount; testIndex++)
170 {
171 if (runTest(rng, mode, method, points, eps, maxError))
172 successfulTestsCount++;
173 }
174 //cout << maxError << " " << successfulTestsCount << endl;
175 if (successfulTestsCount < 0.7*totalTestsCount)
176 {
177 ts->printf( cvtest::TS::LOG, "Invalid accuracy for method %d, failed %d tests from %d, maximum error equals %f, distortion mode equals %d\n",
178 method, totalTestsCount - successfulTestsCount, totalTestsCount, maxError, mode);
179 ts->set_failed_test_info(cvtest::TS::FAIL_BAD_ACCURACY);
180 }
181 }
182 }
183 }
184 double eps[3];
185 int totalTestsCount;
186};

188class CV_solvePnP_Test : public CV_solvePnPRansac_Test
189{
190public:
191 CV_solvePnP_Test()
192 {
193 eps[CV_ITERATIVE] = 1.0e-6;
194 eps[CV_EPNP] = 1.0e-6;
195 eps[CV_P3P] = 1.0e-4;
196 totalTestsCount = 1000;
197 }
198
199 ~CV_solvePnP_Test() {}
200protected:
201 virtual bool runTest(RNG& rng, int mode, int method, const vector<Point3f>& points, const double* epsilon, double& maxError)
202 {
203 Mat rvec, tvec;
204 Mat trueRvec, trueTvec;
205 Mat intrinsics, distCoeffs;
206 generateCameraMatrix(intrinsics, rng);
207 if (mode == 0)
208 distCoeffs = Mat::zeros(4, 1, CV_64FC1);
209 else
210 generateDistCoeffs(distCoeffs, rng);
211 generatePose(trueRvec, trueTvec, rng);
212
213 std::vector<Point3f> opoints;
214 if (method == 2)
215 {
216 opoints = std::vector<Point3f>(points.begin(), points.begin()+4);
217 }
218 else
219 opoints = points;
220
221 vector<Point2f> projectedPoints;
222 projectedPoints.resize(opoints.size());
223 projectPoints(Mat(opoints), trueRvec, trueTvec, intrinsics, distCoeffs, projectedPoints);
224
225 solvePnP(opoints, projectedPoints, intrinsics, distCoeffs, rvec, tvec,
226 false, method);
227
228 double rvecDiff = norm(rvec-trueRvec), tvecDiff = norm(tvec-trueTvec);
229 bool isTestSuccess = rvecDiff < epsilon[method] && tvecDiff < epsilon[method];
230
231 double error = rvecDiff > tvecDiff ? rvecDiff : tvecDiff;
232 if (error > maxError)
233 maxError = error;
234
235 return isTestSuccess;
236 }
237};

249 Mat object(1, count, CV_32FC3);

252 Mat camera_mat(3, 3, CV_32FC1);
253 randu(camera_mat, 0.5, 1);
254 camera_mat.at<float>(0, 1) = 0.f;
255 camera_mat.at<float>(1, 0) = 0.f;
256 camera_mat.at<float>(2, 0) = 0.f;
257 camera_mat.at<float>(2, 1) = 0.f;
258
259 Mat dist_coef(1, 8, CV_32F, cv::Scalar::all(0));
260
261 vector<cv::Point2f> image_vec;
262 Mat rvec_gold(1, 3, CV_32FC1);
263 randu(rvec_gold, 0, 1);
264 Mat tvec_gold(1, 3, CV_32FC1);
265 randu(tvec_gold, 0, 1);
266 projectPoints(object, rvec_gold, tvec_gold, camera_mat, dist_coef, image_vec);
267
268 Mat image(1, count, CV_32FC2, &image_vec[0]);
269
270 Mat rvec1, rvec2;
271 Mat tvec1, tvec2;
272
273 {
274 // limit concurrency to get deterministic result
275 cv::theRNG().state = 20121010;
276 cv::Ptr<tbb::task_scheduler_init> one_thread = new tbb::task_scheduler_init(1);
277 solvePnPRansac(object, image, camera_mat, dist_coef, rvec1, tvec1);
278 }
279
280 if(1)
281 {
282 Mat rvec;
283 Mat tvec;
284 // parallel executions
285 for(int i = 0; i < 10; ++i)
286 {
287 cv::theRNG().state = 20121010;
288 solvePnPRansac(object, image, camera_mat, dist_coef, rvec, tvec);
289 }
290 }
291
292 {
293 // single thread again
294 cv::theRNG().state = 20121010;
295 cv::Ptr<tbb::task_scheduler_init> one_thread = new tbb::task_scheduler_init(1);
296 solvePnPRansac(object, image, camera_mat, dist_coef, rvec2, tvec2);
297 }
298
299 double rnorm = cv::norm(rvec1, rvec2, NORM_INF);
300 double tnorm = cv::norm(tvec1, tvec2, NORM_INF);
301
302 EXPECT_LT(rnorm, 1e-6);
303 EXPECT_LT(tnorm, 1e-6);
304
305}
