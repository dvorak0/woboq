145/*!
146 The Main Class for Image Filtering.
147
148 The class can be used to apply an arbitrary filtering operation to an image.
149 It contains all the necessary intermediate buffers, it computes extrapolated values
150 of the "virtual" pixels outside of the image etc.
151 Pointers to the initialized cv::FilterEngine instances
152 are returned by various OpenCV functions, such as cv::createSeparableLinearFilter(),
153 cv::createLinearFilter(), cv::createGaussianFilter(), cv::createDerivFilter(),
154 cv::createBoxFilter() and cv::createMorphologyFilter().
155
156 Using the class you can process large images by parts and build complex pipelines
157 that include filtering as some of the stages. If all you need is to apply some pre-defined
158 filtering operation, you may use cv::filter2D(), cv::erode(), cv::dilate() etc.
159 functions that create FilterEngine internally.
160
161 Here is the example on how to use the class to implement Laplacian operator, which is the sum of
162 second-order derivatives. More complex variant for different types is implemented in cv::Laplacian().
163
164 \code
165 void laplace_f(const Mat& src, Mat& dst)
166 {
167 CV_Assert( src.type() == CV_32F );
168 // make sure the destination array has the proper size and type
169 dst.create(src.size(), src.type());
170
171 // get the derivative and smooth kernels for d2I/dx2.
172 // for d2I/dy2 we could use the same kernels, just swapped
173 Mat kd, ks;
174 getSobelKernels( kd, ks, 2, 0, ksize, false, ktype );
175
176 // let's process 10 source rows at once
177 int DELTA = std::min(10, src.rows);
178 Ptr<FilterEngine> Fxx = createSeparableLinearFilter(src.type(),
179 dst.type(), kd, ks, Point(-1,-1), 0, borderType, borderType, Scalar() );
180 Ptr<FilterEngine> Fyy = createSeparableLinearFilter(src.type(),
181 dst.type(), ks, kd, Point(-1,-1), 0, borderType, borderType, Scalar() );
182
183 int y = Fxx->start(src), dsty = 0, dy = 0;
184 Fyy->start(src);
185 const uchar* sptr = src.data + y*src.step;
186
187 // allocate the buffers for the spatial image derivatives;
188 // the buffers need to have more than DELTA rows, because at the
189 // last iteration the output may take max(kd.rows-1,ks.rows-1)
190 // rows more than the input.
191 Mat Ixx( DELTA + kd.rows - 1, src.cols, dst.type() );
192 Mat Iyy( DELTA + kd.rows - 1, src.cols, dst.type() );
193
194 // inside the loop we always pass DELTA rows to the filter
195 // (note that the "proceed" method takes care of possibe overflow, since
196 // it was given the actual image height in the "start" method)
197 // on output we can get:
198 // * < DELTA rows (the initial buffer accumulation stage)
199 // * = DELTA rows (settled state in the middle)
200 // * > DELTA rows (then the input image is over, but we generate
201 // "virtual" rows using the border mode and filter them)
202 // this variable number of output rows is dy.
203 // dsty is the current output row.
204 // sptr is the pointer to the first input row in the portion to process
205 for( ; dsty < dst.rows; sptr += DELTA*src.step, dsty += dy )
206 {
207 Fxx->proceed( sptr, (int)src.step, DELTA, Ixx.data, (int)Ixx.step );
208 dy = Fyy->proceed( sptr, (int)src.step, DELTA, d2y.data, (int)Iyy.step );
209 if( dy > 0 )
210 {
211 Mat dstripe = dst.rowRange(dsty, dsty + dy);
212 add(Ixx.rowRange(0, dy), Iyy.rowRange(0, dy), dstripe);
213 }
214 }
215 }
216 \endcode
217*/
218class CV_EXPORTS FilterEngine
219{
220public:
221 //! the default constructor
222 FilterEngine();
223 //! the full constructor. Either _filter2D or both _rowFilter and _columnFilter must be non-empty.
224 FilterEngine(const Ptr<BaseFilter>& _filter2D,
225 const Ptr<BaseRowFilter>& _rowFilter,
226 const Ptr<BaseColumnFilter>& _columnFilter,
227 int srcType, int dstType, int bufType,
228 int _rowBorderType=BORDER_REPLICATE,
229 int _columnBorderType=-1,
230 const Scalar& _borderValue=Scalar());
231 //! the destructor
232 virtual ~FilterEngine();
233 //! reinitializes the engine. The previously assigned filters are released.
234 void init(const Ptr<BaseFilter>& _filter2D,
235 const Ptr<BaseRowFilter>& _rowFilter,
236 const Ptr<BaseColumnFilter>& _columnFilter,
237 int srcType, int dstType, int bufType,
238 int _rowBorderType=BORDER_REPLICATE, int _columnBorderType=-1,
239 const Scalar& _borderValue=Scalar());
240 //! starts filtering of the specified ROI of an image of size wholeSize.
241 virtual int start(Size wholeSize, Rect roi, int maxBufRows=-1);
242 //! starts filtering of the specified ROI of the specified image.
243 virtual int start(const Mat& src, const Rect& srcRoi=Rect(0,0,-1,-1),
244 bool isolated=false, int maxBufRows=-1);
245 //! processes the next srcCount rows of the image.
246 virtual int proceed(const uchar* src, int srcStep, int srcCount,
247 uchar* dst, int dstStep);
248 //! applies filter to the specified ROI of the image. if srcRoi=(0,0,-1,-1), the whole image is filtered.
249 virtual void apply( const Mat& src, Mat& dst,
250 const Rect& srcRoi=Rect(0,0,-1,-1),
251 Point dstOfs=Point(0,0),
252 bool isolated=false);
253 //! returns true if the filter is separable
254 bool isSeparable() const { return (const BaseFilter*)filter2D == 0; }
255 //! returns the number
256 int remainingInputRows() const;
257 int remainingOutputRows() const;
258
259 int srcType, dstType, bufType;
260 Size ksize;
261 Point anchor;
262 int maxWidth;
263 Size wholeSize;
264 Rect roi;
265 int dx1, dx2;
266 int rowBorderType, columnBorderType;
267 vector<int> borderTab;
268 int borderElemSize;
269 vector<uchar> ringBuf;
270 vector<uchar> srcRow;
271 vector<uchar> constBorderValue;
272 vector<uchar> constBorderRow;
273 int bufStep, startY, startY0, endY, rowCount, dstY;
274 vector<uchar*> rows;
275
276 Ptr<BaseFilter> filter2D;
277 Ptr<BaseRowFilter> rowFilter;
278 Ptr<BaseColumnFilter> columnFilter;
279};
