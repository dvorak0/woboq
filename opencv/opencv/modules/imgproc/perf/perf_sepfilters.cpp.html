32PERF_TEST_P(Size_MatType_dx_dy_Border3x3, sobelFilter,
33 testing::Combine(
34 testing::Values(FILTER_SRC_SIZES),
35 testing::Values(CV_16S, CV_32F),
36 testing::Values(make_tuple(0, 1), make_tuple(1, 0), make_tuple(1, 1), make_tuple(0, 2), make_tuple(2, 0), make_tuple(2, 2)),
37 BorderType3x3::all()
38 )
39 )
40{
41 Size size = get<0>(GetParam());
42 int ddepth = get<1>(GetParam());
43 int dx = get<0>(get<2>(GetParam()));
44 int dy = get<1>(get<2>(GetParam()));
45 BorderType3x3 border = get<3>(GetParam());
46
47 Mat src(size, CV_8U);
48 Mat dst(size, ddepth);
49
50 declare.in(src, WARMUP_RNG).out(dst);
51
52 TEST_CYCLE() Sobel(src, dst, ddepth, dx, dy, 3, 1, 0, border);
53
54 SANITY_CHECK(dst);
55}
