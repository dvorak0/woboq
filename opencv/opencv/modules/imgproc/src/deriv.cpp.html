48/* lightweight convolution with 3x3 kernel */
49void icvSepConvSmall3_32f( float* src, int src_step, float* dst, int dst_step,
50 CvSize src_size, const float* kx, const float* ky, float* buffer )
51{
52 int dst_width, buffer_step = 0;
53 int x, y;
54 bool fast_kx = true, fast_ky = true;
55
56 assert( src && dst && src_size.width > 2 && src_size.height > 2 &&
57 (src_step & 3) == 0 && (dst_step & 3) == 0 &&
58 (kx || ky) && (buffer || !kx || !ky));
59
60 src_step /= sizeof(src[0]);
61 dst_step /= sizeof(dst[0]);
62
63 dst_width = src_size.width - 2;
64
65 if( !kx )
66 {
67 /* set vars, so that vertical convolution
68 will write results into destination ROI and
69 horizontal convolution won't run */
70 src_size.width = dst_width;
71 buffer_step = dst_step;
72 buffer = dst;
73 dst_width = 0;
74 }
75 else
76 fast_kx = kx[1] == 0.f && kx[0] == -kx[2] && kx[0] == -1.f;
77
78 assert( src_step >= src_size.width && dst_step >= dst_width );
79
80 src_size.height -= 2;
81 if( !ky )
82 {
83 /* set vars, so that vertical convolution won't run and
84 horizontal convolution will write results into destination ROI */
85 src_size.height += 2;
86 buffer_step = src_step;
87 buffer = src;
88 src_size.width = 0;
89 }
90 else
91 fast_ky = ky[1] == 0.f && ky[0] == -ky[2] && ky[0] == -1.f;
92
93 for( y = 0; y < src_size.height; y++, src += src_step,
94 dst += dst_step,
95 buffer += buffer_step )
96 {
97 float* src2 = src + src_step;
98 float* src3 = src + src_step*2;
99 if( fast_ky )
100 for( x = 0; x < src_size.width; x++ )
101 {
102 buffer[x] = (float)(src3[x] - src[x]);
103 }
104 else
105 for( x = 0; x < src_size.width; x++ )
106 {
107 buffer[x] = (float)(ky[0]*src[x] + ky[1]*src2[x] + ky[2]*src3[x]);
108 }
109
110 if( fast_kx )
111 for( x = 0; x < dst_width; x++ )
112 {
113 dst[x] = (float)(buffer[x+2] - buffer[x]);
114 }
115 else
116 for( x = 0; x < dst_width; x++ )
117 {
118 dst[x] = (float)(kx[0]*buffer[x] + kx[1]*buffer[x+1] + kx[2]*buffer[x+2]);
119 }
120 }
121}

131static void getScharrKernels( OutputArray _kx, OutputArray _ky,
132 int dx, int dy, bool normalize, int ktype )
133{
134 const int ksize = 3;
135
136 CV_Assert( ktype == CV_32F || ktype == CV_64F );
137 _kx.create(ksize, 1, ktype, -1, true);
138 _ky.create(ksize, 1, ktype, -1, true);
139 Mat kx = _kx.getMat();
140 Mat ky = _ky.getMat();
141
142 CV_Assert( dx >= 0 && dy >= 0 && dx+dy == 1 );
143
144 for( int k = 0; k < 2; k++ )
145 {
146 Mat* kernel = k == 0 ? &kx : &ky;
147 int order = k == 0 ? dx : dy;
148 int kerI[3];
149
150 if( order == 0 )
151 kerI[0] = 3, kerI[1] = 10, kerI[2] = 3;
152 else if( order == 1 )
153 kerI[0] = -1, kerI[1] = 0, kerI[2] = 1;
154
155 Mat temp(kernel->rows, kernel->cols, CV_32S, &kerI[0]);
156 double scale = !normalize || order == 1 ? 1. : 1./32;
157 temp.convertTo(*kernel, ktype, scale);
158 }
159}

162static void getSobelKernels( OutputArray _kx, OutputArray _ky,
163 int dx, int dy, int _ksize, bool normalize, int ktype )
164{
165 int i, j, ksizeX = _ksize, ksizeY = _ksize;
166 if( ksizeX == 1 && dx > 0 )
167 ksizeX = 3;
168 if( ksizeY == 1 && dy > 0 )
169 ksizeY = 3;
170
171 CV_Assert( ktype == CV_32F || ktype == CV_64F );
172
173 _kx.create(ksizeX, 1, ktype, -1, true);
174 _ky.create(ksizeY, 1, ktype, -1, true);
175 Mat kx = _kx.getMat();
176 Mat ky = _ky.getMat();
177
178 if( _ksize % 2 == 0 || _ksize > 31 )
179 CV_Error( CV_StsOutOfRange, "The kernel size must be odd and not larger than 31" );
180 vector<int> kerI(std::max(ksizeX, ksizeY) + 1);
181
182 CV_Assert( dx >= 0 && dy >= 0 && dx+dy > 0 );
183
184 for( int k = 0; k < 2; k++ )
185 {
186 Mat* kernel = k == 0 ? &kx : &ky;
187 int order = k == 0 ? dx : dy;
188 int ksize = k == 0 ? ksizeX : ksizeY;
189
190 CV_Assert( ksize > order );
191
192 if( ksize == 1 )
193 kerI[0] = 1;
194 else if( ksize == 3 )
195 {
196 if( order == 0 )
197 kerI[0] = 1, kerI[1] = 2, kerI[2] = 1;
198 else if( order == 1 )
199 kerI[0] = -1, kerI[1] = 0, kerI[2] = 1;
200 else
201 kerI[0] = 1, kerI[1] = -2, kerI[2] = 1;
202 }
203 else
204 {
205 int oldval, newval;
206 kerI[0] = 1;
207 for( i = 0; i < ksize; i++ )
208 kerI[i+1] = 0;
209
210 for( i = 0; i < ksize - order - 1; i++ )
211 {
212 oldval = kerI[0];
213 for( j = 1; j <= ksize; j++ )
214 {
215 newval = kerI[j]+kerI[j-1];
216 kerI[j-1] = oldval;
217 oldval = newval;
218 }
219 }
220
221 for( i = 0; i < order; i++ )
222 {
223 oldval = -kerI[0];
224 for( j = 1; j <= ksize; j++ )
225 {
226 newval = kerI[j-1] - kerI[j];
227 kerI[j-1] = oldval;
228 oldval = newval;
229 }
230 }
231 }
232
233 Mat temp(kernel->rows, kernel->cols, CV_32S, &kerI[0]);
234 double scale = !normalize ? 1. : 1./(1 << (ksize-order-1));
235 temp.convertTo(*kernel, ktype, scale);
236 }
237}

241void cv::getDerivKernels( OutputArray kx, OutputArray ky, int dx, int dy,
242 int ksize, bool normalize, int ktype )
243{
244 if( ksize <= 0 )
245 getScharrKernels( kx, ky, dx, dy, normalize, ktype );
246 else
247 getSobelKernels( kx, ky, dx, dy, ksize, normalize, ktype );
248}

251cv::Ptr<cv::FilterEngine> cv::createDerivFilter(int srcType, int dstType,
252 int dx, int dy, int ksize, int borderType )
253{
254 Mat kx, ky;
255 getDerivKernels( kx, ky, dx, dy, ksize, false, CV_32F );
256 return createSeparableLinearFilter(srcType, dstType,
257 kx, ky, Point(-1,-1), 0, borderType );
258}

265static bool IPPDerivScharr(const Mat& src, Mat& dst, int ddepth, int dx, int dy, double scale)
266{
267 int bufSize = 0;
268 cv::AutoBuffer<char> buffer;
269 IppiSize roi = ippiSize(src.cols, src.rows);
270
271 if( ddepth < 0 )
272 ddepth = src.depth();
273
274 dst.create( src.size(), CV_MAKETYPE(ddepth, src.channels()) );
275
276 switch(src.type())
277 {
278 case CV_8U:
279 {
280 if(scale != 1)
281 return false;
282
283 switch(dst.type())
284 {
285 case CV_16S:
286 {
287 if((dx == 1) && (dy == 0))
288 {
289 ippiFilterScharrVertGetBufferSize_8u16s_C1R(roi,&bufSize);
290 buffer.allocate(bufSize);
291
292 ippiFilterScharrVertBorder_8u16s_C1R((const Ipp8u*)src.data, (int)src.step,
293 (Ipp16s*)dst.data, (int)dst.step, roi, ippBorderRepl, 0, (Ipp8u*)(char*)buffer);
294
295 return true;
296 }
297
298 if((dx == 0) && (dy == 1))
299 {
300 ippiFilterScharrHorizGetBufferSize_8u16s_C1R(roi,&bufSize);
301 buffer.allocate(bufSize);
302
303 ippiFilterScharrHorizBorder_8u16s_C1R((const Ipp8u*)src.data, (int)src.step,
304 (Ipp16s*)dst.data, (int)dst.step, roi, ippBorderRepl, 0, (Ipp8u*)(char*)buffer);
305
306 return true;
307 }
308 }
309
310 default:
311 return false;
312 }
313 }
314
315 case CV_32F:
316 {
317 switch(dst.type())
318 {
319 case CV_32F:
320 if((dx == 1) && (dy == 0))
321 {
322 ippiFilterScharrVertGetBufferSize_32f_C1R(ippiSize(src.cols, src.rows),&bufSize);
323 buffer.allocate(bufSize);
324
325 ippiFilterScharrVertBorder_32f_C1R((const Ipp32f*)src.data, (int)src.step,
326 (Ipp32f*)dst.data, (int)dst.step, ippiSize(src.cols, src.rows),
327 ippBorderRepl, 0, (Ipp8u*)(char*)buffer);
328 if(scale != 1)
329 /* IPP is fast, so MulC produce very little perf degradation */
330 ippiMulC_32f_C1IR((Ipp32f)scale, (Ipp32f*)dst.data, (int)dst.step, ippiSize(dst.cols*dst.channels(), dst.rows));
331
332 return true;
333 }
334
335 if((dx == 0) && (dy == 1))
336 {
337 ippiFilterScharrHorizGetBufferSize_32f_C1R(ippiSize(src.cols, src.rows),&bufSize);
338 buffer.allocate(bufSize);
339
340 ippiFilterScharrHorizBorder_32f_C1R((const Ipp32f*)src.data, (int)src.step,
341 (Ipp32f*)dst.data, (int)dst.step, ippiSize(src.cols, src.rows),
342 ippBorderRepl, 0, (Ipp8u*)(char*)buffer);
343 if(scale != 1)
344 ippiMulC_32f_C1IR((Ipp32f)scale, (Ipp32f *)dst.data, (int)dst.step, ippiSize(dst.cols*dst.channels(), dst.rows));
345
346 return true;
347 }
348
349 default:
350 return false;
351 }
352 }
353
354 default:
355 return false;
356 }

490void cv::Sobel( InputArray _src, OutputArray _dst, int ddepth, int dx, int dy,
491 int ksize, double scale, double delta, int borderType )
492{
493 Mat src = _src.getMat();
494 if (ddepth < 0)
495 ddepth = src.depth();
496 _dst.create( src.size(), CV_MAKETYPE(ddepth, src.channels()) );
497 Mat dst = _dst.getMat();
498
499#ifdef HAVE_TEGRA_OPTIMIZATION
500 if (scale == 1.0 && delta == 0)
501 {
502 if (ksize == 3 && tegra::sobel3x3(src, dst, dx, dy, borderType))
503 return;
504 if (ksize == -1 && tegra::scharr(src, dst, dx, dy, borderType))
505 return;
506 }
507#endif
508
509#if defined (HAVE_IPP) && (IPP_VERSION_MAJOR >= 7)
510 if(dx < 3 && dy < 3 && src.channels() == 1 && borderType == 1)
511 {
512 if(IPPDeriv(src, dst, ddepth, dx, dy, ksize,scale))
513 return;
514 }
515#endif
516 int ktype = std::max(CV_32F, std::max(ddepth, src.depth()));
517
518 Mat kx, ky;
519 getDerivKernels( kx, ky, dx, dy, ksize, false, ktype );
520 if( scale != 1 )
521 {
522 // usually the smoothing part is the slowest to compute,
523 // so try to scale it instead of the faster differenciating part
524 if( dx == 0 )
525 kx *= scale;
526 else
527 ky *= scale;
528 }
529 sepFilter2D( src, dst, ddepth, kx, ky, Point(-1,-1), delta, borderType );
530}

533void cv::Scharr( InputArray _src, OutputArray _dst, int ddepth, int dx, int dy,
534 double scale, double delta, int borderType )
535{
536 Mat src = _src.getMat();
537 if (ddepth < 0)
538 ddepth = src.depth();
539 _dst.create( src.size(), CV_MAKETYPE(ddepth, src.channels()) );
540 Mat dst = _dst.getMat();
541
542#ifdef HAVE_TEGRA_OPTIMIZATION
543 if (scale == 1.0 && delta == 0)
544 if (tegra::scharr(src, dst, dx, dy, borderType))
545 return;
546#endif
547
548#if defined (HAVE_IPP) && (IPP_VERSION_MAJOR >= 7)
549 if(dx < 2 && dy < 2 && src.channels() == 1 && borderType == 1)
550 {
551 if(IPPDerivScharr(src, dst, ddepth, dx, dy, scale))
552 return;
553 }
554#endif
555 int ktype = std::max(CV_32F, std::max(ddepth, src.depth()));
556
557 Mat kx, ky;
558 getScharrKernels( kx, ky, dx, dy, false, ktype );
559 if( scale != 1 )
560 {
561 // usually the smoothing part is the slowest to compute,
562 // so try to scale it instead of the faster differenciating part
563 if( dx == 0 )
564 kx *= scale;
565 else
566 ky *= scale;
567 }
568 sepFilter2D( src, dst, ddepth, kx, ky, Point(-1,-1), delta, borderType );
569}

572void cv::Laplacian( InputArray _src, OutputArray _dst, int ddepth, int ksize,
573 double scale, double delta, int borderType )
574{
575 Mat src = _src.getMat();
576 if (ddepth < 0)
577 ddepth = src.depth();
578 _dst.create( src.size(), CV_MAKETYPE(ddepth, src.channels()) );
579 Mat dst = _dst.getMat();
580
581#ifdef HAVE_TEGRA_OPTIMIZATION
582 if (scale == 1.0 && delta == 0)
583 {
584 if (ksize == 1 && tegra::laplace1(src, dst, borderType))
585 return;
586 if (ksize == 3 && tegra::laplace3(src, dst, borderType))
587 return;
588 if (ksize == 5 && tegra::laplace5(src, dst, borderType))
589 return;
590 }
591#endif
592
593 if( ksize == 1 || ksize == 3 )
594 {
595 float K[2][9] =
596 {{0, 1, 0, 1, -4, 1, 0, 1, 0},
597 {2, 0, 2, 0, -8, 0, 2, 0, 2}};
598 Mat kernel(3, 3, CV_32F, K[ksize == 3]);
599 if( scale != 1 )
600 kernel *= scale;
601 filter2D( src, dst, ddepth, kernel, Point(-1,-1), delta, borderType );
602 }
603 else
604 {
605 const size_t STRIPE_SIZE = 1 << 14;
606
607 int depth = src.depth();
608 int ktype = std::max(CV_32F, std::max(ddepth, depth));
609 int wdepth = depth == CV_8U && ksize <= 5 ? CV_16S : depth <= CV_32F ? CV_32F : CV_64F;
610 int wtype = CV_MAKETYPE(wdepth, src.channels());
611 Mat kd, ks;
612 getSobelKernels( kd, ks, 2, 0, ksize, false, ktype );
613 if( ddepth < 0 )
614 ddepth = src.depth();
615 int dtype = CV_MAKETYPE(ddepth, src.channels());
616
617 int dy0 = std::min(std::max((int)(STRIPE_SIZE/(getElemSize(src.type())*src.cols)), 1), src.rows);
618 Ptr<FilterEngine> fx = createSeparableLinearFilter(src.type(),
619 wtype, kd, ks, Point(-1,-1), 0, borderType, borderType, Scalar() );
620 Ptr<FilterEngine> fy = createSeparableLinearFilter(src.type(),
621 wtype, ks, kd, Point(-1,-1), 0, borderType, borderType, Scalar() );
622
623 int y = fx->start(src), dsty = 0, dy = 0;
624 fy->start(src);
625 const uchar* sptr = src.data + y*src.step;
626
627 Mat d2x( dy0 + kd.rows - 1, src.cols, wtype );
628 Mat d2y( dy0 + kd.rows - 1, src.cols, wtype );
629
630 for( ; dsty < src.rows; sptr += dy0*src.step, dsty += dy )
631 {
632 fx->proceed( sptr, (int)src.step, dy0, d2x.data, (int)d2x.step );
633 dy = fy->proceed( sptr, (int)src.step, dy0, d2y.data, (int)d2y.step );
634 if( dy > 0 )
635 {
636 Mat dstripe = dst.rowRange(dsty, dsty + dy);
637 d2x.rows = d2y.rows = dy; // modify the headers, which should work
638 d2x += d2y;
639 d2x.convertTo( dstripe, dtype, scale, delta );
640 }
641 }
642 }
643}
