41static void magSpectrums( InputArray _src, OutputArray _dst)
42{
43 Mat src = _src.getMat();
44 int depth = src.depth(), cn = src.channels(), type = src.type();
45 int rows = src.rows, cols = src.cols;
46 int j, k;
47
48 CV_Assert( type == CV_32FC1 || type == CV_32FC2 || type == CV_64FC1 || type == CV_64FC2 );
49
50 if(src.depth() == CV_32F)
51 _dst.create( src.rows, src.cols, CV_32FC1 );
52 else
53 _dst.create( src.rows, src.cols, CV_64FC1 );
54
55 Mat dst = _dst.getMat();
56 dst.setTo(0);//Mat elements are not equal to zero by default!
57
58 bool is_1d = (rows == 1 || (cols == 1 && src.isContinuous() && dst.isContinuous()));
59
60 if( is_1d )
61 cols = cols + rows - 1, rows = 1;
62
63 int ncols = cols*cn;
64 int j0 = cn == 1;
65 int j1 = ncols - (cols % 2 == 0 && cn == 1);
66
67 if( depth == CV_32F )
68 {
69 const float* dataSrc = (const float*)src.data;
70 float* dataDst = (float*)dst.data;
71
72 size_t stepSrc = src.step/sizeof(dataSrc[0]);
73 size_t stepDst = dst.step/sizeof(dataDst[0]);
74
75 if( !is_1d && cn == 1 )
76 {
77 for( k = 0; k < (cols % 2 ? 1 : 2); k++ )
78 {
79 if( k == 1 )
80 dataSrc += cols - 1, dataDst += cols - 1;
81 dataDst[0] = dataSrc[0]*dataSrc[0];
82 if( rows % 2 == 0 )
83 dataDst[(rows-1)*stepDst] = dataSrc[(rows-1)*stepSrc]*dataSrc[(rows-1)*stepSrc];
84
85 for( j = 1; j <= rows - 2; j += 2 )
86 {
87 dataDst[j*stepDst] = (float)std::sqrt((double)dataSrc[j*stepSrc]*dataSrc[j*stepSrc] +
88 (double)dataSrc[(j+1)*stepSrc]*dataSrc[(j+1)*stepSrc]);
89 }
90
91 if( k == 1 )
92 dataSrc -= cols - 1, dataDst -= cols - 1;
93 }
94 }
95
96 for( ; rows--; dataSrc += stepSrc, dataDst += stepDst )
97 {
98 if( is_1d && cn == 1 )
99 {
100 dataDst[0] = dataSrc[0]*dataSrc[0];
101 if( cols % 2 == 0 )
102 dataDst[j1] = dataSrc[j1]*dataSrc[j1];
103 }
104
105 for( j = j0; j < j1; j += 2 )
106 {
107 dataDst[j] = (float)std::sqrt((double)dataSrc[j]*dataSrc[j] + (double)dataSrc[j+1]*dataSrc[j+1]);
108 }
109 }
110 }
111 else
112 {
113 const double* dataSrc = (const double*)src.data;
114 double* dataDst = (double*)dst.data;
115
116 size_t stepSrc = src.step/sizeof(dataSrc[0]);
117 size_t stepDst = dst.step/sizeof(dataDst[0]);
118
119 if( !is_1d && cn == 1 )
120 {
121 for( k = 0; k < (cols % 2 ? 1 : 2); k++ )
122 {
123 if( k == 1 )
124 dataSrc += cols - 1, dataDst += cols - 1;
125 dataDst[0] = dataSrc[0]*dataSrc[0];
126 if( rows % 2 == 0 )
127 dataDst[(rows-1)*stepDst] = dataSrc[(rows-1)*stepSrc]*dataSrc[(rows-1)*stepSrc];
128
129 for( j = 1; j <= rows - 2; j += 2 )
130 {
131 dataDst[j*stepDst] = std::sqrt(dataSrc[j*stepSrc]*dataSrc[j*stepSrc] +
132 dataSrc[(j+1)*stepSrc]*dataSrc[(j+1)*stepSrc]);
133 }
134
135 if( k == 1 )
136 dataSrc -= cols - 1, dataDst -= cols - 1;
137 }
138 }
139
140 for( ; rows--; dataSrc += stepSrc, dataDst += stepDst )
141 {
142 if( is_1d && cn == 1 )
143 {
144 dataDst[0] = dataSrc[0]*dataSrc[0];
145 if( cols % 2 == 0 )
146 dataDst[j1] = dataSrc[j1]*dataSrc[j1];
147 }
148
149 for( j = j0; j < j1; j += 2 )
150 {
151 dataDst[j] = std::sqrt(dataSrc[j]*dataSrc[j] + dataSrc[j+1]*dataSrc[j+1]);
152 }
153 }
154 }

157static void divSpectrums( InputArray _srcA, InputArray _srcB, OutputArray _dst, int flags, bool conjB)
158{
159 Mat srcA = _srcA.getMat(), srcB = _srcB.getMat();
160 int depth = srcA.depth(), cn = srcA.channels(), type = srcA.type();
161 int rows = srcA.rows, cols = srcA.cols;
162 int j, k;
163
164 CV_Assert( type == srcB.type() && srcA.size() == srcB.size() );
165 CV_Assert( type == CV_32FC1 || type == CV_32FC2 || type == CV_64FC1 || type == CV_64FC2 );
166
167 _dst.create( srcA.rows, srcA.cols, type );
168 Mat dst = _dst.getMat();
169
170 CV_Assert(dst.data != srcA.data); // non-inplace check
171 CV_Assert(dst.data != srcB.data); // non-inplace check
172
173 bool is_1d = (flags & DFT_ROWS) || (rows == 1 || (cols == 1 &&
174 srcA.isContinuous() && srcB.isContinuous() && dst.isContinuous()));
175
176 if( is_1d && !(flags & DFT_ROWS) )
177 cols = cols + rows - 1, rows = 1;
178
179 int ncols = cols*cn;
180 int j0 = cn == 1;
181 int j1 = ncols - (cols % 2 == 0 && cn == 1);
182
183 if( depth == CV_32F )
184 {
185 const float* dataA = (const float*)srcA.data;
186 const float* dataB = (const float*)srcB.data;
187 float* dataC = (float*)dst.data;
188 float eps = FLT_EPSILON; // prevent div0 problems
189
190 size_t stepA = srcA.step/sizeof(dataA[0]);
191 size_t stepB = srcB.step/sizeof(dataB[0]);
192 size_t stepC = dst.step/sizeof(dataC[0]);
193
194 if( !is_1d && cn == 1 )
195 {
196 for( k = 0; k < (cols % 2 ? 1 : 2); k++ )
197 {
198 if( k == 1 )
199 dataA += cols - 1, dataB += cols - 1, dataC += cols - 1;
200 dataC[0] = dataA[0] / (dataB[0] + eps);
201 if( rows % 2 == 0 )
202 dataC[(rows-1)*stepC] = dataA[(rows-1)*stepA] / (dataB[(rows-1)*stepB] + eps);
203 if( !conjB )
204 for( j = 1; j <= rows - 2; j += 2 )
205 {
206 double denom = (double)dataB[j*stepB]*dataB[j*stepB] +
207 (double)dataB[(j+1)*stepB]*dataB[(j+1)*stepB] + (double)eps;
208
209 double re = (double)dataA[j*stepA]*dataB[j*stepB] +
210 (double)dataA[(j+1)*stepA]*dataB[(j+1)*stepB];
211
212 double im = (double)dataA[(j+1)*stepA]*dataB[j*stepB] -
213 (double)dataA[j*stepA]*dataB[(j+1)*stepB];
214
215 dataC[j*stepC] = (float)(re / denom);
216 dataC[(j+1)*stepC] = (float)(im / denom);
217 }
218 else
219 for( j = 1; j <= rows - 2; j += 2 )
220 {
221
222 double denom = (double)dataB[j*stepB]*dataB[j*stepB] +
223 (double)dataB[(j+1)*stepB]*dataB[(j+1)*stepB] + (double)eps;
224
225 double re = (double)dataA[j*stepA]*dataB[j*stepB] -
226 (double)dataA[(j+1)*stepA]*dataB[(j+1)*stepB];
227
228 double im = (double)dataA[(j+1)*stepA]*dataB[j*stepB] +
229 (double)dataA[j*stepA]*dataB[(j+1)*stepB];
230
231 dataC[j*stepC] = (float)(re / denom);
232 dataC[(j+1)*stepC] = (float)(im / denom);
233 }
234 if( k == 1 )
235 dataA -= cols - 1, dataB -= cols - 1, dataC -= cols - 1;
236 }
237 }
238
239 for( ; rows--; dataA += stepA, dataB += stepB, dataC += stepC )
240 {
241 if( is_1d && cn == 1 )
242 {
243 dataC[0] = dataA[0] / (dataB[0] + eps);
244 if( cols % 2 == 0 )
245 dataC[j1] = dataA[j1] / (dataB[j1] + eps);
246 }
247
248 if( !conjB )
249 for( j = j0; j < j1; j += 2 )
250 {
251 double denom = (double)(dataB[j]*dataB[j] + dataB[j+1]*dataB[j+1] + eps);
252 double re = (double)(dataA[j]*dataB[j] + dataA[j+1]*dataB[j+1]);
253 double im = (double)(dataA[j+1]*dataB[j] - dataA[j]*dataB[j+1]);
254 dataC[j] = (float)(re / denom);
255 dataC[j+1] = (float)(im / denom);
256 }
257 else
258 for( j = j0; j < j1; j += 2 )
259 {
260 double denom = (double)(dataB[j]*dataB[j] + dataB[j+1]*dataB[j+1] + eps);
261 double re = (double)(dataA[j]*dataB[j] - dataA[j+1]*dataB[j+1]);
262 double im = (double)(dataA[j+1]*dataB[j] + dataA[j]*dataB[j+1]);
263 dataC[j] = (float)(re / denom);
264 dataC[j+1] = (float)(im / denom);
265 }
266 }
267 }
268 else
269 {
270 const double* dataA = (const double*)srcA.data;
271 const double* dataB = (const double*)srcB.data;
272 double* dataC = (double*)dst.data;
273 double eps = DBL_EPSILON; // prevent div0 problems
274
275 size_t stepA = srcA.step/sizeof(dataA[0]);
276 size_t stepB = srcB.step/sizeof(dataB[0]);
277 size_t stepC = dst.step/sizeof(dataC[0]);
278
279 if( !is_1d && cn == 1 )
280 {
281 for( k = 0; k < (cols % 2 ? 1 : 2); k++ )
282 {
283 if( k == 1 )
284 dataA += cols - 1, dataB += cols - 1, dataC += cols - 1;
285 dataC[0] = dataA[0] / (dataB[0] + eps);
286 if( rows % 2 == 0 )
287 dataC[(rows-1)*stepC] = dataA[(rows-1)*stepA] / (dataB[(rows-1)*stepB] + eps);
288 if( !conjB )
289 for( j = 1; j <= rows - 2; j += 2 )
290 {
291 double denom = dataB[j*stepB]*dataB[j*stepB] +
292 dataB[(j+1)*stepB]*dataB[(j+1)*stepB] + eps;
293
294 double re = dataA[j*stepA]*dataB[j*stepB] +
295 dataA[(j+1)*stepA]*dataB[(j+1)*stepB];
296
297 double im = dataA[(j+1)*stepA]*dataB[j*stepB] -
298 dataA[j*stepA]*dataB[(j+1)*stepB];
299
300 dataC[j*stepC] = re / denom;
301 dataC[(j+1)*stepC] = im / denom;
302 }
303 else
304 for( j = 1; j <= rows - 2; j += 2 )
305 {
306 double denom = dataB[j*stepB]*dataB[j*stepB] +
307 dataB[(j+1)*stepB]*dataB[(j+1)*stepB] + eps;
308
309 double re = dataA[j*stepA]*dataB[j*stepB] -
310 dataA[(j+1)*stepA]*dataB[(j+1)*stepB];
311
312 double im = dataA[(j+1)*stepA]*dataB[j*stepB] +
313 dataA[j*stepA]*dataB[(j+1)*stepB];
314
315 dataC[j*stepC] = re / denom;
316 dataC[(j+1)*stepC] = im / denom;
317 }
318 if( k == 1 )
319 dataA -= cols - 1, dataB -= cols - 1, dataC -= cols - 1;
320 }
321 }
322
323 for( ; rows--; dataA += stepA, dataB += stepB, dataC += stepC )
324 {
325 if( is_1d && cn == 1 )
326 {
327 dataC[0] = dataA[0] / (dataB[0] + eps);
328 if( cols % 2 == 0 )
329 dataC[j1] = dataA[j1] / (dataB[j1] + eps);
330 }
331
332 if( !conjB )
333 for( j = j0; j < j1; j += 2 )
334 {
335 double denom = dataB[j]*dataB[j] + dataB[j+1]*dataB[j+1] + eps;
336 double re = dataA[j]*dataB[j] + dataA[j+1]*dataB[j+1];
337 double im = dataA[j+1]*dataB[j] - dataA[j]*dataB[j+1];
338 dataC[j] = re / denom;
339 dataC[j+1] = im / denom;
340 }
341 else
342 for( j = j0; j < j1; j += 2 )
343 {
344 double denom = dataB[j]*dataB[j] + dataB[j+1]*dataB[j+1] + eps;
345 double re = dataA[j]*dataB[j] - dataA[j+1]*dataB[j+1];
346 double im = dataA[j+1]*dataB[j] + dataA[j]*dataB[j+1];
347 dataC[j] = re / denom;
348 dataC[j+1] = im / denom;
349 }
350 }
351 }

355static void fftShift(InputOutputArray _out)
356{
357 Mat out = _out.getMat();
358
359 if(out.rows == 1 && out.cols == 1)
360 {
361 // trivially shifted.
362 return;
363 }
364
365 vector<Mat> planes;
366 split(out, planes);
367
368 int xMid = out.cols >> 1;
369 int yMid = out.rows >> 1;
370
371 bool is_1d = xMid == 0 || yMid == 0;
372
373 if(is_1d)
374 {
375 xMid = xMid + yMid;
376
377 for(size_t i = 0; i < planes.size(); i++)
378 {
379 Mat tmp;
380 Mat half0(planes[i], Rect(0, 0, xMid, 1));
381 Mat half1(planes[i], Rect(xMid, 0, xMid, 1));
382
383 half0.copyTo(tmp);
384 half1.copyTo(half0);
385 tmp.copyTo(half1);
386 }
387 }
388 else
389 {
390 for(size_t i = 0; i < planes.size(); i++)
391 {
392 // perform quadrant swaps...
393 Mat tmp;
394 Mat q0(planes[i], Rect(0, 0, xMid, yMid));
395 Mat q1(planes[i], Rect(xMid, 0, xMid, yMid));
396 Mat q2(planes[i], Rect(0, yMid, xMid, yMid));
397 Mat q3(planes[i], Rect(xMid, yMid, xMid, yMid));
398
399 q0.copyTo(tmp);
400 q3.copyTo(q0);
401 tmp.copyTo(q3);
402
403 q1.copyTo(tmp);
404 q2.copyTo(q1);
405 tmp.copyTo(q2);
406 }
407 }
408
409 merge(planes, out);
410}

412static Point2d weightedCentroid(InputArray _src, cv::Point peakLocation, cv::Size weightBoxSize, double* response)
413{
414 Mat src = _src.getMat();
415
416 int type = src.type();
417 CV_Assert( type == CV_32FC1 || type == CV_64FC1 );
418
419 int minr = peakLocation.y - (weightBoxSize.height >> 1);
420 int maxr = peakLocation.y + (weightBoxSize.height >> 1);
421 int minc = peakLocation.x - (weightBoxSize.width >> 1);
422 int maxc = peakLocation.x + (weightBoxSize.width >> 1);
423
424 Point2d centroid;
425 double sumIntensity = 0.0;
426
427 // clamp the values to min and max if needed.
428 if(minr < 0)
429 {
430 minr = 0;
431 }
432
433 if(minc < 0)
434 {
435 minc = 0;
436 }
437
438 if(maxr > src.rows - 1)
439 {
440 maxr = src.rows - 1;
441 }
442
443 if(maxc > src.cols - 1)
444 {
445 maxc = src.cols - 1;
446 }
447
448 if(type == CV_32FC1)
449 {
450 const float* dataIn = (const float*)src.data;
451 dataIn += minr*src.cols;
452 for(int y = minr; y <= maxr; y++)
453 {
454 for(int x = minc; x <= maxc; x++)
455 {
456 centroid.x += (double)x*dataIn[x];
457 centroid.y += (double)y*dataIn[x];
458 sumIntensity += (double)dataIn[x];
459 }
460
461 dataIn += src.cols;
462 }
463 }
464 else
465 {
466 const double* dataIn = (const double*)src.data;
467 dataIn += minr*src.cols;
468 for(int y = minr; y <= maxr; y++)
469 {
470 for(int x = minc; x <= maxc; x++)
471 {
472 centroid.x += (double)x*dataIn[x];
473 centroid.y += (double)y*dataIn[x];
474 sumIntensity += dataIn[x];
475 }
476
477 dataIn += src.cols;
478 }
479 }
480
481 if(response)
482 *response = sumIntensity;
483
484 sumIntensity += DBL_EPSILON; // prevent div0 problems...
485
486 centroid.x /= sumIntensity;
487 centroid.y /= sumIntensity;
488
489 return centroid;
490}

494cv::Point2d cv::phaseCorrelateRes(InputArray _src1, InputArray _src2, InputArray _window, double* response)
495{
496 Mat src1 = _src1.getMat();
497 Mat src2 = _src2.getMat();
498 Mat window = _window.getMat();
499
500 CV_Assert( src1.type() == src2.type());
501 CV_Assert( src1.type() == CV_32FC1 || src1.type() == CV_64FC1 );
502 CV_Assert( src1.size == src2.size);
503
504 if(!window.empty())
505 {
506 CV_Assert( src1.type() == window.type());
507 CV_Assert( src1.size == window.size);
508 }
509
510 int M = getOptimalDFTSize(src1.rows);
511 int N = getOptimalDFTSize(src1.cols);
512
513 Mat padded1, padded2, paddedWin;
514
515 if(M != src1.rows || N != src1.cols)
516 {
517 copyMakeBorder(src1, padded1, 0, M - src1.rows, 0, N - src1.cols, BORDER_CONSTANT, Scalar::all(0));
518 copyMakeBorder(src2, padded2, 0, M - src2.rows, 0, N - src2.cols, BORDER_CONSTANT, Scalar::all(0));
519
520 if(!window.empty())
521 {
522 copyMakeBorder(window, paddedWin, 0, M - window.rows, 0, N - window.cols, BORDER_CONSTANT, Scalar::all(0));
523 }
524 }
525 else
526 {
527 padded1 = src1;
528 padded2 = src2;
529 paddedWin = window;
530 }
531
532 Mat FFT1, FFT2, P, Pm, C;
533
534 // perform window multiplication if available
535 if(!paddedWin.empty())
536 {
537 // apply window to both images before proceeding...
538 multiply(paddedWin, padded1, padded1);
539 multiply(paddedWin, padded2, padded2);
540 }
541
542 // execute phase correlation equation
543 // Reference: http://en.wikipedia.org/wiki/Phase_correlation
544 dft(padded1, FFT1, DFT_REAL_OUTPUT);
545 dft(padded2, FFT2, DFT_REAL_OUTPUT);
546
547 mulSpectrums(FFT1, FFT2, P, 0, true);
548
549 magSpectrums(P, Pm);
550 divSpectrums(P, Pm, C, 0, false); // FF* / |FF*| (phase correlation equation completed here...)
551
552 idft(C, C); // gives us the nice peak shift location...
553
554 fftShift(C); // shift the energy to the center of the frame.
555
556 // locate the highest peak
557 Point peakLoc;
558 minMaxLoc(C, NULL, NULL, NULL, &peakLoc);
559
560 // get the phase shift with sub-pixel accuracy, 5x5 window seems about right here...
561 Point2d t;
562 t = weightedCentroid(C, peakLoc, Size(5, 5), response);
563
564 // max response is M*N (not exactly, might be slightly larger due to rounding errors)
565 if(response)
566 *response /= M*N;
567
568 // adjust shift relative to image center...
569 Point2d center((double)padded1.cols / 2.0, (double)padded1.rows / 2.0);
570
571 return (center - t);
572}

574cv::Point2d cv::phaseCorrelate(InputArray _src1, InputArray _src2, InputArray _window)
575{
576 return phaseCorrelateRes(_src1, _src2, _window, 0);
577}

579void cv::createHanningWindow(OutputArray _dst, cv::Size winSize, int type)
580{
581 CV_Assert( type == CV_32FC1 || type == CV_64FC1 );
582
583 _dst.create(winSize, type);
584 Mat dst = _dst.getMat();
585
586 int rows = dst.rows;
587 int cols = dst.cols;
588
589 if(dst.depth() == CV_32F)
590 {
591 for(int i = 0; i < rows; i++)
592 {
593 float* dstData = dst.ptr<float>(i);
594 double wr = 0.5 * (1.0f - cos(2.0f * CV_PI * (double)i / (double)(rows - 1)));
595 for(int j = 0; j < cols; j++)
596 {
597 double wc = 0.5 * (1.0f - cos(2.0f * CV_PI * (double)j / (double)(cols - 1)));
598 dstData[j] = (float)(wr * wc);
599 }
600 }
601 }
602 else
603 {
604 for(int i = 0; i < rows; i++)
605 {
606 double* dstData = dst.ptr<double>(i);
607 double wr = 0.5 * (1.0 - cos(2.0 * CV_PI * (double)i / (double)(rows - 1)));
608 for(int j = 0; j < cols; j++)
609 {
610 double wc = 0.5 * (1.0 - cos(2.0 * CV_PI * (double)j / (double)(cols - 1)));
611 dstData[j] = wr * wc;
612 }
613 }
614 }
615
616 // perform batch sqrt for SSE performance gains
617 cv::sqrt(dst, dst);
618}
