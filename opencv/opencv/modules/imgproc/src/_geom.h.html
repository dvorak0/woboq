45/* Finds distance between two points */
46CV_INLINE float icvDistanceL2_32f( CvPoint2D32f pt1, CvPoint2D32f pt2 )
47{
48 float dx = pt2.x - pt1.x;
49 float dy = pt2.y - pt1.y;
50
51 return std::sqrt( dx*dx + dy*dy );
52}
