50static void
51calcMinEigenVal( const Mat& _cov, Mat& _dst )
52{
53 int i, j;
54 Size size = _cov.size();
55#if CV_SSE
56 volatile bool simd = checkHardwareSupport(CV_CPU_SSE);
57#endif
58
59 if( _cov.isContinuous() && _dst.isContinuous() )
60 {
61 size.width *= size.height;
62 size.height = 1;
63 }
64
65 for( i = 0; i < size.height; i++ )
66 {
67 const float* cov = (const float*)(_cov.data + _cov.step*i);
68 float* dst = (float*)(_dst.data + _dst.step*i);
69 j = 0;
70 #if CV_SSE
71 if( simd )
72 {
73 __m128 half = _mm_set1_ps(0.5f);
74 for( ; j <= size.width - 5; j += 4 )
75 {
76 __m128 t0 = _mm_loadu_ps(cov + j*3); // a0 b0 c0 x
77 __m128 t1 = _mm_loadu_ps(cov + j*3 + 3); // a1 b1 c1 x
78 __m128 t2 = _mm_loadu_ps(cov + j*3 + 6); // a2 b2 c2 x
79 __m128 t3 = _mm_loadu_ps(cov + j*3 + 9); // a3 b3 c3 x
80 __m128 a, b, c, t;
81 t = _mm_unpacklo_ps(t0, t1); // a0 a1 b0 b1
82 c = _mm_unpackhi_ps(t0, t1); // c0 c1 x x
83 b = _mm_unpacklo_ps(t2, t3); // a2 a3 b2 b3
84 c = _mm_movelh_ps(c, _mm_unpackhi_ps(t2, t3)); // c0 c1 c2 c3
85 a = _mm_movelh_ps(t, b);
86 b = _mm_movehl_ps(b, t);
87 a = _mm_mul_ps(a, half);
88 c = _mm_mul_ps(c, half);
89 t = _mm_sub_ps(a, c);
90 t = _mm_add_ps(_mm_mul_ps(t, t), _mm_mul_ps(b,b));
91 a = _mm_sub_ps(_mm_add_ps(a, c), _mm_sqrt_ps(t));
92 _mm_storeu_ps(dst + j, a);
93 }
94 }
95 #endif
96 for( ; j < size.width; j++ )
97 {
98 float a = cov[j*3]*0.5f;
99 float b = cov[j*3+1];
100 float c = cov[j*3+2]*0.5f;
101 dst[j] = (float)((a + c) - std::sqrt((a - c)*(a - c) + b*b));
102 }
103 }
104}

107static void
108calcHarris( const Mat& _cov, Mat& _dst, double k )
109{
110 int i, j;
111 Size size = _cov.size();
112#if CV_SSE
113 volatile bool simd = checkHardwareSupport(CV_CPU_SSE);
114#endif
115
116 if( _cov.isContinuous() && _dst.isContinuous() )
117 {
118 size.width *= size.height;
119 size.height = 1;
120 }
121
122 for( i = 0; i < size.height; i++ )
123 {
124 const float* cov = (const float*)(_cov.data + _cov.step*i);
125 float* dst = (float*)(_dst.data + _dst.step*i);
126 j = 0;
127
128 #if CV_SSE
129 if( simd )
130 {
131 __m128 k4 = _mm_set1_ps((float)k);
132 for( ; j <= size.width - 5; j += 4 )
133 {
134 __m128 t0 = _mm_loadu_ps(cov + j*3); // a0 b0 c0 x
135 __m128 t1 = _mm_loadu_ps(cov + j*3 + 3); // a1 b1 c1 x
136 __m128 t2 = _mm_loadu_ps(cov + j*3 + 6); // a2 b2 c2 x
137 __m128 t3 = _mm_loadu_ps(cov + j*3 + 9); // a3 b3 c3 x
138 __m128 a, b, c, t;
139 t = _mm_unpacklo_ps(t0, t1); // a0 a1 b0 b1
140 c = _mm_unpackhi_ps(t0, t1); // c0 c1 x x
141 b = _mm_unpacklo_ps(t2, t3); // a2 a3 b2 b3
142 c = _mm_movelh_ps(c, _mm_unpackhi_ps(t2, t3)); // c0 c1 c2 c3
143 a = _mm_movelh_ps(t, b);
144 b = _mm_movehl_ps(b, t);
145 t = _mm_add_ps(a, c);
146 a = _mm_sub_ps(_mm_mul_ps(a, c), _mm_mul_ps(b, b));
147 t = _mm_mul_ps(_mm_mul_ps(k4, t), t);
148 a = _mm_sub_ps(a, t);
149 _mm_storeu_ps(dst + j, a);
150 }
151 }
152 #endif
153
154 for( ; j < size.width; j++ )
155 {
156 float a = cov[j*3];
157 float b = cov[j*3+1];
158 float c = cov[j*3+2];
159 dst[j] = (float)(a*c - b*b - k*(a + c)*(a + c));
160 }
161 }
162}

165void eigen2x2( const float* cov, float* dst, int n )
166{
167 for( int j = 0; j < n; j++ )
168 {
169 double a = cov[j*3];
170 double b = cov[j*3+1];
171 double c = cov[j*3+2];
172
173 double u = (a + c)*0.5;
174 double v = std::sqrt((a - c)*(a - c)*0.25 + b*b);
175 double l1 = u + v;
176 double l2 = u - v;
177
178 double x = b;
179 double y = l1 - a;
180 double e = fabs(x);
181
182 if( e + fabs(y) < 1e-4 )
183 {
184 y = b;
185 x = l1 - c;
186 e = fabs(x);
187 if( e + fabs(y) < 1e-4 )
188 {
189 e = 1./(e + fabs(y) + FLT_EPSILON);
190 x *= e, y *= e;
191 }
192 }
193
194 double d = 1./std::sqrt(x*x + y*y + DBL_EPSILON);
195 dst[6*j] = (float)l1;
196 dst[6*j + 2] = (float)(x*d);
197 dst[6*j + 3] = (float)(y*d);
198
199 x = b;
200 y = l2 - a;
201 e = fabs(x);
202
203 if( e + fabs(y) < 1e-4 )
204 {
205 y = b;
206 x = l2 - c;
207 e = fabs(x);
208 if( e + fabs(y) < 1e-4 )
209 {
210 e = 1./(e + fabs(y) + FLT_EPSILON);
211 x *= e, y *= e;
212 }
213 }
214
215 d = 1./std::sqrt(x*x + y*y + DBL_EPSILON);
216 dst[6*j + 1] = (float)l2;
217 dst[6*j + 4] = (float)(x*d);
218 dst[6*j + 5] = (float)(y*d);
219 }

222static void
223calcEigenValsVecs( const Mat& _cov, Mat& _dst )
224{
225 Size size = _cov.size();
226 if( _cov.isContinuous() && _dst.isContinuous() )
227 {
228 size.width *= size.height;
229 size.height = 1;
230 }
231
232 for( int i = 0; i < size.height; i++ )
233 {
234 const float* cov = (const float*)(_cov.data + _cov.step*i);
235 float* dst = (float*)(_dst.data + _dst.step*i);
236
237 eigen2x2(cov, dst, size.width);
238 }
239}

245static void
246cornerEigenValsVecs( const Mat& src, Mat& eigenv, int block_size,
247 int aperture_size, int op_type, double k=0.,
248 int borderType=BORDER_DEFAULT )
249{
250#ifdef HAVE_TEGRA_OPTIMIZATION
251 if (tegra::cornerEigenValsVecs(src, eigenv, block_size, aperture_size, op_type, k, borderType))
252 return;
253#endif
254
255 int depth = src.depth();
256 double scale = (double)(1 << ((aperture_size > 0 ? aperture_size : 3) - 1)) * block_size;
257 if( aperture_size < 0 )
258 scale *= 2.;
259 if( depth == CV_8U )
260 scale *= 255.;
261 scale = 1./scale;
262
263 CV_Assert( src.type() == CV_8UC1 || src.type() == CV_32FC1 );
264
265 Mat Dx, Dy;
266 if( aperture_size > 0 )
267 {
268 Sobel( src, Dx, CV_32F, 1, 0, aperture_size, scale, 0, borderType );
269 Sobel( src, Dy, CV_32F, 0, 1, aperture_size, scale, 0, borderType );
270 }
271 else
272 {
273 Scharr( src, Dx, CV_32F, 1, 0, scale, 0, borderType );
274 Scharr( src, Dy, CV_32F, 0, 1, scale, 0, borderType );
275 }
276
277 Size size = src.size();
278 Mat cov( size, CV_32FC3 );
279 int i, j;
280
281 for( i = 0; i < size.height; i++ )
282 {
283 float* cov_data = (float*)(cov.data + i*cov.step);
284 const float* dxdata = (const float*)(Dx.data + i*Dx.step);
285 const float* dydata = (const float*)(Dy.data + i*Dy.step);
286
287 for( j = 0; j < size.width; j++ )
288 {
289 float dx = dxdata[j];
290 float dy = dydata[j];
291
292 cov_data[j*3] = dx*dx;
293 cov_data[j*3+1] = dx*dy;
294 cov_data[j*3+2] = dy*dy;
295 }
296 }
297
298 boxFilter(cov, cov, cov.depth(), Size(block_size, block_size),
299 Point(-1,-1), false, borderType );
300
301 if( op_type == MINEIGENVAL )
302 calcMinEigenVal( cov, eigenv );
303 else if( op_type == HARRIS )
304 calcHarris( cov, eigenv, k );
305 else if( op_type == EIGENVALSVECS )
306 calcEigenValsVecs( cov, eigenv );
307}

311void cv::cornerMinEigenVal( InputArray _src, OutputArray _dst, int blockSize, int ksize, int borderType )
312{
313 Mat src = _src.getMat();
314 _dst.create( src.size(), CV_32F );
315 Mat dst = _dst.getMat();
316 cornerEigenValsVecs( src, dst, blockSize, ksize, MINEIGENVAL, 0, borderType );
317}

329void cv::cornerEigenValsAndVecs( InputArray _src, OutputArray _dst, int blockSize, int ksize, int borderType )
330{
331 Mat src = _src.getMat();
332 Size dsz = _dst.size();
333 int dtype = _dst.type();
334
335 if( dsz.height != src.rows || dsz.width*CV_MAT_CN(dtype) != src.cols*6 || CV_MAT_DEPTH(dtype) != CV_32F )
336 _dst.create( src.size(), CV_32FC(6) );
337 Mat dst = _dst.getMat();
338 cornerEigenValsVecs( src, dst, blockSize, ksize, EIGENVALSVECS, 0, borderType );
339}

342void cv::preCornerDetect( InputArray _src, OutputArray _dst, int ksize, int borderType )
343{
344 Mat Dx, Dy, D2x, D2y, Dxy, src = _src.getMat();
345
346 CV_Assert( src.type() == CV_8UC1 || src.type() == CV_32FC1 );
347 _dst.create( src.size(), CV_32F );
348 Mat dst = _dst.getMat();
349
350 Sobel( src, Dx, CV_32F, 1, 0, ksize, 1, 0, borderType );
351 Sobel( src, Dy, CV_32F, 0, 1, ksize, 1, 0, borderType );
352 Sobel( src, D2x, CV_32F, 2, 0, ksize, 1, 0, borderType );
353 Sobel( src, D2y, CV_32F, 0, 2, ksize, 1, 0, borderType );
354 Sobel( src, Dxy, CV_32F, 1, 1, ksize, 1, 0, borderType );
355
356 double factor = 1 << (ksize - 1);
357 if( src.depth() == CV_8U )
358 factor *= 255;
359 factor = 1./(factor * factor * factor);
360
361 Size size = src.size();
362 int i, j;
363 for( i = 0; i < size.height; i++ )
364 {
365 float* dstdata = (float*)(dst.data + i*dst.step);
366 const float* dxdata = (const float*)(Dx.data + i*Dx.step);
367 const float* dydata = (const float*)(Dy.data + i*Dy.step);
368 const float* d2xdata = (const float*)(D2x.data + i*D2x.step);
369 const float* d2ydata = (const float*)(D2y.data + i*D2y.step);
370 const float* dxydata = (const float*)(Dxy.data + i*Dxy.step);
371
372 for( j = 0; j < size.width; j++ )
373 {
374 float dx = dxdata[j];
375 float dy = dydata[j];
376 dstdata[j] = (float)(factor*(dx*dx*d2ydata[j] + dy*dy*d2xdata[j] - 2*dx*dy*dxydata[j]));
377 }
378 }
379}
