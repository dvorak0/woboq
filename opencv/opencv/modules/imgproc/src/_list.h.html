84#define DECLARE_LIST(type, prefix)\
85 /* Basic element of a list*/\
86 struct prefix##element_##type\
87 {\
88 struct prefix##element_##type* m_prev;\
89 struct prefix##element_##type* m_next;\
90 type m_data;\
91 };\
92 typedef struct prefix##element_##type ELEMENT_##type;\
93 /* Initialization and destruction*/\
94 _LIST_INLINE _CVLIST* prefix##create_list_##type(long);\
95 _LIST_INLINE void prefix##destroy_list_##type(_CVLIST*);\
96 /* Access functions*/\
97 _LIST_INLINE CVPOS prefix##get_head_pos_##type(_CVLIST*);\
98 _LIST_INLINE CVPOS prefix##get_tail_pos_##type(_CVLIST*);\
99 _LIST_INLINE type* prefix##get_next_##type(CVPOS*);\
100 _LIST_INLINE type* prefix##get_prev_##type(CVPOS*);\
101 _LIST_INLINE int prefix##is_pos_##type(CVPOS pos);\
102 /* Modification functions*/\
103 _LIST_INLINE void prefix##clear_list_##type(_CVLIST*);\
104 _LIST_INLINE CVPOS prefix##add_head_##type(_CVLIST*, type*);\
105 _LIST_INLINE CVPOS prefix##add_tail_##type(_CVLIST*, type*);\
106 _LIST_INLINE void prefix##remove_head_##type(_CVLIST*);\
107 _LIST_INLINE void prefix##remove_tail_##type(_CVLIST*);\
108 _LIST_INLINE CVPOS prefix##insert_before_##type(_CVLIST*, CVPOS, type*);\
109 _LIST_INLINE CVPOS prefix##insert_after_##type(_CVLIST*, CVPOS, type*);\
110 _LIST_INLINE void prefix##remove_at_##type(_CVLIST*, CVPOS);\
111 _LIST_INLINE void prefix##set_##type(CVPOS, type*);\
112 _LIST_INLINE type* prefix##get_##type(CVPOS);\
113 /* Statistics functions*/\
114 _LIST_INLINE int prefix##get_count_##type(_CVLIST*);

116/* This macro finds a space for a new element and puts in into 'element' pointer */
117#define INSERT_NEW(element_type, l, element)\
118 l->m_size++;\
119 if(l->m_head_free.m_pos != NULL)\
120 {\
121 element = (element_type*)(l->m_head_free.m_pos);\
122 if(element->m_next != NULL)\
123 {\
124 element->m_next->m_prev = NULL;\
125 l->m_head_free.m_pos = element->m_next;\
126 }\
127 else\
128 {\
129 l->m_head_free.m_pos = NULL;\
130 }\
131 }\
132 else\
133 {\
134 if(l->m_buf_size < l->m_size && l->m_head_free.m_pos == NULL)\
135 {\
136 *(void**)l->m_buffer = cvAlloc(l->m_buf_size*sizeof(element_type) + sizeof(void*));\
137 l->m_buffer = *(void**)l->m_buffer;\
138 *(void**)l->m_buffer = NULL;\
139 element = (element_type*)((char*)l->m_buffer + sizeof(void*));\
140 }\
141 else\
142 {\
143 element = (element_type*)((char*)l->m_buffer + sizeof(void*)) + l->m_size - 1;\
144 }\
145 }
146
147/* This macro adds 'element' to the list of free elements*/
148#define INSERT_FREE(element_type, l, element)\
149 if(l->m_head_free.m_pos != NULL)\
150 {\
151 ((element_type*)l->m_head_free.m_pos)->m_prev = element;\
152 }\
153 element->m_next = ((element_type*)l->m_head_free.m_pos);\
154 l->m_head_free.m_pos = element;

159#define IMPLEMENT_LIST(type, prefix)\
160_CVLIST* prefix##create_list_##type(long size)\
161{\
162 _CVLIST* pl = (_CVLIST*)cvAlloc(sizeof(_CVLIST));\
163 pl->m_buf_size = size > 0 ? size : default_size;\
164 pl->m_first_buffer = cvAlloc(pl->m_buf_size*sizeof(ELEMENT_##type) + sizeof(void*));\
165 pl->m_buffer = pl->m_first_buffer;\
166 *(void**)pl->m_buffer = NULL;\
167 pl->m_size = 0;\
168 pl->m_head.m_pos = NULL;\
169 pl->m_tail.m_pos = NULL;\
170 pl->m_head_free.m_pos = NULL;\
171 return pl;\
172}\
173void prefix##destroy_list_##type(_CVLIST* l)\
174{\
175 void* cur = l->m_first_buffer;\
176 void* next;\
177 while(cur)\
178 {\
179 next = *(void**)cur;\
180 cvFree(&cur);\
181 cur = next;\
182 }\
183 cvFree(&l);\
184}\
185CVPOS prefix##get_head_pos_##type(_CVLIST* l)\
186{\
187 return l->m_head;\
188}\
189CVPOS prefix##get_tail_pos_##type(_CVLIST* l)\
190{\
191 return l->m_tail;\
192}\
193type* prefix##get_next_##type(CVPOS* pos)\
194{\
195 if(pos->m_pos)\
196 {\
197 ELEMENT_##type* element = (ELEMENT_##type*)(pos->m_pos);\
198 pos->m_pos = element->m_next;\
199 return &element->m_data;\
200 }\
201 else\
202 {\
203 return NULL;\
204 }\
205}\
206type* prefix##get_prev_##type(CVPOS* pos)\
207{\
208 if(pos->m_pos)\
209 {\
210 ELEMENT_##type* element = (ELEMENT_##type*)(pos->m_pos);\
211 pos->m_pos = element->m_prev;\
212 return &element->m_data;\
213 }\
214 else\
215 {\
216 return NULL;\
217 }\
218}\
219int prefix##is_pos_##type(CVPOS pos)\
220{\
221 return !!pos.m_pos;\
222}\
223void prefix##clear_list_##type(_CVLIST* l)\
224{\
225 l->m_head.m_pos = NULL;\
226 l->m_tail.m_pos = NULL;\
227 l->m_size = 0;\
228 l->m_head_free.m_pos = NULL;\
229}\
230CVPOS prefix##add_head_##type(_CVLIST* l, type* data)\
231{\
232 ELEMENT_##type* element;\
233 INSERT_NEW(ELEMENT_##type, l, element);\
234 element->m_prev = NULL;\
235 element->m_next = (ELEMENT_##type*)(l->m_head.m_pos);\
236 memcpy(&(element->m_data), data, sizeof(*data));\
237 if(element->m_next)\
238 {\
239 element->m_next->m_prev = element;\
240 }\
241 else\
242 {\
243 l->m_tail.m_pos = element;\
244 }\
245 l->m_head.m_pos = element;\
246 return l->m_head;\
247}\
248CVPOS prefix##add_tail_##type(_CVLIST* l, type* data)\
249{\
250 ELEMENT_##type* element;\
251 INSERT_NEW(ELEMENT_##type, l, element);\
252 element->m_next = NULL;\
253 element->m_prev = (ELEMENT_##type*)(l->m_tail.m_pos);\
254 memcpy(&(element->m_data), data, sizeof(*data));\
255 if(element->m_prev)\
256 {\
257 element->m_prev->m_next = element;\
258 }\
259 else\
260 {\
261 l->m_head.m_pos = element;\
262 }\
263 l->m_tail.m_pos = element;\
264 return l->m_tail;\
265}\
266void prefix##remove_head_##type(_CVLIST* l)\
267{\
268 ELEMENT_##type* element = ((ELEMENT_##type*)(l->m_head.m_pos));\
269 if(element->m_next != NULL)\
270 {\
271 element->m_next->m_prev = NULL;\
272 }\
273 l->m_head.m_pos = element->m_next;\
274 INSERT_FREE(ELEMENT_##type, l, element);\
275 l->m_size--;\
276}\
277void prefix##remove_tail_##type(_CVLIST* l)\
278{\
279 ELEMENT_##type* element = ((ELEMENT_##type*)(l->m_tail.m_pos));\
280 if(element->m_prev != NULL)\
281 {\
282 element->m_prev->m_next = NULL;\
283 }\
284 l->m_tail.m_pos = element->m_prev;\
285 INSERT_FREE(ELEMENT_##type, l, element);\
286 l->m_size--;\
287}\
288CVPOS prefix##insert_after_##type(_CVLIST* l, CVPOS pos, type* data)\
289{\
290 ELEMENT_##type* element;\
291 ELEMENT_##type* before;\
292 CVPOS newpos;\
293 INSERT_NEW(ELEMENT_##type, l, element);\
294 memcpy(&(element->m_data), data, sizeof(*data));\
295 before = (ELEMENT_##type*)pos.m_pos;\
296 element->m_prev = before;\
297 element->m_next = before->m_next;\
298 before->m_next = element;\
299 if(element->m_next != NULL)\
300 element->m_next->m_prev = element;\
301 else\
302 l->m_tail.m_pos = element;\
303 newpos.m_pos = element;\
304 return newpos;\
305}\
306CVPOS prefix##insert_before_##type(_CVLIST* l, CVPOS pos, type* data)\
307{\
308 ELEMENT_##type* element;\
309 ELEMENT_##type* after;\
310 CVPOS newpos;\
311 INSERT_NEW(ELEMENT_##type, l, element);\
312 memcpy(&(element->m_data), data, sizeof(*data));\
313 after = (ELEMENT_##type*)pos.m_pos;\
314 element->m_prev = after->m_prev;\
315 element->m_next = after;\
316 after->m_prev = element;\
317 if(element->m_prev != NULL)\
318 element->m_prev->m_next = element;\
319 else\
320 l->m_head.m_pos = element;\
321 newpos.m_pos = element;\
322 return newpos;\
323}\
324void prefix##remove_at_##type(_CVLIST* l, CVPOS pos)\
325{\
326 ELEMENT_##type* element = ((ELEMENT_##type*)pos.m_pos);\
327 if(element->m_prev != NULL)\
328 {\
329 element->m_prev->m_next = element->m_next;\
330 }\
331 else\
332 {\
333 l->m_head.m_pos = element->m_next;\
334 }\
335 if(element->m_next != NULL)\
336 {\
337 element->m_next->m_prev = element->m_prev;\
338 }\
339 else\
340 {\
341 l->m_tail.m_pos = element->m_prev;\
342 }\
343 INSERT_FREE(ELEMENT_##type, l, element);\
344 l->m_size--;\
345}\
346void prefix##set_##type(CVPOS pos, type* data)\
347{\
348 ELEMENT_##type* element = ((ELEMENT_##type*)(pos.m_pos));\
349 memcpy(&(element->m_data), data, sizeof(*data));\
350}\
351type* prefix##get_##type(CVPOS pos)\
352{\
353 ELEMENT_##type* element = ((ELEMENT_##type*)(pos.m_pos));\
354 return &(element->m_data);\
355}\
356int prefix##get_count_##type(_CVLIST* list)\
357{\
358 return (int)list->m_size;\
359}
