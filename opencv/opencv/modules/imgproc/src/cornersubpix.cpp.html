43CV_IMPL void
44cvFindCornerSubPix( const void* srcarr, CvPoint2D32f* corners,
45 int count, CvSize win, CvSize zeroZone,
46 CvTermCriteria criteria )
47{
48 cv::AutoBuffer<float> buffer;
49
50 const int MAX_ITERS = 100;
51 const float drv[] = { -1.f, 0.f, 1.f };
52 float *maskX;
53 float *maskY;
54 float *mask;
55 float *src_buffer;
56 float *gx_buffer;
57 float *gy_buffer;
58 int win_w = win.width * 2 + 1, win_h = win.height * 2 + 1;
59 int win_rect_size = (win_w + 4) * (win_h + 4);
60 double coeff;
61 CvSize size, src_buf_size;
62 int i, j, k, pt_i;
63 int max_iters = 10;
64 double eps = 0;
65
66 CvMat stub, *src = (CvMat*)srcarr;
67 src = cvGetMat( srcarr, &stub );
68
69 if( CV_MAT_TYPE( src->type ) != CV_8UC1 )
70 CV_Error( CV_StsUnsupportedFormat, "The source image must be 8-bit single-channel (CV_8UC1)" );
71
72 if( !corners )
73 CV_Error( CV_StsNullPtr, "" );
74
75 if( count < 0 )
76 CV_Error( CV_StsBadSize, "" );
77
78 if( count == 0 )
79 return;
80
81 if( win.width <= 0 || win.height <= 0 )
82 CV_Error( CV_StsBadSize, "" );
83
84 size = cvGetMatSize( src );
85
86 if( size.width < win_w + 4 || size.height < win_h + 4 )
87 CV_Error( CV_StsBadSize, "" );
88
89 /* initialize variables, controlling loop termination */
90 switch( criteria.type )
91 {
92 case CV_TERMCRIT_ITER:
93 eps = 0.f;
94 max_iters = criteria.max_iter;
95 break;
96 case CV_TERMCRIT_EPS:
97 eps = criteria.epsilon;
98 max_iters = MAX_ITERS;
99 break;
100 case CV_TERMCRIT_ITER | CV_TERMCRIT_EPS:
101 eps = criteria.epsilon;
102 max_iters = criteria.max_iter;
103 break;
104 default:
105 assert( 0 );
106 CV_Error( CV_StsBadFlag, "" );
107 }
108
109 eps = MAX( eps, 0 );
110 eps *= eps; /* use square of error in comparsion operations. */
111
112 max_iters = MAX( max_iters, 1 );
113 max_iters = MIN( max_iters, MAX_ITERS );
114
115 buffer.allocate( win_rect_size * 5 + win_w + win_h + 32 );
116
117 /* assign pointers */
118 maskX = buffer;
119 maskY = maskX + win_w + 4;
120 mask = maskY + win_h + 4;
121 src_buffer = mask + win_w * win_h;
122 gx_buffer = src_buffer + win_rect_size;
123 gy_buffer = gx_buffer + win_rect_size;
124
125 coeff = 1. / (win.width * win.width);
126
127 /* calculate mask */
128 for( i = -win.width, k = 0; i <= win.width; i++, k++ )
129 {
130 maskX[k] = (float)exp( -i * i * coeff );
131 }
132
133 if( win.width == win.height )
134 {
135 maskY = maskX;
136 }
137 else
138 {
139 coeff = 1. / (win.height * win.height);
140 for( i = -win.height, k = 0; i <= win.height; i++, k++ )
141 {
142 maskY[k] = (float) exp( -i * i * coeff );
143 }
144 }
145
146 for( i = 0; i < win_h; i++ )
147 {
148 for( j = 0; j < win_w; j++ )
149 {
150 mask[i * win_w + j] = maskX[j] * maskY[i];
151 }
152 }
153
154
155 /* make zero_zone */
156 if( zeroZone.width >= 0 && zeroZone.height >= 0 &&
157 zeroZone.width * 2 + 1 < win_w && zeroZone.height * 2 + 1 < win_h )
158 {
159 for( i = win.height - zeroZone.height; i <= win.height + zeroZone.height; i++ )
160 {
161 for( j = win.width - zeroZone.width; j <= win.width + zeroZone.width; j++ )
162 {
163 mask[i * win_w + j] = 0;
164 }
165 }
166 }
167
168 /* set sizes of image rectangles, used in convolutions */
169 src_buf_size.width = win_w + 2;
170 src_buf_size.height = win_h + 2;
171
172 /* do optimization loop for all the points */
173 for( pt_i = 0; pt_i < count; pt_i++ )
174 {
175 CvPoint2D32f cT = corners[pt_i], cI = cT;
176 int iter = 0;
177 double err;
178
179 do
180 {
181 CvPoint2D32f cI2;
182 double a, b, c, bb1, bb2;
183
184 IPPI_CALL( icvGetRectSubPix_8u32f_C1R( (uchar*)src->data.ptr, src->step, size,
185 src_buffer, (win_w + 2) * sizeof( src_buffer[0] ),
186 cvSize( win_w + 2, win_h + 2 ), cI ));
187
188 /* calc derivatives */
189 icvSepConvSmall3_32f( src_buffer+src_buf_size.width, src_buf_size.width * sizeof(src_buffer[0]),
190 gx_buffer, win_w * sizeof(gx_buffer[0]),
191 src_buf_size, drv, NULL, NULL );
192 icvSepConvSmall3_32f( src_buffer+1, src_buf_size.width * sizeof(src_buffer[0]),
193 gy_buffer, win_w * sizeof(gy_buffer[0]),
194 src_buf_size, NULL, drv, NULL );
195
196 a = b = c = bb1 = bb2 = 0;
197
198 /* process gradient */
199 for( i = 0, k = 0; i < win_h; i++ )
200 {
201 double py = i - win.height;
202
203 for( j = 0; j < win_w; j++, k++ )
204 {
205 double m = mask[k];
206 double tgx = gx_buffer[k];
207 double tgy = gy_buffer[k];
208 double gxx = tgx * tgx * m;
209 double gxy = tgx * tgy * m;
210 double gyy = tgy * tgy * m;
211 double px = j - win.width;
212
213 a += gxx;
214 b += gxy;
215 c += gyy;
216
217 bb1 += gxx * px + gxy * py;
218 bb2 += gxy * px + gyy * py;
219 }
220 }
221
222 double det=a*c-b*b;
223 if( fabs( det ) > DBL_EPSILON*DBL_EPSILON )
224 {
225 // 2x2 matrix inversion
226 double scale=1.0/det;
227 cI2.x = (float)(cI.x + c*scale*bb1 - b*scale*bb2);
228 cI2.y = (float)(cI.y - b*scale*bb1 + a*scale*bb2);
229 }
230 else
231 {
232 cI2 = cI;
233 }
234
235 err = (cI2.x - cI.x) * (cI2.x - cI.x) + (cI2.y - cI.y) * (cI2.y - cI.y);
236 cI = cI2;
237
238 /* if new point is too far from initial, it means poor convergence. */
239 if (fabs(cI.x - cT.x) > win.width || fabs(cI.y - cT.y) > win.height)
240 {
241 cI = cT;
242 break;
243 }
244 cI.x = std::max(0.0f, std::min((float)size.width, cI.x));
245 cI.y = std::max(0.0f, std::min((float)size.height, cI.y));
246 }
247 while( ++iter < max_iters && err > eps );
248
249 /* if new point is too far from initial, it means poor convergence.
250 leave initial point as the result */
251 if( fabs( cI.x - cT.x ) > win.width || fabs( cI.y - cT.y ) > win.height )
252 {
253 cI = cT;
254 }
255
256 corners[pt_i] = cI; /* store result */
257 }
258}
259
260void cv::cornerSubPix( InputArray _image, InputOutputArray _corners,
261 Size winSize, Size zeroZone,
262 TermCriteria criteria )
263{
264 Mat corners = _corners.getMat();
265 int ncorners = corners.checkVector(2);
266 CV_Assert( ncorners >= 0 && corners.depth() == CV_32F );
267 Mat image = _image.getMat();
268 CvMat c_image = image;
269
270 cvFindCornerSubPix( &c_image, (CvPoint2D32f*)corners.data, ncorners,
271 winSize, zeroZone, criteria );
272}
