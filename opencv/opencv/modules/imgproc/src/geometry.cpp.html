44CV_IMPL CvRect
45cvMaxRect( const CvRect* rect1, const CvRect* rect2 )
46{
47 if( rect1 && rect2 )
48 {
49 CvRect max_rect;
50 int a, b;
51
52 max_rect.x = a = rect1->x;
53 b = rect2->x;
54 if( max_rect.x > b )
55 max_rect.x = b;
56
57 max_rect.width = a += rect1->width;
58 b += rect2->width;
59
60 if( max_rect.width < b )
61 max_rect.width = b;
62 max_rect.width -= max_rect.x;
63
64 max_rect.y = a = rect1->y;
65 b = rect2->y;
66 if( max_rect.y > b )
67 max_rect.y = b;
68
69 max_rect.height = a += rect1->height;
70 b += rect2->height;
71
72 if( max_rect.height < b )
73 max_rect.height = b;
74 max_rect.height -= max_rect.y;
75 return max_rect;
76 }
77 else if( rect1 )
78 return *rect1;
79 else if( rect2 )
80 return *rect2;
81 else
82 return cvRect(0,0,0,0);
83}

95int
96icvIntersectLines( double x1, double dx1, double y1, double dy1,
97 double x2, double dx2, double y2, double dy2, double *t2 )
98{
99 double d = dx1 * dy2 - dx2 * dy1;
100 int result = -1;
101
102 if( d != 0 )
103 {
104 *t2 = ((x2 - x1) * dy1 - (y2 - y1) * dx1) / d;
105 result = 0;
106 }
107 return result;
108}

111void
112icvIntersectLines3( double *a0, double *b0, double *c0,
113 double *a1, double *b1, double *c1, CvPoint2D32f * point )
114{
115 double det = a0[0] * b1[0] - a1[0] * b0[0];
116
117 if( det != 0 )
118 {
119 det = 1. / det;
120 point->x = (float) ((b0[0] * c1[0] - b1[0] * c0[0]) * det);
121 point->y = (float) ((a1[0] * c0[0] - a0[0] * c1[0]) * det);
122 }
123 else
124 {
125 point->x = point->y = FLT_MAX;
126 }

130CV_IMPL double
131cvPointPolygonTest( const CvArr* _contour, CvPoint2D32f pt, int measure_dist )
132{
133 double result = 0;
134
135 CvSeqBlock block;
136 CvContour header;
137 CvSeq* contour = (CvSeq*)_contour;
138 CvSeqReader reader;
139 int i, total, counter = 0;
140 int is_float;
141 double min_dist_num = FLT_MAX, min_dist_denom = 1;
142 CvPoint ip = {0,0};
143
144 if( !CV_IS_SEQ(contour) )
145 {
146 contour = cvPointSeqFromMat( CV_SEQ_KIND_CURVE + CV_SEQ_FLAG_CLOSED,
147 _contour, &header, &block );
148 }
149 else if( CV_IS_SEQ_POINT_SET(contour) )
150 {
151 if( contour->header_size == sizeof(CvContour) && !measure_dist )
152 {
153 CvRect r = ((CvContour*)contour)->rect;
154 if( pt.x < r.x || pt.y < r.y ||
155 pt.x >= r.x + r.width || pt.y >= r.y + r.height )
156 return -1;
157 }
158 }
159 else if( CV_IS_SEQ_CHAIN(contour) )
160 {
161 CV_Error( CV_StsBadArg,
162 "Chains are not supported. Convert them to polygonal representation using cvApproxChains()" );
163 }
164 else
165 CV_Error( CV_StsBadArg, "Input contour is neither a valid sequence nor a matrix" );
166
167 total = contour->total;
168 is_float = CV_SEQ_ELTYPE(contour) == CV_32FC2;
169 cvStartReadSeq( contour, &reader, -1 );
170
171 if( !is_float && !measure_dist && (ip.x = cvRound(pt.x)) == pt.x && (ip.y = cvRound(pt.y)) == pt.y )
172 {
173 // the fastest "pure integer" branch
174 CvPoint v0, v;
175 CV_READ_SEQ_ELEM( v, reader );
176
177 for( i = 0; i < total; i++ )
178 {
179 int dist;
180 v0 = v;
181 CV_READ_SEQ_ELEM( v, reader );
182
183 if( (v0.y <= ip.y && v.y <= ip.y) ||
184 (v0.y > ip.y && v.y > ip.y) ||
185 (v0.x < ip.x && v.x < ip.x) )
186 {
187 if( ip.y == v.y && (ip.x == v.x || (ip.y == v0.y &&
188 ((v0.x <= ip.x && ip.x <= v.x) || (v.x <= ip.x && ip.x <= v0.x)))) )
189 return 0;
190 continue;
191 }
192
193 dist = (ip.y - v0.y)*(v.x - v0.x) - (ip.x - v0.x)*(v.y - v0.y);
194 if( dist == 0 )
195 return 0;
196 if( v.y < v0.y )
197 dist = -dist;
198 counter += dist > 0;
199 }
200
201 result = counter % 2 == 0 ? -1 : 1;
202 }
203 else
204 {
205 CvPoint2D32f v0, v;
206 CvPoint iv;
207
208 if( is_float )
209 {
210 CV_READ_SEQ_ELEM( v, reader );
211 }
212 else
213 {
214 CV_READ_SEQ_ELEM( iv, reader );
215 v = cvPointTo32f( iv );
216 }
217
218 if( !measure_dist )
219 {
220 for( i = 0; i < total; i++ )
221 {
222 double dist;
223 v0 = v;
224 if( is_float )
225 {
226 CV_READ_SEQ_ELEM( v, reader );
227 }
228 else
229 {
230 CV_READ_SEQ_ELEM( iv, reader );
231 v = cvPointTo32f( iv );
232 }
233
234 if( (v0.y <= pt.y && v.y <= pt.y) ||
235 (v0.y > pt.y && v.y > pt.y) ||
236 (v0.x < pt.x && v.x < pt.x) )
237 {
238 if( pt.y == v.y && (pt.x == v.x || (pt.y == v0.y &&
239 ((v0.x <= pt.x && pt.x <= v.x) || (v.x <= pt.x && pt.x <= v0.x)))) )
240 return 0;
241 continue;
242 }
243
244 dist = (double)(pt.y - v0.y)*(v.x - v0.x) - (double)(pt.x - v0.x)*(v.y - v0.y);
245 if( dist == 0 )
246 return 0;
247 if( v.y < v0.y )
248 dist = -dist;
249 counter += dist > 0;
250 }
251
252 result = counter % 2 == 0 ? -1 : 1;
253 }
254 else
255 {
256 for( i = 0; i < total; i++ )
257 {
258 double dx, dy, dx1, dy1, dx2, dy2, dist_num, dist_denom = 1;
259
260 v0 = v;
261 if( is_float )
262 {
263 CV_READ_SEQ_ELEM( v, reader );
264 }
265 else
266 {
267 CV_READ_SEQ_ELEM( iv, reader );
268 v = cvPointTo32f( iv );
269 }
270
271 dx = v.x - v0.x; dy = v.y - v0.y;
272 dx1 = pt.x - v0.x; dy1 = pt.y - v0.y;
273 dx2 = pt.x - v.x; dy2 = pt.y - v.y;
274
275 if( dx1*dx + dy1*dy <= 0 )
276 dist_num = dx1*dx1 + dy1*dy1;
277 else if( dx2*dx + dy2*dy >= 0 )
278 dist_num = dx2*dx2 + dy2*dy2;
279 else
280 {
281 dist_num = (dy1*dx - dx1*dy);
282 dist_num *= dist_num;
283 dist_denom = dx*dx + dy*dy;
284 }
285
286 if( dist_num*min_dist_denom < min_dist_num*dist_denom )
287 {
288 min_dist_num = dist_num;
289 min_dist_denom = dist_denom;
290 if( min_dist_num == 0 )
291 break;
292 }
293
294 if( (v0.y <= pt.y && v.y <= pt.y) ||
295 (v0.y > pt.y && v.y > pt.y) ||
296 (v0.x < pt.x && v.x < pt.x) )
297 continue;
298
299 dist_num = dy1*dx - dx1*dy;
300 if( dy < 0 )
301 dist_num = -dist_num;
302 counter += dist_num > 0;
303 }
304
305 result = sqrt(min_dist_num/min_dist_denom);
306 if( counter % 2 == 0 )
307 result = -result;
308 }
309 }
310
311 return result;
312}

334static int areaSign( Point2f a, Point2f b, Point2f c )
335{
336 static const double eps = 1e-5;
337 double area2 = (b.x - a.x) * (double)(c.y - a.y) - (c.x - a.x ) * (double)(b.y - a.y);
338 return area2 > eps ? 1 : area2 < -eps ? -1 : 0;
339}

341//---------------------------------------------------------------------
342// Returns true iff point c lies on the closed segement ab.
343// Assumes it is already known that abc are collinear.
344//---------------------------------------------------------------------
345static bool between( Point2f a, Point2f b, Point2f c )
346{
347 Point2f ba, ca;
348
349 // If ab not vertical, check betweenness on x; else on y.
350 if ( a.x != b.x )
351 return ((a.x <= c.x) && (c.x <= b.x)) ||
352 ((a.x >= c.x) && (c.x >= b.x));
353 else
354 return ((a.y <= c.y) && (c.y <= b.y)) ||
355 ((a.y >= c.y) && (c.y >= b.y));
356}

358static char parallelInt( Point2f a, Point2f b, Point2f c, Point2f d, Point2f& p, Point2f& q )
359{
360 char code = 'e';
361 if( areaSign(a, b, c) != 0 )
362 code = '0';
363 else if( between(a, b, c) && between(a, b, d))
364 p = c, q = d;
365 else if( between(c, d, a) && between(c, d, b))
366 p = a, q = b;
367 else if( between(a, b, c) && between(c, d, b))
368 p = c, q = b;
369 else if( between(a, b, c) && between(c, d, a))
370 p = c, q = a;
371 else if( between(a, b, d) && between(c, d, b))
372 p = d, q = b;
373 else if( between(a, b, d) && between(c, d, a))
374 p = d, q = a;
375 else
376 code = '0';
377 return code;
378}

380//---------------------------------------------------------------------
381// segSegInt: Finds the point of intersection p between two closed
382// segments ab and cd. Returns p and a char with the following meaning:
383// 'e': The segments collinearly overlap, sharing a point.
384// 'v': An endpoint (vertex) of one segment is on the other segment,
385// but 'e' doesn't hold.
386// '1': The segments intersect properly (i.e., they share a point and
387// neither 'v' nor 'e' holds).
388// '0': The segments do not intersect (i.e., they share no points).
389// Note that two collinear segments that share just one point, an endpoint
390// of each, returns 'e' rather than 'v' as one might expect.
391//---------------------------------------------------------------------
392static char segSegInt( Point2f a, Point2f b, Point2f c, Point2f d, Point2f& p, Point2f& q )
393{
394 double s, t; // The two parameters of the parametric eqns.
395 double num, denom; // Numerator and denoninator of equations.
396 char code = '?'; // Return char characterizing intersection.
397
398 denom = a.x * (double)( d.y - c.y ) +
399 b.x * (double)( c.y - d.y ) +
400 d.x * (double)( b.y - a.y ) +
401 c.x * (double)( a.y - b.y );
402
403 // If denom is zero, then segments are parallel: handle separately.
404 if (denom == 0.0)
405 return parallelInt(a, b, c, d, p, q);
406
407 num = a.x * (double)( d.y - c.y ) +
408 c.x * (double)( a.y - d.y ) +
409 d.x * (double)( c.y - a.y );
410 if ( (num == 0.0) || (num == denom) ) code = 'v';
411 s = num / denom;
412
413 num = -( a.x * (double)( c.y - b.y ) +
414 b.x * (double)( a.y - c.y ) +
415 c.x * (double)( b.y - a.y ) );
416 if ( (num == 0.0) || (num == denom) ) code = 'v';
417 t = num / denom;
418
419 if ( (0.0 < s) && (s < 1.0) &&
420 (0.0 < t) && (t < 1.0) )
421 code = '1';
422 else if ( (0.0 > s) || (s > 1.0) ||
423 (0.0 > t) || (t > 1.0) )
424 code = '0';
425
426 p.x = (float)(a.x + s*(b.x - a.x));
427 p.y = (float)(a.y + s*(b.y - a.y));
428
429 return code;
430}

432static tInFlag inOut( Point2f p, tInFlag inflag, int aHB, int bHA, Point2f*& result )
433{
434 if( p != result[-1] )
435 *result++ = p;
436 // Update inflag.
437 return aHB > 0 ? Pin : bHA > 0 ? Qin : inflag;
438}

451static void addSharedSeg( Point2f p, Point2f q, Point2f*& result )

453 if( p != result[-1] )
454 *result++ = p;
455 if( q != result[-1] )
456 *result++ = q;

460static int intersectConvexConvex_( const Point2f* P, int n, const Point2f* Q, int m,
461 Point2f* result, float* _area )
462{
463 Point2f* result0 = result;
464 // P has n vertices, Q has m vertices.
465 int a=0, b=0; // indices on P and Q (resp.)
466 Point2f Origin(0,0);
467 tInFlag inflag=Unknown; // {Pin, Qin, Unknown}: which inside
468 int aa=0, ba=0; // # advances on a & b indices (after 1st inter.)
469 bool FirstPoint=true;// Is this the first point? (used to initialize).
470 Point2f p0; // The first point.
471 *result++ = Point2f(FLT_MAX, FLT_MAX);
472
473 do
474 {
475 // Computations of key variables.
476 int a1 = (a + n - 1) % n; // a-1, b-1 (resp.)
477 int b1 = (b + m - 1) % m;
478
479 Point2f A = P[a] - P[a1], B = Q[b] - Q[b1]; // directed edges on P and Q (resp.)
480
481 int cross = areaSign( Origin, A, B ); // sign of z-component of A x B
482 int aHB = areaSign( Q[b1], Q[b], P[a] ); // a in H(b).
483 int bHA = areaSign( P[a1], P[a], Q[b] ); // b in H(A);
484
485 // If A & B intersect, update inflag.
486 Point2f p, q;
487 int code = segSegInt( P[a1], P[a], Q[b1], Q[b], p, q );
488 if( code == '1' || code == 'v' )
489 {
490 if( inflag == Unknown && FirstPoint )
491 {
492 aa = ba = 0;
493 FirstPoint = false;
494 p0 = p;
495 *result++ = p;
496 }
497 inflag = inOut( p, inflag, aHB, bHA, result );
498 }
499
500 //-----Advance rules-----
501
502 // Special case: A & B overlap and oppositely oriented.
503 if( code == 'e' && A.ddot(B) < 0 )
504 {
505 addSharedSeg( p, q, result );
506 return (int)(result - result0);
507 }
508
509 // Special case: A & B parallel and separated.
510 if( cross == 0 && aHB < 0 && bHA < 0 )
511 return (int)(result - result0);
512
513 // Special case: A & B collinear.
514 else if ( cross == 0 && aHB == 0 && bHA == 0 ) {
515 // Advance but do not output point.
516 if ( inflag == Pin )
517 b = advance( b, &ba, m, inflag == Qin, Q[b], result );
518 else
519 a = advance( a, &aa, n, inflag == Pin, P[a], result );
520 }
521
522 // Generic cases.
523 else if( cross >= 0 )
524 {
525 if( bHA > 0)
526 a = advance( a, &aa, n, inflag == Pin, P[a], result );
527 else
528 b = advance( b, &ba, m, inflag == Qin, Q[b], result );
529 }
530 else
531 {
532 if( aHB > 0)
533 b = advance( b, &ba, m, inflag == Qin, Q[b], result );
534 else
535 a = advance( a, &aa, n, inflag == Pin, P[a], result );
536 }
537 // Quit when both adv. indices have cycled, or one has cycled twice.
538 }
539 while ( ((aa < n) || (ba < m)) && (aa < 2*n) && (ba < 2*m) );
540
541 // Deal with special cases: not implemented.
542 if( inflag == Unknown )
543 {
544 // The boundaries of P and Q do not cross.
545 // ...
546 }
547
548 int i, nr = (int)(result - result0);
549 double area = 0;
550 Point2f prev = result0[nr-1];
551 for( i = 1; i < nr; i++ )
552 {
553 result0[i-1] = result0[i];
554 area += (double)prev.x*result0[i].y - (double)prev.y*result0[i].x;
555 prev = result0[i];
556 }
557
558 *_area = (float)(area*0.5);
559
560 if( result0[nr-2] == result0[0] && nr > 1 )
561 nr--;
562 return nr-1;
563}

567float cv::intersectConvexConvex( InputArray _p1, InputArray _p2, OutputArray _p12, bool handleNested )
568{
569 Mat p1 = _p1.getMat(), p2 = _p2.getMat();
570 CV_Assert( p1.depth() == CV_32S || p1.depth() == CV_32F );
571 CV_Assert( p2.depth() == CV_32S || p2.depth() == CV_32F );
572
573 int n = p1.checkVector(2, p1.depth(), true);
574 int m = p2.checkVector(2, p2.depth(), true);
575
576 CV_Assert( n >= 0 && m >= 0 );
577
578 if( n < 2 || m < 2 )
579 {
580 _p12.release();
581 return 0.f;
582 }
583
584 AutoBuffer<Point2f> _result(n*2 + m*2 + 1);
585 Point2f *fp1 = _result, *fp2 = fp1 + n;
586 Point2f* result = fp2 + m;
587 int orientation = 0;
588
589 for( int k = 1; k <= 2; k++ )
590 {
591 Mat& p = k == 1 ? p1 : p2;
592 int len = k == 1 ? n : m;
593 Point2f* dst = k == 1 ? fp1 : fp2;
594
595 Mat temp(p.size(), CV_MAKETYPE(CV_32F, p.channels()), dst);
596 p.convertTo(temp, CV_32F);
597 CV_Assert( temp.ptr<Point2f>() == dst );
598 Point2f diff0 = dst[0] - dst[len-1];
599 for( int i = 1; i < len; i++ )
600 {
601 double s = diff0.cross(dst[i] - dst[i-1]);
602 if( s != 0 )
603 {
604 if( s < 0 )
605 {
606 orientation++;
607 flip( temp, temp, temp.rows > 1 ? 0 : 1 );
608 }
609 break;
610 }
611 }
612 }
613
614 float area = 0.f;
615 int nr = intersectConvexConvex_(fp1, n, fp2, m, result, &area);
616 if( nr == 0 )
617 {
618 if( !handleNested )
619 {
620 _p12.release();
621 return 0.f;
622 }
623
624 if( pointPolygonTest(_InputArray(fp1, n), fp2[0], false) >= 0 )
625 {
626 result = fp2;
627 nr = m;
628 }
629 else if( pointPolygonTest(_InputArray(fp2, n), fp1[0], false) >= 0 )
630 {
631 result = fp1;
632 nr = n;
633 }
634 else
635 {
636 _p12.release();
637 return 0.f;
638 }
639 area = (float)contourArea(_InputArray(result, nr), false);
640 }
641
642 if( _p12.needed() )
643 {
644 Mat temp(nr, 1, CV_32FC2, result);
645 // if both input contours were reflected,
646 // let's orient the result as the input vectors
647 if( orientation == 2 )
648 flip(temp, temp, 0);
649
650 temp.copyTo(_p12);
651 }
652 return (float)fabs(area);
653}
