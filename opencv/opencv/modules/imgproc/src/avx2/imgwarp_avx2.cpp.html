47const int INTER_RESIZE_COEF_BITS=11;
48const int INTER_RESIZE_COEF_SCALE=1 << INTER_RESIZE_COEF_BITS;

50int VResizeLinearVec_32s8u_avx2(const uchar** _src, uchar* dst, const uchar* _beta, int width )
51{
52 const int** src = (const int**)_src;
53 const short* beta = (const short*)_beta;
54 const int *S0 = src[0], *S1 = src[1];
55 int x = 0;
56 __m256i b0 = _mm256_set1_epi16(beta[0]), b1 = _mm256_set1_epi16(beta[1]);
57 __m256i delta = _mm256_set1_epi16(2);
58 const int index[8] = { 0, 4, 1, 5, 2, 6, 3, 7 };
59 __m256i shuffle = _mm256_load_si256((const __m256i*)index);
60
61 if( (((size_t)S0|(size_t)S1)&31) == 0 )
62 for( ; x <= width - 32; x += 32 )
63 {
64 __m256i x0, x1, x2, y0, y1, y2;
65 x0 = _mm256_load_si256((const __m256i*)(S0 + x));
66 x1 = _mm256_load_si256((const __m256i*)(S0 + x + 8));
67 y0 = _mm256_load_si256((const __m256i*)(S1 + x));
68 y1 = _mm256_load_si256((const __m256i*)(S1 + x + 8));
69 x0 = _mm256_packs_epi32(_mm256_srai_epi32(x0, 4), _mm256_srai_epi32(x1, 4));
70 y0 = _mm256_packs_epi32(_mm256_srai_epi32(y0, 4), _mm256_srai_epi32(y1, 4));
71
72 x1 = _mm256_load_si256((const __m256i*)(S0 + x + 16));
73 x2 = _mm256_load_si256((const __m256i*)(S0 + x + 24));
74 y1 = _mm256_load_si256((const __m256i*)(S1 + x + 16));
75 y2 = _mm256_load_si256((const __m256i*)(S1 + x + 24));
76 x1 = _mm256_packs_epi32(_mm256_srai_epi32(x1, 4), _mm256_srai_epi32(x2, 4));
77 y1 = _mm256_packs_epi32(_mm256_srai_epi32(y1, 4), _mm256_srai_epi32(y2, 4));
78
79 x0 = _mm256_adds_epi16(_mm256_mulhi_epi16(x0, b0), _mm256_mulhi_epi16(y0, b1));
80 x1 = _mm256_adds_epi16(_mm256_mulhi_epi16(x1, b0), _mm256_mulhi_epi16(y1, b1));
81
82 x0 = _mm256_srai_epi16(_mm256_adds_epi16(x0, delta), 2);
83 x1 = _mm256_srai_epi16(_mm256_adds_epi16(x1, delta), 2);
84 x0 = _mm256_packus_epi16(x0, x1);
85 x0 = _mm256_permutevar8x32_epi32(x0, shuffle);
86 _mm256_storeu_si256( (__m256i*)(dst + x), x0);
87 }
88 else
89 for( ; x <= width - 32; x += 32 )
90 {
91 __m256i x0, x1, x2, y0, y1, y2;
92 x0 = _mm256_loadu_si256((const __m256i*)(S0 + x));
93 x1 = _mm256_loadu_si256((const __m256i*)(S0 + x + 8));
94 y0 = _mm256_loadu_si256((const __m256i*)(S1 + x));
95 y1 = _mm256_loadu_si256((const __m256i*)(S1 + x + 8));
96 x0 = _mm256_packs_epi32(_mm256_srai_epi32(x0, 4), _mm256_srai_epi32(x1, 4));
97 y0 = _mm256_packs_epi32(_mm256_srai_epi32(y0, 4), _mm256_srai_epi32(y1, 4));
98
99 x1 = _mm256_loadu_si256((const __m256i*)(S0 + x + 16));
100 x2 = _mm256_loadu_si256((const __m256i*)(S0 + x + 24));
101 y1 = _mm256_loadu_si256((const __m256i*)(S1 + x + 16));
102 y2 = _mm256_loadu_si256((const __m256i*)(S1 + x + 24));
103 x1 = _mm256_packs_epi32(_mm256_srai_epi32(x1, 4), _mm256_srai_epi32(x2, 4));
104 y1 = _mm256_packs_epi32(_mm256_srai_epi32(y1, 4), _mm256_srai_epi32(y2, 4));
105
106 x0 = _mm256_adds_epi16(_mm256_mulhi_epi16(x0, b0), _mm256_mulhi_epi16(y0, b1));
107 x1 = _mm256_adds_epi16(_mm256_mulhi_epi16(x1, b0), _mm256_mulhi_epi16(y1, b1));
108
109 x0 = _mm256_srai_epi16(_mm256_adds_epi16(x0, delta), 2);
110 x1 = _mm256_srai_epi16(_mm256_adds_epi16(x1, delta), 2);
111 x0 = _mm256_packus_epi16(x0, x1);
112 x0 = _mm256_permutevar8x32_epi32(x0, shuffle);
113 _mm256_storeu_si256( (__m256i*)(dst + x), x0);
114 }
115
116 for( ; x < width - 8; x += 8 )
117 {
118 __m256i x0, y0;
119 x0 = _mm256_srai_epi32(_mm256_loadu_si256((const __m256i*)(S0 + x)), 4);
120 y0 = _mm256_srai_epi32(_mm256_loadu_si256((const __m256i*)(S1 + x)), 4);
121 x0 = _mm256_packs_epi32(x0, x0);
122 y0 = _mm256_packs_epi32(y0, y0);
123 x0 = _mm256_adds_epi16(_mm256_mulhi_epi16(x0, b0), _mm256_mulhi_epi16(y0, b1));
124 x0 = _mm256_srai_epi16(_mm256_adds_epi16(x0, delta), 2);
125 x0 = _mm256_packus_epi16(x0, x0);
126 *(int*)(dst + x) = _mm_cvtsi128_si32(_mm256_extracti128_si256(x0, 0));
127 *(int*)(dst + x + 4) = _mm_cvtsi128_si32(_mm256_extracti128_si256(x0, 1));
128 }
129
130 return x;
131}

133template<int shiftval>
134int VResizeLinearVec_32f16_avx2(const uchar** _src, uchar* _dst, const uchar* _beta, int width )
135{
136 const float** src = (const float**)_src;
137 const float* beta = (const float*)_beta;
138 const float *S0 = src[0], *S1 = src[1];
139 ushort* dst = (ushort*)_dst;
140 int x = 0;
141
142 __m256 b0 = _mm256_set1_ps(beta[0]), b1 = _mm256_set1_ps(beta[1]);
143 __m256i preshift = _mm256_set1_epi32(shiftval);
144 __m256i postshift = _mm256_set1_epi16((short)shiftval);
145
146 if( (((size_t)S0|(size_t)S1)&31) == 0 )
147 for( ; x <= width - 32; x += 32 )
148 {
149 __m256 x0, x1, y0, y1;
150 __m256i t0, t1, t2;
151 x0 = _mm256_load_ps(S0 + x);
152 x1 = _mm256_load_ps(S0 + x + 8);
153 y0 = _mm256_load_ps(S1 + x);
154 y1 = _mm256_load_ps(S1 + x + 8);
155
156 x0 = _mm256_add_ps(_mm256_mul_ps(x0, b0), _mm256_mul_ps(y0, b1));
157 x1 = _mm256_add_ps(_mm256_mul_ps(x1, b0), _mm256_mul_ps(y1, b1));
158 t0 = _mm256_add_epi32(_mm256_cvtps_epi32(x0), preshift);
159 t2 = _mm256_add_epi32(_mm256_cvtps_epi32(x1), preshift);
160 t0 = _mm256_add_epi16(_mm256_packs_epi32(t0, t2), postshift);
161
162 x0 = _mm256_load_ps(S0 + x + 16);
163 x1 = _mm256_load_ps(S0 + x + 24);
164 y0 = _mm256_load_ps(S1 + x + 16);
165 y1 = _mm256_load_ps(S1 + x + 24);
166
167 x0 = _mm256_add_ps(_mm256_mul_ps(x0, b0), _mm256_mul_ps(y0, b1));
168 x1 = _mm256_add_ps(_mm256_mul_ps(x1, b0), _mm256_mul_ps(y1, b1));
169 t1 = _mm256_add_epi32(_mm256_cvtps_epi32(x0), preshift);
170 t2 = _mm256_add_epi32(_mm256_cvtps_epi32(x1), preshift);
171 t1 = _mm256_add_epi16(_mm256_packs_epi32(t1, t2), postshift);
172
173 _mm256_storeu_si256( (__m256i*)(dst + x), t0);
174 _mm256_storeu_si256( (__m256i*)(dst + x + 16), t1);
175 }
176 else
177 for( ; x <= width - 32; x += 32 )
178 {
179 __m256 x0, x1, y0, y1;
180 __m256i t0, t1, t2;
181 x0 = _mm256_loadu_ps(S0 + x);
182 x1 = _mm256_loadu_ps(S0 + x + 8);
183 y0 = _mm256_loadu_ps(S1 + x);
184 y1 = _mm256_loadu_ps(S1 + x + 8);
185
186 x0 = _mm256_add_ps(_mm256_mul_ps(x0, b0), _mm256_mul_ps(y0, b1));
187 x1 = _mm256_add_ps(_mm256_mul_ps(x1, b0), _mm256_mul_ps(y1, b1));
188 t0 = _mm256_add_epi32(_mm256_cvtps_epi32(x0), preshift);
189 t2 = _mm256_add_epi32(_mm256_cvtps_epi32(x1), preshift);
190 t0 = _mm256_add_epi16(_mm256_packs_epi32(t0, t2), postshift);
191
192 x0 = _mm256_loadu_ps(S0 + x + 16);
193 x1 = _mm256_loadu_ps(S0 + x + 24);
194 y0 = _mm256_loadu_ps(S1 + x + 16);
195 y1 = _mm256_loadu_ps(S1 + x + 24);
196
197 x0 = _mm256_add_ps(_mm256_mul_ps(x0, b0), _mm256_mul_ps(y0, b1));
198 x1 = _mm256_add_ps(_mm256_mul_ps(x1, b0), _mm256_mul_ps(y1, b1));
199 t1 = _mm256_add_epi32(_mm256_cvtps_epi32(x0), preshift);
200 t2 = _mm256_add_epi32(_mm256_cvtps_epi32(x1), preshift);
201 t1 = _mm256_add_epi16(_mm256_packs_epi32(t1, t2), postshift);
202
203 _mm256_storeu_si256( (__m256i*)(dst + x), t0);
204 _mm256_storeu_si256( (__m256i*)(dst + x + 16), t1);
205 }
206
207 for( ; x < width - 8; x += 8 )
208 {
209 __m256 x0, y0;
210 __m256i t0;
211 x0 = _mm256_loadu_ps(S0 + x);
212 y0 = _mm256_loadu_ps(S1 + x);
213
214 x0 = _mm256_add_ps(_mm256_mul_ps(x0, b0), _mm256_mul_ps(y0, b1));
215 t0 = _mm256_add_epi32(_mm256_cvtps_epi32(x0), preshift);
216 t0 = _mm256_add_epi16(_mm256_packs_epi32(t0, t0), postshift);
217 _mm_storel_epi64( (__m128i*)(dst + x), _mm256_extracti128_si256(t0, 0));
218 _mm_storel_epi64( (__m128i*)(dst + x + 4), _mm256_extracti128_si256(t0, 1));
219 }
220
221 return x;
222}

224int VResizeCubicVec_32s8u_avx2(const uchar** _src, uchar* dst, const uchar* _beta, int width )
225{
226 const int** src = (const int**)_src;
227 const short* beta = (const short*)_beta;
228 const int *S0 = src[0], *S1 = src[1], *S2 = src[2], *S3 = src[3];
229 int x = 0;
230 float scale = 1.f/(INTER_RESIZE_COEF_SCALE*INTER_RESIZE_COEF_SCALE);
231 __m256 b0 = _mm256_set1_ps(beta[0]*scale), b1 = _mm256_set1_ps(beta[1]*scale),
232 b2 = _mm256_set1_ps(beta[2]*scale), b3 = _mm256_set1_ps(beta[3]*scale);
233 const int shuffle = 0xd8; // 11 | 01 | 10 | 00
234
235 if( (((size_t)S0|(size_t)S1|(size_t)S2|(size_t)S3)&31) == 0 )
236 for( ; x <= width - 16; x += 16 )
237 {
238 __m256i x0, x1, y0, y1;
239 __m256 s0, s1, f0, f1;
240 x0 = _mm256_load_si256((const __m256i*)(S0 + x));
241 x1 = _mm256_load_si256((const __m256i*)(S0 + x + 8));
242 y0 = _mm256_load_si256((const __m256i*)(S1 + x));
243 y1 = _mm256_load_si256((const __m256i*)(S1 + x + 8));
244
245 s0 = _mm256_mul_ps(_mm256_cvtepi32_ps(x0), b0);
246 s1 = _mm256_mul_ps(_mm256_cvtepi32_ps(x1), b0);
247 f0 = _mm256_mul_ps(_mm256_cvtepi32_ps(y0), b1);
248 f1 = _mm256_mul_ps(_mm256_cvtepi32_ps(y1), b1);
249 s0 = _mm256_add_ps(s0, f0);
250 s1 = _mm256_add_ps(s1, f1);
251
252 x0 = _mm256_load_si256((const __m256i*)(S2 + x));
253 x1 = _mm256_load_si256((const __m256i*)(S2 + x + 8));
254 y0 = _mm256_load_si256((const __m256i*)(S3 + x));
255 y1 = _mm256_load_si256((const __m256i*)(S3 + x + 8));
256
257 f0 = _mm256_mul_ps(_mm256_cvtepi32_ps(x0), b2);
258 f1 = _mm256_mul_ps(_mm256_cvtepi32_ps(x1), b2);
259 s0 = _mm256_add_ps(s0, f0);
260 s1 = _mm256_add_ps(s1, f1);
261 f0 = _mm256_mul_ps(_mm256_cvtepi32_ps(y0), b3);
262 f1 = _mm256_mul_ps(_mm256_cvtepi32_ps(y1), b3);
263 s0 = _mm256_add_ps(s0, f0);
264 s1 = _mm256_add_ps(s1, f1);
265
266 x0 = _mm256_cvtps_epi32(s0);
267 x1 = _mm256_cvtps_epi32(s1);
268
269 x0 = _mm256_packs_epi32(x0, x1);
270 x0 = _mm256_permute4x64_epi64(x0, shuffle);
271 x0 = _mm256_packus_epi16(x0, x0);
272 _mm_storel_epi64( (__m128i*)(dst + x), _mm256_extracti128_si256(x0, 0));
273 _mm_storel_epi64( (__m128i*)(dst + x + 8), _mm256_extracti128_si256(x0, 1));
274 }
275 else
276 for( ; x <= width - 16; x += 16 )
277 {
278 __m256i x0, x1, y0, y1;
279 __m256 s0, s1, f0, f1;
280 x0 = _mm256_loadu_si256((const __m256i*)(S0 + x));
281 x1 = _mm256_loadu_si256((const __m256i*)(S0 + x + 8));
282 y0 = _mm256_loadu_si256((const __m256i*)(S1 + x));
283 y1 = _mm256_loadu_si256((const __m256i*)(S1 + x + 8));
284
285 s0 = _mm256_mul_ps(_mm256_cvtepi32_ps(x0), b0);
286 s1 = _mm256_mul_ps(_mm256_cvtepi32_ps(x1), b0);
287 f0 = _mm256_mul_ps(_mm256_cvtepi32_ps(y0), b1);
288 f1 = _mm256_mul_ps(_mm256_cvtepi32_ps(y1), b1);
289 s0 = _mm256_add_ps(s0, f0);
290 s1 = _mm256_add_ps(s1, f1);
291
292 x0 = _mm256_loadu_si256((const __m256i*)(S2 + x));
293 x1 = _mm256_loadu_si256((const __m256i*)(S2 + x + 8));
294 y0 = _mm256_loadu_si256((const __m256i*)(S3 + x));
295 y1 = _mm256_loadu_si256((const __m256i*)(S3 + x + 8));
296
297 f0 = _mm256_mul_ps(_mm256_cvtepi32_ps(x0), b2);
298 f1 = _mm256_mul_ps(_mm256_cvtepi32_ps(x1), b2);
299 s0 = _mm256_add_ps(s0, f0);
300 s1 = _mm256_add_ps(s1, f1);
301 f0 = _mm256_mul_ps(_mm256_cvtepi32_ps(y0), b3);
302 f1 = _mm256_mul_ps(_mm256_cvtepi32_ps(y1), b3);
303 s0 = _mm256_add_ps(s0, f0);
304 s1 = _mm256_add_ps(s1, f1);
305
306 x0 = _mm256_cvtps_epi32(s0);
307 x1 = _mm256_cvtps_epi32(s1);
308
309 x0 = _mm256_packs_epi32(x0, x1);
310 x0 = _mm256_permute4x64_epi64(x0, shuffle);
311 x0 = _mm256_packus_epi16(x0, x0);
312 _mm_storel_epi64( (__m128i*)(dst + x), _mm256_extracti128_si256(x0, 0));
313 _mm_storel_epi64( (__m128i*)(dst + x + 8), _mm256_extracti128_si256(x0, 1));
314 }
315
316 return x;
317}

341 y1 = _mm256_load_ps(S1 + x + 8);

344 s1 = _mm256_mul_ps(x1, b0);
345 y0 = _mm256_mul_ps(y0, b1);
346 y1 = _mm256_mul_ps(y1, b1);
347 s0 = _mm256_add_ps(s0, y0);
348 s1 = _mm256_add_ps(s1, y1);
349
350 x0 = _mm256_load_ps(S2 + x);
351 x1 = _mm256_load_ps(S2 + x + 8);
352 y0 = _mm256_load_ps(S3 + x);
353 y1 = _mm256_load_ps(S3 + x + 8);
354
355 x0 = _mm256_mul_ps(x0, b2);

358 y1 = _mm256_mul_ps(y1, b3);

363
364 t0 = _mm256_add_epi32(_mm256_cvtps_epi32(s0), preshift);
365 t1 = _mm256_add_epi32(_mm256_cvtps_epi32(s1), preshift);
366
367 t0 = _mm256_add_epi16(_mm256_packs_epi32(t0, t1), postshift);
368 t0 = _mm256_permute4x64_epi64(t0, shuffle);
369 _mm256_storeu_si256( (__m256i*)(dst + x), t0);
370 }
371 else
372 for( ; x <= width - 16; x += 16 )
373 {
374 __m256 x0, x1, y0, y1, s0, s1;
375 __m256i t0, t1;
376 x0 = _mm256_loadu_ps(S0 + x);
377 x1 = _mm256_loadu_ps(S0 + x + 8);
378 y0 = _mm256_loadu_ps(S1 + x);
379 y1 = _mm256_loadu_ps(S1 + x + 8);
380
381 s0 = _mm256_mul_ps(x0, b0);
382 s1 = _mm256_mul_ps(x1, b0);
383 y0 = _mm256_mul_ps(y0, b1);
384 y1 = _mm256_mul_ps(y1, b1);
385 s0 = _mm256_add_ps(s0, y0);
386 s1 = _mm256_add_ps(s1, y1);
387
388 x0 = _mm256_loadu_ps(S2 + x);
389 x1 = _mm256_loadu_ps(S2 + x + 8);
390 y0 = _mm256_loadu_ps(S3 + x);
391 y1 = _mm256_loadu_ps(S3 + x + 8);
392
393 x0 = _mm256_mul_ps(x0, b2);
394 x1 = _mm256_mul_ps(x1, b2);
395 y0 = _mm256_mul_ps(y0, b3);
396 y1 = _mm256_mul_ps(y1, b3);
397 s0 = _mm256_add_ps(s0, x0);
398 s1 = _mm256_add_ps(s1, x1);
399 s0 = _mm256_add_ps(s0, y0);
400 s1 = _mm256_add_ps(s1, y1);
401
402 t0 = _mm256_add_epi32(_mm256_cvtps_epi32(s0), preshift);
403 t1 = _mm256_add_epi32(_mm256_cvtps_epi32(s1), preshift);
404
