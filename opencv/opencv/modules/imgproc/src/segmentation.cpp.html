48typedef struct CvWSNode
49{
50 struct CvWSNode* next;
51 int mask_ofs;
52 int img_ofs;
53}
54CvWSNode;
55
56typedef struct CvWSQueue
57{
58 CvWSNode* first;
59 CvWSNode* last;
60}
61CvWSQueue;
62
63static CvWSNode*
64icvAllocWSNodes( CvMemStorage* storage )
65{
66 CvWSNode* n = 0;
67
68 int i, count = (storage->block_size - sizeof(CvMemBlock))/sizeof(*n) - 1;
69
70 n = (CvWSNode*)cvMemStorageAlloc( storage, count*sizeof(*n) );
71 for( i = 0; i < count-1; i++ )
72 n[i].next = n + i + 1;
73 n[count-1].next = 0;
74
75 return n;
76}

79CV_IMPL void
80cvWatershed( const CvArr* srcarr, CvArr* dstarr )
81{
82 const int IN_QUEUE = -2;
83 const int WSHED = -1;
84 const int NQ = 256;
85 cv::Ptr<CvMemStorage> storage;
86
87 CvMat sstub, *src;
88 CvMat dstub, *dst;
89 CvSize size;
90 CvWSNode* free_node = 0, *node;
91 CvWSQueue q[NQ];
92 int active_queue;
93 int i, j;
94 int db, dg, dr;
95 int* mask;
96 uchar* img;
97 int mstep, istep;
98 int subs_tab[513];
99
100 // MAX(a,b) = b + MAX(a-b,0)
101 #define ws_max(a,b) ((b) + subs_tab[(a)-(b)+NQ])
102 // MIN(a,b) = a - MAX(a-b,0)
103 #define ws_min(a,b) ((a) - subs_tab[(a)-(b)+NQ])
104
105 #define ws_push(idx,mofs,iofs) \
106 { \
107 if( !free_node ) \
108 free_node = icvAllocWSNodes( storage );\
109 node = free_node; \
110 free_node = free_node->next;\
111 node->next = 0; \
112 node->mask_ofs = mofs; \
113 node->img_ofs = iofs; \
114 if( q[idx].last ) \
115 q[idx].last->next=node; \
116 else \
117 q[idx].first = node; \
118 q[idx].last = node; \
119 }
120
121 #define ws_pop(idx,mofs,iofs) \
122 { \
123 node = q[idx].first; \
124 q[idx].first = node->next; \
125 if( !node->next ) \
126 q[idx].last = 0; \
127 node->next = free_node; \
128 free_node = node; \
129 mofs = node->mask_ofs; \
130 iofs = node->img_ofs; \
131 }
132
133 #define c_diff(ptr1,ptr2,diff) \
134 { \
135 db = abs((ptr1)[0] - (ptr2)[0]);\
136 dg = abs((ptr1)[1] - (ptr2)[1]);\
137 dr = abs((ptr1)[2] - (ptr2)[2]);\
138 diff = ws_max(db,dg); \
139 diff = ws_max(diff,dr); \
140 assert( 0 <= diff && diff <= 255 ); \
141 }
142
143 src = cvGetMat( srcarr, &sstub );
144 dst = cvGetMat( dstarr, &dstub );
145
146 if( CV_MAT_TYPE(src->type) != CV_8UC3 )
147 CV_Error( CV_StsUnsupportedFormat, "Only 8-bit, 3-channel input images are supported" );
148
149 if( CV_MAT_TYPE(dst->type) != CV_32SC1 )
150 CV_Error( CV_StsUnsupportedFormat,
151 "Only 32-bit, 1-channel output images are supported" );
152
153 if( !CV_ARE_SIZES_EQ( src, dst ))
154 CV_Error( CV_StsUnmatchedSizes, "The input and output images must have the same size" );
155
156 size = cvGetMatSize(src);
157 storage = cvCreateMemStorage();
158
159 istep = src->step;
160 img = src->data.ptr;
161 mstep = dst->step / sizeof(mask[0]);
162 mask = dst->data.i;
163
164 memset( q, 0, NQ*sizeof(q[0]) );
165
166 for( i = 0; i < 256; i++ )
167 subs_tab[i] = 0;
168 for( i = 256; i <= 512; i++ )
169 subs_tab[i] = i - 256;
170
171 // draw a pixel-wide border of dummy "watershed" (i.e. boundary) pixels
172 for( j = 0; j < size.width; j++ )
173 mask[j] = mask[j + mstep*(size.height-1)] = WSHED;
174
175 // initial phase: put all the neighbor pixels of each marker to the ordered queue -
176 // determine the initial boundaries of the basins
177 for( i = 1; i < size.height-1; i++ )
178 {
179 img += istep; mask += mstep;
180 mask[0] = mask[size.width-1] = WSHED;
181
182 for( j = 1; j < size.width-1; j++ )
183 {
184 int* m = mask + j;
185 if( m[0] < 0 ) m[0] = 0;
186 if( m[0] == 0 && (m[-1] > 0 || m[1] > 0 || m[-mstep] > 0 || m[mstep] > 0) )
187 {
188 uchar* ptr = img + j*3;
189 int idx = 256, t;
190 if( m[-1] > 0 )
191 c_diff( ptr, ptr - 3, idx );
192 if( m[1] > 0 )
193 {
194 c_diff( ptr, ptr + 3, t );
195 idx = ws_min( idx, t );
196 }
197 if( m[-mstep] > 0 )
198 {
199 c_diff( ptr, ptr - istep, t );
200 idx = ws_min( idx, t );
201 }
202 if( m[mstep] > 0 )
203 {
204 c_diff( ptr, ptr + istep, t );
205 idx = ws_min( idx, t );
206 }
207 assert( 0 <= idx && idx <= 255 );
208 ws_push( idx, i*mstep + j, i*istep + j*3 );
209 m[0] = IN_QUEUE;
210 }
211 }
212 }
213
214 // find the first non-empty queue
215 for( i = 0; i < NQ; i++ )
216 if( q[i].first )
217 break;
218
219 // if there is no markers, exit immediately
220 if( i == NQ )
221 return;
222
223 active_queue = i;
224 img = src->data.ptr;
225 mask = dst->data.i;
226
227 // recursively fill the basins
228 for(;;)
229 {
230 int mofs, iofs;
231 int lab = 0, t;
232 int* m;
233 uchar* ptr;
234
235 if( q[active_queue].first == 0 )
236 {
237 for( i = active_queue+1; i < NQ; i++ )
238 if( q[i].first )
239 break;
240 if( i == NQ )
241 break;
242 active_queue = i;
243 }
244
245 ws_pop( active_queue, mofs, iofs );
246
247 m = mask + mofs;
248 ptr = img + iofs;
249 t = m[-1];
250 if( t > 0 ) lab = t;
251 t = m[1];
252 if( t > 0 )
253 {
254 if( lab == 0 ) lab = t;
255 else if( t != lab ) lab = WSHED;
256 }
257 t = m[-mstep];
258 if( t > 0 )
259 {
260 if( lab == 0 ) lab = t;
261 else if( t != lab ) lab = WSHED;
262 }
263 t = m[mstep];
264 if( t > 0 )
265 {
266 if( lab == 0 ) lab = t;
267 else if( t != lab ) lab = WSHED;
268 }
269 assert( lab != 0 );
270 m[0] = lab;
271 if( lab == WSHED )
272 continue;
273
274 if( m[-1] == 0 )
275 {
276 c_diff( ptr, ptr - 3, t );
277 ws_push( t, mofs - 1, iofs - 3 );
278 active_queue = ws_min( active_queue, t );
279 m[-1] = IN_QUEUE;
280 }
281 if( m[1] == 0 )
282 {
283 c_diff( ptr, ptr + 3, t );
284 ws_push( t, mofs + 1, iofs + 3 );
285 active_queue = ws_min( active_queue, t );
286 m[1] = IN_QUEUE;
287 }
288 if( m[-mstep] == 0 )
289 {
290 c_diff( ptr, ptr - istep, t );
291 ws_push( t, mofs - mstep, iofs - istep );
292 active_queue = ws_min( active_queue, t );
293 m[-mstep] = IN_QUEUE;
294 }
295 if( m[mstep] == 0 )
296 {
297 c_diff( ptr, ptr + istep, t );
298 ws_push( t, mofs + mstep, iofs + istep );
299 active_queue = ws_min( active_queue, t );
300 m[mstep] = IN_QUEUE;
301 }
302 }
303}

306void cv::watershed( InputArray _src, InputOutputArray markers )
307{
308 Mat src = _src.getMat();
309 CvMat c_src = _src.getMat(), c_markers = markers.getMat();
310 cvWatershed( &c_src, &c_markers );

318CV_IMPL void
319cvPyrMeanShiftFiltering( const CvArr* srcarr, CvArr* dstarr,
320 double sp0, double sr, int max_level,
321 CvTermCriteria termcrit )
322{
323 const int cn = 3;
324 const int MAX_LEVELS = 8;
325
326 if( (unsigned)max_level > (unsigned)MAX_LEVELS )
327 CV_Error( CV_StsOutOfRange, "The number of pyramid levels is too large or negative" );
328
329 std::vector<cv::Mat> src_pyramid(max_level+1);
330 std::vector<cv::Mat> dst_pyramid(max_level+1);
331 cv::Mat mask0;
332 int i, j, level;
333 //uchar* submask = 0;
334
335 #define cdiff(ofs0) (tab[c0-dptr[ofs0]+255] + \
336 tab[c1-dptr[(ofs0)+1]+255] + tab[c2-dptr[(ofs0)+2]+255] >= isr22)
337
338 double sr2 = sr * sr;
339 int isr2 = cvRound(sr2), isr22 = MAX(isr2,16);
340 int tab[768];
341 cv::Mat src0 = cv::cvarrToMat(srcarr);
342 cv::Mat dst0 = cv::cvarrToMat(dstarr);
343
344 if( src0.type() != CV_8UC3 )
345 CV_Error( CV_StsUnsupportedFormat, "Only 8-bit, 3-channel images are supported" );
346
347 if( src0.type() != dst0.type() )
348 CV_Error( CV_StsUnmatchedFormats, "The input and output images must have the same type" );
349
350 if( src0.size() != dst0.size() )
351 CV_Error( CV_StsUnmatchedSizes, "The input and output images must have the same size" );
352
353 if( !(termcrit.type & CV_TERMCRIT_ITER) )
354 termcrit.max_iter = 5;
355 termcrit.max_iter = MAX(termcrit.max_iter,1);
356 termcrit.max_iter = MIN(termcrit.max_iter,100);
357 if( !(termcrit.type & CV_TERMCRIT_EPS) )
358 termcrit.epsilon = 1.f;
359 termcrit.epsilon = MAX(termcrit.epsilon, 0.f);
360
361 for( i = 0; i < 768; i++ )
362 tab[i] = (i - 255)*(i - 255);
363
364 // 1. construct pyramid
365 src_pyramid[0] = src0;
366 dst_pyramid[0] = dst0;
367 for( level = 1; level <= max_level; level++ )
368 {
369 src_pyramid[level].create( (src_pyramid[level-1].rows+1)/2,
370 (src_pyramid[level-1].cols+1)/2, src_pyramid[level-1].type() );
371 dst_pyramid[level].create( src_pyramid[level].rows,
372 src_pyramid[level].cols, src_pyramid[level].type() );
373 cv::pyrDown( src_pyramid[level-1], src_pyramid[level], src_pyramid[level].size() );
374 //CV_CALL( cvResize( src_pyramid[level-1], src_pyramid[level], CV_INTER_AREA ));
375 }
376
377 mask0.create(src0.rows, src0.cols, CV_8UC1);
378 //CV_CALL( submask = (uchar*)cvAlloc( (sp+2)*(sp+2) ));
379
380 // 2. apply meanshift, starting from the pyramid top (i.e. the smallest layer)
381 for( level = max_level; level >= 0; level-- )
382 {
383 cv::Mat src = src_pyramid[level];
384 cv::Size size = src.size();
385 uchar* sptr = src.data;
386 int sstep = (int)src.step;
387 uchar* mask = 0;
388 int mstep = 0;
389 uchar* dptr;
390 int dstep;
391 float sp = (float)(sp0 / (1 << level));
392 sp = MAX( sp, 1 );
393
394 if( level < max_level )
395 {
396 cv::Size size1 = dst_pyramid[level+1].size();
397 cv::Mat m( size.height, size.width, CV_8UC1, mask0.data );
398 dstep = (int)dst_pyramid[level+1].step;
399 dptr = dst_pyramid[level+1].data + dstep + cn;
400 mstep = (int)m.step;
401 mask = m.data + mstep;
402 //cvResize( dst_pyramid[level+1], dst_pyramid[level], CV_INTER_CUBIC );
403 cv::pyrUp( dst_pyramid[level+1], dst_pyramid[level], dst_pyramid[level].size() );
404 m.setTo(cv::Scalar::all(0));
405
406 for( i = 1; i < size1.height-1; i++, dptr += dstep - (size1.width-2)*3, mask += mstep*2 )
407 {
408 for( j = 1; j < size1.width-1; j++, dptr += cn )
409 {
410 int c0 = dptr[0], c1 = dptr[1], c2 = dptr[2];
411 mask[j*2 - 1] = cdiff(-3) || cdiff(3) || cdiff(-dstep-3) || cdiff(-dstep) ||
412 cdiff(-dstep+3) || cdiff(dstep-3) || cdiff(dstep) || cdiff(dstep+3);
413 }
414 }
415
416 cv::dilate( m, m, cv::Mat() );
417 mask = m.data;
418 }
419
420 dptr = dst_pyramid[level].data;
421 dstep = (int)dst_pyramid[level].step;
422
423 for( i = 0; i < size.height; i++, sptr += sstep - size.width*3,
424 dptr += dstep - size.width*3,
425 mask += mstep )
426 {
427 for( j = 0; j < size.width; j++, sptr += 3, dptr += 3 )
428 {
429 int x0 = j, y0 = i, x1, y1, iter;
430 int c0, c1, c2;
431
432 if( mask && !mask[j] )
433 continue;
434
435 c0 = sptr[0], c1 = sptr[1], c2 = sptr[2];
436
437 // iterate meanshift procedure
438 for( iter = 0; iter < termcrit.max_iter; iter++ )
439 {
440 uchar* ptr;
441 int x, y, count = 0;
442 int minx, miny, maxx, maxy;
443 int s0 = 0, s1 = 0, s2 = 0, sx = 0, sy = 0;
444 double icount;
445 int stop_flag;
446
447 //mean shift: process pixels in window (p-sigmaSp)x(p+sigmaSp)
448 minx = cvRound(x0 - sp); minx = MAX(minx, 0);
449 miny = cvRound(y0 - sp); miny = MAX(miny, 0);
450 maxx = cvRound(x0 + sp); maxx = MIN(maxx, size.width-1);
451 maxy = cvRound(y0 + sp); maxy = MIN(maxy, size.height-1);
452 ptr = sptr + (miny - i)*sstep + (minx - j)*3;
453
454 for( y = miny; y <= maxy; y++, ptr += sstep - (maxx-minx+1)*3 )
455 {
456 int row_count = 0;
457 x = minx;
458 #if CV_ENABLE_UNROLLED
459 for( ; x + 3 <= maxx; x += 4, ptr += 12 )
460 {
461 int t0 = ptr[0], t1 = ptr[1], t2 = ptr[2];
462 if( tab[t0-c0+255] + tab[t1-c1+255] + tab[t2-c2+255] <= isr2 )
463 {
464 s0 += t0; s1 += t1; s2 += t2;
465 sx += x; row_count++;
466 }
467 t0 = ptr[3], t1 = ptr[4], t2 = ptr[5];
468 if( tab[t0-c0+255] + tab[t1-c1+255] + tab[t2-c2+255] <= isr2 )
469 {
470 s0 += t0; s1 += t1; s2 += t2;
471 sx += x+1; row_count++;
472 }
473 t0 = ptr[6], t1 = ptr[7], t2 = ptr[8];
474 if( tab[t0-c0+255] + tab[t1-c1+255] + tab[t2-c2+255] <= isr2 )
475 {
476 s0 += t0; s1 += t1; s2 += t2;
477 sx += x+2; row_count++;
478 }
479 t0 = ptr[9], t1 = ptr[10], t2 = ptr[11];
480 if( tab[t0-c0+255] + tab[t1-c1+255] + tab[t2-c2+255] <= isr2 )
481 {
482 s0 += t0; s1 += t1; s2 += t2;
483 sx += x+3; row_count++;
484 }
485 }
486 #endif
487 for( ; x <= maxx; x++, ptr += 3 )
488 {
489 int t0 = ptr[0], t1 = ptr[1], t2 = ptr[2];
490 if( tab[t0-c0+255] + tab[t1-c1+255] + tab[t2-c2+255] <= isr2 )
491 {
492 s0 += t0; s1 += t1; s2 += t2;
493 sx += x; row_count++;
494 }
495 }
496 count += row_count;
497 sy += y*row_count;
498 }
499
500 if( count == 0 )
501 break;
502
503 icount = 1./count;
504 x1 = cvRound(sx*icount);
505 y1 = cvRound(sy*icount);
506 s0 = cvRound(s0*icount);
507 s1 = cvRound(s1*icount);
508 s2 = cvRound(s2*icount);
509
510 stop_flag = (x0 == x1 && y0 == y1) || abs(x1-x0) + abs(y1-y0) +
511 tab[s0 - c0 + 255] + tab[s1 - c1 + 255] +
512 tab[s2 - c2 + 255] <= termcrit.epsilon;
513
514 x0 = x1; y0 = y1;
515 c0 = s0; c1 = s1; c2 = s2;
516
517 if( stop_flag )
518 break;
519 }
520
521 dptr[0] = (uchar)c0;
522 dptr[1] = (uchar)c1;
523 dptr[2] = (uchar)c2;
524 }
525 }
526 }
527}

529void cv::pyrMeanShiftFiltering( InputArray _src, OutputArray _dst,
530 double sp, double sr, int maxLevel,
531 TermCriteria termcrit )
532{
533 Mat src = _src.getMat();
534
535 if( src.empty() )
536 return;
537
538 _dst.create( src.size(), src.type() );
539 CvMat c_src = src, c_dst = _dst.getMat();
540 cvPyrMeanShiftFiltering( &c_src, &c_dst, sp, sr, maxLevel, termcrit );
541}
