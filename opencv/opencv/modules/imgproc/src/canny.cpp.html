54static bool ippCanny(const Mat& _src, Mat& _dst, float low, float high)
55{
56 int size = 0, size1 = 0;
57 IppiSize roi = { _src.cols, _src.rows };
58
59 ippiFilterSobelNegVertGetBufferSize_8u16s_C1R(roi, ippMskSize3x3, &size);
60 ippiFilterSobelHorizGetBufferSize_8u16s_C1R(roi, ippMskSize3x3, &size1);
61 size = std::max(size, size1);
62 ippiCannyGetSize(roi, &size1);
63 size = std::max(size, size1);
64
65 AutoBuffer<uchar> buf(size + 64);
66 uchar* buffer = alignPtr((uchar*)buf, 32);
67
68 Mat _dx(_src.rows, _src.cols, CV_16S);
69 if( ippiFilterSobelNegVertBorder_8u16s_C1R(_src.data, (int)_src.step,
70 _dx.ptr<short>(), (int)_dx.step, roi,
71 ippMskSize3x3, ippBorderRepl, 0, buffer) < 0 )
72 return false;
73
74 Mat _dy(_src.rows, _src.cols, CV_16S);
75 if( ippiFilterSobelHorizBorder_8u16s_C1R(_src.data, (int)_src.step,
76 _dy.ptr<short>(), (int)_dy.step, roi,
77 ippMskSize3x3, ippBorderRepl, 0, buffer) < 0 )
78 return false;
79
80 if( ippiCanny_16s8u_C1R(_dx.ptr<short>(), (int)_dx.step,
81 _dy.ptr<short>(), (int)_dy.step,
82 _dst.data, (int)_dst.step, roi, low, high, buffer) < 0 )
83 return false;
84 return true;
85}
86}

89void cv::Canny( InputArray _src, OutputArray _dst,
90 double low_thresh, double high_thresh,
91 int aperture_size, bool L2gradient )
92{
93 Mat src = _src.getMat();
94 CV_Assert( src.depth() == CV_8U );
95
96 _dst.create(src.size(), CV_8U);
97 Mat dst = _dst.getMat();
98
99 if (!L2gradient && (aperture_size & CV_CANNY_L2_GRADIENT) == CV_CANNY_L2_GRADIENT)
100 {
101 //backward compatibility
102 aperture_size &= ~CV_CANNY_L2_GRADIENT;
103 L2gradient = true;
104 }
105
106 if ((aperture_size & 1) == 0 || (aperture_size != -1 && (aperture_size < 3 || aperture_size > 7)))
107 CV_Error(CV_StsBadFlag, "");
108
109 if (low_thresh > high_thresh)
110 std::swap(low_thresh, high_thresh);
111
112#ifdef HAVE_TEGRA_OPTIMIZATION
113 if (tegra::canny(src, dst, low_thresh, high_thresh, aperture_size, L2gradient))
114 return;
115#endif
116
117#ifdef USE_IPP_CANNY
118 if( aperture_size == 3 && !L2gradient &&
119 ippCanny(src, dst, (float)low_thresh, (float)high_thresh) )
120 return;
121#endif
122
123 const int cn = src.channels();
124 Mat dx(src.rows, src.cols, CV_16SC(cn));
125 Mat dy(src.rows, src.cols, CV_16SC(cn));
126
127 Sobel(src, dx, CV_16S, 1, 0, aperture_size, 1, 0, cv::BORDER_REPLICATE);
128 Sobel(src, dy, CV_16S, 0, 1, aperture_size, 1, 0, cv::BORDER_REPLICATE);
129
130 if (L2gradient)
131 {
132 low_thresh = std::min(32767.0, low_thresh);
133 high_thresh = std::min(32767.0, high_thresh);
134
135 if (low_thresh > 0) low_thresh *= low_thresh;
136 if (high_thresh > 0) high_thresh *= high_thresh;
137 }
138 int low = cvFloor(low_thresh);
139 int high = cvFloor(high_thresh);
140
141 ptrdiff_t mapstep = src.cols + 2;
142 AutoBuffer<uchar> buffer((src.cols+2)*(src.rows+2) + cn * mapstep * 3 * sizeof(int));
143
144 int* mag_buf[3];
145 mag_buf[0] = (int*)(uchar*)buffer;
146 mag_buf[1] = mag_buf[0] + mapstep*cn;
147 mag_buf[2] = mag_buf[1] + mapstep*cn;
148 memset(mag_buf[0], 0, /* cn* */mapstep*sizeof(int));
149
150 uchar* map = (uchar*)(mag_buf[2] + mapstep*cn);
151 memset(map, 1, mapstep);
152 memset(map + mapstep*(src.rows + 1), 1, mapstep);
153
154 int maxsize = std::max(1 << 10, src.cols * src.rows / 10);
155 std::vector<uchar*> stack(maxsize);
156 uchar **stack_top = &stack[0];
157 uchar **stack_bottom = &stack[0];
158
159 /* sector numbers
160 (Top-Left Origin)
161
162 1 2 3
163 * * *
164 * * *
165 0*******0
166 * * *
167 * * *
168 3 2 1
169 */
170
171 #define CANNY_PUSH(d) *(d) = uchar(2), *stack_top++ = (d)
172 #define CANNY_POP(d) (d) = *--stack_top
173
174 // calculate magnitude and angle of gradient, perform non-maxima suppression.
175 // fill the map with one of the following values:
176 // 0 - the pixel might belong to an edge
177 // 1 - the pixel can not belong to an edge
178 // 2 - the pixel does belong to an edge
179 for (int i = 0; i <= src.rows; i++)
180 {
181 int* _norm = mag_buf[(i > 0) + 1] + 1;
182 if (i < src.rows)
183 {
184 short* _dx = dx.ptr<short>(i);
185 short* _dy = dy.ptr<short>(i);
186
187 if (!L2gradient)
188 {
189 for (int j = 0; j < src.cols*cn; j++)
190 _norm[j] = std::abs(int(_dx[j])) + std::abs(int(_dy[j]));
191 }
192 else
193 {
194 for (int j = 0; j < src.cols*cn; j++)
195 _norm[j] = int(_dx[j])*_dx[j] + int(_dy[j])*_dy[j];
196 }
197
198 if (cn > 1)
199 {
200 for(int j = 0, jn = 0; j < src.cols; ++j, jn += cn)
201 {
202 int maxIdx = jn;
203 for(int k = 1; k < cn; ++k)
204 if(_norm[jn + k] > _norm[maxIdx]) maxIdx = jn + k;
205 _norm[j] = _norm[maxIdx];
206 _dx[j] = _dx[maxIdx];
207 _dy[j] = _dy[maxIdx];
208 }
209 }
210 _norm[-1] = _norm[src.cols] = 0;
211 }
212 else
213 memset(_norm-1, 0, /* cn* */mapstep*sizeof(int));
214
215 // at the very beginning we do not have a complete ring
216 // buffer of 3 magnitude rows for non-maxima suppression
217 if (i == 0)
218 continue;
219
220 uchar* _map = map + mapstep*i + 1;
221 _map[-1] = _map[src.cols] = 1;
222
223 int* _mag = mag_buf[1] + 1; // take the central row
224 ptrdiff_t magstep1 = mag_buf[2] - mag_buf[1];
225 ptrdiff_t magstep2 = mag_buf[0] - mag_buf[1];
226
227 const short* _x = dx.ptr<short>(i-1);
228 const short* _y = dy.ptr<short>(i-1);
229
230 if ((stack_top - stack_bottom) + src.cols > maxsize)
231 {
232 int sz = (int)(stack_top - stack_bottom);
233 maxsize = std::max(sz + src.cols, maxsize * 3/2);
234 stack.resize(maxsize);
235 stack_bottom = &stack[0];
236 stack_top = stack_bottom + sz;
237 }
238
239 int prev_flag = 0;
240 for (int j = 0; j < src.cols; j++)
241 {
242 #define CANNY_SHIFT 15
243 const int TG22 = (int)(0.4142135623730950488016887242097*(1<<CANNY_SHIFT) + 0.5);
244
245 int m = _mag[j];
246
247 if (m > low)
248 {
249 int xs = _x[j];
250 int ys = _y[j];
251 int x = std::abs(xs);
252 int y = std::abs(ys) << CANNY_SHIFT;
253
254 int tg22x = x * TG22;
255
256 if (y < tg22x)
257 {
258 if (m > _mag[j-1] && m >= _mag[j+1]) goto __ocv_canny_push;
259 }
260 else
261 {
262 int tg67x = tg22x + (x << (CANNY_SHIFT+1));
263 if (y > tg67x)
264 {
265 if (m > _mag[j+magstep2] && m >= _mag[j+magstep1]) goto __ocv_canny_push;
266 }
267 else
268 {
269 int s = (xs ^ ys) < 0 ? -1 : 1;
270 if (m > _mag[j+magstep2-s] && m > _mag[j+magstep1+s]) goto __ocv_canny_push;
271 }
272 }
273 }
274 prev_flag = 0;
275 _map[j] = uchar(1);
276 continue;
277__ocv_canny_push:
278 if (!prev_flag && m > high && _map[j-mapstep] != 2)
279 {
280 CANNY_PUSH(_map + j);
281 prev_flag = 1;
282 }
283 else
284 _map[j] = 0;
285 }
286
287 // scroll the ring buffer
288 _mag = mag_buf[0];
289 mag_buf[0] = mag_buf[1];
290 mag_buf[1] = mag_buf[2];
291 mag_buf[2] = _mag;
292 }
293
294 // now track the edges (hysteresis thresholding)
295 while (stack_top > stack_bottom)
296 {
297 uchar* m;
298 if ((stack_top - stack_bottom) + 8 > maxsize)
299 {
300 int sz = (int)(stack_top - stack_bottom);
301 maxsize = maxsize * 3/2;
302 stack.resize(maxsize);
303 stack_bottom = &stack[0];
304 stack_top = stack_bottom + sz;
305 }
306
307 CANNY_POP(m);
308
309 if (!m[-1]) CANNY_PUSH(m - 1);
310 if (!m[1]) CANNY_PUSH(m + 1);
311 if (!m[-mapstep-1]) CANNY_PUSH(m - mapstep - 1);
312 if (!m[-mapstep]) CANNY_PUSH(m - mapstep);
313 if (!m[-mapstep+1]) CANNY_PUSH(m - mapstep + 1);
314 if (!m[mapstep-1]) CANNY_PUSH(m + mapstep - 1);
315 if (!m[mapstep]) CANNY_PUSH(m + mapstep);
316 if (!m[mapstep+1]) CANNY_PUSH(m + mapstep + 1);
317 }
318
319 // the final pass, form the final image
320 const uchar* pmap = map + mapstep + 1;
321 uchar* pdst = dst.ptr();
322 for (int i = 0; i < src.rows; i++, pmap += mapstep, pdst += dst.step)
323 {
324 for (int j = 0; j < src.cols; j++)
325 pdst[j] = (uchar)-(pmap[j] >> 1);
326 }
327}

329void cvCanny( const CvArr* image, CvArr* edges, double threshold1,
330 double threshold2, int aperture_size )
331{
332 cv::Mat src = cv::cvarrToMat(image), dst = cv::cvarrToMat(edges);
333 CV_Assert( src.size == dst.size && src.depth() == CV_8U && dst.type() == CV_8U );
334
335 cv::Canny(src, dst, threshold1, threshold2, aperture_size & 255,
336 (aperture_size & CV_CANNY_L2_GRADIENT) != 0);
337}
