82static inline Point normalizeAnchor( Point anchor, Size ksize )
83{
84 if( anchor.x == -1 )
85 anchor.x = ksize.width/2;
86 if( anchor.y == -1 )
87 anchor.y = ksize.height/2;
88 CV_Assert( anchor.inside(Rect(0, 0, ksize.width, ksize.height)) );
89 return anchor;
90}
