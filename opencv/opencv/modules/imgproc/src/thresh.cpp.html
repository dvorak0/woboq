48static void
49thresh_8u( const Mat& _src, Mat& _dst, uchar thresh, uchar maxval, int type )
50{
51 int i, j, j_scalar = 0;
52 uchar tab[256];
53 Size roi = _src.size();
54 roi.width *= _src.channels();
55
56 if( _src.isContinuous() && _dst.isContinuous() )
57 {
58 roi.width *= roi.height;
59 roi.height = 1;
60 }
61
62#ifdef HAVE_TEGRA_OPTIMIZATION
63 if (tegra::thresh_8u(_src, _dst, roi.width, roi.height, thresh, maxval, type))
64 return;
65#endif
66
67 switch( type )
68 {
69 case THRESH_BINARY:
70 for( i = 0; i <= thresh; i++ )
71 tab[i] = 0;
72 for( ; i < 256; i++ )
73 tab[i] = maxval;
74 break;
75 case THRESH_BINARY_INV:
76 for( i = 0; i <= thresh; i++ )
77 tab[i] = maxval;
78 for( ; i < 256; i++ )
79 tab[i] = 0;
80 break;
81 case THRESH_TRUNC:
82 for( i = 0; i <= thresh; i++ )
83 tab[i] = (uchar)i;
84 for( ; i < 256; i++ )
85 tab[i] = thresh;
86 break;
87 case THRESH_TOZERO:
88 for( i = 0; i <= thresh; i++ )
89 tab[i] = 0;
90 for( ; i < 256; i++ )
91 tab[i] = (uchar)i;
92 break;
93 case THRESH_TOZERO_INV:
94 for( i = 0; i <= thresh; i++ )
95 tab[i] = (uchar)i;
96 for( ; i < 256; i++ )
97 tab[i] = 0;
98 break;
99 default:
100 CV_Error( CV_StsBadArg, "Unknown threshold type" );
101 }
102
103#if CV_SSE2
104 if( checkHardwareSupport(CV_CPU_SSE2) )
105 {
106 __m128i _x80 = _mm_set1_epi8('\x80');
107 __m128i thresh_u = _mm_set1_epi8(thresh);
108 __m128i thresh_s = _mm_set1_epi8(thresh ^ 0x80);
109 __m128i maxval_ = _mm_set1_epi8(maxval);
110 j_scalar = roi.width & -8;
111
112 for( i = 0; i < roi.height; i++ )
113 {
114 const uchar* src = (const uchar*)(_src.data + _src.step*i);
115 uchar* dst = (uchar*)(_dst.data + _dst.step*i);
116
117 switch( type )
118 {
119 case THRESH_BINARY:
120 for( j = 0; j <= roi.width - 32; j += 32 )
121 {
122 __m128i v0, v1;
123 v0 = _mm_loadu_si128( (const __m128i*)(src + j) );
124 v1 = _mm_loadu_si128( (const __m128i*)(src + j + 16) );
125 v0 = _mm_cmpgt_epi8( _mm_xor_si128(v0, _x80), thresh_s );
126 v1 = _mm_cmpgt_epi8( _mm_xor_si128(v1, _x80), thresh_s );
127 v0 = _mm_and_si128( v0, maxval_ );
128 v1 = _mm_and_si128( v1, maxval_ );
129 _mm_storeu_si128( (__m128i*)(dst + j), v0 );
130 _mm_storeu_si128( (__m128i*)(dst + j + 16), v1 );
131 }
132
133 for( ; j <= roi.width - 8; j += 8 )
134 {
135 __m128i v0 = _mm_loadl_epi64( (const __m128i*)(src + j) );
136 v0 = _mm_cmpgt_epi8( _mm_xor_si128(v0, _x80), thresh_s );
137 v0 = _mm_and_si128( v0, maxval_ );
138 _mm_storel_epi64( (__m128i*)(dst + j), v0 );
139 }
140 break;
141
142 case THRESH_BINARY_INV:
143 for( j = 0; j <= roi.width - 32; j += 32 )
144 {
145 __m128i v0, v1;
146 v0 = _mm_loadu_si128( (const __m128i*)(src + j) );
147 v1 = _mm_loadu_si128( (const __m128i*)(src + j + 16) );
148 v0 = _mm_cmpgt_epi8( _mm_xor_si128(v0, _x80), thresh_s );
149 v1 = _mm_cmpgt_epi8( _mm_xor_si128(v1, _x80), thresh_s );
150 v0 = _mm_andnot_si128( v0, maxval_ );
151 v1 = _mm_andnot_si128( v1, maxval_ );
152 _mm_storeu_si128( (__m128i*)(dst + j), v0 );
153 _mm_storeu_si128( (__m128i*)(dst + j + 16), v1 );
154 }
155
156 for( ; j <= roi.width - 8; j += 8 )
157 {
158 __m128i v0 = _mm_loadl_epi64( (const __m128i*)(src + j) );
159 v0 = _mm_cmpgt_epi8( _mm_xor_si128(v0, _x80), thresh_s );
160 v0 = _mm_andnot_si128( v0, maxval_ );
161 _mm_storel_epi64( (__m128i*)(dst + j), v0 );
162 }
163 break;
164
165 case THRESH_TRUNC:
166 for( j = 0; j <= roi.width - 32; j += 32 )
167 {
168 __m128i v0, v1;
169 v0 = _mm_loadu_si128( (const __m128i*)(src + j) );
170 v1 = _mm_loadu_si128( (const __m128i*)(src + j + 16) );
171 v0 = _mm_subs_epu8( v0, _mm_subs_epu8( v0, thresh_u ));
172 v1 = _mm_subs_epu8( v1, _mm_subs_epu8( v1, thresh_u ));
173 _mm_storeu_si128( (__m128i*)(dst + j), v0 );
174 _mm_storeu_si128( (__m128i*)(dst + j + 16), v1 );
175 }
176
177 for( ; j <= roi.width - 8; j += 8 )
178 {
179 __m128i v0 = _mm_loadl_epi64( (const __m128i*)(src + j) );
180 v0 = _mm_subs_epu8( v0, _mm_subs_epu8( v0, thresh_u ));
181 _mm_storel_epi64( (__m128i*)(dst + j), v0 );
182 }
183 break;
184
185 case THRESH_TOZERO:
186 for( j = 0; j <= roi.width - 32; j += 32 )
187 {
188 __m128i v0, v1;
189 v0 = _mm_loadu_si128( (const __m128i*)(src + j) );
190 v1 = _mm_loadu_si128( (const __m128i*)(src + j + 16) );
191 v0 = _mm_and_si128( v0, _mm_cmpgt_epi8(_mm_xor_si128(v0, _x80), thresh_s ));
192 v1 = _mm_and_si128( v1, _mm_cmpgt_epi8(_mm_xor_si128(v1, _x80), thresh_s ));
193 _mm_storeu_si128( (__m128i*)(dst + j), v0 );
194 _mm_storeu_si128( (__m128i*)(dst + j + 16), v1 );
195 }
196
197 for( ; j <= roi.width - 8; j += 8 )
198 {
199 __m128i v0 = _mm_loadl_epi64( (const __m128i*)(src + j) );
200 v0 = _mm_and_si128( v0, _mm_cmpgt_epi8(_mm_xor_si128(v0, _x80), thresh_s ));
201 _mm_storel_epi64( (__m128i*)(dst + j), v0 );
202 }
203 break;
204
205 case THRESH_TOZERO_INV:
206 for( j = 0; j <= roi.width - 32; j += 32 )
207 {
208 __m128i v0, v1;
209 v0 = _mm_loadu_si128( (const __m128i*)(src + j) );
210 v1 = _mm_loadu_si128( (const __m128i*)(src + j + 16) );
211 v0 = _mm_andnot_si128( _mm_cmpgt_epi8(_mm_xor_si128(v0, _x80), thresh_s ), v0 );
212 v1 = _mm_andnot_si128( _mm_cmpgt_epi8(_mm_xor_si128(v1, _x80), thresh_s ), v1 );
213 _mm_storeu_si128( (__m128i*)(dst + j), v0 );
214 _mm_storeu_si128( (__m128i*)(dst + j + 16), v1 );
215 }
216
217 for( ; j <= roi.width - 8; j += 8 )
218 {
219 __m128i v0 = _mm_loadl_epi64( (const __m128i*)(src + j) );
220 v0 = _mm_andnot_si128( _mm_cmpgt_epi8(_mm_xor_si128(v0, _x80), thresh_s ), v0 );
221 _mm_storel_epi64( (__m128i*)(dst + j), v0 );
222 }
223 break;
224 }
225 }
226 }
227#endif
228
229 if( j_scalar < roi.width )
230 {
231 for( i = 0; i < roi.height; i++ )
232 {
233 const uchar* src = (const uchar*)(_src.data + _src.step*i);
234 uchar* dst = (uchar*)(_dst.data + _dst.step*i);
235 j = j_scalar;
236#if CV_ENABLE_UNROLLED
237 for( ; j <= roi.width - 4; j += 4 )
238 {
239 uchar t0 = tab[src[j]];
240 uchar t1 = tab[src[j+1]];
241
242 dst[j] = t0;
243 dst[j+1] = t1;
244
245 t0 = tab[src[j+2]];
246 t1 = tab[src[j+3]];
247
248 dst[j+2] = t0;
249 dst[j+3] = t1;
250 }
251#endif
252 for( ; j < roi.width; j++ )
253 dst[j] = tab[src[j]];
254 }
255 }
256}

259static void
260thresh_16s( const Mat& _src, Mat& _dst, short thresh, short maxval, int type )
261{
262 int i, j;
263 Size roi = _src.size();
264 roi.width *= _src.channels();
265 const short* src = (const short*)_src.data;
266 short* dst = (short*)_dst.data;
267 size_t src_step = _src.step/sizeof(src[0]);
268 size_t dst_step = _dst.step/sizeof(dst[0]);
269
270#if CV_SSE2
271 volatile bool useSIMD = checkHardwareSupport(CV_CPU_SSE);
272#endif
273
274 if( _src.isContinuous() && _dst.isContinuous() )
275 {
276 roi.width *= roi.height;
277 roi.height = 1;
278 }
279
280#ifdef HAVE_TEGRA_OPTIMIZATION
281 if (tegra::thresh_16s(_src, _dst, roi.width, roi.height, thresh, maxval, type))
282 return;
283#endif
284
285 switch( type )
286 {
287 case THRESH_BINARY:
288 for( i = 0; i < roi.height; i++, src += src_step, dst += dst_step )
289 {
290 j = 0;
291 #if CV_SSE2
292 if( useSIMD )
293 {
294 __m128i thresh8 = _mm_set1_epi16(thresh), maxval8 = _mm_set1_epi16(maxval);
295 for( ; j <= roi.width - 16; j += 16 )
296 {
297 __m128i v0, v1;
298 v0 = _mm_loadu_si128( (const __m128i*)(src + j) );
299 v1 = _mm_loadu_si128( (const __m128i*)(src + j + 8) );
300 v0 = _mm_cmpgt_epi16( v0, thresh8 );
301 v1 = _mm_cmpgt_epi16( v1, thresh8 );
302 v0 = _mm_and_si128( v0, maxval8 );
303 v1 = _mm_and_si128( v1, maxval8 );
304 _mm_storeu_si128((__m128i*)(dst + j), v0 );
305 _mm_storeu_si128((__m128i*)(dst + j + 8), v1 );
306 }
307 }
308 #endif
309
310 for( ; j < roi.width; j++ )
311 dst[j] = src[j] > thresh ? maxval : 0;
312 }
313 break;
314
315 case THRESH_BINARY_INV:
316 for( i = 0; i < roi.height; i++, src += src_step, dst += dst_step )
317 {
318 j = 0;
319 #if CV_SSE2
320 if( useSIMD )
321 {
322 __m128i thresh8 = _mm_set1_epi16(thresh), maxval8 = _mm_set1_epi16(maxval);
323 for( ; j <= roi.width - 16; j += 16 )
324 {
325 __m128i v0, v1;
326 v0 = _mm_loadu_si128( (const __m128i*)(src + j) );
327 v1 = _mm_loadu_si128( (const __m128i*)(src + j + 8) );
328 v0 = _mm_cmpgt_epi16( v0, thresh8 );
329 v1 = _mm_cmpgt_epi16( v1, thresh8 );
330 v0 = _mm_andnot_si128( v0, maxval8 );
331 v1 = _mm_andnot_si128( v1, maxval8 );
332 _mm_storeu_si128((__m128i*)(dst + j), v0 );
333 _mm_storeu_si128((__m128i*)(dst + j + 8), v1 );
334 }
335 }
336 #endif
337
338 for( ; j < roi.width; j++ )
339 dst[j] = src[j] <= thresh ? maxval : 0;
340 }
341 break;
342
343 case THRESH_TRUNC:
344 for( i = 0; i < roi.height; i++, src += src_step, dst += dst_step )
345 {
346 j = 0;
347 #if CV_SSE2
348 if( useSIMD )
349 {
350 __m128i thresh8 = _mm_set1_epi16(thresh);
351 for( ; j <= roi.width - 16; j += 16 )
352 {
353 __m128i v0, v1;
354 v0 = _mm_loadu_si128( (const __m128i*)(src + j) );
355 v1 = _mm_loadu_si128( (const __m128i*)(src + j + 8) );
356 v0 = _mm_min_epi16( v0, thresh8 );
357 v1 = _mm_min_epi16( v1, thresh8 );
358 _mm_storeu_si128((__m128i*)(dst + j), v0 );
359 _mm_storeu_si128((__m128i*)(dst + j + 8), v1 );
360 }
361 }
362 #endif
363
364 for( ; j < roi.width; j++ )
365 dst[j] = std::min(src[j], thresh);
366 }
367 break;
368
369 case THRESH_TOZERO:
370 for( i = 0; i < roi.height; i++, src += src_step, dst += dst_step )
371 {
372 j = 0;
373 #if CV_SSE2
374 if( useSIMD )
375 {
376 __m128i thresh8 = _mm_set1_epi16(thresh);
377 for( ; j <= roi.width - 16; j += 16 )
378 {
379 __m128i v0, v1;
380 v0 = _mm_loadu_si128( (const __m128i*)(src + j) );
381 v1 = _mm_loadu_si128( (const __m128i*)(src + j + 8) );
382 v0 = _mm_and_si128(v0, _mm_cmpgt_epi16(v0, thresh8));
383 v1 = _mm_and_si128(v1, _mm_cmpgt_epi16(v1, thresh8));
384 _mm_storeu_si128((__m128i*)(dst + j), v0 );
385 _mm_storeu_si128((__m128i*)(dst + j + 8), v1 );
386 }
387 }
388 #endif
389
390 for( ; j < roi.width; j++ )
391 {
392 short v = src[j];
393 dst[j] = v > thresh ? v : 0;
394 }
395 }
396 break;
397
398 case THRESH_TOZERO_INV:
399 for( i = 0; i < roi.height; i++, src += src_step, dst += dst_step )
400 {
401 j = 0;
402 #if CV_SSE2
403 if( useSIMD )
404 {
405 __m128i thresh8 = _mm_set1_epi16(thresh);
406 for( ; j <= roi.width - 16; j += 16 )
407 {
408 __m128i v0, v1;
409 v0 = _mm_loadu_si128( (const __m128i*)(src + j) );
410 v1 = _mm_loadu_si128( (const __m128i*)(src + j + 8) );
411 v0 = _mm_andnot_si128(_mm_cmpgt_epi16(v0, thresh8), v0);
412 v1 = _mm_andnot_si128(_mm_cmpgt_epi16(v1, thresh8), v1);
413 _mm_storeu_si128((__m128i*)(dst + j), v0 );
414 _mm_storeu_si128((__m128i*)(dst + j + 8), v1 );
415 }
416 }
417 #endif
418 for( ; j < roi.width; j++ )
419 {
420 short v = src[j];
421 dst[j] = v <= thresh ? v : 0;
422 }
423 }
424 break;
425 default:
426 return CV_Error( CV_StsBadArg, "" );
427 }
428}

431static void
432thresh_32f( const Mat& _src, Mat& _dst, float thresh, float maxval, int type )
433{
434 int i, j;
435 Size roi = _src.size();
436 roi.width *= _src.channels();
437 const float* src = (const float*)_src.data;
438 float* dst = (float*)_dst.data;
439 size_t src_step = _src.step/sizeof(src[0]);
440 size_t dst_step = _dst.step/sizeof(dst[0]);
441
442#if CV_SSE2
443 volatile bool useSIMD = checkHardwareSupport(CV_CPU_SSE);
444#endif
445
446 if( _src.isContinuous() && _dst.isContinuous() )
447 {
448 roi.width *= roi.height;
449 roi.height = 1;
450 }
451
452#ifdef HAVE_TEGRA_OPTIMIZATION
453 if (tegra::thresh_32f(_src, _dst, roi.width, roi.height, thresh, maxval, type))
454 return;
455#endif
456
457 switch( type )
458 {
459 case THRESH_BINARY:
460 for( i = 0; i < roi.height; i++, src += src_step, dst += dst_step )
461 {
462 j = 0;
463#if CV_SSE2
464 if( useSIMD )
465 {
466 __m128 thresh4 = _mm_set1_ps(thresh), maxval4 = _mm_set1_ps(maxval);
467 for( ; j <= roi.width - 8; j += 8 )
468 {
469 __m128 v0, v1;
470 v0 = _mm_loadu_ps( src + j );
471 v1 = _mm_loadu_ps( src + j + 4 );
472 v0 = _mm_cmpgt_ps( v0, thresh4 );
473 v1 = _mm_cmpgt_ps( v1, thresh4 );
474 v0 = _mm_and_ps( v0, maxval4 );
475 v1 = _mm_and_ps( v1, maxval4 );
476 _mm_storeu_ps( dst + j, v0 );
477 _mm_storeu_ps( dst + j + 4, v1 );
478 }
479 }
480#endif
481
482 for( ; j < roi.width; j++ )
483 dst[j] = src[j] > thresh ? maxval : 0;
484 }
485 break;
486
487 case THRESH_BINARY_INV:
488 for( i = 0; i < roi.height; i++, src += src_step, dst += dst_step )
489 {
490 j = 0;
491#if CV_SSE2
492 if( useSIMD )
493 {
494 __m128 thresh4 = _mm_set1_ps(thresh), maxval4 = _mm_set1_ps(maxval);
495 for( ; j <= roi.width - 8; j += 8 )
496 {
497 __m128 v0, v1;
498 v0 = _mm_loadu_ps( src + j );
499 v1 = _mm_loadu_ps( src + j + 4 );
500 v0 = _mm_cmple_ps( v0, thresh4 );
501 v1 = _mm_cmple_ps( v1, thresh4 );
502 v0 = _mm_and_ps( v0, maxval4 );
503 v1 = _mm_and_ps( v1, maxval4 );
504 _mm_storeu_ps( dst + j, v0 );
505 _mm_storeu_ps( dst + j + 4, v1 );
506 }
507 }
508#endif
509
510 for( ; j < roi.width; j++ )
511 dst[j] = src[j] <= thresh ? maxval : 0;
512 }
513 break;
514
515 case THRESH_TRUNC:
516 for( i = 0; i < roi.height; i++, src += src_step, dst += dst_step )
517 {
518 j = 0;
519#if CV_SSE2
520 if( useSIMD )
521 {
522 __m128 thresh4 = _mm_set1_ps(thresh);
523 for( ; j <= roi.width - 8; j += 8 )
524 {
525 __m128 v0, v1;
526 v0 = _mm_loadu_ps( src + j );
527 v1 = _mm_loadu_ps( src + j + 4 );
528 v0 = _mm_min_ps( v0, thresh4 );
529 v1 = _mm_min_ps( v1, thresh4 );
530 _mm_storeu_ps( dst + j, v0 );
531 _mm_storeu_ps( dst + j + 4, v1 );
532 }
533 }
534#endif
535
536 for( ; j < roi.width; j++ )
537 dst[j] = std::min(src[j], thresh);
538 }
539 break;
540
541 case THRESH_TOZERO:
542 for( i = 0; i < roi.height; i++, src += src_step, dst += dst_step )
543 {
544 j = 0;
545#if CV_SSE2
546 if( useSIMD )
547 {
548 __m128 thresh4 = _mm_set1_ps(thresh);
549 for( ; j <= roi.width - 8; j += 8 )
550 {
551 __m128 v0, v1;
552 v0 = _mm_loadu_ps( src + j );
553 v1 = _mm_loadu_ps( src + j + 4 );
554 v0 = _mm_and_ps(v0, _mm_cmpgt_ps(v0, thresh4));
555 v1 = _mm_and_ps(v1, _mm_cmpgt_ps(v1, thresh4));
556 _mm_storeu_ps( dst + j, v0 );
557 _mm_storeu_ps( dst + j + 4, v1 );
558 }
559 }
560#endif
561
562 for( ; j < roi.width; j++ )
563 {
564 float v = src[j];
565 dst[j] = v > thresh ? v : 0;
566 }
567 }
568 break;
569
570 case THRESH_TOZERO_INV:
571 for( i = 0; i < roi.height; i++, src += src_step, dst += dst_step )
572 {
573 j = 0;
574#if CV_SSE2
575 if( useSIMD )
576 {
577 __m128 thresh4 = _mm_set1_ps(thresh);
578 for( ; j <= roi.width - 8; j += 8 )
579 {
580 __m128 v0, v1;
581 v0 = _mm_loadu_ps( src + j );
582 v1 = _mm_loadu_ps( src + j + 4 );
583 v0 = _mm_and_ps(v0, _mm_cmple_ps(v0, thresh4));
584 v1 = _mm_and_ps(v1, _mm_cmple_ps(v1, thresh4));
585 _mm_storeu_ps( dst + j, v0 );
586 _mm_storeu_ps( dst + j + 4, v1 );
587 }
588 }
589#endif
590 for( ; j < roi.width; j++ )
591 {
592 float v = src[j];
593 dst[j] = v <= thresh ? v : 0;
594 }
595 }
596 break;
597 default:
598 return CV_Error( CV_StsBadArg, "" );
599 }
600}

602
603static double
604getThreshVal_Otsu_8u( const Mat& _src )
605{
606 Size size = _src.size();
607 if( _src.isContinuous() )
608 {
609 size.width *= size.height;
610 size.height = 1;
611 }
612 const int N = 256;
613 int i, j, h[N] = {0};
614 for( i = 0; i < size.height; i++ )
615 {
616 const uchar* src = _src.data + _src.step*i;
617 j = 0;
618 #if CV_ENABLE_UNROLLED
619 for( ; j <= size.width - 4; j += 4 )
620 {
621 int v0 = src[j], v1 = src[j+1];
622 h[v0]++; h[v1]++;
623 v0 = src[j+2]; v1 = src[j+3];
624 h[v0]++; h[v1]++;
625 }
626 #endif
627 for( ; j < size.width; j++ )
628 h[src[j]]++;
629 }
630
631 double mu = 0, scale = 1./(size.width*size.height);
632 for( i = 0; i < N; i++ )
633 mu += i*(double)h[i];
634
635 mu *= scale;
636 double mu1 = 0, q1 = 0;
637 double max_sigma = 0, max_val = 0;
638
639 for( i = 0; i < N; i++ )
640 {
641 double p_i, q2, mu2, sigma;
642
643 p_i = h[i]*scale;
644 mu1 *= q1;
645 q1 += p_i;
646 q2 = 1. - q1;
647
648 if( std::min(q1,q2) < FLT_EPSILON || std::max(q1,q2) > 1. - FLT_EPSILON )
649 continue;
650
651 mu1 = (mu1 + i*p_i)/q1;
652 mu2 = (mu - q1*mu1)/q2;
653 sigma = q1*q2*(mu1 - mu2)*(mu1 - mu2);
654 if( sigma > max_sigma )
655 {
656 max_sigma = sigma;
657 max_val = i;
658 }
659 }
660
661 return max_val;
662}

664class ThresholdRunner : public ParallelLoopBody
665{
666public:
667 ThresholdRunner(Mat _src, Mat _dst, double _thresh, double _maxval, int _thresholdType)
668 {
669 src = _src;
670 dst = _dst;
671
672 thresh = _thresh;
673 maxval = _maxval;
674 thresholdType = _thresholdType;
675 }
676
677 void operator () ( const Range& range ) const
678 {
679 int row0 = range.start;
680 int row1 = range.end;
681
682 Mat srcStripe = src.rowRange(row0, row1);
683 Mat dstStripe = dst.rowRange(row0, row1);
684
685 if (srcStripe.depth() == CV_8U)
686 {
687 thresh_8u( srcStripe, dstStripe, (uchar)thresh, (uchar)maxval, thresholdType );
688 }
689 else if( srcStripe.depth() == CV_16S )
690 {
691 thresh_16s( srcStripe, dstStripe, (short)thresh, (short)maxval, thresholdType );
692 }
693 else if( srcStripe.depth() == CV_32F )
694 {
695 thresh_32f( srcStripe, dstStripe, (float)thresh, (float)maxval, thresholdType );
696 }
697 }
698
699private:
700 Mat src;
701 Mat dst;
702
703 double thresh;
704 double maxval;
705 int thresholdType;
706};

710double cv::threshold( InputArray _src, OutputArray _dst, double thresh, double maxval, int type )
711{
712 Mat src = _src.getMat();
713 bool use_otsu = (type & THRESH_OTSU) != 0;
714 type &= THRESH_MASK;
715
716 if( use_otsu )
717 {
718 CV_Assert( src.type() == CV_8UC1 );
719 thresh = getThreshVal_Otsu_8u(src);
720 }
721
722 _dst.create( src.size(), src.type() );
723 Mat dst = _dst.getMat();
724
725 if( src.depth() == CV_8U )
726 {
727 int ithresh = cvFloor(thresh);
728 thresh = ithresh;
729 int imaxval = cvRound(maxval);
730 if( type == THRESH_TRUNC )
731 imaxval = ithresh;
732 imaxval = saturate_cast<uchar>(imaxval);
733
734 if( ithresh < 0 || ithresh >= 255 )
735 {
736 if( type == THRESH_BINARY || type == THRESH_BINARY_INV ||
737 ((type == THRESH_TRUNC || type == THRESH_TOZERO_INV) && ithresh < 0) ||
738 (type == THRESH_TOZERO && ithresh >= 255) )
739 {
740 int v = type == THRESH_BINARY ? (ithresh >= 255 ? 0 : imaxval) :
741 type == THRESH_BINARY_INV ? (ithresh >= 255 ? imaxval : 0) :
742 /*type == THRESH_TRUNC ? imaxval :*/ 0;
743 dst.setTo(v);
744 }
745 else
746 src.copyTo(dst);
747 return thresh;
748 }
749 thresh = ithresh;
750 maxval = imaxval;
751 }
752 else if( src.depth() == CV_16S )
753 {
754 int ithresh = cvFloor(thresh);
755 thresh = ithresh;
756 int imaxval = cvRound(maxval);
757 if( type == THRESH_TRUNC )
758 imaxval = ithresh;
759 imaxval = saturate_cast<short>(imaxval);
760
761 if( ithresh < SHRT_MIN || ithresh >= SHRT_MAX )
762 {
763 if( type == THRESH_BINARY || type == THRESH_BINARY_INV ||
764 ((type == THRESH_TRUNC || type == THRESH_TOZERO_INV) && ithresh < SHRT_MIN) ||
765 (type == THRESH_TOZERO && ithresh >= SHRT_MAX) )
766 {
767 int v = type == THRESH_BINARY ? (ithresh >= SHRT_MAX ? 0 : imaxval) :
768 type == THRESH_BINARY_INV ? (ithresh >= SHRT_MAX ? imaxval : 0) :
769 /*type == THRESH_TRUNC ? imaxval :*/ 0;
770 dst.setTo(v);
771 }
772 else
773 src.copyTo(dst);
774 return thresh;
775 }
776 thresh = ithresh;
777 maxval = imaxval;
778 }
779 else if( src.depth() == CV_32F )
780 ;
781 else
782 CV_Error( CV_StsUnsupportedFormat, "" );
783
784 parallel_for_(Range(0, dst.rows),
785 ThresholdRunner(src, dst, thresh, maxval, type),
786 dst.total()/(double)(1<<16));
787 return thresh;
788}

791void cv::adaptiveThreshold( InputArray _src, OutputArray _dst, double maxValue,
792 int method, int type, int blockSize, double delta )
793{
794 Mat src = _src.getMat();
795 CV_Assert( src.type() == CV_8UC1 );
796 CV_Assert( blockSize % 2 == 1 && blockSize > 1 );
797 Size size = src.size();
798
799 _dst.create( size, src.type() );
800 Mat dst = _dst.getMat();
801
802 if( maxValue < 0 )
803 {
804 dst = Scalar(0);
805 return;
806 }
807
808 Mat mean;
809
810 if( src.data != dst.data )
811 mean = dst;
812
813 if( method == ADAPTIVE_THRESH_MEAN_C )
814 boxFilter( src, mean, src.type(), Size(blockSize, blockSize),
815 Point(-1,-1), true, BORDER_REPLICATE );
816 else if( method == ADAPTIVE_THRESH_GAUSSIAN_C )
817 GaussianBlur( src, mean, Size(blockSize, blockSize), 0, 0, BORDER_REPLICATE );
818 else
819 CV_Error( CV_StsBadFlag, "Unknown/unsupported adaptive threshold method" );
820
821 int i, j;
822 uchar imaxval = saturate_cast<uchar>(maxValue);
823 int idelta = type == THRESH_BINARY ? cvCeil(delta) : cvFloor(delta);
824 uchar tab[768];
825
826 if( type == CV_THRESH_BINARY )
827 for( i = 0; i < 768; i++ )
828 tab[i] = (uchar)(i - 255 > -idelta ? imaxval : 0);
829 else if( type == CV_THRESH_BINARY_INV )
830 for( i = 0; i < 768; i++ )
831 tab[i] = (uchar)(i - 255 <= -idelta ? imaxval : 0);
832 else
833 CV_Error( CV_StsBadFlag, "Unknown/unsupported threshold type" );
834
835 if( src.isContinuous() && mean.isContinuous() && dst.isContinuous() )
836 {
837 size.width *= size.height;
838 size.height = 1;
839 }
840
841 for( i = 0; i < size.height; i++ )
842 {
843 const uchar* sdata = src.data + src.step*i;
844 const uchar* mdata = mean.data + mean.step*i;
845 uchar* ddata = dst.data + dst.step*i;
846
847 for( j = 0; j < size.width; j++ )
848 ddata[j] = tab[sdata[j] - mdata[j] + 255];
849 }
850}

852CV_IMPL double
853cvThreshold( const void* srcarr, void* dstarr, double thresh, double maxval, int type )
854{
855 cv::Mat src = cv::cvarrToMat(srcarr), dst = cv::cvarrToMat(dstarr), dst0 = dst;
856
857 CV_Assert( src.size == dst.size && src.channels() == dst.channels() &&
858 (src.depth() == dst.depth() || dst.depth() == CV_8U));
859
860 thresh = cv::threshold( src, dst, thresh, maxval, type );
861 if( dst0.data != dst.data )
862 dst.convertTo( dst0, dst0.depth() );
863 return thresh;
864}
