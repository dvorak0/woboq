48template<typename T, int shift> struct FixPtCast
49{
50 typedef int type1;
51 typedef T rtype;
52 rtype operator ()(type1 arg) const { return (T)((arg + (1 << (shift-1))) >> shift); }
53};
54
55template<typename T, int shift> struct FltCast
56{
57 typedef T type1;
58 typedef T rtype;
59 rtype operator ()(type1 arg) const { return arg*(T)(1./(1 << shift)); }
60};

69struct PyrDownVec_32s8u
70{
71 int operator()(int** src, uchar* dst, int, int width) const
72 {
73 if( !checkHardwareSupport(CV_CPU_SSE2) )
74 return 0;
75
76 int x = 0;
77 const int *row0 = src[0], *row1 = src[1], *row2 = src[2], *row3 = src[3], *row4 = src[4];
78 __m128i delta = _mm_set1_epi16(128);
79
80 for( ; x <= width - 16; x += 16 )
81 {
82 __m128i r0, r1, r2, r3, r4, t0, t1;
83 r0 = _mm_packs_epi32(_mm_load_si128((const __m128i*)(row0 + x)),
84 _mm_load_si128((const __m128i*)(row0 + x + 4)));
85 r1 = _mm_packs_epi32(_mm_load_si128((const __m128i*)(row1 + x)),
86 _mm_load_si128((const __m128i*)(row1 + x + 4)));
87 r2 = _mm_packs_epi32(_mm_load_si128((const __m128i*)(row2 + x)),
88 _mm_load_si128((const __m128i*)(row2 + x + 4)));
89 r3 = _mm_packs_epi32(_mm_load_si128((const __m128i*)(row3 + x)),
90 _mm_load_si128((const __m128i*)(row3 + x + 4)));
91 r4 = _mm_packs_epi32(_mm_load_si128((const __m128i*)(row4 + x)),
92 _mm_load_si128((const __m128i*)(row4 + x + 4)));
93 r0 = _mm_add_epi16(r0, r4);
94 r1 = _mm_add_epi16(_mm_add_epi16(r1, r3), r2);
95 r0 = _mm_add_epi16(r0, _mm_add_epi16(r2, r2));
96 t0 = _mm_add_epi16(r0, _mm_slli_epi16(r1, 2));
97 r0 = _mm_packs_epi32(_mm_load_si128((const __m128i*)(row0 + x + 8)),
98 _mm_load_si128((const __m128i*)(row0 + x + 12)));
99 r1 = _mm_packs_epi32(_mm_load_si128((const __m128i*)(row1 + x + 8)),
100 _mm_load_si128((const __m128i*)(row1 + x + 12)));
101 r2 = _mm_packs_epi32(_mm_load_si128((const __m128i*)(row2 + x + 8)),
102 _mm_load_si128((const __m128i*)(row2 + x + 12)));
103 r3 = _mm_packs_epi32(_mm_load_si128((const __m128i*)(row3 + x + 8)),
104 _mm_load_si128((const __m128i*)(row3 + x + 12)));
105 r4 = _mm_packs_epi32(_mm_load_si128((const __m128i*)(row4 + x + 8)),
106 _mm_load_si128((const __m128i*)(row4 + x + 12)));
107 r0 = _mm_add_epi16(r0, r4);
108 r1 = _mm_add_epi16(_mm_add_epi16(r1, r3), r2);
109 r0 = _mm_add_epi16(r0, _mm_add_epi16(r2, r2));
110 t1 = _mm_add_epi16(r0, _mm_slli_epi16(r1, 2));
111 t0 = _mm_srli_epi16(_mm_add_epi16(t0, delta), 8);
112 t1 = _mm_srli_epi16(_mm_add_epi16(t1, delta), 8);
113 _mm_storeu_si128((__m128i*)(dst + x), _mm_packus_epi16(t0, t1));
114 }
115
116 for( ; x <= width - 4; x += 4 )
117 {
118 __m128i r0, r1, r2, r3, r4, z = _mm_setzero_si128();
119 r0 = _mm_packs_epi32(_mm_load_si128((const __m128i*)(row0 + x)), z);
120 r1 = _mm_packs_epi32(_mm_load_si128((const __m128i*)(row1 + x)), z);
121 r2 = _mm_packs_epi32(_mm_load_si128((const __m128i*)(row2 + x)), z);
122 r3 = _mm_packs_epi32(_mm_load_si128((const __m128i*)(row3 + x)), z);
123 r4 = _mm_packs_epi32(_mm_load_si128((const __m128i*)(row4 + x)), z);
124 r0 = _mm_add_epi16(r0, r4);
125 r1 = _mm_add_epi16(_mm_add_epi16(r1, r3), r2);
126 r0 = _mm_add_epi16(r0, _mm_add_epi16(r2, r2));
127 r0 = _mm_add_epi16(r0, _mm_slli_epi16(r1, 2));
128 r0 = _mm_srli_epi16(_mm_add_epi16(r0, delta), 8);
129 *(int*)(dst + x) = _mm_cvtsi128_si32(_mm_packus_epi16(r0, r0));
130 }
131
132 return x;
133 }
134};

136struct PyrDownVec_32f
137{
138 int operator()(float** src, float* dst, int, int width) const
139 {
140 if( !checkHardwareSupport(CV_CPU_SSE) )
141 return 0;
142
143 int x = 0;
144 const float *row0 = src[0], *row1 = src[1], *row2 = src[2], *row3 = src[3], *row4 = src[4];
145 __m128 _4 = _mm_set1_ps(4.f), _scale = _mm_set1_ps(1.f/256);
146 for( ; x <= width - 8; x += 8 )
147 {
148 __m128 r0, r1, r2, r3, r4, t0, t1;
149 r0 = _mm_load_ps(row0 + x);
150 r1 = _mm_load_ps(row1 + x);
151 r2 = _mm_load_ps(row2 + x);
152 r3 = _mm_load_ps(row3 + x);
153 r4 = _mm_load_ps(row4 + x);
154 r0 = _mm_add_ps(r0, r4);
155 r1 = _mm_add_ps(_mm_add_ps(r1, r3), r2);
156 r0 = _mm_add_ps(r0, _mm_add_ps(r2, r2));
157 t0 = _mm_add_ps(r0, _mm_mul_ps(r1, _4));
158
159 r0 = _mm_load_ps(row0 + x + 4);
160 r1 = _mm_load_ps(row1 + x + 4);
161 r2 = _mm_load_ps(row2 + x + 4);
162 r3 = _mm_load_ps(row3 + x + 4);
163 r4 = _mm_load_ps(row4 + x + 4);
164 r0 = _mm_add_ps(r0, r4);
165 r1 = _mm_add_ps(_mm_add_ps(r1, r3), r2);
166 r0 = _mm_add_ps(r0, _mm_add_ps(r2, r2));
167 t1 = _mm_add_ps(r0, _mm_mul_ps(r1, _4));
168
169 t0 = _mm_mul_ps(t0, _scale);
170 t1 = _mm_mul_ps(t1, _scale);
171
172 _mm_storeu_ps(dst + x, t0);
173 _mm_storeu_ps(dst + x + 4, t1);
174 }
175
176 return x;
177 }
178};

187template<class CastOp, class VecOp> void
188pyrDown_( const Mat& _src, Mat& _dst, int borderType )
189{
190 const int PD_SZ = 5;
191 typedef typename CastOp::type1 WT;
192 typedef typename CastOp::rtype T;
193
194 CV_Assert( !_src.empty() );
195 Size ssize = _src.size(), dsize = _dst.size();
196 int cn = _src.channels();
197 int bufstep = (int)alignSize(dsize.width*cn, 16);
198 AutoBuffer<WT> _buf(bufstep*PD_SZ + 16);
199 WT* buf = alignPtr((WT*)_buf, 16);
200 int tabL[CV_CN_MAX*(PD_SZ+2)], tabR[CV_CN_MAX*(PD_SZ+2)];
201 AutoBuffer<int> _tabM(dsize.width*cn);
202 int* tabM = _tabM;
203 WT* rows[PD_SZ];
204 CastOp castOp;
205 VecOp vecOp;
206
207 CV_Assert( std::abs(dsize.width*2 - ssize.width) <= 2 &&
208 std::abs(dsize.height*2 - ssize.height) <= 2 );
209 int k, x, sy0 = -PD_SZ/2, sy = sy0, width0 = std::min((ssize.width-PD_SZ/2-1)/2 + 1, dsize.width);
210
211 for( x = 0; x <= PD_SZ+1; x++ )
212 {
213 int sx0 = borderInterpolate(x - PD_SZ/2, ssize.width, borderType)*cn;
214 int sx1 = borderInterpolate(x + width0*2 - PD_SZ/2, ssize.width, borderType)*cn;
215 for( k = 0; k < cn; k++ )
216 {
217 tabL[x*cn + k] = sx0 + k;
218 tabR[x*cn + k] = sx1 + k;
219 }
220 }
221
222 ssize.width *= cn;
223 dsize.width *= cn;
224 width0 *= cn;
225
226 for( x = 0; x < dsize.width; x++ )
227 tabM[x] = (x/cn)*2*cn + x % cn;
228
229 for( int y = 0; y < dsize.height; y++ )
230 {
231 T* dst = (T*)(_dst.data + _dst.step*y);
232 WT *row0, *row1, *row2, *row3, *row4;
233
234 // fill the ring buffer (horizontal convolution and decimation)
235 for( ; sy <= y*2 + 2; sy++ )
236 {
237 WT* row = buf + ((sy - sy0) % PD_SZ)*bufstep;
238 int _sy = borderInterpolate(sy, ssize.height, borderType);
239 const T* src = (const T*)(_src.data + _src.step*_sy);
240 int limit = cn;
241 const int* tab = tabL;
242
243 for( x = 0;;)
244 {
245 for( ; x < limit; x++ )
246 {
247 row[x] = src[tab[x+cn*2]]*6 + (src[tab[x+cn]] + src[tab[x+cn*3]])*4 +
248 src[tab[x]] + src[tab[x+cn*4]];
249 }
250
251 if( x == dsize.width )
252 break;
253
254 if( cn == 1 )
255 {
256 for( ; x < width0; x++ )
257 row[x] = src[x*2]*6 + (src[x*2 - 1] + src[x*2 + 1])*4 +
258 src[x*2 - 2] + src[x*2 + 2];
259 }
260 else if( cn == 3 )
261 {
262 for( ; x < width0; x += 3 )
263 {
264 const T* s = src + x*2;
265 WT t0 = s[0]*6 + (s[-3] + s[3])*4 + s[-6] + s[6];
266 WT t1 = s[1]*6 + (s[-2] + s[4])*4 + s[-5] + s[7];
267 WT t2 = s[2]*6 + (s[-1] + s[5])*4 + s[-4] + s[8];
268 row[x] = t0; row[x+1] = t1; row[x+2] = t2;
269 }
270 }
271 else if( cn == 4 )
272 {
273 for( ; x < width0; x += 4 )
274 {
275 const T* s = src + x*2;
276 WT t0 = s[0]*6 + (s[-4] + s[4])*4 + s[-8] + s[8];
277 WT t1 = s[1]*6 + (s[-3] + s[5])*4 + s[-7] + s[9];
278 row[x] = t0; row[x+1] = t1;
279 t0 = s[2]*6 + (s[-2] + s[6])*4 + s[-6] + s[10];
280 t1 = s[3]*6 + (s[-1] + s[7])*4 + s[-5] + s[11];
281 row[x+2] = t0; row[x+3] = t1;
282 }
283 }
284 else
285 {
286 for( ; x < width0; x++ )
287 {
288 int sx = tabM[x];
289 row[x] = src[sx]*6 + (src[sx - cn] + src[sx + cn])*4 +
290 src[sx - cn*2] + src[sx + cn*2];
291 }
292 }
293
294 limit = dsize.width;
295 tab = tabR - x;
296 }
297 }
298
299 // do vertical convolution and decimation and write the result to the destination image
300 for( k = 0; k < PD_SZ; k++ )
301 rows[k] = buf + ((y*2 - PD_SZ/2 + k - sy0) % PD_SZ)*bufstep;
302 row0 = rows[0]; row1 = rows[1]; row2 = rows[2]; row3 = rows[3]; row4 = rows[4];
303
304 x = vecOp(rows, dst, (int)_dst.step, dsize.width);
305 for( ; x < dsize.width; x++ )
306 dst[x] = castOp(row2[x]*6 + (row1[x] + row3[x])*4 + row0[x] + row4[x]);
307 }
308}

311template<class CastOp, class VecOp> void
312pyrUp_( const Mat& _src, Mat& _dst, int)
313{
314 const int PU_SZ = 3;
315 typedef typename CastOp::type1 WT;
316 typedef typename CastOp::rtype T;
317
318 Size ssize = _src.size(), dsize = _dst.size();
319 int cn = _src.channels();
320 int bufstep = (int)alignSize((dsize.width+1)*cn, 16);
321 AutoBuffer<WT> _buf(bufstep*PU_SZ + 16);
322 WT* buf = alignPtr((WT*)_buf, 16);
323 AutoBuffer<int> _dtab(ssize.width*cn);
324 int* dtab = _dtab;
325 WT* rows[PU_SZ];
326 CastOp castOp;
327 VecOp vecOp;
328
329 CV_Assert( std::abs(dsize.width - ssize.width*2) == dsize.width % 2 &&
330 std::abs(dsize.height - ssize.height*2) == dsize.height % 2);
331 int k, x, sy0 = -PU_SZ/2, sy = sy0;
332
333 ssize.width *= cn;
334 dsize.width *= cn;
335
336 for( x = 0; x < ssize.width; x++ )
337 dtab[x] = (x/cn)*2*cn + x % cn;
338
339 for( int y = 0; y < ssize.height; y++ )
340 {
341 T* dst0 = (T*)(_dst.data + _dst.step*y*2);
342 T* dst1 = (T*)(_dst.data + _dst.step*(y*2+1));
343 WT *row0, *row1, *row2;
344
345 if( y*2+1 >= dsize.height )
346 dst1 = dst0;
347
348 // fill the ring buffer (horizontal convolution and decimation)
349 for( ; sy <= y + 1; sy++ )
350 {
351 WT* row = buf + ((sy - sy0) % PU_SZ)*bufstep;
352 int _sy = borderInterpolate(sy*2, ssize.height*2, BORDER_REFLECT_101)/2;
353 const T* src = (const T*)(_src.data + _src.step*_sy);
354
355 if( ssize.width == cn )
356 {
357 for( x = 0; x < cn; x++ )
358 row[x] = row[x + cn] = src[x]*8;
359 continue;
360 }
361
362 for( x = 0; x < cn; x++ )
363 {
364 int dx = dtab[x];
365 WT t0 = src[x]*6 + src[x + cn]*2;
366 WT t1 = (src[x] + src[x + cn])*4;
367 row[dx] = t0; row[dx + cn] = t1;
368 dx = dtab[ssize.width - cn + x];
369 int sx = ssize.width - cn + x;
370 t0 = src[sx - cn] + src[sx]*7;
371 t1 = src[sx]*8;
372 row[dx] = t0; row[dx + cn] = t1;
373
374 if (dsize.width > ssize.width*2)
375 {
376 row[(_dst.cols-1) + x] = row[dx + cn];
377 }
378 }
379
380 for( x = cn; x < ssize.width - cn; x++ )
381 {
382 int dx = dtab[x];
383 WT t0 = src[x-cn] + src[x]*6 + src[x+cn];
384 WT t1 = (src[x] + src[x+cn])*4;
385 row[dx] = t0;
386 row[dx+cn] = t1;
387 }
388 }
389
390 // do vertical convolution and decimation and write the result to the destination image
391 for( k = 0; k < PU_SZ; k++ )
392 rows[k] = buf + ((y - PU_SZ/2 + k - sy0) % PU_SZ)*bufstep;
393 row0 = rows[0]; row1 = rows[1]; row2 = rows[2];
394
395 x = vecOp(rows, dst0, (int)_dst.step, dsize.width);
396 for( ; x < dsize.width; x++ )
397 {
398 T t1 = castOp((row1[x] + row2[x])*4);
399 T t0 = castOp(row0[x] + row1[x]*6 + row2[x]);
400 dst1[x] = t1; dst0[x] = t0;
401 }
402 }
403
404 if (dsize.height > ssize.height*2)
405 {
406 T* dst0 = _dst.ptr<T>(ssize.height*2-2);
407 T* dst2 = _dst.ptr<T>(ssize.height*2);
408
409 for(x = 0; x < dsize.width ; x++ )
410 {
411 dst2[x] = dst0[x];
412 }
413 }
414}

420void cv::pyrDown( InputArray _src, OutputArray _dst, const Size& _dsz, int borderType )
421{
422 CV_Assert(borderType != BORDER_CONSTANT);
423
424 Mat src = _src.getMat();
425 Size dsz = _dsz == Size() ? Size((src.cols + 1)/2, (src.rows + 1)/2) : _dsz;
426 _dst.create( dsz, src.type() );
427 Mat dst = _dst.getMat();
428
429#ifdef HAVE_TEGRA_OPTIMIZATION
430 if(borderType == BORDER_DEFAULT && tegra::pyrDown(src, dst))
431 return;
432#endif
433
434 int depth = src.depth();
435 PyrFunc func = 0;
436 if( depth == CV_8U )
437 func = pyrDown_<FixPtCast<uchar, 8>, PyrDownVec_32s8u>;
438 else if( depth == CV_16S )
439 func = pyrDown_<FixPtCast<short, 8>, NoVec<int, short> >;
440 else if( depth == CV_16U )
441 func = pyrDown_<FixPtCast<ushort, 8>, NoVec<int, ushort> >;
442 else if( depth == CV_32F )
443 func = pyrDown_<FltCast<float, 8>, PyrDownVec_32f>;
444 else if( depth == CV_64F )
445 func = pyrDown_<FltCast<double, 8>, NoVec<double, double> >;
446 else
447 CV_Error( CV_StsUnsupportedFormat, "" );
448
449 func( src, dst, borderType );
450}

452void cv::pyrUp( InputArray _src, OutputArray _dst, const Size& _dsz, int borderType )
453{
454 CV_Assert(borderType == BORDER_DEFAULT);
455
456 Mat src = _src.getMat();
457 Size dsz = _dsz == Size() ? Size(src.cols*2, src.rows*2) : _dsz;
458 _dst.create( dsz, src.type() );
459 Mat dst = _dst.getMat();
460
461#ifdef HAVE_TEGRA_OPTIMIZATION
462 if(borderType == BORDER_DEFAULT && tegra::pyrUp(src, dst))
463 return;
464#endif
465
466 int depth = src.depth();
467 PyrFunc func = 0;
468 if( depth == CV_8U )
469 func = pyrUp_<FixPtCast<uchar, 6>, NoVec<int, uchar> >;
470 else if( depth == CV_16S )
471 func = pyrUp_<FixPtCast<short, 6>, NoVec<int, short> >;
472 else if( depth == CV_16U )
473 func = pyrUp_<FixPtCast<ushort, 6>, NoVec<int, ushort> >;
474 else if( depth == CV_32F )
475 func = pyrUp_<FltCast<float, 6>, NoVec<float, float> >;
476 else if( depth == CV_64F )
477 func = pyrUp_<FltCast<double, 6>, NoVec<double, double> >;
478 else
479 CV_Error( CV_StsUnsupportedFormat, "" );
480
481 func( src, dst, borderType );
482}

484void cv::buildPyramid( InputArray _src, OutputArrayOfArrays _dst, int maxlevel, int borderType )
485{
486 CV_Assert(borderType != BORDER_CONSTANT);
487
488 Mat src = _src.getMat();
489 _dst.create( maxlevel + 1, 1, 0 );
490 _dst.getMatRef(0) = src;
491 for( int i = 1; i <= maxlevel; i++ )
492 pyrDown( _dst.getMatRef(i-1), _dst.getMatRef(i), Size(), borderType );
493}

512CV_IMPL void
513cvReleasePyramid( CvMat*** _pyramid, int extra_layers )
514{
515 if( !_pyramid )
516 CV_Error( CV_StsNullPtr, "" );
517
518 if( *_pyramid )

526CV_IMPL CvMat**
527cvCreatePyramid( const CvArr* srcarr, int extra_layers, double rate,
528 const CvSize* layer_sizes, CvArr* bufarr,
529 int calc, int filter )
530{
531 const float eps = 0.1f;
532 uchar* ptr = 0;
533
534 CvMat stub, *src = cvGetMat( srcarr, &stub );
535
536 if( extra_layers < 0 )
537 CV_Error( CV_StsOutOfRange, "The number of extra layers must be non negative" );
538
539 int i, layer_step, elem_size = CV_ELEM_SIZE(src->type);
540 CvSize layer_size, size = cvGetMatSize(src);
541
542 if( bufarr )
543 {
544 CvMat bstub, *buf;
545 int bufsize = 0;
546
547 buf = cvGetMat( bufarr, &bstub );
548 bufsize = buf->rows*buf->cols*CV_ELEM_SIZE(buf->type);
549 layer_size = size;
550 for( i = 1; i <= extra_layers; i++ )
551 {
552 if( !layer_sizes )
553 {
554 layer_size.width = cvRound(layer_size.width*rate+eps);
555 layer_size.height = cvRound(layer_size.height*rate+eps);
556 }
557 else
558 layer_size = layer_sizes[i-1];
559 layer_step = layer_size.width*elem_size;
560 bufsize -= layer_step*layer_size.height;
561 }
562
563 if( bufsize < 0 )
564 CV_Error( CV_StsOutOfRange, "The buffer is too small to fit the pyramid" );
565 ptr = buf->data.ptr;
566 }
567
568 CvMat** pyramid = (CvMat**)cvAlloc( (extra_layers+1)*sizeof(pyramid[0]) );
569 memset( pyramid, 0, (extra_layers+1)*sizeof(pyramid[0]) );
570
571 pyramid[0] = cvCreateMatHeader( size.height, size.width, src->type );
572 cvSetData( pyramid[0], src->data.ptr, src->step );
573 layer_size = size;
574
575 for( i = 1; i <= extra_layers; i++ )
576 {
577 if( !layer_sizes )
578 {
579 layer_size.width = cvRound(layer_size.width*rate + eps);
580 layer_size.height = cvRound(layer_size.height*rate + eps);
581 }
582 else
583 layer_size = layer_sizes[i];
584
585 if( bufarr )
586 {
587 pyramid[i] = cvCreateMatHeader( layer_size.height, layer_size.width, src->type );
588 layer_step = layer_size.width*elem_size;
589 cvSetData( pyramid[i], ptr, layer_step );
590 ptr += layer_step*layer_size.height;
591 }
592 else
593 pyramid[i] = cvCreateMat( layer_size.height, layer_size.width, src->type );
594
595 if( calc )
596 cvPyrDown( pyramid[i-1], pyramid[i], filter );
597 //cvResize( pyramid[i-1], pyramid[i], CV_INTER_LINEAR );
598 }
599
600 return pyramid;
601}
