52static const size_t OUT_OF_RANGE = (size_t)1 << (sizeof(size_t)*8 - 2);
53
54static void
55calcHistLookupTables_8u( const Mat& hist, const SparseMat& shist,
56 int dims, const float** ranges, const double* uniranges,
57 bool uniform, bool issparse, vector<size_t>& _tab )
58{
59 const int low = 0, high = 256;
60 int i, j;
61 _tab.resize((high-low)*dims);
62 size_t* tab = &_tab[0];
63
64 if( uniform )
65 {
66 for( i = 0; i < dims; i++ )
67 {
68 double a = uniranges[i*2];
69 double b = uniranges[i*2+1];
70 int sz = !issparse ? hist.size[i] : shist.size(i);
71 size_t step = !issparse ? hist.step[i] : 1;
72
73 for( j = low; j < high; j++ )
74 {
75 int idx = cvFloor(j*a + b);
76 size_t written_idx;
77 if( (unsigned)idx < (unsigned)sz )
78 written_idx = idx*step;
79 else
80 written_idx = OUT_OF_RANGE;
81
82 tab[i*(high - low) + j - low] = written_idx;
83 }
84 }
85 }
86 else
87 {
88 for( i = 0; i < dims; i++ )
89 {
90 int limit = std::min(cvCeil(ranges[i][0]), high);
91 int idx = -1, sz = !issparse ? hist.size[i] : shist.size(i);
92 size_t written_idx = OUT_OF_RANGE;
93 size_t step = !issparse ? hist.step[i] : 1;
94
95 for(j = low;;)
96 {
97 for( ; j < limit; j++ )
98 tab[i*(high - low) + j - low] = written_idx;
99
100 if( (unsigned)(++idx) < (unsigned)sz )
101 {
102 limit = std::min(cvCeil(ranges[i][idx+1]), high);
103 written_idx = idx*step;
104 }
105 else
106 {
107 for( ; j < high; j++ )
108 tab[i*(high - low) + j - low] = OUT_OF_RANGE;
109 break;
110 }
111 }
112 }
113 }
114}

117static void histPrepareImages( const Mat* images, int nimages, const int* channels,
118 const Mat& mask, int dims, const int* histSize,
119 const float** ranges, bool uniform,
120 vector<uchar*>& ptrs, vector<int>& deltas,
121 Size& imsize, vector<double>& uniranges )
122{
123 int i, j, c;
124 CV_Assert( channels != 0 || nimages == dims );
125
126 imsize = images[0].size();
127 int depth = images[0].depth(), esz1 = (int)images[0].elemSize1();
128 bool isContinuous = true;
129
130 ptrs.resize(dims + 1);
131 deltas.resize((dims + 1)*2);
132
133 for( i = 0; i < dims; i++ )
134 {
135 if(!channels)
136 {
137 j = i;
138 c = 0;
139 CV_Assert( images[j].channels() == 1 );
140 }
141 else
142 {
143 c = channels[i];
144 CV_Assert( c >= 0 );
145 for( j = 0; j < nimages; c -= images[j].channels(), j++ )
146 if( c < images[j].channels() )
147 break;
148 CV_Assert( j < nimages );
149 }
150
151 CV_Assert( images[j].size() == imsize && images[j].depth() == depth );
152 if( !images[j].isContinuous() )
153 isContinuous = false;
154 ptrs[i] = images[j].data + c*esz1;
155 deltas[i*2] = images[j].channels();
156 deltas[i*2+1] = (int)(images[j].step/esz1 - imsize.width*deltas[i*2]);
157 }
158
159 if( mask.data )
160 {
161 CV_Assert( mask.size() == imsize && mask.channels() == 1 );
162 isContinuous = isContinuous && mask.isContinuous();
163 ptrs[dims] = mask.data;
164 deltas[dims*2] = 1;
165 deltas[dims*2 + 1] = (int)(mask.step/mask.elemSize1());
166 }
167
168#ifndef HAVE_TBB
169 if( isContinuous )
170 {
171 imsize.width *= imsize.height;
172 imsize.height = 1;
173 }
174#endif
175
176 if( !ranges )
177 {
178 CV_Assert( depth == CV_8U );
179
180 uniranges.resize( dims*2 );
181 for( i = 0; i < dims; i++ )
182 {
183 uniranges[i*2] = histSize[i]/256.;
184 uniranges[i*2+1] = 0;
185 }
186 }
187 else if( uniform )
188 {
189 uniranges.resize( dims*2 );
190 for( i = 0; i < dims; i++ )
191 {
192 CV_Assert( ranges[i] && ranges[i][0] < ranges[i][1] );
193 double low = ranges[i][0], high = ranges[i][1];
194 double t = histSize[i]/(high - low);
195 uniranges[i*2] = t;
196 uniranges[i*2+1] = -t*low;
197 }
198 }
199 else
200 {
201 for( i = 0; i < dims; i++ )
202 {
203 size_t n = histSize[i];
204 for(size_t k = 0; k < n; k++ )
205 CV_Assert( ranges[i][k] < ranges[i][k+1] );
206 }
207 }
208}

215template<typename T>
216class calcHist1D_Invoker
217{
218public:
219 calcHist1D_Invoker( const vector<uchar*>& _ptrs, const vector<int>& _deltas,
220 Mat& hist, const double* _uniranges, int sz, int dims,
221 Size& imageSize )
222 : mask_(_ptrs[dims]),
223 mstep_(_deltas[dims*2 + 1]),
224 imageWidth_(imageSize.width),
225 histogramSize_(hist.size()), histogramType_(hist.type()),
226 globalHistogram_((tbb::atomic<int>*)hist.data)
227 {
228 p_[0] = ((T**)&_ptrs[0])[0];
229 step_[0] = (&_deltas[0])[1];
230 d_[0] = (&_deltas[0])[0];
231 a_[0] = (&_uniranges[0])[0];
232 b_[0] = (&_uniranges[0])[1];
233 size_[0] = sz;
234 }
235
236 void operator()( const BlockedRange& range ) const
237 {
238 T* p0 = p_[0] + range.begin() * (step_[0] + imageWidth_*d_[0]);
239 uchar* mask = mask_ + range.begin()*mstep_;
240
241 for( int row = range.begin(); row < range.end(); row++, p0 += step_[0] )
242 {
243 if( !mask_ )
244 {
245 for( int x = 0; x < imageWidth_; x++, p0 += d_[0] )
246 {
247 int idx = cvFloor(*p0*a_[0] + b_[0]);
248 if( (unsigned)idx < (unsigned)size_[0] )
249 {
250 globalHistogram_[idx].fetch_and_add(1);
251 }
252 }
253 }
254 else
255 {
256 for( int x = 0; x < imageWidth_; x++, p0 += d_[0] )
257 {
258 if( mask[x] )
259 {
260 int idx = cvFloor(*p0*a_[0] + b_[0]);
261 if( (unsigned)idx < (unsigned)size_[0] )
262 {
263 globalHistogram_[idx].fetch_and_add(1);
264 }
265 }
266 }
267 mask += mstep_;
268 }
269 }
270 }
271
272private:
273 calcHist1D_Invoker operator=(const calcHist1D_Invoker&);
274
275 T* p_[one];
276 uchar* mask_;
277 int step_[one];
278 int d_[one];
279 int mstep_;
280 double a_[one];
281 double b_[one];
282 int size_[one];
283 int imageWidth_;
284 Size histogramSize_;
285 int histogramType_;
286 tbb::atomic<int>* globalHistogram_;
287};

454class CalcHist1D_8uInvoker
455{
456public:
457 CalcHist1D_8uInvoker( const vector<uchar*>& ptrs, const vector<int>& deltas,
458 Size imsize, Mat& hist, int dims, const vector<size_t>& tab,
459 tbb::mutex* lock )
460 : mask_(ptrs[dims]),
461 mstep_(deltas[dims*2 + 1]),
462 imageWidth_(imsize.width),
463 imageSize_(imsize),
464 histSize_(hist.size()), histType_(hist.type()),
465 tab_((size_t*)&tab[0]),
466 histogramWriteLock_(lock),
467 globalHistogram_(hist.data)
468 {
469 p_[0] = (&ptrs[0])[0];
470 step_[0] = (&deltas[0])[1];
471 d_[0] = (&deltas[0])[0];
472 }
473
474 void operator()( const BlockedRange& range ) const
475 {
476 int localHistogram[256] = { 0, };
477 uchar* mask = mask_;
478 uchar* p0 = p_[0];
479 int x;
480 tbb::mutex::scoped_lock lock;
481
482 if( !mask_ )
483 {
484 int n = (imageWidth_ - 4) / 4 + 1;
485 int tail = imageWidth_ - n*4;
486
487 int xN = 4*n;
488 p0 += (xN*d_[0] + tail*d_[0] + step_[0]) * range.begin();
489 }
490 else
491 {
492 p0 += (imageWidth_*d_[0] + step_[0]) * range.begin();
493 mask += mstep_*range.begin();
494 }
495
496 for( int i = range.begin(); i < range.end(); i++, p0 += step_[0] )
497 {
498 if( !mask_ )
499 {
500 if( d_[0] == 1 )
501 {
502 for( x = 0; x <= imageWidth_ - 4; x += 4 )
503 {
504 int t0 = p0[x], t1 = p0[x+1];
505 localHistogram[t0]++; localHistogram[t1]++;
506 t0 = p0[x+2]; t1 = p0[x+3];
507 localHistogram[t0]++; localHistogram[t1]++;
508 }
509 p0 += x;
510 }
511 else
512 {
513 for( x = 0; x <= imageWidth_ - 4; x += 4 )
514 {
515 int t0 = p0[0], t1 = p0[d_[0]];
516 localHistogram[t0]++; localHistogram[t1]++;
517 p0 += d_[0]*2;
518 t0 = p0[0]; t1 = p0[d_[0]];
519 localHistogram[t0]++; localHistogram[t1]++;
520 p0 += d_[0]*2;
521 }
522 }
523
524 for( ; x < imageWidth_; x++, p0 += d_[0] )
525 {
526 localHistogram[*p0]++;
527 }
528 }
529 else
530 {
531 for( x = 0; x < imageWidth_; x++, p0 += d_[0] )
532 {
533 if( mask[x] )
534 {
535 localHistogram[*p0]++;
536 }
537 }
538 mask += mstep_;
539 }
540 }
541
542 lock.acquire(*histogramWriteLock_);
543 for(int i = 0; i < 256; i++ )
544 {
545 size_t hidx = tab_[i];
546 if( hidx < OUT_OF_RANGE )
547 {
548 *(int*)((globalHistogram_ + hidx)) += localHistogram[i];
549 }
550 }
551 lock.release();
552 }

555 {
556 return ( histogram.total() >= 8
557 && imageSize.width * imageSize.height >= 160*120 );

561 uchar* p_[one];
562 uchar* mask_;
563 int mstep_;
564 int step_[one];
565 int d_[one];
566 int imageWidth_;
567 Size imageSize_;
568 Size histSize_;
569 int histType_;
570 size_t* tab_;

577public:
578 CalcHist2D_8uInvoker( const vector<uchar*>& _ptrs, const vector<int>& _deltas,
579 Size imsize, Mat& hist, int dims, const vector<size_t>& _tab,

632 for(int i = 0; i < histSize_.width*histSize_.height; i++)
633 {
634 ((int*)globalHistogram_)[i] += ((int*)localHistData)[i];
635 }
636 lock.release();
637 }

674 d_[0] = (&_deltas[0])[0]; d_[1] = (&_deltas[0])[2]; d_[2] = (&_deltas[0])[4];
675 }

677 void operator()( const BlockedRange& range ) const
678 {
679 uchar* p0 = p_[0] + range.begin()*(step_[0] + imageWidth_*d_[0]);
680 uchar* p1 = p_[1] + range.begin()*(step_[1] + imageWidth_*d_[1]);
681 uchar* p2 = p_[2] + range.begin()*(step_[2] + imageWidth_*d_[2]);
682 uchar* mask = mask_ + range.begin()*mstep_;
683
684 for(int i = range.begin(); i < range.end(); i++, p0 += step_[0], p1 += step_[1], p2 += step_[2] )
685 {
686 if( !mask_ )
687 {
688 for( int x = 0; x < imageWidth_; x++, p0 += d_[0], p1 += d_[1], p2 += d_[2] )
689 {
690 size_t idx = tab_[*p0] + tab_[*p1 + 256] + tab_[*p2 + 512];
691 if( idx < OUT_OF_RANGE )
692 {
693 ( *(tbb::atomic<int>*)(globalHistogram_ + idx) ).fetch_and_add(1);
694 }
695 }
696 }
697 else
698 {
699 for( int x = 0; x < imageWidth_; x++, p0 += d_[0], p1 += d_[1], p2 += d_[2] )
700 {
701 size_t idx;
702 if( mask[x] && (idx = tab_[*p0] + tab_[*p1 + 256] + tab_[*p2 + 512]) < OUT_OF_RANGE )
703 {
704 (*(tbb::atomic<int>*)(globalHistogram_ + idx)).fetch_and_add(1);
705 }
706 }
707 mask += mstep_;
708 }
709 }

755{
756 T** ptrs = (T**)&_ptrs[0];
757 const int* deltas = &_deltas[0];
758 uchar* H = hist.data;
759 int i, x;
760 const uchar* mask = _ptrs[dims];
761 int mstep = _deltas[dims*2 + 1];
762 int size[CV_MAX_DIM];
763 size_t hstep[CV_MAX_DIM];
764
765 for( i = 0; i < dims; i++ )
766 {
767 size[i] = hist.size[i];
768 hstep[i] = hist.step[i];

772 {

774
775 if( dims == 1 )
776 {
777#ifdef HAVE_TBB

779 parallel_for(BlockedRange(0, imsize.height), body);
780#else

782 int sz = size[0], d0 = deltas[0], step0 = deltas[1];

786 {

789 {

793 }
794 else
795 for( x = 0; x < imsize.width; x++, p0 += d0 )
796 if( mask[x] )
797 {
798 int idx = cvFloor(*p0*a + b);
799 if( (unsigned)idx < (unsigned)sz )
800 ((int*)H)[idx]++;
801 }
802 }
803#endif //HAVE_TBB
804 return;
805 }
806 else if( dims == 2 )
807 {
808#ifdef HAVE_TBB
809 calcHist2D_Invoker<T> body(_ptrs, _deltas, hist, _uniranges, size, dims, imsize, hstep);
810 parallel_for(BlockedRange(0, imsize.height), body);
811#else
812 double a0 = uniranges[0], b0 = uniranges[1], a1 = uniranges[2], b1 = uniranges[3];
813 int sz0 = size[0], sz1 = size[1];
814 int d0 = deltas[0], step0 = deltas[1],
815 d1 = deltas[2], step1 = deltas[3];
816 size_t hstep0 = hstep[0];
817 const T* p0 = (const T*)ptrs[0];
818 const T* p1 = (const T*)ptrs[1];

834 int idx0 = cvFloor(*p0*a0 + b0);
835 int idx1 = cvFloor(*p1*a1 + b1);

837 ((int*)(H + hstep0*idx0))[idx1]++;

839 }
840#endif //HAVE_TBB
841 return;
842 }
843 else if( dims == 3 )
844 {
845#ifdef HAVE_TBB
846 if( calcHist3D_Invoker<T>::isFit(hist, imsize) )
847 {
848 calcHist3D_Invoker<T> body(_ptrs, _deltas, imsize, hist, uniranges, dims, hstep, size);
849 parallel_for(BlockedRange(0, imsize.height), body);

851 }

854 a1 = uniranges[2], b1 = uniranges[3],
855 a2 = uniranges[4], b2 = uniranges[5];

863 const T* p2 = (const T*)ptrs[2];
864
865 for( ; imsize.height--; p0 += step0, p1 += step1, p2 += step2, mask += mstep )
866 {
867 if( !mask )
868 for( x = 0; x < imsize.width; x++, p0 += d0, p1 += d1, p2 += d2 )
869 {
870 int idx0 = cvFloor(*p0*a0 + b0);
871 int idx1 = cvFloor(*p1*a1 + b1);
872 int idx2 = cvFloor(*p2*a2 + b2);
873 if( (unsigned)idx0 < (unsigned)sz0 &&
874 (unsigned)idx1 < (unsigned)sz1 &&
875 (unsigned)idx2 < (unsigned)sz2 )
876 ((int*)(H + hstep0*idx0 + hstep1*idx1))[idx2]++;

878 else
879 for( x = 0; x < imsize.width; x++, p0 += d0, p1 += d1, p2 += d2 )
880 if( mask[x] )
881 {
882 int idx0 = cvFloor(*p0*a0 + b0);
883 int idx1 = cvFloor(*p1*a1 + b1);
884 int idx2 = cvFloor(*p2*a2 + b2);
885 if( (unsigned)idx0 < (unsigned)sz0 &&
886 (unsigned)idx1 < (unsigned)sz1 &&

890 }
891 }
892 else
893 {

897 for( x = 0; x < imsize.width; x++ )
898 {
899 uchar* Hptr = H;

901 {
902 int idx = cvFloor(*ptrs[i]*uniranges[i*2] + uniranges[i*2+1]);
903 if( (unsigned)idx >= (unsigned)size[i] )
904 break;
905 ptrs[i] += deltas[i*2];
906 Hptr += idx*hstep[i];
907 }
908
909 if( i == dims )
910 ++*((int*)Hptr);
911 else
912 for( ; i < dims; i++ )
913 ptrs[i] += deltas[i*2];
914 }
915 else
916 for( x = 0; x < imsize.width; x++ )
917 {
918 uchar* Hptr = H;
919 i = 0;
920 if( mask[x] )
921 for( ; i < dims; i++ )
922 {
923 int idx = cvFloor(*ptrs[i]*uniranges[i*2] + uniranges[i*2+1]);
924 if( (unsigned)idx >= (unsigned)size[i] )
925 break;
926 ptrs[i] += deltas[i*2];
927 Hptr += idx*hstep[i];
928 }
929
930 if( i == dims )
931 ++*((int*)Hptr);
932 else
933 for( ; i < dims; i++ )
934 ptrs[i] += deltas[i*2];
935 }
936 for( i = 0; i < dims; i++ )

939 }

946 ranges[i] = &_ranges[i][0];
947
948 for( ; imsize.height--; mask += mstep )
949 {
950 for( x = 0; x < imsize.width; x++ )
951 {
952 uchar* Hptr = H;
953 i = 0;
954
955 if( !mask || mask[x] )
956 for( ; i < dims; i++ )
957 {
958 float v = (float)*ptrs[i];
959 const float* R = ranges[i];
960 int idx = -1, sz = size[i];
961
962 while( v >= R[idx+1] && ++idx < sz )
963 ; // nop
964
965 if( (unsigned)idx >= (unsigned)sz )
966 break;
967
968 ptrs[i] += deltas[i*2];
969 Hptr += idx*hstep[i];
970 }
971
972 if( i == dims )
973 ++*((int*)Hptr);
974 else
975 for( ; i < dims; i++ )
976 ptrs[i] += deltas[i*2];
977 }
978
979 for( i = 0; i < dims; i++ )
980 ptrs[i] += deltas[i*2 + 1];
981 }
982 }

1450template<typename T, typename BT> static void
1451calcBackProj_( vector<uchar*>& _ptrs, const vector<int>& _deltas,
1452 Size imsize, const Mat& hist, int dims, const float** _ranges,
1453 const double* _uniranges, float scale, bool uniform )
1454{
1455 T** ptrs = (T**)&_ptrs[0];
1456 const int* deltas = &_deltas[0];
1457 uchar* H = hist.data;
1458 int i, x;
1459 BT* bproj = (BT*)_ptrs[dims];
1460 int bpstep = _deltas[dims*2 + 1];
1461 int size[CV_MAX_DIM];
1462 size_t hstep[CV_MAX_DIM];
1463
1464 for( i = 0; i < dims; i++ )
1465 {
1466 size[i] = hist.size[i];
1467 hstep[i] = hist.step[i];
1468 }
1469
1470 if( uniform )
1471 {
1472 const double* uniranges = &_uniranges[0];
1473
1474 if( dims == 1 )
1475 {
1476 double a = uniranges[0], b = uniranges[1];
1477 int sz = size[0], d0 = deltas[0], step0 = deltas[1];
1478 const T* p0 = (const T*)ptrs[0];
1479
1480 for( ; imsize.height--; p0 += step0, bproj += bpstep )
1481 {
1482 for( x = 0; x < imsize.width; x++, p0 += d0 )
1483 {
1484 int idx = cvFloor(*p0*a + b);
1485 bproj[x] = (unsigned)idx < (unsigned)sz ? saturate_cast<BT>(((float*)H)[idx]*scale) : 0;
1486 }
1487 }
1488 }
1489 else if( dims == 2 )
1490 {
1491 double a0 = uniranges[0], b0 = uniranges[1],
1492 a1 = uniranges[2], b1 = uniranges[3];
1493 int sz0 = size[0], sz1 = size[1];
1494 int d0 = deltas[0], step0 = deltas[1],
1495 d1 = deltas[2], step1 = deltas[3];
1496 size_t hstep0 = hstep[0];
1497 const T* p0 = (const T*)ptrs[0];
1498 const T* p1 = (const T*)ptrs[1];
1499
1500 for( ; imsize.height--; p0 += step0, p1 += step1, bproj += bpstep )
1501 {
1502 for( x = 0; x < imsize.width; x++, p0 += d0, p1 += d1 )
1503 {
1504 int idx0 = cvFloor(*p0*a0 + b0);
1505 int idx1 = cvFloor(*p1*a1 + b1);
1506 bproj[x] = (unsigned)idx0 < (unsigned)sz0 &&
1507 (unsigned)idx1 < (unsigned)sz1 ?
1508 saturate_cast<BT>(((float*)(H + hstep0*idx0))[idx1]*scale) : 0;
1509 }
1510 }
1511 }
1512 else if( dims == 3 )
1513 {
1514 double a0 = uniranges[0], b0 = uniranges[1],
1515 a1 = uniranges[2], b1 = uniranges[3],
1516 a2 = uniranges[4], b2 = uniranges[5];
1517 int sz0 = size[0], sz1 = size[1], sz2 = size[2];
1518 int d0 = deltas[0], step0 = deltas[1],
1519 d1 = deltas[2], step1 = deltas[3],
1520 d2 = deltas[4], step2 = deltas[5];
1521 size_t hstep0 = hstep[0], hstep1 = hstep[1];
1522 const T* p0 = (const T*)ptrs[0];
1523 const T* p1 = (const T*)ptrs[1];
1524 const T* p2 = (const T*)ptrs[2];
1525
1526 for( ; imsize.height--; p0 += step0, p1 += step1, p2 += step2, bproj += bpstep )
1527 {
1528 for( x = 0; x < imsize.width; x++, p0 += d0, p1 += d1, p2 += d2 )
1529 {
1530 int idx0 = cvFloor(*p0*a0 + b0);
1531 int idx1 = cvFloor(*p1*a1 + b1);
1532 int idx2 = cvFloor(*p2*a2 + b2);
1533 bproj[x] = (unsigned)idx0 < (unsigned)sz0 &&
1534 (unsigned)idx1 < (unsigned)sz1 &&
1535 (unsigned)idx2 < (unsigned)sz2 ?
1536 saturate_cast<BT>(((float*)(H + hstep0*idx0 + hstep1*idx1))[idx2]*scale) : 0;
1537 }
1538 }
1539 }
1540 else
1541 {
1542 for( ; imsize.height--; bproj += bpstep )
1543 {
1544 for( x = 0; x < imsize.width; x++ )
1545 {
1546 uchar* Hptr = H;
1547 for( i = 0; i < dims; i++ )
1548 {
1549 int idx = cvFloor(*ptrs[i]*uniranges[i*2] + uniranges[i*2+1]);
1550 if( (unsigned)idx >= (unsigned)size[i] || (_ranges && *ptrs[i] >= _ranges[i][1]))
1551 break;
1552 ptrs[i] += deltas[i*2];
1553 Hptr += idx*hstep[i];
1554 }
1555
1556 if( i == dims )
1557 bproj[x] = saturate_cast<BT>(*(float*)Hptr*scale);
1558 else
1559 {
1560 bproj[x] = 0;
1561 for( ; i < dims; i++ )
1562 ptrs[i] += deltas[i*2];
1563 }
1564 }
1565 for( i = 0; i < dims; i++ )
1566 ptrs[i] += deltas[i*2 + 1];
1567 }
1568 }
1569 }
1570 else
1571 {
1572 // non-uniform histogram
1573 const float* ranges[CV_MAX_DIM];
1574 for( i = 0; i < dims; i++ )
1575 ranges[i] = &_ranges[i][0];
1576
1577 for( ; imsize.height--; bproj += bpstep )
1578 {
1579 for( x = 0; x < imsize.width; x++ )
1580 {
1581 uchar* Hptr = H;
1582 for( i = 0; i < dims; i++ )
1583 {
1584 float v = (float)*ptrs[i];
1585 const float* R = ranges[i];
1586 int idx = -1, sz = size[i];
1587
1588 while( v >= R[idx+1] && ++idx < sz )
1589 ; // nop
1590
1591 if( (unsigned)idx >= (unsigned)sz )
1592 break;
1593
1594 ptrs[i] += deltas[i*2];
1595 Hptr += idx*hstep[i];
1596 }
1597
1598 if( i == dims )
1599 bproj[x] = saturate_cast<BT>(*(float*)Hptr*scale);
1600 else
1601 {
1602 bproj[x] = 0;
1603 for( ; i < dims; i++ )
1604 ptrs[i] += deltas[i*2];
1605 }
1606 }
1607
1608 for( i = 0; i < dims; i++ )
1609 ptrs[i] += deltas[i*2 + 1];
1610 }
1611 }

1651 bproj[x] = t0; bproj[x+1] = t1;
1652 t0 = matH[p0[x+2]]; t1 = matH[p0[x+3]];
1653 bproj[x+2] = t0; bproj[x+3] = t1;
1654 }
1655 p0 += x;
1656 }
1657 else
1658 for( x = 0; x <= imsize.width - 4; x += 4 )
1659 {
1660 uchar t0 = matH[p0[0]], t1 = matH[p0[d0]];

1662 p0 += d0*2;

1665 p0 += d0*2;
1666 }

1669 bproj[x] = matH[*p0];
1670 }
1671 }
1672 else if( dims == 2 )
1673 {

1675 d1 = deltas[2], step1 = deltas[3];

1677 const uchar* p1 = (const uchar*)ptrs[1];
1678
1679 for( ; imsize.height--; p0 += step0, p1 += step1, bproj += bpstep )
1680 {
1681 for( x = 0; x < imsize.width; x++, p0 += d0, p1 += d1 )
1682 {
1683 size_t idx = tab[*p0] + tab[*p1 + 256];
1684 bproj[x] = idx < OUT_OF_RANGE ? saturate_cast<uchar>(*(float*)(H + idx)*scale) : 0;
1685 }
1686 }
1687 }
1688 else if( dims == 3 )
1689 {
1690 int d0 = deltas[0], step0 = deltas[1],

1692 d2 = deltas[4], step2 = deltas[5];

1704 }
1705 }
1706 else
1707 {
1708 for( ; imsize.height--; bproj += bpstep )

1712 uchar* Hptr = H;

1975
1976double cv::compareHist( InputArray _H1, InputArray _H2, int method )
1977{
1978 Mat H1 = _H1.getMat(), H2 = _H2.getMat();
1979 const Mat* arrays[] = {&H1, &H2, 0};
1980 Mat planes[2];
1981 NAryMatIterator it(arrays, planes);
1982 double result = 0;
1983 int j, len = (int)it.size;
1984
1985 CV_Assert( H1.type() == H2.type() && H1.type() == CV_32F );
1986
1987 double s1 = 0, s2 = 0, s11 = 0, s12 = 0, s22 = 0;
1988
1989 CV_Assert( it.planes[0].isContinuous() && it.planes[1].isContinuous() );
1990
1991 for( size_t i = 0; i < it.nplanes; i++, ++it )
1992 {
1993 const float* h1 = (const float*)it.planes[0].data;
1994 const float* h2 = (const float*)it.planes[1].data;
1995 len = it.planes[0].rows*it.planes[0].cols;
1996
1997 if( method == CV_COMP_CHISQR )
1998 {
1999 for( j = 0; j < len; j++ )
2000 {
2001 double a = h1[j] - h2[j];
2002 double b = h1[j];
2003 if( fabs(b) > DBL_EPSILON )
2004 result += a*a/b;
2005 }
2006 }
2007 else if( method == CV_COMP_CORREL )
2008 {
2009 for( j = 0; j < len; j++ )
2010 {
2011 double a = h1[j];
2012 double b = h2[j];
2013
2014 s12 += a*b;
2015 s1 += a;
2016 s11 += a*a;
2017 s2 += b;
2018 s22 += b*b;
2019 }
2020 }
2021 else if( method == CV_COMP_INTERSECT )
2022 {
2023 for( j = 0; j < len; j++ )
2024 result += std::min(h1[j], h2[j]);
2025 }
2026 else if( method == CV_COMP_BHATTACHARYYA )
2027 {
2028 for( j = 0; j < len; j++ )
2029 {
2030 double a = h1[j];
2031 double b = h2[j];
2032 result += std::sqrt(a*b);
2033 s1 += a;
2034 s2 += b;
2035 }
2036 }
2037 else
2038 CV_Error( CV_StsBadArg, "Unknown comparison method" );
2039 }
2040
2041 if( method == CV_COMP_CORREL )
2042 {
2043 size_t total = H1.total();
2044 double scale = 1./total;
2045 double num = s12 - s1*s2*scale;
2046 double denom2 = (s11 - s1*s1*scale)*(s22 - s2*s2*scale);
2047 result = std::abs(denom2) > DBL_EPSILON ? num/std::sqrt(denom2) : 1.;
2048 }
2049 else if( method == CV_COMP_BHATTACHARYYA )
2050 {
2051 s1 *= s2;
2052 s1 = fabs(s1) > FLT_EPSILON ? 1./std::sqrt(s1) : 1.;
2053 result = std::sqrt(std::max(1. - result*s1, 0.));
2054 }
2055
2056 return result;
2057}

2060double cv::compareHist( const SparseMat& H1, const SparseMat& H2, int method )
2061{
2062 double result = 0;
2063 int i, dims = H1.dims();
2064
2065 CV_Assert( dims > 0 && dims == H2.dims() && H1.type() == H2.type() && H1.type() == CV_32F );
2066 for( i = 0; i < dims; i++ )
2067 CV_Assert( H1.size(i) == H2.size(i) );
2068
2069 const SparseMat *PH1 = &H1, *PH2 = &H2;
2070 if( PH1->nzcount() > PH2->nzcount() && method != CV_COMP_CHISQR )
2071 std::swap(PH1, PH2);
2072
2073 SparseMatConstIterator it = PH1->begin();
2074 int N1 = (int)PH1->nzcount(), N2 = (int)PH2->nzcount();
2075
2076 if( method == CV_COMP_CHISQR )
2077 {
2078 for( i = 0; i < N1; i++, ++it )
2079 {
2080 float v1 = it.value<float>();
2081 const SparseMat::Node* node = it.node();
2082 float v2 = PH2->value<float>(node->idx, (size_t*)&node->hashval);
2083 double a = v1 - v2;
2084 double b = v1;
2085 if( fabs(b) > DBL_EPSILON )
2086 result += a*a/b;
2087 }
2088 }
2089 else if( method == CV_COMP_CORREL )
2090 {
2091 double s1 = 0, s2 = 0, s11 = 0, s12 = 0, s22 = 0;
2092
2093 for( i = 0; i < N1; i++, ++it )
2094 {
2095 double v1 = it.value<float>();
2096 const SparseMat::Node* node = it.node();
2097 s12 += v1*PH2->value<float>(node->idx, (size_t*)&node->hashval);
2098 s1 += v1;
2099 s11 += v1*v1;
2100 }
2101
2102 it = PH2->begin();
2103 for( i = 0; i < N2; i++, ++it )
2104 {
2105 double v2 = it.value<float>();
2106 s2 += v2;
2107 s22 += v2*v2;
2108 }
2109
2110 size_t total = 1;
2111 for( i = 0; i < H1.dims(); i++ )
2112 total *= H1.size(i);
2113 double scale = 1./total;
2114 double num = s12 - s1*s2*scale;
2115 double denom2 = (s11 - s1*s1*scale)*(s22 - s2*s2*scale);
2116 result = std::abs(denom2) > DBL_EPSILON ? num/std::sqrt(denom2) : 1.;
2117 }
2118 else if( method == CV_COMP_INTERSECT )
2119 {
2120 for( i = 0; i < N1; i++, ++it )
2121 {
2122 float v1 = it.value<float>();
2123 const SparseMat::Node* node = it.node();
2124 float v2 = PH2->value<float>(node->idx, (size_t*)&node->hashval);
2125 if( v2 )
2126 result += std::min(v1, v2);
2127 }
2128 }
2129 else if( method == CV_COMP_BHATTACHARYYA )
2130 {
2131 double s1 = 0, s2 = 0;
2132
2133 for( i = 0; i < N1; i++, ++it )
2134 {
2135 double v1 = it.value<float>();
2136 const SparseMat::Node* node = it.node();
2137 double v2 = PH2->value<float>(node->idx, (size_t*)&node->hashval);
2138 result += std::sqrt(v1*v2);
2139 s1 += v1;
2140 }
2141
2142 it = PH2->begin();
2143 for( i = 0; i < N2; i++, ++it )
2144 s2 += it.value<float>();
2145
2146 s1 *= s2;
2147 s1 = fabs(s1) > FLT_EPSILON ? 1./std::sqrt(s1) : 1.;
2148 result = std::sqrt(std::max(1. - result*s1, 0.));
2149 }
2150 else
2151 CV_Error( CV_StsBadArg, "Unknown comparison method" );
2152
2153 return result;
2154}

2451CV_IMPL double
2452cvCompareHist( const CvHistogram* hist1,

2475 CV_Error( CV_StsUnmatchedSizes,
2476 "The histograms have different numbers of dimensions" );

2479 {

2481 CV_Error( CV_StsUnmatchedSizes, "The histograms have different sizes" );
2482 total *= size1[i];

2487 CvSparseMat* mat2 = (CvSparseMat*)(hist2->bins);

2495 }

2512 {

2523 0, 0, &node1->hashval );
2524 if( node2_data )
2525 {
2526 double v2 = *(float*)node2_data;
2527 s12 += v1*v2;
2528 }
2529 s1 += v1;
2530 s11 += v1*v1;
2531 }
2532
2533 for( node2 = cvInitSparseMatIterator( mat2, &iterator );
2534 node2 != 0; node2 = cvGetNextSparseNode( &iterator ))
2535 {
2536 double v2 = *(float*)CV_NODE_VAL(mat2,node2);
2537 s2 += v2;
2538 s22 += v2*v2;
2539 }
2540

2543 result = fabs(denom2) > DBL_EPSILON ? num/sqrt(denom2) : 1;
2544 }
2545 else if( method == CV_COMP_INTERSECT )
2546 {
2547 for( node1 = cvInitSparseMatIterator( mat1, &iterator );
2548 node1 != 0; node1 = cvGetNextSparseNode( &iterator ))
