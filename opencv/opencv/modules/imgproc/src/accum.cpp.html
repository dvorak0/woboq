48template<typename T, typename AT> void
49acc_( const T* src, AT* dst, const uchar* mask, int len, int cn )
50{
51 int i = 0;
52
53 if( !mask )
54 {
55 len *= cn;
56 #if CV_ENABLE_UNROLLED
57 for( ; i <= len - 4; i += 4 )
58 {
59 AT t0, t1;
60 t0 = src[i] + dst[i];
61 t1 = src[i+1] + dst[i+1];
62 dst[i] = t0; dst[i+1] = t1;
63
64 t0 = src[i+2] + dst[i+2];
65 t1 = src[i+3] + dst[i+3];
66 dst[i+2] = t0; dst[i+3] = t1;
67 }
68 #endif
69 for( ; i < len; i++ )
70 dst[i] += src[i];
71 }
72 else if( cn == 1 )
73 {
74 for( ; i < len; i++ )
75 {
76 if( mask[i] )
77 dst[i] += src[i];
78 }
79 }
80 else if( cn == 3 )
81 {
82 for( ; i < len; i++, src += 3, dst += 3 )
83 {
84 if( mask[i] )
85 {
86 AT t0 = src[0] + dst[0];
87 AT t1 = src[1] + dst[1];
88 AT t2 = src[2] + dst[2];
89
90 dst[0] = t0; dst[1] = t1; dst[2] = t2;
91 }
92 }
93 }
94 else
95 {
96 for( ; i < len; i++, src += cn, dst += cn )
97 if( mask[i] )
98 {
99 for( int k = 0; k < cn; k++ )
100 dst[k] += src[k];
101 }
102 }
103}

106template<typename T, typename AT> void
107accSqr_( const T* src, AT* dst, const uchar* mask, int len, int cn )
108{
109 int i = 0;

113 len *= cn;
114 #if CV_ENABLE_UNROLLED
115 for( ; i <= len - 4; i += 4 )

119 t1 = (AT)src[i+1]*src[i+1] + dst[i+1];
120 dst[i] = t0; dst[i+1] = t1;
121
122 t0 = (AT)src[i+2]*src[i+2] + dst[i+2];
123 t1 = (AT)src[i+3]*src[i+3] + dst[i+3];

222template<typename T, typename AT> void
223accW_( const T* src, AT* dst, const uchar* mask, int len, int cn, double alpha )
224{
225 AT a = (AT)alpha, b = 1 - a;
226 int i = 0;
227
228 if( !mask )
229 {
230 len *= cn;
231 #if CV_ENABLE_UNROLLED
232 for( ; i <= len - 4; i += 4 )
233 {
234 AT t0, t1;
235 t0 = src[i]*a + dst[i]*b;
236 t1 = src[i+1]*a + dst[i+1]*b;
237 dst[i] = t0; dst[i+1] = t1;
238
239 t0 = src[i+2]*a + dst[i+2]*b;
240 t1 = src[i+3]*a + dst[i+3]*b;
241 dst[i+2] = t0; dst[i+3] = t1;
242 }
243 #endif
244 for( ; i < len; i++ )
245 dst[i] = src[i]*a + dst[i]*b;
246 }
247 else if( cn == 1 )
248 {
249 for( ; i < len; i++ )
250 {
251 if( mask[i] )
252 dst[i] = src[i]*a + dst[i]*b;
253 }
254 }
255 else if( cn == 3 )
256 {
257 for( ; i < len; i++, src += 3, dst += 3 )
258 {
259 if( mask[i] )
260 {
261 AT t0 = src[0]*a + dst[0]*b;
262 AT t1 = src[1]*a + dst[1]*b;
263 AT t2 = src[2]*a + dst[2]*b;
264
265 dst[0] = t0; dst[1] = t1; dst[2] = t2;
266 }
267 }
268 }
269 else
270 {
271 for( ; i < len; i++, src += cn, dst += cn )
272 if( mask[i] )
273 {
274 for( int k = 0; k < cn; k++ )
275 dst[k] = src[k]*a + dst[k]*b;
276 }
277 }

308typedef void (*AccFunc)(const uchar*, uchar*, const uchar*, int, int);
309typedef void (*AccProdFunc)(const uchar*, const uchar*, uchar*, const uchar*, int, int);
310typedef void (*AccWFunc)(const uchar*, uchar*, const uchar*, int, int, double);

312static AccFunc accTab[] =
313{
314 (AccFunc)acc_8u32f, (AccFunc)acc_8u64f,

344inline int getAccTabIdx(int sdepth, int ddepth)
345{
346 return sdepth == CV_8U && ddepth == CV_32F ? 0 :
347 sdepth == CV_8U && ddepth == CV_64F ? 1 :
348 sdepth == CV_16U && ddepth == CV_32F ? 2 :
349 sdepth == CV_16U && ddepth == CV_64F ? 3 :
350 sdepth == CV_32F && ddepth == CV_32F ? 4 :
351 sdepth == CV_32F && ddepth == CV_64F ? 5 :
352 sdepth == CV_64F && ddepth == CV_64F ? 6 : -1;
353}

357void cv::accumulate( InputArray _src, InputOutputArray _dst, InputArray _mask )
358{
359 Mat src = _src.getMat(), dst = _dst.getMat(), mask = _mask.getMat();
360 int sdepth = src.depth(), ddepth = dst.depth(), cn = src.channels();
361
362 CV_Assert( dst.size == src.size && dst.channels() == cn );
363 CV_Assert( mask.empty() || (mask.size == src.size && mask.type() == CV_8U) );
364
365 int fidx = getAccTabIdx(sdepth, ddepth);
366 AccFunc func = fidx >= 0 ? accTab[fidx] : 0;
367 CV_Assert( func != 0 );
368
369 const Mat* arrays[] = {&src, &dst, &mask, 0};
370 uchar* ptrs[3];
371 NAryMatIterator it(arrays, ptrs);
372 int len = (int)it.size;
373
374 for( size_t i = 0; i < it.nplanes; i++, ++it )
375 func(ptrs[0], ptrs[1], ptrs[2], len, cn);
376}

400void cv::accumulateProduct( InputArray _src1, InputArray _src2,
401 InputOutputArray _dst, InputArray _mask )
402{
403 Mat src1 = _src1.getMat(), src2 = _src2.getMat(), dst = _dst.getMat(), mask = _mask.getMat();
404 int sdepth = src1.depth(), ddepth = dst.depth(), cn = src1.channels();
405
406 CV_Assert( src2.size && src1.size && src2.type() == src1.type() );
407 CV_Assert( dst.size == src1.size && dst.channels() == cn );
408 CV_Assert( mask.empty() || (mask.size == src1.size && mask.type() == CV_8U) );
409
410 int fidx = getAccTabIdx(sdepth, ddepth);
411 AccProdFunc func = fidx >= 0 ? accProdTab[fidx] : 0;
412 CV_Assert( func != 0 );
413
414 const Mat* arrays[] = {&src1, &src2, &dst, &mask, 0};
415 uchar* ptrs[4];
416 NAryMatIterator it(arrays, ptrs);
417 int len = (int)it.size;
418
419 for( size_t i = 0; i < it.nplanes; i++, ++it )
420 func(ptrs[0], ptrs[1], ptrs[2], ptrs[3], len, cn);
421}

448cvAcc( const void* arr, void* sumarr, const void* maskarr )
