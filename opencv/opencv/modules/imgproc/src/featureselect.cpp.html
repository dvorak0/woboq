55void cv::goodFeaturesToTrack( InputArray _image, OutputArray _corners,
56 int maxCorners, double qualityLevel, double minDistance,
57 InputArray _mask, int blockSize,
58 bool useHarrisDetector, double harrisK )
59{
60 Mat image = _image.getMat(), mask = _mask.getMat();
61
62 CV_Assert( qualityLevel > 0 && minDistance >= 0 && maxCorners >= 0 );
63 CV_Assert( mask.empty() || (mask.type() == CV_8UC1 && mask.size() == image.size()) );
64
65 Mat eig, tmp;
66 if( useHarrisDetector )
67 cornerHarris( image, eig, blockSize, 3, harrisK );
68 else
69 cornerMinEigenVal( image, eig, blockSize, 3 );
70
71 double maxVal = 0;
72 minMaxLoc( eig, 0, &maxVal, 0, 0, mask );
73 threshold( eig, eig, maxVal*qualityLevel, 0, THRESH_TOZERO );
74 dilate( eig, tmp, Mat());
75
76 Size imgsize = image.size();
77
78 vector<const float*> tmpCorners;
79
80 // collect list of pointers to features - put them into temporary image
81 for( int y = 1; y < imgsize.height - 1; y++ )
82 {
83 const float* eig_data = (const float*)eig.ptr(y);
84 const float* tmp_data = (const float*)tmp.ptr(y);
85 const uchar* mask_data = mask.data ? mask.ptr(y) : 0;
86
87 for( int x = 1; x < imgsize.width - 1; x++ )
88 {
89 float val = eig_data[x];
90 if( val != 0 && val == tmp_data[x] && (!mask_data || mask_data[x]) )
91 tmpCorners.push_back(eig_data + x);
92 }
93 }
94
95 sort( tmpCorners, greaterThanPtr<float>() );
96 vector<Point2f> corners;
97 size_t i, j, total = tmpCorners.size(), ncorners = 0;
98
99 if(minDistance >= 1)
100 {
101 // Partition the image into larger grids
102 int w = image.cols;
103 int h = image.rows;
104
105 const int cell_size = cvRound(minDistance);
106 const int grid_width = (w + cell_size - 1) / cell_size;
107 const int grid_height = (h + cell_size - 1) / cell_size;
108
109 std::vector<std::vector<Point2f> > grid(grid_width*grid_height);
110
111 minDistance *= minDistance;
112
113 for( i = 0; i < total; i++ )
114 {
115 int ofs = (int)((const uchar*)tmpCorners[i] - eig.data);
116 int y = (int)(ofs / eig.step);
117 int x = (int)((ofs - y*eig.step)/sizeof(float));
118
119 bool good = true;
120
121 int x_cell = x / cell_size;
122 int y_cell = y / cell_size;
123
124 int x1 = x_cell - 1;
125 int y1 = y_cell - 1;
126 int x2 = x_cell + 1;
127 int y2 = y_cell + 1;
128
129 // boundary check
130 x1 = std::max(0, x1);
131 y1 = std::max(0, y1);
132 x2 = std::min(grid_width-1, x2);
133 y2 = std::min(grid_height-1, y2);
134
135 for( int yy = y1; yy <= y2; yy++ )
136 {
137 for( int xx = x1; xx <= x2; xx++ )
138 {
139 vector <Point2f> &m = grid[yy*grid_width + xx];
140
141 if( m.size() )
142 {
143 for(j = 0; j < m.size(); j++)
144 {
145 float dx = x - m[j].x;
146 float dy = y - m[j].y;
147
148 if( dx*dx + dy*dy < minDistance )
149 {
150 good = false;
151 goto break_out;
152 }
153 }
154 }
155 }
156 }
157
158 break_out:
159
160 if(good)
161 {
162 // printf("%d: %d %d -> %d %d, %d, %d -- %d %d %d %d, %d %d, c=%d\n",
163 // i,x, y, x_cell, y_cell, (int)minDistance, cell_size,x1,y1,x2,y2, grid_width,grid_height,c);
164 grid[y_cell*grid_width + x_cell].push_back(Point2f((float)x, (float)y));
165
166 corners.push_back(Point2f((float)x, (float)y));
167 ++ncorners;
168
169 if( maxCorners > 0 && (int)ncorners == maxCorners )
170 break;
171 }
172 }
173 }
174 else
175 {
176 for( i = 0; i < total; i++ )
177 {
178 int ofs = (int)((const uchar*)tmpCorners[i] - eig.data);
179 int y = (int)(ofs / eig.step);
180 int x = (int)((ofs - y*eig.step)/sizeof(float));
181
182 corners.push_back(Point2f((float)x, (float)y));
183 ++ncorners;
184 if( maxCorners > 0 && (int)ncorners == maxCorners )
185 break;
186 }
187 }
188
189 Mat(corners).convertTo(_corners, _corners.fixedType() ? _corners.type() : CV_32F);
190
191 /*
192 for( i = 0; i < total; i++ )
193 {
194 int ofs = (int)((const uchar*)tmpCorners[i] - eig.data);
195 int y = (int)(ofs / eig.step);
196 int x = (int)((ofs - y*eig.step)/sizeof(float));
197
198 if( minDistance > 0 )
199 {
200 for( j = 0; j < ncorners; j++ )
201 {
202 float dx = x - corners[j].x;
203 float dy = y - corners[j].y;
204 if( dx*dx + dy*dy < minDistance )
205 break;
206 }
207 if( j < ncorners )
208 continue;
209 }
210
211 corners.push_back(Point2f((float)x, (float)y));
212 ++ncorners;
213 if( maxCorners > 0 && (int)ncorners == maxCorners )
214 break;
215 }
216*/
217}

219CV_IMPL void
220cvGoodFeaturesToTrack( const void* _image, void*, void*,
221 CvPoint2D32f* _corners, int *_corner_count,
222 double quality_level, double min_distance,
223 const void* _maskImage, int block_size,
224 int use_harris, double harris_k )
225{
226 cv::Mat image = cv::cvarrToMat(_image), mask;
227 cv::vector<cv::Point2f> corners;
228
229 if( _maskImage )
230 mask = cv::cvarrToMat(_maskImage);
231
232 CV_Assert( _corners && _corner_count );
233 cv::goodFeaturesToTrack( image, corners, *_corner_count, quality_level,
234 min_distance, mask, block_size, use_harris != 0, harris_k );
235
236 size_t i, ncorners = corners.size();
237 for( i = 0; i < ncorners; i++ )
238 _corners[i] = corners[i];
239 *_corner_count = (int)ncorners;
240}
