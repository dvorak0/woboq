49 class CLAHE_CalcLut_Body : public cv::ParallelLoopBody
50 {
51 public:
52 CLAHE_CalcLut_Body(const cv::Mat& src, cv::Mat& lut, cv::Size tileSize, int tilesX, int clipLimit, float lutScale) :
53 src_(src), lut_(lut), tileSize_(tileSize), tilesX_(tilesX), clipLimit_(clipLimit), lutScale_(lutScale)
54 {
55 }
56
57 void operator ()(const cv::Range& range) const;
58
59 private:
60 cv::Mat src_;
61 mutable cv::Mat lut_;
62
63 cv::Size tileSize_;
64 int tilesX_;
65 int clipLimit_;
66 float lutScale_;
67 };
68
69 void CLAHE_CalcLut_Body::operator ()(const cv::Range& range) const
70 {
71 const int histSize = 256;
72
73 uchar* tileLut = lut_.ptr(range.start);
74 const size_t lut_step = lut_.step;
75
76 for (int k = range.start; k < range.end; ++k, tileLut += lut_step)
77 {
78 const int ty = k / tilesX_;
79 const int tx = k % tilesX_;
80
81 // retrieve tile submatrix
82
83 cv::Rect tileROI;
84 tileROI.x = tx * tileSize_.width;
85 tileROI.y = ty * tileSize_.height;
86 tileROI.width = tileSize_.width;
87 tileROI.height = tileSize_.height;
88
89 const cv::Mat tile = src_(tileROI);
90
91 // calc histogram
92
93 int tileHist[histSize] = {0, };
94
95 int height = tileROI.height;
96 const size_t sstep = tile.step;
97 for (const uchar* ptr = tile.ptr<uchar>(0); height--; ptr += sstep)
98 {
99 int x = 0;
100 for (; x <= tileROI.width - 4; x += 4)
101 {
102 int t0 = ptr[x], t1 = ptr[x+1];
103 tileHist[t0]++; tileHist[t1]++;
104 t0 = ptr[x+2]; t1 = ptr[x+3];
105 tileHist[t0]++; tileHist[t1]++;
106 }
107
108 for (; x < tileROI.width; ++x)
109 tileHist[ptr[x]]++;
110 }
111
112 // clip histogram
113
114 if (clipLimit_ > 0)
115 {
116 // how many pixels were clipped
117 int clipped = 0;
118 for (int i = 0; i < histSize; ++i)
119 {
120 if (tileHist[i] > clipLimit_)
121 {
122 clipped += tileHist[i] - clipLimit_;
123 tileHist[i] = clipLimit_;
124 }
125 }
126
127 // redistribute clipped pixels
128 int redistBatch = clipped / histSize;
129 int residual = clipped - redistBatch * histSize;
130
131 for (int i = 0; i < histSize; ++i)
132 tileHist[i] += redistBatch;
133
134 for (int i = 0; i < residual; ++i)
135 tileHist[i]++;
136 }
137
138 // calc Lut
139
140 int sum = 0;
141 for (int i = 0; i < histSize; ++i)
142 {
143 sum += tileHist[i];
144 tileLut[i] = cv::saturate_cast<uchar>(sum * lutScale_);
145 }
146 }
147 }

149 class CLAHE_Interpolation_Body : public cv::ParallelLoopBody
150 {
151 public:
152 CLAHE_Interpolation_Body(const cv::Mat& src, cv::Mat& dst, const cv::Mat& lut, cv::Size tileSize, int tilesX, int tilesY) :
153 src_(src), dst_(dst), lut_(lut), tileSize_(tileSize), tilesX_(tilesX), tilesY_(tilesY)
154 {
155 }
156
157 void operator ()(const cv::Range& range) const;
158
159 private:
160 cv::Mat src_;
161 mutable cv::Mat dst_;
162 cv::Mat lut_;
163
164 cv::Size tileSize_;
165 int tilesX_;
166 int tilesY_;
167 };
168
169 void CLAHE_Interpolation_Body::operator ()(const cv::Range& range) const
170 {
171 const size_t lut_step = lut_.step;
172
173 for (int y = range.start; y < range.end; ++y)
174 {
175 const uchar* srcRow = src_.ptr<uchar>(y);
176 uchar* dstRow = dst_.ptr<uchar>(y);
177
178 const float tyf = (static_cast<float>(y) / tileSize_.height) - 0.5f;
179
180 int ty1 = cvFloor(tyf);
181 int ty2 = ty1 + 1;
182
183 const float ya = tyf - ty1;
184
185 ty1 = std::max(ty1, 0);
186 ty2 = std::min(ty2, tilesY_ - 1);
187
188 const uchar* lutPlane1 = lut_.ptr(ty1 * tilesX_);
189 const uchar* lutPlane2 = lut_.ptr(ty2 * tilesX_);
190
191 for (int x = 0; x < src_.cols; ++x)
192 {
193 const float txf = (static_cast<float>(x) / tileSize_.width) - 0.5f;
194
195 int tx1 = cvFloor(txf);
196 int tx2 = tx1 + 1;
197
198 const float xa = txf - tx1;
199
200 tx1 = std::max(tx1, 0);
201 tx2 = std::min(tx2, tilesX_ - 1);
202
203 const int srcVal = srcRow[x];
204
205 const size_t ind1 = tx1 * lut_step + srcVal;
206 const size_t ind2 = tx2 * lut_step + srcVal;
207
208 float res = 0;
209
210 res += lutPlane1[ind1] * ((1.0f - xa) * (1.0f - ya));
211 res += lutPlane1[ind2] * ((xa) * (1.0f - ya));
212 res += lutPlane2[ind1] * ((1.0f - xa) * (ya));
213 res += lutPlane2[ind2] * ((xa) * (ya));
214
215 dstRow[x] = cv::saturate_cast<uchar>(res);
216 }
217 }
218 }

220 class CLAHE_Impl : public cv::CLAHE
221 {

223 CLAHE_Impl(double clipLimit = 40.0, int tilesX = 8, int tilesY = 8);
224
225 cv::AlgorithmInfo* info() const;

227 void apply(cv::InputArray src, cv::OutputArray dst);
228
229 void setClipLimit(double clipLimit);
230 double getClipLimit() const;
231
232 void setTilesGridSize(cv::Size tileGridSize);
233 cv::Size getTilesGridSize() const;
234
235 void collectGarbage();
236
237 private:
238 double clipLimit_;
239 int tilesX_;
240 int tilesY_;
241
242 cv::Mat srcExt_;
243 cv::Mat lut_;
244 };
245
246 CLAHE_Impl::CLAHE_Impl(double clipLimit, int tilesX, int tilesY) :
247 clipLimit_(clipLimit), tilesX_(tilesX), tilesY_(tilesY)
248 {
249 }
250
251 CV_INIT_ALGORITHM(CLAHE_Impl, "CLAHE",
252 obj.info()->addParam(obj, "clipLimit", obj.clipLimit_);
253 obj.info()->addParam(obj, "tilesX", obj.tilesX_);
254 obj.info()->addParam(obj, "tilesY", obj.tilesY_))
255
256 void CLAHE_Impl::apply(cv::InputArray _src, cv::OutputArray _dst)
257 {
258 cv::Mat src = _src.getMat();
259
260 CV_Assert( src.type() == CV_8UC1 );
261
262 _dst.create( src.size(), src.type() );
263 cv::Mat dst = _dst.getMat();
264
265 const int histSize = 256;
266
267 lut_.create(tilesX_ * tilesY_, histSize, CV_8UC1);
268
269 cv::Size tileSize;
270 cv::Mat srcForLut;
271
272 if (src.cols % tilesX_ == 0 && src.rows % tilesY_ == 0)
273 {
274 tileSize = cv::Size(src.cols / tilesX_, src.rows / tilesY_);
275 srcForLut = src;
276 }
277 else
278 {
279 cv::copyMakeBorder(src, srcExt_, 0, tilesY_ - (src.rows % tilesY_), 0, tilesX_ - (src.cols % tilesX_), cv::BORDER_REFLECT_101);
280
281 tileSize = cv::Size(srcExt_.cols / tilesX_, srcExt_.rows / tilesY_);
282 srcForLut = srcExt_;
283 }
284
285 const int tileSizeTotal = tileSize.area();
286 const float lutScale = static_cast<float>(histSize - 1) / tileSizeTotal;
287
288 int clipLimit = 0;
289 if (clipLimit_ > 0.0)
290 {
291 clipLimit = static_cast<int>(clipLimit_ * tileSizeTotal / histSize);
292 clipLimit = std::max(clipLimit, 1);
293 }
294
295 CLAHE_CalcLut_Body calcLutBody(srcForLut, lut_, tileSize, tilesX_, clipLimit, lutScale);
296 cv::parallel_for_(cv::Range(0, tilesX_ * tilesY_), calcLutBody);
297
298 CLAHE_Interpolation_Body interpolationBody(src, dst, lut_, tileSize, tilesX_, tilesY_);
299 cv::parallel_for_(cv::Range(0, src.rows), interpolationBody);
300 }

302 void CLAHE_Impl::setClipLimit(double clipLimit)
303 {
304 clipLimit_ = clipLimit;
305 }
306
307 double CLAHE_Impl::getClipLimit() const

310 }

312 void CLAHE_Impl::setTilesGridSize(cv::Size tileGridSize)
313 {
314 tilesX_ = tileGridSize.width;
315 tilesY_ = tileGridSize.height;
316 }
317
318 cv::Size CLAHE_Impl::getTilesGridSize() const
319 {
320 return cv::Size(tilesX_, tilesY_);
321 }
322
323 void CLAHE_Impl::collectGarbage()
324 {
325 srcExt_.release();
326 lut_.release();
327 }
328}
