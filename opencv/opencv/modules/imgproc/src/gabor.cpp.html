45/*
46 Gabor filters and such. To be greatly extended to have full texture analysis.
47 For the formulas and the explanation of the parameters see:
48 http://en.wikipedia.org/wiki/Gabor_filter
49*/
50
51cv::Mat cv::getGaborKernel( Size ksize, double sigma, double theta,
52 double lambd, double gamma, double psi, int ktype )
53{
54 double sigma_x = sigma;
55 double sigma_y = sigma/gamma;
56 int nstds = 3;
57 int xmin, xmax, ymin, ymax;
58 double c = cos(theta), s = sin(theta);
59
60 if( ksize.width > 0 )
61 xmax = ksize.width/2;
62 else
63 xmax = cvRound(std::max(fabs(nstds*sigma_x*c), fabs(nstds*sigma_y*s)));
64
65 if( ksize.height > 0 )
66 ymax = ksize.height/2;
67 else
68 ymax = cvRound(std::max(fabs(nstds*sigma_x*s), fabs(nstds*sigma_y*c)));
69
70 xmin = -xmax;
71 ymin = -ymax;
72
73 CV_Assert( ktype == CV_32F || ktype == CV_64F );
74
75 Mat kernel(ymax - ymin + 1, xmax - xmin + 1, ktype);
76 double scale = 1;
77 double ex = -0.5/(sigma_x*sigma_x);
78 double ey = -0.5/(sigma_y*sigma_y);
79 double cscale = CV_PI*2/lambd;
80
81 for( int y = ymin; y <= ymax; y++ )
82 for( int x = xmin; x <= xmax; x++ )
83 {
84 double xr = x*c + y*s;
85 double yr = -x*s + y*c;
86
87 double v = scale*exp(ex*xr*xr + ey*yr*yr)*cos(cscale*xr + psi);
88 if( ktype == CV_32F )
89 kernel.at<float>(ymax - y, xmax - x) = (float)v;
90 else
91 kernel.at<double>(ymax - y, xmax - x) = v;
92 }
93
94 return kernel;
95}
