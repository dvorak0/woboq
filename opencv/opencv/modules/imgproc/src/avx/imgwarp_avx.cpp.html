47int VResizeLinearVec_32f_avx(const uchar** _src, uchar* _dst, const uchar* _beta, int width )
48{
49 const float** src = (const float**)_src;
50 const float* beta = (const float*)_beta;
51 const float *S0 = src[0], *S1 = src[1];
52 float* dst = (float*)_dst;
53 int x = 0;
54
55 __m256 b0 = _mm256_set1_ps(beta[0]), b1 = _mm256_set1_ps(beta[1]);
56
57 if( (((size_t)S0|(size_t)S1)&31) == 0 )
58 for( ; x <= width - 16; x += 16 )
59 {
60 __m256 x0, x1, y0, y1;
61 x0 = _mm256_load_ps(S0 + x);
62 x1 = _mm256_load_ps(S0 + x + 8);
63 y0 = _mm256_load_ps(S1 + x);
64 y1 = _mm256_load_ps(S1 + x + 8);
65
66 x0 = _mm256_add_ps(_mm256_mul_ps(x0, b0), _mm256_mul_ps(y0, b1));
67 x1 = _mm256_add_ps(_mm256_mul_ps(x1, b0), _mm256_mul_ps(y1, b1));
68
69 _mm256_storeu_ps( dst + x, x0);
70 _mm256_storeu_ps( dst + x + 8, x1);
71 }
72 else
73 for( ; x <= width - 16; x += 16 )
74 {
75 __m256 x0, x1, y0, y1;
76 x0 = _mm256_loadu_ps(S0 + x);
77 x1 = _mm256_loadu_ps(S0 + x + 8);
78 y0 = _mm256_loadu_ps(S1 + x);
79 y1 = _mm256_loadu_ps(S1 + x + 8);
80
81 x0 = _mm256_add_ps(_mm256_mul_ps(x0, b0), _mm256_mul_ps(y0, b1));
82 x1 = _mm256_add_ps(_mm256_mul_ps(x1, b0), _mm256_mul_ps(y1, b1));
83
84 _mm256_storeu_ps( dst + x, x0);
85 _mm256_storeu_ps( dst + x + 8, x1);
86 }
87
88 return x;
89}

91int VResizeCubicVec_32f_avx(const uchar** _src, uchar* _dst, const uchar* _beta, int width )
92{
93 const float** src = (const float**)_src;
94 const float* beta = (const float*)_beta;
95 const float *S0 = src[0], *S1 = src[1], *S2 = src[2], *S3 = src[3];
96 float* dst = (float*)_dst;
97 int x = 0;
98 __m256 b0 = _mm256_set1_ps(beta[0]), b1 = _mm256_set1_ps(beta[1]),
99 b2 = _mm256_set1_ps(beta[2]), b3 = _mm256_set1_ps(beta[3]);
100
101 if( (((size_t)S0|(size_t)S1|(size_t)S2|(size_t)S3)&31) == 0 )
102 for( ; x <= width - 16; x += 16 )
103 {
104 __m256 x0, x1, y0, y1, s0, s1;
105 x0 = _mm256_load_ps(S0 + x);
106 x1 = _mm256_load_ps(S0 + x + 8);
107 y0 = _mm256_load_ps(S1 + x);
108 y1 = _mm256_load_ps(S1 + x + 8);
109
110 s0 = _mm256_mul_ps(x0, b0);
111 s1 = _mm256_mul_ps(x1, b0);
112 y0 = _mm256_mul_ps(y0, b1);
113 y1 = _mm256_mul_ps(y1, b1);
114 s0 = _mm256_add_ps(s0, y0);
115 s1 = _mm256_add_ps(s1, y1);
116
117 x0 = _mm256_load_ps(S2 + x);
118 x1 = _mm256_load_ps(S2 + x + 8);
119 y0 = _mm256_load_ps(S3 + x);
120 y1 = _mm256_load_ps(S3 + x + 8);
121
122 x0 = _mm256_mul_ps(x0, b2);
123 x1 = _mm256_mul_ps(x1, b2);
124 y0 = _mm256_mul_ps(y0, b3);
125 y1 = _mm256_mul_ps(y1, b3);
126 s0 = _mm256_add_ps(s0, x0);
127 s1 = _mm256_add_ps(s1, x1);
128 s0 = _mm256_add_ps(s0, y0);
129 s1 = _mm256_add_ps(s1, y1);
130
131 _mm256_storeu_ps( dst + x, s0);
132 _mm256_storeu_ps( dst + x + 8, s1);
133 }
134 else
135 for( ; x <= width - 16; x += 16 )
136 {
137 __m256 x0, x1, y0, y1, s0, s1;
138 x0 = _mm256_loadu_ps(S0 + x);
139 x1 = _mm256_loadu_ps(S0 + x + 8);
140 y0 = _mm256_loadu_ps(S1 + x);
141 y1 = _mm256_loadu_ps(S1 + x + 8);
142
143 s0 = _mm256_mul_ps(x0, b0);
144 s1 = _mm256_mul_ps(x1, b0);
145 y0 = _mm256_mul_ps(y0, b1);
146 y1 = _mm256_mul_ps(y1, b1);
147 s0 = _mm256_add_ps(s0, y0);
148 s1 = _mm256_add_ps(s1, y1);
149
150 x0 = _mm256_loadu_ps(S2 + x);
151 x1 = _mm256_loadu_ps(S2 + x + 8);
152 y0 = _mm256_loadu_ps(S3 + x);
153 y1 = _mm256_loadu_ps(S3 + x + 8);
154
155 x0 = _mm256_mul_ps(x0, b2);
156 x1 = _mm256_mul_ps(x1, b2);
157 y0 = _mm256_mul_ps(y0, b3);
158 y1 = _mm256_mul_ps(y1, b3);
159 s0 = _mm256_add_ps(s0, x0);
160 s1 = _mm256_add_ps(s1, x1);
161 s0 = _mm256_add_ps(s0, y0);
162 s1 = _mm256_add_ps(s1, y1);
163
164 _mm256_storeu_ps( dst + x, s0);
165 _mm256_storeu_ps( dst + x + 8, s1);
166 }
167
168 return x;
169}
