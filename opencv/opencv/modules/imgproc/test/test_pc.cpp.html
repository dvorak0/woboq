51/// phase correlation
52class CV_PhaseCorrelatorTest : public cvtest::ArrayTest
53{
54public:
55 CV_PhaseCorrelatorTest();
56protected:
57 void run( int );
58};
59
60CV_PhaseCorrelatorTest::CV_PhaseCorrelatorTest() {}
61
62void CV_PhaseCorrelatorTest::run( int )
63{
64 ts->set_failed_test_info(cvtest::TS::OK);
65
66 Mat r1 = Mat::ones(Size(129, 128), CV_64F);
67 Mat r2 = Mat::ones(Size(129, 128), CV_64F);
68
69 double expectedShiftX = -10.0;
70 double expectedShiftY = -20.0;
71
72 // draw 10x10 rectangles @ (100, 100) and (90, 80) should see ~(-10, -20) shift here...
73 cv::rectangle(r1, Point(100, 100), Point(110, 110), Scalar(0, 0, 0), CV_FILLED);
74 cv::rectangle(r2, Point(90, 80), Point(100, 90), Scalar(0, 0, 0), CV_FILLED);
75
76 Mat hann;
77 createHanningWindow(hann, r1.size(), CV_64F);
78 Point2d phaseShift = phaseCorrelate(r1, r2, hann);
79
80 // test accuracy should be less than 1 pixel...
81 if((expectedShiftX - phaseShift.x) >= 1 || (expectedShiftY - phaseShift.y) >= 1)
82 {
83 ts->set_failed_test_info( cvtest::TS::FAIL_BAD_ACCURACY );
84 }
85}
86
87TEST(Imgproc_PhaseCorrelatorTest, accuracy) { CV_PhaseCorrelatorTest test; test.safe_run(); }
