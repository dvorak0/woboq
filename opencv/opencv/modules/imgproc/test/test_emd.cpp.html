49class CV_EMDTest : public cvtest::BaseTest
50{
51public:
52 CV_EMDTest();
53protected:
54 void run(int);
55};
56
57
58CV_EMDTest::CV_EMDTest()
59{
60}
61
62void CV_EMDTest::run( int )
63{
64 int code = cvtest::TS::OK;
65 const double success_error_level = 1e-6;
66 #define M 10000
67 double emd0 = 2460./210;
68 static float cost[] =
69 {
70 16, 16, 13, 22, 17,
71 14, 14, 13, 19, 15,
72 19, 19, 20, 23, M,
73 M , 0, M, 0, 0
74 };
75 static float w1[] = { 50, 60, 50, 50 },
76 w2[] = { 30, 20, 70, 30, 60 };
77 Mat _w1(4, 1, CV_32F, w1);
78 Mat _w2(5, 1, CV_32F, w2);
79 Mat _cost(_w1.rows, _w2.rows, CV_32F, cost);
80
81 float emd = EMD( _w1, _w2, -1, _cost );
82 if( fabs( emd - emd0 ) > success_error_level*emd0 )
83 {
84 ts->printf( cvtest::TS::LOG,
85 "The computed distance is %.2f, while it should be %.2f\n", emd, emd0 );
86 code = cvtest::TS::FAIL_BAD_ACCURACY;
87 }
88
89 if( code < 0 )
90 ts->set_failed_test_info( code );
91}
92
93TEST(Imgproc_EMD, regression) { CV_EMDTest test; test.safe_run(); }
