74CV_INLINE double
75cvTsDist( CvPoint2D32f a, CvPoint2D32f b )
76{
77 double dx = a.x - b.x;
78 double dy = a.y - b.y;
79 return sqrt(dx*dx + dy*dy);
80}
81
82CV_INLINE double
83cvTsPtLineDist( CvPoint2D32f pt, CvPoint2D32f a, CvPoint2D32f b )
84{
85 double d0 = cvTsDist( pt, a ), d1;
86 double dd = cvTsDist( a, b );
87 if( dd < FLT_EPSILON )
88 return d0;
89 d1 = cvTsDist( pt, b );
90 dd = fabs((double)(pt.x - a.x)*(b.y - a.y) - (double)(pt.y - a.y)*(b.x - a.x))/dd;
91 d0 = MIN( d0, d1 );
92 return MIN( d0, dd );
93}

95static double
96cvTsPointPolygonTest( CvPoint2D32f pt, const CvPoint2D32f* vv, int n, int* _idx=0, int* _on_edge=0 )
97{
98 int i;
99 CvPoint2D32f v = vv[n-1], v0;
100 double min_dist_num = FLT_MAX, min_dist_denom = 1;
101 int min_dist_idx = -1, min_on_edge = 0;
102 int counter = 0;
103 double result;
104
105 for( i = 0; i < n; i++ )
106 {
107 double dx, dy, dx1, dy1, dx2, dy2, dist_num, dist_denom = 1;
108 int on_edge = 0, idx = i;
109
110 v0 = v; v = vv[i];
111 dx = v.x - v0.x; dy = v.y - v0.y;
112 dx1 = pt.x - v0.x; dy1 = pt.y - v0.y;
113 dx2 = pt.x - v.x; dy2 = pt.y - v.y;
114
115 if( dx2*dx + dy2*dy >= 0 )
116 dist_num = dx2*dx2 + dy2*dy2;
117 else if( dx1*dx + dy1*dy <= 0 )
118 {
119 dist_num = dx1*dx1 + dy1*dy1;
120 idx = i - 1;
121 if( idx < 0 ) idx = n-1;
122 }
123 else
124 {
125 dist_num = (dy1*dx - dx1*dy);
126 dist_num *= dist_num;
127 dist_denom = dx*dx + dy*dy;
128 on_edge = 1;
129 }
130
131 if( dist_num*min_dist_denom < min_dist_num*dist_denom )
132 {
133 min_dist_num = dist_num;
134 min_dist_denom = dist_denom;
135 min_dist_idx = idx;
136 min_on_edge = on_edge;
137 if( min_dist_num == 0 )
138 break;
139 }
140
141 if( (v0.y <= pt.y && v.y <= pt.y) ||
142 (v0.y > pt.y && v.y > pt.y) ||
143 (v0.x < pt.x && v.x < pt.x) )
144 continue;
145
146 dist_num = dy1*dx - dx1*dy;
147 if( dy < 0 )
148 dist_num = -dist_num;
149 counter += dist_num > 0;
150 }
151
152 result = sqrt(min_dist_num/min_dist_denom);
153 if( counter % 2 == 0 )
154 result = -result;
155
156 if( _idx )
157 *_idx = min_dist_idx;
158 if( _on_edge )
159 *_on_edge = min_on_edge;
160
161 return result;
162}

319{
320 int size;
321 int use_storage = 0;
322 int point_type;
323 int i;
324 RNG& rng = ts->get_rng();
325
326 cvtest::BaseTest::prepare_test_case( test_case_idx );
327
328 clear();
329 size = cvRound( exp((cvtest::randReal(rng) * (max_log_size - min_log_size) + min_log_size)*CV_LOG2) );
330 use_storage = cvtest::randInt(rng) % 2;
331 point_type = CV_MAKETYPE(cvtest::randInt(rng) %
332 (enable_flt_points ? 2 : 1) ? CV_32F : CV_32S, dims);
333
334 if( use_storage )

336 storage = cvCreateMemStorage( (cvtest::randInt(rng)%10 + 1)*1024 );
337 points1 = cvCreateSeq( point_type, sizeof(CvSeq), CV_ELEM_SIZE(point_type), storage );
338 cvSeqPushMulti( points1, 0, size );
339 points = points1;
340 }
341 else
342 {
343 int rows = 1, cols = size;
344 if( cvtest::randInt(rng) % 2 )
345 rows = size, cols = 1;
346
347 points2 = cvCreateMat( rows, cols, point_type );
348 points = points2;
349 }
350
351 for( i = 0; i < 4; i++ )
352 {
353 low.val[i] = (cvtest::randReal(rng)-0.5)*low_high_range*2;
354 high.val[i] = (cvtest::randReal(rng)-0.5)*low_high_range*2;
355 if( low.val[i] > high.val[i] )
356 {
357 double t;
358 CV_SWAP( low.val[i], high.val[i], t );
359 }
360 if( high.val[i] < low.val[i] + 1 )
361 high.val[i] += 1;
362 }
363
364 generate_point_set( points );
365
366 test_cpp = (cvtest::randInt(rng) & 16) == 0;
367 return 1;
368}

524int CV_ConvHullTest::validate_test_results( int test_case_idx )
525{
526 int code = CV_BaseShapeDescrTest::validate_test_results( test_case_idx );
527 CvMat* hull = 0;
528 CvMat* mask = 0;
529 int i, point_count, hull_count;
530 CvPoint2D32f *p, *h;
531 CvSeq header, hheader, *ptseq, *hseq;
532 CvSeqBlock block, hblock;
533
534 if( points1 )
535 ptseq = points1;
536 else
537 ptseq = cvMakeSeqHeaderForArray( CV_MAT_TYPE(points2->type),
538 sizeof(CvSeq), CV_ELEM_SIZE(points2->type), points2->data.ptr,
539 points2->rows + points2->cols - 1, &header, &block );
540 point_count = ptseq->total;
541 p = (CvPoint2D32f*)(points2->data.ptr);
542
543 if( hull1 )
544 hseq = hull1;
545 else
546 hseq = cvMakeSeqHeaderForArray( CV_MAT_TYPE(hull2->type),
547 sizeof(CvSeq), CV_ELEM_SIZE(hull2->type), hull2->data.ptr,
548 hull2->rows + hull2->cols - 1, &hheader, &hblock );
549 hull_count = hseq->total;
550 hull = cvCreateMat( 1, hull_count, CV_32FC2 );
551 mask = cvCreateMat( 1, hull_count, CV_8UC1 );
552 cvZero( mask );
553 h = (CvPoint2D32f*)(hull->data.ptr);
554
555 // extract convex hull points
556 if( return_points )
557 {
558 cvCvtSeqToArray( hseq, hull->data.ptr );
559 if( CV_SEQ_ELTYPE(hseq) != CV_32FC2 )
560 {
561 CvMat tmp = cvMat( hull->rows, hull->cols, CV_32SC2, hull->data.ptr );
562 cvConvert( &tmp, hull );
563 }
564 }
565 else
566 {
567 CvSeqReader reader;
568 cvStartReadSeq( hseq, &reader );
569
570 for( i = 0; i < hull_count; i++ )
571 {
572 schar* ptr = reader.ptr;
573 int idx;
574 CV_NEXT_SEQ_ELEM( hseq->elem_size, reader );
575
576 if( hull1 )
577 idx = cvSeqElemIdx( ptseq, *(uchar**)ptr );
578 else
579 idx = *(int*)ptr;
580
581 if( idx < 0 || idx >= point_count )
582 {
583 ts->printf( cvtest::TS::LOG, "Invalid convex hull point #%d\n", i );
584 code = cvtest::TS::FAIL_INVALID_OUTPUT;
585 goto _exit_;
586 }
587 h[i] = p[idx];
588 }
589 }
590
591 // check that the convex hull is a convex polygon
592 if( hull_count >= 3 )
593 {
594 CvPoint2D32f pt0 = h[hull_count-1];
595 for( i = 0; i < hull_count; i++ )
596 {
597 int j = i+1;
598 CvPoint2D32f pt1 = h[i], pt2 = h[j < hull_count ? j : 0];
599 float dx0 = pt1.x - pt0.x, dy0 = pt1.y - pt0.y;
600 float dx1 = pt2.x - pt1.x, dy1 = pt2.y - pt1.y;
601 double t = (double)dx0*dy1 - (double)dx1*dy0;
602 if( (t < 0) ^ (orientation != CV_COUNTER_CLOCKWISE) )
603 {
604 ts->printf( cvtest::TS::LOG, "The convex hull is not convex or has a wrong orientation (vtx %d)\n", i );
605 code = cvtest::TS::FAIL_INVALID_OUTPUT;
606 goto _exit_;
607 }
608 pt0 = pt1;
609 }
610 }
611
612 // check that all the points are inside the hull or on the hull edge
613 // and at least hull_point points are at the hull vertices
614 for( i = 0; i < point_count; i++ )
615 {
616 int idx = 0, on_edge = 0;
617 double pptresult = cvTsPointPolygonTest( p[i], h, hull_count, &idx, &on_edge );
618
619 if( pptresult < 0 )
620 {
621 ts->printf( cvtest::TS::LOG, "The point #%d is outside of the convex hull\n", i );
622 code = cvtest::TS::FAIL_BAD_ACCURACY;
623 goto _exit_;
624 }
625
626 if( pptresult < FLT_EPSILON && !on_edge )
627 mask->data.ptr[idx] = (uchar)1;
628 }
629
630 if( cvNorm( mask, 0, CV_L1 ) != hull_count )
631 {
632 ts->printf( cvtest::TS::LOG, "Not every convex hull vertex coincides with some input point\n" );
633 code = cvtest::TS::FAIL_BAD_ACCURACY;
634 goto _exit_;
635 }
636
637_exit_:
638
639 cvReleaseMat( &hull );
640 cvReleaseMat( &mask );
641 if( code < 0 )
642 ts->set_failed_test_info( code );
643 return code;
644}

1360 line0[k] = -line0[k];
1361 if( line[max_k] < 0 )
1362 for( k = 0; k < dims; k++ )
1363 line[k] = -line[k];
1364
1365 for( k = 0; k < dims; k++ )
1366 {
1367 double dt = line[k] - line0[k];

1369 }

1383 }
1384

1386 for( k = 0; k < dims; k++ )
1387 {
1388 double p = line0[k+dims] + t*line0[k] - line[k+dims];
1389 vec_diff += p*p;
1390 }
1391
1392 if( sqrt(vec_diff) > 1*MAX(fabs(t),1) )
1393 {
1394 if( dims == 2 )
1395 ts->printf( cvtest::TS::LOG,
1396 "The computed line point (%.2f,%.2f) is too far from the actual line\n",

1398 else

1400 "The computed line point (%.2f,%.2f,%.2f) is too far from the actual line\n",

1402 code = cvtest::TS::FAIL_BAD_ACCURACY;
1403 goto _exit_;
1404 }
1405
1406_exit_:
1407
1408 if( code < 0 )
1409 {
1410 ts->set_failed_test_info( code );
1411 }
1412 return code;
