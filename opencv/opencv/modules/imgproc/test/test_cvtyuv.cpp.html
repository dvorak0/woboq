12int countOfDifferencies(const Mat& gold, const Mat& result, int maxAllowedDifference = 1)
13{
14 Mat diff;
15 absdiff(gold, result, diff);
16 return countNonZero(diff.reshape(1) > maxAllowedDifference);

319class YUV2RGB_Converter
320{
321public:
322 RGB convert(YUV yuv)
323 {
324 int y = std::max(0, yuv[0] - 16);
325 int u = yuv[1] - 128;
326 int v = yuv[2] - 128;
327 uchar r = saturate_cast<uchar>(1.164f * y + 1.596f * v);
328 uchar g = saturate_cast<uchar>(1.164f * y - 0.813f * v - 0.391f * u);
329 uchar b = saturate_cast<uchar>(1.164f * y + 2.018f * u);
330
331 return RGB(r, g, b);
332 }

345{
346public:
347 YUV convert(RGB rgb)
348 {
349 int r = rgb[0];
350 int g = rgb[1];
351 int b = rgb[2];
352
353 uchar y = saturate_cast<uchar>((int)( 0.257f*r + 0.504f*g + 0.098f*b + 0.5f) + 16);

628TEST_P(Imgproc_ColorYUV, accuracy)
629{
630 int code = GetParam();
631 RNG& random = theRNG();
632
633 ConversionYUV cvt(code);
634
635 const int scn = cvt.getScn();
636 const int dcn = cvt.getDcn();
637 for(int iter = 0; iter < 30; ++iter)
638 {
639 Size sz(random.uniform(1, 641), random.uniform(1, 481));
640
641 if(cvt.requiresEvenWidth()) sz.width += sz.width % 2;
642 if(cvt.requiresEvenHeight()) sz.height += sz.height % 2;
643
644 Size srcSize = cvt.getSrcSize(sz);
645 Mat src = Mat(srcSize.height, srcSize.width * scn, CV_8UC1).reshape(scn);
646
647 Size dstSize = cvt.getDstSize(sz);
648 Mat dst = Mat(dstSize.height, dstSize.width * dcn, CV_8UC1).reshape(dcn);
649 Mat gold(dstSize, CV_8UC(dcn));
650
651 random.fill(src, RNG::UNIFORM, 0, 256);
652
653 if(cvt.rgbWriter_)
654 referenceYUV2RGB<YUV2RGB_Converter> (src, gold, cvt.yuvReader_, cvt.rgbWriter_);
655 else if(cvt.grayWriter_)
656 referenceYUV2GRAY<YUV2GRAY_Converter>(src, gold, cvt.yuvReader_, cvt.grayWriter_);
657 else if(cvt.yuvWriter_)
658 referenceRGB2YUV<RGB2YUV_Converter> (src, gold, cvt.rgbReader_, cvt.yuvWriter_);
659
660 cv::cvtColor(src, dst, code, -1);
661
662 EXPECT_EQ(0, countOfDifferencies(gold, dst));
663 }
664}
