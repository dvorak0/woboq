114int CV_FindContourTest::read_params( CvFileStorage* fs )
115{
116 int t;
117 int code = cvtest::BaseTest::read_params( fs );
118
119 if( code < 0 )
120 return code;
121
122 min_blob_size = cvReadInt( find_param( fs, "min_blob_size" ), min_blob_size );
123 max_blob_size = cvReadInt( find_param( fs, "max_blob_size" ), max_blob_size );
124 max_log_blob_count = cvReadInt( find_param( fs, "max_log_blob_count" ), max_log_blob_count );
125 min_log_img_size = cvReadInt( find_param( fs, "min_log_img_size" ), min_log_img_size );
126 max_log_img_size = cvReadInt( find_param( fs, "max_log_img_size" ), max_log_img_size );
127
128 min_blob_size = cvtest::clipInt( min_blob_size, 1, 100 );
129 max_blob_size = cvtest::clipInt( max_blob_size, 1, 100 );
130
131 if( min_blob_size > max_blob_size )
132 CV_SWAP( min_blob_size, max_blob_size, t );
133
134 max_log_blob_count = cvtest::clipInt( max_log_blob_count, 1, 10 );
135
136 min_log_img_size = cvtest::clipInt( min_log_img_size, 1, 10 );
137 max_log_img_size = cvtest::clipInt( max_log_img_size, 1, 10 );
138
139 if( min_log_img_size > max_log_img_size )
140 CV_SWAP( min_log_img_size, max_log_img_size, t );
141
142 return 0;
143}

146static void
147cvTsGenerateBlobImage( IplImage* img, int min_blob_size, int max_blob_size,
148 int blob_count, int min_brightness, int max_brightness,
149 RNG& rng )
150{
151 int i;
152 CvSize size;
153
154 assert( img->depth == IPL_DEPTH_8U && img->nChannels == 1 );
155
156 cvZero( img );
157
158 // keep the border clear
159 cvSetImageROI( img, cvRect(1,1,img->width-2,img->height-2) );
160 size = cvGetSize( img );
161
162 for( i = 0; i < blob_count; i++ )
163 {
164 CvPoint center;
165 CvSize axes;
166 int angle = cvtest::randInt(rng) % 180;
167 int brightness = cvtest::randInt(rng) %
168 (max_brightness - min_brightness) + min_brightness;
169 center.x = cvtest::randInt(rng) % size.width;
170 center.y = cvtest::randInt(rng) % size.height;
171
172 axes.width = (cvtest::randInt(rng) %
173 (max_blob_size - min_blob_size) + min_blob_size + 1)/2;
174 axes.height = (cvtest::randInt(rng) %
175 (max_blob_size - min_blob_size) + min_blob_size + 1)/2;
176
177 cvEllipse( img, center, axes, angle, 0, 360, cvScalar(brightness), CV_FILLED );
178 }
179
180 cvResetImageROI( img );
181}

184static void
185cvTsMarkContours( IplImage* img, int val )
186{
187 int i, j;
188 int step = img->widthStep;
189
190 assert( img->depth == IPL_DEPTH_8U && img->nChannels == 1 && (val&1) != 0);
191
192 for( i = 1; i < img->height - 1; i++ )
193 for( j = 1; j < img->width - 1; j++ )
194 {
195 uchar* t = (uchar*)(img->imageData + img->widthStep*i + j);
196 if( *t == 1 && (t[-step] == 0 || t[-1] == 0 || t[1] == 0 || t[step] == 0))
197 *t = (uchar)val;
198 }
199
200 cvThreshold( img, img, val - 2, val, CV_THRESH_BINARY );
201}

204int CV_FindContourTest::prepare_test_case( int test_case_idx )
205{
206 RNG& rng = ts->get_rng();
207 const int min_brightness = 0, max_brightness = 2;
208 int i, code = cvtest::BaseTest::prepare_test_case( test_case_idx );
209
210 if( code < 0 )
211 return code;
212
213 clear();
214
215 blob_count = cvRound(exp(cvtest::randReal(rng)*max_log_blob_count*CV_LOG2));
216
217 img_size.width = cvRound(exp((cvtest::randReal(rng)*
218 (max_log_img_size - min_log_img_size) + min_log_img_size)*CV_LOG2));
219 img_size.height = cvRound(exp((cvtest::randReal(rng)*
220 (max_log_img_size - min_log_img_size) + min_log_img_size)*CV_LOG2));
221
222 approx_method = cvtest::randInt( rng ) % 4 + 1;
223 retr_mode = cvtest::randInt( rng ) % 4;
224
225 storage = cvCreateMemStorage( 1 << 10 );
226
227 for( i = 0; i < NUM_IMG; i++ )
228 img[i] = cvCreateImage( img_size, 8, 1 );
229
230 cvTsGenerateBlobImage( img[0], min_blob_size, max_blob_size,
231 blob_count, min_brightness, max_brightness, rng );
232
233 cvCopy( img[0], img[1] );
234 cvCopy( img[0], img[2] );
235
236 cvTsMarkContours( img[1], 255 );
237
238 return 1;
239}

242void CV_FindContourTest::run_func()
243{
244 contours = contours2 = chain = 0;
245 count = cvFindContours( img[2], storage, &contours, sizeof(CvContour), retr_mode, approx_method );
246
247 cvZero( img[3] );
248
249 if( contours && retr_mode != CV_RETR_EXTERNAL && approx_method < CV_CHAIN_APPROX_TC89_L1 )
250 cvDrawContours( img[3], contours, cvScalar(255), cvScalar(255), INT_MAX, -1 );
251
252 cvCopy( img[0], img[2] );
253
254 count2 = cvFindContours( img[2], storage, &chain, sizeof(CvChain), retr_mode, CV_CHAIN_CODE );
255
256 if( chain )
257 contours2 = cvApproxChains( chain, storage, approx_method, 0, 0, 1 );
258
259 cvZero( img[2] );
260
261 if( contours && retr_mode != CV_RETR_EXTERNAL && approx_method < CV_CHAIN_APPROX_TC89_L1 )
262 cvDrawContours( img[2], contours2, cvScalar(255), cvScalar(255), INT_MAX );
263}

266// the whole testing is done here, run_func() is not utilized in this test
267int CV_FindContourTest::validate_test_results( int /*test_case_idx*/ )
268{
269 int code = cvtest::TS::OK;
270
271 cvCmpS( img[0], 0, img[0], CV_CMP_GT );
272
273 if( count != count2 )
274 {
275 ts->printf( cvtest::TS::LOG, "The number of contours retrieved with different "
276 "approximation methods is not the same\n"
277 "(%d contour(s) for method %d vs %d contour(s) for method %d)\n",
278 count, approx_method, count2, CV_CHAIN_CODE );
279 code = cvtest::TS::FAIL_INVALID_OUTPUT;
280 }
281
282 if( retr_mode != CV_RETR_EXTERNAL && approx_method < CV_CHAIN_APPROX_TC89_L1 )
283 {
284 Mat _img[4];
285 for( int i = 0; i < 4; i++ )
286 _img[i] = cvarrToMat(img[i]);
287
288 code = cvtest::cmpEps2(ts, _img[0], _img[3], 0, true, "Comparing original image with the map of filled contours" );
289
290 if( code < 0 )
291 goto _exit_;
292
293 code = cvtest::cmpEps2( ts, _img[1], _img[2], 0, true,
294 "Comparing contour outline vs manually produced edge map" );
295
296 if( code < 0 )
297 goto _exit_;
298 }
299
300 if( contours )
301 {
302 CvTreeNodeIterator iterator1;
303 CvTreeNodeIterator iterator2;
304 int count3;
305
306 for(int i = 0; i < 2; i++ )
307 {
308 CvTreeNodeIterator iterator;
309 cvInitTreeNodeIterator( &iterator, i == 0 ? contours : contours2, INT_MAX );
310
311 for( count3 = 0; cvNextTreeNode( &iterator ) != 0; count3++ )
312 ;
313
314 if( count3 != count )
315 {
316 ts->printf( cvtest::TS::LOG,
317 "The returned number of retrieved contours (using the approx_method = %d) does not match\n"
318 "to the actual number of contours in the tree/list (returned %d, actual %d)\n",
319 i == 0 ? approx_method : 0, count, count3 );
320 code = cvtest::TS::FAIL_INVALID_OUTPUT;
321 goto _exit_;
322 }
323 }
324
325 cvInitTreeNodeIterator( &iterator1, contours, INT_MAX );
326 cvInitTreeNodeIterator( &iterator2, contours2, INT_MAX );
327
328 for( count3 = 0; count3 < count; count3++ )
329 {
330 CvSeq* seq1 = (CvSeq*)cvNextTreeNode( &iterator1 );
331 CvSeq* seq2 = (CvSeq*)cvNextTreeNode( &iterator2 );
332 CvSeqReader reader1;
333 CvSeqReader reader2;
334
335 if( !seq1 || !seq2 )
336 {
337 ts->printf( cvtest::TS::LOG,
338 "There are NULL pointers in the original contour tree or the "
339 "tree produced by cvApproxChains\n" );
340 code = cvtest::TS::FAIL_INVALID_OUTPUT;
341 goto _exit_;
342 }
343
344 cvStartReadSeq( seq1, &reader1 );
345 cvStartReadSeq( seq2, &reader2 );
346
347 if( seq1->total != seq2->total )
348 {
349 ts->printf( cvtest::TS::LOG,
350 "The original contour #%d has %d points, while the corresponding contour has %d point\n",
351 count3, seq1->total, seq2->total );
352 code = cvtest::TS::FAIL_INVALID_OUTPUT;
353 goto _exit_;
354 }
355
356 for(int i = 0; i < seq1->total; i++ )
357 {
358 CvPoint pt1;
359 CvPoint pt2;
360
361 CV_READ_SEQ_ELEM( pt1, reader1 );
362 CV_READ_SEQ_ELEM( pt2, reader2 );
363
364 if( pt1.x != pt2.x || pt1.y != pt2.y )
365 {
366 ts->printf( cvtest::TS::LOG,
367 "The point #%d in the contour #%d is different from the corresponding point "
368 "in the approximated chain ((%d,%d) vs (%d,%d)", count3, i, pt1.x, pt1.y, pt2.x, pt2.y );
369 code = cvtest::TS::FAIL_INVALID_OUTPUT;
370 goto _exit_;
371 }
372 }
373 }
374 }
375
376_exit_:
377 if( code < 0 )
378 {
379#if 0
380 cvNamedWindow( "test", 0 );
381 cvShowImage( "test", img[0] );
382 cvWaitKey();
383#endif
384 ts->set_failed_test_info( code );
385 }
386
387 return code;
388}
