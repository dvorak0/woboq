113const char* CV_ImageWarpBaseTest::interpolation_to_string(int inter) const
114{
115 if (inter == INTER_NEAREST)

139void CV_ImageWarpBaseTest::generate_test_data()
140{
141 RNG& rng = ts->get_rng();
142
143 // generating the src matrix structure
144 Size ssize = randSize(rng), dsize;
145
146 int depth = rng.uniform(0, CV_64F);
147 while (depth == CV_8S || depth == CV_32S)
148 depth = rng.uniform(0, CV_64F);
149
150 int cn = rng.uniform(1, 4);
151 while (cn == 2)
152 cn = rng.uniform(1, 4);
153
154 src.create(ssize, CV_MAKE_TYPE(depth, cn));
155
156 // generating the src matrix
157 int x, y;
158 if (cvtest::randInt(rng) % 2)
159 {
160 for (y = 0; y < ssize.height; y += cell_size)
161 for (x = 0; x < ssize.width; x += cell_size)
162 rectangle(src, Point(x, y), Point(x + std::min<int>(cell_size, ssize.width - x), y +
163 std::min<int>(cell_size, ssize.height - y)), Scalar::all((x + y) % 2 ? 255: 0), CV_FILLED);
164 }
165 else
166 {
167 src = Scalar::all(255);
168 for (y = cell_size; y < src.rows; y += cell_size)
169 line(src, Point2i(0, y), Point2i(src.cols, y), Scalar::all(0), 1);
170 for (x = cell_size; x < src.cols; x += cell_size)
171 line(src, Point2i(x, 0), Point2i(x, src.rows), Scalar::all(0), 1);
172 }
173
174 // generating an interpolation type
175 interpolation = rng.uniform(0, CV_INTER_LANCZOS4 + 1);
176
177 // generating the dst matrix structure
178 double scale_x, scale_y;
179 if (interpolation == INTER_AREA)
180 {
181 bool area_fast = rng.uniform(0., 1.) > 0.5;
182 if (area_fast)
183 {
184 scale_x = rng.uniform(2, 5);
185 scale_y = rng.uniform(2, 5);
186 }
187 else
188 {
189 scale_x = rng.uniform(1.0, 3.0);
190 scale_y = rng.uniform(1.0, 3.0);
191 }
192 }
193 else
194 {
195 scale_x = rng.uniform(0.4, 4.0);
196 scale_y = rng.uniform(0.4, 4.0);
197 }
198 CV_Assert(scale_x > 0.0f && scale_y > 0.0f);
199
200 dsize.width = saturate_cast<int>((ssize.width + scale_x - 1) / scale_x);
201 dsize.height = saturate_cast<int>((ssize.height + scale_y - 1) / scale_y);
202
203 dst = Mat::zeros(dsize, src.type());
204 reference_dst = Mat::zeros(dst.size(), CV_MAKE_TYPE(CV_32F, dst.channels()));
205
206 scale_x = src.cols / static_cast<double>(dst.cols);
207 scale_y = src.rows / static_cast<double>(dst.rows);
208
209 if (interpolation == INTER_AREA && (scale_x < 1.0 || scale_y < 1.0))
210 interpolation = INTER_LINEAR;
211}

213void CV_ImageWarpBaseTest::run(int)
214{
215 for (int i = 0; i < test_case_count; ++i)
216 {
217 generate_test_data();
218 run_func();
219 run_reference_func();
220 if (ts->get_err_code() < 0)
221 break;
222 validate_results();
223 if (ts->get_err_code() < 0)
224 break;
225 ts->update_context(this, i, true);
226 }
227 ts->set_gtest_status();
228}

230void CV_ImageWarpBaseTest::validate_results() const
231{
232 Mat _dst;
233 dst.convertTo(_dst, reference_dst.depth());
234
235 Size dsize = dst.size(), ssize = src.size();
236 int cn = _dst.channels();
237 dsize.width *= cn;
238 float t = 1.0f;
239 if (interpolation == INTER_CUBIC)
240 t = 1.0f;
241 else if (interpolation == INTER_LANCZOS4)
242 t = 1.0f;
243 else if (interpolation == INTER_NEAREST)
244 t = 1.0f;
245 else if (interpolation == INTER_AREA)
246 t = 2.0f;
247
248 for (int dy = 0; dy < dsize.height; ++dy)
249 {
250 const float* rD = reference_dst.ptr<float>(dy);
251 const float* D = _dst.ptr<float>(dy);
252
253 for (int dx = 0; dx < dsize.width; ++dx)
254 if (fabs(rD[dx] - D[dx]) > t &&
255// fabs(rD[dx] - D[dx]) < 250.0f &&
256 rD[dx] <= 255.0f && D[dx] <= 255.0f && rD[dx] >= 0.0f && D[dx] >= 0.0f)
257 {
258 PRINT_TO_LOG("\nNorm of the difference: %lf\n", norm(reference_dst, _dst, NORM_INF));
259 PRINT_TO_LOG("Error in (dx, dy): (%d, %d)\n", dx / cn + 1, dy + 1);
260 PRINT_TO_LOG("Tuple (rD, D): (%f, %f)\n", rD[dx], D[dx]);
261 PRINT_TO_LOG("Dsize: (%d, %d)\n", dsize.width / cn, dsize.height);
262 PRINT_TO_LOG("Ssize: (%d, %d)\n", src.cols, src.rows);
263
264 double scale_x = static_cast<double>(ssize.width) / dsize.width;
265 double scale_y = static_cast<double>(ssize.height) / dsize.height;
266 bool area_fast = interpolation == INTER_AREA &&
267 fabs(scale_x - cvRound(scale_x)) < FLT_EPSILON &&
268 fabs(scale_y - cvRound(scale_y)) < FLT_EPSILON;
269 if (area_fast)
270 {
271 scale_y = cvRound(scale_y);
272 scale_x = cvRound(scale_x);
273 }
274
275 PRINT_TO_LOG("Interpolation: %s\n", interpolation_to_string(area_fast ? INTER_LANCZOS4 + 1 : interpolation));
276 PRINT_TO_LOG("Scale (x, y): (%lf, %lf)\n", scale_x, scale_y);
277 PRINT_TO_LOG("Elemsize: %d\n", src.elemSize1());
278 PRINT_TO_LOG("Channels: %d\n", cn);
279
280#ifdef SHOW_IMAGE
281 const std::string w1("OpenCV impl (run func)"), w2("Reference func"), w3("Src image"), w4("Diff");
282 namedWindow(w1, CV_WINDOW_KEEPRATIO);
283 namedWindow(w2, CV_WINDOW_KEEPRATIO);
284 namedWindow(w3, CV_WINDOW_KEEPRATIO);
285 namedWindow(w4, CV_WINDOW_KEEPRATIO);
286
287 Mat diff;
288 absdiff(reference_dst, _dst, diff);
289
290 imshow(w1, dst);
291 imshow(w2, reference_dst);
292 imshow(w3, src);
293 imshow(w4, diff);
294
295 waitKey();
296#endif
297
298 const int radius = 3;
299 int rmin = MAX(dy - radius, 0), rmax = MIN(dy + radius, dsize.height);
300 int cmin = MAX(dx / cn - radius, 0), cmax = MIN(dx / cn + radius, dsize.width);
301
302 std::cout << "opencv result:\n" << dst(Range(rmin, rmax), Range(cmin, cmax)) << std::endl;
303 std::cout << "reference result:\n" << reference_dst(Range(rmin, rmax), Range(cmin, cmax)) << std::endl;
304
305 ts->set_failed_test_info(cvtest::TS::FAIL_BAD_ACCURACY);
306 return;
307 }
308 }

364 void interpolateLinear(float x, float* coeffs)
365 {
366 coeffs[0] = 1.f - x;
367 coeffs[1] = x;
368 }
369
370 void interpolateCubic(float x, float* coeffs)
371 {
372 const float A = -0.75f;
373
374 coeffs[0] = ((A*(x + 1) - 5*A)*(x + 1) + 8*A)*(x + 1) - 4*A;
375 coeffs[1] = ((A + 2)*x - (A + 3))*x*x + 1;
376 coeffs[2] = ((A + 2)*(1 - x) - (A + 3))*(1 - x)*(1 - x) + 1;
377 coeffs[3] = 1.f - coeffs[0] - coeffs[1] - coeffs[2];
378 }
379
380 void interpolateLanczos4(float x, float* coeffs)
381 {
382 static const double s45 = 0.70710678118654752440084436210485;
383 static const double cs[][2]=
384 {{1, 0}, {-s45, -s45}, {0, 1}, {s45, -s45}, {-1, 0}, {s45, s45}, {0, -1}, {-s45, s45}};
385
386 if( x < FLT_EPSILON )
387 {
388 for( int i = 0; i < 8; i++ )
389 coeffs[i] = 0;
390 coeffs[3] = 1;
391 return;
392 }
393
394 float sum = 0;
395 double y0=-(x+3)*CV_PI*0.25, s0 = sin(y0), c0=cos(y0);
396 for(int i = 0; i < 8; i++ )
397 {
398 double y = -(x+3-i)*CV_PI*0.25;
399 coeffs[i] = (float)((cs[i][0]*s0 + cs[i][1]*c0)/(y*y));
400 sum += coeffs[i];
401 }
402
403 sum = 1.f/sum;
404 for(int i = 0; i < 8; i++ )
405 coeffs[i] *= sum;
406 }
407
408 typedef void (*interpolate_method)(float x, float* coeffs);
409 interpolate_method inter_array[] = { &interpolateLinear, &interpolateCubic, &interpolateLanczos4 };
410}

444double CV_Resize_Test::getWeight(double a, double b, int x)
445{
446 double w = std::min(static_cast<double>(x + 1), b) - std::max(static_cast<double>(x), a);
447 CV_Assert(w >= 0);
448 return w;

451void CV_Resize_Test::resize_area()
452{
453 Size ssize = src.size(), dsize = reference_dst.size();
454 CV_Assert(ssize.area() > 0 && dsize.area() > 0);
455 int cn = src.channels();
456
457 CV_Assert(scale_x >= 1.0 && scale_y >= 1.0);
458
459 double fsy0 = 0, fsy1 = scale_y;
460 for (int dy = 0; dy < dsize.height; ++dy)
461 {
462 float* yD = reference_dst.ptr<float>(dy);
463 int isy0 = cvFloor(fsy0), isy1 = std::min(cvFloor(fsy1), ssize.height - 1);
464 CV_Assert(isy1 <= ssize.height && isy0 < ssize.height);
465
466 double fsx0 = 0, fsx1 = scale_x;
467
468 for (int dx = 0; dx < dsize.width; ++dx)
469 {
470 float* xyD = yD + cn * dx;
471 int isx0 = cvFloor(fsx0), isx1 = std::min(ssize.width - 1, cvFloor(fsx1));
472
473 CV_Assert(isx1 <= ssize.width);
474 CV_Assert(isx0 < ssize.width);
475
476 // for each pixel of dst
477 for (int r = 0; r < cn; ++r)
478 {
479 xyD[r] = 0.0f;
480 double area = 0.0;
481 for (int sy = isy0; sy <= isy1; ++sy)
482 {
483 const float* yS = src.ptr<float>(sy);
484 for (int sx = isx0; sx <= isx1; ++sx)
485 {
486 double wy = getWeight(fsy0, fsy1, sy);
487 double wx = getWeight(fsx0, fsx1, sx);
488 double w = wx * wy;
489 xyD[r] += static_cast<float>(yS[sx * cn + r] * w);
490 area += w;
491 }
492 }
493
494 CV_Assert(area != 0);
495 // norming pixel
496 xyD[r] = static_cast<float>(xyD[r] / area);
497 }
498 fsx1 = std::min((fsx0 = fsx1) + scale_x, static_cast<double>(ssize.width));
499 }
500 fsy1 = std::min((fsy0 = fsy1) + scale_y, static_cast<double>(ssize.height));
501 }
502}

504// for interpolation type : INTER_LINEAR, INTER_LINEAR, INTER_CUBIC, INTER_LANCZOS4
505void CV_Resize_Test::resize_1d(const Mat& _src, Mat& _dst, int dy, const dim& _dim)
506{
507 Size dsize = _dst.size();
508 int cn = _dst.channels();
509 float* yD = _dst.ptr<float>(dy);
510
511 if (interpolation == INTER_NEAREST)
512 {
513 const float* yS = _src.ptr<float>(dy);
514 for (int dx = 0; dx < dsize.width; ++dx)
515 {
516 int isx = _dim[dx].first;
517 const float* xyS = yS + isx * cn;
518 float* xyD = yD + dx * cn;
519
520 for (int r = 0; r < cn; ++r)
521 xyD[r] = xyS[r];
522 }
523 }
524 else if (interpolation == INTER_LINEAR || interpolation == INTER_CUBIC || interpolation == INTER_LANCZOS4)
525 {
526 internal::interpolate_method inter_func = internal::inter_array[interpolation - (interpolation == INTER_LANCZOS4 ? 2 : 1)];
527 size_t elemsize = _src.elemSize();
528
529 int ofs = 0, ksize = 2;
530 if (interpolation == INTER_CUBIC)
531 ofs = 1, ksize = 4;
532 else if (interpolation == INTER_LANCZOS4)
533 ofs = 3, ksize = 8;
534
535 Mat _extended_src_row(1, _src.cols + ksize * 2, _src.type());
536 uchar* srow = _src.data + dy * _src.step;
537 memcpy(_extended_src_row.data + elemsize * ksize, srow, _src.step);
538 for (int k = 0; k < ksize; ++k)
539 {
540 memcpy(_extended_src_row.data + k * elemsize, srow, elemsize);
541 memcpy(_extended_src_row.data + (ksize + k) * elemsize + _src.step, srow + _src.step - elemsize, elemsize);
542 }
543
544 for (int dx = 0; dx < dsize.width; ++dx)
545 {
546 int isx = _dim[dx].first;
547 double fsx = _dim[dx].second;
548
549 float *xyD = yD + dx * cn;
550 const float* xyS = _extended_src_row.ptr<float>(0) + (isx + ksize - ofs) * cn;
551
552 float w[8];
553 inter_func(static_cast<float>(fsx), w);
554
555 for (int r = 0; r < cn; ++r)
556 {
557 xyD[r] = 0;
558 for (int k = 0; k < ksize; ++k)
559 xyD[r] += w[k] * xyS[k * cn + r];
560 }
561 }
562 }
563 else
564 CV_Assert(0);
565}

567void CV_Resize_Test::generate_buffer(double scale, dim& _dim)
568{
569 size_t length = _dim.size();
570 for (size_t dx = 0; dx < length; ++dx)
571 {
572 double fsx = scale * (dx + 0.5) - 0.5;
573 int isx = cvFloor(fsx);
574 _dim[dx] = std::make_pair(isx, fsx - isx);
575 }
576}

578void CV_Resize_Test::resize_generic()
579{
580 Size dsize = reference_dst.size(), ssize = src.size();
581 CV_Assert(dsize.area() > 0 && ssize.area() > 0);
582
583 dim dims[] = { dim(dsize.width), dim(dsize.height) };
584 if (interpolation == INTER_NEAREST)
585 {
586 for (int dx = 0; dx < dsize.width; ++dx)
587 dims[0][dx].first = std::min(cvFloor(dx * scale_x), ssize.width - 1);
588 for (int dy = 0; dy < dsize.height; ++dy)
589 dims[1][dy].first = std::min(cvFloor(dy * scale_y), ssize.height - 1);
590 }
591 else
592 {
593 generate_buffer(scale_x, dims[0]);
594 generate_buffer(scale_y, dims[1]);
595 }
596
597 Mat tmp(ssize.height, dsize.width, reference_dst.type());
598 for (int dy = 0; dy < tmp.rows; ++dy)
599 resize_1d(src, tmp, dy, dims[0]);
600
601 transpose(tmp, tmp);
602 transpose(reference_dst, reference_dst);
603
604 for (int dy = 0; dy < tmp.rows; ++dy)
605 resize_1d(tmp, reference_dst, dy, dims[1]);
606 transpose(reference_dst, reference_dst);
607}

657void CV_Remap_Test::generate_test_data()
658{

670 const int n = std::min(std::min(src.cols, src.rows) / 10 + 1, 2);

673 switch (mapx.type())

686 static const int mapy_types[] = { CV_16UC1, CV_16SC1 };

691 case CV_16UC1:
692 {
693 MatIterator_<ushort> begin_y = mapy.begin<ushort>(), end_y = mapy.end<ushort>();
694 for ( ; begin_y != end_y; ++begin_y)
695 *begin_y = static_cast<ushort>(rng.uniform(0, 1024));
696 }
697 break;
698
699 case CV_16SC1:
700 {
701 MatIterator_<short> begin_y = mapy.begin<short>(), end_y = mapy.end<short>();
702 for ( ; begin_y != end_y; ++begin_y)
703 *begin_y = static_cast<short>(rng.uniform(0, 1024));
704 }
705 break;
706 }
707 }
708 }
709 break;

711 case CV_32FC1:
712 {
713 mapy.create(dst.size(), CV_32FC1);
714 float fscols = static_cast<float>(std::max(src.cols - 1 + n, 0)),
715 fsrows = static_cast<float>(std::max(src.rows - 1 + n, 0));
716 MatIterator_<float> begin_x = mapx.begin<float>(), end_x = mapx.end<float>();
717 MatIterator_<float> begin_y = mapy.begin<float>();
718 for ( ; begin_x != end_x; ++begin_x, ++begin_y)

720 *begin_x = rng.uniform(_n, fscols);

726 case CV_32FC2:
727 {
728 MatIterator_<Vec2f> begin_x = mapx.begin<Vec2f>(), end_x = mapx.end<Vec2f>();

732 {
733 (*begin_x)[0] = rng.uniform(_n, fscols);
734 (*begin_x)[1] = rng.uniform(_n, fsrows);
735 }
736 }
737 break;

739 default:
740 assert(0);
741 break;

793{
794 CV_Assert(_src.depth() == CV_32F && _dst.type() == _src.type());

796
797 Size ssize = _src.size(), dsize = _dst.size();

799 int cn = _src.channels();
800
801 for (int dy = 0; dy < dsize.height; ++dy)
802 {
803 const short* yM = mapx.ptr<short>(dy);

805
806 for (int dx = 0; dx < dsize.width; ++dx)
807 {
808 float* xyD = yD + cn * dx;
809 int sx = yM[dx * 2], sy = yM[dx * 2 + 1];
810
811 if (sx >= 0 && sx < ssize.width && sy >= 0 && sy < ssize.height)
812 {
813 const float *xyS = _src.ptr<float>(sy) + sx * cn;
814
815 for (int r = 0; r < cn; ++r)
816 xyD[r] = xyS[r];
817 }
818 else if (borderType != BORDER_TRANSPARENT)
819 {
820 if (borderType == BORDER_CONSTANT)
821 for (int r = 0; r < cn; ++r)
822 xyD[r] = saturate_cast<float>(borderValue[r]);
823 else
824 {
825 sx = borderInterpolate(sx, ssize.width, borderType);
826 sy = borderInterpolate(sy, ssize.height, borderType);
827 CV_Assert(sx >= 0 && sy >= 0 && sx < ssize.width && sy < ssize.height);
828
829 const float *xyS = _src.ptr<float>(sy) + sx * cn;
830
831 for (int r = 0; r < cn; ++r)
832 xyD[r] = xyS[r];
833 }
834 }
835 }

839void CV_Remap_Test::remap_generic(const Mat& _src, Mat& _dst)
840{
841 CV_Assert(mapx.type() == CV_16SC2 && mapy.type() == CV_16UC1);
842
843 int ksize = 2;
844 if (interpolation == INTER_CUBIC)
845 ksize = 4;
846 else if (interpolation == INTER_LANCZOS4)
847 ksize = 8;
848 else if (interpolation != INTER_LINEAR)
849 assert(0);
850 int ofs = (ksize / 2) - 1;
851
852 CV_Assert(_src.depth() == CV_32F && _dst.type() == _src.type());
853 Size ssize = _src.size(), dsize = _dst.size();
854 int cn = _src.channels(), width1 = std::max(ssize.width - ksize + 1, 0),
855 height1 = std::max(ssize.height - ksize + 1, 0);
856
857 float ix[8], w[16];
858 internal::interpolate_method inter_func = internal::inter_array[interpolation - (interpolation == INTER_LANCZOS4 ? 2 : 1)];
859
860 for (int dy = 0; dy < dsize.height; ++dy)
861 {
862 const short* yMx = mapx.ptr<short>(dy);
863 const ushort* yMy = mapy.ptr<ushort>(dy);
864
865 float* yD = _dst.ptr<float>(dy);
866
867 for (int dx = 0; dx < dsize.width; ++dx)
868 {
869 float* xyD = yD + dx * cn;
870 float sx = yMx[dx * 2], sy = yMx[dx * 2 + 1];
871 int isx = cvFloor(sx), isy = cvFloor(sy);
872
873 inter_func((yMy[dx] & (INTER_TAB_SIZE - 1)) / static_cast<float>(INTER_TAB_SIZE), w);
874 inter_func(((yMy[dx] >> INTER_BITS) & (INTER_TAB_SIZE - 1)) / static_cast<float>(INTER_TAB_SIZE), w + ksize);
875
876 isx -= ofs;
877 isy -= ofs;
878
879 if (isx >= 0 && isx < width1 && isy >= 0 && isy < height1)
880 {
881 for (int r = 0; r < cn; ++r)
882 {
883 for (int y = 0; y < ksize; ++y)
884 {
885 const float* xyS = _src.ptr<float>(isy + y) + isx * cn;
886
887 ix[y] = 0;
888 for (int i = 0; i < ksize; ++i)
889 ix[y] += w[i] * xyS[i * cn + r];
890 }
891 xyD[r] = 0;
892 for (int i = 0; i < ksize; ++i)
893 xyD[r] += w[ksize + i] * ix[i];
894 }
895 }
896 else if (borderType != BORDER_TRANSPARENT)
897 {
898 int ar_x[8], ar_y[8];
899
900 for (int k = 0; k < ksize; k++)
901 {
902 ar_x[k] = borderInterpolate(isx + k, ssize.width, borderType) * cn;
903 ar_y[k] = borderInterpolate(isy + k, ssize.height, borderType);
904 }
905
906 for (int r = 0; r < cn; r++)
907 {
908 xyD[r] = 0;
909 for (int i = 0; i < ksize; ++i)
910 {
911 ix[i] = 0;
912 if (ar_y[i] >= 0)
913 {
914 const float* yS = _src.ptr<float>(ar_y[i]);
915 for (int j = 0; j < ksize; ++j)
916 ix[i] += saturate_cast<float>((ar_x[j] >= 0 ? yS[ar_x[j] + r] : borderValue[r]) * w[j]);
917 }
918 else
919 for (int j = 0; j < ksize; ++j)
920 ix[i] += saturate_cast<float>(borderValue[r] * w[j]);
921 }
922 for (int i = 0; i < ksize; ++i)
923 xyD[r] += saturate_cast<float>(w[ksize + i] * ix[i]);
924 }
925 }
926 }
927 }

1016void CV_WarpAffine_Test::warpAffine(const Mat& _src, Mat& _dst)
1017{
1018 Size dsize = _dst.size();
1019
1020 CV_Assert(_src.size().area() > 0);
1021 CV_Assert(dsize.area() > 0);
1022 CV_Assert(_src.type() == _dst.type());
1023
1024 Mat tM;
1025 M.convertTo(tM, CV_64F);
1026
1027 int inter = interpolation & INTER_MAX;
1028 if (inter == INTER_AREA)
1029 inter = INTER_LINEAR;
1030
1031 mapx.create(dsize, CV_16SC2);
1032 if (inter != INTER_NEAREST)
1033 mapy.create(dsize, CV_16SC1);
1034 else
1035 mapy = Mat();
1036
1037 if (!(interpolation & CV_WARP_INVERSE_MAP))
1038 invertAffineTransform(tM.clone(), tM);
1039
1040 const int AB_BITS = MAX(10, (int)INTER_BITS);
1041 const int AB_SCALE = 1 << AB_BITS;
1042 int round_delta = (inter == INTER_NEAREST) ? AB_SCALE / 2 : (AB_SCALE / INTER_TAB_SIZE / 2);
1043
1044 const double* data_tM = tM.ptr<double>(0);
1045 for (int dy = 0; dy < dsize.height; ++dy)
1046 {
1047 short* yM = mapx.ptr<short>(dy);
1048 for (int dx = 0; dx < dsize.width; ++dx, yM += 2)
1049 {
1050 int v1 = saturate_cast<int>(saturate_cast<int>(data_tM[0] * dx * AB_SCALE) +
1051 saturate_cast<int>((data_tM[1] * dy + data_tM[2]) * AB_SCALE) + round_delta),
1052 v2 = saturate_cast<int>(saturate_cast<int>(data_tM[3] * dx * AB_SCALE) +
1053 saturate_cast<int>((data_tM[4] * dy + data_tM[5]) * AB_SCALE) + round_delta);
1054 v1 >>= AB_BITS - INTER_BITS;
1055 v2 >>= AB_BITS - INTER_BITS;
1056
1057 yM[0] = saturate_cast<short>(v1 >> INTER_BITS);
1058 yM[1] = saturate_cast<short>(v2 >> INTER_BITS);
1059
1060 if (inter != INTER_NEAREST)
1061 mapy.ptr<short>(dy)[dx] = ((v2 & (INTER_TAB_SIZE - 1)) * INTER_TAB_SIZE + (v1 & (INTER_TAB_SIZE - 1)));
1062 }
1063 }
1064
1065 CV_Assert(mapx.type() == CV_16SC2 && ((inter == INTER_NEAREST && !mapy.data) || mapy.type() == CV_16SC1));
1066 cv::remap(_src, _dst, mapx, mapy, inter, borderType, borderValue);
1067}

1132void CV_WarpPerspective_Test::warpPerspective(const Mat& _src, Mat& _dst)
1133{
1134 Size ssize = _src.size(), dsize = _dst.size();
1135
1136 CV_Assert(ssize.area() > 0);
1137 CV_Assert(dsize.area() > 0);
1138 CV_Assert(_src.type() == _dst.type());
1139
1140 if (M.depth() != CV_64F)
1141 {
1142 Mat tmp;
1143 M.convertTo(tmp, CV_64F);
1144 M = tmp;
1145 }
1146
1147 if (!(interpolation & CV_WARP_INVERSE_MAP))
1148 {
1149 Mat tmp;
1150 invert(M, tmp);
1151 M = tmp;
1152 }
1153
1154 int inter = interpolation & INTER_MAX;
1155 if (inter == INTER_AREA)
1156 inter = INTER_LINEAR;
1157
1158 mapx.create(dsize, CV_16SC2);
1159 if (inter != INTER_NEAREST)
1160 mapy.create(dsize, CV_16SC1);
1161 else
1162 mapy = Mat();
1163
1164 double* tM = M.ptr<double>(0);
1165 for (int dy = 0; dy < dsize.height; ++dy)
1166 {
1167 short* yMx = mapx.ptr<short>(dy);
1168
1169 for (int dx = 0; dx < dsize.width; ++dx, yMx += 2)
1170 {
1171 double den = tM[6] * dx + tM[7] * dy + tM[8];
1172 den = den ? 1.0 / den : 0.0;
1173
1174 if (inter == INTER_NEAREST)
1175 {
1176 yMx[0] = saturate_cast<short>((tM[0] * dx + tM[1] * dy + tM[2]) * den);
1177 yMx[1] = saturate_cast<short>((tM[3] * dx + tM[4] * dy + tM[5]) * den);
1178 continue;
1179 }
1180
1181 den *= INTER_TAB_SIZE;
1182 int v0 = saturate_cast<int>((tM[0] * dx + tM[1] * dy + tM[2]) * den);
1183 int v1 = saturate_cast<int>((tM[3] * dx + tM[4] * dy + tM[5]) * den);
1184
1185 yMx[0] = saturate_cast<short>(v0 >> INTER_BITS);
1186 yMx[1] = saturate_cast<short>(v1 >> INTER_BITS);
1187 mapy.ptr<short>(dy)[dx] = saturate_cast<short>((v1 & (INTER_TAB_SIZE - 1)) *
1188 INTER_TAB_SIZE + (v0 & (INTER_TAB_SIZE - 1)));
1189 }
1190 }
1191
1192 CV_Assert(mapx.type() == CV_16SC2 && ((inter == INTER_NEAREST && !mapy.data) || mapy.type() == CV_16SC1));
