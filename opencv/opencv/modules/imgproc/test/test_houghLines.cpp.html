84void CV_HoughLinesTest::run_test(int type)
85{
86 Mat src = imread(string(ts->get_data_path()) + "shared/pic1.png");
87 if (src.empty())
88 {
89 ts->set_failed_test_info(cvtest::TS::FAIL_INVALID_TEST_DATA);
90 return;
91 }
92
93 string xml;
94 if (type == STANDART)
95 xml = string(ts->get_data_path()) + "imgproc/HoughLines.xml";
96 else if (type == PROBABILISTIC)
97 xml = string(ts->get_data_path()) + "imgproc/HoughLinesP.xml";
98 else
99 {
100 ts->printf(cvtest::TS::LOG, "Error: unknown HoughLines algorithm type.\n");
101 ts->set_failed_test_info(cvtest::TS::FAIL_GENERIC);
102 return;
103 }
104
105 Mat dst;
106 Canny(src, dst, 50, 200, 3);
107
108 Mat lines;
109 if (type == STANDART)
110 HoughLines(dst, lines, 1, CV_PI/180, 100, 0, 0);
111 else if (type == PROBABILISTIC)
112 HoughLinesP(dst, lines, 1, CV_PI/180, 100, 0, 0);
113
114 FileStorage fs(xml, FileStorage::READ);
115 if (!fs.isOpened())
116 {
117 fs.open(xml, FileStorage::WRITE);
118 if (!fs.isOpened())
119 {
120 ts->set_failed_test_info(cvtest::TS::FAIL_INVALID_TEST_DATA);
121 return;
122 }
123 fs << "exp_lines" << lines;
124 fs.release();
125 fs.open(xml, FileStorage::READ);
126 if (!fs.isOpened())
127 {
128 ts->set_failed_test_info(cvtest::TS::FAIL_INVALID_TEST_DATA);
129 return;
130 }
131 }
132
133 Mat exp_lines;
134 read( fs["exp_lines"], exp_lines, Mat() );
135 fs.release();
136
137 if ( exp_lines.size != lines.size || norm(exp_lines, lines, NORM_INF) > 1e-4 )
138 {
139 ts->set_failed_test_info(cvtest::TS::FAIL_MISMATCH);
140 return;
141 }
142
143 ts->set_failed_test_info(cvtest::TS::OK);
144}
