71template <typename T> void CV_BoundingRectTest::generate_src_points(vector <Point_<T> >& src, int n)
72{
73 src.clear();
74 for (int i = 0; i < n; ++i)
75 src.push_back(Point_<T>(cv::randu<T>(), cv::randu<T>()));
76}

78template <typename T> cv::Rect CV_BoundingRectTest::get_bounding_rect(const vector <Point_<T> > src)
79{
80 int n = (int)src.size();
81 T min_w = std::numeric_limits<T>::max(), max_w = std::numeric_limits<T>::min();
82 T min_h = min_w, max_h = max_w;
83
84 for (int i = 0; i < n; ++i)
85 {
86 min_w = std::min<T>(src.at(i).x, min_w);
87 max_w = std::max<T>(src.at(i).x, max_w);
88 min_h = std::min<T>(src.at(i).y, min_h);
89 max_h = std::max<T>(src.at(i).y, max_h);
90 }
91
92 return Rect((int)min_w, (int)min_h, (int)max_w-(int)min_w + 1, (int)max_h-(int)min_h + 1);
93}

95template <typename T> bool CV_BoundingRectTest::checking_function_work(vector <Point_<T> >& src, int type)
96{
97 const int MAX_COUNT_OF_POINTS = 1000;
98 const int N = 10000;
99
100 for (int k = 0; k < N; ++k)
101 {
102
103 RNG& rng = ts->get_rng();
104
105 int n = rng.next()%MAX_COUNT_OF_POINTS + 1;
106
107 generate_src_points <T> (src, n);
108
109 cv::Rect right = get_bounding_rect <T> (src);
110
111 cv::Rect rect[2] = { boundingRect(src), boundingRect(Mat(src)) };
112
113 for (int i = 0; i < 2; ++i) if (rect[i] != right)
114 {
115 cout << endl; cout << "Checking for the work of boundingRect function..." << endl;
116 cout << "Type of src points: ";
117 switch (type)
118 {
119 case 0: {cout << "INT"; break;}
120 case 1: {cout << "FLOAT"; break;}
121 default: break;
122 }
123 cout << endl;
124 cout << "Src points are stored as "; if (i == 0) cout << "VECTOR" << endl; else cout << "MAT" << endl;
125 cout << "Number of points: " << n << endl;
126 cout << "Right rect (x, y, w, h): [" << right.x << ", " << right.y << ", " << right.width << ", " << right.height << "]" << endl;
127 cout << "Result rect (x, y, w, h): [" << rect[i].x << ", " << rect[i].y << ", " << rect[i].width << ", " << rect[i].height << "]" << endl;
128 cout << endl;
129 CV_Error(IMGPROC_BOUNDINGRECT_ERROR_DIFF, MESSAGE_ERROR_DIFF);
130 return false;
131 }
132
133 }
134
135 return true;
136}

138void CV_BoundingRectTest::run(int)
139{
140 vector <Point> src_veci; if (!checking_function_work(src_veci, 0)) return;
141 vector <Point2f> src_vecf; checking_function_work(src_vecf, 1);
142}
