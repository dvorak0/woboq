68CV_DisTransTest::CV_DisTransTest()
69{
70 test_array[INPUT].push_back(NULL);
71 test_array[OUTPUT].push_back(NULL);
72 test_array[OUTPUT].push_back(NULL);
73 test_array[REF_OUTPUT].push_back(NULL);
74 test_array[REF_OUTPUT].push_back(NULL);
75 optional_mask = false;
76 element_wise_relative_error = true;
77}

80void CV_DisTransTest::get_test_array_types_and_sizes( int test_case_idx,
81 vector<vector<Size> >& sizes, vector<vector<int> >& types )
82{
83 RNG& rng = ts->get_rng();
84 cvtest::ArrayTest::get_test_array_types_and_sizes( test_case_idx, sizes, types );
85
86 types[INPUT][0] = CV_8UC1;
87 types[OUTPUT][0] = types[REF_OUTPUT][0] = CV_32FC1;
88 types[OUTPUT][1] = types[REF_OUTPUT][1] = CV_32SC1;
89
90 if( cvtest::randInt(rng) & 1 )
91 {
92 mask_size = 3;
93 dist_type = cvtest::randInt(rng) % 4;
94 dist_type = dist_type == 0 ? CV_DIST_C : dist_type == 1 ? CV_DIST_L1 :
95 dist_type == 2 ? CV_DIST_L2 : CV_DIST_USER;
96 }
97 else
98 {
99 mask_size = 5;
100 dist_type = cvtest::randInt(rng) % 10;
101 dist_type = dist_type == 0 ? CV_DIST_C : dist_type == 1 ? CV_DIST_L1 :
102 dist_type < 6 ? CV_DIST_L2 : CV_DIST_USER;
103 }
104
105 // for now, check only the "labeled" distance transform mode
106 fill_labels = 0;
107
108 if( !fill_labels )
109 sizes[OUTPUT][1] = sizes[REF_OUTPUT][1] = cvSize(0,0);
110
111 if( dist_type == CV_DIST_USER )
112 {
113 mask[0] = (float)(1.1 - cvtest::randReal(rng)*0.2);
114 mask[1] = (float)(1.9 - cvtest::randReal(rng)*0.8);
115 mask[2] = (float)(3. - cvtest::randReal(rng));
116 }
117}

120double CV_DisTransTest::get_success_error_level( int /*test_case_idx*/, int /*i*/, int /*j*/ )
121{
122 Size sz = test_mat[INPUT][0].size();
123 return dist_type == CV_DIST_C || dist_type == CV_DIST_L1 ? 0 : 0.01*MAX(sz.width, sz.height);
124}

127void CV_DisTransTest::get_minmax_bounds( int i, int j, int type, Scalar& low, Scalar& high )

137int CV_DisTransTest::prepare_test_case( int test_case_idx )
138{
139 int code = cvtest::ArrayTest::prepare_test_case( test_case_idx );
140 if( code > 0 )
141 {
142 // the function's response to an "all-nonzeros" image is not determined,
143 // so put at least one zero point
144 Mat& mat = test_mat[INPUT][0];
145 RNG& rng = ts->get_rng();
146 int i = cvtest::randInt(rng) % mat.rows;
147 int j = cvtest::randInt(rng) % mat.cols;
148 mat.at<uchar>(i,j) = 0;
149 }
150
151 return code;
152}

162static void
163cvTsDistTransform( const CvMat* _src, CvMat* _dst, int dist_type,
164 int mask_size, float* _mask, CvMat* /*_labels*/ )
165{
166 int i, j, k;
167 int width = _src->cols, height = _src->rows;
168 const float init_val = 1e6;
169 float mask[3] = { 0 };
170 CvMat* temp;
171 int ofs[16] = { 0 };
172 float delta[16] = { 0 };
173 int tstep, count;
174
175 assert( mask_size == 3 || mask_size == 5 );
176
177 if( dist_type == CV_DIST_USER )
178 memcpy( mask, _mask, sizeof(mask) );
179 else if( dist_type == CV_DIST_C )
180 {
181 mask_size = 3;
182 mask[0] = mask[1] = 1.f;
183 }
184 else if( dist_type == CV_DIST_L1 )
185 {
186 mask_size = 3;
187 mask[0] = 1.f;
188 mask[1] = 2.f;
189 }
190 else if( mask_size == 3 )
191 {
192 mask[0] = 0.955f;
193 mask[1] = 1.3693f;
194 }
195 else
196 {
197 mask[0] = 1.0f;
198 mask[1] = 1.4f;
199 mask[2] = 2.1969f;
200 }
201
202 temp = cvCreateMat( height + mask_size-1, width + mask_size-1, CV_32F );
203 tstep = temp->step / sizeof(float);
204
205 if( mask_size == 3 )
206 {
207 count = 4;
208 ofs[0] = -1; delta[0] = mask[0];
209 ofs[1] = -tstep-1; delta[1] = mask[1];
210 ofs[2] = -tstep; delta[2] = mask[0];
211 ofs[3] = -tstep+1; delta[3] = mask[1];
212 }
213 else
214 {
215 count = 8;
216 ofs[0] = -1; delta[0] = mask[0];
217 ofs[1] = -tstep-2; delta[1] = mask[2];
218 ofs[2] = -tstep-1; delta[2] = mask[1];
219 ofs[3] = -tstep; delta[3] = mask[0];
220 ofs[4] = -tstep+1; delta[4] = mask[1];
221 ofs[5] = -tstep+2; delta[5] = mask[2];
222 ofs[6] = -tstep*2-1; delta[6] = mask[2];
223 ofs[7] = -tstep*2+1; delta[7] = mask[2];
224 }
225
226 for( i = 0; i < mask_size/2; i++ )
227 {
228 float* t0 = (float*)(temp->data.ptr + i*temp->step);
229 float* t1 = (float*)(temp->data.ptr + (temp->rows - i - 1)*temp->step);
230
231 for( j = 0; j < width + mask_size - 1; j++ )
232 t0[j] = t1[j] = init_val;
233 }
234
235 for( i = 0; i < height; i++ )
236 {
237 uchar* s = _src->data.ptr + i*_src->step;
238 float* tmp = (float*)(temp->data.ptr + temp->step*(i + (mask_size/2))) + (mask_size/2);
239
240 for( j = 0; j < mask_size/2; j++ )
241 tmp[-j-1] = tmp[j + width] = init_val;
242
243 for( j = 0; j < width; j++ )
244 {
245 if( s[j] == 0 )
246 tmp[j] = 0;
247 else
248 {
249 float min_dist = init_val;
250 for( k = 0; k < count; k++ )
251 {
252 float t = tmp[j+ofs[k]] + delta[k];
253 if( min_dist > t )
254 min_dist = t;
255 }
256 tmp[j] = min_dist;
257 }
258 }
259 }
260
261 for( i = height - 1; i >= 0; i-- )
262 {
263 float* d = (float*)(_dst->data.ptr + i*_dst->step);
264 float* tmp = (float*)(temp->data.ptr + temp->step*(i + (mask_size/2))) + (mask_size/2);
265
266 for( j = width - 1; j >= 0; j-- )
267 {
268 float min_dist = tmp[j];
269 if( min_dist > mask[0] )
270 {
271 for( k = 0; k < count; k++ )
272 {
273 float t = tmp[j-ofs[k]] + delta[k];
274 if( min_dist > t )
275 min_dist = t;
276 }
277 tmp[j] = min_dist;
278 }
279 d[j] = min_dist;
280 }
281 }
282
283 cvReleaseMat( &temp );
284}

289 CvMat _input = test_mat[INPUT][0], _output = test_mat[REF_OUTPUT][0];

292}
