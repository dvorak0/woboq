117void CV_ColorCvtBaseTest::get_test_array_types_and_sizes( int test_case_idx,
118 vector<vector<Size> >& sizes, vector<vector<int> >& types )
119{
120 RNG& rng = ts->get_rng();
121 int depth, cn;
122 cvtest::ArrayTest::get_test_array_types_and_sizes( test_case_idx, sizes, types );
123
124 if( allow_16u && allow_32f )
125 {
126 depth = cvtest::randInt(rng) % 3;
127 depth = depth == 0 ? CV_8U : depth == 1 ? CV_16U : CV_32F;
128 }
129 else if( allow_16u || allow_32f )
130 {
131 depth = cvtest::randInt(rng) % 2;
132 depth = depth == 0 ? CV_8U : allow_16u ? CV_16U : CV_32F;
133 }
134 else
135 depth = CV_8U;
136
137 cn = (cvtest::randInt(rng) & 1) + 3;
138 blue_idx = cvtest::randInt(rng) & 1 ? 2 : 0;
139
140 types[INPUT][0] = CV_MAKETYPE(depth, cn);
141 types[OUTPUT][0] = types[REF_OUTPUT][0] = CV_MAKETYPE(depth, 3);
142 if( test_array[OUTPUT].size() > 1 )
143 types[OUTPUT][1] = types[REF_OUTPUT][1] = CV_MAKETYPE(depth, cn);
144
145 inplace = cn == 3 && cvtest::randInt(rng) % 2 != 0;
146 test_cpp = (cvtest::randInt(rng) & 256) == 0;
147}

158void CV_ColorCvtBaseTest::run_func()
159{
160 CvArr* out0 = test_array[OUTPUT][0];
161 cv::Mat _out0 = cv::cvarrToMat(out0), _out1 = cv::cvarrToMat(test_array[OUTPUT][1]);
162
163 if(!test_cpp)
164 cvCvtColor( inplace ? out0 : test_array[INPUT][0], out0, fwd_code );
165 else
166 cv::cvtColor( cv::cvarrToMat(inplace ? out0 : test_array[INPUT][0]), _out0, fwd_code, _out0.channels());
167
168 if( inplace )
169 {
170 cvCopy( out0, test_array[OUTPUT][1] );
171 out0 = test_array[OUTPUT][1];
172 }
173 if(!test_cpp)
174 cvCvtColor( out0, test_array[OUTPUT][1], inv_code );
175 else
176 cv::cvtColor(cv::cvarrToMat(out0), _out1, inv_code, _out1.channels());
177}

182 convert_forward( test_mat[INPUT][0], test_mat[REF_OUTPUT][0] );
183 convert_backward( test_mat[INPUT][0], test_mat[REF_OUTPUT][0],
184 test_mat[REF_OUTPUT][1] );
185 int depth = test_mat[REF_OUTPUT][0].depth();

187 {
188 for( int y = 0; y < test_mat[REF_OUTPUT][0].rows; y++ )
189 {
190 uchar* h0 = test_mat[REF_OUTPUT][0].ptr(y);
191 uchar* h = test_mat[OUTPUT][0].ptr(y);
192
193 for( int x = 0; x < test_mat[REF_OUTPUT][0].cols; x++, h0 += 3, h += 3 )
194 {
195 if( abs(*h - *h0) >= hue_range-1 && (*h <= 1 || *h0 <= 1) )
196 *h = *h0 = 0;
197 }
198 }

203void CV_ColorCvtBaseTest::convert_forward( const Mat& src, Mat& dst )
204{
205 const float c8u = 0.0039215686274509803f; // 1./255
206 const float c16u = 1.5259021896696422e-005f; // 1./65535
207 int depth = src.depth();
208 int cn = src.channels(), dst_cn = dst.channels();
209 int cols = src.cols, dst_cols_n = dst.cols*dst_cn;
210 vector<float> _src_buf(src.cols*3);
211 vector<float> _dst_buf(dst.cols*3);
212 float* src_buf = &_src_buf[0];
213 float* dst_buf = &_dst_buf[0];
214 int i, j;
215
216 assert( (cn == 3 || cn == 4) && (dst_cn == 3 || dst_cn == 1) );
217
218 for( i = 0; i < src.rows; i++ )
219 {
220 switch( depth )
221 {
222 case CV_8U:
223 {
224 const uchar* src_row = src.ptr(i);
225 uchar* dst_row = dst.ptr(i);
226
227 for( j = 0; j < cols; j++ )
228 {
229 src_buf[j*3] = src_row[j*cn + blue_idx]*c8u;
230 src_buf[j*3+1] = src_row[j*cn + 1]*c8u;
231 src_buf[j*3+2] = src_row[j*cn + (blue_idx^2)]*c8u;
232 }
233
234 convert_row_bgr2abc_32f_c3( src_buf, dst_buf, cols );
235
236 for( j = 0; j < dst_cols_n; j++ )
237 {
238 int t = cvRound( dst_buf[j] );
239 dst_row[j] = saturate_cast<uchar>(t);
240 }
241 }
242 break;
243 case CV_16U:
244 {
245 const ushort* src_row = src.ptr<ushort>(i);
246 ushort* dst_row = dst.ptr<ushort>(i);
247
248 for( j = 0; j < cols; j++ )
249 {
250 src_buf[j*3] = src_row[j*cn + blue_idx]*c16u;
251 src_buf[j*3+1] = src_row[j*cn + 1]*c16u;
252 src_buf[j*3+2] = src_row[j*cn + (blue_idx^2)]*c16u;
253 }
254
255 convert_row_bgr2abc_32f_c3( src_buf, dst_buf, cols );
256
257 for( j = 0; j < dst_cols_n; j++ )
258 {
259 int t = cvRound( dst_buf[j] );
260 dst_row[j] = saturate_cast<ushort>(t);
261 }
262 }
263 break;
264 case CV_32F:
265 {
266 const float* src_row = src.ptr<float>(i);
267 float* dst_row = dst.ptr<float>(i);
268
269 for( j = 0; j < cols; j++ )
270 {
271 src_buf[j*3] = src_row[j*cn + blue_idx];
272 src_buf[j*3+1] = src_row[j*cn + 1];
273 src_buf[j*3+2] = src_row[j*cn + (blue_idx^2)];
274 }
275
276 convert_row_bgr2abc_32f_c3( src_buf, dst_row, cols );
277 }
278 break;
279 default:
280 assert(0);
281 }
282 }
283}

298void CV_ColorCvtBaseTest::convert_backward( const Mat& src, const Mat& dst, Mat& dst2 )
299{
300 if( custom_inv_transform )
301 {
302 int depth = src.depth();
303 int src_cn = dst.channels(), cn = dst2.channels();
304 int cols_n = src.cols*src_cn, dst_cols = dst.cols;
305 vector<float> _src_buf(src.cols*3);
306 vector<float> _dst_buf(dst.cols*3);
307 float* src_buf = &_src_buf[0];
308 float* dst_buf = &_dst_buf[0];
309 int i, j;
310
311 assert( cn == 3 || cn == 4 );
312
313 for( i = 0; i < src.rows; i++ )
314 {
315 switch( depth )
316 {
317 case CV_8U:
318 {
319 const uchar* src_row = dst.ptr(i);
320 uchar* dst_row = dst2.ptr(i);
321
322 for( j = 0; j < cols_n; j++ )
323 src_buf[j] = src_row[j];
324
325 convert_row_abc2bgr_32f_c3( src_buf, dst_buf, dst_cols );
326
327 for( j = 0; j < dst_cols; j++ )
328 {
329 int b = cvRound( dst_buf[j*3]*255. );
330 int g = cvRound( dst_buf[j*3+1]*255. );
331 int r = cvRound( dst_buf[j*3+2]*255. );
332 dst_row[j*cn + blue_idx] = saturate_cast<uchar>(b);
333 dst_row[j*cn + 1] = saturate_cast<uchar>(g);
334 dst_row[j*cn + (blue_idx^2)] = saturate_cast<uchar>(r);
335 if( cn == 4 )
336 dst_row[j*cn + 3] = 255;
337 }
338 }
339 break;
340 case CV_16U:
341 {
342 const ushort* src_row = dst.ptr<ushort>(i);
343 ushort* dst_row = dst2.ptr<ushort>(i);
344
345 for( j = 0; j < cols_n; j++ )
346 src_buf[j] = src_row[j];
347
348 convert_row_abc2bgr_32f_c3( src_buf, dst_buf, dst_cols );
349
350 for( j = 0; j < dst_cols; j++ )
351 {
352 int b = cvRound( dst_buf[j*3]*65535. );
353 int g = cvRound( dst_buf[j*3+1]*65535. );
354 int r = cvRound( dst_buf[j*3+2]*65535. );
355 dst_row[j*cn + blue_idx] = saturate_cast<ushort>(b);
356 dst_row[j*cn + 1] = saturate_cast<ushort>(g);
357 dst_row[j*cn + (blue_idx^2)] = saturate_cast<ushort>(r);
358 if( cn == 4 )
359 dst_row[j*cn + 3] = 65535;
360 }
361 }
362 break;
363 case CV_32F:
364 {
365 const float* src_row = dst.ptr<float>(i);
366 float* dst_row = dst2.ptr<float>(i);
367
368 convert_row_abc2bgr_32f_c3( src_row, dst_buf, dst_cols );
369
370 for( j = 0; j < dst_cols; j++ )
371 {
372 float b = dst_buf[j*3];
373 float g = dst_buf[j*3+1];
374 float r = dst_buf[j*3+2];
375 dst_row[j*cn + blue_idx] = b;
376 dst_row[j*cn + 1] = g;
377 dst_row[j*cn + (blue_idx^2)] = r;
378 if( cn == 4 )
379 dst_row[j*cn + 3] = 1.f;
380 }
381 }
382 break;
383 default:
384 assert(0);
385 }
386 }
387 }
388 else
389 {
390 int i, j, k;
391 int elem_size = (int)src.elemSize(), elem_size1 = (int)src.elemSize1();
392 int width_n = src.cols*elem_size;
393
394 for( i = 0; i < src.rows; i++ )
395 {
396 memcpy( dst2.ptr(i), src.ptr(i), width_n );
397 if( src.channels() == 4 )
398 {
399 // clear the alpha channel
400 uchar* ptr = dst2.ptr(i) + elem_size1*3;
401 for( j = 0; j < width_n; j += elem_size )
402 {
403 for( k = 0; k < elem_size1; k++ )
404 ptr[j + k] = 0;
405 }
406 }
407 }
408 }
409}

527void CV_ColorYCrCbTest::convert_row_bgr2abc_32f_c3( const float* src_row, float* dst_row, int n )
528{
529 int depth = test_mat[INPUT][0].depth();
530 double scale = depth == CV_8U ? 255 : depth == CV_16U ? 65535 : 1;
531 double bias = depth == CV_8U ? 128 : depth == CV_16U ? 32768 : 0.5;
532
533 double M[] = { 0.299, 0.587, 0.114,
534 0.49981, -0.41853, -0.08128,
535 -0.16864, -0.33107, 0.49970 };
536 int j;
537 for( j = 0; j < 9; j++ )
538 M[j] *= scale;
539
540 for( j = 0; j < n*3; j += 3 )
541 {
542 double r = src_row[j+2];
543 double g = src_row[j+1];
544 double b = src_row[j];
545 double y = M[0]*r + M[1]*g + M[2]*b;
546 double cr = M[3]*r + M[4]*g + M[5]*b + bias;
547 double cb = M[6]*r + M[7]*g + M[8]*b + bias;
548 dst_row[j] = (float)y;
549 dst_row[j+1] = (float)cr;
550 dst_row[j+2] = (float)cb;
551 }
552}

554
555void CV_ColorYCrCbTest::convert_row_abc2bgr_32f_c3( const float* src_row, float* dst_row, int n )
556{
557 int depth = test_mat[INPUT][0].depth();
558 double bias = depth == CV_8U ? 128 : depth == CV_16U ? 32768 : 0.5;
559 double scale = depth == CV_8U ? 1./255 : depth == CV_16U ? 1./65535 : 1;
560 double M[] = { 1, 1.40252, 0,
561 1, -0.71440, -0.34434,
562 1, 0, 1.77305 };
563 int j;
564 for( j = 0; j < 9; j++ )
565 M[j] *= scale;
566
567 for( j = 0; j < n*3; j += 3 )
568 {
569 double y = src_row[j];
570 double cr = src_row[j+1] - bias;
571 double cb = src_row[j+2] - bias;
572 double r = M[0]*y + M[1]*cr + M[2]*cb;
573 double g = M[3]*y + M[4]*cr + M[5]*cb;
574 double b = M[6]*y + M[7]*cr + M[8]*cb;
575 dst_row[j] = (float)b;
576 dst_row[j+1] = (float)g;
577 dst_row[j+2] = (float)r;
578 }

1206void CV_ColorLuvTest::convert_row_abc2bgr_32f_c3( const float* src_row, float* dst_row, int n )
1207{
1208 int depth = test_mat[INPUT][0].depth();
1209 float Lscale = depth == CV_8U ? 100.f/255.f : depth == CV_16U ? 100.f/65535.f : 1.f;
1210 int j;
1211 float M[9];
1212 float un = 4.f*Xn/(Xn + 15.f*1.f + 3*Zn);
1213 float vn = 9.f*1.f/(Xn + 15.f*1.f + 3*Zn);
1214 float u_scale = 1.f, u_bias = 0.f;
1215 float v_scale = 1.f, v_bias = 0.f;
1216
1217 for( j = 0; j < 9; j++ )
1218 M[j] = (float)XYZ2RGB[j];
1219
1220 if( depth == CV_8U )
1221 {
1222 u_scale = 1.f/0.720338983f;
1223 u_bias = 96.5254237f;
1224 v_scale = 1.f/0.973282442f;
1225 v_bias = 136.2595419f;
1226 }
1227
1228 for( j = 0; j < n*3; j += 3 )
1229 {
1230 float L = src_row[j]*Lscale;
1231 float u = (src_row[j+1] - u_bias)*u_scale;
1232 float v = (src_row[j+2] - v_bias)*v_scale;
1233 float X, Y, Z;
1234
1235 if( L >= 8 )
1236 {
1237 Y = (L + 16.f)*(1.f/116.f);
1238 Y = Y*Y*Y;
1239 }
1240 else
1241 {
1242 Y = L * (1.f/903.3f);
1243 if( L == 0 )
1244 L = 0.001f;
1245 }
1246
1247 u = u/(13*L) + un;
1248 v = v/(13*L) + vn;
1249
1250 X = -9*Y*u/((u - 4)*v - u*v);
1251 Z = (9*Y - 15*v*Y - v*X)/(3*v);
1252
1253 float r = M[0]*X + M[1]*Y + M[2]*Z;
1254 float g = M[3]*X + M[4]*Y + M[5]*Z;
1255 float b = M[6]*X + M[7]*Y + M[8]*Z;
1256
1257 dst_row[j] = b;
1258 dst_row[j+1] = g;
1259 dst_row[j+2] = r;
1260 }

1586template<typename T>
1587static void bayer2BGR_(const Mat& src, Mat& dst, int code)
1588{
1589 int i, j, cols = src.cols - 2;
1590 int bi = 0;
1591 int step = (int)(src.step/sizeof(T));
1592
1593 if( code == CV_BayerRG2BGR || code == CV_BayerGR2BGR )
1594 bi ^= 2;
1595
1596 for( i = 1; i < src.rows - 1; i++ )
1597 {
1598 const T* ptr = src.ptr<T>(i) + 1;
1599 T* dst_row = dst.ptr<T>(i) + 3;
1600 int save_code = code;
1601 if( cols <= 0 )
1602 {
1603 dst_row[-3] = dst_row[-2] = dst_row[-1] = 0;
1604 dst_row[cols*3] = dst_row[cols*3+1] = dst_row[cols*3+2] = 0;
1605 continue;
1606 }
1607
1608 for( j = 0; j < cols; j++ )
1609 {
1610 int b, g, r;
1611 if( !(code & 1) )
1612 {
1613 b = ptr[j];
1614 g = (ptr[j-1] + ptr[j+1] + ptr[j-step] + ptr[j+step])>>2;
1615 r = (ptr[j-step-1] + ptr[j-step+1] + ptr[j+step-1] + ptr[j+step+1]) >> 2;
1616 }
1617 else
1618 {
1619 b = (ptr[j-1] + ptr[j+1]) >> 1;
1620 g = ptr[j];
1621 r = (ptr[j-step] + ptr[j+step]) >> 1;
1622 }
1623 code ^= 1;
1624 dst_row[j*3 + bi] = (T)b;
1625 dst_row[j*3 + 1] = (T)g;
1626 dst_row[j*3 + (bi^2)] = (T)r;
1627 }
1628
1629 dst_row[-3] = dst_row[0];
1630 dst_row[-2] = dst_row[1];
1631 dst_row[-1] = dst_row[2];
1632 dst_row[cols*3] = dst_row[cols*3-3];
1633 dst_row[cols*3+1] = dst_row[cols*3-2];
1634 dst_row[cols*3+2] = dst_row[cols*3-1];
1635
1636 code = save_code ^ 1;
1637 bi ^= 2;
1638 }
1639
1640 if( src.rows <= 2 )
1641 {
1642 memset( dst.ptr(), 0, (cols+2)*3*sizeof(T) );
1643 memset( dst.ptr(dst.rows-1), 0, (cols+2)*3*sizeof(T) );
1644 }
1645 else
1646 {
1647 T* top_row = dst.ptr<T>();
1648 T* bottom_row = dst.ptr<T>(dst.rows-1);
1649 int dstep = (int)(dst.step/sizeof(T));
1650
1651 for( j = 0; j < (cols+2)*3; j++ )
1652 {
1653 top_row[j] = top_row[j + dstep];
1654 bottom_row[j] = bottom_row[j - dstep];
1655 }
1656 }
1657}

1805 for (int x = 0; x < ssize.width; ++x)
1806 {
1807 if ((x + y) % 2 == 0)
1808 bayer.at<uchar>(y, x) = src.at<Vec3b>(y, x)[1];
1809 else if (x % 2)
1810 bayer.at<uchar>(y, x) = src.at<Vec3b>(y, x)[0];
1811 else
1812 bayer.at<uchar>(y, x) = src.at<Vec3b>(y, x)[2];
1813 }
1814 type = CV_BayerGR2BGR_VNG;
1815 }
1816
1817 // calculating a dst image
1818 cvtColor(bayer, dst, type);
1819
1820 // reading a reference image
1821 full_path = parent_path + pattern[i] + image_name;
1822 reference = imread(full_path, CV_LOAD_IMAGE_UNCHANGED);
1823 if (reference.data == NULL)
1824 {
1825 imwrite(full_path, dst);
1826 continue;
1827 }
1828
1829 if (reference.depth() != dst.depth() || reference.channels() != dst.channels() ||
1830 reference.size() != dst.size())
1831 {
1832 ts.set_failed_test_info(cvtest::TS::FAIL_MISMATCH);
1833 ts.printf(cvtest::TS::SUMMARY, "\nReference channels: %d\n"
1834 "Actual channels: %d\n", reference.channels(), dst.channels());
1835 ts.printf(cvtest::TS::SUMMARY, "\nReference depth: %d\n"
1836 "Actual depth: %d\n", reference.depth(), dst.depth());
1837 ts.printf(cvtest::TS::SUMMARY, "\nReference rows: %d\n"
1838 "Actual rows: %d\n", reference.rows, dst.rows);
1839 ts.printf(cvtest::TS::SUMMARY, "\nReference cols: %d\n"
1840 "Actual cols: %d\n", reference.cols, dst.cols);
1841 ts.set_gtest_status();
1842
1843 return;
1844 }
1845
1846 Mat diff;
1847 absdiff(reference, dst, diff);
1848
1849 int nonZero = countNonZero(diff.reshape(1) > 1);
1850 if (nonZero != 0)
1851 {
1852 ts.set_failed_test_info(cvtest::TS::FAIL_BAD_ACCURACY);
1853 ts.printf(cvtest::TS::SUMMARY, "\nCount non zero in absdiff: %d\n", nonZero);
1854 ts.set_gtest_status();
1855 return;
1856 }
1857 }
1858}

1861void GetTestMatrix(Mat& src)
1862{
1863 Size ssize(1000, 1000);
1864 src.create(ssize, CV_32FC3);
1865 int szm = ssize.width - 1;
1866 float pi2 = 2 * 3.1415f;
1867 // Generate a pretty test image
1868 for (int i = 0; i < ssize.height; i++)
1869 {
1870 for (int j = 0; j < ssize.width; j++)
1871 {
1872 float b = (1 + cos((szm - i) * (szm - j) * pi2 / (10 * float(szm)))) / 2;
1873 float g = (1 + cos((szm - i) * j * pi2 / (10 * float(szm)))) / 2;
1874 float r = (1 + sin(i * j * pi2 / (10 * float(szm)))) / 2;
1875
1876 // The following lines aren't necessary, but just to prove that
1877 // the BGR values all lie in [0,1]...
1878 if (b < 0) b = 0; else if (b > 1) b = 1;
1879 if (g < 0) g = 0; else if (g > 1) g = 1;
1880 if (r < 0) r = 0; else if (r > 1) r = 1;
1881 src.at<cv::Vec3f>(i, j) = cv::Vec3f(b, g, r);
1882 }
1883 }
1884}

1886void validate_result(const Mat& reference, const Mat& actual, const Mat& src = Mat(), int mode = -1)
1887{
1888 cvtest::TS* ts = cvtest::TS::ptr();
1889 Size ssize = reference.size();
1890
1891 int cn = reference.channels();
1892 ssize.width *= cn;
1893 bool next = true;
1894
1895 for (int y = 0; y < ssize.height && next; ++y)
1896 {
1897 const float* rD = reference.ptr<float>(y);
1898 const float* D = actual.ptr<float>(y);
1899 for (int x = 0; x < ssize.width && next; ++x)
1900 if (fabs(rD[x] - D[x]) > 0.0001f)
1901 {
1902 next = false;
1903 ts->printf(cvtest::TS::SUMMARY, "Error in: (%d, %d)\n", x / cn, y);
1904 ts->printf(cvtest::TS::SUMMARY, "Reference value: %f\n", rD[x]);
1905 ts->printf(cvtest::TS::SUMMARY, "Actual value: %f\n", D[x]);
1906 if (!src.empty())
1907 ts->printf(cvtest::TS::SUMMARY, "Src value: %f\n", src.ptr<float>(y)[x]);
1908 ts->printf(cvtest::TS::SUMMARY, "Size: (%d, %d)\n", reference.rows, reference.cols);
1909
1910 if (mode >= 0)
1911 {
1912 cv::Mat lab;
1913 cv::cvtColor(src, lab, mode);
1914 std::cout << "lab: " << lab(cv::Rect(y, x / cn, 1, 1)) << std::endl;
1915 }
1916 std::cout << "src: " << src(cv::Rect(y, x / cn, 1, 1)) << std::endl;
1917
1918 ts->set_failed_test_info(cvtest::TS::FAIL_BAD_ACCURACY);
1919 ts->set_gtest_status();
1920 }
1921 }
