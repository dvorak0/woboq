49class CV_WatershedTest : public cvtest::BaseTest
50{
51public:
52 CV_WatershedTest();
53 ~CV_WatershedTest();
54protected:
55 void run(int);
56};

58CV_WatershedTest::CV_WatershedTest() {}
59CV_WatershedTest::~CV_WatershedTest() {}
60
61void CV_WatershedTest::run( int /* start_from */)
62{
63 string exp_path = string(ts->get_data_path()) + "watershed/wshed_exp.png";
64 Mat exp = imread(exp_path, 0);
65 Mat orig = imread(string(ts->get_data_path()) + "inpaint/orig.png");
66 FileStorage fs(string(ts->get_data_path()) + "watershed/comp.xml", FileStorage::READ);
67
68 if (orig.empty() || !fs.isOpened())
69 {
70 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_TEST_DATA );
71 return;
72 }
73
74 CvSeq* cnts = (CvSeq*)fs["contours"].readObj();
75
76 Mat markers(orig.size(), CV_32SC1);
77 markers = Scalar(0);
78 IplImage iplmrks = markers;
79
80 vector<unsigned char> colors(1);
81 for(int i = 0; cnts != 0; cnts = cnts->h_next, ++i )
82 {
83 cvDrawContours( &iplmrks, cnts, Scalar::all(i + 1), Scalar::all(i + 1), -1, CV_FILLED);
84 Point* p = (Point*)cvGetSeqElem(cnts, 0);
85
86 //expected image was added with 1 in order to save to png
87 //so now we substract 1 to get real color
88 if(exp.data)
89 colors.push_back(exp.ptr(p->y)[p->x] - 1);
90 }
91 fs.release();
92 const int compNum = (int)(colors.size() - 1);
93
94 watershed(orig, markers);
95
96 for(int j = 0; j < markers.rows; ++j)
97 {
98 int* line = markers.ptr<int>(j);
99 for(int i = 0; i < markers.cols; ++i)
100 {
101 int& pixel = line[i];
102
103 if (pixel == -1) // border
104 continue;
105
106 if (pixel <= 0 || pixel > compNum)
107 continue; // bad result, doing nothing and going to get error latter;
108
109 // repaint in saved color to compare with expected;
110 if(exp.data)
111 pixel = colors[pixel];
112 }
113 }
114
115 Mat markers8U;
116 markers.convertTo(markers8U, CV_8U, 1, 1);
117
118 if( exp.empty() || orig.size() != exp.size() )
119 {
120 imwrite(exp_path, markers8U);
121 exp = markers8U;
122 }
123
124 if (0 != norm(markers8U, exp, NORM_INF))
125 {
126 ts->set_failed_test_info( cvtest::TS::FAIL_MISMATCH );
127 return;
128 }
129 ts->set_failed_test_info(cvtest::TS::OK);
130}
