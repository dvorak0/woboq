77void CV_MomentsTest::get_minmax_bounds( int i, int j, int type, Scalar& low, Scalar& high )
78{
79 cvtest::ArrayTest::get_minmax_bounds( i, j, type, low, high );
80 int depth = CV_MAT_DEPTH(type);
81
82 if( depth == CV_16U )
83 {
84 low = Scalar::all(0);
85 high = Scalar::all(1000);
86 }
87 else if( depth == CV_16S )
88 {
89 low = Scalar::all(-1000);
90 high = Scalar::all(1000);
91 }
92 else if( depth == CV_32F )
93 {
94 low = Scalar::all(-1);
95 high = Scalar::all(1);
96 }
97}

100void CV_MomentsTest::get_test_array_types_and_sizes( int test_case_idx,
101 vector<vector<Size> >& sizes, vector<vector<int> >& types )
102{
103 RNG& rng = ts->get_rng();
104 cvtest::ArrayTest::get_test_array_types_and_sizes( test_case_idx, sizes, types );
105 int cn = cvtest::randInt(rng) % 4 + 1;
106 int depth = cvtest::randInt(rng) % 4;
107 depth = depth == 0 ? CV_8U : depth == 1 ? CV_16U : depth == 2 ? CV_16S : CV_32F;
108 if( cn == 2 )
109 cn = 1;
110
111 types[INPUT][0] = CV_MAKETYPE(depth, cn);
112 types[OUTPUT][0] = types[REF_OUTPUT][0] = CV_64FC1;
113 sizes[OUTPUT][0] = sizes[REF_OUTPUT][0] = cvSize(MOMENT_COUNT,1);
114
115 is_binary = cvtest::randInt(rng) % 2 != 0;
116 coi = 0;
117 cvmat_allowed = true;
118 if( cn > 1 )
119 {
120 coi = cvtest::randInt(rng) % cn;
121 cvmat_allowed = false;
122 }
123}

161void CV_MomentsTest::prepare_to_validation( int /*test_case_idx*/ )
162{
163 Mat& src = test_mat[INPUT][0];
164 CvMoments m;
165 double* mdata = test_mat[REF_OUTPUT][0].ptr<double>();
166 int depth = src.depth();
167 int cn = src.channels();
168 int i, y, x, cols = src.cols;
169 double xc = 0., yc = 0.;
170
171 memset( &m, 0, sizeof(m));
172
173 for( y = 0; y < src.rows; y++ )
174 {
175 double s0 = 0, s1 = 0, s2 = 0, s3 = 0;
176 uchar* ptr = src.ptr(y);
177 for( x = 0; x < cols; x++ )
178 {
179 double val;
180 if( depth == CV_8U )
181 val = ptr[x*cn + coi];
182 else if( depth == CV_16U )
183 val = ((ushort*)ptr)[x*cn + coi];
184 else if( depth == CV_16S )
185 val = ((short*)ptr)[x*cn + coi];
186 else
187 val = ((float*)ptr)[x*cn + coi];
188
189 if( is_binary )
190 val = val != 0;
191
192 s0 += val;
193 s1 += val*x;
194 s2 += val*x*x;
195 s3 += ((val*x)*x)*x;
196 }
197
198 m.m00 += s0;
199 m.m01 += s0*y;
200 m.m02 += (s0*y)*y;
201 m.m03 += ((s0*y)*y)*y;
202
203 m.m10 += s1;
204 m.m11 += s1*y;
205 m.m12 += (s1*y)*y;
206
207 m.m20 += s2;
208 m.m21 += s2*y;
209
210 m.m30 += s3;
211 }
212
213 if( m.m00 != 0 )
214 {
215 xc = m.m10/m.m00, yc = m.m01/m.m00;
216 m.inv_sqrt_m00 = 1./sqrt(fabs(m.m00));
217 }
218
219 for( y = 0; y < src.rows; y++ )
220 {
221 double s0 = 0, s1 = 0, s2 = 0, s3 = 0, y1 = y - yc;
222 uchar* ptr = src.ptr(y);
223 for( x = 0; x < cols; x++ )
224 {
225 double val, x1 = x - xc;
226 if( depth == CV_8U )
227 val = ptr[x*cn + coi];
228 else if( depth == CV_16U )
229 val = ((ushort*)ptr)[x*cn + coi];
230 else if( depth == CV_16S )
231 val = ((short*)ptr)[x*cn + coi];
232 else
233 val = ((float*)ptr)[x*cn + coi];
234
235 if( is_binary )
236 val = val != 0;
237
238 s0 += val;
239 s1 += val*x1;
240 s2 += val*x1*x1;
241 s3 += ((val*x1)*x1)*x1;
242 }
243
244 m.mu02 += s0*y1*y1;
245 m.mu03 += ((s0*y1)*y1)*y1;
246
247 m.mu11 += s1*y1;
248 m.mu12 += (s1*y1)*y1;
249
250 m.mu20 += s2;
251 m.mu21 += s2*y1;
252
253 m.mu30 += s3;
254 }
255
256 memcpy( mdata, &m, sizeof(m));
257 mdata += sizeof(m)/sizeof(m.m00);
258
259 /* calc normalized moments */
260 {
261 double inv_m00 = m.inv_sqrt_m00*m.inv_sqrt_m00;
262 double s2 = inv_m00*inv_m00; /* 1./(m00 ^ (2/2 + 1)) */
263 double s3 = s2*m.inv_sqrt_m00; /* 1./(m00 ^ (3/2 + 1)) */
264
265 mdata[0] = m.mu20 * s2;
266 mdata[1] = m.mu11 * s2;
267 mdata[2] = m.mu02 * s2;
268
269 mdata[3] = m.mu30 * s3;
270 mdata[4] = m.mu21 * s3;
271 mdata[5] = m.mu12 * s3;
272 mdata[6] = m.mu03 * s3;
273 }
274
275 double* a = test_mat[REF_OUTPUT][0].ptr<double>();
276 double* b = test_mat[OUTPUT][0].ptr<double>();
277 for( i = 0; i < MOMENT_COUNT; i++ )
278 {
279 if( fabs(a[i]) < 1e-3 )
280 a[i] = 0;
281 if( fabs(b[i]) < 1e-3 )
282 b[i] = 0;
283 }
284}

358void CV_HuMomentsTest::prepare_to_validation( int /*test_case_idx*/ )
359{
360 CvMoments* m = (CvMoments*)test_mat[INPUT][0].data;
361 CvHuMoments* hu = (CvHuMoments*)test_mat[REF_OUTPUT][0].data;
362
363 double inv_m00 = m->inv_sqrt_m00*m->inv_sqrt_m00;
364 double s2 = inv_m00*inv_m00; /* 1./(m00 ^ (2/2 + 1)) */
365 double s3 = s2*m->inv_sqrt_m00; /* 1./(m00 ^ (3/2 + 1)) */
366
367 double nu20 = m->mu20 * s2;
368 double nu11 = m->mu11 * s2;
369 double nu02 = m->mu02 * s2;
370
371 double nu30 = m->mu30 * s3;
372 double nu21 = m->mu21 * s3;
373 double nu12 = m->mu12 * s3;
374 double nu03 = m->mu03 * s3;
375
376 #undef sqr
377 #define sqr(a) ((a)*(a))
378
379 hu->hu1 = nu20 + nu02;
380 hu->hu2 = sqr(nu20 - nu02) + 4*sqr(nu11);
381 hu->hu3 = sqr(nu30 - 3*nu12) + sqr(3*nu21 - nu03);
382 hu->hu4 = sqr(nu30 + nu12) + sqr(nu21 + nu03);
383 hu->hu5 = (nu30 - 3*nu12)*(nu30 + nu12)*(sqr(nu30 + nu12) - 3*sqr(nu21 + nu03)) +
384 (3*nu21 - nu03)*(nu21 + nu03)*(3*sqr(nu30 + nu12) - sqr(nu21 + nu03));
385 hu->hu6 = (nu20 - nu02)*(sqr(nu30 + nu12) - sqr(nu21 + nu03)) +
386 4*nu11*(nu30 + nu12)*(nu21 + nu03);
387 hu->hu7 = (3*nu21 - nu03)*(nu30 + nu12)*(sqr(nu30 + nu12) - 3*sqr(nu21 + nu03)) +
388 (3*nu12 - nu30)*(nu21 + nu03)*(3*sqr(nu30 + nu12) - sqr(nu21 + nu03));
389}
