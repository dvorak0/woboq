63/**
64 * Randomized kd-tree index
65 *
66 * Contains the k-d trees and other information for indexing a set of points
67 * for nearest-neighbor matching.
68 */
69template <typename Distance>
70class KDTreeIndex : public NNIndex<Distance>
71{
72public:
73 typedef typename Distance::ElementType ElementType;
74 typedef typename Distance::ResultType DistanceType;
75
76
77 /**
78 * KDTree constructor
79 *
80 * Params:
81 * inputData = dataset with the input features
82 * params = parameters passed to the kdtree algorithm
83 */
84 KDTreeIndex(const Matrix<ElementType>& inputData, const IndexParams& params = KDTreeIndexParams(),
85 Distance d = Distance() ) :
86 dataset_(inputData), index_params_(params), distance_(d)
87 {
88 size_ = dataset_.rows;
89 veclen_ = dataset_.cols;
90
91 trees_ = get_param(index_params_,"trees",4);
92 tree_roots_ = new NodePtr[trees_];
93
94 // Create a permutable array of indices to the input vectors.
95 vind_.resize(size_);
96 for (size_t i = 0; i < size_; ++i) {
97 vind_[i] = int(i);
98 }
99
100 mean_ = new DistanceType[veclen_];
101 var_ = new DistanceType[veclen_];
102 }
103
104
105 KDTreeIndex(const KDTreeIndex&);
106 KDTreeIndex& operator=(const KDTreeIndex&);
107
108 /**
109 * Standard destructor
110 */
111 ~KDTreeIndex()
112 {
113 if (tree_roots_!=NULL) {
114 delete[] tree_roots_;
115 }
116 delete[] mean_;
117 delete[] var_;
118 }

120 /**
121 * Dummy implementation for other algorithms of addable indexes after that.
122 */
123 void addIndex(const Matrix<ElementType>& /*wholeData*/, const Matrix<ElementType>& /*additionalData*/)
124 {
125 }

127 /**
128 * Builds the index
129 */
130 void buildIndex()
131 {
132 /* Construct the randomized trees. */
133 for (int i = 0; i < trees_; i++) {
134 /* Randomize the order of vectors to allow for unbiased sampling. */
135 std::random_shuffle(vind_.begin(), vind_.end());
136 tree_roots_[i] = divideTree(&vind_[0], int(size_) );
137 }
138 }

197 /**
198 * Find set of nearest neighbors to vec. Their indices are stored inside
199 * the result object.
200 *
201 * Params:
202 * result = the result object in which the indices of the nearest-neighbors are stored
203 * vec = the vector for which to search the nearest neighbors
204 * maxCheck = the maximum number of restarts (in a best-bin-first manner)
205 */
206 void findNeighbors(ResultSet<DistanceType>& result, const ElementType* vec, const SearchParams& searchParams)
207 {
208 int maxChecks = get_param(searchParams,"checks", 32);
209 float epsError = 1+get_param(searchParams,"eps",0.0f);
210
211 if (maxChecks==FLANN_CHECKS_UNLIMITED) {
212 getExactNeighbors(result, vec, epsError);
213 }
214 else {
215 getNeighbors(result, vec, maxChecks, epsError);
216 }
217 }

274 /**
275 * Create a tree node that subdivides the list of vecs from vind[first]
276 * to vind[last]. The routine is called recursively on each sublist.
277 * Place a pointer to this new tree node in the location pTree.
278 *
279 * Params: pTree = the new node to create
280 * first = index of the first vector
281 * last = index of the last vector
282 */
283 NodePtr divideTree(int* ind, int count)
284 {
285 NodePtr node = pool_.allocate<Node>(); // allocate memory
286
287 /* If too few exemplars remain, then make this a leaf node. */
288 if ( count == 1) {
289 node->child1 = node->child2 = NULL; /* Mark as leaf node. */
290 node->divfeat = *ind; /* Store index of this vec. */
291 }
292 else {
293 int idx;
294 int cutfeat;
295 DistanceType cutval;
296 meanSplit(ind, count, idx, cutfeat, cutval);
297
298 node->divfeat = cutfeat;
299 node->divval = cutval;
300 node->child1 = divideTree(ind, idx);
301 node->child2 = divideTree(ind+idx, count-idx);
302 }
303
304 return node;
305 }

308 /**
309 * Choose which feature to use in order to subdivide this set of vectors.
310 * Make a random choice among those with the highest variance, and use
311 * its variance as the threshold value.
312 */
313 void meanSplit(int* ind, int count, int& index, int& cutfeat, DistanceType& cutval)
314 {
315 memset(mean_,0,veclen_*sizeof(DistanceType));
316 memset(var_,0,veclen_*sizeof(DistanceType));
317
318 /* Compute mean values. Only the first SAMPLE_MEAN values need to be
319 sampled to get a good estimate.
320 */
321 int cnt = std::min((int)SAMPLE_MEAN+1, count);
322 for (int j = 0; j < cnt; ++j) {
323 ElementType* v = dataset_[ind[j]];
324 for (size_t k=0; k<veclen_; ++k) {
325 mean_[k] += v[k];
326 }
327 }
328 for (size_t k=0; k<veclen_; ++k) {
329 mean_[k] /= cnt;
330 }
331
332 /* Compute variances (no need to divide by count). */
333 for (int j = 0; j < cnt; ++j) {
334 ElementType* v = dataset_[ind[j]];
335 for (size_t k=0; k<veclen_; ++k) {
336 DistanceType dist = v[k] - mean_[k];
337 var_[k] += dist * dist;
338 }
339 }
340 /* Select one of the highest variance indices at random. */
341 cutfeat = selectDivision(var_);
342 cutval = mean_[cutfeat];
343
344 int lim1, lim2;
345 planeSplit(ind, count, cutfeat, cutval, lim1, lim2);
346
347 if (lim1>count/2) index = lim1;
348 else if (lim2<count/2) index = lim2;
349 else index = count/2;
350
351 /* If either list is empty, it means that all remaining features
352 * are identical. Split in the middle to maintain a balanced tree.
353 */
354 if ((lim1==count)||(lim2==0)) index = count/2;
355 }
356
357
358 /**
359 * Select the top RAND_DIM largest values from v and return the index of
360 * one of these selected at random.
361 */
362 int selectDivision(DistanceType* v)
363 {
364 int num = 0;
365 size_t topind[RAND_DIM];
366
367 /* Create a list of the indices of the top RAND_DIM values. */
368 for (size_t i = 0; i < veclen_; ++i) {
369 if ((num < RAND_DIM)||(v[i] > v[topind[num-1]])) {
370 /* Put this element at end of topind. */
371 if (num < RAND_DIM) {
372 topind[num++] = i; /* Add to list. */
373 }
374 else {
375 topind[num-1] = i; /* Replace last element. */
376 }
377 /* Bubble end value down to right location by repeated swapping. */
378 int j = num - 1;
379 while (j > 0 && v[topind[j]] > v[topind[j-1]]) {
380 std::swap(topind[j], topind[j-1]);
381 --j;
382 }
383 }
384 }
385 /* Select a random integer in range [0,num-1], and return that index. */
386 int rnd = rand_int(num);
387 return (int)topind[rnd];
388 }

391 /**
392 * Subdivide the list of points by a plane perpendicular on axe corresponding
393 * to the 'cutfeat' dimension at 'cutval' position.
394 *
395 * On return:
396 * dataset[ind[0..lim1-1]][cutfeat]<cutval
397 * dataset[ind[lim1..lim2-1]][cutfeat]==cutval
398 * dataset[ind[lim2..count]][cutfeat]>cutval
399 */
400 void planeSplit(int* ind, int count, int cutfeat, DistanceType cutval, int& lim1, int& lim2)
401 {
402 /* Move vector indices for left subtree to front of list. */
403 int left = 0;
404 int right = count-1;
405 for (;; ) {
406 while (left<=right && dataset_[ind[left]][cutfeat]<cutval) ++left;
407 while (left<=right && dataset_[ind[right]][cutfeat]>=cutval) --right;
408 if (left>right) break;
409 std::swap(ind[left], ind[right]); ++left; --right;
410 }
411 lim1 = left;
412 right = count-1;
413 for (;; ) {
414 while (left<=right && dataset_[ind[left]][cutfeat]<=cutval) ++left;
415 while (left<=right && dataset_[ind[right]][cutfeat]>cutval) --right;
416 if (left>right) break;
417 std::swap(ind[left], ind[right]); ++left; --right;
418 }
419 lim2 = left;
420 }

422 /**
423 * Performs an exact nearest neighbor search. The exact search performs a full
424 * traversal of the tree.
425 */
426 void getExactNeighbors(ResultSet<DistanceType>& result, const ElementType* vec, float epsError)
427 {
428 // checkID -= 1; /* Set a different unique ID for each search. */
429
430 if (trees_ > 1) {
431 fprintf(stderr,"It doesn't make any sense to use more than one tree for exact search");
432 }
433 if (trees_>0) {
434 searchLevelExact(result, vec, tree_roots_[0], 0.0, epsError);
435 }
436 assert(result.full());
437 }
438
439 /**
440 * Performs the approximate nearest-neighbor search. The search is approximate
441 * because the tree traversal is abandoned after a given number of descends in
442 * the tree.
443 */
444 void getNeighbors(ResultSet<DistanceType>& result, const ElementType* vec, int maxCheck, float epsError)
445 {
446 int i;
447 BranchSt branch;
448
449 int checkCount = 0;
450 Heap<BranchSt>* heap = new Heap<BranchSt>((int)size_);
451 DynamicBitset checked(size_);
452
453 /* Search once through each tree down to root. */
454 for (i = 0; i < trees_; ++i) {
455 searchLevel(result, vec, tree_roots_[i], 0, checkCount, maxCheck, epsError, heap, checked);
456 }
457
458 /* Keep searching other branches from heap until finished. */
459 while ( heap->popMin(branch) && (checkCount < maxCheck || !result.full() )) {
460 searchLevel(result, vec, branch.node, branch.mindist, checkCount, maxCheck, epsError, heap, checked);
461 }
462
463 delete heap;
464
465 assert(result.full());
466 }

469 /**
470 * Search starting from a given node of the tree. Based on any mismatches at
471 * higher levels, all exemplars below this level must have a distance of
472 * at least "mindistsq".
473 */
474 void searchLevel(ResultSet<DistanceType>& result_set, const ElementType* vec, NodePtr node, DistanceType mindist, int& checkCount, int maxCheck,
475 float epsError, Heap<BranchSt>* heap, DynamicBitset& checked)
476 {
477 if (result_set.worstDist()<mindist) {
478 // printf("Ignoring branch, too far\n");
479 return;
480 }
481
482 /* If this is a leaf node, then do check and return. */
483 if ((node->child1 == NULL)&&(node->child2 == NULL)) {
484 /* Do not check same node more than once when searching multiple trees.
485 Once a vector is checked, we set its location in vind to the
486 current checkID.
487 */
488 int index = node->divfeat;
489 if ( checked.test(index) || ((checkCount>=maxCheck)&& result_set.full()) ) return;
490 checked.set(index);
491 checkCount++;
492
493 DistanceType dist = distance_(dataset_[index], vec, veclen_);
494 result_set.addPoint(dist,index);
495
496 return;
497 }
498
499 /* Which child branch should be taken first? */
500 ElementType val = vec[node->divfeat];
501 DistanceType diff = val - node->divval;
502 NodePtr bestChild = (diff < 0) ? node->child1 : node->child2;
503 NodePtr otherChild = (diff < 0) ? node->child2 : node->child1;
504
505 /* Create a branch record for the branch not taken. Add distance
506 of this feature boundary (we don't attempt to correct for any
507 use of this feature in a parent node, which is unlikely to
508 happen and would have only a small effect). Don't bother
509 adding more branches to heap after halfway point, as cost of
510 adding exceeds their value.
511 */
512
513 DistanceType new_distsq = mindist + distance_.accum_dist(val, node->divval, node->divfeat);
514 // if (2 * checkCount < maxCheck || !result.full()) {
515 if ((new_distsq*epsError < result_set.worstDist())|| !result_set.full()) {
516 heap->insert( BranchSt(otherChild, new_distsq) );
517 }
518
519 /* Call recursively to search next level down. */
520 searchLevel(result_set, vec, bestChild, mindist, checkCount, maxCheck, epsError, heap, checked);
521 }
522
523 /**
524 * Performs an exact search in the tree starting from a node.
525 */
526 void searchLevelExact(ResultSet<DistanceType>& result_set, const ElementType* vec, const NodePtr node, DistanceType mindist, const float epsError)
527 {
528 /* If this is a leaf node, then do check and return. */
529 if ((node->child1 == NULL)&&(node->child2 == NULL)) {
530 int index = node->divfeat;
531 DistanceType dist = distance_(dataset_[index], vec, veclen_);
532 result_set.addPoint(dist,index);
533 return;
534 }
535
536 /* Which child branch should be taken first? */
537 ElementType val = vec[node->divfeat];
538 DistanceType diff = val - node->divval;
539 NodePtr bestChild = (diff < 0) ? node->child1 : node->child2;
540 NodePtr otherChild = (diff < 0) ? node->child2 : node->child1;
541
542 /* Create a branch record for the branch not taken. Add distance
543 of this feature boundary (we don't attempt to correct for any
544 use of this feature in a parent node, which is unlikely to
545 happen and would have only a small effect). Don't bother
546 adding more branches to heap after halfway point, as cost of
547 adding exceeds their value.
548 */
549
550 DistanceType new_distsq = mindist + distance_.accum_dist(val, node->divval, node->divfeat);
551
552 /* Call recursively to search next level down. */
553 searchLevelExact(result_set, vec, bestChild, mindist, epsError);
554
555 if (new_distsq*epsError<=result_set.worstDist()) {
556 searchLevelExact(result_set, vec, otherChild, new_distsq, epsError);
557 }
558 }

563 enum
564 {
565 /**
566 * To improve efficiency, only SAMPLE_MEAN random values are used to
567 * compute the mean and variance at each level when building a tree.
568 * A value of 100 seems to perform as well as using all values.
569 */
570 SAMPLE_MEAN = 100,
571 /**
572 * Top random dimensions to consider
573 *
574 * When creating random trees, the dimension on which to subdivide is
575 * selected at random from among the top RAND_DIM dimensions with the
576 * highest variance. A value of 5 works well.
577 */
578 RAND_DIM=5
579 };
