99 /**
100 * Chooses the initial centers in the k-means clustering in a random manner.
101 *
102 * Params:
103 * k = number of centers
104 * vecs = the dataset of points
105 * indices = indices in the dataset
106 * indices_length = length of indices vector
107 *
108 */
109 void chooseCentersRandom(int k, int* dsindices, int indices_length, int* centers, int& centers_length)
110 {
111 UniqueRandom r(indices_length);
112
113 int index;
114 for (index=0; index<k; ++index) {
115 bool duplicate = true;
116 int rnd;
117 while (duplicate) {
118 duplicate = false;
119 rnd = r.next();
120 if (rnd<0) {
121 centers_length = index;
122 return;
123 }
124
125 centers[index] = dsindices[rnd];
126
127 for (int j=0; j<index; ++j) {
128 DistanceType sq = distance(dataset[centers[index]], dataset[centers[j]], dataset.cols);
129 if (sq<1e-16) {
130 duplicate = true;
131 }
132 }
133 }
134 }
135
136 centers_length = index;
137 }

140 /**
141 * Chooses the initial centers in the k-means using Gonzales' algorithm
142 * so that the centers are spaced apart from each other.
143 *
144 * Params:
145 * k = number of centers
146 * vecs = the dataset of points
147 * indices = indices in the dataset
148 * Returns:
149 */
150 void chooseCentersGonzales(int k, int* dsindices, int indices_length, int* centers, int& centers_length)
151 {
152 int n = indices_length;
153
154 int rnd = rand_int(n);
155 assert(rnd >=0 && rnd < n);
156
157 centers[0] = dsindices[rnd];
158
159 int index;
160 for (index=1; index<k; ++index) {
161
162 int best_index = -1;
163 DistanceType best_val = 0;
164 for (int j=0; j<n; ++j) {
165 DistanceType dist = distance(dataset[centers[0]],dataset[dsindices[j]],dataset.cols);
166 for (int i=1; i<index; ++i) {
167 DistanceType tmp_dist = distance(dataset[centers[i]],dataset[dsindices[j]],dataset.cols);
168 if (tmp_dist<dist) {
169 dist = tmp_dist;
170 }
171 }
172 if (dist>best_val) {
173 best_val = dist;
174 best_index = j;
175 }
176 }
177 if (best_index!=-1) {
178 centers[index] = dsindices[best_index];
179 }
180 else {
181 break;
182 }
183 }
184 centers_length = index;
185 }

188 /**
189 * Chooses the initial centers in the k-means using the algorithm
190 * proposed in the KMeans++ paper:
191 * Arthur, David; Vassilvitskii, Sergei - k-means++: The Advantages of Careful Seeding
192 *
193 * Implementation of this function was converted from the one provided in Arthur's code.
194 *
195 * Params:
196 * k = number of centers
197 * vecs = the dataset of points
198 * indices = indices in the dataset
199 * Returns:
200 */
201 void chooseCentersKMeanspp(int k, int* dsindices, int indices_length, int* centers, int& centers_length)
202 {
203 int n = indices_length;
204
205 double currentPot = 0;
206 DistanceType* closestDistSq = new DistanceType[n];
207
208 // Choose one random center and set the closestDistSq values
209 int index = rand_int(n);
210 assert(index >=0 && index < n);
211 centers[0] = dsindices[index];
212
213 // Computing distance^2 will have the advantage of even higher probability further to pick new centers
214 // far from previous centers (and this complies to "k-means++: the advantages of careful seeding" article)
215 for (int i = 0; i < n; i++) {
216 closestDistSq[i] = distance(dataset[dsindices[i]], dataset[dsindices[index]], dataset.cols);
217 closestDistSq[i] = ensureSquareDistance<Distance>( closestDistSq[i] );
218 currentPot += closestDistSq[i];
219 }
220
221
222 const int numLocalTries = 1;
223
224 // Choose each center
225 int centerCount;
226 for (centerCount = 1; centerCount < k; centerCount++) {
227
228 // Repeat several trials
229 double bestNewPot = -1;
230 int bestNewIndex = 0;
231 for (int localTrial = 0; localTrial < numLocalTries; localTrial++) {
232
233 // Choose our center - have to be slightly careful to return a valid answer even accounting
234 // for possible rounding errors
235 double randVal = rand_double(currentPot);
236 for (index = 0; index < n-1; index++) {
237 if (randVal <= closestDistSq[index]) break;
238 else randVal -= closestDistSq[index];
239 }
240
241 // Compute the new potential
242 double newPot = 0;
243 for (int i = 0; i < n; i++) {
244 DistanceType dist = distance(dataset[dsindices[i]], dataset[dsindices[index]], dataset.cols);
245 newPot += std::min( ensureSquareDistance<Distance>(dist), closestDistSq[i] );
246 }
247
248 // Store the best result
249 if ((bestNewPot < 0)||(newPot < bestNewPot)) {
250 bestNewPot = newPot;
251 bestNewIndex = index;
252 }
253 }
254
255 // Add the appropriate center
256 centers[centerCount] = dsindices[bestNewIndex];
257 currentPot = bestNewPot;
258 for (int i = 0; i < n; i++) {
259 DistanceType dist = distance(dataset[dsindices[i]], dataset[dsindices[bestNewIndex]], dataset.cols);
260 closestDistSq[i] = std::min( ensureSquareDistance<Distance>(dist), closestDistSq[i] );
261 }
262 }
263
264 centers_length = centerCount;
265
266 delete[] closestDistSq;
267 }

273 /**
274 * Index constructor
275 *
276 * Params:
277 * inputData = dataset with the input features
278 * params = parameters passed to the hierarchical k-means algorithm
279 */
280 HierarchicalClusteringIndex(const Matrix<ElementType>& inputData, const IndexParams& index_params = HierarchicalClusteringIndexParams(),
281 Distance d = Distance())
282 : dataset(inputData), params(index_params), root(NULL), indices(NULL), distance(d)
283 {
284 memoryCounter = 0;
285
286 size_ = dataset.rows;
287 veclen_ = dataset.cols;
288
289 branching_ = get_param(params,"branching",32);
290 centers_init_ = get_param(params,"centers_init", FLANN_CENTERS_RANDOM);
291 trees_ = get_param(params,"trees",4);
292 leaf_size_ = get_param(params,"leaf_size",100);
293
294 if (centers_init_==FLANN_CENTERS_RANDOM) {
295 chooseCenters = &HierarchicalClusteringIndex::chooseCentersRandom;
296 }
297 else if (centers_init_==FLANN_CENTERS_GONZALES) {
298 chooseCenters = &HierarchicalClusteringIndex::chooseCentersGonzales;
299 }
300 else if (centers_init_==FLANN_CENTERS_KMEANSPP) {
301 chooseCenters = &HierarchicalClusteringIndex::chooseCentersKMeanspp;
302 }
303 else {
304 throw FLANNException("Unknown algorithm for choosing initial centers.");
305 }
306
307 trees_ = get_param(params,"trees",4);
308 root = new NodePtr[trees_];
309 indices = new int*[trees_];
310
311 for (int i=0; i<trees_; ++i) {
312 root[i] = NULL;
313 indices[i] = NULL;
314 }
315 }

389 /**
390 * Builds the index
391 */
392 void buildIndex()
393 {
394 if (branching_<2) {
395 throw FLANNException("Branching factor must be at least 2");
396 }
397
398 free_elements();
399
400 for (int i=0; i<trees_; ++i) {
401 indices[i] = new int[size_];
402 for (size_t j=0; j<size_; ++j) {
403 indices[i][j] = (int)j;
404 }
405 root[i] = pool.allocate<Node>();
406 computeClustering(root[i], indices[i], (int)size_, branching_,0);
407 }

444 load_value(stream, branching_);
445 load_value(stream, trees_);
446 load_value(stream, centers_init_);
447 load_value(stream, leaf_size_);
448 load_value(stream, memoryCounter);
449
450 indices = new int*[trees_];
451 root = new NodePtr[trees_];
452 for (int i=0; i<trees_; ++i) {
453 indices[i] = new int[size_];
454 load_value(stream, *indices[i], size_);
455 load_tree(stream, root[i], i);
456 }

458 params["algorithm"] = getType();
459 params["branching"] = branching_;
460 params["trees"] = trees_;
461 params["centers_init"] = centers_init_;
462 params["leaf_size"] = leaf_size_;

466 /**
467 * Find set of nearest neighbors to vec. Their indices are stored inside
468 * the result object.
469 *
470 * Params:
471 * result = the result object in which the indices of the nearest-neighbors are stored
472 * vec = the vector for which to search the nearest neighbors
473 * searchParams = parameters that influence the search algorithm (checks)
474 */
475 void findNeighbors(ResultSet<DistanceType>& result, const ElementType* vec, const SearchParams& searchParams)
476 {
477
478 int maxChecks = get_param(searchParams,"checks",32);
479
480 // Priority queue storing intermediate branches in the best-bin-first search
481 Heap<BranchSt>* heap = new Heap<BranchSt>((int)size_);
482
483 std::vector<bool> checked(size_,false);
484 int checks = 0;
485 for (int i=0; i<trees_; ++i) {
486 findNN(root[i], result, vec, checks, maxChecks, heap, checked);
487 }
488
489 BranchSt branch;
490 while (heap->popMin(branch) && (checks<maxChecks || !result.full())) {
491 NodePtr node = branch.node;
492 findNN(node, result, vec, checks, maxChecks, heap, checked);
493 }
494 assert(result.full());
495
496 delete heap;
497
498 }

508 /**
509 * Struture representing a node in the hierarchical k-means tree.
510 */
511 struct Node
512 {
513 /**
514 * The cluster center index
515 */
516 int pivot;
517 /**
518 * The cluster size (number of points in the cluster)
519 */
520 int size;
521 /**
522 * Child nodes (only for non-terminal nodes)
523 */
524 Node** childs;

526 * Node points (only for terminal nodes)
527 */
528 int* indices;
529 /**
530 * Level

545 void save_tree(FILE* stream, NodePtr node, int num)
546 {
547 save_value(stream, *node);
548 if (node->childs==NULL) {
549 int indices_offset = (int)(node->indices - indices[num]);
550 save_value(stream, indices_offset);

552 else {
553 for(int i=0; i<branching_; ++i) {
554 save_tree(stream, node->childs[i], num);

560 void load_tree(FILE* stream, NodePtr& node, int num)
561 {
562 node = pool.allocate<Node>();
563 load_value(stream, *node);
564 if (node->childs==NULL) {
565 int indices_offset;
566 load_value(stream, indices_offset);
567 node->indices = indices[num] + indices_offset;
568 }
569 else {
570 node->childs = pool.allocate<NodePtr>(branching_);
571 for(int i=0; i<branching_; ++i) {
572 load_tree(stream, node->childs[i], num);
573 }
574 }

580 void computeLabels(int* dsindices, int indices_length, int* centers, int centers_length, int* labels, DistanceType& cost)
581 {
582 cost = 0;
583 for (int i=0; i<indices_length; ++i) {
584 ElementType* point = dataset[dsindices[i]];
585 DistanceType dist = distance(point, dataset[centers[0]], veclen_);
586 labels[i] = 0;
587 for (int j=1; j<centers_length; ++j) {
588 DistanceType new_dist = distance(point, dataset[centers[j]], veclen_);
589 if (dist>new_dist) {
590 labels[i] = j;
591 dist = new_dist;
592 }
593 }
594 cost += dist;
595 }
596 }

598 /**
599 * The method responsible with actually doing the recursive hierarchical
600 * clustering
601 *
602 * Params:
603 * node = the node to cluster
604 * indices = indices of the points belonging to the current node
605 * branching = the branching factor to use in the clustering
606 *
607 * TODO: for 1-sized clusters don't store a cluster center (it's the same as the single cluster point)
608 */
609 void computeClustering(NodePtr node, int* dsindices, int indices_length, int branching, int level)
610 {
611 node->size = indices_length;
612 node->level = level;
613
614 if (indices_length < leaf_size_) { // leaf node
615 node->indices = dsindices;
616 std::sort(node->indices,node->indices+indices_length);
617 node->childs = NULL;
618 return;
619 }
620
621 std::vector<int> centers(branching);
622 std::vector<int> labels(indices_length);
623
624 int centers_length;
625 (this->*chooseCenters)(branching, dsindices, indices_length, &centers[0], centers_length);
626
627 if (centers_length<branching) {
628 node->indices = dsindices;
629 std::sort(node->indices,node->indices+indices_length);
630 node->childs = NULL;
631 return;
632 }
633
634
635 // assign points to clusters
636 DistanceType cost;
637 computeLabels(dsindices, indices_length, &centers[0], centers_length, &labels[0], cost);
638
639 node->childs = pool.allocate<NodePtr>(branching);
640 int start = 0;
641 int end = start;
642 for (int i=0; i<branching; ++i) {
643 for (int j=0; j<indices_length; ++j) {
644 if (labels[j]==i) {
645 std::swap(dsindices[j],dsindices[end]);
646 std::swap(labels[j],labels[end]);
647 end++;
648 }
649 }
650
651 node->childs[i] = pool.allocate<Node>();
652 node->childs[i]->pivot = centers[i];
653 node->childs[i]->indices = NULL;
654 computeClustering(node->childs[i],dsindices+start, end-start, branching, level+1);
655 start=end;
656 }
657 }

661 /**
662 * Performs one descent in the hierarchical k-means tree. The branches not
663 * visited are stored in a priority queue.
664 *
665 * Params:
666 * node = node to explore
667 * result = container for the k-nearest neighbors found
668 * vec = query points
669 * checks = how many points in the dataset have been checked so far
670 * maxChecks = maximum dataset points to checks
671 */
672
673
674 void findNN(NodePtr node, ResultSet<DistanceType>& result, const ElementType* vec, int& checks, int maxChecks,
675 Heap<BranchSt>* heap, std::vector<bool>& checked)
676 {
677 if (node->childs==NULL) {
678 if (checks>=maxChecks) {
679 if (result.full()) return;
680 }
681 for (int i=0; i<node->size; ++i) {
682 int index = node->indices[i];
683 if (!checked[index]) {
684 DistanceType dist = distance(dataset[index], vec, veclen_);
685 result.addPoint(dist, index);
686 checked[index] = true;
687 ++checks;
688 }
689 }
690 }
691 else {
692 DistanceType* domain_distances = new DistanceType[branching_];
693 int best_index = 0;
694 domain_distances[best_index] = distance(vec, dataset[node->childs[best_index]->pivot], veclen_);
695 for (int i=1; i<branching_; ++i) {
696 domain_distances[i] = distance(vec, dataset[node->childs[i]->pivot], veclen_);
697 if (domain_distances[i]<domain_distances[best_index]) {
698 best_index = i;
699 }
700 }
701 for (int i=0; i<branching_; ++i) {
702 if (i!=best_index) {
703 heap->insert(BranchSt(node->childs[i],domain_distances[i]));
704 }
705 }
706 delete[] domain_distances;
707 findNN(node->childs[best_index],result,vec, checks, maxChecks, heap, checked);
708 }
709 }
