40struct base_any_policy
41{
42 virtual void static_delete(void** x) = 0;
43 virtual void copy_from_value(void const* src, void** dest) = 0;
44 virtual void clone(void* const* src, void** dest) = 0;
45 virtual void move(void* const* src, void** dest) = 0;
46 virtual void* get_value(void** src) = 0;
47 virtual ::size_t get_size() = 0;
48 virtual const std::type_info& type() = 0;
49 virtual void print(std::ostream& out, void* const* src) = 0;
50
51#ifdef OPENCV_CAN_BREAK_BINARY_COMPATIBILITY
52 virtual ~base_any_policy() {}
53#endif
54};
55
56template<typename T>
57struct typed_base_any_policy : base_any_policy
58{
59 virtual ::size_t get_size() { return sizeof(T); }
60 virtual const std::type_info& type() { return typeid(T); }
61
62};

64template<typename T>
65struct small_any_policy : typed_base_any_policy<T>
66{
67 virtual void static_delete(void**) { }
68 virtual void copy_from_value(void const* src, void** dest)
69 {
70 new (dest) T(* reinterpret_cast<T const*>(src));
71 }
72 virtual void clone(void* const* src, void** dest) { *dest = *src; }
73 virtual void move(void* const* src, void** dest) { *dest = *src; }
74 virtual void* get_value(void** src) { return reinterpret_cast<void*>(src); }
75 virtual void print(std::ostream& out, void* const* src) { out << *reinterpret_cast<T const*>(src); }
76};
77
78template<typename T>
79struct big_any_policy : typed_base_any_policy<T>
80{
81 virtual void static_delete(void** x)
82 {
83 if (* x) { delete (* reinterpret_cast<T**>(x)); *x = NULL; }
84 }
85 virtual void copy_from_value(void const* src, void** dest)
86 {
87 *dest = new T(*reinterpret_cast<T const*>(src));
88 }
89 virtual void clone(void* const* src, void** dest)
90 {
91 *dest = new T(**reinterpret_cast<T* const*>(src));
92 }
93 virtual void move(void* const* src, void** dest)
94 {
95 (*reinterpret_cast<T**>(dest))->~T();
96 **reinterpret_cast<T**>(dest) = **reinterpret_cast<T* const*>(src);
97 }
98 virtual void* get_value(void** src) { return *src; }
99 virtual void print(std::ostream& out, void* const* src) { out << *reinterpret_cast<T const*>(*src); }
100};

102template<> inline void big_any_policy<flann_centers_init_t>::print(std::ostream& out, void* const* src)
103{
104 out << int(*reinterpret_cast<flann_centers_init_t const*>(*src));
105}

153template <typename T>
154class SinglePolicy
155{
156 SinglePolicy();
157 SinglePolicy(const SinglePolicy& other);
158 SinglePolicy& operator=(const SinglePolicy& other);
159
160public:
161 static base_any_policy* get_policy();
162
163private:
164 static typename choose_policy<T>::type policy;
165};
166
167template <typename T>
168typename choose_policy<T>::type SinglePolicy<T>::policy;
169
170/// This function will return a different policy for each type.
171template <typename T>
172inline base_any_policy* SinglePolicy<T>::get_policy() { return &policy; }
173
174} // namespace anyimpl
175
176struct any
177{
178private:
179 // fields
180 anyimpl::base_any_policy* policy;
181 void* object;
182
183public:
184 /// Initializing constructor.
185 template <typename T>
186 any(const T& x)
187 : policy(anyimpl::SinglePolicy<anyimpl::empty_any>::get_policy()), object(NULL)
188 {
189 assign(x);
190 }
191
192 /// Empty constructor.
193 any()
194 : policy(anyimpl::SinglePolicy<anyimpl::empty_any>::get_policy()), object(NULL)
195 { }
196
197 /// Special initializing constructor for string literals.
198 any(const char* x)
199 : policy(anyimpl::SinglePolicy<anyimpl::empty_any>::get_policy()), object(NULL)
200 {
201 assign(x);
202 }
203
204 /// Copy constructor.
205 any(const any& x)
206 : policy(anyimpl::SinglePolicy<anyimpl::empty_any>::get_policy()), object(NULL)
207 {
208 assign(x);
209 }
210
211 /// Destructor.
212 ~any()
213 {
214 policy->static_delete(&object);
215 }
216
217 /// Assignment function from another any.
218 any& assign(const any& x)
219 {
220 reset();
221 policy = x.policy;
222 policy->clone(&x.object, &object);
223 return *this;
224 }
225
226 /// Assignment function.
227 template <typename T>
228 any& assign(const T& x)
229 {
230 reset();
231 policy = anyimpl::SinglePolicy<T>::get_policy();
232 policy->copy_from_value(&x, &object);
233 return *this;
234 }
235
236 /// Assignment operator.
237 template<typename T>
238 any& operator=(const T& x)
239 {
240 return assign(x);
241 }
242
243 /// Assignment operator, specialed for literal strings.
244 /// They have types like const char [6] which don't work as expected.
245 any& operator=(const char* x)
246 {
247 return assign(x);
248 }
249
250 /// Utility functions
251 any& swap(any& x)
252 {
253 std::swap(policy, x.policy);
254 std::swap(object, x.object);
255 return *this;
256 }
257
258 /// Cast operator. You can only cast to the original type.
259 template<typename T>
260 T& cast()
261 {
262 if (policy->type() != typeid(T)) throw anyimpl::bad_any_cast();
263 T* r = reinterpret_cast<T*>(policy->get_value(&object));
264 return *r;
265 }
266
267 /// Cast operator. You can only cast to the original type.
268 template<typename T>
269 const T& cast() const
270 {
271 if (policy->type() != typeid(T)) throw anyimpl::bad_any_cast();
272 T* r = reinterpret_cast<T*>(policy->get_value(const_cast<void **>(&object)));
273 return *r;
274 }
275
276 /// Returns true if the any contains no value.
277 bool empty() const
278 {
279 return policy->type() == typeid(anyimpl::empty_any);
280 }
281
282 /// Frees any allocated memory, and sets the value to NULL.
283 void reset()
284 {
285 policy->static_delete(&object);
286 policy = anyimpl::SinglePolicy<anyimpl::empty_any>::get_policy();
287 }
288
289 /// Returns true if the two types are the same.
290 bool compatible(const any& x) const
291 {
292 return policy->type() == x.policy->type();
293 }
294
295 /// Returns if the type is compatible with the policy
296 template<typename T>
297 bool has_type()
298 {
299 return policy->type() == typeid(T);
300 }
301
302 const std::type_info& type() const
303 {
304 return policy->type();
305 }
306
307 friend std::ostream& operator <<(std::ostream& out, const any& any_val);
308};
309
310inline std::ostream& operator <<(std::ostream& out, const any& any_val)
311{
312 any_val.policy->print(out,&any_val.object);
313 return out;
314}
