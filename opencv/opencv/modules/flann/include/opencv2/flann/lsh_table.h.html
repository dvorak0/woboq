80/** POD for stats about an LSH table
81 */
82struct LshStats
83{
84 std::vector<unsigned int> bucket_sizes_;
85 size_t n_buckets_;
86 size_t bucket_size_mean_;
87 size_t bucket_size_median_;
88 size_t bucket_size_min_;
89 size_t bucket_size_max_;
90 size_t bucket_size_std_dev;
91 /** Each contained vector contains three value: beginning/end for interval, number of elements in the bin
92 */
93 std::vector<std::vector<unsigned int> > size_histogram_;
94};

96/** Overload the << operator for LshStats
97 * @param out the streams
98 * @param stats the stats to display
99 * @return the streams
100 */
101inline std::ostream& operator <<(std::ostream& out, const LshStats& stats)
102{
103 int w = 20;
104 out << "Lsh Table Stats:\n" << std::setw(w) << std::setiosflags(std::ios::right) << "N buckets : "
105 << stats.n_buckets_ << "\n" << std::setw(w) << std::setiosflags(std::ios::right) << "mean size : "
106 << std::setiosflags(std::ios::left) << stats.bucket_size_mean_ << "\n" << std::setw(w)
107 << std::setiosflags(std::ios::right) << "median size : " << stats.bucket_size_median_ << "\n" << std::setw(w)
108 << std::setiosflags(std::ios::right) << "min size : " << std::setiosflags(std::ios::left)
109 << stats.bucket_size_min_ << "\n" << std::setw(w) << std::setiosflags(std::ios::right) << "max size : "
110 << std::setiosflags(std::ios::left) << stats.bucket_size_max_;
111
112 // Display the histogram
113 out << std::endl << std::setw(w) << std::setiosflags(std::ios::right) << "histogram : "
114 << std::setiosflags(std::ios::left);
115 for (std::vector<std::vector<unsigned int> >::const_iterator iterator = stats.size_histogram_.begin(), end =
116 stats.size_histogram_.end(); iterator != end; ++iterator) out << (*iterator)[0] << "-" << (*iterator)[1] << ": " << (*iterator)[2] << ", ";
117
118 return out;
119}

124/** Lsh hash table. As its key is a sub-feature, and as usually
125 * the size of it is pretty small, we keep it as a continuous memory array.
126 * The value is an index in the corpus of features (we keep it as an unsigned
127 * int for pure memory reasons, it could be a size_t)
128 */
129template<typename ElementType>
130class LshTable
131{
132public:
133 /** A container of all the feature indices. Optimized for space
134 */
135#if USE_UNORDERED_MAP
136 typedef std::unordered_map<BucketKey, Bucket> BucketsSpace;
137#else
138 typedef std::map<BucketKey, Bucket> BucketsSpace;
139#endif
140
141 /** A container of all the feature indices. Optimized for speed
142 */
143 typedef std::vector<Bucket> BucketsSpeed;
144
145 /** Default constructor
146 */
147 LshTable()
148 {
149 }
150
151 /** Default constructor
152 * Create the mask and allocate the memory
153 * @param feature_size is the size of the feature (considered as a ElementType[])
154 * @param key_size is the number of bits that are turned on in the feature
155 * @param indices
156 */
157 LshTable(unsigned int feature_size, unsigned int key_size, std::vector<size_t> & indices)
158 {
159 (void)feature_size;
160 (void)key_size;
161 (void)indices;
162 std::cerr << "LSH is not implemented for that type" << std::endl;
163 assert(0);
164 }
165
166 /** Add a feature to the table
167 * @param value the value to store for that feature
168 * @param feature the feature itself
169 */
170 void add(unsigned int value, const ElementType* feature)
171 {
172 // Add the value to the corresponding bucket
173 BucketKey key = (lsh::BucketKey)getKey(feature);
174
175 switch (speed_level_) {
176 case kArray:
177 // That means we get the buckets from an array
178 buckets_speed_[key].push_back(value);
179 break;
180 case kBitsetHash:
181 // That means we can check the bitset for the presence of a key
182 key_bitset_.set(key);
183 buckets_space_[key].push_back(value);
184 break;
185 case kHash:
186 {
187 // That means we have to check for the hash table for the presence of a key
188 buckets_space_[key].push_back(value);
189 break;
190 }
191 }
192 }
193
194 /** Add a set of features to the table
195 * @param indexed_ofst previous indexed offset
196 * @param dataset the values to store
197 */
198 void add(int indexed_ofst, Matrix<ElementType> dataset)
199 {
200#if USE_UNORDERED_MAP
201 buckets_space_.rehash((buckets_space_.size() + dataset.rows) * 1.2);
202#endif
203 // Add the features to the table
204 for (unsigned int i = 0; i < dataset.rows; ++i) add(i + indexed_ofst, dataset[i]);
205 // Now that the table is full, optimize it for speed/space
206 optimize();
207 }
208
209 /** Get a bucket given the key
210 * @param key
211 * @return
212 */
213 inline const Bucket* getBucketFromKey(BucketKey key) const
214 {
215 // Generate other buckets
216 switch (speed_level_) {
217 case kArray:
218 // That means we get the buckets from an array
219 return &buckets_speed_[key];
220 break;
221 case kBitsetHash:
222 // That means we can check the bitset for the presence of a key
223 if (key_bitset_.test(key)) return &buckets_space_.find(key)->second;
224 else return 0;
225 break;
226 case kHash:
227 {
228 // That means we have to check for the hash table for the presence of a key
229 BucketsSpace::const_iterator bucket_it, bucket_end = buckets_space_.end();
230 bucket_it = buckets_space_.find(key);
231 // Stop here if that bucket does not exist
232 if (bucket_it == bucket_end) return 0;
233 else return &bucket_it->second;
234 break;
235 }
236 }
237 return 0;
238 }

240 /** Compute the sub-signature of a feature
241 */
242 size_t getKey(const ElementType* /*feature*/) const
243 {
244 std::cerr << "LSH is not implemented for that type" << std::endl;
245 assert(0);
246 return 1;
247 }

249 /** Get statistics about the table
250 * @return
251 */
252 LshStats getStats() const;
253
254private:
255 /** defines the speed fo the implementation
256 * kArray uses a vector for storing data
257 * kBitsetHash uses a hash map but checks for the validity of a key with a bitset
258 * kHash uses a hash map only
259 */
260 enum SpeedLevel
261 {
262 kArray, kBitsetHash, kHash
263 };
264
265 /** Initialize some variables
266 */
267 void initialize(size_t key_size)
268 {
269 const size_t key_size_lower_bound = 1;
270 //a value (size_t(1) << key_size) must fit the size_t type so key_size has to be strictly less than size of size_t
271 const size_t key_size_upper_bound = std::min(sizeof(BucketKey) * CHAR_BIT + 1, sizeof(size_t) * CHAR_BIT);
272 if (key_size < key_size_lower_bound || key_size >= key_size_upper_bound)
273 {
274 std::stringstream errorMessage;
275 errorMessage << "Invalid key_size (=" << key_size << "). Valid values for your system are " << key_size_lower_bound << " <= key_size < " << key_size_upper_bound << ".";
276 CV_Error(CV_StsBadArg, errorMessage.str());
277 }
278
279 speed_level_ = kHash;
280 key_size_ = (unsigned)key_size;
281 }
282
283 /** Optimize the table for speed/space
284 */
285 void optimize()
286 {
287 // If we are already using the fast storage, no need to do anything
288 if (speed_level_ == kArray) return;
289
290 // Use an array if it will be more than half full
291 if (buckets_space_.size() > ((size_t(1) << key_size_) / 2)) {
292 speed_level_ = kArray;
293 // Fill the array version of it
294 buckets_speed_.resize(size_t(1) << key_size_);
295 for (BucketsSpace::const_iterator key_bucket = buckets_space_.begin(); key_bucket != buckets_space_.end(); ++key_bucket) buckets_speed_[key_bucket->first] = key_bucket->second;
296
297 // Empty the hash table
298 buckets_space_.clear();
299 return;
300 }
301
302 // If the bitset is going to use less than 10% of the RAM of the hash map (at least 1 size_t for the key and two
303 // for the vector) or less than 512MB (key_size_ <= 30)
304 if (((std::max(buckets_space_.size(), buckets_speed_.size()) * CHAR_BIT * 3 * sizeof(BucketKey)) / 10
305 >= (size_t(1) << key_size_)) || (key_size_ <= 32)) {
306 speed_level_ = kBitsetHash;
307 key_bitset_.resize(size_t(1) << key_size_);
308 key_bitset_.reset();
309 // Try with the BucketsSpace
310 for (BucketsSpace::const_iterator key_bucket = buckets_space_.begin(); key_bucket != buckets_space_.end(); ++key_bucket) key_bitset_.set(key_bucket->first);
311 }
312 else {
313 speed_level_ = kHash;
314 key_bitset_.clear();
315 }
316 }
317
318 /** The vector of all the buckets if they are held for speed
319 */
320 BucketsSpeed buckets_speed_;
321
322 /** The hash table of all the buckets in case we cannot use the speed version
323 */
324 BucketsSpace buckets_space_;
325
326 /** What is used to store the data */
327 SpeedLevel speed_level_;
328
329 /** If the subkey is small enough, it will keep track of which subkeys are set through that bitset
330 * That is just a speedup so that we don't look in the hash table (which can be mush slower that checking a bitset)
331 */
332 DynamicBitset key_bitset_;
333
334 /** The size of the sub-signature in bits
335 */
336 unsigned int key_size_;
337
338 // Members only used for the unsigned char specialization
339 /** The mask to apply to a feature to get the hash key
340 * Only used in the unsigned char case
341 */
342 std::vector<size_t> mask_;
343};

346// Specialization for unsigned char
347
348template<>
349inline LshTable<unsigned char>::LshTable( unsigned int feature_size,
350 unsigned int subsignature_size,
351 std::vector<size_t> & indices )
352{
353 initialize(subsignature_size);
354 // Allocate the mask
355 mask_ = std::vector<size_t>((size_t)ceil((float)(feature_size * sizeof(char)) / (float)sizeof(size_t)), 0);
356
357 // Generate a random set of order of subsignature_size_ bits
358 for (unsigned int i = 0; i < key_size_; ++i) {
359 //Ensure the Nth bit will be selected only once among the different LshTables
360 //to avoid having two different tables with signatures sharing many dimensions/many bits
361 size_t index = indices[0];
362 indices.erase( indices.begin() );
363
364 // Set that bit in the mask
365 size_t divisor = CHAR_BIT * sizeof(size_t);
366 size_t idx = index / divisor; //pick the right size_t index
367 mask_[idx] |= size_t(1) << (index % divisor); //use modulo to find the bit offset
368 }
369
370 // Set to 1 if you want to display the mask for debug
371#if 0
372 {
373 size_t bcount = 0;
374 BOOST_FOREACH(size_t mask_block, mask_){
375 out << std::setw(sizeof(size_t) * CHAR_BIT / 4) << std::setfill('0') << std::hex << mask_block
376 << std::endl;
377 bcount += __builtin_popcountll(mask_block);
378 }
379 out << "bit count : " << std::dec << bcount << std::endl;
380 out << "mask size : " << mask_.size() << std::endl;
381 return out;
382 }
383#endif
384}

386/** Return the Subsignature of a feature
387 * @param feature the feature to analyze
388 */
389template<>
390inline size_t LshTable<unsigned char>::getKey(const unsigned char* feature) const
391{
392 // no need to check if T is dividable by sizeof(size_t) like in the Hamming
393 // distance computation as we have a mask
394 const size_t* feature_block_ptr = reinterpret_cast<const size_t*> ((const void*)feature);
395
396 // Figure out the subsignature of the feature
397 // Given the feature ABCDEF, and the mask 001011, the output will be
398 // 000CEF
399 size_t subsignature = 0;
400 size_t bit_index = 1;
401
402 for (std::vector<size_t>::const_iterator pmask_block = mask_.begin(); pmask_block != mask_.end(); ++pmask_block) {
403 // get the mask and signature blocks
404 size_t feature_block = *feature_block_ptr;
405 size_t mask_block = *pmask_block;
406 while (mask_block) {
407 // Get the lowest set bit in the mask block
408 size_t lowest_bit = mask_block & (-(ptrdiff_t)mask_block);
409 // Add it to the current subsignature if necessary
410 subsignature += (feature_block & lowest_bit) ? bit_index : 0;
411 // Reset the bit in the mask block
412 mask_block ^= lowest_bit;
413 // increment the bit index for the subsignature
414 bit_index <<= 1;
415 }
416 // Check the next feature block
417 ++feature_block_ptr;
418 }
419 return subsignature;
420}

422template<>
423inline LshStats LshTable<unsigned char>::getStats() const
424{
425 LshStats stats;
426 stats.bucket_size_mean_ = 0;
427 if ((buckets_speed_.empty()) && (buckets_space_.empty())) {
428 stats.n_buckets_ = 0;
429 stats.bucket_size_median_ = 0;
430 stats.bucket_size_min_ = 0;
431 stats.bucket_size_max_ = 0;
432 return stats;
433 }
434
435 if (!buckets_speed_.empty()) {
436 for (BucketsSpeed::const_iterator pbucket = buckets_speed_.begin(); pbucket != buckets_speed_.end(); ++pbucket) {
437 stats.bucket_sizes_.push_back((lsh::FeatureIndex)pbucket->size());
438 stats.bucket_size_mean_ += pbucket->size();
439 }
440 stats.bucket_size_mean_ /= buckets_speed_.size();
441 stats.n_buckets_ = buckets_speed_.size();
442 }
443 else {
444 for (BucketsSpace::const_iterator x = buckets_space_.begin(); x != buckets_space_.end(); ++x) {
445 stats.bucket_sizes_.push_back((lsh::FeatureIndex)x->second.size());
446 stats.bucket_size_mean_ += x->second.size();
447 }
448 stats.bucket_size_mean_ /= buckets_space_.size();
449 stats.n_buckets_ = buckets_space_.size();
450 }
451
452 std::sort(stats.bucket_sizes_.begin(), stats.bucket_sizes_.end());
453
454 // BOOST_FOREACH(int size, stats.bucket_sizes_)
455 // std::cout << size << " ";
456 // std::cout << std::endl;
457 stats.bucket_size_median_ = stats.bucket_sizes_[stats.bucket_sizes_.size() / 2];
458 stats.bucket_size_min_ = stats.bucket_sizes_.front();
459 stats.bucket_size_max_ = stats.bucket_sizes_.back();
460
461 // TODO compute mean and std
462 /*float mean, stddev;
463 stats.bucket_size_mean_ = mean;
464 stats.bucket_size_std_dev = stddev;*/
465
466 // Include a histogram of the buckets
467 unsigned int bin_start = 0;
468 unsigned int bin_end = 20;
469 bool is_new_bin = true;
470 for (std::vector<unsigned int>::iterator iterator = stats.bucket_sizes_.begin(), end = stats.bucket_sizes_.end(); iterator
471 != end; )
472 if (*iterator < bin_end) {
473 if (is_new_bin) {
474 stats.size_histogram_.push_back(std::vector<unsigned int>(3, 0));
475 stats.size_histogram_.back()[0] = bin_start;
476 stats.size_histogram_.back()[1] = bin_end - 1;
477 is_new_bin = false;
478 }
479 ++stats.size_histogram_.back()[2];
480 ++iterator;
481 }
482 else {
483 bin_start += 20;
484 bin_end += 20;
485 is_new_bin = true;
486 }
487
488 return stats;
489}
