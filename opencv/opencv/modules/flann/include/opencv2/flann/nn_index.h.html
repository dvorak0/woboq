44/**
45 * Nearest-neighbour index base class
46 */
47template <typename Distance>
48class NNIndex
49{
50 typedef typename Distance::ElementType ElementType;
51 typedef typename Distance::ResultType DistanceType;
52
53public:
54
55 virtual ~NNIndex() {}
56
57 /**
58 * \brief Builds the index
59 */
60 virtual void buildIndex() = 0;
61
62 /**
63 * \brief implementation for algorithms of addable indexes after that.
64 */
65 virtual void addIndex(const Matrix<ElementType>& wholeData, const Matrix<ElementType>& additionalData) = 0;
66
67 /**
68 * \brief Perform k-nearest neighbor search
69 * \param[in] queries The query points for which to find the nearest neighbors
70 * \param[out] indices The indices of the nearest neighbors found
71 * \param[out] dists Distances to the nearest neighbors found
72 * \param[in] knn Number of nearest neighbors to return
73 * \param[in] params Search parameters
74 */
75 virtual void knnSearch(const Matrix<ElementType>& queries, Matrix<int>& indices, Matrix<DistanceType>& dists, int knn, const SearchParams& params)
76 {
77 assert(queries.cols == veclen());
78 assert(indices.rows >= queries.rows);
79 assert(dists.rows >= queries.rows);
80 assert(int(indices.cols) >= knn);
81 assert(int(dists.cols) >= knn);
82
83#if 0
84 KNNResultSet<DistanceType> resultSet(knn);
85 for (size_t i = 0; i < queries.rows; i++) {
86 resultSet.init(indices[i], dists[i]);
87 findNeighbors(resultSet, queries[i], params);
88 }
89#else
90 KNNUniqueResultSet<DistanceType> resultSet(knn);
91 for (size_t i = 0; i < queries.rows; i++) {
92 resultSet.clear();
93 findNeighbors(resultSet, queries[i], params);
94 if (get_param(params,"sorted",true)) resultSet.sortAndCopy(indices[i], dists[i], knn);
95 else resultSet.copy(indices[i], dists[i], knn);
96 }
97#endif
98 }
99
100 /**
101 * \brief Perform radius search
102 * \param[in] query The query point
103 * \param[out] indices The indinces of the neighbors found within the given radius
104 * \param[out] dists The distances to the nearest neighbors found
105 * \param[in] radius The radius used for search
106 * \param[in] params Search parameters
107 * \returns Number of neighbors found
108 */
109 virtual int radiusSearch(const Matrix<ElementType>& query, Matrix<int>& indices, Matrix<DistanceType>& dists, float radius, const SearchParams& params)
110 {
111 if (query.rows != 1) {
112 fprintf(stderr, "I can only search one feature at a time for range search\n");
113 return -1;
114 }
115 assert(query.cols == veclen());
116 assert(indices.cols == dists.cols);
117
118 int n = 0;
119 int* indices_ptr = NULL;
120 DistanceType* dists_ptr = NULL;
121 if (indices.cols > 0) {
122 n = (int)indices.cols;
123 indices_ptr = indices[0];
124 dists_ptr = dists[0];
125 }
126
127 RadiusUniqueResultSet<DistanceType> resultSet((DistanceType)radius);
128 resultSet.clear();
129 findNeighbors(resultSet, query[0], params);
130 if (n>0) {
131 if (get_param(params,"sorted",true)) resultSet.sortAndCopy(indices_ptr, dists_ptr, n);
132 else resultSet.copy(indices_ptr, dists_ptr, n);
133 }
134
135 return (int)resultSet.size();
136 }
137
138 /**
139 * \brief Saves the index to a stream
140 * \param stream The stream to save the index to
141 */
142 virtual void saveIndex(FILE* stream) = 0;
143
144 /**
145 * \brief Loads the index from a stream
146 * \param stream The stream from which the index is loaded
147 */
148 virtual void loadIndex(FILE* stream) = 0;
149
150 /**
151 * \returns number of features in this index.
152 */
153 virtual size_t size() const = 0;
154
155 /**
156 * \returns The dimensionality of the features in this index.
157 */
158 virtual size_t veclen() const = 0;
159
160 /**
161 * \returns The amount of memory (in bytes) used by the index.
162 */
163 virtual int usedMemory() const = 0;
164
165 /**
166 * \returns The index type (kdtree, kmeans,...)
167 */
168 virtual flann_algorithm_t getType() const = 0;
169
170 /**
171 * \returns The index parameters
172 */
173 virtual IndexParams getParameters() const = 0;
174
175
176 /**
177 * \brief Method that searches for nearest-neighbours
178 */
179 virtual void findNeighbors(ResultSet<DistanceType>& result, const ElementType* vec, const SearchParams& searchParams) = 0;
180};

182}
