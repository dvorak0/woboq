73template<typename Distance>
74NNIndex<Distance>* load_saved_index(const Matrix<typename Distance::ElementType>& dataset, const std::string& filename, Distance distance)
75{
76 typedef typename Distance::ElementType ElementType;
77
78 FILE* fin = fopen(filename.c_str(), "rb");
79 if (fin == NULL) {
80 return NULL;
81 }
82 IndexHeader header = load_header(fin);
83 if (header.data_type != Datatype<ElementType>::type()) {
84 throw FLANNException("Datatype of saved index is different than of the one to be created.");
85 }
86 if ((size_t(header.rows) != dataset.rows)||(size_t(header.cols) != dataset.cols)) {
87 throw FLANNException("The index saved belongs to a different dataset");
88 }
89
90 IndexParams params;
91 params["algorithm"] = header.index_type;
92 NNIndex<Distance>* nnIndex = create_index_by_type<Distance>(dataset, params, distance);
93 nnIndex->loadIndex(fin);
94 fclose(fin);
95
96 return nnIndex;
97}

100template<typename Distance>
101class Index : public NNIndex<Distance>
102{
103public:
104 typedef typename Distance::ElementType ElementType;
105 typedef typename Distance::ResultType DistanceType;
106
107 Index(const Matrix<ElementType>& features, const IndexParams& params, Distance distance = Distance() )
108 : index_params_(params)
109 {
110 flann_algorithm_t index_type = get_param<flann_algorithm_t>(params,"algorithm");
111 loaded_ = false;
112
113 if (index_type == FLANN_INDEX_SAVED) {
114 nnIndex_ = load_saved_index<Distance>(features, get_param<std::string>(params,"filename"), distance);
115 loaded_ = true;
116 }
117 else {
118 nnIndex_ = create_index_by_type<Distance>(features, params, distance);
119 }
120 }
121
122 ~Index()
123 {
124 delete nnIndex_;
125 }
126
127 /**
128 * implementation for algorithms of addable indexes after that.
129 */
130 void addIndex(const Matrix<ElementType>& wholeData, const Matrix<ElementType>& additionalData)
131 {
132 if (!loaded_) {
133 nnIndex_->addIndex(wholeData, additionalData);
134 }
135 }
136
137 /**
138 * Builds the index.
139 */
140 void buildIndex()
141 {
142 if (!loaded_) {
143 nnIndex_->buildIndex();
144 }
145 }
146
147 void save(std::string filename)
148 {
149 FILE* fout = fopen(filename.c_str(), "wb");
150 if (fout == NULL) {
151 throw FLANNException("Cannot open file");
152 }
153 save_header(fout, *nnIndex_);
154 saveIndex(fout);
155 fclose(fout);
156 }
157
158 /**
159 * \brief Saves the index to a stream
160 * \param stream The stream to save the index to
161 */
162 virtual void saveIndex(FILE* stream)
163 {
164 nnIndex_->saveIndex(stream);
165 }
166
167 /**
168 * \brief Loads the index from a stream
169 * \param stream The stream from which the index is loaded
170 */
171 virtual void loadIndex(FILE* stream)
172 {
173 nnIndex_->loadIndex(stream);
174 }
175
176 /**
177 * \returns number of features in this index.
178 */
179 size_t veclen() const
180 {
181 return nnIndex_->veclen();
182 }
183
184 /**
185 * \returns The dimensionality of the features in this index.
186 */
187 size_t size() const
188 {
189 return nnIndex_->size();
190 }
191
192 /**
193 * \returns The index type (kdtree, kmeans,...)
194 */
195 flann_algorithm_t getType() const
196 {
197 return nnIndex_->getType();
198 }
199
200 /**
201 * \returns The amount of memory (in bytes) used by the index.
202 */
203 virtual int usedMemory() const
204 {
205 return nnIndex_->usedMemory();
206 }
207
208
209 /**
210 * \returns The index parameters
211 */
212 IndexParams getParameters() const
213 {
214 return nnIndex_->getParameters();
215 }
216
217 /**
218 * \brief Perform k-nearest neighbor search
219 * \param[in] queries The query points for which to find the nearest neighbors
220 * \param[out] indices The indices of the nearest neighbors found
221 * \param[out] dists Distances to the nearest neighbors found
222 * \param[in] knn Number of nearest neighbors to return
223 * \param[in] params Search parameters
224 */
225 void knnSearch(const Matrix<ElementType>& queries, Matrix<int>& indices, Matrix<DistanceType>& dists, int knn, const SearchParams& params)
226 {
227 nnIndex_->knnSearch(queries, indices, dists, knn, params);
228 }
229
230 /**
231 * \brief Perform radius search
232 * \param[in] query The query point
233 * \param[out] indices The indinces of the neighbors found within the given radius
234 * \param[out] dists The distances to the nearest neighbors found
235 * \param[in] radius The radius used for search
236 * \param[in] params Search parameters
237 * \returns Number of neighbors found
238 */
239 int radiusSearch(const Matrix<ElementType>& query, Matrix<int>& indices, Matrix<DistanceType>& dists, float radius, const SearchParams& params)
240 {
241 return nnIndex_->radiusSearch(query, indices, dists, radius, params);
242 }
243
244 /**
245 * \brief Method that searches for nearest-neighbours
246 */
247 void findNeighbors(ResultSet<DistanceType>& result, const ElementType* vec, const SearchParams& searchParams)
248 {
249 nnIndex_->findNeighbors(result, vec, searchParams);
250 }
251
252 /**
253 * \brief Returns actual index
254 */
255 FLANN_DEPRECATED NNIndex<Distance>* getIndex()
256 {
257 return nnIndex_;
258 }
259
260 /**
261 * \brief Returns index parameters.
262 * \deprecated use getParameters() instead.
263 */
264 FLANN_DEPRECATED const IndexParams* getIndexParameters()
265 {
266 return &index_params_;
267 }
268
269private:
270 /** Pointer to actual index class */
271 NNIndex<Distance>* nnIndex_;
272 /** Indices if the index was loaded from a file */
273 bool loaded_;
274 /** Parameters passed to the index */
275 IndexParams index_params_;
276};

278/**
279 * Performs a hierarchical clustering of the points passed as argument and then takes a cut in the
280 * the clustering tree to return a flat clustering.
281 * @param[in] points Points to be clustered
282 * @param centers The computed cluster centres. Matrix should be preallocated and centers.rows is the
283 * number of clusters requested.
284 * @param params Clustering parameters (The same as for cvflann::KMeansIndex)
285 * @param d Distance to be used for clustering (eg: cvflann::L2)
286 * @return number of clusters computed (can be different than clusters.rows and is the highest number
287 * of the form (branching-1)*K+1 smaller than clusters.rows).
288 */
289template <typename Distance>
290int hierarchicalClustering(const Matrix<typename Distance::ElementType>& points, Matrix<typename Distance::ResultType>& centers,
291 const KMeansIndexParams& params, Distance d = Distance())
292{
293 KMeansIndex<Distance> kmeans(points, params, d);
294 kmeans.buildIndex();
295
296 int clusterNum = kmeans.getClusterCenters(centers);
297 return clusterNum;
298}
