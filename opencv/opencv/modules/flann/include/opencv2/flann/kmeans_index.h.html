98 /**
99 * Chooses the initial centers in the k-means clustering in a random manner.
100 *
101 * Params:
102 * k = number of centers
103 * vecs = the dataset of points
104 * indices = indices in the dataset
105 * indices_length = length of indices vector
106 *
107 */
108 void chooseCentersRandom(int k, int* indices, int indices_length, int* centers, int& centers_length)
109 {
110 UniqueRandom r(indices_length);
111
112 int index;
113 for (index=0; index<k; ++index) {
114 bool duplicate = true;
115 int rnd;
116 while (duplicate) {
117 duplicate = false;
118 rnd = r.next();
119 if (rnd<0) {
120 centers_length = index;
121 return;
122 }
123
124 centers[index] = indices[rnd];
125
126 for (int j=0; j<index; ++j) {
127 DistanceType sq = distance_(dataset_[centers[index]], dataset_[centers[j]], dataset_.cols);
128 if (sq<1e-16) {
129 duplicate = true;
130 }
131 }
132 }
133 }
134
135 centers_length = index;
136 }

139 /**
140 * Chooses the initial centers in the k-means using Gonzales' algorithm
141 * so that the centers are spaced apart from each other.
142 *
143 * Params:
144 * k = number of centers
145 * vecs = the dataset of points
146 * indices = indices in the dataset
147 * Returns:
148 */
149 void chooseCentersGonzales(int k, int* indices, int indices_length, int* centers, int& centers_length)
150 {
151 int n = indices_length;
152
153 int rnd = rand_int(n);
154 assert(rnd >=0 && rnd < n);
155
156 centers[0] = indices[rnd];
157
158 int index;
159 for (index=1; index<k; ++index) {
160
161 int best_index = -1;
162 DistanceType best_val = 0;
163 for (int j=0; j<n; ++j) {
164 DistanceType dist = distance_(dataset_[centers[0]],dataset_[indices[j]],dataset_.cols);
165 for (int i=1; i<index; ++i) {
166 DistanceType tmp_dist = distance_(dataset_[centers[i]],dataset_[indices[j]],dataset_.cols);
167 if (tmp_dist<dist) {
168 dist = tmp_dist;
169 }
170 }
171 if (dist>best_val) {
172 best_val = dist;
173 best_index = j;
174 }
175 }
176 if (best_index!=-1) {
177 centers[index] = indices[best_index];
178 }
179 else {
180 break;
181 }
182 }
183 centers_length = index;
184 }

187 /**
188 * Chooses the initial centers in the k-means using the algorithm
189 * proposed in the KMeans++ paper:
190 * Arthur, David; Vassilvitskii, Sergei - k-means++: The Advantages of Careful Seeding
191 *
192 * Implementation of this function was converted from the one provided in Arthur's code.
193 *
194 * Params:
195 * k = number of centers
196 * vecs = the dataset of points
197 * indices = indices in the dataset
198 * Returns:
199 */
200 void chooseCentersKMeanspp(int k, int* indices, int indices_length, int* centers, int& centers_length)
201 {
202 int n = indices_length;
203
204 double currentPot = 0;
205 DistanceType* closestDistSq = new DistanceType[n];
206
207 // Choose one random center and set the closestDistSq values
208 int index = rand_int(n);
209 assert(index >=0 && index < n);
210 centers[0] = indices[index];
211
212 for (int i = 0; i < n; i++) {
213 closestDistSq[i] = distance_(dataset_[indices[i]], dataset_[indices[index]], dataset_.cols);
214 closestDistSq[i] = ensureSquareDistance<Distance>( closestDistSq[i] );
215 currentPot += closestDistSq[i];
216 }
217
218
219 const int numLocalTries = 1;
220
221 // Choose each center
222 int centerCount;
223 for (centerCount = 1; centerCount < k; centerCount++) {
224
225 // Repeat several trials
226 double bestNewPot = -1;
227 int bestNewIndex = -1;
228 for (int localTrial = 0; localTrial < numLocalTries; localTrial++) {
229
230 // Choose our center - have to be slightly careful to return a valid answer even accounting
231 // for possible rounding errors
232 double randVal = rand_double(currentPot);
233 for (index = 0; index < n-1; index++) {
234 if (randVal <= closestDistSq[index]) break;
235 else randVal -= closestDistSq[index];
236 }
237
238 // Compute the new potential
239 double newPot = 0;
240 for (int i = 0; i < n; i++) {
241 DistanceType dist = distance_(dataset_[indices[i]], dataset_[indices[index]], dataset_.cols);
242 newPot += std::min( ensureSquareDistance<Distance>(dist), closestDistSq[i] );
243 }
244
245 // Store the best result
246 if ((bestNewPot < 0)||(newPot < bestNewPot)) {
247 bestNewPot = newPot;
248 bestNewIndex = index;
249 }
250 }
251
252 // Add the appropriate center
253 centers[centerCount] = indices[bestNewIndex];
254 currentPot = bestNewPot;
255 for (int i = 0; i < n; i++) {
256 DistanceType dist = distance_(dataset_[indices[i]], dataset_[indices[bestNewIndex]], dataset_.cols);
257 closestDistSq[i] = std::min( ensureSquareDistance<Distance>(dist), closestDistSq[i] );
258 }
259 }
260
261 centers_length = centerCount;
262
263 delete[] closestDistSq;
264 }

275 /**
276 * Index constructor
277 *
278 * Params:
279 * inputData = dataset with the input features
280 * params = parameters passed to the hierarchical k-means algorithm
281 */
282 KMeansIndex(const Matrix<ElementType>& inputData, const IndexParams& params = KMeansIndexParams(),
283 Distance d = Distance())
284 : dataset_(inputData), index_params_(params), root_(NULL), indices_(NULL), distance_(d)
285 {
286 memoryCounter_ = 0;
287
288 size_ = dataset_.rows;
289 veclen_ = dataset_.cols;
290
291 branching_ = get_param(params,"branching",32);
292 iterations_ = get_param(params,"iterations",11);
293 if (iterations_<0) {
294 iterations_ = (std::numeric_limits<int>::max)();
295 }
296 centers_init_ = get_param(params,"centers_init",FLANN_CENTERS_RANDOM);
297
298 if (centers_init_==FLANN_CENTERS_RANDOM) {
299 chooseCenters = &KMeansIndex::chooseCentersRandom;
300 }
301 else if (centers_init_==FLANN_CENTERS_GONZALES) {
302 chooseCenters = &KMeansIndex::chooseCentersGonzales;
303 }
304 else if (centers_init_==FLANN_CENTERS_KMEANSPP) {
305 chooseCenters = &KMeansIndex::chooseCentersKMeanspp;
306 }
307 else {
308 throw FLANNException("Unknown algorithm for choosing initial centers.");
309 }
310 cb_index_ = 0.4f;
311
312 }

372 /**
373 * Builds the index
374 */
375 void buildIndex()
376 {
377 if (branching_<2) {
378 throw FLANNException("Branching factor must be at least 2");
379 }
380
381 indices_ = new int[size_];
382 for (size_t i=0; i<size_; ++i) {
383 indices_[i] = int(i);
384 }
385
386 root_ = pool_.allocate<KMeansNode>();
387 std::memset(root_, 0, sizeof(KMeansNode));
388
389 computeNodeStatistics(root_, indices_, (int)size_);
390 computeClustering(root_, indices_, (int)size_, branching_,0);
391 }

432 /**
433 * Find set of nearest neighbors to vec. Their indices are stored inside
434 * the result object.
435 *
436 * Params:
437 * result = the result object in which the indices of the nearest-neighbors are stored
438 * vec = the vector for which to search the nearest neighbors
439 * searchParams = parameters that influence the search algorithm (checks, cb_index)
440 */
441 void findNeighbors(ResultSet<DistanceType>& result, const ElementType* vec, const SearchParams& searchParams)
442 {
443
444 int maxChecks = get_param(searchParams,"checks",32);
445
446 if (maxChecks==FLANN_CHECKS_UNLIMITED) {
447 findExactNN(root_, result, vec);
448 }
449 else {
450 // Priority queue storing intermediate branches in the best-bin-first search
451 Heap<BranchSt>* heap = new Heap<BranchSt>((int)size_);
452
453 int checks = 0;
454 findNN(root_, result, vec, checks, maxChecks, heap);
455
456 BranchSt branch;
457 while (heap->popMin(branch) && (checks<maxChecks || !result.full())) {
458 KMeansNodePtr node = branch.node;
459 findNN(node, result, vec, checks, maxChecks, heap);
460 }
461 assert(result.full());
462
463 delete heap;
464 }

468 /**
469 * Clustering function that takes a cut in the hierarchical k-means
470 * tree and return the clusters centers of that clustering.
471 * Params:
472 * numClusters = number of clusters to have in the clustering computed
473 * Returns: number of cluster centers
474 */
475 int getClusterCenters(Matrix<DistanceType>& centers)
476 {
477 int numClusters = centers.rows;
478 if (numClusters<1) {
479 throw FLANNException("Number of clusters must be at least 1");
480 }
481
482 DistanceType variance;
483 KMeansNodePtr* clusters = new KMeansNodePtr[numClusters];
484
485 int clusterCount = getMinVarianceClusters(root_, clusters, numClusters, variance);
486
487 Logger::info("Clusters requested: %d, returning %d\n",numClusters, clusterCount);
488
489 for (int i=0; i<clusterCount; ++i) {
490 DistanceType* center = clusters[i]->pivot;
491 for (size_t j=0; j<veclen_; ++j) {
492 centers[i][j] = center[j];
493 }
494 }
495 delete[] clusters;
496
497 return clusterCount;
498 }

604 /**
605 * Computes the statistics of a node (mean, radius, variance).
606 *
607 * Params:
608 * node = the node to use
609 * indices = the indices of the points belonging to the node
610 */
611 void computeNodeStatistics(KMeansNodePtr node, int* indices, int indices_length)
612 {
613
614 DistanceType radius = 0;
615 DistanceType variance = 0;
616 DistanceType* mean = new DistanceType[veclen_];
617 memoryCounter_ += int(veclen_*sizeof(DistanceType));
618
619 memset(mean,0,veclen_*sizeof(DistanceType));
620
621 for (size_t i=0; i<size_; ++i) {
622 ElementType* vec = dataset_[indices[i]];
623 for (size_t j=0; j<veclen_; ++j) {
624 mean[j] += vec[j];
625 }
626 variance += distance_(vec, ZeroIterator<ElementType>(), veclen_);
627 }
628 for (size_t j=0; j<veclen_; ++j) {
629 mean[j] /= size_;
630 }
631 variance /= size_;
632 variance -= distance_(mean, ZeroIterator<ElementType>(), veclen_);
633
634 DistanceType tmp = 0;
635 for (int i=0; i<indices_length; ++i) {
636 tmp = distance_(mean, dataset_[indices[i]], veclen_);
637 if (tmp>radius) {
638 radius = tmp;
639 }
640 }
641
642 node->variance = variance;
643 node->radius = radius;
644 node->pivot = mean;
645 }

648 /**
649 * The method responsible with actually doing the recursive hierarchical
650 * clustering
651 *
652 * Params:
653 * node = the node to cluster
654 * indices = indices of the points belonging to the current node
655 * branching = the branching factor to use in the clustering
656 *
657 * TODO: for 1-sized clusters don't store a cluster center (it's the same as the single cluster point)
658 */
659 void computeClustering(KMeansNodePtr node, int* indices, int indices_length, int branching, int level)
660 {
661 node->size = indices_length;
662 node->level = level;
663
664 if (indices_length < branching) {
665 node->indices = indices;
666 std::sort(node->indices,node->indices+indices_length);
667 node->childs = NULL;
668 return;
669 }
670
671 int* centers_idx = new int[branching];
672 int centers_length;
673 (this->*chooseCenters)(branching, indices, indices_length, centers_idx, centers_length);
674
675 if (centers_length<branching) {
676 node->indices = indices;
677 std::sort(node->indices,node->indices+indices_length);
678 node->childs = NULL;
679 delete [] centers_idx;
680 return;
681 }
682
683
684 Matrix<double> dcenters(new double[branching*veclen_],branching,veclen_);
685 for (int i=0; i<centers_length; ++i) {
686 ElementType* vec = dataset_[centers_idx[i]];
687 for (size_t k=0; k<veclen_; ++k) {
688 dcenters[i][k] = double(vec[k]);
689 }
690 }
691 delete[] centers_idx;
692
693 std::vector<DistanceType> radiuses(branching);
694 int* count = new int[branching];
695 for (int i=0; i<branching; ++i) {
696 radiuses[i] = 0;
697 count[i] = 0;
698 }
699
700 // assign points to clusters
701 int* belongs_to = new int[indices_length];
702 for (int i=0; i<indices_length; ++i) {
703
704 DistanceType sq_dist = distance_(dataset_[indices[i]], dcenters[0], veclen_);
705 belongs_to[i] = 0;
706 for (int j=1; j<branching; ++j) {
707 DistanceType new_sq_dist = distance_(dataset_[indices[i]], dcenters[j], veclen_);
708 if (sq_dist>new_sq_dist) {
709 belongs_to[i] = j;
710 sq_dist = new_sq_dist;
711 }
712 }
713 if (sq_dist>radiuses[belongs_to[i]]) {
714 radiuses[belongs_to[i]] = sq_dist;
715 }
716 count[belongs_to[i]]++;
717 }
718
719 bool converged = false;
720 int iteration = 0;
721 while (!converged && iteration<iterations_) {
722 converged = true;
723 iteration++;
724
725 // compute the new cluster centers
726 for (int i=0; i<branching; ++i) {
727 memset(dcenters[i],0,sizeof(double)*veclen_);
728 radiuses[i] = 0;
729 }
730 for (int i=0; i<indices_length; ++i) {
731 ElementType* vec = dataset_[indices[i]];
732 double* center = dcenters[belongs_to[i]];
733 for (size_t k=0; k<veclen_; ++k) {
734 center[k] += vec[k];
735 }
736 }
737 for (int i=0; i<branching; ++i) {
738 int cnt = count[i];
739 for (size_t k=0; k<veclen_; ++k) {
740 dcenters[i][k] /= cnt;
741 }
742 }
743
744 // reassign points to clusters
745 for (int i=0; i<indices_length; ++i) {
746 DistanceType sq_dist = distance_(dataset_[indices[i]], dcenters[0], veclen_);
747 int new_centroid = 0;
748 for (int j=1; j<branching; ++j) {
749 DistanceType new_sq_dist = distance_(dataset_[indices[i]], dcenters[j], veclen_);
750 if (sq_dist>new_sq_dist) {
751 new_centroid = j;
752 sq_dist = new_sq_dist;
753 }
754 }
755 if (sq_dist>radiuses[new_centroid]) {
756 radiuses[new_centroid] = sq_dist;
757 }
758 if (new_centroid != belongs_to[i]) {
759 count[belongs_to[i]]--;
760 count[new_centroid]++;
761 belongs_to[i] = new_centroid;
762
763 converged = false;
764 }
765 }
766
767 for (int i=0; i<branching; ++i) {
768 // if one cluster converges to an empty cluster,
769 // move an element into that cluster
770 if (count[i]==0) {
771 int j = (i+1)%branching;
772 while (count[j]<=1) {
773 j = (j+1)%branching;
774 }
775
776 for (int k=0; k<indices_length; ++k) {
777 if (belongs_to[k]==j) {
778 // for cluster j, we move the furthest element from the center to the empty cluster i
779 if ( distance_(dataset_[indices[k]], dcenters[j], veclen_) == radiuses[j] ) {
780 belongs_to[k] = i;
781 count[j]--;
782 count[i]++;
783 break;
784 }
785 }
786 }
787 converged = false;
788 }
789 }
790
791 }
792
793 DistanceType** centers = new DistanceType*[branching];
794
795 for (int i=0; i<branching; ++i) {
796 centers[i] = new DistanceType[veclen_];
797 memoryCounter_ += (int)(veclen_*sizeof(DistanceType));
798 for (size_t k=0; k<veclen_; ++k) {
799 centers[i][k] = (DistanceType)dcenters[i][k];
800 }
801 }
802
803
804 // compute kmeans clustering for each of the resulting clusters
805 node->childs = pool_.allocate<KMeansNodePtr>(branching);
806 int start = 0;
807 int end = start;
808 for (int c=0; c<branching; ++c) {
809 int s = count[c];
810
811 DistanceType variance = 0;
812 DistanceType mean_radius =0;
813 for (int i=0; i<indices_length; ++i) {
814 if (belongs_to[i]==c) {
815 DistanceType d = distance_(dataset_[indices[i]], ZeroIterator<ElementType>(), veclen_);
816 variance += d;
817 mean_radius += sqrt(d);
818 std::swap(indices[i],indices[end]);
819 std::swap(belongs_to[i],belongs_to[end]);
820 end++;
821 }
822 }
823 variance /= s;
824 mean_radius /= s;
825 variance -= distance_(centers[c], ZeroIterator<ElementType>(), veclen_);
826
827 node->childs[c] = pool_.allocate<KMeansNode>();
828 std::memset(node->childs[c], 0, sizeof(KMeansNode));
829 node->childs[c]->radius = radiuses[c];
830 node->childs[c]->pivot = centers[c];
831 node->childs[c]->variance = variance;
832 node->childs[c]->mean_radius = mean_radius;
833 computeClustering(node->childs[c],indices+start, end-start, branching, level+1);
834 start=end;
835 }
836
837 delete[] dcenters.data;
838 delete[] centers;
839 delete[] count;
840 delete[] belongs_to;
841 }

845 /**
846 * Performs one descent in the hierarchical k-means tree. The branches not
847 * visited are stored in a priority queue.
848 *
849 * Params:
850 * node = node to explore
851 * result = container for the k-nearest neighbors found
852 * vec = query points
853 * checks = how many points in the dataset have been checked so far
854 * maxChecks = maximum dataset points to checks
855 */
856
857
858 void findNN(KMeansNodePtr node, ResultSet<DistanceType>& result, const ElementType* vec, int& checks, int maxChecks,
859 Heap<BranchSt>* heap)
860 {
861 // Ignore those clusters that are too far away
862 {
863 DistanceType bsq = distance_(vec, node->pivot, veclen_);
864 DistanceType rsq = node->radius;
865 DistanceType wsq = result.worstDist();
866
867 DistanceType val = bsq-rsq-wsq;
868 DistanceType val2 = val*val-4*rsq*wsq;
869
870 //if (val>0) {
871 if ((val>0)&&(val2>0)) {
872 return;
873 }
874 }
875
876 if (node->childs==NULL) {
877 if (checks>=maxChecks) {
878 if (result.full()) return;
879 }
880 checks += node->size;
881 for (int i=0; i<node->size; ++i) {
882 int index = node->indices[i];
883 DistanceType dist = distance_(dataset_[index], vec, veclen_);
884 result.addPoint(dist, index);
885 }
886 }
887 else {
888 DistanceType* domain_distances = new DistanceType[branching_];
889 int closest_center = exploreNodeBranches(node, vec, domain_distances, heap);
890 delete[] domain_distances;
891 findNN(node->childs[closest_center],result,vec, checks, maxChecks, heap);
892 }
893 }

895 /**
896 * Helper function that computes the nearest childs of a node to a given query point.
897 * Params:
898 * node = the node
899 * q = the query point
900 * distances = array with the distances to each child node.
901 * Returns:
902 */
903 int exploreNodeBranches(KMeansNodePtr node, const ElementType* q, DistanceType* domain_distances, Heap<BranchSt>* heap)
904 {
905
906 int best_index = 0;
907 domain_distances[best_index] = distance_(q, node->childs[best_index]->pivot, veclen_);
908 for (int i=1; i<branching_; ++i) {
909 domain_distances[i] = distance_(q, node->childs[i]->pivot, veclen_);
910 if (domain_distances[i]<domain_distances[best_index]) {
911 best_index = i;
912 }
913 }
914
915 // float* best_center = node->childs[best_index]->pivot;
916 for (int i=0; i<branching_; ++i) {
917 if (i != best_index) {
918 domain_distances[i] -= cb_index_*node->childs[i]->variance;
919
920 // float dist_to_border = getDistanceToBorder(node.childs[i].pivot,best_center,q);
921 // if (domain_distances[i]<dist_to_border) {
922 // domain_distances[i] = dist_to_border;
923 // }
924 heap->insert(BranchSt(node->childs[i],domain_distances[i]));
925 }
926 }
927
928 return best_index;
929 }

933 * Function the performs exact nearest neighbor search by traversing the entire tree.
934 */
935 void findExactNN(KMeansNodePtr node, ResultSet<DistanceType>& result, const ElementType* vec)
936 {

938 {
939 DistanceType bsq = distance_(vec, node->pivot, veclen_);
940 DistanceType rsq = node->radius;
941 DistanceType wsq = result.worstDist();
942
943 DistanceType val = bsq-rsq-wsq;
944 DistanceType val2 = val*val-4*rsq*wsq;
945
946 // if (val>0) {
947 if ((val>0)&&(val2>0)) {

950 }

952

974 /**
975 * Helper function.
976 *
977 * I computes the order in which to traverse the child nodes of a particular node.
978 */
979 void getCenterOrdering(KMeansNodePtr node, const ElementType* q, int* sort_indices)
980 {
981 DistanceType* domain_distances = new DistanceType[branching_];
982 for (int i=0; i<branching_; ++i) {
983 DistanceType dist = distance_(q, node->childs[i]->pivot, veclen_);
984
985 int j=0;
986 while (domain_distances[j]<dist && j<i) j++;
987 for (int k=i; k>j; --k) {
988 domain_distances[k] = domain_distances[k-1];
989 sort_indices[k] = sort_indices[k-1];
990 }
991 domain_distances[j] = dist;
992 sort_indices[j] = i;
993 }
994 delete[] domain_distances;
995 }

997 /**
998 * Method that computes the squared distance from the query point q
999 * from inside region with center c to the border between this
1000 * region and the region with center p
1001 */
1002 DistanceType getDistanceToBorder(DistanceType* p, DistanceType* c, DistanceType* q)
1003 {
1004 DistanceType sum = 0;
1005 DistanceType sum2 = 0;
1006
1007 for (int i=0; i<veclen_; ++i) {
1008 DistanceType t = c[i]-p[i];
1009 sum += t*(q[i]-(c[i]+p[i])/2);
1010 sum2 += t*t;
1011 }
1012
1013 return sum*sum/sum2;
1014 }

1017 /**
1018 * Helper function the descends in the hierarchical k-means tree by spliting those clusters that minimize
1019 * the overall variance of the clustering.
1020 * Params:
1021 * root = root node
1022 * clusters = array with clusters centers (return value)
1023 * varianceValue = variance of the clustering (return value)
1024 * Returns:
1025 */
1026 int getMinVarianceClusters(KMeansNodePtr root, KMeansNodePtr* clusters, int clusters_length, DistanceType& varianceValue)
1027 {
1028 int clusterCount = 1;
1029 clusters[0] = root;
1030
1031 DistanceType meanVariance = root->variance*root->size;
1032
1033 while (clusterCount<clusters_length) {
1034 DistanceType minVariance = (std::numeric_limits<DistanceType>::max)();
1035 int splitIndex = -1;
1036
1037 for (int i=0; i<clusterCount; ++i) {
1038 if (clusters[i]->childs != NULL) {
1039
1040 DistanceType variance = meanVariance - clusters[i]->variance*clusters[i]->size;
1041
1042 for (int j=0; j<branching_; ++j) {
1043 variance += clusters[i]->childs[j]->variance*clusters[i]->childs[j]->size;
1044 }
1045 if (variance<minVariance) {
1046 minVariance = variance;
1047 splitIndex = i;
1048 }
1049 }
1050 }
1051
1052 if (splitIndex==-1) break;
1053 if ( (branching_+clusterCount-1) > clusters_length) break;
1054
1055 meanVariance = minVariance;
1056
1057 // split node
1058 KMeansNodePtr toSplit = clusters[splitIndex];
1059 clusters[splitIndex] = toSplit->childs[0];
1060 for (int i=1; i<branching_; ++i) {
1061 clusters[clusterCount++] = toSplit->childs[i];
1062 }
1063 }
1064
1065 varianceValue = meanVariance/root->size;
1066 return clusterCount;
1067 }
