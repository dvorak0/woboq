42#include "test_precomp.hpp"
43
44using namespace cv;
45
46class CV_LshTableBadArgTest : public cvtest::BadArgTest
47{
48protected:
49 void run(int);
50 void run_func(void) {};
51
52 struct Caller
53 {
54 int table_number, key_size, multi_probe_level;
55 Mat features;
56
57 void operator()() const
58 {
59 flann::LshIndexParams indexParams(table_number, key_size, multi_probe_level);
60 flann::Index lsh(features, indexParams);
61 }
62 };
63};
64
65void CV_LshTableBadArgTest::run( int /* start_from */ )
66{
67 RNG &rng = ts->get_rng();
68
69 Caller caller;
70 Size featuresSize = cvtest::randomSize(rng, 10.0);
71 caller.features = cvtest::randomMat(rng, featuresSize, CV_8UC1, 0, 255, false);
72 caller.table_number = 12;
73 caller.multi_probe_level = 2;
74
75 int errors = 0;
76 caller.key_size = 0;
77 errors += run_test_case(CV_StsBadArg, "key_size is zero", caller);
78
79 caller.key_size = static_cast<int>(sizeof(size_t) * CHAR_BIT);
80 errors += run_test_case(CV_StsBadArg, "key_size is too big", caller);
81
82 caller.key_size += cvtest::randInt(rng) % 100;
83 errors += run_test_case(CV_StsBadArg, "key_size is too big", caller);
84
85 if (errors != 0)
86 ts->set_failed_test_info(cvtest::TS::FAIL_MISMATCH);
87 else
88 ts->set_failed_test_info(cvtest::TS::OK);
89}

91TEST(Flann_LshTable, badarg) { CV_LshTableBadArgTest test; test.safe_run(); }
