91void IndexParams::getAll(std::vector<std::string>& names,
92 std::vector<int>& types,
93 std::vector<std::string>& strValues,
94 std::vector<double>& numValues) const
95{
96 names.clear();
97 types.clear();
98 strValues.clear();
99 numValues.clear();
100
101 ::cvflann::IndexParams& p = get_params(*this);
102 ::cvflann::IndexParams::const_iterator it = p.begin(), it_end = p.end();
103
104 for( ; it != it_end; ++it )
105 {
106 names.push_back(it->first);
107 try
108 {
109 std::string val = it->second.cast<std::string>();
110 types.push_back(CV_USRTYPE1);
111 strValues.push_back(val);
112 numValues.push_back(-1);
113 continue;
114 }
115 catch (...) {}
116
117 strValues.push_back(it->second.type().name());
118
119 try
120 {
121 double val = it->second.cast<double>();
122 types.push_back( CV_64F );
123 numValues.push_back(val);
124 continue;
125 }
126 catch (...) {}
127 try
128 {
129 float val = it->second.cast<float>();
130 types.push_back( CV_32F );
131 numValues.push_back(val);
132 continue;
133 }
134 catch (...) {}
135 try
136 {
137 int val = it->second.cast<int>();
138 types.push_back( CV_32S );
139 numValues.push_back(val);
140 continue;
141 }
142 catch (...) {}
143 try
144 {
145 short val = it->second.cast<short>();
146 types.push_back( CV_16S );
147 numValues.push_back(val);
148 continue;
149 }
150 catch (...) {}
151 try
152 {
153 ushort val = it->second.cast<ushort>();
154 types.push_back( CV_16U );
155 numValues.push_back(val);
156 continue;
157 }
158 catch (...) {}
159 try
160 {
161 char val = it->second.cast<char>();
162 types.push_back( CV_8S );
163 numValues.push_back(val);
164 continue;
165 }
166 catch (...) {}
167 try
168 {
169 uchar val = it->second.cast<uchar>();
170 types.push_back( CV_8U );
171 numValues.push_back(val);
172 continue;
173 }
174 catch (...) {}
175 try
176 {
177 bool val = it->second.cast<bool>();
178 types.push_back( CV_MAKETYPE(CV_USRTYPE1,2) );
179 numValues.push_back(val);
180 continue;
181 }
182 catch (...) {}
183 try
184 {
185 cvflann::flann_algorithm_t val = it->second.cast<cvflann::flann_algorithm_t>();
186 types.push_back( CV_MAKETYPE(CV_USRTYPE1,3) );
187 numValues.push_back(val);
188 continue;
189 }
190 catch (...) {}
191
192
193 types.push_back(-1); // unknown type
194 numValues.push_back(-1);
195 }
196}

310template<typename Distance, typename IndexType> void
311buildIndex_(void*& index, const Mat& wholedata, const Mat& data, const IndexParams& params, const Distance& dist = Distance())
312{
313 typedef typename Distance::ElementType ElementType;
314 if(DataType<ElementType>::type != data.type())
315 CV_Error_(CV_StsUnsupportedFormat, ("type=%d\n", data.type()));
316 if(!data.isContinuous())
317 CV_Error(CV_StsBadArg, "Only continuous arrays are supported");
318
319 ::cvflann::Matrix<ElementType> dataset((ElementType*)data.data, data.rows, data.cols);
320
321 // currently, additional index support is the lsh algorithm only.
322 if( !index || getParam<flann_algorithm_t>(params, "algorithm", FLANN_INDEX_LINEAR) != FLANN_INDEX_LSH)
323 {
324 Ptr<IndexType> _index = makePtr<IndexType>(dataset, get_params(params), dist);
325 _index->buildIndex();
326 index = _index;
327 // HACK to prevent object destruction
328 _index.obj = NULL;
329 }
330 else // build additional lsh index
331 {
332 ::cvflann::Matrix<ElementType> wholedataset((ElementType*)wholedata.data, wholedata.rows, wholedata.cols);
333 ((IndexType*)index)->addIndex(wholedataset, dataset);
334 }
335}

366void Index::build(InputArray _wholedata, InputArray _data, const IndexParams& params, flann_distance_t _distType)
367{
368 algo = getParam<flann_algorithm_t>(params, "algorithm", FLANN_INDEX_LINEAR);
369
370 if (algo != FLANN_INDEX_LSH) // do not release if algo == FLANN_INDEX_LSH
371 {
372 release();
373 }
374 if (algo == FLANN_INDEX_SAVED)
375 {
376 load(_data, getParam<std::string>(params, "filename", std::string()));
377 return;
378 }
379
380 Mat data = _data.getMat();
381 if (algo != FLANN_INDEX_LSH) // do not clear if algo == FLANN_INDEX_LSH
382 {
383 index = 0;
384 }
385 featureType = data.type();
386 distType = _distType;
387
388 if ( algo == FLANN_INDEX_LSH)
389 {
390 distType = FLANN_DIST_HAMMING;
391 }
392
393 switch( distType )
394 {
395 case FLANN_DIST_HAMMING:
396 buildIndex< HammingDistance >(index, _wholedata.getMat(), data, params);
397 break;
398 case FLANN_DIST_L2:
399 buildIndex< ::cvflann::L2<float> >(index, _wholedata.getMat(), data, params);
400 break;
401 case FLANN_DIST_L1:
402 buildIndex< ::cvflann::L1<float> >(index, _wholedata.getMat(), data, params);
403 break;
404#if MINIFLANN_SUPPORT_EXOTIC_DISTANCE_TYPES
405 case FLANN_DIST_MAX:
406 buildIndex< ::cvflann::MaxDistance<float> >(index, _wholedata.getMat(), data, params);
407 break;
408 case FLANN_DIST_HIST_INTERSECT:
409 buildIndex< ::cvflann::HistIntersectionDistance<float> >(index, _wholedata.getMat(), data, params);
410 break;
411 case FLANN_DIST_HELLINGER:
412 buildIndex< ::cvflann::HellingerDistance<float> >(index, _wholedata.getMat(), data, params);
413 break;
414 case FLANN_DIST_CHI_SQUARE:
415 buildIndex< ::cvflann::ChiSquareDistance<float> >(index, _wholedata.getMat(), data, params);
416 break;
417 case FLANN_DIST_KL:
418 buildIndex< ::cvflann::KL_Divergence<float> >(index, _wholedata.getMat(), data, params);
419 break;
420#endif
421 default:
422 CV_Error(CV_StsBadArg, "Unknown/unsupported distance type");
423 }
424}

480template<typename Distance, typename IndexType>
481void runKnnSearch_(void* index, const Mat& query, Mat& indices, Mat& dists,
482 int knn, const SearchParams& params)
483{
484 typedef typename Distance::ElementType ElementType;
485 typedef typename Distance::ResultType DistanceType;
486 int type = DataType<ElementType>::type;
487 int dtype = DataType<DistanceType>::type;
488 CV_Assert(query.type() == type && indices.type() == CV_32S && dists.type() == dtype);
489 CV_Assert(query.isContinuous() && indices.isContinuous() && dists.isContinuous());
490
491 ::cvflann::Matrix<ElementType> _query((ElementType*)query.data, query.rows, query.cols);
492 ::cvflann::Matrix<int> _indices((int*)indices.data, indices.rows, indices.cols);
493 ::cvflann::Matrix<DistanceType> _dists((DistanceType*)dists.data, dists.rows, dists.cols);
494
495 ((IndexType*)index)->knnSearch(_query, _indices, _dists, knn,
496 (const ::cvflann::SearchParams&)get_params(params));
497}

506template<typename Distance, typename IndexType>
507int runRadiusSearch_(void* index, const Mat& query, Mat& indices, Mat& dists,
508 double radius, const SearchParams& params)
509{
510 typedef typename Distance::ElementType ElementType;
511 typedef typename Distance::ResultType DistanceType;
512 int type = DataType<ElementType>::type;
513 int dtype = DataType<DistanceType>::type;
514 CV_Assert(query.type() == type && indices.type() == CV_32S && dists.type() == dtype);
515 CV_Assert(query.isContinuous() && indices.isContinuous() && dists.isContinuous());
516
517 ::cvflann::Matrix<ElementType> _query((ElementType*)query.data, query.rows, query.cols);
518 ::cvflann::Matrix<int> _indices((int*)indices.data, indices.rows, indices.cols);
519 ::cvflann::Matrix<DistanceType> _dists((DistanceType*)dists.data, dists.rows, dists.cols);

522 saturate_cast<float>(radius),
523 (const ::cvflann::SearchParams&)get_params(params));

534static void createIndicesDists(OutputArray _indices, OutputArray _dists,
535 Mat& indices, Mat& dists, int rows,
536 int minCols, int maxCols, int dtype)
537{
538 if( _indices.needed() )
539 {
540 indices = _indices.getMat();
541 if( !indices.isContinuous() || indices.type() != CV_32S ||
542 indices.rows != rows || indices.cols < minCols || indices.cols > maxCols )
543 {
544 if( !indices.isContinuous() )
545 _indices.release();
546 _indices.create( rows, minCols, CV_32S );
547 indices = _indices.getMat();
548 }
549 }
550 else
551 indices.create( rows, minCols, CV_32S );
552
553 if( _dists.needed() )
554 {
555 dists = _dists.getMat();
556 if( !dists.isContinuous() || dists.type() != dtype ||
557 dists.rows != rows || dists.cols < minCols || dists.cols > maxCols )
558 {
559 if( !indices.isContinuous() )
560 _dists.release();
561 _dists.create( rows, minCols, dtype );
562 dists = _dists.getMat();
563 }
564 }
565 else
566 dists.create( rows, minCols, dtype );
567}

570void Index::knnSearch(InputArray _query, OutputArray _indices,
571 OutputArray _dists, int knn, const SearchParams& params)
572{
573 Mat query = _query.getMat(), indices, dists;
574 int dtype = distType == FLANN_DIST_HAMMING ? CV_32S : CV_32F;
575
576 createIndicesDists( _indices, _dists, indices, dists, query.rows, knn, knn, dtype );
577
578 switch( distType )
579 {
580 case FLANN_DIST_HAMMING:
581 runKnnSearch<HammingDistance>(index, query, indices, dists, knn, params);
582 break;
583 case FLANN_DIST_L2:
584 runKnnSearch< ::cvflann::L2<float> >(index, query, indices, dists, knn, params);
585 break;
586 case FLANN_DIST_L1:
587 runKnnSearch< ::cvflann::L1<float> >(index, query, indices, dists, knn, params);
588 break;
589#if MINIFLANN_SUPPORT_EXOTIC_DISTANCE_TYPES
590 case FLANN_DIST_MAX:
591 runKnnSearch< ::cvflann::MaxDistance<float> >(index, query, indices, dists, knn, params);
592 break;
593 case FLANN_DIST_HIST_INTERSECT:
594 runKnnSearch< ::cvflann::HistIntersectionDistance<float> >(index, query, indices, dists, knn, params);
595 break;
596 case FLANN_DIST_HELLINGER:
597 runKnnSearch< ::cvflann::HellingerDistance<float> >(index, query, indices, dists, knn, params);
598 break;
599 case FLANN_DIST_CHI_SQUARE:
600 runKnnSearch< ::cvflann::ChiSquareDistance<float> >(index, query, indices, dists, knn, params);
601 break;
602 case FLANN_DIST_KL:
603 runKnnSearch< ::cvflann::KL_Divergence<float> >(index, query, indices, dists, knn, params);
604 break;
605#endif
606 default:
607 CV_Error(CV_StsBadArg, "Unknown/unsupported distance type");
608 }

611int Index::radiusSearch(InputArray _query, OutputArray _indices,
612 OutputArray _dists, double radius, int maxResults,
613 const SearchParams& params)
614{
615 Mat query = _query.getMat(), indices, dists;
616 int dtype = distType == FLANN_DIST_HAMMING ? CV_32S : CV_32F;
617 CV_Assert( maxResults > 0 );
618 createIndicesDists( _indices, _dists, indices, dists, query.rows, maxResults, INT_MAX, dtype );
619
620 if( algo == FLANN_INDEX_LSH )
621 CV_Error( CV_StsNotImplemented, "LSH index does not support radiusSearch operation" );
622
623 switch( distType )
624 {
625 case FLANN_DIST_HAMMING:
626 return runRadiusSearch< HammingDistance >(index, query, indices, dists, radius, params);
627
628 case FLANN_DIST_L2:
629 return runRadiusSearch< ::cvflann::L2<float> >(index, query, indices, dists, radius, params);
630 case FLANN_DIST_L1:
631 return runRadiusSearch< ::cvflann::L1<float> >(index, query, indices, dists, radius, params);
632#if MINIFLANN_SUPPORT_EXOTIC_DISTANCE_TYPES
633 case FLANN_DIST_MAX:
634 return runRadiusSearch< ::cvflann::MaxDistance<float> >(index, query, indices, dists, radius, params);
635 case FLANN_DIST_HIST_INTERSECT:
636 return runRadiusSearch< ::cvflann::HistIntersectionDistance<float> >(index, query, indices, dists, radius, params);
637 case FLANN_DIST_HELLINGER:
638 return runRadiusSearch< ::cvflann::HellingerDistance<float> >(index, query, indices, dists, radius, params);
639 case FLANN_DIST_CHI_SQUARE:
640 return runRadiusSearch< ::cvflann::ChiSquareDistance<float> >(index, query, indices, dists, radius, params);
641 case FLANN_DIST_KL:
642 return runRadiusSearch< ::cvflann::KL_Divergence<float> >(index, query, indices, dists, radius, params);
643#endif
644 default:
645 CV_Error(CV_StsBadArg, "Unknown/unsupported distance type");
646 }
647 return -1;
648}

676void Index::save(const std::string& filename) const
677{
678 FILE* fout = fopen(filename.c_str(), "wb");
679 if (fout == NULL)
680 CV_Error_( CV_StsError, ("Can not open file %s for writing FLANN index\n", filename.c_str()) );
681
682 switch( distType )
683 {
684 case FLANN_DIST_HAMMING:
685 saveIndex< HammingDistance >(this, index, fout);
686 break;
687 case FLANN_DIST_L2:
688 saveIndex< ::cvflann::L2<float> >(this, index, fout);
689 break;
690 case FLANN_DIST_L1:
691 saveIndex< ::cvflann::L1<float> >(this, index, fout);
692 break;
693#if MINIFLANN_SUPPORT_EXOTIC_DISTANCE_TYPES
694 case FLANN_DIST_MAX:
695 saveIndex< ::cvflann::MaxDistance<float> >(this, index, fout);
696 break;
697 case FLANN_DIST_HIST_INTERSECT:
698 saveIndex< ::cvflann::HistIntersectionDistance<float> >(this, index, fout);
699 break;
700 case FLANN_DIST_HELLINGER:
701 saveIndex< ::cvflann::HellingerDistance<float> >(this, index, fout);
702 break;
703 case FLANN_DIST_CHI_SQUARE:
704 saveIndex< ::cvflann::ChiSquareDistance<float> >(this, index, fout);
705 break;
706 case FLANN_DIST_KL:
707 saveIndex< ::cvflann::KL_Divergence<float> >(this, index, fout);
708 break;
709#endif
710 default:
711 fclose(fout);
712 fout = 0;
713 CV_Error(CV_StsBadArg, "Unknown/unsupported distance type");
714 }
715 if( fout )
716 fclose(fout);
717}

720template<typename Distance, typename IndexType>
721bool loadIndex_(Index* index0, void*& index, const Mat& data, FILE* fin, const Distance& dist=Distance())
722{
723 typedef typename Distance::ElementType ElementType;
724 CV_Assert(DataType<ElementType>::type == data.type() && data.isContinuous());
725
726 ::cvflann::Matrix<ElementType> dataset((ElementType*)data.data, data.rows, data.cols);
727
728 ::cvflann::IndexParams params;
729 params["algorithm"] = index0->getAlgorithm();
730 IndexType* _index = new IndexType(dataset, params, dist);
731 _index->loadIndex(fin);
732 index = _index;
733 return true;
734}
735
736template<typename Distance>
737bool loadIndex(Index* index0, void*& index, const Mat& data, FILE* fin, const Distance& dist=Distance())
738{
739 return loadIndex_<Distance, ::cvflann::Index<Distance> >(index0, index, data, fin, dist);
740}

742bool Index::load(InputArray _data, const std::string& filename)
743{
744 Mat data = _data.getMat();
745 bool ok = true;
746 release();
747 FILE* fin = fopen(filename.c_str(), "rb");
748 if (fin == NULL)
749 return false;
750
751 ::cvflann::IndexHeader header = ::cvflann::load_header(fin);
752 algo = header.index_type;
753 featureType = header.data_type == FLANN_UINT8 ? CV_8U :
754 header.data_type == FLANN_INT8 ? CV_8S :
755 header.data_type == FLANN_UINT16 ? CV_16U :
756 header.data_type == FLANN_INT16 ? CV_16S :
757 header.data_type == FLANN_INT32 ? CV_32S :
758 header.data_type == FLANN_FLOAT32 ? CV_32F :
759 header.data_type == FLANN_FLOAT64 ? CV_64F : -1;
760
761 if( (int)header.rows != data.rows || (int)header.cols != data.cols ||
762 featureType != data.type() )
763 {
764 fprintf(stderr, "Reading FLANN index error: the saved data size (%d, %d) or type (%d) is different from the passed one (%d, %d), %d\n",
765 (int)header.rows, (int)header.cols, featureType, data.rows, data.cols, data.type());
766 fclose(fin);
767 return false;
768 }
769
770 int idistType = 0;
771 ::cvflann::load_value(fin, idistType);
772 distType = (flann_distance_t)idistType;
773
774 if( !((distType == FLANN_DIST_HAMMING && featureType == CV_8U) ||
775 (distType != FLANN_DIST_HAMMING && featureType == CV_32F)) )
776 {
777 fprintf(stderr, "Reading FLANN index error: unsupported feature type %d for the index type %d\n", featureType, algo);
778 fclose(fin);
779 return false;
780 }
781
782 switch( distType )
783 {
784 case FLANN_DIST_HAMMING:
785 loadIndex< HammingDistance >(this, index, data, fin);
786 break;
787 case FLANN_DIST_L2:
788 loadIndex< ::cvflann::L2<float> >(this, index, data, fin);
789 break;
790 case FLANN_DIST_L1:
791 loadIndex< ::cvflann::L1<float> >(this, index, data, fin);
792 break;
793#if MINIFLANN_SUPPORT_EXOTIC_DISTANCE_TYPES
794 case FLANN_DIST_MAX:
795 loadIndex< ::cvflann::MaxDistance<float> >(this, index, data, fin);
796 break;
797 case FLANN_DIST_HIST_INTERSECT:
798 loadIndex< ::cvflann::HistIntersectionDistance<float> >(index, data, fin);
799 break;
800 case FLANN_DIST_HELLINGER:
801 loadIndex< ::cvflann::HellingerDistance<float> >(this, index, data, fin);
802 break;
803 case FLANN_DIST_CHI_SQUARE:
804 loadIndex< ::cvflann::ChiSquareDistance<float> >(this, index, data, fin);
805 break;
806 case FLANN_DIST_KL:
807 loadIndex< ::cvflann::KL_Divergence<float> >(this, index, data, fin);
808 break;
809#endif
810 default:
811 fprintf(stderr, "Reading FLANN index error: unsupported distance type %d\n", distType);
812 ok = false;
813 }
814
815 if( fin )
816 fclose(fin);
817 return ok;
818}
