111template<typename _Tp> void CV_OperationsTest::TestType(Size sz, _Tp value)
112{
113 cv::Mat_<_Tp> m(sz);
114 CV_Assert(m.cols == sz.width && m.rows == sz.height && m.depth() == DataType<_Tp>::depth &&
115 m.channels() == DataType<_Tp>::channels &&
116 m.elemSize() == sizeof(_Tp) && m.step == m.elemSize()*m.cols);
117 for( int y = 0; y < sz.height; y++ )
118 for( int x = 0; x < sz.width; x++ )
119 {
120 m(y,x) = value;
121 }
122
123 double s = sum(Mat(m).reshape(1))[0];
124 CV_Assert( s == (double)sz.width*sz.height );
125}
