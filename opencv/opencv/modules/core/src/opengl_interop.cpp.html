72 #define cudaSafeCall(expr) ___cudaSafeCall(expr, __FILE__, __LINE__, CV_Func)
73
74 void ___cudaSafeCall(cudaError_t err, const char* file, const int line, const char* func = "")

76 if (cudaSuccess != err)

83bool cv::ogl::checkError(const char* file, const int line, const char* func)
84{
85#ifndef HAVE_OPENGL
86 (void) file;
87 (void) line;
88 (void) func;
89 return true;
90#else
91 GLenum err = gl::GetError();
92
93 if (err != gl::NO_ERROR_)
94 {
95 const char* msg;
96
97 switch (err)
98 {
99 case gl::INVALID_ENUM:
100 msg = "An unacceptable value is specified for an enumerated argument";
101 break;
102
103 case gl::INVALID_VALUE:
104 msg = "A numeric argument is out of range";
105 break;
106
107 case gl::INVALID_OPERATION:
108 msg = "The specified operation is not allowed in the current state";
109 break;
110
111 case gl::OUT_OF_MEMORY:
112 msg = "There is not enough memory left to execute the command";
113 break;
114
115 default:
116 msg = "Unknown error";
117 };
118
119 cvError(CV_OpenGlApiCallError, func, msg, file, line);
120
121 return false;
122 }
123
124 return true;
125#endif
126}

162 public:

164 ~CudaResource();

166 void registerBuffer(GLuint buffer);

181
182 CudaResource::CudaResource() : resource_(0), buffer_(0)
183 {
184 }
185
186 CudaResource::~CudaResource()
187 {
188 release();
189 }
190
191 void CudaResource::registerBuffer(GLuint buffer)
192 {
193 CV_DbgAssert( buffer != 0 );
194
195 if (buffer_ == buffer)
196 return;
197
198 cudaGraphicsResource_t resource;
199 cudaSafeCall( cudaGraphicsGLRegisterBuffer(&resource, buffer, cudaGraphicsMapFlagsNone) );
200
201 release();
202
203 resource_ = resource;
204 buffer_ = buffer;
205 }
206
207 void CudaResource::release()
208 {
209 if (resource_)
210 cudaGraphicsUnregisterResource(resource_);
211
212 resource_ = 0;
213 buffer_ = 0;
214 }
215
216 class CudaResource::GraphicsMapHolder
217 {
218 public:
219 GraphicsMapHolder(cudaGraphicsResource_t* resource, cudaStream_t stream);
220 ~GraphicsMapHolder();
221
222 void reset();
223
224 private:
225 cudaGraphicsResource_t* resource_;
226 cudaStream_t stream_;
227 };
228
229 CudaResource::GraphicsMapHolder::GraphicsMapHolder(cudaGraphicsResource_t* resource, cudaStream_t stream) : resource_(resource), stream_(stream)
230 {
231 if (resource_)
232 cudaSafeCall( cudaGraphicsMapResources(1, resource_, stream_) );
233 }
234
235 CudaResource::GraphicsMapHolder::~GraphicsMapHolder()
236 {
237 if (resource_)
238 cudaGraphicsUnmapResources(1, resource_, stream_);
239 }
240
241 void CudaResource::GraphicsMapHolder::reset()
242 {
243 resource_ = 0;
244 }
245
246 void CudaResource::copyFrom(const void* src, size_t spitch, size_t width, size_t height, cudaStream_t stream)
247 {
248 CV_DbgAssert( resource_ != 0 );
249
250 GraphicsMapHolder h(&resource_, stream);
251 (void) h;
252
253 void* dst;
254 size_t size;
255 cudaSafeCall( cudaGraphicsResourceGetMappedPointer(&dst, &size, resource_) );
256
257 CV_DbgAssert( width * height == size );
258
259 if (stream == 0)
260 cudaSafeCall( cudaMemcpy2D(dst, width, src, spitch, width, height, cudaMemcpyDeviceToDevice) );
261 else
262 cudaSafeCall( cudaMemcpy2DAsync(dst, width, src, spitch, width, height, cudaMemcpyDeviceToDevice, stream) );
263 }
264
265 void CudaResource::copyTo(void* dst, size_t dpitch, size_t width, size_t height, cudaStream_t stream)
266 {
267 CV_DbgAssert( resource_ != 0 );
268
269 GraphicsMapHolder h(&resource_, stream);
270 (void) h;
271
272 void* src;
273 size_t size;
274 cudaSafeCall( cudaGraphicsResourceGetMappedPointer(&src, &size, resource_) );
275
276 CV_DbgAssert( width * height == size );
277
278 if (stream == 0)
279 cudaSafeCall( cudaMemcpy2D(dst, dpitch, src, width, width, height, cudaMemcpyDeviceToDevice) );
280 else
281 cudaSafeCall( cudaMemcpy2DAsync(dst, dpitch, src, width, width, height, cudaMemcpyDeviceToDevice, stream) );
282 }
283
284 void* CudaResource::map(cudaStream_t stream)
285 {
286 CV_DbgAssert( resource_ != 0 );
287
288 GraphicsMapHolder h(&resource_, stream);
289
290 void* ptr;
291 size_t size;
292 cudaSafeCall( cudaGraphicsResourceGetMappedPointer(&ptr, &size, resource_) );
293
294 h.reset();
295
296 return ptr;
297 }
298
299 void CudaResource::unmap(cudaStream_t stream)
300 {
301 CV_Assert( resource_ != 0 );
302
303 cudaGraphicsUnmapResources(1, &resource_, stream);
304 }
305}

376cv::ogl::Buffer::Impl::Impl(GLsizeiptr size, const GLvoid* data, GLenum target, bool autoRelease) : bufId_(0), autoRelease_(autoRelease)
377{
378 gl::GenBuffers(1, &bufId_);
379 CV_CheckGlError();
380
381 CV_Assert( bufId_ != 0 );
382
383 gl::BindBuffer(target, bufId_);
384 CV_CheckGlError();
385
386 gl::BufferData(target, size, data, gl::DYNAMIC_DRAW);
387 CV_CheckGlError();
388
389 gl::BindBuffer(target, 0);
390 CV_CheckGlError();
391}

393cv::ogl::Buffer::Impl::~Impl()
394{
395 if (autoRelease_ && bufId_)
396 gl::DeleteBuffers(1, &bufId_);
397}

530cv::ogl::Buffer::Buffer(InputArray arr, Target target, bool autoRelease) : rows_(0), cols_(0), type_(0)
531{
532#ifndef HAVE_OPENGL
533 (void) arr;
534 (void) target;
535 (void) autoRelease;
536 throw_nogl();
537#else
538 const int kind = arr.kind();
539
540 switch (kind)
541 {
542 case _InputArray::OPENGL_BUFFER:
543 {
544 copyFrom(arr, target, autoRelease);
545 break;
546 }
547
548 case _InputArray::OPENGL_TEXTURE:
549 {
550 copyFrom(arr, target, autoRelease);
551 break;
552 }
553
554 case _InputArray::GPU_MAT:
555 {
556 copyFrom(arr, target, autoRelease);
557 break;
558 }
559
560 default:
561 {
562 Mat mat = arr.getMat();
563 CV_Assert( mat.isContinuous() );
564 const GLsizeiptr asize = mat.rows * mat.cols * mat.elemSize();
565 impl_ = new Impl(asize, mat.data, target, autoRelease);
566 rows_ = mat.rows;
567 cols_ = mat.cols;
568 type_ = mat.type();
569 break;
570 }
571 }
572#endif
573}

618void cv::ogl::Buffer::copyFrom(InputArray arr, Target target, bool autoRelease)
619{
620#ifndef HAVE_OPENGL
621 (void) arr;
622 (void) target;
623 (void) autoRelease;
624 throw_nogl();
625#else
626 const int kind = arr.kind();
627
628 if (kind == _InputArray::OPENGL_TEXTURE)
629 {
630 ogl::Texture2D tex = arr.getOGlTexture2D();
631 tex.copyTo(*this);
632 setAutoRelease(autoRelease);
633 return;
634 }
635
636 const Size asize = arr.size();
637 const int atype = arr.type();
638 create(asize, atype, target, autoRelease);
639
640 switch (kind)
641 {
642 case _InputArray::OPENGL_BUFFER:
643 {
644 ogl::Buffer buf = arr.getOGlBuffer();
645 impl_->copyFrom(buf.bufId(), asize.area() * CV_ELEM_SIZE(atype));
646 break;
647 }
648
649 case _InputArray::GPU_MAT:
650 {
651 #if !defined HAVE_CUDA || defined(CUDA_DISABLER)
652 throw_nocuda();
653 #else
654 GpuMat dmat = arr.getGpuMat();
655 impl_->copyFrom(dmat.data, dmat.step, dmat.cols * dmat.elemSize(), dmat.rows);
656 #endif
657
658 break;
659 }
660
661 default:
662 {
663 Mat mat = arr.getMat();
664 CV_Assert( mat.isContinuous() );
665 impl_->copyFrom(asize.area() * CV_ELEM_SIZE(atype), mat.data);
666 }
667 }
668#endif
669}

671void cv::ogl::Buffer::copyTo(OutputArray arr, Target target, bool autoRelease) const
672{
673#ifndef HAVE_OPENGL
674 (void) arr;
675 (void) target;
676 (void) autoRelease;
677 throw_nogl();
678#else
679 const int kind = arr.kind();
680
681 switch (kind)
682 {
683 case _InputArray::OPENGL_BUFFER:
684 {
685 arr.getOGlBufferRef().copyFrom(*this, target, autoRelease);
686 break;
687 }
688
689 case _InputArray::OPENGL_TEXTURE:
690 {
691 arr.getOGlTexture2DRef().copyFrom(*this, autoRelease);
692 break;
693 }
694
695 case _InputArray::GPU_MAT:
696 {
697 #if !defined HAVE_CUDA || defined(CUDA_DISABLER)
698 throw_nocuda();
699 #else
700 GpuMat& dmat = arr.getGpuMatRef();
701 dmat.create(rows_, cols_, type_);
702 impl_->copyTo(dmat.data, dmat.step, dmat.cols * dmat.elemSize(), dmat.rows);
703 #endif
704
705 break;
706 }
707
708 default:
709 {
710 arr.create(rows_, cols_, type_);
711 Mat mat = arr.getMat();
712 CV_Assert( mat.isContinuous() );
713 impl_->copyTo(mat.rows * mat.cols * mat.elemSize(), mat.data);
714 }
715 }
716#endif
717}
