78bool clipLine( Size img_size, Point& pt1, Point& pt2 )
79{
80 Point2l p1(pt1.x, pt1.y);
81 Point2l p2(pt2.x, pt2.y);
82 bool inside = clipLine(Size2l(img_size.width, img_size.height), p1, p2);
83 pt1.x = (int)p1.x;
84 pt1.y = (int)p1.y;
85 pt2.x = (int)p2.x;
86 pt2.y = (int)p2.y;
87 return inside;
88}

90bool clipLine( Size2l img_size, Point2l& pt1, Point2l& pt2 )
91{
92 int c1, c2;
93 int64 right = img_size.width-1, bottom = img_size.height-1;
94
95 if( img_size.width <= 0 || img_size.height <= 0 )
96 return false;
97
98 int64 &x1 = pt1.x, &y1 = pt1.y, &x2 = pt2.x, &y2 = pt2.y;
99 c1 = (x1 < 0) + (x1 > right) * 2 + (y1 < 0) * 4 + (y1 > bottom) * 8;
100 c2 = (x2 < 0) + (x2 > right) * 2 + (y2 < 0) * 4 + (y2 > bottom) * 8;
101
102 if( (c1 & c2) == 0 && (c1 | c2) != 0 )
103 {
104 int64 a;
105 if( c1 & 12 )
106 {
107 a = c1 < 8 ? 0 : bottom;
108 x1 += (a - y1) * (x2 - x1) / (y2 - y1);
109 y1 = a;
110 c1 = (x1 < 0) + (x1 > right) * 2;
111 }
112 if( c2 & 12 )
113 {
114 a = c2 < 8 ? 0 : bottom;
115 x2 += (a - y2) * (x2 - x1) / (y2 - y1);
116 y2 = a;
117 c2 = (x2 < 0) + (x2 > right) * 2;
118 }
119 if( (c1 & c2) == 0 && (c1 | c2) != 0 )
120 {
121 if( c1 )
122 {
123 a = c1 == 1 ? 0 : right;
124 y1 += (a - x1) * (y2 - y1) / (x2 - x1);
125 x1 = a;
126 c1 = 0;
127 }
128 if( c2 )
129 {
130 a = c2 == 1 ? 0 : right;
131 y2 += (a - x2) * (y2 - y1) / (x2 - x1);
132 x2 = a;
133 c2 = 0;
134 }
135 }
136
137 assert( (c1 & c2) != 0 || (x1 | y1 | x2 | y2) >= 0 );
138 }
139
140 return (c1 | c2) == 0;
141}

143bool clipLine( Rect img_rect, Point& pt1, Point& pt2 )
144{
145 Point tl = img_rect.tl();
146 pt1 -= tl; pt2 -= tl;
147 bool inside = clipLine(img_rect.size(), pt1, pt2);
148 pt1 += tl; pt2 += tl;

150 return inside;

153/*
154 Initializes line iterator.
155 Returns number of points on the line or negative number if error.
156*/
157LineIterator::LineIterator(const Mat& img, Point pt1, Point pt2,
158 int connectivity, bool left_to_right)
159{
160 count = -1;
161
162 CV_Assert( connectivity == 8 || connectivity == 4 );
163
164 if( (unsigned)pt1.x >= (unsigned)(img.cols) ||
165 (unsigned)pt2.x >= (unsigned)(img.cols) ||
166 (unsigned)pt1.y >= (unsigned)(img.rows) ||
167 (unsigned)pt2.y >= (unsigned)(img.rows) )
168 {
169 if( !clipLine( img.size(), pt1, pt2 ) )
170 {
171 ptr = img.data;
172 err = plusDelta = minusDelta = plusStep = minusStep = count = 0;
173 return;
174 }
175 }
176
177 int bt_pix0 = (int)img.elemSize(), bt_pix = bt_pix0;
178 size_t istep = img.step;
179
180 int dx = pt2.x - pt1.x;
181 int dy = pt2.y - pt1.y;
182 int s = dx < 0 ? -1 : 0;
183
184 if( left_to_right )
185 {
186 dx = (dx ^ s) - s;
187 dy = (dy ^ s) - s;
188 pt1.x ^= (pt1.x ^ pt2.x) & s;
189 pt1.y ^= (pt1.y ^ pt2.y) & s;
190 }
191 else
192 {
193 dx = (dx ^ s) - s;
194 bt_pix = (bt_pix ^ s) - s;
195 }
196
197 ptr = (uchar*)(img.data + pt1.y * istep + pt1.x * bt_pix0);
198
199 s = dy < 0 ? -1 : 0;
200 dy = (dy ^ s) - s;
201 istep = (istep ^ s) - s;
202
203 s = dy > dx ? -1 : 0;
204
205 /* conditional swaps */
206 dx ^= dy & s;
207 dy ^= dx & s;
208 dx ^= dy & s;
209
210 bt_pix ^= istep & s;
211 istep ^= bt_pix & s;
212 bt_pix ^= istep & s;
213
214 if( connectivity == 8 )
215 {
216 assert( dx >= 0 && dy >= 0 );
217
218 err = dx - (dy + dy);
219 plusDelta = dx + dx;
220 minusDelta = -(dy + dy);
221 plusStep = (int)istep;
222 minusStep = bt_pix;
223 count = dx + 1;
224 }
225 else /* connectivity == 4 */
226 {
227 assert( dx >= 0 && dy >= 0 );
228
229 err = 0;
230 plusDelta = (dx + dx) + (dy + dy);
231 minusDelta = -(dy + dy);
232 plusStep = (int)istep - bt_pix;
233 minusStep = bt_pix;
234 count = dx + dy + 1;
235 }
236
237 this->ptr0 = img.data;
238 this->step = (int)img.step;
239 this->elemSize = bt_pix0;
240}

242static void
243Line( Mat& img, Point pt1, Point pt2,
244 const void* _color, int connectivity = 8 )
245{
246 if( connectivity == 0 )
247 connectivity = 8;
248 else if( connectivity == 1 )
249 connectivity = 4;
250
251 LineIterator iterator(img, pt1, pt2, connectivity, true);
252 int i, count = iterator.count;
253 int pix_size = (int)img.elemSize();
254 const uchar* color = (const uchar*)_color;
255
256 for( i = 0; i < count; i++, ++iterator )
257 {
258 uchar* ptr = *iterator;
259 if( pix_size == 1 )
260 ptr[0] = color[0];
261 else if( pix_size == 3 )
262 {
263 ptr[0] = color[0];
264 ptr[1] = color[1];
265 ptr[2] = color[2];
266 }
267 else
268 memcpy( *iterator, color, pix_size );
269 }
270}

287static void
288LineAA( Mat& img, Point2l pt1, Point2l pt2, const void* color )
289{
290 int64 dx, dy;
291 int ecount, scount = 0;
292 int slope;
293 int64 ax, ay;
294 int64 x_step, y_step;
295 int64 i, j;
296 int ep_table[9];
297 int cb = ((uchar*)color)[0], cg = ((uchar*)color)[1], cr = ((uchar*)color)[2], ca = ((uchar*)color)[3];
298 int _cb, _cg, _cr, _ca;
299 int nch = img.channels();
300 uchar* ptr = img.data;
301 size_t step = img.step;
302 Size2l size(img.cols, img.rows);
303
304 if( !((nch == 1 || nch == 3 || nch == 4) && img.depth() == CV_8U) )
305 {
306 Line(img, Point((int)(pt1.x<<XY_SHIFT), (int)(pt1.y<<XY_SHIFT)), Point((int)(pt2.x<<XY_SHIFT), (int)(pt2.y<<XY_SHIFT)), color);
307 return;
308 }
309
310 pt1.x -= XY_ONE*2;
311 pt1.y -= XY_ONE*2;
312 pt2.x -= XY_ONE*2;
313 pt2.y -= XY_ONE*2;
314 ptr += img.step*2 + 2*nch;
315
316 size.width = ((size.width - 5) << XY_SHIFT) + 1;
317 size.height = ((size.height - 5) << XY_SHIFT) + 1;
318
319 if( !clipLine( size, pt1, pt2 ))
320 return;
321
322 dx = pt2.x - pt1.x;
323 dy = pt2.y - pt1.y;
324
325 j = dx < 0 ? -1 : 0;
326 ax = (dx ^ j) - j;
327 i = dy < 0 ? -1 : 0;
328 ay = (dy ^ i) - i;
329
330 if( ax > ay )
331 {
332 dx = ax;
333 dy = (dy ^ j) - j;
334 pt1.x ^= pt2.x & j;
335 pt2.x ^= pt1.x & j;
336 pt1.x ^= pt2.x & j;
337 pt1.y ^= pt2.y & j;
338 pt2.y ^= pt1.y & j;
339 pt1.y ^= pt2.y & j;
340
341 x_step = XY_ONE;
342 y_step = (dy << XY_SHIFT) / (ax | 1);
343 pt2.x += XY_ONE;
344 ecount = (int)((pt2.x >> XY_SHIFT) - (pt1.x >> XY_SHIFT));
345 j = -(pt1.x & (XY_ONE - 1));
346 pt1.y += ((y_step * j) >> XY_SHIFT) + (XY_ONE >> 1);
347 slope = (y_step >> (XY_SHIFT - 5)) & 0x3f;
348 slope ^= (y_step < 0 ? 0x3f : 0);
349
350 /* Get 4-bit fractions for end-point adjustments */
351 i = (pt1.x >> (XY_SHIFT - 7)) & 0x78;
352 j = (pt2.x >> (XY_SHIFT - 7)) & 0x78;
353 }
354 else
355 {
356 dy = ay;
357 dx = (dx ^ i) - i;
358 pt1.x ^= pt2.x & i;
359 pt2.x ^= pt1.x & i;
360 pt1.x ^= pt2.x & i;
361 pt1.y ^= pt2.y & i;
362 pt2.y ^= pt1.y & i;
363 pt1.y ^= pt2.y & i;
364
365 x_step = (dx << XY_SHIFT) / (ay | 1);
366 y_step = XY_ONE;
367 pt2.y += XY_ONE;
368 ecount = (int)((pt2.y >> XY_SHIFT) - (pt1.y >> XY_SHIFT));
369 j = -(pt1.y & (XY_ONE - 1));
370 pt1.x += ((x_step * j) >> XY_SHIFT) + (XY_ONE >> 1);
371 slope = (x_step >> (XY_SHIFT - 5)) & 0x3f;
372 slope ^= (x_step < 0 ? 0x3f : 0);
373
374 /* Get 4-bit fractions for end-point adjustments */
375 i = (pt1.y >> (XY_SHIFT - 7)) & 0x78;
376 j = (pt2.y >> (XY_SHIFT - 7)) & 0x78;
377 }
378
379 slope = (slope & 0x20) ? 0x100 : SlopeCorrTable[slope];
380
381 /* Calc end point correction table */
382 {
383 int t0 = slope << 7;
384 int t1 = ((0x78 - (int)i) | 4) * slope;
385 int t2 = ((int)j | 4) * slope;
386
387 ep_table[0] = 0;
388 ep_table[8] = slope;
389 ep_table[1] = ep_table[3] = ((((j - i) & 0x78) | 4) * slope >> 8) & 0x1ff;
390 ep_table[2] = (t1 >> 8) & 0x1ff;
391 ep_table[4] = ((((j - i) + 0x80) | 4) * slope >> 8) & 0x1ff;
392 ep_table[5] = ((t1 + t0) >> 8) & 0x1ff;
393 ep_table[6] = (t2 >> 8) & 0x1ff;
394 ep_table[7] = ((t2 + t0) >> 8) & 0x1ff;
395 }
396
397 if( nch == 3 )
398 {
399 #define ICV_PUT_POINT() \
400 { \
401 _cb = tptr[0]; \
402 _cb += ((cb - _cb)*a + 127)>> 8;\
403 _cg = tptr[1]; \
404 _cg += ((cg - _cg)*a + 127)>> 8;\
405 _cr = tptr[2]; \
406 _cr += ((cr - _cr)*a + 127)>> 8;\
407 tptr[0] = (uchar)_cb; \
408 tptr[1] = (uchar)_cg; \
409 tptr[2] = (uchar)_cr; \
410 }
411 if( ax > ay )
412 {
413 ptr += (pt1.x >> XY_SHIFT) * 3;
414
415 while( ecount >= 0 )
416 {
417 uchar *tptr = ptr + ((pt1.y >> XY_SHIFT) - 1) * step;
418
419 int ep_corr = ep_table[(((scount >= 2) + 1) & (scount | 2)) * 3 +
420 (((ecount >= 2) + 1) & (ecount | 2))];
421 int a, dist = (pt1.y >> (XY_SHIFT - 5)) & 31;
422
423 a = (ep_corr * FilterTable[dist + 32] >> 8) & 0xff;
424 ICV_PUT_POINT();
425 ICV_PUT_POINT();
426
427 tptr += step;
428 a = (ep_corr * FilterTable[dist] >> 8) & 0xff;
429 ICV_PUT_POINT();
430 ICV_PUT_POINT();
431
432 tptr += step;
433 a = (ep_corr * FilterTable[63 - dist] >> 8) & 0xff;
434 ICV_PUT_POINT();
435 ICV_PUT_POINT();
436
437 pt1.y += y_step;
438 ptr += 3;
439 scount++;
440 ecount--;
441 }
442 }
443 else
444 {
445 ptr += (pt1.y >> XY_SHIFT) * step;
446
447 while( ecount >= 0 )
448 {
449 uchar *tptr = ptr + ((pt1.x >> XY_SHIFT) - 1) * 3;
450
451 int ep_corr = ep_table[(((scount >= 2) + 1) & (scount | 2)) * 3 +
452 (((ecount >= 2) + 1) & (ecount | 2))];
453 int a, dist = (pt1.x >> (XY_SHIFT - 5)) & 31;
454
455 a = (ep_corr * FilterTable[dist + 32] >> 8) & 0xff;
456 ICV_PUT_POINT();
457 ICV_PUT_POINT();
458
459 tptr += 3;
460 a = (ep_corr * FilterTable[dist] >> 8) & 0xff;
461 ICV_PUT_POINT();
462 ICV_PUT_POINT();
463
464 tptr += 3;
465 a = (ep_corr * FilterTable[63 - dist] >> 8) & 0xff;
466 ICV_PUT_POINT();
467 ICV_PUT_POINT();
468
469 pt1.x += x_step;
470 ptr += step;
471 scount++;
472 ecount--;
473 }
474 }
475 #undef ICV_PUT_POINT
476 }
477 else if (nch == 1)
478 {
479 #define ICV_PUT_POINT() \
480 { \
481 _cb = tptr[0]; \
482 _cb += ((cb - _cb)*a + 127)>> 8;\
483 tptr[0] = (uchar)_cb; \

485

487 {

638static void
639Line2( Mat& img, Point2l pt1, Point2l pt2, const void* color)
640{
641 int64 dx, dy;
642 int ecount;
643 int64 ax, ay;
644 int64 i, j;
645 int x, y;
646 int64 x_step, y_step;
647 int cb = ((uchar*)color)[0];
648 int cg = ((uchar*)color)[1];
649 int cr = ((uchar*)color)[2];
650 int pix_size = (int)img.elemSize();
651 uchar *ptr = img.data, *tptr;
652 size_t step = img.step;
653 Size size = img.size();
654
655 //assert( img && (nch == 1 || nch == 3) && img.depth() == CV_8U );
656
657 Size2l sizeScaled(((int64)size.width) << XY_SHIFT, ((int64)size.height) << XY_SHIFT);
658 if( !clipLine( sizeScaled, pt1, pt2 ))
659 return;
660
661 dx = pt2.x - pt1.x;
662 dy = pt2.y - pt1.y;
663
664 j = dx < 0 ? -1 : 0;
665 ax = (dx ^ j) - j;
666 i = dy < 0 ? -1 : 0;
667 ay = (dy ^ i) - i;
668
669 if( ax > ay )
670 {
671 dx = ax;
672 dy = (dy ^ j) - j;
673 pt1.x ^= pt2.x & j;
674 pt2.x ^= pt1.x & j;
675 pt1.x ^= pt2.x & j;
676 pt1.y ^= pt2.y & j;
677 pt2.y ^= pt1.y & j;
678 pt1.y ^= pt2.y & j;
679
680 x_step = XY_ONE;
681 y_step = (dy << XY_SHIFT) / (ax | 1);
682 ecount = (int)((pt2.x - pt1.x) >> XY_SHIFT);
683 }
684 else
685 {

952void ellipse2Poly( Point2d center, Size2d axes, int angle,
953 int arc_start, int arc_end,
954 int delta, vector<Point2d>& pts )
955{
956 float alpha, beta;
957 int i;
958
959 while( angle < 0 )
960 angle += 360;
961 while( angle > 360 )
962 angle -= 360;
963
964 if( arc_start > arc_end )
965 {
966 i = arc_start;
967 arc_start = arc_end;
968 arc_end = i;
969 }
970 while( arc_start < 0 )
971 {
972 arc_start += 360;
973 arc_end += 360;
974 }
975 while( arc_end > 360 )
976 {
977 arc_end -= 360;
978 arc_start -= 360;
979 }
980 if( arc_end - arc_start > 360 )
981 {
982 arc_start = 0;
983 arc_end = 360;
984 }
985 sincos( angle, alpha, beta );
986 pts.resize(0);
987
988 for( i = arc_start; i < arc_end + delta; i += delta )
989 {
990 double x, y;
991 angle = i;
992 if( angle > arc_end )
993 angle = arc_end;
994 if( angle < 0 )
995 angle += 360;
996
997 x = axes.width * SinTable[450-angle];
998 y = axes.height * SinTable[angle];
999 Point2d pt;
1000 pt.x = center.x + x * alpha - y * beta;
1001 pt.y = center.y + x * beta + y * alpha;
1002 pts.push_back(pt);
1003 }
1004
1005 if( pts.size() == 1 )
1006 pts.push_back(pts[0]);
1007}

1168 {
1169 for( i = 0; i < 2; i++ )
1170 {
1171 if( y >= edge[i].ye )
1172 {
1173 int idx = edge[i].idx, di = edge[i].di;
1174 int64 xs = 0, xe;
1175 int ty = 0;
1176
1177 for(;;)
1178 {
1179 ty = (int)((v[idx].y + delta) >> shift);
1180 if( ty > y || edges == 0 )
1181 break;
1182 xs = v[idx].x;
1183 idx += di;
1184 idx -= ((idx < npts) - 1) & npts; /* idx -= idx >= npts ? npts : 0 */
1185 edges--;
1186 }
1187
1188 xe = v[idx].x;
1189 if (XY_SHIFT - shift != 0)
1190 {
1191 xs <<= XY_SHIFT - shift;
1192 xe <<= XY_SHIFT - shift;
1193 }
1194
1195 /* no more edges */
1196 if( y >= ty)
1197 return;
1198
1199 edge[i].ye = ty;
1200 edge[i].dx = ((xe - xs)*2 + (ty - y)) / (2 * (ty - y));
1201 edge[i].x = xs;
1202 edge[i].idx = idx;
1203 }
1204 }
1205 }
1206
1207 if( edge[left].x > edge[right].x )
1208 {
1209 left ^= 1;
1210 right ^= 1;
1211 }
1212
1213 if( y >= 0 )
1214 {
1215 int xx1 = (int)((edge[left].x + delta1) >> XY_SHIFT);
1216 int xx2 = (int)((edge[right].x + delta2) >> XY_SHIFT);
1217
1218 if( xx2 >= 0 && xx1 < size.width )
1219 {
1220 if( xx1 < 0 )
1221 xx1 = 0;
1222 if( xx2 >= size.width )
1223 xx2 = size.width - 1;
1224 ICV_HLINE( ptr, xx1, xx2, color, pix_size );
1225 }
1226 }
1227
1228 edge[left].x += edge[left].dx;
1229 edge[right].x += edge[right].dx;
1230 ptr += img.step;
1231 }
1232 while( ++y <= (int)ymax );
1233}

1237
1238static void
1239CollectPolyEdges( Mat& img, const Point2l* v, int count, vector<PolyEdge>& edges,
1240 const void* color, int line_type, int shift, Point offset )
1241{
1242 int i, delta = offset.y + ((1 << shift) >> 1);
1243 Point2l pt0 = v[count-1], pt1;
1244 pt0.x = (pt0.x + offset.x) << (XY_SHIFT - shift);
1245 pt0.y = (pt0.y + delta) >> shift;
1246
1247 edges.reserve( edges.size() + count );
1248
1249 for( i = 0; i < count; i++, pt0 = pt1 )
1250 {
1251 Point2l t0, t1;
1252 PolyEdge edge;
1253
1254 pt1 = v[i];
1255 pt1.x = (pt1.x + offset.x) << (XY_SHIFT - shift);
1256 pt1.y = (pt1.y + delta) >> shift;
1257
1258 if( line_type < CV_AA )
1259 {
1260 t0.y = pt0.y; t1.y = pt1.y;
1261 t0.x = (pt0.x + (XY_ONE >> 1)) >> XY_SHIFT;
1262 t1.x = (pt1.x + (XY_ONE >> 1)) >> XY_SHIFT;
1263 Line( img, Point((int)(t0.x), (int)(t0.y)), Point((int)(t1.x), (int)(t1.y)), color, line_type );
1264 }
1265 else
1266 {
1267 t0.x = pt0.x; t1.x = pt1.x;
1268 t0.y = pt0.y << XY_SHIFT;
1269 t1.y = pt1.y << XY_SHIFT;
1270 LineAA( img, t0, t1, color );
1271 }
1272
1273 if( pt0.y == pt1.y )
1274 continue;
1275
1276 if( pt0.y < pt1.y )
1277 {
1278 edge.y0 = (int)(pt0.y);
1279 edge.y1 = (int)(pt1.y);
1280 edge.x = pt0.x;
1281 }
1282 else
1283 {
1284 edge.y0 = (int)(pt1.y);
1285 edge.y1 = (int)(pt0.y);
1286 edge.x = pt1.x;
1287 }
1288 edge.dx = (pt1.x - pt0.x) / (pt1.y - pt0.y);
1289 edges.push_back(edge);
1290 }
1291}

1293struct CmpEdges
1294{
1295 bool operator ()(const PolyEdge& e1, const PolyEdge& e2)
1296 {
1297 return e1.y0 - e2.y0 ? e1.y0 < e2.y0 :
1298 e1.x - e2.x ? e1.x < e2.x : e1.dx < e2.dx;
1299 }
1300};

1304static void
1305FillEdgeCollection( Mat& img, vector<PolyEdge>& edges, const void* color )
1306{
1307 PolyEdge tmp;
1308 int i, y, total = (int)edges.size();
1309 Size size = img.size();
1310 PolyEdge* e;
1311 int y_max = INT_MIN, y_min = INT_MAX;
1312 int64 x_max = 0xFFFFFFFFFFFFFFFF, x_min = 0x7FFFFFFFFFFFFFFF;
1313 int pix_size = (int)img.elemSize();
1314
1315 if( total < 2 )
1316 return;
1317
1318 for( i = 0; i < total; i++ )
1319 {
1320 PolyEdge& e1 = edges[i];
1321 assert( e1.y0 < e1.y1 );
1322 // Determine x-coordinate of the end of the edge.
1323 // (This is not necessary x-coordinate of any vertex in the array.)
1324 int64 x1 = e1.x + (e1.y1 - e1.y0) * e1.dx;
1325 y_min = std::min( y_min, e1.y0 );
1326 y_max = std::max( y_max, e1.y1 );
1327 x_min = std::min( x_min, e1.x );
1328 x_max = std::max( x_max, e1.x );
1329 x_min = std::min( x_min, x1 );
1330 x_max = std::max( x_max, x1 );
1331 }
1332
1333 if( y_max < 0 || y_min >= size.height || x_max < 0 || x_min >= ((int64)size.width<<XY_SHIFT) )
1334 return;
1335
1336 std::sort( edges.begin(), edges.end(), CmpEdges() );
1337
1338 // start drawing
1339 tmp.y0 = INT_MAX;
1340 edges.push_back(tmp); // after this point we do not add
1341 // any elements to edges, thus we can use pointers
1342 i = 0;
1343 tmp.next = 0;
1344 e = &edges[i];
1345 y_max = MIN( y_max, size.height );
1346
1347 for( y = e->y0; y < y_max; y++ )
1348 {
1349 PolyEdge *last, *prelast, *keep_prelast;
1350 int sort_flag = 0;
1351 int draw = 0;
1352 int clipline = y < 0;
1353
1354 prelast = &tmp;
1355 last = tmp.next;
1356 while( last || e->y0 == y )
1357 {
1358 if( last && last->y1 == y )
1359 {
1360 // exclude edge if y reachs its lower point
1361 prelast->next = last->next;
1362 last = last->next;
1363 continue;
1364 }
1365 keep_prelast = prelast;
1366 if( last && (e->y0 > y || last->x < e->x) )
1367 {
1368 // go to the next edge in active list
1369 prelast = last;
1370 last = last->next;
1371 }
1372 else if( i < total )
1373 {
1374 // insert new edge into active list if y reachs its upper point
1375 prelast->next = e;
1376 e->next = last;
1377 prelast = e;
1378 e = &edges[++i];
1379 }
1380 else
1381 break;
1382
1383 if( draw )
1384 {
1385 if( !clipline )
1386 {
1387 // convert x's from fixed-point to image coordinates
1388 uchar *timg = img.data + y * img.step;
1389 int x1, x2;
1390
1391 if (keep_prelast->x > prelast->x)
1392 {
1393 x1 = (int)((prelast->x + XY_ONE - 1) >> XY_SHIFT);
1394 x2 = (int)(keep_prelast->x >> XY_SHIFT);
1395 }
1396 else
1397 {
1398 x1 = (int)((keep_prelast->x + XY_ONE - 1) >> XY_SHIFT);
1399 x2 = (int)(prelast->x >> XY_SHIFT);
1400 }
1401
1402 // clip and draw the line
1403 if( x1 < size.width && x2 >= 0 )
1404 {
1405 if( x1 < 0 )
1406 x1 = 0;
1407 if( x2 >= size.width )
1408 x2 = size.width - 1;
1409 ICV_HLINE( timg, x1, x2, color, pix_size );
1410 }
1411 }
1412 keep_prelast->x += keep_prelast->dx;
1413 prelast->x += prelast->dx;
1414 }
1415 draw ^= 1;
1416 }
1417
1418 // sort edges (using bubble sort)
1419 keep_prelast = 0;
1420
1421 do
1422 {
1423 prelast = &tmp;
1424 last = tmp.next;
1425
1426 while( last != keep_prelast && last->next != 0 )
1427 {
1428 PolyEdge *te = last->next;
1429
1430 // swap edges
1431 if( last->x > te->x )
1432 {
1433 prelast->next = te;
1434 last->next = te->next;
1435 te->next = last;
1436 prelast = te;
1437 sort_flag = 1;
1438 }
1439 else
1440 {
1441 prelast = last;
1442 last = te;
1443 }
1444 }
1445 keep_prelast = prelast;
1446 }
1447 while( sort_flag && keep_prelast != tmp.next && keep_prelast != &tmp );
1448 }
1449}

1454Circle( Mat& img, Point center, int radius, const void* color, int fill )

1458 int pix_size = (int)img.elemSize();

1471 int x11 = center.x - dx, x12 = center.x + dx, x21 = center.x - dy, x22 = center.x + dy;
1472

1516 {
1517 uchar *tptr = ptr + y11 * step;
1518

1520 {

1522 ICV_PUT_POINT( tptr, x11 );

1542 ICV_HLINE( tptr, x11, x12, color, pix_size );
1543 }
1544
1545 if( x21 < size.width && x22 >= 0 )
1546 {
1547 if( fill )
1548 {
1549 x21 = std::max( x21, 0 );
1550 x22 = MIN( x22, size.width - 1 );

1552
1553 if( (unsigned)y21 < (unsigned)size.height )
1554 {
1555 uchar *tptr = ptr + y21 * step;
1556
1557 if( !fill )
1558 {
1559 if( x21 >= 0 )
1560 ICV_PUT_POINT( tptr, x21 );

1562 ICV_PUT_POINT( tptr, x22 );

1564 else

1566 }
1567
1568 if( (unsigned)y22 < (unsigned)size.height )

1572 if( !fill )
1573 {

1599static void
1600ThickLine( Mat& img, Point2l p0, Point2l p1, const void* color,
1601 int thickness, int line_type, int flags, int shift )
1602{
1603 static const double INV_XY_ONE = 1./XY_ONE;
1604
1605 p0.x <<= XY_SHIFT - shift;
1606 p0.y <<= XY_SHIFT - shift;
1607 p1.x <<= XY_SHIFT - shift;
1608 p1.y <<= XY_SHIFT - shift;
1609
1610 if( thickness <= 1 )
1611 {
1612 if( line_type < CV_AA )
1613 {
1614 if( line_type == 1 || line_type == 4 || shift == 0 )
1615 {
1616 p0.x = (p0.x + (XY_ONE>>1)) >> XY_SHIFT;
1617 p0.y = (p0.y + (XY_ONE>>1)) >> XY_SHIFT;
1618 p1.x = (p1.x + (XY_ONE>>1)) >> XY_SHIFT;
1619 p1.y = (p1.y + (XY_ONE>>1)) >> XY_SHIFT;
1620 Line( img, Point((int)(p0.x), (int)(p0.y)), Point((int)(p1.x), (int)(p1.y)), color, line_type );
1621 }
1622 else
1623 Line2( img, p0, p1, color );
1624 }
1625 else
1626 LineAA( img, p0, p1, color );
1627 }
1628 else
1629 {
1630 Point2l pt[4], dp = Point2l(0,0);
1631 double dx = (p0.x - p1.x)*INV_XY_ONE, dy = (p1.y - p0.y)*INV_XY_ONE;
1632 double r = dx * dx + dy * dy;
1633 int i, oddThickness = thickness & 1;
1634 thickness <<= XY_SHIFT - 1;
1635
1636 if( fabs(r) > DBL_EPSILON )
1637 {
1638 r = (thickness + oddThickness*XY_ONE*0.5)/std::sqrt(r);
1639 dp.x = cvRound( dy * r );
1640 dp.y = cvRound( dx * r );
1641
1642 pt[0].x = p0.x + dp.x;
1643 pt[0].y = p0.y + dp.y;
1644 pt[1].x = p0.x - dp.x;
1645 pt[1].y = p0.y - dp.y;
1646 pt[2].x = p1.x - dp.x;
1647 pt[2].y = p1.y - dp.y;
1648 pt[3].x = p1.x + dp.x;
1649 pt[3].y = p1.y + dp.y;
1650
1651 FillConvexPoly( img, pt, 4, color, line_type, XY_SHIFT );
1652 }
1653
1654 for( i = 0; i < 2; i++ )
1655 {
1656 if( flags & (i+1) )
1657 {
1658 if( line_type < CV_AA )
1659 {
1660 Point center;
1661 center.x = (int)((p0.x + (XY_ONE>>1)) >> XY_SHIFT);
1662 center.y = (int)((p0.y + (XY_ONE>>1)) >> XY_SHIFT);
1663 Circle( img, center, (thickness + (XY_ONE>>1)) >> XY_SHIFT, color, 1 );
1664 }
1665 else
1666 {
1667 EllipseEx( img, p0, Size2l(thickness, thickness),
1668 0, 0, 360, color, -1, line_type );
1669 }
1670 }
1671 p0 = p1;
1672 }
1673 }
1674}
