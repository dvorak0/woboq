113static const std::string getCudaSupportLibName()
114{
115 Dl_info dl_info;
116 if(0 != dladdr((void *)getCudaSupportLibName, &dl_info))
117 {
118 LOGD("Library name: %s", dl_info.dli_fname);
119 LOGD("Library base address: %p", dl_info.dli_fbase);
120
121 const char* libName=dl_info.dli_fname;
122 while( ((*libName)=='/') || ((*libName)=='.') )
123 libName++;
124
125 char lineBuf[2048];
126 FILE* file = fopen("/proc/self/smaps", "rt");
127
128 if(file)
129 {
130 while (fgets(lineBuf, sizeof lineBuf, file) != NULL)
131 {
132 //verify that line ends with library name
133 int lineLength = strlen(lineBuf);
134 int libNameLength = strlen(libName);
135
136 //trim end
137 for(int i = lineLength - 1; i >= 0 && isspace(lineBuf[i]); --i)
138 {
139 lineBuf[i] = 0;
140 --lineLength;
141 }
142
143 if (0 != strncmp(lineBuf + lineLength - libNameLength, libName, libNameLength))
144 {
145 //the line does not contain the library name
146 continue;
147 }
148
149 //extract path from smaps line
150 char* pathBegin = strchr(lineBuf, '/');
151 if (0 == pathBegin)
152 {
153 LOGE("Strange error: could not find path beginning in lin \"%s\"", lineBuf);
154 continue;
155 }
156
157 char* pathEnd = strrchr(pathBegin, '/');
158 pathEnd[1] = 0;
159
160 LOGD("Libraries folder found: %s", pathBegin);
161
162 fclose(file);
163 return std::string(pathBegin) + DYNAMIC_CUDA_LIB_NAME;
164 }
165 fclose(file);
166 LOGE("Could not find library path");
167 }
168 else
169 {
170 LOGE("Could not read /proc/self/smaps");
171 }
172 }
173 else
174 {
175 LOGE("Could not get library name and base address");
176 }
177
178 return string();
179}

188static bool loadCudaSupportLib()
189{
190 void* handle;
191 const std::string name = getCudaSupportLibName();
192 dlerror();
193 handle = dlopen(name.c_str(), RTLD_LAZY);
194 if (!handle)
195 {
196 LOGE("Cannot dlopen %s: %s", name.c_str(), dlerror());
197 return false;
198 }
199
200 deviceInfoFactory = (DeviceInfoFactoryType)dlsym(handle, "deviceInfoFactory");
201 if (!deviceInfoFactory)
202 {
203 LOGE("Cannot dlsym deviceInfoFactory: %s", dlerror());
204 dlclose(handle);
205 return false;
206 }
207
208 gpuFactory = (GpuFactoryType)dlsym(handle, "gpuFactory");
209 if (!gpuFactory)
210 {
211 LOGE("Cannot dlsym gpuFactory: %s", dlerror());
212 dlclose(handle);
213 return false;
214 }
215
216 return true;
217}

320cv::gpu::GpuMat::GpuMat(int rows_, int cols_, int type_, void* data_, size_t step_) :
321 flags(Mat::MAGIC_VAL + (type_ & TYPE_MASK)), rows(rows_), cols(cols_),
322 step(step_), data((uchar*)data_), refcount(0),
323 datastart((uchar*)data_), dataend((uchar*)data_)
324{
325 size_t minstep = cols * elemSize();
326
327 if (step == Mat::AUTO_STEP)
328 {
329 step = minstep;
330 flags |= Mat::CONTINUOUS_FLAG;
331 }
332 else
333 {
334 if (rows == 1)
335 step = minstep;
336
337 CV_DbgAssert(step >= minstep);
338
339 flags |= step == minstep ? Mat::CONTINUOUS_FLAG : 0;
340 }
341 dataend += step * (rows - 1) + minstep;
342}

368cv::gpu::GpuMat::GpuMat(const GpuMat& m, Range _rowRange, Range _colRange)
369{
370 flags = m.flags;
371 step = m.step; refcount = m.refcount;
372 data = m.data; datastart = m.datastart; dataend = m.dataend;
373
374 if (_rowRange == Range::all())
375 rows = m.rows;
376 else
377 {
378 CV_Assert(0 <= _rowRange.start && _rowRange.start <= _rowRange.end && _rowRange.end <= m.rows);
379
380 rows = _rowRange.size();
381 data += step*_rowRange.start;
382 }
383
384 if (_colRange == Range::all())
385 cols = m.cols;
386 else
387 {
388 CV_Assert(0 <= _colRange.start && _colRange.start <= _colRange.end && _colRange.end <= m.cols);
389
390 cols = _colRange.size();
391 data += _colRange.start*elemSize();
392 flags &= cols < m.cols ? ~Mat::CONTINUOUS_FLAG : -1;
393 }
394
395 if (rows == 1)
396 flags |= Mat::CONTINUOUS_FLAG;
397
398 if (refcount)
399 CV_XADD(refcount, 1);
400
401 if (rows <= 0 || cols <= 0)
402 rows = cols = 0;
403}

405cv::gpu::GpuMat::GpuMat(const GpuMat& m, Rect roi) :
406 flags(m.flags), rows(roi.height), cols(roi.width),
407 step(m.step), data(m.data + roi.y*step), refcount(m.refcount),

414
415 if (refcount)
416 CV_XADD(refcount, 1);
417
418 if (rows <= 0 || cols <= 0)
419 rows = cols = 0;
420}

451void cv::gpu::GpuMat::locateROI(Size& wholeSize, Point& ofs) const
452{
453 size_t esz = elemSize();
454 ptrdiff_t delta1 = data - datastart;
455 ptrdiff_t delta2 = dataend - datastart;
456
457 CV_DbgAssert(step > 0);
458
459 if (delta1 == 0)
460 ofs.x = ofs.y = 0;
461 else
462 {
463 ofs.y = static_cast<int>(delta1 / step);
464 ofs.x = static_cast<int>((delta1 - step * ofs.y) / esz);
465
466 CV_DbgAssert(data == datastart + ofs.y * step + ofs.x * esz);
467 }
468
469 size_t minstep = (ofs.x + cols) * esz;
470
471 wholeSize.height = std::max(static_cast<int>((delta2 - minstep) / step + 1), ofs.y + rows);
472 wholeSize.width = std::max(static_cast<int>((delta2 - step * (wholeSize.height - 1)) / esz), ofs.x + cols);
473}

475GpuMat& cv::gpu::GpuMat::adjustROI(int dtop, int dbottom, int dleft, int dright)
476{
477 Size wholeSize;
478 Point ofs;
479 locateROI(wholeSize, ofs);
480
481 size_t esz = elemSize();
482
483 int row1 = std::max(ofs.y - dtop, 0);
484 int row2 = std::min(ofs.y + rows + dbottom, wholeSize.height);
485
486 int col1 = std::max(ofs.x - dleft, 0);
487 int col2 = std::min(ofs.x + cols + dright, wholeSize.width);
488
489 data += (row1 - ofs.y) * step + (col1 - ofs.x) * esz;
490 rows = row2 - row1;
491 cols = col2 - col1;
492
493 if (esz * cols == step || rows == 1)
494 flags |= Mat::CONTINUOUS_FLAG;
495 else
496 flags &= ~Mat::CONTINUOUS_FLAG;
497
498 return *this;
499}

501GpuMat cv::gpu::GpuMat::reshape(int new_cn, int new_rows) const
502{
503 GpuMat hdr = *this;
504
505 int cn = channels();
506 if (new_cn == 0)
507 new_cn = cn;
508
509 int total_width = cols * cn;
510
511 if ((new_cn > total_width || total_width % new_cn != 0) && new_rows == 0)
512 new_rows = rows * total_width / new_cn;
513
514 if (new_rows != 0 && new_rows != rows)
515 {
516 int total_size = total_width * rows;
517
518 if (!isContinuous())
519 CV_Error(CV_BadStep, "The matrix is not continuous, thus its number of rows can not be changed");
520
521 if ((unsigned)new_rows > (unsigned)total_size)
522 CV_Error(CV_StsOutOfRange, "Bad new number of rows");
523
524 total_width = total_size / new_rows;
525
526 if (total_width * new_rows != total_size)
527 CV_Error(CV_StsBadArg, "The total number of matrix elements is not divisible by the new number of rows");
528
529 hdr.rows = new_rows;
530 hdr.step = total_width * elemSize1();
531 }
532
533 int new_width = total_width / new_cn;
534
535 if (new_width * new_cn != total_width)
536 CV_Error(CV_BadNumChannels, "The total width is not divisible by the new number of channels");
537
538 hdr.cols = new_width;
539 hdr.flags = (hdr.flags & ~CV_MAT_CN_MASK) | ((new_cn - 1) << CV_CN_SHIFT);
540
541 return hdr;
542}

550{

554
555 m.cols = cols;

561void cv::gpu::ensureSizeIsEnough(int rows, int cols, int type, GpuMat& m)
562{
563 if (m.empty() || m.type() != type || m.data != m.datastart)
564 m.create(rows, cols, type);
565 else
566 {
567 const size_t esz = m.elemSize();
568 const ptrdiff_t delta2 = m.dataend - m.datastart;
569
570 const size_t minstep = m.cols * esz;
571
572 Size wholeSize;
573 wholeSize.height = std::max(static_cast<int>((delta2 - minstep) / m.step + 1), m.rows);
574 wholeSize.width = std::max(static_cast<int>((delta2 - m.step * (wholeSize.height - 1)) / esz), m.cols);
575
576 if (wholeSize.height < rows || wholeSize.width < cols)
577 m.create(rows, cols, type);
578 else
579 {
580 m.cols = cols;
581 m.rows = rows;
582 }
583 }
584}

640void cv::gpu::GpuMat::convertTo(GpuMat& dst, int rtype, double alpha, double beta) const
641{
642 bool noScale = fabs(alpha - 1) < numeric_limits<double>::epsilon() && fabs(beta) < numeric_limits<double>::epsilon();
643
644 if (rtype < 0)
645 rtype = type();
646 else
647 rtype = CV_MAKETYPE(CV_MAT_DEPTH(rtype), channels());
648
649 int sdepth = depth();
650 int ddepth = CV_MAT_DEPTH(rtype);
651 if (sdepth == ddepth && noScale)
652 {
653 copyTo(dst);
654 return;
655 }
656
657 GpuMat temp;
658 const GpuMat* psrc = this;
659 if (sdepth != ddepth && psrc == &dst)
660 {
661 temp = *this;
662 psrc = &temp;
663 }
664
665 dst.create(size(), rtype);
666
667 if (noScale)
668 cv::gpu::convertTo(*psrc, dst);
669 else
670 cv::gpu::convertTo(*psrc, dst, alpha, beta);
671}

673GpuMat& cv::gpu::GpuMat::setTo(Scalar s, const GpuMat& mask)
674{
675 CV_Assert(mask.empty() || mask.type() == CV_8UC1);
676 CV_DbgAssert(!empty());
677
678 gpu::setTo(*this, s, mask);
679
680 return *this;
681}
682
683void cv::gpu::GpuMat::create(int _rows, int _cols, int _type)
684{
685 _type &= TYPE_MASK;
686
687 if (rows == _rows && cols == _cols && type() == _type && data)
688 return;
689
690 if (data)
691 release();
692
693 CV_DbgAssert(_rows >= 0 && _cols >= 0);
694
695 if (_rows > 0 && _cols > 0)
696 {
697 flags = Mat::MAGIC_VAL + _type;
698 rows = _rows;
699 cols = _cols;
700
701 size_t esz = elemSize();
702
703 void* devPtr = NULL;
704 gpuFuncTable()->mallocPitch(&devPtr, &step, esz * cols, rows);
705
706 // Single row must be continuous
707 if (rows == 1)
708 step = esz * cols;
709
710 if (esz * cols == step)
711 flags |= Mat::CONTINUOUS_FLAG;
712
713 int64 _nettosize = static_cast<int64>(step) * rows;
714 size_t nettosize = static_cast<size_t>(_nettosize);
715
716 datastart = data = static_cast<uchar*>(devPtr);
717 dataend = data + nettosize;
718
719 refcount = static_cast<int*>(fastMalloc(sizeof(*refcount)));
720 *refcount = 1;
721 }
722}
723
724void cv::gpu::GpuMat::release()
725{
726 if (refcount && CV_XADD(refcount, -1) == 1)
727 {
728 fastFree(refcount);
729
730 gpuFuncTable()->free(datastart);
731 }
732
733 data = datastart = dataend = 0;
734 step = rows = cols = 0;
735 refcount = 0;
736}
