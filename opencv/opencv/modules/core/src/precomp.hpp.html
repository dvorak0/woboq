90template<typename T1, typename T2=T1, typename T3=T1> struct OpAdd
91{
92 typedef T1 type1;
93 typedef T2 type2;
94 typedef T3 rtype;
95 T3 operator ()(const T1 a, const T2 b) const { return saturate_cast<T3>(a + b); }
96};
97
98template<typename T1, typename T2=T1, typename T3=T1> struct OpSub
99{
100 typedef T1 type1;
101 typedef T2 type2;
102 typedef T3 rtype;
103 T3 operator ()(const T1 a, const T2 b) const { return saturate_cast<T3>(a - b); }
104};

106template<typename T1, typename T2=T1, typename T3=T1> struct OpRSub

108 typedef T1 type1;
109 typedef T2 type2;
110 typedef T3 rtype;
111 T3 operator ()(const T1 a, const T2 b) const { return saturate_cast<T3>(b - a); }

130inline Size getContinuousSize_(int flags, int cols, int rows, int widthScale)
131{
132 int64 sz = (int64)cols * rows * widthScale;
133 return (flags & Mat::CONTINUOUS_FLAG) != 0 &&
134 (int)sz == sz ? Size((int)sz, 1) : Size(cols * widthScale, rows);
135}
136
137inline Size getContinuousSize(const Mat& m1, int widthScale = 1)
138{
139 return getContinuousSize_(m1.flags,
140 m1.cols, m1.rows, widthScale);
141}
142
143inline Size getContinuousSize(const Mat& m1, const Mat& m2, int widthScale = 1)
144{
145 return getContinuousSize_(m1.flags & m2.flags,
146 m1.cols, m1.rows, widthScale);
147}

196inline bool checkScalar(const Mat& sc, int atype, int sckind, int akind)
197{
198 if( sc.dims > 2 || (sc.cols != 1 && sc.rows != 1) || !sc.isContinuous() )
199 return false;
200 int cn = CV_MAT_CN(atype);
201 if( akind == _InputArray::MATX && sckind != _InputArray::MATX )
202 return false;
203 return sc.size() == Size(1, 1) || sc.size() == Size(1, cn) || sc.size() == Size(cn, 1) ||
204 (sc.size() == Size(1, 4) && sc.type() == CV_64F && cn <= 4);
205}
