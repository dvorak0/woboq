212static inline bool isIdentity(const MatExpr& e) { return e.op == &g_MatOp_Identity; }
213static inline bool isAddEx(const MatExpr& e) { return e.op == &g_MatOp_AddEx; }
214static inline bool isScaled(const MatExpr& e) { return isAddEx(e) && (!e.b.data || e.beta == 0) && e.s == Scalar(); }
215static inline bool isBin(const MatExpr& e, char c) { return e.op == &g_MatOp_Bin && e.flags == c; }
216static inline bool isCmp(const MatExpr& e) { return e.op == &g_MatOp_Cmp; }
217static inline bool isReciprocal(const MatExpr& e) { return isBin(e,'/') && (!e.b.data || e.beta == 0); }
218static inline bool isT(const MatExpr& e) { return e.op == &g_MatOp_T; }
219static inline bool isInv(const MatExpr& e) { return e.op == &g_MatOp_Invert; }
220static inline bool isSolve(const MatExpr& e) { return e.op == &g_MatOp_Solve; }
221static inline bool isGEMM(const MatExpr& e) { return e.op == &g_MatOp_GEMM; }
222static inline bool isMatProd(const MatExpr& e) { return e.op == &g_MatOp_GEMM && (!e.c.data || e.beta == 0); }
223static inline bool isInitializer(const MatExpr& e) { return e.op == getGlobalMatOpInitializer(); }

232void MatOp::roi(const MatExpr& expr, const Range& rowRange, const Range& colRange, MatExpr& e) const
233{
234 if( elementWise(expr) )
235 {
236 e = MatExpr(expr.op, expr.flags, Mat(), Mat(), Mat(),
237 expr.alpha, expr.beta, expr.s);
238 if(expr.a.data)
239 e.a = expr.a(rowRange, colRange);
240 if(expr.b.data)
241 e.b = expr.b(rowRange, colRange);
242 if(expr.c.data)
243 e.c = expr.c(rowRange, colRange);
244 }
245 else
246 {
247 Mat m;
248 expr.op->assign(expr, m);
249 e = MatExpr(&g_MatOp_Identity, 0, m(rowRange, colRange), Mat(), Mat());
250 }
251}

253void MatOp::diag(const MatExpr& expr, int d, MatExpr& e) const
254{
255 if( elementWise(expr) )
256 {
257 e = MatExpr(expr.op, expr.flags, Mat(), Mat(), Mat(),
258 expr.alpha, expr.beta, expr.s);
259 if(expr.a.data)
260 e.a = expr.a.diag(d);
261 if(expr.b.data)
262 e.b = expr.b.diag(d);
263 if(expr.c.data)
264 e.c = expr.c.diag(d);
265 }
266 else
267 {
268 Mat m;
269 expr.op->assign(expr, m);
270 e = MatExpr(&g_MatOp_Identity, 0, m.diag(d), Mat(), Mat());
271 }
272}

331void MatOp::add(const MatExpr& e1, const MatExpr& e2, MatExpr& res) const
332{
333 if( this == e2.op )
334 {
335 double alpha = 1, beta = 1;
336 Scalar s;
337 Mat m1, m2;
338 if( isAddEx(e1) && (!e1.b.data || e1.beta == 0) )
339 {
340 m1 = e1.a;
341 alpha = e1.alpha;
342 s = e1.s;
343 }
344 else
345 e1.op->assign(e1, m1);
346
347 if( isAddEx(e2) && (!e2.b.data || e2.beta == 0) )
348 {
349 m2 = e2.a;
350 beta = e2.alpha;
351 s += e2.s;
352 }
353 else
354 e2.op->assign(e2, m2);
355 MatOp_AddEx::makeExpr(res, m1, m2, alpha, beta, s);
356 }
357 else
358 e2.op->add(e1, e2, res);
359}

370void MatOp::subtract(const MatExpr& e1, const MatExpr& e2, MatExpr& res) const
371{
372 if( this == e2.op )
373 {
374 double alpha = 1, beta = -1;
375 Scalar s;
376 Mat m1, m2;
377 if( isAddEx(e1) && (!e1.b.data || e1.beta == 0) )
378 {
379 m1 = e1.a;
380 alpha = e1.alpha;
381 s = e1.s;
382 }
383 else
384 e1.op->assign(e1, m1);
385
386 if( isAddEx(e2) && (!e2.b.data || e2.beta == 0) )
387 {
388 m2 = e2.a;
389 beta = -e2.alpha;
390 s -= e2.s;
391 }
392 else
393 e2.op->assign(e2, m2);
394 MatOp_AddEx::makeExpr(res, m1, m2, alpha, beta, s);
395 }
396 else
397 e2.op->subtract(e1, e2, res);
398}

409void MatOp::multiply(const MatExpr& e1, const MatExpr& e2, MatExpr& res, double scale) const
410{
411 if( this == e2.op )
412 {
413 Mat m1, m2;
414
415 if( isReciprocal(e1) )
416 {
417 if( isScaled(e2) )
418 {
419 scale *= e2.alpha;
420 m2 = e2.a;
421 }
422 else
423 e2.op->assign(e2, m2);
424
425 MatOp_Bin::makeExpr(res, '/', m2, e1.a, scale/e1.alpha);
426 }
427 else
428 {
429 char op = '*';
430 if( isScaled(e1) )
431 {
432 m1 = e1.a;
433 scale *= e1.alpha;
434 }
435 else
436 e1.op->assign(e1, m1);
437
438 if( isScaled(e2) )
439 {
440 m2 = e2.a;
441 scale *= e2.alpha;
442 }
443 else if( isReciprocal(e2) )
444 {
445 op = '/';
446 m2 = e2.a;
447 scale /= e2.alpha;
448 }
449 else
450 e2.op->assign(e2, m2);
451
452 MatOp_Bin::makeExpr(res, op, m1, m2, scale);
453 }
454 }
455 else
456 e2.op->multiply(e1, e2, res, scale);
457}

468void MatOp::divide(const MatExpr& e1, const MatExpr& e2, MatExpr& res, double scale) const
469{
470 if( this == e2.op )
471 {
472 if( isReciprocal(e1) && isReciprocal(e2) )
473 MatOp_Bin::makeExpr(res, '/', e2.a, e1.a, e1.alpha/e2.alpha);
474 else
475 {
476 Mat m1, m2;
477 char op = '/';
478
479 if( isScaled(e1) )
480 {
481 m1 = e1.a;
482 scale *= e1.alpha;
483 }
484 else
485 e1.op->assign(e1, m1);
486
487 if( isScaled(e2) )
488 {
489 m2 = e2.a;
490 scale /= e2.alpha;
491 }
492 else if( isReciprocal(e2) )
493 {
494 m2 = e2.a;
495 scale /= e2.alpha;
496 op = '*';
497 }
498 else
499 e2.op->assign(e2, m2);
500 MatOp_Bin::makeExpr(res, op, m1, m2, scale);
501 }
502 }
503 else
504 e2.op->divide(e1, e2, res, scale);
505}

532void MatOp::matmul(const MatExpr& e1, const MatExpr& e2, MatExpr& res) const
533{
534 if( this == e2.op )
535 {
536 double scale = 1;
537 int flags = 0;
538 Mat m1, m2;
539
540 if( isT(e1) )
541 {
542 flags = CV_GEMM_A_T;
543 scale = e1.alpha;
544 m1 = e1.a;
545 }
546 else if( isScaled(e1) )
547 {
548 scale = e1.alpha;
549 m1 = e1.a;
550 }
551 else
552 e1.op->assign(e1, m1);
553
554 if( isT(e2) )
555 {
556 flags |= CV_GEMM_B_T;
557 scale *= e2.alpha;
558 m2 = e2.a;
559 }
560 else if( isScaled(e2) )
561 {
562 scale *= e2.alpha;
563 m2 = e2.a;
564 }
565 else
566 e2.op->assign(e2, m2);
567
568 MatOp_GEMM::makeExpr(res, flags, m1, m2, scale);
569 }
570 else
571 e2.op->matmul(e1, e2, res);
572}

1205/////////////////////////////////////////////////////////////////////////////////////////////////////
1206
1207void MatOp_AddEx::assign(const MatExpr& e, Mat& m, int _type) const
1208{
1209 Mat temp, &dst = _type == -1 || e.a.type() == _type ? m : temp;
1210 if( e.b.data )
1211 {
1212 if( e.s == Scalar() || !e.s.isReal() )
1213 {
1214 if( e.alpha == 1 )
1215 {
1216 if( e.beta == 1 )
1217 cv::add(e.a, e.b, dst);
1218 else if( e.beta == -1 )
1219 cv::subtract(e.a, e.b, dst);
1220 else
1221 cv::scaleAdd(e.b, e.beta, e.a, dst);
1222 }
1223 else if( e.beta == 1 )
1224 {
1225 if( e.alpha == -1 )
1226 cv::subtract(e.b, e.a, dst);
1227 else
1228 cv::scaleAdd(e.a, e.alpha, e.b, dst);
1229 }
1230 else
1231 cv::addWeighted(e.a, e.alpha, e.b, e.beta, 0, dst);
1232
1233 if( !e.s.isReal() )
1234 cv::add(dst, e.s, dst);
1235 }
1236 else
1237 cv::addWeighted(e.a, e.alpha, e.b, e.beta, e.s[0], dst);
1238 }
1239 else if( e.s.isReal() && (dst.data != m.data || fabs(e.alpha) != 1))
1240 {
1241 e.a.convertTo(m, _type, e.alpha, e.s[0]);
1242 return;
1243 }
1244 else if( e.alpha == 1 )
1245 cv::add(e.a, e.s, dst);
1246 else if( e.alpha == -1 )
1247 cv::subtract(e.s, e.a, dst);
1248 else
1249 {
1250 e.a.convertTo(dst, e.a.type(), e.alpha);
1251 cv::add(dst, e.s, dst);
1252 }
1253
1254 if( dst.data != m.data )
1255 dst.convertTo(m, m.type());
1256}

1315
1316void MatOp_Bin::assign(const MatExpr& e, Mat& m, int _type) const
1317{
1318 Mat temp, &dst = _type == -1 || e.a.type() == _type ? m : temp;
1319
1320 if( e.flags == '*' )
1321 cv::multiply(e.a, e.b, dst, e.alpha);
1322 else if( e.flags == '/' && e.b.data )
1323 cv::divide(e.a, e.b, dst, e.alpha);
1324 else if( e.flags == '/' && !e.b.data )
1325 cv::divide(e.alpha, e.a, dst );
1326 else if( e.flags == '&' && e.b.data )
1327 bitwise_and(e.a, e.b, dst);
1328 else if( e.flags == '&' && !e.b.data )
1329 bitwise_and(e.a, e.s, dst);
1330 else if( e.flags == '|' && e.b.data )
1331 bitwise_or(e.a, e.b, dst);
1332 else if( e.flags == '|' && !e.b.data )
1333 bitwise_or(e.a, e.s, dst);
1334 else if( e.flags == '^' && e.b.data )
1335 bitwise_xor(e.a, e.b, dst);
1336 else if( e.flags == '^' && !e.b.data )
1337 bitwise_xor(e.a, e.s, dst);
1338 else if( e.flags == '~' && !e.b.data )
1339 bitwise_not(e.a, dst);
1340 else if( e.flags == 'm' )
1341 cv::min(e.a, e.b, dst);
1342 else if( e.flags == 'n' )
1343 cv::min(e.a, e.s[0], dst);
1344 else if( e.flags == 'M' )
1345 cv::max(e.a, e.b, dst);
1346 else if( e.flags == 'N' )
1347 cv::max(e.a, e.s[0], dst);
1348 else if( e.flags == 'a' && e.b.data )
1349 cv::absdiff(e.a, e.b, dst);
1350 else if( e.flags == 'a' && !e.b.data )
1351 cv::absdiff(e.a, e.s, dst);
1352 else
1353 CV_Error(CV_StsError, "Unknown operation");
1354
1355 if( dst.data != m.data )
1356 dst.convertTo(m, _type);
1357}

1455void MatOp_GEMM::add(const MatExpr& e1, const MatExpr& e2, MatExpr& res) const
1456{
1457 bool i1 = isIdentity(e1), i2 = isIdentity(e2);
1458 double alpha1 = i1 ? 1 : e1.alpha, alpha2 = i2 ? 1 : e2.alpha;
1459
1460 if( isMatProd(e1) && (i2 || isScaled(e2) || isT(e2)) )
1461 MatOp_GEMM::makeExpr(res, (e1.flags & ~CV_GEMM_C_T)|(isT(e2) ? CV_GEMM_C_T : 0),
1462 e1.a, e1.b, alpha1, e2.a, alpha2);
1463 else if( isMatProd(e2) && (i1 || isScaled(e1) || isT(e1)) )
1464 MatOp_GEMM::makeExpr(res, (e2.flags & ~CV_GEMM_C_T)|(isT(e1) ? CV_GEMM_C_T : 0),
1465 e2.a, e2.b, alpha2, e1.a, alpha1);
1466 else if( this == e2.op )
1467 MatOp::add(e1, e2, res);
1468 else
1469 e2.op->add(e1, e2, res);
1470}
1471
1472void MatOp_GEMM::subtract(const MatExpr& e1, const MatExpr& e2, MatExpr& res) const
1473{
1474 bool i1 = isIdentity(e1), i2 = isIdentity(e2);
1475 double alpha1 = i1 ? 1 : e1.alpha, alpha2 = i2 ? 1 : e2.alpha;
1476
1477 if( isMatProd(e1) && (i2 || isScaled(e2) || isT(e2)) )
1478 MatOp_GEMM::makeExpr(res, (e1.flags & ~CV_GEMM_C_T)|(isT(e2) ? CV_GEMM_C_T : 0),
1479 e1.a, e1.b, alpha1, e2.a, -alpha2);
1480 else if( isMatProd(e2) && (i1 || isScaled(e1) || isT(e1)) )
1481 MatOp_GEMM::makeExpr(res, (e2.flags & ~CV_GEMM_C_T)|(isT(e1) ? CV_GEMM_C_T : 0),
1482 e2.a, e2.b, -alpha2, e1.a, alpha1);
1483 else if( this == e2.op )

1485 else
