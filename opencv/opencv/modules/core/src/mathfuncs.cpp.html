51static const float atan2_p1 = 0.9997878412794807f*(float)(180/CV_PI);
52static const float atan2_p3 = -0.3258083974640975f*(float)(180/CV_PI);
53static const float atan2_p5 = 0.1555786518463281f*(float)(180/CV_PI);
54static const float atan2_p7 = -0.04432655554792128f*(float)(180/CV_PI);

56float fastAtan2( float y, float x )
57{
58 float ax = std::abs(x), ay = std::abs(y);
59 float a, c, c2;
60 if( ax >= ay )
61 {
62 c = ay/(ax + (float)DBL_EPSILON);
63 c2 = c*c;
64 a = (((atan2_p7*c2 + atan2_p5)*c2 + atan2_p3)*c2 + atan2_p1)*c;
65 }
66 else
67 {
68 c = ax/(ay + (float)DBL_EPSILON);
69 c2 = c*c;
70 a = 90.f - (((atan2_p7*c2 + atan2_p5)*c2 + atan2_p3)*c2 + atan2_p1)*c;
71 }
72 if( x < 0 )
73 a = 180.f - a;
74 if( y < 0 )
75 a = 360.f - a;
76 return a;
77}

79static void FastAtan2_32f(const float *Y, const float *X, float *angle, int len, bool angleInDegrees=true )
80{
81 int i = 0;
82 float scale = angleInDegrees ? 1 : (float)(CV_PI/180);
83
84#ifdef HAVE_TEGRA_OPTIMIZATION
85 if (tegra::FastAtan2_32f(Y, X, angle, len, scale))
86 return;
87#endif
88
89#if CV_SSE2
90 if( USE_SSE2 )
91 {
92 Cv32suf iabsmask; iabsmask.i = 0x7fffffff;
93 __m128 eps = _mm_set1_ps((float)DBL_EPSILON), absmask = _mm_set1_ps(iabsmask.f);
94 __m128 _90 = _mm_set1_ps(90.f), _180 = _mm_set1_ps(180.f), _360 = _mm_set1_ps(360.f);
95 __m128 z = _mm_setzero_ps(), scale4 = _mm_set1_ps(scale);
96 __m128 p1 = _mm_set1_ps(atan2_p1), p3 = _mm_set1_ps(atan2_p3);
97 __m128 p5 = _mm_set1_ps(atan2_p5), p7 = _mm_set1_ps(atan2_p7);
98
99 for( ; i <= len - 4; i += 4 )
100 {
101 __m128 x = _mm_loadu_ps(X + i), y = _mm_loadu_ps(Y + i);
102 __m128 ax = _mm_and_ps(x, absmask), ay = _mm_and_ps(y, absmask);
103 __m128 mask = _mm_cmplt_ps(ax, ay);
104 __m128 tmin = _mm_min_ps(ax, ay), tmax = _mm_max_ps(ax, ay);
105 __m128 c = _mm_div_ps(tmin, _mm_add_ps(tmax, eps));
106 __m128 c2 = _mm_mul_ps(c, c);
107 __m128 a = _mm_mul_ps(c2, p7);
108 a = _mm_mul_ps(_mm_add_ps(a, p5), c2);
109 a = _mm_mul_ps(_mm_add_ps(a, p3), c2);
110 a = _mm_mul_ps(_mm_add_ps(a, p1), c);
111
112 __m128 b = _mm_sub_ps(_90, a);
113 a = _mm_xor_ps(a, _mm_and_ps(_mm_xor_ps(a, b), mask));
114
115 b = _mm_sub_ps(_180, a);
116 mask = _mm_cmplt_ps(x, z);
117 a = _mm_xor_ps(a, _mm_and_ps(_mm_xor_ps(a, b), mask));
118
119 b = _mm_sub_ps(_360, a);
120 mask = _mm_cmplt_ps(y, z);
121 a = _mm_xor_ps(a, _mm_and_ps(_mm_xor_ps(a, b), mask));
122
123 a = _mm_mul_ps(a, scale4);
124 _mm_storeu_ps(angle + i, a);
125 }
126 }
127#endif
128
129 for( ; i < len; i++ )
130 {
131 float x = X[i], y = Y[i];
132 float ax = std::abs(x), ay = std::abs(y);
133 float a, c, c2;
134 if( ax >= ay )
135 {
136 c = ay/(ax + (float)DBL_EPSILON);
137 c2 = c*c;
138 a = (((atan2_p7*c2 + atan2_p5)*c2 + atan2_p3)*c2 + atan2_p1)*c;
139 }
140 else
141 {
142 c = ax/(ay + (float)DBL_EPSILON);
143 c2 = c*c;
144 a = 90.f - (((atan2_p7*c2 + atan2_p5)*c2 + atan2_p3)*c2 + atan2_p1)*c;
145 }
146 if( x < 0 )
147 a = 180.f - a;
148 if( y < 0 )
149 a = 360.f - a;
150 angle[i] = (float)(a*scale);
151 }
152}

155/* ************************************************************************** *\
156 Fast cube root by Ken Turkowski
157 (http://www.worldserver.com/turk/computergraphics/papers.html)
158\* ************************************************************************** */
159float cubeRoot( float value )
160{
161 float fr;
162 Cv32suf v, m;
163 int ix, s;
164 int ex, shx;
165
166 v.f = value;
167 ix = v.i & 0x7fffffff;
168 s = v.i & 0x80000000;
169 ex = (ix >> 23) - 127;
170 shx = ex % 3;
171 shx -= shx >= 0 ? 3 : 0;
172 ex = (ex - shx) / 3; /* exponent of cube root */
173 v.i = (ix & ((1<<23)-1)) | ((shx + 127)<<23);
174 fr = v.f;
175
176 /* 0.125 <= fr < 1.0 */
177 /* Use quartic rational polynomial with error < 2^(-24) */
178 fr = (float)(((((45.2548339756803022511987494 * fr +
179 192.2798368355061050458134625) * fr +
180 119.1654824285581628956914143) * fr +
181 13.43250139086239872172837314) * fr +
182 0.1636161226585754240958355063)/
183 ((((14.80884093219134573786480845 * fr +
184 151.9714051044435648658557668) * fr +
185 168.5254414101568283957668343) * fr +
186 33.9905941350215598754191872) * fr +
187 1.0));
188
189 /* fr *= 2^ex * sign */
190 m.f = value;
191 v.f = fr;
192 v.i = (v.i + (ex << 23) + s) & (m.i*2 != 0 ? -1 : 0);
193 return v.f;
194}

196static void Magnitude_32f(const float* x, const float* y, float* mag, int len)
197{
198 int i = 0;
199
200#if CV_SSE
201 if( USE_SSE2 )
202 {
203 for( ; i <= len - 8; i += 8 )
204 {
205 __m128 x0 = _mm_loadu_ps(x + i), x1 = _mm_loadu_ps(x + i + 4);
206 __m128 y0 = _mm_loadu_ps(y + i), y1 = _mm_loadu_ps(y + i + 4);
207 x0 = _mm_add_ps(_mm_mul_ps(x0, x0), _mm_mul_ps(y0, y0));
208 x1 = _mm_add_ps(_mm_mul_ps(x1, x1), _mm_mul_ps(y1, y1));
209 x0 = _mm_sqrt_ps(x0); x1 = _mm_sqrt_ps(x1);
210 _mm_storeu_ps(mag + i, x0); _mm_storeu_ps(mag + i + 4, x1);
211 }
212 }
213#endif
214
215 for( ; i < len; i++ )
216 {
217 float x0 = x[i], y0 = y[i];
218 mag[i] = std::sqrt(x0*x0 + y0*y0);
219 }
220}

249static void InvSqrt_32f(const float* src, float* dst, int len)
250{
251 int i = 0;
252
253#if CV_SSE
254 if( USE_SSE2 )
255 {
256 __m128 _0_5 = _mm_set1_ps(0.5f), _1_5 = _mm_set1_ps(1.5f);
257 if( (((size_t)src|(size_t)dst) & 15) == 0 )
258 for( ; i <= len - 8; i += 8 )
259 {
260 __m128 t0 = _mm_load_ps(src + i), t1 = _mm_load_ps(src + i + 4);
261 __m128 h0 = _mm_mul_ps(t0, _0_5), h1 = _mm_mul_ps(t1, _0_5);
262 t0 = _mm_rsqrt_ps(t0); t1 = _mm_rsqrt_ps(t1);
263 t0 = _mm_mul_ps(t0, _mm_sub_ps(_1_5, _mm_mul_ps(_mm_mul_ps(t0,t0),h0)));
264 t1 = _mm_mul_ps(t1, _mm_sub_ps(_1_5, _mm_mul_ps(_mm_mul_ps(t1,t1),h1)));
265 _mm_store_ps(dst + i, t0); _mm_store_ps(dst + i + 4, t1);
266 }
267 else
268 for( ; i <= len - 8; i += 8 )
269 {
270 __m128 t0 = _mm_loadu_ps(src + i), t1 = _mm_loadu_ps(src + i + 4);
271 __m128 h0 = _mm_mul_ps(t0, _0_5), h1 = _mm_mul_ps(t1, _0_5);
272 t0 = _mm_rsqrt_ps(t0); t1 = _mm_rsqrt_ps(t1);
273 t0 = _mm_mul_ps(t0, _mm_sub_ps(_1_5, _mm_mul_ps(_mm_mul_ps(t0,t0),h0)));
274 t1 = _mm_mul_ps(t1, _mm_sub_ps(_1_5, _mm_mul_ps(_mm_mul_ps(t1,t1),h1)));
275 _mm_storeu_ps(dst + i, t0); _mm_storeu_ps(dst + i + 4, t1);
276 }
277 }
278#endif
279
280 for( ; i < len; i++ )
281 dst[i] = 1/std::sqrt(src[i]);
282}

354void magnitude( InputArray src1, InputArray src2, OutputArray dst )
355{
356 Mat X = src1.getMat(), Y = src2.getMat();
357 int type = X.type(), depth = X.depth(), cn = X.channels();
358 CV_Assert( X.size == Y.size && type == Y.type() && (depth == CV_32F || depth == CV_64F));
359 dst.create(X.dims, X.size, X.type());
360 Mat Mag = dst.getMat();
361
362 const Mat* arrays[] = {&X, &Y, &Mag, 0};
363 uchar* ptrs[3];
364 NAryMatIterator it(arrays, ptrs);
365 int len = (int)it.size*cn;
366
367 for( size_t i = 0; i < it.nplanes; i++, ++it )
368 {
369 if( depth == CV_32F )
370 {
371 const float *x = (const float*)ptrs[0], *y = (const float*)ptrs[1];
372 float *mag = (float*)ptrs[2];
373 Magnitude_32f( x, y, mag, len );
374 }
375 else
376 {
377 const double *x = (const double*)ptrs[0], *y = (const double*)ptrs[1];
378 double *mag = (double*)ptrs[2];
379 Magnitude_64f( x, y, mag, len );
380 }
381 }

385void phase( InputArray src1, InputArray src2, OutputArray dst, bool angleInDegrees )
386{
387 Mat X = src1.getMat(), Y = src2.getMat();
388 int type = X.type(), depth = X.depth(), cn = X.channels();
389 CV_Assert( X.size == Y.size && type == Y.type() && (depth == CV_32F || depth == CV_64F));
390 dst.create( X.dims, X.size, type );
391 Mat Angle = dst.getMat();
392
393 const Mat* arrays[] = {&X, &Y, &Angle, 0};
394 uchar* ptrs[3];
395 NAryMatIterator it(arrays, ptrs);
396 cv::AutoBuffer<float> _buf;
397 float* buf[2] = {0, 0};
398 int j, k, total = (int)(it.size*cn), blockSize = total;
399 size_t esz1 = X.elemSize1();
400
401 if( depth == CV_64F )
402 {
403 blockSize = std::min(blockSize, ((BLOCK_SIZE+cn-1)/cn)*cn);
404 _buf.allocate(blockSize*2);
405 buf[0] = _buf;
406 buf[1] = buf[0] + blockSize;
407 }
408
409 for( size_t i = 0; i < it.nplanes; i++, ++it )
410 {
411 for( j = 0; j < total; j += blockSize )
412 {
413 int len = std::min(total - j, blockSize);
414 if( depth == CV_32F )
415 {
416 const float *x = (const float*)ptrs[0], *y = (const float*)ptrs[1];
417 float *angle = (float*)ptrs[2];
418 FastAtan2_32f( y, x, angle, len, angleInDegrees );
419 }
420 else
421 {
422 const double *x = (const double*)ptrs[0], *y = (const double*)ptrs[1];
423 double *angle = (double*)ptrs[2];
424 for( k = 0; k < len; k++ )
425 {
426 buf[0][k] = (float)x[k];
427 buf[1][k] = (float)y[k];
428 }
429
430 FastAtan2_32f( buf[1], buf[0], buf[0], len, angleInDegrees );
431 for( k = 0; k < len; k++ )
432 angle[k] = buf[0][k];
433 }
434 ptrs[0] += len*esz1;
435 ptrs[1] += len*esz1;
436 ptrs[2] += len*esz1;
437 }
438 }

1893{

1896

1901 bool is_ipower = false;
1902
1903 if( fabs(ipower - power) < DBL_EPSILON )
1904 {
1905 if( ipower < 0 )
1906 {
1907 divide( 1., src, dst );
1908 if( ipower == -1 )
1909 return;
1910 ipower = -ipower;
1911 src = dst;
1912 }
1913
1914 switch( ipower )
1915 {
1916 case 0:
1917 dst = Scalar::all(1);
1918 return;
1919 case 1:
1920 src.copyTo(dst);
1921 return;
1922 case 2:
1923 multiply(src, src, dst);
1924 return;
1925 default:
1926 is_ipower = true;

1928 }

1930 CV_Assert( depth == CV_32F || depth == CV_64F );
1931

2340int cv::solveCubic( InputArray _coeffs, OutputArray _roots )
2341{
2342 const int n0 = 3;
2343 Mat coeffs = _coeffs.getMat();
2344 int ctype = coeffs.type();
2345
2346 CV_Assert( ctype == CV_32F || ctype == CV_64F );
2347 CV_Assert( (coeffs.size() == Size(n0, 1) ||
2348 coeffs.size() == Size(n0+1, 1) ||
2349 coeffs.size() == Size(1, n0) ||
2350 coeffs.size() == Size(1, n0+1)) );
2351
2352 _roots.create(n0, 1, ctype, -1, true, DEPTH_MASK_FLT);
2353 Mat roots = _roots.getMat();
2354
2355 int i = -1, n = 0;
2356 double a0 = 1., a1, a2, a3;
2357 double x0 = 0., x1 = 0., x2 = 0.;
2358 int ncoeffs = coeffs.rows + coeffs.cols - 1;
2359
2360 if( ctype == CV_32FC1 )
2361 {
2362 if( ncoeffs == 4 )
2363 a0 = coeffs.at<float>(++i);
2364
2365 a1 = coeffs.at<float>(i+1);
2366 a2 = coeffs.at<float>(i+2);
2367 a3 = coeffs.at<float>(i+3);
2368 }
2369 else
2370 {
2371 if( ncoeffs == 4 )
2372 a0 = coeffs.at<double>(++i);
2373
2374 a1 = coeffs.at<double>(i+1);
2375 a2 = coeffs.at<double>(i+2);
2376 a3 = coeffs.at<double>(i+3);
2377 }
2378
2379 if( a0 == 0 )
2380 {
2381 if( a1 == 0 )
2382 {
2383 if( a2 == 0 )
2384 n = a3 == 0 ? -1 : 0;
2385 else
2386 {
2387 // linear equation
2388 x0 = -a3/a2;
2389 n = 1;
2390 }
2391 }
2392 else
2393 {
2394 // quadratic equation
2395 double d = a2*a2 - 4*a1*a3;
2396 if( d >= 0 )
2397 {
2398 d = sqrt(d);
2399 double q1 = (-a2 + d) * 0.5;
2400 double q2 = (a2 + d) * -0.5;
2401 if( fabs(q1) > fabs(q2) )
2402 {
2403 x0 = q1 / a1;
2404 x1 = a3 / q1;
2405 }
2406 else
2407 {
2408 x0 = q2 / a1;
2409 x1 = a3 / q2;
2410 }
2411 n = d > 0 ? 2 : 1;
2412 }
2413 }
2414 }
2415 else
2416 {
2417 a0 = 1./a0;
2418 a1 *= a0;
2419 a2 *= a0;
2420 a3 *= a0;
2421
2422 double Q = (a1 * a1 - 3 * a2) * (1./9);
2423 double R = (2 * a1 * a1 * a1 - 9 * a1 * a2 + 27 * a3) * (1./54);
2424 double Qcubed = Q * Q * Q;
2425 double d = Qcubed - R * R;
2426
2427 if( d > 0 )
2428 {
2429 double theta = acos(R / sqrt(Qcubed));
2430 double sqrtQ = sqrt(Q);
2431 double t0 = -2 * sqrtQ;
2432 double t1 = theta * (1./3);
2433 double t2 = a1 * (1./3);
2434 x0 = t0 * cos(t1) - t2;
2435 x1 = t0 * cos(t1 + (2.*CV_PI/3)) - t2;
2436 x2 = t0 * cos(t1 + (4.*CV_PI/3)) - t2;
2437 n = 3;
2438 }
2439 else if( d == 0 )
2440 {
2441 if(R >= 0)
2442 {
2443 x0 = -2*pow(R, 1./3) - a1/3;
2444 x1 = pow(R, 1./3) - a1/3;
2445 }
2446 else
2447 {
2448 x0 = 2*pow(-R, 1./3) - a1/3;
2449 x1 = -pow(-R, 1./3) - a1/3;
2450 }
2451 x2 = 0;
2452 n = x0 == x1 ? 1 : 2;
2453 x1 = x0 == x1 ? 0 : x1;
2454 }
2455 else
2456 {
2457 double e;
2458 d = sqrt(-d);
2459 e = pow(d + fabs(R), 1./3);
2460 if( R > 0 )
2461 e = -e;
2462 x0 = (e + Q / e) - a1 * (1./3);
2463 n = 1;
2464 }
2465 }
2466
2467 if( roots.type() == CV_32FC1 )
2468 {
2469 roots.at<float>(0) = (float)x0;
2470 roots.at<float>(1) = (float)x1;
2471 roots.at<float>(2) = (float)x2;
2472 }
2473 else
2474 {
2475 roots.at<double>(0) = x0;
2476 roots.at<double>(1) = x1;
2477 roots.at<double>(2) = x2;
2478 }
2479
2480 return n;
2481}

2483/* finds complex roots of a polynomial using Durand-Kerner method:
2484 http://en.wikipedia.org/wiki/Durand%E2%80%93Kerner_method */
2485double cv::solvePoly( InputArray _coeffs0, OutputArray _roots0, int maxIters )
2486{
2487 typedef Complex<double> C;
2488
2489 double maxDiff = 0;
2490 int iter, i, j;
2491 Mat coeffs0 = _coeffs0.getMat();
2492 int ctype = _coeffs0.type();
2493 int cdepth = CV_MAT_DEPTH(ctype);
2494
2495 CV_Assert( CV_MAT_DEPTH(ctype) >= CV_32F && CV_MAT_CN(ctype) <= 2 );
2496 CV_Assert( coeffs0.rows == 1 || coeffs0.cols == 1 );
2497
2498 int n = coeffs0.cols + coeffs0.rows - 2;
2499
2500 _roots0.create(n, 1, CV_MAKETYPE(cdepth, 2), -1, true, DEPTH_MASK_FLT);
2501 Mat roots0 = _roots0.getMat();
2502
2503 AutoBuffer<C> buf(n*2+2);
2504 C *coeffs = buf, *roots = coeffs + n + 1;
2505 Mat coeffs1(coeffs0.size(), CV_MAKETYPE(CV_64F, coeffs0.channels()), coeffs0.channels() == 2 ? coeffs : roots);
2506 coeffs0.convertTo(coeffs1, coeffs1.type());
2507 if( coeffs0.channels() == 1 )
2508 {
2509 const double* rcoeffs = (const double*)roots;
2510 for( i = 0; i <= n; i++ )
2511 coeffs[i] = C(rcoeffs[i], 0);
2512 }
2513
2514 C p(1, 0), r(1, 1);
2515 for( i = 0; i < n; i++ )
2516 {
2517 roots[i] = p;
2518 p = p * r;
2519 }
2520
2521 maxIters = maxIters <= 0 ? 1000 : maxIters;
2522 for( iter = 0; iter < maxIters; iter++ )
2523 {
2524 maxDiff = 0;
2525 for( i = 0; i < n; i++ )
2526 {
2527 p = roots[i];
2528 C num = coeffs[n], denom = coeffs[n];
2529 int num_same_root = 1;
2530 for( j = 0; j < n; j++ )
2531 {
2532 num = num*p + coeffs[n-j-1];
2533 if( j != i )
2534 {
2535 if ( (p - roots[j]).re != 0 || (p - roots[j]).im != 0 )
2536 denom = denom * (p - roots[j]);
2537 else
2538 num_same_root++;
2539 }
2540 }
2541 num /= denom;
2542 if( num_same_root > 1)
2543 {
2544 double old_num_re = num.re;
2545 double old_num_im = num.im;
2546 int square_root_times = num_same_root % 2 == 0 ? num_same_root / 2 : num_same_root / 2 - 1;
2547
2548 for( j = 0; j < square_root_times; j++)
2549 {
2550 num.re = old_num_re*old_num_re + old_num_im*old_num_im;
2551 num.re = sqrt(num.re);
2552 num.re += old_num_re;
2553 num.im = num.re - old_num_re;
2554 num.re /= 2;
2555 num.re = sqrt(num.re);
2556
2557 num.im /= 2;
2558 num.im = sqrt(num.im);
2559 if( old_num_re < 0 ) num.im = -num.im;
2560 }
2561
2562 if( num_same_root % 2 != 0){
2563 Mat cube_coefs(4, 1, CV_64FC1);
2564 Mat cube_roots(3, 1, CV_64FC2);
2565 cube_coefs.at<double>(3) = -(pow(old_num_re, 3));
2566 cube_coefs.at<double>(2) = -(15*pow(old_num_re, 2) + 27*pow(old_num_im, 2));
2567 cube_coefs.at<double>(1) = -48*old_num_re;
2568 cube_coefs.at<double>(0) = 64;
2569 solveCubic(cube_coefs, cube_roots);
2570
2571 if(cube_roots.at<double>(0) >= 0) num.re = pow(cube_roots.at<double>(0), 1./3);
2572 else num.re = -pow(-cube_roots.at<double>(0), 1./3);
2573 num.im = sqrt(pow(num.re, 2) / 3 - old_num_re / (3*num.re));
2574 }
2575 }
2576
2577 roots[i] = p - num;
2578 maxDiff = max(maxDiff, abs(num));
2579 }
2580 if( maxDiff <= 0 )
2581 break;
2582 }
2583
2584 if( coeffs0.channels() == 1 )
2585 {
2586 const double verySmallEps = 1e-100;
2587 for( i = 0; i < n; i++ )
2588 if( fabs(roots[i].im) < verySmallEps )
2589 roots[i].im = 0;
2590 }
2591
2592 Mat(roots0.size(), CV_64FC2, roots).convertTo(roots0, roots0.type());
2593 return maxDiff;
2594}
