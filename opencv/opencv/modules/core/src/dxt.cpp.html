122#define BitRev(i,shift) \
123 ((int)((((unsigned)bitrevTab[(i)&255] << 24)+ \
124 ((unsigned)bitrevTab[((i)>> 8)&255] << 16)+ \
125 ((unsigned)bitrevTab[((i)>>16)&255] << 8)+ \

128static int
129DFTFactorize( int n, int* factors )
130{
131 int nf = 0, f, i, j;
132
133 if( n <= 5 )
134 {
135 factors[0] = n;
136 return 1;
137 }
138
139 f = (((n - 1)^n)+1) >> 1;
140 if( f > 1 )
141 {
142 factors[nf++] = f;
143 n = f == n ? 1 : n/f;
144 }
145
146 for( f = 3; n > 1; )
147 {
148 int d = n/f;
149 if( d*f == n )
150 {
151 factors[nf++] = f;
152 n = d;
153 }
154 else
155 {
156 f += 2;
157 if( f*f > n )
158 break;
159 }
160 }
161
162 if( n > 1 )
163 factors[nf++] = n;
164
165 f = (factors[0] & 1) == 0;
166 for( i = f; i < (nf+f)/2; i++ )
167 CV_SWAP( factors[i], factors[nf-i-1+f], j );
168
169 return nf;
170}

172static void
173DFTInit( int n0, int nf, int* factors, int* itab, int elem_size, void* _wave, int inv_itab )
174{
175 int digits[34], radix[34];
176 int n = factors[0], m = 0;
177 int* itab0 = itab;
178 int i, j, k;
179 Complex<double> w, w1;
180 double t;
181
182 if( n0 <= 5 )
183 {
184 itab[0] = 0;
185 itab[n0-1] = n0-1;
186
187 if( n0 != 4 )
188 {
189 for( i = 1; i < n0-1; i++ )
190 itab[i] = i;
191 }
192 else
193 {
194 itab[1] = 2;
195 itab[2] = 1;
196 }
197 if( n0 == 5 )
198 {
199 if( elem_size == sizeof(Complex<double>) )
200 ((Complex<double>*)_wave)[0] = Complex<double>(1.,0.);
201 else
202 ((Complex<float>*)_wave)[0] = Complex<float>(1.f,0.f);
203 }
204 if( n0 != 4 )
205 return;
206 m = 2;
207 }
208 else
209 {
210 // radix[] is initialized from index 'nf' down to zero
211 assert (nf < 34);
212 radix[nf] = 1;
213 digits[nf] = 0;
214 for( i = 0; i < nf; i++ )
215 {
216 digits[i] = 0;
217 radix[nf-i-1] = radix[nf-i]*factors[nf-i-1];
218 }
219
220 if( inv_itab && factors[0] != factors[nf-1] )
221 itab = (int*)_wave;
222
223 if( (n & 1) == 0 )
224 {
225 int a = radix[1], na2 = n*a>>1, na4 = na2 >> 1;
226 for( m = 0; (unsigned)(1 << m) < (unsigned)n; m++ )
227 ;
228 if( n <= 2 )
229 {
230 itab[0] = 0;
231 itab[1] = na2;
232 }
233 else if( n <= 256 )
234 {
235 int shift = 10 - m;
236 for( i = 0; i <= n - 4; i += 4 )
237 {
238 j = (bitrevTab[i>>2]>>shift)*a;
239 itab[i] = j;
240 itab[i+1] = j + na2;
241 itab[i+2] = j + na4;
242 itab[i+3] = j + na2 + na4;
243 }
244 }
245 else
246 {
247 int shift = 34 - m;
248 for( i = 0; i < n; i += 4 )
249 {
250 int i4 = i >> 2;
251 j = BitRev(i4,shift)*a;
252 itab[i] = j;
253 itab[i+1] = j + na2;
254 itab[i+2] = j + na4;
255 itab[i+3] = j + na2 + na4;
256 }
257 }
258
259 digits[1]++;
260
261 if( nf >= 2 )
262 {
263 for( i = n, j = radix[2]; i < n0; )
264 {
265 for( k = 0; k < n; k++ )
266 itab[i+k] = itab[k] + j;
267 if( (i += n) >= n0 )
268 break;
269 j += radix[2];
270 for( k = 1; ++digits[k] >= factors[k]; k++ )
271 {
272 digits[k] = 0;
273 j += radix[k+2] - radix[k];
274 }
275 }
276 }
277 }
278 else
279 {
280 for( i = 0, j = 0;; )
281 {
282 itab[i] = j;
283 if( ++i >= n0 )
284 break;
285 j += radix[1];
286 for( k = 0; ++digits[k] >= factors[k]; k++ )
287 {
288 digits[k] = 0;
289 j += radix[k+2] - radix[k];
290 }
291 }
292 }
293
294 if( itab != itab0 )
295 {
296 itab0[0] = 0;
297 for( i = n0 & 1; i < n0; i += 2 )
298 {
299 int k0 = itab[i];
300 int k1 = itab[i+1];
301 itab0[k0] = i;
302 itab0[k1] = i+1;
303 }
304 }
305 }
306
307 if( (n0 & (n0-1)) == 0 )
308 {
309 w.re = w1.re = DFTTab[m][0];
310 w.im = w1.im = -DFTTab[m][1];
311 }
312 else
313 {
314 t = -CV_PI*2/n0;
315 w.im = w1.im = sin(t);
316 w.re = w1.re = std::sqrt(1. - w1.im*w1.im);
317 }
318 n = (n0+1)/2;
319
320 if( elem_size == sizeof(Complex<double>) )
321 {
322 Complex<double>* wave = (Complex<double>*)_wave;
323
324 wave[0].re = 1.;
325 wave[0].im = 0.;
326
327 if( (n0 & 1) == 0 )
328 {
329 wave[n].re = -1.;
330 wave[n].im = 0;
331 }
332
333 for( i = 1; i < n; i++ )
334 {
335 wave[i] = w;
336 wave[n0-i].re = w.re;
337 wave[n0-i].im = -w.im;
338
339 t = w.re*w1.re - w.im*w1.im;
340 w.im = w.re*w1.im + w.im*w1.re;
341 w.re = t;
342 }
343 }
344 else
345 {
346 Complex<float>* wave = (Complex<float>*)_wave;
347 assert( elem_size == sizeof(Complex<float>) );
348
349 wave[0].re = 1.f;
350 wave[0].im = 0.f;
351
352 if( (n0 & 1) == 0 )
353 {
354 wave[n].re = -1.f;
355 wave[n].im = 0.f;
356 }
357
358 for( i = 1; i < n; i++ )
359 {
360 wave[i].re = (float)w.re;
361 wave[i].im = (float)w.im;
362 wave[n0-i].re = (float)w.re;
363 wave[n0-i].im = (float)-w.im;
364
365 t = w.re*w1.re - w.im*w1.im;
366 w.im = w.re*w1.im + w.im*w1.re;
367 w.re = t;
368 }
369 }
370}

384 int n = 1, i, j, nx, dw, dw0 = _dw0;
385 __m128 z = _mm_setzero_ps(), x02=z, x13=z, w01=z, w23=z, y01, y23, t0, t1;
386 Cv32suf t; t.i = 0x80000000;
387 __m128 neg0_mask = _mm_load_ss(&t.f);
388 __m128 neg3_mask = _mm_shuffle_ps(neg0_mask, neg0_mask, _MM_SHUFFLE(0,1,2,3));
389
390 for( ; n*4 <= N; )
391 {
392 nx = n;
393 n *= 4;
394 dw0 /= 4;
395
396 for( i = 0; i < n0; i += n )
397 {
398 Complexf *v0, *v1;
399
400 v0 = dst + i;
401 v1 = v0 + nx*2;
402
403 x02 = _mm_loadl_pi(x02, (const __m64*)&v0[0]);
404 x13 = _mm_loadl_pi(x13, (const __m64*)&v0[nx]);
405 x02 = _mm_loadh_pi(x02, (const __m64*)&v1[0]);
406 x13 = _mm_loadh_pi(x13, (const __m64*)&v1[nx]);
407
408 y01 = _mm_add_ps(x02, x13);
409 y23 = _mm_sub_ps(x02, x13);
410 t1 = _mm_xor_ps(_mm_shuffle_ps(y01, y23, _MM_SHUFFLE(2,3,3,2)), neg3_mask);
411 t0 = _mm_movelh_ps(y01, y23);
412 y01 = _mm_add_ps(t0, t1);
413 y23 = _mm_sub_ps(t0, t1);
414
415 _mm_storel_pi((__m64*)&v0[0], y01);
416 _mm_storeh_pi((__m64*)&v0[nx], y01);
417 _mm_storel_pi((__m64*)&v1[0], y23);
418 _mm_storeh_pi((__m64*)&v1[nx], y23);
419
420 for( j = 1, dw = dw0; j < nx; j++, dw += dw0 )
421 {
422 v0 = dst + i + j;
423 v1 = v0 + nx*2;
424
425 x13 = _mm_loadl_pi(x13, (const __m64*)&v0[nx]);
426 w23 = _mm_loadl_pi(w23, (const __m64*)&wave[dw*2]);
427 x13 = _mm_loadh_pi(x13, (const __m64*)&v1[nx]); // x1, x3 = r1 i1 r3 i3
428 w23 = _mm_loadh_pi(w23, (const __m64*)&wave[dw*3]); // w2, w3 = wr2 wi2 wr3 wi3
429
430 t0 = _mm_mul_ps(_mm_moveldup_ps(x13), w23);
431 t1 = _mm_mul_ps(_mm_movehdup_ps(x13), _mm_shuffle_ps(w23, w23, _MM_SHUFFLE(2,3,0,1)));
432 x13 = _mm_addsub_ps(t0, t1);
433 // re(x1*w2), im(x1*w2), re(x3*w3), im(x3*w3)
434 x02 = _mm_loadl_pi(x02, (const __m64*)&v1[0]); // x2 = r2 i2
435 w01 = _mm_loadl_pi(w01, (const __m64*)&wave[dw]); // w1 = wr1 wi1
436 x02 = _mm_shuffle_ps(x02, x02, _MM_SHUFFLE(0,0,1,1));
437 w01 = _mm_shuffle_ps(w01, w01, _MM_SHUFFLE(1,0,0,1));
438 x02 = _mm_mul_ps(x02, w01);
439 x02 = _mm_addsub_ps(x02, _mm_movelh_ps(x02, x02));
440 // re(x0) im(x0) re(x2*w1), im(x2*w1)
441 x02 = _mm_loadl_pi(x02, (const __m64*)&v0[0]);
442
443 y01 = _mm_add_ps(x02, x13);
444 y23 = _mm_sub_ps(x02, x13);
445 t1 = _mm_xor_ps(_mm_shuffle_ps(y01, y23, _MM_SHUFFLE(2,3,3,2)), neg3_mask);
446 t0 = _mm_movelh_ps(y01, y23);
447 y01 = _mm_add_ps(t0, t1);
448 y23 = _mm_sub_ps(t0, t1);
449
450 _mm_storel_pi((__m64*)&v0[0], y01);
451 _mm_storeh_pi((__m64*)&v0[nx], y01);

453 _mm_storeh_pi((__m64*)&v1[nx], y23);

455 }
456 }

460 }

521// mixed-radix complex discrete Fourier transform: double-precision version
522template<typename T> static void
523DFT( const Complex<T>* src, Complex<T>* dst, int n,
524 int nf, const int* factors, const int* itab,
525 const Complex<T>* wave, int tab_size,
526 const void*
527#ifdef USE_IPP_DFT
528 spec
529#endif
530 , Complex<T>* buf,
531 int flags, double _scale )
532{
533 static const T sin_120 = (T)0.86602540378443864676372317075294;
534 static const T fft5_2 = (T)0.559016994374947424102293417182819;
535 static const T fft5_3 = (T)-0.951056516295153572116439333379382;
536 static const T fft5_4 = (T)-1.538841768587626701285145288018455;
537 static const T fft5_5 = (T)0.363271264002680442947733378740309;
538
539 int n0 = n, f_idx, nx;
540 int inv = flags & DFT_INVERSE;
541 int dw0 = tab_size, dw;
542 int i, j, k;
543 Complex<T> t;
544 T scale = (T)_scale;
545 int tab_step;
546
547#ifdef USE_IPP_DFT
548 if( spec )
549 {
550 if( !inv )
551 ippsDFTFwd_CToC( src, dst, spec, (uchar*)buf );
552 else
553 ippsDFTInv_CToC( src, dst, spec, (uchar*)buf );
554 return;
555 }
556#endif
557
558 tab_step = tab_size == n ? 1 : tab_size == n*2 ? 2 : tab_size/n;
559
560 // 0. shuffle data
561 if( dst != src )
562 {
563 assert( (flags & DFT_NO_PERMUTE) == 0 );
564 if( !inv )
565 {
566 for( i = 0; i <= n - 2; i += 2, itab += 2*tab_step )
567 {
568 int k0 = itab[0], k1 = itab[tab_step];
569 assert( (unsigned)k0 < (unsigned)n && (unsigned)k1 < (unsigned)n );
570 dst[i] = src[k0]; dst[i+1] = src[k1];
571 }
572
573 if( i < n )
574 dst[n-1] = src[n-1];
575 }
576 else
577 {
578 for( i = 0; i <= n - 2; i += 2, itab += 2*tab_step )
579 {
580 int k0 = itab[0], k1 = itab[tab_step];
581 assert( (unsigned)k0 < (unsigned)n && (unsigned)k1 < (unsigned)n );
582 t.re = src[k0].re; t.im = -src[k0].im;
583 dst[i] = t;
584 t.re = src[k1].re; t.im = -src[k1].im;
585 dst[i+1] = t;
586 }
587
588 if( i < n )
589 {
590 t.re = src[n-1].re; t.im = -src[n-1].im;
591 dst[i] = t;
592 }
593 }
594 }
595 else
596 {
597 if( (flags & DFT_NO_PERMUTE) == 0 )
598 {
599 CV_Assert( factors[0] == factors[nf-1] );
600 if( nf == 1 )
601 {
602 if( (n & 3) == 0 )
603 {
604 int n2 = n/2;
605 Complex<T>* dsth = dst + n2;
606
607 for( i = 0; i < n2; i += 2, itab += tab_step*2 )
608 {
609 j = itab[0];
610 assert( (unsigned)j < (unsigned)n2 );
611
612 CV_SWAP(dst[i+1], dsth[j], t);
613 if( j > i )
614 {
615 CV_SWAP(dst[i], dst[j], t);
616 CV_SWAP(dsth[i+1], dsth[j+1], t);
617 }
618 }
619 }
620 // else do nothing
621 }
622 else
623 {
624 for( i = 0; i < n; i++, itab += tab_step )
625 {
626 j = itab[0];
627 assert( (unsigned)j < (unsigned)n );
628 if( j > i )
629 CV_SWAP(dst[i], dst[j], t);
630 }
631 }
632 }
633
634 if( inv )
635 {
636 for( i = 0; i <= n - 2; i += 2 )
637 {
638 T t0 = -dst[i].im;
639 T t1 = -dst[i+1].im;
640 dst[i].im = t0; dst[i+1].im = t1;
641 }
642
643 if( i < n )
644 dst[n-1].im = -dst[n-1].im;
645 }
646 }
647
648 n = 1;
649 // 1. power-2 transforms
650 if( (factors[0] & 1) == 0 )
651 {
652 if( factors[0] >= 4 && checkHardwareSupport(CV_CPU_SSE3))
653 {
654 DFT_VecR4<T> vr4;
655 n = vr4(dst, factors[0], n0, dw0, wave);
656 }
657
658 // radix-4 transform
659 for( ; n*4 <= factors[0]; )
660 {
661 nx = n;
662 n *= 4;
663 dw0 /= 4;
664
665 for( i = 0; i < n0; i += n )
666 {
667 Complex<T> *v0, *v1;
668 T r0, i0, r1, i1, r2, i2, r3, i3, r4, i4;
669
670 v0 = dst + i;
671 v1 = v0 + nx*2;
672
673 r0 = v1[0].re; i0 = v1[0].im;
674 r4 = v1[nx].re; i4 = v1[nx].im;
675
676 r1 = r0 + r4; i1 = i0 + i4;
677 r3 = i0 - i4; i3 = r4 - r0;
678
679 r2 = v0[0].re; i2 = v0[0].im;
680 r4 = v0[nx].re; i4 = v0[nx].im;
681
682 r0 = r2 + r4; i0 = i2 + i4;
683 r2 -= r4; i2 -= i4;
684
685 v0[0].re = r0 + r1; v0[0].im = i0 + i1;
686 v1[0].re = r0 - r1; v1[0].im = i0 - i1;
687 v0[nx].re = r2 + r3; v0[nx].im = i2 + i3;
688 v1[nx].re = r2 - r3; v1[nx].im = i2 - i3;
689
690 for( j = 1, dw = dw0; j < nx; j++, dw += dw0 )
691 {
692 v0 = dst + i + j;
693 v1 = v0 + nx*2;
694
695 r2 = v0[nx].re*wave[dw*2].re - v0[nx].im*wave[dw*2].im;
696 i2 = v0[nx].re*wave[dw*2].im + v0[nx].im*wave[dw*2].re;
697 r0 = v1[0].re*wave[dw].im + v1[0].im*wave[dw].re;
698 i0 = v1[0].re*wave[dw].re - v1[0].im*wave[dw].im;
699 r3 = v1[nx].re*wave[dw*3].im + v1[nx].im*wave[dw*3].re;
700 i3 = v1[nx].re*wave[dw*3].re - v1[nx].im*wave[dw*3].im;
701
702 r1 = i0 + i3; i1 = r0 + r3;
703 r3 = r0 - r3; i3 = i3 - i0;
704 r4 = v0[0].re; i4 = v0[0].im;
705
706 r0 = r4 + r2; i0 = i4 + i2;
707 r2 = r4 - r2; i2 = i4 - i2;
708
709 v0[0].re = r0 + r1; v0[0].im = i0 + i1;
710 v1[0].re = r0 - r1; v1[0].im = i0 - i1;
711 v0[nx].re = r2 + r3; v0[nx].im = i2 + i3;
712 v1[nx].re = r2 - r3; v1[nx].im = i2 - i3;
713 }
714 }
715 }
716
717 for( ; n < factors[0]; )
718 {
719 // do the remaining radix-2 transform
720 nx = n;
721 n *= 2;
722 dw0 /= 2;
723
724 for( i = 0; i < n0; i += n )
725 {
726 Complex<T>* v = dst + i;
727 T r0 = v[0].re + v[nx].re;
728 T i0 = v[0].im + v[nx].im;
729 T r1 = v[0].re - v[nx].re;
730 T i1 = v[0].im - v[nx].im;
731 v[0].re = r0; v[0].im = i0;
732 v[nx].re = r1; v[nx].im = i1;
733
734 for( j = 1, dw = dw0; j < nx; j++, dw += dw0 )
735 {
736 v = dst + i + j;
737 r1 = v[nx].re*wave[dw].re - v[nx].im*wave[dw].im;
738 i1 = v[nx].im*wave[dw].re + v[nx].re*wave[dw].im;
739 r0 = v[0].re; i0 = v[0].im;
740
741 v[0].re = r0 + r1; v[0].im = i0 + i1;
742 v[nx].re = r0 - r1; v[nx].im = i0 - i1;
743 }
744 }
745 }
746 }
747
748 // 2. all the other transforms
749 for( f_idx = (factors[0]&1) ? 0 : 1; f_idx < nf; f_idx++ )
750 {
751 int factor = factors[f_idx];
752 nx = n;
753 n *= factor;
754 dw0 /= factor;
755
756 if( factor == 3 )
757 {
758 // radix-3
759 for( i = 0; i < n0; i += n )
760 {
761 Complex<T>* v = dst + i;
762
763 T r1 = v[nx].re + v[nx*2].re;
764 T i1 = v[nx].im + v[nx*2].im;
765 T r0 = v[0].re;
766 T i0 = v[0].im;
767 T r2 = sin_120*(v[nx].im - v[nx*2].im);
768 T i2 = sin_120*(v[nx*2].re - v[nx].re);
769 v[0].re = r0 + r1; v[0].im = i0 + i1;
770 r0 -= (T)0.5*r1; i0 -= (T)0.5*i1;
771 v[nx].re = r0 + r2; v[nx].im = i0 + i2;
772 v[nx*2].re = r0 - r2; v[nx*2].im = i0 - i2;
773
774 for( j = 1, dw = dw0; j < nx; j++, dw += dw0 )
775 {
776 v = dst + i + j;
777 r0 = v[nx].re*wave[dw].re - v[nx].im*wave[dw].im;
778 i0 = v[nx].re*wave[dw].im + v[nx].im*wave[dw].re;
779 i2 = v[nx*2].re*wave[dw*2].re - v[nx*2].im*wave[dw*2].im;
780 r2 = v[nx*2].re*wave[dw*2].im + v[nx*2].im*wave[dw*2].re;
781 r1 = r0 + i2; i1 = i0 + r2;
782
783 r2 = sin_120*(i0 - r2); i2 = sin_120*(i2 - r0);
784 r0 = v[0].re; i0 = v[0].im;
785 v[0].re = r0 + r1; v[0].im = i0 + i1;
786 r0 -= (T)0.5*r1; i0 -= (T)0.5*i1;
787 v[nx].re = r0 + r2; v[nx].im = i0 + i2;
788 v[nx*2].re = r0 - r2; v[nx*2].im = i0 - i2;
789 }
790 }
791 }
792 else if( factor == 5 )
793 {
794 // radix-5
795 for( i = 0; i < n0; i += n )
796 {
797 for( j = 0, dw = 0; j < nx; j++, dw += dw0 )
798 {
799 Complex<T>* v0 = dst + i + j;
800 Complex<T>* v1 = v0 + nx*2;
801 Complex<T>* v2 = v1 + nx*2;
802
803 T r0, i0, r1, i1, r2, i2, r3, i3, r4, i4, r5, i5;
804
805 r3 = v0[nx].re*wave[dw].re - v0[nx].im*wave[dw].im;
806 i3 = v0[nx].re*wave[dw].im + v0[nx].im*wave[dw].re;
807 r2 = v2[0].re*wave[dw*4].re - v2[0].im*wave[dw*4].im;
808 i2 = v2[0].re*wave[dw*4].im + v2[0].im*wave[dw*4].re;
809
810 r1 = r3 + r2; i1 = i3 + i2;
811 r3 -= r2; i3 -= i2;
812
813 r4 = v1[nx].re*wave[dw*3].re - v1[nx].im*wave[dw*3].im;
814 i4 = v1[nx].re*wave[dw*3].im + v1[nx].im*wave[dw*3].re;
815 r0 = v1[0].re*wave[dw*2].re - v1[0].im*wave[dw*2].im;
816 i0 = v1[0].re*wave[dw*2].im + v1[0].im*wave[dw*2].re;
817
818 r2 = r4 + r0; i2 = i4 + i0;
819 r4 -= r0; i4 -= i0;
820
821 r0 = v0[0].re; i0 = v0[0].im;
822 r5 = r1 + r2; i5 = i1 + i2;
823
824 v0[0].re = r0 + r5; v0[0].im = i0 + i5;
825
826 r0 -= (T)0.25*r5; i0 -= (T)0.25*i5;
827 r1 = fft5_2*(r1 - r2); i1 = fft5_2*(i1 - i2);
828 r2 = -fft5_3*(i3 + i4); i2 = fft5_3*(r3 + r4);
829
830 i3 *= -fft5_5; r3 *= fft5_5;
831 i4 *= -fft5_4; r4 *= fft5_4;
832
833 r5 = r2 + i3; i5 = i2 + r3;
834 r2 -= i4; i2 -= r4;
835
836 r3 = r0 + r1; i3 = i0 + i1;
837 r0 -= r1; i0 -= i1;
838
839 v0[nx].re = r3 + r2; v0[nx].im = i3 + i2;
840 v2[0].re = r3 - r2; v2[0].im = i3 - i2;
841
842 v1[0].re = r0 + r5; v1[0].im = i0 + i5;
843 v1[nx].re = r0 - r5; v1[nx].im = i0 - i5;
844 }
845 }
846 }
847 else
848 {
849 // radix-"factor" - an odd number
850 int p, q, factor2 = (factor - 1)/2;
851 int d, dd, dw_f = tab_size/factor;
852 Complex<T>* a = buf;
853 Complex<T>* b = buf + factor2;
854
855 for( i = 0; i < n0; i += n )
856 {
857 for( j = 0, dw = 0; j < nx; j++, dw += dw0 )
858 {
859 Complex<T>* v = dst + i + j;
860 Complex<T> v_0 = v[0];
861 Complex<T> vn_0 = v_0;
862
863 if( j == 0 )
864 {
865 for( p = 1, k = nx; p <= factor2; p++, k += nx )
866 {
867 T r0 = v[k].re + v[n-k].re;
868 T i0 = v[k].im - v[n-k].im;
869 T r1 = v[k].re - v[n-k].re;
870 T i1 = v[k].im + v[n-k].im;
871
872 vn_0.re += r0; vn_0.im += i1;
873 a[p-1].re = r0; a[p-1].im = i0;
874 b[p-1].re = r1; b[p-1].im = i1;
875 }
876 }
877 else
878 {
879 const Complex<T>* wave_ = wave + dw*factor;
880 d = dw;
881
882 for( p = 1, k = nx; p <= factor2; p++, k += nx, d += dw )
883 {
884 T r2 = v[k].re*wave[d].re - v[k].im*wave[d].im;
885 T i2 = v[k].re*wave[d].im + v[k].im*wave[d].re;
886
887 T r1 = v[n-k].re*wave_[-d].re - v[n-k].im*wave_[-d].im;
888 T i1 = v[n-k].re*wave_[-d].im + v[n-k].im*wave_[-d].re;
889
890 T r0 = r2 + r1;
891 T i0 = i2 - i1;
892 r1 = r2 - r1;
893 i1 = i2 + i1;
894
895 vn_0.re += r0; vn_0.im += i1;
896 a[p-1].re = r0; a[p-1].im = i0;
897 b[p-1].re = r1; b[p-1].im = i1;
898 }
899 }
900
901 v[0] = vn_0;
902
903 for( p = 1, k = nx; p <= factor2; p++, k += nx )
904 {
905 Complex<T> s0 = v_0, s1 = v_0;
906 d = dd = dw_f*p;
907
908 for( q = 0; q < factor2; q++ )
909 {
910 T r0 = wave[d].re * a[q].re;
911 T i0 = wave[d].im * a[q].im;
912 T r1 = wave[d].re * b[q].im;
913 T i1 = wave[d].im * b[q].re;
914
915 s1.re += r0 + i0; s0.re += r0 - i0;
916 s1.im += r1 - i1; s0.im += r1 + i1;
917
918 d += dd;
919 d -= -(d >= tab_size) & tab_size;
920 }
921
922 v[k] = s0;
923 v[n-k] = s1;
924 }
925 }
926 }
927 }
928 }
929
930 if( scale != 1 )
931 {
932 T re_scale = scale, im_scale = scale;
933 if( inv )
934 im_scale = -im_scale;
935
936 for( i = 0; i < n0; i++ )
937 {
938 T t0 = dst[i].re*re_scale;
939 T t1 = dst[i].im*im_scale;
940 dst[i].re = t0;
941 dst[i].im = t1;
942 }
943 }
944 else if( inv )
945 {
946 for( i = 0; i <= n0 - 2; i += 2 )
947 {
948 T t0 = -dst[i].im;
949 T t1 = -dst[i+1].im;
950 dst[i].im = t0;
951 dst[i+1].im = t1;
952 }
953
954 if( i < n0 )
955 dst[n0-1].im = -dst[n0-1].im;
956 }
957}

960/* FFT of real vector
961 output vector format:
962 re(0), re(1), im(1), ... , re(n/2-1), im((n+1)/2-1) [, re((n+1)/2)] OR ...
963 re(0), 0, re(1), im(1), ..., re(n/2-1), im((n+1)/2-1) [, re((n+1)/2), 0] */
964template<typename T> static void
965RealDFT( const T* src, T* dst, int n, int nf, int* factors, const int* itab,
966 const Complex<T>* wave, int tab_size, const void*
967#ifdef USE_IPP_DFT
968 spec
969#endif
970 ,
971 Complex<T>* buf, int flags, double _scale )
972{
973 int complex_output = (flags & DFT_COMPLEX_INPUT_OR_OUTPUT) != 0;
974 T scale = (T)_scale;
975 int j, n2 = n >> 1;
976 dst += complex_output;
977
978#ifdef USE_IPP_DFT
979 if( spec )
980 {
981 ippsDFTFwd_RToPack( src, dst, spec, (uchar*)buf );
982 if( complex_output )
983 {
984 dst[-1] = dst[0];
985 dst[0] = 0;
986 if( (n & 1) == 0 )
987 dst[n] = 0;
988 }
989 return;
990 }
991#endif
992 assert( tab_size == n );
993
994 if( n == 1 )
995 {
996 dst[0] = src[0]*scale;
997 }
998 else if( n == 2 )
999 {
1000 T t = (src[0] + src[1])*scale;
1001 dst[1] = (src[0] - src[1])*scale;
1002 dst[0] = t;
1003 }
1004 else if( n & 1 )
1005 {
1006 dst -= complex_output;
1007 Complex<T>* _dst = (Complex<T>*)dst;
1008 _dst[0].re = src[0]*scale;
1009 _dst[0].im = 0;
1010 for( j = 1; j < n; j += 2 )
1011 {
1012 T t0 = src[itab[j]]*scale;
1013 T t1 = src[itab[j+1]]*scale;
1014 _dst[j].re = t0;
1015 _dst[j].im = 0;
1016 _dst[j+1].re = t1;
1017 _dst[j+1].im = 0;
1018 }
1019 DFT( _dst, _dst, n, nf, factors, itab, wave,
1020 tab_size, 0, buf, DFT_NO_PERMUTE, 1 );
1021 if( !complex_output )
1022 dst[1] = dst[0];
1023 }
1024 else
1025 {
1026 T t0, t;
1027 T h1_re, h1_im, h2_re, h2_im;
1028 T scale2 = scale*(T)0.5;
1029 factors[0] >>= 1;
1030
1031 DFT( (Complex<T>*)src, (Complex<T>*)dst, n2, nf - (factors[0] == 1),
1032 factors + (factors[0] == 1),
1033 itab, wave, tab_size, 0, buf, 0, 1 );
1034 factors[0] <<= 1;
1035
1036 t = dst[0] - dst[1];
1037 dst[0] = (dst[0] + dst[1])*scale;
1038 dst[1] = t*scale;
1039
1040 t0 = dst[n2];
1041 t = dst[n-1];
1042 dst[n-1] = dst[1];
1043
1044 for( j = 2, wave++; j < n2; j += 2, wave++ )
1045 {
1046 /* calc odd */
1047 h2_re = scale2*(dst[j+1] + t);
1048 h2_im = scale2*(dst[n-j] - dst[j]);
1049
1050 /* calc even */
1051 h1_re = scale2*(dst[j] + dst[n-j]);
1052 h1_im = scale2*(dst[j+1] - t);
1053
1054 /* rotate */
1055 t = h2_re*wave->re - h2_im*wave->im;
1056 h2_im = h2_re*wave->im + h2_im*wave->re;
1057 h2_re = t;
1058 t = dst[n-j-1];
1059
1060 dst[j-1] = h1_re + h2_re;
1061 dst[n-j-1] = h1_re - h2_re;
1062 dst[j] = h1_im + h2_im;
1063 dst[n-j] = h2_im - h1_im;
1064 }
1065
1066 if( j <= n2 )
1067 {
1068 dst[n2-1] = t0*scale;
1069 dst[n2] = -t*scale;
1070 }
1071 }
1072
1073 if (complex_output && ((n & 1) == 0 || n == 1))
1074 {
1075 dst[-1] = dst[0];
1076 dst[0] = 0;
1077 if (n > 1)
1078 dst[n] = 0;
1079 }
1080}
1081
1082/* Inverse FFT of complex conjugate-symmetric vector
1083 input vector format:
1084 re[0], re[1], im[1], ... , re[n/2-1], im[n/2-1], re[n/2] OR
1085 re(0), 0, re(1), im(1), ..., re(n/2-1), im((n+1)/2-1) [, re((n+1)/2), 0] */
1086template<typename T> static void
1087CCSIDFT( const T* src, T* dst, int n, int nf, int* factors, const int* itab,
1088 const Complex<T>* wave, int tab_size,
1089 const void*
1090#ifdef USE_IPP_DFT
1091 spec
1092#endif
1093 , Complex<T>* buf,
1094 int flags, double _scale )
1095{
1096 int complex_input = (flags & DFT_COMPLEX_INPUT_OR_OUTPUT) != 0;
1097 int j, k, n2 = (n+1) >> 1;
1098 T scale = (T)_scale;
1099 T save_s1 = 0.;
1100 T t0, t1, t2, t3, t;
1101
1102 assert( tab_size == n );
1103
1104 if( complex_input )
1105 {
1106 assert( src != dst );
1107 save_s1 = src[1];
1108 ((T*)src)[1] = src[0];
1109 src++;
1110 }
1111#ifdef USE_IPP_DFT
1112 if( spec )
1113 {
1114 ippsDFTInv_PackToR( src, dst, spec, (uchar*)buf );
1115 goto finalize;
1116 }
1117#endif
1118 if( n == 1 )
1119 {
1120 dst[0] = (T)(src[0]*scale);
1121 }
1122 else if( n == 2 )
1123 {
1124 t = (src[0] + src[1])*scale;
1125 dst[1] = (src[0] - src[1])*scale;
1126 dst[0] = t;
1127 }
1128 else if( n & 1 )
1129 {
1130 Complex<T>* _src = (Complex<T>*)(src-1);
1131 Complex<T>* _dst = (Complex<T>*)dst;
1132
1133 _dst[0].re = src[0];
1134 _dst[0].im = 0;
1135 for( j = 1; j < n2; j++ )
1136 {
1137 int k0 = itab[j], k1 = itab[n-j];
1138 t0 = _src[j].re; t1 = _src[j].im;
1139 _dst[k0].re = t0; _dst[k0].im = -t1;
1140 _dst[k1].re = t0; _dst[k1].im = t1;
1141 }
1142
1143 DFT( _dst, _dst, n, nf, factors, itab, wave,
1144 tab_size, 0, buf, DFT_NO_PERMUTE, 1. );
1145 dst[0] *= scale;
1146 for( j = 1; j < n; j += 2 )
1147 {
1148 t0 = dst[j*2]*scale;
1149 t1 = dst[j*2+2]*scale;
1150 dst[j] = t0;
1151 dst[j+1] = t1;
1152 }
1153 }
1154 else
1155 {
1156 int inplace = src == dst;
1157 const Complex<T>* w = wave;
1158
1159 t = src[1];
1160 t0 = (src[0] + src[n-1]);
1161 t1 = (src[n-1] - src[0]);
1162 dst[0] = t0;
1163 dst[1] = t1;
1164
1165 for( j = 2, w++; j < n2; j += 2, w++ )
1166 {
1167 T h1_re, h1_im, h2_re, h2_im;
1168
1169 h1_re = (t + src[n-j-1]);
1170 h1_im = (src[j] - src[n-j]);
1171
1172 h2_re = (t - src[n-j-1]);
1173 h2_im = (src[j] + src[n-j]);
1174
1175 t = h2_re*w->re + h2_im*w->im;
1176 h2_im = h2_im*w->re - h2_re*w->im;
1177 h2_re = t;
1178
1179 t = src[j+1];
1180 t0 = h1_re - h2_im;
1181 t1 = -h1_im - h2_re;
1182 t2 = h1_re + h2_im;
1183 t3 = h1_im - h2_re;
1184
1185 if( inplace )
1186 {
1187 dst[j] = t0;
1188 dst[j+1] = t1;
1189 dst[n-j] = t2;
1190 dst[n-j+1]= t3;
1191 }
1192 else
1193 {
1194 int j2 = j >> 1;
1195 k = itab[j2];
1196 dst[k] = t0;
1197 dst[k+1] = t1;
1198 k = itab[n2-j2];
1199 dst[k] = t2;
1200 dst[k+1]= t3;
1201 }
1202 }
1203
1204 if( j <= n2 )
1205 {
1206 t0 = t*2;
1207 t1 = src[n2]*2;
1208
1209 if( inplace )
1210 {
1211 dst[n2] = t0;
1212 dst[n2+1] = t1;
1213 }
1214 else
1215 {
1216 k = itab[n2];
1217 dst[k*2] = t0;
1218 dst[k*2+1] = t1;
1219 }
1220 }
1221
1222 factors[0] >>= 1;
1223 DFT( (Complex<T>*)dst, (Complex<T>*)dst, n2,
1224 nf - (factors[0] == 1),
1225 factors + (factors[0] == 1), itab,
1226 wave, tab_size, 0, buf,
1227 inplace ? 0 : DFT_NO_PERMUTE, 1. );
1228 factors[0] <<= 1;
1229
1230 for( j = 0; j < n; j += 2 )
1231 {
1232 t0 = dst[j]*scale;
1233 t1 = dst[j+1]*(-scale);
1234 dst[j] = t0;
1235 dst[j+1] = t1;
1236 }
1237 }
1238
1239#ifdef USE_IPP_DFT
1240finalize:
1241#endif
1242 if( complex_input )
1243 ((T*)src)[0] = (T)save_s1;
1244}

1519{

1521 {
1522 (DFTFunc)DFT_32f,
1523 (DFTFunc)RealDFT_32f,
1524 (DFTFunc)CCSIDFT_32f,
1525 (DFTFunc)DFT_64f,
1526 (DFTFunc)RealDFT_64f,
1527 (DFTFunc)CCSIDFT_64f

1529
1530 AutoBuffer<uchar> buf;
1531 void *spec = 0;
1532
1533 Mat src0 = _src0.getMat(), src = src0;
1534 int prev_len = 0, stage = 0;
1535 bool inv = (flags & DFT_INVERSE) != 0;
1536 int nf = 0, real_transform = src.channels() == 1 || (inv && (flags & DFT_REAL_OUTPUT)!=0);
1537 int type = src.type(), depth = src.depth();
1538 int elem_size = (int)src.elemSize1(), complex_elem_size = elem_size*2;

1541#ifdef USE_IPP_DFT

1544#endif
1545
1546 CV_Assert( type == CV_32FC1 || type == CV_32FC2 || type == CV_64FC1 || type == CV_64FC2 );
1547
1548 if( !inv && src.channels() == 1 && (flags & DFT_COMPLEX_OUTPUT) )
1549 _dst.create( src.size(), CV_MAKETYPE(depth, 2) );
1550 else if( inv && src.channels() == 2 && (flags & DFT_REAL_OUTPUT) )
1551 _dst.create( src.size(), depth );
1552 else
1553 _dst.create( src.size(), type );
1554
1555 Mat dst = _dst.getMat();
1556
1557 if( !real_transform )
1558 elem_size = complex_elem_size;
1559
1560 if( src.cols == 1 && nonzero_rows > 0 )
1561 CV_Error( CV_StsNotImplemented,
1562 "This mode (using nonzero_rows with a single-column matrix) breaks the function's logic, so it is prohibited.\n"
1563 "For fast convolution/correlation use 2-column matrix or single-row matrix instead" );
1564

1566 // (stage 0) or column-wise (stage 1) transform
1567 if( !(flags & DFT_ROWS) && src.rows > 1 &&
1568 ((src.cols == 1 && (!src.isContinuous() || !dst.isContinuous())) ||
1569 (src.cols > 1 && inv && real_transform)) )
1570 stage = 1;
1571
1572 for(;;)
1573 {
1574 double scale = 1;
1575 uchar* wave = 0;
1576 int* itab = 0;
1577 uchar* ptr;
1578 int i, len, count, sz = 0;
1579 int use_buf = 0, odd_real = 0;
1580 DFTFunc dft_func;
1581
1582 if( stage == 0 ) // row-wise transform
1583 {
1584 len = !inv ? src.cols : dst.cols;
1585 count = src.rows;
1586 if( len == 1 && !(flags & DFT_ROWS) )
1587 {
1588 len = !inv ? src.rows : dst.rows;
1589 count = 1;
1590 }
1591 odd_real = real_transform && (len & 1);
1592 }
1593 else
1594 {
1595 len = dst.rows;
1596 count = !inv ? src0.cols : dst.cols;
1597 sz = 2*len*complex_elem_size;
1598 }
1599
1600 spec = 0;
1601#ifdef USE_IPP_DFT
1602 if( len*count >= 64 ) // use IPP DFT if available
1603 {
1604 int specsize=0, initsize=0, worksize=0;
1605 IppDFTGetSizeFunc getSizeFunc = 0;
1606 IppDFTInitFunc initFunc = 0;

1669 // twiddle factors and permutation table

1689 int _flags = (int)inv + (src.channels() != dst.channels() ?

1764 else if( flags & CV_DXT_SCALE )

1797 else

1800 if( even )
1801 {
1802 CopyColumn( sptr0 + b*complex_elem_size, src.step,
1803 buf1, complex_elem_size, len, complex_elem_size );
1804 }
1805 sptr0 += complex_elem_size;
1806 }

1808 if( even )
1809 dft_func( buf1, dbuf1, len, nf, factors, itab,
1810 wave, len, spec, ptr, inv, scale );

1812 wave, len, spec, ptr, inv, scale );
1813
1814 if( dst.channels() == 1 )
1815 {
1816 if( !inv )
1817 {
1818 // copy the half of output vector to the first/last column.
1819 // before doing that, defgragment the vector
1820 memcpy( dbuf0 + elem_size, dbuf0, elem_size );
1821 CopyColumn( dbuf0 + elem_size, elem_size, dptr0,
1822 dst.step, len, elem_size );
1823 if( even )
1824 {
1825 memcpy( dbuf1 + elem_size, dbuf1, elem_size );
1826 CopyColumn( dbuf1 + elem_size, elem_size,
1827 dptr0 + (count-1)*elem_size,
1828 dst.step, len, elem_size );
1829 }
1830 dptr0 += elem_size;
1831 }
1832 else
1833 {
1834 // copy the real part of the complex vector to the first/last column
1835 CopyColumn( dbuf0, complex_elem_size, dptr0, dst.step, len, elem_size );
1836 if( even )
1837 CopyColumn( dbuf1, complex_elem_size, dptr0 + (count-1)*elem_size,
1838 dst.step, len, elem_size );
1839 dptr0 += elem_size;
1840 }
1841 }
1842 else
1843 {
1844 assert( !inv );
1845 CopyColumn( dbuf0, complex_elem_size, dptr0,
1846 dst.step, len, complex_elem_size );
1847 if( even )
1848 CopyColumn( dbuf1, complex_elem_size,
1849 dptr0 + b*complex_elem_size,
1850 dst.step, len, complex_elem_size );
1851 dptr0 += complex_elem_size;
1852 }
1853 }
1854
1855 for( i = a; i < b; i += 2 )
1856 {
1857 if( i+1 < b )
1858 {
1859 CopyFrom2Columns( sptr0, src.step, buf0, buf1, len, complex_elem_size );
1860 dft_func( buf1, dbuf1, len, nf, factors, itab,
1861 wave, len, spec, ptr, inv, scale );
1862 }
1863 else
1864 CopyColumn( sptr0, src.step, buf0, complex_elem_size, len, complex_elem_size );
1865
1866 dft_func( buf0, dbuf0, len, nf, factors, itab,
1867 wave, len, spec, ptr, inv, scale );
1868
1869 if( i+1 < b )
1870 CopyTo2Columns( dbuf0, dbuf1, dptr0, dst.step, len, complex_elem_size );
1871 else
1872 CopyColumn( dbuf0, complex_elem_size, dptr0, dst.step, len, complex_elem_size );
1873 sptr0 += 2*complex_elem_size;
1874 dptr0 += 2*complex_elem_size;
1875 }
1876
1877 if( stage != 0 )
1878 {
1879 if( !inv && real_transform && dst.channels() == 2 && len > 1 )
1880 complementComplexOutput(dst, len, 2);
1881 break;
1882 }
1883 src = dst;
1884 }

2076template<typename T> static void
2077DCT( const T* src, int src_step, T* dft_src, T* dft_dst, T* dst, int dst_step,
2078 int n, int nf, int* factors, const int* itab, const Complex<T>* dft_wave,
2079 const Complex<T>* dct_wave, const void* spec, Complex<T>* buf )
2080{
2081 static const T sin_45 = (T)0.70710678118654752440084436210485;
2082 int j, n2 = n >> 1;
2083
2084 src_step /= sizeof(src[0]);
2085 dst_step /= sizeof(dst[0]);
2086 T* dst1 = dst + (n-1)*dst_step;
2087
2088 if( n == 1 )
2089 {
2090 dst[0] = src[0];
2091 return;
2092 }
2093
2094 for( j = 0; j < n2; j++, src += src_step*2 )
2095 {
2096 dft_src[j] = src[0];
2097 dft_src[n-j-1] = src[src_step];
2098 }
2099
2100 RealDFT( dft_src, dft_dst, n, nf, factors,
2101 itab, dft_wave, n, spec, buf, 0, 1.0 );
2102 src = dft_dst;
2103
2104 dst[0] = (T)(src[0]*dct_wave->re*sin_45);
2105 dst += dst_step;
2106 for( j = 1, dct_wave++; j < n2; j++, dct_wave++,
2107 dst += dst_step, dst1 -= dst_step )
2108 {
2109 T t0 = dct_wave->re*src[j*2-1] - dct_wave->im*src[j*2];
2110 T t1 = -dct_wave->im*src[j*2-1] - dct_wave->re*src[j*2];
2111 dst[0] = t0;
2112 dst1[0] = t1;
2113 }
2114
2115 dst[0] = src[n-1]*dct_wave->re;
2116}
