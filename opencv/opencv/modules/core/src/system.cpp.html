95std::wstring GetTempPathWinRT()
96{
97#ifdef __cplusplus_winrt
98 return std::wstring(Windows::Storage::ApplicationData::Current->TemporaryFolder->Path->Data());
99#else
100 Microsoft::WRL::ComPtr<ABI::Windows::Storage::IApplicationDataStatics> appdataFactory;
101 Microsoft::WRL::ComPtr<ABI::Windows::Storage::IApplicationData> appdataRef;
102 Microsoft::WRL::ComPtr<ABI::Windows::Storage::IStorageFolder> storagefolderRef;
103 Microsoft::WRL::ComPtr<ABI::Windows::Storage::IStorageItem> storageitemRef;
104 HSTRING str;
105 HSTRING_HEADER hstrHead;
106 std::wstring wstr;
107 if (FAILED(WindowsCreateStringReference(RuntimeClass_Windows_Storage_ApplicationData,
108 (UINT32)wcslen(RuntimeClass_Windows_Storage_ApplicationData), &hstrHead, &str)))
109 return wstr;
110 if (FAILED(RoGetActivationFactory(str, IID_PPV_ARGS(appdataFactory.ReleaseAndGetAddressOf()))))
111 return wstr;
112 if (FAILED(appdataFactory->get_Current(appdataRef.ReleaseAndGetAddressOf())))
113 return wstr;
114 if (FAILED(appdataRef->get_TemporaryFolder(storagefolderRef.ReleaseAndGetAddressOf())))
115 return wstr;
116 if (FAILED(storagefolderRef.As(&storageitemRef)))
117 return wstr;
118 str = NULL;
119 if (FAILED(storageitemRef->get_Path(&str)))
120 return wstr;
121 wstr = WindowsGetStringRawBuffer(str, NULL);
122 WindowsDeleteString(str);
123 return wstr;
124#endif
125}

127std::wstring GetTempFileNameWinRT(std::wstring prefix)
128{
129 wchar_t guidStr[40];
130 GUID g;
131 CoCreateGuid(&g);
132 wchar_t* mask = L"%08x_%04x_%04x_%02x%02x_%02x%02x%02x%02x%02x%02x";
133 swprintf(&guidStr[0], sizeof(guidStr)/sizeof(wchar_t), mask,
134 g.Data1, g.Data2, g.Data3, UINT(g.Data4[0]), UINT(g.Data4[1]),
135 UINT(g.Data4[2]), UINT(g.Data4[3]), UINT(g.Data4[4]),
136 UINT(g.Data4[5]), UINT(g.Data4[6]), UINT(g.Data4[7]));
137
138 return prefix + std::wstring(guidStr);
139}

191void Exception::formatMessage()
192{
193 if( func.size() > 0 )
194 msg = format("%s:%d: error: (%d) %s in function %s\n", file.c_str(), line, code, err.c_str(), func.c_str());
195 else
196 msg = format("%s:%d: error: (%d) %s\n", file.c_str(), line, code, err.c_str());
197}

199struct HWFeatures
200{
201 enum { MAX_FEATURE = CV_HARDWARE_MAX_FEATURE };
202
203 HWFeatures(void)
204 {
205 memset( have, 0, sizeof(have) );
206 x86_family = 0;
207 }
208
209 static HWFeatures initialize(void)
210 {
211 HWFeatures f;
212 int cpuid_data[4] = { 0, 0, 0, 0 };
213
214 #if defined _MSC_VER && (defined _M_IX86 || defined _M_X64)
215 __cpuid(cpuid_data, 1);
216 #elif defined __GNUC__ && (defined __i386__ || defined __x86_64__)
217 #ifdef __x86_64__
218 asm __volatile__
219 (
220 "movl $1, %%eax\n\t"
221 "cpuid\n\t"
222 :[eax]"=a"(cpuid_data[0]),[ebx]"=b"(cpuid_data[1]),[ecx]"=c"(cpuid_data[2]),[edx]"=d"(cpuid_data[3])
223 :
224 : "cc"
225 );
226 #else
227 asm volatile
228 (
229 "pushl %%ebx\n\t"
230 "movl $1,%%eax\n\t"
231 "cpuid\n\t"
232 "popl %%ebx\n\t"
233 : "=a"(cpuid_data[0]), "=c"(cpuid_data[2]), "=d"(cpuid_data[3])
234 :
235 : "cc"
236 );
237 #endif
238 #endif
239
240 f.x86_family = (cpuid_data[0] >> 8) & 15;
241 if( f.x86_family >= 6 )
242 {
243 f.have[CV_CPU_MMX] = (cpuid_data[3] & (1 << 23)) != 0;
244 f.have[CV_CPU_SSE] = (cpuid_data[3] & (1<<25)) != 0;
245 f.have[CV_CPU_SSE2] = (cpuid_data[3] & (1<<26)) != 0;
246 f.have[CV_CPU_SSE3] = (cpuid_data[2] & (1<<0)) != 0;
247 f.have[CV_CPU_SSSE3] = (cpuid_data[2] & (1<<9)) != 0;
248 f.have[CV_CPU_SSE4_1] = (cpuid_data[2] & (1<<19)) != 0;
249 f.have[CV_CPU_SSE4_2] = (cpuid_data[2] & (1<<20)) != 0;
250 f.have[CV_CPU_POPCNT] = (cpuid_data[2] & (1<<23)) != 0;
251 f.have[CV_CPU_AVX] = (((cpuid_data[2] & (1<<28)) != 0)&&((cpuid_data[2] & (1<<27)) != 0));//OS uses XSAVE_XRSTORE and CPU support AVX
252 }
253
254 #if defined _MSC_VER && (defined _M_IX86 || defined _M_X64)
255 __cpuidex(cpuid_data, 7, 0);
256 #elif defined __GNUC__ && (defined __i386__ || defined __x86_64__)
257 #ifdef __x86_64__
258 asm __volatile__
259 (
260 "movl $7, %%eax\n\t"
261 "movl $0, %%ecx\n\t"
262 "cpuid\n\t"
263 :[eax]"=a"(cpuid_data[0]),[ebx]"=b"(cpuid_data[1]),[ecx]"=c"(cpuid_data[2]),[edx]"=d"(cpuid_data[3])
264 :
265 : "cc"
266 );
267 #else
268 // We need to preserve ebx since we are compiling PIC code.
269 // This means we cannot use "=b" for the 2nd output register.
270 asm volatile
271 (
272 "pushl %%ebx\n\t"
273 "movl $7,%%eax\n\t"
274 "movl $0,%%ecx\n\t"
275 "cpuid\n\t"
276 "movl %%ebx,%1\n\t"
277 "popl %%ebx\n\t"
278 : "=a"(cpuid_data[0]), "=r"(cpuid_data[1]), "=c"(cpuid_data[2]), "=d"(cpuid_data[3])
279 :
280 : "cc"
281 );
282 #endif
283 #endif
284
285 if( f.x86_family >= 6 )
286 {
287 f.have[CV_CPU_AVX2] = (cpuid_data[1] & (1<<5)) != 0;
288 }
289
290 return f;
291 }
292
293 int x86_family;
294 bool have[MAX_FEATURE+1];
295};

333int64 getTickCount(void)
334{
335#if defined WIN32 || defined _WIN32 || defined WINCE
336 LARGE_INTEGER counter;
337 QueryPerformanceCounter( &counter );
338 return (int64)counter.QuadPart;
339#elif defined __linux || defined __linux__
340 struct timespec tp;
341 clock_gettime(CLOCK_MONOTONIC, &tp);
342 return (int64)tp.tv_sec*1000000000 + tp.tv_nsec;
343#elif defined __MACH__ && defined __APPLE__
344 return (int64)mach_absolute_time();
345#else
346 struct timeval tv;
347 struct timezone tz;
348 gettimeofday( &tv, &tz );
349 return (int64)tv.tv_sec*1000000 + tv.tv_usec;
350#endif
351}

353double getTickFrequency(void)
354{
355#if defined WIN32 || defined _WIN32 || defined WINCE
356 LARGE_INTEGER freq;
357 QueryPerformanceFrequency(&freq);
358 return (double)freq.QuadPart;
359#elif defined __linux || defined __linux__
360 return 1e9;
361#elif defined __MACH__ && defined __APPLE__
362 static double freq = 0;
363 if( freq == 0 )
364 {
365 mach_timebase_info_data_t sTimebaseInfo;
366 mach_timebase_info(&sTimebaseInfo);
367 freq = sTimebaseInfo.denom*1e9/sTimebaseInfo.numer;
368 }
369 return freq;
370#else
371 return 1e6;
372#endif
373}

390 return (int64)lo | ((int64)hi << 32);
391}

395int64 getCPUTickCount(void)
396{
397 int64 result = 0;
398 unsigned upper, lower, tmp;
399 __asm__ volatile(
400 "0: \n"
401 "\tmftbu %0 \n"
402 "\tmftb %1 \n"
403 "\tmftbu %2 \n"
404 "\tcmpw %2,%0 \n"
405 "\tbne 0b \n"
406 : "=r"(upper),"=r"(lower),"=r"(tmp)
407 );
408 return lower | ((int64)upper << 32);
409}

458string tempfile( const char* suffix )
459{
460 string fname;
461#ifndef HAVE_WINRT
462 const char *temp_dir = getenv("OPENCV_TEMP_PATH");
463#endif
464
465#if defined WIN32 || defined _WIN32
466#ifdef HAVE_WINRT
467 RoInitialize(RO_INIT_MULTITHREADED);
468 std::wstring temp_dir = L"";
469 const wchar_t* opencv_temp_dir = GetTempPathWinRT().c_str();
470 if (opencv_temp_dir)
471 temp_dir = std::wstring(opencv_temp_dir);
472
473 std::wstring temp_file;
474 temp_file = GetTempFileNameWinRT(L"ocv");
475 if (temp_file.empty())
476 return string();
477
478 temp_file = temp_dir + std::wstring(L"\\") + temp_file;
479 DeleteFileW(temp_file.c_str());
480
481 char aname[MAX_PATH];
482 size_t copied = wcstombs(aname, temp_file.c_str(), MAX_PATH);
483 CV_Assert((copied != MAX_PATH) && (copied != (size_t)-1));
484 fname = string(aname);
485 RoUninitialize();
486#else
487 char temp_dir2[MAX_PATH] = { 0 };
488 char temp_file[MAX_PATH] = { 0 };
489
490 if (temp_dir == 0 || temp_dir[0] == 0)
491 {
492 ::GetTempPathA(sizeof(temp_dir2), temp_dir2);
493 temp_dir = temp_dir2;
494 }
495 if(0 == ::GetTempFileNameA(temp_dir, "ocv", 0, temp_file))
496 return string();
497
498 DeleteFileA(temp_file);
499
500 fname = temp_file;
501#endif
502# else
503# ifdef ANDROID
504 //char defaultTemplate[] = "/mnt/sdcard/__opencv_temp.XXXXXX";
505 char defaultTemplate[] = "/data/local/tmp/__opencv_temp.XXXXXX";
506# else
507 char defaultTemplate[] = "/tmp/__opencv_temp.XXXXXX";
508# endif
509
510 if (temp_dir == 0 || temp_dir[0] == 0)
511 fname = defaultTemplate;
512 else
513 {
514 fname = temp_dir;
515 char ech = fname[fname.size() - 1];
516 if(ech != '/' && ech != '\\')
517 fname += "/";
518 fname += "__opencv_temp.XXXXXX";
519 }
520
521 const int fd = mkstemp((char*)fname.c_str());
522 if (fd == -1) return string();
523
524 close(fd);
525 remove(fname.c_str());
526# endif
527
528 if (suffix)
529 {
530 if (suffix[0] != '.')
531 return fname + "." + suffix;
532 else
533 return fname + suffix;
534 }
535 return fname;
536}

538static CvErrorCallback customErrorCallback = 0;
539static void* customErrorCallbackData = 0;
540static bool breakOnError = false;
541
542bool setBreakOnError(bool value)
543{
544 bool prevVal = breakOnError;
545 breakOnError = value;
546 return prevVal;
547}
548
549void error( const Exception& exc )
550{
551 if (customErrorCallback != 0)
552 customErrorCallback(exc.code, exc.func.c_str(), exc.err.c_str(),
553 exc.file.c_str(), exc.line, customErrorCallbackData);
554 else
555 {
556 const char* errorStr = cvErrorStr(exc.code);
557 char buf[1 << 16];
558
559 sprintf( buf, "OpenCV Error: %s (%s) in %s, file %s, line %d",
560 errorStr, exc.err.c_str(), exc.func.size() > 0 ?
561 exc.func.c_str() : "unknown function", exc.file.c_str(), exc.line );
562 fprintf( stderr, "%s\n", buf );
563 fflush( stderr );
564# ifdef __ANDROID__
565 __android_log_print(ANDROID_LOG_ERROR, "cv::error()", "%s", buf);
566# endif
567 }
568
569 if(breakOnError)
570 {
571 static volatile int* p = 0;
572 *p = 0;
573 }
574
575 throw exc;
576}

590}

649 static char buf[256];

656 case CV_StsInternal : return "Internal error";

662 case CV_StsNullPtr : return "Null pointer";
663 case CV_StsDivByZero : return "Division by zero occured";

672 case CV_BadCOI : return "Input COI is not supported";
673 case CV_BadNumChannels : return "Bad number of channels";
674 case CV_StsBadFlag : return "Bad flag (parameter or structure field)";
675 case CV_StsBadPoint : return "Bad parameter of type CvPoint";
676 case CV_StsBadMask : return "Bad type of mask argument";
677 case CV_StsParseError : return "Parsing error";
678 case CV_StsNotImplemented : return "The function/feature is not implemented";
679 case CV_StsBadMemBlock : return "Memory block has been corrupted";

718/* function, which converts int to int */
719CV_IMPL int
720cvErrorFromIppStatus( int status )
721{
722 switch (status)
723 {
724 case CV_BADSIZE_ERR: return CV_StsBadSize;
725 case CV_BADMEMBLOCK_ERR: return CV_StsBadMemBlock;
726 case CV_NULLPTR_ERR: return CV_StsNullPtr;
727 case CV_DIV_BY_ZERO_ERR: return CV_StsDivByZero;

782CV_IMPL int
783cvRegisterModule( const CvModuleInfo* module )
784{
785 CV_Assert( module != 0 && module->name != 0 && module->version != 0 );
786
787 size_t name_len = strlen(module->name);
788 size_t version_len = strlen(module->version);
789
790 CvModuleInfo* module_copy = (CvModuleInfo*)malloc( sizeof(*module_copy) +
791 name_len + 1 + version_len + 1 );
792
793 *module_copy = *module;
794 module_copy->name = (char*)(module_copy + 1);
795 module_copy->version = (char*)(module_copy + 1) + name_len + 1;
796
797 memcpy( (void*)module_copy->name, module->name, name_len + 1 );
798 memcpy( (void*)module_copy->version, module->version, version_len + 1 );
799 module_copy->next = 0;
800
801 if( CvModule::first == 0 )
802 CvModule::first = module_copy;
803 else
804 CvModule::last->next = module_copy;
805
806 CvModule::last = module_copy;
807
808 return 0;
809}

813CV_IMPL void
814cvGetModuleInfo( const char* name, const char **version, const char **plugin_list )
815{
816 static char joint_verinfo[1024] = "";
817 static char plugin_list_buf[1024] = "";
818
819 if( version )
820 *version = 0;
821
822 if( plugin_list )
823 *plugin_list = 0;
824
825 CvModuleInfo* module;
826
827 if( version )
828 {
829 if( name )
830 {
831 size_t i, name_len = strlen(name);
832
833 for( module = CvModule::first; module != 0; module = module->next )
834 {
835 if( strlen(module->name) == name_len )
836 {
837 for( i = 0; i < name_len; i++ )
838 {
839 int c0 = toupper(module->name[i]), c1 = toupper(name[i]);
840 if( c0 != c1 )
841 break;
842 }
843 if( i == name_len )
844 break;
845 }
846 }
847 if( !module )
848 CV_Error( CV_StsObjectNotFound, "The module is not found" );
849
850 *version = module->version;
851 }
852 else
853 {
854 char* ptr = joint_verinfo;
855
856 for( module = CvModule::first; module != 0; module = module->next )
857 {
858 sprintf( ptr, "%s: %s%s", module->name, module->version, module->next ? ", " : "" );
859 ptr += strlen(ptr);
860 }
861
862 *version = joint_verinfo;
863 }
864 }
865
866 if( plugin_list )
867 *plugin_list = plugin_list_buf;
868}
