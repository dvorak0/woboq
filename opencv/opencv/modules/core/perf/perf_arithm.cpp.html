74{

77 cv::Mat a = Mat(sz, type);
78 cv::Mat b = Mat(sz, type);
79 cv::Mat c = Mat(sz, type);
80
81 declare.in(a, b, WARMUP_RNG).out(c);
82
83 if (CV_MAT_DEPTH(type) == CV_32S)

85 //see ticket 1529: absdiff can be without saturation on 32S
