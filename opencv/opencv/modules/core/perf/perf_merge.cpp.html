12PERF_TEST_P( Size_SrcDepth_DstChannels, merge,
13 testing::Combine
14 (
15 testing::Values(TYPICAL_MAT_SIZES),
16 testing::Values(CV_8U, CV_16S, CV_32S, CV_32F, CV_64F),
17 testing::Values(2, 3, 4)
18 )
19 )
20{
21 Size sz = get<0>(GetParam());

23 int dstChannels = get<2>(GetParam());
24
25 vector<Mat> mv;
26 for( int i = 0; i < dstChannels; ++i )
27 {
28 mv.push_back( Mat(sz, CV_MAKETYPE(srcDepth, 1)) );
29 randu(mv[i], 0, 255);
30 }
31
32 Mat dst;
33 int runs = (sz.width <= 640) ? 8 : 1;
34 TEST_CYCLE_MULTIRUN(runs) merge( (vector<Mat> &)mv, dst );
35
36#ifdef __aarch64__
37 // looks like random generator produces a little bit
38 // different source data on aarch64 platform and
39 // eps should be increased to allow the tests pass
40 SANITY_CHECK(dst, (srcDepth == CV_32F ? 1.55e-5 : 1e-12));
41#else
42 SANITY_CHECK(dst, 1e-12);

44}
