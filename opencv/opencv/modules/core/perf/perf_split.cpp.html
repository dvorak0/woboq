12PERF_TEST_P( Size_Depth_Channels, split,
13 testing::Combine

18 )

30 TEST_CYCLE_MULTIRUN(runs) split(m, (vector<Mat>&)mv);
31
32#ifdef __aarch64__
33 // looks like random generator produces a little bit
34 // different source data on aarch64 platform and
