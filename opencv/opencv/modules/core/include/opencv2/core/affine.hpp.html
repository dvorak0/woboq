192template<typename T> inline
193cv::Affine3<T>::Affine3(const cv::Mat& data, const Vec3& t)
194{
195 CV_Assert(data.type() == cv::DataType<T>::type);
196
197 if (data.cols == 4 && data.rows == 4)
198 {
199 data.copyTo(matrix);
200 return;
201 }
202 else if (data.cols == 4 && data.rows == 3)
203 {
204 rotation(data(Rect(0, 0, 3, 3)));
205 translation(data(Rect(3, 0, 1, 3)));
206 }
207 else
208 {
209 rotation(data);
210 translation(t);
211 }
212
213 matrix.val[12] = matrix.val[13] = matrix.val[14] = 0;
214 matrix.val[15] = 1;
215}

233template<typename T> inline
234void cv::Affine3<T>::rotation(const Vec3& _rvec)
235{
236 double rx = _rvec[0], ry = _rvec[1], rz = _rvec[2];
237 double theta = std::sqrt(rx*rx + ry*ry + rz*rz);
238
239 if (theta < DBL_EPSILON)
240 rotation(Mat3::eye());
241 else
242 {
243 const double I[] = { 1, 0, 0, 0, 1, 0, 0, 0, 1 };
244
245 double c = std::cos(theta);
246 double s = std::sin(theta);
247 double c1 = 1. - c;
248 double itheta = (theta != 0) ? 1./theta : 0.;
249
250 rx *= itheta; ry *= itheta; rz *= itheta;
251
252 double rrt[] = { rx*rx, rx*ry, rx*rz, rx*ry, ry*ry, ry*rz, rx*rz, ry*rz, rz*rz };
253 double _r_x_[] = { 0, -rz, ry, rz, 0, -rx, -ry, rx, 0 };
254 Mat3 R;
255
256 // R = cos(theta)*I + (1 - cos(theta))*r*rT + sin(theta)*[r_x]
257 // where [r_x] is [0 -rz ry; rz 0 -rx; -ry rx 0]
258 for(int k = 0; k < 9; ++k)
259 R.val[k] = static_cast<float_type>(c*I[k] + c1*rrt[k] + s*_r_x_[k]);
260
261 rotation(R);
262 }
263}

265//Combines rotation methods above. Suports 3x3, 1x3, 3x1 sizes of data matrix;
266template<typename T> inline
267void cv::Affine3<T>::rotation(const cv::Mat& data)
268{
269 CV_Assert(data.type() == cv::DataType<T>::type);
270
271 if (data.cols == 3 && data.rows == 3)
272 {
273 Mat3 R;
274 data.copyTo(R);
275 rotation(R);
276 }
277 else if ((data.cols == 3 && data.rows == 1) || (data.cols == 1 && data.rows == 3))
278 {
279 Vec3 _rvec;
280 data.reshape(1, 3).copyTo(_rvec);
281 rotation(_rvec);
282 }
283 else
284 CV_Assert(!"Input marix can be 3x3, 1x3 or 3x1");
285}

287template<typename T> inline
288void cv::Affine3<T>::linear(const Mat3& L)
289{
290 matrix.val[0] = L.val[0]; matrix.val[1] = L.val[1]; matrix.val[ 2] = L.val[2];
291 matrix.val[4] = L.val[3]; matrix.val[5] = L.val[4]; matrix.val[ 6] = L.val[5];
292 matrix.val[8] = L.val[6]; matrix.val[9] = L.val[7]; matrix.val[10] = L.val[8];
293}

295template<typename T> inline
296void cv::Affine3<T>::translation(const Vec3& t)
297{
298 matrix.val[3] = t[0]; matrix.val[7] = t[1]; matrix.val[11] = t[2];
299}

301template<typename T> inline
302typename cv::Affine3<T>::Mat3 cv::Affine3<T>::rotation() const
303{
304 return linear();

307template<typename T> inline
308typename cv::Affine3<T>::Mat3 cv::Affine3<T>::linear() const
309{
310 typename cv::Affine3<T>::Mat3 R;
311 R.val[0] = matrix.val[0]; R.val[1] = matrix.val[1]; R.val[2] = matrix.val[ 2];
312 R.val[3] = matrix.val[4]; R.val[4] = matrix.val[5]; R.val[5] = matrix.val[ 6];
313 R.val[6] = matrix.val[8]; R.val[7] = matrix.val[9]; R.val[8] = matrix.val[10];
314 return R;
315}

317template<typename T> inline
318typename cv::Affine3<T>::Vec3 cv::Affine3<T>::translation() const
319{
320 return Vec3(matrix.val[3], matrix.val[7], matrix.val[11]);
321}

323template<typename T> inline
324typename cv::Affine3<T>::Vec3 cv::Affine3<T>::rvec() const
325{
326 cv::Vec3d w;
327 cv::Matx33d u, vt, R = rotation();
328 cv::SVD::compute(R, w, u, vt, cv::SVD::FULL_UV + cv::SVD::MODIFY_A);
329 R = u * vt;
330
331 double rx = R.val[7] - R.val[5];
332 double ry = R.val[2] - R.val[6];
333 double rz = R.val[3] - R.val[1];
334
335 double s = std::sqrt((rx*rx + ry*ry + rz*rz)*0.25);
336 double c = (R.val[0] + R.val[4] + R.val[8] - 1) * 0.5;
337 c = c > 1.0 ? 1.0 : c < -1.0 ? -1.0 : c;
338 double theta = acos(c);
339
340 if( s < 1e-5 )
341 {
342 if( c > 0 )
343 rx = ry = rz = 0;
344 else
345 {
346 double t;
347 t = (R.val[0] + 1) * 0.5;
348 rx = std::sqrt(std::max(t, 0.0));
349 t = (R.val[4] + 1) * 0.5;
350 ry = std::sqrt(std::max(t, 0.0)) * (R.val[1] < 0 ? -1.0 : 1.0);
351 t = (R.val[8] + 1) * 0.5;
352 rz = std::sqrt(std::max(t, 0.0)) * (R.val[2] < 0 ? -1.0 : 1.0);
353
354 if( fabs(rx) < fabs(ry) && fabs(rx) < fabs(rz) && (R.val[5] > 0) != (ry*rz > 0) )
355 rz = -rz;
356 theta /= std::sqrt(rx*rx + ry*ry + rz*rz);
357 rx *= theta;
358 ry *= theta;
359 rz *= theta;
360 }
361 }
362 else
363 {
364 double vth = 1/(2*s);
365 vth *= theta;
366 rx *= vth; ry *= vth; rz *= vth;
367 }
368
369 return cv::Vec3d(rx, ry, rz);
370}

378template<typename T> inline
379cv::Affine3<T> cv::Affine3<T>::rotate(const Mat3& R) const
380{
381 Mat3 Lc = linear();
382 Vec3 tc = translation();
383 Mat4 result;
384 result.val[12] = result.val[13] = result.val[14] = 0;
385 result.val[15] = 1;
386
387 for(int j = 0; j < 3; ++j)
388 {
389 for(int i = 0; i < 3; ++i)
390 {
391 float_type value = 0;
392 for(int k = 0; k < 3; ++k)
393 value += R(j, k) * Lc(k, i);
394 result(j, i) = value;
395 }
396
397 result(j, 3) = R.row(j).dot(tc.t());
398 }
399 return result;
400}

408template<typename T> inline
409cv::Affine3<T> cv::Affine3<T>::translate(const Vec3& t) const
410{
411 Mat4 m = matrix;
412 m.val[ 3] += t[0];
413 m.val[ 7] += t[1];
414 m.val[11] += t[2];
415 return m;
416}

418template<typename T> inline
419cv::Affine3<T> cv::Affine3<T>::concatenate(const Affine3<T>& affine) const
420{
421 return (*this).rotate(affine.rotation()).translate(affine.translation());
422}

443template<typename T, typename V> inline
444V cv::operator*(const cv::Affine3<T>& affine, const V& v)
445{
446 const typename Affine3<T>::Mat4& m = affine.matrix;
447
448 V r;
449 r.x = m.val[0] * v.x + m.val[1] * v.y + m.val[ 2] * v.z + m.val[ 3];
450 r.y = m.val[4] * v.x + m.val[5] * v.y + m.val[ 6] * v.z + m.val[ 7];
451 r.z = m.val[8] * v.x + m.val[9] * v.y + m.val[10] * v.z + m.val[11];
452 return r;
453}
454/** @endcond */

456static inline
457cv::Vec3f cv::operator*(const cv::Affine3f& affine, const cv::Vec3f& v)
458{
459 const cv::Matx44f& m = affine.matrix;
460 cv::Vec3f r;
461 r.val[0] = m.val[0] * v[0] + m.val[1] * v[1] + m.val[ 2] * v[2] + m.val[ 3];
462 r.val[1] = m.val[4] * v[0] + m.val[5] * v[1] + m.val[ 6] * v[2] + m.val[ 7];
463 r.val[2] = m.val[8] * v[0] + m.val[9] * v[1] + m.val[10] * v[2] + m.val[11];
464 return r;
465}

467static inline
468cv::Vec3d cv::operator*(const cv::Affine3d& affine, const cv::Vec3d& v)
469{
470 const cv::Matx44d& m = affine.matrix;
471 cv::Vec3d r;
472 r.val[0] = m.val[0] * v[0] + m.val[1] * v[1] + m.val[ 2] * v[2] + m.val[ 3];
473 r.val[1] = m.val[4] * v[0] + m.val[5] * v[1] + m.val[ 6] * v[2] + m.val[ 7];
474 r.val[2] = m.val[8] * v[0] + m.val[9] * v[1] + m.val[10] * v[2] + m.val[11];
475 return r;
476}
