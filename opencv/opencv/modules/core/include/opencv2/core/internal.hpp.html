465/****************************************************************************************\
466
467 Generic implementation of QuickSort algorithm.
468 ----------------------------------------------
469 Using this macro user can declare customized sort function that can be much faster
470 than built-in qsort function because of lower overhead on elements
471 comparison and exchange. The macro takes less_than (or LT) argument - a macro or function
472 that takes 2 arguments returns non-zero if the first argument should be before the second
473 one in the sorted sequence and zero otherwise.
474
475 Example:
476
477 Suppose that the task is to sort points by ascending of y coordinates and if
478 y's are equal x's should ascend.
479
480 The code is:
481 ------------------------------------------------------------------------------
482 #define cmp_pts( pt1, pt2 ) \
483 ((pt1).y < (pt2).y || ((pt1).y < (pt2).y && (pt1).x < (pt2).x))
484
485 [static] CV_IMPLEMENT_QSORT( icvSortPoints, CvPoint, cmp_pts )
486 ------------------------------------------------------------------------------
487
488 After that the function "void icvSortPoints( CvPoint* array, size_t total, int aux );"
489 is available to user.
490
491 aux is an additional parameter, which can be used when comparing elements.
492 The current implementation was derived from *BSD system qsort():

527#define CV_IMPLEMENT_QSORT_EX( func_name, T, LT, user_data_type ) \
528void func_name( T *array, size_t total, user_data_type aux ) \
529{ \
530 int isort_thresh = 7; \
531 T t; \
532 int sp = 0; \
533 \
534 struct \
535 { \
536 T *lb; \
537 T *ub; \
538 } \
539 stack[48]; \
540 \
541 (void)aux; \
542 \
543 if( total <= 1 ) \
544 return; \
545 \
546 stack[0].lb = array; \
547 stack[0].ub = array + (total - 1); \
548 \
549 while( sp >= 0 ) \
550 { \
551 T* left = stack[sp].lb; \
552 T* right = stack[sp--].ub; \
553 \
554 for(;;) \
555 { \
556 int i, n = (int)(right - left) + 1, m; \
557 T* ptr; \
558 T* ptr2; \
559 \
560 if( n <= isort_thresh ) \
561 { \
562 insert_sort: \
563 for( ptr = left + 1; ptr <= right; ptr++ ) \
564 { \
565 for( ptr2 = ptr; ptr2 > left && LT(ptr2[0],ptr2[-1]); ptr2--) \
566 CV_SWAP( ptr2[0], ptr2[-1], t ); \
567 } \
568 break; \
569 } \
570 else \
571 { \
572 T* left0; \
573 T* left1; \
574 T* right0; \
575 T* right1; \
576 T* pivot; \
577 T* a; \
578 T* b; \
579 T* c; \
580 int swap_cnt = 0; \
581 \
582 left0 = left; \
583 right0 = right; \
584 pivot = left + (n/2); \
585 \
586 if( n > 40 ) \
587 { \
588 int d = n / 8; \
589 a = left, b = left + d, c = left + 2*d; \
590 left = LT(*a, *b) ? (LT(*b, *c) ? b : (LT(*a, *c) ? c : a)) \
591 : (LT(*c, *b) ? b : (LT(*a, *c) ? a : c)); \
592 \
593 a = pivot - d, b = pivot, c = pivot + d; \
594 pivot = LT(*a, *b) ? (LT(*b, *c) ? b : (LT(*a, *c) ? c : a)) \
595 : (LT(*c, *b) ? b : (LT(*a, *c) ? a : c)); \
596 \
597 a = right - 2*d, b = right - d, c = right; \
598 right = LT(*a, *b) ? (LT(*b, *c) ? b : (LT(*a, *c) ? c : a)) \
599 : (LT(*c, *b) ? b : (LT(*a, *c) ? a : c)); \
600 } \
601 \
602 a = left, b = pivot, c = right; \
603 pivot = LT(*a, *b) ? (LT(*b, *c) ? b : (LT(*a, *c) ? c : a)) \
604 : (LT(*c, *b) ? b : (LT(*a, *c) ? a : c)); \
605 if( pivot != left0 ) \
606 { \
607 CV_SWAP( *pivot, *left0, t ); \
608 pivot = left0; \
609 } \
610 left = left1 = left0 + 1; \
611 right = right1 = right0; \
612 \
613 for(;;) \
614 { \
615 while( left <= right && !LT(*pivot, *left) ) \
616 { \
617 if( !LT(*left, *pivot) ) \
618 { \
619 if( left > left1 ) \
620 CV_SWAP( *left1, *left, t ); \
621 swap_cnt = 1; \
622 left1++; \
623 } \
624 left++; \
625 } \
626 \
627 while( left <= right && !LT(*right, *pivot) ) \
628 { \
629 if( !LT(*pivot, *right) ) \
630 { \
631 if( right < right1 ) \
632 CV_SWAP( *right1, *right, t ); \
633 swap_cnt = 1; \
634 right1--; \
635 } \
636 right--; \
637 } \
638 \
639 if( left > right ) \
640 break; \
641 CV_SWAP( *left, *right, t ); \
642 swap_cnt = 1; \
643 left++; \
644 right--; \
645 } \
646 \
647 if( swap_cnt == 0 ) \
648 { \
649 left = left0, right = right0; \
650 goto insert_sort; \
651 } \
652 \
653 n = MIN( (int)(left1 - left0), (int)(left - left1) ); \
654 for( i = 0; i < n; i++ ) \
655 CV_SWAP( left0[i], left[i-n], t ); \
656 \
657 n = MIN( (int)(right0 - right1), (int)(right1 - right) ); \
658 for( i = 0; i < n; i++ ) \
659 CV_SWAP( left[i], right0[i-n+1], t ); \
660 n = (int)(left - left1); \
661 m = (int)(right1 - right); \
662 if( n > 1 ) \
663 { \
664 if( m > 1 ) \
665 { \
666 if( n > m ) \
667 { \
668 stack[++sp].lb = left0; \
669 stack[sp].ub = left0 + n - 1; \
670 left = right0 - m + 1, right = right0; \
671 } \
672 else \
673 { \
674 stack[++sp].lb = right0 - m + 1; \
675 stack[sp].ub = right0; \
676 left = left0, right = left0 + n - 1; \
677 } \
678 } \
679 else \
680 left = left0, right = left0 + n - 1; \
681 } \
682 else if( m > 1 ) \
683 left = right0 - m + 1, right = right0; \
684 else \
685 break; \
686 } \
687 } \
688 } \
689}
