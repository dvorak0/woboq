342CV_INLINE int cvRound( double value )
343{
344#if (defined _MSC_VER && defined _M_X64) || (defined __GNUC__ && defined __x86_64__ && defined __SSE2__ && !defined __APPLE__)
345 __m128d t = _mm_set_sd( value );
346 return _mm_cvtsd_si32(t);
347#elif defined _MSC_VER && defined _M_IX86
348 int t;
349 __asm
350 {
351 fld value;
352 fistp t;
353 }
354 return t;
355#elif defined _MSC_VER && defined _M_ARM && defined HAVE_TEGRA_OPTIMIZATION
356 TEGRA_ROUND(value);
357#elif defined CV_ICC || defined __GNUC__
358# ifdef HAVE_TEGRA_OPTIMIZATION
359 TEGRA_ROUND(value);
360# elif CV_VFP
361 ARM_ROUND_DBL(value)
362# else
363 return (int)lrint(value);
364# endif
365#else
366 double intpart, fractpart;
367 fractpart = modf(value, &intpart);
368 if ((fabs(fractpart) != 0.5) || ((((int)intpart) % 2) != 0))
369 return (int)(value + (value >= 0 ? 0.5 : -0.5));
370 else
371 return (int)intpart;
372#endif
373}

379CV_INLINE int cvFloor( double value )
380{
381#if defined _MSC_VER && defined _M_X64 || (defined __GNUC__ && defined __SSE2__ && !defined __APPLE__)
382 __m128d t = _mm_set_sd( value );
383 int i = _mm_cvtsd_si32(t);
384 return i - _mm_movemask_pd(_mm_cmplt_sd(t, _mm_cvtsi32_sd(t,i)));
385#elif defined __GNUC__
386 int i = (int)value;
387 return i - (i > value);
388#else
389 int i = cvRound(value);
390 float diff = (float)(value - i);
391 return i - (diff < 0);
392#endif
393}

416{
417 Cv64suf ieee754;

445/* Return random 32-bit unsigned integer: */
446CV_INLINE unsigned cvRandInt( CvRNG* rng )
447{
448 uint64 temp = *rng;
449 temp = (uint64)(unsigned)temp*CV_RNG_COEFF + (temp >> 32);

452}
