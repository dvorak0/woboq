153 //! Smart pointer for GPU memory with reference counting. Its interface is mostly similar with cv::Mat.
154 class CV_EXPORTS GpuMat
155 {
156 public:
157 //! default constructor
158 GpuMat();
159
160 //! constructs GpuMatrix of the specified size and type (_type is CV_8UC1, CV_64FC3, CV_32SC(12) etc.)
161 GpuMat(int rows, int cols, int type);
162 GpuMat(Size size, int type);
163
164 //! constucts GpuMatrix and fills it with the specified value _s.
165 GpuMat(int rows, int cols, int type, Scalar s);
166 GpuMat(Size size, int type, Scalar s);
167
168 //! copy constructor
169 GpuMat(const GpuMat& m);
170
171 //! constructor for GpuMatrix headers pointing to user-allocated data
172 GpuMat(int rows, int cols, int type, void* data, size_t step = Mat::AUTO_STEP);
173 GpuMat(Size size, int type, void* data, size_t step = Mat::AUTO_STEP);
174
175 //! creates a matrix header for a part of the bigger matrix
176 GpuMat(const GpuMat& m, Range rowRange, Range colRange);
177 GpuMat(const GpuMat& m, Rect roi);
178
179 //! builds GpuMat from Mat. Perfom blocking upload to device.
180 explicit GpuMat(const Mat& m);
181
182 //! destructor - calls release()
183 ~GpuMat();
184
185 //! assignment operators
186 GpuMat& operator = (const GpuMat& m);
187
188 //! pefroms blocking upload data to GpuMat.
189 void upload(const Mat& m);
190
191 //! downloads data from device to host memory. Blocking calls.
192 void download(Mat& m) const;
193
194 //! returns a new GpuMatrix header for the specified row
195 GpuMat row(int y) const;
196 //! returns a new GpuMatrix header for the specified column
197 GpuMat col(int x) const;
198 //! ... for the specified row span
199 GpuMat rowRange(int startrow, int endrow) const;
200 GpuMat rowRange(Range r) const;
201 //! ... for the specified column span
202 GpuMat colRange(int startcol, int endcol) const;
203 GpuMat colRange(Range r) const;
204
205 //! returns deep copy of the GpuMatrix, i.e. the data is copied
206 GpuMat clone() const;
207 //! copies the GpuMatrix content to "m".
208 // It calls m.create(this->size(), this->type()).
209 void copyTo(GpuMat& m) const;
210 //! copies those GpuMatrix elements to "m" that are marked with non-zero mask elements.
211 void copyTo(GpuMat& m, const GpuMat& mask) const;
212 //! converts GpuMatrix to another datatype with optional scalng. See cvConvertScale.
213 void convertTo(GpuMat& m, int rtype, double alpha = 1, double beta = 0) const;
214
215 void assignTo(GpuMat& m, int type=-1) const;
216
217 //! sets every GpuMatrix element to s
218 GpuMat& operator = (Scalar s);
219 //! sets some of the GpuMatrix elements to s, according to the mask
220 GpuMat& setTo(Scalar s, const GpuMat& mask = GpuMat());
221 //! creates alternative GpuMatrix header for the same data, with different
222 // number of channels and/or different number of rows. see cvReshape.
223 GpuMat reshape(int cn, int rows = 0) const;
224
225 //! allocates new GpuMatrix data unless the GpuMatrix already has specified size and type.
226 // previous data is unreferenced if needed.
227 void create(int rows, int cols, int type);
228 void create(Size size, int type);
229 //! decreases reference counter;
230 // deallocate the data when reference counter reaches 0.
231 void release();
232
233 //! swaps with other smart pointer
234 void swap(GpuMat& mat);
235
236 //! locates GpuMatrix header within a parent GpuMatrix. See below
237 void locateROI(Size& wholeSize, Point& ofs) const;
238 //! moves/resizes the current GpuMatrix ROI inside the parent GpuMatrix.
239 GpuMat& adjustROI(int dtop, int dbottom, int dleft, int dright);
240 //! extracts a rectangular sub-GpuMatrix
241 // (this is a generalized form of row, rowRange etc.)
242 GpuMat operator()(Range rowRange, Range colRange) const;
243 GpuMat operator()(Rect roi) const;
244
245 //! returns true iff the GpuMatrix data is continuous
246 // (i.e. when there are no gaps between successive rows).
247 // similar to CV_IS_GpuMat_CONT(cvGpuMat->type)
248 bool isContinuous() const;
249 //! returns element size in bytes,
250 // similar to CV_ELEM_SIZE(cvMat->type)
251 size_t elemSize() const;
252 //! returns the size of element channel in bytes.
253 size_t elemSize1() const;
254 //! returns element type, similar to CV_MAT_TYPE(cvMat->type)
255 int type() const;
256 //! returns element type, similar to CV_MAT_DEPTH(cvMat->type)
257 int depth() const;
258 //! returns element type, similar to CV_MAT_CN(cvMat->type)
259 int channels() const;
260 //! returns step/elemSize1()
261 size_t step1() const;
262 //! returns GpuMatrix size:
263 // width == number of columns, height == number of rows
264 Size size() const;
265 //! returns true if GpuMatrix data is NULL
266 bool empty() const;
267
268 //! returns pointer to y-th row
269 uchar* ptr(int y = 0);
270 const uchar* ptr(int y = 0) const;
271
272 //! template version of the above method
273 template<typename _Tp> _Tp* ptr(int y = 0);
274 template<typename _Tp> const _Tp* ptr(int y = 0) const;
275
276 template <typename _Tp> operator PtrStepSz<_Tp>() const;
277 template <typename _Tp> operator PtrStep<_Tp>() const;
278
279 // Deprecated function
280 template <typename _Tp> CV_GPU_DEPRECATED operator DevMem2D_<_Tp>() const;
281 template <typename _Tp> CV_GPU_DEPRECATED operator PtrStep_<_Tp>() const;
282 #undef CV_GPU_DEPRECATED
283
284 /*! includes several bit-fields:
285 - the magic signature
286 - continuity flag
287 - depth
288 - number of channels
289 */
290 int flags;
291
292 //! the number of rows and columns
293 int rows, cols;
294
295 //! a distance between successive rows in bytes; includes the gap if any
296 size_t step;
297
298 //! pointer to the data
299 uchar* data;
300
301 //! pointer to the reference counter;
302 // when GpuMatrix points to user-allocated data, the pointer is NULL
303 int* refcount;
304
305 //! helper fields used in locateROI and adjustROI
306 uchar* datastart;
307 uchar* dataend;
308 };

332 inline GpuMat::GpuMat()
333 : flags(0), rows(0), cols(0), step(0), data(0), refcount(0), datastart(0), dataend(0)
334 {
335 }
336
337 inline GpuMat::GpuMat(int rows_, int cols_, int type_)
338 : flags(0), rows(0), cols(0), step(0), data(0), refcount(0), datastart(0), dataend(0)
339 {
340 if (rows_ > 0 && cols_ > 0)
341 create(rows_, cols_, type_);
342 }
343
344 inline GpuMat::GpuMat(Size size_, int type_)
345 : flags(0), rows(0), cols(0), step(0), data(0), refcount(0), datastart(0), dataend(0)
346 {
347 if (size_.height > 0 && size_.width > 0)
348 create(size_.height, size_.width, type_);
349 }
350
351 inline GpuMat::GpuMat(int rows_, int cols_, int type_, Scalar s_)
352 : flags(0), rows(0), cols(0), step(0), data(0), refcount(0), datastart(0), dataend(0)
353 {
354 if (rows_ > 0 && cols_ > 0)
355 {
356 create(rows_, cols_, type_);
357 setTo(s_);
358 }
359 }
360
361 inline GpuMat::GpuMat(Size size_, int type_, Scalar s_)
362 : flags(0), rows(0), cols(0), step(0), data(0), refcount(0), datastart(0), dataend(0)
363 {
364 if (size_.height > 0 && size_.width > 0)
365 {
366 create(size_.height, size_.width, type_);
367 setTo(s_);
368 }
369 }

371 inline GpuMat::~GpuMat()

373 release();
374 }

376 inline GpuMat GpuMat::clone() const
377 {
378 GpuMat m;
379 copyTo(m);

381 }

383 inline void GpuMat::assignTo(GpuMat& m, int _type) const
384 {
385 if (_type < 0)
386 m = *this;
387 else
388 convertTo(m, _type);
389 }

391 inline size_t GpuMat::step1() const
392 {
393 return step / elemSize1();
394 }
