51// Remap

53enum { HALF_SIZE=0, UPSIDE_DOWN, REFLECTION_X, REFLECTION_BOTH };
54CV_ENUM(RemapMode, HALF_SIZE, UPSIDE_DOWN, REFLECTION_X, REFLECTION_BOTH)
55
56void generateMap(cv::Mat& map_x, cv::Mat& map_y, int remapMode)
57{
58 for (int j = 0; j < map_x.rows; ++j)
59 {
60 for (int i = 0; i < map_x.cols; ++i)
61 {
62 switch (remapMode)
63 {
64 case HALF_SIZE:
65 if (i > map_x.cols*0.25 && i < map_x.cols*0.75 && j > map_x.rows*0.25 && j < map_x.rows*0.75)
66 {
67 map_x.at<float>(j,i) = 2.f * (i - map_x.cols * 0.25f) + 0.5f;
68 map_y.at<float>(j,i) = 2.f * (j - map_x.rows * 0.25f) + 0.5f;
69 }
70 else
71 {
72 map_x.at<float>(j,i) = 0.f;
73 map_y.at<float>(j,i) = 0.f;
74 }
75 break;
76 case UPSIDE_DOWN:
77 map_x.at<float>(j,i) = static_cast<float>(i);
78 map_y.at<float>(j,i) = static_cast<float>(map_x.rows - j);
79 break;
80 case REFLECTION_X:
81 map_x.at<float>(j,i) = static_cast<float>(map_x.cols - i);
82 map_y.at<float>(j,i) = static_cast<float>(j);
83 break;
84 case REFLECTION_BOTH:
85 map_x.at<float>(j,i) = static_cast<float>(map_x.cols - i);
86 map_y.at<float>(j,i) = static_cast<float>(map_x.rows - j);
87 break;
88 } // end of switch
89 }
90 }
91}
