78PERF_TEST_P(ImagePair, Video_InterpolateFrames,
79 Values<pair_string>(make_pair("gpu/opticalflow/frame0.png", "gpu/opticalflow/frame1.png")))
80{
81 cv::Mat frame0 = readImage(GetParam().first, cv::IMREAD_GRAYSCALE);
82 ASSERT_FALSE(frame0.empty());
83
84 cv::Mat frame1 = readImage(GetParam().second, cv::IMREAD_GRAYSCALE);
85 ASSERT_FALSE(frame1.empty());
86
87 frame0.convertTo(frame0, CV_32FC1, 1.0 / 255.0);
88 frame1.convertTo(frame1, CV_32FC1, 1.0 / 255.0);
89
90 if (PERF_RUN_GPU())
91 {
92 const cv::gpu::GpuMat d_frame0(frame0);
93 const cv::gpu::GpuMat d_frame1(frame1);
94 cv::gpu::GpuMat d_fu, d_fv;
95 cv::gpu::GpuMat d_bu, d_bv;
96
97 cv::gpu::BroxOpticalFlow d_flow(0.197f /*alpha*/, 50.0f /*gamma*/, 0.8f /*scale_factor*/,
98 10 /*inner_iterations*/, 77 /*outer_iterations*/, 10 /*solver_iterations*/);
99
100 d_flow(d_frame0, d_frame1, d_fu, d_fv);
101 d_flow(d_frame1, d_frame0, d_bu, d_bv);
102
103 cv::gpu::GpuMat newFrame;
104 cv::gpu::GpuMat d_buf;
105
106 TEST_CYCLE() cv::gpu::interpolateFrames(d_frame0, d_frame1, d_fu, d_fv, d_bu, d_bv, 0.5f, newFrame, d_buf);
107
108 GPU_SANITY_CHECK(newFrame, 1e-4);
109 }
110 else
111 {
112 FAIL_NO_CPU();
113 }
114}

234// PyrLKOpticalFlowSparse
235
236DEF_PARAM_TEST(ImagePair_Gray_NPts_WinSz_Levels_Iters, pair_string, bool, int, int, int, int);
237
238PERF_TEST_P(ImagePair_Gray_NPts_WinSz_Levels_Iters, Video_PyrLKOpticalFlowSparse,
239 Combine(Values<pair_string>(make_pair("gpu/opticalflow/frame0.png", "gpu/opticalflow/frame1.png")),
240 Bool(),
241 Values(8000),
242 Values(21),
243 Values(1, 3),
244 Values(1, 30)))
245{
246 declare.time(20.0);
247
248 const pair_string imagePair = GET_PARAM(0);
249 const bool useGray = GET_PARAM(1);
250 const int points = GET_PARAM(2);
251 const int winSize = GET_PARAM(3);
252 const int levels = GET_PARAM(4);
253 const int iters = GET_PARAM(5);
254
255 const cv::Mat frame0 = readImage(imagePair.first, useGray ? cv::IMREAD_GRAYSCALE : cv::IMREAD_COLOR);
256 ASSERT_FALSE(frame0.empty());
257
258 const cv::Mat frame1 = readImage(imagePair.second, useGray ? cv::IMREAD_GRAYSCALE : cv::IMREAD_COLOR);
259 ASSERT_FALSE(frame1.empty());
260
261 cv::Mat gray_frame;
262 if (useGray)
263 gray_frame = frame0;
264 else
265 cv::cvtColor(frame0, gray_frame, cv::COLOR_BGR2GRAY);
266
267 cv::Mat pts;
268 cv::goodFeaturesToTrack(gray_frame, pts, points, 0.01, 0.0);
269
270 if (PERF_RUN_GPU())
271 {
272 const cv::gpu::GpuMat d_pts(pts.reshape(2, 1));
273
274 cv::gpu::PyrLKOpticalFlow d_pyrLK;
275 d_pyrLK.winSize = cv::Size(winSize, winSize);
276 d_pyrLK.maxLevel = levels - 1;
277 d_pyrLK.iters = iters;
278
279 const cv::gpu::GpuMat d_frame0(frame0);
280 const cv::gpu::GpuMat d_frame1(frame1);
281 cv::gpu::GpuMat nextPts;
282 cv::gpu::GpuMat status;
283
284 TEST_CYCLE() d_pyrLK.sparse(d_frame0, d_frame1, d_pts, nextPts, status);
285
286 GPU_SANITY_CHECK(nextPts);
287 GPU_SANITY_CHECK(status);
288 }
289 else
290 {
291 cv::Mat nextPts;
292 cv::Mat status;
293
294 TEST_CYCLE()
295 {
296 cv::calcOpticalFlowPyrLK(frame0, frame1, pts, nextPts, status, cv::noArray(),
297 cv::Size(winSize, winSize), levels - 1,
298 cv::TermCriteria(cv::TermCriteria::COUNT + cv::TermCriteria::EPS, iters, 0.01));
299 }
300
301 CPU_SANITY_CHECK(nextPts);
302 CPU_SANITY_CHECK(status);
303 }
304}
