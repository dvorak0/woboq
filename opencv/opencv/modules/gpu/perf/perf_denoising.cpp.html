53// BilateralFilter
54
55DEF_PARAM_TEST(Sz_Depth_Cn_KernelSz, cv::Size, MatDepth, MatCn, int);
56
57PERF_TEST_P(Sz_Depth_Cn_KernelSz, Denoising_BilateralFilter,
58 Combine(GPU_DENOISING_IMAGE_SIZES,
59 Values(CV_8U, CV_32F),
60 GPU_CHANNELS_1_3,
61 Values(3, 5, 9)))
62{
63 declare.time(60.0);
64
65 const cv::Size size = GET_PARAM(0);
66 const int depth = GET_PARAM(1);
67 const int channels = GET_PARAM(2);
68 const int kernel_size = GET_PARAM(3);
69
70 const float sigma_color = 7;
71 const float sigma_spatial = 5;
72 const int borderMode = cv::BORDER_REFLECT101;
73
74 const int type = CV_MAKE_TYPE(depth, channels);
75
76 cv::Mat src(size, type);
77 declare.in(src, WARMUP_RNG);
78
79 if (PERF_RUN_GPU())
80 {
81 const cv::gpu::GpuMat d_src(src);
82 cv::gpu::GpuMat dst;
83
84 TEST_CYCLE() cv::gpu::bilateralFilter(d_src, dst, kernel_size, sigma_color, sigma_spatial, borderMode);
85
86 GPU_SANITY_CHECK(dst);
87 }
88 else
89 {
90 cv::Mat dst;
91
92 TEST_CYCLE() cv::bilateralFilter(src, dst, kernel_size, sigma_color, sigma_spatial, borderMode);
93
94 CPU_SANITY_CHECK(dst);
95 }
96}
