51// HOG
52
53DEF_PARAM_TEST_1(Image, string);
54
55PERF_TEST_P(Image, ObjDetect_HOG,
56 Values<string>("gpu/hog/road.png",
57 "gpu/caltech/image_00000009_0.png",
58 "gpu/caltech/image_00000032_0.png",
59 "gpu/caltech/image_00000165_0.png",
60 "gpu/caltech/image_00000261_0.png",
61 "gpu/caltech/image_00000469_0.png",
62 "gpu/caltech/image_00000527_0.png",
63 "gpu/caltech/image_00000574_0.png"))
64{
65 declare.time(300.0);
66
67 const cv::Mat img = readImage(GetParam(), cv::IMREAD_GRAYSCALE);
68 ASSERT_FALSE(img.empty());
69
70 if (PERF_RUN_GPU())
71 {
72 const cv::gpu::GpuMat d_img(img);
73 std::vector<cv::Rect> gpu_found_locations;
74
75 cv::gpu::HOGDescriptor d_hog;
76 d_hog.setSVMDetector(cv::gpu::HOGDescriptor::getDefaultPeopleDetector());
77
78 TEST_CYCLE() d_hog.detectMultiScale(d_img, gpu_found_locations);
79
80 SANITY_CHECK(gpu_found_locations);
81 }
82 else
83 {
84 std::vector<cv::Rect> cpu_found_locations;
85
86 cv::HOGDescriptor hog;
87 hog.setSVMDetector(cv::gpu::HOGDescriptor::getDefaultPeopleDetector());
88
89 TEST_CYCLE() hog.detectMultiScale(img, cpu_found_locations);
90
91 SANITY_CHECK(cpu_found_locations);
92 }
93}

96// HaarClassifier
97
98typedef pair<string, string> pair_string;
99DEF_PARAM_TEST_1(ImageAndCascade, pair_string);
100
101PERF_TEST_P(ImageAndCascade, ObjDetect_HaarClassifier,
102 Values<pair_string>(make_pair("gpu/haarcascade/group_1_640x480_VGA.pgm", "gpu/perf/haarcascade_frontalface_alt.xml")))
103{
104 const cv::Mat img = readImage(GetParam().first, cv::IMREAD_GRAYSCALE);
105 ASSERT_FALSE(img.empty());
106
107 if (PERF_RUN_GPU())
108 {
109 cv::gpu::CascadeClassifier_GPU d_cascade;
110 ASSERT_TRUE(d_cascade.load(perf::TestBase::getDataPath(GetParam().second)));
111
112 const cv::gpu::GpuMat d_img(img);
113 cv::gpu::GpuMat objects_buffer;
114 int detections_num = 0;
115
116 TEST_CYCLE() detections_num = d_cascade.detectMultiScale(d_img, objects_buffer);
117
118 std::vector<cv::Rect> gpu_rects(detections_num);
119 cv::Mat gpu_rects_mat(1, detections_num, cv::DataType<cv::Rect>::type, &gpu_rects[0]);
120 objects_buffer.colRange(0, detections_num).download(gpu_rects_mat);
121 cv::groupRectangles(gpu_rects, 3, 0.2);
122 SANITY_CHECK(gpu_rects);
123 }
124 else
125 {
126 cv::CascadeClassifier cascade;
127 ASSERT_TRUE(cascade.load(perf::TestBase::getDataPath("gpu/perf/haarcascade_frontalface_alt.xml")));
128
129 std::vector<cv::Rect> cpu_rects;
130
131 TEST_CYCLE() cascade.detectMultiScale(img, cpu_rects);
132
133 SANITY_CHECK(cpu_rects);
134 }
135}
