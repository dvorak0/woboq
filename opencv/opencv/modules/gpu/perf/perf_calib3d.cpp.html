53typedef std::tr1::tuple<string, string> pair_string;
54DEF_PARAM_TEST_1(ImagePair, pair_string);
55
56PERF_TEST_P(ImagePair, Calib3D_StereoBM,
57 Values(pair_string("gpu/perf/aloe.png", "gpu/perf/aloeR.png")))
58{
59 declare.time(300.0);
60
61 const cv::Mat imgLeft = readImage(GET_PARAM(0), cv::IMREAD_GRAYSCALE);
62 ASSERT_FALSE(imgLeft.empty());
63
64 const cv::Mat imgRight = readImage(GET_PARAM(1), cv::IMREAD_GRAYSCALE);
65 ASSERT_FALSE(imgRight.empty());
66
67 const int preset = 0;
68 const int ndisp = 256;
69
70 if (PERF_RUN_GPU())
71 {
72 cv::gpu::StereoBM_GPU d_bm(preset, ndisp);
73
74 const cv::gpu::GpuMat d_imgLeft(imgLeft);
75 const cv::gpu::GpuMat d_imgRight(imgRight);
76 cv::gpu::GpuMat dst;
77
78 TEST_CYCLE() d_bm(d_imgLeft, d_imgRight, dst);
79
80 GPU_SANITY_CHECK(dst);
81 }
82 else
83 {
84 cv::StereoBM bm(preset, ndisp);
85
86 cv::Mat dst;
87
88 TEST_CYCLE() bm(imgLeft, imgRight, dst);
89
90 CPU_SANITY_CHECK(dst);
91 }
92}

259//////////////////////////////////////////////////////////////////////
260// SolvePnPRansac
261
262PERF_TEST_P(Count, Calib3D_SolvePnPRansac,
263 Values(5000, 10000, 20000))
264{
265 declare.time(10.0);
266
267 const int count = GetParam();
268
269 cv::Mat object(1, count, CV_32FC3);
270 declare.in(object, WARMUP_RNG);
271
272 cv::Mat camera_mat(3, 3, CV_32FC1);
273 cv::randu(camera_mat, 0.5, 1);
274 camera_mat.at<float>(0, 1) = 0.f;
275 camera_mat.at<float>(1, 0) = 0.f;
276 camera_mat.at<float>(2, 0) = 0.f;
277 camera_mat.at<float>(2, 1) = 0.f;
278
279 const cv::Mat dist_coef(1, 8, CV_32F, cv::Scalar::all(0));
280
281 cv::Mat rvec_gold(1, 3, CV_32FC1);
282 cv::randu(rvec_gold, 0, 1);
283
284 cv::Mat tvec_gold(1, 3, CV_32FC1);
285 cv::randu(tvec_gold, 0, 1);
286
287 std::vector<cv::Point2f> image_vec;
288 cv::projectPoints(object, rvec_gold, tvec_gold, camera_mat, dist_coef, image_vec);
289
290 const cv::Mat image(1, count, CV_32FC2, &image_vec[0]);
291
292 cv::Mat rvec;
293 cv::Mat tvec;
294
295 if (PERF_RUN_GPU())
296 {
297 TEST_CYCLE() cv::gpu::solvePnPRansac(object, image, camera_mat, dist_coef, rvec, tvec);
298
299 GPU_SANITY_CHECK(rvec, 1e-3);
300 GPU_SANITY_CHECK(tvec, 1e-3);
301 }
302 else
303 {
304 TEST_CYCLE() cv::solvePnPRansac(object, image, camera_mat, dist_coef, rvec, tvec);
305
306 CPU_SANITY_CHECK(rvec, 1e-6);
307 CPU_SANITY_CHECK(tvec, 1e-6);
308 }
309}
