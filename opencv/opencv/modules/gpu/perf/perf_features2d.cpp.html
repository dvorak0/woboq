53DEF_PARAM_TEST(Image_Threshold_NonMaxSuppression, string, int, bool);
54
55PERF_TEST_P(Image_Threshold_NonMaxSuppression, Features2D_FAST,
56 Combine(Values<string>("gpu/perf/aloe.png"),
57 Values(20),
58 Bool()))
59{
60 const cv::Mat img = readImage(GET_PARAM(0), cv::IMREAD_GRAYSCALE);
61 ASSERT_FALSE(img.empty());
62
63 const int threshold = GET_PARAM(1);
64 const bool nonMaxSuppersion = GET_PARAM(2);
65
66 if (PERF_RUN_GPU())
67 {
68 cv::gpu::FAST_GPU d_fast(threshold, nonMaxSuppersion, 0.5);
69
70 const cv::gpu::GpuMat d_img(img);
71 cv::gpu::GpuMat d_keypoints;
72
73 TEST_CYCLE() d_fast(d_img, cv::gpu::GpuMat(), d_keypoints);
74
75 std::vector<cv::KeyPoint> gpu_keypoints;
76 d_fast.downloadKeypoints(d_keypoints, gpu_keypoints);
77
78 sortKeyPoints(gpu_keypoints);
79
80 SANITY_CHECK_KEYPOINTS(gpu_keypoints);
81 }
82 else
83 {
84 std::vector<cv::KeyPoint> cpu_keypoints;
85
86 TEST_CYCLE() cv::FAST(img, cpu_keypoints, threshold, nonMaxSuppersion);
87
88 SANITY_CHECK_KEYPOINTS(cpu_keypoints);
89 }
90}

93// ORB
94
95DEF_PARAM_TEST(Image_NFeatures, string, int);
96
97PERF_TEST_P(Image_NFeatures, Features2D_ORB,
98 Combine(Values<string>("gpu/perf/aloe.png"),
99 Values(4000)))
100{
101 declare.time(300.0);
102
103 const cv::Mat img = readImage(GET_PARAM(0), cv::IMREAD_GRAYSCALE);
104 ASSERT_FALSE(img.empty());
105
106 const int nFeatures = GET_PARAM(1);
107
108 if (PERF_RUN_GPU())
109 {
110 cv::gpu::ORB_GPU d_orb(nFeatures);
111
112 const cv::gpu::GpuMat d_img(img);
113 cv::gpu::GpuMat d_keypoints, d_descriptors;
114
115 TEST_CYCLE() d_orb(d_img, cv::gpu::GpuMat(), d_keypoints, d_descriptors);
116
117 std::vector<cv::KeyPoint> gpu_keypoints;
118 d_orb.downloadKeyPoints(d_keypoints, gpu_keypoints);
119
120 cv::Mat gpu_descriptors(d_descriptors);
121
122 gpu_keypoints.resize(10);
123 gpu_descriptors = gpu_descriptors.rowRange(0, 10);
124
125 sortKeyPoints(gpu_keypoints, gpu_descriptors);
126
127 SANITY_CHECK_KEYPOINTS(gpu_keypoints, 1e-10);
128 SANITY_CHECK(gpu_descriptors);
129 }
130 else
131 {
132 cv::ORB orb(nFeatures);
133
134 std::vector<cv::KeyPoint> cpu_keypoints;
135 cv::Mat cpu_descriptors;
136
137 TEST_CYCLE() orb(img, cv::noArray(), cpu_keypoints, cpu_descriptors);
138
139 SANITY_CHECK_KEYPOINTS(cpu_keypoints);
140 SANITY_CHECK(cpu_descriptors);
141 }
142}

147DEF_PARAM_TEST(DescSize_Norm, int, NormType);
148
149#ifdef OPENCV_TINY_GPU_MODULE
150PERF_TEST_P(DescSize_Norm, Features2D_BFMatch, Combine(
151 Values(64, 128, 256),
152 Values(NormType(cv::NORM_L2), NormType(cv::NORM_HAMMING))
153))
154#else
155PERF_TEST_P(DescSize_Norm, Features2D_BFMatch, Combine(
156 Values(64, 128, 256),
157 Values(NormType(cv::NORM_L1), NormType(cv::NORM_L2), NormType(cv::NORM_HAMMING))
158))
159#endif
160{
161 declare.time(20.0);
162
163 const int desc_size = GET_PARAM(0);
164 const int normType = GET_PARAM(1);
165
166 const int type = normType == cv::NORM_HAMMING ? CV_8U : CV_32F;
167
168 cv::Mat query(3000, desc_size, type);
169 declare.in(query, WARMUP_RNG);
170
171 cv::Mat train(3000, desc_size, type);
172 declare.in(train, WARMUP_RNG);
173
174 if (PERF_RUN_GPU())
175 {
176 cv::gpu::BFMatcher_GPU d_matcher(normType);
177
178 const cv::gpu::GpuMat d_query(query);
179 const cv::gpu::GpuMat d_train(train);
180 cv::gpu::GpuMat d_trainIdx, d_distance;
181
182 TEST_CYCLE() d_matcher.matchSingle(d_query, d_train, d_trainIdx, d_distance);
183
184 std::vector<cv::DMatch> gpu_matches;
185 d_matcher.matchDownload(d_trainIdx, d_distance, gpu_matches);
186
187 SANITY_CHECK_MATCHES(gpu_matches);
188 }
189 else
190 {
191 cv::BFMatcher matcher(normType);
192
193 std::vector<cv::DMatch> cpu_matches;
194
195 TEST_CYCLE() matcher.match(query, train, cpu_matches);
196
197 SANITY_CHECK_MATCHES(cpu_matches);
198 }
199}

204static void toOneRowMatches(const std::vector< std::vector<cv::DMatch> >& src, std::vector<cv::DMatch>& dst)
205{
206 dst.clear();
207 for (size_t i = 0; i < src.size(); ++i)
208 for (size_t j = 0; j < src[i].size(); ++j)
209 dst.push_back(src[i][j]);
210}

212DEF_PARAM_TEST(DescSize_K_Norm, int, int, NormType);
213
214#ifdef OPENCV_TINY_GPU_MODULE
215PERF_TEST_P(DescSize_K_Norm, Features2D_BFKnnMatch, Combine(
216 Values(64, 128, 256),
217 Values(2, 3),
218 Values(NormType(cv::NORM_L2))
219))
220#else
221PERF_TEST_P(DescSize_K_Norm, Features2D_BFKnnMatch, Combine(
222 Values(64, 128, 256),
223 Values(2, 3),
224 Values(NormType(cv::NORM_L1), NormType(cv::NORM_L2))
225))
226#endif
227{
228 declare.time(30.0);
229
230 const int desc_size = GET_PARAM(0);
231 const int k = GET_PARAM(1);
232 const int normType = GET_PARAM(2);
233
234 const int type = normType == cv::NORM_HAMMING ? CV_8U : CV_32F;
235
236 cv::Mat query(3000, desc_size, type);
237 declare.in(query, WARMUP_RNG);
238
239 cv::Mat train(3000, desc_size, type);
240 declare.in(train, WARMUP_RNG);
241
242 if (PERF_RUN_GPU())
243 {
244 cv::gpu::BFMatcher_GPU d_matcher(normType);
245
246 const cv::gpu::GpuMat d_query(query);
247 const cv::gpu::GpuMat d_train(train);
248 cv::gpu::GpuMat d_trainIdx, d_distance, d_allDist;
249
250 TEST_CYCLE() d_matcher.knnMatchSingle(d_query, d_train, d_trainIdx, d_distance, d_allDist, k);
251
252 std::vector< std::vector<cv::DMatch> > matchesTbl;
253 d_matcher.knnMatchDownload(d_trainIdx, d_distance, matchesTbl);
254
255 std::vector<cv::DMatch> gpu_matches;
256 toOneRowMatches(matchesTbl, gpu_matches);
257
258 SANITY_CHECK_MATCHES(gpu_matches);
259 }
260 else
261 {
262 cv::BFMatcher matcher(normType);
263
264 std::vector< std::vector<cv::DMatch> > matchesTbl;
265
266 TEST_CYCLE() matcher.knnMatch(query, train, matchesTbl, k);
267
268 std::vector<cv::DMatch> cpu_matches;
269 toOneRowMatches(matchesTbl, cpu_matches);
270
271 SANITY_CHECK_MATCHES(cpu_matches);
272 }
273}
