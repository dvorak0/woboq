58
59PERF_TEST_P(Sz_Depth_Cn, Core_Merge,
60 Combine(GPU_TYPICAL_MAT_SIZES,
61 ARITHM_MAT_DEPTH,
62 Values(2, 3, 4)))
63{
64 const cv::Size size = GET_PARAM(0);
65 const int depth = GET_PARAM(1);
66 const int channels = GET_PARAM(2);
67
68 std::vector<cv::Mat> src(channels);
69 for (int i = 0; i < channels; ++i)
70 {
71 src[i].create(size, depth);
72 declare.in(src[i], WARMUP_RNG);
73 }
74
75 if (PERF_RUN_GPU())
76 {
77 std::vector<cv::gpu::GpuMat> d_src(channels);
78 for (int i = 0; i < channels; ++i)
79 d_src[i].upload(src[i]);
80
81 cv::gpu::GpuMat dst;
82
83 TEST_CYCLE() cv::gpu::merge(d_src, dst);
84
85 GPU_SANITY_CHECK(dst, 1e-10);
86 }
87 else
88 {
89 cv::Mat dst;
90
91 TEST_CYCLE() cv::merge(src, dst);
92
93 CPU_SANITY_CHECK(dst);
94 }
95}
