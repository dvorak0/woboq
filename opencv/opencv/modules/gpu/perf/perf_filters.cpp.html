51// Blur
52
53DEF_PARAM_TEST(Sz_Type_KernelSz, cv::Size, MatType, int);
54
55PERF_TEST_P(Sz_Type_KernelSz, Filters_Blur,
56 Combine(GPU_TYPICAL_MAT_SIZES,
57 Values(CV_8UC1, CV_8UC4),
58 Values(3, 5, 7)))
59{
60 declare.time(20.0);
61
62 const cv::Size size = GET_PARAM(0);
63 const int type = GET_PARAM(1);
64 const int ksize = GET_PARAM(2);
65
66 cv::Mat src(size, type);
67 declare.in(src, WARMUP_RNG);
68
69 if (PERF_RUN_GPU())
70 {
71 const cv::gpu::GpuMat d_src(src);
72 cv::gpu::GpuMat dst;
73
74 TEST_CYCLE() cv::gpu::blur(d_src, dst, cv::Size(ksize, ksize));
75
76 GPU_SANITY_CHECK(dst, 1);
77 }
78 else
79 {
80 cv::Mat dst;
81
82 TEST_CYCLE() cv::blur(src, dst, cv::Size(ksize, ksize));
83
84 CPU_SANITY_CHECK(dst);
85 }
86}
