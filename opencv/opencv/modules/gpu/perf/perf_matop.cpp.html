51// SetTo
52
53PERF_TEST_P(Sz_Depth_Cn, MatOp_SetTo,
54 Combine(GPU_TYPICAL_MAT_SIZES,
55 Values(CV_8U, CV_16U, CV_32F, CV_64F),
56 GPU_CHANNELS_1_3_4))
57{
58 const cv::Size size = GET_PARAM(0);
59 const int depth = GET_PARAM(1);
60 const int channels = GET_PARAM(2);
61
62 const int type = CV_MAKE_TYPE(depth, channels);
63
64 const cv::Scalar val(1, 2, 3, 4);
65
66 if (PERF_RUN_GPU())
67 {
68 cv::gpu::GpuMat dst(size, type);
69
70 TEST_CYCLE() dst.setTo(val);
71
72 GPU_SANITY_CHECK(dst);
73 }
74 else
75 {
76 cv::Mat dst(size, type);
77
78 TEST_CYCLE() dst.setTo(val);
79
80 CPU_SANITY_CHECK(dst);
81 }
82}

84//////////////////////////////////////////////////////////////////////
85// SetToMasked
86
87PERF_TEST_P(Sz_Depth_Cn, MatOp_SetToMasked,
88 Combine(GPU_TYPICAL_MAT_SIZES,
89 Values(CV_8U, CV_16U, CV_32F, CV_64F),
90 GPU_CHANNELS_1_3_4))
91{
92 const cv::Size size = GET_PARAM(0);
93 const int depth = GET_PARAM(1);
94 const int channels = GET_PARAM(2);
95
96 const int type = CV_MAKE_TYPE(depth, channels);
97
98 cv::Mat src(size, type);
99 cv::Mat mask(size, CV_8UC1);
100 declare.in(src, mask, WARMUP_RNG);
101
102 const cv::Scalar val(1, 2, 3, 4);
103
104 if (PERF_RUN_GPU())
105 {
106 cv::gpu::GpuMat dst(src);
107 const cv::gpu::GpuMat d_mask(mask);
108
109 TEST_CYCLE() dst.setTo(val, d_mask);
110
111 GPU_SANITY_CHECK(dst, 1e-10);
112 }
113 else
114 {
115 cv::Mat dst = src;
116
117 TEST_CYCLE() dst.setTo(val, mask);
118
119 CPU_SANITY_CHECK(dst);
120 }
121}

124// CopyToMasked
125
126PERF_TEST_P(Sz_Depth_Cn, MatOp_CopyToMasked,
127 Combine(GPU_TYPICAL_MAT_SIZES,
128 Values(CV_8U, CV_16U, CV_32F, CV_64F),
129 GPU_CHANNELS_1_3_4))
130{
131 const cv::Size size = GET_PARAM(0);
132 const int depth = GET_PARAM(1);
133 const int channels = GET_PARAM(2);
134
135 const int type = CV_MAKE_TYPE(depth, channels);
136
137 cv::Mat src(size, type);
138 cv::Mat mask(size, CV_8UC1);
139 declare.in(src, mask, WARMUP_RNG);
140
141 if (PERF_RUN_GPU())
142 {
143 const cv::gpu::GpuMat d_src(src);
144 const cv::gpu::GpuMat d_mask(mask);
145 cv::gpu::GpuMat dst(d_src.size(), d_src.type(), cv::Scalar::all(0));
146
147 TEST_CYCLE() d_src.copyTo(dst, d_mask);
148
149 GPU_SANITY_CHECK(dst, 1e-10);
150 }
151 else
152 {
153 cv::Mat dst(src.size(), src.type(), cv::Scalar::all(0));
154
155 TEST_CYCLE() src.copyTo(dst, mask);
156
157 CPU_SANITY_CHECK(dst);
158 }
159}

162// ConvertTo
163
164DEF_PARAM_TEST(Sz_2Depth, cv::Size, MatDepth, MatDepth);
165
166PERF_TEST_P(Sz_2Depth, MatOp_ConvertTo,
167 Combine(GPU_TYPICAL_MAT_SIZES,
168 Values(CV_8U, CV_16U, CV_32F, CV_64F),
169 Values(CV_8U, CV_16U, CV_32F, CV_64F)))
170{
171 const cv::Size size = GET_PARAM(0);
172 const int depth1 = GET_PARAM(1);
173 const int depth2 = GET_PARAM(2);
174
175 cv::Mat src(size, depth1);
176 declare.in(src, WARMUP_RNG);
177
178 const double a = 0.5;
179 const double b = 1.0;
180
181 if (PERF_RUN_GPU())
182 {
183 const cv::gpu::GpuMat d_src(src);
184 cv::gpu::GpuMat dst;
185
186 TEST_CYCLE() d_src.convertTo(dst, depth2, a, b);
187
188 GPU_SANITY_CHECK(dst, 1e-10);
189 }
190 else
191 {
192 cv::Mat dst;
193
194 TEST_CYCLE() src.convertTo(dst, depth2, a, b);
195
196 CPU_SANITY_CHECK(dst);
197 }
198}
