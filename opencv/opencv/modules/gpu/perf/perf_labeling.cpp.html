52struct GreedyLabeling
53{
54 struct dot
55 {
56 int x;
57 int y;
58
59 static dot make(int i, int j)
60 {
61 dot d; d.x = i; d.y = j;
62 return d;
63 }
64 };
65
66 struct InInterval
67 {
68 InInterval(const int& _lo, const int& _hi) : lo(-_lo), hi(_hi) {}
69 const int lo, hi;
70
71 bool operator() (const unsigned char a, const unsigned char b) const
72 {
73 int d = a - b;
74 return lo <= d && d <= hi;
75 }
76
77 private:
78 InInterval& operator=(const InInterval&);
79
80
81 };
82
83 GreedyLabeling(cv::Mat img)
84 : image(img), _labels(image.size(), CV_32SC1, cv::Scalar::all(-1)) {stack = new dot[image.cols * image.rows];}
85
86 ~GreedyLabeling(){delete[] stack;}
87
88 void operator() (cv::Mat labels) const
89 {
90 labels.setTo(cv::Scalar::all(-1));
91 InInterval inInt(0, 2);
92 int cc = -1;
93
94 int* dist_labels = (int*)labels.data;
95 int pitch = static_cast<int>(labels.step1());
96
97 unsigned char* source = (unsigned char*)image.data;
98 int width = image.cols;
99 int height = image.rows;
100
101 for (int j = 0; j < image.rows; ++j)
102 for (int i = 0; i < image.cols; ++i)
103 {
104 if (dist_labels[j * pitch + i] != -1) continue;
105
106 dot* top = stack;
107 dot p = dot::make(i, j);
108 cc++;
109
110 dist_labels[j * pitch + i] = cc;
111
112 while (top >= stack)
113 {
114 int* dl = &dist_labels[p.y * pitch + p.x];
115 unsigned char* sp = &source[p.y * image.step1() + p.x];
116
117 dl[0] = cc;
118
119 //right
120 if( p.x < (width - 1) && dl[ +1] == -1 && inInt(sp[0], sp[+1]))
121 *top++ = dot::make(p.x + 1, p.y);
122
123 //left
124 if( p.x > 0 && dl[-1] == -1 && inInt(sp[0], sp[-1]))
125 *top++ = dot::make(p.x - 1, p.y);
126
127 //bottom
128 if( p.y < (height - 1) && dl[+pitch] == -1 && inInt(sp[0], sp[+image.step1()]))
129 *top++ = dot::make(p.x, p.y + 1);
130
131 //top
132 if( p.y > 0 && dl[-pitch] == -1 && inInt(sp[0], sp[-static_cast<int>(image.step1())]))
133 *top++ = dot::make(p.x, p.y - 1);
134
135 p = *--top;
136 }
137 }
138 }
139
140 cv::Mat image;
141 cv::Mat _labels;
142 dot* stack;
143};

145PERF_TEST_P(Image, DISABLED_Labeling_ConnectivityMask,
146 Values<string>("gpu/labeling/aloe-disp.png"))
147{
148 declare.time(1.0);
149
150 const cv::Mat image = readImage(GetParam(), cv::IMREAD_GRAYSCALE);
151 ASSERT_FALSE(image.empty());
152
153 if (PERF_RUN_GPU())
154 {
155 cv::gpu::GpuMat d_image(image);
156 cv::gpu::GpuMat mask;
157
158 TEST_CYCLE() cv::gpu::connectivityMask(d_image, mask, cv::Scalar::all(0), cv::Scalar::all(2));
159
160 GPU_SANITY_CHECK(mask);
161 }
162 else
163 {
164 FAIL_NO_CPU();
165 }
166}

168PERF_TEST_P(Image, DISABLED_Labeling_ConnectedComponents,
169 Values<string>("gpu/labeling/aloe-disp.png"))
170{
171 declare.time(1.0);
172
173 const cv::Mat image = readImage(GetParam(), cv::IMREAD_GRAYSCALE);
174 ASSERT_FALSE(image.empty());
175
176 if (PERF_RUN_GPU())
177 {
178 cv::gpu::GpuMat d_mask;
179 cv::gpu::connectivityMask(cv::gpu::GpuMat(image), d_mask, cv::Scalar::all(0), cv::Scalar::all(2));
180
181 cv::gpu::GpuMat components;
182
183 TEST_CYCLE() cv::gpu::labelComponents(d_mask, components);
184
185 GPU_SANITY_CHECK(components);
186 }
187 else
188 {
189 GreedyLabeling host(image);
190
191 TEST_CYCLE() host(host._labels);
192
193 cv::Mat components = host._labels;
194 CPU_SANITY_CHECK(components);
195 }
