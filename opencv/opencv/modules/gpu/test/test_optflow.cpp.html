66GPU_TEST_P(BroxOpticalFlow, Regression)
67{
68 cv::Mat frame0 = readImageType("opticalflow/frame0.png", CV_32FC1);

70
71 cv::Mat frame1 = readImageType("opticalflow/frame1.png", CV_32FC1);

73
74 cv::gpu::BroxOpticalFlow brox(0.197f /*alpha*/, 50.0f /*gamma*/, 0.8f /*scale_factor*/,
75 10 /*inner_iterations*/, 77 /*outer_iterations*/, 10 /*solver_iterations*/);
76
77 cv::gpu::GpuMat u;
78 cv::gpu::GpuMat v;
79 brox(loadMat(frame0), loadMat(frame1), u, v);
80
81 std::string fname(cvtest::TS::ptr()->get_data_path());
82 if (devInfo.majorVersion() >= 2)
83 fname += "opticalflow/brox_optical_flow_cc20.bin";
84 else
85 fname += "opticalflow/brox_optical_flow.bin";
86
87#ifndef BROX_DUMP
88 std::ifstream f(fname.c_str(), std::ios_base::binary);
89
90 int rows, cols;
91
92 f.read((char*) &rows, sizeof(rows));
93 f.read((char*) &cols, sizeof(cols));
94
95 cv::Mat u_gold(rows, cols, CV_32FC1);
96
97 for (int i = 0; i < u_gold.rows; ++i)
98 f.read(u_gold.ptr<char>(i), u_gold.cols * sizeof(float));
99
100 cv::Mat v_gold(rows, cols, CV_32FC1);
101
102 for (int i = 0; i < v_gold.rows; ++i)
103 f.read(v_gold.ptr<char>(i), v_gold.cols * sizeof(float));
104
105 EXPECT_MAT_SIMILAR(u_gold, u, 1e-3);
106 EXPECT_MAT_SIMILAR(v_gold, v, 1e-3);
107#else
108 std::ofstream f(fname.c_str(), std::ios_base::binary);
109
110 f.write((char*) &u.rows, sizeof(u.rows));
111 f.write((char*) &u.cols, sizeof(u.cols));
112
113 cv::Mat h_u(u);
114 cv::Mat h_v(v);
115
116 for (int i = 0; i < u.rows; ++i)
117 f.write(h_u.ptr<char>(i), u.cols * sizeof(float));
118
119 for (int i = 0; i < v.rows; ++i)
120 f.write(h_v.ptr<char>(i), v.cols * sizeof(float));

175GPU_TEST_P(GoodFeaturesToTrack, Accuracy)
176{
177 cv::Mat image = readImage("opticalflow/frame0.png", cv::IMREAD_GRAYSCALE);
178 ASSERT_FALSE(image.empty());
179
180 int maxCorners = 1000;
181 double qualityLevel = 0.01;
182
183 cv::gpu::GoodFeaturesToTrackDetector_GPU detector(maxCorners, qualityLevel, minDistance);
184
185 cv::gpu::GpuMat d_pts;
186 detector(loadMat(image), d_pts);
187
188 ASSERT_FALSE(d_pts.empty());
189
190 std::vector<cv::Point2f> pts(d_pts.cols);
191 cv::Mat pts_mat(1, d_pts.cols, CV_32FC2, (void*) &pts[0]);
192 d_pts.download(pts_mat);
193
194 std::vector<cv::Point2f> pts_gold;
195 cv::goodFeaturesToTrack(image, pts_gold, maxCorners, qualityLevel, minDistance);
196
197 ASSERT_EQ(pts_gold.size(), pts.size());
198
199 size_t mistmatch = 0;
200 for (size_t i = 0; i < pts.size(); ++i)
201 {
202 cv::Point2i a = pts_gold[i];
203 cv::Point2i b = pts[i];
204
205 bool eq = std::abs(a.x - b.x) < 1 && std::abs(a.y - b.y) < 1;
206
207 if (!eq)
208 ++mistmatch;
209 }
210
211 double bad_ratio = static_cast<double>(mistmatch) / pts.size();
212
213 ASSERT_LE(bad_ratio, 0.01);
214}

257GPU_TEST_P(PyrLKOpticalFlow, Sparse)
258{
259 cv::Mat frame0 = readImage("opticalflow/frame0.png", useGray ? cv::IMREAD_GRAYSCALE : cv::IMREAD_COLOR);
260 ASSERT_FALSE(frame0.empty());
261
262 cv::Mat frame1 = readImage("opticalflow/frame1.png", useGray ? cv::IMREAD_GRAYSCALE : cv::IMREAD_COLOR);
263 ASSERT_FALSE(frame1.empty());
264
265 cv::Mat gray_frame;
266 if (useGray)
267 gray_frame = frame0;
268 else
269 cv::cvtColor(frame0, gray_frame, cv::COLOR_BGR2GRAY);
270
271 std::vector<cv::Point2f> pts;
272 cv::goodFeaturesToTrack(gray_frame, pts, 1000, 0.01, 0.0);
273
274 cv::gpu::GpuMat d_pts;
275 cv::Mat pts_mat(1, (int) pts.size(), CV_32FC2, (void*) &pts[0]);
276 d_pts.upload(pts_mat);
277
278 cv::gpu::PyrLKOpticalFlow pyrLK;
279
280 cv::gpu::GpuMat d_nextPts;
281 cv::gpu::GpuMat d_status;
282 pyrLK.sparse(loadMat(frame0), loadMat(frame1), d_pts, d_nextPts, d_status);
283
284 std::vector<cv::Point2f> nextPts(d_nextPts.cols);
285 cv::Mat nextPts_mat(1, d_nextPts.cols, CV_32FC2, (void*) &nextPts[0]);
286 d_nextPts.download(nextPts_mat);
287
288 std::vector<unsigned char> status(d_status.cols);
289 cv::Mat status_mat(1, d_status.cols, CV_8UC1, (void*) &status[0]);
290 d_status.download(status_mat);
291
292 std::vector<cv::Point2f> nextPts_gold;
293 std::vector<unsigned char> status_gold;
294 cv::calcOpticalFlowPyrLK(frame0, frame1, pts, nextPts_gold, status_gold, cv::noArray());
295
296 ASSERT_EQ(nextPts_gold.size(), nextPts.size());
297 ASSERT_EQ(status_gold.size(), status.size());
298
299 size_t mistmatch = 0;
300 for (size_t i = 0; i < nextPts.size(); ++i)
301 {
302 cv::Point2i a = nextPts[i];
303 cv::Point2i b = nextPts_gold[i];
304
305 if (status[i] != status_gold[i])
306 {
307 ++mistmatch;
308 continue;
309 }
310
311 if (status[i])
312 {
313 bool eq = std::abs(a.x - b.x) <= 1 && std::abs(a.y - b.y) <= 1;
314
315 if (!eq)
316 ++mistmatch;
317 }
318 }
319
320 double bad_ratio = static_cast<double>(mistmatch) / nextPts.size();
321
322 ASSERT_LE(bad_ratio, 0.01);
323}

379 cv::Mat flow;

381 {
382 cv::Mat flowxy[] = {cv::Mat(d_flowx), cv::Mat(d_flowy)};
383 cv::merge(flowxy, 2, flow);
384
385 farn.flags |= cv::OPTFLOW_USE_INITIAL_FLOW;
386 farn(loadMat(frame0), loadMat(frame1), d_flowx, d_flowy);
387 }
388
389 cv::calcOpticalFlowFarneback(
390 frame0, frame1, flow, farn.pyrScale, farn.numLevels, farn.winSize,
391 farn.numIters, farn.polyN, farn.polySigma, farn.flags);
392
393 std::vector<cv::Mat> flowxy;

515 void FastOpticalFlowBM_gold(const cv::Mat_<uchar>& I0, const cv::Mat_<uchar>& I1, cv::Mat_<float>& velx, cv::Mat_<float>& vely, int search_window, int block_window)
516 {
517 velx.create(I0.size());
518 vely.create(I0.size());
519
520 int search_radius = search_window / 2;
521 int block_radius = block_window / 2;
522
523 for (int y = 0; y < I0.rows; ++y)
524 {
525 for (int x = 0; x < I0.cols; ++x)
526 {
527 int bestDist = std::numeric_limits<int>::max();
528 int bestDx = 0;
529 int bestDy = 0;
530
531 for (int dy = -search_radius; dy <= search_radius; ++dy)
532 {
533 for (int dx = -search_radius; dx <= search_radius; ++dx)
534 {
535 int dist = 0;
536
537 for (int by = -block_radius; by <= block_radius; ++by)
538 {
539 for (int bx = -block_radius; bx <= block_radius; ++bx)
540 {
541 int I0_val = I0(cv::borderInterpolate(y + by, I0.rows, cv::BORDER_DEFAULT), cv::borderInterpolate(x + bx, I0.cols, cv::BORDER_DEFAULT));
542 int I1_val = I1(cv::borderInterpolate(y + dy + by, I0.rows, cv::BORDER_DEFAULT), cv::borderInterpolate(x + dx + bx, I0.cols, cv::BORDER_DEFAULT));
543
544 dist += std::abs(I0_val - I1_val);
545 }
546 }
547
548 if (dist < bestDist)
549 {
550 bestDist = dist;
551 bestDx = dx;
552 bestDy = dy;
553 }
554 }
555 }
556
557 velx(y, x) = (float) bestDx;
558 vely(y, x) = (float) bestDy;
559 }
560 }
561 }
562
563 double calc_rmse(const cv::Mat_<float>& flow1, const cv::Mat_<float>& flow2)
564 {
565 double sum = 0.0;
566
567 for (int y = 0; y < flow1.rows; ++y)
568 {
569 for (int x = 0; x < flow1.cols; ++x)
570 {
571 double diff = flow1(y, x) - flow2(y, x);
572 sum += diff * diff;
573 }
574 }
575
576 return std::sqrt(sum / flow1.size().area());
577 }
578}

584GPU_TEST_P(FastOpticalFlowBM, Accuracy)
585{
586 const double MAX_RMSE = 0.6;
587
588 int search_window = 15;
589 int block_window = 5;
590
591 cv::gpu::DeviceInfo devInfo = GetParam();
592 cv::gpu::setDevice(devInfo.deviceID());
593
594 cv::Mat frame0 = readImage("opticalflow/rubberwhale1.png", cv::IMREAD_GRAYSCALE);
595 ASSERT_FALSE(frame0.empty());
596
597 cv::Mat frame1 = readImage("opticalflow/rubberwhale2.png", cv::IMREAD_GRAYSCALE);
598 ASSERT_FALSE(frame1.empty());
599
600 cv::Size smallSize(320, 240);
601 cv::Mat frame0_small;
602 cv::Mat frame1_small;
603
604 cv::resize(frame0, frame0_small, smallSize);
605 cv::resize(frame1, frame1_small, smallSize);
606
607 cv::gpu::GpuMat d_flowx;
608 cv::gpu::GpuMat d_flowy;
609 cv::gpu::FastOpticalFlowBM fastBM;
610
611 fastBM(loadMat(frame0_small), loadMat(frame1_small), d_flowx, d_flowy, search_window, block_window);
612
613 cv::Mat_<float> flowx;
614 cv::Mat_<float> flowy;
615 FastOpticalFlowBM_gold(frame0_small, frame1_small, flowx, flowy, search_window, block_window);
616
617 double err;
618
619 err = calc_rmse(flowx, cv::Mat(d_flowx));
620 EXPECT_LE(err, MAX_RMSE);
621
622 err = calc_rmse(flowy, cv::Mat(d_flowy));
623 EXPECT_LE(err, MAX_RMSE);
624}
