50PARAM_TEST_CASE(VideoReader, cv::gpu::DeviceInfo, std::string)
51{
52 cv::gpu::DeviceInfo devInfo;
53 std::string inputFile;
54
55 virtual void SetUp()
56 {
57 devInfo = GET_PARAM(0);
58 inputFile = GET_PARAM(1);
59
60 cv::gpu::setDevice(devInfo.deviceID());
61
62 inputFile = std::string(cvtest::TS::ptr()->get_data_path()) + "video/" + inputFile;
63 }
64};
65
66GPU_TEST_P(VideoReader, Regression)
67{
68 cv::gpu::VideoReader_GPU reader(inputFile);
69 ASSERT_TRUE(reader.isOpened());
70
71 cv::gpu::GpuMat frame;
72
73 for (int i = 0; i < 10; ++i)
74 {
75 ASSERT_TRUE(reader.read(frame));
76 ASSERT_FALSE(frame.empty());
77 }
78
79 reader.close();
80 ASSERT_FALSE(reader.isOpened());
81}
82
83INSTANTIATE_TEST_CASE_P(GPU_Video, VideoReader, testing::Combine(
84 ALL_DEVICES,
85 testing::Values(std::string("768x576.avi"), std::string("1920x1080.avi"))));

92PARAM_TEST_CASE(VideoWriter, cv::gpu::DeviceInfo, std::string)
93{
94 cv::gpu::DeviceInfo devInfo;
95 std::string inputFile;
96
97 virtual void SetUp()
98 {
99 devInfo = GET_PARAM(0);
100 inputFile = GET_PARAM(1);
101
102 cv::gpu::setDevice(devInfo.deviceID());
103
104 inputFile = std::string(cvtest::TS::ptr()->get_data_path()) + std::string("video/") + inputFile;
105 }
106};
107
108GPU_TEST_P(VideoWriter, Regression)
109{
110 std::string outputFile = cv::tempfile(".avi");
111 const double FPS = 25.0;
112
113 cv::VideoCapture reader(inputFile);
114 ASSERT_TRUE(reader.isOpened());
115
116 cv::gpu::VideoWriter_GPU d_writer;
117
118 cv::Mat frame;
119 cv::gpu::GpuMat d_frame;
120
121 for (int i = 0; i < 10; ++i)
122 {
123 reader >> frame;
124 ASSERT_FALSE(frame.empty());
125
126 d_frame.upload(frame);
127
128 if (!d_writer.isOpened())
129 d_writer.open(outputFile, frame.size(), FPS);
130
131 d_writer.write(d_frame);
132 }
133
134 reader.release();
135 d_writer.close();
136
137 reader.open(outputFile);
138 ASSERT_TRUE(reader.isOpened());
139
140 for (int i = 0; i < 5; ++i)
141 {
142 reader >> frame;
143 ASSERT_FALSE(frame.empty());
144 }
145}
146
147INSTANTIATE_TEST_CASE_P(GPU_Video, VideoWriter, testing::Combine(
148 ALL_DEVICES,
149 testing::Values(std::string("768x576.avi"), std::string("1920x1080.avi"))));
150
151#endif // WIN32
