52PARAM_TEST_CASE(SetTo, cv::gpu::DeviceInfo, cv::Size, MatType, UseRoi)
53{
54 cv::gpu::DeviceInfo devInfo;
55 cv::Size size;
56 int type;
57 bool useRoi;
58
59 virtual void SetUp()
60 {
61 devInfo = GET_PARAM(0);
62 size = GET_PARAM(1);
63 type = GET_PARAM(2);
64 useRoi = GET_PARAM(3);
65
66 cv::gpu::setDevice(devInfo.deviceID());
67 }
68};
69
70GPU_TEST_P(SetTo, Zero)
71{
72 cv::Scalar zero = cv::Scalar::all(0);
73
74 cv::gpu::GpuMat mat = createMat(size, type, useRoi);
75 mat.setTo(zero);
76
77 EXPECT_MAT_NEAR(cv::Mat::zeros(size, type), mat, 0.0);
78}

80GPU_TEST_P(SetTo, SameVal)
81{
82 cv::Scalar val = cv::Scalar::all(randomDouble(0.0, 255.0));
83
84 if (CV_MAT_DEPTH(type) == CV_64F && !supportFeature(devInfo, cv::gpu::NATIVE_DOUBLE))
85 {
86 try
87 {
88 cv::gpu::GpuMat mat = createMat(size, type, useRoi);
89 mat.setTo(val);
90 }
91 catch (const cv::Exception& e)
92 {
93 ASSERT_EQ(CV_StsUnsupportedFormat, e.code);
94 }
95 }
96 else
97 {
98 cv::gpu::GpuMat mat = createMat(size, type, useRoi);
99 mat.setTo(val);
100
101 EXPECT_MAT_NEAR(cv::Mat(size, type, val), mat, 0.0);
102 }
103}

130GPU_TEST_P(SetTo, Masked)
131{
132 cv::Scalar val = randomScalar(0.0, 255.0);
133 cv::Mat mat_gold = randomMat(size, type);
134 cv::Mat mask = randomMat(size, CV_8UC1, 0.0, 2.0);
135
136 if (CV_MAT_DEPTH(type) == CV_64F && !supportFeature(devInfo, cv::gpu::NATIVE_DOUBLE))
137 {
138 try
139 {
140 cv::gpu::GpuMat mat = createMat(size, type, useRoi);
141 mat.setTo(val, loadMat(mask));
142 }
143 catch (const cv::Exception& e)
144 {
145 ASSERT_EQ(CV_StsUnsupportedFormat, e.code);
146 }
147 }
148 else
149 {
150 cv::gpu::GpuMat mat = loadMat(mat_gold, useRoi);
151 mat.setTo(val, loadMat(mask, useRoi));
152
153 mat_gold.setTo(val, mask);
154
155 EXPECT_MAT_NEAR(mat_gold, mat, 0.0);
156 }
157}

187GPU_TEST_P(CopyTo, WithOutMask)
188{
189 cv::Mat src = randomMat(size, type);
190
191 cv::gpu::GpuMat d_src = loadMat(src, useRoi);
192 cv::gpu::GpuMat dst = createMat(size, type, useRoi);
193 d_src.copyTo(dst);
194
195 EXPECT_MAT_NEAR(src, dst, 0.0);
196}

198GPU_TEST_P(CopyTo, Masked)
199{
200 cv::Mat src = randomMat(size, type);
201 cv::Mat mask = randomMat(size, CV_8UC1, 0.0, 2.0);
202
203 if (CV_MAT_DEPTH(type) == CV_64F && !supportFeature(devInfo, cv::gpu::NATIVE_DOUBLE))
204 {
205 try
206 {
207 cv::gpu::GpuMat d_src = loadMat(src);
208 cv::gpu::GpuMat dst;
209 d_src.copyTo(dst, loadMat(mask, useRoi));
210 }
211 catch (const cv::Exception& e)
212 {
213 ASSERT_EQ(CV_StsUnsupportedFormat, e.code);
214 }
215 }
216 else
217 {
218 cv::gpu::GpuMat d_src = loadMat(src, useRoi);
219 cv::gpu::GpuMat dst = loadMat(cv::Mat::zeros(size, type), useRoi);
220 d_src.copyTo(dst, loadMat(mask, useRoi));
221
222 cv::Mat dst_gold = cv::Mat::zeros(size, type);
223 src.copyTo(dst_gold, mask);
224
225 EXPECT_MAT_NEAR(dst_gold, dst, 0.0);
226 }
227}

238PARAM_TEST_CASE(ConvertTo, cv::gpu::DeviceInfo, cv::Size, MatDepth, MatDepth, UseRoi)
239{
240 cv::gpu::DeviceInfo devInfo;
241 cv::Size size;
242 int depth1;
243 int depth2;
244 bool useRoi;
245
246 virtual void SetUp()
247 {
248 devInfo = GET_PARAM(0);
249 size = GET_PARAM(1);
250 depth1 = GET_PARAM(2);
251 depth2 = GET_PARAM(3);
252 useRoi = GET_PARAM(4);
253
254 cv::gpu::setDevice(devInfo.deviceID());
255 }
256};
257
258GPU_TEST_P(ConvertTo, WithOutScaling)
259{
260 cv::Mat src = randomMat(size, depth1);
261
262 if ((depth1 == CV_64F || depth2 == CV_64F) && !supportFeature(devInfo, cv::gpu::NATIVE_DOUBLE))
263 {
264 try
265 {
266 cv::gpu::GpuMat d_src = loadMat(src);
267 cv::gpu::GpuMat dst;
268 d_src.convertTo(dst, depth2);
269 }
270 catch (const cv::Exception& e)
271 {
272 ASSERT_EQ(CV_StsUnsupportedFormat, e.code);
273 }
274 }
275 else
276 {
277 cv::gpu::GpuMat d_src = loadMat(src, useRoi);
278 cv::gpu::GpuMat dst = createMat(size, depth2, useRoi);
279 d_src.convertTo(dst, depth2);
280
281 cv::Mat dst_gold;
282 src.convertTo(dst_gold, depth2);
283
284 EXPECT_MAT_NEAR(dst_gold, dst, depth2 < CV_32F ? 1.0 : 1e-4);
285 }
286}

295 {

305 }
306 }

308 {

316 EXPECT_MAT_NEAR(dst_gold, dst, depth2 < CV_32F ? 1.0 : 1e-4);
317 }
318}

328// ensureSizeIsEnough
329
330struct EnsureSizeIsEnough : testing::TestWithParam<cv::gpu::DeviceInfo>
331{
332 virtual void SetUp()
333 {
334 cv::gpu::DeviceInfo devInfo = GetParam();
335 cv::gpu::setDevice(devInfo.deviceID());
336 }
337};
338
339GPU_TEST_P(EnsureSizeIsEnough, BufferReuse)
340{
341 cv::gpu::GpuMat buffer(100, 100, CV_8U);
342 cv::gpu::GpuMat old = buffer;
343
344 // don't reallocate memory
345 cv::gpu::ensureSizeIsEnough(10, 20, CV_8U, buffer);
346 EXPECT_EQ(10, buffer.rows);
347 EXPECT_EQ(20, buffer.cols);
348 EXPECT_EQ(CV_8UC1, buffer.type());
349 EXPECT_EQ(reinterpret_cast<intptr_t>(old.data), reinterpret_cast<intptr_t>(buffer.data));
350
351 // don't reallocate memory
352 cv::gpu::ensureSizeIsEnough(20, 30, CV_8U, buffer);
353 EXPECT_EQ(20, buffer.rows);
354 EXPECT_EQ(30, buffer.cols);
355 EXPECT_EQ(CV_8UC1, buffer.type());
356 EXPECT_EQ(reinterpret_cast<intptr_t>(old.data), reinterpret_cast<intptr_t>(buffer.data));
357}
358
359INSTANTIATE_TEST_CASE_P(GPU_GpuMat, EnsureSizeIsEnough, ALL_DEVICES);
