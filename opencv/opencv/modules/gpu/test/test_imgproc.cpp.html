502namespace
503{
504 template <typename T>
505 void blendLinearGold(const cv::Mat& img1, const cv::Mat& img2, const cv::Mat& weights1, const cv::Mat& weights2, cv::Mat& result_gold)
506 {
507 result_gold.create(img1.size(), img1.type());
508
509 int cn = img1.channels();
510
511 for (int y = 0; y < img1.rows; ++y)
512 {
513 const float* weights1_row = weights1.ptr<float>(y);
514 const float* weights2_row = weights2.ptr<float>(y);
515 const T* img1_row = img1.ptr<T>(y);
516 const T* img2_row = img2.ptr<T>(y);
517 T* result_gold_row = result_gold.ptr<T>(y);
518
519 for (int x = 0; x < img1.cols * cn; ++x)
520 {
521 float w1 = weights1_row[x / cn];
522 float w2 = weights2_row[x / cn];
523 result_gold_row[x] = static_cast<T>((img1_row[x] * w1 + img2_row[x] * w2) / (w1 + w2 + 1e-5f));
524 }
525 }
526 }
527}

577// Convolve

579namespace
580{
581 void convolveDFT(const cv::Mat& A, const cv::Mat& B, cv::Mat& C, bool ccorr = false)
582 {
583 // reallocate the output array if needed
584 C.create(std::abs(A.rows - B.rows) + 1, std::abs(A.cols - B.cols) + 1, A.type());
585 cv::Size dftSize;
586
587 // compute the size of DFT transform
588 dftSize.width = cv::getOptimalDFTSize(A.cols + B.cols - 1);
589 dftSize.height = cv::getOptimalDFTSize(A.rows + B.rows - 1);
590
591 // allocate temporary buffers and initialize them with 0s
592 cv::Mat tempA(dftSize, A.type(), cv::Scalar::all(0));
593 cv::Mat tempB(dftSize, B.type(), cv::Scalar::all(0));
594
595 // copy A and B to the top-left corners of tempA and tempB, respectively
596 cv::Mat roiA(tempA, cv::Rect(0, 0, A.cols, A.rows));
597 A.copyTo(roiA);
598 cv::Mat roiB(tempB, cv::Rect(0, 0, B.cols, B.rows));
599 B.copyTo(roiB);
600
601 // now transform the padded A & B in-place;
602 // use "nonzeroRows" hint for faster processing
603 cv::dft(tempA, tempA, 0, A.rows);
604 cv::dft(tempB, tempB, 0, B.rows);
605
606 // multiply the spectrums;
607 // the function handles packed spectrum representations well
608 cv::mulSpectrums(tempA, tempB, tempA, 0, ccorr);
609
610 // transform the product back from the frequency domain.
611 // Even though all the result rows will be non-zero,
612 // you need only the first C.rows of them, and thus you
613 // pass nonzeroRows == C.rows
614 cv::dft(tempA, tempA, cv::DFT_INVERSE + cv::DFT_SCALE, C.rows);
615
616 // now copy the result back to C.
617 tempA(cv::Rect(0, 0, C.cols, C.rows)).copyTo(C);
618 }
619
620 IMPLEMENT_PARAM_CLASS(KSize, int);
621 IMPLEMENT_PARAM_CLASS(Ccorr, bool);
622}

1064 void testR2CThenC2R(const std::string& hint, int cols, int rows, bool inplace)
1065 {
1066 SCOPED_TRACE(hint);
1067
1068 cv::Mat a = randomMat(cv::Size(cols, rows), CV_32FC1, 0.0, 10.0);
1069
1070 cv::gpu::GpuMat d_b, d_c;
1071 cv::gpu::GpuMat d_b_data, d_c_data;
1072 if (inplace)
1073 {
1074 if (a.cols == 1)
1075 {
1076 d_b_data.create(1, (a.rows / 2 + 1) * a.cols, CV_32FC2);
1077 d_b = cv::gpu::GpuMat(a.rows / 2 + 1, a.cols, CV_32FC2, d_b_data.ptr(), a.cols * d_b_data.elemSize());
1078 }
1079 else
1080 {
1081 d_b_data.create(1, a.rows * (a.cols / 2 + 1), CV_32FC2);
1082 d_b = cv::gpu::GpuMat(a.rows, a.cols / 2 + 1, CV_32FC2, d_b_data.ptr(), (a.cols / 2 + 1) * d_b_data.elemSize());
1083 }
1084 d_c_data.create(1, a.size().area(), CV_32F);
1085 d_c = cv::gpu::GpuMat(a.rows, a.cols, CV_32F, d_c_data.ptr(), a.cols * d_c_data.elemSize());
1086 }
1087
1088 cv::gpu::dft(loadMat(a), d_b, cv::Size(cols, rows), 0);
1089 cv::gpu::dft(d_b, d_c, cv::Size(cols, rows), cv::DFT_REAL_OUTPUT | cv::DFT_SCALE);
1090

1092 EXPECT_TRUE(!inplace || d_c.ptr() == d_c_data.ptr());
1093 ASSERT_EQ(CV_32F, d_c.depth());
1094 ASSERT_EQ(1, d_c.channels());
1095
1096 cv::Mat c(d_c);
1097 EXPECT_MAT_NEAR(a, c, rows * cols * 1e-5);
