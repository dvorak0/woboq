50// pyrDown
51
52PARAM_TEST_CASE(PyrDown, cv::gpu::DeviceInfo, cv::Size, MatType, UseRoi)
53{
54 cv::gpu::DeviceInfo devInfo;
55 cv::Size size;
56 int type;
57 bool useRoi;
58
59 virtual void SetUp()
60 {
61 devInfo = GET_PARAM(0);
62 size = GET_PARAM(1);
63 type = GET_PARAM(2);
64 useRoi = GET_PARAM(3);
65
66 cv::gpu::setDevice(devInfo.deviceID());
67 }
68};
69
70GPU_TEST_P(PyrDown, Accuracy)
71{
72 cv::Mat src = randomMat(size, type);
73
74 cv::gpu::GpuMat dst = createMat(cv::Size((size.width + 1) / 2, (size.height + 1) / 2), type, useRoi);
75 cv::gpu::pyrDown(loadMat(src, useRoi), dst);
76
77 cv::Mat dst_gold;
78 cv::pyrDown(src, dst_gold);
79
80 EXPECT_MAT_NEAR(dst_gold, dst, src.depth() == CV_32F ? 1e-4 : 1.0);
81}
82
83#ifdef OPENCV_TINY_GPU_MODULE
84INSTANTIATE_TEST_CASE_P(GPU_ImgProc, PyrDown, testing::Combine(
85 ALL_DEVICES,
86 DIFFERENT_SIZES,
87 testing::Values(MatType(CV_8UC1), MatType(CV_8UC3), MatType(CV_8UC4), MatType(CV_32FC1), MatType(CV_32FC3), MatType(CV_32FC4)),
88 WHOLE_SUBMAT));
89#else
90INSTANTIATE_TEST_CASE_P(GPU_ImgProc, PyrDown, testing::Combine(
91 ALL_DEVICES,
92 DIFFERENT_SIZES,
93 testing::Values(MatType(CV_8UC1), MatType(CV_8UC3), MatType(CV_8UC4), MatType(CV_16UC1), MatType(CV_16UC3), MatType(CV_16UC4), MatType(CV_32FC1), MatType(CV_32FC3), MatType(CV_32FC4)),
