50// HoughLines
51
52PARAM_TEST_CASE(HoughLines, cv::gpu::DeviceInfo, cv::Size, UseRoi)
53{
54 static void generateLines(cv::Mat& img)
55 {
56 img.setTo(cv::Scalar::all(0));
57
58 cv::line(img, cv::Point(20, 0), cv::Point(20, img.rows), cv::Scalar::all(255));
59 cv::line(img, cv::Point(0, 50), cv::Point(img.cols, 50), cv::Scalar::all(255));
60 cv::line(img, cv::Point(0, 0), cv::Point(img.cols, img.rows), cv::Scalar::all(255));
61 cv::line(img, cv::Point(img.cols, 0), cv::Point(0, img.rows), cv::Scalar::all(255));
62 }
63
64 static void drawLines(cv::Mat& dst, const std::vector<cv::Vec2f>& lines)
65 {
66 dst.setTo(cv::Scalar::all(0));
67
68 for (size_t i = 0; i < lines.size(); ++i)
69 {
70 float rho = lines[i][0], theta = lines[i][1];
71 cv::Point pt1, pt2;
72 double a = std::cos(theta), b = std::sin(theta);
73 double x0 = a*rho, y0 = b*rho;
74 pt1.x = cvRound(x0 + 1000*(-b));
75 pt1.y = cvRound(y0 + 1000*(a));
76 pt2.x = cvRound(x0 - 1000*(-b));
77 pt2.y = cvRound(y0 - 1000*(a));
78 cv::line(dst, pt1, pt2, cv::Scalar::all(255));
79 }
80 }
81};
82
83GPU_TEST_P(HoughLines, Accuracy)
84{
85 const cv::gpu::DeviceInfo devInfo = GET_PARAM(0);
86 cv::gpu::setDevice(devInfo.deviceID());
87 const cv::Size size = GET_PARAM(1);
88 const bool useRoi = GET_PARAM(2);
89
90 const float rho = 1.0f;
91 const float theta = (float) (1.5 * CV_PI / 180.0);
92 const int threshold = 100;
93
94 cv::Mat src(size, CV_8UC1);
95 generateLines(src);
96
97 cv::gpu::GpuMat d_lines;
98 cv::gpu::HoughLines(loadMat(src, useRoi), d_lines, rho, theta, threshold);
99
100 std::vector<cv::Vec2f> lines;
101 cv::gpu::HoughLinesDownload(d_lines, lines);
102
103 cv::Mat dst(size, CV_8UC1);
104 drawLines(dst, lines);
105
106 ASSERT_MAT_NEAR(src, dst, 0.0);
107}
108
109INSTANTIATE_TEST_CASE_P(GPU_ImgProc, HoughLines, testing::Combine(
110 ALL_DEVICES,
111 DIFFERENT_SIZES,
112 WHOLE_SUBMAT));

116
117PARAM_TEST_CASE(HoughCircles, cv::gpu::DeviceInfo, cv::Size, UseRoi)
118{
119 static void drawCircles(cv::Mat& dst, const std::vector<cv::Vec3f>& circles, bool fill)
120 {
121 dst.setTo(cv::Scalar::all(0));
122
123 for (size_t i = 0; i < circles.size(); ++i)
124 cv::circle(dst, cv::Point2f(circles[i][0], circles[i][1]), (int)circles[i][2], cv::Scalar::all(255), fill ? -1 : 1);
125 }
126};
127
128GPU_TEST_P(HoughCircles, Accuracy)
129{
130 const cv::gpu::DeviceInfo devInfo = GET_PARAM(0);
131 cv::gpu::setDevice(devInfo.deviceID());
132 const cv::Size size = GET_PARAM(1);
133 const bool useRoi = GET_PARAM(2);
134
135 const float dp = 2.0f;
136 const float minDist = 0.0f;
137 const int minRadius = 10;
138 const int maxRadius = 20;
139 const int cannyThreshold = 100;
140 const int votesThreshold = 20;
141
142 std::vector<cv::Vec3f> circles_gold(4);
143 circles_gold[0] = cv::Vec3i(20, 20, minRadius);
144 circles_gold[1] = cv::Vec3i(90, 87, minRadius + 3);
145 circles_gold[2] = cv::Vec3i(30, 70, minRadius + 8);
146 circles_gold[3] = cv::Vec3i(80, 10, maxRadius);
147
148 cv::Mat src(size, CV_8UC1);
149 drawCircles(src, circles_gold, true);
150
151 cv::gpu::GpuMat d_circles;
152 cv::gpu::HoughCircles(loadMat(src, useRoi), d_circles, CV_HOUGH_GRADIENT, dp, minDist, cannyThreshold, votesThreshold, minRadius, maxRadius);
153
154 std::vector<cv::Vec3f> circles;
155 cv::gpu::HoughCirclesDownload(d_circles, circles);
156
157 ASSERT_FALSE(circles.empty());
158
159 for (size_t i = 0; i < circles.size(); ++i)
160 {
161 cv::Vec3f cur = circles[i];
162
163 bool found = false;
164
165 for (size_t j = 0; j < circles_gold.size(); ++j)
166 {
167 cv::Vec3f gold = circles_gold[j];
168
169 if (std::fabs(cur[0] - gold[0]) < 5 && std::fabs(cur[1] - gold[1]) < 5 && std::fabs(cur[2] - gold[2]) < 5)
170 {
171 found = true;
172 break;
173 }
174 }
175
176 ASSERT_TRUE(found);
177 }
178}
179
180INSTANTIATE_TEST_CASE_P(GPU_ImgProc, HoughCircles, testing::Combine(
181 ALL_DEVICES,
182 DIFFERENT_SIZES,
183 WHOLE_SUBMAT));

194GPU_TEST_P(GeneralizedHough, POSITION)
195{
196 const cv::gpu::DeviceInfo devInfo = GET_PARAM(0);
197 cv::gpu::setDevice(devInfo.deviceID());
198 const bool useRoi = GET_PARAM(1);
199
200 cv::Mat templ = readImage("../cv/shared/templ.png", cv::IMREAD_GRAYSCALE);
201 ASSERT_FALSE(templ.empty());
202
203 cv::Point templCenter(templ.cols / 2, templ.rows / 2);
204
205 const size_t gold_count = 3;
206 cv::Point pos_gold[gold_count];
207 pos_gold[0] = cv::Point(templCenter.x + 10, templCenter.y + 10);
208 pos_gold[1] = cv::Point(2 * templCenter.x + 40, templCenter.y + 10);
209 pos_gold[2] = cv::Point(2 * templCenter.x + 40, 2 * templCenter.y + 40);
210
211 cv::Mat image(templ.rows * 3, templ.cols * 3, CV_8UC1, cv::Scalar::all(0));
212 for (size_t i = 0; i < gold_count; ++i)
213 {
214 cv::Rect rec(pos_gold[i].x - templCenter.x, pos_gold[i].y - templCenter.y, templ.cols, templ.rows);
215 cv::Mat imageROI = image(rec);
216 templ.copyTo(imageROI);
217 }
218
219 cv::Ptr<cv::gpu::GeneralizedHough_GPU> hough = cv::gpu::GeneralizedHough_GPU::create(cv::GHT_POSITION);
220 hough->set("votesThreshold", 200);
221
222 hough->setTemplate(loadMat(templ, useRoi));
223
224 cv::gpu::GpuMat d_pos;
225 hough->detect(loadMat(image, useRoi), d_pos);
226
227 std::vector<cv::Vec4f> pos;
228 hough->download(d_pos, pos);
229
230 ASSERT_EQ(gold_count, pos.size());
231
232 for (size_t i = 0; i < gold_count; ++i)
233 {
234 cv::Point gold = pos_gold[i];
235
236 bool found = false;
237
238 for (size_t j = 0; j < pos.size(); ++j)
239 {
240 cv::Point2f p(pos[j][0], pos[j][1]);
241
242 if (::fabs(p.x - gold.x) < 2 && ::fabs(p.y - gold.y) < 2)
243 {
244 found = true;
245 break;
246 }
247 }
248
249 ASSERT_TRUE(found);
250 }
251}
252
253INSTANTIATE_TEST_CASE_P(GPU_ImgProc, GeneralizedHough, testing::Combine(
254 ALL_DEVICES,
255 WHOLE_SUBMAT));
