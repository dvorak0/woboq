335PARAM_TEST_CASE(Texture2D, cv::Size, MatType)
336{
337 static void SetUpTestCase()
338 {
339 cv::namedWindow("test", cv::WINDOW_OPENGL);
340 }
341
342 static void TearDownTestCase()
343 {
344 cv::destroyAllWindows();

346
347 cv::Size size;
348 int type;
349 int depth;
350 int cn;
351 cv::ogl::Texture2D::Format format;
352
353 virtual void SetUp()
354 {
355 size = GET_PARAM(0);
356 type = GET_PARAM(1);
357
358 depth = CV_MAT_DEPTH(type);
359 cn = CV_MAT_CN(type);
360 format = cn == 1 ? cv::ogl::Texture2D::DEPTH_COMPONENT : cn == 3 ? cv::ogl::Texture2D::RGB : cn == 4 ? cv::ogl::Texture2D::RGBA : cv::ogl::Texture2D::NONE;
361 }
362};
363

370 EXPECT_EQ(format, tex.format());
371}
