54 template <typename T, template <typename> class Interpolator>
55 void resizeImpl(const cv::Mat& src, cv::Mat& dst, double fx, double fy)
56 {
57 const int cn = src.channels();
58
59 cv::Size dsize(cv::saturate_cast<int>(src.cols * fx), cv::saturate_cast<int>(src.rows * fy));
60
61 dst.create(dsize, src.type());
62
63 float ifx = static_cast<float>(1.0 / fx);
64 float ify = static_cast<float>(1.0 / fy);
65
66 for (int y = 0; y < dsize.height; ++y)
67 {
68 for (int x = 0; x < dsize.width; ++x)
69 {
70 for (int c = 0; c < cn; ++c)
71 dst.at<T>(y, x * cn + c) = Interpolator<T>::getValue(src, y * ify, x * ifx, c, cv::BORDER_REPLICATE);
72 }
73 }
74 }
75
76 template <typename T, template <typename> class Interpolator>
77 void resizeLinearImpl(const cv::Mat& src, cv::Mat& dst, double fx, double fy)
78 {
79 const int cn = src.channels();
80
81 cv::Size dsize(cv::saturate_cast<int>(src.cols * fx), cv::saturate_cast<int>(src.rows * fy));
82
83 dst.create(dsize, src.type());
84
85 float ifx = static_cast<float>(1.0 / fx);
86 float ify = static_cast<float>(1.0 / fy);
87
88 for (int y = 0; y < dsize.height; ++y)
89 {
90 for (int x = 0; x < dsize.width; ++x)
91 {
92 for (int c = 0; c < cn; ++c)
93 dst.at<T>(y, x * cn + c) = Interpolator<T>::getValue(src, (y + 0.5f) * ify - 0.5f, (x + 0.5f) * ifx - 0.5f, c, cv::BORDER_REPLICATE);
94 }
95 }
96 }
97
98 void resizeGold(const cv::Mat& src, cv::Mat& dst, double fx, double fy, int interpolation)
99 {
100 typedef void (*func_t)(const cv::Mat& src, cv::Mat& dst, double fx, double fy);
101
102 static const func_t nearest_funcs[] =
103 {
104 resizeImpl<unsigned char, NearestInterpolator>,
105 resizeImpl<signed char, NearestInterpolator>,
106 resizeImpl<unsigned short, NearestInterpolator>,
107 resizeImpl<short, NearestInterpolator>,
108 resizeImpl<int, NearestInterpolator>,
109 resizeImpl<float, NearestInterpolator>
110 };
111
112
113 static const func_t linear_funcs[] =
114 {
115 resizeLinearImpl<unsigned char, LinearInterpolator>,
116 resizeLinearImpl<signed char, LinearInterpolator>,
117 resizeLinearImpl<unsigned short, LinearInterpolator>,
118 resizeLinearImpl<short, LinearInterpolator>,
119 resizeLinearImpl<int, LinearInterpolator>,
120 resizeLinearImpl<float, LinearInterpolator>
121 };
122
123 static const func_t cubic_funcs[] =
124 {
125 resizeImpl<unsigned char, CubicInterpolator>,
126 resizeImpl<signed char, CubicInterpolator>,
127 resizeImpl<unsigned short, CubicInterpolator>,
128 resizeImpl<short, CubicInterpolator>,
129 resizeImpl<int, CubicInterpolator>,
130 resizeImpl<float, CubicInterpolator>
131 };
132
133 static const func_t* funcs[] = {nearest_funcs, linear_funcs, cubic_funcs};
134
135 funcs[interpolation][src.depth()](src, dst, fx, fy);
136 }
137}

142PARAM_TEST_CASE(Resize, cv::gpu::DeviceInfo, cv::Size, MatType, double, Interpolation, UseRoi)
143{
144 cv::gpu::DeviceInfo devInfo;
145 cv::Size size;
146 double coeff;
147 int interpolation;
148 int type;
149 bool useRoi;
150
151 virtual void SetUp()
152 {
153 devInfo = GET_PARAM(0);
154 size = GET_PARAM(1);
155 type = GET_PARAM(2);
156 coeff = GET_PARAM(3);
157 interpolation = GET_PARAM(4);
158 useRoi = GET_PARAM(5);
159
160 cv::gpu::setDevice(devInfo.deviceID());
161 }
162};
163
164GPU_TEST_P(Resize, Accuracy)
165{
166 cv::Mat src = randomMat(size, type);
167
168 cv::gpu::GpuMat dst = createMat(cv::Size(cv::saturate_cast<int>(src.cols * coeff), cv::saturate_cast<int>(src.rows * coeff)), type, useRoi);
169 cv::gpu::resize(loadMat(src, useRoi), dst, cv::Size(), coeff, coeff, interpolation);
170
171 cv::Mat dst_gold;
172 resizeGold(src, dst_gold, coeff, coeff, interpolation);
173
174 EXPECT_MAT_NEAR(dst_gold, dst, src.depth() == CV_32F ? 1e-2 : 1.0);
175}

197PARAM_TEST_CASE(ResizeSameAsHost, cv::gpu::DeviceInfo, cv::Size, MatType, double, Interpolation, UseRoi)
198{
199 cv::gpu::DeviceInfo devInfo;
200 cv::Size size;
201 double coeff;
202 int interpolation;
203 int type;
204 bool useRoi;
205
206 virtual void SetUp()
207 {
208 devInfo = GET_PARAM(0);
209 size = GET_PARAM(1);
210 type = GET_PARAM(2);
211 coeff = GET_PARAM(3);
212 interpolation = GET_PARAM(4);
213 useRoi = GET_PARAM(5);
214
215 cv::gpu::setDevice(devInfo.deviceID());
216 }
217};
218
219// downscaling only: used for classifiers
220GPU_TEST_P(ResizeSameAsHost, Accuracy)
221{
222 cv::Mat src = randomMat(size, type);
223
224 cv::gpu::GpuMat dst = createMat(cv::Size(cv::saturate_cast<int>(src.cols * coeff), cv::saturate_cast<int>(src.rows * coeff)), type, useRoi);
225 cv::gpu::resize(loadMat(src, useRoi), dst, cv::Size(), coeff, coeff, interpolation);
226
227 cv::Mat dst_gold;
228 cv::resize(src, dst_gold, cv::Size(), coeff, coeff, interpolation);
229
230 // CPU test for cv::resize uses 16 as error threshold for CV_8U, we uses 4 as error threshold for CV_8U
231 EXPECT_MAT_NEAR(dst_gold, dst, src.depth() == CV_32F ? 1e-2 : src.depth() == CV_8U ? 4.0 : 1.0);
232}
