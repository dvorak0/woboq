75PARAM_TEST_CASE(FGDStatModel, cv::gpu::DeviceInfo, std::string, Channels)
76{
77 cv::gpu::DeviceInfo devInfo;
78 std::string inputFile;
79 int out_cn;
80
81 virtual void SetUp()
82 {
83 devInfo = GET_PARAM(0);
84 cv::gpu::setDevice(devInfo.deviceID());
85
86 inputFile = std::string(cvtest::TS::ptr()->get_data_path()) + "video/" + GET_PARAM(1);
87
88 out_cn = GET_PARAM(2);
89 }
90};
91
92GPU_TEST_P(FGDStatModel, Update)
93{
94 cv::VideoCapture cap(inputFile);
95 ASSERT_TRUE(cap.isOpened());
96
97 cv::Mat frame;
98 cap >> frame;
99 ASSERT_FALSE(frame.empty());
100
101 cv::Mat frameSmall;
102 cv::resize(frame, frameSmall, cv::Size(), 0.5, 0.5);
103
104 IplImage ipl_frame = frameSmall;
105 cv::Ptr<CvBGStatModel> model(cvCreateFGDStatModel(&ipl_frame));
106
107 cv::gpu::GpuMat d_frame(frameSmall);
108 cv::gpu::FGDStatModel d_model(out_cn);
109 d_model.create(d_frame);
110
111 cv::Mat h_background;
112 cv::Mat h_foreground;
113 cv::Mat h_background3;
114
115 for (int i = 0; i < 5; ++i)
116 {
117 cap >> frame;
118 ASSERT_FALSE(frame.empty());
119
120 cv::resize(frame, frameSmall, cv::Size(), 0.5, 0.5);
121
122 ipl_frame = frameSmall;
123 int gold_count = cvUpdateBGStatModel(&ipl_frame, model);
124
125 d_frame.upload(frameSmall);
126
127 int count = d_model.update(d_frame);
128
129 ASSERT_EQ(gold_count, count);
130
131 cv::Mat gold_background(model->background);
132 cv::Mat gold_foreground(model->foreground);
133
134 if (out_cn == 3)
135 d_model.background.download(h_background3);
136 else
137 {
138 d_model.background.download(h_background);
139 cv::cvtColor(h_background, h_background3, cv::COLOR_BGRA2BGR);
140 }
141 d_model.foreground.download(h_foreground);
142
143 ASSERT_MAT_NEAR(gold_background, h_background3, 1.0);
144 ASSERT_MAT_NEAR(gold_foreground, h_foreground, 0.0);
145 }
146}

149 ALL_DEVICES,
150 testing::Values(std::string("768x576.avi")),
151 testing::Values(Channels(3), Channels(4))));

166PARAM_TEST_CASE(MOG, cv::gpu::DeviceInfo, std::string, UseGray, LearningRate, UseRoi)
167{
168 cv::gpu::DeviceInfo devInfo;
169 std::string inputFile;
170 bool useGray;
171 double learningRate;
172 bool useRoi;
173
174 virtual void SetUp()
175 {
176 devInfo = GET_PARAM(0);
177 cv::gpu::setDevice(devInfo.deviceID());
178
179 inputFile = std::string(cvtest::TS::ptr()->get_data_path()) + "video/" + GET_PARAM(1);
180
181 useGray = GET_PARAM(2);
182
183 learningRate = GET_PARAM(3);
184
185 useRoi = GET_PARAM(4);
186 }
187};
188
189GPU_TEST_P(MOG, Update)
190{
191 cv::VideoCapture cap(inputFile);
192 ASSERT_TRUE(cap.isOpened());
193
194 cv::Mat frame;
195 cap >> frame;
196 ASSERT_FALSE(frame.empty());
197
198 cv::gpu::MOG_GPU mog;
199 cv::gpu::GpuMat foreground = createMat(frame.size(), CV_8UC1, useRoi);
200
201 cv::BackgroundSubtractorMOG mog_gold;
202 cv::Mat foreground_gold;
203
204 for (int i = 0; i < 10; ++i)
205 {
206 cap >> frame;
207 ASSERT_FALSE(frame.empty());
208
209 if (useGray)
210 {
211 cv::Mat temp;
212 cv::cvtColor(frame, temp, cv::COLOR_BGR2GRAY);
213 cv::swap(temp, frame);
214 }
215
216 mog(loadMat(frame, useRoi), foreground, (float)learningRate);
217
218 mog_gold(frame, foreground_gold, learningRate);
219
220 ASSERT_MAT_NEAR(foreground_gold, foreground, 0.0);
221 }
222}

243PARAM_TEST_CASE(MOG2, cv::gpu::DeviceInfo, std::string, UseGray, DetectShadow, UseRoi)
244{
245 cv::gpu::DeviceInfo devInfo;
246 std::string inputFile;
247 bool useGray;
248 bool detectShadow;
249 bool useRoi;
250
251 virtual void SetUp()
252 {
253 devInfo = GET_PARAM(0);
254 cv::gpu::setDevice(devInfo.deviceID());
255
256 inputFile = std::string(cvtest::TS::ptr()->get_data_path()) + "video/" + GET_PARAM(1);
257 useGray = GET_PARAM(2);
258 detectShadow = GET_PARAM(3);
259 useRoi = GET_PARAM(4);
260 }
261};
262
263GPU_TEST_P(MOG2, Update)
264{
265 cv::VideoCapture cap(inputFile);
266 ASSERT_TRUE(cap.isOpened());
267
268 cv::Mat frame;
269 cap >> frame;
270 ASSERT_FALSE(frame.empty());
271
272 cv::gpu::MOG2_GPU mog2;
273 mog2.bShadowDetection = detectShadow;
274 cv::gpu::GpuMat foreground = createMat(frame.size(), CV_8UC1, useRoi);
275
276 cv::BackgroundSubtractorMOG2 mog2_gold;
277 mog2_gold.set("detectShadows", detectShadow);
278 cv::Mat foreground_gold;
279
280 for (int i = 0; i < 10; ++i)
281 {
282 cap >> frame;
283 ASSERT_FALSE(frame.empty());
284
285 if (useGray)
286 {
287 cv::Mat temp;
288 cv::cvtColor(frame, temp, cv::COLOR_BGR2GRAY);
289 cv::swap(temp, frame);
290 }
291
292 mog2(loadMat(frame, useRoi), foreground);
293
294 mog2_gold(frame, foreground_gold);
295
296 if (detectShadow)
297 {
298 ASSERT_MAT_SIMILAR(foreground_gold, foreground, 1e-2);
299 }
300 else
301 {
302 ASSERT_MAT_NEAR(foreground_gold, foreground, 0);
303 }
304 }
305}
306
307GPU_TEST_P(MOG2, getBackgroundImage)
308{
309 if (useGray)
310 return;
311
312 cv::VideoCapture cap(inputFile);
313 ASSERT_TRUE(cap.isOpened());
314
315 cv::Mat frame;
316
317 cv::gpu::MOG2_GPU mog2;
318 mog2.bShadowDetection = detectShadow;
319 cv::gpu::GpuMat foreground;
320
321 cv::BackgroundSubtractorMOG2 mog2_gold;
322 mog2_gold.set("detectShadows", detectShadow);
323 cv::Mat foreground_gold;
324
325 for (int i = 0; i < 10; ++i)
326 {
327 cap >> frame;
328 ASSERT_FALSE(frame.empty());
329
330 mog2(loadMat(frame, useRoi), foreground);
331
332 mog2_gold(frame, foreground_gold);
333 }
334
335 cv::gpu::GpuMat background = createMat(frame.size(), frame.type(), useRoi);
336 mog2.getBackgroundImage(background);
337
338 cv::Mat background_gold;
339 mog2_gold.getBackgroundImage(background_gold);
340
341 ASSERT_MAT_NEAR(background_gold, background, 1);
342}

356PARAM_TEST_CASE(GMG, cv::gpu::DeviceInfo, cv::Size, MatDepth, Channels, UseRoi)
357{
358};
359
360GPU_TEST_P(GMG, Accuracy)
361{
362 const cv::gpu::DeviceInfo devInfo = GET_PARAM(0);
363 cv::gpu::setDevice(devInfo.deviceID());
364 const cv::Size size = GET_PARAM(1);
365 const int depth = GET_PARAM(2);
366 const int channels = GET_PARAM(3);
367 const bool useRoi = GET_PARAM(4);
368
369 const int type = CV_MAKE_TYPE(depth, channels);
370
371 const cv::Mat zeros(size, CV_8UC1, cv::Scalar::all(0));
372 const cv::Mat fullfg(size, CV_8UC1, cv::Scalar::all(255));
373
374 cv::Mat frame = randomMat(size, type, 0, 100);
375 cv::gpu::GpuMat d_frame = loadMat(frame, useRoi);
376
377 cv::gpu::GMG_GPU gmg;
378 gmg.numInitializationFrames = 5;
379 gmg.smoothingRadius = 0;
380 gmg.initialize(d_frame.size(), 0, 255);
381
382 cv::gpu::GpuMat d_fgmask = createMat(size, CV_8UC1, useRoi);
383
384 for (int i = 0; i < gmg.numInitializationFrames; ++i)
385 {
386 gmg(d_frame, d_fgmask);
387
388 // fgmask should be entirely background during training
389 ASSERT_MAT_NEAR(zeros, d_fgmask, 0);
390 }
391
392 frame = randomMat(size, type, 160, 255);
393 d_frame = loadMat(frame, useRoi);
394 gmg(d_frame, d_fgmask);
395
396 // now fgmask should be entirely foreground
397 ASSERT_MAT_NEAR(fullfg, d_fgmask, 0);
398}
399
400INSTANTIATE_TEST_CASE_P(GPU_Video, GMG, testing::Combine(
401 ALL_DEVICES,
402 DIFFERENT_SIZES,
403 testing::Values(MatType(CV_8U), MatType(CV_16U), MatType(CV_32F)),
404 testing::Values(Channels(1), Channels(3), Channels(4)),
405 WHOLE_SUBMAT));
