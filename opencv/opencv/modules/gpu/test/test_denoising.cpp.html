50// BilateralFilter
51
52PARAM_TEST_CASE(BilateralFilter, cv::gpu::DeviceInfo, cv::Size, MatType)
53{
54 cv::gpu::DeviceInfo devInfo;
55 cv::Size size;
56 int type;
57 int kernel_size;
58 float sigma_color;
59 float sigma_spatial;
60
61 virtual void SetUp()
62 {
63 devInfo = GET_PARAM(0);
64 size = GET_PARAM(1);
65 type = GET_PARAM(2);
66
67 kernel_size = 5;
68 sigma_color = 10.f;
69 sigma_spatial = 3.5f;
70
71 cv::gpu::setDevice(devInfo.deviceID());
72 }
73};
74
75GPU_TEST_P(BilateralFilter, Accuracy)
76{
77 cv::Mat src = randomMat(size, type);
78
79 src.convertTo(src, type);
80 cv::gpu::GpuMat dst;
81
82 cv::gpu::bilateralFilter(loadMat(src), dst, kernel_size, sigma_color, sigma_spatial);
83
84 cv::Mat dst_gold;
85 cv::bilateralFilter(src, dst_gold, kernel_size, sigma_color, sigma_spatial);
86
87 EXPECT_MAT_NEAR(dst_gold, dst, src.depth() == CV_32F ? 1e-3 : 1.0);
88}
89
90INSTANTIATE_TEST_CASE_P(GPU_Denoising, BilateralFilter, testing::Combine(
91 ALL_DEVICES,
92 testing::Values(cv::Size(128, 128), cv::Size(113, 113), cv::Size(639, 481)),
93 testing::Values(MatType(CV_8UC1), MatType(CV_8UC3), MatType(CV_32FC1), MatType(CV_32FC3))
94 ));

98// Brute Force Non local means
99
100struct BruteForceNonLocalMeans: testing::TestWithParam<cv::gpu::DeviceInfo>
101{
102 cv::gpu::DeviceInfo devInfo;
103
104 virtual void SetUp()
105 {
106 devInfo = GetParam();
107 cv::gpu::setDevice(devInfo.deviceID());
108 }
109};
110
111GPU_TEST_P(BruteForceNonLocalMeans, Regression)
112{
113 using cv::gpu::GpuMat;
114
115 cv::Mat bgr = readImage("denoising/lena_noised_gaussian_sigma=20_multi_0.png", cv::IMREAD_COLOR);
116 ASSERT_FALSE(bgr.empty());
117 cv::resize(bgr, bgr, cv::Size(256, 256));
118
119 cv::Mat gray;
120 cv::cvtColor(bgr, gray, CV_BGR2GRAY);
121
122 GpuMat dbgr, dgray;
123 cv::gpu::nonLocalMeans(GpuMat(bgr), dbgr, 20);
124 cv::gpu::nonLocalMeans(GpuMat(gray), dgray, 20);
125
126#if 0
127 dumpImage("denoising/nlm_denoised_lena_bgr.png", cv::Mat(dbgr));
128 dumpImage("denoising/nlm_denoised_lena_gray.png", cv::Mat(dgray));
129#endif
130
131 cv::Mat bgr_gold = readImage("denoising/nlm_denoised_lena_bgr.png", cv::IMREAD_COLOR);
132 cv::Mat gray_gold = readImage("denoising/nlm_denoised_lena_gray.png", cv::IMREAD_GRAYSCALE);
133 ASSERT_FALSE(bgr_gold.empty() || gray_gold.empty());
134 cv::resize(bgr_gold, bgr_gold, cv::Size(256, 256));
135 cv::resize(gray_gold, gray_gold, cv::Size(256, 256));
136
137 EXPECT_MAT_NEAR(bgr_gold, dbgr, 1);
138 EXPECT_MAT_NEAR(gray_gold, dgray, 1);
139}
140
141INSTANTIATE_TEST_CASE_P(GPU_Denoising, BruteForceNonLocalMeans, ALL_DEVICES);
