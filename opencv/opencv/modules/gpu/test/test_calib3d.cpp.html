64GPU_TEST_P(StereoBM, Regression)
65{
66 cv::Mat left_image = readImage("stereobm/aloe-L.png", cv::IMREAD_GRAYSCALE);
67 cv::Mat right_image = readImage("stereobm/aloe-R.png", cv::IMREAD_GRAYSCALE);

71 ASSERT_FALSE(right_image.empty());

76
77 bm(loadMat(left_image), loadMat(right_image), disp);
78
79 EXPECT_MAT_NEAR(disp_gold, disp, 0.0);
80}

171struct TransformPoints : testing::TestWithParam<cv::gpu::DeviceInfo>
172{
173 cv::gpu::DeviceInfo devInfo;
174
175 virtual void SetUp()
176 {
177 devInfo = GetParam();
178
179 cv::gpu::setDevice(devInfo.deviceID());
180 }
181};
182
183GPU_TEST_P(TransformPoints, Accuracy)
184{
185 cv::Mat src = randomMat(cv::Size(1000, 1), CV_32FC3, 0, 10);
186 cv::Mat rvec = randomMat(cv::Size(3, 1), CV_32F, 0, 1);
187 cv::Mat tvec = randomMat(cv::Size(3, 1), CV_32F, 0, 1);
188
189 cv::gpu::GpuMat dst;
190 cv::gpu::transformPoints(loadMat(src), rvec, tvec, dst);
191
192 ASSERT_EQ(src.size(), dst.size());
193 ASSERT_EQ(src.type(), dst.type());
194
195 cv::Mat h_dst(dst);
196
197 cv::Mat rot;
198 cv::Rodrigues(rvec, rot);
199
200 for (int i = 0; i < h_dst.cols; ++i)
201 {
202 cv::Point3f res = h_dst.at<cv::Point3f>(0, i);
203
204 cv::Point3f p = src.at<cv::Point3f>(0, i);
205 cv::Point3f res_gold(
206 rot.at<float>(0, 0) * p.x + rot.at<float>(0, 1) * p.y + rot.at<float>(0, 2) * p.z + tvec.at<float>(0, 0),
207 rot.at<float>(1, 0) * p.x + rot.at<float>(1, 1) * p.y + rot.at<float>(1, 2) * p.z + tvec.at<float>(0, 1),
208 rot.at<float>(2, 0) * p.x + rot.at<float>(2, 1) * p.y + rot.at<float>(2, 2) * p.z + tvec.at<float>(0, 2));
209
210 ASSERT_POINT3_NEAR(res_gold, res, 1e-5);
211 }
212}
213
214INSTANTIATE_TEST_CASE_P(GPU_Calib3D, TransformPoints, ALL_DEVICES);

218
219struct ProjectPoints : testing::TestWithParam<cv::gpu::DeviceInfo>
220{
221 cv::gpu::DeviceInfo devInfo;
222
223 virtual void SetUp()
224 {
225 devInfo = GetParam();
226
227 cv::gpu::setDevice(devInfo.deviceID());
228 }
229};
230
231GPU_TEST_P(ProjectPoints, Accuracy)
232{
233 cv::Mat src = randomMat(cv::Size(1000, 1), CV_32FC3, 0, 10);
234 cv::Mat rvec = randomMat(cv::Size(3, 1), CV_32F, 0, 1);
235 cv::Mat tvec = randomMat(cv::Size(3, 1), CV_32F, 0, 1);
236 cv::Mat camera_mat = randomMat(cv::Size(3, 3), CV_32F, 0.5, 1);
237 camera_mat.at<float>(0, 1) = 0.f;
238 camera_mat.at<float>(1, 0) = 0.f;
239 camera_mat.at<float>(2, 0) = 0.f;
240 camera_mat.at<float>(2, 1) = 0.f;
241
242 cv::gpu::GpuMat dst;
243 cv::gpu::projectPoints(loadMat(src), rvec, tvec, camera_mat, cv::Mat(), dst);
244
245 ASSERT_EQ(1, dst.rows);
246 ASSERT_EQ(MatType(CV_32FC2), MatType(dst.type()));
247
248 std::vector<cv::Point2f> dst_gold;
249 cv::projectPoints(src, rvec, tvec, camera_mat, cv::Mat(1, 8, CV_32F, cv::Scalar::all(0)), dst_gold);
250
251 ASSERT_EQ(dst_gold.size(), static_cast<size_t>(dst.cols));
252
253 cv::Mat h_dst(dst);
254
255 for (size_t i = 0; i < dst_gold.size(); ++i)
256 {
257 cv::Point2f res = h_dst.at<cv::Point2f>(0, (int)i);
258 cv::Point2f res_gold = dst_gold[i];
259
260 ASSERT_LE(cv::norm(res_gold - res) / cv::norm(res_gold), 1e-3f);
261 }
262}
263
264INSTANTIATE_TEST_CASE_P(GPU_Calib3D, ProjectPoints, ALL_DEVICES);

268
269struct SolvePnPRansac : testing::TestWithParam<cv::gpu::DeviceInfo>
270{
271 cv::gpu::DeviceInfo devInfo;
272
273 virtual void SetUp()
274 {
275 devInfo = GetParam();
276
277 cv::gpu::setDevice(devInfo.deviceID());
278 }
279};
280
281GPU_TEST_P(SolvePnPRansac, Accuracy)
282{
283 // Use RNG with fixed seed to be reproducable
284 cv::RNG rng(123456789);
285 cv::theRNG() = rng;
286
287 cv::Mat object = cvtest::randomMat(rng, cv::Size(5000, 1), CV_32FC3, 0, 100, false);
288 cv::Mat camera_mat = cvtest::randomMat(rng, cv::Size(3, 3), CV_32FC1, 0.5, 1, false);
289 camera_mat.at<float>(0, 1) = 0.f;
290 camera_mat.at<float>(1, 0) = 0.f;
291 camera_mat.at<float>(2, 0) = 0.f;
292 camera_mat.at<float>(2, 1) = 0.f;
293
294 std::vector<cv::Point2f> image_vec;
295 cv::Mat rvec_gold = cvtest::randomMat(rng, cv::Size(3, 1), CV_32F, 0, 1, false);
296 cv::Mat tvec_gold = cvtest::randomMat(rng, cv::Size(3, 1), CV_32F, 0, 1, false);
297 cv::projectPoints(object, rvec_gold, tvec_gold, camera_mat, cv::Mat(1, 8, CV_32F, cv::Scalar::all(0)), image_vec);
298
299 cv::Mat rvec, tvec;
300 std::vector<int> inliers;
301 cv::gpu::solvePnPRansac(object, cv::Mat(1, (int)image_vec.size(), CV_32FC2, &image_vec[0]),
302 camera_mat, cv::Mat(1, 8, CV_32F, cv::Scalar::all(0)),
303 rvec, tvec, false, 200, 2.f, 100, &inliers);
304
305 ASSERT_LE(cv::norm(rvec - rvec_gold), 2e-3);
306 ASSERT_LE(cv::norm(tvec - tvec_gold), 2e-3);
307}
308
309INSTANTIATE_TEST_CASE_P(GPU_Calib3D, SolvePnPRansac, ALL_DEVICES);

314PARAM_TEST_CASE(ReprojectImageTo3D, cv::gpu::DeviceInfo, cv::Size, MatDepth, UseRoi)
315{
316 cv::gpu::DeviceInfo devInfo;
317 cv::Size size;
318 int depth;
319 bool useRoi;
320
321 virtual void SetUp()
322 {
323 devInfo = GET_PARAM(0);
324 size = GET_PARAM(1);
325 depth = GET_PARAM(2);
326 useRoi = GET_PARAM(3);
327
328 cv::gpu::setDevice(devInfo.deviceID());
329 }
330};
331
332GPU_TEST_P(ReprojectImageTo3D, Accuracy)
333{
334 cv::Mat disp = randomMat(size, depth, 5.0, 30.0);
335 cv::Mat Q = randomMat(cv::Size(4, 4), CV_32FC1, 0.1, 1.0);
336
337 cv::gpu::GpuMat dst;
338 cv::gpu::reprojectImageTo3D(loadMat(disp, useRoi), dst, Q, 3);
339
340 cv::Mat dst_gold;
341 cv::reprojectImageTo3D(disp, dst_gold, Q, false);
342
343 EXPECT_MAT_NEAR(dst_gold, dst, 1e-5);
344}
