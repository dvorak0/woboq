43#include "test_precomp.hpp"
44
45#ifdef HAVE_CUDA
46
47#include "opencv2/gpu/stream_accessor.hpp"
48
49using namespace cvtest;
50
51#if CUDART_VERSION >= 5000
52
53struct Async : testing::TestWithParam<cv::gpu::DeviceInfo>
54{
55 cv::gpu::CudaMem src;
56 cv::gpu::GpuMat d_src;
57
58 cv::gpu::CudaMem dst;
59 cv::gpu::GpuMat d_dst;
60
61 virtual void SetUp()
62 {
63 cv::gpu::DeviceInfo devInfo = GetParam();
64 cv::gpu::setDevice(devInfo.deviceID());
65
66 cv::Mat m = randomMat(cv::Size(128, 128), CV_8UC1);
67 src.create(m.size(), m.type(), cv::gpu::CudaMem::ALLOC_PAGE_LOCKED);
68 m.copyTo(src.createMatHeader());
69 }
70};
71
72void checkMemSet(cv::gpu::Stream&, int status, void* userData)
73{
74 ASSERT_EQ(cudaSuccess, status);
75
76 Async* test = reinterpret_cast<Async*>(userData);
77
78 cv::Mat src = test->src;
79 cv::Mat dst = test->dst;
80
81 cv::Mat dst_gold = cv::Mat::zeros(src.size(), src.type());
82
83 ASSERT_MAT_NEAR(dst_gold, dst, 0);
84}
85
86GPU_TEST_P(Async, MemSet)
87{
88 cv::gpu::Stream stream;
89
90 d_dst.upload(src);
91
92 stream.enqueueMemSet(d_dst, cv::Scalar::all(0));
93 stream.enqueueDownload(d_dst, dst);
94
95 Async* test = this;
96 stream.enqueueHostCallback(checkMemSet, test);
97
98 stream.waitForCompletion();
99}
100
101void checkConvert(cv::gpu::Stream&, int status, void* userData)
102{
103 ASSERT_EQ(cudaSuccess, status);
104
105 Async* test = reinterpret_cast<Async*>(userData);
106
107 cv::Mat src = test->src;
108 cv::Mat dst = test->dst;
109
110 cv::Mat dst_gold;
111 src.convertTo(dst_gold, CV_32S);
112
113 ASSERT_MAT_NEAR(dst_gold, dst, 0);
114}
115
116GPU_TEST_P(Async, Convert)
117{
118 cv::gpu::Stream stream;
119
120 stream.enqueueUpload(src, d_src);
121 stream.enqueueConvert(d_src, d_dst, CV_32S);
122 stream.enqueueDownload(d_dst, dst);
123
124 Async* test = this;
125 stream.enqueueHostCallback(checkConvert, test);
126
127 stream.waitForCompletion();
128}
129
130GPU_TEST_P(Async, WrapStream)
131{
132 cudaStream_t cuda_stream = NULL;
133 ASSERT_EQ(cudaSuccess, cudaStreamCreate(&cuda_stream));
134
135 cv::gpu::Stream stream = cv::gpu::StreamAccessor::wrapStream(cuda_stream);
136
137 stream.enqueueUpload(src, d_src);
138 stream.enqueueConvert(d_src, d_dst, CV_32S);
139 stream.enqueueDownload(d_dst, dst);
140
141 Async* test = this;
142 stream.enqueueHostCallback(checkConvert, test);
143
144 stream.waitForCompletion();
145
146 ASSERT_EQ(cudaSuccess, cudaStreamDestroy(cuda_stream));
147}
148
149INSTANTIATE_TEST_CASE_P(GPU_Stream, Async, ALL_DEVICES);
