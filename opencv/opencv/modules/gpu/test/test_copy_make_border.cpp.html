43#include "test_precomp.hpp"

45#ifdef HAVE_CUDA
46
47using namespace cvtest;
48
49namespace
50{
51 IMPLEMENT_PARAM_CLASS(Border, int)
52}
53
54PARAM_TEST_CASE(CopyMakeBorder, cv::gpu::DeviceInfo, cv::Size, MatType, Border, BorderType, UseRoi)
55{
56 cv::gpu::DeviceInfo devInfo;
57 cv::Size size;
58 int type;
59 int border;
60 int borderType;
61 bool useRoi;
62
63 virtual void SetUp()
64 {
65 devInfo = GET_PARAM(0);
66 size = GET_PARAM(1);
67 type = GET_PARAM(2);
68 border = GET_PARAM(3);
69 borderType = GET_PARAM(4);
70 useRoi = GET_PARAM(5);
71
72 cv::gpu::setDevice(devInfo.deviceID());
73 }
74};
75
76GPU_TEST_P(CopyMakeBorder, Accuracy)
77{
78 cv::Mat src = randomMat(size, type);
79 cv::Scalar val = randomScalar(0, 255);
80
81 cv::gpu::GpuMat dst = createMat(cv::Size(size.width + 2 * border, size.height + 2 * border), type, useRoi);
82 cv::gpu::copyMakeBorder(loadMat(src, useRoi), dst, border, border, border, border, borderType, val);
83
84 cv::Mat dst_gold;
85 cv::copyMakeBorder(src, dst_gold, border, border, border, border, borderType, val);
86
87 EXPECT_MAT_NEAR(dst_gold, dst, 0.0);
88}
