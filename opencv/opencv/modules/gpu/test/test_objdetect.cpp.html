91#ifdef DUMP
92 void dump(const cv::Mat& blockHists, const std::vector<cv::Point>& locations)
93 {
94 f.write((char*)&blockHists.rows, sizeof(blockHists.rows));
95 f.write((char*)&blockHists.cols, sizeof(blockHists.cols));
96
97 for (int i = 0; i < blockHists.rows; ++i)
98 {
99 for (int j = 0; j < blockHists.cols; ++j)
100 {
101 float val = blockHists.at<float>(i, j);
102 f.write((char*)&val, sizeof(val));
103 }
104 }
105
106 int nlocations = locations.size();
107 f.write((char*)&nlocations, sizeof(nlocations));
108
109 for (int i = 0; i < locations.size(); ++i)
110 f.write((char*)&locations[i], sizeof(locations[i]));
111 }
112#else
113 void compare(const cv::Mat& blockHists, const std::vector<cv::Point>& locations)
114 {
115 int rows, cols;
116 f.read((char*)&rows, sizeof(rows));
117 f.read((char*)&cols, sizeof(cols));
118 ASSERT_EQ(rows, blockHists.rows);
119 ASSERT_EQ(cols, blockHists.cols);
120
121 for (int i = 0; i < blockHists.rows; ++i)
122 {
123 for (int j = 0; j < blockHists.cols; ++j)
124 {
125 float val;
126 f.read((char*)&val, sizeof(val));
127 ASSERT_NEAR(val, blockHists.at<float>(i, j), 1e-3);
128 }
129 }
130
131 int nlocations;
132 f.read((char*)&nlocations, sizeof(nlocations));
133 ASSERT_EQ(nlocations, static_cast<int>(locations.size()));
134
135 for (int i = 0; i < nlocations; ++i)
136 {
137 cv::Point location;
138 f.read((char*)&location, sizeof(location));
139 ASSERT_EQ(location, locations[i]);
140 }
141 }
142#endif

144 void testDetect(const cv::Mat& img)
145 {
146 hog->gamma_correction = false;
147 hog->setSVMDetector(cv::gpu::HOGDescriptor::getDefaultPeopleDetector());
148
149 std::vector<cv::Point> locations;
150
151 // Test detect
152 hog->detect(loadMat(img), locations, 0);
153
154#ifdef DUMP
155 dump(cv::Mat(hog->getBlockHist()), locations);
156#else
157 compare(cv::Mat(hog->getBlockHists()), locations);
158#endif
159
160 // Test detect on smaller image
161 cv::Mat img2;
162 cv::resize(img, img2, cv::Size(img.cols / 2, img.rows / 2));
163 hog->detect(loadMat(img2), locations, 0);
164
165#ifdef DUMP
166 dump(cv::Mat(hog->getBlockHist()), locations);
167#else
168 compare(cv::Mat(hog->getBlockHists()), locations);
169#endif
170
171 // Test detect on greater image
172 cv::resize(img, img2, cv::Size(img.cols * 2, img.rows * 2));
173 hog->detect(loadMat(img2), locations, 0);
174
175#ifdef DUMP
176 dump(cv::Mat(hog->getBlockHist()), locations);
177#else
178 compare(cv::Mat(hog->getBlockHists()), locations);
179#endif
180 }
181
182 // Does not compare border value, as interpolation leads to delta
183 void compare_inner_parts(cv::Mat d1, cv::Mat d2)
184 {
185 for (int i = 1; i < blocks_per_win_y - 1; ++i)
186 for (int j = 1; j < blocks_per_win_x - 1; ++j)
187 for (int k = 0; k < block_hist_size; ++k)
188 {
189 float a = d1.at<float>(0, (i * blocks_per_win_x + j) * block_hist_size);
190 float b = d2.at<float>(0, (i * blocks_per_win_x + j) * block_hist_size);
191 ASSERT_FLOAT_EQ(a, b);
192 }
193 }
194};

222GPU_TEST_P(HOG, GetDescriptors)
223{
224 // Load image (e.g. train data, composed from windows)
225 cv::Mat img_rgb = readImage("hog/train_data.png");
226 ASSERT_FALSE(img_rgb.empty());
227
228 // Convert to C4
229 cv::Mat img;
230 cv::cvtColor(img_rgb, img, CV_BGR2BGRA);
231
232 cv::gpu::GpuMat d_img(img);
233
234 // Convert train images into feature vectors (train table)
235 cv::gpu::GpuMat descriptors, descriptors_by_cols;
236 hog->getDescriptors(d_img, hog->win_size, descriptors, cv::gpu::HOGDescriptor::DESCR_FORMAT_ROW_BY_ROW);
237 hog->getDescriptors(d_img, hog->win_size, descriptors_by_cols, cv::gpu::HOGDescriptor::DESCR_FORMAT_COL_BY_COL);
238
239 // Check size of the result train table
240 wins_per_img_x = 3;
241 wins_per_img_y = 2;
242 blocks_per_win_x = 7;
243 blocks_per_win_y = 15;
244 block_hist_size = 36;
245 cv::Size descr_size_expected = cv::Size(blocks_per_win_x * blocks_per_win_y * block_hist_size,
246 wins_per_img_x * wins_per_img_y);
247 ASSERT_EQ(descr_size_expected, descriptors.size());
248
249 // Check both formats of output descriptors are handled correctly
250 cv::Mat dr(descriptors);
251 cv::Mat dc(descriptors_by_cols);
252 for (int i = 0; i < wins_per_img_x * wins_per_img_y; ++i)
253 {
254 const float* l = dr.rowRange(i, i + 1).ptr<float>();
255 const float* r = dc.rowRange(i, i + 1).ptr<float>();
256 for (int y = 0; y < blocks_per_win_y; ++y)
257 for (int x = 0; x < blocks_per_win_x; ++x)
258 for (int k = 0; k < block_hist_size; ++k)
259 ASSERT_EQ(l[(y * blocks_per_win_x + x) * block_hist_size + k],
260 r[(x * blocks_per_win_y + y) * block_hist_size + k]);
261 }
262
263 /* Now we want to extract the same feature vectors, but from single images. NOTE: results will
264 be defferent, due to border values interpolation. Using of many small images is slower, however we
265 wont't call getDescriptors and will use computeBlockHistograms instead of. computeBlockHistograms
266 works good, it can be checked in the gpu_hog sample */
267
268 img_rgb = readImage("hog/positive1.png");
269 ASSERT_TRUE(!img_rgb.empty());
270 cv::cvtColor(img_rgb, img, CV_BGR2BGRA);
271 hog->computeBlockHistograms(cv::gpu::GpuMat(img));
272 // Everything is fine with interpolation for left top subimage
273 ASSERT_EQ(0.0, cv::norm(cv::Mat(hog->getBlockHists()), (cv::Mat)descriptors.rowRange(0, 1)));
274
275 img_rgb = readImage("hog/positive2.png");
276 ASSERT_TRUE(!img_rgb.empty());
277 cv::cvtColor(img_rgb, img, CV_BGR2BGRA);
278 hog->computeBlockHistograms(cv::gpu::GpuMat(img));
279 compare_inner_parts(cv::Mat(hog->getBlockHists()), cv::Mat(descriptors.rowRange(1, 2)));
280
281 img_rgb = readImage("hog/negative1.png");
282 ASSERT_TRUE(!img_rgb.empty());
283 cv::cvtColor(img_rgb, img, CV_BGR2BGRA);
284 hog->computeBlockHistograms(cv::gpu::GpuMat(img));
285 compare_inner_parts(cv::Mat(hog->getBlockHists()), cv::Mat(descriptors.rowRange(2, 3)));
286
287 img_rgb = readImage("hog/negative2.png");
288 ASSERT_TRUE(!img_rgb.empty());
289 cv::cvtColor(img_rgb, img, CV_BGR2BGRA);
290 hog->computeBlockHistograms(cv::gpu::GpuMat(img));
291 compare_inner_parts(cv::Mat(hog->getBlockHists()), cv::Mat(descriptors.rowRange(3, 4)));
292
293 img_rgb = readImage("hog/positive3.png");
294 ASSERT_TRUE(!img_rgb.empty());
295 cv::cvtColor(img_rgb, img, CV_BGR2BGRA);
296 hog->computeBlockHistograms(cv::gpu::GpuMat(img));
297 compare_inner_parts(cv::Mat(hog->getBlockHists()), cv::Mat(descriptors.rowRange(4, 5)));
298
299 img_rgb = readImage("hog/negative3.png");
300 ASSERT_TRUE(!img_rgb.empty());
301 cv::cvtColor(img_rgb, img, CV_BGR2BGRA);
302 hog->computeBlockHistograms(cv::gpu::GpuMat(img));
303 compare_inner_parts(cv::Mat(hog->getBlockHists()), cv::Mat(descriptors.rowRange(5, 6)));
304}

325GPU_TEST_P(CalTech, HOG)
326{
327 cv::gpu::GpuMat d_img(img);
328 cv::Mat markedImage(img.clone());
329
330 cv::gpu::HOGDescriptor d_hog;
331 d_hog.setSVMDetector(cv::gpu::HOGDescriptor::getDefaultPeopleDetector());
332 d_hog.nlevels = d_hog.nlevels + 32;
333
334 std::vector<cv::Rect> found_locations;
335 d_hog.detectMultiScale(d_img, found_locations);
336
337#if defined (LOG_CASCADE_STATISTIC)
338 for (int i = 0; i < (int)found_locations.size(); i++)
339 {
340 cv::Rect r = found_locations[i];
341
342 std::cout << r.x << " " << r.y << " " << r.width << " " << r.height << std::endl;
343 cv::rectangle(markedImage, r , CV_RGB(255, 0, 0));
344 }
345
346 cv::imshow("Res", markedImage); cv::waitKey();
347#endif
348}

383PARAM_TEST_CASE(LBP_classify, cv::gpu::DeviceInfo, int)
384{
385 cv::gpu::DeviceInfo devInfo;
386
387 virtual void SetUp()
388 {
389 devInfo = GET_PARAM(0);
390 cv::gpu::setDevice(devInfo.deviceID());
391 }
392};
393
394GPU_TEST_P(LBP_classify, Accuracy)
395{
396 std::string classifierXmlPath = std::string(cvtest::TS::ptr()->get_data_path()) + "lbpcascade/lbpcascade_frontalface.xml";
397 std::string imagePath = std::string(cvtest::TS::ptr()->get_data_path()) + "lbpcascade/er.png";
398
399 cv::CascadeClassifier cpuClassifier(classifierXmlPath);
400 ASSERT_FALSE(cpuClassifier.empty());
401
402 cv::Mat image = cv::imread(imagePath);
403 image = image.colRange(0, image.cols/2);
404 cv::Mat grey;
405 cvtColor(image, grey, CV_BGR2GRAY);
406 ASSERT_FALSE(image.empty());
407
408 std::vector<cv::Rect> rects;
409 cpuClassifier.detectMultiScale(grey, rects);
410 cv::Mat markedImage = image.clone();
411
412 std::vector<cv::Rect>::iterator it = rects.begin();
413 for (; it != rects.end(); ++it)
414 cv::rectangle(markedImage, *it, CV_RGB(0, 0, 255));
415
416 cv::gpu::CascadeClassifier_GPU gpuClassifier;
417 ASSERT_TRUE(gpuClassifier.load(classifierXmlPath));
418
419 cv::gpu::GpuMat gpu_rects;
420 cv::gpu::GpuMat tested(grey);
421 int count = gpuClassifier.detectMultiScale(tested, gpu_rects);
422
423#if defined (LOG_CASCADE_STATISTIC)
424 cv::Mat downloaded(gpu_rects);
425 const cv::Rect* faces = downloaded.ptr<cv::Rect>();
426 for (int i = 0; i < count; i++)
427 {
428 cv::Rect r = faces[i];
429
430 std::cout << r.x << " " << r.y << " " << r.width << " " << r.height << std::endl;
431 cv::rectangle(markedImage, r , CV_RGB(255, 0, 0));
432 }
433#endif
434
435#if defined (LOG_CASCADE_STATISTIC)
436 cv::imshow("Res", markedImage); cv::waitKey();
437#endif
438 (void)count;
439}
