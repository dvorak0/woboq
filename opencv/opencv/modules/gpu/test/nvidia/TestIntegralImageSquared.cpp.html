73bool TestIntegralImageSquared::process()
74{
75 NCVStatus ncvStat;
76 bool rcode = false;
77
78 Ncv32u widthSII = this->width + 1;
79 Ncv32u heightSII = this->height + 1;
80
81 NCVMatrixAlloc<Ncv8u> d_img(*this->allocatorGPU.get(), this->width, this->height);
82 ncvAssertReturn(d_img.isMemAllocated(), false);
83 NCVMatrixAlloc<Ncv8u> h_img(*this->allocatorCPU.get(), this->width, this->height);
84 ncvAssertReturn(h_img.isMemAllocated(), false);
85 NCVMatrixAlloc<Ncv64u> d_imgSII(*this->allocatorGPU.get(), widthSII, heightSII);
86 ncvAssertReturn(d_imgSII.isMemAllocated(), false);
87 NCVMatrixAlloc<Ncv64u> h_imgSII(*this->allocatorCPU.get(), widthSII, heightSII);
88 ncvAssertReturn(h_imgSII.isMemAllocated(), false);
89 NCVMatrixAlloc<Ncv64u> h_imgSII_d(*this->allocatorCPU.get(), widthSII, heightSII);
90 ncvAssertReturn(h_imgSII_d.isMemAllocated(), false);
91
92 Ncv32u bufSize;
93 ncvStat = nppiStSqrIntegralGetSize_8u64u(NcvSize32u(this->width, this->height), &bufSize, this->devProp);
94 ncvAssertReturn(NPPST_SUCCESS == ncvStat, false);
95 NCVVectorAlloc<Ncv8u> d_tmpBuf(*this->allocatorGPU.get(), bufSize);
96 ncvAssertReturn(d_tmpBuf.isMemAllocated(), false);
97
98 NCV_SET_SKIP_COND(this->allocatorGPU.get()->isCounting());
99 NCV_SKIP_COND_BEGIN
100
101 ncvAssertReturn(this->src.fill(h_img), false);
102
103 ncvStat = h_img.copySolid(d_img, 0);
104 ncvAssertReturn(ncvStat == NPPST_SUCCESS, false);
105
106 ncvStat = nppiStSqrIntegral_8u64u_C1R(d_img.ptr(), d_img.pitch(),
107 d_imgSII.ptr(), d_imgSII.pitch(),
108 NcvSize32u(this->width, this->height),
109 d_tmpBuf.ptr(), bufSize, this->devProp);
110 ncvAssertReturn(ncvStat == NPPST_SUCCESS, false);
111
112 ncvStat = d_imgSII.copySolid(h_imgSII_d, 0);
113 ncvAssertReturn(ncvStat == NPPST_SUCCESS, false);
114
115 ncvStat = nppiStSqrIntegral_8u64u_C1R_host(h_img.ptr(), h_img.pitch(),
116 h_imgSII.ptr(), h_imgSII.pitch(),
117 NcvSize32u(this->width, this->height));
118 ncvAssertReturn(ncvStat == NPPST_SUCCESS, false);
119
120 NCV_SKIP_COND_END
121
122 //bit-to-bit check
123 bool bLoopVirgin = true;
124
125 NCV_SKIP_COND_BEGIN
126 for (Ncv32u i=0; bLoopVirgin && i < h_img.height() + 1; i++)
127 {
128 for (Ncv32u j=0; bLoopVirgin && j < h_img.width() + 1; j++)
129 {
130 if (h_imgSII.ptr()[h_imgSII.stride()*i+j] != h_imgSII_d.ptr()[h_imgSII_d.stride()*i+j])
131 {
132 bLoopVirgin = false;
133 }
134 }
135 }
136 NCV_SKIP_COND_END
137
138 if (bLoopVirgin)
139 {
140 rcode = true;
141 }
142
143 return rcode;
144}
