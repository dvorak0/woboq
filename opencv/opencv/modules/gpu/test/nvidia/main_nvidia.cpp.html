76template <class T_in, class T_out>
77void generateIntegralTests(NCVAutoTestLister &testLister,
78 NCVTestSourceProvider<T_in> &src,
79 Ncv32u maxWidth, Ncv32u maxHeight)
80{
81 for (Ncv32f _i=1.0; _i<maxWidth; _i*=1.2f)
82 {
83 Ncv32u i = (Ncv32u)_i;
84 char testName[80];
85 sprintf(testName, "LinIntImgW%dH%d", i, 2);
86 testLister.add(new TestIntegralImage<T_in, T_out>(testName, src, i, 2));
87 }
88 for (Ncv32f _i=1.0; _i<maxHeight; _i*=1.2f)
89 {
90 Ncv32u i = (Ncv32u)_i;
91 char testName[80];
92 sprintf(testName, "LinIntImgW%dH%d", 2, i);
93 testLister.add(new TestIntegralImage<T_in, T_out>(testName, src, 2, i));
94 }
95
96 testLister.add(new TestIntegralImage<T_in, T_out>("LinIntImg_VGA", src, 640, 480));
97}

222template <class T>
223void generateDrawRectsTests(NCVAutoTestLister &testLister,
224 NCVTestSourceProvider<T> &src,
225 NCVTestSourceProvider<Ncv32u> &src32u,
226 Ncv32u maxWidth, Ncv32u maxHeight)
227{
228 for (Ncv32f _i=16.0; _i<maxWidth; _i*=1.1f)
229 {
230 Ncv32u i = (Ncv32u)_i;
231 Ncv32u j = maxHeight * i / maxWidth;
232 if (!j) continue;
233 char testName[80];
234 sprintf(testName, "DrawRectsW%dH%d", i, j);
235
236 if (sizeof(T) == sizeof(Ncv32u))
237 {
238 testLister.add(new TestDrawRects<T>(testName, src, src32u, i, j, i*j/1000+1, (T)0xFFFFFFFF));
239 }
240 else if (sizeof(T) == sizeof(Ncv8u))
241 {
242 testLister.add(new TestDrawRects<T>(testName, src, src32u, i, j, i*j/1000+1, (T)0xFF));
243 }
244 else
245 {
246 ncvAssertPrintCheck(false, "Attempted to instantiate non-existing DrawRects test suite");
247 }
248 }
249
250 //test VGA
251 testLister.add(new TestDrawRects<T>("DrawRects_VGA", src, src32u, 640, 480, 640*480/1000, (T)0xFF));
252}

323bool nvidia_NPPST_Integral_Image(const std::string& test_data_path, OutputLevel outputLevel)
324{
325 path = test_data_path.c_str();
326 if (outputLevel != OutputLevelFull)
327 ncvSetDebugOutputHandler(devNullOutput);
328
329 NCVAutoTestLister testListerII("NPPST Integral Image", outputLevel);
330
331 NCVTestSourceProvider<Ncv8u> testSrcRandom_8u(2010, 0, 255, 2048, 2048);
332 NCVTestSourceProvider<Ncv32f> testSrcRandom_32f(2010, -1.0f, 1.0f, 2048, 2048);
333
334 generateIntegralTests<Ncv8u, Ncv32u>(testListerII, testSrcRandom_8u, 2048, 2048);
335 generateIntegralTests<Ncv32f, Ncv32f>(testListerII, testSrcRandom_32f, 2048, 2048);
