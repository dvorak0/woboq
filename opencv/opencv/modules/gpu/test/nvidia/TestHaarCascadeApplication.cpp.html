53 // http://www.christian-seiler.de/projekte/fpmath/
54 class FpuControl
55 {
56 public:
57 FpuControl();
58 ~FpuControl();
59
60 private:
61 #if defined(__GNUC__) && !defined(__APPLE__) && !defined(__arm__) && !defined (__aarch64__)
62 fpu_control_t fpu_oldcw, fpu_cw;
63 #elif defined(_WIN32) && !defined(_WIN64)
64 unsigned int fpu_oldcw, fpu_cw;
65 #endif
66 };
67
68 FpuControl::FpuControl()
69 {
70 #if defined(__GNUC__) && !defined(__APPLE__) && !defined(__arm__) && !defined(__aarch64__)
71 _FPU_GETCW(fpu_oldcw);
72 fpu_cw = (fpu_oldcw & ~_FPU_EXTENDED & ~_FPU_DOUBLE & ~_FPU_SINGLE) | _FPU_SINGLE;
73 _FPU_SETCW(fpu_cw);
74 #elif defined(_WIN32) && !defined(_WIN64)
75 _controlfp_s(&fpu_cw, 0, 0);
76 fpu_oldcw = fpu_cw;
77 _controlfp_s(&fpu_cw, _PC_24, _MCW_PC);
78 #endif
79 }
80
81 FpuControl::~FpuControl()
82 {
83 #if defined(__GNUC__) && !defined(__APPLE__) && !defined(__arm__) && !defined(__aarch64__)
84 _FPU_SETCW(fpu_oldcw);
85 #elif defined(_WIN32) && !defined(_WIN64)
86 _controlfp_s(&fpu_cw, fpu_oldcw, _MCW_PC);
87 #endif
88 }
89}

121bool TestHaarCascadeApplication::process()
122{
123 NCVStatus ncvStat;
124 bool rcode = false;
125
126 Ncv32u numStages, numNodes, numFeatures;
127
128 ncvStat = ncvHaarGetClassifierSize(this->cascadeName, numStages, numNodes, numFeatures);
129 ncvAssertReturn(ncvStat == NCV_SUCCESS, false);
130
131 NCVVectorAlloc<HaarStage64> h_HaarStages(*this->allocatorCPU.get(), numStages);
132 ncvAssertReturn(h_HaarStages.isMemAllocated(), false);
133 NCVVectorAlloc<HaarClassifierNode128> h_HaarNodes(*this->allocatorCPU.get(), numNodes);
134 ncvAssertReturn(h_HaarNodes.isMemAllocated(), false);
135 NCVVectorAlloc<HaarFeature64> h_HaarFeatures(*this->allocatorCPU.get(), numFeatures);
136 ncvAssertReturn(h_HaarFeatures.isMemAllocated(), false);
137
138 NCVVectorAlloc<HaarStage64> d_HaarStages(*this->allocatorGPU.get(), numStages);
139 ncvAssertReturn(d_HaarStages.isMemAllocated(), false);
140 NCVVectorAlloc<HaarClassifierNode128> d_HaarNodes(*this->allocatorGPU.get(), numNodes);
141 ncvAssertReturn(d_HaarNodes.isMemAllocated(), false);
142 NCVVectorAlloc<HaarFeature64> d_HaarFeatures(*this->allocatorGPU.get(), numFeatures);
143 ncvAssertReturn(d_HaarFeatures.isMemAllocated(), false);
144
145 HaarClassifierCascadeDescriptor haar;
146 haar.ClassifierSize.width = haar.ClassifierSize.height = 1;
147 haar.bNeedsTiltedII = false;
148 haar.NumClassifierRootNodes = numNodes;
149 haar.NumClassifierTotalNodes = numNodes;
150 haar.NumFeatures = numFeatures;
151 haar.NumStages = numStages;
152
153 NCV_SET_SKIP_COND(this->allocatorGPU.get()->isCounting());
154 NCV_SKIP_COND_BEGIN
155
156 ncvStat = ncvHaarLoadFromFile_host(this->cascadeName, haar, h_HaarStages, h_HaarNodes, h_HaarFeatures);
157 ncvAssertReturn(ncvStat == NCV_SUCCESS, false);
158
159 ncvAssertReturn(NCV_SUCCESS == h_HaarStages.copySolid(d_HaarStages, 0), false);
160 ncvAssertReturn(NCV_SUCCESS == h_HaarNodes.copySolid(d_HaarNodes, 0), false);
161 ncvAssertReturn(NCV_SUCCESS == h_HaarFeatures.copySolid(d_HaarFeatures, 0), false);
162 ncvAssertCUDAReturn(cudaStreamSynchronize(0), false);
163
164 NCV_SKIP_COND_END
165
166 NcvSize32s srcRoi, srcIIRoi, searchRoi;
167 srcRoi.width = this->width;
168 srcRoi.height = this->height;
169 srcIIRoi.width = srcRoi.width + 1;
170 srcIIRoi.height = srcRoi.height + 1;
171 searchRoi.width = srcIIRoi.width - haar.ClassifierSize.width;
172 searchRoi.height = srcIIRoi.height - haar.ClassifierSize.height;
173 if (searchRoi.width <= 0 || searchRoi.height <= 0)
174 {
175 return false;
176 }
177 NcvSize32u searchRoiU(searchRoi.width, searchRoi.height);
178
179 NCVMatrixAlloc<Ncv8u> d_img(*this->allocatorGPU.get(), this->width, this->height);
180 ncvAssertReturn(d_img.isMemAllocated(), false);
181 NCVMatrixAlloc<Ncv8u> h_img(*this->allocatorCPU.get(), this->width, this->height);
182 ncvAssertReturn(h_img.isMemAllocated(), false);
183
184 Ncv32u integralWidth = this->width + 1;
185 Ncv32u integralHeight = this->height + 1;
186
187 NCVMatrixAlloc<Ncv32u> d_integralImage(*this->allocatorGPU.get(), integralWidth, integralHeight);
188 ncvAssertReturn(d_integralImage.isMemAllocated(), false);
189 NCVMatrixAlloc<Ncv64u> d_sqIntegralImage(*this->allocatorGPU.get(), integralWidth, integralHeight);
190 ncvAssertReturn(d_sqIntegralImage.isMemAllocated(), false);
191 NCVMatrixAlloc<Ncv32u> h_integralImage(*this->allocatorCPU.get(), integralWidth, integralHeight);
192 ncvAssertReturn(h_integralImage.isMemAllocated(), false);
193 NCVMatrixAlloc<Ncv64u> h_sqIntegralImage(*this->allocatorCPU.get(), integralWidth, integralHeight);
194 ncvAssertReturn(h_sqIntegralImage.isMemAllocated(), false);
195
196 NCVMatrixAlloc<Ncv32f> d_rectStdDev(*this->allocatorGPU.get(), this->width, this->height);
197 ncvAssertReturn(d_rectStdDev.isMemAllocated(), false);
198 NCVMatrixAlloc<Ncv32u> d_pixelMask(*this->allocatorGPU.get(), this->width, this->height);
199 ncvAssertReturn(d_pixelMask.isMemAllocated(), false);
200 NCVMatrixAlloc<Ncv32f> h_rectStdDev(*this->allocatorCPU.get(), this->width, this->height);
201 ncvAssertReturn(h_rectStdDev.isMemAllocated(), false);
202 NCVMatrixAlloc<Ncv32u> h_pixelMask(*this->allocatorCPU.get(), this->width, this->height);
203 ncvAssertReturn(h_pixelMask.isMemAllocated(), false);
204
205 NCVVectorAlloc<NcvRect32u> d_hypotheses(*this->allocatorGPU.get(), this->width * this->height);
206 ncvAssertReturn(d_hypotheses.isMemAllocated(), false);
207 NCVVectorAlloc<NcvRect32u> h_hypotheses(*this->allocatorCPU.get(), this->width * this->height);
208 ncvAssertReturn(h_hypotheses.isMemAllocated(), false);
209
210 NCVStatus nppStat;
211 Ncv32u szTmpBufIntegral, szTmpBufSqIntegral;
212 nppStat = nppiStIntegralGetSize_8u32u(NcvSize32u(this->width, this->height), &szTmpBufIntegral, this->devProp);
213 ncvAssertReturn(nppStat == NPPST_SUCCESS, false);
214 nppStat = nppiStSqrIntegralGetSize_8u64u(NcvSize32u(this->width, this->height), &szTmpBufSqIntegral, this->devProp);
215 ncvAssertReturn(nppStat == NPPST_SUCCESS, false);
216 NCVVectorAlloc<Ncv8u> d_tmpIIbuf(*this->allocatorGPU.get(), std::max(szTmpBufIntegral, szTmpBufSqIntegral));
217 ncvAssertReturn(d_tmpIIbuf.isMemAllocated(), false);
218
219 Ncv32u detectionsOnThisScale_d = 0;
220 Ncv32u detectionsOnThisScale_h = 0;
221
222 NCV_SKIP_COND_BEGIN
223
224 ncvAssertReturn(this->src.fill(h_img), false);
225 ncvStat = h_img.copySolid(d_img, 0);
226 ncvAssertReturn(ncvStat == NCV_SUCCESS, false);
227 ncvAssertCUDAReturn(cudaStreamSynchronize(0), false);
228
229 nppStat = nppiStIntegral_8u32u_C1R(d_img.ptr(), d_img.pitch(),
230 d_integralImage.ptr(), d_integralImage.pitch(),
231 NcvSize32u(d_img.width(), d_img.height()),
232 d_tmpIIbuf.ptr(), szTmpBufIntegral, this->devProp);
233 ncvAssertReturn(nppStat == NPPST_SUCCESS, false);
234
235 nppStat = nppiStSqrIntegral_8u64u_C1R(d_img.ptr(), d_img.pitch(),
236 d_sqIntegralImage.ptr(), d_sqIntegralImage.pitch(),
237 NcvSize32u(d_img.width(), d_img.height()),
238 d_tmpIIbuf.ptr(), szTmpBufSqIntegral, this->devProp);
239 ncvAssertReturn(nppStat == NPPST_SUCCESS, false);
240
241 const NcvRect32u rect(
242 HAAR_STDDEV_BORDER,
243 HAAR_STDDEV_BORDER,
244 haar.ClassifierSize.width - 2*HAAR_STDDEV_BORDER,
245 haar.ClassifierSize.height - 2*HAAR_STDDEV_BORDER);
246 nppStat = nppiStRectStdDev_32f_C1R(
247 d_integralImage.ptr(), d_integralImage.pitch(),
248 d_sqIntegralImage.ptr(), d_sqIntegralImage.pitch(),
249 d_rectStdDev.ptr(), d_rectStdDev.pitch(),
250 NcvSize32u(searchRoi.width, searchRoi.height), rect,
251 1.0f, true);
252 ncvAssertReturn(nppStat == NPPST_SUCCESS, false);
253
254 ncvStat = d_integralImage.copySolid(h_integralImage, 0);
255 ncvAssertReturn(ncvStat == NCV_SUCCESS, false);
256 ncvStat = d_rectStdDev.copySolid(h_rectStdDev, 0);
257 ncvAssertReturn(ncvStat == NCV_SUCCESS, false);
258
259 for (Ncv32u i=0; i<searchRoiU.height; i++)
260 {
261 for (Ncv32u j=0; j<h_pixelMask.stride(); j++)
262 {
263 if (j<searchRoiU.width)
264 {
265 h_pixelMask.ptr()[i*h_pixelMask.stride()+j] = (i << 16) | j;
266 }
267 else
268 {
269 h_pixelMask.ptr()[i*h_pixelMask.stride()+j] = OBJDET_MASK_ELEMENT_INVALID_32U;
270 }
271 }
272 }
273 ncvAssertReturn(cudaSuccess == cudaStreamSynchronize(0), false);
274
275 {
276 // calculations here
277 FpuControl fpu;
278 (void) fpu;
279
280 ncvStat = ncvApplyHaarClassifierCascade_host(
281 h_integralImage, h_rectStdDev, h_pixelMask,
282 detectionsOnThisScale_h,
283 haar, h_HaarStages, h_HaarNodes, h_HaarFeatures, false,
284 searchRoiU, 1, 1.0f);
285 ncvAssertReturn(ncvStat == NCV_SUCCESS, false);
286 }
287
288 NCV_SKIP_COND_END
289
290 int devId;
291 ncvAssertCUDAReturn(cudaGetDevice(&devId), false);
292 cudaDeviceProp _devProp;
293 ncvAssertCUDAReturn(cudaGetDeviceProperties(&_devProp, devId), false);
294
295 ncvStat = ncvApplyHaarClassifierCascade_device(
296 d_integralImage, d_rectStdDev, d_pixelMask,
297 detectionsOnThisScale_d,
298 haar, h_HaarStages, d_HaarStages, d_HaarNodes, d_HaarFeatures, false,
299 searchRoiU, 1, 1.0f,
300 *this->allocatorGPU.get(), *this->allocatorCPU.get(),
301 _devProp, 0);
302 ncvAssertReturn(ncvStat == NCV_SUCCESS, false);
303
304 NCVMatrixAlloc<Ncv32u> h_pixelMask_d(*this->allocatorCPU.get(), this->width, this->height);
305 ncvAssertReturn(h_pixelMask_d.isMemAllocated(), false);
306
307 //bit-to-bit check
308 bool bLoopVirgin = true;
309
310 NCV_SKIP_COND_BEGIN
311
312 ncvStat = d_pixelMask.copySolid(h_pixelMask_d, 0);
313 ncvAssertReturn(ncvStat == NCV_SUCCESS, false);
314
315 if (detectionsOnThisScale_d != detectionsOnThisScale_h)
316 {
317 bLoopVirgin = false;
318 }
319 else
320 {
321 std::sort(h_pixelMask_d.ptr(), h_pixelMask_d.ptr() + detectionsOnThisScale_d);
322 for (Ncv32u i=0; i<detectionsOnThisScale_d && bLoopVirgin; i++)
323 {
324 if (h_pixelMask.ptr()[i] != h_pixelMask_d.ptr()[i])
325 {
326 bLoopVirgin = false;
327 }
328 }
329 }
330
331 NCV_SKIP_COND_END
332
333 if (bLoopVirgin)
334 {
335 rcode = true;
336 }
337
338 return rcode;
339}
