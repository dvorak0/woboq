82bool TestRectStdDev::process()
83{
84 NCVStatus ncvStat;
85 bool rcode = false;
86
87 Ncv32s _normWidth = (Ncv32s)this->width - this->rect.x - this->rect.width + 1;
88 Ncv32s _normHeight = (Ncv32s)this->height - this->rect.y - this->rect.height + 1;
89 if (_normWidth <= 0 || _normHeight <= 0)
90 {
91 return true;
92 }
93 Ncv32u normWidth = (Ncv32u)_normWidth;
94 Ncv32u normHeight = (Ncv32u)_normHeight;
95 NcvSize32u szNormRoi(normWidth, normHeight);
96
97 Ncv32u widthII = this->width + 1;
98 Ncv32u heightII = this->height + 1;
99 Ncv32u widthSII = this->width + 1;
100 Ncv32u heightSII = this->height + 1;
101
102 NCVMatrixAlloc<Ncv8u> d_img(*this->allocatorGPU.get(), this->width, this->height);
103 ncvAssertReturn(d_img.isMemAllocated(), false);
104 NCVMatrixAlloc<Ncv8u> h_img(*this->allocatorCPU.get(), this->width, this->height);
105 ncvAssertReturn(h_img.isMemAllocated(), false);
106
107 NCVMatrixAlloc<Ncv32u> d_imgII(*this->allocatorGPU.get(), widthII, heightII);
108 ncvAssertReturn(d_imgII.isMemAllocated(), false);
109 NCVMatrixAlloc<Ncv32u> h_imgII(*this->allocatorCPU.get(), widthII, heightII);
110 ncvAssertReturn(h_imgII.isMemAllocated(), false);
111
112 NCVMatrixAlloc<Ncv64u> d_imgSII(*this->allocatorGPU.get(), widthSII, heightSII);
113 ncvAssertReturn(d_imgSII.isMemAllocated(), false);
114 NCVMatrixAlloc<Ncv64u> h_imgSII(*this->allocatorCPU.get(), widthSII, heightSII);
115 ncvAssertReturn(h_imgSII.isMemAllocated(), false);
116
117 NCVMatrixAlloc<Ncv32f> d_norm(*this->allocatorGPU.get(), normWidth, normHeight);
118 ncvAssertReturn(d_norm.isMemAllocated(), false);
119 NCVMatrixAlloc<Ncv32f> h_norm(*this->allocatorCPU.get(), normWidth, normHeight);
120 ncvAssertReturn(h_norm.isMemAllocated(), false);
121 NCVMatrixAlloc<Ncv32f> h_norm_d(*this->allocatorCPU.get(), normWidth, normHeight);
122 ncvAssertReturn(h_norm_d.isMemAllocated(), false);
123
124 Ncv32u bufSizeII, bufSizeSII;
125 ncvStat = nppiStIntegralGetSize_8u32u(NcvSize32u(this->width, this->height), &bufSizeII, this->devProp);
126 ncvAssertReturn(NPPST_SUCCESS == ncvStat, false);
127 ncvStat = nppiStSqrIntegralGetSize_8u64u(NcvSize32u(this->width, this->height), &bufSizeSII, this->devProp);
128 ncvAssertReturn(NPPST_SUCCESS == ncvStat, false);
129 Ncv32u bufSize = bufSizeII > bufSizeSII ? bufSizeII : bufSizeSII;
130 NCVVectorAlloc<Ncv8u> d_tmpBuf(*this->allocatorGPU.get(), bufSize);
131 ncvAssertReturn(d_tmpBuf.isMemAllocated(), false);
132
133 NCV_SET_SKIP_COND(this->allocatorGPU.get()->isCounting());
134 NCV_SKIP_COND_BEGIN
135 ncvAssertReturn(this->src.fill(h_img), false);
136
137 ncvStat = h_img.copySolid(d_img, 0);
138 ncvAssertReturn(ncvStat == NPPST_SUCCESS, false);
139
140 ncvStat = nppiStIntegral_8u32u_C1R(d_img.ptr(), d_img.pitch(),
141 d_imgII.ptr(), d_imgII.pitch(),
142 NcvSize32u(this->width, this->height),
143 d_tmpBuf.ptr(), bufSize, this->devProp);
144 ncvAssertReturn(ncvStat == NPPST_SUCCESS, false);
145
146 ncvStat = nppiStSqrIntegral_8u64u_C1R(d_img.ptr(), d_img.pitch(),
147 d_imgSII.ptr(), d_imgSII.pitch(),
148 NcvSize32u(this->width, this->height),
149 d_tmpBuf.ptr(), bufSize, this->devProp);
150 ncvAssertReturn(ncvStat == NPPST_SUCCESS, false);
151
152 ncvStat = nppiStRectStdDev_32f_C1R(d_imgII.ptr(), d_imgII.pitch(),
153 d_imgSII.ptr(), d_imgSII.pitch(),
154 d_norm.ptr(), d_norm.pitch(),
155 szNormRoi, this->rect,
156 this->scaleFactor,
157 this->bTextureCache);
158 ncvAssertReturn(ncvStat == NPPST_SUCCESS, false);
159
160 ncvStat = d_norm.copySolid(h_norm_d, 0);
161 ncvAssertReturn(ncvStat == NPPST_SUCCESS, false);
162
163 ncvStat = nppiStIntegral_8u32u_C1R_host(h_img.ptr(), h_img.pitch(),
164 h_imgII.ptr(), h_imgII.pitch(),
165 NcvSize32u(this->width, this->height));
166 ncvAssertReturn(ncvStat == NPPST_SUCCESS, false);
167
168 ncvStat = nppiStSqrIntegral_8u64u_C1R_host(h_img.ptr(), h_img.pitch(),
169 h_imgSII.ptr(), h_imgSII.pitch(),
170 NcvSize32u(this->width, this->height));
171 ncvAssertReturn(ncvStat == NPPST_SUCCESS, false);
172
173 ncvStat = nppiStRectStdDev_32f_C1R_host(h_imgII.ptr(), h_imgII.pitch(),
174 h_imgSII.ptr(), h_imgSII.pitch(),
175 h_norm.ptr(), h_norm.pitch(),
176 szNormRoi, this->rect,
177 this->scaleFactor);
178 ncvAssertReturn(ncvStat == NPPST_SUCCESS, false);
179 NCV_SKIP_COND_END
180
181 //bit-to-bit check
182 bool bLoopVirgin = true;
183
184 NCV_SKIP_COND_BEGIN
185 const Ncv64f relEPS = 0.005;
186 for (Ncv32u i=0; bLoopVirgin && i < h_norm.height(); i++)
187 {
188 for (Ncv32u j=0; bLoopVirgin && j < h_norm.width(); j++)
189 {
190 Ncv64f absErr = fabs(h_norm.ptr()[h_norm.stride()*i+j] - h_norm_d.ptr()[h_norm_d.stride()*i+j]);
191 Ncv64f relErr = absErr / h_norm.ptr()[h_norm.stride()*i+j];
192
193 if (relErr > relEPS)
194 {
195 bLoopVirgin = false;
196 }
197 }
198 }
199 NCV_SKIP_COND_END
200
201 if (bLoopVirgin)
202 {
203 rcode = true;
204 }
205
206 return rcode;
207}
