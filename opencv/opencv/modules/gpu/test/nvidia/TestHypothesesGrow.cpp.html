85bool TestHypothesesGrow::process()
86{
87 NCVStatus ncvStat;
88 bool rcode = false;
89
90 NCVVectorAlloc<Ncv32u> h_vecSrc(*this->allocatorCPU.get(), this->maxLenSrc);
91 ncvAssertReturn(h_vecSrc.isMemAllocated(), false);
92 NCVVectorAlloc<Ncv32u> d_vecSrc(*this->allocatorGPU.get(), this->maxLenSrc);
93 ncvAssertReturn(d_vecSrc.isMemAllocated(), false);
94
95 NCVVectorAlloc<NcvRect32u> h_vecDst(*this->allocatorCPU.get(), this->maxLenDst);
96 ncvAssertReturn(h_vecDst.isMemAllocated(), false);
97 NCVVectorAlloc<NcvRect32u> d_vecDst(*this->allocatorGPU.get(), this->maxLenDst);
98 ncvAssertReturn(d_vecDst.isMemAllocated(), false);
99 NCVVectorAlloc<NcvRect32u> h_vecDst_d(*this->allocatorCPU.get(), this->maxLenDst);
100 ncvAssertReturn(h_vecDst_d.isMemAllocated(), false);
101
102 NCV_SET_SKIP_COND(this->allocatorGPU.get()->isCounting());
103
104 NCV_SKIP_COND_BEGIN
105 ncvAssertReturn(this->src.fill(h_vecSrc), false);
106 memset(h_vecDst.ptr(), 0, h_vecDst.length() * sizeof(NcvRect32u));
107 NCVVectorReuse<Ncv32u> h_vecDst_as32u(h_vecDst.getSegment(), lenDst * sizeof(NcvRect32u) / sizeof(Ncv32u));
108 ncvAssertReturn(h_vecDst_as32u.isMemReused(), false);
109 ncvAssertReturn(this->src.fill(h_vecDst_as32u), false);
110 memcpy(h_vecDst_d.ptr(), h_vecDst.ptr(), h_vecDst.length() * sizeof(NcvRect32u));
111 NCV_SKIP_COND_END
112
113 ncvStat = h_vecSrc.copySolid(d_vecSrc, 0);
114 ncvAssertReturn(ncvStat == NCV_SUCCESS, false);
115 ncvStat = h_vecDst.copySolid(d_vecDst, 0);
116 ncvAssertReturn(ncvStat == NCV_SUCCESS, false);
117 ncvAssertCUDAReturn(cudaStreamSynchronize(0), false);
118
119 Ncv32u h_outElemNum_d = 0;
120 Ncv32u h_outElemNum_h = 0;
121 NCV_SKIP_COND_BEGIN
122 h_outElemNum_d = this->lenDst;
123 ncvStat = ncvGrowDetectionsVector_device(d_vecSrc, this->lenSrc,
124 d_vecDst, h_outElemNum_d, this->maxLenDst,
125 this->rectWidth, this->rectHeight, this->rectScale, 0);
126 ncvAssertReturn(ncvStat == NCV_SUCCESS, false);
127 ncvStat = d_vecDst.copySolid(h_vecDst_d, 0);
128 ncvAssertReturn(ncvStat == NCV_SUCCESS, false);
129 ncvAssertCUDAReturn(cudaStreamSynchronize(0), false);
130
131 h_outElemNum_h = this->lenDst;
132 ncvStat = ncvGrowDetectionsVector_host(h_vecSrc, this->lenSrc,
133 h_vecDst, h_outElemNum_h, this->maxLenDst,
134 this->rectWidth, this->rectHeight, this->rectScale);
135 ncvAssertReturn(ncvStat == NCV_SUCCESS, false);
136 NCV_SKIP_COND_END
137
138 //bit-to-bit check
139 bool bLoopVirgin = true;
140
141 NCV_SKIP_COND_BEGIN
142 if (h_outElemNum_d != h_outElemNum_h)
143 {
144 bLoopVirgin = false;
145 }
146 else
147 {
148 if (memcmp(h_vecDst.ptr(), h_vecDst_d.ptr(), this->maxLenDst * sizeof(NcvRect32u)))
149 {
150 bLoopVirgin = false;
151 }
152 }
153 NCV_SKIP_COND_END
154
155 if (bLoopVirgin)
156 {
157 rcode = true;
158 }
159
160 return rcode;
161}
