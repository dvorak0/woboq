70bool TestHaarCascadeLoader::process()
71{
72 NCVStatus ncvStat;
73 bool rcode = false;
74
75 Ncv32u numStages, numNodes, numFeatures;
76 Ncv32u numStages_2 = 0, numNodes_2 = 0, numFeatures_2 = 0;
77
78 ncvStat = ncvHaarGetClassifierSize(this->cascadeName, numStages, numNodes, numFeatures);
79 ncvAssertReturn(ncvStat == NCV_SUCCESS, false);
80
81 NCVVectorAlloc<HaarStage64> h_HaarStages(*this->allocatorCPU.get(), numStages);
82 ncvAssertReturn(h_HaarStages.isMemAllocated(), false);
83 NCVVectorAlloc<HaarClassifierNode128> h_HaarNodes(*this->allocatorCPU.get(), numNodes);
84 ncvAssertReturn(h_HaarNodes.isMemAllocated(), false);
85 NCVVectorAlloc<HaarFeature64> h_HaarFeatures(*this->allocatorCPU.get(), numFeatures);
86 ncvAssertReturn(h_HaarFeatures.isMemAllocated(), false);
87
88 NCVVectorAlloc<HaarStage64> h_HaarStages_2(*this->allocatorCPU.get(), numStages);
89 ncvAssertReturn(h_HaarStages_2.isMemAllocated(), false);
90 NCVVectorAlloc<HaarClassifierNode128> h_HaarNodes_2(*this->allocatorCPU.get(), numNodes);
91 ncvAssertReturn(h_HaarNodes_2.isMemAllocated(), false);
92 NCVVectorAlloc<HaarFeature64> h_HaarFeatures_2(*this->allocatorCPU.get(), numFeatures);
93 ncvAssertReturn(h_HaarFeatures_2.isMemAllocated(), false);
94
95 HaarClassifierCascadeDescriptor haar;
96 HaarClassifierCascadeDescriptor haar_2;
97
98 NCV_SET_SKIP_COND(this->allocatorGPU.get()->isCounting());
99 NCV_SKIP_COND_BEGIN
100
101 const std::string testNvbinName = cv::tempfile("test.nvbin");
102 ncvStat = ncvHaarLoadFromFile_host(this->cascadeName, haar, h_HaarStages, h_HaarNodes, h_HaarFeatures);
103 ncvAssertReturn(ncvStat == NCV_SUCCESS, false);
104
105 ncvStat = ncvHaarStoreNVBIN_host(testNvbinName, haar, h_HaarStages, h_HaarNodes, h_HaarFeatures);
106 ncvAssertReturn(ncvStat == NCV_SUCCESS, false);
107
108 ncvStat = ncvHaarGetClassifierSize(testNvbinName, numStages_2, numNodes_2, numFeatures_2);
109 ncvAssertReturn(ncvStat == NCV_SUCCESS, false);
110
111 ncvStat = ncvHaarLoadFromFile_host(testNvbinName, haar_2, h_HaarStages_2, h_HaarNodes_2, h_HaarFeatures_2);
112 ncvAssertReturn(ncvStat == NCV_SUCCESS, false);
113
114 NCV_SKIP_COND_END
115
116 //bit-to-bit check
117 bool bLoopVirgin = true;
118
119 NCV_SKIP_COND_BEGIN
120
121 if (
122 numStages_2 != numStages ||
123 numNodes_2 != numNodes ||
124 numFeatures_2 != numFeatures ||
125 haar.NumStages != haar_2.NumStages ||
126 haar.NumClassifierRootNodes != haar_2.NumClassifierRootNodes ||
127 haar.NumClassifierTotalNodes != haar_2.NumClassifierTotalNodes ||
128 haar.NumFeatures != haar_2.NumFeatures ||
129 haar.ClassifierSize.width != haar_2.ClassifierSize.width ||
130 haar.ClassifierSize.height != haar_2.ClassifierSize.height ||
131 haar.bNeedsTiltedII != haar_2.bNeedsTiltedII ||
132 haar.bHasStumpsOnly != haar_2.bHasStumpsOnly )
133 {
134 bLoopVirgin = false;
135 }
136 if (memcmp(h_HaarStages.ptr(), h_HaarStages_2.ptr(), haar.NumStages * sizeof(HaarStage64)) ||
137 memcmp(h_HaarNodes.ptr(), h_HaarNodes_2.ptr(), haar.NumClassifierTotalNodes * sizeof(HaarClassifierNode128)) ||
138 memcmp(h_HaarFeatures.ptr(), h_HaarFeatures_2.ptr(), haar.NumFeatures * sizeof(HaarFeature64)) )
139 {
140 bLoopVirgin = false;
141 }
142 NCV_SKIP_COND_END
143
144 if (bLoopVirgin)
145 {
146 rcode = true;
147 }
148
149 return rcode;
150}
