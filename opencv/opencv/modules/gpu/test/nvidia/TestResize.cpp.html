50template <class T>
51TestResize<T>::TestResize(std::string testName_, NCVTestSourceProvider<T> &src_,
52 Ncv32u width_, Ncv32u height_, Ncv32u scaleFactor_, NcvBool bTextureCache_)
53 :
54 NCVTestProvider(testName_),
55 src(src_),
56 width(width_),
57 height(height_),
58 scaleFactor(scaleFactor_),
59 bTextureCache(bTextureCache_)

61}

82template <class T>
83bool TestResize<T>::process()
84{
85 NCVStatus ncvStat;
86 bool rcode = false;
87
88 Ncv32s smallWidth = this->width / this->scaleFactor;
89 Ncv32s smallHeight = this->height / this->scaleFactor;
90 if (smallWidth == 0 || smallHeight == 0)
91 {
92 return true;
93 }
94
95 NcvSize32u srcSize(this->width, this->height);
96
97 NCVMatrixAlloc<T> d_img(*this->allocatorGPU.get(), this->width, this->height);
98 ncvAssertReturn(d_img.isMemAllocated(), false);
99 NCVMatrixAlloc<T> h_img(*this->allocatorCPU.get(), this->width, this->height);
100 ncvAssertReturn(h_img.isMemAllocated(), false);
101
102 NCVMatrixAlloc<T> d_small(*this->allocatorGPU.get(), smallWidth, smallHeight);
103 ncvAssertReturn(d_small.isMemAllocated(), false);
104 NCVMatrixAlloc<T> h_small(*this->allocatorCPU.get(), smallWidth, smallHeight);
105 ncvAssertReturn(h_small.isMemAllocated(), false);
106 NCVMatrixAlloc<T> h_small_d(*this->allocatorCPU.get(), smallWidth, smallHeight);
107 ncvAssertReturn(h_small_d.isMemAllocated(), false);
108
109 NCV_SET_SKIP_COND(this->allocatorGPU.get()->isCounting());
110 NCV_SKIP_COND_BEGIN
111 ncvAssertReturn(this->src.fill(h_img), false);
112 NCV_SKIP_COND_END
113
114 ncvStat = h_img.copySolid(d_img, 0);
115 ncvAssertReturn(ncvStat == NPPST_SUCCESS, false);
116 NCV_SKIP_COND_BEGIN
117 if (sizeof(T) == sizeof(Ncv32u))
118 {
119 ncvStat = nppiStDecimate_32u_C1R((Ncv32u *)d_img.ptr(), d_img.pitch(),
120 (Ncv32u *)d_small.ptr(), d_small.pitch(),
121 srcSize, this->scaleFactor,
122 this->bTextureCache);
123 }
124 else if (sizeof(T) == sizeof(Ncv64u))
125 {
126 ncvStat = nppiStDecimate_64u_C1R((Ncv64u *)d_img.ptr(), d_img.pitch(),
127 (Ncv64u *)d_small.ptr(), d_small.pitch(),
128 srcSize, this->scaleFactor,
129 this->bTextureCache);
130 }
131 else
132 {
133 ncvAssertPrintReturn(false, "Incorrect downsample test instance", false);
134 }
135 ncvAssertReturn(ncvStat == NPPST_SUCCESS, false);
136 NCV_SKIP_COND_END
137 ncvStat = d_small.copySolid(h_small_d, 0);
138 ncvAssertReturn(ncvStat == NPPST_SUCCESS, false);
139
140 NCV_SKIP_COND_BEGIN
141 if (sizeof(T) == sizeof(Ncv32u))
142 {
143 ncvStat = nppiStDecimate_32u_C1R_host((Ncv32u *)h_img.ptr(), h_img.pitch(),
144 (Ncv32u *)h_small.ptr(), h_small.pitch(),
145 srcSize, this->scaleFactor);
146 }
147 else if (sizeof(T) == sizeof(Ncv64u))
148 {
149 ncvStat = nppiStDecimate_64u_C1R_host((Ncv64u *)h_img.ptr(), h_img.pitch(),
150 (Ncv64u *)h_small.ptr(), h_small.pitch(),
151 srcSize, this->scaleFactor);
152 }
153 else
154 {
155 ncvAssertPrintReturn(false, "Incorrect downsample test instance", false);
156 }
157 ncvAssertReturn(ncvStat == NPPST_SUCCESS, false);
158 NCV_SKIP_COND_END
159
160 //bit-to-bit check
161 bool bLoopVirgin = true;
162
163 NCV_SKIP_COND_BEGIN
164 //const Ncv64f relEPS = 0.005;
165 for (Ncv32u i=0; bLoopVirgin && i < h_small.height(); i++)
166 {
167 for (Ncv32u j=0; bLoopVirgin && j < h_small.width(); j++)
168 {
169 if (h_small.ptr()[h_small.stride()*i+j] != h_small_d.ptr()[h_small_d.stride()*i+j])
170 {
171 bLoopVirgin = false;
172 }
173 }
174 }
175 NCV_SKIP_COND_END
176
177 if (bLoopVirgin)
178 {
179 rcode = true;
180 }
181
182 return rcode;
183}
