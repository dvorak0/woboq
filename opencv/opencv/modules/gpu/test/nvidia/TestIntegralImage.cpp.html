79template <class T_in, class T_out>
80bool TestIntegralImage<T_in, T_out>::process()
81{
82 NCVStatus ncvStat;
83 bool rcode = false;
84
85 Ncv32u widthII = this->width + 1;
86 Ncv32u heightII = this->height + 1;
87
88 NCVMatrixAlloc<T_in> d_img(*this->allocatorGPU.get(), this->width, this->height);
89 ncvAssertReturn(d_img.isMemAllocated(), false);
90 NCVMatrixAlloc<T_in> h_img(*this->allocatorCPU.get(), this->width, this->height);
91 ncvAssertReturn(h_img.isMemAllocated(), false);
92 NCVMatrixAlloc<T_out> d_imgII(*this->allocatorGPU.get(), widthII, heightII);
93 ncvAssertReturn(d_imgII.isMemAllocated(), false);
94 NCVMatrixAlloc<T_out> h_imgII(*this->allocatorCPU.get(), widthII, heightII);
95 ncvAssertReturn(h_imgII.isMemAllocated(), false);
96 NCVMatrixAlloc<T_out> h_imgII_d(*this->allocatorCPU.get(), widthII, heightII);
97 ncvAssertReturn(h_imgII_d.isMemAllocated(), false);
98
99 Ncv32u bufSize;
100 if (sizeof(T_in) == sizeof(Ncv8u))
101 {
102 ncvStat = nppiStIntegralGetSize_8u32u(NcvSize32u(this->width, this->height), &bufSize, this->devProp);
103 ncvAssertReturn(NPPST_SUCCESS == ncvStat, false);
104 }
105 else if (sizeof(T_in) == sizeof(Ncv32f))
106 {
107 ncvStat = nppiStIntegralGetSize_32f32f(NcvSize32u(this->width, this->height), &bufSize, this->devProp);
108 ncvAssertReturn(NPPST_SUCCESS == ncvStat, false);
109 }
110 else
111 {
112 ncvAssertPrintReturn(false, "Incorrect integral image test instance", false);
113 }
114
115 NCVVectorAlloc<Ncv8u> d_tmpBuf(*this->allocatorGPU.get(), bufSize);
116 ncvAssertReturn(d_tmpBuf.isMemAllocated(), false);
117
118 NCV_SET_SKIP_COND(this->allocatorGPU.get()->isCounting());
119 NCV_SKIP_COND_BEGIN
120
121 ncvAssertReturn(this->src.fill(h_img), false);
122
123 ncvStat = h_img.copySolid(d_img, 0);
124 ncvAssertReturn(ncvStat == NPPST_SUCCESS, false);
125
126 if (sizeof(T_in) == sizeof(Ncv8u))
127 {
128 ncvStat = nppiStIntegral_8u32u_C1R((Ncv8u *)d_img.ptr(), d_img.pitch(),
129 (Ncv32u *)d_imgII.ptr(), d_imgII.pitch(),
130 NcvSize32u(this->width, this->height),
131 d_tmpBuf.ptr(), bufSize, this->devProp);
132 ncvAssertReturn(ncvStat == NPPST_SUCCESS, false);
133 }
134 else if (sizeof(T_in) == sizeof(Ncv32f))
135 {
136 ncvStat = nppiStIntegral_32f32f_C1R((Ncv32f *)d_img.ptr(), d_img.pitch(),
137 (Ncv32f *)d_imgII.ptr(), d_imgII.pitch(),
138 NcvSize32u(this->width, this->height),
139 d_tmpBuf.ptr(), bufSize, this->devProp);
140 ncvAssertReturn(ncvStat == NPPST_SUCCESS, false);
141 }
142 else
143 {
144 ncvAssertPrintReturn(false, "Incorrect integral image test instance", false);
145 }
146
147 ncvStat = d_imgII.copySolid(h_imgII_d, 0);
148 ncvAssertReturn(ncvStat == NPPST_SUCCESS, false);
149
150 if (sizeof(T_in) == sizeof(Ncv8u))
151 {
152 ncvStat = nppiStIntegral_8u32u_C1R_host((Ncv8u *)h_img.ptr(), h_img.pitch(),
153 (Ncv32u *)h_imgII.ptr(), h_imgII.pitch(),
154 NcvSize32u(this->width, this->height));
155 ncvAssertReturn(ncvStat == NPPST_SUCCESS, false);
156 }
157 else if (sizeof(T_in) == sizeof(Ncv32f))
158 {
159 ncvStat = nppiStIntegral_32f32f_C1R_host((Ncv32f *)h_img.ptr(), h_img.pitch(),
160 (Ncv32f *)h_imgII.ptr(), h_imgII.pitch(),
161 NcvSize32u(this->width, this->height));
162 ncvAssertReturn(ncvStat == NPPST_SUCCESS, false);
163 }
164 else
165 {
166 ncvAssertPrintReturn(false, "Incorrect integral image test instance", false);
167 }
168
169 NCV_SKIP_COND_END
170
171 //bit-to-bit check
172 bool bLoopVirgin = true;
173
174 NCV_SKIP_COND_BEGIN
175 for (Ncv32u i=0; bLoopVirgin && i < h_img.height() + 1; i++)
176 {
177 for (Ncv32u j=0; bLoopVirgin && j < h_img.width() + 1; j++)
178 {
179 if (sizeof(T_in) == sizeof(Ncv8u))
180 {
181 if (h_imgII.ptr()[h_imgII.stride()*i+j] != h_imgII_d.ptr()[h_imgII_d.stride()*i+j])
182 {
183 bLoopVirgin = false;
184 }
185 }
186 else if (sizeof(T_in) == sizeof(Ncv32f))
187 {
188 if (fabsf((float)h_imgII.ptr()[h_imgII.stride()*i+j] - (float)h_imgII_d.ptr()[h_imgII_d.stride()*i+j]) > 0.01f)
189 {
190 bLoopVirgin = false;
191 }
192 }
193 else
194 {
195 ncvAssertPrintReturn(false, "Incorrect integral image test instance", false);
196 }
197 }
198 }
199 NCV_SKIP_COND_END
200
201 if (bLoopVirgin)
202 {
203 rcode = true;
204 }
205
206 return rcode;
207}
