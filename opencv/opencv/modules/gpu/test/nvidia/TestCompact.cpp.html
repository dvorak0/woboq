75bool TestCompact::process()
76{
77 NCVStatus ncvStat;
78 bool rcode = false;
79
80 NCVVectorAlloc<Ncv32u> h_vecSrc(*this->allocatorCPU.get(), this->length);
81 ncvAssertReturn(h_vecSrc.isMemAllocated(), false);
82 NCVVectorAlloc<Ncv32u> d_vecSrc(*this->allocatorGPU.get(), this->length);
83 ncvAssertReturn(d_vecSrc.isMemAllocated(), false);
84
85 NCVVectorAlloc<Ncv32u> h_vecDst(*this->allocatorCPU.get(), this->length);
86 ncvAssertReturn(h_vecDst.isMemAllocated(), false);
87 NCVVectorAlloc<Ncv32u> d_vecDst(*this->allocatorGPU.get(), this->length);
88 ncvAssertReturn(d_vecDst.isMemAllocated(), false);
89 NCVVectorAlloc<Ncv32u> h_vecDst_d(*this->allocatorCPU.get(), this->length);
90 ncvAssertReturn(h_vecDst_d.isMemAllocated(), false);
91
92 NCV_SET_SKIP_COND(this->allocatorGPU.get()->isCounting());
93 NCV_SKIP_COND_BEGIN
94 ncvAssertReturn(this->src.fill(h_vecSrc), false);
95 for (Ncv32u i=0; i<this->length; i++)
96 {
97 Ncv32u tmp = (h_vecSrc.ptr()[i]) & 0xFF;
98 tmp = tmp * 99 / 255;
99 if (tmp < this->badElemPercentage)
100 {
101 h_vecSrc.ptr()[i] = this->badElem;
102 }
103 }
104 NCV_SKIP_COND_END
105
106 NCVVectorAlloc<Ncv32u> h_dstLen(*this->allocatorCPU.get(), 1);
107 ncvAssertReturn(h_dstLen.isMemAllocated(), false);
108 Ncv32u bufSize;
109 ncvStat = nppsStCompactGetSize_32u(this->length, &bufSize, this->devProp);
110 ncvAssertReturn(NPPST_SUCCESS == ncvStat, false);
111 NCVVectorAlloc<Ncv8u> d_tmpBuf(*this->allocatorGPU.get(), bufSize);
112 ncvAssertReturn(d_tmpBuf.isMemAllocated(), false);
113
114 Ncv32u h_outElemNum_h = 0;
115
116 NCV_SKIP_COND_BEGIN
117 ncvStat = h_vecSrc.copySolid(d_vecSrc, 0);
118 ncvAssertReturn(ncvStat == NPPST_SUCCESS, false);
119 ncvStat = nppsStCompact_32u(d_vecSrc.ptr(), this->length,
120 d_vecDst.ptr(), h_dstLen.ptr(), this->badElem,
121 d_tmpBuf.ptr(), bufSize, this->devProp);
122 ncvAssertReturn(ncvStat == NPPST_SUCCESS, false);
123 ncvStat = d_vecDst.copySolid(h_vecDst_d, 0);
124 ncvAssertReturn(ncvStat == NPPST_SUCCESS, false);
125
126 ncvStat = nppsStCompact_32u_host(h_vecSrc.ptr(), this->length, h_vecDst.ptr(), &h_outElemNum_h, this->badElem);
127 ncvAssertReturn(ncvStat == NPPST_SUCCESS, false);
128 NCV_SKIP_COND_END
129
130 //bit-to-bit check
131 bool bLoopVirgin = true;
132
133 NCV_SKIP_COND_BEGIN
134 if (h_dstLen.ptr()[0] != h_outElemNum_h)
135 {
136 bLoopVirgin = false;
137 }
138 else
139 {
140 for (Ncv32u i=0; bLoopVirgin && i < h_outElemNum_h; i++)
141 {
142 if (h_vecDst.ptr()[i] != h_vecDst_d.ptr()[i])
143 {
144 bLoopVirgin = false;
145 }
146 }
147 }
148 NCV_SKIP_COND_END
149
150 if (bLoopVirgin)
151 {
152 rcode = true;
153 }
154
155 return rcode;
156}
