50TestDrawRects<T>::TestDrawRects(std::string testName_, NCVTestSourceProvider<T> &src_,
51 NCVTestSourceProvider<Ncv32u> &src32u_,
52 Ncv32u width_, Ncv32u height_, Ncv32u numRects_, T color_)

54 NCVTestProvider(testName_),

84template <class T>
85bool TestDrawRects<T>::process()
86{
87 NCVStatus ncvStat;
88 bool rcode = false;
89
90 NCVMatrixAlloc<T> d_img(*this->allocatorGPU.get(), this->width, this->height);
91 ncvAssertReturn(d_img.isMemAllocated(), false);
92 NCVMatrixAlloc<T> h_img(*this->allocatorCPU.get(), this->width, this->height);
93 ncvAssertReturn(h_img.isMemAllocated(), false);
94 NCVMatrixAlloc<T> h_img_d(*this->allocatorCPU.get(), this->width, this->height);
95 ncvAssertReturn(h_img_d.isMemAllocated(), false);
96
97 NCVVectorAlloc<NcvRect32u> d_rects(*this->allocatorGPU.get(), this->numRects);
98 ncvAssertReturn(d_rects.isMemAllocated(), false);
99 NCVVectorAlloc<NcvRect32u> h_rects(*this->allocatorCPU.get(), this->numRects);
100 ncvAssertReturn(h_rects.isMemAllocated(), false);
101
102 NCV_SET_SKIP_COND(this->allocatorGPU.get()->isCounting());
103 NCV_SKIP_COND_BEGIN
104 ncvAssertReturn(this->src.fill(h_img), false);
105 ncvStat = h_img.copySolid(d_img, 0);
106 ncvAssertReturn(ncvStat == NCV_SUCCESS, false);
107 ncvAssertCUDAReturn(cudaStreamSynchronize(0), false);
108
109 //fill vector of rectangles with random rects covering the input
110 NCVVectorReuse<Ncv32u> h_rects_as32u(h_rects.getSegment());
111 ncvAssertReturn(h_rects_as32u.isMemReused(), false);
112 ncvAssertReturn(this->src32u.fill(h_rects_as32u), false);
113 for (Ncv32u i=0; i<this->numRects; i++)
114 {
115 h_rects.ptr()[i].x = (Ncv32u)(((1.0 * h_rects.ptr()[i].x) / RAND_MAX) * (this->width-2));
116 h_rects.ptr()[i].y = (Ncv32u)(((1.0 * h_rects.ptr()[i].y) / RAND_MAX) * (this->height-2));
117 h_rects.ptr()[i].width = (Ncv32u)(((1.0 * h_rects.ptr()[i].width) / RAND_MAX) * (this->width+10 - h_rects.ptr()[i].x));
118 h_rects.ptr()[i].height = (Ncv32u)(((1.0 * h_rects.ptr()[i].height) / RAND_MAX) * (this->height+10 - h_rects.ptr()[i].y));
119 }
120 ncvStat = h_rects.copySolid(d_rects, 0);
121 ncvAssertReturn(ncvStat == NCV_SUCCESS, false);
122 ncvAssertCUDAReturn(cudaStreamSynchronize(0), false);
123
124 if (sizeof(T) == sizeof(Ncv32u))
125 {
126 ncvStat = ncvDrawRects_32u_device((Ncv32u *)d_img.ptr(), d_img.stride(), this->width, this->height,
127 (NcvRect32u *)d_rects.ptr(), this->numRects, this->color, 0);
128 }
129 else if (sizeof(T) == sizeof(Ncv8u))
130 {
131 ncvStat = ncvDrawRects_8u_device((Ncv8u *)d_img.ptr(), d_img.stride(), this->width, this->height,
132 (NcvRect32u *)d_rects.ptr(), this->numRects, (Ncv8u)this->color, 0);
133 }
134 else
135 {
136 ncvAssertPrintReturn(false, "Incorrect drawrects test instance", false);
137 }
138 ncvAssertReturn(ncvStat == NCV_SUCCESS, false);
139 NCV_SKIP_COND_END
140
141 ncvStat = d_img.copySolid(h_img_d, 0);
142 ncvAssertReturn(ncvStat == NCV_SUCCESS, false);
143 ncvAssertCUDAReturn(cudaStreamSynchronize(0), false);
144
145 NCV_SKIP_COND_BEGIN
146 if (sizeof(T) == sizeof(Ncv32u))
147 {
148 ncvStat = ncvDrawRects_32u_host((Ncv32u *)h_img.ptr(), h_img.stride(), this->width, this->height,
149 (NcvRect32u *)h_rects.ptr(), this->numRects, this->color);
150 }
151 else if (sizeof(T) == sizeof(Ncv8u))
152 {
153 ncvStat = ncvDrawRects_8u_host((Ncv8u *)h_img.ptr(), h_img.stride(), this->width, this->height,
154 (NcvRect32u *)h_rects.ptr(), this->numRects, (Ncv8u)this->color);
155 }
156 else
157 {
158 ncvAssertPrintReturn(false, "Incorrect drawrects test instance", false);
159 }
160 ncvAssertReturn(ncvStat == NCV_SUCCESS, false);
161 NCV_SKIP_COND_END
162
163 //bit-to-bit check
164 bool bLoopVirgin = true;
165
166 NCV_SKIP_COND_BEGIN
167 //const Ncv64f relEPS = 0.005;
168 for (Ncv32u i=0; bLoopVirgin && i < h_img.height(); i++)
169 {
170 for (Ncv32u j=0; bLoopVirgin && j < h_img.width(); j++)
171 {
172 if (h_img.ptr()[h_img.stride()*i+j] != h_img_d.ptr()[h_img_d.stride()*i+j])
173 {
174 bLoopVirgin = false;
175 }
176 }
177 }
178 NCV_SKIP_COND_END
179
180 if (bLoopVirgin)
181 {
182 rcode = true;
183 }
184
185 return rcode;
186}
