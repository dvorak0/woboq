44#define _ncvautotestlister_hpp_
45
46#include <vector>
47
48#include "NCVTest.hpp"
49#include <main_test_nvidia.h>
50//enum OutputLevel

52// OutputLevelNone,
53// OutputLevelCompact,

56
57class NCVAutoTestLister
58{
59public:
60
61 NCVAutoTestLister(std::string testSuiteName_, OutputLevel outputLevel_ = OutputLevelCompact, NcvBool bStopOnFirstFail_=false)
62 :
63 testSuiteName(testSuiteName_),
64 outputLevel(outputLevel_),
65 bStopOnFirstFail(bStopOnFirstFail_)
66 {
67 }
68
69 void add(INCVTest *test)
70 {
71 this->tests.push_back(test);
72 }
73
74 bool invoke()
75 {
76 Ncv32u nPassed = 0;
77 Ncv32u nFailed = 0;
78 Ncv32u nFailedMem = 0;
79
80 if (outputLevel == OutputLevelCompact)
81 {
82 printf("Test suite '%s' with %d tests\n",
83 testSuiteName.c_str(),
84 (int)(this->tests.size()));
85 }
86
87 for (Ncv32u i=0; i<this->tests.size(); i++)
88 {
89 INCVTest &curTest = *tests[i];
90
91 NCVTestReport curReport;
92 bool res = curTest.executeTest(curReport);
93
94 if (outputLevel == OutputLevelFull)
95 {
96 printf("Test %3i %16s; Consumed mem GPU = %8d, CPU = %8d; %s\n",
97 i,
98 curTest.getName().c_str(),
99 curReport.statsNums["MemGPU"],
100 curReport.statsNums["MemCPU"],
101 curReport.statsText["rcode"].c_str());
102 }
103
104 if (res)
105 {
106 nPassed++;
107 if (outputLevel == OutputLevelCompact)
108 {
109 printf(".");
110 }
111 }
112 else
113 {
114 if (!curReport.statsText["rcode"].compare("FAILED"))
115 {
116 nFailed++;
117 if (outputLevel == OutputLevelCompact)
118 {
119 printf("x");
120 }
121 if (bStopOnFirstFail)
122 {
123 break;
124 }
125 }
126 else
127 {
128 nFailedMem++;
129 if (outputLevel == OutputLevelCompact)
130 {
131 printf("m");
132 }
133 }
134 }
135 fflush(stdout);
136 }
137 if (outputLevel == OutputLevelCompact)
138 {
139 printf("\n");
140 }
141
142 if (outputLevel != OutputLevelNone)
143 {
144 printf("Test suite '%s' complete: %d total, %d passed, %d memory errors, %d failed\n\n",
145 testSuiteName.c_str(),
146 (int)(this->tests.size()),
147 nPassed,
148 nFailedMem,
149 nFailed);
150 }
151
152 bool passed = nFailed == 0 && nFailedMem == 0;
153 return passed;
154 }
155
156 ~NCVAutoTestLister()
157 {
158 for (Ncv32u i=0; i<this->tests.size(); i++)
159 {
160 delete tests[i];
161 }
162 }
163
164private:
165
166 std::string testSuiteName;
167 OutputLevel outputLevel;
168 NcvBool bStopOnFirstFail;
169 std::vector<INCVTest *> tests;
170};
