61struct NCVTestReport
62{
63 std::map<std::string, Ncv32u> statsNums;
64 std::map<std::string, std::string> statsText;
65};
66
67
68class INCVTest
69{
70public:
71 virtual bool executeTest(NCVTestReport &report) = 0;
72 virtual std::string getName() const = 0;
73 virtual ~INCVTest(){}
74};
75
76
77class NCVTestProvider : public INCVTest
78{
79public:
80
81 NCVTestProvider(std::string testName_)
82 :
83 testName(testName_)
84 {
85 int devId;
86 ncvAssertPrintReturn(cudaSuccess == cudaGetDevice(&devId), "Error returned from cudaGetDevice", );
87 ncvAssertPrintReturn(cudaSuccess == cudaGetDeviceProperties(&this->devProp, devId), "Error returned from cudaGetDeviceProperties", );
88 }
89
90 virtual bool init() = 0;
91 virtual bool process() = 0;
92 virtual bool deinit() = 0;
93 virtual bool toString(std::ofstream &strOut) = 0;
94
95 virtual std::string getName() const
96 {
97 return this->testName;
98 }
99
100 virtual ~NCVTestProvider()
101 {
102 deinitMemory();
103 }
104
105 virtual bool executeTest(NCVTestReport &report)
106 {
107 bool res;
108 report.statsText["rcode"] = "FAILED";
109
110 res = initMemory(report);
111 if (!res)
112 {
113 dumpToFile(report);
114 deinitMemory();
115 return false;
116 }
117
118 res = init();
119 if (!res)
120 {
121 dumpToFile(report);
122 deinit();
123 deinitMemory();
124 return false;
125 }
126
127 res = process();
128 if (!res)
129 {
130 dumpToFile(report);
131 deinit();
132 deinitMemory();
133 return false;
134 }
135
136 res = deinit();
137 if (!res)
138 {
139 dumpToFile(report);
140 deinitMemory();
141 return false;
142 }
143
144 deinitMemory();
145
146 report.statsText["rcode"] = "Passed";
147 return true;
148 }
149
150protected:
151
152 cudaDeviceProp devProp;
153 std::auto_ptr<INCVMemAllocator> allocatorGPU;
154 std::auto_ptr<INCVMemAllocator> allocatorCPU;
155
156private:
157
158 std::string testName;
159
160 bool initMemory(NCVTestReport &report)
161 {
162 this->allocatorGPU.reset(new NCVMemStackAllocator(static_cast<Ncv32u>(devProp.textureAlignment)));
163 this->allocatorCPU.reset(new NCVMemStackAllocator(static_cast<Ncv32u>(devProp.textureAlignment)));
164
165 if (!this->allocatorGPU.get()->isInitialized() ||
166 !this->allocatorCPU.get()->isInitialized())
167 {
168 report.statsText["rcode"] = "Memory FAILED";
169 return false;
170 }
171
172 if (!this->process())
173 {
174 report.statsText["rcode"] = "Memory FAILED";
175 return false;
176 }
177
178 Ncv32u maxGPUsize = (Ncv32u)this->allocatorGPU.get()->maxSize();
179 Ncv32u maxCPUsize = (Ncv32u)this->allocatorCPU.get()->maxSize();
180
181 report.statsNums["MemGPU"] = maxGPUsize;
182 report.statsNums["MemCPU"] = maxCPUsize;
183
184 this->allocatorGPU.reset(new NCVMemStackAllocator(NCVMemoryTypeDevice, maxGPUsize, static_cast<Ncv32u>(devProp.textureAlignment)));
185
186 this->allocatorCPU.reset(new NCVMemStackAllocator(NCVMemoryTypeHostPinned, maxCPUsize, static_cast<Ncv32u>(devProp.textureAlignment)));
187
188 if (!this->allocatorGPU.get()->isInitialized() ||
189 !this->allocatorCPU.get()->isInitialized())
190 {
191 report.statsText["rcode"] = "Memory FAILED";
192 return false;
193 }
194
195 return true;
196 }
197
198 void deinitMemory()
199 {
200 this->allocatorGPU.reset();
201 this->allocatorCPU.reset();
202 }
203
204 void dumpToFile(NCVTestReport &report)
205 {
206 bool bReasonMem = (0 == report.statsText["rcode"].compare("Memory FAILED"));
207 std::string fname = "TestDump_";
208 fname += (bReasonMem ? "m_" : "") + this->testName + ".log";
209 std::ofstream stream(fname.c_str(), std::ios::trunc | std::ios::out);
210 if (!stream.is_open()) return;
211
212 stream << "NCV Test Failure Log: " << this->testName << std::endl;
213 stream << "====================================================" << std::endl << std::endl;
214 stream << "Test initialization report: " << std::endl;
215 for (std::map<std::string,std::string>::iterator it=report.statsText.begin();
216 it != report.statsText.end(); it++)
217 {
218 stream << it->first << "=" << it->second << std::endl;
219 }
220 for (std::map<std::string,Ncv32u>::iterator it=report.statsNums.begin();
221 it != report.statsNums.end(); it++)
222 {
223 stream << it->first << "=" << it->second << std::endl;
224 }
225 stream << std::endl;
226
227 stream << "Test initialization parameters: " << std::endl;
228 bool bSerializeRes = false;
229 try
230 {
231 bSerializeRes = this->toString(stream);
232 }
233 catch (...)
234 {
235 }
236
237 if (!bSerializeRes)
238 {
239 stream << "Couldn't retrieve object dump" << std::endl;
240 }
241
242 stream.flush();
243 }
244};
