100void cv::gpu::FarnebackOpticalFlow::prepareGaussian(
101 int n, double sigma, float *g, float *xg, float *xxg,
102 double &ig11, double &ig03, double &ig33, double &ig55)
103{
104 double s = 0.;
105 for (int x = -n; x <= n; x++)
106 {
107 g[x] = (float)std::exp(-x*x/(2*sigma*sigma));
108 s += g[x];
109 }
110
111 s = 1./s;
112 for (int x = -n; x <= n; x++)
113 {
114 g[x] = (float)(g[x]*s);
115 xg[x] = (float)(x*g[x]);
116 xxg[x] = (float)(x*x*g[x]);
117 }
118
119 Mat_<double> G(6, 6);
120 G.setTo(0);
121
122 for (int y = -n; y <= n; y++)
123 {
124 for (int x = -n; x <= n; x++)
125 {
126 G(0,0) += g[y]*g[x];
127 G(1,1) += g[y]*g[x]*x*x;
128 G(3,3) += g[y]*g[x]*x*x*x*x;
129 G(5,5) += g[y]*g[x]*x*x*y*y;
130 }
131 }
132
133 //G[0][0] = 1.;
134 G(2,2) = G(0,3) = G(0,4) = G(3,0) = G(4,0) = G(1,1);
135 G(4,4) = G(3,3);
136 G(3,4) = G(4,3) = G(5,5);
137
138 // invG:
139 // [ x e e ]
140 // [ y ]
141 // [ y ]
142 // [ e z ]
143 // [ e z ]
144 // [ u ]
145 Mat_<double> invG = G.inv(DECOMP_CHOLESKY);
146
147 ig11 = invG(1,1);
148 ig03 = invG(0,3);
149 ig33 = invG(3,3);
150 ig55 = invG(5,5);
151}

154void cv::gpu::FarnebackOpticalFlow::setPolynomialExpansionConsts(int n, double sigma)
155{
156 vector<float> buf(n*6 + 3);
157 float* g = &buf[0] + n;
158 float* xg = g + n*2 + 1;
159 float* xxg = xg + n*2 + 1;
160
161 if (sigma < FLT_EPSILON)
162 sigma = n*0.3;
163
164 double ig11, ig03, ig33, ig55;
165 prepareGaussian(n, sigma, g, xg, xxg, ig11, ig03, ig33, ig55);
166
167 device::optflow_farneback::setPolynomialExpansionConsts(n, g, xg, xxg, static_cast<float>(ig11), static_cast<float>(ig03), static_cast<float>(ig33), static_cast<float>(ig55));
168}

171void cv::gpu::FarnebackOpticalFlow::updateFlow_boxFilter(
172 const GpuMat& R0, const GpuMat& R1, GpuMat& flowx, GpuMat &flowy,
173 GpuMat& M, GpuMat &bufM, int blockSize, bool updateMatrices, Stream streams[])
174{
175 if (deviceSupports(FEATURE_SET_COMPUTE_12))
176 device::optflow_farneback::boxFilter5Gpu(M, blockSize/2, bufM, S(streams[0]));
177 else
178 device::optflow_farneback::boxFilter5Gpu_CC11(M, blockSize/2, bufM, S(streams[0]));
179 swap(M, bufM);
180
181 for (int i = 1; i < 5; ++i)
182 streams[i].waitForCompletion();
183 device::optflow_farneback::updateFlowGpu(M, flowx, flowy, S(streams[0]));
184
185 if (updateMatrices)
186 device::optflow_farneback::updateMatricesGpu(flowx, flowy, R0, R1, M, S(streams[0]));
187}

209void cv::gpu::FarnebackOpticalFlow::operator ()(
210 const GpuMat &frame0, const GpuMat &frame1, GpuMat &flowx, GpuMat &flowy, Stream &s)
211{
212 CV_Assert(frame0.channels() == 1 && frame1.channels() == 1);
213 CV_Assert(frame0.size() == frame1.size());
214 CV_Assert(polyN == 5 || polyN == 7);
215 CV_Assert(!fastPyramids || std::abs(pyrScale - 0.5) < 1e-6);
216
217 Stream streams[5];
218 if (S(s))
219 streams[0] = s;
220
221 Size size = frame0.size();
222 GpuMat prevFlowX, prevFlowY, curFlowX, curFlowY;
223
224 flowx.create(size, CV_32F);
225 flowy.create(size, CV_32F);
226 GpuMat flowx0 = flowx;
227 GpuMat flowy0 = flowy;
228
229 // Crop unnecessary levels
230 double scale = 1;
231 int numLevelsCropped = 0;
232 for (; numLevelsCropped < numLevels; numLevelsCropped++)
233 {
234 scale *= pyrScale;
235 if (size.width*scale < MIN_SIZE || size.height*scale < MIN_SIZE)
236 break;
237 }
238
239 streams[0].enqueueConvert(frame0, frames_[0], CV_32F);
240 streams[1].enqueueConvert(frame1, frames_[1], CV_32F);
241
242 if (fastPyramids)
243 {
244 // Build Gaussian pyramids using pyrDown()
245 pyramid0_.resize(numLevelsCropped + 1);
246 pyramid1_.resize(numLevelsCropped + 1);
247 pyramid0_[0] = frames_[0];
248 pyramid1_[0] = frames_[1];
249 for (int i = 1; i <= numLevelsCropped; ++i)
250 {
251 pyrDown(pyramid0_[i - 1], pyramid0_[i], streams[0]);
252 pyrDown(pyramid1_[i - 1], pyramid1_[i], streams[1]);
253 }
254 }
255
256 setPolynomialExpansionConsts(polyN, polySigma);
257 device::optflow_farneback::setUpdateMatricesConsts();
258
259 for (int k = numLevelsCropped; k >= 0; k--)
260 {
261 streams[0].waitForCompletion();
262
263 scale = 1;
264 for (int i = 0; i < k; i++)
265 scale *= pyrScale;
266
267 double sigma = (1./scale - 1) * 0.5;
268 int smoothSize = cvRound(sigma*5) | 1;
269 smoothSize = std::max(smoothSize, 3);
270
271 int width = cvRound(size.width*scale);
272 int height = cvRound(size.height*scale);
273
274 if (fastPyramids)
275 {
276 width = pyramid0_[k].cols;
277 height = pyramid0_[k].rows;
278 }
279
280 if (k > 0)
281 {
282 curFlowX.create(height, width, CV_32F);
283 curFlowY.create(height, width, CV_32F);
284 }
285 else
286 {
287 curFlowX = flowx0;
288 curFlowY = flowy0;
289 }
290
291 if (!prevFlowX.data)
292 {
293 if (flags & OPTFLOW_USE_INITIAL_FLOW)
294 {
295#if ENABLE_GPU_RESIZE
296 resize(flowx0, curFlowX, Size(width, height), 0, 0, INTER_LINEAR, streams[0]);
297 resize(flowy0, curFlowY, Size(width, height), 0, 0, INTER_LINEAR, streams[1]);
298 streams[0].enqueueConvert(curFlowX, curFlowX, curFlowX.depth(), scale);
299 streams[1].enqueueConvert(curFlowY, curFlowY, curFlowY.depth(), scale);
300#else
301 Mat tmp1, tmp2;
302 flowx0.download(tmp1);
303 resize(tmp1, tmp2, Size(width, height), 0, 0, INTER_AREA);
304 tmp2 *= scale;
305 curFlowX.upload(tmp2);
306 flowy0.download(tmp1);
307 resize(tmp1, tmp2, Size(width, height), 0, 0, INTER_AREA);
308 tmp2 *= scale;
309 curFlowY.upload(tmp2);
310#endif
311 }
312 else
313 {
314 streams[0].enqueueMemSet(curFlowX, 0);
315 streams[1].enqueueMemSet(curFlowY, 0);
316 }
317 }
318 else
319 {
320#if ENABLE_GPU_RESIZE
321 resize(prevFlowX, curFlowX, Size(width, height), 0, 0, INTER_LINEAR, streams[0]);
322 resize(prevFlowY, curFlowY, Size(width, height), 0, 0, INTER_LINEAR, streams[1]);
323 streams[0].enqueueConvert(curFlowX, curFlowX, curFlowX.depth(), 1./pyrScale);
324 streams[1].enqueueConvert(curFlowY, curFlowY, curFlowY.depth(), 1./pyrScale);
325#else
326 Mat tmp1, tmp2;
327 prevFlowX.download(tmp1);
328 resize(tmp1, tmp2, Size(width, height), 0, 0, INTER_LINEAR);
329 tmp2 *= 1./pyrScale;
330 curFlowX.upload(tmp2);
331 prevFlowY.download(tmp1);
332 resize(tmp1, tmp2, Size(width, height), 0, 0, INTER_LINEAR);
333 tmp2 *= 1./pyrScale;
334 curFlowY.upload(tmp2);
335#endif
336 }
337
338 GpuMat M = allocMatFromBuf(5*height, width, CV_32F, M_);
339 GpuMat bufM = allocMatFromBuf(5*height, width, CV_32F, bufM_);
340 GpuMat R[2] =
341 {
342 allocMatFromBuf(5*height, width, CV_32F, R_[0]),
343 allocMatFromBuf(5*height, width, CV_32F, R_[1])
344 };
345
346 if (fastPyramids)
347 {
348 device::optflow_farneback::polynomialExpansionGpu(pyramid0_[k], polyN, R[0], S(streams[0]));
349 device::optflow_farneback::polynomialExpansionGpu(pyramid1_[k], polyN, R[1], S(streams[1]));
350 }
351 else
352 {
353 GpuMat blurredFrame[2] =
354 {
355 allocMatFromBuf(size.height, size.width, CV_32F, blurredFrame_[0]),
356 allocMatFromBuf(size.height, size.width, CV_32F, blurredFrame_[1])
357 };
358 GpuMat pyrLevel[2] =
359 {
360 allocMatFromBuf(height, width, CV_32F, pyrLevel_[0]),
361 allocMatFromBuf(height, width, CV_32F, pyrLevel_[1])
362 };
363
364 Mat g = getGaussianKernel(smoothSize, sigma, CV_32F);
365 device::optflow_farneback::setGaussianBlurKernel(g.ptr<float>(smoothSize/2), smoothSize/2);
366
367 for (int i = 0; i < 2; i++)
368 {
369 device::optflow_farneback::gaussianBlurGpu(
370 frames_[i], smoothSize/2, blurredFrame[i], BORDER_REFLECT101_GPU, S(streams[i]));
371#if ENABLE_GPU_RESIZE
372 resize(blurredFrame[i], pyrLevel[i], Size(width, height), INTER_LINEAR, streams[i]);
373#else
374 Mat tmp1, tmp2;
375 tmp[i].download(tmp1);
376 resize(tmp1, tmp2, Size(width, height), INTER_LINEAR);
377 I[i].upload(tmp2);
378#endif
379 device::optflow_farneback::polynomialExpansionGpu(pyrLevel[i], polyN, R[i], S(streams[i]));
380 }
381 }
382
383 streams[1].waitForCompletion();
384 device::optflow_farneback::updateMatricesGpu(curFlowX, curFlowY, R[0], R[1], M, S(streams[0]));
385
386 if (flags & OPTFLOW_FARNEBACK_GAUSSIAN)
387 {
388 Mat g = getGaussianKernel(winSize, winSize/2*0.3f, CV_32F);
389 device::optflow_farneback::setGaussianBlurKernel(g.ptr<float>(winSize/2), winSize/2);
390 }
391 for (int i = 0; i < numIters; i++)
392 {
393 if (flags & OPTFLOW_FARNEBACK_GAUSSIAN)
394 updateFlow_gaussianBlur(R[0], R[1], curFlowX, curFlowY, M, bufM, winSize, i < numIters-1, streams);
395 else
396 updateFlow_boxFilter(R[0], R[1], curFlowX, curFlowY, M, bufM, winSize, i < numIters-1, streams);
397 }
398
399 prevFlowX = curFlowX;
400 prevFlowY = curFlowY;
401 }
402
403 flowx = curFlowX;
404 flowy = curFlowY;
405
406 if (!S(s))
407 streams[0].waitForCompletion();
408}
