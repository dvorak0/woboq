67static float4 scalarToCudaType(const cv::Scalar& in)
68{
69 return make_float4((float)in[0], (float)in[1], (float)in[2], (float)in[3]);
70}

72void cv::gpu::connectivityMask(const GpuMat& image, GpuMat& mask, const cv::Scalar& lo, const cv::Scalar& hi, Stream& s)
73{
74 CV_Assert(!image.empty());
75
76 int ch = image.channels();
77 CV_Assert(ch <= 4);
78
79 int depth = image.depth();
80
81 typedef void (*func_t)(const PtrStepSzb& image, PtrStepSzb edges, const float4& lo, const float4& hi, cudaStream_t stream);
82
83 static const func_t suppotLookup[8][4] =
84 { // 1, 2, 3, 4
85 { device::ccl::computeEdges<uchar>, 0, device::ccl::computeEdges<uchar3>, device::ccl::computeEdges<uchar4> },// CV_8U
86 { 0, 0, 0, 0 },// CV_16U
87 { device::ccl::computeEdges<ushort>, 0, device::ccl::computeEdges<ushort3>, device::ccl::computeEdges<ushort4> },// CV_8S
88 { 0, 0, 0, 0 },// CV_16S
89 { device::ccl::computeEdges<int>, 0, 0, 0 },// CV_32S
90 { device::ccl::computeEdges<float>, 0, 0, 0 },// CV_32F
91 { 0, 0, 0, 0 },// CV_64F
92 { 0, 0, 0, 0 } // CV_USRTYPE1
93 };
94
95 func_t f = suppotLookup[depth][ch - 1];
96 CV_Assert(f);
97
98 if (image.size() != mask.size() || mask.type() != CV_8UC1)
99 mask.create(image.size(), CV_8UC1);
100
101 cudaStream_t stream = StreamAccessor::getStream(s);
102 float4 culo = scalarToCudaType(lo), cuhi = scalarToCudaType(hi);
103 f(image, mask, culo, cuhi, stream);
104}

106void cv::gpu::labelComponents(const GpuMat& mask, GpuMat& components, int flags, Stream& s)
107{
108 CV_Assert(!mask.empty() && mask.type() == CV_8U);
109
110 if (!deviceSupports(SHARED_ATOMICS))
111 CV_Error(CV_StsNotImplemented, "The device doesn't support shared atomics and communicative synchronization!");
112
113 components.create(mask.size(), CV_32SC1);
114
115 cudaStream_t stream = StreamAccessor::getStream(s);
116 device::ccl::labelComponents(mask, components, flags, stream);
117}

121 typedef NppStatus (*init_func_t)(NppiSize oSize, NppiGraphcutState** ppState, Npp8u* pDeviceMem);

123 class NppiGraphcutStateHandler
124 {
125 public:
126 NppiGraphcutStateHandler(NppiSize sznpp, Npp8u* pDeviceMem, const init_func_t func)
127 {
128 nppSafeCall( func(sznpp, &pState, pDeviceMem) );
129 }
130
131 ~NppiGraphcutStateHandler()
132 {
133 nppSafeCall( nppiGraphcutFree(pState) );
134 }
135
136 operator NppiGraphcutState*()
137 {
138 return pState;
139 }
140
141 private:
142 NppiGraphcutState* pState;
143 };
144}

146void cv::gpu::graphcut(GpuMat& terminals, GpuMat& leftTransp, GpuMat& rightTransp, GpuMat& top, GpuMat& bottom, GpuMat& labels, GpuMat& buf, Stream& s)
147{
148#if (CUDART_VERSION < 5000)
149 CV_Assert(terminals.type() == CV_32S);
150#else
151 CV_Assert(terminals.type() == CV_32S || terminals.type() == CV_32F);
152#endif
153
154 Size src_size = terminals.size();
155
156 CV_Assert(leftTransp.size() == Size(src_size.height, src_size.width));
157 CV_Assert(leftTransp.type() == terminals.type());
158
159 CV_Assert(rightTransp.size() == Size(src_size.height, src_size.width));
160 CV_Assert(rightTransp.type() == terminals.type());
161
162 CV_Assert(top.size() == src_size);
163 CV_Assert(top.type() == terminals.type());
164
165 CV_Assert(bottom.size() == src_size);
166 CV_Assert(bottom.type() == terminals.type());
167
168 labels.create(src_size, CV_8U);
169
170 NppiSize sznpp;
171 sznpp.width = src_size.width;
172 sznpp.height = src_size.height;
173
174 int bufsz;
175 nppSafeCall( nppiGraphcutGetSize(sznpp, &bufsz) );
176
177 ensureSizeIsEnough(1, bufsz, CV_8U, buf);
178
179 cudaStream_t stream = StreamAccessor::getStream(s);
180
181 NppStreamHandler h(stream);
182
183 NppiGraphcutStateHandler state(sznpp, buf.ptr<Npp8u>(), nppiGraphcutInitAlloc);
184
185#if (CUDART_VERSION < 5000)
186 nppSafeCall( nppiGraphcut_32s8u(terminals.ptr<Npp32s>(), leftTransp.ptr<Npp32s>(), rightTransp.ptr<Npp32s>(), top.ptr<Npp32s>(), bottom.ptr<Npp32s>(),
187 static_cast<int>(terminals.step), static_cast<int>(leftTransp.step), sznpp, labels.ptr<Npp8u>(), static_cast<int>(labels.step), state) );
188#else
189 if (terminals.type() == CV_32S)
190 {
191 nppSafeCall( nppiGraphcut_32s8u(terminals.ptr<Npp32s>(), leftTransp.ptr<Npp32s>(), rightTransp.ptr<Npp32s>(), top.ptr<Npp32s>(), bottom.ptr<Npp32s>(),
192 static_cast<int>(terminals.step), static_cast<int>(leftTransp.step), sznpp, labels.ptr<Npp8u>(), static_cast<int>(labels.step), state) );
193 }
194 else
195 {
196 nppSafeCall( nppiGraphcut_32f8u(terminals.ptr<Npp32f>(), leftTransp.ptr<Npp32f>(), rightTransp.ptr<Npp32f>(), top.ptr<Npp32f>(), bottom.ptr<Npp32f>(),
197 static_cast<int>(terminals.step), static_cast<int>(leftTransp.step), sznpp, labels.ptr<Npp8u>(), static_cast<int>(labels.step), state) );
198 }
199#endif
200
201 if (stream == 0)
202 cudaSafeCall( cudaDeviceSynchronize() );
203}

205void cv::gpu::graphcut(GpuMat& terminals, GpuMat& leftTransp, GpuMat& rightTransp, GpuMat& top, GpuMat& topLeft, GpuMat& topRight,
206 GpuMat& bottom, GpuMat& bottomLeft, GpuMat& bottomRight, GpuMat& labels, GpuMat& buf, Stream& s)
207{
208#if (CUDART_VERSION < 5000)
209 CV_Assert(terminals.type() == CV_32S);
210#else
211 CV_Assert(terminals.type() == CV_32S || terminals.type() == CV_32F);
212#endif
213
214 Size src_size = terminals.size();
215
216 CV_Assert(leftTransp.size() == Size(src_size.height, src_size.width));
217 CV_Assert(leftTransp.type() == terminals.type());
218
219 CV_Assert(rightTransp.size() == Size(src_size.height, src_size.width));
220 CV_Assert(rightTransp.type() == terminals.type());
221
222 CV_Assert(top.size() == src_size);
223 CV_Assert(top.type() == terminals.type());
224
225 CV_Assert(topLeft.size() == src_size);
226 CV_Assert(topLeft.type() == terminals.type());
227
228 CV_Assert(topRight.size() == src_size);
229 CV_Assert(topRight.type() == terminals.type());
230
231 CV_Assert(bottom.size() == src_size);
232 CV_Assert(bottom.type() == terminals.type());
233
234 CV_Assert(bottomLeft.size() == src_size);
235 CV_Assert(bottomLeft.type() == terminals.type());
236
237 CV_Assert(bottomRight.size() == src_size);
238 CV_Assert(bottomRight.type() == terminals.type());
239
240 labels.create(src_size, CV_8U);
241
242 NppiSize sznpp;
243 sznpp.width = src_size.width;
244 sznpp.height = src_size.height;
245
246 int bufsz;
247 nppSafeCall( nppiGraphcut8GetSize(sznpp, &bufsz) );
248
249 ensureSizeIsEnough(1, bufsz, CV_8U, buf);
250
251 cudaStream_t stream = StreamAccessor::getStream(s);
252
253 NppStreamHandler h(stream);
254
255 NppiGraphcutStateHandler state(sznpp, buf.ptr<Npp8u>(), nppiGraphcut8InitAlloc);
256
257#if (CUDART_VERSION < 5000)
258 nppSafeCall( nppiGraphcut8_32s8u(terminals.ptr<Npp32s>(), leftTransp.ptr<Npp32s>(), rightTransp.ptr<Npp32s>(),
259 top.ptr<Npp32s>(), topLeft.ptr<Npp32s>(), topRight.ptr<Npp32s>(),
260 bottom.ptr<Npp32s>(), bottomLeft.ptr<Npp32s>(), bottomRight.ptr<Npp32s>(),
261 static_cast<int>(terminals.step), static_cast<int>(leftTransp.step), sznpp, labels.ptr<Npp8u>(), static_cast<int>(labels.step), state) );
262#else
263 if (terminals.type() == CV_32S)
264 {
265 nppSafeCall( nppiGraphcut8_32s8u(terminals.ptr<Npp32s>(), leftTransp.ptr<Npp32s>(), rightTransp.ptr<Npp32s>(),
266 top.ptr<Npp32s>(), topLeft.ptr<Npp32s>(), topRight.ptr<Npp32s>(),
267 bottom.ptr<Npp32s>(), bottomLeft.ptr<Npp32s>(), bottomRight.ptr<Npp32s>(),
268 static_cast<int>(terminals.step), static_cast<int>(leftTransp.step), sznpp, labels.ptr<Npp8u>(), static_cast<int>(labels.step), state) );
269 }
270 else
271 {
272 nppSafeCall( nppiGraphcut8_32f8u(terminals.ptr<Npp32f>(), leftTransp.ptr<Npp32f>(), rightTransp.ptr<Npp32f>(),
273 top.ptr<Npp32f>(), topLeft.ptr<Npp32f>(), topRight.ptr<Npp32f>(),
274 bottom.ptr<Npp32f>(), bottomLeft.ptr<Npp32f>(), bottomRight.ptr<Npp32f>(),
275 static_cast<int>(terminals.step), static_cast<int>(leftTransp.step), sznpp, labels.ptr<Npp8u>(), static_cast<int>(labels.step), state) );
276 }
277#endif
278
279 if (stream == 0)
280 cudaSafeCall( cudaDeviceSynchronize() );
281}
