57void cv::gpu::resize(const GpuMat& src, GpuMat& dst, Size dsize, double fx, double fy, int interpolation, Stream& stream)
58{
59 typedef void (*func_t)(const PtrStepSzb& src, const PtrStepSzb& srcWhole, int yoff, int xoff, const PtrStepSzb& dst, float fy, float fx, int interpolation, cudaStream_t stream);
60
61#ifdef OPENCV_TINY_GPU_MODULE
62 static const func_t funcs[6][4] =
63 {
64 {device::resize<uchar> , 0 /*device::resize<uchar2>*/ , device::resize<uchar3> , device::resize<uchar4> },
65 {0, 0, 0, 0},
66 {0, 0, 0, 0},
67 {0, 0, 0, 0},
68 {0, 0, 0, 0},
69 {device::resize<float> , 0 /*device::resize<float2>*/ , device::resize<float3> , device::resize<float4> }
70 };
71#else
72 static const func_t funcs[6][4] =
73 {
74 {device::resize<uchar> , 0 /*device::resize<uchar2>*/ , device::resize<uchar3> , device::resize<uchar4> },
75 {0 /*device::resize<schar>*/, 0 /*device::resize<char2>*/ , 0 /*device::resize<char3>*/, 0 /*device::resize<char4>*/},
76 {device::resize<ushort> , 0 /*device::resize<ushort2>*/, device::resize<ushort3> , device::resize<ushort4> },
77 {device::resize<short> , 0 /*device::resize<short2>*/ , device::resize<short3> , device::resize<short4> },
78 {0 /*device::resize<int>*/ , 0 /*device::resize<int2>*/ , 0 /*device::resize<int3>*/ , 0 /*device::resize<int4>*/ },
79 {device::resize<float> , 0 /*device::resize<float2>*/ , device::resize<float3> , device::resize<float4> }
80 };
81#endif
82
83 CV_Assert( src.depth() <= CV_32F && src.channels() <= 4 );
84 CV_Assert( interpolation == INTER_NEAREST || interpolation == INTER_LINEAR || interpolation == INTER_CUBIC || interpolation == INTER_AREA );
85 CV_Assert( !(dsize == Size()) || (fx > 0 && fy > 0) );
86
87 if (dsize == Size())
88 {
89 dsize = Size(saturate_cast<int>(src.cols * fx), saturate_cast<int>(src.rows * fy));
90 }
91 else
92 {
93 fx = static_cast<double>(dsize.width) / src.cols;
94 fy = static_cast<double>(dsize.height) / src.rows;
95 }
96
97 dst.create(dsize, src.type());
98
99 if (dsize == src.size())
100 {
101 if (stream)
102 stream.enqueueCopy(src, dst);
103 else
104 src.copyTo(dst);
105 return;
106 }
107
108 const func_t func = funcs[src.depth()][src.channels() - 1];
109
110 if (!func)
111 CV_Error(CV_StsUnsupportedFormat, "Unsupported combination of source and destination types");
112
113 Size wholeSize;
114 Point ofs;
115 src.locateROI(wholeSize, ofs);
116 PtrStepSzb wholeSrc(wholeSize.height, wholeSize.width, src.datastart, src.step);
117
118 func(src, wholeSrc, ofs.y, ofs.x, dst, static_cast<float>(1.0 / fy), static_cast<float>(1.0 / fx), interpolation, StreamAccessor::getStream(stream));
119}
