109struct SegmLinkVal
110{
111 SegmLinkVal() {}
112 SegmLinkVal(int dr_, int dsp_) : dr(dr_), dsp(dsp_) {}
113 bool operator <(const SegmLinkVal& other) const
114 {
115 return dr + dsp < other.dr + other.dsp;
116 }
117 int dr;
118 int dsp;
119};
120
121
122struct SegmLink
123{
124 SegmLink() {}
125 SegmLink(int from_, int to_, const SegmLinkVal& val_)
126 : from(from_), to(to_), val(val_) {}
127 bool operator <(const SegmLink& other) const
128 {
129 return val < other.val;
130 }
131 int from;
132 int to;
133 SegmLinkVal val;
134};

140DjSets::DjSets(int n) : parent(n), rank(n, 0), size(n, 1)
141{
142 for (int i = 0; i < n; ++i)
143 parent[i] = i;
144}
145
146
147inline int DjSets::find(int elem)
148{
149 int set = elem;
150 while (set != parent[set])
151 set = parent[set];
152 while (elem != parent[elem])
153 {
154 int next = parent[elem];
155 parent[elem] = set;
156 elem = next;
157 }
158 return set;
159}
160
161
162inline int DjSets::merge(int set1, int set2)
163{
164 if (rank[set1] < rank[set2])
165 {
166 parent[set1] = set2;
167 size[set2] += size[set1];
168 return set2;
169 }
170 if (rank[set2] < rank[set1])
171 {
172 parent[set2] = set1;
173 size[set1] += size[set2];
174 return set1;
175 }
176 parent[set1] = set2;
177 rank[set2]++;
178 size[set2] += size[set1];
179 return set2;
180}

183template <typename T>
184Graph<T>::Graph(int numv_, int nume_max_) : start(numv_, -1), edges(nume_max_)
185{
186 this->numv = numv_;
187 this->nume_max = nume_max_;
188 nume = 0;
189}

192template <typename T>
193inline void Graph<T>::addEdge(int from, int to, const T& val)
194{
195 edges[nume] = Edge(to, start[from], val);
196 start[from] = nume;
197 nume++;
198}

201inline int pix(int y, int x, int ncols)

207inline int sqr(int x)
208{
209 return x * x;
210}

212
213inline int dist2(const cv::Vec4b& lhs, const cv::Vec4b& rhs)
214{
215 return sqr(lhs[0] - rhs[0]) + sqr(lhs[1] - rhs[1]) + sqr(lhs[2] - rhs[2]);
216}
217
218
219inline int dist2(const cv::Vec2s& lhs, const cv::Vec2s& rhs)
220{
221 return sqr(lhs[0] - rhs[0]) + sqr(lhs[1] - rhs[1]);
222}

227void cv::gpu::meanShiftSegmentation(const GpuMat& src, Mat& dst, int sp, int sr, int minsize, TermCriteria criteria)
228{
229 CV_Assert(src.type() == CV_8UC4);
230 const int nrows = src.rows;
231 const int ncols = src.cols;
232 const int hr = sr;
233 const int hsp = sp;
234
235 // Perform mean shift procedure and obtain region and spatial maps
236 GpuMat d_rmap, d_spmap;
237 meanShiftProc(src, d_rmap, d_spmap, sp, sr, criteria);
238 Mat rmap(d_rmap);
239 Mat spmap(d_spmap);
240
241 Graph<SegmLinkVal> g(nrows * ncols, 4 * (nrows - 1) * (ncols - 1)
242 + (nrows - 1) + (ncols - 1));
243
244 // Make region adjacent graph from image
245 Vec4b r1;
246 Vec4b r2[4];
247 Vec2s sp1;
248 Vec2s sp2[4];
249 int dr[4];
250 int dsp[4];
251 for (int y = 0; y < nrows - 1; ++y)
252 {
253 Vec4b* ry = rmap.ptr<Vec4b>(y);
254 Vec4b* ryp = rmap.ptr<Vec4b>(y + 1);
255 Vec2s* spy = spmap.ptr<Vec2s>(y);
256 Vec2s* spyp = spmap.ptr<Vec2s>(y + 1);
257 for (int x = 0; x < ncols - 1; ++x)
258 {
259 r1 = ry[x];
260 sp1 = spy[x];
261
262 r2[0] = ry[x + 1];
263 r2[1] = ryp[x];
264 r2[2] = ryp[x + 1];
265 r2[3] = ryp[x];
266
267 sp2[0] = spy[x + 1];
268 sp2[1] = spyp[x];
269 sp2[2] = spyp[x + 1];
270 sp2[3] = spyp[x];
271
272 dr[0] = dist2(r1, r2[0]);
273 dr[1] = dist2(r1, r2[1]);
274 dr[2] = dist2(r1, r2[2]);
275 dsp[0] = dist2(sp1, sp2[0]);
276 dsp[1] = dist2(sp1, sp2[1]);
277 dsp[2] = dist2(sp1, sp2[2]);
278
279 r1 = ry[x + 1];
280 sp1 = spy[x + 1];
281
282 dr[3] = dist2(r1, r2[3]);
283 dsp[3] = dist2(sp1, sp2[3]);
284
285 g.addEdge(pix(y, x, ncols), pix(y, x + 1, ncols), SegmLinkVal(dr[0], dsp[0]));
286 g.addEdge(pix(y, x, ncols), pix(y + 1, x, ncols), SegmLinkVal(dr[1], dsp[1]));
287 g.addEdge(pix(y, x, ncols), pix(y + 1, x + 1, ncols), SegmLinkVal(dr[2], dsp[2]));
288 g.addEdge(pix(y, x + 1, ncols), pix(y + 1, x, ncols), SegmLinkVal(dr[3], dsp[3]));
289 }
290 }
291 for (int y = 0; y < nrows - 1; ++y)
292 {
293 r1 = rmap.at<Vec4b>(y, ncols - 1);
294 r2[0] = rmap.at<Vec4b>(y + 1, ncols - 1);
295 sp1 = spmap.at<Vec2s>(y, ncols - 1);
296 sp2[0] = spmap.at<Vec2s>(y + 1, ncols - 1);
297 dr[0] = dist2(r1, r2[0]);
298 dsp[0] = dist2(sp1, sp2[0]);
299 g.addEdge(pix(y, ncols - 1, ncols), pix(y + 1, ncols - 1, ncols), SegmLinkVal(dr[0], dsp[0]));
300 }
301 for (int x = 0; x < ncols - 1; ++x)
302 {
303 r1 = rmap.at<Vec4b>(nrows - 1, x);
304 r2[0] = rmap.at<Vec4b>(nrows - 1, x + 1);
305 sp1 = spmap.at<Vec2s>(nrows - 1, x);
306 sp2[0] = spmap.at<Vec2s>(nrows - 1, x + 1);
307 dr[0] = dist2(r1, r2[0]);
308 dsp[0] = dist2(sp1, sp2[0]);
309 g.addEdge(pix(nrows - 1, x, ncols), pix(nrows - 1, x + 1, ncols), SegmLinkVal(dr[0], dsp[0]));
310 }
311
312 DjSets comps(g.numv);
313
314 // Find adjacent components
315 for (int v = 0; v < g.numv; ++v)
316 {
317 for (int e_it = g.start[v]; e_it != -1; e_it = g.edges[e_it].next)
318 {
319 int c1 = comps.find(v);
320 int c2 = comps.find(g.edges[e_it].to);
321 if (c1 != c2 && g.edges[e_it].val.dr < hr && g.edges[e_it].val.dsp < hsp)
322 comps.merge(c1, c2);
323 }
324 }
325
326 vector<SegmLink> edges;
327 edges.reserve(g.numv);
328
329 // Prepare edges connecting differnet components
330 for (int v = 0; v < g.numv; ++v)
331 {
332 int c1 = comps.find(v);
333 for (int e_it = g.start[v]; e_it != -1; e_it = g.edges[e_it].next)
334 {
335 int c2 = comps.find(g.edges[e_it].to);
336 if (c1 != c2)
337 edges.push_back(SegmLink(c1, c2, g.edges[e_it].val));
338 }
339 }
340
341 // Sort all graph's edges connecting differnet components (in asceding order)
342 sort(edges.begin(), edges.end());
343
344 // Exclude small components (starting from the nearest couple)
345 for (size_t i = 0; i < edges.size(); ++i)
346 {
347 int c1 = comps.find(edges[i].from);
348 int c2 = comps.find(edges[i].to);
349 if (c1 != c2 && (comps.size[c1] < minsize || comps.size[c2] < minsize))
350 comps.merge(c1, c2);
351 }
352
353 // Compute sum of the pixel's colors which are in the same segment
354 Mat h_src(src);
355 vector<Vec4i> sumcols(nrows * ncols, Vec4i(0, 0, 0, 0));
356 for (int y = 0; y < nrows; ++y)
357 {
358 Vec4b* h_srcy = h_src.ptr<Vec4b>(y);
359 for (int x = 0; x < ncols; ++x)
360 {
361 int parent = comps.find(pix(y, x, ncols));
362 Vec4b col = h_srcy[x];
363 Vec4i& sumcol = sumcols[parent];
364 sumcol[0] += col[0];
365 sumcol[1] += col[1];
366 sumcol[2] += col[2];
367 }
368 }
369
370 // Create final image, color of each segment is the average color of its pixels
371 dst.create(src.size(), src.type());
372
373 for (int y = 0; y < nrows; ++y)
374 {
375 Vec4b* dsty = dst.ptr<Vec4b>(y);
376 for (int x = 0; x < ncols; ++x)
377 {
378 int parent = comps.find(pix(y, x, ncols));
379 const Vec4i& sumcol = sumcols[parent];
380 Vec4b& dstcol = dsty[x];
381 dstcol[0] = static_cast<uchar>(sumcol[0] / comps.size[parent]);
382 dstcol[1] = static_cast<uchar>(sumcol[1] / comps.size[parent]);
383 dstcol[2] = static_cast<uchar>(sumcol[2] / comps.size[parent]);
384 dstcol[3] = 255;
385 }
386 }
387}
