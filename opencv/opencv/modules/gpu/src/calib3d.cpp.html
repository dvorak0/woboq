86 void transformPointsCaller(const GpuMat& src, const Mat& rvec, const Mat& tvec, GpuMat& dst, cudaStream_t stream)
87 {
88 CV_Assert(src.rows == 1 && src.cols > 0 && src.type() == CV_32FC3);
89 CV_Assert(rvec.size() == Size(3, 1) && rvec.type() == CV_32F);
90 CV_Assert(tvec.size() == Size(3, 1) && tvec.type() == CV_32F);
91
92 // Convert rotation vector into matrix
93 Mat rot;
94 Rodrigues(rvec, rot);
95
96 dst.create(src.size(), src.type());
97 transform_points::call(src, rot.ptr<float>(), tvec.ptr<float>(), dst, stream);
98 }
99}

132 // Selects subset_size random different points from [0, num_points - 1] range
133 void selectRandom(int subset_size, int num_points, vector<int>& subset)
134 {
135 subset.resize(subset_size);
136 for (int i = 0; i < subset_size; ++i)
137 {
138 bool was;
139 do
140 {
141 subset[i] = cv::theRNG().uniform(0, num_points);
142 was = false;
143 for (int j = 0; j < i; ++j)
144 {
145 if (subset[j] == subset[i])
146 {
147 was = true;
148 break;
149 }
150 }
151 } while (was);
152 }
153 }

155 // Computes rotation, translation pair for small subsets if the input data
156 class TransformHypothesesGenerator : public ParallelLoopBody
157 {
158 public:
159 TransformHypothesesGenerator(const Mat& object_, const Mat& image_, const Mat& dist_coef_,
160 const Mat& camera_mat_, int num_points_, int subset_size_,
161 Mat rot_matrices_, Mat transl_vectors_)
162 : object(&object_), image(&image_), dist_coef(&dist_coef_), camera_mat(&camera_mat_),
163 num_points(num_points_), subset_size(subset_size_), rot_matrices(rot_matrices_),
164 transl_vectors(transl_vectors_) {}
165
166 void operator()(const Range& range) const
167 {
168 // Input data for generation of the current hypothesis
169 vector<int> subset_indices(subset_size);
170 Mat_<Point3f> object_subset(1, subset_size);
171 Mat_<Point2f> image_subset(1, subset_size);
172
173 // Current hypothesis data
174 Mat rot_vec(1, 3, CV_64F);
175 Mat rot_mat(3, 3, CV_64F);
176 Mat transl_vec(1, 3, CV_64F);
177
178 for (int iter = range.start; iter < range.end; ++iter)
179 {
180 selectRandom(subset_size, num_points, subset_indices);
181 for (int i = 0; i < subset_size; ++i)
182 {
183 object_subset(0, i) = object->at<Point3f>(subset_indices[i]);
184 image_subset(0, i) = image->at<Point2f>(subset_indices[i]);
185 }
186
187 solvePnP(object_subset, image_subset, *camera_mat, *dist_coef, rot_vec, transl_vec);
188
189 // Remember translation vector
190 Mat transl_vec_ = transl_vectors.colRange(iter * 3, (iter + 1) * 3);
191 transl_vec = transl_vec.reshape(0, 1);
192 transl_vec.convertTo(transl_vec_, CV_32F);
193
194 // Remember rotation matrix
195 Rodrigues(rot_vec, rot_mat);
196 Mat rot_mat_ = rot_matrices.colRange(iter * 9, (iter + 1) * 9).reshape(0, 3);
197 rot_mat.convertTo(rot_mat_, CV_32F);
198 }
199 }
200
201 const Mat* object;
202 const Mat* image;
203 const Mat* dist_coef;
204 const Mat* camera_mat;
205 int num_points;
206 int subset_size;
207
208 // Hypotheses storage (global)
209 Mat rot_matrices;
210 Mat transl_vectors;
211 };

214void cv::gpu::solvePnPRansac(const Mat& object, const Mat& image, const Mat& camera_mat,
215 const Mat& dist_coef, Mat& rvec, Mat& tvec, bool use_extrinsic_guess,
216 int num_iters, float max_dist, int min_inlier_count,
217 vector<int>* inliers)
218{
219 (void)min_inlier_count;
220 CV_Assert(object.rows == 1 && object.cols > 0 && object.type() == CV_32FC3);
221 CV_Assert(image.rows == 1 && image.cols > 0 && image.type() == CV_32FC2);
222 CV_Assert(object.cols == image.cols);
223 CV_Assert(camera_mat.size() == Size(3, 3) && camera_mat.type() == CV_32F);
224 CV_Assert(!use_extrinsic_guess); // We don't support initial guess for now
225 CV_Assert(num_iters <= solve_pnp_ransac::maxNumIters());
226
227 const int subset_size = 4;
228 const int num_points = object.cols;
229 CV_Assert(num_points >= subset_size);
230
231 // Unapply distortion and intrinsic camera transformations
232 Mat eye_camera_mat = Mat::eye(3, 3, CV_32F);
233 Mat empty_dist_coef;
234 Mat image_normalized;
235 undistortPoints(image, image_normalized, camera_mat, dist_coef, Mat(), eye_camera_mat);
236
237 // Hypotheses storage (global)
238 Mat rot_matrices(1, num_iters * 9, CV_32F);
239 Mat transl_vectors(1, num_iters * 3, CV_32F);
240
241 // Generate set of hypotheses using small subsets of the input data
242 TransformHypothesesGenerator body(object, image_normalized, empty_dist_coef, eye_camera_mat,
243 num_points, subset_size, rot_matrices, transl_vectors);
244 parallel_for_(Range(0, num_iters), body);
245
246 // Compute scores (i.e. number of inliers) for each hypothesis
247 GpuMat d_object(object);
248 GpuMat d_image_normalized(image_normalized);
249 GpuMat d_hypothesis_scores(1, num_iters, CV_32S);
250 solve_pnp_ransac::computeHypothesisScores(
251 num_iters, num_points, rot_matrices.ptr<float>(), transl_vectors.ptr<float3>(),
252 d_object.ptr<float3>(), d_image_normalized.ptr<float2>(), max_dist * max_dist,
253 d_hypothesis_scores.ptr<int>());
254
255 // Find the best hypothesis index
256 Point best_idx;
257 double best_score;
258 minMaxLoc(d_hypothesis_scores, NULL, &best_score, NULL, &best_idx);
259 int num_inliers = static_cast<int>(best_score);
260
261 // Extract the best hypothesis data
262
263 Mat rot_mat = rot_matrices.colRange(best_idx.x * 9, (best_idx.x + 1) * 9).reshape(0, 3);
264 Rodrigues(rot_mat, rvec);
265 rvec = rvec.reshape(0, 1);
266
267 tvec = transl_vectors.colRange(best_idx.x * 3, (best_idx.x + 1) * 3).clone();
268 tvec = tvec.reshape(0, 1);
269
270 // Build vector of inlier indices
271 if (inliers != NULL)
272 {
273 inliers->clear();
274 inliers->reserve(num_inliers);
275
276 Point3f p, p_transf;
277 Point2f p_proj;
278 const float* rot = rot_mat.ptr<float>();
279 const float* transl = tvec.ptr<float>();
280
281 for (int i = 0; i < num_points; ++i)
282 {
283 p = object.at<Point3f>(0, i);
284 p_transf.x = rot[0] * p.x + rot[1] * p.y + rot[2] * p.z + transl[0];
285 p_transf.y = rot[3] * p.x + rot[4] * p.y + rot[5] * p.z + transl[1];
286 p_transf.z = rot[6] * p.x + rot[7] * p.y + rot[8] * p.z + transl[2];
287 p_proj.x = p_transf.x / p_transf.z;
288 p_proj.y = p_transf.y / p_transf.z;
289 if (norm(p_proj - image_normalized.at<Point2f>(0, i)) < max_dist)
290 inliers->push_back(i);
291 }
292 }
293}
