102void cv::gpu::StereoConstantSpaceBP::estimateRecommendedParams(int width, int height, int& ndisp, int& iters, int& levels, int& nr_plane)
103{
104 ndisp = (int) ((float) width / 3.14f);
105 if ((ndisp & 1) != 0)
106 ndisp++;
107
108 int mm = ::max(width, height);
109 iters = mm / 100 + ((mm > 1200)? - 4 : 4);
110
111 levels = (int)::log(static_cast<double>(mm)) * 2 / 3;
112 if (levels == 0) levels++;
113
114 nr_plane = (int) ((float) ndisp / std::pow(2.0, levels + 1));
115}

139template<class T>
140static void csbp_operator(StereoConstantSpaceBP& rthis, GpuMat& mbuf, GpuMat& temp, GpuMat& out, const GpuMat& left, const GpuMat& right, GpuMat& disp, Stream& stream)
141{
142 CV_DbgAssert(0 < rthis.ndisp && 0 < rthis.iters && 0 < rthis.levels && 0 < rthis.nr_plane
143 && left.rows == right.rows && left.cols == right.cols && left.type() == right.type());
144
145 CV_Assert(rthis.levels <= 8 && (left.type() == CV_8UC1 || left.type() == CV_8UC3 || left.type() == CV_8UC4));
146
147 const Scalar zero = Scalar::all(0);
148
149 cudaStream_t cudaStream = StreamAccessor::getStream(stream);
150
151 ////////////////////////////////////////////////////////////////////////////////////////////
152 // Init
153
154 int rows = left.rows;
155 int cols = left.cols;
156
157 rthis.levels = min(rthis.levels, int(log((double)rthis.ndisp) / log(2.0)));
158 int levels = rthis.levels;
159
160 // compute sizes
161 AutoBuffer<int> buf(levels * 3);
162 int* cols_pyr = buf;
163 int* rows_pyr = cols_pyr + levels;
164 int* nr_plane_pyr = rows_pyr + levels;
165
166 cols_pyr[0] = cols;
167 rows_pyr[0] = rows;
168 nr_plane_pyr[0] = rthis.nr_plane;
169
170 for (int i = 1; i < levels; i++)
171 {
172 cols_pyr[i] = cols_pyr[i-1] / 2;
173 rows_pyr[i] = rows_pyr[i-1] / 2;
174 nr_plane_pyr[i] = nr_plane_pyr[i-1] * 2;
175 }
176
177
178 GpuMat u[2], d[2], l[2], r[2], disp_selected_pyr[2], data_cost, data_cost_selected;
179
180
181 //allocate buffers
182 int buffers_count = 10; // (up + down + left + right + disp_selected_pyr) * 2
183 buffers_count += 2; // data_cost has twice more rows than other buffers, what's why +2, not +1;
184 buffers_count += 1; // data_cost_selected
185 mbuf.create(rows * rthis.nr_plane * buffers_count, cols, DataType<T>::type);
186
187 data_cost = mbuf.rowRange(0, rows * rthis.nr_plane * 2);
188 data_cost_selected = mbuf.rowRange(data_cost.rows, data_cost.rows + rows * rthis.nr_plane);
189
190 for(int k = 0; k < 2; ++k) // in/out
191 {
192 GpuMat sub1 = mbuf.rowRange(data_cost.rows + data_cost_selected.rows, mbuf.rows);
193 GpuMat sub2 = sub1.rowRange((k+0)*sub1.rows/2, (k+1)*sub1.rows/2);
194
195 GpuMat *buf_ptrs[] = { &u[k], &d[k], &l[k], &r[k], &disp_selected_pyr[k] };
196 for(int _r = 0; _r < 5; ++_r)
197 {
198 *buf_ptrs[_r] = sub2.rowRange(_r * sub2.rows/5, (_r+1) * sub2.rows/5);
199 assert(buf_ptrs[_r]->cols == cols && buf_ptrs[_r]->rows == rows * rthis.nr_plane);
200 }
201 };
202
203 size_t elem_step = mbuf.step / sizeof(T);
204
205 Size temp_size = data_cost.size();
206 if ((size_t)temp_size.area() < elem_step * rows_pyr[levels - 1] * rthis.ndisp)
207 temp_size = Size(static_cast<int>(elem_step), rows_pyr[levels - 1] * rthis.ndisp);
208
209 temp.create(temp_size, DataType<T>::type);
210
211 ////////////////////////////////////////////////////////////////////////////
212 // Compute
213
214 load_constants(rthis.ndisp, rthis.max_data_term, rthis.data_weight, rthis.max_disc_term, rthis.disc_single_jump, rthis.min_disp_th, left, right, temp);
215
216 if (stream)
217 {
218 stream.enqueueMemSet(l[0], zero);
219 stream.enqueueMemSet(d[0], zero);
220 stream.enqueueMemSet(r[0], zero);
221 stream.enqueueMemSet(u[0], zero);
222
223 stream.enqueueMemSet(l[1], zero);
224 stream.enqueueMemSet(d[1], zero);
225 stream.enqueueMemSet(r[1], zero);
226 stream.enqueueMemSet(u[1], zero);
227
228 stream.enqueueMemSet(data_cost, zero);
229 stream.enqueueMemSet(data_cost_selected, zero);
230 }
231 else
232 {
233 l[0].setTo(zero);
234 d[0].setTo(zero);
235 r[0].setTo(zero);
236 u[0].setTo(zero);
237
238 l[1].setTo(zero);
239 d[1].setTo(zero);
240 r[1].setTo(zero);
241 u[1].setTo(zero);
242
243 data_cost.setTo(zero);
244 data_cost_selected.setTo(zero);
245 }
246
247 int cur_idx = 0;
248
249 for (int i = levels - 1; i >= 0; i--)
250 {
251 if (i == levels - 1)
252 {
253 init_data_cost(left.rows, left.cols, disp_selected_pyr[cur_idx].ptr<T>(), data_cost_selected.ptr<T>(),
254 elem_step, rows_pyr[i], cols_pyr[i], i, nr_plane_pyr[i], rthis.ndisp, left.channels(), rthis.use_local_init_data_cost, cudaStream);
255 }
256 else
257 {
258 compute_data_cost(disp_selected_pyr[cur_idx].ptr<T>(), data_cost.ptr<T>(), elem_step,
259 left.rows, left.cols, rows_pyr[i], cols_pyr[i], rows_pyr[i+1], i, nr_plane_pyr[i+1], left.channels(), cudaStream);
260
261 int new_idx = (cur_idx + 1) & 1;
262
263 init_message(u[new_idx].ptr<T>(), d[new_idx].ptr<T>(), l[new_idx].ptr<T>(), r[new_idx].ptr<T>(),
264 u[cur_idx].ptr<T>(), d[cur_idx].ptr<T>(), l[cur_idx].ptr<T>(), r[cur_idx].ptr<T>(),
265 disp_selected_pyr[new_idx].ptr<T>(), disp_selected_pyr[cur_idx].ptr<T>(),
266 data_cost_selected.ptr<T>(), data_cost.ptr<T>(), elem_step, rows_pyr[i],
267 cols_pyr[i], nr_plane_pyr[i], rows_pyr[i+1], cols_pyr[i+1], nr_plane_pyr[i+1], cudaStream);
268
269 cur_idx = new_idx;
270 }
271
272 calc_all_iterations(u[cur_idx].ptr<T>(), d[cur_idx].ptr<T>(), l[cur_idx].ptr<T>(), r[cur_idx].ptr<T>(),
273 data_cost_selected.ptr<T>(), disp_selected_pyr[cur_idx].ptr<T>(), elem_step,
274 rows_pyr[i], cols_pyr[i], nr_plane_pyr[i], rthis.iters, cudaStream);
275 }
276
277 if (disp.empty())
278 disp.create(rows, cols, CV_16S);
279
280 out = ((disp.type() == CV_16S) ? disp : (out.create(rows, cols, CV_16S), out));
281
282 if (stream)
283 stream.enqueueMemSet(out, zero);
284 else
285 out.setTo(zero);
286
287 compute_disp(u[cur_idx].ptr<T>(), d[cur_idx].ptr<T>(), l[cur_idx].ptr<T>(), r[cur_idx].ptr<T>(),
288 data_cost_selected.ptr<T>(), disp_selected_pyr[cur_idx].ptr<T>(), elem_step, out, nr_plane_pyr[0], cudaStream);
289
290 if (disp.type() != CV_16S)
291 {
292 if (stream)
293 stream.enqueueConvert(out, disp, disp.type());
294 else
295 out.convertTo(disp, disp.type());
296 }
297}

300typedef void (*csbp_operator_t)(StereoConstantSpaceBP& rthis, GpuMat& mbuf,
301 GpuMat& temp, GpuMat& out, const GpuMat& left, const GpuMat& right, GpuMat& disp, Stream& stream);
302
303const static csbp_operator_t operators[] = {0, 0, 0, csbp_operator<short>, 0, csbp_operator<float>, 0, 0};
304
305void cv::gpu::StereoConstantSpaceBP::operator()(const GpuMat& left, const GpuMat& right, GpuMat& disp, Stream& stream)
306{
307 CV_Assert(msg_type == CV_32F || msg_type == CV_16S);
308 operators[msg_type](*this, messages_buffers, temp, out, left, right, disp, stream);
309}
