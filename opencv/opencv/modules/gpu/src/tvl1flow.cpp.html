58cv::gpu::OpticalFlowDual_TVL1_GPU::OpticalFlowDual_TVL1_GPU()
59{
60 tau = 0.25;
61 lambda = 0.15;
62 theta = 0.3;
63 nscales = 5;
64 warps = 5;
65 epsilon = 0.01;
66 iterations = 300;
67 useInitialFlow = false;
68}

70void cv::gpu::OpticalFlowDual_TVL1_GPU::operator ()(const GpuMat& I0, const GpuMat& I1, GpuMat& flowx, GpuMat& flowy)
71{
72 CV_Assert( I0.type() == CV_8UC1 || I0.type() == CV_32FC1 );
73 CV_Assert( I0.size() == I1.size() );
74 CV_Assert( I0.type() == I1.type() );
75 CV_Assert( !useInitialFlow || (flowx.size() == I0.size() && flowx.type() == CV_32FC1 && flowy.size() == flowx.size() && flowy.type() == flowx.type()) );
76 CV_Assert( nscales > 0 );
77
78 // allocate memory for the pyramid structure
79 I0s.resize(nscales);
80 I1s.resize(nscales);
81 u1s.resize(nscales);
82 u2s.resize(nscales);
83
84 I0.convertTo(I0s[0], CV_32F, I0.depth() == CV_8U ? 1.0 : 255.0);
85 I1.convertTo(I1s[0], CV_32F, I1.depth() == CV_8U ? 1.0 : 255.0);
86
87 if (!useInitialFlow)
88 {
89 flowx.create(I0.size(), CV_32FC1);
90 flowy.create(I0.size(), CV_32FC1);
91 }
92
93 u1s[0] = flowx;
94 u2s[0] = flowy;
95
96 I1x_buf.create(I0.size(), CV_32FC1);
97 I1y_buf.create(I0.size(), CV_32FC1);
98
99 I1w_buf.create(I0.size(), CV_32FC1);
100 I1wx_buf.create(I0.size(), CV_32FC1);
101 I1wy_buf.create(I0.size(), CV_32FC1);
102
103 grad_buf.create(I0.size(), CV_32FC1);
104 rho_c_buf.create(I0.size(), CV_32FC1);
105
106 p11_buf.create(I0.size(), CV_32FC1);
107 p12_buf.create(I0.size(), CV_32FC1);
108 p21_buf.create(I0.size(), CV_32FC1);
109 p22_buf.create(I0.size(), CV_32FC1);
110
111 diff_buf.create(I0.size(), CV_32FC1);
112
113 // create the scales
114 for (int s = 1; s < nscales; ++s)
115 {
116 gpu::pyrDown(I0s[s - 1], I0s[s]);
117 gpu::pyrDown(I1s[s - 1], I1s[s]);
118
119 if (I0s[s].cols < 16 || I0s[s].rows < 16)
120 {
121 nscales = s;
122 break;
123 }
124
125 if (useInitialFlow)
126 {
127 gpu::pyrDown(u1s[s - 1], u1s[s]);
128 gpu::pyrDown(u2s[s - 1], u2s[s]);
129
130 gpu::multiply(u1s[s], Scalar::all(0.5), u1s[s]);
131 gpu::multiply(u2s[s], Scalar::all(0.5), u2s[s]);
132 }
133 else
134 {
135 u1s[s].create(I0s[s].size(), CV_32FC1);
136 u2s[s].create(I0s[s].size(), CV_32FC1);
137 }
138 }
139
140 if (!useInitialFlow)
141 {
142 u1s[nscales-1].setTo(Scalar::all(0));
143 u2s[nscales-1].setTo(Scalar::all(0));
144 }
145
146 // pyramidal structure for computing the optical flow
147 for (int s = nscales - 1; s >= 0; --s)
148 {
149 // compute the optical flow at the current scale
150 procOneScale(I0s[s], I1s[s], u1s[s], u2s[s]);
151
152 // if this was the last scale, finish now
153 if (s == 0)
154 break;
155
156 // otherwise, upsample the optical flow
157
158 // zoom the optical flow for the next finer scale
159 gpu::resize(u1s[s], u1s[s - 1], I0s[s - 1].size());
160 gpu::resize(u2s[s], u2s[s - 1], I0s[s - 1].size());
161
162 // scale the optical flow with the appropriate zoom factor
163 gpu::multiply(u1s[s - 1], Scalar::all(2), u1s[s - 1]);
164 gpu::multiply(u2s[s - 1], Scalar::all(2), u2s[s - 1]);
165 }
166}

180void cv::gpu::OpticalFlowDual_TVL1_GPU::procOneScale(const GpuMat& I0, const GpuMat& I1, GpuMat& u1, GpuMat& u2)
181{
182 using namespace tvl1flow;
183
184 const double scaledEpsilon = epsilon * epsilon * I0.size().area();
185
186 CV_DbgAssert( I1.size() == I0.size() );
187 CV_DbgAssert( I1.type() == I0.type() );
188 CV_DbgAssert( u1.size() == I0.size() );
189 CV_DbgAssert( u2.size() == u1.size() );
190
191 GpuMat I1x = I1x_buf(Rect(0, 0, I0.cols, I0.rows));
192 GpuMat I1y = I1y_buf(Rect(0, 0, I0.cols, I0.rows));
193 centeredGradient(I1, I1x, I1y);
194
195 GpuMat I1w = I1w_buf(Rect(0, 0, I0.cols, I0.rows));
196 GpuMat I1wx = I1wx_buf(Rect(0, 0, I0.cols, I0.rows));
197 GpuMat I1wy = I1wy_buf(Rect(0, 0, I0.cols, I0.rows));
198
199 GpuMat grad = grad_buf(Rect(0, 0, I0.cols, I0.rows));
200 GpuMat rho_c = rho_c_buf(Rect(0, 0, I0.cols, I0.rows));
201
202 GpuMat p11 = p11_buf(Rect(0, 0, I0.cols, I0.rows));
203 GpuMat p12 = p12_buf(Rect(0, 0, I0.cols, I0.rows));
204 GpuMat p21 = p21_buf(Rect(0, 0, I0.cols, I0.rows));
205 GpuMat p22 = p22_buf(Rect(0, 0, I0.cols, I0.rows));
206 p11.setTo(Scalar::all(0));
207 p12.setTo(Scalar::all(0));
208 p21.setTo(Scalar::all(0));
209 p22.setTo(Scalar::all(0));
210
211 GpuMat diff = diff_buf(Rect(0, 0, I0.cols, I0.rows));
212
213 const float l_t = static_cast<float>(lambda * theta);
214 const float taut = static_cast<float>(tau / theta);
215
216 for (int warpings = 0; warpings < warps; ++warpings)
217 {
218 warpBackward(I0, I1, I1x, I1y, u1, u2, I1w, I1wx, I1wy, grad, rho_c);
219
220 double error = numeric_limits<double>::max();
221 double prevError = 0.0;
222 for (int n = 0; error > scaledEpsilon && n < iterations; ++n)
223 {
224 // some tweaks to make sum operation less frequently
225 bool calcError = (epsilon > 0) && (n & 0x1) && (prevError < scaledEpsilon);
226
227 estimateU(I1wx, I1wy, grad, rho_c, p11, p12, p21, p22, u1, u2, diff, l_t, static_cast<float>(theta), calcError);
228
229 if (calcError)
230 {
231 error = gpu::sum(diff, norm_buf)[0];
232 prevError = error;
233 }
234 else
235 {
236 error = numeric_limits<double>::max();
237 prevError -= scaledEpsilon;
238 }
239
240 estimateDualVariables(u1, u2, p11, p12, p21, p22, taut);
241 }
242 }

246{

248 I1s.clear();
249 u1s.clear();
250 u2s.clear();

253 I1y_buf.release();
254
255 I1w_buf.release();
256 I1wx_buf.release();
257 I1wy_buf.release();
258
259 grad_buf.release();
260 rho_c_buf.release();
261
262 p11_buf.release();
263 p12_buf.release();
264 p21_buf.release();
265 p22_buf.release();
266
267 diff_buf.release();
268 norm_buf.release();
269}
