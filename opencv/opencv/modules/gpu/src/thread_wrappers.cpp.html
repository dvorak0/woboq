83 class cv::gpu::detail::CriticalSection::Impl
84 {
85 public:
86 Impl()
87 {
88 pthread_mutexattr_t mutex_attribute;
89 pthread_mutexattr_init(&mutex_attribute);
90 pthread_mutexattr_settype(&mutex_attribute, PTHREAD_MUTEX_RECURSIVE);
91 pthread_mutex_init(&mutex_, 0);
92 pthread_mutexattr_destroy(&mutex_attribute);
93 }
94
95 ~Impl()
96 {
97 pthread_mutex_destroy(&mutex_);
98 }
99
100 void enter()
101 {
102 pthread_mutex_lock(&mutex_);
103 }
104
105 void leave()
106 {
107 pthread_mutex_unlock(&mutex_);
108 }
109
110 private:
111 pthread_mutex_t mutex_;
112 };
113#endif

153 class cv::gpu::detail::Thread::Impl
154 {
155 public:
156 Impl(void (*func)(void* userData), void* userData)
157 {
158 userData_.func = func;
159 userData_.param = userData;
160
161 thread_ = CreateThread(
162 NULL, // default security attributes
163 0, // use default stack size
164 WinThreadFunction, // thread function name
165 &userData_, // argument to thread function
166 0, // use default creation flags
167 &threadId_); // returns the thread identifier
168 }
169
170 ~Impl()
171 {
172 CloseHandle(thread_);
173 }
174
175 void wait()
176 {
177 WaitForSingleObject(thread_, INFINITE);
178 }
179
180 private:
181 UserData userData_;
182 HANDLE thread_;
183 DWORD threadId_;
184 };
185#else
186 namespace
187 {
188 struct UserData
189 {
190 void (*func)(void* userData);
191 void* param;
192 };
193
194 void* PThreadFunction(void* lpParam)
195 {
196 UserData* userData = static_cast<UserData*>(lpParam);
197
198 userData->func(userData->param);
199
200 return 0;
201 }
202 }

204 class cv::gpu::detail::Thread::Impl
205 {
206 public:
207 Impl(void (*func)(void* userData), void* userData)
208 {
209 userData_.func = func;
210 userData_.param = userData;
211
212 pthread_create(&thread_, NULL, PThreadFunction, &userData_);
213 }
214
215 ~Impl()
216 {
217 pthread_detach(thread_);
218 }
219
220 void wait()
221 {
222 pthread_join(thread_, NULL);
223 }
224
225 private:
226 pthread_t thread_;
227 UserData userData_;
228 };
229#endif
