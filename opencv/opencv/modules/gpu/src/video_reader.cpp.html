100cv::gpu::VideoReader_GPU::Impl::Impl(const cv::Ptr<VideoSource>& source) :
101 videoSource_(source),
102 lock_(0)
103{
104 // init context
105 GpuMat temp(1, 1, CV_8UC1);
106 temp.release();
107
108 DeviceInfo devInfo;
109 CV_Assert( devInfo.supports(FEATURE_SET_COMPUTE_11) );
110
111 CUcontext ctx;
112 cuSafeCall( cuCtxGetCurrent(&ctx) );
113 cuSafeCall( cuvidCtxLockCreate(&lock_, ctx) );
114
115 frameQueue_.reset(new detail::FrameQueue);
116 videoDecoder_.reset(new detail::VideoDecoder(videoSource_->format(), lock_));
117 videoParser_.reset(new detail::VideoParser(videoDecoder_.get(), frameQueue_.get()));
118
119 videoSource_->setFrameQueue(frameQueue_.get());
120 videoSource_->setVideoParser(videoParser_.get());
121
122 videoSource_->start();
123}
124
125cv::gpu::VideoReader_GPU::Impl::~Impl()
126{
127 frameQueue_->endDecode();
128 videoSource_->stop();
129}

151 enum ColorSpace
152 {
153 ITU601 = 1,
154 ITU709 = 2
155 };
156
157 void setColorSpaceMatrix(ColorSpace CSC, float hueCSC[9], float hue)
158 {
159 float hueSin = std::sin(hue);
160 float hueCos = std::cos(hue);
161
162 if (CSC == ITU601)
163 {
164 //CCIR 601
165 hueCSC[0] = 1.1644f;
166 hueCSC[1] = hueSin * 1.5960f;
167 hueCSC[2] = hueCos * 1.5960f;
168 hueCSC[3] = 1.1644f;
169 hueCSC[4] = (hueCos * -0.3918f) - (hueSin * 0.8130f);
170 hueCSC[5] = (hueSin * 0.3918f) - (hueCos * 0.8130f);
171 hueCSC[6] = 1.1644f;
172 hueCSC[7] = hueCos * 2.0172f;
173 hueCSC[8] = hueSin * -2.0172f;
174 }
175 else if (CSC == ITU709)
176 {
177 //CCIR 709
178 hueCSC[0] = 1.0f;
179 hueCSC[1] = hueSin * 1.57480f;
180 hueCSC[2] = hueCos * 1.57480f;
181 hueCSC[3] = 1.0;
182 hueCSC[4] = (hueCos * -0.18732f) - (hueSin * 0.46812f);
183 hueCSC[5] = (hueSin * 0.18732f) - (hueCos * 0.46812f);
184 hueCSC[6] = 1.0f;
185 hueCSC[7] = hueCos * 1.85560f;
186 hueCSC[8] = hueSin * -1.85560f;
187 }
188 }

190 void cudaPostProcessFrame(const cv::gpu::GpuMat& decodedFrame, cv::gpu::GpuMat& interopFrame, int width, int height)
191 {
192 using namespace cv::gpu::device::video_decoding;
193
194 static bool updateCSC = true;
195 static float hueColorSpaceMat[9];
196
197 // Upload the Color Space Conversion Matrices
198 if (updateCSC)
199 {
200 const ColorSpace colorSpace = ITU601;
201 const float hue = 0.0f;
202
203 // CCIR 601/709
204 setColorSpaceMatrix(colorSpace, hueColorSpaceMat, hue);
205
206 updateCSC = false;
207 }
208
209 // Final Stage: NV12toARGB color space conversion
210
211 interopFrame.create(height, width, CV_8UC4);
212
213 loadHueCSC(hueColorSpaceMat);
214
215 NV12ToARGB_gpu(decodedFrame, interopFrame);
216 }

219bool cv::gpu::VideoReader_GPU::Impl::grab(GpuMat& frame)
220{
221 if (videoSource_->hasError() || videoParser_->hasError())
222 CV_Error(CV_StsUnsupportedFormat, "Unsupported video source");
223
224 if (!videoSource_->isStarted() || frameQueue_->isEndOfDecode())
225 return false;
226
227 if (frames_.empty())
228 {
229 CUVIDPARSERDISPINFO displayInfo;
230
231 for (;;)
232 {
233 if (frameQueue_->dequeue(displayInfo))
234 break;
235
236 if (videoSource_->hasError() || videoParser_->hasError())
237 CV_Error(CV_StsUnsupportedFormat, "Unsupported video source");
238
239 if (frameQueue_->isEndOfDecode())
240 return false;
241
242 // Wait a bit
243 detail::Thread::sleep(1);
244 }
245
246 bool isProgressive = displayInfo.progressive_frame != 0;
247 const int num_fields = isProgressive ? 1 : 2 + displayInfo.repeat_first_field;
248
249 for (int active_field = 0; active_field < num_fields; ++active_field)
250 {
251 CUVIDPROCPARAMS videoProcParams;
252 std::memset(&videoProcParams, 0, sizeof(CUVIDPROCPARAMS));
253
254 videoProcParams.progressive_frame = displayInfo.progressive_frame;
255 videoProcParams.second_field = active_field;
256 videoProcParams.top_field_first = displayInfo.top_field_first;
257 videoProcParams.unpaired_field = (num_fields == 1);
258
259 frames_.push_back(std::make_pair(displayInfo, videoProcParams));
260 }
261 }
262
263 if (frames_.empty())
264 return false;
265
266 std::pair<CUVIDPARSERDISPINFO, CUVIDPROCPARAMS> frameInfo = frames_.front();
267 frames_.pop_front();
268
269 {
270 VideoCtxAutoLock autoLock(lock_);
271
272 // map decoded video frame to CUDA surface
273 cv::gpu::GpuMat decodedFrame = videoDecoder_->mapFrame(frameInfo.first.picture_index, frameInfo.second);
274
275 // perform post processing on the CUDA surface (performs colors space conversion and post processing)
276 // comment this out if we inclue the line of code seen above
277 cudaPostProcessFrame(decodedFrame, frame, videoDecoder_->targetWidth(), videoDecoder_->targetHeight());
278
279 // unmap video frame
280 // unmapFrame() synchronizes with the VideoDecode API (ensures the frame has finished decoding)
281 videoDecoder_->unmapFrame(decodedFrame);
282 }
283
284 // release the frame, so it can be re-used in decoder
285 if (frames_.empty())
286 frameQueue_->releaseFrame(frameInfo.first);
287
288 return true;
289}

312void cv::gpu::VideoReader_GPU::open(const std::string& filename)
313{
314 CV_Assert( !filename.empty() );
315
316#ifndef __APPLE__
317 try
318 {
319 cv::Ptr<VideoSource> source(new detail::CuvidVideoSource(filename));
320 open(source);
321 }
322 catch (const std::runtime_error&)
323#endif
324 {
325 cv::Ptr<VideoSource> source(new cv::gpu::detail::FFmpegVideoSource(filename));
326 open(source);
327 }
328}

330void cv::gpu::VideoReader_GPU::open(const cv::Ptr<VideoSource>& source)
331{
332 CV_Assert( !source.empty() );
333 close();
334 impl_.reset(new Impl(source));
335}
336
337bool cv::gpu::VideoReader_GPU::isOpened() const
338{
339 return impl_.get() != 0;
340}
341
342void cv::gpu::VideoReader_GPU::close()
343{
344 impl_.reset();
345}
346
347bool cv::gpu::VideoReader_GPU::read(GpuMat& image)
348{
349 if (!isOpened())
350 return false;
351
352 if (!impl_->grab(image))
353 {
354 close();
355 return false;
356 }
357
358 return true;
359}

372void cv::gpu::VideoReader_GPU::dumpFormat(std::ostream& st)
373{
374 static const char* codecs[] =
375 {
376 "MPEG1",
377 "MPEG2",
378 "MPEG4",
379 "VC1",
380 "H264",
381 "JPEG",
382 "H264_SVC",
383 "H264_MVC"
384 };
385
386 static const char* chromas[] =
387 {
388 "Monochrome",
389 "YUV420",
390 "YUV422",
391 "YUV444"
392 };
393
394 FormatInfo _format = this->format();
395
396 st << "Frame Size : " << _format.width << "x" << _format.height << std::endl;
397 st << "Codec : " << (_format.codec <= H264_MVC ? codecs[_format.codec] : "Uncompressed YUV") << std::endl;
398 st << "Chroma Format : " << chromas[_format.chromaFormat] << std::endl;
399}
