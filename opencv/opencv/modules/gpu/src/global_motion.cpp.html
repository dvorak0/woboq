67void cv::gpu::compactPoints(GpuMat &points0, GpuMat &points1, const GpuMat &mask)
68{
69 CV_Assert(points0.rows == 1 && points1.rows == 1 && mask.rows == 1);
70 CV_Assert(points0.type() == CV_32FC2 && points1.type() == CV_32FC2 && mask.type() == CV_8U);
71 CV_Assert(points0.cols == mask.cols && points1.cols == mask.cols);
72
73 int npoints = points0.cols;
74 int remaining = cv::gpu::device::globmotion::compactPoints(
75 npoints, (float*)points0.data, (float*)points1.data, mask.data);
76
77 points0 = points0.colRange(0, remaining);
78 points1 = points1.colRange(0, remaining);
79}

82void cv::gpu::calcWobbleSuppressionMaps(
83 int left, int idx, int right, Size size, const Mat &ml, const Mat &mr,
84 GpuMat &mapx, GpuMat &mapy)
85{
86 CV_Assert(ml.size() == Size(3, 3) && ml.type() == CV_32F && ml.isContinuous());
87 CV_Assert(mr.size() == Size(3, 3) && mr.type() == CV_32F && mr.isContinuous());
88
89 mapx.create(size, CV_32F);
90 mapy.create(size, CV_32F);
91
92 cv::gpu::device::globmotion::calcWobbleSuppressionMaps(
93 left, idx, right, size.width, size.height,
94 ml.ptr<float>(), mr.ptr<float>(), mapx, mapy);
95}
