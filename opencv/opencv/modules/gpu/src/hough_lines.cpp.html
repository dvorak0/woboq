80void cv::gpu::HoughLines(const GpuMat& src, GpuMat& lines, HoughLinesBuf& buf, float rho, float theta, int threshold, bool doSort, int maxLines)
81{
82 using namespace cv::gpu::device::hough;
83
84 CV_Assert(src.type() == CV_8UC1);
85 CV_Assert(src.cols < std::numeric_limits<unsigned short>::max());
86 CV_Assert(src.rows < std::numeric_limits<unsigned short>::max());
87
88 ensureSizeIsEnough(1, src.size().area(), CV_32SC1, buf.list);
89 unsigned int* srcPoints = buf.list.ptr<unsigned int>();
90
91 const int pointsCount = buildPointList_gpu(src, srcPoints);
92 if (pointsCount == 0)
93 {
94 lines.release();
95 return;
96 }
97
98 const int numangle = cvRound(CV_PI / theta);
99 const int numrho = cvRound(((src.cols + src.rows) * 2 + 1) / rho);
100 CV_Assert(numangle > 0 && numrho > 0);
101
102 ensureSizeIsEnough(numangle + 2, numrho + 2, CV_32SC1, buf.accum);
103 buf.accum.setTo(Scalar::all(0));
104
105 DeviceInfo devInfo;
106 linesAccum_gpu(srcPoints, pointsCount, buf.accum, rho, theta, devInfo.sharedMemPerBlock(), devInfo.supports(FEATURE_SET_COMPUTE_20));
107
108 ensureSizeIsEnough(2, maxLines, CV_32FC2, lines);
109
110 int linesCount = linesGetResult_gpu(buf.accum, lines.ptr<float2>(0), lines.ptr<int>(1), maxLines, rho, theta, threshold, doSort);
111 if (linesCount > 0)
112 lines.cols = linesCount;
113 else
114 lines.release();
115}

117void cv::gpu::HoughLinesDownload(const GpuMat& d_lines, OutputArray h_lines_, OutputArray h_votes_)
118{
119 if (d_lines.empty())
120 {
121 h_lines_.release();
122 if (h_votes_.needed())
123 h_votes_.release();
124 return;
125 }
126
127 CV_Assert(d_lines.rows == 2 && d_lines.type() == CV_32FC2);
128
129 h_lines_.create(1, d_lines.cols, CV_32FC2);
130 Mat h_lines = h_lines_.getMat();
131 d_lines.row(0).download(h_lines);
132
133 if (h_votes_.needed())
134 {
135 h_votes_.create(1, d_lines.cols, CV_32SC1);
136 Mat h_votes = h_votes_.getMat();
137 GpuMat d_votes(1, d_lines.cols, CV_32SC1, const_cast<int*>(d_lines.ptr<int>(1)));
138 d_votes.download(h_votes);
139 }
