66cv::gpu::GMG_GPU::GMG_GPU()
67{
68 maxFeatures = 64;
69 learningRate = 0.025f;
70 numInitializationFrames = 120;
71 quantizationLevels = 16;
72 backgroundPrior = 0.8f;
73 decisionThreshold = 0.8f;
74 smoothingRadius = 7;
75 updateBackgroundModel = true;
76}

78void cv::gpu::GMG_GPU::initialize(cv::Size frameSize, float min, float max)
79{
80 using namespace cv::gpu::device::bgfg_gmg;
81
82 CV_Assert(min < max);
83 CV_Assert(maxFeatures > 0);
84 CV_Assert(learningRate >= 0.0f && learningRate <= 1.0f);
85 CV_Assert(numInitializationFrames >= 1);
86 CV_Assert(quantizationLevels >= 1 && quantizationLevels <= 255);
87 CV_Assert(backgroundPrior >= 0.0f && backgroundPrior <= 1.0f);
88
89 minVal_ = min;
90 maxVal_ = max;
91
92 frameSize_ = frameSize;
93
94 frameNum_ = 0;
95
96 nfeatures_.create(frameSize_, CV_32SC1);
97 colors_.create(maxFeatures * frameSize_.height, frameSize_.width, CV_32SC1);
98 weights_.create(maxFeatures * frameSize_.height, frameSize_.width, CV_32FC1);
99
100 nfeatures_.setTo(cv::Scalar::all(0));
101
102 if (smoothingRadius > 0)
103 boxFilter_ = cv::gpu::createBoxFilter_GPU(CV_8UC1, CV_8UC1, cv::Size(smoothingRadius, smoothingRadius));
104
105 loadConstants(frameSize_.width, frameSize_.height, minVal_, maxVal_, quantizationLevels, backgroundPrior, decisionThreshold, maxFeatures, numInitializationFrames);
106}

108void cv::gpu::GMG_GPU::operator ()(const cv::gpu::GpuMat& frame, cv::gpu::GpuMat& fgmask, float newLearningRate, cv::gpu::Stream& stream)
109{
110 using namespace cv::gpu::device::bgfg_gmg;
111
112 typedef void (*func_t)(PtrStepSzb frame, PtrStepb fgmask, PtrStepSzi colors, PtrStepf weights, PtrStepi nfeatures,
113 int frameNum, float learningRate, bool updateBackgroundModel, cudaStream_t stream);
114 static const func_t funcs[6][4] =
115 {
116 {update_gpu<uchar>, 0, update_gpu<uchar3>, update_gpu<uchar4>},
117 {0,0,0,0},
118 {update_gpu<ushort>, 0, update_gpu<ushort3>, update_gpu<ushort4>},
119 {0,0,0,0},
120 {0,0,0,0},
121 {update_gpu<float>, 0, update_gpu<float3>, update_gpu<float4>}
122 };
123
124 CV_Assert(frame.depth() == CV_8U || frame.depth() == CV_16U || frame.depth() == CV_32F);
125 CV_Assert(frame.channels() == 1 || frame.channels() == 3 || frame.channels() == 4);
126
127 if (newLearningRate != -1.0f)
128 {
129 CV_Assert(newLearningRate >= 0.0f && newLearningRate <= 1.0f);
130 learningRate = newLearningRate;
131 }
132
133 if (frame.size() != frameSize_)
134 initialize(frame.size(), 0.0f, frame.depth() == CV_8U ? 255.0f : frame.depth() == CV_16U ? std::numeric_limits<ushort>::max() : 1.0f);
135
136 fgmask.create(frameSize_, CV_8UC1);
137 if (stream)
138 stream.enqueueMemSet(fgmask, cv::Scalar::all(0));
139 else
140 fgmask.setTo(cv::Scalar::all(0));
141
142 funcs[frame.depth()][frame.channels() - 1](frame, fgmask, colors_, weights_, nfeatures_, frameNum_, learningRate, updateBackgroundModel, cv::gpu::StreamAccessor::getStream(stream));
143
144 // medianBlur
145 if (smoothingRadius > 0)
146 {
147 boxFilter_->apply(fgmask, buf_, cv::Rect(0,0,-1,-1), stream);
148 int minCount = (smoothingRadius * smoothingRadius + 1) / 2;
149 double thresh = 255.0 * minCount / (smoothingRadius * smoothingRadius);
150 cv::gpu::threshold(buf_, fgmask, thresh, 255.0, cv::THRESH_BINARY, stream);
151 }
152
153 // keep track of how many frames we have processed
154 ++frameNum_;
155}
156
157void cv::gpu::GMG_GPU::release()
158{
159 frameSize_ = Size();
160
161 nfeatures_.release();
162 colors_.release();
163 weights_.release();
164 boxFilter_.release();
165 buf_.release();
166}
