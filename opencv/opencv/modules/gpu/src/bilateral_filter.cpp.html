73 const float DEFAULT_EDGE_THRESHOLD = 0.1f;
74 const float DEFAULT_MAX_DISC_THRESHOLD = 0.2f;
75 const float DEFAULT_SIGMA_RANGE = 10.0f;
76
77 inline void calc_color_weighted_table(GpuMat& table_color, float sigma_range, int len)
78 {
79 Mat cpu_table_color(1, len, CV_32F);
80
81 float* line = cpu_table_color.ptr<float>();
82
83 for(int i = 0; i < len; i++)
84 line[i] = static_cast<float>(std::exp(-double(i * i) / (2 * sigma_range * sigma_range)));
85
86 table_color.upload(cpu_table_color);
87 }
88
89 inline void calc_space_weighted_filter(GpuMat& table_space, int win_size, float dist_space)
90 {
91 int half = (win_size >> 1);
92
93 Mat cpu_table_space(half + 1, half + 1, CV_32F);
94
95 for (int y = 0; y <= half; ++y)
96 {
97 float* row = cpu_table_space.ptr<float>(y);
98 for (int x = 0; x <= half; ++x)
99 row[x] = exp(-sqrt(float(y * y) + float(x * x)) / dist_space);
100 }
101
102 table_space.upload(cpu_table_space);
103 }

105 template <typename T>
106 void disp_bilateral_filter_operator(int ndisp, int radius, int iters, float edge_threshold,float max_disc_threshold,
107 GpuMat& table_color, GpuMat& table_space,
108 const GpuMat& disp, const GpuMat& img, GpuMat& dst, Stream& stream)
109 {
110 short edge_disc = max<short>(short(1), short(ndisp * edge_threshold + 0.5));
111 short max_disc = short(ndisp * max_disc_threshold + 0.5);
112
113 disp_load_constants(table_color.ptr<float>(), table_space, ndisp, radius, edge_disc, max_disc);
114
115 if (&dst != &disp)
116 {
117 if (stream)
118 stream.enqueueCopy(disp, dst);
119 else
120 disp.copyTo(dst);
121 }
122
123 disp_bilateral_filter<T>(dst, img, img.channels(), iters, StreamAccessor::getStream(stream));
124 }
125
126 typedef void (*bilateral_filter_operator_t)(int ndisp, int radius, int iters, float edge_threshold, float max_disc_threshold,
127 GpuMat& table_color, GpuMat& table_space,
128 const GpuMat& disp, const GpuMat& img, GpuMat& dst, Stream& stream);
129
130 const bilateral_filter_operator_t operators[] =
131 {disp_bilateral_filter_operator<unsigned char>, 0, 0, disp_bilateral_filter_operator<short>, 0, 0, 0, 0};
132}

134cv::gpu::DisparityBilateralFilter::DisparityBilateralFilter(int ndisp_, int radius_, int iters_)
135 : ndisp(ndisp_), radius(radius_), iters(iters_), edge_threshold(DEFAULT_EDGE_THRESHOLD), max_disc_threshold(DEFAULT_MAX_DISC_THRESHOLD),
136 sigma_range(DEFAULT_SIGMA_RANGE)
137{
138 calc_color_weighted_table(table_color, sigma_range, 255);
139 calc_space_weighted_filter(table_space, radius * 2 + 1, radius + 1.0f);
140}
141
142cv::gpu::DisparityBilateralFilter::DisparityBilateralFilter(int ndisp_, int radius_, int iters_, float edge_threshold_,
143 float max_disc_threshold_, float sigma_range_)
144 : ndisp(ndisp_), radius(radius_), iters(iters_), edge_threshold(edge_threshold_), max_disc_threshold(max_disc_threshold_),
145 sigma_range(sigma_range_)
146{
147 calc_color_weighted_table(table_color, sigma_range, 255);
148 calc_space_weighted_filter(table_space, radius * 2 + 1, radius + 1.0f);
149}

151void cv::gpu::DisparityBilateralFilter::operator()(const GpuMat& disp, const GpuMat& img, GpuMat& dst, Stream& stream)
152{
153 CV_DbgAssert(0 < ndisp && 0 < radius && 0 < iters);
154 CV_Assert(disp.rows == img.rows && disp.cols == img.cols && (disp.type() == CV_8U || disp.type() == CV_16S) && (img.type() == CV_8UC1 || img.type() == CV_8UC3));
155 operators[disp.type()](ndisp, radius, iters, edge_threshold, max_disc_threshold, table_color, table_space, disp, img, dst, stream);
156}
