193 void GHT_Pos::setTemplateImpl(const GpuMat& edges, const GpuMat& dx, const GpuMat& dy, Point templCenter_)
194 {
195 templSize = edges.size();
196 templCenter = templCenter_;

199 ensureSizeIsEnough(templSize, dx.type(), templDx);

203 dx.copyTo(templDx);
204 dy.copyTo(templDy);

206 processTempl();

209 void GHT_Pos::detectImpl(const GpuMat& edges, const GpuMat& dx, const GpuMat& dy, GpuMat& positions)
210 {
211 imageSize = edges.size();
212
213 ensureSizeIsEnough(imageSize, edges.type(), imageEdges);
214 ensureSizeIsEnough(imageSize, dx.type(), imageDx);
215 ensureSizeIsEnough(imageSize, dy.type(), imageDy);
216
217 edges.copyTo(imageEdges);
218 dx.copyTo(imageDx);
219 dy.copyTo(imageDy);
220
221 posCount = 0;
222
223 processImage();
224
225 if (posCount == 0)
226 positions.release();
227 else
228 {
229 if (minDist > 1)
230 filterMinDist();
231 convertTo(positions);
232 }
233 }

260 void GHT_Pos::buildEdgePointList(const GpuMat& edges, const GpuMat& dx, const GpuMat& dy)
261 {
262 using namespace cv::gpu::device::hough;
263
264 typedef int (*func_t)(PtrStepSzb edges, PtrStepSzb dx, PtrStepSzb dy, unsigned int* coordList, float* thetaList);
265 static const func_t funcs[] =
266 {
267 0,
268 0,
269 0,
270 buildEdgePointList_gpu<short>,
271 buildEdgePointList_gpu<int>,
272 buildEdgePointList_gpu<float>,
273 0
274 };
275
276 CV_Assert(edges.type() == CV_8UC1);
277 CV_Assert(dx.size() == edges.size());
278 CV_Assert(dy.type() == dx.type() && dy.size() == edges.size());
279
280 const func_t func = funcs[dx.depth()];
281 CV_Assert(func != 0);
282
283 edgePointList.cols = (int) (edgePointList.step / sizeof(int));
284 ensureSizeIsEnough(2, edges.size().area(), CV_32SC1, edgePointList);
285
286 edgePointList.cols = func(edges, dx, dy, edgePointList.ptr<unsigned int>(0), edgePointList.ptr<float>(1));
287 }
288
289 #define votes_cmp_gt(l1, l2) (aux[l1].x > aux[l2].x)
290 static CV_IMPLEMENT_QSORT_EX( sortIndexies, int, votes_cmp_gt, const int3* )
291
292 void GHT_Pos::filterMinDist()
293 {
294 oldPosBuf.resize(posCount);
295 oldVoteBuf.resize(posCount);
296
297 cudaSafeCall( cudaMemcpy(&oldPosBuf[0], outBuf.ptr(0), posCount * sizeof(float4), cudaMemcpyDeviceToHost) );
298 cudaSafeCall( cudaMemcpy(&oldVoteBuf[0], outBuf.ptr(1), posCount * sizeof(int3), cudaMemcpyDeviceToHost) );
299
300 indexies.resize(posCount);
301 for (int i = 0; i < posCount; ++i)
302 indexies[i] = i;
303 sortIndexies(&indexies[0], posCount, &oldVoteBuf[0]);
304
305 newPosBuf.clear();
306 newVoteBuf.clear();
307 newPosBuf.reserve(posCount);
308 newVoteBuf.reserve(posCount);
309
310 const int cellSize = cvRound(minDist);
311 const int gridWidth = (imageSize.width + cellSize - 1) / cellSize;
312 const int gridHeight = (imageSize.height + cellSize - 1) / cellSize;
313
314 vector< vector<Point2f> > grid(gridWidth * gridHeight);
315
316 const double minDist2 = minDist * minDist;
317
318 for (int i = 0; i < posCount; ++i)
319 {
320 const int ind = indexies[i];
321
322 Point2f p(oldPosBuf[ind].x, oldPosBuf[ind].y);
323
324 bool good = true;
325
326 const int xCell = static_cast<int>(p.x / cellSize);
327 const int yCell = static_cast<int>(p.y / cellSize);
328
329 int x1 = xCell - 1;
330 int y1 = yCell - 1;
331 int x2 = xCell + 1;
332 int y2 = yCell + 1;
333
334 // boundary check
335 x1 = std::max(0, x1);
336 y1 = std::max(0, y1);
337 x2 = std::min(gridWidth - 1, x2);
338 y2 = std::min(gridHeight - 1, y2);
339
340 for (int yy = y1; yy <= y2; ++yy)
341 {
342 for (int xx = x1; xx <= x2; ++xx)
343 {
344 const vector<Point2f>& m = grid[yy * gridWidth + xx];
345
346 for(size_t j = 0; j < m.size(); ++j)
347 {
348 const Point2f d = p - m[j];
349
350 if (d.ddot(d) < minDist2)
351 {
352 good = false;
353 goto break_out;
354 }
355 }
356 }
357 }
358
359 break_out:
360
361 if(good)
362 {
363 grid[yCell * gridWidth + xCell].push_back(p);
364
365 newPosBuf.push_back(oldPosBuf[ind]);
366 newVoteBuf.push_back(oldVoteBuf[ind]);
367 }
368 }
369
370 posCount = static_cast<int>(newPosBuf.size());
371 cudaSafeCall( cudaMemcpy(outBuf.ptr(0), &newPosBuf[0], posCount * sizeof(float4), cudaMemcpyHostToDevice) );
372 cudaSafeCall( cudaMemcpy(outBuf.ptr(1), &newVoteBuf[0], posCount * sizeof(int3), cudaMemcpyHostToDevice) );
373 }

376 {
377 ensureSizeIsEnough(2, posCount, CV_32FC4, positions);
378 GpuMat(2, posCount, CV_32FC4, outBuf.data, outBuf.step).copyTo(positions);
379 }

439 void GHT_Ballard_Pos::processTempl()
440 {
441 using namespace cv::gpu::device::hough;
442
443 CV_Assert(levels > 0);
444
445 buildEdgePointList(templEdges, templDx, templDy);
446
447 ensureSizeIsEnough(levels + 1, maxSize, CV_16SC2, r_table);
448 ensureSizeIsEnough(1, levels + 1, CV_32SC1, r_sizes);
449 r_sizes.setTo(Scalar::all(0));
450
451 if (edgePointList.cols > 0)
452 {
453 buildRTable_gpu(edgePointList.ptr<unsigned int>(0), edgePointList.ptr<float>(1), edgePointList.cols,
454 r_table, r_sizes.ptr<int>(), make_short2(templCenter.x, templCenter.y), levels);
455 min(r_sizes, maxSize, r_sizes);
456 }
457 }

459 void GHT_Ballard_Pos::processImage()
460 {
461 calcHist();
462 findPosInHist();
463 }

465 void GHT_Ballard_Pos::calcHist()
466 {
467 using namespace cv::gpu::device::hough;
468
469 CV_Assert(levels > 0 && r_table.rows == (levels + 1) && r_sizes.cols == (levels + 1));
470 CV_Assert(dp > 0.0);
471
472 const double idp = 1.0 / dp;
473
474 buildEdgePointList(imageEdges, imageDx, imageDy);
475
476 ensureSizeIsEnough(cvCeil(imageSize.height * idp) + 2, cvCeil(imageSize.width * idp) + 2, CV_32SC1, hist);
477 hist.setTo(Scalar::all(0));
478
479 if (edgePointList.cols > 0)
480 {
481 GHT_Ballard_Pos_calcHist_gpu(edgePointList.ptr<unsigned int>(0), edgePointList.ptr<float>(1), edgePointList.cols,
482 r_table, r_sizes.ptr<int>(),
483 hist,
484 (float)dp, levels);
485 }
486 }
487
488 void GHT_Ballard_Pos::findPosInHist()
489 {
490 using namespace cv::gpu::device::hough;
491
492 CV_Assert(votesThreshold > 0);
493
494 ensureSizeIsEnough(2, maxSize, CV_32FC4, outBuf);
495
496 posCount = GHT_Ballard_Pos_findPosInHist_gpu(hist, outBuf.ptr<float4>(0), outBuf.ptr<int3>(1), maxSize, (float)dp, votesThreshold);
497 }

633 using namespace cv::gpu::device::hough;

635 CV_Assert(levels > 0 && r_table.rows == (levels + 1) && r_sizes.cols == (levels + 1));

637 CV_Assert(minAngle >= 0.0 && minAngle < maxAngle && maxAngle <= 360.0);

640 const double idp = 1.0 / dp;

647 ensureSizeIsEnough((angleRange + 2) * (rows + 2), cols + 2, CV_32SC1, hist);
648 hist.setTo(Scalar::all(0));
649

651 {
652 GHT_Ballard_PosRotation_calcHist_gpu(edgePointList.ptr<unsigned int>(0), edgePointList.ptr<float>(1), edgePointList.cols,
653 r_table, r_sizes.ptr<int>(),
654 hist, rows, cols,
655 (float)minAngle, (float)angleStep, angleRange, (float)dp, levels);
656 }

667 const int rows = cvCeil(imageSize.height * idp);
668 const int cols = cvCeil(imageSize.width * idp);
669
670 ensureSizeIsEnough(2, maxSize, CV_32FC4, outBuf);
671
672 posCount = GHT_Ballard_PosRotation_findPosInHist_gpu(hist, rows, cols, angleRange, outBuf.ptr<float4>(0), outBuf.ptr<int3>(1), maxSize, (float)minAngle, (float)angleStep, (float)dp, votesThreshold);
673 }

829 void GHT_Guil_Full::processImage()
830 {
831 using namespace cv::gpu::device::hough;
832
833 CV_Assert(levels > 0);
834 CV_Assert(templFeatures.sizes.cols == levels + 1);
835 CV_Assert(minAngle >= 0.0 && minAngle < maxAngle && maxAngle <= 360.0);
836 CV_Assert(angleStep > 0.0 && angleStep < 360.0);
837 CV_Assert(angleThresh > 0);
838 CV_Assert(minScale > 0.0 && minScale < maxScale);
839 CV_Assert(scaleStep > 0.0);
840 CV_Assert(scaleThresh > 0);
841 CV_Assert(dp > 0.0);
842 CV_Assert(posThresh > 0);
843
844 const double iAngleStep = 1.0 / angleStep;
845 const int angleRange = cvCeil((maxAngle - minAngle) * iAngleStep);
846
847 const double iScaleStep = 1.0 / scaleStep;
848 const int scaleRange = cvCeil((maxScale - minScale) * iScaleStep);
849
850 const double idp = 1.0 / dp;
851 const int histRows = cvCeil(imageSize.height * idp);
852 const int histCols = cvCeil(imageSize.width * idp);
853
854 ensureSizeIsEnough(histRows + 2, std::max(angleRange + 1, std::max(scaleRange + 1, histCols + 2)), CV_32SC1, hist);
855 h_buf.resize(std::max(angleRange + 1, scaleRange + 1));
856
857 ensureSizeIsEnough(2, maxSize, CV_32FC4, outBuf);
858
859 buildFeatureList(imageEdges, imageDx, imageDy, imageFeatures,
860 GHT_Guil_Full_setImageFeatures, GHT_Guil_Full_buildImageFeatureList_gpu,
861 false);
862
863 calcOrientation();
864
865 for (size_t i = 0; i < angles.size(); ++i)
866 {
867 const double angle = angles[i].first;
868 const int angleVotes = angles[i].second;
869
870 calcScale(angle);
871
872 for (size_t j = 0; j < scales.size(); ++j)
873 {
874 const double scale = scales[j].first;
875 const int scaleVotes = scales[j].second;
876
877 calcPosition(angle, angleVotes, scale, scaleVotes);
878 }
879 }

941 void GHT_Guil_Full::calcOrientation()
942 {
943 using namespace cv::gpu::device::hough;
944
945 const double iAngleStep = 1.0 / angleStep;
946 const int angleRange = cvCeil((maxAngle - minAngle) * iAngleStep);
947
948 hist.setTo(Scalar::all(0));
949 GHT_Guil_Full_calcOHist_gpu(templFeatures.sizes.ptr<int>(), imageFeatures.sizes.ptr<int>(0),
950 hist.ptr<int>(), (float)minAngle, (float)maxAngle, (float)angleStep, angleRange, levels, templFeatures.maxSize);
951 cudaSafeCall( cudaMemcpy(&h_buf[0], hist.data, h_buf.size() * sizeof(int), cudaMemcpyDeviceToHost) );
952
953 angles.clear();
954
955 for (int n = 0; n < angleRange; ++n)
956 {
957 if (h_buf[n] >= angleThresh)
958 {
959 const double angle = minAngle + n * angleStep;
960 angles.push_back(make_pair(angle, h_buf[n]));
961 }
962 }
963 }

965 void GHT_Guil_Full::calcScale(double angle)
966 {
967 using namespace cv::gpu::device::hough;
968
969 const double iScaleStep = 1.0 / scaleStep;
970 const int scaleRange = cvCeil((maxScale - minScale) * iScaleStep);
971
972 hist.setTo(Scalar::all(0));
973 GHT_Guil_Full_calcSHist_gpu(templFeatures.sizes.ptr<int>(), imageFeatures.sizes.ptr<int>(0),
974 hist.ptr<int>(), (float)angle, (float)angleEpsilon, (float)minScale, (float)maxScale, (float)iScaleStep, scaleRange, levels, templFeatures.maxSize);
975 cudaSafeCall( cudaMemcpy(&h_buf[0], hist.data, h_buf.size() * sizeof(int), cudaMemcpyDeviceToHost) );
976
977 scales.clear();
978
979 for (int s = 0; s < scaleRange; ++s)
980 {
981 if (h_buf[s] >= scaleThresh)
982 {
983 const double scale = minScale + s * scaleStep;
984 scales.push_back(make_pair(scale, h_buf[s]));
985 }
986 }

1003{
1004 switch (method)
1005 {
1006 case GHT_POSITION:
1007 CV_Assert( !GHT_Ballard_Pos_info_auto.name().empty() );
1008 return new GHT_Ballard_Pos();
1009
1010 case (GHT_POSITION | GHT_SCALE):
1011 CV_Assert( !GHT_Ballard_PosScale_info_auto.name().empty() );
1012 return new GHT_Ballard_PosScale();
1013
1014 case (GHT_POSITION | GHT_ROTATION):
1015 CV_Assert( !GHT_Ballard_PosRotation_info_auto.name().empty() );
1016 return new GHT_Ballard_PosRotation();
1017
1018 case (GHT_POSITION | GHT_SCALE | GHT_ROTATION):
1019 CV_Assert( !GHT_Guil_Full_info_auto.name().empty() );
1020 return new GHT_Guil_Full();
1021 }
1022
1023 CV_Error(CV_StsBadArg, "Unsupported method");
1024 return Ptr<GeneralizedHough_GPU>();
