49 #define error_entry(entry) { entry, #entry }
50
51 struct ErrorEntry
52 {
53 int code;
54 std::string str;
55 };
56
57 class ErrorEntryComparer
58 {
59 public:
60 inline ErrorEntryComparer(int code) : code_(code) {}
61
62 inline bool operator()(const ErrorEntry& e) const { return e.code == code_; }
63
64 private:
65 int code_;
66 };
67
68 std::string getErrorString(int code, const ErrorEntry* errors, size_t n)
69 {
70 size_t idx = std::find_if(errors, errors + n, ErrorEntryComparer(code)) - errors;
71
72 const std::string& msg = (idx != n) ? errors[idx].str : std::string("Unknown error code");
73
74 std::ostringstream ostr;
75 ostr << msg << " [Code = " << code << "]";
76
77 return ostr.str();
78 }
