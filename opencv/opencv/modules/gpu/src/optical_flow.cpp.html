60 NCVMatrix<Ncv32f>& u, NCVMatrix<Ncv32f>& v, const cudaDeviceProp& devProp)

75void cv::gpu::BroxOpticalFlow::operator ()(const GpuMat& frame0, const GpuMat& frame1, GpuMat& u, GpuMat& v, Stream& s)
76{
77 ncvSetDebugOutputHandler(outputHandler);
78
79 CV_Assert(frame0.type() == CV_32FC1);
80 CV_Assert(frame1.size() == frame0.size() && frame1.type() == frame0.type());
81
82 u.create(frame0.size(), CV_32FC1);
83 v.create(frame0.size(), CV_32FC1);
84
85 cudaDeviceProp devProp;
86 cudaSafeCall( cudaGetDeviceProperties(&devProp, getDevice()) );
87
88 NCVBroxOpticalFlowDescriptor desc;
89
90 desc.alpha = alpha;
91 desc.gamma = gamma;
92 desc.scale_factor = scale_factor;
93 desc.number_of_inner_iterations = inner_iterations;
94 desc.number_of_outer_iterations = outer_iterations;
95 desc.number_of_solver_iterations = solver_iterations;
96
97 NCVMemSegment frame0MemSeg;
98 frame0MemSeg.begin.memtype = NCVMemoryTypeDevice;
99 frame0MemSeg.begin.ptr = const_cast<uchar*>(frame0.data);
100 frame0MemSeg.size = frame0.step * frame0.rows;
101
102 NCVMemSegment frame1MemSeg;
103 frame1MemSeg.begin.memtype = NCVMemoryTypeDevice;
104 frame1MemSeg.begin.ptr = const_cast<uchar*>(frame1.data);
105 frame1MemSeg.size = frame1.step * frame1.rows;
106
107 NCVMemSegment uMemSeg;
108 uMemSeg.begin.memtype = NCVMemoryTypeDevice;
109 uMemSeg.begin.ptr = u.ptr();
110 uMemSeg.size = u.step * u.rows;
111
112 NCVMemSegment vMemSeg;
113 vMemSeg.begin.memtype = NCVMemoryTypeDevice;
114 vMemSeg.begin.ptr = v.ptr();
115 vMemSeg.size = v.step * v.rows;
116
117 NCVMatrixReuse<Ncv32f> frame0Mat(frame0MemSeg, static_cast<Ncv32u>(devProp.textureAlignment), frame0.cols, frame0.rows, static_cast<Ncv32u>(frame0.step));
118 NCVMatrixReuse<Ncv32f> frame1Mat(frame1MemSeg, static_cast<Ncv32u>(devProp.textureAlignment), frame1.cols, frame1.rows, static_cast<Ncv32u>(frame1.step));
119 NCVMatrixReuse<Ncv32f> uMat(uMemSeg, static_cast<Ncv32u>(devProp.textureAlignment), u.cols, u.rows, static_cast<Ncv32u>(u.step));
120 NCVMatrixReuse<Ncv32f> vMat(vMemSeg, static_cast<Ncv32u>(devProp.textureAlignment), v.cols, v.rows, static_cast<Ncv32u>(v.step));
121
122 cudaStream_t stream = StreamAccessor::getStream(s);
123
124 size_t bufSize = getBufSize(desc, frame0Mat, frame1Mat, uMat, vMat, devProp);
125
126 ensureSizeIsEnough(1, static_cast<int>(bufSize), CV_8UC1, buf);
127
128 NCVMemStackAllocator gpuAllocator(NCVMemoryTypeDevice, bufSize, static_cast<Ncv32u>(devProp.textureAlignment), buf.ptr());
129
130 ncvSafeCall( NCVBroxOpticalFlow(desc, gpuAllocator, frame0Mat, frame1Mat, uMat, vMat, stream) );
131}

133void cv::gpu::interpolateFrames(const GpuMat& frame0, const GpuMat& frame1, const GpuMat& fu, const GpuMat& fv, const GpuMat& bu, const GpuMat& bv,
134 float pos, GpuMat& newFrame, GpuMat& buf, Stream& s)
135{
136 CV_Assert(frame0.type() == CV_32FC1);
137 CV_Assert(frame1.size() == frame0.size() && frame1.type() == frame0.type());
138 CV_Assert(fu.size() == frame0.size() && fu.type() == frame0.type());
139 CV_Assert(fv.size() == frame0.size() && fv.type() == frame0.type());
140 CV_Assert(bu.size() == frame0.size() && bu.type() == frame0.type());
141 CV_Assert(bv.size() == frame0.size() && bv.type() == frame0.type());
142
143 newFrame.create(frame0.size(), frame0.type());
144
145 buf.create(6 * frame0.rows, frame0.cols, CV_32FC1);
146 buf.setTo(Scalar::all(0));
147
148 // occlusion masks
149 GpuMat occ0 = buf.rowRange(0 * frame0.rows, 1 * frame0.rows);
150 GpuMat occ1 = buf.rowRange(1 * frame0.rows, 2 * frame0.rows);
151
152 // interpolated forward flow
153 GpuMat fui = buf.rowRange(2 * frame0.rows, 3 * frame0.rows);
154 GpuMat fvi = buf.rowRange(3 * frame0.rows, 4 * frame0.rows);
155
156 // interpolated backward flow
157 GpuMat bui = buf.rowRange(4 * frame0.rows, 5 * frame0.rows);
158 GpuMat bvi = buf.rowRange(5 * frame0.rows, 6 * frame0.rows);
159
160 size_t step = frame0.step;
161
162 CV_Assert(frame1.step == step && fu.step == step && fv.step == step && bu.step == step && bv.step == step && newFrame.step == step && buf.step == step);
163
164 cudaStream_t stream = StreamAccessor::getStream(s);
165 NppStStreamHandler h(stream);
166
167 NppStInterpolationState state;
168
169 state.size = NcvSize32u(frame0.cols, frame0.rows);
170 state.nStep = static_cast<Ncv32u>(step);
171 state.pSrcFrame0 = const_cast<Ncv32f*>(frame0.ptr<Ncv32f>());
172 state.pSrcFrame1 = const_cast<Ncv32f*>(frame1.ptr<Ncv32f>());
173 state.pFU = const_cast<Ncv32f*>(fu.ptr<Ncv32f>());
174 state.pFV = const_cast<Ncv32f*>(fv.ptr<Ncv32f>());
175 state.pBU = const_cast<Ncv32f*>(bu.ptr<Ncv32f>());
176 state.pBV = const_cast<Ncv32f*>(bv.ptr<Ncv32f>());
177 state.pos = pos;
178 state.pNewFrame = newFrame.ptr<Ncv32f>();
179 state.ppBuffers[0] = occ0.ptr<Ncv32f>();
180 state.ppBuffers[1] = occ1.ptr<Ncv32f>();
181 state.ppBuffers[2] = fui.ptr<Ncv32f>();
182 state.ppBuffers[3] = fvi.ptr<Ncv32f>();
183 state.ppBuffers[4] = bui.ptr<Ncv32f>();
184 state.ppBuffers[5] = bvi.ptr<Ncv32f>();
185
186 ncvSafeCall( nppiStInterpolateFrames(&state) );
187
188 if (stream == 0)
189 cudaSafeCall( cudaDeviceSynchronize() );
190}

201void cv::gpu::createOpticalFlowNeedleMap(const GpuMat& u, const GpuMat& v, GpuMat& vertex, GpuMat& colors)
202{
203 using namespace cv::gpu::device::optical_flow;
204
205 CV_Assert(u.type() == CV_32FC1);
206 CV_Assert(v.type() == u.type() && v.size() == u.size());
207
208 const int NEEDLE_MAP_SCALE = 16;
209
210 const int x_needles = u.cols / NEEDLE_MAP_SCALE;
211 const int y_needles = u.rows / NEEDLE_MAP_SCALE;
212
213 GpuMat u_avg(y_needles, x_needles, CV_32FC1);
214 GpuMat v_avg(y_needles, x_needles, CV_32FC1);
215
216 NeedleMapAverage_gpu(u, v, u_avg, v_avg);
217
218 const int NUM_VERTS_PER_ARROW = 6;
219
220 const int num_arrows = x_needles * y_needles * NUM_VERTS_PER_ARROW;
221
222 vertex.create(1, num_arrows, CV_32FC3);
223 colors.create(1, num_arrows, CV_32FC3);
224
225 colors.setTo(Scalar::all(1.0));
226
227 double uMax, vMax;
228 minMax(u_avg, 0, &uMax);
229 minMax(v_avg, 0, &vMax);
230
231 float max_flow = static_cast<float>(std::sqrt(uMax * uMax + vMax * vMax));
232
233 CreateOpticalFlowNeedleMap_gpu(u_avg, v_avg, vertex.ptr<float>(), colors.ptr<float>(), max_flow, 1.0f / u.cols, 1.0f / u.rows);
234
235 cvtColor(colors, colors, COLOR_HSV2RGB);
236}
