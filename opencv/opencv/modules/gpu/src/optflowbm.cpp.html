63void cv::gpu::calcOpticalFlowBM(const GpuMat& prev, const GpuMat& curr, Size blockSize, Size shiftSize, Size maxRange, bool usePrevious, GpuMat& velx, GpuMat& vely, GpuMat& buf, Stream& st)
64{
65 CV_Assert( prev.type() == CV_8UC1 );
66 CV_Assert( curr.size() == prev.size() && curr.type() == prev.type() );
67
68 const Size velSize((prev.cols - blockSize.width + shiftSize.width) / shiftSize.width,
69 (prev.rows - blockSize.height + shiftSize.height) / shiftSize.height);
70
71 velx.create(velSize, CV_32FC1);
72 vely.create(velSize, CV_32FC1);
73
74 // scanning scheme coordinates
75 vector<short2> ss((2 * maxRange.width + 1) * (2 * maxRange.height + 1));
76 int ssCount = 0;
77
78 // Calculate scanning scheme
79 const int minCount = std::min(maxRange.width, maxRange.height);
80
81 // use spiral search pattern
82 //
83 // 9 10 11 12
84 // 8 1 2 13
85 // 7 * 3 14
86 // 6 5 4 15
87 //... 20 19 18 17
88 //
89
90 for (int i = 0; i < minCount; ++i)
91 {
92 // four cycles along sides
93 int x = -i - 1, y = x;
94
95 // upper side
96 for (int j = -i; j <= i + 1; ++j, ++ssCount)
97 {
98 ss[ssCount].x = ++x;
99 ss[ssCount].y = y;
100 }
101
102 // right side
103 for (int j = -i; j <= i + 1; ++j, ++ssCount)
104 {
105 ss[ssCount].x = x;
106 ss[ssCount].y = ++y;
107 }
108
109 // bottom side
110 for (int j = -i; j <= i + 1; ++j, ++ssCount)
111 {
112 ss[ssCount].x = --x;
113 ss[ssCount].y = y;
114 }
115
116 // left side
117 for (int j = -i; j <= i + 1; ++j, ++ssCount)
118 {
119 ss[ssCount].x = x;
120 ss[ssCount].y = --y;
121 }
122 }
123
124 // the rest part
125 if (maxRange.width < maxRange.height)
126 {
127 const int xleft = -minCount;
128
129 // cycle by neighbor rings
130 for (int i = minCount; i < maxRange.height; ++i)
131 {
132 // two cycles by x
133 int y = -(i + 1);
134 int x = xleft;
135
136 // upper side
137 for (int j = -maxRange.width; j <= maxRange.width; ++j, ++ssCount, ++x)
138 {
139 ss[ssCount].x = x;
140 ss[ssCount].y = y;
141 }
142
143 x = xleft;
144 y = -y;
145
146 // bottom side
147 for (int j = -maxRange.width; j <= maxRange.width; ++j, ++ssCount, ++x)
148 {
149 ss[ssCount].x = x;
150 ss[ssCount].y = y;
151 }
152 }
153 }
154 else if (maxRange.width > maxRange.height)
155 {
156 const int yupper = -minCount;
157
158 // cycle by neighbor rings
159 for (int i = minCount; i < maxRange.width; ++i)
160 {
161 // two cycles by y
162 int x = -(i + 1);
163 int y = yupper;
164
165 // left side
166 for (int j = -maxRange.height; j <= maxRange.height; ++j, ++ssCount, ++y)
167 {
168 ss[ssCount].x = x;
169 ss[ssCount].y = y;
170 }
171
172 y = yupper;
173 x = -x;
174
175 // right side
176 for (int j = -maxRange.height; j <= maxRange.height; ++j, ++ssCount, ++y)
177 {
178 ss[ssCount].x = x;
179 ss[ssCount].y = y;
180 }
181 }
182 }
183
184 const cudaStream_t stream = StreamAccessor::getStream(st);
185
186 ensureSizeIsEnough(1, ssCount, CV_16SC2, buf);
187 if (stream == 0)
188 cudaSafeCall( cudaMemcpy(buf.data, &ss[0], ssCount * sizeof(short2), cudaMemcpyHostToDevice) );
189 else
190 cudaSafeCall( cudaMemcpyAsync(buf.data, &ss[0], ssCount * sizeof(short2), cudaMemcpyHostToDevice, stream) );
191
192 const int maxX = prev.cols - blockSize.width;
193 const int maxY = prev.rows - blockSize.height;
194
195 const int SMALL_DIFF = 2;
196 const int BIG_DIFF = 128;
197
198 const int blSize = blockSize.area();
199 const int acceptLevel = blSize * SMALL_DIFF;
200 const int escapeLevel = blSize * BIG_DIFF;
201
202 optflowbm::calc(prev, curr, velx, vely,
203 make_int2(blockSize.width, blockSize.height), make_int2(shiftSize.width, shiftSize.height), usePrevious,
204 maxX, maxY, acceptLevel, escapeLevel, buf.ptr<short2>(), ssCount, stream);
205}

215void cv::gpu::FastOpticalFlowBM::operator ()(const GpuMat& I0, const GpuMat& I1, GpuMat& flowx, GpuMat& flowy, int search_window, int block_window, Stream& stream)
216{
217 CV_Assert( I0.type() == CV_8UC1 );
218 CV_Assert( I1.size() == I0.size() && I1.type() == I0.type() );
219
220 int border_size = search_window / 2 + block_window / 2;
221 Size esize = I0.size() + Size(border_size, border_size) * 2;
222
223 ensureSizeIsEnough(esize, I0.type(), extended_I0);
224 ensureSizeIsEnough(esize, I0.type(), extended_I1);
225
226 copyMakeBorder(I0, extended_I0, border_size, border_size, border_size, border_size, cv::BORDER_DEFAULT, Scalar(), stream);
227 copyMakeBorder(I1, extended_I1, border_size, border_size, border_size, border_size, cv::BORDER_DEFAULT, Scalar(), stream);
228
229 GpuMat I0_hdr = extended_I0(Rect(Point2i(border_size, border_size), I0.size()));
230 GpuMat I1_hdr = extended_I1(Rect(Point2i(border_size, border_size), I0.size()));
231
232 int bcols, brows;
233 optflowbm_fast::get_buffer_size(I0.cols, I0.rows, search_window, block_window, bcols, brows);
234
235 ensureSizeIsEnough(brows, bcols, CV_32SC1, buffer);
236
237 flowx.create(I0.size(), CV_32FC1);
238 flowy.create(I0.size(), CV_32FC1);
239
240 optflowbm_fast::calc<uchar>(I0_hdr, I1_hdr, flowx, flowy, buffer, search_window, block_window, StreamAccessor::getStream(stream));
241}
