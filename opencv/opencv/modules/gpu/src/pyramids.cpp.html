65void cv::gpu::pyrDown(const GpuMat& src, GpuMat& dst, Stream& stream)
66{
67 using namespace cv::gpu::device::imgproc;
68
69 typedef void (*func_t)(PtrStepSzb src, PtrStepSzb dst, cudaStream_t stream);
70
71#ifdef OPENCV_TINY_GPU_MODULE
72 static const func_t funcs[6][4] =
73 {
74 {pyrDown_gpu<uchar> , 0 /*pyrDown_gpu<uchar2>*/ , pyrDown_gpu<uchar3> , pyrDown_gpu<uchar4> },
75 {0, 0, 0, 0},
76 {0, 0, 0, 0},
77 {0, 0, 0, 0},
78 {0, 0, 0, 0},
79 {pyrDown_gpu<float> , 0 /*pyrDown_gpu<float2>*/ , pyrDown_gpu<float3> , pyrDown_gpu<float4> }
80 };
81#else
82 static const func_t funcs[6][4] =
83 {
84 {pyrDown_gpu<uchar> , 0 /*pyrDown_gpu<uchar2>*/ , pyrDown_gpu<uchar3> , pyrDown_gpu<uchar4> },
85 {0 /*pyrDown_gpu<schar>*/, 0 /*pyrDown_gpu<schar2>*/ , 0 /*pyrDown_gpu<schar3>*/, 0 /*pyrDown_gpu<schar4>*/},
86 {pyrDown_gpu<ushort> , 0 /*pyrDown_gpu<ushort2>*/, pyrDown_gpu<ushort3> , pyrDown_gpu<ushort4> },
87 {pyrDown_gpu<short> , 0 /*pyrDown_gpu<short2>*/ , pyrDown_gpu<short3> , pyrDown_gpu<short4> },
88 {0 /*pyrDown_gpu<int>*/ , 0 /*pyrDown_gpu<int2>*/ , 0 /*pyrDown_gpu<int3>*/ , 0 /*pyrDown_gpu<int4>*/ },
89 {pyrDown_gpu<float> , 0 /*pyrDown_gpu<float2>*/ , pyrDown_gpu<float3> , pyrDown_gpu<float4> }
90 };
91#endif
92
93 CV_Assert(src.depth() <= CV_32F && src.channels() <= 4);
94
95 const func_t func = funcs[src.depth()][src.channels() - 1];

166void cv::gpu::ImagePyramid::build(const GpuMat& img, int numLayers, Stream& stream)
167{
168 using namespace cv::gpu::device::pyramid;
169
170 typedef void (*func_t)(PtrStepSzb src, PtrStepSzb dst, cudaStream_t stream);
171
172 static const func_t funcs[6][4] =
173 {
174 {kernelDownsampleX2_gpu<uchar1> , 0 /*kernelDownsampleX2_gpu<uchar2>*/ , kernelDownsampleX2_gpu<uchar3> , kernelDownsampleX2_gpu<uchar4> },
175 {0 /*kernelDownsampleX2_gpu<char1>*/ , 0 /*kernelDownsampleX2_gpu<char2>*/ , 0 /*kernelDownsampleX2_gpu<char3>*/ , 0 /*kernelDownsampleX2_gpu<char4>*/ },
176 {kernelDownsampleX2_gpu<ushort1> , 0 /*kernelDownsampleX2_gpu<ushort2>*/, kernelDownsampleX2_gpu<ushort3> , kernelDownsampleX2_gpu<ushort4> },
177 {0 /*kernelDownsampleX2_gpu<short1>*/ , 0 /*kernelDownsampleX2_gpu<short2>*/ , 0 /*kernelDownsampleX2_gpu<short3>*/, 0 /*kernelDownsampleX2_gpu<short4>*/},
178 {0 /*kernelDownsampleX2_gpu<int1>*/ , 0 /*kernelDownsampleX2_gpu<int2>*/ , 0 /*kernelDownsampleX2_gpu<int3>*/ , 0 /*kernelDownsampleX2_gpu<int4>*/ },
179 {kernelDownsampleX2_gpu<float1> , 0 /*kernelDownsampleX2_gpu<float2>*/ , kernelDownsampleX2_gpu<float3> , kernelDownsampleX2_gpu<float4> }
180 };
181
182 CV_Assert(img.depth() <= CV_32F && img.channels() <= 4);
183
184 const func_t func = funcs[img.depth()][img.channels() - 1];
185 CV_Assert(func != 0);
186
187 layer0_ = img;
188 Size szLastLayer = img.size();
189 nLayers_ = 1;
190
191 if (numLayers <= 0)
192 numLayers = 255; //it will cut-off when any of the dimensions goes 1
193
194 pyramid_.resize(numLayers);
195
196 for (int i = 0; i < numLayers - 1; ++i)
197 {
198 Size szCurLayer(szLastLayer.width / 2, szLastLayer.height / 2);
199
200 if (szCurLayer.width == 0 || szCurLayer.height == 0)
201 break;
202
203 ensureSizeIsEnough(szCurLayer, img.type(), pyramid_[i]);
204 nLayers_++;
205
206 const GpuMat& prevLayer = i == 0 ? layer0_ : pyramid_[i - 1];
207
208 func(prevLayer, pyramid_[i], StreamAccessor::getStream(stream));
209
210 szLastLayer = szCurLayer;
211 }

214void cv::gpu::ImagePyramid::getLayer(GpuMat& outImg, Size outRoi, Stream& stream) const
215{
216 using namespace cv::gpu::device::pyramid;
217
218 typedef void (*func_t)(PtrStepSzb src, PtrStepSzb dst, cudaStream_t stream);
219
220 static const func_t funcs[6][4] =
221 {
222 {kernelInterpolateFrom1_gpu<uchar1> , 0 /*kernelInterpolateFrom1_gpu<uchar2>*/ , kernelInterpolateFrom1_gpu<uchar3> , kernelInterpolateFrom1_gpu<uchar4> },
223 {0 /*kernelInterpolateFrom1_gpu<char1>*/ , 0 /*kernelInterpolateFrom1_gpu<char2>*/ , 0 /*kernelInterpolateFrom1_gpu<char3>*/ , 0 /*kernelInterpolateFrom1_gpu<char4>*/ },
224 {kernelInterpolateFrom1_gpu<ushort1> , 0 /*kernelInterpolateFrom1_gpu<ushort2>*/, kernelInterpolateFrom1_gpu<ushort3> , kernelInterpolateFrom1_gpu<ushort4> },
225 {0 /*kernelInterpolateFrom1_gpu<short1>*/, 0 /*kernelInterpolateFrom1_gpu<short2>*/ , 0 /*kernelInterpolateFrom1_gpu<short3>*/, 0 /*kernelInterpolateFrom1_gpu<short4>*/},
226 {0 /*kernelInterpolateFrom1_gpu<int1>*/ , 0 /*kernelInterpolateFrom1_gpu<int2>*/ , 0 /*kernelInterpolateFrom1_gpu<int3>*/ , 0 /*kernelInterpolateFrom1_gpu<int4>*/ },
227 {kernelInterpolateFrom1_gpu<float1> , 0 /*kernelInterpolateFrom1_gpu<float2>*/ , kernelInterpolateFrom1_gpu<float3> , kernelInterpolateFrom1_gpu<float4> }
228 };
229
230 CV_Assert(outRoi.width <= layer0_.cols && outRoi.height <= layer0_.rows && outRoi.width > 0 && outRoi.height > 0);
231
232 ensureSizeIsEnough(outRoi, layer0_.type(), outImg);
233
234 const func_t func = funcs[outImg.depth()][outImg.channels() - 1];
235 CV_Assert(func != 0);
236
237 if (outRoi.width == layer0_.cols && outRoi.height == layer0_.rows)
238 {
239 if (stream)
240 stream.enqueueCopy(layer0_, outImg);
241 else
242 layer0_.copyTo(outImg);
243 }
244
245 float lastScale = 1.0f;
246 float curScale;
247 GpuMat lastLayer = layer0_;
248 GpuMat curLayer;
249
250 for (int i = 0; i < nLayers_ - 1; ++i)
251 {
252 curScale = lastScale * 0.5f;
253 curLayer = pyramid_[i];
254
255 if (outRoi.width == curLayer.cols && outRoi.height == curLayer.rows)
256 {
257 if (stream)
258 stream.enqueueCopy(curLayer, outImg);
259 else
260 curLayer.copyTo(outImg);
261 }
262
263 if (outRoi.width >= curLayer.cols && outRoi.height >= curLayer.rows)
264 break;
265
266 lastScale = curScale;
267 lastLayer = curLayer;
268 }
269
270 func(lastLayer, outImg, StreamAccessor::getStream(stream));
271}
