47cv::gpu::detail::FrameQueue::FrameQueue() :
48 endOfDecode_(0),
49 framesInQueue_(0),
50 readPosition_(0)
51{
52 std::memset(displayQueue_, 0, sizeof(displayQueue_));
53 std::memset((void*)isFrameInUse_, 0, sizeof(isFrameInUse_));
54}
55
56bool cv::gpu::detail::FrameQueue::waitUntilFrameAvailable(int pictureIndex)
57{
58 while (isInUse(pictureIndex))
59 {
60 // Decoder is getting too far ahead from display
61 Thread::sleep(1);
62
63 if (isEndOfDecode())
64 return false;
65 }
66
67 return true;
68}
69
70void cv::gpu::detail::FrameQueue::enqueue(const CUVIDPARSERDISPINFO* picParams)
71{
72 // Mark the frame as 'in-use' so we don't re-use it for decoding until it is no longer needed
73 // for display
74 isFrameInUse_[picParams->picture_index] = true;
75
76 // Wait until we have a free entry in the display queue (should never block if we have enough entries)
77 do
78 {
79 bool isFramePlaced = false;
80
81 {
82 CriticalSection::AutoLock autoLock(criticalSection_);
83
84 if (framesInQueue_ < MaximumSize)
85 {
86 int writePosition = (readPosition_ + framesInQueue_) % MaximumSize;
87 displayQueue_[writePosition] = *picParams;
88 framesInQueue_++;
89 isFramePlaced = true;
90 }
91 }
92
93 if (isFramePlaced) // Done
94 break;
95
96 // Wait a bit
97 Thread::sleep(1);
98 } while (!isEndOfDecode());
99}
100
101bool cv::gpu::detail::FrameQueue::dequeue(CUVIDPARSERDISPINFO& displayInfo)
102{
103 CriticalSection::AutoLock autoLock(criticalSection_);
104
105 if (framesInQueue_ > 0)
106 {
107 int entry = readPosition_;
108 displayInfo = displayQueue_[entry];
109 readPosition_ = (entry + 1) % MaximumSize;
110 framesInQueue_--;
111 return true;
112 }
113
114 return false;
115}
