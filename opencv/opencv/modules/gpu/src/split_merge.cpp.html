69 void merge(const GpuMat* src, size_t n, GpuMat& dst, const cudaStream_t& stream)
70 {
71 using namespace ::cv::gpu::device::split_merge;
72
73 CV_Assert(src);
74 CV_Assert(n > 0);
75
76 int depth = src[0].depth();
77 Size size = src[0].size();
78
79 if (depth == CV_64F)
80 {
81 if (!deviceSupports(NATIVE_DOUBLE))
82 CV_Error(CV_StsUnsupportedFormat, "The device doesn't support double");
83 }
84
85 bool single_channel_only = true;
86 int total_channels = 0;
87
88 for (size_t i = 0; i < n; ++i)
89 {
90 CV_Assert(src[i].size() == size);
91 CV_Assert(src[i].depth() == depth);
92 single_channel_only = single_channel_only && src[i].channels() == 1;
93 total_channels += src[i].channels();
94 }
95
96 CV_Assert(single_channel_only);
97 CV_Assert(total_channels <= 4);
98
99 if (total_channels == 1)
100 src[0].copyTo(dst);
101 else
102 {
103 dst.create(size, CV_MAKETYPE(depth, total_channels));
104
105 PtrStepSzb src_as_devmem[4];
106 for(size_t i = 0; i < n; ++i)
107 src_as_devmem[i] = src[i];
108
109 PtrStepSzb dst_as_devmem(dst);
110 merge_caller(src_as_devmem, dst_as_devmem, total_channels, CV_ELEM_SIZE(depth), stream);
111 }
112 }

114 void split(const GpuMat& src, GpuMat* dst, const cudaStream_t& stream)
115 {
116 using namespace ::cv::gpu::device::split_merge;
117
118 CV_Assert(dst);
119
120 int depth = src.depth();
121 int num_channels = src.channels();
122
123 if (depth == CV_64F)
124 {
125 if (!deviceSupports(NATIVE_DOUBLE))
126 CV_Error(CV_StsUnsupportedFormat, "The device doesn't support double");
127 }
128
129 if (num_channels == 1)
130 {
131 src.copyTo(dst[0]);
132 return;
133 }
134
135 for (int i = 0; i < num_channels; ++i)
136 dst[i].create(src.size(), depth);
137
138 CV_Assert(num_channels <= 4);
139
140 PtrStepSzb dst_as_devmem[4];
141 for (int i = 0; i < num_channels; ++i)
142 dst_as_devmem[i] = dst[i];
143
144 PtrStepSzb src_as_devmem(src);
145 split_caller(src_as_devmem, dst_as_devmem, num_channels, src.elemSize1(), stream);
146 }

149void cv::gpu::merge(const GpuMat* src, size_t n, GpuMat& dst, Stream& stream)
150{
151 ::merge(src, n, dst, StreamAccessor::getStream(stream));
152}

155void cv::gpu::merge(const vector<GpuMat>& src, GpuMat& dst, Stream& stream)
156{
157 ::merge(&src[0], src.size(), dst, StreamAccessor::getStream(stream));
158}
159
160void cv::gpu::split(const GpuMat& src, GpuMat* dst, Stream& stream)
161{
162 ::split(src, dst, StreamAccessor::getStream(stream));
163}
164
165void cv::gpu::split(const GpuMat& src, vector<GpuMat>& dst, Stream& stream)
166{
167 dst.resize(src.channels());
168 if(src.channels() > 0)
169 ::split(src, &dst[0], StreamAccessor::getStream(stream));
170}
