194void cv::gpu::BruteForceMatcher_GPU_base::matchSingle(const GpuMat& query, const GpuMat& train,
195 GpuMat& trainIdx, GpuMat& distance,
196 const GpuMat& mask, Stream& stream)
197{
198 if (query.empty() || train.empty())
199 return;
200
201 using namespace cv::gpu::device::bf_match;
202
203 typedef void (*caller_t)(const PtrStepSzb& query, const PtrStepSzb& train, const PtrStepSzb& mask,
204 const PtrStepSzi& trainIdx, const PtrStepSzf& distance,
205 cudaStream_t stream);
206
207#ifdef OPENCV_TINY_GPU_MODULE
208 static const caller_t callers[3][6] =
209 {
210 {
211 0/*matchL1_gpu<unsigned char>*/, 0/*matchL1_gpu<signed char>*/,
212 0/*matchL1_gpu<unsigned short>*/, 0/*matchL1_gpu<short>*/,
213 0/*matchL1_gpu<int>*/, 0/*matchL1_gpu<float>*/
214 },
215 {
216 0/*matchL2_gpu<unsigned char>*/, 0/*matchL2_gpu<signed char>*/,
217 0/*matchL2_gpu<unsigned short>*/, 0/*matchL2_gpu<short>*/,
218 0/*matchL2_gpu<int>*/, matchL2_gpu<float>
219 },
220 {
221 matchHamming_gpu<unsigned char>, 0/*matchHamming_gpu<signed char>*/,
222 0/*matchHamming_gpu<unsigned short>*/, 0/*matchHamming_gpu<short>*/,
223 0/*matchHamming_gpu<int>*/, 0/*matchHamming_gpu<float>*/
224 }
225 };
226#else
227 static const caller_t callers[3][6] =
228 {
229 {
230 matchL1_gpu<unsigned char>, 0/*matchL1_gpu<signed char>*/,
231 matchL1_gpu<unsigned short>, matchL1_gpu<short>,
232 matchL1_gpu<int>, matchL1_gpu<float>
233 },
234 {
235 0/*matchL2_gpu<unsigned char>*/, 0/*matchL2_gpu<signed char>*/,
236 0/*matchL2_gpu<unsigned short>*/, 0/*matchL2_gpu<short>*/,
237 0/*matchL2_gpu<int>*/, matchL2_gpu<float>
238 },
239 {
240 matchHamming_gpu<unsigned char>, 0/*matchHamming_gpu<signed char>*/,
241 matchHamming_gpu<unsigned short>, 0/*matchHamming_gpu<short>*/,
242 matchHamming_gpu<int>, 0/*matchHamming_gpu<float>*/
243 }
244 };
245#endif
246
247 CV_Assert(query.channels() == 1 && query.depth() < CV_64F);
248 CV_Assert(train.cols == query.cols && train.type() == query.type());
249
250 const int nQuery = query.rows;
251
252 ensureSizeIsEnough(1, nQuery, CV_32S, trainIdx);
253 ensureSizeIsEnough(1, nQuery, CV_32F, distance);
254
255 caller_t func = callers[distType][query.depth()];
256 CV_Assert(func != 0);
257
258 func(query, train, mask, trainIdx, distance, StreamAccessor::getStream(stream));
259}

272void cv::gpu::BruteForceMatcher_GPU_base::matchConvert(const Mat& trainIdx, const Mat& distance, vector<DMatch>& matches)
273{
274 if (trainIdx.empty() || distance.empty())
275 return;
276
277 CV_Assert(trainIdx.type() == CV_32SC1);
278 CV_Assert(distance.type() == CV_32FC1 && distance.cols == trainIdx.cols);
279
280 const int nQuery = trainIdx.cols;
281
282 matches.clear();
283 matches.reserve(nQuery);
284
285 const int* trainIdx_ptr = trainIdx.ptr<int>();
286 const float* distance_ptr = distance.ptr<float>();
287 for (int queryIdx = 0; queryIdx < nQuery; ++queryIdx, ++trainIdx_ptr, ++distance_ptr)
288 {
289 int train_idx = *trainIdx_ptr;
290
291 if (train_idx == -1)
292 continue;
293
294 float distance_local = *distance_ptr;
295
296 DMatch m(queryIdx, train_idx, 0, distance_local);
297
298 matches.push_back(m);
299 }
300}

310void cv::gpu::BruteForceMatcher_GPU_base::makeGpuCollection(GpuMat& trainCollection, GpuMat& maskCollection,
311 const vector<GpuMat>& masks)
312{
313 if (empty())
314 return;
315
316 if (masks.empty())
317 {
318 Mat trainCollectionCPU(1, static_cast<int>(trainDescCollection.size()), CV_8UC(sizeof(PtrStepSzb)));
319
320 PtrStepSzb* trainCollectionCPU_ptr = trainCollectionCPU.ptr<PtrStepSzb>();
321
322 for (size_t i = 0, size = trainDescCollection.size(); i < size; ++i, ++trainCollectionCPU_ptr)
323 *trainCollectionCPU_ptr = trainDescCollection[i];
324
325 trainCollection.upload(trainCollectionCPU);
326 maskCollection.release();
327 }
328 else
329 {
330 CV_Assert(masks.size() == trainDescCollection.size());
331
332 Mat trainCollectionCPU(1, static_cast<int>(trainDescCollection.size()), CV_8UC(sizeof(PtrStepSzb)));
333 Mat maskCollectionCPU(1, static_cast<int>(trainDescCollection.size()), CV_8UC(sizeof(PtrStepb)));
334
335 PtrStepSzb* trainCollectionCPU_ptr = trainCollectionCPU.ptr<PtrStepSzb>();
336 PtrStepb* maskCollectionCPU_ptr = maskCollectionCPU.ptr<PtrStepb>();
337
338 for (size_t i = 0, size = trainDescCollection.size(); i < size; ++i, ++trainCollectionCPU_ptr, ++maskCollectionCPU_ptr)
339 {
340 const GpuMat& train = trainDescCollection[i];
341 const GpuMat& mask = masks[i];
342
343 CV_Assert(mask.empty() || (mask.type() == CV_8UC1 && mask.cols == train.rows));
344
345 *trainCollectionCPU_ptr = train;
346 *maskCollectionCPU_ptr = mask;
347 }
348
349 trainCollection.upload(trainCollectionCPU);
350 maskCollection.upload(maskCollectionCPU);
351 }
352}

354void cv::gpu::BruteForceMatcher_GPU_base::matchCollection(const GpuMat& query, const GpuMat& trainCollection,
355 GpuMat& trainIdx, GpuMat& imgIdx, GpuMat& distance,
356 const GpuMat& masks, Stream& stream)
357{
358#ifdef OPENCV_TINY_GPU_MODULE
359 (void)query;
360 (void)trainCollection;
361 (void)trainIdx;
362 (void)imgIdx;
363 (void)distance;
364 (void)masks;
365 (void)stream;
366 CV_Error(CV_StsNotImplemented, "not available in tiny build");
367#else
368 if (query.empty() || trainCollection.empty())
369 return;
370
371 using namespace cv::gpu::device::bf_match;
372
373 typedef void (*caller_t)(const PtrStepSzb& query, const PtrStepSzb& trains, const PtrStepSz<PtrStepb>& masks,
374 const PtrStepSzi& trainIdx, const PtrStepSzi& imgIdx, const PtrStepSzf& distance,
375 cudaStream_t stream);
376
377 static const caller_t callers[3][6] =
378 {
379 {
380 matchL1_gpu<unsigned char>, 0/*matchL1_gpu<signed char>*/,
381 matchL1_gpu<unsigned short>, matchL1_gpu<short>,
382 matchL1_gpu<int>, matchL1_gpu<float>
383 },
384 {
385 0/*matchL2_gpu<unsigned char>*/, 0/*matchL2_gpu<signed char>*/,
386 0/*matchL2_gpu<unsigned short>*/, 0/*matchL2_gpu<short>*/,
387 0/*matchL2_gpu<int>*/, matchL2_gpu<float>
388 },
389 {
390 matchHamming_gpu<unsigned char>, 0/*matchHamming_gpu<signed char>*/,
391 matchHamming_gpu<unsigned short>, 0/*matchHamming_gpu<short>*/,
392 matchHamming_gpu<int>, 0/*matchHamming_gpu<float>*/
393 }
394 };
395
396 CV_Assert(query.channels() == 1 && query.depth() < CV_64F);
397
398 const int nQuery = query.rows;
399
400 ensureSizeIsEnough(1, nQuery, CV_32S, trainIdx);
401 ensureSizeIsEnough(1, nQuery, CV_32S, imgIdx);
402 ensureSizeIsEnough(1, nQuery, CV_32F, distance);
403
404 caller_t func = callers[distType][query.depth()];
405 CV_Assert(func != 0);
406
407 func(query, trainCollection, masks, trainIdx, imgIdx, distance, StreamAccessor::getStream(stream));
408#endif
409}

423void cv::gpu::BruteForceMatcher_GPU_base::matchConvert(const Mat& trainIdx, const Mat& imgIdx, const Mat& distance, std::vector<DMatch>& matches)
424{
425 if (trainIdx.empty() || imgIdx.empty() || distance.empty())
426 return;
427
428 CV_Assert(trainIdx.type() == CV_32SC1);
429 CV_Assert(imgIdx.type() == CV_32SC1 && imgIdx.cols == trainIdx.cols);
430 CV_Assert(distance.type() == CV_32FC1 && distance.cols == trainIdx.cols);
431
432 const int nQuery = trainIdx.cols;
433
434 matches.clear();
435 matches.reserve(nQuery);
436
437 const int* trainIdx_ptr = trainIdx.ptr<int>();
438 const int* imgIdx_ptr = imgIdx.ptr<int>();
439 const float* distance_ptr = distance.ptr<float>();
440 for (int queryIdx = 0; queryIdx < nQuery; ++queryIdx, ++trainIdx_ptr, ++imgIdx_ptr, ++distance_ptr)
441 {
442 int _trainIdx = *trainIdx_ptr;
443
444 if (_trainIdx == -1)
445 continue;
446
447 int _imgIdx = *imgIdx_ptr;
448
449 float _distance = *distance_ptr;
450
451 DMatch m(queryIdx, _trainIdx, _imgIdx, _distance);
452
453 matches.push_back(m);
454 }
455}

457void cv::gpu::BruteForceMatcher_GPU_base::match(const GpuMat& query, vector<DMatch>& matches, const vector<GpuMat>& masks)
458{
459 GpuMat trainCollection;
460 GpuMat maskCollection;
461
462 makeGpuCollection(trainCollection, maskCollection, masks);
463
464 GpuMat trainIdx, imgIdx, distance;
465
466 matchCollection(query, trainCollection, trainIdx, imgIdx, distance, maskCollection);
467 matchDownload(trainIdx, imgIdx, distance, matches);
468}

480 using namespace cv::gpu::device::bf_knnmatch;

483 const PtrStepSzb& trainIdx, const PtrStepSzb& distance, const PtrStepSzf& allDist,
484 cudaStream_t stream);
485
486#ifdef OPENCV_TINY_GPU_MODULE
487 static const caller_t callers[3][6] =
488 {
489 {
490 0/*matchL1_gpu<unsigned char>*/, 0/*matchL1_gpu<signed char>*/,
491 0/*matchL1_gpu<unsigned short>*/, 0/*matchL1_gpu<short>*/,
492 0/*matchL1_gpu<int>*/, 0/*matchL1_gpu<float>*/
493 },
494 {
495 0/*matchL2_gpu<unsigned char>*/, 0/*matchL2_gpu<signed char>*/,
496 0/*matchL2_gpu<unsigned short>*/, 0/*matchL2_gpu<short>*/,
497 0/*matchL2_gpu<int>*/, matchL2_gpu<float>

751void cv::gpu::BruteForceMatcher_GPU_base::knnMatch(const GpuMat& query, vector< vector<DMatch> >& matches, int k,
752 const vector<GpuMat>& masks, bool compactResult)
753{
754 if (k == 2)
755 {
756 GpuMat trainCollection;
757 GpuMat maskCollection;
758
759 makeGpuCollection(trainCollection, maskCollection, masks);
760
761 GpuMat trainIdx, imgIdx, distance;
762
763 knnMatch2Collection(query, trainCollection, trainIdx, imgIdx, distance, maskCollection);
764 knnMatch2Download(trainIdx, imgIdx, distance, matches);
765 }
766 else
767 {
768 if (query.empty() || empty())
769 return;
770
771 vector< vector<DMatch> > curMatches;
772 vector<DMatch> temp;
773 temp.reserve(2 * k);
774
775 matches.resize(query.rows);
776 for_each(matches.begin(), matches.end(), bind2nd(mem_fun_ref(&vector<DMatch>::reserve), k));
777
778 for (size_t imgIdx = 0, size = trainDescCollection.size(); imgIdx < size; ++imgIdx)
779 {
780 knnMatch(query, trainDescCollection[imgIdx], curMatches, k, masks.empty() ? GpuMat() : masks[imgIdx]);
781
782 for (int queryIdx = 0; queryIdx < query.rows; ++queryIdx)
783 {
784 vector<DMatch>& localMatch = curMatches[queryIdx];
785 vector<DMatch>& globalMatch = matches[queryIdx];
786
787 for_each(localMatch.begin(), localMatch.end(), ImgIdxSetter(static_cast<int>(imgIdx)));
788
789 temp.clear();
790 merge(globalMatch.begin(), globalMatch.end(), localMatch.begin(), localMatch.end(), back_inserter(temp));
791
792 globalMatch.clear();
793 const size_t count = std::min((size_t)k, temp.size());
794 copy(temp.begin(), temp.begin() + count, back_inserter(globalMatch));
795 }
796 }
797
798 if (compactResult)
799 {
800 vector< vector<DMatch> >::iterator new_end = remove_if(matches.begin(), matches.end(), mem_fun_ref(&vector<DMatch>::empty));
801 matches.erase(new_end, matches.end());
802 }
803 }
