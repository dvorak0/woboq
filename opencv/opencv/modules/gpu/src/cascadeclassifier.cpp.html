92 NCVStatus process(const GpuMat& src, GpuMat& objects, float scaleStep, int minNeighbors,
93 bool findLargestObject, bool visualizeInPlace, cv::Size ncvMinSize,
94 /*out*/unsigned int& numDetections)
95 {
96 calculateMemReqsAndAllocate(src.size());
97
98 NCVMemPtr src_beg;
99 src_beg.ptr = (void*)src.ptr<Ncv8u>();
100 src_beg.memtype = NCVMemoryTypeDevice;
101
102 NCVMemSegment src_seg;
103 src_seg.begin = src_beg;
104 src_seg.size = src.step * src.rows;
105
106 NCVMatrixReuse<Ncv8u> d_src(src_seg, static_cast<int>(devProp.textureAlignment), src.cols, src.rows, static_cast<int>(src.step), true);
107 ncvAssertReturn(d_src.isMemReused(), NCV_ALLOCATOR_BAD_REUSE);
108
109 CV_Assert(objects.rows == 1);
110
111 NCVMemPtr objects_beg;
112 objects_beg.ptr = (void*)objects.ptr<NcvRect32u>();
113 objects_beg.memtype = NCVMemoryTypeDevice;
114
115 NCVMemSegment objects_seg;
116 objects_seg.begin = objects_beg;
117 objects_seg.size = objects.step * objects.rows;
118 NCVVectorReuse<NcvRect32u> d_rects(objects_seg, objects.cols);
119 ncvAssertReturn(d_rects.isMemReused(), NCV_ALLOCATOR_BAD_REUSE);
120
121 NcvSize32u roi;
122 roi.width = d_src.width();
123 roi.height = d_src.height();
124
125 NcvSize32u winMinSize(ncvMinSize.width, ncvMinSize.height);
126
127 Ncv32u flags = 0;
128 flags |= findLargestObject? NCVPipeObjDet_FindLargestObject : 0;
129 flags |= visualizeInPlace ? NCVPipeObjDet_VisualizeInPlace : 0;
130
131 ncvStat = ncvDetectObjectsMultiScale_device(
132 d_src, roi, d_rects, numDetections, haar, *h_haarStages,
133 *d_haarStages, *d_haarNodes, *d_haarFeatures,
134 winMinSize,
135 minNeighbors,
136 scaleStep, 1,
137 flags,
138 *gpuAllocator, *cpuAllocator, devProp, 0);
139 ncvAssertReturnNcvStat(ncvStat);
140 ncvAssertCUDAReturn(cudaStreamSynchronize(0), NCV_CUDA_ERROR);
141
142 return NCV_SUCCESS;
143 }

146 bool findLargestObject, bool visualizeInPlace, cv::Size minSize, cv::Size /*maxObjectSize*/)

161 ncvMinSize.height = minSize.height;

163

165 ncvSafeCall(this->process(image, objectsBuf, (float)scaleFactor, minNeighbors, findLargestObject, visualizeInPlace, ncvMinSize, numDetections));
166
167 return numDetections;
168 }

175 NCVStatus load(const string& classifierFile)
176 {
177 int devId = cv::gpu::getDevice();
178 ncvAssertCUDAReturn(cudaGetDeviceProperties(&devProp, devId), NCV_CUDA_ERROR);
179
180 // Load the classifier from file (assuming its size is about 1 mb) using a simple allocator
181 gpuCascadeAllocator = new NCVMemNativeAllocator(NCVMemoryTypeDevice, static_cast<int>(devProp.textureAlignment));
182 cpuCascadeAllocator = new NCVMemNativeAllocator(NCVMemoryTypeHostPinned, static_cast<int>(devProp.textureAlignment));
183
184 ncvAssertPrintReturn(gpuCascadeAllocator->isInitialized(), "Error creating cascade GPU allocator", NCV_CUDA_ERROR);
185 ncvAssertPrintReturn(cpuCascadeAllocator->isInitialized(), "Error creating cascade CPU allocator", NCV_CUDA_ERROR);
186
187 Ncv32u haarNumStages, haarNumNodes, haarNumFeatures;
188 ncvStat = ncvHaarGetClassifierSize(classifierFile, haarNumStages, haarNumNodes, haarNumFeatures);
189 ncvAssertPrintReturn(ncvStat == NCV_SUCCESS, "Error reading classifier size (check the file)", NCV_FILE_ERROR);
190
191 h_haarStages = new NCVVectorAlloc<HaarStage64>(*cpuCascadeAllocator, haarNumStages);
192 h_haarNodes = new NCVVectorAlloc<HaarClassifierNode128>(*cpuCascadeAllocator, haarNumNodes);
193 h_haarFeatures = new NCVVectorAlloc<HaarFeature64>(*cpuCascadeAllocator, haarNumFeatures);
194
195 ncvAssertPrintReturn(h_haarStages->isMemAllocated(), "Error in cascade CPU allocator", NCV_CUDA_ERROR);
196 ncvAssertPrintReturn(h_haarNodes->isMemAllocated(), "Error in cascade CPU allocator", NCV_CUDA_ERROR);
197 ncvAssertPrintReturn(h_haarFeatures->isMemAllocated(), "Error in cascade CPU allocator", NCV_CUDA_ERROR);
198
199 ncvStat = ncvHaarLoadFromFile_host(classifierFile, haar, *h_haarStages, *h_haarNodes, *h_haarFeatures);
200 ncvAssertPrintReturn(ncvStat == NCV_SUCCESS, "Error loading classifier", NCV_FILE_ERROR);
201
202 d_haarStages = new NCVVectorAlloc<HaarStage64>(*gpuCascadeAllocator, haarNumStages);
203 d_haarNodes = new NCVVectorAlloc<HaarClassifierNode128>(*gpuCascadeAllocator, haarNumNodes);
204 d_haarFeatures = new NCVVectorAlloc<HaarFeature64>(*gpuCascadeAllocator, haarNumFeatures);
205
206 ncvAssertPrintReturn(d_haarStages->isMemAllocated(), "Error in cascade GPU allocator", NCV_CUDA_ERROR);
207 ncvAssertPrintReturn(d_haarNodes->isMemAllocated(), "Error in cascade GPU allocator", NCV_CUDA_ERROR);
208 ncvAssertPrintReturn(d_haarFeatures->isMemAllocated(), "Error in cascade GPU allocator", NCV_CUDA_ERROR);
209
210 ncvStat = h_haarStages->copySolid(*d_haarStages, 0);
211 ncvAssertPrintReturn(ncvStat == NCV_SUCCESS, "Error copying cascade to GPU", NCV_CUDA_ERROR);
212 ncvStat = h_haarNodes->copySolid(*d_haarNodes, 0);
213 ncvAssertPrintReturn(ncvStat == NCV_SUCCESS, "Error copying cascade to GPU", NCV_CUDA_ERROR);
214 ncvStat = h_haarFeatures->copySolid(*d_haarFeatures, 0);
215 ncvAssertPrintReturn(ncvStat == NCV_SUCCESS, "Error copying cascade to GPU", NCV_CUDA_ERROR);
216
217 return NCV_SUCCESS;
218 }

220 NCVStatus calculateMemReqsAndAllocate(const Size& frameSize)
221 {
222 if (lastAllocatedFrameSize == frameSize)
223 {
224 return NCV_SUCCESS;
225 }
226
227 // Calculate memory requirements and create real allocators
228 NCVMemStackAllocator gpuCounter(static_cast<int>(devProp.textureAlignment));
229 NCVMemStackAllocator cpuCounter(static_cast<int>(devProp.textureAlignment));
230
231 ncvAssertPrintReturn(gpuCounter.isInitialized(), "Error creating GPU memory counter", NCV_CUDA_ERROR);
232 ncvAssertPrintReturn(cpuCounter.isInitialized(), "Error creating CPU memory counter", NCV_CUDA_ERROR);
233
234 NCVMatrixAlloc<Ncv8u> d_src(gpuCounter, frameSize.width, frameSize.height);
235 NCVMatrixAlloc<Ncv8u> h_src(cpuCounter, frameSize.width, frameSize.height);
236
237 ncvAssertReturn(d_src.isMemAllocated(), NCV_ALLOCATOR_BAD_ALLOC);
238 ncvAssertReturn(h_src.isMemAllocated(), NCV_ALLOCATOR_BAD_ALLOC);
239
240 NCVVectorAlloc<NcvRect32u> d_rects(gpuCounter, 100);
241 ncvAssertReturn(d_rects.isMemAllocated(), NCV_ALLOCATOR_BAD_ALLOC);
242
243 NcvSize32u roi;
244 roi.width = d_src.width();
245 roi.height = d_src.height();
246 Ncv32u numDetections;
247 ncvStat = ncvDetectObjectsMultiScale_device(d_src, roi, d_rects, numDetections, haar, *h_haarStages,
248 *d_haarStages, *d_haarNodes, *d_haarFeatures, haar.ClassifierSize, 4, 1.2f, 1, 0, gpuCounter, cpuCounter, devProp, 0);
249
250 ncvAssertReturnNcvStat(ncvStat);
251 ncvAssertCUDAReturn(cudaStreamSynchronize(0), NCV_CUDA_ERROR);
252
253 gpuAllocator = new NCVMemStackAllocator(NCVMemoryTypeDevice, gpuCounter.maxSize(), static_cast<int>(devProp.textureAlignment));
254 cpuAllocator = new NCVMemStackAllocator(NCVMemoryTypeHostPinned, cpuCounter.maxSize(), static_cast<int>(devProp.textureAlignment));
255
256 ncvAssertPrintReturn(gpuAllocator->isInitialized(), "Error creating GPU memory allocator", NCV_CUDA_ERROR);
257 ncvAssertPrintReturn(cpuAllocator->isInitialized(), "Error creating CPU memory allocator", NCV_CUDA_ERROR);
258
259 lastAllocatedFrameSize = frameSize;
260 return NCV_SUCCESS;
261 }

312struct PyrLavel
313{
314 PyrLavel(int _order, float _scale, cv::Size frame, cv::Size window, cv::Size minObjectSize)
315 {
316 do
317 {
318 order = _order;
319 scale = pow(_scale, order);
320 sFrame = frame / scale;
321 workArea = sFrame - window + 1;
322 sWindow = window * scale;
323 _order++;
324 } while (sWindow <= minObjectSize);
325 }
326
327 bool isFeasible(cv::Size maxObj)
328 {
329 return workArea.width > 0 && workArea.height > 0 && sWindow <= maxObj;
330 }
331
332 PyrLavel next(float factor, cv::Size frame, cv::Size window, cv::Size minObjectSize)
333 {
334 return PyrLavel(order + 1, factor, frame, window, minObjectSize);
335 }
336
337 int order;
338 float scale;
339 cv::Size sFrame;
340 cv::Size workArea;
341 cv::Size sWindow;
342};

370struct cv::gpu::CascadeClassifier_GPU::LbpCascade : cv::gpu::CascadeClassifier_GPU::CascadeClassifierImpl
371{
372public:
373 struct Stage
374 {
375 int first;
376 int ntrees;
377 float threshold;
378 };
379
380 LbpCascade(){}
381 virtual ~LbpCascade(){}
382
383 virtual unsigned int process(const GpuMat& image, GpuMat& objects, float scaleFactor, int groupThreshold, bool /*findLargestObject*/,
384 bool /*visualizeInPlace*/, cv::Size minObjectSize, cv::Size maxObjectSize)
385 {
386 CV_Assert(scaleFactor > 1 && image.depth() == CV_8U);
387
388 // const int defaultObjSearchNum = 100;
389 const float grouping_eps = 0.2f;
390
391 if( !objects.empty() && objects.depth() == CV_32S)
392 objects.reshape(4, 1);
393 else
394 objects.create(1 , image.cols >> 4, CV_32SC4);
395
396 // used for debug
397 // candidates.setTo(cv::Scalar::all(0));
398 // objects.setTo(cv::Scalar::all(0));
399
400 if (maxObjectSize == cv::Size())
401 maxObjectSize = image.size();
402
403 allocateBuffers(image.size());
404
405 unsigned int classified = 0;
406 GpuMat dclassified(1, 1, CV_32S);
407 cudaSafeCall( cudaMemcpy(dclassified.ptr(), &classified, sizeof(int), cudaMemcpyHostToDevice) );
408
409 PyrLavel level(0, scaleFactor, image.size(), NxM, minObjectSize);
410
411 while (level.isFeasible(maxObjectSize))
412 {
413 int acc = level.sFrame.width + 1;
414 float iniScale = level.scale;
415
416 cv::Size area = level.workArea;
417 int step = 1 + (level.scale <= 2.f);
418
419 int total = 0, prev = 0;
420
421 while (acc <= integralFactor * (image.cols + 1) && level.isFeasible(maxObjectSize))
422 {
423 // create sutable matrix headers
424 GpuMat src = resuzeBuffer(cv::Rect(0, 0, level.sFrame.width, level.sFrame.height));
425 GpuMat sint = integral(cv::Rect(prev, 0, level.sFrame.width + 1, level.sFrame.height + 1));
426 GpuMat buff = integralBuffer;
427
428 // generate integral for scale
429 gpu::resize(image, src, level.sFrame, 0, 0, CV_INTER_LINEAR);
430 gpu::integralBuffered(src, sint, buff);
431
432 // calculate job
433 int totalWidth = level.workArea.width / step;
434 total += totalWidth * (level.workArea.height / step);
435
436 // go to next pyramide level
437 level = level.next(scaleFactor, image.size(), NxM, minObjectSize);
438 area = level.workArea;
439
440 step = (1 + (level.scale <= 2.f));
441 prev = acc;
442 acc += level.sFrame.width + 1;
443 }
444
445 device::lbp::classifyPyramid(image.cols, image.rows, NxM.width - 1, NxM.height - 1, iniScale, scaleFactor, total, stage_mat, stage_mat.cols / sizeof(Stage), nodes_mat,
446 leaves_mat, subsets_mat, features_mat, subsetSize, candidates, dclassified.ptr<unsigned int>(), integral);
447 }
448
449 if (groupThreshold <= 0 || objects.empty())
450 return 0;
451
452 cudaSafeCall( cudaMemcpy(&classified, dclassified.ptr(), sizeof(int), cudaMemcpyDeviceToHost) );
453 device::lbp::connectedConmonents(candidates, classified, objects, groupThreshold, grouping_eps, dclassified.ptr<unsigned int>());
454
455 cudaSafeCall( cudaMemcpy(&classified, dclassified.ptr(), sizeof(int), cudaMemcpyDeviceToHost) );
456 cudaSafeCall( cudaDeviceSynchronize() );
457 return classified;
458 }

472 if (frame == cv::Size())
473 return;
474
475 if (resuzeBuffer.empty() || frame.width > resuzeBuffer.cols || frame.height > resuzeBuffer.rows)

495 bool read(const FileNode &root)
496 {
497 const char *GPU_CC_STAGE_TYPE = "stageType";
498 const char *GPU_CC_FEATURE_TYPE = "featureType";
499 const char *GPU_CC_BOOST = "BOOST";
500 const char *GPU_CC_LBP = "LBP";
501 const char *GPU_CC_MAX_CAT_COUNT = "maxCatCount";
502 const char *GPU_CC_HEIGHT = "height";
503 const char *GPU_CC_WIDTH = "width";
504 const char *GPU_CC_STAGE_PARAMS = "stageParams";
505 const char *GPU_CC_MAX_DEPTH = "maxDepth";
506 const char *GPU_CC_FEATURE_PARAMS = "featureParams";
507 const char *GPU_CC_STAGES = "stages";
508 const char *GPU_CC_STAGE_THRESHOLD = "stageThreshold";
509 const float GPU_THRESHOLD_EPS = 1e-5f;
510 const char *GPU_CC_WEAK_CLASSIFIERS = "weakClassifiers";
511 const char *GPU_CC_INTERNAL_NODES = "internalNodes";
512 const char *GPU_CC_LEAF_VALUES = "leafValues";
513 const char *GPU_CC_FEATURES = "features";
514 const char *GPU_CC_RECT = "rect";
515
516 std::string stageTypeStr = (string)root[GPU_CC_STAGE_TYPE];
517 CV_Assert(stageTypeStr == GPU_CC_BOOST);
518
519 string featureTypeStr = (string)root[GPU_CC_FEATURE_TYPE];
520 CV_Assert(featureTypeStr == GPU_CC_LBP);
521
522 NxM.width = (int)root[GPU_CC_WIDTH];
523 NxM.height = (int)root[GPU_CC_HEIGHT];
524 CV_Assert( NxM.height > 0 && NxM.width > 0 );
525
526 isStumps = ((int)(root[GPU_CC_STAGE_PARAMS][GPU_CC_MAX_DEPTH]) == 1) ? true : false;
527 CV_Assert(isStumps);
528
529 FileNode fn = root[GPU_CC_FEATURE_PARAMS];
530 if (fn.empty())
531 return false;
532
533 ncategories = fn[GPU_CC_MAX_CAT_COUNT];
534
535 subsetSize = (ncategories + 31) / 32;
536 nodeStep = 3 + ( ncategories > 0 ? subsetSize : 1 );
537
538 fn = root[GPU_CC_STAGES];
539 if (fn.empty())
540 return false;
541
542 std::vector<Stage> stages;
543 stages.reserve(fn.size());
544
545 std::vector<int> cl_trees;
546 std::vector<int> cl_nodes;
547 std::vector<float> cl_leaves;
548 std::vector<int> subsets;
549
550 FileNodeIterator it = fn.begin(), it_end = fn.end();
551 for (size_t si = 0; it != it_end; si++, ++it )
552 {
553 FileNode fns = *it;
554 Stage st;
555 st.threshold = (float)fns[GPU_CC_STAGE_THRESHOLD] - GPU_THRESHOLD_EPS;
556
557 fns = fns[GPU_CC_WEAK_CLASSIFIERS];
558 if (fns.empty())
559 return false;
560
561 st.ntrees = (int)fns.size();
562 st.first = (int)cl_trees.size();
563
564 stages.push_back(st);// (int, int, float)
565
566 cl_trees.reserve(stages[si].first + stages[si].ntrees);
567
568 // weak trees
569 FileNodeIterator it1 = fns.begin(), it1_end = fns.end();
570 for ( ; it1 != it1_end; ++it1 )
571 {
572 FileNode fnw = *it1;
573
574 FileNode internalNodes = fnw[GPU_CC_INTERNAL_NODES];
575 FileNode leafValues = fnw[GPU_CC_LEAF_VALUES];
576 if ( internalNodes.empty() || leafValues.empty() )
577 return false;
578
579 int nodeCount = (int)internalNodes.size()/nodeStep;
580 cl_trees.push_back(nodeCount);
581
582 cl_nodes.reserve((cl_nodes.size() + nodeCount) * 3);
583 cl_leaves.reserve(cl_leaves.size() + leafValues.size());
584
585 if( subsetSize > 0 )
586 subsets.reserve(subsets.size() + nodeCount * subsetSize);
587
588 // nodes
589 FileNodeIterator iIt = internalNodes.begin(), iEnd = internalNodes.end();
590
591 for( ; iIt != iEnd; )
592 {
593 cl_nodes.push_back((int)*(iIt++));
594 cl_nodes.push_back((int)*(iIt++));
595 cl_nodes.push_back((int)*(iIt++));
596
597 if( subsetSize > 0 )
598 for( int j = 0; j < subsetSize; j++, ++iIt )
599 subsets.push_back((int)*iIt);
600 }
601
602 // leaves
603 iIt = leafValues.begin(), iEnd = leafValues.end();
604 for( ; iIt != iEnd; ++iIt )
605 cl_leaves.push_back((float)*iIt);
606 }
607 }
608
609 fn = root[GPU_CC_FEATURES];
610 if( fn.empty() )
611 return false;
612 std::vector<uchar> features;
613 features.reserve(fn.size() * 4);
614 FileNodeIterator f_it = fn.begin(), f_end = fn.end();
615 for (; f_it != f_end; ++f_it)
616 {
617 FileNode rect = (*f_it)[GPU_CC_RECT];
618 FileNodeIterator r_it = rect.begin();
619 features.push_back(saturate_cast<uchar>((int)*(r_it++)));
620 features.push_back(saturate_cast<uchar>((int)*(r_it++)));
621 features.push_back(saturate_cast<uchar>((int)*(r_it++)));
622 features.push_back(saturate_cast<uchar>((int)*(r_it++)));
623 }
624
625 // copy data structures on gpu
626 stage_mat.upload(cv::Mat(1, (int) (stages.size() * sizeof(Stage)), CV_8UC1, (uchar*)&(stages[0]) ));
627 trees_mat.upload(cv::Mat(cl_trees).reshape(1,1));
628 nodes_mat.upload(cv::Mat(cl_nodes).reshape(1,1));
629 leaves_mat.upload(cv::Mat(cl_leaves).reshape(1,1));
630 subsets_mat.upload(cv::Mat(subsets).reshape(1,1));
631 features_mat.upload(cv::Mat(features).reshape(4,1));
632
633 return true;
634 }

692bool cv::gpu::CascadeClassifier_GPU::load(const string& filename)
693{
694 release();
695
696 std::string fext = filename.substr(filename.find_last_of(".") + 1);
697 std::transform(fext.begin(), fext.end(), fext.begin(), ::tolower);
698
699 if (fext == "nvbin")
700 {
701 impl = new HaarCascade();
702 return impl->read(filename);
703 }
704
705 FileStorage fs(filename, FileStorage::READ);
706
707 if (!fs.isOpened())
708 {
709 impl = new HaarCascade();
710 return impl->read(filename);
711 }
712
713 const char *GPU_CC_LBP = "LBP";
714 string featureTypeStr = (string)fs.getFirstTopLevelNode()["featureType"];
715 if (featureTypeStr == GPU_CC_LBP)
716 impl = new LbpCascade();
717 else
718 impl = new HaarCascade();
719
720 impl->read(filename);
721 return !this->empty();
722}

763NCVStatus loadFromXML(const std::string &filename,
764 HaarClassifierCascadeDescriptor &haar,
765 std::vector<HaarStage64> &haarStages,
766 std::vector<HaarClassifierNode128> &haarClassifierNodes,
767 std::vector<HaarFeature64> &haarFeatures)
768{
769 NCVStatus ncvStat;
770
771 haar.NumStages = 0;
772 haar.NumClassifierRootNodes = 0;
773 haar.NumClassifierTotalNodes = 0;
774 haar.NumFeatures = 0;
775 haar.ClassifierSize.width = 0;
776 haar.ClassifierSize.height = 0;
777 haar.bHasStumpsOnly = true;
778 haar.bNeedsTiltedII = false;
779 Ncv32u curMaxTreeDepth;
780
781 std::vector<HaarClassifierNode128> h_TmpClassifierNotRootNodes;
782 haarStages.resize(0);
783 haarClassifierNodes.resize(0);
784 haarFeatures.resize(0);
785
786 Ptr<CvHaarClassifierCascade> oldCascade = (CvHaarClassifierCascade*)cvLoad(filename.c_str(), 0, 0, 0);
787 if (oldCascade.empty())
788 {
789 return NCV_HAAR_XML_LOADING_EXCEPTION;
790 }
791
792 haar.ClassifierSize.width = oldCascade->orig_window_size.width;
793 haar.ClassifierSize.height = oldCascade->orig_window_size.height;
794
795 int stagesCound = oldCascade->count;
796 for(int s = 0; s < stagesCound; ++s) // by stages
797 {
798 HaarStage64 curStage;
799 curStage.setStartClassifierRootNodeOffset(static_cast<Ncv32u>(haarClassifierNodes.size()));
800
801 curStage.setStageThreshold(oldCascade->stage_classifier[s].threshold);
802
803 int treesCount = oldCascade->stage_classifier[s].count;
804 for(int t = 0; t < treesCount; ++t) // by trees
805 {
806 Ncv32u nodeId = 0;
807 CvHaarClassifier* tree = &oldCascade->stage_classifier[s].classifier[t];
808
809 int nodesCount = tree->count;
810 for(int n = 0; n < nodesCount; ++n) //by features
811 {
812 CvHaarFeature* feature = &tree->haar_feature[n];
813
814 HaarClassifierNode128 curNode;
815 curNode.setThreshold(tree->threshold[n]);
816
817 NcvBool bIsLeftNodeLeaf = false;
818 NcvBool bIsRightNodeLeaf = false;
819
820 HaarClassifierNodeDescriptor32 nodeLeft;
821 if ( tree->left[n] <= 0 )
822 {
823 Ncv32f leftVal = tree->alpha[-tree->left[n]];
824 ncvStat = nodeLeft.create(leftVal);
825 ncvAssertReturn(ncvStat == NCV_SUCCESS, ncvStat);
826 bIsLeftNodeLeaf = true;
827 }
828 else
829 {
830 Ncv32u leftNodeOffset = tree->left[n];
831 nodeLeft.create((Ncv32u)(h_TmpClassifierNotRootNodes.size() + leftNodeOffset - 1));
832 haar.bHasStumpsOnly = false;
833 }
834 curNode.setLeftNodeDesc(nodeLeft);
835
836 HaarClassifierNodeDescriptor32 nodeRight;
837 if ( tree->right[n] <= 0 )
838 {
839 Ncv32f rightVal = tree->alpha[-tree->right[n]];
840 ncvStat = nodeRight.create(rightVal);
841 ncvAssertReturn(ncvStat == NCV_SUCCESS, ncvStat);
842 bIsRightNodeLeaf = true;
843 }
844 else
845 {
846 Ncv32u rightNodeOffset = tree->right[n];
847 nodeRight.create((Ncv32u)(h_TmpClassifierNotRootNodes.size() + rightNodeOffset - 1));
848 haar.bHasStumpsOnly = false;
849 }
850 curNode.setRightNodeDesc(nodeRight);
851
852 Ncv32u tiltedVal = feature->tilted;
853 haar.bNeedsTiltedII = (tiltedVal != 0);
854
855 Ncv32u featureId = 0;
856 for(int l = 0; l < CV_HAAR_FEATURE_MAX; ++l) //by rects
857 {
858 Ncv32u rectX = feature->rect[l].r.x;
859 Ncv32u rectY = feature->rect[l].r.y;
860 Ncv32u rectWidth = feature->rect[l].r.width;
861 Ncv32u rectHeight = feature->rect[l].r.height;
862
863 Ncv32f rectWeight = feature->rect[l].weight;
864
865 if (rectWeight == 0/* && rectX == 0 &&rectY == 0 && rectWidth == 0 && rectHeight == 0*/)
866 break;
867
868 HaarFeature64 curFeature;
869 ncvStat = curFeature.setRect(rectX, rectY, rectWidth, rectHeight, haar.ClassifierSize.width, haar.ClassifierSize.height);
870 curFeature.setWeight(rectWeight);
871 ncvAssertReturn(NCV_SUCCESS == ncvStat, ncvStat);
872 haarFeatures.push_back(curFeature);
873
874 featureId++;
875 }
876
877 HaarFeatureDescriptor32 tmpFeatureDesc;
878 ncvStat = tmpFeatureDesc.create(haar.bNeedsTiltedII, bIsLeftNodeLeaf, bIsRightNodeLeaf,
879 featureId, static_cast<Ncv32u>(haarFeatures.size()) - featureId);
880 ncvAssertReturn(NCV_SUCCESS == ncvStat, ncvStat);
881 curNode.setFeatureDesc(tmpFeatureDesc);
882
883 if (!nodeId)
884 {
885 //root node
886 haarClassifierNodes.push_back(curNode);
887 curMaxTreeDepth = 1;
888 }
889 else
890 {
891 //other node
892 h_TmpClassifierNotRootNodes.push_back(curNode);
893 curMaxTreeDepth++;
894 }
895
896 nodeId++;
897 }
898 }
899
900 curStage.setNumClassifierRootNodes(treesCount);
901 haarStages.push_back(curStage);
902 }
903
904 //fill in cascade stats
905 haar.NumStages = static_cast<Ncv32u>(haarStages.size());
906 haar.NumClassifierRootNodes = static_cast<Ncv32u>(haarClassifierNodes.size());
907 haar.NumClassifierTotalNodes = static_cast<Ncv32u>(haar.NumClassifierRootNodes + h_TmpClassifierNotRootNodes.size());
908 haar.NumFeatures = static_cast<Ncv32u>(haarFeatures.size());
909
910 //merge root and leaf nodes in one classifiers array
911 Ncv32u offsetRoot = static_cast<Ncv32u>(haarClassifierNodes.size());
912 for (Ncv32u i=0; i<haarClassifierNodes.size(); i++)
913 {
914 HaarFeatureDescriptor32 featureDesc = haarClassifierNodes[i].getFeatureDesc();
915
916 HaarClassifierNodeDescriptor32 nodeLeft = haarClassifierNodes[i].getLeftNodeDesc();
917 if (!featureDesc.isLeftNodeLeaf())
918 {
919 Ncv32u newOffset = nodeLeft.getNextNodeOffset() + offsetRoot;
920 nodeLeft.create(newOffset);
921 }
922 haarClassifierNodes[i].setLeftNodeDesc(nodeLeft);
923
924 HaarClassifierNodeDescriptor32 nodeRight = haarClassifierNodes[i].getRightNodeDesc();
925 if (!featureDesc.isRightNodeLeaf())
926 {
927 Ncv32u newOffset = nodeRight.getNextNodeOffset() + offsetRoot;
928 nodeRight.create(newOffset);
929 }
930 haarClassifierNodes[i].setRightNodeDesc(nodeRight);
931 }
932
933 for (Ncv32u i=0; i<h_TmpClassifierNotRootNodes.size(); i++)
934 {
935 HaarFeatureDescriptor32 featureDesc = h_TmpClassifierNotRootNodes[i].getFeatureDesc();
936
937 HaarClassifierNodeDescriptor32 nodeLeft = h_TmpClassifierNotRootNodes[i].getLeftNodeDesc();
938 if (!featureDesc.isLeftNodeLeaf())
939 {
940 Ncv32u newOffset = nodeLeft.getNextNodeOffset() + offsetRoot;
941 nodeLeft.create(newOffset);
942 }
943 h_TmpClassifierNotRootNodes[i].setLeftNodeDesc(nodeLeft);
944
945 HaarClassifierNodeDescriptor32 nodeRight = h_TmpClassifierNotRootNodes[i].getRightNodeDesc();
946 if (!featureDesc.isRightNodeLeaf())
947 {
948 Ncv32u newOffset = nodeRight.getNextNodeOffset() + offsetRoot;
949 nodeRight.create(newOffset);
950 }
951 h_TmpClassifierNotRootNodes[i].setRightNodeDesc(nodeRight);
952
953 haarClassifierNodes.push_back(h_TmpClassifierNotRootNodes[i]);
954 }
955
956 return NCV_SUCCESS;
957}
