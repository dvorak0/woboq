64 class NppStreamHandler
65 {
66 public:
67 inline explicit NppStreamHandler(cudaStream_t newStream = 0)
68 {
69 oldStream = nppGetStream();
70 nppSetStream(newStream);
71 }
72
73 inline ~NppStreamHandler()
74 {
75 nppSetStream(oldStream);
76 }
77
78 private:
79 cudaStream_t oldStream;
80 };
81
82 class NppStStreamHandler
83 {
84 public:
85 inline explicit NppStStreamHandler(cudaStream_t newStream = 0)
86 {
87 oldStream = nppStSetActiveCUDAstream(newStream);
88 }
89
90 inline ~NppStStreamHandler()
91 {
92 nppStSetActiveCUDAstream(oldStream);
93 }
94
95 private:
96 cudaStream_t oldStream;
97 };
