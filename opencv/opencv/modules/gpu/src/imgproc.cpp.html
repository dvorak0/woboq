114void cv::gpu::meanShiftFiltering(const GpuMat& src, GpuMat& dst, int sp, int sr, TermCriteria criteria, Stream& stream)
115{
116 using namespace ::cv::gpu::device::imgproc;
117
118 if( src.empty() )
119 CV_Error( CV_StsBadArg, "The input image is empty" );
120
121 if( src.depth() != CV_8U || src.channels() != 4 )
122 CV_Error( CV_StsUnsupportedFormat, "Only 8-bit, 4-channel images are supported" );
123
124 dst.create( src.size(), CV_8UC4 );
125
126 if( !(criteria.type & TermCriteria::MAX_ITER) )
127 criteria.maxCount = 5;
128
129 int maxIter = std::min(std::max(criteria.maxCount, 1), 100);
130
131 float eps;
132 if( !(criteria.type & TermCriteria::EPS) )
133 eps = 1.f;
134 eps = (float)std::max(criteria.epsilon, 0.0);
135
136 meanShiftFiltering_gpu(src, dst, sp, sr, maxIter, eps, StreamAccessor::getStream(stream));
137}

277void cv::gpu::copyMakeBorder(const GpuMat& src, GpuMat& dst, int top, int bottom, int left, int right, int borderType, const Scalar& value, Stream& s)
278{
279 CV_Assert(src.depth() <= CV_32F && src.channels() <= 4);
280 CV_Assert(borderType == BORDER_REFLECT101 || borderType == BORDER_REPLICATE || borderType == BORDER_CONSTANT || borderType == BORDER_REFLECT || borderType == BORDER_WRAP);
281
282 dst.create(src.rows + top + bottom, src.cols + left + right, src.type());
283
284 cudaStream_t stream = StreamAccessor::getStream(s);
285
286#ifdef USE_NPP
287 if (borderType == BORDER_CONSTANT && (src.type() == CV_8UC1 || src.type() == CV_8UC4 || src.type() == CV_32SC1 || src.type() == CV_32FC1))
288 {
289 NppiSize srcsz;
290 srcsz.width = src.cols;
291 srcsz.height = src.rows;
292
293 NppiSize dstsz;
294 dstsz.width = dst.cols;
295 dstsz.height = dst.rows;
296
297 NppStreamHandler h(stream);
298
299 switch (src.type())
300 {
301 case CV_8UC1:
302 {
303 Npp8u nVal = saturate_cast<Npp8u>(value[0]);
304 nppSafeCall( nppiCopyConstBorder_8u_C1R(src.ptr<Npp8u>(), static_cast<int>(src.step), srcsz,
305 dst.ptr<Npp8u>(), static_cast<int>(dst.step), dstsz, top, left, nVal) );
306 break;
307 }
308 case CV_8UC4:
309 {
310 Npp8u nVal[] = {saturate_cast<Npp8u>(value[0]), saturate_cast<Npp8u>(value[1]), saturate_cast<Npp8u>(value[2]), saturate_cast<Npp8u>(value[3])};
311 nppSafeCall( nppiCopyConstBorder_8u_C4R(src.ptr<Npp8u>(), static_cast<int>(src.step), srcsz,
312 dst.ptr<Npp8u>(), static_cast<int>(dst.step), dstsz, top, left, nVal) );
313 break;
314 }
315 case CV_32SC1:
316 {
317 Npp32s nVal = saturate_cast<Npp32s>(value[0]);
318 nppSafeCall( nppiCopyConstBorder_32s_C1R(src.ptr<Npp32s>(), static_cast<int>(src.step), srcsz,
319 dst.ptr<Npp32s>(), static_cast<int>(dst.step), dstsz, top, left, nVal) );
320 break;
321 }
322 case CV_32FC1:
323 {
324 Npp32f val = saturate_cast<Npp32f>(value[0]);
325 Npp32s nVal = *(reinterpret_cast<Npp32s_a*>(&val));
326 nppSafeCall( nppiCopyConstBorder_32s_C1R(src.ptr<Npp32s>(), static_cast<int>(src.step), srcsz,
327 dst.ptr<Npp32s>(), static_cast<int>(dst.step), dstsz, top, left, nVal) );
328 break;
329 }
330 }
331
332 if (stream == 0)
333 cudaSafeCall( cudaDeviceSynchronize() );
334 }
335 else
336#endif
337 {
338 typedef void (*caller_t)(const PtrStepSzb& src, const PtrStepSzb& dst, int top, int left, int borderType, const Scalar& value, cudaStream_t stream);
339#ifdef OPENCV_TINY_GPU_MODULE
340 static const caller_t callers[6][4] =
341 {
342 { copyMakeBorder_caller<uchar, 1> , copyMakeBorder_caller<uchar, 2> , copyMakeBorder_caller<uchar, 3> , copyMakeBorder_caller<uchar, 4>},
343 {0, 0, 0, 0},
344 {0, 0, 0, 0},
345 {0, 0, 0, 0},
346 {0, 0, 0, 0},
347 { copyMakeBorder_caller<float, 1> , 0/*copyMakeBorder_caller<float, 2>*/, copyMakeBorder_caller<float, 3> , copyMakeBorder_caller<float ,4>}
348 };
349#else
350 static const caller_t callers[6][4] =
351 {
352 { copyMakeBorder_caller<uchar, 1> , copyMakeBorder_caller<uchar, 2> , copyMakeBorder_caller<uchar, 3> , copyMakeBorder_caller<uchar, 4>},
353 {0/*copyMakeBorder_caller<schar, 1>*/, 0/*copyMakeBorder_caller<schar, 2>*/ , 0/*copyMakeBorder_caller<schar, 3>*/, 0/*copyMakeBorder_caller<schar, 4>*/},
354 { copyMakeBorder_caller<ushort, 1> , 0/*copyMakeBorder_caller<ushort, 2>*/, copyMakeBorder_caller<ushort, 3> , copyMakeBorder_caller<ushort, 4>},
355 { copyMakeBorder_caller<short, 1> , 0/*copyMakeBorder_caller<short, 2>*/ , copyMakeBorder_caller<short, 3> , copyMakeBorder_caller<short, 4>},
356 {0/*copyMakeBorder_caller<int, 1>*/, 0/*copyMakeBorder_caller<int, 2>*/ , 0/*copyMakeBorder_caller<int, 3>*/, 0/*copyMakeBorder_caller<int , 4>*/},
357 { copyMakeBorder_caller<float, 1> , 0/*copyMakeBorder_caller<float, 2>*/ , copyMakeBorder_caller<float, 3> , copyMakeBorder_caller<float ,4>}
358 };
359#endif
360
361 caller_t func = callers[src.depth()][src.channels() - 1];
362 CV_Assert(func != 0);
363
364 int gpuBorderType;
365 CV_Assert(tryConvertToGpuBorderType(borderType, gpuBorderType));
366
367 func(src, dst, top, left, gpuBorderType, value, stream);
368 }

1266// dft
1267
1268void cv::gpu::dft(const GpuMat& src, GpuMat& dst, Size dft_size, int flags, Stream& stream)
1269{
1270#ifndef HAVE_CUFFT
1271
1272 OPENCV_GPU_UNUSED(src);
1273 OPENCV_GPU_UNUSED(dst);
1274 OPENCV_GPU_UNUSED(dft_size);
1275 OPENCV_GPU_UNUSED(flags);
1276 OPENCV_GPU_UNUSED(stream);
1277
1278 throw_nogpu();
1279
1280#else
1281
1282 CV_Assert(src.type() == CV_32F || src.type() == CV_32FC2);
1283
1284 // We don't support unpacked output (in the case of real input)
1285 CV_Assert(!(flags & DFT_COMPLEX_OUTPUT));
1286
1287 bool is_1d_input = (dft_size.height == 1) || (dft_size.width == 1);
1288 int is_row_dft = flags & DFT_ROWS;
1289 int is_scaled_dft = flags & DFT_SCALE;
1290 int is_inverse = flags & DFT_INVERSE;
1291 bool is_complex_input = src.channels() == 2;
1292 bool is_complex_output = !(flags & DFT_REAL_OUTPUT);
1293
1294 // We don't support real-to-real transform
1295 CV_Assert(is_complex_input || is_complex_output);
1296
1297 GpuMat src_data;
1298
1299 // Make sure here we work with the continuous input,
1300 // as CUFFT can't handle gaps
1301 src_data = src;
1302 createContinuous(src.rows, src.cols, src.type(), src_data);
1303 if (src_data.data != src.data)
1304 src.copyTo(src_data);
1305
1306 Size dft_size_opt = dft_size;
1307 if (is_1d_input && !is_row_dft)
1308 {
1309 // If the source matrix is single column handle it as single row
1310 dft_size_opt.width = std::max(dft_size.width, dft_size.height);
1311 dft_size_opt.height = std::min(dft_size.width, dft_size.height);
1312 }
1313
1314 cufftType dft_type = CUFFT_R2C;
1315 if (is_complex_input)
1316 dft_type = is_complex_output ? CUFFT_C2C : CUFFT_C2R;
1317
1318 CV_Assert(dft_size_opt.width > 1);
1319
1320 cufftHandle plan;
1321 if (is_1d_input || is_row_dft)
1322 cufftPlan1d(&plan, dft_size_opt.width, dft_type, dft_size_opt.height);
1323 else
1324 cufftPlan2d(&plan, dft_size_opt.height, dft_size_opt.width, dft_type);
1325
1326 cufftSafeCall( cufftSetStream(plan, StreamAccessor::getStream(stream)) );
1327
1328 if (is_complex_input)
1329 {
1330 if (is_complex_output)
1331 {
1332 createContinuous(dft_size, CV_32FC2, dst);
1333 cufftSafeCall(cufftExecC2C(
1334 plan, src_data.ptr<cufftComplex>(), dst.ptr<cufftComplex>(),
1335 is_inverse ? CUFFT_INVERSE : CUFFT_FORWARD));
1336 }
1337 else
1338 {
1339 createContinuous(dft_size, CV_32F, dst);
1340 cufftSafeCall(cufftExecC2R(
1341 plan, src_data.ptr<cufftComplex>(), dst.ptr<cufftReal>()));
1342 }
1343 }
1344 else
1345 {
1346 // We could swap dft_size for efficiency. Here we must reflect it
1347 if (dft_size == dft_size_opt)
1348 createContinuous(Size(dft_size.width / 2 + 1, dft_size.height), CV_32FC2, dst);
1349 else
1350 createContinuous(Size(dft_size.width, dft_size.height / 2 + 1), CV_32FC2, dst);
1351
1352 cufftSafeCall(cufftExecR2C(
1353 plan, src_data.ptr<cufftReal>(), dst.ptr<cufftComplex>()));
1354 }
1355
1356 cufftSafeCall(cufftDestroy(plan));
1357
1358 if (is_scaled_dft)
1359 multiply(dst, Scalar::all(1. / dft_size.area()), dst, 1, -1, stream);
1360
1361#endif
1362}

1366
1367void cv::gpu::ConvolveBuf::create(Size image_size, Size templ_size)
1368{
1369 result_size = Size(image_size.width - templ_size.width + 1,
1370 image_size.height - templ_size.height + 1);
1371
1372 block_size = user_block_size;
1373 if (user_block_size.width == 0 || user_block_size.height == 0)
1374 block_size = estimateBlockSize(result_size, templ_size);
1375
1376 dft_size.width = 1 << int(ceil(std::log(block_size.width + templ_size.width - 1.) / std::log(2.)));
1377 dft_size.height = 1 << int(ceil(std::log(block_size.height + templ_size.height - 1.) / std::log(2.)));
1378
1379 // CUFFT has hard-coded kernels for power-of-2 sizes (up to 8192),
1380 // see CUDA Toolkit 4.1 CUFFT Library Programming Guide
1381 if (dft_size.width > 8192)
1382 dft_size.width = getOptimalDFTSize(block_size.width + templ_size.width - 1);
1383 if (dft_size.height > 8192)
1384 dft_size.height = getOptimalDFTSize(block_size.height + templ_size.height - 1);
1385
1386 // To avoid wasting time doing small DFTs
1387 dft_size.width = std::max(dft_size.width, 512);
1388 dft_size.height = std::max(dft_size.height, 512);
1389
1390 createContinuous(dft_size, CV_32F, image_block);
1391 createContinuous(dft_size, CV_32F, templ_block);
1392 createContinuous(dft_size, CV_32F, result_data);
1393
1394 spect_len = dft_size.height * (dft_size.width / 2 + 1);
1395 createContinuous(1, spect_len, CV_32FC2, image_spect);
1396 createContinuous(1, spect_len, CV_32FC2, templ_spect);
1397 createContinuous(1, spect_len, CV_32FC2, result_spect);
1398
1399 // Use maximum result matrix block size for the estimated DFT block size
1400 block_size.width = std::min(dft_size.width - templ_size.width + 1, result_size.width);
1401 block_size.height = std::min(dft_size.height - templ_size.height + 1, result_size.height);
1402}

1421void cv::gpu::convolve(const GpuMat& image, const GpuMat& templ, GpuMat& result, bool ccorr, ConvolveBuf& buf, Stream& stream)
1422{
1423 using namespace ::cv::gpu::device::imgproc;
1424
1425#ifndef HAVE_CUFFT
1426 throw_nogpu();
1427#else
1428 StaticAssert<sizeof(float) == sizeof(cufftReal)>::check();
1429 StaticAssert<sizeof(float) * 2 == sizeof(cufftComplex)>::check();
1430
1431 CV_Assert(image.type() == CV_32F);
1432 CV_Assert(templ.type() == CV_32F);
1433
1434 buf.create(image.size(), templ.size());
1435 result.create(buf.result_size, CV_32F);
1436
1437 Size& block_size = buf.block_size;
1438 Size& dft_size = buf.dft_size;
1439
1440 GpuMat& image_block = buf.image_block;
1441 GpuMat& templ_block = buf.templ_block;
1442 GpuMat& result_data = buf.result_data;
1443
1444 GpuMat& image_spect = buf.image_spect;
1445 GpuMat& templ_spect = buf.templ_spect;
1446 GpuMat& result_spect = buf.result_spect;
1447
1448 cufftHandle planR2C, planC2R;
1449 cufftSafeCall(cufftPlan2d(&planC2R, dft_size.height, dft_size.width, CUFFT_C2R));
1450 cufftSafeCall(cufftPlan2d(&planR2C, dft_size.height, dft_size.width, CUFFT_R2C));
1451
1452 cufftSafeCall( cufftSetStream(planR2C, StreamAccessor::getStream(stream)) );
1453 cufftSafeCall( cufftSetStream(planC2R, StreamAccessor::getStream(stream)) );
1454
1455 GpuMat templ_roi(templ.size(), CV_32F, templ.data, templ.step);
1456 copyMakeBorder(templ_roi, templ_block, 0, templ_block.rows - templ_roi.rows, 0,
1457 templ_block.cols - templ_roi.cols, 0, Scalar(), stream);
1458
1459 cufftSafeCall(cufftExecR2C(planR2C, templ_block.ptr<cufftReal>(),
1460 templ_spect.ptr<cufftComplex>()));
1461
1462 // Process all blocks of the result matrix
1463 for (int y = 0; y < result.rows; y += block_size.height)
1464 {
1465 for (int x = 0; x < result.cols; x += block_size.width)
1466 {
1467 Size image_roi_size(std::min(x + dft_size.width, image.cols) - x,
1468 std::min(y + dft_size.height, image.rows) - y);
1469 GpuMat image_roi(image_roi_size, CV_32F, (void*)(image.ptr<float>(y) + x),
1470 image.step);
1471 copyMakeBorder(image_roi, image_block, 0, image_block.rows - image_roi.rows,
1472 0, image_block.cols - image_roi.cols, 0, Scalar(), stream);
1473
1474 cufftSafeCall(cufftExecR2C(planR2C, image_block.ptr<cufftReal>(),
1475 image_spect.ptr<cufftComplex>()));
1476 mulAndScaleSpectrums(image_spect, templ_spect, result_spect, 0,
1477 1.f / dft_size.area(), ccorr, stream);
1478 cufftSafeCall(cufftExecC2R(planC2R, result_spect.ptr<cufftComplex>(),
1479 result_data.ptr<cufftReal>()));
1480
1481 Size result_roi_size(std::min(x + block_size.width, result.cols) - x,
1482 std::min(y + block_size.height, result.rows) - y);
1483 GpuMat result_roi(result_roi_size, result.type(),
1484 (void*)(result.ptr<float>(y) + x), result.step);
1485 GpuMat result_block(result_roi_size, result_data.type(),
1486 result_data.ptr(), result_data.step);
1487
1488 if (stream)
1489 stream.enqueueCopy(result_block, result_roi);
1490 else
1491 result_block.copyTo(result_roi);
1492 }
1493 }
1494
1495 cufftSafeCall(cufftDestroy(planR2C));
1496 cufftSafeCall(cufftDestroy(planC2R));
1497#endif
1498}
