68void cv::gpu::blendLinear(const GpuMat& img1, const GpuMat& img2, const GpuMat& weights1, const GpuMat& weights2,
69 GpuMat& result, Stream& stream)
70{
71 CV_Assert(img1.size() == img2.size());
72 CV_Assert(img1.type() == img2.type());
73 CV_Assert(weights1.size() == img1.size());
74 CV_Assert(weights2.size() == img2.size());
75 CV_Assert(weights1.type() == CV_32F);
76 CV_Assert(weights2.type() == CV_32F);
77
78 const Size size = img1.size();
79 const int depth = img1.depth();
80 const int cn = img1.channels();
81
82 result.create(size, CV_MAKE_TYPE(depth, cn));
83
84 switch (depth)
85 {
86 case CV_8U:
87 if (cn != 4)
88 blendLinearCaller<uchar>(size.height, size.width, cn, img1, img2, weights1, weights2, result, StreamAccessor::getStream(stream));
89 else
90 blendLinearCaller8UC4(size.height, size.width, img1, img2, weights1, weights2, result, StreamAccessor::getStream(stream));
91 break;
92 case CV_32F:
93 blendLinearCaller<float>(size.height, size.width, cn, img1, img2, weights1, weights2, result, StreamAccessor::getStream(stream));
94 break;
95 default:
96 CV_Error(CV_StsUnsupportedFormat, "bad image depth in linear blending function");
97 }
98}
