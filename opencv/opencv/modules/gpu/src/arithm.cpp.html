70void cv::gpu::gemm(const GpuMat& src1, const GpuMat& src2, double alpha, const GpuMat& src3, double beta, GpuMat& dst, int flags, Stream& stream)
71{
72#ifndef HAVE_CUBLAS
73 (void)src1;
74 (void)src2;
75 (void)alpha;
76 (void)src3;
77 (void)beta;
78 (void)dst;
79 (void)flags;
80 (void)stream;
81 CV_Error(CV_StsNotImplemented, "The library was build without CUBLAS");
82#else
83 // CUBLAS works with column-major matrices
84
85 CV_Assert(src1.type() == CV_32FC1 || src1.type() == CV_32FC2 || src1.type() == CV_64FC1 || src1.type() == CV_64FC2);
86 CV_Assert(src2.type() == src1.type() && (src3.empty() || src3.type() == src1.type()));
87
88 if (src1.depth() == CV_64F)
89 {
90 if (!deviceSupports(NATIVE_DOUBLE))
91 CV_Error(CV_StsUnsupportedFormat, "The device doesn't support double");
92 }
93
94 bool tr1 = (flags & GEMM_1_T) != 0;
95 bool tr2 = (flags & GEMM_2_T) != 0;
96 bool tr3 = (flags & GEMM_3_T) != 0;
97
98 if (src1.type() == CV_64FC2)
99 {
100 if (tr1 || tr2 || tr3)
101 CV_Error(CV_StsNotImplemented, "transpose operation doesn't implemented for CV_64FC2 type");
102 }
103
104 Size src1Size = tr1 ? Size(src1.rows, src1.cols) : src1.size();
105 Size src2Size = tr2 ? Size(src2.rows, src2.cols) : src2.size();
106 Size src3Size = tr3 ? Size(src3.rows, src3.cols) : src3.size();
107 Size dstSize(src2Size.width, src1Size.height);
108
109 CV_Assert(src1Size.width == src2Size.height);
110 CV_Assert(src3.empty() || src3Size == dstSize);
111
112 dst.create(dstSize, src1.type());
113
114 if (beta != 0)
115 {
116 if (src3.empty())
117 {
118 if (stream)
119 stream.enqueueMemSet(dst, Scalar::all(0));
120 else
121 dst.setTo(Scalar::all(0));
122 }
123 else
124 {
125 if (tr3)
126 {
127 transpose(src3, dst, stream);
128 }
129 else
130 {
131 if (stream)
132 stream.enqueueCopy(src3, dst);
133 else
134 src3.copyTo(dst);
135 }
136 }
137 }
138
139 cublasHandle_t handle;
140 cublasSafeCall( cublasCreate_v2(&handle) );
141
142 cublasSafeCall( cublasSetStream_v2(handle, StreamAccessor::getStream(stream)) );
143
144 cublasSafeCall( cublasSetPointerMode_v2(handle, CUBLAS_POINTER_MODE_HOST) );
145
146 const float alphaf = static_cast<float>(alpha);
147 const float betaf = static_cast<float>(beta);
148
149 const cuComplex alphacf = make_cuComplex(alphaf, 0);
150 const cuComplex betacf = make_cuComplex(betaf, 0);
151
152 const cuDoubleComplex alphac = make_cuDoubleComplex(alpha, 0);
153 const cuDoubleComplex betac = make_cuDoubleComplex(beta, 0);
154
155 cublasOperation_t transa = tr2 ? CUBLAS_OP_T : CUBLAS_OP_N;
156 cublasOperation_t transb = tr1 ? CUBLAS_OP_T : CUBLAS_OP_N;
157
158 switch (src1.type())
159 {
160 case CV_32FC1:
161 cublasSafeCall( cublasSgemm_v2(handle, transa, transb, tr2 ? src2.rows : src2.cols, tr1 ? src1.cols : src1.rows, tr2 ? src2.cols : src2.rows,
162 &alphaf,
163 src2.ptr<float>(), static_cast<int>(src2.step / sizeof(float)),
164 src1.ptr<float>(), static_cast<int>(src1.step / sizeof(float)),
165 &betaf,
166 dst.ptr<float>(), static_cast<int>(dst.step / sizeof(float))) );
167 break;
168
169 case CV_64FC1:
170 cublasSafeCall( cublasDgemm_v2(handle, transa, transb, tr2 ? src2.rows : src2.cols, tr1 ? src1.cols : src1.rows, tr2 ? src2.cols : src2.rows,
171 &alpha,
172 src2.ptr<double>(), static_cast<int>(src2.step / sizeof(double)),
173 src1.ptr<double>(), static_cast<int>(src1.step / sizeof(double)),
174 &beta,
175 dst.ptr<double>(), static_cast<int>(dst.step / sizeof(double))) );
176 break;
177
178 case CV_32FC2:
179 cublasSafeCall( cublasCgemm_v2(handle, transa, transb, tr2 ? src2.rows : src2.cols, tr1 ? src1.cols : src1.rows, tr2 ? src2.cols : src2.rows,
180 &alphacf,
181 src2.ptr<cuComplex>(), static_cast<int>(src2.step / sizeof(cuComplex)),
182 src1.ptr<cuComplex>(), static_cast<int>(src1.step / sizeof(cuComplex)),
183 &betacf,
184 dst.ptr<cuComplex>(), static_cast<int>(dst.step / sizeof(cuComplex))) );
185 break;
186
187 case CV_64FC2:
188 cublasSafeCall( cublasZgemm_v2(handle, transa, transb, tr2 ? src2.rows : src2.cols, tr1 ? src1.cols : src1.rows, tr2 ? src2.cols : src2.rows,
189 &alphac,
190 src2.ptr<cuDoubleComplex>(), static_cast<int>(src2.step / sizeof(cuDoubleComplex)),
191 src1.ptr<cuDoubleComplex>(), static_cast<int>(src1.step / sizeof(cuDoubleComplex)),
192 &betac,
193 dst.ptr<cuDoubleComplex>(), static_cast<int>(dst.step / sizeof(cuDoubleComplex))) );
194 break;
195 }
196
197 cublasSafeCall( cublasDestroy_v2(handle) );
198#endif
199}

204void cv::gpu::transpose(const GpuMat& src, GpuMat& dst, Stream& s)
205{
206 CV_Assert(src.elemSize() == 1 || src.elemSize() == 4 || src.elemSize() == 8);
207
208 dst.create( src.cols, src.rows, src.type() );
209
210 cudaStream_t stream = StreamAccessor::getStream(s);
211
212 if (src.elemSize() == 1)
213 {
214 NppStreamHandler h(stream);
215
216 NppiSize sz;
217 sz.width = src.cols;
218 sz.height = src.rows;
219
220 nppSafeCall( nppiTranspose_8u_C1R(src.ptr<Npp8u>(), static_cast<int>(src.step),
221 dst.ptr<Npp8u>(), static_cast<int>(dst.step), sz) );
222 }
223 else if (src.elemSize() == 4)
224 {
225 NppStStreamHandler h(stream);
226
227 NcvSize32u sz;
228 sz.width = src.cols;
229 sz.height = src.rows;
230
231 ncvSafeCall( nppiStTranspose_32u_C1R(const_cast<Ncv32u*>(src.ptr<Ncv32u>()), static_cast<int>(src.step),
232 dst.ptr<Ncv32u>(), static_cast<int>(dst.step), sz) );
233 }
234 else // if (src.elemSize() == 8)
235 {
236 if (!deviceSupports(NATIVE_DOUBLE))
237 CV_Error(CV_StsUnsupportedFormat, "The device doesn't support double");
238
239 NppStStreamHandler h(stream);
240
241 NcvSize32u sz;
242 sz.width = src.cols;
243 sz.height = src.rows;
244
245 ncvSafeCall( nppiStTranspose_64u_C1R(const_cast<Ncv64u*>(src.ptr<Ncv64u>()), static_cast<int>(src.step),
246 dst.ptr<Ncv64u>(), static_cast<int>(dst.step), sz) );
247 }
248
249 if (stream == 0)
250 cudaSafeCall( cudaDeviceSynchronize() );
251}

254// flip
255
256namespace
257{
258 template<int DEPTH> struct NppTypeTraits;
259 template<> struct NppTypeTraits<CV_8U> { typedef Npp8u npp_t; };
260 template<> struct NppTypeTraits<CV_8S> { typedef Npp8s npp_t; };
261 template<> struct NppTypeTraits<CV_16U> { typedef Npp16u npp_t; };
262 template<> struct NppTypeTraits<CV_16S> { typedef Npp16s npp_t; };
263 template<> struct NppTypeTraits<CV_32S> { typedef Npp32s npp_t; };
264 template<> struct NppTypeTraits<CV_32F> { typedef Npp32f npp_t; };
265 template<> struct NppTypeTraits<CV_64F> { typedef Npp64f npp_t; };
266
267 template <int DEPTH> struct NppMirrorFunc
268 {
269 typedef typename NppTypeTraits<DEPTH>::npp_t npp_t;
270
271 typedef NppStatus (*func_t)(const npp_t* pSrc, int nSrcStep, npp_t* pDst, int nDstStep, NppiSize oROI, NppiAxis flip);
272 };
273
274 template <int DEPTH, typename NppMirrorFunc<DEPTH>::func_t func> struct NppMirror
275 {
276 typedef typename NppMirrorFunc<DEPTH>::npp_t npp_t;
277
278 static void call(const GpuMat& src, GpuMat& dst, int flipCode, cudaStream_t stream)
279 {
280 NppStreamHandler h(stream);
281
282 NppiSize sz;
283 sz.width = src.cols;
284 sz.height = src.rows;
285
286 nppSafeCall( func(src.ptr<npp_t>(), static_cast<int>(src.step),
287 dst.ptr<npp_t>(), static_cast<int>(dst.step), sz,
288 (flipCode == 0 ? NPP_HORIZONTAL_AXIS : (flipCode > 0 ? NPP_VERTICAL_AXIS : NPP_BOTH_AXIS))) );
289
290 if (stream == 0)
291 cudaSafeCall( cudaDeviceSynchronize() );
292 }
293 };
294}

296void cv::gpu::flip(const GpuMat& src, GpuMat& dst, int flipCode, Stream& stream)
297{
298 typedef void (*func_t)(const GpuMat& src, GpuMat& dst, int flipCode, cudaStream_t stream);
299 static const func_t funcs[6][4] =
300 {
301 {NppMirror<CV_8U, nppiMirror_8u_C1R>::call, 0, NppMirror<CV_8U, nppiMirror_8u_C3R>::call, NppMirror<CV_8U, nppiMirror_8u_C4R>::call},
302 {0,0,0,0},
303 {NppMirror<CV_16U, nppiMirror_16u_C1R>::call, 0, NppMirror<CV_16U, nppiMirror_16u_C3R>::call, NppMirror<CV_16U, nppiMirror_16u_C4R>::call},
304 {0,0,0,0},
305 {NppMirror<CV_32S, nppiMirror_32s_C1R>::call, 0, NppMirror<CV_32S, nppiMirror_32s_C3R>::call, NppMirror<CV_32S, nppiMirror_32s_C4R>::call},
306 {NppMirror<CV_32F, nppiMirror_32f_C1R>::call, 0, NppMirror<CV_32F, nppiMirror_32f_C3R>::call, NppMirror<CV_32F, nppiMirror_32f_C4R>::call}
307 };
308
309 CV_Assert(src.depth() == CV_8U || src.depth() == CV_16U || src.depth() == CV_32S || src.depth() == CV_32F);
310 CV_Assert(src.channels() == 1 || src.channels() == 3 || src.channels() == 4);
311
312 dst.create(src.size(), src.type());
313
314 funcs[src.depth()][src.channels() - 1](src, dst, flipCode, StreamAccessor::getStream(stream));
315}

324
325void cv::gpu::LUT(const GpuMat& src, const Mat& lut, GpuMat& dst, Stream& s)
326{
327 const int cn = src.channels();
328
329 CV_Assert( src.type() == CV_8UC1 || src.type() == CV_8UC3 );
330 CV_Assert( lut.depth() == CV_8U );
331 CV_Assert( lut.channels() == 1 || lut.channels() == cn );
332 CV_Assert( lut.rows * lut.cols == 256 && lut.isContinuous() );
333
334 dst.create(src.size(), CV_MAKE_TYPE(lut.depth(), cn));
335
336 GpuMat d_lut;
337 d_lut.upload(Mat(1, 256, lut.type(), lut.data));
338
339 int lut_cn = d_lut.channels();
340 bool cc30 = deviceSupports(FEATURE_SET_COMPUTE_30);
341 cudaStream_t stream = StreamAccessor::getStream(s);
342
343 if (lut_cn == 1)
344 {
345 arithm::lut(src.reshape(1), d_lut.data, lut_cn, dst.reshape(1), cc30, stream);
346 }
347 else if (lut_cn == 3)
348 {
349 arithm::lut(src, d_lut.data, lut_cn, dst, cc30, stream);
350 }
351}

403 inline void cartToPolar_caller(const GpuMat& x, const GpuMat& y, GpuMat* mag, bool magSqr, GpuMat* angle, bool angleInDegrees, cudaStream_t stream)
404 {
405 using namespace ::cv::gpu::device::mathfunc;
406
407 CV_Assert(x.size() == y.size() && x.type() == y.type());
408 CV_Assert(x.depth() == CV_32F);
409
410 if (mag)
411 mag->create(x.size(), x.type());
412 if (angle)
413 angle->create(x.size(), x.type());
414
415 GpuMat x1cn = x.reshape(1);
416 GpuMat y1cn = y.reshape(1);
417 GpuMat mag1cn = mag ? mag->reshape(1) : GpuMat();
418 GpuMat angle1cn = angle ? angle->reshape(1) : GpuMat();
419
420 cartToPolar_gpu(x1cn, y1cn, mag1cn, magSqr, angle1cn, angleInDegrees, stream);
421 }
422
423 inline void polarToCart_caller(const GpuMat& mag, const GpuMat& angle, GpuMat& x, GpuMat& y, bool angleInDegrees, cudaStream_t stream)
424 {
425 using namespace ::cv::gpu::device::mathfunc;
426
427 CV_Assert((mag.empty() || mag.size() == angle.size()) && mag.type() == angle.type());
428 CV_Assert(mag.depth() == CV_32F);
429
430 x.create(mag.size(), mag.type());
431 y.create(mag.size(), mag.type());
432
433 GpuMat mag1cn = mag.reshape(1);
434 GpuMat angle1cn = angle.reshape(1);
435 GpuMat x1cn = x.reshape(1);
436 GpuMat y1cn = y.reshape(1);
437
438 polarToCart_gpu(mag1cn, angle1cn, x1cn, y1cn, angleInDegrees, stream);
439 }

476
477void cv::gpu::normalize(const GpuMat& src, GpuMat& dst, double a, double b, int norm_type, int dtype, const GpuMat& mask, GpuMat& norm_buf, GpuMat& cvt_buf)
478{
479 double scale = 1, shift = 0;
480 if (norm_type == NORM_MINMAX)
481 {
482 double smin = 0, smax = 0;
483 double dmin = std::min(a, b), dmax = std::max(a, b);
484 minMax(src, &smin, &smax, mask, norm_buf);
485 scale = (dmax - dmin) * (smax - smin > numeric_limits<double>::epsilon() ? 1.0 / (smax - smin) : 0.0);
486 shift = dmin - smin * scale;
487 }
488 else if (norm_type == NORM_L2 || norm_type == NORM_L1 || norm_type == NORM_INF)
489 {
490 scale = norm(src, norm_type, mask, norm_buf);
491 scale = scale > numeric_limits<double>::epsilon() ? a / scale : 0.0;
492 shift = 0;
493 }
494 else
495 {
496 CV_Error(CV_StsBadArg, "Unknown/unsupported norm type");
497 }
498
499 if (mask.empty())
500 {
501 src.convertTo(dst, dtype, scale, shift);
502 }
503 else
504 {
505 src.convertTo(cvt_buf, dtype, scale, shift);
506 cvt_buf.copyTo(dst, mask);
507 }
508}
