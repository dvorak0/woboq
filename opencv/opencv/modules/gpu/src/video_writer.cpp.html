198void cv::gpu::VideoWriter_GPU::Impl::initEncoder(double fps)
199{
200 int err;
201
202 // Set codec
203
204 static const unsigned long codecs_id[] =
205 {
206 NV_CODEC_TYPE_MPEG1, NV_CODEC_TYPE_MPEG2, NV_CODEC_TYPE_MPEG4, NV_CODEC_TYPE_H264, NV_CODEC_TYPE_VC1
207 };
208 err = NVSetCodec(encoder_, codecs_id[codec_]);
209 if (err)
210 CV_Error(CV_StsNotImplemented, "Codec format is not supported");
211
212 // Set default params
213
214 err = NVSetDefaultParam(encoder_);
215 CV_Assert( err == 0 );
216
217 // Set some common params
218
219 int inputSize[] = { frameSize_.width, frameSize_.height };
220 err = NVSetParamValue(encoder_, NVVE_IN_SIZE, &inputSize);
221 CV_Assert( err == 0 );
222 err = NVSetParamValue(encoder_, NVVE_OUT_SIZE, &inputSize);
223 CV_Assert( err == 0 );
224
225 int aspectRatio[] = { frameSize_.width, frameSize_.height, ASPECT_RATIO_DAR };
226 err = NVSetParamValue(encoder_, NVVE_ASPECT_RATIO, &aspectRatio);
227 CV_Assert( err == 0 );
228
229 // FPS
230
231 int frame_rate = static_cast<int>(fps + 0.5);
232 int frame_rate_base = 1;
233 while (fabs(static_cast<double>(frame_rate) / frame_rate_base) - fps > 0.001)
234 {
235 frame_rate_base *= 10;
236 frame_rate = static_cast<int>(fps*frame_rate_base + 0.5);
237 }
238 int FrameRate[] = { frame_rate, frame_rate_base };
239 err = NVSetParamValue(encoder_, NVVE_FRAME_RATE, &FrameRate);
240 CV_Assert( err == 0 );
241
242 // Select device for encoding
243
244 int gpuID = cv::gpu::getDevice();
245 err = NVSetParamValue(encoder_, NVVE_FORCE_GPU_SELECTION, &gpuID);
246 CV_Assert( err == 0 );
247}

434void cv::gpu::VideoWriter_GPU::Impl::initGpuMemory()
435{
436 int err;
437 CUresult cuRes;
438
439 // initialize context
440 cv::gpu::GpuMat temp(1, 1, CV_8U);
441 temp.release();
442
443 static const int bpp[] =
444 {
445 16, // UYVY, 4:2:2
446 16, // YUY2, 4:2:2
447 12, // YV12, 4:2:0
448 12, // NV12, 4:2:0
449 12, // IYUV, 4:2:0
450 };
451
452 CUcontext cuContext;
453 cuRes = cuCtxGetCurrent(&cuContext);
454 CV_Assert( cuRes == CUDA_SUCCESS );
455
456 // Allocate the CUDA memory Pitched Surface
457 if (surfaceFormat_ == UYVY || surfaceFormat_ == YUY2)
458 videoFrame_.create(frameSize_.height, (frameSize_.width * bpp[surfaceFormat_]) / 8, CV_8UC1);
459 else
460 videoFrame_.create((frameSize_.height * bpp[surfaceFormat_]) / 8, frameSize_.width, CV_8UC1);
461
462 // Create the Video Context Lock (used for synchronization)
463 cuRes = cuvidCtxLockCreate(&cuCtxLock_, cuContext);
464 CV_Assert( cuRes == CUDA_SUCCESS );
465
466 // If we are using GPU Device Memory with NVCUVENC, it is necessary to create a
467 // CUDA Context with a Context Lock cuvidCtxLock. The Context Lock needs to be passed to NVCUVENC
468
469 int iUseDeviceMem = 1;
470 err = NVSetParamValue(encoder_, NVVE_DEVICE_MEMORY_INPUT, &iUseDeviceMem);
471 CV_Assert ( err == 0 );
472
473 err = NVSetParamValue(encoder_, NVVE_DEVICE_CTX_LOCK, &cuCtxLock_);
474 CV_Assert ( err == 0 );
475}

509 // UYVY/YUY2 are both 4:2:2 formats (16bpc)
510 // Luma, U, V are interleaved, chroma is subsampled (w/2,h)
511 void copyUYVYorYUY2Frame(cv::Size frameSize, const cv::gpu::GpuMat& src, cv::gpu::GpuMat& dst)
512 {
513 CUresult res;
514
515 // Source is YUVY/YUY2 4:2:2, the YUV data in a packed and interleaved
516
517 // YUV Copy setup
518 CUDA_MEMCPY2D stCopyYUV422;
519 memset((void*)&stCopyYUV422, 0, sizeof(stCopyYUV422));
520 stCopyYUV422.srcXInBytes = 0;
521 stCopyYUV422.srcY = 0;
522 stCopyYUV422.srcMemoryType = CU_MEMORYTYPE_DEVICE;
523 stCopyYUV422.srcHost = 0;
524 stCopyYUV422.srcDevice = (CUdeviceptr) src.data;
525 stCopyYUV422.srcArray = 0;
526 stCopyYUV422.srcPitch = src.step;
527
528 stCopyYUV422.dstXInBytes = 0;
529 stCopyYUV422.dstY = 0;
530 stCopyYUV422.dstMemoryType = CU_MEMORYTYPE_DEVICE;
531 stCopyYUV422.dstHost = 0;
532 stCopyYUV422.dstDevice = (CUdeviceptr) dst.data;
533 stCopyYUV422.dstArray = 0;
534 stCopyYUV422.dstPitch = dst.step;
535
536 stCopyYUV422.WidthInBytes = frameSize.width * 2;
537 stCopyYUV422.Height = frameSize.height;
538
539 // DMA Luma/Chroma
540 res = cuMemcpy2D(&stCopyYUV422);
541 CV_Assert( res == CUDA_SUCCESS );
542 }

544 // YV12/IYUV are both 4:2:0 planar formats (12bpc)
545 // Luma, U, V chroma planar (12bpc), chroma is subsampled (w/2,h/2)
546 void copyYV12orIYUVFrame(cv::Size frameSize, const cv::gpu::GpuMat& src, cv::gpu::GpuMat& dst)
547 {
548 CUresult res;
549
550 // Source is YV12/IYUV, this native format is converted to NV12 format by the video encoder
551
552 // (1) luma copy setup
553 CUDA_MEMCPY2D stCopyLuma;
554 memset((void*)&stCopyLuma, 0, sizeof(stCopyLuma));
555 stCopyLuma.srcXInBytes = 0;
556 stCopyLuma.srcY = 0;
557 stCopyLuma.srcMemoryType = CU_MEMORYTYPE_DEVICE;
558 stCopyLuma.srcHost = 0;
559 stCopyLuma.srcDevice = (CUdeviceptr) src.data;
560 stCopyLuma.srcArray = 0;
561 stCopyLuma.srcPitch = src.step;
562
563 stCopyLuma.dstXInBytes = 0;
564 stCopyLuma.dstY = 0;
565 stCopyLuma.dstMemoryType = CU_MEMORYTYPE_DEVICE;
566 stCopyLuma.dstHost = 0;
567 stCopyLuma.dstDevice = (CUdeviceptr) dst.data;
568 stCopyLuma.dstArray = 0;
569 stCopyLuma.dstPitch = dst.step;
570
571 stCopyLuma.WidthInBytes = frameSize.width;
572 stCopyLuma.Height = frameSize.height;
573
574 // (2) chroma copy setup, U/V can be done together
575 CUDA_MEMCPY2D stCopyChroma;
576 memset((void*)&stCopyChroma, 0, sizeof(stCopyChroma));
577 stCopyChroma.srcXInBytes = 0;
578 stCopyChroma.srcY = frameSize.height << 1; // U/V chroma offset
579 stCopyChroma.srcMemoryType = CU_MEMORYTYPE_DEVICE;
580 stCopyChroma.srcHost = 0;
581 stCopyChroma.srcDevice = (CUdeviceptr) src.data;
582 stCopyChroma.srcArray = 0;
583 stCopyChroma.srcPitch = src.step >> 1; // chroma is subsampled by 2 (but it has U/V are next to each other)
584
585 stCopyChroma.dstXInBytes = 0;
586 stCopyChroma.dstY = frameSize.height << 1; // chroma offset (srcY*srcPitch now points to the chroma planes)
587 stCopyChroma.dstMemoryType = CU_MEMORYTYPE_DEVICE;
588 stCopyChroma.dstHost = 0;
589 stCopyChroma.dstDevice = (CUdeviceptr) dst.data;
590 stCopyChroma.dstArray = 0;
591 stCopyChroma.dstPitch = dst.step >> 1;
592
593 stCopyChroma.WidthInBytes = frameSize.width >> 1;
594 stCopyChroma.Height = frameSize.height; // U/V are sent together
595
596 // DMA Luma
597 res = cuMemcpy2D(&stCopyLuma);
598 CV_Assert( res == CUDA_SUCCESS );
599
600 // DMA Chroma channels (UV side by side)
601 res = cuMemcpy2D(&stCopyChroma);
602 CV_Assert( res == CUDA_SUCCESS );
603 }

605 // NV12 is 4:2:0 format (12bpc)
606 // Luma followed by U/V chroma interleaved (12bpc), chroma is subsampled (w/2,h/2)
607 void copyNV12Frame(cv::Size frameSize, const cv::gpu::GpuMat& src, cv::gpu::GpuMat& dst)
608 {
609 CUresult res;
610
611 // Source is NV12 in pitch linear memory
612 // Because we are assume input is NV12 (if we take input in the native format), the encoder handles NV12 as a native format in pitch linear memory
613
614 // Luma/Chroma can be done in a single transfer
615 CUDA_MEMCPY2D stCopyNV12;
616 memset((void*)&stCopyNV12, 0, sizeof(stCopyNV12));
617 stCopyNV12.srcXInBytes = 0;
618 stCopyNV12.srcY = 0;
619 stCopyNV12.srcMemoryType = CU_MEMORYTYPE_DEVICE;
620 stCopyNV12.srcHost = 0;
621 stCopyNV12.srcDevice = (CUdeviceptr) src.data;
622 stCopyNV12.srcArray = 0;
623 stCopyNV12.srcPitch = src.step;
624
625 stCopyNV12.dstXInBytes = 0;
626 stCopyNV12.dstY = 0;
627 stCopyNV12.dstMemoryType = CU_MEMORYTYPE_DEVICE;
628 stCopyNV12.dstHost = 0;
629 stCopyNV12.dstDevice = (CUdeviceptr) dst.data;
630 stCopyNV12.dstArray = 0;
631 stCopyNV12.dstPitch = dst.step;
632
633 stCopyNV12.WidthInBytes = frameSize.width;
634 stCopyNV12.Height =(frameSize.height * 3) >> 1;
635
636 // DMA Luma/Chroma
637 res = cuMemcpy2D(&stCopyNV12);
638 CV_Assert( res == CUDA_SUCCESS );
639 }
640}

642void cv::gpu::VideoWriter_GPU::Impl::write(const cv::gpu::GpuMat& frame, bool lastFrame)
643{
644 if (inputFormat_ == SF_BGR)
645 {
646 CV_Assert( frame.size() == frameSize_ );
647 CV_Assert( frame.type() == CV_8UC1 || frame.type() == CV_8UC3 || frame.type() == CV_8UC4 );
648 }
649 else
650 {
651 CV_Assert( frame.size() == videoFrame_.size() );
652 CV_Assert( frame.type() == videoFrame_.type() );
653 }
654
655 NVVE_EncodeFrameParams efparams;
656 efparams.Width = frameSize_.width;
657 efparams.Height = frameSize_.height;
658 efparams.Pitch = static_cast<int>(videoFrame_.step);
659 efparams.SurfFmt = surfaceFormat_;
660 efparams.PictureStruc = FRAME_PICTURE;
661 efparams.topfieldfirst = 0;
662 efparams.repeatFirstField = 0;
663 efparams.progressiveFrame = (surfaceFormat_ == NV12) ? 1 : 0;
664 efparams.bLast = lastFrame;
665 efparams.picBuf = 0; // Must be set to NULL in order to support device memory input
666
667 // Don't forget we need to lock/unlock between memcopies
668 CUresult res = cuvidCtxLock(cuCtxLock_, 0);
669 CV_Assert( res == CUDA_SUCCESS );
670
671 if (inputFormat_ == SF_BGR)
672 cv::gpu::device::video_encoding::YV12_gpu(frame, frame.channels(), videoFrame_);
673 else
674 {
675 switch (surfaceFormat_)
676 {
677 case UYVY: // UYVY (4:2:2)
678 case YUY2: // YUY2 (4:2:2)
679 copyUYVYorYUY2Frame(frameSize_, frame, videoFrame_);
680 break;
681
682 case YV12: // YV12 (4:2:0), Y V U
683 case IYUV: // IYUV (4:2:0), Y U V
684 copyYV12orIYUVFrame(frameSize_, frame, videoFrame_);
685 break;
686
687 case NV12: // NV12 (4:2:0)
688 copyNV12Frame(frameSize_, frame, videoFrame_);
689 break;
690 }
691 }
692
693 res = cuvidCtxUnlock(cuCtxLock_, 0);
694 CV_Assert( res == CUDA_SUCCESS );
695
696 int err = NVEncodeFrame(encoder_, &efparams, 0, videoFrame_.data);
697 CV_Assert( err == 0 );
698}

758 {
759 static bool initialized = 0;
760
761 if (!initialized)
762 {
763 #if defined WIN32 || defined _WIN32
764 const char* module_name = "opencv_ffmpeg"
765 CVAUX_STR(CV_VERSION_EPOCH) CVAUX_STR(CV_VERSION_MAJOR) CVAUX_STR(CV_VERSION_MINOR)

768 #endif
769 ".dll";

771 static HMODULE cvFFOpenCV = LoadLibrary(module_name);
772
773 if (cvFFOpenCV)

781
782 initialized = create_OutputMediaStream_FFMPEG_p != 0 && release_OutputMediaStream_FFMPEG_p != 0 && write_OutputMediaStream_FFMPEG_p != 0;
783 }
784 #elif defined HAVE_FFMPEG
785 create_OutputMediaStream_FFMPEG_p = create_OutputMediaStream_FFMPEG;

787 write_OutputMediaStream_FFMPEG_p = write_OutputMediaStream_FFMPEG;

791 }

793 return initialized;
794 }
