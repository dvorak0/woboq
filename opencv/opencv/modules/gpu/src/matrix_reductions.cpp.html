77 class DeviceBuffer
78 {
79 public:
80 explicit DeviceBuffer(int count_ = 1) : count(count_)
81 {
82 cudaSafeCall( cudaMalloc(&pdev, count * sizeof(double)) );
83 }
84 ~DeviceBuffer()
85 {
86 cudaSafeCall( cudaFree(pdev) );
87 }
88
89 operator double*() {return pdev;}
90
91 void download(double* hptr)
92 {
93 double hbuf;
94 cudaSafeCall( cudaMemcpy(&hbuf, pdev, sizeof(double), cudaMemcpyDeviceToHost) );
95 *hptr = hbuf;
96 }
97 void download(double** hptrs)
98 {
99 AutoBuffer<double, 2 * sizeof(double)> hbuf(count);
100 cudaSafeCall( cudaMemcpy((void*)hbuf, pdev, count * sizeof(double), cudaMemcpyDeviceToHost) );
101 for (int i = 0; i < count; ++i)
102 *hptrs[i] = hbuf[i];
103 }
104
105 private:
106 double* pdev;
107 int count;
108 };
109}

121void cv::gpu::meanStdDev(const GpuMat& src, Scalar& mean, Scalar& stddev, GpuMat& buf)
122{
123 CV_Assert(src.type() == CV_8UC1);
124
125 if (!deviceSupports(FEATURE_SET_COMPUTE_13))
126 CV_Error(CV_StsNotImplemented, "Not sufficient compute capebility");
127
128 NppiSize sz;
129 sz.width = src.cols;
130 sz.height = src.rows;
131
132 DeviceBuffer dbuf(2);
133
134 int bufSize;
135#if (CUDART_VERSION <= 4020)
136 nppSafeCall( nppiMeanStdDev8uC1RGetBufferHostSize(sz, &bufSize) );
137#else
138 nppSafeCall( nppiMeanStdDevGetBufferHostSize_8u_C1R(sz, &bufSize) );
139#endif
140
141 ensureSizeIsEnough(1, bufSize, CV_8UC1, buf);
142
143 nppSafeCall( nppiMean_StdDev_8u_C1R(src.ptr<Npp8u>(), static_cast<int>(src.step), sz, buf.ptr<Npp8u>(), dbuf, (double*)dbuf + 1) );
144
145 cudaSafeCall( cudaDeviceSynchronize() );
146
147 double* ptrs[2] = {mean.val, stddev.val};
148 dbuf.download(ptrs);
149}

165double cv::gpu::norm(const GpuMat& src, int normType, const GpuMat& mask, GpuMat& buf)
166{
167 CV_Assert(normType == NORM_INF || normType == NORM_L1 || normType == NORM_L2);
168 CV_Assert(mask.empty() || (mask.type() == CV_8UC1 && mask.size() == src.size() && src.channels() == 1));
169
170 GpuMat src_single_channel = src.reshape(1);
171
172 if (normType == NORM_L1)
173 return absSum(src_single_channel, mask, buf)[0];
174
175 if (normType == NORM_L2)
176 return std::sqrt(sqrSum(src_single_channel, mask, buf)[0]);
177
178 // NORM_INF
179 double min_val, max_val;
180 minMax(src_single_channel, &min_val, &max_val, mask, buf);
181 return std::max(std::abs(min_val), std::abs(max_val));
182}

184double cv::gpu::norm(const GpuMat& src1, const GpuMat& src2, int normType)
185{
186 CV_Assert(src1.type() == CV_8UC1);
187 CV_Assert(src1.size() == src2.size() && src1.type() == src2.type());
188 CV_Assert(normType == NORM_INF || normType == NORM_L1 || normType == NORM_L2);
189
190#if CUDART_VERSION < 5050
191 typedef NppStatus (*func_t)(const Npp8u* pSrc1, int nSrcStep1, const Npp8u* pSrc2, int nSrcStep2, NppiSize oSizeROI, Npp64f* pRetVal);
192
193 static const func_t funcs[] = {nppiNormDiff_Inf_8u_C1R, nppiNormDiff_L1_8u_C1R, nppiNormDiff_L2_8u_C1R};
194#else
195 typedef NppStatus (*func_t)(const Npp8u* pSrc1, int nSrcStep1, const Npp8u* pSrc2, int nSrcStep2,
196 NppiSize oSizeROI, Npp64f* pRetVal, Npp8u * pDeviceBuffer);
197
198 typedef NppStatus (*buf_size_func_t)(NppiSize oSizeROI, int* hpBufferSize);
199
200 static const func_t funcs[] = {nppiNormDiff_Inf_8u_C1R, nppiNormDiff_L1_8u_C1R, nppiNormDiff_L2_8u_C1R};
201
202 static const buf_size_func_t buf_size_funcs[] = {nppiNormDiffInfGetBufferHostSize_8u_C1R, nppiNormDiffL1GetBufferHostSize_8u_C1R, nppiNormDiffL2GetBufferHostSize_8u_C1R};
203#endif
204
205 NppiSize sz;
206 sz.width = src1.cols;
207 sz.height = src1.rows;
208
209 int funcIdx = normType >> 1;
210
211 double retVal;
212
213 DeviceBuffer dbuf;
214
215#if CUDART_VERSION < 5050
216 nppSafeCall( funcs[funcIdx](src1.ptr<Npp8u>(), static_cast<int>(src1.step), src2.ptr<Npp8u>(), static_cast<int>(src2.step), sz, dbuf) );
217#else
218 int bufSize;
219 buf_size_funcs[funcIdx](sz, &bufSize);
220
221 GpuMat buf(1, bufSize, CV_8UC1);
222
223 nppSafeCall( funcs[funcIdx](src1.ptr<Npp8u>(), static_cast<int>(src1.step), src2.ptr<Npp8u>(), static_cast<int>(src2.step), sz, dbuf, buf.data) );
224#endif
225
226 cudaSafeCall( cudaDeviceSynchronize() );
227
228 dbuf.download(&retVal);
229
230 return retVal;
231}
