73cv::gpu::StereoBM_GPU::StereoBM_GPU()

78cv::gpu::StereoBM_GPU::StereoBM_GPU(int preset_, int ndisparities_, int winSize_)
79 : preset(preset_), ndisp(ndisparities_), winSize(winSize_), avergeTexThreshold(defaultAvgTexThreshold)
80{
81 const int max_supported_ndisp = 1 << (sizeof(unsigned char) * 8);
82 CV_Assert(0 < ndisp && ndisp <= max_supported_ndisp);
83 CV_Assert(ndisp % 8 == 0);
84 CV_Assert(winSize % 2 == 1);
85}

87bool cv::gpu::StereoBM_GPU::checkIfGpuCallReasonable()
88{
89 if (0 == getCudaEnabledDeviceCount())
90 return false;
91
92 DeviceInfo device_info;
93
94 if (device_info.majorVersion() > 1 || device_info.multiProcessorCount() > 16)
95 return true;
96
97 return false;
98}

100namespace

102 void stereo_bm_gpu_operator( GpuMat& minSSD, GpuMat& leBuf, GpuMat& riBuf, int preset, int ndisp, int winSize, float avergeTexThreshold, const GpuMat& left, const GpuMat& right, GpuMat& disparity, cudaStream_t stream)
103 {
104 using namespace ::cv::gpu::device::stereobm;
105
106 CV_Assert(left.rows == right.rows && left.cols == right.cols);
107 CV_Assert(left.type() == CV_8UC1);
108 CV_Assert(right.type() == CV_8UC1);
109
110 disparity.create(left.size(), CV_8U);
111 minSSD.create(left.size(), CV_32S);
112
113 GpuMat le_for_bm = left;
114 GpuMat ri_for_bm = right;
115
116 if (preset == StereoBM_GPU::PREFILTER_XSOBEL)
117 {
118 leBuf.create( left.size(), left.type());
119 riBuf.create(right.size(), right.type());
120
121 prefilter_xsobel( left, leBuf, 31, stream);
122 prefilter_xsobel(right, riBuf, 31, stream);
123
124 le_for_bm = leBuf;
125 ri_for_bm = riBuf;
126 }
127
128 stereoBM_GPU(le_for_bm, ri_for_bm, disparity, ndisp, winSize, minSSD, stream);
129
130 if (avergeTexThreshold)
131 postfilter_textureness(le_for_bm, winSize, avergeTexThreshold, disparity, stream);
132 }
133}

135void cv::gpu::StereoBM_GPU::operator() ( const GpuMat& left, const GpuMat& right, GpuMat& disparity, Stream& stream)
136{
137 stereo_bm_gpu_operator(minSSD, leBuf, riBuf, preset, ndisp, winSize, avergeTexThreshold, left, right, disparity, StreamAccessor::getStream(stream));
138}
