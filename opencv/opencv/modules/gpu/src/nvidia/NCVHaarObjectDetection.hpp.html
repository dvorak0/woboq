82struct HaarFeature64
83{
84 uint2 _ui2;
85
86#define HaarFeature64_CreateCheck_MaxRectField 0xFF
87
88 __host__ NCVStatus setRect(Ncv32u rectX, Ncv32u rectY, Ncv32u rectWidth, Ncv32u rectHeight, Ncv32u /*clsWidth*/, Ncv32u /*clsHeight*/)
89 {
90 ncvAssertReturn(rectWidth <= HaarFeature64_CreateCheck_MaxRectField && rectHeight <= HaarFeature64_CreateCheck_MaxRectField, NCV_HAAR_TOO_LARGE_FEATURES);
91 ((NcvRect8u*)&(this->_ui2.x))->x = (Ncv8u)rectX;
92 ((NcvRect8u*)&(this->_ui2.x))->y = (Ncv8u)rectY;
93 ((NcvRect8u*)&(this->_ui2.x))->width = (Ncv8u)rectWidth;
94 ((NcvRect8u*)&(this->_ui2.x))->height = (Ncv8u)rectHeight;
95 return NCV_SUCCESS;
96 }
97
98 __host__ NCVStatus setWeight(Ncv32f weight)
99 {
100 ((Ncv32f_a*)&(this->_ui2.y))[0] = weight;
101 return NCV_SUCCESS;
102 }
103
104 __device__ __host__ void getRect(Ncv32u *rectX, Ncv32u *rectY, Ncv32u *rectWidth, Ncv32u *rectHeight)
105 {
106 NcvRect8u tmpRect = *(NcvRect8u*)(&this->_ui2.x);
107 *rectX = tmpRect.x;
108 *rectY = tmpRect.y;
109 *rectWidth = tmpRect.width;
110 *rectHeight = tmpRect.height;
111 }
112
113 __device__ __host__ Ncv32f getWeight(void)
114 {
115 return *(Ncv32f_a*)(&this->_ui2.y);
116 }
117};

120struct HaarFeatureDescriptor32
121{
122private:
123
124#define HaarFeatureDescriptor32_Interpret_MaskFlagTilted 0x80000000
125#define HaarFeatureDescriptor32_Interpret_MaskFlagLeftNodeLeaf 0x40000000
126#define HaarFeatureDescriptor32_Interpret_MaskFlagRightNodeLeaf 0x20000000
127#define HaarFeatureDescriptor32_CreateCheck_MaxNumFeatures 0x1F
128#define HaarFeatureDescriptor32_NumFeatures_Shift 24
129#define HaarFeatureDescriptor32_CreateCheck_MaxFeatureOffset 0x00FFFFFF
130
131 Ncv32u desc;
132
133public:
134
135 __host__ NCVStatus create(NcvBool bTilted, NcvBool bLeftLeaf, NcvBool bRightLeaf,
136 Ncv32u numFeatures, Ncv32u offsetFeatures)
137 {
138 if (numFeatures > HaarFeatureDescriptor32_CreateCheck_MaxNumFeatures)
139 {
140 return NCV_HAAR_TOO_MANY_FEATURES_IN_CLASSIFIER;
141 }
142 if (offsetFeatures > HaarFeatureDescriptor32_CreateCheck_MaxFeatureOffset)
143 {
144 return NCV_HAAR_TOO_MANY_FEATURES_IN_CASCADE;
145 }
146 this->desc = 0;
147 this->desc |= (bTilted ? HaarFeatureDescriptor32_Interpret_MaskFlagTilted : 0);
148 this->desc |= (bLeftLeaf ? HaarFeatureDescriptor32_Interpret_MaskFlagLeftNodeLeaf : 0);
149 this->desc |= (bRightLeaf ? HaarFeatureDescriptor32_Interpret_MaskFlagRightNodeLeaf : 0);
150 this->desc |= (numFeatures << HaarFeatureDescriptor32_NumFeatures_Shift);
151 this->desc |= offsetFeatures;
152 return NCV_SUCCESS;
153 }
154
155 __device__ __host__ NcvBool isTilted(void)
156 {
157 return (this->desc & HaarFeatureDescriptor32_Interpret_MaskFlagTilted) != 0;
158 }
159
160 __device__ __host__ NcvBool isLeftNodeLeaf(void)
161 {
162 return (this->desc & HaarFeatureDescriptor32_Interpret_MaskFlagLeftNodeLeaf) != 0;
163 }
164
165 __device__ __host__ NcvBool isRightNodeLeaf(void)
166 {
167 return (this->desc & HaarFeatureDescriptor32_Interpret_MaskFlagRightNodeLeaf) != 0;
168 }
169
170 __device__ __host__ Ncv32u getNumFeatures(void)
171 {
172 return (this->desc >> HaarFeatureDescriptor32_NumFeatures_Shift) & HaarFeatureDescriptor32_CreateCheck_MaxNumFeatures;
173 }
174
175 __device__ __host__ Ncv32u getFeaturesOffset(void)
176 {
177 return this->desc & HaarFeatureDescriptor32_CreateCheck_MaxFeatureOffset;
178 }
179};

271struct HaarStage64
272{
273#define HaarStage64_Interpret_MaskRootNodes 0x0000FFFF
274#define HaarStage64_Interpret_MaskRootNodeOffset 0xFFFF0000
275#define HaarStage64_Interpret_ShiftRootNodeOffset 16
276
277 uint2 _ui2;
278
279 __host__ NCVStatus setStageThreshold(Ncv32f t)
280 {
281 this->_ui2.x = *(Ncv32u_a *)&t;
282 return NCV_SUCCESS;
283 }
284
285 __host__ NCVStatus setStartClassifierRootNodeOffset(Ncv32u val)
286 {
287 if (val > (HaarStage64_Interpret_MaskRootNodeOffset >> HaarStage64_Interpret_ShiftRootNodeOffset))
288 {
289 return NCV_HAAR_XML_LOADING_EXCEPTION;
290 }
291 this->_ui2.y = (val << HaarStage64_Interpret_ShiftRootNodeOffset) | (this->_ui2.y & HaarStage64_Interpret_MaskRootNodes);
292 return NCV_SUCCESS;
293 }
294
295 __host__ NCVStatus setNumClassifierRootNodes(Ncv32u val)
296 {
297 if (val > HaarStage64_Interpret_MaskRootNodes)
298 {
299 return NCV_HAAR_XML_LOADING_EXCEPTION;
300 }
301 this->_ui2.y = val | (this->_ui2.y & HaarStage64_Interpret_MaskRootNodeOffset);
302 return NCV_SUCCESS;
303 }
304
305 __host__ __device__ Ncv32f getStageThreshold(void)
306 {
307 return *(Ncv32f_a*)&this->_ui2.x;
308 }
309
310 __host__ __device__ Ncv32u getStartClassifierRootNodeOffset(void)
311 {
312 return (this->_ui2.y >> HaarStage64_Interpret_ShiftRootNodeOffset);
313 }
314
315 __host__ __device__ Ncv32u getNumClassifierRootNodes(void)
316 {
317 return (this->_ui2.y & HaarStage64_Interpret_MaskRootNodes);
318 }
319};
