88/** Border type
89 *
90 * Filtering operations assume that each pixel has a neighborhood of pixels.
91 * The following structure describes possible ways to define non-existent pixels.
92 */
93enum NppStBorderType
94{
95 nppStBorderNone = 0, ///< There is no need to define additional pixels, image is extended already
96 nppStBorderClamp = 1, ///< Clamp out of range position to borders
97 nppStBorderWrap = 2, ///< Wrap out of range position. Image becomes periodic.
98 nppStBorderMirror = 3 ///< reflect out of range position across borders
99};

112/** Frame interpolation state
113 *
114 * This structure holds parameters required for frame interpolation.
115 * Forward displacement field is a per-pixel mapping from frame 0 to frame 1.
116 * Backward displacement field is a per-pixel mapping from frame 1 to frame 0.
117 */
118
119 struct NppStInterpolationState
120{
121 NcvSize32u size; ///< frame size
122 Ncv32u nStep; ///< pitch
123 Ncv32f pos; ///< new frame position
124 Ncv32f *pSrcFrame0; ///< frame 0
125 Ncv32f *pSrcFrame1; ///< frame 1
126 Ncv32f *pFU; ///< forward horizontal displacement
127 Ncv32f *pFV; ///< forward vertical displacement
128 Ncv32f *pBU; ///< backward horizontal displacement
129 Ncv32f *pBV; ///< backward vertical displacement
130 Ncv32f *pNewFrame; ///< new frame
131 Ncv32f *ppBuffers[6]; ///< temporary buffers
132};

242/** Warp image using provided 2D vector field and 1x1 point spread function.
243 * 32-bit floating point image, single channel
244 *
245 * During warping pixels from the source image may fall between pixels of the destination image.
246 * PSF (point spread function) describes how the source image pixel affects pixels of the destination.
247 * For 1x1 PSF only single pixel with the largest intersection is affected (similar to nearest interpolation).
248 *
249 * Destination image size and line step must be the same as the source image size and line step
250 *
251 * \param pSrc [IN] Source image pointer (CUDA device memory)
252 * \param srcSize [IN] Source image size
253 * \param nSrcStep [IN] Source image line step
254 * \param pU [IN] Pointer to horizontal displacement field (CUDA device memory)
255 * \param pV [IN] Pointer to vertical displacement field (CUDA device memory)
256 * \param nVFStep [IN] Displacement field line step
257 * \param timeScale [IN] Value by which displacement field will be scaled for warping
258 * \param pDst [OUT] Destination image pointer (CUDA device memory)
259 *
260 * \return NCV status code
261 */
262NCV_EXPORTS
263NCVStatus nppiStVectorWarp_PSF1x1_32f_C1(const Ncv32f *pSrc,
264 NcvSize32u srcSize,
265 Ncv32u nSrcStep,
266 const Ncv32f *pU,
267 const Ncv32f *pV,
268 Ncv32u nVFStep,
269 Ncv32f timeScale,
270 Ncv32f *pDst);
