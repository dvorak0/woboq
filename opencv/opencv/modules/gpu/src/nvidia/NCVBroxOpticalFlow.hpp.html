47// Algorithm is explained in the original paper:
48// T. Brox, A. Bruhn, N. Papenberg, J. Weickert:
49// High accuracy optical flow estimation based on a theory for warping.

60#ifndef _ncv_optical_flow_h_
61#define _ncv_optical_flow_h_
62
63#include "NCV.hpp"
64
65/// \brief Model and solver parameters
66struct NCVBroxOpticalFlowDescriptor
67{
68 /// flow smoothness
69 Ncv32f alpha;
70 /// gradient constancy importance
71 Ncv32f gamma;
72 /// pyramid scale factor
73 Ncv32f scale_factor;
74 /// number of lagged non-linearity iterations (inner loop)
75 Ncv32u number_of_inner_iterations;
76 /// number of warping iterations (number of pyramid levels)
77 Ncv32u number_of_outer_iterations;
78 /// number of linear system solver iterations
79 Ncv32u number_of_solver_iterations;
80};
81
82/////////////////////////////////////////////////////////////////////////////////////////
83/// \brief Compute optical flow
84///
85/// Based on method by Brox et al [2004]
86/// \param [in] desc model and solver parameters
87/// \param [in] gpu_mem_allocator GPU memory allocator
88/// \param [in] frame0 source frame
89/// \param [in] frame1 frame to track
90/// \param [out] u flow horizontal component (along \b x axis)
91/// \param [out] v flow vertical component (along \b y axis)
92/// \return computation status
93/////////////////////////////////////////////////////////////////////////////////////////
94
95NCV_EXPORTS
96NCVStatus NCVBroxOpticalFlow(const NCVBroxOpticalFlowDescriptor desc,
97 INCVMemAllocator &gpu_mem_allocator,
98 const NCVMatrix<Ncv32f> &frame0,
99 const NCVMatrix<Ncv32f> &frame1,
100 NCVMatrix<Ncv32f> &u,
101 NCVMatrix<Ncv32f> &v,
102 cudaStream_t stream);
103
104#endif
