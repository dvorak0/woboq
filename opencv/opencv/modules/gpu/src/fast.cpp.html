76void cv::gpu::FAST_GPU::downloadKeypoints(const GpuMat& d_keypoints, std::vector<KeyPoint>& keypoints)

79 return;

82 convertKeypoints(h_keypoints, keypoints);

85void cv::gpu::FAST_GPU::convertKeypoints(const Mat& h_keypoints, std::vector<KeyPoint>& keypoints)
86{
87 if (h_keypoints.empty())
88 return;
89
90 CV_Assert(h_keypoints.rows == ROWS_COUNT && h_keypoints.elemSize() == 4);
91
92 int npoints = h_keypoints.cols;
93
94 keypoints.resize(npoints);
95
96 const short2* loc_row = h_keypoints.ptr<short2>(LOCATION_ROW);
97 const float* response_row = h_keypoints.ptr<float>(RESPONSE_ROW);
98
99 for (int i = 0; i < npoints; ++i)
100 {
101 KeyPoint kp(loc_row[i].x, loc_row[i].y, static_cast<float>(FEATURE_SIZE), -1, response_row[i]);
102 keypoints[i] = kp;
103 }
104}

106void cv::gpu::FAST_GPU::operator ()(const GpuMat& img, const GpuMat& mask, GpuMat& keypoints)
107{
108 calcKeyPointsLocation(img, mask);
109 keypoints.cols = getKeyPoints(keypoints);
110}

121int cv::gpu::FAST_GPU::calcKeyPointsLocation(const GpuMat& img, const GpuMat& mask)
122{
123 using namespace cv::gpu::device::fast;
124
125 CV_Assert(img.type() == CV_8UC1);
126 CV_Assert(mask.empty() || (mask.type() == CV_8UC1 && mask.size() == img.size()));
127
128 int maxKeypoints = static_cast<int>(keypointsRatio * img.size().area());
129
130 ensureSizeIsEnough(1, maxKeypoints, CV_16SC2, kpLoc_);
131
132 if (nonmaxSuppression)
133 {
134 ensureSizeIsEnough(img.size(), CV_32SC1, score_);
135 score_.setTo(Scalar::all(0));
136 }
137
138 count_ = calcKeypoints_gpu(img, mask, kpLoc_.ptr<short2>(), maxKeypoints, nonmaxSuppression ? score_ : PtrStepSzi(), threshold);
139 count_ = std::min(count_, maxKeypoints);
140
141 return count_;
142}
143
144int cv::gpu::FAST_GPU::getKeyPoints(GpuMat& keypoints)
145{
146 using namespace cv::gpu::device::fast;
147
148 if (count_ == 0)
149 return 0;
150
151 ensureSizeIsEnough(ROWS_COUNT, count_, CV_32FC1, keypoints);
152
153 if (nonmaxSuppression)
154 return nonmaxSuppression_gpu(kpLoc_.ptr<short2>(), count_, score_, keypoints.ptr<short2>(LOCATION_ROW), keypoints.ptr<float>(RESPONSE_ROW));
155
156 GpuMat locRow(1, count_, kpLoc_.type(), keypoints.ptr(0));
157 kpLoc_.colRange(0, count_).copyTo(locRow);
158 keypoints.row(1).setTo(Scalar::all(0));
159
160 return count_;
161}
