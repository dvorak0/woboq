64void cv::gpu::GoodFeaturesToTrackDetector_GPU::operator ()(const GpuMat& image, GpuMat& corners, const GpuMat& mask)
65{
66 using namespace cv::gpu::device::gfft;
67
68 CV_Assert(qualityLevel > 0 && minDistance >= 0 && maxCorners >= 0);
69 CV_Assert(mask.empty() || (mask.type() == CV_8UC1 && mask.size() == image.size()));
70
71 ensureSizeIsEnough(image.size(), CV_32F, eig_);
72
73 if (useHarrisDetector)
74 cornerHarris(image, eig_, Dx_, Dy_, buf_, blockSize, 3, harrisK);
75 else
76 cornerMinEigenVal(image, eig_, Dx_, Dy_, buf_, blockSize, 3);
77
78 double maxVal = 0;
79 minMax(eig_, 0, &maxVal, GpuMat(), minMaxbuf_);
80
81 ensureSizeIsEnough(1, std::max(1000, static_cast<int>(image.size().area() * 0.05)), CV_32FC2, tmpCorners_);
82
83 int total = findCorners_gpu(eig_, static_cast<float>(maxVal * qualityLevel), mask, tmpCorners_.ptr<float2>(), tmpCorners_.cols);
84
85 if (total == 0)
86 {
87 corners.release();
88 return;
89 }
90
91 sortCorners_gpu(eig_, tmpCorners_.ptr<float2>(), total);
92
93 if (minDistance < 1)
94 tmpCorners_.colRange(0, maxCorners > 0 ? std::min(maxCorners, total) : total).copyTo(corners);
95 else
96 {
97 vector<Point2f> tmp(total);
98 Mat tmpMat(1, total, CV_32FC2, (void*)&tmp[0]);
99 tmpCorners_.colRange(0, total).download(tmpMat);
100
101 vector<Point2f> tmp2;
102 tmp2.reserve(total);
103
104 const int cell_size = cvRound(minDistance);
105 const int grid_width = (image.cols + cell_size - 1) / cell_size;
106 const int grid_height = (image.rows + cell_size - 1) / cell_size;
107
108 std::vector< std::vector<Point2f> > grid(grid_width * grid_height);
109
110 for (int i = 0; i < total; ++i)
111 {
112 Point2f p = tmp[i];
113
114 bool good = true;
115
116 int x_cell = static_cast<int>(p.x / cell_size);
117 int y_cell = static_cast<int>(p.y / cell_size);
118
119 int x1 = x_cell - 1;
120 int y1 = y_cell - 1;
121 int x2 = x_cell + 1;
122 int y2 = y_cell + 1;
123
124 // boundary check
125 x1 = std::max(0, x1);
126 y1 = std::max(0, y1);
127 x2 = std::min(grid_width - 1, x2);
128 y2 = std::min(grid_height - 1, y2);
129
130 for (int yy = y1; yy <= y2; yy++)
131 {
132 for (int xx = x1; xx <= x2; xx++)
133 {
134 vector<Point2f>& m = grid[yy * grid_width + xx];
135
136 if (!m.empty())
137 {
138 for(size_t j = 0; j < m.size(); j++)
139 {
140 float dx = p.x - m[j].x;
141 float dy = p.y - m[j].y;
142
143 if (dx * dx + dy * dy < minDistance * minDistance)
144 {
145 good = false;
146 goto break_out;
147 }
148 }
149 }
150 }
151 }
152
153 break_out:
154
155 if(good)
156 {
157 grid[y_cell * grid_width + x_cell].push_back(p);
158
159 tmp2.push_back(p);
160
161 if (maxCorners > 0 && tmp2.size() == static_cast<size_t>(maxCorners))
162 break;
163 }
164 }
165
166 corners.upload(Mat(1, static_cast<int>(tmp2.size()), CV_32FC2, &tmp2[0]));
167 }
168}
