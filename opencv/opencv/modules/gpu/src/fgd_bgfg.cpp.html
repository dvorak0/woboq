111
112 cv::gpu::ensureSizeIsEnough(params.N2c * size.height, size.width, CV_8UC(out_cn), ctable_v_);

115 cv::gpu::ensureSizeIsEnough(params.N2cc * size.height, size.width, CV_32FC1, cctable_Pv_);

119 cctable_Pvb_.setTo(cv::Scalar::all(0));
120
121 cv::gpu::ensureSizeIsEnough(params.N2cc * size.height, size.width, CV_8UC(out_cn), cctable_v1_);

255 void copyChannels(const cv::gpu::GpuMat& src, cv::gpu::GpuMat& dst, int dst_cn = -1)
256 {
257 const int src_cn = src.channels();
258
259 if (dst_cn < 0)
260 dst_cn = src_cn;
261
262 cv::gpu::ensureSizeIsEnough(src.size(), CV_MAKE_TYPE(src.depth(), dst_cn), dst);
263
264 if (src_cn == dst_cn)
265 src.copyTo(dst);
266 else
267 {
268 static const int cvt_codes[4][4] =
269 {
270 {-1, -1, cv::COLOR_GRAY2BGR, cv::COLOR_GRAY2BGRA},
271 {-1, -1, -1, -1},
272 {cv::COLOR_BGR2GRAY, -1, -1, cv::COLOR_BGR2BGRA},
273 {cv::COLOR_BGRA2GRAY, -1, cv::COLOR_BGRA2BGR, -1}
274 };
275
276 const int cvt_code = cvt_codes[src_cn - 1][dst_cn - 1];
277 CV_DbgAssert( cvt_code >= 0 );
278
279 cv::gpu::cvtColor(src, dst, cvt_code, dst_cn);
280 }
281 }

284void cv::gpu::FGDStatModel::Impl::create(const cv::gpu::GpuMat& firstFrame, const cv::gpu::FGDStatModel::Params& params)
285{
286 CV_Assert(firstFrame.type() == CV_8UC3 || firstFrame.type() == CV_8UC4);
287
288 params_ = params;
289
290 cv::gpu::ensureSizeIsEnough(firstFrame.size(), CV_8UC1, foreground_);
291
292 copyChannels(firstFrame, background_, out_cn_);
293
294 copyChannels(firstFrame, prevFrame_);
295
296 cv::gpu::ensureSizeIsEnough(firstFrame.size(), CV_8UC1, Ftd_);
297 cv::gpu::ensureSizeIsEnough(firstFrame.size(), CV_8UC1, Fbd_);
298
299 stat_.create(firstFrame.size(), params_, out_cn_);
300 bgfg::setBGPixelStat(stat_);
301
302 if (params_.perform_morphing > 0)
303 {
304 cv::Mat kernel = cv::getStructuringElement(cv::MORPH_RECT, cv::Size(1 + params_.perform_morphing * 2, 1 + params_.perform_morphing * 2));
305 cv::Point anchor(params_.perform_morphing, params_.perform_morphing);
306
307 dilateFilter_ = cv::gpu::createMorphologyFilter_GPU(cv::MORPH_DILATE, CV_8UC1, kernel, filterBuf_, anchor);
308 erodeFilter_ = cv::gpu::createMorphologyFilter_GPU(cv::MORPH_ERODE, CV_8UC1, kernel, filterBuf_, anchor);
309 }
310}

337 void calcDiffHistogram(const cv::gpu::GpuMat& prevFrame, const cv::gpu::GpuMat& curFrame, cv::gpu::GpuMat& hist, cv::gpu::GpuMat& histBuf)
338 {
339 typedef void (*func_t)(cv::gpu::PtrStepSzb prevFrame, cv::gpu::PtrStepSzb curFrame, unsigned int* hist0, unsigned int* hist1, unsigned int* hist2, unsigned int* partialBuf0, unsigned int* partialBuf1, unsigned int* partialBuf2, bool cc20, cudaStream_t stream);
340 static const func_t funcs[4][4] =
341 {
342 {0,0,0,0},
343 {0,0,0,0},
344 {0,0,bgfg::calcDiffHistogram_gpu<uchar3, uchar3>,bgfg::calcDiffHistogram_gpu<uchar3, uchar4>},
345 {0,0,bgfg::calcDiffHistogram_gpu<uchar4, uchar3>,bgfg::calcDiffHistogram_gpu<uchar4, uchar4>}
346 };
347
348 hist.create(3, 256, CV_32SC1);
349 histBuf.create(3, bgfg::PARTIAL_HISTOGRAM_COUNT * bgfg::HISTOGRAM_BIN_COUNT, CV_32SC1);
350
351 funcs[prevFrame.channels() - 1][curFrame.channels() - 1](
352 prevFrame, curFrame,
353 hist.ptr<unsigned int>(0), hist.ptr<unsigned int>(1), hist.ptr<unsigned int>(2),
354 histBuf.ptr<unsigned int>(0), histBuf.ptr<unsigned int>(1), histBuf.ptr<unsigned int>(2),
355 cv::gpu::deviceSupports(cv::gpu::FEATURE_SET_COMPUTE_20), 0);
356 }
357
358 void calcRelativeVariance(unsigned int hist[3 * 256], double relativeVariance[3][bgfg::HISTOGRAM_BIN_COUNT])
359 {
360 std::memset(relativeVariance, 0, 3 * bgfg::HISTOGRAM_BIN_COUNT * sizeof(double));
361
362 for (int thres = bgfg::HISTOGRAM_BIN_COUNT - 2; thres >= 0; --thres)
363 {
364 cv::Vec3d sum(0.0, 0.0, 0.0);
365 cv::Vec3d sqsum(0.0, 0.0, 0.0);
366 cv::Vec3i count(0, 0, 0);
367
368 for (int j = thres; j < bgfg::HISTOGRAM_BIN_COUNT; ++j)
369 {
370 sum[0] += static_cast<double>(j) * hist[j];
371 sqsum[0] += static_cast<double>(j * j) * hist[j];
372 count[0] += hist[j];
373
374 sum[1] += static_cast<double>(j) * hist[j + 256];
375 sqsum[1] += static_cast<double>(j * j) * hist[j + 256];
376 count[1] += hist[j + 256];
377
378 sum[2] += static_cast<double>(j) * hist[j + 512];
379 sqsum[2] += static_cast<double>(j * j) * hist[j + 512];
380 count[2] += hist[j + 512];
381 }
382
383 count[0] = std::max(count[0], 1);
384 count[1] = std::max(count[1], 1);
385 count[2] = std::max(count[2], 1);
386
387 cv::Vec3d my(
388 sum[0] / count[0],
389 sum[1] / count[1],
390 sum[2] / count[2]
391 );
392
393 relativeVariance[0][thres] = std::sqrt(sqsum[0] / count[0] - my[0] * my[0]);
394 relativeVariance[1][thres] = std::sqrt(sqsum[1] / count[1] - my[1] * my[1]);
395 relativeVariance[2][thres] = std::sqrt(sqsum[2] / count[2] - my[2] * my[2]);
396 }
397 }

399 void calcDiffThreshMask(const cv::gpu::GpuMat& prevFrame, const cv::gpu::GpuMat& curFrame, cv::Vec3d bestThres, cv::gpu::GpuMat& changeMask)
400 {
401 typedef void (*func_t)(cv::gpu::PtrStepSzb prevFrame, cv::gpu::PtrStepSzb curFrame, uchar3 bestThres, cv::gpu::PtrStepSzb changeMask, cudaStream_t stream);
402 static const func_t funcs[4][4] =
403 {
404 {0,0,0,0},
405 {0,0,0,0},
406 {0,0,bgfg::calcDiffThreshMask_gpu<uchar3, uchar3>,bgfg::calcDiffThreshMask_gpu<uchar3, uchar4>},
407 {0,0,bgfg::calcDiffThreshMask_gpu<uchar4, uchar3>,bgfg::calcDiffThreshMask_gpu<uchar4, uchar4>}
408 };
409
410 changeMask.setTo(cv::Scalar::all(0));
411
412 funcs[prevFrame.channels() - 1][curFrame.channels() - 1](prevFrame, curFrame, make_uchar3((uchar)bestThres[0], (uchar)bestThres[1], (uchar)bestThres[2]), changeMask, 0);
413 }
414
415 // performs change detection for Foreground detection algorithm
416 void changeDetection(const cv::gpu::GpuMat& prevFrame, const cv::gpu::GpuMat& curFrame, cv::gpu::GpuMat& changeMask, cv::gpu::GpuMat& hist, cv::gpu::GpuMat& histBuf)
417 {
418 calcDiffHistogram(prevFrame, curFrame, hist, histBuf);
419
420 unsigned int histData[3 * 256];
421 cv::Mat h_hist(3, 256, CV_32SC1, histData);
422 hist.download(h_hist);
423
424 double relativeVariance[3][bgfg::HISTOGRAM_BIN_COUNT];
425 calcRelativeVariance(histData, relativeVariance);
426
427 // Find maximum:
428 cv::Vec3d bestThres(10.0, 10.0, 10.0);
429 for (int i = 0; i < bgfg::HISTOGRAM_BIN_COUNT; ++i)
430 {
431 bestThres[0] = std::max(bestThres[0], relativeVariance[0][i]);
432 bestThres[1] = std::max(bestThres[1], relativeVariance[1][i]);
433 bestThres[2] = std::max(bestThres[2], relativeVariance[2][i]);
434 }
435
436 calcDiffThreshMask(prevFrame, curFrame, bestThres, changeMask);
437 }
438}

468 {0,0,bgfg::bgfgClassification_gpu<uchar4, uchar4, uchar3>,bgfg::bgfgClassification_gpu<uchar4, uchar4, uchar4>}

470 };
471
472 const int deltaC = cvRound(params.delta * 256 / params.Lc);

480
481 return count;
482 }

520 void seqToContours(CvSeq* _ccontours, CvMemStorage* storage, cv::OutputArrayOfArrays _contours)
521 {
522 cv::Seq<CvSeq*> all_contours(cvTreeToNodeSeq(_ccontours, sizeof(CvSeq), storage));
523
524 size_t total = all_contours.size();
525
526 _contours.create((int) total, 1, 0, -1, true);
527
528 cv::SeqIterator<CvSeq*> it = all_contours.begin();
529 for (size_t i = 0; i < total; ++i, ++it)
530 {
531 CvSeq* c = *it;
532 ((CvContour*)c)->color = (int)i;
533 _contours.create((int)c->total, 1, CV_32SC2, (int)i, true);
534 cv::Mat ci = _contours.getMat((int)i);
535 CV_Assert( ci.isContinuous() );
536 cvCvtSeqToArray(c, ci.data);
537 }
538 }

540 int findForegroundRegions(cv::gpu::GpuMat& d_foreground, cv::Mat& h_foreground, std::vector< std::vector<cv::Point> >& foreground_regions,
541 CvMemStorage* storage, const cv::gpu::FGDStatModel::Params& params)
542 {
543 int region_count = 0;
544
545 // Discard under-size foreground regions:
546
547 d_foreground.download(h_foreground);
548 IplImage ipl_foreground = h_foreground;
549 CvSeq* first_seq = 0;
550
551 cvFindContours(&ipl_foreground, storage, &first_seq, sizeof(CvContour), CV_RETR_LIST);
552
553 for (CvSeq* seq = first_seq; seq; seq = seq->h_next)
554 {
555 CvContour* cnt = reinterpret_cast<CvContour*>(seq);
556
557 if (cnt->rect.width * cnt->rect.height < params.minArea || (params.is_obj_without_holes && CV_IS_SEQ_HOLE(seq)))
558 {
559 // Delete under-size contour:
560 CvSeq* prev_seq = seq->h_prev;
561 if (prev_seq)
562 {
563 prev_seq->h_next = seq->h_next;
564
565 if (seq->h_next)
566 seq->h_next->h_prev = prev_seq;
567 }
568 else
569 {
570 first_seq = seq->h_next;
571
572 if (seq->h_next)
573 seq->h_next->h_prev = NULL;
574 }
575 }
576 else
577 {
578 region_count++;
579 }
580 }
581
582 seqToContours(first_seq, storage, foreground_regions);
583 h_foreground.setTo(0);
584
585 cv::drawContours(h_foreground, foreground_regions, -1, cv::Scalar::all(255), -1);
586
587 d_foreground.upload(h_foreground);
588
589 return region_count;
590 }

635/////////////////////////////////////////////////////////////////////////
636// Impl::update
637
638int cv::gpu::FGDStatModel::Impl::update(const cv::gpu::GpuMat& curFrame)
639{
640 CV_Assert(curFrame.type() == CV_8UC3 || curFrame.type() == CV_8UC4);
641 CV_Assert(curFrame.size() == prevFrame_.size());
642
643 cvClearMemStorage(storage_);
644 foreground_regions_.clear();
645 foreground_.setTo(cv::Scalar::all(0));
646
647 changeDetection(prevFrame_, curFrame, Ftd_, hist_, histBuf_);
648 changeDetection(background_, curFrame, Fbd_, hist_, histBuf_);
649
650 int FG_pixels_count = bgfgClassification(prevFrame_, curFrame, Ftd_, Fbd_, foreground_, countBuf_, params_, out_cn_);
651
652 if (params_.perform_morphing > 0)
653 smoothForeground(foreground_, filterBrd_, buf_, erodeFilter_, dilateFilter_, params_);
654
655 int region_count = 0;
656 if (params_.minArea > 0 || params_.is_obj_without_holes)
657 region_count = findForegroundRegions(foreground_, h_foreground_, foreground_regions_, storage_, params_);
658
659 // Check ALL BG update condition:
660 const double BGFG_FGD_BG_UPDATE_TRESH = 0.5;
661 if (static_cast<double>(FG_pixels_count) / Ftd_.size().area() > BGFG_FGD_BG_UPDATE_TRESH)
662 stat_.setTrained();
663
664 updateBackgroundModel(prevFrame_, curFrame, Ftd_, Fbd_, foreground_, background_, params_);
665
666 copyChannels(curFrame, prevFrame_);
667
668 return region_count;
669}
