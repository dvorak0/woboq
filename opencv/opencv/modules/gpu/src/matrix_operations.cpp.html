89CudaMem& cv::gpu::CudaMem::operator = (const CudaMem& m)
90{
91 if( this != &m )
92 {
93 if( m.refcount )
94 CV_XADD(m.refcount, 1);
95 release();
96 flags = m.flags;
97 rows = m.rows; cols = m.cols;
98 step = m.step; data = m.data;
99 datastart = m.datastart;
100 dataend = m.dataend;
101 refcount = m.refcount;
102 alloc_type = m.alloc_type;
103 }
104 return *this;
105}

211 size_t alignUpStep(size_t what, size_t alignment)
212 {
213 size_t alignMask = alignment-1;
214 size_t inverseAlignMask = ~alignMask;
215 size_t res = (what + alignMask) & inverseAlignMask;
216 return res;
217 }

220void cv::gpu::CudaMem::create(int _rows, int _cols, int _type, int _alloc_type)
221{
222 if (_alloc_type == ALLOC_ZEROCOPY && !canMapHostMemory())
223 cv::gpu::error("ZeroCopy is not supported by current device", __FILE__, __LINE__);
224
225 _type &= TYPE_MASK;
226 if( rows == _rows && cols == _cols && type() == _type && data )
227 return;
228 if( data )
229 release();
230 CV_DbgAssert( _rows >= 0 && _cols >= 0 );
231 if( _rows > 0 && _cols > 0 )
232 {
233 flags = Mat::MAGIC_VAL + Mat::CONTINUOUS_FLAG + _type;
234 rows = _rows;
235 cols = _cols;
236 step = elemSize()*cols;
237 if (_alloc_type == ALLOC_ZEROCOPY)
238 {
239 cudaDeviceProp prop;
240 cudaSafeCall( cudaGetDeviceProperties(&prop, getDevice()) );
241 step = alignUpStep(step, prop.textureAlignment);
242 }
243 int64 _nettosize = (int64)step*rows;
244 size_t nettosize = (size_t)_nettosize;
245 if( _nettosize != (int64)nettosize )
246 CV_Error(CV_StsNoMem, "Too big buffer is allocated");
247 size_t datasize = alignSize(nettosize, (int)sizeof(*refcount));
248
249 //datastart = data = (uchar*)fastMalloc(datasize + sizeof(*refcount));
250 alloc_type = _alloc_type;
251 void *ptr;
252
253 switch (alloc_type)
254 {
255 case ALLOC_PAGE_LOCKED: cudaSafeCall( cudaHostAlloc( &ptr, datasize, cudaHostAllocDefault) ); break;
256 case ALLOC_ZEROCOPY: cudaSafeCall( cudaHostAlloc( &ptr, datasize, cudaHostAllocMapped) ); break;
257 case ALLOC_WRITE_COMBINED: cudaSafeCall( cudaHostAlloc( &ptr, datasize, cudaHostAllocWriteCombined) ); break;
258 default: cv::gpu::error("Invalid alloc type", __FILE__, __LINE__);
259 }
260
261 datastart = data = (uchar*)ptr;
262 dataend = data + nettosize;
263
264 refcount = (int*)cv::fastMalloc(sizeof(*refcount));
265 *refcount = 1;
266 }
267}

269GpuMat cv::gpu::CudaMem::createGpuMatHeader () const
270{
271 GpuMat res;
272 if (alloc_type == ALLOC_ZEROCOPY)
273 {
274 void *pdev;
275 cudaSafeCall( cudaHostGetDevicePointer( &pdev, data, 0 ) );
276 res = GpuMat(rows, cols, type(), pdev, step);
277 }
278 else
279 cv::gpu::error("Zero-copy is not supported or memory was allocated without zero-copy flag", __FILE__, __LINE__);
280
281 return res;
282}

284void cv::gpu::CudaMem::release()
285{
286 if( refcount && CV_XADD(refcount, -1) == 1 )
287 {
288 cudaSafeCall( cudaFreeHost(datastart ) );
289 fastFree(refcount);
290 }
291 data = datastart = dataend = 0;
292 step = rows = cols = 0;
293 refcount = 0;
294}
