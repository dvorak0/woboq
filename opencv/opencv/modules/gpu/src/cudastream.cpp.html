84struct Stream::Impl
85{
86 static cudaStream_t getStream(const Impl* impl)
87 {
88 return impl ? impl->stream : 0;

90
91 cudaStream_t stream;
92 bool own_stream;
93 int ref_counter;
94};

130bool cv::gpu::Stream::queryIfComplete()
131{
132 cudaStream_t stream = Impl::getStream(impl);
133 cudaError_t err = cudaStreamQuery(stream);
134
135 if (err == cudaErrorNotReady || err == cudaSuccess)
136 return err == cudaSuccess;
137
138 cudaSafeCall(err);
139 return false;
140}

148void cv::gpu::Stream::enqueueDownload(const GpuMat& src, Mat& dst)
149{
150 // if not -> allocation will be done, but after that dst will not point to page locked memory
151 CV_Assert( src.size() == dst.size() && src.type() == dst.type() );
152
153 cudaStream_t stream = Impl::getStream(impl);
154 size_t bwidth = src.cols * src.elemSize();
155 cudaSafeCall( cudaMemcpy2DAsync(dst.data, dst.step, src.data, src.step, bwidth, src.rows, cudaMemcpyDeviceToHost, stream) );
156}

194void cv::gpu::Stream::enqueueMemSet(GpuMat& src, Scalar val)
195{
196 const int sdepth = src.depth();
197
198 if (sdepth == CV_64F)
199 {
200 if (!deviceSupports(NATIVE_DOUBLE))
201 CV_Error(CV_StsUnsupportedFormat, "The device doesn't support double");
202 }
203
204 cudaStream_t stream = Impl::getStream(impl);
205
206 if (val[0] == 0.0 && val[1] == 0.0 && val[2] == 0.0 && val[3] == 0.0)
207 {
208 cudaSafeCall( cudaMemset2DAsync(src.data, src.step, 0, src.cols * src.elemSize(), src.rows, stream) );
209 return;
210 }
211
212 if (sdepth == CV_8U)
213 {
214 int cn = src.channels();
215
216 if (cn == 1 || (cn == 2 && val[0] == val[1]) || (cn == 3 && val[0] == val[1] && val[0] == val[2]) || (cn == 4 && val[0] == val[1] && val[0] == val[2] && val[0] == val[3]))
217 {
218 int ival = saturate_cast<uchar>(val[0]);
219 cudaSafeCall( cudaMemset2DAsync(src.data, src.step, ival, src.cols * src.elemSize(), src.rows, stream) );
220 return;
221 }
222 }
223
224 setTo(src, val, stream);
225}

227void cv::gpu::Stream::enqueueMemSet(GpuMat& src, Scalar val, const GpuMat& mask)
228{
229 const int sdepth = src.depth();
230
231 if (sdepth == CV_64F)
232 {
233 if (!deviceSupports(NATIVE_DOUBLE))
234 CV_Error(CV_StsUnsupportedFormat, "The device doesn't support double");
235 }
236
237 CV_Assert(mask.type() == CV_8UC1);
238
239 cudaStream_t stream = Impl::getStream(impl);
240
241 setTo(src, val, mask, stream);
242}

244void cv::gpu::Stream::enqueueConvert(const GpuMat& src, GpuMat& dst, int dtype, double alpha, double beta)
245{
246 if (dtype < 0)
247 dtype = src.type();
248 else
249 dtype = CV_MAKE_TYPE(CV_MAT_DEPTH(dtype), src.channels());
250
251 const int sdepth = src.depth();
252 const int ddepth = CV_MAT_DEPTH(dtype);
253
254 if (sdepth == CV_64F || ddepth == CV_64F)
255 {
256 if (!deviceSupports(NATIVE_DOUBLE))
257 CV_Error(CV_StsUnsupportedFormat, "The device doesn't support double");
258 }
259
260 bool noScale = fabs(alpha - 1) < numeric_limits<double>::epsilon() && fabs(beta) < numeric_limits<double>::epsilon();
261
262 if (sdepth == ddepth && noScale)
263 {
264 enqueueCopy(src, dst);
265 return;
266 }
267
268 dst.create(src.size(), dtype);
269
270 cudaStream_t stream = Impl::getStream(impl);
271 convertTo(src, dst, alpha, beta, stream);
272}

285 void CUDART_CB cudaStreamCallback(cudaStream_t, cudaError_t status, void* userData)
286 {
287 CallbackData* data = reinterpret_cast<CallbackData*>(userData);
288 data->callback(data->stream, static_cast<int>(status), data->userData);
289 delete data;
290 }

295void cv::gpu::Stream::enqueueHostCallback(StreamCallback callback, void* userData)
296{
297#if CUDART_VERSION >= 5000
298 CallbackData* data = new CallbackData;
299 data->callback = callback;
300 data->userData = userData;
301 data->stream = *this;
302
303 cudaStream_t stream = Impl::getStream(impl);
304
305 cudaSafeCall( cudaStreamAddCallback(stream, cudaStreamCallback, data, 0) );
306#else
307 (void) callback;
308 (void) userData;
309 CV_Error(CV_StsNotImplemented, "This function requires CUDA 5.0");
310#endif
311}

328void cv::gpu::Stream::create()
329{
330 if (impl)
331 release();
332
333 cudaStream_t stream;
334 cudaSafeCall( cudaStreamCreate( &stream ) );
335
336 impl = (Stream::Impl*) fastMalloc(sizeof(Stream::Impl));
337
338 impl->stream = stream;
339 impl->own_stream = true;
340 impl->ref_counter = 1;
341}
342
343void cv::gpu::Stream::release()
344{
345 if (impl && CV_XADD(&impl->ref_counter, -1) == 1)
346 {
347 if (impl->own_stream)
348 {
349 cudaSafeCall( cudaStreamDestroy(impl->stream) );
350 }
351 cv::fastFree(impl);
352 }
353}
354
355Stream StreamAccessor::wrapStream(cudaStream_t stream)
356{
357 Stream::Impl* impl = (Stream::Impl*) fastMalloc(sizeof(Stream::Impl));
358
359 impl->stream = stream;
360 impl->own_stream = false;
361 impl->ref_counter = 1;
362
363 return Stream(impl);
364}
