107cv::gpu::HOGDescriptor::HOGDescriptor(Size win_size_, Size block_size_, Size block_stride_, Size cell_size_,
108 int nbins_, double win_sigma_, double threshold_L2hys_, bool gamma_correction_, int nlevels_)
109 : win_size(win_size_),
110 block_size(block_size_),
111 block_stride(block_stride_),
112 cell_size(cell_size_),
113 nbins(nbins_),
114 win_sigma(win_sigma_),
115 threshold_L2hys(threshold_L2hys_),
116 gamma_correction(gamma_correction_),
117 nlevels(nlevels_)
118{
119 CV_Assert((win_size.width - block_size.width ) % block_stride.width == 0 &&
120 (win_size.height - block_size.height) % block_stride.height == 0);
121
122 CV_Assert(block_size.width % cell_size.width == 0 && block_size.height % cell_size.height == 0);
123
124 CV_Assert(block_stride == cell_size);
125
126 CV_Assert(cell_size == Size(8, 8));
127
128 Size cells_per_block = Size(block_size.width / cell_size.width, block_size.height / cell_size.height);
129 CV_Assert(cells_per_block == Size(2, 2));
130}

155void cv::gpu::HOGDescriptor::setSVMDetector(const vector<float>& _detector)
156{
157 std::vector<float> detector_reordered(_detector.size());
158
159 size_t block_hist_size = getBlockHistogramSize();
160 cv::Size blocks_per_img = numPartsWithin(win_size, block_size, block_stride);
161
162 for (int i = 0; i < blocks_per_img.height; ++i)
163 for (int j = 0; j < blocks_per_img.width; ++j)
164 {
165 const float* src = &_detector[0] + (j * blocks_per_img.height + i) * block_hist_size;
166 float* dst = &detector_reordered[0] + (i * blocks_per_img.width + j) * block_hist_size;
167 for (size_t k = 0; k < block_hist_size; ++k)
168 dst[k] = src[k];
169 }
170
171 this->detector.upload(Mat(detector_reordered).reshape(1, 1));
172
173 size_t descriptor_size = getDescriptorSize();
174 free_coef = _detector.size() > descriptor_size ? _detector[descriptor_size] : 0;
175
176 CV_Assert(checkDetectorSize());
177}

196void cv::gpu::HOGDescriptor::computeGradient(const GpuMat& img, GpuMat& _grad, GpuMat& _qangle)
197{
198 CV_Assert(img.type() == CV_8UC1 || img.type() == CV_8UC4);
199
200 // grad.create(img.size(), CV_32FC2);
201 _grad = getBuffer(img.size(), CV_32FC2, grad_buf);
202
203 // qangle.create(img.size(), CV_8UC2);
204 _qangle = getBuffer(img.size(), CV_8UC2, qangle_buf);
205
206 float angleScale = (float)(nbins / CV_PI);
207 switch (img.type())
208 {
209 case CV_8UC1:
210 hog::compute_gradients_8UC1(nbins, img.rows, img.cols, img, angleScale, _grad, _qangle, gamma_correction);
211 break;
212 case CV_8UC4:
213 hog::compute_gradients_8UC4(nbins, img.rows, img.cols, img, angleScale, _grad, _qangle, gamma_correction);
214 break;
215 }
216}

219void cv::gpu::HOGDescriptor::computeBlockHistograms(const GpuMat& img)
220{
221 cv::Size blocks_per_win = numPartsWithin(win_size, block_size, block_stride);
222 hog::set_up_constants(nbins, block_stride.width, block_stride.height, blocks_per_win.width, blocks_per_win.height);
223
224 computeGradient(img, grad, qangle);
225
226 size_t block_hist_size = getBlockHistogramSize();
227 Size blocks_per_img = numPartsWithin(img.size(), block_size, block_stride);
228
229 // block_hists.create(1, block_hist_size * blocks_per_img.area(), CV_32F);
230 block_hists = getBuffer(1, static_cast<int>(block_hist_size * blocks_per_img.area()), CV_32F, block_hists_buf);
231
232 hog::compute_hists(nbins, block_stride.width, block_stride.height, img.rows, img.cols,
233 grad, qangle, (float)getWinSigma(), block_hists.ptr<float>());
234
235 hog::normalize_hists(nbins, block_stride.width, block_stride.height, img.rows, img.cols,
236 block_hists.ptr<float>(), (float)threshold_L2hys);
237}

240void cv::gpu::HOGDescriptor::getDescriptors(const GpuMat& img, Size win_stride, GpuMat& descriptors, int descr_format)
241{
242 CV_Assert(win_stride.width % block_stride.width == 0 && win_stride.height % block_stride.height == 0);
243
244 computeBlockHistograms(img);
245
246 const size_t block_hist_size = getBlockHistogramSize();
247 Size blocks_per_win = numPartsWithin(win_size, block_size, block_stride);
248 Size wins_per_img = numPartsWithin(img.size(), win_size, win_stride);
249
250 descriptors.create(wins_per_img.area(), static_cast<int>(blocks_per_win.area() * block_hist_size), CV_32F);
251
252 switch (descr_format)
253 {
254 case DESCR_FORMAT_ROW_BY_ROW:
255 hog::extract_descrs_by_rows(win_size.height, win_size.width, block_stride.height, block_stride.width,
256 win_stride.height, win_stride.width, img.rows, img.cols, block_hists.ptr<float>(), descriptors);
257 break;
258 case DESCR_FORMAT_COL_BY_COL:
259 hog::extract_descrs_by_cols(win_size.height, win_size.width, block_stride.height, block_stride.width,
260 win_stride.height, win_stride.width, img.rows, img.cols, block_hists.ptr<float>(), descriptors);
261 break;
262 default:
263 CV_Error(CV_StsBadArg, "Unknown descriptor format");
264 }
265}

267void cv::gpu::HOGDescriptor::computeConfidence(const GpuMat& img, vector<Point>& hits, double hit_threshold,
268 Size win_stride, Size padding, vector<Point>& locations, vector<double>& confidences)
269{
270 CV_Assert(padding == Size(0, 0));
271
272 hits.clear();
273 if (detector.empty())
274 return;
275
276 computeBlockHistograms(img);
277
278 if (win_stride == Size())
279 win_stride = block_stride;
280 else
281 CV_Assert(win_stride.width % block_stride.width == 0 &&
282 win_stride.height % block_stride.height == 0);
283
284 Size wins_per_img = numPartsWithin(img.size(), win_size, win_stride);
285 labels.create(1, wins_per_img.area(), CV_32F);
286
287 hog::compute_confidence_hists(win_size.height, win_size.width, block_stride.height, block_stride.width,
288 win_stride.height, win_stride.width, img.rows, img.cols, block_hists.ptr<float>(),
289 detector.ptr<float>(), (float)free_coef, (float)hit_threshold, labels.ptr<float>());
290
291 labels.download(labels_host);
292 float* vec = labels_host.ptr<float>();
293
294 // does not support roi for now..
295 locations.clear();
296 confidences.clear();
297 for (int i = 0; i < wins_per_img.area(); i++)
298 {
299 int y = i / wins_per_img.width;
300 int x = i - wins_per_img.width * y;
301 if (vec[i] >= hit_threshold)
302 hits.push_back(Point(x * win_stride.width, y * win_stride.height));
303
304 Point pt(win_stride.width * x, win_stride.height * y);
305 locations.push_back(pt);
306 confidences.push_back((double)vec[i]);
307 }
308}

310void cv::gpu::HOGDescriptor::computeConfidenceMultiScale(const GpuMat& img, vector<Rect>& found_locations,
311 double hit_threshold, Size win_stride, Size padding,
312 vector<HOGConfidence> &conf_out, int group_threshold)
313{
314 vector<double> level_scale;
315 double scale = 1.;
316 int levels = 0;
317
318 for (levels = 0; levels < (int)conf_out.size(); levels++)
319 {
320 scale = conf_out[levels].scale;
321 level_scale.push_back(scale);
322 if (cvRound(img.cols/scale) < win_size.width || cvRound(img.rows/scale) < win_size.height)
323 break;
324 }
325
326 levels = std::max(levels, 1);
327 level_scale.resize(levels);
328
329 std::vector<Rect> all_candidates;
330 vector<Point> locations;
331
332 for (size_t i = 0; i < level_scale.size(); i++)
333 {
334 scale = level_scale[i];
335 Size sz(cvRound(img.cols / scale), cvRound(img.rows / scale));
336 GpuMat smaller_img;
337
338 if (sz == img.size())
339 smaller_img = img;
340 else
341 {
342 smaller_img.create(sz, img.type());
343 switch (img.type())
344 {
345 case CV_8UC1: hog::resize_8UC1(img, smaller_img); break;
346 case CV_8UC4: hog::resize_8UC4(img, smaller_img); break;
347 }
348 }
349
350 computeConfidence(smaller_img, locations, hit_threshold, win_stride, padding, conf_out[i].locations, conf_out[i].confidences);
351
352 Size scaled_win_size(cvRound(win_size.width * scale), cvRound(win_size.height * scale));
353 for (size_t j = 0; j < locations.size(); j++)
354 all_candidates.push_back(Rect(Point2d((CvPoint)locations[j]) * scale, scaled_win_size));
355 }
356
357 found_locations.assign(all_candidates.begin(), all_candidates.end());
358 groupRectangles(found_locations, group_threshold, 0.2/*magic number copied from CPU version*/);
359}

362void cv::gpu::HOGDescriptor::detect(const GpuMat& img, vector<Point>& hits, double hit_threshold, Size win_stride, Size padding)
363{
364 CV_Assert(img.type() == CV_8UC1 || img.type() == CV_8UC4);
365 CV_Assert(padding == Size(0, 0));
366
367 hits.clear();
368 if (detector.empty())
369 return;
370
371 computeBlockHistograms(img);
372
373 if (win_stride == Size())
374 win_stride = block_stride;
375 else
376 CV_Assert(win_stride.width % block_stride.width == 0 && win_stride.height % block_stride.height == 0);
377
378 Size wins_per_img = numPartsWithin(img.size(), win_size, win_stride);
379 // labels.create(1, wins_per_img.area(), CV_8U);
380 labels = getBuffer(1, wins_per_img.area(), CV_8U, labels_buf);
381
382 hog::classify_hists(win_size.height, win_size.width, block_stride.height, block_stride.width,
383 win_stride.height, win_stride.width, img.rows, img.cols, block_hists.ptr<float>(),
384 detector.ptr<float>(), (float)free_coef, (float)hit_threshold, labels.ptr());
385
386 labels.download(labels_host);
387 unsigned char* vec = labels_host.ptr();
388 for (int i = 0; i < wins_per_img.area(); i++)
389 {
390 int y = i / wins_per_img.width;
391 int x = i - wins_per_img.width * y;
392 if (vec[i])
393 hits.push_back(Point(x * win_stride.width, y * win_stride.height));
394 }
395}

399void cv::gpu::HOGDescriptor::detectMultiScale(const GpuMat& img, vector<Rect>& found_locations, double hit_threshold,
400 Size win_stride, Size padding, double scale0, int group_threshold)
401{
402
403 CV_Assert(img.type() == CV_8UC1 || img.type() == CV_8UC4);
404
405 vector<double> level_scale;
406 double scale = 1.;
407 int levels = 0;
408
409 for (levels = 0; levels < nlevels; levels++)
410 {
411 level_scale.push_back(scale);
412 if (cvRound(img.cols/scale) < win_size.width ||
413 cvRound(img.rows/scale) < win_size.height || scale0 <= 1)
414 break;
415 scale *= scale0;
416 }
417 levels = std::max(levels, 1);
418 level_scale.resize(levels);
419 image_scales.resize(levels);
420
421 std::vector<Rect> all_candidates;
422 vector<Point> locations;
423
424 for (size_t i = 0; i < level_scale.size(); i++)
425 {
426 scale = level_scale[i];
427 Size sz(cvRound(img.cols / scale), cvRound(img.rows / scale));
428 GpuMat smaller_img;
429
430 if (sz == img.size())
431 smaller_img = img;
432 else
433 {
434 image_scales[i].create(sz, img.type());
435 switch (img.type())
436 {
437 case CV_8UC1: hog::resize_8UC1(img, image_scales[i]); break;
438 case CV_8UC4: hog::resize_8UC4(img, image_scales[i]); break;
439 }
440 smaller_img = image_scales[i];
441 }
442
443 detect(smaller_img, locations, hit_threshold, win_stride, padding);
444 Size scaled_win_size(cvRound(win_size.width * scale), cvRound(win_size.height * scale));
445 for (size_t j = 0; j < locations.size(); j++)
446 all_candidates.push_back(Rect(Point2d((CvPoint)locations[j]) * scale, scaled_win_size));
447 }
448
449 found_locations.assign(all_candidates.begin(), all_candidates.end());
450 groupRectangles(found_locations, group_threshold, 0.2/*magic number copied from CPU version*/);
451}
