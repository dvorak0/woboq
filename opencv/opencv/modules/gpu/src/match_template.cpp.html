147 // Evaluates optimal template's area threshold. If
148 // template's area is less than the threshold, we use naive match
149 // template version, otherwise FFT-based (if available)
150 int getTemplateThreshold(int method, int depth)
151 {
152 switch (method)
153 {
154 case CV_TM_CCORR:
155 if (depth == CV_32F) return 250;
156 if (depth == CV_8U) return 300;
157 break;
158 case CV_TM_SQDIFF:
159 if (depth == CV_8U) return 300;
160 break;
161 }
162 CV_Error(CV_StsBadArg, "getTemplateThreshold: unsupported match template mode");
163 return 0;
164 }

167 void matchTemplate_CCORR_32F(
168 const GpuMat& image, const GpuMat& templ, GpuMat& result, MatchTemplateBuf &buf, Stream& stream)
169 {
170 result.create(image.rows - templ.rows + 1, image.cols - templ.cols + 1, CV_32F);
171 if (templ.size().area() < getTemplateThreshold(CV_TM_CCORR, CV_32F))
172 {
173 matchTemplateNaive_CCORR_32F(image, templ, result, image.channels(), StreamAccessor::getStream(stream));
174 return;
175 }
176
177 ConvolveBuf convolve_buf;
178 convolve_buf.user_block_size = buf.user_block_size;
179
180 if (image.channels() == 1)
181 convolve(image.reshape(1), templ.reshape(1), result, true, convolve_buf, stream);
182 else
183 {
184 GpuMat result_;
185 convolve(image.reshape(1), templ.reshape(1), result_, true, convolve_buf, stream);
186 extractFirstChannel_32F(result_, result, image.channels(), StreamAccessor::getStream(stream));
187 }
188 }

191 void matchTemplate_CCORR_8U(
192 const GpuMat& image, const GpuMat& templ, GpuMat& result, MatchTemplateBuf &buf, Stream& stream)
193 {
194 if (templ.size().area() < getTemplateThreshold(CV_TM_CCORR, CV_8U))
195 {
196 result.create(image.rows - templ.rows + 1, image.cols - templ.cols + 1, CV_32F);
197 matchTemplateNaive_CCORR_8U(image, templ, result, image.channels(), StreamAccessor::getStream(stream));
198 return;
199 }
200
201 if (stream)
202 {
203 stream.enqueueConvert(image, buf.imagef, CV_32F);
204 stream.enqueueConvert(templ, buf.templf, CV_32F);
205 }
206 else
207 {
208 image.convertTo(buf.imagef, CV_32F);
209 templ.convertTo(buf.templf, CV_32F);
210 }
211 matchTemplate_CCORR_32F(buf.imagef, buf.templf, result, buf, stream);
212 }

215 void matchTemplate_CCORR_NORMED_8U(
216 const GpuMat& image, const GpuMat& templ, GpuMat& result, MatchTemplateBuf &buf, Stream& stream)
217 {
218 matchTemplate_CCORR_8U(image, templ, result, buf, stream);
219
220 buf.image_sqsums.resize(1);
221 sqrIntegral(image.reshape(1), buf.image_sqsums[0], stream);
222
223 unsigned long long templ_sqsum = (unsigned long long)sqrSum(templ.reshape(1))[0];
224 normalize_8U(templ.cols, templ.rows, buf.image_sqsums[0], templ_sqsum, result, image.channels(), StreamAccessor::getStream(stream));
225 }

237 void matchTemplate_SQDIFF_8U(
238 const GpuMat& image, const GpuMat& templ, GpuMat& result, MatchTemplateBuf &buf, Stream& stream)
239 {
240 if (templ.size().area() < getTemplateThreshold(CV_TM_SQDIFF, CV_8U))
241 {
242 result.create(image.rows - templ.rows + 1, image.cols - templ.cols + 1, CV_32F);
243 matchTemplateNaive_SQDIFF_8U(image, templ, result, image.channels(), StreamAccessor::getStream(stream));
244 return;
245 }
246
247 buf.image_sqsums.resize(1);
248 sqrIntegral(image.reshape(1), buf.image_sqsums[0], stream);
249
250 unsigned long long templ_sqsum = (unsigned long long)sqrSum(templ.reshape(1))[0];
251
252 matchTemplate_CCORR_8U(image, templ, result, buf, stream);
253 matchTemplatePrepared_SQDIFF_8U(templ.cols, templ.rows, buf.image_sqsums[0], templ_sqsum, result, image.channels(), StreamAccessor::getStream(stream));
254 }

270 void matchTemplate_CCOFF_8U(
271 const GpuMat& image, const GpuMat& templ, GpuMat& result, MatchTemplateBuf &buf, Stream& stream)
272 {
273 matchTemplate_CCORR_8U(image, templ, result, buf, stream);
274
275 if (image.channels() == 1)
276 {
277 buf.image_sums.resize(1);
278 integral(image, buf.image_sums[0], stream);
279
280 unsigned int templ_sum = (unsigned int)sum(templ)[0];
281 matchTemplatePrepared_CCOFF_8U(templ.cols, templ.rows, buf.image_sums[0], templ_sum, result, StreamAccessor::getStream(stream));
282 }
283 else
284 {
285 split(image, buf.images);
286 buf.image_sums.resize(buf.images.size());
287 for (int i = 0; i < image.channels(); ++i)
288 integral(buf.images[i], buf.image_sums[i], stream);
289
290 Scalar templ_sum = sum(templ);
291
292 switch (image.channels())
293 {
294 case 2:
295 matchTemplatePrepared_CCOFF_8UC2(
296 templ.cols, templ.rows, buf.image_sums[0], buf.image_sums[1],
297 (unsigned int)templ_sum[0], (unsigned int)templ_sum[1],
298 result, StreamAccessor::getStream(stream));
299 break;
300 case 3:
301 matchTemplatePrepared_CCOFF_8UC3(
302 templ.cols, templ.rows, buf.image_sums[0], buf.image_sums[1], buf.image_sums[2],
303 (unsigned int)templ_sum[0], (unsigned int)templ_sum[1], (unsigned int)templ_sum[2],
304 result, StreamAccessor::getStream(stream));
305 break;
306 case 4:
307 matchTemplatePrepared_CCOFF_8UC4(
308 templ.cols, templ.rows, buf.image_sums[0], buf.image_sums[1], buf.image_sums[2], buf.image_sums[3],
309 (unsigned int)templ_sum[0], (unsigned int)templ_sum[1], (unsigned int)templ_sum[2],
310 (unsigned int)templ_sum[3], result, StreamAccessor::getStream(stream));
311 break;
312 default:
313 CV_Error(CV_StsBadArg, "matchTemplate: unsupported number of channels");
314 }
315 }
316 }

319 void matchTemplate_CCOFF_NORMED_8U(
320 const GpuMat& image, const GpuMat& templ, GpuMat& result, MatchTemplateBuf &buf, Stream& stream)
321 {
322 if (stream)
323 {
324 stream.enqueueConvert(image, buf.imagef, CV_32F);
325 stream.enqueueConvert(templ, buf.templf, CV_32F);
326 }
327 else
328 {
329 image.convertTo(buf.imagef, CV_32F);
330 templ.convertTo(buf.templf, CV_32F);
331 }
332
333 matchTemplate_CCORR_32F(buf.imagef, buf.templf, result, buf, stream);
334
335 if (image.channels() == 1)
336 {
337 buf.image_sums.resize(1);
338 integral(image, buf.image_sums[0], stream);
339 buf.image_sqsums.resize(1);
340 sqrIntegral(image, buf.image_sqsums[0], stream);
341
342 unsigned int templ_sum = (unsigned int)sum(templ)[0];
343 unsigned long long templ_sqsum = (unsigned long long)sqrSum(templ)[0];
344
345 matchTemplatePrepared_CCOFF_NORMED_8U(
346 templ.cols, templ.rows, buf.image_sums[0], buf.image_sqsums[0],
347 templ_sum, templ_sqsum, result, StreamAccessor::getStream(stream));
348 }
349 else
350 {
351 split(image, buf.images);
352 buf.image_sums.resize(buf.images.size());
353 buf.image_sqsums.resize(buf.images.size());
354 for (int i = 0; i < image.channels(); ++i)
355 {
356 integral(buf.images[i], buf.image_sums[i], stream);
357 sqrIntegral(buf.images[i], buf.image_sqsums[i], stream);
358 }
359
360 Scalar templ_sum = sum(templ);
361 Scalar templ_sqsum = sqrSum(templ);
362
363 switch (image.channels())
364 {
365 case 2:
366 matchTemplatePrepared_CCOFF_NORMED_8UC2(
367 templ.cols, templ.rows,
368 buf.image_sums[0], buf.image_sqsums[0],
369 buf.image_sums[1], buf.image_sqsums[1],
370 (unsigned int)templ_sum[0], (unsigned long long)templ_sqsum[0],
371 (unsigned int)templ_sum[1], (unsigned long long)templ_sqsum[1],
372 result, StreamAccessor::getStream(stream));
373 break;
374 case 3:
375 matchTemplatePrepared_CCOFF_NORMED_8UC3(
376 templ.cols, templ.rows,
377 buf.image_sums[0], buf.image_sqsums[0],
378 buf.image_sums[1], buf.image_sqsums[1],
379 buf.image_sums[2], buf.image_sqsums[2],
380 (unsigned int)templ_sum[0], (unsigned long long)templ_sqsum[0],
381 (unsigned int)templ_sum[1], (unsigned long long)templ_sqsum[1],
382 (unsigned int)templ_sum[2], (unsigned long long)templ_sqsum[2],
383 result, StreamAccessor::getStream(stream));
384 break;
385 case 4:
386 matchTemplatePrepared_CCOFF_NORMED_8UC4(
387 templ.cols, templ.rows,
388 buf.image_sums[0], buf.image_sqsums[0],
389 buf.image_sums[1], buf.image_sqsums[1],
390 buf.image_sums[2], buf.image_sqsums[2],
391 buf.image_sums[3], buf.image_sqsums[3],
392 (unsigned int)templ_sum[0], (unsigned long long)templ_sqsum[0],
393 (unsigned int)templ_sum[1], (unsigned long long)templ_sqsum[1],
394 (unsigned int)templ_sum[2], (unsigned long long)templ_sqsum[2],
395 (unsigned int)templ_sum[3], (unsigned long long)templ_sqsum[3],
396 result, StreamAccessor::getStream(stream));
397 break;
398 default:
399 CV_Error(CV_StsBadArg, "matchTemplate: unsupported number of channels");
400 }
401 }

413void cv::gpu::matchTemplate(
414 const GpuMat& image, const GpuMat& templ, GpuMat& result, int method,
415 MatchTemplateBuf &buf, Stream& stream)
416{
417 CV_Assert(image.type() == templ.type());
418 CV_Assert(image.cols >= templ.cols && image.rows >= templ.rows);
419
420 typedef void (*Caller)(const GpuMat&, const GpuMat&, GpuMat&, MatchTemplateBuf&, Stream& stream);
421
422 static const Caller callers8U[] = { ::matchTemplate_SQDIFF_8U, ::matchTemplate_SQDIFF_NORMED_8U,
423 ::matchTemplate_CCORR_8U, ::matchTemplate_CCORR_NORMED_8U,
424 ::matchTemplate_CCOFF_8U, ::matchTemplate_CCOFF_NORMED_8U };
425 static const Caller callers32F[] = { ::matchTemplate_SQDIFF_32F, 0,
426 ::matchTemplate_CCORR_32F, 0, 0, 0 };
427
428 const Caller* callers = 0;
429 switch (image.depth())
430 {
431 case CV_8U: callers = callers8U; break;
432 case CV_32F: callers = callers32F; break;
433 default: CV_Error(CV_StsBadArg, "matchTemplate: unsupported data type");
434 }
435
436 Caller caller = callers[method];
437 CV_Assert(caller);
438 caller(image, templ, result, buf, stream);
439}
