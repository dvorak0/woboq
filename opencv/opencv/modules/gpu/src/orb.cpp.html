350 void initializeOrbPattern(const Point* pattern0, Mat& pattern, int ntuples, int tupleSize, int poolSize)
351 {
352 RNG rng(0x12345678);
353
354 pattern.create(2, ntuples * tupleSize, CV_32SC1);
355 pattern.setTo(Scalar::all(0));
356
357 int* pattern_x_ptr = pattern.ptr<int>(0);
358 int* pattern_y_ptr = pattern.ptr<int>(1);
359
360 for (int i = 0; i < ntuples; i++)
361 {
362 for (int k = 0; k < tupleSize; k++)
363 {
364 for(;;)
365 {
366 int idx = rng.uniform(0, poolSize);
367 Point pt = pattern0[idx];
368
369 int k1;
370 for (k1 = 0; k1 < k; k1++)
371 if (pattern_x_ptr[tupleSize * i + k1] == pt.x && pattern_y_ptr[tupleSize * i + k1] == pt.y)
372 break;
373
374 if (k1 == k)
375 {
376 pattern_x_ptr[tupleSize * i + k] = pt.x;
377 pattern_y_ptr[tupleSize * i + k] = pt.y;
378 break;
379 }
380 }
381 }
382 }
383 }

385 void makeRandomPattern(int patchSize, Point* pattern, int npoints)
386 {
387 // we always start with a fixed seed,
388 // to make patterns the same on each run
389 RNG rng(0x34985739);
390
391 for (int i = 0; i < npoints; i++)
392 {
393 pattern[i].x = rng.uniform(-patchSize / 2, patchSize / 2 + 1);
394 pattern[i].y = rng.uniform(-patchSize / 2, patchSize / 2 + 1);
395 }
396 }

399cv::gpu::ORB_GPU::ORB_GPU(int nFeatures, float scaleFactor, int nLevels, int edgeThreshold, int firstLevel, int WTA_K, int scoreType, int patchSize) :
400 nFeatures_(nFeatures), scaleFactor_(scaleFactor), nLevels_(nLevels), edgeThreshold_(edgeThreshold), firstLevel_(firstLevel), WTA_K_(WTA_K),
401 scoreType_(scoreType), patchSize_(patchSize),
402 fastDetector_(DEFAULT_FAST_THRESHOLD)
403{
404 CV_Assert(patchSize_ >= 2);
405
406 // fill the extractors and descriptors for the corresponding scales
407 float factor = 1.0f / scaleFactor_;
408 float n_desired_features_per_scale = nFeatures_ * (1.0f - factor) / (1.0f - std::pow(factor, nLevels_));
409
410 n_features_per_level_.resize(nLevels_);
411 size_t sum_n_features = 0;
412 for (int level = 0; level < nLevels_ - 1; ++level)
413 {
414 n_features_per_level_[level] = cvRound(n_desired_features_per_scale);
415 sum_n_features += n_features_per_level_[level];
416 n_desired_features_per_scale *= factor;
417 }
418 n_features_per_level_[nLevels_ - 1] = nFeatures - sum_n_features;
419
420 // pre-compute the end of a row in a circular patch
421 int half_patch_size = patchSize_ / 2;
422 vector<int> u_max(half_patch_size + 2);
423 for (int v = 0; v <= half_patch_size * std::sqrt(2.f) / 2 + 1; ++v)
424 u_max[v] = cvRound(std::sqrt(static_cast<float>(half_patch_size * half_patch_size - v * v)));
425
426 // Make sure we are symmetric
427 for (int v = half_patch_size, v_0 = 0; v >= half_patch_size * std::sqrt(2.f) / 2; --v)
428 {
429 while (u_max[v_0] == u_max[v_0 + 1])
430 ++v_0;
431 u_max[v] = v_0;
432 ++v_0;
433 }
434 CV_Assert(u_max.size() < 32);
435 cv::gpu::device::orb::loadUMax(&u_max[0], static_cast<int>(u_max.size()));
436
437 // Calc pattern
438 const int npoints = 512;
439 Point pattern_buf[npoints];
440 const Point* pattern0 = (const Point*)bit_pattern_31_;
441 if (patchSize_ != 31)
442 {
443 pattern0 = pattern_buf;
444 makeRandomPattern(patchSize_, pattern_buf, npoints);
445 }
446
447 CV_Assert(WTA_K_ == 2 || WTA_K_ == 3 || WTA_K_ == 4);
448
449 Mat h_pattern;
450
451 if (WTA_K_ == 2)
452 {
453 h_pattern.create(2, npoints, CV_32SC1);
454
455 int* pattern_x_ptr = h_pattern.ptr<int>(0);
456 int* pattern_y_ptr = h_pattern.ptr<int>(1);
457
458 for (int i = 0; i < npoints; ++i)
459 {
460 pattern_x_ptr[i] = pattern0[i].x;
461 pattern_y_ptr[i] = pattern0[i].y;
462 }
463 }
464 else
465 {
466 int ntuples = descriptorSize() * 4;
467 initializeOrbPattern(pattern0, h_pattern, ntuples, WTA_K_, npoints);
468 }
469
470 pattern_.upload(h_pattern);
471
472 blurFilter = createGaussianFilter_GPU(CV_8UC1, Size(7, 7), 2, 2, BORDER_REFLECT_101);
473
474 blurForDescriptor = false;
475}

480 {
481 return pow(scaleFactor, level - firstLevel);
482 }

485void cv::gpu::ORB_GPU::buildScalePyramids(const GpuMat& image, const GpuMat& mask)
486{
487 CV_Assert(image.type() == CV_8UC1);
488 CV_Assert(mask.empty() || (mask.type() == CV_8UC1 && mask.size() == image.size()));
489
490 imagePyr_.resize(nLevels_);
491 maskPyr_.resize(nLevels_);
492
493 for (int level = 0; level < nLevels_; ++level)
494 {
495 float scale = 1.0f / getScale(scaleFactor_, firstLevel_, level);
496
497 Size sz(cvRound(image.cols * scale), cvRound(image.rows * scale));
498
499 ensureSizeIsEnough(sz, image.type(), imagePyr_[level]);
500 ensureSizeIsEnough(sz, CV_8UC1, maskPyr_[level]);
501 maskPyr_[level].setTo(Scalar::all(255));
502
503 // Compute the resized image
504 if (level != firstLevel_)
505 {
506 if (level < firstLevel_)
507 {
508 resize(image, imagePyr_[level], sz, 0, 0, INTER_LINEAR);
509
510 if (!mask.empty())
511 resize(mask, maskPyr_[level], sz, 0, 0, INTER_LINEAR);
512 }
513 else
514 {
515 resize(imagePyr_[level - 1], imagePyr_[level], sz, 0, 0, INTER_LINEAR);
516
517 if (!mask.empty())
518 {
519 resize(maskPyr_[level - 1], maskPyr_[level], sz, 0, 0, INTER_LINEAR);
520 threshold(maskPyr_[level], maskPyr_[level], 254, 0, THRESH_TOZERO);
521 }
522 }
523 }
524 else
525 {
526 image.copyTo(imagePyr_[level]);
527
528 if (!mask.empty())
529 mask.copyTo(maskPyr_[level]);
530 }
531
532 // Filter keypoints by image border
533 ensureSizeIsEnough(sz, CV_8UC1, buf_);
534 buf_.setTo(Scalar::all(0));
535 Rect inner(edgeThreshold_, edgeThreshold_, sz.width - 2 * edgeThreshold_, sz.height - 2 * edgeThreshold_);
536 buf_(inner).setTo(Scalar::all(255));
537
538 bitwise_and(maskPyr_[level], buf_, maskPyr_[level]);
539 }

544 //takes keypoints and culls them by the response
545 void cull(GpuMat& keypoints, int& count, int n_points)
546 {
547 using namespace cv::gpu::device::orb;
548
549 //this is only necessary if the keypoints size is greater than the number of desired points.
550 if (count > n_points)
551 {
552 if (n_points == 0)
553 {
554 keypoints.release();
555 return;
556 }
557
558 count = cull_gpu(keypoints.ptr<int>(FAST_GPU::LOCATION_ROW), keypoints.ptr<float>(FAST_GPU::RESPONSE_ROW), count, n_points);
559 }

563void cv::gpu::ORB_GPU::computeKeyPointsPyramid()
564{
565 using namespace cv::gpu::device::orb;
566
567 int half_patch_size = patchSize_ / 2;
568
569 keyPointsPyr_.resize(nLevels_);
570 keyPointsCount_.resize(nLevels_);
571
572 for (int level = 0; level < nLevels_; ++level)
573 {
574 keyPointsCount_[level] = fastDetector_.calcKeyPointsLocation(imagePyr_[level], maskPyr_[level]);
575
576 if (keyPointsCount_[level] == 0)
577 continue;
578
579 ensureSizeIsEnough(3, keyPointsCount_[level], CV_32FC1, keyPointsPyr_[level]);
580
581 GpuMat fastKpRange = keyPointsPyr_[level].rowRange(0, 2);
582 keyPointsCount_[level] = fastDetector_.getKeyPoints(fastKpRange);
583
584 if (keyPointsCount_[level] == 0)
585 continue;
586
587 int n_features = static_cast<int>(n_features_per_level_[level]);
588
589 if (scoreType_ == ORB::HARRIS_SCORE)
590 {
591 // Keep more points than necessary as FAST does not give amazing corners
592 cull(keyPointsPyr_[level], keyPointsCount_[level], 2 * n_features);
593
594 // Compute the Harris cornerness (better scoring than FAST)
595 HarrisResponses_gpu(imagePyr_[level], keyPointsPyr_[level].ptr<short2>(0), keyPointsPyr_[level].ptr<float>(1), keyPointsCount_[level], 7, HARRIS_K, 0);
596 }
597
598 //cull to the final desired level, using the new Harris scores or the original FAST scores.
599 cull(keyPointsPyr_[level], keyPointsCount_[level], n_features);
600
601 // Compute orientation
602 IC_Angle_gpu(imagePyr_[level], keyPointsPyr_[level].ptr<short2>(0), keyPointsPyr_[level].ptr<float>(2), keyPointsCount_[level], half_patch_size, 0);
603 }

606void cv::gpu::ORB_GPU::computeDescriptors(GpuMat& descriptors)
607{
608 using namespace cv::gpu::device::orb;
609
610 int nAllkeypoints = 0;
611
612 for (int level = 0; level < nLevels_; ++level)
613 nAllkeypoints += keyPointsCount_[level];
614
615 if (nAllkeypoints == 0)
616 {
617 descriptors.release();
618 return;
619 }
620
621 ensureSizeIsEnough(nAllkeypoints, descriptorSize(), CV_8UC1, descriptors);
622
623 int offset = 0;
624
625 for (int level = 0; level < nLevels_; ++level)
626 {
627 if (keyPointsCount_[level] == 0)
628 continue;
629
630 GpuMat descRange = descriptors.rowRange(offset, offset + keyPointsCount_[level]);
631
632 if (blurForDescriptor)
633 {
634 // preprocess the resized image
635 ensureSizeIsEnough(imagePyr_[level].size(), imagePyr_[level].type(), buf_);
636 blurFilter->apply(imagePyr_[level], buf_, Rect(0, 0, imagePyr_[level].cols, imagePyr_[level].rows));
637 }
638
639 computeOrbDescriptor_gpu(blurForDescriptor ? buf_ : imagePyr_[level], keyPointsPyr_[level].ptr<short2>(0), keyPointsPyr_[level].ptr<float>(2),
640 keyPointsCount_[level], pattern_.ptr<int>(0), pattern_.ptr<int>(1), descRange, descriptorSize(), WTA_K_, 0);
641
642 offset += keyPointsCount_[level];
643 }
644}

646void cv::gpu::ORB_GPU::mergeKeyPoints(GpuMat& keypoints)
647{
648 using namespace cv::gpu::device::orb;
649
650 int nAllkeypoints = 0;
651
652 for (int level = 0; level < nLevels_; ++level)
653 nAllkeypoints += keyPointsCount_[level];
654
655 if (nAllkeypoints == 0)
656 {
657 keypoints.release();
658 return;
659 }
660
661 ensureSizeIsEnough(ROWS_COUNT, nAllkeypoints, CV_32FC1, keypoints);
662
663 int offset = 0;
664
665 for (int level = 0; level < nLevels_; ++level)
666 {
667 if (keyPointsCount_[level] == 0)
668 continue;
669
670 float sf = getScale(scaleFactor_, firstLevel_, level);
671
672 GpuMat keyPointsRange = keypoints.colRange(offset, offset + keyPointsCount_[level]);
673
674 float locScale = level != firstLevel_ ? sf : 1.0f;
675
676 mergeLocation_gpu(keyPointsPyr_[level].ptr<short2>(0), keyPointsRange.ptr<float>(0), keyPointsRange.ptr<float>(1), keyPointsCount_[level], locScale, 0);
677
678 GpuMat range = keyPointsRange.rowRange(2, 4);
679 keyPointsPyr_[level](Range(1, 3), Range(0, keyPointsCount_[level])).copyTo(range);
680
681 keyPointsRange.row(4).setTo(Scalar::all(level));
682 keyPointsRange.row(5).setTo(Scalar::all(patchSize_ * sf));
683
684 offset += keyPointsCount_[level];
685 }

701void cv::gpu::ORB_GPU::convertKeyPoints(Mat &d_keypoints, std::vector<KeyPoint>& keypoints)
702{
703 if (d_keypoints.empty())
704 {
705 keypoints.clear();
706 return;
707 }
708
709 CV_Assert(d_keypoints.type() == CV_32FC1 && d_keypoints.rows == ROWS_COUNT);
710
711 const float* x_ptr = d_keypoints.ptr<float>(X_ROW);
712 const float* y_ptr = d_keypoints.ptr<float>(Y_ROW);
713 const float* response_ptr = d_keypoints.ptr<float>(RESPONSE_ROW);
714 const float* angle_ptr = d_keypoints.ptr<float>(ANGLE_ROW);
715 const float* octave_ptr = d_keypoints.ptr<float>(OCTAVE_ROW);
716 const float* size_ptr = d_keypoints.ptr<float>(SIZE_ROW);
717
718 keypoints.resize(d_keypoints.cols);
719
720 for (int i = 0; i < d_keypoints.cols; ++i)
721 {
722 KeyPoint kp;
723
724 kp.pt.x = x_ptr[i];
725 kp.pt.y = y_ptr[i];
726 kp.response = response_ptr[i];
727 kp.angle = angle_ptr[i];
728 kp.octave = static_cast<int>(octave_ptr[i]);
729 kp.size = size_ptr[i];
730
731 keypoints[i] = kp;
732 }
