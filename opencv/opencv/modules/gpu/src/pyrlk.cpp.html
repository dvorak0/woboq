81 void calcPatchSize(cv::Size winSize, dim3& block, dim3& patch)
82 {
83 if (winSize.width > 32 && winSize.width > 2 * winSize.height)
84 {
85 block.x = deviceSupports(FEATURE_SET_COMPUTE_12) ? 32 : 16;
86 block.y = 8;
87 }
88 else
89 {
90 block.x = 16;
91 block.y = deviceSupports(FEATURE_SET_COMPUTE_12) ? 16 : 8;
92 }
93
94 patch.x = (winSize.width + block.x - 1) / block.x;
95 patch.y = (winSize.height + block.y - 1) / block.y;
96
97 block.z = patch.z = 1;
98 }
99}

101void cv::gpu::PyrLKOpticalFlow::sparse(const GpuMat& prevImg, const GpuMat& nextImg, const GpuMat& prevPts, GpuMat& nextPts, GpuMat& status, GpuMat* err)
102{
103 if (prevPts.empty())
104 {
105 nextPts.release();
106 status.release();
107 if (err) err->release();
108 return;
109 }
110
111 dim3 block, patch;
112 calcPatchSize(winSize, block, patch);
113
114 CV_Assert(prevImg.channels() == 1 || prevImg.channels() == 3 || prevImg.channels() == 4);
115 CV_Assert(prevImg.size() == nextImg.size() && prevImg.type() == nextImg.type());
116 CV_Assert(maxLevel >= 0);
117 CV_Assert(winSize.width > 2 && winSize.height > 2);
118 CV_Assert(patch.x > 0 && patch.x < 6 && patch.y > 0 && patch.y < 6);
119 CV_Assert(prevPts.rows == 1 && prevPts.type() == CV_32FC2);
120
121 if (useInitialFlow)
122 CV_Assert(nextPts.size() == prevPts.size() && nextPts.type() == CV_32FC2);
123 else
124 ensureSizeIsEnough(1, prevPts.cols, prevPts.type(), nextPts);
125
126 GpuMat temp1 = (useInitialFlow ? nextPts : prevPts).reshape(1);
127 GpuMat temp2 = nextPts.reshape(1);
128 multiply(temp1, Scalar::all(1.0 / (1 << maxLevel) / 2.0), temp2);
129
130 ensureSizeIsEnough(1, prevPts.cols, CV_8UC1, status);
131 status.setTo(Scalar::all(1));
132
133 if (err)
134 ensureSizeIsEnough(1, prevPts.cols, CV_32FC1, *err);
135
136 // build the image pyramids.
137
138 prevPyr_.resize(maxLevel + 1);
139 nextPyr_.resize(maxLevel + 1);
140
141 int cn = prevImg.channels();
142
143 if (cn == 1 || cn == 4)
144 {
145 prevImg.convertTo(prevPyr_[0], CV_32F);
146 nextImg.convertTo(nextPyr_[0], CV_32F);
147 }
148 else
149 {
150 buf_.resize(1);
151
152 cvtColor(prevImg, buf_[0], COLOR_BGR2BGRA);
153 buf_[0].convertTo(prevPyr_[0], CV_32F);
154
155 cvtColor(nextImg, buf_[0], COLOR_BGR2BGRA);
156 buf_[0].convertTo(nextPyr_[0], CV_32F);
157 }
158
159 for (int level = 1; level <= maxLevel; ++level)
160 {
161 pyrDown(prevPyr_[level - 1], prevPyr_[level]);
162 pyrDown(nextPyr_[level - 1], nextPyr_[level]);
163 }
164
165 pyrlk::loadConstants(make_int2(winSize.width, winSize.height), iters);
166
167 for (int level = maxLevel; level >= 0; level--)
168 {
169 if (cn == 1)
170 {
171 pyrlk::sparse1(prevPyr_[level], nextPyr_[level],
172 prevPts.ptr<float2>(), nextPts.ptr<float2>(), status.ptr(), level == 0 && err ? err->ptr<float>() : 0, prevPts.cols,
173 level, block, patch);
174 }
175 else
176 {
177 pyrlk::sparse4(prevPyr_[level], nextPyr_[level],
178 prevPts.ptr<float2>(), nextPts.ptr<float2>(), status.ptr(), level == 0 && err ? err->ptr<float>() : 0, prevPts.cols,
179 level, block, patch);
180 }
181 }
182}

184void cv::gpu::PyrLKOpticalFlow::dense(const GpuMat& prevImg, const GpuMat& nextImg, GpuMat& u, GpuMat& v, GpuMat* err)
185{
186 CV_Assert(prevImg.type() == CV_8UC1);
187 CV_Assert(prevImg.size() == nextImg.size() && prevImg.type() == nextImg.type());
188 CV_Assert(maxLevel >= 0);
189 CV_Assert(winSize.width > 2 && winSize.height > 2);
190
191 if (err)
192 err->create(prevImg.size(), CV_32FC1);
193
194 // build the image pyramids.
195
196 prevPyr_.resize(maxLevel + 1);
197 nextPyr_.resize(maxLevel + 1);
198
199 prevPyr_[0] = prevImg;
200 nextImg.convertTo(nextPyr_[0], CV_32F);
201
202 for (int level = 1; level <= maxLevel; ++level)
203 {
204 pyrDown(prevPyr_[level - 1], prevPyr_[level]);
205 pyrDown(nextPyr_[level - 1], nextPyr_[level]);
206 }
207
208 ensureSizeIsEnough(prevImg.size(), CV_32FC1, uPyr_[0]);
209 ensureSizeIsEnough(prevImg.size(), CV_32FC1, vPyr_[0]);
210 ensureSizeIsEnough(prevImg.size(), CV_32FC1, uPyr_[1]);
211 ensureSizeIsEnough(prevImg.size(), CV_32FC1, vPyr_[1]);
212 uPyr_[0].setTo(Scalar::all(0));
213 vPyr_[0].setTo(Scalar::all(0));
214 uPyr_[1].setTo(Scalar::all(0));
215 vPyr_[1].setTo(Scalar::all(0));
216
217 int2 winSize2i = make_int2(winSize.width, winSize.height);
218 pyrlk::loadConstants(winSize2i, iters);
219
220 PtrStepSzf derr = err ? *err : PtrStepSzf();
221
222 int idx = 0;
223
224 for (int level = maxLevel; level >= 0; level--)
225 {
226 int idx2 = (idx + 1) & 1;
227
228 pyrlk::dense(prevPyr_[level], nextPyr_[level], uPyr_[idx], vPyr_[idx], uPyr_[idx2], vPyr_[idx2],
229 level == 0 ? derr : PtrStepSzf(), winSize2i);
230
231 if (level > 0)
232 idx = idx2;
233 }
234
235 uPyr_[idx].copyTo(u);
236 vPyr_[idx].copyTo(v);
237}

239void cv::gpu::PyrLKOpticalFlow::releaseMemory()
240{
241 prevPyr_.clear();
242 nextPyr_.clear();
243
244 buf_.clear();
245
246 uPyr_[0].release();
247 vPyr_[0].release();
248
249 uPyr_[1].release();
250 vPyr_[1].release();
251}
