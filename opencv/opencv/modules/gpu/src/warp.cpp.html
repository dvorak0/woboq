74void cv::gpu::buildWarpAffineMaps(const Mat& M, bool inverse, Size dsize, GpuMat& xmap, GpuMat& ymap, Stream& stream)
75{
76 using namespace cv::gpu::device::imgproc;
77
78 CV_Assert(M.rows == 2 && M.cols == 3);
79
80 xmap.create(dsize, CV_32FC1);
81 ymap.create(dsize, CV_32FC1);
82
83 float coeffs[2 * 3];
84 Mat coeffsMat(2, 3, CV_32F, (void*)coeffs);
85
86 if (inverse)
87 M.convertTo(coeffsMat, coeffsMat.type());
88 else
89 {
90 cv::Mat iM;
91 invertAffineTransform(M, iM);
92 iM.convertTo(coeffsMat, coeffsMat.type());
93 }
94
95 buildWarpAffineMaps_gpu(coeffs, xmap, ymap, StreamAccessor::getStream(stream));
96}

124 template<int DEPTH> struct NppTypeTraits;
125 template<> struct NppTypeTraits<CV_8U> { typedef Npp8u npp_t; };
126 template<> struct NppTypeTraits<CV_8S> { typedef Npp8s npp_t; };
127 template<> struct NppTypeTraits<CV_16U> { typedef Npp16u npp_t; };
128 template<> struct NppTypeTraits<CV_16S> { typedef Npp16s npp_t; typedef Npp16sc npp_complex_type; };
129 template<> struct NppTypeTraits<CV_32S> { typedef Npp32s npp_t; typedef Npp32sc npp_complex_type; };
130 template<> struct NppTypeTraits<CV_32F> { typedef Npp32f npp_t; typedef Npp32fc npp_complex_type; };
131 template<> struct NppTypeTraits<CV_64F> { typedef Npp64f npp_t; typedef Npp64fc npp_complex_type; };
132
133 template <int DEPTH> struct NppWarpFunc
134 {
135 typedef typename NppTypeTraits<DEPTH>::npp_t npp_t;
136
137 typedef NppStatus (*func_t)(const npp_t* pSrc, NppiSize srcSize, int srcStep, NppiRect srcRoi, npp_t* pDst,
138 int dstStep, NppiRect dstRoi, const double coeffs[][3],
139 int interpolation);
140 };
141
142 template <int DEPTH, typename NppWarpFunc<DEPTH>::func_t func> struct NppWarp
143 {
144 typedef typename NppWarpFunc<DEPTH>::npp_t npp_t;
145
146 static void call(const cv::gpu::GpuMat& src, cv::gpu::GpuMat& dst, double coeffs[][3], int interpolation, cudaStream_t stream)
147 {
148 static const int npp_inter[] = {NPPI_INTER_NN, NPPI_INTER_LINEAR, NPPI_INTER_CUBIC};
149
150 NppiSize srcsz;
151 srcsz.height = src.rows;
152 srcsz.width = src.cols;
153
154 NppiRect srcroi;
155 srcroi.x = 0;
156 srcroi.y = 0;
157 srcroi.height = src.rows;
158 srcroi.width = src.cols;
159
160 NppiRect dstroi;
161 dstroi.x = 0;
162 dstroi.y = 0;
163 dstroi.height = dst.rows;
164 dstroi.width = dst.cols;
165
166 cv::gpu::NppStreamHandler h(stream);
167
168 nppSafeCall( func(src.ptr<npp_t>(), srcsz, static_cast<int>(src.step), srcroi,
169 dst.ptr<npp_t>(), static_cast<int>(dst.step), dstroi,
170 coeffs, npp_inter[interpolation]) );
171
172 if (stream == 0)
173 cudaSafeCall( cudaDeviceSynchronize() );
174 }
175 };
