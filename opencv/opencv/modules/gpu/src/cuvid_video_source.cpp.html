48cv::gpu::detail::CuvidVideoSource::CuvidVideoSource(const std::string& fname)
49{
50 CUVIDSOURCEPARAMS params;
51 std::memset(&params, 0, sizeof(CUVIDSOURCEPARAMS));
52
53 // Fill parameter struct
54 params.pUserData = this; // will be passed to data handlers
55 params.pfnVideoDataHandler = HandleVideoData; // our local video-handler callback
56 params.pfnAudioDataHandler = 0;
57
58 // now create the actual source
59 CUresult res = cuvidCreateVideoSource(&videoSource_, fname.c_str(), &params);
60 if (res == CUDA_ERROR_INVALID_SOURCE)
61 throw std::runtime_error("Unsupported video source");
62 cuSafeCall( res );
63
64 CUVIDEOFORMAT vidfmt;
65 cuSafeCall( cuvidGetSourceVideoFormat(videoSource_, &vidfmt, 0) );
66
67 format_.codec = static_cast<VideoReader_GPU::Codec>(vidfmt.codec);
68 format_.chromaFormat = static_cast<VideoReader_GPU::ChromaFormat>(vidfmt.chroma_format);
69 format_.width = vidfmt.coded_width;
70 format_.height = vidfmt.coded_height;
71}
72
73cv::gpu::VideoReader_GPU::FormatInfo cv::gpu::detail::CuvidVideoSource::format() const
74{
75 return format_;
76}
77
78void cv::gpu::detail::CuvidVideoSource::start()
79{
80 cuSafeCall( cuvidSetVideoSourceState(videoSource_, cudaVideoState_Started) );
81}
82
83void cv::gpu::detail::CuvidVideoSource::stop()
84{
85 cuSafeCall( cuvidSetVideoSourceState(videoSource_, cudaVideoState_Stopped) );
86}
87
88bool cv::gpu::detail::CuvidVideoSource::isStarted() const
89{
90 return (cuvidGetVideoSourceState(videoSource_) == cudaVideoState_Started);
91}
92
93bool cv::gpu::detail::CuvidVideoSource::hasError() const
94{
95 return (cuvidGetVideoSourceState(videoSource_) == cudaVideoState_Error);
96}
97
98int CUDAAPI cv::gpu::detail::CuvidVideoSource::HandleVideoData(void* userData, CUVIDSOURCEDATAPACKET* packet)
99{
100 CuvidVideoSource* thiz = static_cast<CuvidVideoSource*>(userData);
101
102 return thiz->parseVideoData(packet->payload, packet->payload_size, (packet->flags & CUVID_PKT_ENDOFSTREAM) != 0);
103}
