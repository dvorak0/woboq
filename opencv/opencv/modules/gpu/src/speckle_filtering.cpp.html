50//! Speckle filtering - filters small connected components on diparity image.
51//! It sets pixel (x,y) to newVal if it coresponds to small CC with size < maxSpeckleSize.
52//! Threshold for border between CC is diffThreshold;
53 CV_EXPORTS void filterSpeckles(Mat& img, uchar newVal, int maxSpeckleSize, uchar diffThreshold, Mat& buf);
54}
55
56typedef Point_<short> Point2s;
57
58void cv::filterSpeckles( Mat& img, uchar newVal, int maxSpeckleSize, uchar maxDiff, Mat& _buf)
59{
60 int MaxD = 1024;
61 int WinSz = 64;
62
63 int bufSize0 = (MaxD + 2)*sizeof(int) + (img.rows+WinSz+2)*MaxD*sizeof(int) +
64 (img.rows + WinSz + 2)*sizeof(int) +
65 (img.rows+WinSz+2)*MaxD*(WinSz+1)*sizeof(uchar) + 256;
66 int bufSize1 = (img.cols + 9 + 2) * sizeof(int) + 256;
67 int bufSz = max(bufSize0 * 1, bufSize1 * 2);
68
69 _buf.create(1, bufSz, CV_8U);
70
71 CV_Assert( img.type() == CV_8U );
72
73 int width = img.cols, height = img.rows, npixels = width*height;
74 size_t bufSize = npixels*(int)(sizeof(Point2s) + sizeof(int) + sizeof(uchar));
75 if( !_buf.isContinuous() || !_buf.data || _buf.cols*_buf.rows*_buf.elemSize() < bufSize )
76 _buf.create(1, (int)bufSize, CV_8U);
77
78 uchar* buf = _buf.data;
79 int i, j, dstep = (int)(img.step/sizeof(uchar));
80 int* labels = (int*)buf;
81 buf += npixels*sizeof(labels[0]);
82 Point2s* wbuf = (Point2s*)buf;
83 buf += npixels*sizeof(wbuf[0]);
84 uchar* rtype = (uchar*)buf;
85 int curlabel = 0;
86
87 // clear out label assignments
88 memset(labels, 0, npixels*sizeof(labels[0]));
89
90 for( i = 0; i < height; i++ )
91 {
92 uchar* ds = img.ptr<uchar>(i);
93 int* ls = labels + width*i;
94
95 for( j = 0; j < width; j++ )
96 {
97 if( ds[j] != newVal ) // not a bad disparity
98 {
99 if( ls[j] ) // has a label, check for bad label
100 {
101 if( rtype[ls[j]] ) // small region, zero out disparity
102 ds[j] = (uchar)newVal;
103 }
104 // no label, assign and propagate
105 else
106 {
107 Point2s* ws = wbuf; // initialize wavefront
108 Point2s p((short)j, (short)i); // current pixel
109 curlabel++; // next label
110 int count = 0; // current region size
111 ls[j] = curlabel;
112
113 // wavefront propagation
114 while( ws >= wbuf ) // wavefront not empty
115 {
116 count++;
117 // put neighbors onto wavefront
118 uchar* dpp = &img.at<uchar>(p.y, p.x);
119 uchar dp = *dpp;
120 int* lpp = labels + width*p.y + p.x;
121
122 if( p.x < width-1 && !lpp[+1] && dpp[+1] != newVal && std::abs(dp - dpp[+1]) <= maxDiff )
123 {
124 lpp[+1] = curlabel;
125 *ws++ = Point2s(p.x+1, p.y);
126 }
127
128 if( p.x > 0 && !lpp[-1] && dpp[-1] != newVal && std::abs(dp - dpp[-1]) <= maxDiff )
129 {
130 lpp[-1] = curlabel;
131 *ws++ = Point2s(p.x-1, p.y);
132 }
133
134 if( p.y < height-1 && !lpp[+width] && dpp[+dstep] != newVal && std::abs(dp - dpp[+dstep]) <= maxDiff )
135 {
136 lpp[+width] = curlabel;
137 *ws++ = Point2s(p.x, p.y+1);
138 }
139
140 if( p.y > 0 && !lpp[-width] && dpp[-dstep] != newVal && std::abs(dp - dpp[-dstep]) <= maxDiff )
141 {
142 lpp[-width] = curlabel;
143 *ws++ = Point2s(p.x, p.y-1);
144 }
145
146 // pop most recent and propagate
147 // NB: could try least recent, maybe better convergence
148 p = *--ws;
149 }
150
151 // assign label type
152 if( count <= maxSpeckleSize ) // speckle region
153 {
154 //printf("count = %d\n", count);
155 rtype[ls[j]] = 1; // small region label
156 ds[j] = (uchar)newVal;
157 }
158 else
159 {
160 //printf("count = %d\n", count);
161 rtype[ls[j]] = 0; // large region label
162 }
163 }
164 }
165 }
166 }
