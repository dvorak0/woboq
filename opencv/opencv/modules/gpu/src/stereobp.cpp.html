92void cv::gpu::StereoBeliefPropagation::estimateRecommendedParams(int width, int height, int& ndisp, int& iters, int& levels)
93{
94 ndisp = width / 4;
95 if ((ndisp & 1) != 0)
96 ndisp++;
97
98 int mm = ::max(width, height);
99 iters = mm / 100 + 2;
100
101 levels = (int)(::log(static_cast<double>(mm)) + 1) * 4 / 5;
102 if (levels == 0) levels++;
103}

123 class StereoBeliefPropagationImpl
124 {
125 public:
126 StereoBeliefPropagationImpl(StereoBeliefPropagation& rthis_,
127 GpuMat& u_, GpuMat& d_, GpuMat& l_, GpuMat& r_,
128 GpuMat& u2_, GpuMat& d2_, GpuMat& l2_, GpuMat& r2_,
129 vector<GpuMat>& datas_, GpuMat& out_)
130 : rthis(rthis_), u(u_), d(d_), l(l_), r(r_), u2(u2_), d2(d2_), l2(l2_), r2(r2_), datas(datas_), out(out_),
131 zero(Scalar::all(0)), scale(rthis_.msg_type == CV_32F ? 1.0f : 10.0f)
132 {
133 CV_Assert(0 < rthis.ndisp && 0 < rthis.iters && 0 < rthis.levels);
134 CV_Assert(rthis.msg_type == CV_32F || rthis.msg_type == CV_16S);
135 CV_Assert(rthis.msg_type == CV_32F || (1 << (rthis.levels - 1)) * scale * rthis.max_data_term < numeric_limits<short>::max());
136 }
137
138 void operator()(const GpuMat& left, const GpuMat& right, GpuMat& disp, Stream& stream)
139 {
140 typedef void (*comp_data_t)(const PtrStepSzb& left, const PtrStepSzb& right, const PtrStepSzb& data, cudaStream_t stream);
141 static const comp_data_t comp_data_callers[2][5] =
142 {
143 {0, comp_data_gpu<unsigned char, short>, 0, comp_data_gpu<uchar3, short>, comp_data_gpu<uchar4, short>},
144 {0, comp_data_gpu<unsigned char, float>, 0, comp_data_gpu<uchar3, float>, comp_data_gpu<uchar4, float>}
145 };
146
147 CV_Assert(left.size() == right.size() && left.type() == right.type());
148 CV_Assert(left.type() == CV_8UC1 || left.type() == CV_8UC3 || left.type() == CV_8UC4);
149
150 rows = left.rows;
151 cols = left.cols;
152
153 int divisor = (int)pow(2.f, rthis.levels - 1.0f);
154 int lowest_cols = cols / divisor;
155 int lowest_rows = rows / divisor;
156 const int min_image_dim_size = 2;
157 CV_Assert(min(lowest_cols, lowest_rows) > min_image_dim_size);
158
159 init(stream);
160
161 createContinuous(rows * rthis.ndisp, cols, rthis.msg_type, datas[0]);
162
163 comp_data_callers[rthis.msg_type == CV_32F][left.channels()](left, right, datas[0], StreamAccessor::getStream(stream));
164
165 calcBP(disp, stream);
166 }
167
168 void operator()(const GpuMat& data, GpuMat& disp, Stream& stream)
169 {
170 CV_Assert((data.type() == rthis.msg_type) && (data.rows % rthis.ndisp == 0));
171
172 rows = data.rows / rthis.ndisp;
173 cols = data.cols;
174
175 int divisor = (int)pow(2.f, rthis.levels - 1.0f);
176 int lowest_cols = cols / divisor;
177 int lowest_rows = rows / divisor;
178 const int min_image_dim_size = 2;
179 CV_Assert(min(lowest_cols, lowest_rows) > min_image_dim_size);
180
181 init(stream);
182
183 datas[0] = data;
184
185 calcBP(disp, stream);
186 }
187 private:
188 void init(Stream& stream)
189 {
190 createContinuous(rows * rthis.ndisp, cols, rthis.msg_type, u);
191 createContinuous(rows * rthis.ndisp, cols, rthis.msg_type, d);
192 createContinuous(rows * rthis.ndisp, cols, rthis.msg_type, l);
193 createContinuous(rows * rthis.ndisp, cols, rthis.msg_type, r);
194
195 if (rthis.levels & 1)
196 {
197 //can clear less area
198 if (stream)
199 {
200 stream.enqueueMemSet(u, zero);
201 stream.enqueueMemSet(d, zero);
202 stream.enqueueMemSet(l, zero);
203 stream.enqueueMemSet(r, zero);
204 }
205 else
206 {
207 u.setTo(zero);
208 d.setTo(zero);
209 l.setTo(zero);
210 r.setTo(zero);
211 }
212 }
213
214 if (rthis.levels > 1)
215 {
216 int less_rows = (rows + 1) / 2;
217 int less_cols = (cols + 1) / 2;
218
219 createContinuous(less_rows * rthis.ndisp, less_cols, rthis.msg_type, u2);
220 createContinuous(less_rows * rthis.ndisp, less_cols, rthis.msg_type, d2);
221 createContinuous(less_rows * rthis.ndisp, less_cols, rthis.msg_type, l2);
222 createContinuous(less_rows * rthis.ndisp, less_cols, rthis.msg_type, r2);
223
224 if ((rthis.levels & 1) == 0)
225 {
226 if (stream)
227 {
228 stream.enqueueMemSet(u2, zero);
229 stream.enqueueMemSet(d2, zero);
230 stream.enqueueMemSet(l2, zero);
231 stream.enqueueMemSet(r2, zero);
232 }
233 else
234 {
235 u2.setTo(zero);
236 d2.setTo(zero);
237 l2.setTo(zero);
238 r2.setTo(zero);
239 }
240 }
241 }
242
243 load_constants(rthis.ndisp, rthis.max_data_term, scale * rthis.data_weight, scale * rthis.max_disc_term, scale * rthis.disc_single_jump);
244
245 datas.resize(rthis.levels);
246
247 cols_all.resize(rthis.levels);
248 rows_all.resize(rthis.levels);
249
250 cols_all[0] = cols;
251 rows_all[0] = rows;
252 }
253
254 void calcBP(GpuMat& disp, Stream& stream)
255 {
256 typedef void (*data_step_down_t)(int dst_cols, int dst_rows, int src_cols, int src_rows, const PtrStepSzb& src, const PtrStepSzb& dst, cudaStream_t stream);
257 static const data_step_down_t data_step_down_callers[2] =
258 {
259 data_step_down_gpu<short>, data_step_down_gpu<float>
260 };
261
262 typedef void (*level_up_messages_t)(int dst_idx, int dst_cols, int dst_rows, int src_rows, PtrStepSzb* mus, PtrStepSzb* mds, PtrStepSzb* mls, PtrStepSzb* mrs, cudaStream_t stream);
263 static const level_up_messages_t level_up_messages_callers[2] =
264 {
265 level_up_messages_gpu<short>, level_up_messages_gpu<float>
266 };
267
268 typedef void (*calc_all_iterations_t)(int cols, int rows, int iters, const PtrStepSzb& u, const PtrStepSzb& d, const PtrStepSzb& l, const PtrStepSzb& r, const PtrStepSzb& data, cudaStream_t stream);
269 static const calc_all_iterations_t calc_all_iterations_callers[2] =
270 {
271 calc_all_iterations_gpu<short>, calc_all_iterations_gpu<float>
272 };
273
274 typedef void (*output_t)(const PtrStepSzb& u, const PtrStepSzb& d, const PtrStepSzb& l, const PtrStepSzb& r, const PtrStepSzb& data, const PtrStepSz<short>& disp, cudaStream_t stream);
275 static const output_t output_callers[2] =
276 {
277 output_gpu<short>, output_gpu<float>
278 };
279
280 const int funcIdx = rthis.msg_type == CV_32F;
281
282 cudaStream_t cudaStream = StreamAccessor::getStream(stream);
283
284 for (int i = 1; i < rthis.levels; ++i)
285 {
286 cols_all[i] = (cols_all[i-1] + 1) / 2;
287 rows_all[i] = (rows_all[i-1] + 1) / 2;
288
289 createContinuous(rows_all[i] * rthis.ndisp, cols_all[i], rthis.msg_type, datas[i]);
290
291 data_step_down_callers[funcIdx](cols_all[i], rows_all[i], cols_all[i-1], rows_all[i-1], datas[i-1], datas[i], cudaStream);
292 }
293
294 PtrStepSzb mus[] = {u, u2};
295 PtrStepSzb mds[] = {d, d2};
296 PtrStepSzb mrs[] = {r, r2};
297 PtrStepSzb mls[] = {l, l2};
298
299 int mem_idx = (rthis.levels & 1) ? 0 : 1;
300
301 for (int i = rthis.levels - 1; i >= 0; --i)
302 {
303 // for lower level we have already computed messages by setting to zero
304 if (i != rthis.levels - 1)
305 level_up_messages_callers[funcIdx](mem_idx, cols_all[i], rows_all[i], rows_all[i+1], mus, mds, mls, mrs, cudaStream);
306
307 calc_all_iterations_callers[funcIdx](cols_all[i], rows_all[i], rthis.iters, mus[mem_idx], mds[mem_idx], mls[mem_idx], mrs[mem_idx], datas[i], cudaStream);
308
309 mem_idx = (mem_idx + 1) & 1;
310 }
311
312 if (disp.empty())
313 disp.create(rows, cols, CV_16S);
314
315 out = ((disp.type() == CV_16S) ? disp : (out.create(rows, cols, CV_16S), out));
316
317 if (stream)
318 stream.enqueueMemSet(out, zero);
319 else
320 out.setTo(zero);
321
322 output_callers[funcIdx](u, d, l, r, datas.front(), out, cudaStream);
323
324 if (disp.type() != CV_16S)
325 {
326 if (stream)
327 stream.enqueueConvert(out, disp, disp.type());
328 else
329 out.convertTo(disp, disp.type());
330 }
331 }
332
333 StereoBeliefPropagation& rthis;
334
335 GpuMat& u;
336 GpuMat& d;
337 GpuMat& l;
338 GpuMat& r;
339
340 GpuMat& u2;
341 GpuMat& d2;
342 GpuMat& l2;
343 GpuMat& r2;
344
345 vector<GpuMat>& datas;
346 GpuMat& out;
347
348 const Scalar zero;
349 const float scale;
350
351 int rows, cols;
352
353 vector<int> cols_all, rows_all;
354 };
355}
