86cv::gpu::MOG_GPU::MOG_GPU(int nmixtures) :
87 frameSize_(0, 0), frameType_(0), nframes_(0)
88{
89 nmixtures_ = std::min(nmixtures > 0 ? nmixtures : mog::defaultNMixtures, 8);
90 history = mog::defaultHistory;
91 varThreshold = mog::defaultVarThreshold;
92 backgroundRatio = mog::defaultBackgroundRatio;
93 noiseSigma = mog::defaultNoiseSigma;

96void cv::gpu::MOG_GPU::initialize(cv::Size frameSize, int frameType)
97{
98 CV_Assert(frameType == CV_8UC1 || frameType == CV_8UC3 || frameType == CV_8UC4);
99
100 frameSize_ = frameSize;
101 frameType_ = frameType;
102
103 int ch = CV_MAT_CN(frameType);
104 int work_ch = ch;
105
106 // for each gaussian mixture of each pixel bg model we store
107 // the mixture sort key (w/sum_of_variances), the mixture weight (w),
108 // the mean (nchannels values) and
109 // the diagonal covariance matrix (another nchannels values)
110
111 weight_.create(frameSize.height * nmixtures_, frameSize_.width, CV_32FC1);
112 sortKey_.create(frameSize.height * nmixtures_, frameSize_.width, CV_32FC1);
113 mean_.create(frameSize.height * nmixtures_, frameSize_.width, CV_32FC(work_ch));
114 var_.create(frameSize.height * nmixtures_, frameSize_.width, CV_32FC(work_ch));
115
116 weight_.setTo(cv::Scalar::all(0));
117 sortKey_.setTo(cv::Scalar::all(0));
118 mean_.setTo(cv::Scalar::all(0));
119 var_.setTo(cv::Scalar::all(0));
120
121 nframes_ = 0;
122}

124void cv::gpu::MOG_GPU::operator()(const cv::gpu::GpuMat& frame, cv::gpu::GpuMat& fgmask, float learningRate, Stream& stream)
125{
126 using namespace cv::gpu::device::mog;
127
128 CV_Assert(frame.depth() == CV_8U);
129
130 int ch = frame.channels();
131 int work_ch = ch;
132
133 if (nframes_ == 0 || learningRate >= 1.0 || frame.size() != frameSize_ || work_ch != mean_.channels())
134 initialize(frame.size(), frame.type());
135
136 fgmask.create(frameSize_, CV_8UC1);
137
138 ++nframes_;
139 learningRate = learningRate >= 0.0f && nframes_ > 1 ? learningRate : 1.0f / std::min(nframes_, history);
140 CV_Assert(learningRate >= 0.0f);
141
142 mog_gpu(frame, ch, fgmask, weight_, sortKey_, mean_, var_, nmixtures_,
143 varThreshold, learningRate, backgroundRatio, noiseSigma,
144 StreamAccessor::getStream(stream));
145}

147void cv::gpu::MOG_GPU::getBackgroundImage(GpuMat& backgroundImage, Stream& stream) const
148{
149 using namespace cv::gpu::device::mog;
150
151 backgroundImage.create(frameSize_, frameType_);
152
153 getBackgroundImage_gpu(backgroundImage.channels(), weight_, mean_, backgroundImage, nmixtures_, backgroundRatio, StreamAccessor::getStream(stream));
154}

173 // default parameters of gaussian background detection algorithm
174 const int defaultHistory = 500; // Learning rate; alpha = 1/defaultHistory2
175 const float defaultVarThreshold = 4.0f * 4.0f;
176 const int defaultNMixtures = 5; // maximal number of Gaussians in mixture
177 const float defaultBackgroundRatio = 0.9f; // threshold sum of weights for background test
178 const float defaultVarThresholdGen = 3.0f * 3.0f;
179 const float defaultVarInit = 15.0f; // initial variance for new components
180 const float defaultVarMax = 5.0f * defaultVarInit;
181 const float defaultVarMin = 4.0f;
182
183 // additional parameters
184 const float defaultfCT = 0.05f; // complexity reduction prior constant 0 - no reduction of number of components
185 const unsigned char defaultnShadowDetection = 127; // value to use in the segmentation mask for shadows, set 0 not to do shadow detection
186 const float defaultfTau = 0.5f; // Tau - shadow threshold, see the paper for explanation
187}

189cv::gpu::MOG2_GPU::MOG2_GPU(int nmixtures) :
190 frameSize_(0, 0), frameType_(0), nframes_(0)
191{
192 nmixtures_ = nmixtures > 0 ? nmixtures : mog2::defaultNMixtures;
193
194 history = mog2::defaultHistory;
195 varThreshold = mog2::defaultVarThreshold;
196 bShadowDetection = true;
197
198 backgroundRatio = mog2::defaultBackgroundRatio;
199 fVarInit = mog2::defaultVarInit;
200 fVarMax = mog2::defaultVarMax;
201 fVarMin = mog2::defaultVarMin;
202
203 varThresholdGen = mog2::defaultVarThresholdGen;
204 fCT = mog2::defaultfCT;
205 nShadowDetection = mog2::defaultnShadowDetection;
206 fTau = mog2::defaultfTau;
207}

209void cv::gpu::MOG2_GPU::initialize(cv::Size frameSize, int frameType)
210{
211 using namespace cv::gpu::device::mog;
212
213 CV_Assert(frameType == CV_8UC1 || frameType == CV_8UC3 || frameType == CV_8UC4);
214
215 frameSize_ = frameSize;
216 frameType_ = frameType;
217 nframes_ = 0;
218
219 int ch = CV_MAT_CN(frameType);
220 int work_ch = ch;
221
222 // for each gaussian mixture of each pixel bg model we store ...
223 // the mixture weight (w),
224 // the mean (nchannels values) and
225 // the covariance
226 weight_.create(frameSize.height * nmixtures_, frameSize_.width, CV_32FC1);
227 variance_.create(frameSize.height * nmixtures_, frameSize_.width, CV_32FC1);
228 mean_.create(frameSize.height * nmixtures_, frameSize_.width, CV_32FC(work_ch));
229
230 //make the array for keeping track of the used modes per pixel - all zeros at start
231 bgmodelUsedModes_.create(frameSize_, CV_8UC1);
232 bgmodelUsedModes_.setTo(cv::Scalar::all(0));
233
234 loadConstants(nmixtures_, varThreshold, backgroundRatio, varThresholdGen, fVarInit, fVarMin, fVarMax, fTau, nShadowDetection);
235}

237void cv::gpu::MOG2_GPU::operator()(const GpuMat& frame, GpuMat& fgmask, float learningRate, Stream& stream)
238{
239 using namespace cv::gpu::device::mog;
240
241 int ch = frame.channels();
242 int work_ch = ch;
243
244 if (nframes_ == 0 || learningRate >= 1.0f || frame.size() != frameSize_ || work_ch != mean_.channels())
245 initialize(frame.size(), frame.type());
246
247 fgmask.create(frameSize_, CV_8UC1);
248 fgmask.setTo(cv::Scalar::all(0));
249
250 ++nframes_;
251 learningRate = learningRate >= 0.0f && nframes_ > 1 ? learningRate : 1.0f / std::min(2 * nframes_, history);
252 CV_Assert(learningRate >= 0.0f);
253
254 mog2_gpu(frame, frame.channels(), fgmask, bgmodelUsedModes_, weight_, variance_, mean_, learningRate, -learningRate * fCT, bShadowDetection, StreamAccessor::getStream(stream));
255}

257void cv::gpu::MOG2_GPU::getBackgroundImage(GpuMat& backgroundImage, Stream& stream) const

259 using namespace cv::gpu::device::mog;
260
261 backgroundImage.create(frameSize_, frameType_);
262
263 getBackgroundImage2_gpu(backgroundImage.channels(), bgmodelUsedModes_, weight_, mean_, backgroundImage, StreamAccessor::getStream(stream));
264}
