61void cv::gpu::remap(const GpuMat& src, GpuMat& dst, const GpuMat& xmap, const GpuMat& ymap, int interpolation, int borderMode, Scalar borderValue, Stream& stream)
62{
63 using namespace cv::gpu::device::imgproc;
64
65 typedef void (*func_t)(PtrStepSzb src, PtrStepSzb srcWhole, int xoff, int yoff, PtrStepSzf xmap, PtrStepSzf ymap, PtrStepSzb dst, int interpolation,
66 int borderMode, const float* borderValue, cudaStream_t stream, bool cc20);
67
68#ifdef OPENCV_TINY_GPU_MODULE
69 static const func_t funcs[6][4] =
70 {
71 {remap_gpu<uchar> , 0 /*remap_gpu<uchar2>*/ , remap_gpu<uchar3> , remap_gpu<uchar4> },
72 {0, 0, 0, 0},
73 {0, 0, 0, 0},
74 {0, 0, 0, 0},
75 {0, 0, 0, 0},
76 {remap_gpu<float> , 0 /*remap_gpu<float2>*/ , remap_gpu<float3> , remap_gpu<float4> }
77 };
78#else
79 static const func_t funcs[6][4] =
80 {
81 {remap_gpu<uchar> , 0 /*remap_gpu<uchar2>*/ , remap_gpu<uchar3> , remap_gpu<uchar4> },
82 {0 /*remap_gpu<schar>*/, 0 /*remap_gpu<char2>*/ , 0 /*remap_gpu<char3>*/, 0 /*remap_gpu<char4>*/},
83 {remap_gpu<ushort> , 0 /*remap_gpu<ushort2>*/, remap_gpu<ushort3> , remap_gpu<ushort4> },
84 {remap_gpu<short> , 0 /*remap_gpu<short2>*/ , remap_gpu<short3> , remap_gpu<short4> },
85 {0 /*remap_gpu<int>*/ , 0 /*remap_gpu<int2>*/ , 0 /*remap_gpu<int3>*/ , 0 /*remap_gpu<int4>*/ },
86 {remap_gpu<float> , 0 /*remap_gpu<float2>*/ , remap_gpu<float3> , remap_gpu<float4> }
87 };
88#endif
89
90 CV_Assert(src.depth() <= CV_32F && src.channels() <= 4);
91 CV_Assert(xmap.type() == CV_32F && ymap.type() == CV_32F && xmap.size() == ymap.size());
92 CV_Assert(interpolation == INTER_NEAREST || interpolation == INTER_LINEAR || interpolation == INTER_CUBIC);
93 CV_Assert(borderMode == BORDER_REFLECT101 || borderMode == BORDER_REPLICATE || borderMode == BORDER_CONSTANT || borderMode == BORDER_REFLECT || borderMode == BORDER_WRAP);
94
95 const func_t func = funcs[src.depth()][src.channels() - 1];
96 CV_Assert(func != 0);
97
98 int gpuBorderType;
99 CV_Assert(tryConvertToGpuBorderType(borderMode, gpuBorderType));
100
101 dst.create(xmap.size(), src.type());
102
103 Scalar_<float> borderValueFloat;
104 borderValueFloat = borderValue;
105
106 Size wholeSize;
107 Point ofs;
108 src.locateROI(wholeSize, ofs);
109
110 func(src, PtrStepSzb(wholeSize.height, wholeSize.width, src.datastart, src.step), ofs.x, ofs.y, xmap, ymap,
111 dst, interpolation, gpuBorderType, borderValueFloat.val, StreamAccessor::getStream(stream), deviceSupports(FEATURE_SET_COMPUTE_20));
112}
