74void cv::gpu::bilateralFilter(const GpuMat& src, GpuMat& dst, int kernel_size, float sigma_color, float sigma_spatial, int borderMode, Stream& s)
75{
76 using cv::gpu::device::imgproc::bilateral_filter_gpu;
77
78 typedef void (*func_t)(const PtrStepSzb& src, PtrStepSzb dst, int kernel_size, float sigma_spatial, float sigma_color, int borderMode, cudaStream_t s);
79
80#ifdef OPENCV_TINY_GPU_MODULE
81 static const func_t funcs[6][4] =
82 {
83 {bilateral_filter_gpu<uchar> , 0 /*bilateral_filter_gpu<uchar2>*/ , bilateral_filter_gpu<uchar3> , bilateral_filter_gpu<uchar4> },
84 {0 /*bilateral_filter_gpu<schar>*/ , 0 /*bilateral_filter_gpu<schar2>*/ , 0 /*bilateral_filter_gpu<schar3>*/ , 0 /*bilateral_filter_gpu<schar4>*/ },
85 {0 /*bilateral_filter_gpu<ushort>*/, 0 /*bilateral_filter_gpu<ushort2>*/, 0 /*bilateral_filter_gpu<ushort3>*/, 0 /*bilateral_filter_gpu<ushort4>*/},
86 {0 /*bilateral_filter_gpu<short>*/ , 0 /*bilateral_filter_gpu<short2>*/ , 0 /*bilateral_filter_gpu<short3>*/ , 0 /*bilateral_filter_gpu<short4>*/ },
87 {0 /*bilateral_filter_gpu<int>*/ , 0 /*bilateral_filter_gpu<int2>*/ , 0 /*bilateral_filter_gpu<int3>*/ , 0 /*bilateral_filter_gpu<int4>*/ },
88 {bilateral_filter_gpu<float> , 0 /*bilateral_filter_gpu<float2>*/ , bilateral_filter_gpu<float3> , bilateral_filter_gpu<float4> }
89 };
90#else
91 static const func_t funcs[6][4] =
92 {
93 {bilateral_filter_gpu<uchar> , 0 /*bilateral_filter_gpu<uchar2>*/ , bilateral_filter_gpu<uchar3> , bilateral_filter_gpu<uchar4> },
94 {0 /*bilateral_filter_gpu<schar>*/, 0 /*bilateral_filter_gpu<schar2>*/ , 0 /*bilateral_filter_gpu<schar3>*/, 0 /*bilateral_filter_gpu<schar4>*/},
95 {bilateral_filter_gpu<ushort> , 0 /*bilateral_filter_gpu<ushort2>*/, bilateral_filter_gpu<ushort3> , bilateral_filter_gpu<ushort4> },
96 {bilateral_filter_gpu<short> , 0 /*bilateral_filter_gpu<short2>*/ , bilateral_filter_gpu<short3> , bilateral_filter_gpu<short4> },
97 {0 /*bilateral_filter_gpu<int>*/ , 0 /*bilateral_filter_gpu<int2>*/ , 0 /*bilateral_filter_gpu<int3>*/ , 0 /*bilateral_filter_gpu<int4>*/ },
98 {bilateral_filter_gpu<float> , 0 /*bilateral_filter_gpu<float2>*/ , bilateral_filter_gpu<float3> , bilateral_filter_gpu<float4> }
99 };
100#endif
101
102 sigma_color = (sigma_color <= 0 ) ? 1 : sigma_color;
103 sigma_spatial = (sigma_spatial <= 0 ) ? 1 : sigma_spatial;
104
105
106 int radius = (kernel_size <= 0) ? cvRound(sigma_spatial*1.5) : kernel_size/2;
107 kernel_size = std::max(radius, 1)*2 + 1;
108
109 CV_Assert(src.depth() <= CV_32F && src.channels() <= 4);
110 const func_t func = funcs[src.depth()][src.channels() - 1];
111 CV_Assert(func != 0);
112
113 CV_Assert(borderMode == BORDER_REFLECT101 || borderMode == BORDER_REPLICATE || borderMode == BORDER_CONSTANT || borderMode == BORDER_REFLECT || borderMode == BORDER_WRAP);
114
115 int gpuBorderType;
116 CV_Assert(tryConvertToGpuBorderType(borderMode, gpuBorderType));
117
118 dst.create(src.size(), src.type());
119 func(src, dst, kernel_size, sigma_spatial, sigma_color, gpuBorderType, StreamAccessor::getStream(s));
120}

122void cv::gpu::nonLocalMeans(const GpuMat& src, GpuMat& dst, float h, int search_window, int block_window, int borderMode, Stream& s)
123{
124 using cv::gpu::device::imgproc::nlm_bruteforce_gpu;
125 typedef void (*func_t)(const PtrStepSzb& src, PtrStepSzb dst, int search_radius, int block_radius, float h, int borderMode, cudaStream_t stream);
126
127 static const func_t funcs[4] = { nlm_bruteforce_gpu<uchar>, nlm_bruteforce_gpu<uchar2>, nlm_bruteforce_gpu<uchar3>, 0/*nlm_bruteforce_gpu<uchar4>,*/ };
128
129 CV_Assert(src.type() == CV_8U || src.type() == CV_8UC2 || src.type() == CV_8UC3);
130
131 const func_t func = funcs[src.channels() - 1];
132 CV_Assert(func != 0);
133
134 int b = borderMode;
135 CV_Assert(b == BORDER_REFLECT101 || b == BORDER_REPLICATE || b == BORDER_CONSTANT || b == BORDER_REFLECT || b == BORDER_WRAP);
136
137 int gpuBorderType;
138 CV_Assert(tryConvertToGpuBorderType(borderMode, gpuBorderType));
139
140 dst.create(src.size(), src.type());
141 func(src, dst, search_window/2, block_window/2, h, gpuBorderType, StreamAccessor::getStream(s));
142}

164void cv::gpu::FastNonLocalMeansDenoising::simpleMethod(const GpuMat& src, GpuMat& dst, float h, int search_window, int block_window, Stream& s)
165{
166 CV_Assert(src.depth() == CV_8U && src.channels() < 4);
167
168 int border_size = search_window/2 + block_window/2;
169 Size esize = src.size() + Size(border_size, border_size) * 2;
170
171 cv::gpu::ensureSizeIsEnough(esize, CV_8UC3, extended_src_buffer);
172 GpuMat extended_src(esize, src.type(), extended_src_buffer.ptr(), extended_src_buffer.step);
173
174 cv::gpu::copyMakeBorder(src, extended_src, border_size, border_size, border_size, border_size, cv::BORDER_DEFAULT, Scalar(), s);
175 GpuMat src_hdr = extended_src(Rect(Point2i(border_size, border_size), src.size()));
176
177 int bcols, brows;
178 device::imgproc::nln_fast_get_buffer_size(src_hdr, search_window, block_window, bcols, brows);
179 buffer.create(brows, bcols, CV_32S);
180
181 using namespace cv::gpu::device::imgproc;
182 typedef void (*nlm_fast_t)(const PtrStepSzb&, PtrStepSzb, PtrStepi, int, int, float, cudaStream_t);
183 static const nlm_fast_t funcs[] = { nlm_fast_gpu<uchar>, nlm_fast_gpu<uchar2>, nlm_fast_gpu<uchar3>, 0};
184
185 dst.create(src.size(), src.type());
186 funcs[src.channels()-1](src_hdr, dst, buffer, search_window, block_window, h, StreamAccessor::getStream(s));
187}

189void cv::gpu::FastNonLocalMeansDenoising::labMethod( const GpuMat& src, GpuMat& dst, float h_luminance, float h_color, int search_window, int block_window, Stream& s)
190{
191 CV_Assert(src.type() == CV_8UC3);
192
193 lab.create(src.size(), src.type());
194 cv::gpu::cvtColor(src, lab, CV_BGR2Lab, 0, s);
195
196 l.create(src.size(), CV_8U);
197 ab.create(src.size(), CV_8UC2);
198 device::imgproc::fnlm_split_channels(lab, l, ab, StreamAccessor::getStream(s));
199
200 simpleMethod(l, l, h_luminance, search_window, block_window, s);
201 simpleMethod(ab, ab, h_color, search_window, block_window, s);
202
203 device::imgproc::fnlm_merge_channels(l, ab, lab, StreamAccessor::getStream(s));
204 cv::gpu::cvtColor(lab, dst, CV_Lab2BGR, 0, s);
205}
