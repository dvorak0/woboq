65//////////////////////////////// CudaMem ////////////////////////////////
66// CudaMem is limited cv::Mat with page locked memory allocation.
67// Page locked memory is only needed for async and faster coping to GPU.
68// It is convertable to cv::Mat header without reference counting
69// so you can use it with other opencv functions.
70
71// Page-locks the matrix m memory and maps it for the device(s)
72CV_EXPORTS void registerPageLocked(Mat& m);
73// Unmaps the memory of matrix m, and makes it pageable again.
74CV_EXPORTS void unregisterPageLocked(Mat& m);
75
76class CV_EXPORTS CudaMem
77{
78public:
79 enum { ALLOC_PAGE_LOCKED = 1, ALLOC_ZEROCOPY = 2, ALLOC_WRITE_COMBINED = 4 };
80
81 CudaMem();
82 CudaMem(const CudaMem& m);
83
84 CudaMem(int rows, int cols, int type, int _alloc_type = ALLOC_PAGE_LOCKED);
85 CudaMem(Size size, int type, int alloc_type = ALLOC_PAGE_LOCKED);
86
87
88 //! creates from cv::Mat with coping data
89 explicit CudaMem(const Mat& m, int alloc_type = ALLOC_PAGE_LOCKED);
90
91 ~CudaMem();
92
93 CudaMem& operator = (const CudaMem& m);
94
95 //! returns deep copy of the matrix, i.e. the data is copied
96 CudaMem clone() const;
97
98 //! allocates new matrix data unless the matrix already has specified size and type.
99 void create(int rows, int cols, int type, int alloc_type = ALLOC_PAGE_LOCKED);
100 void create(Size size, int type, int alloc_type = ALLOC_PAGE_LOCKED);
101
102 //! decrements reference counter and released memory if needed.
103 void release();
104
105 //! returns matrix header with disabled reference counting for CudaMem data.
106 Mat createMatHeader() const;
107 operator Mat() const;
108
109 //! maps host memory into device address space and returns GpuMat header for it. Throws exception if not supported by hardware.
110 GpuMat createGpuMatHeader() const;
111 operator GpuMat() const;
112
113 //returns if host memory can be mapperd to gpu address space;
114 static bool canMapHostMemory();
115
116 // Please see cv::Mat for descriptions
117 bool isContinuous() const;
118 size_t elemSize() const;
119 size_t elemSize1() const;
120 int type() const;
121 int depth() const;
122 int channels() const;
123 size_t step1() const;
124 Size size() const;
125 bool empty() const;
126
127
128 // Please see cv::Mat for descriptions
129 int flags;
130 int rows, cols;
131 size_t step;
132
133 uchar* data;
134 int* refcount;
135
136 uchar* datastart;
137 uchar* dataend;
138
139 int alloc_type;
140};

142//////////////////////////////// CudaStream ////////////////////////////////
143// Encapculates Cuda Stream. Provides interface for async coping.
144// Passed to each function that supports async kernel execution.
145// Reference counting is enabled
146
147class CV_EXPORTS Stream
148{
149public:
150 Stream();
151 ~Stream();
152
153 Stream(const Stream&);
154 Stream& operator =(const Stream&);
155
156 bool queryIfComplete();
157 void waitForCompletion();
158
159 //! downloads asynchronously
160 // Warning! cv::Mat must point to page locked memory (i.e. to CudaMem data or to its subMat)
161 void enqueueDownload(const GpuMat& src, CudaMem& dst);
162 void enqueueDownload(const GpuMat& src, Mat& dst);
163
164 //! uploads asynchronously
165 // Warning! cv::Mat must point to page locked memory (i.e. to CudaMem data or to its ROI)
166 void enqueueUpload(const CudaMem& src, GpuMat& dst);
167 void enqueueUpload(const Mat& src, GpuMat& dst);
168
169 //! copy asynchronously
170 void enqueueCopy(const GpuMat& src, GpuMat& dst);
171
172 //! memory set asynchronously
173 void enqueueMemSet(GpuMat& src, Scalar val);
174 void enqueueMemSet(GpuMat& src, Scalar val, const GpuMat& mask);
175
176 //! converts matrix type, ex from float to uchar depending on type
177 void enqueueConvert(const GpuMat& src, GpuMat& dst, int dtype, double a = 1, double b = 0);
178
179 //! adds a callback to be called on the host after all currently enqueued items in the stream have completed
180 typedef void (*StreamCallback)(Stream& stream, int status, void* userData);
181 void enqueueHostCallback(StreamCallback callback, void* userData);
182
183 static Stream& Null();
184
185 operator bool() const;
186
187private:
188 struct Impl;
189
190 explicit Stream(Impl* impl);
191 void create();
192 void release();
193
194 Impl *impl;
195
196 friend struct StreamAccessor;
197};
