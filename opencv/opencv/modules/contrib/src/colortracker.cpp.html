56void CvMeanShiftTracker::newTrackingWindow(Mat image, Rect selection)
57{
58 hist.release();
59 int channels[] = { 0, 0 , 1, 1};
60 float hrange[] = { 0, 180 };
61 float srange[] = { 0, 1 };
62 const float* ranges[] = {hrange, srange};
63
64 cvtColor(image, hsv, CV_BGR2HSV);
65 inRange(hsv, Scalar(0, 30, MIN(10, 256)), Scalar(180, 256, MAX(10, 256)), mask);
66
67 hue.create(hsv.size(), CV_8UC2);
68 mixChannels(&hsv, 1, &hue, 1, channels, 2);
69
70 Mat roi(hue, selection);
71 Mat mskroi(mask, selection);
72 int ch[] = {0, 1};
73 int chsize[] = {32, 32};
74 calcHist(&roi, 1, ch, mskroi, hist, 1, chsize, ranges);
75 normalize(hist, hist, 0, 255, CV_MINMAX);
76
77 prev_trackwindow = selection;
78}

80RotatedRect CvMeanShiftTracker::updateTrackingWindow(Mat image)
81{
82 int channels[] = { 0, 0 , 1, 1};
83 float hrange[] = { 0, 180 };
84 float srange[] = { 0, 1 };
85 const float* ranges[] = {hrange, srange};
86
87 cvtColor(image, hsv, CV_BGR2HSV);
88 inRange(hsv, Scalar(0, 30, MIN(10, 256)), Scalar(180, 256, MAX(10, 256)), mask);
89 hue.create(hsv.size(), CV_8UC2);
90 mixChannels(&hsv, 1, &hue, 1, channels, 2);
91 int ch[] = {0, 1};
92 calcBackProject(&hue, 1, ch, hist, backproj, ranges);
93 backproj &= mask;
94
95 prev_trackbox = CamShift(backproj, prev_trackwindow, TermCriteria(CV_TERMCRIT_EPS | CV_TERMCRIT_ITER, 10, 1));
96 int cols = backproj.cols, rows = backproj.rows, r = (MIN(cols, rows) + 5) / 6;
97 prev_trackwindow = Rect(prev_trackwindow.x - r, prev_trackwindow.y - r, prev_trackwindow.x + r,
98 prev_trackwindow.y + r) & Rect(0, 0, cols, rows);
99
100 prev_center.x = (float)(prev_trackwindow.x + prev_trackwindow.width / 2);
101 prev_center.y = (float)(prev_trackwindow.y + prev_trackwindow.height / 2);
102
103#ifdef DEBUG_HYTRACKER
104 ellipse(image, prev_trackbox, Scalar(0, 0, 255), 1, CV_AA);
105#endif
106
107 return prev_trackbox;
108}

110Mat CvMeanShiftTracker::getHistogramProjection(int type)
111{

113 backproj.convertTo(ms_backproj_f, type);
114 return ms_backproj_f;
