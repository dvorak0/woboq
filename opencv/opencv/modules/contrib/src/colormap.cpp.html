28static Mat linspace(float x0, float x1, int n)
29{
30 Mat pts(n, 1, CV_32FC1);
31 float step = (x1-x0)/(n-1);
32 for(int i = 0; i < n; i++)
33 pts.at<float>(i,0) = x0+i*step;
34 return pts;
35}

40static void sortMatrixRowsByIndices(InputArray _src, InputArray _indices, OutputArray _dst)
41{
42 if(_indices.getMat().type() != CV_32SC1)
43 CV_Error(CV_StsUnsupportedFormat, "cv::sortRowsByIndices only works on integer indices!");
44 Mat src = _src.getMat();
45 vector<int> indices = _indices.getMat();
46 _dst.create(src.rows, src.cols, src.type());
47 Mat dst = _dst.getMat();
48 for(size_t idx = 0; idx < indices.size(); idx++) {
49 Mat originalRow = src.row(indices[idx]);
50 Mat sortedRow = dst.row((int)idx);
51 originalRow.copyTo(sortedRow);
52 }
53}

55static Mat sortMatrixRowsByIndices(InputArray src, InputArray indices)
56{
57 Mat dst;
58 sortMatrixRowsByIndices(src, indices, dst);
59 return dst;
60}

63static Mat argsort(InputArray _src, bool ascending=true)
64{
65 Mat src = _src.getMat();
66 if (src.rows != 1 && src.cols != 1)
67 CV_Error(CV_StsBadArg, "cv::argsort only sorts 1D matrices.");
68 int flags = CV_SORT_EVERY_ROW+(ascending ? CV_SORT_ASCENDING : CV_SORT_DESCENDING);
69 Mat sorted_indices;
70 sortIdx(src.reshape(1,1),sorted_indices,flags);
71 return sorted_indices;
72}

74template <typename _Tp> static
75Mat interp1_(const Mat& X_, const Mat& Y_, const Mat& XI)
76{
77 int n = XI.rows;
78 // sort input table
79 vector<int> sort_indices = argsort(X_);
80
81 Mat X = sortMatrixRowsByIndices(X_,sort_indices);
82 Mat Y = sortMatrixRowsByIndices(Y_,sort_indices);
83 // interpolated values
84 Mat yi = Mat::zeros(XI.size(), XI.type());
85 for(int i = 0; i < n; i++) {
86 int c = 0;
87 int low = 0;
88 int high = X.rows - 1;
89 // set bounds
90 if(XI.at<_Tp>(i,0) < X.at<_Tp>(low, 0))
91 high = 1;
92 if(XI.at<_Tp>(i,0) > X.at<_Tp>(high, 0))
93 low = high - 1;
94 // binary search
95 while((high-low)>1) {
96 c = low + ((high - low) >> 1);
97 if(XI.at<_Tp>(i,0) > X.at<_Tp>(c,0)) {
98 low = c;
99 } else {
100 high = c;
101 }
102 }
103 // linear interpolation
104 yi.at<_Tp>(i,0) += Y.at<_Tp>(low,0)
105 + (XI.at<_Tp>(i,0) - X.at<_Tp>(low,0))
106 * (Y.at<_Tp>(high,0) - Y.at<_Tp>(low,0))
107 / (X.at<_Tp>(high,0) - X.at<_Tp>(low,0));
108 }
109 return yi;
110}

112static Mat interp1(InputArray _x, InputArray _Y, InputArray _xi)
113{
114 // get matrices
115 Mat x = _x.getMat();
116 Mat Y = _Y.getMat();
117 Mat xi = _xi.getMat();
118 // check types & alignment
119 assert((x.type() == Y.type()) && (Y.type() == xi.type()));
120 assert((x.cols == 1) && (x.rows == Y.rows) && (x.cols == Y.cols));
121 // call templated interp1
122 switch(x.type()) {
123 case CV_8SC1: return interp1_<char>(x,Y,xi); break;
124 case CV_8UC1: return interp1_<unsigned char>(x,Y,xi); break;
125 case CV_16SC1: return interp1_<short>(x,Y,xi); break;
126 case CV_16UC1: return interp1_<unsigned short>(x,Y,xi); break;
127 case CV_32SC1: return interp1_<int>(x,Y,xi); break;
128 case CV_32FC1: return interp1_<float>(x,Y,xi); break;
129 case CV_64FC1: return interp1_<double>(x,Y,xi); break;
130 default: CV_Error(CV_StsUnsupportedFormat, ""); break;
131 }
132 return Mat();
133}

473 void ColorMap::operator()(InputArray _src, OutputArray _dst) const
474 {
475 if(_lut.total() != 256)
476 CV_Error(CV_StsAssert, "cv::LUT only supports tables of size 256.");
477 Mat src = _src.getMat();
478 // Return original matrix if wrong type is given (is fail loud better here?)
479 if(src.type() != CV_8UC1 && src.type() != CV_8UC3)
480 {
481 src.copyTo(_dst);
482 return;
483 }
484 // Turn into a BGR matrix into its grayscale representation.
485 if(src.type() == CV_8UC3)
486 cvtColor(src.clone(), src, CV_BGR2GRAY);
487 cvtColor(src.clone(), src, CV_GRAY2BGR);
488 // Apply the ColorMap.
489 LUT(src, _lut, _dst);
490 }

497 interp1(X, b, xi),
498 interp1(X, g, xi),
499 interp1(X, r, xi)};
