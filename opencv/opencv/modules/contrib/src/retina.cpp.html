115void Retina::setup(std::string retinaParameterFile, const bool applyDefaultSetupOnFailure)
116{
117 try
118 {
119 // opening retinaParameterFile in read mode
120 cv::FileStorage fs(retinaParameterFile, cv::FileStorage::READ);
121 setup(fs, applyDefaultSetupOnFailure);
122 }catch(Exception &e)
123 {
124 std::cout<<"Retina::setup: wrong/unappropriate xml parameter file : error report :`n=>"<<e.what()<<std::endl;
125 if (applyDefaultSetupOnFailure)
126 {
127 std::cout<<"Retina::setup: resetting retina with default parameters"<<std::endl;
128 setupOPLandIPLParvoChannel();
129 setupIPLMagnoChannel();
130 }
131 else
132 {
133 std::cout<<"=> keeping current parameters"<<std::endl;
134 }
135 }

138void Retina::setup(cv::FileStorage &fs, const bool applyDefaultSetupOnFailure)
139{
140 try
141 {
142 // read parameters file if it exists or apply default setup if asked for
143 if (!fs.isOpened())
144 {
145 std::cout<<"Retina::setup: provided parameters file could not be open... skeeping configuration"<<std::endl;
146 return;
147 // implicit else case : retinaParameterFile could be open (it exists at least)
148 }
149 // OPL and Parvo init first... update at the same time the parameters structure and the retina core
150 cv::FileNode rootFn = fs.root(), currFn=rootFn["OPLandIPLparvo"];
151 currFn["colorMode"]>>_retinaParameters.OPLandIplParvo.colorMode;
152 currFn["normaliseOutput"]>>_retinaParameters.OPLandIplParvo.normaliseOutput;
153 currFn["photoreceptorsLocalAdaptationSensitivity"]>>_retinaParameters.OPLandIplParvo.photoreceptorsLocalAdaptationSensitivity;
154 currFn["photoreceptorsTemporalConstant"]>>_retinaParameters.OPLandIplParvo.photoreceptorsTemporalConstant;
155 currFn["photoreceptorsSpatialConstant"]>>_retinaParameters.OPLandIplParvo.photoreceptorsSpatialConstant;
156 currFn["horizontalCellsGain"]>>_retinaParameters.OPLandIplParvo.horizontalCellsGain;
157 currFn["hcellsTemporalConstant"]>>_retinaParameters.OPLandIplParvo.hcellsTemporalConstant;
158 currFn["hcellsSpatialConstant"]>>_retinaParameters.OPLandIplParvo.hcellsSpatialConstant;
159 currFn["ganglionCellsSensitivity"]>>_retinaParameters.OPLandIplParvo.ganglionCellsSensitivity;
160 setupOPLandIPLParvoChannel(_retinaParameters.OPLandIplParvo.colorMode, _retinaParameters.OPLandIplParvo.normaliseOutput, _retinaParameters.OPLandIplParvo.photoreceptorsLocalAdaptationSensitivity, _retinaParameters.OPLandIplParvo.photoreceptorsTemporalConstant, _retinaParameters.OPLandIplParvo.photoreceptorsSpatialConstant, _retinaParameters.OPLandIplParvo.horizontalCellsGain, _retinaParameters.OPLandIplParvo.hcellsTemporalConstant, _retinaParameters.OPLandIplParvo.hcellsSpatialConstant, _retinaParameters.OPLandIplParvo.ganglionCellsSensitivity);
161
162 // init retina IPL magno setup... update at the same time the parameters structure and the retina core
163 currFn=rootFn["IPLmagno"];
164 currFn["normaliseOutput"]>>_retinaParameters.IplMagno.normaliseOutput;
165 currFn["parasolCells_beta"]>>_retinaParameters.IplMagno.parasolCells_beta;
166 currFn["parasolCells_tau"]>>_retinaParameters.IplMagno.parasolCells_tau;
167 currFn["parasolCells_k"]>>_retinaParameters.IplMagno.parasolCells_k;
168 currFn["amacrinCellsTemporalCutFrequency"]>>_retinaParameters.IplMagno.amacrinCellsTemporalCutFrequency;
169 currFn["V0CompressionParameter"]>>_retinaParameters.IplMagno.V0CompressionParameter;
170 currFn["localAdaptintegration_tau"]>>_retinaParameters.IplMagno.localAdaptintegration_tau;
171 currFn["localAdaptintegration_k"]>>_retinaParameters.IplMagno.localAdaptintegration_k;
172
173 setupIPLMagnoChannel(_retinaParameters.IplMagno.normaliseOutput, _retinaParameters.IplMagno.parasolCells_beta, _retinaParameters.IplMagno.parasolCells_tau, _retinaParameters.IplMagno.parasolCells_k, _retinaParameters.IplMagno.amacrinCellsTemporalCutFrequency,_retinaParameters.IplMagno.V0CompressionParameter, _retinaParameters.IplMagno.localAdaptintegration_tau, _retinaParameters.IplMagno.localAdaptintegration_k);
174
175 }catch(Exception &e)
176 {
177 std::cout<<"Retina::setup: resetting retina with default parameters"<<std::endl;
178 if (applyDefaultSetupOnFailure)
179 {
180 setupOPLandIPLParvoChannel();
181 setupIPLMagnoChannel();
182 }
183 std::cout<<"Retina::setup: wrong/unappropriate xml parameter file : error report :`n=>"<<e.what()<<std::endl;
184 std::cout<<"=> keeping current parameters"<<std::endl;
185 }
186
187 // report current configuration
188 std::cout<<printSetup()<<std::endl;
189}

197 setupIPLMagnoChannel(_retinaParameters.IplMagno.normaliseOutput, _retinaParameters.IplMagno.parasolCells_beta, _retinaParameters.IplMagno.parasolCells_tau, _retinaParameters.IplMagno.parasolCells_k, _retinaParameters.IplMagno.amacrinCellsTemporalCutFrequency,_retinaParameters.IplMagno.V0CompressionParameter, _retinaParameters.IplMagno.localAdaptintegration_tau, _retinaParameters.IplMagno.localAdaptintegration_k);

275 _retinaFilter->activateNormalizeParvoOutput_0_maxOutputValue(normaliseOutput);
276
277 // update parameters struture
278
279 _retinaParameters.OPLandIplParvo.colorMode = colorMode;
280 _retinaParameters.OPLandIplParvo.normaliseOutput = normaliseOutput;
281 _retinaParameters.OPLandIplParvo.photoreceptorsLocalAdaptationSensitivity = photoreceptorsLocalAdaptationSensitivity;
282 _retinaParameters.OPLandIplParvo.photoreceptorsTemporalConstant = photoreceptorsTemporalConstant;
283 _retinaParameters.OPLandIplParvo.photoreceptorsSpatialConstant = photoreceptorsSpatialConstant;
284 _retinaParameters.OPLandIplParvo.horizontalCellsGain = horizontalCellsGain;
285 _retinaParameters.OPLandIplParvo.hcellsTemporalConstant = HcellsTemporalConstant;
286 _retinaParameters.OPLandIplParvo.hcellsSpatialConstant = HcellsSpatialConstant;

308void Retina::run(const cv::Mat &inputMatToConvert)
309{
310 // first convert input image to the compatible format : std::valarray<float>
311 const bool colorMode = _convertCvMat2ValarrayBuffer(inputMatToConvert, _inputBuffer);
312 // process the retina
313 if (!_retinaFilter->runFilter(_inputBuffer, colorMode, false, _retinaParameters.OPLandIplParvo.colorMode && colorMode, false))
314 throw cv::Exception(-1, "Retina cannot be applied, wrong input buffer size", "Retina::run", "Retina.h", 0);
315}

344// private method called by constructirs
345void Retina::_init(const cv::Size inputSz, const bool colorMode, RETINA_COLORSAMPLINGMETHOD colorSamplingMethod, const bool useRetinaLogSampling, const double reductionFactor, const double samplingStrenght)
346{
347 // basic error check
348 if (inputSz.height*inputSz.width <= 0)
349 throw cv::Exception(-1, "Bad retina size setup : size height and with must be superior to zero", "Retina::setup", "Retina.h", 0);
350
351 unsigned int nbPixels=inputSz.height*inputSz.width;
352 // resize buffers if size does not match
353 _inputBuffer.resize(nbPixels*3); // buffer supports gray images but also 3 channels color buffers... (larger is better...)
354
355 // allocate the retina model
356 if (_retinaFilter)
357 delete _retinaFilter;
358 _retinaFilter = new RetinaFilter(inputSz.height, inputSz.width, colorMode, colorSamplingMethod, useRetinaLogSampling, reductionFactor, samplingStrenght);
359
360 _retinaParameters.OPLandIplParvo.colorMode = colorMode;
361 // prepare the default parameter XML file with default setup
362 setup(_retinaParameters);
363
364 // init retina
365 _retinaFilter->clearAllBuffers();
366
367 // report current configuration
368 std::cout<<printSetup()<<std::endl;
369}

371void Retina::_convertValarrayBuffer2cvMat(const std::valarray<float> &grayMatrixToConvert, const unsigned int nbRows, const unsigned int nbColumns, const bool colorMode, cv::Mat &outBuffer)
372{
373 // fill output buffer with the valarray buffer
374 const float *valarrayPTR=get_data(grayMatrixToConvert);
375 if (!colorMode)
376 {
377 outBuffer.create(cv::Size(nbColumns, nbRows), CV_8U);
378 for (unsigned int i=0;i<nbRows;++i)
379 {
380 for (unsigned int j=0;j<nbColumns;++j)
381 {
382 cv::Point2d pixel(j,i);
383 outBuffer.at<unsigned char>(pixel)=(unsigned char)*(valarrayPTR++);
384 }
385 }
386 }else
387 {
388 const unsigned int doubleNBpixels=_retinaFilter->getOutputNBpixels()*2;
389 outBuffer.create(cv::Size(nbColumns, nbRows), CV_8UC3);
390 for (unsigned int i=0;i<nbRows;++i)
391 {
392 for (unsigned int j=0;j<nbColumns;++j,++valarrayPTR)
393 {
394 cv::Point2d pixel(j,i);
395 cv::Vec3b pixelValues;
396 pixelValues[2]=(unsigned char)*(valarrayPTR);
397 pixelValues[1]=(unsigned char)*(valarrayPTR+_retinaFilter->getOutputNBpixels());
398 pixelValues[0]=(unsigned char)*(valarrayPTR+doubleNBpixels);
399
400 outBuffer.at<cv::Vec3b>(pixel)=pixelValues;
401 }
402 }
403 }

406bool Retina::_convertCvMat2ValarrayBuffer(const cv::Mat inputMatToConvert, std::valarray<float> &outputValarrayMatrix)
407{
408 // first check input consistency
409 if (inputMatToConvert.empty())
410 throw cv::Exception(-1, "Retina cannot be applied, input buffer is empty", "Retina::run", "Retina.h", 0);
411
412 // retreive color mode from image input
413 int imageNumberOfChannels = inputMatToConvert.channels();
414
415 // convert to float AND fill the valarray buffer
416 typedef float T; // define here the target pixel format, here, float
417 const int dsttype = DataType<T>::depth; // output buffer is float format
418
419
420 if(imageNumberOfChannels==4)
421 {
422 // create a cv::Mat table (for RGBA planes)
423 cv::Mat planes[4] =
424 {
425 cv::Mat(inputMatToConvert.size(), dsttype, &outputValarrayMatrix[_retinaFilter->getInputNBpixels()*2]),
426 cv::Mat(inputMatToConvert.size(), dsttype, &outputValarrayMatrix[_retinaFilter->getInputNBpixels()]),
427 cv::Mat(inputMatToConvert.size(), dsttype, &outputValarrayMatrix[0])
428 };
429 planes[3] = cv::Mat(inputMatToConvert.size(), dsttype); // last channel (alpha) does not point on the valarray (not usefull in our case)
430 // split color cv::Mat in 4 planes... it fills valarray directely
431 cv::split(cv::Mat_<Vec<T, 4> >(inputMatToConvert), planes);
432 }
433 else if (imageNumberOfChannels==3)
434 {
435 // create a cv::Mat table (for RGB planes)
436 cv::Mat planes[] =
437 {
438 cv::Mat(inputMatToConvert.size(), dsttype, &outputValarrayMatrix[_retinaFilter->getInputNBpixels()*2]),
439 cv::Mat(inputMatToConvert.size(), dsttype, &outputValarrayMatrix[_retinaFilter->getInputNBpixels()]),
440 cv::Mat(inputMatToConvert.size(), dsttype, &outputValarrayMatrix[0])
441 };
442 // split color cv::Mat in 3 planes... it fills valarray directely
443 cv::split(cv::Mat_<Vec<T, 3> >(inputMatToConvert), planes);
444 }
445 else if(imageNumberOfChannels==1)
446 {
447 // create a cv::Mat header for the valarray
448 cv::Mat dst(inputMatToConvert.size(), dsttype, &outputValarrayMatrix[0]);
449 inputMatToConvert.convertTo(dst, dsttype);
450 }
451 else
452 CV_Error(CV_StsUnsupportedFormat, "input image must be single channel (gray levels), bgr format (color) or bgra (color with transparency which won't be considered");
453
454 return imageNumberOfChannels>1; // return bool : false for gray level image processing, true for color mode
455}
