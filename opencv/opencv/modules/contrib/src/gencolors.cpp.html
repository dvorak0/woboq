49static void downsamplePoints( const Mat& src, Mat& dst, size_t count )
50{
51 CV_Assert( count >= 2 );
52 CV_Assert( src.cols == 1 || src.rows == 1 );
53 CV_Assert( src.total() >= count );
54 CV_Assert( src.type() == CV_8UC3);
55
56 dst.create( 1, (int)count, CV_8UC3 );
57 //TODO: optimize by exploiting symmetry in the distance matrix
58 Mat dists( (int)src.total(), (int)src.total(), CV_32FC1, Scalar(0) );
59 if( dists.empty() )
60 std::cerr << "Such big matrix cann't be created." << std::endl;
61
62 for( int i = 0; i < dists.rows; i++ )
63 {
64 for( int j = i; j < dists.cols; j++ )
65 {
66 float dist = (float)norm(src.at<Point3_<uchar> >(i) - src.at<Point3_<uchar> >(j));
67 dists.at<float>(j, i) = dists.at<float>(i, j) = dist;
68 }
69 }
70
71 double maxVal;
72 Point maxLoc;
73 minMaxLoc(dists, 0, &maxVal, 0, &maxLoc);
74
75 dst.at<Point3_<uchar> >(0) = src.at<Point3_<uchar> >(maxLoc.x);
76 dst.at<Point3_<uchar> >(1) = src.at<Point3_<uchar> >(maxLoc.y);
77
78 Mat activedDists( 0, dists.cols, dists.type() );
79 Mat candidatePointsMask( 1, dists.cols, CV_8UC1, Scalar(255) );
80 activedDists.push_back( dists.row(maxLoc.y) );
81 candidatePointsMask.at<uchar>(0, maxLoc.y) = 0;
82
83 for( size_t i = 2; i < count; i++ )
84 {
85 activedDists.push_back(dists.row(maxLoc.x));
86 candidatePointsMask.at<uchar>(0, maxLoc.x) = 0;
87
88 Mat minDists;
89 reduce( activedDists, minDists, 0, CV_REDUCE_MIN );
90 minMaxLoc( minDists, 0, &maxVal, 0, &maxLoc, candidatePointsMask );
91 dst.at<Point3_<uchar> >((int)i) = src.at<Point3_<uchar> >(maxLoc.x);
92 }
93}

95void cv::generateColors( std::vector<Scalar>& colors, size_t count, size_t factor )
96{
97 if( count < 1 )
98 return;
99
100 colors.resize(count);
101
102 if( count == 1 )
103 {
104 colors[0] = Scalar(0,0,255); // red
105 return;
106 }
107 if( count == 2 )
108 {
109 colors[0] = Scalar(0,0,255); // red
110 colors[1] = Scalar(0,255,0); // green
111 return;
112 }
113
114 // Generate a set of colors in RGB space. A size of the set is severel times (=factor) larger then
115 // the needed count of colors.
116 Mat bgr( 1, (int)(count*factor), CV_8UC3 );
117 randu( bgr, 0, 256 );
118
119 // Convert the colors set to Lab space.
120 // Distances between colors in this space correspond a human perception.
121 Mat lab;
122 cvtColor( bgr, lab, CV_BGR2Lab);
123
124 // Subsample colors from the generated set so that
125 // to maximize the minimum distances between each other.
126 // Douglas-Peucker algorithm is used for this.
127 Mat lab_subset;
128 downsamplePoints( lab, lab_subset, count );
129
130 // Convert subsampled colors back to RGB
131 Mat bgr_subset;
132 cvtColor( lab_subset, bgr_subset, CV_Lab2BGR );
133
134 CV_Assert( bgr_subset.total() == count );
135 for( size_t i = 0; i < count; i++ )
136 {
137 Point3_<uchar> c = bgr_subset.at<Point3_<uchar> >((int)i);
138 colors[i] = Scalar(c.x, c.y, c.z);
139 }
140}
