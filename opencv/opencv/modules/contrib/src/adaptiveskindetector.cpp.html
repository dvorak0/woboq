41#define ASD_IS_IN_MOTION(pointer, v, threshold) ((abs((*(pointer)) - (v)) > (threshold)) ? true : false)

43void CvAdaptiveSkinDetector::initData(IplImage *src, int widthDivider, int heightDivider)
44{
45 CvSize imageSize = cvSize(src->width/widthDivider, src->height/heightDivider);
46
47 imgHueFrame = cvCreateImage(imageSize, IPL_DEPTH_8U, 1);
48 imgShrinked = cvCreateImage(imageSize, IPL_DEPTH_8U, src->nChannels);
49 imgSaturationFrame = cvCreateImage(imageSize, IPL_DEPTH_8U, 1);
50 imgMotionFrame = cvCreateImage(imageSize, IPL_DEPTH_8U, 1);
51 imgTemp = cvCreateImage(imageSize, IPL_DEPTH_8U, 1);
52 imgFilteredFrame = cvCreateImage(imageSize, IPL_DEPTH_8U, 1);
53 imgGrayFrame = cvCreateImage(imageSize, IPL_DEPTH_8U, 1);
54 imgLastGrayFrame = cvCreateImage(imageSize, IPL_DEPTH_8U, 1);
55 imgHSVFrame = cvCreateImage(imageSize, IPL_DEPTH_8U, 3);
56}

58CvAdaptiveSkinDetector::CvAdaptiveSkinDetector(int samplingDivider, int morphingMethod)
59{
60 nSkinHueLowerBound = GSD_HUE_LT;
61 nSkinHueUpperBound = GSD_HUE_UT;
62
63 fHistogramMergeFactor = 0.05; // empirical result
64 fHuePercentCovered = 0.95; // empirical result
65
66 nMorphingMethod = morphingMethod;
67 nSamplingDivider = samplingDivider;
68
69 nFrameCount = 0;
70 nStartCounter = 0;
71
72 imgHueFrame = NULL;
73 imgMotionFrame = NULL;
74 imgTemp = NULL;
75 imgFilteredFrame = NULL;
76 imgShrinked = NULL;
77 imgGrayFrame = NULL;
78 imgLastGrayFrame = NULL;
79 imgSaturationFrame = NULL;
80 imgHSVFrame = NULL;
81}

96void CvAdaptiveSkinDetector::process(IplImage *inputBGRImage, IplImage *outputHueMask)
97{
98 IplImage *src = inputBGRImage;
99
100 int h, v, i, l;
101 bool isInit = false;
102
103 nFrameCount++;
104
105 if (imgHueFrame == NULL)
106 {
107 isInit = true;
108 initData(src, nSamplingDivider, nSamplingDivider);
109 }
110
111 unsigned char *pShrinked, *pHueFrame, *pMotionFrame, *pLastGrayFrame, *pFilteredFrame, *pGrayFrame;
112 pShrinked = (unsigned char *)imgShrinked->imageData;
113 pHueFrame = (unsigned char *)imgHueFrame->imageData;
114 pMotionFrame = (unsigned char *)imgMotionFrame->imageData;
115 pLastGrayFrame = (unsigned char *)imgLastGrayFrame->imageData;
116 pFilteredFrame = (unsigned char *)imgFilteredFrame->imageData;
117 pGrayFrame = (unsigned char *)imgGrayFrame->imageData;
118
119 if ((src->width != imgHueFrame->width) || (src->height != imgHueFrame->height))
120 {
121 cvResize(src, imgShrinked);
122 cvCvtColor(imgShrinked, imgHSVFrame, CV_BGR2HSV);
123 }
124 else
125 {
126 cvCvtColor(src, imgHSVFrame, CV_BGR2HSV);
127 }
128
129 cvSplit(imgHSVFrame, imgHueFrame, imgSaturationFrame, imgGrayFrame, 0);
130
131 cvSetZero(imgMotionFrame);
132 cvSetZero(imgFilteredFrame);
133
134 l = imgHueFrame->height * imgHueFrame->width;
135
136 for (i = 0; i < l; i++)
137 {
138 v = (*pGrayFrame);
139 if ((v >= GSD_INTENSITY_LT) && (v <= GSD_INTENSITY_UT))
140 {
141 h = (*pHueFrame);
142 if ((h >= GSD_HUE_LT) && (h <= GSD_HUE_UT))
143 {
144 if ((h >= nSkinHueLowerBound) && (h <= nSkinHueUpperBound))
145 ASD_INTENSITY_SET_PIXEL(pFilteredFrame, h);
146
147 if (ASD_IS_IN_MOTION(pLastGrayFrame, v, 7))
148 ASD_INTENSITY_SET_PIXEL(pMotionFrame, h);
149 }
150 }
151 pShrinked += 3;
152 pGrayFrame++;
153 pLastGrayFrame++;
154 pMotionFrame++;
155 pHueFrame++;
156 pFilteredFrame++;
157 }
158
159 if (isInit)
160 cvCalcHist(&imgHueFrame, skinHueHistogram.fHistogram);
161
162 cvCopy(imgGrayFrame, imgLastGrayFrame);
163
164 cvErode(imgMotionFrame, imgTemp); // eliminate disperse pixels, which occur because of the camera noise
165 cvDilate(imgTemp, imgMotionFrame);
166
167 cvCalcHist(&imgMotionFrame, histogramHueMotion.fHistogram);
168
169 skinHueHistogram.mergeWith(&histogramHueMotion, fHistogramMergeFactor);
170
171 skinHueHistogram.findCurveThresholds(nSkinHueLowerBound, nSkinHueUpperBound, 1 - fHuePercentCovered);
172
173 switch (nMorphingMethod)
174 {
175 case MORPHING_METHOD_ERODE :
176 cvErode(imgFilteredFrame, imgTemp);
177 cvCopy(imgTemp, imgFilteredFrame);
178 break;
179 case MORPHING_METHOD_ERODE_ERODE :
180 cvErode(imgFilteredFrame, imgTemp);
181 cvErode(imgTemp, imgFilteredFrame);
182 break;
183 case MORPHING_METHOD_ERODE_DILATE :
184 cvErode(imgFilteredFrame, imgTemp);
185 cvDilate(imgTemp, imgFilteredFrame);
186 break;
187 }
188
189 if (outputHueMask != NULL)
190 cvCopy(imgFilteredFrame, outputHueMask);
191}

196CvAdaptiveSkinDetector::Histogram::Histogram()
197{
198 int histogramSize[] = { HistogramSize };
199 float range[] = { GSD_HUE_LT, GSD_HUE_UT };
200 float *ranges[] = { range };
201 fHistogram = cvCreateHist(1, histogramSize, CV_HIST_ARRAY, ranges, 1);

210int CvAdaptiveSkinDetector::Histogram::findCoverageIndex(double surfaceToCover, int defaultValue)
211{
212 double s = 0;
213 for (int i = 0; i < HistogramSize; i++)
214 {
215 s += cvGetReal1D( fHistogram->bins, i );
216 if (s >= surfaceToCover)
217 {
218 return i;
219 }
220 }
221 return defaultValue;
222}

224void CvAdaptiveSkinDetector::Histogram::findCurveThresholds(int &x1, int &x2, double percent)
225{
226 double sum = 0;
227
228 for (int i = 0; i < HistogramSize; i++)
229 {
230 sum += cvGetReal1D( fHistogram->bins, i );
231 }
232
233 x1 = findCoverageIndex(sum * percent, -1);
234 x2 = findCoverageIndex(sum * (1-percent), -1);
235
236 if (x1 == -1)
237 x1 = GSD_HUE_LT;
238 else
239 x1 += GSD_HUE_LT;
240
241 if (x2 == -1)
242 x2 = GSD_HUE_UT;
243 else
244 x2 += GSD_HUE_LT;
245}

247void CvAdaptiveSkinDetector::Histogram::mergeWith(CvAdaptiveSkinDetector::Histogram *source, double weight)
248{
249 float myweight = (float)(1-weight);
250 float maxVal1 = 0, maxVal2 = 0, *f1, *f2, ff1, ff2;
251
252 cvGetMinMaxHistValue(source->fHistogram, NULL, &maxVal2);
253
254 if (maxVal2 > 0 )
255 {
256 cvGetMinMaxHistValue(fHistogram, NULL, &maxVal1);
257 if (maxVal1 <= 0)
258 {
259 for (int i = 0; i < HistogramSize; i++)
260 {
261 f1 = (float*)cvPtr1D(fHistogram->bins, i);
262 f2 = (float*)cvPtr1D(source->fHistogram->bins, i);
263 (*f1) = (*f2);
264 }
265 }
266 else
267 {
268 for (int i = 0; i < HistogramSize; i++)
269 {
270 f1 = (float*)cvPtr1D(fHistogram->bins, i);
271 f2 = (float*)cvPtr1D(source->fHistogram->bins, i);
272
273 ff1 = ((*f1)/maxVal1)*myweight;
274 if (ff1 < 0)
275 ff1 = -ff1;
276
277 ff2 = (float)(((*f2)/maxVal2)*weight);
278 if (ff2 < 0)
279 ff2 = -ff2;
280
281 (*f1) = (ff1 + ff2);
282
283 }
284 }
