26// Reads a sequence from a FileNode::SEQ with type _Tp into a result vector.
27template<typename _Tp>
28inline void readFileNodeList(const FileNode& fn, vector<_Tp>& result) {
29 if (fn.type() == FileNode::SEQ) {
30 for (FileNodeIterator it = fn.begin(); it != fn.end();) {
31 _Tp item;
32 it >> item;
33 result.push_back(item);
34 }
35 }
36}

38// Writes the a list of given items to a cv::FileStorage.
39template<typename _Tp>
40inline void writeFileNodeList(FileStorage& fs, const string& name,
41 const vector<_Tp>& items) {
42 // typedefs
43 typedef typename vector<_Tp>::const_iterator constVecIterator;
44 // write the elements in item to fs
45 fs << name << "[";
46 for (constVecIterator it = items.begin(); it != items.end(); ++it) {
47 fs << *it;
48 }
49 fs << "]";
50}

52static Mat asRowMatrix(InputArrayOfArrays src, int rtype, double alpha=1, double beta=0) {
53 // make sure the input data is a vector of matrices or vector of vector
54 if(src.kind() != _InputArray::STD_VECTOR_MAT && src.kind() != _InputArray::STD_VECTOR_VECTOR) {
55 string error_message = "The data is expected as InputArray::STD_VECTOR_MAT (a std::vector<Mat>) or _InputArray::STD_VECTOR_VECTOR (a std::vector< vector<...> >).";
56 CV_Error(CV_StsBadArg, error_message);
57 }
58 // number of samples
59 size_t n = src.total();
60 // return empty matrix if no matrices given
61 if(n == 0)
62 return Mat();
63 // dimensionality of (reshaped) samples
64 size_t d = src.getMat(0).total();
65 // create data matrix
66 Mat data((int)n, (int)d, rtype);
67 // now copy data
68 for(unsigned int i = 0; i < n; i++) {
69 // make sure data can be reshaped, throw exception if not!
70 if(src.getMat(i).total() != d) {
71 string error_message = format("Wrong number of elements in matrix #%d! Expected %d was %d.", i, d, src.getMat(i).total());
72 CV_Error(CV_StsBadArg, error_message);
73 }
74 // get a hold of the current row
75 Mat xi = data.row(i);
76 // make reshape happy by cloning for non-continuous matrices
77 if(src.getMat(i).isContinuous()) {
78 src.getMat(i).reshape(1, 1).convertTo(xi, rtype, alpha, beta);
79 } else {
80 src.getMat(i).clone().reshape(1, 1).convertTo(xi, rtype, alpha, beta);
81 }
82 }
83 return data;
84}

87// Removes duplicate elements in a given vector.
88template<typename _Tp>
89inline vector<_Tp> remove_dups(const vector<_Tp>& src) {
90 typedef typename set<_Tp>::const_iterator constSetIterator;
91 typedef typename vector<_Tp>::const_iterator constVecIterator;
92 set<_Tp> set_elems;
93 for (constVecIterator it = src.begin(); it != src.end(); ++it)
94 set_elems.insert(*it);
95 vector<_Tp> elems;
96 for (constSetIterator it = set_elems.begin(); it != set_elems.end(); ++it)
97 elems.push_back(*it);
98 return elems;
99}

101// The FaceRecognizer2 class is introduced to keep the FaceRecognizer binary backward compatibility in 2.4
102// In master setLabelInfo/getLabelInfo/getLabelsByString should be virtual and _labelsInfo should be moved
103// to FaceRecognizer, that allows to avoid FaceRecognizer2 in master
104class FaceRecognizer2 : public FaceRecognizer
105{
106protected:
107 // Stored pairs "label id - string info"
108 std::map<int, string> _labelsInfo;
109
110public:
111 // Sets additional information as pairs label - info.
112 virtual void setLabelsInfo(const std::map<int, string>& labelsInfo)
113 {
114 _labelsInfo = labelsInfo;
115 }
116
117 // Gets string information by label
118 virtual string getLabelInfo(int label) const
119 {
120 std::map<int, string>::const_iterator iter(_labelsInfo.find(label));
121 return iter != _labelsInfo.end() ? iter->second : "";
122 }
123
124 // Gets labels by string
125 virtual vector<int> getLabelsByString(const string& str)
126 {
127 vector<int> labels;
128 for(std::map<int,string>::const_iterator it = _labelsInfo.begin(); it != _labelsInfo.end(); it++)
129 {
130 size_t found = (it->second).find(str);
131 if(found != string::npos)
132 labels.push_back(it->first);
133 }
134 return labels;
135 }
136
137};

139// Utility structure to load/save face label info (a pair of int and string) via FileStorage
140struct LabelInfo
141{
142 LabelInfo():label(-1), value("") {}
143 LabelInfo(int _label, const std::string &_value): label(_label), value(_value) {}
144 int label;
145 std::string value;
146 void write(cv::FileStorage& fs) const
147 {
148 fs << "{" << "label" << label << "value" << value << "}";
149 }
150 void read(const cv::FileNode& node)
151 {
152 label = (int)node["label"];
153 value = (std::string)node["value"];
154 }
155 std::ostream& operator<<(std::ostream& out)
156 {
157 out << "{ label = " << label << ", " << "value = " << value << "}";
158 return out;
159 }
160};

163{
164 x.write(fs);

167static void read(const cv::FileNode& node, LabelInfo& x, const LabelInfo& default_value = LabelInfo())
168{
169 if(node.empty())
170 x = default_value;
171 else
172 x.read(node);

298 Mat _labels;

301 // corresponding labels in labels, possibly preserving
302 // old model data.
303 void train(InputArrayOfArrays src, InputArray labels, bool preserveData);
304
305public:
306 using FaceRecognizer::save;
307 using FaceRecognizer::load;
308
309 // Initializes this LBPH Model. The current implementation is rather fixed
310 // as it uses the Extended Local Binary Patterns per default.
311 //
312 // radius, neighbors are used in the local binary patterns creation.
313 // grid_x, grid_y control the grid size of the spatial histograms.
314 LBPH(int radius_=1, int neighbors_=8,
315 int gridx=8, int gridy=8,

317 _grid_x(gridx),
318 _grid_y(gridy),

330 int radius_=1, int neighbors_=8,

427void Eigenfaces::train(InputArrayOfArrays _src, InputArray _local_labels) {
428 if(_src.total() == 0) {
429 string error_message = format("Empty training data was given. You'll need more than one sample to learn a model.");
430 CV_Error(CV_StsBadArg, error_message);
431 } else if(_local_labels.getMat().type() != CV_32SC1) {
432 string error_message = format("Labels must be given as integer (CV_32SC1). Expected %d, but was %d.", CV_32SC1, _local_labels.type());
433 CV_Error(CV_StsBadArg, error_message);
434 }
435 // make sure data has correct size
436 if(_src.total() > 1) {
437 for(int i = 1; i < static_cast<int>(_src.total()); i++) {
438 if(_src.getMat(i-1).total() != _src.getMat(i).total()) {
439 string error_message = format("In the Eigenfaces method all input samples (training images) must be of equal size! Expected %d pixels, but was %d pixels.", _src.getMat(i-1).total(), _src.getMat(i).total());
440 CV_Error(CV_StsUnsupportedFormat, error_message);
441 }
442 }
443 }
444 // get labels
445 Mat labels = _local_labels.getMat();
446 // observations in row
447 Mat data = asRowMatrix(_src, CV_64FC1);
448
449 // number of samples
450 int n = data.rows;
451 // assert there are as much samples as labels
452 if(static_cast<int>(labels.total()) != n) {
453 string error_message = format("The number of samples (src) must equal the number of labels (labels)! len(src)=%d, len(labels)=%d.", n, labels.total());
454 CV_Error(CV_StsBadArg, error_message);
455 }
456 // clear existing model data
457 _labels.release();
458 _projections.clear();
459 // clip number of components to be valid
460 if((_num_components <= 0) || (_num_components > n))
461 _num_components = n;
462
463 // perform the PCA
464 PCA pca(data, Mat(), CV_PCA_DATA_AS_ROW, _num_components);
465 // copy the PCA results
466 _mean = pca.mean.reshape(1,1); // store the mean vector
467 _eigenvalues = pca.eigenvalues.clone(); // eigenvalues by row
468 transpose(pca.eigenvectors, _eigenvectors); // eigenvectors by column
469 // store labels for prediction
470 _labels = labels.clone();
471 // save projections
472 for(int sampleIdx = 0; sampleIdx < data.rows; sampleIdx++) {
473 Mat p = subspaceProject(_eigenvectors, _mean, data.row(sampleIdx));
474 _projections.push_back(p);
475 }
476}

478void Eigenfaces::predict(InputArray _src, int &minClass, double &minDist) const {
479 // get data
480 Mat src = _src.getMat();
481 // make sure the user is passing correct data
482 if(_projections.empty()) {
483 // throw error if no data (or simply return -1?)
484 string error_message = "This Eigenfaces model is not computed yet. Did you call Eigenfaces::train?";
485 CV_Error(CV_StsError, error_message);
486 } else if(_eigenvectors.rows != static_cast<int>(src.total())) {
487 // check data alignment just for clearer exception messages
488 string error_message = format("Wrong input image size. Reason: Training and Test images must be of equal size! Expected an image with %d elements, but got %d.", _eigenvectors.rows, src.total());
489 CV_Error(CV_StsBadArg, error_message);
490 }
491 // project into PCA subspace
492 Mat q = subspaceProject(_eigenvectors, _mean, src.reshape(1,1));
493 minDist = DBL_MAX;
494 minClass = -1;
495 for(size_t sampleIdx = 0; sampleIdx < _projections.size(); sampleIdx++) {
496 double dist = norm(_projections[sampleIdx], q, NORM_L2);
497 if((dist < minDist) && (dist < _threshold)) {
498 minDist = dist;
499 minClass = _labels.at<int>((int)sampleIdx);
500 }
501 }
502}

504int Eigenfaces::predict(InputArray _src) const {
505 int label;
506 double dummy;
507 predict(_src, label, dummy);
508 return label;
509}

548//------------------------------------------------------------------------------
549// Fisherfaces
550//------------------------------------------------------------------------------
551void Fisherfaces::train(InputArrayOfArrays src, InputArray _lbls) {
552 if(src.total() == 0) {
553 string error_message = format("Empty training data was given. You'll need more than one sample to learn a model.");
554 CV_Error(CV_StsBadArg, error_message);
555 } else if(_lbls.getMat().type() != CV_32SC1) {
556 string error_message = format("Labels must be given as integer (CV_32SC1). Expected %d, but was %d.", CV_32SC1, _lbls.type());
557 CV_Error(CV_StsBadArg, error_message);
558 }
559 // make sure data has correct size
560 if(src.total() > 1) {
561 for(int i = 1; i < static_cast<int>(src.total()); i++) {
562 if(src.getMat(i-1).total() != src.getMat(i).total()) {
563 string error_message = format("In the Fisherfaces method all input samples (training images) must be of equal size! Expected %d pixels, but was %d pixels.", src.getMat(i-1).total(), src.getMat(i).total());
564 CV_Error(CV_StsUnsupportedFormat, error_message);
565 }
566 }
567 }
568 // get data
569 Mat labels = _lbls.getMat();
570 Mat data = asRowMatrix(src, CV_64FC1);
571 // number of samples
572 int N = data.rows;
573 // make sure labels are passed in correct shape
574 if(labels.total() != (size_t) N) {
575 string error_message = format("The number of samples (src) must equal the number of labels (labels)! len(src)=%d, len(labels)=%d.", N, labels.total());
576 CV_Error(CV_StsBadArg, error_message);
577 } else if(labels.rows != 1 && labels.cols != 1) {
578 string error_message = format("Expected the labels in a matrix with one row or column! Given dimensions are rows=%s, cols=%d.", labels.rows, labels.cols);
579 CV_Error(CV_StsBadArg, error_message);
580 }
581 // clear existing model data
582 _labels.release();
583 _projections.clear();
584 // safely copy from cv::Mat to std::vector
585 vector<int> ll;
586 for(unsigned int i = 0; i < labels.total(); i++) {
587 ll.push_back(labels.at<int>(i));
588 }
589 // get the number of unique classes
590 int C = (int) remove_dups(ll).size();
591 // clip number of components to be a valid number
592 if((_num_components <= 0) || (_num_components > (C-1)))
593 _num_components = (C-1);
594 // perform a PCA and keep (N-C) components
595 PCA pca(data, Mat(), CV_PCA_DATA_AS_ROW, (N-C));
596 // project the data and perform a LDA on it
597 LDA lda(pca.project(data),labels, _num_components);
598 // store the total mean vector
599 _mean = pca.mean.reshape(1,1);
600 // store labels
601 _labels = labels.clone();
602 // store the eigenvalues of the discriminants
603 lda.eigenvalues().convertTo(_eigenvalues, CV_64FC1);
604 // Now calculate the projection matrix as pca.eigenvectors * lda.eigenvectors.
605 // Note: OpenCV stores the eigenvectors by row, so we need to transpose it!
606 gemm(pca.eigenvectors, lda.eigenvectors(), 1.0, Mat(), 0.0, _eigenvectors, GEMM_1_T);
607 // store the projections of the original data
608 for(int sampleIdx = 0; sampleIdx < data.rows; sampleIdx++) {
609 Mat p = subspaceProject(_eigenvectors, _mean, data.row(sampleIdx));
610 _projections.push_back(p);
611 }
612}

614void Fisherfaces::predict(InputArray _src, int &minClass, double &minDist) const {
615 Mat src = _src.getMat();
616 // check data alignment just for clearer exception messages
617 if(_projections.empty()) {
618 // throw error if no data (or simply return -1?)
619 string error_message = "This Fisherfaces model is not computed yet. Did you call Fisherfaces::train?";
620 CV_Error(CV_StsBadArg, error_message);
621 } else if(src.total() != (size_t) _eigenvectors.rows) {
622 string error_message = format("Wrong input image size. Reason: Training and Test images must be of equal size! Expected an image with %d elements, but got %d.", _eigenvectors.rows, src.total());
623 CV_Error(CV_StsBadArg, error_message);
624 }
625 // project into LDA subspace
626 Mat q = subspaceProject(_eigenvectors, _mean, src.reshape(1,1));
627 // find 1-nearest neighbor
628 minDist = DBL_MAX;
629 minClass = -1;
630 for(size_t sampleIdx = 0; sampleIdx < _projections.size(); sampleIdx++) {
631 double dist = norm(_projections[sampleIdx], q, NORM_L2);
632 if((dist < minDist) && (dist < _threshold)) {
633 minDist = dist;
634 minClass = _labels.at<int>((int)sampleIdx);
635 }
636 }
637}

689template <typename _Tp> static
690void olbp_(InputArray _src, OutputArray _dst) {
691 // get matrices
692 Mat src = _src.getMat();
693 // allocate memory for result
694 _dst.create(src.rows-2, src.cols-2, CV_8UC1);
695 Mat dst = _dst.getMat();
696 // zero the result matrix
697 dst.setTo(0);
698 // calculate patterns
699 for(int i=1;i<src.rows-1;i++) {
700 for(int j=1;j<src.cols-1;j++) {
701 _Tp center = src.at<_Tp>(i,j);
702 unsigned char code = 0;
703 code |= (src.at<_Tp>(i-1,j-1) >= center) << 7;
704 code |= (src.at<_Tp>(i-1,j) >= center) << 6;
705 code |= (src.at<_Tp>(i-1,j+1) >= center) << 5;
706 code |= (src.at<_Tp>(i,j+1) >= center) << 4;
707 code |= (src.at<_Tp>(i+1,j+1) >= center) << 3;
708 code |= (src.at<_Tp>(i+1,j) >= center) << 2;
709 code |= (src.at<_Tp>(i+1,j-1) >= center) << 1;
710 code |= (src.at<_Tp>(i,j-1) >= center) << 0;
711 dst.at<unsigned char>(i-1,j-1) = code;
712 }
713 }
714}

716//------------------------------------------------------------------------------
717// cv::elbp
718//------------------------------------------------------------------------------
719template <typename _Tp> static
720inline void elbp_(InputArray _src, OutputArray _dst, int radius, int neighbors) {
721 //get matrices
722 Mat src = _src.getMat();
723 // allocate memory for result
724 _dst.create(src.rows-2*radius, src.cols-2*radius, CV_32SC1);
725 Mat dst = _dst.getMat();
726 // zero
727 dst.setTo(0);
728 for(int n=0; n<neighbors; n++) {
729 // sample points
730 float x = static_cast<float>(radius * cos(2.0*CV_PI*n/static_cast<float>(neighbors)));
731 float y = static_cast<float>(-radius * sin(2.0*CV_PI*n/static_cast<float>(neighbors)));
732 // relative indices
733 int fx = static_cast<int>(floor(x));
734 int fy = static_cast<int>(floor(y));
735 int cx = static_cast<int>(ceil(x));
736 int cy = static_cast<int>(ceil(y));
737 // fractional part
738 float ty = y - fy;
739 float tx = x - fx;
740 // set interpolation weights
741 float w1 = (1 - tx) * (1 - ty);
742 float w2 = tx * (1 - ty);
743 float w3 = (1 - tx) * ty;
744 float w4 = tx * ty;
745 // iterate through your data
746 for(int i=radius; i < src.rows-radius;i++) {
747 for(int j=radius;j < src.cols-radius;j++) {
748 // calculate interpolated value
749 float t = static_cast<float>(w1*src.at<_Tp>(i+fy,j+fx) + w2*src.at<_Tp>(i+fy,j+cx) + w3*src.at<_Tp>(i+cy,j+fx) + w4*src.at<_Tp>(i+cy,j+cx));
750 // floating point precision, so check some machine-dependent epsilon
751 dst.at<int>(i-radius,j-radius) += ((t > src.at<_Tp>(i,j)) || (std::abs(t-src.at<_Tp>(i,j)) < std::numeric_limits<float>::epsilon())) << n;
752 }
753 }
754 }
755}

757static void elbp(InputArray src, OutputArray dst, int radius, int neighbors)
758{
759 int type = src.type();
760 switch (type) {
761 case CV_8SC1: elbp_<char>(src,dst, radius, neighbors); break;
762 case CV_8UC1: elbp_<unsigned char>(src, dst, radius, neighbors); break;
763 case CV_16SC1: elbp_<short>(src,dst, radius, neighbors); break;
764 case CV_16UC1: elbp_<unsigned short>(src,dst, radius, neighbors); break;
765 case CV_32SC1: elbp_<int>(src,dst, radius, neighbors); break;
766 case CV_32FC1: elbp_<float>(src,dst, radius, neighbors); break;
767 case CV_64FC1: elbp_<double>(src,dst, radius, neighbors); break;
768 default:
769 string error_msg = format("Using Original Local Binary Patterns for feature extraction only works on single-channel images (given %d). Please pass the image data as a grayscale image!", type);
770 CV_Error(CV_StsNotImplemented, error_msg);
771 break;
772 }
773}

775static Mat
776histc_(const Mat& src, int minVal=0, int maxVal=255, bool normed=false)
777{
778 Mat result;
779 // Establish the number of bins.
780 int histSize = maxVal-minVal+1;
781 // Set the ranges.
782 float range[] = { static_cast<float>(minVal), static_cast<float>(maxVal+1) };
783 const float* histRange = { range };
784 // calc histogram
785 calcHist(&src, 1, 0, Mat(), result, 1, &histSize, &histRange, true, false);
786 // normalize
787 if(normed) {
788 result /= (int)src.total();
789 }
790 return result.reshape(1,1);
791}

793static Mat histc(InputArray _src, int minVal, int maxVal, bool normed)
794{
795 Mat src = _src.getMat();
796 switch (src.type()) {
797 case CV_8SC1:
798 return histc_(Mat_<float>(src), minVal, maxVal, normed);
799 break;
800 case CV_8UC1:
801 return histc_(src, minVal, maxVal, normed);
802 break;
803 case CV_16SC1:
804 return histc_(Mat_<float>(src), minVal, maxVal, normed);
805 break;
806 case CV_16UC1:
807 return histc_(src, minVal, maxVal, normed);
808 break;
809 case CV_32SC1:
810 return histc_(Mat_<float>(src), minVal, maxVal, normed);
811 break;
812 case CV_32FC1:
813 return histc_(src, minVal, maxVal, normed);
814 break;
815 default:
816 CV_Error(CV_StsUnmatchedFormats, "This type is not implemented yet."); break;
817 }
818 return Mat();
819}

822static Mat spatial_histogram(InputArray _src, int numPatterns,
823 int grid_x, int grid_y, bool /*normed*/)
824{
825 Mat src = _src.getMat();
826 // calculate LBP patch size
827 int width = src.cols/grid_x;
828 int height = src.rows/grid_y;
829 // allocate memory for the spatial histogram
830 Mat result = Mat::zeros(grid_x * grid_y, numPatterns, CV_32FC1);
831 // return matrix with zeros if no data was given
832 if(src.empty())
833 return result.reshape(1,1);
834 // initial result_row
835 int resultRowIdx = 0;
836 // iterate through grid
837 for(int i = 0; i < grid_y; i++) {
838 for(int j = 0; j < grid_x; j++) {
839 Mat src_cell = Mat(src, Range(i*height,(i+1)*height), Range(j*width,(j+1)*width));
840 Mat cell_hist = histc(src_cell, 0, (numPatterns-1), true);
841 // copy to the result matrix
842 Mat result_row = result.row(resultRowIdx);
843 cell_hist.reshape(1,1).convertTo(result_row, CV_32FC1);
844 // increase row count in result matrix
845 resultRowIdx++;
846 }
847 }
848 // return result as reshaped feature vector
849 return result.reshape(1,1);
850}

910void LBPH::train(InputArrayOfArrays _in_src, InputArray _in_labels, bool preserveData) {
911 if(_in_src.kind() != _InputArray::STD_VECTOR_MAT && _in_src.kind() != _InputArray::STD_VECTOR_VECTOR) {
912 string error_message = "The images are expected as InputArray::STD_VECTOR_MAT (a std::vector<Mat>) or _InputArray::STD_VECTOR_VECTOR (a std::vector< vector<...> >).";
913 CV_Error(CV_StsBadArg, error_message);
914 }
915 if(_in_src.total() == 0) {
916 string error_message = format("Empty training data was given. You'll need more than one sample to learn a model.");
917 CV_Error(CV_StsUnsupportedFormat, error_message);
918 } else if(_in_labels.getMat().type() != CV_32SC1) {
919 string error_message = format("Labels must be given as integer (CV_32SC1). Expected %d, but was %d.", CV_32SC1, _in_labels.type());
920 CV_Error(CV_StsUnsupportedFormat, error_message);
921 }
922 // get the vector of matrices
923 vector<Mat> src;
924 _in_src.getMatVector(src);
925 // get the label matrix
926 Mat labels = _in_labels.getMat();
927 // check if data is well- aligned
928 if(labels.total() != src.size()) {
929 string error_message = format("The number of samples (src) must equal the number of labels (labels). Was len(samples)=%d, len(labels)=%d.", src.size(), _labels.total());
930 CV_Error(CV_StsBadArg, error_message);
931 }
932 // if this model should be trained without preserving old data, delete old model data
933 if(!preserveData) {
934 _labels.release();
935 _histograms.clear();
936 }
937 // append labels to _labels matrix
938 for(size_t labelIdx = 0; labelIdx < labels.total(); labelIdx++) {
939 _labels.push_back(labels.at<int>((int)labelIdx));
940 }
941 // store the spatial histograms of the original data
942 for(size_t sampleIdx = 0; sampleIdx < src.size(); sampleIdx++) {
943 // calculate lbp image
944 Mat lbp_image = elbp(src[sampleIdx], _radius, _neighbors);
945 // get spatial histogram from this lbp image
946 Mat p = spatial_histogram(
947 lbp_image, /* lbp_image */
948 static_cast<int>(std::pow(2.0, static_cast<double>(_neighbors))), /* number of possible patterns */
949 _grid_x, /* grid size x */
950 _grid_y, /* grid size y */
951 true);
952 // add to templates
953 _histograms.push_back(p);
954 }
955}
956
957void LBPH::predict(InputArray _src, int &minClass, double &minDist) const {
958 if(_histograms.empty()) {
959 // throw error if no data (or simply return -1?)
960 string error_message = "This LBPH model is not computed yet. Did you call the train method?";
961 CV_Error(CV_StsBadArg, error_message);
962 }
963 Mat src = _src.getMat();
964 // get the spatial histogram from input image
965 Mat lbp_image = elbp(src, _radius, _neighbors);
966 Mat query = spatial_histogram(
967 lbp_image, /* lbp_image */
968 static_cast<int>(std::pow(2.0, static_cast<double>(_neighbors))), /* number of possible patterns */
969 _grid_x, /* grid size x */
970 _grid_y, /* grid size y */
971 true /* normed histograms */);
972 // find 1-nearest neighbor
973 minDist = DBL_MAX;
974 minClass = -1;
975 for(size_t sampleIdx = 0; sampleIdx < _histograms.size(); sampleIdx++) {
976 double dist = compareHist(_histograms[sampleIdx], query, CV_COMP_CHISQR);
977 if((dist < minDist) && (dist < _threshold)) {
978 minDist = dist;
979 minClass = _labels.at<int>((int) sampleIdx);
980 }
981 }
982}

985 int label;
986 double dummy;
987 predict(_src, label, dummy);
988 return label;
989}
