68/**
69* @class MagnoRetinaFilter
70* @brief class which describes the magnocellular channel of the retina:
71* -> performs a moving contours extraction with powerfull local data enhancement
72*
73* TYPICAL USE:
74*
75* // create object at a specified picture size
76* MagnoRetinaFilter *movingContoursExtractor;
77* movingContoursExtractor =new MagnoRetinaFilter(frameSizeRows, frameSizeColumns);
78*
79* // init gain, spatial and temporal parameters:
80* movingContoursExtractor->setCoefficientsTable(0, 0.7, 5, 3);
81*
82* // during program execution, call the filter for contours extraction for an input picture called "FrameBuffer":
83* movingContoursExtractor->runfilter(FrameBuffer);
84*
85* // get the output frame, check in the class description below for more outputs:
86* const float *movingContours=movingContoursExtractor->getMagnoYsaturated();
87*
88* // at the end of the program, destroy object:
89* delete movingContoursExtractor;
90
91* @author Alexandre BENOIT, benoit.alexandre.vision@gmail.com, LISTIC : www.listic.univ-savoie.fr, Gipsa-Lab, France: www.gipsa-lab.inpg.fr/
92* Creation date 2007
93* Based on Alexandre BENOIT thesis: "Le syst√®me visuel humain au secours de la vision par ordinateur"
94*/

124 /**
125 * resize retina magno filter object (resize all allocated buffers)
126 * @param NBrows: the new height size

131 /**
132 * set parameters values
133 * @param parasolCells_beta: the low pass filter gain used for local contrast adaptation at the IPL level of the retina (for ganglion cells local adaptation), typical value is 0
134 * @param parasolCells_tau: the low pass filter time constant used for local contrast adaptation at the IPL level of the retina (for ganglion cells local adaptation), unit is frame, typical value is 0 (immediate response)
135 * @param parasolCells_k: the low pass filter spatial constant used for local contrast adaptation at the IPL level of the retina (for ganglion cells local adaptation), unit is pixels, typical value is 5
136 * @param amacrinCellsTemporalCutFrequency: the time constant of the first order high pass fiter of the magnocellular way (motion information channel), unit is frames, tipicall value is 5
137 * @param localAdaptIntegration_tau: specifies the temporal constant of the low pas filter involved in the computation of the local "motion mean" for the local adaptation computation
138 * @param localAdaptIntegration_k: specifies the spatial constant of the low pas filter involved in the computation of the local "motion mean" for the local adaptation computation
139 */
140 void setCoefficientsTable(const float parasolCells_beta, const float parasolCells_tau, const float parasolCells_k, const float amacrinCellsTemporalCutFrequency, const float localAdaptIntegration_tau, const float localAdaptIntegration_k);

142 /**
143 * launch filter that runs all the IPL magno filter (model of the magnocellular channel of the Inner Plexiform Layer of the retina)
144 * @param OPL_ON: the output of the bipolar ON cells of the retina (available from the ParvoRetinaFilter class (getBipolarCellsON() function)
145 * @param OPL_OFF: the output of the bipolar OFF cells of the retina (available from the ParvoRetinaFilter class (getBipolarCellsOFF() function)
146 * @return the processed result without post-processing
147 */
148 const std::valarray<float> &runFilter(const std::valarray<float> &OPL_ON, const std::valarray<float> &OPL_OFF);

195#ifdef MAKE_PARALLEL
196 /******************************************************
197 ** IF some parallelizing thread methods are available, then, main loops are parallelized using these functors
198 ** ==> main idea parallelize main filters loops, then, only the most used methods are parallelized... TODO : increase the number of parallelized methods as necessary
199 ** ==> functors names = Parallel_$$$ where $$$= the name of the serial method that is parallelized
200 ** ==> functors constructors can differ from the parameters used with their related serial functions
201 */
202 class Parallel_amacrineCellsComputing: public cv::ParallelLoopBody
203 {
204 private:
205 const float *OPL_ON, *OPL_OFF;
206 float *previousInput_ON, *previousInput_OFF, *amacrinCellsTempOutput_ON, *amacrinCellsTempOutput_OFF;
207 float temporalCoefficient;
208 public:
209 Parallel_amacrineCellsComputing(const float *OPL_ON_PTR, const float *OPL_OFF_PTR, float *previousInput_ON_PTR, float *previousInput_OFF_PTR, float *amacrinCellsTempOutput_ON_PTR, float *amacrinCellsTempOutput_OFF_PTR, float temporalCoefficientVal)
210 :OPL_ON(OPL_ON_PTR), OPL_OFF(OPL_OFF_PTR), previousInput_ON(previousInput_ON_PTR), previousInput_OFF(previousInput_OFF_PTR), amacrinCellsTempOutput_ON(amacrinCellsTempOutput_ON_PTR), amacrinCellsTempOutput_OFF(amacrinCellsTempOutput_OFF_PTR), temporalCoefficient(temporalCoefficientVal) {}
211
212 virtual void operator()( const Range& r ) const {
213 const float *OPL_ON_PTR=OPL_ON+r.start;
214 const float *OPL_OFF_PTR=OPL_OFF+r.start;
215 float *previousInput_ON_PTR= previousInput_ON+r.start;
216 float *previousInput_OFF_PTR= previousInput_OFF+r.start;
217 float *amacrinCellsTempOutput_ON_PTR= amacrinCellsTempOutput_ON+r.start;
218 float *amacrinCellsTempOutput_OFF_PTR= amacrinCellsTempOutput_OFF+r.start;
219
220 for (int IDpixel=r.start ; IDpixel!=r.end; ++IDpixel)
221 {
222
223 /* Compute ON and OFF amacrin cells high pass temporal filter */
224 float magnoXonPixelResult = temporalCoefficient*(*amacrinCellsTempOutput_ON_PTR+ *OPL_ON_PTR-*previousInput_ON_PTR);
225 *(amacrinCellsTempOutput_ON_PTR++)=((float)(magnoXonPixelResult>0))*magnoXonPixelResult;
226
227 float magnoXoffPixelResult = temporalCoefficient*(*amacrinCellsTempOutput_OFF_PTR+ *OPL_OFF_PTR-*previousInput_OFF_PTR);
228 *(amacrinCellsTempOutput_OFF_PTR++)=((float)(magnoXoffPixelResult>0))*magnoXoffPixelResult;
229
230 /* prepare next loop */
231 *(previousInput_ON_PTR++)=*(OPL_ON_PTR++);
232 *(previousInput_OFF_PTR++)=*(OPL_OFF_PTR++);
233
234 }
235 }
236
237 };
238#endif
239 };
