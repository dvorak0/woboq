14 std::vector<std::string> Directory::GetListFiles( const std::string& path, const std::string & exten, bool addPath )
15 {
16 std::vector<std::string> list;
17 list.clear();
18 std::string path_f = path + "/" + exten;
19 #ifdef WIN32
20 #ifdef HAVE_WINRT
21 WIN32_FIND_DATAW FindFileData;
22 #else
23 WIN32_FIND_DATAA FindFileData;
24 #endif
25 HANDLE hFind;
26
27 #ifdef HAVE_WINRT
28 wchar_t wpath[MAX_PATH];
29 size_t copied = mbstowcs(wpath, path_f.c_str(), MAX_PATH);
30 CV_Assert((copied != MAX_PATH) && (copied != (size_t)-1));
31 hFind = FindFirstFileExW(wpath, FindExInfoStandard, &FindFileData, FindExSearchNameMatch, NULL, 0);
32 #else
33 hFind = FindFirstFileA((LPCSTR)path_f.c_str(), &FindFileData);
34 #endif
35 if (hFind == INVALID_HANDLE_VALUE)
36 {
37 return list;
38 }
39 else
40 {
41 do
42 {
43 if (FindFileData.dwFileAttributes == FILE_ATTRIBUTE_NORMAL ||
44 FindFileData.dwFileAttributes == FILE_ATTRIBUTE_ARCHIVE ||
45 FindFileData.dwFileAttributes == FILE_ATTRIBUTE_HIDDEN ||
46 FindFileData.dwFileAttributes == FILE_ATTRIBUTE_SYSTEM ||
47 FindFileData.dwFileAttributes == FILE_ATTRIBUTE_READONLY)
48 {
49 char* fname;
50 #ifdef HAVE_WINRT
51 char fname_tmp[MAX_PATH] = {0};
52 size_t copied = wcstombs(fname_tmp, FindFileData.cFileName, MAX_PATH);
53 CV_Assert((copied != MAX_PATH) && (copied != (size_t)-1));
54 fname = fname_tmp;
55 #else
56 fname = FindFileData.cFileName;
57 #endif
58 if (addPath)
59 list.push_back(path + "/" + std::string(fname));
60 else
61 list.push_back(std::string(fname));
62 }
63 }
64 #ifdef HAVE_WINRT
65 while(FindNextFileW(hFind, &FindFileData));
66 #else
67 while(FindNextFileA(hFind, &FindFileData));
68 #endif
69 FindClose(hFind);
70 }
71 #else
72 (void)addPath;
73 DIR *dp;
74 struct dirent *dirp;
75 if((dp = opendir(path.c_str())) == NULL)
76 {
77 return list;
78 }
79
80 #ifdef __QNX__
81 // you have to ask QNX to please include more file information
82 // and not to report duplicate names as a result of file system unions
83 if ( -1 == dircntl(dp, D_SETFLAG, D_FLAG_STAT|D_FLAG_FILTER) )
84 return list;
85 #endif
86
87 while ((dirp = readdir(dp)) != NULL)
88 {
89 #ifdef __QNX__
90 // QNX looks at the world a little differently
91 dirent_extra *extra;
92 dirent_extra_stat *extra_stat;
93 for (extra = _DEXTRA_FIRST(dirp),
94 extra_stat = reinterpret_cast<dirent_extra_stat *>(extra);
95 _DEXTRA_VALID(extra, dirp);
96 extra = _DEXTRA_NEXT(extra),
97 extra_stat = reinterpret_cast<dirent_extra_stat *>(extra))
98 if ((extra->d_type != _DTYPE_NONE) && S_ISREG(extra_stat->d_stat.st_mode))
99 #else
100 if (dirp->d_type == DT_REG)
101 #endif
102 {
103 if (exten.compare("*") == 0)
104 list.push_back(static_cast<std::string>(dirp->d_name));
105 else
106 if (std::string(dirp->d_name).find(exten) != std::string::npos)
107 list.push_back(static_cast<std::string>(dirp->d_name));
108 }
109 }
110 closedir(dp);
111 #endif
112
113 return list;
114 }

116 std::vector<std::string> Directory::GetListFolders( const std::string& path, const std::string & exten, bool addPath )
117 {
118 std::vector<std::string> list;
119 std::string path_f = path + "/" + exten;
120 list.clear();
121 #ifdef WIN32
122 #ifdef HAVE_WINRT
123 WIN32_FIND_DATAW FindFileData;
124 #else
125 WIN32_FIND_DATAA FindFileData;
126 #endif
127 HANDLE hFind;
128
129 #ifdef HAVE_WINRT
130 wchar_t wpath [MAX_PATH];
131 size_t copied = mbstowcs(wpath, path_f.c_str(), path_f.size());
132 CV_Assert((copied != MAX_PATH) && (copied != (size_t)-1));
133
134 hFind = FindFirstFileExW(wpath, FindExInfoStandard, &FindFileData, FindExSearchNameMatch, NULL, 0);
135 #else
136 hFind = FindFirstFileA((LPCSTR)path_f.c_str(), &FindFileData);
137 #endif
138 if (hFind == INVALID_HANDLE_VALUE)
139 {
140 return list;
141 }
142 else
143 {
144 do
145 {
146 #ifdef HAVE_WINRT
147 if (FindFileData.dwFileAttributes == FILE_ATTRIBUTE_DIRECTORY &&
148 wcscmp(FindFileData.cFileName, L".") != 0 &&
149 wcscmp(FindFileData.cFileName, L"..") != 0)
150 #else
151 if (FindFileData.dwFileAttributes == FILE_ATTRIBUTE_DIRECTORY &&
152 strcmp(FindFileData.cFileName, ".") != 0 &&
153 strcmp(FindFileData.cFileName, "..") != 0)
154 #endif
155 {
156 char* fname;
157 #ifdef HAVE_WINRT
158 char fname_tmp[MAX_PATH];
159 size_t copied = wcstombs(fname_tmp, FindFileData.cFileName, MAX_PATH);
160 CV_Assert((copied != MAX_PATH) && (copied != (size_t)-1));
161 fname = fname_tmp;
162 #else
163 fname = FindFileData.cFileName;
164 #endif
165
166 if (addPath)
167 list.push_back(path + "/" + std::string(fname));
168 else
169 list.push_back(std::string(fname));
170 }
171 }
172 #ifdef HAVE_WINRT
173 while(FindNextFileW(hFind, &FindFileData));
174 #else
175 while(FindNextFileA(hFind, &FindFileData));
176 #endif
177 FindClose(hFind);
178 }
179
180 #else
181 (void)addPath;
182 DIR *dp;
183 struct dirent *dirp;
184 if((dp = opendir(path_f.c_str())) == NULL)
185 {
186 return list;
187 }
188
189 #ifdef __QNX__
190 // you have to ask QNX to please include more file information
191 // and not to report duplicate names as a result of file system unions
192 if ( -1 == dircntl(dp, D_SETFLAG, D_FLAG_STAT|D_FLAG_FILTER) )
193 return list;
194 #endif
195
196 while ((dirp = readdir(dp)) != NULL)
197 {
198 #ifdef __QNX__
199 // QNX looks at the world a little differently
200 dirent_extra *extra;
201 dirent_extra_stat *extra_stat;
202 for (extra = _DEXTRA_FIRST(dirp),
203 extra_stat = reinterpret_cast<dirent_extra_stat *>(extra);
204 _DEXTRA_VALID(extra, dirp);
205 extra = _DEXTRA_NEXT(extra),
206 extra_stat = reinterpret_cast<dirent_extra_stat *>(extra))
207 if ((extra->d_type != _DTYPE_NONE) &&
208 S_ISDIR(extra_stat->d_stat.st_mode) &&
209 #else
210 if (dirp->d_type == DT_DIR &&
211 #endif
212 strcmp(dirp->d_name, ".") != 0 &&
213 strcmp(dirp->d_name, "..") != 0 )
214 {
215 if (exten.compare("*") == 0)
216 list.push_back(static_cast<std::string>(dirp->d_name));
217 else
218 if (std::string(dirp->d_name).find(exten) != std::string::npos)
219 list.push_back(static_cast<std::string>(dirp->d_name));
220 }
221 }
222 closedir(dp);
223 #endif
224
225 return list;
226 }

228 std::vector<std::string> Directory::GetListFilesR ( const std::string& path, const std::string & exten, bool addPath )
229 {
230 std::vector<std::string> list = Directory::GetListFiles(path, exten, addPath);
231
232 std::vector<std::string> dirs = Directory::GetListFolders(path, exten, addPath);
233
234 std::vector<std::string>::const_iterator it;
235 for (it = dirs.begin(); it != dirs.end(); ++it)
236 {
237 std::vector<std::string> cl = Directory::GetListFiles(*it, exten, addPath);
238 list.insert(list.end(), cl.begin(), cl.end());
239 }
240
241 return list;
242 }
