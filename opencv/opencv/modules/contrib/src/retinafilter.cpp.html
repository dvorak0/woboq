194 void RetinaFilter::_createHybridTable()
195 {
196 // create hybrid output and related coefficient table
197 _retinaParvoMagnoMappedFrame.resize(_photoreceptorsPrefilter.getNBpixels());
198
199 _retinaParvoMagnoMapCoefTable.resize(_photoreceptorsPrefilter.getNBpixels()*2);
200
201 // fill _hybridParvoMagnoCoefTable
202 int i, j, halfRows=_photoreceptorsPrefilter.getNBrows()/2, halfColumns=_photoreceptorsPrefilter.getNBcolumns()/2;
203 float *hybridParvoMagnoCoefTablePTR= &_retinaParvoMagnoMapCoefTable[0];
204 float minDistance=MIN(halfRows, halfColumns)*0.7f;
205 for (i=0;i<(int)_photoreceptorsPrefilter.getNBrows();++i)
206 {
207 for (j=0;j<(int)_photoreceptorsPrefilter.getNBcolumns();++j)
208 {
209 float distanceToCenter=sqrt(((float)(i-halfRows)*(i-halfRows)+(j-halfColumns)*(j-halfColumns)));
210 if (distanceToCenter<minDistance)
211 {
212 float a=*(hybridParvoMagnoCoefTablePTR++)=0.5f+0.5f*(float)cos(CV_PI*distanceToCenter/minDistance);
213 *(hybridParvoMagnoCoefTablePTR++)=1.f-a;
214 }else
215 {
216 *(hybridParvoMagnoCoefTablePTR++)=0.f;
217 *(hybridParvoMagnoCoefTablePTR++)=1.f;
218 }
219 }
220 }
221 }

243 bool RetinaFilter::checkInput(const std::valarray<float> &input, const bool)
244 {
245
246 BasicRetinaFilter *inputTarget=&_photoreceptorsPrefilter;
247 if (_photoreceptorsLogSampling)
248 inputTarget=_photoreceptorsLogSampling;
249
250 bool test=input.size()==inputTarget->getNBpixels() || input.size()==(inputTarget->getNBpixels()*3) ;
251 if (!test)
252 {
253 std::cerr<<"RetinaFilter::checkInput: input buffer does not match retina buffer size, conversion aborted"<<std::endl;
254 std::cout<<"RetinaFilter::checkInput: input size="<<input.size()<<" / "<<"retina size="<<inputTarget->getNBpixels()<<std::endl;
255 return false;
256 }
257
258 return true;
259 }

261 // main function that runs the filter for a given input frame
262 bool RetinaFilter::runFilter(const std::valarray<float> &imageInput, const bool useAdaptiveFiltering, const bool processRetinaParvoMagnoMapping, const bool useColorMode, const bool inputIsColorMultiplexed)
263 {
264 // preliminary check
265 bool processSuccess=true;
266 if (!checkInput(imageInput, useColorMode))
267 return false;
268
269 // run the color multiplexing if needed and compute each suub filter of the retina:
270 // -> local adaptation
271 // -> contours OPL extraction
272 // -> moving contours extraction
273
274 // stability controls value update
275 ++_ellapsedFramesSinceLastReset;
276
277 _useColorMode=useColorMode;
278
279 /* pointer to the appropriate input data after,
280 * by default, if graylevel mode, the input is processed,
281 * if color or something else must be considered, specific preprocessing are applied
282 */
283
284 const std::valarray<float> *selectedPhotoreceptorsLocalAdaptationInput= &imageInput;
285 const std::valarray<float> *selectedPhotoreceptorsColorInput=&imageInput;
286
287 //********** Following is input data specific photoreceptors processing
288 if (_photoreceptorsLogSampling)
289 {
290 _photoreceptorsLogSampling->runProjection(imageInput, useColorMode);
291 selectedPhotoreceptorsColorInput=selectedPhotoreceptorsLocalAdaptationInput=&(_photoreceptorsLogSampling->getSampledFrame());
292 }
293
294 if (useColorMode&& (!inputIsColorMultiplexed)) // not multiplexed color input case
295 {
296 _colorEngine.runColorMultiplexing(*selectedPhotoreceptorsColorInput);
297 selectedPhotoreceptorsLocalAdaptationInput=&(_colorEngine.getMultiplexedFrame());
298 }
299
300 //********** Following is generic Retina processing
301
302 // photoreceptors local adaptation
303 _photoreceptorsPrefilter.runFilter_LocalAdapdation(*selectedPhotoreceptorsLocalAdaptationInput, _ParvoRetinaFilter.getHorizontalCellsOutput());
304 // safety pixel values checks
305 //_photoreceptorsPrefilter.normalizeGrayOutput_0_maxOutputValue(_maxOutputValue);
306
307 // run parvo filter
308 _ParvoRetinaFilter.runFilter(_photoreceptorsPrefilter.getOutput(), _useParvoOutput);
309
310 if (_useParvoOutput)
311 {
312 _ParvoRetinaFilter.normalizeGrayOutputCentredSigmoide(); // models the saturation of the cells, usefull for visualisation of the ON-OFF Parvo Output, Bipolar cells outputs do not change !!!
313 _ParvoRetinaFilter.centerReductImageLuminance(); // best for further spectrum analysis
314
315 if (_normalizeParvoOutput_0_maxOutputValue)
316 _ParvoRetinaFilter.normalizeGrayOutput_0_maxOutputValue(_maxOutputValue);
317 }
318
319 if (_useParvoOutput&&_useMagnoOutput)
320 {
321 _MagnoRetinaFilter.runFilter(_ParvoRetinaFilter.getBipolarCellsON(), _ParvoRetinaFilter.getBipolarCellsOFF());
322 if (_normalizeMagnoOutput_0_maxOutputValue)
323 {
324 _MagnoRetinaFilter.normalizeGrayOutput_0_maxOutputValue(_maxOutputValue);
325 }
326 _MagnoRetinaFilter.normalizeGrayOutputNearZeroCentreredSigmoide();
327 }
328
329 if (_useParvoOutput&&_useMagnoOutput&&processRetinaParvoMagnoMapping)
330 {
331 _processRetinaParvoMagnoMapping();
332 if (_useColorMode)
333 _colorEngine.runColorDemultiplexing(_retinaParvoMagnoMappedFrame, useAdaptiveFiltering, _maxOutputValue);//_ColorEngine->getMultiplexedFrame());//_ParvoRetinaFilter->getPhotoreceptorsLPfilteringOutput());
334
335 return processSuccess;
336 }
337
338 if (_useParvoOutput&&_useColorMode)
339 {
340 _colorEngine.runColorDemultiplexing(_ParvoRetinaFilter.getOutput(), useAdaptiveFiltering, _maxOutputValue);//_ColorEngine->getMultiplexedFrame());//_ParvoRetinaFilter->getPhotoreceptorsLPfilteringOutput());
341 // compute A Cr1 Cr2 to LMS color space conversion
342 //if (true)
343 // _applyImageColorSpaceConversion(_ColorEngine->getChrominance(), lmsTempBuffer.Buffer(), _LMStoACr1Cr2);
344 }
345
346 return processSuccess;
347 }

408
409 // rescaling result between 0 and 255
410 _colorEngine.normalizeRGBOutput_0_maxOutputValue(255.0);
411
412 // return the result
413 RGBimageOutput=_colorEngine.getDemultiplexedColorFrame();
414 }
