161// Change coefficients table
162void BasicRetinaFilter::setLPfilterParameters(const float beta, const float tau, const float desired_k, const unsigned int filterIndex)
163{
164 float _beta = beta+tau;
165 float k=desired_k;
166 // check if the spatial constant is correct (avoid 0 value to avoid division by 0)
167 if (desired_k<=0)
168 {
169 k=0.001f;
170 std::cerr<<"BasicRetinaFilter::spatial constant of the low pass filter must be superior to zero !!! correcting parameter setting to 0,001"<<std::endl;
171 }
172
173 float _alpha = k*k;
174 float _mu = 0.8f;
175 unsigned int tableOffset=filterIndex*3;
176 if (k<=0)
177 {
178 std::cerr<<"BasicRetinaFilter::spatial filtering coefficient must be superior to zero, correcting value to 0.01"<<std::endl;
179 _alpha=0.0001f;
180 }
181
182 float _temp = (1.0f+_beta)/(2.0f*_mu*_alpha);
183 float a = _filteringCoeficientsTable[tableOffset] = 1.0f + _temp - (float)sqrt( (1.0f+_temp)*(1.0f+_temp) - 1.0f);
184 _filteringCoeficientsTable[1+tableOffset]=(1.0f-a)*(1.0f-a)*(1.0f-a)*(1.0f-a)/(1.0f+_beta);
185 _filteringCoeficientsTable[2+tableOffset] =tau;
186
187 //std::cout<<"BasicRetinaFilter::normal:"<<(1.0-a)*(1.0-a)*(1.0-a)*(1.0-a)/(1.0+_beta)<<" -> old:"<<(1-a)*(1-a)*(1-a)*(1-a)/(1+_beta)<<std::endl;
188
189 //std::cout<<"BasicRetinaFilter::a="<<a<<", gain="<<_filteringCoeficientsTable[1+tableOffset]<<", tau="<<tau<<std::endl;
190}

192void BasicRetinaFilter::setProgressiveFilterConstants_CentredAccuracy(const float beta, const float tau, const float alpha0, const unsigned int filterIndex)
193{
194 // check if dedicated buffers are already allocated, if not create them
195 if (_progressiveSpatialConstant.size()!=_filterOutput.size())
196 {
197 _progressiveSpatialConstant.resize(_filterOutput.size());
198 _progressiveGain.resize(_filterOutput.size());
199 }
200
201 float _beta = beta+tau;
202 float _mu=0.8f;
203 if (alpha0<=0)
204 {
205 std::cerr<<"BasicRetinaFilter::spatial filtering coefficient must be superior to zero, correcting value to 0.01"<<std::endl;
206 //alpha0=0.0001;
207 }
208
209 unsigned int tableOffset=filterIndex*3;
210
211 float _alpha=0.8f;
212 float _temp = (1.0f+_beta)/(2.0f*_mu*_alpha);
213 float a=_filteringCoeficientsTable[tableOffset] = 1.0f + _temp - (float)sqrt( (1.0f+_temp)*(1.0f+_temp) - 1.0f);
214 _filteringCoeficientsTable[tableOffset+1]=(1.0f-a)*(1.0f-a)*(1.0f-a)*(1.0f-a)/(1.0f+_beta);
215 _filteringCoeficientsTable[tableOffset+2] =tau;
216
217 float commonFactor=alpha0/(float)sqrt(_halfNBcolumns*_halfNBcolumns+_halfNBrows*_halfNBrows+1.0f);
218 //memset(_progressiveSpatialConstant, 255, _filterOutput.getNBpixels());
219 for (unsigned int idColumn=0;idColumn<_halfNBcolumns; ++idColumn)
220 for (unsigned int idRow=0;idRow<_halfNBrows; ++idRow)
221 {
222 // computing local spatial constant
223 float localSpatialConstantValue=commonFactor*sqrt((float)(idColumn*idColumn)+(float)(idRow*idRow));
224 if (localSpatialConstantValue>1.0f)
225 localSpatialConstantValue=1.0f;
226
227 _progressiveSpatialConstant[_halfNBcolumns-1+idColumn+_filterOutput.getNBcolumns()*(_halfNBrows-1+idRow)]=localSpatialConstantValue;
228 _progressiveSpatialConstant[_halfNBcolumns-1-idColumn+_filterOutput.getNBcolumns()*(_halfNBrows-1+idRow)]=localSpatialConstantValue;
229 _progressiveSpatialConstant[_halfNBcolumns-1+idColumn+_filterOutput.getNBcolumns()*(_halfNBrows-1-idRow)]=localSpatialConstantValue;
230 _progressiveSpatialConstant[_halfNBcolumns-1-idColumn+_filterOutput.getNBcolumns()*(_halfNBrows-1-idRow)]=localSpatialConstantValue;
231
232 // computing local gain
233 float localGain=(1-localSpatialConstantValue)*(1-localSpatialConstantValue)*(1-localSpatialConstantValue)*(1-localSpatialConstantValue)/(1+_beta);
234 _progressiveGain[_halfNBcolumns-1+idColumn+_filterOutput.getNBcolumns()*(_halfNBrows-1+idRow)]=localGain;
235 _progressiveGain[_halfNBcolumns-1-idColumn+_filterOutput.getNBcolumns()*(_halfNBrows-1+idRow)]=localGain;
236 _progressiveGain[_halfNBcolumns-1+idColumn+_filterOutput.getNBcolumns()*(_halfNBrows-1-idRow)]=localGain;
237 _progressiveGain[_halfNBcolumns-1-idColumn+_filterOutput.getNBcolumns()*(_halfNBrows-1-idRow)]=localGain;
238
239 //std::cout<<commonFactor<<", "<<sqrt((_halfNBcolumns-1-idColumn)+(_halfNBrows-idRow-1))<<", "<<(_halfNBcolumns-1-idColumn)<<", "<<(_halfNBrows-idRow-1)<<", "<<localSpatialConstantValue<<std::endl;
240 }
241}

243void BasicRetinaFilter::setProgressiveFilterConstants_CustomAccuracy(const float beta, const float tau, const float k, const std::valarray<float> &accuracyMap, const unsigned int filterIndex)
244{
245
246 if (accuracyMap.size()!=_filterOutput.size())
247 {
248 std::cerr<<"BasicRetinaFilter::setProgressiveFilterConstants_CustomAccuracy: error: input accuracy map does not match filter size, init skept"<<std::endl;
249 return ;
250 }
251
252 // check if dedicated buffers are already allocated, if not create them
253 if (_progressiveSpatialConstant.size()!=_filterOutput.size())
254 {
255 _progressiveSpatialConstant.resize(accuracyMap.size());
256 _progressiveGain.resize(accuracyMap.size());
257 }
258
259 float _beta = beta+tau;
260 float _alpha=k*k;
261 float _mu=0.8f;
262 if (k<=0)
263 {
264 std::cerr<<"BasicRetinaFilter::spatial filtering coefficient must be superior to zero, correcting value to 0.01"<<std::endl;
265 //alpha0=0.0001;
266 }
267 unsigned int tableOffset=filterIndex*3;
268 float _temp = (1.0f+_beta)/(2.0f*_mu*_alpha);
269 float a=_filteringCoeficientsTable[tableOffset] = 1.0f + _temp - (float)sqrt( (1.0f+_temp)*(1.0f+_temp) - 1.0f);
270 _filteringCoeficientsTable[tableOffset+1]=(1.0f-a)*(1.0f-a)*(1.0f-a)*(1.0f-a)/(1.0f+_beta);
271 _filteringCoeficientsTable[tableOffset+2] =tau;
272
273 //memset(_progressiveSpatialConstant, 255, _filterOutput.getNBpixels());
274 for (unsigned int idColumn=0;idColumn<_filterOutput.getNBcolumns(); ++idColumn)
275 for (unsigned int idRow=0;idRow<_filterOutput.getNBrows(); ++idRow)
276 {
277 // computing local spatial constant
278 unsigned int index=idColumn+idRow*_filterOutput.getNBcolumns();
279 float localSpatialConstantValue=_a*accuracyMap[index];
280 if (localSpatialConstantValue>1)
281 localSpatialConstantValue=1;
282
283 _progressiveSpatialConstant[index]=localSpatialConstantValue;
284
285 // computing local gain
286 float localGain=(1.0f-localSpatialConstantValue)*(1.0f-localSpatialConstantValue)*(1.0f-localSpatialConstantValue)*(1.0f-localSpatialConstantValue)/(1.0f+_beta);
287 _progressiveGain[index]=localGain;
288
289 //std::cout<<commonFactor<<", "<<sqrt((_halfNBcolumns-1-idColumn)+(_halfNBrows-idRow-1))<<", "<<(_halfNBcolumns-1-idColumn)<<", "<<(_halfNBrows-idRow-1)<<", "<<localSpatialConstantValue<<std::endl;
290 }

395// run LP filter on the input data and rewrite it
396void BasicRetinaFilter::runFilter_LPfilter_Autonomous(std::valarray<float> &inputOutputFrame, const unsigned int filterIndex)
397{
398 unsigned int coefTableOffset=filterIndex*3;
399
400 /**********/
401 _a=_filteringCoeficientsTable[coefTableOffset];
402 _gain=_filteringCoeficientsTable[1+coefTableOffset];
403 _tau=_filteringCoeficientsTable[2+coefTableOffset];
404
405 // launch the serie of 1D directional filters in order to compute the 2D low pass filter
406 _horizontalCausalFilter(&inputOutputFrame[0], 0, _filterOutput.getNBrows());
407 _horizontalAnticausalFilter(&inputOutputFrame[0], 0, _filterOutput.getNBrows());
408 _verticalCausalFilter(&inputOutputFrame[0], 0, _filterOutput.getNBcolumns());
409 _verticalAnticausalFilter_multGain(&inputOutputFrame[0], 0, _filterOutput.getNBcolumns());
410
411}

419 _tau=_filteringCoeficientsTable[2+coefTableOffset];

449// horizontal causal filter which adds the input inside
450void BasicRetinaFilter::_horizontalCausalFilter(float *outputFrame, unsigned int IDrowStart, unsigned int IDrowEnd)
451{
452
453
454 //#pragma omp parallel for
455 for (unsigned int IDrow=IDrowStart; IDrow<IDrowEnd; ++IDrow)
456 {
457 float* outputPTR=outputFrame+(IDrowStart+IDrow)*_filterOutput.getNBcolumns();
458 float result=0;
459 for (unsigned int index=0; index<_filterOutput.getNBcolumns(); ++index)
460 {
461 result = *(outputPTR)+ _a* result;
462 *(outputPTR++) = result;
463 }
464 }
465}
466// horizontal causal filter which adds the input inside
467void BasicRetinaFilter::_horizontalCausalFilter_addInput(const float *inputFrame, float *outputFrame, unsigned int IDrowStart, unsigned int IDrowEnd)
468{
469#ifdef MAKE_PARALLEL
470 cv::parallel_for_(cv::Range(IDrowStart,IDrowEnd), Parallel_horizontalCausalFilter_addInput(inputFrame, outputFrame, IDrowStart, _filterOutput.getNBcolumns(), _a, _tau));
471#else
472 for (unsigned int IDrow=IDrowStart; IDrow<IDrowEnd; ++IDrow)
473 {
474 float* outputPTR=outputFrame+(IDrowStart+IDrow)*_filterOutput.getNBcolumns();
475 const float* inputPTR=inputFrame+(IDrowStart+IDrow)*_filterOutput.getNBcolumns();
476 float result=0;
477 for (unsigned int index=0; index<_filterOutput.getNBcolumns(); ++index)
478 {
479 result = *(inputPTR++) + _tau**(outputPTR)+ _a* result;
480 *(outputPTR++) = result;
481 }
482 }
483#endif
484}

486// horizontal anticausal filter (basic way, no add on)
487void BasicRetinaFilter::_horizontalAnticausalFilter(float *outputFrame, unsigned int IDrowStart, unsigned int IDrowEnd)
