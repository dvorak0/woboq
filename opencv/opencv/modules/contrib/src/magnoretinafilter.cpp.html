145void MagnoRetinaFilter::setCoefficientsTable(const float parasolCells_beta, const float parasolCells_tau, const float parasolCells_k, const float amacrinCellsTemporalCutFrequency, const float localAdaptIntegration_tau, const float localAdaptIntegration_k )
146{
147 _temporalCoefficient=(float)exp(-1.0f/amacrinCellsTemporalCutFrequency);
148 // the first set of parameters is dedicated to the low pass filtering property of the ganglion cells
149 BasicRetinaFilter::setLPfilterParameters(parasolCells_beta, parasolCells_tau, parasolCells_k, 0);
150 // the second set of parameters is dedicated to the ganglion cells output intergartion for their local adaptation property
151 BasicRetinaFilter::setLPfilterParameters(0, localAdaptIntegration_tau, localAdaptIntegration_k, 1);
152}

154void MagnoRetinaFilter::_amacrineCellsComputing(const float *OPL_ON, const float *OPL_OFF)
155{
156#ifdef MAKE_PARALLEL
157 cv::parallel_for_(cv::Range(0,_filterOutput.getNBpixels()), Parallel_amacrineCellsComputing(OPL_ON, OPL_OFF, &_previousInput_ON[0], &_previousInput_OFF[0], &_amacrinCellsTempOutput_ON[0], &_amacrinCellsTempOutput_OFF[0], _temporalCoefficient));
158#else
159 const float *OPL_ON_PTR=OPL_ON;
160 const float *OPL_OFF_PTR=OPL_OFF;
161 float *previousInput_ON_PTR= &_previousInput_ON[0];
162 float *previousInput_OFF_PTR= &_previousInput_OFF[0];
163 float *amacrinCellsTempOutput_ON_PTR= &_amacrinCellsTempOutput_ON[0];
164 float *amacrinCellsTempOutput_OFF_PTR= &_amacrinCellsTempOutput_OFF[0];
165
166 for (unsigned int IDpixel=0 ; IDpixel<this->getNBpixels(); ++IDpixel)
167 {
168
169 /* Compute ON and OFF amacrin cells high pass temporal filter */
170 float magnoXonPixelResult = _temporalCoefficient*(*amacrinCellsTempOutput_ON_PTR+ *OPL_ON_PTR-*previousInput_ON_PTR);
171 *(amacrinCellsTempOutput_ON_PTR++)=((float)(magnoXonPixelResult>0))*magnoXonPixelResult;
172
173 float magnoXoffPixelResult = _temporalCoefficient*(*amacrinCellsTempOutput_OFF_PTR+ *OPL_OFF_PTR-*previousInput_OFF_PTR);
174 *(amacrinCellsTempOutput_OFF_PTR++)=((float)(magnoXoffPixelResult>0))*magnoXoffPixelResult;
175
176 /* prepare next loop */
177 *(previousInput_ON_PTR++)=*(OPL_ON_PTR++);
178 *(previousInput_OFF_PTR++)=*(OPL_OFF_PTR++);
179
180 }
181#endif
182}

184// launch filter that runs all the IPL filter
185const std::valarray<float> &MagnoRetinaFilter::runFilter(const std::valarray<float> &OPL_ON, const std::valarray<float> &OPL_OFF)
186{
187 // Compute the high pass temporal filter
188 _amacrineCellsComputing(get_data(OPL_ON), get_data(OPL_OFF));
189
190 // apply low pass filtering on ON and OFF ways after temporal high pass filtering
191 _spatiotemporalLPfilter(&_amacrinCellsTempOutput_ON[0], &_magnoXOutputON[0], 0);
192 _spatiotemporalLPfilter(&_amacrinCellsTempOutput_OFF[0], &_magnoXOutputOFF[0], 0);
193
194 // local adaptation of the ganglion cells to the local contrast of the moving contours
195 _spatiotemporalLPfilter(&_magnoXOutputON[0], &_localProcessBufferON[0], 1);
196 _localLuminanceAdaptation(&_magnoXOutputON[0], &_localProcessBufferON[0]);
197 _spatiotemporalLPfilter(&_magnoXOutputOFF[0], &_localProcessBufferOFF[0], 1);
198 _localLuminanceAdaptation(&_magnoXOutputOFF[0], &_localProcessBufferOFF[0]);
199
200 /* Compute MagnoY */
201 float *magnoYOutput= &(*_magnoYOutput)[0];
202 float *magnoXOutputON_PTR= &_magnoXOutputON[0];
203 float *magnoXOutputOFF_PTR= &_magnoXOutputOFF[0];
204 for (unsigned int IDpixel=0 ; IDpixel<_filterOutput.getNBpixels() ; ++IDpixel)
205 *(magnoYOutput++)=*(magnoXOutputON_PTR++)+*(magnoXOutputOFF_PTR++);
206
207 return (*_magnoYOutput);
208}
