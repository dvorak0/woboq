68/**
69* @class ParvoRetinaFilter
70* @brief class which describes the OPL retina model and the Inner Plexiform Layer parvocellular channel of the retina:
71* -> performs a contours extraction with powerfull local data enhancement as at the retina level
72* -> spectrum whitening occurs at the OPL (Outer Plexiform Layer) of the retina: corrects the 1/f spectrum tendancy of natural images
73* ---> enhances details with mid spatial frequencies, attenuates low spatial frequencies (luminance), attenuates high temporal frequencies and high spatial frequencies, etc.
74*
75* TYPICAL USE:
76*
77* // create object at a specified picture size
78* ParvoRetinaFilter *contoursExtractor;
79* contoursExtractor =new ParvoRetinaFilter(frameSizeRows, frameSizeColumns);
80*
81* // init gain, spatial and temporal parameters:
82* contoursExtractor->setCoefficientsTable(0, 0.7, 1, 0, 7, 1);
83*
84* // during program execution, call the filter for contours extraction for an input picture called "FrameBuffer":
85* contoursExtractor->runfilter(FrameBuffer);
86*
87* // get the output frame, check in the class description below for more outputs:
88* const float *contours=contoursExtractor->getParvoONminusOFF();
89*
90* // at the end of the program, destroy object:
91* delete contoursExtractor;
92
93* @author Alexandre BENOIT, benoit.alexandre.vision@gmail.com, LISTIC : www.listic.univ-savoie.fr, Gipsa-Lab, France: www.gipsa-lab.inpg.fr/
94* Creation date 2007
95* Based on Alexandre BENOIT thesis: "Le système visuel humain au secours de la vision par ordinateur"
96*
97*/

106//retina classes that derivate from the Basic Retrina class
107class ParvoRetinaFilter: public BasicRetinaFilter
108{
109
110public:
111 /**
112 * constructor parameters are only linked to image input size
113 * @param NBrows: number of rows of the input image
114 * @param NBcolumns: number of columns of the input image
115 */
116 ParvoRetinaFilter(const unsigned int NBrows=480, const unsigned int NBcolumns=640);
117
118 /**
119 * standard desctructor
120 */
121 virtual ~ParvoRetinaFilter();
122
123 /**
124 * resize method, keeps initial parameters, all buffers are flushed
125 * @param NBrows: number of rows of the input image
126 * @param NBcolumns: number of columns of the input image
127 */
128 void resize(const unsigned int NBrows, const unsigned int NBcolumns);
129
130 /**
131 * function that clears all buffers of the object
132 */
133 void clearAllBuffers();
134
135 /**
136 * setup the OPL and IPL parvo channels
137 * @param beta1: gain of the horizontal cells network, if 0, then the mean value of the output is zero, if the parameter is near 1, the amplitude is boosted but it should only be used for values rescaling... if needed
138 * @param tau1: the time constant of the first order low pass filter of the photoreceptors, use it to cut high temporal frequencies (noise or fast motion), unit is frames, typical value is 1 frame
139 * @param k1: the spatial constant of the first order low pass filter of the photoreceptors, use it to cut high spatial frequencies (noise or thick contours), unit is pixels, typical value is 1 pixel
140 * @param beta2: gain of the horizontal cells network, if 0, then the mean value of the output is zero, if the parameter is near 1, then, the luminance is not filtered and is still reachable at the output, typicall value is 0
141 * @param tau2: the time constant of the first order low pass filter of the horizontal cells, use it to cut low temporal frequencies (local luminance variations), unit is frames, typical value is 1 frame, as the photoreceptors
142 * @param k2: the spatial constant of the first order low pass filter of the horizontal cells, use it to cut low spatial frequencies (local luminance), unit is pixels, typical value is 5 pixel, this value is also used for local contrast computing when computing the local contrast adaptation at the ganglion cells level (Inner Plexiform Layer parvocellular channel model)
143 */
144 void setOPLandParvoFiltersParameters(const float beta1, const float tau1, const float k1, const float beta2, const float tau2, const float k2);
145
146 /**
147 * setup more precisely the low pass filter used for the ganglion cells low pass filtering (used for local luminance adaptation)
148 * @param tau: time constant of the filter (unit is frame for video processing)
149 * @param k: spatial constant of the filter (unit is pixels)
150 */
151 void setGanglionCellsLocalAdaptationLPfilterParameters(const float tau, const float k){BasicRetinaFilter::setLPfilterParameters(0, tau, k, 2);}; // change the parameters of the filter
152
153
154 /**
155 * launch filter that runs the OPL spatiotemporal filtering and optionally finalizes IPL Pagno filter (model of the Parvocellular channel of the Inner Plexiform Layer of the retina)
156 * @param inputFrame: the input image to be processed, this can be the direct gray level input frame, but a better efficacy is expected if the input is preliminary processed by the photoreceptors local adaptation possible to acheive with the help of a BasicRetinaFilter object
157 * @param useParvoOutput: set true if the final IPL filtering step has to be computed (local contrast enhancement)
158 * @return the processed Parvocellular channel output (updated only if useParvoOutput is true)
159 * @details: in any case, after this function call, photoreceptors and horizontal cells output are updated, use getPhotoreceptorsLPfilteringOutput() and getHorizontalCellsOutput() to get them
160 * also, bipolar cells output are accessible (difference between photoreceptors and horizontal cells, ON output has positive values, OFF ouput has negative values), use the following access methods: getBipolarCellsON() and getBipolarCellsOFF()if useParvoOutput is true,
161 * if useParvoOutput is true, the complete Parvocellular channel is computed, more outputs are updated and can be accessed threw: getParvoON(), getParvoOFF() and their difference with getOutput()
162 */
163 const std::valarray<float> &runFilter(const std::valarray<float> &inputFrame, const bool useParvoOutput=true); // output return is _parvocellularOutputONminusOFF
164
165 /**
166 * @return the output of the photoreceptors filtering step (high cut frequency spatio-temporal low pass filter)
167 */
168 inline const std::valarray<float> &getPhotoreceptorsLPfilteringOutput() const {return _photoreceptorsOutput;};
169
170 /**
171 * @return the output of the photoreceptors filtering step (low cut frequency spatio-temporal low pass filter)
172 */
173 inline const std::valarray<float> &getHorizontalCellsOutput() const { return _horizontalCellsOutput;};
174
175 /**
176 * @return the output Parvocellular ON channel of the retina model
177 */
178 inline const std::valarray<float> &getParvoON() const {return _parvocellularOutputON;};
179
180 /**
181 * @return the output Parvocellular OFF channel of the retina model
182 */
183 inline const std::valarray<float> &getParvoOFF() const {return _parvocellularOutputOFF;};
184
185 /**
186 * @return the output of the Bipolar cells of the ON channel of the retina model same as function getParvoON() but without luminance local adaptation
187 */
188 inline const std::valarray<float> &getBipolarCellsON() const {return _bipolarCellsOutputON;};
189
190 /**
191 * @return the output of the Bipolar cells of the OFF channel of the retina model same as function getParvoON() but without luminance local adaptation
192 */
193 inline const std::valarray<float> &getBipolarCellsOFF() const {return _bipolarCellsOutputOFF;};
194
195 /**
196 * @return the photoreceptors's temporal constant
197 */
198 inline float getPhotoreceptorsTemporalConstant(){return this->_filteringCoeficientsTable[2];};
199
200 /**
201 * @return the horizontal cells' temporal constant
202 */
203 inline float getHcellsTemporalConstant(){return this->_filteringCoeficientsTable[5];};

205private:
206 // template buffers
207 std::valarray <float>_photoreceptorsOutput;
208 std::valarray <float>_horizontalCellsOutput;
209 std::valarray <float>_parvocellularOutputON;
210 std::valarray <float>_parvocellularOutputOFF;
211 std::valarray <float>_bipolarCellsOutputON;
212 std::valarray <float>_bipolarCellsOutputOFF;
213 std::valarray <float>_localAdaptationOFF;
214 std::valarray <float> *_localAdaptationON;
215 TemplateBuffer<float> *_parvocellularOutputONminusOFF;
216 // private functions
217 void _OPL_OnOffWaysComputing();

219#ifdef MAKE_PARALLEL
220/******************************************************
221** IF some parallelizing thread methods are available, then, main loops are parallelized using these functors
222** ==> main idea parallelize main filters loops, then, only the most used methods are parallelized... TODO : increase the number of parallelized methods as necessary
223** ==> functors names = Parallel_$$$ where $$$= the name of the serial method that is parallelized
224** ==> functors constructors can differ from the parameters used with their related serial functions
225*/
226 class Parallel_OPL_OnOffWaysComputing: public cv::ParallelLoopBody
227 {
228 private:
229 float *photoreceptorsOutput, *horizontalCellsOutput, *bipolarCellsON, *bipolarCellsOFF, *parvocellularOutputON, *parvocellularOutputOFF;
230 public:
231 Parallel_OPL_OnOffWaysComputing(float *photoreceptorsOutput_PTR, float *horizontalCellsOutput_PTR, float *bipolarCellsON_PTR, float *bipolarCellsOFF_PTR, float *parvocellularOutputON_PTR, float *parvocellularOutputOFF_PTR)
232 :photoreceptorsOutput(photoreceptorsOutput_PTR), horizontalCellsOutput(horizontalCellsOutput_PTR), bipolarCellsON(bipolarCellsON_PTR), bipolarCellsOFF(bipolarCellsOFF_PTR), parvocellularOutputON(parvocellularOutputON_PTR), parvocellularOutputOFF(parvocellularOutputOFF_PTR) {}
233
234 virtual void operator()( const Range& r ) const {
235 // compute bipolar cells response equal to photoreceptors minus horizontal cells response
236 // and copy the result on parvo cellular outputs... keeping time before their local contrast adaptation for final result
237 float *photoreceptorsOutput_PTR= photoreceptorsOutput+r.start;
238 float *horizontalCellsOutput_PTR= horizontalCellsOutput+r.start;
239 float *bipolarCellsON_PTR = bipolarCellsON+r.start;
240 float *bipolarCellsOFF_PTR = bipolarCellsOFF+r.start;
241 float *parvocellularOutputON_PTR= parvocellularOutputON+r.start;
242 float *parvocellularOutputOFF_PTR= parvocellularOutputOFF+r.start;
243
244 for (int IDpixel=r.start ; IDpixel!=r.end ; ++IDpixel)
245 {
246 float pixelDifference = *(photoreceptorsOutput_PTR++) -*(horizontalCellsOutput_PTR++);
247 // test condition to allow write pixelDifference in ON or OFF buffer and 0 in the over
248 float isPositive=(float) (pixelDifference>0.0f);
249
250 // ON and OFF channels writing step
251 *(parvocellularOutputON_PTR++)=*(bipolarCellsON_PTR++) = isPositive*pixelDifference;
252 *(parvocellularOutputOFF_PTR++)=*(bipolarCellsOFF_PTR++)= (isPositive-1.0f)*pixelDifference;
253 }
254 }
255 };
256#endif
257
258};
259}
