93void computeNormals( const Octree& Octree, const vector<Point3f>& centers, vector<Point3f>& normals,
94 vector<uchar>& mask, float normalRadius, int minNeighbors = 20)
95{
96 size_t normals_size = centers.size();
97 normals.resize(normals_size);
98
99 if (mask.size() != normals_size)
100 {
101 size_t m = mask.size();
102 mask.resize(normals_size);
103 if (normals_size > m)
104 for(; m < normals_size; ++m)
105 mask[m] = 1;
106 }
107
108 vector<Point3f> buffer;
109 buffer.reserve(128);
110 SVD svd;
111
112 const static Point3f zero(0.f, 0.f, 0.f);
113
114 for(size_t n = 0; n < normals_size; ++n)
115 {
116 if (mask[n] == 0)
117 continue;
118
119 const Point3f& center = centers[n];
120 Octree.getPointsWithinSphere(center, normalRadius, buffer);
121
122 int buf_size = (int)buffer.size();
123 if (buf_size < minNeighbors)
124 {
125 normals[n] = Mesh3D::allzero;
126 mask[n] = 0;
127 continue;
128 }
129
130 //find the mean point for normalization
131 Point3f mean(Mesh3D::allzero);
132 for(int i = 0; i < buf_size; ++i)
133 mean += buffer[i];
134
135 mean.x /= buf_size;
136 mean.y /= buf_size;
137 mean.z /= buf_size;
138
139 double pxpx = 0;
140 double pypy = 0;
141 double pzpz = 0;
142
143 double pxpy = 0;
144 double pxpz = 0;
145 double pypz = 0;
146
147 for(int i = 0; i < buf_size; ++i)
148 {
149 const Point3f& p = buffer[i];
150
151 pxpx += (p.x - mean.x) * (p.x - mean.x);
152 pypy += (p.y - mean.y) * (p.y - mean.y);
153 pzpz += (p.z - mean.z) * (p.z - mean.z);
154
155 pxpy += (p.x - mean.x) * (p.y - mean.y);
156 pxpz += (p.x - mean.x) * (p.z - mean.z);
157 pypz += (p.y - mean.y) * (p.z - mean.z);
158 }
159
160 //create and populate matrix with normalized nbrs
161 double M_data[] = { pxpx, pxpy, pxpz, /**/ pxpy, pypy, pypz, /**/ pxpz, pypz, pzpz };
162 Mat M(3, 3, CV_64F, M_data);
163
164 svd(M, SVD::MODIFY_A);
165
166 /*normals[n] = Point3f( (float)((double*)svd.vt.data)[6],
167 (float)((double*)svd.vt.data)[7],
168 (float)((double*)svd.vt.data)[8] );*/
169 normals[n] = reinterpret_cast<Point3d*>(svd.vt.data)[2];
170 mask[n] = 1;
171 }
172}

174void initRotationMat(const Point3f& n, float out[9])
175{
176 double pitch = atan2(n.x, n.z);
177 double pmat[] = { cos(pitch), 0, -sin(pitch) ,
178 0 , 1, 0 ,
179 sin(pitch), 0, cos(pitch) };
180
181 double roll = atan2((double)n.y, n.x * pmat[3*2+0] + n.z * pmat[3*2+2]);
182
183 double rmat[] = { 1, 0, 0,
184 0, cos(roll), -sin(roll) ,
185 0, sin(roll), cos(roll) };
186
187 for(int i = 0; i < 3; ++i)
188 for(int j = 0; j < 3; ++j)
189 out[3*i+j] = (float)(rmat[3*i+0]*pmat[3*0+j] +
190 rmat[3*i+1]*pmat[3*1+j] + rmat[3*i+2]*pmat[3*2+j]);
191}

193void transform(const Point3f& in, float matrix[9], Point3f& out)
194{
195 out.x = in.x * matrix[3*0+0] + in.y * matrix[3*0+1] + in.z * matrix[3*0+2];
196 out.y = in.x * matrix[3*1+0] + in.y * matrix[3*1+1] + in.z * matrix[3*1+2];
197 out.z = in.x * matrix[3*2+0] + in.y * matrix[3*2+1] + in.z * matrix[3*2+2];
198}

201void convertTransformMatrix(const float* matrix, float* sseMatrix)
202{
203 sseMatrix[0] = matrix[0]; sseMatrix[1] = matrix[3]; sseMatrix[2] = matrix[6]; sseMatrix[3] = 0;
204 sseMatrix[4] = matrix[1]; sseMatrix[5] = matrix[4]; sseMatrix[6] = matrix[7]; sseMatrix[7] = 0;
205 sseMatrix[8] = matrix[2]; sseMatrix[9] = matrix[5]; sseMatrix[10] = matrix[8]; sseMatrix[11] = 0;
206}

208inline __m128 transformSSE(const __m128* matrix, const __m128& in)
209{
210 assert(((size_t)matrix & 15) == 0);
211 __m128 a0 = _mm_mul_ps(_mm_load_ps((float*)(matrix+0)), _mm_shuffle_ps(in,in,_MM_SHUFFLE(0,0,0,0)));
212 __m128 a1 = _mm_mul_ps(_mm_load_ps((float*)(matrix+1)), _mm_shuffle_ps(in,in,_MM_SHUFFLE(1,1,1,1)));
213 __m128 a2 = _mm_mul_ps(_mm_load_ps((float*)(matrix+2)), _mm_shuffle_ps(in,in,_MM_SHUFFLE(2,2,2,2)));
214
215 return _mm_add_ps(_mm_add_ps(a0,a1),a2);
216}

218inline __m128i _mm_mullo_epi32_emul(const __m128i& a, __m128i& b)
219{
220 __m128i pack = _mm_packs_epi32(a, a);
221 return _mm_unpacklo_epi16(_mm_mullo_epi16(pack, b), _mm_mulhi_epi16(pack, b));
222}

226void computeSpinImages( const Octree& Octree, const vector<Point3f>& points, const vector<Point3f>& normals,
227 vector<uchar>& mask, Mat& spinImages, int imageWidth, float binSize)
228{
229 float pixelsPerMeter = 1.f / binSize;
230 float support = imageWidth * binSize;
231
232 assert(normals.size() == points.size());
233 assert(mask.size() == points.size());
234
235 size_t points_size = points.size();
236 mask.resize(points_size);
237
238 int height = imageWidth;
239 int width = imageWidth;
240
241 spinImages.create( (int)points_size, width*height, CV_32F );
242
243 int nthreads = getNumThreads();
244 int i;
245
246 vector< vector<Point3f> > pointsInSpherePool(nthreads);
247 for(i = 0; i < nthreads; i++)
248 pointsInSpherePool[i].reserve(2048);
249
250 float halfSuppport = support / 2;
251 float searchRad = support * sqrt(5.f) / 2; // sqrt(sup*sup + (sup/2) * (sup/2) )
252
253#ifdef _OPENMP
254 #pragma omp parallel for num_threads(nthreads)
255#endif
256 for(i = 0; i < (int)points_size; ++i)
257 {
258 if (mask[i] == 0)
259 continue;
260
261 int t = cvGetThreadNum();
262 vector<Point3f>& pointsInSphere = pointsInSpherePool[t];
263
264 const Point3f& center = points[i];
265 Octree.getPointsWithinSphere(center, searchRad, pointsInSphere);
266
267 size_t inSphere_size = pointsInSphere.size();
268 if (inSphere_size == 0)
269 {
270 mask[i] = 0;
271 continue;
272 }
273
274 const Point3f& normal = normals[i];
275
276 float rotmat[9];
277 initRotationMat(normal, rotmat);
278 Point3f new_center;
279 transform(center, rotmat, new_center);
280
281 Mat spinImage = spinImages.row(i).reshape(1, height);
282 float* spinImageData = (float*)spinImage.data;
283 int step = width;
284 spinImage = Scalar(0.);
285
286 float alpha, beta;
287 size_t j = 0;
288#if CV_SSE2
289 if (inSphere_size > 4 && checkHardwareSupport(CV_CPU_SSE2))
290 {
291 __m128 rotmatSSE[3];
292 convertTransformMatrix(rotmat, (float*)rotmatSSE);
293
294 __m128 center_x4 = _mm_set1_ps(new_center.x);
295 __m128 center_y4 = _mm_set1_ps(new_center.y);
296 __m128 center_z4 = _mm_set1_ps(new_center.z + halfSuppport);
297 __m128 ppm4 = _mm_set1_ps(pixelsPerMeter);
298 __m128i height4m1 = _mm_set1_epi32(spinImage.rows-1);
299 __m128i width4m1 = _mm_set1_epi32(spinImage.cols-1);
300 assert( spinImage.step <= 0xffff );
301 __m128i step4 = _mm_set1_epi16((short)step);
302 __m128i zero4 = _mm_setzero_si128();
303 __m128i one4i = _mm_set1_epi32(1);
304 __m128 zero4f = _mm_setzero_ps();
305 __m128 one4f = _mm_set1_ps(1.f);
306 //__m128 two4f = _mm_set1_ps(2.f);
307 int CV_DECL_ALIGNED(16) o[4];
308
309 for (; j <= inSphere_size - 5; j += 4)
310 {
311 __m128 pt0 = transformSSE(rotmatSSE, _mm_loadu_ps((float*)&pointsInSphere[j+0])); // x0 y0 z0 .
312 __m128 pt1 = transformSSE(rotmatSSE, _mm_loadu_ps((float*)&pointsInSphere[j+1])); // x1 y1 z1 .
313 __m128 pt2 = transformSSE(rotmatSSE, _mm_loadu_ps((float*)&pointsInSphere[j+2])); // x2 y2 z2 .
314 __m128 pt3 = transformSSE(rotmatSSE, _mm_loadu_ps((float*)&pointsInSphere[j+3])); // x3 y3 z3 .
315
316 __m128 z0 = _mm_unpackhi_ps(pt0, pt1); // z0 z1 . .
317 __m128 z1 = _mm_unpackhi_ps(pt2, pt3); // z2 z3 . .
318 __m128 beta4 = _mm_sub_ps(center_z4, _mm_movelh_ps(z0, z1)); // b0 b1 b2 b3
319
320 __m128 xy0 = _mm_unpacklo_ps(pt0, pt1); // x0 x1 y0 y1
321 __m128 xy1 = _mm_unpacklo_ps(pt2, pt3); // x2 x3 y2 y3
322 __m128 x4 = _mm_movelh_ps(xy0, xy1); // x0 x1 x2 x3
323 __m128 y4 = _mm_movehl_ps(xy1, xy0); // y0 y1 y2 y3
324
325 x4 = _mm_sub_ps(x4, center_x4);
326 y4 = _mm_sub_ps(y4, center_y4);
327 __m128 alpha4 = _mm_sqrt_ps(_mm_add_ps(_mm_mul_ps(x4,x4),_mm_mul_ps(y4,y4)));
328
329 __m128 n1f4 = _mm_mul_ps( beta4, ppm4); /* beta4 float */
330 __m128 n2f4 = _mm_mul_ps(alpha4, ppm4); /* alpha4 float */
331
332 /* floor */
333 __m128i n1 = _mm_sub_epi32(_mm_cvttps_epi32( _mm_add_ps( n1f4, one4f ) ), one4i);
334 __m128i n2 = _mm_sub_epi32(_mm_cvttps_epi32( _mm_add_ps( n2f4, one4f ) ), one4i);
335
336 __m128 f1 = _mm_sub_ps( n1f4, _mm_cvtepi32_ps(n1) ); /* { beta4 } */
337 __m128 f2 = _mm_sub_ps( n2f4, _mm_cvtepi32_ps(n2) ); /* { alpha4 } */
338
339 __m128 f1f2 = _mm_mul_ps(f1, f2); // f1 * f2
340 __m128 omf1omf2 = _mm_add_ps(_mm_sub_ps(_mm_sub_ps(one4f, f2), f1), f1f2); // (1-f1) * (1-f2)
341
342 __m128i _mask = _mm_and_si128(
343 _mm_andnot_si128(_mm_cmpgt_epi32(zero4, n1), _mm_cmpgt_epi32(height4m1, n1)),
344 _mm_andnot_si128(_mm_cmpgt_epi32(zero4, n2), _mm_cmpgt_epi32(width4m1, n2)));
345
346 __m128 maskf = _mm_cmpneq_ps(_mm_cvtepi32_ps(_mask), zero4f);
347
348 __m128 v00 = _mm_and_ps( omf1omf2 , maskf); // a00 b00 c00 d00
349 __m128 v01 = _mm_and_ps( _mm_sub_ps( f2, f1f2 ), maskf); // a01 b01 c01 d01
350 __m128 v10 = _mm_and_ps( _mm_sub_ps( f1, f1f2 ), maskf); // a10 b10 c10 d10
351 __m128 v11 = _mm_and_ps( f1f2 , maskf); // a11 b11 c11 d11
352
353 __m128i ofs4 = _mm_and_si128(_mm_add_epi32(_mm_mullo_epi32_emul(n1, step4), n2), _mask);
354 _mm_store_si128((__m128i*)o, ofs4);
355
356 __m128 t0 = _mm_unpacklo_ps(v00, v01); // a00 a01 b00 b01
357 __m128 t1 = _mm_unpacklo_ps(v10, v11); // a10 a11 b10 b11
358 __m128 u0 = _mm_movelh_ps(t0, t1); // a00 a01 a10 a11
359 __m128 u1 = _mm_movehl_ps(t1, t0); // b00 b01 b10 b11
360
361 __m128 x0 = _mm_loadl_pi(u0, (__m64*)(spinImageData+o[0])); // x00 x01
362 x0 = _mm_loadh_pi(x0, (__m64*)(spinImageData+o[0]+step)); // x00 x01 x10 x11
363 x0 = _mm_add_ps(x0, u0);
364 _mm_storel_pi((__m64*)(spinImageData+o[0]), x0);
365 _mm_storeh_pi((__m64*)(spinImageData+o[0]+step), x0);
366
367 x0 = _mm_loadl_pi(x0, (__m64*)(spinImageData+o[1])); // y00 y01
368 x0 = _mm_loadh_pi(x0, (__m64*)(spinImageData+o[1]+step)); // y00 y01 y10 y11
369 x0 = _mm_add_ps(x0, u1);
370 _mm_storel_pi((__m64*)(spinImageData+o[1]), x0);
371 _mm_storeh_pi((__m64*)(spinImageData+o[1]+step), x0);
372
373 t0 = _mm_unpackhi_ps(v00, v01); // c00 c01 d00 d01
374 t1 = _mm_unpackhi_ps(v10, v11); // c10 c11 d10 d11
375 u0 = _mm_movelh_ps(t0, t1); // c00 c01 c10 c11
376 u1 = _mm_movehl_ps(t1, t0); // d00 d01 d10 d11
377
378 x0 = _mm_loadl_pi(x0, (__m64*)(spinImageData+o[2])); // z00 z01
379 x0 = _mm_loadh_pi(x0, (__m64*)(spinImageData+o[2]+step)); // z00 z01 z10 z11
380 x0 = _mm_add_ps(x0, u0);
381 _mm_storel_pi((__m64*)(spinImageData+o[2]), x0);
382 _mm_storeh_pi((__m64*)(spinImageData+o[2]+step), x0);
383
384 x0 = _mm_loadl_pi(x0, (__m64*)(spinImageData+o[3])); // w00 w01
385 x0 = _mm_loadh_pi(x0, (__m64*)(spinImageData+o[3]+step)); // w00 w01 w10 w11
386 x0 = _mm_add_ps(x0, u1);
387 _mm_storel_pi((__m64*)(spinImageData+o[3]), x0);
388 _mm_storeh_pi((__m64*)(spinImageData+o[3]+step), x0);
389 }
390 }
391#endif
392 for (; j < inSphere_size; ++j)
393 {
394 Point3f pt;
395 transform(pointsInSphere[j], rotmat, pt);
396
397 beta = halfSuppport - (pt.z - new_center.z);
398 if (beta >= support || beta < 0)
399 continue;
400
401 alpha = sqrt( (new_center.x - pt.x) * (new_center.x - pt.x) +
402 (new_center.y - pt.y) * (new_center.y - pt.y) );
403
404 float n1f = beta * pixelsPerMeter;
405 float n2f = alpha * pixelsPerMeter;
406
407 int n1 = cvFloor(n1f);
408 int n2 = cvFloor(n2f);
409
410 float f1 = n1f - n1;
411 float f2 = n2f - n2;
412
413 if ((unsigned)n1 >= (unsigned)(spinImage.rows-1) ||
414 (unsigned)n2 >= (unsigned)(spinImage.cols-1))
415 continue;
416
417 float *cellptr = spinImageData + step * n1 + n2;
418 float f1f2 = f1*f2;
419 cellptr[0] += 1 - f1 - f2 + f1f2;
420 cellptr[1] += f2 - f1f2;
421 cellptr[step] += f1 - f1f2;
422 cellptr[step+1] += f1f2;
423 }
424 mask[i] = 1;
425 }
426}

541bool cv::SpinImageModel::spinCorrelation(const Mat& spin1, const Mat& spin2, float lambda, float& result)
542{
543 struct Math { static double atanh(double x) { return 0.5 * std::log( (1 + x) / (1 - x) ); } };
544
545 const float* s1 = spin1.ptr<float>();
546 const float* s2 = spin2.ptr<float>();
547
548 int spin_sz = spin1.cols * spin1.rows;
549 double sum1 = 0.0, sum2 = 0.0, sum12 = 0.0, sum11 = 0.0, sum22 = 0.0;
550
551 int N = 0;
552 int i = 0;
553#if CV_SSE2//____________TEMPORARY_DISABLED_____________
554 float CV_DECL_ALIGNED(16) su1[4], su2[4], su11[4], su22[4], su12[4], n[4];
555
556 __m128 zerof4 = _mm_setzero_ps();
557 __m128 onef4 = _mm_set1_ps(1.f);
558 __m128 Nf4 = zerof4;
559 __m128 sum1f4 = zerof4;
560 __m128 sum2f4 = zerof4;
561 __m128 sum11f4 = zerof4;
562 __m128 sum22f4 = zerof4;
563 __m128 sum12f4 = zerof4;
564 for(; i < spin_sz - 5; i += 4)
565 {
566 __m128 v1f4 = _mm_loadu_ps(s1 + i);
567 __m128 v2f4 = _mm_loadu_ps(s2 + i);
568
569 __m128 mskf4 = _mm_and_ps(_mm_cmpneq_ps(v1f4, zerof4), _mm_cmpneq_ps(v2f4, zerof4));
570 if( !_mm_movemask_ps(mskf4) )
571 continue;
572
573 Nf4 = _mm_add_ps(Nf4, _mm_and_ps(onef4, mskf4));
574
575 v1f4 = _mm_and_ps(v1f4, mskf4);
576 v2f4 = _mm_and_ps(v2f4, mskf4);
577
578 sum1f4 = _mm_add_ps(sum1f4, v1f4);
579 sum2f4 = _mm_add_ps(sum2f4, v2f4);
580 sum11f4 = _mm_add_ps(sum11f4, _mm_mul_ps(v1f4, v1f4));
581 sum22f4 = _mm_add_ps(sum22f4, _mm_mul_ps(v2f4, v2f4));
582 sum12f4 = _mm_add_ps(sum12f4, _mm_mul_ps(v1f4, v2f4));
583 }
584 _mm_store_ps( su1, sum1f4 );
585 _mm_store_ps( su2, sum2f4 );
586 _mm_store_ps(su11, sum11f4 );
587 _mm_store_ps(su22, sum22f4 );
588 _mm_store_ps(su12, sum12f4 );
589 _mm_store_ps(n, Nf4 );
590
591 N = static_cast<int>(n[0] + n[1] + n[2] + n[3]);
592 sum1 = su1[0] + su1[1] + su1[2] + su1[3];
593 sum2 = su2[0] + su2[1] + su2[2] + su2[3];
594 sum11 = su11[0] + su11[1] + su11[2] + su11[3];
595 sum22 = su22[0] + su22[1] + su22[2] + su22[3];
596 sum12 = su12[0] + su12[1] + su12[2] + su12[3];
597#endif
598
599 for(; i < spin_sz; ++i)
600 {
601 float v1 = s1[i];
602 float v2 = s2[i];
603
604 if( !v1 || !v2 )
605 continue;
606 N++;
607
608 sum1 += v1;
609 sum2 += v2;
610 sum11 += v1 * v1;
611 sum22 += v2 * v2;
612 sum12 += v1 * v2;
613 }
614 if( N < 4 )
615 return false;
616
617 double sum1sum1 = sum1 * sum1;
618 double sum2sum2 = sum2 * sum2;
619
620 double Nsum12 = N * sum12;
621 double Nsum11 = N * sum11;
622 double Nsum22 = N * sum22;
623
624 if (Nsum11 == sum1sum1 || Nsum22 == sum2sum2)
625 return false;
626
627 double corr = (Nsum12 - sum1 * sum2) / sqrt( (Nsum11 - sum1sum1) * (Nsum22 - sum2sum2) );
628 double atanh = Math::atanh(corr);
629 result = (float)( atanh * atanh - lambda * ( 1.0 / (N - 3) ) );
630 return true;
631}

633inline Point2f cv::SpinImageModel::calcSpinMapCoo(const Point3f& p, const Point3f& v, const Point3f& n)
634{
635 /*Point3f PmV(p.x - v.x, p.y - v.y, p.z - v.z);
636 float normalNorm = (float)norm(n);
637 float beta = PmV.dot(n) / normalNorm;
638 float pmcNorm = (float)norm(PmV);
639 float alpha = sqrt( pmcNorm * pmcNorm - beta * beta);
640 return Point2f(alpha, beta);*/
641
642 float pmv_x = p.x - v.x, pmv_y = p.y - v.y, pmv_z = p.z - v.z;
643
644 float beta = (pmv_x * n.x + pmv_y + n.y + pmv_z * n.z) / sqrt(n.x * n.x + n.y * n.y + n.z * n.z);
645 float alpha = sqrt( pmv_x * pmv_x + pmv_y * pmv_y + pmv_z * pmv_z - beta * beta);
646 return Point2f(alpha, beta);
647}

649inline float cv::SpinImageModel::geometricConsistency(const Point3f& pointScene1, const Point3f& normalScene1,
650 const Point3f& pointModel1, const Point3f& normalModel1,
651 const Point3f& pointScene2, const Point3f& normalScene2,
652 const Point3f& pointModel2, const Point3f& normalModel2)
653{
654 Point2f Sm2_to_m1, Ss2_to_s1;
655 Point2f Sm1_to_m2, Ss1_to_s2;
656
657 double n_Sm2_to_m1 = norm(Sm2_to_m1 = calcSpinMapCoo(pointModel2, pointModel1, normalModel1));
658 double n_Ss2_to_s1 = norm(Ss2_to_s1 = calcSpinMapCoo(pointScene2, pointScene1, normalScene1));
659
660 double gc21 = 2 * norm(Sm2_to_m1 - Ss2_to_s1) / (n_Sm2_to_m1 + n_Ss2_to_s1 ) ;
661
662 double n_Sm1_to_m2 = norm(Sm1_to_m2 = calcSpinMapCoo(pointModel1, pointModel2, normalModel2));
663 double n_Ss1_to_s2 = norm(Ss1_to_s2 = calcSpinMapCoo(pointScene1, pointScene2, normalScene2));
664
665 double gc12 = 2 * norm(Sm1_to_m2 - Ss1_to_s2) / (n_Sm1_to_m2 + n_Ss1_to_s2 ) ;
666
667 return (float)max(gc12, gc21);
668}

670inline float cv::SpinImageModel::groupingCreteria(const Point3f& pointScene1, const Point3f& normalScene1,
671 const Point3f& pointModel1, const Point3f& normalModel1,
672 const Point3f& pointScene2, const Point3f& normalScene2,
673 const Point3f& pointModel2, const Point3f& normalModel2,
674 float gamma)
675{
676 Point2f Sm2_to_m1, Ss2_to_s1;
677 Point2f Sm1_to_m2, Ss1_to_s2;
678
679 float gamma05_inv = 0.5f/gamma;
680
681 double n_Sm2_to_m1 = norm(Sm2_to_m1 = calcSpinMapCoo(pointModel2, pointModel1, normalModel1));
682 double n_Ss2_to_s1 = norm(Ss2_to_s1 = calcSpinMapCoo(pointScene2, pointScene1, normalScene1));
683
684 double gc21 = 2 * norm(Sm2_to_m1 - Ss2_to_s1) / (n_Sm2_to_m1 + n_Ss2_to_s1 );
685 double wgc21 = gc21 / (1 - exp( -(n_Sm2_to_m1 + n_Ss2_to_s1) * gamma05_inv ) );
686
687 double n_Sm1_to_m2 = norm(Sm1_to_m2 = calcSpinMapCoo(pointModel1, pointModel2, normalModel2));
688 double n_Ss1_to_s2 = norm(Ss1_to_s2 = calcSpinMapCoo(pointScene1, pointScene2, normalScene2));
689
690 double gc12 = 2 * norm(Sm1_to_m2 - Ss1_to_s2) / (n_Sm1_to_m2 + n_Ss1_to_s2 );
691 double wgc12 = gc12 / (1 - exp( -(n_Sm1_to_m2 + n_Ss1_to_s2) * gamma05_inv ) );
692
693 return (float)max(wgc12, wgc21);
694}

723Mat cv::SpinImageModel::packRandomScaledSpins(bool separateScale, size_t xCount, size_t yCount) const
724{
725 int spinNum = (int)getSpinCount();
726 int num = min(spinNum, (int)(xCount * yCount));
727
728 if (num == 0)
729 return Mat();
730
731 RNG& rng = theRNG();
732
733 vector<Mat> spins;
734 for(int i = 0; i < num; ++i)
735 spins.push_back(getSpinImage( rng.next() % spinNum ).reshape(1, imageWidth));
736
737 if (separateScale)
738 for(int i = 0; i < num; ++i)
739 {
740 double max;
741 Mat spin8u;
742 minMaxLoc(spins[i], 0, &max);
743 spins[i].convertTo(spin8u, CV_8U, -255.0/max, 255.0);
744 spins[i] = spin8u;
745 }
746 else
747 {
748 double totalMax = 0;
749 for(int i = 0; i < num; ++i)
750 {
751 double m;
752 minMaxLoc(spins[i], 0, &m);
753 totalMax = max(m, totalMax);
754 }
755
756 for(int i = 0; i < num; ++i)
757 {
758 Mat spin8u;
759 spins[i].convertTo(spin8u, CV_8U, -255.0/totalMax, 255.0);
760 spins[i] = spin8u;
761 }
762 }
763
764 int sz = spins.front().cols;
765
766 Mat result((int)(yCount * sz + (yCount - 1)), (int)(xCount * sz + (xCount - 1)), CV_8UC3);
767 result = colors[(static_cast<int64>(cvGetTickCount()/cvGetTickFrequency())/1000) % colors_mum];
768
769 int pos = 0;
770 for(int y = 0; y < (int)yCount; ++y)
771 for(int x = 0; x < (int)xCount; ++x)
772 if (pos < num)
773 {
774 int starty = (y + 0) * sz + y;
775 int endy = (y + 1) * sz + y;
776
777 int startx = (x + 0) * sz + x;
778 int endx = (x + 1) * sz + x;
779
780 Mat color;
781 cvtColor(spins[pos++], color, CV_GRAY2BGR);
782 Mat roi = result(Range(starty, endy), Range(startx, endx));
783 color.copyTo(roi);
784 }
785 return result;
786}

788void cv::SpinImageModel::selectRandomSubset(float ratio)
789{
790 ratio = min(max(ratio, 0.f), 1.f);
791
792 size_t vtxSize = mesh.vtx.size();
793 size_t setSize = static_cast<size_t>(vtxSize * ratio);
794
795 if (setSize == 0)
796 {
797 subset.clear();
798 }
799 else if (setSize == vtxSize)
800 {
801 subset.resize(vtxSize);
802 _iota(subset.begin(), subset.end(), 0);
803 }
804 else
805 {
806 RNG& rnd = theRNG();
807
808 vector<size_t> left(vtxSize);
809 _iota(left.begin(), left.end(), (size_t)0);
810
811 subset.resize(setSize);
812 for(size_t i = 0; i < setSize; ++i)
813 {
814 int pos = rnd.next() % (int)left.size();
815 subset[i] = (int)left[pos];
816
817 left[pos] = left.back();
818 left.resize(left.size() - 1);
819 }
820 sort(subset, less<int>());
821 }

829void cv::SpinImageModel::repackSpinImages(const vector<uchar>& mask, Mat& _spinImages, bool reAlloc) const
830{
831 if (reAlloc)
832 {
833 size_t spinCount = mask.size() - count(mask.begin(), mask.end(), (uchar)0);
834 Mat newImgs((int)spinCount, _spinImages.cols, _spinImages.type());
835
836 int pos = 0;
837 for(size_t t = 0; t < mask.size(); ++t)
838 if (mask[t])
839 {
840 Mat row = newImgs.row(pos++);
841 _spinImages.row((int)t).copyTo(row);
842 }
843 _spinImages = newImgs;
844 }
845 else
846 {
847 int last = (int)mask.size();
848
849 int dest = (int)(find(mask.begin(), mask.end(), (uchar)0) - mask.begin());
850 if (dest == last)
851 return;
852
853 int first = dest + 1;
854 for (; first != last; ++first)
855 if (mask[first] != 0)
856 {
857 Mat row = _spinImages.row(dest);
858 _spinImages.row(first).copyTo(row);
859 ++dest;
860 }
861 _spinImages = _spinImages.rowRange(0, dest);
862 }

865void cv::SpinImageModel::compute()
866{
867 /* estimate binSize */
868 if (binSize == 0.f)
869 {
870 if (mesh.resolution == -1.f)
871 mesh.estimateResolution();
872 binSize = mesh.resolution;
873 }
874 /* estimate normalRadius */
875 normalRadius = normalRadius != 0.f ? normalRadius : binSize * imageWidth / 2;
876
877 mesh.buildOctree();
878 if (subset.empty())
879 {
880 mesh.computeNormals(normalRadius, minNeighbors);
881 subset.resize(mesh.vtx.size());
882 _iota(subset.begin(), subset.end(), 0);
883 }
884 else
885 mesh.computeNormals(subset, normalRadius, minNeighbors);
886
887 vector<uchar> mask(mesh.vtx.size(), 0);
888 for(size_t i = 0; i < subset.size(); ++i)
889 if (mesh.normals[subset[i]] == Mesh3D::allzero)
890 subset[i] = -1;
891 else
892 mask[subset[i]] = 1;
893 subset.resize( remove(subset.begin(), subset.end(), -1) - subset.begin() );
894
895 vector<Point3f> vtx;
896 vector<Point3f> normals;
897 for(size_t i = 0; i < mask.size(); ++i)
898 if(mask[i])
899 {
900 vtx.push_back(mesh.vtx[i]);
901 normals.push_back(mesh.normals[i]);
902 }
903
904 vector<uchar> spinMask(vtx.size(), 1);
905 computeSpinImages( mesh.octree, vtx, normals, spinMask, spinImages, imageWidth, binSize);
906 repackSpinImages(spinMask, spinImages);
907
908 size_t mask_pos = 0;
909 for(size_t i = 0; i < mask.size(); ++i)
910 if(mask[i])
911 if (spinMask[mask_pos++] == 0)
912 subset.resize( remove(subset.begin(), subset.end(), (int)i) - subset.begin() );
913}

915void cv::SpinImageModel::matchSpinToModel(const Mat& spin, vector<int>& indeces, vector<float>& corrCoeffs, bool useExtremeOutliers) const
916{
917 const SpinImageModel& model = *this;
918
919 indeces.clear();
920 corrCoeffs.clear();
921
922 vector<float> corrs(model.spinImages.rows);
923 vector<uchar> masks(model.spinImages.rows);
924 vector<float> cleanCorrs;
925 cleanCorrs.reserve(model.spinImages.rows);
926
927 for(int i = 0; i < model.spinImages.rows; ++i)
928 {
929 masks[i] = spinCorrelation(spin, model.spinImages.row(i), model.lambda, corrs[i]);
930 if (masks[i])
931 cleanCorrs.push_back(corrs[i]);
932 }
933
934 /* Filtering by measure histogram */
935 size_t total = cleanCorrs.size();
936 if(total < 5)
937 return;
938
939 sort(cleanCorrs, less<float>());
940
941 float lower_fourth = cleanCorrs[(1 * total) / 4 - 1];
942 float upper_fourth = cleanCorrs[(3 * total) / 4 - 0];
943 float fourth_spread = upper_fourth - lower_fourth;
944
945 //extreme or moderate?
946 float coef = useExtremeOutliers ? 3.0f : 1.5f;
947
948 float histThresHi = upper_fourth + coef * fourth_spread;
949 //float histThresLo = lower_fourth - coef * fourth_spread;
950
951 for(size_t i = 0; i < corrs.size(); ++i)
952 if (masks[i])
953 if (/* corrs[i] < histThresLo || */ corrs[i] > histThresHi)
954 {
955 indeces.push_back((int)i);
956 corrCoeffs.push_back(corrs[i]);
957 }
958}

1002 void cv::SpinImageModel::match(const SpinImageModel& scene, vector< vector<Vec2i> >& result)
1003{
1004 if (mesh.vtx.empty())
1005 throw Mesh3D::EmptyMeshException();
1006
1007 result.clear();
1008
1009 SpinImageModel& model = *this;
1010 const float infinity = numeric_limits<float>::infinity();
1011 const float float_max = numeric_limits<float>::max();
1012
1013 /* estimate gamma */
1014 if (model.gamma == 0.f)
1015 {
1016 if (model.mesh.resolution == -1.f)
1017 model.mesh.estimateResolution();
1018 model.gamma = 4 * model.mesh.resolution;
1019 }
1020
1021 /* estimate lambda */
1022 if (model.lambda == 0.f)
1023 {
1024 vector<int> nonzero(model.spinImages.rows);
1025 for(int i = 0; i < model.spinImages.rows; ++i)
1026 nonzero[i] = countNonZero(model.spinImages.row(i));
1027 sort(nonzero, less<int>());
1028 model.lambda = static_cast<float>( nonzero[ nonzero.size()/2 ] ) / 2;
1029 }
1030
1031 TickMeter corr_timer;
1032 corr_timer.start();
1033 vector<Match> allMatches;
1034 for(int i = 0; i < scene.spinImages.rows; ++i)
1035 {
1036 vector<int> indeces;
1037 vector<float> coeffs;
1038 matchSpinToModel(scene.spinImages.row(i), indeces, coeffs);
1039 for(size_t t = 0; t < indeces.size(); ++t)
1040 allMatches.push_back(Match(i, indeces[t], coeffs[t]));
1041
1042 if (out) if (i % 100 == 0) *out << "Comparing scene spinimage " << i << " of " << scene.spinImages.rows << endl;
1043 }
1044 corr_timer.stop();
1045 if (out) *out << "Spin correlation time = " << corr_timer << endl;
1046 if (out) *out << "Matches number = " << allMatches.size() << endl;
1047
1048 if(allMatches.empty())
1049 return;
1050
1051 /* filtering by similarity measure */
1052 const float fraction = 0.5f;
1053 float maxMeasure = max_element(allMatches.begin(), allMatches.end(), less<float>())->measure;
1054 allMatches.erase(
1055 remove_if(allMatches.begin(), allMatches.end(), bind2nd(less<float>(), maxMeasure * fraction)),
1056 allMatches.end());
1057 if (out) *out << "Matches number [filtered by similarity measure] = " << allMatches.size() << endl;
1058
1059 int matchesSize = (int)allMatches.size();
1060 if(matchesSize == 0)
1061 return;
1062
1063 /* filtering by geometric consistency */
1064 for(int i = 0; i < matchesSize; ++i)
1065 {
1066 int consistNum = 1;
1067 float gc = float_max;
1068
1069 for(int j = 0; j < matchesSize; ++j)
1070 if (i != j)
1071 {
1072 const Match& mi = allMatches[i];
1073 const Match& mj = allMatches[j];
1074
1075 if (mi.sceneInd == mj.sceneInd || mi.modelInd == mj.modelInd)
1076 gc = float_max;
1077 else
1078 {
1079 const Point3f& pointSceneI = scene.getSpinVertex(mi.sceneInd);
1080 const Point3f& normalSceneI = scene.getSpinNormal(mi.sceneInd);
1081
1082 const Point3f& pointModelI = model.getSpinVertex(mi.modelInd);
1083 const Point3f& normalModelI = model.getSpinNormal(mi.modelInd);
1084
1085 const Point3f& pointSceneJ = scene.getSpinVertex(mj.sceneInd);
1086 const Point3f& normalSceneJ = scene.getSpinNormal(mj.sceneInd);
1087
1088 const Point3f& pointModelJ = model.getSpinVertex(mj.modelInd);
1089 const Point3f& normalModelJ = model.getSpinNormal(mj.modelInd);
1090
1091 gc = geometricConsistency(pointSceneI, normalSceneI, pointModelI, normalModelI,
1092 pointSceneJ, normalSceneJ, pointModelJ, normalModelJ);
1093 }
1094
1095 if (gc < model.T_GeometriccConsistency)
1096 ++consistNum;
1097 }
1098
1099
1100 if (consistNum < matchesSize / 4) /* failed consistensy test */
1101 allMatches[i].measure = infinity;
1102 }
1103 allMatches.erase(
1104 remove_if(allMatches.begin(), allMatches.end(), bind2nd(equal_to<float>(), infinity)),
1105 allMatches.end());
1106 if (out) *out << "Matches number [filtered by geometric consistency] = " << allMatches.size() << endl;
1107
1108
1109 matchesSize = (int)allMatches.size();
1110 if(matchesSize == 0)
1111 return;
1112
1113 if (out) *out << "grouping ..." << endl;
1114
1115 Mat groupingMat((int)matchesSize, (int)matchesSize, CV_32F);
1116 groupingMat = Scalar(0);
1117
1118 /* grouping */
1119 for(int j = 0; j < matchesSize; ++j)
1120 for(int i = j + 1; i < matchesSize; ++i)
1121 {
1122 const Match& mi = allMatches[i];
1123 const Match& mj = allMatches[j];
1124
1125 if (mi.sceneInd == mj.sceneInd || mi.modelInd == mj.modelInd)
1126 {
1127 groupingMat.ptr<float>(i)[j] = float_max;
1128 groupingMat.ptr<float>(j)[i] = float_max;
1129 continue;
1130 }
1131
1132 const Point3f& pointSceneI = scene.getSpinVertex(mi.sceneInd);
1133 const Point3f& normalSceneI = scene.getSpinNormal(mi.sceneInd);
1134
1135 const Point3f& pointModelI = model.getSpinVertex(mi.modelInd);
1136 const Point3f& normalModelI = model.getSpinNormal(mi.modelInd);
1137
1138 const Point3f& pointSceneJ = scene.getSpinVertex(mj.sceneInd);
