71inline static
72void computeC_RigidBodyMotion( double* C, double dIdx, double dIdy, const Point3f& p3d, double fx, double fy )
73{
74 double invz = 1. / p3d.z,
75 v0 = dIdx * fx * invz,
76 v1 = dIdy * fy * invz,
77 v2 = -(v0 * p3d.x + v1 * p3d.y) * invz;
78
79 C[0] = -p3d.z * v1 + p3d.y * v2;
80 C[1] = p3d.z * v0 - p3d.x * v2;
81 C[2] = -p3d.y * v0 + p3d.x * v1;
82 C[3] = v0;
83 C[4] = v1;
84 C[5] = v2;
85}
86
87inline static
88void computeC_Rotation( double* C, double dIdx, double dIdy, const Point3f& p3d, double fx, double fy )
89{
90 double invz = 1. / p3d.z,
91 v0 = dIdx * fx * invz,
92 v1 = dIdy * fy * invz,
93 v2 = -(v0 * p3d.x + v1 * p3d.y) * invz;
94
95 C[0] = -p3d.z * v1 + p3d.y * v2;
96 C[1] = p3d.z * v0 - p3d.x * v2;
97 C[2] = -p3d.y * v0 + p3d.x * v1;
98}
99
100inline static
101void computeC_Translation( double* C, double dIdx, double dIdy, const Point3f& p3d, double fx, double fy )
102{
103 double invz = 1. / p3d.z,
104 v0 = dIdx * fx * invz,
105 v1 = dIdy * fy * invz,
106 v2 = -(v0 * p3d.x + v1 * p3d.y) * invz;
107
108 C[0] = v0;
109 C[1] = v1;
110 C[2] = v2;
111}

113inline static
114void computeProjectiveMatrix( const Mat& ksi, Mat& Rt )
115{
116 CV_Assert( ksi.size() == Size(1,6) && ksi.type() == CV_64FC1 );
117
118#if defined(HAVE_EIGEN) && EIGEN_WORLD_VERSION == 3 && (!defined _MSC_VER || !defined _M_X64 || _MSC_VER > 1500)
119 const double* ksi_ptr = reinterpret_cast<const double*>(ksi.ptr(0));
120 Eigen::Matrix<double,4,4> twist, g;
121 twist << 0., -ksi_ptr[2], ksi_ptr[1], ksi_ptr[3],
122 ksi_ptr[2], 0., -ksi_ptr[0], ksi_ptr[4],
123 -ksi_ptr[1], ksi_ptr[0], 0, ksi_ptr[5],
124 0., 0., 0., 0.;
125 g = twist.exp();
126
127
128 eigen2cv(g, Rt);
129#else
130 // for infinitesimal transformation
131 Rt = Mat::eye(4, 4, CV_64FC1);
132
133 Mat R = Rt(Rect(0,0,3,3));
134 Mat rvec = ksi.rowRange(0,3);
135
136 Rodrigues( rvec, R );
137
138 Rt.at<double>(0,3) = ksi.at<double>(3);
139 Rt.at<double>(1,3) = ksi.at<double>(4);
140 Rt.at<double>(2,3) = ksi.at<double>(5);
141#endif
142}

144static
145void cvtDepth2Cloud( const Mat& depth, Mat& cloud, const Mat& cameraMatrix )
146{
147 CV_Assert( cameraMatrix.type() == CV_64FC1 );
148 const double inv_fx = 1.f/cameraMatrix.at<double>(0,0);
149 const double inv_fy = 1.f/cameraMatrix.at<double>(1,1);
150 const double ox = cameraMatrix.at<double>(0,2);
151 const double oy = cameraMatrix.at<double>(1,2);
152 cloud.create( depth.size(), CV_32FC3 );
153 for( int y = 0; y < cloud.rows; y++ )
154 {
155 Point3f* cloud_ptr = reinterpret_cast<Point3f*>(cloud.ptr(y));
156 const float* depth_prt = reinterpret_cast<const float*>(depth.ptr(y));
157 for( int x = 0; x < cloud.cols; x++ )
158 {
159 float z = depth_prt[x];
160 cloud_ptr[x].x = (float)((x - ox) * z * inv_fx);
161 cloud_ptr[x].y = (float)((y - oy) * z * inv_fy);
162 cloud_ptr[x].z = z;
163 }
164 }
165}

168template<class ImageElemType>
169static void warpImage( const Mat& image, const Mat& depth,
170 const Mat& Rt, const Mat& cameraMatrix, const Mat& distCoeff,
171 Mat& warpedImage )
172{
173 const Rect rect = Rect(0, 0, image.cols, image.rows);
174
175 vector<Point2f> points2d;
176 Mat cloud, transformedCloud;
177
178 cvtDepth2Cloud( depth, cloud, cameraMatrix );
179 perspectiveTransform( cloud, transformedCloud, Rt );
180 projectPoints( transformedCloud.reshape(3,1), Mat::eye(3,3,CV_64FC1), Mat::zeros(3,1,CV_64FC1), cameraMatrix, distCoeff, points2d );
181
182 Mat pointsPositions( points2d );
183 pointsPositions = pointsPositions.reshape( 2, image.rows );
184
185 warpedImage.create( image.size(), image.type() );
186 warpedImage = Scalar::all(0);
187
188 Mat zBuffer( image.size(), CV_32FC1, FLT_MAX );
189 for( int y = 0; y < image.rows; y++ )
190 {
191 for( int x = 0; x < image.cols; x++ )
192 {
193 const Point3f p3d = transformedCloud.at<Point3f>(y,x);
194 const Point p2d = pointsPositions.at<Point2f>(y,x);
195 if( !cvIsNaN(cloud.at<Point3f>(y,x).z) && cloud.at<Point3f>(y,x).z > 0 &&
196 rect.contains(p2d) && zBuffer.at<float>(p2d) > p3d.z )
197 {
198 warpedImage.at<ImageElemType>(p2d) = image.at<ImageElemType>(y,x);
199 zBuffer.at<float>(p2d) = p3d.z;
200 }
201 }
202 }
203}

206static inline
207void set2shorts( int& dst, int short_v1, int short_v2 )
208{
209 unsigned short* ptr = reinterpret_cast<unsigned short*>(&dst);
210 ptr[0] = static_cast<unsigned short>(short_v1);
211 ptr[1] = static_cast<unsigned short>(short_v2);
212}

214static inline
215void get2shorts( int src, int& short_v1, int& short_v2 )
216{
217 typedef union { int vint32; unsigned short vuint16[2]; } s32tou16;
218 const unsigned short* ptr = (reinterpret_cast<s32tou16*>(&src))->vuint16;
219 short_v1 = ptr[0];
220 short_v2 = ptr[1];
221}

223static
224int computeCorresp( const Mat& K, const Mat& K_inv, const Mat& Rt,
225 const Mat& depth0, const Mat& depth1, const Mat& texturedMask1, float maxDepthDiff,
226 Mat& corresps )
227{
228 CV_Assert( K.type() == CV_64FC1 );
229 CV_Assert( K_inv.type() == CV_64FC1 );
230 CV_Assert( Rt.type() == CV_64FC1 );
231
232 corresps.create( depth1.size(), CV_32SC1 );
233
234 Mat R = Rt(Rect(0,0,3,3)).clone();
235
236 Mat KRK_inv = K * R * K_inv;
237 const double * KRK_inv_ptr = reinterpret_cast<const double *>(KRK_inv.ptr());
238
239 Mat Kt = Rt(Rect(3,0,1,3)).clone();
240 Kt = K * Kt;
241 const double * Kt_ptr = reinterpret_cast<const double *>(Kt.ptr());
242
243 Rect r(0, 0, depth1.cols, depth1.rows);
244
245 corresps = Scalar(-1);
246 int correspCount = 0;
247 for( int v1 = 0; v1 < depth1.rows; v1++ )
248 {
249 for( int u1 = 0; u1 < depth1.cols; u1++ )
250 {
251 float d1 = depth1.at<float>(v1,u1);
252 if( !cvIsNaN(d1) && texturedMask1.at<uchar>(v1,u1) )
253 {
254 float transformed_d1 = (float)(d1 * (KRK_inv_ptr[6] * u1 + KRK_inv_ptr[7] * v1 + KRK_inv_ptr[8]) + Kt_ptr[2]);
255 int u0 = cvRound((d1 * (KRK_inv_ptr[0] * u1 + KRK_inv_ptr[1] * v1 + KRK_inv_ptr[2]) + Kt_ptr[0]) / transformed_d1);
256 int v0 = cvRound((d1 * (KRK_inv_ptr[3] * u1 + KRK_inv_ptr[4] * v1 + KRK_inv_ptr[5]) + Kt_ptr[1]) / transformed_d1);
257
258 if( r.contains(Point(u0,v0)) )
259 {
260 float d0 = depth0.at<float>(v0,u0);
261 if( !cvIsNaN(d0) && std::abs(transformed_d1 - d0) <= maxDepthDiff )
262 {
263 int c = corresps.at<int>(v0,u0);
264 if( c != -1 )
265 {
266 int exist_u1, exist_v1;
267 get2shorts( c, exist_u1, exist_v1);
268
269 float exist_d1 = (float)(depth1.at<float>(exist_v1,exist_u1) * (KRK_inv_ptr[6] * exist_u1 + KRK_inv_ptr[7] * exist_v1 + KRK_inv_ptr[8]) + Kt_ptr[2]);
270
271 if( transformed_d1 > exist_d1 )
272 continue;
273 }
274 else
275 correspCount++;
276
277 set2shorts( corresps.at<int>(v0,u0), u1, v1 );
278 }
279 }
280 }
281 }
282 }
283
284 return correspCount;
285}

287static inline
288void preprocessDepth( Mat depth0, Mat depth1,
289 const Mat& validMask0, const Mat& validMask1,
290 float minDepth, float maxDepth )
291{
292 CV_DbgAssert( depth0.size() == depth1.size() );
293
294 for( int y = 0; y < depth0.rows; y++ )
295 {
296 for( int x = 0; x < depth0.cols; x++ )
297 {
298 float& d0 = depth0.at<float>(y,x);
299 if( !cvIsNaN(d0) && (d0 > maxDepth || d0 < minDepth || d0 <= 0 || (!validMask0.empty() && !validMask0.at<uchar>(y,x))) )
300 d0 = std::numeric_limits<float>::quiet_NaN();
301
302 float& d1 = depth1.at<float>(y,x);
303 if( !cvIsNaN(d1) && (d1 > maxDepth || d1 < minDepth || d1 <= 0 || (!validMask1.empty() && !validMask1.at<uchar>(y,x))) )
304 d1 = std::numeric_limits<float>::quiet_NaN();
305 }
306 }
307}

309static
310void buildPyramids( const Mat& image0, const Mat& image1,
311 const Mat& depth0, const Mat& depth1,
312 const Mat& cameraMatrix, int sobelSize, double sobelScale,
313 const vector<float>& minGradMagnitudes,
314 vector<Mat>& pyramidImage0, vector<Mat>& pyramidDepth0,
315 vector<Mat>& pyramidImage1, vector<Mat>& pyramidDepth1,
316 vector<Mat>& pyramid_dI_dx1, vector<Mat>& pyramid_dI_dy1,
317 vector<Mat>& pyramidTexturedMask1, vector<Mat>& pyramidCameraMatrix )
318{
319 const int pyramidMaxLevel = (int)minGradMagnitudes.size() - 1;
320
321 buildPyramid( image0, pyramidImage0, pyramidMaxLevel );
322 buildPyramid( image1, pyramidImage1, pyramidMaxLevel );
323
324 pyramid_dI_dx1.resize( pyramidImage1.size() );
325 pyramid_dI_dy1.resize( pyramidImage1.size() );
326 pyramidTexturedMask1.resize( pyramidImage1.size() );
327
328 pyramidCameraMatrix.reserve( pyramidImage1.size() );
329
330 Mat cameraMatrix_dbl;
331 cameraMatrix.convertTo( cameraMatrix_dbl, CV_64FC1 );
332
333 for( size_t i = 0; i < pyramidImage1.size(); i++ )
334 {
335 Sobel( pyramidImage1[i], pyramid_dI_dx1[i], CV_16S, 1, 0, sobelSize );
336 Sobel( pyramidImage1[i], pyramid_dI_dy1[i], CV_16S, 0, 1, sobelSize );
337
338 const Mat& dx = pyramid_dI_dx1[i];
339 const Mat& dy = pyramid_dI_dy1[i];
340
341 Mat texturedMask( dx.size(), CV_8UC1, Scalar(0) );
342 const float minScalesGradMagnitude2 = (float)((minGradMagnitudes[i] * minGradMagnitudes[i]) / (sobelScale * sobelScale));
343 for( int y = 0; y < dx.rows; y++ )
344 {
345 for( int x = 0; x < dx.cols; x++ )
346 {
347 float m2 = (float)(dx.at<short>(y,x)*dx.at<short>(y,x) + dy.at<short>(y,x)*dy.at<short>(y,x));
348 if( m2 >= minScalesGradMagnitude2 )
349 texturedMask.at<uchar>(y,x) = 255;
350 }
351 }
352 pyramidTexturedMask1[i] = texturedMask;
353 Mat levelCameraMatrix = i == 0 ? cameraMatrix_dbl : 0.5f * pyramidCameraMatrix[i-1];
354 levelCameraMatrix.at<double>(2,2) = 1.;
355 pyramidCameraMatrix.push_back( levelCameraMatrix );
356 }
357
358 buildPyramid( depth0, pyramidDepth0, pyramidMaxLevel );
359 buildPyramid( depth1, pyramidDepth1, pyramidMaxLevel );
360}

362static
363bool solveSystem( const Mat& C, const Mat& dI_dt, double detThreshold, Mat& ksi )
364{
365#if defined(HAVE_EIGEN) && EIGEN_WORLD_VERSION == 3
366 Eigen::Matrix<double, Eigen::Dynamic, Eigen::Dynamic> eC, eCt, edI_dt;
367 cv2eigen(C, eC);
368 cv2eigen(dI_dt, edI_dt);
369 eCt = eC.transpose();
370
371 Eigen::Matrix<double, Eigen::Dynamic, Eigen::Dynamic> A, B, eksi;
372
373 A = eCt * eC;
374 double det = A.determinant();
375 if( fabs (det) < detThreshold || cvIsNaN(det) || cvIsInf(det) )
376 return false;
377
378 B = -eCt * edI_dt;
379
380 eksi = A.ldlt().solve(B);
381 eigen2cv( eksi, ksi );
382
383#else
384 Mat A = C.t() * C;
385
386 double det = cv::determinant(A);
387
388 if( fabs (det) < detThreshold || cvIsNaN(det) || cvIsInf(det) )
389 return false;
390
391 Mat B = -C.t() * dI_dt;
392 cv::solve( A, B, ksi, DECOMP_CHOLESKY );
393#endif
394
395 return true;
396}

398typedef void (*ComputeCFuncPtr)( double* C, double dIdx, double dIdy, const Point3f& p3d, double fx, double fy );
399
400static
401bool computeKsi( int transformType,
402 const Mat& image0, const Mat& cloud0,
403 const Mat& image1, const Mat& dI_dx1, const Mat& dI_dy1,
404 const Mat& corresps, int correspsCount,
405 double fx, double fy, double sobelScale, double determinantThreshold,
406 Mat& ksi )
407{
408 int Cwidth = -1;
409 ComputeCFuncPtr computeCFuncPtr = 0;
410 if( transformType == RIGID_BODY_MOTION )
411 {
412 Cwidth = 6;
413 computeCFuncPtr = computeC_RigidBodyMotion;
414 }
415 else if( transformType == ROTATION )
416 {
417 Cwidth = 3;
418 computeCFuncPtr = computeC_Rotation;
419 }
420 else if( transformType == TRANSLATION )
421 {
422 Cwidth = 3;
423 computeCFuncPtr = computeC_Translation;
424 }
425 else
426 CV_Error( CV_StsBadFlag, "Unsupported value of transformation type flag.");
427
428 Mat C( correspsCount, Cwidth, CV_64FC1 );
429 Mat dI_dt( correspsCount, 1, CV_64FC1 );
430
431 double sigma = 0;
432 int pointCount = 0;
433 for( int v0 = 0; v0 < corresps.rows; v0++ )
434 {
435 for( int u0 = 0; u0 < corresps.cols; u0++ )
436 {
437 if( corresps.at<int>(v0,u0) != -1 )
438 {
439 int u1, v1;
440 get2shorts( corresps.at<int>(v0,u0), u1, v1 );
441 double diff = static_cast<double>(image1.at<uchar>(v1,u1)) -
442 static_cast<double>(image0.at<uchar>(v0,u0));
443 sigma += diff * diff;
444 pointCount++;
445 }
446 }
447 }
448 sigma = std::sqrt(sigma/pointCount);
449
450 pointCount = 0;
451 for( int v0 = 0; v0 < corresps.rows; v0++ )
452 {
453 for( int u0 = 0; u0 < corresps.cols; u0++ )
454 {
455 if( corresps.at<int>(v0,u0) != -1 )
456 {
457 int u1, v1;
458 get2shorts( corresps.at<int>(v0,u0), u1, v1 );
459
460 double diff = static_cast<double>(image1.at<uchar>(v1,u1)) -
461 static_cast<double>(image0.at<uchar>(v0,u0));
462 double w = sigma + std::abs(diff);
463 w = w > DBL_EPSILON ? 1./w : 1.;
464
465 (*computeCFuncPtr)( (double*)C.ptr(pointCount),
466 w * sobelScale * dI_dx1.at<short int>(v1,u1),
467 w * sobelScale * dI_dy1.at<short int>(v1,u1),
468 cloud0.at<Point3f>(v0,u0), fx, fy);
469
470 dI_dt.at<double>(pointCount) = w * diff;
471 pointCount++;
472 }
473 }
474 }
475
476 Mat sln;
477 bool solutionExist = solveSystem( C, dI_dt, determinantThreshold, sln );
478
479 if( solutionExist )
480 {
481 ksi.create(6,1,CV_64FC1);
482 ksi = Scalar(0);
483
484 Mat subksi;
485 if( transformType == RIGID_BODY_MOTION )
486 {
487 subksi = ksi;
488 }
489 else if( transformType == ROTATION )
490 {
491 subksi = ksi.rowRange(0,3);
492 }
493 else if( transformType == TRANSLATION )
494 {
495 subksi = ksi.rowRange(3,6);
496 }
497
498 sln.copyTo( subksi );
499 }
500
501 return solutionExist;
502}

504bool cv::RGBDOdometry( cv::Mat& Rt, const Mat& initRt,
505 const cv::Mat& image0, const cv::Mat& _depth0, const cv::Mat& validMask0,
506 const cv::Mat& image1, const cv::Mat& _depth1, const cv::Mat& validMask1,
507 const cv::Mat& cameraMatrix, float minDepth, float maxDepth, float maxDepthDiff,
508 const std::vector<int>& iterCounts, const std::vector<float>& minGradientMagnitudes,
509 int transformType )
510{
511 const int sobelSize = 3;
512 const double sobelScale = 1./8;
513
514 Mat depth0 = _depth0.clone(),
515 depth1 = _depth1.clone();
516
517 // check RGB-D input data
518 CV_Assert( !image0.empty() );
519 CV_Assert( image0.type() == CV_8UC1 );
520 CV_Assert( depth0.type() == CV_32FC1 && depth0.size() == image0.size() );
521
522 CV_Assert( image1.size() == image0.size() );
523 CV_Assert( image1.type() == CV_8UC1 );
524 CV_Assert( depth1.type() == CV_32FC1 && depth1.size() == image0.size() );
525
526 // check masks
527 CV_Assert( validMask0.empty() || (validMask0.type() == CV_8UC1 && validMask0.size() == image0.size()) );
528 CV_Assert( validMask1.empty() || (validMask1.type() == CV_8UC1 && validMask1.size() == image0.size()) );
529
530 // check camera params
531 CV_Assert( cameraMatrix.type() == CV_32FC1 && cameraMatrix.size() == Size(3,3) );
532
533 // other checks
534 CV_Assert( iterCounts.empty() || minGradientMagnitudes.empty() ||
535 minGradientMagnitudes.size() == iterCounts.size() );
536 CV_Assert( initRt.empty() || (initRt.type()==CV_64FC1 && initRt.size()==Size(4,4) ) );
537
538 vector<int> defaultIterCounts;
539 vector<float> defaultMinGradMagnitudes;
540 vector<int> const* iterCountsPtr = &iterCounts;
541 vector<float> const* minGradientMagnitudesPtr = &minGradientMagnitudes;
542
543 if( iterCounts.empty() || minGradientMagnitudes.empty() )
544 {
545 defaultIterCounts.resize(4);
546 defaultIterCounts[0] = 7;
547 defaultIterCounts[1] = 7;
548 defaultIterCounts[2] = 7;
549 defaultIterCounts[3] = 10;
550
551 defaultMinGradMagnitudes.resize(4);
552 defaultMinGradMagnitudes[0] = 12;
553 defaultMinGradMagnitudes[1] = 5;
554 defaultMinGradMagnitudes[2] = 3;
555 defaultMinGradMagnitudes[3] = 1;
556
557 iterCountsPtr = &defaultIterCounts;
558 minGradientMagnitudesPtr = &defaultMinGradMagnitudes;
559 }
560
561 preprocessDepth( depth0, depth1, validMask0, validMask1, minDepth, maxDepth );
562
563 vector<Mat> pyramidImage0, pyramidDepth0,
564 pyramidImage1, pyramidDepth1, pyramid_dI_dx1, pyramid_dI_dy1, pyramidTexturedMask1,
565 pyramidCameraMatrix;
566 buildPyramids( image0, image1, depth0, depth1, cameraMatrix, sobelSize, sobelScale, *minGradientMagnitudesPtr,
567 pyramidImage0, pyramidDepth0, pyramidImage1, pyramidDepth1,
568 pyramid_dI_dx1, pyramid_dI_dy1, pyramidTexturedMask1, pyramidCameraMatrix );
569
570 Mat resultRt = initRt.empty() ? Mat::eye(4,4,CV_64FC1) : initRt.clone();
571 Mat currRt, ksi;
572 for( int level = (int)iterCountsPtr->size() - 1; level >= 0; level-- )
573 {
574 const Mat& levelCameraMatrix = pyramidCameraMatrix[level];
575
576 const Mat& levelImage0 = pyramidImage0[level];
577 const Mat& levelDepth0 = pyramidDepth0[level];
578 Mat levelCloud0;
579 cvtDepth2Cloud( pyramidDepth0[level], levelCloud0, levelCameraMatrix );
580
581 const Mat& levelImage1 = pyramidImage1[level];
582 const Mat& levelDepth1 = pyramidDepth1[level];
583 const Mat& level_dI_dx1 = pyramid_dI_dx1[level];
584 const Mat& level_dI_dy1 = pyramid_dI_dy1[level];
585
586 CV_Assert( level_dI_dx1.type() == CV_16S );
587 CV_Assert( level_dI_dy1.type() == CV_16S );
588
589 const double fx = levelCameraMatrix.at<double>(0,0);
590 const double fy = levelCameraMatrix.at<double>(1,1);
591 const double determinantThreshold = 1e-6;
592
593 Mat corresps( levelImage0.size(), levelImage0.type() );
594
595 // Run transformation search on current level iteratively.
596 for( int iter = 0; iter < (*iterCountsPtr)[level]; iter ++ )
597 {
598 int correspsCount = computeCorresp( levelCameraMatrix, levelCameraMatrix.inv(), resultRt.inv(DECOMP_SVD),
599 levelDepth0, levelDepth1, pyramidTexturedMask1[level], maxDepthDiff,
600 corresps );
601
602 if( correspsCount == 0 )
603 break;
604
605 bool solutionExist = computeKsi( transformType,
606 levelImage0, levelCloud0,
607 levelImage1, level_dI_dx1, level_dI_dy1,
608 corresps, correspsCount,
609 fx, fy, sobelScale, determinantThreshold,
610 ksi );
611
612 if( !solutionExist )
613 break;
614
615 computeProjectiveMatrix( ksi, currRt );
616
617 resultRt = currRt * resultRt;
618
619#if SHOW_DEBUG_IMAGES
620 std::cout << "currRt " << currRt << std::endl;
621 Mat warpedImage0;
622 const Mat distCoeff(1,5,CV_32FC1,Scalar(0));
623 warpImage<uchar>( levelImage0, levelDepth0, resultRt, levelCameraMatrix, distCoeff, warpedImage0 );
624
625 imshow( "im0", levelImage0 );
626 imshow( "wim0", warpedImage0 );
627 imshow( "im1", levelImage1 );
628 waitKey();
629#endif
630 }
631 }
632
633 Rt = resultRt;
634
635 return !Rt.empty();
636}
