68/**
69* @class ImageLogPolProjection
70* @brief class able to perform a log sampling of an image input (models the log sampling of the photoreceptors of the retina)
71* or a log polar projection which models the retina information projection on the primary visual cortex: a linear projection in the center for detail analysis and a log projection of the borders (low spatial frequency motion information in general)
72*
73* collaboration: Barthelemy DURETTE who experimented the retina log projection
74-> "Traitement visuels Bio mimtiques pour la supplance perceptive", internal technical report, May 2005, Gipsa-lab/DIS, Grenoble, FRANCE
75*
76* * TYPICAL USE:
77*
78* // create object, here for a log sampling (keyword:RETINALOGPROJECTION): (dynamic object allocation sample)
79* ImageLogPolProjection *imageSamplingTool;
80* imageSamplingTool = new ImageLogPolProjection(frameSizeRows, frameSizeColumns, RETINALOGPROJECTION);
81*
82* // init log projection:
83* imageSamplingTool->initProjection(1.0, 15.0);
84*
85* // during program execution, call the log transform applied to a frame called "FrameBuffer" :
86* imageSamplingTool->runProjection(FrameBuffer);
87* // get output frame and its size:
88* const unsigned int logSampledFrame_nbRows=imageSamplingTool->getOutputNBrows();
89* const unsigned int logSampledFrame_nbColumns=imageSamplingTool->getOutputNBcolumns();
90* const double *logSampledFrame=imageSamplingTool->getSampledFrame();
91*
92* // at the end of the program, destroy object:
93* delete imageSamplingTool;
94*
95* @author Alexandre BENOIT, benoit.alexandre.vision@gmail.com, LISTIC : www.listic.univ-savoie.fr, Gipsa-Lab, France: www.gipsa-lab.inpg.fr/
96* Creation date 2007
97*/

101#include "basicretinafilter.hpp"
102
103
104namespace cv
105{
106
107class ImageLogPolProjection:public BasicRetinaFilter
108{
109public:
110
111 enum PROJECTIONTYPE{RETINALOGPROJECTION, CORTEXLOGPOLARPROJECTION};
112
113 /**
114 * constructor, just specifies the image input size and the projection type, no projection initialisation is done
115 * -> use initLogRetinaSampling() or initLogPolarCortexSampling() for that
116 * @param nbRows: number of rows of the input image
117 * @param nbColumns: number of columns of the input image
118 * @param projection: the type of projection, RETINALOGPROJECTION or CORTEXLOGPOLARPROJECTION
119 * @param colorMode: specifies if the projection is applied on a grayscale image (false) or color images (3 layers) (true)
120 */
121 ImageLogPolProjection(const unsigned int nbRows, const unsigned int nbColumns, const PROJECTIONTYPE projection, const bool colorMode=false);
122
123 /**
124 * standard destructor
125 */
126 virtual ~ImageLogPolProjection();
127
128 /**
129 * function that clears all buffers of the object
130 */
131 void clearAllBuffers();
132
133 /**
134 * resize retina color filter object (resize all allocated buffers)
135 * @param NBrows: the new height size
136 * @param NBcolumns: the new width size
137 */
138 void resize(const unsigned int NBrows, const unsigned int NBcolumns);
139
140 /**
141 * init function depending on the projection type
142 * @param reductionFactor: the size reduction factor of the ouptup image in regard of the size of the input image, must be superior to 1
143 * @param samplingStrenght: specifies the strenght of the log compression effect (magnifying coefficient)
144 * @return true if the init was performed without any errors
145 */
146 bool initProjection(const double reductionFactor, const double samplingStrenght);
147
148 /**
149 * main funtion of the class: run projection function
150 * @param inputFrame: the input frame to be processed
151 * @param colorMode: the input buffer color mode: false=gray levels, true = 3 color channels mode
152 * @return the output frame
153 */
154 std::valarray<float> &runProjection(const std::valarray<float> &inputFrame, const bool colorMode=false);
155
156 /**
157 * @return the numbers of rows (height) of the images OUTPUTS of the object
158 */
159 inline unsigned int getOutputNBrows(){return _outputNBrows;};
160
161 /**
162 * @return the numbers of columns (width) of the images OUTPUTS of the object
163 */
164 inline unsigned int getOutputNBcolumns(){return _outputNBcolumns;};
165
166 /**
167 * main funtion of the class: run projection function
168 * @param size: one of the input frame initial dimensions to be processed
169 * @return the output frame dimension
170 */
171 inline static unsigned int predictOutputSize(const unsigned int size, const double reductionFactor){return (unsigned int)((double)size/reductionFactor);};
172
173 /**
174 * @return the output of the filter which applies an irregular Low Pass spatial filter to the imag input (see function
175 */
176 inline const std::valarray<float> &getIrregularLPfilteredInputFrame() const {return _irregularLPfilteredFrame;};
177
178 /**
179 * function which allows to retrieve the output frame which was updated after the "runProjection(...) function BasicRetinaFilter::runProgressiveFilter(...)
180 * @return the projection result
181 */
182 inline const std::valarray<float> &getSampledFrame() const {return _sampledFrame;};
183
184 /**
185 * function which allows gives the tranformation table, its size is (getNBrows()*getNBcolumns()*2)
186 * @return the transformation matrix [outputPixIndex_i, inputPixIndex_i, outputPixIndex_i+1, inputPixIndex_i+1....]
187 */
188 inline const std::valarray<unsigned int> &getSamplingMap() const {return _transformTable;};
189
190 inline double getOriginalRadiusLength(const double projectedRadiusLength){return _azero/(_alim-projectedRadiusLength*2.0/_minDimension);};
191
192 // unsigned int getInputPixelIndex(const unsigned int ){ return _transformTable[index*2+1]};
193
194private:
195 PROJECTIONTYPE _selectedProjection;
196
197 // size of the image output
198 unsigned int _outputNBrows;
199 unsigned int _outputNBcolumns;
200 unsigned int _outputNBpixels;
201 unsigned int _outputDoubleNBpixels;
202 unsigned int _inputDoubleNBpixels;
203
204 // is the object able to manage color flag
205 bool _colorModeCapable;
206 // sampling strenght factor
207 double _samplingStrenght;
208 // sampling reduction factor
209 double _reductionFactor;
210
211 // log sampling parameters
212 double _azero;
213 double _alim;
214 double _minDimension;
215
216 // template buffers
217 std::valarray<float>_sampledFrame;
218 std::valarray<float>&_tempBuffer;
219 std::valarray<unsigned int>_transformTable;
220
221 std::valarray<float> &_irregularLPfilteredFrame; // just a reference for easier understanding
222 unsigned int _usefullpixelIndex;
223
224 // init transformation tables
225 bool _computeLogProjection();
226 bool _computeLogPolarProjection();
227
228 // specifies if init was done correctly
229 bool _initOK;
230 // private init projections functions called by "initProjection(...)" function
231 bool _initLogRetinaSampling(const double reductionFactor, const double samplingStrenght);
232 bool _initLogPolarCortexSampling(const double reductionFactor, const double samplingStrenght);
233
234 ImageLogPolProjection(const ImageLogPolProjection&);
235 ImageLogPolProjection& operator=(const ImageLogPolProjection&);
236
237};
238
239}
240#endif /*IMAGELOGPOLPROJECTION_H_*/
