171// run filter for a new frame input
172// output return is (*_parvocellularOutputONminusOFF)
173const std::valarray<float> &ParvoRetinaFilter::runFilter(const std::valarray<float> &inputFrame, const bool useParvoOutput)
174{
175 _spatiotemporalLPfilter(get_data(inputFrame), &_photoreceptorsOutput[0]);
176 _spatiotemporalLPfilter(&_photoreceptorsOutput[0], &_horizontalCellsOutput[0], 1);
177 _OPL_OnOffWaysComputing();
178
179 if (useParvoOutput)
180 {
181 // local adaptation processes on ON and OFF ways
182 _spatiotemporalLPfilter(&_bipolarCellsOutputON[0], &(*_localAdaptationON)[0], 2);
183 _localLuminanceAdaptation(&_parvocellularOutputON[0], &(*_localAdaptationON)[0]);
184
185 _spatiotemporalLPfilter(&_bipolarCellsOutputOFF[0], &_localAdaptationOFF[0], 2);
186 _localLuminanceAdaptation(&_parvocellularOutputOFF[0], &_localAdaptationOFF[0]);
187
188 //// Final loop that computes the main output of this filter
189 //
190 //// loop that makes the difference between photoreceptor cells output and horizontal cells
191 //// positive part goes on the ON way, negative pat goes on the OFF way
192 float *parvocellularOutputONminusOFF_PTR=&(*_parvocellularOutputONminusOFF)[0];
193 float *parvocellularOutputON_PTR=&_parvocellularOutputON[0];
194 float *parvocellularOutputOFF_PTR=&_parvocellularOutputOFF[0];
195
196 for (unsigned int IDpixel=0 ; IDpixel<_filterOutput.getNBpixels() ; ++IDpixel)
197 *(parvocellularOutputONminusOFF_PTR++)= (*(parvocellularOutputON_PTR++)-*(parvocellularOutputOFF_PTR++));
198 }
199 return (*_parvocellularOutputONminusOFF);
200}

202void ParvoRetinaFilter::_OPL_OnOffWaysComputing() // WARNING : this method requires many buffer accesses, parallelizing can increase bandwith & core efficacy
203{
204 // loop that makes the difference between photoreceptor cells output and horizontal cells
205 // positive part goes on the ON way, negative pat goes on the OFF way
206
207#ifdef MAKE_PARALLEL
208 cv::parallel_for_(cv::Range(0,_filterOutput.getNBpixels()), Parallel_OPL_OnOffWaysComputing(&_photoreceptorsOutput[0], &_horizontalCellsOutput[0], &_bipolarCellsOutputON[0], &_bipolarCellsOutputOFF[0], &_parvocellularOutputON[0], &_parvocellularOutputOFF[0]));
209#else
210 float *photoreceptorsOutput_PTR= &_photoreceptorsOutput[0];
211 float *horizontalCellsOutput_PTR= &_horizontalCellsOutput[0];
212 float *bipolarCellsON_PTR = &_bipolarCellsOutputON[0];
213 float *bipolarCellsOFF_PTR = &_bipolarCellsOutputOFF[0];
214 float *parvocellularOutputON_PTR= &_parvocellularOutputON[0];
215 float *parvocellularOutputOFF_PTR= &_parvocellularOutputOFF[0];
216 // compute bipolar cells response equal to photoreceptors minus horizontal cells response
217 // and copy the result on parvo cellular outputs... keeping time before their local contrast adaptation for final result
218 for (unsigned int IDpixel=0 ; IDpixel<_filterOutput.getNBpixels() ; ++IDpixel)
219 {
220 float pixelDifference = *(photoreceptorsOutput_PTR++) -*(horizontalCellsOutput_PTR++);
221 // test condition to allow write pixelDifference in ON or OFF buffer and 0 in the over
222 float isPositive=(float) (pixelDifference>0.0f);
223
224 // ON and OFF channels writing step
225 *(parvocellularOutputON_PTR++)=*(bipolarCellsON_PTR++) = isPositive*pixelDifference;
226 *(parvocellularOutputOFF_PTR++)=*(bipolarCellsOFF_PTR++)= (isPositive-1.0f)*pixelDifference;
227 }
228#endif
229}
