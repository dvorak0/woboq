134// init functions depending on the projection type
135bool ImageLogPolProjection::initProjection(const double reductionFactor, const double samplingStrenght)
136{
137 switch(_selectedProjection)
138 {
139 case RETINALOGPROJECTION:
140 return _initLogRetinaSampling(reductionFactor, samplingStrenght);
141 break;
142 case CORTEXLOGPOLARPROJECTION:
143 return _initLogPolarCortexSampling(reductionFactor, samplingStrenght);
144 break;
145 default:
146 std::cout<<"ImageLogPolProjection::no projection setted up... performing default retina projection... take care"<<std::endl;
147 return _initLogRetinaSampling(reductionFactor, samplingStrenght);
148 break;
149 }
150}

152// -> private init functions dedicated to each projection
153bool ImageLogPolProjection::_initLogRetinaSampling(const double reductionFactor, const double samplingStrenght)
154{
155 _initOK=false;
156
157 if (_selectedProjection!=RETINALOGPROJECTION)
158 {
159 std::cerr<<"ImageLogPolProjection::initLogRetinaSampling: could not initialize logPolar projection for a log projection system\n -> you probably chose the wrong init function, use initLogPolarCortexSampling() instead"<<std::endl;
160 return false;
161 }
162 if (reductionFactor<1.0)
163 {
164 std::cerr<<"ImageLogPolProjection::initLogRetinaSampling: reduction factor must be superior to 0, skeeping initialisation..."<<std::endl;
165 return false;
166 }
167
168 // compute image output size
169 _outputNBrows=predictOutputSize(this->getNBrows(), reductionFactor);
170 _outputNBcolumns=predictOutputSize(this->getNBcolumns(), reductionFactor);
171 _outputNBpixels=_outputNBrows*_outputNBcolumns;
172 _outputDoubleNBpixels=_outputNBrows*_outputNBcolumns*2;
173
174#ifdef IMAGELOGPOLPROJECTION_DEBUG
175 std::cout<<"ImageLogPolProjection::initLogRetinaSampling: Log resampled image resampling factor: "<<reductionFactor<<", strenght:"<<samplingStrenght<<std::endl;
176 std::cout<<"ImageLogPolProjection::initLogRetinaSampling: Log resampled image size: "<<_outputNBrows<<"*"<<_outputNBcolumns<<std::endl;
177#endif
178
179 // setup progressive prefilter that will be applied BEFORE log sampling
180 setProgressiveFilterConstants_CentredAccuracy(0.f, 0.f, 0.99f);
181
182 // (re)create the image output buffer and transform table if the reduction factor changed
183 _sampledFrame.resize(_outputNBpixels*(1+(unsigned int)_colorModeCapable*2));
184
185 // specifiying new reduction factor after preliminar checks
186 _reductionFactor=reductionFactor;
187 _samplingStrenght=samplingStrenght;
188
189 // compute the rlim for symetric rows/columns sampling, then, the rlim is based on the smallest dimension
190 _minDimension=(double)(_filterOutput.getNBrows() < _filterOutput.getNBcolumns() ? _filterOutput.getNBrows() : _filterOutput.getNBcolumns());
191
192 // input frame dimensions dependent log sampling:
193 //double rlim=1.0/reductionFactor*(minDimension/2.0+samplingStrenght);
194
195 // input frame dimensions INdependent log sampling:
196 _azero=(1.0+reductionFactor*sqrt(samplingStrenght))/(reductionFactor*reductionFactor*samplingStrenght-1.0);
197 _alim=(1.0+_azero)/reductionFactor;
198#ifdef IMAGELOGPOLPROJECTION_DEBUG
199 std::cout<<"ImageLogPolProjection::initLogRetinaSampling: rlim= "<<rlim<<std::endl;
200 std::cout<<"ImageLogPolProjection::initLogRetinaSampling: alim= "<<alim<<std::endl;
201#endif
202
203 // get half frame size
204 unsigned int halfOutputRows = _outputNBrows/2-1;
205 unsigned int halfOutputColumns = _outputNBcolumns/2-1;
206 unsigned int halfInputRows = _filterOutput.getNBrows()/2-1;
207 unsigned int halfInputColumns = _filterOutput.getNBcolumns()/2-1;
208
209 // computing log sampling matrix by computing quarters of images
210 // the original new image center (_filterOutput.getNBrows()/2, _filterOutput.getNBcolumns()/2) being at coordinate (_filterOutput.getNBrows()/(2*_reductionFactor), _filterOutput.getNBcolumns()/(2*_reductionFactor))
211
212 // -> use a temporary transform table which is bigger than the final one, we only report pixels coordinates that are included in the sampled picture
213 std::valarray<unsigned int> tempTransformTable(2*_outputNBpixels); // the structure would be: (pixelInputCoordinate n)(pixelOutputCoordinate n)(pixelInputCoordinate n+1)(pixelOutputCoordinate n+1)
214 _usefullpixelIndex=0;
215
216 double rMax=0;
217 halfInputRows<halfInputColumns ? rMax=(double)(halfInputRows*halfInputRows):rMax=(double)(halfInputColumns*halfInputColumns);
218
219 for (unsigned int idRow=0;idRow<halfOutputRows; ++idRow)
220 {
221 for (unsigned int idColumn=0;idColumn<halfOutputColumns; ++idColumn)
222 {
223 // get the pixel position in the original picture
224
225 // -> input frame dimensions dependent log sampling:
226 //double scale = samplingStrenght/(rlim-(double)sqrt(idRow*idRow+idColumn*idColumn));
227
228 // -> input frame dimensions INdependent log sampling:
229 double scale=getOriginalRadiusLength((double)sqrt((double)(idRow*idRow+idColumn*idColumn)));
230#ifdef IMAGELOGPOLPROJECTION_DEBUG
231 std::cout<<"ImageLogPolProjection::initLogRetinaSampling: scale= "<<scale<<std::endl;
232 std::cout<<"ImageLogPolProjection::initLogRetinaSampling: scale2= "<<scale2<<std::endl;
233#endif
234 if (scale < 0) ///check it later
235 scale = 10000;
236
237#ifdef IMAGELOGPOLPROJECTION_DEBUG
238 // std::cout<<"ImageLogPolProjection::initLogRetinaSampling: scale= "<<scale<<std::endl;
239#endif
240
241 unsigned int u=(unsigned int)floor((double)idRow*scale);
242 unsigned int v=(unsigned int)floor((double)idColumn*scale);
243
244 // manage border effects
245 double length=u*u+v*v;
246 double radiusRatio=sqrt(rMax/length);
247
248#ifdef IMAGELOGPOLPROJECTION_DEBUG
249 std::cout<<"ImageLogPolProjection::(inputH, inputW)="<<halfInputRows<<", "<<halfInputColumns<<", Rmax2="<<rMax<<std::endl;
250 std::cout<<"before ==> ImageLogPolProjection::(u, v)="<<u<<", "<<v<<", r="<<u*u+v*v<<std::endl;
251 std::cout<<"ratio ="<<radiusRatio<<std::endl;
252#endif
253
254 if (radiusRatio < 1.0)
255 {
256 u=(unsigned int)floor(radiusRatio*double(u));
257 v=(unsigned int)floor(radiusRatio*double(v));
258 }
259#ifdef IMAGELOGPOLPROJECTION_DEBUG
260 std::cout<<"after ==> ImageLogPolProjection::(u, v)="<<u<<", "<<v<<", r="<<u*u+v*v<<std::endl;
261 std::cout<<"ImageLogPolProjection::("<<(halfOutputRows-idRow)<<", "<<idColumn+halfOutputColumns<<") <- ("<<halfInputRows-u<<", "<<v+halfInputColumns<<")"<<std::endl;
262 std::cout<<(halfOutputRows-idRow)+(halfOutputColumns+idColumn)*_outputNBrows<<" -> "<<(halfInputRows-u)+_filterOutput.getNBrows()*(halfInputColumns+v)<<std::endl;
263#endif
264
265 if ((u<halfInputRows)&&(v<halfInputColumns))
266 {
267
268#ifdef IMAGELOGPOLPROJECTION_DEBUG
269 std::cout<<"*** VALID ***"<<std::endl;
270#endif
271
272 // set pixel coordinate of the input picture in the transform table at the current log sampled pixel
273 // 1st quadrant
274 tempTransformTable[_usefullpixelIndex++]=(halfOutputColumns+idColumn)+(halfOutputRows-idRow)*_outputNBcolumns;
275 tempTransformTable[_usefullpixelIndex++]=_filterOutput.getNBcolumns()*(halfInputRows-u)+(halfInputColumns+v);
276 // 2nd quadrant
277 tempTransformTable[_usefullpixelIndex++]=(halfOutputColumns+idColumn)+(halfOutputRows+idRow)*_outputNBcolumns;
278 tempTransformTable[_usefullpixelIndex++]=_filterOutput.getNBcolumns()*(halfInputRows+u)+(halfInputColumns+v);
279 // 3rd quadrant
280 tempTransformTable[_usefullpixelIndex++]=(halfOutputColumns-idColumn)+(halfOutputRows-idRow)*_outputNBcolumns;
281 tempTransformTable[_usefullpixelIndex++]=_filterOutput.getNBcolumns()*(halfInputRows-u)+(halfInputColumns-v);
282 // 4td quadrant
283 tempTransformTable[_usefullpixelIndex++]=(halfOutputColumns-idColumn)+(halfOutputRows+idRow)*_outputNBcolumns;
284 tempTransformTable[_usefullpixelIndex++]=_filterOutput.getNBcolumns()*(halfInputRows+u)+(halfInputColumns-v);
285 }
286 }
287 }
288
289 // (re)creating and filling the transform table
290 _transformTable.resize(_usefullpixelIndex);
291 memcpy(&_transformTable[0], &tempTransformTable[0], sizeof(unsigned int)*_usefullpixelIndex);
292
293 // reset all buffers
294 clearAllBuffers();
295
296#ifdef IMAGELOGPOLPROJECTION_DEBUG
297 std::cout<<"ImageLogPolProjection::initLogRetinaSampling: init done successfully"<<std::endl;
298#endif
299 _initOK=true;
300 return _initOK;
301}

303bool ImageLogPolProjection::_initLogPolarCortexSampling(const double reductionFactor, const double)
304{
305 _initOK=false;
306
307 if (_selectedProjection!=CORTEXLOGPOLARPROJECTION)
308 {
309 std::cerr<<"ImageLogPolProjection::could not initialize log projection for a logPolar projection system\n -> you probably chose the wrong init function, use initLogRetinaSampling() instead"<<std::endl;
310 return false;
311 }
312
313 if (reductionFactor<1.0)
314 {
315 std::cerr<<"ImageLogPolProjection::reduction factor must be superior to 0, skeeping initialisation..."<<std::endl;
316 return false;
317 }
318
319 // compute the smallest image size
320 unsigned int minDimension=(_filterOutput.getNBrows() < _filterOutput.getNBcolumns() ? _filterOutput.getNBrows() : _filterOutput.getNBcolumns());
321 // specifiying new reduction factor after preliminar checks
322 _reductionFactor=reductionFactor;
323 // compute image output size
324 _outputNBrows=(unsigned int)((double)minDimension/reductionFactor);
325 _outputNBcolumns=(unsigned int)((double)minDimension/reductionFactor);
326 _outputNBpixels=_outputNBrows*_outputNBcolumns;
327 _outputDoubleNBpixels=_outputNBrows*_outputNBcolumns*2;
328
329 // get half frame size
330 //unsigned int halfOutputRows = _outputNBrows/2-1;
331 //unsigned int halfOutputColumns = _outputNBcolumns/2-1;
332 unsigned int halfInputRows = _filterOutput.getNBrows()/2-1;
333 unsigned int halfInputColumns = _filterOutput.getNBcolumns()/2-1;
334
335
336#ifdef IMAGELOGPOLPROJECTION_DEBUG
337 std::cout<<"ImageLogPolProjection::Log resampled image size: "<<_outputNBrows<<"*"<<_outputNBcolumns<<std::endl;
338#endif
339
340 // setup progressive prefilter that will be applied BEFORE log sampling
341 setProgressiveFilterConstants_CentredAccuracy(0.f, 0.f, 0.99f);
342
343 // (re)create the image output buffer and transform table if the reduction factor changed
344 _sampledFrame.resize(_outputNBpixels*(1+(unsigned int)_colorModeCapable*2));
345
346 // create the radius and orientation axis and fill them, radius E [0;1], orientation E[-pi, pi]
347 std::valarray<double> radiusAxis(_outputNBcolumns);
348 double radiusStep=2.30/(double)_outputNBcolumns;
349 for (unsigned int i=0;i<_outputNBcolumns;++i)
350 {
351 radiusAxis[i]=i*radiusStep;
352 }
353 std::valarray<double> orientationAxis(_outputNBrows);
354 double orientationStep=-2.0*CV_PI/(double)_outputNBrows;
355 for (unsigned int io=0;io<_outputNBrows;++io)
356 {
357 orientationAxis[io]=io*orientationStep;
358 }
359 // -> use a temporay transform table which is bigger than the final one, we only report pixels coordinates that are included in the sampled picture
360 std::valarray<unsigned int> tempTransformTable(2*_outputNBpixels); // the structure would be: (pixelInputCoordinate n)(pixelOutputCoordinate n)(pixelInputCoordinate n+1)(pixelOutputCoordinate n+1)
361 _usefullpixelIndex=0;
362
363 //std::cout<<"ImageLogPolProjection::Starting cortex projection"<<std::endl;
364 // compute transformation, get theta and Radius in reagrd of the output sampled pixel
365 double diagonalLength=sqrt((double)(_outputNBcolumns*_outputNBcolumns+_outputNBrows*_outputNBrows));
366 for (unsigned int radiusIndex=0;radiusIndex<_outputNBcolumns;++radiusIndex)
367 for(unsigned int orientationIndex=0;orientationIndex<_outputNBrows;++orientationIndex)
368 {
369 double x=1.0+sinh(radiusAxis[radiusIndex])*cos(orientationAxis[orientationIndex]);
370 double y=sinh(radiusAxis[radiusIndex])*sin(orientationAxis[orientationIndex]);
371 // get the input picture coordinate
372 double R=diagonalLength*sqrt(x*x+y*y)/(5.0+sqrt(x*x+y*y));
373 double theta=atan2(y,x);
374 // convert input polar coord into cartesian/C compatble coordinate
375 unsigned int columnIndex=(unsigned int)(cos(theta)*R)+halfInputColumns;
376 unsigned int rowIndex=(unsigned int)(sin(theta)*R)+halfInputRows;
377 //std::cout<<"ImageLogPolProjection::R="<<R<<" / Theta="<<theta<<" / (x, y)="<<columnIndex<<", "<<rowIndex<<std::endl;
378 if ((columnIndex<_filterOutput.getNBcolumns())&&(columnIndex>0)&&(rowIndex<_filterOutput.getNBrows())&&(rowIndex>0))
379 {
380 // set coordinate
381 tempTransformTable[_usefullpixelIndex++]=radiusIndex+orientationIndex*_outputNBcolumns;
382 tempTransformTable[_usefullpixelIndex++]= columnIndex+rowIndex*_filterOutput.getNBcolumns();
383 }
384 }
385
386 // (re)creating and filling the transform table
387 _transformTable.resize(_usefullpixelIndex);
388 memcpy(&_transformTable[0], &tempTransformTable[0], sizeof(unsigned int)*_usefullpixelIndex);
389
390 // reset all buffers
391 clearAllBuffers();
392 _initOK=true;
393 return true;
394}

396// action function
397std::valarray<float> &ImageLogPolProjection::runProjection(const std::valarray<float> &inputFrame, const bool colorMode)
398{
399 if (_colorModeCapable&&colorMode)
400 {
401 // progressive filtering and storage of the result in _tempBuffer
402 _spatiotemporalLPfilter_Irregular(get_data(inputFrame), &_irregularLPfilteredFrame[0]);
403 _spatiotemporalLPfilter_Irregular(&_irregularLPfilteredFrame[0], &_tempBuffer[0]); // warning, temporal issue may occur, if the temporal constant is not NULL !!!
404
405 _spatiotemporalLPfilter_Irregular(get_data(inputFrame)+_filterOutput.getNBpixels(), &_irregularLPfilteredFrame[0]);
406 _spatiotemporalLPfilter_Irregular(&_irregularLPfilteredFrame[0], &_tempBuffer[0]+_filterOutput.getNBpixels());
407
408 _spatiotemporalLPfilter_Irregular(get_data(inputFrame)+_filterOutput.getNBpixels()*2, &_irregularLPfilteredFrame[0]);
409 _spatiotemporalLPfilter_Irregular(&_irregularLPfilteredFrame[0], &_tempBuffer[0]+_filterOutput.getNBpixels()*2);
410
411 // applying image projection/resampling
412 unsigned int *transformTablePTR=&_transformTable[0];
413 for (unsigned int i=0 ; i<_usefullpixelIndex ; i+=2, transformTablePTR+=2)
414 {
415#ifdef IMAGELOGPOLPROJECTION_DEBUG
416 std::cout<<"ImageLogPolProjection::i:"<<i<<"output(max="<<_outputNBpixels<<")="<<_transformTable[i]<<" / intput(max="<<_filterOutput.getNBpixels()<<")="<<_transformTable[i+1]<<std::endl;
417#endif
418 _sampledFrame[*(transformTablePTR)]=_tempBuffer[*(transformTablePTR+1)];
419 _sampledFrame[*(transformTablePTR)+_outputNBpixels]=_tempBuffer[*(transformTablePTR+1)+_filterOutput.getNBpixels()];
420 _sampledFrame[*(transformTablePTR)+_outputDoubleNBpixels]=_tempBuffer[*(transformTablePTR+1)+_inputDoubleNBpixels];
421 }
422
423#ifdef IMAGELOGPOLPROJECTION_DEBUG
424 std::cout<<"ImageLogPolProjection::runProjection: color image projection OK"<<std::endl;
425#endif
426 //normalizeGrayOutput_0_maxOutputValue(_sampledFrame, _outputNBpixels);
427 }else
428 {
429 _spatiotemporalLPfilter_Irregular(get_data(inputFrame), &_irregularLPfilteredFrame[0]);
430 _spatiotemporalLPfilter_Irregular(&_irregularLPfilteredFrame[0], &_irregularLPfilteredFrame[0]);
431 // applying image projection/resampling
432 unsigned int *transformTablePTR=&_transformTable[0];
433 for (unsigned int i=0 ; i<_usefullpixelIndex ; i+=2, transformTablePTR+=2)
434 {
435#ifdef IMAGELOGPOLPROJECTION_DEBUG
436 std::cout<<"i:"<<i<<"output(max="<<_outputNBpixels<<")="<<_transformTable[i]<<" / intput(max="<<_filterOutput.getNBpixels()<<")="<<_transformTable[i+1]<<std::endl;
437#endif
438 _sampledFrame[*(transformTablePTR)]=_irregularLPfilteredFrame[*(transformTablePTR+1)];
439 }
440 //normalizeGrayOutput_0_maxOutputValue(_sampledFrame, _outputNBpixels);
441#ifdef IMAGELOGPOLPROJECTION_DEBUG
442 std::cout<<"ImageLogPolProjection::runProjection: gray level image projection OK"<<std::endl;
443#endif
444 }
445
446 return _sampledFrame;
447}
