83inline float CvHybridTracker::getL2Norm(Point2f p1, Point2f p2) {
84 float distance = (p1.x - p2.x) * (p1.x - p2.x) + (p1.y - p2.y) * (p1.y
85 - p2.y);
86 return sqrt(distance);
87}

89Mat CvHybridTracker::getDistanceProjection(Mat image, Point2f center) {
90 Mat hist(image.size(), CV_64F);
91
92 double lu = getL2Norm(Point(0, 0), center);
93 double ru = getL2Norm(Point(0, image.size().width), center);
94 double rd = getL2Norm(Point(image.size().height, image.size().width),
95 center);
96 double ld = getL2Norm(Point(image.size().height, 0), center);
97
98 double max = (lu < ru) ? lu : ru;
99 max = (max < rd) ? max : rd;
100 max = (max < ld) ? max : ld;
101
102 for (int i = 0; i < hist.rows; i++)
103 for (int j = 0; j < hist.cols; j++)
104 hist.at<double> (i, j) = 1.0 - (getL2Norm(Point(i, j), center)
105 / max);
106
107 return hist;
108}

110Mat CvHybridTracker::getGaussianProjection(Mat image, int ksize, double sigma,
111 Point2f center) {
112 Mat kernel = getGaussianKernel(ksize, sigma, CV_64F);
113 double max = kernel.at<double> (ksize / 2);
114
115 Mat hist(image.size(), CV_64F);
116 for (int i = 0; i < hist.rows; i++)
117 for (int j = 0; j < hist.cols; j++) {
118 int pos = cvRound(getL2Norm(Point(i, j), center));
119 if (pos < ksize / 2.0)
120 hist.at<double> (i, j) = 1.0 - (kernel.at<double> (pos) / max);
121 }
122
123 return hist;
124}

126void CvHybridTracker::newTracker(Mat image, Rect selection) {
127 prev_proj = Mat::zeros(image.size(), CV_64FC1);
128 prev_center = Point2f(selection.x + selection.width / 2.0f, selection.y
129 + selection.height / 2.0f);
130 prev_window = selection;
131
132 mstracker->newTrackingWindow(image, selection);
133 fttracker->newTrackingWindow(image, selection);
134
135 samples = cvCreateMat(2, 1, CV_32FC1);
136 labels = cvCreateMat(2, 1, CV_32SC1);
137
138 ittr = 0;
139}

141void CvHybridTracker::updateTracker(Mat image) {
142 ittr++;
143
144 //copy over clean images: TODO
145 mstracker->updateTrackingWindow(image);
146 fttracker->updateTrackingWindowWithFlow(image);
147
148 if (params.motion_model == CvMotionModel::EM)
149 updateTrackerWithEM(image);
150 else
151 updateTrackerWithLowPassFilter(image);
152
153 // Regression to find new weights
154 Point2f ms_center = mstracker->getTrackingEllipse().center;
155 Point2f ft_center = fttracker->getTrackingCenter();
156
157#ifdef DEBUG_HYTRACKER
158 circle(image, ms_center, 3, Scalar(0, 0, 255), -1, 8);
159 circle(image, ft_center, 3, Scalar(255, 0, 0), -1, 8);
160 putText(image, "ms", Point(ms_center.x+2, ms_center.y), FONT_HERSHEY_PLAIN, 0.75, Scalar(255, 255, 255));
161 putText(image, "ft", Point(ft_center.x+2, ft_center.y), FONT_HERSHEY_PLAIN, 0.75, Scalar(255, 255, 255));
162#endif
163
164 double ms_len = getL2Norm(ms_center, curr_center);
165 double ft_len = getL2Norm(ft_center, curr_center);
166 double total_len = ms_len + ft_len;
167
168 params.ms_tracker_weight *= (ittr - 1);
169 params.ms_tracker_weight += (float)((ms_len / total_len));
170 params.ms_tracker_weight /= ittr;
171 params.ft_tracker_weight *= (ittr - 1);
172 params.ft_tracker_weight += (float)((ft_len / total_len));
173 params.ft_tracker_weight /= ittr;
174
175 circle(image, prev_center, 3, Scalar(0, 0, 0), -1, 8);
176 circle(image, curr_center, 3, Scalar(255, 255, 255), -1, 8);
177
178 prev_center = curr_center;
179 prev_window.x = (int)(curr_center.x-prev_window.width/2.0);
180 prev_window.y = (int)(curr_center.y-prev_window.height/2.0);
181
182 mstracker->setTrackingWindow(prev_window);
183 fttracker->setTrackingWindow(prev_window);
184}

186void CvHybridTracker::updateTrackerWithEM(Mat image) {
187 Mat ms_backproj = mstracker->getHistogramProjection(CV_64F);
188 Mat ms_distproj = getDistanceProjection(image, mstracker->getTrackingCenter());
189 Mat ms_proj = ms_backproj.mul(ms_distproj);
190
191 float dist_err = getL2Norm(mstracker->getTrackingCenter(), fttracker->getTrackingCenter());
192 Mat ft_gaussproj = getGaussianProjection(image, cvRound(dist_err), -1, fttracker->getTrackingCenter());
193 Mat ft_distproj = getDistanceProjection(image, fttracker->getTrackingCenter());
194 Mat ft_proj = ft_gaussproj.mul(ft_distproj);
195
196 Mat proj = params.ms_tracker_weight * ms_proj + params.ft_tracker_weight * ft_proj + prev_proj;
197
198 int sample_count = countNonZero(proj);
199 cvReleaseMat(&samples);
200 cvReleaseMat(&labels);
201 samples = cvCreateMat(sample_count, 2, CV_32FC1);
202 labels = cvCreateMat(sample_count, 1, CV_32SC1);
203
204 int count = 0;
205 for (int i = 0; i < proj.rows; i++)
206 for (int j = 0; j < proj.cols; j++)
207 if (proj.at<double> (i, j) > 0) {
208 samples->data.fl[count * 2] = (float)i;
209 samples->data.fl[count * 2 + 1] = (float)j;
210 count++;
211 }
212
213 cv::Mat lbls;
214
215 EM em_model(1, EM::COV_MAT_SPHERICAL, TermCriteria(TermCriteria::COUNT + TermCriteria::EPS, 10000, 0.001));
216 em_model.train(cvarrToMat(samples), noArray(), lbls);
217 if(labels)
218 lbls.copyTo(cvarrToMat(labels));
219
220 Mat em_means = em_model.get<Mat>("means");
221 curr_center.x = (float)em_means.at<float>(0, 0);
222 curr_center.y = (float)em_means.at<float>(0, 1);
223}

225void CvHybridTracker::updateTrackerWithLowPassFilter(Mat) {
226 RotatedRect ms_track = mstracker->getTrackingEllipse();
227 Point2f ft_center = fttracker->getTrackingCenter();
228
229 float a = params.low_pass_gain;
230 curr_center.x = (1 - a) * prev_center.x + a * (params.ms_tracker_weight * ms_track.center.x + params.ft_tracker_weight * ft_center.x);
231 curr_center.y = (1 - a) * prev_center.y + a * (params.ms_tracker_weight * ms_track.center.y + params.ft_tracker_weight * ft_center.y);
232}
