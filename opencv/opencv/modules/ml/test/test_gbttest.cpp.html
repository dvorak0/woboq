48CV_GBTreesTest::CV_GBTreesTest()
49{
50 int64 seeds[] = { CV_BIG_INT(0x00009fff4f9c8d52),
51 CV_BIG_INT(0x0000a17166072c7c),
52 CV_BIG_INT(0x0201b32115cd1f9a),
53 CV_BIG_INT(0x0513cb37abcd1234),
54 CV_BIG_INT(0x0001a2b3c4d5f678)
55 };
56
57 int seedCount = sizeof(seeds)/sizeof(seeds[0]);
58 cv::RNG& rng = cv::theRNG();
59 initSeed = rng.state;
60 rng.state = seeds[rng(seedCount)];
61
62 datasets = 0;
63 data = 0;
64 gtb = 0;
65}

67CV_GBTreesTest::~CV_GBTreesTest()

69 if (data)

72 cv::theRNG().state = initSeed;

76int CV_GBTreesTest::TestTrainPredict(int test_num)
77{
78 int code = cvtest::TS::OK;
79
80 int weak_count = 200;
81 float shrinkage = 0.1f;
82 float subsample_portion = 0.5f;
83 int max_depth = 5;
84 bool use_surrogates = false;
85 int loss_function_type = 0;
86 switch (test_num)
87 {
88 case (1) : loss_function_type = CvGBTrees::SQUARED_LOSS; break;
89 case (2) : loss_function_type = CvGBTrees::ABSOLUTE_LOSS; break;
90 case (3) : loss_function_type = CvGBTrees::HUBER_LOSS; break;
91 case (0) : loss_function_type = CvGBTrees::DEVIANCE_LOSS; break;
92 default :
93 {
94 ts->printf( cvtest::TS::LOG, "Bad test_num value in CV_GBTreesTest::TestTrainPredict(..) function." );
95 return cvtest::TS::FAIL_BAD_ARG_CHECK;
96 }
97 }
98
99 int dataset_num = test_num == 0 ? 0 : 1;
100 if (!data)
101 {
102 data = new CvMLData();
103 data->set_delimiter(',');
104
105 if (data->read_csv(datasets[dataset_num].c_str()))
106 {
107 ts->printf( cvtest::TS::LOG, "File reading error." );
108 return cvtest::TS::FAIL_INVALID_TEST_DATA;
109 }
110
111 if (test_num == 0)
112 {
113 data->set_response_idx(57);
114 data->set_var_types("ord[0-56],cat[57]");
115 }
116 else
117 {
118 data->set_response_idx(13);
119 data->set_var_types("ord[0-2,4-13],cat[3]");
120 subsample_portion = 0.7f;
121 }
122
123 int train_sample_count = cvFloor(_get_len(data->get_responses())*0.5f);
124 CvTrainTestSplit spl( train_sample_count );
125 data->set_train_test_split( &spl );
126 }
127
128 data->mix_train_and_test_idx();
129
130
131 if (gtb) delete gtb;
132 gtb = new CvGBTrees();
133 bool tmp_code = true;
134 tmp_code = gtb->train(data, CvGBTreesParams(loss_function_type, weak_count,
135 shrinkage, subsample_portion,
136 max_depth, use_surrogates));
137
138 if (!tmp_code)
139 {
140 ts->printf( cvtest::TS::LOG, "Model training was failed.");
141 return cvtest::TS::FAIL_INVALID_OUTPUT;
142 }
143
144 code = checkPredictError(test_num);
145
146 return code;
147
148}

151int CV_GBTreesTest::checkPredictError(int test_num)
152{
153 if (!gtb)
154 return cvtest::TS::FAIL_GENERIC;
155
156 //float mean[] = {5.430247f, 13.5654f, 12.6569f, 13.1661f};
157 //float sigma[] = {0.4162694f, 3.21161f, 3.43297f, 3.00624f};
158 float mean[] = {5.80226f, 12.68689f, 13.49095f, 13.19628f};
159 float sigma[] = {0.4764534f, 3.166919f, 3.022405f, 2.868722f};
160
161 float current_error = gtb->calc_error(data, CV_TEST_ERROR);
162
163 if ( abs( current_error - mean[test_num]) > 6*sigma[test_num] )
164 {
165 ts->printf( cvtest::TS::LOG, "Test error is out of range:\n"
166 "abs(%f/*curEr*/ - %f/*mean*/ > %f/*6*sigma*/",
167 current_error, mean[test_num], 6*sigma[test_num] );
168 return cvtest::TS::FAIL_BAD_ACCURACY;
169 }
170
171 return cvtest::TS::OK;
172
173}

176int CV_GBTreesTest::TestSaveLoad()
177{
178 if (!gtb)
179 return cvtest::TS::FAIL_GENERIC;
180
181 model_file_name1 = cv::tempfile();
182 model_file_name2 = cv::tempfile();
183
184 gtb->save(model_file_name1.c_str());
185 gtb->calc_error(data, CV_TEST_ERROR, &test_resps1);
186 gtb->load(model_file_name1.c_str());
187 gtb->calc_error(data, CV_TEST_ERROR, &test_resps2);
188 gtb->save(model_file_name2.c_str());
189
190 return checkLoadSave();
191
192}

196int CV_GBTreesTest::checkLoadSave()
197{
198 int code = cvtest::TS::OK;
199
200 // 1. compare files
201 ifstream f1( model_file_name1.c_str() ), f2( model_file_name2.c_str() );
202 string s1, s2;
203 int lineIdx = 0;
204 CV_Assert( f1.is_open() && f2.is_open() );
205 for( ; !f1.eof() && !f2.eof(); lineIdx++ )
206 {
207 getline( f1, s1 );
208 getline( f2, s2 );
209 if( s1.compare(s2) )
210 {
211 ts->printf( cvtest::TS::LOG, "first and second saved files differ in %n-line; first %n line: %s; second %n-line: %s",
212 lineIdx, lineIdx, s1.c_str(), lineIdx, s2.c_str() );
213 code = cvtest::TS::FAIL_INVALID_OUTPUT;
214 }
215 }
216 if( !f1.eof() || !f2.eof() )
217 {
218 ts->printf( cvtest::TS::LOG, "First and second saved files differ in %n-line; first %n line: %s; second %n-line: %s",
219 lineIdx, lineIdx, s1.c_str(), lineIdx, s2.c_str() );
220 code = cvtest::TS::FAIL_INVALID_OUTPUT;
221 }
222 f1.close();
223 f2.close();
224 // delete temporary files
225 remove( model_file_name1.c_str() );
226 remove( model_file_name2.c_str() );
227
228 // 2. compare responses
229 CV_Assert( test_resps1.size() == test_resps2.size() );
230 vector<float>::const_iterator it1 = test_resps1.begin(), it2 = test_resps2.begin();
231 for( ; it1 != test_resps1.end(); ++it1, ++it2 )
232 {
233 if( fabs(*it1 - *it2) > FLT_EPSILON )
234 {
235 ts->printf( cvtest::TS::LOG, "Responses predicted before saving and after loading are different" );
236 code = cvtest::TS::FAIL_INVALID_OUTPUT;
237 }
238 }
239 return code;
240}

244void CV_GBTreesTest::run(int)
245{
246
247 string dataPath = string(ts->get_data_path());
248 datasets = new string[2];
249 datasets[0] = dataPath + string("spambase.data"); /*string("dataset_classification.csv");*/
250 datasets[1] = dataPath + string("housing_.data"); /*string("dataset_regression.csv");*/
251
252 int code = cvtest::TS::OK;
253
254 for (int i = 0; i < 4; i++)
255 {
256
257 int temp_code = TestTrainPredict(i);
258 if (temp_code != cvtest::TS::OK)
259 {
260 code = temp_code;
261 break;
262 }
263
264 else if (i==0)
265 {
266 temp_code = TestSaveLoad();
267 if (temp_code != cvtest::TS::OK)
268 code = temp_code;
269 delete data;
270 data = 0;
271 }
272
273 delete gtb;
274 gtb = 0;
275 }
276 delete data;
277 data = 0;
278
279 ts->set_failed_test_info( code );
280}
