48// 1. nbayes
49void nbayes_check_data( CvMLData* _data )
50{
51 if( _data->get_missing() )
52 CV_Error( CV_StsBadArg, "missing values are not supported" );
53 const CvMat* var_types = _data->get_var_types();
54 bool is_classifier = var_types->data.ptr[var_types->cols-1] == CV_VAR_CATEGORICAL;
55 if( ( fabs( cvNorm( var_types, 0, CV_L1 ) -
56 (var_types->rows + var_types->cols - 2)*CV_VAR_ORDERED - CV_VAR_CATEGORICAL ) > FLT_EPSILON ) ||
57 !is_classifier )
58 CV_Error( CV_StsBadArg, "incorrect types of predictors or responses" );
59}
60bool nbayes_train( CvNormalBayesClassifier* nbayes, CvMLData* _data )
61{
62 nbayes_check_data( _data );
63 const CvMat* values = _data->get_values();
64 const CvMat* responses = _data->get_responses();
65 const CvMat* train_sidx = _data->get_train_sample_idx();
66 const CvMat* var_idx = _data->get_var_idx();
67 return nbayes->train( values, responses, var_idx, train_sidx );
68}
69float nbayes_calc_error( CvNormalBayesClassifier* nbayes, CvMLData* _data, int type, vector<float> *resp )
70{
71 float err = 0;
72 nbayes_check_data( _data );
73 const CvMat* values = _data->get_values();
74 const CvMat* response = _data->get_responses();
75 const CvMat* sample_idx = (type == CV_TEST_ERROR) ? _data->get_test_sample_idx() : _data->get_train_sample_idx();
76 int* sidx = sample_idx ? sample_idx->data.i : 0;
77 int r_step = CV_IS_MAT_CONT(response->type) ?
78 1 : response->step / CV_ELEM_SIZE(response->type);
79 int sample_count = sample_idx ? sample_idx->cols : 0;
80 sample_count = (type == CV_TRAIN_ERROR && sample_count == 0) ? values->rows : sample_count;
81 float* pred_resp = 0;
82 if( resp && (sample_count > 0) )
83 {
84 resp->resize( sample_count );
85 pred_resp = &((*resp)[0]);
86 }
87
88 for( int i = 0; i < sample_count; i++ )
89 {
90 CvMat sample;
91 int si = sidx ? sidx[i] : i;
92 cvGetRow( values, &sample, si );
93 float r = (float)nbayes->predict( &sample, 0 );
94 if( pred_resp )
95 pred_resp[i] = r;
96 int d = fabs((double)r - response->data.fl[si*r_step]) <= FLT_EPSILON ? 0 : 1;
97 err += d;
98 }
99 err = sample_count ? err / (float)sample_count * 100 : -FLT_MAX;
100 return err;
101}

103// 2. knearest
104void knearest_check_data_and_get_predictors( CvMLData* _data, CvMat* _predictors )
105{
106 const CvMat* values = _data->get_values();
107 const CvMat* var_idx = _data->get_var_idx();
108 if( var_idx->cols + var_idx->rows != values->cols )
109 CV_Error( CV_StsBadArg, "var_idx is not supported" );
110 if( _data->get_missing() )
111 CV_Error( CV_StsBadArg, "missing values are not supported" );
112 int resp_idx = _data->get_response_idx();
113 if( resp_idx == 0)
114 cvGetCols( values, _predictors, 1, values->cols );
115 else if( resp_idx == values->cols - 1 )
116 cvGetCols( values, _predictors, 0, values->cols - 1 );
117 else
118 CV_Error( CV_StsBadArg, "responses must be in the first or last column; other cases are not supported" );
119}
120bool knearest_train( CvKNearest* knearest, CvMLData* _data )
121{
122 const CvMat* responses = _data->get_responses();
123 const CvMat* train_sidx = _data->get_train_sample_idx();
124 bool is_regression = _data->get_var_type( _data->get_response_idx() ) == CV_VAR_ORDERED;
125 CvMat predictors;
126 knearest_check_data_and_get_predictors( _data, &predictors );
127 return knearest->train( &predictors, responses, train_sidx, is_regression );
128}
129float knearest_calc_error( CvKNearest* knearest, CvMLData* _data, int k, int type, vector<float> *resp )
130{
131 float err = 0;
132 const CvMat* response = _data->get_responses();
133 const CvMat* sample_idx = (type == CV_TEST_ERROR) ? _data->get_test_sample_idx() : _data->get_train_sample_idx();
134 int* sidx = sample_idx ? sample_idx->data.i : 0;
135 int r_step = CV_IS_MAT_CONT(response->type) ?
136 1 : response->step / CV_ELEM_SIZE(response->type);
137 bool is_regression = _data->get_var_type( _data->get_response_idx() ) == CV_VAR_ORDERED;
138 CvMat predictors;
139 knearest_check_data_and_get_predictors( _data, &predictors );
140 int sample_count = sample_idx ? sample_idx->cols : 0;
141 sample_count = (type == CV_TRAIN_ERROR && sample_count == 0) ? predictors.rows : sample_count;
142 float* pred_resp = 0;
143 if( resp && (sample_count > 0) )
144 {
145 resp->resize( sample_count );
146 pred_resp = &((*resp)[0]);
147 }
148 if ( !is_regression )
149 {
150 for( int i = 0; i < sample_count; i++ )
151 {
152 CvMat sample;
153 int si = sidx ? sidx[i] : i;
154 cvGetRow( &predictors, &sample, si );
155 float r = knearest->find_nearest( &sample, k );
156 if( pred_resp )
157 pred_resp[i] = r;
158 int d = fabs((double)r - response->data.fl[si*r_step]) <= FLT_EPSILON ? 0 : 1;
159 err += d;
160 }
161 err = sample_count ? err / (float)sample_count * 100 : -FLT_MAX;
162 }
163 else
164 {
165 for( int i = 0; i < sample_count; i++ )
166 {
167 CvMat sample;
168 int si = sidx ? sidx[i] : i;
169 cvGetRow( &predictors, &sample, si );
170 float r = knearest->find_nearest( &sample, k );
171 if( pred_resp )
172 pred_resp[i] = r;
173 float d = r - response->data.fl[si*r_step];
174 err += d*d;
175 }
176 err = sample_count ? err / (float)sample_count : -FLT_MAX;
177 }
178 return err;
179}

309void ann_check_data_and_get_predictors( CvMLData* _data, CvMat* _inputs )
310{
311 const CvMat* values = _data->get_values();
312 const CvMat* var_idx = _data->get_var_idx();
313 if( var_idx->cols + var_idx->rows != values->cols )
314 CV_Error( CV_StsBadArg, "var_idx is not supported" );
315 if( _data->get_missing() )
316 CV_Error( CV_StsBadArg, "missing values are not supported" );
317 int resp_idx = _data->get_response_idx();
318 if( resp_idx == 0)
319 cvGetCols( values, _inputs, 1, values->cols );
320 else if( resp_idx == values->cols - 1 )
321 cvGetCols( values, _inputs, 0, values->cols - 1 );
322 else
323 CV_Error( CV_StsBadArg, "outputs must be in the first or last column; other cases are not supported" );
324}
325void ann_get_new_responses( CvMLData* _data, Mat& new_responses, map<int, int>& cls_map )
326{
327 const CvMat* train_sidx = _data->get_train_sample_idx();
328 int* train_sidx_ptr = train_sidx->data.i;
329 const CvMat* responses = _data->get_responses();
330 float* responses_ptr = responses->data.fl;
331 int r_step = CV_IS_MAT_CONT(responses->type) ?
332 1 : responses->step / CV_ELEM_SIZE(responses->type);
333 int cls_count = 0;
334 // construct cls_map
335 cls_map.clear();
336 for( int si = 0; si < train_sidx->cols; si++ )
337 {
338 int sidx = train_sidx_ptr[si];
339 int r = cvRound(responses_ptr[sidx*r_step]);
340 CV_DbgAssert( fabs(responses_ptr[sidx*r_step]-r) < FLT_EPSILON );
341 int cls_map_size = (int)cls_map.size();
342 cls_map[r];
343 if ( (int)cls_map.size() > cls_map_size )
344 cls_map[r] = cls_count++;
345 }
346 new_responses.create( responses->rows, cls_count, CV_32F );
347 new_responses.setTo( 0 );
348 for( int si = 0; si < train_sidx->cols; si++ )
349 {
350 int sidx = train_sidx_ptr[si];
351 int r = cvRound(responses_ptr[sidx*r_step]);
352 int cidx = cls_map[r];
353 new_responses.ptr<float>(sidx)[cidx] = 1;
354 }
355}
356int ann_train( CvANN_MLP* ann, CvMLData* _data, Mat& new_responses, CvANN_MLP_TrainParams _params, int flags = 0 )
357{
358 const CvMat* train_sidx = _data->get_train_sample_idx();
359 CvMat predictors;
360 ann_check_data_and_get_predictors( _data, &predictors );
361 CvMat _new_responses = CvMat( new_responses );
362 return ann->train( &predictors, &_new_responses, 0, train_sidx, _params, flags );
363}
364float ann_calc_error( CvANN_MLP* ann, CvMLData* _data, map<int, int>& cls_map, int type , vector<float> *resp_labels )
365{
366 float err = 0;
367 const CvMat* responses = _data->get_responses();
368 const CvMat* sample_idx = (type == CV_TEST_ERROR) ? _data->get_test_sample_idx() : _data->get_train_sample_idx();
369 int* sidx = sample_idx ? sample_idx->data.i : 0;
370 int r_step = CV_IS_MAT_CONT(responses->type) ?
371 1 : responses->step / CV_ELEM_SIZE(responses->type);
372 CvMat predictors;
373 ann_check_data_and_get_predictors( _data, &predictors );
374 int sample_count = sample_idx ? sample_idx->cols : 0;
375 sample_count = (type == CV_TRAIN_ERROR && sample_count == 0) ? predictors.rows : sample_count;
376 float* pred_resp = 0;
377 vector<float> innresp;
378 if( sample_count > 0 )
379 {
380 if( resp_labels )
381 {
382 resp_labels->resize( sample_count );
383 pred_resp = &((*resp_labels)[0]);
384 }
385 else
386 {
387 innresp.resize( sample_count );
388 pred_resp = &(innresp[0]);
389 }
390 }
391 int cls_count = (int)cls_map.size();
392 Mat output( 1, cls_count, CV_32FC1 );
393 CvMat _output = CvMat(output);
394 for( int i = 0; i < sample_count; i++ )
395 {
396 CvMat sample;
397 int si = sidx ? sidx[i] : i;
398 cvGetRow( &predictors, &sample, si );
399 ann->predict( &sample, &_output );
400 CvPoint best_cls = {0,0};
401 cvMinMaxLoc( &_output, 0, 0, 0, &best_cls, 0 );
402 int r = cvRound(responses->data.fl[si*r_step]);
403 CV_DbgAssert( fabs(responses->data.fl[si*r_step]-r) < FLT_EPSILON );
404 r = cls_map[r];
405 int d = best_cls.x == r ? 0 : 1;
406 err += d;
407 pred_resp[i] = (float)best_cls.x;
408 }
409 err = sample_count ? err / (float)sample_count * 100 : -FLT_MAX;
410 return err;
411}

434CV_MLBaseTest::CV_MLBaseTest(const char* _modelName)
435{
436 int64 seeds[] = { CV_BIG_INT(0x00009fff4f9c8d52),
437 CV_BIG_INT(0x0000a17166072c7c),
438 CV_BIG_INT(0x0201b32115cd1f9a),
439 CV_BIG_INT(0x0513cb37abcd1234),
440 CV_BIG_INT(0x0001a2b3c4d5f678)
441 };
442
443 int seedCount = sizeof(seeds)/sizeof(seeds[0]);
444 RNG& rng = theRNG();
445
446 initSeed = rng.state;
447
448 rng.state = seeds[rng(seedCount)];
449
450 modelName = _modelName;
451 nbayes = 0;
452 knearest = 0;
453 svm = 0;
454 ann = 0;
455 dtree = 0;
456 boost = 0;
457 rtrees = 0;
458 ertrees = 0;
459 if( !modelName.compare(CV_NBAYES) )
460 nbayes = new CvNormalBayesClassifier;
461 else if( !modelName.compare(CV_KNEAREST) )
462 knearest = new CvKNearest;
463 else if( !modelName.compare(CV_SVM) )
464 svm = new CvSVM;
465 else if( !modelName.compare(CV_ANN) )
466 ann = new CvANN_MLP;
467 else if( !modelName.compare(CV_DTREE) )
468 dtree = new CvDTree;
469 else if( !modelName.compare(CV_BOOST) )
470 boost = new CvBoost;
471 else if( !modelName.compare(CV_RTREES) )
472 rtrees = new CvRTrees;
473 else if( !modelName.compare(CV_ERTREES) )
474 ertrees = new CvERTrees;
475}

500int CV_MLBaseTest::read_params( CvFileStorage* _fs )
501{
502 if( !_fs )
503 test_case_count = -1;
504 else
505 {
506 CvFileNode* fn = cvGetRootFileNode( _fs, 0 );
507 fn = (CvFileNode*)cvGetSeqElem( fn->data.seq, 0 );
508 fn = cvGetFileNodeByName( _fs, fn, "run_params" );
509 CvSeq* dataSetNamesSeq = cvGetFileNodeByName( _fs, fn, modelName.c_str() )->data.seq;
510 test_case_count = dataSetNamesSeq ? dataSetNamesSeq->total : -1;
511 if( test_case_count > 0 )
512 {
513 dataSetNames.resize( test_case_count );
514 vector<string>::iterator it = dataSetNames.begin();
515 for( int i = 0; i < test_case_count; i++, it++ )
516 *it = ((CvFileNode*)cvGetSeqElem( dataSetNamesSeq, i ))->data.str.ptr;
517 }
518 }
519 return cvtest::TS::OK;;
520}

522void CV_MLBaseTest::run( int )
523{
524 string filename = ts->get_data_path();
525 filename += get_validation_filename();
526 validationFS.open( filename, FileStorage::READ );
527 read_params( *validationFS );
528
529 int code = cvtest::TS::OK;
530 for (int i = 0; i < test_case_count; i++)
531 {
532 int temp_code = run_test_case( i );
533 if (temp_code == cvtest::TS::OK)
534 temp_code = validate_test_results( i );
535 if (temp_code != cvtest::TS::OK)
536 code = temp_code;
537 }
538 if ( test_case_count <= 0)
539 {
540 ts->printf( cvtest::TS::LOG, "validation file is not determined or not correct" );
541 code = cvtest::TS::FAIL_INVALID_TEST_DATA;
542 }
543 ts->set_failed_test_info( code );
544}
545
546int CV_MLBaseTest::prepare_test_case( int test_case_idx )
547{
548 int trainSampleCount, respIdx;
549 string varTypes;
550 clear();
551
552 string dataPath = ts->get_data_path();
553 if ( dataPath.empty() )
554 {
555 ts->printf( cvtest::TS::LOG, "data path is empty" );
556 return cvtest::TS::FAIL_INVALID_TEST_DATA;
557 }
558
559 string dataName = dataSetNames[test_case_idx],
560 filename = dataPath + dataName + ".data";
561 if ( data.read_csv( filename.c_str() ) != 0)
562 {
563 char msg[100];
564 sprintf( msg, "file %s can not be read", filename.c_str() );
565 ts->printf( cvtest::TS::LOG, msg );
566 return cvtest::TS::FAIL_INVALID_TEST_DATA;
567 }
568
569 FileNode dataParamsNode = validationFS.getFirstTopLevelNode()["validation"][modelName][dataName]["data_params"];
570 CV_DbgAssert( !dataParamsNode.empty() );
571
572 CV_DbgAssert( !dataParamsNode["LS"].empty() );
573 dataParamsNode["LS"] >> trainSampleCount;
574 CvTrainTestSplit spl( trainSampleCount );
575 data.set_train_test_split( &spl );
576
577 CV_DbgAssert( !dataParamsNode["resp_idx"].empty() );
578 dataParamsNode["resp_idx"] >> respIdx;

580
581 CV_DbgAssert( !dataParamsNode["types"].empty() );
582 dataParamsNode["types"] >> varTypes;
583 data.set_var_types( varTypes.c_str() );
584
585 return cvtest::TS::OK;

593int CV_MLBaseTest::train( int testCaseIdx )
594{
595 bool is_trained = false;
596 FileNode modelParamsNode =
597 validationFS.getFirstTopLevelNode()["validation"][modelName][dataSetNames[testCaseIdx]]["model_params"];
598
599 if( !modelName.compare(CV_NBAYES) )
600 is_trained = nbayes_train( nbayes, &data );
601 else if( !modelName.compare(CV_KNEAREST) )
602 {
603 assert( 0 );
604 //is_trained = knearest->train( &data );
605 }
606 else if( !modelName.compare(CV_SVM) )
607 {
608 string svm_type_str, kernel_type_str;
609 modelParamsNode["svm_type"] >> svm_type_str;
610 modelParamsNode["kernel_type"] >> kernel_type_str;
611 CvSVMParams params;
612 params.svm_type = str_to_svm_type( svm_type_str );
613 params.kernel_type = str_to_svm_kernel_type( kernel_type_str );
614 modelParamsNode["degree"] >> params.degree;
615 modelParamsNode["gamma"] >> params.gamma;
616 modelParamsNode["coef0"] >> params.coef0;
617 modelParamsNode["C"] >> params.C;
618 modelParamsNode["nu"] >> params.nu;
619 modelParamsNode["p"] >> params.p;
620 is_trained = svm_train( svm, &data, params );
621 }
622 else if( !modelName.compare(CV_EM) )
623 {
624 assert( 0 );
625 }
626 else if( !modelName.compare(CV_ANN) )
627 {
628 string train_method_str;
629 double param1, param2;
630 modelParamsNode["train_method"] >> train_method_str;
631 modelParamsNode["param1"] >> param1;
632 modelParamsNode["param2"] >> param2;
633 Mat new_responses;
634 ann_get_new_responses( &data, new_responses, cls_map );
635 int layer_sz[] = { data.get_values()->cols - 1, 100, 100, (int)cls_map.size() };
636 CvMat layer_sizes =
637 cvMat( 1, (int)(sizeof(layer_sz)/sizeof(layer_sz[0])), CV_32S, layer_sz );
638 ann->create( &layer_sizes );
639 is_trained = ann_train( ann, &data, new_responses, CvANN_MLP_TrainParams(cvTermCriteria(CV_TERMCRIT_ITER,300,0.01),
640 str_to_ann_train_method(train_method_str), param1, param2) ) >= 0;
641 }
642 else if( !modelName.compare(CV_DTREE) )
643 {
644 int MAX_DEPTH, MIN_SAMPLE_COUNT, MAX_CATEGORIES, CV_FOLDS;
645 float REG_ACCURACY = 0;
646 bool USE_SURROGATE, IS_PRUNED;
647 modelParamsNode["max_depth"] >> MAX_DEPTH;
648 modelParamsNode["min_sample_count"] >> MIN_SAMPLE_COUNT;
649 modelParamsNode["use_surrogate"] >> USE_SURROGATE;
650 modelParamsNode["max_categories"] >> MAX_CATEGORIES;
651 modelParamsNode["cv_folds"] >> CV_FOLDS;
652 modelParamsNode["is_pruned"] >> IS_PRUNED;
653 is_trained = dtree->train( &data,
654 CvDTreeParams(MAX_DEPTH, MIN_SAMPLE_COUNT, REG_ACCURACY, USE_SURROGATE,
655 MAX_CATEGORIES, CV_FOLDS, false, IS_PRUNED, 0 )) != 0;
656 }
657 else if( !modelName.compare(CV_BOOST) )
658 {
659 int BOOST_TYPE, WEAK_COUNT, MAX_DEPTH;
660 float WEIGHT_TRIM_RATE;
661 bool USE_SURROGATE;
662 string typeStr;
663 modelParamsNode["type"] >> typeStr;
664 BOOST_TYPE = str_to_boost_type( typeStr );
665 modelParamsNode["weak_count"] >> WEAK_COUNT;
666 modelParamsNode["weight_trim_rate"] >> WEIGHT_TRIM_RATE;
667 modelParamsNode["max_depth"] >> MAX_DEPTH;
668 modelParamsNode["use_surrogate"] >> USE_SURROGATE;
669 is_trained = boost->train( &data,
670 CvBoostParams(BOOST_TYPE, WEAK_COUNT, WEIGHT_TRIM_RATE, MAX_DEPTH, USE_SURROGATE, 0) ) != 0;
671 }
672 else if( !modelName.compare(CV_RTREES) )
673 {
674 int MAX_DEPTH, MIN_SAMPLE_COUNT, MAX_CATEGORIES, CV_FOLDS, NACTIVE_VARS, MAX_TREES_NUM;
675 float REG_ACCURACY = 0, OOB_EPS = 0.0;
676 bool USE_SURROGATE, IS_PRUNED;
677 modelParamsNode["max_depth"] >> MAX_DEPTH;
678 modelParamsNode["min_sample_count"] >> MIN_SAMPLE_COUNT;
679 modelParamsNode["use_surrogate"] >> USE_SURROGATE;
680 modelParamsNode["max_categories"] >> MAX_CATEGORIES;
681 modelParamsNode["cv_folds"] >> CV_FOLDS;
682 modelParamsNode["is_pruned"] >> IS_PRUNED;
683 modelParamsNode["nactive_vars"] >> NACTIVE_VARS;
684 modelParamsNode["max_trees_num"] >> MAX_TREES_NUM;
685 is_trained = rtrees->train( &data, CvRTParams( MAX_DEPTH, MIN_SAMPLE_COUNT, REG_ACCURACY,
686 USE_SURROGATE, MAX_CATEGORIES, 0, true, // (calc_var_importance == true) <=> RF processes variable importance
687 NACTIVE_VARS, MAX_TREES_NUM, OOB_EPS, CV_TERMCRIT_ITER)) != 0;
688 }
689 else if( !modelName.compare(CV_ERTREES) )
690 {
691 int MAX_DEPTH, MIN_SAMPLE_COUNT, MAX_CATEGORIES, CV_FOLDS, NACTIVE_VARS, MAX_TREES_NUM;
692 float REG_ACCURACY = 0, OOB_EPS = 0.0;
693 bool USE_SURROGATE, IS_PRUNED;
694 modelParamsNode["max_depth"] >> MAX_DEPTH;
695 modelParamsNode["min_sample_count"] >> MIN_SAMPLE_COUNT;
696 modelParamsNode["use_surrogate"] >> USE_SURROGATE;
697 modelParamsNode["max_categories"] >> MAX_CATEGORIES;
698 modelParamsNode["cv_folds"] >> CV_FOLDS;
699 modelParamsNode["is_pruned"] >> IS_PRUNED;
700 modelParamsNode["nactive_vars"] >> NACTIVE_VARS;
701 modelParamsNode["max_trees_num"] >> MAX_TREES_NUM;
702 is_trained = ertrees->train( &data, CvRTParams( MAX_DEPTH, MIN_SAMPLE_COUNT, REG_ACCURACY,
703 USE_SURROGATE, MAX_CATEGORIES, 0, false, // (calc_var_importance == true) <=> RF processes variable importance
704 NACTIVE_VARS, MAX_TREES_NUM, OOB_EPS, CV_TERMCRIT_ITER)) != 0;
705 }
706
707 if( !is_trained )
708 {
709 ts->printf( cvtest::TS::LOG, "in test case %d model training was failed", testCaseIdx );
710 return cvtest::TS::FAIL_INVALID_OUTPUT;
711 }
712 return cvtest::TS::OK;
713}
