61#define CV_MAT_ELEM_FLAG( mat, type, comp, vect, tflag ) \
62 (( tflag == CV_ROW_SAMPLE ) \
63 ? (CV_MAT_ELEM( mat, type, comp, vect )) \
64 : (CV_MAT_ELEM( mat, type, vect, comp )))

66/* Convert matrix to vector */
67#define ICV_MAT2VEC( mat, vdata, vstep, num ) \
68 if( MIN( (mat).rows, (mat).cols ) != 1 ) \
69 CV_ERROR( CV_StsBadArg, "" ); \
70 (vdata) = ((mat).data.ptr); \
71 if( (mat).rows == 1 ) \
72 { \
73 (vstep) = CV_ELEM_SIZE( (mat).type ); \
74 (num) = (mat).cols; \
75 } \
76 else \
77 { \
78 (vstep) = (mat).step; \
79 (num) = (mat).rows; \
80 }
81
82/* get raw data */
83#define ICV_RAWDATA( mat, flags, rdata, sstep, cstep, m, n ) \
84 (rdata) = (mat).data.ptr; \
85 if( CV_IS_ROW_SAMPLE( flags ) ) \
86 { \
87 (sstep) = (mat).step; \
88 (cstep) = CV_ELEM_SIZE( (mat).type ); \
89 (m) = (mat).rows; \
90 (n) = (mat).cols; \
91 } \
92 else \
93 { \
94 (cstep) = (mat).step; \
95 (sstep) = CV_ELEM_SIZE( (mat).type ); \
96 (n) = (mat).rows; \
97 (m) = (mat).cols; \
98 }
99
100#define ICV_IS_MAT_OF_TYPE( mat, mat_type) \
101 (CV_IS_MAT( mat ) && CV_MAT_TYPE( mat->type ) == (mat_type) && \
102 (mat)->cols > 0 && (mat)->rows > 0)
103
104/*
105 uchar* data; int sstep, cstep; - trainData->data
106 uchar* classes; int clstep; int ncl;- trainClasses
107 uchar* tmask; int tmstep; int ntm; - typeMask
108 uchar* missed;int msstep, mcstep; -missedMeasurements...
109 int mm, mn; == m,n == size,dim
110 uchar* sidx;int sistep; - sampleIdx
111 uchar* cidx;int cistep; - compIdx
112 int k, l; == n,m == dim,size (length of cidx, sidx)
113 int m, n; == size,dim
114*/
115#define ICV_DECLARE_TRAIN_ARGS() \
116 uchar* data; \
117 int sstep, cstep; \
118 uchar* classes; \
119 int clstep; \
120 int ncl; \
121 uchar* tmask; \
122 int tmstep; \
123 int ntm; \
124 uchar* missed; \
125 int msstep, mcstep; \
126 int mm, mn; \
127 uchar* sidx; \
128 int sistep; \
129 uchar* cidx; \
130 int cistep; \
131 int k, l; \
132 int m, n; \
133 \
134 data = classes = tmask = missed = sidx = cidx = NULL; \
135 sstep = cstep = clstep = ncl = tmstep = ntm = msstep = mcstep = mm = mn = 0; \
136 sistep = cistep = k = l = m = n = 0;
137
138#define ICV_TRAIN_DATA_REQUIRED( param, flags ) \
139 if( !ICV_IS_MAT_OF_TYPE( (param), CV_32FC1 ) ) \
140 { \
141 CV_ERROR( CV_StsBadArg, "Invalid " #param " parameter" ); \
142 } \
143 else \
144 { \
145 ICV_RAWDATA( *(param), (flags), data, sstep, cstep, m, n ); \
146 k = n; \
147 l = m; \
148 }
149
150#define ICV_TRAIN_CLASSES_REQUIRED( param ) \
151 if( !ICV_IS_MAT_OF_TYPE( (param), CV_32FC1 ) ) \
152 { \
153 CV_ERROR( CV_StsBadArg, "Invalid " #param " parameter" ); \
154 } \
155 else \
156 { \
157 ICV_MAT2VEC( *(param), classes, clstep, ncl ); \
158 if( m != ncl ) \
159 { \
160 CV_ERROR( CV_StsBadArg, "Unmatched sizes" ); \
161 } \
162 }
163
164#define ICV_ARG_NULL( param ) \
165 if( (param) != NULL ) \
166 { \
167 CV_ERROR( CV_StsBadArg, #param " parameter must be NULL" ); \
168 }
169
170#define ICV_MISSED_MEASUREMENTS_OPTIONAL( param, flags ) \
171 if( param ) \
172 { \
173 if( !ICV_IS_MAT_OF_TYPE( param, CV_8UC1 ) ) \
174 { \
175 CV_ERROR( CV_StsBadArg, "Invalid " #param " parameter" ); \
176 } \
177 else \
178 { \
179 ICV_RAWDATA( *(param), (flags), missed, msstep, mcstep, mm, mn ); \
180 if( mm != m || mn != n ) \
181 { \
182 CV_ERROR( CV_StsBadArg, "Unmatched sizes" ); \
183 } \
184 } \
185 }
186
187#define ICV_COMP_IDX_OPTIONAL( param ) \
188 if( param ) \
189 { \
190 if( !ICV_IS_MAT_OF_TYPE( param, CV_32SC1 ) ) \
191 { \
192 CV_ERROR( CV_StsBadArg, "Invalid " #param " parameter" ); \
193 } \
194 else \
195 { \
196 ICV_MAT2VEC( *(param), cidx, cistep, k ); \
197 if( k > n ) \
198 CV_ERROR( CV_StsBadArg, "Invalid " #param " parameter" ); \
199 } \
200 }
201
202#define ICV_SAMPLE_IDX_OPTIONAL( param ) \
203 if( param ) \
204 { \
205 if( !ICV_IS_MAT_OF_TYPE( param, CV_32SC1 ) ) \
206 { \
207 CV_ERROR( CV_StsBadArg, "Invalid " #param " parameter" ); \
208 } \
209 else \
210 { \
211 ICV_MAT2VEC( *sampleIdx, sidx, sistep, l ); \
212 if( l > m ) \
213 CV_ERROR( CV_StsBadArg, "Invalid " #param " parameter" ); \
214 } \
215 }

218#define ICV_CONVERT_FLOAT_ARRAY_TO_MATRICE( array, matrice ) \
219{ \
220 CvMat a, b; \
221 int dims = (matrice)->cols; \
222 int nsamples = (matrice)->rows; \
223 int type = CV_MAT_TYPE((matrice)->type); \
224 int i, offset = dims; \
225 \
226 CV_ASSERT( type == CV_32FC1 || type == CV_64FC1 ); \
227 offset *= ((type == CV_32FC1) ? sizeof(float) : sizeof(double));\
228 \
229 b = cvMat( 1, dims, CV_32FC1 ); \
230 cvGetRow( matrice, &a, 0 ); \
231 for( i = 0; i < nsamples; i++, a.data.ptr += offset ) \
232 { \
233 b.data.fl = (float*)array[i]; \
234 CV_CALL( cvConvert( &b, &a ) ); \
235 } \
236}

242/* Generates a set of classes centers in quantity <num_of_clusters> that are generated as
243 uniform random vectors in parallelepiped, where <data> is concentrated. Vectors in
244 <data> should have horizontal orientation. If <centers> != NULL, the function doesn't
245 allocate any memory and stores generated centers in <centers>, returns <centers>.
246 If <centers> == NULL, the function allocates memory and creates the matrice. Centers
247 are supposed to be oriented horizontally. */
248CvMat* icvGenerateRandomClusterCenters( int seed,
249 const CvMat* data,
250 int num_of_clusters,
251 CvMat* centers CV_DEFAULT(0));
252
253/* Fills the <labels> using <probs> by choosing the maximal probability. Outliers are
254 fixed by <oulier_tresh> and have cluster label (-1). Function also controls that there
255 weren't "empty" clusters by filling empty clusters with the maximal probability vector.
256 If probs_sums != NULL, filles it with the sums of probabilities for each sample (it is
257 useful for normalizing probabilities' matrice of FCM) */
258void icvFindClusterLabels( const CvMat* probs, float outlier_thresh, float r,
259 const CvMat* labels );
