69void CvKNearest::clear()
70{
71 while( samples )
72 {
73 CvVectors* next_samples = samples->next;
74 cvFree( &samples->data.fl );
75 cvFree( &samples );
76 samples = next_samples;
77 }
78 var_count = 0;
79 total = 0;
80 max_k = 0;
81}

92bool CvKNearest::train( const CvMat* _train_data, const CvMat* _responses,
93 const CvMat* _sample_idx, bool _is_regression,
94 int _max_k, bool _update_base )
95{
96 bool ok = false;
97 CvMat* responses = 0;
98
99 CV_FUNCNAME( "CvKNearest::train" );
100
101 __BEGIN__;
102
103 CvVectors* _samples = 0;
104 float** _data = 0;
105 int _count = 0, _dims = 0, _dims_all = 0, _rsize = 0;
106
107 if( !_update_base )
108 clear();
109
110 // Prepare training data and related parameters.
111 // Treat categorical responses as ordered - to prevent class label compression and
112 // to enable entering new classes in the updates
113 CV_CALL( cvPrepareTrainData( "CvKNearest::train", _train_data, CV_ROW_SAMPLE,
114 _responses, CV_VAR_ORDERED, 0, _sample_idx, true, (const float***)&_data,
115 &_count, &_dims, &_dims_all, &responses, 0, 0 ));
116
117 if( !responses )
118 CV_ERROR( CV_StsNoMem, "Could not allocate memory for responses" );
119
120 if( _update_base && _dims != var_count )
121 CV_ERROR( CV_StsBadArg, "The newly added data have different dimensionality" );
122
123 if( !_update_base )
124 {
125 if( _max_k < 1 )
126 CV_ERROR( CV_StsOutOfRange, "max_k must be a positive number" );
127
128 regression = _is_regression;
129 var_count = _dims;
130 max_k = _max_k;
131 }
132
133 _rsize = _count*sizeof(float);
134 CV_CALL( _samples = (CvVectors*)cvAlloc( sizeof(*_samples) + _rsize ));
135 _samples->next = samples;
136 _samples->type = CV_32F;
137 _samples->data.fl = _data;
138 _samples->count = _count;
139 total += _count;
140
141 samples = _samples;
142 memcpy( _samples + 1, responses->data.fl, _rsize );
143
144 ok = true;
145
146 __END__;
147
148 if( responses && responses->data.ptr != _responses->data.ptr )
149 cvReleaseMat(&responses);
150
151 return ok;
152}

156void CvKNearest::find_neighbors_direct( const CvMat* _samples, int k, int start, int end,
157 float* neighbor_responses, const float** neighbors, float* dist ) const
158{
159 int i, j, count = end - start, k1 = 0, k2 = 0, d = var_count;
160 CvVectors* s = samples;
161
162 for( ; s != 0; s = s->next )
163 {
164 int n = s->count;
165 for( j = 0; j < n; j++ )
166 {
167 for( i = 0; i < count; i++ )
168 {
169 double sum = 0;
170 Cv32suf si;
171 const float* v = s->data.fl[j];
172 const float* u = (float*)(_samples->data.ptr + _samples->step*(start + i));
173 Cv32suf* dd = (Cv32suf*)(dist + i*k);
174 float* nr;
175 const float** nn;
176 int t, ii, ii1;
177
178 for( t = 0; t <= d - 4; t += 4 )
179 {
180 double t0 = u[t] - v[t], t1 = u[t+1] - v[t+1];
181 double t2 = u[t+2] - v[t+2], t3 = u[t+3] - v[t+3];
182 sum += t0*t0 + t1*t1 + t2*t2 + t3*t3;
183 }
184
185 for( ; t < d; t++ )
186 {
187 double t0 = u[t] - v[t];
188 sum += t0*t0;
189 }
190
191 si.f = (float)sum;
192 for( ii = k1-1; ii >= 0; ii-- )
193 if( si.i > dd[ii].i )
194 break;
195 if( ii >= k-1 )
196 continue;
197
198 nr = neighbor_responses + i*k;
199 nn = neighbors ? neighbors + (start + i)*k : 0;
200 for( ii1 = k2 - 1; ii1 > ii; ii1-- )
201 {
202 dd[ii1+1].i = dd[ii1].i;
203 nr[ii1+1] = nr[ii1];
204 if( nn ) nn[ii1+1] = nn[ii1];
205 }
206 dd[ii+1].i = si.i;
207 nr[ii+1] = ((float*)(s + 1))[j];
208 if( nn )
209 nn[ii+1] = v;
210 }
211 k1 = MIN( k1+1, k );
212 k2 = MIN( k1, k-1 );
213 }
214 }
215}

218float CvKNearest::write_results( int k, int k1, int start, int end,
219 const float* neighbor_responses, const float* dist,
220 CvMat* _results, CvMat* _neighbor_responses,
221 CvMat* _dist, Cv32suf* sort_buf ) const
222{
223 float result = 0.f;
224 int i, j, j1, count = end - start;
225 double inv_scale = 1./k1;
226 int rstep = _results && !CV_IS_MAT_CONT(_results->type) ? _results->step/sizeof(result) : 1;
227
228 for( i = 0; i < count; i++ )
229 {
230 const Cv32suf* nr = (const Cv32suf*)(neighbor_responses + i*k);
231 float* dst;
232 float r;
233 if( _results || start+i == 0 )
234 {
235 if( regression )
236 {
237 double s = 0;
238 for( j = 0; j < k1; j++ )
239 s += nr[j].f;
240 r = (float)(s*inv_scale);
241 }
242 else
243 {
244 int prev_start = 0, best_count = 0, cur_count;
245 Cv32suf best_val;
246
247 for( j = 0; j < k1; j++ )
248 sort_buf[j].i = nr[j].i;
249
250 for( j = k1-1; j > 0; j-- )
251 {
252 bool swap_fl = false;
253 for( j1 = 0; j1 < j; j1++ )
254 if( sort_buf[j1].i > sort_buf[j1+1].i )
255 {
256 int t;
257 CV_SWAP( sort_buf[j1].i, sort_buf[j1+1].i, t );
258 swap_fl = true;
259 }
260 if( !swap_fl )
261 break;
262 }
263
264 best_val.i = 0;
265 for( j = 1; j <= k1; j++ )
266 if( j == k1 || sort_buf[j].i != sort_buf[j-1].i )
267 {
268 cur_count = j - prev_start;
269 if( best_count < cur_count )
270 {
271 best_count = cur_count;
272 best_val.i = sort_buf[j-1].i;
273 }
274 prev_start = j;
275 }
276 r = best_val.f;
277 }
278
279 if( start+i == 0 )
280 result = r;
281
282 if( _results )
283 _results->data.fl[(start + i)*rstep] = r;
284 }
285
286 if( _neighbor_responses )
287 {
288 dst = (float*)(_neighbor_responses->data.ptr +
289 (start + i)*_neighbor_responses->step);
290 for( j = 0; j < k1; j++ )
291 dst[j] = nr[j].f;
292 for( ; j < k; j++ )
293 dst[j] = 0.f;
294 }
295
296 if( _dist )
297 {
298 dst = (float*)(_dist->data.ptr + (start + i)*_dist->step);
299 for( j = 0; j < k1; j++ )
300 dst[j] = dist[j + i*k];
301 for( ; j < k; j++ )
302 dst[j] = 0.f;
303 }
304 }
305
306 return result;
307}

309struct P1 : cv::ParallelLoopBody {
310 P1(const CvKNearest* _pointer, int _buf_sz, int _k, const CvMat* __samples, const float** __neighbors,
311 int _k1, CvMat* __results, CvMat* __neighbor_responses, CvMat* __dist, float* _result)
312 {
313 pointer = _pointer;
314 k = _k;
315 _samples = __samples;
316 _neighbors = __neighbors;
317 k1 = _k1;
318 _results = __results;
319 _neighbor_responses = __neighbor_responses;
320 _dist = __dist;
321 result = _result;
322 buf_sz = _buf_sz;
323 }
324
325 const CvKNearest* pointer;
326 int k;
327 const CvMat* _samples;
328 const float** _neighbors;
329 int k1;
330 CvMat* _results;
331 CvMat* _neighbor_responses;
332 CvMat* _dist;
333 float* result;
334 int buf_sz;
335
336 void operator()( const cv::Range& range ) const
337 {
338 cv::AutoBuffer<float> buf(buf_sz);
339 for(int i = range.start; i < range.end; i += 1 )
340 {
341 float* neighbor_responses = &buf[0];
342 float* dist = neighbor_responses + 1*k;
343 Cv32suf* sort_buf = (Cv32suf*)(dist + 1*k);
344
345 pointer->find_neighbors_direct( _samples, k, i, i + 1,
346 neighbor_responses, _neighbors, dist );
347
348 float r = pointer->write_results( k, k1, i, i + 1, neighbor_responses, dist,
349 _results, _neighbor_responses, _dist, sort_buf );
350
351 if( i == 0 )
352 *result = r;
353 }
354 }
355
356};

358float CvKNearest::find_nearest( const CvMat* _samples, int k, CvMat* _results,
359 const float** _neighbors, CvMat* _neighbor_responses, CvMat* _dist ) const
360{
361 float result = 0.f;
362 const int max_blk_count = 128, max_buf_sz = 1 << 12;
363
364 if( !samples )
365 CV_Error( CV_StsError, "The search tree must be constructed first using train method" );
366
367 if( !CV_IS_MAT(_samples) ||
368 CV_MAT_TYPE(_samples->type) != CV_32FC1 ||
369 _samples->cols != var_count )
370 CV_Error( CV_StsBadArg, "Input samples must be floating-point matrix (<num_samples>x<var_count>)" );
371
372 if( _results && (!CV_IS_MAT(_results) ||
373 (_results->cols != 1 && _results->rows != 1) ||
374 _results->cols + _results->rows - 1 != _samples->rows) )
375 CV_Error( CV_StsBadArg,
376 "The results must be 1d vector containing as much elements as the number of samples" );
377
378 if( _results && CV_MAT_TYPE(_results->type) != CV_32FC1 &&
379 (CV_MAT_TYPE(_results->type) != CV_32SC1 || regression))
380 CV_Error( CV_StsUnsupportedFormat,
381 "The results must be floating-point or integer (in case of classification) vector" );
382
383 if( k < 1 || k > max_k )
384 CV_Error( CV_StsOutOfRange, "k must be within 1..max_k range" );
385
386 if( _neighbor_responses )
387 {
388 if( !CV_IS_MAT(_neighbor_responses) || CV_MAT_TYPE(_neighbor_responses->type) != CV_32FC1 ||
389 _neighbor_responses->rows != _samples->rows || _neighbor_responses->cols != k )
390 CV_Error( CV_StsBadArg,
391 "The neighbor responses (if present) must be floating-point matrix of <num_samples> x <k> size" );
392 }
393
394 if( _dist )
395 {
396 if( !CV_IS_MAT(_dist) || CV_MAT_TYPE(_dist->type) != CV_32FC1 ||
397 _dist->rows != _samples->rows || _dist->cols != k )
398 CV_Error( CV_StsBadArg,
399 "The distances from the neighbors (if present) must be floating-point matrix of <num_samples> x <k> size" );
400 }
401
402 int count = _samples->rows;
403 int count_scale = k*2;
404 int blk_count0 = MIN( count, max_blk_count );
405 int buf_sz = MIN( blk_count0 * count_scale, max_buf_sz );
406 blk_count0 = MAX( buf_sz/count_scale, 1 );
407 blk_count0 += blk_count0 % 2;
408 blk_count0 = MIN( blk_count0, count );
409 buf_sz = blk_count0 * count_scale + k;
410 int k1 = get_sample_count();
411 k1 = MIN( k1, k );
412
413 cv::parallel_for_(cv::Range(0, count), P1(this, buf_sz, k, _samples, _neighbors, k1,
414 _results, _neighbor_responses, _dist, &result)
415 );
416
417 return result;
418}

440float CvKNearest::find_nearest( const Mat& _samples, int k, Mat* _results,
441 const float** _neighbors, Mat* _neighbor_responses,
442 Mat* _dist ) const
443{
444 CvMat s = _samples, results, *presults = 0, nresponses, *pnresponses = 0, dist, *pdist = 0;
445
446 if( _results )
447 {
448 if(!(_results->data && (_results->type() == CV_32F ||
449 (_results->type() == CV_32S && regression)) &&
450 (_results->cols == 1 || _results->rows == 1) &&
451 _results->cols + _results->rows - 1 == _samples.rows) )
452 _results->create(_samples.rows, 1, CV_32F);
453 presults = &(results = *_results);
454 }
455
456 if( _neighbor_responses )
457 {
458 if(!(_neighbor_responses->data && _neighbor_responses->type() == CV_32F &&
459 _neighbor_responses->cols == k && _neighbor_responses->rows == _samples.rows) )
460 _neighbor_responses->create(_samples.rows, k, CV_32F);
461 pnresponses = &(nresponses = *_neighbor_responses);
462 }
463
464 if( _dist )
465 {
466 if(!(_dist->data && _dist->type() == CV_32F &&
467 _dist->cols == k && _dist->rows == _samples.rows) )
468 _dist->create(_samples.rows, k, CV_32F);
469 pdist = &(dist = *_dist);
470 }
471
472 return find_nearest(&s, k, presults, _neighbors, pnresponses, pdist );
473}

476float CvKNearest::find_nearest( const cv::Mat& _samples, int k, CV_OUT cv::Mat& results,
