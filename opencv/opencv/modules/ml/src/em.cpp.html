51EM::EM(int _nclusters, int _covMatType, const TermCriteria& _termCrit)
52{
53 nclusters = _nclusters;
54 covMatType = _covMatType;
55 maxIters = (_termCrit.type & TermCriteria::MAX_ITER) ? _termCrit.maxCount : DEFAULT_MAX_ITERS;
56 epsilon = (_termCrit.type & TermCriteria::EPS) ? _termCrit.epsilon : 0;

83bool EM::train(InputArray samples,
84 OutputArray logLikelihoods,
85 OutputArray labels,
86 OutputArray probs)
87{
88 Mat samplesMat = samples.getMat();
89 setTrainData(START_AUTO_STEP, samplesMat, 0, 0, 0, 0);
90 return doTrain(START_AUTO_STEP, logLikelihoods, labels, probs);

93bool EM::trainE(InputArray samples,
94 InputArray _means0,
95 InputArray _covs0,
96 InputArray _weights0,
97 OutputArray logLikelihoods,
98 OutputArray labels,
99 OutputArray probs)
100{
101 Mat samplesMat = samples.getMat();
102 vector<Mat> covs0;
103 _covs0.getMatVector(covs0);
104
105 Mat means0 = _means0.getMat(), weights0 = _weights0.getMat();
106
107 setTrainData(START_E_STEP, samplesMat, 0, !_means0.empty() ? &means0 : 0,
108 !_covs0.empty() ? &covs0 : 0, !_weights0.empty() ? &weights0 : 0);
109 return doTrain(START_E_STEP, logLikelihoods, labels, probs);
110}

126Vec2d EM::predict(InputArray _sample, OutputArray _probs) const
127{
128 Mat sample = _sample.getMat();
129 CV_Assert(isTrained());
130
131 CV_Assert(!sample.empty());
132 if(sample.type() != CV_64FC1)
133 {
134 Mat tmp;
135 sample.convertTo(tmp, CV_64FC1);
136 sample = tmp;
137 }
138 sample = sample.reshape(1, 1);
139
140 Mat probs;
141 if( _probs.needed() )
142 {
143 _probs.create(1, nclusters, CV_64FC1);
144 probs = _probs.getMat();
145 }
146
147 return computeProbabilities(sample, !probs.empty() ? &probs : 0);
148}

156static
157void checkTrainData(int startStep, const Mat& samples,
158 int nclusters, int covMatType, const Mat* probs, const Mat* means,
159 const vector<Mat>* covs, const Mat* weights)
160{
161 // Check samples.
162 CV_Assert(!samples.empty());
163 CV_Assert(samples.channels() == 1);
164
165 int nsamples = samples.rows;
166 int dim = samples.cols;
167
168 // Check training params.
169 CV_Assert(nclusters > 0);
170 CV_Assert(nclusters <= nsamples);
171 CV_Assert(startStep == EM::START_AUTO_STEP ||
172 startStep == EM::START_E_STEP ||
173 startStep == EM::START_M_STEP);
174 CV_Assert(covMatType == EM::COV_MAT_GENERIC ||
175 covMatType == EM::COV_MAT_DIAGONAL ||
176 covMatType == EM::COV_MAT_SPHERICAL);
177
178 CV_Assert(!probs ||
179 (!probs->empty() &&
180 probs->rows == nsamples && probs->cols == nclusters &&
181 (probs->type() == CV_32FC1 || probs->type() == CV_64FC1)));
182
183 CV_Assert(!weights ||
184 (!weights->empty() &&
185 (weights->cols == 1 || weights->rows == 1) && static_cast<int>(weights->total()) == nclusters &&
186 (weights->type() == CV_32FC1 || weights->type() == CV_64FC1)));
187
188 CV_Assert(!means ||
189 (!means->empty() &&
190 means->rows == nclusters && means->cols == dim &&
191 means->channels() == 1));
192
193 CV_Assert(!covs ||
194 (!covs->empty() &&
195 static_cast<int>(covs->size()) == nclusters));
196 if(covs)
197 {
198 const Size covSize(dim, dim);
199 for(size_t i = 0; i < covs->size(); i++)
200 {
201 const Mat& m = (*covs)[i];
202 CV_Assert(!m.empty() && m.size() == covSize && (m.channels() == 1));
203 }
204 }
205
206 if(startStep == EM::START_E_STEP)
207 {
208 CV_Assert(means);
209 }
210 else if(startStep == EM::START_M_STEP)
211 {
212 CV_Assert(probs);
213 }
214}

216static
217void preprocessSampleData(const Mat& src, Mat& dst, int dstType, bool isAlwaysClone)
218{
219 if(src.type() == dstType && !isAlwaysClone)
220 dst = src;
221 else
222 src.convertTo(dst, dstType);
223}
224
225static
226void preprocessProbability(Mat& probs)
227{
228 max(probs, 0., probs);
229
230 const double uniformProbability = (double)(1./probs.cols);
231 for(int y = 0; y < probs.rows; y++)
232 {
233 Mat sampleProbs = probs.row(y);
234
235 double maxVal = 0;
236 minMaxLoc(sampleProbs, 0, &maxVal);
237 if(maxVal < FLT_EPSILON)
238 sampleProbs.setTo(uniformProbability);
239 else
240 normalize(sampleProbs, sampleProbs, 1, 0, NORM_L1);
241 }
242}

244void EM::setTrainData(int startStep, const Mat& samples,
245 const Mat* probs0,
246 const Mat* means0,
247 const vector<Mat>* covs0,
248 const Mat* weights0)
249{
250 clear();
251
252 checkTrainData(startStep, samples, nclusters, covMatType, probs0, means0, covs0, weights0);
253
254 bool isKMeansInit = (startStep == EM::START_AUTO_STEP) || (startStep == EM::START_E_STEP && (covs0 == 0 || weights0 == 0));
255 // Set checked data
256 preprocessSampleData(samples, trainSamples, isKMeansInit ? CV_32FC1 : CV_64FC1, false);
257
258 // set probs
259 if(probs0 && startStep == EM::START_M_STEP)
260 {
261 preprocessSampleData(*probs0, trainProbs, CV_64FC1, true);
262 preprocessProbability(trainProbs);
263 }
264
265 // set weights
266 if(weights0 && (startStep == EM::START_E_STEP && covs0))
267 {
268 weights0->convertTo(weights, CV_64FC1);
269 weights = weights.reshape(1,1);
270 preprocessProbability(weights);
271 }
272
273 // set means
274 if(means0 && (startStep == EM::START_E_STEP/* || startStep == EM::START_AUTO_STEP*/))
275 means0->convertTo(means, isKMeansInit ? CV_32FC1 : CV_64FC1);
276
277 // set covs
278 if(covs0 && (startStep == EM::START_E_STEP && weights0))
279 {
280 covs.resize(nclusters);
281 for(size_t i = 0; i < covs0->size(); i++)
282 (*covs0)[i].convertTo(covs[i], CV_64FC1);
283 }
284}

286void EM::decomposeCovs()
287{
288 CV_Assert(!covs.empty());
289 covsEigenValues.resize(nclusters);
290 if(covMatType == EM::COV_MAT_GENERIC)
291 covsRotateMats.resize(nclusters);
292 invCovsEigenValues.resize(nclusters);
293 for(int clusterIndex = 0; clusterIndex < nclusters; clusterIndex++)
294 {
295 CV_Assert(!covs[clusterIndex].empty());
296
297 SVD svd(covs[clusterIndex], SVD::MODIFY_A + SVD::FULL_UV);
298
299 if(covMatType == EM::COV_MAT_SPHERICAL)
300 {
301 double maxSingularVal = svd.w.at<double>(0);
302 covsEigenValues[clusterIndex] = Mat(1, 1, CV_64FC1, Scalar(maxSingularVal));
303 }
304 else if(covMatType == EM::COV_MAT_DIAGONAL)
305 {
306 covsEigenValues[clusterIndex] = svd.w;
307 }
308 else //EM::COV_MAT_GENERIC
309 {
310 covsEigenValues[clusterIndex] = svd.w;
311 covsRotateMats[clusterIndex] = svd.u;
312 }
313 max(covsEigenValues[clusterIndex], minEigenValue, covsEigenValues[clusterIndex]);
314 invCovsEigenValues[clusterIndex] = 1./covsEigenValues[clusterIndex];
315 }
316}

318void EM::clusterTrainSamples()
319{
320 int nsamples = trainSamples.rows;
321
322 // Cluster samples, compute/update means
323
324 // Convert samples and means to 32F, because kmeans requires this type.
325 Mat trainSamplesFlt, meansFlt;
326 if(trainSamples.type() != CV_32FC1)
327 trainSamples.convertTo(trainSamplesFlt, CV_32FC1);
328 else
329 trainSamplesFlt = trainSamples;
330 if(!means.empty())
331 {
332 if(means.type() != CV_32FC1)
333 means.convertTo(meansFlt, CV_32FC1);
334 else
335 meansFlt = means;
336 }
337
338 Mat labels;
339 kmeans(trainSamplesFlt, nclusters, labels, TermCriteria(TermCriteria::COUNT, means.empty() ? 10 : 1, 0.5), 10, KMEANS_PP_CENTERS, meansFlt);
340
341 // Convert samples and means back to 64F.
342 CV_Assert(meansFlt.type() == CV_32FC1);
343 if(trainSamples.type() != CV_64FC1)
344 {
345 Mat trainSamplesBuffer;
346 trainSamplesFlt.convertTo(trainSamplesBuffer, CV_64FC1);
347 trainSamples = trainSamplesBuffer;
348 }
349 meansFlt.convertTo(means, CV_64FC1);
350
351 // Compute weights and covs
352 weights = Mat(1, nclusters, CV_64FC1, Scalar(0));
353 covs.resize(nclusters);
354 for(int clusterIndex = 0; clusterIndex < nclusters; clusterIndex++)
355 {
356 Mat clusterSamples;
357 for(int sampleIndex = 0; sampleIndex < nsamples; sampleIndex++)
358 {
359 if(labels.at<int>(sampleIndex) == clusterIndex)
360 {
361 const Mat sample = trainSamples.row(sampleIndex);
362 clusterSamples.push_back(sample);
363 }
364 }
365 CV_Assert(!clusterSamples.empty());
366
367 calcCovarMatrix(clusterSamples, covs[clusterIndex], means.row(clusterIndex),
368 CV_COVAR_NORMAL + CV_COVAR_ROWS + CV_COVAR_USE_AVG + CV_COVAR_SCALE, CV_64FC1);
369 weights.at<double>(clusterIndex) = static_cast<double>(clusterSamples.rows)/static_cast<double>(nsamples);
370 }
371
372 decomposeCovs();
373}

375void EM::computeLogWeightDivDet()
376{
377 CV_Assert(!covsEigenValues.empty());
378
379 Mat logWeights;
380 cv::max(weights, DBL_MIN, weights);
381 log(weights, logWeights);
382
383 logWeightDivDet.create(1, nclusters, CV_64FC1);
384 // note: logWeightDivDet = log(weight_k) - 0.5 * log(|det(cov_k)|)
385
386 for(int clusterIndex = 0; clusterIndex < nclusters; clusterIndex++)
387 {
388 double logDetCov = 0.;
389 const int evalCount = static_cast<int>(covsEigenValues[clusterIndex].total());
390 for(int di = 0; di < evalCount; di++)
391 logDetCov += std::log(covsEigenValues[clusterIndex].at<double>(covMatType != EM::COV_MAT_SPHERICAL ? di : 0));
392
393 logWeightDivDet.at<double>(clusterIndex) = logWeights.at<double>(clusterIndex) - 0.5 * logDetCov;
394 }
395}

397bool EM::doTrain(int startStep, OutputArray logLikelihoods, OutputArray labels, OutputArray probs)
398{
399 int dim = trainSamples.cols;
400 // Precompute the empty initial train data in the cases of EM::START_E_STEP and START_AUTO_STEP
401 if(startStep != EM::START_M_STEP)
402 {
403 if(covs.empty())
404 {
405 CV_Assert(weights.empty());
406 clusterTrainSamples();
407 }
408 }
409
410 if(!covs.empty() && covsEigenValues.empty() )
411 {
412 CV_Assert(invCovsEigenValues.empty());
413 decomposeCovs();
414 }
415
416 if(startStep == EM::START_M_STEP)
417 mStep();
418
419 double trainLogLikelihood, prevTrainLogLikelihood = 0.;
420 for(int iter = 0; ; iter++)
421 {
422 eStep();
423 trainLogLikelihood = sum(trainLogLikelihoods)[0];
424
425 if(iter >= maxIters - 1)
426 break;
427
428 double trainLogLikelihoodDelta = trainLogLikelihood - prevTrainLogLikelihood;
429 if( iter != 0 &&
430 (trainLogLikelihoodDelta < -DBL_EPSILON ||
431 trainLogLikelihoodDelta < epsilon * std::fabs(trainLogLikelihood)))
432 break;
433
434 mStep();
435
436 prevTrainLogLikelihood = trainLogLikelihood;
437 }
438
439 if( trainLogLikelihood <= -DBL_MAX/10000. )
440 {
441 clear();
442 return false;
443 }
444
445 // postprocess covs
446 covs.resize(nclusters);
447 for(int clusterIndex = 0; clusterIndex < nclusters; clusterIndex++)
448 {
449 if(covMatType == EM::COV_MAT_SPHERICAL)
450 {
451 covs[clusterIndex].create(dim, dim, CV_64FC1);
452 setIdentity(covs[clusterIndex], Scalar(covsEigenValues[clusterIndex].at<double>(0)));
453 }
454 else if(covMatType == EM::COV_MAT_DIAGONAL)
455 {
456 covs[clusterIndex] = Mat::diag(covsEigenValues[clusterIndex]);
457 }
458 }
459
460 if(labels.needed())
461 trainLabels.copyTo(labels);
462 if(probs.needed())
463 trainProbs.copyTo(probs);
464 if(logLikelihoods.needed())
465 trainLogLikelihoods.copyTo(logLikelihoods);
466
467 trainSamples.release();
468 trainProbs.release();
469 trainLabels.release();
470 trainLogLikelihoods.release();
471
472 return true;
473}

475Vec2d EM::computeProbabilities(const Mat& sample, Mat* probs) const
476{
477 // L_ik = log(weight_k) - 0.5 * log(|det(cov_k)|) - 0.5 *(x_i - mean_k)' cov_k^(-1) (x_i - mean_k)]
478 // q = arg(max_k(L_ik))
479 // probs_ik = exp(L_ik - L_iq) / (1 + sum_j!=q (exp(L_ij - L_iq))
480 // see Alex Smola's blog http://blog.smola.org/page/2 for
481 // details on the log-sum-exp trick
482
483 CV_Assert(!means.empty());
484 CV_Assert(sample.type() == CV_64FC1);
485 CV_Assert(sample.rows == 1);
486 CV_Assert(sample.cols == means.cols);
487
488 int dim = sample.cols;
489
490 Mat L(1, nclusters, CV_64FC1);
491 int label = 0;
492 for(int clusterIndex = 0; clusterIndex < nclusters; clusterIndex++)
493 {
494 const Mat centeredSample = sample - means.row(clusterIndex);
495
496 Mat rotatedCenteredSample = covMatType != EM::COV_MAT_GENERIC ?
497 centeredSample : centeredSample * covsRotateMats[clusterIndex];
498
499 double Lval = 0;
500 for(int di = 0; di < dim; di++)
501 {
502 double w = invCovsEigenValues[clusterIndex].at<double>(covMatType != EM::COV_MAT_SPHERICAL ? di : 0);
503 double val = rotatedCenteredSample.at<double>(di);
504 Lval += w * val * val;
505 }
506 CV_DbgAssert(!logWeightDivDet.empty());
507 L.at<double>(clusterIndex) = logWeightDivDet.at<double>(clusterIndex) - 0.5 * Lval;
508
509 if(L.at<double>(clusterIndex) > L.at<double>(label))
510 label = clusterIndex;
511 }
512
513 double maxLVal = L.at<double>(label);
514 Mat expL_Lmax = L; // exp(L_ij - L_iq)
515 for(int i = 0; i < L.cols; i++)
516 expL_Lmax.at<double>(i) = std::exp(L.at<double>(i) - maxLVal);
517 double expDiffSum = sum(expL_Lmax)[0]; // sum_j(exp(L_ij - L_iq))
518
519 if(probs)
520 {
521 probs->create(1, nclusters, CV_64FC1);
522 double factor = 1./expDiffSum;
523 expL_Lmax *= factor;
524 expL_Lmax.copyTo(*probs);
525 }
526
527 Vec2d res;
528 res[0] = std::log(expDiffSum) + maxLVal - 0.5 * dim * CV_LOG2PI;
529 res[1] = label;
530
531 return res;
532}

534void EM::eStep()
535{
536 // Compute probs_ik from means_k, covs_k and weights_k.
537 trainProbs.create(trainSamples.rows, nclusters, CV_64FC1);
538 trainLabels.create(trainSamples.rows, 1, CV_32SC1);
539 trainLogLikelihoods.create(trainSamples.rows, 1, CV_64FC1);
540
541 computeLogWeightDivDet();
542
543 CV_DbgAssert(trainSamples.type() == CV_64FC1);
544 CV_DbgAssert(means.type() == CV_64FC1);
545
546 for(int sampleIndex = 0; sampleIndex < trainSamples.rows; sampleIndex++)
547 {
548 Mat sampleProbs = trainProbs.row(sampleIndex);
549 Vec2d res = computeProbabilities(trainSamples.row(sampleIndex), &sampleProbs);
550 trainLogLikelihoods.at<double>(sampleIndex) = res[0];
551 trainLabels.at<int>(sampleIndex) = static_cast<int>(res[1]);
552 }
553}
554
555void EM::mStep()
556{
557 // Update means_k, covs_k and weights_k from probs_ik
558 int dim = trainSamples.cols;
559
560 // Update weights
561 // not normalized first
562 reduce(trainProbs, weights, 0, CV_REDUCE_SUM);
563
564 // Update means
565 means.create(nclusters, dim, CV_64FC1);
566 means = Scalar(0);
567
568 const double minPosWeight = trainSamples.rows * DBL_EPSILON;
569 double minWeight = DBL_MAX;
570 int minWeightClusterIndex = -1;
571 for(int clusterIndex = 0; clusterIndex < nclusters; clusterIndex++)
572 {
573 if(weights.at<double>(clusterIndex) <= minPosWeight)
574 continue;
575
576 if(weights.at<double>(clusterIndex) < minWeight)
577 {
578 minWeight = weights.at<double>(clusterIndex);
579 minWeightClusterIndex = clusterIndex;
580 }
581
582 Mat clusterMean = means.row(clusterIndex);
583 for(int sampleIndex = 0; sampleIndex < trainSamples.rows; sampleIndex++)
584 clusterMean += trainProbs.at<double>(sampleIndex, clusterIndex) * trainSamples.row(sampleIndex);
585 clusterMean /= weights.at<double>(clusterIndex);
586 }
587
588 // Update covsEigenValues and invCovsEigenValues
589 covs.resize(nclusters);
590 covsEigenValues.resize(nclusters);
591 if(covMatType == EM::COV_MAT_GENERIC)
592 covsRotateMats.resize(nclusters);
593 invCovsEigenValues.resize(nclusters);
594 for(int clusterIndex = 0; clusterIndex < nclusters; clusterIndex++)
595 {
596 if(weights.at<double>(clusterIndex) <= minPosWeight)
597 continue;
598
599 if(covMatType != EM::COV_MAT_SPHERICAL)
600 covsEigenValues[clusterIndex].create(1, dim, CV_64FC1);
601 else
602 covsEigenValues[clusterIndex].create(1, 1, CV_64FC1);
603
604 if(covMatType == EM::COV_MAT_GENERIC)
605 covs[clusterIndex].create(dim, dim, CV_64FC1);
606
607 Mat clusterCov = covMatType != EM::COV_MAT_GENERIC ?
608 covsEigenValues[clusterIndex] : covs[clusterIndex];
609
610 clusterCov = Scalar(0);
611
612 Mat centeredSample;
613 for(int sampleIndex = 0; sampleIndex < trainSamples.rows; sampleIndex++)
614 {
615 centeredSample = trainSamples.row(sampleIndex) - means.row(clusterIndex);
616
617 if(covMatType == EM::COV_MAT_GENERIC)
618 clusterCov += trainProbs.at<double>(sampleIndex, clusterIndex) * centeredSample.t() * centeredSample;
619 else
620 {
621 double p = trainProbs.at<double>(sampleIndex, clusterIndex);
622 for(int di = 0; di < dim; di++ )
623 {
624 double val = centeredSample.at<double>(di);
625 clusterCov.at<double>(covMatType != EM::COV_MAT_SPHERICAL ? di : 0) += p*val*val;
626 }
627 }
628 }
629
630 if(covMatType == EM::COV_MAT_SPHERICAL)
631 clusterCov /= dim;
632
633 clusterCov /= weights.at<double>(clusterIndex);
634
635 // Update covsRotateMats for EM::COV_MAT_GENERIC only
636 if(covMatType == EM::COV_MAT_GENERIC)
637 {
638 SVD svd(covs[clusterIndex], SVD::MODIFY_A + SVD::FULL_UV);
639 covsEigenValues[clusterIndex] = svd.w;
640 covsRotateMats[clusterIndex] = svd.u;
641 }
642
643 max(covsEigenValues[clusterIndex], minEigenValue, covsEigenValues[clusterIndex]);
644
645 // update invCovsEigenValues
646 invCovsEigenValues[clusterIndex] = 1./covsEigenValues[clusterIndex];
647 }
648
649 for(int clusterIndex = 0; clusterIndex < nclusters; clusterIndex++)
650 {
651 if(weights.at<double>(clusterIndex) <= minPosWeight)
652 {
653 Mat clusterMean = means.row(clusterIndex);
654 means.row(minWeightClusterIndex).copyTo(clusterMean);
655 covs[minWeightClusterIndex].copyTo(covs[clusterIndex]);
656 covsEigenValues[minWeightClusterIndex].copyTo(covsEigenValues[clusterIndex]);
657 if(covMatType == EM::COV_MAT_GENERIC)
658 covsRotateMats[minWeightClusterIndex].copyTo(covsRotateMats[clusterIndex]);
659 invCovsEigenValues[minWeightClusterIndex].copyTo(invCovsEigenValues[clusterIndex]);
660 }
661 }
662
663 // Normalize weights
664 weights /= trainSamples.rows;
665}
