107bool CvNormalBayesClassifier::train( const CvMat* _train_data, const CvMat* _responses,
108 const CvMat* _var_idx, const CvMat* _sample_idx, bool update )
109{
110 const float min_variation = FLT_EPSILON;
111 bool result = false;
112 CvMat* responses = 0;
113 const float** train_data = 0;
114 CvMat* __cls_labels = 0;
115 CvMat* __var_idx = 0;
116 CvMat* cov = 0;
117
118 CV_FUNCNAME( "CvNormalBayesClassifier::train" );
119
120 __BEGIN__;
121
122 int cls, nsamples = 0, _var_count = 0, _var_all = 0, nclasses = 0;
123 int s, c1, c2;
124 const int* responses_data;
125
126 CV_CALL( cvPrepareTrainData( 0,
127 _train_data, CV_ROW_SAMPLE, _responses, CV_VAR_CATEGORICAL,
128 _var_idx, _sample_idx, false, &train_data,
129 &nsamples, &_var_count, &_var_all, &responses,
130 &__cls_labels, &__var_idx ));
131
132 if( !update )
133 {
134 const size_t mat_size = sizeof(CvMat*);
135 size_t data_size;
136
137 clear();
138
139 var_idx = __var_idx;
140 cls_labels = __cls_labels;
141 __var_idx = __cls_labels = 0;
142 var_count = _var_count;
143 var_all = _var_all;
144
145 nclasses = cls_labels->cols;
146 data_size = nclasses*6*mat_size;
147
148 CV_CALL( count = (CvMat**)cvAlloc( data_size ));
149 memset( count, 0, data_size );
150
151 sum = count + nclasses;
152 productsum = sum + nclasses;
153 avg = productsum + nclasses;
154 inv_eigen_values= avg + nclasses;
155 cov_rotate_mats = inv_eigen_values + nclasses;
156
157 CV_CALL( c = cvCreateMat( 1, nclasses, CV_64FC1 ));
158
159 for( cls = 0; cls < nclasses; cls++ )
160 {
161 CV_CALL(count[cls] = cvCreateMat( 1, var_count, CV_32SC1 ));
162 CV_CALL(sum[cls] = cvCreateMat( 1, var_count, CV_64FC1 ));
163 CV_CALL(productsum[cls] = cvCreateMat( var_count, var_count, CV_64FC1 ));
164 CV_CALL(avg[cls] = cvCreateMat( 1, var_count, CV_64FC1 ));
165 CV_CALL(inv_eigen_values[cls] = cvCreateMat( 1, var_count, CV_64FC1 ));
166 CV_CALL(cov_rotate_mats[cls] = cvCreateMat( var_count, var_count, CV_64FC1 ));
167 CV_CALL(cvZero( count[cls] ));
168 CV_CALL(cvZero( sum[cls] ));
169 CV_CALL(cvZero( productsum[cls] ));
170 CV_CALL(cvZero( avg[cls] ));
171 CV_CALL(cvZero( inv_eigen_values[cls] ));
172 CV_CALL(cvZero( cov_rotate_mats[cls] ));
173 }
174 }
175 else
176 {
177 // check that the new training data has the same dimensionality etc.
178 if( _var_count != var_count || _var_all != var_all || !((!_var_idx && !var_idx) ||
179 (_var_idx && var_idx && cvNorm(_var_idx,var_idx,CV_C) < DBL_EPSILON)) )
180 CV_ERROR( CV_StsBadArg,
181 "The new training data is inconsistent with the original training data" );
182
183 if( cls_labels->cols != __cls_labels->cols ||
184 cvNorm(cls_labels, __cls_labels, CV_C) > DBL_EPSILON )
185 CV_ERROR( CV_StsNotImplemented,
186 "In the current implementation the new training data must have absolutely "
187 "the same set of class labels as used in the original training data" );
188
189 nclasses = cls_labels->cols;
190 }
191
192 responses_data = responses->data.i;
193 CV_CALL( cov = cvCreateMat( _var_count, _var_count, CV_64FC1 ));
194
195 /* process train data (count, sum , productsum) */
196 for( s = 0; s < nsamples; s++ )
197 {
198 cls = responses_data[s];
199 int* count_data = count[cls]->data.i;
200 double* sum_data = sum[cls]->data.db;
201 double* prod_data = productsum[cls]->data.db;
202 const float* train_vec = train_data[s];
203
204 for( c1 = 0; c1 < _var_count; c1++, prod_data += _var_count )
205 {
206 double val1 = train_vec[c1];
207 sum_data[c1] += val1;
208 count_data[c1]++;
209 for( c2 = c1; c2 < _var_count; c2++ )
210 prod_data[c2] += train_vec[c2]*val1;
211 }
212 }
213 cvReleaseMat( &responses );
214 responses = 0;
215
216 /* calculate avg, covariance matrix, c */
217 for( cls = 0; cls < nclasses; cls++ )
218 {
219 double det = 1;
220 int i, j;
221 CvMat* w = inv_eigen_values[cls];
222 int* count_data = count[cls]->data.i;
223 double* avg_data = avg[cls]->data.db;
224 double* sum1 = sum[cls]->data.db;
225
226 cvCompleteSymm( productsum[cls], 0 );
227
228 for( j = 0; j < _var_count; j++ )
229 {
230 int n = count_data[j];
231 avg_data[j] = n ? sum1[j] / n : 0.;
232 }
233
234 count_data = count[cls]->data.i;
235 avg_data = avg[cls]->data.db;
236 sum1 = sum[cls]->data.db;
237
238 for( i = 0; i < _var_count; i++ )
239 {
240 double* avg2_data = avg[cls]->data.db;
241 double* sum2 = sum[cls]->data.db;
242 double* prod_data = productsum[cls]->data.db + i*_var_count;
243 double* cov_data = cov->data.db + i*_var_count;
244 double s1val = sum1[i];
245 double avg1 = avg_data[i];
246 int _count = count_data[i];
247
248 for( j = 0; j <= i; j++ )
249 {
250 double avg2 = avg2_data[j];
251 double cov_val = prod_data[j] - avg1 * sum2[j] - avg2 * s1val + avg1 * avg2 * _count;
252 cov_val = (_count > 1) ? cov_val / (_count - 1) : cov_val;
253 cov_data[j] = cov_val;
254 }
255 }
256
257 CV_CALL( cvCompleteSymm( cov, 1 ));
258 CV_CALL( cvSVD( cov, w, cov_rotate_mats[cls], 0, CV_SVD_U_T ));
259 CV_CALL( cvMaxS( w, min_variation, w ));
260 for( j = 0; j < _var_count; j++ )
261 det *= w->data.db[j];
262
263 CV_CALL( cvDiv( NULL, w, w ));
264 c->data.db[cls] = det > 0 ? log(det) : -700;
265 }
266
267 result = true;
268
269 __END__;
270
271 if( !result || cvGetErrStatus() < 0 )
272 clear();
273
274 cvReleaseMat( &cov );
275 cvReleaseMat( &__cls_labels );
276 cvReleaseMat( &__var_idx );
277 cvFree( &train_data );
278
279 return result;
280}

282struct predict_body : cv::ParallelLoopBody {
283 predict_body(CvMat* _c, CvMat** _cov_rotate_mats, CvMat** _inv_eigen_values, CvMat** _avg,
284 const CvMat* _samples, const int* _vidx, CvMat* _cls_labels,
285 CvMat* _results, float* _value, int _var_count1
286 )
287 {
288 c = _c;
289 cov_rotate_mats = _cov_rotate_mats;
290 inv_eigen_values = _inv_eigen_values;
291 avg = _avg;
292 samples = _samples;
293 vidx = _vidx;
294 cls_labels = _cls_labels;
295 results = _results;
296 value = _value;
297 var_count1 = _var_count1;
298 }
299
300 CvMat* c;
301 CvMat** cov_rotate_mats;
302 CvMat** inv_eigen_values;
303 CvMat** avg;
304 const CvMat* samples;
305 const int* vidx;
306 CvMat* cls_labels;
307
308 CvMat* results;
309 float* value;
310 int var_count1;
311
312 void operator()( const cv::Range& range ) const
313 {
314
315 int cls = -1;
316 int rtype = 0, rstep = 0;
317 int nclasses = cls_labels->cols;
318 int _var_count = avg[0]->cols;
319
320 if (results)
321 {
322 rtype = CV_MAT_TYPE(results->type);
323 rstep = CV_IS_MAT_CONT(results->type) ? 1 : results->step/CV_ELEM_SIZE(rtype);
324 }
325 // allocate memory and initializing headers for calculating
326 cv::AutoBuffer<double> buffer(nclasses + var_count1);
327 CvMat diff = cvMat( 1, var_count1, CV_64FC1, &buffer[0] );
328
329 for(int k = range.start; k < range.end; k += 1 )
330 {
331 int ival;
332 double opt = FLT_MAX;
333
334 for(int i = 0; i < nclasses; i++ )
335 {
336
337 double cur = c->data.db[i];
338 CvMat* u = cov_rotate_mats[i];
339 CvMat* w = inv_eigen_values[i];
340
341 const double* avg_data = avg[i]->data.db;
342 const float* x = (const float*)(samples->data.ptr + samples->step*k);
343
344 // cov = u w u' --> cov^(-1) = u w^(-1) u'
345 for(int j = 0; j < _var_count; j++ )
346 diff.data.db[j] = avg_data[j] - x[vidx ? vidx[j] : j];
347
348 cvGEMM( &diff, u, 1, 0, 0, &diff, CV_GEMM_B_T );
349 for(int j = 0; j < _var_count; j++ )
350 {
351 double d = diff.data.db[j];
352 cur += d*d*w->data.db[j];
353 }
354
355 if( cur < opt )
356 {
357 cls = i;
358 opt = cur;
359 }
360 /* probability = exp( -0.5 * cur ) */
361 }
362
363 ival = cls_labels->data.i[cls];
364 if( results )
365 {
366 if( rtype == CV_32SC1 )
367 results->data.i[k*rstep] = ival;
368 else
369 results->data.fl[k*rstep] = (float)ival;
370 }
371 if( k == 0 )
372 *value = (float)ival;
373 }
374 }
375};

378float CvNormalBayesClassifier::predict( const CvMat* samples, CvMat* results ) const
379{
380 float value = 0;
381
382 if( !CV_IS_MAT(samples) || CV_MAT_TYPE(samples->type) != CV_32FC1 || samples->cols != var_all )
383 CV_Error( CV_StsBadArg,
384 "The input samples must be 32f matrix with the number of columns = var_all" );
385
386 if( samples->rows > 1 && !results )
387 CV_Error( CV_StsNullPtr,
388 "When the number of input samples is >1, the output vector of results must be passed" );
389
390 if( results )
391 {
392 if( !CV_IS_MAT(results) || (CV_MAT_TYPE(results->type) != CV_32FC1 &&
393 CV_MAT_TYPE(results->type) != CV_32SC1) ||
394 (results->cols != 1 && results->rows != 1) ||
395 results->cols + results->rows - 1 != samples->rows )
396 CV_Error( CV_StsBadArg, "The output array must be integer or floating-point vector "
397 "with the number of elements = number of rows in the input matrix" );
398 }
399
400 const int* vidx = var_idx ? var_idx->data.i : 0;
401
402 cv::parallel_for_(cv::Range(0, samples->rows),
403 predict_body(c, cov_rotate_mats, inv_eigen_values, avg, samples,
404 vidx, cls_labels, results, &value, var_count));
405
406 return value;
407}

410void CvNormalBayesClassifier::write( CvFileStorage* fs, const char* name ) const
411{
412 CV_FUNCNAME( "CvNormalBayesClassifier::write" );
413
414 __BEGIN__;
415
416 int nclasses, i;
417
418 nclasses = cls_labels->cols;
419
420 cvStartWriteStruct( fs, name, CV_NODE_MAP, CV_TYPE_NAME_ML_NBAYES );
421
422 CV_CALL( cvWriteInt( fs, "var_count", var_count ));
423 CV_CALL( cvWriteInt( fs, "var_all", var_all ));
424
425 if( var_idx )
426 CV_CALL( cvWrite( fs, "var_idx", var_idx ));
427 CV_CALL( cvWrite( fs, "cls_labels", cls_labels ));
428
429 CV_CALL( cvStartWriteStruct( fs, "count", CV_NODE_SEQ ));
430 for( i = 0; i < nclasses; i++ )
431 CV_CALL( cvWrite( fs, NULL, count[i] ));
432 CV_CALL( cvEndWriteStruct( fs ));
433
434 CV_CALL( cvStartWriteStruct( fs, "sum", CV_NODE_SEQ ));
435 for( i = 0; i < nclasses; i++ )
436 CV_CALL( cvWrite( fs, NULL, sum[i] ));
437 CV_CALL( cvEndWriteStruct( fs ));
438
439 CV_CALL( cvStartWriteStruct( fs, "productsum", CV_NODE_SEQ ));
440 for( i = 0; i < nclasses; i++ )
441 CV_CALL( cvWrite( fs, NULL, productsum[i] ));
442 CV_CALL( cvEndWriteStruct( fs ));
443
444 CV_CALL( cvStartWriteStruct( fs, "avg", CV_NODE_SEQ ));
445 for( i = 0; i < nclasses; i++ )
446 CV_CALL( cvWrite( fs, NULL, avg[i] ));
447 CV_CALL( cvEndWriteStruct( fs ));
448
449 CV_CALL( cvStartWriteStruct( fs, "inv_eigen_values", CV_NODE_SEQ ));
450 for( i = 0; i < nclasses; i++ )
451 CV_CALL( cvWrite( fs, NULL, inv_eigen_values[i] ));
452 CV_CALL( cvEndWriteStruct( fs ));
453
454 CV_CALL( cvStartWriteStruct( fs, "cov_rotate_mats", CV_NODE_SEQ ));
455 for( i = 0; i < nclasses; i++ )
456 CV_CALL( cvWrite( fs, NULL, cov_rotate_mats[i] ));
457 CV_CALL( cvEndWriteStruct( fs ));
458
459 CV_CALL( cvWrite( fs, "c", c ));
460
461 cvEndWriteStruct( fs );
462
463 __END__;
464}

467void CvNormalBayesClassifier::read( CvFileStorage* fs, CvFileNode* root_node )
468{
469 bool ok = false;
470 CV_FUNCNAME( "CvNormalBayesClassifier::read" );
471
472 __BEGIN__;
473
474 int nclasses, i;
475 size_t data_size;
476 CvFileNode* node;
477 CvSeq* seq;
478 CvSeqReader reader;
479
480 clear();
481
482 CV_CALL( var_count = cvReadIntByName( fs, root_node, "var_count", -1 ));
483 CV_CALL( var_all = cvReadIntByName( fs, root_node, "var_all", -1 ));
484 CV_CALL( var_idx = (CvMat*)cvReadByName( fs, root_node, "var_idx" ));
485 CV_CALL( cls_labels = (CvMat*)cvReadByName( fs, root_node, "cls_labels" ));
486 if( !cls_labels )
487 CV_ERROR( CV_StsParseError, "No \"cls_labels\" in NBayes classifier" );
488 if( cls_labels->cols < 1 )
489 CV_ERROR( CV_StsBadArg, "Number of classes is less 1" );
490 if( var_count <= 0 )
491 CV_ERROR( CV_StsParseError,
492 "The field \"var_count\" of NBayes classifier is missing" );
493 nclasses = cls_labels->cols;
494
495 data_size = nclasses*6*sizeof(CvMat*);
496 CV_CALL( count = (CvMat**)cvAlloc( data_size ));
497 memset( count, 0, data_size );
498
499 sum = count + nclasses;
500 productsum = sum + nclasses;
501 avg = productsum + nclasses;
502 inv_eigen_values = avg + nclasses;
503 cov_rotate_mats = inv_eigen_values + nclasses;
504
505 CV_CALL( node = cvGetFileNodeByName( fs, root_node, "count" ));
506 seq = node->data.seq;
507 if( !CV_NODE_IS_SEQ(node->tag) || seq->total != nclasses)
508 CV_ERROR( CV_StsBadArg, "" );
509 CV_CALL( cvStartReadSeq( seq, &reader, 0 ));
510 for( i = 0; i < nclasses; i++ )
511 {
512 CV_CALL( count[i] = (CvMat*)cvRead( fs, (CvFileNode*)reader.ptr ));
513 CV_NEXT_SEQ_ELEM( seq->elem_size, reader );
514 }
515
516 CV_CALL( node = cvGetFileNodeByName( fs, root_node, "sum" ));
517 seq = node->data.seq;
518 if( !CV_NODE_IS_SEQ(node->tag) || seq->total != nclasses)
519 CV_ERROR( CV_StsBadArg, "" );
520 CV_CALL( cvStartReadSeq( seq, &reader, 0 ));
521 for( i = 0; i < nclasses; i++ )
522 {
523 CV_CALL( sum[i] = (CvMat*)cvRead( fs, (CvFileNode*)reader.ptr ));
524 CV_NEXT_SEQ_ELEM( seq->elem_size, reader );
525 }
526
527 CV_CALL( node = cvGetFileNodeByName( fs, root_node, "productsum" ));
528 seq = node->data.seq;
529 if( !CV_NODE_IS_SEQ(node->tag) || seq->total != nclasses)
530 CV_ERROR( CV_StsBadArg, "" );
531 CV_CALL( cvStartReadSeq( seq, &reader, 0 ));
532 for( i = 0; i < nclasses; i++ )
533 {
534 CV_CALL( productsum[i] = (CvMat*)cvRead( fs, (CvFileNode*)reader.ptr ));
535 CV_NEXT_SEQ_ELEM( seq->elem_size, reader );
536 }
537
538 CV_CALL( node = cvGetFileNodeByName( fs, root_node, "avg" ));
539 seq = node->data.seq;
540 if( !CV_NODE_IS_SEQ(node->tag) || seq->total != nclasses)
541 CV_ERROR( CV_StsBadArg, "" );
542 CV_CALL( cvStartReadSeq( seq, &reader, 0 ));
543 for( i = 0; i < nclasses; i++ )
544 {
545 CV_CALL( avg[i] = (CvMat*)cvRead( fs, (CvFileNode*)reader.ptr ));
546 CV_NEXT_SEQ_ELEM( seq->elem_size, reader );
547 }
548
549 CV_CALL( node = cvGetFileNodeByName( fs, root_node, "inv_eigen_values" ));
550 seq = node->data.seq;
551 if( !CV_NODE_IS_SEQ(node->tag) || seq->total != nclasses)
552 CV_ERROR( CV_StsBadArg, "" );
553 CV_CALL( cvStartReadSeq( seq, &reader, 0 ));
554 for( i = 0; i < nclasses; i++ )
555 {
556 CV_CALL( inv_eigen_values[i] = (CvMat*)cvRead( fs, (CvFileNode*)reader.ptr ));
557 CV_NEXT_SEQ_ELEM( seq->elem_size, reader );
558 }
559
560 CV_CALL( node = cvGetFileNodeByName( fs, root_node, "cov_rotate_mats" ));
561 seq = node->data.seq;
562 if( !CV_NODE_IS_SEQ(node->tag) || seq->total != nclasses)
563 CV_ERROR( CV_StsBadArg, "" );
564 CV_CALL( cvStartReadSeq( seq, &reader, 0 ));
565 for( i = 0; i < nclasses; i++ )
566 {
567 CV_CALL( cov_rotate_mats[i] = (CvMat*)cvRead( fs, (CvFileNode*)reader.ptr ));
568 CV_NEXT_SEQ_ELEM( seq->elem_size, reader );
569 }
570
571 CV_CALL( c = (CvMat*)cvReadByName( fs, root_node, "c" ));
572
573 ok = true;
574
575 __END__;
576
577 if( !ok )
578 clear();
579}

611float CvNormalBayesClassifier::predict( const Mat& _samples, Mat* _results ) const
612{
613 CvMat samples = _samples, results, *presults = 0;
614
615 if( _results )
616 {
617 if( !(_results->data && _results->type() == CV_32F &&
618 (_results->cols == 1 || _results->rows == 1) &&
619 _results->cols + _results->rows - 1 == _samples.rows) )
620 _results->create(_samples.rows, 1, CV_32F);
621 presults = &(results = *_results);
622 }
623
624 return predict(&samples, presults);
625}
