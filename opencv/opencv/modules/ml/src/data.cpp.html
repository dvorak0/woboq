120static char *fgets_chomp(char *str, int n, FILE *stream)
121{
122 char *head = fgets(str, n, stream);
123 if( head )
124 {
125 for(char *tail = head + strlen(head) - 1; tail >= head; --tail)
126 {
127 if( *tail != '\r' && *tail != '\n' )
128 break;
129 *tail = '\0';
130 }
131 }
132 return head;
133}

136int CvMLData::read_csv(const char* filename)
137{
138 const int M = 1000000;
139 const char str_delimiter[3] = { ' ', delimiter, '\0' };
140 FILE* file = 0;
141 CvMemStorage* storage;
142 CvSeq* seq;
143 char *ptr;
144 float* el_ptr;
145 CvSeqReader reader;
146 int cols_count = 0;
147 uchar *var_types_ptr = 0;
148
149 clear();
150
151 file = fopen( filename, "rt" );
152
153 if( !file )
154 return -1;
155
156 // read the first line and determine the number of variables
157 std::vector<char> _buf(M);
158 char* buf = &_buf[0];
159 if( !fgets_chomp( buf, M, file ))
160 {
161 fclose(file);
162 return -1;
163 }
164
165 ptr = buf;
166 while( *ptr == ' ' )
167 ptr++;
168 for( ; *ptr != '\0'; )
169 {
170 if(*ptr == delimiter || *ptr == ' ')
171 {
172 cols_count++;
173 ptr++;
174 while( *ptr == ' ' ) ptr++;
175 }
176 else
177 ptr++;
178 }
179
180 cols_count++;
181
182 if ( cols_count == 0)
183 {
184 fclose(file);
185 return -1;
186 }
187
188 // create temporary memory storage to store the whole database
189 el_ptr = new float[cols_count];
190 storage = cvCreateMemStorage();
191 seq = cvCreateSeq( 0, sizeof(*seq), cols_count*sizeof(float), storage );
192
193 var_types = cvCreateMat( 1, cols_count, CV_8U );
194 cvZero( var_types );
195 var_types_ptr = var_types->data.ptr;
196
197 for(;;)
198 {
199 char *token = NULL;
200 int type;
201 token = strtok(buf, str_delimiter);
202 if (!token)
203 break;
204 for (int i = 0; i < cols_count-1; i++)
205 {
206 str_to_flt_elem( token, el_ptr[i], type);
207 var_types_ptr[i] |= type;
208 token = strtok(NULL, str_delimiter);
209 if (!token)
210 {
211 fclose(file);
212 delete [] el_ptr;
213 return -1;
214 }
215 }
216 str_to_flt_elem( token, el_ptr[cols_count-1], type);
217 var_types_ptr[cols_count-1] |= type;
218 cvSeqPush( seq, el_ptr );
219 if( !fgets_chomp( buf, M, file ) )
220 break;
221 }
222 fclose(file);
223
224 values = cvCreateMat( seq->total, cols_count, CV_32FC1 );
225 missing = cvCreateMat( seq->total, cols_count, CV_8U );
226 var_idx_mask = cvCreateMat( 1, values->cols, CV_8UC1 );
227 cvSet( var_idx_mask, cvRealScalar(1) );
228 train_sample_count = seq->total;
229
230 cvStartReadSeq( seq, &reader );
231 for(int i = 0; i < seq->total; i++ )
232 {
233 const float* sdata = (float*)reader.ptr;
234 float* ddata = values->data.fl + cols_count*i;
235 uchar* dm = missing->data.ptr + cols_count*i;
236
237 for( int j = 0; j < cols_count; j++ )
238 {
239 ddata[j] = sdata[j];
240 dm[j] = ( fabs( MISS_VAL - sdata[j] ) <= FLT_EPSILON );
241 }
242 CV_NEXT_SEQ_ELEM( seq->elem_size, reader );
243 }
244
245 if ( cvNorm( missing, 0, CV_L1 ) <= FLT_EPSILON )
246 cvReleaseMat( &missing );
247
248 cvReleaseMemStorage( &storage );
249 delete []el_ptr;
250 return 0;
251}

276void CvMLData::str_to_flt_elem( const char* token, float& flt_elem, int& type)
277{
278
279 char* stopstring = NULL;
280 flt_elem = (float)strtod( token, &stopstring );
281 assert( stopstring );
282 type = CV_VAR_ORDERED;
283 if ( *stopstring == miss_ch && strlen(stopstring) == 1 ) // missed value
284 {
285 flt_elem = MISS_VAL;
286 type = CV_VAR_MISS;
287 }
288 else
289 {
290 if ( (*stopstring != 0) && (*stopstring != '\n') && (strcmp(stopstring, "\r\n") != 0) ) // class label
291 {
292 int idx = class_map[token];
293 if ( idx == 0)
294 {
295 total_class_count++;
296 idx = total_class_count;
297 class_map[token] = idx;
298 }
299 flt_elem = (float)idx;
300 type = CV_VAR_CATEGORICAL;
301 }
302 }

400void CvMLData::set_var_types( const char* str )
401{
402 CV_FUNCNAME( "CvMLData::set_var_types" );
403 __BEGIN__;
404
405 const char* ord = 0, *cat = 0;
406 int var_count = 0, set_var_type_count = 0;
407 if ( !values )
408 CV_ERROR( CV_StsInternal, "data is empty" );
409
410 var_count = values->cols;
411
412 assert( var_types );
413
414 ord = strstr( str, "ord" );
415 cat = strstr( str, "cat" );
416 if ( !ord && !cat )
417 CV_ERROR( CV_StsBadArg, "types string is not correct" );
418
419 if ( !ord && strlen(cat) == 3 ) // str == "cat"
420 {
421 cvSet( var_types, cvScalarAll(CV_VAR_CATEGORICAL) );
422 return;
423 }
424
425 if ( !cat && strlen(ord) == 3 ) // str == "ord"
426 {
427 cvSet( var_types, cvScalarAll(CV_VAR_ORDERED) );
428 return;
429 }
430
431 if ( ord ) // parse ord str
432 {
433 char* stopstring = NULL;
434 if ( ord[3] != '[')
435 CV_ERROR( CV_StsBadArg, "types string is not correct" );
436
437 ord += 4; // pass "ord["
438 do
439 {
440 int b1 = (int)strtod( ord, &stopstring );
441 if ( *stopstring == 0 || (*stopstring != ',' && *stopstring != ']' && *stopstring != '-') )
442 CV_ERROR( CV_StsBadArg, "types string is not correct" );
443 ord = stopstring + 1;
444 if ( (stopstring[0] == ',') || (stopstring[0] == ']'))
445 {
446 if ( var_types->data.ptr[b1] == CV_VAR_CATEGORICAL)
447 CV_ERROR( CV_StsBadArg, "it`s impossible to assign CV_VAR_ORDERED type to categorical variable" );
448 var_types->data.ptr[b1] = CV_VAR_ORDERED;
449 set_var_type_count++;
450 }
451 else
452 {
453 if ( stopstring[0] == '-')
454 {
455 int b2 = (int)strtod( ord, &stopstring);
456 if ( (*stopstring == 0) || (*stopstring != ',' && *stopstring != ']') )
457 CV_ERROR( CV_StsBadArg, "types string is not correct" );
458 ord = stopstring + 1;
459 for (int i = b1; i <= b2; i++)
460 {
461 if ( var_types->data.ptr[i] == CV_VAR_CATEGORICAL)
462 CV_ERROR( CV_StsBadArg, "it`s impossible to assign CV_VAR_ORDERED type to categorical variable" );
463 var_types->data.ptr[i] = CV_VAR_ORDERED;
464 }
465 set_var_type_count += b2 - b1 + 1;
466 }
467 else
468 CV_ERROR( CV_StsBadArg, "types string is not correct" );
469
470 }
471 }
472 while (*stopstring != ']');
473
474 if ( stopstring[1] != '\0' && stopstring[1] != ',')
475 CV_ERROR( CV_StsBadArg, "types string is not correct" );
476 }
477
478 if ( cat ) // parse cat str
479 {
480 char* stopstring = NULL;
481 if ( cat[3] != '[')
482 CV_ERROR( CV_StsBadArg, "types string is not correct" );
483
484 cat += 4; // pass "cat["
485 do
486 {
487 int b1 = (int)strtod( cat, &stopstring );
488 if ( *stopstring == 0 || (*stopstring != ',' && *stopstring != ']' && *stopstring != '-') )
489 CV_ERROR( CV_StsBadArg, "types string is not correct" );
490 cat = stopstring + 1;
491 if ( (stopstring[0] == ',') || (stopstring[0] == ']'))
492 {
493 var_types->data.ptr[b1] = CV_VAR_CATEGORICAL;
494 set_var_type_count++;
495 }
496 else
497 {
498 if ( stopstring[0] == '-')
499 {
500 int b2 = (int)strtod( cat, &stopstring);
501 if ( (*stopstring == 0) || (*stopstring != ',' && *stopstring != ']') )
502 CV_ERROR( CV_StsBadArg, "types string is not correct" );
503 cat = stopstring + 1;
504 for (int i = b1; i <= b2; i++)
505 var_types->data.ptr[i] = CV_VAR_CATEGORICAL;
506 set_var_type_count += b2 - b1 + 1;
507 }
508 else
509 CV_ERROR( CV_StsBadArg, "types string is not correct" );
510
511 }
512 }
513 while (*stopstring != ']');
514
515 if ( stopstring[1] != '\0' && stopstring[1] != ',')
516 CV_ERROR( CV_StsBadArg, "types string is not correct" );
517 }
518
519 if (set_var_type_count != var_count)
520 CV_ERROR( CV_StsBadArg, "types string is not correct" );
521
522 __END__;
523}

525const CvMat* CvMLData::get_var_types()
526{
527 CV_FUNCNAME( "CvMLData::get_var_types" );
528 __BEGIN__;
529
530 uchar *var_types_out_ptr = 0;
531 int avcount, vt_size;
532 if ( !values )
533 CV_ERROR( CV_StsInternal, "data is empty" );
534
535 assert( var_idx_mask );
536
537 avcount = cvFloor( cvNorm( var_idx_mask, 0, CV_L1 ) );
538 vt_size = avcount + (response_idx >= 0);
539
540 if ( avcount == values->cols || (avcount == values->cols-1 && response_idx == values->cols-1) )
541 return var_types;
542
543 if ( !var_types_out || ( var_types_out && var_types_out->cols != vt_size ) )
544 {
545 cvReleaseMat( &var_types_out );
546 var_types_out = cvCreateMat( 1, vt_size, CV_8UC1 );
547 }
548
549 var_types_out_ptr = var_types_out->data.ptr;
550 for( int i = 0; i < var_types->cols; i++)
551 {
552 if (i == response_idx || !var_idx_mask->data.ptr[i]) continue;
553 *var_types_out_ptr = var_types->data.ptr[i];
554 var_types_out_ptr++;
555 }
556 if ( response_idx >= 0 )
557 *var_types_out_ptr = var_types->data.ptr[response_idx];
558
559 __END__;
560
561 return var_types_out;
562}

593void CvMLData::set_train_test_split( const CvTrainTestSplit * spl)
594{
595 CV_FUNCNAME( "CvMLData::set_division" );
596 __BEGIN__;
597
598 int sample_count = 0;
599
600 if ( !values )
601 CV_ERROR( CV_StsInternal, "data is empty" );
602
603 sample_count = values->rows;
604
605 float train_sample_portion;
606
607 if (spl->train_sample_part_mode == CV_COUNT)
608 {
609 train_sample_count = spl->train_sample_part.count;
610 if (train_sample_count > sample_count)
611 CV_ERROR( CV_StsBadArg, "train samples count is not correct" );
612 train_sample_count = train_sample_count<=0 ? sample_count : train_sample_count;
613 }
614 else // dtype.train_sample_part_mode == CV_PORTION
615 {
616 train_sample_portion = spl->train_sample_part.portion;
617 if ( train_sample_portion > 1)
618 CV_ERROR( CV_StsBadArg, "train samples count is not correct" );
619 train_sample_portion = train_sample_portion <= FLT_EPSILON ||
620 1 - train_sample_portion <= FLT_EPSILON ? 1 : train_sample_portion;
621 train_sample_count = std::max(1, cvFloor( train_sample_portion * sample_count ));
622 }
623
624 if ( train_sample_count == sample_count )
625 {
626 free_train_test_idx();
627 return;
628 }
629
630 if ( train_sample_idx && train_sample_idx->cols != train_sample_count )
631 free_train_test_idx();
632
633 if ( !sample_idx)
634 {
635 int test_sample_count = sample_count- train_sample_count;
636 sample_idx = (int*)cvAlloc( sample_count * sizeof(sample_idx[0]) );
637 for (int i = 0; i < sample_count; i++ )
638 sample_idx[i] = i;
639 train_sample_idx = cvCreateMatHeader( 1, train_sample_count, CV_32SC1 );
640 *train_sample_idx = cvMat( 1, train_sample_count, CV_32SC1, &sample_idx[0] );
641
642 CV_Assert(test_sample_count > 0);
643 test_sample_idx = cvCreateMatHeader( 1, test_sample_count, CV_32SC1 );
644 *test_sample_idx = cvMat( 1, test_sample_count, CV_32SC1, &sample_idx[train_sample_count] );
645 }
646
647 mix = spl->mix;
648 if ( mix )
649 mix_train_and_test_idx();
650
651 __END__;
652}

703const CvMat* CvMLData::get_var_idx()
704{
705 CV_FUNCNAME( "CvMLData::get_var_idx" );
706 __BEGIN__;
707
708 int avcount = 0;
709
710 if ( !values )
711 CV_ERROR( CV_StsInternal, "data is empty" );
712
713 assert( var_idx_mask );
714
715 avcount = cvFloor( cvNorm( var_idx_mask, 0, CV_L1 ) );
716 int* vidx;
717
718 if ( avcount == values->cols )
719 return 0;
720
721 if ( !var_idx_out || ( var_idx_out && var_idx_out->cols != avcount ) )
722 {
723 cvReleaseMat( &var_idx_out );
724 var_idx_out = cvCreateMat( 1, avcount, CV_32SC1);
725 if ( response_idx >=0 )
726 var_idx_mask->data.ptr[response_idx] = 0;
727 }
728
729 vidx = var_idx_out->data.i;
730
731 for(int i = 0; i < var_idx_mask->cols; i++)
732 if ( var_idx_mask->data.ptr[i] )
733 {
734 *vidx = i;
735 vidx++;
736 }
737
738 __END__;
739
740 return var_idx_out;
741}

748void CvMLData::change_var_idx( int vi, bool state )
749{
750 CV_FUNCNAME( "CvMLData::change_var_idx" );
751 __BEGIN__;
752
753 int var_count = 0;
754
755 if ( !values )
756 CV_ERROR( CV_StsInternal, "data is empty" );
757
758 var_count = values->cols;
759
760 if ( vi < 0 || vi >= var_count)
761 CV_ERROR( CV_StsBadArg, "variable index is not correct" );
