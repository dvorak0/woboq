95void ForestTreeBestSplitFinder::operator()(const BlockedRange& range)
96{
97 int vi, vi1 = range.begin(), vi2 = range.end();
98 int n = node->sample_count;
99 CvDTreeTrainData* data = tree->get_data();
100 AutoBuffer<uchar> inn_buf(2*n*(sizeof(int) + sizeof(float)));
101
102 CvForestTree* ftree = (CvForestTree*)tree;
103 const CvMat* active_var_mask = ftree->forest->get_active_var_mask();
104
105 for( vi = vi1; vi < vi2; vi++ )
106 {
107 CvDTreeSplit *res;
108 int ci = data->var_type->data.i[vi];
109 if( node->num_valid[vi] <= 1
110 || (active_var_mask && !active_var_mask->data.ptr[vi]) )
111 continue;
112
113 if( data->is_classifier )
114 {
115 if( ci >= 0 )
116 res = ftree->find_split_cat_class( node, vi, bestSplit->quality, split, (uchar*)inn_buf );
117 else
118 res = ftree->find_split_ord_class( node, vi, bestSplit->quality, split, (uchar*)inn_buf );
119 }
120 else
121 {
122 if( ci >= 0 )
123 res = ftree->find_split_cat_reg( node, vi, bestSplit->quality, split, (uchar*)inn_buf );
124 else
125 res = ftree->find_split_ord_reg( node, vi, bestSplit->quality, split, (uchar*)inn_buf );
126 }
127
128 if( res && bestSplit->quality < split->quality )
129 memcpy( (CvDTreeSplit*)bestSplit, (CvDTreeSplit*)split, splitSize );
130 }
131}
132}

134CvDTreeSplit* CvForestTree::find_best_split( CvDTreeNode* node )
135{
136 CvMat* active_var_mask = 0;
137 if( forest )
138 {
139 int var_count;
140 CvRNG* rng = forest->get_rng();
141
142 active_var_mask = forest->get_active_var_mask();
143 var_count = active_var_mask->cols;
144
145 CV_Assert( var_count == data->var_count );
146
147 for( int vi = 0; vi < var_count; vi++ )
148 {
149 uchar temp;
150 int i1 = cvRandInt(rng) % var_count;
151 int i2 = cvRandInt(rng) % var_count;
152 CV_SWAP( active_var_mask->data.ptr[i1],
153 active_var_mask->data.ptr[i2], temp );
154 }
155 }
156
157 cv::ForestTreeBestSplitFinder finder( this, node );
158
159 cv::parallel_reduce(cv::BlockedRange(0, data->var_count), finder);
160
161 CvDTreeSplit *bestSplit = 0;
162 if( finder.bestSplit->quality > 0 )
163 {
164 bestSplit = data->new_split_cat( 0, -1.0f );
165 memcpy( bestSplit, finder.bestSplit, finder.splitSize );
166 }
167
168 return bestSplit;
169}

265bool CvRTrees::train( const CvMat* _train_data, int _tflag,
266 const CvMat* _responses, const CvMat* _var_idx,
267 const CvMat* _sample_idx, const CvMat* _var_type,
268 const CvMat* _missing_mask, CvRTParams params )
269{
270 clear();
271
272 CvDTreeParams tree_params( params.max_depth, params.min_sample_count,
273 params.regression_accuracy, params.use_surrogates, params.max_categories,
274 params.cv_folds, params.use_1se_rule, false, params.priors );
275
276 data = new CvDTreeTrainData();
277 data->set_data( _train_data, _tflag, _responses, _var_idx,
278 _sample_idx, _var_type, _missing_mask, tree_params, true);
279
280 int var_count = data->var_count;
281 if( params.nactive_vars > var_count )
282 params.nactive_vars = var_count;
283 else if( params.nactive_vars == 0 )
284 params.nactive_vars = (int)sqrt((double)var_count);
285 else if( params.nactive_vars < 0 )
286 CV_Error( CV_StsBadArg, "<nactive_vars> must be non-negative" );
287
288 // Create mask of active variables at the tree nodes
289 active_var_mask = cvCreateMat( 1, var_count, CV_8UC1 );
290 if( params.calc_var_importance )
291 {
292 var_importance = cvCreateMat( 1, var_count, CV_32FC1 );
293 cvZero(var_importance);
294 }
295 { // initialize active variables mask
296 CvMat submask1, submask2;
297 CV_Assert( (active_var_mask->cols >= 1) && (params.nactive_vars > 0) && (params.nactive_vars <= active_var_mask->cols) );
298 cvGetCols( active_var_mask, &submask1, 0, params.nactive_vars );
299 cvSet( &submask1, cvScalar(1) );
300 if( params.nactive_vars < active_var_mask->cols )
301 {
302 cvGetCols( active_var_mask, &submask2, params.nactive_vars, var_count );
303 cvZero( &submask2 );
304 }
305 }
306
307 return grow_forest( params.term_crit );
308}

323bool CvRTrees::grow_forest( const CvTermCriteria term_crit )
324{
325 CvMat* sample_idx_mask_for_tree = 0;
326 CvMat* sample_idx_for_tree = 0;
327
328 const int max_ntrees = term_crit.max_iter;
329 const double max_oob_err = term_crit.epsilon;
330
331 const int dims = data->var_count;
332 float maximal_response = 0;
333
334 CvMat* oob_sample_votes = 0;
335 CvMat* oob_responses = 0;
336
337 float* oob_samples_perm_ptr= 0;
338
339 float* samples_ptr = 0;
340 uchar* missing_ptr = 0;
341 float* true_resp_ptr = 0;
342 bool is_oob_or_vimportance = (max_oob_err > 0 && term_crit.type != CV_TERMCRIT_ITER) || var_importance;
343
344 // oob_predictions_sum[i] = sum of predicted values for the i-th sample
345 // oob_num_of_predictions[i] = number of summands
346 // (number of predictions for the i-th sample)
347 // initialize these variable to avoid warning C4701
348 CvMat oob_predictions_sum = cvMat( 1, 1, CV_32FC1 );
349 CvMat oob_num_of_predictions = cvMat( 1, 1, CV_32FC1 );
350
351 nsamples = data->sample_count;
352 nclasses = data->get_num_classes();
353
354 if ( is_oob_or_vimportance )
355 {
356 if( data->is_classifier )
357 {
358 oob_sample_votes = cvCreateMat( nsamples, nclasses, CV_32SC1 );
359 cvZero(oob_sample_votes);
360 }
361 else
362 {
363 // oob_responses[0,i] = oob_predictions_sum[i]
364 // = sum of predicted values for the i-th sample
365 // oob_responses[1,i] = oob_num_of_predictions[i]
366 // = number of summands (number of predictions for the i-th sample)
367 oob_responses = cvCreateMat( 2, nsamples, CV_32FC1 );
368 cvZero(oob_responses);
369 cvGetRow( oob_responses, &oob_predictions_sum, 0 );
370 cvGetRow( oob_responses, &oob_num_of_predictions, 1 );
371 }
372
373 oob_samples_perm_ptr = (float*)cvAlloc( sizeof(float)*nsamples*dims );
374 samples_ptr = (float*)cvAlloc( sizeof(float)*nsamples*dims );
375 missing_ptr = (uchar*)cvAlloc( sizeof(uchar)*nsamples*dims );
376 true_resp_ptr = (float*)cvAlloc( sizeof(float)*nsamples );
377
378 data->get_vectors( 0, samples_ptr, missing_ptr, true_resp_ptr );
379
380 double minval, maxval;
381 CvMat responses = cvMat(1, nsamples, CV_32FC1, true_resp_ptr);
382 cvMinMaxLoc( &responses, &minval, &maxval );
383 maximal_response = (float)MAX( MAX( fabs(minval), fabs(maxval) ), 0 );
384 }
385
386 trees = (CvForestTree**)cvAlloc( sizeof(trees[0])*max_ntrees );
387 memset( trees, 0, sizeof(trees[0])*max_ntrees );
388
389 sample_idx_mask_for_tree = cvCreateMat( 1, nsamples, CV_8UC1 );
390 sample_idx_for_tree = cvCreateMat( 1, nsamples, CV_32SC1 );
391
392 ntrees = 0;
393 while( ntrees < max_ntrees )
394 {
395 int i, oob_samples_count = 0;
396 double ncorrect_responses = 0; // used for estimation of variable importance
397 CvForestTree* tree = 0;
398
399 cvZero( sample_idx_mask_for_tree );
400 for(i = 0; i < nsamples; i++ ) //form sample for creation one tree
401 {
402 int idx = (*rng)(nsamples);
403 sample_idx_for_tree->data.i[i] = idx;
404 sample_idx_mask_for_tree->data.ptr[idx] = 0xFF;
405 }
406
407 trees[ntrees] = new CvForestTree();
408 tree = trees[ntrees];
409 tree->train( data, sample_idx_for_tree, this );
410
411 if ( is_oob_or_vimportance )
412 {
413 CvMat sample, missing;
414 // form array of OOB samples indices and get these samples
415 sample = cvMat( 1, dims, CV_32FC1, samples_ptr );
416 missing = cvMat( 1, dims, CV_8UC1, missing_ptr );
417
418 oob_error = 0;
419 for( i = 0; i < nsamples; i++,
420 sample.data.fl += dims, missing.data.ptr += dims )
421 {
422 CvDTreeNode* predicted_node = 0;
423 // check if the sample is OOB
424 if( sample_idx_mask_for_tree->data.ptr[i] )
425 continue;
426
427 // predict oob samples
428 if( !predicted_node )
429 predicted_node = tree->predict(&sample, &missing, true);
430
431 if( !data->is_classifier ) //regression
432 {
433 double avg_resp, resp = predicted_node->value;
434 oob_predictions_sum.data.fl[i] += (float)resp;
435 oob_num_of_predictions.data.fl[i] += 1;
436
437 // compute oob error
438 avg_resp = oob_predictions_sum.data.fl[i]/oob_num_of_predictions.data.fl[i];
439 avg_resp -= true_resp_ptr[i];
440 oob_error += avg_resp*avg_resp;
441 resp = (resp - true_resp_ptr[i])/maximal_response;
442 ncorrect_responses += exp( -resp*resp );
443 }
444 else //classification
445 {
446 double prdct_resp;
447 CvPoint max_loc;
448 CvMat votes;
449
450 cvGetRow(oob_sample_votes, &votes, i);
451 votes.data.i[predicted_node->class_idx]++;
452
453 // compute oob error
454 cvMinMaxLoc( &votes, 0, 0, 0, &max_loc );
455
456 prdct_resp = data->cat_map->data.i[max_loc.x];
457 oob_error += (fabs(prdct_resp - true_resp_ptr[i]) < FLT_EPSILON) ? 0 : 1;
458
459 ncorrect_responses += cvRound(predicted_node->value - true_resp_ptr[i]) == 0;
460 }
461 oob_samples_count++;
462 }
463 if( oob_samples_count > 0 )
464 oob_error /= (double)oob_samples_count;
465
466 // estimate variable importance
467 if( var_importance && oob_samples_count > 0 )
468 {
469 int m;
470
471 memcpy( oob_samples_perm_ptr, samples_ptr, dims*nsamples*sizeof(float));
472 for( m = 0; m < dims; m++ )
473 {
474 double ncorrect_responses_permuted = 0;
475 // randomly permute values of the m-th variable in the oob samples
476 float* mth_var_ptr = oob_samples_perm_ptr + m;
477
478 for( i = 0; i < nsamples; i++ )
479 {
480 int i1, i2;
481 float temp;
482
483 if( sample_idx_mask_for_tree->data.ptr[i] ) //the sample is not OOB
484 continue;
485 i1 = (*rng)(nsamples);
486 i2 = (*rng)(nsamples);
487 CV_SWAP( mth_var_ptr[i1*dims], mth_var_ptr[i2*dims], temp );
488
489 // turn values of (m-1)-th variable, that were permuted
490 // at the previous iteration, untouched
491 if( m > 1 )
492 oob_samples_perm_ptr[i*dims+m-1] = samples_ptr[i*dims+m-1];
493 }
494
495 // predict "permuted" cases and calculate the number of votes for the
496 // correct class in the variable-m-permuted oob data
497 sample = cvMat( 1, dims, CV_32FC1, oob_samples_perm_ptr );
498 missing = cvMat( 1, dims, CV_8UC1, missing_ptr );
499 for( i = 0; i < nsamples; i++,
500 sample.data.fl += dims, missing.data.ptr += dims )
501 {
502 double predct_resp, true_resp;
503
504 if( sample_idx_mask_for_tree->data.ptr[i] ) //the sample is not OOB
505 continue;
506
507 predct_resp = tree->predict(&sample, &missing, true)->value;
508 true_resp = true_resp_ptr[i];
509 if( data->is_classifier )
510 ncorrect_responses_permuted += cvRound(true_resp - predct_resp) == 0;
511 else
512 {
513 true_resp = (true_resp - predct_resp)/maximal_response;
514 ncorrect_responses_permuted += exp( -true_resp*true_resp );
515 }
516 }
517 var_importance->data.fl[m] += (float)(ncorrect_responses
518 - ncorrect_responses_permuted);
519 }
520 }
521 }
522 ntrees++;
523 if( term_crit.type != CV_TERMCRIT_ITER && oob_error < max_oob_err )
524 break;
525 }
526
527 if( var_importance )
528 {
529 for ( int vi = 0; vi < var_importance->cols; vi++ )
530 var_importance->data.fl[vi] = ( var_importance->data.fl[vi] > 0 ) ?
531 var_importance->data.fl[vi] : 0;
532 cvNormalize( var_importance, var_importance, 1., 0, CV_L1 );
533 }
534
535 cvFree( &oob_samples_perm_ptr );
536 cvFree( &samples_ptr );
537 cvFree( &missing_ptr );
538 cvFree( &true_resp_ptr );
539
540 cvReleaseMat( &sample_idx_mask_for_tree );
541 cvReleaseMat( &sample_idx_for_tree );
542
543 cvReleaseMat( &oob_sample_votes );
544 cvReleaseMat( &oob_responses );
545
546 return true;
547}

556float CvRTrees::get_proximity( const CvMat* sample1, const CvMat* sample2,
557 const CvMat* missing1, const CvMat* missing2 ) const
558{
559 float result = 0;
560
561 for( int i = 0; i < ntrees; i++ )
562 result += trees[i]->predict( sample1, missing1 ) ==
563 trees[i]->predict( sample2, missing2 ) ? 1 : 0;
564 result = result/(float)ntrees;
565
566 return result;
567}

569float CvRTrees::calc_error( CvMLData* _data, int type , std::vector<float> *resp )
570{
571 float err = 0;
572 const CvMat* values = _data->get_values();
573 const CvMat* response = _data->get_responses();
574 const CvMat* missing = _data->get_missing();
575 const CvMat* sample_idx = (type == CV_TEST_ERROR) ? _data->get_test_sample_idx() : _data->get_train_sample_idx();
576 const CvMat* var_types = _data->get_var_types();
577 int* sidx = sample_idx ? sample_idx->data.i : 0;
578 int r_step = CV_IS_MAT_CONT(response->type) ?
579 1 : response->step / CV_ELEM_SIZE(response->type);
580 bool is_classifier = var_types->data.ptr[var_types->cols-1] == CV_VAR_CATEGORICAL;
581 int sample_count = sample_idx ? sample_idx->cols : 0;
582 sample_count = (type == CV_TRAIN_ERROR && sample_count == 0) ? values->rows : sample_count;
583 float* pred_resp = 0;
584 if( resp && (sample_count > 0) )
585 {
586 resp->resize( sample_count );
587 pred_resp = &((*resp)[0]);
588 }
589 if ( is_classifier )
590 {
591 for( int i = 0; i < sample_count; i++ )
592 {
593 CvMat sample, miss;
594 int si = sidx ? sidx[i] : i;
595 cvGetRow( values, &sample, si );
596 if( missing )
597 cvGetRow( missing, &miss, si );
598 float r = (float)predict( &sample, missing ? &miss : 0 );
599 if( pred_resp )
600 pred_resp[i] = r;
601 int d = fabs((double)r - response->data.fl[si*r_step]) <= FLT_EPSILON ? 0 : 1;
602 err += d;
603 }
604 err = sample_count ? err / (float)sample_count * 100 : -FLT_MAX;
605 }
606 else
607 {
608 for( int i = 0; i < sample_count; i++ )
609 {
610 CvMat sample, miss;
611 int si = sidx ? sidx[i] : i;
612 cvGetRow( values, &sample, si );
613 if( missing )
614 cvGetRow( missing, &miss, si );
615 float r = (float)predict( &sample, missing ? &miss : 0 );
616 if( pred_resp )
617 pred_resp[i] = r;
618 float d = r - response->data.fl[si*r_step];
619 err += d*d;
620 }
621 err = sample_count ? err / (float)sample_count : -FLT_MAX;
622 }
623 return err;
624}
625
626float CvRTrees::get_train_error()
627{
628 float err = -1;
629
630 int sample_count = data->sample_count;
631 int var_count = data->var_count;
632
633 float *values_ptr = (float*)cvAlloc( sizeof(float)*sample_count*var_count );
634 uchar *missing_ptr = (uchar*)cvAlloc( sizeof(uchar)*sample_count*var_count );
635 float *responses_ptr = (float*)cvAlloc( sizeof(float)*sample_count );
636
637 data->get_vectors( 0, values_ptr, missing_ptr, responses_ptr);
638
639 if (data->is_classifier)
640 {
641 int err_count = 0;
642 float *vp = values_ptr;
643 uchar *mp = missing_ptr;
644 for (int si = 0; si < sample_count; si++, vp += var_count, mp += var_count)
645 {
646 CvMat sample = cvMat( 1, var_count, CV_32FC1, vp );
647 CvMat missing = cvMat( 1, var_count, CV_8UC1, mp );
648 float r = predict( &sample, &missing );
649 if (fabs(r - responses_ptr[si]) >= FLT_EPSILON)
650 err_count++;
651 }
652 err = (float)err_count / (float)sample_count;
653 }
654 else
655 CV_Error( CV_StsBadArg, "This method is not supported for regression problems" );
656
657 cvFree( &values_ptr );
658 cvFree( &missing_ptr );
659 cvFree( &responses_ptr );
660
661 return err;
662}

664
665float CvRTrees::predict( const CvMat* sample, const CvMat* missing ) const
666{
667 double result = -1;
668 int k;
669
670 if( nclasses > 0 ) //classification
671 {
672 int max_nvotes = 0;
673 cv::AutoBuffer<int> _votes(nclasses);
674 int* votes = _votes;
675 memset( votes, 0, sizeof(*votes)*nclasses );
676 for( k = 0; k < ntrees; k++ )
677 {
678 CvDTreeNode* predicted_node = trees[k]->predict( sample, missing );
679 int nvotes;
680 int class_idx = predicted_node->class_idx;
681 CV_Assert( 0 <= class_idx && class_idx < nclasses );
682
683 nvotes = ++votes[class_idx];
684 if( nvotes > max_nvotes )
685 {
686 max_nvotes = nvotes;
687 result = predicted_node->value;
688 }
689 }
690 }
691 else // regression
692 {
693 result = 0;
694 for( k = 0; k < ntrees; k++ )
695 result += trees[k]->predict( sample, missing )->value;
696 result /= (double)ntrees;
697 }
698
699 return (float)result;
700}
701
702float CvRTrees::predict_prob( const CvMat* sample, const CvMat* missing) const
703{
704 if( nclasses == 2 ) //classification
705 {
706 cv::AutoBuffer<int> _votes(nclasses);
707 int* votes = _votes;
708 memset( votes, 0, sizeof(*votes)*nclasses );
709 for( int k = 0; k < ntrees; k++ )
710 {
711 CvDTreeNode* predicted_node = trees[k]->predict( sample, missing );
712 int class_idx = predicted_node->class_idx;
713 CV_Assert( 0 <= class_idx && class_idx < nclasses );
714
715 ++votes[class_idx];
716 }
717
718 return float(votes[1])/ntrees;
719 }
720 else // regression
721 CV_Error(CV_StsBadArg, "This function works for binary classification problems only...");
722
723 return -1;
724}

726void CvRTrees::write( CvFileStorage* fs, const char* name ) const
727{
728 int k;
729
730 if( ntrees < 1 || !trees || nsamples < 1 )
731 CV_Error( CV_StsBadArg, "Invalid CvRTrees object" );
732
733 std::string modelNodeName = this->getName();
734 cvStartWriteStruct( fs, name, CV_NODE_MAP, modelNodeName.c_str() );
735
736 cvWriteInt( fs, "nclasses", nclasses );
737 cvWriteInt( fs, "nsamples", nsamples );
738 cvWriteInt( fs, "nactive_vars", (int)cvSum(active_var_mask).val[0] );
739 cvWriteReal( fs, "oob_error", oob_error );
740
741 if( var_importance )
742 cvWrite( fs, "var_importance", var_importance );
743
744 cvWriteInt( fs, "ntrees", ntrees );
745
746 data->write_params( fs );
747
748 cvStartWriteStruct( fs, "trees", CV_NODE_SEQ );
749
750 for( k = 0; k < ntrees; k++ )
751 {
752 cvStartWriteStruct( fs, 0, CV_NODE_MAP );
753 trees[k]->write( fs );
754 cvEndWriteStruct( fs );
755 }
756
757 cvEndWriteStruct( fs ); //trees
758 cvEndWriteStruct( fs ); //CV_TYPE_NAME_ML_RTREES
759}

762void CvRTrees::read( CvFileStorage* fs, CvFileNode* fnode )
763{
764 int nactive_vars, var_count, k;
765 CvSeqReader reader;
766 CvFileNode* trees_fnode = 0;
767
768 clear();
769
770 nclasses = cvReadIntByName( fs, fnode, "nclasses", -1 );
771 nsamples = cvReadIntByName( fs, fnode, "nsamples" );
772 nactive_vars = cvReadIntByName( fs, fnode, "nactive_vars", -1 );
773 oob_error = cvReadRealByName(fs, fnode, "oob_error", -1 );
774 ntrees = cvReadIntByName( fs, fnode, "ntrees", -1 );
775
776 var_importance = (CvMat*)cvReadByName( fs, fnode, "var_importance" );
777
778 if( nclasses < 0 || nsamples <= 0 || nactive_vars < 0 || oob_error < 0 || ntrees <= 0)
779 CV_Error( CV_StsParseError, "Some <nclasses>, <nsamples>, <var_count>, "
780 "<nactive_vars>, <oob_error>, <ntrees> of tags are missing" );
781
782 rng = &cv::theRNG();
783
784 trees = (CvForestTree**)cvAlloc( sizeof(trees[0])*ntrees );
785 memset( trees, 0, sizeof(trees[0])*ntrees );
786
787 data = new CvDTreeTrainData();
788 data->read_params( fs, fnode );
789 data->shared = true;
790
791 trees_fnode = cvGetFileNodeByName( fs, fnode, "trees" );
792 if( !trees_fnode || !CV_NODE_IS_SEQ(trees_fnode->tag) )
793 CV_Error( CV_StsParseError, "<trees> tag is missing" );
794
795 cvStartReadSeq( trees_fnode->data.seq, &reader );
796 if( reader.seq->total != ntrees )
797 CV_Error( CV_StsParseError,
798 "<ntrees> is not equal to the number of trees saved in file" );
799
800 for( k = 0; k < ntrees; k++ )
801 {
802 trees[k] = new CvForestTree();
803 trees[k]->read( fs, (CvFileNode*)reader.ptr, this, data );
804 CV_NEXT_SEQ_ELEM( reader.seq->elem_size, reader );
805 }
806
807 var_count = data->var_count;
808 active_var_mask = cvCreateMat( 1, var_count, CV_8UC1 );
809 {
810 // initialize active variables mask
811 CvMat submask1;
812 cvGetCols( active_var_mask, &submask1, 0, nactive_vars );
813 cvSet( &submask1, cvScalar(1) );
814
815 if( nactive_vars < var_count )
816 {
817 CvMat submask2;
818 cvGetCols( active_var_mask, &submask2, nactive_vars, var_count );
819 cvZero( &submask2 );
820 }
821 }
822}
