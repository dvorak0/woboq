121CvParamGrid CvSVM::get_default_grid( int param_id )
122{
123 CvParamGrid grid;
124 if( param_id == CvSVM::C )
125 {
126 grid.min_val = 0.1;
127 grid.max_val = 500;
128 grid.step = 5; // total iterations = 5
129 }
130 else if( param_id == CvSVM::GAMMA )
131 {
132 grid.min_val = 1e-5;
133 grid.max_val = 0.6;
134 grid.step = 15; // total iterations = 4
135 }
136 else if( param_id == CvSVM::P )
137 {
138 grid.min_val = 0.01;
139 grid.max_val = 100;
140 grid.step = 7; // total iterations = 4
141 }
142 else if( param_id == CvSVM::NU )
143 {
144 grid.min_val = 0.01;
145 grid.max_val = 0.2;
146 grid.step = 3; // total iterations = 3
147 }
148 else if( param_id == CvSVM::COEF )
149 {
150 grid.min_val = 0.1;
151 grid.max_val = 300;
152 grid.step = 14; // total iterations = 3
153 }
154 else if( param_id == CvSVM::DEGREE )
155 {
156 grid.min_val = 0.01;
157 grid.max_val = 4;
158 grid.step = 7; // total iterations = 3
159 }
160 else
161 cvError( CV_StsBadArg, "CvSVM::get_default_grid", "Invalid type of parameter "
162 "(use one of CvSVM::C, CvSVM::GAMMA et al.)", __FILE__, __LINE__ );
163 return grid;
164}

229void CvSVMKernel::calc_non_rbf_base( int vcount, int var_count, const float** vecs,
230 const float* another, Qfloat* results,
231 double alpha, double beta )
232{
233 int j, k;
234 for( j = 0; j < vcount; j++ )
235 {
236 const float* sample = vecs[j];
237 double s = 0;
238 for( k = 0; k <= var_count - 4; k += 4 )
239 s += sample[k]*another[k] + sample[k+1]*another[k+1] +
240 sample[k+2]*another[k+2] + sample[k+3]*another[k+3];
241 for( ; k < var_count; k++ )
242 s += sample[k]*another[k];
243 results[j] = (Qfloat)(s*alpha + beta);
244 }
245}

265void CvSVMKernel::calc_sigmoid( int vcount, int var_count, const float** vecs,
266 const float* another, Qfloat* results )
267{
268 int j;
269 calc_non_rbf_base( vcount, var_count, vecs, another, results,
270 -2*params->gamma, -2*params->coef0 );
271 // TODO: speedup this
272 for( j = 0; j < vcount; j++ )
273 {
274 Qfloat t = results[j];
275 double e = exp(-fabs(t));
276 if( t > 0 )
277 results[j] = (Qfloat)((1. - e)/(1. + e));
278 else
279 results[j] = (Qfloat)((e - 1.)/(e + 1.));
280 }
281}

284void CvSVMKernel::calc_rbf( int vcount, int var_count, const float** vecs,
285 const float* another, Qfloat* results )
286{
287 CvMat R = cvMat( 1, vcount, QFLOAT_TYPE, results );
288 double gamma = -params->gamma;
289 int j, k;
290
291 for( j = 0; j < vcount; j++ )
292 {
293 const float* sample = vecs[j];
294 double s = 0;
295
296 for( k = 0; k <= var_count - 4; k += 4 )
297 {
298 double t0 = sample[k] - another[k];
299 double t1 = sample[k+1] - another[k+1];
300
301 s += t0*t0 + t1*t1;
302
303 t0 = sample[k+2] - another[k+2];
304 t1 = sample[k+3] - another[k+3];
305
306 s += t0*t0 + t1*t1;
307 }
308
309 for( ; k < var_count; k++ )
310 {
311 double t0 = sample[k] - another[k];
312 s += t0*t0;
313 }
314 results[j] = (Qfloat)(s*gamma);
315 }
316
317 if( vcount > 0 )
318 cvExp( &R, &R );
319}

322void CvSVMKernel::calc( int vcount, int var_count, const float** vecs,
323 const float* another, Qfloat* results )
324{
325 const Qfloat max_val = (Qfloat)(FLT_MAX*1e-3);
326 int j;
327 (this->*calc_func)( vcount, var_count, vecs, another, results );
328 for( j = 0; j < vcount; j++ )
329 {
330 if( results[j] > max_val )
331 results[j] = max_val;
332 }
333}

336// Generalized SMO+SVMlight algorithm
337// Solves:
338//
339// min [0.5(\alpha^T Q \alpha) + b^T \alpha]
340//
341// y^T \alpha = \delta
342// y_i = +1 or -1
343// 0 <= alpha_i <= Cp for y_i = 1
344// 0 <= alpha_i <= Cn for y_i = -1
345//
346// Given:
347//
348// Q, b, y, Cp, Cn, and an initial feasible point \alpha
349// l is the size of vectors and matrices
350// eps is the stopping criterion
351//
352// solution will be put in \alpha, objective value will be put in obj
353//
354

397bool CvSVMSolver::create( int _sample_count, int _var_count, const float** _samples, schar* _y,
398 int _alpha_count, double* _alpha, double _Cp, double _Cn,
399 CvMemStorage* _storage, CvSVMKernel* _kernel, GetRow _get_row,
400 SelectWorkingSet _select_working_set, CalcRho _calc_rho )
401{
402 bool ok = false;
403 int i, svm_type;
404
405 CV_FUNCNAME( "CvSVMSolver::create" );
406
407 __BEGIN__;
408
409 int rows_hdr_size;
410
411 clear();
412
413 sample_count = _sample_count;
414 var_count = _var_count;
415 samples = _samples;
416 y = _y;
417 alpha_count = _alpha_count;
418 alpha = _alpha;
419 kernel = _kernel;
420
421 C[0] = _Cn;
422 C[1] = _Cp;
423 eps = kernel->params->term_crit.epsilon;
424 max_iter = kernel->params->term_crit.max_iter;
425 storage = cvCreateChildMemStorage( _storage );
426
427 b = (double*)cvMemStorageAlloc( storage, alpha_count*sizeof(b[0]));
428 alpha_status = (schar*)cvMemStorageAlloc( storage, alpha_count*sizeof(alpha_status[0]));
429 G = (double*)cvMemStorageAlloc( storage, alpha_count*sizeof(G[0]));
430 for( i = 0; i < 2; i++ )
431 buf[i] = (Qfloat*)cvMemStorageAlloc( storage, sample_count*2*sizeof(buf[i][0]) );
432 svm_type = kernel->params->svm_type;
433
434 select_working_set_func = _select_working_set;
435 if( !select_working_set_func )
436 select_working_set_func = svm_type == CvSVM::NU_SVC || svm_type == CvSVM::NU_SVR ?
437 &CvSVMSolver::select_working_set_nu_svm : &CvSVMSolver::select_working_set;
438
439 calc_rho_func = _calc_rho;
440 if( !calc_rho_func )
441 calc_rho_func = svm_type == CvSVM::NU_SVC || svm_type == CvSVM::NU_SVR ?
442 &CvSVMSolver::calc_rho_nu_svm : &CvSVMSolver::calc_rho;
443
444 get_row_func = _get_row;
445 if( !get_row_func )
446 get_row_func = params->svm_type == CvSVM::EPS_SVR ||
447 params->svm_type == CvSVM::NU_SVR ? &CvSVMSolver::get_row_svr :
448 params->svm_type == CvSVM::C_SVC ||
449 params->svm_type == CvSVM::NU_SVC ? &CvSVMSolver::get_row_svc :
450 &CvSVMSolver::get_row_one_class;
451
452 cache_line_size = sample_count*sizeof(Qfloat);
453 // cache size = max(num_of_samples^2*sizeof(Qfloat)*0.25, 64Kb)
454 // (assuming that for large training sets ~25% of Q matrix is used)
455 cache_size = MAX( cache_line_size*sample_count/4, CV_SVM_MIN_CACHE_SIZE );
456
457 // the size of Q matrix row headers
458 rows_hdr_size = sample_count*sizeof(rows[0]);
459 if( rows_hdr_size > storage->block_size )
460 CV_ERROR( CV_StsOutOfRange, "Too small storage block size" );
461
462 lru_list.prev = lru_list.next = &lru_list;
463 rows = (CvSVMKernelRow*)cvMemStorageAlloc( storage, rows_hdr_size );
464 memset( rows, 0, rows_hdr_size );
465
466 ok = true;
467
468 __END__;
469
470 return ok;
471}

474float* CvSVMSolver::get_row_base( int i, bool* _existed )
475{
476 int i1 = i < sample_count ? i : i - sample_count;
477 CvSVMKernelRow* row = rows + i1;
478 bool existed = row->data != 0;
479 Qfloat* data;
480
481 if( existed || cache_size <= 0 )
482 {
483 CvSVMKernelRow* del_row = existed ? row : lru_list.prev;
484 data = del_row->data;
485 assert( data != 0 );
486
487 // delete row from the LRU list
488 del_row->data = 0;
489 del_row->prev->next = del_row->next;
490 del_row->next->prev = del_row->prev;
491 }
492 else
493 {
494 data = (Qfloat*)cvMemStorageAlloc( storage, cache_line_size );
495 cache_size -= cache_line_size;
496 }
497
498 // insert row into the LRU list
499 row->data = data;
500 row->prev = &lru_list;
501 row->next = lru_list.next;
502 row->prev->next = row->next->prev = row;
503
504 if( !existed )
505 {
506 kernel->calc( sample_count, var_count, samples, samples[i1], row->data );
507 }
508
509 if( _existed )
510 *_existed = existed;
511
512 return row->data;
513}

532 row[j] = -_y[j]*row[j];
533 }

595bool CvSVMSolver::solve_generic( CvSVMSolutionInfo& si )
596{
597 int iter = 0;
598 int i, j, k;
599
600 // 1. initialize gradient and alpha status
601 for( i = 0; i < alpha_count; i++ )
602 {
603 update_alpha_status(i);
604 G[i] = b[i];
605 if( fabs(G[i]) > 1e200 )
606 return false;
607 }
608
609 for( i = 0; i < alpha_count; i++ )
610 {
611 if( !is_lower_bound(i) )
612 {
613 const Qfloat *Q_i = get_row( i, buf[0] );
614 double alpha_i = alpha[i];
615
616 for( j = 0; j < alpha_count; j++ )
617 G[j] += alpha_i*Q_i[j];
618 }
619 }
620
621 // 2. optimization loop
622 for(;;)
623 {
624 const Qfloat *Q_i, *Q_j;
625 double C_i, C_j;
626 double old_alpha_i, old_alpha_j, alpha_i, alpha_j;
627 double delta_alpha_i, delta_alpha_j;
628
629#ifdef _DEBUG
630 for( i = 0; i < alpha_count; i++ )
631 {
632 if( fabs(G[i]) > 1e+300 )
633 return false;
634
635 if( fabs(alpha[i]) > 1e16 )
636 return false;
637 }
638#endif
639
640 if( (this->*select_working_set_func)( i, j ) != 0 || iter++ >= max_iter )
641 break;
642
643 Q_i = get_row( i, buf[0] );
644 Q_j = get_row( j, buf[1] );
645
646 C_i = get_C(i);
647 C_j = get_C(j);
648
649 alpha_i = old_alpha_i = alpha[i];
650 alpha_j = old_alpha_j = alpha[j];
651
652 if( y[i] != y[j] )
653 {
654 double denom = Q_i[i]+Q_j[j]+2*Q_i[j];
655 double delta = (-G[i]-G[j])/MAX(fabs(denom),FLT_EPSILON);
656 double diff = alpha_i - alpha_j;
657 alpha_i += delta;
658 alpha_j += delta;
659
660 if( diff > 0 && alpha_j < 0 )
661 {
662 alpha_j = 0;
663 alpha_i = diff;
664 }
665 else if( diff <= 0 && alpha_i < 0 )
666 {
667 alpha_i = 0;
668 alpha_j = -diff;
669 }
670
671 if( diff > C_i - C_j && alpha_i > C_i )
672 {
673 alpha_i = C_i;
674 alpha_j = C_i - diff;
675 }
676 else if( diff <= C_i - C_j && alpha_j > C_j )
677 {
678 alpha_j = C_j;
679 alpha_i = C_j + diff;
680 }
681 }
682 else
683 {
684 double denom = Q_i[i]+Q_j[j]-2*Q_i[j];
685 double delta = (G[i]-G[j])/MAX(fabs(denom),FLT_EPSILON);
686 double sum = alpha_i + alpha_j;
687 alpha_i -= delta;
688 alpha_j += delta;
689
690 if( sum > C_i && alpha_i > C_i )
691 {
692 alpha_i = C_i;
693 alpha_j = sum - C_i;
694 }
695 else if( sum <= C_i && alpha_j < 0)
696 {
697 alpha_j = 0;
698 alpha_i = sum;
699 }
700
701 if( sum > C_j && alpha_j > C_j )
702 {
703 alpha_j = C_j;
704 alpha_i = sum - C_j;
705 }
706 else if( sum <= C_j && alpha_i < 0 )
707 {
708 alpha_i = 0;
709 alpha_j = sum;
710 }
711 }
712
713 // update alpha
714 alpha[i] = alpha_i;
715 alpha[j] = alpha_j;
716 update_alpha_status(i);
717 update_alpha_status(j);
718
719 // update G
720 delta_alpha_i = alpha_i - old_alpha_i;
721 delta_alpha_j = alpha_j - old_alpha_j;
722
723 for( k = 0; k < alpha_count; k++ )
724 G[k] += Q_i[k]*delta_alpha_i + Q_j[k]*delta_alpha_j;
725 }
726
727 // calculate rho
728 (this->*calc_rho_func)( si.rho, si.r );
729
730 // calculate objective value
731 for( i = 0, si.obj = 0; i < alpha_count; i++ )
732 si.obj += alpha[i] * (G[i] + b[i]);
733
734 si.obj *= 0.5;
735
736 si.upper_bound_p = C[1];
737 si.upper_bound_n = C[0];
738
739 return true;
740}

743// return 1 if already optimal, return 0 otherwise
744bool
745CvSVMSolver::select_working_set( int& out_i, int& out_j )
746{
747 // return i,j which maximize -grad(f)^T d , under constraint
748 // if alpha_i == C, d != +1
749 // if alpha_i == 0, d != -1
750 double Gmax1 = -DBL_MAX; // max { -grad(f)_i * d | y_i*d = +1 }
751 int Gmax1_idx = -1;
752
753 double Gmax2 = -DBL_MAX; // max { -grad(f)_i * d | y_i*d = -1 }
754 int Gmax2_idx = -1;
755
756 int i;
757
758 for( i = 0; i < alpha_count; i++ )
759 {
760 double t;
761
762 if( y[i] > 0 ) // y = +1
763 {
764 if( !is_upper_bound(i) && (t = -G[i]) > Gmax1 ) // d = +1
765 {
766 Gmax1 = t;
767 Gmax1_idx = i;
768 }
769 if( !is_lower_bound(i) && (t = G[i]) > Gmax2 ) // d = -1
770 {
771 Gmax2 = t;
772 Gmax2_idx = i;
773 }
774 }
775 else // y = -1
776 {
777 if( !is_upper_bound(i) && (t = -G[i]) > Gmax2 ) // d = +1
778 {
779 Gmax2 = t;
780 Gmax2_idx = i;
781 }
782 if( !is_lower_bound(i) && (t = G[i]) > Gmax1 ) // d = -1
783 {
784 Gmax1 = t;
785 Gmax1_idx = i;
786 }
787 }
788 }
789
790 out_i = Gmax1_idx;
791 out_j = Gmax2_idx;
792
793 return Gmax1 + Gmax2 < eps;
794}

797void
798CvSVMSolver::calc_rho( double& rho, double& r )
799{
800 int i, nr_free = 0;
801 double ub = DBL_MAX, lb = -DBL_MAX, sum_free = 0;
802
803 for( i = 0; i < alpha_count; i++ )
804 {
805 double yG = y[i]*G[i];
806
807 if( is_lower_bound(i) )
808 {
809 if( y[i] > 0 )
810 ub = MIN(ub,yG);
811 else
812 lb = MAX(lb,yG);
813 }
814 else if( is_upper_bound(i) )
815 {
816 if( y[i] < 0)
817 ub = MIN(ub,yG);
818 else
819 lb = MAX(lb,yG);
820 }
821 else
822 {
823 ++nr_free;
824 sum_free += yG;
825 }
826 }
827
828 rho = nr_free > 0 ? sum_free/nr_free : (ub + lb)*0.5;
829 r = 0;
830}

902void
903CvSVMSolver::calc_rho_nu_svm( double& rho, double& r )
904{
905 int nr_free1 = 0, nr_free2 = 0;
906 double ub1 = DBL_MAX, ub2 = DBL_MAX;
907 double lb1 = -DBL_MAX, lb2 = -DBL_MAX;
908 double sum_free1 = 0, sum_free2 = 0;
909 double r1, r2;
910
911 int i;
912
913 for( i = 0; i < alpha_count; i++ )
914 {
915 double G_i = G[i];
916 if( y[i] > 0 )
917 {
918 if( is_lower_bound(i) )
919 ub1 = MIN( ub1, G_i );
920 else if( is_upper_bound(i) )
921 lb1 = MAX( lb1, G_i );
922 else
923 {
924 ++nr_free1;
925 sum_free1 += G_i;
926 }
927 }
928 else
929 {
930 if( is_lower_bound(i) )
931 ub2 = MIN( ub2, G_i );
932 else if( is_upper_bound(i) )
933 lb2 = MAX( lb2, G_i );
934 else
935 {
936 ++nr_free2;
937 sum_free2 += G_i;
938 }
939 }
940 }
941
942 r1 = nr_free1 > 0 ? sum_free1/nr_free1 : (ub1 + lb1)*0.5;
943 r2 = nr_free2 > 0 ? sum_free2/nr_free2 : (ub2 + lb2)*0.5;
944
945 rho = (r1 - r2)*0.5;
946 r = (r1 + r2)*0.5;
947}

980
981bool CvSVMSolver::solve_nu_svc( int _sample_count, int _var_count, const float** _samples, schar* _y,
982 CvMemStorage* _storage, CvSVMKernel* _kernel,
983 double* _alpha, CvSVMSolutionInfo& _si )
984{
985 int i;
986 double sum_pos, sum_neg, inv_r;
987
988 if( !create( _sample_count, _var_count, _samples, _y, _sample_count,
989 _alpha, 1., 1., _storage, _kernel, &CvSVMSolver::get_row_svc,
990 &CvSVMSolver::select_working_set_nu_svm, &CvSVMSolver::calc_rho_nu_svm ))
991 return false;
992
993 sum_pos = kernel->params->nu * sample_count * 0.5;
994 sum_neg = kernel->params->nu * sample_count * 0.5;
995
996 for( i = 0; i < sample_count; i++ )
997 {
998 if( y[i] > 0 )
999 {
1000 alpha[i] = MIN(1.0, sum_pos);
1001 sum_pos -= alpha[i];
1002 }
1003 else
1004 {
1005 alpha[i] = MIN(1.0, sum_neg);
1006 sum_neg -= alpha[i];
1007 }
1008 b[i] = 0;
1009 }
1010
1011 if( !solve_generic( _si ))
1012 return false;
1013
1014 inv_r = 1./_si.r;
1015
1016 for( i = 0; i < sample_count; i++ )
1017 alpha[i] *= y[i]*inv_r;
1018
1019 _si.rho *= inv_r;
1020 _si.obj *= (inv_r*inv_r);
1021 _si.upper_bound_p = inv_r;
1022 _si.upper_bound_n = inv_r;
1023
1024 return true;
1025}

1028bool CvSVMSolver::solve_one_class( int _sample_count, int _var_count, const float** _samples,
1029 CvMemStorage* _storage, CvSVMKernel* _kernel,
1030 double* _alpha, CvSVMSolutionInfo& _si )
1031{
1032 int i, n;
1033 double nu = _kernel->params->nu;
1034
1035 if( !create( _sample_count, _var_count, _samples, 0, _sample_count,
1036 _alpha, 1., 1., _storage, _kernel, &CvSVMSolver::get_row_one_class,
1037 &CvSVMSolver::select_working_set, &CvSVMSolver::calc_rho ))
1038 return false;
1039
1040 y = (schar*)cvMemStorageAlloc( storage, sample_count*sizeof(y[0]) );
1041 n = cvRound( nu*sample_count );
1042
1043 for( i = 0; i < sample_count; i++ )
1044 {
1045 y[i] = 1;
1046 b[i] = 0;
1047 alpha[i] = i < n ? 1 : 0;
1048 }
1049
1050 if( n < sample_count )
1051 alpha[n] = nu * sample_count - n;
1052 else
1053 alpha[n-1] = nu * sample_count - (n-1);
1054
1055 return solve_generic(_si);
1056}

1059bool CvSVMSolver::solve_eps_svr( int _sample_count, int _var_count, const float** _samples,
1060 const float* _y, CvMemStorage* _storage,
1061 CvSVMKernel* _kernel, double* _alpha, CvSVMSolutionInfo& _si )
1062{
1063 int i;
1064 double p = _kernel->params->p, kernel_param_c = _kernel->params->C;
1065
1066 if( !create( _sample_count, _var_count, _samples, 0,
1067 _sample_count*2, 0, kernel_param_c, kernel_param_c, _storage, _kernel, &CvSVMSolver::get_row_svr,
1068 &CvSVMSolver::select_working_set, &CvSVMSolver::calc_rho ))
1069 return false;
1070
1071 y = (schar*)cvMemStorageAlloc( storage, sample_count*2*sizeof(y[0]) );
1072 alpha = (double*)cvMemStorageAlloc( storage, alpha_count*sizeof(alpha[0]) );
1073
1074 for( i = 0; i < sample_count; i++ )
1075 {
1076 alpha[i] = 0;
1077 b[i] = p - _y[i];
1078 y[i] = 1;
1079
1080 alpha[i+sample_count] = 0;
1081 b[i+sample_count] = p + _y[i];
1082 y[i+sample_count] = -1;
1083 }
1084
1085 if( !solve_generic( _si ))
1086 return false;
1087
1088 for( i = 0; i < sample_count; i++ )
1089 _alpha[i] = alpha[i] - alpha[i+sample_count];
1090
1091 return true;
1092}

1095bool CvSVMSolver::solve_nu_svr( int _sample_count, int _var_count, const float** _samples,
1096 const float* _y, CvMemStorage* _storage,
1097 CvSVMKernel* _kernel, double* _alpha, CvSVMSolutionInfo& _si )
1098{
1099 int i;
1100 double kernel_param_c = _kernel->params->C, sum;
1101
1102 if( !create( _sample_count, _var_count, _samples, 0,
1103 _sample_count*2, 0, 1., 1., _storage, _kernel, &CvSVMSolver::get_row_svr,
1104 &CvSVMSolver::select_working_set_nu_svm, &CvSVMSolver::calc_rho_nu_svm ))
1105 return false;
1106
1107 y = (schar*)cvMemStorageAlloc( storage, sample_count*2*sizeof(y[0]) );
1108 alpha = (double*)cvMemStorageAlloc( storage, alpha_count*sizeof(alpha[0]) );
1109 sum = kernel_param_c * _kernel->params->nu * sample_count * 0.5;
1110
1111 for( i = 0; i < sample_count; i++ )
1112 {
1113 alpha[i] = alpha[i + sample_count] = MIN(sum, kernel_param_c);
1114 sum -= alpha[i];
1115
1116 b[i] = -_y[i];
1117 y[i] = 1;
1118
1119 b[i + sample_count] = _y[i];
1120 y[i + sample_count] = -1;
1121 }
1122
1123 if( !solve_generic( _si ))
1124 return false;
1125
1126 for( i = 0; i < sample_count; i++ )
1127 _alpha[i] = alpha[i] - alpha[i+sample_count];
1128
1129 return true;
1130}

1316bool CvSVM::do_train( int svm_type, int sample_count, int var_count, const float** samples,
1317 const CvMat* responses, CvMemStorage* temp_storage, double* alpha )
1318{
1319 bool ok = false;
1320
1321 CV_FUNCNAME( "CvSVM::do_train" );
1322
1323 __BEGIN__;
1324
1325 CvSVMDecisionFunc* df = 0;
1326 const int sample_size = var_count*sizeof(samples[0][0]);
1327 int i, j, k;
1328
1329 cvClearMemStorage( storage );
1330
1331 if( svm_type == ONE_CLASS || svm_type == EPS_SVR || svm_type == NU_SVR )
1332 {
1333 int sv_count = 0;
1334
1335 CV_CALL( decision_func = df =
1336 (CvSVMDecisionFunc*)cvAlloc( sizeof(df[0]) ));
1337
1338 df->rho = 0;
1339 if( !train1( sample_count, var_count, samples, svm_type == ONE_CLASS ? 0 :
1340 responses->data.i, 0, 0, temp_storage, alpha, df->rho ))
1341 EXIT;
1342
1343 for( i = 0; i < sample_count; i++ )
1344 sv_count += fabs(alpha[i]) > 0;
1345
1346 CV_Assert(sv_count != 0);
1347
1348 sv_total = df->sv_count = sv_count;
1349 CV_CALL( df->alpha = (double*)cvMemStorageAlloc( storage, sv_count*sizeof(df->alpha[0])) );
1350 CV_CALL( sv = (float**)cvMemStorageAlloc( storage, sv_count*sizeof(sv[0])));
1351
1352 for( i = k = 0; i < sample_count; i++ )
1353 {
1354 if( fabs(alpha[i]) > 0 )
1355 {
1356 CV_CALL( sv[k] = (float*)cvMemStorageAlloc( storage, sample_size ));
1357 memcpy( sv[k], samples[i], sample_size );
1358 df->alpha[k++] = alpha[i];
1359 }
1360 }
1361 }
1362 else
1363 {
1364 int class_count = class_labels->cols;
1365 int* sv_tab = 0;
1366 const float** temp_samples = 0;
1367 int* class_ranges = 0;
1368 schar* temp_y = 0;
1369 assert( svm_type == CvSVM::C_SVC || svm_type == CvSVM::NU_SVC );
1370
1371 if( svm_type == CvSVM::C_SVC && params.class_weights )
1372 {
1373 const CvMat* cw = params.class_weights;
1374
1375 if( !CV_IS_MAT(cw) || (cw->cols != 1 && cw->rows != 1) ||
1376 cw->rows + cw->cols - 1 != class_count ||
1377 (CV_MAT_TYPE(cw->type) != CV_32FC1 && CV_MAT_TYPE(cw->type) != CV_64FC1) )
1378 CV_ERROR( CV_StsBadArg, "params.class_weights must be 1d floating-point vector "
1379 "containing as many elements as the number of classes" );
1380
1381 CV_CALL( class_weights = cvCreateMat( cw->rows, cw->cols, CV_64F ));
1382 CV_CALL( cvConvert( cw, class_weights ));
1383 CV_CALL( cvScale( class_weights, class_weights, params.C ));
1384 }
1385
1386 CV_CALL( decision_func = df = (CvSVMDecisionFunc*)cvAlloc(
1387 (class_count*(class_count-1)/2)*sizeof(df[0])));
1388
1389 CV_CALL( sv_tab = (int*)cvMemStorageAlloc( temp_storage, sample_count*sizeof(sv_tab[0]) ));
1390 memset( sv_tab, 0, sample_count*sizeof(sv_tab[0]) );
1391 CV_CALL( class_ranges = (int*)cvMemStorageAlloc( temp_storage,
1392 (class_count + 1)*sizeof(class_ranges[0])));
1393 CV_CALL( temp_samples = (const float**)cvMemStorageAlloc( temp_storage,
1394 sample_count*sizeof(temp_samples[0])));
1395 CV_CALL( temp_y = (schar*)cvMemStorageAlloc( temp_storage, sample_count));
1396
1397 class_ranges[class_count] = 0;
1398 cvSortSamplesByClasses( samples, responses, class_ranges, 0 );
1399 //check that while cross-validation there were the samples from all the classes
1400 if( class_ranges[class_count] <= 0 )
1401 CV_ERROR( CV_StsBadArg, "While cross-validation one or more of the classes have "
1402 "been fell out of the sample. Try to enlarge <CvSVMParams::k_fold>" );
1403
1404 if( svm_type == NU_SVC )
1405 {
1406 // check if nu is feasible
1407 for(i = 0; i < class_count; i++ )
1408 {
1409 int ci = class_ranges[i+1] - class_ranges[i];
1410 for( j = i+1; j< class_count; j++ )
1411 {
1412 int cj = class_ranges[j+1] - class_ranges[j];
1413 if( params.nu*(ci + cj)*0.5 > MIN( ci, cj ) )
1414 {
1415 // !!!TODO!!! add some diagnostic
1416 EXIT; // exit immediately; will release the model and return NULL pointer
1417 }
1418 }
1419 }
1420 }
1421
1422 // train n*(n-1)/2 classifiers
1423 for( i = 0; i < class_count; i++ )
1424 {
1425 for( j = i+1; j < class_count; j++, df++ )
1426 {
1427 int si = class_ranges[i], ci = class_ranges[i+1] - si;
1428 int sj = class_ranges[j], cj = class_ranges[j+1] - sj;
1429 double Cp = params.C, Cn = Cp;
1430 int k1 = 0, sv_count = 0;
1431
1432 for( k = 0; k < ci; k++ )
1433 {
1434 temp_samples[k] = samples[si + k];
1435 temp_y[k] = 1;
1436 }
1437
1438 for( k = 0; k < cj; k++ )
1439 {
1440 temp_samples[ci + k] = samples[sj + k];
1441 temp_y[ci + k] = -1;
1442 }
1443
1444 if( class_weights )
1445 {
1446 Cp = class_weights->data.db[i];
1447 Cn = class_weights->data.db[j];
1448 }
1449
1450 if( !train1( ci + cj, var_count, temp_samples, temp_y,
1451 Cp, Cn, temp_storage, alpha, df->rho ))
1452 EXIT;
1453
1454 for( k = 0; k < ci + cj; k++ )
1455 sv_count += fabs(alpha[k]) > 0;
1456
1457 df->sv_count = sv_count;
1458
1459 CV_CALL( df->alpha = (double*)cvMemStorageAlloc( temp_storage,
1460 sv_count*sizeof(df->alpha[0])));
1461 CV_CALL( df->sv_index = (int*)cvMemStorageAlloc( temp_storage,
1462 sv_count*sizeof(df->sv_index[0])));
1463
1464 for( k = 0; k < ci; k++ )
1465 {
1466 if( fabs(alpha[k]) > 0 )
1467 {
1468 sv_tab[si + k] = 1;
1469 df->sv_index[k1] = si + k;
1470 df->alpha[k1++] = alpha[k];
1471 }
1472 }
1473
1474 for( k = 0; k < cj; k++ )
1475 {
1476 if( fabs(alpha[ci + k]) > 0 )
1477 {
1478 sv_tab[sj + k] = 1;
1479 df->sv_index[k1] = sj + k;
1480 df->alpha[k1++] = alpha[ci + k];
1481 }
1482 }
1483 }
1484 }
1485
1486 // allocate support vectors and initialize sv_tab
1487 for( i = 0, k = 0; i < sample_count; i++ )
1488 {
1489 if( sv_tab[i] )
1490 sv_tab[i] = ++k;
1491 }
1492
1493 sv_total = k;
1494 CV_CALL( sv = (float**)cvMemStorageAlloc( storage, sv_total*sizeof(sv[0])));
1495
1496 for( i = 0, k = 0; i < sample_count; i++ )
1497 {
1498 if( sv_tab[i] )
1499 {
1500 CV_CALL( sv[k] = (float*)cvMemStorageAlloc( storage, sample_size ));
1501 memcpy( sv[k], samples[i], sample_size );
1502 k++;
1503 }
1504 }
1505
1506 df = (CvSVMDecisionFunc*)decision_func;
1507
1508 // set sv pointers
1509 for( i = 0; i < class_count; i++ )
1510 {
1511 for( j = i+1; j < class_count; j++, df++ )
1512 {
1513 for( k = 0; k < df->sv_count; k++ )
1514 {
1515 df->sv_index[k] = sv_tab[df->sv_index[k]]-1;
1516 assert( (unsigned)df->sv_index[k] < (unsigned)sv_total );
1517 }
1518 }
1519 }
1520 }
1521
1522 optimize_linear_svm();
1523 ok = true;
1524
1525 __END__;
1526
1527 return ok;
1528}

1531void CvSVM::optimize_linear_svm()
1532{
1533 // we optimize only linear SVM: compress all the support vectors into one.
1534 if( params.kernel_type != LINEAR )
1535 return;
1536
1537 int class_count = class_labels ? class_labels->cols :
1538 params.svm_type == CvSVM::ONE_CLASS ? 1 : 0;
1539
1540 int i, df_count = class_count > 1 ? class_count*(class_count-1)/2 : 1;
1541 CvSVMDecisionFunc* df = decision_func;
1542
1543 for( i = 0; i < df_count; i++ )
1544 {
1545 int sv_count = df[i].sv_count;
1546 if( sv_count != 1 )
1547 break;
1548 }
1549
1550 // if every decision functions uses a single support vector;
1551 // it's already compressed. skip it then.
1552 if( i == df_count )
1553 return;
1554
1555 int var_count = get_var_count();
1556 cv::AutoBuffer<double> vbuf(var_count);
1557 double* v = vbuf;
1558 float** new_sv = (float**)cvMemStorageAlloc(storage, df_count*sizeof(new_sv[0]));
1559
1560 for( i = 0; i < df_count; i++ )
1561 {
1562 new_sv[i] = (float*)cvMemStorageAlloc(storage, var_count*sizeof(new_sv[i][0]));
1563 float* dst = new_sv[i];
1564 memset(v, 0, var_count*sizeof(v[0]));
1565 int j, k, sv_count = df[i].sv_count;
1566 for( j = 0; j < sv_count; j++ )
1567 {
1568 const float* src = class_count > 1 && df[i].sv_index ? sv[df[i].sv_index[j]] : sv[j];
1569 double a = df[i].alpha[j];
1570 for( k = 0; k < var_count; k++ )
1571 v[k] += src[k]*a;
1572 }
1573 for( k = 0; k < var_count; k++ )
1574 dst[k] = (float)v[k];
1575 df[i].sv_count = 1;
1576 df[i].alpha[0] = 1.;
1577 if( class_count > 1 && df[i].sv_index )
1578 df[i].sv_index[0] = i;
1579 }
1580
1581 sv = new_sv;
1582 sv_total = df_count;
1583}

1600 double* alpha;

1651struct indexedratio
1652{
1653 double val;
1654 int ind;
1655 int count_smallest, count_biggest;
1656 void eval() { val = (double) count_smallest/(count_smallest+count_biggest); }
1657};
1658
1659static int CV_CDECL
1660icvCmpIndexedratio( const void* a, const void* b )
1661{
1662 return ((const indexedratio*)a)->val < ((const indexedratio*)b)->val ? -1
1663 : ((const indexedratio*)a)->val > ((const indexedratio*)b)->val ? 1
1664 : 0;

1667bool CvSVM::train_auto( const CvMat* _train_data, const CvMat* _responses,
1668 const CvMat* _var_idx, const CvMat* _sample_idx, CvSVMParams _params, int k_fold,
1669 CvParamGrid C_grid, CvParamGrid gamma_grid, CvParamGrid p_grid,
1670 CvParamGrid nu_grid, CvParamGrid coef_grid, CvParamGrid degree_grid,
1671 bool balanced)
1672{
1673 bool ok = false;
1674 CvMat* responses = 0;
1675 CvMat* responses_local = 0;
1676 CvMemStorage* temp_storage = 0;
1677 const float** samples = 0;
1678 const float** samples_local = 0;
1679
1680 CV_FUNCNAME( "CvSVM::train_auto" );
1681 __BEGIN__;
1682
1683 int svm_type, sample_count, var_count, sample_size;
1684 int block_size = 1 << 16;
1685 double* alpha;
1686 RNG* rng = &theRNG();
1687
1688 // all steps are logarithmic and must be > 1
1689 double degree_step = 10, g_step = 10, coef_step = 10, C_step = 10, nu_step = 10, p_step = 10;
1690 double gamma = 0, curr_c = 0, degree = 0, coef = 0, p = 0, nu = 0;
1691 double best_degree = 0, best_gamma = 0, best_coef = 0, best_C = 0, best_nu = 0, best_p = 0;
1692 float min_error = FLT_MAX, error;
1693
1694 if( _params.svm_type == CvSVM::ONE_CLASS )
1695 {
1696 if(!train( _train_data, _responses, _var_idx, _sample_idx, _params ))
1697 EXIT;
1698 return true;
1699 }
1700
1701 clear();
1702
1703 if( k_fold < 2 )
1704 CV_ERROR( CV_StsBadArg, "Parameter <k_fold> must be > 1" );
1705
1706 CV_CALL(set_params( _params ));
1707 svm_type = _params.svm_type;
1708
1709 // All the parameters except, possibly, <coef0> are positive.
1710 // <coef0> is nonnegative
1711 if( C_grid.step <= 1 )
1712 {
1713 C_grid.min_val = C_grid.max_val = params.C;
1714 C_grid.step = 10;
1715 }
1716 else
1717 CV_CALL(C_grid.check());
1718
1719 if( gamma_grid.step <= 1 )
1720 {
1721 gamma_grid.min_val = gamma_grid.max_val = params.gamma;
1722 gamma_grid.step = 10;
1723 }
1724 else
1725 CV_CALL(gamma_grid.check());
1726
1727 if( p_grid.step <= 1 )
1728 {
1729 p_grid.min_val = p_grid.max_val = params.p;
1730 p_grid.step = 10;
1731 }
1732 else
1733 CV_CALL(p_grid.check());
1734
1735 if( nu_grid.step <= 1 )
1736 {
1737 nu_grid.min_val = nu_grid.max_val = params.nu;
1738 nu_grid.step = 10;
1739 }
1740 else
1741 CV_CALL(nu_grid.check());
1742
1743 if( coef_grid.step <= 1 )
1744 {
1745 coef_grid.min_val = coef_grid.max_val = params.coef0;
1746 coef_grid.step = 10;
1747 }
1748 else
1749 CV_CALL(coef_grid.check());
1750
1751 if( degree_grid.step <= 1 )
1752 {
1753 degree_grid.min_val = degree_grid.max_val = params.degree;
1754 degree_grid.step = 10;
1755 }
1756 else
1757 CV_CALL(degree_grid.check());
1758
1759 // these parameters are not used:
1760 if( params.kernel_type != CvSVM::POLY )
1761 degree_grid.min_val = degree_grid.max_val = params.degree;
1762 if( params.kernel_type == CvSVM::LINEAR )
1763 gamma_grid.min_val = gamma_grid.max_val = params.gamma;
1764 if( params.kernel_type != CvSVM::POLY && params.kernel_type != CvSVM::SIGMOID )
1765 coef_grid.min_val = coef_grid.max_val = params.coef0;
1766 if( svm_type == CvSVM::NU_SVC || svm_type == CvSVM::ONE_CLASS )
1767 C_grid.min_val = C_grid.max_val = params.C;
1768 if( svm_type == CvSVM::C_SVC || svm_type == CvSVM::EPS_SVR )
1769 nu_grid.min_val = nu_grid.max_val = params.nu;
1770 if( svm_type != CvSVM::EPS_SVR )
1771 p_grid.min_val = p_grid.max_val = params.p;
1772
1773 CV_ASSERT( g_step > 1 && degree_step > 1 && coef_step > 1);
1774 CV_ASSERT( p_step > 1 && C_step > 1 && nu_step > 1 );
1775
1776 /* Prepare training data and related parameters */
1777 CV_CALL(cvPrepareTrainData( "CvSVM::train_auto", _train_data, CV_ROW_SAMPLE,
1778 svm_type != CvSVM::ONE_CLASS ? _responses : 0,
1779 svm_type == CvSVM::C_SVC ||
1780 svm_type == CvSVM::NU_SVC ? CV_VAR_CATEGORICAL :
1781 CV_VAR_ORDERED, _var_idx, _sample_idx,
1782 false, &samples, &sample_count, &var_count, &var_all,
1783 &responses, &class_labels, &var_idx ));
1784
1785 sample_size = var_count*sizeof(samples[0][0]);
1786
1787 // make the storage block size large enough to fit all
1788 // the temporary vectors and output support vectors.
1789 block_size = MAX( block_size, sample_count*(int)sizeof(CvSVMKernelRow));
1790 block_size = MAX( block_size, sample_count*2*(int)sizeof(double) + 1024 );
1791 block_size = MAX( block_size, sample_size*2 + 1024 );
1792
1793 CV_CALL( storage = cvCreateMemStorage(block_size + sizeof(CvMemBlock) + sizeof(CvSeqBlock)));
1794 CV_CALL(temp_storage = cvCreateChildMemStorage(storage));
1795 CV_CALL(alpha = (double*)cvMemStorageAlloc(temp_storage, sample_count*sizeof(double)));
1796
1797 create_kernel();
1798 create_solver();
1799
1800 {
1801 const int testset_size = sample_count/k_fold;
1802 const int trainset_size = sample_count - testset_size;
1803 const int last_testset_size = sample_count - testset_size*(k_fold-1);
1804 const int last_trainset_size = sample_count - last_testset_size;
1805 const bool is_regression = (svm_type == EPS_SVR) || (svm_type == NU_SVR);
1806
1807 size_t resp_elem_size = CV_ELEM_SIZE(responses->type);
1808 size_t size = 2*last_trainset_size*sizeof(samples[0]);
1809
1810 samples_local = (const float**) cvAlloc( size );
1811 memset( samples_local, 0, size );
1812
1813 responses_local = cvCreateMat( 1, trainset_size, CV_MAT_TYPE(responses->type) );
1814 cvZero( responses_local );
1815
1816 // randomly permute samples and responses
1817 for(int i = 0; i < sample_count; i++ )
1818 {
1819 int i1 = (*rng)(sample_count);
1820 int i2 = (*rng)(sample_count);
1821 const float* temp;
1822 float t;
1823 int y;
1824
1825 CV_SWAP( samples[i1], samples[i2], temp );
1826 if( is_regression )
1827 CV_SWAP( responses->data.fl[i1], responses->data.fl[i2], t );
1828 else
1829 CV_SWAP( responses->data.i[i1], responses->data.i[i2], y );
1830 }
1831
1832 if (!is_regression && class_labels->cols==2 && balanced)
1833 {
1834 // count class samples
1835 int num_0=0,num_1=0;
1836 for (int i=0; i<sample_count; ++i)
1837 {
1838 if (responses->data.i[i]==class_labels->data.i[0])
1839 ++num_0;
1840 else
1841 ++num_1;
1842 }
1843
1844 int label_smallest_class;
1845 int label_biggest_class;
1846 if (num_0 < num_1)
1847 {
1848 label_biggest_class = class_labels->data.i[1];
1849 label_smallest_class = class_labels->data.i[0];
1850 }
1851 else
1852 {
1853 label_biggest_class = class_labels->data.i[0];
1854 label_smallest_class = class_labels->data.i[1];
1855 int y;
1856 CV_SWAP(num_0,num_1,y);
1857 }
1858 const double class_ratio = (double) num_0/sample_count;
1859 // calculate class ratio of each fold
1860 indexedratio *ratios=0;
1861 ratios = (indexedratio*) cvAlloc(k_fold*sizeof(*ratios));
1862 for (int k=0, i_begin=0; k<k_fold; ++k, i_begin+=testset_size)
1863 {
1864 int count0=0;
1865 int count1=0;
1866 int i_end = i_begin + (k<k_fold-1 ? testset_size : last_testset_size);
1867 for (int i=i_begin; i<i_end; ++i)
1868 {
1869 if (responses->data.i[i]==label_smallest_class)
1870 ++count0;
1871 else
1872 ++count1;
1873 }
1874 ratios[k].ind = k;
1875 ratios[k].count_smallest = count0;
1876 ratios[k].count_biggest = count1;
1877 ratios[k].eval();
1878 }
1879 // initial distance
1880 qsort(ratios, k_fold, sizeof(ratios[0]), icvCmpIndexedratio);
1881 double old_dist = 0.0;
1882 for (int k=0; k<k_fold; ++k)
1883 old_dist += std::abs(ratios[k].val-class_ratio);
1884 double new_dist = 1.0;
1885 // iterate to make the folds more balanced
1886 while (new_dist > 0.0)
1887 {
1888 if (ratios[0].count_biggest==0 || ratios[k_fold-1].count_smallest==0)
1889 break; // we are not able to swap samples anymore
1890 // what if we swap the samples, calculate the new distance
1891 ratios[0].count_smallest++;
1892 ratios[0].count_biggest--;
1893 ratios[0].eval();
1894 ratios[k_fold-1].count_smallest--;
1895 ratios[k_fold-1].count_biggest++;
1896 ratios[k_fold-1].eval();
1897 qsort(ratios, k_fold, sizeof(ratios[0]), icvCmpIndexedratio);
1898 new_dist = 0.0;
1899 for (int k=0; k<k_fold; ++k)
1900 new_dist += std::abs(ratios[k].val-class_ratio);
1901 if (new_dist < old_dist)
1902 {
1903 // swapping really improves, so swap the samples
1904 // index of the biggest_class sample from the minimum ratio fold
1905 int i1 = ratios[0].ind * testset_size;
1906 for ( ; i1<sample_count; ++i1)
1907 {
1908 if (responses->data.i[i1]==label_biggest_class)
1909 break;
1910 }
1911 // index of the smallest_class sample from the maximum ratio fold
1912 int i2 = ratios[k_fold-1].ind * testset_size;
1913 for ( ; i2<sample_count; ++i2)
1914 {
1915 if (responses->data.i[i2]==label_smallest_class)
1916 break;
1917 }
1918 // swap
1919 const float* temp;
1920 int y;
1921 CV_SWAP( samples[i1], samples[i2], temp );
1922 CV_SWAP( responses->data.i[i1], responses->data.i[i2], y );
1923 old_dist = new_dist;
1924 }
1925 else
1926 break; // does not improve, so break the loop
1927 }
1928 cvFree(&ratios);
1929 }
1930
1931 int* cls_lbls = class_labels ? class_labels->data.i : 0;
1932 curr_c = C_grid.min_val;
1933 do
1934 {
1935 params.C = curr_c;
1936 gamma = gamma_grid.min_val;
1937 do
1938 {
1939 params.gamma = gamma;
1940 p = p_grid.min_val;
1941 do
1942 {
1943 params.p = p;
1944 nu = nu_grid.min_val;
1945 do
1946 {
1947 params.nu = nu;
1948 coef = coef_grid.min_val;
1949 do
1950 {
1951 params.coef0 = coef;
1952 degree = degree_grid.min_val;
1953 do
1954 {
1955 params.degree = degree;
1956
1957 float** test_samples_ptr = (float**)samples;
1958 uchar* true_resp = responses->data.ptr;
1959 int test_size = testset_size;
1960 int train_size = trainset_size;
1961
1962 error = 0;
1963 for(int k = 0; k < k_fold; k++ )
1964 {
1965 memcpy( samples_local, samples, sizeof(samples[0])*test_size*k );
1966 memcpy( samples_local + test_size*k, test_samples_ptr + test_size,
1967 sizeof(samples[0])*(sample_count - testset_size*(k+1)) );
1968
1969 memcpy( responses_local->data.ptr, responses->data.ptr, resp_elem_size*test_size*k );
1970 memcpy( responses_local->data.ptr + resp_elem_size*test_size*k,
1971 true_resp + resp_elem_size*test_size,
1972 resp_elem_size*(sample_count - testset_size*(k+1)) );
1973
1974 if( k == k_fold - 1 )
1975 {
1976 test_size = last_testset_size;
1977 train_size = last_trainset_size;
1978 responses_local->cols = last_trainset_size;
1979 }
1980
1981 // Train SVM on <train_size> samples
1982 if( !do_train( svm_type, train_size, var_count,
1983 (const float**)samples_local, responses_local, temp_storage, alpha ) )
1984 EXIT;
1985
1986 // Compute test set error on <test_size> samples
1987 for(int i = 0; i < test_size; i++, true_resp += resp_elem_size, test_samples_ptr++ )
1988 {
1989 float resp = predict( *test_samples_ptr, var_count );
1990 error += is_regression ? powf( resp - *(float*)true_resp, 2 )
1991 : ((int)resp != cls_lbls[*(int*)true_resp]);
1992 }
1993 }
1994 if( min_error > error )
1995 {
1996 min_error = error;
1997 best_degree = degree;
1998 best_gamma = gamma;
1999 best_coef = coef;
2000 best_C = curr_c;
2001 best_nu = nu;
2002 best_p = p;
2003 }
2004 degree *= degree_grid.step;
2005 }
2006 while( degree < degree_grid.max_val );
2007 coef *= coef_grid.step;
2008 }
2009 while( coef < coef_grid.max_val );
2010 nu *= nu_grid.step;
2011 }
2012 while( nu < nu_grid.max_val );
2013 p *= p_grid.step;
2014 }
2015 while( p < p_grid.max_val );
2016 gamma *= gamma_grid.step;
2017 }
