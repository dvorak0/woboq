55void CvERTreeTrainData::set_data( const CvMat* _train_data, int _tflag,
56 const CvMat* _responses, const CvMat* _var_idx, const CvMat* _sample_idx,
57 const CvMat* _var_type, const CvMat* _missing_mask, const CvDTreeParams& _params,
58 bool _shared, bool _add_labels, bool _update_data )
59{
60 CvMat* sample_indices = 0;
61 CvMat* var_type0 = 0;
62 CvMat* tmp_map = 0;
63 int** int_ptr = 0;
64 CvPair16u32s* pair16u32s_ptr = 0;
65 CvDTreeTrainData* data = 0;
66 float *_fdst = 0;
67 int *_idst = 0;
68 unsigned short* udst = 0;
69 int* idst = 0;
70
71 CV_FUNCNAME( "CvERTreeTrainData::set_data" );
72
73 __BEGIN__;
74
75 int sample_all = 0, r_type, cv_n;
76 int total_c_count = 0;
77 int tree_block_size, temp_block_size, max_split_size, nv_size, cv_size = 0;
78 int ds_step, dv_step, ms_step = 0, mv_step = 0; // {data|mask}{sample|var}_step
79 int vi, i, size;
80 char err[100];
81 const int *sidx = 0, *vidx = 0;
82
83 uint64 effective_buf_size = 0;
84 int effective_buf_height = 0, effective_buf_width = 0;
85
86 if ( _params.use_surrogates )
87 CV_ERROR(CV_StsBadArg, "CvERTrees do not support surrogate splits");
88
89 if( _update_data && data_root )
90 {
91 CV_ERROR(CV_StsBadArg, "CvERTrees do not support data update");
92 }
93
94 clear();
95
96 var_all = 0;
97 rng = &cv::theRNG();
98
99 CV_CALL( set_params( _params ));
100
101 // check parameter types and sizes
102 CV_CALL( cvCheckTrainData( _train_data, _tflag, _missing_mask, &var_all, &sample_all ));
103
104 train_data = _train_data;
105 responses = _responses;
106 missing_mask = _missing_mask;
107
108 if( _tflag == CV_ROW_SAMPLE )
109 {
110 ds_step = _train_data->step/CV_ELEM_SIZE(_train_data->type);
111 dv_step = 1;
112 if( _missing_mask )
113 ms_step = _missing_mask->step, mv_step = 1;
114 }
115 else
116 {
117 dv_step = _train_data->step/CV_ELEM_SIZE(_train_data->type);
118 ds_step = 1;
119 if( _missing_mask )
120 mv_step = _missing_mask->step, ms_step = 1;
121 }
122 tflag = _tflag;
123
124 sample_count = sample_all;
125 var_count = var_all;
126
127 if( _sample_idx )
128 {
129 CV_CALL( sample_indices = cvPreprocessIndexArray( _sample_idx, sample_all ));
130 sidx = sample_indices->data.i;
131 sample_count = sample_indices->rows + sample_indices->cols - 1;
132 }
133
134 if( _var_idx )
135 {
136 CV_CALL( var_idx = cvPreprocessIndexArray( _var_idx, var_all ));
137 vidx = var_idx->data.i;
138 var_count = var_idx->rows + var_idx->cols - 1;
139 }
140
141 if( !CV_IS_MAT(_responses) ||
142 (CV_MAT_TYPE(_responses->type) != CV_32SC1 &&
143 CV_MAT_TYPE(_responses->type) != CV_32FC1) ||
144 (_responses->rows != 1 && _responses->cols != 1) ||
145 _responses->rows + _responses->cols - 1 != sample_all )
146 CV_ERROR( CV_StsBadArg, "The array of _responses must be an integer or "
147 "floating-point vector containing as many elements as "
148 "the total number of samples in the training data matrix" );
149
150 is_buf_16u = false;
151 if ( sample_count < 65536 )
152 is_buf_16u = true;
153
154 r_type = CV_VAR_CATEGORICAL;
155 if( _var_type )
156 CV_CALL( var_type0 = cvPreprocessVarType( _var_type, var_idx, var_count, &r_type ));
157
158 CV_CALL( var_type = cvCreateMat( 1, var_count+2, CV_32SC1 ));
159
160 cat_var_count = 0;
161 ord_var_count = -1;
162
163 is_classifier = r_type == CV_VAR_CATEGORICAL;
164
165 // step 0. calc the number of categorical vars
166 for( vi = 0; vi < var_count; vi++ )
167 {
168 char vt = var_type0 ? var_type0->data.ptr[vi] : CV_VAR_ORDERED;
169 var_type->data.i[vi] = vt == CV_VAR_CATEGORICAL ? cat_var_count++ : ord_var_count--;
170 }
171
172 ord_var_count = ~ord_var_count;
173 cv_n = params.cv_folds;
174 // set the two last elements of var_type array to be able
175 // to locate responses and cross-validation labels using
176 // the corresponding get_* functions.
177 var_type->data.i[var_count] = cat_var_count;
178 var_type->data.i[var_count+1] = cat_var_count+1;
179
180 // in case of single ordered predictor we need dummy cv_labels
181 // for safe split_node_data() operation
182 have_labels = cv_n > 0 || (ord_var_count == 1 && cat_var_count == 0) || _add_labels;
183
184 work_var_count = cat_var_count + (is_classifier ? 1 : 0) + (have_labels ? 1 : 0);
185
186 shared = _shared;
187 buf_count = shared ? 2 : 1;
188
189 buf_size = -1; // the member buf_size is obsolete
190
191 effective_buf_size = (uint64)(work_var_count + 1)*(uint64)sample_count * buf_count; // this is the total size of "CvMat buf" to be allocated
192 effective_buf_width = sample_count;
193 effective_buf_height = work_var_count+1;
194
195 if (effective_buf_width >= effective_buf_height)
196 effective_buf_height *= buf_count;
197 else
198 effective_buf_width *= buf_count;
199
200 if ((uint64)effective_buf_width * (uint64)effective_buf_height != effective_buf_size)
201 {
202 CV_Error(CV_StsBadArg, "The memory buffer cannot be allocated since its size exceeds integer fields limit");
203 }
204
205 if ( is_buf_16u )
206 {
207 CV_CALL( buf = cvCreateMat( effective_buf_height, effective_buf_width, CV_16UC1 ));
208 CV_CALL( pair16u32s_ptr = (CvPair16u32s*)cvAlloc( sample_count*sizeof(pair16u32s_ptr[0]) ));
209 }
210 else
211 {
212 CV_CALL( buf = cvCreateMat( effective_buf_height, effective_buf_width, CV_32SC1 ));
213 CV_CALL( int_ptr = (int**)cvAlloc( sample_count*sizeof(int_ptr[0]) ));
214 }
215
216 size = is_classifier ? cat_var_count+1 : cat_var_count;
217 size = !size ? 1 : size;
218 CV_CALL( cat_count = cvCreateMat( 1, size, CV_32SC1 ));
219 CV_CALL( cat_ofs = cvCreateMat( 1, size, CV_32SC1 ));
220
221 size = is_classifier ? (cat_var_count + 1)*params.max_categories : cat_var_count*params.max_categories;
222 size = !size ? 1 : size;
223 CV_CALL( cat_map = cvCreateMat( 1, size, CV_32SC1 ));
224
225 // now calculate the maximum size of split,
226 // create memory storage that will keep nodes and splits of the decision tree
227 // allocate root node and the buffer for the whole training data
228 max_split_size = cvAlign(sizeof(CvDTreeSplit) +
229 (MAX(0,sample_count - 33)/32)*sizeof(int),sizeof(void*));
230 tree_block_size = MAX((int)sizeof(CvDTreeNode)*8, max_split_size);
231 tree_block_size = MAX(tree_block_size + block_size_delta, min_block_size);
232 CV_CALL( tree_storage = cvCreateMemStorage( tree_block_size ));
233 CV_CALL( node_heap = cvCreateSet( 0, sizeof(*node_heap), sizeof(CvDTreeNode), tree_storage ));
234
235 nv_size = var_count*sizeof(int);
236 nv_size = cvAlign(MAX( nv_size, (int)sizeof(CvSetElem) ), sizeof(void*));
237
238 temp_block_size = nv_size;
239
240 if( cv_n )
241 {
242 if( sample_count < cv_n*MAX(params.min_sample_count,10) )
243 CV_ERROR( CV_StsOutOfRange,
244 "The many folds in cross-validation for such a small dataset" );
245
246 cv_size = cvAlign( cv_n*(sizeof(int) + sizeof(double)*2), sizeof(double) );
247 temp_block_size = MAX(temp_block_size, cv_size);
248 }
249
250 temp_block_size = MAX( temp_block_size + block_size_delta, min_block_size );
251 CV_CALL( temp_storage = cvCreateMemStorage( temp_block_size ));
252 CV_CALL( nv_heap = cvCreateSet( 0, sizeof(*nv_heap), nv_size, temp_storage ));
253 if( cv_size )
254 CV_CALL( cv_heap = cvCreateSet( 0, sizeof(*cv_heap), cv_size, temp_storage ));
255
256 CV_CALL( data_root = new_node( 0, sample_count, 0, 0 ));
257
258 max_c_count = 1;
259
260 _fdst = 0;
261 _idst = 0;
262 if (ord_var_count)
263 _fdst = (float*)cvAlloc(sample_count*sizeof(_fdst[0]));
264 if (is_buf_16u && (cat_var_count || is_classifier))
265 _idst = (int*)cvAlloc(sample_count*sizeof(_idst[0]));
266
267 // transform the training data to convenient representation
268 for( vi = 0; vi <= var_count; vi++ )
269 {
270 int ci;
271 const uchar* mask = 0;
272 int m_step = 0, step;
273 const int* idata = 0;
274 const float* fdata = 0;
275 int num_valid = 0;
276
277 if( vi < var_count ) // analyze i-th input variable
278 {
279 int vi0 = vidx ? vidx[vi] : vi;
280 ci = get_var_type(vi);
281 step = ds_step; m_step = ms_step;
282 if( CV_MAT_TYPE(_train_data->type) == CV_32SC1 )
283 idata = _train_data->data.i + vi0*dv_step;
284 else
285 fdata = _train_data->data.fl + vi0*dv_step;
286 if( _missing_mask )
287 mask = _missing_mask->data.ptr + vi0*mv_step;
288 }
289 else // analyze _responses
290 {
291 ci = cat_var_count;
292 step = CV_IS_MAT_CONT(_responses->type) ?
293 1 : _responses->step / CV_ELEM_SIZE(_responses->type);
294 if( CV_MAT_TYPE(_responses->type) == CV_32SC1 )
295 idata = _responses->data.i;
296 else
297 fdata = _responses->data.fl;
298 }
299
300 if( (vi < var_count && ci>=0) ||
301 (vi == var_count && is_classifier) ) // process categorical variable or response
302 {
303 int c_count, prev_label;
304 int* c_map;
305
306 if (is_buf_16u)
307 udst = (unsigned short*)(buf->data.s + ci*sample_count);
308 else
309 idst = buf->data.i + ci*sample_count;
310
311 // copy data
312 for( i = 0; i < sample_count; i++ )
313 {
314 int val = INT_MAX, si = sidx ? sidx[i] : i;
315 if( !mask || !mask[(size_t)si*m_step] )
316 {
317 if( idata )
318 val = idata[(size_t)si*step];
319 else
320 {
321 float t = fdata[(size_t)si*step];
322 val = cvRound(t);
323 if( val != t )
324 {
325 sprintf( err, "%d-th value of %d-th (categorical) "
326 "variable is not an integer", i, vi );
327 CV_ERROR( CV_StsBadArg, err );
328 }
329 }
330
331 if( val == INT_MAX )
332 {
333 sprintf( err, "%d-th value of %d-th (categorical) "
334 "variable is too large", i, vi );
335 CV_ERROR( CV_StsBadArg, err );
336 }
337 num_valid++;
338 }
339 if (is_buf_16u)
340 {
341 _idst[i] = val;
342 pair16u32s_ptr[i].u = udst + i;
343 pair16u32s_ptr[i].i = _idst + i;
344 }
345 else
346 {
347 idst[i] = val;
348 int_ptr[i] = idst + i;
349 }
350 }
351
352 c_count = num_valid > 0;
353
354 if (is_buf_16u)
355 {
356 icvSortPairs( pair16u32s_ptr, sample_count, 0 );
357 // count the categories
358 for( i = 1; i < num_valid; i++ )
359 if (*pair16u32s_ptr[i].i != *pair16u32s_ptr[i-1].i)
360 c_count ++ ;
361 }
362 else
363 {
364 icvSortIntPtr( int_ptr, sample_count, 0 );
365 // count the categories
366 for( i = 1; i < num_valid; i++ )
367 c_count += *int_ptr[i] != *int_ptr[i-1];
368 }
369
370 if( vi > 0 )
371 max_c_count = MAX( max_c_count, c_count );
372 cat_count->data.i[ci] = c_count;
373 cat_ofs->data.i[ci] = total_c_count;
374
375 // resize cat_map, if need
376 if( cat_map->cols < total_c_count + c_count )
377 {
378 tmp_map = cat_map;
379 CV_CALL( cat_map = cvCreateMat( 1,
380 MAX(cat_map->cols*3/2,total_c_count+c_count), CV_32SC1 ));
381 for( i = 0; i < total_c_count; i++ )
382 cat_map->data.i[i] = tmp_map->data.i[i];
383 cvReleaseMat( &tmp_map );
384 }
385
386 c_map = cat_map->data.i + total_c_count;
387 total_c_count += c_count;
388
389 c_count = -1;
390 if (is_buf_16u)
391 {
392 // compact the class indices and build the map
393 prev_label = ~*pair16u32s_ptr[0].i;
394 for( i = 0; i < num_valid; i++ )
395 {
396 int cur_label = *pair16u32s_ptr[i].i;
397 if( cur_label != prev_label )
398 c_map[++c_count] = prev_label = cur_label;
399 *pair16u32s_ptr[i].u = (unsigned short)c_count;
400 }
401 // replace labels for missing values with 65535
402 for( ; i < sample_count; i++ )
403 *pair16u32s_ptr[i].u = 65535;
404 }
405 else
406 {
407 // compact the class indices and build the map
408 prev_label = ~*int_ptr[0];
409 for( i = 0; i < num_valid; i++ )
410 {
411 int cur_label = *int_ptr[i];
412 if( cur_label != prev_label )
413 c_map[++c_count] = prev_label = cur_label;
414 *int_ptr[i] = c_count;
415 }
416 // replace labels for missing values with -1
417 for( ; i < sample_count; i++ )
418 *int_ptr[i] = -1;
419 }
420 }
421 else if( ci < 0 ) // process ordered variable
422 {
423 for( i = 0; i < sample_count; i++ )
424 {
425 float val = ord_nan;
426 int si = sidx ? sidx[i] : i;
427 if( !mask || !mask[(size_t)si*m_step] )
428 {
429 if( idata )
430 val = (float)idata[(size_t)si*step];
431 else
432 val = fdata[(size_t)si*step];
433
434 if( fabs(val) >= ord_nan )
435 {
436 sprintf( err, "%d-th value of %d-th (ordered) "
437 "variable (=%g) is too large", i, vi, val );
438 CV_ERROR( CV_StsBadArg, err );
439 }
440 num_valid++;
441 }
442 }
443 }
444 if( vi < var_count )
445 data_root->set_num_valid(vi, num_valid);
446 }
447
448 // set sample labels
449 if (is_buf_16u)
450 udst = (unsigned short*)(buf->data.s + get_work_var_count()*sample_count);
451 else
452 idst = buf->data.i + get_work_var_count()*sample_count;
453
454 for (i = 0; i < sample_count; i++)
455 {
456 if (udst)
457 udst[i] = sidx ? (unsigned short)sidx[i] : (unsigned short)i;
458 else
459 idst[i] = sidx ? sidx[i] : i;
460 }
461
462 if( cv_n )
463 {
464 unsigned short* usdst = 0;
465 int* idst2 = 0;
466
467 if (is_buf_16u)
468 {
469 usdst = (unsigned short*)(buf->data.s + (get_work_var_count()-1)*sample_count);
470 for( i = vi = 0; i < sample_count; i++ )
471 {
472 usdst[i] = (unsigned short)vi++;
473 vi &= vi < cv_n ? -1 : 0;
474 }
475
476 for( i = 0; i < sample_count; i++ )
477 {
478 int a = (*rng)(sample_count);
479 int b = (*rng)(sample_count);
480 unsigned short unsh = (unsigned short)vi;
481 CV_SWAP( usdst[a], usdst[b], unsh );
482 }
483 }
484 else
485 {
486 idst2 = buf->data.i + (get_work_var_count()-1)*sample_count;
487 for( i = vi = 0; i < sample_count; i++ )
488 {
489 idst2[i] = vi++;
490 vi &= vi < cv_n ? -1 : 0;
491 }
492
493 for( i = 0; i < sample_count; i++ )
494 {
495 int a = (*rng)(sample_count);
496 int b = (*rng)(sample_count);
497 CV_SWAP( idst2[a], idst2[b], vi );
498 }
499 }
500 }
501
502 if ( cat_map )
503 cat_map->cols = MAX( total_c_count, 1 );
504
505 max_split_size = cvAlign(sizeof(CvDTreeSplit) +
506 (MAX(0,max_c_count - 33)/32)*sizeof(int),sizeof(void*));
507 CV_CALL( split_heap = cvCreateSet( 0, sizeof(*split_heap), max_split_size, tree_storage ));
508
509 have_priors = is_classifier && params.priors;
510 if( is_classifier )
511 {
512 int m = get_num_classes();
513 double sum = 0;
514 CV_CALL( priors = cvCreateMat( 1, m, CV_64F ));
515 for( i = 0; i < m; i++ )
516 {
517 double val = have_priors ? params.priors[i] : 1.;
518 if( val <= 0 )
519 CV_ERROR( CV_StsOutOfRange, "Every class weight should be positive" );
520 priors->data.db[i] = val;
521 sum += val;
522 }
523
524 // normalize weights
525 if( have_priors )
526 cvScale( priors, priors, 1./sum );
527
528 CV_CALL( priors_mult = cvCloneMat( priors ));
529 CV_CALL( counts = cvCreateMat( 1, m, CV_32SC1 ));
530 }
531
532 CV_CALL( direction = cvCreateMat( 1, sample_count, CV_8UC1 ));
533 CV_CALL( split_buf = cvCreateMat( 1, sample_count, CV_32SC1 ));
534
535 __END__;
536
537 if( data )
538 delete data;
539
540 if ( pair16u32s_ptr )
541 cvFree( &pair16u32s_ptr );
542
543 if (_fdst)
544 cvFree( &_fdst );
545 if (_idst)
546 cvFree( &_idst );
547 cvFree( &int_ptr );
548 cvReleaseMat( &var_type0 );
549 cvReleaseMat( &sample_indices );
550 cvReleaseMat( &tmp_map );
551}

553void CvERTreeTrainData::get_ord_var_data( CvDTreeNode* n, int vi, float* ord_values_buf, int* missing_buf,
554 const float** ord_values, const int** missing, int* sample_indices_buf )
555{
556 int vidx = var_idx ? var_idx->data.i[vi] : vi;
557 int node_sample_count = n->sample_count;
558 // may use missing_buf as buffer for sample indices!
559 const int* sample_indices = get_sample_indices(n, sample_indices_buf ? sample_indices_buf : missing_buf);
560
561 int td_step = train_data->step/CV_ELEM_SIZE(train_data->type);
562 int m_step = missing_mask ? missing_mask->step/CV_ELEM_SIZE(missing_mask->type) : 1;
563 if( tflag == CV_ROW_SAMPLE )
564 {
565 for( int i = 0; i < node_sample_count; i++ )
566 {
567 int idx = sample_indices[i];
568 missing_buf[i] = missing_mask ? *(missing_mask->data.ptr + idx * m_step + vi) : 0;
569 ord_values_buf[i] = *(train_data->data.fl + idx * td_step + vidx);
570 }
571 }
572 else
573 for( int i = 0; i < node_sample_count; i++ )
574 {
575 int idx = sample_indices[i];
576 missing_buf[i] = missing_mask ? *(missing_mask->data.ptr + vi* m_step + idx) : 0;
577 ord_values_buf[i] = *(train_data->data.fl + vidx* td_step + idx);
578 }
579 *ord_values = ord_values_buf;
580 *missing = missing_buf;
581}

598const int* CvERTreeTrainData::get_cat_var_data( CvDTreeNode* n, int vi, int* cat_values_buf )
599{
600 int ci = get_var_type( vi);
601 const int* cat_values = 0;
602 if( !is_buf_16u )
603 cat_values = buf->data.i + n->buf_idx*get_length_subbuf() + ci*sample_count + n->offset;
604 else {
605 const unsigned short* short_values = (const unsigned short*)(buf->data.s + n->buf_idx*get_length_subbuf() +
606 ci*sample_count + n->offset);
607 for( int i = 0; i < n->sample_count; i++ )
608 cat_values_buf[i] = short_values[i];
609 cat_values = cat_values_buf;
610 }
611 return cat_values;
612}

614void CvERTreeTrainData::get_vectors( const CvMat* _subsample_idx,
615 float* values, uchar* missing,
616 float* _responses, bool get_class_idx )
617{
618 CvMat* subsample_idx = 0;
619 CvMat* subsample_co = 0;
620
621 cv::AutoBuffer<uchar> inn_buf(sample_count*(sizeof(float) + sizeof(int)));
622
623 CV_FUNCNAME( "CvERTreeTrainData::get_vectors" );
624
625 __BEGIN__;
626
627 int i, vi, total = sample_count, count = total, cur_ofs = 0;
628 int* sidx = 0;
629 int* co = 0;
630
631 if( _subsample_idx )
632 {
633 CV_CALL( subsample_idx = cvPreprocessIndexArray( _subsample_idx, sample_count ));
634 sidx = subsample_idx->data.i;

682 get_ord_var_data(data_root, vi, values + vi, mis_buf, &dst, &mis, 0);

686 }

688 // copy responses
689 if( _responses )

691 if( is_classifier )
692 {
693 int* lbls_buf = (int*)(uchar*)inn_buf;
694 const int* src = get_class_labels(data_root, lbls_buf);

714 }

716 __END__;

719 cvReleaseMat( &subsample_co );

758double CvForestERTree::calc_node_dir( CvDTreeNode* node )
759{
760 char* dir = (char*)data->direction->data.ptr;
761 int i, n = node->sample_count, vi = node->split->var_idx;
762 double L, R;
763
764 assert( !node->split->inversed );
765
766 if( data->get_var_type(vi) >= 0 ) // split on categorical var
767 {
768 cv::AutoBuffer<uchar> inn_buf(n*sizeof(int)*(!data->have_priors ? 1 : 2));
769 int* labels_buf = (int*)(uchar*)inn_buf;
770 const int* labels = data->get_cat_var_data( node, vi, labels_buf );
771 const int* subset = node->split->subset;
772 if( !data->have_priors )
773 {
774 int sum = 0, sum_abs = 0;
775
776 for( i = 0; i < n; i++ )
777 {
778 int idx = labels[i];
779 int d = ( ((idx >= 0)&&(!data->is_buf_16u)) || ((idx != 65535)&&(data->is_buf_16u)) ) ?
780 CV_DTREE_CAT_DIR(idx,subset) : 0;
781 sum += d; sum_abs += d & 1;
782 dir[i] = (char)d;
783 }
784
785 R = (sum_abs + sum) >> 1;
786 L = (sum_abs - sum) >> 1;
787 }
788 else
789 {
790 const double* priors = data->priors_mult->data.db;
791 double sum = 0, sum_abs = 0;
792 int *responses_buf = labels_buf + n;
793 const int* responses = data->get_class_labels(node, responses_buf);
794
795 for( i = 0; i < n; i++ )
796 {
797 int idx = labels[i];
798 double w = priors[responses[i]];
799 int d = idx >= 0 ? CV_DTREE_CAT_DIR(idx,subset) : 0;
800 sum += d*w; sum_abs += (d & 1)*w;
801 dir[i] = (char)d;
802 }
803
804 R = (sum_abs + sum) * 0.5;
805 L = (sum_abs - sum) * 0.5;
806 }
807 }
808 else // split on ordered var
809 {
810 float split_val = node->split->ord.c;
811 cv::AutoBuffer<uchar> inn_buf(n*(sizeof(int)*(!data->have_priors ? 1 : 2) + sizeof(float)));
812 float* val_buf = (float*)(uchar*)inn_buf;
813 int* missing_buf = (int*)(val_buf + n);
814 const float* val = 0;
815 const int* missing = 0;
816 data->get_ord_var_data( node, vi, val_buf, missing_buf, &val, &missing, 0 );
817
818 if( !data->have_priors )
819 {
820 L = R = 0;
821 for( i = 0; i < n; i++ )
822 {
823 if ( missing[i] )
824 dir[i] = (char)0;
825 else
826 {
827 if ( val[i] < split_val)
828 {
829 dir[i] = (char)-1;
830 L++;
831 }
832 else
833 {
834 dir[i] = (char)1;
835 R++;
836 }
837 }
838 }
839 }
840 else
841 {
842 const double* priors = data->priors_mult->data.db;
843 int* responses_buf = missing_buf + n;
844 const int* responses = data->get_class_labels(node, responses_buf);
845 L = R = 0;
846 for( i = 0; i < n; i++ )
847 {
848 if ( missing[i] )
849 dir[i] = (char)0;
850 else
851 {
852 double w = priors[responses[i]];
853 if ( val[i] < split_val)
854 {
855 dir[i] = (char)-1;
856 L += w;
857 }
858 else
859 {
860 dir[i] = (char)1;
861 R += w;
862 }
863 }
864 }
865 }
866 }
867
868 node->maxlr = MAX( L, R );
869 return node->split->quality/(L + R);
870}

872CvDTreeSplit* CvForestERTree::find_split_ord_class( CvDTreeNode* node, int vi, float init_quality, CvDTreeSplit* _split,
873 uchar* _ext_buf )
874{
875 const float epsilon = FLT_EPSILON*2;
876 const float split_delta = (1 + FLT_EPSILON) * FLT_EPSILON;
877
878 int n = node->sample_count;
879 int m = data->get_num_classes();
880
881 cv::AutoBuffer<uchar> inn_buf;
882 if( !_ext_buf )
883 inn_buf.allocate(n*(2*sizeof(int) + sizeof(float)));
884 uchar* ext_buf = _ext_buf ? _ext_buf : (uchar*)inn_buf;
885 float* values_buf = (float*)ext_buf;
886 int* missing_buf = (int*)(values_buf + n);
887 const float* values = 0;
888 const int* missing = 0;
889 data->get_ord_var_data( node, vi, values_buf, missing_buf, &values, &missing, 0 );
890 int* responses_buf = missing_buf + n;
891 const int* responses = data->get_class_labels( node, responses_buf );
892
893 double lbest_val = 0, rbest_val = 0, best_val = init_quality, split_val = 0;
894 const double* priors = data->have_priors ? data->priors_mult->data.db : 0;
895 bool is_find_split = false;
896 float pmin, pmax;
897 int smpi = 0;
898 while ( missing[smpi] && (smpi < n) )
899 smpi++;
900 assert(smpi < n);
901
902 pmin = values[smpi];
903 pmax = pmin;
904 for (; smpi < n; smpi++)
905 {
906 float ptemp = values[smpi];
907 int ms = missing[smpi];
908 if (ms) continue;
909 if ( ptemp < pmin)
910 pmin = ptemp;
911 if ( ptemp > pmax)
912 pmax = ptemp;
913 }
914 float fdiff = pmax-pmin;
915 if (fdiff > epsilon)
916 {
917 is_find_split = true;
918 cv::RNG* rng = data->rng;
919 split_val = pmin + rng->uniform(0.f, 1.f) * fdiff ;
920 if (split_val - pmin <= FLT_EPSILON)
921 split_val = pmin + split_delta;
922 if (pmax - split_val <= FLT_EPSILON)
923 split_val = pmax - split_delta;
924
925 // calculate Gini index
926 if ( !priors )
927 {
928 cv::AutoBuffer<int> lrc(m*2);
929 int *lc = lrc, *rc = lc + m;
930 int L = 0, R = 0;
931
932 // init arrays of class instance counters on both sides of the split
933 for(int i = 0; i < m; i++ )
934 {
935 lc[i] = 0;
936 rc[i] = 0;
937 }
938 for( int si = 0; si < n; si++ )
939 {
940 int r = responses[si];
941 float val = values[si];
942 int ms = missing[si];
943 if (ms) continue;
944 if ( val < split_val )
945 {
946 lc[r]++;
947 L++;
948 }
949 else
950 {
951 rc[r]++;
952 R++;
953 }
954 }
955 for (int i = 0; i < m; i++)
956 {
957 lbest_val += lc[i]*lc[i];
958 rbest_val += rc[i]*rc[i];
959 }
960 best_val = (lbest_val*R + rbest_val*L) / ((double)(L*R));
961 }
962 else
963 {
964 cv::AutoBuffer<double> lrc(m*2);
965 double *lc = lrc, *rc = lc + m;
966 double L = 0, R = 0;
967
968 // init arrays of class instance counters on both sides of the split
969 for(int i = 0; i < m; i++ )
970 {
971 lc[i] = 0;
972 rc[i] = 0;
973 }
974 for( int si = 0; si < n; si++ )
975 {
976 int r = responses[si];
977 float val = values[si];
978 int ms = missing[si];
979 double p = priors[r];
980 if (ms) continue;
981 if ( val < split_val )
982 {
983 lc[r] += p;
984 L += p;
985 }
986 else
987 {
988 rc[r] += p;
989 R += p;
990 }
991 }
992 for (int i = 0; i < m; i++)
993 {
994 lbest_val += lc[i]*lc[i];
995 rbest_val += rc[i]*rc[i];
996 }
997 best_val = (lbest_val*R + rbest_val*L) / (L*R);
998 }
999
1000 }
1001
1002 CvDTreeSplit* split = 0;
1003 if( is_find_split )
1004 {
1005 split = _split ? _split : data->new_split_ord( 0, 0.0f, 0, 0, 0.0f );
1006 split->var_idx = vi;
1007 split->ord.c = (float)split_val;
1008 split->ord.split_point = -1;
1009 split->inversed = 0;
1010 split->quality = (float)best_val;
1011 }
1012 return split;
1013}

1299 {
1300 CvRNG* rng = forest->get_rng();
1301 int l_cval_count = 1 + cvRandInt(rng) % (valid_ccount-1);
1302
1303 CvMat* var_class_mask = cvCreateMat( 1, valid_ccount, CV_8UC1 );
1304 CvMat submask;
1305 memset(var_class_mask->data.ptr, 0, valid_ccount*CV_ELEM_SIZE(var_class_mask->type));
1306 cvGetCols( var_class_mask, &submask, 0, l_cval_count );
1307 cvSet( &submask, cvScalar(1) );
1308 for (int i = 0; i < valid_ccount; i++)
1309 {
1310 uchar temp;
1311 int i1 = cvRandInt( rng ) % valid_ccount;
1312 int i2 = cvRandInt( rng ) % valid_ccount;
1313 CV_SWAP( var_class_mask->data.ptr[i1], var_class_mask->data.ptr[i2], temp );
1314 }
1315
1316 split = _split ? _split : data->new_split_cat( 0, -1.0f);
1317 split->var_idx = vi;
1318 memset( split->subset, 0, (data->max_c_count + 31)/32 * sizeof(int));
1319
1320 int L = 0, R = 0;
1321 for( int si = 0; si < n; si++ )
1322 {
1323 float r = responses[si];
1324 int var_class_idx = labels[si];
1325 if ( ((var_class_idx == 65535) && data->is_buf_16u) || ((var_class_idx<0) && (!data->is_buf_16u)) )
1326 continue;
1327 int mask_class_idx = valid_cidx[var_class_idx];
1328 if (var_class_mask->data.ptr[mask_class_idx])
1329 {
1330 lsum += r;
1331 L++;
1332 split->subset[var_class_idx >> 5] |= 1 << (var_class_idx & 31);
1333 }
1334 else
1335 {
1336 rsum += r;
1337 R++;
1338 }
1339 }
1340 best_val = (lsum*lsum*R + rsum*rsum*L)/((double)L*R);
1341
1342 split->quality = (float)best_val;
1343
1344 cvReleaseMat(&var_class_mask);
1345 }
1346 }
1347
1348 return split;

1485 {

1487 pos*scount + left->offset);

1620bool CvERTrees::grow_forest( const CvTermCriteria term_crit )
1621{
1622 bool result = false;
1623
1624 CvMat* sample_idx_for_tree = 0;
1625
1626 CV_FUNCNAME("CvERTrees::grow_forest");
1627 __BEGIN__;
1628
1629 const int max_ntrees = term_crit.max_iter;
1630 const double max_oob_err = term_crit.epsilon;
1631
1632 const int dims = data->var_count;
1633 float maximal_response = 0;
1634
1635 CvMat* oob_sample_votes = 0;
1636 CvMat* oob_responses = 0;
1637
1638 float* oob_samples_perm_ptr= 0;
1639
1640 float* samples_ptr = 0;
1641 uchar* missing_ptr = 0;
1642 float* true_resp_ptr = 0;
1643 bool is_oob_or_vimportance = ((max_oob_err > 0) && (term_crit.type != CV_TERMCRIT_ITER)) || var_importance;
1644
1645 // oob_predictions_sum[i] = sum of predicted values for the i-th sample
1646 // oob_num_of_predictions[i] = number of summands
1647 // (number of predictions for the i-th sample)
1648 // initialize these variable to avoid warning C4701
1649 CvMat oob_predictions_sum = cvMat( 1, 1, CV_32FC1 );
1650 CvMat oob_num_of_predictions = cvMat( 1, 1, CV_32FC1 );
1651
1652 nsamples = data->sample_count;
1653 nclasses = data->get_num_classes();
1654
1655 if ( is_oob_or_vimportance )
1656 {
1657 if( data->is_classifier )
1658 {
1659 CV_CALL(oob_sample_votes = cvCreateMat( nsamples, nclasses, CV_32SC1 ));
1660 cvZero(oob_sample_votes);
1661 }
1662 else
1663 {
1664 // oob_responses[0,i] = oob_predictions_sum[i]
1665 // = sum of predicted values for the i-th sample
1666 // oob_responses[1,i] = oob_num_of_predictions[i]
1667 // = number of summands (number of predictions for the i-th sample)
1668 CV_CALL(oob_responses = cvCreateMat( 2, nsamples, CV_32FC1 ));
1669 cvZero(oob_responses);
1670 cvGetRow( oob_responses, &oob_predictions_sum, 0 );
1671 cvGetRow( oob_responses, &oob_num_of_predictions, 1 );
1672 }
1673
1674 CV_CALL(oob_samples_perm_ptr = (float*)cvAlloc( sizeof(float)*nsamples*dims ));
1675 CV_CALL(samples_ptr = (float*)cvAlloc( sizeof(float)*nsamples*dims ));
1676 CV_CALL(missing_ptr = (uchar*)cvAlloc( sizeof(uchar)*nsamples*dims ));
1677 CV_CALL(true_resp_ptr = (float*)cvAlloc( sizeof(float)*nsamples ));
1678
1679 CV_CALL(data->get_vectors( 0, samples_ptr, missing_ptr, true_resp_ptr ));
1680 {
1681 double minval, maxval;
1682 CvMat responses = cvMat(1, nsamples, CV_32FC1, true_resp_ptr);
1683 cvMinMaxLoc( &responses, &minval, &maxval );
1684 maximal_response = (float)MAX( MAX( fabs(minval), fabs(maxval) ), 0 );
1685 }
1686 }
1687
1688 trees = (CvForestTree**)cvAlloc( sizeof(trees[0])*max_ntrees );
1689 memset( trees, 0, sizeof(trees[0])*max_ntrees );
1690
1691 CV_CALL(sample_idx_for_tree = cvCreateMat( 1, nsamples, CV_32SC1 ));
1692
1693 for (int i = 0; i < nsamples; i++)
1694 sample_idx_for_tree->data.i[i] = i;
1695 ntrees = 0;
1696 while( ntrees < max_ntrees )
1697 {
1698 int i, oob_samples_count = 0;
1699 double ncorrect_responses = 0; // used for estimation of variable importance
1700 CvForestTree* tree = 0;
1701
1702 trees[ntrees] = new CvForestERTree();
1703 tree = (CvForestERTree*)trees[ntrees];
1704 CV_CALL(tree->train( data, 0, this ));
1705
1706 if ( is_oob_or_vimportance )
1707 {
1708 CvMat sample, missing;
1709 // form array of OOB samples indices and get these samples
1710 sample = cvMat( 1, dims, CV_32FC1, samples_ptr );
1711 missing = cvMat( 1, dims, CV_8UC1, missing_ptr );
1712
1713 oob_error = 0;
1714 for( i = 0; i < nsamples; i++,
1715 sample.data.fl += dims, missing.data.ptr += dims )
1716 {
1717 CvDTreeNode* predicted_node = 0;
1718
1719 // predict oob samples
1720 if( !predicted_node )
1721 CV_CALL(predicted_node = tree->predict(&sample, &missing, true));
1722
1723 if( !data->is_classifier ) //regression
1724 {
1725 double avg_resp, resp = predicted_node->value;
1726 oob_predictions_sum.data.fl[i] += (float)resp;
1727 oob_num_of_predictions.data.fl[i] += 1;
1728
1729 // compute oob error
1730 avg_resp = oob_predictions_sum.data.fl[i]/oob_num_of_predictions.data.fl[i];
1731 avg_resp -= true_resp_ptr[i];
1732 oob_error += avg_resp*avg_resp;
1733 resp = (resp - true_resp_ptr[i])/maximal_response;
1734 ncorrect_responses += exp( -resp*resp );
1735 }
1736 else //classification
1737 {
1738 double prdct_resp;
1739 CvPoint max_loc;
1740 CvMat votes;
1741
1742 cvGetRow(oob_sample_votes, &votes, i);
1743 votes.data.i[predicted_node->class_idx]++;
1744
1745 // compute oob error
1746 cvMinMaxLoc( &votes, 0, 0, 0, &max_loc );
1747
1748 prdct_resp = data->cat_map->data.i[max_loc.x];
1749 oob_error += (fabs(prdct_resp - true_resp_ptr[i]) < FLT_EPSILON) ? 0 : 1;
1750
1751 ncorrect_responses += cvRound(predicted_node->value - true_resp_ptr[i]) == 0;
1752 }
1753 oob_samples_count++;
1754 }
1755 if( oob_samples_count > 0 )
1756 oob_error /= (double)oob_samples_count;
1757
1758 // estimate variable importance
1759 if( var_importance && oob_samples_count > 0 )
1760 {
1761 int m;
1762
1763 memcpy( oob_samples_perm_ptr, samples_ptr, dims*nsamples*sizeof(float));
1764 for( m = 0; m < dims; m++ )
1765 {
1766 double ncorrect_responses_permuted = 0;
1767 // randomly permute values of the m-th variable in the oob samples
1768 float* mth_var_ptr = oob_samples_perm_ptr + m;
1769
1770 for( i = 0; i < nsamples; i++ )
1771 {
1772 int i1, i2;
1773 float temp;
1774
1775 i1 = (*rng)(nsamples);
1776 i2 = (*rng)(nsamples);
1777 CV_SWAP( mth_var_ptr[i1*dims], mth_var_ptr[i2*dims], temp );
1778
1779 // turn values of (m-1)-th variable, that were permuted
1780 // at the previous iteration, untouched
1781 if( m > 1 )
1782 oob_samples_perm_ptr[i*dims+m-1] = samples_ptr[i*dims+m-1];
1783 }
1784
1785 // predict "permuted" cases and calculate the number of votes for the
1786 // correct class in the variable-m-permuted oob data
1787 sample = cvMat( 1, dims, CV_32FC1, oob_samples_perm_ptr );
1788 missing = cvMat( 1, dims, CV_8UC1, missing_ptr );
1789 for( i = 0; i < nsamples; i++,
1790 sample.data.fl += dims, missing.data.ptr += dims )
1791 {
1792 double predct_resp, true_resp;
1793
1794 predct_resp = tree->predict(&sample, &missing, true)->value;
1795 true_resp = true_resp_ptr[i];
1796 if( data->is_classifier )
1797 ncorrect_responses_permuted += cvRound(true_resp - predct_resp) == 0;
1798 else
1799 {
1800 true_resp = (true_resp - predct_resp)/maximal_response;
1801 ncorrect_responses_permuted += exp( -true_resp*true_resp );
1802 }
1803 }
1804 var_importance->data.fl[m] += (float)(ncorrect_responses
1805 - ncorrect_responses_permuted);
1806 }
1807 }
1808 }
1809 ntrees++;
1810 if( term_crit.type != CV_TERMCRIT_ITER && oob_error < max_oob_err )
1811 break;
1812 }
1813 if( var_importance )
1814 {
1815 for ( int vi = 0; vi < var_importance->cols; vi++ )
1816 var_importance->data.fl[vi] = ( var_importance->data.fl[vi] > 0 ) ?
1817 var_importance->data.fl[vi] : 0;
1818 cvNormalize( var_importance, var_importance, 1., 0, CV_L1 );
1819 }
1820
1821 result = true;
1822
1823 cvFree( &oob_samples_perm_ptr );
1824 cvFree( &samples_ptr );
1825 cvFree( &missing_ptr );
1826 cvFree( &true_resp_ptr );
1827
1828 cvReleaseMat( &sample_idx_for_tree );
1829
1830 cvReleaseMat( &oob_sample_votes );
1831 cvReleaseMat( &oob_responses );
1832
1833 __END__;
1834
1835 return result;
1836}
