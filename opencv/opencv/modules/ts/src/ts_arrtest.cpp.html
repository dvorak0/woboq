50ArrayTest::ArrayTest()
51{
52 test_case_count = default_test_case_count;

71void ArrayTest::clear()
72{
73 for( size_t i = 0; i < test_array.size(); i++ )
74 {
75 for( size_t j = 0; j < test_array[i].size(); j++ )
76 cvRelease( &test_array[i][j] );
77 }
78 BaseTest::clear();
79}

82int ArrayTest::read_params( CvFileStorage* fs )
83{
84 int code = BaseTest::read_params( fs );
85 if( code < 0 )
86 return code;
87
88 min_log_array_size = cvReadInt( find_param( fs, "min_log_array_size" ), min_log_array_size );
89 max_log_array_size = cvReadInt( find_param( fs, "max_log_array_size" ), max_log_array_size );
90 test_case_count = cvReadInt( find_param( fs, "test_case_count" ), test_case_count );
91 test_case_count = cvRound( test_case_count*ts->get_test_case_count_scale() );
92
93 min_log_array_size = clipInt( min_log_array_size, 0, 20 );
94 max_log_array_size = clipInt( max_log_array_size, min_log_array_size, 20 );
95 test_case_count = clipInt( test_case_count, 0, 100000 );
96
97 return code;
98}

101void ArrayTest::get_test_array_types_and_sizes( int /*test_case_idx*/, vector<vector<Size> >& sizes, vector<vector<int> >& types )
102{
103 RNG& rng = ts->get_rng();
104 Size size;
105 double val;
106 size_t i, j;
107
108 val = randReal(rng) * (max_log_array_size - min_log_array_size) + min_log_array_size;
109 size.width = cvRound( exp(val*CV_LOG2) );
110 val = randReal(rng) * (max_log_array_size - min_log_array_size) + min_log_array_size;
111 size.height = cvRound( exp(val*CV_LOG2) );
112
113 for( i = 0; i < test_array.size(); i++ )
114 {
115 size_t sizei = test_array[i].size();
116 for( j = 0; j < sizei; j++ )
117 {
118 sizes[i][j] = size;
119 types[i][j] = CV_8UC1;
120 }
121 }
122}

132int ArrayTest::prepare_test_case( int test_case_idx )
133{
134 int code = 1;
135 size_t max_arr = test_array.size();
136 vector<vector<Size> > sizes(max_arr);
137 vector<vector<Size> > whole_sizes(max_arr);
138 vector<vector<int> > types(max_arr);
139 size_t i, j;
140 RNG& rng = ts->get_rng();
141 bool is_image = false;
142
143 for( i = 0; i < max_arr; i++ )
144 {
145 size_t sizei = std::max(test_array[i].size(), (size_t)1);
146 sizes[i].resize(sizei);
147 types[i].resize(sizei);
148 whole_sizes[i].resize(sizei);
149 }
150
151 get_test_array_types_and_sizes( test_case_idx, sizes, types );
152
153 for( i = 0; i < max_arr; i++ )
154 {
155 size_t sizei = test_array[i].size();
156 for( j = 0; j < sizei; j++ )
157 {
158 unsigned t = randInt(rng);
159 bool create_mask = true, use_roi = false;
160 CvSize size = sizes[i][j], whole_size = size;
161 CvRect roi = {0,0,0,0};
162
163 is_image = !cvmat_allowed ? true : iplimage_allowed ? (t & 1) != 0 : false;
164 create_mask = (t & 6) == 0; // ~ each of 3 tests will use mask
165 use_roi = (t & 8) != 0;
166 if( use_roi )
167 {
168 whole_size.width += randInt(rng) % 10;
169 whole_size.height += randInt(rng) % 10;
170 }
171
172 cvRelease( &test_array[i][j] );
173 if( size.width > 0 && size.height > 0 &&
174 types[i][j] >= 0 && (i != MASK || create_mask) )
175 {
176 if( use_roi )
177 {
178 roi.width = size.width;
179 roi.height = size.height;
180 if( whole_size.width > size.width )
181 roi.x = randInt(rng) % (whole_size.width - size.width);
182
183 if( whole_size.height > size.height )
184 roi.y = randInt(rng) % (whole_size.height - size.height);
185 }
186
187 if( is_image )
188 {
189 test_array[i][j] = cvCreateImage( whole_size,
190 icvTsTypeToDepth[CV_MAT_DEPTH(types[i][j])], CV_MAT_CN(types[i][j]) );
191 if( use_roi )
192 cvSetImageROI( (IplImage*)test_array[i][j], roi );
193 }
194 else
195 {
196 test_array[i][j] = cvCreateMat( whole_size.height, whole_size.width, types[i][j] );
197 if( use_roi )
198 {
199 CvMat submat, *mat = (CvMat*)test_array[i][j];
200 cvGetSubRect( test_array[i][j], &submat, roi );
201 submat.refcount = mat->refcount;
202 *mat = submat;
203 }
204 }
205 }
206 }
207 }
208
209 test_mat.resize(test_array.size());
210 for( i = 0; i < max_arr; i++ )
211 {
212 size_t sizei = test_array[i].size();
213 test_mat[i].resize(sizei);
214 for( j = 0; j < sizei; j++ )
215 {
216 CvArr* arr = test_array[i][j];
217 test_mat[i][j] = cv::cvarrToMat(arr);
218 if( !test_mat[i][j].empty() )
219 fill_array( test_case_idx, (int)i, (int)j, test_mat[i][j] );
220 }
221 }
222
223 return code;
224}

227void ArrayTest::get_minmax_bounds( int i, int /*j*/, int type, Scalar& low, Scalar& high )
228{
229 double l, u;
230 int depth = CV_MAT_DEPTH(type);
231
232 if( i == MASK )
233 {
234 l = -2;
235 u = 2;
236 }
237 else if( depth < CV_32S )
238 {
239 l = getMinVal(type);
240 u = getMaxVal(type);
241 }
242 else
243 {
244 u = depth == CV_32S ? 1000000 : 1000.;
245 l = -u;
246 }
247
248 low = Scalar::all(l);
249 high = Scalar::all(u);
250}

253void ArrayTest::fill_array( int /*test_case_idx*/, int i, int j, Mat& arr )
254{
255 if( i == REF_INPUT_OUTPUT )
256 cvtest::copy( test_mat[INPUT_OUTPUT][j], arr, Mat() );
257 else if( i == INPUT || i == INPUT_OUTPUT || i == MASK )
258 {
259 Scalar low, high;
260
261 get_minmax_bounds( i, j, arr.type(), low, high );
262 randUni( ts->get_rng(), arr, low, high );
263 }
264}

267double ArrayTest::get_success_error_level( int /*test_case_idx*/, int i, int j )
268{
269 int elem_depth = CV_MAT_DEPTH(cvGetElemType(test_array[i][j]));
270 assert( i == OUTPUT || i == INPUT_OUTPUT );
271 return elem_depth < CV_32F ? 0 : elem_depth == CV_32F ? FLT_EPSILON*100: DBL_EPSILON*5000;
272}

281int ArrayTest::validate_test_results( int test_case_idx )
282{
283 static const char* arr_names[] = { "input", "input/output", "output",
284 "ref input/output", "ref output",
285 "temporary", "mask" };
286 size_t i, j;
287 prepare_to_validation( test_case_idx );
288
289 for( i = 0; i < 2; i++ )
290 {
291 int i0 = i == 0 ? OUTPUT : INPUT_OUTPUT;
292 int i1 = i == 0 ? REF_OUTPUT : REF_INPUT_OUTPUT;
293 size_t sizei = test_array[i0].size();
294
295 assert( sizei == test_array[i1].size() );
296 for( j = 0; j < sizei; j++ )
297 {
298 double err_level;
299 int code;
300
301 if( !test_array[i1][j] )
302 continue;
303
304 err_level = get_success_error_level( test_case_idx, i0, (int)j );
305 code = cmpEps2(ts, test_mat[i0][j], test_mat[i1][j], err_level, element_wise_relative_error, arr_names[i0]);
306
307 if (code == 0) continue;
308
309 for( i0 = 0; i0 < (int)test_array.size(); i0++ )
310 {
311 size_t sizei0 = test_array[i0].size();
312 if( i0 == REF_INPUT_OUTPUT || i0 == OUTPUT || i0 == TEMP )
313 continue;
314 for( i1 = 0; i1 < (int)sizei0; i1++ )
315 {
316 const Mat& arr = test_mat[i0][i1];
317 if( !arr.empty() )
318 {
319 string sizestr = vec2str(", ", &arr.size[0], arr.dims);
320 ts->printf( TS::LOG, "%s array %d type=%sC%d, size=(%s)\n",
321 arr_names[i0], i1, getTypeName(arr.depth()),
322 arr.channels(), sizestr.c_str() );
323 }
324 }
325 }
326 ts->set_failed_test_info( code );
327 return code;
328 }
329 }
330
331 return 0;
332}
