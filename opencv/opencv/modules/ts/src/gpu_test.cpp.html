89 //////////////////////////////////////////////////////////////////////
90 // GpuMat create
91
92 GpuMat createMat(Size size, int type, bool useRoi)
93 {
94 Size size0 = size;
95
96 if (useRoi)
97 {
98 size0.width += randomInt(5, 15);
99 size0.height += randomInt(5, 15);
100 }
101
102 GpuMat d_m(size0, type);
103
104 if (size0 != size)
105 d_m = d_m(Rect((size0.width - size.width) / 2, (size0.height - size.height) / 2, size.width, size.height));
106
107 return d_m;
108 }
109
110 GpuMat loadMat(const Mat& m, bool useRoi)
111 {
112 GpuMat d_m = createMat(m.size(), m.type(), useRoi);
113 d_m.upload(m);
114 return d_m;
115 }

126 {

128 if (CV_MAT_CN(type) == 4)
129 {
130 Mat temp;
131 cvtColor(src, temp, COLOR_BGR2BGRA);
132 swap(src, temp);
133 }
134 src.convertTo(src, CV_MAT_DEPTH(type), CV_MAT_DEPTH(type) == CV_32F ? 1.0 / 255.0 : 1.0);
135 return src;
136 }

152 void DeviceManager::load(int i)
153 {
154 devices_.clear();
155 devices_.reserve(1);
156
157 std::ostringstream msg;
158
159 if (i < 0 || i >= getCudaEnabledDeviceCount())
160 {
161 msg << "Incorrect device number - " << i;
162 throw std::runtime_error(msg.str());
163 }
164
165 DeviceInfo info(i);
166
167 if (!info.isCompatible())
168 {
169 msg << "Device " << i << " [" << info.name() << "] is NOT compatible with current GPU module build";
170 throw std::runtime_error(msg.str());
171 }
172
173 devices_.push_back(info);
174 }
175
176 void DeviceManager::loadAll()
177 {
178 int deviceCount = getCudaEnabledDeviceCount();
179
180 devices_.clear();
181 devices_.reserve(deviceCount);
182
183 for (int i = 0; i < deviceCount; ++i)
184 {
185 DeviceInfo info(i);
186 if (info.isCompatible())
187 {
188 devices_.push_back(info);
189 }
190 }

198 template <typename T, typename OutT> std::string printMatValImpl(const Mat& m, Point p)
199 {
200 const int cn = m.channels();
201
202 std::ostringstream ostr;
203 ostr << "(";
204
205 p.x /= cn;
206
207 ostr << static_cast<OutT>(m.at<T>(p.y, p.x * cn));
208 for (int c = 1; c < m.channels(); ++c)
209 {
210 ostr << ", " << static_cast<OutT>(m.at<T>(p.y, p.x * cn + c));
211 }
212 ostr << ")";
213
214 return ostr.str();
215 }
216
217 std::string printMatVal(const Mat& m, Point p)
218 {
219 typedef std::string (*func_t)(const Mat& m, Point p);
220
221 static const func_t funcs[] =
222 {
223 printMatValImpl<uchar, int>, printMatValImpl<schar, int>, printMatValImpl<ushort, int>, printMatValImpl<short, int>,
224 printMatValImpl<int, int>, printMatValImpl<float, float>, printMatValImpl<double, double>
225 };
226
227 return funcs[m.depth()](m, p);
228 }
229 }

231 void minMaxLocGold(const Mat& src, double* minVal_, double* maxVal_, Point* minLoc_, Point* maxLoc_, const Mat& mask)
232 {
233 if (src.depth() != CV_8S)
234 {
235 minMaxLoc(src, minVal_, maxVal_, minLoc_, maxLoc_, mask);
236 return;
237 }
238
239 // OpenCV's minMaxLoc doesn't support CV_8S type
240 double minVal = std::numeric_limits<double>::max();
241 Point minLoc(-1, -1);
242
243 double maxVal = -std::numeric_limits<double>::max();
244 Point maxLoc(-1, -1);
245
246 for (int y = 0; y < src.rows; ++y)
247 {
248 const schar* src_row = src.ptr<schar>(y);
249 const uchar* mask_row = mask.empty() ? 0 : mask.ptr<uchar>(y);
250
251 for (int x = 0; x < src.cols; ++x)
252 {
253 if (!mask_row || mask_row[x])
254 {
255 schar val = src_row[x];
256
257 if (val < minVal)
258 {
259 minVal = val;
260 minLoc = cv::Point(x, y);
261 }
262
263 if (val > maxVal)
264 {
265 maxVal = val;
266 maxLoc = cv::Point(x, y);
267 }
268 }
269 }
270 }
271
272 if (minVal_) *minVal_ = minVal;
273 if (maxVal_) *maxVal_ = maxVal;
274
275 if (minLoc_) *minLoc_ = minLoc;
276 if (maxLoc_) *maxLoc_ = maxLoc;
277 }

291 AssertionResult assertMatNear(const char* expr1, const char* expr2, const char* eps_expr, InputArray m1_, InputArray m2_, double eps)
292 {
293 Mat m1 = getMat(m1_);
294 Mat m2 = getMat(m2_);
295
296 if (m1.size() != m2.size())
297 {
298 return AssertionFailure() << "Matrices \"" << expr1 << "\" and \"" << expr2 << "\" have different sizes : \""
299 << expr1 << "\" [" << PrintToString(m1.size()) << "] vs \""
300 << expr2 << "\" [" << PrintToString(m2.size()) << "]";
301 }
302
303 if (m1.type() != m2.type())
304 {
305 return AssertionFailure() << "Matrices \"" << expr1 << "\" and \"" << expr2 << "\" have different types : \""
306 << expr1 << "\" [" << PrintToString(MatType(m1.type())) << "] vs \""
307 << expr2 << "\" [" << PrintToString(MatType(m2.type())) << "]";
308 }
309
310 Mat diff;
311 absdiff(m1.reshape(1), m2.reshape(1), diff);
312
313 double maxVal = 0.0;
314 Point maxLoc;
315 minMaxLocGold(diff, 0, &maxVal, 0, &maxLoc);
316
317 if (maxVal > eps)
318 {
319 return AssertionFailure() << "The max difference between matrices \"" << expr1 << "\" and \"" << expr2
320 << "\" is " << maxVal << " at (" << maxLoc.y << ", " << maxLoc.x / m1.channels() << ")"
321 << ", which exceeds \"" << eps_expr << "\", where \""
322 << expr1 << "\" at (" << maxLoc.y << ", " << maxLoc.x / m1.channels() << ") evaluates to " << printMatVal(m1, maxLoc) << ", \""
323 << expr2 << "\" at (" << maxLoc.y << ", " << maxLoc.x / m1.channels() << ") evaluates to " << printMatVal(m2, maxLoc) << ", \""
324 << eps_expr << "\" evaluates to " << eps;
325 }
326
327 return AssertionSuccess();
328 }

340 vector<MatType> types(int depth_start, int depth_end, int cn_start, int cn_end)
341 {
342 vector<MatType> v;
343
344 v.reserve((depth_end - depth_start + 1) * (cn_end - cn_start + 1));
345
346 for (int depth = depth_start; depth <= depth_end; ++depth)
347 {
348 for (int cn = cn_start; cn <= cn_end; ++cn)
349 {
350 v.push_back(MatType(CV_MAKE_TYPE(depth, cn)));
351 }
352 }
353
354 return v;
355 }

388 void showDiff(InputArray gold_, InputArray actual_, double eps)
389 {
390 Mat gold = getMat(gold_);
391 Mat actual = getMat(actual_);
392
393 Mat diff;
394 absdiff(gold, actual, diff);
395 threshold(diff, diff, eps, 255.0, cv::THRESH_BINARY);
396
397 namedWindow("gold", WINDOW_NORMAL);
398 namedWindow("actual", WINDOW_NORMAL);
399 namedWindow("diff", WINDOW_NORMAL);
400
401 imshow("gold", gold);
402 imshow("actual", actual);
403 imshow("diff", diff);
404
405 waitKey();
406 }

410 bool keyPointsEquals(const cv::KeyPoint& p1, const cv::KeyPoint& p2)
411 {
412 const double maxPtDif = 1.0;
413 const double maxSizeDif = 1.0;
414 const double maxAngleDif = 2.0;
415 const double maxResponseDif = 0.1;
416
417 double dist = cv::norm(p1.pt - p2.pt);
418
419 if (dist < maxPtDif &&
420 fabs(p1.size - p2.size) < maxSizeDif &&
421 std::abs(p1.angle - p2.angle) < maxAngleDif &&
422 std::abs(p1.response - p2.response) < maxResponseDif &&
423 p1.octave == p2.octave &&
424 p1.class_id == p2.class_id)
425 {
426 return true;
427 }
428
429 return false;
430 }
431
432 struct KeyPointLess : std::binary_function<cv::KeyPoint, cv::KeyPoint, bool>
433 {
434 bool operator()(const cv::KeyPoint& kp1, const cv::KeyPoint& kp2) const
435 {
436 return kp1.pt.y < kp2.pt.y || (kp1.pt.y == kp2.pt.y && kp1.pt.x < kp2.pt.x);
437 }
438 };
439 }
440
441 testing::AssertionResult assertKeyPointsEquals(const char* gold_expr, const char* actual_expr, std::vector<cv::KeyPoint>& gold, std::vector<cv::KeyPoint>& actual)
442 {
443 if (gold.size() != actual.size())
444 {
445 return testing::AssertionFailure() << "KeyPoints size mistmach\n"
446 << "\"" << gold_expr << "\" : " << gold.size() << "\n"
447 << "\"" << actual_expr << "\" : " << actual.size();
448 }
449
450 std::sort(actual.begin(), actual.end(), KeyPointLess());
451 std::sort(gold.begin(), gold.end(), KeyPointLess());
452
453 for (size_t i = 0; i < gold.size(); ++i)
454 {
455 const cv::KeyPoint& p1 = gold[i];
456 const cv::KeyPoint& p2 = actual[i];
457
458 if (!keyPointsEquals(p1, p2))
459 {
460 return testing::AssertionFailure() << "KeyPoints differ at " << i << "\n"
461 << "\"" << gold_expr << "\" vs \"" << actual_expr << "\" : \n"
462 << "pt : " << testing::PrintToString(p1.pt) << " vs " << testing::PrintToString(p2.pt) << "\n"
463 << "size : " << p1.size << " vs " << p2.size << "\n"
464 << "angle : " << p1.angle << " vs " << p2.angle << "\n"
465 << "response : " << p1.response << " vs " << p2.response << "\n"
466 << "octave : " << p1.octave << " vs " << p2.octave << "\n"
467 << "class_id : " << p1.class_id << " vs " << p2.class_id;
468 }
469 }
470
471 return ::testing::AssertionSuccess();
472 }
473
474 int getMatchedPointsCount(std::vector<cv::KeyPoint>& gold, std::vector<cv::KeyPoint>& actual)
475 {
476 std::sort(actual.begin(), actual.end(), KeyPointLess());
477 std::sort(gold.begin(), gold.end(), KeyPointLess());
478
479 int validCount = 0;
480
481 for (size_t i = 0; i < gold.size(); ++i)
482 {
483 const cv::KeyPoint& p1 = gold[i];
484 const cv::KeyPoint& p2 = actual[i];
485
486 if (keyPointsEquals(p1, p2))
487 ++validCount;
488 }
489
490 return validCount;
491 }
492
493 int getMatchedPointsCount(const std::vector<cv::KeyPoint>& keypoints1, const std::vector<cv::KeyPoint>& keypoints2, const std::vector<cv::DMatch>& matches)
494 {
495 int validCount = 0;
496
497 for (size_t i = 0; i < matches.size(); ++i)
498 {
499 const cv::DMatch& m = matches[i];
500
501 const cv::KeyPoint& p1 = keypoints1[m.queryIdx];
502 const cv::KeyPoint& p2 = keypoints2[m.trainIdx];
503
504 if (keyPointsEquals(p1, p2))
505 ++validCount;
506 }
507
508 return validCount;
509 }
