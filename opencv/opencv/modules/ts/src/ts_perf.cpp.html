71static void randu(cv::Mat& m)
72{
73 const int bigValue = 0x00000FFF;
74 if (m.depth() < CV_32F)
75 {
76 int minmax[] = {0, 256};
77 cv::Mat mr = cv::Mat(m.rows, (int)(m.cols * m.elemSize()), CV_8U, m.ptr(), m.step[0]);
78 cv::randu(mr, cv::Mat(1, 1, CV_32S, minmax), cv::Mat(1, 1, CV_32S, minmax + 1));
79 }
80 else if (m.depth() == CV_32F)
81 {
82 //float minmax[] = {-FLT_MAX, FLT_MAX};
83 float minmax[] = {-bigValue, bigValue};
84 cv::Mat mr = m.reshape(1);
85 cv::randu(mr, cv::Mat(1, 1, CV_32F, minmax), cv::Mat(1, 1, CV_32F, minmax + 1));
86 }
87 else
88 {
89 //double minmax[] = {-DBL_MAX, DBL_MAX};
90 double minmax[] = {-bigValue, bigValue};
91 cv::Mat mr = m.reshape(1);
92 cv::randu(mr, cv::Mat(1, 1, CV_64F, minmax), cv::Mat(1, 1, CV_64F, minmax + 1));
93 }
94}

155void Regression::init(const std::string& testSuitName, const std::string& ext)
156{
157 if (!storageInPath.empty())

159 LOGE("Subsequent initialization of Regression utility is not allowed.");
160 return;
161 }
162

164 const char *path_separator = "/";
165
166 if (data_path_dir)
167 {
168 int len = (int)strlen(data_path_dir)-1;
169 if (len < 0) len = 0;
170 std::string path_base = (data_path_dir[0] == 0 ? std::string(".") : std::string(data_path_dir))
171 + (data_path_dir[len] == '/' || data_path_dir[len] == '\\' ? "" : path_separator)
172 + "perf"
173 + path_separator;
174
175 storageInPath = path_base + testSuitName + ext;
176 storageOutPath = path_base + testSuitName;
177 }
178 else
179 {
180 storageInPath = testSuitName + ext;
181 storageOutPath = testSuitName;
182 }
183
184 suiteName = testSuitName;
185
186 try
187 {
188 if (storageIn.open(storageInPath, cv::FileStorage::READ))
189 {
190 rootIn = storageIn.root();
191 if (storageInPath.length() > 3 && storageInPath.substr(storageInPath.length()-3) == ".gz")

193 storageOutPath += ext;

195 }

197 {
198 LOGE("Failed to open sanity data for reading: %s", storageInPath.c_str());
199 }
200
201 if(!storageIn.isOpened())
202 storageOutPath = storageInPath;

241std::string Regression::getCurrentTestNodeName()
242{
243 const ::testing::TestInfo* const test_info =
244 ::testing::UnitTest::GetInstance()->current_test_info();
245
246 if (test_info == 0)
247 return "undefined";
248
249 std::string nodename = std::string(test_info->test_case_name()) + "--" + test_info->name();
250 size_t idx = nodename.find_first_of('/');
251 if (idx != std::string::npos)
252 nodename.erase(idx);
253
254 const char* type_param = test_info->type_param();
255 if (type_param != 0)
256 (nodename += "--") += type_param;
257
258 const char* value_param = test_info->value_param();
259 if (value_param != 0)
260 (nodename += "--") += value_param;
261
262 for(size_t i = 0; i < nodename.length(); ++i)
263 if (!isalnum(nodename[i]) && '_' != nodename[i])
264 nodename[i] = '-';
265
266 return nodename;

274double Regression::getElem(cv::Mat& m, int y, int x, int cn)
275{
276 switch (m.depth())
277 {
278 case CV_8U: return *(m.ptr<unsigned char>(y, x) + cn);
279 case CV_8S: return *(m.ptr<signed char>(y, x) + cn);
280 case CV_16U: return *(m.ptr<unsigned short>(y, x) + cn);
281 case CV_16S: return *(m.ptr<signed short>(y, x) + cn);
282 case CV_32S: return *(m.ptr<signed int>(y, x) + cn);
283 case CV_32F: return *(m.ptr<float>(y, x) + cn);
284 case CV_64F: return *(m.ptr<double>(y, x) + cn);
285 default: return 0;
286 }
287}

289void Regression::write(cv::Mat m)
290{
291 if (!m.empty() && m.dims < 2) return;
292
293 double min, max;
294 cv::minMaxIdx(m, &min, &max);
295 write() << "min" << min << "max" << max;
296
297 write() << "last" << "{" << "x" << m.size.p[1] - 1 << "y" << m.size.p[0] - 1
298 << "val" << getElem(m, m.size.p[0] - 1, m.size.p[1] - 1, m.channels() - 1) << "}";
299
300 int x, y, cn;
301 x = regRNG.uniform(0, m.size.p[1]);
302 y = regRNG.uniform(0, m.size.p[0]);
303 cn = regRNG.uniform(0, m.channels());
304 write() << "rng1" << "{" << "x" << x << "y" << y;
305 if(cn > 0) write() << "cn" << cn;
306 write() << "val" << getElem(m, y, x, cn) << "}";
307
308 x = regRNG.uniform(0, m.size.p[1]);
309 y = regRNG.uniform(0, m.size.p[0]);
310 cn = regRNG.uniform(0, m.channels());
311 write() << "rng2" << "{" << "x" << x << "y" << y;
312 if (cn > 0) write() << "cn" << cn;
313 write() << "val" << getElem(m, y, x, cn) << "}";
314}

316void Regression::verify(cv::FileNode node, cv::Mat actual, double eps, std::string argname, ERROR_TYPE err)
317{
318 if (!actual.empty() && actual.dims < 2) return;
319
320 double expect_min = (double)node["min"];
321 double expect_max = (double)node["max"];
322
323 if (err == ERROR_RELATIVE)
324 eps *= std::max(std::abs(expect_min), std::abs(expect_max));
325
326 double actual_min, actual_max;
327 cv::minMaxIdx(actual, &actual_min, &actual_max);
328
329 ASSERT_NEAR(expect_min, actual_min, eps)
330 << argname << " has unexpected minimal value" << std::endl;
331 ASSERT_NEAR(expect_max, actual_max, eps)
332 << argname << " has unexpected maximal value" << std::endl;
333
334 cv::FileNode last = node["last"];
335 double actual_last = getElem(actual, actual.size.p[0] - 1, actual.size.p[1] - 1, actual.channels() - 1);
336 int expect_cols = (int)last["x"] + 1;
337 int expect_rows = (int)last["y"] + 1;
338 ASSERT_EQ(expect_cols, actual.size.p[1])
339 << argname << " has unexpected number of columns" << std::endl;
340 ASSERT_EQ(expect_rows, actual.size.p[0])
341 << argname << " has unexpected number of rows" << std::endl;
342
343 double expect_last = (double)last["val"];
344 ASSERT_NEAR(expect_last, actual_last, eps)
345 << argname << " has unexpected value of the last element" << std::endl;
346
347 cv::FileNode rng1 = node["rng1"];
348 int x1 = rng1["x"];
349 int y1 = rng1["y"];
350 int cn1 = rng1["cn"];
351
352 double expect_rng1 = (double)rng1["val"];
353 // it is safe to use x1 and y1 without checks here because we have already
354 // verified that mat size is the same as recorded
355 double actual_rng1 = getElem(actual, y1, x1, cn1);
356
357 ASSERT_NEAR(expect_rng1, actual_rng1, eps)
358 << argname << " has unexpected value of the ["<< x1 << ":" << y1 << ":" << cn1 <<"] element" << std::endl;
359
360 cv::FileNode rng2 = node["rng2"];
361 int x2 = rng2["x"];
362 int y2 = rng2["y"];
363 int cn2 = rng2["cn"];
364
365 double expect_rng2 = (double)rng2["val"];
366 double actual_rng2 = getElem(actual, y2, x2, cn2);
367
368 ASSERT_NEAR(expect_rng2, actual_rng2, eps)
369 << argname << " has unexpected value of the ["<< x2 << ":" << y2 << ":" << cn2 <<"] element" << std::endl;
370}

399static int countViolations(const cv::Mat& expected, const cv::Mat& actual, const cv::Mat& diff, double eps, double* max_violation = 0, double* max_allowed = 0)
400{
401 cv::Mat diff64f;
402 diff.reshape(1).convertTo(diff64f, CV_64F);
403
404 cv::Mat expected_abs = cv::abs(expected.reshape(1));
405 cv::Mat actual_abs = cv::abs(actual.reshape(1));
406 cv::Mat maximum, mask;
407 cv::max(expected_abs, actual_abs, maximum);
408 cv::multiply(maximum, cv::Vec<double, 1>(eps), maximum, CV_64F);
409 cv::compare(diff64f, maximum, mask, cv::CMP_GT);
410
411 int v = cv::countNonZero(mask);
412
413 if (v > 0 && max_violation != 0 && max_allowed != 0)
414 {
415 int loc[10];
416 cv::minMaxIdx(maximum, 0, max_allowed, 0, loc, mask);
417 *max_violation = diff64f.at<double>(loc[1], loc[0]);
418 }
419
420 return v;
421}

423void Regression::verify(cv::FileNode node, cv::InputArray array, double eps, ERROR_TYPE err)
424{
425 int expected_kind = (int)node["kind"];
426 int expected_type = (int)node["type"];
427 ASSERT_EQ(expected_kind, array.kind()) << " Argument \"" << node.name() << "\" has unexpected kind";
428 ASSERT_EQ(expected_type, array.type()) << " Argument \"" << node.name() << "\" has unexpected type";
429
430 cv::FileNode valnode = node["val"];
431 if (isVector(array))
432 {
433 int expected_length = (int)node["len"];
434 ASSERT_EQ(expected_length, (int)array.total()) << " Vector \"" << node.name() << "\" has unexpected length";
435 int idx = node["idx"];
436
437 cv::Mat actual = array.getMat(idx);
438
439 if (valnode.isNone())
440 {
441 ASSERT_LE((size_t)26, actual.total() * (size_t)actual.channels())
442 << " \"" << node.name() << "[" << idx << "]\" has unexpected number of elements";
443 verify(node, actual, eps, cv::format("%s[%d]", node.name().c_str(), idx), err);
444 }
445 else
446 {
447 cv::Mat expected;
448 valnode >> expected;
449
450 if(expected.empty())
451 {
452 ASSERT_TRUE(actual.empty())
453 << " expected empty " << node.name() << "[" << idx<< "]";
454 }
455 else
456 {
457 ASSERT_EQ(expected.size(), actual.size())
458 << " " << node.name() << "[" << idx<< "] has unexpected size";
459
460 cv::Mat diff;
461 cv::absdiff(expected, actual, diff);
462
463 if (err == ERROR_ABSOLUTE)
464 {
465 if (!cv::checkRange(diff, true, 0, 0, eps))
466 {
467 if(expected.total() * expected.channels() < 12)
468 std::cout << " Expected: " << std::endl << expected << std::endl << " Actual:" << std::endl << actual << std::endl;
469
470 double max;
471 cv::minMaxIdx(diff.reshape(1), 0, &max);
472
473 FAIL() << " Absolute difference (=" << max << ") between argument \""
474 << node.name() << "[" << idx << "]\" and expected value is greater than " << eps;
475 }
476 }
477 else if (err == ERROR_RELATIVE)
478 {
479 double maxv, maxa;
480 int violations = countViolations(expected, actual, diff, eps, &maxv, &maxa);
481 if (violations > 0)
482 {
483 if(expected.total() * expected.channels() < 12)
484 std::cout << " Expected: " << std::endl << expected << std::endl << " Actual:" << std::endl << actual << std::endl;
485
486 FAIL() << " Relative difference (" << maxv << " of " << maxa << " allowed) between argument \""
487 << node.name() << "[" << idx << "]\" and expected value is greater than " << eps << " in " << violations << " points";
488 }
489 }
490 }
491 }
492 }
493 else
494 {
495 if (valnode.isNone())
496 {
497 ASSERT_LE((size_t)26, array.total() * (size_t)array.channels())
498 << " Argument \"" << node.name() << "\" has unexpected number of elements";
499 verify(node, array.getMat(), eps, "Argument \"" + node.name() + "\"", err);
500 }
501 else
502 {
503 cv::Mat expected;
504 valnode >> expected;
505 cv::Mat actual = array.getMat();
506
507 if(expected.empty())
508 {
509 ASSERT_TRUE(actual.empty())
510 << " expected empty " << node.name();
511 }
512 else
513 {
514 ASSERT_EQ(expected.size(), actual.size())
515 << " Argument \"" << node.name() << "\" has unexpected size";
516
517 cv::Mat diff;
518 cv::absdiff(expected, actual, diff);
519
520 if (err == ERROR_ABSOLUTE)
521 {
522 if (!cv::checkRange(diff, true, 0, 0, eps))
523 {
524 if(expected.total() * expected.channels() < 12)
525 std::cout << " Expected: " << std::endl << expected << std::endl << " Actual:" << std::endl << actual << std::endl;
526
527 double max;
528 cv::minMaxIdx(diff.reshape(1), 0, &max);
529
530 FAIL() << " Difference (=" << max << ") between argument1 \"" << node.name()
531 << "\" and expected value is greater than " << eps;
532 }
533 }
534 else if (err == ERROR_RELATIVE)
535 {
536 double maxv, maxa;
537 int violations = countViolations(expected, actual, diff, eps, &maxv, &maxa);
538 if (violations > 0)
539 {
540 if(expected.total() * expected.channels() < 12)
541 std::cout << " Expected: " << std::endl << expected << std::endl << " Actual:" << std::endl << actual << std::endl;
542
543 FAIL() << " Relative difference (" << maxv << " of " << maxa << " allowed) between argument \"" << node.name()
544 << "\" and expected value is greater than " << eps << " in " << violations << " points";
545 }
546 }
547 }
548 }
549 }

552Regression& Regression::operator() (const std::string& name, cv::InputArray array, double eps, ERROR_TYPE err)
553{
554 // exit if current test is already failed
555 if(::testing::UnitTest::GetInstance()->current_test_info()->result()->Failed()) return *this;
556
557 if(!array.empty() && array.depth() == CV_USRTYPE1)
558 {
559 ADD_FAILURE() << " Can not check regression for CV_USRTYPE1 data type for " << name;
560 return *this;
561 }
562
563 std::string nodename = getCurrentTestNodeName();
564
565 cv::FileNode n = rootIn[nodename];
566 if(n.isNone())
567 {
568 if(param_write_sanity)
569 {
570 if (nodename != currentTestNodeName)
571 {
572 if (!currentTestNodeName.empty())
573 write() << "}";
574 currentTestNodeName = nodename;
575
576 write() << nodename << "{";
577 }
578 // TODO: verify that name is alphanumeric, current error message is useless
579 write() << name << "{";
580 write(array);
581 write() << "}";
582 }
583 else if(param_verify_sanity)
584 {
585 ADD_FAILURE() << " No regression data for " << name << " argument";
586 }
587 }
588 else
589 {
590 cv::FileNode this_arg = n[name];
591 if (!this_arg.isMap())
592 ADD_FAILURE() << " No regression data for " << name << " argument";
593 else
594 verify(this_arg, array, eps, err);
595 }
596
597 return *this;
598}

638void TestBase::Init(const std::vector<std::string> & availableImpls,

640{
641 available_impls = availableImpls;
642
643 const std::string command_line_keys =
644 "{ |perf_max_outliers |8 |percent of allowed outliers}"
645 "{ |perf_min_samples |10 |minimal required numer of samples}"
646 "{ |perf_force_samples |100 |force set maximum number of samples for all tests}"
647 "{ |perf_seed |809564 |seed for random numbers generator}"
648 "{ |perf_threads |-1 |the number of worker threads, if parallel execution is enabled}"
649 "{ |perf_write_sanity |false |create new records for sanity checks}"
650 "{ |perf_verify_sanity |false |fail tests having no regression data for sanity checks}"
651 "{ |perf_impl |" + available_impls[0] +
652 "|the implementation variant of functions under test}"
653 "{ |perf_list_impls |false |list available implementation variants and exit}"
654 "{ |perf_run_cpu |false |deprecated, equivalent to --perf_impl=plain}"
655 "{ |perf_strategy |default |specifies performance measuring strategy: default, base or simple (weak restrictions)}"
656#ifdef ANDROID
657 "{ |perf_time_limit |6.0 |default time limit for a single test (in seconds)}"
658 "{ |perf_affinity_mask |0 |set affinity mask for the main thread}"

806int64 TestBase::_calibrate()
807{
808 class _helper : public ::perf::TestBase
809 {
810 public:
811 performance_metrics& getMetrics() { return calcMetrics(); }
812 virtual void TestBody() {}
813 virtual void PerfTestBody()
814 {
815 //the whole system warmup
816 SetUp();
817 cv::Mat a(2048, 2048, CV_32S, cv::Scalar(1));
818 cv::Mat b(2048, 2048, CV_32S, cv::Scalar(2));
819 declare.time(30);
820 double s = 0;
821 for(declare.iterations(20); startTimer(), next(); stopTimer())
822 s+=a.dot(b);
823 declare.time(s);
824
825 //self calibration
826 SetUp();
827 for(declare.iterations(1000); startTimer(), next(); stopTimer()){}
828 }
829 };
830
831 _timeadjustment = 0;
832 _helper h;
833 h.PerfTestBody();
834 double compensation = h.getMetrics().min;
835 if (param_strategy == PERF_STRATEGY_SIMPLE)
836 {
837 CV_Assert(compensation < 0.01 * cv::getTickFrequency());
838 compensation = 0.0f; // simple strategy doesn't require any compensation
839 }
840 LOGD("Time compensation is %.0f", compensation);
841 return (int64)compensation;
842}

901bool TestBase::next()
902{
903 static int64 lastActivityPrintTime = 0;
904
905 if (currentIter != (unsigned int)-1)
906 {
907 if (currentIter + 1 != times.size())
908 ADD_FAILURE() << " next() is called before stopTimer()";
909 }
910 else
911 {
912 lastActivityPrintTime = 0;
913 metrics.clear();
914 }
915
916 cv::theRNG().state = param_seed; //this rng should generate same numbers for each run
917 ++currentIter;
918
919 bool has_next = false;
920
921 do {
922 assert(currentIter == times.size());
923 if (currentIter == 0)
924 {
925 has_next = true;
926 break;
927 }
928
929 if (param_strategy == PERF_STRATEGY_BASE)
930 {
931 has_next = currentIter < nIters && totalTime < timeLimit;
932 }
933 else
934 {
935 assert(param_strategy == PERF_STRATEGY_SIMPLE);
936 if (totalTime - lastActivityPrintTime >= cv::getTickFrequency() * 10)
937 {
938 std::cout << '.' << std::endl;
939 lastActivityPrintTime = totalTime;
940 }
941 if (currentIter >= nIters)
942 {
943 has_next = false;
944 break;
945 }
946 if (currentIter < param_min_samples)
947 {
948 has_next = true;
949 break;
950 }
951
952 calcMetrics();
953
954 double criteria = 0.03; // 3%
955 if (fabs(metrics.mean) > 1e-6)
956 has_next = metrics.stddev > criteria * fabs(metrics.mean);
957 else
958 has_next = true;
959 }
960 } while (false);
961
962#ifdef ANDROID
963 if (log_power_checkpoints)
964 {
965 timeval tim;
966 gettimeofday(&tim, NULL);
967 unsigned long long t1 = tim.tv_sec * 1000LLU + (unsigned long long)(tim.tv_usec / 1000.f);
968
969 if (currentIter == 1) RecordProperty("test_start", cv::format("%llu",t1).c_str());
970 if (!has_next) RecordProperty("test_complete", cv::format("%llu",t1).c_str());
971 }
972#endif
973
974 if (has_next)
975 startTimer(); // really we should measure activity from this moment, so reset start time
976 return has_next;
977}

990 randu(m);

993 return;

1021 if (lastTime == 0)
1022 ADD_FAILURE() << " stopTimer() is called before startTimer()/next()";
1023 lastTime = time - lastTime;

1031performance_metrics& TestBase::calcMetrics()
1032{
1033 CV_Assert(metrics.samples <= (unsigned int)currentIter);
1034 if ((metrics.samples == (unsigned int)currentIter) || times.size() == 0)
1035 return metrics;
1036
1037 metrics.bytesIn = getTotalInputSize();
1038 metrics.bytesOut = getTotalOutputSize();
1039 metrics.frequency = cv::getTickFrequency();
1040 metrics.samples = (unsigned int)times.size();
1041 metrics.outliers = 0;
1042
1043 if (metrics.terminationReason != performance_metrics::TERM_INTERRUPT && metrics.terminationReason != performance_metrics::TERM_EXCEPTION)
1044 {
1045 if (currentIter == nIters)
1046 metrics.terminationReason = performance_metrics::TERM_ITERATIONS;
1047 else if (totalTime >= timeLimit)
1048 metrics.terminationReason = performance_metrics::TERM_TIME;
1049 else
1050 metrics.terminationReason = performance_metrics::TERM_UNKNOWN;
1051 }
1052
1053 std::sort(times.begin(), times.end());
1054
1055 TimeVector::const_iterator start = times.begin();
1056 TimeVector::const_iterator end = times.end();
1057
1058 if (param_strategy == PERF_STRATEGY_BASE)
1059 {
1060 //estimate mean and stddev for log(time)
1061 double gmean = 0;
1062 double gstddev = 0;
1063 int n = 0;
1064 for(TimeVector::const_iterator i = times.begin(); i != times.end(); ++i)
1065 {
1066 double x = static_cast<double>(*i)/runsPerIteration;
1067 if (x < DBL_EPSILON) continue;
1068 double lx = log(x);
1069
1070 ++n;
1071 double delta = lx - gmean;
1072 gmean += delta / n;
1073 gstddev += delta * (lx - gmean);
1074 }
1075
1076 gstddev = n > 1 ? sqrt(gstddev / (n - 1)) : 0;
1077
1078 //filter outliers assuming log-normal distribution
1079 //http://stackoverflow.com/questions/1867426/modeling-distribution-of-performance-measurements
1080 if (gstddev > DBL_EPSILON)
1081 {
1082 double minout = exp(gmean - 3 * gstddev) * runsPerIteration;
1083 double maxout = exp(gmean + 3 * gstddev) * runsPerIteration;
1084 while(*start < minout) ++start, ++metrics.outliers;
1085 do --end, ++metrics.outliers; while(*end > maxout);
1086 ++end, --metrics.outliers;
1087 }
1088 }
1089 else if (param_strategy == PERF_STRATEGY_SIMPLE)
1090 {
1091 metrics.outliers = static_cast<int>(times.size() * param_max_outliers / 100);
1092 for (unsigned int i = 0; i < metrics.outliers; i++)
1093 --end;
1094 }
1095 else
1096 {
1097 assert(false);
1098 }
1099
1100 int offset = static_cast<int>(start - times.begin());
1101
1102 metrics.min = static_cast<double>(*start)/runsPerIteration;
1103 //calc final metrics
1104 unsigned int n = 0;
1105 double gmean = 0;
1106 double gstddev = 0;
1107 double mean = 0;
1108 double stddev = 0;
1109 unsigned int m = 0;
1110 for(; start != end; ++start)
1111 {
1112 double x = static_cast<double>(*start)/runsPerIteration;
1113 if (x > DBL_EPSILON)
1114 {
1115 double lx = log(x);
1116 ++m;
1117 double gdelta = lx - gmean;
1118 gmean += gdelta / m;
1119 gstddev += gdelta * (lx - gmean);
1120 }
1121 ++n;
1122 double delta = x - mean;
1123 mean += delta / n;
1124 stddev += delta * (x - mean);
1125 }
1126
1127 metrics.mean = mean;
1128 metrics.gmean = exp(gmean);
1129 metrics.gstddev = m > 1 ? sqrt(gstddev / (m - 1)) : 0;
1130 metrics.stddev = n > 1 ? sqrt(stddev / (n - 1)) : 0;
1131 metrics.median = (n % 2
1132 ? (double)times[offset + n / 2]
1133 : 0.5 * (times[offset + n / 2] + times[offset + n / 2 - 1])
1134 ) / runsPerIteration;
1135
1136 return metrics;
1137}
1138
1139void TestBase::validateMetrics()
1140{
1141 performance_metrics& m = calcMetrics();
1142
1143 if (HasFailure()) return;
1144
1145 ASSERT_GE(m.samples, 1u)
1146 << " No time measurements was performed.\nstartTimer() and stopTimer() commands are required for performance tests.";
1147
1148 if (param_strategy == PERF_STRATEGY_BASE)
1149 {
1150 EXPECT_GE(m.samples, param_min_samples)
1151 << " Only a few samples are collected.\nPlease increase number of iterations or/and time limit to get reliable performance measurements.";
1152
1153 if (m.gstddev > DBL_EPSILON)
1154 {
1155 EXPECT_GT(/*m.gmean * */1., /*m.gmean * */ 2 * sinh(m.gstddev * param_max_deviation))
1156 << " Test results are not reliable ((mean-sigma,mean+sigma) deviation interval is greater than measured time interval).";
1157 }
1158
1159 EXPECT_LE(m.outliers, std::max((unsigned int)cvCeil(m.samples * param_max_outliers / 100.), 1u))
1160 << " Test results are not reliable (too many outliers).";
1161 }
1162 else if (param_strategy == PERF_STRATEGY_SIMPLE)
1163 {
1164 double mean = metrics.mean * 1000.0f / metrics.frequency;
1165 double stddev = metrics.stddev * 1000.0f / metrics.frequency;
1166 double percents = stddev / mean * 100.f;
1167 printf(" samples = %d, mean = %.2f, stddev = %.2f (%.1f%%)\n", (int)metrics.samples, mean, stddev, percents);
1168 }
1169 else
1170 {
1171 assert(false);
1172 }
1173}

1198 RecordProperty("mean", cv::format("%.0f", m.mean).c_str());
1199 RecordProperty("stddev", cv::format("%.0f", m.stddev).c_str());
1200 }
1201 else
1202 {
1203 const ::testing::TestInfo* const test_info = ::testing::UnitTest::GetInstance()->current_test_info();
1204 const char* type_param = test_info->type_param();
1205 const char* value_param = test_info->value_param();
1206
1207#if defined(ANDROID) && defined(USE_ANDROID_LOGGING)
1208 LOGD("[ FAILED ] %s.%s", test_info->test_case_name(), test_info->name());
1209#endif
1210
1211 if (type_param) LOGD("type = %11s", type_param);
1212 if (value_param) LOGD("params = %11s", value_param);
1213
1214 switch (m.terminationReason)
1215 {
1216 case performance_metrics::TERM_ITERATIONS:
1217 LOGD("termination reason: reached maximum number of iterations");
1218 break;
1219 case performance_metrics::TERM_TIME:
1220 LOGD("termination reason: reached time limit");
1221 break;
1222 case performance_metrics::TERM_INTERRUPT:
1223 LOGD("termination reason: aborted by the performance testing framework");
1224 break;
1225 case performance_metrics::TERM_EXCEPTION:
1226 LOGD("termination reason: unhandled exception");
1227 break;
1228 case performance_metrics::TERM_UNKNOWN:
1229 default:
1230 LOGD("termination reason: unknown");
1231 break;
1232 };
1233
1234 LOGD("bytesIn =%11lu", (unsigned long)m.bytesIn);
1235 LOGD("bytesOut =%11lu", (unsigned long)m.bytesOut);
1236 if (nIters == (unsigned int)-1 || m.terminationReason == performance_metrics::TERM_ITERATIONS)
1237 LOGD("samples =%11u", m.samples);
1238 else
1239 LOGD("samples =%11u of %u", m.samples, nIters);
1240 LOGD("outliers =%11u", m.outliers);
1241 LOGD("frequency =%11.0f", m.frequency);
1242 if (m.samples > 0)
1243 {
1244 LOGD("min =%11.0f = %.2fms", m.min, m.min * 1e3 / m.frequency);
1245 LOGD("median =%11.0f = %.2fms", m.median, m.median * 1e3 / m.frequency);
1246 LOGD("gmean =%11.0f = %.2fms", m.gmean, m.gmean * 1e3 / m.frequency);
1247 LOGD("gstddev =%11.8f = %.2fms for 97%% dispersion interval", m.gstddev, m.gmean * 2 * sinh(m.gstddev * 3) * 1e3 / m.frequency);
1248 LOGD("mean =%11.0f = %.2fms", m.mean, m.mean * 1e3 / m.frequency);
1249 LOGD("stddev =%11.0f = %.2fms", m.stddev, m.stddev * 1e3 / m.frequency);
1250 }
1251 }

1304std::string TestBase::getDataPath(const std::string& relativePath)
1305{
1306 if (relativePath.empty())
1307 {
1308 ADD_FAILURE() << " Bad path to test resource";
1309 throw PerfEarlyExitException();
1310 }
1311
1312 const char *data_path_dir = getenv("OPENCV_TEST_DATA_PATH");
1313 const char *path_separator = "/";
1314
1315 std::string path;
1316 if (data_path_dir)
1317 {
1318 int len = (int)strlen(data_path_dir) - 1;
1319 if (len < 0) len = 0;
1320 path = (data_path_dir[0] == 0 ? std::string(".") : std::string(data_path_dir))
1321 + (data_path_dir[len] == '/' || data_path_dir[len] == '\\' ? "" : path_separator);
1322 }
1323 else
1324 {
1325 path = ".";
1326 path += path_separator;
1327 }
1328
1329 if (relativePath[0] == '/' || relativePath[0] == '\\')
1330 path += relativePath.substr(1);
1331 else
1332 path += relativePath;
1333
1334 FILE* fp = fopen(path.c_str(), "r");
1335 if (fp)
1336 fclose(fp);
1337 else
1338 {

1492namespace {
1493struct KeypointComparator
1494{
1495 std::vector<cv::KeyPoint>& pts_;
1496 comparators::KeypointGreater cmp;
1497
1498 KeypointComparator(std::vector<cv::KeyPoint>& pts) : pts_(pts), cmp() {}
1499
1500 bool operator()(int idx1, int idx2) const
1501 {
1502 return cmp(pts_[idx1], pts_[idx2]);
1503 }
1504private:
1505 const KeypointComparator& operator=(const KeypointComparator&); // quiet MSVC
1506};
1507}//namespace
1508
1509void perf::sort(std::vector<cv::KeyPoint>& pts, cv::InputOutputArray descriptors)
1510{
1511 cv::Mat desc = descriptors.getMat();
1512
1513 CV_Assert(pts.size() == (size_t)desc.rows);
1514 cv::AutoBuffer<int> idxs(desc.rows);
1515
1516 for (int i = 0; i < desc.rows; ++i)
1517 idxs[i] = i;
1518
1519 std::sort((int*)idxs, (int*)idxs + desc.rows, KeypointComparator(pts));
1520
1521 std::vector<cv::KeyPoint> spts(pts.size());
1522 cv::Mat sdesc(desc.size(), desc.type());
1523
1524 for(int j = 0; j < desc.rows; ++j)
1525 {
1526 spts[j] = pts[idxs[j]];
1527 cv::Mat row = sdesc.row(j);
1528 desc.row(idxs[j]).copyTo(row);
1529 }
1530
1531 spts.swap(pts);
1532 sdesc.copyTo(desc);
1533}
