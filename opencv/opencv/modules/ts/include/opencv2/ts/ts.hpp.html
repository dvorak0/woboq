366 // test error codes
367 enum
368 {
369 // everything is Ok
370 OK=0,
371
372 // generic error: stub value to be used
373 // temporarily if the error's cause is unknown
374 FAIL_GENERIC=-1,
375
376 // the test is missing some essential data to proceed further
377 FAIL_MISSING_TEST_DATA=-2,
378
379 // the tested function raised an error via cxcore error handler
380 FAIL_ERROR_IN_CALLED_FUNC=-3,
381
382 // an exception has been raised;
383 // for memory and arithmetic exception
384 // there are two specialized codes (see below...)
385 FAIL_EXCEPTION=-4,
386
387 // a memory exception
388 // (access violation, access to missed page, stack overflow etc.)
389 FAIL_MEMORY_EXCEPTION=-5,
390
391 // arithmetic exception (overflow, division by zero etc.)
392 FAIL_ARITHM_EXCEPTION=-6,
393
394 // the tested function corrupted memory (no exception have been raised)
395 FAIL_MEMORY_CORRUPTION_BEGIN=-7,
396 FAIL_MEMORY_CORRUPTION_END=-8,
397
398 // the tested function (or test ifself) do not deallocate some memory
399 FAIL_MEMORY_LEAK=-9,
400
401 // the tested function returned invalid object, e.g. matrix, containing NaNs,
402 // structure with NULL or out-of-range fields (while it should not)
403 FAIL_INVALID_OUTPUT=-10,
404
405 // the tested function returned valid object, but it does not match to
406 // the original (or produced by the test) object
407 FAIL_MISMATCH=-11,
408
409 // the tested function returned valid object (a single number or numerical array),
410 // but it differs too much from the original (or produced by the test) object
411 FAIL_BAD_ACCURACY=-12,
412
413 // the tested function hung. Sometimes, can be determined by unexpectedly long
414 // processing time (in this case there should be possibility to interrupt such a function
415 FAIL_HANG=-13,
416
417 // unexpected response on passing bad arguments to the tested function
418 // (the function crashed, proceed successfully (while it should not), or returned
419 // error code that is different from what is expected)
420 FAIL_BAD_ARG_CHECK=-14,
421
422 // the test data (in whole or for the particular test case) is invalid
423 FAIL_INVALID_TEST_DATA=-15,
424
425 // the test has been skipped because it is not in the selected subset of the tests to run,
426 // because it has been run already within the same run with the same parameters, or because
427 // of some other reason and this is not considered as an error.
428 // Normally TS::run() (or overridden method in the derived class) takes care of what
429 // needs to be run, so this code should not occur.
430 SKIPPED=1
431 };

507class CV_EXPORTS BadArgTest : public BaseTest
508{
509public:
510 // constructor(s) and destructor
511 BadArgTest();
512 virtual ~BadArgTest();
513
514protected:
515 virtual int run_test_case( int expected_code, const string& descr );
516 virtual void run_func(void) = 0;
517 int test_case_idx;
518 int progress;
519 double t, freq;
520
521 template<class F>
522 int run_test_case( int expected_code, const string& _descr, F f)
523 {
524 double new_t = (double)cv::getTickCount(), dt;
525 if( test_case_idx < 0 )
526 {
527 test_case_idx = 0;
528 progress = 0;
529 dt = 0;
530 }
531 else
532 {
533 dt = (new_t - t)/(freq*1000);
534 t = new_t;
535 }
536 progress = update_progress(progress, test_case_idx, 0, dt);
537
538 int errcount = 0;
539 bool thrown = false;
540 const char* descr = _descr.c_str() ? _descr.c_str() : "";
541
542 try
543 {
544 f();
545 }
546 catch(const cv::Exception& e)
547 {
548 thrown = true;
549 if( e.code != expected_code )
550 {
551 ts->printf(TS::LOG, "%s (test case #%d): the error code %d is different from the expected %d\n",
552 descr, test_case_idx, e.code, expected_code);
553 errcount = 1;
554 }
555 }
556 catch(...)
557 {
558 thrown = true;
559 ts->printf(TS::LOG, "%s (test case #%d): unknown exception was thrown (the function has likely crashed)\n",
560 descr, test_case_idx);
561 errcount = 1;
562 }
563 if(!thrown)
564 {
565 ts->printf(TS::LOG, "%s (test case #%d): no expected exception was thrown\n",
566 descr, test_case_idx);
567 errcount = 1;
568 }
569 test_case_idx++;
570
571 return errcount;
572 }
573};
