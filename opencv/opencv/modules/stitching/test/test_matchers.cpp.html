47using namespace cv;
48using namespace std;

50TEST(SurfFeaturesFinder, CanFindInROIs)
51{
52 Ptr<detail::FeaturesFinder> finder = new detail::SurfFeaturesFinder();
53 Mat img = imread(string(cvtest::TS::ptr()->get_data_path()) + "cv/shared/lena.png");
54
55 vector<Rect> rois;
56 rois.push_back(Rect(0, 0, img.cols / 2, img.rows / 2));
57 rois.push_back(Rect(img.cols / 2, img.rows / 2, img.cols - img.cols / 2, img.rows - img.rows / 2));
58 detail::ImageFeatures roi_features;
59 (*finder)(img, roi_features, rois);
60
61 int tl_rect_count = 0, br_rect_count = 0, bad_count = 0;
62 for (size_t i = 0; i < roi_features.keypoints.size(); ++i)
63 {
64 Point2f pt = roi_features.keypoints[i].pt;
65 if (pt.x >= rois[0].x && pt.y >= rois[0].y && pt.x <= rois[0].br().x && pt.y <= rois[0].br().y)
66 tl_rect_count++;
67 else if (pt.x >= rois[1].x && pt.y >= rois[1].y && pt.x <= rois[1].br().x && pt.y <= rois[1].br().y)
68 br_rect_count++;
69 else
70 bad_count++;
71 }
72
73 ASSERT_GT(tl_rect_count, 0);
74 ASSERT_GT(br_rect_count, 0);
75 ASSERT_EQ(bad_count, 0);
76}
