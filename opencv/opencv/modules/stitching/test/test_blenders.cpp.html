42#include "test_precomp.hpp"
43
44using namespace cv;
45using namespace std;
46
47TEST(MultiBandBlender, CanBlendTwoImages)
48{
49 Mat image1 = imread(string(cvtest::TS::ptr()->get_data_path()) + "cv/shared/baboon.png");
50 Mat image2 = imread(string(cvtest::TS::ptr()->get_data_path()) + "cv/shared/lena.png");
51 ASSERT_EQ(image1.rows, image2.rows); ASSERT_EQ(image1.cols, image2.cols);
52
53 Mat image1s, image2s;
54 image1.convertTo(image1s, CV_16S);
55 image2.convertTo(image2s, CV_16S);
56
57 Mat mask1(image1s.size(), CV_8U);
58 mask1(Rect(0, 0, mask1.cols/2, mask1.rows)).setTo(255);
59 mask1(Rect(mask1.cols/2, 0, mask1.cols - mask1.cols/2, mask1.rows)).setTo(0);
60
61 Mat mask2(image2s.size(), CV_8U);
62 mask2(Rect(0, 0, mask2.cols/2, mask2.rows)).setTo(0);
63 mask2(Rect(mask2.cols/2, 0, mask2.cols - mask2.cols/2, mask2.rows)).setTo(255);
64
65 detail::MultiBandBlender blender(false, 5);
66
67 blender.prepare(Rect(0, 0, max(image1s.cols, image2s.cols), max(image1s.rows, image2s.rows)));
68 blender.feed(image1s, mask1, Point(0,0));
69 blender.feed(image2s, mask2, Point(0,0));
70
71 Mat result_s, result_mask;
72 blender.blend(result_s, result_mask);
73 Mat result; result_s.convertTo(result, CV_8U);
74
75 Mat expected = imread(string(cvtest::TS::ptr()->get_data_path()) + "stitching/baboon_lena.png");
76 double rmsErr = norm(expected, result, NORM_L2) / sqrt(double(expected.size().area()));
77 ASSERT_LT(rmsErr, 1e-3);
78}
