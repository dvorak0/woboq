50Ptr<ExposureCompensator> ExposureCompensator::createDefault(int type)
51{
52 if (type == NO)
53 return new NoExposureCompensator();
54 if (type == GAIN)
55 return new GainCompensator();
56 if (type == GAIN_BLOCKS)
57 return new BlocksGainCompensator();
58 CV_Error(CV_StsBadArg, "unsupported exposure compensation method");
59 return NULL;
60}

63void ExposureCompensator::feed(const vector<Point> &corners, const vector<Mat> &images,
64 const vector<Mat> &masks)
65{
66 vector<pair<Mat,uchar> > level_masks;
67 for (size_t i = 0; i < masks.size(); ++i)
68 level_masks.push_back(make_pair(masks[i], 255));
69 feed(corners, images, level_masks);
70}

73void GainCompensator::feed(const vector<Point> &corners, const vector<Mat> &images,
74 const vector<pair<Mat,uchar> > &masks)
75{
76 LOGLN("Exposure compensation...");
77#if ENABLE_LOG
78 int64 t = getTickCount();
79#endif
80
81 CV_Assert(corners.size() == images.size() && images.size() == masks.size());
82
83 const int num_images = static_cast<int>(images.size());
84 Mat_<int> N(num_images, num_images); N.setTo(0);
85 Mat_<double> I(num_images, num_images); I.setTo(0);
86
87 //Rect dst_roi = resultRoi(corners, images);
88 Mat subimg1, subimg2;
89 Mat_<uchar> submask1, submask2, intersect;
90
91 for (int i = 0; i < num_images; ++i)
92 {
93 for (int j = i; j < num_images; ++j)
94 {
95 Rect roi;
96 if (overlapRoi(corners[i], corners[j], images[i].size(), images[j].size(), roi))
97 {
98 subimg1 = images[i](Rect(roi.tl() - corners[i], roi.br() - corners[i]));
99 subimg2 = images[j](Rect(roi.tl() - corners[j], roi.br() - corners[j]));
100
101 submask1 = masks[i].first(Rect(roi.tl() - corners[i], roi.br() - corners[i]));
102 submask2 = masks[j].first(Rect(roi.tl() - corners[j], roi.br() - corners[j]));
103 intersect = (submask1 == masks[i].second) & (submask2 == masks[j].second);
104
105 N(i, j) = N(j, i) = max(1, countNonZero(intersect));
106
107 double Isum1 = 0, Isum2 = 0;
108 for (int y = 0; y < roi.height; ++y)
109 {
110 const Point3_<uchar>* r1 = subimg1.ptr<Point3_<uchar> >(y);
111 const Point3_<uchar>* r2 = subimg2.ptr<Point3_<uchar> >(y);
112 for (int x = 0; x < roi.width; ++x)
113 {
114 if (intersect(y, x))
115 {
116 Isum1 += sqrt(static_cast<double>(sqr(r1[x].x) + sqr(r1[x].y) + sqr(r1[x].z)));
117 Isum2 += sqrt(static_cast<double>(sqr(r2[x].x) + sqr(r2[x].y) + sqr(r2[x].z)));
118 }
119 }
120 }
121 I(i, j) = Isum1 / N(i, j);
122 I(j, i) = Isum2 / N(i, j);
123 }
124 }
125 }
126
127 double alpha = 0.01;
128 double beta = 100;
129
130 Mat_<double> A(num_images, num_images); A.setTo(0);
131 Mat_<double> b(num_images, 1); b.setTo(0);
132 for (int i = 0; i < num_images; ++i)
133 {
134 for (int j = 0; j < num_images; ++j)
135 {
136 b(i, 0) += beta * N(i, j);
137 A(i, i) += beta * N(i, j);
138 if (j == i) continue;
139 A(i, i) += 2 * alpha * I(i, j) * I(i, j) * N(i, j);
140 A(i, j) -= 2 * alpha * I(i, j) * I(j, i) * N(i, j);
141 }
142 }
143
144 solve(A, b, gains_);
145
146 LOGLN("Exposure compensation, time: " << ((getTickCount() - t) / getTickFrequency()) << " sec");
147}

150void GainCompensator::apply(int index, Point /*corner*/, Mat &image, const Mat &/*mask*/)
151{
152 image *= gains_(index, 0);
153}

156vector<double> GainCompensator::gains() const
157{
158 vector<double> gains_vec(gains_.rows);
159 for (int i = 0; i < gains_.rows; ++i)
160 gains_vec[i] = gains_(i, 0);
161 return gains_vec;
162}

165void BlocksGainCompensator::feed(const vector<Point> &corners, const vector<Mat> &images,
166 const vector<pair<Mat,uchar> > &masks)
167{
168 CV_Assert(corners.size() == images.size() && images.size() == masks.size());
169
170 const int num_images = static_cast<int>(images.size());
171
172 vector<Size> bl_per_imgs(num_images);
173 vector<Point> block_corners;
174 vector<Mat> block_images;
175 vector<pair<Mat,uchar> > block_masks;
176
177 // Construct blocks for gain compensator
178 for (int img_idx = 0; img_idx < num_images; ++img_idx)
179 {
180 Size bl_per_img((images[img_idx].cols + bl_width_ - 1) / bl_width_,
181 (images[img_idx].rows + bl_height_ - 1) / bl_height_);
182 int bl_width = (images[img_idx].cols + bl_per_img.width - 1) / bl_per_img.width;
183 int bl_height = (images[img_idx].rows + bl_per_img.height - 1) / bl_per_img.height;
184 bl_per_imgs[img_idx] = bl_per_img;
185 for (int by = 0; by < bl_per_img.height; ++by)
186 {
187 for (int bx = 0; bx < bl_per_img.width; ++bx)
188 {
189 Point bl_tl(bx * bl_width, by * bl_height);
190 Point bl_br(min(bl_tl.x + bl_width, images[img_idx].cols),
191 min(bl_tl.y + bl_height, images[img_idx].rows));
192
193 block_corners.push_back(corners[img_idx] + bl_tl);
194 block_images.push_back(images[img_idx](Rect(bl_tl, bl_br)));
195 block_masks.push_back(make_pair(masks[img_idx].first(Rect(bl_tl, bl_br)),
196 masks[img_idx].second));
197 }
198 }
199 }
200
201 GainCompensator compensator;
202 compensator.feed(block_corners, block_images, block_masks);
203 vector<double> gains = compensator.gains();
204 gain_maps_.resize(num_images);
205
206 Mat_<float> ker(1, 3);
207 ker(0,0) = 0.25; ker(0,1) = 0.5; ker(0,2) = 0.25;
208
209 int bl_idx = 0;
210 for (int img_idx = 0; img_idx < num_images; ++img_idx)
211 {
212 Size bl_per_img = bl_per_imgs[img_idx];
213 gain_maps_[img_idx].create(bl_per_img);
214
215 for (int by = 0; by < bl_per_img.height; ++by)
216 for (int bx = 0; bx < bl_per_img.width; ++bx, ++bl_idx)
217 gain_maps_[img_idx](by, bx) = static_cast<float>(gains[bl_idx]);
218
219 sepFilter2D(gain_maps_[img_idx], gain_maps_[img_idx], CV_32F, ker, ker);
220 sepFilter2D(gain_maps_[img_idx], gain_maps_[img_idx], CV_32F, ker, ker);
221 }
222}

225void BlocksGainCompensator::apply(int index, Point /*corner*/, Mat &image, const Mat &/*mask*/)
226{
227 CV_Assert(image.type() == CV_8UC3);
228
229 Mat_<float> gain_map;
230 if (gain_maps_[index].size() == image.size())
231 gain_map = gain_maps_[index];
232 else
233 resize(gain_maps_[index], gain_map, image.size(), 0, 0, INTER_LINEAR);
234
235 for (int y = 0; y < image.rows; ++y)
236 {
237 const float* gain_row = gain_map.ptr<float>(y);
238 Point3_<uchar>* row = image.ptr<Point3_<uchar> >(y);
239 for (int x = 0; x < image.cols; ++x)
240 {
241 row[x].x = saturate_cast<uchar>(row[x].x * gain_row[x]);
242 row[x].y = saturate_cast<uchar>(row[x].y * gain_row[x]);
243 row[x].z = saturate_cast<uchar>(row[x].z * gain_row[x]);
244 }
245 }
