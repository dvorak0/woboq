50static const float WEIGHT_EPS = 1e-5f;

52Ptr<Blender> Blender::createDefault(int type, bool try_gpu)
53{
54 if (type == NO)
55 return new Blender();
56 if (type == FEATHER)
57 return new FeatherBlender();
58 if (type == MULTI_BAND)
59 return new MultiBandBlender(try_gpu);
60 CV_Error(CV_StsBadArg, "unsupported blending method");
61 return NULL;
62}

81void Blender::feed(const Mat &img, const Mat &mask, Point tl)
82{
83 CV_Assert(img.type() == CV_16SC3);
84 CV_Assert(mask.type() == CV_8U);
85 int dx = tl.x - dst_roi_.x;
86 int dy = tl.y - dst_roi_.y;
87
88 for (int y = 0; y < img.rows; ++y)
89 {
90 const Point3_<short> *src_row = img.ptr<Point3_<short> >(y);
91 Point3_<short> *dst_row = dst_.ptr<Point3_<short> >(dy + y);
92 const uchar *mask_row = mask.ptr<uchar>(y);
93 uchar *dst_mask_row = dst_mask_.ptr<uchar>(dy + y);
94
95 for (int x = 0; x < img.cols; ++x)
96 {
97 if (mask_row[x])
98 dst_row[dx + x] = src_row[x];
99 dst_mask_row[dx + x] |= mask_row[x];
100 }
101 }
102}

105void Blender::blend(Mat &dst, Mat &dst_mask)
106{
107 dst_.setTo(Scalar::all(0), dst_mask_ == 0);
108 dst = dst_;
109 dst_mask = dst_mask_;
110 dst_.release();
111 dst_mask_.release();
112}

123void FeatherBlender::feed(const Mat &img, const Mat &mask, Point tl)
124{
125 CV_Assert(img.type() == CV_16SC3);
126 CV_Assert(mask.type() == CV_8U);
127
128 createWeightMap(mask, sharpness_, weight_map_);
129 int dx = tl.x - dst_roi_.x;
130 int dy = tl.y - dst_roi_.y;
131
132 for (int y = 0; y < img.rows; ++y)
133 {
134 const Point3_<short>* src_row = img.ptr<Point3_<short> >(y);
135 Point3_<short>* dst_row = dst_.ptr<Point3_<short> >(dy + y);
136 const float* weight_row = weight_map_.ptr<float>(y);
137 float* dst_weight_row = dst_weight_map_.ptr<float>(dy + y);
138
139 for (int x = 0; x < img.cols; ++x)
140 {
141 dst_row[dx + x].x += static_cast<short>(src_row[x].x * weight_row[x]);
142 dst_row[dx + x].y += static_cast<short>(src_row[x].y * weight_row[x]);
143 dst_row[dx + x].z += static_cast<short>(src_row[x].z * weight_row[x]);
144 dst_weight_row[dx + x] += weight_row[x];
145 }
146 }
147}

150void FeatherBlender::blend(Mat &dst, Mat &dst_mask)
151{
152 normalizeUsingWeightMap(dst_weight_map_, dst_);
153 dst_mask_ = dst_weight_map_ > WEIGHT_EPS;
154 Blender::blend(dst, dst_mask);
155}

158Rect FeatherBlender::createWeightMaps(const vector<Mat> &masks, const vector<Point> &corners,
159 vector<Mat> &weight_maps)
160{
161 weight_maps.resize(masks.size());
162 for (size_t i = 0; i < masks.size(); ++i)
163 createWeightMap(masks[i], sharpness_, weight_maps[i]);
164
165 Rect dst_roi = resultRoi(corners, masks);
166 Mat weights_sum(dst_roi.size(), CV_32F);
167 weights_sum.setTo(0);
168
169 for (size_t i = 0; i < weight_maps.size(); ++i)
170 {
171 Rect roi(corners[i].x - dst_roi.x, corners[i].y - dst_roi.y,
172 weight_maps[i].cols, weight_maps[i].rows);
173 weights_sum(roi) += weight_maps[i];
174 }
175
176 for (size_t i = 0; i < weight_maps.size(); ++i)
177 {
178 Rect roi(corners[i].x - dst_roi.x, corners[i].y - dst_roi.y,
179 weight_maps[i].cols, weight_maps[i].rows);
180 Mat tmp = weights_sum(roi);
181 tmp.setTo(1, tmp < numeric_limits<float>::epsilon());
182 divide(weight_maps[i], tmp, weight_maps[i]);
183 }
184
185 return dst_roi;
186}

189MultiBandBlender::MultiBandBlender(int try_gpu, int num_bands, int weight_type)
190{
191 setNumBands(num_bands);
192#if defined(HAVE_OPENCV_GPU) && !defined(DYNAMIC_CUDA_SUPPORT)
193 can_use_gpu_ = try_gpu && gpu::getCudaEnabledDeviceCount();
194#else
195 (void)try_gpu;
196 can_use_gpu_ = false;
197#endif
198 CV_Assert(weight_type == CV_32F || weight_type == CV_16S);
199 weight_type_ = weight_type;
200}

203void MultiBandBlender::prepare(Rect dst_roi)
204{
205 dst_roi_final_ = dst_roi;
206
207 // Crop unnecessary bands
208 double max_len = static_cast<double>(max(dst_roi.width, dst_roi.height));
209 num_bands_ = min(actual_num_bands_, static_cast<int>(ceil(log(max_len) / log(2.0))));
210
211 // Add border to the final image, to ensure sizes are divided by (1 << num_bands_)
212 dst_roi.width += ((1 << num_bands_) - dst_roi.width % (1 << num_bands_)) % (1 << num_bands_);
213 dst_roi.height += ((1 << num_bands_) - dst_roi.height % (1 << num_bands_)) % (1 << num_bands_);
214
215 Blender::prepare(dst_roi);
216
217 dst_pyr_laplace_.resize(num_bands_ + 1);
218 dst_pyr_laplace_[0] = dst_;
219
220 dst_band_weights_.resize(num_bands_ + 1);
221 dst_band_weights_[0].create(dst_roi.size(), weight_type_);
222 dst_band_weights_[0].setTo(0);
223
224 for (int i = 1; i <= num_bands_; ++i)
225 {
226 dst_pyr_laplace_[i].create((dst_pyr_laplace_[i - 1].rows + 1) / 2,
227 (dst_pyr_laplace_[i - 1].cols + 1) / 2, CV_16SC3);
228 dst_band_weights_[i].create((dst_band_weights_[i - 1].rows + 1) / 2,
229 (dst_band_weights_[i - 1].cols + 1) / 2, weight_type_);
230 dst_pyr_laplace_[i].setTo(Scalar::all(0));
231 dst_band_weights_[i].setTo(0);
232 }
233}

236void MultiBandBlender::feed(const Mat &img, const Mat &mask, Point tl)
237{
238 CV_Assert(img.type() == CV_16SC3 || img.type() == CV_8UC3);
239 CV_Assert(mask.type() == CV_8U);
240
241 // Keep source image in memory with small border
242 int gap = 3 * (1 << num_bands_);
243 Point tl_new(max(dst_roi_.x, tl.x - gap),
244 max(dst_roi_.y, tl.y - gap));
245 Point br_new(min(dst_roi_.br().x, tl.x + img.cols + gap),
246 min(dst_roi_.br().y, tl.y + img.rows + gap));
247
248 // Ensure coordinates of top-left, bottom-right corners are divided by (1 << num_bands_).
249 // After that scale between layers is exactly 2.
250 //
251 // We do it to avoid interpolation problems when keeping sub-images only. There is no such problem when
252 // image is bordered to have size equal to the final image size, but this is too memory hungry approach.
253 tl_new.x = dst_roi_.x + (((tl_new.x - dst_roi_.x) >> num_bands_) << num_bands_);
254 tl_new.y = dst_roi_.y + (((tl_new.y - dst_roi_.y) >> num_bands_) << num_bands_);
255 int width = br_new.x - tl_new.x;
256 int height = br_new.y - tl_new.y;
257 width += ((1 << num_bands_) - width % (1 << num_bands_)) % (1 << num_bands_);
258 height += ((1 << num_bands_) - height % (1 << num_bands_)) % (1 << num_bands_);
259 br_new.x = tl_new.x + width;
260 br_new.y = tl_new.y + height;
261 int dy = max(br_new.y - dst_roi_.br().y, 0);
262 int dx = max(br_new.x - dst_roi_.br().x, 0);
263 tl_new.x -= dx; br_new.x -= dx;
264 tl_new.y -= dy; br_new.y -= dy;
265
266 int top = tl.y - tl_new.y;
267 int left = tl.x - tl_new.x;
268 int bottom = br_new.y - tl.y - img.rows;
269 int right = br_new.x - tl.x - img.cols;
270
271 // Create the source image Laplacian pyramid
272 Mat img_with_border;
273 copyMakeBorder(img, img_with_border, top, bottom, left, right,
274 BORDER_REFLECT);
275 vector<Mat> src_pyr_laplace;
276 if (can_use_gpu_ && img_with_border.depth() == CV_16S)
277 createLaplacePyrGpu(img_with_border, num_bands_, src_pyr_laplace);
278 else
279 createLaplacePyr(img_with_border, num_bands_, src_pyr_laplace);
280
281 // Create the weight map Gaussian pyramid
282 Mat weight_map;
283 vector<Mat> weight_pyr_gauss(num_bands_ + 1);
284
285 if(weight_type_ == CV_32F)
286 {
287 mask.convertTo(weight_map, CV_32F, 1./255.);
288 }
289 else// weight_type_ == CV_16S
290 {
291 mask.convertTo(weight_map, CV_16S);
292 add(weight_map, 1, weight_map, mask != 0);
293 }
294
295 copyMakeBorder(weight_map, weight_pyr_gauss[0], top, bottom, left, right, BORDER_CONSTANT);
296
297 for (int i = 0; i < num_bands_; ++i)
298 pyrDown(weight_pyr_gauss[i], weight_pyr_gauss[i + 1]);
299
300 int y_tl = tl_new.y - dst_roi_.y;
301 int y_br = br_new.y - dst_roi_.y;
302 int x_tl = tl_new.x - dst_roi_.x;
303 int x_br = br_new.x - dst_roi_.x;
304
305 // Add weighted layer of the source image to the final Laplacian pyramid layer
306 if(weight_type_ == CV_32F)
307 {
308 for (int i = 0; i <= num_bands_; ++i)
309 {
310 for (int y = y_tl; y < y_br; ++y)
311 {
312 int y_ = y - y_tl;
313 const Point3_<short>* src_row = src_pyr_laplace[i].ptr<Point3_<short> >(y_);
314 Point3_<short>* dst_row = dst_pyr_laplace_[i].ptr<Point3_<short> >(y);
315 const float* weight_row = weight_pyr_gauss[i].ptr<float>(y_);
316 float* dst_weight_row = dst_band_weights_[i].ptr<float>(y);
317
318 for (int x = x_tl; x < x_br; ++x)
319 {
320 int x_ = x - x_tl;
321 dst_row[x].x += static_cast<short>(src_row[x_].x * weight_row[x_]);
322 dst_row[x].y += static_cast<short>(src_row[x_].y * weight_row[x_]);
323 dst_row[x].z += static_cast<short>(src_row[x_].z * weight_row[x_]);
324 dst_weight_row[x] += weight_row[x_];
325 }
326 }
327 x_tl /= 2; y_tl /= 2;
328 x_br /= 2; y_br /= 2;
329 }
330 }
331 else// weight_type_ == CV_16S
332 {
333 for (int i = 0; i <= num_bands_; ++i)
334 {
335 for (int y = y_tl; y < y_br; ++y)
336 {
337 int y_ = y - y_tl;
338 const Point3_<short>* src_row = src_pyr_laplace[i].ptr<Point3_<short> >(y_);
339 Point3_<short>* dst_row = dst_pyr_laplace_[i].ptr<Point3_<short> >(y);
340 const short* weight_row = weight_pyr_gauss[i].ptr<short>(y_);
341 short* dst_weight_row = dst_band_weights_[i].ptr<short>(y);
342
343 for (int x = x_tl; x < x_br; ++x)
344 {
345 int x_ = x - x_tl;
346 dst_row[x].x += short((src_row[x_].x * weight_row[x_]) >> 8);
347 dst_row[x].y += short((src_row[x_].y * weight_row[x_]) >> 8);
348 dst_row[x].z += short((src_row[x_].z * weight_row[x_]) >> 8);
349 dst_weight_row[x] += weight_row[x_];
350 }
351 }
352 x_tl /= 2; y_tl /= 2;
353 x_br /= 2; y_br /= 2;
354 }
355 }
356}

359void MultiBandBlender::blend(Mat &dst, Mat &dst_mask)
360{
361 for (int i = 0; i <= num_bands_; ++i)
362 normalizeUsingWeightMap(dst_band_weights_[i], dst_pyr_laplace_[i]);
363
364 if (can_use_gpu_)
365 restoreImageFromLaplacePyrGpu(dst_pyr_laplace_);
366 else
367 restoreImageFromLaplacePyr(dst_pyr_laplace_);
368
369 dst_ = dst_pyr_laplace_[0];
370 dst_ = dst_(Range(0, dst_roi_final_.height), Range(0, dst_roi_final_.width));
371 dst_mask_ = dst_band_weights_[0] > WEIGHT_EPS;
372 dst_mask_ = dst_mask_(Range(0, dst_roi_final_.height), Range(0, dst_roi_final_.width));
373 dst_pyr_laplace_.clear();
374 dst_band_weights_.clear();
375
376 Blender::blend(dst, dst_mask);
377}

383void normalizeUsingWeightMap(const Mat& weight, Mat& src)
384{
385#ifdef HAVE_TEGRA_OPTIMIZATION
386 if(tegra::normalizeUsingWeightMap(weight, src))
387 return;
388#endif
389 CV_Assert(src.type() == CV_16SC3);
390
391 if(weight.type() == CV_32FC1)
392 {
393 for (int y = 0; y < src.rows; ++y)
394 {
395 Point3_<short> *row = src.ptr<Point3_<short> >(y);
396 const float *weight_row = weight.ptr<float>(y);
397
398 for (int x = 0; x < src.cols; ++x)
399 {
400 row[x].x = static_cast<short>(row[x].x / (weight_row[x] + WEIGHT_EPS));
401 row[x].y = static_cast<short>(row[x].y / (weight_row[x] + WEIGHT_EPS));
402 row[x].z = static_cast<short>(row[x].z / (weight_row[x] + WEIGHT_EPS));
403 }
404 }
405 }
406 else
407 {
408 CV_Assert(weight.type() == CV_16SC1);
409
410 for (int y = 0; y < src.rows; ++y)
411 {
412 const short *weight_row = weight.ptr<short>(y);
413 Point3_<short> *row = src.ptr<Point3_<short> >(y);
414
415 for (int x = 0; x < src.cols; ++x)
416 {
417 int w = weight_row[x] + 1;
418 row[x].x = static_cast<short>((row[x].x << 8) / w);
419 row[x].y = static_cast<short>((row[x].y << 8) / w);
420 row[x].z = static_cast<short>((row[x].z << 8) / w);
421 }
422 }
423 }
424}

427void createWeightMap(const Mat &mask, float sharpness, Mat &weight)
428{
429 CV_Assert(mask.type() == CV_8U);
430 distanceTransform(mask, weight, CV_DIST_L1, 3);
431 threshold(weight * sharpness, weight, 1.f, 1.f, THRESH_TRUNC);
432}

435void createLaplacePyr(const Mat &img, int num_levels, vector<Mat> &pyr)
436{
437#ifdef HAVE_TEGRA_OPTIMIZATION
438 if(tegra::createLaplacePyr(img, num_levels, pyr))
439 return;
440#endif
441
442 pyr.resize(num_levels + 1);
443
444 if(img.depth() == CV_8U)
445 {
446 if(num_levels == 0)
447 {
448 img.convertTo(pyr[0], CV_16S);
449 return;
450 }
451
452 Mat downNext;
453 Mat current = img;
454 pyrDown(img, downNext);
455
456 for(int i = 1; i < num_levels; ++i)
457 {
458 Mat lvl_up;
459 Mat lvl_down;
460
461 pyrDown(downNext, lvl_down);
462 pyrUp(downNext, lvl_up, current.size());
463 subtract(current, lvl_up, pyr[i-1], noArray(), CV_16S);
464
465 current = downNext;
466 downNext = lvl_down;
467 }
468
469 {
470 Mat lvl_up;
471 pyrUp(downNext, lvl_up, current.size());
472 subtract(current, lvl_up, pyr[num_levels-1], noArray(), CV_16S);
473
474 downNext.convertTo(pyr[num_levels], CV_16S);
475 }
476 }
477 else
478 {
479 pyr[0] = img;
480 for (int i = 0; i < num_levels; ++i)
481 pyrDown(pyr[i], pyr[i + 1]);
482 Mat tmp;
483 for (int i = 0; i < num_levels; ++i)
484 {
485 pyrUp(pyr[i + 1], tmp, pyr[i].size());
486 subtract(pyr[i], tmp, pyr[i]);
487 }
488 }
489}

492void createLaplacePyrGpu(const Mat &img, int num_levels, vector<Mat> &pyr)
493{
494#if defined(HAVE_OPENCV_GPU) && !defined(DYNAMIC_CUDA_SUPPORT)
495 pyr.resize(num_levels + 1);
496
497 vector<gpu::GpuMat> gpu_pyr(num_levels + 1);
498 gpu_pyr[0].upload(img);
499 for (int i = 0; i < num_levels; ++i)
500 gpu::pyrDown(gpu_pyr[i], gpu_pyr[i + 1]);
501
502 gpu::GpuMat tmp;
503 for (int i = 0; i < num_levels; ++i)
504 {
505 gpu::pyrUp(gpu_pyr[i + 1], tmp);
506 gpu::subtract(gpu_pyr[i], tmp, gpu_pyr[i]);
507 gpu_pyr[i].download(pyr[i]);
508 }
509
510 gpu_pyr[num_levels].download(pyr[num_levels]);
511#else
512 (void)img;
513 (void)num_levels;
514 (void)pyr;
515 CV_Error(CV_StsNotImplemented, "CUDA optimization is unavailable");
516#endif
517}

519
520void restoreImageFromLaplacePyr(vector<Mat> &pyr)
521{
522 if (pyr.empty())
523 return;
524 Mat tmp;
525 for (size_t i = pyr.size() - 1; i > 0; --i)
526 {
527 pyrUp(pyr[i], tmp, pyr[i - 1].size());
528 add(tmp, pyr[i - 1], pyr[i - 1]);
529 }
530}
531
532
533void restoreImageFromLaplacePyrGpu(vector<Mat> &pyr)
534{
535#if defined(HAVE_OPENCV_GPU) && !defined(DYNAMIC_CUDA_SUPPORT)
536 if (pyr.empty())
537 return;
538
539 vector<gpu::GpuMat> gpu_pyr(pyr.size());
540 for (size_t i = 0; i < pyr.size(); ++i)
541 gpu_pyr[i].upload(pyr[i]);
542
543 gpu::GpuMat tmp;
544 for (size_t i = pyr.size() - 1; i > 0; --i)
545 {
546 gpu::pyrUp(gpu_pyr[i], tmp);
547 gpu::add(tmp, gpu_pyr[i - 1], gpu_pyr[i - 1]);
548 }
549
550 gpu_pyr[0].download(pyr[0]);
551#else
552 (void)pyr;
553 CV_Error(CV_StsNotImplemented, "CUDA optimization is unavailable");
554#endif
555}
