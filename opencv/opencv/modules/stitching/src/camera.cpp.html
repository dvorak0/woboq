45using namespace std;

47namespace cv {
48namespace detail {
49
50CameraParams::CameraParams() : focal(1), aspect(1), ppx(0), ppy(0),
51 R(Mat::eye(3, 3, CV_64F)), t(Mat::zeros(3, 1, CV_64F)) {}
52
53CameraParams::CameraParams(const CameraParams &other) { *this = other; }
54
55const CameraParams& CameraParams::operator =(const CameraParams &other)
56{
57 focal = other.focal;
58 ppx = other.ppx;
59 ppy = other.ppy;
60 aspect = other.aspect;
61 R = other.R.clone();
62 t = other.t.clone();
63 return *this;
64}
65
66Mat CameraParams::K() const
67{
68 Mat_<double> k = Mat::eye(3, 3, CV_64F);
69 k(0,0) = focal; k(0,2) = ppx;
70 k(1,1) = focal * aspect; k(1,2) = ppy;
71 return k;
72}
