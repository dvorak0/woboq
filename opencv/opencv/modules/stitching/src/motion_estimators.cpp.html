51struct IncDistance
52{
53 IncDistance(vector<int> &vdists) : dists(&vdists[0]) {}
54 void operator ()(const GraphEdge &edge) { dists[edge.to] = dists[edge.from] + 1; }
55 int* dists;
56};

58
59struct CalcRotation
60{
61 CalcRotation(int _num_images, const vector<MatchesInfo> &_pairwise_matches, vector<CameraParams> &_cameras)
62 : num_images(_num_images), pairwise_matches(&_pairwise_matches[0]), cameras(&_cameras[0]) {}
63
64 void operator ()(const GraphEdge &edge)
65 {
66 int pair_idx = edge.from * num_images + edge.to;
67
68 Mat_<double> K_from = Mat::eye(3, 3, CV_64F);
69 K_from(0,0) = cameras[edge.from].focal;
70 K_from(1,1) = cameras[edge.from].focal * cameras[edge.from].aspect;
71 K_from(0,2) = cameras[edge.from].ppx;
72 K_from(1,2) = cameras[edge.from].ppy;
73
74 Mat_<double> K_to = Mat::eye(3, 3, CV_64F);
75 K_to(0,0) = cameras[edge.to].focal;
76 K_to(1,1) = cameras[edge.to].focal * cameras[edge.to].aspect;
77 K_to(0,2) = cameras[edge.to].ppx;
78 K_to(1,2) = cameras[edge.to].ppy;
79
80 Mat R = K_from.inv() * pairwise_matches[pair_idx].H.inv() * K_to;
81 cameras[edge.to].R = cameras[edge.from].R * R;
82 }
83
84 int num_images;
85 const MatchesInfo* pairwise_matches;
86 CameraParams* cameras;
87};

92void calcDeriv(const Mat &err1, const Mat &err2, double h, Mat res)
93{
94 for (int i = 0; i < err1.rows; ++i)
95 res.at<double>(i, 0) = (err2.at<double>(i, 0) - err1.at<double>(i, 0)) / h;
96}

172void BundleAdjusterBase::estimate(const vector<ImageFeatures> &features,
173 const vector<MatchesInfo> &pairwise_matches,
174 vector<CameraParams> &cameras)
175{
176 LOG_CHAT("Bundle adjustment");
177#if ENABLE_LOG
178 int64 t = getTickCount();
179#endif
180
181 num_images_ = static_cast<int>(features.size());
182 features_ = &features[0];
183 pairwise_matches_ = &pairwise_matches[0];
184
185 setUpInitialCameraParams(cameras);
186
187 // Leave only consistent image pairs
188 edges_.clear();
189 for (int i = 0; i < num_images_ - 1; ++i)
190 {
191 for (int j = i + 1; j < num_images_; ++j)
192 {
193 const MatchesInfo& matches_info = pairwise_matches_[i * num_images_ + j];
194 if (matches_info.confidence > conf_thresh_)
195 edges_.push_back(make_pair(i, j));
196 }
197 }
198
199 // Compute number of correspondences
200 total_num_matches_ = 0;
201 for (size_t i = 0; i < edges_.size(); ++i)
202 total_num_matches_ += static_cast<int>(pairwise_matches[edges_[i].first * num_images_ +
203 edges_[i].second].num_inliers);
204
205 CvLevMarq solver(num_images_ * num_params_per_cam_,
206 total_num_matches_ * num_errs_per_measurement_,
207 term_criteria_);
208
209 Mat err, jac;
210 CvMat matParams = cam_params_;
211 cvCopy(&matParams, solver.param);
212
213 int iter = 0;
214 for(;;)
215 {
216 const CvMat* _param = 0;
217 CvMat* _jac = 0;
218 CvMat* _err = 0;
219
220 bool proceed = solver.update(_param, _jac, _err);
221
222 cvCopy(_param, &matParams);
223
224 if (!proceed || !_err)
225 break;
226
227 if (_jac)
228 {
229 calcJacobian(jac);
230 CvMat tmp = jac;
231 cvCopy(&tmp, _jac);
232 }
233
234 if (_err)
235 {
236 calcError(err);
237 LOG_CHAT(".");
238 iter++;
239 CvMat tmp = err;
240 cvCopy(&tmp, _err);
241 }
242 }
243
244 LOGLN_CHAT("");
245 LOGLN_CHAT("Bundle adjustment, final RMS error: " << sqrt(err.dot(err) / total_num_matches_));
246 LOGLN_CHAT("Bundle adjustment, iterations done: " << iter);
247
248 obtainRefinedCameraParams(cameras);
249
250 // Normalize motion to center image
251 Graph span_tree;
252 vector<int> span_tree_centers;
253 findMaxSpanningTree(num_images_, pairwise_matches, span_tree, span_tree_centers);
254 Mat R_inv = cameras[span_tree_centers[0]].R.inv();
255 for (int i = 0; i < num_images_; ++i)
256 cameras[i].R = R_inv * cameras[i].R;
257
258 LOGLN_CHAT("Bundle adjustment, time: " << ((getTickCount() - t) / getTickFrequency()) << " sec");
259}

264void BundleAdjusterReproj::setUpInitialCameraParams(const vector<CameraParams> &cameras)
265{
266 cam_params_.create(num_images_ * 7, 1, CV_64F);
267 SVD svd;
268 for (int i = 0; i < num_images_; ++i)
269 {
270 cam_params_.at<double>(i * 7, 0) = cameras[i].focal;
271 cam_params_.at<double>(i * 7 + 1, 0) = cameras[i].ppx;
272 cam_params_.at<double>(i * 7 + 2, 0) = cameras[i].ppy;
273 cam_params_.at<double>(i * 7 + 3, 0) = cameras[i].aspect;
274
275 svd(cameras[i].R, SVD::FULL_UV);
276 Mat R = svd.u * svd.vt;
277 if (determinant(R) < 0)
278 R *= -1;
279
280 Mat rvec;
281 Rodrigues(R, rvec);
282 CV_Assert(rvec.type() == CV_32F);
283 cam_params_.at<double>(i * 7 + 4, 0) = rvec.at<float>(0, 0);
284 cam_params_.at<double>(i * 7 + 5, 0) = rvec.at<float>(1, 0);
285 cam_params_.at<double>(i * 7 + 6, 0) = rvec.at<float>(2, 0);
286 }

291{
292 for (int i = 0; i < num_images_; ++i)
293 {
294 cameras[i].focal = cam_params_.at<double>(i * 7, 0);
295 cameras[i].ppx = cam_params_.at<double>(i * 7 + 1, 0);
296 cameras[i].ppy = cam_params_.at<double>(i * 7 + 2, 0);
297 cameras[i].aspect = cam_params_.at<double>(i * 7 + 3, 0);
298
299 Mat rvec(3, 1, CV_64F);
300 rvec.at<double>(0, 0) = cam_params_.at<double>(i * 7 + 4, 0);
301 rvec.at<double>(1, 0) = cam_params_.at<double>(i * 7 + 5, 0);
302 rvec.at<double>(2, 0) = cam_params_.at<double>(i * 7 + 6, 0);
303 Rodrigues(rvec, cameras[i].R);
304
305 Mat tmp;
306 cameras[i].R.convertTo(tmp, CV_32F);
307 cameras[i].R = tmp;
308 }

312void BundleAdjusterReproj::calcError(Mat &err)
313{
314 err.create(total_num_matches_ * 2, 1, CV_64F);
315
316 int match_idx = 0;
317 for (size_t edge_idx = 0; edge_idx < edges_.size(); ++edge_idx)
318 {
319 int i = edges_[edge_idx].first;
320 int j = edges_[edge_idx].second;
321 double f1 = cam_params_.at<double>(i * 7, 0);
322 double f2 = cam_params_.at<double>(j * 7, 0);
323 double ppx1 = cam_params_.at<double>(i * 7 + 1, 0);
324 double ppx2 = cam_params_.at<double>(j * 7 + 1, 0);
325 double ppy1 = cam_params_.at<double>(i * 7 + 2, 0);
326 double ppy2 = cam_params_.at<double>(j * 7 + 2, 0);
327 double a1 = cam_params_.at<double>(i * 7 + 3, 0);
328 double a2 = cam_params_.at<double>(j * 7 + 3, 0);
329
330 double R1[9];
331 Mat R1_(3, 3, CV_64F, R1);
332 Mat rvec(3, 1, CV_64F);
333 rvec.at<double>(0, 0) = cam_params_.at<double>(i * 7 + 4, 0);
334 rvec.at<double>(1, 0) = cam_params_.at<double>(i * 7 + 5, 0);
335 rvec.at<double>(2, 0) = cam_params_.at<double>(i * 7 + 6, 0);
336 Rodrigues(rvec, R1_);
337
338 double R2[9];
339 Mat R2_(3, 3, CV_64F, R2);
340 rvec.at<double>(0, 0) = cam_params_.at<double>(j * 7 + 4, 0);
341 rvec.at<double>(1, 0) = cam_params_.at<double>(j * 7 + 5, 0);
342 rvec.at<double>(2, 0) = cam_params_.at<double>(j * 7 + 6, 0);
343 Rodrigues(rvec, R2_);
344
345 const ImageFeatures& features1 = features_[i];
346 const ImageFeatures& features2 = features_[j];
347 const MatchesInfo& matches_info = pairwise_matches_[i * num_images_ + j];
348
349 Mat_<double> K1 = Mat::eye(3, 3, CV_64F);
350 K1(0,0) = f1; K1(0,2) = ppx1;
351 K1(1,1) = f1*a1; K1(1,2) = ppy1;
352
353 Mat_<double> K2 = Mat::eye(3, 3, CV_64F);
354 K2(0,0) = f2; K2(0,2) = ppx2;
355 K2(1,1) = f2*a2; K2(1,2) = ppy2;
356
357 Mat_<double> H = K2 * R2_.inv() * R1_ * K1.inv();
358
359 for (size_t k = 0; k < matches_info.matches.size(); ++k)
360 {
361 if (!matches_info.inliers_mask[k])
362 continue;
363
364 const DMatch& m = matches_info.matches[k];
365 Point2f p1 = features1.keypoints[m.queryIdx].pt;
366 Point2f p2 = features2.keypoints[m.trainIdx].pt;
367 double x = H(0,0)*p1.x + H(0,1)*p1.y + H(0,2);
368 double y = H(1,0)*p1.x + H(1,1)*p1.y + H(1,2);
369 double z = H(2,0)*p1.x + H(2,1)*p1.y + H(2,2);
370
371 err.at<double>(2 * match_idx, 0) = p2.x - x/z;
372 err.at<double>(2 * match_idx + 1, 0) = p2.y - y/z;
373 match_idx++;
374 }
375 }
376}

379void BundleAdjusterReproj::calcJacobian(Mat &jac)
380{
381 jac.create(total_num_matches_ * 2, num_images_ * 7, CV_64F);
382 jac.setTo(0);
383
384 double val;
385 const double step = 1e-4;
386
387 for (int i = 0; i < num_images_; ++i)
388 {
389 if (refinement_mask_.at<uchar>(0, 0))
390 {
391 val = cam_params_.at<double>(i * 7, 0);
392 cam_params_.at<double>(i * 7, 0) = val - step;
393 calcError(err1_);
394 cam_params_.at<double>(i * 7, 0) = val + step;
395 calcError(err2_);
396 calcDeriv(err1_, err2_, 2 * step, jac.col(i * 7));
397 cam_params_.at<double>(i * 7, 0) = val;
398 }
399 if (refinement_mask_.at<uchar>(0, 2))
400 {
401 val = cam_params_.at<double>(i * 7 + 1, 0);
402 cam_params_.at<double>(i * 7 + 1, 0) = val - step;
403 calcError(err1_);
404 cam_params_.at<double>(i * 7 + 1, 0) = val + step;
405 calcError(err2_);
406 calcDeriv(err1_, err2_, 2 * step, jac.col(i * 7 + 1));
407 cam_params_.at<double>(i * 7 + 1, 0) = val;
408 }
409 if (refinement_mask_.at<uchar>(1, 2))
410 {
411 val = cam_params_.at<double>(i * 7 + 2, 0);
412 cam_params_.at<double>(i * 7 + 2, 0) = val - step;
413 calcError(err1_);
414 cam_params_.at<double>(i * 7 + 2, 0) = val + step;
415 calcError(err2_);
416 calcDeriv(err1_, err2_, 2 * step, jac.col(i * 7 + 2));
417 cam_params_.at<double>(i * 7 + 2, 0) = val;
418 }
419 if (refinement_mask_.at<uchar>(1, 1))
420 {
421 val = cam_params_.at<double>(i * 7 + 3, 0);
422 cam_params_.at<double>(i * 7 + 3, 0) = val - step;
423 calcError(err1_);
424 cam_params_.at<double>(i * 7 + 3, 0) = val + step;
425 calcError(err2_);
426 calcDeriv(err1_, err2_, 2 * step, jac.col(i * 7 + 3));
427 cam_params_.at<double>(i * 7 + 3, 0) = val;
428 }
429 for (int j = 4; j < 7; ++j)
430 {
431 val = cam_params_.at<double>(i * 7 + j, 0);
432 cam_params_.at<double>(i * 7 + j, 0) = val - step;
433 calcError(err1_);
434 cam_params_.at<double>(i * 7 + j, 0) = val + step;
435 calcError(err2_);
436 calcDeriv(err1_, err2_, 2 * step, jac.col(i * 7 + j));
437 cam_params_.at<double>(i * 7 + j, 0) = val;
438 }
439 }
440}

487void BundleAdjusterRay::calcError(Mat &err)
488{
489 err.create(total_num_matches_ * 3, 1, CV_64F);
490
491 int match_idx = 0;
492 for (size_t edge_idx = 0; edge_idx < edges_.size(); ++edge_idx)
493 {
494 int i = edges_[edge_idx].first;
495 int j = edges_[edge_idx].second;
496 double f1 = cam_params_.at<double>(i * 4, 0);
497 double f2 = cam_params_.at<double>(j * 4, 0);
498
499 double R1[9];
500 Mat R1_(3, 3, CV_64F, R1);
501 Mat rvec(3, 1, CV_64F);
502 rvec.at<double>(0, 0) = cam_params_.at<double>(i * 4 + 1, 0);
503 rvec.at<double>(1, 0) = cam_params_.at<double>(i * 4 + 2, 0);
504 rvec.at<double>(2, 0) = cam_params_.at<double>(i * 4 + 3, 0);
505 Rodrigues(rvec, R1_);
506
507 double R2[9];
508 Mat R2_(3, 3, CV_64F, R2);
509 rvec.at<double>(0, 0) = cam_params_.at<double>(j * 4 + 1, 0);
510 rvec.at<double>(1, 0) = cam_params_.at<double>(j * 4 + 2, 0);
511 rvec.at<double>(2, 0) = cam_params_.at<double>(j * 4 + 3, 0);
512 Rodrigues(rvec, R2_);
513
514 const ImageFeatures& features1 = features_[i];
515 const ImageFeatures& features2 = features_[j];
516 const MatchesInfo& matches_info = pairwise_matches_[i * num_images_ + j];
517
518 Mat_<double> K1 = Mat::eye(3, 3, CV_64F);
519 K1(0,0) = f1; K1(0,2) = features1.img_size.width * 0.5;
520 K1(1,1) = f1; K1(1,2) = features1.img_size.height * 0.5;
521
522 Mat_<double> K2 = Mat::eye(3, 3, CV_64F);
523 K2(0,0) = f2; K2(0,2) = features2.img_size.width * 0.5;
524 K2(1,1) = f2; K2(1,2) = features2.img_size.height * 0.5;
525
526 Mat_<double> H1 = R1_ * K1.inv();
527 Mat_<double> H2 = R2_ * K2.inv();
528
529 for (size_t k = 0; k < matches_info.matches.size(); ++k)
530 {
531 if (!matches_info.inliers_mask[k])
532 continue;
533
534 const DMatch& m = matches_info.matches[k];
535
536 Point2f p1 = features1.keypoints[m.queryIdx].pt;
537 double x1 = H1(0,0)*p1.x + H1(0,1)*p1.y + H1(0,2);
538 double y1 = H1(1,0)*p1.x + H1(1,1)*p1.y + H1(1,2);
539 double z1 = H1(2,0)*p1.x + H1(2,1)*p1.y + H1(2,2);
540 double len = sqrt(x1*x1 + y1*y1 + z1*z1);
541 x1 /= len; y1 /= len; z1 /= len;
542
543 Point2f p2 = features2.keypoints[m.trainIdx].pt;
544 double x2 = H2(0,0)*p2.x + H2(0,1)*p2.y + H2(0,2);
545 double y2 = H2(1,0)*p2.x + H2(1,1)*p2.y + H2(1,2);
546 double z2 = H2(2,0)*p2.x + H2(2,1)*p2.y + H2(2,2);
547 len = sqrt(x2*x2 + y2*y2 + z2*z2);
548 x2 /= len; y2 /= len; z2 /= len;
549
550 double mult = sqrt(f1 * f2);
551 err.at<double>(3 * match_idx, 0) = mult * (x1 - x2);
552 err.at<double>(3 * match_idx + 1, 0) = mult * (y1 - y2);
553 err.at<double>(3 * match_idx + 2, 0) = mult * (z1 - z2);
554
555 match_idx++;
556 }
557 }
558}

561void BundleAdjusterRay::calcJacobian(Mat &jac)
562{
563 jac.create(total_num_matches_ * 3, num_images_ * 4, CV_64F);
564
565 double val;
566 const double step = 1e-3;
567
568 for (int i = 0; i < num_images_; ++i)
569 {
570 for (int j = 0; j < 4; ++j)
571 {
572 val = cam_params_.at<double>(i * 4 + j, 0);
573 cam_params_.at<double>(i * 4 + j, 0) = val - step;
574 calcError(err1_);
575 cam_params_.at<double>(i * 4 + j, 0) = val + step;
576 calcError(err2_);
577 calcDeriv(err1_, err2_, 2 * step, jac.col(i * 4 + j));
578 cam_params_.at<double>(i * 4 + j, 0) = val;
579 }
580 }

586void waveCorrect(vector<Mat> &rmats, WaveCorrectKind kind)
587{
588 LOGLN("Wave correcting...");
589#if ENABLE_LOG
590 int64 t = getTickCount();
591#endif
592 if (rmats.size() <= 1)
593 {
594 LOGLN("Wave correcting, time: " << ((getTickCount() - t) / getTickFrequency()) << " sec");
595 return;
596 }
597
598 Mat moment = Mat::zeros(3, 3, CV_32F);
599 for (size_t i = 0; i < rmats.size(); ++i)
600 {
601 Mat col = rmats[i].col(0);
602 moment += col * col.t();
603 }
604 Mat eigen_vals, eigen_vecs;
605 eigen(moment, eigen_vals, eigen_vecs);
606
607 Mat rg1;
608 if (kind == WAVE_CORRECT_HORIZ)
609 rg1 = eigen_vecs.row(2).t();
610 else if (kind == WAVE_CORRECT_VERT)
611 rg1 = eigen_vecs.row(0).t();
612 else
613 CV_Error(CV_StsBadArg, "unsupported kind of wave correction");
614
615 Mat img_k = Mat::zeros(3, 1, CV_32F);
616 for (size_t i = 0; i < rmats.size(); ++i)
617 img_k += rmats[i].col(2);
618 Mat rg0 = rg1.cross(img_k);
619 double rg0_norm = norm(rg0);
620
621 if( rg0_norm <= DBL_MIN )
622 {
623 return;
624 }
625
626 rg0 /= rg0_norm;
627
628 Mat rg2 = rg0.cross(rg1);
629
630 double conf = 0;
631 if (kind == WAVE_CORRECT_HORIZ)
632 {
633 for (size_t i = 0; i < rmats.size(); ++i)
634 conf += rg0.dot(rmats[i].col(0));
635 if (conf < 0)
636 {
637 rg0 *= -1;
638 rg1 *= -1;
639 }
640 }
641 else if (kind == WAVE_CORRECT_VERT)
642 {
643 for (size_t i = 0; i < rmats.size(); ++i)
644 conf -= rg1.dot(rmats[i].col(0));
645 if (conf < 0)
646 {
647 rg0 *= -1;
648 rg1 *= -1;
649 }
650 }
651
652 Mat R = Mat::zeros(3, 3, CV_32F);
653 Mat tmp = R.row(0);
654 Mat(rg0.t()).copyTo(tmp);
655 tmp = R.row(1);
656 Mat(rg1.t()).copyTo(tmp);
657 tmp = R.row(2);
658 Mat(rg2.t()).copyTo(tmp);
659
660 for (size_t i = 0; i < rmats.size(); ++i)
661 rmats[i] = R * rmats[i];
662
663 LOGLN("Wave correcting, time: " << ((getTickCount() - t) / getTickFrequency()) << " sec");
664}

669string matchesGraphAsString(vector<string> &pathes, vector<MatchesInfo> &pairwise_matches,
670 float conf_threshold)
671{
672 stringstream str;
673 str << "graph matches_graph{\n";
674
675 const int num_images = static_cast<int>(pathes.size());
676 set<pair<int,int> > span_tree_edges;
677 DisjointSets comps(num_images);
678
679 for (int i = 0; i < num_images; ++i)
680 {
681 for (int j = 0; j < num_images; ++j)
682 {
683 if (pairwise_matches[i*num_images + j].confidence < conf_threshold)
684 continue;
685 int comp1 = comps.findSetByElem(i);
686 int comp2 = comps.findSetByElem(j);
687 if (comp1 != comp2)
688 {
689 comps.mergeSets(comp1, comp2);
690 span_tree_edges.insert(make_pair(i, j));
691 }
692 }
693 }
694
695 for (set<pair<int,int> >::const_iterator itr = span_tree_edges.begin();
696 itr != span_tree_edges.end(); ++itr)
697 {
698 pair<int,int> edge = *itr;
699 if (span_tree_edges.find(edge) != span_tree_edges.end())
700 {
701 string name_src = pathes[edge.first];
702 size_t prefix_len = name_src.find_last_of("/\\");
703 if (prefix_len != string::npos) prefix_len++; else prefix_len = 0;
704 name_src = name_src.substr(prefix_len, name_src.size() - prefix_len);
705
706 string name_dst = pathes[edge.second];
707 prefix_len = name_dst.find_last_of("/\\");
708 if (prefix_len != string::npos) prefix_len++; else prefix_len = 0;
709 name_dst = name_dst.substr(prefix_len, name_dst.size() - prefix_len);
710
711 int pos = edge.first*num_images + edge.second;
712 str << "\"" << name_src << "\" -- \"" << name_dst << "\""
713 << "[label=\"Nm=" << pairwise_matches[pos].matches.size()
714 << ", Ni=" << pairwise_matches[pos].num_inliers
715 << ", C=" << pairwise_matches[pos].confidence << "\"];\n";
716 }
717 }
718
719 for (size_t i = 0; i < comps.size.size(); ++i)
720 {
721 if (comps.size[comps.findSetByElem((int)i)] == 1)
722 {
723 string name = pathes[i];
724 size_t prefix_len = name.find_last_of("/\\");
725 if (prefix_len != string::npos) prefix_len++; else prefix_len = 0;
726 name = name.substr(prefix_len, name.size() - prefix_len);
727 str << "\"" << name << "\";\n";
728 }
729 }
730
731 str << "}";
732 return str.str();
733}

735vector<int> leaveBiggestComponent(vector<ImageFeatures> &features, vector<MatchesInfo> &pairwise_matches,
736 float conf_threshold)
737{
738 const int num_images = static_cast<int>(features.size());
739
740 DisjointSets comps(num_images);
741 for (int i = 0; i < num_images; ++i)
742 {
743 for (int j = 0; j < num_images; ++j)
744 {
745 if (pairwise_matches[i*num_images + j].confidence < conf_threshold)
746 continue;
747 int comp1 = comps.findSetByElem(i);
748 int comp2 = comps.findSetByElem(j);
749 if (comp1 != comp2)
750 comps.mergeSets(comp1, comp2);
751 }
752 }
753
754 int max_comp = static_cast<int>(max_element(comps.size.begin(), comps.size.end()) - comps.size.begin());
755
756 vector<int> indices;
757 vector<int> indices_removed;
758 for (int i = 0; i < num_images; ++i)
759 if (comps.findSetByElem(i) == max_comp)
760 indices.push_back(i);
761 else
762 indices_removed.push_back(i);
763
764 vector<ImageFeatures> features_subset;
765 vector<MatchesInfo> pairwise_matches_subset;
766 for (size_t i = 0; i < indices.size(); ++i)
767 {
768 features_subset.push_back(features[indices[i]]);
769 for (size_t j = 0; j < indices.size(); ++j)
770 {
771 pairwise_matches_subset.push_back(pairwise_matches[indices[i]*num_images + indices[j]]);
772 pairwise_matches_subset.back().src_img_idx = static_cast<int>(i);
773 pairwise_matches_subset.back().dst_img_idx = static_cast<int>(j);
774 }
775 }
776
777 if (static_cast<int>(features_subset.size()) == num_images)
778 return indices;
779
780 LOG("Removed some images, because can't match them or there are too similar images: (");
781 LOG(indices_removed[0] + 1);
782 for (size_t i = 1; i < indices_removed.size(); ++i)
783 LOG(", " << indices_removed[i]+1);
784 LOGLN(").");
785 LOGLN("Try to decrease the match confidence threshold and/or check if you're stitching duplicates.");
786
787 features = features_subset;
788 pairwise_matches = pairwise_matches_subset;
789
790 return indices;
791}

794void findMaxSpanningTree(int num_images, const vector<MatchesInfo> &pairwise_matches,
795 Graph &span_tree, vector<int> &centers)
796{
797 Graph graph(num_images);
798 vector<GraphEdge> edges;
799
800 // Construct images graph and remember its edges
801 for (int i = 0; i < num_images; ++i)
802 {
803 for (int j = 0; j < num_images; ++j)
804 {
805 if (pairwise_matches[i * num_images + j].H.empty())
806 continue;
807 float conf = static_cast<float>(pairwise_matches[i * num_images + j].num_inliers);
808 graph.addEdge(i, j, conf);
809 edges.push_back(GraphEdge(i, j, conf));
810 }
811 }
812
813 DisjointSets comps(num_images);
814 span_tree.create(num_images);
815 vector<int> span_tree_powers(num_images, 0);
816
817 // Find maximum spanning tree
818 sort(edges.begin(), edges.end(), greater<GraphEdge>());
819 for (size_t i = 0; i < edges.size(); ++i)
820 {
821 int comp1 = comps.findSetByElem(edges[i].from);
822 int comp2 = comps.findSetByElem(edges[i].to);
823 if (comp1 != comp2)
824 {
825 comps.mergeSets(comp1, comp2);
826 span_tree.addEdge(edges[i].from, edges[i].to, edges[i].weight);
827 span_tree.addEdge(edges[i].to, edges[i].from, edges[i].weight);
828 span_tree_powers[edges[i].from]++;
829 span_tree_powers[edges[i].to]++;
830 }
831 }
832
833 // Find spanning tree leafs
834 vector<int> span_tree_leafs;
835 for (int i = 0; i < num_images; ++i)
836 if (span_tree_powers[i] == 1)
837 span_tree_leafs.push_back(i);
838
839 // Find maximum distance from each spanning tree vertex
840 vector<int> max_dists(num_images, 0);
841 vector<int> cur_dists;
842 for (size_t i = 0; i < span_tree_leafs.size(); ++i)
843 {
844 cur_dists.assign(num_images, 0);
845 span_tree.walkBreadthFirst(span_tree_leafs[i], IncDistance(cur_dists));
846 for (int j = 0; j < num_images; ++j)
847 max_dists[j] = max(max_dists[j], cur_dists[j]);
848 }
849
850 // Find min-max distance
851 int min_max_dist = max_dists[0];
852 for (int i = 1; i < num_images; ++i)
853 if (min_max_dist > max_dists[i])
854 min_max_dist = max_dists[i];
855
856 // Find spanning tree centers
857 centers.clear();
858 for (int i = 0; i < num_images; ++i)
859 if (max_dists[i] == min_max_dist)
860 centers.push_back(i);
861 CV_Assert(centers.size() > 0 && centers.size() <= 2);
862}
