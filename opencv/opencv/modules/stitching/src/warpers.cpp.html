50void ProjectorBase::setCameraParams(const Mat &K, const Mat &R, const Mat &T)
51{
52 CV_Assert(K.size() == Size(3, 3) && K.type() == CV_32F);
53 CV_Assert(R.size() == Size(3, 3) && R.type() == CV_32F);
54 CV_Assert((T.size() == Size(1, 3) || T.size() == Size(3, 1)) && T.type() == CV_32F);
55
56 Mat_<float> K_(K);
57 k[0] = K_(0,0); k[1] = K_(0,1); k[2] = K_(0,2);
58 k[3] = K_(1,0); k[4] = K_(1,1); k[5] = K_(1,2);
59 k[6] = K_(2,0); k[7] = K_(2,1); k[8] = K_(2,2);
60
61 Mat_<float> Rinv = R.t();
62 rinv[0] = Rinv(0,0); rinv[1] = Rinv(0,1); rinv[2] = Rinv(0,2);
63 rinv[3] = Rinv(1,0); rinv[4] = Rinv(1,1); rinv[5] = Rinv(1,2);
64 rinv[6] = Rinv(2,0); rinv[7] = Rinv(2,1); rinv[8] = Rinv(2,2);
65
66 Mat_<float> R_Kinv = R * K.inv();
67 r_kinv[0] = R_Kinv(0,0); r_kinv[1] = R_Kinv(0,1); r_kinv[2] = R_Kinv(0,2);
68 r_kinv[3] = R_Kinv(1,0); r_kinv[4] = R_Kinv(1,1); r_kinv[5] = R_Kinv(1,2);
69 r_kinv[6] = R_Kinv(2,0); r_kinv[7] = R_Kinv(2,1); r_kinv[8] = R_Kinv(2,2);
70
71 Mat_<float> K_Rinv = K * Rinv;
72 k_rinv[0] = K_Rinv(0,0); k_rinv[1] = K_Rinv(0,1); k_rinv[2] = K_Rinv(0,2);
73 k_rinv[3] = K_Rinv(1,0); k_rinv[4] = K_Rinv(1,1); k_rinv[5] = K_Rinv(1,2);
74 k_rinv[6] = K_Rinv(2,0); k_rinv[7] = K_Rinv(2,1); k_rinv[8] = K_Rinv(2,2);
75
76 Mat_<float> T_(T.reshape(0, 3));
77 t[0] = T_(0,0); t[1] = T_(1,0); t[2] = T_(2,0);
78}

81Point2f PlaneWarper::warpPoint(const Point2f &pt, const Mat &K, const Mat &R, const Mat &T)
82{
83 projector_.setCameraParams(K, R, T);
84 Point2f uv;
85 projector_.mapForward(pt.x, pt.y, uv.x, uv.y);
86 return uv;
87}

90Rect PlaneWarper::buildMaps(Size src_size, const Mat &K, const Mat &R, const Mat &T, Mat &xmap, Mat &ymap)
91{
92 projector_.setCameraParams(K, R, T);
93
94 Point dst_tl, dst_br;
95 detectResultRoi(src_size, dst_tl, dst_br);
96
97 xmap.create(dst_br.y - dst_tl.y + 1, dst_br.x - dst_tl.x + 1, CV_32F);
98 ymap.create(dst_br.y - dst_tl.y + 1, dst_br.x - dst_tl.x + 1, CV_32F);
99
100 float x, y;
101 for (int v = dst_tl.y; v <= dst_br.y; ++v)
102 {
103 for (int u = dst_tl.x; u <= dst_br.x; ++u)
104 {
105 projector_.mapBackward(static_cast<float>(u), static_cast<float>(v), x, y);
106 xmap.at<float>(v - dst_tl.y, u - dst_tl.x) = x;
107 ymap.at<float>(v - dst_tl.y, u - dst_tl.x) = y;
108 }
109 }
110
111 return Rect(dst_tl, dst_br);
112}

115Point PlaneWarper::warp(const Mat &src, const Mat &K, const Mat &R, const Mat &T, int interp_mode, int border_mode,
116 Mat &dst)
117{
118 Mat xmap, ymap;
119 Rect dst_roi = buildMaps(src.size(), K, R, T, xmap, ymap);
120
121 dst.create(dst_roi.height + 1, dst_roi.width + 1, src.type());
122 remap(src, dst, xmap, ymap, interp_mode, border_mode);
123
124 return dst_roi.tl();
125}

139void PlaneWarper::detectResultRoi(Size src_size, Point &dst_tl, Point &dst_br)
140{
141 float tl_uf = numeric_limits<float>::max();
142 float tl_vf = numeric_limits<float>::max();
143 float br_uf = -numeric_limits<float>::max();
144 float br_vf = -numeric_limits<float>::max();
145
146 float u, v;
147
148 projector_.mapForward(0, 0, u, v);
149 tl_uf = min(tl_uf, u); tl_vf = min(tl_vf, v);
150 br_uf = max(br_uf, u); br_vf = max(br_vf, v);
151
152 projector_.mapForward(0, static_cast<float>(src_size.height - 1), u, v);
153 tl_uf = min(tl_uf, u); tl_vf = min(tl_vf, v);
154 br_uf = max(br_uf, u); br_vf = max(br_vf, v);
155
156 projector_.mapForward(static_cast<float>(src_size.width - 1), 0, u, v);
157 tl_uf = min(tl_uf, u); tl_vf = min(tl_vf, v);
158 br_uf = max(br_uf, u); br_vf = max(br_vf, v);
159
160 projector_.mapForward(static_cast<float>(src_size.width - 1), static_cast<float>(src_size.height - 1), u, v);
161 tl_uf = min(tl_uf, u); tl_vf = min(tl_vf, v);
162 br_uf = max(br_uf, u); br_vf = max(br_vf, v);
163
164 dst_tl.x = static_cast<int>(tl_uf);
165 dst_tl.y = static_cast<int>(tl_vf);
166 dst_br.x = static_cast<int>(br_uf);
167 dst_br.y = static_cast<int>(br_vf);
168}

171void SphericalWarper::detectResultRoi(Size src_size, Point &dst_tl, Point &dst_br)
172{
173 detectResultRoiByBorder(src_size, dst_tl, dst_br);
174
175 float tl_uf = static_cast<float>(dst_tl.x);
176 float tl_vf = static_cast<float>(dst_tl.y);
177 float br_uf = static_cast<float>(dst_br.x);
178 float br_vf = static_cast<float>(dst_br.y);
179
180 float x = projector_.rinv[1];
181 float y = projector_.rinv[4];
182 float z = projector_.rinv[7];
183 if (y > 0.f)
184 {
185 float x_ = (projector_.k[0] * x + projector_.k[1] * y) / z + projector_.k[2];
186 float y_ = projector_.k[4] * y / z + projector_.k[5];
187 if (x_ > 0.f && x_ < src_size.width && y_ > 0.f && y_ < src_size.height)
188 {
189 tl_uf = min(tl_uf, 0.f); tl_vf = min(tl_vf, static_cast<float>(CV_PI * projector_.scale));
190 br_uf = max(br_uf, 0.f); br_vf = max(br_vf, static_cast<float>(CV_PI * projector_.scale));
191 }
192 }
193
194 x = projector_.rinv[1];
195 y = -projector_.rinv[4];
196 z = projector_.rinv[7];
197 if (y > 0.f)
198 {
199 float x_ = (projector_.k[0] * x + projector_.k[1] * y) / z + projector_.k[2];
200 float y_ = projector_.k[4] * y / z + projector_.k[5];
201 if (x_ > 0.f && x_ < src_size.width && y_ > 0.f && y_ < src_size.height)
202 {
203 tl_uf = min(tl_uf, 0.f); tl_vf = min(tl_vf, static_cast<float>(0));
204 br_uf = max(br_uf, 0.f); br_vf = max(br_vf, static_cast<float>(0));
205 }
206 }
207
208 dst_tl.x = static_cast<int>(tl_uf);
209 dst_tl.y = static_cast<int>(tl_vf);
210 dst_br.x = static_cast<int>(br_uf);
211 dst_br.y = static_cast<int>(br_vf);
212}
