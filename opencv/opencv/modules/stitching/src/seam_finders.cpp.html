51void PairwiseSeamFinder::find(const vector<Mat> &src, const vector<Point> &corners,
52 vector<Mat> &masks)
53{
54 LOGLN("Finding seams...");
55 if (src.size() == 0)
56 return;
57
58#if ENABLE_LOG
59 int64 t = getTickCount();
60#endif
61
62 images_ = src;
63 sizes_.resize(src.size());
64 for (size_t i = 0; i < src.size(); ++i)
65 sizes_[i] = src[i].size();
66 corners_ = corners;
67 masks_ = masks;
68 run();
69
70 LOGLN("Finding seams, time: " << ((getTickCount() - t) / getTickFrequency()) << " sec");
71}

74void PairwiseSeamFinder::run()
75{
76 for (size_t i = 0; i < sizes_.size() - 1; ++i)
77 {
78 for (size_t j = i + 1; j < sizes_.size(); ++j)
79 {
80 Rect roi;
81 if (overlapRoi(corners_[i], corners_[j], sizes_[i], sizes_[j], roi))
82 findInPair(i, j, roi);
83 }
84 }

108void VoronoiSeamFinder::findInPair(size_t first, size_t second, Rect roi)
109{
110 const int gap = 10;
111 Mat submask1(roi.height + 2 * gap, roi.width + 2 * gap, CV_8U);
112 Mat submask2(roi.height + 2 * gap, roi.width + 2 * gap, CV_8U);
113
114 Size img1 = sizes_[first], img2 = sizes_[second];
115 Mat mask1 = masks_[first], mask2 = masks_[second];
116 Point tl1 = corners_[first], tl2 = corners_[second];
117
118 // Cut submasks with some gap
119 for (int y = -gap; y < roi.height + gap; ++y)
120 {
121 for (int x = -gap; x < roi.width + gap; ++x)
122 {
123 int y1 = roi.y - tl1.y + y;
124 int x1 = roi.x - tl1.x + x;
125 if (y1 >= 0 && x1 >= 0 && y1 < img1.height && x1 < img1.width)
126 submask1.at<uchar>(y + gap, x + gap) = mask1.at<uchar>(y1, x1);
127 else
128 submask1.at<uchar>(y + gap, x + gap) = 0;
129
130 int y2 = roi.y - tl2.y + y;
131 int x2 = roi.x - tl2.x + x;
132 if (y2 >= 0 && x2 >= 0 && y2 < img2.height && x2 < img2.width)
133 submask2.at<uchar>(y + gap, x + gap) = mask2.at<uchar>(y2, x2);
134 else
135 submask2.at<uchar>(y + gap, x + gap) = 0;
136 }
137 }
138
139 Mat collision = (submask1 != 0) & (submask2 != 0);
140 Mat unique1 = submask1.clone(); unique1.setTo(0, collision);
141 Mat unique2 = submask2.clone(); unique2.setTo(0, collision);
142
143 Mat dist1, dist2;
144 distanceTransform(unique1 == 0, dist1, CV_DIST_L1, 3);
145 distanceTransform(unique2 == 0, dist2, CV_DIST_L1, 3);
146
147 Mat seam = dist1 < dist2;
148
149 for (int y = 0; y < roi.height; ++y)
150 {
151 for (int x = 0; x < roi.width; ++x)
152 {
153 if (seam.at<uchar>(y + gap, x + gap))
154 mask2.at<uchar>(roi.y - tl2.y + y, roi.x - tl2.x + x) = 0;
155 else
156 mask1.at<uchar>(roi.y - tl1.y + y, roi.x - tl1.x + x) = 0;
157 }
158 }
159}

165void DpSeamFinder::find(const vector<Mat> &src, const vector<Point> &corners, vector<Mat> &masks)
166{
167 LOGLN("Finding seams...");
168#if ENABLE_LOG
169 int64 t = getTickCount();
170#endif
171
172 if (src.size() == 0)
173 return;
174
175 vector<pair<size_t, size_t> > pairs;
176
177 for (size_t i = 0; i+1 < src.size(); ++i)
178 for (size_t j = i+1; j < src.size(); ++j)
179 pairs.push_back(make_pair(i, j));
180
181 sort(pairs.begin(), pairs.end(), ImagePairLess(src, corners));
182 reverse(pairs.begin(), pairs.end());
183
184 for (size_t i = 0; i < pairs.size(); ++i)
185 {
186 size_t i0 = pairs[i].first, i1 = pairs[i].second;
187 process(src[i0], src[i1], corners[i0], corners[i1], masks[i0], masks[i1]);
188 }
189
190 LOGLN("Finding seams, time: " << ((getTickCount() - t) / getTickFrequency()) << " sec");
191}

194void DpSeamFinder::process(
195 const Mat &image1, const Mat &image2, Point tl1, Point tl2,
196 Mat &mask1, Mat &mask2)
197{
198 CV_Assert(image1.size() == mask1.size());
199 CV_Assert(image2.size() == mask2.size());
200
201 Point intersectTl(std::max(tl1.x, tl2.x), std::max(tl1.y, tl2.y));
202
203 Point intersectBr(std::min(tl1.x + image1.cols, tl2.x + image2.cols),
204 std::min(tl1.y + image1.rows, tl2.y + image2.rows));
205
206 if (intersectTl.x >= intersectBr.x || intersectTl.y >= intersectBr.y)
207 return; // there are no conflicts
208
209 unionTl_ = Point(std::min(tl1.x, tl2.x), std::min(tl1.y, tl2.y));
210
211 unionBr_ = Point(std::max(tl1.x + image1.cols, tl2.x + image2.cols),
212 std::max(tl1.y + image1.rows, tl2.y + image2.rows));
213
214 unionSize_ = Size(unionBr_.x - unionTl_.x, unionBr_.y - unionTl_.y);
215
216 mask1_ = Mat::zeros(unionSize_, CV_8U);
217 mask2_ = Mat::zeros(unionSize_, CV_8U);
218
219 Mat tmp = mask1_(Rect(tl1.x - unionTl_.x, tl1.y - unionTl_.y, mask1.cols, mask1.rows));
220 mask1.copyTo(tmp);
221
222 tmp = mask2_(Rect(tl2.x - unionTl_.x, tl2.y - unionTl_.y, mask2.cols, mask2.rows));
223 mask2.copyTo(tmp);
224
225 // find both images contour masks
226
227 contour1mask_ = Mat::zeros(unionSize_, CV_8U);
228 contour2mask_ = Mat::zeros(unionSize_, CV_8U);
229
230 for (int y = 0; y < unionSize_.height; ++y)
231 {
232 for (int x = 0; x < unionSize_.width; ++x)
233 {
234 if (mask1_(y, x) &&
235 ((x == 0 || !mask1_(y, x-1)) || (x == unionSize_.width-1 || !mask1_(y, x+1)) ||
236 (y == 0 || !mask1_(y-1, x)) || (y == unionSize_.height-1 || !mask1_(y+1, x))))
237 {
238 contour1mask_(y, x) = 255;
239 }
240
241 if (mask2_(y, x) &&
242 ((x == 0 || !mask2_(y, x-1)) || (x == unionSize_.width-1 || !mask2_(y, x+1)) ||
243 (y == 0 || !mask2_(y-1, x)) || (y == unionSize_.height-1 || !mask2_(y+1, x))))
244 {
245 contour2mask_(y, x) = 255;
246 }
247 }
248 }
249
250 findComponents();
251
252 findEdges();
253
254 resolveConflicts(image1, image2, tl1, tl2, mask1, mask2);
255}

258void DpSeamFinder::findComponents()
259{
260 // label all connected components and get information about them
261
262 ncomps_ = 0;
263 labels_.create(unionSize_);
264 states_.clear();
265 tls_.clear();
266 brs_.clear();
267 contours_.clear();
268
269 for (int y = 0; y < unionSize_.height; ++y)
270 {
271 for (int x = 0; x < unionSize_.width; ++x)
272 {
273 if (mask1_(y, x) && mask2_(y, x))
274 labels_(y, x) = numeric_limits<int>::max();
275 else if (mask1_(y, x))
276 labels_(y, x) = numeric_limits<int>::max()-1;
277 else if (mask2_(y, x))
278 labels_(y, x) = numeric_limits<int>::max()-2;
279 else
280 labels_(y, x) = 0;
281 }
282 }
283
284 for (int y = 0; y < unionSize_.height; ++y)
285 {
286 for (int x = 0; x < unionSize_.width; ++x)
287 {
288 if (labels_(y, x) >= numeric_limits<int>::max()-2)
289 {
290 if (labels_(y, x) == numeric_limits<int>::max())
291 states_.push_back(INTERS);
292 else if (labels_(y, x) == numeric_limits<int>::max()-1)
293 states_.push_back(FIRST);
294 else if (labels_(y, x) == numeric_limits<int>::max()-2)
295 states_.push_back(SECOND);
296
297 floodFill(labels_, Point(x, y), ++ncomps_);
298 tls_.push_back(Point(x, y));
299 brs_.push_back(Point(x+1, y+1));
300 contours_.push_back(vector<Point>());
301 }
302
303 if (labels_(y, x))
304 {
305 int l = labels_(y, x);
306 int ci = l-1;
307
308 tls_[ci].x = std::min(tls_[ci].x, x);
309 tls_[ci].y = std::min(tls_[ci].y, y);
310 brs_[ci].x = std::max(brs_[ci].x, x+1);
311 brs_[ci].y = std::max(brs_[ci].y, y+1);
312
313 if ((x == 0 || labels_(y, x-1) != l) || (x == unionSize_.width-1 || labels_(y, x+1) != l) ||
314 (y == 0 || labels_(y-1, x) != l) || (y == unionSize_.height-1 || labels_(y+1, x) != l))
315 {
316 contours_[ci].push_back(Point(x, y));
317 }
318 }
319 }
320 }
321}

324void DpSeamFinder::findEdges()
325{
326 // find edges between components
327
328 map<pair<int, int>, int> wedges; // weighted edges
329
330 for (int ci = 0; ci < ncomps_-1; ++ci)
331 {
332 for (int cj = ci+1; cj < ncomps_; ++cj)
333 {
334 wedges[make_pair(ci, cj)] = 0;
335 wedges[make_pair(cj, ci)] = 0;
336 }
337 }
338
339 for (int ci = 0; ci < ncomps_; ++ci)
340 {
341 for (size_t i = 0; i < contours_[ci].size(); ++i)
342 {
343 int x = contours_[ci][i].x;
344 int y = contours_[ci][i].y;
345 int l = ci + 1;
346
347 if (x > 0 && labels_(y, x-1) && labels_(y, x-1) != l)
348 {
349 wedges[make_pair(ci, labels_(y, x-1)-1)]++;
350 wedges[make_pair(labels_(y, x-1)-1, ci)]++;
351 }
352
353 if (y > 0 && labels_(y-1, x) && labels_(y-1, x) != l)
354 {
355 wedges[make_pair(ci, labels_(y-1, x)-1)]++;
356 wedges[make_pair(labels_(y-1, x)-1, ci)]++;
357 }
358
359 if (x < unionSize_.width-1 && labels_(y, x+1) && labels_(y, x+1) != l)
360 {
361 wedges[make_pair(ci, labels_(y, x+1)-1)]++;
362 wedges[make_pair(labels_(y, x+1)-1, ci)]++;
363 }
364
365 if (y < unionSize_.height-1 && labels_(y+1, x) && labels_(y+1, x) != l)
366 {
367 wedges[make_pair(ci, labels_(y+1, x)-1)]++;
368 wedges[make_pair(labels_(y+1, x)-1, ci)]++;
369 }
370 }
371 }
372
373 edges_.clear();
374
375 for (int ci = 0; ci < ncomps_-1; ++ci)
376 {
377 for (int cj = ci+1; cj < ncomps_; ++cj)
378 {
379 map<pair<int, int>, int>::iterator itr = wedges.find(make_pair(ci, cj));
380 if (itr != wedges.end() && itr->second > 0)
381 edges_.insert(itr->first);
382
383 itr = wedges.find(make_pair(cj, ci));
384 if (itr != wedges.end() && itr->second > 0)
385 edges_.insert(itr->first);
386 }
387 }
388}

391void DpSeamFinder::resolveConflicts(
392 const Mat &image1, const Mat &image2, Point tl1, Point tl2, Mat &mask1, Mat &mask2)
393{
394 if (costFunc_ == COLOR_GRAD)
395 computeGradients(image1, image2);
396
397 // resolve conflicts between components
398
399 bool hasConflict = true;
400 while (hasConflict)
401 {
402 int c1 = 0, c2 = 0;
403 hasConflict = false;
404
405 for (set<pair<int, int> >::iterator itr = edges_.begin(); itr != edges_.end(); ++itr)
406 {
407 c1 = itr->first;
408 c2 = itr->second;
409
410 if ((states_[c1] & INTERS) && (states_[c1] & (~INTERS)) != states_[c2])
411 {
412 hasConflict = true;
413 break;
414 }
415 }
416
417 if (hasConflict)
418 {
419 int l1 = c1+1, l2 = c2+1;
420
421 if (hasOnlyOneNeighbor(c1))
422 {
423 // if the first components has only one adjacent component
424
425 for (int y = tls_[c1].y; y < brs_[c1].y; ++y)
426 for (int x = tls_[c1].x; x < brs_[c1].x; ++x)
427 if (labels_(y, x) == l1)
428 labels_(y, x) = l2;
429
430 states_[c1] = states_[c2] == FIRST ? SECOND : FIRST;
431 }
432 else
433 {
434 // if the first component has more than one adjacent component
435
436 Point p1, p2;
437 if (getSeamTips(c1, c2, p1, p2))
438 {
439 vector<Point> seam;
440 bool isHorizontalSeam;
441
442 if (estimateSeam(image1, image2, tl1, tl2, c1, p1, p2, seam, isHorizontalSeam))
443 updateLabelsUsingSeam(c1, c2, seam, isHorizontalSeam);
444 }
445
446 states_[c1] = states_[c2] == FIRST ? INTERS_SECOND : INTERS_FIRST;
447 }
448
449 const int c[] = {c1, c2};
450 const int l[] = {l1, l2};
451
452 for (int i = 0; i < 2; ++i)
453 {
454 // update information about the (i+1)-th component
455
456 int x0 = tls_[c[i]].x, x1 = brs_[c[i]].x;
457 int y0 = tls_[c[i]].y, y1 = brs_[c[i]].y;
458
459 tls_[c[i]] = Point(numeric_limits<int>::max(), numeric_limits<int>::max());
460 brs_[c[i]] = Point(numeric_limits<int>::min(), numeric_limits<int>::min());
461 contours_[c[i]].clear();
462
463 for (int y = y0; y < y1; ++y)
464 {
465 for (int x = x0; x < x1; ++x)
466 {
467 if (labels_(y, x) == l[i])
468 {
469 tls_[c[i]].x = std::min(tls_[c[i]].x, x);
470 tls_[c[i]].y = std::min(tls_[c[i]].y, y);
471 brs_[c[i]].x = std::max(brs_[c[i]].x, x+1);
472 brs_[c[i]].y = std::max(brs_[c[i]].y, y+1);
473
474 if ((x == 0 || labels_(y, x-1) != l[i]) || (x == unionSize_.width-1 || labels_(y, x+1) != l[i]) ||
475 (y == 0 || labels_(y-1, x) != l[i]) || (y == unionSize_.height-1 || labels_(y+1, x) != l[i]))
476 {
477 contours_[c[i]].push_back(Point(x, y));
478 }
479 }
480 }
481 }
482 }
483
484 // remove edges
485
486 edges_.erase(make_pair(c1, c2));
487 edges_.erase(make_pair(c2, c1));
488 }
489 }
490
491 // update masks
492
493 int dx1 = unionTl_.x - tl1.x, dy1 = unionTl_.y - tl1.y;
494 int dx2 = unionTl_.x - tl2.x, dy2 = unionTl_.y - tl2.y;
495
496 for (int y = 0; y < mask2.rows; ++y)
497 {
498 for (int x = 0; x < mask2.cols; ++x)
499 {
500 int l = labels_(y - dy2, x - dx2);
501 if (l > 0 && (states_[l-1] & FIRST) && mask1.at<uchar>(y - dy2 + dy1, x - dx2 + dx1))
502 mask2.at<uchar>(y, x) = 0;
503 }
504 }
505
506 for (int y = 0; y < mask1.rows; ++y)
507 {
508 for (int x = 0; x < mask1.cols; ++x)
509 {
510 int l = labels_(y - dy1, x - dx1);
511 if (l > 0 && (states_[l-1] & SECOND) && mask2.at<uchar>(y - dy1 + dy2, x - dx1 + dx2))
512 mask1.at<uchar>(y, x) = 0;
513 }
514 }
515}

518void DpSeamFinder::computeGradients(const Mat &image1, const Mat &image2)
519{
520 CV_Assert(image1.channels() == 3 || image1.channels() == 4);
521 CV_Assert(image2.channels() == 3 || image2.channels() == 4);
522 CV_Assert(costFunction() == COLOR_GRAD);
523
524 Mat gray;
525
526 if (image1.channels() == 3)
527 cvtColor(image1, gray, CV_BGR2GRAY);
528 else if (image1.channels() == 4)
529 cvtColor(image1, gray, CV_BGRA2GRAY);
530
531 Sobel(gray, gradx1_, CV_32F, 1, 0);
532 Sobel(gray, grady1_, CV_32F, 0, 1);
533
534 if (image2.channels() == 3)
535 cvtColor(image2, gray, CV_BGR2GRAY);
536 else if (image2.channels() == 4)
537 cvtColor(image2, gray, CV_BGRA2GRAY);
538
539 Sobel(gray, gradx2_, CV_32F, 1, 0);
540 Sobel(gray, grady2_, CV_32F, 0, 1);
541}

544bool DpSeamFinder::hasOnlyOneNeighbor(int comp)
545{
546 set<pair<int, int> >::iterator begin, end;
547 begin = lower_bound(edges_.begin(), edges_.end(), make_pair(comp, numeric_limits<int>::min()));
548 end = upper_bound(edges_.begin(), edges_.end(), make_pair(comp, numeric_limits<int>::max()));
549 return ++begin == end;
550}

553bool DpSeamFinder::closeToContour(int y, int x, const Mat_<uchar> &contourMask)
554{
555 const int rad = 2;
556
557 for (int dy = -rad; dy <= rad; ++dy)
558 {
559 if (y + dy >= 0 && y + dy < unionSize_.height)
560 {
561 for (int dx = -rad; dx <= rad; ++dx)
562 {
563 if (x + dx >= 0 && x + dx < unionSize_.width &&
564 contourMask(y + dy, x + dx))
565 {
566 return true;
567 }
568 }
569 }
570 }
571
572 return false;
573}

576bool DpSeamFinder::getSeamTips(int comp1, int comp2, Point &p1, Point &p2)
577{
578 CV_Assert(states_[comp1] & INTERS);
579
580 // find special points
581
582 vector<Point> specialPoints;
583 int l2 = comp2+1;
584
585 for (size_t i = 0; i < contours_[comp1].size(); ++i)
586 {
587 int x = contours_[comp1][i].x;
588 int y = contours_[comp1][i].y;
589
590 if (closeToContour(y, x, contour1mask_) &&
591 closeToContour(y, x, contour2mask_) &&
592 ((x > 0 && labels_(y, x-1) == l2) ||
593 (y > 0 && labels_(y-1, x) == l2) ||
594 (x < unionSize_.width-1 && labels_(y, x+1) == l2) ||
595 (y < unionSize_.height-1 && labels_(y+1, x) == l2)))
596 {
597 specialPoints.push_back(Point(x, y));
598 }
599 }
600
601 if (specialPoints.size() < 2)
602 return false;
603
604 // find clusters
605
606 vector<int> labels;
607 cv::partition(specialPoints, labels, ClosePoints(10));
608
609 int nlabels = *max_element(labels.begin(), labels.end()) + 1;
610 if (nlabels < 2)
611 return false;
612
613 vector<Point> sum(nlabels);
614 vector<vector<Point> > points(nlabels);
615
616 for (size_t i = 0; i < specialPoints.size(); ++i)
617 {
618 sum[labels[i]] += specialPoints[i];
619 points[labels[i]].push_back(specialPoints[i]);
620 }
621
622 // select two most distant clusters
623
624 int idx[2] = {-1,-1};
625 double maxDist = -numeric_limits<double>::max();
626
627 for (int i = 0; i < nlabels-1; ++i)
628 {
629 for (int j = i+1; j < nlabels; ++j)
630 {
631 double size1 = static_cast<double>(points[i].size()), size2 = static_cast<double>(points[j].size());
632 double cx1 = cvRound(sum[i].x / size1), cy1 = cvRound(sum[i].y / size1);
633 double cx2 = cvRound(sum[j].x / size2), cy2 = cvRound(sum[j].y / size2);
634
635 double dist = (cx1 - cx2) * (cx1 - cx2) + (cy1 - cy2) * (cy1 - cy2);
636 if (dist > maxDist)
637 {
638 maxDist = dist;
639 idx[0] = i;
640 idx[1] = j;
641 }
642 }
643 }
644
645 // select two points closest to the clusters' centers
646
647 Point p[2];
648
649 for (int i = 0; i < 2; ++i)
650 {
651 double size = static_cast<double>(points[idx[i]].size());
652 double cx = cvRound(sum[idx[i]].x / size);
653 double cy = cvRound(sum[idx[i]].y / size);
654
655 size_t closest = points[idx[i]].size();
656 double minDist = numeric_limits<double>::max();
657
658 for (size_t j = 0; j < points[idx[i]].size(); ++j)
659 {
660 double dist = (points[idx[i]][j].x - cx) * (points[idx[i]][j].x - cx) +
661 (points[idx[i]][j].y - cy) * (points[idx[i]][j].y - cy);
662 if (dist < minDist)
663 {
664 minDist = dist;
665 closest = j;
666 }
667 }
668
669 p[i] = points[idx[i]][closest];
670 }
671
672 p1 = p[0];
673 p2 = p[1];
674 return true;
675}

688}

697 sqr(r1[4*x1+2] - r2[4*x2+2]));

703void DpSeamFinder::computeCosts(
704 const Mat &image1, const Mat &image2, Point tl1, Point tl2,
705 int comp, Mat_<float> &costV, Mat_<float> &costH)
706{
707 CV_Assert(states_[comp] & INTERS);
708
709 // compute costs
710
711 float (*diff)(const Mat&, int, int, const Mat&, int, int) = 0;
712 if (image1.type() == CV_32FC3 && image2.type() == CV_32FC3)
713 diff = diffL2Square3<float>;
714 else if (image1.type() == CV_8UC3 && image2.type() == CV_8UC3)
715 diff = diffL2Square3<uchar>;
716 else if (image1.type() == CV_32FC4 && image2.type() == CV_32FC4)
717 diff = diffL2Square4<float>;
718 else if (image1.type() == CV_8UC4 && image2.type() == CV_8UC4)
719 diff = diffL2Square4<uchar>;
720 else
721 CV_Error(CV_StsBadArg, "both images must have CV_32FC3(4) or CV_8UC3(4) type");
722
723 int l = comp+1;
724 Rect roi(tls_[comp], brs_[comp]);
725
726 int dx1 = unionTl_.x - tl1.x, dy1 = unionTl_.y - tl1.y;
727 int dx2 = unionTl_.x - tl2.x, dy2 = unionTl_.y - tl2.y;
728
729 const float badRegionCost = normL2(Point3f(255.f, 255.f, 255.f),
730 Point3f(0.f, 0.f, 0.f));
731
732 costV.create(roi.height, roi.width+1);
733
734 for (int y = roi.y; y < roi.br().y; ++y)
735 {
736 for (int x = roi.x; x < roi.br().x+1; ++x)
737 {
738 if (labels_(y, x) == l && x > 0 && labels_(y, x-1) == l)
739 {
740 float costColor = (diff(image1, y + dy1, x + dx1 - 1, image2, y + dy2, x + dx2) +
741 diff(image1, y + dy1, x + dx1, image2, y + dy2, x + dx2 - 1)) / 2;
742 if (costFunc_ == COLOR)
743 costV(y - roi.y, x - roi.x) = costColor;
744 else if (costFunc_ == COLOR_GRAD)
745 {
746 float costGrad = std::abs(gradx1_(y + dy1, x + dx1)) + std::abs(gradx1_(y + dy1, x + dx1 - 1)) +
747 std::abs(gradx2_(y + dy2, x + dx2)) + std::abs(gradx2_(y + dy2, x + dx2 - 1)) + 1.f;
748 costV(y - roi.y, x - roi.x) = costColor / costGrad;
749 }
750 }
751 else
752 costV(y - roi.y, x - roi.x) = badRegionCost;
753 }
754 }
755
756 costH.create(roi.height+1, roi.width);
757
758 for (int y = roi.y; y < roi.br().y+1; ++y)
759 {
760 for (int x = roi.x; x < roi.br().x; ++x)
761 {
762 if (labels_(y, x) == l && y > 0 && labels_(y-1, x) == l)
763 {
764 float costColor = (diff(image1, y + dy1 - 1, x + dx1, image2, y + dy2, x + dx2) +
765 diff(image1, y + dy1, x + dx1, image2, y + dy2 - 1, x + dx2)) / 2;
766 if (costFunc_ == COLOR)
767 costH(y - roi.y, x - roi.x) = costColor;
768 else if (costFunc_ == COLOR_GRAD)
769 {
770 float costGrad = std::abs(grady1_(y + dy1, x + dx1)) + std::abs(grady1_(y + dy1 - 1, x + dx1)) +
771 std::abs(grady2_(y + dy2, x + dx2)) + std::abs(grady2_(y + dy2 - 1, x + dx2)) + 1.f;
772 costH(y - roi.y, x - roi.x) = costColor / costGrad;
773 }
774 }
775 else
776 costH(y - roi.y, x - roi.x) = badRegionCost;
777 }
778 }
779}

782bool DpSeamFinder::estimateSeam(
783 const Mat &image1, const Mat &image2, Point tl1, Point tl2, int comp,
784 Point p1, Point p2, vector<Point> &seam, bool &isHorizontal)
785{
786 CV_Assert(states_[comp] & INTERS);
787
788 Mat_<float> costV, costH;
789 computeCosts(image1, image2, tl1, tl2, comp, costV, costH);
790
791 Rect roi(tls_[comp], brs_[comp]);
792 Point src = p1 - roi.tl();
793 Point dst = p2 - roi.tl();
794 int l = comp+1;
795
796 // estimate seam direction
797
798 bool swapped = false;
799 isHorizontal = std::abs(dst.x - src.x) > std::abs(dst.y - src.y);
800
801 if (isHorizontal)
802 {
803 if (src.x > dst.x)
804 {
805 std::swap(src, dst);
806 swapped = true;
807 }
808 }
809 else if (src.y > dst.y)
810 {
811 swapped = true;
812 std::swap(src, dst);
813 }
814
815 // find optimal control
816
817 Mat_<uchar> control = Mat::zeros(roi.size(), CV_8U);
818 Mat_<uchar> reachable = Mat::zeros(roi.size(), CV_8U);
819 Mat_<float> cost = Mat::zeros(roi.size(), CV_32F);
820
821 reachable(src) = 1;
822 cost(src) = 0.f;
823
824 int nsteps;
825 pair<float, int> steps[3];
826
827 if (isHorizontal)
828 {
829 for (int x = src.x+1; x <= dst.x; ++x)
830 {
831 for (int y = 0; y < roi.height; ++y)
832 {
833 // seam follows along upper side of pixels
834
835 nsteps = 0;
836
837 if (labels_(y + roi.y, x + roi.x) == l)
838 {
839 if (reachable(y, x-1))
840 steps[nsteps++] = make_pair(cost(y, x-1) + costH(y, x-1), 1);
841 if (y > 0 && reachable(y-1, x-1))
842 steps[nsteps++] = make_pair(cost(y-1, x-1) + costH(y-1, x-1) + costV(y-1, x), 2);
843 if (y < roi.height-1 && reachable(y+1, x-1))
844 steps[nsteps++] = make_pair(cost(y+1, x-1) + costH(y+1, x-1) + costV(y, x), 3);
845 }
846
847 if (nsteps)
848 {
849 pair<float, int> opt = *min_element(steps, steps + nsteps);
850 cost(y, x) = opt.first;
851 control(y, x) = (uchar)opt.second;
852 reachable(y, x) = 255;
853 }
854 }
855 }
856 }
857 else
858 {
859 for (int y = src.y+1; y <= dst.y; ++y)
860 {
861 for (int x = 0; x < roi.width; ++x)
862 {
863 // seam follows along left side of pixels
864
865 nsteps = 0;
866
867 if (labels_(y + roi.y, x + roi.x) == l)
868 {
869 if (reachable(y-1, x))
870 steps[nsteps++] = make_pair(cost(y-1, x) + costV(y-1, x), 1);
871 if (x > 0 && reachable(y-1, x-1))
872 steps[nsteps++] = make_pair(cost(y-1, x-1) + costV(y-1, x-1) + costH(y, x-1), 2);
873 if (x < roi.width-1 && reachable(y-1, x+1))
874 steps[nsteps++] = make_pair(cost(y-1, x+1) + costV(y-1, x+1) + costH(y, x), 3);
875 }
876
877 if (nsteps)
878 {
879 pair<float, int> opt = *min_element(steps, steps + nsteps);
880 cost(y, x) = opt.first;
881 control(y, x) = (uchar)opt.second;
882 reachable(y, x) = 255;
883 }
884 }
885 }
886 }
887
888 if (!reachable(dst))
889 return false;
890
891 // restore seam
892
893 Point p = dst;
894 seam.clear();
895 seam.push_back(p + roi.tl());
896
897 if (isHorizontal)
898 {
899 for (; p.x != src.x; seam.push_back(p + roi.tl()))
900 {
901 if (control(p) == 2) p.y--;
902 else if (control(p) == 3) p.y++;
903 p.x--;
904 }
905 }
906 else
907 {
908 for (; p.y != src.y; seam.push_back(p + roi.tl()))
909 {
910 if (control(p) == 2) p.x--;
911 else if (control(p) == 3) p.x++;
912 p.y--;
913 }
914 }
915
916 if (!swapped)
917 reverse(seam.begin(), seam.end());
918
919 CV_Assert(seam.front() == p1);
920 CV_Assert(seam.back() == p2);
921 return true;
922}

925void DpSeamFinder::updateLabelsUsingSeam(
926 int comp1, int comp2, const vector<Point> &seam, bool isHorizontalSeam)
927{
928 Mat_<int> mask = Mat::zeros(brs_[comp1].y - tls_[comp1].y,
929 brs_[comp1].x - tls_[comp1].x, CV_32S);
930
931 for (size_t i = 0; i < contours_[comp1].size(); ++i)
932 mask(contours_[comp1][i] - tls_[comp1]) = 255;
933
934 for (size_t i = 0; i < seam.size(); ++i)
935 mask(seam[i] - tls_[comp1]) = 255;
936
937 // find connected components after seam carving
938
939 int l1 = comp1+1, l2 = comp2+1;
940
941 int ncomps = 0;
942
943 for (int y = 0; y < mask.rows; ++y)
944 for (int x = 0; x < mask.cols; ++x)
945 if (!mask(y, x) && labels_(y + tls_[comp1].y, x + tls_[comp1].x) == l1)
946 floodFill(mask, Point(x, y), ++ncomps);
947
948 for (size_t i = 0; i < contours_[comp1].size(); ++i)
949 {
950 int x = contours_[comp1][i].x - tls_[comp1].x;
951 int y = contours_[comp1][i].y - tls_[comp1].y;
952
953 bool ok = false;
954 static const int dx[] = {-1, +1, 0, 0, -1, +1, -1, +1};
955 static const int dy[] = {0, 0, -1, +1, -1, -1, +1, +1};
956
957 for (int j = 0; j < 8; ++j)
958 {
959 int c = x + dx[j];
960 int r = y + dy[j];
961
962 if (c >= 0 && c < mask.cols && r >= 0 && r < mask.rows &&
963 mask(r, c) && mask(r, c) != 255)
964 {
965 ok = true;
966 mask(y, x) = mask(r, c);
967 }
968 }
969
970 if (!ok)
971 mask(y, x) = 0;
972 }
973
974 if (isHorizontalSeam)
975 {
976 for (size_t i = 0; i < seam.size(); ++i)
977 {
978 int x = seam[i].x - tls_[comp1].x;
979 int y = seam[i].y - tls_[comp1].y;
980
981 if (y < mask.rows-1 && mask(y+1, x) && mask(y+1, x) != 255)
982 mask(y, x) = mask(y+1, x);
983 else
984 mask(y, x) = 0;
985 }
986 }
987 else
988 {
989 for (size_t i = 0; i < seam.size(); ++i)
990 {
991 int x = seam[i].x - tls_[comp1].x;
992 int y = seam[i].y - tls_[comp1].y;
993
994 if (x < mask.cols-1 && mask(y, x+1) && mask(y, x+1) != 255)
995 mask(y, x) = mask(y, x+1);
996 else
997 mask(y, x) = 0;
998 }
999 }
1000
1001 // find new components connected with the second component and
1002 // with other components except the ones we are working with
1003
1004 map<int, int> connect2;
1005 map<int, int> connectOther;
1006
1007 for (int i = 1; i <= ncomps; ++i)
1008 {
1009 connect2.insert(make_pair(i, 0));
1010 connectOther.insert(make_pair(i, 0));
1011 }
1012
1013 for (size_t i = 0; i < contours_[comp1].size(); ++i)
1014 {
1015 int x = contours_[comp1][i].x;
1016 int y = contours_[comp1][i].y;
1017
1018 if ((x > 0 && labels_(y, x-1) == l2) ||
1019 (y > 0 && labels_(y-1, x) == l2) ||
1020 (x < unionSize_.width-1 && labels_(y, x+1) == l2) ||
1021 (y < unionSize_.height-1 && labels_(y+1, x) == l2))
1022 {
1023 connect2[mask(y - tls_[comp1].y, x - tls_[comp1].x)]++;
1024 }
1025
1026 if ((x > 0 && labels_(y, x-1) != l1 && labels_(y, x-1) != l2) ||
1027 (y > 0 && labels_(y-1, x) != l1 && labels_(y-1, x) != l2) ||
1028 (x < unionSize_.width-1 && labels_(y, x+1) != l1 && labels_(y, x+1) != l2) ||
1029 (y < unionSize_.height-1 && labels_(y+1, x) != l1 && labels_(y+1, x) != l2))
1030 {
1031 connectOther[mask(y - tls_[comp1].y, x - tls_[comp1].x)]++;
1032 }
1033 }
1034
1035 vector<int> isAdjComp(ncomps + 1, 0);
1036
1037 for (map<int, int>::iterator itr = connect2.begin(); itr != connect2.end(); ++itr)
1038 {
1039 double len = static_cast<double>(contours_[comp1].size());
1040 isAdjComp[itr->first] = itr->second / len > 0.05 && connectOther.find(itr->first)->second / len < 0.1;
1041 }
1042
1043 // update labels
1044
1045 for (int y = 0; y < mask.rows; ++y)
1046 for (int x = 0; x < mask.cols; ++x)
1047 if (mask(y, x) && isAdjComp[mask(y, x)])
1048 labels_(y + tls_[comp1].y, x + tls_[comp1].x) = l2;
1049}
