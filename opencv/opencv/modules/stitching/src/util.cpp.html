50void DisjointSets::createOneElemSets(int n)
51{
52 rank_.assign(n, 0);
53 size.assign(n, 1);
54 parent.resize(n);
55 for (int i = 0; i < n; ++i)
56 parent[i] = i;
57}
58
59
60int DisjointSets::findSetByElem(int elem)
61{
62 int set = elem;
63 while (set != parent[set])
64 set = parent[set];
65 int next;
66 while (elem != parent[elem])
67 {
68 next = parent[elem];
69 parent[elem] = set;
70 elem = next;
71 }
72 return set;
73}

75
76int DisjointSets::mergeSets(int set1, int set2)
77{
78 if (rank_[set1] < rank_[set2])
79 {
80 parent[set1] = set2;
81 size[set2] += size[set1];
82 return set2;
83 }
84 if (rank_[set2] < rank_[set1])
85 {
86 parent[set2] = set1;
87 size[set1] += size[set2];
88 return set1;
89 }
90 parent[set1] = set2;
91 rank_[set2]++;
92 size[set2] += size[set1];
93 return set2;
94}

103bool overlapRoi(Point tl1, Point tl2, Size sz1, Size sz2, Rect &roi)
104{
105 int x_tl = max(tl1.x, tl2.x);
106 int y_tl = max(tl1.y, tl2.y);
107 int x_br = min(tl1.x + sz1.width, tl2.x + sz2.width);
108 int y_br = min(tl1.y + sz1.height, tl2.y + sz2.height);
109 if (x_tl < x_br && y_tl < y_br)
110 {
111 roi = Rect(x_tl, y_tl, x_br - x_tl, y_br - y_tl);
112 return true;
113 }
114 return false;
115}

118Rect resultRoi(const vector<Point> &corners, const vector<Mat> &images)
119{
120 vector<Size> sizes(images.size());
121 for (size_t i = 0; i < images.size(); ++i)
122 sizes[i] = images[i].size();
123 return resultRoi(corners, sizes);
124}

126
127Rect resultRoi(const vector<Point> &corners, const vector<Size> &sizes)
128{
129 CV_Assert(sizes.size() == corners.size());
130 Point tl(numeric_limits<int>::max(), numeric_limits<int>::max());
131 Point br(numeric_limits<int>::min(), numeric_limits<int>::min());
132 for (size_t i = 0; i < corners.size(); ++i)
133 {
134 tl.x = min(tl.x, corners[i].x);
135 tl.y = min(tl.y, corners[i].y);
136 br.x = max(br.x, corners[i].x + sizes[i].width);
137 br.y = max(br.y, corners[i].y + sizes[i].height);
138 }
139 return Rect(tl, br);
140}

143Point resultTl(const vector<Point> &corners)
144{
145 Point tl(numeric_limits<int>::max(), numeric_limits<int>::max());
146 for (size_t i = 0; i < corners.size(); ++i)
147 {
148 tl.x = min(tl.x, corners[i].x);
149 tl.y = min(tl.y, corners[i].y);
150 }
151 return tl;
152}

155void selectRandomSubset(int count, int size, vector<int> &subset)
156{
157 subset.clear();
158 for (int i = 0; i < size; ++i)
159 {
160 if (randu<int>() % (size - i) < count)
161 {
162 subset.push_back(i);
163 count--;
164 }
165 }
166}
