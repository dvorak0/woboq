50template<typename _Tp> static inline bool
51decomposeCholesky(_Tp* A, size_t astep, int m)
52{
53 if (!Cholesky(A, astep, m, 0, 0, 0))
54 return false;
55 astep /= sizeof(A[0]);
56 for (int i = 0; i < m; ++i)
57 A[i*astep + i] = (_Tp)(1./A[i*astep + i]);
58 return true;
59}

67void focalsFromHomography(const Mat& H, double &f0, double &f1, bool &f0_ok, bool &f1_ok)
68{
69 CV_Assert(H.type() == CV_64F && H.size() == Size(3, 3));
70
71 const double* h = reinterpret_cast<const double*>(H.data);
72
73 double d1, d2; // Denominators
74 double v1, v2; // Focal squares value candidates
75
76 f1_ok = true;
77 d1 = h[6] * h[7];
78 d2 = (h[7] - h[6]) * (h[7] + h[6]);
79 v1 = -(h[0] * h[1] + h[3] * h[4]) / d1;
80 v2 = (h[0] * h[0] + h[3] * h[3] - h[1] * h[1] - h[4] * h[4]) / d2;
81 if (v1 < v2) std::swap(v1, v2);
82 if (v1 > 0 && v2 > 0) f1 = sqrt(std::abs(d1) > std::abs(d2) ? v1 : v2);
83 else if (v1 > 0) f1 = sqrt(v1);
84 else f1_ok = false;
85
86 f0_ok = true;
87 d1 = h[0] * h[3] + h[1] * h[4];
88 d2 = h[0] * h[0] + h[1] * h[1] - h[3] * h[3] - h[4] * h[4];
89 v1 = -h[2] * h[5] / d1;
90 v2 = (h[5] * h[5] - h[2] * h[2]) / d2;
91 if (v1 < v2) std::swap(v1, v2);
92 if (v1 > 0 && v2 > 0) f0 = sqrt(std::abs(d1) > std::abs(d2) ? v1 : v2);
93 else if (v1 > 0) f0 = sqrt(v1);
94 else f0_ok = false;
95}

98void estimateFocal(const vector<ImageFeatures> &features, const vector<MatchesInfo> &pairwise_matches,
99 vector<double> &focals)
100{
101 const int num_images = static_cast<int>(features.size());
102 focals.resize(num_images);
103
104 vector<double> all_focals;
105
106 for (int i = 0; i < num_images; ++i)
107 {
108 for (int j = 0; j < num_images; ++j)
109 {
110 const MatchesInfo &m = pairwise_matches[i*num_images + j];
111 if (m.H.empty())
112 continue;
113 double f0, f1;
114 bool f0ok, f1ok;
115 focalsFromHomography(m.H, f0, f1, f0ok, f1ok);
116 if (f0ok && f1ok)
117 all_focals.push_back(sqrt(f0 * f1));
118 }
119 }
120
121 if (static_cast<int>(all_focals.size()) >= num_images - 1)
122 {
123 double median;
124
125 std::sort(all_focals.begin(), all_focals.end());
126 if (all_focals.size() % 2 == 1)
127 median = all_focals[all_focals.size() / 2];
128 else
129 median = (all_focals[all_focals.size() / 2 - 1] + all_focals[all_focals.size() / 2]) * 0.5;
130
131 for (int i = 0; i < num_images; ++i)
132 focals[i] = median;
133 }
134 else
135 {
136 LOGLN("Can't estimate focal length, will use naive approach");
137 double focals_sum = 0;
138 for (int i = 0; i < num_images; ++i)
139 focals_sum += features[i].img_size.width + features[i].img_size.height;
140 for (int i = 0; i < num_images; ++i)
141 focals[i] = focals_sum / num_images;
142 }
143}

146bool calibrateRotatingCamera(const vector<Mat> &Hs, Mat &K)
147{
148 int m = static_cast<int>(Hs.size());
149 CV_Assert(m >= 1);
150
151 vector<Mat> Hs_(m);
152 for (int i = 0; i < m; ++i)
153 {
154 CV_Assert(Hs[i].size() == Size(3, 3) && Hs[i].type() == CV_64F);
155 Hs_[i] = Hs[i] / pow(determinant(Hs[i]), 1./3.);
156 }
157
158 const int idx_map[3][3] = {{0, 1, 2}, {1, 3, 4}, {2, 4, 5}};
159 Mat_<double> A(6*m, 6);
160 A.setTo(0);
161
162 int eq_idx = 0;
163 for (int k = 0; k < m; ++k)
164 {
165 Mat_<double> H(Hs_[k]);
166 for (int i = 0; i < 3; ++i)
167 {
168 for (int j = i; j < 3; ++j, ++eq_idx)
169 {
170 for (int l = 0; l < 3; ++l)
171 {
172 for (int s = 0; s < 3; ++s)
173 {
174 int idx = idx_map[l][s];
175 A(eq_idx, idx) += H(i,l) * H(j,s);
176 }
177 }
178 A(eq_idx, idx_map[i][j]) -= 1;
179 }
180 }
181 }
182
183 Mat_<double> wcoef;
184 SVD::solveZ(A, wcoef);
185
186 Mat_<double> W(3,3);
187 for (int i = 0; i < 3; ++i)
188 for (int j = i; j < 3; ++j)
189 W(i,j) = W(j,i) = wcoef(idx_map[i][j], 0) / wcoef(5,0);
190 if (!decomposeCholesky(W.ptr<double>(), W.step, 3))
191 return false;
192 W(0,1) = W(0,2) = W(1,2) = 0;
193 K = W.t();
194 return true;
195}
