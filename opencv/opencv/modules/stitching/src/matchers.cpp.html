67struct MatchPairsBody : ParallelLoopBody
68{
69 MatchPairsBody(FeaturesMatcher &_matcher, const vector<ImageFeatures> &_features,
70 vector<MatchesInfo> &_pairwise_matches, vector<pair<int,int> > &_near_pairs)
71 : matcher(_matcher), features(_features),
72 pairwise_matches(_pairwise_matches), near_pairs(_near_pairs) {}
73
74 void operator ()(const Range &r) const
75 {
76 const int num_images = static_cast<int>(features.size());
77 for (int i = r.start; i < r.end; ++i)
78 {
79 int from = near_pairs[i].first;
80 int to = near_pairs[i].second;
81 int pair_idx = from*num_images + to;
82
83 matcher(features[from], features[to], pairwise_matches[pair_idx]);
84 pairwise_matches[pair_idx].src_img_idx = from;
85 pairwise_matches[pair_idx].dst_img_idx = to;
86
87 size_t dual_pair_idx = to*num_images + from;
88
89 pairwise_matches[dual_pair_idx] = pairwise_matches[pair_idx];
90 pairwise_matches[dual_pair_idx].src_img_idx = to;
91 pairwise_matches[dual_pair_idx].dst_img_idx = from;
92
93 if (!pairwise_matches[pair_idx].H.empty())
94 pairwise_matches[dual_pair_idx].H = pairwise_matches[pair_idx].H.inv();
95
96 for (size_t j = 0; j < pairwise_matches[dual_pair_idx].matches.size(); ++j)
97 std::swap(pairwise_matches[dual_pair_idx].matches[j].queryIdx,
98 pairwise_matches[dual_pair_idx].matches[j].trainIdx);
99 LOG(".");
100 }
101 }
102
103 FeaturesMatcher &matcher;
104 const vector<ImageFeatures> &features;
105 vector<MatchesInfo> &pairwise_matches;
106 vector<pair<int,int> > &near_pairs;
107
108private:
109 void operator =(const MatchPairsBody&);
110};

148void CpuMatcher::match(const ImageFeatures &features1, const ImageFeatures &features2, MatchesInfo& matches_info)
149{
150 CV_Assert(features1.descriptors.type() == features2.descriptors.type());
151 CV_Assert(features2.descriptors.depth() == CV_8U || features2.descriptors.depth() == CV_32F);
152
153#ifdef HAVE_TEGRA_OPTIMIZATION
154 if (tegra::match2nearest(features1, features2, matches_info, match_conf_))
155 return;
156#endif
157
158 matches_info.matches.clear();
159
160 Ptr<flann::IndexParams> indexParams = new flann::KDTreeIndexParams();
161 Ptr<flann::SearchParams> searchParams = new flann::SearchParams();
162
163 if (features2.descriptors.depth() == CV_8U)
164 {
165 indexParams->setAlgorithm(cvflann::FLANN_INDEX_LSH);
166 searchParams->setAlgorithm(cvflann::FLANN_INDEX_LSH);
167 }
168
169 FlannBasedMatcher matcher(indexParams, searchParams);
170 vector< vector<DMatch> > pair_matches;
171 MatchesSet matches;
172
173 // Find 1->2 matches
174 matcher.knnMatch(features1.descriptors, features2.descriptors, pair_matches, 2);
175 for (size_t i = 0; i < pair_matches.size(); ++i)
176 {
177 if (pair_matches[i].size() < 2)
178 continue;
179 const DMatch& m0 = pair_matches[i][0];
180 const DMatch& m1 = pair_matches[i][1];
181 if (m0.distance < (1.f - match_conf_) * m1.distance)
182 {
183 matches_info.matches.push_back(m0);
184 matches.insert(make_pair(m0.queryIdx, m0.trainIdx));
185 }
186 }
187 LOG("\n1->2 matches: " << matches_info.matches.size() << endl);
188
189 // Find 2->1 matches
190 pair_matches.clear();
191 matcher.knnMatch(features2.descriptors, features1.descriptors, pair_matches, 2);
192 for (size_t i = 0; i < pair_matches.size(); ++i)
193 {
194 if (pair_matches[i].size() < 2)
195 continue;
196 const DMatch& m0 = pair_matches[i][0];
197 const DMatch& m1 = pair_matches[i][1];
198 if (m0.distance < (1.f - match_conf_) * m1.distance)
199 if (matches.find(make_pair(m0.trainIdx, m0.queryIdx)) == matches.end())
200 matches_info.matches.push_back(DMatch(m0.trainIdx, m0.queryIdx, m0.distance));
201 }
202 LOG("1->2 & 2->1 matches: " << matches_info.matches.size() << endl);
203}

205#if defined(HAVE_OPENCV_GPU) && !defined(DYNAMIC_CUDA_SUPPORT)
206void GpuMatcher::match(const ImageFeatures &features1, const ImageFeatures &features2, MatchesInfo& matches_info)
207{
208 matches_info.matches.clear();
209
210 ensureSizeIsEnough(features1.descriptors.size(), features1.descriptors.type(), descriptors1_);
211 ensureSizeIsEnough(features2.descriptors.size(), features2.descriptors.type(), descriptors2_);
212
213 descriptors1_.upload(features1.descriptors);
214 descriptors2_.upload(features2.descriptors);
215
216 //TODO: NORM_L1 allows to avoid matcher crashes for ORB features, but is not absolutely correct for them.
217 // The best choice for ORB features is NORM_HAMMING, but it is incorrect for SURF features.
218 // More accurate fix in this place should be done in the future -- the type of the norm
219 // should be either a parameter of this method, or a field of the class.
220 BFMatcher_GPU matcher(NORM_L1);
221 MatchesSet matches;
222
223 // Find 1->2 matches
224 pair_matches.clear();
225 matcher.knnMatchSingle(descriptors1_, descriptors2_, train_idx_, distance_, all_dist_, 2);
226 matcher.knnMatchDownload(train_idx_, distance_, pair_matches);
227 for (size_t i = 0; i < pair_matches.size(); ++i)
228 {
229 if (pair_matches[i].size() < 2)
230 continue;
231 const DMatch& m0 = pair_matches[i][0];
232 const DMatch& m1 = pair_matches[i][1];
233 if (m0.distance < (1.f - match_conf_) * m1.distance)
234 {
235 matches_info.matches.push_back(m0);
236 matches.insert(make_pair(m0.queryIdx, m0.trainIdx));
237 }
238 }
239
240 // Find 2->1 matches
241 pair_matches.clear();
242 matcher.knnMatchSingle(descriptors2_, descriptors1_, train_idx_, distance_, all_dist_, 2);
243 matcher.knnMatchDownload(train_idx_, distance_, pair_matches);
244 for (size_t i = 0; i < pair_matches.size(); ++i)
245 {
246 if (pair_matches[i].size() < 2)
247 continue;
248 const DMatch& m0 = pair_matches[i][0];
249 const DMatch& m1 = pair_matches[i][1];
250 if (m0.distance < (1.f - match_conf_) * m1.distance)
251 if (matches.find(make_pair(m0.trainIdx, m0.queryIdx)) == matches.end())
252 matches_info.matches.push_back(DMatch(m0.trainIdx, m0.queryIdx, m0.distance));
253 }
254}

279
280void FeaturesFinder::operator ()(const Mat &image, ImageFeatures &features, const vector<Rect> &rois)
281{
282 vector<ImageFeatures> roi_features(rois.size());
283 size_t total_kps_count = 0;
284 int total_descriptors_height = 0;
285
286 for (size_t i = 0; i < rois.size(); ++i)
287 {
288 find(image(rois[i]), roi_features[i]);
289 total_kps_count += roi_features[i].keypoints.size();
290 total_descriptors_height += roi_features[i].descriptors.rows;
291 }
292
293 features.img_size = image.size();
294 features.keypoints.resize(total_kps_count);
295 features.descriptors.create(total_descriptors_height,
296 roi_features[0].descriptors.cols,
297 roi_features[0].descriptors.type());
298
299 int kp_idx = 0;
300 int descr_offset = 0;
301 for (size_t i = 0; i < rois.size(); ++i)
302 {
303 for (size_t j = 0; j < roi_features[i].keypoints.size(); ++j, ++kp_idx)
304 {
305 features.keypoints[kp_idx] = roi_features[i].keypoints[j];
306 features.keypoints[kp_idx].pt.x += (float)rois[i].x;
307 features.keypoints[kp_idx].pt.y += (float)rois[i].y;
308 }
309 Mat subdescr = features.descriptors.rowRange(
310 descr_offset, descr_offset + roi_features[i].descriptors.rows);
311 roi_features[i].descriptors.copyTo(subdescr);
312 descr_offset += roi_features[i].descriptors.rows;
313 }
314}

317SurfFeaturesFinder::SurfFeaturesFinder(double hess_thresh, int num_octaves, int num_layers,
318 int num_octaves_descr, int num_layers_descr)
319{
320 if (num_octaves_descr == num_octaves && num_layers_descr == num_layers)
321 {
322 surf = Algorithm::create<Feature2D>("Feature2D.SURF");
323 if( surf.empty() )
324 CV_Error( CV_StsNotImplemented, "OpenCV was built without SURF support" );
325 surf->set("hessianThreshold", hess_thresh);
326 surf->set("nOctaves", num_octaves);
327 surf->set("nOctaveLayers", num_layers);
328 }
329 else
330 {
331 detector_ = Algorithm::create<FeatureDetector>("Feature2D.SURF");
332 extractor_ = Algorithm::create<DescriptorExtractor>("Feature2D.SURF");
333
334 if( detector_.empty() || extractor_.empty() )
335 CV_Error( CV_StsNotImplemented, "OpenCV was built without SURF support" );
336
337 detector_->set("hessianThreshold", hess_thresh);
338 detector_->set("nOctaves", num_octaves);
339 detector_->set("nOctaveLayers", num_layers);
340
341 extractor_->set("nOctaves", num_octaves_descr);
342 extractor_->set("nOctaveLayers", num_layers_descr);
343 }
344}

346void SurfFeaturesFinder::find(const Mat &image, ImageFeatures &features)
347{
348 Mat gray_image;
349 CV_Assert((image.type() == CV_8UC3) || (image.type() == CV_8UC1));
350 if(image.type() == CV_8UC3)
351 {
352 cvtColor(image, gray_image, CV_BGR2GRAY);
353 }
354 else
355 {
356 gray_image = image;
357 }
358 if (surf == 0)
359 {
360 detector_->detect(gray_image, features.keypoints);
361 extractor_->compute(gray_image, features.keypoints, features.descriptors);
362 }
363 else
364 {
365 Mat descriptors;
366 (*surf)(gray_image, Mat(), features.keypoints, descriptors);
367 features.descriptors = descriptors.reshape(1, (int)features.keypoints.size());
368 }
369}

377void OrbFeaturesFinder::find(const Mat &image, ImageFeatures &features)
378{
379 Mat gray_image;
380
381 CV_Assert((image.type() == CV_8UC3) || (image.type() == CV_8UC4) || (image.type() == CV_8UC1));
382
383 if (image.type() == CV_8UC3) {
384 cvtColor(image, gray_image, CV_BGR2GRAY);
385 } else if (image.type() == CV_8UC4) {
386 cvtColor(image, gray_image, CV_BGRA2GRAY);
387 } else if (image.type() == CV_8UC1) {
388 gray_image=image;
389 } else {
390 CV_Error(CV_StsUnsupportedFormat, "");
391 }
392
393 if (grid_size.area() == 1)
394 (*orb)(gray_image, Mat(), features.keypoints, features.descriptors);
395 else
396 {
397 features.keypoints.clear();
398 features.descriptors.release();
399
400 std::vector<KeyPoint> points;
401 Mat descriptors;
402
403 for (int r = 0; r < grid_size.height; ++r)
404 for (int c = 0; c < grid_size.width; ++c)
405 {
406 int xl = c * gray_image.cols / grid_size.width;
407 int yl = r * gray_image.rows / grid_size.height;
408 int xr = (c+1) * gray_image.cols / grid_size.width;
409 int yr = (r+1) * gray_image.rows / grid_size.height;
410
411 // LOGLN("OrbFeaturesFinder::find: gray_image.empty=" << (gray_image.empty()?"true":"false") << ", "
412 // << " gray_image.size()=(" << gray_image.size().width << "x" << gray_image.size().height << "), "
413 // << " yl=" << yl << ", yr=" << yr << ", "
414 // << " xl=" << xl << ", xr=" << xr << ", gray_image.data=" << ((size_t)gray_image.data) << ", "
415 // << "gray_image.dims=" << gray_image.dims << "\n");
416
417 Mat gray_image_part=gray_image(Range(yl, yr), Range(xl, xr));
418 // LOGLN("OrbFeaturesFinder::find: gray_image_part.empty=" << (gray_image_part.empty()?"true":"false") << ", "
419 // << " gray_image_part.size()=(" << gray_image_part.size().width << "x" << gray_image_part.size().height << "), "
420 // << " gray_image_part.dims=" << gray_image_part.dims << ", "
421 // << " gray_image_part.data=" << ((size_t)gray_image_part.data) << "\n");
422
423 (*orb)(gray_image_part, Mat(), points, descriptors);
424
425 features.keypoints.reserve(features.keypoints.size() + points.size());
426 for (std::vector<KeyPoint>::iterator kp = points.begin(); kp != points.end(); ++kp)
427 {
428 kp->pt.x += xl;
429 kp->pt.y += yl;
430 features.keypoints.push_back(*kp);
431 }
432 features.descriptors.push_back(descriptors);
433 }
434 }
435}

471

473}

530void FeaturesMatcher::operator ()(const vector<ImageFeatures> &features, vector<MatchesInfo> &pairwise_matches,
531 const Mat &mask)
532{
533 const int num_images = static_cast<int>(features.size());
534
535 CV_Assert(mask.empty() || (mask.type() == CV_8U && mask.cols == num_images && mask.rows));
536 Mat_<uchar> mask_(mask);
537 if (mask_.empty())
538 mask_ = Mat::ones(num_images, num_images, CV_8U);
539
540 vector<pair<int,int> > near_pairs;
541 for (int i = 0; i < num_images - 1; ++i)
542 for (int j = i + 1; j < num_images; ++j)
543 if (features[i].keypoints.size() > 0 && features[j].keypoints.size() > 0 && mask_(i, j))
544 near_pairs.push_back(make_pair(i, j));
545
546 pairwise_matches.resize(num_images * num_images);
547 MatchPairsBody body(*this, features, pairwise_matches, near_pairs);
548
549 if (is_thread_safe_)
550 parallel_for_(Range(0, static_cast<int>(near_pairs.size())), body);
551 else
552 body(Range(0, static_cast<int>(near_pairs.size())));
553 LOGLN_CHAT("");
554}

572 num_matches_thresh2_ = num_matches_thresh2;

575
576void BestOf2NearestMatcher::match(const ImageFeatures &features1, const ImageFeatures &features2,
577 MatchesInfo &matches_info)
578{
579 (*impl_)(features1, features2, matches_info);
580
581 // Check if it makes sense to find homography
582 if (matches_info.matches.size() < static_cast<size_t>(num_matches_thresh1_))
583 return;
584
585 // Construct point-point correspondences for homography estimation
586 Mat src_points(1, static_cast<int>(matches_info.matches.size()), CV_32FC2);
587 Mat dst_points(1, static_cast<int>(matches_info.matches.size()), CV_32FC2);
588 for (size_t i = 0; i < matches_info.matches.size(); ++i)
589 {
590 const DMatch& m = matches_info.matches[i];
591
592 Point2f p = features1.keypoints[m.queryIdx].pt;
593 p.x -= features1.img_size.width * 0.5f;
594 p.y -= features1.img_size.height * 0.5f;
595 src_points.at<Point2f>(0, static_cast<int>(i)) = p;
596
597 p = features2.keypoints[m.trainIdx].pt;
598 p.x -= features2.img_size.width * 0.5f;
599 p.y -= features2.img_size.height * 0.5f;
600 dst_points.at<Point2f>(0, static_cast<int>(i)) = p;
601 }
602
603 // Find pair-wise motion
604 matches_info.H = findHomography(src_points, dst_points, matches_info.inliers_mask, CV_RANSAC);
605 if (std::abs(determinant(matches_info.H)) < numeric_limits<double>::epsilon())
606 return;
607
608 // Find number of inliers
609 matches_info.num_inliers = 0;
610 for (size_t i = 0; i < matches_info.inliers_mask.size(); ++i)
611 if (matches_info.inliers_mask[i])
612 matches_info.num_inliers++;
613
614 // These coeffs are from paper M. Brown and D. Lowe. "Automatic Panoramic Image Stitching
615 // using Invariant Features"
616 matches_info.confidence = matches_info.num_inliers / (8 + 0.3 * matches_info.matches.size());
617
618 // Set zero confidence to remove matches between too close images, as they don't provide
619 // additional information anyway. The threshold was set experimentally.
620 matches_info.confidence = matches_info.confidence > 3. ? 0. : matches_info.confidence;
621
622 // Check if we should try to refine motion
623 if (matches_info.num_inliers < num_matches_thresh2_)
624 return;
625
626 // Construct point-point correspondences for inliers only
627 src_points.create(1, matches_info.num_inliers, CV_32FC2);
628 dst_points.create(1, matches_info.num_inliers, CV_32FC2);
629 int inlier_idx = 0;
630 for (size_t i = 0; i < matches_info.matches.size(); ++i)
631 {
632 if (!matches_info.inliers_mask[i])
633 continue;
634
635 const DMatch& m = matches_info.matches[i];
636
637 Point2f p = features1.keypoints[m.queryIdx].pt;
638 p.x -= features1.img_size.width * 0.5f;
639 p.y -= features1.img_size.height * 0.5f;
640 src_points.at<Point2f>(0, inlier_idx) = p;
641
642 p = features2.keypoints[m.trainIdx].pt;
643 p.x -= features2.img_size.width * 0.5f;
644 p.y -= features2.img_size.height * 0.5f;
645 dst_points.at<Point2f>(0, inlier_idx) = p;
646
647 inlier_idx++;
648 }
649
650 // Rerun motion estimation on inliers only
651 matches_info.H = findHomography(src_points, dst_points, CV_RANSAC);
652}
