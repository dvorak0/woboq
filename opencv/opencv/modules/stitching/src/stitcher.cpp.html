49Stitcher Stitcher::createDefault(bool try_use_gpu)
50{
51 Stitcher stitcher;
52 stitcher.setRegistrationResol(0.6);
53 stitcher.setSeamEstimationResol(0.1);
54 stitcher.setCompositingResol(ORIG_RESOL);
55 stitcher.setPanoConfidenceThresh(1);
56 stitcher.setWaveCorrection(true);
57 stitcher.setWaveCorrectKind(detail::WAVE_CORRECT_HORIZ);
58 stitcher.setFeaturesMatcher(new detail::BestOf2NearestMatcher(try_use_gpu));
59 stitcher.setBundleAdjuster(new detail::BundleAdjusterRay());
60
61#if defined(HAVE_OPENCV_GPU) && !defined(DYNAMIC_CUDA_SUPPORT)
62 if (try_use_gpu && gpu::getCudaEnabledDeviceCount() > 0)
63 {
64#if defined(HAVE_OPENCV_NONFREE)
65 stitcher.setFeaturesFinder(new detail::SurfFeaturesFinderGpu());
66#else
67 stitcher.setFeaturesFinder(new detail::OrbFeaturesFinder());
68#endif
69 stitcher.setWarper(new SphericalWarperGpu());
70 stitcher.setSeamFinder(new detail::GraphCutSeamFinderGpu());
71 }
72 else
73#endif
74 {
75#ifdef HAVE_OPENCV_NONFREE
76 stitcher.setFeaturesFinder(new detail::SurfFeaturesFinder());
77#else
78 stitcher.setFeaturesFinder(new detail::OrbFeaturesFinder());
79#endif
80 stitcher.setWarper(new SphericalWarper());
81 stitcher.setSeamFinder(new detail::GraphCutSeamFinder(detail::GraphCutSeamFinderBase::COST_COLOR));
82 }
83
84 stitcher.setExposureCompensator(new detail::BlocksGainCompensator());
85 stitcher.setBlender(new detail::MultiBandBlender(try_use_gpu));
86
87 return stitcher;
88}

97Stitcher::Status Stitcher::estimateTransform(InputArray images, const vector<vector<Rect> > &rois)
98{
99 images.getMatVector(imgs_);
100 rois_ = rois;
101
102 Status status;
103
104 if ((status = matchImages()) != OK)
105 return status;
106
107 estimateCameraParams();
108
109 return OK;
110}

120Stitcher::Status Stitcher::composePanorama(InputArray images, OutputArray pano)
121{
122 LOGLN("Warping images (auxiliary)... ");
123
124 vector<Mat> imgs;
125 images.getMatVector(imgs);
126 if (!imgs.empty())
127 {
128 CV_Assert(imgs.size() == imgs_.size());
129
130 Mat img;
131 seam_est_imgs_.resize(imgs.size());
132
133 for (size_t i = 0; i < imgs.size(); ++i)
134 {
135 imgs_[i] = imgs[i];
136 resize(imgs[i], img, Size(), seam_scale_, seam_scale_);
137 seam_est_imgs_[i] = img.clone();
138 }
139
140 vector<Mat> seam_est_imgs_subset;
141 vector<Mat> imgs_subset;
142
143 for (size_t i = 0; i < indices_.size(); ++i)
144 {
145 imgs_subset.push_back(imgs_[indices_[i]]);
146 seam_est_imgs_subset.push_back(seam_est_imgs_[indices_[i]]);
147 }
148
149 seam_est_imgs_ = seam_est_imgs_subset;
150 imgs_ = imgs_subset;
151 }
152
153 Mat &pano_ = pano.getMatRef();
154
155#if ENABLE_LOG
156 int64 t = getTickCount();
157#endif
158
159 vector<Point> corners(imgs_.size());
160 vector<Mat> masks_warped(imgs_.size());
161 vector<Mat> images_warped(imgs_.size());
162 vector<Size> sizes(imgs_.size());
163 vector<Mat> masks(imgs_.size());
164
165 // Prepare image masks
166 for (size_t i = 0; i < imgs_.size(); ++i)
167 {
168 masks[i].create(seam_est_imgs_[i].size(), CV_8U);
169 masks[i].setTo(Scalar::all(255));
170 }
171
172 // Warp images and their masks
173 Ptr<detail::RotationWarper> w = warper_->create(float(warped_image_scale_ * seam_work_aspect_));
174 for (size_t i = 0; i < imgs_.size(); ++i)
175 {
176 Mat_<float> K;
177 cameras_[i].K().convertTo(K, CV_32F);
178 K(0,0) *= (float)seam_work_aspect_;
179 K(0,2) *= (float)seam_work_aspect_;
180 K(1,1) *= (float)seam_work_aspect_;
181 K(1,2) *= (float)seam_work_aspect_;
182
183 corners[i] = w->warp(seam_est_imgs_[i], K, cameras_[i].R, INTER_LINEAR, BORDER_REFLECT, images_warped[i]);
184 sizes[i] = images_warped[i].size();
185
186 w->warp(masks[i], K, cameras_[i].R, INTER_NEAREST, BORDER_CONSTANT, masks_warped[i]);
187 }
188
189 vector<Mat> images_warped_f(imgs_.size());
190 for (size_t i = 0; i < imgs_.size(); ++i)
191 images_warped[i].convertTo(images_warped_f[i], CV_32F);
192
193 LOGLN("Warping images, time: " << ((getTickCount() - t) / getTickFrequency()) << " sec");
194
195 // Find seams
196 exposure_comp_->feed(corners, images_warped, masks_warped);
197 seam_finder_->find(images_warped_f, corners, masks_warped);
198
199 // Release unused memory
200 seam_est_imgs_.clear();
201 images_warped.clear();
202 images_warped_f.clear();
203 masks.clear();
204
205 LOGLN("Compositing...");
206#if ENABLE_LOG
207 t = getTickCount();
208#endif
209
210 Mat img_warped, img_warped_s;
211 Mat dilated_mask, seam_mask, mask, mask_warped;
212
213 //double compose_seam_aspect = 1;
214 double compose_work_aspect = 1;
215 bool is_blender_prepared = false;
216
217 double compose_scale = 1;
218 bool is_compose_scale_set = false;
219
220 Mat full_img, img;
221 for (size_t img_idx = 0; img_idx < imgs_.size(); ++img_idx)
222 {
223 LOGLN("Compositing image #" << indices_[img_idx] + 1);
224
225 // Read image and resize it if necessary
226 full_img = imgs_[img_idx];
227 if (!is_compose_scale_set)
228 {
229 if (compose_resol_ > 0)
230 compose_scale = min(1.0, sqrt(compose_resol_ * 1e6 / full_img.size().area()));
231 is_compose_scale_set = true;
232
233 // Compute relative scales
234 //compose_seam_aspect = compose_scale / seam_scale_;
235 compose_work_aspect = compose_scale / work_scale_;
236
237 // Update warped image scale
238 warped_image_scale_ *= static_cast<float>(compose_work_aspect);
239 w = warper_->create((float)warped_image_scale_);
240
241 // Update corners and sizes
242 for (size_t i = 0; i < imgs_.size(); ++i)
243 {
244 // Update intrinsics
245 cameras_[i].focal *= compose_work_aspect;
246 cameras_[i].ppx *= compose_work_aspect;
247 cameras_[i].ppy *= compose_work_aspect;
248
249 // Update corner and size
250 Size sz = full_img_sizes_[i];
251 if (std::abs(compose_scale - 1) > 1e-1)
252 {
253 sz.width = cvRound(full_img_sizes_[i].width * compose_scale);
254 sz.height = cvRound(full_img_sizes_[i].height * compose_scale);
255 }
256
257 Mat K;
258 cameras_[i].K().convertTo(K, CV_32F);
259 Rect roi = w->warpRoi(sz, K, cameras_[i].R);
260 corners[i] = roi.tl();
261 sizes[i] = roi.size();
262 }
263 }
264 if (std::abs(compose_scale - 1) > 1e-1)
265 resize(full_img, img, Size(), compose_scale, compose_scale);
266 else
267 img = full_img;
268 full_img.release();
269 Size img_size = img.size();
270
271 Mat K;
272 cameras_[img_idx].K().convertTo(K, CV_32F);
273
274 // Warp the current image
275 w->warp(img, K, cameras_[img_idx].R, INTER_LINEAR, BORDER_REFLECT, img_warped);
276
277 // Warp the current image mask
278 mask.create(img_size, CV_8U);
279 mask.setTo(Scalar::all(255));
280 w->warp(mask, K, cameras_[img_idx].R, INTER_NEAREST, BORDER_CONSTANT, mask_warped);
281
282 // Compensate exposure
283 exposure_comp_->apply((int)img_idx, corners[img_idx], img_warped, mask_warped);
284
285 img_warped.convertTo(img_warped_s, CV_16S);
286 img_warped.release();
287 img.release();
288 mask.release();
289
290 // Make sure seam mask has proper size
291 dilate(masks_warped[img_idx], dilated_mask, Mat());
292 resize(dilated_mask, seam_mask, mask_warped.size());
293
294 mask_warped = seam_mask & mask_warped;
295
296 if (!is_blender_prepared)
297 {
298 blender_->prepare(corners, sizes);
299 is_blender_prepared = true;
300 }
301
302 // Blend the current image
303 blender_->feed(img_warped_s, mask_warped, corners[img_idx]);
304 }
305
306 Mat result, result_mask;
307 blender_->blend(result, result_mask);
308
309 LOGLN("Compositing, time: " << ((getTickCount() - t) / getTickFrequency()) << " sec");
310
311 // Preliminary result is in CV_16SC3 format, but all values are in [0,255] range,
312 // so convert it to avoid user confusing
313 result.convertTo(pano_, CV_8U);
314
315 return OK;
316}

337Stitcher::Status Stitcher::matchImages()
338{
339 if ((int)imgs_.size() < 2)
340 {
341 LOGLN("Need more images");
342 return ERR_NEED_MORE_IMGS;
343 }
344
345 work_scale_ = 1;
346 seam_work_aspect_ = 1;
347 seam_scale_ = 1;
348 bool is_work_scale_set = false;
349 bool is_seam_scale_set = false;
350 Mat full_img, img;
351 features_.resize(imgs_.size());
352 seam_est_imgs_.resize(imgs_.size());
353 full_img_sizes_.resize(imgs_.size());
354
355 LOGLN("Finding features...");
356#if ENABLE_LOG
357 int64 t = getTickCount();
358#endif
359
360 for (size_t i = 0; i < imgs_.size(); ++i)
361 {
362 full_img = imgs_[i];
363 full_img_sizes_[i] = full_img.size();
364
365 if (registr_resol_ < 0)
366 {
367 img = full_img;
368 work_scale_ = 1;
369 is_work_scale_set = true;
370 }
371 else
372 {
373 if (!is_work_scale_set)
374 {
375 work_scale_ = min(1.0, sqrt(registr_resol_ * 1e6 / full_img.size().area()));
376 is_work_scale_set = true;
377 }
378 resize(full_img, img, Size(), work_scale_, work_scale_);
379 }
380 if (!is_seam_scale_set)
381 {
382 seam_scale_ = min(1.0, sqrt(seam_est_resol_ * 1e6 / full_img.size().area()));
383 seam_work_aspect_ = seam_scale_ / work_scale_;
384 is_seam_scale_set = true;
385 }
386
387 if (rois_.empty())
388 (*features_finder_)(img, features_[i]);
389 else
390 {
391 vector<Rect> rois(rois_[i].size());
392 for (size_t j = 0; j < rois_[i].size(); ++j)
393 {
394 Point tl(cvRound(rois_[i][j].x * work_scale_), cvRound(rois_[i][j].y * work_scale_));
395 Point br(cvRound(rois_[i][j].br().x * work_scale_), cvRound(rois_[i][j].br().y * work_scale_));
396 rois[j] = Rect(tl, br);
397 }
398 (*features_finder_)(img, features_[i], rois);
399 }
400 features_[i].img_idx = (int)i;
401 LOGLN("Features in image #" << i+1 << ": " << features_[i].keypoints.size());
402
403 resize(full_img, img, Size(), seam_scale_, seam_scale_);
404 seam_est_imgs_[i] = img.clone();
405 }
406
407 // Do it to save memory
408 features_finder_->collectGarbage();
409 full_img.release();
410 img.release();
411
412 LOGLN("Finding features, time: " << ((getTickCount() - t) / getTickFrequency()) << " sec");
413
414 LOG("Pairwise matching");
415#if ENABLE_LOG
416 t = getTickCount();
417#endif
418 (*features_matcher_)(features_, pairwise_matches_, matching_mask_);
419 features_matcher_->collectGarbage();
420 LOGLN("Pairwise matching, time: " << ((getTickCount() - t) / getTickFrequency()) << " sec");
421
422 // Leave only images we are sure are from the same panorama
423 indices_ = detail::leaveBiggestComponent(features_, pairwise_matches_, (float)conf_thresh_);
424 vector<Mat> seam_est_imgs_subset;
425 vector<Mat> imgs_subset;
426 vector<Size> full_img_sizes_subset;
427 for (size_t i = 0; i < indices_.size(); ++i)
428 {
429 imgs_subset.push_back(imgs_[indices_[i]]);
430 seam_est_imgs_subset.push_back(seam_est_imgs_[indices_[i]]);
431 full_img_sizes_subset.push_back(full_img_sizes_[indices_[i]]);
432 }
433 seam_est_imgs_ = seam_est_imgs_subset;
434 imgs_ = imgs_subset;
435 full_img_sizes_ = full_img_sizes_subset;
436
437 if ((int)imgs_.size() < 2)
438 {
439 LOGLN("Need more images");
440 return ERR_NEED_MORE_IMGS;
441 }
442
443 return OK;
444}

447void Stitcher::estimateCameraParams()
448{
449 detail::HomographyBasedEstimator estimator;
450 estimator(features_, pairwise_matches_, cameras_);
451
452 for (size_t i = 0; i < cameras_.size(); ++i)
453 {
454 Mat R;
455 cameras_[i].R.convertTo(R, CV_32F);
456 cameras_[i].R = R;
457 LOGLN("Initial intrinsic parameters #" << indices_[i] + 1 << ":\n " << cameras_[i].K());
458 }
459
460 bundle_adjuster_->setConfThresh(conf_thresh_);
461 (*bundle_adjuster_)(features_, pairwise_matches_, cameras_);
462
463 // Find median focal length and use it as final image scale
464 vector<double> focals;
465 for (size_t i = 0; i < cameras_.size(); ++i)
466 {
467 LOGLN("Camera #" << indices_[i] + 1 << ":\n" << cameras_[i].K());
468 focals.push_back(cameras_[i].focal);
469 }
470
471 std::sort(focals.begin(), focals.end());
472 if (focals.size() % 2 == 1)
473 warped_image_scale_ = static_cast<float>(focals[focals.size() / 2]);
474 else
475 warped_image_scale_ = static_cast<float>(focals[focals.size() / 2 - 1] + focals[focals.size() / 2]) * 0.5f;
476
477 if (do_wave_correct_)
478 {
479 vector<Mat> rmats;
480 for (size_t i = 0; i < cameras_.size(); ++i)
481 rmats.push_back(cameras_[i].R.clone());
482 detail::waveCorrect(rmats, wave_correct_kind_);
483 for (size_t i = 0; i < cameras_.size(); ++i)
484 cameras_[i].R = rmats[i];
485 }
486}
