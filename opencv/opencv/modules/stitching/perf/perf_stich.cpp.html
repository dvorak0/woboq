28PERF_TEST_P(stitch, a123, TEST_DETECTORS)
29{
30 Mat pano;
31
32 vector<Mat> imgs;
33 imgs.push_back( imread( getDataPath("stitching/a1.png") ) );
34 imgs.push_back( imread( getDataPath("stitching/a2.png") ) );
35 imgs.push_back( imread( getDataPath("stitching/a3.png") ) );
36
37 Ptr<detail::FeaturesFinder> featuresFinder = GetParam() == "orb"
38 ? (detail::FeaturesFinder*)new detail::OrbFeaturesFinder()
39 : (detail::FeaturesFinder*)new detail::SurfFeaturesFinder();
40
41 Ptr<detail::FeaturesMatcher> featuresMatcher = GetParam() == "orb"
42 ? new detail::BestOf2NearestMatcher(false, ORB_MATCH_CONFIDENCE)
43 : new detail::BestOf2NearestMatcher(false, SURF_MATCH_CONFIDENCE);
44
45 declare.time(30 * 20).iterations(20);
46
47 while(next())
48 {
49 Stitcher stitcher = Stitcher::createDefault();
50 stitcher.setFeaturesFinder(featuresFinder);
51 stitcher.setFeaturesMatcher(featuresMatcher);
52 stitcher.setWarper(new SphericalWarper());
53 stitcher.setRegistrationResol(WORK_MEGAPIX);
54
55 startTimer();
56 stitcher.stitch(imgs, pano);
57 stopTimer();
58 }
59
60 Mat pano_small;
61 if (!pano.empty())
62 resize(pano, pano_small, Size(320, 240), 0, 0, INTER_AREA);
63
64 SANITY_CHECK(pano_small, 5);
65}

105PERF_TEST_P( match, bestOf2Nearest, TEST_DETECTORS)
106{
107 Mat img1, img1_full = imread( getDataPath("stitching/b1.png") );
108 Mat img2, img2_full = imread( getDataPath("stitching/b2.png") );
109 float scale1 = (float)std::min(1.0, sqrt(WORK_MEGAPIX * 1e6 / img1_full.total()));
110 float scale2 = (float)std::min(1.0, sqrt(WORK_MEGAPIX * 1e6 / img2_full.total()));
111 resize(img1_full, img1, Size(), scale1, scale1);
112 resize(img2_full, img2, Size(), scale2, scale2);
113
114 Ptr<detail::FeaturesFinder> finder;
115 Ptr<detail::FeaturesMatcher> matcher;
116 if (GetParam() == "surf")
117 {
118 finder = new detail::SurfFeaturesFinder();
119 matcher = new detail::BestOf2NearestMatcher(false, SURF_MATCH_CONFIDENCE);
120 }
121 else if (GetParam() == "orb")
122 {
123 finder = new detail::OrbFeaturesFinder();
124 matcher = new detail::BestOf2NearestMatcher(false, ORB_MATCH_CONFIDENCE);
125 }
126 else
127 {
128 FAIL() << "Unknown 2D features type: " << GetParam();
129 }
130
131 detail::ImageFeatures features1, features2;
132 (*finder)(img1, features1);
133 (*finder)(img2, features2);
134
135 detail::MatchesInfo pairwise_matches;
136
137 declare.in(features1.descriptors, features2.descriptors);
138
139 while(next())
140 {
141 cvflann::seed_random(42);//for predictive FlannBasedMatcher
142 startTimer();
143 (*matcher)(features1, features2, pairwise_matches);
144 stopTimer();
145 matcher->collectGarbage();
146 }
147
148 std::vector<DMatch>& matches = pairwise_matches.matches;
149 if (GetParam() == "orb") matches.resize(0);
150 for(size_t q = 0; q < matches.size(); ++q)
151 if (matches[q].imgIdx < 0) { matches.resize(q); break;}
152 SANITY_CHECK_MATCHES(matches);
153}

155PERF_TEST_P( matchVector, bestOf2NearestVectorFeatures, testing::Combine(
156 TEST_DETECTORS,
157 testing::Values(2, 4, 8))
158 )
159{
160 Mat img1, img1_full = imread( getDataPath("stitching/b1.png") );
161 Mat img2, img2_full = imread( getDataPath("stitching/b2.png") );
162 float scale1 = (float)std::min(1.0, sqrt(WORK_MEGAPIX * 1e6 / img1_full.total()));
163 float scale2 = (float)std::min(1.0, sqrt(WORK_MEGAPIX * 1e6 / img2_full.total()));
164 resize(img1_full, img1, Size(), scale1, scale1);
165 resize(img2_full, img2, Size(), scale2, scale2);
166
167 Ptr<detail::FeaturesFinder> finder;
168 Ptr<detail::FeaturesMatcher> matcher;
169 String detectorName = get<0>(GetParam());
170 int featuresVectorSize = get<1>(GetParam());
171 if (detectorName == "surf")
172 {
173 finder = new detail::SurfFeaturesFinder();
174 matcher = new detail::BestOf2NearestMatcher(false, SURF_MATCH_CONFIDENCE);
175 }
176 else if (detectorName == "orb")
177 {
178 finder = new detail::OrbFeaturesFinder();
179 matcher = new detail::BestOf2NearestMatcher(false, ORB_MATCH_CONFIDENCE);
180 }
181 else
182 {
183 FAIL() << "Unknown 2D features type: " << get<0>(GetParam());
184 }
185
186 detail::ImageFeatures features1, features2;
187 (*finder)(img1, features1);
188 (*finder)(img2, features2);
189 vector<detail::ImageFeatures> features;
190 vector<detail::MatchesInfo> pairwise_matches;
191 for(int i = 0; i < featuresVectorSize/2; i++)
192 {
193 features.push_back(features1);
194 features.push_back(features2);
195 }
196
197 declare.time(200);
198 while(next())
199 {
200 cvflann::seed_random(42);//for predictive FlannBasedMatcher
201 startTimer();
202 (*matcher)(features, pairwise_matches);
203 stopTimer();
204 matcher->collectGarbage();
205 }
206
207
208 std::vector<DMatch>& matches = pairwise_matches[detectorName == "surf" ? 1 : 0].matches;
209 for(size_t q = 0; q < matches.size(); ++q)
210 if (matches[q].imgIdx < 0) { matches.resize(q); break;}
211 SANITY_CHECK_MATCHES(matches);
212}
