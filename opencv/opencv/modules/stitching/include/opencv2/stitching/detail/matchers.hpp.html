59struct CV_EXPORTS ImageFeatures
60{
61 int img_idx;
62 Size img_size;
63 std::vector<KeyPoint> keypoints;
64 Mat descriptors;
65};

68class CV_EXPORTS FeaturesFinder
69{
70public:
71 virtual ~FeaturesFinder() {}
72 void operator ()(const Mat &image, ImageFeatures &features);
73 void operator ()(const Mat &image, ImageFeatures &features, const std::vector<cv::Rect> &rois);
74 virtual void collectGarbage() {}
75
76protected:
77 virtual void find(const Mat &image, ImageFeatures &features) = 0;
78};

133struct CV_EXPORTS MatchesInfo
134{
135 MatchesInfo();
136 MatchesInfo(const MatchesInfo &other);
137 const MatchesInfo& operator =(const MatchesInfo &other);
138
139 int src_img_idx, dst_img_idx; // Images indices (optional)
140 std::vector<DMatch> matches;
141 std::vector<uchar> inliers_mask; // Geometrically consistent matches mask
142 int num_inliers; // Number of geometrically consistent matches
143 Mat H; // Estimated homography
144 double confidence; // Confidence two images are from the same panorama
145};
146
147
148class CV_EXPORTS FeaturesMatcher
149{
150public:
151 virtual ~FeaturesMatcher() {}
152
153 void operator ()(const ImageFeatures &features1, const ImageFeatures &features2,
154 MatchesInfo& matches_info) { match(features1, features2, matches_info); }
155
156 void operator ()(const std::vector<ImageFeatures> &features, std::vector<MatchesInfo> &pairwise_matches,
157 const cv::Mat &mask = cv::Mat());
158
159 bool isThreadSafe() const { return is_thread_safe_; }
160
161 virtual void collectGarbage() {}
162
163protected:
164 FeaturesMatcher(bool is_thread_safe = false) : is_thread_safe_(is_thread_safe) {}
165
166 virtual void match(const ImageFeatures &features1, const ImageFeatures &features2,
167 MatchesInfo& matches_info) = 0;
168
169 bool is_thread_safe_;
170};
