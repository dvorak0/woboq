61namespace cv {
62
63class CV_EXPORTS Stitcher
64{
65public:
66 enum { ORIG_RESOL = -1 };
67 enum Status { OK, ERR_NEED_MORE_IMGS };
68
69 // Creates stitcher with default parameters
70 static Stitcher createDefault(bool try_use_gpu = false);
71
72 double registrationResol() const { return registr_resol_; }
73 void setRegistrationResol(double resol_mpx) { registr_resol_ = resol_mpx; }
74
75 double seamEstimationResol() const { return seam_est_resol_; }
76 void setSeamEstimationResol(double resol_mpx) { seam_est_resol_ = resol_mpx; }
77
78 double compositingResol() const { return compose_resol_; }
79 void setCompositingResol(double resol_mpx) { compose_resol_ = resol_mpx; }
80
81 double panoConfidenceThresh() const { return conf_thresh_; }
82 void setPanoConfidenceThresh(double conf_thresh) { conf_thresh_ = conf_thresh; }
83
84 bool waveCorrection() const { return do_wave_correct_; }
85 void setWaveCorrection(bool flag) { do_wave_correct_ = flag; }
86
87 detail::WaveCorrectKind waveCorrectKind() const { return wave_correct_kind_; }
88 void setWaveCorrectKind(detail::WaveCorrectKind kind) { wave_correct_kind_ = kind; }
89
90 Ptr<detail::FeaturesFinder> featuresFinder() { return features_finder_; }
91 const Ptr<detail::FeaturesFinder> featuresFinder() const { return features_finder_; }
92 void setFeaturesFinder(Ptr<detail::FeaturesFinder> features_finder)
93 { features_finder_ = features_finder; }
94
95 Ptr<detail::FeaturesMatcher> featuresMatcher() { return features_matcher_; }
96 const Ptr<detail::FeaturesMatcher> featuresMatcher() const { return features_matcher_; }
97 void setFeaturesMatcher(Ptr<detail::FeaturesMatcher> features_matcher)
98 { features_matcher_ = features_matcher; }
99
100 const cv::Mat& matchingMask() const { return matching_mask_; }
101 void setMatchingMask(const cv::Mat &mask)
102 {
103 CV_Assert(mask.type() == CV_8U && mask.cols == mask.rows);
104 matching_mask_ = mask.clone();
105 }
106
107 Ptr<detail::BundleAdjusterBase> bundleAdjuster() { return bundle_adjuster_; }
108 const Ptr<detail::BundleAdjusterBase> bundleAdjuster() const { return bundle_adjuster_; }
109 void setBundleAdjuster(Ptr<detail::BundleAdjusterBase> bundle_adjuster)
110 { bundle_adjuster_ = bundle_adjuster; }
111
112 Ptr<WarperCreator> warper() { return warper_; }
113 const Ptr<WarperCreator> warper() const { return warper_; }
114 void setWarper(Ptr<WarperCreator> creator) { warper_ = creator; }
115
116 Ptr<detail::ExposureCompensator> exposureCompensator() { return exposure_comp_; }
117 const Ptr<detail::ExposureCompensator> exposureCompensator() const { return exposure_comp_; }
118 void setExposureCompensator(Ptr<detail::ExposureCompensator> exposure_comp)
119 { exposure_comp_ = exposure_comp; }
120
121 Ptr<detail::SeamFinder> seamFinder() { return seam_finder_; }
122 const Ptr<detail::SeamFinder> seamFinder() const { return seam_finder_; }
123 void setSeamFinder(Ptr<detail::SeamFinder> seam_finder) { seam_finder_ = seam_finder; }
124
125 Ptr<detail::Blender> blender() { return blender_; }
126 const Ptr<detail::Blender> blender() const { return blender_; }
127 void setBlender(Ptr<detail::Blender> b) { blender_ = b; }
128
129 Status estimateTransform(InputArray images);
130 Status estimateTransform(InputArray images, const std::vector<std::vector<Rect> > &rois);
131
132 Status composePanorama(OutputArray pano);
133 Status composePanorama(InputArray images, OutputArray pano);
134
135 Status stitch(InputArray images, OutputArray pano);
136 Status stitch(InputArray images, const std::vector<std::vector<Rect> > &rois, OutputArray pano);
137
138 std::vector<int> component() const { return indices_; }
139 std::vector<detail::CameraParams> cameras() const { return cameras_; }
140 double workScale() const { return work_scale_; }
141
142private:
143 Stitcher() {}
144
145 Status matchImages();
146 void estimateCameraParams();
147
148 double registr_resol_;
149 double seam_est_resol_;
150 double compose_resol_;
151 double conf_thresh_;
152 Ptr<detail::FeaturesFinder> features_finder_;
153 Ptr<detail::FeaturesMatcher> features_matcher_;
154 cv::Mat matching_mask_;
155 Ptr<detail::BundleAdjusterBase> bundle_adjuster_;
156 bool do_wave_correct_;
157 detail::WaveCorrectKind wave_correct_kind_;
158 Ptr<WarperCreator> warper_;
159 Ptr<detail::ExposureCompensator> exposure_comp_;
160 Ptr<detail::SeamFinder> seam_finder_;
161 Ptr<detail::Blender> blender_;
162
163 std::vector<cv::Mat> imgs_;
164 std::vector<std::vector<cv::Rect> > rois_;
165 std::vector<cv::Size> full_img_sizes_;
166 std::vector<detail::ImageFeatures> features_;
167 std::vector<detail::MatchesInfo> pairwise_matches_;
168 std::vector<cv::Mat> seam_est_imgs_;
169 std::vector<int> indices_;
170 std::vector<detail::CameraParams> cameras_;
171 double work_scale_;
172 double seam_scale_;
173 double seam_work_aspect_;
174 double warped_image_scale_;
175};
