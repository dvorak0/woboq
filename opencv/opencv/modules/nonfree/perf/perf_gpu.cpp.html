55//////////////////////////////////////////////////////////////////////
56// SURF
57
58DEF_PARAM_TEST_1(Image, string);
59
60PERF_TEST_P(Image, GPU_SURF,
61 Values<string>("gpu/perf/aloe.png"))
62{
63 declare.time(50.0);
64
65 const cv::Mat img = readImage(GetParam(), cv::IMREAD_GRAYSCALE);
66 ASSERT_FALSE(img.empty());
67
68 if (PERF_RUN_GPU())
69 {
70 cv::gpu::SURF_GPU d_surf;
71
72 const cv::gpu::GpuMat d_img(img);
73 cv::gpu::GpuMat d_keypoints, d_descriptors;
74
75 TEST_CYCLE() d_surf(d_img, cv::gpu::GpuMat(), d_keypoints, d_descriptors);
76
77 std::vector<cv::KeyPoint> gpu_keypoints;
78 d_surf.downloadKeypoints(d_keypoints, gpu_keypoints);
79
80 cv::Mat gpu_descriptors(d_descriptors);
81
82 sortKeyPoints(gpu_keypoints, gpu_descriptors);
83
84 SANITY_CHECK_KEYPOINTS(gpu_keypoints);
85 SANITY_CHECK(gpu_descriptors, 1e-3);
86 }
87 else
88 {
89 cv::SURF surf;
90
91 std::vector<cv::KeyPoint> cpu_keypoints;
92 cv::Mat cpu_descriptors;
93
94 TEST_CYCLE() surf(img, cv::noArray(), cpu_keypoints, cpu_descriptors);
95
96 SANITY_CHECK_KEYPOINTS(cpu_keypoints);
97 SANITY_CHECK(cpu_descriptors);
98 }
99}
