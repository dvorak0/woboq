50using namespace cv;

54typedef perf::TestBaseWithParam<std::string> OCL_SURF;
55
56#define SURF_IMAGES \
57 "cv/detectors_descriptors_evaluation/images_datasets/leuven/img1.png",\
58 "stitching/a3.png"
59
60#define OCL_TEST_CYCLE() for( ; startTimer(), next(); cv::ocl::finish(), stopTimer())
61
62PERF_TEST_P(OCL_SURF, DISABLED_with_data_transfer, testing::Values(SURF_IMAGES))
63{
64 string filename = getDataPath(GetParam());
65 Mat src = imread(filename, IMREAD_GRAYSCALE);
66 ASSERT_FALSE(src.empty());
67
68 Mat cpu_kp, cpu_dp;
69 declare.time(60);
70
71 if (getSelectedImpl() == "ocl")
72 {
73 SURF_OCL d_surf;
74 oclMat d_keypoints, d_descriptors;
75
76 OCL_TEST_CYCLE()
77 {
78 oclMat d_src(src);
79
80 d_surf(d_src, oclMat(), d_keypoints, d_descriptors);
81
82 d_keypoints.download(cpu_kp);
83 d_descriptors.download(cpu_dp);
84 }
85 }
86 else if (getSelectedImpl() == "plain")
87 {
88 cv::SURF surf;
89 std::vector<cv::KeyPoint> kp;
90
91 TEST_CYCLE() surf(src, Mat(), kp, cpu_dp);
92 }
93
94 SANITY_CHECK_NOTHING();
95}

126PERF_TEST_P(OCL_SURF, DISABLED_detect, testing::Values(SURF_IMAGES))
127{
128 String filename = getDataPath(GetParam());
129 Mat frame = imread(filename, IMREAD_GRAYSCALE);
130 ASSERT_FALSE(frame.empty()) << "Unable to load source image " << filename;
131
132 declare.in(frame);
133
134 Mat mask;
135 vector<KeyPoint> points;
136 Ptr<Feature2D> detector;
137
138 if (getSelectedImpl() == "plain")
139 {
140 detector = new SURF;
141 TEST_CYCLE() detector->operator()(frame, mask, points, noArray());
142 }
143 else if (getSelectedImpl() == "ocl")
144 {
145 detector = new ocl::SURF_OCL;
146 OCL_TEST_CYCLE() detector->operator()(frame, mask, points, noArray());
147 }
148 else CV_TEST_FAIL_NO_IMPL();
149
150 SANITY_CHECK_KEYPOINTS(points, 1e-3);
151}
