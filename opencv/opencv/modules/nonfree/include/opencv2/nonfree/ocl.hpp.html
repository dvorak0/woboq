44#ifndef __OPENCV_NONFREE_OCL_HPP__
45#define __OPENCV_NONFREE_OCL_HPP__
46
47#include "opencv2/ocl/ocl.hpp"
48
49namespace cv
50{
51 namespace ocl
52 {
53 //! Speeded up robust features, port from GPU module.
54 ////////////////////////////////// SURF //////////////////////////////////////////
55
56 class CV_EXPORTS SURF_OCL : public cv::Feature2D
57 {
58 public:
59 enum KeypointLayout
60 {
61 X_ROW = 0,
62 Y_ROW,
63 LAPLACIAN_ROW,
64 OCTAVE_ROW,
65 SIZE_ROW,
66 ANGLE_ROW,
67 HESSIAN_ROW,
68 ROWS_COUNT
69 };
70
71 //! the default constructor
72 SURF_OCL();
73 //! the full constructor taking all the necessary parameters
74 explicit SURF_OCL(double _hessianThreshold, int _nOctaves = 4,
75 int _nOctaveLayers = 2, bool _extended = true, float _keypointsRatio = 0.01f, bool _upright = false);
76
77 //! returns the descriptor size in float's (64 or 128)
78 int descriptorSize() const;
79
80 int descriptorType() const;
81
82 //! upload host keypoints to device memory
83 void uploadKeypoints(const vector<cv::KeyPoint> &keypoints, oclMat &keypointsocl);
84 //! download keypoints from device to host memory
85 void downloadKeypoints(const oclMat &keypointsocl, vector<KeyPoint> &keypoints);
86 //! download descriptors from device to host memory
87 void downloadDescriptors(const oclMat &descriptorsocl, vector<float> &descriptors);
88 //! finds the keypoints using fast hessian detector used in SURF
89 //! supports CV_8UC1 images
90 //! keypoints will have nFeature cols and 6 rows
91 //! keypoints.ptr<float>(X_ROW)[i] will contain x coordinate of i'th feature
92 //! keypoints.ptr<float>(Y_ROW)[i] will contain y coordinate of i'th feature
93 //! keypoints.ptr<float>(LAPLACIAN_ROW)[i] will contain laplacian sign of i'th feature
94 //! keypoints.ptr<float>(OCTAVE_ROW)[i] will contain octave of i'th feature
95 //! keypoints.ptr<float>(SIZE_ROW)[i] will contain size of i'th feature
96 //! keypoints.ptr<float>(ANGLE_ROW)[i] will contain orientation of i'th feature
97 //! keypoints.ptr<float>(HESSIAN_ROW)[i] will contain response of i'th feature
98 void operator()(const oclMat &img, const oclMat &mask, oclMat &keypoints);
99 //! finds the keypoints and computes their descriptors.
100 //! Optionally it can compute descriptors for the user-provided keypoints and recompute keypoints direction
101 void operator()(const oclMat &img, const oclMat &mask, oclMat &keypoints, oclMat &descriptors,
102 bool useProvidedKeypoints = false);
103 void operator()(const oclMat &img, const oclMat &mask, std::vector<KeyPoint> &keypoints);
104 void operator()(const oclMat &img, const oclMat &mask, std::vector<KeyPoint> &keypoints, oclMat &descriptors,
105 bool useProvidedKeypoints = false);
106 void operator()(const oclMat &img, const oclMat &mask, std::vector<KeyPoint> &keypoints, std::vector<float> &descriptors,
107 bool useProvidedKeypoints = false);
108
109 //! finds the keypoints using fast hessian detector used in SURF
110 void operator()(InputArray img, InputArray mask,
111 CV_OUT vector<KeyPoint>& keypoints) const;
112 //! finds the keypoints and computes their descriptors. Optionally it can compute descriptors for the user-provided keypoints
113 void operator()(InputArray img, InputArray mask,
114 CV_OUT vector<KeyPoint>& keypoints,
115 OutputArray descriptors,
116 bool useProvidedKeypoints=false) const;
117
118 AlgorithmInfo* info() const;
119
120 void releaseMemory();
121
122 // SURF parameters
123 float hessianThreshold;
124 int nOctaves;
125 int nOctaveLayers;
126 bool extended;
127 bool upright;
128 //! max keypoints = min(keypointsRatio * img.size().area(), 65535)
129 float keypointsRatio;
130 oclMat sum, mask1, maskSum, intBuffer;
131 oclMat det, trace;
132 oclMat maxPosBuffer;
133 protected:
134 void detectImpl( const Mat& image, vector<KeyPoint>& keypoints, const Mat& mask) const;
135 void computeImpl( const Mat& image, vector<KeyPoint>& keypoints, Mat& descriptors) const;
136 };
137 }
138}
139
140#endif //__OPENCV_NONFREE_OCL_HPP__
