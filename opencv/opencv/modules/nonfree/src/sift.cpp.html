114// default width of descriptor histogram array
115static const int SIFT_DESCR_WIDTH = 4;
116
117// default number of bins per histogram in descriptor array
118static const int SIFT_DESCR_HIST_BINS = 8;
119
120// assumed gaussian blur for input image
121static const float SIFT_INIT_SIGMA = 0.5f;
122
123// width of border in which to ignore keypoints
124static const int SIFT_IMG_BORDER = 5;
125
126// maximum steps of keypoint interpolation before failure
127static const int SIFT_MAX_INTERP_STEPS = 5;
128
129// default number of bins in histogram for orientation assignment
130static const int SIFT_ORI_HIST_BINS = 36;

132// determines gaussian sigma for orientation assignment
133static const float SIFT_ORI_SIG_FCTR = 1.5f;

135// determines the radius of the region used in orientation assignment
136static const float SIFT_ORI_RADIUS = 3 * SIFT_ORI_SIG_FCTR;

138// orientation magnitude relative to max that results in new feature
139static const float SIFT_ORI_PEAK_RATIO = 0.8f;

160static inline void
161unpackOctave(const KeyPoint& kpt, int& octave, int& layer, float& scale)
162{
163 octave = kpt.octave & 255;
164 layer = (kpt.octave >> 8) & 255;
165 octave = octave < 128 ? octave : (-128 | octave);
166 scale = octave >= 0 ? 1.f/(1 << octave) : (float)(1 << -octave);
167}

169static Mat createInitialImage( const Mat& img, bool doubleImageSize, float sigma )
170{
171 Mat gray, gray_fpt;
172 if( img.channels() == 3 || img.channels() == 4 )
173 {
174 cvtColor(img, gray, COLOR_BGR2GRAY);
175 gray.convertTo(gray_fpt, DataType<sift_wt>::type, SIFT_FIXPT_SCALE, 0);
176 }
177 else
178 img.convertTo(gray_fpt, DataType<sift_wt>::type, SIFT_FIXPT_SCALE, 0);
179
180 float sig_diff;
181
182 if( doubleImageSize )
183 {
184 sig_diff = sqrtf( std::max(sigma * sigma - SIFT_INIT_SIGMA * SIFT_INIT_SIGMA * 4, 0.01f) );
185 Mat dbl;
186 resize(gray_fpt, dbl, Size(gray_fpt.cols*2, gray_fpt.rows*2), 0, 0, INTER_LINEAR);
187 GaussianBlur(dbl, dbl, Size(), sig_diff, sig_diff);
188 return dbl;
189 }
190 else
191 {
192 sig_diff = sqrtf( std::max(sigma * sigma - SIFT_INIT_SIGMA * SIFT_INIT_SIGMA, 0.01f) );
193 GaussianBlur(gray_fpt, gray_fpt, Size(), sig_diff, sig_diff);
194 return gray_fpt;
195 }
196}

199void SIFT::buildGaussianPyramid( const Mat& base, vector<Mat>& pyr, int nOctaves ) const
200{
201 vector<double> sig(nOctaveLayers + 3);
202 pyr.resize(nOctaves*(nOctaveLayers + 3));
203
204 // precompute Gaussian sigmas using the following formula:
205 // \sigma_{total}^2 = \sigma_{i}^2 + \sigma_{i-1}^2
206 sig[0] = sigma;
207 double k = pow( 2., 1. / nOctaveLayers );
208 for( int i = 1; i < nOctaveLayers + 3; i++ )
209 {
210 double sig_prev = pow(k, (double)(i-1))*sigma;
211 double sig_total = sig_prev*k;
212 sig[i] = std::sqrt(sig_total*sig_total - sig_prev*sig_prev);
213 }
214
215 for( int o = 0; o < nOctaves; o++ )
216 {
217 for( int i = 0; i < nOctaveLayers + 3; i++ )
218 {
219 Mat& dst = pyr[o*(nOctaveLayers + 3) + i];
220 if( o == 0 && i == 0 )
221 dst = base;
222 // base of new octave is halved image from end of previous octave
223 else if( i == 0 )
224 {
225 const Mat& src = pyr[(o-1)*(nOctaveLayers + 3) + nOctaveLayers];
226 resize(src, dst, Size(src.cols/2, src.rows/2),
227 0, 0, INTER_NEAREST);
228 }
229 else
230 {
231 const Mat& src = pyr[o*(nOctaveLayers + 3) + i-1];
232 GaussianBlur(src, dst, Size(), sig[i], sig[i]);
233 }
234 }
235 }
236}

239void SIFT::buildDoGPyramid( const vector<Mat>& gpyr, vector<Mat>& dogpyr ) const
240{
241 int nOctaves = (int)gpyr.size()/(nOctaveLayers + 3);
242 dogpyr.resize( nOctaves*(nOctaveLayers + 2) );
243
244 for( int o = 0; o < nOctaves; o++ )
245 {
246 for( int i = 0; i < nOctaveLayers + 2; i++ )
247 {
248 const Mat& src1 = gpyr[o*(nOctaveLayers + 3) + i];
249 const Mat& src2 = gpyr[o*(nOctaveLayers + 3) + i + 1];
250 Mat& dst = dogpyr[o*(nOctaveLayers + 2) + i];
251 subtract(src2, src1, dst, noArray(), DataType<sift_wt>::type);
252 }
253 }
254}

257// Computes a gradient orientation histogram at a specified pixel
258static float calcOrientationHist( const Mat& img, Point pt, int radius,
259 float sigma, float* hist, int n )
260{
261 int i, j, k, len = (radius*2+1)*(radius*2+1);
262
263 float expf_scale = -1.f/(2.f * sigma * sigma);
264 AutoBuffer<float> buf(len*4 + n+4);
265 float *X = buf, *Y = X + len, *Mag = X, *Ori = Y + len, *W = Ori + len;
266 float* temphist = W + len + 2;
267
268 for( i = 0; i < n; i++ )
269 temphist[i] = 0.f;
270
271 for( i = -radius, k = 0; i <= radius; i++ )
272 {
273 int y = pt.y + i;
274 if( y <= 0 || y >= img.rows - 1 )
275 continue;
276 for( j = -radius; j <= radius; j++ )
277 {
278 int x = pt.x + j;
279 if( x <= 0 || x >= img.cols - 1 )
280 continue;
281
282 float dx = (float)(img.at<sift_wt>(y, x+1) - img.at<sift_wt>(y, x-1));
283 float dy = (float)(img.at<sift_wt>(y-1, x) - img.at<sift_wt>(y+1, x));
284
285 X[k] = dx; Y[k] = dy; W[k] = (i*i + j*j)*expf_scale;
286 k++;
287 }
288 }
289
290 len = k;
291
292 // compute gradient values, orientations and the weights over the pixel neighborhood
293 exp(W, W, len);
294 fastAtan2(Y, X, Ori, len, true);
295 magnitude(X, Y, Mag, len);
296
297 for( k = 0; k < len; k++ )
298 {
299 int bin = cvRound((n/360.f)*Ori[k]);
300 if( bin >= n )
301 bin -= n;
302 if( bin < 0 )
303 bin += n;
304 temphist[bin] += W[k]*Mag[k];
305 }
306
307 // smooth the histogram
308 temphist[-1] = temphist[n-1];
309 temphist[-2] = temphist[n-2];
310 temphist[n] = temphist[0];
311 temphist[n+1] = temphist[1];
312 for( i = 0; i < n; i++ )
313 {
314 hist[i] = (temphist[i-2] + temphist[i+2])*(1.f/16.f) +
315 (temphist[i-1] + temphist[i+1])*(4.f/16.f) +
316 temphist[i]*(6.f/16.f);
317 }
318
319 float maxval = hist[0];
320 for( i = 1; i < n; i++ )
321 maxval = std::max(maxval, hist[i]);
322
323 return maxval;
324}

327//
328// Interpolates a scale-space extremum's location and scale to subpixel
329// accuracy to form an image feature. Rejects features with low contrast.
330// Based on Section 4 of Lowe's paper.
331static bool adjustLocalExtrema( const vector<Mat>& dog_pyr, KeyPoint& kpt, int octv,
332 int& layer, int& r, int& c, int nOctaveLayers,
333 float contrastThreshold, float edgeThreshold, float sigma )
334{
335 const float img_scale = 1.f/(255*SIFT_FIXPT_SCALE);
336 const float deriv_scale = img_scale*0.5f;
337 const float second_deriv_scale = img_scale;
338 const float cross_deriv_scale = img_scale*0.25f;
339
340 float xi=0, xr=0, xc=0, contr=0;
341 int i = 0;
342
343 for( ; i < SIFT_MAX_INTERP_STEPS; i++ )
344 {
345 int idx = octv*(nOctaveLayers+2) + layer;
346 const Mat& img = dog_pyr[idx];
347 const Mat& prev = dog_pyr[idx-1];
348 const Mat& next = dog_pyr[idx+1];
349
350 Vec3f dD((img.at<sift_wt>(r, c+1) - img.at<sift_wt>(r, c-1))*deriv_scale,
351 (img.at<sift_wt>(r+1, c) - img.at<sift_wt>(r-1, c))*deriv_scale,
352 (next.at<sift_wt>(r, c) - prev.at<sift_wt>(r, c))*deriv_scale);
353
354 float v2 = (float)img.at<sift_wt>(r, c)*2;
355 float dxx = (img.at<sift_wt>(r, c+1) + img.at<sift_wt>(r, c-1) - v2)*second_deriv_scale;
356 float dyy = (img.at<sift_wt>(r+1, c) + img.at<sift_wt>(r-1, c) - v2)*second_deriv_scale;
357 float dss = (next.at<sift_wt>(r, c) + prev.at<sift_wt>(r, c) - v2)*second_deriv_scale;
358 float dxy = (img.at<sift_wt>(r+1, c+1) - img.at<sift_wt>(r+1, c-1) -
359 img.at<sift_wt>(r-1, c+1) + img.at<sift_wt>(r-1, c-1))*cross_deriv_scale;
360 float dxs = (next.at<sift_wt>(r, c+1) - next.at<sift_wt>(r, c-1) -
361 prev.at<sift_wt>(r, c+1) + prev.at<sift_wt>(r, c-1))*cross_deriv_scale;
362 float dys = (next.at<sift_wt>(r+1, c) - next.at<sift_wt>(r-1, c) -
363 prev.at<sift_wt>(r+1, c) + prev.at<sift_wt>(r-1, c))*cross_deriv_scale;
364
365 Matx33f H(dxx, dxy, dxs,
366 dxy, dyy, dys,
367 dxs, dys, dss);
368
369 Vec3f X = H.solve(dD, DECOMP_LU);
370
371 xi = -X[2];
372 xr = -X[1];
373 xc = -X[0];
374
375 if( std::abs(xi) < 0.5f && std::abs(xr) < 0.5f && std::abs(xc) < 0.5f )
376 break;
377
378 if( std::abs(xi) > (float)(INT_MAX/3) ||
379 std::abs(xr) > (float)(INT_MAX/3) ||
380 std::abs(xc) > (float)(INT_MAX/3) )
381 return false;
382
383 c += cvRound(xc);
384 r += cvRound(xr);
385 layer += cvRound(xi);
386
387 if( layer < 1 || layer > nOctaveLayers ||
388 c < SIFT_IMG_BORDER || c >= img.cols - SIFT_IMG_BORDER ||
389 r < SIFT_IMG_BORDER || r >= img.rows - SIFT_IMG_BORDER )
390 return false;
391 }
392
393 // ensure convergence of interpolation
394 if( i >= SIFT_MAX_INTERP_STEPS )
395 return false;
396
397 {
398 int idx = octv*(nOctaveLayers+2) + layer;
399 const Mat& img = dog_pyr[idx];
400 const Mat& prev = dog_pyr[idx-1];
401 const Mat& next = dog_pyr[idx+1];
402 Matx31f dD((img.at<sift_wt>(r, c+1) - img.at<sift_wt>(r, c-1))*deriv_scale,
403 (img.at<sift_wt>(r+1, c) - img.at<sift_wt>(r-1, c))*deriv_scale,
404 (next.at<sift_wt>(r, c) - prev.at<sift_wt>(r, c))*deriv_scale);
405 float t = dD.dot(Matx31f(xc, xr, xi));
406
407 contr = img.at<sift_wt>(r, c)*img_scale + t * 0.5f;
408 if( std::abs( contr ) * nOctaveLayers < contrastThreshold )
409 return false;
410
411 // principal curvatures are computed using the trace and det of Hessian
412 float v2 = img.at<sift_wt>(r, c)*2.f;
413 float dxx = (img.at<sift_wt>(r, c+1) + img.at<sift_wt>(r, c-1) - v2)*second_deriv_scale;
414 float dyy = (img.at<sift_wt>(r+1, c) + img.at<sift_wt>(r-1, c) - v2)*second_deriv_scale;
415 float dxy = (img.at<sift_wt>(r+1, c+1) - img.at<sift_wt>(r+1, c-1) -
416 img.at<sift_wt>(r-1, c+1) + img.at<sift_wt>(r-1, c-1)) * cross_deriv_scale;
417 float tr = dxx + dyy;
418 float det = dxx * dyy - dxy * dxy;
419
420 if( det <= 0 || tr*tr*edgeThreshold >= (edgeThreshold + 1)*(edgeThreshold + 1)*det )
421 return false;
422 }
423
424 kpt.pt.x = (c + xc) * (1 << octv);
425 kpt.pt.y = (r + xr) * (1 << octv);
426 kpt.octave = octv + (layer << 8) + (cvRound((xi + 0.5)*255) << 16);
427 kpt.size = sigma*powf(2.f, (layer + xi) / nOctaveLayers)*(1 << octv)*2;
428 kpt.response = std::abs(contr);
429
430 return true;
431}

434//
435// Detects features at extrema in DoG scale space. Bad features are discarded
436// based on contrast and ratio of principal curvatures.
437void SIFT::findScaleSpaceExtrema( const vector<Mat>& gauss_pyr, const vector<Mat>& dog_pyr,
438 vector<KeyPoint>& keypoints ) const
439{
440 int nOctaves = (int)gauss_pyr.size()/(nOctaveLayers + 3);
441 int threshold = cvFloor(0.5 * contrastThreshold / nOctaveLayers * 255 * SIFT_FIXPT_SCALE);
442 const int n = SIFT_ORI_HIST_BINS;
443 float hist[n];
444 KeyPoint kpt;
445
446 keypoints.clear();
447
448 for( int o = 0; o < nOctaves; o++ )
449 for( int i = 1; i <= nOctaveLayers; i++ )
450 {
451 int idx = o*(nOctaveLayers+2)+i;
452 const Mat& img = dog_pyr[idx];
453 const Mat& prev = dog_pyr[idx-1];
454 const Mat& next = dog_pyr[idx+1];
455 int step = (int)img.step1();
456 int rows = img.rows, cols = img.cols;
457
458 for( int r = SIFT_IMG_BORDER; r < rows-SIFT_IMG_BORDER; r++)
459 {
460 const sift_wt* currptr = img.ptr<sift_wt>(r);
461 const sift_wt* prevptr = prev.ptr<sift_wt>(r);
462 const sift_wt* nextptr = next.ptr<sift_wt>(r);
463
464 for( int c = SIFT_IMG_BORDER; c < cols-SIFT_IMG_BORDER; c++)
465 {
466 sift_wt val = currptr[c];
467
468 // find local extrema with pixel accuracy
469 if( std::abs(val) > threshold &&
470 ((val > 0 && val >= currptr[c-1] && val >= currptr[c+1] &&
471 val >= currptr[c-step-1] && val >= currptr[c-step] && val >= currptr[c-step+1] &&
472 val >= currptr[c+step-1] && val >= currptr[c+step] && val >= currptr[c+step+1] &&
473 val >= nextptr[c] && val >= nextptr[c-1] && val >= nextptr[c+1] &&
474 val >= nextptr[c-step-1] && val >= nextptr[c-step] && val >= nextptr[c-step+1] &&
475 val >= nextptr[c+step-1] && val >= nextptr[c+step] && val >= nextptr[c+step+1] &&
476 val >= prevptr[c] && val >= prevptr[c-1] && val >= prevptr[c+1] &&
477 val >= prevptr[c-step-1] && val >= prevptr[c-step] && val >= prevptr[c-step+1] &&
478 val >= prevptr[c+step-1] && val >= prevptr[c+step] && val >= prevptr[c+step+1]) ||
479 (val < 0 && val <= currptr[c-1] && val <= currptr[c+1] &&
480 val <= currptr[c-step-1] && val <= currptr[c-step] && val <= currptr[c-step+1] &&
481 val <= currptr[c+step-1] && val <= currptr[c+step] && val <= currptr[c+step+1] &&
482 val <= nextptr[c] && val <= nextptr[c-1] && val <= nextptr[c+1] &&
483 val <= nextptr[c-step-1] && val <= nextptr[c-step] && val <= nextptr[c-step+1] &&
484 val <= nextptr[c+step-1] && val <= nextptr[c+step] && val <= nextptr[c+step+1] &&
485 val <= prevptr[c] && val <= prevptr[c-1] && val <= prevptr[c+1] &&
486 val <= prevptr[c-step-1] && val <= prevptr[c-step] && val <= prevptr[c-step+1] &&
487 val <= prevptr[c+step-1] && val <= prevptr[c+step] && val <= prevptr[c+step+1])))
488 {
489 int r1 = r, c1 = c, layer = i;
490 if( !adjustLocalExtrema(dog_pyr, kpt, o, layer, r1, c1,
491 nOctaveLayers, (float)contrastThreshold,
492 (float)edgeThreshold, (float)sigma) )
493 continue;
494 float scl_octv = kpt.size*0.5f/(1 << o);
495 float omax = calcOrientationHist(gauss_pyr[o*(nOctaveLayers+3) + layer],
496 Point(c1, r1),
497 cvRound(SIFT_ORI_RADIUS * scl_octv),
498 SIFT_ORI_SIG_FCTR * scl_octv,
499 hist, n);
500 float mag_thr = (float)(omax * SIFT_ORI_PEAK_RATIO);
501 for( int j = 0; j < n; j++ )
502 {
503 int l = j > 0 ? j - 1 : n - 1;
504 int r2 = j < n-1 ? j + 1 : 0;
505
506 if( hist[j] > hist[l] && hist[j] > hist[r2] && hist[j] >= mag_thr )
507 {
508 float bin = j + 0.5f * (hist[l]-hist[r2]) / (hist[l] - 2*hist[j] + hist[r2]);
509 bin = bin < 0 ? n + bin : bin >= n ? bin - n : bin;
510 kpt.angle = 360.f - (float)((360.f/n) * bin);
511 if(std::abs(kpt.angle - 360.f) < FLT_EPSILON)
512 kpt.angle = 0.f;
513 keypoints.push_back(kpt);
514 }
515 }
516 }
517 }
518 }
519 }

523static void calcSIFTDescriptor( const Mat& img, Point2f ptf, float ori, float scl,
524 int d, int n, float* dst )
525{
526 Point pt(cvRound(ptf.x), cvRound(ptf.y));
527 float cos_t = cosf(ori*(float)(CV_PI/180));
528 float sin_t = sinf(ori*(float)(CV_PI/180));
529 float bins_per_rad = n / 360.f;
530 float exp_scale = -1.f/(d * d * 0.5f);
531 float hist_width = SIFT_DESCR_SCL_FCTR * scl;
532 int radius = cvRound(hist_width * 1.4142135623730951f * (d + 1) * 0.5f);
533 // Clip the radius to the diagonal of the image to avoid autobuffer too large exception
534 radius = std::min(radius, (int) sqrt((double) img.cols*img.cols + img.rows*img.rows));
535 cos_t /= hist_width;
536 sin_t /= hist_width;
537
538 int i, j, k, len = (radius*2+1)*(radius*2+1), histlen = (d+2)*(d+2)*(n+2);
539 int rows = img.rows, cols = img.cols;
540
541 AutoBuffer<float> buf(len*6 + histlen);
542 float *X = buf, *Y = X + len, *Mag = Y, *Ori = Mag + len, *W = Ori + len;
543 float *RBin = W + len, *CBin = RBin + len, *hist = CBin + len;
544
545 for( i = 0; i < d+2; i++ )
546 {
547 for( j = 0; j < d+2; j++ )
548 for( k = 0; k < n+2; k++ )
549 hist[(i*(d+2) + j)*(n+2) + k] = 0.;
550 }
551
552 for( i = -radius, k = 0; i <= radius; i++ )
553 for( j = -radius; j <= radius; j++ )
554 {
555 // Calculate sample's histogram array coords rotated relative to ori.
556 // Subtract 0.5 so samples that fall e.g. in the center of row 1 (i.e.
557 // r_rot = 1.5) have full weight placed in row 1 after interpolation.
558 float c_rot = j * cos_t - i * sin_t;
559 float r_rot = j * sin_t + i * cos_t;
560 float rbin = r_rot + d/2 - 0.5f;
561 float cbin = c_rot + d/2 - 0.5f;
562 int r = pt.y + i, c = pt.x + j;
563
564 if( rbin > -1 && rbin < d && cbin > -1 && cbin < d &&
565 r > 0 && r < rows - 1 && c > 0 && c < cols - 1 )
566 {
567 float dx = (float)(img.at<sift_wt>(r, c+1) - img.at<sift_wt>(r, c-1));
568 float dy = (float)(img.at<sift_wt>(r-1, c) - img.at<sift_wt>(r+1, c));
569 X[k] = dx; Y[k] = dy; RBin[k] = rbin; CBin[k] = cbin;
570 W[k] = (c_rot * c_rot + r_rot * r_rot)*exp_scale;
571 k++;
572 }
573 }
574
575 len = k;
576 fastAtan2(Y, X, Ori, len, true);
577 magnitude(X, Y, Mag, len);
578 exp(W, W, len);
579
580 for( k = 0; k < len; k++ )
581 {
582 float rbin = RBin[k], cbin = CBin[k];
583 float obin = (Ori[k] - ori)*bins_per_rad;
584 float mag = Mag[k]*W[k];
585
586 int r0 = cvFloor( rbin );
587 int c0 = cvFloor( cbin );
588 int o0 = cvFloor( obin );
589 rbin -= r0;
590 cbin -= c0;
591 obin -= o0;
592
593 if( o0 < 0 )
594 o0 += n;
595 if( o0 >= n )
596 o0 -= n;
597
598 // histogram update using tri-linear interpolation
599 float v_r1 = mag*rbin, v_r0 = mag - v_r1;
600 float v_rc11 = v_r1*cbin, v_rc10 = v_r1 - v_rc11;
601 float v_rc01 = v_r0*cbin, v_rc00 = v_r0 - v_rc01;
602 float v_rco111 = v_rc11*obin, v_rco110 = v_rc11 - v_rco111;
603 float v_rco101 = v_rc10*obin, v_rco100 = v_rc10 - v_rco101;
604 float v_rco011 = v_rc01*obin, v_rco010 = v_rc01 - v_rco011;
605 float v_rco001 = v_rc00*obin, v_rco000 = v_rc00 - v_rco001;
606
607 int idx = ((r0+1)*(d+2) + c0+1)*(n+2) + o0;
608 hist[idx] += v_rco000;
609 hist[idx+1] += v_rco001;
610 hist[idx+(n+2)] += v_rco010;
611 hist[idx+(n+3)] += v_rco011;
612 hist[idx+(d+2)*(n+2)] += v_rco100;
613 hist[idx+(d+2)*(n+2)+1] += v_rco101;
614 hist[idx+(d+3)*(n+2)] += v_rco110;
615 hist[idx+(d+3)*(n+2)+1] += v_rco111;
616 }
617
618 // finalize histogram, since the orientation histograms are circular
619 for( i = 0; i < d; i++ )
620 for( j = 0; j < d; j++ )
621 {
622 int idx = ((i+1)*(d+2) + (j+1))*(n+2);
623 hist[idx] += hist[idx+n];
624 hist[idx+1] += hist[idx+n+1];
625 for( k = 0; k < n; k++ )
626 dst[(i*d + j)*n + k] = hist[idx+k];
627 }
628 // copy histogram to the descriptor,
629 // apply hysteresis thresholding
630 // and scale the result, so that it can be easily converted
631 // to byte array
632 float nrm2 = 0;
633 len = d*d*n;
634 for( k = 0; k < len; k++ )
635 nrm2 += dst[k]*dst[k];
636 float thr = std::sqrt(nrm2)*SIFT_DESCR_MAG_THR;
637 for( i = 0, nrm2 = 0; i < k; i++ )
638 {
639 float val = std::min(dst[i], thr);
640 dst[i] = val;
641 nrm2 += val*val;
642 }
643 nrm2 = SIFT_INT_DESCR_FCTR/std::max(std::sqrt(nrm2), FLT_EPSILON);
644
645#if 1
646 for( k = 0; k < len; k++ )
647 {
648 dst[k] = saturate_cast<uchar>(dst[k]*nrm2);
649 }
650#else
651 float nrm1 = 0;
652 for( k = 0; k < len; k++ )
653 {
654 dst[k] *= nrm2;
655 nrm1 += dst[k];
656 }
657 nrm1 = 1.f/std::max(nrm1, FLT_EPSILON);
658 for( k = 0; k < len; k++ )
659 {
660 dst[k] = std::sqrt(dst[k] * nrm1);//saturate_cast<uchar>(std::sqrt(dst[k] * nrm1)*SIFT_INT_DESCR_FCTR);
661 }
662#endif
663}

665static void calcDescriptors(const vector<Mat>& gpyr, const vector<KeyPoint>& keypoints,
666 Mat& descriptors, int nOctaveLayers, int firstOctave )
667{
668 int d = SIFT_DESCR_WIDTH, n = SIFT_DESCR_HIST_BINS;
669
670 for( size_t i = 0; i < keypoints.size(); i++ )
671 {
672 KeyPoint kpt = keypoints[i];
673 int octave, layer;
674 float scale;
675 unpackOctave(kpt, octave, layer, scale);
676 CV_Assert(octave >= firstOctave && layer <= nOctaveLayers+2);
677 float size=kpt.size*scale;
678 Point2f ptf(kpt.pt.x*scale, kpt.pt.y*scale);
679 const Mat& img = gpyr[(octave - firstOctave)*(nOctaveLayers + 3) + layer];
680
681 float angle = 360.f - kpt.angle;
682 if(std::abs(angle - 360.f) < FLT_EPSILON)
683 angle = 0.f;
684 calcSIFTDescriptor(img, ptf, angle, size*0.5f, d, n, descriptors.ptr<float>((int)i));
685 }
686}

715void SIFT::operator()(InputArray _image, InputArray _mask,
716 vector<KeyPoint>& keypoints,
717 OutputArray _descriptors,
718 bool useProvidedKeypoints) const
719{
720 int firstOctave = -1, actualNOctaves = 0, actualNLayers = 0;
721 Mat image = _image.getMat(), mask = _mask.getMat();
722
723 if( image.empty() || image.depth() != CV_8U )
724 CV_Error( CV_StsBadArg, "image is empty or has incorrect depth (!=CV_8U)" );
725
726 if( !mask.empty() && mask.type() != CV_8UC1 )
727 CV_Error( CV_StsBadArg, "mask has incorrect type (!=CV_8UC1)" );
728
729 if( useProvidedKeypoints )
730 {
731 firstOctave = 0;
732 int maxOctave = INT_MIN;
733 for( size_t i = 0; i < keypoints.size(); i++ )
734 {
735 int octave, layer;
736 float scale;
737 unpackOctave(keypoints[i], octave, layer, scale);
738 firstOctave = std::min(firstOctave, octave);
739 maxOctave = std::max(maxOctave, octave);
740 actualNLayers = std::max(actualNLayers, layer-2);
741 }
742
743 firstOctave = std::min(firstOctave, 0);
744 CV_Assert( firstOctave >= -1 && actualNLayers <= nOctaveLayers );
745 actualNOctaves = maxOctave - firstOctave + 1;
746 }
747
748 Mat base = createInitialImage(image, firstOctave < 0, (float)sigma);
749 vector<Mat> gpyr, dogpyr;
750 int nOctaves = actualNOctaves > 0 ? actualNOctaves : cvRound(log( (double)std::min( base.cols, base.rows ) ) / log(2.) - 2) - firstOctave;
751
752 //double t, tf = getTickFrequency();
753 //t = (double)getTickCount();
754 buildGaussianPyramid(base, gpyr, nOctaves);
755 buildDoGPyramid(gpyr, dogpyr);
756
757 //t = (double)getTickCount() - t;
758 //printf("pyramid construction time: %g\n", t*1000./tf);
759
760 if( !useProvidedKeypoints )
761 {
762 //t = (double)getTickCount();
763 findScaleSpaceExtrema(gpyr, dogpyr, keypoints);
764 KeyPointsFilter::removeDuplicated( keypoints );
765
766 if( nfeatures > 0 )
767 KeyPointsFilter::retainBest(keypoints, nfeatures);
768 //t = (double)getTickCount() - t;
769 //printf("keypoint detection time: %g\n", t*1000./tf);
770
771 if( firstOctave < 0 )
772 for( size_t i = 0; i < keypoints.size(); i++ )
773 {
774 KeyPoint& kpt = keypoints[i];
775 float scale = 1.f/(float)(1 << -firstOctave);
776 kpt.octave = (kpt.octave & ~255) | ((kpt.octave + firstOctave) & 255);
777 kpt.pt *= scale;
778 kpt.size *= scale;
779 }
780
781 if( !mask.empty() )
782 KeyPointsFilter::runByPixelsMask( keypoints, mask );
783 }
784 else
785 {
786 // filter keypoints by mask
787 //KeyPointsFilter::runByPixelsMask( keypoints, mask );
788 }
789
790 if( _descriptors.needed() )
791 {
792 //t = (double)getTickCount();
793 int dsize = descriptorSize();
794 _descriptors.create((int)keypoints.size(), dsize, CV_32F);
795 Mat descriptors = _descriptors.getMat();
796
797 calcDescriptors(gpyr, keypoints, descriptors, nOctaveLayers, firstOctave);
798 //t = (double)getTickCount() - t;
799 //printf("descriptor extraction time: %g\n", t*1000./tf);
800 }
801}
