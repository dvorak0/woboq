59 // The number of degrees between orientation samples in calcOrientation
60 const static int ORI_SEARCH_INC = 5;
61 // The local size of the calcOrientation kernel
62 const static int ORI_LOCAL_SIZE = (360 / ORI_SEARCH_INC);

64 static void openCLExecuteKernelSURF(Context *clCxt, const cv::ocl::ProgramEntry* source, string kernelName, size_t globalThreads[3],
65 size_t localThreads[3], std::vector< std::pair<size_t, const void *> > &args, int channels, int depth)
66 {
67 std::stringstream optsStr;
68 optsStr << "-D ORI_LOCAL_SIZE=" << ORI_LOCAL_SIZE << " ";
69 optsStr << "-D ORI_SEARCH_INC=" << ORI_SEARCH_INC << " ";
70 cl_kernel kernel;
71 kernel = openCLGetKernelFromSource(clCxt, source, kernelName, optsStr.str().c_str());
72 size_t wave_size = queryWaveFrontSize(kernel);
73 CV_Assert(clReleaseKernel(kernel) == CL_SUCCESS);
74 optsStr << "-D WAVE_SIZE=" << wave_size;
75 openCLExecuteKernel(clCxt, source, kernelName, globalThreads, localThreads, args, channels, depth, optsStr.str().c_str());
76 }
77
78 }

81static inline int calcSize(int octave, int layer)
82{
83 /* Wavelet size at first layer of first octave. */
84 const int HAAR_SIZE0 = 9;
85
86 /* Wavelet size increment between layers. This should be an even number,
87 such that the wavelet sizes in an octave are either all even or all odd.
88 This ensures that when looking for the neighbors of a sample, the layers
89
90 above and below are aligned correctly. */
91 const int HAAR_SIZE_INC = 6;
92
93 return (HAAR_SIZE0 + HAAR_SIZE_INC * layer) << octave;
94}

171 void detectKeypoints(oclMat &keypoints)
172 {
173 // create image pyramid buffers
174 // different layers have same sized buffers, but they are sampled from Gaussian kernel.
175 ensureSizeIsEnough(img_rows * (surf_.nOctaveLayers + 2), img_cols, CV_32FC1, surf_.det);
176 ensureSizeIsEnough(img_rows * (surf_.nOctaveLayers + 2), img_cols, CV_32FC1, surf_.trace);
177
178 ensureSizeIsEnough(1, maxCandidates, CV_32SC4, surf_.maxPosBuffer);
179 ensureSizeIsEnough(SURF_OCL::ROWS_COUNT, maxFeatures, CV_32FC1, keypoints);
180 keypoints.setTo(Scalar::all(0));
181
182 for (int octave = 0; octave < surf_.nOctaves; ++octave)
183 {
184 const int layer_rows = img_rows >> octave;
185 const int layer_cols = img_cols >> octave;
186
187 //loadOctaveConstants(octave, layer_rows, layer_cols);
188
189 icvCalcLayerDetAndTrace_gpu(surf_.det, surf_.trace, octave, surf_.nOctaveLayers, layer_rows);
190
191 icvFindMaximaInLayer_gpu(surf_.det, surf_.trace, surf_.maxPosBuffer, counters, 1 + octave,
192 octave, use_mask, surf_.nOctaveLayers, layer_rows, layer_cols);
193
194 int maxCounter = ((Mat)counters).at<int>(1 + octave);
195 maxCounter = std::min(maxCounter, static_cast<int>(maxCandidates));
196
197 if (maxCounter > 0)
198 {
199 icvInterpolateKeypoint_gpu(surf_.det, surf_.maxPosBuffer, maxCounter,
200 keypoints, counters, octave, layer_rows, maxFeatures);
201 }
202 }
203 int featureCounter = Mat(counters).at<int>(0);
204 featureCounter = std::min(featureCounter, static_cast<int>(maxFeatures));
205
206 keypoints.cols = featureCounter;
207
208 if (surf_.upright)
209 {
210 //keypoints.row(SURF_OCL::ANGLE_ROW).setTo(Scalar::all(90.0));
211 setUpright(keypoints);
212 }
213 else
214 {
215 findOrientation(keypoints);
216 }
217 }

225 }
226 }

234 }
235 }

237 void computeDescriptors(const oclMat &keypoints, oclMat &descriptors, int descriptorSize)

242 ensureSizeIsEnough(nFeatures, descriptorSize, CV_32F, descriptors);
243 compute_descriptors_gpu(descriptors, keypoints, nFeatures);
244 }
245 }

309void cv::ocl::SURF_OCL::uploadKeypoints(const vector<KeyPoint> &keypoints, oclMat &keypointsGPU)
310{
311 if (keypoints.empty())
312 keypointsGPU.release();
313 else
314 {
315 Mat keypointsCPU(SURF_OCL::ROWS_COUNT, static_cast<int>(keypoints.size()), CV_32FC1);
316
317 float *kp_x = keypointsCPU.ptr<float>(SURF_OCL::X_ROW);
318 float *kp_y = keypointsCPU.ptr<float>(SURF_OCL::Y_ROW);
319 int *kp_laplacian = keypointsCPU.ptr<int>(SURF_OCL::LAPLACIAN_ROW);
320 int *kp_octave = keypointsCPU.ptr<int>(SURF_OCL::OCTAVE_ROW);
321 float *kp_size = keypointsCPU.ptr<float>(SURF_OCL::SIZE_ROW);
322 float *kp_dir = keypointsCPU.ptr<float>(SURF_OCL::ANGLE_ROW);
323 float *kp_hessian = keypointsCPU.ptr<float>(SURF_OCL::HESSIAN_ROW);
324
325 for (size_t i = 0, size = keypoints.size(); i < size; ++i)
326 {
327 const KeyPoint &kp = keypoints[i];
328 kp_x[i] = kp.pt.x;
329 kp_y[i] = kp.pt.y;
330 kp_octave[i] = kp.octave;
331 kp_size[i] = kp.size;
332 kp_dir[i] = kp.angle;
333 kp_hessian[i] = kp.response;
334 kp_laplacian[i] = 1;
335 }
336
337 keypointsGPU.upload(keypointsCPU);
338 }
339}

341void cv::ocl::SURF_OCL::downloadKeypoints(const oclMat &keypointsGPU, vector<KeyPoint> &keypoints)
342{
343 const int nFeatures = keypointsGPU.cols;
344
345 if (nFeatures == 0)
346 keypoints.clear();
347 else
348 {
349 CV_Assert(keypointsGPU.type() == CV_32FC1 && keypointsGPU.rows == ROWS_COUNT);
350
351 Mat keypointsCPU(keypointsGPU);
352
353 keypoints.resize(nFeatures);
354
355 float *kp_x = keypointsCPU.ptr<float>(SURF_OCL::X_ROW);
356 float *kp_y = keypointsCPU.ptr<float>(SURF_OCL::Y_ROW);
357 int *kp_laplacian = keypointsCPU.ptr<int>(SURF_OCL::LAPLACIAN_ROW);
358 int *kp_octave = keypointsCPU.ptr<int>(SURF_OCL::OCTAVE_ROW);
359 float *kp_size = keypointsCPU.ptr<float>(SURF_OCL::SIZE_ROW);
360 float *kp_dir = keypointsCPU.ptr<float>(SURF_OCL::ANGLE_ROW);
361 float *kp_hessian = keypointsCPU.ptr<float>(SURF_OCL::HESSIAN_ROW);
362
363 for (int i = 0; i < nFeatures; ++i)
364 {
365 KeyPoint &kp = keypoints[i];
366 kp.pt.x = kp_x[i];
367 kp.pt.y = kp_y[i];
368 kp.class_id = kp_laplacian[i];
369 kp.octave = kp_octave[i];
370 kp.size = kp_size[i];
371 kp.angle = kp_dir[i];
372 kp.response = kp_hessian[i];
373 }
374 }
375}

458void cv::ocl::SURF_OCL::operator()(InputArray img, InputArray mask, vector<KeyPoint> &keypoints,
459 OutputArray descriptors, bool useProvidedKeypoints) const
460{
461 oclMat _img, _mask;
462 if(img.kind() == _InputArray::OCL_MAT)
463 _img = *(oclMat*)img.obj;
464 else
465 _img.upload(img.getMat());
466 if(_img.channels() != 1)
467 {
468 oclMat temp;
469 cvtColor(_img, temp, COLOR_BGR2GRAY);
470 _img = temp;
471 }
472
473 if( !mask.empty() )
474 {
475 if(mask.kind() == _InputArray::OCL_MAT)
476 _mask = *(oclMat*)mask.obj;
477 else
478 _mask.upload(mask.getMat());
479 }
480
481 SURF_OCL_Invoker theSurf((SURF_OCL&)*this, _img, _mask);
482 oclMat keypointsGPU;
483
484 if (!useProvidedKeypoints || !upright)
485 ((SURF_OCL*)this)->uploadKeypoints(keypoints, keypointsGPU);
486
487 if (!useProvidedKeypoints)
488 theSurf.detectKeypoints(keypointsGPU);
489 else if (!upright)
490 theSurf.findOrientation(keypointsGPU);
491 if(keypointsGPU.cols*keypointsGPU.rows != 0)
492 ((SURF_OCL*)this)->downloadKeypoints(keypointsGPU, keypoints);
493
494 if( descriptors.needed() )
495 {
496 oclMat descriptorsGPU;
497 theSurf.computeDescriptors(keypointsGPU, descriptorsGPU, descriptorSize());
498 Size sz = descriptorsGPU.size();
499 if( descriptors.kind() == _InputArray::STD_VECTOR )
500 {
501 CV_Assert(descriptors.type() == CV_32F);
502 std::vector<float>* v = (std::vector<float>*)descriptors.obj;
503 v->resize(sz.width*sz.height);
504 Mat m(sz, CV_32F, &v->at(0));
505 descriptorsGPU.download(m);
506 }
507 else
508 {
509 descriptors.create(sz, CV_32F);
510 Mat m = descriptors.getMat();
511 descriptorsGPU.download(m);
512 }
513 }
514}

550void SURF_OCL_Invoker::icvCalcLayerDetAndTrace_gpu(oclMat &det, oclMat &trace, int octave, int nOctaveLayers, int c_layer_rows)
551{
552 const int min_size = calcSize(octave, 0);
553 const int max_samples_i = 1 + ((img_rows - min_size) >> octave);
554 const int max_samples_j = 1 + ((img_cols - min_size) >> octave);
555
556 Context *clCxt = det.clCxt;
557 string kernelName = "icvCalcLayerDetAndTrace";
558 std::vector< std::pair<size_t, const void *> > args;
559
560 if(sumTex)
561 {
562 args.push_back( std::make_pair( sizeof(cl_mem), (void *)&sumTex));
563 }
564 else
565 {
566 args.push_back( std::make_pair( sizeof(cl_mem), (void *)&surf_.sum.data)); // if image2d is not supported
567 }
568 args.push_back( std::make_pair( sizeof(cl_mem), (void *)&det.data));
569 args.push_back( std::make_pair( sizeof(cl_mem), (void *)&trace.data));
570 args.push_back( std::make_pair( sizeof(cl_int), (void *)&det.step));
571 args.push_back( std::make_pair( sizeof(cl_int), (void *)&trace.step));
572 args.push_back( std::make_pair( sizeof(cl_int), (void *)&img_rows));
573 args.push_back( std::make_pair( sizeof(cl_int), (void *)&img_cols));
574 args.push_back( std::make_pair( sizeof(cl_int), (void *)&nOctaveLayers));
575 args.push_back( std::make_pair( sizeof(cl_int), (void *)&octave));
576 args.push_back( std::make_pair( sizeof(cl_int), (void *)&c_layer_rows));
577 args.push_back( std::make_pair( sizeof(cl_int), (void *)&surf_.sum.step));
578
579 size_t localThreads[3] = {16, 16, 1};
580 size_t globalThreads[3] =
581 {
582 divUp(max_samples_j, localThreads[0]) * localThreads[0],
583 divUp(max_samples_i, localThreads[1]) * localThreads[1] *(nOctaveLayers + 2),
584 1
585 };
586 openCLExecuteKernelSURF(clCxt, &surf, kernelName, globalThreads, localThreads, args, -1, -1);
587}

589void SURF_OCL_Invoker::icvFindMaximaInLayer_gpu(const oclMat &det, const oclMat &trace, oclMat &maxPosBuffer, oclMat &maxCounter, int counterOffset,
590 int octave, bool useMask, int nLayers, int layer_rows, int layer_cols)
591{
592 const int min_margin = ((calcSize(octave, 2) >> 1) >> octave) + 1;
593
594 Context *clCxt = det.clCxt;
595 string kernelName = useMask ? "icvFindMaximaInLayer_withmask" : "icvFindMaximaInLayer";
596 std::vector< std::pair<size_t, const void *> > args;
597
598 args.push_back( std::make_pair( sizeof(cl_mem), (void *)&det.data));
599 args.push_back( std::make_pair( sizeof(cl_mem), (void *)&trace.data));
600 args.push_back( std::make_pair( sizeof(cl_mem), (void *)&maxPosBuffer.data));
601 args.push_back( std::make_pair( sizeof(cl_mem), (void *)&maxCounter.data));
602 args.push_back( std::make_pair( sizeof(cl_int), (void *)&counterOffset));
603 args.push_back( std::make_pair( sizeof(cl_int), (void *)&det.step));
604 args.push_back( std::make_pair( sizeof(cl_int), (void *)&trace.step));
605 args.push_back( std::make_pair( sizeof(cl_int), (void *)&img_rows));
606 args.push_back( std::make_pair( sizeof(cl_int), (void *)&img_cols));
607 args.push_back( std::make_pair( sizeof(cl_int), (void *)&nLayers));
608 args.push_back( std::make_pair( sizeof(cl_int), (void *)&octave));
609 args.push_back( std::make_pair( sizeof(cl_int), (void *)&layer_rows));
610 args.push_back( std::make_pair( sizeof(cl_int), (void *)&layer_cols));
611 args.push_back( std::make_pair( sizeof(cl_int), (void *)&maxCandidates));
612 args.push_back( std::make_pair( sizeof(cl_float), (void *)&surf_.hessianThreshold));
613
614 if(useMask)
615 {
616 if(maskSumTex)
617 {
618 args.push_back( std::make_pair( sizeof(cl_mem), (void *)&maskSumTex));
619 }

705void SURF_OCL_Invoker::compute_descriptors_gpu(const oclMat &descriptors, const oclMat &keypoints, int nFeatures)
706{
707 // compute unnormalized descriptors, then normalize them - odd indexing since grid must be 2D
708 Context *clCxt = descriptors.clCxt;
709 string kernelName;
710 std::vector< std::pair<size_t, const void *> > args;
711 size_t localThreads[3] = {1, 1, 1};
712 size_t globalThreads[3] = {1, 1, 1};
713
714 if(descriptors.cols == 64)
715 {
716 kernelName = "compute_descriptors64";
717
718 localThreads[0] = 6;
719 localThreads[1] = 6;
720
721 globalThreads[0] = nFeatures * localThreads[0];
722 globalThreads[1] = 16 * localThreads[1];
723
724 args.clear();
725 if(imgTex)
726 {
727 args.push_back( std::make_pair( sizeof(cl_mem), (void *)&imgTex));
728 }
729 else
730 {
731 args.push_back( std::make_pair( sizeof(cl_mem), (void *)&_img.data));
732 }
733 args.push_back( std::make_pair( sizeof(cl_mem), (void *)&descriptors.data));
734 args.push_back( std::make_pair( sizeof(cl_mem), (void *)&keypoints.data));
735 args.push_back( std::make_pair( sizeof(cl_int), (void *)&descriptors.step));
736 args.push_back( std::make_pair( sizeof(cl_int), (void *)&keypoints.step));
737 args.push_back( std::make_pair( sizeof(cl_int), (void *)&_img.rows));
738 args.push_back( std::make_pair( sizeof(cl_int), (void *)&_img.cols));
739 args.push_back( std::make_pair( sizeof(cl_int), (void *)&_img.step));
740
741 openCLExecuteKernelSURF(clCxt, &surf, kernelName, globalThreads, localThreads, args, -1, -1);
742
743 kernelName = "normalize_descriptors64";
744
745 localThreads[0] = 64;
746 localThreads[1] = 1;
747
748 globalThreads[0] = nFeatures * localThreads[0];
749 globalThreads[1] = localThreads[1];
750
751 args.clear();
752 args.push_back( std::make_pair( sizeof(cl_mem), (void *)&descriptors.data));
753 args.push_back( std::make_pair( sizeof(cl_int), (void *)&descriptors.step));
754
755 openCLExecuteKernelSURF(clCxt, &surf, kernelName, globalThreads, localThreads, args, -1, -1);
756 }
757 else
758 {
759 kernelName = "compute_descriptors128";
760
761 localThreads[0] = 6;
762 localThreads[1] = 6;
763
764 globalThreads[0] = nFeatures * localThreads[0];
765 globalThreads[1] = 16 * localThreads[1];
766
767 args.clear();
768 if(imgTex)
769 {
770 args.push_back( std::make_pair( sizeof(cl_mem), (void *)&imgTex));
771 }
772 else
773 {
774 args.push_back( std::make_pair( sizeof(cl_mem), (void *)&_img.data));
775 }
776 args.push_back( std::make_pair( sizeof(cl_mem), (void *)&descriptors.data));
777 args.push_back( std::make_pair( sizeof(cl_mem), (void *)&keypoints.data));
778 args.push_back( std::make_pair( sizeof(cl_int), (void *)&descriptors.step));
779 args.push_back( std::make_pair( sizeof(cl_int), (void *)&keypoints.step));
780 args.push_back( std::make_pair( sizeof(cl_int), (void *)&_img.rows));
781 args.push_back( std::make_pair( sizeof(cl_int), (void *)&_img.cols));
782 args.push_back( std::make_pair( sizeof(cl_int), (void *)&_img.step));
783
784 openCLExecuteKernelSURF(clCxt, &surf, kernelName, globalThreads, localThreads, args, -1, -1);
785
786 kernelName = "normalize_descriptors128";
787
788 localThreads[0] = 128;
789 localThreads[1] = 1;
790
791 globalThreads[0] = nFeatures * localThreads[0];
792 globalThreads[1] = localThreads[1];
793
794 args.clear();
795 args.push_back( std::make_pair( sizeof(cl_mem), (void *)&descriptors.data));
796 args.push_back( std::make_pair( sizeof(cl_int), (void *)&descriptors.step));
797
798 openCLExecuteKernelSURF(clCxt, &surf, kernelName, globalThreads, localThreads, args, -1, -1);
799 }
