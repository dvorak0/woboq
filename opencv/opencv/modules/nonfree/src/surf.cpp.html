46/*
47 The following changes have been made, comparing to the original contribution:
48 1. A lot of small optimizations, less memory allocations, got rid of global buffers
49 2. Reversed order of cvGetQuadrangleSubPix and cvResize calls; probably less accurate, but much faster
50 3. The descriptor computing part (which is most expensive) is threaded using OpenMP
51 (subpixel-accurate keypoint localization and scale estimation are still TBD)
52*/

54/*
55KeyPoint position and scale interpolation has been implemented as described in
56the Brown and Lowe paper cited by the SURF paper.
57
58The sampling step along the x and y axes of the image for the determinant of the
59Hessian is now the same for each layer in an octave. While this increases the
60computation time, it ensures that a true 3x3x3 neighbourhood exists, with
61samples calculated at the same position in the layers above and below. This
62results in improved maxima detection and non-maxima suppression, and I think it
63is consistent with the description in the SURF paper.
64
65The wavelet size sampling interval has also been made consistent. The wavelet
66size at the first layer of the first octave is now 9 instead of 7. Along with
67regular position sampling steps, this makes location and scale interpolation
68easy. I think this is consistent with the SURF paper and original
69implementation.
70
71The scaling of the wavelet parameters has been fixed to ensure that the patterns
72are symmetric around the centre. Previously the truncation caused by integer
73division in the scaling ratio caused a bias towards the top left of the wavelet,
74resulting in inconsistent keypoint positions.
75
76The matrices for the determinant and trace of the Hessian are now reused in each
77octave.
78
79The extraction of the patch of pixels surrounding a keypoint used to build a
80descriptor has been simplified.
81
82KeyPoint descriptor normalisation has been changed from normalising each 4x4
83cell (resulting in a descriptor of magnitude 16) to normalising the entire
84descriptor to magnitude 1.
85
86The default number of octaves has been increased from 3 to 4 to match the
87original SURF binary default. The increase in computation time is minimal since
88the higher octaves are sampled sparsely.
89
90The default number of layers per octave has been reduced from 3 to 2, to prevent
91redundant calculation of similar sizes in consecutive octaves. This decreases
92computation time. The number of features extracted may be less, however the
93additional features were mostly redundant.
94
95The radius of the circle of gradient samples used to assign an orientation has
96been increased from 4 to 6 to match the description in the SURF paper. This is
97now defined by ORI_RADIUS, and could be made into a parameter.
98
99The size of the sliding window used in orientation assignment has been reduced
100from 120 to 60 degrees to match the description in the SURF paper. This is now
101defined by ORI_WIN, and could be made into a parameter.
102
103Other options like HAAR_SIZE0, HAAR_SIZE_INC, SAMPLE_STEP0, ORI_SEARCH_INC,
104ORI_SIGMA and DESC_SIGMA have been separated from the code and documented.
105These could also be made into parameters.
106
107Modifications by Ian Mahon
108
109*/

129struct SurfHF
130{
131 int p0, p1, p2, p3;
132 float w;
133
134 SurfHF(): p0(0), p1(0), p2(0), p3(0), w(0) {}
135};
136
137inline float calcHaarPattern( const int* origin, const SurfHF* f, int n )
138{
139 double d = 0;
140 for( int k = 0; k < n; k++ )
141 d += (origin[f[k].p0] + origin[f[k].p3] - origin[f[k].p1] - origin[f[k].p2])*f[k].w;
142 return (float)d;
143}
144
145static void
146resizeHaarPattern( const int src[][5], SurfHF* dst, int n, int oldSize, int newSize, int widthStep )
147{
148 float ratio = (float)newSize/oldSize;
149 for( int k = 0; k < n; k++ )
150 {
151 int dx1 = cvRound( ratio*src[k][0] );
152 int dy1 = cvRound( ratio*src[k][1] );
153 int dx2 = cvRound( ratio*src[k][2] );
154 int dy2 = cvRound( ratio*src[k][3] );
155 dst[k].p0 = dy1*widthStep + dx1;
156 dst[k].p1 = dy2*widthStep + dx1;
157 dst[k].p2 = dy1*widthStep + dx2;
158 dst[k].p3 = dy2*widthStep + dx2;
159 dst[k].w = src[k][4]/((float)(dx2-dx1)*(dy2-dy1));
160 }
161}

163/*
164 * Calculate the determinant and trace of the Hessian for a layer of the
165 * scale-space pyramid
166 */
167static void calcLayerDetAndTrace( const Mat& sum, int size, int sampleStep,
168 Mat& det, Mat& trace )
169{
170 const int NX=3, NY=3, NXY=4;
171 const int dx_s[NX][5] = { {0, 2, 3, 7, 1}, {3, 2, 6, 7, -2}, {6, 2, 9, 7, 1} };
172 const int dy_s[NY][5] = { {2, 0, 7, 3, 1}, {2, 3, 7, 6, -2}, {2, 6, 7, 9, 1} };
173 const int dxy_s[NXY][5] = { {1, 1, 4, 4, 1}, {5, 1, 8, 4, -1}, {1, 5, 4, 8, -1}, {5, 5, 8, 8, 1} };
174
175 SurfHF Dx[NX], Dy[NY], Dxy[NXY];
176
177 if( size > sum.rows-1 || size > sum.cols-1 )
178 return;
179
180 resizeHaarPattern( dx_s , Dx , NX , 9, size, sum.cols );
181 resizeHaarPattern( dy_s , Dy , NY , 9, size, sum.cols );
182 resizeHaarPattern( dxy_s, Dxy, NXY, 9, size, sum.cols );
183
184 /* The integral image 'sum' is one pixel bigger than the source image */
185 int samples_i = 1+(sum.rows-1-size)/sampleStep;
186 int samples_j = 1+(sum.cols-1-size)/sampleStep;
187
188 /* Ignore pixels where some of the kernel is outside the image */
189 int margin = (size/2)/sampleStep;
190
191 for( int i = 0; i < samples_i; i++ )
192 {
193 const int* sum_ptr = sum.ptr<int>(i*sampleStep);
194 float* det_ptr = &det.at<float>(i+margin, margin);
195 float* trace_ptr = &trace.at<float>(i+margin, margin);
196 for( int j = 0; j < samples_j; j++ )
197 {
198 float dx = calcHaarPattern( sum_ptr, Dx , 3 );
199 float dy = calcHaarPattern( sum_ptr, Dy , 3 );
200 float dxy = calcHaarPattern( sum_ptr, Dxy, 4 );
201 sum_ptr += sampleStep;
202 det_ptr[j] = dx*dy - 0.81f*dxy*dxy;
203 trace_ptr[j] = dx + dy;
204 }
205 }
206}

209/*
210 * Maxima location interpolation as described in "Invariant Features from
211 * Interest Point Groups" by Matthew Brown and David Lowe. This is performed by
212 * fitting a 3D quadratic to a set of neighbouring samples.
213 *
214 * The gradient vector and Hessian matrix at the initial keypoint location are
215 * approximated using central differences. The linear system Ax = b is then
216 * solved, where A is the Hessian, b is the negative gradient, and x is the
217 * offset of the interpolated maxima coordinates from the initial estimate.
218 * This is equivalent to an iteration of Netwon's optimisation algorithm.
219 *
220 * N9 contains the samples in the 3x3x3 neighbourhood of the maxima
221 * dx is the sampling step in x
222 * dy is the sampling step in y
223 * ds is the sampling step in size
224 * point contains the keypoint coordinates and scale to be modified
225 *
226 * Return value is 1 if interpolation was successful, 0 on failure.
227 */
228static int
229interpolateKeypoint( float N9[3][9], int dx, int dy, int ds, KeyPoint& kpt )
230{
231 Vec3f b(-(N9[1][5]-N9[1][3])/2, // Negative 1st deriv with respect to x
232 -(N9[1][7]-N9[1][1])/2, // Negative 1st deriv with respect to y
233 -(N9[2][4]-N9[0][4])/2); // Negative 1st deriv with respect to s
234
235 Matx33f A(
236 N9[1][3]-2*N9[1][4]+N9[1][5], // 2nd deriv x, x
237 (N9[1][8]-N9[1][6]-N9[1][2]+N9[1][0])/4, // 2nd deriv x, y
238 (N9[2][5]-N9[2][3]-N9[0][5]+N9[0][3])/4, // 2nd deriv x, s
239 (N9[1][8]-N9[1][6]-N9[1][2]+N9[1][0])/4, // 2nd deriv x, y
240 N9[1][1]-2*N9[1][4]+N9[1][7], // 2nd deriv y, y
241 (N9[2][7]-N9[2][1]-N9[0][7]+N9[0][1])/4, // 2nd deriv y, s
242 (N9[2][5]-N9[2][3]-N9[0][5]+N9[0][3])/4, // 2nd deriv x, s
243 (N9[2][7]-N9[2][1]-N9[0][7]+N9[0][1])/4, // 2nd deriv y, s
244 N9[0][4]-2*N9[1][4]+N9[2][4]); // 2nd deriv s, s
245
246 Vec3f x = A.solve(b, DECOMP_LU);
247
248 bool ok = (x[0] != 0 || x[1] != 0 || x[2] != 0) &&
249 std::abs(x[0]) <= 1 && std::abs(x[1]) <= 1 && std::abs(x[2]) <= 1;
250
251 if( ok )
252 {
253 kpt.pt.x += x[0]*dx;
254 kpt.pt.y += x[1]*dy;
255 kpt.size = (float)cvRound( kpt.size + x[2]*ds );
256 }
257 return ok;
258}

342/*
343 * Find the maxima in the determinant of the Hessian in a layer of the
344 * scale-space pyramid
345 */
346void SURFFindInvoker::findMaximaInLayer( const Mat& sum, const Mat& mask_sum,
347 const vector<Mat>& dets, const vector<Mat>& traces,
348 const vector<int>& sizes, vector<KeyPoint>& keypoints,
349 int octave, int layer, float hessianThreshold, int sampleStep )
350{
351 // Wavelet Data
352 const int NM=1;
353 const int dm[NM][5] = { {0, 0, 9, 9, 1} };
354 SurfHF Dm;
355
356 int size = sizes[layer];
357
358 // The integral image 'sum' is one pixel bigger than the source image
359 int layer_rows = (sum.rows-1)/sampleStep;
360 int layer_cols = (sum.cols-1)/sampleStep;
361
362 // Ignore pixels without a 3x3x3 neighbourhood in the layer above
363 int margin = (sizes[layer+1]/2)/sampleStep+1;
364
365 if( !mask_sum.empty() )
366 resizeHaarPattern( dm, &Dm, NM, 9, size, mask_sum.cols );
367
368 int step = (int)(dets[layer].step/dets[layer].elemSize());
369
370 for( int i = margin; i < layer_rows - margin; i++ )
371 {
372 const float* det_ptr = dets[layer].ptr<float>(i);
373 const float* trace_ptr = traces[layer].ptr<float>(i);
374 for( int j = margin; j < layer_cols-margin; j++ )
375 {
376 float val0 = det_ptr[j];
377 if( val0 > hessianThreshold )
378 {
379 /* Coordinates for the start of the wavelet in the sum image. There
380 is some integer division involved, so don't try to simplify this
381 (cancel out sampleStep) without checking the result is the same */
382 int sum_i = sampleStep*(i-(size/2)/sampleStep);
383 int sum_j = sampleStep*(j-(size/2)/sampleStep);
384
385 /* The 3x3x3 neighbouring samples around the maxima.
386 The maxima is included at N9[1][4] */
387
388 const float *det1 = &dets[layer-1].at<float>(i, j);
389 const float *det2 = &dets[layer].at<float>(i, j);
390 const float *det3 = &dets[layer+1].at<float>(i, j);
391 float N9[3][9] = { { det1[-step-1], det1[-step], det1[-step+1],
392 det1[-1] , det1[0] , det1[1],
393 det1[step-1] , det1[step] , det1[step+1] },
394 { det2[-step-1], det2[-step], det2[-step+1],
395 det2[-1] , det2[0] , det2[1],
396 det2[step-1] , det2[step] , det2[step+1] },
397 { det3[-step-1], det3[-step], det3[-step+1],
398 det3[-1] , det3[0] , det3[1],
399 det3[step-1] , det3[step] , det3[step+1] } };
400
401 /* Check the mask - why not just check the mask at the center of the wavelet? */
402 if( !mask_sum.empty() )
403 {
404 const int* mask_ptr = &mask_sum.at<int>(sum_i, sum_j);
405 float mval = calcHaarPattern( mask_ptr, &Dm, 1 );
406 if( mval < 0.5 )
407 continue;
408 }
409
410 /* Non-maxima suppression. val0 is at N9[1][4]*/
411 if( val0 > N9[0][0] && val0 > N9[0][1] && val0 > N9[0][2] &&
412 val0 > N9[0][3] && val0 > N9[0][4] && val0 > N9[0][5] &&
413 val0 > N9[0][6] && val0 > N9[0][7] && val0 > N9[0][8] &&
414 val0 > N9[1][0] && val0 > N9[1][1] && val0 > N9[1][2] &&
415 val0 > N9[1][3] && val0 > N9[1][5] &&
416 val0 > N9[1][6] && val0 > N9[1][7] && val0 > N9[1][8] &&
417 val0 > N9[2][0] && val0 > N9[2][1] && val0 > N9[2][2] &&
418 val0 > N9[2][3] && val0 > N9[2][4] && val0 > N9[2][5] &&
419 val0 > N9[2][6] && val0 > N9[2][7] && val0 > N9[2][8] )
420 {
421 /* Calculate the wavelet center coordinates for the maxima */
422 float center_i = sum_i + (size-1)*0.5f;
423 float center_j = sum_j + (size-1)*0.5f;
424
425 KeyPoint kpt( center_j, center_i, (float)sizes[layer],
426 -1, val0, octave, CV_SIGN(trace_ptr[j]) );
427
428 /* Interpolate maxima location within the 3x3x3 neighbourhood */
429 int ds = size - sizes[layer-1];
430 int interp_ok = interpolateKeypoint( N9, sampleStep, sampleStep, ds, kpt );
431
432 /* Sometimes the interpolation step gives a negative size etc. */
433 if( interp_ok )
434 {
435 /*printf( "KeyPoint %f %f %d\n", point.pt.x, point.pt.y, point.size );*/
436 cv::AutoLock lock(findMaximaInLayer_m);
437 keypoints.push_back(kpt);
438 }
439 }
440 }
441 }
442 }
443}

445struct KeypointGreater
446{
447 inline bool operator()(const KeyPoint& kp1, const KeyPoint& kp2) const
448 {
449 if(kp1.response > kp2.response) return true;
450 if(kp1.response < kp2.response) return false;
451 if(kp1.size > kp2.size) return true;
452 if(kp1.size < kp2.size) return false;
453 if(kp1.octave > kp2.octave) return true;
454 if(kp1.octave < kp2.octave) return false;
455 if(kp1.pt.y < kp2.pt.y) return false;
456 if(kp1.pt.y > kp2.pt.y) return true;
457 return kp1.pt.x < kp2.pt.x;
458 }
459};

462static void fastHessianDetector( const Mat& sum, const Mat& mask_sum, vector<KeyPoint>& keypoints,
463 int nOctaves, int nOctaveLayers, float hessianThreshold )
464{
465 /* Sampling step along image x and y axes at first octave. This is doubled
466 for each additional octave. WARNING: Increasing this improves speed,
467 however keypoint extraction becomes unreliable. */
468 const int SAMPLE_STEP0 = 1;
469
470 int nTotalLayers = (nOctaveLayers+2)*nOctaves;
471 int nMiddleLayers = nOctaveLayers*nOctaves;
472
473 vector<Mat> dets(nTotalLayers);
474 vector<Mat> traces(nTotalLayers);
475 vector<int> sizes(nTotalLayers);
476 vector<int> sampleSteps(nTotalLayers);
477 vector<int> middleIndices(nMiddleLayers);
478
479 keypoints.clear();
480
481 // Allocate space and calculate properties of each layer
482 int index = 0, middleIndex = 0, step = SAMPLE_STEP0;
483
484 for( int octave = 0; octave < nOctaves; octave++ )
485 {
486 for( int layer = 0; layer < nOctaveLayers+2; layer++ )
487 {
488 /* The integral image sum is one pixel bigger than the source image*/
489 dets[index].create( (sum.rows-1)/step, (sum.cols-1)/step, CV_32F );
490 traces[index].create( (sum.rows-1)/step, (sum.cols-1)/step, CV_32F );
491 sizes[index] = (SURF_HAAR_SIZE0 + SURF_HAAR_SIZE_INC*layer) << octave;
492 sampleSteps[index] = step;
493
494 if( 0 < layer && layer <= nOctaveLayers )
495 middleIndices[middleIndex++] = index;
496 index++;
497 }
498 step *= 2;
499 }
500
501 // Calculate hessian determinant and trace samples in each layer
502 parallel_for_( Range(0, nTotalLayers),
503 SURFBuildInvoker(sum, sizes, sampleSteps, dets, traces) );
504
505 // Find maxima in the determinant of the hessian
506 parallel_for_( Range(0, nMiddleLayers),
507 SURFFindInvoker(sum, mask_sum, dets, traces, sizes,
508 sampleSteps, middleIndices, keypoints,
509 nOctaveLayers, hessianThreshold) );
510
511 std::sort(keypoints.begin(), keypoints.end(), KeypointGreater());
512}

515struct SURFInvoker : ParallelLoopBody
516{
517 enum { ORI_RADIUS = 6, ORI_WIN = 60, PATCH_SZ = 20 };
518
519 SURFInvoker( const Mat& _img, const Mat& _sum,
520 vector<KeyPoint>& _keypoints, Mat& _descriptors,
521 bool _extended, bool _upright )
522 {
523 keypoints = &_keypoints;
524 descriptors = &_descriptors;
525 img = &_img;
526 sum = &_sum;
527 extended = _extended;
528 upright = _upright;
529
530 // Simple bound for number of grid points in circle of radius ORI_RADIUS
531 const int nOriSampleBound = (2*ORI_RADIUS+1)*(2*ORI_RADIUS+1);
532
533 // Allocate arrays
534 apt.resize(nOriSampleBound);
535 aptw.resize(nOriSampleBound);
536 DW.resize(PATCH_SZ*PATCH_SZ);
537
538 /* Coordinates and weights of samples used to calculate orientation */
539 Mat G_ori = getGaussianKernel( 2*ORI_RADIUS+1, SURF_ORI_SIGMA, CV_32F );
540 nOriSamples = 0;
541 for( int i = -ORI_RADIUS; i <= ORI_RADIUS; i++ )
542 {
543 for( int j = -ORI_RADIUS; j <= ORI_RADIUS; j++ )
544 {
545 if( i*i + j*j <= ORI_RADIUS*ORI_RADIUS )
546 {
547 apt[nOriSamples] = cvPoint(i,j);
548 aptw[nOriSamples++] = G_ori.at<float>(i+ORI_RADIUS,0) * G_ori.at<float>(j+ORI_RADIUS,0);
549 }
550 }
551 }
552 CV_Assert( nOriSamples <= nOriSampleBound );
553
554 /* Gaussian used to weight descriptor samples */
555 Mat G_desc = getGaussianKernel( PATCH_SZ, SURF_DESC_SIGMA, CV_32F );
556 for( int i = 0; i < PATCH_SZ; i++ )
557 {
558 for( int j = 0; j < PATCH_SZ; j++ )
559 DW[i*PATCH_SZ+j] = G_desc.at<float>(i,0) * G_desc.at<float>(j,0);
560 }
561 }
562
563 void operator()(const Range& range) const
564 {
565 /* X and Y gradient wavelet data */
566 const int NX=2, NY=2;
567 const int dx_s[NX][5] = {{0, 0, 2, 4, -1}, {2, 0, 4, 4, 1}};
568 const int dy_s[NY][5] = {{0, 0, 4, 2, 1}, {0, 2, 4, 4, -1}};
569
570 // Optimisation is better using nOriSampleBound than nOriSamples for
571 // array lengths. Maybe because it is a constant known at compile time
572 const int nOriSampleBound =(2*ORI_RADIUS+1)*(2*ORI_RADIUS+1);
573
574 float X[nOriSampleBound], Y[nOriSampleBound], angle[nOriSampleBound];
575 uchar PATCH[PATCH_SZ+1][PATCH_SZ+1];
576 float DX[PATCH_SZ][PATCH_SZ], DY[PATCH_SZ][PATCH_SZ];
577 CvMat matX = cvMat(1, nOriSampleBound, CV_32F, X);
578 CvMat matY = cvMat(1, nOriSampleBound, CV_32F, Y);
579 CvMat _angle = cvMat(1, nOriSampleBound, CV_32F, angle);
580 Mat _patch(PATCH_SZ+1, PATCH_SZ+1, CV_8U, PATCH);
581
582 int dsize = extended ? 128 : 64;
583
584 int k, k1 = range.start, k2 = range.end;
585 float maxSize = 0;
586 for( k = k1; k < k2; k++ )
587 {
588 maxSize = std::max(maxSize, (*keypoints)[k].size);
589 }
590 int imaxSize = std::max(cvCeil((PATCH_SZ+1)*maxSize*1.2f/9.0f), 1);
591 Ptr<CvMat> winbuf = cvCreateMat( 1, imaxSize*imaxSize, CV_8U );
592 for( k = k1; k < k2; k++ )
593 {
594 int i, j, kk, nangle;
595 float* vec;
596 SurfHF dx_t[NX], dy_t[NY];
597 KeyPoint& kp = (*keypoints)[k];
598 float size = kp.size;
599 Point2f center = kp.pt;
600 /* The sampling intervals and wavelet sized for selecting an orientation
601 and building the keypoint descriptor are defined relative to 's' */
602 float s = size*1.2f/9.0f;
603 /* To find the dominant orientation, the gradients in x and y are
604 sampled in a circle of radius 6s using wavelets of size 4s.
605 We ensure the gradient wavelet size is even to ensure the
606 wavelet pattern is balanced and symmetric around its center */
607 int grad_wav_size = 2*cvRound( 2*s );
608 if( sum->rows < grad_wav_size || sum->cols < grad_wav_size )
609 {
610 /* when grad_wav_size is too big,
611 * the sampling of gradient will be meaningless
612 * mark keypoint for deletion. */
613 kp.size = -1;
614 continue;
615 }
616
617 float descriptor_dir = 360.f - 90.f;
618 if (upright == 0)
619 {
620 resizeHaarPattern( dx_s, dx_t, NX, 4, grad_wav_size, sum->cols );
621 resizeHaarPattern( dy_s, dy_t, NY, 4, grad_wav_size, sum->cols );
622 for( kk = 0, nangle = 0; kk < nOriSamples; kk++ )
623 {
624 int x = cvRound( center.x + apt[kk].x*s - (float)(grad_wav_size-1)/2 );
625 int y = cvRound( center.y + apt[kk].y*s - (float)(grad_wav_size-1)/2 );
626 if( y < 0 || y >= sum->rows - grad_wav_size ||
627 x < 0 || x >= sum->cols - grad_wav_size )
628 continue;
629 const int* ptr = &sum->at<int>(y, x);
630 float vx = calcHaarPattern( ptr, dx_t, 2 );
631 float vy = calcHaarPattern( ptr, dy_t, 2 );
632 X[nangle] = vx*aptw[kk];
633 Y[nangle] = vy*aptw[kk];
634 nangle++;
635 }
636 if( nangle == 0 )
637 {
638 // No gradient could be sampled because the keypoint is too
639 // near too one or more of the sides of the image. As we
640 // therefore cannot find a dominant direction, we skip this
641 // keypoint and mark it for later deletion from the sequence.
642 kp.size = -1;
643 continue;
644 }
645 matX.cols = matY.cols = _angle.cols = nangle;
646 cvCartToPolar( &matX, &matY, 0, &_angle, 1 );
647
648 float bestx = 0, besty = 0, descriptor_mod = 0;
649 for( i = 0; i < 360; i += SURF_ORI_SEARCH_INC )
650 {
651 float sumx = 0, sumy = 0, temp_mod;
652 for( j = 0; j < nangle; j++ )
653 {
654 int d = std::abs(cvRound(angle[j]) - i);
655 if( d < ORI_WIN/2 || d > 360-ORI_WIN/2 )
656 {
657 sumx += X[j];
658 sumy += Y[j];
659 }
660 }
661 temp_mod = sumx*sumx + sumy*sumy;
662 if( temp_mod > descriptor_mod )
663 {
664 descriptor_mod = temp_mod;
665 bestx = sumx;
666 besty = sumy;
667 }
668 }
669 descriptor_dir = fastAtan2( -besty, bestx );
670 }
671 kp.angle = descriptor_dir;
672 if( !descriptors || !descriptors->data )
673 continue;
674
675 /* Extract a window of pixels around the keypoint of size 20s */
676 int win_size = (int)((PATCH_SZ+1)*s);
677 CV_Assert( winbuf->cols >= win_size*win_size );
678 Mat win(win_size, win_size, CV_8U, winbuf->data.ptr);
679
680 if( !upright )
681 {
682 descriptor_dir *= (float)(CV_PI/180);
683 float sin_dir = -std::sin(descriptor_dir);
684 float cos_dir = std::cos(descriptor_dir);
685
686 /* Subpixel interpolation version (slower). Subpixel not required since
687 the pixels will all get averaged when we scale down to 20 pixels */
688 /*
689 float w[] = { cos_dir, sin_dir, center.x,
690 -sin_dir, cos_dir , center.y };
691 CvMat W = cvMat(2, 3, CV_32F, w);
692 cvGetQuadrangleSubPix( img, &win, &W );
693 */
694
695 float win_offset = -(float)(win_size-1)/2;
696 float start_x = center.x + win_offset*cos_dir + win_offset*sin_dir;
697 float start_y = center.y - win_offset*sin_dir + win_offset*cos_dir;
698 uchar* WIN = win.data;
699#if 0
700 // Nearest neighbour version (faster)
701 for( i = 0; i < win_size; i++, start_x += sin_dir, start_y += cos_dir )
702 {
703 float pixel_x = start_x;
704 float pixel_y = start_y;
705 for( j = 0; j < win_size; j++, pixel_x += cos_dir, pixel_y -= sin_dir )
706 {
707 int x = std::min(std::max(cvRound(pixel_x), 0), img->cols-1);
708 int y = std::min(std::max(cvRound(pixel_y), 0), img->rows-1);
709 WIN[i*win_size + j] = img->at<uchar>(y, x);
710 }
711 }
712#else
713 int ncols1 = img->cols-1, nrows1 = img->rows-1;
714 size_t imgstep = img->step;
715 for( i = 0; i < win_size; i++, start_x += sin_dir, start_y += cos_dir )
716 {
717 double pixel_x = start_x;
718 double pixel_y = start_y;
719 for( j = 0; j < win_size; j++, pixel_x += cos_dir, pixel_y -= sin_dir )
720 {
721 int ix = cvFloor(pixel_x), iy = cvFloor(pixel_y);
722 if( (unsigned)ix < (unsigned)ncols1 &&
723 (unsigned)iy < (unsigned)nrows1 )
724 {
725 float a = (float)(pixel_x - ix), b = (float)(pixel_y - iy);
726 const uchar* imgptr = &img->at<uchar>(iy, ix);
727 WIN[i*win_size + j] = (uchar)
728 cvRound(imgptr[0]*(1.f - a)*(1.f - b) +
729 imgptr[1]*a*(1.f - b) +
730 imgptr[imgstep]*(1.f - a)*b +
731 imgptr[imgstep+1]*a*b);
732 }
733 else
734 {
735 int x = std::min(std::max(cvRound(pixel_x), 0), ncols1);
736 int y = std::min(std::max(cvRound(pixel_y), 0), nrows1);
737 WIN[i*win_size + j] = img->at<uchar>(y, x);
738 }
739 }
740 }
741#endif
742 }
743 else
744 {
745 // extract rect - slightly optimized version of the code above
746 // TODO: find faster code, as this is simply an extract rect operation,
747 // e.g. by using cvGetSubRect, problem is the border processing
748 // descriptor_dir == 90 grad
749 // sin_dir == 1
750 // cos_dir == 0
751
752 float win_offset = -(float)(win_size-1)/2;
753 int start_x = cvRound(center.x + win_offset);
754 int start_y = cvRound(center.y - win_offset);
755 uchar* WIN = win.data;
756 for( i = 0; i < win_size; i++, start_x++ )
757 {
758 int pixel_x = start_x;
759 int pixel_y = start_y;
760 for( j = 0; j < win_size; j++, pixel_y-- )
761 {
762 int x = MAX( pixel_x, 0 );
763 int y = MAX( pixel_y, 0 );
764 x = MIN( x, img->cols-1 );
765 y = MIN( y, img->rows-1 );
766 WIN[i*win_size + j] = img->at<uchar>(y, x);
767 }
768 }
769 }
770 // Scale the window to size PATCH_SZ so each pixel's size is s. This
771 // makes calculating the gradients with wavelets of size 2s easy
772 resize(win, _patch, _patch.size(), 0, 0, INTER_AREA);
773
774 // Calculate gradients in x and y with wavelets of size 2s
775 for( i = 0; i < PATCH_SZ; i++ )
776 for( j = 0; j < PATCH_SZ; j++ )
777 {
778 float dw = DW[i*PATCH_SZ + j];
779 float vx = (PATCH[i][j+1] - PATCH[i][j] + PATCH[i+1][j+1] - PATCH[i+1][j])*dw;
780 float vy = (PATCH[i+1][j] - PATCH[i][j] + PATCH[i+1][j+1] - PATCH[i][j+1])*dw;
781 DX[i][j] = vx;
782 DY[i][j] = vy;
783 }
784
785 // Construct the descriptor
786 vec = descriptors->ptr<float>(k);
787 for( kk = 0; kk < dsize; kk++ )
788 vec[kk] = 0;
789 double square_mag = 0;
790 if( extended )
791 {
792 // 128-bin descriptor
793 for( i = 0; i < 4; i++ )
794 for( j = 0; j < 4; j++ )
795 {
796 for(int y = i*5; y < i*5+5; y++ )
797 {
798 for(int x = j*5; x < j*5+5; x++ )
799 {
800 float tx = DX[y][x], ty = DY[y][x];
801 if( ty >= 0 )
802 {
803 vec[0] += tx;
804 vec[1] += (float)fabs(tx);
805 } else {
806 vec[2] += tx;
807 vec[3] += (float)fabs(tx);
808 }
809 if ( tx >= 0 )
810 {
811 vec[4] += ty;
812 vec[5] += (float)fabs(ty);
813 } else {
814 vec[6] += ty;
815 vec[7] += (float)fabs(ty);
816 }
817 }
818 }
819 for( kk = 0; kk < 8; kk++ )
820 square_mag += vec[kk]*vec[kk];
821 vec += 8;
822 }
823 }
824 else
825 {
826 // 64-bin descriptor
827 for( i = 0; i < 4; i++ )
828 for( j = 0; j < 4; j++ )
829 {
830 for(int y = i*5; y < i*5+5; y++ )
831 {
832 for(int x = j*5; x < j*5+5; x++ )
833 {
834 float tx = DX[y][x], ty = DY[y][x];
835 vec[0] += tx; vec[1] += ty;
836 vec[2] += (float)fabs(tx); vec[3] += (float)fabs(ty);
837 }
838 }
839 for( kk = 0; kk < 4; kk++ )
840 square_mag += vec[kk]*vec[kk];
841 vec+=4;
842 }
843 }
844
845 // unit vector is essential for contrast invariance
846 vec = descriptors->ptr<float>(k);
847 float scale = (float)(1./(sqrt(square_mag) + DBL_EPSILON));
848 for( kk = 0; kk < dsize; kk++ )
849 vec[kk] *= scale;
850 }
851 }

854 const Mat* img;
855 const Mat* sum;
856 vector<KeyPoint>* keypoints;
857 Mat* descriptors;
858 bool extended;
859 bool upright;
860
861 // Pre-calculated values
862 int nOriSamples;
863 vector<Point> apt;
864 vector<float> aptw;
865 vector<float> DW;

895
896void SURF::operator()(InputArray _img, InputArray _mask,
897 CV_OUT vector<KeyPoint>& keypoints,
898 OutputArray _descriptors,
899 bool useProvidedKeypoints) const
900{
901 Mat img = _img.getMat(), mask = _mask.getMat(), mask1, sum, msum;
902 bool doDescriptors = _descriptors.needed();
903
904 CV_Assert(!img.empty() && img.depth() == CV_8U);
905 if( img.channels() > 1 )
906 cvtColor(img, img, COLOR_BGR2GRAY);
907
908 CV_Assert(mask.empty() || (mask.type() == CV_8U && mask.size() == img.size()));
909 CV_Assert(hessianThreshold >= 0);
910 CV_Assert(nOctaves > 0);
911 CV_Assert(nOctaveLayers > 0);
912
913 integral(img, sum, CV_32S);
914
915 // Compute keypoints only if we are not asked for evaluating the descriptors are some given locations:
916 if( !useProvidedKeypoints )
917 {
918 if( !mask.empty() )
919 {
920 cv::min(mask, 1, mask1);
921 integral(mask1, msum, CV_32S);
922 }
923 fastHessianDetector( sum, msum, keypoints, nOctaves, nOctaveLayers, (float)hessianThreshold );
924 }
925
926 int i, j, N = (int)keypoints.size();
927 if( N > 0 )
928 {
929 Mat descriptors;
930 bool _1d = false;
931 int dcols = extended ? 128 : 64;
932 size_t dsize = dcols*sizeof(float);
933
934 if( doDescriptors )
935 {
936 _1d = _descriptors.kind() == _InputArray::STD_VECTOR && _descriptors.type() == CV_32F;
937 if( _1d )
938 {
939 _descriptors.create(N*dcols, 1, CV_32F);
940 descriptors = _descriptors.getMat().reshape(1, N);
941 }
942 else
943 {
944 _descriptors.create(N, dcols, CV_32F);
945 descriptors = _descriptors.getMat();
946 }
947 }
948
949 // we call SURFInvoker in any case, even if we do not need descriptors,
950 // since it computes orientation of each feature.
951 parallel_for_(Range(0, N), SURFInvoker(img, sum, keypoints, descriptors, extended, upright) );
952
953 // remove keypoints that were marked for deletion
954 for( i = j = 0; i < N; i++ )
955 {
956 if( keypoints[i].size > 0 )
957 {
958 if( i > j )
959 {
960 keypoints[j] = keypoints[i];
961 if( doDescriptors )
962 memcpy( descriptors.ptr(j), descriptors.ptr(i), dsize);
963 }
964 j++;
965 }
966 }
967 if( N > j )
968 {
969 N = j;
970 keypoints.resize(N);
971 if( doDescriptors )
972 {
973 Mat d = descriptors.rowRange(0, N);
974 if( _1d )
975 d = d.reshape(1, N*dcols);
976 d.copyTo(_descriptors);
977 }
978 }
979 }
980}
