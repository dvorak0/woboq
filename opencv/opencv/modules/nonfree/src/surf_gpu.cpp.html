96 int calcSize(int octave, int layer)
97 {
98 /* Wavelet size at first layer of first octave. */
99 const int HAAR_SIZE0 = 9;
100
101 /* Wavelet size increment between layers. This should be an even number,
102 such that the wavelet sizes in an octave are either all even or all odd.
103 This ensures that when looking for the neighbours of a sample, the layers
104
105 above and below are aligned correctly. */
106 const int HAAR_SIZE_INC = 6;
107
108 return (HAAR_SIZE0 + HAAR_SIZE_INC * layer) << octave;
109 }
110
111 class SURF_GPU_Invoker
112 {
113 public:
114 SURF_GPU_Invoker(SURF_GPU& surf, const GpuMat& img, const GpuMat& mask) :
115 surf_(surf),
116 img_cols(img.cols), img_rows(img.rows),
117 use_mask(!mask.empty())
118 {
119 CV_Assert(!img.empty() && img.type() == CV_8UC1);
120 CV_Assert(mask.empty() || (mask.size() == img.size() && mask.type() == CV_8UC1));
121 CV_Assert(surf_.nOctaves > 0 && surf_.nOctaveLayers > 0);
122
123 const int min_size = calcSize(surf_.nOctaves - 1, 0);
124 CV_Assert(img_rows - min_size >= 0);
125 CV_Assert(img_cols - min_size >= 0);
126
127 const int layer_rows = img_rows >> (surf_.nOctaves - 1);
128 const int layer_cols = img_cols >> (surf_.nOctaves - 1);
129 const int min_margin = ((calcSize((surf_.nOctaves - 1), 2) >> 1) >> (surf_.nOctaves - 1)) + 1;
130 CV_Assert(layer_rows - 2 * min_margin > 0);
131 CV_Assert(layer_cols - 2 * min_margin > 0);
132
133 maxFeatures = min(static_cast<int>(img.size().area() * surf.keypointsRatio), 65535);
134 maxCandidates = min(static_cast<int>(1.5 * maxFeatures), 65535);
135
136 CV_Assert(maxFeatures > 0);
137
138 counters.create(1, surf_.nOctaves + 1, CV_32SC1);
139 counters.setTo(Scalar::all(0));
140
141 loadGlobalConstants(maxCandidates, maxFeatures, img_rows, img_cols, surf_.nOctaveLayers, static_cast<float>(surf_.hessianThreshold));
142
143 bindImgTex(img);
144
145 integralBuffered(img, surf_.sum, surf_.intBuffer);
146 sumOffset = bindSumTex(surf_.sum);
147
148 if (use_mask)
149 {
150 min(mask, 1.0, surf_.mask1);
151 integralBuffered(surf_.mask1, surf_.maskSum, surf_.intBuffer);
152 maskOffset = bindMaskSumTex(surf_.maskSum);
153 }
154 }
155
156 void detectKeypoints(GpuMat& keypoints)
157 {
158 ensureSizeIsEnough(img_rows * (surf_.nOctaveLayers + 2), img_cols, CV_32FC1, surf_.det);
159 ensureSizeIsEnough(img_rows * (surf_.nOctaveLayers + 2), img_cols, CV_32FC1, surf_.trace);
160
161 ensureSizeIsEnough(1, maxCandidates, CV_32SC4, surf_.maxPosBuffer);
162 ensureSizeIsEnough(SURF_GPU::ROWS_COUNT, maxFeatures, CV_32FC1, keypoints);
163 keypoints.setTo(Scalar::all(0));
164
165 for (int octave = 0; octave < surf_.nOctaves; ++octave)
166 {
167 const int layer_rows = img_rows >> octave;
168 const int layer_cols = img_cols >> octave;
169
170 loadOctaveConstants(octave, layer_rows, layer_cols);
171
172 icvCalcLayerDetAndTrace_gpu(surf_.det, surf_.trace, img_rows, img_cols, octave, surf_.nOctaveLayers);
173
174 icvFindMaximaInLayer_gpu(surf_.det, surf_.trace, surf_.maxPosBuffer.ptr<int4>(), counters.ptr<unsigned int>() + 1 + octave,
175 img_rows, img_cols, octave, use_mask, surf_.nOctaveLayers);
176
177 unsigned int maxCounter;
178 cudaSafeCall( cudaMemcpy(&maxCounter, counters.ptr<unsigned int>() + 1 + octave, sizeof(unsigned int), cudaMemcpyDeviceToHost) );
179 maxCounter = std::min(maxCounter, static_cast<unsigned int>(maxCandidates));
180
181 if (maxCounter > 0)
182 {
183 icvInterpolateKeypoint_gpu(surf_.det, surf_.maxPosBuffer.ptr<int4>(), maxCounter,
184 keypoints.ptr<float>(SURF_GPU::X_ROW), keypoints.ptr<float>(SURF_GPU::Y_ROW),
185 keypoints.ptr<int>(SURF_GPU::LAPLACIAN_ROW), keypoints.ptr<int>(SURF_GPU::OCTAVE_ROW),
186 keypoints.ptr<float>(SURF_GPU::SIZE_ROW), keypoints.ptr<float>(SURF_GPU::HESSIAN_ROW),
187 counters.ptr<unsigned int>());
188 }
189 }
190 unsigned int featureCounter;
191 cudaSafeCall( cudaMemcpy(&featureCounter, counters.ptr<unsigned int>(), sizeof(unsigned int), cudaMemcpyDeviceToHost) );
192 featureCounter = std::min(featureCounter, static_cast<unsigned int>(maxFeatures));
193
194 keypoints.cols = featureCounter;
195
196 if (surf_.upright)
197 keypoints.row(SURF_GPU::ANGLE_ROW).setTo(Scalar::all(360.0 - 90.0));
198 else
199 findOrientation(keypoints);
200 }
201
202 void findOrientation(GpuMat& keypoints)
203 {
204 const int nFeatures = keypoints.cols;
205 if (nFeatures > 0)
206 {
207 icvCalcOrientation_gpu(keypoints.ptr<float>(SURF_GPU::X_ROW), keypoints.ptr<float>(SURF_GPU::Y_ROW),
208 keypoints.ptr<float>(SURF_GPU::SIZE_ROW), keypoints.ptr<float>(SURF_GPU::ANGLE_ROW), nFeatures);
209 }
210 }
211
212 void computeDescriptors(const GpuMat& keypoints, GpuMat& descriptors, int descriptorSize)
213 {
214 const int nFeatures = keypoints.cols;
215 if (nFeatures > 0)
216 {
217 ensureSizeIsEnough(nFeatures, descriptorSize, CV_32F, descriptors);
218 compute_descriptors_gpu(descriptors, keypoints.ptr<float>(SURF_GPU::X_ROW), keypoints.ptr<float>(SURF_GPU::Y_ROW),
219 keypoints.ptr<float>(SURF_GPU::SIZE_ROW), keypoints.ptr<float>(SURF_GPU::ANGLE_ROW), nFeatures);
220 }
221 }

268void cv::gpu::SURF_GPU::uploadKeypoints(const vector<KeyPoint>& keypoints, GpuMat& keypointsGPU)
269{
270 if (keypoints.empty())
271 keypointsGPU.release();
272 else
273 {
274 Mat keypointsCPU(SURF_GPU::ROWS_COUNT, static_cast<int>(keypoints.size()), CV_32FC1);
275
276 float* kp_x = keypointsCPU.ptr<float>(SURF_GPU::X_ROW);
277 float* kp_y = keypointsCPU.ptr<float>(SURF_GPU::Y_ROW);
278 int* kp_laplacian = keypointsCPU.ptr<int>(SURF_GPU::LAPLACIAN_ROW);
279 int* kp_octave = keypointsCPU.ptr<int>(SURF_GPU::OCTAVE_ROW);
280 float* kp_size = keypointsCPU.ptr<float>(SURF_GPU::SIZE_ROW);
281 float* kp_dir = keypointsCPU.ptr<float>(SURF_GPU::ANGLE_ROW);
282 float* kp_hessian = keypointsCPU.ptr<float>(SURF_GPU::HESSIAN_ROW);
283
284 for (size_t i = 0, size = keypoints.size(); i < size; ++i)
285 {
286 const KeyPoint& kp = keypoints[i];
287 kp_x[i] = kp.pt.x;
288 kp_y[i] = kp.pt.y;
289 kp_octave[i] = kp.octave;
290 kp_size[i] = kp.size;
291 kp_dir[i] = kp.angle;
292 kp_hessian[i] = kp.response;
293 kp_laplacian[i] = 1;
294 }
295
296 keypointsGPU.upload(keypointsCPU);
297 }
298}

300void cv::gpu::SURF_GPU::downloadKeypoints(const GpuMat& keypointsGPU, vector<KeyPoint>& keypoints)
301{
302 const int nFeatures = keypointsGPU.cols;
303
304 if (nFeatures == 0)
305 keypoints.clear();
306 else
307 {
308 CV_Assert(keypointsGPU.type() == CV_32FC1 && keypointsGPU.rows == ROWS_COUNT);
309
310 Mat keypointsCPU(keypointsGPU);
311
312 keypoints.resize(nFeatures);
313
314 float* kp_x = keypointsCPU.ptr<float>(SURF_GPU::X_ROW);
315 float* kp_y = keypointsCPU.ptr<float>(SURF_GPU::Y_ROW);
316 int* kp_laplacian = keypointsCPU.ptr<int>(SURF_GPU::LAPLACIAN_ROW);
317 int* kp_octave = keypointsCPU.ptr<int>(SURF_GPU::OCTAVE_ROW);
318 float* kp_size = keypointsCPU.ptr<float>(SURF_GPU::SIZE_ROW);
319 float* kp_dir = keypointsCPU.ptr<float>(SURF_GPU::ANGLE_ROW);
320 float* kp_hessian = keypointsCPU.ptr<float>(SURF_GPU::HESSIAN_ROW);
321
322 for (int i = 0; i < nFeatures; ++i)
323 {
324 KeyPoint& kp = keypoints[i];
325 kp.pt.x = kp_x[i];
326 kp.pt.y = kp_y[i];
327 kp.class_id = kp_laplacian[i];
328 kp.octave = kp_octave[i];
329 kp.size = kp_size[i];
330 kp.angle = kp_dir[i];
331 kp.response = kp_hessian[i];
332 }
333 }

336void cv::gpu::SURF_GPU::downloadDescriptors(const GpuMat& descriptorsGPU, vector<float>& descriptors)
337{
338 if (descriptorsGPU.empty())
339 descriptors.clear();
340 else
341 {
342 CV_Assert(descriptorsGPU.type() == CV_32F);
343
344 descriptors.resize(descriptorsGPU.rows * descriptorsGPU.cols);
345 Mat descriptorsCPU(descriptorsGPU.size(), CV_32F, &descriptors[0]);
346 descriptorsGPU.download(descriptorsCPU);
347 }
348}

350void cv::gpu::SURF_GPU::operator()(const GpuMat& img, const GpuMat& mask, GpuMat& keypoints)
351{
352 if (!img.empty())
353 {
354 SURF_GPU_Invoker surf(*this, img, mask);
355
356 surf.detectKeypoints(keypoints);
357 }
358}
359
360void cv::gpu::SURF_GPU::operator()(const GpuMat& img, const GpuMat& mask, GpuMat& keypoints, GpuMat& descriptors,
361 bool useProvidedKeypoints)
362{
363 if (!img.empty())
364 {
365 SURF_GPU_Invoker surf(*this, img, mask);
366
367 if (!useProvidedKeypoints)
368 surf.detectKeypoints(keypoints);
369 else if (!upright)
370 {
371 surf.findOrientation(keypoints);
372 }
373
374 surf.computeDescriptors(keypoints, descriptors, descriptorSize());
375 }
376}

398}
