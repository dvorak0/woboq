59static
60Mat generateHomography(float angle)
61{
62 // angle - rotation around Oz in degrees
63 float angleRadian = static_cast<float>(angle * CV_PI / 180);
64 Mat H = Mat::eye(3, 3, CV_32FC1);
65 H.at<float>(0,0) = H.at<float>(1,1) = std::cos(angleRadian);
66 H.at<float>(0,1) = -std::sin(angleRadian);
67 H.at<float>(1,0) = std::sin(angleRadian);
68
69 return H;
70}

72static
73Mat rotateImage(const Mat& srcImage, float angle, Mat& dstImage, Mat& dstMask)
74{
75 // angle - rotation around Oz in degrees
76 float diag = std::sqrt(static_cast<float>(srcImage.cols * srcImage.cols + srcImage.rows * srcImage.rows));
77 Mat LUShift = Mat::eye(3, 3, CV_32FC1); // left up
78 LUShift.at<float>(0,2) = static_cast<float>(-srcImage.cols/2);
79 LUShift.at<float>(1,2) = static_cast<float>(-srcImage.rows/2);
80 Mat RDShift = Mat::eye(3, 3, CV_32FC1); // right down
81 RDShift.at<float>(0,2) = diag/2;
82 RDShift.at<float>(1,2) = diag/2;
83 Size sz(cvRound(diag), cvRound(diag));
84
85 Mat srcMask(srcImage.size(), CV_8UC1, Scalar(255));
86
87 Mat H = RDShift * generateHomography(angle) * LUShift;
88 warpPerspective(srcImage, dstImage, H, sz);
89 warpPerspective(srcMask, dstMask, H, sz);
90
91 return H;
92}

94void rotateKeyPoints(const vector<KeyPoint>& src, const Mat& H, float angle, vector<KeyPoint>& dst)
95{
96 // suppose that H is rotation given from rotateImage() and angle has value passed to rotateImage()
97 vector<Point2f> srcCenters, dstCenters;
98 KeyPoint::convert(src, srcCenters);
99
100 perspectiveTransform(srcCenters, dstCenters, H);
101
102 dst = src;
103 for(size_t i = 0; i < dst.size(); i++)
104 {
105 dst[i].pt = dstCenters[i];
106 float dstAngle = src[i].angle + angle;
107 if(dstAngle >= 360.f)
108 dstAngle -= 360.f;
109 dst[i].angle = dstAngle;
110 }
111}

113void scaleKeyPoints(const vector<KeyPoint>& src, vector<KeyPoint>& dst, float scale)
114{
115 dst.resize(src.size());
116 for(size_t i = 0; i < src.size(); i++)
117 dst[i] = KeyPoint(src[i].pt.x * scale, src[i].pt.y * scale, src[i].size * scale, src[i].angle);
118}

120static
121float calcCirclesIntersectArea(const Point2f& p0, float r0, const Point2f& p1, float r1)
122{
123 float c = static_cast<float>(norm(p0 - p1)), sqr_c = c * c;
124
125 float sqr_r0 = r0 * r0;
126 float sqr_r1 = r1 * r1;
127
128 if(r0 + r1 <= c)
129 return 0;
130
131 float minR = std::min(r0, r1);
132 float maxR = std::max(r0, r1);
133 if(c + minR <= maxR)
134 return static_cast<float>(CV_PI * minR * minR);
135
136 float cos_halfA0 = (sqr_r0 + sqr_c - sqr_r1) / (2 * r0 * c);
137 float cos_halfA1 = (sqr_r1 + sqr_c - sqr_r0) / (2 * r1 * c);
138
139 float A0 = 2 * acos(cos_halfA0);
140 float A1 = 2 * acos(cos_halfA1);
141
142 return 0.5f * sqr_r0 * (A0 - sin(A0)) +
143 0.5f * sqr_r1 * (A1 - sin(A1));
144}
145
146static
147float calcIntersectRatio(const Point2f& p0, float r0, const Point2f& p1, float r1)
148{
149 float intersectArea = calcCirclesIntersectArea(p0, r0, p1, r1);
150 float unionArea = static_cast<float>(CV_PI) * (r0 * r0 + r1 * r1) - intersectArea;
151 return intersectArea / unionArea;
152}

154static
155void matchKeyPoints(const vector<KeyPoint>& keypoints0, const Mat& H,
156 const vector<KeyPoint>& keypoints1,
157 vector<DMatch>& matches)
158{
159 vector<Point2f> points0;
160 KeyPoint::convert(keypoints0, points0);
161 Mat points0t;
162 if(H.empty())
163 points0t = Mat(points0);
164 else
165 perspectiveTransform(Mat(points0), points0t, H);
166
167 matches.clear();
168 vector<uchar> usedMask(keypoints1.size(), 0);
169 for(int i0 = 0; i0 < static_cast<int>(keypoints0.size()); i0++)
170 {
171 int nearestPointIndex = -1;
172 float maxIntersectRatio = 0.f;
173 const float r0 = 0.5f * keypoints0[i0].size;
174 for(size_t i1 = 0; i1 < keypoints1.size(); i1++)
175 {
176 if(nearestPointIndex >= 0 && usedMask[i1])
177 continue;
178
179 float r1 = 0.5f * keypoints1[i1].size;
180 float intersectRatio = calcIntersectRatio(points0t.at<Point2f>(i0), r0,
181 keypoints1[i1].pt, r1);
182 if(intersectRatio > maxIntersectRatio)
183 {
184 maxIntersectRatio = intersectRatio;
185 nearestPointIndex = static_cast<int>(i1);
186 }
187 }
188
189 matches.push_back(DMatch(i0, nearestPointIndex, maxIntersectRatio));
190 if(nearestPointIndex >= 0)
191 usedMask[nearestPointIndex] = 1;
192 }
193}

195static void removeVerySmallKeypoints(vector<KeyPoint>& keypoints)
196{
197 size_t i, j = 0, n = keypoints.size();
198 for( i = 0; i < n; i++ )
199 {
200 if( (keypoints[i].octave & 128) != 0 )
201 ;
202 else
203 keypoints[j++] = keypoints[i];
204 }
205 keypoints.resize(j);
206}

209class DetectorRotationInvarianceTest : public cvtest::BaseTest
210{
211public:
212 DetectorRotationInvarianceTest(const Ptr<FeatureDetector>& _featureDetector,
213 float _minKeyPointMatchesRatio,
214 float _minAngleInliersRatio) :
215 featureDetector(_featureDetector),
216 minKeyPointMatchesRatio(_minKeyPointMatchesRatio),
217 minAngleInliersRatio(_minAngleInliersRatio)
218 {
219 CV_Assert(!featureDetector.empty());
220 }
221
222protected:
223
224 void run(int)
225 {
226 const string imageFilename = string(ts->get_data_path()) + IMAGE_TSUKUBA;
227
228 // Read test data
229 Mat image0 = imread(imageFilename), image1, mask1;
230 if(image0.empty())
231 {
232 ts->printf(cvtest::TS::LOG, "Image %s can not be read.\n", imageFilename.c_str());
233 ts->set_failed_test_info(cvtest::TS::FAIL_INVALID_TEST_DATA);
234 return;
235 }
236
237 vector<KeyPoint> keypoints0;
238 featureDetector->detect(image0, keypoints0);
239 removeVerySmallKeypoints(keypoints0);
240 if(keypoints0.size() < 15)
241 CV_Error(CV_StsAssert, "Detector gives too few points in a test image\n");
242
243 const int maxAngle = 360, angleStep = 15;
244 for(int angle = 0; angle < maxAngle; angle += angleStep)
245 {
246 Mat H = rotateImage(image0, static_cast<float>(angle), image1, mask1);
247
248 vector<KeyPoint> keypoints1;
249 featureDetector->detect(image1, keypoints1, mask1);
250 removeVerySmallKeypoints(keypoints1);
251
252 vector<DMatch> matches;
253 matchKeyPoints(keypoints0, H, keypoints1, matches);
254
255 int angleInliersCount = 0;
256
257 const float minIntersectRatio = 0.5f;
258 int keyPointMatchesCount = 0;
259 for(size_t m = 0; m < matches.size(); m++)
260 {
261 if(matches[m].distance < minIntersectRatio)
262 continue;
263
264 keyPointMatchesCount++;
265
266 // Check does this inlier have consistent angles
267 const float maxAngleDiff = 15.f; // grad
268 float angle0 = keypoints0[matches[m].queryIdx].angle;
269 float angle1 = keypoints1[matches[m].trainIdx].angle;
270 if(angle0 == -1 || angle1 == -1)
271 CV_Error(CV_StsBadArg, "Given FeatureDetector is not rotation invariant, it can not be tested here.\n");
272 CV_Assert(angle0 >= 0.f && angle0 < 360.f);
273 CV_Assert(angle1 >= 0.f && angle1 < 360.f);
274
275 float rotAngle0 = angle0 + angle;
276 if(rotAngle0 >= 360.f)
277 rotAngle0 -= 360.f;
278
279 float angleDiff = std::max(rotAngle0, angle1) - std::min(rotAngle0, angle1);
280 angleDiff = std::min(angleDiff, static_cast<float>(360.f - angleDiff));
281 CV_Assert(angleDiff >= 0.f);
282 bool isAngleCorrect = angleDiff < maxAngleDiff;
283 if(isAngleCorrect)
284 angleInliersCount++;
285 }
286
287 float keyPointMatchesRatio = static_cast<float>(keyPointMatchesCount) / keypoints0.size();
288 if(keyPointMatchesRatio < minKeyPointMatchesRatio)
289 {
290 ts->printf(cvtest::TS::LOG, "Incorrect keyPointMatchesRatio: curr = %f, min = %f.\n",
291 keyPointMatchesRatio, minKeyPointMatchesRatio);
292 ts->set_failed_test_info(cvtest::TS::FAIL_BAD_ACCURACY);
293 return;
294 }
295
296 if(keyPointMatchesCount)
297 {
298 float angleInliersRatio = static_cast<float>(angleInliersCount) / keyPointMatchesCount;
299 if(angleInliersRatio < minAngleInliersRatio)
300 {
301 ts->printf(cvtest::TS::LOG, "Incorrect angleInliersRatio: curr = %f, min = %f.\n",
302 angleInliersRatio, minAngleInliersRatio);
303 ts->set_failed_test_info(cvtest::TS::FAIL_BAD_ACCURACY);
304 return;
305 }
306 }
307#if SHOW_DEBUG_LOG
308 std::cout << "keyPointMatchesRatio - " << keyPointMatchesRatio
309 << " - angleInliersRatio " << static_cast<float>(angleInliersCount) / keyPointMatchesCount << std::endl;
310#endif
311 }
312 ts->set_failed_test_info( cvtest::TS::OK );
313 }
314
315 Ptr<FeatureDetector> featureDetector;
316 float minKeyPointMatchesRatio;
317 float minAngleInliersRatio;

501 }
502

512 }
513 }
514#if SHOW_DEBUG_LOG

517#endif
518 }
