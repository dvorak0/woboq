128bool CV_FeatureDetectorTest::isSimilarKeypoints( const KeyPoint& p1, const KeyPoint& p2 )
129{
130 const float maxPtDif = 1.f;
131 const float maxSizeDif = 1.f;
132 const float maxAngleDif = 2.f;
133 const float maxResponseDif = 0.1f;
134
135 float dist = (float)norm( p1.pt - p2.pt );
136 return (dist < maxPtDif &&
137 fabs(p1.size - p2.size) < maxSizeDif &&
138 abs(p1.angle - p2.angle) < maxAngleDif &&
139 abs(p1.response - p2.response) < maxResponseDif &&
140 p1.octave == p2.octave &&
141 p1.class_id == p2.class_id );
142}

144void CV_FeatureDetectorTest::compareKeypointSets( const vector<KeyPoint>& validKeypoints, const vector<KeyPoint>& calcKeypoints )
145{
146 const float maxCountRatioDif = 0.01f;
147
148 // Compare counts of validation and calculated keypoints.
149 float countRatio = (float)validKeypoints.size() / (float)calcKeypoints.size();
150 if( countRatio < 1 - maxCountRatioDif || countRatio > 1.f + maxCountRatioDif )
151 {
152 ts->printf( cvtest::TS::LOG, "Bad keypoints count ratio (validCount = %d, calcCount = %d).\n",
153 validKeypoints.size(), calcKeypoints.size() );
154 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_OUTPUT );
155 return;
156 }
157
158 int progress = 0, progressCount = (int)(validKeypoints.size() * calcKeypoints.size());
159 int badPointCount = 0, commonPointCount = max((int)validKeypoints.size(), (int)calcKeypoints.size());
160 for( size_t v = 0; v < validKeypoints.size(); v++ )
161 {
162 int nearestIdx = -1;
163 float minDist = std::numeric_limits<float>::max();
164
165 for( size_t c = 0; c < calcKeypoints.size(); c++ )
166 {
167 progress = update_progress( progress, (int)(v*calcKeypoints.size() + c), progressCount, 0 );
168 float curDist = (float)norm( calcKeypoints[c].pt - validKeypoints[v].pt );
169 if( curDist < minDist )
170 {
171 minDist = curDist;
172 nearestIdx = (int)c;
173 }
174 }
175
176 assert( minDist >= 0 );
177 if( !isSimilarKeypoints( validKeypoints[v], calcKeypoints[nearestIdx] ) )
178 badPointCount++;
179 }
180 ts->printf( cvtest::TS::LOG, "badPointCount = %d; validPointCount = %d; calcPointCount = %d\n",
181 badPointCount, validKeypoints.size(), calcKeypoints.size() );
182 if( badPointCount > 0.9 * commonPointCount )
183 {
184 ts->printf( cvtest::TS::LOG, " - Bad accuracy!\n" );
185 ts->set_failed_test_info( cvtest::TS::FAIL_BAD_ACCURACY );
186 return;
187 }
188 ts->printf( cvtest::TS::LOG, " - OK\n" );
189}

191void CV_FeatureDetectorTest::regressionTest()
192{
193 assert( !fdetector.empty() );
194 string imgFilename = string(ts->get_data_path()) + FEATURES2D_DIR + "/" + IMAGE_FILENAME;
195 string resFilename = string(ts->get_data_path()) + DETECTOR_DIR + "/" + string(name) + ".xml.gz";
196
197 // Read the test image.
198 Mat image = imread( imgFilename );
199 if( image.empty() )
200 {
201 ts->printf( cvtest::TS::LOG, "Image %s can not be read.\n", imgFilename.c_str() );
202 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_TEST_DATA );
203 return;
204 }
205
206 FileStorage fs( resFilename, FileStorage::READ );
207
208 // Compute keypoints.
209 vector<KeyPoint> calcKeypoints;
210 fdetector->detect( image, calcKeypoints );
211
212 if( fs.isOpened() ) // Compare computed and valid keypoints.
213 {
214 // TODO compare saved feature detector params with current ones
215
216 // Read validation keypoints set.
217 vector<KeyPoint> validKeypoints;
218 read( fs["keypoints"], validKeypoints );
219 if( validKeypoints.empty() )
220 {
221 ts->printf( cvtest::TS::LOG, "Keypoints can not be read.\n" );
222 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_TEST_DATA );
223 return;
224 }
225
226 compareKeypointSets( validKeypoints, calcKeypoints );
227 }
228 else // Write detector parameters and computed keypoints as validation data.
229 {
230 fs.open( resFilename, FileStorage::WRITE );
231 if( !fs.isOpened() )
232 {
233 ts->printf( cvtest::TS::LOG, "File %s can not be opened to write.\n", resFilename.c_str() );
234 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_TEST_DATA );
235 return;
236 }
237 else
238 {
239 fs << "detector_params" << "{";
240 fdetector->write( fs );
241 fs << "}";
242
243 write( fs, "keypoints", calcKeypoints );
244 }
245 }

266static void writeMatInBin( const Mat& mat, const string& filename )
267{
268 FILE* f = fopen( filename.c_str(), "wb");
269 if( f )
270 {
271 int type = mat.type();
272 fwrite( (void*)&mat.rows, sizeof(int), 1, f );
273 fwrite( (void*)&mat.cols, sizeof(int), 1, f );
274 fwrite( (void*)&type, sizeof(int), 1, f );
275 int dataSize = (int)(mat.step * mat.rows * mat.channels());
276 fwrite( (void*)&dataSize, sizeof(int), 1, f );
277 fwrite( (void*)mat.data, 1, dataSize, f );
278 fclose(f);
279 }
280}
281
282static Mat readMatFromBin( const string& filename )
283{
284 FILE* f = fopen( filename.c_str(), "rb" );
285 if( f )
286 {
287 int rows, cols, type, dataSize;
288 size_t elements_read1 = fread( (void*)&rows, sizeof(int), 1, f );
289 size_t elements_read2 = fread( (void*)&cols, sizeof(int), 1, f );
290 size_t elements_read3 = fread( (void*)&type, sizeof(int), 1, f );
291 size_t elements_read4 = fread( (void*)&dataSize, sizeof(int), 1, f );
292 CV_Assert(elements_read1 == 1 && elements_read2 == 1 && elements_read3 == 1 && elements_read4 == 1);
293
294 uchar* data = (uchar*)cvAlloc(dataSize);
295 size_t elements_read = fread( (void*)data, 1, dataSize, f );
296 CV_Assert(elements_read == (size_t)(dataSize));
297 fclose(f);
298
299 return Mat( rows, cols, type, data );
300 }
301 return Mat();
302}

304template<class Distance>
305class CV_DescriptorExtractorTest : public cvtest::BaseTest
306{
307public:
308 typedef typename Distance::ValueType ValueType;
309 typedef typename Distance::ResultType DistanceType;
310
311 CV_DescriptorExtractorTest( const string _name, DistanceType _maxDist, const Ptr<DescriptorExtractor>& _dextractor,
312 Distance d = Distance() ):
313 name(_name), maxDist(_maxDist), dextractor(_dextractor), distance(d) {}
314protected:
315 virtual void createDescriptorExtractor() {}
316
317 void compareDescriptors( const Mat& validDescriptors, const Mat& calcDescriptors )
318 {
319 if( validDescriptors.size != calcDescriptors.size || validDescriptors.type() != calcDescriptors.type() )
320 {
321 ts->printf(cvtest::TS::LOG, "Valid and computed descriptors matrices must have the same size and type.\n");
322 ts->printf(cvtest::TS::LOG, "Valid size is (%d x %d) actual size is (%d x %d).\n", validDescriptors.rows, validDescriptors.cols, calcDescriptors.rows, calcDescriptors.cols);
323 ts->printf(cvtest::TS::LOG, "Valid type is %d actual type is %d.\n", validDescriptors.type(), calcDescriptors.type());
324 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_TEST_DATA );
325 return;
326 }
327
328 CV_Assert( DataType<ValueType>::type == validDescriptors.type() );
329
330 int dimension = validDescriptors.cols;
331 DistanceType curMaxDist = std::numeric_limits<DistanceType>::min();
332 for( int y = 0; y < validDescriptors.rows; y++ )
333 {
334 DistanceType dist = distance( validDescriptors.ptr<ValueType>(y), calcDescriptors.ptr<ValueType>(y), dimension );
335 if( dist > curMaxDist )
336 curMaxDist = dist;
337 }
338
339 stringstream ss;
340 ss << "Max distance between valid and computed descriptors " << curMaxDist;
341 if( curMaxDist < maxDist )
342 ss << "." << endl;
343 else
344 {
345 ss << ">" << maxDist << " - bad accuracy!"<< endl;
346 ts->set_failed_test_info( cvtest::TS::FAIL_BAD_ACCURACY );
347 }
348 ts->printf(cvtest::TS::LOG, ss.str().c_str() );
349 }

396 void regressionTest()
397 {
398 assert( !dextractor.empty() );
399
400 // Read the test image.
401 string imgFilename = string(ts->get_data_path()) + FEATURES2D_DIR + "/" + IMAGE_FILENAME;
402
403 Mat img = imread( imgFilename );
404 if( img.empty() )
405 {
406 ts->printf( cvtest::TS::LOG, "Image %s can not be read.\n", imgFilename.c_str() );
407 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_TEST_DATA );
408 return;
409 }
410
411 vector<KeyPoint> keypoints;
412 FileStorage fs( string(ts->get_data_path()) + FEATURES2D_DIR + "/keypoints.xml.gz", FileStorage::READ );
413 if( fs.isOpened() )
414 {
415 read( fs.getFirstTopLevelNode(), keypoints );
416
417 Mat calcDescriptors;
418 double t = (double)getTickCount();
419 dextractor->compute( img, keypoints, calcDescriptors );
420 t = getTickCount() - t;
421 ts->printf(cvtest::TS::LOG, "\nAverage time of computing one descriptor = %g ms.\n", t/((double)cvGetTickFrequency()*1000.)/calcDescriptors.rows );
422
423 if( calcDescriptors.rows != (int)keypoints.size() )
424 {
425 ts->printf( cvtest::TS::LOG, "Count of computed descriptors and keypoints count must be equal.\n" );
426 ts->printf( cvtest::TS::LOG, "Count of keypoints is %d.\n", (int)keypoints.size() );
427 ts->printf( cvtest::TS::LOG, "Count of computed descriptors is %d.\n", calcDescriptors.rows );
428 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_OUTPUT );
429 return;
430 }
431
432 if( calcDescriptors.cols != dextractor->descriptorSize() || calcDescriptors.type() != dextractor->descriptorType() )
433 {
434 ts->printf( cvtest::TS::LOG, "Incorrect descriptor size or descriptor type.\n" );
435 ts->printf( cvtest::TS::LOG, "Expected size is %d.\n", dextractor->descriptorSize() );
436 ts->printf( cvtest::TS::LOG, "Calculated size is %d.\n", calcDescriptors.cols );
437 ts->printf( cvtest::TS::LOG, "Expected type is %d.\n", dextractor->descriptorType() );
438 ts->printf( cvtest::TS::LOG, "Calculated type is %d.\n", calcDescriptors.type() );
439 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_OUTPUT );
440 return;
441 }
442
443 // TODO read and write descriptor extractor parameters and check them
444 Mat validDescriptors = readDescriptors();
445 if( !validDescriptors.empty() )
446 compareDescriptors( validDescriptors, calcDescriptors );
447 else
448 {
449 if( !writeDescriptors( calcDescriptors ) )
450 {
451 ts->printf( cvtest::TS::LOG, "Descriptors can not be written.\n" );
452 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_TEST_DATA );
453 return;
454 }
455 }
456 }
457 else
458 {
459 ts->printf( cvtest::TS::LOG, "Compute and write keypoints.\n" );
460 fs.open( string(ts->get_data_path()) + FEATURES2D_DIR + "/keypoints.xml.gz", FileStorage::WRITE );
461 if( fs.isOpened() )
462 {
463 SurfFeatureDetector fd;
464 fd.detect(img, keypoints);
465 write( fs, "keypoints", keypoints );
466 }
467 else
468 {
469 ts->printf(cvtest::TS::LOG, "File for writting keypoints can not be opened.\n");
470 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_TEST_DATA );
471 return;
472 }
473 }

640void CV_DescriptorMatcherTest::generateData( Mat& query, Mat& train )
641{
642 RNG& rng = theRNG();
643
644 // Generate query descriptors randomly.
645 // Descriptor vector elements are integer values.
646 Mat buf( queryDescCount, dim, CV_32SC1 );
647 rng.fill( buf, RNG::UNIFORM, Scalar::all(0), Scalar(3) );
648 buf.convertTo( query, CV_32FC1 );
649
650 // Generate train decriptors as follows:
651 // copy each query descriptor to train set countFactor times
652 // and perturb some one element of the copied descriptors in
653 // in ascending order. General boundaries of the perturbation
654 // are (0.f, 1.f).
655 train.create( query.rows*countFactor, query.cols, CV_32FC1 );
656 float step = 1.f / countFactor;
657 for( int qIdx = 0; qIdx < query.rows; qIdx++ )
658 {
659 Mat queryDescriptor = query.row(qIdx);
660 for( int c = 0; c < countFactor; c++ )
661 {
662 int tIdx = qIdx * countFactor + c;
663 Mat trainDescriptor = train.row(tIdx);
664 queryDescriptor.copyTo( trainDescriptor );
665 int elem = rng(dim);
666 float diff = rng.uniform( step*c, step*(c+1) );
667 trainDescriptor.at<float>(0, elem) += diff;
668 }
669 }
670}

672void CV_DescriptorMatcherTest::matchTest( const Mat& query, const Mat& train )
673{
674 dmatcher->clear();
675
676 // test const version of match()
677 {
678 vector<DMatch> matches;
679 dmatcher->match( query, train, matches );
680
681 if( (int)matches.size() != queryDescCount )
