73void getRotation(const Mat& img, Mat& aff, Mat& out)
74{
75 Point center(img.cols/2, img.rows/2);
76 aff = getRotationMatrix2D(center, 30, 1);
77 warpAffine( img, out, aff, img.size());
78}
79
80void getZoom(const Mat& img, Mat& aff, Mat& out)
81{
82 const double mult = 1.2;
83
84 aff.create(2, 3, CV_64F);
85 double *data = aff.ptr<double>();
86 data[0] = mult; data[1] = 0; data[2] = 0;
87 data[3] = 0; data[4] = mult; data[5] = 0;
88
89 warpAffine( img, out, aff, img.size());
90}
91
92void getBlur(const Mat& img, Mat& aff, Mat& out)
93{
94 aff.create(2, 3, CV_64F);
95 double *data = aff.ptr<double>();
96 data[0] = 1; data[1] = 0; data[2] = 0;
97 data[3] = 0; data[4] = 1; data[5] = 0;
98
99 GaussianBlur(img, out, Size(5, 5), 2);
100}
101
102void getBrightness(const Mat& img, Mat& aff, Mat& out)
103{
104 aff.create(2, 3, CV_64F);
105 double *data = aff.ptr<double>();
106 data[0] = 1; data[1] = 0; data[2] = 0;
107 data[3] = 0; data[4] = 1; data[5] = 0;
108
109 add(img, Mat(img.size(), img.type(), Scalar(15)), out);
110}

139struct WrapPoint
140{
141 const double* R;
142 WrapPoint(const Mat& rmat) : R(rmat.ptr<double>()) { };
143
144 KeyPoint operator()(const KeyPoint& kp) const
145 {
146 KeyPoint res = kp;
147 res.pt.x = static_cast<float>(kp.pt.x * R[0] + kp.pt.y * R[1] + R[2]);
148 res.pt.y = static_cast<float>(kp.pt.x * R[3] + kp.pt.y * R[4] + R[5]);
149 return res;
150 }
151};
152
153struct sortByR { bool operator()(const KeyPoint& kp1, const KeyPoint& kp2) { return norm(kp1.pt) < norm(kp2.pt); } };

155template <class T> bool CV_DetectorsTest::testDetector(const Mat& img, const T& detector, vector<KeyPoint>& exp)
156{
157 vector<KeyPoint> orig_kpts;
158 detector(img, orig_kpts);
159
160 typedef void (*TransfFunc )(const Mat&, Mat&, Mat& FransfFunc);
161 const TransfFunc transfFunc[] = { getRotation, getZoom, getBlur, getBrightness };
162 //const string names[] = { "Rotation", "Zoom", "Blur", "Brightness" };
163 const size_t case_num = sizeof(transfFunc)/sizeof(transfFunc[0]);
164
165 vector<Mat> affs(case_num);
166 vector<Mat> new_imgs(case_num);
167
168 vector< vector<KeyPoint> > new_kpts(case_num);
169 vector< vector<KeyPoint> > transf_kpts(case_num);
170
171 //showOrig(img, orig_kpts);
172 for(size_t i = 0; i < case_num; ++i)
173 {
174 transfFunc[i](img, affs[i], new_imgs[i]);
175 detector(new_imgs[i], new_kpts[i]);
176 transform(orig_kpts.begin(), orig_kpts.end(), back_inserter(transf_kpts[i]), WrapPoint(affs[i]));
177 //show(names[i], new_imgs[i], new_kpts[i], transf_kpts[i]);
178 }
179
180 const float thres = 3;
181 const float nthres = 3;
182
183 vector<KeyPoint> result;
184 for(size_t i = 0; i < orig_kpts.size(); ++i)
185 {
186 const KeyPoint& okp = orig_kpts[i];
187 int foundCounter = 0;
188 for(size_t j = 0; j < case_num; ++j)
189 {
190 const KeyPoint& tkp = transf_kpts[j][i];
191
192 size_t k = 0;
193
194 for(; k < new_kpts[j].size(); ++k)
195 if (norm(new_kpts[j][k].pt - tkp.pt) < nthres && fabs(new_kpts[j][k].size - tkp.size) < thres)
196 break;
197
198 if (k != new_kpts[j].size())
199 ++foundCounter;
200
201 }
202 if (foundCounter == (int)case_num)
203 result.push_back(okp);
204 }
205
206 sort(result.begin(), result.end(), sortByR());
207 sort(exp.begin(), exp.end(), sortByR());
208
209 if (result.size() != exp.size())
210 {
211 ts->set_failed_test_info(cvtest::TS::FAIL_INVALID_TEST_DATA);
212 return false;
213 }
214
215 int foundCounter1 = 0;
216 for(size_t i = 0; i < exp.size(); ++i)
217 {
218 const KeyPoint& e = exp[i];
219 size_t j = 0;
220 for(; j < result.size(); ++j)
221 {
222 const KeyPoint& r = result[i];
223 if (norm(r.pt-e.pt) < nthres && fabs(r.size - e.size) < thres)
224 break;
225 }
226 if (j != result.size())
227 ++foundCounter1;
228 }
229
230 int foundCounter2 = 0;
231 for(size_t i = 0; i < result.size(); ++i)
232 {
233 const KeyPoint& r = result[i];
234 size_t j = 0;
235 for(; j < exp.size(); ++j)
236 {
237 const KeyPoint& e = exp[i];
238 if (norm(r.pt-e.pt) < nthres && fabs(r.size - e.size) < thres)
239 break;
240 }
241 if (j != exp.size())
242 ++foundCounter2;
243 }
244 //showOrig(img, result); waitKey();
245
246 const float errorRate = 0.9f;
247 if (float(foundCounter1)/exp.size() < errorRate || float(foundCounter2)/result.size() < errorRate)
248 {
249 ts->set_failed_test_info( cvtest::TS::FAIL_MISMATCH);
250 return false;
251 }
252 return true;
253}

263void CV_DetectorsTest::LoadExpected(const string& file, vector<KeyPoint>& out)
264{
265 Mat mat_exp;
266 FileStorage fs(file, FileStorage::READ);
267 if (fs.isOpened())
268 {
269 read( fs["ResultVectorData"], mat_exp, Mat() );
270 out.resize(mat_exp.cols / sizeof(KeyPoint));
271 copy(mat_exp.ptr<KeyPoint>(), mat_exp.ptr<KeyPoint>() + out.size(), out.begin());
272 }
273 else
274 {
275 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_TEST_DATA);
276 out.clear();
277 }
278}

280void CV_DetectorsTest::run( int /*start_from*/ )
281{
282 Mat img = imread(string(ts->get_data_path()) + "shared/graffiti.png", 0);
283
284 if (img.empty())
285 {
286 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_TEST_DATA );
287 return;
288 }
289
290 Mat to_test(img.size() * 2, img.type(), Scalar(0));
291 Mat roi = to_test(Rect(img.rows/2, img.cols/2, img.cols, img.rows));
292 img.copyTo(roi);
293 GaussianBlur(to_test, to_test, Size(3, 3), 1.5);
294
295 vector<KeyPoint> exp;
296 LoadExpected(string(ts->get_data_path()) + "detectors/surf.xml", exp);
297 if (exp.empty())
298 return;
299
300 if (!testDetector(to_test, SurfNoMaskWrap(SURF(1536+512+512, 2)), exp))
301 return;
302
303 LoadExpected(string(ts->get_data_path()) + "detectors/star.xml", exp);
304 if (exp.empty())
305 return;
306
307 if (!testDetector(to_test, StarDetector(45, 30, 10, 8, 5), exp))
308 return;
309
310 ts->set_failed_test_info( cvtest::TS::OK);
311}
