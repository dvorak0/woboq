63PARAM_TEST_CASE(SURF, SURF_HessianThreshold, SURF_Octaves, SURF_OctaveLayers, SURF_Extended, SURF_Upright)
64{
65 double hessianThreshold;
66 int nOctaves;
67 int nOctaveLayers;
68 bool extended;
69 bool upright;
70
71 virtual void SetUp()
72 {
73 hessianThreshold = GET_PARAM(0);
74 nOctaves = GET_PARAM(1);
75 nOctaveLayers = GET_PARAM(2);
76 extended = GET_PARAM(3);
77 upright = GET_PARAM(4);
78 }
79};

81GPU_TEST_P(SURF, Detector)
82{
83 cv::Mat image = readImage("../gpu/features2d/aloe.png", cv::IMREAD_GRAYSCALE);
84 ASSERT_FALSE(image.empty());
85
86 cv::gpu::SURF_GPU surf;
87 surf.hessianThreshold = hessianThreshold;
88 surf.nOctaves = nOctaves;
89 surf.nOctaveLayers = nOctaveLayers;
90 surf.extended = extended;
91 surf.upright = upright;
92 surf.keypointsRatio = 0.05f;
93
94 std::vector<cv::KeyPoint> keypoints;
95 surf(loadMat(image), cv::gpu::GpuMat(), keypoints);
96
97 cv::SURF surf_gold;
98 surf_gold.hessianThreshold = hessianThreshold;
99 surf_gold.nOctaves = nOctaves;
100 surf_gold.nOctaveLayers = nOctaveLayers;
101 surf_gold.extended = extended;
102 surf_gold.upright = upright;
103
104 std::vector<cv::KeyPoint> keypoints_gold;
105 surf_gold(image, cv::noArray(), keypoints_gold);
106
107 ASSERT_EQ(keypoints_gold.size(), keypoints.size());
108 int matchedCount = getMatchedPointsCount(keypoints_gold, keypoints);
109 double matchedRatio = static_cast<double>(matchedCount) / keypoints_gold.size();
110
111 EXPECT_GT(matchedRatio, 0.95);
112}

114GPU_TEST_P(SURF, Detector_Masked)
115{

118

126 surf.extended = extended;
127 surf.upright = upright;
128 surf.keypointsRatio = 0.05f;
129
130 std::vector<cv::KeyPoint> keypoints;
131 surf(loadMat(image), loadMat(mask), keypoints);
132
133 cv::SURF surf_gold;
134 surf_gold.hessianThreshold = hessianThreshold;
135 surf_gold.nOctaves = nOctaves;
136 surf_gold.nOctaveLayers = nOctaveLayers;
137 surf_gold.extended = extended;
138 surf_gold.upright = upright;
139
140 std::vector<cv::KeyPoint> keypoints_gold;
141 surf_gold(image, mask, keypoints_gold);

148}

150GPU_TEST_P(SURF, Descriptor)
151{
152 cv::Mat image = readImage("../gpu/features2d/aloe.png", cv::IMREAD_GRAYSCALE);
153 ASSERT_FALSE(image.empty());
154
155 cv::gpu::SURF_GPU surf;
156 surf.hessianThreshold = hessianThreshold;
157 surf.nOctaves = nOctaves;
158 surf.nOctaveLayers = nOctaveLayers;
159 surf.extended = extended;
160 surf.upright = upright;
161 surf.keypointsRatio = 0.05f;
162
163 cv::SURF surf_gold;
164 surf_gold.hessianThreshold = hessianThreshold;
165 surf_gold.nOctaves = nOctaves;
166 surf_gold.nOctaveLayers = nOctaveLayers;
167 surf_gold.extended = extended;
168 surf_gold.upright = upright;
169
170 std::vector<cv::KeyPoint> keypoints;
171 surf_gold(image, cv::noArray(), keypoints);
172
173 cv::gpu::GpuMat descriptors;
174 surf(loadMat(image), cv::gpu::GpuMat(), keypoints, descriptors, true);
175
176 cv::Mat descriptors_gold;
177 surf_gold(image, cv::noArray(), keypoints, descriptors_gold, true);
178
179 cv::BFMatcher matcher(cv::NORM_L2);
180 std::vector<cv::DMatch> matches;
181 matcher.match(descriptors_gold, cv::Mat(descriptors), matches);
182
183 int matchedCount = getMatchedPointsCount(keypoints, keypoints, matches);
184 double matchedRatio = static_cast<double>(matchedCount) / keypoints.size();
185
186 EXPECT_GT(matchedRatio, 0.6);
187}
