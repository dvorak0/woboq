91void btv_l1_device_ocl::buildMotionMaps(const oclMat& forwardMotionX, const oclMat& forwardMotionY,
92 const oclMat& backwardMotionX, const oclMat& backwardMotionY,
93 oclMat& forwardMapX, oclMat& forwardMapY,
94 oclMat& backwardMapX, oclMat& backwardMapY)
95{
96 Context* clCxt = Context::getContext();
97
98 size_t local_thread[] = {32, 8, 1};
99 size_t global_thread[] = {(size_t)forwardMapX.cols, (size_t)forwardMapX.rows, 1};
100
101 int forwardMotionX_step = (int)(forwardMotionX.step/forwardMotionX.elemSize());
102 int forwardMotionY_step = (int)(forwardMotionY.step/forwardMotionY.elemSize());
103 int backwardMotionX_step = (int)(backwardMotionX.step/backwardMotionX.elemSize());
104 int backwardMotionY_step = (int)(backwardMotionY.step/backwardMotionY.elemSize());
105 int forwardMapX_step = (int)(forwardMapX.step/forwardMapX.elemSize());
106 int forwardMapY_step = (int)(forwardMapY.step/forwardMapY.elemSize());
107 int backwardMapX_step = (int)(backwardMapX.step/backwardMapX.elemSize());
108 int backwardMapY_step = (int)(backwardMapY.step/backwardMapY.elemSize());
109
110 String kernel_name = "buildMotionMapsKernel";
111 vector< pair<size_t, const void*> > args;
112
113 args.push_back(make_pair(sizeof(cl_mem), (void*)&forwardMotionX.data));
114 args.push_back(make_pair(sizeof(cl_mem), (void*)&forwardMotionY.data));
115 args.push_back(make_pair(sizeof(cl_mem), (void*)&backwardMotionX.data));
116 args.push_back(make_pair(sizeof(cl_mem), (void*)&backwardMotionY.data));
117 args.push_back(make_pair(sizeof(cl_mem), (void*)&forwardMapX.data));
118 args.push_back(make_pair(sizeof(cl_mem), (void*)&forwardMapY.data));
119 args.push_back(make_pair(sizeof(cl_mem), (void*)&backwardMapX.data));
120 args.push_back(make_pair(sizeof(cl_mem), (void*)&backwardMapY.data));
121
122 args.push_back(make_pair(sizeof(cl_int), (void*)&forwardMotionX.rows));
123 args.push_back(make_pair(sizeof(cl_int), (void*)&forwardMotionY.cols));
124
125 args.push_back(make_pair(sizeof(cl_int), (void*)&forwardMotionX_step));
126 args.push_back(make_pair(sizeof(cl_int), (void*)&forwardMotionY_step));
127 args.push_back(make_pair(sizeof(cl_int), (void*)&backwardMotionX_step));
128 args.push_back(make_pair(sizeof(cl_int), (void*)&backwardMotionY_step));
129 args.push_back(make_pair(sizeof(cl_int), (void*)&forwardMapX_step));
130 args.push_back(make_pair(sizeof(cl_int), (void*)&forwardMapY_step));
131 args.push_back(make_pair(sizeof(cl_int), (void*)&backwardMapX_step));
132 args.push_back(make_pair(sizeof(cl_int), (void*)&backwardMapY_step));

134 openCLExecuteKernel(clCxt, &superres_btvl1, kernel_name, global_thread, local_thread, args, -1, -1);

229 void calcRelativeMotions(const vector<pair<oclMat, oclMat> >& forwardMotions, const vector<pair<oclMat, oclMat> >& backwardMotions,
230 vector<pair<oclMat, oclMat> >& relForwardMotions, vector<pair<oclMat, oclMat> >& relBackwardMotions,
231 int baseIdx, Size size)
232 {
233 const int count = static_cast<int>(forwardMotions.size());
234
235 relForwardMotions.resize(count);
236 relForwardMotions[baseIdx].first.create(size, CV_32FC1);
237 relForwardMotions[baseIdx].first.setTo(Scalar::all(0));
238 relForwardMotions[baseIdx].second.create(size, CV_32FC1);
239 relForwardMotions[baseIdx].second.setTo(Scalar::all(0));
240
241 relBackwardMotions.resize(count);
242 relBackwardMotions[baseIdx].first.create(size, CV_32FC1);
243 relBackwardMotions[baseIdx].first.setTo(Scalar::all(0));
244 relBackwardMotions[baseIdx].second.create(size, CV_32FC1);
245 relBackwardMotions[baseIdx].second.setTo(Scalar::all(0));
246
247 for (int i = baseIdx - 1; i >= 0; --i)
248 {
249 ocl::add(relForwardMotions[i + 1].first, forwardMotions[i].first, relForwardMotions[i].first);
250 ocl::add(relForwardMotions[i + 1].second, forwardMotions[i].second, relForwardMotions[i].second);
251
252 ocl::add(relBackwardMotions[i + 1].first, backwardMotions[i + 1].first, relBackwardMotions[i].first);
253 ocl::add(relBackwardMotions[i + 1].second, backwardMotions[i + 1].second, relBackwardMotions[i].second);
254 }
255
256 for (int i = baseIdx + 1; i < count; ++i)
257 {
258 ocl::add(relForwardMotions[i - 1].first, backwardMotions[i].first, relForwardMotions[i].first);
259 ocl::add(relForwardMotions[i - 1].second, backwardMotions[i].second, relForwardMotions[i].second);
260
261 ocl::add(relBackwardMotions[i - 1].first, forwardMotions[i - 1].first, relBackwardMotions[i].first);
262 ocl::add(relBackwardMotions[i - 1].second, forwardMotions[i - 1].second, relBackwardMotions[i].second);
263 }
264 }

266 void upscaleMotions(const vector<pair<oclMat, oclMat> >& lowResMotions, vector<pair<oclMat, oclMat> >& highResMotions, int scale)
267 {
268 highResMotions.resize(lowResMotions.size());
269
270 for (size_t i = 0; i < lowResMotions.size(); ++i)
271 {
272 ocl::resize(lowResMotions[i].first, highResMotions[i].first, Size(), scale, scale, INTER_LINEAR);
273 ocl::resize(lowResMotions[i].second, highResMotions[i].second, Size(), scale, scale, INTER_LINEAR);
274
275 ocl::multiply(scale, highResMotions[i].first, highResMotions[i].first);
276 ocl::multiply(scale, highResMotions[i].second, highResMotions[i].second);
277 }

309 void calcBtvWeights(int btvKernelSize, double alpha, vector<float>& btvWeights)
310 {
311 const size_t size = btvKernelSize * btvKernelSize;
312
313 btvWeights.resize(size);
314
315 const int ksize = (btvKernelSize - 1) / 2;
316 const float alpha_f = static_cast<float>(alpha);
317
318 for (int m = 0, ind = 0; m <= ksize; ++m)
319 {
320 for (int l = ksize; l + m >= 0; --l, ++ind)
321 btvWeights[ind] = pow(alpha_f, std::abs(m) + std::abs(l));
322 }
323
324 btvWeights_ = &btvWeights[0];
325 btvWeights_size = size;
326 Mat btvWeights_mheader(1, static_cast<int>(size), CV_32FC1, btvWeights_);
327 c_btvRegWeights = btvWeights_mheader;
328 }

330 void calcBtvRegularization(const oclMat& src, oclMat& dst, int btvKernelSize)
331 {
332 dst.create(src.size(), src.type());
333
334 const int ksize = (btvKernelSize - 1) / 2;
335
336 btv_l1_device_ocl::calcBtvRegularization(src, dst, ksize);
337 }

407 void BTVL1_OCL_Base::process(const vector<oclMat>& src, oclMat& dst,
408 const vector<pair<oclMat, oclMat> >& forwardMotions, const vector<pair<oclMat, oclMat> >& backwardMotions,
409 int baseIdx)
410 {
411 CV_Assert( scale_ > 1 );
412 CV_Assert( iterations_ > 0 );
413 CV_Assert( tau_ > 0.0 );
414 CV_Assert( alpha_ > 0.0 );
415 CV_Assert( btvKernelSize_ > 0 && btvKernelSize_ <= 16 );
416 CV_Assert( blurKernelSize_ > 0 );
417 CV_Assert( blurSigma_ >= 0.0 );
418
419 // update blur filter and btv weights
420
421 if (filters_.size() != src.size() || blurKernelSize_ != curBlurKernelSize_ || blurSigma_ != curBlurSigma_ || src[0].type() != curSrcType_)
422 {
423 filters_.resize(src.size());
424 for (size_t i = 0; i < src.size(); ++i)
425 filters_[i] = cv::ocl::createGaussianFilter_GPU(src[0].type(), Size(blurKernelSize_, blurKernelSize_), blurSigma_);
426 curBlurKernelSize_ = blurKernelSize_;
427 curBlurSigma_ = blurSigma_;
428 curSrcType_ = src[0].type();
429 }
430
431 if (btvWeights_.empty() || btvKernelSize_ != curBtvKernelSize_ || alpha_ != curAlpha_)
432 {
433 calcBtvWeights(btvKernelSize_, alpha_, btvWeights_);
434 curBtvKernelSize_ = btvKernelSize_;
435 curAlpha_ = alpha_;
436 }
437
438 // calc motions between input frames
439
440 calcRelativeMotions(forwardMotions, backwardMotions,
441 lowResForwardMotions_, lowResBackwardMotions_,
442 baseIdx, src[0].size());
443
444 upscaleMotions(lowResForwardMotions_, highResForwardMotions_, scale_);
445 upscaleMotions(lowResBackwardMotions_, highResBackwardMotions_, scale_);
446
447 forwardMaps_.resize(highResForwardMotions_.size());
448 backwardMaps_.resize(highResForwardMotions_.size());
449 for (size_t i = 0; i < highResForwardMotions_.size(); ++i)
450 {
451 buildMotionMaps(highResForwardMotions_[i], highResBackwardMotions_[i], forwardMaps_[i], backwardMaps_[i]);
452 }
453 // initial estimation
454
455 const Size lowResSize = src[0].size();
456 const Size highResSize(lowResSize.width * scale_, lowResSize.height * scale_);
457
458 ocl::resize(src[baseIdx], highRes_, highResSize, 0, 0, INTER_LINEAR);
459
460 // iterations
461
462 diffTerms_.resize(src.size());
463 bool d_inited = false;
464 a_.create(highRes_.size(), highRes_.type());
465 b_.create(highRes_.size(), highRes_.type());
466 c_.create(lowResSize, highRes_.type());
467 d_.create(highRes_.rows, highRes_.cols, highRes_.type());
468 for (int i = 0; i < iterations_; ++i)
469 {
470 if(!d_inited)
471 {
472 d_.setTo(0);
473 d_inited = true;
474 }
475 for (size_t k = 0; k < src.size(); ++k)
476 {
477 diffTerms_[k].create(highRes_.size(), highRes_.type());
478 // a = M * Ih
479 ocl::remap(highRes_, a_, backwardMaps_[k].first, backwardMaps_[k].second, INTER_NEAREST, BORDER_CONSTANT, Scalar());
480 // b = HM * Ih
481 filters_[k]->apply(a_, b_, Rect(0,0,-1,-1));
482 // c = DHF * Ih
483 ocl::resize(b_, c_, lowResSize, 0, 0, INTER_NEAREST);
484
485 diffSign(src[k], c_, c_);
486
487 // a = Dt * diff
488 upscale(c_, d_, scale_);
489 // b = HtDt * diff
490 filters_[k]->apply(d_, b_, Rect(0,0,-1,-1));
491 // diffTerm = MtHtDt * diff
492 ocl::remap(b_, diffTerms_[k], forwardMaps_[k].first, forwardMaps_[k].second, INTER_NEAREST, BORDER_CONSTANT, Scalar());
493 }
494
495 if (lambda_ > 0)
496 {
497 calcBtvRegularization(highRes_, regTerm_, btvKernelSize_);
498 ocl::addWeighted(highRes_, 1.0, regTerm_, -tau_ * lambda_, 0.0, highRes_);
499 }
500
501 for (size_t k = 0; k < src.size(); ++k)
502 {
503 ocl::addWeighted(highRes_, 1.0, diffTerms_[k], tau_, 0.0, highRes_);
504 }
505 }
506
507 Rect inner(btvKernelSize_, btvKernelSize_, highRes_.cols - 2 * btvKernelSize_, highRes_.rows - 2 * btvKernelSize_);
508 highRes_(inner).copyTo(dst);
509 }

609 void BTVL1_OCL::initImpl(Ptr<FrameSource>& frameSource)
610 {
611 const int cacheSize = 2 * temporalAreaRadius_ + 1;
612
613 frames_.resize(cacheSize);
614 forwardMotions_.resize(cacheSize);
615 backwardMotions_.resize(cacheSize);
616 outputs_.resize(cacheSize);
617
618 storePos_ = -1;
619
620 for (int t = -temporalAreaRadius_; t <= temporalAreaRadius_; ++t)
621 readNextFrame(frameSource);
622
623 for (int i = 0; i <= temporalAreaRadius_; ++i)
624 processFrame(i);
625
626 procPos_ = temporalAreaRadius_;
627 outPos_ = -1;
628 }
629
630 void BTVL1_OCL::processImpl(Ptr<FrameSource>& frameSource, OutputArray _output)
631 {
632 if (outPos_ >= storePos_)
633 {
634 if(_output.kind() == _InputArray::OCL_MAT)
635 {
636 getOclMatRef(_output).release();
637 }
638 else
639 {
640 _output.release();
641 }
642 return;
643 }
644
645 readNextFrame(frameSource);
646
647 if (procPos_ < storePos_)
648 {
649 ++procPos_;
650 processFrame(procPos_);
651 }
652
653 ++outPos_;
654 const oclMat& curOutput = at(outPos_, outputs_);
655
656 if (_output.kind() == _InputArray::OCL_MAT)
657 curOutput.convertTo(getOclMatRef(_output), CV_8U);
658 else
659 {
660 curOutput.convertTo(finalOutput_, CV_8U);
661 arrCopy(finalOutput_, _output);
662 }
663 }
664
665 void BTVL1_OCL::readNextFrame(Ptr<FrameSource>& frameSource)
666 {
667 curFrame_.release();
668 frameSource->nextFrame(curFrame_);
669
670 if (curFrame_.empty())
671 return;
672
673 ++storePos_;
674 curFrame_.convertTo(at(storePos_, frames_), CV_32F);
675
676 if (storePos_ > 0)
677 {
678 pair<oclMat, oclMat>& forwardMotion = at(storePos_ - 1, forwardMotions_);
679 pair<oclMat, oclMat>& backwardMotion = at(storePos_, backwardMotions_);
680
681 opticalFlow_->calc(prevFrame_, curFrame_, forwardMotion.first, forwardMotion.second);
682 opticalFlow_->calc(curFrame_, prevFrame_, backwardMotion.first, backwardMotion.second);
683 }
684
685 curFrame_.copyTo(prevFrame_);
686 }
687
688 void BTVL1_OCL::processFrame(int idx)
689 {
690 const int startIdx = max(idx - temporalAreaRadius_, 0);
691 const int procIdx = idx;
692 const int endIdx = min(startIdx + 2 * temporalAreaRadius_, storePos_);
693
694 const int count = endIdx - startIdx + 1;
695
696 srcFrames_.resize(count);
697 srcForwardMotions_.resize(count);
698 srcBackwardMotions_.resize(count);
699
700 int baseIdx = -1;
701
702 for (int i = startIdx, k = 0; i <= endIdx; ++i, ++k)
703 {
704 if (i == procIdx)
705 baseIdx = k;
706
707 srcFrames_[k] = at(i, frames_);
708
709 if (i < endIdx)
710 srcForwardMotions_[k] = at(i, forwardMotions_);
711 if (i > startIdx)
712 srcBackwardMotions_[k] = at(i, backwardMotions_);
713 }
714
715 process(srcFrames_, at(idx, outputs_), srcForwardMotions_, srcBackwardMotions_, baseIdx);
716 }
