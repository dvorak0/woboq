82 void calcRelativeMotions(const vector<pair<GpuMat, GpuMat> >& forwardMotions, const vector<pair<GpuMat, GpuMat> >& backwardMotions,
83 vector<pair<GpuMat, GpuMat> >& relForwardMotions, vector<pair<GpuMat, GpuMat> >& relBackwardMotions,
84 int baseIdx, Size size)
85 {
86 const int count = static_cast<int>(forwardMotions.size());
87
88 relForwardMotions.resize(count);
89 relForwardMotions[baseIdx].first.create(size, CV_32FC1);
90 relForwardMotions[baseIdx].first.setTo(Scalar::all(0));
91 relForwardMotions[baseIdx].second.create(size, CV_32FC1);
92 relForwardMotions[baseIdx].second.setTo(Scalar::all(0));
93
94 relBackwardMotions.resize(count);
95 relBackwardMotions[baseIdx].first.create(size, CV_32FC1);
96 relBackwardMotions[baseIdx].first.setTo(Scalar::all(0));
97 relBackwardMotions[baseIdx].second.create(size, CV_32FC1);
98 relBackwardMotions[baseIdx].second.setTo(Scalar::all(0));
99
100 for (int i = baseIdx - 1; i >= 0; --i)
101 {
102 gpu::add(relForwardMotions[i + 1].first, forwardMotions[i].first, relForwardMotions[i].first);
103 gpu::add(relForwardMotions[i + 1].second, forwardMotions[i].second, relForwardMotions[i].second);
104
105 gpu::add(relBackwardMotions[i + 1].first, backwardMotions[i + 1].first, relBackwardMotions[i].first);
106 gpu::add(relBackwardMotions[i + 1].second, backwardMotions[i + 1].second, relBackwardMotions[i].second);
107 }
108
109 for (int i = baseIdx + 1; i < count; ++i)
110 {
111 gpu::add(relForwardMotions[i - 1].first, backwardMotions[i].first, relForwardMotions[i].first);
112 gpu::add(relForwardMotions[i - 1].second, backwardMotions[i].second, relForwardMotions[i].second);
113
114 gpu::add(relBackwardMotions[i - 1].first, forwardMotions[i - 1].first, relBackwardMotions[i].first);
115 gpu::add(relBackwardMotions[i - 1].second, forwardMotions[i - 1].second, relBackwardMotions[i].second);
116 }
117 }

119 void upscaleMotions(const vector<pair<GpuMat, GpuMat> >& lowResMotions, vector<pair<GpuMat, GpuMat> >& highResMotions, int scale)
120 {
121 highResMotions.resize(lowResMotions.size());
122
123 for (size_t i = 0; i < lowResMotions.size(); ++i)
124 {
125 gpu::resize(lowResMotions[i].first, highResMotions[i].first, Size(), scale, scale, INTER_CUBIC);
126 gpu::resize(lowResMotions[i].second, highResMotions[i].second, Size(), scale, scale, INTER_CUBIC);
127
128 gpu::multiply(highResMotions[i].first, Scalar::all(scale), highResMotions[i].first);
129 gpu::multiply(highResMotions[i].second, Scalar::all(scale), highResMotions[i].second);
130 }
131 }

148 void upscale(const GpuMat& src, GpuMat& dst, int scale, Stream& stream)
149 {
150 typedef void (*func_t)(const PtrStepSzb src, PtrStepSzb dst, int scale, cudaStream_t stream);
151 static const func_t funcs[] =
152 {
153 0, btv_l1_device::upscale<1>, 0, btv_l1_device::upscale<3>, btv_l1_device::upscale<4>
154 };
155
156 CV_Assert( src.channels() == 1 || src.channels() == 3 || src.channels() == 4 );
157
158 dst.create(src.rows * scale, src.cols * scale, src.type());
159 dst.setTo(Scalar::all(0));
160
161 const func_t func = funcs[src.channels()];
162
163 func(src, dst, scale, StreamAccessor::getStream(stream));
164 }

173 void calcBtvWeights(int btvKernelSize, double alpha, vector<float>& btvWeights)
174 {
175 const size_t size = btvKernelSize * btvKernelSize;
176
177 btvWeights.resize(size);
178
179 const int ksize = (btvKernelSize - 1) / 2;
180 const float alpha_f = static_cast<float>(alpha);
181
182 for (int m = 0, ind = 0; m <= ksize; ++m)
183 {
184 for (int l = ksize; l + m >= 0; --l, ++ind)
185 btvWeights[ind] = pow(alpha_f, std::abs(m) + std::abs(l));
186 }
187
188 btv_l1_device::loadBtvWeights(&btvWeights[0], size);
189 }

191 void calcBtvRegularization(const GpuMat& src, GpuMat& dst, int btvKernelSize)
192 {
193 typedef void (*func_t)(PtrStepSzb src, PtrStepSzb dst, int ksize);
194 static const func_t funcs[] =
195 {
196 0,
197 btv_l1_device::calcBtvRegularization<1>,
198 0,
199 btv_l1_device::calcBtvRegularization<3>,
200 btv_l1_device::calcBtvRegularization<4>
201 };
202
203 dst.create(src.size(), src.type());
204 dst.setTo(Scalar::all(0));
205
206 const int ksize = (btvKernelSize - 1) / 2;
207
208 funcs[src.channels()](src, dst, ksize);
209 }

280 void BTVL1_GPU_Base::process(const vector<GpuMat>& src, GpuMat& dst,
281 const vector<pair<GpuMat, GpuMat> >& forwardMotions, const vector<pair<GpuMat, GpuMat> >& backwardMotions,
282 int baseIdx)
283 {
284 CV_Assert( scale_ > 1 );
285 CV_Assert( iterations_ > 0 );
286 CV_Assert( tau_ > 0.0 );
287 CV_Assert( alpha_ > 0.0 );
288 CV_Assert( btvKernelSize_ > 0 && btvKernelSize_ <= 16 );
289 CV_Assert( blurKernelSize_ > 0 );
290 CV_Assert( blurSigma_ >= 0.0 );
291
292 // update blur filter and btv weights
293
294 if (filters_.size() != src.size() || blurKernelSize_ != curBlurKernelSize_ || blurSigma_ != curBlurSigma_ || src[0].type() != curSrcType_)
295 {
296 filters_.resize(src.size());
297 for (size_t i = 0; i < src.size(); ++i)
298 filters_[i] = createGaussianFilter_GPU(src[0].type(), Size(blurKernelSize_, blurKernelSize_), blurSigma_);
299 curBlurKernelSize_ = blurKernelSize_;
300 curBlurSigma_ = blurSigma_;
301 curSrcType_ = src[0].type();
302 }
303
304 if (btvWeights_.empty() || btvKernelSize_ != curBtvKernelSize_ || alpha_ != curAlpha_)
305 {
306 calcBtvWeights(btvKernelSize_, alpha_, btvWeights_);
307 curBtvKernelSize_ = btvKernelSize_;
308 curAlpha_ = alpha_;
309 }
310
311 // calc motions between input frames
312
313 calcRelativeMotions(forwardMotions, backwardMotions, lowResForwardMotions_, lowResBackwardMotions_, baseIdx, src[0].size());
314
315 upscaleMotions(lowResForwardMotions_, highResForwardMotions_, scale_);
316 upscaleMotions(lowResBackwardMotions_, highResBackwardMotions_, scale_);
317
318 forwardMaps_.resize(highResForwardMotions_.size());
319 backwardMaps_.resize(highResForwardMotions_.size());
320 for (size_t i = 0; i < highResForwardMotions_.size(); ++i)
321 buildMotionMaps(highResForwardMotions_[i], highResBackwardMotions_[i], forwardMaps_[i], backwardMaps_[i]);
322
323 // initial estimation
324
325 const Size lowResSize = src[0].size();
326 const Size highResSize(lowResSize.width * scale_, lowResSize.height * scale_);
327
328 gpu::resize(src[baseIdx], highRes_, highResSize, 0, 0, INTER_CUBIC);
329
330 // iterations
331
332 streams_.resize(src.size());
333 diffTerms_.resize(src.size());
334 a_.resize(src.size());
335 b_.resize(src.size());
336 c_.resize(src.size());
337
338 for (int i = 0; i < iterations_; ++i)
339 {
340 for (size_t k = 0; k < src.size(); ++k)
341 {
342 // a = M * Ih
343 gpu::remap(highRes_, a_[k], backwardMaps_[k].first, backwardMaps_[k].second, INTER_NEAREST, BORDER_REPLICATE, Scalar(), streams_[k]);
344 // b = HM * Ih
345 filters_[k]->apply(a_[k], b_[k], Rect(0,0,-1,-1), streams_[k]);
346 // c = DHF * Ih
347 gpu::resize(b_[k], c_[k], lowResSize, 0, 0, INTER_NEAREST, streams_[k]);
348
349 diffSign(src[k], c_[k], c_[k], streams_[k]);
350
351 // a = Dt * diff
352 upscale(c_[k], a_[k], scale_, streams_[k]);
353 // b = HtDt * diff
354 filters_[k]->apply(a_[k], b_[k], Rect(0,0,-1,-1), streams_[k]);
355 // diffTerm = MtHtDt * diff
356 gpu::remap(b_[k], diffTerms_[k], forwardMaps_[k].first, forwardMaps_[k].second, INTER_NEAREST, BORDER_REPLICATE, Scalar(), streams_[k]);
357 }
358
359 if (lambda_ > 0)
360 {
361 calcBtvRegularization(highRes_, regTerm_, btvKernelSize_);
362 gpu::addWeighted(highRes_, 1.0, regTerm_, -tau_ * lambda_, 0.0, highRes_);
363 }
364
365 for (size_t k = 0; k < src.size(); ++k)
366 {
367 streams_[k].waitForCompletion();
368 gpu::addWeighted(highRes_, 1.0, diffTerms_[k], tau_, 0.0, highRes_);
369 }
370 }
371
372 Rect inner(btvKernelSize_, btvKernelSize_, highRes_.cols - 2 * btvKernelSize_, highRes_.rows - 2 * btvKernelSize_);
373 highRes_(inner).copyTo(dst);
374 }

473 void BTVL1_GPU::initImpl(Ptr<FrameSource>& frameSource)
474 {
475 const int cacheSize = 2 * temporalAreaRadius_ + 1;
476
477 frames_.resize(cacheSize);
478 forwardMotions_.resize(cacheSize);
479 backwardMotions_.resize(cacheSize);
480 outputs_.resize(cacheSize);
481
482 storePos_ = -1;
483
484 for (int t = -temporalAreaRadius_; t <= temporalAreaRadius_; ++t)
485 readNextFrame(frameSource);
486
487 for (int i = 0; i <= temporalAreaRadius_; ++i)
488 processFrame(i);
489
490 procPos_ = temporalAreaRadius_;
491 outPos_ = -1;
492 }
493
494 void BTVL1_GPU::processImpl(Ptr<FrameSource>& frameSource, OutputArray _output)
495 {
496 if (outPos_ >= storePos_)
497 {
498 _output.release();
499 return;
500 }
501
502 readNextFrame(frameSource);
503
504 if (procPos_ < storePos_)
505 {
506 ++procPos_;
507 processFrame(procPos_);
508 }
509
510 ++outPos_;
511 const GpuMat& curOutput = at(outPos_, outputs_);
512
513 if (_output.kind() == _InputArray::GPU_MAT)
514 curOutput.convertTo(_output.getGpuMatRef(), CV_8U);
515 else
516 {
517 curOutput.convertTo(finalOutput_, CV_8U);
518 arrCopy(finalOutput_, _output);
519 }
520 }
521
522 void BTVL1_GPU::readNextFrame(Ptr<FrameSource>& frameSource)
523 {
524 frameSource->nextFrame(curFrame_);
525
526 if (curFrame_.empty())
527 return;
528
529 ++storePos_;
530 curFrame_.convertTo(at(storePos_, frames_), CV_32F);
531
532 if (storePos_ > 0)
533 {
534 pair<GpuMat, GpuMat>& forwardMotion = at(storePos_ - 1, forwardMotions_);
535 pair<GpuMat, GpuMat>& backwardMotion = at(storePos_, backwardMotions_);
536
537 opticalFlow_->calc(prevFrame_, curFrame_, forwardMotion.first, forwardMotion.second);
538 opticalFlow_->calc(curFrame_, prevFrame_, backwardMotion.first, backwardMotion.second);
539 }
540
541 curFrame_.copyTo(prevFrame_);
542 }
543
544 void BTVL1_GPU::processFrame(int idx)
545 {
546 const int startIdx = max(idx - temporalAreaRadius_, 0);
547 const int procIdx = idx;
548 const int endIdx = min(startIdx + 2 * temporalAreaRadius_, storePos_);
549
550 const int count = endIdx - startIdx + 1;
551
552 srcFrames_.resize(count);
553 srcForwardMotions_.resize(count);
554 srcBackwardMotions_.resize(count);
555
556 int baseIdx = -1;
557
558 for (int i = startIdx, k = 0; i <= endIdx; ++i, ++k)
559 {
560 if (i == procIdx)
561 baseIdx = k;
562
563 srcFrames_[k] = at(i, frames_);
564
565 if (i < endIdx)
566 srcForwardMotions_[k] = at(i, forwardMotions_);
567 if (i > startIdx)
568 srcBackwardMotions_[k] = at(i, backwardMotions_);
569 }
570
571 process(srcFrames_, at(idx, outputs_), srcForwardMotions_, srcBackwardMotions_, baseIdx);
572 }
