55 void calcRelativeMotions(const vector<Mat>& forwardMotions, const vector<Mat>& backwardMotions,
56 vector<Mat>& relForwardMotions, vector<Mat>& relBackwardMotions,
57 int baseIdx, Size size)
58 {
59 const int count = static_cast<int>(forwardMotions.size());
60
61 relForwardMotions.resize(count);
62 relForwardMotions[baseIdx].create(size, CV_32FC2);
63 relForwardMotions[baseIdx].setTo(Scalar::all(0));
64
65 relBackwardMotions.resize(count);
66 relBackwardMotions[baseIdx].create(size, CV_32FC2);
67 relBackwardMotions[baseIdx].setTo(Scalar::all(0));
68
69 for (int i = baseIdx - 1; i >= 0; --i)
70 {
71 add(relForwardMotions[i + 1], forwardMotions[i], relForwardMotions[i]);
72
73 add(relBackwardMotions[i + 1], backwardMotions[i + 1], relBackwardMotions[i]);
74 }
75
76 for (int i = baseIdx + 1; i < count; ++i)
77 {
78 add(relForwardMotions[i - 1], backwardMotions[i], relForwardMotions[i]);
79
80 add(relBackwardMotions[i - 1], forwardMotions[i - 1], relBackwardMotions[i]);
81 }
82 }

84 void upscaleMotions(const vector<Mat>& lowResMotions, vector<Mat>& highResMotions, int scale)
85 {
86 highResMotions.resize(lowResMotions.size());
87
88 for (size_t i = 0; i < lowResMotions.size(); ++i)
89 {
90 resize(lowResMotions[i], highResMotions[i], Size(), scale, scale, INTER_CUBIC);
91 multiply(highResMotions[i], Scalar::all(scale), highResMotions[i]);
92 }
93 }

95 void buildMotionMaps(const Mat& forwardMotion, const Mat& backwardMotion, Mat& forwardMap, Mat& backwardMap)
96 {
97 forwardMap.create(forwardMotion.size(), CV_32FC2);
98 backwardMap.create(forwardMotion.size(), CV_32FC2);
99
100 for (int y = 0; y < forwardMotion.rows; ++y)
101 {
102 const Point2f* forwardMotionRow = forwardMotion.ptr<Point2f>(y);
103 const Point2f* backwardMotionRow = backwardMotion.ptr<Point2f>(y);
104 Point2f* forwardMapRow = forwardMap.ptr<Point2f>(y);
105 Point2f* backwardMapRow = backwardMap.ptr<Point2f>(y);
106
107 for (int x = 0; x < forwardMotion.cols; ++x)
108 {
109 Point2f base(static_cast<float>(x), static_cast<float>(y));
110
111 forwardMapRow[x] = base + backwardMotionRow[x];
112 backwardMapRow[x] = base + forwardMotionRow[x];
113 }
114 }
115 }

117 template <typename T>
118 void upscaleImpl(const Mat& src, Mat& dst, int scale)
119 {
120 dst.create(src.rows * scale, src.cols * scale, src.type());
121 dst.setTo(Scalar::all(0));
122
123 for (int y = 0, Y = 0; y < src.rows; ++y, Y += scale)
124 {
125 const T* srcRow = src.ptr<T>(y);
126 T* dstRow = dst.ptr<T>(Y);
127
128 for (int x = 0, X = 0; x < src.cols; ++x, X += scale)
129 dstRow[X] = srcRow[x];
130 }
131 }

133 void upscale(const Mat& src, Mat& dst, int scale)
134 {
135 typedef void (*func_t)(const Mat& src, Mat& dst, int scale);
136 static const func_t funcs[] =
137 {
138 0, upscaleImpl<float>, 0, upscaleImpl<Point3f>
139 };
140
141 CV_Assert( src.channels() == 1 || src.channels() == 3 || src.channels() == 4 );
142
143 const func_t func = funcs[src.channels()];
144
145 func(src, dst, scale);
146 }

148 float diffSign(float a, float b)
149 {
150 return a > b ? 1.0f : a < b ? -1.0f : 0.0f;
151 }
152 Point3f diffSign(Point3f a, Point3f b)
153 {
154 return Point3f(
155 a.x > b.x ? 1.0f : a.x < b.x ? -1.0f : 0.0f,
156 a.y > b.y ? 1.0f : a.y < b.y ? -1.0f : 0.0f,
157 a.z > b.z ? 1.0f : a.z < b.z ? -1.0f : 0.0f
158 );
159 }
160
161 void diffSign(const Mat& src1, const Mat& src2, Mat& dst)
162 {
163 const int count = src1.cols * src1.channels();
164
165 dst.create(src1.size(), src1.type());
166
167 for (int y = 0; y < src1.rows; ++y)
168 {
169 const float* src1Ptr = src1.ptr<float>(y);
170 const float* src2Ptr = src2.ptr<float>(y);
171 float* dstPtr = dst.ptr<float>(y);
172
173 for (int x = 0; x < count; ++x)
174 dstPtr[x] = diffSign(src1Ptr[x], src2Ptr[x]);
175 }
176 }

178 void calcBtvWeights(int btvKernelSize, double alpha, vector<float>& btvWeights)
179 {
180 const size_t size = btvKernelSize * btvKernelSize;
181
182 btvWeights.resize(size);
183
184 const int ksize = (btvKernelSize - 1) / 2;
185 const float alpha_f = static_cast<float>(alpha);
186
187 for (int m = 0, ind = 0; m <= ksize; ++m)
188 {
189 for (int l = ksize; l + m >= 0; --l, ++ind)
190 btvWeights[ind] = pow(alpha_f, std::abs(m) + std::abs(l));
191 }
192 }

194 template <typename T>
195 struct BtvRegularizationBody : ParallelLoopBody
196 {
197 void operator ()(const Range& range) const;
198
199 Mat src;
200 mutable Mat dst;
201 int ksize;
202 const float* btvWeights;
203 };
204
205 template <typename T>
206 void BtvRegularizationBody<T>::operator ()(const Range& range) const
207 {
208 for (int i = range.start; i < range.end; ++i)
209 {
210 const T* srcRow = src.ptr<T>(i);
211 T* dstRow = dst.ptr<T>(i);
212
213 for(int j = ksize; j < src.cols - ksize; ++j)
214 {
215 const T srcVal = srcRow[j];
216
217 for (int m = 0, ind = 0; m <= ksize; ++m)
218 {
219 const T* srcRow2 = src.ptr<T>(i - m);
220 const T* srcRow3 = src.ptr<T>(i + m);
221
222 for (int l = ksize; l + m >= 0; --l, ++ind)
223 {
224 dstRow[j] += btvWeights[ind] * (diffSign(srcVal, srcRow3[j + l]) - diffSign(srcRow2[j - l], srcVal));
225 }
226 }
227 }
228 }
229 }

231 template <typename T>
232 void calcBtvRegularizationImpl(const Mat& src, Mat& dst, int btvKernelSize, const vector<float>& btvWeights)
233 {
234 dst.create(src.size(), src.type());
235 dst.setTo(Scalar::all(0));
236
237 const int ksize = (btvKernelSize - 1) / 2;
238
239 BtvRegularizationBody<T> body;
240
241 body.src = src;
242 body.dst = dst;
243 body.ksize = ksize;
244 body.btvWeights = &btvWeights[0];
245
246 parallel_for_(Range(ksize, src.rows - ksize), body);
247 }
248
249 void calcBtvRegularization(const Mat& src, Mat& dst, int btvKernelSize, const vector<float>& btvWeights)
250 {
251 typedef void (*func_t)(const Mat& src, Mat& dst, int btvKernelSize, const vector<float>& btvWeights);
252 static const func_t funcs[] =
253 {
254 0, calcBtvRegularizationImpl<float>, 0, calcBtvRegularizationImpl<Point3f>
255 };
256
257 const func_t func = funcs[src.channels()];
258
259 func(src, dst, btvKernelSize, btvWeights);
260 }

329 void BTVL1_Base::process(const vector<Mat>& src, Mat& dst, const vector<Mat>& forwardMotions, const vector<Mat>& backwardMotions, int baseIdx)
330 {
331 CV_Assert( scale_ > 1 );
332 CV_Assert( iterations_ > 0 );
333 CV_Assert( tau_ > 0.0 );
334 CV_Assert( alpha_ > 0.0 );
335 CV_Assert( btvKernelSize_ > 0 );
336 CV_Assert( blurKernelSize_ > 0 );
337 CV_Assert( blurSigma_ >= 0.0 );
338
339 // update blur filter and btv weights
340
341 if (filter_.empty() || blurKernelSize_ != curBlurKernelSize_ || blurSigma_ != curBlurSigma_ || src[0].type() != curSrcType_)
342 {
343 filter_ = createGaussianFilter(src[0].type(), Size(blurKernelSize_, blurKernelSize_), blurSigma_);
344 curBlurKernelSize_ = blurKernelSize_;
345 curBlurSigma_ = blurSigma_;
346 curSrcType_ = src[0].type();
347 }
348
349 if (btvWeights_.empty() || btvKernelSize_ != curBtvKernelSize_ || alpha_ != curAlpha_)
350 {
351 calcBtvWeights(btvKernelSize_, alpha_, btvWeights_);
352 curBtvKernelSize_ = btvKernelSize_;
353 curAlpha_ = alpha_;
354 }
355
356 // calc high res motions
357
358 calcRelativeMotions(forwardMotions, backwardMotions, lowResForwardMotions_, lowResBackwardMotions_, baseIdx, src[0].size());
359
360 upscaleMotions(lowResForwardMotions_, highResForwardMotions_, scale_);
361 upscaleMotions(lowResBackwardMotions_, highResBackwardMotions_, scale_);
362
363 forwardMaps_.resize(highResForwardMotions_.size());
364 backwardMaps_.resize(highResForwardMotions_.size());
365 for (size_t i = 0; i < highResForwardMotions_.size(); ++i)
366 buildMotionMaps(highResForwardMotions_[i], highResBackwardMotions_[i], forwardMaps_[i], backwardMaps_[i]);
367
368 // initial estimation
369
370 const Size lowResSize = src[0].size();
371 const Size highResSize(lowResSize.width * scale_, lowResSize.height * scale_);
372
373 resize(src[baseIdx], highRes_, highResSize, 0, 0, INTER_CUBIC);
374
375 // iterations
376
377 diffTerm_.create(highResSize, highRes_.type());
378 a_.create(highResSize, highRes_.type());
379 b_.create(highResSize, highRes_.type());
380 c_.create(lowResSize, highRes_.type());
381
382 for (int i = 0; i < iterations_; ++i)
383 {
384 diffTerm_.setTo(Scalar::all(0));
385
386 for (size_t k = 0; k < src.size(); ++k)
387 {
388 // a = M * Ih
389 remap(highRes_, a_, backwardMaps_[k], noArray(), INTER_NEAREST);
390 // b = HM * Ih
391 filter_->apply(a_, b_);
392 // c = DHM * Ih
393 resize(b_, c_, lowResSize, 0, 0, INTER_NEAREST);
394
395 diffSign(src[k], c_, c_);
396
397 // a = Dt * diff
398 upscale(c_, a_, scale_);
399 // b = HtDt * diff
400 filter_->apply(a_, b_);
401 // a = MtHtDt * diff
402 remap(b_, a_, forwardMaps_[k], noArray(), INTER_NEAREST);
403
404 add(diffTerm_, a_, diffTerm_);
405 }
406
407 if (lambda_ > 0)
408 {
409 calcBtvRegularization(highRes_, regTerm_, btvKernelSize_, btvWeights_);
410 addWeighted(diffTerm_, 1.0, regTerm_, -lambda_, 0.0, diffTerm_);
411 }
412
413 addWeighted(highRes_, 1.0, diffTerm_, tau_, 0.0, highRes_);
414 }
415
416 Rect inner(btvKernelSize_, btvKernelSize_, highRes_.cols - 2 * btvKernelSize_, highRes_.rows - 2 * btvKernelSize_);
417 highRes_(inner).copyTo(dst);
418 }

517 void BTVL1::initImpl(Ptr<FrameSource>& frameSource)
518 {
519 const int cacheSize = 2 * temporalAreaRadius_ + 1;
520
521 frames_.resize(cacheSize);
522 forwardMotions_.resize(cacheSize);
523 backwardMotions_.resize(cacheSize);
524 outputs_.resize(cacheSize);
525
526 storePos_ = -1;
527
528 for (int t = -temporalAreaRadius_; t <= temporalAreaRadius_; ++t)
529 readNextFrame(frameSource);
530
531 for (int i = 0; i <= temporalAreaRadius_; ++i)
532 processFrame(i);
533
534 procPos_ = temporalAreaRadius_;
535 outPos_ = -1;
536 }
537
538 void BTVL1::processImpl(Ptr<FrameSource>& frameSource, OutputArray _output)
539 {
540 if (outPos_ >= storePos_)
541 {
542 _output.release();
543 return;
544 }
545
546 readNextFrame(frameSource);
547
548 if (procPos_ < storePos_)
549 {
550 ++procPos_;
551 processFrame(procPos_);
552 }
553
554 ++outPos_;
555 const Mat& curOutput = at(outPos_, outputs_);
556
557 if (_output.kind() < _InputArray::OPENGL_BUFFER)
558 curOutput.convertTo(_output, CV_8U);
559 else
560 {
561 curOutput.convertTo(finalOutput_, CV_8U);
562 arrCopy(finalOutput_, _output);
563 }
564 }
565
566 void BTVL1::readNextFrame(Ptr<FrameSource>& frameSource)
567 {
568 frameSource->nextFrame(curFrame_);
569
570 if (curFrame_.empty())
571 return;
572
573 ++storePos_;
574 curFrame_.convertTo(at(storePos_, frames_), CV_32F);
575
576 if (storePos_ > 0)
577 {
578 opticalFlow_->calc(prevFrame_, curFrame_, at(storePos_ - 1, forwardMotions_));
579 opticalFlow_->calc(curFrame_, prevFrame_, at(storePos_, backwardMotions_));
580 }
581
582 curFrame_.copyTo(prevFrame_);
583 }
584
585 void BTVL1::processFrame(int idx)
586 {
587 const int startIdx = max(idx - temporalAreaRadius_, 0);
588 const int procIdx = idx;
589 const int endIdx = min(startIdx + 2 * temporalAreaRadius_, storePos_);
590
591 const int count = endIdx - startIdx + 1;
592
593 srcFrames_.resize(count);
594 srcForwardMotions_.resize(count);
595 srcBackwardMotions_.resize(count);
596
597 int baseIdx = -1;
598
599 for (int i = startIdx, k = 0; i <= endIdx; ++i, ++k)
600 {
601 if (i == procIdx)
602 baseIdx = k;
603
604 srcFrames_[k] = at(i, frames_);
605
606 if (i < endIdx)
607 srcForwardMotions_[k] = at(i, forwardMotions_);
608 if (i > startIdx)
609 srcBackwardMotions_[k] = at(i, backwardMotions_);
610 }
611
612 process(srcFrames_, at(idx, outputs_), srcForwardMotions_, srcBackwardMotions_, baseIdx);
613 }
