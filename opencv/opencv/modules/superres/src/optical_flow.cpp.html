65///////////////////////////////////////////////////////////////////
66// CpuOpticalFlow
67
68namespace
69{
70 class CpuOpticalFlow : public DenseOpticalFlowExt
71 {
72 public:
73 explicit CpuOpticalFlow(int work_type);
74
75 void calc(InputArray frame0, InputArray frame1, OutputArray flow1, OutputArray flow2);
76 void collectGarbage();
77
78 protected:
79 virtual void impl(const Mat& input0, const Mat& input1, OutputArray dst) = 0;
80
81 private:
82 int work_type_;
83 Mat buf_[6];
84 Mat flow_;
85 Mat flows_[2];
86 };
87
88 CpuOpticalFlow::CpuOpticalFlow(int work_type) : work_type_(work_type)
89 {
90 }
91
92 void CpuOpticalFlow::calc(InputArray _frame0, InputArray _frame1, OutputArray _flow1, OutputArray _flow2)
93 {
94 Mat frame0 = arrGetMat(_frame0, buf_[0]);
95 Mat frame1 = arrGetMat(_frame1, buf_[1]);
96
97 CV_Assert( frame1.type() == frame0.type() );
98 CV_Assert( frame1.size() == frame0.size() );
99
100 Mat input0 = convertToType(frame0, work_type_, buf_[2], buf_[3]);
101 Mat input1 = convertToType(frame1, work_type_, buf_[4], buf_[5]);
102
103 if (!_flow2.needed() && _flow1.kind() < _InputArray::OPENGL_BUFFER)
104 {
105 impl(input0, input1, _flow1);
106 return;
107 }
108
109 impl(input0, input1, flow_);
110
111 if (!_flow2.needed())
112 {
113 arrCopy(flow_, _flow1);
114 }
115 else
116 {
117 split(flow_, flows_);
118
119 arrCopy(flows_[0], _flow1);
120 arrCopy(flows_[1], _flow2);
121 }
122 }
123
124 void CpuOpticalFlow::collectGarbage()
125 {
126 for (int i = 0; i < 6; ++i)
127 buf_[i].release();
128 flow_.release();
129 flows_[0].release();
130 flows_[1].release();
131 }
132}

389namespace

391 class GpuOpticalFlow : public DenseOpticalFlowExt
392 {
393 public:
394 explicit GpuOpticalFlow(int work_type);
395
396 void calc(InputArray frame0, InputArray frame1, OutputArray flow1, OutputArray flow2);
397 void collectGarbage();
398
399 protected:
400 virtual void impl(const GpuMat& input0, const GpuMat& input1, GpuMat& dst1, GpuMat& dst2) = 0;
401
402 private:
403 int work_type_;
404 GpuMat buf_[6];
405 GpuMat u_, v_, flow_;
406 };
407
408 GpuOpticalFlow::GpuOpticalFlow(int work_type) : work_type_(work_type)
409 {
410 }
411
412 void GpuOpticalFlow::calc(InputArray _frame0, InputArray _frame1, OutputArray _flow1, OutputArray _flow2)
413 {
414 GpuMat frame0 = arrGetGpuMat(_frame0, buf_[0]);
415 GpuMat frame1 = arrGetGpuMat(_frame1, buf_[1]);
416
417 CV_Assert( frame1.type() == frame0.type() );
418 CV_Assert( frame1.size() == frame0.size() );
419
420 GpuMat input0 = convertToType(frame0, work_type_, buf_[2], buf_[3]);
421 GpuMat input1 = convertToType(frame1, work_type_, buf_[4], buf_[5]);
422
423 if (_flow2.needed() && _flow1.kind() == _InputArray::GPU_MAT && _flow2.kind() == _InputArray::GPU_MAT)
424 {
425 impl(input0, input1, _flow1.getGpuMatRef(), _flow2.getGpuMatRef());
426 return;
427 }
428
429 impl(input0, input1, u_, v_);
430
431 if (_flow2.needed())
432 {
433 arrCopy(u_, _flow1);
434 arrCopy(v_, _flow2);
435 }
436 else
437 {
438 GpuMat src[] = {u_, v_};
439 merge(src, 2, flow_);
440 arrCopy(flow_, _flow1);
441 }
442 }
443
444 void GpuOpticalFlow::collectGarbage()
445 {
446 for (int i = 0; i < 6; ++i)
447 buf_[i].release();
448 u_.release();
449 v_.release();
450 flow_.release();
451 }
452}

757 oclOpticalFlow::oclOpticalFlow(int work_type) : work_type_(work_type)
758 {
759 }
760
761 void oclOpticalFlow::calc(InputArray frame0, InputArray frame1, OutputArray flow1, OutputArray flow2)
762 {
763 ocl::oclMat& _frame0 = ocl::getOclMatRef(frame0);
764 ocl::oclMat& _frame1 = ocl::getOclMatRef(frame1);
765 ocl::oclMat& _flow1 = ocl::getOclMatRef(flow1);
766 ocl::oclMat& _flow2 = ocl::getOclMatRef(flow2);
767
768 CV_Assert( _frame1.type() == _frame0.type() );
769 CV_Assert( _frame1.size() == _frame0.size() );
770
771 cv::ocl::oclMat input0_ = convertToType(_frame0, work_type_, buf_[2], buf_[3]);
772 cv::ocl::oclMat input1_ = convertToType(_frame1, work_type_, buf_[4], buf_[5]);
773
774 impl(input0_, input1_, u_, v_);//go to tvl1 algorithm
775
776 u_.copyTo(_flow1);
777 v_.copyTo(_flow2);
778 }
779
780 void oclOpticalFlow::collectGarbage()
781 {
782 for (int i = 0; i < 6; ++i)
783 buf_[i].release();
784 u_.release();
785 v_.release();
786 flow_.release();
787 }
