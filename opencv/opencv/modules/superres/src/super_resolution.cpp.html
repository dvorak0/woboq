45using namespace std;
46using namespace cv;
47using namespace cv::superres;
48
49bool cv::superres::initModule_superres()
50{
51 return !createSuperResolution_BTVL1().empty();
52}
53
54cv::superres::SuperResolution::SuperResolution()
55{
56 frameSource_ = createFrameSource_Empty();
57 firstCall_ = true;
58}
59
60void cv::superres::SuperResolution::setInput(const Ptr<FrameSource>& frameSource)
61{
62 frameSource_ = frameSource;
63 firstCall_ = true;
64}
65
66void cv::superres::SuperResolution::nextFrame(OutputArray frame)
67{
68 if (firstCall_)
69 {
70 initImpl(frameSource_);
71 firstCall_ = false;
72 }
73
74 processImpl(frameSource_, frame);
75}
76
77void cv::superres::SuperResolution::reset()
78{
79 frameSource_->reset();
80 firstCall_ = true;
81}

85}
