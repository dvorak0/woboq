47class AllignedFrameSource : public cv::superres::FrameSource
48{
49public:
50 AllignedFrameSource(const cv::Ptr<cv::superres::FrameSource>& base, int scale);
51
52 void nextFrame(cv::OutputArray frame);
53 void reset();
54
55private:
56 cv::Ptr<cv::superres::FrameSource> base_;
57 cv::Mat origFrame_;
58 int scale_;
59};
60
61AllignedFrameSource::AllignedFrameSource(const cv::Ptr<cv::superres::FrameSource>& base, int scale) :
62 base_(base), scale_(scale)
63{
64 CV_Assert( !base_.empty() );
65}
66
67void AllignedFrameSource::nextFrame(cv::OutputArray frame)
68{
69 base_->nextFrame(origFrame_);
70
71 if (origFrame_.rows % scale_ == 0 && origFrame_.cols % scale_ == 0)
72 {
73 cv::superres::arrCopy(origFrame_, frame);
74 }
75 else
76 {
77 cv::Rect ROI(0, 0, (origFrame_.cols / scale_) * scale_, (origFrame_.rows / scale_) * scale_);
78 cv::superres::arrCopy(origFrame_(ROI), frame);
79 }
80}
81
82void AllignedFrameSource::reset()
83{
84 base_->reset();
85}

87class DegradeFrameSource : public cv::superres::FrameSource
88{
89public:
90 DegradeFrameSource(const cv::Ptr<cv::superres::FrameSource>& base, int scale);
91
92 void nextFrame(cv::OutputArray frame);
93 void reset();
94
95private:
96 cv::Ptr<cv::superres::FrameSource> base_;
97 cv::Mat origFrame_;
98 cv::Mat blurred_;
99 cv::Mat deg_;
100 double iscale_;
101};
102
103DegradeFrameSource::DegradeFrameSource(const cv::Ptr<cv::superres::FrameSource>& base, int scale) :
104 base_(base), iscale_(1.0 / scale)
105{
106 CV_Assert( !base_.empty() );
107}

109void addGaussNoise(cv::Mat& image, double sigma)
110{
111 cv::Mat noise(image.size(), CV_32FC(image.channels()));
112 cvtest::TS::ptr()->get_rng().fill(noise, cv::RNG::NORMAL, 0.0, sigma);
113
114 cv::addWeighted(image, 1.0, noise, 1.0, 0.0, image, image.depth());
115}
116
117void addSpikeNoise(cv::Mat& image, int frequency)
118{
119 cv::Mat_<uchar> mask(image.size(), 0);
120
121 for (int y = 0; y < mask.rows; ++y)
122 {
123 for (int x = 0; x < mask.cols; ++x)
124 {
125 if (cvtest::TS::ptr()->get_rng().uniform(0, frequency) < 1)
126 mask(y, x) = 255;
127 }
128 }
129
130 image.setTo(cv::Scalar::all(255), mask);
131}
132
133void DegradeFrameSource::nextFrame(cv::OutputArray frame)
134{
135 base_->nextFrame(origFrame_);
136
137 cv::GaussianBlur(origFrame_, blurred_, cv::Size(5, 5), 0);
138 cv::resize(blurred_, deg_, cv::Size(), iscale_, iscale_, cv::INTER_NEAREST);
139
140 addGaussNoise(deg_, 10.0);
141 addSpikeNoise(deg_, 500);
142
143 cv::superres::arrCopy(deg_, frame);
144}
145
146void DegradeFrameSource::reset()
147{
148 base_->reset();
149}

151double MSSIM(const cv::Mat& i1, const cv::Mat& i2)
152{
153 const double C1 = 6.5025;
154 const double C2 = 58.5225;
155
156 const int depth = CV_32F;
157
158 cv::Mat I1, I2;
159 i1.convertTo(I1, depth);
160 i2.convertTo(I2, depth);
161
162 cv::Mat I2_2 = I2.mul(I2); // I2^2
163 cv::Mat I1_2 = I1.mul(I1); // I1^2
164 cv::Mat I1_I2 = I1.mul(I2); // I1 * I2
165
166 cv::Mat mu1, mu2;
167 cv::GaussianBlur(I1, mu1, cv::Size(11, 11), 1.5);
168 cv::GaussianBlur(I2, mu2, cv::Size(11, 11), 1.5);
169
170 cv::Mat mu1_2 = mu1.mul(mu1);
171 cv::Mat mu2_2 = mu2.mul(mu2);
172 cv::Mat mu1_mu2 = mu1.mul(mu2);
173
174 cv::Mat sigma1_2, sigma2_2, sigma12;
175
176 cv::GaussianBlur(I1_2, sigma1_2, cv::Size(11, 11), 1.5);
177 sigma1_2 -= mu1_2;
178
179 cv::GaussianBlur(I2_2, sigma2_2, cv::Size(11, 11), 1.5);
180 sigma2_2 -= mu2_2;
181
182 cv::GaussianBlur(I1_I2, sigma12, cv::Size(11, 11), 1.5);
183 sigma12 -= mu1_mu2;
184
185 cv::Mat t1, t2;
186 cv::Mat numerator;
187 cv::Mat denominator;
188
189 // t3 = ((2*mu1_mu2 + C1).*(2*sigma12 + C2))
190 t1 = 2 * mu1_mu2 + C1;
191 t2 = 2 * sigma12 + C2;
192 numerator = t1.mul(t2);
193
194 // t1 =((mu1_2 + mu2_2 + C1).*(sigma1_2 + sigma2_2 + C2))
195 t1 = mu1_2 + mu2_2 + C1;
196 t2 = sigma1_2 + sigma2_2 + C2;
197 denominator = t1.mul(t2);
198
199 // ssim_map = numerator./denominator;
200 cv::Mat ssim_map;
201 cv::divide(numerator, denominator, ssim_map);
202
203 // mssim = average of ssim map
204 cv::Scalar mssim = cv::mean(ssim_map);
205
206 if (i1.channels() == 1)
207 return mssim[0];
208
209 return (mssim[0] + mssim[1] + mssim[3]) / 3;
210}

218void SuperResolution::RunTest(cv::Ptr<cv::superres::SuperResolution> superRes)
219{
220 const std::string inputVideoName = cvtest::TS::ptr()->get_data_path() + "car.avi";
221 const int scale = 2;
222 const int iterations = 100;
223 const int temporalAreaRadius = 2;
224
225 ASSERT_FALSE( superRes.empty() );
226
227 const int btvKernelSize = superRes->getInt("btvKernelSize");
228
229 superRes->set("scale", scale);
230 superRes->set("iterations", iterations);
231 superRes->set("temporalAreaRadius", temporalAreaRadius);
232
233 cv::Ptr<cv::superres::FrameSource> goldSource(new AllignedFrameSource(cv::superres::createFrameSource_Video(inputVideoName), scale));
234 cv::Ptr<cv::superres::FrameSource> lowResSource(new DegradeFrameSource(new AllignedFrameSource(cv::superres::createFrameSource_Video(inputVideoName), scale), scale));
235
236 // skip first frame
237 cv::Mat frame;
238
239 lowResSource->nextFrame(frame);
240 goldSource->nextFrame(frame);
241
242 cv::Rect inner(btvKernelSize, btvKernelSize, frame.cols - 2 * btvKernelSize, frame.rows - 2 * btvKernelSize);
243
244 superRes->setInput(lowResSource);
245
246 double srAvgMSSIM = 0.0;
247 const int count = 10;
248
249 cv::Mat goldFrame, superResFrame;
250 for (int i = 0; i < count; ++i)
251 {
252 goldSource->nextFrame(goldFrame);
253 ASSERT_FALSE( goldFrame.empty() );
254
255 superRes->nextFrame(superResFrame);
256 ASSERT_FALSE( superResFrame.empty() );
257
258 const double srMSSIM = MSSIM(goldFrame(inner), superResFrame);
259
260 srAvgMSSIM += srMSSIM;
261 }
262
263 srAvgMSSIM /= count;
264
265 EXPECT_GE( srAvgMSSIM, 0.5 );
266}
