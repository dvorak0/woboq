55 class OneFrameSource_CPU : public FrameSource
56 {
57 public:
58 explicit OneFrameSource_CPU(const Mat& frame) : frame_(frame) {}
59
60 void nextFrame(OutputArray frame)
61 {
62 frame.getMatRef() = frame_;
63 }
64
65 void reset()
66 {
67 }
68
69 private:
70 Mat frame_;
71 };

91 class ZeroOpticalFlow : public DenseOpticalFlowExt
92 {
93 public:
94 void calc(InputArray frame0, InputArray, OutputArray flow1, OutputArray flow2)
95 {
96 cv::Size size = frame0.size();
97
98 if (!flow2.needed())
99 {
100 flow1.create(size, CV_32FC2);
101
102 if (flow1.kind() == cv::_InputArray::GPU_MAT)
103 flow1.getGpuMatRef().setTo(cv::Scalar::all(0));
104 else
105 flow1.getMatRef().setTo(cv::Scalar::all(0));
106 }
107 else
108 {
109 flow1.create(size, CV_32FC1);
110 flow2.create(size, CV_32FC1);
111
112 if (flow1.kind() == cv::_InputArray::GPU_MAT)
113 flow1.getGpuMatRef().setTo(cv::Scalar::all(0));
114 else
115 flow1.getMatRef().setTo(cv::Scalar::all(0));
116
117 if (flow2.kind() == cv::_InputArray::GPU_MAT)
118 flow2.getGpuMatRef().setTo(cv::Scalar::all(0));
119 else
120 flow2.getMatRef().setTo(cv::Scalar::all(0));
121 }
122 }
123
124 void collectGarbage()
125 {
126 }
127 };
128}

130PERF_TEST_P(Size_MatType, SuperResolution_BTVL1,
131 Combine(Values(szSmall64, szSmall128),
132 Values(MatType(CV_8UC1), MatType(CV_8UC3))))
133{
134 declare.time(5 * 60);
135
136 const Size size = get<0>(GetParam());
137 const int type = get<1>(GetParam());
138
139 Mat frame(size, type);
140 declare.in(frame, WARMUP_RNG);
141
142 const int scale = 2;
143 const int iterations = 50;
144 const int temporalAreaRadius = 1;
145 Ptr<DenseOpticalFlowExt> opticalFlow(new ZeroOpticalFlow);
146
147 if (PERF_RUN_GPU())
148 {
149 Ptr<SuperResolution> superRes = createSuperResolution_BTVL1_GPU();
150
151 superRes->set("scale", scale);
152 superRes->set("iterations", iterations);
153 superRes->set("temporalAreaRadius", temporalAreaRadius);
154 superRes->set("opticalFlow", opticalFlow);
155
156 superRes->setInput(new OneFrameSource_GPU(GpuMat(frame)));
157
158 GpuMat dst;
159 superRes->nextFrame(dst);
160
161 TEST_CYCLE_N(10) superRes->nextFrame(dst);
162
163 GPU_SANITY_CHECK(dst);
164 }
165 else
166 {
167 Ptr<SuperResolution> superRes = createSuperResolution_BTVL1();
168
169 superRes->set("scale", scale);
170 superRes->set("iterations", iterations);
171 superRes->set("temporalAreaRadius", temporalAreaRadius);
172 superRes->set("opticalFlow", opticalFlow);
173
174 superRes->setInput(new OneFrameSource_CPU(frame));
175
176 Mat dst;
177 superRes->nextFrame(dst);
178
179 TEST_CYCLE_N(10) superRes->nextFrame(dst);
180
181 CPU_SANITY_CHECK(dst);
182 }
183}
