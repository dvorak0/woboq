59inline void normalizeAnchor(int &anchor, int ksize)
60{
61 if (anchor < 0)
62 anchor = ksize >> 1;
63
64 CV_Assert(0 <= anchor && anchor < ksize);
65}
66
67inline void normalizeAnchor(Point &anchor, const Size &ksize)
68{
69 normalizeAnchor(anchor.x, ksize.width);
70 normalizeAnchor(anchor.y, ksize.height);
71}
72
73inline void normalizeROI(Rect &roi, const Size &ksize, const Point &/*anchor*/, const Size &src_size)
74{
75 if (roi == Rect(0, 0, -1, -1))
76 roi = Rect(0, 0, src_size.width, src_size.height);
77
78 CV_Assert(ksize.height > 0 && ksize.width > 0 && ((ksize.height & 1) == 1) && ((ksize.width & 1) == 1));
79 CV_Assert(roi.x >= 0 && roi.y >= 0 && roi.width <= src_size.width && roi.height <= src_size.height);
80}

168static void GPUErode(const oclMat &src, oclMat &dst, oclMat &mat_kernel,
169 Size &ksize, const Point anchor, bool rectKernel)
170{
171 //Normalize the result by default
172 //float alpha = ksize.height * ksize.width;
173 CV_Assert(src.clCxt == dst.clCxt);
174 CV_Assert((src.cols == dst.cols) &&
175 (src.rows == dst.rows));
176 CV_Assert((src.oclchannels() == dst.oclchannels()));
177
178 int srcStep = src.step / src.elemSize();
179 int dstStep = dst.step / dst.elemSize();
180 int srcOffset = src.offset / src.elemSize();
181 int dstOffset = dst.offset / dst.elemSize();
182
183 int srcOffset_x = srcOffset % srcStep;
184 int srcOffset_y = srcOffset / srcStep;
185 Context *clCxt = src.clCxt;
186 string kernelName;
187#ifdef ANDROID
188 size_t localThreads[3] = {16, 8, 1};
189#else
190 size_t localThreads[3] = {16, 16, 1};
191#endif
192 size_t globalThreads[3] = {(src.cols + localThreads[0] - 1) / localThreads[0] *localThreads[0], (src.rows + localThreads[1] - 1) / localThreads[1] *localThreads[1], 1};
193
194 if (src.type() == CV_8UC1)
195 {
196 kernelName = "morph_C1_D0";
197 globalThreads[0] = ((src.cols + 3) / 4 + localThreads[0] - 1) / localThreads[0] * localThreads[0];
198 CV_Assert(localThreads[0]*localThreads[1] * 8 >= (localThreads[0] * 4 + ksize.width - 1) * (localThreads[1] + ksize.height - 1));
199 }
200 else
201 {
202 kernelName = "morph";
203 CV_Assert(localThreads[0]*localThreads[1] * 2 >= (localThreads[0] + ksize.width - 1) * (localThreads[1] + ksize.height - 1));
204 }
205
206 char s[64];
207
208 switch (src.type())
209 {
210 case CV_8UC1:
211 sprintf(s, "-D VAL=255");
212 break;
213 case CV_8UC3:
214 case CV_8UC4:
215 sprintf(s, "-D VAL=255 -D GENTYPE=uchar4");
216 break;
217 case CV_32FC1:
218 sprintf(s, "-D VAL=FLT_MAX -D GENTYPE=float");
219 break;
220 case CV_32FC3:
221 case CV_32FC4:
222 sprintf(s, "-D VAL=FLT_MAX -D GENTYPE=float4");
223 break;
224 default:
225 CV_Error(CV_StsUnsupportedFormat, "unsupported type");
226 }
227
228 char compile_option[128];
229 sprintf(compile_option, "-D RADIUSX=%d -D RADIUSY=%d -D LSIZE0=%d -D LSIZE1=%d -D ERODE %s %s",
230 anchor.x, anchor.y, (int)localThreads[0], (int)localThreads[1],
231 s, rectKernel?"-D RECTKERNEL":"");
232
233 vector< pair<size_t, const void *> > args;
234 args.push_back(make_pair(sizeof(cl_mem), (void *)&src.data));
235 args.push_back(make_pair(sizeof(cl_mem), (void *)&dst.data));
236 args.push_back(make_pair(sizeof(cl_int), (void *)&srcOffset_x));
237 args.push_back(make_pair(sizeof(cl_int), (void *)&srcOffset_y));
238 args.push_back(make_pair(sizeof(cl_int), (void *)&src.cols));
239 args.push_back(make_pair(sizeof(cl_int), (void *)&src.rows));
240 args.push_back(make_pair(sizeof(cl_int), (void *)&srcStep));
241 args.push_back(make_pair(sizeof(cl_int), (void *)&dstStep));
242 args.push_back(make_pair(sizeof(cl_mem), (void *)&mat_kernel.data));
243 args.push_back(make_pair(sizeof(cl_int), (void *)&src.wholecols));
244 args.push_back(make_pair(sizeof(cl_int), (void *)&src.wholerows));
245 args.push_back(make_pair(sizeof(cl_int), (void *)&dstOffset));
246
247 openCLExecuteKernel(clCxt, &filtering_morph, kernelName, globalThreads, localThreads, args, -1, -1, compile_option);
248}

333Ptr<BaseFilter_GPU> cv::ocl::getMorphologyFilter_GPU(int op, int type, const Mat &_kernel, const Size &ksize, Point anchor)
334{
335 CV_Assert(op == MORPH_ERODE || op == MORPH_DILATE);
336 CV_Assert(type == CV_8UC1 || type == CV_8UC3 || type == CV_8UC4 || type == CV_32FC1 || type == CV_32FC3 || type == CV_32FC4);
337
338 normalizeAnchor(anchor, ksize);
339 Mat kernel8U;
340 _kernel.convertTo(kernel8U, CV_8U);
341 Mat kernel = kernel8U.reshape(1, 1);
342
343 bool noZero = true;
344 for(int i = 0; i < kernel.rows * kernel.cols; ++i)
345 if(kernel.at<uchar>(i) != 1)
346 noZero = false;
347
348 MorphFilter_GPU* mfgpu = new MorphFilter_GPU(ksize, anchor, kernel, op == MORPH_ERODE ? GPUErode : GPUDilate);
349 if(noZero)
350 mfgpu->rectKernel = true;
351
352 return Ptr<BaseFilter_GPU>(mfgpu);
353}

399void morphOp(int op, const oclMat &src, oclMat &dst, const Mat &_kernel, Point anchor, int iterations, int borderType, const Scalar &borderValue)
400{
401 if ((borderType != cv::BORDER_CONSTANT) || (borderValue != morphologyDefaultBorderValue()))
402 {
403 CV_Error(CV_StsBadArg, "unsupported border type");
404 }
405
406 Mat kernel;
407 Size ksize = _kernel.data ? _kernel.size() : Size(3, 3);
408
409 normalizeAnchor(anchor, ksize);
410
411 if (iterations == 0 || _kernel.rows *_kernel.cols == 1)
412 {
413 src.copyTo(dst);
414 return;
415 }
416
417 dst.create(src.size(), src.type());
418
419 if (!_kernel.data)
420 {
421 kernel = getStructuringElement(MORPH_RECT, Size(1 + iterations * 2, 1 + iterations * 2));
422 anchor = Point(iterations, iterations);
423 iterations = 1;
424 }
425 else if (iterations > 1 && countNonZero(_kernel) == _kernel.rows * _kernel.cols)
426 {
427 anchor = Point(anchor.x * iterations, anchor.y * iterations);
428 kernel = getStructuringElement(MORPH_RECT, Size(ksize.width + (iterations - 1) * (ksize.width - 1),
429 ksize.height + (iterations - 1) * (ksize.height - 1)), anchor);
430 iterations = 1;
431 }
432 else
433 kernel = _kernel;
434
435 Ptr<MorphologyFilterEngine_GPU> f = createMorphologyFilter_GPU(op, src.type(), kernel, anchor, iterations);
436
437 f->apply(src, dst);
438}

441void cv::ocl::erode(const oclMat &src, oclMat &dst, const Mat &kernel, Point anchor, int iterations,
442 int borderType, const Scalar &borderValue)
443{
444 bool allZero = true;
445

449

451 kernel.data[0] = 1;

462void cv::ocl::morphologyEx(const oclMat &src, oclMat &dst, int op, const Mat &kernel, Point anchor, int iterations,
463 int borderType, const Scalar &borderValue)
464{
465 oclMat temp;
466
467 switch (op)
468 {
469 case MORPH_ERODE:
470 erode(src, dst, kernel, anchor, iterations, borderType, borderValue);
471 break;
472 case MORPH_DILATE:
473 dilate(src, dst, kernel, anchor, iterations, borderType, borderValue);
474 break;
475 case MORPH_OPEN:
476 erode(src, temp, kernel, anchor, iterations, borderType, borderValue);
477 dilate(temp, dst, kernel, anchor, iterations, borderType, borderValue);
478 break;
479 case CV_MOP_CLOSE:
480 dilate(src, temp, kernel, anchor, iterations, borderType, borderValue);
481 erode(temp, dst, kernel, anchor, iterations, borderType, borderValue);
482 break;
483 case CV_MOP_GRADIENT:
484 erode(src, temp, kernel, anchor, iterations, borderType, borderValue);
485 dilate(src, dst, kernel, anchor, iterations, borderType, borderValue);
486 subtract(dst, temp, dst);
487 break;
488 case CV_MOP_TOPHAT:
489 erode(src, dst, kernel, anchor, iterations, borderType, borderValue);
490 dilate(dst, temp, kernel, anchor, iterations, borderType, borderValue);
491 subtract(src, temp, dst);
492 break;
493 case CV_MOP_BLACKHAT:
494 dilate(src, dst, kernel, anchor, iterations, borderType, borderValue);
495 erode(dst, temp, kernel, anchor, iterations, borderType, borderValue);
496 subtract(temp, src, dst);
497 break;
498 default:
499 CV_Error(CV_StsBadArg, "unknown morphological operation");
500 }
501}

527// prepare kernel: transpose and make double rows (+align). Returns size of aligned row
528// Samples:
529// a b c
530// Input: d e f
531// g h i
532// Output, last two zeros is the alignment:
533// a d g a d g 0 0
534// b e h b e h 0 0
535// c f i c f i 0 0
536template <typename T>
537static int _prepareKernelFilter2D(std::vector<T>& data, const Mat &kernel)
538{
539 Mat _kernel; kernel.convertTo(_kernel, DataDepth<T>::value);
540 int size_y_aligned = roundUp(kernel.rows * 2, 4);
541 data.clear(); data.resize(size_y_aligned * kernel.cols, 0);
542 for (int x = 0; x < kernel.cols; x++)
543 {
544 for (int y = 0; y < kernel.rows; y++)
545 {
546 data[x * size_y_aligned + y] = _kernel.at<T>(y, x);
547 data[x * size_y_aligned + y + kernel.rows] = _kernel.at<T>(y, x);
548 }
549 }
550 return size_y_aligned;
551}

553static void GPUFilter2D(const oclMat &src, oclMat &dst, const Mat &kernel,
554 const Size &ksize, const Point& anchor, const int borderType)
555{
556 CV_Assert(src.clCxt == dst.clCxt);
557 CV_Assert((src.cols == dst.cols) &&
558 (src.rows == dst.rows));
559 CV_Assert(src.oclchannels() == dst.oclchannels());
560
561 CV_Assert(kernel.cols == ksize.width && kernel.rows == ksize.height);
562 CV_Assert(kernel.channels() == 1);
563
564 CV_Assert(anchor.x >= 0 && anchor.x < kernel.cols);
565 CV_Assert(anchor.y >= 0 && anchor.y < kernel.rows);
566
567 bool useDouble = src.depth() == CV_64F;
568
569 std::vector<float> kernelDataFloat;
570 std::vector<double> kernelDataDouble;
571 int kernel_size_y2_aligned = useDouble ?
572 _prepareKernelFilter2D<double>(kernelDataDouble, kernel)
573 : _prepareKernelFilter2D<float>(kernelDataFloat, kernel);
574 oclMat oclKernelParameter;
575 if (useDouble)
576 {
577 oclKernelParameter.createEx(1, kernelDataDouble.size(), CV_64FC1, DEVICE_MEM_R_ONLY, DEVICE_MEM_DEFAULT);
578 openCLMemcpy2D(src.clCxt, oclKernelParameter.data, kernelDataDouble.size()*sizeof(double),
579 &kernelDataDouble[0], kernelDataDouble.size()*sizeof(double),
580 kernelDataDouble.size()*sizeof(double), 1, clMemcpyHostToDevice);
581 }
582 else
583 {
584 oclKernelParameter.createEx(1, kernelDataFloat.size(), CV_32FC1, DEVICE_MEM_R_ONLY, DEVICE_MEM_DEFAULT);
585 openCLMemcpy2D(src.clCxt, oclKernelParameter.data, kernelDataFloat.size()*sizeof(float),
586 &kernelDataFloat[0], kernelDataFloat.size()*sizeof(float),
587 kernelDataFloat.size()*sizeof(float), 1, clMemcpyHostToDevice);
588 }
589
590 size_t tryWorkItems = src.clCxt->getDeviceInfo().maxWorkItemSizes[0];
591 do {
592 size_t BLOCK_SIZE = tryWorkItems;
593 while (BLOCK_SIZE > 32 && BLOCK_SIZE >= (size_t)ksize.width * 2 && BLOCK_SIZE > (size_t)src.cols * 2)
594 BLOCK_SIZE /= 2;
595#if 1 // TODO Mode with several blocks requires a much more VGPRs, so this optimization is not actual for the current devices
596 size_t BLOCK_SIZE_Y = 1;
597#else
598 size_t BLOCK_SIZE_Y = 8; // TODO Check heuristic value on devices
599 while (BLOCK_SIZE_Y < BLOCK_SIZE / 8 && BLOCK_SIZE_Y * src.clCxt->getDeviceInfo().maxComputeUnits * 32 < (size_t)src.rows)
