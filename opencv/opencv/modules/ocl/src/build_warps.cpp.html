53// buildWarpPlaneMaps
54
55void cv::ocl::buildWarpPlaneMaps(Size /*src_size*/, Rect dst_roi, const Mat &K, const Mat &R, const Mat &T,
56 float scale, oclMat &xmap, oclMat &ymap)
57{
58 CV_Assert(K.size() == Size(3, 3) && K.type() == CV_32F);
59 CV_Assert(R.size() == Size(3, 3) && R.type() == CV_32F);
60 CV_Assert((T.size() == Size(3, 1) || T.size() == Size(1, 3)) && T.type() == CV_32F && T.isContinuous());
61
62 Mat K_Rinv = K * R.t();
63 CV_Assert(K_Rinv.isContinuous());
64
65 Mat KRT_mat(1, 12, CV_32FC1); // 9 + 3
66 KRT_mat(Range::all(), Range(0, 8)) = K_Rinv.reshape(1, 1);
67 KRT_mat(Range::all(), Range(9, 11)) = T;
68
69 oclMat KRT_oclMat(KRT_mat);
70 // transfer K_Rinv and T into a single cl_mem
71 xmap.create(dst_roi.size(), CV_32F);
72 ymap.create(dst_roi.size(), CV_32F);
73
74 int tl_u = dst_roi.tl().x;
75 int tl_v = dst_roi.tl().y;
76
77 int xmap_step = xmap.step / xmap.elemSize(), xmap_offset = xmap.offset / xmap.elemSize();
78 int ymap_step = ymap.step / ymap.elemSize(), ymap_offset = ymap.offset / ymap.elemSize();
79
80 vector< pair<size_t, const void *> > args;
81 args.push_back( make_pair( sizeof(cl_mem), (void *)&xmap.data));
82 args.push_back( make_pair( sizeof(cl_mem), (void *)&ymap.data));
83 args.push_back( make_pair( sizeof(cl_mem), (void *)&KRT_mat.data));
84 args.push_back( make_pair( sizeof(cl_int), (void *)&tl_u));
85 args.push_back( make_pair( sizeof(cl_int), (void *)&tl_v));
86 args.push_back( make_pair( sizeof(cl_int), (void *)&xmap.cols));
87 args.push_back( make_pair( sizeof(cl_int), (void *)&xmap.rows));
88 args.push_back( make_pair( sizeof(cl_int), (void *)&xmap_step));
89 args.push_back( make_pair( sizeof(cl_int), (void *)&ymap_step));
90 args.push_back( make_pair( sizeof(cl_int), (void *)&xmap_offset));
91 args.push_back( make_pair( sizeof(cl_int), (void *)&ymap_offset));
92 args.push_back( make_pair( sizeof(cl_float), (void *)&scale));
93
94 size_t globalThreads[3] = { (size_t)xmap.cols, (size_t)xmap.rows, 1 };
95#ifdef ANDROID
96 size_t localThreads[3] = {32, 4, 1};
97#else
98 size_t localThreads[3] = {32, 8, 1};
99#endif
100 openCLExecuteKernel(Context::getContext(), &build_warps, "buildWarpPlaneMaps", globalThreads, localThreads, args, -1, -1);
101}

196// buildWarpAffineMaps
197
198void cv::ocl::buildWarpAffineMaps(const Mat &M, bool inverse, Size dsize, oclMat &xmap, oclMat &ymap)
199{
200 CV_Assert(M.rows == 2 && M.cols == 3);
201 CV_Assert(dsize.area());
202
203 xmap.create(dsize, CV_32FC1);
204 ymap.create(dsize, CV_32FC1);
205
206 float coeffs[2 * 3];
207 Mat coeffsMat(2, 3, CV_32F, (void *)coeffs);
208
209 if (inverse)
210 M.convertTo(coeffsMat, coeffsMat.type());
211 else
212 {
213 cv::Mat iM;
214 invertAffineTransform(M, iM);
215 iM.convertTo(coeffsMat, coeffsMat.type());
216 }
217
218 int xmap_step = xmap.step / xmap.elemSize(), xmap_offset = xmap.offset / xmap.elemSize();
219 int ymap_step = ymap.step / ymap.elemSize(), ymap_offset = ymap.offset / ymap.elemSize();
220
221 oclMat coeffsOclMat(coeffsMat.reshape(1, 1));
222
223 vector< pair<size_t, const void *> > args;
224 args.push_back( make_pair( sizeof(cl_mem), (void *)&xmap.data));
225 args.push_back( make_pair( sizeof(cl_mem), (void *)&ymap.data));
226 args.push_back( make_pair( sizeof(cl_mem), (void *)&coeffsOclMat.data));
227 args.push_back( make_pair( sizeof(cl_int), (void *)&xmap.cols));
228 args.push_back( make_pair( sizeof(cl_int), (void *)&xmap.rows));
229 args.push_back( make_pair( sizeof(cl_int), (void *)&xmap_step));
230 args.push_back( make_pair( sizeof(cl_int), (void *)&ymap_step));
231 args.push_back( make_pair( sizeof(cl_int), (void *)&xmap_offset));
232 args.push_back( make_pair( sizeof(cl_int), (void *)&ymap_offset));
233
234 size_t globalThreads[3] = { (size_t)xmap.cols, (size_t)xmap.rows, 1 };
235#ifdef ANDROID
236 size_t localThreads[3] = {32, 4, 1};
237#else
238 size_t localThreads[3] = {32, 8, 1};
239#endif
240 openCLExecuteKernel(Context::getContext(), &build_warps, "buildWarpAffineMaps", globalThreads, localThreads, args, -1, -1);
241}

245
246void cv::ocl::buildWarpPerspectiveMaps(const Mat &M, bool inverse, Size dsize, oclMat &xmap, oclMat &ymap)
247{
248 CV_Assert(M.rows == 3 && M.cols == 3);
249 CV_Assert(dsize.area() > 0);
250
251 xmap.create(dsize, CV_32FC1);
252 ymap.create(dsize, CV_32FC1);
253
254 float coeffs[3 * 3];
255 Mat coeffsMat(3, 3, CV_32F, (void *)coeffs);
256
257 if (inverse)
258 M.convertTo(coeffsMat, coeffsMat.type());
259 else
260 {
261 cv::Mat iM;
262 invert(M, iM);
263 iM.convertTo(coeffsMat, coeffsMat.type());
264 }
265
266 oclMat coeffsOclMat(coeffsMat.reshape(1, 1));
267
268 int xmap_step = xmap.step / xmap.elemSize(), xmap_offset = xmap.offset / xmap.elemSize();
269 int ymap_step = ymap.step / ymap.elemSize(), ymap_offset = ymap.offset / ymap.elemSize();
270
271 vector< pair<size_t, const void *> > args;
272 args.push_back( make_pair( sizeof(cl_mem), (void *)&xmap.data));
273 args.push_back( make_pair( sizeof(cl_mem), (void *)&ymap.data));
274 args.push_back( make_pair( sizeof(cl_mem), (void *)&coeffsOclMat.data));
275 args.push_back( make_pair( sizeof(cl_int), (void *)&xmap.cols));
276 args.push_back( make_pair( sizeof(cl_int), (void *)&xmap.rows));
277 args.push_back( make_pair( sizeof(cl_int), (void *)&xmap_step));
278 args.push_back( make_pair( sizeof(cl_int), (void *)&ymap_step));
279 args.push_back( make_pair( sizeof(cl_int), (void *)&xmap_offset));
280 args.push_back( make_pair( sizeof(cl_int), (void *)&ymap_offset));
281
282 size_t globalThreads[3] = { (size_t)xmap.cols, (size_t)xmap.rows, 1 };
283
284 openCLExecuteKernel(Context::getContext(), &build_warps, "buildWarpPerspectiveMaps", globalThreads, NULL, args, -1, -1);
285}
