46#include "precomp.hpp"
47#include "opencl_kernels.hpp"
48
49using namespace cv;
50using namespace cv::ocl;
51
52void cv::ocl::blendLinear(const oclMat &src1, const oclMat &src2, const oclMat &weights1, const oclMat &weights2,
53 oclMat &dst)
54{
55 CV_Assert(src1.depth() <= CV_32F);
56 CV_Assert(src1.size() == src2.size() && src1.type() == src2.type());
57 CV_Assert(weights1.size() == weights2.size() && weights1.size() == src1.size() &&
58 weights1.type() == CV_32FC1 && weights2.type() == CV_32FC1);
59
60 dst.create(src1.size(), src1.type());
61
62 size_t globalSize[] = { (size_t)dst.cols, (size_t)dst.rows, 1};
63 size_t localSize[] = { 16, 16, 1 };
64
65 int depth = dst.depth(), ocn = dst.oclchannels();
66 int src1_step = src1.step / src1.elemSize(), src1_offset = src1.offset / src1.elemSize();
67 int src2_step = src2.step / src2.elemSize(), src2_offset = src2.offset / src2.elemSize();
68 int weight1_step = weights1.step / weights1.elemSize(), weight1_offset = weights1.offset / weights1.elemSize();
69 int weight2_step = weights2.step / weights2.elemSize(), weight2_offset = weights2.offset / weights2.elemSize();
70 int dst_step = dst.step / dst.elemSize(), dst_offset = dst.offset / dst.elemSize();
71
72 const char * const channelMap[] = { "", "", "2", "4", "4" };
73 const char * const typeMap[] = { "uchar", "char", "ushort", "short", "int", "float", "double" };
74 std::string buildOptions = format("-D T=%s%s -D convertToT=convert_%s%s%s -D FT=float%s -D convertToFT=convert_float%s",
75 typeMap[depth], channelMap[ocn], typeMap[depth], channelMap[ocn],
76 depth >= CV_32S ? "" : "_sat_rte", channelMap[ocn], channelMap[ocn]);
77
78 vector< pair<size_t, const void *> > args;
79 args.push_back( make_pair( sizeof(cl_mem), (void *)&src1.data ));
80 args.push_back( make_pair( sizeof(cl_int), (void *)&src1_offset ));
81 args.push_back( make_pair( sizeof(cl_int), (void *)&src1_step ));
82 args.push_back( make_pair( sizeof(cl_mem), (void *)&src2.data ));
83 args.push_back( make_pair( sizeof(cl_int), (void *)&src2_offset ));
84 args.push_back( make_pair( sizeof(cl_int), (void *)&src2_step ));
85 args.push_back( make_pair( sizeof(cl_mem), (void *)&weights1.data ));
86 args.push_back( make_pair( sizeof(cl_int), (void *)&weight1_offset ));
87 args.push_back( make_pair( sizeof(cl_int), (void *)&weight1_step ));
88 args.push_back( make_pair( sizeof(cl_mem), (void *)&weights2.data ));
89 args.push_back( make_pair( sizeof(cl_int), (void *)&weight2_offset ));
90 args.push_back( make_pair( sizeof(cl_int), (void *)&weight2_step ));
91 args.push_back( make_pair( sizeof(cl_mem), (void *)&dst.data ));
92 args.push_back( make_pair( sizeof(cl_int), (void *)&dst_offset ));
93 args.push_back( make_pair( sizeof(cl_int), (void *)&dst_step ));
94 args.push_back( make_pair( sizeof(cl_int), (void *)&dst.rows ));
95 args.push_back( make_pair( sizeof(cl_int), (void *)&dst.cols ));
96
97 openCLExecuteKernel(src1.clCxt, &blend_linear, "blendLinear", globalSize, localSize, args,
98 -1, -1, buildOptions.c_str());
99}
