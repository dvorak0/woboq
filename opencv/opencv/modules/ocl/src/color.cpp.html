53static void fromRGB_caller(const oclMat &src, oclMat &dst, int bidx, const std::string & kernelName,
54 const std::string & additionalOptions = std::string(),
55 const oclMat & data1 = oclMat(), const oclMat & data2 = oclMat())
56{
57 int src_offset = src.offset / src.elemSize1(), src_step = src.step1();
58 int dst_offset = dst.offset / dst.elemSize1(), dst_step = dst.step1();
59 int pixels_per_work_item = 1;
60
61 if (Context::getContext()->supportsFeature(FEATURE_CL_INTEL_DEVICE))
62 {
63 if ((src.cols % 4 == 0) && (src.depth() == CV_8U))
64 pixels_per_work_item = 4;
65 else if (src.cols % 2 == 0)
66 pixels_per_work_item = 2;
67 else
68 pixels_per_work_item = 1;
69 }
70
71 std::string build_options = format("-D DEPTH_%d -D scn=%d -D bidx=%d -D pixels_per_work_item=%d", src.depth(), src.oclchannels(), bidx, pixels_per_work_item);
72 if (!additionalOptions.empty())
73 build_options += additionalOptions;
74
75 vector<pair<size_t , const void *> > args;
76 args.push_back( make_pair( sizeof(cl_int) , (void *)&dst.cols));
77 args.push_back( make_pair( sizeof(cl_int) , (void *)&dst.rows));
78 args.push_back( make_pair( sizeof(cl_int) , (void *)&src_step));
79 args.push_back( make_pair( sizeof(cl_int) , (void *)&dst_step));
80 args.push_back( make_pair( sizeof(cl_mem) , (void *)&src.data));
81 args.push_back( make_pair( sizeof(cl_mem) , (void *)&dst.data));
82 args.push_back( make_pair( sizeof(cl_int) , (void *)&src_offset ));
83 args.push_back( make_pair( sizeof(cl_int) , (void *)&dst_offset ));
84
85 if (!data1.empty())
86 args.push_back( make_pair( sizeof(cl_mem) , (void *)&data1.data ));
87 if (!data2.empty())
88 args.push_back( make_pair( sizeof(cl_mem) , (void *)&data2.data ));
89
90 size_t gt[3] = { (size_t)dst.cols/pixels_per_work_item, (size_t)dst.rows, 1 };
91#ifdef ANDROID
92 size_t lt[3] = { 16, 10, 1 };
93#else
94 size_t lt[3] = { 16, 16, 1 };
95#endif
96 openCLExecuteKernel(src.clCxt, &cvt_color, kernelName.c_str(), gt, lt, args, -1, -1, build_options.c_str());
97}

352static void cvtColor_caller(const oclMat &src, oclMat &dst, int code, int dcn)
353{
354 Size sz = src.size();
355 int scn = src.channels(), depth = src.depth(), bidx;
356
357 CV_Assert(depth == CV_8U || depth == CV_16U || depth == CV_32F);
358
359 switch (code)
360 {
361 case CV_BGR2BGRA: case CV_RGB2BGRA: case CV_BGRA2BGR:
362 case CV_RGBA2BGR: case CV_RGB2BGR: case CV_BGRA2RGBA:
363 {
364 CV_Assert(scn == 3 || scn == 4);
365 dcn = code == CV_BGR2BGRA || code == CV_RGB2BGRA || code == CV_BGRA2RGBA ? 4 : 3;
366 bool reverse = !(code == CV_BGR2BGRA || code == CV_BGRA2BGR);
367 dst.create(sz, CV_MAKE_TYPE(depth, dcn));
368 RGB_caller(src, dst, reverse);
369 break;
370 }
371 case CV_BGR2BGR565: case CV_BGR2BGR555: case CV_RGB2BGR565: case CV_RGB2BGR555:
372 case CV_BGRA2BGR565: case CV_BGRA2BGR555: case CV_RGBA2BGR565: case CV_RGBA2BGR555:
373 {
374 CV_Assert((scn == 3 || scn == 4) && depth == CV_8U );
375 bidx = code == CV_BGR2BGR565 || code == CV_BGR2BGR555 ||
376 code == CV_BGRA2BGR565 || code == CV_BGRA2BGR555 ? 0 : 2;
377 int greenbits = code == CV_BGR2BGR565 || code == CV_RGB2BGR565 ||
378 code == CV_BGRA2BGR565 || code == CV_RGBA2BGR565 ? 6 : 5;
379 dst.create(sz, CV_8UC2);
380 toRGB5x5_caller(src, dst, bidx, greenbits, "RGB2RGB5x5");
381 break;
382 }
383 case CV_BGR5652BGR: case CV_BGR5552BGR: case CV_BGR5652RGB: case CV_BGR5552RGB:
384 case CV_BGR5652BGRA: case CV_BGR5552BGRA: case CV_BGR5652RGBA: case CV_BGR5552RGBA:
385 {
386 dcn = code == CV_BGR5652BGRA || code == CV_BGR5552BGRA || code == CV_BGR5652RGBA || code == CV_BGR5552RGBA ? 4 : 3;
387 CV_Assert((dcn == 3 || dcn == 4) && scn == 2 && depth == CV_8U);
388 bidx = code == CV_BGR5652BGR || code == CV_BGR5552BGR ||
389 code == CV_BGR5652BGRA || code == CV_BGR5552BGRA ? 0 : 2;
390 int greenbits = code == CV_BGR5652BGR || code == CV_BGR5652RGB ||
391 code == CV_BGR5652BGRA || code == CV_BGR5652RGBA ? 6 : 5;
392 dst.create(sz, CV_MAKETYPE(depth, dcn));
393 fromRGB5x5_caller(src, dst, bidx, greenbits, "RGB5x52RGB");
394 break;
395 }
396 case CV_BGR5652GRAY: case CV_BGR5552GRAY:
397 {
398 CV_Assert(scn == 2 && depth == CV_8U);
399 dst.create(sz, CV_8UC1);
400 int greenbits = code == CV_BGR5652GRAY ? 6 : 5;
401 fromRGB5x5_caller(src, dst, -1, greenbits, "BGR5x52Gray");
402 break;
403 }
404 case CV_GRAY2BGR565: case CV_GRAY2BGR555:
405 {
406 CV_Assert(scn == 1 && depth == CV_8U);
407 dst.create(sz, CV_8UC2);
408 int greenbits = code == CV_GRAY2BGR565 ? 6 : 5;
409 toRGB5x5_caller(src, dst, -1, greenbits, "Gray2BGR5x5");
410 break;
411 }
412 case CV_RGB2GRAY: case CV_BGR2GRAY: case CV_RGBA2GRAY: case CV_BGRA2GRAY:
413 {
414 CV_Assert(scn == 3 || scn == 4);
415 bidx = code == CV_BGR2GRAY || code == CV_BGRA2GRAY ? 0 : 2;
416 dst.create(sz, CV_MAKETYPE(depth, 1));
417 fromRGB_caller(src, dst, bidx, "RGB2Gray");
418 break;
419 }
420 case CV_GRAY2BGR: case CV_GRAY2BGRA:
421 {
422 CV_Assert(scn == 1);
423 dcn = code == CV_GRAY2BGRA ? 4 : 3;
424 dst.create(sz, CV_MAKETYPE(depth, dcn));
425 fromGray_caller(src, dst, 0, "Gray2RGB");
426 break;
427 }
428 case CV_BGR2YUV: case CV_RGB2YUV:
429 {
430 CV_Assert(scn == 3 || scn == 4);
431 bidx = code == CV_BGR2YUV ? 0 : 2;
432 dst.create(sz, CV_MAKETYPE(depth, 3));
433 fromRGB_caller(src, dst, bidx, "RGB2YUV");
434 break;
435 }
436 case CV_YUV2BGR: case CV_YUV2RGB:
437 {
438 if( dcn <= 0 )
439 dcn = 3;
440 CV_Assert(scn == 3 && (dcn == 3 || dcn == 4));
441 bidx = code == CV_YUV2BGR ? 0 : 2;
442 dst.create(sz, CV_MAKETYPE(depth, dcn));
443 toRGB_caller(src, dst, bidx, "YUV2RGB");
444 break;
445 }
446 case CV_YUV2RGB_NV12: case CV_YUV2BGR_NV12:
447 case CV_YUV2RGBA_NV12: case CV_YUV2BGRA_NV12:
448 {
449 CV_Assert(scn == 1);
