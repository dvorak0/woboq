45#include "precomp.hpp"
46
47namespace cv {
48namespace ocl {
49
50class ProgramCache
51{
52protected:
53 ProgramCache();
54 ~ProgramCache();
55public:
56 static ProgramCache *getProgramCache();
57
58 cl_program getProgram(const Context *ctx, const cv::ocl::ProgramEntry* source,
59 const char *build_options);
60
61 void releaseProgram();
62protected:
63 //lookup the binary given the file name
64 // (with acquired mutexCache)
65 cl_program progLookup(const string& srcsign);
66
67 //add program to the cache
68 // (with acquired mutexCache)
69 void addProgram(const string& srcsign, cl_program program);
70
71 map <string, cl_program> codeCache;
72 unsigned int cacheSize;
73
74 //The presumed watermark for the cache volume (256MB). Is it enough?
75 //We may need more delicate algorithms when necessary later.
76 //Right now, let's just leave it along.
77 static const unsigned MAX_PROG_CACHE_SIZE = 1024;
78
79 // acquire both mutexes in this order: 1) mutexFiles 2) mutexCache
80 static cv::Mutex mutexFiles;
81 static cv::Mutex mutexCache;
82};

84}//namespace ocl
