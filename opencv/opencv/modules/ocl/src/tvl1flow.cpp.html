51cv::ocl::OpticalFlowDual_TVL1_OCL::OpticalFlowDual_TVL1_OCL()
52{
53 tau = 0.25;
54 lambda = 0.15;
55 theta = 0.3;
56 nscales = 5;
57 warps = 5;
58 epsilon = 0.01;
59 iterations = 300;
60 useInitialFlow = false;
61}

63void cv::ocl::OpticalFlowDual_TVL1_OCL::operator()(const oclMat& I0, const oclMat& I1, oclMat& flowx, oclMat& flowy)
64{
65 CV_Assert( I0.type() == CV_8UC1 || I0.type() == CV_32FC1 );
66 CV_Assert( I0.size() == I1.size() );
67 CV_Assert( I0.type() == I1.type() );
68 CV_Assert( !useInitialFlow || (flowx.size() == I0.size() && flowx.type() == CV_32FC1 && flowy.size() == flowx.size() && flowy.type() == flowx.type()) );
69 CV_Assert( nscales > 0 );
70
71 // allocate memory for the pyramid structure
72 I0s.resize(nscales);
73 I1s.resize(nscales);
74 u1s.resize(nscales);
75 u2s.resize(nscales);
76 //I0s_step == I1s_step
77 I0.convertTo(I0s[0], CV_32F, I0.depth() == CV_8U ? 1.0 : 255.0);
78 I1.convertTo(I1s[0], CV_32F, I1.depth() == CV_8U ? 1.0 : 255.0);
79
80
81 if (!useInitialFlow)
82 {
83 flowx.create(I0.size(), CV_32FC1);
84 flowy.create(I0.size(), CV_32FC1);
85 }
86 //u1s_step != u2s_step
87 u1s[0] = flowx;
88 u2s[0] = flowy;
89
90 I1x_buf.create(I0.size(), CV_32FC1);
91 I1y_buf.create(I0.size(), CV_32FC1);
92
93 I1w_buf.create(I0.size(), CV_32FC1);
94 I1wx_buf.create(I0.size(), CV_32FC1);
95 I1wy_buf.create(I0.size(), CV_32FC1);
96
97 grad_buf.create(I0.size(), CV_32FC1);
98 rho_c_buf.create(I0.size(), CV_32FC1);
99
100 p11_buf.create(I0.size(), CV_32FC1);
101 p12_buf.create(I0.size(), CV_32FC1);
102 p21_buf.create(I0.size(), CV_32FC1);
103 p22_buf.create(I0.size(), CV_32FC1);
104
105 diff_buf.create(I0.size(), CV_32FC1);
106
107 // create the scales
108 for (int s = 1; s < nscales; ++s)
109 {
110 ocl::pyrDown(I0s[s - 1], I0s[s]);
111 ocl::pyrDown(I1s[s - 1], I1s[s]);
112
113 if (I0s[s].cols < 16 || I0s[s].rows < 16)
114 {
115 nscales = s;
116 break;
117 }
118
119 if (useInitialFlow)
120 {
121 ocl::pyrDown(u1s[s - 1], u1s[s]);
122 ocl::pyrDown(u2s[s - 1], u2s[s]);
123
124 ocl::multiply(0.5, u1s[s], u1s[s]);
125 ocl::multiply(0.5, u2s[s], u2s[s]);
126 }
127 }
128
129 // pyramidal structure for computing the optical flow
130 for (int s = nscales - 1; s >= 0; --s)
131 {
132 // compute the optical flow at the current scale
133 procOneScale(I0s[s], I1s[s], u1s[s], u2s[s]);
134
135 // if this was the last scale, finish now
136 if (s == 0)
137 break;
138
139 // otherwise, upsample the optical flow
140
141 // zoom the optical flow for the next finer scale
142 ocl::resize(u1s[s], u1s[s - 1], I0s[s - 1].size());
143 ocl::resize(u2s[s], u2s[s - 1], I0s[s - 1].size());
144
145 // scale the optical flow with the appropriate zoom factor
146 multiply(2, u1s[s - 1], u1s[s - 1]);
147 multiply(2, u2s[s - 1], u2s[s - 1]);
148
149 }

170void cv::ocl::OpticalFlowDual_TVL1_OCL::procOneScale(const oclMat &I0, const oclMat &I1, oclMat &u1, oclMat &u2)
171{
172 using namespace ocl_tvl1flow;
173
174 const double scaledEpsilon = epsilon * epsilon * I0.size().area();
175
176 CV_DbgAssert( I1.size() == I0.size() );
177 CV_DbgAssert( I1.type() == I0.type() );
178 CV_DbgAssert( u1.empty() || u1.size() == I0.size() );
179 CV_DbgAssert( u2.size() == u1.size() );
180
181 if (u1.empty())
182 {
183 u1.create(I0.size(), CV_32FC1);
184 u1.setTo(Scalar::all(0));
185
186 u2.create(I0.size(), CV_32FC1);
187 u2.setTo(Scalar::all(0));
188 }
189
190 oclMat I1x = I1x_buf(Rect(0, 0, I0.cols, I0.rows));
191 oclMat I1y = I1y_buf(Rect(0, 0, I0.cols, I0.rows));
192
193 centeredGradient(I1, I1x, I1y);
194
195 oclMat I1w = I1w_buf(Rect(0, 0, I0.cols, I0.rows));
196 oclMat I1wx = I1wx_buf(Rect(0, 0, I0.cols, I0.rows));
197 oclMat I1wy = I1wy_buf(Rect(0, 0, I0.cols, I0.rows));
198
199 oclMat grad = grad_buf(Rect(0, 0, I0.cols, I0.rows));
200 oclMat rho_c = rho_c_buf(Rect(0, 0, I0.cols, I0.rows));
201
202 oclMat p11 = p11_buf(Rect(0, 0, I0.cols, I0.rows));
203 oclMat p12 = p12_buf(Rect(0, 0, I0.cols, I0.rows));
204 oclMat p21 = p21_buf(Rect(0, 0, I0.cols, I0.rows));
205 oclMat p22 = p22_buf(Rect(0, 0, I0.cols, I0.rows));
206 p11.setTo(Scalar::all(0));
207 p12.setTo(Scalar::all(0));
208 p21.setTo(Scalar::all(0));
209 p22.setTo(Scalar::all(0));
210
211 oclMat diff = diff_buf(Rect(0, 0, I0.cols, I0.rows));
212
213 const float l_t = static_cast<float>(lambda * theta);
214 const float taut = static_cast<float>(tau / theta);
215
216 for (int warpings = 0; warpings < warps; ++warpings)
217 {
218 warpBackward(I0, I1, I1x, I1y, u1, u2, I1w, I1wx, I1wy, grad, rho_c);
219
220 double error = numeric_limits<double>::max();
221 double prev_error = 0;
222 for (int n = 0; error > scaledEpsilon && n < iterations; ++n)
223 {
224 // some tweaks to make sum operation less frequently
225 char calc_error = (n & 0x1) && (prev_error < scaledEpsilon);
226 estimateU(I1wx, I1wy, grad, rho_c, p11, p12, p21, p22,
227 u1, u2, diff, l_t, static_cast<float>(theta), calc_error);
228 if(calc_error)
229 {
230 error = ocl::sum(diff)[0];
231 prev_error = error;
232 }
233 else
234 {
235 error = numeric_limits<double>::max();
236 prev_error -= scaledEpsilon;
237 }
238 estimateDualVariables(u1, u2, p11, p12, p21, p22, taut);
239
240 }
241 }

272void ocl_tvl1flow::centeredGradient(const oclMat &src, oclMat &dx, oclMat &dy)
273{
274 Context *clCxt = src.clCxt;
275 size_t localThreads[3] = {32, 8, 1};
276 size_t globalThreads[3] = {(size_t)src.cols, (size_t)src.rows, 1};
277
278 int srcElementSize = src.elemSize();
279 int src_step = src.step/srcElementSize;
280
281 int dElememntSize = dx.elemSize();
282 int dx_step = dx.step/dElememntSize;
283
284 string kernelName = "centeredGradientKernel";
285 vector< pair<size_t, const void *> > args;
286 args.push_back( make_pair( sizeof(cl_mem), (void*)&src.data));
287 args.push_back( make_pair( sizeof(cl_int), (void*)&src.cols));
288 args.push_back( make_pair( sizeof(cl_int), (void*)&src.rows));
289 args.push_back( make_pair( sizeof(cl_int), (void*)&src_step));
290 args.push_back( make_pair( sizeof(cl_mem), (void*)&dx.data));
291 args.push_back( make_pair( sizeof(cl_mem), (void*)&dy.data));
292 args.push_back( make_pair( sizeof(cl_int), (void*)&dx_step));
293 openCLExecuteKernel(clCxt, &tvl1flow, kernelName, globalThreads, localThreads, args, -1, -1);
294
295}

409void ocl_tvl1flow::warpBackward(const oclMat &I0, const oclMat &I1, oclMat &I1x, oclMat &I1y, oclMat &u1, oclMat &u2, oclMat &I1w, oclMat &I1wx, oclMat &I1wy, oclMat &grad, oclMat &rho)
410{
411 Context* clCxt = I0.clCxt;
412
413 int u1ElementSize = u1.elemSize();
414 int u1Step = u1.step/u1ElementSize;
415
416 int u2ElementSize = u2.elemSize();
417 int u2Step = u2.step/u2ElementSize;
418
419 int I0ElementSize = I0.elemSize();
420 int I0Step = I0.step/I0ElementSize;
421
422 int I1w_element_size = I1w.elemSize();
423 int I1w_step = I1w.step/I1w_element_size;
424
425 int u1_offset_y = u1.offset/u1.step;
426 int u1_offset_x = u1.offset%u1.step;
427 u1_offset_x = u1_offset_x/u1.elemSize();
428
429 int u2_offset_y = u2.offset/u2.step;
430 int u2_offset_x = u2.offset%u2.step;
431 u2_offset_x = u2_offset_x/u2.elemSize();
432
433 size_t localThread[] = {32, 8, 1};
434 size_t globalThread[] =
435 {
436 (size_t)I0.cols,
437 (size_t)I0.rows,
438 1
439 };
440
441 cl_mem I1_tex;
442 cl_mem I1x_tex;
443 cl_mem I1y_tex;
444 I1_tex = bindTexture(I1);
445 I1x_tex = bindTexture(I1x);
446 I1y_tex = bindTexture(I1y);
447
448 string kernelName = "warpBackwardKernel";
449 vector< pair<size_t, const void *> > args;
450 args.push_back( make_pair( sizeof(cl_mem), (void*)&I0.data));
451 args.push_back( make_pair( sizeof(cl_int), (void*)&I0Step));
452 args.push_back( make_pair( sizeof(cl_int), (void*)&I0.cols));
453 args.push_back( make_pair( sizeof(cl_int), (void*)&I0.rows));
454 args.push_back( make_pair( sizeof(cl_mem), (void*)&I1_tex));
455 args.push_back( make_pair( sizeof(cl_mem), (void*)&I1x_tex));
456 args.push_back( make_pair( sizeof(cl_mem), (void*)&I1y_tex));
457 args.push_back( make_pair( sizeof(cl_mem), (void*)&u1.data));
458 args.push_back( make_pair( sizeof(cl_int), (void*)&u1Step));
459 args.push_back( make_pair( sizeof(cl_mem), (void*)&u2.data));
460 args.push_back( make_pair( sizeof(cl_mem), (void*)&I1w.data));
461 args.push_back( make_pair( sizeof(cl_mem), (void*)&I1wx.data));
462 args.push_back( make_pair( sizeof(cl_mem), (void*)&I1wy.data));
463 args.push_back( make_pair( sizeof(cl_mem), (void*)&grad.data));
464 args.push_back( make_pair( sizeof(cl_mem), (void*)&rho.data));
465 args.push_back( make_pair( sizeof(cl_int), (void*)&I1w_step));
466 args.push_back( make_pair( sizeof(cl_int), (void*)&u2Step));
467 args.push_back( make_pair( sizeof(cl_int), (void*)&u1_offset_x));
468 args.push_back( make_pair( sizeof(cl_int), (void*)&u1_offset_y));
469 args.push_back( make_pair( sizeof(cl_int), (void*)&u2_offset_x));
470 args.push_back( make_pair( sizeof(cl_int), (void*)&u2_offset_y));
471
472 openCLExecuteKernel(clCxt, &tvl1flow, kernelName, globalThread, localThread, args, -1, -1);
473
474 releaseTexture(I1_tex);
475 releaseTexture(I1x_tex);
476 releaseTexture(I1y_tex);
477}
