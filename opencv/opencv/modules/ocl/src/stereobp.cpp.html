62 typedef struct
63 {
64 int cndisp;
65 float cmax_data_term;
66 float cdata_weight;
67 float cmax_disc_term;
68 float cdisc_single_jump;
69 } con_struct_t;
70
71 cl_mem cl_con_struct = NULL;
72 static void load_constants(int ndisp, float max_data_term, float data_weight,
73 float max_disc_term, float disc_single_jump)
74 {
75 con_struct_t *con_struct = new con_struct_t;
76 con_struct -> cndisp = ndisp;
77 con_struct -> cmax_data_term = max_data_term;
78 con_struct -> cdata_weight = data_weight;
79 con_struct -> cmax_disc_term = max_disc_term;
80 con_struct -> cdisc_single_jump = disc_single_jump;
81
82 Context* clCtx = Context::getContext();
83 cl_context clContext = *(cl_context*)(clCtx->getOpenCLContextPtr());
84 cl_command_queue clCmdQueue = *(cl_command_queue*)(clCtx->getOpenCLCommandQueuePtr());
85 cl_con_struct = load_constant(clContext, clCmdQueue, (void *)con_struct,
86 sizeof(con_struct_t));
87
88 delete con_struct;
89 }
90 static void release_constants()
91 {
92 openCLFree(cl_con_struct);
93 }

98 static void comp_data_call(const oclMat &left, const oclMat &right, oclMat &data, int /*disp*/,
99 float /*cmax_data_term*/, float /*cdata_weight*/)
100 {
101 Context *clCxt = left.clCxt;
102 int channels = left.oclchannels();
103 int data_type = data.type();
104
105 string kernelName = "comp_data";
106
107 vector<pair<size_t , const void *> > args;
108
109 args.push_back( make_pair( sizeof(cl_mem) , (void *)&left.data));
110 args.push_back( make_pair( sizeof(cl_int) , (void *)&left.rows));
111 args.push_back( make_pair( sizeof(cl_int) , (void *)&left.cols));
112 args.push_back( make_pair( sizeof(cl_int) , (void *)&left.step));
113 args.push_back( make_pair( sizeof(cl_mem) , (void *)&right.data));
114 args.push_back( make_pair( sizeof(cl_int) , (void *)&right.step));
115 args.push_back( make_pair( sizeof(cl_mem) , (void *)&data.data));
116 args.push_back( make_pair( sizeof(cl_int) , (void *)&data.step));
117 args.push_back( make_pair( sizeof(cl_mem) , (void *)&cl_con_struct));
118
119 size_t gt[3] = {(size_t)left.cols, (size_t)left.rows, 1}, lt[3] = {16, 16, 1};
120
121 const int OPT_SIZE = 50;
122 char cn_opt [OPT_SIZE] = "";
123 sprintf( cn_opt, "%s -D CN=%d",
124 (data_type == CV_16S ? "-D T_SHORT":"-D T_FLOAT"),
125 channels
126 );
127 openCLExecuteKernel(clCxt, &stereobp, kernelName, gt, lt, args, -1, -1, cn_opt);
128 }

132 static void data_step_down_call(int dst_cols, int dst_rows, int src_rows,
133 const oclMat &src, oclMat &dst, int disp)

135 Context *clCxt = src.clCxt;
136 int data_type = src.type();
137
138 string kernelName = "data_step_down";
139
140 vector<pair<size_t , const void *> > args;
141
142 args.push_back( make_pair( sizeof(cl_mem) , (void *)&src.data));
143 args.push_back( make_pair( sizeof(cl_int) , (void *)&src_rows));
144 args.push_back( make_pair( sizeof(cl_mem) , (void *)&dst.data));
145 args.push_back( make_pair( sizeof(cl_int) , (void *)&dst_rows));
146 args.push_back( make_pair( sizeof(cl_int) , (void *)&dst_cols));
147 args.push_back( make_pair( sizeof(cl_int) , (void *)&src.step));
148 args.push_back( make_pair( sizeof(cl_int) , (void *)&dst.step));
149 args.push_back( make_pair( sizeof(cl_int) , (void *)&disp));
150
151 size_t gt[3] = {(size_t)dst_cols, (size_t)dst_rows, 1}, lt[3] = {16, 16, 1};
152 const char* t_opt = data_type == CV_16S ? "-D T_SHORT":"-D T_FLOAT";
153 openCLExecuteKernel(clCxt, &stereobp, kernelName, gt, lt, args, -1, -1, t_opt);
154 }

180 static void level_up_messages_calls(int dst_idx, int dst_cols, int dst_rows, int src_rows,
181 oclMat *mus, oclMat *mds, oclMat *mls, oclMat *mrs,
182 int ndisp)
183 {
184 int src_idx = (dst_idx + 1) & 1;
185
186 level_up_message_call(dst_cols, dst_rows, src_rows,
187 mus[src_idx], mus[dst_idx], ndisp);
188
189 level_up_message_call(dst_cols, dst_rows, src_rows,
190 mds[src_idx], mds[dst_idx], ndisp);
191
192 level_up_message_call(dst_cols, dst_rows, src_rows,
193 mls[src_idx], mls[dst_idx], ndisp);
194
195 level_up_message_call(dst_cols, dst_rows, src_rows,
196 mrs[src_idx], mrs[dst_idx], ndisp);
197 }

201 static void calc_all_iterations_call(int cols, int rows, oclMat &u, oclMat &d,
202 oclMat &l, oclMat &r, oclMat &data,
203 int t, int cndisp, float cmax_disc_term,
204 float cdisc_single_jump)
205 {
206 Context *clCxt = l.clCxt;

208
209 string kernelName = "one_iteration";
210
211 vector<pair<size_t , const void *> > args;
212
213 args.push_back( make_pair( sizeof(cl_mem) , (void *)&u.data));
214 args.push_back( make_pair( sizeof(cl_int) , (void *)&u.step));
215 args.push_back( make_pair( sizeof(cl_mem) , (void *)&data.data));
216 args.push_back( make_pair( sizeof(cl_int) , (void *)&data.step));
217 args.push_back( make_pair( sizeof(cl_mem) , (void *)&d.data));
218 args.push_back( make_pair( sizeof(cl_mem) , (void *)&l.data));
219 args.push_back( make_pair( sizeof(cl_mem) , (void *)&r.data));
220 args.push_back( make_pair( sizeof(cl_int) , (void *)&t));
221 args.push_back( make_pair( sizeof(cl_int) , (void *)&cols));
222 args.push_back( make_pair( sizeof(cl_int) , (void *)&rows));
223 args.push_back( make_pair( sizeof(cl_float) , (void *)&cmax_disc_term));
224 args.push_back( make_pair( sizeof(cl_float) , (void *)&cdisc_single_jump));
225
226 size_t gt[3] = {(size_t)cols, (size_t)rows, 1}, lt[3] = {16, 16, 1};
227 char opt[80] = "";
228 sprintf(opt, "-D %s -D CNDISP=%d", data_type == CV_16S ? "T_SHORT":"T_FLOAT", cndisp);
229 openCLExecuteKernel(clCxt, &stereobp, kernelName, gt, lt, args, -1, -1, opt);
230 }

232 static void calc_all_iterations_calls(int cols, int rows, int iters, oclMat &u,
233 oclMat &d, oclMat &l, oclMat &r,
234 oclMat &data, int cndisp, float cmax_disc_term,
235 float cdisc_single_jump)
236 {
237 for(int t = 0; t < iters; ++t)
238 calc_all_iterations_call(cols, rows, u, d, l, r, data, t, cndisp,
239 cmax_disc_term, cdisc_single_jump);
240 }

281void cv::ocl::StereoBeliefPropagation::estimateRecommendedParams(int width, int height, int &ndisp, int &iters, int &levels)
282{
283 ndisp = width / 4;
284 if ((ndisp & 1) != 0)
285 ndisp++;
286
287 int mm = ::max(width, height);
288 iters = mm / 100 + 2;
289
290 levels = (int)(::log(static_cast<double>(mm)) + 1) * 4 / 5;
291 if (levels == 0) levels++;
292}

312 class StereoBeliefPropagationImpl
313 {
314 public:
315 StereoBeliefPropagationImpl(StereoBeliefPropagation &rthis_,
316 oclMat &u_, oclMat &d_, oclMat &l_, oclMat &r_,
317 oclMat &u2_, oclMat &d2_, oclMat &l2_, oclMat &r2_,
318 vector<oclMat> &datas_, oclMat &out_)
319 : rthis(rthis_), u(u_), d(d_), l(l_), r(r_), u2(u2_), d2(d2_), l2(l2_), r2(r2_), datas(datas_), out(out_),
320 zero(Scalar::all(0)), scale(rthis_.msg_type == CV_32F ? 1.0f : 10.0f)
321 {
322 CV_Assert(0 < rthis.ndisp && 0 < rthis.iters && 0 < rthis.levels);
323 CV_Assert(rthis.msg_type == CV_32F || rthis.msg_type == CV_16S);
324 CV_Assert(rthis.msg_type == CV_32F || (1 << (rthis.levels - 1)) * scale * rthis.max_data_term < numeric_limits<short>::max());
325 }
326
327 void operator()(const oclMat &left, const oclMat &right, oclMat &disp)
328 {
329 CV_Assert(left.size() == right.size() && left.type() == right.type());
330 CV_Assert(left.type() == CV_8UC1 || left.type() == CV_8UC3 || left.type() == CV_8UC4);
331
332 rows = left.rows;
333 cols = left.cols;
334
335 int divisor = (int)pow(2.f, rthis.levels - 1.0f);
336 int lowest_cols = cols / divisor;
337 int lowest_rows = rows / divisor;
338 const int min_image_dim_size = 2;
339 CV_Assert(min(lowest_cols, lowest_rows) > min_image_dim_size);
340
341 init();
342
343 datas[0].create(rows * rthis.ndisp, cols, rthis.msg_type);
344 datas[0].setTo(Scalar_<short>::all(0));
345
346 cv::ocl::stereoBP::comp_data_call(left, right, datas[0], rthis.ndisp, rthis.max_data_term, scale * rthis.data_weight);
347 calcBP(disp);
348 }
349
350 void operator()(const oclMat &data, oclMat &disp)
351 {
352 CV_Assert((data.type() == rthis.msg_type) && (data.rows % rthis.ndisp == 0));
353
354 rows = data.rows / rthis.ndisp;
355 cols = data.cols;
356
357 int divisor = (int)pow(2.f, rthis.levels - 1.0f);
358 int lowest_cols = cols / divisor;
359 int lowest_rows = rows / divisor;
360 const int min_image_dim_size = 2;
361 CV_Assert(min(lowest_cols, lowest_rows) > min_image_dim_size);
362
363 init();
364
365 datas[0] = data;
366
367 calcBP(disp);
368 }
369 private:
370 void init()
371 {
372 u.create(rows * rthis.ndisp, cols, rthis.msg_type);
373 d.create(rows * rthis.ndisp, cols, rthis.msg_type);
374 l.create(rows * rthis.ndisp, cols, rthis.msg_type);
375 r.create(rows * rthis.ndisp, cols, rthis.msg_type);
376
377 if (rthis.levels & 1)
378 {
379 //can clear less area
380 u = zero;
381 d = zero;
382 l = zero;
383 r = zero;
384 }
385
386 if (rthis.levels > 1)
387 {
388 int less_rows = (rows + 1) / 2;
389 int less_cols = (cols + 1) / 2;
390
391 u2.create(less_rows * rthis.ndisp, less_cols, rthis.msg_type);
392 d2.create(less_rows * rthis.ndisp, less_cols, rthis.msg_type);
393 l2.create(less_rows * rthis.ndisp, less_cols, rthis.msg_type);
394 r2.create(less_rows * rthis.ndisp, less_cols, rthis.msg_type);
395
396 if ((rthis.levels & 1) == 0)
397 {
398 u2 = zero;
399 d2 = zero;
400 l2 = zero;
401 r2 = zero;
402 }
403 }
404
405 cv::ocl::stereoBP::load_constants(rthis.ndisp, rthis.max_data_term, scale * rthis.data_weight,
406 scale * rthis.max_disc_term, scale * rthis.disc_single_jump);
407
408 datas.resize(rthis.levels);
409 cols_all.resize(rthis.levels);
410 rows_all.resize(rthis.levels);
411
412 cols_all[0] = cols;
413 rows_all[0] = rows;
414 }
415
416 void calcBP(oclMat &disp)
417 {
418 using namespace cv::ocl::stereoBP;
419
420 for (int i = 1; i < rthis.levels; ++i)
421 {
422 cols_all[i] = (cols_all[i - 1] + 1) / 2;
423 rows_all[i] = (rows_all[i - 1] + 1) / 2;
424
425 datas[i].create(rows_all[i] * rthis.ndisp, cols_all[i], rthis.msg_type);
426 datas[i].setTo(Scalar_<short>::all(0));
427
428 data_step_down_call(cols_all[i], rows_all[i], rows_all[i - 1],
429 datas[i - 1], datas[i], rthis.ndisp);
430 }
431
432 oclMat mus[] = {u, u2};
433 oclMat mds[] = {d, d2};
434 oclMat mrs[] = {r, r2};
435 oclMat mls[] = {l, l2};
436
437 int mem_idx = (rthis.levels & 1) ? 0 : 1;
438
439 for (int i = rthis.levels - 1; i >= 0; --i)
440 {
441 // for lower level we have already computed messages by setting to zero
442 if (i != rthis.levels - 1)
443 level_up_messages_calls(mem_idx, cols_all[i], rows_all[i], rows_all[i + 1],
444 mus, mds, mls, mrs, rthis.ndisp);
445
446 calc_all_iterations_calls(cols_all[i], rows_all[i], rthis.iters, mus[mem_idx],
447 mds[mem_idx], mls[mem_idx], mrs[mem_idx], datas[i],
448 rthis.ndisp, scale * rthis.max_disc_term,
449 scale * rthis.disc_single_jump);
450
451 mem_idx = (mem_idx + 1) & 1;
452 }
453 if (disp.empty())
454 disp.create(rows, cols, CV_16S);
455
456 out = ((disp.type() == CV_16S) ? disp : (out.create(rows, cols, CV_16S), out));
457 out = zero;
458
459 output_call(u, d, l, r, datas.front(), out, rthis.ndisp);
460
461 if (disp.type() != CV_16S)
462 out.convertTo(disp, disp.type());
463
464 release_constants();
465 }

470 oclMat &u;
471 oclMat &d;
472 oclMat &l;
473 oclMat &r;
474
475 oclMat &u2;
476 oclMat &d2;
477 oclMat &l2;
478 oclMat &r2;
479
480 vector<oclMat> &datas;
481 oclMat &out;
482
483 const Scalar zero;
484 const float scale;
485
486 int rows, cols;
487
488 vector<int> cols_all, rows_all;
489 };
