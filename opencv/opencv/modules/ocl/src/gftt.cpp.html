51// currently sort procedure on the host is more efficient
52static bool use_cpu_sorter = true;

54// compact structure for corners
55struct DefCorner
56{
57 float eig; //eigenvalue of corner
58 short x; //x coordinate of corner point
59 short y; //y coordinate of corner point
60} ;
61
62// compare procedure for corner
63//it is used for sort on the host side
64struct DefCornerCompare
65{
66 bool operator()(const DefCorner a, const DefCorner b) const
67 {
68 return a.eig > b.eig;
69 }
70};

72// sort corner point using opencl bitonicosrt implementation
73static void sortCorners_caller(oclMat& corners, const int count)
74{
75 Context * cxt = Context::getContext();
76 int GS = count/2;
77 int LS = min(255,GS);
78 size_t globalThreads[3] = {(size_t)GS, 1, 1};
79 size_t localThreads[3] = {(size_t)LS, 1, 1};
80
81 // 2^numStages should be equal to count or the output is invalid
82 int numStages = 0;
83 for(int i = count; i > 1; i >>= 1)
84 {
85 ++numStages;
86 }
87 const int argc = 4;
88 std::vector< std::pair<size_t, const void *> > args(argc);
89 std::string kernelname = "sortCorners_bitonicSort";
90 args[0] = std::make_pair(sizeof(cl_mem), (void *)&corners.data);
91 args[1] = std::make_pair(sizeof(cl_int), (void *)&count);
92 for(int stage = 0; stage < numStages; ++stage)
93 {
94 args[2] = std::make_pair(sizeof(cl_int), (void *)&stage);
95 for(int passOfStage = 0; passOfStage < stage + 1; ++passOfStage)
96 {
97 args[3] = std::make_pair(sizeof(cl_int), (void *)&passOfStage);
98 openCLExecuteKernel(cxt, &imgproc_gftt, kernelname, globalThreads, localThreads, args, -1, -1);
99 }
100 }
101}

103// find corners on matrix and put it into array
104static void findCorners_caller(
105 const oclMat& eig_mat, //input matrix worth eigenvalues
106 oclMat& eigMinMax, //input with min and max values of eigenvalues
107 const float qualityLevel,
108 const oclMat& mask,
109 oclMat& corners, //output array with detected corners
110 oclMat& counter) //output value with number of detected corners, have to be 0 before call
111{
112 string opt;
113 Context * cxt = Context::getContext();
114
115 std::vector< std::pair<size_t, const void*> > args;
116
117 const int mask_strip = mask.step / mask.elemSize1();
118
119 args.push_back(make_pair( sizeof(cl_mem), (void*)&(eig_mat.data)));
120
121 int src_pitch = (int)eig_mat.step;
122 args.push_back(make_pair( sizeof(cl_int), (void*)&src_pitch ));
123 args.push_back(make_pair( sizeof(cl_mem), (void*)&mask.data ));
124 args.push_back(make_pair( sizeof(cl_mem), (void*)&corners.data ));
125 args.push_back(make_pair( sizeof(cl_int), (void*)&mask_strip));
126 args.push_back(make_pair( sizeof(cl_mem), (void*)&eigMinMax.data ));
127 args.push_back(make_pair( sizeof(cl_float), (void*)&qualityLevel ));
128 args.push_back(make_pair( sizeof(cl_int), (void*)&eig_mat.rows ));
129 args.push_back(make_pair( sizeof(cl_int), (void*)&eig_mat.cols ));
130 args.push_back(make_pair( sizeof(cl_int), (void*)&corners.cols ));
131 args.push_back(make_pair( sizeof(cl_mem), (void*)&counter.data ));
132
133 size_t globalThreads[3] = {(size_t)eig_mat.cols, (size_t)eig_mat.rows, 1};
134 size_t localThreads[3] = {16, 16, 1};
135 if(!mask.empty())
136 opt += " -D WITH_MASK=1";
137
138 openCLExecuteKernel(cxt, &imgproc_gftt, "findCorners", globalThreads, localThreads, args, -1, -1, opt.c_str());
139}

142static void minMaxEig_caller(const oclMat &src, oclMat &dst, oclMat & tozero)
143{
144 size_t groupnum = src.clCxt->getDeviceInfo().maxComputeUnits;
145 CV_Assert(groupnum != 0);
146
147 int dbsize = groupnum * 2 * src.elemSize();
148 ensureSizeIsEnough(1, dbsize, CV_8UC1, dst);
149
150 cl_mem dst_data = reinterpret_cast<cl_mem>(dst.data);
151
152 int vElemSize = src.elemSize1();
153 int src_step = src.step / vElemSize, src_offset = src.offset / vElemSize;
154 int total = src.size().area();
155
156 {
157 // first parallel pass
158 vector<pair<size_t , const void *> > args;
159 args.push_back( make_pair( sizeof(cl_mem) , (void *)&src.data));
160 args.push_back( make_pair( sizeof(cl_int) , (void *)&src_step));
161 args.push_back( make_pair( sizeof(cl_int) , (void *)&src_offset));
162 args.push_back( make_pair( sizeof(cl_int) , (void *)&src.rows ));
163 args.push_back( make_pair( sizeof(cl_int) , (void *)&src.cols ));
164 args.push_back( make_pair( sizeof(cl_int) , (void *)&total));
165 args.push_back( make_pair( sizeof(cl_int) , (void *)&groupnum));
166 args.push_back( make_pair( sizeof(cl_mem) , (void *)&dst_data ));
167 size_t globalThreads[3] = {(size_t)groupnum * 256, 1, 1};
168 size_t localThreads[3] = {256, 1, 1};
169 openCLExecuteKernel(src.clCxt, &arithm_minMax, "arithm_op_minMax", globalThreads, localThreads,
170 args, -1, -1, "-D T=float -D DEPTH_5 -D vlen=1");
171 }
172
173 {
174 // run final "serial" kernel to find accumulate results from threads and reset corner counter
175 vector<pair<size_t , const void *> > args;
176 args.push_back( make_pair( sizeof(cl_mem) , (void *)&dst_data ));
177 args.push_back( make_pair( sizeof(cl_int) , (void *)&groupnum ));
178 args.push_back( make_pair( sizeof(cl_mem) , (void *)&tozero.data ));
179 size_t globalThreads[3] = {1, 1, 1};
180 size_t localThreads[3] = {1, 1, 1};
181 openCLExecuteKernel(src.clCxt, &imgproc_gftt, "arithm_op_minMax_final", globalThreads, localThreads,
182 args, -1, -1);
183 }

186void cv::ocl::GoodFeaturesToTrackDetector_OCL::operator ()(const oclMat& image, oclMat& corners, const oclMat& mask)
187{
188 CV_Assert(qualityLevel > 0 && minDistance >= 0 && maxCorners >= 0);
189 CV_Assert(mask.empty() || (mask.type() == CV_8UC1 && mask.size() == image.size()));
190
191 ensureSizeIsEnough(image.size(), CV_32F, eig_);
192
193 if (useHarrisDetector)
194 cornerHarris_dxdy(image, eig_, Dx_, Dy_, blockSize, 3, harrisK);
195 else
196 cornerMinEigenVal_dxdy(image, eig_, Dx_, Dy_, blockSize, 3);
197
198 ensureSizeIsEnough(1,1, CV_32SC1, counter_);
199
200 // find max eigenvalue and reset detected counters
201 minMaxEig_caller(eig_,eig_minmax_,counter_);
202
203 // allocate buffer for kernels
204 int corner_array_size = std::max(1024, static_cast<int>(image.size().area() * 0.05));
205
206 if(!use_cpu_sorter)
207 { // round to 2^n
208 unsigned int n=1;
209 for(n=1;n<(unsigned int)corner_array_size;n<<=1) ;
210 corner_array_size = (int)n;
211
212 ensureSizeIsEnough(1, corner_array_size , CV_32FC2, tmpCorners_);
213
214 // set to 0 to be able use bitonic sort on whole 2^n array
215 tmpCorners_.setTo(0);
216 }
217 else
218 {
219 ensureSizeIsEnough(1, corner_array_size , CV_32FC2, tmpCorners_);
220 }
221
222 int total = tmpCorners_.cols; // by default the number of corner is full array
223 vector<DefCorner> tmp(tmpCorners_.cols); // input buffer with corner for HOST part of algorithm
224
225 //find points with high eigenvalue and put it into the output array
226 findCorners_caller(
227 eig_,
228 eig_minmax_,
229 static_cast<float>(qualityLevel),
230 mask,
231 tmpCorners_,
232 counter_);
233
234 if(!use_cpu_sorter)
235 {// sort detected corners on deivce side
236 sortCorners_caller(tmpCorners_, corner_array_size);
237 }
238 else
239 {// send non-blocking request to read real non-zero number of corners to sort it on the HOST side
240 openCLVerifyCall(clEnqueueReadBuffer(getClCommandQueue(counter_.clCxt), (cl_mem)counter_.data, CL_FALSE, 0,sizeof(int), &total, 0, NULL, NULL));
241 }
242
243 //blocking read whole corners array (sorted or not sorted)
244 openCLReadBuffer(tmpCorners_.clCxt,(cl_mem)tmpCorners_.data,&tmp[0],tmpCorners_.cols*sizeof(DefCorner));
245
246 if (total == 0)
247 {// check for trivial case
248 corners.release();
249 return;
250 }
251
252 if(use_cpu_sorter)
253 {// sort detected corners on cpu side.
254 tmp.resize(total);
255 cv::sort(tmp,DefCornerCompare());
256 }
257
258 //estimate maximal size of final output array
259 int total_max = maxCorners > 0 ? std::min(maxCorners, total) : total;
260 int D2 = (int)ceil(minDistance * minDistance);
261 // allocate output buffer
262 vector<Point2f> tmp2;
263 tmp2.reserve(total_max);
264
265
266 if (minDistance < 1)
267 {// we have not distance restriction. then just copy with conversion maximal allowed points into output array
268 for(int i=0;i<total_max && tmp[i].eig>0.0f;++i)
269 {
270 tmp2.push_back(Point2f(tmp[i].x,tmp[i].y));
271 }
272 }
273 else
274 {// we have distance restriction. then start coping to output array from the first element and check distance for each next one
275 const int cell_size = cvRound(minDistance);
276 const int grid_width = (image.cols + cell_size - 1) / cell_size;
277 const int grid_height = (image.rows + cell_size - 1) / cell_size;
278
279 std::vector< std::vector<Point2i> > grid(grid_width * grid_height);
280
281 for (int i = 0; i < total ; ++i)
282 {
283 DefCorner p = tmp[i];
284
285 if(p.eig<=0.0f)
286 break; // condition to stop that is needed for GPU bitonic sort usage.
287
288 bool good = true;
289
290 int x_cell = static_cast<int>(p.x / cell_size);
291 int y_cell = static_cast<int>(p.y / cell_size);
292
293 int x1 = x_cell - 1;
294 int y1 = y_cell - 1;
295 int x2 = x_cell + 1;
296 int y2 = y_cell + 1;
297
298 // boundary check
299 x1 = std::max(0, x1);
300 y1 = std::max(0, y1);
301 x2 = std::min(grid_width - 1, x2);
302 y2 = std::min(grid_height - 1, y2);
303
304 for (int yy = y1; yy <= y2; yy++)
305 {
306 for (int xx = x1; xx <= x2; xx++)
307 {
308 vector<Point2i>& m = grid[yy * grid_width + xx];
309 if (m.empty())
310 continue;
311 for(size_t j = 0; j < m.size(); j++)
312 {
313 int dx = p.x - m[j].x;
314 int dy = p.y - m[j].y;
315
316 if (dx * dx + dy * dy < D2)
317 {
318 good = false;
319 goto break_out_;
320 }
321 }
322 }
323 }
324
325 break_out_:
326
327 if(good)
328 {
329 grid[y_cell * grid_width + x_cell].push_back(Point2i(p.x,p.y));
330
331 tmp2.push_back(Point2f(p.x,p.y));
332
333 if (maxCorners > 0 && tmp2.size() == static_cast<size_t>(maxCorners))
334 break;
335 }
336 }
337
338 }
339 int final_size = static_cast<int>(tmp2.size());
340 if(final_size>0)
341 corners.upload(Mat(1, final_size, CV_32FC2, &tmp2[0]));
342 else
343 corners.release();
344}
345void cv::ocl::GoodFeaturesToTrackDetector_OCL::downloadPoints(const oclMat &points, vector<Point2f> &points_v)
346{
347 CV_DbgAssert(points.type() == CV_32FC2);
348 points_v.resize(points.cols);
349 openCLSafeCall(clEnqueueReadBuffer(
350 *(cl_command_queue*)getClCommandQueuePtr(),
351 reinterpret_cast<cl_mem>(points.data),
352 CL_TRUE,
353 0,
354 points.cols * sizeof(Point2f),
355 &points_v[0],
356 0,
357 NULL,
358 NULL));
359}
