67 enum FftType
68 {
69 C2R = 1, // complex to complex
70 R2C = 2, // real to opencl HERMITIAN_INTERLEAVED
71 C2C = 3 // opencl HERMITIAN_INTERLEAVED to real
72 };
73 struct FftPlan
74 {
75 protected:
76 clAmdFftPlanHandle plHandle;
77 FftPlan& operator=(const FftPlan&);
78 public:
79 FftPlan(Size _dft_size, int _src_step, int _dst_step, int _flags, FftType _type);
80 ~FftPlan();
81 inline clAmdFftPlanHandle getPlanHandle() { return plHandle; }
82
83 const Size dft_size;
84 const int src_step, dst_step;
85 const int flags;
86 const FftType type;
87 };
88 class PlanCache
89 {
90 protected:
91 PlanCache();
92 ~PlanCache();
93 static PlanCache* planCache;
94
95 bool started;
96 vector<FftPlan *> planStore;
97 clAmdFftSetupData *setupData;
98 public:
99 friend void fft_setup();
100 friend void fft_teardown();
101
102 static PlanCache* getPlanCache()
103 {
104 if (NULL == planCache)
105 planCache = new PlanCache();
106 return planCache;
107 }
108 // return a baked plan->
109 // if there is one matched plan, return it
110 // if not, bake a new one, put it into the planStore and return it.
111 static FftPlan* getPlan(Size _dft_size, int _src_step, int _dst_step, int _flags, FftType _type);
112
113 // remove a single plan from the store
114 // return true if the plan is successfully removed
115 // else
116 static bool removePlan(clAmdFftPlanHandle );
117 };
118 }

122void cv::ocl::fft_setup()
123{
124 PlanCache& pCache = *PlanCache::getPlanCache();
125 if(pCache.started)
126 {
127 return;
128 }
129 if (pCache.setupData == NULL)
130 pCache.setupData = new clAmdFftSetupData;
131 openCLSafeCall(clAmdFftInitSetupData( pCache.setupData ));
132 pCache.started = true;
133}
134void cv::ocl::fft_teardown()
135{
136 PlanCache& pCache = *PlanCache::getPlanCache();
137
138 if(!pCache.started)
139 return;
140
141 for(size_t i = 0; i < pCache.planStore.size(); i ++)
142 delete pCache.planStore[i];
143 pCache.planStore.clear();
144
145 try
146 {
147 openCLSafeCall( clAmdFftTeardown( ) );
148 }
149 catch (const std::bad_alloc &)
150 { }
151
152 delete pCache.setupData; pCache.setupData = NULL;
153 pCache.started = false;
154}

156// bake a new plan
157cv::ocl::FftPlan::FftPlan(Size _dft_size, int _src_step, int _dst_step, int _flags, FftType _type)
158 : plHandle(0), dft_size(_dft_size), src_step(_src_step), dst_step(_dst_step), flags(_flags), type(_type)
159{
160 fft_setup();
161
162 bool is_1d_input = (_dft_size.height == 1);
163 int is_row_dft = flags & DFT_ROWS;
164 int is_scaled_dft = flags & DFT_SCALE;
165 int is_inverse = flags & DFT_INVERSE;
166
167 //clAmdFftResultLocation place;
168 clAmdFftLayout inLayout;
169 clAmdFftLayout outLayout;
170 clAmdFftDim dim = is_1d_input || is_row_dft ? CLFFT_1D : CLFFT_2D;
171
172 size_t batchSize = is_row_dft ? dft_size.height : 1;
173 size_t clLengthsIn[ 3 ] = {1, 1, 1};
174 size_t clStridesIn[ 3 ] = {1, 1, 1};
175 //size_t clLengthsOut[ 3 ] = {1, 1, 1};
176 size_t clStridesOut[ 3 ] = {1, 1, 1};
177 clLengthsIn[0] = dft_size.width;
178 clLengthsIn[1] = is_row_dft ? 1 : dft_size.height;
179 clStridesIn[0] = 1;
180 clStridesOut[0] = 1;
181
182 switch(_type)
183 {
184 case C2C:
185 inLayout = CLFFT_COMPLEX_INTERLEAVED;
186 outLayout = CLFFT_COMPLEX_INTERLEAVED;
187 clStridesIn[1] = src_step / sizeof(std::complex<float>);
188 clStridesOut[1] = clStridesIn[1];
189 break;
190 case R2C:
191 inLayout = CLFFT_REAL;
192 outLayout = CLFFT_HERMITIAN_INTERLEAVED;
193 clStridesIn[1] = src_step / sizeof(float);
194 clStridesOut[1] = dst_step / sizeof(std::complex<float>);
195 break;
196 case C2R:
197 inLayout = CLFFT_HERMITIAN_INTERLEAVED;
198 outLayout = CLFFT_REAL;
199 clStridesIn[1] = src_step / sizeof(std::complex<float>);
200 clStridesOut[1] = dst_step / sizeof(float);
201 break;
202 default:
203 //std::runtime_error("does not support this conversion!");
204 cout << "Does not support this conversion!" << endl;
205 throw exception();
206 break;
207 }
208
209 clStridesIn[2] = is_row_dft ? clStridesIn[1] : dft_size.width * clStridesIn[1];
210 clStridesOut[2] = is_row_dft ? clStridesOut[1] : dft_size.width * clStridesOut[1];
211
212 openCLSafeCall( clAmdFftCreateDefaultPlan( &plHandle, *(cl_context*)getClContextPtr(), dim, clLengthsIn ) );
213
214 openCLSafeCall( clAmdFftSetResultLocation( plHandle, CLFFT_OUTOFPLACE ) );
215 openCLSafeCall( clAmdFftSetLayout( plHandle, inLayout, outLayout ) );
216 openCLSafeCall( clAmdFftSetPlanBatchSize( plHandle, batchSize ) );
217
218 openCLSafeCall( clAmdFftSetPlanInStride ( plHandle, dim, clStridesIn ) );
219 openCLSafeCall( clAmdFftSetPlanOutStride ( plHandle, dim, clStridesOut ) );
220 openCLSafeCall( clAmdFftSetPlanDistance ( plHandle, clStridesIn[ dim ], clStridesOut[ dim ]) );
221
222 float scale_ = is_scaled_dft ? 1.f / _dft_size.area() : 1.f;
223 openCLSafeCall( clAmdFftSetPlanScale ( plHandle, is_inverse ? CLFFT_BACKWARD : CLFFT_FORWARD, scale_ ) );
224
225 //ready to bake
226 openCLSafeCall( clAmdFftBakePlan( plHandle, 1, (cl_command_queue*)getClCommandQueuePtr(), NULL, NULL ) );
227}
228cv::ocl::FftPlan::~FftPlan()
229{
230 openCLSafeCall( clAmdFftDestroyPlan( &plHandle ) );
231}

245FftPlan* cv::ocl::PlanCache::getPlan(Size _dft_size, int _src_step, int _dst_step, int _flags, FftType _type)
246{
247 PlanCache& pCache = *PlanCache::getPlanCache();
248 vector<FftPlan *>& pStore = pCache.planStore;
249 // go through search
250 for(size_t i = 0; i < pStore.size(); i ++)
251 {
252 FftPlan *plan = pStore[i];
253 if(
254 plan->dft_size.width == _dft_size.width &&
255 plan->dft_size.height == _dft_size.height &&
256 plan->flags == _flags &&
257 plan->src_step == _src_step &&
258 plan->dst_step == _dst_step &&
259 plan->type == _type
260 )
261 {
262 return plan;
263 }
264 }
265 // no baked plan is found
266 FftPlan *newPlan = new FftPlan(_dft_size, _src_step, _dst_step, _flags, _type);
267 pStore.push_back(newPlan);
268 return newPlan;
269}

271bool cv::ocl::PlanCache::removePlan(clAmdFftPlanHandle plHandle)
272{
273 PlanCache& pCache = *PlanCache::getPlanCache();
274 vector<FftPlan *>& pStore = pCache.planStore;
275 for(size_t i = 0; i < pStore.size(); i ++)
276 {
277 if(pStore[i]->getPlanHandle() == plHandle)
278 {
279 pStore.erase(pStore.begin() + i);
280 delete pStore[i];
281 return true;
282 }
283 }
284 return false;
285}

287void cv::ocl::dft(const oclMat &src, oclMat &dst, Size dft_size, int flags)
288{
289 if(dft_size == Size(0, 0))
290 {
291 dft_size = src.size();
292 }
293 // check if the given dft size is of optimal dft size
294 CV_Assert(dft_size.area() == getOptimalDFTSize(dft_size.area()));
295
296 // the two flags are not compatible
297 CV_Assert( !((flags & DFT_SCALE) && (flags & DFT_ROWS)) );
298
299 // similar assertions with cuda module
300 CV_Assert(src.type() == CV_32F || src.type() == CV_32FC2);
301
302 //bool is_1d_input = (src.rows == 1);
303 //int is_row_dft = flags & DFT_ROWS;
304 //int is_scaled_dft = flags & DFT_SCALE;
305 int is_inverse = flags & DFT_INVERSE;
306 bool is_complex_input = src.channels() == 2;
307 bool is_complex_output = !(flags & DFT_REAL_OUTPUT);
308
309
310 // We don't support real-to-real transform
311 CV_Assert(is_complex_input || is_complex_output);
312 FftType type = (FftType)(is_complex_input << 0 | is_complex_output << 1);
313
314 switch(type)
315 {
316 case C2C:
317 dst.create(src.rows, src.cols, CV_32FC2);
318 break;
319 case R2C:
320 dst.create(src.rows, src.cols / 2 + 1, CV_32FC2);
321 break;
322 case C2R:
323 CV_Assert(dft_size.width / 2 + 1 == src.cols && dft_size.height == src.rows);
324 dst.create(src.rows, dft_size.width, CV_32FC1);
325 break;
326 default:
327 //std::runtime_error("does not support this conversion!");
328 cout << "Does not support this conversion!" << endl;
329 throw exception();
330 break;
331 }
332 clAmdFftPlanHandle plHandle = PlanCache::getPlan(dft_size, src.step, dst.step, flags, type)->getPlanHandle();
333
334 //get the buffersize
335 size_t buffersize = 0;
336 openCLSafeCall( clAmdFftGetTmpBufSize(plHandle, &buffersize ) );
337
338 //allocate the intermediate buffer
339 // TODO, bind this with the current FftPlan
340 cl_mem clMedBuffer = NULL;
341 if (buffersize)
342 {
343 cl_int medstatus;
344 clMedBuffer = clCreateBuffer ( *(cl_context*)(src.clCxt->getOpenCLContextPtr()), CL_MEM_READ_WRITE, buffersize, 0, &medstatus);
345 openCLSafeCall( medstatus );
346 }
347 cl_command_queue clq = *(cl_command_queue*)(src.clCxt->getOpenCLCommandQueuePtr());
348 openCLSafeCall( clAmdFftEnqueueTransform( plHandle,
349 is_inverse ? CLFFT_BACKWARD : CLFFT_FORWARD,
350 1,
351 &clq,
352 0, NULL, NULL,
353 (cl_mem *)&src.data, (cl_mem *)&dst.data, clMedBuffer ) );
354 openCLSafeCall( clFinish(clq) );
355 if(clMedBuffer)
356 {
357 openCLFree(clMedBuffer);
358 }
359 //fft_teardown();
360}
