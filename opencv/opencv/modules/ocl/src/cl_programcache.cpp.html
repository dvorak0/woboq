55/*
56 * The binary caching system to eliminate redundant program source compilation.
57 * Strictly, this is not a cache because we do not implement evictions right now.
58 * We shall add such features to trade-off memory consumption and performance when necessary.

64ProgramCache* _programCache = NULL;
65ProgramCache* ProgramCache::getProgramCache()
66{
67 if (NULL == _programCache)
68 {
69 cv::AutoLock lock(getInitializationMutex());
70 if (NULL == _programCache)
71 _programCache = new ProgramCache();
72 }
73 return _programCache;
74}
75
76ProgramCache::ProgramCache()
77{
78 codeCache.clear();
79 cacheSize = 0;
80}
81
82ProgramCache::~ProgramCache()
83{
84 releaseProgram();
85 if (this == _programCache)
86 {
87 cv::AutoLock lock(getInitializationMutex());
88 if (this == _programCache)
89 _programCache = NULL;
90 }
91}

93cl_program ProgramCache::progLookup(const string& srcsign)
94{
95 map<string, cl_program>::iterator iter;
96 iter = codeCache.find(srcsign);
97 if(iter != codeCache.end())
98 return iter->second;
99 else
100 return NULL;
101}
102
103void ProgramCache::addProgram(const string& srcsign, cl_program program)
104{
105 if (!progLookup(srcsign))
106 {
107 clRetainProgram(program);
108 codeCache.insert(map<string, cl_program>::value_type(srcsign, program));
109 }
110}
111
112void ProgramCache::releaseProgram()
113{
114 map<string, cl_program>::iterator iter;
115 for(iter = codeCache.begin(); iter != codeCache.end(); iter++)
116 {
117 openCLSafeCall(clReleaseProgram(iter->second));
118 }
119 codeCache.clear();
120 cacheSize = 0;
121}

123static bool enable_disk_cache = true;
124static String binpath = "";
125
126void setBinaryDiskCache(int mode, String path)
127{
128 enable_disk_cache = false;
129 binpath = "";
130
131 if(mode == CACHE_NONE)
132 {
133 return;
134 }
135 enable_disk_cache =
136#if defined(_DEBUG) || defined(DEBUG)
137 (mode & CACHE_DEBUG) == CACHE_DEBUG;
138#else
139 (mode & CACHE_RELEASE) == CACHE_RELEASE;
140#endif
141 if(enable_disk_cache && !path.empty())
142 {
143 binpath = path;
144 }
145}

152static const int MAX_ENTRIES = 64;

154struct ProgramFileCache
155{
156 struct CV_DECL_ALIGNED(1) ProgramFileHeader
157 {
158 int hashLength;
159 //char hash[];
160 };
161
162 struct CV_DECL_ALIGNED(1) ProgramFileTable
163 {
164 int numberOfEntries;
165 //int firstEntryOffset[];
166 };
167
168 struct CV_DECL_ALIGNED(1) ProgramFileConfigurationEntry
169 {
170 int nextEntry;
171 int dataSize;
172 int optionsLength;
173 //char options[];
174 // char data[];
175 };

177 string fileName_;
178 const char* hash_;
179 std::fstream f;
180
181 ProgramFileCache(const string& fileName, const char* hash)
182 : fileName_(fileName), hash_(hash)
183 {
184 if (hash_ != NULL)
185 {
186 f.open(fileName_.c_str(), ios::in|ios::out|ios::binary);
187 if(f.is_open())
188 {
189 int hashLength = 0;
190 f.read((char*)&hashLength, sizeof(int));
191 std::vector<char> fhash(hashLength + 1);
192 f.read(&fhash[0], hashLength);
193 if (f.eof() || strncmp(hash_, &fhash[0], hashLength) != 0)
194 {
195 f.close();
196 remove(fileName_.c_str());
197 return;
198 }
199 }
200 }
201 }

203 int getHash(const string& options)
204 {
205 int hash = 0;
206 for (size_t i = 0; i < options.length(); i++)
207 {
208 hash = (hash << 2) ^ (hash >> 17) ^ options[i];
209 }
210 return (hash + (hash >> 16)) & (MAX_ENTRIES - 1);
211 }
212
213 bool readConfigurationFromFile(const string& options, std::vector<char>& buf)
214 {
215 if (hash_ == NULL)
216 return false;
217
218 if (!f.is_open())
219 return false;
220
221 f.seekg(0, std::fstream::end);
222 size_t fileSize = (size_t)f.tellg();
223 if (fileSize == 0)
224 {
225 std::cerr << "Invalid file (empty): " << fileName_ << std::endl;
226 f.close();
227 remove(fileName_.c_str());
228 return false;
229 }
230 f.seekg(0, std::fstream::beg);
231
232 int hashLength = 0;
233 f.read((char*)&hashLength, sizeof(int));
234 CV_Assert(hashLength > 0);
235 f.seekg(sizeof(hashLength) + hashLength, std::fstream::beg);
236
237 int numberOfEntries = 0;
238 f.read((char*)&numberOfEntries, sizeof(int));
239 CV_Assert(numberOfEntries > 0);
240 if (numberOfEntries != MAX_ENTRIES)
241 {
242 std::cerr << "Invalid file: " << fileName_ << std::endl;
243 f.close();
244 remove(fileName_.c_str());
245 return false;
246 }
247
248 std::vector<int> firstEntryOffset(numberOfEntries);
249 f.read((char*)&firstEntryOffset[0], sizeof(int)*numberOfEntries);
250
251 int entryNum = getHash(options);
252
253 int entryOffset = firstEntryOffset[entryNum];
254 ProgramFileConfigurationEntry entry;
255 while (entryOffset > 0)
256 {
257 f.seekg(entryOffset, std::fstream::beg);
258 assert(sizeof(entry) == sizeof(int)*3);
259 f.read((char*)&entry, sizeof(entry));
260 std::vector<char> foptions(entry.optionsLength);
261 if ((int)options.length() == entry.optionsLength)
262 {
263 if (entry.optionsLength > 0)
264 f.read(&foptions[0], entry.optionsLength);
265 if (memcmp(&foptions[0], options.c_str(), entry.optionsLength) == 0)
266 {
267 buf.resize(entry.dataSize);
268 f.read(&buf[0], entry.dataSize);
269 f.seekg(0, std::fstream::beg);
270 return true;
271 }
272 }
273 if (entry.nextEntry <= 0)
274 break;
275 entryOffset = entry.nextEntry;
276 }
277 return false;
278 }

280 bool writeConfigurationToFile(const string& options, std::vector<char>& buf)
281 {
282 if (hash_ == NULL)
283 return true; // don't save programs without hash
284
285 if (!f.is_open())
286 {
287 f.open(fileName_.c_str(), ios::in|ios::out|ios::binary);
288 if (!f.is_open())
289 {
290 f.open(fileName_.c_str(), ios::out|ios::binary);
291 if (!f.is_open())
292 return false;
293 }
294 }
295
296 f.seekg(0, std::fstream::end);
297 size_t fileSize = (size_t)f.tellg();
298 if (fileSize == 0)
299 {
300 f.seekp(0, std::fstream::beg);
301 int hashLength = strlen(hash_);
302 f.write((char*)&hashLength, sizeof(int));
303 f.write(hash_, hashLength);
304
305 int numberOfEntries = MAX_ENTRIES;
306 f.write((char*)&numberOfEntries, sizeof(int));
307 std::vector<int> firstEntryOffset(MAX_ENTRIES, 0);
308 f.write((char*)&firstEntryOffset[0], sizeof(int)*numberOfEntries);
309 f.close();
310 f.open(fileName_.c_str(), ios::in|ios::out|ios::binary);
311 CV_Assert(f.is_open());
312 f.seekg(0, std::fstream::end);
313 fileSize = (size_t)f.tellg();
314 }
315 f.seekg(0, std::fstream::beg);
316
317 int hashLength = 0;
318 f.read((char*)&hashLength, sizeof(int));
319 CV_Assert(hashLength > 0);
320 f.seekg(sizeof(hashLength) + hashLength, std::fstream::beg);
321
322 int numberOfEntries = 0;
323 f.read((char*)&numberOfEntries, sizeof(int));
324 CV_Assert(numberOfEntries > 0);
325 if (numberOfEntries != MAX_ENTRIES)
326 {
327 std::cerr << "Invalid file: " << fileName_ << std::endl;
328 f.close();
329 remove(fileName_.c_str());
330 return false;
331 }
332
333 size_t tableEntriesOffset = (size_t)f.tellg();
334 std::vector<int> firstEntryOffset(numberOfEntries);
335 f.read((char*)&firstEntryOffset[0], sizeof(int)*numberOfEntries);
336
337 int entryNum = getHash(options);
338
339 int entryOffset = firstEntryOffset[entryNum];
340 ProgramFileConfigurationEntry entry;
341 while (entryOffset > 0)
342 {
343 f.seekg(entryOffset, std::fstream::beg);
344 assert(sizeof(entry) == sizeof(int)*3);
345 f.read((char*)&entry, sizeof(entry));
346 std::vector<char> foptions(entry.optionsLength);
347 if ((int)options.length() == entry.optionsLength)
348 {
349 if (entry.optionsLength > 0)
350 f.read(&foptions[0], entry.optionsLength);
351 CV_Assert(memcmp(&foptions, options.c_str(), entry.optionsLength) != 0);
352 }
353 if (entry.nextEntry <= 0)
354 break;
355 entryOffset = entry.nextEntry;
356 }
357 if (entryOffset > 0)
358 {
359 f.seekp(entryOffset, std::fstream::beg);
360 entry.nextEntry = fileSize;
361 f.write((char*)&entry, sizeof(entry));
362 }
363 else
364 {
365 firstEntryOffset[entryNum] = fileSize;
366 f.seekp(tableEntriesOffset, std::fstream::beg);
367 f.write((char*)&firstEntryOffset[0], sizeof(int)*numberOfEntries);
368 }
369 f.seekp(fileSize, std::fstream::beg);
370 entry.nextEntry = 0;
371 entry.dataSize = buf.size();
372 entry.optionsLength = options.length();
373 f.write((char*)&entry, sizeof(entry));
374 f.write(options.c_str(), entry.optionsLength);
375 f.write(&buf[0], entry.dataSize);
376 return true;
377 }

379 cl_program getOrBuildProgram(const Context* ctx, const cv::ocl::ProgramEntry* source, const string& options)
380 {
381 cl_int status = 0;
382 cl_program program = NULL;
383 std::vector<char> binary;
384 if (!enable_disk_cache || !readConfigurationFromFile(options, binary))
385 {
386 program = clCreateProgramWithSource(getClContext(ctx), 1, (const char**)&source->programStr, NULL, &status);
387 openCLVerifyCall(status);
388 cl_device_id device = getClDeviceID(ctx);
389 status = clBuildProgram(program, 1, &device, options.c_str(), NULL, NULL);
390 if(status == CL_SUCCESS)
391 {
392 if (enable_disk_cache)
393 {
394 size_t binarySize;
395 openCLSafeCall(clGetProgramInfo(program,
396 CL_PROGRAM_BINARY_SIZES,
397 sizeof(size_t),
398 &binarySize, NULL));
399
400 std::vector<char> localBinary(binarySize);
401
402 char* ptr = &localBinary[0];
403 openCLSafeCall(clGetProgramInfo(program,
404 CL_PROGRAM_BINARIES,
405 sizeof(char*),
406 &ptr,
407 NULL));
408
409 if (!writeConfigurationToFile(options, localBinary))
410 {
411 std::cerr << "Can't write data to file: " << fileName_ << std::endl;
412 }
413 }
414 }
415 }
416 else
417 {
418 cl_device_id device = getClDeviceID(ctx);
419 size_t size = binary.size();
420 const char* ptr = &binary[0];
421 program = clCreateProgramWithBinary(getClContext(ctx),
422 1, &device,
423 (const size_t *)&size, (const unsigned char **)&ptr,
424 NULL, &status);
425 openCLVerifyCall(status);
426 status = clBuildProgram(program, 1, &device, options.c_str(), NULL, NULL);
427 }
428
429 if(status != CL_SUCCESS)
430 {
431 if (status == CL_BUILD_PROGRAM_FAILURE || status == CL_INVALID_BUILD_OPTIONS)
432 {
433 size_t buildLogSize = 0;
434 openCLSafeCall(clGetProgramBuildInfo(program, getClDeviceID(ctx),
435 CL_PROGRAM_BUILD_LOG, 0, NULL, &buildLogSize));
436 std::vector<char> buildLog; buildLog.resize(buildLogSize);
437 memset(&buildLog[0], 0, buildLogSize);
438 openCLSafeCall(clGetProgramBuildInfo(program, getClDeviceID(ctx),
439 CL_PROGRAM_BUILD_LOG, buildLogSize, &buildLog[0], NULL));
440 std::cout << std::endl << "BUILD LOG: "
441 << (source->name ? source->name : "dynamic program") << ": "
442 << options << "\n";
443 std::cout << &buildLog[0] << endl;
444 }
445 openCLVerifyCall(status);
446 }
447 return program;
448 }

451cl_program ProgramCache::getProgram(const Context *ctx, const cv::ocl::ProgramEntry* source,
452 const char *build_options)
453{
454 stringstream src_sign;
455
456 if (source->name)
457 {
458 src_sign << source->name;
459 src_sign << getClContext(ctx);
460 if (NULL != build_options)
461 {
462 src_sign << "_" << build_options;
463 }
464
465 {
466 cv::AutoLock localLockCache(mutexCache);
467 cl_program program = ProgramCache::getProgramCache()->progLookup(src_sign.str());
468 if (!!program)
469 {
470 clRetainProgram(program);
471 return program;
472 }
473 }
474 }
475
476 cv::AutoLock lockCache(mutexFiles);
477
478 // second check
479 if (source->name)
480 {
481 cv::AutoLock localLockCache(mutexCache);
482 cl_program program = ProgramCache::getProgramCache()->progLookup(src_sign.str());
483 if (!!program)
484 {
485 clRetainProgram(program);
486 return program;
487 }
488 }
489
490 string all_build_options;
491 if (!ctx->getDeviceInfo().compilationExtraOptions.empty())
492 all_build_options += ctx->getDeviceInfo().compilationExtraOptions;
493 if (build_options != NULL)
494 {
495 all_build_options += " ";
496 all_build_options += build_options;
497 }
498 const DeviceInfo& devInfo = ctx->getDeviceInfo();
499 string filename = binpath + (source->name ? source->name : "NULL") + "_" + devInfo.platform->platformName + "_" + devInfo.deviceName + ".clb";
500
501 ProgramFileCache programFileCache(filename, source->programHash);
502 cl_program program = programFileCache.getOrBuildProgram(ctx, source, all_build_options);
503
504 //Cache the binary for future use if build_options is null
505 if (source->name)
506 {
507 cv::AutoLock localLockCache(mutexCache);
508 this->addProgram(src_sign.str(), program);
509 }
510 return program;
511}
