59 static inline int divUp(int total, int grain)
60 {
61 return (total + grain - 1) / grain;
62 }
63 static string get_kernel_name(string kernel_name, int data_type)
64 {
65 stringstream idxStr;
66 if(data_type == CV_16S)
67 idxStr << "0";
68 else
69 idxStr << "1";
70 kernel_name += idxStr.str();
71
72 return kernel_name;
73 }

78 static void init_data_cost_caller(const oclMat &left, const oclMat &right, oclMat &temp,
79 StereoConstantSpaceBP *pThis,
80 int msg_step, int h, int w, int level)
81 {
82 Context *clCxt = left.clCxt;
83 int data_type = pThis->msg_type;
84 int channels = left.oclchannels();
85
86 string kernelName = get_kernel_name("init_data_cost_", data_type);
87
88 cl_kernel kernel = openCLGetKernelFromSource(clCxt, &stereocsbp, kernelName);
89
90 //size_t blockSize = 256;
91 size_t localThreads[] = {32, 8 ,1};
92 size_t globalThreads[] = {divUp(w, localThreads[0]) *localThreads[0],
93 divUp(h, localThreads[1]) *localThreads[1],
94 1
95 };
96
97 int cdisp_step1 = msg_step * h;
98 openCLVerifyKernel(clCxt, kernel, localThreads);
99 openCLSafeCall(clSetKernelArg(kernel, 0, sizeof(cl_mem), (void *)&temp.data));
100 openCLSafeCall(clSetKernelArg(kernel, 1, sizeof(cl_mem), (void *)&left.data));
101 openCLSafeCall(clSetKernelArg(kernel, 2, sizeof(cl_mem), (void *)&right.data));
102 openCLSafeCall(clSetKernelArg(kernel, 3, sizeof(cl_int), (void *)&h));
103 openCLSafeCall(clSetKernelArg(kernel, 4, sizeof(cl_int), (void *)&w));
104 openCLSafeCall(clSetKernelArg(kernel, 5, sizeof(cl_int), (void *)&level));
105 openCLSafeCall(clSetKernelArg(kernel, 6, sizeof(cl_int), (void *)&channels));
106 openCLSafeCall(clSetKernelArg(kernel, 7, sizeof(cl_int), (void *)&msg_step));
107 openCLSafeCall(clSetKernelArg(kernel, 8, sizeof(cl_float), (void *)&pThis->data_weight));
108 openCLSafeCall(clSetKernelArg(kernel, 9, sizeof(cl_float), (void *)&pThis->max_data_term));
109 openCLSafeCall(clSetKernelArg(kernel, 10, sizeof(cl_int), (void *)&cdisp_step1));
110 openCLSafeCall(clSetKernelArg(kernel, 11, sizeof(cl_int), (void *)&pThis->min_disp_th));
111 openCLSafeCall(clSetKernelArg(kernel, 12, sizeof(cl_int), (void *)&left.step));
112 openCLSafeCall(clSetKernelArg(kernel, 13, sizeof(cl_int), (void *)&pThis->ndisp));
113 openCLSafeCall(clEnqueueNDRangeKernel(*(cl_command_queue*)getClCommandQueuePtr(), kernel, 2, NULL,
114 globalThreads, localThreads, 0, NULL, NULL));
115
116 clFinish(*(cl_command_queue*)getClCommandQueuePtr());
117 openCLSafeCall(clReleaseKernel(kernel));
118 }

120 static void init_data_cost_reduce_caller(const oclMat &left, const oclMat &right, oclMat &temp,
121 StereoConstantSpaceBP *pThis,
122 int msg_step, int h, int w, int level)
123 {
124
125 Context *clCxt = left.clCxt;
126 int data_type = pThis->msg_type;
127 int channels = left.oclchannels();
128 int win_size = (int)std::pow(2.f, level);
129
130 string kernelName = get_kernel_name("init_data_cost_reduce_", data_type);
131
132 cl_kernel kernel = openCLGetKernelFromSource(clCxt, &stereocsbp, kernelName);
133
134 const int threadsNum = 256;
135 //size_t blockSize = threadsNum;
136 size_t localThreads[3] = {(size_t)win_size, 1, (size_t)threadsNum / win_size};
137 size_t globalThreads[3] = { w *localThreads[0],
138 h * divUp(pThis->ndisp, localThreads[2]) *localThreads[1], 1 * localThreads[2]
139 };
140
141 int local_mem_size = threadsNum * sizeof(float);
142 int cdisp_step1 = msg_step * h;
143
144 openCLVerifyKernel(clCxt, kernel, localThreads);
145
146 openCLSafeCall(clSetKernelArg(kernel, 0, sizeof(cl_mem), (void *)&temp.data));
147 openCLSafeCall(clSetKernelArg(kernel, 1, sizeof(cl_mem), (void *)&left.data));
148 openCLSafeCall(clSetKernelArg(kernel, 2, sizeof(cl_mem), (void *)&right.data));
149 openCLSafeCall(clSetKernelArg(kernel, 3, local_mem_size, (void *)NULL));
150 openCLSafeCall(clSetKernelArg(kernel, 4, sizeof(cl_int), (void *)&level));
151 openCLSafeCall(clSetKernelArg(kernel, 5, sizeof(cl_int), (void *)&left.rows));
152 openCLSafeCall(clSetKernelArg(kernel, 6, sizeof(cl_int), (void *)&left.cols));
153 openCLSafeCall(clSetKernelArg(kernel, 7, sizeof(cl_int), (void *)&h));
154 openCLSafeCall(clSetKernelArg(kernel, 8, sizeof(cl_int), (void *)&win_size));
155 openCLSafeCall(clSetKernelArg(kernel, 9, sizeof(cl_int), (void *)&channels));
156 openCLSafeCall(clSetKernelArg(kernel, 10, sizeof(cl_int), (void *)&pThis->ndisp));
157 openCLSafeCall(clSetKernelArg(kernel, 11, sizeof(cl_int), (void *)&left.step));
158 openCLSafeCall(clSetKernelArg(kernel, 12, sizeof(cl_float), (void *)&pThis->data_weight));
159 openCLSafeCall(clSetKernelArg(kernel, 13, sizeof(cl_float), (void *)&pThis->max_data_term));
160 openCLSafeCall(clSetKernelArg(kernel, 14, sizeof(cl_int), (void *)&pThis->min_disp_th));
161 openCLSafeCall(clSetKernelArg(kernel, 15, sizeof(cl_int), (void *)&cdisp_step1));
162 openCLSafeCall(clSetKernelArg(kernel, 16, sizeof(cl_int), (void *)&msg_step));
163 openCLSafeCall(clEnqueueNDRangeKernel(*(cl_command_queue*)getClCommandQueuePtr(), kernel, 3, NULL,
164 globalThreads, localThreads, 0, NULL, NULL));
165 clFinish(*(cl_command_queue*)getClCommandQueuePtr());
166 openCLSafeCall(clReleaseKernel(kernel));
167 }

237 static void init_data_cost(const oclMat &left, const oclMat &right, oclMat &temp, StereoConstantSpaceBP *pThis,
238 uchar *disp_selected_pyr, uchar *data_cost_selected,
239 size_t msg_step, int h, int w, int level, int nr_plane)
240 {
241
242 if(level <= 1)
243 init_data_cost_caller(left, right, temp, pThis, msg_step, h, w, level);
244 else
245 init_data_cost_reduce_caller(left, right, temp, pThis, msg_step, h, w, level);
246
247 if(pThis->use_local_init_data_cost == true)
248 {
249 get_first_initial_local_caller(data_cost_selected, disp_selected_pyr, temp, pThis, h, w, nr_plane, msg_step);
250 }
251 else
252 {
253 get_first_initial_global_caller(data_cost_selected, disp_selected_pyr, temp, pThis, h, w,
254 nr_plane, msg_step);
255 }
256 }

421 static void calc_all_iterations_caller(uchar *u, uchar *d, uchar *l, uchar *r, uchar *data_cost_selected,
422 uchar *disp_selected_pyr, oclMat &temp, StereoConstantSpaceBP *pThis,
423 int msg_step, int h, int w, int nr_plane, int i)
424 {
425 Context *clCxt = temp.clCxt;
426 int data_type = pThis->msg_type;
427
428 string kernelName = get_kernel_name("compute_message_", data_type);
429
430 cl_kernel kernel = openCLGetKernelFromSource(clCxt, &stereocsbp, kernelName);
431 size_t localThreads[] = {32, 8, 1};
432 size_t globalThreads[] = {divUp(w, (localThreads[0]) << 1) *localThreads[0],
433 divUp(h, localThreads[1]) *localThreads[1],
434 1
435 };
436
437 int disp_step = msg_step * h;
438 openCLVerifyKernel(clCxt, kernel, localThreads);
439 openCLSafeCall(clSetKernelArg(kernel, 0, sizeof(cl_mem), (void *)&u));
440 openCLSafeCall(clSetKernelArg(kernel, 1, sizeof(cl_mem), (void *)&d));
441 openCLSafeCall(clSetKernelArg(kernel, 2, sizeof(cl_mem), (void *)&l));
442 openCLSafeCall(clSetKernelArg(kernel, 3, sizeof(cl_mem), (void *)&r));
443 openCLSafeCall(clSetKernelArg(kernel, 4, sizeof(cl_mem), (void *)&data_cost_selected));
444 openCLSafeCall(clSetKernelArg(kernel, 5, sizeof(cl_mem), (void *)&disp_selected_pyr));
445 openCLSafeCall(clSetKernelArg(kernel, 6, sizeof(cl_mem), (void *)&temp.data));
446 openCLSafeCall(clSetKernelArg(kernel, 7, sizeof(cl_int), (void *)&h));
447 openCLSafeCall(clSetKernelArg(kernel, 8, sizeof(cl_int), (void *)&w));
448 openCLSafeCall(clSetKernelArg(kernel, 9, sizeof(cl_int), (void *)&nr_plane));
449 openCLSafeCall(clSetKernelArg(kernel, 10, sizeof(cl_int), (void *)&i));
450 openCLSafeCall(clSetKernelArg(kernel, 11, sizeof(cl_float), (void *)&pThis->max_disc_term));
451 openCLSafeCall(clSetKernelArg(kernel, 12, sizeof(cl_int), (void *)&disp_step));
452 openCLSafeCall(clSetKernelArg(kernel, 13, sizeof(cl_int), (void *)&msg_step));
453 openCLSafeCall(clSetKernelArg(kernel, 14, sizeof(cl_float), (void *)&pThis->disc_single_jump));
454 openCLSafeCall(clEnqueueNDRangeKernel(*(cl_command_queue*)getClCommandQueuePtr(), kernel, 2, NULL,
455 globalThreads, localThreads, 0, NULL, NULL));
456
457 clFinish(*(cl_command_queue*)getClCommandQueuePtr());
458 openCLSafeCall(clReleaseKernel(kernel));
459 }
460 static void calc_all_iterations(uchar *u, uchar *d, uchar *l, uchar *r, uchar *data_cost_selected,
461 uchar *disp_selected_pyr, oclMat &temp, StereoConstantSpaceBP *pThis,
462 int msg_step, int h, int w, int nr_plane)
463 {
464 for(int t = 0; t < pThis->iters; t++)
465 calc_all_iterations_caller(u, d, l, r, data_cost_selected, disp_selected_pyr, temp, pThis,
466 msg_step, h, w, nr_plane, t & 1);
467 }

520void cv::ocl::StereoConstantSpaceBP::estimateRecommendedParams(int width, int height, int &ndisp, int &iters, int &levels, int &nr_plane)
521{
522 ndisp = (int) ((float) width / 3.14f);
523 if ((ndisp & 1) != 0)
524 ndisp++;
525
526 int mm = ::max(width, height);
527 iters = mm / 100 + ((mm > 1200) ? - 4 : 4);
528
529 levels = (int)::log(static_cast<double>(mm)) * 2 / 3;
530 if (levels == 0) levels++;
531
532 nr_plane = (int) ((float) ndisp / std::pow(2.0, levels + 1));
533}

558template<class T>
559static void csbp_operator(StereoConstantSpaceBP *pThis, oclMat u[2], oclMat d[2], oclMat l[2], oclMat r[2],
560 oclMat disp_selected_pyr[2], oclMat &data_cost, oclMat &data_cost_selected,
561 oclMat &temp, oclMat &out, const oclMat &left, const oclMat &right, oclMat &disp)
562{
563 CV_DbgAssert(0 < pThis->ndisp && 0 < pThis->iters && 0 < pThis->levels && 0 < pThis->nr_plane
564 && left.rows == right.rows && left.cols == right.cols && left.type() == right.type());
565
566 CV_Assert(pThis->levels <= 8 && (left.type() == CV_8UC1 || left.type() == CV_8UC3));
567
568 const Scalar zero = Scalar::all(0);
569
570 ////////////////////////////////////Init///////////////////////////////////////////////////
571 int rows = left.rows;
572 int cols = left.cols;
573
574 pThis->levels = min(pThis->levels, int(log((double)pThis->ndisp) / log(2.0)));
575 int levels = pThis->levels;
576
577 AutoBuffer<int> buf(levels * 4);
578
579 int *cols_pyr = buf;
580 int *rows_pyr = cols_pyr + levels;
581 int *nr_plane_pyr = rows_pyr + levels;
582 int *step_pyr = nr_plane_pyr + levels;
583
584 cols_pyr[0] = cols;
585 rows_pyr[0] = rows;
586 nr_plane_pyr[0] = pThis->nr_plane;
587
588 const int n = 64;
589 step_pyr[0] = alignSize(cols * sizeof(T), n) / sizeof(T);
590 for (int i = 1; i < levels; i++)
591 {
592 cols_pyr[i] = cols_pyr[i - 1] / 2;
593 rows_pyr[i] = rows_pyr[i - 1]/ 2;
594
595 nr_plane_pyr[i] = nr_plane_pyr[i - 1] * 2;
596
597 step_pyr[i] = alignSize(cols_pyr[i] * sizeof(T), n) / sizeof(T);
598 }
599
600 Size msg_size(step_pyr[0], rows * nr_plane_pyr[0]);
601 Size data_cost_size(step_pyr[0], rows * nr_plane_pyr[0] * 2);
602
603 u[0].create(msg_size, DataType<T>::type);
604 d[0].create(msg_size, DataType<T>::type);
605 l[0].create(msg_size, DataType<T>::type);
606 r[0].create(msg_size, DataType<T>::type);
607
608 u[1].create(msg_size, DataType<T>::type);
609 d[1].create(msg_size, DataType<T>::type);
610 l[1].create(msg_size, DataType<T>::type);
611 r[1].create(msg_size, DataType<T>::type);
612
613 disp_selected_pyr[0].create(msg_size, DataType<T>::type);
614 disp_selected_pyr[1].create(msg_size, DataType<T>::type);
615
616 data_cost.create(data_cost_size, DataType<T>::type);
617 data_cost_selected.create(msg_size, DataType<T>::type);
618
619 Size temp_size = data_cost_size;
620 if (data_cost_size.width * data_cost_size.height < step_pyr[0] * rows_pyr[levels - 1] * pThis->ndisp)
621 temp_size = Size(step_pyr[0], rows_pyr[levels - 1] * pThis->ndisp);
622
623 temp.create(temp_size, DataType<T>::type);
624 temp = zero;
625
626 ///////////////////////////////// Compute////////////////////////////////////////////////
627
628 //csbp::load_constants(pThis->ndisp, pThis->max_data_term, pThis->data_weight,
629 // pThis->max_disc_term, pThis->disc_single_jump, pThis->min_disp_th, left, right, temp);
630
631 l[0] = zero;
632 d[0] = zero;
633 r[0] = zero;
634 u[0] = zero;
635 disp_selected_pyr[0] = zero;
636
637 l[1] = zero;
638 d[1] = zero;
639 r[1] = zero;
640 u[1] = zero;
641 disp_selected_pyr[1] = zero;
642
643 data_cost = zero;
644
645 data_cost_selected = zero;
646
647 int cur_idx = 0;
648
649 for (int i = levels - 1; i >= 0; i--)
650 {
651 if (i == levels - 1)
652 {
653 cv::ocl::stereoCSBP::init_data_cost(left, right, temp, pThis, disp_selected_pyr[cur_idx].data,
654 data_cost_selected.data, step_pyr[0], rows_pyr[i], cols_pyr[i],
655 i, nr_plane_pyr[i]);
656 }
657 else
658 {
659 cv::ocl::stereoCSBP::compute_data_cost(
660 disp_selected_pyr[cur_idx].data, data_cost.data, pThis, step_pyr[0],
661 step_pyr[0], left, right, rows_pyr[i], cols_pyr[i], rows_pyr[i + 1], i,
662 nr_plane_pyr[i + 1]);
663
664 int new_idx = (cur_idx + 1) & 1;
665
666 cv::ocl::stereoCSBP::init_message(u[new_idx].data, d[new_idx].data, l[new_idx].data, r[new_idx].data,
667 u[cur_idx].data, d[cur_idx].data, l[cur_idx].data, r[cur_idx].data,
668 disp_selected_pyr[new_idx].data, disp_selected_pyr[cur_idx].data,
669 data_cost_selected.data, data_cost.data, temp, pThis, step_pyr[0],
670 step_pyr[0], rows_pyr[i], cols_pyr[i], nr_plane_pyr[i], rows_pyr[i + 1],
671 cols_pyr[i + 1], nr_plane_pyr[i + 1]);
672 cur_idx = new_idx;
673 }
674 cv::ocl::stereoCSBP::calc_all_iterations(u[cur_idx].data, d[cur_idx].data, l[cur_idx].data, r[cur_idx].data,
675 data_cost_selected.data, disp_selected_pyr[cur_idx].data, temp,
676 pThis, step_pyr[0], rows_pyr[i], cols_pyr[i], nr_plane_pyr[i]);
677 }
678
679 if (disp.empty())
680 disp.create(rows, cols, CV_16S);
681
682 out = ((disp.type() == CV_16S) ? disp : (out.create(rows, cols, CV_16S), out));
683 out = zero;
684
685 stereoCSBP::compute_disp(u[cur_idx].data, d[cur_idx].data, l[cur_idx].data, r[cur_idx].data,
686 data_cost_selected.data, disp_selected_pyr[cur_idx].data, pThis, step_pyr[0],
687 out, nr_plane_pyr[0]);
688 if (disp.type() != CV_16S)
689 out.convertTo(disp, disp.type());
690}
