79void cv::ocl::interpolateFrames(const oclMat &frame0, const oclMat &frame1,
80 const oclMat &fu, const oclMat &fv,
81 const oclMat &bu, const oclMat &bv,
82 float pos, oclMat &newFrame, oclMat &buf)
83{
84 CV_Assert(frame0.type() == CV_32FC1);
85 CV_Assert(frame1.size() == frame0.size() && frame1.type() == frame0.type());
86 CV_Assert(fu.size() == frame0.size() && fu.type() == frame0.type());
87 CV_Assert(fv.size() == frame0.size() && fv.type() == frame0.type());
88 CV_Assert(bu.size() == frame0.size() && bu.type() == frame0.type());
89 CV_Assert(bv.size() == frame0.size() && bv.type() == frame0.type());
90
91 newFrame.create(frame0.size(), frame0.type());
92
93 buf.create(6 * frame0.rows, frame0.cols, CV_32FC1);
94 buf.setTo(Scalar::all(0));
95
96 size_t step = frame0.step;
97
98 CV_Assert(frame1.step == step && fu.step == step && fv.step == step && bu.step == step && bv.step == step && newFrame.step == step && buf.step == step);
99 cl_mem tex_src0 = 0, tex_src1 = 0;
100
101 // warp flow
102 using namespace interpolate;
103
104 bindImgTex(frame0, tex_src0);
105 bindImgTex(frame1, tex_src1);
106
107 // CUDA Offsets
108 enum
109 {
110 cov0 = 0,
111 cov1,
112 fwdU,
113 fwdV,
114 bwdU,
115 bwdV
116 };
117
118 vectorWarp(fu, fu, fv, buf, cov0, pos, fwdU);
119 vectorWarp(fv, fu, fv, buf, cov0, pos, fwdV);
120 vectorWarp(bu, bu, bv, buf, cov1, 1.0f - pos, bwdU);
121 vectorWarp(bv, bu, bv, buf, cov1, 1.0f - pos, bwdU);
122
123 blendFrames(frame0, frame1, buf, pos, newFrame, tex_src0, tex_src1);
124
125 openCLFree(tex_src0);
126 openCLFree(tex_src1);
127}

129void interpolate::memsetKernel(float val, oclMat &img, int height, int offset)
130{
131 Context *clCxt = Context::getContext();
132 string kernelName = "memsetKernel";
133 vector< pair<size_t, const void *> > args;
134 int step = img.step / sizeof(float);
135 offset = step * height * offset;
136
137 args.push_back( make_pair( sizeof(cl_float), (void *)&val));
138 args.push_back( make_pair( sizeof(cl_mem), (void *)&img.data));
139 args.push_back( make_pair( sizeof(cl_int), (void *)&img.cols));
140 args.push_back( make_pair( sizeof(cl_int), (void *)&height));
141 args.push_back( make_pair( sizeof(cl_int), (void *)&step));
142 args.push_back( make_pair( sizeof(cl_int), (void *)&offset));
143
144 size_t globalThreads[3] = {(size_t)img.cols, (size_t)height, 1};
145 size_t localThreads[3] = {16, 16, 1};
146 openCLExecuteKernel(clCxt, &interpolate_frames, kernelName, globalThreads, localThreads, args, -1, -1);
147}
148void interpolate::normalizeKernel(oclMat &buffer, int height, int factor_offset, int dst_offset)
149{
150 Context *clCxt = Context::getContext();
151 string kernelName = "normalizeKernel";
152 vector< pair<size_t, const void *> > args;
153 int step = buffer.step / sizeof(float);
154 factor_offset = step * height * factor_offset;
155 dst_offset = step * height * dst_offset;
156
157 args.push_back( make_pair( sizeof(cl_mem), (void *)&buffer.data));
158 args.push_back( make_pair( sizeof(cl_int), (void *)&buffer.cols));
159 args.push_back( make_pair( sizeof(cl_int), (void *)&height));
160 args.push_back( make_pair( sizeof(cl_int), (void *)&step));
161 args.push_back( make_pair( sizeof(cl_int), (void *)&factor_offset));
162 args.push_back( make_pair( sizeof(cl_int), (void *)&dst_offset));
163
164 size_t globalThreads[3] = {(size_t)buffer.cols, (size_t)height, 1};
165 size_t localThreads[3] = {16, 16, 1};
166 openCLExecuteKernel(clCxt, &interpolate_frames, kernelName, globalThreads, localThreads, args, -1, -1);
167}

169void interpolate::forwardWarpKernel(const oclMat &src, oclMat &buffer, const oclMat &u, const oclMat &v, const float time_scale,
170 int b_offset, int d_offset)
171{
172 Context *clCxt = Context::getContext();
173 string kernelName = "forwardWarpKernel";
174 vector< pair<size_t, const void *> > args;
175 int f_step = u.step / sizeof(float); // flow step
176 int b_step = buffer.step / sizeof(float);
177
178 b_offset = b_step * src.rows * b_offset;
179 d_offset = b_step * src.rows * d_offset;
180
181 args.push_back( make_pair( sizeof(cl_mem), (void *)&src.data));
182 args.push_back( make_pair( sizeof(cl_mem), (void *)&buffer.data));
183 args.push_back( make_pair( sizeof(cl_mem), (void *)&u.data));
184 args.push_back( make_pair( sizeof(cl_mem), (void *)&v.data));
185 args.push_back( make_pair( sizeof(cl_int), (void *)&src.cols));
186 args.push_back( make_pair( sizeof(cl_int), (void *)&src.rows));
187 args.push_back( make_pair( sizeof(cl_int), (void *)&f_step));
188 args.push_back( make_pair( sizeof(cl_int), (void *)&b_step));
189 args.push_back( make_pair( sizeof(cl_int), (void *)&b_offset));
190 args.push_back( make_pair( sizeof(cl_int), (void *)&d_offset));
191 args.push_back( make_pair( sizeof(cl_float), (void *)&time_scale));
192
193 size_t globalThreads[3] = {(size_t)src.cols, (size_t)src.rows, 1};
194 size_t localThreads[3] = {16, 16, 1};
195 openCLExecuteKernel(clCxt, &interpolate_frames, kernelName, globalThreads, localThreads, args, -1, -1);
196}

198void interpolate::vectorWarp(const oclMat &src, const oclMat &u, const oclMat &v,
199 oclMat &buffer, int b_offset, float timeScale, int d_offset)
200{
201 memsetKernel(0, buffer, src.rows, b_offset);
202 forwardWarpKernel(src, buffer, u, v, timeScale, b_offset, d_offset);
203 normalizeKernel(buffer, src.rows, b_offset, d_offset);
204}

206void interpolate::blendFrames(const oclMat &frame0, const oclMat &/*frame1*/, const oclMat &buffer, float pos, oclMat &newFrame, cl_mem &tex_src0, cl_mem &tex_src1)
207{
208 int step = buffer.step / sizeof(float);
209
210 Context *clCxt = Context::getContext();
211 string kernelName = "blendFramesKernel";
212 vector< pair<size_t, const void *> > args;
213
214 args.push_back( make_pair( sizeof(cl_mem), (void *)&tex_src0));
215 args.push_back( make_pair( sizeof(cl_mem), (void *)&tex_src1));
216 args.push_back( make_pair( sizeof(cl_mem), (void *)&buffer.data));
217 args.push_back( make_pair( sizeof(cl_mem), (void *)&newFrame.data));
218 args.push_back( make_pair( sizeof(cl_int), (void *)&frame0.cols));
219 args.push_back( make_pair( sizeof(cl_int), (void *)&frame0.rows));
220 args.push_back( make_pair( sizeof(cl_int), (void *)&step));
221 args.push_back( make_pair( sizeof(cl_float), (void *)&pos));
222
223 size_t globalThreads[3] = {(size_t)frame0.cols, (size_t)frame0.rows, 1};
224 size_t localThreads[3] = {16, 16, 1};
225 openCLExecuteKernel(clCxt, &interpolate_frames, kernelName, globalThreads, localThreads, args, -1, -1);
226}
