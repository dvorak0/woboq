68// convert_C3C4
69
70static void convert_C3C4(const cl_mem &src, oclMat &dst)
71{
72 Context *clCxt = dst.clCxt;
73 int pixel_end = dst.wholecols * dst.wholerows - 1;
74 int dstStep_in_pixel = dst.step1() / dst.oclchannels();
75
76 const char * const typeMap[] = { "uchar", "char", "ushort", "short", "int", "float", "double" };
77 std::string buildOptions = format("-D GENTYPE4=%s4", typeMap[dst.depth()]);
78
79 vector< pair<size_t, const void *> > args;
80 args.push_back( make_pair( sizeof(cl_mem), (void *)&src));
81 args.push_back( make_pair( sizeof(cl_mem), (void *)&dst.data));
82 args.push_back( make_pair( sizeof(cl_int), (void *)&dst.wholecols));
83 args.push_back( make_pair( sizeof(cl_int), (void *)&dst.wholerows));
84 args.push_back( make_pair( sizeof(cl_int), (void *)&dstStep_in_pixel));
85 args.push_back( make_pair( sizeof(cl_int), (void *)&pixel_end));
86
87 size_t globalThreads[3] = { divUp(dst.wholecols * dst.wholerows, 4), 1, 1 };
88
89#ifdef ANDROID
90 openCLExecuteKernel(clCxt, &convertC3C4, "convertC3C4", globalThreads, NULL,
91 args, -1, -1, buildOptions.c_str());
92#else
93 size_t localThreads[3] = { 256, 1, 1 };
94 openCLExecuteKernel(clCxt, &convertC3C4, "convertC3C4", globalThreads, localThreads,
95 args, -1, -1, buildOptions.c_str());
96#endif
97}

129void cv::ocl::oclMat::upload(const Mat &m)
130{
131 if (!Context::getContext()->supportsFeature(FEATURE_CL_DOUBLE) && m.depth() == CV_64F)
132 {
133 CV_Error(CV_OpenCLDoubleNotSupported, "Selected device doesn't support double");
134 return;
135 }
136
137 CV_DbgAssert(!m.empty());
138 Size wholeSize;
139 Point ofs;
140 m.locateROI(wholeSize, ofs);
141 create(wholeSize, m.type());
142
143 if (m.channels() == 3)
144 {
145 int pitch = wholeSize.width * 3 * m.elemSize1();
146 int tail_padding = m.elemSize1() * 3072;
147 int err;
148 cl_mem temp = clCreateBuffer(*(cl_context*)clCxt->getOpenCLContextPtr(), CL_MEM_READ_WRITE,
149 (pitch * wholeSize.height + tail_padding - 1) / tail_padding * tail_padding, 0, &err);
150 openCLVerifyCall(err);
151
152 openCLMemcpy2D(clCxt, temp, pitch, m.datastart, m.step, wholeSize.width * m.elemSize(), wholeSize.height, clMemcpyHostToDevice, 3);
153 convert_C3C4(temp, *this);
154 openCLSafeCall(clReleaseMemObject(temp));
155 }
156 else
157 openCLMemcpy2D(clCxt, data, step, m.datastart, m.step, wholeSize.width * elemSize(), wholeSize.height, clMemcpyHostToDevice);
158
159 rows = m.rows;
160 cols = m.cols;
161 offset = ofs.y * step + ofs.x * elemSize();
162}

192void cv::ocl::oclMat::download(cv::Mat &m) const
193{
194 CV_DbgAssert(!this->empty());
195 m.create(wholerows, wholecols, type());
196
197 if(m.channels() == 3)
198 {
199 int pitch = wholecols * 3 * m.elemSize1();
200 int tail_padding = m.elemSize1() * 3072;
201 int err;
202 cl_mem temp = clCreateBuffer(*(cl_context*)clCxt->getOpenCLContextPtr(), CL_MEM_READ_WRITE,
203 (pitch * wholerows + tail_padding - 1) / tail_padding * tail_padding, 0, &err);
204 openCLVerifyCall(err);
205
206 convert_C4C3(*this, temp);
207 openCLMemcpy2D(clCxt, m.data, m.step, temp, pitch, wholecols * m.elemSize(), wholerows, clMemcpyDeviceToHost, 3);
208 openCLSafeCall(clReleaseMemObject(temp));
209 }
210 else
211 {
212 openCLMemcpy2D(clCxt, m.data, m.step, data, step, wholecols * elemSize(), wholerows, clMemcpyDeviceToHost);
213 }
214
215 Size wholesize;
216 Point ofs;
217 locateROI(wholesize, ofs);
218 m.adjustROI(-ofs.y, ofs.y + rows - wholerows, -ofs.x, ofs.x + cols - wholecols);
219}

224static void copy_to_with_mask(const oclMat &src, oclMat &dst, const oclMat &mask, string kernelName)
225{
226 CV_DbgAssert( dst.rows == mask.rows && dst.cols == mask.cols &&
227 src.rows == dst.rows && src.cols == dst.cols
228 && mask.type() == CV_8UC1);
229
230 vector<pair<size_t , const void *> > args;
231
232 std::string string_types[4][7] = {{"uchar", "char", "ushort", "short", "int", "float", "double"},
233 {"uchar2", "char2", "ushort2", "short2", "int2", "float2", "double2"},
234 {"uchar3", "char3", "ushort3", "short3", "int3", "float3", "double3"},
235 {"uchar4", "char4", "ushort4", "short4", "int4", "float4", "double4"}
236 };
237
238 char compile_option[32];
239 sprintf(compile_option, "-D GENTYPE=%s", string_types[dst.oclchannels() - 1][dst.depth()].c_str());
240 size_t localThreads[3] = {16, 16, 1};
241 size_t globalThreads[3] = { (size_t)dst.cols, (size_t)dst.rows, 1 };
242
243 int dststep_in_pixel = dst.step / dst.elemSize(), dstoffset_in_pixel = dst.offset / dst.elemSize();
244 int srcstep_in_pixel = src.step / src.elemSize(), srcoffset_in_pixel = src.offset / src.elemSize();
245
246 args.push_back( make_pair( sizeof(cl_mem) , (void *)&src.data ));
247 args.push_back( make_pair( sizeof(cl_mem) , (void *)&dst.data ));
248 args.push_back( make_pair( sizeof(cl_mem) , (void *)&mask.data ));
249 args.push_back( make_pair( sizeof(cl_int) , (void *)&src.cols ));
250 args.push_back( make_pair( sizeof(cl_int) , (void *)&src.rows ));
251 args.push_back( make_pair( sizeof(cl_int) , (void *)&srcstep_in_pixel ));
252 args.push_back( make_pair( sizeof(cl_int) , (void *)&srcoffset_in_pixel ));
253 args.push_back( make_pair( sizeof(cl_int) , (void *)&dststep_in_pixel ));
254 args.push_back( make_pair( sizeof(cl_int) , (void *)&dstoffset_in_pixel ));
255 args.push_back( make_pair( sizeof(cl_int) , (void *)&mask.step ));
256 args.push_back( make_pair( sizeof(cl_int) , (void *)&mask.offset ));
257
258 openCLExecuteKernel(dst.clCxt , &operator_copyToM, kernelName, globalThreads,
259 localThreads, args, -1, -1, compile_option);
260}

262void cv::ocl::oclMat::copyTo( oclMat &mat, const oclMat &mask) const
263{
264 if (mask.empty())
265 {
266 CV_DbgAssert(!this->empty());
267 mat.create(size(), type());
268 openCLCopyBuffer2D(clCxt, mat.data, mat.step, mat.offset,
269 data, step, cols * elemSize(), rows, offset);
270 }
271 else
272 {
273 mat.create(size(), type());
274 copy_to_with_mask(*this, mat, mask, "copy_to_with_mask");
275 }
276}

282static void convert_run(const oclMat &src, oclMat &dst, double alpha, double beta)
283{
284 string kernelName = "convert_to";
285 float alpha_f = alpha, beta_f = beta;
286 int sdepth = src.depth(), ddepth = dst.depth();
287 int sstep1 = (int)src.step1(), dstep1 = (int)dst.step1();
288 int cols1 = src.cols * src.oclchannels();
289
290 char buildOptions[150], convertString[50];
291 const char * typeMap[] = { "uchar", "char", "ushort", "short", "int", "float", "double" };
292 sprintf(convertString, "convert_%s_sat_rte", typeMap[ddepth]);
293 sprintf(buildOptions, "-D srcT=%s -D dstT=%s -D convertToDstType=%s", typeMap[sdepth],
294 typeMap[ddepth], CV_32F == ddepth || ddepth == CV_64F ? "" : convertString);
295
296 CV_DbgAssert(src.rows == dst.rows && src.cols == dst.cols);
297 vector<pair<size_t , const void *> > args;
298
299 size_t localThreads[3] = { 16, 16, 1 };
300 size_t globalThreads[3] = { divUp(cols1, localThreads[0]) * localThreads[0],
301 divUp(dst.rows, localThreads[1]) * localThreads[1], 1 };
302
303 int doffset1 = dst.offset / dst.elemSize1();
304 int soffset1 = src.offset / src.elemSize1();
305
306 args.push_back( make_pair( sizeof(cl_mem) , (void *)&src.data ));
307 args.push_back( make_pair( sizeof(cl_mem) , (void *)&dst.data ));
308 args.push_back( make_pair( sizeof(cl_int) , (void *)&cols1 ));
309 args.push_back( make_pair( sizeof(cl_int) , (void *)&src.rows ));
310 args.push_back( make_pair( sizeof(cl_int) , (void *)&sstep1 ));
311 args.push_back( make_pair( sizeof(cl_int) , (void *)&soffset1 ));
312 args.push_back( make_pair( sizeof(cl_int) , (void *)&dstep1 ));
313 args.push_back( make_pair( sizeof(cl_int) , (void *)&doffset1 ));
314 args.push_back( make_pair( sizeof(cl_float) , (void *)&alpha_f ));
315 args.push_back( make_pair( sizeof(cl_float) , (void *)&beta_f ));
316
317 openCLExecuteKernel(dst.clCxt , &operator_convertTo, kernelName, globalThreads,
318 localThreads, args, -1, -1, buildOptions);
319}

321void cv::ocl::oclMat::convertTo( oclMat &dst, int rtype, double alpha, double beta ) const
322{
323 if (!clCxt->supportsFeature(FEATURE_CL_DOUBLE) &&
324 (depth() == CV_64F || dst.depth() == CV_64F))
325 {
326 CV_Error(CV_OpenCLDoubleNotSupported, "Selected device doesn't support double");
327 return;
328 }
329
330 bool noScale = fabs(alpha - 1) < std::numeric_limits<double>::epsilon()
331 && fabs(beta) < std::numeric_limits<double>::epsilon();
332
333 if( rtype < 0 )
334 rtype = type();
335 else
336 rtype = CV_MAKETYPE(CV_MAT_DEPTH(rtype), channels());
337
338 int sdepth = depth(), ddepth = CV_MAT_DEPTH(rtype);
339 if( sdepth == ddepth && noScale )
340 {
341 copyTo(dst);
342 return;
343 }
344
345 oclMat temp;
346 const oclMat *psrc = this;
347 if( sdepth != ddepth && psrc == &dst )
348 psrc = &(temp = *this);
349
350 dst.create( size(), rtype );
351 convert_run(*psrc, dst, alpha, beta);
352}

381 buf->s[1] = saturate_cast<PT>(s[1]);

390 buf->s[0] = saturate_cast<PT>(s[0]);
391 buf->s[1] = saturate_cast<PT>(s[1]);
392 buf->s[2] = saturate_cast<PT>(s[2]);
393 buf->s[3] = saturate_cast<PT>(s[3]);

399static std::vector<uchar> scalarToCLVector(const cv::Scalar & s, int type)
400{
401 const int depth = CV_MAT_DEPTH(type);
402 const int channels = CV_MAT_CN(type);
403
404 static const ConvertFunc funcs[4][7] =
405 {
406 { cvt1<cl_uchar, uchar>, cvt1<cl_char, char>, cvt1<cl_ushort, ushort>, cvt1<cl_short, short>,
407 cvt1<cl_int, int>, cvt1<cl_float, float>, cvt1<cl_double, double> },
408
409 { cvt2<cl_uchar2, uchar>, cvt2<cl_char2, char>, cvt2<cl_ushort2, ushort>, cvt2<cl_short2, short>,
410 cvt2<cl_int2, int>, cvt2<cl_float2, float>, cvt2<cl_double2, double> },
411
412 { 0, 0, 0, 0, 0, 0, 0 },
413
414 { cvt4<cl_uchar4, uchar>, cvt4<cl_char4, char>, cvt4<cl_ushort4, ushort>, cvt4<cl_short4, short>,
415 cvt4<cl_int4, int>, cvt4<cl_float4, float>, cvt4<cl_double4, double> }
416 };
417
418 ConvertFunc func = funcs[channels - 1][depth];
419 return func(s);
420}

424static void set_to_withoutmask_run(const oclMat &dst, const Scalar &scalar, string kernelName)
425{
426 vector<pair<size_t , const void *> > args;
427
428 size_t localThreads[3] = {16, 16, 1};
429 size_t globalThreads[3] = { (size_t)dst.cols, (size_t)dst.rows, 1 };
430 int step_in_pixel = dst.step / dst.elemSize(), offset_in_pixel = dst.offset / dst.elemSize();
431
432 if (dst.type() == CV_8UC1)
433 globalThreads[0] = ((dst.cols + 4) / 4 + localThreads[0] - 1) / localThreads[0] * localThreads[0];
434
435 const char * const typeMap[] = { "uchar", "char", "ushort", "short", "int", "float", "double" };
436 const char channelMap[] = { ' ', ' ', '2', '4', '4' };
437 std::string buildOptions = format("-D GENTYPE=%s%c", typeMap[dst.depth()], channelMap[dst.channels()]);
438
439 Mat mat(1, 1, dst.type(), scalar);
440
441#ifdef CL_VERSION_1_2
442 // this enables backwards portability to
443 // run on OpenCL 1.1 platform if library binaries are compiled with OpenCL 1.2 support
444 if (Context::getContext()->supportsFeature(FEATURE_CL_VER_1_2) && dst.isContinuous())
445 {
446 std::vector<uchar> p = ::scalarToCLVector(scalar, CV_MAKE_TYPE(dst.depth(), dst.oclchannels()));
447 clEnqueueFillBuffer(getClCommandQueue(dst.clCxt),
448 (cl_mem)dst.data, (void*)&p[0], p.size(),
449 0, dst.step * dst.rows, 0, NULL, NULL);
450 }
451 else
452#endif
453 {
454 oclMat m(mat);
455 args.push_back( make_pair( sizeof(cl_mem) , (void*)&m.data ));
456 args.push_back( make_pair( sizeof(cl_mem) , (void *)&dst.data ));
457 args.push_back( make_pair( sizeof(cl_int) , (void *)&dst.cols ));
458 args.push_back( make_pair( sizeof(cl_int) , (void *)&dst.rows ));
459 args.push_back( make_pair( sizeof(cl_int) , (void *)&step_in_pixel ));
460 args.push_back( make_pair( sizeof(cl_int) , (void *)&offset_in_pixel ));
461
462 openCLExecuteKernel(dst.clCxt , &operator_setTo, kernelName, globalThreads,
463 localThreads, args, -1, -1, buildOptions.c_str());
464 }
465}

468{
469 CV_DbgAssert( dst.rows == mask.rows && dst.cols == mask.cols);

486 args.push_back( make_pair( sizeof(cl_mem) , (void *)&mask.data ));
487 args.push_back( make_pair( sizeof(cl_int) , (void *)&mask.step ));

493oclMat &cv::ocl::oclMat::setTo(const Scalar &scalar, const oclMat &mask)
494{
495 CV_Assert(mask.type() == CV_8UC1);
496 CV_Assert( this->depth() >= 0 && this->depth() <= 6 );
497 CV_DbgAssert( !this->empty());
498 if (mask.empty())
499 {
500 set_to_withoutmask_run(*this, scalar, type() == CV_8UC1 ?
501 "set_to_without_mask_C1_D0" : "set_to_without_mask");
502 }
503 else
504 set_to_withmask_run(*this, scalar, mask, "set_to_with_mask");
505
506 return *this;
507}

509oclMat cv::ocl::oclMat::reshape(int new_cn, int new_rows) const
510{
511 if( new_rows != 0 && new_rows != rows)
512 CV_Error( CV_StsBadFunc, "oclMat's number of rows can not be changed for current version" );
513
514 oclMat hdr = *this;
515
516 int cn = oclchannels();
517 if (new_cn == 0)
518 new_cn = cn;
519
520 int total_width = cols * cn;
521 if ((new_cn > total_width || total_width % new_cn != 0) && new_rows == 0)
522 new_rows = rows * total_width / new_cn;
523
524 if (new_rows != 0 && new_rows != rows)
525 {
526 int total_size = total_width * rows;
527
528 if (!isContinuous())
529 CV_Error(CV_BadStep, "The matrix is not continuous, thus its number of rows can not be changed");
530
531 if ((unsigned)new_rows > (unsigned)total_size)
532 CV_Error(CV_StsOutOfRange, "Bad new number of rows");
533
534 total_width = total_size / new_rows;
535 if (total_width * new_rows != total_size)
536 CV_Error(CV_StsBadArg, "The total number of matrix elements is not divisible by the new number of rows");
537
538 hdr.rows = new_rows;
539 hdr.step = total_width * elemSize1();
540 }
541
542 int new_width = total_width / new_cn;
543 if (new_width * new_cn != total_width)
544 CV_Error(CV_BadNumChannels, "The total width is not divisible by the new number of channels");
545
546 hdr.cols = new_width;
547 hdr.wholecols = new_width;
548 hdr.flags = (hdr.flags & ~CV_MAT_CN_MASK) | ((new_cn - 1) << CV_CN_SHIFT);
549 return hdr;
550
551}

563void cv::ocl::oclMat::createEx(int _rows, int _cols, int _type, DevMemRW rw_type, DevMemType mem_type)
564{
565 clCxt = Context::getContext();
566 /* core logic */
567 _type &= TYPE_MASK;
568 if( rows == _rows && cols == _cols && type() == _type && data )
569 return;
570 if( data )
571 release();
572 CV_DbgAssert( _rows >= 0 && _cols >= 0 );
573 if( _rows > 0 && _cols > 0 )
574 {
575 flags = Mat::MAGIC_VAL + _type;
576 rows = _rows;
577 cols = _cols;
578 wholerows = _rows;
579 wholecols = _cols;
580 size_t esz = elemSize();
581
582 void *dev_ptr;
583 openCLMallocPitchEx(clCxt, &dev_ptr, &step, GPU_MATRIX_MALLOC_STEP(esz * cols), rows, rw_type, mem_type);
584
585 if (esz * cols == step)
586 flags |= Mat::CONTINUOUS_FLAG;
587
588 int64 _nettosize = (int64)step * rows;
589 size_t nettosize = (size_t)_nettosize;
590
591 datastart = data = (uchar *)dev_ptr;
592 dataend = data + nettosize;
593
594 refcount = (int *)fastMalloc(sizeof(*refcount));
595 *refcount = 1;
596 }
597}

599void cv::ocl::oclMat::release()
600{
601 if( refcount && CV_XADD(refcount, -1) == 1 )
602 {
603 fastFree(refcount);
604 openCLFree(datastart);
605 }
606 data = datastart = dataend = 0;
607 step = rows = cols = 0;
608 offset = wholerows = wholecols = 0;
609 refcount = 0;
610}
