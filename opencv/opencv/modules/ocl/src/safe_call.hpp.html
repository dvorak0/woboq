46#ifndef __OPENCV_OPENCL_SAFE_CALL_HPP__
47#define __OPENCV_OPENCL_SAFE_CALL_HPP__
48
49#include "opencv2/ocl/cl_runtime/cl_runtime.hpp"
50
51#define openCLSafeCall(expr) ___openCLSafeCall(expr, __FILE__, __LINE__, CV_Func)
52#define openCLVerifyCall(res) ___openCLSafeCall(res, __FILE__, __LINE__, CV_Func)
53
54
55namespace cv
56{
57 namespace ocl
58 {
59 void error( const char *error_string, const char *file, const int line, const char *func = "");
60 const char *getOpenCLErrorString( int err );
61
62 static inline void ___openCLSafeCall(int err, const char *file, const int line, const char *func = "")
63 {
64 if (CL_SUCCESS != err)
65 cv::ocl::error(getOpenCLErrorString(err), file, line, func);
66 }
67 }
68}
69
70#endif /* __OPENCV_OPENCL_SAFE_CALL_HPP__ */
