56 // The function calculates center of gravity and the central second order moments
57 static void icvCompleteMomentState( CvMoments* lmoments )
58 {
59 double cx = 0, cy = 0;
60 double mu20, mu11, mu02;
61
62 assert( lmoments != 0 );
63 lmoments->inv_sqrt_m00 = 0;
64
65 if( fabs(lmoments->m00) > DBL_EPSILON )
66 {
67 double inv_m00 = 1. / lmoments->m00;
68 cx = lmoments->m10 * inv_m00;
69 cy = lmoments->m01 * inv_m00;
70 lmoments->inv_sqrt_m00 = std::sqrt( fabs(inv_m00) );
71 }
72
73 // mu20 = m20 - m10*cx
74 mu20 = lmoments->m20 - lmoments->m10 * cx;
75 // mu11 = m11 - m10*cy
76 mu11 = lmoments->m11 - lmoments->m10 * cy;
77 // mu02 = m02 - m01*cy
78 mu02 = lmoments->m02 - lmoments->m01 * cy;
79
80 lmoments->mu20 = mu20;
81 lmoments->mu11 = mu11;
82 lmoments->mu02 = mu02;
83
84 // mu30 = m30 - cx*(3*mu20 + cx*m10)
85 lmoments->mu30 = lmoments->m30 - cx * (3 * mu20 + cx * lmoments->m10);
86 mu11 += mu11;
87 // mu21 = m21 - cx*(2*mu11 + cx*m01) - cy*mu20
88 lmoments->mu21 = lmoments->m21 - cx * (mu11 + cx * lmoments->m01) - cy * mu20;
89 // mu12 = m12 - cy*(2*mu11 + cy*m10) - cx*mu02
90 lmoments->mu12 = lmoments->m12 - cy * (mu11 + cy * lmoments->m10) - cx * mu02;
91 // mu03 = m03 - cy*(3*mu02 + cy*m01)
92 lmoments->mu03 = lmoments->m03 - cy * (3 * mu02 + cy * lmoments->m01);
93 }

96 static void icvContourMoments( CvSeq* contour, CvMoments* mom )
97 {
98 if( contour->total )
99 {
100 CvSeqReader reader;
101 int lpt = contour->total;
102 double a00, a10, a01, a20, a11, a02, a30, a21, a12, a03;
103
104 cvStartReadSeq( contour, &reader, 0 );
105
106 size_t reader_size = lpt << 1;
107 cv::Mat reader_mat(1,reader_size,CV_32FC1);
108
109 bool is_float = CV_SEQ_ELTYPE(contour) == CV_32FC2;
110
111 if (!cv::ocl::Context::getContext()->supportsFeature(FEATURE_CL_DOUBLE) && is_float)
112 {
113 CV_Error(CV_StsUnsupportedFormat, "Moments - double is not supported by your GPU!");
114 }
115
116 if( is_float )
117 {
118 for(size_t i = 0; i < reader_size; ++i)
119 {
120 reader_mat.at<float>(0, i++) = ((CvPoint2D32f*)(reader.ptr))->x;
121 reader_mat.at<float>(0, i) = ((CvPoint2D32f*)(reader.ptr))->y;
122 CV_NEXT_SEQ_ELEM( contour->elem_size, reader );
123 }
124 }
125 else
126 {
127 for(size_t i = 0; i < reader_size; ++i)
128 {
129 reader_mat.at<float>(0, i++) = ((CvPoint*)(reader.ptr))->x;
130 reader_mat.at<float>(0, i) = ((CvPoint*)(reader.ptr))->y;
131 CV_NEXT_SEQ_ELEM( contour->elem_size, reader );
132 }
133 }
134
135 cv::ocl::oclMat dst_a(10, lpt, CV_64FC1);
136 cv::ocl::oclMat reader_oclmat(reader_mat);
137 int llength = std::min(lpt,128);
138 size_t localThreads[3] = { (size_t)llength, 1, 1};
139 size_t globalThreads[3] = { (size_t)lpt, 1, 1};
140 vector<pair<size_t , const void *> > args;
141 args.push_back( make_pair( sizeof(cl_int) , (void *)&contour->total ));
142 args.push_back( make_pair( sizeof(cl_mem) , (void *)&reader_oclmat.data ));
143 args.push_back( make_pair( sizeof(cl_mem) , (void *)&dst_a.data ));
144 cl_int dst_step = (cl_int)dst_a.step;
145 args.push_back( make_pair( sizeof(cl_int) , (void *)&dst_step ));
146
147 char builOption[128];
148 snprintf(builOption, 128, "-D CV_8UC1");
149
150 openCLExecuteKernel(dst_a.clCxt, &moments, "icvContourMoments", globalThreads, localThreads, args, -1, -1, builOption);
151
152 cv::Mat dst(dst_a);
153 a00 = a10 = a01 = a20 = a11 = a02 = a30 = a21 = a12 = a03 = 0.0;
154 if (!cv::ocl::Context::getContext()->supportsFeature(FEATURE_CL_DOUBLE))
155 {
156 for (int i = 0; i < contour->total; ++i)
157 {
158 a00 += dst.at<cl_long>(0, i);
159 a10 += dst.at<cl_long>(1, i);
160 a01 += dst.at<cl_long>(2, i);
161 a20 += dst.at<cl_long>(3, i);
162 a11 += dst.at<cl_long>(4, i);
163 a02 += dst.at<cl_long>(5, i);
164 a30 += dst.at<cl_long>(6, i);
165 a21 += dst.at<cl_long>(7, i);
166 a12 += dst.at<cl_long>(8, i);
167 a03 += dst.at<cl_long>(9, i);
168 }
169 }
170 else
171 {
172 a00 = cv::sum(dst.row(0))[0];
173 a10 = cv::sum(dst.row(1))[0];
174 a01 = cv::sum(dst.row(2))[0];
175 a20 = cv::sum(dst.row(3))[0];
176 a11 = cv::sum(dst.row(4))[0];
177 a02 = cv::sum(dst.row(5))[0];
178 a30 = cv::sum(dst.row(6))[0];
179 a21 = cv::sum(dst.row(7))[0];
180 a12 = cv::sum(dst.row(8))[0];
181 a03 = cv::sum(dst.row(9))[0];
182 }
183
184 double db1_2, db1_6, db1_12, db1_24, db1_20, db1_60;
185 if( fabs(a00) > FLT_EPSILON )
186 {
187 if( a00 > 0 )
188 {
189 db1_2 = 0.5;
190 db1_6 = 0.16666666666666666666666666666667;
191 db1_12 = 0.083333333333333333333333333333333;
192 db1_24 = 0.041666666666666666666666666666667;
193 db1_20 = 0.05;
194 db1_60 = 0.016666666666666666666666666666667;
195 }
196 else
197 {
198 db1_2 = -0.5;
199 db1_6 = -0.16666666666666666666666666666667;
200 db1_12 = -0.083333333333333333333333333333333;
201 db1_24 = -0.041666666666666666666666666666667;
202 db1_20 = -0.05;
203 db1_60 = -0.016666666666666666666666666666667;
204 }
205
206 // spatial moments
207 mom->m00 = a00 * db1_2;
208 mom->m10 = a10 * db1_6;
209 mom->m01 = a01 * db1_6;
210 mom->m20 = a20 * db1_12;
211 mom->m11 = a11 * db1_24;
212 mom->m02 = a02 * db1_12;
213 mom->m30 = a30 * db1_20;
214 mom->m21 = a21 * db1_60;
215 mom->m12 = a12 * db1_60;
216 mom->m03 = a03 * db1_20;
217
218 icvCompleteMomentState( mom );
219 }
220 }
221 }

223 Moments ocl_moments(oclMat& src, bool binary) //for image
224 {
225 CV_Assert(src.oclchannels() == 1);
226 if(src.type() == CV_64FC1 && !Context::getContext()->supportsFeature(FEATURE_CL_DOUBLE))
227 {
228 CV_Error(CV_StsUnsupportedFormat, "Moments - double is not supported by your GPU!");
229 }
230
231 if(binary)
232 {
233 oclMat mask;
234 if(src.type() != CV_8UC1)
235 {
236 src.convertTo(mask, CV_8UC1);
237 }
238 oclMat src8u(src.size(), CV_8UC1);
239 src8u.setTo(Scalar(255), mask);
240 src = src8u;
241 }
242 const int TILE_SIZE = 256;
243
244 CvMoments mom;
245 memset(&mom, 0, sizeof(mom));
246
247 cv::Size size = src.size();
248 int blockx, blocky;
249 blockx = (size.width + TILE_SIZE - 1)/TILE_SIZE;
250 blocky = (size.height + TILE_SIZE - 1)/TILE_SIZE;
251
252 oclMat dst_m;
253 int tile_height = TILE_SIZE;
254
255 size_t localThreads[3] = {1, (size_t)tile_height, 1};
256 size_t globalThreads[3] = {(size_t)blockx, (size_t)size.height, 1};
257
258 if(Context::getContext()->supportsFeature(FEATURE_CL_DOUBLE))
259 {
260 dst_m.create(blocky * 10, blockx, CV_64FC1);
261 }else
262 {
263 dst_m.create(blocky * 10, blockx, CV_32FC1);
264 }
265
266 int src_step = (int)(src.step/src.elemSize());
267 int dstm_step = (int)(dst_m.step/dst_m.elemSize());
268
269 vector<pair<size_t , const void *> > args;
270 args.push_back( make_pair( sizeof(cl_mem) , (void *)&src.data ));
271 args.push_back( make_pair( sizeof(cl_int) , (void *)&src.rows ));
272 args.push_back( make_pair( sizeof(cl_int) , (void *)&src.cols ));
273 args.push_back( make_pair( sizeof(cl_int) , (void *)&src_step ));
274 args.push_back( make_pair( sizeof(cl_mem) , (void *)&dst_m.data ));
275 args.push_back( make_pair( sizeof(cl_int) , (void *)&dst_m.cols ));
276 args.push_back( make_pair( sizeof(cl_int) , (void *)&dstm_step ));
277
278 int binary_;
279 if(binary)
280 binary_ = 1;
281 else
282 binary_ = 0;
283 args.push_back( make_pair( sizeof(cl_int) , (void *)&binary_));
284
285 char builOption[128];
286 if(binary || src.type() == CV_8UC1)
287 {
288 snprintf(builOption, 128, "-D CV_8UC1");
289 }else if(src.type() == CV_16UC1)
290 {
291 snprintf(builOption, 128, "-D CV_16UC1");
292 }else if(src.type() == CV_16SC1)
293 {
294 snprintf(builOption, 128, "-D CV_16SC1");
295 }else if(src.type() == CV_32FC1)
296 {
297 snprintf(builOption, 128, "-D CV_32FC1");
298 }else if(src.type() == CV_64FC1)
299 {
300 snprintf(builOption, 128, "-D CV_64FC1");
301 }else
302 {
303 CV_Error( CV_StsUnsupportedFormat, "" );
304 }
305
306 openCLExecuteKernel(Context::getContext(), &moments, "CvMoments", globalThreads, localThreads, args, -1, -1, builOption);
307
308 Mat tmp(dst_m);
309 tmp.convertTo(tmp, CV_64FC1);
310
311 double tmp_m[10] = {0};
312
313 for(int j = 0; j < tmp.rows; j += 10)
314 {
315 for(int i = 0; i < tmp.cols; i++)
316 {
317 tmp_m[0] += tmp.at<double>(j, i);
318 tmp_m[1] += tmp.at<double>(j + 1, i);
319 tmp_m[2] += tmp.at<double>(j + 2, i);
320 tmp_m[3] += tmp.at<double>(j + 3, i);
321 tmp_m[4] += tmp.at<double>(j + 4, i);
322 tmp_m[5] += tmp.at<double>(j + 5, i);
323 tmp_m[6] += tmp.at<double>(j + 6, i);
324 tmp_m[7] += tmp.at<double>(j + 7, i);
325 tmp_m[8] += tmp.at<double>(j + 8, i);
326 tmp_m[9] += tmp.at<double>(j + 9, i);
327 }
328 }
329
330 mom.m00 = tmp_m[0];
331 mom.m10 = tmp_m[1];
332 mom.m01 = tmp_m[2];
333 mom.m20 = tmp_m[3];
334 mom.m11 = tmp_m[4];
335 mom.m02 = tmp_m[5];
336 mom.m30 = tmp_m[6];
337 mom.m21 = tmp_m[7];
338 mom.m12 = tmp_m[8];
339 mom.m03 = tmp_m[9];
340 icvCompleteMomentState( &mom );
341 return mom;
342 }

344 Moments ocl_moments(InputArray _contour) //for contour
345 {
346 CvMoments mom;
347 memset(&mom, 0, sizeof(mom));
348
349 Mat arr = _contour.getMat();
350 CvMat c_array = arr;
351
352 const void* array = &c_array;
353
354 CvSeq* contour = 0;
355 if( CV_IS_SEQ( array ))
356 {
357 contour = (CvSeq*)(array);
358 if( !CV_IS_SEQ_POINT_SET( contour ))
359 CV_Error( CV_StsBadArg, "The passed sequence is not a valid contour" );
360 }
361
362 int type, coi = 0;
363
364 CvMat stub, *mat = (CvMat*)(array);
365 CvContour contourHeader;
366 CvSeqBlock block;
367
368 if( !contour )
369 {
370 mat = cvGetMat( mat, &stub, &coi );
371 type = CV_MAT_TYPE( mat->type );
372
373 if( type == CV_32SC2 || type == CV_32FC2 )
374 {
375 contour = cvPointSeqFromMat(
376 CV_SEQ_KIND_CURVE | CV_SEQ_FLAG_CLOSED,
377 mat, &contourHeader, &block );
378 }
379 }
380
381 CV_Assert(contour);
382
383 icvContourMoments(contour, &mom);
384 return mom;
385 }
