113cv::ocl::MOG::MOG(int nmixtures) :
114frameSize_(0, 0), frameType_(0), nframes_(0)
115{
116 nmixtures_ = std::min(nmixtures > 0 ? nmixtures : mog::defaultNMixtures, 8);
117 history = mog::defaultHistory;
118 varThreshold = mog::defaultVarThreshold;
119 backgroundRatio = mog::defaultBackgroundRatio;
120 noiseSigma = mog::defaultNoiseSigma;

123void cv::ocl::MOG::initialize(cv::Size frameSize, int frameType)
124{
125 CV_Assert(frameType == CV_8UC1 || frameType == CV_8UC3 || frameType == CV_8UC4);
126
127 frameSize_ = frameSize;
128 frameType_ = frameType;
129
130 int ch = CV_MAT_CN(frameType);
131 int work_ch = ch;
132
133 // for each gaussian mixture of each pixel bg model we store
134 // the mixture sort key (w/sum_of_variances), the mixture weight (w),
135 // the mean (nchannels values) and
136 // the diagonal covariance matrix (another nchannels values)
137
138 weight_.create(frameSize.height * nmixtures_, frameSize_.width, CV_32FC1);
139 sortKey_.create(frameSize.height * nmixtures_, frameSize_.width, CV_32FC1);
140 mean_.create(frameSize.height * nmixtures_, frameSize_.width, CV_32FC(work_ch));
141 var_.create(frameSize.height * nmixtures_, frameSize_.width, CV_32FC(work_ch));
142
143 weight_.setTo(cv::Scalar::all(0));
144 sortKey_.setTo(cv::Scalar::all(0));
145 mean_.setTo(cv::Scalar::all(0));
146 var_.setTo(cv::Scalar::all(0));
147
148 nframes_ = 0;
149}

151void cv::ocl::MOG::operator()(const cv::ocl::oclMat& frame, cv::ocl::oclMat& fgmask, float learningRate)
152{
153 using namespace cv::ocl::device::mog;
154
155 CV_Assert(frame.depth() == CV_8U);
156
157 int ch = frame.oclchannels();
158 int work_ch = ch;
159
160 if (nframes_ == 0 || learningRate >= 1.0 || frame.size() != frameSize_ || work_ch != mean_.oclchannels())
161 initialize(frame.size(), frame.type());
162
163 fgmask.create(frameSize_, CV_8UC1);
164
165 ++nframes_;
166 learningRate = learningRate >= 0.0f && nframes_ > 1 ? learningRate : 1.0f / std::min(nframes_, history);
167 CV_Assert(learningRate >= 0.0f);
168
169 mog_ocl(frame, ch, fgmask, weight_, sortKey_, mean_, var_, nmixtures_,
170 varThreshold, learningRate, backgroundRatio, noiseSigma);
171}

195static void mog_withoutLearning(const oclMat& frame, int cn, oclMat& fgmask, oclMat& weight, oclMat& mean, oclMat& var,
196 int nmixtures, float varThreshold, float backgroundRatio)
197{
198 Context* clCxt = Context::getContext();
199
200 size_t local_thread[] = {32, 8, 1};
201 size_t global_thread[] = {(size_t)frame.cols, (size_t)frame.rows, 1};
202
203 int frame_step = (int)(frame.step/frame.elemSize());
204 int fgmask_step = (int)(fgmask.step/fgmask.elemSize());
205 int weight_step = (int)(weight.step/weight.elemSize());
206 int mean_step = (int)(mean.step/mean.elemSize());
207 int var_step = (int)(var.step/var.elemSize());
208
209 int fgmask_offset_y = (int)(fgmask.offset/fgmask.step);
210 int fgmask_offset_x = (int)(fgmask.offset%fgmask.step);
211 fgmask_offset_x = fgmask_offset_x/(int)fgmask.elemSize();
212
213 int frame_offset_y = (int)(frame.offset/frame.step);
214 int frame_offset_x = (int)(frame.offset%frame.step);
215 frame_offset_x = frame_offset_x/(int)frame.elemSize();
216
217 char build_option[50];
218 if(cn == 1)
219 {
220 snprintf(build_option, 50, "-D CN1 -D NMIXTURES=%d", nmixtures);
221 }else
222 {
223 snprintf(build_option, 50, "-D NMIXTURES=%d", nmixtures);
224 }
225
226 String kernel_name = "mog_withoutLearning_kernel";
227 vector< pair<size_t, const void*> > args;
228
229 args.push_back(make_pair(sizeof(cl_mem), (void*)&frame.data));
230 args.push_back(make_pair(sizeof(cl_mem), (void*)&fgmask.data));
231 args.push_back(make_pair(sizeof(cl_mem), (void*)&weight.data));
232 args.push_back(make_pair(sizeof(cl_mem), (void*)&mean.data));
233 args.push_back(make_pair(sizeof(cl_mem), (void*)&var.data));
234
235 args.push_back(make_pair(sizeof(cl_int), (void*)&frame.rows));
236 args.push_back(make_pair(sizeof(cl_int), (void*)&frame.cols));
237
238 args.push_back(make_pair(sizeof(cl_int), (void*)&frame_step));
239 args.push_back(make_pair(sizeof(cl_int), (void*)&fgmask_step));
240 args.push_back(make_pair(sizeof(cl_int), (void*)&weight_step));
241 args.push_back(make_pair(sizeof(cl_int), (void*)&mean_step));
242 args.push_back(make_pair(sizeof(cl_int), (void*)&var_step));
243
244 args.push_back(make_pair(sizeof(cl_float), (void*)&varThreshold));
245 args.push_back(make_pair(sizeof(cl_float), (void*)&backgroundRatio));
246
247 args.push_back(make_pair(sizeof(cl_int), (void*)&fgmask_offset_x));
248 args.push_back(make_pair(sizeof(cl_int), (void*)&fgmask_offset_y));
249
250 args.push_back(make_pair(sizeof(cl_int), (void*)&frame_offset_x));
251 args.push_back(make_pair(sizeof(cl_int), (void*)&frame_offset_y));
252
253 openCLExecuteKernel(clCxt, &bgfg_mog, kernel_name, global_thread, local_thread, args, -1, -1, build_option);
254}

327void cv::ocl::device::mog::mog_ocl(const oclMat& frame, int cn, oclMat& fgmask, oclMat& weight, oclMat& sortKey, oclMat& mean, oclMat& var,
328 int nmixtures, float varThreshold, float learningRate, float backgroundRatio, float noiseSigma)
329{
330 const float minVar = noiseSigma * noiseSigma;
331
332 if(learningRate > 0.0f)
333 mog_withLearning(frame, cn, fgmask, weight, sortKey, mean, var, nmixtures,
334 varThreshold, backgroundRatio, learningRate, minVar);
335 else

378void cv::ocl::device::mog::loadConstants(float Tb, float TB, float Tg, float varInit, float varMin, float varMax, float tau, unsigned char shadowVal)
379{
380 varMin = cv::min(varMin, varMax);
381 varMax = cv::max(varMin, varMax);
382
383 c_TB = TB;
384
385 _contant_struct *constants = new _contant_struct;
386 constants->c_Tb = Tb;
387 constants->c_TB = TB;
388 constants->c_Tg = Tg;
389 constants->c_varInit = varInit;
390 constants->c_varMin = varMin;
391 constants->c_varMax = varMax;
392 constants->c_tau = tau;
393 constants->c_shadowVal = shadowVal;
394
395 cl_constants = load_constant(*((cl_context*)getClContextPtr()), *((cl_command_queue*)getClCommandQueuePtr()),
396 (void *)constants, sizeof(_contant_struct));
397}

524}

531 // default parameters of gaussian background detection algorithm
532 const int defaultHistory = 500; // Learning rate; alpha = 1/defaultHistory2
533 const float defaultVarThreshold = 4.0f * 4.0f;
534 const int defaultNMixtures = 5; // maximal number of Gaussians in mixture
535 const float defaultBackgroundRatio = 0.9f; // threshold sum of weights for background test
536 const float defaultVarThresholdGen = 3.0f * 3.0f;
537 const float defaultVarInit = 15.0f; // initial variance for new components
538 const float defaultVarMax = 5.0f * defaultVarInit;
539 const float defaultVarMin = 4.0f;
540
541 // additional parameters
542 const float defaultfCT = 0.05f; // complexity reduction prior constant 0 - no reduction of number of components
543 const unsigned char defaultnShadowDetection = 127; // value to use in the segmentation mask for shadows, set 0 not to do shadow detection
544 const float defaultfTau = 0.5f; // Tau - shadow threshold, see the paper for explanation
545}

547cv::ocl::MOG2::MOG2(int nmixtures) : frameSize_(0, 0), frameType_(0), nframes_(0)
548{
549 nmixtures_ = nmixtures > 0 ? nmixtures : mog2::defaultNMixtures;
550
551 history = mog2::defaultHistory;
552 varThreshold = mog2::defaultVarThreshold;
553 bShadowDetection = true;
554
555 backgroundRatio = mog2::defaultBackgroundRatio;
556 fVarInit = mog2::defaultVarInit;
557 fVarMax = mog2::defaultVarMax;
558 fVarMin = mog2::defaultVarMin;
559
560 varThresholdGen = mog2::defaultVarThresholdGen;
561 fCT = mog2::defaultfCT;
562 nShadowDetection = mog2::defaultnShadowDetection;
563 fTau = mog2::defaultfTau;
564}

566void cv::ocl::MOG2::initialize(cv::Size frameSize, int frameType)
567{
568 using namespace cv::ocl::device::mog;
569 CV_Assert(frameType == CV_8UC1 || frameType == CV_8UC3 || frameType == CV_8UC4);
570
571 frameSize_ = frameSize;
572 frameType_ = frameType;
573 nframes_ = 0;
574
575 int ch = CV_MAT_CN(frameType);
576 int work_ch = ch;
577
578 // for each gaussian mixture of each pixel bg model we store ...
579 // the mixture weight (w),
580 // the mean (nchannels values) and
581 // the covariance
582 weight_.create(frameSize.height * nmixtures_, frameSize_.width, CV_32FC1);
583 weight_.setTo(Scalar::all(0));
584
585 variance_.create(frameSize.height * nmixtures_, frameSize_.width, CV_32FC1);
586 variance_.setTo(Scalar::all(0));
587
588 mean_.create(frameSize.height * nmixtures_, frameSize_.width, CV_32FC(work_ch)); //4 channels
589 mean_.setTo(Scalar::all(0));
590
591 //make the array for keeping track of the used modes per pixel - all zeros at start
592 bgmodelUsedModes_.create(frameSize_, CV_32FC1);
593 bgmodelUsedModes_.setTo(cv::Scalar::all(0));
594
595 loadConstants(varThreshold, backgroundRatio, varThresholdGen, fVarInit, fVarMin, fVarMax, fTau, nShadowDetection);
596}
597
598void cv::ocl::MOG2::operator()(const oclMat& frame, oclMat& fgmask, float learningRate)
599{
600 using namespace cv::ocl::device::mog;
601
602 int ch = frame.oclchannels();
603 int work_ch = ch;
604
605 if (nframes_ == 0 || learningRate >= 1.0f || frame.size() != frameSize_ || work_ch != mean_.oclchannels())
606 initialize(frame.size(), frame.type());
607
608 fgmask.create(frameSize_, CV_8UC1);
609 fgmask.setTo(cv::Scalar::all(0));
610
611 ++nframes_;
612 learningRate = learningRate >= 0.0f && nframes_ > 1 ? learningRate : 1.0f / std::min(2 * nframes_, history);
613 CV_Assert(learningRate >= 0.0f);
614
615 mog2_ocl(frame, frame.oclchannels(), fgmask, bgmodelUsedModes_, weight_, variance_, mean_, learningRate, -learningRate * fCT, bShadowDetection, nmixtures_);
616}
