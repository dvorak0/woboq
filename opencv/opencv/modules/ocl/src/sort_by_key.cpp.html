74void static genSortBuildOption(const oclMat& keys, const oclMat& vals, bool isGreaterThan, char * build_opt_buf)
75{
76 sprintf(build_opt_buf, "-D IS_GT=%d -D K_T=%s -D V_T=%s",

90static void sortByKey(oclMat& keys, oclMat& vals, size_t vecSize, bool isGreaterThan)
91{
92 CV_Assert(isSizePowerOf2(vecSize));
93
94 Context * cxt = Context::getContext();
95 size_t globalThreads[3] = {vecSize / 2, 1, 1};
96
97 // 2^numStages should be equal to vecSize or the output is invalid
98 int numStages = 0;
99 for(int i = vecSize; i > 1; i >>= 1)
100 {
101 ++numStages;
102 }
103 char build_opt_buf [100];
104 genSortBuildOption(keys, vals, isGreaterThan, build_opt_buf);
105 const int argc = 5;
106 std::vector< std::pair<size_t, const void *> > args(argc);
107 String kernelname = "bitonicSort";
108
109 args[0] = std::make_pair(sizeof(cl_mem), (void *)&keys.data);
110 args[1] = std::make_pair(sizeof(cl_mem), (void *)&vals.data);
111 args[2] = std::make_pair(sizeof(cl_int), (void *)&vecSize);
112
113 for(int stage = 0; stage < numStages; ++stage)
114 {
115 args[3] = std::make_pair(sizeof(cl_int), (void *)&stage);
116 for(int passOfStage = 0; passOfStage < stage + 1; ++passOfStage)
117 {
118 args[4] = std::make_pair(sizeof(cl_int), (void *)&passOfStage);
119#ifdef ANDROID
120 openCLExecuteKernel(cxt, &kernel_sort_by_key, kernelname, globalThreads, NULL, args, -1, -1, build_opt_buf);
121#else
122 size_t localThreads[3] = {GROUP_SIZE, 1, 1};
123 openCLExecuteKernel(cxt, &kernel_sort_by_key, kernelname, globalThreads, localThreads, args, -1, -1, build_opt_buf);
124#endif
125 }
126 }
127}

197void static naive_scan_addition_cpu(oclMat& input, oclMat& output)
198{
199 Mat m_input = input, m_output(output.size(), output.type());
200 MatIterator_<int> i_mit = m_input.begin<int>();
201 MatIterator_<int> o_mit = m_output.begin<int>();
202 *o_mit = 0;
203 ++i_mit;
204 ++o_mit;
205 for(; i_mit != m_input.end<int>(); ++i_mit, ++o_mit)
206 {
207 *o_mit = *(o_mit - 1) + *(i_mit - 1);
208 }
209 output = m_output;
210}

213//radix sort ported from Bolt
214static void sortByKey(oclMat& keys, oclMat& vals, size_t origVecSize, bool isGreaterThan)
215{
216 CV_Assert(keys.depth() == CV_32S || keys.depth() == CV_32F); // we assume keys are 4 bytes
217
218 bool isKeyFloat = keys.type() == CV_32F;
219
220 const int RADIX = 4; //Now you cannot replace this with Radix 8 since there is a
221 //local array of 16 elements in the histogram kernel.
222 const int RADICES = (1 << RADIX); //Values handeled by each work-item?
223
224 bool newBuffer = false;
225 size_t vecSize = origVecSize;
226
227 unsigned int groupSize = RADICES;
228
229 size_t mulFactor = groupSize * RADICES;
230
231 oclMat buffer_keys, buffer_vals;
232
233 if(origVecSize % mulFactor != 0)
234 {
235 vecSize = ((vecSize + mulFactor) / mulFactor) * mulFactor;
236 buffer_keys.create(1, vecSize, keys.type());
237 buffer_vals.create(1, vecSize, vals.type());
238 Scalar padding_value;
239 oclMat roi_buffer_vals = buffer_vals(Rect(0,0,origVecSize,1));
240
241 if(isGreaterThan)
242 {
243 switch(buffer_keys.depth())
244 {
245 case CV_32F:
246 padding_value = Scalar::all(-FLT_MAX);
247 break;
248 case CV_32S:
249 padding_value = Scalar::all(INT_MIN);
250 break;
251 }
252 }
253 else
254 {
255 switch(buffer_keys.depth())
256 {
257 case CV_32F:
258 padding_value = Scalar::all(FLT_MAX);
259 break;
260 case CV_32S:
261 padding_value = Scalar::all(INT_MAX);
262 break;
263 }
264 }
265 ocl::copyMakeBorder(
266 keys(Rect(0,0,origVecSize,1)), buffer_keys,
267 0, 0, 0, vecSize - origVecSize,
268 BORDER_CONSTANT, padding_value);
269 vals(Rect(0,0,origVecSize,1)).copyTo(roi_buffer_vals);
270 newBuffer = true;
271 }
272 else
273 {
274 buffer_keys = keys;
275 buffer_vals = vals;
276 newBuffer = false;
277 }
278 oclMat swap_input_keys(1, vecSize, keys.type());
279 oclMat swap_input_vals(1, vecSize, vals.type());
280 oclMat hist_bin_keys(1, vecSize, CV_32SC1);
281 oclMat hist_bin_dest_keys(1, vecSize, CV_32SC1);
282
283 Context * cxt = Context::getContext();
284
285 size_t globalThreads[3] = {vecSize / RADICES, 1, 1};
286 size_t localThreads[3] = {groupSize, 1, 1};
287
288 std::vector< std::pair<size_t, const void *> > args;
289 char build_opt_buf [100];
290 genSortBuildOption(keys, vals, isGreaterThan, build_opt_buf);
291
292 //additional build option for radix sort
293 sprintf(build_opt_buf + strlen(build_opt_buf), " -D K_%s", isKeyFloat?"FLT":"INT");
294
295 String kernelnames[2] = {String("histogramRadixN"), String("permuteRadixN")};
296
297 int swap = 0;
298 for(int bits = 0; bits < (static_cast<int>(keys.elemSize()) * 8); bits += RADIX)
299 {
300 args.clear();
301 //Do a histogram pass locally
302 if(swap == 0)
303 {
304 args.push_back(std::make_pair(sizeof(cl_mem), (void *)&buffer_keys.data));
305 }
306 else
307 {
308 args.push_back(std::make_pair(sizeof(cl_mem), (void *)&swap_input_keys.data));
309 }
310 args.push_back(std::make_pair(sizeof(cl_mem), (void *)&hist_bin_keys.data));
311 args.push_back(std::make_pair(sizeof(cl_int), (void *)&bits));
312 openCLExecuteKernel(cxt, &kernel_radix_sort_by_key, kernelnames[0], globalThreads, localThreads,
313 args, -1, -1, build_opt_buf);
314
315 args.clear();
316 //Perform a global scan
317 naive_scan_addition_cpu(hist_bin_keys, hist_bin_dest_keys);
318 // end of scan
319 if(swap == 0)
320 {
321 args.push_back(std::make_pair(sizeof(cl_mem), (void *)&buffer_keys.data));
322 args.push_back(std::make_pair(sizeof(cl_mem), (void *)&buffer_vals.data));
323 }
324 else
325 {
326 args.push_back(std::make_pair(sizeof(cl_mem), (void *)&swap_input_keys.data));
327 args.push_back(std::make_pair(sizeof(cl_mem), (void *)&swap_input_vals.data));
328 }
329 args.push_back(std::make_pair(sizeof(cl_mem), (void *)&hist_bin_dest_keys.data));
330 args.push_back(std::make_pair(sizeof(cl_int), (void *)&bits));
331
332 if(swap == 0)
333 {
334 args.push_back(std::make_pair(sizeof(cl_mem), (void *)&swap_input_keys.data));
335 args.push_back(std::make_pair(sizeof(cl_mem), (void *)&swap_input_vals.data));
336 }
337 else
338 {
339 args.push_back(std::make_pair(sizeof(cl_mem), (void *)&buffer_keys.data));
340 args.push_back(std::make_pair(sizeof(cl_mem), (void *)&buffer_vals.data));
341 }
342 openCLExecuteKernel(cxt, &kernel_radix_sort_by_key, kernelnames[1], globalThreads, localThreads,
343 args, -1, -1, build_opt_buf);
344 swap = swap ? 0 : 1;
345 }
346 if(newBuffer)
347 {
348 buffer_keys(Rect(0,0,origVecSize,1)).copyTo(keys);
349 buffer_vals(Rect(0,0,origVecSize,1)).copyTo(vals);
350 }
351}

355namespace merge_sort
356{
357static void sortByKey(oclMat& keys, oclMat& vals, size_t vecSize, bool isGreaterThan)
358{
359 Context * cxt = Context::getContext();
360
361 const size_t groupSize = cxt->getDeviceInfo().maxWorkGroupSize >= 256 ? 256: 128;
362
363 size_t globalThreads[3] = {vecSize, 1, 1};
364 size_t localThreads[3] = {groupSize, 1, 1};
365
366 std::vector< std::pair<size_t, const void *> > args;
367 char build_opt_buf [100];
368 genSortBuildOption(keys, vals, isGreaterThan, build_opt_buf);
369
370 String kernelname[] = {String("blockInsertionSort"), String("merge")};
371 int keylds_size = groupSize * keys.elemSize();
372 int vallds_size = groupSize * vals.elemSize();
373 args.push_back(std::make_pair(sizeof(cl_mem), (void *)&keys.data));
374 args.push_back(std::make_pair(sizeof(cl_mem), (void *)&vals.data));
375 args.push_back(std::make_pair(sizeof(cl_uint), (void *)&vecSize));
376 args.push_back(std::make_pair(keylds_size, (void*)NULL));
377 args.push_back(std::make_pair(vallds_size, (void*)NULL));
378
379 openCLExecuteKernel(cxt, &kernel_stablesort_by_key, kernelname[0], globalThreads, localThreads, args, -1, -1, build_opt_buf);
380
381 // Early exit for the case of no merge passes, values are already in destination vector
382 if(vecSize <= groupSize)
383 {
384 return;
385 }
386
387 // An odd number of elements requires an extra merge pass to sort
388 size_t numMerges = 0;
389 // Calculate the log2 of vecSize, taking into acvecSize our block size from kernel 1 is 64
390 // this is how many merge passes we want
391 size_t log2BlockSize = vecSize >> 6;
392 for( ; log2BlockSize > 1; log2BlockSize >>= 1 )
393 {
394 ++numMerges;
395 }
396 // Check to see if the input vector size is a power of 2, if not we will need last merge pass
397 numMerges += isSizePowerOf2(vecSize)? 1: 0;
398
399 // Allocate a flipflop buffer because the merge passes are out of place
400 oclMat tmpKeyBuffer(keys.size(), keys.type());
401 oclMat tmpValBuffer(vals.size(), vals.type());
402 args.resize(8);
403
404 args[4] = std::make_pair(sizeof(cl_uint), (void *)&vecSize);
405 args[6] = std::make_pair(keylds_size, (void*)NULL);
406 args[7] = std::make_pair(vallds_size, (void*)NULL);
407
408 for(size_t pass = 1; pass <= numMerges; ++pass )
409 {
410 // For each pass, flip the input-output buffers
411 if( pass & 0x1 )
412 {
413 args[0] = std::make_pair(sizeof(cl_mem), (void *)&keys.data);
414 args[1] = std::make_pair(sizeof(cl_mem), (void *)&vals.data);
415 args[2] = std::make_pair(sizeof(cl_mem), (void *)&tmpKeyBuffer.data);
416 args[3] = std::make_pair(sizeof(cl_mem), (void *)&tmpValBuffer.data);
417 }
418 else
419 {
420 args[0] = std::make_pair(sizeof(cl_mem), (void *)&tmpKeyBuffer.data);
421 args[1] = std::make_pair(sizeof(cl_mem), (void *)&tmpValBuffer.data);
422 args[2] = std::make_pair(sizeof(cl_mem), (void *)&keys.data);
423 args[3] = std::make_pair(sizeof(cl_mem), (void *)&vals.data);
424 }
425 // For each pass, the merge window doubles
426 unsigned int srcLogicalBlockSize = static_cast<unsigned int>( localThreads[0] << (pass-1) );
427 args[5] = std::make_pair(sizeof(cl_uint), (void *)&srcLogicalBlockSize);
428 openCLExecuteKernel(cxt, &kernel_stablesort_by_key, kernelname[1], globalThreads, localThreads, args, -1, -1, build_opt_buf);
429 }
430 // If there are an odd number of merges, then the output data is sitting in the temp buffer. We need to copy
431 // the results back into the input array
432 if( numMerges & 1 )
433 {
434 tmpKeyBuffer.copyTo(keys);
435 tmpValBuffer.copyTo(vals);
436 }
437}
438} /* merge_sort */

444void cv::ocl::sortByKey(oclMat& keys, oclMat& vals, size_t vecSize, int method, bool isGreaterThan)
445{
446 CV_Assert( keys.rows == 1 ); // we only allow one dimensional input
447 CV_Assert( keys.channels() == 1 ); // we only allow one channel keys
448 CV_Assert( vecSize <= static_cast<size_t>(keys.cols) );
449 switch(method)
450 {
451 case SORT_BITONIC:
452 bitonic_sort::sortByKey(keys, vals, vecSize, isGreaterThan);
453 break;
454 case SORT_SELECTION:
455 selection_sort::sortByKey(keys, vals, vecSize, isGreaterThan);
456 break;
457 case SORT_RADIX:
458 radix_sort::sortByKey(keys, vals, vecSize, isGreaterThan);
459 break;
460 case SORT_MERGE:
461 merge_sort::sortByKey(keys, vals, vecSize, isGreaterThan);
462 break;
463 }
464}
465
466void cv::ocl::sortByKey(oclMat& keys, oclMat& vals, int method, bool isGreaterThan)
467{
468 CV_Assert( keys.size() == vals.size() );
469 CV_Assert( keys.rows == 1 ); // we only allow one dimensional input
470 size_t vecSize = static_cast<size_t>(keys.cols);
471 sortByKey(keys, vals, vecSize, method, isGreaterThan);
472}
