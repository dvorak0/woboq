61//////////////////////////prefilter_xsbel////////////////////////////////
62////////////////////////////////////////////////////////////////////////
63static void prefilter_xsobel(const oclMat &input, oclMat &output, int prefilterCap)
64{
65 string kernelName = "prefilter_xsobel";
66
67 size_t blockSize = 1;
68 size_t globalThreads[3] = { (size_t)input.cols, (size_t)input.rows, 1 };
69 size_t localThreads[3] = { blockSize, blockSize, 1 };
70
71 std::vector< std::pair<size_t, const void *> > args;
72 args.push_back(std::make_pair(sizeof(cl_mem), (void *)&input.data));
73 args.push_back(std::make_pair(sizeof(cl_mem), (void *)&output.data));
74 args.push_back(std::make_pair(sizeof(cl_int), (void *)&input.rows));
75 args.push_back(std::make_pair(sizeof(cl_int), (void *)&input.cols));
76 args.push_back(std::make_pair(sizeof(cl_int), (void *)&prefilterCap));
77
78 openCLExecuteKernel(Context::getContext(), &stereobm, kernelName,
79 globalThreads, localThreads, args, -1, -1);
80}

84#define N_DISPARITIES 8
85#define ROWSperTHREAD 21
86#define BLOCK_W 128

88////////////////////////////////////////////////////////////////////////////
89///////////////////////////////stereoBM_GPU////////////////////////////////
90////////////////////////////////////////////////////////////////////////////
91static void stereo_bm(const oclMat &left, const oclMat &right, oclMat &disp,
92 int maxdisp, int winSize, oclMat &minSSD_buf)
93{
94 int winsz2 = winSize >> 1;
95
96 string kernelName = "stereoKernel";
97
98 disp.setTo(Scalar_<unsigned char>::all(0));
99 minSSD_buf.setTo(Scalar_<unsigned int>::all(0xFFFFFFFF));
100
101 size_t minssd_step = minSSD_buf.step / minSSD_buf.elemSize();
102 size_t local_mem_size = (N_DISPARITIES * (BLOCK_W + 2 * winsz2)) *
103 sizeof(cl_uint);
104 //size_t blockSize = 1;
105 size_t localThreads[] = { BLOCK_W, 1, 1 };
106 size_t globalThreads[] = { (size_t)left.cols - maxdisp - 2 * winsz2,
107 divUp(left.rows - 2 * winsz2, ROWSperTHREAD),
108 1 };
109
110 std::vector< std::pair<size_t, const void *> > args;
111 args.push_back(std::make_pair(sizeof(cl_mem), (void *)&left.data));
112 args.push_back(std::make_pair(sizeof(cl_mem), (void *)&right.data));
113 args.push_back(std::make_pair(sizeof(cl_mem), (void *)&minSSD_buf.data));
114 args.push_back(std::make_pair(sizeof(cl_int), (void *)&minssd_step));
115 args.push_back(std::make_pair(sizeof(cl_mem), (void *)&disp.data));
116 args.push_back(std::make_pair(sizeof(cl_int), (void *)&disp.step));
117 args.push_back(std::make_pair(sizeof(cl_int), (void *)&left.cols));
118 args.push_back(std::make_pair(sizeof(cl_int), (void *)&left.rows));
119 args.push_back(std::make_pair(sizeof(cl_int), (void *)&left.step));
120 args.push_back(std::make_pair(sizeof(cl_int), (void *)&maxdisp));
121 args.push_back(std::make_pair(local_mem_size, (void *)NULL));
122
123 char opt [128];
124 sprintf(opt, "-D radius=%d", winsz2);
125 openCLExecuteKernel(Context::getContext(), &stereobm, kernelName,
126 globalThreads, localThreads, args, -1, -1, opt);
127}
128////////////////////////////////////////////////////////////////////////////
129///////////////////////////////postfilter_textureness///////////////////////
130////////////////////////////////////////////////////////////////////////////
131static void postfilter_textureness(oclMat &left, int winSize,
132 float avergeTexThreshold, oclMat &disparity)
133{
134 string kernelName = "textureness_kernel";
135
136 size_t blockSize = 1;
137 size_t localThreads[] = { BLOCK_W, blockSize ,1};
138 size_t globalThreads[] = { (size_t)left.cols,
139 divUp(left.rows, 2 * ROWSperTHREAD),
140 1 };
141
142 size_t local_mem_size = (localThreads[0] + localThreads[0] + (winSize / 2) * 2) * sizeof(float);
143
144 std::vector< std::pair<size_t, const void *> > args;
145 args.push_back(std::make_pair(sizeof(cl_mem), (void *)&disparity.data));
146 args.push_back(std::make_pair(sizeof(cl_int), (void *)&disparity.rows));
147 args.push_back(std::make_pair(sizeof(cl_int), (void *)&disparity.cols));
148 args.push_back(std::make_pair(sizeof(cl_int), (void *)&disparity.step));
149 args.push_back(std::make_pair(sizeof(cl_mem), (void *)&left.data));
150 args.push_back(std::make_pair(sizeof(cl_int), (void *)&left.rows));
151 args.push_back(std::make_pair(sizeof(cl_int), (void *)&left.cols));
152 args.push_back(std::make_pair(sizeof(cl_int), (void *)&winSize));
153 args.push_back(std::make_pair(sizeof(cl_float), (void *)&avergeTexThreshold));
154 args.push_back(std::make_pair(local_mem_size, (void*)NULL));
155 openCLExecuteKernel(Context::getContext(), &stereobm, kernelName,
156 globalThreads, localThreads, args, -1, -1);
157}
158//////////////////////////////////////////////////////////////////////////////
159/////////////////////////////////////operator/////////////////////////////////
160/////////////////////////////////////////////////////////////////////////////
161static void operator_(oclMat &minSSD, oclMat &leBuf, oclMat &riBuf, int preset, int ndisp,
162 int winSize, float avergeTexThreshold, const oclMat &left,
163 const oclMat &right, oclMat &disparity)
164
165{
166 CV_DbgAssert(left.rows == right.rows && left.cols == right.cols);
167 CV_DbgAssert(left.type() == CV_8UC1);
168 CV_DbgAssert(right.type() == CV_8UC1);
169
170 disparity.create(left.size(), CV_8UC1);
171 minSSD.create(left.size(), CV_32SC1);
172
173 oclMat le_for_bm = left;
174 oclMat ri_for_bm = right;
175
176 if (preset == cv::ocl::StereoBM_OCL::PREFILTER_XSOBEL)
177 {
178 leBuf.create( left.size(), left.type());
179 riBuf.create(right.size(), right.type());
180
181 prefilter_xsobel( left, leBuf, 31);
182 prefilter_xsobel(right, riBuf, 31);
183
184 le_for_bm = leBuf;
185 ri_for_bm = riBuf;
186 }
187
188 stereo_bm(le_for_bm, ri_for_bm, disparity, ndisp, winSize, minSSD);
189
190 if (avergeTexThreshold)
191 {
192 postfilter_textureness(le_for_bm, winSize, avergeTexThreshold, disparity);
193 }
194}
195}
196}

200cv::ocl::StereoBM_OCL::StereoBM_OCL()
201 : preset(BASIC_PRESET), ndisp(DEFAULT_NDISP), winSize(DEFAULT_WINSZ),
202 avergeTexThreshold(defaultAvgTexThreshold) {}
203
204cv::ocl::StereoBM_OCL::StereoBM_OCL(int preset_, int ndisparities_, int winSize_)
205 : preset(preset_), ndisp(ndisparities_), winSize(winSize_),
206 avergeTexThreshold(defaultAvgTexThreshold)
207{
208 const int max_supported_ndisp = 1 << (sizeof(unsigned char) * 8);
209 CV_Assert(0 < ndisp && ndisp <= max_supported_ndisp);
210 CV_Assert(ndisp % 8 == 0);
211 CV_Assert(winSize % 2 == 1);
212}
