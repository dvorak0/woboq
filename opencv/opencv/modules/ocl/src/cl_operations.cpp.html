112#define MEMORY_CORRUPTION_GUARD
113#ifdef MEMORY_CORRUPTION_GUARD
114//#define CHECK_MEMORY_CORRUPTION
115#define CHECK_MEMORY_CORRUPTION_PRINT_ERROR
116#define CHECK_MEMORY_CORRUPTION_RAISE_ERROR
117static const int __memory_corruption_guard_bytes = 64*1024;
118#ifdef CHECK_MEMORY_CORRUPTION
119static const int __memory_corruption_check_pattern = 0x14326547; // change pattern for sizeof(int)==8
120#endif
121struct CheckBuffers
122{
123 cl_mem mainBuffer;
124 size_t size;
125 size_t widthInBytes, height;
126 CheckBuffers()
127 : mainBuffer(NULL), size(0), widthInBytes(0), height(0)
128 {
129 // nothing
130 }
131 CheckBuffers(cl_mem _mainBuffer, size_t _size, size_t _widthInBytes, size_t _height)
132 : mainBuffer(_mainBuffer), size(_size), widthInBytes(_widthInBytes), height(_height)
133 {
134 // nothing
135 }
136};
137static std::map<cl_mem, CheckBuffers> __check_buffers;
138#endif

146void openCLMallocPitchEx(Context *ctx, void **dev_ptr, size_t *pitch,
147 size_t widthInBytes, size_t height, DevMemRW rw_type, DevMemType mem_type)
148{
149 cl_int status;
150 size_t size = widthInBytes * height;
151 bool useSubBuffers =
152#ifndef MEMORY_CORRUPTION_GUARD
153 false;
154#else
155 true;
156#endif
157 const DeviceInfo& devInfo = ctx->getDeviceInfo();
158 if (useSubBuffers && devInfo.isIntelDevice)
159 {
160 useSubBuffers = false; // TODO FIXIT We observe memory leaks then we working with sub-buffers
161 // on the CPU device of Intel OpenCL SDK (Linux). We will investigate this later.
162 }
163 if (!useSubBuffers)
164 {
165 *dev_ptr = clCreateBuffer(getClContext(ctx), gDevMemRWValueMap[rw_type]|gDevMemTypeValueMap[mem_type],
166 size, 0, &status);
167 openCLVerifyCall(status);
168 }
169#ifdef MEMORY_CORRUPTION_GUARD
170 else
171 {
172 size_t allocSize = size + __memory_corruption_guard_bytes * 2;
173 cl_mem mainBuffer = clCreateBuffer(getClContext(ctx), gDevMemRWValueMap[rw_type]|gDevMemTypeValueMap[mem_type],
174 allocSize, 0, &status);
175 openCLVerifyCall(status);
176 cl_buffer_region r = {__memory_corruption_guard_bytes, size};
177 *dev_ptr = clCreateSubBuffer(mainBuffer,
178 gDevMemRWValueMap[rw_type]|gDevMemTypeValueMap[mem_type],
179 CL_BUFFER_CREATE_TYPE_REGION, &r,
180 &status);
181 openCLVerifyCall(status);
182#ifdef CHECK_MEMORY_CORRUPTION
183 std::vector<int> tmp(__memory_corruption_guard_bytes / sizeof(int),
184 __memory_corruption_check_pattern);
185 CV_Assert(tmp.size() * sizeof(int) == __memory_corruption_guard_bytes);
186 openCLVerifyCall(clEnqueueWriteBuffer(getClCommandQueue(ctx),
187 mainBuffer, CL_FALSE, 0, __memory_corruption_guard_bytes, &tmp[0],
188 0, NULL, NULL));
189 openCLVerifyCall(clEnqueueWriteBuffer(getClCommandQueue(ctx),
190 mainBuffer, CL_FALSE, __memory_corruption_guard_bytes + size, __memory_corruption_guard_bytes, &tmp[0],
191 0, NULL, NULL));
192 clFinish(getClCommandQueue(ctx));
193#endif
194 CheckBuffers data(mainBuffer, size, widthInBytes, height);
195 cv::AutoLock lock(getInitializationMutex());
196 __check_buffers.insert(std::pair<cl_mem, CheckBuffers>((cl_mem)*dev_ptr, data));
197 }
198#endif
199 *pitch = widthInBytes;
200}

202void openCLMemcpy2D(Context *ctx, void *dst, size_t dpitch,
203 const void *src, size_t spitch,
204 size_t width, size_t height, openCLMemcpyKind kind, int channels)
205{
206 size_t buffer_origin[3] = {0, 0, 0};
207 size_t host_origin[3] = {0, 0, 0};
208 size_t region[3] = {width, height, 1};
209 if(kind == clMemcpyHostToDevice)
210 {
211 if(dpitch == width || channels == 3 || height == 1)
212 {
213 openCLSafeCall(clEnqueueWriteBuffer(getClCommandQueue(ctx), (cl_mem)dst, CL_TRUE,
214 0, width * height, src, 0, NULL, NULL));
215 }
216 else
217 {
218 openCLSafeCall(clEnqueueWriteBufferRect(getClCommandQueue(ctx), (cl_mem)dst, CL_TRUE,
219 buffer_origin, host_origin, region, dpitch, 0, spitch, 0, src, 0, 0, 0));
220 }
221 }
222 else if(kind == clMemcpyDeviceToHost)
223 {
224 if(spitch == width || channels == 3 || height == 1)
225 {
226 openCLSafeCall(clEnqueueReadBuffer(getClCommandQueue(ctx), (cl_mem)src, CL_TRUE,
227 0, width * height, dst, 0, NULL, NULL));
228 }
229 else
230 {
231 openCLSafeCall(clEnqueueReadBufferRect(getClCommandQueue(ctx), (cl_mem)src, CL_TRUE,
232 buffer_origin, host_origin, region, spitch, 0, dpitch, 0, dst, 0, 0, 0));
233 }
234 }

246 region, spitch, 0, dpitch, 0, 0, 0, 0));
247}

249void openCLFree(void *devPtr)
250{
251 openCLSafeCall(clReleaseMemObject((cl_mem)devPtr));
252#ifdef MEMORY_CORRUPTION_GUARD
253#ifdef CHECK_MEMORY_CORRUPTION
254 bool failBefore = false, failAfter = false;
255#endif
256 CheckBuffers data;
257 {
258 cv::AutoLock lock(getInitializationMutex());
259 std::map<cl_mem, CheckBuffers>::iterator i = __check_buffers.find((cl_mem)devPtr);
260 if (i != __check_buffers.end())
261 {
262 data = i->second;
263 __check_buffers.erase(i);
264 }
265 }
266 if (data.mainBuffer != NULL)
267 {
268#ifdef CHECK_MEMORY_CORRUPTION
269 Context* ctx = Context::getContext();
270 std::vector<uchar> checkBefore(__memory_corruption_guard_bytes);
271 std::vector<uchar> checkAfter(__memory_corruption_guard_bytes);
272 openCLVerifyCall(clEnqueueReadBuffer(getClCommandQueue(ctx),
273 data.mainBuffer, CL_FALSE, 0, __memory_corruption_guard_bytes, &checkBefore[0],
274 0, NULL, NULL));
275 openCLVerifyCall(clEnqueueReadBuffer(getClCommandQueue(ctx),
276 data.mainBuffer, CL_FALSE, __memory_corruption_guard_bytes + data.size, __memory_corruption_guard_bytes, &checkAfter[0],
277 0, NULL, NULL));
278 clFinish(getClCommandQueue(ctx));
279
280 std::vector<int> tmp(__memory_corruption_guard_bytes / sizeof(int),
281 __memory_corruption_check_pattern);
282
283 if (memcmp(&checkBefore[0], &tmp[0], __memory_corruption_guard_bytes) != 0)
284 {
285 failBefore = true;
286 }
287 if (memcmp(&checkAfter[0], &tmp[0], __memory_corruption_guard_bytes) != 0)
288 {
289 failAfter = true;
290 }
291#else
292 // TODO FIXIT Attach clReleaseMemObject call to event completion callback
293 // TODO 2013/12/04 Disable workaround
294 // Context* ctx = Context::getContext();
295 // clFinish(getClCommandQueue(ctx));
296#endif
297 openCLSafeCall(clReleaseMemObject(data.mainBuffer));
298 }
299#if defined(CHECK_MEMORY_CORRUPTION)
300 if (failBefore)
301 {
302#ifdef CHECK_MEMORY_CORRUPTION_PRINT_ERROR
303 std::cerr << "ERROR: Memory corruption detected: before buffer: " << cv::format("widthInBytes=%d height=%d", (int)data.widthInBytes, (int)data.height) << std::endl;
304#endif
305#ifdef CHECK_MEMORY_CORRUPTION_RAISE_ERROR
306 CV_Error(CV_StsInternal, "Memory corruption detected: before buffer");
307#endif
308 }
309 if (failAfter)
310 {
311#ifdef CHECK_MEMORY_CORRUPTION_PRINT_ERROR
312 std::cerr << "ERROR: Memory corruption detected: after buffer: " << cv::format("widthInBytes=%d height=%d", (int)data.widthInBytes, (int)data.height) << std::endl;
313#endif
314#ifdef CHECK_MEMORY_CORRUPTION_RAISE_ERROR
315 CV_Error(CV_StsInternal, "Memory corruption detected: after buffer");
316#endif
317 }
318#endif // CHECK_MEMORY_CORRUPTION
319#endif // MEMORY_CORRUPTION_GUARD
320}

327cl_kernel openCLGetKernelFromSource(const Context *ctx, const cv::ocl::ProgramEntry* source, string kernelName,
328 const char *build_options)
329{
330 cl_kernel kernel;
331 cl_int status = 0;
332 CV_Assert(ProgramCache::getProgramCache() != NULL);
333 cl_program program = ProgramCache::getProgramCache()->getProgram(ctx, source, build_options);
334 CV_Assert(program != NULL);
335 kernel = clCreateKernel(program, kernelName.c_str(), &status);
336 openCLVerifyCall(status);
337 openCLVerifyCall(clReleaseProgram(program));
338 return kernel;
339}

341void openCLVerifyKernel(const Context *ctx, cl_kernel kernel, size_t *localThreads)
342{
343 size_t kernelWorkGroupSize;
344 openCLSafeCall(clGetKernelWorkGroupInfo(kernel, getClDeviceID(ctx),
345 CL_KERNEL_WORK_GROUP_SIZE, sizeof(size_t), &kernelWorkGroupSize, 0));
346 CV_Assert( localThreads[0] <= ctx->getDeviceInfo().maxWorkItemSizes[0] );
347 CV_Assert( localThreads[1] <= ctx->getDeviceInfo().maxWorkItemSizes[1] );
348 CV_Assert( localThreads[2] <= ctx->getDeviceInfo().maxWorkItemSizes[2] );
349 CV_Assert( localThreads[0] * localThreads[1] * localThreads[2] <= kernelWorkGroupSize );
350 CV_Assert( localThreads[0] * localThreads[1] * localThreads[2] <= ctx->getDeviceInfo().maxWorkGroupSize );
351}

358static std::string removeDuplicatedWhiteSpaces(const char * buildOptions)
359{
360 if (buildOptions == NULL)
361 return "";
362
363 size_t length = strlen(buildOptions), didx = 0, sidx = 0;
364 while (sidx < length && buildOptions[sidx] == 0)
365 ++sidx;
366
367 std::string opt;
368 opt.resize(length);
369
370 for ( ; sidx < length; ++sidx)
371 if (buildOptions[sidx] != ' ')
372 opt[didx++] = buildOptions[sidx];
373 else if ( !(didx > 0 && opt[didx - 1] == ' ') )
374 opt[didx++] = buildOptions[sidx];
375
376 return opt;
377}

379cl_kernel openCLGetKernelFromSource(Context *ctx, const cv::ocl::ProgramEntry* source, string kernelName, int channels,
380 int depth, const char *build_options)
381{
382 //construct kernel name
383 //The rule is functionName_Cn_Dn, C represent Channels, D Represent DataType Depth, n represent an integer number
384 //for example split_C2_D3, represent the split kernel with channels = 2 and dataType Depth = 3(Data type is short)
385 stringstream idxStr;
386 if(channels != -1)
387 idxStr << "_C" << channels;
388 if(depth != -1)
389 idxStr << "_D" << depth;
390 kernelName += idxStr.str();
391
392 std::string fixedOptions = removeDuplicatedWhiteSpaces(build_options);
393 cl_kernel kernel = openCLGetKernelFromSource(ctx, source, kernelName, fixedOptions.c_str());
394 return kernel;
395}

397void openCLExecuteKernel(Context *ctx, cl_kernel kernel, size_t globalThreads[3],
398 size_t localThreads[3], vector< pair<size_t, const void *> > &args)
399{
400 if ( localThreads != NULL)
401 {
402 globalThreads[0] = roundUp(globalThreads[0], localThreads[0]);
403 globalThreads[1] = roundUp(globalThreads[1], localThreads[1]);
404 globalThreads[2] = roundUp(globalThreads[2], localThreads[2]);
405
406 cv::ocl::openCLVerifyKernel(ctx, kernel, localThreads);
407 }
408 for(size_t i = 0; i < args.size(); i ++)
409 openCLSafeCall(clSetKernelArg(kernel, i, args[i].first, args[i].second));
410
411#ifndef PRINT_KERNEL_RUN_TIME
412 openCLSafeCall(clEnqueueNDRangeKernel(getClCommandQueue(ctx), kernel, 3, NULL, globalThreads,
413 localThreads, 0, NULL, NULL));
414#else
415 cl_event event = NULL;
416 openCLSafeCall(clEnqueueNDRangeKernel(getClCommandQueue(ctx), kernel, 3, NULL, globalThreads,
417 localThreads, 0, NULL, &event));
418
419 cl_ulong start_time, end_time, queue_time;
420 double execute_time = 0;
421 double total_time = 0;
422
423 openCLSafeCall(clWaitForEvents(1, &event));
424 openCLSafeCall(clGetEventProfilingInfo(event, CL_PROFILING_COMMAND_START,
425 sizeof(cl_ulong), &start_time, 0));
426
427 openCLSafeCall(clGetEventProfilingInfo(event, CL_PROFILING_COMMAND_END,
428 sizeof(cl_ulong), &end_time, 0));
429
430 openCLSafeCall(clGetEventProfilingInfo(event, CL_PROFILING_COMMAND_QUEUED,
431 sizeof(cl_ulong), &queue_time, 0));
432
433 execute_time = (double)(end_time - start_time) / (1000 * 1000);
434 total_time = (double)(end_time - queue_time) / (1000 * 1000);
435
436 total_execute_time += execute_time;
437 total_kernel_time += total_time;
438 clReleaseEvent(event);
439#endif
440
441 clFlush(getClCommandQueue(ctx));
442 openCLSafeCall(clReleaseKernel(kernel));
443}

495void openCLExecuteKernelInterop(Context *ctx, const cv::ocl::ProgramSource& source, string kernelName,
496 size_t globalThreads[3], size_t localThreads[3],
497 vector< pair<size_t, const void *> > &args, int channels, int depth, const char *build_options)
498
499{
500 //construct kernel name
501 //The rule is functionName_Cn_Dn, C represent Channels, D Represent DataType Depth, n represent an integer number
502 //for example split_C2_D2, represent the split kernel with channels = 2 and dataType Depth = 2 (Data type is char)
503 stringstream idxStr;
504 if(channels != -1)
505 idxStr << "_C" << channels;
506 if(depth != -1)
507 idxStr << "_D" << depth;
508 kernelName += idxStr.str();
509
510 std::string name = std::string("custom_") + source.name;
511 ProgramEntry program = { name.c_str(), source.programStr, source.programHash };
512 cl_kernel kernel = openCLGetKernelFromSource(ctx, &program, kernelName, build_options);
513
514 CV_Assert(globalThreads != NULL);
515 if ( localThreads != NULL)
516 {
517 globalThreads[0] = roundUp(globalThreads[0], localThreads[0]);
518 globalThreads[1] = roundUp(globalThreads[1], localThreads[1]);
519 globalThreads[2] = roundUp(globalThreads[2], localThreads[2]);
520
521 cv::ocl::openCLVerifyKernel(ctx, kernel, localThreads);
522 }
523 for(size_t i = 0; i < args.size(); i ++)
524 openCLSafeCall(clSetKernelArg(kernel, i, args[i].first, args[i].second));
525
526 openCLSafeCall(clEnqueueNDRangeKernel(getClCommandQueue(ctx), kernel, 3, NULL, globalThreads,
527 localThreads, 0, NULL, NULL));
528
529 clFinish(getClCommandQueue(ctx));
530 openCLSafeCall(clReleaseKernel(kernel));
531}

534 const size_t size)
