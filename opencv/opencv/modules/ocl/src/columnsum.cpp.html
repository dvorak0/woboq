46#include "precomp.hpp"
47#include "opencl_kernels.hpp"
48
49using namespace cv;
50using namespace cv::ocl;
51
52void cv::ocl::columnSum(const oclMat &src, oclMat &dst)
53{
54 CV_Assert(src.type() == CV_32FC1);
55 dst.create(src.size(), src.type());
56
57 int src_step = src.step / src.elemSize(), src_offset = src.offset / src.elemSize();
58 int dst_step = dst.step / dst.elemSize(), dst_offset = dst.offset / dst.elemSize();
59
60 std::vector< pair<size_t, const void *> > args;
61 args.push_back( make_pair( sizeof(cl_mem), (void *)&src.data));
62 args.push_back( make_pair( sizeof(cl_mem), (void *)&dst.data));
63 args.push_back( make_pair( sizeof(cl_int), (void *)&src.cols));
64 args.push_back( make_pair( sizeof(cl_int), (void *)&src.rows));
65 args.push_back( make_pair( sizeof(cl_int), (void *)&src_step));
66 args.push_back( make_pair( sizeof(cl_int), (void *)&dst_step));
67 args.push_back( make_pair( sizeof(cl_int), (void *)&src_offset));
68 args.push_back( make_pair( sizeof(cl_int), (void *)&dst_offset));
69
70 size_t globalThreads[3] = {(size_t)dst.cols, 1, 1};
71 size_t localThreads[3] = {256, 1, 1};
72
73 openCLExecuteKernel(src.clCxt, &imgproc_columnsum, "columnSum", globalThreads, localThreads, args, src.oclchannels(), src.depth());
74
75}
