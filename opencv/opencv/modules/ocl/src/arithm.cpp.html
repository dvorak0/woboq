59static std::vector<uchar> scalarToVector(const cv::Scalar & sc, int depth, int ocn, int cn)
60{
61 CV_Assert(ocn == cn || (ocn == 4 && cn == 3));
62
63 static const int sizeMap[] = { sizeof(uchar), sizeof(char), sizeof(ushort),
64 sizeof(short), sizeof(int), sizeof(float), sizeof(double) };
65
66 int elemSize1 = sizeMap[depth];
67 int bufSize = elemSize1 * ocn;
68 std::vector<uchar> _buf(bufSize);
69 uchar * buf = &_buf[0];
70 scalarToRawData(sc, buf, CV_MAKE_TYPE(depth, cn));
71 memset(buf + elemSize1 * cn, 0, (ocn - cn) * elemSize1);
72
73 return _buf;
74}

82static void arithmetic_run_generic(const oclMat &src1, const oclMat &src2, const Scalar & scalar, const oclMat & mask,
83 oclMat &dst, int op_type, bool use_scalar = false)
84{
85 Context *clCxt = src1.clCxt;
86 bool hasDouble = clCxt->supportsFeature(FEATURE_CL_DOUBLE);
87 if (!hasDouble && (src1.depth() == CV_64F || src2.depth() == CV_64F || dst.depth() == CV_64F))
88 {
89 CV_Error(CV_OpenCLDoubleNotSupported, "Selected device doesn't support double");
90 return;
91 }
92
93 CV_Assert(src2.empty() || (!src2.empty() && src1.type() == src2.type() && src1.size() == src2.size()));
94 CV_Assert(mask.empty() || (!mask.empty() && mask.type() == CV_8UC1 && mask.size() == src1.size()));
95 CV_Assert(op_type >= ADD && op_type <= MAX);
96
97 dst.create(src1.size(), src1.type());
98
99 int oclChannels = src1.oclchannels(), depth = src1.depth();
100 int src1step1 = src1.step / src1.elemSize(), src1offset1 = src1.offset / src1.elemSize();
101 int src2step1 = src2.step / src2.elemSize(), src2offset1 = src2.offset / src2.elemSize();
102 int maskstep1 = mask.step, maskoffset1 = mask.offset / mask.elemSize();
103 int dststep1 = dst.step / dst.elemSize(), dstoffset1 = dst.offset / dst.elemSize();
104 std::vector<uchar> m;
105
106#ifdef ANDROID
107 size_t localThreads[3] = { 16, 10, 1 };
108#else
109 size_t localThreads[3] = { 16, 16, 1 };
110#endif
111 size_t globalThreads[3] = { (size_t)dst.cols, (size_t)dst.rows, 1 };
112
113 std::string kernelName = "arithm_binary_op";
114
115 const char * const typeMap[] = { "uchar", "char", "ushort", "short", "int", "float", "double" };
116 const char * const WTypeMap[] = { "short", "short", "int", "int", "int", "float", "double" };
117 const char * const funcMap[] = { "FUNC_ADD", "FUNC_SUB", "FUNC_MUL", "FUNC_DIV", "FUNC_ABS", "FUNC_ABS_DIFF", "FUNC_MIN", "FUNC_MAX" };
118 const char * const channelMap[] = { "", "", "2", "4", "4" };
119 bool haveScalar = use_scalar || src2.empty();
120
121 int WDepth = depth;
122 if (haveScalar)
123 WDepth = hasDouble && WDepth == CV_64F ? CV_64F : CV_32F;
124 if (op_type == DIV)
125 WDepth = hasDouble ? CV_64F : CV_32F;
126 else if (op_type == MUL)
127 WDepth = hasDouble && (depth == CV_32S || depth == CV_64F) ? CV_64F : CV_32F;
128
129 std::string buildOptions = format("-D T=%s%s -D WT=%s%s -D convertToT=convert_%s%s%s -D %s "
130 "-D convertToWT=convert_%s%s",
131 typeMap[depth], channelMap[oclChannels],
132 WTypeMap[WDepth], channelMap[oclChannels],
133 typeMap[depth], channelMap[oclChannels], (depth >= CV_32F ? "" : (depth == CV_32S ? "_rte" : "_sat_rte")),
134 funcMap[op_type], WTypeMap[WDepth], channelMap[oclChannels]);
135
136 vector<pair<size_t , const void *> > args;
137 args.push_back( make_pair( sizeof(cl_mem), (void *)&src1.data ));
138 args.push_back( make_pair( sizeof(cl_int), (void *)&src1step1 ));
139 args.push_back( make_pair( sizeof(cl_int), (void *)&src1offset1 ));
140
141 if (!src2.empty())
142 {
143 args.push_back( make_pair( sizeof(cl_mem), (void *)&src2.data ));
144 args.push_back( make_pair( sizeof(cl_int), (void *)&src2step1 ));
145 args.push_back( make_pair( sizeof(cl_int), (void *)&src2offset1 ));
146
147 kernelName += "_mat";
148
149 if (haveScalar)
150 buildOptions += " -D HAVE_SCALAR";
151 }
152
153 if (haveScalar)
154 {
155 const int WDepthMap[] = { CV_16S, CV_16S, CV_32S, CV_32S, CV_32S, CV_32F, CV_64F };
156 m = scalarToVector(scalar, WDepthMap[WDepth], oclChannels, src1.channels());
157
158 args.push_back( make_pair( m.size(), (void *)&m[0]));
159
160 kernelName += "_scalar";
161 }
162
163 if (!mask.empty())
164 {
165 args.push_back( make_pair( sizeof(cl_mem), (void *)&mask.data ));
166 args.push_back( make_pair( sizeof(cl_int), (void *)&maskstep1 ));
167 args.push_back( make_pair( sizeof(cl_int), (void *)&maskoffset1 ));
168
169 kernelName += "_mask";
170 }
171
172 args.push_back( make_pair( sizeof(cl_mem), (void *)&dst.data ));
173 args.push_back( make_pair( sizeof(cl_int), (void *)&dststep1 ));
174 args.push_back( make_pair( sizeof(cl_int), (void *)&dstoffset1 ));
175
176 args.push_back( make_pair( sizeof(cl_int), (void *)&src1.cols ));
177 args.push_back( make_pair( sizeof(cl_int), (void *)&src1.rows ));
178
179 openCLExecuteKernel(clCxt, mask.empty() ?
180 (!src2.empty() ? &arithm_add : &arithm_add_scalar) :
181 (!src2.empty() ? &arithm_add_mask : &arithm_add_scalar_mask),
182 kernelName, globalThreads, localThreads,
183 args, -1, -1, buildOptions.c_str());
184}

262static void compare_run(const oclMat &src1, const oclMat &src2, oclMat &dst, int cmpOp,
263 string kernelName, const cv::ocl::ProgramEntry* source)
264{
265 dst.create(src1.size(), CV_8UC1);
266
267 int depth = src1.depth();
268 size_t localThreads[3] = { 64, 4, 1 };
269 size_t globalThreads[3] = { (size_t)dst.cols, (size_t)dst.rows, 1 };
270
271 int src1step1 = src1.step1(), src1offset1 = src1.offset / src1.elemSize1();
272 int src2step1 = src2.step1(), src2offset1 = src2.offset / src2.elemSize1();
273 int dststep1 = dst.step1(), dstoffset1 = dst.offset / dst.elemSize1();
274
275 const char * const typeMap[] = { "uchar", "char", "ushort", "short", "int", "float", "double" };
276 const char * operationMap[] = { "==", ">", ">=", "<", "<=", "!=" };
277 std::string buildOptions = format("-D T=%s -D Operation=%s", typeMap[depth], operationMap[cmpOp]);
278
279 vector<pair<size_t , const void *> > args;
280 args.push_back( make_pair( sizeof(cl_mem), (void *)&src1.data ));
281 args.push_back( make_pair( sizeof(cl_int), (void *)&src1step1 ));
282 args.push_back( make_pair( sizeof(cl_int), (void *)&src1offset1 ));
283 args.push_back( make_pair( sizeof(cl_mem), (void *)&src2.data ));
284 args.push_back( make_pair( sizeof(cl_int), (void *)&src2step1 ));
285 args.push_back( make_pair( sizeof(cl_int), (void *)&src2offset1 ));
286 args.push_back( make_pair( sizeof(cl_mem), (void *)&dst.data ));
287 args.push_back( make_pair( sizeof(cl_int), (void *)&dststep1 ));
288 args.push_back( make_pair( sizeof(cl_int), (void *)&dstoffset1 ));
289 args.push_back( make_pair( sizeof(cl_int), (void *)&src1.cols ));
290 args.push_back( make_pair( sizeof(cl_int), (void *)&src1.rows ));
291
292 openCLExecuteKernel(src1.clCxt, source, kernelName, globalThreads, localThreads,
293 args, -1, -1, buildOptions.c_str());
294}

296void cv::ocl::compare(const oclMat &src1, const oclMat &src2, oclMat &dst , int cmpOp)
297{
298 if (!src1.clCxt->supportsFeature(FEATURE_CL_DOUBLE) && src1.depth() == CV_64F)
299 {
300 CV_Error(CV_OpenCLDoubleNotSupported, "Selected device doesn't support double");
301 return;
302 }
303
304 CV_Assert(src1.type() == src2.type() && src1.channels() == 1);
305 CV_Assert(cmpOp >= CMP_EQ && cmpOp <= CMP_NE);
306
307 compare_run(src1, src2, dst, cmpOp, "arithm_compare", &arithm_compare);
308}

316static void arithmetic_sum_buffer_run(const oclMat &src, cl_mem &dst, int groupnum, int type, int ddepth, int vlen)
317{
318 int vElemSize = vlen * src.elemSize();
319 int src_offset = src.offset / vElemSize, src_step = src.step / vElemSize;
320 int src_cols = src.cols / vlen, total = src.size().area() / vlen;
321
322 vlen *= src.oclchannels();
323
324 const char * const typeMap[] = { "uchar", "char", "ushort", "short", "int", "float", "double" };
325 const char * const funcMap[] = { "FUNC_SUM", "FUNC_ABS_SUM", "FUNC_SQR_SUM" };
326 const char * const channelMap[] = { " ", " ", "2", "4", "4", "", "", "", "8" };
327 string buildOptions = format("-D srcT=%s%s -D dstT=%s%s -D convertToDstT=convert_%s%s -D %s",
328 typeMap[src.depth()], channelMap[vlen], typeMap[ddepth],
329 channelMap[vlen], typeMap[ddepth], channelMap[vlen], funcMap[type]);
330
331 vector<pair<size_t , const void *> > args;
332 args.push_back( make_pair( sizeof(cl_mem) , (void *)&src.data));
333 args.push_back( make_pair( sizeof(cl_int) , (void *)&src_step ));
334 args.push_back( make_pair( sizeof(cl_int) , (void *)&src_offset ));
335 args.push_back( make_pair( sizeof(cl_int) , (void *)&src_cols ));
336 args.push_back( make_pair( sizeof(cl_int) , (void *)&total ));
337 args.push_back( make_pair( sizeof(cl_int) , (void *)&groupnum ));
338 args.push_back( make_pair( sizeof(cl_mem) , (void *)&dst ));
339 size_t globalThreads[3] = { (size_t)groupnum * 256, 1, 1 };
340
341#ifdef ANDROID
342 openCLExecuteKernel(src.clCxt, &arithm_sum, "arithm_op_sum", globalThreads, NULL,
343 args, -1, -1, buildOptions.c_str());
344#else
345 size_t localThreads[3] = { 256, 1, 1 };
346 openCLExecuteKernel(src.clCxt, &arithm_sum, "arithm_op_sum", globalThreads, localThreads,
347 args, -1, -1, buildOptions.c_str());
348#endif
349}

351template <typename T>
352Scalar arithmetic_sum(const oclMat &src, int type, int ddepth)
353{
354 CV_Assert(src.step % src.elemSize() == 0);
355
356 size_t groupnum = src.clCxt->getDeviceInfo().maxComputeUnits;
357 CV_Assert(groupnum != 0);
358
359 int vlen = 8 / src.channels(), vElemSize = vlen * src.elemSize1();
360 while (src.offset % vElemSize != 0 || src.step % vElemSize != 0 || src.cols % vlen != 0)
361 vlen >>= 1, vElemSize >>= 1;
362
363 int dbsize = groupnum * src.oclchannels() * vlen;
364 Context *clCxt = src.clCxt;
365
366 AutoBuffer<T> _buf(dbsize);
367 T *p = (T*)_buf;
368 memset(p, 0, dbsize * sizeof(T));
369
370 cl_mem dstBuffer = openCLCreateBuffer(clCxt, CL_MEM_WRITE_ONLY, dbsize * sizeof(T));
371 arithmetic_sum_buffer_run(src, dstBuffer, groupnum, type, ddepth, vlen);
372 openCLReadBuffer(clCxt, dstBuffer, (void *)p, dbsize * sizeof(T));
373 openCLFree(dstBuffer);
374
375 Scalar s = Scalar::all(0.0);
376 for (int i = 0; i < dbsize; )
377 for (int j = 0; j < src.oclchannels(); j++, i++)
378 s.val[j] += p[i];
379
380 return s;
381}

385Scalar cv::ocl::sum(const oclMat &src)
386{
387 if (!src.clCxt->supportsFeature(FEATURE_CL_DOUBLE) && src.depth() == CV_64F)

450
451void cv::ocl::meanStdDev(const oclMat &src, Scalar &mean, Scalar &stddev)
452{
453 if (src.depth() == CV_64F && !src.clCxt->supportsFeature(FEATURE_CL_DOUBLE))
454 {
455 CV_Error(CV_OpenCLDoubleNotSupported, "Selected device doesn't support double");
456 return;
457 }
458
459 double total = 1.0 / src.size().area();
460
461 mean = sum(src);
462 stddev = sqrSum(src);
463
464 for (int i = 0; i < 4; ++i)
465 {
466 mean[i] *= total;
467 stddev[i] = std::sqrt(std::max(stddev[i] * total - mean.val[i] * mean.val[i] , 0.));
468 }
469}

477{
478 const char * const typeMap[] = { "uchar", "char", "ushort", "short", "int", "float", "double" };

480
481 ostringstream stream;
482 stream << "-D T=" << typeMap[src.depth()] << channelMap[vlen];
483 if (numeric_limits<T>::is_integer)
484 {
485 stream << " -D MAX_VAL=" << (WT)numeric_limits<T>::max();
486 stream << " -D MIN_VAL=" << (WT)numeric_limits<T>::min();
487 }
488 else
489 stream << " -D DEPTH_" << src.depth();
490 stream << " -D vlen=" << vlen;
491 std::string buildOptions = stream.str();
492
493 int vElemSize = src.elemSize1() * vlen, src_cols = src.cols / vlen;
494 int src_step = src.step / vElemSize, src_offset = src.offset / vElemSize;
495 int mask_step = mask.step / vlen, mask_offset = mask.offset / vlen;
496 int total = src.size().area() / vlen;
497
498 vector<pair<size_t , const void *> > args;
499 args.push_back( make_pair( sizeof(cl_mem) , (void *)&src.data));
500 args.push_back( make_pair( sizeof(cl_int) , (void *)&src_step ));
501 args.push_back( make_pair( sizeof(cl_int) , (void *)&src_offset ));
502 args.push_back( make_pair( sizeof(cl_int) , (void *)&src.rows ));
503 args.push_back( make_pair( sizeof(cl_int) , (void *)&src_cols ));
504 args.push_back( make_pair( sizeof(cl_int) , (void *)&total));
505 args.push_back( make_pair( sizeof(cl_int) , (void *)&groupnum));
506 args.push_back( make_pair( sizeof(cl_mem) , (void *)&dst));
507
508 if (!mask.empty())
509 {
510 args.push_back( make_pair( sizeof(cl_mem) , (void *)&mask.data ));
511 args.push_back( make_pair( sizeof(cl_int) , (void *)&mask_step ));
512 args.push_back( make_pair( sizeof(cl_int) , (void *)&mask_offset ));
513
514 buildOptions += " -D WITH_MASK";
515 }
516
517 size_t globalThreads[3] = { (size_t)groupnum * 256, 1, 1 };
518 size_t localThreads[3] = { 256, 1, 1 };
519
520 // kernel use fixed grid size, replace lt on NULL is impossible without kernel changes
521 openCLExecuteKernel(src.clCxt, &arithm_minMax, "arithm_op_minMax", globalThreads, localThreads,
522 args, -1, -1, buildOptions.c_str());
523}

525template <typename T, typename WT>
526void arithmetic_minMax(const oclMat &src, double *minVal, double *maxVal, const oclMat &mask)
527{
528 size_t groupnum = src.clCxt->getDeviceInfo().maxComputeUnits;
529 CV_Assert(groupnum != 0);
530
531 int vlen = mask.empty() ? 8 : 1, vElemSize = vlen * src.elemSize1();
532 while (src.offset % vElemSize != 0 || src.step % vElemSize != 0 || src.cols % vlen != 0)
533 {
534 vlen >>= 1;
535 vElemSize >>= 1;
536 }
537
538 int dbsize = groupnum * 2 * vElemSize;
539 oclMat buf;
540 ensureSizeIsEnough(1, dbsize, CV_8UC1, buf);
541
542 cl_mem buf_data = reinterpret_cast<cl_mem>(buf.data);
543 arithmetic_minMax_run<T, WT>(src, mask, buf_data, vlen, groupnum);
544
545 Mat matbuf = Mat(buf);
546 T *p = matbuf.ptr<T>();
547
548 if (minVal != NULL)
549 {
550 *minVal = std::numeric_limits<double>::max();
551 for (int i = 0, end = vlen * (int)groupnum; i < end; i++)
552 *minVal = *minVal < p[i] ? *minVal : p[i];
553 }
554 if (maxVal != NULL)
555 {
556 *maxVal = -std::numeric_limits<double>::max();
557 for (int i = vlen * (int)groupnum, end = i << 1; i < end; i++)
558 *maxVal = *maxVal > p[i] ? *maxVal : p[i];
559 }
560}

678double cv::ocl::norm(const oclMat &src1, const oclMat &src2, int normType)
679{
680 if (!src1.clCxt->supportsFeature(FEATURE_CL_DOUBLE) && src1.depth() == CV_64F)
681 {
682 CV_Error(CV_OpenCLDoubleNotSupported, "Selected device doesn't support double");
683 return -1;
684 }
685 CV_Assert(src2.empty() || (src1.type() == src2.type() && src1.size() == src2.size()));
686
687 bool isRelative = (normType & NORM_RELATIVE) != 0;
688 normType &= NORM_TYPE_MASK;
689 CV_Assert(normType == NORM_INF || normType == NORM_L1 || normType == NORM_L2);
690
691 Scalar s;
692 int cn = src1.channels();
693 double r = 0;
694 oclMat diff;
695
696 arithm_absdiff_nonsaturate_run(src1, src2, diff, normType);
697
698 switch (normType)
699 {
700 case NORM_INF:
701 diff = diff.reshape(1);
702 minMax(diff, NULL, &r);
703 break;
704 case NORM_L1:
705 s = sum(diff);
706 for (int i = 0; i < cn; ++i)
707 r += s[i];
708 break;
709 case NORM_L2:
710 s = sqrSum(diff);
711 for (int i = 0; i < cn; ++i)
712 r += s[i];
713 r = std::sqrt(r);
714 break;
715 }
716 if (isRelative)
717 r = r / (norm(src2, normType) + DBL_EPSILON);
718
719 return r;
720}

1179void arithmetic_minMaxLoc(const oclMat &src, double *minVal, double *maxVal,
1180 Point *minLoc, Point *maxLoc, const oclMat &mask)

1184 CV_Assert(groupnum != 0);
1185 int minloc = -1 , maxloc = -1;

1189 *minVal = std::numeric_limits<double>::max() , *maxVal = -std::numeric_limits<double>::max();
1190
1191 if (mask.empty())
1192 arithmetic_minMaxLoc_run(src, dstBuffer, vlen, groupnum);
1193 else
1194 arithmetic_minMaxLoc_mask_run(src, mask, dstBuffer, vlen, groupnum);

1198 memset(p, 0, dbsize);
1199
1200 openCLReadBuffer(clCxt, dstBuffer, (void *)p, dbsize);
1201 for (int i = 0; i < vlen * (int)groupnum; i++)
1202 {
1203 *minVal = (*minVal < p[i] || p[i + 2 * vlen * groupnum] == -1) ? *minVal : p[i];
1204 minloc = (*minVal < p[i] || p[i + 2 * vlen * groupnum] == -1) ? minloc : cvRound(p[i + 2 * vlen * groupnum]);
1205 }
1206 for (int i = vlen * (int)groupnum; i < 2 * vlen * (int)groupnum; i++)
1207 {
1208 *maxVal = (*maxVal > p[i] || p[i + 2 * vlen * groupnum] == -1) ? *maxVal : p[i];
1209 maxloc = (*maxVal > p[i] || p[i + 2 * vlen * groupnum] == -1) ? maxloc : cvRound(p[i + 2 * vlen * groupnum]);
1210 }

1213 int pre_cols = (src.offset % src.step) / src.elemSize1();
1214 int wholecols = src.step / src.elemSize1();
1215 if ( minLoc )
1216 {
1217 if ( minloc >= 0 )
1218 {
1219 minLoc->y = minloc / wholecols - pre_rows;
1220 minLoc->x = minloc % wholecols - pre_cols;
1221 }
1222 else
1223 minLoc->x = minLoc->y = -1;
1224 }
1225 if ( maxLoc )
1226 {
1227 if ( maxloc >= 0 )
1228 {
1229 maxLoc->y = maxloc / wholecols - pre_rows;
1230 maxLoc->x = maxloc % wholecols - pre_cols;
1231 }
1232 else
1233 maxLoc->x = maxLoc->y = -1;
1234 }
1235
1236 openCLSafeCall(clReleaseMemObject(dstBuffer));
1237}

1355 const char * const channelMap[] = { "", "", "2", "4", "4", "", "", "", "8", "", "", "", "", "", "", "", "16" };

1587void cv::ocl::addWeighted(const oclMat &src1, double alpha, const oclMat &src2, double beta, double gama, oclMat &dst)
1588{
1589 Context *clCxt = src1.clCxt;
1590 bool hasDouble = clCxt->supportsFeature(FEATURE_CL_DOUBLE);
1591 if (!hasDouble && src1.depth() == CV_64F)
1592 {
1593 CV_Error(CV_OpenCLDoubleNotSupported, "Selected device doesn't support double");
1594 return;
1595 }
1596
1597 CV_Assert(src1.size() == src2.size() && src1.type() == src2.type());
1598 dst.create(src1.size(), src1.type());
1599
1600 int channels = dst.oclchannels();
1601 int depth = dst.depth();
1602
1603 int cols1 = src1.cols * channels;
1604 int src1step1 = src1.step1(), src1offset1 = src1.offset / src1.elemSize1();
1605 int src2step1 = src2.step1(), src2offset1 = src2.offset / src1.elemSize1();
1606 int dststep1 = dst.step1(), dstoffset1 = dst.offset / dst.elemSize1();
1607
1608 const char * const typeMap[] = { "uchar", "char", "ushort", "short", "int", "float", "double" };
1609 std::string buildOptions = format("-D T=%s -D WT=%s -D convertToT=convert_%s%s",
1610 typeMap[depth], hasDouble ? "double" : "float", typeMap[depth],
1611 depth >= CV_32F ? "" : "_sat_rte");
1612
1613 size_t globalThreads[3] = { (size_t)cols1, (size_t)dst.rows, 1};
1614
1615 float alpha_f = static_cast<float>(alpha),
1616 beta_f = static_cast<float>(beta),
1617 gama_f = static_cast<float>(gama);
1618
1619 vector<pair<size_t , const void *> > args;
1620 args.push_back( make_pair( sizeof(cl_mem), (void *)&src1.data ));
1621 args.push_back( make_pair( sizeof(cl_int), (void *)&src1step1 ));
1622 args.push_back( make_pair( sizeof(cl_int), (void *)&src1offset1));
1623 args.push_back( make_pair( sizeof(cl_mem), (void *)&src2.data ));
1624 args.push_back( make_pair( sizeof(cl_int), (void *)&src2step1 ));
1625 args.push_back( make_pair( sizeof(cl_int), (void *)&src2offset1));
1626 args.push_back( make_pair( sizeof(cl_mem), (void *)&dst.data ));
1627 args.push_back( make_pair( sizeof(cl_int), (void *)&dststep1 ));
1628 args.push_back( make_pair( sizeof(cl_int), (void *)&dstoffset1));
1629
1630 if (!hasDouble)
1631 {
1632 args.push_back( make_pair( sizeof(cl_float), (void *)&alpha_f ));
1633 args.push_back( make_pair( sizeof(cl_float), (void *)&beta_f ));
1634 args.push_back( make_pair( sizeof(cl_float), (void *)&gama_f ));
1635 }
1636 else
1637 {
1638 args.push_back( make_pair( sizeof(cl_double), (void *)&alpha ));
1639 args.push_back( make_pair( sizeof(cl_double), (void *)&beta ));
1640 args.push_back( make_pair( sizeof(cl_double), (void *)&gama ));
1641 }
1642
1643 args.push_back( make_pair( sizeof(cl_int), (void *)&cols1 ));
1644 args.push_back( make_pair( sizeof(cl_int), (void *)&src1.rows ));
1645

1648 args, -1, -1, buildOptions.c_str());
