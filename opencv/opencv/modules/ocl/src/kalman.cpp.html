60void KalmanFilter::init(int DP, int MP, int CP, int type)
61{
62 CV_Assert( DP > 0 && MP > 0 );
63 CV_Assert( type == CV_32F || type == CV_64F );
64 CP = cv::max(CP, 0);
65
66 statePre.create(DP, 1, type);
67 statePre.setTo(Scalar::all(0));
68
69 statePost.create(DP, 1, type);
70 statePost.setTo(Scalar::all(0));
71
72 transitionMatrix.create(DP, DP, type);
73 setIdentity(transitionMatrix, 1);
74
75 processNoiseCov.create(DP, DP, type);
76 setIdentity(processNoiseCov, 1);
77
78 measurementNoiseCov.create(MP, MP, type);
79 setIdentity(measurementNoiseCov, 1);
80
81 measurementMatrix.create(MP, DP, type);
82 measurementMatrix.setTo(Scalar::all(0));
83
84 errorCovPre.create(DP, DP, type);
85 errorCovPre.setTo(Scalar::all(0));
86
87 errorCovPost.create(DP, DP, type);
88 errorCovPost.setTo(Scalar::all(0));
89
90 gain.create(DP, MP, type);
91 gain.setTo(Scalar::all(0));
92
93 if( CP > 0 )
94 {
95 controlMatrix.create(DP, CP, type);
96 controlMatrix.setTo(Scalar::all(0));
97 }
98 else
99 controlMatrix.release();
100
101 temp1.create(DP, DP, type);
102 temp2.create(MP, DP, type);
103 temp3.create(MP, MP, type);
104 temp4.create(MP, DP, type);
105 temp5.create(MP, 1, type);
106}
107
108CV_EXPORTS const oclMat& KalmanFilter::predict(const oclMat& control)
109{
110 gemm(transitionMatrix, statePost, 1, oclMat(), 0, statePre);
111 oclMat temp;
112
113 if(control.data)
114 gemm(controlMatrix, control, 1, statePre, 1, statePre);
115 gemm(transitionMatrix, errorCovPost, 1, oclMat(), 0, temp1);
116 gemm(temp1, transitionMatrix, 1, processNoiseCov, 1, errorCovPre, GEMM_2_T);
117 statePre.copyTo(statePost);
118 return statePre;
119}
120
121CV_EXPORTS const oclMat& KalmanFilter::correct(const oclMat& measurement)
122{
123 CV_Assert(measurement.empty() == false);
124 gemm(measurementMatrix, errorCovPre, 1, oclMat(), 0, temp2);
125 gemm(temp2, measurementMatrix, 1, measurementNoiseCov, 1, temp3, GEMM_2_T);
126 Mat temp;
127 solve(Mat(temp3), Mat(temp2), temp, DECOMP_SVD);
128 temp4.upload(temp);
129 gain = temp4.t();
130 gemm(measurementMatrix, statePre, -1, measurement, 1, temp5);
131 gemm(gain, temp5, 1, statePre, 1, statePost);
132 gemm(gain, temp2, -1, errorCovPre, 1, errorCovPost);
133 return statePost;
134}
