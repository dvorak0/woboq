105void cvPreparePredictData( const CvArr* sample, int dims_all, const CvMat* comp_idx,
106 int class_count, const CvMat* prob, float** row_sample,
107 int as_sparse CV_DEFAULT(0) );
108
109void cvPreparePredictData( const CvArr* _sample, int dims_all,
110 const CvMat* comp_idx, int class_count,
111 const CvMat* prob, float** _row_sample,
112 int as_sparse )
113{
114 float* row_sample = 0;
115 int* inverse_comp_idx = 0;
116
117 CV_FUNCNAME( "cvPreparePredictData" );
118
119 __BEGIN__;
120
121 const CvMat* sample = (const CvMat*)_sample;
122 float* sample_data;
123 int sample_step;
124 int is_sparse = CV_IS_SPARSE_MAT(sample);
125 int d, sizes[CV_MAX_DIM];
126 int i, dims_selected;
127 int vec_size;
128
129 if( !is_sparse && !CV_IS_MAT(sample) )
130 {
131 CV_ERROR( !sample ? CV_StsNullPtr : CV_StsBadArg, "The sample is not a valid vector" );
132 }
133
134 if( cvGetElemType( sample ) != CV_32FC1 )
135 {
136 CV_ERROR( CV_StsUnsupportedFormat, "Input sample must have 32fC1 type" );
137 }
138
139 CV_CALL( d = cvGetDims( sample, sizes ));
140
141 if( !((is_sparse && d == 1) || (!is_sparse && d == 2 && (sample->rows == 1 || sample->cols == 1))) )
142 {
143 CV_ERROR( CV_StsBadSize, "Input sample must be 1-dimensional vector" );
144 }
145
146 if( d == 1 )
147 sizes[1] = 1;
148
149 if( sizes[0] + sizes[1] - 1 != dims_all )
150 CV_ERROR( CV_StsUnmatchedSizes,
151 "The sample size is different from what has been used for training" );
152
153 if( !_row_sample )
154 {
155 CV_ERROR( CV_StsNullPtr, "INTERNAL ERROR: The row_sample pointer is NULL" );
156 }
157
158 if( comp_idx && (!CV_IS_MAT(comp_idx) || comp_idx->rows != 1 ||
159 CV_MAT_TYPE(comp_idx->type) != CV_32SC1) )
160 {
161 CV_ERROR( CV_StsBadArg, "INTERNAL ERROR: invalid comp_idx" );
162 }
163
164 dims_selected = comp_idx ? comp_idx->cols : dims_all;
165
166 if( prob )
167 {
168 if( !CV_IS_MAT(prob) )
169 {
170 CV_ERROR( CV_StsBadArg, "The output matrix of probabilities is invalid" );
171 }
172
173 if( (prob->rows != 1 && prob->cols != 1) ||
174 (CV_MAT_TYPE(prob->type) != CV_32FC1 &&
175 CV_MAT_TYPE(prob->type) != CV_64FC1) )
176 CV_ERROR( CV_StsBadSize,
177 "The matrix of probabilities must be 1-dimensional vector of 32fC1 type" );
178
179 if( prob->rows + prob->cols - 1 != class_count )
180 CV_ERROR( CV_StsUnmatchedSizes,
181 "The vector of probabilities must contain as many elements as "
182 "the number of classes in the training set" );
183 }
184
185 vec_size = !as_sparse ? dims_selected * sizeof(row_sample[0]) :
186 (dims_selected + 1) * sizeof(CvSparseVecElem32f);
187
188 if( CV_IS_MAT(sample) )
189 {
190 sample_data = sample->data.fl;
191 sample_step = CV_IS_MAT_CONT(sample->type) ? 1 : sample->step / sizeof(row_sample[0]);
192
193 if( !comp_idx && CV_IS_MAT_CONT(sample->type) && !as_sparse )
194 *_row_sample = sample_data;
195 else
196 {
197 CV_CALL( row_sample = (float*)cvAlloc( vec_size ));
198
199 if( !comp_idx )
200 for( i = 0; i < dims_selected; i++ )
201 row_sample[i] = sample_data[sample_step * i];
202 else
203 {
204 int* comp = comp_idx->data.i;
205 for( i = 0; i < dims_selected; i++ )
206 row_sample[i] = sample_data[sample_step * comp[i]];
207 }
208
209 *_row_sample = row_sample;
210 }
211
212 if( as_sparse )
213 {
214 const float* src = (const float*)row_sample;
215 CvSparseVecElem32f* dst = (CvSparseVecElem32f*)row_sample;
216
217 dst[dims_selected].idx = -1;
218 for( i = dims_selected - 1; i >= 0; i-- )
219 {
220 dst[i].idx = i;
221 dst[i].val = src[i];
222 }
223 }
224 }
225 else
226 {
227 CvSparseNode* node;
228 CvSparseMatIterator mat_iterator;
229 const CvSparseMat* sparse = (const CvSparseMat*)sample;
230 assert( is_sparse );
231
232 node = cvInitSparseMatIterator( sparse, &mat_iterator );
233 CV_CALL( row_sample = (float*)cvAlloc( vec_size ));
234
235 if( comp_idx )
236 {
237 CV_CALL( inverse_comp_idx = (int*)cvAlloc( dims_all * sizeof(int) ));
238 memset( inverse_comp_idx, -1, dims_all * sizeof(int) );
239 for( i = 0; i < dims_selected; i++ )
240 inverse_comp_idx[comp_idx->data.i[i]] = i;
241 }
242
243 if( !as_sparse )
244 {
245 memset( row_sample, 0, vec_size );
246
247 for( ; node != 0; node = cvGetNextSparseNode(&mat_iterator) )
248 {
249 int idx = *CV_NODE_IDX( sparse, node );
250 if( inverse_comp_idx )
251 {
252 idx = inverse_comp_idx[idx];
253 if( idx < 0 )
254 continue;
255 }
256 row_sample[idx] = *(float*)CV_NODE_VAL( sparse, node );
257 }
258 }
259 else
260 {
261 CvSparseVecElem32f* ptr = (CvSparseVecElem32f*)row_sample;
262
263 for( ; node != 0; node = cvGetNextSparseNode(&mat_iterator) )
264 {
265 int idx = *CV_NODE_IDX( sparse, node );
266 if( inverse_comp_idx )
267 {
268 idx = inverse_comp_idx[idx];
269 if( idx < 0 )
270 continue;
271 }
272 ptr->idx = idx;
273 ptr->val = *(float*)CV_NODE_VAL( sparse, node );
274 ptr++;
275 }
276
277 qsort( row_sample, ptr - (CvSparseVecElem32f*)row_sample,
278 sizeof(ptr[0]), icvCmpSparseVecElems );
279 ptr->idx = -1;
280 }
281
282 *_row_sample = row_sample;
283 }
284
285 __END__;
286
287 if( inverse_comp_idx )
288 cvFree( &inverse_comp_idx );
289
290 if( cvGetErrStatus() < 0 && _row_sample )
291 {
292 cvFree( &row_sample );
293 *_row_sample = 0;
294 }
295}

297float CvSVM_OCL::predict( const int row_index, int row_len, Mat& src, bool returnDFVal ) const
298{
299 assert( kernel );
300
301 (void)row_len;
302
303 int class_count = class_labels ? class_labels->cols :
304 params.svm_type == ONE_CLASS ? 1 : 0;
305
306 float result = 0;
307 cv::AutoBuffer<float> _buffer(sv_total + (class_count + 1) * 2);
308 float* buffer = _buffer;
309
310 if( params.svm_type == EPS_SVR ||
311 params.svm_type == NU_SVR ||
312 params.svm_type == ONE_CLASS )
313 {
314 CvSVMDecisionFunc* df = (CvSVMDecisionFunc*)decision_func;
315 int i, sv_count = df->sv_count;
316 double sum = -df->rho;
317
318 ((CvSVMKernel_ocl*)kernel)->calc( sv_count, row_index, buffer, src);
319 for( i = 0; i < sv_count; i++ )
320 sum += buffer[i] * df->alpha[i];
321
322 result = params.svm_type == ONE_CLASS ? (float)(sum > 0) : (float)sum;
323 }
324 else if( params.svm_type == C_SVC ||
325 params.svm_type == NU_SVC )
326 {
327 CvSVMDecisionFunc* df = (CvSVMDecisionFunc*)decision_func;
328 int* vote = (int*)(buffer + sv_total);
329 int i, j, k;
330
331 memset( vote, 0, class_count * sizeof(vote[0]));
332 ((CvSVMKernel_ocl*)kernel)->calc( sv_total, row_index, buffer, src);
333 double sum = 0.;
334
335 for( i = 0; i < class_count; i++ )
336 for( j = i + 1; j < class_count; j++, df++ )
337 {
338 sum = -df->rho;
339 int sv_count = df->sv_count;
340 for( k = 0; k < sv_count; k++ )
341 sum += df->alpha[k] * buffer[df->sv_index[k]];
342
343 vote[sum > 0 ? i : j]++;
344 }
345
346 for( i = 1, k = 0; i < class_count; i++ )
347 if( vote[i] > vote[k] )
348 k = i;
349
350 result = returnDFVal && class_count == 2 ? (float)sum : (float)(class_labels->data.i[k]);
351 }
352 else
353 CV_Error( CV_StsBadArg, "INTERNAL ERROR: Unknown SVM type, "
354 "the SVM structure is probably corrupted" );
355
356 return result;
357}

396float* CvSVMSolver_ocl::get_row_base( int i, bool* _existed, Mat& src )
397{
398 int i1 = i < sample_count ? i : i - sample_count;
399 CvSVMKernelRow* row = rows + i1;
400 bool existed = row->data != 0;
401 Qfloat* data;
402
403 if( existed || cache_size <= 0 )
404 {
405 CvSVMKernelRow* del_row = existed ? row : lru_list.prev;
406 data = del_row->data;
407 assert( data != 0 );
408
409 // delete row from the LRU list
410 del_row->data = 0;
411 del_row->prev->next = del_row->next;
412 del_row->next->prev = del_row->prev;
413 }
414 else
415 {
416 data = (Qfloat*)cvMemStorageAlloc( storage, cache_line_size );
417 cache_size -= cache_line_size;
418 }
419
420 // insert row into the LRU list
421 row->data = data;
422 row->prev = &lru_list;
423 row->next = lru_list.next;
424 row->prev->next = row->next->prev = row;
425
426 if( !existed )
427 ((CvSVMKernel_ocl*)kernel)->calc( sample_count, i1, row->data, src);
428
429 if( _existed )
430 *_existed = existed;
431
432 return row->data;
433}

437static void matmul_sigmod(oclMat & src, oclMat & src2, oclMat & dst, int src_rows, int src2_cols, int var_count, double alpha1, double beta1)
438{
439 Context *clCxt = Context::getContext();
440 string kernelName = "svm_sigmod";
441 int src_step = (int)src.step / src.elemSize();
442 int src2_step = (int)src2.step / src2.elemSize();
443 int dst_step = (int)dst.step / dst.elemSize();
444 int x = MIN(16, src_rows);
445 int y = MIN(16, src2_cols);
446 size_t localThreads[] = {(size_t)x, (size_t)y, 1};
447 size_t globalThreads[] = {(size_t)src2_cols, (size_t)src_rows, 1};
448 int width = var_count;
449
450 vector< pair<size_t, const void *> > args;
451 args.push_back(make_pair(sizeof(cl_mem), (void* )&src.data));
452 args.push_back(make_pair(sizeof(cl_int), (void* )&src_step));
453 args.push_back(make_pair(sizeof(cl_mem), (void* )&src2.data));
454 args.push_back(make_pair(sizeof(cl_int), (void* )&src2_step));
455 args.push_back(make_pair(sizeof(cl_mem), (void* )&dst.data));
456 args.push_back(make_pair(sizeof(cl_int), (void* )&dst_step));
457 args.push_back(make_pair(sizeof(cl_int), (void* )&src_rows));
458 args.push_back(make_pair(sizeof(cl_int), (void* )&src2_cols));
459 args.push_back(make_pair(sizeof(cl_int), (void* )&width));
460
461 float alpha = 0.0f, beta = 0.0f;
462 if(!Context::getContext()->supportsFeature(FEATURE_CL_DOUBLE))
463 {
464 alpha = (float)alpha1;
465 beta = (float)beta1;
466 args.push_back(make_pair(sizeof(cl_float), (void* )&alpha));
467 args.push_back(make_pair(sizeof(cl_float), (void* )&beta));
468 }
469 else
470 {
471 args.push_back(make_pair(sizeof(cl_double), (void* )&alpha1));
472 args.push_back(make_pair(sizeof(cl_double), (void* )&beta1));
473 }
474 openCLExecuteKernel(clCxt, &svm, kernelName, globalThreads, localThreads, args, -1, -1);
475}

568static void matmul_rbf(oclMat& src, oclMat& src_e, oclMat& dst, int src_rows, int src2_cols, int var_count, double gamma1, bool flag)
569{
570
571 Context *clCxt = Context::getContext();
572
573 string kernelName = "svm_rbf";
574
575 int width = var_count;
576 int src_step = (int)src.step / src.elemSize();

596 args.push_back(make_pair(sizeof(cl_int), (void* )&src_rows));
597 args.push_back(make_pair(sizeof(cl_int), (void* )&src2_cols));
598 args.push_back(make_pair(sizeof(cl_int), (void* )&width));
599 float gamma = 0.0f;
600 if(!Context::getContext()->supportsFeature(FEATURE_CL_DOUBLE))
601 {
602 gamma = (float)gamma1;
603 args.push_back(make_pair(sizeof(cl_float), (void* )&gamma));
604 }
605 else
606 args.push_back(make_pair(sizeof(cl_double), (void* )&gamma1));
607
608 openCLExecuteKernel(clCxt, &svm, kernelName, globalThreads, localThreads, args, -1, -1, build_options);
609}

611float CvSVM_OCL::predict(const CvMat* samples, CV_OUT CvMat* results) const
612{
613 int var_count = get_var_count();
614 int sample_count = samples->rows;
615
616 //float* row_sample = 0;
617 Mat src_temp = Mat(sample_count, var_count, CV_32FC1);
618 CV_FUNCNAME( "CvSVM::predict" );
619
620
621 for(int i = 0; i < samples->rows; i++)
622 {
623 __BEGIN__;
624 CvMat sample;
625 float* row_sample = 0;
626 cvGetRow( samples, &sample, i );
627 int class_count;
628 if( !kernel )
629 {
630 CV_ERROR( CV_StsBadArg, "The SVM should be trained first" );
631 }
632
633 class_count = class_labels ? class_labels->cols :
634 params.svm_type == ONE_CLASS ? 1 : 0;
635
636 CV_CALL( cvPreparePredictData(&sample, var_all, var_idx,
637 class_count, 0, &row_sample ));
638 for(int j = 0; j < var_count; ++j)
639 src_temp.at<float>(i, j) = row_sample[j];
640 __END__;
641 }
642
643 Mat dst1;
644 double alpha1 = 0.0, beta1 = 0.0, gamma1 = 0.0;
645 if(params.kernel_type == CvSVM::LINEAR)
646 {
647 alpha1 = 1;
648 beta1 = 0;
649 }
650 if(params.kernel_type == CvSVM::POLY)
651 {
652 alpha1 = params.gamma;
653 beta1 = params.coef0;
654 }
655 if(params.kernel_type == CvSVM::SIGMOID)
656 {
657 alpha1 = - 2 * params.gamma;
658 beta1 = - 2 * params.coef0;
659 }
660 if(params.kernel_type == CvSVM::RBF)
661 gamma1 = - params.gamma;
662
663 Mat sv_temp = Mat(sv_total, var_count, CV_32FC1, Scalar::all(0));
664
665
666 for(int i = 0; i < sv_total; ++i)
667 for(int j = 0; j < var_count; ++j)
668 sv_temp.at<float>(i, j) = sv[i][j];
669
670 oclMat src(sample_count, var_count, CV_32FC1, Scalar::all(0));
671 oclMat sv_;
672
673 src.upload(src_temp);
674 oclMat dst;
675
676#ifdef HAVE_CLAMDBLAS
677
678 dst = oclMat(sample_count, sv_total, CV_32FC1);
679 oclMat src3(sample_count, sv_total, CV_32FC1, Scalar::all(1));
680 if(params.kernel_type != CvSVM::RBF)
681 {
682 Mat sv_temp1;
683 transpose(sv_temp, sv_temp1);
684 sv_.upload(sv_temp1);
685 gemm(src, sv_, alpha1, src3, beta1, dst);
686 }
687
688#else
689 double degree1 = 0.0;
690 if (params.kernel_type == CvSVM::POLY)
691 degree1 = params.degree;
692
693 if(!Context::getContext()->supportsFeature(FEATURE_CL_DOUBLE))
694 dst = oclMat(sample_count, sv_total, CV_32FC1);
695 else
696 dst = oclMat(sample_count, sv_total, CV_64FC1);
697
698 if(params.kernel_type == CvSVM::LINEAR)
699 {
700 sv_.upload(sv_temp);
701 matmul_linear(src, sv_, dst, sample_count, sv_total, var_count, alpha1, beta1);
702 }
703 if( params.kernel_type == CvSVM::SIGMOID)
704 {
705 sv_.upload(sv_temp);
706 matmul_sigmod(src, sv_, dst, sample_count, sv_total, var_count, alpha1, beta1);
707 }
708
709 if(params.kernel_type == CvSVM::POLY)
710 {
711 sv_.upload(sv_temp);
712 if(sample_count > 0)
713 matmul_poly(src, sv_, dst, sample_count, sv_total, var_count, alpha1, beta1, degree1, true);
714 else
715 matmul_poly(src, sv_, dst, sample_count, sv_total, var_count, alpha1, beta1, degree1, false);
716 }
717#endif
718
719 if(params.kernel_type == CvSVM::RBF)
720 {
721 sv_.upload(sv_temp);
722 if(!Context::getContext()->supportsFeature(FEATURE_CL_DOUBLE))
723 dst = oclMat(sample_count, sv_total, CV_32FC1);
724 else
725 dst = oclMat(sample_count, sv_total, CV_64FC1);
726
727 if(sample_count > 0)
728 matmul_rbf(src, sv_, dst, sample_count, sv_total, var_count, gamma1, true);
729 else
730 matmul_rbf(src, sv_, dst, sample_count, sv_total, var_count, gamma1, false);
731 }
732 dst.download(dst1);
733
734 float result = 0;
735 for(int i = 0; i < samples->rows; i++ )
736 {
737 int r = (int)this->predict(i, dst1);
738 if (results)
739 results->data.fl[i] = (float)r;
740 if (i == 0)
741 result = (float)r;
742 }
743 return result;
744}

753bool CvSVMSolver_ocl::solve_generic( CvSVMSolutionInfo& si )
754{
755 int iter = 0;
756 int i, j, k;
757
758 // 1. initialize gradient and alpha status
759 for( i = 0; i < alpha_count; i++ )
760 {
761 update_alpha_status(i);
762 G[i] = b[i];
763 if( fabs(G[i]) > 1e200 )
764 {
765 return false;
766 }
767 }
768 Mat dst1;
769 double alpha1 = 0.0, beta1 = 0.0, gamma1 = 0.0;
770 if(params->kernel_type == CvSVM::LINEAR)
771 {
772 alpha1 = 1;
773 beta1 = 0;
774 }
775 if(params->kernel_type == CvSVM::POLY)
776 {
777 alpha1 = params->gamma;
778 beta1 = params->coef0;
779 }
780 if(params->kernel_type == CvSVM::SIGMOID)
781 {
782 alpha1 = -2 * params->gamma;
783 beta1 = -2 * params->coef0;
784 }
785 if(params->kernel_type == CvSVM::RBF)
786 {
787 gamma1 = -params->gamma;
788 }
789 Mat src1 = Mat(sample_count, var_count, CV_32FC1);
790
791 for(int i2 = 0; i2 < sample_count; ++i2)
792 {
793 for(int j2 = 0; j2 < var_count; ++j2)
794 {
795 src1.at<float>(i2, j2) = samples[i2][j2];
796 }
797 }
798 oclMat src, src_e;
799 src.upload(src1);
800 oclMat dst;
801
802#ifdef HAVE_CLAMDBLAS
803
804 dst = oclMat(sample_count, sample_count, CV_32FC1);
805 oclMat src3(sample_count, sample_count, CV_32FC1, Scalar::all(1));
806 if(params->kernel_type != CvSVM::RBF)
807 {
808 ocl::transpose(src, src_e);
809 gemm(src, src_e, alpha1, src3, beta1, dst);
810 }
811
812#else
813 double degree1 = 0.0;
814 if(params->kernel_type == CvSVM::POLY)
815 degree1 = params->degree;
816
817 if(!Context::getContext()->supportsFeature(FEATURE_CL_DOUBLE))
818 dst = oclMat(sample_count, sample_count, CV_32FC1);
819 else
820 dst = oclMat(sample_count, sample_count, CV_64FC1);
821
822 if(params->kernel_type == CvSVM::LINEAR )
823 {
824 src_e = src;
825 matmul_linear(src, src_e, dst, sample_count, sample_count, var_count, alpha1, beta1);
826 }
827 if( params->kernel_type == CvSVM::SIGMOID)
828 {
829 src_e = src;
830 matmul_sigmod(src, src_e, dst, sample_count, sample_count, var_count, alpha1, beta1);
831 }
832
833 if(params->kernel_type == CvSVM::POLY)
834 {
835 src_e = src;
836 if(sample_count > 0)
837 matmul_poly(src, src_e, dst, sample_count, sample_count, var_count, alpha1, beta1, degree1, true);
838 else
839 matmul_poly(src, src_e, dst, sample_count, sample_count, var_count, alpha1, beta1, degree1, false);
840 }
841
842#endif
843
844 if(params->kernel_type == CvSVM::RBF)
845 {
846 src_e = src;
847 if(!Context::getContext()->supportsFeature(FEATURE_CL_DOUBLE))
848 dst = oclMat(sample_count, sample_count, CV_32FC1);
849 else
850 dst = oclMat(sample_count, sample_count, CV_64FC1);
851
852 if(sample_count > 0)
853 matmul_rbf(src, src_e, dst, sample_count, sample_count, var_count, gamma1, true);
854 else
855 matmul_rbf(src, src_e, dst, sample_count, sample_count, var_count, gamma1, false);
856 }
857 dst.download(dst1);
858 for( i = 0; i < alpha_count; i++ )
859 {
860 if( !is_lower_bound(i) )
861 {
862 const Qfloat *Q_i = CvSVMSolver::get_row( i, buf[0]);
863 double alpha_i = alpha[i];
864
865 for( j = 0; j < alpha_count; j++ )
866 G[j] += alpha_i * Q_i[j];
867 }
868 }
869
870 // 2. optimization loop
871 for(;;)
872 {
873 const Qfloat *Q_i, *Q_j;
874 double C_i, C_j;
875 double old_alpha_i, old_alpha_j, alpha_i, alpha_j;
876 double delta_alpha_i, delta_alpha_j;
877
878#ifdef _DEBUG
879 for( i = 0; i < alpha_count; i++ )
880 {
881 if( fabs(G[i]) > 1e+300 )
882 return false;
883
884 if( fabs(alpha[i]) > 1e16 )
885 return false;
886 }
887#endif
888
889 if( (this->*select_working_set_func)( i, j ) != 0 || iter++ >= max_iter )
890 {
891 break;
892 }
893 Q_i = get_row( i, buf[0], dst1);
894 Q_j = get_row( j, buf[1], dst1);
895
896 C_i = get_C(i);
897 C_j = get_C(j);
898
899 alpha_i = old_alpha_i = alpha[i];
900 alpha_j = old_alpha_j = alpha[j];
901
902 if( y[i] != y[j] )
903 {
904 double denom = Q_i[i] + Q_j[j] + 2 * Q_i[j];
905 double delta = (-G[i] - G[j]) / MAX(fabs(denom), FLT_EPSILON);
906 double diff = alpha_i - alpha_j;
907 alpha_i += delta;
908 alpha_j += delta;
909
910 if( diff > 0 && alpha_j < 0 )
911 {
912 alpha_j = 0;
913 alpha_i = diff;
914 }
915 else if( diff <= 0 && alpha_i < 0 )
916 {
917 alpha_i = 0;
918 alpha_j = -diff;
919 }
920
921 if( diff > C_i - C_j && alpha_i > C_i )
922 {
923 alpha_i = C_i;
924 alpha_j = C_i - diff;
925 }
926 else if( diff <= C_i - C_j && alpha_j > C_j )
927 {
928 alpha_j = C_j;
929 alpha_i = C_j + diff;
930 }
931 }
932 else
933 {
934 double denom = Q_i[i] + Q_j[j] - 2 * Q_i[j];
935 double delta = (G[i] - G[j]) / MAX(fabs(denom), FLT_EPSILON);
936 double sum = alpha_i + alpha_j;
937 alpha_i -= delta;
938 alpha_j += delta;
939
940 if( sum > C_i && alpha_i > C_i )
941 {
942 alpha_i = C_i;
943 alpha_j = sum - C_i;
944 }
945 else if( sum <= C_i && alpha_j < 0)
946 {
947 alpha_j = 0;
948 alpha_i = sum;
949 }
950
951 if( sum > C_j && alpha_j > C_j )
952 {
953 alpha_j = C_j;
954 alpha_i = sum - C_j;
955 }
956 else if( sum <= C_j && alpha_i < 0 )
957 {
958 alpha_i = 0;
959 alpha_j = sum;
960 }
961 }
962 // update alpha
963 alpha[i] = alpha_i;
964 alpha[j] = alpha_j;
965 update_alpha_status(i);
966 update_alpha_status(j);
967
968 // update G
969 delta_alpha_i = alpha_i - old_alpha_i;
970 delta_alpha_j = alpha_j - old_alpha_j;
971
972 for( k = 0; k < alpha_count; k++ )
973 G[k] += Q_i[k] * delta_alpha_i + Q_j[k] * delta_alpha_j;
974 }
975
976 // calculate rho
977 (this->*calc_rho_func)( si.rho, si.r );
978
979 // calculate objective value
980 for( i = 0, si.obj = 0; i < alpha_count; i++ )
981 si.obj += alpha[i] * (G[i] + b[i]);
982
983 si.obj *= 0.5;
984
985 si.upper_bound_p = C[1];
986 si.upper_bound_n = C[0];
987
988 return true;
989}
