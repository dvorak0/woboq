55 static void openCLExecuteKernel_2(Context *clCxt, const cv::ocl::ProgramEntry* source, string kernelName, size_t globalThreads[3],
56 size_t localThreads[3], vector< pair<size_t, const void *> > &args, int channels,
57 int depth, char *build_options, FLUSH_MODE finish_mode)
58 {
59 //construct kernel name
60 //The rule is functionName_Cn_Dn, C represent Channels, D Represent DataType Depth, n represent an integer number
61 //for exmaple split_C2_D2, represent the split kernel with channels =2 and dataType Depth = 2(Data type is char)
62 stringstream idxStr;
63 if(channels != -1)
64 idxStr << "_C" << channels;
65 if(depth != -1)
66 idxStr << "_D" << depth;
67 kernelName += idxStr.str();
68
69 cl_kernel kernel;
70 kernel = openCLGetKernelFromSource(clCxt, source, kernelName, build_options);
71
72 if ( localThreads != NULL)
73 {
74 globalThreads[0] = divUp(globalThreads[0], localThreads[0]) * localThreads[0];
75 globalThreads[1] = divUp(globalThreads[1], localThreads[1]) * localThreads[1];
76 globalThreads[2] = divUp(globalThreads[2], localThreads[2]) * localThreads[2];
77
78 //size_t blockSize = localThreads[0] * localThreads[1] * localThreads[2];
79 cv::ocl::openCLVerifyKernel(clCxt, kernel, localThreads);
80 }
81 for(size_t i = 0; i < args.size(); i ++)
82 openCLSafeCall(clSetKernelArg(kernel, i, args[i].first, args[i].second));
83
84 openCLSafeCall(clEnqueueNDRangeKernel(*(cl_command_queue*)clCxt->getOpenCLCommandQueuePtr(), kernel, 3, NULL, globalThreads,
85 localThreads, 0, NULL, NULL));
86
87 switch(finish_mode)
88 {
89 case CLFINISH:
90 clFinish(*(cl_command_queue*)clCxt->getOpenCLCommandQueuePtr());
91 case CLFLUSH:
92 clFlush(*(cl_command_queue*)clCxt->getOpenCLCommandQueuePtr());
93 break;
94 case DISABLE:
95 default:
96 break;
97 }
98 openCLSafeCall(clReleaseKernel(kernel));
99 }

117 cl_mem bindTexture(const oclMat &mat)
118 {
119 cl_mem texture;
120 cl_image_format format;
121 int err;
122 int depth = mat.depth();
123 int channels = mat.oclchannels();
124
125 switch(depth)
126 {
127 case CV_8U:
128 format.image_channel_data_type = CL_UNSIGNED_INT8;
129 break;
130 case CV_32S:
131 format.image_channel_data_type = CL_UNSIGNED_INT32;
132 break;
133 case CV_32F:
134 format.image_channel_data_type = CL_FLOAT;
135 break;
136 default:
137 CV_Error(-1, "Image forma is not supported");
138 break;
139 }
140 switch(channels)
141 {
142 case 1:
143 format.image_channel_order = CL_R;
144 break;
145 case 3:
146 format.image_channel_order = CL_RGB;
147 break;
148 case 4:
149 format.image_channel_order = CL_RGBA;
150 break;
151 default:
152 CV_Error(-1, "Image format is not supported");
153 break;
154 }
155#ifdef CL_VERSION_1_2
156 //this enables backwards portability to
157 //run on OpenCL 1.1 platform if library binaries are compiled with OpenCL 1.2 support
158 if(Context::getContext()->supportsFeature(FEATURE_CL_VER_1_2))
159 {
160 cl_image_desc desc;
161 desc.image_type = CL_MEM_OBJECT_IMAGE2D;
162 desc.image_width = mat.cols;
163 desc.image_height = mat.rows;
164 desc.image_depth = 0;
165 desc.image_array_size = 1;
166 desc.image_row_pitch = 0;
167 desc.image_slice_pitch = 0;
168 desc.buffer = NULL;
169 desc.num_mip_levels = 0;
170 desc.num_samples = 0;
171 texture = clCreateImage(*(cl_context*)mat.clCxt->getOpenCLContextPtr(), CL_MEM_READ_WRITE, &format, &desc, NULL, &err);
172 }
173 else
174#endif
175 {
176 texture = clCreateImage2D(
177 *(cl_context*)mat.clCxt->getOpenCLContextPtr(),
178 CL_MEM_READ_WRITE,
179 &format,
180 mat.cols,
181 mat.rows,
182 0,
183 NULL,
184 &err);
185 }
186 size_t origin[] = { 0, 0, 0 };
187 size_t region[] = { (size_t)mat.cols, (size_t)mat.rows, 1 };
188
189 cl_mem devData;
190 if (mat.cols * mat.elemSize() != mat.step)
191 {
192 devData = clCreateBuffer(*(cl_context*)mat.clCxt->getOpenCLContextPtr(), CL_MEM_READ_ONLY, mat.cols * mat.rows
193 * mat.elemSize(), NULL, NULL);
194 const size_t regin[3] = {(size_t)mat.cols * mat.elemSize(), (size_t)mat.rows, 1};
195 clEnqueueCopyBufferRect(*(cl_command_queue*)mat.clCxt->getOpenCLCommandQueuePtr(), (cl_mem)mat.data, devData, origin, origin,
196 regin, mat.step, 0, mat.cols * mat.elemSize(), 0, 0, NULL, NULL);
197 clFlush(*(cl_command_queue*)mat.clCxt->getOpenCLCommandQueuePtr());
198 }
199 else
200 {
201 devData = (cl_mem)mat.data;
202 }
203
204 clEnqueueCopyBufferToImage(*(cl_command_queue*)mat.clCxt->getOpenCLCommandQueuePtr(), devData, texture, 0, origin, region, 0, NULL, 0);
205 if ((mat.cols * mat.elemSize() != mat.step))
206 {
207 clFlush(*(cl_command_queue*)mat.clCxt->getOpenCLCommandQueuePtr());
208 clReleaseMemObject(devData);
209 }
210
211 openCLSafeCall(err);
212 return texture;
213 }

216 {
217 return Ptr<TextureCL>(new TextureCL(bindTexture(mat), mat.rows, mat.cols, mat.type()));
