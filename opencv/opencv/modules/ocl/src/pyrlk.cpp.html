54struct dim3
55{
56 unsigned int x, y, z;
57};

59static void calcPatchSize(cv::Size winSize, int cn, dim3 &block, dim3 &patch, bool isDeviceArch11)
60{
61 winSize.width *= cn;
62
63 if (winSize.width > 32 && winSize.width > 2 * winSize.height)
64 {
65 block.x = isDeviceArch11 ? 16 : 32;
66 block.y = 8;
67 }
68 else
69 {
70 block.x = 16;
71 block.y = isDeviceArch11 ? 8 : 16;
72 }
73
74 patch.x = (winSize.width + block.x - 1) / block.x;
75 patch.y = (winSize.height + block.y - 1) / block.y;
76
77 block.z = patch.z = 1;
78}

80static void lkSparse_run(oclMat &I, oclMat &J,
81 const oclMat &prevPts, oclMat &nextPts, oclMat &status, oclMat& err, bool /*GET_MIN_EIGENVALS*/, int ptcount,
82 int level, dim3 patch, Size winSize, int iters)
83{
84 Context *clCxt = I.clCxt;
85 string kernelName = "lkSparse";
86 size_t localThreads[3] = { 8, 8, 1 };
87 size_t globalThreads[3] = { 8 * (size_t)ptcount, 8, 1};
88 int cn = I.oclchannels();
89 char calcErr = level==0?1:0;
90
91 vector<pair<size_t , const void *> > args;
92
93 cl_mem ITex = bindTexture(I);
94 cl_mem JTex = bindTexture(J);
95
96 args.push_back( make_pair( sizeof(cl_mem), (void *)&ITex ));
97 args.push_back( make_pair( sizeof(cl_mem), (void *)&JTex ));
98 args.push_back( make_pair( sizeof(cl_mem), (void *)&prevPts.data ));
99 args.push_back( make_pair( sizeof(cl_int), (void *)&prevPts.step ));
100 args.push_back( make_pair( sizeof(cl_mem), (void *)&nextPts.data ));
101 args.push_back( make_pair( sizeof(cl_int), (void *)&nextPts.step ));
102 args.push_back( make_pair( sizeof(cl_mem), (void *)&status.data ));
103 args.push_back( make_pair( sizeof(cl_mem), (void *)&err.data ));
104 args.push_back( make_pair( sizeof(cl_int), (void *)&level ));
105 args.push_back( make_pair( sizeof(cl_int), (void *)&I.rows ));
106 args.push_back( make_pair( sizeof(cl_int), (void *)&I.cols ));
107 args.push_back( make_pair( sizeof(cl_int), (void *)&patch.x ));
108 args.push_back( make_pair( sizeof(cl_int), (void *)&patch.y ));
109 args.push_back( make_pair( sizeof(cl_int), (void *)&cn ));
110 args.push_back( make_pair( sizeof(cl_int), (void *)&winSize.width ));
111 args.push_back( make_pair( sizeof(cl_int), (void *)&winSize.height ));
112 args.push_back( make_pair( sizeof(cl_int), (void *)&iters ));
113 args.push_back( make_pair( sizeof(cl_char), (void *)&calcErr ));
114
115 bool is_cpu = isCpuDevice();
116 if (is_cpu)
117 {
118 openCLExecuteKernel(clCxt, &pyrlk, kernelName, globalThreads, localThreads, args, I.oclchannels(), I.depth(), (char*)" -D CPU");
119 }
120 else
121 {
122 stringstream idxStr;
123 idxStr << kernelName << "_C" << I.oclchannels() << "_D" << I.depth();
124 cl_kernel kernel = openCLGetKernelFromSource(clCxt, &pyrlk, idxStr.str());
125 int wave_size = (int)queryWaveFrontSize(kernel);
126 openCLSafeCall(clReleaseKernel(kernel));
127
128 static char opt[32] = {0};
129 sprintf(opt, "-D WAVE_SIZE=%d", wave_size);
130
131 openCLExecuteKernel(clCxt, &pyrlk, kernelName, globalThreads, localThreads,
132 args, I.oclchannels(), I.depth(), opt);
133 }
134 releaseTexture(ITex);
135 releaseTexture(JTex);
136}

138void cv::ocl::PyrLKOpticalFlow::sparse(const oclMat &prevImg, const oclMat &nextImg, const oclMat &prevPts, oclMat &nextPts, oclMat &status, oclMat *err)
139{
140 if (prevPts.empty())
141 {
142 nextPts.release();
143 status.release();
144 if (err) err->release();
145 return;
146 }
147
148 derivLambda = std::min(std::max(derivLambda, 0.0), 1.0);
149
150 iters = std::min(std::max(iters, 0), 100);
151
152 const int cn = prevImg.oclchannels();
153
154 dim3 block, patch;
155 calcPatchSize(winSize, cn, block, patch, isDeviceArch11_);
156
157 CV_Assert(derivLambda >= 0);
158 CV_Assert(maxLevel >= 0 && winSize.width > 2 && winSize.height > 2);
159 CV_Assert(prevImg.size() == nextImg.size() && prevImg.type() == nextImg.type());
160 CV_Assert(patch.x > 0 && patch.x < 6 && patch.y > 0 && patch.y < 6);
161 CV_Assert(prevPts.rows == 1 && prevPts.type() == CV_32FC2);
162
163 if (useInitialFlow)
164 CV_Assert(nextPts.size() == prevPts.size() && nextPts.type() == CV_32FC2);
165 else
166 ensureSizeIsEnough(1, prevPts.cols, prevPts.type(), nextPts);
167
168 oclMat temp1 = (useInitialFlow ? nextPts : prevPts).reshape(1);
169 oclMat temp2 = nextPts.reshape(1);
170 multiply(1.0f/(1<<maxLevel)/2.0f, temp1, temp2);
171
172 ensureSizeIsEnough(1, prevPts.cols, CV_8UC1, status);
173 status.setTo(Scalar::all(1));
174
175 bool errMat = false;
176 if (!err)
177 {
178 err = new oclMat(1, prevPts.cols, CV_32FC1);
179 errMat = true;
180 }
181 else
182 ensureSizeIsEnough(1, prevPts.cols, CV_32FC1, *err);
183
184 // build the image pyramids.
185 prevPyr_.resize(maxLevel + 1);
186 nextPyr_.resize(maxLevel + 1);
187
188 if (cn == 1 || cn == 4)
189 {
190 prevImg.convertTo(prevPyr_[0], CV_32F);
191 nextImg.convertTo(nextPyr_[0], CV_32F);
192 }
193
194 for (int level = 1; level <= maxLevel; ++level)
195 {
196 pyrDown(prevPyr_[level - 1], prevPyr_[level]);
197 pyrDown(nextPyr_[level - 1], nextPyr_[level]);
198 }
199
200 // dI/dx ~ Ix, dI/dy ~ Iy
201 for (int level = maxLevel; level >= 0; level--)
202 {
203 lkSparse_run(prevPyr_[level], nextPyr_[level],
204 prevPts, nextPts, status, *err, getMinEigenVals, prevPts.cols,
205 level, patch, winSize, iters);
206 }
207
208 if(errMat)
209 delete err;
210}

212static void lkDense_run(oclMat &I, oclMat &J, oclMat &u, oclMat &v,
213 oclMat &prevU, oclMat &prevV, oclMat *err, Size winSize, int iters)
214{
215 Context *clCxt = I.clCxt;
216
217 string kernelName = "lkDense";
218
219 size_t localThreads[3] = { 16, 16, 1 };
220 size_t globalThreads[3] = { (size_t)I.cols, (size_t)I.rows, 1};
221
222 cl_char calcErr = err ? 1 : 0;
223
224 cl_mem ITex;
225 cl_mem JTex;
226
227 ITex = bindTexture(I);
228 JTex = bindTexture(J);
229
230 vector<pair<size_t , const void *> > args;
231
232 args.push_back( make_pair( sizeof(cl_mem), (void *)&ITex ));
233 args.push_back( make_pair( sizeof(cl_mem), (void *)&JTex ));
234
235 args.push_back( make_pair( sizeof(cl_mem), (void *)&u.data ));
236 args.push_back( make_pair( sizeof(cl_int), (void *)&u.step ));
237 args.push_back( make_pair( sizeof(cl_mem), (void *)&v.data ));
238 args.push_back( make_pair( sizeof(cl_int), (void *)&v.step ));
239 args.push_back( make_pair( sizeof(cl_mem), (void *)&prevU.data ));
240 args.push_back( make_pair( sizeof(cl_int), (void *)&prevU.step ));
241 args.push_back( make_pair( sizeof(cl_mem), (void *)&prevV.data ));
242 args.push_back( make_pair( sizeof(cl_int), (void *)&prevV.step ));
243 args.push_back( make_pair( sizeof(cl_int), (void *)&I.rows ));
244 args.push_back( make_pair( sizeof(cl_int), (void *)&I.cols ));
245 args.push_back( make_pair( sizeof(cl_int), (void *)&winSize.width ));
246 args.push_back( make_pair( sizeof(cl_int), (void *)&winSize.height ));
247 args.push_back( make_pair( sizeof(cl_int), (void *)&iters ));
248 args.push_back( make_pair( sizeof(cl_char), (void *)&calcErr ));
249
250 openCLExecuteKernel(clCxt, &pyrlk, kernelName, globalThreads, localThreads, args, I.oclchannels(), I.depth());
251
252 releaseTexture(ITex);
253 releaseTexture(JTex);
254}

256void cv::ocl::PyrLKOpticalFlow::dense(const oclMat &prevImg, const oclMat &nextImg, oclMat &u, oclMat &v, oclMat *err)
257{
258 CV_Assert(prevImg.type() == CV_8UC1);
259 CV_Assert(prevImg.size() == nextImg.size() && prevImg.type() == nextImg.type());
260 CV_Assert(maxLevel >= 0);
261 CV_Assert(winSize.width > 2 && winSize.height > 2);
262
263 if (err)
264 err->create(prevImg.size(), CV_32FC1);
265
266 prevPyr_.resize(maxLevel + 1);
267 nextPyr_.resize(maxLevel + 1);
268
269 prevPyr_[0] = prevImg;
270 nextImg.convertTo(nextPyr_[0], CV_32F);
271
272 for (int level = 1; level <= maxLevel; ++level)
273 {
274 pyrDown(prevPyr_[level - 1], prevPyr_[level]);
275 pyrDown(nextPyr_[level - 1], nextPyr_[level]);
276 }
277
278 ensureSizeIsEnough(prevImg.size(), CV_32FC1, uPyr_[0]);
279 ensureSizeIsEnough(prevImg.size(), CV_32FC1, vPyr_[0]);
280 ensureSizeIsEnough(prevImg.size(), CV_32FC1, uPyr_[1]);
281 ensureSizeIsEnough(prevImg.size(), CV_32FC1, vPyr_[1]);
282 uPyr_[1].setTo(Scalar::all(0));
283 vPyr_[1].setTo(Scalar::all(0));
284
285 Size winSize2i(winSize.width, winSize.height);
286
287 int idx = 0;
288
289 for (int level = maxLevel; level >= 0; level--)
290 {
291 int idx2 = (idx + 1) & 1;
292
293 lkDense_run(prevPyr_[level], nextPyr_[level], uPyr_[idx], vPyr_[idx], uPyr_[idx2], vPyr_[idx2],
294 level == 0 ? err : 0, winSize2i, iters);
295
296 if (level > 0)
297 idx = idx2;
298 }
299
300 uPyr_[idx].copyTo(u);
301 vPyr_[idx].copyTo(v);
302}
