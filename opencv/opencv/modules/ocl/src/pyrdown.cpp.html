56static void pyrdown_run(const oclMat &src, const oclMat &dst)
57{
58
59 CV_Assert(src.type() == dst.type());
60 CV_Assert(src.depth() != CV_8S);
61
62 Context *clCxt = src.clCxt;
63 string kernelName = "pyrDown";
64
65 size_t localThreads[3] = { 256, 1, 1 };
66 size_t globalThreads[3] = { (size_t)src.cols, (size_t)dst.rows, 1};
67
68 vector<pair<size_t , const void *> > args;
69 args.push_back( make_pair( sizeof(cl_mem), (void *)&src.data ));
70 args.push_back( make_pair( sizeof(cl_int), (void *)&src.step ));
71 args.push_back( make_pair( sizeof(cl_int), (void *)&src.rows));
72 args.push_back( make_pair( sizeof(cl_int), (void *)&src.cols));
73 args.push_back( make_pair( sizeof(cl_mem), (void *)&dst.data ));
74 args.push_back( make_pair( sizeof(cl_int), (void *)&dst.step ));
75 args.push_back( make_pair( sizeof(cl_int), (void *)&dst.cols));
76
77 openCLExecuteKernel(clCxt, &pyr_down, kernelName, globalThreads, localThreads, args, src.oclchannels(), src.depth());
78}
79//////////////////////////////////////////////////////////////////////////////
80// pyrDown
81
82void cv::ocl::pyrDown(const oclMat &src, oclMat &dst)
83{
84 int depth = src.depth(), channels = src.channels();
85 CV_Assert(depth == CV_8U || depth == CV_16U || depth == CV_16S || depth == CV_32F);
86 CV_Assert(channels == 1 || channels == 3 || channels == 4);
87
88 dst.create((src.rows + 1) / 2, (src.cols + 1) / 2, src.type());
89
90 pyrdown_run(src, dst);
91}
