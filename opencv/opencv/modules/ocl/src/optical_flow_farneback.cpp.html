61static void updateMatricesOcl(const oclMat &flowx, const oclMat &flowy, const oclMat &R0, const oclMat &R1, oclMat &M)
62{
63 string kernelName("updateMatrices");
64#ifdef ANDROID
65 size_t localThreads[3] = { 32, 4, 1 };
66#else
67 size_t localThreads[3] = { 32, 8, 1 };
68#endif
69 size_t globalThreads[3] = { (size_t)flowx.cols, (size_t)flowx.rows, 1 };
70
71 std::vector< std::pair<size_t, const void *> > args;
72 args.push_back(std::make_pair(sizeof(cl_mem), (void *)&M.data));
73 args.push_back(std::make_pair(sizeof(cl_mem), (void *)&flowx.data));
74 args.push_back(std::make_pair(sizeof(cl_mem), (void *)&flowy.data));
75 args.push_back(std::make_pair(sizeof(cl_mem), (void *)&R0.data));
76 args.push_back(std::make_pair(sizeof(cl_mem), (void *)&R1.data));
77 args.push_back(std::make_pair(sizeof(cl_int), (void *)&flowx.rows));
78 args.push_back(std::make_pair(sizeof(cl_int), (void *)&flowx.cols));
79 args.push_back(std::make_pair(sizeof(cl_int), (void *)&M.step));
80 args.push_back(std::make_pair(sizeof(cl_int), (void *)&flowx.step));
81 args.push_back(std::make_pair(sizeof(cl_int), (void *)&flowy.step));
82 args.push_back(std::make_pair(sizeof(cl_int), (void *)&R0.step));
83 args.push_back(std::make_pair(sizeof(cl_int), (void *)&R1.step));
84
85 openCLExecuteKernel(Context::getContext(), &optical_flow_farneback, kernelName,
86 globalThreads, localThreads, args, -1, -1);
87}

143static oclMat allocMatFromBuf(int rows, int cols, int type, oclMat &mat)
144{
145 if (!mat.empty() && mat.type() == type && mat.rows >= rows && mat.cols >= cols)
146 return mat(Rect(0, 0, cols, rows));
147 return mat = oclMat(rows, cols, type);

270void cv::ocl::FarnebackOpticalFlow::prepareGaussian(
271 int n, double sigma, float *g, float *xg, float *xxg,
272 double &ig11, double &ig03, double &ig33, double &ig55)
273{
274 double s = 0.;
275 for (int x = -n; x <= n; x++)
276 {
277 g[x] = (float)std::exp(-x*x/(2*sigma*sigma));
278 s += g[x];
279 }
280
281 s = 1./s;
282 for (int x = -n; x <= n; x++)
283 {
284 g[x] = (float)(g[x]*s);
285 xg[x] = (float)(x*g[x]);
286 xxg[x] = (float)(x*x*g[x]);
287 }
288
289 Mat_<double> G(6, 6);
290 G.setTo(0);
291
292 for (int y = -n; y <= n; y++)
293 {
294 for (int x = -n; x <= n; x++)
295 {
296 G(0,0) += g[y]*g[x];
297 G(1,1) += g[y]*g[x]*x*x;
298 G(3,3) += g[y]*g[x]*x*x*x*x;
299 G(5,5) += g[y]*g[x]*x*x*y*y;
300 }
301 }
302
303 //G[0][0] = 1.;
304 G(2,2) = G(0,3) = G(0,4) = G(3,0) = G(4,0) = G(1,1);
305 G(4,4) = G(3,3);
306 G(3,4) = G(4,3) = G(5,5);
307
308 // invG:
309 // [ x e e ]
310 // [ y ]
311 // [ y ]
312 // [ e z ]
313 // [ e z ]
314 // [ u ]
315 Mat_<double> invG = G.inv(DECOMP_CHOLESKY);
316
317 ig11 = invG(1,1);
318 ig03 = invG(0,3);
319 ig33 = invG(3,3);
320 ig55 = invG(5,5);
321}

323void cv::ocl::FarnebackOpticalFlow::setPolynomialExpansionConsts(int n, double sigma)
324{
325 vector<float> buf(n*6 + 3);
326 float* g = &buf[0] + n;
327 float* xg = g + n*2 + 1;
328 float* xxg = xg + n*2 + 1;
329
330 if (sigma < FLT_EPSILON)
331 sigma = n*0.3;
332
333 double ig11, ig03, ig33, ig55;
334 prepareGaussian(n, sigma, g, xg, xxg, ig11, ig03, ig33, ig55);
335
336 cv::Mat t_g(1, n + 1, CV_32FC1, g);
337 cv::Mat t_xg(1, n + 1, CV_32FC1, xg);
338 cv::Mat t_xxg(1, n + 1, CV_32FC1, xxg);
339
340 gMat.upload(t_g);
341 xgMat.upload(t_xg);
342 xxgMat.upload(t_xxg);
343
344 ig[0] = static_cast<float>(ig11);
345 ig[1] = static_cast<float>(ig03);
346 ig[2] = static_cast<float>(ig33);
347 ig[3] = static_cast<float>(ig55);
348}

380void cv::ocl::FarnebackOpticalFlow::operator ()(
381 const oclMat &frame0, const oclMat &frame1, oclMat &flowx, oclMat &flowy)
382{
383 CV_Assert(frame0.channels() == 1 && frame1.channels() == 1);
384 CV_Assert(frame0.size() == frame1.size());
385 CV_Assert(polyN == 5 || polyN == 7);
386 CV_Assert(!fastPyramids || std::abs(pyrScale - 0.5) < 1e-6);
387
388 Size size = frame0.size();
389 oclMat prevFlowX, prevFlowY, curFlowX, curFlowY;
390
391 flowx.create(size, CV_32F);
392 flowy.create(size, CV_32F);
393 oclMat flowx0 = flowx;
394 oclMat flowy0 = flowy;
395
396 // Crop unnecessary levels
397 double scale = 1;
398 int numLevelsCropped = 0;
399 for (; numLevelsCropped < numLevels; numLevelsCropped++)
400 {
401 scale *= pyrScale;
402 if (size.width*scale < MIN_SIZE || size.height*scale < MIN_SIZE)
403 break;
404 }
405
406 frame0.convertTo(frames_[0], CV_32F);
407 frame1.convertTo(frames_[1], CV_32F);
408
409 if (fastPyramids)
410 {
411 // Build Gaussian pyramids using pyrDown()
412 pyramid0_.resize(numLevelsCropped + 1);
413 pyramid1_.resize(numLevelsCropped + 1);
414 pyramid0_[0] = frames_[0];
415 pyramid1_[0] = frames_[1];
416 for (int i = 1; i <= numLevelsCropped; ++i)
417 {
418 pyrDown(pyramid0_[i - 1], pyramid0_[i]);
419 pyrDown(pyramid1_[i - 1], pyramid1_[i]);
420 }
421 }
422
423 setPolynomialExpansionConsts(polyN, polySigma);
424
425 for (int k = numLevelsCropped; k >= 0; k--)
426 {
427 scale = 1;
428 for (int i = 0; i < k; i++)
429 scale *= pyrScale;
430
431 double sigma = (1./scale - 1) * 0.5;
432 int smoothSize = cvRound(sigma*5) | 1;
433 smoothSize = std::max(smoothSize, 3);
434
435 int width = cvRound(size.width*scale);
436 int height = cvRound(size.height*scale);
437
438 if (fastPyramids)
439 {
440 width = pyramid0_[k].cols;
441 height = pyramid0_[k].rows;
442 }
443
444 if (k > 0)
445 {
446 curFlowX.create(height, width, CV_32F);
447 curFlowY.create(height, width, CV_32F);
448 }
449 else
450 {
451 curFlowX = flowx0;
452 curFlowY = flowy0;
453 }
454
455 if (!prevFlowX.data)
456 {
457 if (flags & cv::OPTFLOW_USE_INITIAL_FLOW)
458 {
459 resize(flowx0, curFlowX, Size(width, height), 0, 0, INTER_LINEAR);
460 resize(flowy0, curFlowY, Size(width, height), 0, 0, INTER_LINEAR);
461 multiply(scale, curFlowX, curFlowX);
462 multiply(scale, curFlowY, curFlowY);
463 }
464 else
465 {
466 curFlowX.setTo(0);
467 curFlowY.setTo(0);
468 }
469 }
470 else
471 {
472 resize(prevFlowX, curFlowX, Size(width, height), 0, 0, INTER_LINEAR);
473 resize(prevFlowY, curFlowY, Size(width, height), 0, 0, INTER_LINEAR);
474 multiply(1./pyrScale, curFlowX, curFlowX);
475 multiply(1./pyrScale, curFlowY, curFlowY);
476 }
477
478 oclMat M = allocMatFromBuf(5*height, width, CV_32F, M_);
479 oclMat bufM = allocMatFromBuf(5*height, width, CV_32F, bufM_);
480 oclMat R[2] =
481 {
482 allocMatFromBuf(5*height, width, CV_32F, R_[0]),
483 allocMatFromBuf(5*height, width, CV_32F, R_[1])
484 };
485
486 if (fastPyramids)
487 {
488 polynomialExpansionOcl(pyramid0_[k], polyN, R[0]);
489 polynomialExpansionOcl(pyramid1_[k], polyN, R[1]);
490 }
491 else
492 {
493 oclMat blurredFrame[2] =
494 {
495 allocMatFromBuf(size.height, size.width, CV_32F, blurredFrame_[0]),
496 allocMatFromBuf(size.height, size.width, CV_32F, blurredFrame_[1])
497 };
498 oclMat pyrLevel[2] =
499 {
500 allocMatFromBuf(height, width, CV_32F, pyrLevel_[0]),
501 allocMatFromBuf(height, width, CV_32F, pyrLevel_[1])
502 };
503
504 Mat g = getGaussianKernel(smoothSize, sigma, CV_32F);
505 setGaussianBlurKernel(g.ptr<float>(smoothSize/2), smoothSize/2);
506
507 for (int i = 0; i < 2; i++)
508 {
509 gaussianBlurOcl(frames_[i], smoothSize/2, blurredFrame[i]);
510 resize(blurredFrame[i], pyrLevel[i], Size(width, height), INTER_LINEAR);
511 polynomialExpansionOcl(pyrLevel[i], polyN, R[i]);
512 }
513 }
514
515 optflow_farneback::updateMatricesOcl(curFlowX, curFlowY, R[0], R[1], M);
516
517 if (flags & OPTFLOW_FARNEBACK_GAUSSIAN)
518 {
519 Mat g = getGaussianKernel(winSize, winSize/2*0.3f, CV_32F);
520 setGaussianBlurKernel(g.ptr<float>(winSize/2), winSize/2);
521 }
522 for (int i = 0; i < numIters; i++)
523 {
524 if (flags & OPTFLOW_FARNEBACK_GAUSSIAN)
525 updateFlow_gaussianBlur(R[0], R[1], curFlowX, curFlowY, M, bufM, winSize, i < numIters-1);
526 else
527 updateFlow_boxFilter(R[0], R[1], curFlowX, curFlowY, M, bufM, winSize, i < numIters-1);
528 }
529
530 prevFlowX = curFlowX;
531 prevFlowY = curFlowY;
532 }
533
534 flowx = curFlowX;
535 flowy = curFlowY;
536}
