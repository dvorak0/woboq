78static bool clBlasInitialized = false;
79
80void cv::ocl::clBlasSetup()
81{
82 if(!clBlasInitialized)
83 {
84 AutoLock lock(getInitializationMutex());
85 if(!clBlasInitialized)
86 {
87 openCLSafeCall(clAmdBlasSetup());
88 clBlasInitialized = true;
89 }
90 }
91}
92
93void cv::ocl::clBlasTeardown()
94{
95 AutoLock lock(getInitializationMutex());
96 if(clBlasInitialized)
97 {
98 clAmdBlasTeardown();
99 clBlasInitialized = false;
100 }
101}

103void cv::ocl::gemm(const oclMat &src1, const oclMat &src2, double alpha,
104 const oclMat &src3, double beta, oclMat &dst, int flags)
105{
106 CV_Assert(src1.cols == src2.rows &&
107 (src3.empty() || (src1.rows == src3.rows && src2.cols == src3.cols)));
108 CV_Assert(!(cv::GEMM_3_T & flags)); // cv::GEMM_3_T is not supported
109 if(!src3.empty())
110 {
111 src3.copyTo(dst);
112 }
113 else
114 {
115 dst.create(src1.rows, src2.cols, src1.type());
116 dst.setTo(Scalar::all(0));
117 }
118
119 clBlasSetup();
120
121 const clAmdBlasTranspose transA = (cv::GEMM_1_T & flags) ? clAmdBlasTrans : clAmdBlasNoTrans;
122 const clAmdBlasTranspose transB = (cv::GEMM_2_T & flags) ? clAmdBlasTrans : clAmdBlasNoTrans;
123 const clAmdBlasOrder order = clAmdBlasRowMajor;
124
125 const int M = src1.rows;
126 const int N = src2.cols;
127 const int K = src1.cols;
128 int lda = src1.step;
129 int ldb = src2.step;
130 int ldc = dst.step;
131 int offa = src1.offset;
132 int offb = src2.offset;
133 int offc = dst.offset;
134
135 cl_command_queue clq = *(cl_command_queue*)src1.clCxt->getOpenCLCommandQueuePtr();
136 switch(src1.type())
137 {
138 case CV_32FC1:
139 lda /= sizeof(float);
140 ldb /= sizeof(float);
141 ldc /= sizeof(float);
142 offa /= sizeof(float);
143 offb /= sizeof(float);
144 offc /= sizeof(float);
145
146 openCLSafeCall
147 (
148 clAmdBlasSgemmEx(order, transA, transB, M, N, K,
149 alpha, (const cl_mem)src1.data, offa, lda, (const cl_mem)src2.data, offb, ldb,
150 beta, (cl_mem)dst.data, offc, ldc, 1, &clq, 0, NULL, NULL)
151 );
152 break;
153 case CV_64FC1:
154 lda /= sizeof(double);
155 ldb /= sizeof(double);
156 ldc /= sizeof(double);
157 offa /= sizeof(double);
158 offb /= sizeof(double);
159 offc /= sizeof(double);
160 openCLSafeCall
161 (
162 clAmdBlasDgemmEx(order, transA, transB, M, N, K,
163 alpha, (const cl_mem)src1.data, offa, lda, (const cl_mem)src2.data, offb, ldb,
164 beta, (cl_mem)dst.data, offc, ldc, 1, &clq, 0, NULL, NULL)
165 );
166 break;
167 case CV_32FC2:
168 {
169 lda /= sizeof(std::complex<float>);
170 ldb /= sizeof(std::complex<float>);
171 ldc /= sizeof(std::complex<float>);
172 offa /= sizeof(std::complex<float>);
173 offb /= sizeof(std::complex<float>);
174 offc /= sizeof(std::complex<float>);
175 cl_float2 alpha_2 = {{alpha, 0}};
176 cl_float2 beta_2 = {{beta, 0}};
177 openCLSafeCall
178 (
179 clAmdBlasCgemmEx(order, transA, transB, M, N, K,
180 alpha_2, (const cl_mem)src1.data, offa, lda, (const cl_mem)src2.data, offb, ldb,
181 beta_2, (cl_mem)dst.data, offc, ldc, 1, &clq, 0, NULL, NULL)
182 );
183 }
184 break;
185 case CV_64FC2:
186 {
187 lda /= sizeof(std::complex<double>);
188 ldb /= sizeof(std::complex<double>);
189 ldc /= sizeof(std::complex<double>);
190 offa /= sizeof(std::complex<double>);
191 offb /= sizeof(std::complex<double>);
192 offc /= sizeof(std::complex<double>);
193 cl_double2 alpha_2 = {{alpha, 0}};
194 cl_double2 beta_2 = {{beta, 0}};
195 openCLSafeCall
196 (
197 clAmdBlasZgemmEx(order, transA, transB, M, N, K,
198 alpha_2, (const cl_mem)src1.data, offa, lda, (const cl_mem)src2.data, offb, ldb,
199 beta_2, (cl_mem)dst.data, offc, ldc, 1, &clq, 0, NULL, NULL)
200 );
201 }
202 break;
203 }
204}
