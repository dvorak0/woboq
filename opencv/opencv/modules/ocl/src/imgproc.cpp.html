101 static std::vector<uchar> scalarToVector(const cv::Scalar & sc, int depth, int ocn, int cn)
102 {
103 CV_Assert(ocn == cn || (ocn == 4 && cn == 3));
104
105 static const int sizeMap[] = { sizeof(uchar), sizeof(char), sizeof(ushort),
106 sizeof(short), sizeof(int), sizeof(float), sizeof(double) };
107
108 int elemSize1 = sizeMap[depth];
109 int bufSize = elemSize1 * ocn;
110 std::vector<uchar> _buf(bufSize);
111 uchar * buf = &_buf[0];
112 scalarToRawData(sc, buf, CV_MAKE_TYPE(depth, cn));
113 memset(buf + elemSize1 * cn, 0, (ocn - cn) * elemSize1);
114
115 return _buf;
116 }

118 static void threshold_runner(const oclMat &src, oclMat &dst, double thresh, double maxVal, int thresholdType)
119 {
120 bool ival = src.depth() < CV_32F;
121 int cn = src.channels(), vecSize = 4, depth = src.depth();
122 std::vector<uchar> thresholdValue = scalarToVector(cv::Scalar::all(ival ? cvFloor(thresh) : thresh), dst.depth(),
123 dst.oclchannels(), dst.channels());
124 std::vector<uchar> maxValue = scalarToVector(cv::Scalar::all(maxVal), dst.depth(), dst.oclchannels(), dst.channels());
125
126 const char * const thresholdMap[] = { "THRESH_BINARY", "THRESH_BINARY_INV", "THRESH_TRUNC",
127 "THRESH_TOZERO", "THRESH_TOZERO_INV" };
128 const char * const channelMap[] = { "", "", "2", "4", "4" };
129 const char * const typeMap[] = { "uchar", "char", "ushort", "short", "int", "float", "double" };
130 std::string buildOptions = format("-D T=%s%s -D %s", typeMap[depth], channelMap[cn], thresholdMap[thresholdType]);
131
132 int elemSize = src.elemSize();
133 int src_step = src.step / elemSize, src_offset = src.offset / elemSize;
134 int dst_step = dst.step / elemSize, dst_offset = dst.offset / elemSize;
135
136 vector< pair<size_t, const void *> > args;
137 args.push_back( make_pair(sizeof(cl_mem), (void *)&src.data));
138 args.push_back( make_pair(sizeof(cl_int), (void *)&src_offset));
139 args.push_back( make_pair(sizeof(cl_int), (void *)&src_step));
140 args.push_back( make_pair(sizeof(cl_mem), (void *)&dst.data));
141 args.push_back( make_pair(sizeof(cl_int), (void *)&dst_offset));
142 args.push_back( make_pair(sizeof(cl_int), (void *)&dst_step));
143 args.push_back( make_pair(thresholdValue.size(), (void *)&thresholdValue[0]));
144 args.push_back( make_pair(maxValue.size(), (void *)&maxValue[0]));
145
146 int max_index = dst.cols, cols = dst.cols;
147 if (cn == 1 && vecSize > 1)
148 {
149 CV_Assert(((vecSize - 1) & vecSize) == 0 && vecSize <= 16);
150 cols = divUp(cols, vecSize);
151 buildOptions += format(" -D VECTORIZED -D VT=%s%d -D VLOADN=vload%d -D VECSIZE=%d -D VSTOREN=vstore%d",
152 typeMap[depth], vecSize, vecSize, vecSize, vecSize);
153
154 int vecSizeBytes = vecSize * dst.elemSize1();
155 if ((dst.offset % dst.step) % vecSizeBytes == 0 && dst.step % vecSizeBytes == 0)
156 buildOptions += " -D DST_ALIGNED";
157 if ((src.offset % src.step) % vecSizeBytes == 0 && src.step % vecSizeBytes == 0)
158 buildOptions += " -D SRC_ALIGNED";
159
160 args.push_back( make_pair(sizeof(cl_int), (void *)&max_index));
161 }
162
163 args.push_back( make_pair(sizeof(cl_int), (void *)&dst.rows));
164 args.push_back( make_pair(sizeof(cl_int), (void *)&cols));
165
166 size_t localThreads[3] = { 16, 16, 1 };
167 size_t globalThreads[3] = { (size_t)cols, (size_t)dst.rows, 1 };
168
169 openCLExecuteKernel(src.clCxt, &imgproc_threshold, "threshold", globalThreads, localThreads, args,
170 -1, -1, buildOptions.c_str());
171 }

173 double threshold(const oclMat &src, oclMat &dst, double thresh, double maxVal, int thresholdType)
174 {
175 CV_Assert(thresholdType == THRESH_BINARY || thresholdType == THRESH_BINARY_INV || thresholdType == THRESH_TRUNC
176 || thresholdType == THRESH_TOZERO || thresholdType == THRESH_TOZERO_INV);
177
178 dst.create(src.size(), src.type());
179 threshold_runner(src, dst, thresh, maxVal, thresholdType);
180
181 return thresh;
182 }

188 void remap( const oclMat &src, oclMat &dst, oclMat &map1, oclMat &map2, int interpolation, int borderType, const Scalar &borderValue )
189 {
190 Context *clCxt = src.clCxt;
191 bool supportsDouble = clCxt->supportsFeature(FEATURE_CL_DOUBLE);
192 if (!supportsDouble && src.depth() == CV_64F)
193 {
194 CV_Error(CV_OpenCLDoubleNotSupported, "Selected device does not support double");
195 return;
196 }
197
198 if (map1.empty())
199 map1.swap(map2);
200
201 CV_Assert(interpolation == INTER_LINEAR || interpolation == INTER_NEAREST);
202 CV_Assert((map1.type() == CV_16SC2 && (map2.empty() || (map2.type() == CV_16UC1 || map2.type() == CV_16SC1)) ) ||
203 (map1.type() == CV_32FC2 && !map2.data) ||
204 (map1.type() == CV_32FC1 && map2.type() == CV_32FC1));
205 CV_Assert(!map2.data || map2.size() == map1.size());
206 CV_Assert(borderType == BORDER_CONSTANT || borderType == BORDER_REPLICATE || borderType == BORDER_WRAP
207 || borderType == BORDER_REFLECT_101 || borderType == BORDER_REFLECT);
208
209 dst.create(map1.size(), src.type());
210
211 const char * const typeMap[] = { "uchar", "char", "ushort", "short", "int", "float", "double" };
212 const char * const channelMap[] = { "", "", "2", "4", "4" };
213 const char * const interMap[] = { "INTER_NEAREST", "INTER_LINEAR", "INTER_CUBIC", "INTER_LINEAR", "INTER_LANCZOS" };
214 const char * const borderMap[] = { "BORDER_CONSTANT", "BORDER_REPLICATE", "BORDER_REFLECT", "BORDER_WRAP",
215 "BORDER_REFLECT_101", "BORDER_TRANSPARENT" };
216
217 string kernelName = "remap";
218 if (map1.type() == CV_32FC2 && map2.empty())
219 kernelName += "_32FC2";
220 else if (map1.type() == CV_16SC2)
221 {
222 kernelName += "_16SC2";
223 if (!map2.empty())
224 kernelName += "_16UC1";
225 }
226 else if (map1.type() == CV_32FC1 && map2.type() == CV_32FC1)
227 kernelName += "_2_32FC1";
228 else
229 CV_Error(CV_StsBadArg, "Unsupported map types");
230
231 int ocn = dst.oclchannels();
232 size_t globalThreads[3] = { (size_t)dst.cols, (size_t)dst.rows, 1 };
233
234 Mat scalar(1, 1, CV_MAKE_TYPE(dst.depth(), ocn), borderValue);
235 std::string buildOptions = format("-D %s -D %s -D T=%s%s", interMap[interpolation],
236 borderMap[borderType], typeMap[src.depth()], channelMap[ocn]);
237
238 if (interpolation != INTER_NEAREST)
239 {
240 int wdepth = std::max(CV_32F, dst.depth());
241 buildOptions += format(" -D WT=%s%s -D convertToT=convert_%s%s%s -D convertToWT=convert_%s%s"
242 " -D convertToWT2=convert_%s2 -D WT2=%s2",
243 typeMap[wdepth], channelMap[ocn],
244 typeMap[src.depth()], channelMap[ocn], src.depth() < CV_32F ? "_sat_rte" : "",
245 typeMap[wdepth], channelMap[ocn],
246 typeMap[wdepth], typeMap[wdepth]);
247 }
248
249 int src_step = src.step / src.elemSize(), src_offset = src.offset / src.elemSize();
250 int map1_step = map1.step / map1.elemSize(), map1_offset = map1.offset / map1.elemSize();
251 int map2_step = map2.step / map2.elemSize(), map2_offset = map2.offset / map2.elemSize();
252 int dst_step = dst.step / dst.elemSize(), dst_offset = dst.offset / dst.elemSize();
253
254 vector< pair<size_t, const void *> > args;
255 args.push_back( make_pair(sizeof(cl_mem), (void *)&src.data));
256 args.push_back( make_pair(sizeof(cl_mem), (void *)&dst.data));
257 args.push_back( make_pair(sizeof(cl_mem), (void *)&map1.data));
258 if (!map2.empty())
259 args.push_back( make_pair(sizeof(cl_mem), (void *)&map2.data));
260 args.push_back( make_pair(sizeof(cl_int), (void *)&src_offset));
261 args.push_back( make_pair(sizeof(cl_int), (void *)&dst_offset));
262 args.push_back( make_pair(sizeof(cl_int), (void *)&map1_offset));
263 if (!map2.empty())
264 args.push_back( make_pair(sizeof(cl_int), (void *)&map2_offset));
265 args.push_back( make_pair(sizeof(cl_int), (void *)&src_step));
266 args.push_back( make_pair(sizeof(cl_int), (void *)&dst_step));
267 args.push_back( make_pair(sizeof(cl_int), (void *)&map1_step));
268 if (!map2.empty())
269 args.push_back( make_pair(sizeof(cl_int), (void *)&map2_step));
270 args.push_back( make_pair(sizeof(cl_int), (void *)&src.cols));
271 args.push_back( make_pair(sizeof(cl_int), (void *)&src.rows));
272 args.push_back( make_pair(sizeof(cl_int), (void *)&dst.cols));
273 args.push_back( make_pair(sizeof(cl_int), (void *)&dst.rows));
274 args.push_back( make_pair(scalar.elemSize(), (void *)scalar.data));
275
276#ifdef ANDROID
277 openCLExecuteKernel(clCxt, &imgproc_remap, kernelName, globalThreads, NULL, args, -1, -1, buildOptions.c_str());
278#else
279 size_t localThreads[3] = { 256, 1, 1 };
280 openCLExecuteKernel(clCxt, &imgproc_remap, kernelName, globalThreads, localThreads, args, -1, -1, buildOptions.c_str());
281#endif
282 }

287 static void computeResizeAreaTabs(int ssize, int dsize, double scale, int * const map_tab,
288 float * const alpha_tab, int * const ofs_tab)
289 {
290 int k = 0, dx = 0;
291 for ( ; dx < dsize; dx++)
292 {
293 ofs_tab[dx] = k;
294
295 double fsx1 = dx * scale;
296 double fsx2 = fsx1 + scale;
297 double cellWidth = std::min(scale, ssize - fsx1);
298
299 int sx1 = cvCeil(fsx1), sx2 = cvFloor(fsx2);
300
301 sx2 = std::min(sx2, ssize - 1);
302 sx1 = std::min(sx1, sx2);
303
304 if (sx1 - fsx1 > 1e-3)
305 {
306 map_tab[k] = sx1 - 1;
307 alpha_tab[k++] = (float)((sx1 - fsx1) / cellWidth);
308 }
309
310 for (int sx = sx1; sx < sx2; sx++)
311 {
312 map_tab[k] = sx;
313 alpha_tab[k++] = float(1.0 / cellWidth);
314 }
315
316 if (fsx2 - sx2 > 1e-3)
317 {
318 map_tab[k] = sx2;
319 alpha_tab[k++] = (float)(std::min(std::min(fsx2 - sx2, 1.), cellWidth) / cellWidth);
320 }
321 }
322 ofs_tab[dx] = k;
323 }

325 static void computeResizeAreaFastTabs(int * dmap_tab, int * smap_tab, int scale, int dcols, int scol)
326 {
327 for (int i = 0; i < dcols; ++i)
328 dmap_tab[i] = scale * i;
329
330 for (int i = 0, size = dcols * scale; i < size; ++i)
331 smap_tab[i] = std::min(scol - 1, i);
332 }

334 static void resize_gpu( const oclMat &src, oclMat &dst, double ifx, double ify, int interpolation)
335 {
336 float ifxf = (float)ifx, ifyf = (float)ify;
337 int src_step = src.step / src.elemSize(), src_offset = src.offset / src.elemSize();
338 int dst_step = dst.step / dst.elemSize(), dst_offset = dst.offset / dst.elemSize();
339 int ocn = dst.oclchannels(), depth = dst.depth();
340
341 const char * const interMap[] = { "NN", "LN", "CUBIC", "AREA", "LAN4" };
342 std::string kernelName = std::string("resize") + interMap[interpolation];
343
344 const char * const typeMap[] = { "uchar", "char", "ushort", "short", "int", "float", "double" };
345 const char * const channelMap[] = { "" , "", "2", "4", "4" };
346 std::string buildOption = format("-D %s -D T=%s%s", interMap[interpolation], typeMap[depth], channelMap[ocn]);
347
348 int wdepth = std::max(src.depth(), CV_32F);
349
350 // check if fx, fy is integer and then we have inter area fast mode
351 int iscale_x = saturate_cast<int>(ifx);
352 int iscale_y = saturate_cast<int>(ify);
353
354 bool is_area_fast = std::abs(ifx - iscale_x) < DBL_EPSILON &&
355 std::abs(ify - iscale_y) < DBL_EPSILON;
356 if (is_area_fast)
357 wdepth = std::max(src.depth(), CV_32S);
358
359 if (interpolation != INTER_NEAREST)
360 {
361 buildOption += format(" -D WT=%s -D WTV=%s%s -D convertToWTV=convert_%s%s -D convertToT=convert_%s%s%s",
362 typeMap[wdepth], typeMap[wdepth], channelMap[ocn],
363 typeMap[wdepth], channelMap[ocn],
364 typeMap[src.depth()], channelMap[ocn], src.depth() <= CV_32S ? "_sat_rte" : "");
365 }
366
367#ifdef ANDROID
368 size_t blkSizeX = 16, blkSizeY = 8;
369#else
370 size_t blkSizeX = 16, blkSizeY = 16;
371#endif
372 size_t glbSizeX;
373 if (src.type() == CV_8UC1 && interpolation == INTER_LINEAR)
374 {
375 size_t cols = (dst.cols + dst.offset % 4 + 3) / 4;
376 glbSizeX = cols % blkSizeX == 0 && cols != 0 ? cols : (cols / blkSizeX + 1) * blkSizeX;
377 }
378 else
379 glbSizeX = dst.cols;
380
381 oclMat alphaOcl, mapOcl, tabofsOcl;
382 if (interpolation == INTER_AREA)
383 {
384 if (is_area_fast)
385 {
386 kernelName += "_FAST";
387 int wdepth2 = std::max(CV_32F, src.depth());
388 buildOption += format(" -D WT2V=%s%s -D convertToWT2V=convert_%s%s -D AREA_FAST -D XSCALE=%d -D YSCALE=%d -D SCALE=%f",
389 typeMap[wdepth2], channelMap[ocn], typeMap[wdepth2], channelMap[ocn],
390 iscale_x, iscale_y, 1.0f / (iscale_x * iscale_y));
391
392 int smap_tab_size = dst.cols * iscale_x + dst.rows * iscale_y;
393 AutoBuffer<int> dmap_tab(dst.cols + dst.rows), smap_tab(smap_tab_size);
394 int * dxmap_tab = dmap_tab, * dymap_tab = dxmap_tab + dst.cols;
395 int * sxmap_tab = smap_tab, * symap_tab = smap_tab + dst.cols * iscale_y;
396
397 computeResizeAreaFastTabs(dxmap_tab, sxmap_tab, iscale_x, dst.cols, src.cols);
398 computeResizeAreaFastTabs(dymap_tab, symap_tab, iscale_y, dst.rows, src.rows);
399
400 tabofsOcl = oclMat(1, dst.cols + dst.rows, CV_32SC1, (void *)dmap_tab);
401 mapOcl = oclMat(1, smap_tab_size, CV_32SC1, (void *)smap_tab);
402 }
403 else
404 {
405 Size ssize = src.size(), dsize = dst.size();
406 int xytab_size = (ssize.width + ssize.height) << 1;
407 int tabofs_size = dsize.height + dsize.width + 2;
408
409 AutoBuffer<int> _xymap_tab(xytab_size), _xyofs_tab(tabofs_size);
410 AutoBuffer<float> _xyalpha_tab(xytab_size);
411 int * xmap_tab = _xymap_tab, * ymap_tab = _xymap_tab + (ssize.width << 1);
412 float * xalpha_tab = _xyalpha_tab, * yalpha_tab = _xyalpha_tab + (ssize.width << 1);
413 int * xofs_tab = _xyofs_tab, * yofs_tab = _xyofs_tab + dsize.width + 1;
414
415 computeResizeAreaTabs(ssize.width, dsize.width, ifx, xmap_tab, xalpha_tab, xofs_tab);
416 computeResizeAreaTabs(ssize.height, dsize.height, ify, ymap_tab, yalpha_tab, yofs_tab);
417
418 // loading precomputed arrays to GPU
419 alphaOcl = oclMat(1, xytab_size, CV_32FC1, (void *)_xyalpha_tab);
420 mapOcl = oclMat(1, xytab_size, CV_32SC1, (void *)_xymap_tab);
421 tabofsOcl = oclMat(1, tabofs_size, CV_32SC1, (void *)_xyofs_tab);
422 }
423 }
424
425 size_t globalThreads[3] = { glbSizeX, (size_t)dst.rows, 1 };
426 size_t localThreads[3] = { blkSizeX, blkSizeY, 1 };
427
428 std::vector< std::pair<size_t, const void *> > args;
429 args.push_back( make_pair(sizeof(cl_mem), (void *)&dst.data));
430 args.push_back( make_pair(sizeof(cl_mem), (void *)&src.data));
431 args.push_back( make_pair(sizeof(cl_int), (void *)&dst_offset));
432 args.push_back( make_pair(sizeof(cl_int), (void *)&src_offset));
433 args.push_back( make_pair(sizeof(cl_int), (void *)&dst_step));
434 args.push_back( make_pair(sizeof(cl_int), (void *)&src_step));
435 args.push_back( make_pair(sizeof(cl_int), (void *)&src.cols));
436 args.push_back( make_pair(sizeof(cl_int), (void *)&src.rows));
437 args.push_back( make_pair(sizeof(cl_int), (void *)&dst.cols));
438 args.push_back( make_pair(sizeof(cl_int), (void *)&dst.rows));
439
440 if (wdepth == CV_64F)
441 {
442 args.push_back( make_pair(sizeof(cl_double), (void *)&ifx));
443 args.push_back( make_pair(sizeof(cl_double), (void *)&ify));
444 }
445 else
446 {
447 args.push_back( make_pair(sizeof(cl_float), (void *)&ifxf));
448 args.push_back( make_pair(sizeof(cl_float), (void *)&ifyf));
449 }
450
451 // precomputed tabs
452 if (!tabofsOcl.empty())
453 args.push_back( make_pair(sizeof(cl_mem), (void *)&tabofsOcl.data));
454
455 if (!mapOcl.empty())
456 args.push_back( make_pair(sizeof(cl_mem), (void *)&mapOcl.data));
457
458 if (!alphaOcl.empty())
459 args.push_back( make_pair(sizeof(cl_mem), (void *)&alphaOcl.data));
460
461 ocn = interpolation == INTER_LINEAR ? ocn : -1;
462 depth = interpolation == INTER_LINEAR ? depth : -1;
463
464 openCLExecuteKernel(src.clCxt, &imgproc_resize, kernelName, globalThreads, localThreads, args,
465 ocn, depth, buildOption.c_str());
466 }

468 void resize(const oclMat &src, oclMat &dst, Size dsize, double fx, double fy, int interpolation)
469 {
470 if (!src.clCxt->supportsFeature(FEATURE_CL_DOUBLE) && src.depth() == CV_64F)
471 {
472 CV_Error(CV_OpenCLDoubleNotSupported, "Selected device does not support double");
473 return;
474 }
475
476 CV_Assert(src.type() == CV_8UC1 || src.type() == CV_8UC3 || src.type() == CV_8UC4
477 || src.type() == CV_32FC1 || src.type() == CV_32FC3 || src.type() == CV_32FC4);
478 CV_Assert(dsize.area() > 0 || (fx > 0 && fy > 0));
479
480 if (dsize.area() == 0)
481 {
482 dsize = Size(saturate_cast<int>(src.cols * fx), saturate_cast<int>(src.rows * fy));
483 CV_Assert(dsize.area() > 0);
484 }
485 else
486 {
487 fx = (double)dsize.width / src.cols;
488 fy = (double)dsize.height / src.rows;
489 }
490
491 double inv_fy = 1 / fy, inv_fx = 1 / fx;
492 CV_Assert(interpolation == INTER_LINEAR || interpolation == INTER_NEAREST ||
493 (interpolation == INTER_AREA && inv_fx >= 1 && inv_fy >= 1));
494
495 dst.create(dsize, src.type());
496
497 resize_gpu( src, dst, inv_fx, inv_fy, interpolation);
498 }

503 void medianFilter(const oclMat &src, oclMat &dst, int m)
504 {
505 CV_Assert( m % 2 == 1 && m > 1 );
506 CV_Assert( (src.depth() == CV_8U || src.depth() == CV_32F) && (src.channels() == 1 || src.channels() == 4));
507 dst.create(src.size(), src.type());
508
509 int srcStep = src.step / src.elemSize(), dstStep = dst.step / dst.elemSize();
510 int srcOffset = src.offset / src.elemSize(), dstOffset = dst.offset / dst.elemSize();

634 void convert_coeffs(F *M)
635 {
636 double D = M[0] * M[4] - M[1] * M[3];
637 D = D != 0 ? 1. / D : 0;
638 double A11 = M[4] * D, A22 = M[0] * D;
639 M[0] = A11;
640 M[1] *= -D;
641 M[3] *= -D;
642 M[4] = A22;
643 double b1 = -M[0] * M[2] - M[1] * M[5];
644 double b2 = -M[3] * M[2] - M[4] * M[5];
645 M[2] = b1;
646 M[5] = b2;
647 }
648
649 double invert(double *M)
650 {
651#define Sd(y,x) (Sd[y*3+x])
652#define Dd(y,x) (Dd[y*3+x])
653#define det3(m) (m(0,0)*(m(1,1)*m(2,2) - m(1,2)*m(2,1)) - \
654 m(0,1)*(m(1,0)*m(2,2) - m(1,2)*m(2,0)) + \
655 m(0,2)*(m(1,0)*m(2,1) - m(1,1)*m(2,0)))
656 double *Sd = M;
657 double *Dd = M;
658 double d = det3(Sd);
659 double result = 0;
660 if ( d != 0)
661 {
662 double t[9];
663 result = d;
664 d = 1. / d;
665
666 t[0] = (Sd(1, 1) * Sd(2, 2) - Sd(1, 2) * Sd(2, 1)) * d;
667 t[1] = (Sd(0, 2) * Sd(2, 1) - Sd(0, 1) * Sd(2, 2)) * d;
668 t[2] = (Sd(0, 1) * Sd(1, 2) - Sd(0, 2) * Sd(1, 1)) * d;
669
670 t[3] = (Sd(1, 2) * Sd(2, 0) - Sd(1, 0) * Sd(2, 2)) * d;
671 t[4] = (Sd(0, 0) * Sd(2, 2) - Sd(0, 2) * Sd(2, 0)) * d;
672 t[5] = (Sd(0, 2) * Sd(1, 0) - Sd(0, 0) * Sd(1, 2)) * d;
673
674 t[6] = (Sd(1, 0) * Sd(2, 1) - Sd(1, 1) * Sd(2, 0)) * d;
675 t[7] = (Sd(0, 1) * Sd(2, 0) - Sd(0, 0) * Sd(2, 1)) * d;
676 t[8] = (Sd(0, 0) * Sd(1, 1) - Sd(0, 1) * Sd(1, 0)) * d;
677
678 Dd(0, 0) = t[0];
679 Dd(0, 1) = t[1];
680 Dd(0, 2) = t[2];
681 Dd(1, 0) = t[3];
682 Dd(1, 1) = t[4];
683 Dd(1, 2) = t[5];
684 Dd(2, 0) = t[6];
685 Dd(2, 1) = t[7];
686 Dd(2, 2) = t[8];
687 }
688 return result;

1692 static void oclbilateralFilter_8u( const oclMat &src, oclMat &dst, int d,
1693 double sigma_color, double sigma_space,
1694 int borderType )
1695 {
1696 int cn = src.channels();
1697 int i, j, maxk, radius;
1698
1699 CV_Assert( (src.channels() == 1 || src.channels() == 3) &&
1700 src.type() == dst.type() && src.size() == dst.size() &&
1701 src.data != dst.data );
1702
1703 if ( sigma_color <= 0 )
1704 sigma_color = 1;
1705 if ( sigma_space <= 0 )
1706 sigma_space = 1;
1707
1708 double gauss_color_coeff = -0.5 / (sigma_color * sigma_color);
1709 double gauss_space_coeff = -0.5 / (sigma_space * sigma_space);
1710
1711 if ( d <= 0 )
1712 radius = cvRound(sigma_space * 1.5);
1713 else
1714 radius = d / 2;
1715 radius = MAX(radius, 1);
1716 d = radius * 2 + 1;
1717
1718 oclMat temp;
1719 copyMakeBorder( src, temp, radius, radius, radius, radius, borderType );
1720
1721 vector<float> _color_weight(cn * 256);
1722 vector<float> _space_weight(d * d);
1723 vector<int> _space_ofs(d * d);
1724 float *color_weight = &_color_weight[0];
1725 float *space_weight = &_space_weight[0];
1726 int *space_ofs = &_space_ofs[0];
1727
1728 int dst_step_in_pixel = dst.step / dst.elemSize();
1729 int dst_offset_in_pixel = dst.offset / dst.elemSize();
1730 int temp_step_in_pixel = temp.step / temp.elemSize();
1731
1732 // initialize color-related bilateral filter coefficients
1733 for( i = 0; i < 256 * cn; i++ )
1734 color_weight[i] = (float)std::exp(i * i * gauss_color_coeff);
1735
1736 // initialize space-related bilateral filter coefficients
1737 for( i = -radius, maxk = 0; i <= radius; i++ )
1738 for( j = -radius; j <= radius; j++ )
1739 {
1740 double r = std::sqrt((double)i * i + (double)j * j);
1741 if ( r > radius )
1742 continue;
1743 space_weight[maxk] = (float)std::exp(r * r * gauss_space_coeff);
1744 space_ofs[maxk++] = (int)(i * temp_step_in_pixel + j);
1745 }
1746
1747 oclMat oclcolor_weight(1, cn * 256, CV_32FC1, color_weight);
1748 oclMat oclspace_weight(1, d * d, CV_32FC1, space_weight);
1749 oclMat oclspace_ofs(1, d * d, CV_32SC1, space_ofs);
1750
1751 string kernelName = "bilateral";
1752#ifdef ANDROID
1753 size_t localThreads[3] = { 16, 8, 1 };
1754#else
1755 size_t localThreads[3] = { 16, 16, 1 };
1756#endif
1757 size_t globalThreads[3] = { (size_t)dst.cols, (size_t)dst.rows, 1 };
1758
1759 if ((dst.type() == CV_8UC1) && ((dst.offset & 3) == 0) && ((dst.cols & 3) == 0))
1760 {
1761 kernelName = "bilateral2";
1762 globalThreads[0] = dst.cols >> 2;
1763 }
1764
1765 vector<pair<size_t , const void *> > args;
1766 args.push_back( make_pair( sizeof(cl_mem), (void *)&dst.data ));
1767 args.push_back( make_pair( sizeof(cl_mem), (void *)&temp.data ));
1768 args.push_back( make_pair( sizeof(cl_int), (void *)&dst.rows ));
1769 args.push_back( make_pair( sizeof(cl_int), (void *)&dst.cols ));
1770 args.push_back( make_pair( sizeof(cl_int), (void *)&maxk ));
1771 args.push_back( make_pair( sizeof(cl_int), (void *)&radius ));
1772 args.push_back( make_pair( sizeof(cl_int), (void *)&dst_step_in_pixel ));
1773 args.push_back( make_pair( sizeof(cl_int), (void *)&dst_offset_in_pixel ));
1774 args.push_back( make_pair( sizeof(cl_int), (void *)&temp_step_in_pixel ));
1775 args.push_back( make_pair( sizeof(cl_int), (void *)&temp.rows ));
1776 args.push_back( make_pair( sizeof(cl_int), (void *)&temp.cols ));
1777 args.push_back( make_pair( sizeof(cl_mem), (void *)&oclcolor_weight.data ));
1778 args.push_back( make_pair( sizeof(cl_mem), (void *)&oclspace_weight.data ));
1779 args.push_back( make_pair( sizeof(cl_mem), (void *)&oclspace_ofs.data ));
1780
1781 openCLExecuteKernel(src.clCxt, &imgproc_bilateral, kernelName, globalThreads, localThreads, args, dst.oclchannels(), dst.depth());
1782 }
