68bool KNearestNeighbour::train(const Mat& trainData, Mat& labels, Mat& sampleIdx,
69 bool isRegression, int _max_k, bool updateBase)
70{
71 max_k = _max_k;
72 bool cv_knn_train = CvKNearest::train(trainData, labels, sampleIdx, isRegression, max_k, updateBase);
73
74 CvVectors* s = CvKNearest::samples;
75
76 cv::Mat samples_mat(s->count, CvKNearest::var_count + 1, s->type);
77
78 float* s1 = (float*)(s + 1);
79 for(int i = 0; i < s->count; i++)
80 {
81 float* t1 = s->data.fl[i];
82 for(int j = 0; j < CvKNearest::var_count; j++)
83 {
84 Point pos(j, i);
85 samples_mat.at<float>(pos) = t1[j];
86 }
87
88 Point pos_label(CvKNearest::var_count, i);
89 samples_mat.at<float>(pos_label) = s1[i];
90 }
91
92 samples_ocl = samples_mat;
93 return cv_knn_train;
94}

96void KNearestNeighbour::find_nearest(const oclMat& lsamples, int k, oclMat& lables)
97{
98 CV_Assert(!samples_ocl.empty());
99 lables.create(lsamples.rows, 1, CV_32FC1);
100
101 CV_Assert(lsamples.cols == CvKNearest::var_count);
102 CV_Assert(lsamples.type() == CV_32FC1);
103 CV_Assert(k >= 1 && k <= max_k);
104
105 int k1 = KNearest::get_sample_count();
106 k1 = MIN( k1, k );
107
108 String kernel_name = "knn_find_nearest";
109 cl_ulong local_memory_size = (cl_ulong)Context::getContext()->getDeviceInfo().localMemorySize;
110 int nThreads = local_memory_size / (2 * k * 4);
111 if(nThreads >= 256)
112 nThreads = 256;
113
114 int smem_size = nThreads * k * 4 * 2;
115 size_t local_thread[] = {1, (size_t)nThreads, 1};
116 size_t global_thread[] = {1, (size_t)lsamples.rows, 1};
117
118 char build_option[50];
119 if(!Context::getContext()->supportsFeature(FEATURE_CL_DOUBLE))
120 {
121 sprintf(build_option, " ");
122 }else
123 sprintf(build_option, "-D DOUBLE_SUPPORT");
124
125 std::vector< std::pair<size_t, const void*> > args;
126
127 int samples_ocl_step = samples_ocl.step/samples_ocl.elemSize();
128 int samples_step = lsamples.step/lsamples.elemSize();
129 int lables_step = lables.step/lables.elemSize();
130
131 int _regression = 0;
132 if(CvKNearest::regression)
133 _regression = 1;
134
135 args.push_back(make_pair(sizeof(cl_mem), (void*)&lsamples.data));
136 args.push_back(make_pair(sizeof(cl_int), (void*)&lsamples.rows));
137 args.push_back(make_pair(sizeof(cl_int), (void*)&lsamples.cols));
138 args.push_back(make_pair(sizeof(cl_int), (void*)&samples_step));
139 args.push_back(make_pair(sizeof(cl_int), (void*)&k));
140 args.push_back(make_pair(sizeof(cl_mem), (void*)&samples_ocl.data));
141 args.push_back(make_pair(sizeof(cl_int), (void*)&samples_ocl.rows));
142 args.push_back(make_pair(sizeof(cl_int), (void*)&samples_ocl_step));
143 args.push_back(make_pair(sizeof(cl_mem), (void*)&lables.data));
144 args.push_back(make_pair(sizeof(cl_int), (void*)&lables_step));
145 args.push_back(make_pair(sizeof(cl_int), (void*)&_regression));
146 args.push_back(make_pair(sizeof(cl_int), (void*)&k1));
147 args.push_back(make_pair(sizeof(cl_int), (void*)&samples_ocl.cols));
148 args.push_back(make_pair(sizeof(cl_int), (void*)&nThreads));
149 args.push_back(make_pair(smem_size, (void*)NULL));
150 openCLExecuteKernel(Context::getContext(), &knearest, kernel_name, global_thread, local_thread, args, -1, -1, build_option);
151}
