59void cv::ocl::CannyBuf::create(const Size &image_size, int apperture_size)
60{
61 ensureSizeIsEnough(image_size, CV_32SC1, dx);
62 ensureSizeIsEnough(image_size, CV_32SC1, dy);
63
64 if(apperture_size == 3)
65 {
66 ensureSizeIsEnough(image_size, CV_32SC1, dx_buf);
67 ensureSizeIsEnough(image_size, CV_32SC1, dy_buf);
68 }
69 else if(apperture_size > 0)
70 {
71 Mat kx, ky;
72 if (!filterDX)
73 {
74 filterDX = createDerivFilter_GPU(CV_8U, CV_32S, 1, 0, apperture_size, BORDER_REPLICATE);
75 }
76 if (!filterDY)
77 {
78 filterDY = createDerivFilter_GPU(CV_8U, CV_32S, 0, 1, apperture_size, BORDER_REPLICATE);
79 }
80 }
81 ensureSizeIsEnough(2 * (image_size.height + 2), image_size.width + 2, CV_32FC1, edgeBuf);
82
83 ensureSizeIsEnough(1, image_size.area(), CV_16UC2, trackBuf1);
84 ensureSizeIsEnough(1, image_size.area(), CV_16UC2, trackBuf2);
85}

122 void CannyCaller(CannyBuf &buf, oclMat &dst, float low_thresh, float high_thresh)
123 {
124 using namespace ::cv::ocl::canny;
125 oclMat magBuf = buf.edgeBuf(Rect(0, 0, buf.edgeBuf.cols, buf.edgeBuf.rows / 2));
126 oclMat mapBuf = buf.edgeBuf(Rect(0, buf.edgeBuf.rows / 2, buf.edgeBuf.cols, buf.edgeBuf.rows / 2));
127
128 calcMap_gpu(buf.dx, buf.dy, magBuf, mapBuf, dst.rows, dst.cols, low_thresh, high_thresh);
129
130 edgesHysteresisLocal_gpu(mapBuf, buf.trackBuf1, buf.counter, dst.rows, dst.cols);
131
132 edgesHysteresisGlobal_gpu(mapBuf, buf.trackBuf1, buf.trackBuf2, buf.counter, dst.rows, dst.cols);
133
134 getEdges_gpu(mapBuf, dst, dst.rows, dst.cols);
135 }

144void cv::ocl::Canny(const oclMat &src, CannyBuf &buf, oclMat &dst, double low_thresh, double high_thresh, int apperture_size, bool L2gradient)
145{
146 using namespace ::cv::ocl::canny;
147
148 CV_Assert(src.type() == CV_8UC1);
149
150 if( low_thresh > high_thresh )
151 std::swap( low_thresh, high_thresh );
152
153 dst.create(src.size(), CV_8U);
154 dst.setTo(Scalar::all(0));
155
156 buf.create(src.size(), apperture_size);
157 buf.edgeBuf.setTo(Scalar::all(0));
158
159 oclMat magBuf = buf.edgeBuf(Rect(0, 0, buf.edgeBuf.cols, buf.edgeBuf.rows / 2));
160
161 if (apperture_size == 3)
162 {
163 calcSobelRowPass_gpu(src, buf.dx_buf, buf.dy_buf, src.rows, src.cols);
164
165 calcMagnitude_gpu(buf.dx_buf, buf.dy_buf, buf.dx, buf.dy, magBuf, src.rows, src.cols, L2gradient);
166 }
167 else
168 {
169 buf.filterDX->apply(src, buf.dx);
170 buf.filterDY->apply(src, buf.dy);
171
172 calcMagnitude_gpu(buf.dx, buf.dy, magBuf, src.rows, src.cols, L2gradient);
173 }
174 CannyCaller(buf, dst, static_cast<float>(low_thresh), static_cast<float>(high_thresh));
175}

182void cv::ocl::Canny(const oclMat &dx, const oclMat &dy, CannyBuf &buf, oclMat &dst, double low_thresh, double high_thresh, bool L2gradient)
183{
184 using namespace ::cv::ocl::canny;
185
186 CV_Assert(dx.type() == CV_32SC1 && dy.type() == CV_32SC1 && dx.size() == dy.size());
187
188 if( low_thresh > high_thresh )
189 std::swap( low_thresh, high_thresh);
190
191 dst.create(dx.size(), CV_8U);
192 dst.setTo(Scalar::all(0));
193
194 buf.dx = dx;
195 buf.dy = dy;
196 buf.create(dx.size(), -1);
197 buf.edgeBuf.setTo(Scalar::all(0));
198
199 oclMat magBuf = buf.edgeBuf(Rect(0, 0, buf.edgeBuf.cols, buf.edgeBuf.rows / 2));
200
201 calcMagnitude_gpu(buf.dx, buf.dy, magBuf, dx.rows, dx.cols, L2gradient);
202
203 CannyCaller(buf, dst, static_cast<float>(low_thresh), static_cast<float>(high_thresh));
204}

207{
208 Context *clCxt = src.clCxt;

213 args.push_back( make_pair( sizeof(cl_mem), (void *)&dx_buf.data));
214 args.push_back( make_pair( sizeof(cl_mem), (void *)&dy_buf.data));
215 args.push_back( make_pair( sizeof(cl_int), (void *)&rows));
216 args.push_back( make_pair( sizeof(cl_int), (void *)&cols));
217 args.push_back( make_pair( sizeof(cl_int), (void *)&src.step));
218 args.push_back( make_pair( sizeof(cl_int), (void *)&src.offset));
219 args.push_back( make_pair( sizeof(cl_int), (void *)&dx_buf.step));
220 args.push_back( make_pair( sizeof(cl_int), (void *)&dx_buf.offset));
221 args.push_back( make_pair( sizeof(cl_int), (void *)&dy_buf.step));
222 args.push_back( make_pair( sizeof(cl_int), (void *)&dy_buf.offset));
223
224 size_t globalThreads[3] = {(size_t)cols, (size_t)rows, 1};
225 size_t localThreads[3] = {16, 16, 1};
226 openCLExecuteKernel(clCxt, &imgproc_canny, kernelName, globalThreads, localThreads, args, -1, -1);
227}

340void canny::edgesHysteresisGlobal_gpu(oclMat &map, oclMat &st1, oclMat &st2, oclMat& counter, int rows, int cols)
341{
342 Context *clCxt = map.clCxt;
343 vector< pair<size_t, const void *> > args;
344 size_t localThreads[3] = {128, 1, 1};
345
346 while(1 > 0)
347 {
348 Mat counterMat; counter.download(counterMat);
349 int count = counterMat.at<int>(0, 0);
350 CV_Assert(count >= 0);
351 if (count == 0)
352 break;
353
354 counterMat.at<int>(0, 0) = 0;
355 counter.upload(counterMat);
356
357 args.clear();
358 size_t globalThreads[3] = {(size_t)std::min((unsigned)count, 65535u) * 128, divUp(count, 65535), 1};
359 args.push_back( make_pair( sizeof(cl_mem), (void *)&map.data));
360 args.push_back( make_pair( sizeof(cl_mem), (void *)&st1.data));
361 args.push_back( make_pair( sizeof(cl_mem), (void *)&st2.data));
362 args.push_back( make_pair( sizeof(cl_mem), (void *)&counter.data));
363 args.push_back( make_pair( sizeof(cl_int), (void *)&rows));
364 args.push_back( make_pair( sizeof(cl_int), (void *)&cols));
365 args.push_back( make_pair( sizeof(cl_int), (void *)&count));
366 args.push_back( make_pair( sizeof(cl_int), (void *)&map.step));
367 args.push_back( make_pair( sizeof(cl_int), (void *)&map.offset));
368
369 openCLExecuteKernel(clCxt, &imgproc_canny, "edgesHysteresisGlobal", globalThreads, localThreads, args, -1, -1);
370 std::swap(st1, st2);
371 }
