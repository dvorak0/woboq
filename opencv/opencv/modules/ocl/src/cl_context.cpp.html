74static bool parseOpenCLVersion(const std::string& versionStr, int& major, int& minor)
75{
76 size_t p0 = versionStr.find(' ');
77 while (true)
78 {
79 if (p0 == std::string::npos)
80 break;
81 if (p0 + 1 >= versionStr.length())
82 break;
83 char c = versionStr[p0 + 1];
84 if (isdigit(c))
85 break;
86 p0 = versionStr.find(' ', p0 + 1);
87 }
88 size_t p1 = versionStr.find('.', p0);
89 size_t p2 = versionStr.find(' ', p1);
90 if (p0 == std::string::npos || p1 == std::string::npos || p2 == std::string::npos)
91 {
92 major = 0;
93 minor = 0;
94 return false;
95 }
96 std::string majorStr = versionStr.substr(p0 + 1, p1 - p0 - 1);
97 std::string minorStr = versionStr.substr(p1 + 1, p2 - p1 - 1);
98 major = atoi(majorStr.c_str());
99 minor = atoi(minorStr.c_str());
100 return true;
101}

134{
135 cl_platform_id platform_id;
136 cl_device_id device_id;

138 DeviceInfoImpl()
139 : platform_id(NULL), device_id(NULL)
140 {
141 }

143 void init(int id, PlatformInfoImpl& platformInfoImpl, cl_device_id device)
144 {
145 CV_Assert(device_id == NULL);
146
147 this->_id = id;
148 platform_id = platformInfoImpl.platform_id;
149 device_id = device;
150
151 this->platform = &platformInfoImpl;
152
153 cl_device_type type = cl_device_type(-1);
154 openCLSafeCall(getScalarInfo(clGetDeviceInfo, device, CL_DEVICE_TYPE, type));
155 this->deviceType = DeviceType(type);
156
157 openCLSafeCall(getStringInfo(clGetDeviceInfo, device, CL_DEVICE_PROFILE, this->deviceProfile));
158 openCLSafeCall(getStringInfo(clGetDeviceInfo, device, CL_DEVICE_VERSION, this->deviceVersion));
159 openCLSafeCall(getStringInfo(clGetDeviceInfo, device, CL_DEVICE_NAME, this->deviceName));
160 openCLSafeCall(getStringInfo(clGetDeviceInfo, device, CL_DEVICE_VENDOR, this->deviceVendor));
161 cl_uint vendorID = 0;
162 openCLSafeCall(getScalarInfo(clGetDeviceInfo, device, CL_DEVICE_VENDOR_ID, vendorID));
163 this->deviceVendorId = vendorID;
164 openCLSafeCall(getStringInfo(clGetDeviceInfo, device, CL_DRIVER_VERSION, this->deviceDriverVersion));
165 openCLSafeCall(getStringInfo(clGetDeviceInfo, device, CL_DEVICE_EXTENSIONS, this->deviceExtensions));
166
167 parseOpenCLVersion(this->deviceVersion,
168 this->deviceVersionMajor, this->deviceVersionMinor);
169
170 size_t maxWGS = 0;
171 openCLSafeCall(getScalarInfo(clGetDeviceInfo, device, CL_DEVICE_MAX_WORK_GROUP_SIZE, maxWGS));
172 this->maxWorkGroupSize = maxWGS;
173
174 cl_uint maxDimensions = 0;
175 openCLSafeCall(getScalarInfo(clGetDeviceInfo, device, CL_DEVICE_MAX_WORK_ITEM_DIMENSIONS, maxDimensions));
176 std::vector<size_t> maxWIS(maxDimensions);
177 openCLSafeCall(clGetDeviceInfo(device, CL_DEVICE_MAX_WORK_ITEM_SIZES, sizeof(size_t) * maxDimensions,
178 (void *)&maxWIS[0], 0));
179 this->maxWorkItemSizes = maxWIS;
180
181 cl_uint maxCU = 0;
182 openCLSafeCall(getScalarInfo(clGetDeviceInfo, device, CL_DEVICE_MAX_COMPUTE_UNITS, maxCU));
183 this->maxComputeUnits = maxCU;
184
185 cl_ulong localMS = 0;
186 openCLSafeCall(getScalarInfo(clGetDeviceInfo, device, CL_DEVICE_LOCAL_MEM_SIZE, localMS));
187 this->localMemorySize = (size_t)localMS;
188
189 cl_ulong maxMAS = 0;
190 openCLSafeCall(getScalarInfo(clGetDeviceInfo, device, CL_DEVICE_MAX_MEM_ALLOC_SIZE, maxMAS));
191 this->maxMemAllocSize = (size_t)maxMAS;
192
193 cl_bool unifiedMemory = false;
194 openCLSafeCall(getScalarInfo(clGetDeviceInfo, device, CL_DEVICE_HOST_UNIFIED_MEMORY, unifiedMemory));
195 this->isUnifiedMemory = unifiedMemory != 0;
196
197 //initialize extra options for compilation. Currently only fp64 is included.
198 //Assume 4KB is enough to store all possible extensions.
199 openCLSafeCall(getStringInfo(clGetDeviceInfo, device, CL_DEVICE_EXTENSIONS, this->deviceExtensions));
200
201 size_t fp64_khr = this->deviceExtensions.find("cl_khr_fp64");
202 if(fp64_khr != std::string::npos)
203 {
204 this->compilationExtraOptions += "-D DOUBLE_SUPPORT";
205 this->haveDoubleSupport = true;
206 }
207 else
208 {
209 this->haveDoubleSupport = false;
210 }
211
212 size_t intel_platform = platformInfoImpl.platformVendor.find("Intel");
213 if(intel_platform != std::string::npos)
214 {
215 this->compilationExtraOptions += " -D INTEL_DEVICE";
216 this->isIntelDevice = true;
217 }
218 else
219 {
220 this->isIntelDevice = false;
221 }
222
223 if (id < 0)
224 {
225#ifdef CL_VERSION_1_2
226 if (this->deviceVersionMajor > 1 || (this->deviceVersionMajor == 1 && this->deviceVersionMinor >= 2))
227 {
228 ::clRetainDevice(device);
229 }
230#endif
231 }
232 }

264// Layout: <Platform>:<CPU|GPU|ACCELERATOR|nothing=GPU/CPU>:<deviceName>
265// Sample: AMD:GPU:
266// Sample: AMD:GPU:Tahiti
267// Sample: :GPU|CPU: = '' = ':' = '::'
268static bool parseOpenCLDeviceConfiguration(const std::string& configurationStr,
269 std::string& platform, std::vector<std::string>& deviceTypes, std::string& deviceNameOrID)
270{
271 std::string deviceTypesStr;
272 size_t p0 = configurationStr.find(':');
273 if (p0 != std::string::npos)
274 {
275 size_t p1 = configurationStr.find(':', p0 + 1);
276 if (p1 != std::string::npos)
277 {
278 size_t p2 = configurationStr.find(':', p1 + 1);
279 if (p2 != std::string::npos)
280 {
281 std::cerr << "ERROR: Invalid configuration string for OpenCL device" << std::endl;
282 return false;
283 }
284 else
285 {
286 // assume platform + device types + device name/id
287 platform = configurationStr.substr(0, p0);
288 deviceTypesStr = configurationStr.substr(p0 + 1, p1 - (p0 + 1));
289 deviceNameOrID = configurationStr.substr(p1 + 1, configurationStr.length() - (p1 + 1));
290 }
291 }
292 else
293 {
294 // assume platform + device types
295 platform = configurationStr.substr(0, p0);
296 deviceTypesStr = configurationStr.substr(p0 + 1, configurationStr.length() - (p0 + 1));
297 }
298 }
299 else
300 {
301 // assume only platform
302 platform = configurationStr;
303 }
304 deviceTypes = split(deviceTypesStr, '|');
305 return true;
306}

308static bool selectOpenCLDevice()
309{
310 std::string platform;
311 std::vector<std::string> deviceTypes;
312 std::string deviceName;
313 const char* configuration = getenv("OPENCV_OPENCL_DEVICE");
314 if (configuration)
315 {
316 if (!parseOpenCLDeviceConfiguration(std::string(configuration), platform, deviceTypes, deviceName))
317 return false;
318 }
319
320 bool isID = false;
321 int deviceID = -1;
322 if (deviceName.length() == 1)
323 // We limit ID range to 0..9, because we want to write:
324 // - '2500' to mean i5-2500
325 // - '8350' to mean AMD FX-8350
326 // - '650' to mean GeForce 650
327 // To extend ID range change condition to '> 0'
328 {
329 isID = true;
330 for (size_t i = 0; i < deviceName.length(); i++)
331 {
332 if (!isdigit(deviceName[i]))
333 {
334 isID = false;
335 break;
336 }
337 }
338 if (isID)
339 {
340 deviceID = atoi(deviceName.c_str());
341 CV_Assert(deviceID >= 0);
342 }
343 }
344
345 const PlatformInfo* platformInfo = NULL;
346 if (platform.length() > 0)
347 {
348 PlatformsInfo platforms;
349 getOpenCLPlatforms(platforms);
350 for (size_t i = 0; i < platforms.size(); i++)
351 {
352 if (platforms[i]->platformName.find(platform) != std::string::npos)
353 {
354 platformInfo = platforms[i];
355 break;
356 }
357 }
358 if (platformInfo == NULL)
359 {
360 std::cerr << "ERROR: Can't find OpenCL platform by name: " << platform << std::endl;
361 goto not_found;
362 }
363 }
364
365 if (deviceTypes.size() == 0)
366 {
367 if (!isID)
368 {
369 deviceTypes.push_back("GPU");
370 deviceTypes.push_back("CPU");
371 }
372 else
373 {
374 deviceTypes.push_back("ALL");
375 }
376 }
377 for (size_t t = 0; t < deviceTypes.size(); t++)
378 {
379 int deviceType = 0;
380 if (deviceTypes[t] == "GPU")
381 {
382 deviceType = CVCL_DEVICE_TYPE_GPU;
383 }
384 else if (deviceTypes[t] == "CPU")
385 {
386 deviceType = CVCL_DEVICE_TYPE_CPU;
387 }
388 else if (deviceTypes[t] == "ACCELERATOR")
389 {
390 deviceType = CVCL_DEVICE_TYPE_ACCELERATOR;
391 }
392 else if (deviceTypes[t] == "ALL")
393 {
394 deviceType = CVCL_DEVICE_TYPE_ALL;
395 }
396 else
397 {
398 std::cerr << "ERROR: Unsupported device type for OpenCL device (GPU, CPU, ACCELERATOR): " << deviceTypes[t] << std::endl;
399 goto not_found;
400 }
401
402 DevicesInfo devices;
403 getOpenCLDevices(devices, deviceType, platformInfo);
404
405 for (size_t i = (isID ? deviceID : 0);
406 (isID ? (i == (size_t)deviceID) : true) && (i < devices.size());
407 i++)
408 {
409 if (isID || devices[i]->deviceName.find(deviceName) != std::string::npos)
410 {
411 // check for OpenCL 1.1
412 if (devices[i]->deviceVersionMajor < 1 ||
413 (devices[i]->deviceVersionMajor == 1 && devices[i]->deviceVersionMinor < 1))
414 {
415 std::cerr << "Skip unsupported version of OpenCL device: " << devices[i]->deviceName
416 << "(" << devices[i]->platform->platformName << ")" << std::endl;
417 continue; // unsupported version of device, skip it
418 }
419 try
420 {
421 setDevice(devices[i]);
422 }
423 catch (...)
424 {
425 std::cerr << "ERROR: Can't select OpenCL device: " << devices[i]->deviceName
426 << "(" << devices[i]->platform->platformName << ")" << std::endl;
427 goto not_found;
428 }
429 return true;
430 }
431 }
432 }
433not_found:
434 std::cerr << "ERROR: Required OpenCL device not found, check configuration: " << (configuration == NULL ? "" : configuration) << std::endl
435 << " Platform: " << (platform.length() == 0 ? "any" : platform) << std::endl
436 << " Device types: ";
437 for (size_t t = 0; t < deviceTypes.size(); t++)
438 {
439 std::cerr << deviceTypes[t] << " ";
440 }
441 std::cerr << std::endl << " Device name: " << (deviceName.length() == 0 ? "any" : deviceName) << std::endl;
442 return false;
443}

445static bool __initialized = false;
446static int initializeOpenCLDevices()
447{
448 assert(!__initialized);
449 __initialized = true;
450
451 assert(global_devices.size() == 0);
452
453 std::vector<cl_platform_id> platforms;
454 try
455 {
456 openCLSafeCall(getPlatforms(platforms));
457 }
458 catch (cv::Exception&)
459 {
460 return 0; // OpenCL not found
461 }
462
463 global_platforms.resize(platforms.size());
464
465 for (size_t i = 0; i < platforms.size(); ++i)
466 {
467 PlatformInfoImpl& platformInfo = global_platforms[i];
468
469 cl_platform_id platform = platforms[i];
470 platformInfo.init(i, platform);
471
472 std::vector<cl_device_id> devices;
473 cl_int status = getDevices(platform, CL_DEVICE_TYPE_ALL, devices);
474 if(status != CL_DEVICE_NOT_FOUND)
475 openCLVerifyCall(status);
476
477 if(devices.size() > 0)
478 {
479 int baseIndx = global_devices.size();
480 global_devices.resize(baseIndx + devices.size());
481 platformInfo.deviceIDs.resize(devices.size());
482 platformInfo.devices.resize(devices.size());
483
484 for(size_t j = 0; j < devices.size(); ++j)
485 {
486 cl_device_id device = devices[j];
487
488 DeviceInfoImpl& deviceInfo = global_devices[baseIndx + j];
489 platformInfo.deviceIDs[j] = baseIndx + j;
490 deviceInfo.init(baseIndx + j, platformInfo, device);
491 }
492 }
493 }
494
495 for (size_t i = 0; i < platforms.size(); ++i)
496 {
497 PlatformInfoImpl& platformInfo = global_platforms[i];
498 for(size_t j = 0; j < platformInfo.deviceIDs.size(); ++j)
499 {
500 DeviceInfoImpl& deviceInfo = global_devices[platformInfo.deviceIDs[j]];
501 platformInfo.devices[j] = &deviceInfo;
502 }
503 }
504
505 return global_devices.size();
506}

564//This is a global singleton class used to represent a OpenCL context.
565class ContextImpl : public Context
566{
567public:
568 cl_device_id clDeviceID;
569 cl_context clContext;
570 const DeviceInfoImpl& deviceInfoImpl;
571
572protected:
573 ContextImpl(const DeviceInfoImpl& _deviceInfoImpl, cl_context context)
574 : clDeviceID(_deviceInfoImpl.device_id), clContext(context), deviceInfoImpl(_deviceInfoImpl)
575 {
576#ifdef CL_VERSION_1_2
577 if (supportsFeature(FEATURE_CL_VER_1_2))
578 {
579 openCLSafeCall(clRetainDevice(clDeviceID));
580 }
581#endif
582 openCLSafeCall(clRetainContext(clContext));
583
584 ContextImpl* old = NULL;
585 {
586 cv::AutoLock lock(getCurrentContextMutex());
587 old = currentContext;
588 currentContext = this;
589 }
590 if (old != NULL)
591 {
592 delete old;
593 }
594 }
595 ~ContextImpl()
596 {
597 CV_Assert(this != currentContext);
598
599#ifdef CL_VERSION_1_2
600#ifdef WIN32
601 // if process is on termination stage (ExitProcess was called and other threads were terminated)
602 // then disable device release because it may cause program hang
603 if (!__termination)
604#endif
605 {
606 if (supportsFeature(FEATURE_CL_VER_1_2))
607 {
608 openCLSafeCall(clReleaseDevice(clDeviceID));
609 }
610 }
611#endif
612 if (deviceInfoImpl._id < 0) // not in the global registry, so we should cleanup it
613 {
614#ifdef CL_VERSION_1_2
615 if (supportsFeature(FEATURE_CL_VER_1_2))
616 {
617 openCLSafeCall(clReleaseDevice(deviceInfoImpl.device_id));
618 }
619#endif
620 PlatformInfoImpl* platformImpl = (PlatformInfoImpl*)(deviceInfoImpl.platform);
621 delete platformImpl;
622 delete const_cast<DeviceInfoImpl*>(&deviceInfoImpl);
623 }
624 clDeviceID = NULL;
625
626#ifdef WIN32
627 // if process is on termination stage (ExitProcess was called and other threads were terminated)
628 // then disable context release because it may cause program hang
629 if (!__termination)
630#endif
631 {
632 if(clContext)
633 {
634 openCLSafeCall(clReleaseContext(clContext));
635 }
636 }
637 clContext = NULL;
638 }
639public:
640 static void setContext(const DeviceInfo* deviceInfo);
641 static void initializeContext(void* pClPlatform, void* pClContext, void* pClDevice);
642
643 bool supportsFeature(FEATURE_TYPE featureType) const;
644
645 static void cleanupContext(void);
646
647 static ContextImpl* getContext();
648private:
649 ContextImpl(const ContextImpl&); // disabled
650 ContextImpl& operator=(const ContextImpl&); // disabled
651
652 static ContextImpl* currentContext;
653};

664ContextImpl* ContextImpl::getContext()
665{
666 if (currentContext == NULL)
667 {
668 static bool defaultInitiaization = false;
669 if (!defaultInitiaization)
670 {
671 cv::AutoLock lock(getInitializationMutex());
672 try
673 {
674 if (!__initialized)
675 {
676 if (initializeOpenCLDevices() == 0)
677 {
678 CV_Error(CV_OpenCLInitError, "OpenCL not available");
679 }
680 }
681 if (!__deviceSelected)
682 {
683 if (!selectOpenCLDevice())
684 {
685 CV_Error(CV_OpenCLInitError, "Can't select OpenCL device");
686 }
687 }
688 defaultInitiaization = true;
689 }
690 catch (...)
691 {
692 defaultInitiaization = true;
693 throw;
694 }
695 }
696 CV_Assert(currentContext != NULL);
697 }
698 return currentContext;
699}

767void ContextImpl::setContext(const DeviceInfo* deviceInfo)
768{
769 CV_Assert(deviceInfo->_id >= 0); // we can't specify custom devices
770 CV_Assert(deviceInfo->_id < (int)global_devices.size());
771
772 {
773 cv::AutoLock lock(getCurrentContextMutex());
774 if (currentContext)
775 {
776 if (currentContext->deviceInfoImpl._id == deviceInfo->_id)
777 return;
778 }
779 }
780
781 DeviceInfoImpl& infoImpl = global_devices[deviceInfo->_id];
782 CV_Assert(deviceInfo == &infoImpl);
783
784 cl_int status = 0;
785 cl_context_properties cps[3] = { CL_CONTEXT_PLATFORM, (cl_context_properties)(infoImpl.platform_id), 0 };
786 cl_context clContext = clCreateContext(cps, 1, &infoImpl.device_id, NULL, NULL, &status);
787 openCLVerifyCall(status);
788
789 ContextImpl* ctx = new ContextImpl(infoImpl, clContext);
790 clReleaseContext(clContext);
791 (void)ctx;
792}

794void ContextImpl::initializeContext(void* pClPlatform, void* pClContext, void* pClDevice)
795{
796 CV_Assert(pClPlatform != NULL);
797 CV_Assert(pClContext != NULL);
798 CV_Assert(pClDevice != NULL);
799 cl_platform_id platform = *(cl_platform_id*)pClPlatform;
800 cl_context context = *(cl_context*)pClContext;
801 cl_device_id device = *(cl_device_id*)pClDevice;
802
803 PlatformInfoImpl* platformInfoImpl = new PlatformInfoImpl();
804 platformInfoImpl->init(-1, platform);
805 DeviceInfoImpl* deviceInfoImpl = new DeviceInfoImpl();
806 deviceInfoImpl->init(-1, *platformInfoImpl, device);
807
808 ContextImpl* ctx = new ContextImpl(*deviceInfoImpl, context);
809 (void)ctx;
810}

839int getOpenCLDevices(std::vector<const DeviceInfo*> &devices, int deviceType, const PlatformInfo* platform)
840{
841 if (!__initialized)
842 initializeOpenCLDevices();
843
844 devices.clear();
845
846 switch(deviceType)
847 {
848 case CVCL_DEVICE_TYPE_DEFAULT:
849 case CVCL_DEVICE_TYPE_CPU:
850 case CVCL_DEVICE_TYPE_GPU:
851 case CVCL_DEVICE_TYPE_ACCELERATOR:
852 case CVCL_DEVICE_TYPE_ALL:
853 break;
854 default:
855 return 0;
856 }
857
858 if (platform == NULL)
859 {
860 for (size_t id = 0; id < global_devices.size(); ++id)
861 {
862 DeviceInfoImpl& deviceInfo = global_devices[id];
863 if (((int)deviceInfo.deviceType & deviceType) != 0)
864 {
865 devices.push_back(&deviceInfo);
866 }
867 }
868 }
869 else
870 {
871 for (size_t id = 0; id < platform->devices.size(); ++id)
872 {
873 const DeviceInfo* deviceInfo = platform->devices[id];
874 if (((int)deviceInfo->deviceType & deviceType) == deviceType)
875 {
876 devices.push_back(deviceInfo);
877 }
878 }
879 }
880
881 return (int)devices.size();
882}
