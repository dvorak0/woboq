123cv::ocl::HOGDescriptor::HOGDescriptor(Size win_size_, Size block_size_, Size block_stride_,
124 Size cell_size_, int nbins_, double win_sigma_,
125 double threshold_L2hys_, bool gamma_correction_, int nlevels_)
126 : win_size(win_size_),
127 block_size(block_size_),
128 block_stride(block_stride_),
129 cell_size(cell_size_),
130 nbins(nbins_),
131 win_sigma(win_sigma_),
132 threshold_L2hys(threshold_L2hys_),
133 gamma_correction(gamma_correction_),
134 nlevels(nlevels_)
135{
136 CV_Assert((win_size.width - block_size.width ) % block_stride.width == 0 &&
137 (win_size.height - block_size.height) % block_stride.height == 0);
138
139 CV_Assert(block_size.width % cell_size.width == 0 &&
140 block_size.height % cell_size.height == 0);
141
142 CV_Assert(block_stride == cell_size);
143
144 CV_Assert(cell_size == Size(8, 8));
145
146 Size cells_per_block(block_size.width / cell_size.width,
147 block_size.height / cell_size.height);
148 CV_Assert(cells_per_block == Size(2, 2));
149
150 cv::Size blocks_per_win = numPartsWithin(win_size, block_size, block_stride);
151 hog::set_up_constants(nbins, block_stride.width, block_stride.height,
152 blocks_per_win.width, blocks_per_win.height);
153
154 effect_size = Size(0, 0);
155
156 if (isCpuDevice())
157 hog_device_cpu = true;
158 else
159 hog_device_cpu = false;
160
161}

188void cv::ocl::HOGDescriptor::setSVMDetector(const vector<float> &_detector)
189{
190 std::vector<float> detector_reordered(_detector.size());
191
192 size_t block_hist_size = getBlockHistogramSize();
193 cv::Size blocks_per_img = numPartsWithin(win_size, block_size, block_stride);
194
195 for (int i = 0; i < blocks_per_img.height; ++i)
196 for (int j = 0; j < blocks_per_img.width; ++j)
197 {
198 const float *src = &_detector[0] + (j * blocks_per_img.height + i) * block_hist_size;
199 float *dst = &detector_reordered[0] + (i * blocks_per_img.width + j) * block_hist_size;
200 for (size_t k = 0; k < block_hist_size; ++k)
201 dst[k] = src[k];
202 }
203
204 this->detector.upload(Mat(detector_reordered).reshape(1, 1));
205
206 size_t descriptor_size = getDescriptorSize();
207 free_coef = _detector.size() > descriptor_size ? _detector[descriptor_size] : 0;
208
209 CV_Assert(checkDetectorSize());
210}

212void cv::ocl::HOGDescriptor::init_buffer(const oclMat &img, Size win_stride)
213{
214 if (!image_scale.empty())
215 return;
216
217 if (effect_size == Size(0, 0))
218 effect_size = img.size();
219
220 grad.create(img.size(), CV_32FC2);
221 qangle.create(img.size(), hog::qangle_type);
222
223 const size_t block_hist_size = getBlockHistogramSize();
224 const Size blocks_per_img = numPartsWithin(img.size(), block_size, block_stride);
225 block_hists.create(1,
226 static_cast<int>(block_hist_size * blocks_per_img.area()) + 256, CV_32F);
227
228 Size wins_per_img = numPartsWithin(img.size(), win_size, win_stride);
229 labels.create(1, wins_per_img.area(), CV_8U);
230
231 float sigma = getWinSigma();
232 float scale = 1.f / (2.f * sigma * sigma);
233 Mat gaussian_lut(1, 512, CV_32FC1);
234 int idx = 0;
235 for(int i=-8; i<8; i++)
236 for(int j=-8; j<8; j++)
237 gaussian_lut.at<float>(idx++) = std::exp(-(j * j + i * i) * scale);
238 for(int i=-8; i<8; i++)
239 for(int j=-8; j<8; j++)
240 gaussian_lut.at<float>(idx++) = (8.f - fabs(j + 0.5f)) * (8.f - fabs(i + 0.5f)) / 64.f;
241
242 gauss_w_lut.upload(gaussian_lut);
243}

245void cv::ocl::HOGDescriptor::computeGradient(const oclMat &img, oclMat &lgrad, oclMat &lqangle)
246{
247 CV_Assert(img.type() == CV_8UC1 || img.type() == CV_8UC4);
248
249 float angleScale = (float)(nbins / CV_PI);
250 switch (img.type())
251 {
252 case CV_8UC1:
253 hog::compute_gradients_8UC1(effect_size.height, effect_size.width, img,
254 angleScale, lgrad, lqangle, gamma_correction);
255 break;
256 case CV_8UC4:
257 hog::compute_gradients_8UC4(effect_size.height, effect_size.width, img,
258 angleScale, lgrad, lqangle, gamma_correction);
259 break;
260 }
261}

264void cv::ocl::HOGDescriptor::computeBlockHistograms(const oclMat &img)
265{
266 computeGradient(img, this->grad, this->qangle);
267
268 hog::compute_hists(nbins, block_stride.width, block_stride.height, effect_size.height,
269 effect_size.width, grad, qangle, gauss_w_lut, block_hists);
270
271 hog::normalize_hists(nbins, block_stride.width, block_stride.height, effect_size.height,
272 effect_size.width, block_hists, (float)threshold_L2hys);
273}

276void cv::ocl::HOGDescriptor::getDescriptors(const oclMat &img, Size win_stride,
277 oclMat &descriptors, int descr_format)
278{
279 CV_Assert(win_stride.width % block_stride.width == 0 &&
280 win_stride.height % block_stride.height == 0);
281
282 init_buffer(img, win_stride);
283
284 computeBlockHistograms(img);
285
286 const size_t block_hist_size = getBlockHistogramSize();
287 Size blocks_per_win = numPartsWithin(win_size, block_size, block_stride);
288 Size wins_per_img = numPartsWithin(effect_size, win_size, win_stride);
289
290 descriptors.create(wins_per_img.area(),
291 static_cast<int>(blocks_per_win.area() * block_hist_size), CV_32F);
292
293 switch (descr_format)
294 {
295 case DESCR_FORMAT_ROW_BY_ROW:
296 hog::extract_descrs_by_rows(win_size.height, win_size.width,
297 block_stride.height, block_stride.width, win_stride.height, win_stride.width,
298 effect_size.height, effect_size.width, block_hists, descriptors);
299 break;
300 case DESCR_FORMAT_COL_BY_COL:
301 hog::extract_descrs_by_cols(win_size.height, win_size.width,
302 block_stride.height, block_stride.width, win_stride.height, win_stride.width,
303 effect_size.height, effect_size.width, block_hists, descriptors);
304 break;
305 default:
306 CV_Error(CV_StsBadArg, "Unknown descriptor format");
307 }
308}

311void cv::ocl::HOGDescriptor::detect(const oclMat &img, vector<Point> &hits,
312 double hit_threshold, Size win_stride, Size padding)
313{
314 CV_Assert(img.type() == CV_8UC1 || img.type() == CV_8UC4);
315 CV_Assert(padding == Size(0, 0));
316
317 hits.clear();
318 if (detector.empty())
319 return;
320
321 if (win_stride == Size())
322 win_stride = block_stride;
323 else
324 CV_Assert(win_stride.width % block_stride.width == 0 &&
325 win_stride.height % block_stride.height == 0);
326 init_buffer(img, win_stride);
327
328 computeBlockHistograms(img);
329
330 hog::classify_hists(win_size.height, win_size.width, block_stride.height,
331 block_stride.width, win_stride.height, win_stride.width,
332 effect_size.height, effect_size.width, block_hists, detector,
333 (float)free_coef, (float)hit_threshold, labels);
334
335 labels.download(labels_host);
336 unsigned char *vec = labels_host.ptr();
337 Size wins_per_img = numPartsWithin(effect_size, win_size, win_stride);
338 for (int i = 0; i < wins_per_img.area(); i++)
339 {
340 int y = i / wins_per_img.width;
341 int x = i - wins_per_img.width * y;
342 if (vec[i])
343 hits.push_back(Point(x * win_stride.width, y * win_stride.height));
344 }
345}

349void cv::ocl::HOGDescriptor::detectMultiScale(const oclMat &img, vector<Rect> &found_locations,
350 double hit_threshold, Size win_stride, Size padding,
351 double scale0, int group_threshold)
352{
353 CV_Assert(img.type() == CV_8UC1 || img.type() == CV_8UC4);
354 CV_Assert(scale0 > 1);
355
356 vector<double> level_scale;
357 double scale = 1.;
358 int levels = 0;
359
360 for (levels = 0; levels < nlevels; levels++)
361 {
362 level_scale.push_back(scale);
363 if (cvRound(img.cols / scale) < win_size.width ||
364 cvRound(img.rows / scale) < win_size.height || scale0 <= 1)
365 break;
366 scale *= scale0;
367 }
368 levels = std::max(levels, 1);
369 level_scale.resize(levels);
370
371 std::vector<Rect> all_candidates;
372 vector<Point> locations;
373
374 if (win_stride == Size())
375 win_stride = block_stride;
376 else
377 CV_Assert(win_stride.width % block_stride.width == 0 &&
378 win_stride.height % block_stride.height == 0);
379 init_buffer(img, win_stride);
380 image_scale.create(img.size(), img.type());
381
382 for (size_t i = 0; i < level_scale.size(); i++)
383 {
384 scale = level_scale[i];
385 effect_size = Size(cvRound(img.cols / scale), cvRound(img.rows / scale));
386 if (effect_size == img.size())
387 {
388 detect(img, locations, hit_threshold, win_stride, padding);
389 }
390 else
391 {
392 resize(img, image_scale, effect_size);
393 detect(image_scale, locations, hit_threshold, win_stride, padding);
394 }
395 Size scaled_win_size(cvRound(win_size.width * scale),
396 cvRound(win_size.height * scale));
397 for (size_t j = 0; j < locations.size(); j++)
398 all_candidates.push_back(Rect(Point2d((CvPoint)locations[j]) * scale,
399 scaled_win_size));
400 }
401
402 found_locations.assign(all_candidates.begin(), all_candidates.end());
403 groupRectangles(found_locations, group_threshold, 0.2);
404}
