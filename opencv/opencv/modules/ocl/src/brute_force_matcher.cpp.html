69template < int BLOCK_SIZE, int MAX_DESC_LEN/*, typename Mask*/ >
70void matchUnrolledCached(const oclMat &query, const oclMat &train, const oclMat &/*mask*/,
71 const oclMat &trainIdx, const oclMat &distance, int distType)
72{
73 cv::ocl::Context *ctx = query.clCxt;
74 size_t globalSize[] = {((size_t)query.rows + BLOCK_SIZE - 1) / BLOCK_SIZE * BLOCK_SIZE, BLOCK_SIZE, 1};
75 size_t localSize[] = {BLOCK_SIZE, BLOCK_SIZE, 1};
76 const size_t smemSize = (BLOCK_SIZE * (MAX_DESC_LEN >= 2 * BLOCK_SIZE ? MAX_DESC_LEN : 2 * BLOCK_SIZE) + BLOCK_SIZE * BLOCK_SIZE) * sizeof(int);
77 int block_size = BLOCK_SIZE;
78 int m_size = MAX_DESC_LEN;
79 vector< pair<size_t, const void *> > args;
80
81 char opt [OPT_SIZE] = "";
82 sprintf(opt,
83 "-D T=%s -D DIST_TYPE=%d -D BLOCK_SIZE=%d -D MAX_DESC_LEN=%d",
84 T_ARR[query.depth()], distType, block_size, m_size);
85
86 if(globalSize[0] != 0)
87 {
88 args.push_back( make_pair( sizeof(cl_mem), (void *)&query.data ));
89 args.push_back( make_pair( sizeof(cl_mem), (void *)&train.data ));
90 //args.push_back( make_pair( sizeof(cl_mem), (void *)&mask.data ));
91 args.push_back( make_pair( sizeof(cl_mem), (void *)&trainIdx.data ));
92 args.push_back( make_pair( sizeof(cl_mem), (void *)&distance.data ));
93 args.push_back( make_pair( smemSize, (void *)NULL));
94 args.push_back( make_pair( sizeof(cl_int), (void *)&query.rows ));
95 args.push_back( make_pair( sizeof(cl_int), (void *)&query.cols ));
96 args.push_back( make_pair( sizeof(cl_int), (void *)&train.rows ));
97 args.push_back( make_pair( sizeof(cl_int), (void *)&train.cols ));
98 args.push_back( make_pair( sizeof(cl_int), (void *)&query.step ));
99
100 std::string kernelName = "BruteForceMatch_UnrollMatch";
101
102 openCLExecuteKernel(ctx, &brute_force_match, kernelName, globalSize, localSize, args, -1, -1, opt);
103 }
104}

236static void matchDispatcher(const oclMat &query, const oclMat &train, const oclMat &mask,
237 const oclMat &trainIdx, const oclMat &distance, int distType)
238{
239 const oclMat zeroMask;
240 const oclMat &tempMask = mask.data ? mask : zeroMask;
241 bool is_cpu = isCpuDevice();
242 if (query.cols <= 64)
243 {
244 matchUnrolledCached<16, 64>(query, train, tempMask, trainIdx, distance, distType);
245 }
246 else if (query.cols <= 128 && !is_cpu)
247 {
248 matchUnrolledCached<16, 128>(query, train, tempMask, trainIdx, distance, distType);
249 }
250 else
251 {
252 match<16>(query, train, tempMask, trainIdx, distance, distType);
253 }
254}

940 {
941 oclMat trainCollection;

946 oclMat trainIdx, imgIdx, distance;

951 else
952 {
953 if (query.empty() || empty())
954 return;
955
956 vector< vector<DMatch> > curMatches;
957 vector<DMatch> temp;
958 temp.reserve(2 * k);

969 vector<DMatch> &localMatch = curMatches[queryIdx];
970 vector<DMatch> &globalMatch = matches[queryIdx];
971
972 for_each(localMatch.begin(), localMatch.end(), ImgIdxSetter(static_cast<int>(imgIdx)));
973
974 temp.clear();
975 merge(globalMatch.begin(), globalMatch.end(), localMatch.begin(), localMatch.end(), back_inserter(temp));
976
977 globalMatch.clear();
978 const size_t count = std::min((size_t)k, temp.size());
979 copy(temp.begin(), temp.begin() + count, back_inserter(globalMatch));
980 }
981 }
982
983 if (compactResult)
984 {
985 vector< vector<DMatch> >::iterator new_end = remove_if(matches.begin(), matches.end(), mem_fun_ref(&vector<DMatch>::empty));
