174 void error(const char *error_string, const char *file, const int line, const char *func)
175 {
176 int code = CV_GpuApiCallError;
177
178 if (std::uncaught_exception())
179 {
180 const char *errorStr = cvErrorStr(code);
181 const char *function = func ? func : "unknown function";
182
183 std::cerr << "OpenCV Error: " << errorStr << "(" << error_string << ") in " << function << ", file " << file << ", line " << line;
184 std::cerr.flush();
185 }
186 else
187 cv::error( cv::Exception(code, error_string, func, file, line) );
188 }
189 }
