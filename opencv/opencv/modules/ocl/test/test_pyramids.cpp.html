47#include "test_precomp.hpp"
48#include <iomanip>

50#ifdef HAVE_OPENCL
51
52using namespace cv;
53using namespace testing;
54using namespace std;
55
56PARAM_TEST_CASE(PyrBase, MatDepth, Channels)
57{
58 int depth;
59 int channels;
60
61 Mat dst_cpu;
62 ocl::oclMat gdst;
63
64 virtual void SetUp()
65 {
66 depth = GET_PARAM(0);
67 channels = GET_PARAM(1);
68 }
69};
70
71/////////////////////// PyrDown //////////////////////////
72
73typedef PyrBase PyrDown;
74
75OCL_TEST_P(PyrDown, Mat)
76{
77 for (int j = 0; j < LOOP_TIMES; j++)
78 {
79 Size size(MWIDTH, MHEIGHT);
80 Mat src = randomMat(size, CV_MAKETYPE(depth, channels), 0, 255);
81 ocl::oclMat gsrc(src);
82
83 pyrDown(src, dst_cpu);
84 ocl::pyrDown(gsrc, gdst);
85
86 EXPECT_MAT_NEAR(dst_cpu, Mat(gdst), depth == CV_32F ? 1e-4f : 1.0f);
87 }
88}
89
90INSTANTIATE_TEST_CASE_P(OCL_ImgProc, PyrDown, Combine(
91 Values(CV_8U, CV_16U, CV_16S, CV_32F),
92 Values(1, 3, 4)));
93
94/////////////////////// PyrUp //////////////////////////
95
96typedef PyrBase PyrUp;
97
98OCL_TEST_P(PyrUp, Accuracy)
99{
100 for (int j = 0; j < LOOP_TIMES; j++)
101 {
102 Size size(MWIDTH, MHEIGHT);
103 Mat src = randomMat(size, CV_MAKETYPE(depth, channels), 0, 255);
104 ocl::oclMat gsrc(src);
105
106 pyrUp(src, dst_cpu);
107 ocl::pyrUp(gsrc, gdst);
108
109 EXPECT_MAT_NEAR(dst_cpu, Mat(gdst), (depth == CV_32F ? 1e-4f : 1.0));
110 }
111}
112
113
114INSTANTIATE_TEST_CASE_P(OCL_ImgProc, PyrUp, Combine(
115 Values(CV_8U, CV_16U, CV_16S, CV_32F),
116 Values(1, 3, 4)));
117#endif // HAVE_OPENCL
