62typedef struct
63{
64 short x;
65 short y;

68COOR do_meanShift(int x0, int y0, uchar *sptr, uchar *dptr, int sstep, Size size, int sp, int sr, int maxIter, float eps, int *tab)
69{
70
71 int isr2 = sr * sr;
72 int c0, c1, c2, c3;
73 int iter;
74 uchar *ptr = NULL;
75 uchar *pstart = NULL;
76 int revx = 0, revy = 0;
77 c0 = sptr[0];
78 c1 = sptr[1];
79 c2 = sptr[2];
80 c3 = sptr[3];
81 // iterate meanshift procedure
82 for(iter = 0; iter < maxIter; iter++ )
83 {
84 int count = 0;
85 int s0 = 0, s1 = 0, s2 = 0, sx = 0, sy = 0;
86
87 //mean shift: process pixels in window (p-sigmaSp)x(p+sigmaSp)
88 int minx = x0 - sp;
89 int miny = y0 - sp;
90 int maxx = x0 + sp;
91 int maxy = y0 + sp;
92
93 //deal with the image boundary
94 if(minx < 0) minx = 0;
95 if(miny < 0) miny = 0;
96 if(maxx >= size.width) maxx = size.width - 1;
97 if(maxy >= size.height) maxy = size.height - 1;
98 if(iter == 0)
99 {
100 pstart = sptr;
101 }
102 else
103 {
104 pstart = pstart + revy * sstep + (revx << 2); //point to the new position
105 }
106 ptr = pstart;
107 ptr = ptr + (miny - y0) * sstep + ((minx - x0) << 2); //point to the start in the row
108
109 for( int y = miny; y <= maxy; y++, ptr += sstep - ((maxx - minx + 1) << 2))
110 {
111 int rowCount = 0;
112 int x = minx;
113#if CV_ENABLE_UNROLLED
114 for( ; x + 4 <= maxx; x += 4, ptr += 16)
115 {
116 int t0, t1, t2;
117 t0 = ptr[0], t1 = ptr[1], t2 = ptr[2];
118 if(tab[t0 - c0 + 255] + tab[t1 - c1 + 255] + tab[t2 - c2 + 255] <= isr2)
119 {
120 s0 += t0;
121 s1 += t1;
122 s2 += t2;
123 sx += x;
124 rowCount++;
125 }
126 t0 = ptr[4], t1 = ptr[5], t2 = ptr[6];
127 if(tab[t0 - c0 + 255] + tab[t1 - c1 + 255] + tab[t2 - c2 + 255] <= isr2)
128 {
129 s0 += t0;
130 s1 += t1;
131 s2 += t2;
132 sx += x + 1;
133 rowCount++;
134 }
135 t0 = ptr[8], t1 = ptr[9], t2 = ptr[10];
136 if(tab[t0 - c0 + 255] + tab[t1 - c1 + 255] + tab[t2 - c2 + 255] <= isr2)
137 {
138 s0 += t0;
139 s1 += t1;
140 s2 += t2;
141 sx += x + 2;
142 rowCount++;
143 }
144 t0 = ptr[12], t1 = ptr[13], t2 = ptr[14];
145 if(tab[t0 - c0 + 255] + tab[t1 - c1 + 255] + tab[t2 - c2 + 255] <= isr2)
146 {
147 s0 += t0;
148 s1 += t1;
149 s2 += t2;
150 sx += x + 3;
151 rowCount++;
152 }
153 }
154#endif
155 for(; x <= maxx; x++, ptr += 4)
156 {
157 int t0 = ptr[0], t1 = ptr[1], t2 = ptr[2];
158 if(tab[t0 - c0 + 255] + tab[t1 - c1 + 255] + tab[t2 - c2 + 255] <= isr2)
159 {
160 s0 += t0;
161 s1 += t1;
162 s2 += t2;
163 sx += x;
164 rowCount++;
165 }
166 }
167 if(rowCount == 0)
168 continue;
169 count += rowCount;
170 sy += y * rowCount;
171 }
172
173 if( count == 0 )
174 break;
175
176 int x1 = sx / count;
177 int y1 = sy / count;
178 s0 = s0 / count;
179 s1 = s1 / count;
180 s2 = s2 / count;
181
182 bool stopFlag = (x0 == x1 && y0 == y1) || (abs(x1 - x0) + abs(y1 - y0) +
183 tab[s0 - c0 + 255] + tab[s1 - c1 + 255] + tab[s2 - c2 + 255] <= eps);
184
185 //revise the pointer corresponding to the new (y0,x0)
186 revx = x1 - x0;
187 revy = y1 - y0;
188
189 x0 = x1;
190 y0 = y1;
191 c0 = s0;
192 c1 = s1;
193 c2 = s2;
194
195 if( stopFlag )
196 break;
197 } //for iter
198
199 dptr[0] = (uchar)c0;
200 dptr[1] = (uchar)c1;
201 dptr[2] = (uchar)c2;
202 dptr[3] = (uchar)c3;
203
204 COOR coor;
205 coor.x = (short)x0;
206 coor.y = (short)y0;
207 return coor;
208}

210void meanShiftFiltering_(const Mat &src_roi, Mat &dst_roi, int sp, int sr, TermCriteria crit)
211{
212 if( src_roi.empty() )
213 CV_Error( CV_StsBadArg, "The input image is empty" );
214
215 if( src_roi.depth() != CV_8U || src_roi.channels() != 4 )
216 CV_Error( CV_StsUnsupportedFormat, "Only 8-bit, 4-channel images are supported" );
217
218 CV_Assert( (src_roi.cols == dst_roi.cols) && (src_roi.rows == dst_roi.rows) );
219 CV_Assert( !(dst_roi.step & 0x3) );
220
221 if( !(crit.type & TermCriteria::MAX_ITER) )
222 crit.maxCount = 5;
223 int maxIter = std::min(std::max(crit.maxCount, 1), 100);
224 float eps;
225 if( !(crit.type & TermCriteria::EPS) )
226 eps = 1.f;
227 eps = (float)std::max(crit.epsilon, 0.0);
228
229 int tab[512];
230 for(int i = 0; i < 512; i++)
231 tab[i] = (i - 255) * (i - 255);
232 uchar *sptr = src_roi.data;
233 uchar *dptr = dst_roi.data;
234 int sstep = (int)src_roi.step;
235 int dstep = (int)dst_roi.step;
236 Size size = src_roi.size();
237
238 for(int i = 0; i < size.height; i++, sptr += sstep - (size.width << 2),
239 dptr += dstep - (size.width << 2))
240 {
241 for(int j = 0; j < size.width; j++, sptr += 4, dptr += 4)
242 {
243 do_meanShift(j, i, sptr, dptr, sstep, size, sp, sr, maxIter, eps, tab);
244 }
245 }
246}

248void meanShiftProc_(const Mat &src_roi, Mat &dst_roi, Mat &dstCoor_roi, int sp, int sr, TermCriteria crit)
249{
250 if( src_roi.empty() )
251 CV_Error( CV_StsBadArg, "The input image is empty" );
252 if( src_roi.depth() != CV_8U || src_roi.channels() != 4 )
253 CV_Error( CV_StsUnsupportedFormat, "Only 8-bit, 4-channel images are supported" );
254 CV_Assert( (src_roi.cols == dst_roi.cols) && (src_roi.rows == dst_roi.rows) &&
255 (src_roi.cols == dstCoor_roi.cols) && (src_roi.rows == dstCoor_roi.rows));
256 CV_Assert( !(dstCoor_roi.step & 0x3) );
257
258 if( !(crit.type & TermCriteria::MAX_ITER) )
259 crit.maxCount = 5;
260 int maxIter = std::min(std::max(crit.maxCount, 1), 100);
261 float eps;
262 if( !(crit.type & TermCriteria::EPS) )
263 eps = 1.f;
264 eps = (float)std::max(crit.epsilon, 0.0);
265
266 int tab[512];
267 for(int i = 0; i < 512; i++)
268 tab[i] = (i - 255) * (i - 255);
269 uchar *sptr = src_roi.data;
270 uchar *dptr = dst_roi.data;
271 short *dCoorptr = (short *)dstCoor_roi.data;
272 int sstep = (int)src_roi.step;
273 int dstep = (int)dst_roi.step;
274 int dCoorstep = (int)dstCoor_roi.step >> 1;
275 Size size = src_roi.size();
276
277 for(int i = 0; i < size.height; i++, sptr += sstep - (size.width << 2),
278 dptr += dstep - (size.width << 2), dCoorptr += dCoorstep - (size.width << 1))
279 {
280 for(int j = 0; j < size.width; j++, sptr += 4, dptr += 4, dCoorptr += 2)
281 {
282 *((COOR *)dCoorptr) = do_meanShift(j, i, sptr, dptr, sstep, size, sp, sr, maxIter, eps, tab);
283 }
284 }
285
286}

290PARAM_TEST_CASE(meanShiftTestBase, MatType, MatType, int, int, TermCriteria, bool)
291{
292 int type, typeCoor;
293 int sp, sr;
294 TermCriteria crit;
295 bool useRoi;
296
297 // src mat
298 Mat src, src_roi;
299 Mat dst, dst_roi;
300 Mat dstCoor, dstCoor_roi;
301
302 // ocl dst mat
303 ocl::oclMat gsrc, gsrc_roi;
304 ocl::oclMat gdst, gdst_roi;
305 ocl::oclMat gdstCoor, gdstCoor_roi;
306
307 virtual void SetUp()
308 {
309 type = GET_PARAM(0);
310 typeCoor = GET_PARAM(1);
311 sp = GET_PARAM(2);
312 sr = GET_PARAM(3);
313 crit = GET_PARAM(4);
314 useRoi = GET_PARAM(5);
315 }
316
317 void random_roi()
318 {
319 Size roiSize = randomSize(1, MAX_VALUE);
320 Border srcBorder = randomBorder(0, useRoi ? MAX_VALUE : 0);
321 randomSubMat(src, src_roi, roiSize, srcBorder, type, 5, 256);
322 generateOclMat(gsrc, gsrc_roi, src, roiSize, srcBorder);
323
324 Border dstBorder = randomBorder(0, useRoi ? MAX_VALUE : 0);
325 randomSubMat(dst, dst_roi, roiSize, dstBorder, type, 5, 256);
326 generateOclMat(gdst, gdst_roi, dst, roiSize, dstBorder);
327
328 randomSubMat(dstCoor, dstCoor_roi, roiSize, dstBorder, typeCoor, 5, 256);
329 generateOclMat(gdstCoor, gdstCoor_roi, dstCoor, roiSize, dstBorder);
330 }
331
332 void Near(double threshold = 0.0)
333 {
334 Mat whole, roi;
335 gdst.download(whole);
336 gdst_roi.download(roi);
337
338 EXPECT_MAT_NEAR(dst, whole, threshold);
339 EXPECT_MAT_NEAR(dst_roi, roi, threshold);
340 }
341
342 void Near1(double threshold = 0.0)
343 {
344 Mat whole, roi;
345 gdstCoor.download(whole);
346 gdstCoor_roi.download(roi);
347
348 EXPECT_MAT_NEAR(dstCoor, whole, threshold);
349 EXPECT_MAT_NEAR(dstCoor_roi, roi, threshold);
350 }
351};
