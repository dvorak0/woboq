58#include "opencv2/ocl/private/opencl_dumpinfo.hpp"

60int LOOP_TIMES = 1;
61
62void readLoopTimes(int argc, char ** argv)
63{
64 const char * const command_line_keys =
65 "{ |test_loop_times |1 |count of iterations per each test}"
66 "{h |help |false |print help info}";
67
68 cv::CommandLineParser parser(argc, argv, command_line_keys);
69 if (parser.get<bool>("help"))
70 {
71 std::cout << "\nAvailable options besides google test option: \n";
72 parser.printParams();
73 }
74
75 LOOP_TIMES = parser.get<int>("test_loop_times");
76 CV_Assert(LOOP_TIMES > 0);
77}
