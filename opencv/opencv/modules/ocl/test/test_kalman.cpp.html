57PARAM_TEST_CASE(Kalman, int, int)
58{
59 int size_;
60 int iteration;
61 virtual void SetUp()
62 {
63 size_ = GET_PARAM(0);
64 iteration = GET_PARAM(1);
65 }
66};
67
68OCL_TEST_P(Kalman, Accuracy)
69{
70 const int Dim = size_;
71 const int Steps = iteration;
72 const double max_init = 1;
73 const double max_noise = 0.1;
74
75 Mat sample_mat(Dim, 1, CV_32F), temp_mat;
76 oclMat Sample(Dim, 1, CV_32F);
77 oclMat Temp(Dim, 1, CV_32F);
78 Mat Temp_cpu(Dim, 1, CV_32F);
79
80 Size size(Sample.cols, Sample.rows);
81
82 sample_mat = randomMat(size, Sample.type(), -max_init, max_init, false);
83 Sample.upload(sample_mat);
84
85 //ocl start
86 cv::ocl::KalmanFilter kalman_filter_ocl;
87 kalman_filter_ocl.init(Dim, Dim);
88
89 cv::ocl::setIdentity(kalman_filter_ocl.errorCovPre, 1);
90 cv::ocl::setIdentity(kalman_filter_ocl.measurementMatrix, 1);
91 cv::ocl::setIdentity(kalman_filter_ocl.errorCovPost, 1);
92
93 kalman_filter_ocl.measurementNoiseCov.setTo(Scalar::all(0));
94 kalman_filter_ocl.statePre.setTo(Scalar::all(0));
95 kalman_filter_ocl.statePost.setTo(Scalar::all(0));
96
97 kalman_filter_ocl.correct(Sample);
98 //ocl end
99
100 //cpu start
101 cv::KalmanFilter kalman_filter_cpu;
102
103 kalman_filter_cpu.init(Dim, Dim);
104
105 cv::setIdentity(kalman_filter_cpu.errorCovPre, 1);
106 cv::setIdentity(kalman_filter_cpu.measurementMatrix, 1);
107 cv::setIdentity(kalman_filter_cpu.errorCovPost, 1);
108
109 kalman_filter_cpu.measurementNoiseCov.setTo(Scalar::all(0));
110 kalman_filter_cpu.statePre.setTo(Scalar::all(0));
111 kalman_filter_cpu.statePost.setTo(Scalar::all(0));
112
113 kalman_filter_cpu.correct(sample_mat);
114 //cpu end
115 //test begin
116 for(int i = 0; i<Steps; i++)
117 {
118 kalman_filter_ocl.predict();
119 kalman_filter_cpu.predict();
120
121 cv::gemm(kalman_filter_cpu.transitionMatrix, sample_mat, 1, cv::Mat(), 0, Temp_cpu);
122
123 Size size1(Temp.cols, Temp.rows);
124 Mat temp = randomMat(size1, Temp.type(), 0, 0xffff, false);
125
126
127 cv::multiply(2, temp, temp);
128
129 cv::subtract(temp, 1, temp);
130
131 cv::multiply(max_noise, temp, temp);
132
133 cv::add(temp, Temp_cpu, Temp_cpu);
134
135 Temp.upload(Temp_cpu);
136 Temp.copyTo(Sample);
137 Temp_cpu.copyTo(sample_mat);
138
139 kalman_filter_ocl.correct(Temp);
140 kalman_filter_cpu.correct(Temp_cpu);
141 }
142 //test end
143 EXPECT_MAT_NEAR(kalman_filter_cpu.statePost, kalman_filter_ocl.statePost, 0);
144}
