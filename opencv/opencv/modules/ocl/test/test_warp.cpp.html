68{
69 int type, interpolation;
70 Size dsize;
71 bool useRoi, mapInverse;
72
73 Mat src, dst_whole, src_roi, dst_roi;

76 virtual void SetUp()
77 {
78 type = GET_PARAM(0);
79 interpolation = GET_PARAM(1);
80 mapInverse = GET_PARAM(2);

82

87 void random_roi()
88 {
89 dsize = randomSize(1, MAX_VALUE);
90
91 Size roiSize = randomSize(1, MAX_VALUE);
92 Border srcBorder = randomBorder(0, useRoi ? MAX_VALUE : 0);

97
98 generateOclMat(gsrc_whole, gsrc_roi, src, roiSize, srcBorder);
99 generateOclMat(gdst_whole, gdst_roi, dst_whole, dsize, dstBorder);
100 }

103 {
104 Mat whole, roi;

210static void buildWarpPerspectiveMaps(const Mat &M, bool inverse, Size dsize, Mat &xmap, Mat &ymap)
211{
212 CV_Assert(M.rows == 3 && M.cols == 3);
213 CV_Assert(dsize.area() > 0);
214
215 xmap.create(dsize, CV_32FC1);
216 ymap.create(dsize, CV_32FC1);
217
218 float coeffs[3 * 3];
219 Mat coeffsMat(3, 3, CV_32F, (void *)coeffs);
220
221 if (inverse)
222 M.convertTo(coeffsMat, coeffsMat.type());
223 else
224 {
225 cv::Mat iM;
226 invert(M, iM);
227 iM.convertTo(coeffsMat, coeffsMat.type());
228 }
229
230 for (int y = 0; y < dsize.height; ++y)
231 {
232 float * const xmap_ptr = xmap.ptr<float>(y);
233 float * const ymap_ptr = ymap.ptr<float>(y);
234
235 for (int x = 0; x < dsize.width; ++x)
236 {
237 float coeff = 1.0f / (x * coeffs[6] + y * coeffs[7] + coeffs[8]);
238 xmap_ptr[x] = (x * coeffs[0] + y * coeffs[1] + coeffs[2]) * coeff;
239 ymap_ptr[x] = (x * coeffs[3] + y * coeffs[4] + coeffs[5]) * coeff;
240 }
241 }
242}

244OCL_TEST_P(BuildWarpPerspectiveMaps, Mat)
245{
246 for (int j = 0; j < LOOP_TIMES; j++)
247 {
248 random_roi();
249
250 float cols = static_cast<float>(MAX_VALUE), rows = static_cast<float>(MAX_VALUE);
251 float cols2 = cols / 2.0f, rows2 = rows / 2.0f;
252 Point2f sp[] = { Point2f(0.0f, 0.0f), Point2f(cols, 0.0f), Point2f(0.0f, rows), Point2f(cols, rows) };
253 Point2f dp[] = { Point2f(rng.uniform(0.0f, cols2), rng.uniform(0.0f, rows2)),
254 Point2f(rng.uniform(cols2, cols), rng.uniform(0.0f, rows2)),
255 Point2f(rng.uniform(0.0f, cols2), rng.uniform(rows2, rows)),
256 Point2f(rng.uniform(cols2, cols), rng.uniform(rows2, rows)) };
257 Mat M = getPerspectiveTransform(sp, dp);
258
259 buildWarpPerspectiveMaps(M, mapInverse, dsize, xmap_roi, ymap_roi);
260 ocl::buildWarpPerspectiveMaps(M, mapInverse, dsize, gxmap_roi, gymap_roi);
261
262 Near(5e-3);
263 Near1(5e-3);
264 }

270PARAM_TEST_CASE(Remap, MatDepth, Channels, pair<MatType, MatType>, Border, bool)
271{
272 int srcType, map1Type, map2Type;
273 int borderType;
274 bool useRoi;
275
276 Scalar val;
277
278 Mat src, src_roi;
279 Mat dst, dst_roi;
280 Mat map1, map1_roi;
281 Mat map2, map2_roi;
282
283 // ocl mat with roi
284 ocl::oclMat gsrc, gsrc_roi;
285 ocl::oclMat gdst, gdst_roi;
286 ocl::oclMat gmap1, gmap1_roi;
287 ocl::oclMat gmap2, gmap2_roi;
288
289 virtual void SetUp()
290 {
291 srcType = CV_MAKE_TYPE(GET_PARAM(0), GET_PARAM(1));
292 map1Type = GET_PARAM(2).first;
293 map2Type = GET_PARAM(2).second;
294 borderType = GET_PARAM(3);
295 useRoi = GET_PARAM(4);
296 }
297
298 void random_roi()
299 {
300 val = randomScalar(-MAX_VALUE, MAX_VALUE);
301 Size srcROISize = randomSize(1, MAX_VALUE);
302 Size dstROISize = randomSize(1, MAX_VALUE);
303
304 Border srcBorder = randomBorder(0, useRoi ? MAX_VALUE : 0);
305 randomSubMat(src, src_roi, srcROISize, srcBorder, srcType, 5, 256);
306
307 Border dstBorder = randomBorder(0, useRoi ? MAX_VALUE : 0);
308 randomSubMat(dst, dst_roi, dstROISize, dstBorder, srcType, -MAX_VALUE, MAX_VALUE);
309
310 int mapMaxValue = MAX_VALUE << 2;
311 Border map1Border = randomBorder(0, useRoi ? MAX_VALUE : 0);
312 randomSubMat(map1, map1_roi, dstROISize, map1Border, map1Type, -mapMaxValue, mapMaxValue);
313
314 Border map2Border = randomBorder(0, useRoi ? MAX_VALUE : 0);
315 if (map2Type != noType)
316 {
317 int mapMinValue = -mapMaxValue;
318 if (map2Type == CV_16UC1 || map2Type == CV_16SC1)
319 mapMinValue = 0, mapMaxValue = INTER_TAB_SIZE2;
320 randomSubMat(map2, map2_roi, dstROISize, map2Border, map2Type, mapMinValue, mapMaxValue);
321 }
322
323 generateOclMat(gsrc, gsrc_roi, src, srcROISize, srcBorder);
324 generateOclMat(gdst, gdst_roi, dst, dstROISize, dstBorder);
325 generateOclMat(gmap1, gmap1_roi, map1, dstROISize, map1Border);
326 if (noType != map2Type)
327 generateOclMat(gmap2, gmap2_roi, map2, dstROISize, map2Border);
328 }
329
330 void Near(double threshold = 0.0)
331 {
332 Mat whole, roi;
333 gdst.download(whole);
334 gdst_roi.download(roi);
335
336 EXPECT_MAT_NEAR(dst, whole, threshold);
337 EXPECT_MAT_NEAR(dst_roi, roi, threshold);
338 }
339};

374PARAM_TEST_CASE(Resize, MatType, double, double, Interpolation, bool)
375{
376 int type, interpolation;
377 double fx, fy;
378 bool useRoi;
379
380 Mat src, dst_whole, src_roi, dst_roi;
381 ocl::oclMat gsrc_whole, gsrc_roi, gdst_whole, gdst_roi;
382
383 virtual void SetUp()
384 {
385 type = GET_PARAM(0);
386 fx = GET_PARAM(1);
387 fy = GET_PARAM(2);
388 interpolation = GET_PARAM(3);
389 useRoi = GET_PARAM(4);
390 }
391
392 void random_roi()
393 {
394 CV_Assert(fx > 0 && fy > 0);
395
396 Size srcRoiSize = randomSize(1, MAX_VALUE), dstRoiSize;
397 dstRoiSize.width = cvRound(srcRoiSize.width * fx);
398 dstRoiSize.height = cvRound(srcRoiSize.height * fy);
399
400 if (dstRoiSize.area() == 0)
401 return random_roi();
402
403 Border srcBorder = randomBorder(0, useRoi ? MAX_VALUE : 0);
404 randomSubMat(src, src_roi, srcRoiSize, srcBorder, type, -MAX_VALUE, MAX_VALUE);
405
406 Border dstBorder = randomBorder(0, useRoi ? MAX_VALUE : 0);
407 randomSubMat(dst_whole, dst_roi, dstRoiSize, dstBorder, type, -MAX_VALUE, MAX_VALUE);
408
409 generateOclMat(gsrc_whole, gsrc_roi, src, srcRoiSize, srcBorder);
410 generateOclMat(gdst_whole, gdst_roi, dst_whole, dstRoiSize, dstBorder);
411 }
412
413 void Near(double threshold = 0.0)
414 {
415 Mat whole, roi;
416 gdst_whole.download(whole);
417 gdst_roi.download(roi);
418
419 EXPECT_MAT_NEAR(dst_whole, whole, threshold);
420 EXPECT_MAT_NEAR(dst_roi, roi, threshold);
421 }
422};
423
424OCL_TEST_P(Resize, Mat)
425{
426 for (int j = 0; j < LOOP_TIMES; j++)
427 {
428 random_roi();
429
430 cv::resize(src_roi, dst_roi, Size(), fx, fy, interpolation);
431 ocl::resize(gsrc_roi, gdst_roi, Size(), fx, fy, interpolation);
432
433 Near(1.0);
434 }
435}
