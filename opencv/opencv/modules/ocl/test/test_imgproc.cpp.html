64PARAM_TEST_CASE(ImgprocTestBase, MatType,
65 int, // blockSize
66 int, // border type
67 bool) // roi or not
68{
69 int type, borderType, blockSize;
70 bool useRoi;
71
72 Mat src, dst_whole, src_roi, dst_roi;
73 ocl::oclMat gsrc_whole, gsrc_roi, gdst_whole, gdst_roi;
74
75 virtual void SetUp()
76 {
77 type = GET_PARAM(0);
78 blockSize = GET_PARAM(1);
79 borderType = GET_PARAM(2);
80 useRoi = GET_PARAM(3);
81 }
82
83 virtual void random_roi()
84 {
85 Size roiSize = randomSize(1, MAX_VALUE);
86 Border srcBorder = randomBorder(0, useRoi ? MAX_VALUE : 0);
87 randomSubMat(src, src_roi, roiSize, srcBorder, type, 5, 256);
88
89 Border dstBorder = randomBorder(0, useRoi ? MAX_VALUE : 0);
90 randomSubMat(dst_whole, dst_roi, roiSize, dstBorder, type, 5, 16);
91
92 generateOclMat(gsrc_whole, gsrc_roi, src, roiSize, srcBorder);
93 generateOclMat(gdst_whole, gdst_roi, dst_whole, roiSize, dstBorder);
94 }
95
96 void Near(double threshold = 0.0, bool relative = false)
97 {
98 Mat roi, whole;
99 gdst_whole.download(whole);
100 gdst_roi.download(roi);
101
102 if (relative)
103 {
104 EXPECT_MAT_NEAR_RELATIVE(dst_whole, whole, threshold);
105 EXPECT_MAT_NEAR_RELATIVE(dst_roi, roi, threshold);
106 }
107 else
108 {
109 EXPECT_MAT_NEAR(dst_whole, whole, threshold);
110 EXPECT_MAT_NEAR(dst_roi, roi, threshold);
111 }
112 }
113};

117PARAM_TEST_CASE(CopyMakeBorder, MatDepth, // depth
118 Channels, // channels
119 bool, // isolated or not
120 Border, // border type
121 bool) // roi or not
122{
123 int type, borderType;
124 bool useRoi;
125
126 Border border;
127 Scalar val;
128
129 Mat src, dst_whole, src_roi, dst_roi;
130 ocl::oclMat gsrc_whole, gsrc_roi, gdst_whole, gdst_roi;
131
132 virtual void SetUp()
133 {
134 type = CV_MAKE_TYPE(GET_PARAM(0), GET_PARAM(1));
135 borderType = GET_PARAM(3);
136
137 if (GET_PARAM(2))
138 borderType |= BORDER_ISOLATED;
139
140 useRoi = GET_PARAM(4);
141 }
142
143 void random_roi()
144 {
145 border = randomBorder(0, MAX_VALUE << 2);
146 val = randomScalar(-MAX_VALUE, MAX_VALUE);
147
148 Size roiSize = randomSize(1, MAX_VALUE);
149 Border srcBorder = randomBorder(0, useRoi ? MAX_VALUE : 0);
150 randomSubMat(src, src_roi, roiSize, srcBorder, type, -MAX_VALUE, MAX_VALUE);
151
152 Border dstBorder = randomBorder(0, useRoi ? MAX_VALUE : 0);
153 dstBorder.top += border.top;
154 dstBorder.lef += border.lef;
155 dstBorder.rig += border.rig;
156 dstBorder.bot += border.bot;
157
158 randomSubMat(dst_whole, dst_roi, roiSize, dstBorder, type, -MAX_VALUE, MAX_VALUE);
159
160 generateOclMat(gsrc_whole, gsrc_roi, src, roiSize, srcBorder);
161 generateOclMat(gdst_whole, gdst_roi, dst_whole, roiSize, dstBorder);
162 }
163
164 void Near(double threshold = 0.0)
165 {
166 Mat whole, roi;
167 gdst_whole.download(whole);
168 gdst_roi.download(roi);
169
170 EXPECT_MAT_NEAR(dst_whole, whole, threshold);
171 EXPECT_MAT_NEAR(dst_roi, roi, threshold);
172 }
173};

184 Near();
185 }

207struct CornerTestBase :
208 public ImgprocTestBase
209{
210 virtual void random_roi()
211 {
212 Mat image = readImageType("gpu/stereobm/aloe-L.png", type);
213 ASSERT_FALSE(image.empty());
214
215 bool isFP = CV_MAT_DEPTH(type) >= CV_32F;
216 float val = 255.0f;
217 if (isFP)
218 {
219 image.convertTo(image, -1, 1.0 / 255);
220 val /= 255.0f;
221 }
222
223 Size roiSize = image.size();
224 Border srcBorder = randomBorder(0, useRoi ? MAX_VALUE : 0);
225
226 Size wholeSize = Size(roiSize.width + srcBorder.lef + srcBorder.rig, roiSize.height + srcBorder.top + srcBorder.bot);
227 src = randomMat(wholeSize, type, -val, val, false);
228 src_roi = src(Rect(srcBorder.lef, srcBorder.top, roiSize.width, roiSize.height));
229 image.copyTo(src_roi);
230
231 Border dstBorder = randomBorder(0, useRoi ? MAX_VALUE : 0);
232 randomSubMat(dst_whole, dst_roi, roiSize, dstBorder, CV_32FC1, 5, 16);
233
234 generateOclMat(gsrc_whole, gsrc_roi, src, roiSize, srcBorder);
235 generateOclMat(gdst_whole, gdst_roi, dst_whole, roiSize, dstBorder);
236 }
237};

366static void calcHistGold(const Mat &src, Mat &hist)
367{
368 hist = Mat(1, 256, CV_32SC1, Scalar::all(0));
369
370 int * const hist_row = hist.ptr<int>();
371 for (int y = 0; y < src.rows; ++y)
372 {
373 const uchar * const src_row = src.ptr(y);
374
375 for (int x = 0; x < src.cols; ++x)
376 ++hist_row[src_row[x]];
377 }
378}

456static void convolve_gold(const Mat & src, const Mat & kernel, Mat & dst)
457{
458 for (int i = 0; i < src.rows; i++)
459 {
460 float * const dstptr = dst.ptr<float>(i);
461
462 for (int j = 0; j < src.cols; j++)
463 {
464 float temp = 0;
465
466 for (int m = 0; m < kernel.rows; m++)
467 {
468 const float * const kptr = kernel.ptr<float>(m);
469 for (int n = 0; n < kernel.cols; n++)
470 {
471 int r = clipInt(i - kernel.rows / 2 + m, 0, src.rows - 1);
472 int c = clipInt(j - kernel.cols / 2 + n, 0, src.cols - 1);
473
474 temp += src.ptr<float>(r)[c] * kptr[n];
475 }
476 }
477
478 dstptr[j] = temp;
479 }
480 }
481}

506////////////////////////////////// ColumnSum //////////////////////////////////////
507
508static void columnSum_gold(const Mat & src, Mat & dst)
509{
510 float * prevdptr = dst.ptr<float>(0);
511 const float * sptr = src.ptr<float>(0);
512
513 for (int x = 0; x < src.cols; ++x)
514 prevdptr[x] = sptr[x];
515
516 for (int y = 1; y < src.rows; ++y)
517 {
518 sptr = src.ptr<float>(y);
519 float * const dptr = dst.ptr<float>(y);
520
521 for (int x = 0; x < src.cols; ++x)
522 dptr[x] = prevdptr[x] + sptr[x];
523
524 prevdptr = dptr;
525 }
526}
