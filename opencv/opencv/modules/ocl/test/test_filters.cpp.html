59PARAM_TEST_CASE(FilterTestBase, MatType,
60 int, // kernel size
61 Size, // dx, dy
62 int, // border type
63 double, // optional parameter
64 bool) // roi or not
65{
66 bool isFP;
67
68 int type, borderType, ksize;
69 Size size;
70 double param;
71 bool useRoi;
72
73 Mat src, dst_whole, src_roi, dst_roi;
74 ocl::oclMat gsrc_whole, gsrc_roi, gdst_whole, gdst_roi;
75
76 virtual void SetUp()
77 {
78 type = GET_PARAM(0);
79 ksize = GET_PARAM(1);
80 size = GET_PARAM(2);
81 borderType = GET_PARAM(3);
82 param = GET_PARAM(4);
83 useRoi = GET_PARAM(5);
84
85 isFP = (CV_MAT_DEPTH(type) == CV_32F || CV_MAT_DEPTH(type) == CV_64F);
86 }
87
88 void random_roi(int minSize = 1)
89 {
90 if (minSize == 0)
91 minSize = ksize;
92 Size roiSize = randomSize(minSize, MAX_VALUE);
93 Border srcBorder = randomBorder(0, useRoi ? MAX_VALUE : 0);
94 randomSubMat(src, src_roi, roiSize, srcBorder, type, isFP ? 0 : 5, isFP ? 1 : 256);
95
96 Border dstBorder = randomBorder(0, useRoi ? MAX_VALUE : 0);
97 randomSubMat(dst_whole, dst_roi, roiSize, dstBorder, type, isFP ? 0.20 : 60, isFP ? 0.25 : 70);
98
99 generateOclMat(gsrc_whole, gsrc_roi, src, roiSize, srcBorder);
100 generateOclMat(gdst_whole, gdst_roi, dst_whole, roiSize, dstBorder);
101 }
102
103 void Near()
104 {
105 if (isFP)
106 Near(1e-6, true);
107 else
108 Near(1, false);
109 }
110
111 void Near(double threshold, bool relative)
112 {
113 Mat roi, whole;
114 gdst_whole.download(whole);
115 gdst_roi.download(roi);
116
117 if (relative)
118 {
119 EXPECT_MAT_NEAR_RELATIVE(dst_whole, whole, threshold);
120 EXPECT_MAT_NEAR_RELATIVE(dst_roi, roi, threshold);
121 }
122 else
123 {
124 EXPECT_MAT_NEAR(dst_whole, whole, threshold);
125 EXPECT_MAT_NEAR(dst_roi, roi, threshold);
126 }
127 }
128};

284// Filter2D
285
286typedef FilterTestBase Filter2D;
287
288OCL_TEST_P(Filter2D, Mat)
289{
290 for (int j = 0; j < LOOP_TIMES; j++)
291 {
292 random_roi();
293
294 Point anchor(-1, -1);
295 if (size.width >= 0)
296 anchor.x = size.width % ksize;
297 if (size.height >= 0)
298 anchor.y = size.height % ksize;
299
300 const Size kernelSize(ksize, ksize);
301 Mat kernel = randomMat(kernelSize, CV_32FC1, 0, 1.0);
302 kernel *= 1.0 / (double)(ksize * ksize);
303
304 cv::filter2D(src_roi, dst_roi, -1, kernel, anchor, 0.0, borderType);
305 ocl::filter2D(gsrc_roi, gdst_roi, -1, kernel, anchor, 0.0, borderType);
306
307 Near();
308 }
309}
