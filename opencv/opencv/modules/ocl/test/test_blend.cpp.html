53template <typename T>
54static void blendLinearGold(const Mat &img1, const Mat &img2,
55 const Mat &weights1, const Mat &weights2,
56 Mat &result_gold)
57{
58 CV_Assert(img1.size() == img2.size() && img1.type() == img2.type());
59 CV_Assert(weights1.size() == weights2.size() && weights1.size() == img1.size() &&
60 weights1.type() == CV_32FC1 && weights2.type() == CV_32FC1);
61
62 result_gold.create(img1.size(), img1.type());
63
64 int cn = img1.channels();
65 int step1 = img1.cols * img1.channels();
66
67 for (int y = 0; y < img1.rows; ++y)
68 {
69 const float * const weights1_row = weights1.ptr<float>(y);
70 const float * const weights2_row = weights2.ptr<float>(y);
71 const T * const img1_row = img1.ptr<T>(y);
72 const T * const img2_row = img2.ptr<T>(y);
73 T * const result_gold_row = result_gold.ptr<T>(y);
74
75 for (int x = 0; x < step1; ++x)
76 {
77 int x1 = x / cn;
78 float w1 = weights1_row[x1], w2 = weights2_row[x1];
79 result_gold_row[x] = saturate_cast<T>(((float)img1_row[x] * w1
80 + (float)img2_row[x] * w2) / (w1 + w2 + 1e-5f));
81 }
82 }
83}

85PARAM_TEST_CASE(Blend, MatDepth, int, bool)
86{
87 int depth, channels;
88 bool useRoi;
89
90 Mat src1, src2, weights1, weights2, dst;
91 Mat src1_roi, src2_roi, weights1_roi, weights2_roi, dst_roi;
92 oclMat gsrc1, gsrc2, gweights1, gweights2, gdst, gst;
93 oclMat gsrc1_roi, gsrc2_roi, gweights1_roi, gweights2_roi, gdst_roi;
94
95 virtual void SetUp()
96 {
97 depth = GET_PARAM(0);
98 channels = GET_PARAM(1);
99 useRoi = GET_PARAM(2);
100 }
101
102 void random_roi()
103 {
104 const int type = CV_MAKE_TYPE(depth, channels);
105
106 const double upValue = 256;
107 const double sumMinValue = 0.01; // we don't want to divide by "zero"
108
109 Size roiSize = randomSize(1, 20);
110 Border src1Border = randomBorder(0, useRoi ? MAX_VALUE : 0);
111 randomSubMat(src1, src1_roi, roiSize, src1Border, type, -upValue, upValue);
112
113 Border src2Border = randomBorder(0, useRoi ? MAX_VALUE : 0);
114 randomSubMat(src2, src2_roi, roiSize, src2Border, type, -upValue, upValue);
115
116 Border weights1Border = randomBorder(0, useRoi ? MAX_VALUE : 0);
117 randomSubMat(weights1, weights1_roi, roiSize, weights1Border, CV_32FC1, -upValue, upValue);
118
119 Border weights2Border = randomBorder(0, useRoi ? MAX_VALUE : 0);
120 randomSubMat(weights2, weights2_roi, roiSize, weights2Border, CV_32FC1, sumMinValue, upValue); // fill it as a (w1 + w12)
121
122 weights2_roi = weights2_roi - weights1_roi;
123 // check that weights2_roi is still a part of weights2 (not a new matrix)
124 CV_Assert(checkNorm(weights2_roi,
125 weights2(Rect(weights2Border.lef, weights2Border.top, roiSize.width, roiSize.height))) < 1e-6);
126
127 Border dstBorder = randomBorder(0, useRoi ? MAX_VALUE : 0);
128 randomSubMat(dst, dst_roi, roiSize, dstBorder, type, 5, 16);
129
130 generateOclMat(gsrc1, gsrc1_roi, src1, roiSize, src1Border);
131 generateOclMat(gsrc2, gsrc2_roi, src2, roiSize, src2Border);
132 generateOclMat(gweights1, gweights1_roi, weights1, roiSize, weights1Border);
133 generateOclMat(gweights2, gweights2_roi, weights2, roiSize, weights2Border);
134 generateOclMat(gdst, gdst_roi, dst, roiSize, dstBorder);
135 }
136
137 void Near(double eps = 0.0)
138 {
139 Mat whole, roi;
140 gdst.download(whole);
141 gdst_roi.download(roi);
142
143 EXPECT_MAT_NEAR(dst, whole, eps);
144 EXPECT_MAT_NEAR(dst_roi, roi, eps);
145 }
146};

148typedef void (*blendLinearFunc)(const cv::Mat &img1, const cv::Mat &img2, const cv::Mat &weights1, const cv::Mat &weights2, cv::Mat &result_gold);
149
150OCL_TEST_P(Blend, Accuracy)
151{
152 for (int i = 0; i < LOOP_TIMES; ++i)
153 {
154 random_roi();
155
156 cv::ocl::blendLinear(gsrc1_roi, gsrc2_roi, gweights1_roi, gweights2_roi, gdst_roi);
157
158 static blendLinearFunc funcs[] = {
159 blendLinearGold<uchar>,
160 blendLinearGold<schar>,
161 blendLinearGold<ushort>,
162 blendLinearGold<short>,
163 blendLinearGold<int>,
164 blendLinearGold<float>,
165 };
166
167 blendLinearFunc func = funcs[depth];
168 func(src1_roi, src2_roi, weights1_roi, weights2_roi, dst_roi);
169
170 Near(depth <= CV_32S ? 1.0 : 0.2);
171 }
172}
