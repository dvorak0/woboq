50// GEMM

52PARAM_TEST_CASE(Gemm, int, cv::Size, int)
53{
54 int type;
55 cv::Size mat_size;
56 int flags;
57
58 virtual void SetUp()
59 {
60 type = GET_PARAM(0);
61 mat_size = GET_PARAM(1);
62 flags = GET_PARAM(2);
63 }
64};
65
66OCL_TEST_P(Gemm, Accuracy)
67{
68 cv::Mat a = randomMat(mat_size, type, 0.0, 10.0);
69 cv::Mat b = randomMat(mat_size, type, 0.0, 10.0);
70 cv::Mat c = randomMat(mat_size, type, 0.0, 10.0);
71
72 cv::Mat dst;
73 cv::ocl::oclMat ocl_dst;
74
75 cv::gemm(a, b, 1.0, c, 1.0, dst, flags);
76 cv::ocl::gemm(cv::ocl::oclMat(a), cv::ocl::oclMat(b), 1.0, cv::ocl::oclMat(c), 1.0, ocl_dst, flags);
77
78 EXPECT_MAT_NEAR(dst, ocl_dst, mat_size.area() * 1e-4);
79}
80
81INSTANTIATE_TEST_CASE_P(ocl_gemm, Gemm, testing::Combine(
82 testing::Values(CV_32FC1, CV_32FC2/*, CV_64FC1, CV_64FC2*/),
83 testing::Values(cv::Size(20, 20), cv::Size(300, 300)),
84 testing::Values(0, (int)cv::GEMM_1_T, (int)cv::GEMM_2_T, (int)(cv::GEMM_1_T + cv::GEMM_2_T))));
