57static void genTrainData(cv::RNG& rng, Mat& trainData, int trainDataRow, int trainDataCol,
58 Mat& trainLabel = Mat().setTo(Scalar::all(0)), int nClasses = 0)
59{
60 cv::Size size(trainDataCol, trainDataRow);
61 trainData = randomMat(rng, size, CV_32FC1, 1.0, 1000.0, false);
62 if(nClasses != 0)
63 {
64 cv::Size size1(trainDataRow, 1);
65 trainLabel = randomMat(rng, size1, CV_8UC1, 0, nClasses - 1, false);
66 trainLabel.convertTo(trainLabel, CV_32FC1);
67 }
68}

70PARAM_TEST_CASE(KNN, int, Size, int, bool)
71{
72 int k;
73 int trainDataCol;
74 int testDataRow;
75 int nClass;
76 bool regression;
77 virtual void SetUp()
78 {
79 k = GET_PARAM(0);
80 nClass = GET_PARAM(2);
81 trainDataCol = GET_PARAM(1).width;
82 testDataRow = GET_PARAM(1).height;
83 regression = GET_PARAM(3);
84 }
85};
86
87OCL_TEST_P(KNN, Accuracy)
88{
89 Mat trainData, trainLabels;
90 const int trainDataRow = 500;
91 genTrainData(rng, trainData, trainDataRow, trainDataCol, trainLabels, nClass);
92
93 Mat testData, testLabels;
94 genTrainData(rng, testData, testDataRow, trainDataCol);
95
96 KNearestNeighbour knn_ocl;
97 CvKNearest knn_cpu;
98 Mat best_label_cpu;
99 oclMat best_label_ocl;
100
101 /*ocl k-Nearest_Neighbor start*/
102 oclMat trainData_ocl;
103 trainData_ocl.upload(trainData);
104 Mat simpleIdx;
105 knn_ocl.train(trainData, trainLabels, simpleIdx, regression);
106
107 oclMat testdata;
108 testdata.upload(testData);
109 knn_ocl.find_nearest(testdata, k, best_label_ocl);
110 /*ocl k-Nearest_Neighbor end*/
111
112 /*cpu k-Nearest_Neighbor start*/
113 knn_cpu.train(trainData, trainLabels, simpleIdx, regression);
114 knn_cpu.find_nearest(testData, k, &best_label_cpu);
115 /*cpu k-Nearest_Neighbor end*/
116 if(regression)
117 {
118 EXPECT_MAT_SIMILAR(Mat(best_label_ocl), best_label_cpu, 1e-5);
119 }
120 else
121 {
122 EXPECT_MAT_NEAR(Mat(best_label_ocl), best_label_cpu, 0.0);
123 }
124}

131PARAM_TEST_CASE(SVM_OCL, int, int, int)
132{
133 cv::Size size;
134 int kernel_type;
135 int svm_type;
136 Mat src, labels, samples, labels_predict;
137 int K;
138
139 virtual void SetUp()
140 {
141
142 kernel_type = GET_PARAM(0);
143 svm_type = GET_PARAM(1);
144 K = GET_PARAM(2);
145 cv::Size sz = cv::Size(MWIDTH, MHEIGHT);
146 src.create(sz, CV_32FC1);
147 labels.create(1, sz.height, CV_32SC1);
148 int row_idx = 0;
149 const int max_number = sz.height / K - 1;
150 CV_Assert(K <= sz.height);
151 for(int i = 0; i < K; i++ )
152 {
153 Mat center_row_header = src.row(row_idx);
154 center_row_header.setTo(0);
155 int nchannel = center_row_header.channels();
156 for(int j = 0; j < nchannel; j++)
157 {
158 center_row_header.at<float>(0, i * nchannel + j) = 500.0;
159 }
160 labels.at<int>(0, row_idx) = i;
161 for(int j = 0; (j < max_number) ||
162 (i == K - 1 && j < max_number + sz.height % K); j ++)
163 {
164 Mat cur_row_header = src.row(row_idx + 1 + j);
165 center_row_header.copyTo(cur_row_header);
166 Mat tmpmat = randomMat(cur_row_header.size(), cur_row_header.type(), 1, 100, false);
167 cur_row_header += tmpmat;
168 labels.at<int>(0, row_idx + 1 + j) = i;
169 }
170 row_idx += 1 + max_number;
171 }
172 labels.convertTo(labels, CV_32FC1);
173 cv::Size test_size = cv::Size(MWIDTH, 100);
174 samples.create(test_size, CV_32FC1);
175 labels_predict.create(1, test_size.height, CV_32SC1);
176 const int max_number_test = test_size.height / K - 1;
177 row_idx = 0;
178 for(int i = 0; i < K; i++ )
179 {
180 Mat center_row_header = samples.row(row_idx);
181 center_row_header.setTo(0);
182 int nchannel = center_row_header.channels();
183 for(int j = 0; j < nchannel; j++)
184 {
185 center_row_header.at<float>(0, i * nchannel + j) = 500.0;
186 }
187 labels_predict.at<int>(0, row_idx) = i;
188 for(int j = 0; (j < max_number_test) ||
189 (i == K - 1 && j < max_number_test + test_size.height % K); j ++)
190 {
191 Mat cur_row_header = samples.row(row_idx + 1 + j);
192 center_row_header.copyTo(cur_row_header);
193 Mat tmpmat = randomMat(cur_row_header.size(), cur_row_header.type(), 1, 100, false);
194 cur_row_header += tmpmat;
195 labels_predict.at<int>(0, row_idx + 1 + j) = i;
196 }
197 row_idx += 1 + max_number_test;
198 }
199 labels_predict.convertTo(labels_predict, CV_32FC1);
200 }
201};
202
203OCL_TEST_P(SVM_OCL, Accuracy)
204{
205 CvSVMParams params;
206 params.degree = 0.4;
207 params.gamma = 1;
208 params.coef0 = 1;
209 params.C = 1;
210 params.nu = 0.5;
211 params.p = 1;
212 params.svm_type = svm_type;
213 params.kernel_type = kernel_type;
214
215 params.term_crit = cvTermCriteria(CV_TERMCRIT_ITER, 1000, 0.001);
216
217 CvSVM SVM;
218 SVM.train(src, labels, Mat(), Mat(), params);
219
220 cv::ocl::CvSVM_OCL SVM_OCL;
221 SVM_OCL.train(src, labels, Mat(), Mat(), params);
222
223 int c = SVM.get_support_vector_count();
224 int c1 = SVM_OCL.get_support_vector_count();
225
226 Mat sv(c, MHEIGHT, CV_32FC1);
227 Mat sv_ocl(c1, MHEIGHT, CV_32FC1);
228 for(int i = 0; i < c; i++)
229 {
230 const float* v = SVM.get_support_vector(i);
231
232 for(int j = 0; j < MHEIGHT; j++)
233 {
234 sv.at<float>(i, j) = v[j];
235 }
236 }
237 for(int i = 0; i < c1; i++)
238 {
239 const float* v_ocl = SVM_OCL.get_support_vector(i);
240
241 for(int j = 0; j < MHEIGHT; j++)
242 {
243 sv_ocl.at<float>(i, j) = v_ocl[j];
244 }
245 }
246 cv::BFMatcher matcher(cv::NORM_L2);
247 std::vector<cv::DMatch> matches;
248 matcher.match(sv, sv_ocl, matches);
249 int count = 0;
250
251 for(std::vector<cv::DMatch>::iterator itr = matches.begin(); itr != matches.end(); itr++)
252 {
253 if((*itr).distance < 0.1)
254 {
255 count ++;
256 }
257 }
258 if(c != 0)
259 {
260 float matchedRatio = (float)count / c;
261 EXPECT_GT(matchedRatio, 0.95);
262 }
263 if(c != 0)
264 {
265 CvMat *result = cvCreateMat(1, samples.rows, CV_32FC1);
266 CvMat test_samples = samples;
267
268 CvMat *result_ocl = cvCreateMat(1, samples.rows, CV_32FC1);
269
270 SVM.predict(&test_samples, result);
271
272 SVM_OCL.predict(&test_samples, result_ocl);
273
274 int true_resp = 0, true_resp_ocl = 0;
275 for (int i = 0; i < samples.rows; i++)
276 {
277 if (result->data.fl[i] == labels_predict.at<float>(0, i))
278 {
279 true_resp++;
280 }
281 }
282 float matchedRatio = (float)true_resp / samples.rows;
283
284 for (int i = 0; i < samples.rows; i++)
285 {
286 if (result_ocl->data.fl[i] == labels_predict.at<float>(0, i))
287 {
288 true_resp_ocl++;
289 }
290 }
291 float matchedRatio_ocl = (float)true_resp_ocl / samples.rows;
292
293 if(matchedRatio != 0 && true_resp_ocl < true_resp)
294 {
295 EXPECT_NEAR(matchedRatio_ocl, matchedRatio, 0.03);
296 }
297 }
298}
