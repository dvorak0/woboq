56#define OCL_KMEANS_USE_INITIAL_LABELS 1
57#define OCL_KMEANS_PP_CENTERS 2

59PARAM_TEST_CASE(Kmeans, int, int, int)
60{
61 int type;
62 int K;
63 int flags;
64 Mat src ;
65 ocl::oclMat d_src, d_dists;
66
67 Mat labels, centers;
68 ocl::oclMat d_labels, d_centers;
69 virtual void SetUp()
70 {
71 K = GET_PARAM(0);
72 type = GET_PARAM(1);
73 flags = GET_PARAM(2);
74
75 // MWIDTH=256, MHEIGHT=256. defined in utility.hpp
76 Size size = Size(MWIDTH, MHEIGHT);
77 src.create(size, type);
78 int row_idx = 0;
79 const int max_neighbour = MHEIGHT / K - 1;
80 CV_Assert(K <= MWIDTH);
81 for(int i = 0; i < K; i++ )
82 {
83 Mat center_row_header = src.row(row_idx);
84 center_row_header.setTo(0);
85 int nchannel = center_row_header.channels();
86 for(int j = 0; j < nchannel; j++)
87 center_row_header.at<float>(0, i*nchannel+j) = 50000.0;
88
89 for(int j = 0; (j < max_neighbour) ||
90 (i == K-1 && j < max_neighbour + MHEIGHT%K); j ++)
91 {
92 Mat cur_row_header = src.row(row_idx + 1 + j);
93 center_row_header.copyTo(cur_row_header);
94 Mat tmpmat = randomMat(cur_row_header.size(), cur_row_header.type(), -200, 200, false);
95 cur_row_header += tmpmat;
96 }
97 row_idx += 1 + max_neighbour;
98 }
99 }
100};
101OCL_TEST_P(Kmeans, Mat){
102 if(flags & KMEANS_USE_INITIAL_LABELS)
103 {
104 // inital a given labels
105 labels.create(src.rows, 1, CV_32S);
106 int *label = labels.ptr<int>();
107 for(int i = 0; i < src.rows; i++)
108 label[i] = rng.uniform(0, K);
109 d_labels.upload(labels);
110 }
111 d_src.upload(src);
112
113 for(int j = 0; j < LOOP_TIMES; j++)
114 {
115 kmeans(src, K, labels,
116 TermCriteria( CV_TERMCRIT_EPS+CV_TERMCRIT_ITER, 100, 0),
117 1, flags, centers);
118 ocl::kmeans(d_src, K, d_labels,
119 TermCriteria( CV_TERMCRIT_EPS+CV_TERMCRIT_ITER, 100, 0),
120 1, flags, d_centers);
121 Mat dd_labels(d_labels);
122 Mat dd_centers(d_centers);
123 if(flags & KMEANS_USE_INITIAL_LABELS)
124 {
125 EXPECT_MAT_NEAR(labels, dd_labels, 0);
126 EXPECT_MAT_NEAR(centers, dd_centers, 1e-3);
127 }
128 else
129 {
130 int row_idx = 0;
131 for(int i = 0; i < K; i++)
132 {
133 // verify lables with ground truth resutls
134 int label = labels.at<int>(row_idx);
135 int header_label = dd_labels.at<int>(row_idx);
136 for(int j2 = 0; (j2 < MHEIGHT/K)||(i == K-1 && j2 < MHEIGHT/K+MHEIGHT%K); j2++)
137 {
138 ASSERT_NEAR(labels.at<int>(row_idx+j2), label, 0);
139 ASSERT_NEAR(dd_labels.at<int>(row_idx+j2), header_label, 0);
140 }
141
142 // verify centers
143 float *center = centers.ptr<float>(label);
144 float *header_center = dd_centers.ptr<float>(header_label);
145 for(int t = 0; t < centers.cols; t++)
146 ASSERT_NEAR(center[t], header_center[t], 1e-3);
147
148 row_idx += MHEIGHT/K;
149 }
150 }
151 }
152}

162CV_ENUM(DistType, NORM_L1, NORM_L2SQR)
163
164PARAM_TEST_CASE(distanceToCenters, DistType, bool)
165{
166 int distType;
167 bool useRoi;
168
169 Mat src, centers, src_roi, centers_roi;
170 ocl::oclMat ocl_src, ocl_centers, ocl_src_roi, ocl_centers_roi;
171
172 virtual void SetUp()
173 {
174 distType = GET_PARAM(0);
175 useRoi = GET_PARAM(1);
176 }
177
178 void random_roi()
179 {
180 Size roiSizeSrc = randomSize(1, MAX_VALUE);
181 Size roiSizeCenters = randomSize(1, MAX_VALUE);
182 roiSizeSrc.width = roiSizeCenters.width;
183
184 Border srcBorder = randomBorder(0, useRoi ? MAX_VALUE : 0);
185 randomSubMat(src, src_roi, roiSizeSrc, srcBorder, CV_32FC1, -MAX_VALUE, MAX_VALUE);
186
187 Border centersBorder = randomBorder(0, useRoi ? 500 : 0);
188 randomSubMat(centers, centers_roi, roiSizeCenters, centersBorder, CV_32FC1, -MAX_VALUE, MAX_VALUE);
189
190 for (int i = 0; i < centers.rows; i++)
191 centers.at<float>(i, randomInt(0, centers.cols)) = (float)randomDouble(SHRT_MAX, INT_MAX);
192
193 generateOclMat(ocl_src, ocl_src_roi, src, roiSizeSrc, srcBorder);
194 generateOclMat(ocl_centers, ocl_centers_roi, centers, roiSizeCenters, centersBorder);
195 }
196};
197
198OCL_TEST_P(distanceToCenters, Accuracy)
199{
200 for (int j = 0; j < LOOP_TIMES; j++)
201 {
202 random_roi();
203
204 Mat labels, dists;
205 ocl::distanceToCenters(ocl_src_roi, ocl_centers_roi, dists, labels, distType);
206
207 EXPECT_EQ(dists.size(), labels.size());
208
209 Mat batch_dists;
210 cv::batchDistance(src_roi, centers_roi, batch_dists, CV_32FC1, noArray(), distType);
211
212 std::vector<float> gold_dists_v;
213 gold_dists_v.reserve(batch_dists.rows);
214
215 for (int i = 0; i < batch_dists.rows; i++)
216 {
217 Mat r = batch_dists.row(i);
218 double mVal;
219 Point mLoc;
220 minMaxLoc(r, &mVal, NULL, &mLoc, NULL);
221
222 int ocl_label = labels.at<int>(i, 0);
223 EXPECT_EQ(mLoc.x, ocl_label);
224
225 gold_dists_v.push_back(static_cast<float>(mVal));
226 }
227
228 double relative_error = cv::norm(Mat(gold_dists_v), dists, NORM_INF | NORM_RELATIVE);
229 ASSERT_LE(relative_error, 1e-5);
230 }
231}
