55PARAM_TEST_CASE(CvtColor, MatDepth, bool)
56{
57 int depth;
58 bool use_roi;
59
60 // src mat
61 Mat src;
62 Mat dst;
63
64 // src mat with roi
65 Mat src_roi;
66 Mat dst_roi;
67
68 // ocl dst mat for testing
69 ocl::oclMat gsrc_whole;
70 ocl::oclMat gdst_whole;
71
72 // ocl mat with roi
73 ocl::oclMat gsrc_roi;
74 ocl::oclMat gdst_roi;
75
76 virtual void SetUp()
77 {
78 depth = GET_PARAM(0);
79 use_roi = GET_PARAM(1);
80 }
81
82 virtual void random_roi(int channelsIn, int channelsOut)
83 {
84 const int srcType = CV_MAKE_TYPE(depth, channelsIn);
85 const int dstType = CV_MAKE_TYPE(depth, channelsOut);
86
87 Size roiSize = randomSize(1, MAX_VALUE);
88 Border srcBorder = randomBorder(0, use_roi ? MAX_VALUE : 0);
89 randomSubMat(src, src_roi, roiSize, srcBorder, srcType, 2, 100);
90
91 Border dstBorder = randomBorder(0, use_roi ? MAX_VALUE : 0);
92 randomSubMat(dst, dst_roi, roiSize, dstBorder, dstType, 5, 16);
93
94 generateOclMat(gsrc_whole, gsrc_roi, src, roiSize, srcBorder);
95 generateOclMat(gdst_whole, gdst_roi, dst, roiSize, dstBorder);
96 }
97
98 void Near(double threshold)
99 {
100 Mat whole, roi;
101 gdst_whole.download(whole);
102 gdst_roi.download(roi);
103
104 EXPECT_MAT_NEAR(dst, whole, threshold);
105 EXPECT_MAT_NEAR(dst_roi, roi, threshold);
106 }
107
108 void doTest(int channelsIn, int channelsOut, int code, double threshold = 1e-3)
109 {
110 for (int j = 0; j < LOOP_TIMES; j++)
111 {
112 random_roi(channelsIn, channelsOut);
113
114 cvtColor(src_roi, dst_roi, code, channelsOut);
115 ocl::cvtColor(gsrc_roi, gdst_roi, code, channelsOut);
116
117 Near(threshold);
118 }
119 }
120};

269struct CvtColor_YUV420 :
270 public CvtColor
271{
272 void random_roi(int channelsIn, int channelsOut)
273 {
274 const int srcType = CV_MAKE_TYPE(depth, channelsIn);
275 const int dstType = CV_MAKE_TYPE(depth, channelsOut);
276
277 Size roiSize = randomSize(1, MAX_VALUE);
278 roiSize.width *= 2;
279 roiSize.height *= 3;
280 Border srcBorder = randomBorder(0, use_roi ? MAX_VALUE : 0);
281 randomSubMat(src, src_roi, roiSize, srcBorder, srcType, 2, 100);
282
283 Border dstBorder = randomBorder(0, use_roi ? MAX_VALUE : 0);
284 randomSubMat(dst, dst_roi, roiSize, dstBorder, dstType, 5, 16);
285
286 generateOclMat(gsrc_whole, gsrc_roi, src, roiSize, srcBorder);
287 generateOclMat(gdst_whole, gdst_roi, dst, roiSize, dstBorder);
288 }
289};
