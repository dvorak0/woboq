74
75inline double checkNormRelative(const Mat &m1, const Mat &m2)
76{
77 return cv::norm(m1, m2, cv::NORM_INF) /
78 std::max((double)std::numeric_limits<float>::epsilon(),
79 (double)std::max(cv::norm(m1, cv::NORM_INF), norm(m2, cv::NORM_INF)));
80}

82#define EXPECT_MAT_NORM(mat, eps) \
83{ \
84 EXPECT_LE(checkNorm(cv::Mat(mat)), eps) \
85}
86
87#define EXPECT_MAT_NEAR(mat1, mat2, eps) \
88{ \
89 ASSERT_EQ(mat1.type(), mat2.type()); \
90 ASSERT_EQ(mat1.size(), mat2.size()); \
91 EXPECT_LE(checkNorm(cv::Mat(mat1), cv::Mat(mat2)), eps) \
92 << cv::format("Size: %d x %d", mat1.cols, mat1.rows) << std::endl; \
93}
94
95#define EXPECT_MAT_NEAR_RELATIVE(mat1, mat2, eps) \
96{ \
97 ASSERT_EQ(mat1.type(), mat2.type()); \
98 ASSERT_EQ(mat1.size(), mat2.size()); \
99 EXPECT_LE(checkNormRelative(cv::Mat(mat1), cv::Mat(mat2)), eps) \
100 << cv::format("Size: %d x %d", mat1.cols, mat1.rows) << std::endl; \
101}
102
103#define EXPECT_MAT_SIMILAR(mat1, mat2, eps) \
104{ \
105 ASSERT_EQ(mat1.type(), mat2.type()); \
106 ASSERT_EQ(mat1.size(), mat2.size()); \
107 EXPECT_LE(checkSimilarity(cv::Mat(mat1), cv::Mat(mat2)), eps); \
108}

138template <typename T>
139struct TSTestWithParam : public ::testing::TestWithParam<T>
140{
141 cv::RNG rng;
142
143 TSTestWithParam()
144 {
145 rng = cv::RNG(OCL_RNG_SEED);
146 }
147
148 int randomInt(int minVal, int maxVal)
149 {
150 return rng.uniform(minVal, maxVal);
151 }
152
153 double randomDouble(double minVal, double maxVal)
154 {
155 return rng.uniform(minVal, maxVal);
156 }
157
158 double randomDoubleLog(double minVal, double maxVal)
159 {
160 double logMin = log((double)minVal + 1);
161 double logMax = log((double)maxVal + 1);
162 double pow = rng.uniform(logMin, logMax);
163 double v = exp(pow) - 1;
164 CV_Assert(v >= minVal && (v < maxVal || (v == minVal && v == maxVal)));
165 return v;
166 }
167
168 Size randomSize(int minVal, int maxVal)
169 {
170#if 1
171 return cv::Size((int)randomDoubleLog(minVal, maxVal), (int)randomDoubleLog(minVal, maxVal));
172#else
173 return cv::Size(randomInt(minVal, maxVal), randomInt(minVal, maxVal));
174#endif
175 }
176
177 Size randomSize(int minValX, int maxValX, int minValY, int maxValY)
178 {
179#if 1
180 return cv::Size(randomDoubleLog(minValX, maxValX), randomDoubleLog(minValY, maxValY));
181#else
182 return cv::Size(randomInt(minVal, maxVal), randomInt(minVal, maxVal));
183#endif
184 }
185
186 Scalar randomScalar(double minVal, double maxVal)
187 {
188 return Scalar(randomDouble(minVal, maxVal), randomDouble(minVal, maxVal), randomDouble(minVal, maxVal), randomDouble(minVal, maxVal));
189 }
190
191 Mat randomMat(Size size, int type, double minVal, double maxVal, bool useRoi = false)
192 {
193 RNG dataRng(rng.next());
194 return cvtest::randomMat(dataRng, size, type, minVal, maxVal, useRoi);
195 }
196
197 struct Border
198 {
199 int top, bot, lef, rig;
200 };
201
202 Border randomBorder(int minValue = 0, int maxValue = MAX_VALUE)
203 {
204 Border border = {
205 (int)randomDoubleLog(minValue, maxValue),
206 (int)randomDoubleLog(minValue, maxValue),
207 (int)randomDoubleLog(minValue, maxValue),
208 (int)randomDoubleLog(minValue, maxValue)
209 };
210 return border;
211 }
212
213 void randomSubMat(Mat& whole, Mat& subMat, const Size& roiSize, const Border& border, int type, double minVal, double maxVal)
214 {
215 Size wholeSize = Size(roiSize.width + border.lef + border.rig, roiSize.height + border.top + border.bot);
216 whole = randomMat(wholeSize, type, minVal, maxVal, false);
217 subMat = whole(Rect(border.lef, border.top, roiSize.width, roiSize.height));
218 }
219
220 void generateOclMat(cv::ocl::oclMat& whole, cv::ocl::oclMat& subMat, const Mat& wholeMat, const Size& roiSize, const Border& border)
221 {
222 whole = wholeMat;
223 subMat = whole(Rect(border.lef, border.top, roiSize.width, roiSize.height));
224 }
225};

273# define OCL_TEST_P(test_case_name, test_name) \
274 class GTEST_TEST_CLASS_NAME_(test_case_name, test_name) : \
275 public test_case_name { \
276 public: \
277 GTEST_TEST_CLASS_NAME_(test_case_name, test_name)() { } \
278 virtual void TestBody(); \
279 void OCLTestBody(); \
280 private: \
281 static int AddToRegistry() \
282 { \
283 ::testing::UnitTest::GetInstance()->parameterized_test_registry(). \
284 GetTestCasePatternHolder<test_case_name>(\
285 #test_case_name, __FILE__, __LINE__)->AddTestPattern(\
286 #test_case_name, \
287 #test_name, \
288 new ::testing::internal::TestMetaFactory< \
289 GTEST_TEST_CLASS_NAME_(test_case_name, test_name)>()); \
290 return 0; \
291 } \
292 \
293 static int gtest_registering_dummy_; \
294 GTEST_DISALLOW_COPY_AND_ASSIGN_(\
295 GTEST_TEST_CLASS_NAME_(test_case_name, test_name)); \
296 }; \
297 \
298 int GTEST_TEST_CLASS_NAME_(test_case_name, \
299 test_name)::gtest_registering_dummy_ GTEST_ATTRIBUTE_UNUSED_ = \
300 GTEST_TEST_CLASS_NAME_(test_case_name, test_name)::AddToRegistry(); \
301 \
302 void GTEST_TEST_CLASS_NAME_(test_case_name, test_name)::TestBody() \
303 { \
304 try \
305 { \
306 OCLTestBody(); \
307 } \
308 catch (const cv::Exception & ex) \
309 { \
310 if (ex.code == CV_OpenCLDoubleNotSupported)\
311 std::cout << "Test skipped (selected device does not support double)" << std::endl; \
312 else if (ex.code == CV_OpenCLNoAMDBlasFft) \
313 std::cout << "Test skipped (AMD Blas / Fft libraries are not available)" << std::endl; \
314 else \
315 throw; \
316 } \
317 } \
318 \
319 void GTEST_TEST_CLASS_NAME_(test_case_name, test_name)::OCLTestBody()
