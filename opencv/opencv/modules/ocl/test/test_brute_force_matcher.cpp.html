46#include "test_precomp.hpp"
47#ifdef HAVE_OPENCL
48namespace
49{
50 /////////////////////////////////////////////////////////////////////////////////////////////////
51 // BruteForceMatcher
52 CV_ENUM(DistType, BruteForceMatcher_OCL_base::L1Dist,
53 BruteForceMatcher_OCL_base::L2Dist,
54 BruteForceMatcher_OCL_base::HammingDist)
55 IMPLEMENT_PARAM_CLASS(DescriptorSize, int)
56 PARAM_TEST_CASE(BruteForceMatcher, DistType, DescriptorSize)
57 {
58 cv::ocl::BruteForceMatcher_OCL_base::DistType distType;
59 int normCode;
60 int dim;
61
62 int queryDescCount;
63 int countFactor;
64
65 cv::Mat query, train;
66
67 virtual void SetUp()
68 {
69 distType = (cv::ocl::BruteForceMatcher_OCL_base::DistType)(int)GET_PARAM(0);
70 dim = GET_PARAM(1);
71
72 queryDescCount = 300; // must be even number because we split train data in some cases in two
73 countFactor = 4; // do not change it
74
75 cv::Mat queryBuf, trainBuf;
76
77 // Generate query descriptors randomly.
78 // Descriptor vector elements are integer values.
79 queryBuf.create(queryDescCount, dim, CV_32SC1);
80 rng.fill(queryBuf, cv::RNG::UNIFORM, cv::Scalar::all(0), cv::Scalar::all(3));
81 queryBuf.convertTo(queryBuf, CV_32FC1);
82
83 // Generate train decriptors as follows:
84 // copy each query descriptor to train set countFactor times
85 // and perturb some one element of the copied descriptors in
86 // in ascending order. General boundaries of the perturbation
87 // are (0.f, 1.f).
88 trainBuf.create(queryDescCount * countFactor, dim, CV_32FC1);
89 float step = 1.f / countFactor;
90 for (int qIdx = 0; qIdx < queryDescCount; qIdx++)
91 {
92 cv::Mat queryDescriptor = queryBuf.row(qIdx);
93 for (int c = 0; c < countFactor; c++)
94 {
95 int tIdx = qIdx * countFactor + c;
96 cv::Mat trainDescriptor = trainBuf.row(tIdx);
97 queryDescriptor.copyTo(trainDescriptor);
98 int elem = rng(dim);
99 float diff = rng.uniform(step * c, step * (c + 1));
100 trainDescriptor.at<float>(0, elem) += diff;
101 }
102 }
103
104 queryBuf.convertTo(query, CV_32F);
105 trainBuf.convertTo(train, CV_32F);
106 }
107 };
108
109#ifdef ANDROID
110 OCL_TEST_P(BruteForceMatcher, DISABLED_Match_Single)
111#else
112 OCL_TEST_P(BruteForceMatcher, Match_Single)
113#endif
114 {
115 cv::ocl::BruteForceMatcher_OCL_base matcher(distType);
116
117 std::vector<cv::DMatch> matches;
118 matcher.match(cv::ocl::oclMat(query), cv::ocl::oclMat(train), matches);
119
120 ASSERT_EQ(static_cast<size_t>(queryDescCount), matches.size());
121
122 int badCount = 0;
123 for (size_t i = 0; i < matches.size(); i++)
124 {
125 cv::DMatch match = matches[i];
126 if ((match.queryIdx != (int)i) || (match.trainIdx != (int)i * countFactor) || (match.imgIdx != 0))
127 badCount++;
128 }
129
130 ASSERT_EQ(0, badCount);
131 }
132
133#ifdef ANDROID
134 OCL_TEST_P(BruteForceMatcher, DISABLED_KnnMatch_2_Single)
135#else
136 OCL_TEST_P(BruteForceMatcher, KnnMatch_2_Single)
137#endif
138 {
139 const int knn = 2;
140
141 cv::ocl::BruteForceMatcher_OCL_base matcher(distType);
142
143 std::vector< std::vector<cv::DMatch> > matches;
144 matcher.knnMatch(cv::ocl::oclMat(query), cv::ocl::oclMat(train), matches, knn);
145
146 ASSERT_EQ(static_cast<size_t>(queryDescCount), matches.size());
147
148 int badCount = 0;
149 for (size_t i = 0; i < matches.size(); i++)
150 {
151 if ((int)matches[i].size() != knn)
152 badCount++;
153 else
154 {
155 int localBadCount = 0;
156 for (int k = 0; k < knn; k++)
157 {
158 cv::DMatch match = matches[i][k];
159 if ((match.queryIdx != (int)i) || (match.trainIdx != (int)i * countFactor + k) || (match.imgIdx != 0))
160 localBadCount++;
161 }
162 badCount += localBadCount > 0 ? 1 : 0;
163 }
164 }
165
166 ASSERT_EQ(0, badCount);
167 }
168
169#ifdef ANDROID
170 OCL_TEST_P(BruteForceMatcher, DISABLED_RadiusMatch_Single)
171#else
172 OCL_TEST_P(BruteForceMatcher, RadiusMatch_Single)
173#endif
174 {
175 float radius = 1.f / countFactor;
176
177 cv::ocl::BruteForceMatcher_OCL_base matcher(distType);
178
179 std::vector< std::vector<cv::DMatch> > matches;
180 matcher.radiusMatch(cv::ocl::oclMat(query), cv::ocl::oclMat(train), matches, radius);
181
182 ASSERT_EQ(static_cast<size_t>(queryDescCount), matches.size());
183
184 int badCount = 0;
185 for (size_t i = 0; i < matches.size(); i++)
186 {
187 if ((int)matches[i].size() != 1)
188 {
189 badCount++;
190 }
191 else
192 {
193 cv::DMatch match = matches[i][0];
194 if ((match.queryIdx != (int)i) || (match.trainIdx != (int)i * countFactor) || (match.imgIdx != 0))
195 badCount++;
196 }
197 }
198
199 ASSERT_EQ(0, badCount);
200 }
