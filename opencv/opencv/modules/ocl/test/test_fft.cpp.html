53PARAM_TEST_CASE(Dft, cv::Size, int)
54{
55 cv::Size dft_size;
56 int dft_flags;
57 virtual void SetUp()
58 {
59 dft_size = GET_PARAM(0);
60 dft_flags = GET_PARAM(1);
61 }
62};
63
64OCL_TEST_P(Dft, C2C)
65{
66 cv::Mat a = randomMat(dft_size, CV_32FC2, 0.0, 100.0);
67 cv::Mat b_gold;
68
69 cv::ocl::oclMat d_b;
70
71 cv::dft(a, b_gold, dft_flags);
72 cv::ocl::dft(cv::ocl::oclMat(a), d_b, a.size(), dft_flags);
73 EXPECT_MAT_NEAR(b_gold, cv::Mat(d_b), a.size().area() * 1e-4);
74}
75
76OCL_TEST_P(Dft, R2C)
77{
78 cv::Mat a = randomMat(dft_size, CV_32FC1, 0.0, 100.0);
79 cv::Mat b_gold, b_gold_roi;
80
81 cv::ocl::oclMat d_b, d_c;
82 cv::ocl::dft(cv::ocl::oclMat(a), d_b, a.size(), dft_flags);
83 cv::dft(a, b_gold, cv::DFT_COMPLEX_OUTPUT | dft_flags);
84
85 b_gold_roi = b_gold(cv::Rect(0, 0, d_b.cols, d_b.rows));
86 EXPECT_MAT_NEAR(b_gold_roi, cv::Mat(d_b), a.size().area() * 1e-4);
87
88 cv::Mat c_gold;
89 cv::dft(b_gold, c_gold, cv::DFT_INVERSE | cv::DFT_REAL_OUTPUT | cv::DFT_SCALE);
90 EXPECT_MAT_NEAR(b_gold_roi, cv::Mat(d_b), a.size().area() * 1e-4);
91}
92
93OCL_TEST_P(Dft, R2CthenC2R)
94{
95 cv::Mat a = randomMat(dft_size, CV_32FC1, 0.0, 10.0);
96
97 cv::ocl::oclMat d_b, d_c;
98 cv::ocl::dft(cv::ocl::oclMat(a), d_b, a.size(), 0);
99 cv::ocl::dft(d_b, d_c, a.size(), cv::DFT_SCALE | cv::DFT_INVERSE | cv::DFT_REAL_OUTPUT);
100 EXPECT_MAT_NEAR(a, d_c, a.size().area() * 1e-4);
101}
