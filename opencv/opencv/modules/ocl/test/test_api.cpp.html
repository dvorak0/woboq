42#include "test_precomp.hpp"
43#include "opencv2/ocl/cl_runtime/cl_runtime.hpp" // for OpenCL types & functions

45TEST(TestAPI, openCLExecuteKernelInterop)
46{
47 cv::RNG rng;
48 Size sz(10000, 1);
49 cv::Mat cpuMat = cvtest::randomMat(rng, sz, CV_32FC4, -10, 10, false);
50
51 cv::ocl::oclMat gpuMat(cpuMat);
52 cv::ocl::oclMat gpuMatDst(sz, CV_32FC4);
53
54 const char* kernelStr =
55"__kernel void test_kernel(__global float4* src, __global float4* dst) {\n"
56" int x = get_global_id(0);\n"
57" dst[x] = src[x];\n"
58"}\n";
59
60 cv::ocl::ProgramSource program("test_interop", kernelStr);
61
62 using namespace std;
63 vector<pair<size_t , const void *> > args;
64 args.push_back( make_pair( sizeof(cl_mem), (void *) &gpuMat.data ));
65 args.push_back( make_pair( sizeof(cl_mem), (void *) &gpuMatDst.data ));
66
67 size_t globalThreads[3] = { (size_t)sz.width, 1, 1 };
68 cv::ocl::openCLExecuteKernelInterop(
69 gpuMat.clCxt,
70 program,
71 "test_kernel",
72 globalThreads, NULL, args,
73 -1, -1,
74 "");
75
76 cv::Mat dst;
77 gpuMatDst.download(dst);
78
79 EXPECT_LE(checkNorm(cpuMat, dst), 1e-3);
80}

82// This test must be DISABLED by default!
83// (We can't restore original context for other tests)
84TEST(TestAPI, DISABLED_InitializationFromHandles)
85{
86#define MAX_PLATFORMS 16
87 cl_platform_id platforms[MAX_PLATFORMS] = { NULL };
88 cl_uint numPlatforms = 0;
89 cl_int status = ::clGetPlatformIDs(MAX_PLATFORMS, &platforms[0], &numPlatforms);
90 ASSERT_EQ(CL_SUCCESS, status) << "clGetPlatformIDs";
91 ASSERT_NE(0, (int)numPlatforms);
92
93 int selectedPlatform = 0;
94 cl_platform_id platform = platforms[selectedPlatform];
95
96 ASSERT_NE((void*)NULL, platform);
97
98 cl_device_id device = NULL;
99 status = ::clGetDeviceIDs(platform, CL_DEVICE_TYPE_ALL, 1, &device, NULL);
100 ASSERT_EQ(CL_SUCCESS, status) << "clGetDeviceIDs";
101 ASSERT_NE((void*)NULL, device);
102
103 cl_context_properties cps[3] = { CL_CONTEXT_PLATFORM, (cl_context_properties)(platform), 0 };
104 cl_context context = ::clCreateContext(cps, 1, &device, NULL, NULL, &status);
105 ASSERT_EQ(CL_SUCCESS, status) << "clCreateContext";
106 ASSERT_NE((void*)NULL, context);
107
108 ASSERT_NO_THROW(cv::ocl::initializeContext(&platform, &context, &device));
109
110 status = ::clReleaseContext(context);
111 ASSERT_EQ(CL_SUCCESS, status) << "clReleaseContext";
112
113#ifdef CL_VERSION_1_2
114#if 1
115 {
116 cv::ocl::Context* ctx = cv::ocl::Context::getContext();
117 ASSERT_NE((void*)NULL, ctx);
118 if (ctx->supportsFeature(cv::ocl::FEATURE_CL_VER_1_2)) // device supports OpenCL 1.2+
119 {
120 status = ::clReleaseDevice(device);
121 ASSERT_EQ(CL_SUCCESS, status) << "clReleaseDevice";
122 }
123 }
124#else // code below doesn't work on Linux (SEGFAULTs on 1.1- devices are not handled via exceptions)
125 try
126 {
127 status = ::clReleaseDevice(device); // NOTE This works only with !DEVICES! that supports OpenCL 1.2
128 (void)status; // no check
129 }
130 catch (...)
131 {
132 // nothing, there is no problem
133 }
134#endif
135#endif
136
137 // print the name of current device
138 cv::ocl::Context* ctx = cv::ocl::Context::getContext();
139 ASSERT_NE((void*)NULL, ctx);
140 const cv::ocl::DeviceInfo& deviceInfo = ctx->getDeviceInfo();
141 std::cout << "Device name: " << deviceInfo.deviceName << std::endl;
142 std::cout << "Platform name: " << deviceInfo.platform->platformName << std::endl;
143
144 ASSERT_EQ(context, *(cl_context*)ctx->getOpenCLContextPtr());
145 ASSERT_EQ(device, *(cl_device_id*)ctx->getOpenCLDeviceIDPtr());
146
147 // do some calculations and check results
148 cv::RNG rng;
149 Size sz(100, 100);
150 cv::Mat srcMat = cvtest::randomMat(rng, sz, CV_32FC4, -10, 10, false);
151 cv::Mat dstMat;
152
153 cv::ocl::oclMat srcGpuMat(srcMat);
154 cv::ocl::oclMat dstGpuMat;
155
156 cv::Scalar v = cv::Scalar::all(1);
157 cv::add(srcMat, v, dstMat);
158 cv::ocl::add(srcGpuMat, v, dstGpuMat);
159
160 cv::Mat dstGpuMatMap;
161 dstGpuMat.download(dstGpuMatMap);
162
163 EXPECT_LE(checkNorm(dstMat, dstGpuMatMap), 1e-3);
164}
