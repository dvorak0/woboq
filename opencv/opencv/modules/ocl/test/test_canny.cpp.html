46#include "test_precomp.hpp"
47#ifdef HAVE_OPENCL
48
49////////////////////////////////////////////////////////
50// Canny
51IMPLEMENT_PARAM_CLASS(AppertureSize, int)
52IMPLEMENT_PARAM_CLASS(L2gradient, bool)
53
54PARAM_TEST_CASE(Canny, AppertureSize, L2gradient)
55{
56 int apperture_size;
57 bool useL2gradient;
58
59 cv::Mat edges_gold;
60 virtual void SetUp()
61 {
62 apperture_size = GET_PARAM(0);
63 useL2gradient = GET_PARAM(1);
64 }
65};
66
67OCL_TEST_P(Canny, Accuracy)
68{
69 cv::Mat img = readImage("cv/shared/fruits.png", cv::IMREAD_GRAYSCALE);
70 ASSERT_FALSE(img.empty());
71
72 double low_thresh = 50.0;
73 double high_thresh = 100.0;
74
75 cv::ocl::oclMat ocl_img = cv::ocl::oclMat(img);
76
77 cv::ocl::oclMat edges;
78 cv::ocl::Canny(ocl_img, edges, low_thresh, high_thresh, apperture_size, useL2gradient);
79
80 cv::Mat local_edges_gold;
81 cv::Canny(img, local_edges_gold, low_thresh, high_thresh, apperture_size, useL2gradient);
82
83 EXPECT_MAT_SIMILAR(local_edges_gold, edges, 1e-2);
84}
85
86INSTANTIATE_TEST_CASE_P(OCL_ImgProc, Canny, testing::Combine(
87 testing::Values(AppertureSize(3), AppertureSize(5)),
88 testing::Values(L2gradient(false), L2gradient(true))));
