71PARAM_TEST_CASE(Lut, MatDepth, MatDepth, bool, bool)
72{
73 int lut_depth;
74 int cn;
75 bool use_roi, same_cn;
76
77 // src mat
78 cv::Mat src;
79 cv::Mat lut;
80 cv::Mat dst;
81
82 // src mat with roi
83 cv::Mat src_roi;
84 cv::Mat lut_roi;
85 cv::Mat dst_roi;
86
87 // ocl dst mat for testing
88 cv::ocl::oclMat gsrc_whole;
89 cv::ocl::oclMat glut_whole;
90 cv::ocl::oclMat gdst_whole;
91
92 // ocl mat with roi
93 cv::ocl::oclMat gsrc_roi;
94 cv::ocl::oclMat glut_roi;
95 cv::ocl::oclMat gdst_roi;
96
97 virtual void SetUp()
98 {
99 lut_depth = GET_PARAM(0);
100 cn = GET_PARAM(1);
101 same_cn = GET_PARAM(2);
102 use_roi = GET_PARAM(3);
103 }
104
105 void random_roi()
106 {
107 const int src_type = CV_MAKE_TYPE(CV_8U, cn);
108 const int lut_type = CV_MAKE_TYPE(lut_depth, same_cn ? cn : 1);
109 const int dst_type = CV_MAKE_TYPE(lut_depth, cn);
110
111 Size roiSize = randomSize(1, MAX_VALUE);
112 Border srcBorder = randomBorder(0, use_roi ? MAX_VALUE : 0);
113 randomSubMat(src, src_roi, roiSize, srcBorder, src_type, 0, 256);
114
115 Size lutRoiSize = Size(256, 1);
116 Border lutBorder = randomBorder(0, use_roi ? MAX_VALUE : 0);
117 randomSubMat(lut, lut_roi, lutRoiSize, lutBorder, lut_type, 5, 16);
118
119 Border dstBorder = randomBorder(0, use_roi ? MAX_VALUE : 0);
120 randomSubMat(dst, dst_roi, roiSize, dstBorder, dst_type, 5, 16);
121
122 generateOclMat(gsrc_whole, gsrc_roi, src, roiSize, srcBorder);
123 generateOclMat(glut_whole, glut_roi, lut, lutRoiSize, lutBorder);
124 generateOclMat(gdst_whole, gdst_roi, dst, roiSize, dstBorder);
125 }
126
127 void Near(double threshold = 0.)
128 {
129 Mat whole, roi;
130 gdst_whole.download(whole);
131 gdst_roi.download(roi);
132
133 EXPECT_MAT_NEAR(dst, whole, threshold);
134 EXPECT_MAT_NEAR(dst_roi, roi, threshold);
135 }
136};
137
138OCL_TEST_P(Lut, Mat)
139{
140 for (int j = 0; j < LOOP_TIMES; j++)
141 {
142 random_roi();
143
144 cv::LUT(src_roi, lut_roi, dst_roi);
145 cv::ocl::LUT(gsrc_roi, glut_roi, gdst_roi);
146
147 Near();
148 }
149}

153PARAM_TEST_CASE(ArithmTestBase, MatDepth, Channels, bool)
154{
155 int depth;
156 int cn;
157 bool use_roi;
158 cv::Scalar val;
159
160 // src mat
161 cv::Mat src1;
162 cv::Mat src2;
163 cv::Mat mask;
164 cv::Mat dst1;
165 cv::Mat dst2;
166
167 // src mat with roi
168 cv::Mat src1_roi;
169 cv::Mat src2_roi;
170 cv::Mat mask_roi;
171 cv::Mat dst1_roi;
172 cv::Mat dst2_roi;
173
174 // ocl dst mat for testing
175 cv::ocl::oclMat gsrc1_whole;
176 cv::ocl::oclMat gsrc2_whole;
177 cv::ocl::oclMat gdst1_whole;
178 cv::ocl::oclMat gdst2_whole;
179 cv::ocl::oclMat gmask_whole;
180
181 // ocl mat with roi
182 cv::ocl::oclMat gsrc1_roi;
183 cv::ocl::oclMat gsrc2_roi;
184 cv::ocl::oclMat gdst1_roi;
185 cv::ocl::oclMat gdst2_roi;
186 cv::ocl::oclMat gmask_roi;
187
188 virtual void SetUp()
189 {
190 depth = GET_PARAM(0);
191 cn = GET_PARAM(1);
192 use_roi = GET_PARAM(2);
193 }
194
195 virtual void random_roi()
196 {
197 const int type = CV_MAKE_TYPE(depth, cn);
198
199 Size roiSize = randomSize(1, MAX_VALUE);
200 Border src1Border = randomBorder(0, use_roi ? MAX_VALUE : 0);
201 randomSubMat(src1, src1_roi, roiSize, src1Border, type, -11, 11);
202
203 Border src2Border = randomBorder(0, use_roi ? MAX_VALUE : 0);
204 randomSubMat(src2, src2_roi, roiSize, src2Border, type, -1540, 1740);
205
206 Border dst1Border = randomBorder(0, use_roi ? MAX_VALUE : 0);
207 randomSubMat(dst1, dst1_roi, roiSize, dst1Border, type, 5, 16);
208
209 Border dst2Border = randomBorder(0, use_roi ? MAX_VALUE : 0);
210 randomSubMat(dst2, dst2_roi, roiSize, dst2Border, type, 5, 16);
211
212 Border maskBorder = randomBorder(0, use_roi ? MAX_VALUE : 0);
213 randomSubMat(mask, mask_roi, roiSize, maskBorder, CV_8UC1, 0, 2);
214 cv::threshold(mask, mask, 0.5, 255., CV_8UC1);
215
216
217 generateOclMat(gsrc1_whole, gsrc1_roi, src1, roiSize, src1Border);
218 generateOclMat(gsrc2_whole, gsrc2_roi, src2, roiSize, src2Border);
219 generateOclMat(gdst1_whole, gdst1_roi, dst1, roiSize, dst1Border);
220 generateOclMat(gdst2_whole, gdst2_roi, dst2, roiSize, dst2Border);
221 generateOclMat(gmask_whole, gmask_roi, mask, roiSize, maskBorder);
222
223 val = cv::Scalar(rng.uniform(-100.0, 100.0), rng.uniform(-100.0, 100.0),
224 rng.uniform(-100.0, 100.0), rng.uniform(-100.0, 100.0));
225 }
226
227 void Near(double threshold = 0.)
228 {
229 Mat whole, roi;
230 gdst1_whole.download(whole);
231 gdst1_roi.download(roi);
232
233 EXPECT_MAT_NEAR(dst1, whole, threshold);
234 EXPECT_MAT_NEAR(dst1_roi, roi, threshold);
235 }
236
237 void Near1(double threshold = 0.)
238 {
239 Mat whole, roi;
240 gdst2_whole.download(whole);
241 gdst2_roi.download(roi);
242
243 EXPECT_MAT_NEAR(dst2, whole, threshold);
244 EXPECT_MAT_NEAR(dst2_roi, roi, threshold);
245 }
246};

702//////////////////////////////// MinMax /////////////////////////////////////////////////
703
704typedef ArithmTestBase MinMax;
705
706OCL_TEST_P(MinMax, MAT)
707{
708 for (int j = 0; j < LOOP_TIMES; j++)
709 {
710 random_roi();
711
712 double minVal, maxVal;
713
714 if (src1.depth() != CV_8S)
715 cv::minMaxIdx(src1_roi, &minVal, &maxVal, NULL, NULL);
716 else
717 {
718 minVal = std::numeric_limits<double>::max();
719 maxVal = -std::numeric_limits<double>::max();
720 for (int i2 = 0; i2 < src1_roi.rows; ++i2)
721 for (int j2 = 0; j2 < src1_roi.cols; ++j2)
722 {
723 signed char oneVal = src1_roi.at<signed char>(i2, j2);
724 if (oneVal < minVal) minVal = oneVal;
725 if (oneVal > maxVal) maxVal = oneVal;
726 }
727 }
728
729 double minVal_, maxVal_;
730 cv::ocl::minMax(gsrc1_roi, &minVal_, &maxVal_);
731
732 EXPECT_DOUBLE_EQ(minVal_, minVal);
