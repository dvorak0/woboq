76cv::ocl::oclMat createMat_ocl(cv::RNG& rng, Size size, int type, bool useRoi)
77{
78 Size size0 = size;
79
80 if (useRoi)
81 {
82 size0.width += rng.uniform(5, 15);
83 size0.height += rng.uniform(5, 15);
84 }
85
86 cv::ocl::oclMat d_m(size0, type);
87
88 if (size0 != size)
89 d_m = d_m(Rect((size0.width - size.width) / 2, (size0.height - size.height) / 2, size.width, size.height));
90
91 return d_m;
92}

94cv::ocl::oclMat loadMat_ocl(cv::RNG& rng, const Mat& m, bool useRoi)
95{
96 CV_Assert(m.type() == CV_8UC1 || m.type() == CV_8UC3);
97 cv::ocl::oclMat d_m;
98 d_m = createMat_ocl(rng, m.size(), m.type(), useRoi);
99
100 Size ls;
101 Point pt;
102
103 d_m.locateROI(ls, pt);
104
105 Rect roi(pt.x, pt.y, d_m.size().width, d_m.size().height);
106
107 cv::ocl::oclMat m_ocl(m);
108
109 cv::ocl::oclMat d_m_roi(d_m, roi);
110
111 m_ocl.copyTo(d_m);
112 return d_m;
113}

115vector<MatType> types(int depth_start, int depth_end, int cn_start, int cn_end)
116{
117 vector<MatType> v;
118
119 v.reserve((depth_end - depth_start + 1) * (cn_end - cn_start + 1));
120
121 for (int depth = depth_start; depth <= depth_end; ++depth)
122 {
123 for (int cn = cn_start; cn <= cn_end; ++cn)
124 {
125 v.push_back(CV_MAKETYPE(depth, cn));
126 }
127 }
128
129 return v;
130}

144Mat readImageType(const string &fname, int type)
145{
146 Mat src = readImage(fname, CV_MAT_CN(type) == 1 ? IMREAD_GRAYSCALE : IMREAD_COLOR);
147 if (CV_MAT_CN(type) == 4)
148 {
149 Mat temp;
150 cvtColor(src, temp, cv::COLOR_BGR2BGRA);
151 swap(src, temp);
152 }
153 src.convertTo(src, CV_MAT_DEPTH(type));
154 return src;
155}

167double checkSimilarity(const Mat &m1, const Mat &m2)
168{
169 Mat diff;
170 matchTemplate(m1, m2, diff, CV_TM_CCORR_NORMED);
171 return std::abs(diff.at<float>(0, 0) - 1.f);
172}

189double checkRectSimilarity(Size sz, std::vector<Rect>& ob1, std::vector<Rect>& ob2)
190{
191 double final_test_result = 0.0;
192 size_t sz1 = ob1.size();
193 size_t sz2 = ob2.size();
194
195 if(sz1 != sz2)
196 {
197 return sz1 > sz2 ? (double)(sz1 - sz2) : (double)(sz2 - sz1);
198 }
199 else
200 {
201 if(sz1==0 && sz2==0)
202 return 0;
203 cv::Mat cpu_result(sz, CV_8UC1);
204 cpu_result.setTo(0);
205
206 for(vector<Rect>::const_iterator r = ob1.begin(); r != ob1.end(); r++)
207 {
208 cv::Mat cpu_result_roi(cpu_result, *r);
209 cpu_result_roi.setTo(1);
210 cpu_result.copyTo(cpu_result);
211 }
212 int cpu_area = cv::countNonZero(cpu_result > 0);
213
214 cv::Mat gpu_result(sz, CV_8UC1);
215 gpu_result.setTo(0);
216 for(vector<Rect>::const_iterator r2 = ob2.begin(); r2 != ob2.end(); r2++)
217 {
218 cv::Mat gpu_result_roi(gpu_result, *r2);
219 gpu_result_roi.setTo(1);
220 gpu_result.copyTo(gpu_result);
221 }
222
223 cv::Mat result_;
224 multiply(cpu_result, gpu_result, result_);
225 int result = cv::countNonZero(result_ > 0);
226 if(cpu_area!=0 && result!=0)
227 final_test_result = 1.0 - (double)result/(double)cpu_area;
228 else if(cpu_area==0 && result!=0)
229 final_test_result = -1;
230 }
231 return final_test_result;
232}

234void showDiff(const Mat& src, const Mat& gold, const Mat& actual, double eps, bool alwaysShow)
235{
236 Mat diff, diff_thresh;
237 absdiff(gold, actual, diff);
238 diff.convertTo(diff, CV_32F);
239 threshold(diff, diff_thresh, eps, 255.0, cv::THRESH_BINARY);
240
241 if (alwaysShow || cv::countNonZero(diff_thresh.reshape(1)) > 0)
242 {
243#if 0
244 std::cout << "Src: " << std::endl << src << std::endl;
245 std::cout << "Reference: " << std::endl << gold << std::endl;
246 std::cout << "OpenCL: " << std::endl << actual << std::endl;
247#endif
248
249 namedWindow("src", WINDOW_NORMAL);
250 namedWindow("gold", WINDOW_NORMAL);
251 namedWindow("actual", WINDOW_NORMAL);
252 namedWindow("diff", WINDOW_NORMAL);
253
254 imshow("src", src);
255 imshow("gold", gold);
256 imshow("actual", actual);
257 imshow("diff", diff);
258
259 waitKey();
260 }
