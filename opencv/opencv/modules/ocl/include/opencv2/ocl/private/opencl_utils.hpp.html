51inline cl_int getPlatforms(std::vector<cl_platform_id>& platforms)
52{
53 cl_uint n = 0;
54
55 cl_int err = ::clGetPlatformIDs(0, NULL, &n);
56 if (err != CL_SUCCESS)
57 return err;
58
59 platforms.clear(); platforms.resize(n);
60 err = ::clGetPlatformIDs(n, &platforms[0], NULL);
61 if (err != CL_SUCCESS)
62 return err;
63
64 return CL_SUCCESS;
65}
66
67inline cl_int getDevices(cl_platform_id platform, cl_device_type type, std::vector<cl_device_id>& devices)
68{
69 cl_uint n = 0;
70
71 cl_int err = ::clGetDeviceIDs(platform, type, 0, NULL, &n);
72 if (err != CL_SUCCESS)
73 return err;
74
75 devices.clear(); devices.resize(n);
76 err = ::clGetDeviceIDs(platform, type, n, &devices[0], NULL);
77 if (err != CL_SUCCESS)
78 return err;
79
80 return CL_SUCCESS;
81}

86template <typename Functor, typename ObjectType, typename T>
87inline cl_int getScalarInfo(Functor f, ObjectType obj, cl_uint name, T& param)
88{
89 return f(obj, name, sizeof(T), &param, NULL);
90}
91
92template <typename Functor, typename ObjectType>
93inline cl_int getStringInfo(Functor f, ObjectType obj, cl_uint name, std::string& param)
94{
95 ::size_t required;
96 cl_int err = f(obj, name, 0, NULL, &required);
97 if (err != CL_SUCCESS)
98 return err;
99
100 param.clear();
101 if (required > 0)
102 {
103 std::vector<char> buf(required + 1, char(0));
104 err = f(obj, name, required, &buf[0], NULL);
105 if (err != CL_SUCCESS)
106 return err;
107 param = &buf[0];
108 }
109
110 return CL_SUCCESS;
111}
