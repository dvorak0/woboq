147//Represents an image texture object
148class CV_EXPORTS TextureCL
149{
150public:
151 TextureCL(cl_mem tex, int r, int c, int t)
152 : tex_(tex), rows(r), cols(c), type(t) {}
153 ~TextureCL()
154 {
155 openCLFree(tex_);
156 }
157 operator cl_mem()
158 {
159 return tex_;
160 }
161 cl_mem const tex_;
162 const int rows;
163 const int cols;
164 const int type;
165private:
166 //disable assignment
167 void operator=(const TextureCL&);
168};

179inline size_t divUp(size_t total, size_t grain)
180{
181 return (total + grain - 1) / grain;
182}
183
184inline size_t roundUp(size_t sz, size_t n)
185{
186 // we don't assume that n is a power of 2 (see alignSize)
187 // equal to divUp(sz, n) * n
188 size_t t = sz + n - 1;
189 size_t rem = t % n;
190 size_t result = t - rem;
191 return result;
192}
