56static std::string bytesToStringRepr(size_t value)
57{
58 size_t b = value % 1024;
59 value /= 1024;
60
61 size_t kb = value % 1024;
62 value /= 1024;
63
64 size_t mb = value % 1024;
65 value /= 1024;
66
67 size_t gb = value;
68
69 std::ostringstream stream;
70
71 if (gb > 0)
72 stream << gb << " GB ";
73 if (mb > 0)
74 stream << mb << " MB ";
75 if (kb > 0)
76 stream << kb << " kB ";
77 if (b > 0)
78 stream << b << " B";
79
80 return stream.str();
81}

83static void dumpOpenCLDevice()
84{
85 using namespace cv::ocl;
86 try
87 {
88 cv::ocl::PlatformsInfo platforms;
89 cv::ocl::getOpenCLPlatforms(platforms);
90 DUMP_MESSAGE_STDOUT("OpenCL Platforms: ");
91 const char* deviceTypeStr;
92 for(unsigned int i=0; i < platforms.size(); i++)
93 {
94 DUMP_MESSAGE_STDOUT(" " << platforms.at(i)->platformName);
95 const cv::ocl::DevicesInfo& devices = platforms.at(i)->devices;
96 for(unsigned int j=0; j < devices.size(); j++)
97 {
98 const cv::ocl::DeviceInfo& current_device = *devices.at(j);
99 deviceTypeStr = current_device.deviceType == CVCL_DEVICE_TYPE_CPU
100 ? ("CPU") : (current_device.deviceType == CVCL_DEVICE_TYPE_GPU ? "GPU" : "unknown");
101 DUMP_MESSAGE_STDOUT( " " << deviceTypeStr << " : " << current_device.deviceName << " : " << current_device.deviceVersion );
102 DUMP_PROPERTY_XML("cv_ocl_platform_"<< i<<"_device_"<<j, "(Platform=" << current_device.platform->platformName << ")(Type="
103 << deviceTypeStr <<")(Name="<< current_device.deviceName << ")(Version="<< current_device.deviceVersion<<")");
104 }
105 }
106 DUMP_MESSAGE_STDOUT("Current OpenCL device: ");
107
108 const cv::ocl::DeviceInfo& deviceInfo = cv::ocl::Context::getContext()->getDeviceInfo();
109
110 DUMP_MESSAGE_STDOUT(" Platform = "<< deviceInfo.platform->platformName);
111 DUMP_PROPERTY_XML("cv_ocl_current_platformName", deviceInfo.platform->platformName);
112
113 deviceTypeStr = deviceInfo.deviceType == CVCL_DEVICE_TYPE_CPU
114 ? "CPU" : (deviceInfo.deviceType == CVCL_DEVICE_TYPE_GPU ? "GPU" : "unknown");
115 DUMP_MESSAGE_STDOUT(" Type = "<< deviceTypeStr);
116 DUMP_PROPERTY_XML("cv_ocl_current_deviceType", deviceTypeStr);
117
118 DUMP_MESSAGE_STDOUT(" Name = "<< deviceInfo.deviceName);
119 DUMP_PROPERTY_XML("cv_ocl_current_deviceName", deviceInfo.deviceName);
120
121 DUMP_MESSAGE_STDOUT(" Version = " << deviceInfo.deviceVersion);
122 DUMP_PROPERTY_XML("cv_ocl_current_deviceVersion", deviceInfo.deviceVersion);
123
124 DUMP_MESSAGE_STDOUT(" Compute units = "<< deviceInfo.maxComputeUnits);
125 DUMP_PROPERTY_XML("cv_ocl_current_maxComputeUnits", deviceInfo.maxComputeUnits);
126
127 DUMP_MESSAGE_STDOUT(" Max work group size = "<< deviceInfo.maxWorkGroupSize);
128 DUMP_PROPERTY_XML("cv_ocl_current_maxWorkGroupSize", deviceInfo.maxWorkGroupSize);
129
130 std::string localMemorySizeStr = bytesToStringRepr(deviceInfo.localMemorySize);
131 DUMP_MESSAGE_STDOUT(" Local memory size = "<< localMemorySizeStr.c_str());
132 DUMP_PROPERTY_XML("cv_ocl_current_localMemorySize", deviceInfo.localMemorySize);
133
134 std::string maxMemAllocSizeStr = bytesToStringRepr(deviceInfo.maxMemAllocSize);
135 DUMP_MESSAGE_STDOUT(" Max memory allocation size = "<< maxMemAllocSizeStr.c_str());
136 DUMP_PROPERTY_XML("cv_ocl_current_maxMemAllocSize", deviceInfo.maxMemAllocSize);
137
138 const char* doubleSupportStr = deviceInfo.haveDoubleSupport ? "Yes" : "No";
139 DUMP_MESSAGE_STDOUT(" Double support = "<< doubleSupportStr);
140 DUMP_PROPERTY_XML("cv_ocl_current_haveDoubleSupport", deviceInfo.haveDoubleSupport);
141
142 const char* isUnifiedMemoryStr = deviceInfo.isUnifiedMemory ? "Yes" : "No";
143 DUMP_MESSAGE_STDOUT(" Unified memory = "<< isUnifiedMemoryStr);
144 DUMP_PROPERTY_XML("cv_ocl_current_isUnifiedMemory", deviceInfo.isUnifiedMemory);
145 }
146 catch (...)
147 {
148 DUMP_MESSAGE_STDOUT("OpenCL device not available");
149 DUMP_PROPERTY_XML("cv_ocl", "not available");
150 }
151}
