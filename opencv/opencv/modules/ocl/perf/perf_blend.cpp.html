55template <typename T>
56static void blendLinearGold(const Mat &img1, const Mat &img2,
57 const Mat &weights1, const Mat &weights2,
58 Mat &result_gold)
59{
60 CV_Assert(img1.size() == img2.size() && img1.type() == img2.type());
61 CV_Assert(weights1.size() == weights2.size() && weights1.size() == img1.size() &&
62 weights1.type() == CV_32FC1 && weights2.type() == CV_32FC1);
63
64 result_gold.create(img1.size(), img1.type());
65
66 int cn = img1.channels();
67 int step1 = img1.cols * img1.channels();
68
69 for (int y = 0; y < img1.rows; ++y)
70 {
71 const float * const weights1_row = weights1.ptr<float>(y);
72 const float * const weights2_row = weights2.ptr<float>(y);
73 const T * const img1_row = img1.ptr<T>(y);
74 const T * const img2_row = img2.ptr<T>(y);
75 T * const result_gold_row = result_gold.ptr<T>(y);
76
77 for (int x = 0; x < step1; ++x)
78 {
79 int x1 = x / cn;
80 float w1 = weights1_row[x1], w2 = weights2_row[x1];
81 result_gold_row[x] = saturate_cast<T>(((float)img1_row[x] * w1
82 + (float)img2_row[x] * w2) / (w1 + w2 + 1e-5f));
83 }
84 }
85}
86
87typedef void (*blendFunction)(const Mat &img1, const Mat &img2,
88 const Mat &weights1, const Mat &weights2,
89 Mat &result_gold);
90
91typedef Size_MatType BlendLinearFixture;
92
93OCL_PERF_TEST_P(BlendLinearFixture, BlendLinear,
94 ::testing::Combine(OCL_TEST_SIZES, OCL_PERF_ENUM(CV_32FC1, CV_32FC4)))
95{
96 Size_MatType_t params = GetParam();
97 const Size srcSize = get<0>(params);
98 const int srcType = get<1>(params);
99 const double eps = CV_MAT_DEPTH(srcType) <= CV_32S ? 1.0 : 0.2;
100
101 Mat src1(srcSize, srcType), src2(srcSize, srcType), dst(srcSize, srcType);
102 Mat weights1(srcSize, CV_32FC1), weights2(srcSize, CV_32FC1);
103
104 declare.in(src1, src2, WARMUP_RNG).out(dst);
105 randu(weights1, 0.0f, 1.0f);
106 randu(weights2, 0.0f, 1.0f);
107
108 if (RUN_OCL_IMPL)
109 {
110 ocl::oclMat oclSrc1(src1), oclSrc2(src2), oclDst;
111 ocl::oclMat oclWeights1(weights1), oclWeights2(weights2);
112
113 OCL_TEST_CYCLE() ocl::blendLinear(oclSrc1, oclSrc2, oclWeights1, oclWeights2, oclDst);
114
115 oclDst.download(dst);
116
117 SANITY_CHECK(dst, eps);
118 }
119 else if (RUN_PLAIN_IMPL)
120 {
121 blendFunction funcs[] = { (blendFunction)blendLinearGold<uchar>, (blendFunction)blendLinearGold<float> };
122 int funcIdx = CV_MAT_DEPTH(srcType) == CV_8UC1 ? 0 : 1;
123
124 TEST_CYCLE() (funcs[funcIdx])(src1, src2, weights1, weights2, dst);
125
126 SANITY_CHECK(dst, eps);
127 }
128 else
129 OCL_PERF_ELSE
130}
