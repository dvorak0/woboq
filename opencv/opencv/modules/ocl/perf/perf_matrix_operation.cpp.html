52///////////// ConvertTo////////////////////////
53
54typedef Size_MatType ConvertToFixture;
55
56OCL_PERF_TEST_P(ConvertToFixture, ConvertTo, ::testing::Combine(OCL_TEST_SIZES, OCL_TEST_TYPES))
57{
58 const Size_MatType_t params = GetParam();
59 const Size srcSize = get<0>(params);
60 const int type = get<1>(params);
61
62 Mat src(srcSize, type), dst;
63 const int dstType = CV_MAKE_TYPE(CV_32F, src.channels());
64
65 checkDeviceMaxMemoryAllocSize(srcSize, type);
66 checkDeviceMaxMemoryAllocSize(srcSize, dstType);
67
68 dst.create(srcSize, dstType);
69 declare.in(src, WARMUP_RNG).out(dst);
70
71 if (RUN_OCL_IMPL)
72 {
73 ocl::oclMat oclSrc(src), oclDst(srcSize, dstType);
74
75 OCL_TEST_CYCLE() oclSrc.convertTo(oclDst, dstType);
76
77 oclDst.download(dst);
78
79 SANITY_CHECK(dst);
80 }
81 else if (RUN_PLAIN_IMPL)
82 {
83 TEST_CYCLE() src.convertTo(dst, dstType);
84
85 SANITY_CHECK(dst);
86 }
87 else
88 OCL_PERF_ELSE
89}
