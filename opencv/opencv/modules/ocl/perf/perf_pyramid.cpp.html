52///////////// pyrDown //////////////////////
53
54typedef Size_MatType PyrDownFixture;
55
56OCL_PERF_TEST_P(PyrDownFixture, PyrDown,
57 ::testing::Combine(OCL_TEST_SIZES, OCL_TEST_TYPES))
58{
59 const Size_MatType_t params = GetParam();
60 const Size srcSize = get<0>(params);
61 const int type = get<1>(params);
62 Size dstSize((srcSize.height + 1) >> 1, (srcSize.width + 1) >> 1);
63
64 checkDeviceMaxMemoryAllocSize(srcSize, type);
65 checkDeviceMaxMemoryAllocSize(dstSize, type);
66
67 Mat src(srcSize, type), dst;
68 dst.create(dstSize, type);
69 declare.in(src, WARMUP_RNG).out(dst);
70
71 if (RUN_OCL_IMPL)
72 {
73 ocl::oclMat oclSrc(src), oclDst(dstSize, type);
74
75 OCL_TEST_CYCLE() ocl::pyrDown(oclSrc, oclDst);
76
77 oclDst.download(dst);
78
79 SANITY_CHECK(dst, 5e-4);
80 }
81 else if (RUN_PLAIN_IMPL)
82 {
83 TEST_CYCLE() pyrDown(src, dst);
84
85 SANITY_CHECK(dst);
86 }
87 else
88 OCL_PERF_ELSE
89}
