48using namespace perf;
49using std::tr1::tuple;
50using std::tr1::get;
51using std::tr1::make_tuple;

53///////////// cvtColor////////////////////////
54
55CV_ENUM(ConversionTypes, COLOR_RGB2GRAY, COLOR_RGB2BGR, COLOR_RGB2YUV, COLOR_YUV2RGB, COLOR_RGB2YCrCb,
56 COLOR_YCrCb2RGB, COLOR_RGB2XYZ, COLOR_XYZ2RGB, COLOR_RGB2HSV, COLOR_HSV2RGB, COLOR_RGB2HLS,
57 COLOR_HLS2RGB, COLOR_BGR5652BGR, COLOR_BGR2BGR565, COLOR_RGBA2mRGBA, COLOR_mRGBA2RGBA, COLOR_YUV2RGB_NV12)
58
59typedef tuple<Size, tuple<ConversionTypes, int, int> > CvtColorParams;
60typedef TestBaseWithParam<CvtColorParams> CvtColorFixture;
61
62OCL_PERF_TEST_P(CvtColorFixture, CvtColor, testing::Combine(
63 OCL_TEST_SIZES,
64 testing::Values(
65 make_tuple(ConversionTypes(COLOR_RGB2GRAY), 3, 1),
66 make_tuple(ConversionTypes(COLOR_RGB2BGR), 3, 3),
67 make_tuple(ConversionTypes(COLOR_RGB2YUV), 3, 3),
68 make_tuple(ConversionTypes(COLOR_YUV2RGB), 3, 3),
69 make_tuple(ConversionTypes(COLOR_RGB2YCrCb), 3, 3),
70 make_tuple(ConversionTypes(COLOR_YCrCb2RGB), 3, 3),
71 make_tuple(ConversionTypes(COLOR_RGB2XYZ), 3, 3),
72 make_tuple(ConversionTypes(COLOR_XYZ2RGB), 3, 3),
73 make_tuple(ConversionTypes(COLOR_RGB2HSV), 3, 3),
74 make_tuple(ConversionTypes(COLOR_HSV2RGB), 3, 3),
75 make_tuple(ConversionTypes(COLOR_RGB2HLS), 3, 3),
76 make_tuple(ConversionTypes(COLOR_HLS2RGB), 3, 3),
77 make_tuple(ConversionTypes(COLOR_BGR5652BGR), 2, 3),
78 make_tuple(ConversionTypes(COLOR_BGR2BGR565), 3, 2),
79 make_tuple(ConversionTypes(COLOR_RGBA2mRGBA), 4, 4),
80 make_tuple(ConversionTypes(COLOR_mRGBA2RGBA), 4, 4),
81 make_tuple(ConversionTypes(COLOR_YUV2RGB_NV12), 1, 3)
82 )))
83{
84 CvtColorParams params = GetParam();
85 const Size srcSize = get<0>(params);
86 const tuple<int, int, int> conversionParams = get<1>(params);
87 const int code = get<0>(conversionParams), scn = get<1>(conversionParams),
88 dcn = get<2>(conversionParams);
89
90 Mat src(srcSize, CV_8UC(scn)), dst(srcSize, CV_8UC(scn));
91 declare.in(src, WARMUP_RNG).out(dst);
92
93 if (RUN_OCL_IMPL)
94 {
95 ocl::oclMat oclSrc(src), oclDst(src.size(), dst.type());
96
97 OCL_TEST_CYCLE() ocl::cvtColor(oclSrc, oclDst, code, dcn);
98 oclDst.download(dst);
99
100 SANITY_CHECK(dst, 1);
101 }
102 else if (RUN_PLAIN_IMPL)
103 {
104 TEST_CYCLE() cv::cvtColor(src, dst, code, dcn);
105
106 SANITY_CHECK(dst);
107 }
108 else
109 OCL_PERF_ELSE
110}
