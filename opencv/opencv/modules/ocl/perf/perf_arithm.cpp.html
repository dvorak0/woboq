56
57OCL_PERF_TEST_P(LUTFixture, LUT,
58 ::testing::Combine(OCL_TEST_SIZES, OCL_PERF_ENUM(CV_8UC1, CV_32FC1)))
59{
60 // getting params
61 const Size_MatType_t params = GetParam();
62 const Size srcSize = get<0>(params);
63 const int type = get<1>(params), cn = CV_MAT_CN(type);
64
65 // creating src data
66 Mat src(srcSize, CV_8UC(cn)), lut(1, 256, type);
67 int dstType = CV_MAKETYPE(lut.depth(), src.channels());
68 Mat dst(srcSize, dstType);
69
70 declare.in(src, lut, WARMUP_RNG).out(dst);
71
72 // select implementation
73 if (RUN_OCL_IMPL)
74 {
75 ocl::oclMat oclSrc(src), oclLut(lut), oclDst(srcSize, dstType);
76
77 OCL_TEST_CYCLE() cv::ocl::LUT(oclSrc, oclLut, oclDst);
78 oclDst.download(dst);
79
80 SANITY_CHECK(dst);
81 }
82 else if (RUN_PLAIN_IMPL)
83 {
84 TEST_CYCLE() cv::LUT(src, lut, dst);
85
86 SANITY_CHECK(dst);
87 }
88 else
89 OCL_PERF_ELSE
90}
