51PERF_TEST(StereoMatchBMFixture, StereoMatchBM)
52{
53 Mat left_image = imread(getDataPath("gpu/stereobm/aloe-L.png"), cv::IMREAD_GRAYSCALE);
54 Mat right_image = imread(getDataPath("gpu/stereobm/aloe-R.png"), cv::IMREAD_GRAYSCALE);
55
56 ASSERT_TRUE(!left_image.empty()) << "no input image";
57 ASSERT_TRUE(!right_image.empty()) << "no input image";
58 ASSERT_TRUE(right_image.size() == left_image.size());
59 ASSERT_TRUE(right_image.size() == left_image.size());
60
61 const int n_disp = 128, winSize = 19;
62 Mat disp(left_image.size(), CV_16SC1);
63
64 declare.in(left_image, right_image).out(disp);
65
66 if (RUN_OCL_IMPL)
67 {
68 ocl::oclMat oclLeft(left_image), oclRight(right_image),
69 oclDisp(left_image.size(), CV_16SC1);
70 ocl::StereoBM_OCL oclBM(0, n_disp, winSize);
71
72 OCL_TEST_CYCLE() oclBM(oclLeft, oclRight, oclDisp);
73 }
74 else if (RUN_PLAIN_IMPL)
75 {
76 StereoBM bm(0, n_disp, winSize);
77
78 TEST_CYCLE() bm(left_image, right_image, disp);
79 }
80 else
81 OCL_PERF_ELSE
82
83 int value = 0;
84 SANITY_CHECK(value);
85}
