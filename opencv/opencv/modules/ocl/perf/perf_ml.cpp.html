53static void genData(Mat& trainData, Size size, Mat& trainLabel = Mat().setTo(Scalar::all(0)), int nClasses = 0)
54{
55 trainData.create(size, CV_32FC1);
56 randu(trainData, 1.0, 100.0);
57
58 if (nClasses != 0)
59 {
60 trainLabel.create(size.height, 1, CV_8UC1);
61 randu(trainLabel, 0, nClasses - 1);
62 trainLabel.convertTo(trainLabel, CV_32FC1);
63 }
64}

66typedef tuple<int> KNNParamType;
67typedef TestBaseWithParam<KNNParamType> KNNFixture;
68
69PERF_TEST_P(KNNFixture, KNN,
70 testing::Values(1000, 2000, 4000))
71{
72 KNNParamType params = GetParam();
73 const int rows = get<0>(params);
74 int columns = 100;
75 int k = rows/250;
76
77 Mat trainData, trainLabels;
78 Size size(columns, rows);
79 genData(trainData, size, trainLabels, 3);
80
81 Mat testData;
82 genData(testData, size);
83 Mat best_label;
84
85 if (RUN_PLAIN_IMPL)
86 {
87 TEST_CYCLE()
88 {
89 CvKNearest knn_cpu;
90 knn_cpu.train(trainData, trainLabels);
91 knn_cpu.find_nearest(testData, k, &best_label);
92 }
93 }
94 else if (RUN_OCL_IMPL)
95 {
96 cv::ocl::oclMat best_label_ocl;
97 cv::ocl::oclMat testdata;
98 testdata.upload(testData);
99
100 OCL_TEST_CYCLE()
101 {
102 cv::ocl::KNearestNeighbour knn_ocl;
103 knn_ocl.train(trainData, trainLabels);
104 knn_ocl.find_nearest(testdata, k, best_label_ocl);
105 }
106 best_label_ocl.download(best_label);
107 }
108 else
109 OCL_PERF_ELSE
110 SANITY_CHECK(best_label);
111}

114typedef TestBaseWithParam<tuple<int> > SVMFixture;
115
116// code is based on: samples\cpp\tutorial_code\ml\non_linear_svms\non_linear_svms.cpp
117PERF_TEST_P(SVMFixture, DISABLED_SVM,
118 testing::Values(50, 100))
119{
120
121 const int NTRAINING_SAMPLES = get<0>(GetParam()); // Number of training samples per class
122
123 #define FRAC_LINEAR_SEP 0.9f // Fraction of samples which compose the linear separable part
124
125 const int WIDTH = 512, HEIGHT = 512;
126
127 Mat trainData(2*NTRAINING_SAMPLES, 2, CV_32FC1);
128 Mat labels (2*NTRAINING_SAMPLES, 1, CV_32FC1);
129
130 RNG rng(100); // Random value generation class
131
132 // Set up the linearly separable part of the training data
133 int nLinearSamples = (int) (FRAC_LINEAR_SEP * NTRAINING_SAMPLES);
134
135 // Generate random points for the class 1
136 Mat trainClass = trainData.rowRange(0, nLinearSamples);
137 // The x coordinate of the points is in [0, 0.4)
138 Mat c = trainClass.colRange(0, 1);
139 rng.fill(c, RNG::UNIFORM, Scalar(1), Scalar(0.4 * WIDTH));
140 // The y coordinate of the points is in [0, 1)
141 c = trainClass.colRange(1,2);
142 rng.fill(c, RNG::UNIFORM, Scalar(1), Scalar(HEIGHT));
143
144 // Generate random points for the class 2
145 trainClass = trainData.rowRange(2*NTRAINING_SAMPLES-nLinearSamples, 2*NTRAINING_SAMPLES);
146 // The x coordinate of the points is in [0.6, 1]
147 c = trainClass.colRange(0 , 1);
148 rng.fill(c, RNG::UNIFORM, Scalar(0.6*WIDTH), Scalar(WIDTH));
149 // The y coordinate of the points is in [0, 1)
150 c = trainClass.colRange(1,2);
151 rng.fill(c, RNG::UNIFORM, Scalar(1), Scalar(HEIGHT));
152
153 //------------------ Set up the non-linearly separable part of the training data ---------------
154
155 // Generate random points for the classes 1 and 2
156 trainClass = trainData.rowRange( nLinearSamples, 2*NTRAINING_SAMPLES-nLinearSamples);
157 // The x coordinate of the points is in [0.4, 0.6)
158 c = trainClass.colRange(0,1);
159 rng.fill(c, RNG::UNIFORM, Scalar(0.4*WIDTH), Scalar(0.6*WIDTH));
160 // The y coordinate of the points is in [0, 1)
161 c = trainClass.colRange(1,2);
162 rng.fill(c, RNG::UNIFORM, Scalar(1), Scalar(HEIGHT));
163
164 //------------------------- Set up the labels for the classes ---------------------------------
165 labels.rowRange( 0, NTRAINING_SAMPLES).setTo(1); // Class 1
166 labels.rowRange(NTRAINING_SAMPLES, 2*NTRAINING_SAMPLES).setTo(2); // Class 2
167
168 //------------------------ Set up the support vector machines parameters --------------------
169 CvSVMParams params;
170 params.svm_type = SVM::C_SVC;
171 params.C = 0.1;
172 params.kernel_type = SVM::LINEAR;
173 params.term_crit = TermCriteria(CV_TERMCRIT_ITER, (int)1e7, 1e-6);
174
175 Mat dst = Mat::zeros(HEIGHT, WIDTH, CV_8UC1);
176
177 Mat samples(WIDTH*HEIGHT, 2, CV_32FC1);
178 int k = 0;
179 for (int i = 0; i < HEIGHT; ++i)
180 {
181 for (int j = 0; j < WIDTH; ++j)
182 {
183 samples.at<float>(k, 0) = (float)i;
184 samples.at<float>(k, 0) = (float)j;
185 k++;
186 }
187 }
188 Mat results(WIDTH*HEIGHT, 1, CV_32FC1);
189
190 CvMat samples_ = samples;
191 CvMat results_ = results;
192
193 if (RUN_PLAIN_IMPL)
194 {
195 CvSVM svm;
196 svm.train(trainData, labels, Mat(), Mat(), params);
197 TEST_CYCLE()
198 {
199 svm.predict(&samples_, &results_);
200 }
201 }
202 else if (RUN_OCL_IMPL)
203 {
204 CvSVM_OCL svm;
205 svm.train(trainData, labels, Mat(), Mat(), params);
206 OCL_TEST_CYCLE()
207 {
208 svm.predict(&samples_, &results_);
209 }
210 }
211 else
212 OCL_PERF_ELSE
213
214 SANITY_CHECK_NOTHING();
215}
