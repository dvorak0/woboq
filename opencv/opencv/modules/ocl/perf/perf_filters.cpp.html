49using std::tr1::get;

52typedef tuple<Size, MatType, int> FilterParams;
53typedef TestBaseWithParam<FilterParams> FilterFixture;

55///////////// Blur////////////////////////
56
57typedef FilterFixture BlurFixture;
58
59OCL_PERF_TEST_P(BlurFixture, Blur,
60 ::testing::Combine(OCL_TEST_SIZES, OCL_TEST_TYPES, OCL_PERF_ENUM(3, 5)))
61{
62 const FilterParams params = GetParam();
63 const Size srcSize = get<0>(params);
64 const int type = get<1>(params), ksize = get<2>(params), bordertype = BORDER_CONSTANT;
65
66 checkDeviceMaxMemoryAllocSize(srcSize, type);
67
68 Mat src(srcSize, type), dst(srcSize, type);
69 declare.in(src, WARMUP_RNG).out(dst);
70
71 if (RUN_OCL_IMPL)
72 {
73 ocl::oclMat oclSrc(src), oclDst(srcSize, type);
74
75 OCL_TEST_CYCLE() cv::ocl::blur(oclSrc, oclDst, Size(ksize, ksize), Point(-1, -1), bordertype);
76
77 oclDst.download(dst);
78
79 SANITY_CHECK(dst, 1 + DBL_EPSILON);
80 }
81 else if (RUN_PLAIN_IMPL)
82 {
83 TEST_CYCLE() cv::blur(src, dst, Size(ksize, ksize), Point(-1, -1), bordertype);
84
85 SANITY_CHECK(dst, 1 + DBL_EPSILON);
86 }
87 else
88 OCL_PERF_ELSE
89}
