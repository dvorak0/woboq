48using namespace perf;
49using std::tr1::tuple;
50using std::tr1::get;

52///////////// Merge////////////////////////
53
54typedef tuple<Size, MatDepth, int> MergeParams;
55typedef TestBaseWithParam<MergeParams> MergeFixture;
56
57OCL_PERF_TEST_P(MergeFixture, Merge,
58 ::testing::Combine(OCL_TEST_SIZES, OCL_PERF_ENUM(CV_8U, CV_32F),
59 OCL_PERF_ENUM(2, 3)))
60{
61 const MergeParams params = GetParam();
62 const Size srcSize = get<0>(params);
63 const int depth = get<1>(params), cn = get<2>(params),
64 dtype = CV_MAKE_TYPE(depth, cn);
65
66 checkDeviceMaxMemoryAllocSize(srcSize, dtype);
67
68 Mat dst(srcSize, dtype);
69 vector<Mat> src(cn);
70 for (vector<Mat>::iterator i = src.begin(), end = src.end(); i != end; ++i)
71 {
72 i->create(srcSize, CV_MAKE_TYPE(depth, 1));
73 declare.in(*i, WARMUP_RNG);
74 }
75 declare.out(dst);
76
77 if (RUN_OCL_IMPL)
78 {
79 ocl::oclMat oclDst(srcSize, dtype);
80 vector<ocl::oclMat> oclSrc(src.size());
81 for (vector<ocl::oclMat>::size_type i = 0, end = src.size(); i < end; ++i)
82 oclSrc[i] = src[i];
83
84 OCL_TEST_CYCLE() cv::ocl::merge(oclSrc, oclDst);
85
86 oclDst.download(dst);
87
88 SANITY_CHECK(dst);
89 }
90 else if (RUN_PLAIN_IMPL)
91 {
92 TEST_CYCLE() cv::merge(src, dst);
93
94 SANITY_CHECK(dst);
95 }
96 else
97 OCL_PERF_ELSE
98}

100///////////// Split////////////////////////
101
102typedef MergeParams SplitParams;
103typedef TestBaseWithParam<SplitParams> SplitFixture;
104
105OCL_PERF_TEST_P(SplitFixture, Split,
106 ::testing::Combine(OCL_TEST_SIZES, OCL_PERF_ENUM(CV_8U, CV_32F),
107 OCL_PERF_ENUM(2, 3)))
108{
109 const SplitParams params = GetParam();
110 const Size srcSize = get<0>(params);
111 const int depth = get<1>(params), cn = get<2>(params);
112 const int type = CV_MAKE_TYPE(depth, cn);
113
114 checkDeviceMaxMemoryAllocSize(srcSize, type);
115
116 Mat src(srcSize, type);
117 Mat dst0, dst1, dst2;
118 declare.in(src, WARMUP_RNG);
119
120 ASSERT_TRUE(cn == 3 || cn == 2);
121
122 if (RUN_OCL_IMPL)
123 {
124 ocl::oclMat oclSrc(src);
125 vector<ocl::oclMat> oclDst(cn);
126 oclDst[0] = ocl::oclMat(srcSize, depth);
127 oclDst[1] = ocl::oclMat(srcSize, depth);
128 if (cn == 3)
129 oclDst[2] = ocl::oclMat(srcSize, depth);
130
131 OCL_TEST_CYCLE() cv::ocl::split(oclSrc, oclDst);
132
133 oclDst[0].download(dst0);
134 oclDst[1].download(dst1);
135 if (cn == 3)
136 oclDst[2].download(dst2);
137 }
138 else if (RUN_PLAIN_IMPL)
139 {
140 vector<Mat> dst(cn);
141 dst[0] = Mat(srcSize, depth);
142 dst[1] = Mat(srcSize, depth);
143 if (cn == 3)
144 dst[2] = Mat(srcSize, depth);
145
146 TEST_CYCLE() cv::split(src, dst);
147
148 dst0 = dst[0];
149 dst1 = dst[1];
150 if (cn == 3)
151 dst2 = dst[2];
152 }
153 else
154 OCL_PERF_ELSE
155
156 if (cn == 2)
157 {
158 SANITY_CHECK(dst0);
159 SANITY_CHECK(dst1);
160 }
161 else if (cn == 3)
162 {
163 SANITY_CHECK(dst0);
164 SANITY_CHECK(dst1);
165 SANITY_CHECK(dst2);
166 }
167}
