56PERF_TEST(HaarFixture, Haar)
57{
58 vector<Rect> faces;
59
60 Mat img = imread(getDataPath("gpu/haarcascade/basketball1.png"), CV_LOAD_IMAGE_GRAYSCALE);
61 ASSERT_TRUE(!img.empty()) << "can't open basketball1.png";
62 declare.in(img);
63
64 if (RUN_PLAIN_IMPL)
65 {
66 CascadeClassifier faceCascade;
67 ASSERT_TRUE(faceCascade.load(getDataPath("gpu/haarcascade/haarcascade_frontalface_alt.xml")))
68 << "can't load haarcascade_frontalface_alt.xml";
69
70 TEST_CYCLE() faceCascade.detectMultiScale(img, faces,
71 1.1, 2, 0 | CV_HAAR_SCALE_IMAGE, Size(30, 30));
72
73 SANITY_CHECK(faces, 4 + 1e-4);
74 }
75 else if (RUN_OCL_IMPL)
76 {
77 ocl::OclCascadeClassifier faceCascade;
78 ocl::oclMat oclImg(img);
79
80 ASSERT_TRUE(faceCascade.load(getDataPath("gpu/haarcascade/haarcascade_frontalface_alt.xml")))
81 << "can't load haarcascade_frontalface_alt.xml";
82
83 OCL_TEST_CYCLE() faceCascade.detectMultiScale(oclImg, faces,
84 1.1, 2, 0 | CV_HAAR_SCALE_IMAGE, Size(30, 30));
85
86 SANITY_CHECK(faces, 4 + 1e-4);
87 }
88 else
89 OCL_PERF_ELSE
90}

92typedef std::tr1::tuple<std::string, std::string, int> Cascade_Image_MinSize_t;
93typedef perf::TestBaseWithParam<Cascade_Image_MinSize_t> Cascade_Image_MinSize;
94
95OCL_PERF_TEST_P(Cascade_Image_MinSize, CascadeClassifier,
96 testing::Combine(testing::Values( string("cv/cascadeandhog/cascades/haarcascade_frontalface_alt.xml"),
97 string("cv/cascadeandhog/cascades/haarcascade_frontalface_alt2.xml") ),
98 testing::Values( string("cv/shared/lena.png"),
99 string("cv/cascadeandhog/images/bttf301.png"),
100 string("cv/cascadeandhog/images/class57.png") ),
101 testing::Values(30, 64, 90)))
102{
103 const string cascasePath = get<0>(GetParam());
104 const string imagePath = get<1>(GetParam());
105 const int min_size = get<2>(GetParam());
106 Size minSize(min_size, min_size);
107 vector<Rect> faces;
108
109 Mat img = imread(getDataPath(imagePath), IMREAD_GRAYSCALE);
110 ASSERT_FALSE(img.empty()) << "Can't load source image: " << getDataPath(imagePath);
111 equalizeHist(img, img);
112 declare.in(img);
113
114 if (RUN_PLAIN_IMPL)
115 {
116 CascadeClassifier cc;
117 ASSERT_TRUE(cc.load(getDataPath(cascasePath))) << "Can't load cascade file: " << getDataPath(cascasePath);
118
119 while (next())
120 {
121 faces.clear();
122
123 startTimer();
124 cc.detectMultiScale(img, faces, 1.1, 3, CV_HAAR_SCALE_IMAGE, minSize);
125 stopTimer();
126 }
127 }
128 else if (RUN_OCL_IMPL)
129 {
130 ocl::oclMat uimg(img);
131 ocl::OclCascadeClassifier cc;
132 ASSERT_TRUE(cc.load(getDataPath(cascasePath))) << "Can't load cascade file: " << getDataPath(cascasePath);
133
134 while (next())
135 {
136 faces.clear();
137 ocl::finish();
138
139 startTimer();
140 cc.detectMultiScale(uimg, faces, 1.1, 3, CV_HAAR_SCALE_IMAGE, minSize);
141 stopTimer();
142 }
143 }
144 else
145 OCL_PERF_ELSE
146
147 //sort(faces.begin(), faces.end(), comparators::RectLess());
148 SANITY_CHECK_NOTHING();//(faces, min_size/5);
149 // using SANITY_CHECK_NOTHING() since OCL and PLAIN version may find different faces number
150}
