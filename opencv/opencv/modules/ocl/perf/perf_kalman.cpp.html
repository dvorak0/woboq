51using namespace perf;
52using namespace std;
53using namespace cv::ocl;
54using namespace cv;
55using std::tr1::tuple;
56using std::tr1::get;
57
58///////////// Kalman Filter ////////////////////////
59
60typedef tuple<int> KalmanFilterType;
61typedef TestBaseWithParam<KalmanFilterType> KalmanFilterFixture;
62
63PERF_TEST_P(KalmanFilterFixture, KalmanFilter,
64 ::testing::Values(1000, 1500))
65{
66 KalmanFilterType params = GetParam();
67 const int dim = get<0>(params);
68
69 cv::Mat sample(dim, 1, CV_32FC1), dresult;
70 randu(sample, -1, 1);
71
72 cv::Mat statePre_;
73
74 if (RUN_PLAIN_IMPL)
75 {
76 cv::KalmanFilter kalman;
77 TEST_CYCLE()
78 {
79 kalman.init(dim, dim);
80 kalman.correct(sample);
81 kalman.predict();
82 }
83 statePre_ = kalman.statePre;
84 }
85 else if(RUN_OCL_IMPL)
86 {
87 cv::ocl::oclMat dsample(sample);
88 cv::ocl::KalmanFilter kalman_ocl;
89 OCL_TEST_CYCLE()
90 {
91 kalman_ocl.init(dim, dim);
92 kalman_ocl.correct(dsample);
93 kalman_ocl.predict();
94 }
95 kalman_ocl.statePre.download(statePre_);
96 }
97 else
98 OCL_PERF_ELSE
99
100 SANITY_CHECK(statePre_);
101}
102
103#endif // HAVE_CLAMDBLAS
