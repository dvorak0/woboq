53///////////// MinMax ////////////////////////
54
55typedef Size_MatType MinMaxFixture;
56
57PERF_TEST_P(MinMaxFixture, MinMax,
58 ::testing::Combine(OCL_TYPICAL_MAT_SIZES,
59 OCL_PERF_ENUM(CV_8UC1, CV_32FC1)))
60{
61 const Size_MatType_t params = GetParam();
62 const Size srcSize = get<0>(params);
63 const int type = get<1>(params);
64
65 Mat src(srcSize, type);
66 declare.in(src, WARMUP_RNG);
67
68 double min_val = std::numeric_limits<double>::max(),
69 max_val = std::numeric_limits<double>::min();
70
71 if (RUN_OCL_IMPL)
72 {
73 ocl::oclMat oclSrc(src);
74
75 OCL_TEST_CYCLE() cv::ocl::minMax(oclSrc, &min_val, &max_val);
76
77 ASSERT_GE(max_val, min_val);
78 SANITY_CHECK(min_val);
79 SANITY_CHECK(max_val);
80 }
81 else if (RUN_PLAIN_IMPL)
82 {
83 Point min_loc, max_loc;
84
85 TEST_CYCLE() cv::minMaxLoc(src, &min_val, &max_val, &min_loc, &max_loc);
86
87 ASSERT_GE(max_val, min_val);
88 SANITY_CHECK(min_val);
89 SANITY_CHECK(max_val);
90 }
91 else
92 OCL_PERF_ELSE
93}

239///////////// norm////////////////////////
240
241CV_ENUM(NormType, NORM_INF, NORM_L1, NORM_L2)
242
243typedef std::tr1::tuple<Size, MatType, NormType> NormParams;
244typedef TestBaseWithParam<NormParams> NormFixture;
245
246OCL_PERF_TEST_P(NormFixture, Norm,
247 ::testing::Combine(OCL_PERF_ENUM(OCL_SIZE_1, OCL_SIZE_2, OCL_SIZE_3),
248 OCL_TEST_TYPES, NormType::all()))
249{
250 const NormParams params = GetParam();
251 const Size srcSize = get<0>(params);
252 const int type = get<1>(params);
253 const int normType = get<2>(params);
254 perf::ERROR_TYPE errorType = type != NORM_INF ? ERROR_RELATIVE : ERROR_ABSOLUTE;
255 double eps = 1e-5, value;
256
257 Mat src1(srcSize, type), src2(srcSize, type);
258 declare.in(src1, src2, WARMUP_RNG);
259
260 if (RUN_OCL_IMPL)
261 {
262 ocl::oclMat oclSrc1(src1), oclSrc2(src2);
263
264 OCL_TEST_CYCLE() value = cv::ocl::norm(oclSrc1, oclSrc2, normType);
265
266 SANITY_CHECK(value, eps, errorType);
267 }
268 else if (RUN_PLAIN_IMPL)
269 {
270 TEST_CYCLE() value = cv::norm(src1, src2, normType);
271
272 SANITY_CHECK(value, eps, errorType);
273 }
274 else
275 OCL_PERF_ELSE
276}
