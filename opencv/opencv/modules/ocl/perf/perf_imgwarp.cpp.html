59OCL_PERF_TEST_P(WarpAffineFixture, WarpAffine,
60 ::testing::Combine(OCL_TEST_SIZES, OCL_TEST_TYPES_134, InterType::all()))
61{
62 static const double coeffs[2][3] =
63 {
64 { cos(CV_PI / 6), -sin(CV_PI / 6), 100.0 },
65 { sin(CV_PI / 6), cos(CV_PI / 6), -100.0 }
66 };
67 Mat M(2, 3, CV_64F, (void *)coeffs);
68
69 const WarpAffineParams params = GetParam();
70 const Size srcSize = get<0>(params);
71 const int type = get<1>(params), interpolation = get<2>(params);
72
73 checkDeviceMaxMemoryAllocSize(srcSize, type);
74
75 Mat src(srcSize, type), dst(srcSize, type);
76 declare.in(src, WARMUP_RNG).out(dst);
77
78 if (RUN_OCL_IMPL)
79 {
80 ocl::oclMat oclSrc(src), oclDst(srcSize, type);
81
82 OCL_TEST_CYCLE() cv::ocl::warpAffine(oclSrc, oclDst, M, srcSize, interpolation);
83
84 oclDst.download(dst);
85
86 SANITY_CHECK(dst, 5e-4);
87 }

89 {
90 TEST_CYCLE() cv::warpAffine(src, dst, M, srcSize, interpolation);
91
92 SANITY_CHECK(dst, 5e-4);
93 }
94 else
95 OCL_PERF_ELSE
96}

223///////////// Remap ////////////////////////
224
225typedef tuple<Size, MatType, InterType> RemapParams;
226typedef TestBaseWithParam<RemapParams> RemapFixture;
227
228OCL_PERF_TEST_P(RemapFixture, Remap,
229 ::testing::Combine(OCL_TEST_SIZES, OCL_TEST_TYPES, InterType::all()))
230{
231 const RemapParams params = GetParam();
232 const Size srcSize = get<0>(params);
233 const int type = get<1>(params), interpolation = get<2>(params);
234
235 Mat src(srcSize, type), dst(srcSize, type);
236 declare.in(src, WARMUP_RNG).out(dst);
237
238 Mat xmap, ymap;
239 xmap.create(srcSize, CV_32FC1);
240 ymap.create(srcSize, CV_32FC1);
241
242 for (int i = 0; i < srcSize.height; ++i)
243 {
244 float * const xmap_row = xmap.ptr<float>(i);
245 float * const ymap_row = ymap.ptr<float>(i);
246
247 for (int j = 0; j < srcSize.width; ++j)
248 {
249 xmap_row[j] = (j - srcSize.width * 0.5f) * 0.75f + srcSize.width * 0.5f;
250 ymap_row[j] = (i - srcSize.height * 0.5f) * 0.75f + srcSize.height * 0.5f;
251 }
252 }
253
254 const int borderMode = BORDER_CONSTANT;
255
256 if (RUN_OCL_IMPL)
257 {
258 ocl::oclMat oclSrc(src), oclDst(srcSize, type);
259 ocl::oclMat oclXMap(xmap), oclYMap(ymap);
260
261 OCL_TEST_CYCLE() cv::ocl::remap(oclSrc, oclDst, oclXMap, oclYMap, interpolation, borderMode);
262
263 oclDst.download(dst);
264
265 SANITY_CHECK(dst, 1 + DBL_EPSILON);
266 }
267 else if (RUN_PLAIN_IMPL)
268 {
269 TEST_CYCLE() cv::remap(src, dst, xmap, ymap, interpolation, borderMode);
270
271 SANITY_CHECK(dst, 1 + DBL_EPSILON);
272 }
273 else
274 OCL_PERF_ELSE
275}

278///////////// BuildWarpPerspectiveMaps ////////////////////////
279
280static void buildWarpPerspectiveMaps(const Mat &M, bool inverse, Size dsize, Mat &xmap, Mat &ymap)
281{
282 CV_Assert(M.rows == 3 && M.cols == 3);
283 CV_Assert(dsize.area() > 0);
284
285 xmap.create(dsize, CV_32FC1);
286 ymap.create(dsize, CV_32FC1);
287
288 float coeffs[3 * 3];
289 Mat coeffsMat(3, 3, CV_32F, (void *)coeffs);
290
291 if (inverse)
292 M.convertTo(coeffsMat, coeffsMat.type());
293 else
294 {
295 cv::Mat iM;
296 invert(M, iM);
297 iM.convertTo(coeffsMat, coeffsMat.type());
298 }
299
300 for (int y = 0; y < dsize.height; ++y)
301 {
302 float * const xmap_ptr = xmap.ptr<float>(y);
303 float * const ymap_ptr = ymap.ptr<float>(y);
304
305 for (int x = 0; x < dsize.width; ++x)
306 {
307 float coeff = 1.0f / (x * coeffs[6] + y * coeffs[7] + coeffs[8]);
308 xmap_ptr[x] = (x * coeffs[0] + y * coeffs[1] + coeffs[2]) * coeff;
309 ymap_ptr[x] = (x * coeffs[3] + y * coeffs[4] + coeffs[5]) * coeff;
310 }
311 }
312}

314typedef TestBaseWithParam<Size> BuildWarpPerspectiveMapsFixture;
315
316PERF_TEST_P(BuildWarpPerspectiveMapsFixture, Inverse, OCL_TYPICAL_MAT_SIZES)
317{
318 static const double coeffs[3][3] =
319 {
320 {cos(CV_PI / 6), -sin(CV_PI / 6), 100.0},
321 {sin(CV_PI / 6), cos(CV_PI / 6), -100.0},
322 {0.0, 0.0, 1.0}
323 };
324 Mat M(3, 3, CV_64F, (void *)coeffs);
325 const Size dsize = GetParam();
326 const double eps = 5e-4;
327
328 Mat xmap(dsize, CV_32FC1), ymap(dsize, CV_32FC1);
329 declare.in(M).out(xmap, ymap);
330
331 if (RUN_OCL_IMPL)
332 {
333 ocl::oclMat oclXMap(dsize, CV_32FC1), oclYMap(dsize, CV_32FC1);
334
335 OCL_TEST_CYCLE() cv::ocl::buildWarpPerspectiveMaps(M, true, dsize, oclXMap, oclYMap);
336
337 oclXMap.download(xmap);
338 oclYMap.download(ymap);
339
340 SANITY_CHECK(xmap, eps);
341 SANITY_CHECK(ymap, eps);
342 }
343 else if (RUN_PLAIN_IMPL)
344 {
345 TEST_CYCLE() buildWarpPerspectiveMaps(M, true, dsize, xmap, ymap);
346
347 SANITY_CHECK(xmap, eps);
348 SANITY_CHECK(ymap, eps);
349 }
350 else
351 OCL_PERF_ELSE
352}
