311typedef struct _COOR
312{
313 short x;
314 short y;
315} COOR;

317static COOR do_meanShift(int x0, int y0, uchar *sptr, uchar *dptr, int sstep, cv::Size size, int sp, int sr, int maxIter, float eps, int *tab)
318{
319
320 int isr2 = sr * sr;
321 int c0, c1, c2, c3;
322 int iter;
323 uchar *ptr = NULL;
324 uchar *pstart = NULL;
325 int revx = 0, revy = 0;
326 c0 = sptr[0];
327 c1 = sptr[1];
328 c2 = sptr[2];
329 c3 = sptr[3];
330 // iterate meanshift procedure
331 for(iter = 0; iter < maxIter; iter++ )
332 {
333 int count = 0;
334 int s0 = 0, s1 = 0, s2 = 0, sx = 0, sy = 0;
335
336 //mean shift: process pixels in window (p-sigmaSp)x(p+sigmaSp)
337 int minx = x0 - sp;
338 int miny = y0 - sp;
339 int maxx = x0 + sp;
340 int maxy = y0 + sp;
341
342 //deal with the image boundary
343 if(minx < 0) minx = 0;
344 if(miny < 0) miny = 0;
345 if(maxx >= size.width) maxx = size.width - 1;
346 if(maxy >= size.height) maxy = size.height - 1;
347 if(iter == 0)
348 {
349 pstart = sptr;
350 }
351 else
352 {
353 pstart = pstart + revy * sstep + (revx << 2); //point to the new position
354 }
355 ptr = pstart;
356 ptr = ptr + (miny - y0) * sstep + ((minx - x0) << 2); //point to the start in the row
357
358 for( int y = miny; y <= maxy; y++, ptr += sstep - ((maxx - minx + 1) << 2))
359 {
360 int rowCount = 0;
361 int x = minx;
362#if CV_ENABLE_UNROLLED
363 for( ; x + 4 <= maxx; x += 4, ptr += 16)
364 {
365 int t0, t1, t2;
366 t0 = ptr[0], t1 = ptr[1], t2 = ptr[2];
367 if(tab[t0 - c0 + 255] + tab[t1 - c1 + 255] + tab[t2 - c2 + 255] <= isr2)
368 {
369 s0 += t0;
370 s1 += t1;
371 s2 += t2;
372 sx += x;
373 rowCount++;
374 }
375 t0 = ptr[4], t1 = ptr[5], t2 = ptr[6];
376 if(tab[t0 - c0 + 255] + tab[t1 - c1 + 255] + tab[t2 - c2 + 255] <= isr2)
377 {
378 s0 += t0;
379 s1 += t1;
380 s2 += t2;
381 sx += x + 1;
382 rowCount++;
383 }
384 t0 = ptr[8], t1 = ptr[9], t2 = ptr[10];
385 if(tab[t0 - c0 + 255] + tab[t1 - c1 + 255] + tab[t2 - c2 + 255] <= isr2)
386 {
387 s0 += t0;
388 s1 += t1;
389 s2 += t2;
390 sx += x + 2;
391 rowCount++;
392 }
393 t0 = ptr[12], t1 = ptr[13], t2 = ptr[14];
394 if(tab[t0 - c0 + 255] + tab[t1 - c1 + 255] + tab[t2 - c2 + 255] <= isr2)
395 {
396 s0 += t0;
397 s1 += t1;
398 s2 += t2;
399 sx += x + 3;
400 rowCount++;
401 }
402 }
403#endif
404 for(; x <= maxx; x++, ptr += 4)
405 {
406 int t0 = ptr[0], t1 = ptr[1], t2 = ptr[2];
407 if(tab[t0 - c0 + 255] + tab[t1 - c1 + 255] + tab[t2 - c2 + 255] <= isr2)
408 {
409 s0 += t0;
410 s1 += t1;
411 s2 += t2;
412 sx += x;
413 rowCount++;
414 }
415 }
416 if(rowCount == 0)
417 continue;
418 count += rowCount;
419 sy += y * rowCount;
420 }
421
422 if( count == 0 )
423 break;
424
425 int x1 = sx / count;
426 int y1 = sy / count;
427 s0 = s0 / count;
428 s1 = s1 / count;
429 s2 = s2 / count;
430
431 bool stopFlag = (x0 == x1 && y0 == y1) || (abs(x1 - x0) + abs(y1 - y0) +
432 tab[s0 - c0 + 255] + tab[s1 - c1 + 255] + tab[s2 - c2 + 255] <= eps);
433
434 //revise the pointer corresponding to the new (y0,x0)
435 revx = x1 - x0;
436 revy = y1 - y0;
437
438 x0 = x1;
439 y0 = y1;
440 c0 = s0;
441 c1 = s1;
442 c2 = s2;
443
444 if( stopFlag )
445 break;
446 } //for iter
447
448 dptr[0] = (uchar)c0;
449 dptr[1] = (uchar)c1;
450 dptr[2] = (uchar)c2;
451 dptr[3] = (uchar)c3;
452
453 COOR coor;
454 coor.x = static_cast<short>(x0);
455 coor.y = static_cast<short>(y0);
456 return coor;
457}

459static void meanShiftFiltering_(const Mat &src_roi, Mat &dst_roi, int sp, int sr, cv::TermCriteria crit)
460{
461 if( src_roi.empty() )
462 CV_Error( CV_StsBadArg, "The input image is empty" );
463
464 if( src_roi.depth() != CV_8U || src_roi.channels() != 4 )
465 CV_Error( CV_StsUnsupportedFormat, "Only 8-bit, 4-channel images are supported" );
466
467 dst_roi.create(src_roi.size(), src_roi.type());
468
469 CV_Assert( (src_roi.cols == dst_roi.cols) && (src_roi.rows == dst_roi.rows) );
470 CV_Assert( !(dst_roi.step & 0x3) );
471
472 if( !(crit.type & cv::TermCriteria::MAX_ITER) )
473 crit.maxCount = 5;
474 int maxIter = std::min(std::max(crit.maxCount, 1), 100);
475 float eps;
476 if( !(crit.type & cv::TermCriteria::EPS) )
477 eps = 1.f;
478 eps = (float)std::max(crit.epsilon, 0.0);
479
480 int tab[512];
481 for(int i = 0; i < 512; i++)
482 tab[i] = (i - 255) * (i - 255);
483 uchar *sptr = src_roi.data;
484 uchar *dptr = dst_roi.data;
485 int sstep = (int)src_roi.step;
486 int dstep = (int)dst_roi.step;
487 cv::Size size = src_roi.size();
488
489 for(int i = 0; i < size.height; i++, sptr += sstep - (size.width << 2),
490 dptr += dstep - (size.width << 2))
491 {
492 for(int j = 0; j < size.width; j++, sptr += 4, dptr += 4)
493 {
494 do_meanShift(j, i, sptr, dptr, sstep, size, sp, sr, maxIter, eps, tab);
495 }
496 }
497}

531static void meanShiftProc_(const Mat &src_roi, Mat &dst_roi, Mat &dstCoor_roi, int sp, int sr, cv::TermCriteria crit)
532{
533 if (src_roi.empty())
534 {
535 CV_Error(CV_StsBadArg, "The input image is empty");
536 }
537 if (src_roi.depth() != CV_8U || src_roi.channels() != 4)
538 {
539 CV_Error(CV_StsUnsupportedFormat, "Only 8-bit, 4-channel images are supported");
540 }
541
542 dst_roi.create(src_roi.size(), src_roi.type());
543 dstCoor_roi.create(src_roi.size(), CV_16SC2);
544
545 CV_Assert((src_roi.cols == dst_roi.cols) && (src_roi.rows == dst_roi.rows) &&
546 (src_roi.cols == dstCoor_roi.cols) && (src_roi.rows == dstCoor_roi.rows));
547 CV_Assert(!(dstCoor_roi.step & 0x3));
548
549 if (!(crit.type & cv::TermCriteria::MAX_ITER))
550 {
551 crit.maxCount = 5;
552 }
553
554 int maxIter = std::min(std::max(crit.maxCount, 1), 100);
555 float eps;
556
557 if (!(crit.type & cv::TermCriteria::EPS))
558 {
559 eps = 1.f;
560 }
561
562 eps = (float)std::max(crit.epsilon, 0.0);
563
564 int tab[512];
565
566 for (int i = 0; i < 512; i++)
567 {
568 tab[i] = (i - 255) * (i - 255);
569 }
570
571 uchar *sptr = src_roi.data;
572 uchar *dptr = dst_roi.data;
573 short *dCoorptr = (short *)dstCoor_roi.data;
574 int sstep = (int)src_roi.step;
575 int dstep = (int)dst_roi.step;
576 int dCoorstep = (int)dstCoor_roi.step >> 1;
577 cv::Size size = src_roi.size();
578
579 for (int i = 0; i < size.height; i++, sptr += sstep - (size.width << 2),
580 dptr += dstep - (size.width << 2), dCoorptr += dCoorstep - (size.width << 1))
581 {
582 for (int j = 0; j < size.width; j++, sptr += 4, dptr += 4, dCoorptr += 2)
583 {
584 *((COOR *)dCoorptr) = do_meanShift(j, i, sptr, dptr, sstep, size, sp, sr, maxIter, eps, tab);
585 }
586 }
587
588}

664static void columnSumPerfTest(const Mat & src, Mat & dst)
665{
666 for (int j = 0; j < src.cols; j++)
667 dst.at<float>(0, j) = src.at<float>(0, j);
668
669 for (int i = 1; i < src.rows; ++i)
670 for (int j = 0; j < src.cols; ++j)
671 dst.at<float>(i, j) = dst.at<float>(i - 1 , j) + src.at<float>(i , j);
