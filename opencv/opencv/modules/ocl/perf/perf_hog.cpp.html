54struct RectLess :
55 public std::binary_function<cv::Rect, cv::Rect, bool>
56{
57 bool operator()(const cv::Rect& a,
58 const cv::Rect& b) const
59 {
60 if (a.x != b.x)
61 return a.x < b.x;
62 else if (a.y != b.y)
63 return a.y < b.y;
64 else if (a.width != b.width)
65 return a.width < b.width;
66 else
67 return a.height < b.height;
68 }
69};

71OCL_PERF_TEST(HOGFixture, HOG)
72{
73 Mat src = imread(getDataPath("gpu/hog/road.png"), cv::IMREAD_GRAYSCALE);
74 ASSERT_TRUE(!src.empty()) << "can't open input image road.png";
75
76 vector<cv::Rect> found_locations;
77 declare.in(src);
78
79 if (RUN_PLAIN_IMPL)
80 {
81 HOGDescriptor hog;
82 hog.setSVMDetector(hog.getDefaultPeopleDetector());
83
84 TEST_CYCLE() hog.detectMultiScale(src, found_locations);
85
86 std::sort(found_locations.begin(), found_locations.end(), RectLess());
87 SANITY_CHECK(found_locations, 1 + DBL_EPSILON);
88 }
89 else if (RUN_OCL_IMPL)
90 {
91 ocl::HOGDescriptor ocl_hog;
92 ocl_hog.setSVMDetector(ocl_hog.getDefaultPeopleDetector());
93 ocl::oclMat oclSrc(src);
94
95 OCL_TEST_CYCLE() ocl_hog.detectMultiScale(oclSrc, found_locations);
96
97 std::sort(found_locations.begin(), found_locations.end(), RectLess());
98 SANITY_CHECK(found_locations, 1 + DBL_EPSILON);
99 }
100 else
101 OCL_PERF_ELSE
102}
