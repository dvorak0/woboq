55typedef Size_MatType CV_TM_CCORRFixture;
56
57OCL_PERF_TEST_P(CV_TM_CCORRFixture, matchTemplate,
58 ::testing::Combine(::testing::Values(Size(1000, 1000), Size(2000, 2000)),
59 OCL_PERF_ENUM(CV_32FC1, CV_32FC4)))
60{
61 const Size_MatType_t params = GetParam();
62 const Size srcSize = get<0>(params), templSize(5, 5);
63 const int type = get<1>(params);
64
65 Mat src(srcSize, type), templ(templSize, type);
66 const Size dstSize(src.cols - templ.cols + 1, src.rows - templ.rows + 1);
67 Mat dst(dstSize, CV_32F);
68 randu(src, 0.0f, 1.0f);
69 randu(templ, 0.0f, 1.0f);
70 declare.in(src, templ).out(dst);
71
72 if (RUN_OCL_IMPL)
73 {
74 ocl::oclMat oclSrc(src), oclTempl(templ), oclDst(dstSize, CV_32F);
75
76 OCL_TEST_CYCLE() cv::ocl::matchTemplate(oclSrc, oclTempl, oclDst, CV_TM_CCORR);
77
78 oclDst.download(dst);
79
80 SANITY_CHECK(dst, 1e-4);
81 }
82 else if (RUN_PLAIN_IMPL)
83 {
84 TEST_CYCLE() cv::matchTemplate(src, templ, dst, CV_TM_CCORR);
85
86 SANITY_CHECK(dst, 1e-4);
87 }
88 else
89 OCL_PERF_ELSE
90}

94OCL_PERF_TEST_P(CV_TM_CCORR_NORMEDFixture, matchTemplate,
95 ::testing::Values(Size(1000, 1000), Size(2000, 2000), Size(4000, 4000)))
96{
97 const Size srcSize = GetParam(), templSize(5, 5);

100 const Size dstSize(src.cols - templ.cols + 1, src.rows - templ.rows + 1);

105 {

121 OCL_PERF_ELSE
122}

124CV_ENUM(MethodType, TM_SQDIFF, TM_SQDIFF_NORMED, TM_CCORR, TM_CCORR_NORMED, TM_CCOEFF, TM_CCOEFF_NORMED)
125
126typedef std::tr1::tuple<Size, Size, MethodType> ImgSize_TmplSize_Method_t;
127typedef TestBaseWithParam<ImgSize_TmplSize_Method_t> ImgSize_TmplSize_Method;
128
129OCL_PERF_TEST_P(ImgSize_TmplSize_Method, MatchTemplate,
130 ::testing::Combine(
131 testing::Values(szSmall128, cv::Size(320, 240),
132 cv::Size(640, 480), cv::Size(800, 600),
133 cv::Size(1024, 768), cv::Size(1280, 1024)),
134 testing::Values(cv::Size(12, 12), cv::Size(28, 9),
135 cv::Size(8, 30), cv::Size(16, 16)),
136 MethodType::all()
137 )
138 )
139{
140 Size imgSz = get<0>(GetParam());
141 Size tmplSz = get<1>(GetParam());
142 int method = get<2>(GetParam());
143
144 Mat img(imgSz, CV_8UC1);
145 Mat tmpl(tmplSz, CV_8UC1);
146 Mat result(imgSz - tmplSz + Size(1, 1), CV_32F);
147
148 bool isNormed =
149 method == TM_CCORR_NORMED ||
150 method == TM_SQDIFF_NORMED ||
151 method == TM_CCOEFF_NORMED;
152
153 declare.in(img, tmpl, WARMUP_RNG).out(result).time(30);
154
155 if (RUN_OCL_IMPL)
156 {
157 ocl::oclMat oclImg(img), oclTmpl(tmpl), oclResult(result.size(), CV_32F);
158
159 OCL_TEST_CYCLE() matchTemplate(oclImg, oclTmpl, oclResult, method);
160
161 oclResult.download(result);
162 }
163 else if (RUN_PLAIN_IMPL)
164 {
165 OCL_TEST_CYCLE() matchTemplate(img, tmpl, result, method);
166 }
167 else
168 OCL_PERF_ELSE
169
170 double eps = isNormed ? 3e-2
171 : 255 * 255 * tmpl.total() * 1e-4;
172
173 if (method == TM_SQDIFF_NORMED)
174 SANITY_CHECK_NOTHING();
175 else if (isNormed)
176 SANITY_CHECK(result, eps, ERROR_RELATIVE);
177 else
178 SANITY_CHECK(result, eps);
179}
