55typedef TestBaseWithParam<tuple<int> > PyrLKOpticalFlowFixture;
56
57OCL_PERF_TEST_P(PyrLKOpticalFlowFixture,
58 PyrLKOpticalFlow, ::testing::Values(1000, 2000, 4000))
59{
60 const int pointsCount = get<0>(GetParam());
61
62 const string fileName0 = "gpu/opticalflow/rubberwhale1.png",
63 fileName1 = "gpu/opticalflow/rubberwhale2.png";
64 Mat frame0 = imread(getDataPath(fileName0), cv::IMREAD_GRAYSCALE);
65 Mat frame1 = imread(getDataPath(fileName1), cv::IMREAD_GRAYSCALE);
66
67 declare.in(frame0, frame1);
68
69 ASSERT_FALSE(frame0.empty()) << "can't load " << fileName0;
70 ASSERT_FALSE(frame1.empty()) << "can't load " << fileName1;
71
72 vector<Point2f> pts, nextPts;
73 vector<unsigned char> status;
74 vector<float> err;
75 goodFeaturesToTrack(frame0, pts, pointsCount, 0.01, 0.0);
76 Mat ptsMat(1, static_cast<int>(pts.size()), CV_32FC2, (void *)&pts[0]);
77
78 if (RUN_PLAIN_IMPL)
79 {
80 TEST_CYCLE()
81 cv::calcOpticalFlowPyrLK(frame0, frame1, pts, nextPts, status, err);
82 }
83 else if (RUN_OCL_IMPL)
84 {
85 ocl::PyrLKOpticalFlow oclPyrLK;
86 ocl::oclMat oclFrame0(frame0), oclFrame1(frame1);
87 ocl::oclMat oclPts(ptsMat);
88 ocl::oclMat oclNextPts, oclStatus, oclErr;
89
90 OCL_TEST_CYCLE()
91 oclPyrLK.sparse(oclFrame0, oclFrame1, oclPts, oclNextPts, oclStatus, &oclErr);
92 }
93 else
94 OCL_PERF_ELSE
95
96 int value = 0;
97 SANITY_CHECK(value);
98}

100PERF_TEST(tvl1flowFixture, tvl1flow)
101{
102 Mat frame0 = imread(getDataPath("gpu/opticalflow/rubberwhale1.png"), cv::IMREAD_GRAYSCALE);
103 ASSERT_FALSE(frame0.empty()) << "can't load rubberwhale1.png";
104
105 Mat frame1 = imread(getDataPath("gpu/opticalflow/rubberwhale2.png"), cv::IMREAD_GRAYSCALE);
106 ASSERT_FALSE(frame1.empty()) << "can't load rubberwhale2.png";
107
108 const Size srcSize = frame0.size();
109 const double eps = 1.2;
110 Mat flow(srcSize, CV_32FC2), flow1(srcSize, CV_32FC1), flow2(srcSize, CV_32FC1);
111 declare.in(frame0, frame1).out(flow1, flow2);
112
113 if (RUN_PLAIN_IMPL)
114 {
115 Ptr<DenseOpticalFlow> alg = createOptFlow_DualTVL1();
116
117 TEST_CYCLE() alg->calc(frame0, frame1, flow);
118
119 alg->collectGarbage();
120 Mat flows[2] = { flow1, flow2 };
121 split(flow, flows);
122
123 SANITY_CHECK(flow1, eps);
124 SANITY_CHECK(flow2, eps);
125 }
126 else if (RUN_OCL_IMPL)
127 {
128 ocl::OpticalFlowDual_TVL1_OCL oclAlg;
129 ocl::oclMat oclFrame0(frame0), oclFrame1(frame1), oclFlow1(srcSize, CV_32FC1),
130 oclFlow2(srcSize, CV_32FC1);
131
132 OCL_TEST_CYCLE() oclAlg(oclFrame0, oclFrame1, oclFlow1, oclFlow2);
133
134 oclAlg.collectGarbage();
135
136 oclFlow1.download(flow1);
137 oclFlow2.download(flow2);
138
139 SANITY_CHECK(flow1, eps);
140 SANITY_CHECK(flow2, eps);
141 }
142 else
143 OCL_PERF_ELSE
144}

146///////////// FarnebackOpticalFlow ////////////////////////
147
148CV_ENUM(farneFlagType, 0, OPTFLOW_FARNEBACK_GAUSSIAN)
149
150typedef tuple<tuple<int, double>, farneFlagType, bool> FarnebackOpticalFlowParams;
151typedef TestBaseWithParam<FarnebackOpticalFlowParams> FarnebackOpticalFlowFixture;
152
153OCL_PERF_TEST_P(FarnebackOpticalFlowFixture, FarnebackOpticalFlow,
154 ::testing::Combine(
155 ::testing::Values(make_tuple<int, double>(5, 1.1),
156 make_tuple<int, double>(7, 1.5)),
157 farneFlagType::all(), ::testing::Bool()))
158{
159 Mat frame0 = imread(getDataPath("gpu/opticalflow/rubberwhale1.png"), cv::IMREAD_GRAYSCALE);
160 ASSERT_FALSE(frame0.empty()) << "can't load rubberwhale1.png";
161
162 Mat frame1 = imread(getDataPath("gpu/opticalflow/rubberwhale2.png"), cv::IMREAD_GRAYSCALE);
163 ASSERT_FALSE(frame1.empty()) << "can't load rubberwhale2.png";
164
165 const Size srcSize = frame0.size();
166
167 const FarnebackOpticalFlowParams params = GetParam();
168 const tuple<int, double> polyParams = get<0>(params);
169 const int polyN = get<0>(polyParams), flags = get<1>(params);
170 const double polySigma = get<1>(polyParams), pyrScale = 0.5;
171 const bool useInitFlow = get<2>(params);
172 const double eps = 1.5;
173
174 Mat flowx(srcSize, CV_32FC1), flowy(srcSize, CV_32FC1), flow(srcSize, CV_32FC2);
175 declare.in(frame0, frame1).out(flowx, flowy);
176
177 ocl::FarnebackOpticalFlow farn;
178 farn.pyrScale = pyrScale;
179 farn.polyN = polyN;
180 farn.polySigma = polySigma;
181 farn.flags = flags;
182
183 if (RUN_PLAIN_IMPL)
184 {
185 if (useInitFlow)
186 {
187 calcOpticalFlowFarneback(
188 frame0, frame1, flow, farn.pyrScale, farn.numLevels, farn.winSize,
189 farn.numIters, farn.polyN, farn.polySigma, farn.flags);
190 farn.flags |= OPTFLOW_USE_INITIAL_FLOW;
191 }
192
193 TEST_CYCLE()
194 calcOpticalFlowFarneback(
195 frame0, frame1, flow, farn.pyrScale, farn.numLevels, farn.winSize,
196 farn.numIters, farn.polyN, farn.polySigma, farn.flags);
197
198 Mat flowxy[2] = { flowx, flowy };
199 split(flow, flowxy);
200
201 SANITY_CHECK(flowx, eps);
202 SANITY_CHECK(flowy, eps);
203 }
204 else if (RUN_OCL_IMPL)
205 {
206 ocl::oclMat oclFrame0(frame0), oclFrame1(frame1),
207 oclFlowx(srcSize, CV_32FC1), oclFlowy(srcSize, CV_32FC1);
208
209 if (useInitFlow)
210 {
211 farn(oclFrame0, oclFrame1, oclFlowx, oclFlowy);
212 farn.flags |= OPTFLOW_USE_INITIAL_FLOW;
213 }
214
215 OCL_TEST_CYCLE()
216 farn(oclFrame0, oclFrame1, oclFlowx, oclFlowy);
217
218 oclFlowx.download(flowx);
219 oclFlowy.download(flowy);
220
221 SANITY_CHECK(flowx, eps);
222 SANITY_CHECK(flowy, eps);
223 }
224 else
225 OCL_PERF_ELSE
226}
