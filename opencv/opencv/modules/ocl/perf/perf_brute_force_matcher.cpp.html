53typedef Size_MatType BruteForceMatcherFixture;
54
55OCL_PERF_TEST_P(BruteForceMatcherFixture, Match,
56 ::testing::Combine(OCL_PERF_ENUM(OCL_SIZE_1, OCL_SIZE_2, OCL_SIZE_3),
57 OCL_PERF_ENUM(MatType(CV_32FC1))))
58{
59 const Size_MatType_t params = GetParam();
60 const Size srcSize = get<0>(params);
61 const int type = get<1>(params);
62
63 vector<DMatch> matches;
64 Mat query(srcSize, type), train(srcSize, type);
65 declare.in(query, train);
66 randu(query, 0.0f, 1.0f);
67 randu(train, 0.0f, 1.0f);
68
69 if (RUN_PLAIN_IMPL)
70 {
71 BFMatcher matcher(NORM_L2);
72 TEST_CYCLE() matcher.match(query, train, matches);
73
74 SANITY_CHECK_MATCHES(matches);
75 }
76 else if (RUN_OCL_IMPL)
77 {
78 ocl::BruteForceMatcher_OCL_base oclMatcher(ocl::BruteForceMatcher_OCL_base::L2Dist);
79 ocl::oclMat oclQuery(query), oclTrain(train);
80
81 OCL_TEST_CYCLE()
82 oclMatcher.match(oclQuery, oclTrain, matches);
83
84 SANITY_CHECK_MATCHES(matches, 1e-5);
85 }
86 else
87 OCL_PERF_ELSE
88}

90OCL_PERF_TEST_P(BruteForceMatcherFixture, KnnMatch,
91 ::testing::Combine(OCL_PERF_ENUM(OCL_SIZE_1, OCL_SIZE_2, OCL_SIZE_3),
92 OCL_PERF_ENUM(MatType(CV_32FC1))))
93{
94 const Size_MatType_t params = GetParam();
95 const Size srcSize = get<0>(params);
96 const int type = get<1>(params);
97
98 vector<vector<DMatch> > matches(2);
99 Mat query(srcSize, type), train(srcSize, type);
100 randu(query, 0.0f, 1.0f);
101 randu(train, 0.0f, 1.0f);
102
103 declare.in(query, train);
104
105 if (RUN_PLAIN_IMPL)
106 {
107 BFMatcher matcher(NORM_L2);
108 TEST_CYCLE() matcher.knnMatch(query, train, matches, 2);
109
110 std::vector<DMatch> & matches0 = matches[0], & matches1 = matches[1];
111 SANITY_CHECK_MATCHES(matches0);
112 SANITY_CHECK_MATCHES(matches1);
113 }
114 else if (RUN_OCL_IMPL)
115 {
116 ocl::BruteForceMatcher_OCL_base oclMatcher(ocl::BruteForceMatcher_OCL_base::L2Dist);
117 ocl::oclMat oclQuery(query), oclTrain(train);
118 ocl::oclMat oclTrainIdx, oclDistance;
119
120 OCL_TEST_CYCLE()
121 oclMatcher.knnMatch(oclQuery, oclTrain, matches, 2);
122
123 oclMatcher.knnMatchDownload(oclTrainIdx, oclDistance, matches);
124
125 std::vector<DMatch> & matches0 = matches[0], & matches1 = matches[1];
126 SANITY_CHECK_MATCHES(matches0, 1e-5);
127 SANITY_CHECK_MATCHES(matches1, 1e-5);
128 }
129 else
130 OCL_PERF_ELSE
131}

133OCL_PERF_TEST_P(BruteForceMatcherFixture, RadiusMatch,
134 ::testing::Combine(OCL_PERF_ENUM(OCL_SIZE_1, OCL_SIZE_2, OCL_SIZE_3),
135 OCL_PERF_ENUM(MatType(CV_32FC1))))
136{
137 const Size_MatType_t params = GetParam();
138 const Size srcSize = get<0>(params);
139 const int type = get<1>(params);
140
141 const float max_distance = 2.0f;
142 vector<vector<DMatch> > matches(2);
143 Mat query(srcSize, type), train(srcSize, type);
144 declare.in(query, train);
145
146 randu(query, 0.0f, 1.0f);
147 randu(train, 0.0f, 1.0f);
148
149 if (RUN_PLAIN_IMPL)
150 {
151 cv::BFMatcher matcher(NORM_L2);
152 TEST_CYCLE() matcher.radiusMatch(query, train, matches, max_distance);
153
154 std::vector<DMatch> & matches0 = matches[0], & matches1 = matches[1];
155 SANITY_CHECK_MATCHES(matches0);
156 SANITY_CHECK_MATCHES(matches1);
157 }
158 else if (RUN_OCL_IMPL)
159 {
160 ocl::oclMat oclQuery(query), oclTrain(train);
161 ocl::BruteForceMatcher_OCL_base oclMatcher(ocl::BruteForceMatcher_OCL_base::L2Dist);
162 ocl::oclMat oclTrainIdx, oclDistance, oclNMatches;
163
164 OCL_TEST_CYCLE()
165 oclMatcher.radiusMatch(oclQuery, oclTrain, matches, max_distance);
166
167 oclMatcher.radiusMatchDownload(oclTrainIdx, oclDistance, oclNMatches, matches);
168
169 std::vector<DMatch> & matches0 = matches[0], & matches1 = matches[1];
170 SANITY_CHECK_MATCHES(matches0);
171 SANITY_CHECK_MATCHES(matches1);
172 }
173 else
174 OCL_PERF_ELSE
175}
