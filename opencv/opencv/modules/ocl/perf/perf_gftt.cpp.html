49using namespace perf;
50using std::tr1::tuple;
51using std::tr1::get;
52
53///////////// GoodFeaturesToTrack ////////////////////////
54
55typedef tuple<String, double, bool> GoodFeaturesToTrackParams;
56typedef TestBaseWithParam<GoodFeaturesToTrackParams> GoodFeaturesToTrackFixture;
57
58OCL_PERF_TEST_P(GoodFeaturesToTrackFixture, GoodFeaturesToTrack,
59 ::testing::Combine(OCL_PERF_ENUM(String("gpu/opticalflow/rubberwhale1.png")),
60 OCL_PERF_ENUM(0.0, 3.0), testing::Bool()))
61{
62 const GoodFeaturesToTrackParams params = GetParam();
63 const String fileName = get<0>(params);
64 const double minDistance = get<1>(params), qualityLevel = 0.01;
65 const bool harrisDetector = get<2>(params);
66 const int maxCorners = 1000;
67
68 Mat frame = imread(getDataPath(fileName), IMREAD_GRAYSCALE);
69 ASSERT_FALSE(frame.empty()) << "no input image";
70
71 vector<Point2f> pts_gold;
72 declare.in(frame);
73
74 if (RUN_OCL_IMPL)
75 {
76 ocl::oclMat oclFrame(frame), pts_oclmat;
77 ocl::GoodFeaturesToTrackDetector_OCL detector(maxCorners, qualityLevel, minDistance, 3,
78 harrisDetector);
79
80 OCL_TEST_CYCLE() detector(oclFrame, pts_oclmat);
81
82 detector.downloadPoints(pts_oclmat, pts_gold);
83
84 SANITY_CHECK(pts_gold);
85 }
86 else if (RUN_PLAIN_IMPL)
87 {
88 TEST_CYCLE() cv::goodFeaturesToTrack(frame, pts_gold,
89 maxCorners, qualityLevel, minDistance, noArray(), 3, harrisDetector);
90
91 SANITY_CHECK(pts_gold);
92 }
93 else
94 OCL_PERF_ELSE
95}
