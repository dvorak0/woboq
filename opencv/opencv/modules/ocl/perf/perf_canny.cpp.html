46#include "perf_precomp.hpp"
47
48using namespace perf;
49using std::tr1::tuple;
50using std::tr1::get;
51
52///////////// Canny ////////////////////////
53
54typedef tuple<int, bool> CannyParams;
55typedef TestBaseWithParam<CannyParams> CannyFixture;
56
57OCL_PERF_TEST_P(CannyFixture, Canny,
58 ::testing::Combine(OCL_PERF_ENUM(3, 5), testing::Bool()))
59{
60 const CannyParams params = GetParam();
61 int apertureSize = get<0>(params);
62 bool L2Grad = get<1>(params);
63
64 Mat img = imread(getDataPath("gpu/stereobm/aloe-L.png"), cv::IMREAD_GRAYSCALE),
65 edges(img.size(), CV_8UC1);
66 ASSERT_TRUE(!img.empty()) << "can't open aloe-L.png";
67
68 declare.in(img).out(edges);
69
70 if (RUN_OCL_IMPL)
71 {
72 ocl::oclMat oclImg(img), oclEdges(img.size(), CV_8UC1);
73
74 OCL_TEST_CYCLE() ocl::Canny(oclImg, oclEdges, 50.0, 100.0, apertureSize, L2Grad);
75 oclEdges.download(edges);
76 }
77 else if (RUN_PLAIN_IMPL)
78 {
79 TEST_CYCLE() Canny(img, edges, 50.0, 100.0, apertureSize, L2Grad);
80 }
81 else
82 OCL_PERF_ELSE
83
84 SANITY_CHECK_NOTHING();
85}
