41/*//Implementation of the Gaussian mixture model background subtraction from:
42//
43//"Improved adaptive Gausian mixture model for background subtraction"
44//Z.Zivkovic
45//International Conference Pattern Recognition, UK, August, 2004
46//http://www.zoranz.net/Publications/zivkovic2004ICPR.pdf
47//The code is very fast and performs also shadow detection.
48//Number of Gausssian components is adapted per pixel.
49//
50// and
51//
52//"Efficient Adaptive Density Estimapion per Image Pixel for the Task of Background Subtraction"
53//Z.Zivkovic, F. van der Heijden
54//Pattern Recognition Letters, vol. 27, no. 7, pages 773-780, 2006.
55//
56//The algorithm similar to the standard Stauffer&Grimson algorithm with
57//additional selection of the number of the Gaussian components based on:
58//
59//"Recursive unsupervised learning of finite mixture models "
60//Z.Zivkovic, F.van der Heijden
61//IEEE Trans. on Pattern Analysis and Machine Intelligence, vol.26, no.5, pages 651-656, 2004
62//http://www.zoranz.net/Publications/zivkovic2004PAMI.pdf
63//
64//
65//Example usage with as cpp class
66// BackgroundSubtractorMOG2 bg_model;
67//For each new image the model is updates using:
68// bg_model(img, fgmask);
69//
70//Example usage as part of the CvBGStatModel:
71// CvBGStatModel* bg_model = cvCreateGaussianBGModel2( first_frame );
72//
73// //update for each frame
74// cvUpdateBGStatModel( tmp_frame, bg_model );//segmentation result is in bg_model->foreground
75//
76// //release at the program termination
77// cvReleaseBGStatModel( &bg_model );
78//
79//Author: Z.Zivkovic, www.zoranz.net
80//Date: 7-April-2011, Version:1.0

89 Interface of Gaussian mixture algorithm from:
90

97 -fast - number of Gausssian components is constantly adapted per pixel.

102// default parameters of gaussian background detection algorithm
103static const int defaultHistory2 = 500; // Learning rate; alpha = 1/defaultHistory2
104static const float defaultVarThreshold2 = 4.0f*4.0f;
105static const int defaultNMixtures2 = 5; // maximal number of Gaussians in mixture
106static const float defaultBackgroundRatio2 = 0.9f; // threshold sum of weights for background test
107static const float defaultVarThresholdGen2 = 3.0f*3.0f;
108static const float defaultVarInit2 = 15.0f; // initial variance for new components
109static const float defaultVarMax2 = 5*defaultVarInit2;
110static const float defaultVarMin2 = 4.0f;
111
112// additional parameters
113static const float defaultfCT2 = 0.05f; // complexity reduction prior constant 0 - no reduction of number of components
114static const unsigned char defaultnShadowDetection2 = (unsigned char)127; // value to use in the segmentation mask for shadows, set 0 not to do shadow detection
115static const float defaultfTau = 0.5f; // Tau - shadow threshold, see the paper for explanation
116
117struct GaussBGStatModel2Params
118{
119 //image info
120 int nWidth;
121 int nHeight;
122 int nND;//number of data dimensions (image channels)
123
124 bool bPostFiltering;//defult 1 - do postfiltering - will make shadow detection results also give value 255
125 double minArea; // for postfiltering
126
127 bool bInit;//default 1, faster updates at start
128
129 /////////////////////////
130 //very important parameters - things you will change
131 ////////////////////////
132 float fAlphaT;
133 //alpha - speed of update - if the time interval you want to average over is T
134 //set alpha=1/T. It is also usefull at start to make T slowly increase
135 //from 1 until the desired T
136 float fTb;
137 //Tb - threshold on the squared Mahalan. dist. to decide if it is well described
138 //by the background model or not. Related to Cthr from the paper.
139 //This does not influence the update of the background. A typical value could be 4 sigma
140 //and that is Tb=4*4=16;
141
142 /////////////////////////
143 //less important parameters - things you might change but be carefull
144 ////////////////////////
145 float fTg;
146 //Tg - threshold on the squared Mahalan. dist. to decide
147 //when a sample is close to the existing components. If it is not close
148 //to any a new component will be generated. I use 3 sigma => Tg=3*3=9.
149 //Smaller Tg leads to more generated components and higher Tg might make
150 //lead to small number of components but they can grow too large
151 float fTB;//1-cf from the paper
152 //TB - threshold when the component becomes significant enough to be included into
153 //the background model. It is the TB=1-cf from the paper. So I use cf=0.1 => TB=0.
154 //For alpha=0.001 it means that the mode should exist for approximately 105 frames before
155 //it is considered foreground
156 float fVarInit;
157 float fVarMax;
158 float fVarMin;
159 //initial standard deviation for the newly generated components.
160 //It will will influence the speed of adaptation. A good guess should be made.
161 //A simple way is to estimate the typical standard deviation from the images.
162 //I used here 10 as a reasonable value
163 float fCT;//CT - complexity reduction prior
164 //this is related to the number of samples needed to accept that a component
165 //actually exists. We use CT=0.05 of all the samples. By setting CT=0 you get
166 //the standard Stauffer&Grimson algorithm (maybe not exact but very similar)
167
168 //even less important parameters
169 int nM;//max number of modes - const - 4 is usually enough
170
171 //shadow detection parameters
172 bool bShadowDetection;//default 1 - do shadow detection
173 unsigned char nShadowDetection;//do shadow detection - insert this value as the detection result
174 float fTau;
175 // Tau - shadow threshold. The shadow is detected if the pixel is darker
176 //version of the background. Tau is a threshold on how much darker the shadow can be.
177 //Tau= 0.5 means that if pixel is more than 2 times darker then it is not shadow
178 //See: Prati,Mikic,Trivedi,Cucchiarra,"Detecting Moving Shadows...",IEEE PAMI,2003.
179};
180
181struct GMM
182{
183 float weight;
184 float variance;
185};
186
187// shadow detection performed per pixel
188// should work for rgb data, could be usefull for gray scale and depth data as well
189// See: Prati,Mikic,Trivedi,Cucchiarra,"Detecting Moving Shadows...",IEEE PAMI,2003.
190static CV_INLINE bool
191detectShadowGMM(const float* data, int nchannels, int nmodes,
192 const GMM* gmm, const float* mean,
193 float Tb, float TB, float tau)
194{
195 float tWeight = 0;
196
197 // check all the components marked as background:
198 for( int mode = 0; mode < nmodes; mode++, mean += nchannels )
199 {
200 GMM g = gmm[mode];
201
202 float numerator = 0.0f;
203 float denominator = 0.0f;
204 for( int c = 0; c < nchannels; c++ )
205 {
206 numerator += data[c] * mean[c];
207 denominator += mean[c] * mean[c];
208 }
209
210 // no division by zero allowed
211 if( denominator == 0 )
212 return false;
213
214 // if tau < a < 1 then also check the color distortion
215 if( numerator <= denominator && numerator >= tau*denominator )
216 {
217 float a = numerator / denominator;
218 float dist2a = 0.0f;
219
220 for( int c = 0; c < nchannels; c++ )
221 {
222 float dD= a*mean[c] - data[c];
223 dist2a += dD*dD;
224 }
225
226 if (dist2a < Tb*g.variance*a*a)
227 return true;
228 };
229
230 tWeight += g.weight;
231 if( tWeight > TB )
232 return false;
233 };
234 return false;
235}

237//update GMM - the base update function performed per pixel
238//
239//"Efficient Adaptive Density Estimapion per Image Pixel for the Task of Background Subtraction"
240//Z.Zivkovic, F. van der Heijden
241//Pattern Recognition Letters, vol. 27, no. 7, pages 773-780, 2006.

244//additional selection of the number of the Gaussian components based on:

246//"Recursive unsupervised learning of finite mixture models "

251struct MOG2Invoker : ParallelLoopBody
252{
253 MOG2Invoker(const Mat& _src, Mat& _dst,
254 GMM* _gmm, float* _mean,
255 uchar* _modesUsed,
256 int _nmixtures, float _alphaT,
257 float _Tb, float _TB, float _Tg,
258 float _varInit, float _varMin, float _varMax,
259 float _prune, float _tau, bool _detectShadows,
260 uchar _shadowVal)
261 {
262 src = &_src;
263 dst = &_dst;
264 gmm0 = _gmm;
265 mean0 = _mean;
266 modesUsed0 = _modesUsed;
267 nmixtures = _nmixtures;
268 alphaT = _alphaT;
269 Tb = _Tb;
270 TB = _TB;
271 Tg = _Tg;
272 varInit = _varInit;
273 varMin = MIN(_varMin, _varMax);
274 varMax = MAX(_varMin, _varMax);
275 prune = _prune;
276 tau = _tau;
277 detectShadows = _detectShadows;
278 shadowVal = _shadowVal;
279
280 cvtfunc = src->depth() != CV_32F ? getConvertFunc(src->depth(), CV_32F) : 0;
281 }
282
283 void operator()(const Range& range) const
284 {
285 int y0 = range.start, y1 = range.end;
286 int ncols = src->cols, nchannels = src->channels();
287 AutoBuffer<float> buf(src->cols*nchannels);
288 float alpha1 = 1.f - alphaT;
289 float dData[CV_CN_MAX];
290
291 for( int y = y0; y < y1; y++ )
292 {
293 const float* data = buf;
294 if( cvtfunc )
295 cvtfunc( src->ptr(y), src->step, 0, 0, (uchar*)data, 0, Size(ncols*nchannels, 1), 0);
296 else
297 data = src->ptr<float>(y);
298
299 float* mean = mean0 + ncols*nmixtures*nchannels*y;
300 GMM* gmm = gmm0 + ncols*nmixtures*y;
301 uchar* modesUsed = modesUsed0 + ncols*y;
302 uchar* mask = dst->ptr(y);
303
304 for( int x = 0; x < ncols; x++, data += nchannels, gmm += nmixtures, mean += nmixtures*nchannels )
305 {
306 //calculate distances to the modes (+ sort)
307 //here we need to go in descending order!!!
308 bool background = false;//return value -> true - the pixel classified as background
309
310 //internal:
311 bool fitsPDF = false;//if it remains zero a new GMM mode will be added
312 int nmodes = modesUsed[x];//current number of modes in GMM
313 float totalWeight = 0.f;
314
315 float* mean_m = mean;
316
317 //////
318 //go through all modes
319 for( int mode = 0; mode < nmodes; mode++, mean_m += nchannels )
320 {
321 float weight = alpha1*gmm[mode].weight + prune;//need only weight if fit is found
322 int swap_count = 0;
323 ////
324 //fit not found yet
325 if( !fitsPDF )
326 {
327 //check if it belongs to some of the remaining modes
328 float var = gmm[mode].variance;
329
330 //calculate difference and distance
331 float dist2;
332
333 if( nchannels == 3 )
334 {
335 dData[0] = mean_m[0] - data[0];
336 dData[1] = mean_m[1] - data[1];
337 dData[2] = mean_m[2] - data[2];
338 dist2 = dData[0]*dData[0] + dData[1]*dData[1] + dData[2]*dData[2];
339 }
340 else
341 {
342 dist2 = 0.f;
343 for( int c = 0; c < nchannels; c++ )
344 {
345 dData[c] = mean_m[c] - data[c];
346 dist2 += dData[c]*dData[c];
347 }
348 }
349
350 //background? - Tb - usually larger than Tg
351 if( totalWeight < TB && dist2 < Tb*var )
352 background = true;
353
354 //check fit
355 if( dist2 < Tg*var )
356 {
357 /////
358 //belongs to the mode
359 fitsPDF = true;
360
361 //update distribution
362
363 //update weight
364 weight += alphaT;
365 float k = alphaT/weight;
366
367 //update mean
368 for( int c = 0; c < nchannels; c++ )
369 mean_m[c] -= k*dData[c];
370
371 //update variance
372 float varnew = var + k*(dist2-var);
373 //limit the variance
374 varnew = MAX(varnew, varMin);
375 varnew = MIN(varnew, varMax);
376 gmm[mode].variance = varnew;
377
378 //sort
379 //all other weights are at the same place and
380 //only the matched (iModes) is higher -> just find the new place for it
381 for( int i = mode; i > 0; i-- )
382 {
383 //check one up
384 if( weight < gmm[i-1].weight )
385 break;
386
387 swap_count++;
388 //swap one up
389 std::swap(gmm[i], gmm[i-1]);
390 for( int c = 0; c < nchannels; c++ )
391 std::swap(mean[i*nchannels + c], mean[(i-1)*nchannels + c]);
392 }
393 //belongs to the mode - bFitsPDF becomes 1
394 /////
395 }
396 }//!bFitsPDF)
397
398 //check prune
399 if( weight < -prune )
400 {
401 weight = 0.0;
402 nmodes--;
403 }
404
405 gmm[mode-swap_count].weight = weight;//update weight by the calculated value
406 totalWeight += weight;
407 }
408 //go through all modes
409 //////
410
411 //renormalize weights
412 totalWeight = 1.f/totalWeight;
413 for( int mode = 0; mode < nmodes; mode++ )
414 {
415 gmm[mode].weight *= totalWeight;
416 }
417
418 //make new mode if needed and exit
419 if( !fitsPDF )
420 {
421 // replace the weakest or add a new one
422 int mode = nmodes == nmixtures ? nmixtures-1 : nmodes++;
423
424 if (nmodes==1)
425 gmm[mode].weight = 1.f;
426 else
427 {
428 gmm[mode].weight = alphaT;
429
430 // renormalize all other weights
431 for( int i = 0; i < nmodes-1; i++ )
432 gmm[i].weight *= alpha1;
433 }
434
435 // init
436 for( int c = 0; c < nchannels; c++ )
437 mean[mode*nchannels + c] = data[c];
438
439 gmm[mode].variance = varInit;
440
441 //sort
442 //find the new place for it
443 for( int i = nmodes - 1; i > 0; i-- )
444 {
445 // check one up
446 if( alphaT < gmm[i-1].weight )
447 break;
448
449 // swap one up
450 std::swap(gmm[i], gmm[i-1]);
451 for( int c = 0; c < nchannels; c++ )
452 std::swap(mean[i*nchannels + c], mean[(i-1)*nchannels + c]);
453 }
454 }
455
456 //set the number of modes
457 modesUsed[x] = uchar(nmodes);
458 mask[x] = background ? 0 :
459 detectShadows && detectShadowGMM(data, nchannels, nmodes, gmm, mean, Tb, TB, tau) ?
460 shadowVal : 255;
461 }
462 }
463 }
464
465 const Mat* src;
466 Mat* dst;
467 GMM* gmm0;
468 float* mean0;
469 uchar* modesUsed0;
470
471 int nmixtures;
472 float alphaT, Tb, TB, Tg;
473 float varInit, varMin, varMax, prune, tau;
474
475 bool detectShadows;
476 uchar shadowVal;
477
478 BinaryFunc cvtfunc;
479};

530void BackgroundSubtractorMOG2::initialize(Size _frameSize, int _frameType)
531{
532 frameSize = _frameSize;
533 frameType = _frameType;
534 nframes = 0;
535
536 int nchannels = CV_MAT_CN(frameType);
537 CV_Assert( nchannels <= CV_CN_MAX );
538
539 // for each gaussian mixture of each pixel bg model we store ...
540 // the mixture weight (w),
541 // the mean (nchannels values) and
542 // the covariance
543 bgmodel.create( 1, frameSize.height*frameSize.width*nmixtures*(2 + nchannels), CV_32F );
544 //make the array for keeping track of the used modes per pixel - all zeros at start
545 bgmodelUsedModes.create(frameSize,CV_8U);
546 bgmodelUsedModes = Scalar::all(0);
547}
548
549void BackgroundSubtractorMOG2::operator()(InputArray _image, OutputArray _fgmask, double learningRate)
550{
551 Mat image = _image.getMat();
552 bool needToInitialize = nframes == 0 || learningRate >= 1 || image.size() != frameSize || image.type() != frameType;
553
554 if( needToInitialize )
555 initialize(image.size(), image.type());
556
557 _fgmask.create( image.size(), CV_8U );
558 Mat fgmask = _fgmask.getMat();
559
560 ++nframes;
561 learningRate = learningRate >= 0 && nframes > 1 ? learningRate : 1./min( 2*nframes, history );
562 CV_Assert(learningRate >= 0);
563
564 parallel_for_(Range(0, image.rows),
565 MOG2Invoker(image, fgmask,
566 (GMM*)bgmodel.data,
567 (float*)(bgmodel.data + sizeof(GMM)*nmixtures*image.rows*image.cols),
568 bgmodelUsedModes.data, nmixtures, (float)learningRate,
569 (float)varThreshold,
570 backgroundRatio, varThresholdGen,
571 fVarInit, fVarMin, fVarMax, float(-learningRate*fCT), fTau,
572 bShadowDetection, nShadowDetection));
573}

575void BackgroundSubtractorMOG2::getBackgroundImage(OutputArray backgroundImage) const
576{
577 int nchannels = CV_MAT_CN(frameType);
578 CV_Assert(nchannels == 1 || nchannels == 3);
579 Mat meanBackground(frameSize, CV_MAKETYPE(CV_8U, nchannels), Scalar::all(0));
580 int firstGaussianIdx = 0;
581 const GMM* gmm = (GMM*)bgmodel.data;
582 const float* mean = reinterpret_cast<const float*>(gmm + frameSize.width*frameSize.height*nmixtures);
583 std::vector<float> meanVal(nchannels, 0.f);
584 for(int row=0; row<meanBackground.rows; row++)
585 {
586 for(int col=0; col<meanBackground.cols; col++)
587 {
588 int nmodes = bgmodelUsedModes.at<uchar>(row, col);
589 float totalWeight = 0.f;
590 for(int gaussianIdx = firstGaussianIdx; gaussianIdx < firstGaussianIdx + nmodes; gaussianIdx++)
591 {
592 GMM gaussian = gmm[gaussianIdx];
593 size_t meanPosition = gaussianIdx*nchannels;
594 for(int chn = 0; chn < nchannels; chn++)
595 {
596 meanVal[chn] += gaussian.weight * mean[meanPosition + chn];
597 }
598 totalWeight += gaussian.weight;
599
600 if(totalWeight > backgroundRatio)
601 break;
602 }
603 float invWeight = 1.f/totalWeight;
604 switch(nchannels)
605 {
606 case 1:
607 meanBackground.at<uchar>(row, col) = (uchar)(meanVal[0] * invWeight);
608 meanVal[0] = 0.f;
609 break;
610 case 3:
611 Vec3f& meanVec = *reinterpret_cast<Vec3f*>(&meanVal[0]);
612 meanBackground.at<Vec3b>(row, col) = Vec3b(meanVec * invWeight);
613 meanVec = 0.f;
614 break;
615 }
616 firstGaussianIdx += nmixtures;
617 }
618 }
619 meanBackground.copyTo(backgroundImage);
620}
