48static void calcSharrDeriv(const cv::Mat& src, cv::Mat& dst)
49{
50 using namespace cv;
51 using cv::detail::deriv_type;
52 int rows = src.rows, cols = src.cols, cn = src.channels(), colsn = cols*cn, depth = src.depth();
53 CV_Assert(depth == CV_8U);
54 dst.create(rows, cols, CV_MAKETYPE(DataType<deriv_type>::depth, cn*2));
55
56#ifdef HAVE_TEGRA_OPTIMIZATION
57 if (tegra::calcSharrDeriv(src, dst))
58 return;
59#endif
60
61 int x, y, delta = (int)alignSize((cols + 2)*cn, 16);
62 AutoBuffer<deriv_type> _tempBuf(delta*2 + 64);
63 deriv_type *trow0 = alignPtr(_tempBuf + cn, 16), *trow1 = alignPtr(trow0 + delta, 16);
64
65#if CV_SSE2
66 __m128i z = _mm_setzero_si128(), c3 = _mm_set1_epi16(3), c10 = _mm_set1_epi16(10);
67#endif
68
69 for( y = 0; y < rows; y++ )
70 {
71 const uchar* srow0 = src.ptr<uchar>(y > 0 ? y-1 : rows > 1 ? 1 : 0);
72 const uchar* srow1 = src.ptr<uchar>(y);
73 const uchar* srow2 = src.ptr<uchar>(y < rows-1 ? y+1 : rows > 1 ? rows-2 : 0);
74 deriv_type* drow = dst.ptr<deriv_type>(y);
75
76 // do vertical convolution
77 x = 0;
78#if CV_SSE2
79 for( ; x <= colsn - 8; x += 8 )
80 {
81 __m128i s0 = _mm_unpacklo_epi8(_mm_loadl_epi64((const __m128i*)(srow0 + x)), z);
82 __m128i s1 = _mm_unpacklo_epi8(_mm_loadl_epi64((const __m128i*)(srow1 + x)), z);
83 __m128i s2 = _mm_unpacklo_epi8(_mm_loadl_epi64((const __m128i*)(srow2 + x)), z);
84 __m128i t0 = _mm_add_epi16(_mm_mullo_epi16(_mm_add_epi16(s0, s2), c3), _mm_mullo_epi16(s1, c10));
85 __m128i t1 = _mm_sub_epi16(s2, s0);
86 _mm_store_si128((__m128i*)(trow0 + x), t0);
87 _mm_store_si128((__m128i*)(trow1 + x), t1);
88 }
89#endif
90
91#if CV_NEON
92
93 const uint16x8_t q8 = vdupq_n_u16(3);
94 const uint8x8_t d18 = vdup_n_u8(10);
95
96 const int16x8_t q8i = vdupq_n_s16(3);
97 const int16x8_t q9 = vdupq_n_s16(10);
98
99 for( ; x <= colsn - 8; x += 8)
100 {
101 uint8x8_t d0 = vld1_u8((const uint8_t*)&srow0[x]);
102 uint8x8_t d1 = vld1_u8((const uint8_t*)&srow1[x]);
103 uint8x8_t d2 = vld1_u8((const uint8_t*)&srow2[x]);
104 uint16x8_t q4 = vaddl_u8(d0, d2);
105 uint16x8_t q11 = vsubl_u8(d2, d0);
106 uint16x8_t q5 = vmulq_u16(q4, q8);
107 uint16x8_t q6 = vmull_u8(d1, d18);
108 uint16x8_t q10 = vaddq_u16(q6, q5);
109 vst1q_u16((uint16_t*)&trow0[x], q10);
110 vst1q_u16((uint16_t*)&trow1[x], q11);
111
112 }
113#endif
114 for( ; x < colsn; x++ )
115 {
116 int t0 = (srow0[x] + srow2[x])*3 + srow1[x]*10;
117 int t1 = srow2[x] - srow0[x];
118 trow0[x] = (deriv_type)t0;
119 trow1[x] = (deriv_type)t1;
120 }
121
122 // make border
123 int x0 = (cols > 1 ? 1 : 0)*cn, x1 = (cols > 1 ? cols-2 : 0)*cn;
124 for( int k = 0; k < cn; k++ )
125 {
126 trow0[-cn + k] = trow0[x0 + k]; trow0[colsn + k] = trow0[x1 + k];
127 trow1[-cn + k] = trow1[x0 + k]; trow1[colsn + k] = trow1[x1 + k];
128 }
129
130 // do horizontal convolution, interleave the results and store them to dst
131 x = 0;
132#if CV_SSE2
133 for( ; x <= colsn - 8; x += 8 )
134 {
135 __m128i s0 = _mm_loadu_si128((const __m128i*)(trow0 + x - cn));
136 __m128i s1 = _mm_loadu_si128((const __m128i*)(trow0 + x + cn));
137 __m128i s2 = _mm_loadu_si128((const __m128i*)(trow1 + x - cn));
138 __m128i s3 = _mm_load_si128((const __m128i*)(trow1 + x));
139 __m128i s4 = _mm_loadu_si128((const __m128i*)(trow1 + x + cn));
140
141 __m128i t0 = _mm_sub_epi16(s1, s0);
142 __m128i t1 = _mm_add_epi16(_mm_mullo_epi16(_mm_add_epi16(s2, s4), c3), _mm_mullo_epi16(s3, c10));
143 __m128i t2 = _mm_unpacklo_epi16(t0, t1);
144 t0 = _mm_unpackhi_epi16(t0, t1);
145 // this can probably be replaced with aligned stores if we aligned dst properly.
146 _mm_storeu_si128((__m128i*)(drow + x*2), t2);
147 _mm_storeu_si128((__m128i*)(drow + x*2 + 8), t0);
148 }
149#endif
150
151#if CV_NEON
152 for( ; x <= colsn - 8; x += 8 )
153 {
154
155 int16x8_t q0 = vld1q_s16((const int16_t*)&trow0[x+cn]);
156 int16x8_t q1 = vld1q_s16((const int16_t*)&trow0[x-cn]);
157 int16x8_t q2 = vld1q_s16((const int16_t*)&trow1[x+cn]);
158 int16x8_t q3 = vld1q_s16((const int16_t*)&trow1[x-cn]);
159 int16x8_t q5 = vsubq_s16(q0, q1);
160 int16x8_t q6 = vaddq_s16(q2, q3);
161 int16x8_t q4 = vld1q_s16((const int16_t*)&trow1[x]);
162 int16x8_t q7 = vmulq_s16(q6, q8i);
163 int16x8_t q10 = vmulq_s16(q4, q9);
164 int16x8_t q11 = vaddq_s16(q7, q10);
165 int16x4_t d22 = vget_low_s16(q11);
166 int16x4_t d23 = vget_high_s16(q11);
167 int16x4_t d11 = vget_high_s16(q5);
168 int16x4_t d10 = vget_low_s16(q5);
169 int16x4x2_t q5x2, q11x2;
170 q5x2.val[0] = d10; q5x2.val[1] = d22;
171 q11x2.val[0] = d11; q11x2.val[1] = d23;
172 vst2_s16((int16_t*)&drow[x*2], q5x2);
173 vst2_s16((int16_t*)&drow[(x*2)+8], q11x2);
174
175 }
176
177#endif
178
179 for( ; x < colsn; x++ )
180 {
181 deriv_type t0 = (deriv_type)(trow0[x+cn] - trow0[x-cn]);
182 deriv_type t1 = (deriv_type)((trow1[x+cn] + trow1[x-cn])*3 + trow1[x]*10);
183 drow[x*2] = t0; drow[x*2+1] = t1;
184 }
185 }
186}

191 const Mat& _prevImg, const Mat& _prevDeriv, const Mat& _nextImg,
192 const Point2f* _prevPts, Point2f* _nextPts,
193 uchar* _status, float* _err,
194 Size _winSize, TermCriteria _criteria,
195 int _level, int _maxLevel, int _flags, float _minEigThreshold )
196{
197 prevImg = &_prevImg;
198 prevDeriv = &_prevDeriv;
199 nextImg = &_nextImg;
200 prevPts = _prevPts;
201 nextPts = _nextPts;
202 status = _status;
203 err = _err;
204 winSize = _winSize;
205 criteria = _criteria;
206 level = _level;
207 maxLevel = _maxLevel;
208 flags = _flags;
209 minEigThreshold = _minEigThreshold;
210}

212void cv::detail::LKTrackerInvoker::operator()(const Range& range) const
213{
214 Point2f halfWin((winSize.width-1)*0.5f, (winSize.height-1)*0.5f);
215 const Mat& I = *prevImg;
216 const Mat& J = *nextImg;
217 const Mat& derivI = *prevDeriv;
218
219 int j, cn = I.channels(), cn2 = cn*2;
220 cv::AutoBuffer<deriv_type> _buf(winSize.area()*(cn + cn2));
221 int derivDepth = DataType<deriv_type>::depth;
222
223 Mat IWinBuf(winSize, CV_MAKETYPE(derivDepth, cn), (deriv_type*)_buf);
224 Mat derivIWinBuf(winSize, CV_MAKETYPE(derivDepth, cn2), (deriv_type*)_buf + winSize.area()*cn);
225
226 for( int ptidx = range.start; ptidx < range.end; ptidx++ )
227 {
228 Point2f prevPt = prevPts[ptidx]*(float)(1./(1 << level));
229 Point2f nextPt;
230 if( level == maxLevel )
231 {
232 if( flags & OPTFLOW_USE_INITIAL_FLOW )
233 nextPt = nextPts[ptidx]*(float)(1./(1 << level));
234 else
235 nextPt = prevPt;
236 }
237 else
238 nextPt = nextPts[ptidx]*2.f;
239 nextPts[ptidx] = nextPt;
240
241 Point2i iprevPt, inextPt;
242 prevPt -= halfWin;
243 iprevPt.x = cvFloor(prevPt.x);
244 iprevPt.y = cvFloor(prevPt.y);
245
246 if( iprevPt.x < -winSize.width || iprevPt.x >= derivI.cols ||
247 iprevPt.y < -winSize.height || iprevPt.y >= derivI.rows )
248 {
249 if( level == 0 )
250 {
251 if( status )
252 status[ptidx] = false;
253 if( err )
254 err[ptidx] = 0;
255 }
256 continue;
257 }
258
259 float a = prevPt.x - iprevPt.x;
260 float b = prevPt.y - iprevPt.y;
261 const int W_BITS = 14, W_BITS1 = 14;
262 const float FLT_SCALE = 1.f/(1 << 20);
263 int iw00 = cvRound((1.f - a)*(1.f - b)*(1 << W_BITS));
264 int iw01 = cvRound(a*(1.f - b)*(1 << W_BITS));
265 int iw10 = cvRound((1.f - a)*b*(1 << W_BITS));
266 int iw11 = (1 << W_BITS) - iw00 - iw01 - iw10;
267
268 int dstep = (int)(derivI.step/derivI.elemSize1());
269 int stepI = (int)(I.step/I.elemSize1());
270 int stepJ = (int)(J.step/J.elemSize1());
271 float A11 = 0, A12 = 0, A22 = 0;
272
273#if CV_SSE2
274 __m128i qw0 = _mm_set1_epi32(iw00 + (iw01 << 16));
275 __m128i qw1 = _mm_set1_epi32(iw10 + (iw11 << 16));
276 __m128i z = _mm_setzero_si128();
277 __m128i qdelta_d = _mm_set1_epi32(1 << (W_BITS1-1));
278 __m128i qdelta = _mm_set1_epi32(1 << (W_BITS1-5-1));
279 __m128 qA11 = _mm_setzero_ps(), qA12 = _mm_setzero_ps(), qA22 = _mm_setzero_ps();
280#endif
281
282#if CV_NEON
283
284 int CV_DECL_ALIGNED(16) nA11[] = {0, 0, 0, 0}, nA12[] = {0, 0, 0, 0}, nA22[] = {0, 0, 0, 0};
285 const int shifter1 = -(W_BITS - 5); //negative so it shifts right
286 const int shifter2 = -(W_BITS);
287
288 const int16x4_t d26 = vdup_n_s16((int16_t)iw00);
289 const int16x4_t d27 = vdup_n_s16((int16_t)iw01);
290 const int16x4_t d28 = vdup_n_s16((int16_t)iw10);
291 const int16x4_t d29 = vdup_n_s16((int16_t)iw11);
292 const int32x4_t q11 = vdupq_n_s32((int32_t)shifter1);
293 const int32x4_t q12 = vdupq_n_s32((int32_t)shifter2);
294
295#endif
296
297 // extract the patch from the first image, compute covariation matrix of derivatives
298 int x, y;
299 for( y = 0; y < winSize.height; y++ )
300 {
301 const uchar* src = (const uchar*)I.data + (y + iprevPt.y)*stepI + iprevPt.x*cn;
302 const deriv_type* dsrc = (const deriv_type*)derivI.data + (y + iprevPt.y)*dstep + iprevPt.x*cn2;
303
304 deriv_type* Iptr = (deriv_type*)(IWinBuf.data + y*IWinBuf.step);
305 deriv_type* dIptr = (deriv_type*)(derivIWinBuf.data + y*derivIWinBuf.step);
306
307 x = 0;
308
309#if CV_SSE2
310 for( ; x <= winSize.width*cn - 4; x += 4, dsrc += 4*2, dIptr += 4*2 )
311 {
312 __m128i v00, v01, v10, v11, t0, t1;
313
314 v00 = _mm_unpacklo_epi8(_mm_cvtsi32_si128(*(const int*)(src + x)), z);
315 v01 = _mm_unpacklo_epi8(_mm_cvtsi32_si128(*(const int*)(src + x + cn)), z);
316 v10 = _mm_unpacklo_epi8(_mm_cvtsi32_si128(*(const int*)(src + x + stepI)), z);
317 v11 = _mm_unpacklo_epi8(_mm_cvtsi32_si128(*(const int*)(src + x + stepI + cn)), z);
318
319 t0 = _mm_add_epi32(_mm_madd_epi16(_mm_unpacklo_epi16(v00, v01), qw0),
320 _mm_madd_epi16(_mm_unpacklo_epi16(v10, v11), qw1));
321 t0 = _mm_srai_epi32(_mm_add_epi32(t0, qdelta), W_BITS1-5);
322 _mm_storel_epi64((__m128i*)(Iptr + x), _mm_packs_epi32(t0,t0));
323
324 v00 = _mm_loadu_si128((const __m128i*)(dsrc));
325 v01 = _mm_loadu_si128((const __m128i*)(dsrc + cn2));
326 v10 = _mm_loadu_si128((const __m128i*)(dsrc + dstep));
327 v11 = _mm_loadu_si128((const __m128i*)(dsrc + dstep + cn2));
328
329 t0 = _mm_add_epi32(_mm_madd_epi16(_mm_unpacklo_epi16(v00, v01), qw0),
330 _mm_madd_epi16(_mm_unpacklo_epi16(v10, v11), qw1));
331 t1 = _mm_add_epi32(_mm_madd_epi16(_mm_unpackhi_epi16(v00, v01), qw0),
332 _mm_madd_epi16(_mm_unpackhi_epi16(v10, v11), qw1));
333 t0 = _mm_srai_epi32(_mm_add_epi32(t0, qdelta_d), W_BITS1);
334 t1 = _mm_srai_epi32(_mm_add_epi32(t1, qdelta_d), W_BITS1);
335 v00 = _mm_packs_epi32(t0, t1); // Ix0 Iy0 Ix1 Iy1 ...
336
337 _mm_storeu_si128((__m128i*)dIptr, v00);
338 t0 = _mm_srai_epi32(v00, 16); // Iy0 Iy1 Iy2 Iy3
339 t1 = _mm_srai_epi32(_mm_slli_epi32(v00, 16), 16); // Ix0 Ix1 Ix2 Ix3
340
341 __m128 fy = _mm_cvtepi32_ps(t0);
342 __m128 fx = _mm_cvtepi32_ps(t1);
343
344 qA22 = _mm_add_ps(qA22, _mm_mul_ps(fy, fy));
345 qA12 = _mm_add_ps(qA12, _mm_mul_ps(fx, fy));
346 qA11 = _mm_add_ps(qA11, _mm_mul_ps(fx, fx));
347 }
348#endif
349
350#if CV_NEON
351
352 for( ; x <= winSize.width*cn - 4; x += 4, dsrc += 4*2, dIptr += 4*2 )
353 {
354
355 uint8x8_t d0 = vld1_u8(&src[x]);
356 uint8x8_t d2 = vld1_u8(&src[x+cn]);
357 uint16x8_t q0 = vmovl_u8(d0);
358 uint16x8_t q1 = vmovl_u8(d2);
359
360 int32x4_t q5 = vmull_s16(vget_low_s16(vreinterpretq_s16_u16(q0)), d26);
361 int32x4_t q6 = vmull_s16(vget_low_s16(vreinterpretq_s16_u16(q1)), d27);
362
363 uint8x8_t d4 = vld1_u8(&src[x + stepI]);
364 uint8x8_t d6 = vld1_u8(&src[x + stepI + cn]);
365 uint16x8_t q2 = vmovl_u8(d4);
366 uint16x8_t q3 = vmovl_u8(d6);
367
368 int32x4_t q7 = vmull_s16(vget_low_s16(vreinterpretq_s16_u16(q2)), d28);
369 int32x4_t q8 = vmull_s16(vget_low_s16(vreinterpretq_s16_u16(q3)), d29);
370
371 q5 = vaddq_s32(q5, q6);
372 q7 = vaddq_s32(q7, q8);
373 q5 = vaddq_s32(q5, q7);
374
375 int16x4x2_t d0d1 = vld2_s16(dsrc);
376 int16x4x2_t d2d3 = vld2_s16(&dsrc[cn2]);
377
378 q5 = vqrshlq_s32(q5, q11);
379
380 int32x4_t q4 = vmull_s16(d0d1.val[0], d26);
381 q6 = vmull_s16(d0d1.val[1], d26);
382
383 int16x4_t nd0 = vmovn_s32(q5);
384
385 q7 = vmull_s16(d2d3.val[0], d27);
386 q8 = vmull_s16(d2d3.val[1], d27);
387
388 vst1_s16(&Iptr[x], nd0);
389
390 int16x4x2_t d4d5 = vld2_s16(&dsrc[dstep]);
391 int16x4x2_t d6d7 = vld2_s16(&dsrc[dstep+cn2]);
392
393 q4 = vaddq_s32(q4, q7);
394 q6 = vaddq_s32(q6, q8);
395
396 q7 = vmull_s16(d4d5.val[0], d28);
397 int32x4_t nq0 = vmull_s16(d4d5.val[1], d28);
398 q8 = vmull_s16(d6d7.val[0], d29);
399 int32x4_t q15 = vmull_s16(d6d7.val[1], d29);
400
401 q7 = vaddq_s32(q7, q8);
402 nq0 = vaddq_s32(nq0, q15);
403
404 q4 = vaddq_s32(q4, q7);
405 q6 = vaddq_s32(q6, nq0);
406
407 int32x4_t nq1 = vld1q_s32(nA12);
408 int32x4_t nq2 = vld1q_s32(nA22);
409 nq0 = vld1q_s32(nA11);
410
411 q4 = vqrshlq_s32(q4, q12);
412 q6 = vqrshlq_s32(q6, q12);
413
414 q7 = vmulq_s32(q4, q4);
415 q8 = vmulq_s32(q4, q6);
416 q15 = vmulq_s32(q6, q6);
417
418 nq0 = vaddq_s32(nq0, q7);
419 nq1 = vaddq_s32(nq1, q8);
420 nq2 = vaddq_s32(nq2, q15);
421
422 vst1q_s32(nA11, nq0);
423 vst1q_s32(nA12, nq1);
424 vst1q_s32(nA22, nq2);
425
426 int16x4_t d8 = vmovn_s32(q4);
427 int16x4_t d12 = vmovn_s32(q6);
428
429 int16x4x2_t d8d12;
430 d8d12.val[0] = d8; d8d12.val[1] = d12;
431 vst2_s16(dIptr, d8d12);
432 }
433
434#endif
435
436 for( ; x < winSize.width*cn; x++, dsrc += 2, dIptr += 2 )
437 {
438 int ival = CV_DESCALE(src[x]*iw00 + src[x+cn]*iw01 +
439 src[x+stepI]*iw10 + src[x+stepI+cn]*iw11, W_BITS1-5);
440 int ixval = CV_DESCALE(dsrc[0]*iw00 + dsrc[cn2]*iw01 +
441 dsrc[dstep]*iw10 + dsrc[dstep+cn2]*iw11, W_BITS1);
442 int iyval = CV_DESCALE(dsrc[1]*iw00 + dsrc[cn2+1]*iw01 + dsrc[dstep+1]*iw10 +
443 dsrc[dstep+cn2+1]*iw11, W_BITS1);
444
445 Iptr[x] = (short)ival;
446 dIptr[0] = (short)ixval;
447 dIptr[1] = (short)iyval;
448
449 A11 += (float)(ixval*ixval);
450 A12 += (float)(ixval*iyval);
451 A22 += (float)(iyval*iyval);
452 }
453 }
454
455#if CV_SSE2
456 float CV_DECL_ALIGNED(16) A11buf[4], A12buf[4], A22buf[4];
457 _mm_store_ps(A11buf, qA11);
458 _mm_store_ps(A12buf, qA12);
459 _mm_store_ps(A22buf, qA22);
460 A11 += A11buf[0] + A11buf[1] + A11buf[2] + A11buf[3];
461 A12 += A12buf[0] + A12buf[1] + A12buf[2] + A12buf[3];
462 A22 += A22buf[0] + A22buf[1] + A22buf[2] + A22buf[3];
463#endif
464
465#if CV_NEON
466 A11 += (float)(nA11[0] + nA11[1] + nA11[2] + nA11[3]);
467 A12 += (float)(nA12[0] + nA12[1] + nA12[2] + nA12[3]);
468 A22 += (float)(nA22[0] + nA22[1] + nA22[2] + nA22[3]);
469#endif
470
471 A11 *= FLT_SCALE;
472 A12 *= FLT_SCALE;
473 A22 *= FLT_SCALE;
474
475 float D = A11*A22 - A12*A12;
476 float minEig = (A22 + A11 - std::sqrt((A11-A22)*(A11-A22) +
477 4.f*A12*A12))/(2*winSize.width*winSize.height);
478
479 if( err && (flags & CV_LKFLOW_GET_MIN_EIGENVALS) != 0 )
480 err[ptidx] = (float)minEig;
481
482 if( minEig < minEigThreshold || D < FLT_EPSILON )
483 {
484 if( level == 0 && status )
485 status[ptidx] = false;
486 continue;
487 }
488
489 D = 1.f/D;
490
491 nextPt -= halfWin;
492 Point2f prevDelta;
493
494 for( j = 0; j < criteria.maxCount; j++ )
495 {
496 inextPt.x = cvFloor(nextPt.x);
497 inextPt.y = cvFloor(nextPt.y);
498
499 if( inextPt.x < -winSize.width || inextPt.x >= J.cols ||
500 inextPt.y < -winSize.height || inextPt.y >= J.rows )
501 {
502 if( level == 0 && status )
503 status[ptidx] = false;
504 break;
505 }
506
507 a = nextPt.x - inextPt.x;
508 b = nextPt.y - inextPt.y;
509 iw00 = cvRound((1.f - a)*(1.f - b)*(1 << W_BITS));
510 iw01 = cvRound(a*(1.f - b)*(1 << W_BITS));
511 iw10 = cvRound((1.f - a)*b*(1 << W_BITS));
512 iw11 = (1 << W_BITS) - iw00 - iw01 - iw10;
513 float b1 = 0, b2 = 0;
514#if CV_SSE2
515 qw0 = _mm_set1_epi32(iw00 + (iw01 << 16));
516 qw1 = _mm_set1_epi32(iw10 + (iw11 << 16));
517 __m128 qb0 = _mm_setzero_ps(), qb1 = _mm_setzero_ps();
518#endif
519
520#if CV_NEON
521
522 int CV_DECL_ALIGNED(16) nB1[] = {0,0,0,0}, nB2[] = {0,0,0,0};
523
524 const int16x4_t d26_2 = vdup_n_s16((int16_t)iw00);
525 const int16x4_t d27_2 = vdup_n_s16((int16_t)iw01);
526 const int16x4_t d28_2 = vdup_n_s16((int16_t)iw10);
527 const int16x4_t d29_2 = vdup_n_s16((int16_t)iw11);
528
529#endif
530
531 for( y = 0; y < winSize.height; y++ )
532 {
533 const uchar* Jptr = (const uchar*)J.data + (y + inextPt.y)*stepJ + inextPt.x*cn;
534 const deriv_type* Iptr = (const deriv_type*)(IWinBuf.data + y*IWinBuf.step);
535 const deriv_type* dIptr = (const deriv_type*)(derivIWinBuf.data + y*derivIWinBuf.step);
536
537 x = 0;
538
539#if CV_SSE2
540 for( ; x <= winSize.width*cn - 8; x += 8, dIptr += 8*2 )
541 {
542 __m128i diff0 = _mm_loadu_si128((const __m128i*)(Iptr + x)), diff1;
543 __m128i v00 = _mm_unpacklo_epi8(_mm_loadl_epi64((const __m128i*)(Jptr + x)), z);
544 __m128i v01 = _mm_unpacklo_epi8(_mm_loadl_epi64((const __m128i*)(Jptr + x + cn)), z);
545 __m128i v10 = _mm_unpacklo_epi8(_mm_loadl_epi64((const __m128i*)(Jptr + x + stepJ)), z);
546 __m128i v11 = _mm_unpacklo_epi8(_mm_loadl_epi64((const __m128i*)(Jptr + x + stepJ + cn)), z);
547
548 __m128i t0 = _mm_add_epi32(_mm_madd_epi16(_mm_unpacklo_epi16(v00, v01), qw0),
549 _mm_madd_epi16(_mm_unpacklo_epi16(v10, v11), qw1));
550 __m128i t1 = _mm_add_epi32(_mm_madd_epi16(_mm_unpackhi_epi16(v00, v01), qw0),
551 _mm_madd_epi16(_mm_unpackhi_epi16(v10, v11), qw1));
552 t0 = _mm_srai_epi32(_mm_add_epi32(t0, qdelta), W_BITS1-5);
553 t1 = _mm_srai_epi32(_mm_add_epi32(t1, qdelta), W_BITS1-5);
554 diff0 = _mm_subs_epi16(_mm_packs_epi32(t0, t1), diff0);
555 diff1 = _mm_unpackhi_epi16(diff0, diff0);
556 diff0 = _mm_unpacklo_epi16(diff0, diff0); // It0 It0 It1 It1 ...
557 v00 = _mm_loadu_si128((const __m128i*)(dIptr)); // Ix0 Iy0 Ix1 Iy1 ...
558 v01 = _mm_loadu_si128((const __m128i*)(dIptr + 8));
559 v10 = _mm_mullo_epi16(v00, diff0);
560 v11 = _mm_mulhi_epi16(v00, diff0);
561 v00 = _mm_unpacklo_epi16(v10, v11);
562 v10 = _mm_unpackhi_epi16(v10, v11);
563 qb0 = _mm_add_ps(qb0, _mm_cvtepi32_ps(v00));
564 qb1 = _mm_add_ps(qb1, _mm_cvtepi32_ps(v10));
565 v10 = _mm_mullo_epi16(v01, diff1);
566 v11 = _mm_mulhi_epi16(v01, diff1);
567 v00 = _mm_unpacklo_epi16(v10, v11);
568 v10 = _mm_unpackhi_epi16(v10, v11);
569 qb0 = _mm_add_ps(qb0, _mm_cvtepi32_ps(v00));
570 qb1 = _mm_add_ps(qb1, _mm_cvtepi32_ps(v10));
571 }
572#endif
573
574#if CV_NEON
575
576 for( ; x <= winSize.width*cn - 8; x += 8, dIptr += 8*2 )
577 {
578
579 uint8x8_t d0 = vld1_u8(&Jptr[x]);
580 uint8x8_t d2 = vld1_u8(&Jptr[x+cn]);
581 uint8x8_t d4 = vld1_u8(&Jptr[x+stepJ]);
582 uint8x8_t d6 = vld1_u8(&Jptr[x+stepJ+cn]);
583
584 uint16x8_t q0 = vmovl_u8(d0);
585 uint16x8_t q1 = vmovl_u8(d2);
586 uint16x8_t q2 = vmovl_u8(d4);
587 uint16x8_t q3 = vmovl_u8(d6);
588
589 int32x4_t nq4 = vmull_s16(vget_low_s16(vreinterpretq_s16_u16(q0)), d26_2);
590 int32x4_t nq5 = vmull_s16(vget_high_s16(vreinterpretq_s16_u16(q0)), d26_2);
591
592 int32x4_t nq6 = vmull_s16(vget_low_s16(vreinterpretq_s16_u16(q1)), d27_2);
593 int32x4_t nq7 = vmull_s16(vget_high_s16(vreinterpretq_s16_u16(q1)), d27_2);
594
595 int32x4_t nq8 = vmull_s16(vget_low_s16(vreinterpretq_s16_u16(q2)), d28_2);
596 int32x4_t nq9 = vmull_s16(vget_high_s16(vreinterpretq_s16_u16(q2)), d28_2);
597
598 int32x4_t nq10 = vmull_s16(vget_low_s16(vreinterpretq_s16_u16(q3)), d29_2);
599 int32x4_t nq11 = vmull_s16(vget_high_s16(vreinterpretq_s16_u16(q3)), d29_2);
600
601 nq4 = vaddq_s32(nq4, nq6);
602 nq5 = vaddq_s32(nq5, nq7);
603 nq8 = vaddq_s32(nq8, nq10);
604 nq9 = vaddq_s32(nq9, nq11);
605
606 int16x8_t q6 = vld1q_s16(&Iptr[x]);
607
608 nq4 = vaddq_s32(nq4, nq8);
609 nq5 = vaddq_s32(nq5, nq9);
610
611 nq8 = vmovl_s16(vget_high_s16(q6));
612 nq6 = vmovl_s16(vget_low_s16(q6));
613
614 nq4 = vqrshlq_s32(nq4, q11);
615 nq5 = vqrshlq_s32(nq5, q11);
616
617 int16x8x2_t q0q1 = vld2q_s16(dIptr);
618 nq11 = vld1q_s32(nB1);
619 int32x4_t nq15 = vld1q_s32(nB2);
620
621 nq4 = vsubq_s32(nq4, nq6);
622 nq5 = vsubq_s32(nq5, nq8);
623
624 int32x4_t nq2 = vmovl_s16(vget_low_s16(q0q1.val[0]));
625 int32x4_t nq3 = vmovl_s16(vget_high_s16(q0q1.val[0]));
626
627 nq7 = vmovl_s16(vget_low_s16(q0q1.val[1]));
628 nq8 = vmovl_s16(vget_high_s16(q0q1.val[1]));
629
630 nq9 = vmulq_s32(nq4, nq2);
631 nq10 = vmulq_s32(nq5, nq3);
632
633 nq4 = vmulq_s32(nq4, nq7);
634 nq5 = vmulq_s32(nq5, nq8);
635
636 nq9 = vaddq_s32(nq9, nq10);
637 nq4 = vaddq_s32(nq4, nq5);
638
639 nq11 = vaddq_s32(nq11, nq9);
640 nq15 = vaddq_s32(nq15, nq4);
641
642 vst1q_s32(nB1, nq11);
643 vst1q_s32(nB2, nq15);
644 }
645
646#endif
647
648 for( ; x < winSize.width*cn; x++, dIptr += 2 )
649 {
650 int diff = CV_DESCALE(Jptr[x]*iw00 + Jptr[x+cn]*iw01 +
651 Jptr[x+stepJ]*iw10 + Jptr[x+stepJ+cn]*iw11,
652 W_BITS1-5) - Iptr[x];
653 b1 += (float)(diff*dIptr[0]);
654 b2 += (float)(diff*dIptr[1]);
655 }
656 }
657
658#if CV_SSE2
659 float CV_DECL_ALIGNED(16) bbuf[4];
660 _mm_store_ps(bbuf, _mm_add_ps(qb0, qb1));
661 b1 += bbuf[0] + bbuf[2];
662 b2 += bbuf[1] + bbuf[3];
663#endif
664
665#if CV_NEON
666
667 b1 += (float)(nB1[0] + nB1[1] + nB1[2] + nB1[3]);
668 b2 += (float)(nB2[0] + nB2[1] + nB2[2] + nB2[3]);
669
670#endif
671
672 b1 *= FLT_SCALE;
673 b2 *= FLT_SCALE;
674
675 Point2f delta( (float)((A12*b2 - A22*b1) * D),
676 (float)((A12*b1 - A11*b2) * D));
677 //delta = -delta;
678
679 nextPt += delta;
680 nextPts[ptidx] = nextPt + halfWin;
681
682 if( delta.ddot(delta) <= criteria.epsilon )
683 break;
684
685 if( j > 0 && std::abs(delta.x + prevDelta.x) < 0.01 &&
686 std::abs(delta.y + prevDelta.y) < 0.01 )
687 {
688 nextPts[ptidx] -= delta*0.5f;
689 break;
690 }
691 prevDelta = delta;
692 }
693
694 if( status[ptidx] && err && level == 0 && (flags & CV_LKFLOW_GET_MIN_EIGENVALS) == 0 )
695 {
696 Point2f nextPoint = nextPts[ptidx] - halfWin;
697 Point inextPoint;
698
699 inextPoint.x = cvFloor(nextPoint.x);
700 inextPoint.y = cvFloor(nextPoint.y);
701
702 if( inextPoint.x < -winSize.width || inextPoint.x >= J.cols ||
703 inextPoint.y < -winSize.height || inextPoint.y >= J.rows )
704 {
705 if( status )
706 status[ptidx] = false;
707 continue;
708 }
709
710 float aa = nextPoint.x - inextPoint.x;
711 float bb = nextPoint.y - inextPoint.y;
712 iw00 = cvRound((1.f - aa)*(1.f - bb)*(1 << W_BITS));
713 iw01 = cvRound(aa*(1.f - bb)*(1 << W_BITS));
714 iw10 = cvRound((1.f - aa)*bb*(1 << W_BITS));
715 iw11 = (1 << W_BITS) - iw00 - iw01 - iw10;
716 float errval = 0.f;
717
718 for( y = 0; y < winSize.height; y++ )
719 {
720 const uchar* Jptr = (const uchar*)J.data + (y + inextPoint.y)*stepJ + inextPoint.x*cn;
721 const deriv_type* Iptr = (const deriv_type*)(IWinBuf.data + y*IWinBuf.step);
722
723 for( x = 0; x < winSize.width*cn; x++ )
724 {
725 int diff = CV_DESCALE(Jptr[x]*iw00 + Jptr[x+cn]*iw01 +
726 Jptr[x+stepJ]*iw10 + Jptr[x+stepJ+cn]*iw11,
727 W_BITS1-5) - Iptr[x];
728 errval += std::abs((float)diff);
729 }
730 }
731 err[ptidx] = errval * 1.f/(32*winSize.width*cn*winSize.height);
732 }
733 }
734}

736int cv::buildOpticalFlowPyramid(InputArray _img, OutputArrayOfArrays pyramid, Size winSize, int maxLevel, bool withDerivatives,
737 int pyrBorder, int derivBorder, bool tryReuseInputImage)
738{
739 Mat img = _img.getMat();
740 CV_Assert(img.depth() == CV_8U && winSize.width > 2 && winSize.height > 2 );
741 int pyrstep = withDerivatives ? 2 : 1;
742
743 pyramid.create(1, (maxLevel + 1) * pyrstep, 0 /*type*/, -1, true, 0);
744
745 int derivType = CV_MAKETYPE(DataType<cv::detail::deriv_type>::depth, img.channels() * 2);
746
747 //level 0
748 bool lvl0IsSet = false;
749 if(tryReuseInputImage && img.isSubmatrix() && (pyrBorder & BORDER_ISOLATED) == 0)
750 {
751 Size wholeSize;
752 Point ofs;
753 img.locateROI(wholeSize, ofs);
754 if (ofs.x >= winSize.width && ofs.y >= winSize.height
755 && ofs.x + img.cols + winSize.width <= wholeSize.width
756 && ofs.y + img.rows + winSize.height <= wholeSize.height)
757 {
758 pyramid.getMatRef(0) = img;
759 lvl0IsSet = true;
760 }
761 }
762
763 if(!lvl0IsSet)
764 {
765 Mat& temp = pyramid.getMatRef(0);
766
767 if(!temp.empty())
768 temp.adjustROI(winSize.height, winSize.height, winSize.width, winSize.width);
769 if(temp.type() != img.type() || temp.cols != winSize.width*2 + img.cols || temp.rows != winSize.height * 2 + img.rows)
770 temp.create(img.rows + winSize.height*2, img.cols + winSize.width*2, img.type());
771
772 if(pyrBorder == BORDER_TRANSPARENT)
773 img.copyTo(temp(Rect(winSize.width, winSize.height, img.cols, img.rows)));
774 else
775 copyMakeBorder(img, temp, winSize.height, winSize.height, winSize.width, winSize.width, pyrBorder);
776 temp.adjustROI(-winSize.height, -winSize.height, -winSize.width, -winSize.width);
777 }
778
779 Size sz = img.size();
780 Mat prevLevel = pyramid.getMatRef(0);
781 Mat thisLevel = prevLevel;
782
783 for(int level = 0; level <= maxLevel; ++level)
784 {
785 if (level != 0)
786 {
787 Mat& temp = pyramid.getMatRef(level * pyrstep);
788
789 if(!temp.empty())
790 temp.adjustROI(winSize.height, winSize.height, winSize.width, winSize.width);
791 if(temp.type() != img.type() || temp.cols != winSize.width*2 + sz.width || temp.rows != winSize.height * 2 + sz.height)
792 temp.create(sz.height + winSize.height*2, sz.width + winSize.width*2, img.type());
793
794 thisLevel = temp(Rect(winSize.width, winSize.height, sz.width, sz.height));
795 pyrDown(prevLevel, thisLevel, sz);
796
797 if(pyrBorder != BORDER_TRANSPARENT)
798 copyMakeBorder(thisLevel, temp, winSize.height, winSize.height, winSize.width, winSize.width, pyrBorder|BORDER_ISOLATED);
799 temp.adjustROI(-winSize.height, -winSize.height, -winSize.width, -winSize.width);
800 }
801
802 if(withDerivatives)
803 {
804 Mat& deriv = pyramid.getMatRef(level * pyrstep + 1);
805
806 if(!deriv.empty())
807 deriv.adjustROI(winSize.height, winSize.height, winSize.width, winSize.width);
808 if(deriv.type() != derivType || deriv.cols != winSize.width*2 + sz.width || deriv.rows != winSize.height * 2 + sz.height)
809 deriv.create(sz.height + winSize.height*2, sz.width + winSize.width*2, derivType);
810
811 Mat derivI = deriv(Rect(winSize.width, winSize.height, sz.width, sz.height));
812 calcSharrDeriv(thisLevel, derivI);
813
814 if(derivBorder != BORDER_TRANSPARENT)
815 copyMakeBorder(derivI, deriv, winSize.height, winSize.height, winSize.width, winSize.width, derivBorder|BORDER_ISOLATED);
816 deriv.adjustROI(-winSize.height, -winSize.height, -winSize.width, -winSize.width);
817 }
818
819 sz = Size((sz.width+1)/2, (sz.height+1)/2);
820 if( sz.width <= winSize.width || sz.height <= winSize.height )
821 {
822 pyramid.create(1, (level + 1) * pyrstep, 0 /*type*/, -1, true, 0);//check this
823 return level;
824 }
825
826 prevLevel = thisLevel;
827 }
828
829 return maxLevel;
830}

832void cv::calcOpticalFlowPyrLK( InputArray _prevImg, InputArray _nextImg,
833 InputArray _prevPts, InputOutputArray _nextPts,
834 OutputArray _status, OutputArray _err,
835 Size winSize, int maxLevel,
836 TermCriteria criteria,
837 int flags, double minEigThreshold )
838{
839 Mat prevPtsMat = _prevPts.getMat();
840 const int derivDepth = DataType<cv::detail::deriv_type>::depth;
841
842 CV_Assert( maxLevel >= 0 && winSize.width > 2 && winSize.height > 2 );
843
844 int level=0, i, npoints;
845 CV_Assert( (npoints = prevPtsMat.checkVector(2, CV_32F, true)) >= 0 );
846
847 if( npoints == 0 )
848 {
849 _nextPts.release();
850 _status.release();
851 _err.release();
852 return;
853 }
854
855 if( !(flags & OPTFLOW_USE_INITIAL_FLOW) )
856 _nextPts.create(prevPtsMat.size(), prevPtsMat.type(), -1, true);
857
858 Mat nextPtsMat = _nextPts.getMat();
859 CV_Assert( nextPtsMat.checkVector(2, CV_32F, true) == npoints );
860
861 const Point2f* prevPts = (const Point2f*)prevPtsMat.data;
862 Point2f* nextPts = (Point2f*)nextPtsMat.data;
863
864 _status.create((int)npoints, 1, CV_8U, -1, true);
865 Mat statusMat = _status.getMat(), errMat;
866 CV_Assert( statusMat.isContinuous() );
867 uchar* status = statusMat.data;
868 float* err = 0;
869
870 for( i = 0; i < npoints; i++ )
871 status[i] = true;
872
873 if( _err.needed() )
874 {
875 _err.create((int)npoints, 1, CV_32F, -1, true);
876 errMat = _err.getMat();
877 CV_Assert( errMat.isContinuous() );
878 err = (float*)errMat.data;
879 }
880
881 vector<Mat> prevPyr, nextPyr;
882 int levels1 = -1;
883 int lvlStep1 = 1;
884 int levels2 = -1;
885 int lvlStep2 = 1;
886
887 if(_prevImg.kind() == _InputArray::STD_VECTOR_MAT)
888 {
889 _prevImg.getMatVector(prevPyr);
890
891 levels1 = int(prevPyr.size()) - 1;
892 CV_Assert(levels1 >= 0);
893
894 if (levels1 % 2 == 1 && prevPyr[0].channels() * 2 == prevPyr[1].channels() && prevPyr[1].depth() == derivDepth)
895 {
896 lvlStep1 = 2;
897 levels1 /= 2;
898 }
899
900 // ensure that pyramid has reqired padding
901 if(levels1 > 0)
902 {
903 Size fullSize;
904 Point ofs;
905 prevPyr[lvlStep1].locateROI(fullSize, ofs);
906 CV_Assert(ofs.x >= winSize.width && ofs.y >= winSize.height
907 && ofs.x + prevPyr[lvlStep1].cols + winSize.width <= fullSize.width
908 && ofs.y + prevPyr[lvlStep1].rows + winSize.height <= fullSize.height);
909 }
910
911 if(levels1 < maxLevel)
912 maxLevel = levels1;
913 }
914
915 if(_nextImg.kind() == _InputArray::STD_VECTOR_MAT)
916 {
917 _nextImg.getMatVector(nextPyr);
918
919 levels2 = int(nextPyr.size()) - 1;
920 CV_Assert(levels2 >= 0);
921
922 if (levels2 % 2 == 1 && nextPyr[0].channels() * 2 == nextPyr[1].channels() && nextPyr[1].depth() == derivDepth)
923 {
924 lvlStep2 = 2;
925 levels2 /= 2;
926 }
927
928 // ensure that pyramid has reqired padding
929 if(levels2 > 0)
930 {
931 Size fullSize;
932 Point ofs;
933 nextPyr[lvlStep2].locateROI(fullSize, ofs);
934 CV_Assert(ofs.x >= winSize.width && ofs.y >= winSize.height
935 && ofs.x + nextPyr[lvlStep2].cols + winSize.width <= fullSize.width
936 && ofs.y + nextPyr[lvlStep2].rows + winSize.height <= fullSize.height);
937 }
938
939 if(levels2 < maxLevel)
940 maxLevel = levels2;
941 }
942
943 if (levels1 < 0)
944 maxLevel = buildOpticalFlowPyramid(_prevImg, prevPyr, winSize, maxLevel, false);
945
946 if (levels2 < 0)
947 maxLevel = buildOpticalFlowPyramid(_nextImg, nextPyr, winSize, maxLevel, false);
948
949 if( (criteria.type & TermCriteria::COUNT) == 0 )
950 criteria.maxCount = 30;
951 else
952 criteria.maxCount = std::min(std::max(criteria.maxCount, 0), 100);
953 if( (criteria.type & TermCriteria::EPS) == 0 )
954 criteria.epsilon = 0.01;
955 else
956 criteria.epsilon = std::min(std::max(criteria.epsilon, 0.), 10.);
957 criteria.epsilon *= criteria.epsilon;
958
959 // dI/dx ~ Ix, dI/dy ~ Iy
960 Mat derivIBuf;
961 if(lvlStep1 == 1)
962 derivIBuf.create(prevPyr[0].rows + winSize.height*2, prevPyr[0].cols + winSize.width*2, CV_MAKETYPE(derivDepth, prevPyr[0].channels() * 2));
963
964 for( level = maxLevel; level >= 0; level-- )
965 {
966 Mat derivI;
967 if(lvlStep1 == 1)
968 {
969 Size imgSize = prevPyr[level * lvlStep1].size();
970 Mat _derivI( imgSize.height + winSize.height*2,
971 imgSize.width + winSize.width*2, derivIBuf.type(), derivIBuf.data );
972 derivI = _derivI(Rect(winSize.width, winSize.height, imgSize.width, imgSize.height));
973 calcSharrDeriv(prevPyr[level * lvlStep1], derivI);
974 copyMakeBorder(derivI, _derivI, winSize.height, winSize.height, winSize.width, winSize.width, BORDER_CONSTANT|BORDER_ISOLATED);
975 }
976 else
977 derivI = prevPyr[level * lvlStep1 + 1];
978
979 CV_Assert(prevPyr[level * lvlStep1].size() == nextPyr[level * lvlStep2].size());
980 CV_Assert(prevPyr[level * lvlStep1].type() == nextPyr[level * lvlStep2].type());
981
982#ifdef HAVE_TEGRA_OPTIMIZATION
983 typedef tegra::LKTrackerInvoker<cv::detail::LKTrackerInvoker> LKTrackerInvoker;
984#else
985 typedef cv::detail::LKTrackerInvoker LKTrackerInvoker;
986#endif
987

1010 double **scale, cv::AutoBuffer<uchar>* buffer )
1011{
1012 const int ALIGN = 8;
1013 int pyrBytes, bufferBytes = 0, elem_size;
1014 int level1 = level + 1;
1015
1016 int i;
1017 CvSize imgSize, levelSize;
1018
1019 *imgI = *imgJ = 0;
1020 *step = 0;
1021 *scale = 0;
1022 *size = 0;
1023
1024 /* check input arguments */
1025 if( ((flags & CV_LKFLOW_PYR_A_READY) != 0 && !pyrA) ||
1026 ((flags & CV_LKFLOW_PYR_B_READY) != 0 && !pyrB) )
1027 CV_Error( CV_StsNullPtr, "Some of the precomputed pyramids are missing" );
1028
1029 if( level < 0 )
1030 CV_Error( CV_StsOutOfRange, "The number of pyramid levels is negative" );
1031
1032 switch( criteria->type )
1033 {
1034 case CV_TERMCRIT_ITER:
1035 criteria->epsilon = 0.f;
1036 break;
1037 case CV_TERMCRIT_EPS:
1038 criteria->max_iter = max_iters;
1039 break;
1040 case CV_TERMCRIT_ITER | CV_TERMCRIT_EPS:
1041 break;
1042 default:
1043 assert( 0 );
1044 CV_Error( CV_StsBadArg, "Invalid termination criteria" );
1045 }
1046
1047 /* compare squared values */
1048 criteria->epsilon *= criteria->epsilon;
1049
1050 /* set pointers and step for every level */
1051 pyrBytes = 0;
1052
1053 imgSize = cvGetSize(imgA);
1054 elem_size = CV_ELEM_SIZE(imgA->type);
1055 levelSize = imgSize;
1056
1057 for( i = 1; i < level1; i++ )

1059 levelSize.width = (levelSize.width + 1) >> 1;
1060 levelSize.height = (levelSize.height + 1) >> 1;
1061

1063 pyrBytes += tstep * levelSize.height;
1064 }
1065
1066 assert( pyrBytes <= imgSize.width * imgSize.height * elem_size * 4 / 3 );
1067
1068 /* buffer_size = <size for patches> + <size for pyramids> */
1069 bufferBytes = (int)((level1 >= 0) * ((pyrA->data.ptr == 0) +
1070 (pyrB->data.ptr == 0)) * pyrBytes +
1071 (sizeof(imgI[0][0]) * 2 + sizeof(step[0][0]) +
1072 sizeof(size[0][0]) + sizeof(scale[0][0])) * level1);

1074 buffer->allocate( bufferBytes );

1079 *scale = (double *) (*step + level1);
1080 *size = (CvSize *)(*scale + level1);

1082 imgI[0][0] = imgA->data.ptr;
1083 imgJ[0][0] = imgB->data.ptr;
1084 step[0][0] = imgA->step;
1085 scale[0][0] = 1;
1086 size[0][0] = imgSize;
1087
1088 if( level > 0 )
1089 {
1090 uchar *bufPtr = (uchar *) (*size + level1);
1091 uchar *ptrA = pyrA->data.ptr;
1092 uchar *ptrB = pyrB->data.ptr;
1093
1094 if( !ptrA )
1095 {
1096 ptrA = bufPtr;
1097 bufPtr += pyrBytes;
1098 }
1099
1100 if( !ptrB )
1101 ptrB = bufPtr;
1102
1103 levelSize = imgSize;
1104
1105 /* build pyramids for both frames */
1106 for( i = 1; i <= level; i++ )
1107 {
1108 int levelBytes;
1109 CvMat prev_level, next_level;
1110
1111 levelSize.width = (levelSize.width + 1) >> 1;
1112 levelSize.height = (levelSize.height + 1) >> 1;
1113

1115 step[0][i] = cvAlign( levelSize.width, ALIGN ) * elem_size;
1116 scale[0][i] = scale[0][i - 1] * 0.5;
1117
1118 levelBytes = step[0][i] * levelSize.height;
1119 imgI[0][i] = (uchar *) ptrA;
1120 ptrA += levelBytes;
1121
1122 if( !(flags & CV_LKFLOW_PYR_A_READY) )
1123 {
1124 prev_level = cvMat( size[0][i-1].height, size[0][i-1].width, CV_8UC1 );
1125 next_level = cvMat( size[0][i].height, size[0][i].width, CV_8UC1 );
1126 cvSetData( &prev_level, imgI[0][i-1], step[0][i-1] );
1127 cvSetData( &next_level, imgI[0][i], step[0][i] );
1128 cvPyrDown( &prev_level, &next_level );
1129 }
1130
1131 imgJ[0][i] = (uchar *) ptrB;
1132 ptrB += levelBytes;
1133
1134 if( !(flags & CV_LKFLOW_PYR_B_READY) )
1135 {
1136 prev_level = cvMat( size[0][i-1].height, size[0][i-1].width, CV_8UC1 );

1142 }

1604 to the bottom (original image) */

1630 Av[5] = featuresB[i].y += featuresB[i].y;

1689 G[15] += xy * ixix;
1690 G[16] += xx * ixiy;
1691 G[17] += xy * ixiy;

1698 G[23] += yy * ixiy;

1774 b[5] += y * iyt;

1780 G[k*6+3]*b[3] + G[k*6+4]*b[4] + G[k*6+5]*b[5];

1783 Av[5] = (float)(Av[5] + Av[3] * eta[0] + Av[4] * eta[1]);

1785 t0 = Av[0] * (1 + eta[2]) + Av[1] * eta[4];
1786 t1 = Av[0] * eta[3] + Av[1] * (1 + eta[5]);

1788 Av[1] = (float)t1;
1789

1791 t1 = Av[3] * eta[3] + Av[4] * (1 + eta[5]);
1792 Av[3] = (float)t0;
1793 Av[4] = (float)t1;
1794
1795 if( eta[0] * eta[0] + eta[1] * eta[1] < criteria.epsilon )
1796 break;
1797 }
1798
1799 if( pt_status != 0 || l == 0 )
1800 {
1801 status[i] = (char)pt_status;
1802 featuresB[i].x = Av[2];
1803 featuresB[i].y = Av[5];
1804
1805 matrices[i*4] = Av[0];
1806 matrices[i*4+1] = Av[1];
1807 matrices[i*4+2] = Av[3];
1808 matrices[i*4+3] = Av[4];
1809 }
1810
1811 if( pt_status && l == 0 && error )
1812 {
1813 /* calc error */
1814 double err = 0;
1815
1816 for( y = 0, k = 0; y < patchSize.height; y++ )
1817 {
1818 for( x = 0; x < patchSize.width; x++, k++ )
1819 {
1820 double t = patchI[k] - patchJ[k] + meanJ;
1821 err += t * t;
1822 }
1823 }

1825 }
1826 }
1827 }

2131 fabs( m[3]*pA[i].x + m[4]*pA[i].y + m[5] - pB[i].y ) < MAX(brect.width,brect.height)*0.05 )

2136 break;
2137 }
2138
2139 if( k >= RANSAC_MAX_ITERS )
2140 return 0;
2141
2142 if( good_count < count )
2143 {
2144 for( i = 0; i < good_count; i++ )
2145 {
2146 j = good_idx[i];
2147 pA[i] = pA[j];
2148 pB[i] = pB[j];
2149 }
2150 }
2151
2152 icvGetRTMatrix( pA, pB, good_count, &M, full_affine );

2154 m[5] /= scale;
2155 cvConvert( &M, matM );
