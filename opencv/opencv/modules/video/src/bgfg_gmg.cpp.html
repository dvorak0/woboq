70void cv::BackgroundSubtractorGMG::initialize(cv::Size frameSize, double min, double max)
71{
72 CV_Assert(min < max);
73 CV_Assert(maxFeatures > 0);
74 CV_Assert(learningRate >= 0.0 && learningRate <= 1.0);
75 CV_Assert(numInitializationFrames >= 1);
76 CV_Assert(quantizationLevels >= 1 && quantizationLevels <= 255);
77 CV_Assert(backgroundPrior >= 0.0 && backgroundPrior <= 1.0);
78
79 minVal_ = min;
80 maxVal_ = max;
81
82 frameSize_ = frameSize;
83 frameNum_ = 0;
84
85 nfeatures_.create(frameSize_);
86 colors_.create(frameSize_.area(), maxFeatures);
87 weights_.create(frameSize_.area(), maxFeatures);
88
89 nfeatures_.setTo(cv::Scalar::all(0));
90}

94 float findFeature(unsigned int color, const unsigned int* colors, const float* weights, int nfeatures)
95 {
96 for (int i = 0; i < nfeatures; ++i)
97 {
98 if (color == colors[i])
99 return weights[i];
100 }
101
102 // not in histogram, so return 0.
103 return 0.0f;
104 }
105
106 void normalizeHistogram(float* weights, int nfeatures)
107 {
108 float total = 0.0f;
109 for (int i = 0; i < nfeatures; ++i)
110 total += weights[i];
111
112 if (total != 0.0f)
113 {
114 for (int i = 0; i < nfeatures; ++i)
115 weights[i] /= total;
116 }
117 }
118
119 bool insertFeature(unsigned int color, float weight, unsigned int* colors, float* weights, int& nfeatures, int maxFeatures)
120 {
121 int idx = -1;
122 for (int i = 0; i < nfeatures; ++i)
123 {
124 if (color == colors[i])
125 {
126 // feature in histogram
127 weight += weights[i];
128 idx = i;
129 break;
130 }
131 }
132
133 if (idx >= 0)
134 {
135 // move feature to beginning of list
136
137 ::memmove(colors + 1, colors, idx * sizeof(unsigned int));
138 ::memmove(weights + 1, weights, idx * sizeof(float));
139
140 colors[0] = color;
141 weights[0] = weight;
142 }
143 else if (nfeatures == maxFeatures)
144 {
145 // discard oldest feature
146
147 ::memmove(colors + 1, colors, (nfeatures - 1) * sizeof(unsigned int));
148 ::memmove(weights + 1, weights, (nfeatures - 1) * sizeof(float));
149
150 colors[0] = color;
151 weights[0] = weight;
152 }
153 else
154 {
155 colors[nfeatures] = color;
156 weights[nfeatures] = weight;
157
158 ++nfeatures;
159
160 return true;
161 }
162
163 return false;
164 }
165}

169 template <typename T> struct Quantization
170 {
171 static unsigned int apply(const void* src_, int x, int cn, double minVal, double maxVal, int quantizationLevels)
172 {
173 const T* src = static_cast<const T*>(src_);
174 src += x * cn;
175
176 unsigned int res = 0;
177 for (int i = 0, shift = 0; i < cn; ++i, ++src, shift += 8)
178 res |= static_cast<int>((*src - minVal) * quantizationLevels / (maxVal - minVal)) << shift;
179
180 return res;
181 }
182 };

184 class GMG_LoopBody : public cv::ParallelLoopBody
185 {
186 public:
187 GMG_LoopBody(const cv::Mat& frame, const cv::Mat& fgmask, const cv::Mat_<int>& nfeatures, const cv::Mat_<unsigned int>& colors, const cv::Mat_<float>& weights,
188 int maxFeatures, double learningRate, int numInitializationFrames, int quantizationLevels, double backgroundPrior, double decisionThreshold,
189 double maxVal, double minVal, int frameNum, bool updateBackgroundModel) :
190 frame_(frame), fgmask_(fgmask), nfeatures_(nfeatures), colors_(colors), weights_(weights),
191 maxFeatures_(maxFeatures), learningRate_(learningRate), numInitializationFrames_(numInitializationFrames), quantizationLevels_(quantizationLevels),
192 backgroundPrior_(backgroundPrior), decisionThreshold_(decisionThreshold), updateBackgroundModel_(updateBackgroundModel),
193 maxVal_(maxVal), minVal_(minVal), frameNum_(frameNum)
194 {
195 }
196
197 void operator() (const cv::Range& range) const;
198
199 private:
200 cv::Mat frame_;
201
202 mutable cv::Mat_<uchar> fgmask_;
203
204 mutable cv::Mat_<int> nfeatures_;
205 mutable cv::Mat_<unsigned int> colors_;
206 mutable cv::Mat_<float> weights_;
207
208 int maxFeatures_;
209 double learningRate_;
210 int numInitializationFrames_;
211 int quantizationLevels_;
212 double backgroundPrior_;
213 double decisionThreshold_;
214 bool updateBackgroundModel_;
215
216 double maxVal_;
217 double minVal_;
218 int frameNum_;
219 };
220
221 void GMG_LoopBody::operator() (const cv::Range& range) const
222 {
223 typedef unsigned int (*func_t)(const void* src_, int x, int cn, double minVal, double maxVal, int quantizationLevels);
224 static const func_t funcs[] =
225 {
226 Quantization<uchar>::apply,
227 Quantization<schar>::apply,
228 Quantization<ushort>::apply,
229 Quantization<short>::apply,
230 Quantization<int>::apply,
231 Quantization<float>::apply,
232 Quantization<double>::apply
233 };
234
235 const func_t func = funcs[frame_.depth()];
236 CV_Assert(func != 0);
237
238 const int cn = frame_.channels();
239
240 for (int y = range.start, featureIdx = y * frame_.cols; y < range.end; ++y)
241 {
242 const uchar* frame_row = frame_.ptr(y);
243 int* nfeatures_row = nfeatures_[y];
244 uchar* fgmask_row = fgmask_[y];
245
246 for (int x = 0; x < frame_.cols; ++x, ++featureIdx)
247 {
248 int nfeatures = nfeatures_row[x];
249 unsigned int* colors = colors_[featureIdx];
250 float* weights = weights_[featureIdx];
251
252 unsigned int newFeatureColor = func(frame_row, x, cn, minVal_, maxVal_, quantizationLevels_);
253
254 bool isForeground = false;
255
256 if (frameNum_ >= numInitializationFrames_)
257 {
258 // typical operation
259
260 const double weight = findFeature(newFeatureColor, colors, weights, nfeatures);
261
262 // see Godbehere, Matsukawa, Goldberg (2012) for reasoning behind this implementation of Bayes rule
263 const double posterior = (weight * backgroundPrior_) / (weight * backgroundPrior_ + (1.0 - weight) * (1.0 - backgroundPrior_));
264
265 isForeground = ((1.0 - posterior) > decisionThreshold_);
266
267 // update histogram.
268
269 if (updateBackgroundModel_)
270 {
271 for (int i = 0; i < nfeatures; ++i)
272 weights[i] *= (float)(1.0f - learningRate_);
273
274 bool inserted = insertFeature(newFeatureColor, (float)learningRate_, colors, weights, nfeatures, maxFeatures_);
275
276 if (inserted)
277 {
278 normalizeHistogram(weights, nfeatures);
279 nfeatures_row[x] = nfeatures;
280 }
281 }
282 }
283 else
284 {
285 // training-mode update
286
287 insertFeature(newFeatureColor, 1.0f, colors, weights, nfeatures, maxFeatures_);
288
289 if (frameNum_ == numInitializationFrames_ - 1)
290 normalizeHistogram(weights, nfeatures);
291 }
292
293 fgmask_row[x] = (uchar)(-(schar)isForeground);
294 }
295 }
296 }
297}

299void cv::BackgroundSubtractorGMG::operator ()(InputArray _frame, OutputArray _fgmask, double newLearningRate)
300{
301 Mat frame = _frame.getMat();
302
303 CV_Assert(frame.depth() == CV_8U || frame.depth() == CV_16U || frame.depth() == CV_32F);
304 CV_Assert(frame.channels() == 1 || frame.channels() == 3 || frame.channels() == 4);
305
306 if (newLearningRate != -1.0)
307 {
308 CV_Assert(newLearningRate >= 0.0 && newLearningRate <= 1.0);
309 learningRate = newLearningRate;
310 }
311
312 if (frame.size() != frameSize_)
313 initialize(frame.size(), 0.0, frame.depth() == CV_8U ? 255.0 : frame.depth() == CV_16U ? std::numeric_limits<ushort>::max() : 1.0);
314
315 _fgmask.create(frameSize_, CV_8UC1);
316 Mat fgmask = _fgmask.getMat();
317
318 GMG_LoopBody body(frame, fgmask, nfeatures_, colors_, weights_,
319 maxFeatures, learningRate, numInitializationFrames, quantizationLevels, backgroundPrior, decisionThreshold,
320 maxVal_, minVal_, frameNum_, updateBackgroundModel);
321 parallel_for_(Range(0, frame.rows), body, frame.total()/(double)(1<<16));
322
323 if (smoothingRadius > 0)
324 {
325 medianBlur(fgmask, buf_, smoothingRadius);
326 cv::swap(fgmask, buf_);
327 }
328
329 // keep track of how many frames we have processed
330 ++frameNum_;
331}

334{

337 nfeatures_.release();

340 buf_.release();
341}
