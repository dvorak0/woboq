70static const int defaultNMixtures = 5;
71static const int defaultHistory = 200;
72static const double defaultBackgroundRatio = 0.7;
73static const double defaultVarThreshold = 2.5*2.5;
74static const double defaultNoiseSigma = 30*0.5;
75static const double defaultInitialWeight = 0.05;

77BackgroundSubtractorMOG::BackgroundSubtractorMOG()
78{
79 frameSize = Size(0,0);
80 frameType = 0;
81
82 nframes = 0;
83 nmixtures = defaultNMixtures;
84 history = defaultHistory;
85 varThreshold = defaultVarThreshold;
86 backgroundRatio = defaultBackgroundRatio;
87 noiseSigma = defaultNoiseSigma;
88}

90BackgroundSubtractorMOG::BackgroundSubtractorMOG(int _history, int _nmixtures,
91 double _backgroundRatio,
92 double _noiseSigma)
93{
94 frameSize = Size(0,0);
95 frameType = 0;
96
97 nframes = 0;
98 nmixtures = min(_nmixtures > 0 ? _nmixtures : defaultNMixtures, 8);
99 history = _history > 0 ? _history : defaultHistory;
100 varThreshold = defaultVarThreshold;
101 backgroundRatio = min(_backgroundRatio > 0 ? _backgroundRatio : 0.95, 1.);
102 noiseSigma = _noiseSigma <= 0 ? defaultNoiseSigma : _noiseSigma;
103}

110void BackgroundSubtractorMOG::initialize(Size _frameSize, int _frameType)
111{
112 frameSize = _frameSize;
113 frameType = _frameType;
114 nframes = 0;
115
116 int nchannels = CV_MAT_CN(frameType);
117 CV_Assert( CV_MAT_DEPTH(frameType) == CV_8U );
118
119 // for each gaussian mixture of each pixel bg model we store ...
120 // the mixture sort key (w/sum_of_variances), the mixture weight (w),
121 // the mean (nchannels values) and
122 // the diagonal covariance matrix (another nchannels values)
123 bgmodel.create( 1, frameSize.height*frameSize.width*nmixtures*(2 + 2*nchannels), CV_32F );
124 bgmodel = Scalar::all(0);
125}

128template<typename VT> struct MixData
129{
130 float sortKey;
131 float weight;
132 VT mean;
133 VT var;
134};

137static void process8uC1( const Mat& image, Mat& fgmask, double learningRate,
138 Mat& bgmodel, int nmixtures, double backgroundRatio,
139 double varThreshold, double noiseSigma )
140{
141 int x, y, k, k1, rows = image.rows, cols = image.cols;
142 float alpha = (float)learningRate, T = (float)backgroundRatio, vT = (float)varThreshold;
143 int K = nmixtures;
144 MixData<float>* mptr = (MixData<float>*)bgmodel.data;
145
146 const float w0 = (float)defaultInitialWeight;
147 const float sk0 = (float)(w0/(defaultNoiseSigma*2));
148 const float var0 = (float)(defaultNoiseSigma*defaultNoiseSigma*4);
149 const float minVar = (float)(noiseSigma*noiseSigma);
150
151 for( y = 0; y < rows; y++ )
152 {
153 const uchar* src = image.ptr<uchar>(y);
154 uchar* dst = fgmask.ptr<uchar>(y);
155
156 if( alpha > 0 )
157 {
158 for( x = 0; x < cols; x++, mptr += K )
159 {
160 float wsum = 0;
161 float pix = src[x];
162 int kHit = -1, kForeground = -1;
163
164 for( k = 0; k < K; k++ )
165 {
166 float w = mptr[k].weight;
167 wsum += w;
168 if( w < FLT_EPSILON )
169 break;
170 float mu = mptr[k].mean;
171 float var = mptr[k].var;
172 float diff = pix - mu;
173 float d2 = diff*diff;
174 if( d2 < vT*var )
175 {
176 wsum -= w;
177 float dw = alpha*(1.f - w);
178 mptr[k].weight = w + dw;
179 mptr[k].mean = mu + alpha*diff;
180 var = max(var + alpha*(d2 - var), minVar);
181 mptr[k].var = var;
182 mptr[k].sortKey = w/sqrt(var);
183
184 for( k1 = k-1; k1 >= 0; k1-- )
185 {
186 if( mptr[k1].sortKey >= mptr[k1+1].sortKey )
187 break;
188 std::swap( mptr[k1], mptr[k1+1] );
189 }
190
191 kHit = k1+1;
192 break;
193 }
194 }
195
196 if( kHit < 0 ) // no appropriate gaussian mixture found at all, remove the weakest mixture and create a new one
197 {
198 kHit = k = min(k, K-1);
199 wsum += w0 - mptr[k].weight;
200 mptr[k].weight = w0;
201 mptr[k].mean = pix;
202 mptr[k].var = var0;
203 mptr[k].sortKey = sk0;
204 }
205 else
206 for( ; k < K; k++ )
207 wsum += mptr[k].weight;
208
209 float wscale = 1.f/wsum;
210 wsum = 0;
211 for( k = 0; k < K; k++ )
212 {
213 wsum += mptr[k].weight *= wscale;
214 mptr[k].sortKey *= wscale;
215 if( wsum > T && kForeground < 0 )
216 kForeground = k+1;
217 }
218
219 dst[x] = (uchar)(-(kHit >= kForeground));
220 }
221 }
222 else
223 {
224 for( x = 0; x < cols; x++, mptr += K )
225 {
226 float pix = src[x];
227 int kHit = -1, kForeground = -1;
228
229 for( k = 0; k < K; k++ )
230 {
231 if( mptr[k].weight < FLT_EPSILON )
232 break;
233 float mu = mptr[k].mean;
234 float var = mptr[k].var;
235 float diff = pix - mu;
236 float d2 = diff*diff;
237 if( d2 < vT*var )
238 {
239 kHit = k;
240 break;
241 }
242 }
243
244 if( kHit >= 0 )
245 {
246 float wsum = 0;
247 for( k = 0; k < K; k++ )
248 {
249 wsum += mptr[k].weight;
250 if( wsum > T )
251 {
252 kForeground = k+1;
253 break;
254 }
255 }
256 }
257
258 dst[x] = (uchar)(kHit < 0 || kHit >= kForeground ? 255 : 0);
259 }
260 }
261 }

265static void process8uC3( const Mat& image, Mat& fgmask, double learningRate,
266 Mat& bgmodel, int nmixtures, double backgroundRatio,
267 double varThreshold, double noiseSigma )
268{
269 int x, y, k, k1, rows = image.rows, cols = image.cols;
270 float alpha = (float)learningRate, T = (float)backgroundRatio, vT = (float)varThreshold;
271 int K = nmixtures;
272
273 const float w0 = (float)defaultInitialWeight;
274 const float sk0 = (float)(w0/(defaultNoiseSigma*2*sqrt(3.)));
275 const float var0 = (float)(defaultNoiseSigma*defaultNoiseSigma*4);
276 const float minVar = (float)(noiseSigma*noiseSigma);
277 MixData<Vec3f>* mptr = (MixData<Vec3f>*)bgmodel.data;
278
279 for( y = 0; y < rows; y++ )
280 {
281 const uchar* src = image.ptr<uchar>(y);
282 uchar* dst = fgmask.ptr<uchar>(y);
283
284 if( alpha > 0 )
285 {
286 for( x = 0; x < cols; x++, mptr += K )
287 {
288 float wsum = 0;
289 Vec3f pix(src[x*3], src[x*3+1], src[x*3+2]);
290 int kHit = -1, kForeground = -1;
291
292 for( k = 0; k < K; k++ )
293 {
294 float w = mptr[k].weight;
295 wsum += w;
296 if( w < FLT_EPSILON )
297 break;
298 Vec3f mu = mptr[k].mean;
299 Vec3f var = mptr[k].var;
300 Vec3f diff = pix - mu;
301 float d2 = diff.dot(diff);
302 if( d2 < vT*(var[0] + var[1] + var[2]) )
303 {
304 wsum -= w;
305 float dw = alpha*(1.f - w);
306 mptr[k].weight = w + dw;
307 mptr[k].mean = mu + alpha*diff;
308 var = Vec3f(max(var[0] + alpha*(diff[0]*diff[0] - var[0]), minVar),
309 max(var[1] + alpha*(diff[1]*diff[1] - var[1]), minVar),
310 max(var[2] + alpha*(diff[2]*diff[2] - var[2]), minVar));
311 mptr[k].var = var;
312 mptr[k].sortKey = w/sqrt(var[0] + var[1] + var[2]);
313
314 for( k1 = k-1; k1 >= 0; k1-- )
315 {
316 if( mptr[k1].sortKey >= mptr[k1+1].sortKey )
317 break;
318 std::swap( mptr[k1], mptr[k1+1] );
319 }
320
321 kHit = k1+1;
322 break;
323 }
324 }
325
326 if( kHit < 0 ) // no appropriate gaussian mixture found at all, remove the weakest mixture and create a new one
327 {
328 kHit = k = min(k, K-1);
329 wsum += w0 - mptr[k].weight;
330 mptr[k].weight = w0;
331 mptr[k].mean = pix;
332 mptr[k].var = Vec3f(var0, var0, var0);
333 mptr[k].sortKey = sk0;
334 }
335 else
336 for( ; k < K; k++ )
337 wsum += mptr[k].weight;
338
339 float wscale = 1.f/wsum;
340 wsum = 0;
341 for( k = 0; k < K; k++ )
342 {
343 wsum += mptr[k].weight *= wscale;
344 mptr[k].sortKey *= wscale;
345 if( wsum > T && kForeground < 0 )
346 kForeground = k+1;
347 }
348
349 dst[x] = (uchar)(-(kHit >= kForeground));
350 }
351 }
352 else
353 {
354 for( x = 0; x < cols; x++, mptr += K )
355 {
356 Vec3f pix(src[x*3], src[x*3+1], src[x*3+2]);
357 int kHit = -1, kForeground = -1;
358
359 for( k = 0; k < K; k++ )
360 {
361 if( mptr[k].weight < FLT_EPSILON )
362 break;
363 Vec3f mu = mptr[k].mean;
364 Vec3f var = mptr[k].var;
365 Vec3f diff = pix - mu;
366 float d2 = diff.dot(diff);
367 if( d2 < vT*(var[0] + var[1] + var[2]) )
368 {
369 kHit = k;
370 break;
371 }
372 }
373
374 if( kHit >= 0 )
375 {
376 float wsum = 0;
377 for( k = 0; k < K; k++ )
378 {
379 wsum += mptr[k].weight;
380 if( wsum > T )
381 {
382 kForeground = k+1;
383 break;
384 }
385 }
386 }
387
388 dst[x] = (uchar)(kHit < 0 || kHit >= kForeground ? 255 : 0);
389 }
390 }
391 }

394void BackgroundSubtractorMOG::operator()(InputArray _image, OutputArray _fgmask, double learningRate)
395{
396 Mat image = _image.getMat();
397 bool needToInitialize = nframes == 0 || learningRate >= 1 || image.size() != frameSize || image.type() != frameType;
398
399 if( needToInitialize )
400 initialize(image.size(), image.type());
401
402 CV_Assert( image.depth() == CV_8U );
403 _fgmask.create( image.size(), CV_8U );
404 Mat fgmask = _fgmask.getMat();
405
406 ++nframes;
407 learningRate = learningRate >= 0 && nframes > 1 ? learningRate : 1./min( nframes, history );
408 CV_Assert(learningRate >= 0);
409
410 if( image.type() == CV_8UC1 )
411 process8uC1( image, fgmask, learningRate, bgmodel, nmixtures, backgroundRatio, varThreshold, noiseSigma );
412 else if( image.type() == CV_8UC3 )
413 process8uC3( image, fgmask, learningRate, bgmodel, nmixtures, backgroundRatio, varThreshold, noiseSigma );
414 else
415 CV_Error( CV_StsUnsupportedFormat, "Only 1- and 3-channel 8-bit images are supported in BackgroundSubtractorMOG" );
416}
