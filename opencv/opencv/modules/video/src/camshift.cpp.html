44/*F///////////////////////////////////////////////////////////////////////////////////////
45// Name: cvMeanShift
46// Purpose: MeanShift algorithm
47// Context:
48// Parameters:
49// imgProb - 2D object probability distribution
50// windowIn - CvRect of CAMSHIFT Window intial size
51// numIters - If CAMSHIFT iterates this many times, stop
52// windowOut - Location, height and width of converged CAMSHIFT window
53// len - If != NULL, return equivalent len
54// width - If != NULL, return equivalent width
55// Returns:
56// Number of iterations CAMSHIFT took to converge
57// Notes:
58//F*/
59CV_IMPL int
60cvMeanShift( const void* imgProb, CvRect windowIn,
61 CvTermCriteria criteria, CvConnectedComp* comp )
62{
63 CvMoments moments;
64 int i = 0, eps;
65 CvMat stub, *mat = (CvMat*)imgProb;
66 CvMat cur_win;
67 CvRect cur_rect = windowIn;
68
69 if( comp )
70 comp->rect = windowIn;
71
72 moments.m00 = moments.m10 = moments.m01 = 0;
73
74 mat = cvGetMat( mat, &stub );
75
76 if( CV_MAT_CN( mat->type ) > 1 )
77 CV_Error( CV_BadNumChannels, cvUnsupportedFormat );
78
79 if( windowIn.height <= 0 || windowIn.width <= 0 )
80 CV_Error( CV_StsBadArg, "Input window has non-positive sizes" );
81
82 windowIn = cv::Rect(windowIn) & cv::Rect(0, 0, mat->cols, mat->rows);
83
84 criteria = cvCheckTermCriteria( criteria, 1., 100 );
85 eps = cvRound( criteria.epsilon * criteria.epsilon );
86
87 for( i = 0; i < criteria.max_iter; i++ )
88 {
89 int dx, dy, nx, ny;
90 double inv_m00;
91 cur_rect = cv::Rect(cur_rect) & cv::Rect(0, 0, mat->cols, mat->rows);
92 if( cv::Rect(cur_rect) == cv::Rect() )
93 {
94 cur_rect.x = mat->cols/2;
95 cur_rect.y = mat->rows/2;
96 }
97 cur_rect.width = MAX(cur_rect.width, 1);
98 cur_rect.height = MAX(cur_rect.height, 1);
99
100 cvGetSubRect( mat, &cur_win, cur_rect );
101 cvMoments( &cur_win, &moments );
102
103 /* Calculating center of mass */
104 if( fabs(moments.m00) < DBL_EPSILON )
105 break;
106
107 inv_m00 = moments.inv_sqrt_m00*moments.inv_sqrt_m00;
108 dx = cvRound( moments.m10 * inv_m00 - windowIn.width*0.5 );
109 dy = cvRound( moments.m01 * inv_m00 - windowIn.height*0.5 );
110
111 nx = cur_rect.x + dx;
112 ny = cur_rect.y + dy;
113
114 if( nx < 0 )
115 nx = 0;
116 else if( nx + cur_rect.width > mat->cols )
117 nx = mat->cols - cur_rect.width;
118
119 if( ny < 0 )
120 ny = 0;
121 else if( ny + cur_rect.height > mat->rows )
122 ny = mat->rows - cur_rect.height;
123
124 dx = nx - cur_rect.x;
125 dy = ny - cur_rect.y;
126 cur_rect.x = nx;
127 cur_rect.y = ny;
128
129 /* Check for coverage centers mass & window */
130 if( dx*dx + dy*dy < eps )
131 break;
132 }
133
134 if( comp )
135 {
136 comp->rect = cur_rect;
137 comp->area = (float)moments.m00;
138 }
139
140 return i;
141}

144/*F///////////////////////////////////////////////////////////////////////////////////////
145// Name: cvCamShift
146// Purpose: CAMSHIFT algorithm
147// Context:
148// Parameters:
149// imgProb - 2D object probability distribution
150// windowIn - CvRect of CAMSHIFT Window intial size
151// criteria - criteria of stop finding window
152// windowOut - Location, height and width of converged CAMSHIFT window
153// orientation - If != NULL, return distribution orientation
154// len - If != NULL, return equivalent len
155// width - If != NULL, return equivalent width
156// area - sum of all elements in result window
157// Returns:
158// Number of iterations CAMSHIFT took to converge
159// Notes:
160//F*/
161CV_IMPL int
162cvCamShift( const void* imgProb, CvRect windowIn,
163 CvTermCriteria criteria,
164 CvConnectedComp* _comp,
165 CvBox2D* box )
166{
167 const int TOLERANCE = 10;
168 CvMoments moments;
169 double m00 = 0, m10, m01, mu20, mu11, mu02, inv_m00;
170 double a, b, c, xc, yc;
171 double rotate_a, rotate_c;
172 double theta = 0, square;
173 double cs, sn;
174 double length = 0, width = 0;
175 int itersUsed = 0;
176 CvConnectedComp comp;
177 CvMat cur_win, stub, *mat = (CvMat*)imgProb;
178
179 comp.rect = windowIn;
180
181 mat = cvGetMat( mat, &stub );
182
183 itersUsed = cvMeanShift( mat, windowIn, criteria, &comp );
184 windowIn = comp.rect;
185
186 windowIn.x -= TOLERANCE;
187 if( windowIn.x < 0 )
188 windowIn.x = 0;
189
190 windowIn.y -= TOLERANCE;
191 if( windowIn.y < 0 )
192 windowIn.y = 0;
193
194 windowIn.width += 2 * TOLERANCE;
195 if( windowIn.x + windowIn.width > mat->width )
196 windowIn.width = mat->width - windowIn.x;
197
198 windowIn.height += 2 * TOLERANCE;
199 if( windowIn.y + windowIn.height > mat->height )
200 windowIn.height = mat->height - windowIn.y;
201
202 cvGetSubRect( mat, &cur_win, windowIn );
203
204 /* Calculating moments in new center mass */
205 cvMoments( &cur_win, &moments );
206
207 m00 = moments.m00;
208 m10 = moments.m10;
209 m01 = moments.m01;
210 mu11 = moments.mu11;
211 mu20 = moments.mu20;
212 mu02 = moments.mu02;
213
214 if( fabs(m00) < DBL_EPSILON )
215 return -1;
216
217 inv_m00 = 1. / m00;
218 xc = cvRound( m10 * inv_m00 + windowIn.x );
219 yc = cvRound( m01 * inv_m00 + windowIn.y );
220 a = mu20 * inv_m00;
221 b = mu11 * inv_m00;
222 c = mu02 * inv_m00;
223
224 /* Calculating width & height */
225 square = sqrt( 4 * b * b + (a - c) * (a - c) );
226
227 /* Calculating orientation */
228 theta = atan2( 2 * b, a - c + square );
229
230 /* Calculating width & length of figure */
231 cs = cos( theta );
232 sn = sin( theta );
233
234 rotate_a = cs * cs * mu20 + 2 * cs * sn * mu11 + sn * sn * mu02;
235 rotate_c = sn * sn * mu20 - 2 * cs * sn * mu11 + cs * cs * mu02;
236 length = sqrt( rotate_a * inv_m00 ) * 4;
237 width = sqrt( rotate_c * inv_m00 ) * 4;
238
239 /* In case, when tetta is 0 or 1.57... the Length & Width may be exchanged */
240 if( length < width )
241 {
242 double t;
243
244 CV_SWAP( length, width, t );
245 CV_SWAP( cs, sn, t );
246 theta = CV_PI*0.5 - theta;
247 }
248
249 /* Saving results */
250 if( _comp || box )
251 {
252 int t0, t1;
253 int _xc = cvRound( xc );
254 int _yc = cvRound( yc );
255
256 t0 = cvRound( fabs( length * cs ));
257 t1 = cvRound( fabs( width * sn ));
258
259 t0 = MAX( t0, t1 ) + 2;
260 comp.rect.width = MIN( t0, (mat->width - _xc) * 2 );
261
262 t0 = cvRound( fabs( length * sn ));
263 t1 = cvRound( fabs( width * cs ));
264
265 t0 = MAX( t0, t1 ) + 2;
266 comp.rect.height = MIN( t0, (mat->height - _yc) * 2 );
267
268 comp.rect.x = MAX( 0, _xc - comp.rect.width / 2 );
269 comp.rect.y = MAX( 0, _yc - comp.rect.height / 2 );
270
271 comp.rect.width = MIN( mat->width - comp.rect.x, comp.rect.width );
272 comp.rect.height = MIN( mat->height - comp.rect.y, comp.rect.height );
273 comp.area = (float) m00;
274 }
275
276 if( _comp )
277 *_comp = comp;
278
279 if( box )
280 {
281 box->size.height = (float)length;
282 box->size.width = (float)width;
283 box->angle = (float)((CV_PI*0.5+theta)*180./CV_PI);
284 while(box->angle < 0)
285 box->angle += 360;
286 while(box->angle >= 360)
287 box->angle -= 360;
288 if(box->angle >= 180)
289 box->angle -= 180;
290 box->center = cvPoint2D32f( comp.rect.x + comp.rect.width*0.5f,
291 comp.rect.y + comp.rect.height*0.5f);
292 }
293
294 return itersUsed;
295}

298cv::RotatedRect cv::CamShift( InputArray _probImage, Rect& window,
299 TermCriteria criteria )
300{
301 CvConnectedComp comp;
302 CvBox2D box;
303
304 box.center.x = box.center.y = 0; box.angle = 0; box.size.width = box.size.height = 0;
305 comp.rect.x = comp.rect.y = comp.rect.width = comp.rect.height = 0;
306
307 Mat probImage = _probImage.getMat();
308 CvMat c_probImage = probImage;
309 cvCamShift(&c_probImage, window, (CvTermCriteria)criteria, &comp, &box);
310 window = comp.rect;
311 return RotatedRect(Point2f(box.center), Size2f(box.size), box.angle);
312}
313
314int cv::meanShift( InputArray _probImage, Rect& window, TermCriteria criteria )
315{
316 CvConnectedComp comp;
317 Mat probImage = _probImage.getMat();
318 CvMat c_probImage = probImage;
319 int iters = cvMeanShift(&c_probImage, window, (CvTermCriteria)criteria, &comp );
320 window = comp.rect;
321 return iters;
322}
