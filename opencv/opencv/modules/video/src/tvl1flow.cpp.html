140OpticalFlowDual_TVL1::OpticalFlowDual_TVL1()
141{
142 tau = 0.25;
143 lambda = 0.15;
144 theta = 0.3;
145 nscales = 5;
146 warps = 5;
147 epsilon = 0.01;
148 iterations = 300;
149 useInitialFlow = false;

152void OpticalFlowDual_TVL1::calc(InputArray _I0, InputArray _I1, InputOutputArray _flow)
153{
154 Mat I0 = _I0.getMat();
155 Mat I1 = _I1.getMat();
156
157 CV_Assert( I0.type() == CV_8UC1 || I0.type() == CV_32FC1 );
158 CV_Assert( I0.size() == I1.size() );
159 CV_Assert( I0.type() == I1.type() );
160 CV_Assert( !useInitialFlow || (_flow.size() == I0.size() && _flow.type() == CV_32FC2) );
161 CV_Assert( nscales > 0 );
162
163 // allocate memory for the pyramid structure
164 I0s.resize(nscales);
165 I1s.resize(nscales);
166 u1s.resize(nscales);
167 u2s.resize(nscales);
168
169 I0.convertTo(I0s[0], I0s[0].depth(), I0.depth() == CV_8U ? 1.0 : 255.0);
170 I1.convertTo(I1s[0], I1s[0].depth(), I1.depth() == CV_8U ? 1.0 : 255.0);
171
172 u1s[0].create(I0.size());
173 u2s[0].create(I0.size());
174
175 if (useInitialFlow)
176 {
177 Mat_<float> mv[] = {u1s[0], u2s[0]};
178 split(_flow.getMat(), mv);
179 }
180
181 I1x_buf.create(I0.size());
182 I1y_buf.create(I0.size());
183
184 flowMap1_buf.create(I0.size());
185 flowMap2_buf.create(I0.size());
186
187 I1w_buf.create(I0.size());
188 I1wx_buf.create(I0.size());
189 I1wy_buf.create(I0.size());
190
191 grad_buf.create(I0.size());
192 rho_c_buf.create(I0.size());
193
194 v1_buf.create(I0.size());
195 v2_buf.create(I0.size());
196
197 p11_buf.create(I0.size());
198 p12_buf.create(I0.size());
199 p21_buf.create(I0.size());
200 p22_buf.create(I0.size());
201
202 div_p1_buf.create(I0.size());
203 div_p2_buf.create(I0.size());
204
205 u1x_buf.create(I0.size());
206 u1y_buf.create(I0.size());
207 u2x_buf.create(I0.size());
208 u2y_buf.create(I0.size());
209
210 // create the scales
211 for (int s = 1; s < nscales; ++s)
212 {
213 pyrDown(I0s[s - 1], I0s[s]);
214 pyrDown(I1s[s - 1], I1s[s]);
215
216 if (I0s[s].cols < 16 || I0s[s].rows < 16)
217 {
218 nscales = s;
219 break;
220 }
221
222 if (useInitialFlow)
223 {
224 pyrDown(u1s[s - 1], u1s[s]);
225 pyrDown(u2s[s - 1], u2s[s]);
226
227 multiply(u1s[s], Scalar::all(0.5), u1s[s]);
228 multiply(u2s[s], Scalar::all(0.5), u2s[s]);
229 }
230 else
231 {
232 u1s[s].create(I0s[s].size());
233 u2s[s].create(I0s[s].size());
234 }
235 }
236
237 if (!useInitialFlow)
238 {
239 u1s[nscales-1].setTo(Scalar::all(0));
240 u2s[nscales-1].setTo(Scalar::all(0));
241 }
242
243 // pyramidal structure for computing the optical flow
244 for (int s = nscales - 1; s >= 0; --s)
245 {
246 // compute the optical flow at the current scale
247 procOneScale(I0s[s], I1s[s], u1s[s], u2s[s]);
248
249 // if this was the last scale, finish now
250 if (s == 0)
251 break;
252
253 // otherwise, upsample the optical flow
254
255 // zoom the optical flow for the next finer scale
256 resize(u1s[s], u1s[s - 1], I0s[s - 1].size());
257 resize(u2s[s], u2s[s - 1], I0s[s - 1].size());
258
259 // scale the optical flow with the appropriate zoom factor
260 multiply(u1s[s - 1], Scalar::all(2), u1s[s - 1]);
261 multiply(u2s[s - 1], Scalar::all(2), u2s[s - 1]);
262 }
263
264 Mat uxy[] = {u1s[0], u2s[0]};
265 merge(uxy, 2, _flow);
266}

269// buildFlowMap
270
271struct BuildFlowMapBody : ParallelLoopBody
272{
273 void operator() (const Range& range) const;
274
275 Mat_<float> u1;
276 Mat_<float> u2;
277 mutable Mat_<float> map1;
278 mutable Mat_<float> map2;
279};
280
281void BuildFlowMapBody::operator() (const Range& range) const
282{
283 for (int y = range.start; y < range.end; ++y)
284 {
285 const float* u1Row = u1[y];
286 const float* u2Row = u2[y];
287
288 float* map1Row = map1[y];
289 float* map2Row = map2[y];
290
291 for (int x = 0; x < u1.cols; ++x)
292 {
293 map1Row[x] = x + u1Row[x];
294 map2Row[x] = y + u2Row[x];
295 }
296 }
297}
298
299void buildFlowMap(const Mat_<float>& u1, const Mat_<float>& u2, Mat_<float>& map1, Mat_<float>& map2)
300{
301 CV_DbgAssert( u2.size() == u1.size() );
302 CV_DbgAssert( map1.size() == u1.size() );
303 CV_DbgAssert( map2.size() == u1.size() );
304
305 BuildFlowMapBody body;
306
307 body.u1 = u1;
308 body.u2 = u2;
309 body.map1 = map1;
310 body.map2 = map2;
311
312 parallel_for_(Range(0, u1.rows), body);
313}

318struct CenteredGradientBody : ParallelLoopBody
319{
320 void operator() (const Range& range) const;
321
322 Mat_<float> src;
323 mutable Mat_<float> dx;
324 mutable Mat_<float> dy;
325};
326
327void CenteredGradientBody::operator() (const Range& range) const
328{
329 const int last_col = src.cols - 1;
330
331 for (int y = range.start; y < range.end; ++y)
332 {
333 const float* srcPrevRow = src[y - 1];
334 const float* srcCurRow = src[y];
335 const float* srcNextRow = src[y + 1];
336
337 float* dxRow = dx[y];
338 float* dyRow = dy[y];
339
340 for (int x = 1; x < last_col; ++x)
341 {
342 dxRow[x] = 0.5f * (srcCurRow[x + 1] - srcCurRow[x - 1]);
343 dyRow[x] = 0.5f * (srcNextRow[x] - srcPrevRow[x]);
344 }
345 }
346}
347
348void centeredGradient(const Mat_<float>& src, Mat_<float>& dx, Mat_<float>& dy)
349{
350 CV_DbgAssert( src.rows > 2 && src.cols > 2 );
351 CV_DbgAssert( dx.size() == src.size() );
352 CV_DbgAssert( dy.size() == src.size() );
353
354 const int last_row = src.rows - 1;
355 const int last_col = src.cols - 1;
356
357 // compute the gradient on the center body of the image
358 {
359 CenteredGradientBody body;
360
361 body.src = src;
362 body.dx = dx;
363 body.dy = dy;
364
365 parallel_for_(Range(1, last_row), body);
366 }
367
368 // compute the gradient on the first and last rows
369 for (int x = 1; x < last_col; ++x)
370 {
371 dx(0, x) = 0.5f * (src(0, x + 1) - src(0, x - 1));
372 dy(0, x) = 0.5f * (src(1, x) - src(0, x));
373
374 dx(last_row, x) = 0.5f * (src(last_row, x + 1) - src(last_row, x - 1));
375 dy(last_row, x) = 0.5f * (src(last_row, x) - src(last_row - 1, x));
376 }
377
378 // compute the gradient on the first and last columns
379 for (int y = 1; y < last_row; ++y)
380 {
381 dx(y, 0) = 0.5f * (src(y, 1) - src(y, 0));
382 dy(y, 0) = 0.5f * (src(y + 1, 0) - src(y - 1, 0));
383
384 dx(y, last_col) = 0.5f * (src(y, last_col) - src(y, last_col - 1));
385 dy(y, last_col) = 0.5f * (src(y + 1, last_col) - src(y - 1, last_col));
386 }
387
388 // compute the gradient at the four corners
389 dx(0, 0) = 0.5f * (src(0, 1) - src(0, 0));
390 dy(0, 0) = 0.5f * (src(1, 0) - src(0, 0));
391
392 dx(0, last_col) = 0.5f * (src(0, last_col) - src(0, last_col - 1));
393 dy(0, last_col) = 0.5f * (src(1, last_col) - src(0, last_col));
394
395 dx(last_row, 0) = 0.5f * (src(last_row, 1) - src(last_row, 0));
396 dy(last_row, 0) = 0.5f * (src(last_row, 0) - src(last_row - 1, 0));
397
398 dx(last_row, last_col) = 0.5f * (src(last_row, last_col) - src(last_row, last_col - 1));
399 dy(last_row, last_col) = 0.5f * (src(last_row, last_col) - src(last_row - 1, last_col));
400}

403// forwardGradient
404
405struct ForwardGradientBody : ParallelLoopBody
406{
407 void operator() (const Range& range) const;
408
409 Mat_<float> src;
410 mutable Mat_<float> dx;
411 mutable Mat_<float> dy;
412};
413
414void ForwardGradientBody::operator() (const Range& range) const
415{
416 const int last_col = src.cols - 1;
417
418 for (int y = range.start; y < range.end; ++y)
419 {
420 const float* srcCurRow = src[y];
421 const float* srcNextRow = src[y + 1];
422
423 float* dxRow = dx[y];
424 float* dyRow = dy[y];
425
426 for (int x = 0; x < last_col; ++x)
427 {
428 dxRow[x] = srcCurRow[x + 1] - srcCurRow[x];
429 dyRow[x] = srcNextRow[x] - srcCurRow[x];
430 }
431 }
432}
433
434void forwardGradient(const Mat_<float>& src, Mat_<float>& dx, Mat_<float>& dy)
435{
436 CV_DbgAssert( src.rows > 2 && src.cols > 2 );
437 CV_DbgAssert( dx.size() == src.size() );
438 CV_DbgAssert( dy.size() == src.size() );
439
440 const int last_row = src.rows - 1;
441 const int last_col = src.cols - 1;
442
443 // compute the gradient on the central body of the image
444 {
445 ForwardGradientBody body;
446
447 body.src = src;
448 body.dx = dx;
449 body.dy = dy;
450
451 parallel_for_(Range(0, last_row), body);
452 }
453
454 // compute the gradient on the last row
455 for (int x = 0; x < last_col; ++x)
456 {
457 dx(last_row, x) = src(last_row, x + 1) - src(last_row, x);
458 dy(last_row, x) = 0.0f;
459 }
460
461 // compute the gradient on the last column
462 for (int y = 0; y < last_row; ++y)
463 {
464 dx(y, last_col) = 0.0f;
465 dy(y, last_col) = src(y + 1, last_col) - src(y, last_col);
466 }
467
468 dx(last_row, last_col) = 0.0f;
469 dy(last_row, last_col) = 0.0f;
470}

474
475struct DivergenceBody : ParallelLoopBody
476{
477 void operator() (const Range& range) const;
478
479 Mat_<float> v1;
480 Mat_<float> v2;
481 mutable Mat_<float> div;
482};
483
484void DivergenceBody::operator() (const Range& range) const
485{
486 for (int y = range.start; y < range.end; ++y)
487 {
488 const float* v1Row = v1[y];
489 const float* v2PrevRow = v2[y - 1];
490 const float* v2CurRow = v2[y];
491
492 float* divRow = div[y];
493
494 for(int x = 1; x < v1.cols; ++x)
495 {
496 const float v1x = v1Row[x] - v1Row[x - 1];
497 const float v2y = v2CurRow[x] - v2PrevRow[x];
498
499 divRow[x] = v1x + v2y;
500 }
501 }
502}
503
504void divergence(const Mat_<float>& v1, const Mat_<float>& v2, Mat_<float>& div)
505{
506 CV_DbgAssert( v1.rows > 2 && v1.cols > 2 );
507 CV_DbgAssert( v2.size() == v1.size() );
508 CV_DbgAssert( div.size() == v1.size() );
509
510 {
511 DivergenceBody body;
512
513 body.v1 = v1;
514 body.v2 = v2;
515 body.div = div;
516
517 parallel_for_(Range(1, v1.rows), body);
518 }
519
520 // compute the divergence on the first row
521 for(int x = 1; x < v1.cols; ++x)
522 div(0, x) = v1(0, x) - v1(0, x - 1) + v2(0, x);
523
524 // compute the divergence on the first column
525 for (int y = 1; y < v1.rows; ++y)
526 div(y, 0) = v1(y, 0) + v2(y, 0) - v2(y - 1, 0);
527
528 div(0, 0) = v1(0, 0) + v2(0, 0);
529}

531////////////////////////////////////////////////////////////
532// calcGradRho
533
534struct CalcGradRhoBody : ParallelLoopBody
535{
536 void operator() (const Range& range) const;
537
538 Mat_<float> I0;
539 Mat_<float> I1w;
540 Mat_<float> I1wx;
541 Mat_<float> I1wy;
542 Mat_<float> u1;
543 Mat_<float> u2;
544 mutable Mat_<float> grad;
545 mutable Mat_<float> rho_c;
546};
547
548void CalcGradRhoBody::operator() (const Range& range) const
549{
550 for (int y = range.start; y < range.end; ++y)
551 {
552 const float* I0Row = I0[y];
553 const float* I1wRow = I1w[y];
554 const float* I1wxRow = I1wx[y];
555 const float* I1wyRow = I1wy[y];
556 const float* u1Row = u1[y];
557 const float* u2Row = u2[y];
558
559 float* gradRow = grad[y];
560 float* rhoRow = rho_c[y];
561
562 for (int x = 0; x < I0.cols; ++x)
563 {
564 const float Ix2 = I1wxRow[x] * I1wxRow[x];
565 const float Iy2 = I1wyRow[x] * I1wyRow[x];
566
567 // store the |Grad(I1)|^2
568 gradRow[x] = Ix2 + Iy2;
569
570 // compute the constant part of the rho function
571 rhoRow[x] = (I1wRow[x] - I1wxRow[x] * u1Row[x] - I1wyRow[x] * u2Row[x] - I0Row[x]);
572 }
573 }
574}
575
576void calcGradRho(const Mat_<float>& I0, const Mat_<float>& I1w, const Mat_<float>& I1wx, const Mat_<float>& I1wy, const Mat_<float>& u1, const Mat_<float>& u2,
577 Mat_<float>& grad, Mat_<float>& rho_c)
578{
579 CV_DbgAssert( I1w.size() == I0.size() );
580 CV_DbgAssert( I1wx.size() == I0.size() );
581 CV_DbgAssert( I1wy.size() == I0.size() );
582 CV_DbgAssert( u1.size() == I0.size() );
583 CV_DbgAssert( u2.size() == I0.size() );
584 CV_DbgAssert( grad.size() == I0.size() );
585 CV_DbgAssert( rho_c.size() == I0.size() );
586
587 CalcGradRhoBody body;
588
589 body.I0 = I0;
590 body.I1w = I1w;
591 body.I1wx = I1wx;
592 body.I1wy = I1wy;
593 body.u1 = u1;
594 body.u2 = u2;
595 body.grad = grad;
596 body.rho_c = rho_c;
597
598 parallel_for_(Range(0, I0.rows), body);
599}

601////////////////////////////////////////////////////////////
602// estimateV
603
604struct EstimateVBody : ParallelLoopBody
605{
606 void operator() (const Range& range) const;
607
608 Mat_<float> I1wx;
609 Mat_<float> I1wy;
610 Mat_<float> u1;
611 Mat_<float> u2;
612 Mat_<float> grad;
613 Mat_<float> rho_c;
614 mutable Mat_<float> v1;
615 mutable Mat_<float> v2;
616 float l_t;
617};
618
619void EstimateVBody::operator() (const Range& range) const
620{
621 for (int y = range.start; y < range.end; ++y)
622 {
623 const float* I1wxRow = I1wx[y];
624 const float* I1wyRow = I1wy[y];
625 const float* u1Row = u1[y];
626 const float* u2Row = u2[y];
627 const float* gradRow = grad[y];
628 const float* rhoRow = rho_c[y];
629
630 float* v1Row = v1[y];
631 float* v2Row = v2[y];
632
633 for (int x = 0; x < I1wx.cols; ++x)
634 {
635 const float rho = rhoRow[x] + (I1wxRow[x] * u1Row[x] + I1wyRow[x] * u2Row[x]);
636
637 float d1 = 0.0f;
638 float d2 = 0.0f;
639
640 if (rho < -l_t * gradRow[x])
641 {
642 d1 = l_t * I1wxRow[x];
643 d2 = l_t * I1wyRow[x];
644 }
645 else if (rho > l_t * gradRow[x])
646 {
647 d1 = -l_t * I1wxRow[x];
648 d2 = -l_t * I1wyRow[x];
649 }
650 else if (gradRow[x] > numeric_limits<float>::epsilon())
651 {
652 float fi = -rho / gradRow[x];
653 d1 = fi * I1wxRow[x];
654 d2 = fi * I1wyRow[x];
655 }
656
657 v1Row[x] = u1Row[x] + d1;
658 v2Row[x] = u2Row[x] + d2;
659 }
660 }
661}
662
663void estimateV(const Mat_<float>& I1wx, const Mat_<float>& I1wy, const Mat_<float>& u1, const Mat_<float>& u2, const Mat_<float>& grad, const Mat_<float>& rho_c,
664 Mat_<float>& v1, Mat_<float>& v2, float l_t)
665{
666 CV_DbgAssert( I1wy.size() == I1wx.size() );
667 CV_DbgAssert( u1.size() == I1wx.size() );
668 CV_DbgAssert( u2.size() == I1wx.size() );
669 CV_DbgAssert( grad.size() == I1wx.size() );
670 CV_DbgAssert( rho_c.size() == I1wx.size() );
671 CV_DbgAssert( v1.size() == I1wx.size() );
672 CV_DbgAssert( v2.size() == I1wx.size() );
673
674 EstimateVBody body;
675
676 body.I1wx = I1wx;
677 body.I1wy = I1wy;
678 body.u1 = u1;
679 body.u2 = u2;
680 body.grad = grad;
681 body.rho_c = rho_c;
682 body.v1 = v1;
683 body.v2 = v2;
684 body.l_t = l_t;
685
686 parallel_for_(Range(0, I1wx.rows), body);
687}

689////////////////////////////////////////////////////////////
690// estimateU
691
692float estimateU(const Mat_<float>& v1, const Mat_<float>& v2, const Mat_<float>& div_p1, const Mat_<float>& div_p2, Mat_<float>& u1, Mat_<float>& u2, float theta)
693{
694 CV_DbgAssert( v2.size() == v1.size() );
695 CV_DbgAssert( div_p1.size() == v1.size() );
696 CV_DbgAssert( div_p2.size() == v1.size() );
697 CV_DbgAssert( u1.size() == v1.size() );
698 CV_DbgAssert( u2.size() == v1.size() );
699
700 float error = 0.0f;
701 for (int y = 0; y < v1.rows; ++y)
702 {
703 const float* v1Row = v1[y];
704 const float* v2Row = v2[y];
705 const float* divP1Row = div_p1[y];
706 const float* divP2Row = div_p2[y];
707
708 float* u1Row = u1[y];
709 float* u2Row = u2[y];
710
711 for (int x = 0; x < v1.cols; ++x)
712 {
713 const float u1k = u1Row[x];
714 const float u2k = u2Row[x];
715
716 u1Row[x] = v1Row[x] + theta * divP1Row[x];
717 u2Row[x] = v2Row[x] + theta * divP2Row[x];
718
719 error += (u1Row[x] - u1k) * (u1Row[x] - u1k) + (u2Row[x] - u2k) * (u2Row[x] - u2k);
720 }
721 }
722
723 return error;
724}

727// estimateDualVariables
728
729struct EstimateDualVariablesBody : ParallelLoopBody
730{
731 void operator() (const Range& range) const;
732
733 Mat_<float> u1x;
734 Mat_<float> u1y;
735 Mat_<float> u2x;
736 Mat_<float> u2y;
737 mutable Mat_<float> p11;
738 mutable Mat_<float> p12;
739 mutable Mat_<float> p21;
740 mutable Mat_<float> p22;
741 float taut;
742};
743
744void EstimateDualVariablesBody::operator() (const Range& range) const
745{
746 for (int y = range.start; y < range.end; ++y)
747 {
748 const float* u1xRow = u1x[y];
749 const float* u1yRow = u1y[y];
750 const float* u2xRow = u2x[y];
751 const float* u2yRow = u2y[y];
752
753 float* p11Row = p11[y];
754 float* p12Row = p12[y];
755 float* p21Row = p21[y];
756 float* p22Row = p22[y];
757
758 for (int x = 0; x < u1x.cols; ++x)
759 {
760 const float g1 = static_cast<float>(hypot(u1xRow[x], u1yRow[x]));
761 const float g2 = static_cast<float>(hypot(u2xRow[x], u2yRow[x]));
762
763 const float ng1 = 1.0f + taut * g1;
764 const float ng2 = 1.0f + taut * g2;
765
766 p11Row[x] = (p11Row[x] + taut * u1xRow[x]) / ng1;
767 p12Row[x] = (p12Row[x] + taut * u1yRow[x]) / ng1;
768 p21Row[x] = (p21Row[x] + taut * u2xRow[x]) / ng2;
769 p22Row[x] = (p22Row[x] + taut * u2yRow[x]) / ng2;
770 }
771 }
772}

774void estimateDualVariables(const Mat_<float>& u1x, const Mat_<float>& u1y, const Mat_<float>& u2x, const Mat_<float>& u2y,
775 Mat_<float>& p11, Mat_<float>& p12, Mat_<float>& p21, Mat_<float>& p22, float taut)
776{
777 CV_DbgAssert( u1y.size() == u1x.size() );
778 CV_DbgAssert( u2x.size() == u1x.size() );
779 CV_DbgAssert( u2y.size() == u1x.size() );
780 CV_DbgAssert( p11.size() == u1x.size() );
781 CV_DbgAssert( p12.size() == u1x.size() );
782 CV_DbgAssert( p21.size() == u1x.size() );
783 CV_DbgAssert( p22.size() == u1x.size() );
784
785 EstimateDualVariablesBody body;
786
787 body.u1x = u1x;
788 body.u1y = u1y;
789 body.u2x = u2x;

792 body.p12 = p12;
793 body.p21 = p21;
794 body.p22 = p22;
795 body.taut = taut;
796
797 parallel_for_(Range(0, u1x.rows), body);
798}

800void OpticalFlowDual_TVL1::procOneScale(const Mat_<float>& I0, const Mat_<float>& I1, Mat_<float>& u1, Mat_<float>& u2)
801{
802 const float scaledEpsilon = static_cast<float>(epsilon * epsilon * I0.size().area());
803
804 CV_DbgAssert( I1.size() == I0.size() );
805 CV_DbgAssert( I1.type() == I0.type() );
806 CV_DbgAssert( u1.size() == I0.size() );
807 CV_DbgAssert( u2.size() == u1.size() );
808
809 Mat_<float> I1x = I1x_buf(Rect(0, 0, I0.cols, I0.rows));
810 Mat_<float> I1y = I1y_buf(Rect(0, 0, I0.cols, I0.rows));
811 centeredGradient(I1, I1x, I1y);
812
813 Mat_<float> flowMap1 = flowMap1_buf(Rect(0, 0, I0.cols, I0.rows));
814 Mat_<float> flowMap2 = flowMap2_buf(Rect(0, 0, I0.cols, I0.rows));
815
816 Mat_<float> I1w = I1w_buf(Rect(0, 0, I0.cols, I0.rows));
817 Mat_<float> I1wx = I1wx_buf(Rect(0, 0, I0.cols, I0.rows));
818 Mat_<float> I1wy = I1wy_buf(Rect(0, 0, I0.cols, I0.rows));
819
820 Mat_<float> grad = grad_buf(Rect(0, 0, I0.cols, I0.rows));
821 Mat_<float> rho_c = rho_c_buf(Rect(0, 0, I0.cols, I0.rows));
822
823 Mat_<float> v1 = v1_buf(Rect(0, 0, I0.cols, I0.rows));
824 Mat_<float> v2 = v2_buf(Rect(0, 0, I0.cols, I0.rows));
825
826 Mat_<float> p11 = p11_buf(Rect(0, 0, I0.cols, I0.rows));
827 Mat_<float> p12 = p12_buf(Rect(0, 0, I0.cols, I0.rows));
828 Mat_<float> p21 = p21_buf(Rect(0, 0, I0.cols, I0.rows));
829 Mat_<float> p22 = p22_buf(Rect(0, 0, I0.cols, I0.rows));
830 p11.setTo(Scalar::all(0));
831 p12.setTo(Scalar::all(0));
832 p21.setTo(Scalar::all(0));
833 p22.setTo(Scalar::all(0));
834
835 Mat_<float> div_p1 = div_p1_buf(Rect(0, 0, I0.cols, I0.rows));
836 Mat_<float> div_p2 = div_p2_buf(Rect(0, 0, I0.cols, I0.rows));
837
838 Mat_<float> u1x = u1x_buf(Rect(0, 0, I0.cols, I0.rows));
839 Mat_<float> u1y = u1y_buf(Rect(0, 0, I0.cols, I0.rows));
840 Mat_<float> u2x = u2x_buf(Rect(0, 0, I0.cols, I0.rows));
841 Mat_<float> u2y = u2y_buf(Rect(0, 0, I0.cols, I0.rows));
842
843 const float l_t = static_cast<float>(lambda * theta);
844 const float taut = static_cast<float>(tau / theta);
845
846 for (int warpings = 0; warpings < warps; ++warpings)
847 {
848 // compute the warping of the target image and its derivatives
849 buildFlowMap(u1, u2, flowMap1, flowMap2);
850 remap(I1, I1w, flowMap1, flowMap2, INTER_CUBIC);
851 remap(I1x, I1wx, flowMap1, flowMap2, INTER_CUBIC);
852 remap(I1y, I1wy, flowMap1, flowMap2, INTER_CUBIC);
853
854 calcGradRho(I0, I1w, I1wx, I1wy, u1, u2, grad, rho_c);
855
856 float error = numeric_limits<float>::max();
857 for (int n = 0; error > scaledEpsilon && n < iterations; ++n)
858 {
859 // estimate the values of the variable (v1, v2) (thresholding operator TH)
860 estimateV(I1wx, I1wy, u1, u2, grad, rho_c, v1, v2, l_t);
861
862 // compute the divergence of the dual variable (p1, p2)
863 divergence(p11, p12, div_p1);
864 divergence(p21, p22, div_p2);
865
866 // estimate the values of the optical flow (u1, u2)
867 error = estimateU(v1, v2, div_p1, div_p2, u1, u2, static_cast<float>(theta));
868
869 // compute the gradient of the optical flow (Du1, Du2)
870 forwardGradient(u1, u1x, u1y);
871 forwardGradient(u2, u2x, u2y);
872
873 // estimate the values of the dual variable (p1, p2)
874 estimateDualVariables(u1x, u1y, u2x, u2y, p11, p12, p21, p22, taut);
875 }
876 }
