44CV_IMPL CvKalman*
45cvCreateKalman( int DP, int MP, int CP )
46{
47 CvKalman *kalman = 0;
48
49 if( DP <= 0 || MP <= 0 )
50 CV_Error( CV_StsOutOfRange,
51 "state and measurement vectors must have positive number of dimensions" );
52
53 if( CP < 0 )
54 CP = DP;
55
56 /* allocating memory for the structure */
57 kalman = (CvKalman *)cvAlloc( sizeof( CvKalman ));
58 memset( kalman, 0, sizeof(*kalman));
59
60 kalman->DP = DP;
61 kalman->MP = MP;
62 kalman->CP = CP;
63
64 kalman->state_pre = cvCreateMat( DP, 1, CV_32FC1 );
65 cvZero( kalman->state_pre );
66
67 kalman->state_post = cvCreateMat( DP, 1, CV_32FC1 );
68 cvZero( kalman->state_post );
69
70 kalman->transition_matrix = cvCreateMat( DP, DP, CV_32FC1 );
71 cvSetIdentity( kalman->transition_matrix );
72
73 kalman->process_noise_cov = cvCreateMat( DP, DP, CV_32FC1 );
74 cvSetIdentity( kalman->process_noise_cov );
75
76 kalman->measurement_matrix = cvCreateMat( MP, DP, CV_32FC1 );
77 cvZero( kalman->measurement_matrix );
78
79 kalman->measurement_noise_cov = cvCreateMat( MP, MP, CV_32FC1 );
80 cvSetIdentity( kalman->measurement_noise_cov );
81
82 kalman->error_cov_pre = cvCreateMat( DP, DP, CV_32FC1 );
83
84 kalman->error_cov_post = cvCreateMat( DP, DP, CV_32FC1 );
85 cvZero( kalman->error_cov_post );
86
87 kalman->gain = cvCreateMat( DP, MP, CV_32FC1 );
88
89 if( CP > 0 )
90 {
91 kalman->control_matrix = cvCreateMat( DP, CP, CV_32FC1 );
92 cvZero( kalman->control_matrix );
93 }
94
95 kalman->temp1 = cvCreateMat( DP, DP, CV_32FC1 );
96 kalman->temp2 = cvCreateMat( MP, DP, CV_32FC1 );
97 kalman->temp3 = cvCreateMat( MP, MP, CV_32FC1 );
98 kalman->temp4 = cvCreateMat( MP, DP, CV_32FC1 );
99 kalman->temp5 = cvCreateMat( MP, 1, CV_32FC1 );
100
101#if 1
102 kalman->PosterState = kalman->state_pre->data.fl;
103 kalman->PriorState = kalman->state_post->data.fl;
104 kalman->DynamMatr = kalman->transition_matrix->data.fl;
105 kalman->MeasurementMatr = kalman->measurement_matrix->data.fl;
106 kalman->MNCovariance = kalman->measurement_noise_cov->data.fl;
107 kalman->PNCovariance = kalman->process_noise_cov->data.fl;
108 kalman->KalmGainMatr = kalman->gain->data.fl;
109 kalman->PriorErrorCovariance = kalman->error_cov_pre->data.fl;
110 kalman->PosterErrorCovariance = kalman->error_cov_post->data.fl;
111#endif
112
113 return kalman;
114}

117CV_IMPL void
118cvReleaseKalman( CvKalman** _kalman )
119{
120 CvKalman *kalman;
121
122 if( !_kalman )
123 CV_Error( CV_StsNullPtr, "" );
124
125 kalman = *_kalman;
126 if( !kalman )
127 return;
128
129 /* freeing the memory */
130 cvReleaseMat( &kalman->state_pre );
131 cvReleaseMat( &kalman->state_post );
132 cvReleaseMat( &kalman->transition_matrix );
133 cvReleaseMat( &kalman->control_matrix );
134 cvReleaseMat( &kalman->measurement_matrix );
135 cvReleaseMat( &kalman->process_noise_cov );
136 cvReleaseMat( &kalman->measurement_noise_cov );
137 cvReleaseMat( &kalman->error_cov_pre );
138 cvReleaseMat( &kalman->gain );
139 cvReleaseMat( &kalman->error_cov_post );
140 cvReleaseMat( &kalman->temp1 );
141 cvReleaseMat( &kalman->temp2 );
142 cvReleaseMat( &kalman->temp3 );
143 cvReleaseMat( &kalman->temp4 );
144 cvReleaseMat( &kalman->temp5 );
145
146 memset( kalman, 0, sizeof(*kalman));
147
148 /* deallocating the structure */
149 cvFree( _kalman );
150}

153CV_IMPL const CvMat*
154cvKalmanPredict( CvKalman* kalman, const CvMat* control )
155{
156 if( !kalman )
157 CV_Error( CV_StsNullPtr, "" );
158
159 /* update the state */
160 /* x'(k) = A*x(k) */
161 cvMatMulAdd( kalman->transition_matrix, kalman->state_post, 0, kalman->state_pre );
162
163 if( control && kalman->CP > 0 )
164 /* x'(k) = x'(k) + B*u(k) */
165 cvMatMulAdd( kalman->control_matrix, control, kalman->state_pre, kalman->state_pre );
166
167 /* update error covariance matrices */
168 /* temp1 = A*P(k) */
169 cvMatMulAdd( kalman->transition_matrix, kalman->error_cov_post, 0, kalman->temp1 );
170
171 /* P'(k) = temp1*At + Q */
172 cvGEMM( kalman->temp1, kalman->transition_matrix, 1, kalman->process_noise_cov, 1,
173 kalman->error_cov_pre, CV_GEMM_B_T );
174
175 /* handle the case when there will be measurement before the next predict */
176 cvCopy(kalman->state_pre, kalman->state_post);
177
178 return kalman->state_pre;
179}

182CV_IMPL const CvMat*
183cvKalmanCorrect( CvKalman* kalman, const CvMat* measurement )
184{
185 if( !kalman || !measurement )
186 CV_Error( CV_StsNullPtr, "" );
187
188 /* temp2 = H*P'(k) */
189 cvMatMulAdd( kalman->measurement_matrix, kalman->error_cov_pre, 0, kalman->temp2 );
190 /* temp3 = temp2*Ht + R */
191 cvGEMM( kalman->temp2, kalman->measurement_matrix, 1,
192 kalman->measurement_noise_cov, 1, kalman->temp3, CV_GEMM_B_T );
193
194 /* temp4 = inv(temp3)*temp2 = Kt(k) */
195 cvSolve( kalman->temp3, kalman->temp2, kalman->temp4, CV_SVD );
196
197 /* K(k) */
198 cvTranspose( kalman->temp4, kalman->gain );
199
200 /* temp5 = z(k) - H*x'(k) */
201 cvGEMM( kalman->measurement_matrix, kalman->state_pre, -1, measurement, 1, kalman->temp5 );
202
203 /* x(k) = x'(k) + K(k)*temp5 */
204 cvMatMulAdd( kalman->gain, kalman->temp5, kalman->state_pre, kalman->state_post );
205
206 /* P(k) = P'(k) - K(k)*temp2 */
207 cvGEMM( kalman->gain, kalman->temp2, -1, kalman->error_cov_pre, 1,
208 kalman->error_cov_post, 0 );
209
210 return kalman->state_post;
211}

216KalmanFilter::KalmanFilter() {}
217KalmanFilter::KalmanFilter(int dynamParams, int measureParams, int controlParams, int type)
218{
219 init(dynamParams, measureParams, controlParams, type);
220}
221
222void KalmanFilter::init(int DP, int MP, int CP, int type)
223{
224 CV_Assert( DP > 0 && MP > 0 );
225 CV_Assert( type == CV_32F || type == CV_64F );
226 CP = std::max(CP, 0);
227
228 statePre = Mat::zeros(DP, 1, type);
229 statePost = Mat::zeros(DP, 1, type);
230 transitionMatrix = Mat::eye(DP, DP, type);
231
232 processNoiseCov = Mat::eye(DP, DP, type);
233 measurementMatrix = Mat::zeros(MP, DP, type);
234 measurementNoiseCov = Mat::eye(MP, MP, type);
235
236 errorCovPre = Mat::zeros(DP, DP, type);
237 errorCovPost = Mat::zeros(DP, DP, type);
238 gain = Mat::zeros(DP, MP, type);
239
240 if( CP > 0 )
241 controlMatrix = Mat::zeros(DP, CP, type);
242 else
243 controlMatrix.release();
244
245 temp1.create(DP, DP, type);
246 temp2.create(MP, DP, type);
247 temp3.create(MP, MP, type);
248 temp4.create(MP, DP, type);
249 temp5.create(MP, 1, type);
250}
251
252const Mat& KalmanFilter::predict(const Mat& control)
253{
254 // update the state: x'(k) = A*x(k)
255 statePre = transitionMatrix*statePost;
256
257 if( control.data )
258 // x'(k) = x'(k) + B*u(k)
259 statePre += controlMatrix*control;
260
261 // update error covariance matrices: temp1 = A*P(k)
262 temp1 = transitionMatrix*errorCovPost;
263
264 // P'(k) = temp1*At + Q
265 gemm(temp1, transitionMatrix, 1, processNoiseCov, 1, errorCovPre, GEMM_2_T);
266
267 // handle the case when there will be measurement before the next predict.
268 statePre.copyTo(statePost);
269 errorCovPre.copyTo(errorCovPost);
270
271 return statePre;
272}
273
274const Mat& KalmanFilter::correct(const Mat& measurement)
275{
276 // temp2 = H*P'(k)
277 temp2 = measurementMatrix * errorCovPre;
278
279 // temp3 = temp2*Ht + R
280 gemm(temp2, measurementMatrix, 1, measurementNoiseCov, 1, temp3, GEMM_2_T);
281
282 // temp4 = inv(temp3)*temp2 = Kt(k)
283 solve(temp3, temp2, temp4, DECOMP_SVD);
284
285 // K(k)
286 gain = temp4.t();
287
288 // temp5 = z(k) - H*x'(k)
289 temp5 = measurement - measurementMatrix*statePre;
290
291 // x(k) = x'(k) + K(k)*temp5
292 statePost = statePre + gain*temp5;
293
294 // P(k) = P'(k) - K(k)*temp2
295 errorCovPost = errorCovPre - gain*temp2;
296
297 return statePost;
298}
