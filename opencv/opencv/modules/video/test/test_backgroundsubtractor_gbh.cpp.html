10using namespace cv;
11
12class CV_BackgroundSubtractorTest : public cvtest::BaseTest
13{
14public:
15 CV_BackgroundSubtractorTest();
16protected:
17 void run(int);
18};
19
20CV_BackgroundSubtractorTest::CV_BackgroundSubtractorTest()
21{
22}
23
24/**
25 * This test checks the following:
26 * (i) BackgroundSubtractorGMG can operate with matrices of various types and sizes
27 * (ii) Training mode returns empty fgmask
28 * (iii) End of training mode, and anomalous frame yields every pixel detected as FG
29 */
30void CV_BackgroundSubtractorTest::run(int)
31{
32 int code = cvtest::TS::OK;
33 RNG& rng = ts->get_rng();
34 int type = ((unsigned int)rng)%7; //!< pick a random type, 0 - 6, defined in types_c.h
35 int channels = 1 + ((unsigned int)rng)%4; //!< random number of channels from 1 to 4.
36 int channelsAndType = CV_MAKETYPE(type,channels);
37 int width = 2 + ((unsigned int)rng)%98; //!< Mat will be 2 to 100 in width and height
38 int height = 2 + ((unsigned int)rng)%98;
39
40 Ptr<BackgroundSubtractorGMG> fgbg =
41 Algorithm::create<BackgroundSubtractorGMG>("BackgroundSubtractor.GMG");
42 Mat fgmask;
43
44 if (fgbg == NULL)
45 CV_Error(CV_StsError,"Failed to create Algorithm\n");
46
47 /**
48 * Set a few parameters
49 */
50 fgbg->set("smoothingRadius",7);
51 fgbg->set("decisionThreshold",0.7);
52 fgbg->set("initializationFrames",120);
53
54 /**
55 * Generate bounds for the values in the matrix for each type
56 */
57 uchar maxuc = 0, minuc = 0;
58 char maxc = 0, minc = 0;
59 unsigned int maxui = 0, minui = 0;
60 int maxi=0, mini = 0;
61 long int maxli = 0, minli = 0;
62 float maxf = 0, minf = 0;
63 double maxd = 0, mind = 0;
64
65 /**
66 * Max value for simulated images picked randomly in upper half of type range
67 * Min value for simulated images picked randomly in lower half of type range
68 */
69 if (type == CV_8U)
70 {
71 uchar half = UCHAR_MAX/2;
72 maxuc = (unsigned char)rng.uniform(half+32, UCHAR_MAX);
73 minuc = (unsigned char)rng.uniform(0, half-32);
74 }
75 else if (type == CV_8S)
76 {
77 maxc = (char)rng.uniform(32, CHAR_MAX);
78 minc = (char)rng.uniform(CHAR_MIN, -32);
79 }
80 else if (type == CV_16U)
81 {
82 ushort half = USHRT_MAX/2;
83 maxui = (unsigned int)rng.uniform(half+32, USHRT_MAX);
84 minui = (unsigned int)rng.uniform(0, half-32);
85 }
86 else if (type == CV_16S)
87 {
88 maxi = rng.uniform(32, SHRT_MAX);
89 mini = rng.uniform(SHRT_MIN, -32);
90 }
91 else if (type == CV_32S)
92 {
93 maxli = rng.uniform(32, INT_MAX);
94 minli = rng.uniform(INT_MIN, -32);
95 }
96 else if (type == CV_32F)
97 {
98 maxf = rng.uniform(32.0f, FLT_MAX);
99 minf = rng.uniform(-FLT_MAX, -32.0f);
100 }
101 else if (type == CV_64F)
102 {
103 maxd = rng.uniform(32.0, DBL_MAX);
104 mind = rng.uniform(-DBL_MAX, -32.0);
105 }
106
107 Mat simImage = Mat::zeros(height, width, channelsAndType);
108 const unsigned int numLearningFrames = 120;
109 for (unsigned int i = 0; i < numLearningFrames; ++i)
110 {
111 /**
112 * Genrate simulated "image" for any type. Values always confined to upper half of range.
113 */
114 if (type == CV_8U)
115 {
116 rng.fill(simImage,RNG::UNIFORM,(unsigned char)(minuc/2+maxuc/2),maxuc);
117 if (i == 0)
118 fgbg->initialize(simImage.size(),minuc,maxuc);
119 }
120 else if (type == CV_8S)
121 {
122 rng.fill(simImage,RNG::UNIFORM,(char)(minc/2+maxc/2),maxc);
123 if (i==0)
124 fgbg->initialize(simImage.size(),minc,maxc);
125 }
126 else if (type == CV_16U)
127 {
128 rng.fill(simImage,RNG::UNIFORM,(unsigned int)(minui/2+maxui/2),maxui);
129 if (i==0)
130 fgbg->initialize(simImage.size(),minui,maxui);
131 }
132 else if (type == CV_16S)
133 {
134 rng.fill(simImage,RNG::UNIFORM,(int)(mini/2+maxi/2),maxi);
135 if (i==0)
136 fgbg->initialize(simImage.size(),mini,maxi);
137 }
138 else if (type == CV_32F)
139 {
140 rng.fill(simImage,RNG::UNIFORM,(float)(minf/2.0+maxf/2.0),maxf);
141 if (i==0)
142 fgbg->initialize(simImage.size(),minf,maxf);
143 }
144 else if (type == CV_32S)
145 {
146 rng.fill(simImage,RNG::UNIFORM,(long int)(minli/2+maxli/2),maxli);
147 if (i==0)
148 fgbg->initialize(simImage.size(),minli,maxli);
149 }
150 else if (type == CV_64F)
151 {
152 rng.fill(simImage,RNG::UNIFORM,(double)(mind/2.0+maxd/2.0),maxd);
153 if (i==0)
154 fgbg->initialize(simImage.size(),mind,maxd);
155 }
156
157 /**
158 * Feed simulated images into background subtractor
159 */
160 (*fgbg)(simImage,fgmask);
161 Mat fullbg = Mat::zeros(simImage.rows, simImage.cols, CV_8U);
162
163 //! fgmask should be entirely background during training
164 code = cvtest::cmpEps2( ts, fgmask, fullbg, 0, false, "The training foreground mask" );
165 if (code < 0)
166 ts->set_failed_test_info( code );
167 }
168 //! generate last image, distinct from training images
169 if (type == CV_8U)
170 rng.fill(simImage,RNG::UNIFORM,minuc,minuc);
171 else if (type == CV_8S)
172 rng.fill(simImage,RNG::UNIFORM,minc,minc);
173 else if (type == CV_16U)
174 rng.fill(simImage,RNG::UNIFORM,minui,minui);
175 else if (type == CV_16S)
176 rng.fill(simImage,RNG::UNIFORM,mini,mini);
177 else if (type == CV_32F)
178 rng.fill(simImage,RNG::UNIFORM,minf,minf);
179 else if (type == CV_32S)
180 rng.fill(simImage,RNG::UNIFORM,minli,minli);
181 else if (type == CV_64F)
182 rng.fill(simImage,RNG::UNIFORM,mind,mind);
183
184 (*fgbg)(simImage,fgmask);
185 //! now fgmask should be entirely foreground
186 Mat fullfg = 255*Mat::ones(simImage.rows, simImage.cols, CV_8U);
187 code = cvtest::cmpEps2( ts, fgmask, fullfg, 255, false, "The final foreground mask" );
188 if (code < 0)
189 {
190 ts->set_failed_test_info( code );
191 }
192
193}

195TEST(VIDEO_BGSUBGMG, accuracy) { CV_BackgroundSubtractorTest test; test.safe_run(); }
