73void CV_MHIBaseTest::get_minmax_bounds( int i, int j, int type, Scalar& low, Scalar& high )
74{
75 cvtest::ArrayTest::get_minmax_bounds( i, j, type, low, high );
76 if( i == INPUT && CV_MAT_DEPTH(type) == CV_8U )
77 {
78 low = Scalar::all(cvRound(-1./silh_ratio)+2.);
79 high = Scalar::all(2);
80 }
81 else if( i == mhi_i || i == mhi_ref_i )
82 {
83 low = Scalar::all(-exp(max_log_duration));
84 high = Scalar::all(0.);
85 }
86}

121///////////////////// update motion history ////////////////////////////
122
123static void test_updateMHI( const Mat& silh, Mat& mhi, double timestamp, double duration )
124{
125 int i, j;
126 float delbound = (float)(timestamp - duration);
127 for( i = 0; i < mhi.rows; i++ )
128 {
129 const uchar* silh_row = silh.ptr(i);
130 float* mhi_row = mhi.ptr<float>(i);
131
132 for( j = 0; j < mhi.cols; j++ )
133 {
134 if( silh_row[j] )
135 mhi_row[j] = (float)timestamp;
136 else if( mhi_row[j] < delbound )
137 mhi_row[j] = 0.f;
138 }
139 }
140}

185///////////////////// calc motion gradient ////////////////////////////
186
187static void test_MHIGradient( const Mat& mhi, Mat& mask, Mat& orientation,
188 double delta1, double delta2, int aperture_size )
189{
190 Point anchor( aperture_size/2, aperture_size/2 );
191 double limit = 1e-4*aperture_size*aperture_size;
192
193 Mat dx, dy, min_mhi, max_mhi;
194
195 Mat kernel = cvtest::calcSobelKernel2D( 1, 0, aperture_size );
196 cvtest::filter2D( mhi, dx, CV_32F, kernel, anchor, 0, BORDER_REPLICATE );
197 kernel = cvtest::calcSobelKernel2D( 0, 1, aperture_size );
198 cvtest::filter2D( mhi, dy, CV_32F, kernel, anchor, 0, BORDER_REPLICATE );
199
200 kernel = Mat::ones(aperture_size, aperture_size, CV_8U);
201 cvtest::erode(mhi, min_mhi, kernel, anchor, 0, BORDER_REPLICATE);
202 cvtest::dilate(mhi, max_mhi, kernel, anchor, 0, BORDER_REPLICATE);
203
204 if( delta1 > delta2 )
205 {
206 double t;
207 CV_SWAP( delta1, delta2, t );
208 }
209
210 for( int i = 0; i < mhi.rows; i++ )
211 {
212 uchar* mask_row = mask.ptr(i);
213 float* orient_row = orientation.ptr<float>(i);
214 const float* dx_row = dx.ptr<float>(i);
215 const float* dy_row = dy.ptr<float>(i);
216 const float* min_row = min_mhi.ptr<float>(i);
217 const float* max_row = max_mhi.ptr<float>(i);
218
219 for( int j = 0; j < mhi.cols; j++ )
220 {
221 double delta = max_row[j] - min_row[j];
222 double _dx = dx_row[j], _dy = dy_row[j];
223
224 if( delta1 <= delta && delta <= delta2 &&
225 (fabs(_dx) > limit || fabs(_dy) > limit) )
226 {
227 mask_row[j] = 1;
228 double angle = atan2( _dy, _dx ) * (180/CV_PI);
229 if( angle < 0 )
230 angle += 360.;
231 orient_row[j] = (float)angle;
232 }
233 else
234 {
235 mask_row[j] = 0;
236 orient_row[j] = 0.f;
237 }
238 }
239 }

312////////////////////// calc global orientation /////////////////////////
313
314static double test_calcGlobalOrientation( const Mat& orient, const Mat& mask,
315 const Mat& mhi, double timestamp, double duration )
316{
317 const int HIST_SIZE = 12;
318 int y, x;
319 int histogram[HIST_SIZE];
320 int max_bin = 0;
321
322 double base_orientation = 0, delta_orientation = 0, weight = 0;
323 double low_time, global_orientation;
324
325 memset( histogram, 0, sizeof( histogram ));
326 timestamp = 0;
327
328 for( y = 0; y < orient.rows; y++ )
329 {
330 const float* orient_data = orient.ptr<float>(y);
331 const uchar* mask_data = mask.ptr(y);
332 const float* mhi_data = mhi.ptr<float>(y);
333 for( x = 0; x < orient.cols; x++ )
334 if( mask_data[x] )
335 {
336 int bin = cvFloor( (orient_data[x]*HIST_SIZE)/360 );
337 histogram[bin < 0 ? 0 : bin >= HIST_SIZE ? HIST_SIZE-1 : bin]++;
338 if( mhi_data[x] > timestamp )
339 timestamp = mhi_data[x];
340 }
341 }
342
343 low_time = timestamp - duration;
344
345 for( x = 1; x < HIST_SIZE; x++ )
346 {
347 if( histogram[x] > histogram[max_bin] )
348 max_bin = x;
349 }
350
351 base_orientation = ((double)max_bin*360)/HIST_SIZE;
352
353 for( y = 0; y < orient.rows; y++ )
354 {
355 const float* orient_data = orient.ptr<float>(y);
356 const float* mhi_data = mhi.ptr<float>(y);
357 const uchar* mask_data = mask.ptr(y);
358
359 for( x = 0; x < orient.cols; x++ )
360 {
361 if( mask_data[x] && mhi_data[x] > low_time )
362 {
363 double diff = orient_data[x] - base_orientation;
364 double delta_weight = (((mhi_data[x] - low_time)/duration)*254 + 1)/255;
365
366 if( diff < -180 ) diff += 360;
367 if( diff > 180 ) diff -= 360;
368
369 if( delta_weight > 0 && fabs(diff) < 45 )
370 {
371 delta_orientation += diff*delta_weight;
372 weight += delta_weight;
373 }
374 }
375 }
376 }
377
378 if( weight == 0 )
379 global_orientation = base_orientation;
380 else
381 {
382 global_orientation = base_orientation + delta_orientation/weight;
383 if( global_orientation < 0 ) global_orientation += 360;
384 if( global_orientation > 360 ) global_orientation -= 360;
385 }
386
387 return global_orientation;
388}

468int CV_MHIGlobalOrientTest::validate_test_results( int test_case_idx )
469{
470 //printf("%d. rows=%d, cols=%d, nzmask=%d\n", test_case_idx, test_mat[INPUT][1].rows, test_mat[INPUT][1].cols,
471 // cvCountNonZero(test_array[INPUT][1]));
472
473 double ref_angle = test_calcGlobalOrientation( test_mat[INPUT][2], test_mat[INPUT][1],
474 test_mat[INPUT][0], timestamp, duration );
475 double err_level = get_success_error_level( test_case_idx, 0, 0 );
476 int code = cvtest::TS::OK;
477 int nz = cvCountNonZero( test_array[INPUT][1] );
478
479 if( nz > 32 && !(min_angle - err_level <= angle &&
480 max_angle + err_level >= angle) &&
481 !(min_angle - err_level <= angle+360 &&
482 max_angle + err_level >= angle+360) )
483 {
484 ts->printf( cvtest::TS::LOG, "The angle=%g is outside (%g,%g) range\n",
485 angle, min_angle - err_level, max_angle + err_level );
486 code = cvtest::TS::FAIL_BAD_ACCURACY;
487 }
488 else if( fabs(angle - ref_angle) > err_level &&
489 fabs(360 - fabs(angle - ref_angle)) > err_level )
490 {
491 ts->printf( cvtest::TS::LOG, "The angle=%g differs too much from reference value=%g\n",
492 angle, ref_angle );
493 code = cvtest::TS::FAIL_BAD_ACCURACY;
494 }
495
496 if( code < 0 )
497 ts->set_failed_test_info( code );
498 return code;
499}
