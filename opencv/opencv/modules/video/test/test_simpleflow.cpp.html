61static bool readOpticalFlowFromFile(FILE* file, cv::Mat& flow) {
62 char header[5];
63 if (fread(header, 1, 4, file) < 4 && (string)header != "PIEH") {
64 return false;
65 }
66
67 int cols, rows;
68 if (fread(&cols, sizeof(int), 1, file) != 1||
69 fread(&rows, sizeof(int), 1, file) != 1) {
70 return false;
71 }
72
73 flow = cv::Mat::zeros(rows, cols, CV_32FC2);
74
75 for (int i = 0; i < rows; ++i) {
76 for (int j = 0; j < cols; ++j) {
77 cv::Vec2f flow_at_point;
78 if (fread(&(flow_at_point[0]), sizeof(float), 1, file) != 1 ||
79 fread(&(flow_at_point[1]), sizeof(float), 1, file) != 1) {
80 return false;
81 }
82 flow.at<cv::Vec2f>(i, j) = flow_at_point;
83 }
84 }
85
86 return true;
87}
88
89static bool isFlowCorrect(float u) {
90 return !cvIsNaN(u) && (fabs(u) < 1e9);
91}
92
93static float calc_rmse(cv::Mat flow1, cv::Mat flow2) {
94 float sum = 0;
95 int counter = 0;
96 const int rows = flow1.rows;
97 const int cols = flow1.cols;
98
99 for (int y = 0; y < rows; ++y) {
100 for (int x = 0; x < cols; ++x) {
101 cv::Vec2f flow1_at_point = flow1.at<cv::Vec2f>(y, x);
102 cv::Vec2f flow2_at_point = flow2.at<cv::Vec2f>(y, x);
103
104 float u1 = flow1_at_point[0];
105 float v1 = flow1_at_point[1];
106 float u2 = flow2_at_point[0];
107 float v2 = flow2_at_point[1];
108
109 if (isFlowCorrect(u1) && isFlowCorrect(u2) && isFlowCorrect(v1) && isFlowCorrect(v2)) {
110 sum += (u1-u2)*(u1-u2) + (v1-v2)*(v1-v2);
111 counter++;
112 }
113 }
114 }
115 return (float)sqrt(sum / (1e-9 + counter));
116}

118void CV_SimpleFlowTest::run(int) {
119 const float MAX_RMSE = 0.6f;
120 const string frame1_path = ts->get_data_path() + "optflow/RubberWhale1.png";
121 const string frame2_path = ts->get_data_path() + "optflow/RubberWhale2.png";
122 const string gt_flow_path = ts->get_data_path() + "optflow/RubberWhale.flo";
123
124 cv::Mat frame1 = cv::imread(frame1_path);
125 cv::Mat frame2 = cv::imread(frame2_path);
126
127 if (frame1.empty()) {
128 ts->printf(cvtest::TS::LOG, "could not read image %s\n", frame2_path.c_str());
129 ts->set_failed_test_info(cvtest::TS::FAIL_MISSING_TEST_DATA);
130 return;
131 }
132
133 if (frame2.empty()) {
134 ts->printf(cvtest::TS::LOG, "could not read image %s\n", frame2_path.c_str());
135 ts->set_failed_test_info(cvtest::TS::FAIL_MISSING_TEST_DATA);
136 return;
137 }
138
139 if (frame1.rows != frame2.rows && frame1.cols != frame2.cols) {
140 ts->printf(cvtest::TS::LOG, "images should be of equal sizes (%s and %s)",
141 frame1_path.c_str(), frame2_path.c_str());
142 ts->set_failed_test_info(cvtest::TS::FAIL_MISSING_TEST_DATA);
143 return;
144 }
145
146 if (frame1.type() != 16 || frame2.type() != 16) {
147 ts->printf(cvtest::TS::LOG, "images should be of equal type CV_8UC3 (%s and %s)",
148 frame1_path.c_str(), frame2_path.c_str());
149 ts->set_failed_test_info(cvtest::TS::FAIL_MISSING_TEST_DATA);
150 return;
151 }
152
153 cv::Mat flow_gt;
154
155 FILE* gt_flow_file = fopen(gt_flow_path.c_str(), "rb");
156 if (gt_flow_file == NULL) {
157 ts->printf(cvtest::TS::LOG, "could not read ground-thuth flow from file %s",
158 gt_flow_path.c_str());
159 ts->set_failed_test_info(cvtest::TS::FAIL_MISSING_TEST_DATA);
160 return;
161 }
162
163 if (!readOpticalFlowFromFile(gt_flow_file, flow_gt)) {
164 ts->printf(cvtest::TS::LOG, "error while reading flow data from file %s",
165 gt_flow_path.c_str());
166 ts->set_failed_test_info(cvtest::TS::FAIL_MISSING_TEST_DATA);
167 return;
168 }
169 fclose(gt_flow_file);
170
171 cv::Mat flow;
172 cv::calcOpticalFlowSF(frame1, frame2, flow, 3, 2, 4);
173
174 float rmse = calc_rmse(flow_gt, flow);
175
176 ts->printf(cvtest::TS::LOG, "Optical flow estimation RMSE for SimpleFlow algorithm : %lf\n",
177 rmse);
178
179 if (rmse > MAX_RMSE) {
180 ts->printf( cvtest::TS::LOG,
181 "Too big rmse error : %lf ( >= %lf )\n", rmse, MAX_RMSE);
182 ts->set_failed_test_info(cvtest::TS::FAIL_BAD_ACCURACY);
183 return;
184 }
185}
