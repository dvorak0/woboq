53 // first four bytes, should be the same in little endian
54 const float FLO_TAG_FLOAT = 202021.25f; // check for this when READING the file
55#ifdef DUMP
56 const char FLO_TAG_STRING[] = "PIEH"; // use this when WRITING the file

58 // binary file format for flow data specified here:
59 // http://vision.middlebury.edu/flow/data/
60 void writeOpticalFlowToFile(const Mat_<Point2f>& flow, const string& fileName)
61 {
62 ofstream file(fileName.c_str(), ios_base::binary);
63
64 file << FLO_TAG_STRING;
65
66 file.write((const char*) &flow.cols, sizeof(int));
67 file.write((const char*) &flow.rows, sizeof(int));
68
69 for (int i = 0; i < flow.rows; ++i)
70 {
71 for (int j = 0; j < flow.cols; ++j)
72 {
73 const Point2f u = flow(i, j);
74
75 file.write((const char*) &u.x, sizeof(float));
76 file.write((const char*) &u.y, sizeof(float));
77 }
78 }
79 }
80#endif
81
82 // binary file format for flow data specified here:
83 // http://vision.middlebury.edu/flow/data/
84 void readOpticalFlowFromFile(Mat_<Point2f>& flow, const string& fileName)
85 {
86 ifstream file(fileName.c_str(), ios_base::binary);
87
88 float tag;
89 file.read((char*) &tag, sizeof(float));
90 CV_Assert( tag == FLO_TAG_FLOAT );
91
92 Size size;
93
94 file.read((char*) &size.width, sizeof(int));
95 file.read((char*) &size.height, sizeof(int));
96
97 flow.create(size);
98
99 for (int i = 0; i < flow.rows; ++i)
100 {
101 for (int j = 0; j < flow.cols; ++j)
102 {
103 Point2f u;
104
105 file.read((char*) &u.x, sizeof(float));
106 file.read((char*) &u.y, sizeof(float));
107
108 flow(i, j) = u;
109 }
110 }
111 }

113 bool isFlowCorrect(Point2f u)
114 {
115 return !cvIsNaN(u.x) && !cvIsNaN(u.y) && (fabs(u.x) < 1e9) && (fabs(u.y) < 1e9);
116 }
117
118 void check(const Mat_<Point2f>& gold, const Mat_<Point2f>& flow, double threshold = 0.1, double expectedAccuracy = 0.95)
119 {
120 threshold = threshold*threshold;
121
122 size_t gold_counter = 0;
123 size_t valid_counter = 0;
124
125 for (int i = 0; i < gold.rows; ++i)
126 {
127 for (int j = 0; j < gold.cols; ++j)
128 {
129 const Point2f u1 = gold(i, j);
130 const Point2f u2 = flow(i, j);
131
132 if (isFlowCorrect(u1))
133 {
134 gold_counter++;
135 if (isFlowCorrect(u2))
136 {
137 const Point2f diff = u1 - u2;
138 double err = diff.ddot(diff);
139 if (err <= threshold)
140 valid_counter++;
141 }
142 }
143 }
144 }
145 EXPECT_GE(valid_counter, expectedAccuracy * gold_counter);
146 }

149TEST(Video_calcOpticalFlowDual_TVL1, Regression)
150{
151 const string frame1_path = TS::ptr()->get_data_path() + "optflow/RubberWhale1.png";
152 const string frame2_path = TS::ptr()->get_data_path() + "optflow/RubberWhale2.png";
153 const string gold_flow_path = TS::ptr()->get_data_path() + "optflow/tvl1_flow.flo";
154
155 Mat frame1 = imread(frame1_path, IMREAD_GRAYSCALE);
156 Mat frame2 = imread(frame2_path, IMREAD_GRAYSCALE);
157 ASSERT_FALSE(frame1.empty());
158 ASSERT_FALSE(frame2.empty());
159
160 Mat_<Point2f> flow;
161 Ptr<DenseOpticalFlow> tvl1 = createOptFlow_DualTVL1();
162
163 tvl1->calc(frame1, frame2, flow);
164
165#ifdef DUMP
166 writeOpticalFlowToFile(flow, gold_flow_path);
167#else
168 Mat_<Point2f> gold;
169 readOpticalFlowFromFile(gold, gold_flow_path);
170
171 ASSERT_EQ(gold.rows, flow.rows);
172 ASSERT_EQ(gold.cols, flow.cols);
173
174 check(gold, flow);
175#endif
176}
