109void FREAK::buildPattern()
110{
111 if( patternScale == patternScale0 && nOctaves == nOctaves0 && !patternLookup.empty() )
112 return;
113
114 nOctaves0 = nOctaves;
115 patternScale0 = patternScale;
116
117 patternLookup.resize(FREAK_NB_SCALES*FREAK_NB_ORIENTATION*FREAK_NB_POINTS);
118 double scaleStep = pow(2.0, (double)(nOctaves)/FREAK_NB_SCALES ); // 2 ^ ( (nOctaves-1) /nbScales)
119 double scalingFactor, alpha, beta, theta = 0;
120
121 // pattern definition, radius normalized to 1.0 (outer point position+sigma=1.0)
122 const int n[8] = {6,6,6,6,6,6,6,1}; // number of points on each concentric circle (from outer to inner)
123 const double bigR(2.0/3.0); // bigger radius
124 const double smallR(2.0/24.0); // smaller radius
125 const double unitSpace( (bigR-smallR)/21.0 ); // define spaces between concentric circles (from center to outer: 1,2,3,4,5,6)
126 // radii of the concentric cirles (from outer to inner)
127 const double radius[8] = {bigR, bigR-6*unitSpace, bigR-11*unitSpace, bigR-15*unitSpace, bigR-18*unitSpace, bigR-20*unitSpace, smallR, 0.0};
128 // sigma of pattern points (each group of 6 points on a concentric cirle has the same sigma)
129 const double sigma[8] = {radius[0]/2.0, radius[1]/2.0, radius[2]/2.0,
130 radius[3]/2.0, radius[4]/2.0, radius[5]/2.0,
131 radius[6]/2.0, radius[6]/2.0
132 };
133 // fill the lookup table
134 for( int scaleIdx=0; scaleIdx < FREAK_NB_SCALES; ++scaleIdx )
135 {
136 patternSizes[scaleIdx] = 0; // proper initialization
137 scalingFactor = pow(scaleStep,scaleIdx); //scale of the pattern, scaleStep ^ scaleIdx
138
139 for( int orientationIdx = 0; orientationIdx < FREAK_NB_ORIENTATION; ++orientationIdx )
140 {
141 theta = double(orientationIdx)* 2*CV_PI/double(FREAK_NB_ORIENTATION); // orientation of the pattern
142 int pointIdx = 0;
143
144 PatternPoint* patternLookupPtr = &patternLookup[0];
145 for( size_t i = 0; i < 8; ++i )
146 {
147 for( int k = 0 ; k < n[i]; ++k )
148 {
149 beta = CV_PI/n[i] * (i%2); // orientation offset so that groups of points on each circles are staggered
150 alpha = double(k)* 2*CV_PI/double(n[i])+beta+theta;
151
152 // add the point to the look-up table
153 PatternPoint& point = patternLookupPtr[ scaleIdx*FREAK_NB_ORIENTATION*FREAK_NB_POINTS+orientationIdx*FREAK_NB_POINTS+pointIdx ];
154 point.x = static_cast<float>(radius[i] * cos(alpha) * scalingFactor * patternScale);
155 point.y = static_cast<float>(radius[i] * sin(alpha) * scalingFactor * patternScale);
156 point.sigma = static_cast<float>(sigma[i] * scalingFactor * patternScale);
157
158 // adapt the sizeList if necessary
159 const int sizeMax = static_cast<int>(ceil((radius[i]+sigma[i])*scalingFactor*patternScale)) + 1;
160 if( patternSizes[scaleIdx] < sizeMax )
161 patternSizes[scaleIdx] = sizeMax;
162
163 ++pointIdx;
164 }
165 }
166 }
167 }
168
169 // build the list of orientation pairs
170 orientationPairs[0].i=0; orientationPairs[0].j=3; orientationPairs[1].i=1; orientationPairs[1].j=4; orientationPairs[2].i=2; orientationPairs[2].j=5;
171 orientationPairs[3].i=0; orientationPairs[3].j=2; orientationPairs[4].i=1; orientationPairs[4].j=3; orientationPairs[5].i=2; orientationPairs[5].j=4;
172 orientationPairs[6].i=3; orientationPairs[6].j=5; orientationPairs[7].i=4; orientationPairs[7].j=0; orientationPairs[8].i=5; orientationPairs[8].j=1;
173
174 orientationPairs[9].i=6; orientationPairs[9].j=9; orientationPairs[10].i=7; orientationPairs[10].j=10; orientationPairs[11].i=8; orientationPairs[11].j=11;
175 orientationPairs[12].i=6; orientationPairs[12].j=8; orientationPairs[13].i=7; orientationPairs[13].j=9; orientationPairs[14].i=8; orientationPairs[14].j=10;
176 orientationPairs[15].i=9; orientationPairs[15].j=11; orientationPairs[16].i=10; orientationPairs[16].j=6; orientationPairs[17].i=11; orientationPairs[17].j=7;
177
178 orientationPairs[18].i=12; orientationPairs[18].j=15; orientationPairs[19].i=13; orientationPairs[19].j=16; orientationPairs[20].i=14; orientationPairs[20].j=17;
179 orientationPairs[21].i=12; orientationPairs[21].j=14; orientationPairs[22].i=13; orientationPairs[22].j=15; orientationPairs[23].i=14; orientationPairs[23].j=16;
180 orientationPairs[24].i=15; orientationPairs[24].j=17; orientationPairs[25].i=16; orientationPairs[25].j=12; orientationPairs[26].i=17; orientationPairs[26].j=13;
181
182 orientationPairs[27].i=18; orientationPairs[27].j=21; orientationPairs[28].i=19; orientationPairs[28].j=22; orientationPairs[29].i=20; orientationPairs[29].j=23;
183 orientationPairs[30].i=18; orientationPairs[30].j=20; orientationPairs[31].i=19; orientationPairs[31].j=21; orientationPairs[32].i=20; orientationPairs[32].j=22;
184 orientationPairs[33].i=21; orientationPairs[33].j=23; orientationPairs[34].i=22; orientationPairs[34].j=18; orientationPairs[35].i=23; orientationPairs[35].j=19;
185
186 orientationPairs[36].i=24; orientationPairs[36].j=27; orientationPairs[37].i=25; orientationPairs[37].j=28; orientationPairs[38].i=26; orientationPairs[38].j=29;
187 orientationPairs[39].i=30; orientationPairs[39].j=33; orientationPairs[40].i=31; orientationPairs[40].j=34; orientationPairs[41].i=32; orientationPairs[41].j=35;
188 orientationPairs[42].i=36; orientationPairs[42].j=39; orientationPairs[43].i=37; orientationPairs[43].j=40; orientationPairs[44].i=38; orientationPairs[44].j=41;
189
190 for( unsigned m = FREAK_NB_ORIENPAIRS; m--; )
191 {
192 const float dx = patternLookup[orientationPairs[m].i].x-patternLookup[orientationPairs[m].j].x;
193 const float dy = patternLookup[orientationPairs[m].i].y-patternLookup[orientationPairs[m].j].y;
194 const float norm_sq = (dx*dx+dy*dy);
195 orientationPairs[m].weight_dx = int((dx/(norm_sq))*4096.0+0.5);
196 orientationPairs[m].weight_dy = int((dy/(norm_sq))*4096.0+0.5);
197 }
198
199 // build the list of description pairs
200 std::vector<DescriptionPair> allPairs;
201 for( unsigned int i = 1; i < (unsigned int)FREAK_NB_POINTS; ++i )
202 {
203 // (generate all the pairs)
204 for( unsigned int j = 0; (unsigned int)j < i; ++j )
205 {
206 DescriptionPair pair = {(uchar)i,(uchar)j};
207 allPairs.push_back(pair);
208 }
209 }
210 // Input vector provided
211 if( !selectedPairs0.empty() )
212 {
213 if( (int)selectedPairs0.size() == FREAK_NB_PAIRS )
214 {
215 for( int i = 0; i < FREAK_NB_PAIRS; ++i )
216 descriptionPairs[i] = allPairs[selectedPairs0.at(i)];
217 }
218 else
219 {
220 CV_Error(CV_StsVecLengthErr, "Input vector does not match the required size");
221 }
222 }
223 else // default selected pairs
224 {
225 for( int i = 0; i < FREAK_NB_PAIRS; ++i )
226 descriptionPairs[i] = allPairs[FREAK_DEF_PAIRS[i]];
227 }
228}

230void FREAK::computeImpl( const Mat& image, std::vector<KeyPoint>& keypoints, Mat& descriptors ) const
231{
232
233 if( image.empty() )
234 return;
235 if( keypoints.empty() )
236 return;
237
238 ((FREAK*)this)->buildPattern();
239
240 Mat imgIntegral;
241 integral(image, imgIntegral);
242 std::vector<int> kpScaleIdx(keypoints.size()); // used to save pattern scale index corresponding to each keypoints
243 const std::vector<int>::iterator ScaleIdxBegin = kpScaleIdx.begin(); // used in std::vector erase function
244 const std::vector<cv::KeyPoint>::iterator kpBegin = keypoints.begin(); // used in std::vector erase function
245 const float sizeCst = static_cast<float>(FREAK_NB_SCALES/(FREAK_LOG2* nOctaves));
246 uchar pointsValue[FREAK_NB_POINTS];
247 int thetaIdx = 0;
248 int direction0;
249 int direction1;
250
251 // compute the scale index corresponding to the keypoint size and remove keypoints close to the border
252 if( scaleNormalized )
253 {
254 for( size_t k = keypoints.size(); k--; )
255 {
256 //Is k non-zero? If so, decrement it and continue"
257 kpScaleIdx[k] = max( (int)(log(keypoints[k].size/FREAK_SMALLEST_KP_SIZE)*sizeCst+0.5) ,0);
258 if( kpScaleIdx[k] >= FREAK_NB_SCALES )
259 kpScaleIdx[k] = FREAK_NB_SCALES-1;
260
261 if( keypoints[k].pt.x <= patternSizes[kpScaleIdx[k]] || //check if the description at this specific position and scale fits inside the image
262 keypoints[k].pt.y <= patternSizes[kpScaleIdx[k]] ||
263 keypoints[k].pt.x >= image.cols-patternSizes[kpScaleIdx[k]] ||
264 keypoints[k].pt.y >= image.rows-patternSizes[kpScaleIdx[k]]
265 )
266 {
267 keypoints.erase(kpBegin+k);
268 kpScaleIdx.erase(ScaleIdxBegin+k);
269 }
270 }
271 }
272 else
273 {
274 const int scIdx = max( (int)(1.0986122886681*sizeCst+0.5) ,0);
275 for( size_t k = keypoints.size(); k--; )
276 {
277 kpScaleIdx[k] = scIdx; // equivalent to the formule when the scale is normalized with a constant size of keypoints[k].size=3*SMALLEST_KP_SIZE
278 if( kpScaleIdx[k] >= FREAK_NB_SCALES )
279 {
280 kpScaleIdx[k] = FREAK_NB_SCALES-1;
281 }
282 if( keypoints[k].pt.x <= patternSizes[kpScaleIdx[k]] ||
283 keypoints[k].pt.y <= patternSizes[kpScaleIdx[k]] ||
284 keypoints[k].pt.x >= image.cols-patternSizes[kpScaleIdx[k]] ||
285 keypoints[k].pt.y >= image.rows-patternSizes[kpScaleIdx[k]]
286 )
287 {
288 keypoints.erase(kpBegin+k);
289 kpScaleIdx.erase(ScaleIdxBegin+k);
290 }
291 }
292 }
293
294 // allocate descriptor memory, estimate orientations, extract descriptors
295 if( !extAll )
296 {
297 // extract the best comparisons only
298 descriptors = cv::Mat::zeros((int)keypoints.size(), FREAK_NB_PAIRS/8, CV_8U);
299#if CV_SSE2
300 __m128i* ptr= (__m128i*) (descriptors.data+(keypoints.size()-1)*descriptors.step[0]);
301#else
302 std::bitset<FREAK_NB_PAIRS>* ptr = (std::bitset<FREAK_NB_PAIRS>*) (descriptors.data+(keypoints.size()-1)*descriptors.step[0]);
303#endif
304 for( size_t k = keypoints.size(); k--; )
305 {
306 // estimate orientation (gradient)
307 if( !orientationNormalized )
308 {
309 thetaIdx = 0; // assign 0 to all keypoints
310 keypoints[k].angle = 0.0;
311 }
312 else
313 {
314 // get the points intensity value in the un-rotated pattern
315 for( int i = FREAK_NB_POINTS; i--; )
316 {
317 pointsValue[i] = meanIntensity(image, imgIntegral, keypoints[k].pt.x,keypoints[k].pt.y, kpScaleIdx[k], 0, i);
318 }
319 direction0 = 0;
320 direction1 = 0;
321 for( int m = 45; m--; )
322 {
323 //iterate through the orientation pairs
324 const int delta = (pointsValue[ orientationPairs[m].i ]-pointsValue[ orientationPairs[m].j ]);
325 direction0 += delta*(orientationPairs[m].weight_dx)/2048;
326 direction1 += delta*(orientationPairs[m].weight_dy)/2048;
327 }
328
329 keypoints[k].angle = static_cast<float>(atan2((float)direction1,(float)direction0)*(180.0/CV_PI));//estimate orientation
330 thetaIdx = int(FREAK_NB_ORIENTATION*keypoints[k].angle*(1/360.0)+0.5);
331 if( thetaIdx < 0 )
332 thetaIdx += FREAK_NB_ORIENTATION;
333
334 if( thetaIdx >= FREAK_NB_ORIENTATION )
335 thetaIdx -= FREAK_NB_ORIENTATION;
336 }
337 // extract descriptor at the computed orientation
338 for( int i = FREAK_NB_POINTS; i--; )
339 {
340 pointsValue[i] = meanIntensity(image, imgIntegral, keypoints[k].pt.x,keypoints[k].pt.y, kpScaleIdx[k], thetaIdx, i);
341 }
342#if CV_SSE2
343 // note that comparisons order is modified in each block (but first 128 comparisons remain globally the same-->does not affect the 128,384 bits segmanted matching strategy)
344 int cnt = 0;
345 for( int n = FREAK_NB_PAIRS/128; n-- ; )
346 {
347 __m128i result128 = _mm_setzero_si128();
348 for( int m = 128/16; m--; cnt += 16 )
349 {
350 __m128i operand1 = _mm_set_epi8(
351 pointsValue[descriptionPairs[cnt+0].i],
352 pointsValue[descriptionPairs[cnt+1].i],
353 pointsValue[descriptionPairs[cnt+2].i],
354 pointsValue[descriptionPairs[cnt+3].i],
355 pointsValue[descriptionPairs[cnt+4].i],
356 pointsValue[descriptionPairs[cnt+5].i],
357 pointsValue[descriptionPairs[cnt+6].i],
358 pointsValue[descriptionPairs[cnt+7].i],
359 pointsValue[descriptionPairs[cnt+8].i],
360 pointsValue[descriptionPairs[cnt+9].i],
361 pointsValue[descriptionPairs[cnt+10].i],
362 pointsValue[descriptionPairs[cnt+11].i],
363 pointsValue[descriptionPairs[cnt+12].i],
364 pointsValue[descriptionPairs[cnt+13].i],
365 pointsValue[descriptionPairs[cnt+14].i],
366 pointsValue[descriptionPairs[cnt+15].i]);
367
368 __m128i operand2 = _mm_set_epi8(
369 pointsValue[descriptionPairs[cnt+0].j],
370 pointsValue[descriptionPairs[cnt+1].j],
371 pointsValue[descriptionPairs[cnt+2].j],
372 pointsValue[descriptionPairs[cnt+3].j],
373 pointsValue[descriptionPairs[cnt+4].j],
374 pointsValue[descriptionPairs[cnt+5].j],
375 pointsValue[descriptionPairs[cnt+6].j],
376 pointsValue[descriptionPairs[cnt+7].j],
377 pointsValue[descriptionPairs[cnt+8].j],
378 pointsValue[descriptionPairs[cnt+9].j],
379 pointsValue[descriptionPairs[cnt+10].j],
380 pointsValue[descriptionPairs[cnt+11].j],
381 pointsValue[descriptionPairs[cnt+12].j],
382 pointsValue[descriptionPairs[cnt+13].j],
383 pointsValue[descriptionPairs[cnt+14].j],
384 pointsValue[descriptionPairs[cnt+15].j]);
385
386 __m128i workReg = _mm_min_epu8(operand1, operand2); // emulated "not less than" for 8-bit UNSIGNED integers
387 workReg = _mm_cmpeq_epi8(workReg, operand2); // emulated "not less than" for 8-bit UNSIGNED integers
388
389 workReg = _mm_and_si128(_mm_set1_epi16(short(0x8080 >> m)), workReg); // merge the last 16 bits with the 128bits std::vector until full
390 result128 = _mm_or_si128(result128, workReg);
391 }
392 (*ptr) = result128;
393 ++ptr;
394 }
395 ptr -= (FREAK_NB_PAIRS/128)*2;
396#else
397 // extracting descriptor preserving the order of SSE version
398 int cnt = 0;
399 for( int n = 7; n < FREAK_NB_PAIRS; n += 128)
400 {
401 for( int m = 8; m--; )
402 {
403 int nm = n-m;
404 for(int kk = nm+15*8; kk >= nm; kk-=8, ++cnt)
405 {
406 ptr->set(kk, pointsValue[descriptionPairs[cnt].i] >= pointsValue[descriptionPairs[cnt].j]);
407 }
408 }
409 }
410 --ptr;
411#endif
412 }
413 }
414 else // extract all possible comparisons for selection
415 {
416 descriptors = cv::Mat::zeros((int)keypoints.size(), 128, CV_8U);
417 std::bitset<1024>* ptr = (std::bitset<1024>*) (descriptors.data+(keypoints.size()-1)*descriptors.step[0]);
418
419 for( size_t k = keypoints.size(); k--; )
420 {
421 //estimate orientation (gradient)
422 if( !orientationNormalized )
423 {
424 thetaIdx = 0;//assign 0 to all keypoints
425 keypoints[k].angle = 0.0;
426 }
427 else
428 {
429 //get the points intensity value in the un-rotated pattern
430 for( int i = FREAK_NB_POINTS;i--; )
431 pointsValue[i] = meanIntensity(image, imgIntegral, keypoints[k].pt.x,keypoints[k].pt.y, kpScaleIdx[k], 0, i);
432
433 direction0 = 0;
434 direction1 = 0;
435 for( int m = 45; m--; )
436 {
437 //iterate through the orientation pairs
438 const int delta = (pointsValue[ orientationPairs[m].i ]-pointsValue[ orientationPairs[m].j ]);
439 direction0 += delta*(orientationPairs[m].weight_dx)/2048;
440 direction1 += delta*(orientationPairs[m].weight_dy)/2048;
441 }
442
443 keypoints[k].angle = static_cast<float>(atan2((float)direction1,(float)direction0)*(180.0/CV_PI)); //estimate orientation
444 thetaIdx = int(FREAK_NB_ORIENTATION*keypoints[k].angle*(1/360.0)+0.5);
445
446 if( thetaIdx < 0 )
447 thetaIdx += FREAK_NB_ORIENTATION;
448
449 if( thetaIdx >= FREAK_NB_ORIENTATION )
450 thetaIdx -= FREAK_NB_ORIENTATION;
451 }
452 // get the points intensity value in the rotated pattern
453 for( int i = FREAK_NB_POINTS; i--; )
454 {
455 pointsValue[i] = meanIntensity(image, imgIntegral, keypoints[k].pt.x,
456 keypoints[k].pt.y, kpScaleIdx[k], thetaIdx, i);
457 }
458
459 int cnt(0);
460 for( int i = 1; i < FREAK_NB_POINTS; ++i )
461 {
462 //(generate all the pairs)
463 for( int j = 0; j < i; ++j )
464 {
465 ptr->set(cnt, pointsValue[i] >= pointsValue[j] );
466 ++cnt;
467 }
468 }
469 --ptr;
470 }
471 }
472}

474// simply take average on a square patch, not even gaussian approx
475uchar FREAK::meanIntensity( const cv::Mat& image, const cv::Mat& integral,
476 const float kp_x,
477 const float kp_y,
478 const unsigned int scale,
479 const unsigned int rot,
480 const unsigned int point) const
481{
482 // get point position in image
483 const PatternPoint& FreakPoint = patternLookup[scale*FREAK_NB_ORIENTATION*FREAK_NB_POINTS + rot*FREAK_NB_POINTS + point];
484 const float xf = FreakPoint.x+kp_x;
485 const float yf = FreakPoint.y+kp_y;
486 const int x = int(xf);
487 const int y = int(yf);
488 const int& imagecols = image.cols;
489
490 // get the sigma:
491 const float radius = FreakPoint.sigma;
492
493 // calculate output:
494 if( radius < 0.5 )
495 {
496 // interpolation multipliers:
497 const int r_x = static_cast<int>((xf-x)*1024);
498 const int r_y = static_cast<int>((yf-y)*1024);
499 const int r_x_1 = (1024-r_x);
500 const int r_y_1 = (1024-r_y);
501 uchar* ptr = image.data+x+y*imagecols;
502 unsigned int ret_val;
503 // linear interpolation:
504 ret_val = (r_x_1*r_y_1*int(*ptr));
505 ptr++;
506 ret_val += (r_x*r_y_1*int(*ptr));
507 ptr += imagecols;
508 ret_val += (r_x*r_y*int(*ptr));
509 ptr--;
510 ret_val += (r_x_1*r_y*int(*ptr));
511 //return the rounded mean
512 ret_val += 2 * 1024 * 1024;
513 return static_cast<uchar>(ret_val / (4 * 1024 * 1024));
514 }
515
516 // expected case:
517
518 // calculate borders
519 const int x_left = int(xf-radius+0.5);
520 const int y_top = int(yf-radius+0.5);
521 const int x_right = int(xf+radius+1.5);//integral image is 1px wider
522 const int y_bottom = int(yf+radius+1.5);//integral image is 1px higher
523 int ret_val;
524
525 ret_val = integral.at<int>(y_bottom,x_right);//bottom right corner
526 ret_val -= integral.at<int>(y_bottom,x_left);
527 ret_val += integral.at<int>(y_top,x_left);
528 ret_val -= integral.at<int>(y_top,x_right);
529 ret_val = ret_val/( (x_right-x_left)* (y_bottom-y_top) );
530 //~ std::cout<<integral.step[1]<<std::endl;
531 return static_cast<uchar>(ret_val);
532}

534// pair selection algorithm from a set of training images and corresponding keypoints
535vector<int> FREAK::selectPairs(const std::vector<Mat>& images
536 , std::vector<std::vector<KeyPoint> >& keypoints
537 , const double corrTresh
538 , bool verbose )
539{
540 extAll = true;
541 // compute descriptors with all pairs
542 Mat descriptors;
543
544 if( verbose )
545 std::cout << "Number of images: " << images.size() << std::endl;
546
547 for( size_t i = 0;i < images.size(); ++i )
548 {
549 Mat descriptorsTmp;
550 computeImpl(images[i],keypoints[i],descriptorsTmp);
551 descriptors.push_back(descriptorsTmp);
552 }
553
554 if( verbose )
555 std::cout << "number of keypoints: " << descriptors.rows << std::endl;
556
557 //descriptor in floating point format (each bit is a float)
558 Mat descriptorsFloat = Mat::zeros(descriptors.rows, 903, CV_32F);
559
560 std::bitset<1024>* ptr = (std::bitset<1024>*) (descriptors.data+(descriptors.rows-1)*descriptors.step[0]);
561 for( int m = descriptors.rows; m--; )
562 {
563 for( int n = 903; n--; )
564 {
565 if( ptr->test(n) == true )
566 descriptorsFloat.at<float>(m,n)=1.0f;
567 }
568 --ptr;
569 }
570
571 std::vector<PairStat> pairStat;
572 for( int n = 903; n--; )
573 {
574 // the higher the variance, the better --> mean = 0.5
575 PairStat tmp = { fabs( mean(descriptorsFloat.col(n))[0]-0.5 ) ,n};
576 pairStat.push_back(tmp);
577 }
578
579 std::sort( pairStat.begin(),pairStat.end(), sortMean() );
580
581 std::vector<PairStat> bestPairs;
582 for( int m = 0; m < 903; ++m )
583 {
584 if( verbose )
585 std::cout << m << ":" << bestPairs.size() << " " << std::flush;
586 double corrMax(0);
587
588 for( size_t n = 0; n < bestPairs.size(); ++n )
589 {
590 int idxA = bestPairs[n].idx;
591 int idxB = pairStat[m].idx;
592 double corr(0);
593 // compute correlation between 2 pairs
594 corr = fabs(compareHist(descriptorsFloat.col(idxA), descriptorsFloat.col(idxB), CV_COMP_CORREL));
595
596 if( corr > corrMax )
597 {
598 corrMax = corr;
599 if( corrMax >= corrTresh )
600 break;
601 }
602 }
603
604 if( corrMax < corrTresh/*0.7*/ )
605 bestPairs.push_back(pairStat[m]);
606
607 if( bestPairs.size() >= 512 )
608 {
609 if( verbose )
610 std::cout << m << std::endl;
611 break;
612 }
613 }
614
615 std::vector<int> idxBestPairs;
616 if( (int)bestPairs.size() >= FREAK_NB_PAIRS )
617 {
618 for( int i = 0; i < FREAK_NB_PAIRS; ++i )
619 idxBestPairs.push_back(bestPairs[i].idx);
620 }
621 else
622 {
623 if( verbose )
624 std::cout << "correlation threshold too small (restrictive)" << std::endl;
625 CV_Error(CV_StsError, "correlation threshold too small (restrictive)");
626 }
627 extAll = false;
628 return idxBestPairs;
629}
