52Mat windowedMatchingMask( const vector<KeyPoint>& keypoints1, const vector<KeyPoint>& keypoints2,
53 float maxDeltaX, float maxDeltaY )
54{
55 if( keypoints1.empty() || keypoints2.empty() )
56 return Mat();
57
58 int n1 = (int)keypoints1.size(), n2 = (int)keypoints2.size();
59 Mat mask( n1, n2, CV_8UC1 );
60 for( int i = 0; i < n1; i++ )
61 {
62 for( int j = 0; j < n2; j++ )
63 {
64 Point2f diff = keypoints2[j].pt - keypoints1[i].pt;
65 mask.at<uchar>(i, j) = std::abs(diff.x) < maxDeltaX && std::abs(diff.y) < maxDeltaY;
66 }
67 }
68 return mask;
69}

86void DescriptorMatcher::DescriptorCollection::set( const vector<Mat>& descriptors )
87{
88 clear();
89
90 size_t imageCount = descriptors.size();
91 CV_Assert( imageCount > 0 );
92
93 startIdxs.resize( imageCount );
94
95 int dim = -1;
96 int type = -1;
97 startIdxs[0] = 0;
98 for( size_t i = 1; i < imageCount; i++ )
99 {
100 int s = 0;
101 if( !descriptors[i-1].empty() )
102 {
103 dim = descriptors[i-1].cols;
104 type = descriptors[i-1].type();
105 s = descriptors[i-1].rows;
106 }
107 startIdxs[i] = startIdxs[i-1] + s;
108 }
109 if( imageCount == 1 )
110 {
111 if( descriptors[0].empty() ) return;
112
113 dim = descriptors[0].cols;
114 type = descriptors[0].type();
115 }
116 assert( dim > 0 );
117
118 int count = startIdxs[imageCount-1] + descriptors[imageCount-1].rows;
119
120 if( count > 0 )
121 {
122 mergedDescriptors.create( count, dim, type );
123 for( size_t i = 0; i < imageCount; i++ )
124 {
125 if( !descriptors[i].empty() )
126 {
127 CV_Assert( descriptors[i].cols == dim && descriptors[i].type() == type );
128 Mat m = mergedDescriptors.rowRange( startIdxs[i], startIdxs[i] + descriptors[i].rows );
129 descriptors[i].copyTo(m);
130 }
131 }
132 }
133}

308 size_t outCount = 0;

340void BFMatcher::knnMatchImpl( const Mat& queryDescriptors, vector<vector<DMatch> >& matches, int knn,
341 const vector<Mat>& masks, bool compactResult )
342{
343 const int IMGIDX_SHIFT = 18;
344 const int IMGIDX_ONE = (1 << IMGIDX_SHIFT);
345
346 if( queryDescriptors.empty() || trainDescCollection.empty() )
347 {
348 matches.clear();
349 return;
350 }
351 CV_Assert( queryDescriptors.type() == trainDescCollection[0].type() );
352
353 matches.reserve(queryDescriptors.rows);
354
355 int iIdx, imgCount = (int)trainDescCollection.size(), update = 0;
356 int dtype = normType == NORM_HAMMING || normType == NORM_HAMMING2 ||
357 (normType == NORM_L1 && queryDescriptors.type() == CV_8U) ? CV_32S : CV_32F;
358 int maxRows = 0;
359
360 CV_Assert( (int64)imgCount*IMGIDX_ONE < INT_MAX );
361
362 for( iIdx = 0; iIdx < imgCount; iIdx++ )
363 maxRows = std::max(maxRows, trainDescCollection[iIdx].rows);
364
365 int m = queryDescriptors.rows;
366 Mat dist(m, knn, dtype), nidx(m, knn, CV_32S);
367 dist = Scalar::all(dtype == CV_32S ? (double)INT_MAX : (double)FLT_MAX);
368 nidx = Scalar::all(-1);
369
370 for( iIdx = 0; iIdx < imgCount; iIdx++ )
371 {
372 CV_Assert( trainDescCollection[iIdx].rows < IMGIDX_ONE );
373 int n = std::min(knn, trainDescCollection[iIdx].rows);
374 Mat dist_i = dist.colRange(0, n), nidx_i = nidx.colRange(0, n);
375 batchDistance(queryDescriptors, trainDescCollection[iIdx], dist_i, dtype, nidx_i,
376 normType, knn, masks.empty() ? Mat() : masks[iIdx], update, crossCheck);
377 update += IMGIDX_ONE;
378 }
379
380 if( dtype == CV_32S )
381 {
382 Mat temp;
383 dist.convertTo(temp, CV_32F);
384 dist = temp;
385 }
386
387 for( int qIdx = 0; qIdx < queryDescriptors.rows; qIdx++ )
388 {
389 const float* distptr = dist.ptr<float>(qIdx);
390 const int* nidxptr = nidx.ptr<int>(qIdx);
391
392 matches.push_back( vector<DMatch>() );
393 vector<DMatch>& mq = matches.back();
394 mq.reserve(knn);
395
396 for( int k = 0; k < nidx.cols; k++ )
397 {
398 if( nidxptr[k] < 0 )
399 break;
400 mq.push_back( DMatch(qIdx, nidxptr[k] & (IMGIDX_ONE - 1),
401 nidxptr[k] >> IMGIDX_SHIFT, distptr[k]) );
402 }
403
404 if( mq.empty() && compactResult )
405 matches.pop_back();
406 }
407}

409
410void BFMatcher::radiusMatchImpl( const Mat& queryDescriptors, vector<vector<DMatch> >& matches,
411 float maxDistance, const vector<Mat>& masks, bool compactResult )
412{
413 if( queryDescriptors.empty() || trainDescCollection.empty() )
414 {
415 matches.clear();
416 return;
417 }
418 CV_Assert( queryDescriptors.type() == trainDescCollection[0].type() );
419
420 matches.resize(queryDescriptors.rows);
421 Mat dist, distf;
422
423 int iIdx, imgCount = (int)trainDescCollection.size();
424 int dtype = normType == NORM_HAMMING ||
425 (normType == NORM_L1 && queryDescriptors.type() == CV_8U) ? CV_32S : CV_32F;
426
427 for( iIdx = 0; iIdx < imgCount; iIdx++ )
428 {
429 batchDistance(queryDescriptors, trainDescCollection[iIdx], dist, dtype, noArray(),
430 normType, 0, masks.empty() ? Mat() : masks[iIdx], 0, false);
431 if( dtype == CV_32S )
432 dist.convertTo(distf, CV_32F);
433 else
434 distf = dist;
435
436 for( int qIdx = 0; qIdx < queryDescriptors.rows; qIdx++ )
437 {
438 const float* distptr = distf.ptr<float>(qIdx);
439
440 vector<DMatch>& mq = matches[qIdx];
441 for( int k = 0; k < distf.cols; k++ )
442 {
443 if( distptr[k] <= maxDistance )
444 mq.push_back( DMatch(qIdx, k, iIdx, distptr[k]) );
445 }
446 }
447 }
448
449 int qIdx0 = 0;
450 for( int qIdx = 0; qIdx < queryDescriptors.rows; qIdx++ )
451 {
452 if( matches[qIdx].empty() && compactResult )
453 continue;
454
455 if( qIdx0 < qIdx )
456 std::swap(matches[qIdx], matches[qIdx0]);
457
458 std::sort( matches[qIdx0].begin(), matches[qIdx0].end() );
459 qIdx0++;
460 }
461}

465/*
466 * Factory function for DescriptorMatcher creating
467 */
468Ptr<DescriptorMatcher> DescriptorMatcher::create( const string& descriptorMatcherType )
469{
470 DescriptorMatcher* dm = 0;
471 if( !descriptorMatcherType.compare( "FlannBased" ) )
472 {
473 dm = new FlannBasedMatcher();
474 }
475 else if( !descriptorMatcherType.compare( "BruteForce" ) ) // L2
476 {
477 dm = new BFMatcher(NORM_L2);
478 }
479 else if( !descriptorMatcherType.compare( "BruteForce-SL2" ) ) // Squared L2
480 {
481 dm = new BFMatcher(NORM_L2SQR);
482 }
483 else if( !descriptorMatcherType.compare( "BruteForce-L1" ) )
484 {
485 dm = new BFMatcher(NORM_L1);
486 }
487 else if( !descriptorMatcherType.compare("BruteForce-Hamming") ||
488 !descriptorMatcherType.compare("BruteForce-HammingLUT") )
489 {
490 dm = new BFMatcher(NORM_HAMMING);
491 }
492 else if( !descriptorMatcherType.compare("BruteForce-Hamming(2)") )
493 {
494 dm = new BFMatcher(NORM_HAMMING2);
495 }
496 else
497 CV_Error( CV_StsBadArg, "Unknown matcher name" );
498
499 return dm;
500}

532void FlannBasedMatcher::train()
533{
534 int trained = mergedDescriptors.size();
535 if (flannIndex.empty() || trained < addedDescCount)
536 {
537 mergedDescriptors.set( trainDescCollection );
538
539 // construct flannIndex class, if empty or Algorithm not equal FLANN_INDEX_LSH
540 if (flannIndex.empty() || flannIndex->getAlgorithm() != cvflann::FLANN_INDEX_LSH)
541 {
542 flannIndex = new flann::Index(mergedDescriptors.getDescriptors(), *indexParams);
543 }
544 else
545 {
546 flannIndex->build(mergedDescriptors.getDescriptors(), mergedDescriptors.getDescriptors().rowRange(trained, mergedDescriptors.size()), *indexParams, cvflann::FLANN_DIST_HAMMING);
547 }
548 }
549}

551void FlannBasedMatcher::read( const FileNode& fn)
552{
553 if (indexParams == 0)
554 indexParams = new flann::IndexParams();
555
556 FileNode ip = fn["indexParams"];
557 CV_Assert(ip.type() == FileNode::SEQ);
558
559 for(int i = 0; i < (int)ip.size(); ++i)
560 {
561 CV_Assert(ip[i].type() == FileNode::MAP);
562 std::string _name = (std::string)ip[i]["name"];
563 int type = (int)ip[i]["type"];
564
565 switch(type)
566 {
567 case CV_8U:
568 case CV_8S:
569 case CV_16U:
570 case CV_16S:
571 case CV_32S:
572 indexParams->setInt(_name, (int) ip[i]["value"]);
573 break;
574 case CV_32F:
575 indexParams->setFloat(_name, (float) ip[i]["value"]);
576 break;
577 case CV_64F:
578 indexParams->setDouble(_name, (double) ip[i]["value"]);
579 break;
580 case CV_USRTYPE1:
581 indexParams->setString(_name, (std::string) ip[i]["value"]);
582 break;
583 case CV_MAKETYPE(CV_USRTYPE1,2):
584 indexParams->setBool(_name, (int) ip[i]["value"] != 0);
585 break;
586 case CV_MAKETYPE(CV_USRTYPE1,3):
587 indexParams->setAlgorithm((int) ip[i]["value"]);
588 break;
589 };
590 }
591
592 if (searchParams == 0)
593 searchParams = new flann::SearchParams();
594
595 FileNode sp = fn["searchParams"];
596 CV_Assert(sp.type() == FileNode::SEQ);
597
598 for(int i = 0; i < (int)sp.size(); ++i)
599 {
600 CV_Assert(sp[i].type() == FileNode::MAP);
601 std::string _name = (std::string)sp[i]["name"];
602 int type = (int)sp[i]["type"];
603
604 switch(type)
605 {
606 case CV_8U:
607 case CV_8S:
608 case CV_16U:
609 case CV_16S:
610 case CV_32S:
611 searchParams->setInt(_name, (int) sp[i]["value"]);
612 break;
613 case CV_32F:
614 searchParams->setFloat(_name, (float) ip[i]["value"]);
615 break;
616 case CV_64F:
617 searchParams->setDouble(_name, (double) ip[i]["value"]);
618 break;
619 case CV_USRTYPE1:
620 searchParams->setString(_name, (std::string) ip[i]["value"]);
621 break;
622 case CV_MAKETYPE(CV_USRTYPE1,2):
623 searchParams->setBool(_name, (int) ip[i]["value"] != 0);
624 break;
625 case CV_MAKETYPE(CV_USRTYPE1,3):
626 searchParams->setAlgorithm((int) ip[i]["value"]);
627 break;
628 };
629 }
630
631 flannIndex.release();
632}

747 FlannBasedMatcher* matcher = new FlannBasedMatcher(indexParams, searchParams);
748 if( !emptyTrainData )
749 {

752 //matcher->flannIndex;

761void FlannBasedMatcher::convertToDMatches( const DescriptorCollection& collection, const Mat& indices, const Mat& dists,
762 vector<vector<DMatch> >& matches )
763{
764 matches.resize( indices.rows );
765 for( int i = 0; i < indices.rows; i++ )
766 {
767 for( int j = 0; j < indices.cols; j++ )
768 {
769 int idx = indices.at<int>(i, j);
770 if( idx >= 0 )
771 {
772 int imgIdx, trainIdx;
773 collection.getLocalIdx( idx, imgIdx, trainIdx );
774 float dist = 0;
775 if (dists.type() == CV_32S)
776 dist = static_cast<float>( dists.at<int>(i,j) );
777 else
778 dist = std::sqrt(dists.at<float>(i,j));
779 matches[i].push_back( DMatch( i, trainIdx, imgIdx, dist ) );
780 }
781 }
782 }
783}

785void FlannBasedMatcher::knnMatchImpl( const Mat& queryDescriptors, vector<vector<DMatch> >& matches, int knn,
786 const vector<Mat>& /*masks*/, bool /*compactResult*/ )
787{
788 Mat indices( queryDescriptors.rows, knn, CV_32SC1 );
789 Mat dists( queryDescriptors.rows, knn, CV_32FC1);
790 flannIndex->knnSearch( queryDescriptors, indices, dists, knn, *searchParams );
791
792 convertToDMatches( mergedDescriptors, indices, dists, matches );
793}
794
795void FlannBasedMatcher::radiusMatchImpl( const Mat& queryDescriptors, vector<vector<DMatch> >& matches, float maxDistance,
796 const vector<Mat>& /*masks*/, bool /*compactResult*/ )
797{
798 const int count = mergedDescriptors.size(); // TODO do count as param?
799 Mat indices( queryDescriptors.rows, count, CV_32SC1, Scalar::all(-1) );
800 Mat dists( queryDescriptors.rows, count, CV_32FC1, Scalar::all(-1) );
801 for( int qIdx = 0; qIdx < queryDescriptors.rows; qIdx++ )
802 {
803 Mat queryDescriptorsRow = queryDescriptors.row(qIdx);
804 Mat indicesRow = indices.row(qIdx);
805 Mat distsRow = dists.row(qIdx);
806 flannIndex->radiusSearch( queryDescriptorsRow, indicesRow, distsRow, maxDistance*maxDistance, count, *searchParams );
807 }
808
809 convertToDMatches( mergedDescriptors, indices, dists, matches );
810}

839

841 keypoints.insert( keypoints.end(), _points.begin(), _points.end() );
842 for( size_t i = 0; i < _points.size(); i++ )

844
845 size_t prevSize = startIndices.size(), addSize = _images.size();
846 startIndices.resize( prevSize + addSize );

848 if( prevSize == 0 )
849 startIndices[prevSize] = 0; //first
850 else
851 startIndices[prevSize] = (int)(startIndices[prevSize-1] + keypoints[prevSize-1].size());
852
853 for( size_t i = prevSize + 1; i < prevSize + addSize; i++ )
854 {
855 startIndices[i] = (int)(startIndices[i - 1] + keypoints[i - 1].size());
856 }

903void GenericDescriptorMatcher::KeyPointCollection::getLocalIdx( int globalPointIdx, int& imgIdx, int& localPointIdx ) const
904{
905 imgIdx = -1;
906 CV_Assert( globalPointIdx < (int)keypointCount() );
