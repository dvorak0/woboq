47/**
48 * Function that computes the Harris responses in a
49 * blockSize x blockSize patch at given points in an image
50 */
51static void
52HarrisResponses(const Mat& img, vector<KeyPoint>& pts, int blockSize, float harris_k)
53{
54 CV_Assert( img.type() == CV_8UC1 && blockSize*blockSize <= 2048 );
55
56 size_t ptidx, ptsize = pts.size();
57
58 const uchar* ptr00 = img.ptr<uchar>();
59 int step = (int)(img.step/img.elemSize1());
60 int r = blockSize/2;
61
62 float scale = (1 << 2) * blockSize * 255.0f;
63 scale = 1.0f / scale;
64 float scale_sq_sq = scale * scale * scale * scale;
65
66 AutoBuffer<int> ofsbuf(blockSize*blockSize);
67 int* ofs = ofsbuf;
68 for( int i = 0; i < blockSize; i++ )
69 for( int j = 0; j < blockSize; j++ )
70 ofs[i*blockSize + j] = (int)(i*step + j);
71
72 for( ptidx = 0; ptidx < ptsize; ptidx++ )
73 {
74 int x0 = cvRound(pts[ptidx].pt.x - r);
75 int y0 = cvRound(pts[ptidx].pt.y - r);
76
77 const uchar* ptr0 = ptr00 + y0*step + x0;
78 int a = 0, b = 0, c = 0;
79
80 for( int k = 0; k < blockSize*blockSize; k++ )
81 {
82 const uchar* ptr = ptr0 + ofs[k];
83 int Ix = (ptr[1] - ptr[-1])*2 + (ptr[-step+1] - ptr[-step-1]) + (ptr[step+1] - ptr[step-1]);
84 int Iy = (ptr[step] - ptr[-step])*2 + (ptr[step-1] - ptr[-step-1]) + (ptr[step+1] - ptr[-step+1]);
85 a += Ix*Ix;
86 b += Iy*Iy;
87 c += Ix*Iy;
88 }
89 pts[ptidx].response = ((float)a * b - (float)c * c -
90 harris_k * ((float)a + b) * ((float)a + b))*scale_sq_sq;
91 }
92}

96static float IC_Angle(const Mat& image, const int half_k, Point2f pt,
97 const vector<int> & u_max)
98{
99 int m_01 = 0, m_10 = 0;
100
101 const uchar* center = &image.at<uchar> (cvRound(pt.y), cvRound(pt.x));
102
103 // Treat the center line differently, v=0
104 for (int u = -half_k; u <= half_k; ++u)
105 m_10 += u * center[u];
106
107 // Go line by line in the circular patch
108 int step = (int)image.step1();
109 for (int v = 1; v <= half_k; ++v)
110 {
111 // Proceed over the two lines
112 int v_sum = 0;
113 int d = u_max[v];
114 for (int u = -d; u <= d; ++u)
115 {
116 int val_plus = center[u + v*step], val_minus = center[u - v*step];
117 v_sum += (val_plus - val_minus);
118 m_10 += u * (val_plus + val_minus);
119 }
120 m_01 += v * v_sum;
121 }
122
123 return fastAtan2((float)m_01, (float)m_10);
124}

128static void computeOrbDescriptor(const KeyPoint& kpt,
129 const Mat& img, const Point* pattern,
130 uchar* desc, int dsize, int WTA_K)
131{
132 float angle = kpt.angle;
133 //angle = cvFloor(angle/12)*12.f;
134 angle *= (float)(CV_PI/180.f);
135 float a = (float)cos(angle), b = (float)sin(angle);
136
137 const uchar* center = &img.at<uchar>(cvRound(kpt.pt.y), cvRound(kpt.pt.x));
138 int step = (int)img.step;
139
140 float x, y;
141 int ix, iy;
142#if 1
143 #define GET_VALUE(idx) \
144 (x = pattern[idx].x*a - pattern[idx].y*b, \
145 y = pattern[idx].x*b + pattern[idx].y*a, \
146 ix = cvRound(x), \
147 iy = cvRound(y), \
148 *(center + iy*step + ix) )
149#else
150 #define GET_VALUE(idx) \
151 (x = pattern[idx].x*a - pattern[idx].y*b, \
152 y = pattern[idx].x*b + pattern[idx].y*a, \
153 ix = cvFloor(x), iy = cvFloor(y), \
154 x -= ix, y -= iy, \
155 cvRound(center[iy*step + ix]*(1-x)*(1-y) + center[(iy+1)*step + ix]*(1-x)*y + \
156 center[iy*step + ix+1]*x*(1-y) + center[(iy+1)*step + ix+1]*x*y))
157#endif
158
159 if( WTA_K == 2 )
160 {
161 for (int i = 0; i < dsize; ++i, pattern += 16)
162 {
163 int t0, t1, val;
164 t0 = GET_VALUE(0); t1 = GET_VALUE(1);
165 val = t0 < t1;
166 t0 = GET_VALUE(2); t1 = GET_VALUE(3);
167 val |= (t0 < t1) << 1;
168 t0 = GET_VALUE(4); t1 = GET_VALUE(5);
169 val |= (t0 < t1) << 2;
170 t0 = GET_VALUE(6); t1 = GET_VALUE(7);
171 val |= (t0 < t1) << 3;
172 t0 = GET_VALUE(8); t1 = GET_VALUE(9);
173 val |= (t0 < t1) << 4;
174 t0 = GET_VALUE(10); t1 = GET_VALUE(11);
175 val |= (t0 < t1) << 5;
176 t0 = GET_VALUE(12); t1 = GET_VALUE(13);
177 val |= (t0 < t1) << 6;
178 t0 = GET_VALUE(14); t1 = GET_VALUE(15);
179 val |= (t0 < t1) << 7;
180
181 desc[i] = (uchar)val;
182 }
183 }
184 else if( WTA_K == 3 )
185 {
186 for (int i = 0; i < dsize; ++i, pattern += 12)
187 {
188 int t0, t1, t2, val;
189 t0 = GET_VALUE(0); t1 = GET_VALUE(1); t2 = GET_VALUE(2);
190 val = t2 > t1 ? (t2 > t0 ? 2 : 0) : (t1 > t0);
191
192 t0 = GET_VALUE(3); t1 = GET_VALUE(4); t2 = GET_VALUE(5);
193 val |= (t2 > t1 ? (t2 > t0 ? 2 : 0) : (t1 > t0)) << 2;
194
195 t0 = GET_VALUE(6); t1 = GET_VALUE(7); t2 = GET_VALUE(8);
196 val |= (t2 > t1 ? (t2 > t0 ? 2 : 0) : (t1 > t0)) << 4;
197
198 t0 = GET_VALUE(9); t1 = GET_VALUE(10); t2 = GET_VALUE(11);
199 val |= (t2 > t1 ? (t2 > t0 ? 2 : 0) : (t1 > t0)) << 6;
200
201 desc[i] = (uchar)val;
202 }
203 }
204 else if( WTA_K == 4 )
205 {
206 for (int i = 0; i < dsize; ++i, pattern += 16)
207 {
208 int t0, t1, t2, t3, u, v, k, val;
209 t0 = GET_VALUE(0); t1 = GET_VALUE(1);
210 t2 = GET_VALUE(2); t3 = GET_VALUE(3);
211 u = 0, v = 2;
212 if( t1 > t0 ) t0 = t1, u = 1;
213 if( t3 > t2 ) t2 = t3, v = 3;
214 k = t0 > t2 ? u : v;
215 val = k;
216
217 t0 = GET_VALUE(4); t1 = GET_VALUE(5);
218 t2 = GET_VALUE(6); t3 = GET_VALUE(7);
219 u = 0, v = 2;
220 if( t1 > t0 ) t0 = t1, u = 1;
221 if( t3 > t2 ) t2 = t3, v = 3;
222 k = t0 > t2 ? u : v;
223 val |= k << 2;
224
225 t0 = GET_VALUE(8); t1 = GET_VALUE(9);
226 t2 = GET_VALUE(10); t3 = GET_VALUE(11);
227 u = 0, v = 2;
228 if( t1 > t0 ) t0 = t1, u = 1;
229 if( t3 > t2 ) t2 = t3, v = 3;
230 k = t0 > t2 ? u : v;
231 val |= k << 4;
232
233 t0 = GET_VALUE(12); t1 = GET_VALUE(13);
234 t2 = GET_VALUE(14); t3 = GET_VALUE(15);
235 u = 0, v = 2;
236 if( t1 > t0 ) t0 = t1, u = 1;
237 if( t3 > t2 ) t2 = t3, v = 3;
238 k = t0 > t2 ? u : v;
239 val |= k << 6;
240
241 desc[i] = (uchar)val;
242 }
243 }
244 else
245 CV_Error( CV_StsBadSize, "Wrong WTA_K. It can be only 2, 3 or 4." );
246
247 #undef GET_VALUE
248}

251static void initializeOrbPattern( const Point* pattern0, vector<Point>& pattern, int ntuples, int tupleSize, int poolSize )
252{
253 RNG rng(0x12345678);
254 int i, k, k1;
255 pattern.resize(ntuples*tupleSize);
256
257 for( i = 0; i < ntuples; i++ )
258 {
259 for( k = 0; k < tupleSize; k++ )
260 {
261 for(;;)
262 {
263 int idx = rng.uniform(0, poolSize);
264 Point pt = pattern0[idx];
265 for( k1 = 0; k1 < k; k1++ )
266 if( pattern[tupleSize*i + k1] == pt )
267 break;
268 if( k1 == k )
269 {
270 pattern[tupleSize*i + k] = pt;
271 break;
272 }
273 }
274 }
275 }
276}

540{

542 // to make patterns the same on each run

544 {

546 pattern[i].y = rng.uniform(-patchSize/2, patchSize/2+1);
547 }

588/** Compute the ORB keypoint orientations
589 * @param image the image to compute the features and descriptors on
590 * @param integral_image the integral image of the iamge (can be empty, but the computation will be slower)
591 * @param scale the scale at which we compute the orientation
592 * @param keypoints the resulting keypoints
593 */
594static void computeOrientation(const Mat& image, vector<KeyPoint>& keypoints,
595 int halfPatchSize, const vector<int>& umax)
596{
597 // Process each keypoint
598 for (vector<KeyPoint>::iterator keypoint = keypoints.begin(),
599 keypointEnd = keypoints.end(); keypoint != keypointEnd; ++keypoint)
600 {
601 keypoint->angle = IC_Angle(image, halfPatchSize, keypoint->pt, umax);
602 }
603}

606/** Compute the ORB keypoints on an image
607 * @param image_pyramid the image pyramid to compute the features and descriptors on
608 * @param mask_pyramid the masks to apply at every level
609 * @param keypoints the resulting keypoints, clustered per level
610 */
611static void computeKeyPoints(const vector<Mat>& imagePyramid,
612 const vector<Mat>& maskPyramid,
613 vector<vector<KeyPoint> >& allKeypoints,
614 int nfeatures, int firstLevel, double scaleFactor,
615 int edgeThreshold, int patchSize, int scoreType )
616{
617 int nlevels = (int)imagePyramid.size();
618 vector<int> nfeaturesPerLevel(nlevels);
619
620 // fill the extractors and descriptors for the corresponding scales
621 float factor = (float)(1.0 / scaleFactor);
622 float ndesiredFeaturesPerScale = nfeatures*(1 - factor)/(1 - (float)pow((double)factor, (double)nlevels));
623
624 int sumFeatures = 0;
625 for( int level = 0; level < nlevels-1; level++ )
626 {
627 nfeaturesPerLevel[level] = cvRound(ndesiredFeaturesPerScale);
628 sumFeatures += nfeaturesPerLevel[level];
629 ndesiredFeaturesPerScale *= factor;
630 }
631 nfeaturesPerLevel[nlevels-1] = std::max(nfeatures - sumFeatures, 0);
632
633 // Make sure we forget about what is too close to the boundary
634 //edge_threshold_ = std::max(edge_threshold_, patch_size_/2 + kKernelWidth / 2 + 2);
635
636 // pre-compute the end of a row in a circular patch
637 int halfPatchSize = patchSize / 2;
638 vector<int> umax(halfPatchSize + 2);
639
640 int v, v0, vmax = cvFloor(halfPatchSize * sqrt(2.f) / 2 + 1);
641 int vmin = cvCeil(halfPatchSize * sqrt(2.f) / 2);
642 for (v = 0; v <= vmax; ++v)
643 umax[v] = cvRound(sqrt((double)halfPatchSize * halfPatchSize - v * v));
644
645 // Make sure we are symmetric
646 for (v = halfPatchSize, v0 = 0; v >= vmin; --v)
647 {
648 while (umax[v0] == umax[v0 + 1])
649 ++v0;
650 umax[v] = v0;
651 ++v0;
652 }
653
654 allKeypoints.resize(nlevels);
655
656 for (int level = 0; level < nlevels; ++level)
657 {
658 int featuresNum = nfeaturesPerLevel[level];
659 allKeypoints[level].reserve(featuresNum*2);
660
661 vector<KeyPoint> & keypoints = allKeypoints[level];
662
663 // Detect FAST features, 20 is a good threshold
664 FastFeatureDetector fd(20, true);
665 fd.detect(imagePyramid[level], keypoints, maskPyramid[level]);
666
667 // Remove keypoints very close to the border
668 KeyPointsFilter::runByImageBorder(keypoints, imagePyramid[level].size(), edgeThreshold);
669
670 if( scoreType == ORB::HARRIS_SCORE )
671 {
672 // Keep more points than necessary as FAST does not give amazing corners
673 KeyPointsFilter::retainBest(keypoints, 2 * featuresNum);
674
675 // Compute the Harris cornerness (better scoring than FAST)
676 HarrisResponses(imagePyramid[level], keypoints, 7, HARRIS_K);
677 }
678
679 //cull to the final desired level, using the new Harris scores or the original FAST scores.
680 KeyPointsFilter::retainBest(keypoints, featuresNum);
681
682 float sf = getScale(level, firstLevel, scaleFactor);
683
684 // Set the level of the coordinates
685 for (vector<KeyPoint>::iterator keypoint = keypoints.begin(),
686 keypointEnd = keypoints.end(); keypoint != keypointEnd; ++keypoint)
687 {
688 keypoint->octave = level;
689 keypoint->size = patchSize*sf;
690 }
691
692 computeOrientation(imagePyramid[level], keypoints, halfPatchSize, umax);
693 }
694}

697/** Compute the ORB decriptors
698 * @param image the image to compute the features and descriptors on
699 * @param integral_image the integral image of the image (can be empty, but the computation will be slower)
700 * @param level the scale at which we compute the orientation
701 * @param keypoints the keypoints to use
702 * @param descriptors the resulting descriptors
703 */
704static void computeDescriptors(const Mat& image, vector<KeyPoint>& keypoints, Mat& descriptors,
705 const vector<Point>& pattern, int dsize, int WTA_K)
706{
707 //convert to grayscale if more than one color
708 CV_Assert(image.type() == CV_8UC1);
709 //create the descriptor mat, keypoints.size() rows, BYTES cols
710 descriptors = Mat::zeros((int)keypoints.size(), dsize, CV_8UC1);
711
712 for (size_t i = 0; i < keypoints.size(); i++)
713 computeOrbDescriptor(keypoints[i], image, &pattern[0], descriptors.ptr((int)i), dsize, WTA_K);
714}

717/** Compute the ORB features and descriptors on an image
718 * @param img the image to compute the features and descriptors on
719 * @param mask the mask to apply
720 * @param keypoints the resulting keypoints
721 * @param descriptors the resulting descriptors
722 * @param do_keypoints if true, the keypoints are computed, otherwise used as an input
723 * @param do_descriptors if true, also computes the descriptors
724 */
725void ORB::operator()( InputArray _image, InputArray _mask, vector<KeyPoint>& _keypoints,
726 OutputArray _descriptors, bool useProvidedKeypoints) const
727{
728 CV_Assert(patchSize >= 2);
729
730 bool do_keypoints = !useProvidedKeypoints;
731 bool do_descriptors = _descriptors.needed();
732
733 if( (!do_keypoints && !do_descriptors) || _image.empty() )
734 return;
735
736 //ROI handling
737 const int HARRIS_BLOCK_SIZE = 9;
738 int halfPatchSize = patchSize / 2;
739 // sqrt(2.0) is for handling patch rotation
740 int descPatchSize = cvCeil(halfPatchSize*sqrt(2.0));
741 int border = std::max(edgeThreshold, std::max(descPatchSize, HARRIS_BLOCK_SIZE/2))+1;
742
743 Mat image = _image.getMat(), mask = _mask.getMat();
744 if( image.type() != CV_8UC1 )
745 cvtColor(_image, image, CV_BGR2GRAY);
746
747 int levelsNum = this->nlevels;
748
749 if( !do_keypoints )
750 {
751 // if we have pre-computed keypoints, they may use more levels than it is set in parameters
752 // !!!TODO!!! implement more correct method, independent from the used keypoint detector.
753 // Namely, the detector should provide correct size of each keypoint. Based on the keypoint size
754 // and the algorithm used (i.e. BRIEF, running on 31x31 patches) we should compute the approximate
755 // scale-factor that we need to apply. Then we should cluster all the computed scale-factors and
756 // for each cluster compute the corresponding image.
757 //
758 // In short, ultimately the descriptor should
759 // ignore octave parameter and deal only with the keypoint size.
760 levelsNum = 0;
761 for( size_t i = 0; i < _keypoints.size(); i++ )
762 levelsNum = std::max(levelsNum, std::max(_keypoints[i].octave, 0));
763 levelsNum++;
764 }
765
766 // Pre-compute the scale pyramids
767 vector<Mat> imagePyramid(levelsNum), maskPyramid(levelsNum);
768 for (int level = 0; level < levelsNum; ++level)
769 {
770 float scale = 1/getScale(level, firstLevel, scaleFactor);
771 Size sz(cvRound(image.cols*scale), cvRound(image.rows*scale));
772 Size wholeSize(sz.width + border*2, sz.height + border*2);
773 Mat temp(wholeSize, image.type()), masktemp;
774 imagePyramid[level] = temp(Rect(border, border, sz.width, sz.height));
775
776 if( !mask.empty() )
777 {
778 masktemp = Mat(wholeSize, mask.type());
779 maskPyramid[level] = masktemp(Rect(border, border, sz.width, sz.height));
780 }
781
782 // Compute the resized image
783 if( level != firstLevel )
784 {
785 if( level < firstLevel )
786 {
787 resize(image, imagePyramid[level], sz, 0, 0, INTER_LINEAR);
788 if (!mask.empty())
789 resize(mask, maskPyramid[level], sz, 0, 0, INTER_LINEAR);
790 }
791 else
792 {
793 resize(imagePyramid[level-1], imagePyramid[level], sz, 0, 0, INTER_LINEAR);
794 if (!mask.empty())
795 {
796 resize(maskPyramid[level-1], maskPyramid[level], sz, 0, 0, INTER_LINEAR);
797 threshold(maskPyramid[level], maskPyramid[level], 254, 0, THRESH_TOZERO);
798 }
799 }
800
801 copyMakeBorder(imagePyramid[level], temp, border, border, border, border,
802 BORDER_REFLECT_101+BORDER_ISOLATED);
803 if (!mask.empty())
804 copyMakeBorder(maskPyramid[level], masktemp, border, border, border, border,
805 BORDER_CONSTANT+BORDER_ISOLATED);
806 }
807 else
808 {
809 copyMakeBorder(image, temp, border, border, border, border,
810 BORDER_REFLECT_101);
811 if( !mask.empty() )
812 copyMakeBorder(mask, masktemp, border, border, border, border,
813 BORDER_CONSTANT+BORDER_ISOLATED);
814 }
815 }
816
817 // Pre-compute the keypoints (we keep the best over all scales, so this has to be done beforehand
818 vector < vector<KeyPoint> > allKeypoints;
819 if( do_keypoints )
820 {
821 // Get keypoints, those will be far enough from the border that no check will be required for the descriptor
822 computeKeyPoints(imagePyramid, maskPyramid, allKeypoints,
823 nfeatures, firstLevel, scaleFactor,
824 edgeThreshold, patchSize, scoreType);
825
826 // make sure we have the right number of keypoints keypoints
827 /*vector<KeyPoint> temp;
828
829 for (int level = 0; level < n_levels; ++level)
830 {
831 vector<KeyPoint>& keypoints = all_keypoints[level];
832 temp.insert(temp.end(), keypoints.begin(), keypoints.end());
833 keypoints.clear();
834 }
835
836 KeyPoint::retainBest(temp, n_features_);
837
838 for (vector<KeyPoint>::iterator keypoint = temp.begin(),
839 keypoint_end = temp.end(); keypoint != keypoint_end; ++keypoint)
840 all_keypoints[keypoint->octave].push_back(*keypoint);*/
841 }
842 else
843 {
844 // Remove keypoints very close to the border
845 KeyPointsFilter::runByImageBorder(_keypoints, image.size(), edgeThreshold);
846
847 // Cluster the input keypoints depending on the level they were computed at
848 allKeypoints.resize(levelsNum);
849 for (vector<KeyPoint>::iterator keypoint = _keypoints.begin(),
850 keypointEnd = _keypoints.end(); keypoint != keypointEnd; ++keypoint)
851 allKeypoints[keypoint->octave].push_back(*keypoint);
852
853 // Make sure we rescale the coordinates
854 for (int level = 0; level < levelsNum; ++level)
855 {
856 if (level == firstLevel)
857 continue;
858
859 vector<KeyPoint> & keypoints = allKeypoints[level];
860 float scale = 1/getScale(level, firstLevel, scaleFactor);
861 for (vector<KeyPoint>::iterator keypoint = keypoints.begin(),
862 keypointEnd = keypoints.end(); keypoint != keypointEnd; ++keypoint)
863 keypoint->pt *= scale;
864 }
865 }
866
867 Mat descriptors;
868 vector<Point> pattern;
869
870 if( do_descriptors )
871 {
872 int nkeypoints = 0;
873 for (int level = 0; level < levelsNum; ++level)
874 nkeypoints += (int)allKeypoints[level].size();
875 if( nkeypoints == 0 )
876 _descriptors.release();
877 else
878 {
879 _descriptors.create(nkeypoints, descriptorSize(), CV_8U);
880 descriptors = _descriptors.getMat();
881 }
882
883 const int npoints = 512;
884 Point patternbuf[npoints];
885 const Point* pattern0 = (const Point*)bit_pattern_31_;
886
887 if( patchSize != 31 )
888 {
889 pattern0 = patternbuf;
890 makeRandomPattern(patchSize, patternbuf, npoints);
891 }
892
893 CV_Assert( WTA_K == 2 || WTA_K == 3 || WTA_K == 4 );
894
895 if( WTA_K == 2 )
896 std::copy(pattern0, pattern0 + npoints, std::back_inserter(pattern));
897 else
898 {
899 int ntuples = descriptorSize()*4;
900 initializeOrbPattern(pattern0, pattern, ntuples, WTA_K, npoints);
901 }
902 }
903
904 _keypoints.clear();
905 int offset = 0;
906 for (int level = 0; level < levelsNum; ++level)
907 {
908 // Get the features and compute their orientation
909 vector<KeyPoint>& keypoints = allKeypoints[level];
910 int nkeypoints = (int)keypoints.size();
911
912 // Compute the descriptors
913 if (do_descriptors)
914 {
915 Mat desc;
916 if (!descriptors.empty())
917 {
918 desc = descriptors.rowRange(offset, offset + nkeypoints);
919 }
920
921 offset += nkeypoints;
922 // preprocess the resized image
923 Mat& workingMat = imagePyramid[level];
924 //boxFilter(working_mat, working_mat, working_mat.depth(), Size(5,5), Point(-1,-1), true, BORDER_REFLECT_101);
925 GaussianBlur(workingMat, workingMat, Size(7, 7), 2, 2, BORDER_REFLECT_101);
926 computeDescriptors(workingMat, keypoints, desc, pattern, descriptorSize(), WTA_K);
927 }
928
929 // Copy to the output data
930 if (level != firstLevel)
931 {
932 float scale = getScale(level, firstLevel, scaleFactor);
933 for (vector<KeyPoint>::iterator keypoint = keypoints.begin(),
934 keypointEnd = keypoints.end(); keypoint != keypointEnd; ++keypoint)
935 keypoint->pt *= scale;
936 }
937 // And add the keypoints to the output
938 _keypoints.insert(_keypoints.end(), keypoints.begin(), keypoints.end());
939 }
940}
