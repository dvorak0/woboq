47DynamicAdaptedFeatureDetector::DynamicAdaptedFeatureDetector(const Ptr<AdjusterAdapter>& a,
48 int min_features, int max_features, int max_iters ) :
49 escape_iters_(max_iters), min_features_(min_features), max_features_(max_features), adjuster_(a)
50{}
51
52bool DynamicAdaptedFeatureDetector::empty() const
53{
54 return adjuster_.empty() || adjuster_->empty();
55}
56
57void DynamicAdaptedFeatureDetector::detectImpl(const Mat& image, vector<KeyPoint>& keypoints, const Mat& mask) const
58{
59 //for oscillation testing
60 bool down = false;
61 bool up = false;
62
63 //flag for whether the correct threshhold has been reached
64 bool thresh_good = false;
65
66 Ptr<AdjusterAdapter> adjuster = adjuster_->clone();
67
68 //break if the desired number hasn't been reached.
69 int iter_count = escape_iters_;
70
71 while( iter_count > 0 && !(down && up) && !thresh_good && adjuster->good() )
72 {
73 keypoints.clear();
74
75 //the adjuster takes care of calling the detector with updated parameters
76 adjuster->detect(image, keypoints,mask);
77
78 if( int(keypoints.size()) < min_features_ )
79 {
80 down = true;
81 adjuster->tooFew(min_features_, (int)keypoints.size());
82 }
83 else if( int(keypoints.size()) > max_features_ )
84 {
85 up = true;
86 adjuster->tooMany(max_features_, (int)keypoints.size());
87 }
88 else
89 thresh_good = true;
90
91 iter_count--;
92 }
93
94}

96FastAdjuster::FastAdjuster( int init_thresh, bool nonmax, int min_thresh, int max_thresh ) :
97 thresh_(init_thresh), nonmax_(nonmax), init_thresh_(init_thresh),
98 min_thresh_(min_thresh), max_thresh_(max_thresh)
99{}
100
101void FastAdjuster::detectImpl(const Mat& image, vector<KeyPoint>& keypoints, const Mat& mask) const
102{
103 FastFeatureDetector(thresh_, nonmax_).detect(image, keypoints, mask);
104}
105
106void FastAdjuster::tooFew(int, int)
107{
108 //fast is easy to adjust
109 thresh_--;
110}
111
112void FastAdjuster::tooMany(int, int)
113{
114 //fast is easy to adjust
115 thresh_++;
116}
117
118//return whether or not the threshhold is beyond
119//a useful point
120bool FastAdjuster::good() const
121{
122 return (thresh_ > min_thresh_) && (thresh_ < max_thresh_);
123}
124
125Ptr<AdjusterAdapter> FastAdjuster::clone() const
126{
127 Ptr<AdjusterAdapter> cloned_obj = new FastAdjuster( init_thresh_, nonmax_, min_thresh_, max_thresh_ );
128 return cloned_obj;
129}

136void StarAdjuster::detectImpl(const Mat& image, vector<KeyPoint>& keypoints, const Mat& mask) const
137{
138 StarFeatureDetector detector_tmp(16, cvRound(thresh_), 10, 8, 3);
139 detector_tmp.detect(image, keypoints, mask);
140}

142void StarAdjuster::tooFew(int, int)
143{
144 thresh_ *= 0.9;
145 if (thresh_ < 1.1)
146 thresh_ = 1.1;
147}
148
149void StarAdjuster::tooMany(int, int)
150{
151 thresh_ *= 1.1;
152}

171{

202Ptr<AdjusterAdapter> AdjusterAdapter::create( const string& detectorType )
203{
204 Ptr<AdjusterAdapter> adapter;
205
206 if( !detectorType.compare( "FAST" ) )
207 {
208 adapter = new FastAdjuster();
209 }
210 else if( !detectorType.compare( "STAR" ) )
211 {
212 adapter = new StarAdjuster();
213 }
214 else if( !detectorType.compare( "SURF" ) )
215 {
216 adapter = new SurfAdjuster();
217 }
218
219 return adapter;
220}
