47size_t KeyPoint::hash() const
48{
49 size_t _Val = 2166136261U, scale = 16777619U;
50 Cv32suf u;
51 u.f = pt.x; _Val = (scale * _Val) ^ u.u;
52 u.f = pt.y; _Val = (scale * _Val) ^ u.u;
53 u.f = size; _Val = (scale * _Val) ^ u.u;
54 u.f = angle; _Val = (scale * _Val) ^ u.u;
55 u.f = response; _Val = (scale * _Val) ^ u.u;
56 _Val = (scale * _Val) ^ ((size_t) octave);
57 _Val = (scale * _Val) ^ ((size_t) class_id);
58 return _Val;
59}

61void write(FileStorage& fs, const string& objname, const vector<KeyPoint>& keypoints)
62{
63 WriteStructContext ws(fs, objname, CV_NODE_SEQ + CV_NODE_FLOW);
64
65 int i, npoints = (int)keypoints.size();
66 for( i = 0; i < npoints; i++ )
67 {
68 const KeyPoint& kpt = keypoints[i];
69 write(fs, kpt.pt.x);
70 write(fs, kpt.pt.y);
71 write(fs, kpt.size);
72 write(fs, kpt.angle);
73 write(fs, kpt.response);
74 write(fs, kpt.octave);
75 write(fs, kpt.class_id);
76 }
77}

80void read(const FileNode& node, vector<KeyPoint>& keypoints)
81{
82 keypoints.resize(0);
83 FileNodeIterator it = node.begin(), it_end = node.end();
84 for( ; it != it_end; )
85 {
86 KeyPoint kpt;
87 it >> kpt.pt.x >> kpt.pt.y >> kpt.size >> kpt.angle >> kpt.response >> kpt.octave >> kpt.class_id;
88 keypoints.push_back(kpt);
89 }

93void KeyPoint::convert(const std::vector<KeyPoint>& keypoints, std::vector<Point2f>& points2f,
94 const vector<int>& keypointIndexes)
95{
96 if( keypointIndexes.empty() )
97 {
98 points2f.resize( keypoints.size() );
99 for( size_t i = 0; i < keypoints.size(); i++ )
100 points2f[i] = keypoints[i].pt;
101 }
102 else
103 {
104 points2f.resize( keypointIndexes.size() );
105 for( size_t i = 0; i < keypointIndexes.size(); i++ )

107 int idx = keypointIndexes[i];
108 if( idx >= 0 )

110 else
111 {
112 CV_Error( CV_StsBadArg, "keypointIndexes has element < 0. TODO: process this case" );
113 //points2f[i] = Point2f(-1, -1);
114 }
115 }
116 }

127float KeyPoint::overlap( const KeyPoint& kp1, const KeyPoint& kp2 )
128{
129 float a = kp1.size * 0.5f;
130 float b = kp2.size * 0.5f;
131 float a_2 = a * a;
132 float b_2 = b * b;
133
134 Point2f p1 = kp1.pt;
135 Point2f p2 = kp2.pt;
136 float c = (float)norm( p1 - p2 );
137
138 float ovrl = 0.f;
139
140 // one circle is completely encovered by the other => no intersection points!
141 if( min( a, b ) + c <= max( a, b ) )
142 return min( a_2, b_2 ) / max( a_2, b_2 );
143
144 if( c < a + b ) // circles intersect
145 {
146 float c_2 = c * c;
147 float cosAlpha = ( b_2 + c_2 - a_2 ) / ( kp2.size * c );
148 float cosBeta = ( a_2 + c_2 - b_2 ) / ( kp1.size * c );
149 float alpha = acos( cosAlpha );
150 float beta = acos( cosBeta );
151 float sinAlpha = sin(alpha);
152 float sinBeta = sin(beta);
153
154 float segmentAreaA = a_2 * beta;
155 float segmentAreaB = b_2 * alpha;
156
157 float triangleAreaA = a_2 * sinBeta * cosBeta;
158 float triangleAreaB = b_2 * sinAlpha * cosAlpha;
159
160 float intersectionArea = segmentAreaA + segmentAreaB - triangleAreaA - triangleAreaB;
161 float unionArea = (a_2 + b_2) * (float)CV_PI - intersectionArea;
162
163 ovrl = intersectionArea / unionArea;
164 }
165
166 return ovrl;
167}

170struct KeypointResponseGreaterThanThreshold
171{
172 KeypointResponseGreaterThanThreshold(float _value) :
173 value(_value)
174 {
175 }
176 inline bool operator()(const KeyPoint& kpt) const
177 {
178 return kpt.response >= value;
179 }
180 float value;
181};

183struct KeypointResponseGreater
184{
185 inline bool operator()(const KeyPoint& kp1, const KeyPoint& kp2) const
186 {
187 return kp1.response > kp2.response;
188 }
189};

191// takes keypoints and culls them by the response
192void KeyPointsFilter::retainBest(vector<KeyPoint>& keypoints, int n_points)
193{
194 //this is only necessary if the keypoints size is greater than the number of desired points.
195 if( n_points >= 0 && keypoints.size() > (size_t)n_points )
196 {
197 if (n_points==0)
198 {
199 keypoints.clear();
200 return;
201 }
202 //first use nth element to partition the keypoints into the best and worst.
203 std::nth_element(keypoints.begin(), keypoints.begin() + n_points, keypoints.end(), KeypointResponseGreater());
204 //this is the boundary response, and in the case of FAST may be ambigous
205 float ambiguous_response = keypoints[n_points - 1].response;
206 //use std::partition to grab all of the keypoints with the boundary response.
207 vector<KeyPoint>::const_iterator new_end =
208 std::partition(keypoints.begin() + n_points, keypoints.end(),
209 KeypointResponseGreaterThanThreshold(ambiguous_response));
210 //resize the keypoints, given this new end point. nth_element and partition reordered the points inplace
211 keypoints.resize(new_end - keypoints.begin());
212 }

228void KeyPointsFilter::runByImageBorder( vector<KeyPoint>& keypoints, Size imageSize, int borderSize )
229{
230 if( borderSize > 0)
231 {
232 if (imageSize.height <= borderSize * 2 || imageSize.width <= borderSize * 2)
233 keypoints.clear();
234 else
235 keypoints.erase( std::remove_if(keypoints.begin(), keypoints.end(),
236 RoiPredicate(Rect(Point(borderSize, borderSize),
237 Point(imageSize.width - borderSize, imageSize.height - borderSize)))),
238 keypoints.end() );
239 }
240}

256void KeyPointsFilter::runByKeypointSize( vector<KeyPoint>& keypoints, float minSize, float maxSize )
257{
258 CV_Assert( minSize >= 0 );
259 CV_Assert( maxSize >= 0);
260 CV_Assert( minSize <= maxSize );

262 keypoints.erase( std::remove_if(keypoints.begin(), keypoints.end(), SizePredicate(minSize, maxSize)),
263 keypoints.end() );

288struct KeyPoint_LessThan
289{
290 KeyPoint_LessThan(const vector<KeyPoint>& _kp) : kp(&_kp) {}
291 bool operator()(int i, int j) const
292 {
293 const KeyPoint& kp1 = (*kp)[i];
294 const KeyPoint& kp2 = (*kp)[j];
295 if( kp1.pt.x != kp2.pt.x )
296 return kp1.pt.x < kp2.pt.x;
297 if( kp1.pt.y != kp2.pt.y )
298 return kp1.pt.y < kp2.pt.y;
299 if( kp1.size != kp2.size )
300 return kp1.size > kp2.size;
301 if( kp1.angle != kp2.angle )
302 return kp1.angle < kp2.angle;
303 if( kp1.response != kp2.response )
304 return kp1.response > kp2.response;
305 if( kp1.octave != kp2.octave )
306 return kp1.octave > kp2.octave;
307 if( kp1.class_id != kp2.class_id )
308 return kp1.class_id > kp2.class_id;
309
310 return i < j;
311 }
312 const vector<KeyPoint>* kp;
313};

315void KeyPointsFilter::removeDuplicated( vector<KeyPoint>& keypoints )
316{
317 int i, j, n = (int)keypoints.size();
318 vector<int> kpidx(n);
319 vector<uchar> mask(n, (uchar)1);
320
321 for( i = 0; i < n; i++ )
322 kpidx[i] = i;
323 std::sort(kpidx.begin(), kpidx.end(), KeyPoint_LessThan(keypoints));
324 for( i = 1, j = 0; i < n; i++ )
325 {
326 KeyPoint& kp1 = keypoints[kpidx[i]];
327 KeyPoint& kp2 = keypoints[kpidx[j]];
328 if( kp1.pt.x != kp2.pt.x || kp1.pt.y != kp2.pt.y ||
329 kp1.size != kp2.size || kp1.angle != kp2.angle )
330 j = i;
331 else
332 mask[kpidx[i]] = 0;
333 }
334
335 for( i = j = 0; i < n; i++ )
336 {
337 if( mask[i] )
338 {
339 if( i != j )
340 keypoints[j] = keypoints[i];
341 j++;
342 }
343 }
344 keypoints.resize(j);
345}
