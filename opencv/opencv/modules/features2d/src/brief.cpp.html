52inline int smoothedSum(const Mat& sum, const KeyPoint& pt, int y, int x)
53{
54 static const int HALF_KERNEL = BriefDescriptorExtractor::KERNEL_SIZE / 2;
55
56 int img_y = (int)(pt.pt.y + 0.5) + y;
57 int img_x = (int)(pt.pt.x + 0.5) + x;
58 return sum.at<int>(img_y + HALF_KERNEL + 1, img_x + HALF_KERNEL + 1)
59 - sum.at<int>(img_y + HALF_KERNEL + 1, img_x - HALF_KERNEL)
60 - sum.at<int>(img_y - HALF_KERNEL, img_x + HALF_KERNEL + 1)
61 + sum.at<int>(img_y - HALF_KERNEL, img_x - HALF_KERNEL);
62}

99BriefDescriptorExtractor::BriefDescriptorExtractor(int bytes) :
100 bytes_(bytes), test_fn_(NULL)
101{
102 switch (bytes)
103 {
104 case 16:
105 test_fn_ = pixelTests16;
106 break;
107 case 32:
108 test_fn_ = pixelTests32;
109 break;
110 case 64:
111 test_fn_ = pixelTests64;
112 break;
113 default:
114 CV_Error(CV_StsBadArg, "bytes must be 16, 32, or 64");
115 }
116}
117
118int BriefDescriptorExtractor::descriptorSize() const
119{
120 return bytes_;
121}
122
123int BriefDescriptorExtractor::descriptorType() const
124{
125 return CV_8UC1;
126}

128void BriefDescriptorExtractor::read( const FileNode& fn)
129{
130 int dSize = fn["descriptorSize"];
131 switch (dSize)
132 {
133 case 16:
134 test_fn_ = pixelTests16;
135 break;
136 case 32:
137 test_fn_ = pixelTests32;
138 break;
139 case 64:
140 test_fn_ = pixelTests64;
141 break;
142 default:
143 CV_Error(CV_StsBadArg, "descriptorSize must be 16, 32, or 64");
144 }
145 bytes_ = dSize;
146}
147
148void BriefDescriptorExtractor::write( FileStorage& fs) const
149{
150 fs << "descriptorSize" << bytes_;
151}

153void BriefDescriptorExtractor::computeImpl(const Mat& image, std::vector<KeyPoint>& keypoints, Mat& descriptors) const
154{
155 // Construct integral image for fast smoothing (box filter)
156 Mat sum;
157
158 Mat grayImage = image;
159 if( image.type() != CV_8U ) cvtColor( image, grayImage, CV_BGR2GRAY );
160
161 ///TODO allow the user to pass in a precomputed integral image
162 //if(image.type() == CV_32S)
163 // sum = image;
164 //else
165
166 integral( grayImage, sum, CV_32S);
167
168 //Remove keypoints very close to the border
169 KeyPointsFilter::runByImageBorder(keypoints, image.size(), PATCH_SIZE/2 + KERNEL_SIZE/2);
170
171 descriptors = Mat::zeros((int)keypoints.size(), bytes_, CV_8U);
172 test_fn_(sum, keypoints, descriptors);
173}
