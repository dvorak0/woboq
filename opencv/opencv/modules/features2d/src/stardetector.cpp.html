47static void
48computeIntegralImages( const Mat& matI, Mat& matS, Mat& matT, Mat& _FT )
49{
50 CV_Assert( matI.type() == CV_8U );
51
52 int x, y, rows = matI.rows, cols = matI.cols;
53
54 matS.create(rows + 1, cols + 1, CV_32S);
55 matT.create(rows + 1, cols + 1, CV_32S);
56 _FT.create(rows + 1, cols + 1, CV_32S);
57
58 const uchar* I = matI.ptr<uchar>();
59 int *S = matS.ptr<int>(), *T = matT.ptr<int>(), *FT = _FT.ptr<int>();
60 int istep = (int)matI.step, step = (int)(matS.step/sizeof(S[0]));
61
62 for( x = 0; x <= cols; x++ )
63 S[x] = T[x] = FT[x] = 0;
64
65 S += step; T += step; FT += step;
66 S[0] = T[0] = 0;
67 FT[0] = I[0];
68 for( x = 1; x < cols; x++ )
69 {
70 S[x] = S[x-1] + I[x-1];
71 T[x] = I[x-1];
72 FT[x] = I[x] + I[x-1];
73 }
74 S[cols] = S[cols-1] + I[cols-1];
75 T[cols] = FT[cols] = I[cols-1];
76
77 for( y = 2; y <= rows; y++ )
78 {
79 I += istep, S += step, T += step, FT += step;
80
81 S[0] = S[-step]; S[1] = S[-step+1] + I[0];
82 T[0] = T[-step + 1];
83 T[1] = FT[0] = T[-step + 2] + I[-istep] + I[0];
84 FT[1] = FT[-step + 2] + I[-istep] + I[1] + I[0];
85
86 for( x = 2; x < cols; x++ )
87 {
88 S[x] = S[x - 1] + S[-step + x] - S[-step + x - 1] + I[x - 1];
89 T[x] = T[-step + x - 1] + T[-step + x + 1] - T[-step*2 + x] + I[-istep + x - 1] + I[x - 1];
90 FT[x] = FT[-step + x - 1] + FT[-step + x + 1] - FT[-step*2 + x] + I[x] + I[x-1];
91 }
92
93 S[cols] = S[cols - 1] + S[-step + cols] - S[-step + cols - 1] + I[cols - 1];
94 T[cols] = FT[cols] = T[-step + cols - 1] + I[-istep + cols - 1] + I[cols - 1];
95 }
96}

104static int
105StarDetectorComputeResponses( const Mat& img, Mat& responses, Mat& sizes, int maxSize )
106{
107 const int MAX_PATTERN = 17;
108 static const int sizes0[] = {1, 2, 3, 4, 6, 8, 11, 12, 16, 22, 23, 32, 45, 46, 64, 90, 128, -1};
109 static const int pairs[][2] = {{1, 0}, {3, 1}, {4, 2}, {5, 3}, {7, 4}, {8, 5}, {9, 6},
110 {11, 8}, {13, 10}, {14, 11}, {15, 12}, {16, 14}, {-1, -1}};
111 float invSizes[MAX_PATTERN][2];
112 int sizes1[MAX_PATTERN];
113
114#if CV_SSE2
115 __m128 invSizes4[MAX_PATTERN][2];
116 __m128 sizes1_4[MAX_PATTERN];
117 Cv32suf absmask;
118 absmask.i = 0x7fffffff;
119 volatile bool useSIMD = cv::checkHardwareSupport(CV_CPU_SSE2);
120#endif
121 StarFeature f[MAX_PATTERN];
122
123 Mat sum, tilted, flatTilted;
124 int y, rows = img.rows, cols = img.cols;
125 int border, npatterns=0, maxIdx=0;
126
127 CV_Assert( img.type() == CV_8UC1 );
128
129 responses.create( img.size(), CV_32F );
130 sizes.create( img.size(), CV_16S );
131
132 while( pairs[npatterns][0] >= 0 && !
133 ( sizes0[pairs[npatterns][0]] >= maxSize
134 || sizes0[pairs[npatterns+1][0]] + sizes0[pairs[npatterns+1][0]]/2 >= std::min(rows, cols) ) )
135 {
136 ++npatterns;
137 }
138
139 npatterns += (pairs[npatterns-1][0] >= 0);
140 maxIdx = pairs[npatterns-1][0];
141
142 computeIntegralImages( img, sum, tilted, flatTilted );
143 int step = (int)(sum.step/sum.elemSize());
144
145 for(int i = 0; i <= maxIdx; i++ )
146 {
147 int ur_size = sizes0[i], t_size = sizes0[i] + sizes0[i]/2;
148 int ur_area = (2*ur_size + 1)*(2*ur_size + 1);
149 int t_area = t_size*t_size + (t_size + 1)*(t_size + 1);
150
151 f[i].p[0] = sum.ptr<int>() + (ur_size + 1)*step + ur_size + 1;
152 f[i].p[1] = sum.ptr<int>() - ur_size*step + ur_size + 1;
153 f[i].p[2] = sum.ptr<int>() + (ur_size + 1)*step - ur_size;
154 f[i].p[3] = sum.ptr<int>() - ur_size*step - ur_size;
155
156 f[i].p[4] = tilted.ptr<int>() + (t_size + 1)*step + 1;
157 f[i].p[5] = flatTilted.ptr<int>() - t_size;
158 f[i].p[6] = flatTilted.ptr<int>() + t_size + 1;
159 f[i].p[7] = tilted.ptr<int>() - t_size*step + 1;
160
161 f[i].area = ur_area + t_area;
162 sizes1[i] = sizes0[i];
163 }
164 // negate end points of the size range
165 // for a faster rejection of very small or very large features in non-maxima suppression.
166 sizes1[0] = -sizes1[0];
167 sizes1[1] = -sizes1[1];
168 sizes1[maxIdx] = -sizes1[maxIdx];
169 border = sizes0[maxIdx] + sizes0[maxIdx]/2;
170
171 for(int i = 0; i < npatterns; i++ )
172 {
173 int innerArea = f[pairs[i][1]].area;
174 int outerArea = f[pairs[i][0]].area - innerArea;
175 invSizes[i][0] = 1.f/outerArea;
176 invSizes[i][1] = 1.f/innerArea;
177 }
178
179#if CV_SSE2
180 if( useSIMD )
181 {
182 for(int i = 0; i < npatterns; i++ )
183 {
184 _mm_store_ps((float*)&invSizes4[i][0], _mm_set1_ps(invSizes[i][0]));
185 _mm_store_ps((float*)&invSizes4[i][1], _mm_set1_ps(invSizes[i][1]));
186 }
187
188 for(int i = 0; i <= maxIdx; i++ )
189 _mm_store_ps((float*)&sizes1_4[i], _mm_set1_ps((float)sizes1[i]));
190 }
191#endif
192
193 for( y = 0; y < border; y++ )
194 {
195 float* r_ptr = responses.ptr<float>(y);
196 float* r_ptr2 = responses.ptr<float>(rows - 1 - y);
197 short* s_ptr = sizes.ptr<short>(y);
198 short* s_ptr2 = sizes.ptr<short>(rows - 1 - y);
199
200 memset( r_ptr, 0, cols*sizeof(r_ptr[0]));
201 memset( r_ptr2, 0, cols*sizeof(r_ptr2[0]));
202 memset( s_ptr, 0, cols*sizeof(s_ptr[0]));
203 memset( s_ptr2, 0, cols*sizeof(s_ptr2[0]));
204 }
205
206 for( y = border; y < rows - border; y++ )
207 {
208 int x = border;
209 float* r_ptr = responses.ptr<float>(y);
210 short* s_ptr = sizes.ptr<short>(y);
211
212 memset( r_ptr, 0, border*sizeof(r_ptr[0]));
213 memset( s_ptr, 0, border*sizeof(s_ptr[0]));
214 memset( r_ptr + cols - border, 0, border*sizeof(r_ptr[0]));
215 memset( s_ptr + cols - border, 0, border*sizeof(s_ptr[0]));
216
217#if CV_SSE2
218 if( useSIMD )
219 {
220 __m128 absmask4 = _mm_set1_ps(absmask.f);
221 for( ; x <= cols - border - 4; x += 4 )
222 {
223 int ofs = y*step + x;
224 __m128 vals[MAX_PATTERN];
225 __m128 bestResponse = _mm_setzero_ps();
226 __m128 bestSize = _mm_setzero_ps();
227
228 for(int i = 0; i <= maxIdx; i++ )
229 {
230 const int** p = (const int**)&f[i].p[0];
231 __m128i r0 = _mm_sub_epi32(_mm_loadu_si128((const __m128i*)(p[0]+ofs)),
232 _mm_loadu_si128((const __m128i*)(p[1]+ofs)));
233 __m128i r1 = _mm_sub_epi32(_mm_loadu_si128((const __m128i*)(p[3]+ofs)),
234 _mm_loadu_si128((const __m128i*)(p[2]+ofs)));
235 __m128i r2 = _mm_sub_epi32(_mm_loadu_si128((const __m128i*)(p[4]+ofs)),
236 _mm_loadu_si128((const __m128i*)(p[5]+ofs)));
237 __m128i r3 = _mm_sub_epi32(_mm_loadu_si128((const __m128i*)(p[7]+ofs)),
238 _mm_loadu_si128((const __m128i*)(p[6]+ofs)));
239 r0 = _mm_add_epi32(_mm_add_epi32(r0,r1), _mm_add_epi32(r2,r3));
240 _mm_store_ps((float*)&vals[i], _mm_cvtepi32_ps(r0));
241 }
242
243 for(int i = 0; i < npatterns; i++ )
244 {
245 __m128 inner_sum = vals[pairs[i][1]];
246 __m128 outer_sum = _mm_sub_ps(vals[pairs[i][0]], inner_sum);
247 __m128 response = _mm_sub_ps(_mm_mul_ps(inner_sum, invSizes4[i][1]),
248 _mm_mul_ps(outer_sum, invSizes4[i][0]));
249 __m128 swapmask = _mm_cmpgt_ps(_mm_and_ps(response,absmask4),
250 _mm_and_ps(bestResponse,absmask4));
251 bestResponse = _mm_xor_ps(bestResponse,
252 _mm_and_ps(_mm_xor_ps(response,bestResponse), swapmask));
253 bestSize = _mm_xor_ps(bestSize,
254 _mm_and_ps(_mm_xor_ps(sizes1_4[pairs[i][0]], bestSize), swapmask));
255 }
256
257 _mm_storeu_ps(r_ptr + x, bestResponse);
258 _mm_storel_epi64((__m128i*)(s_ptr + x),
259 _mm_packs_epi32(_mm_cvtps_epi32(bestSize),_mm_setzero_si128()));
260 }
261 }
262#endif
263 for( ; x < cols - border; x++ )
264 {
265 int ofs = y*step + x;
266 int vals[MAX_PATTERN];
267 float bestResponse = 0;
268 int bestSize = 0;
269
270 for(int i = 0; i <= maxIdx; i++ )
271 {
272 const int** p = (const int**)&f[i].p[0];
273 vals[i] = p[0][ofs] - p[1][ofs] - p[2][ofs] + p[3][ofs] +
274 p[4][ofs] - p[5][ofs] - p[6][ofs] + p[7][ofs];
275 }
276 for(int i = 0; i < npatterns; i++ )
277 {
278 int inner_sum = vals[pairs[i][1]];
279 int outer_sum = vals[pairs[i][0]] - inner_sum;
280 float response = inner_sum*invSizes[i][1] - outer_sum*invSizes[i][0];
281 if( fabs(response) > fabs(bestResponse) )
282 {
283 bestResponse = response;
284 bestSize = sizes1[pairs[i][0]];
285 }
286 }
287
288 r_ptr[x] = bestResponse;
289 s_ptr[x] = (short)bestSize;
290 }
291 }
292
293 return border;
294}

297static bool StarDetectorSuppressLines( const Mat& responses, const Mat& sizes, Point pt,
298 int lineThresholdProjected, int lineThresholdBinarized )
299{
300 const float* r_ptr = responses.ptr<float>();
301 int rstep = (int)(responses.step/sizeof(r_ptr[0]));
302 const short* s_ptr = sizes.ptr<short>();
303 int sstep = (int)(sizes.step/sizeof(s_ptr[0]));
304 int sz = s_ptr[pt.y*sstep + pt.x];
305 int x, y, delta = sz/4, radius = delta*4;
306 float Lxx = 0, Lyy = 0, Lxy = 0;
307 int Lxxb = 0, Lyyb = 0, Lxyb = 0;
308
309 for( y = pt.y - radius; y <= pt.y + radius; y += delta )
310 for( x = pt.x - radius; x <= pt.x + radius; x += delta )
311 {
312 float Lx = r_ptr[y*rstep + x + 1] - r_ptr[y*rstep + x - 1];
313 float Ly = r_ptr[(y+1)*rstep + x] - r_ptr[(y-1)*rstep + x];
314 Lxx += Lx*Lx; Lyy += Ly*Ly; Lxy += Lx*Ly;
315 }
316
317 if( (Lxx + Lyy)*(Lxx + Lyy) >= lineThresholdProjected*(Lxx*Lyy - Lxy*Lxy) )
318 return true;
319
320 for( y = pt.y - radius; y <= pt.y + radius; y += delta )
321 for( x = pt.x - radius; x <= pt.x + radius; x += delta )
322 {
323 int Lxb = (s_ptr[y*sstep + x + 1] == sz) - (s_ptr[y*sstep + x - 1] == sz);
324 int Lyb = (s_ptr[(y+1)*sstep + x] == sz) - (s_ptr[(y-1)*sstep + x] == sz);
325 Lxxb += Lxb * Lxb; Lyyb += Lyb * Lyb; Lxyb += Lxb * Lyb;
326 }
327
328 if( (Lxxb + Lyyb)*(Lxxb + Lyyb) >= lineThresholdBinarized*(Lxxb*Lyyb - Lxyb*Lxyb) )
329 return true;
330
331 return false;
332}

335static void
336StarDetectorSuppressNonmax( const Mat& responses, const Mat& sizes,
337 vector<KeyPoint>& keypoints, int border,
338 int responseThreshold,
339 int lineThresholdProjected,
340 int lineThresholdBinarized,
341 int suppressNonmaxSize )
342{
343 int x, y, x1, y1, delta = suppressNonmaxSize/2;
344 int rows = responses.rows, cols = responses.cols;
345 const float* r_ptr = responses.ptr<float>();
346 int rstep = (int)(responses.step/sizeof(r_ptr[0]));
347 const short* s_ptr = sizes.ptr<short>();
348 int sstep = (int)(sizes.step/sizeof(s_ptr[0]));
349 short featureSize = 0;
350
351 for( y = border; y < rows - border; y += delta+1 )
352 for( x = border; x < cols - border; x += delta+1 )
353 {
354 float maxResponse = (float)responseThreshold;
355 float minResponse = (float)-responseThreshold;
356 Point maxPt(-1, -1), minPt(-1, -1);
357 int tileEndY = MIN(y + delta, rows - border - 1);
358 int tileEndX = MIN(x + delta, cols - border - 1);
359
360 for( y1 = y; y1 <= tileEndY; y1++ )
361 for( x1 = x; x1 <= tileEndX; x1++ )
362 {
363 float val = r_ptr[y1*rstep + x1];
364 if( maxResponse < val )
365 {
366 maxResponse = val;
367 maxPt = Point(x1, y1);
368 }
369 else if( minResponse > val )
370 {
371 minResponse = val;
372 minPt = Point(x1, y1);
373 }
374 }
375
376 if( maxPt.x >= 0 )
377 {
378 for( y1 = maxPt.y - delta; y1 <= maxPt.y + delta; y1++ )
379 for( x1 = maxPt.x - delta; x1 <= maxPt.x + delta; x1++ )
380 {
381 float val = r_ptr[y1*rstep + x1];
382 if( val >= maxResponse && (y1 != maxPt.y || x1 != maxPt.x))
383 goto skip_max;
384 }
385
386 if( (featureSize = s_ptr[maxPt.y*sstep + maxPt.x]) >= 4 &&
387 !StarDetectorSuppressLines( responses, sizes, maxPt, lineThresholdProjected,
388 lineThresholdBinarized ))
389 {
390 KeyPoint kpt((float)maxPt.x, (float)maxPt.y, featureSize, -1, maxResponse);
391 keypoints.push_back(kpt);
392 }
393 }
394 skip_max:
395 if( minPt.x >= 0 )
396 {
397 for( y1 = minPt.y - delta; y1 <= minPt.y + delta; y1++ )
398 for( x1 = minPt.x - delta; x1 <= minPt.x + delta; x1++ )
399 {
400 float val = r_ptr[y1*rstep + x1];
401 if( val <= minResponse && (y1 != minPt.y || x1 != minPt.x))
402 goto skip_min;
403 }
404
405 if( (featureSize = s_ptr[minPt.y*sstep + minPt.x]) >= 4 &&
406 !StarDetectorSuppressLines( responses, sizes, minPt,
407 lineThresholdProjected, lineThresholdBinarized))
408 {
409 KeyPoint kpt((float)minPt.x, (float)minPt.y, featureSize, -1, maxResponse);
410 keypoints.push_back(kpt);
411 }
412 }
413 skip_min:
414 ;
415 }

429void StarDetector::detectImpl( const Mat& image, vector<KeyPoint>& keypoints, const Mat& mask ) const
430{
431 Mat grayImage = image;
432 if( image.type() != CV_8U ) cvtColor( image, grayImage, CV_BGR2GRAY );
433
434 (*this)(grayImage, keypoints);
435 KeyPointsFilter::runByPixelsMask( keypoints, mask );
436}
437
438void StarDetector::operator()(const Mat& img, vector<KeyPoint>& keypoints) const
439{
440 Mat responses, sizes;
441 int border = StarDetectorComputeResponses( img, responses, sizes, maxSize );
442 keypoints.clear();
443 if( border >= 0 )
444 StarDetectorSuppressNonmax( responses, sizes, keypoints, border,
445 responseThreshold, lineThresholdProjected,
446 lineThresholdBinarized, suppressNonmaxSize );
447}
