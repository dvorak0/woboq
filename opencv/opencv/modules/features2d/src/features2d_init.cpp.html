203bool cv::initModule_features2d(void)
204{
205 bool all = true;
206 all &= !BriefDescriptorExtractor_info_auto.name().empty();
207 all &= !BRISK_info_auto.name().empty();
208 all &= !FastFeatureDetector_info_auto.name().empty();
209 all &= !FastFeatureDetector2_info_auto.name().empty();
210 all &= !StarDetector_info_auto.name().empty();
211 all &= !MSER_info_auto.name().empty();
212 all &= !FREAK_info_auto.name().empty();
213 all &= !ORB_info_auto.name().empty();
214 all &= !GFTTDetector_info_auto.name().empty();
215 all &= !HarrisDetector_info_auto.name().empty();
216 all &= !DenseFeatureDetector_info_auto.name().empty();
217 all &= !GridAdaptedFeatureDetector_info_auto.name().empty();
218 all &= !BFMatcher_info_auto.name().empty();
219 all &= !FlannBasedMatcher_info_auto.name().empty();
220
221 return all;
222}
