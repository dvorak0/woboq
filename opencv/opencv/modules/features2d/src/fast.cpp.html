54template<int patternSize>
55void FAST_t(InputArray _img, std::vector<KeyPoint>& keypoints, int threshold, bool nonmax_suppression)
56{
57 Mat img = _img.getMat();
58 const int K = patternSize/2, N = patternSize + K + 1;
59#if CV_SSE2
60 const int quarterPatternSize = patternSize/4;
61 (void)quarterPatternSize;
62#endif
63 int i, j, k, pixel[25];
64 makeOffsets(pixel, (int)img.step, patternSize);
65
66 keypoints.clear();
67
68 threshold = std::min(std::max(threshold, 0), 255);
69
70#if CV_SSE2
71 __m128i delta = _mm_set1_epi8(-128), t = _mm_set1_epi8((char)threshold), K16 = _mm_set1_epi8((char)K);
72 (void)K16;
73 (void)delta;
74 (void)t;
75#endif
76 uchar threshold_tab[512];
77 for( i = -255; i <= 255; i++ )
78 threshold_tab[i+255] = (uchar)(i < -threshold ? 1 : i > threshold ? 2 : 0);
79
80 AutoBuffer<uchar> _buf((img.cols+16)*3*(sizeof(int) + sizeof(uchar)) + 128);
81 uchar* buf[3];
82 buf[0] = _buf; buf[1] = buf[0] + img.cols; buf[2] = buf[1] + img.cols;
83 int* cpbuf[3];
84 cpbuf[0] = (int*)alignPtr(buf[2] + img.cols, sizeof(int)) + 1;
85 cpbuf[1] = cpbuf[0] + img.cols + 1;
86 cpbuf[2] = cpbuf[1] + img.cols + 1;
87 memset(buf[0], 0, img.cols*3);
88
89 for(i = 3; i < img.rows-2; i++)
90 {
91 const uchar* ptr = img.ptr<uchar>(i) + 3;
92 uchar* curr = buf[(i - 3)%3];
93 int* cornerpos = cpbuf[(i - 3)%3];
94 memset(curr, 0, img.cols);
95 int ncorners = 0;
96
97 if( i < img.rows - 3 )
98 {
99 j = 3;
100 #if CV_SSE2
101 if( patternSize == 16 )
102 {
103 for(; j < img.cols - 16 - 3; j += 16, ptr += 16)
104 {
105 __m128i m0, m1;
106 __m128i v0 = _mm_loadu_si128((const __m128i*)ptr);
107 __m128i v1 = _mm_xor_si128(_mm_subs_epu8(v0, t), delta);
108 v0 = _mm_xor_si128(_mm_adds_epu8(v0, t), delta);
109
110 __m128i x0 = _mm_sub_epi8(_mm_loadu_si128((const __m128i*)(ptr + pixel[0])), delta);
111 __m128i x1 = _mm_sub_epi8(_mm_loadu_si128((const __m128i*)(ptr + pixel[quarterPatternSize])), delta);
112 __m128i x2 = _mm_sub_epi8(_mm_loadu_si128((const __m128i*)(ptr + pixel[2*quarterPatternSize])), delta);
113 __m128i x3 = _mm_sub_epi8(_mm_loadu_si128((const __m128i*)(ptr + pixel[3*quarterPatternSize])), delta);
114 m0 = _mm_and_si128(_mm_cmpgt_epi8(x0, v0), _mm_cmpgt_epi8(x1, v0));
115 m1 = _mm_and_si128(_mm_cmpgt_epi8(v1, x0), _mm_cmpgt_epi8(v1, x1));
116 m0 = _mm_or_si128(m0, _mm_and_si128(_mm_cmpgt_epi8(x1, v0), _mm_cmpgt_epi8(x2, v0)));
117 m1 = _mm_or_si128(m1, _mm_and_si128(_mm_cmpgt_epi8(v1, x1), _mm_cmpgt_epi8(v1, x2)));
118 m0 = _mm_or_si128(m0, _mm_and_si128(_mm_cmpgt_epi8(x2, v0), _mm_cmpgt_epi8(x3, v0)));
119 m1 = _mm_or_si128(m1, _mm_and_si128(_mm_cmpgt_epi8(v1, x2), _mm_cmpgt_epi8(v1, x3)));
120 m0 = _mm_or_si128(m0, _mm_and_si128(_mm_cmpgt_epi8(x3, v0), _mm_cmpgt_epi8(x0, v0)));
121 m1 = _mm_or_si128(m1, _mm_and_si128(_mm_cmpgt_epi8(v1, x3), _mm_cmpgt_epi8(v1, x0)));
122 m0 = _mm_or_si128(m0, m1);
123 int mask = _mm_movemask_epi8(m0);
124 if( mask == 0 )
125 continue;
126 if( (mask & 255) == 0 )
127 {
128 j -= 8;
129 ptr -= 8;
130 continue;
131 }
132
133 __m128i c0 = _mm_setzero_si128(), c1 = c0, max0 = c0, max1 = c0;
134 for( k = 0; k < N; k++ )
135 {
136 __m128i x = _mm_xor_si128(_mm_loadu_si128((const __m128i*)(ptr + pixel[k])), delta);
137 m0 = _mm_cmpgt_epi8(x, v0);
138 m1 = _mm_cmpgt_epi8(v1, x);
139
140 c0 = _mm_and_si128(_mm_sub_epi8(c0, m0), m0);
141 c1 = _mm_and_si128(_mm_sub_epi8(c1, m1), m1);
142
143 max0 = _mm_max_epu8(max0, c0);
144 max1 = _mm_max_epu8(max1, c1);
145 }
146
147 max0 = _mm_max_epu8(max0, max1);
148 int m = _mm_movemask_epi8(_mm_cmpgt_epi8(max0, K16));
149
150 for( k = 0; m > 0 && k < 16; k++, m >>= 1 )
151 if(m & 1)
152 {
153 cornerpos[ncorners++] = j+k;
154 if(nonmax_suppression)
155 curr[j+k] = (uchar)cornerScore<patternSize>(ptr+k, pixel, threshold);
156 }
157 }
158 }
159 #endif
160 for( ; j < img.cols - 3; j++, ptr++ )
161 {
162 int v = ptr[0];
163 const uchar* tab = &threshold_tab[0] - v + 255;
164 int d = tab[ptr[pixel[0]]] | tab[ptr[pixel[8]]];
165
166 if( d == 0 )
167 continue;
168
169 d &= tab[ptr[pixel[2]]] | tab[ptr[pixel[10]]];
170 d &= tab[ptr[pixel[4]]] | tab[ptr[pixel[12]]];
171 d &= tab[ptr[pixel[6]]] | tab[ptr[pixel[14]]];
172
173 if( d == 0 )
174 continue;
175
176 d &= tab[ptr[pixel[1]]] | tab[ptr[pixel[9]]];
177 d &= tab[ptr[pixel[3]]] | tab[ptr[pixel[11]]];
178 d &= tab[ptr[pixel[5]]] | tab[ptr[pixel[13]]];
179 d &= tab[ptr[pixel[7]]] | tab[ptr[pixel[15]]];
180
181 if( d & 1 )
182 {
183 int vt = v - threshold, count = 0;
184
185 for( k = 0; k < N; k++ )
186 {
187 int x = ptr[pixel[k]];
188 if(x < vt)
189 {
190 if( ++count > K )
191 {
192 cornerpos[ncorners++] = j;
193 if(nonmax_suppression)
194 curr[j] = (uchar)cornerScore<patternSize>(ptr, pixel, threshold);
195 break;
196 }
197 }
198 else
199 count = 0;
200 }
201 }
202
203 if( d & 2 )
204 {
205 int vt = v + threshold, count = 0;
206
207 for( k = 0; k < N; k++ )
208 {
209 int x = ptr[pixel[k]];
210 if(x > vt)
211 {
212 if( ++count > K )
213 {
214 cornerpos[ncorners++] = j;
215 if(nonmax_suppression)
216 curr[j] = (uchar)cornerScore<patternSize>(ptr, pixel, threshold);
217 break;
218 }
219 }
220 else
221 count = 0;
222 }
223 }
224 }
225 }
226
227 cornerpos[-1] = ncorners;
228
229 if( i == 3 )
230 continue;
231
232 const uchar* prev = buf[(i - 4 + 3)%3];
233 const uchar* pprev = buf[(i - 5 + 3)%3];
234 cornerpos = cpbuf[(i - 4 + 3)%3];
235 ncorners = cornerpos[-1];
236
237 for( k = 0; k < ncorners; k++ )
238 {
239 j = cornerpos[k];
240 int score = prev[j];
241 if( !nonmax_suppression ||
242 (score > prev[j+1] && score > prev[j-1] &&
243 score > pprev[j-1] && score > pprev[j] && score > pprev[j+1] &&
244 score > curr[j-1] && score > curr[j] && score > curr[j+1]) )
245 {
246 keypoints.push_back(KeyPoint((float)j, (float)(i-1), 7.f, -1, (float)score));
247 }
248 }
249 }
250}

252void FASTX(InputArray _img, std::vector<KeyPoint>& keypoints, int threshold, bool nonmax_suppression, int type)
253{
254 switch(type) {
255 case FastFeatureDetector::TYPE_5_8:
256 FAST_t<8>(_img, keypoints, threshold, nonmax_suppression);
257 break;
258 case FastFeatureDetector::TYPE_7_12:
259 FAST_t<12>(_img, keypoints, threshold, nonmax_suppression);
260 break;
261 case FastFeatureDetector::TYPE_9_16:
262#ifdef HAVE_TEGRA_OPTIMIZATION
263 if(tegra::FAST(_img, keypoints, threshold, nonmax_suppression))
264 break;
265#endif
266 FAST_t<16>(_img, keypoints, threshold, nonmax_suppression);
267 break;
268 }
269}
270
271void FAST(InputArray _img, std::vector<KeyPoint>& keypoints, int threshold, bool nonmax_suppression)
272{
273 FASTX(_img, keypoints, threshold, nonmax_suppression, FastFeatureDetector::TYPE_9_16);
274}

291void FastFeatureDetector::detectImpl( const Mat& image, vector<KeyPoint>& keypoints, const Mat& mask ) const
292{
293 Mat grayImage = image;
294 if( image.type() != CV_8U ) cvtColor( image, grayImage, CV_BGR2GRAY );
295 FAST( grayImage, keypoints, threshold, nonmaxSuppression );
296 KeyPointsFilter::runByPixelsMask( keypoints, mask );
297}
298
299void FastFeatureDetector2::detectImpl( const Mat& image, vector<KeyPoint>& keypoints, const Mat& mask ) const
300{
301 Mat grayImage = image;
302 if( image.type() != CV_8U ) cvtColor( image, grayImage, CV_BGR2GRAY );
303 FASTX( grayImage, keypoints, threshold, nonmaxSuppression, type );
304 KeyPointsFilter::runByPixelsMask( keypoints, mask );
305}
