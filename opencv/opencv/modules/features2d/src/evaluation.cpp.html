49template<typename _Tp> static int solveQuadratic(_Tp a, _Tp b, _Tp c, _Tp& x1, _Tp& x2)
50{
51 if( a == 0 )
52 {
53 if( b == 0 )
54 {
55 x1 = x2 = 0;
56 return c == 0;
57 }
58 x1 = x2 = -c/b;
59 return 1;
60 }
61
62 _Tp d = b*b - 4*a*c;
63 if( d < 0 )
64 {
65 x1 = x2 = 0;
66 return 0;
67 }
68 if( d > 0 )
69 {
70 d = std::sqrt(d);
71 double s = 1/(2*a);
72 x1 = (-b - d)*s;
73 x2 = (-b + d)*s;
74 if( x1 > x2 )
75 std::swap(x1, x2);
76 return 2;
77 }
78 x1 = x2 = -b/(2*a);
79 return 1;
80}

84static inline Point2f applyHomography( const Mat_<double>& H, const Point2f& pt )
85{
86 double z = H(2,0)*pt.x + H(2,1)*pt.y + H(2,2);
87 if( z )
88 {
89 double w = 1./z;
90 return Point2f( (float)((H(0,0)*pt.x + H(0,1)*pt.y + H(0,2))*w), (float)((H(1,0)*pt.x + H(1,1)*pt.y + H(1,2))*w) );
91 }
92 return Point2f( numeric_limits<float>::max(), numeric_limits<float>::max() );
93}

95static inline void linearizeHomographyAt( const Mat_<double>& H, const Point2f& pt, Mat_<double>& A )
96{
97 A.create(2,2);
98 double p1 = H(0,0)*pt.x + H(0,1)*pt.y + H(0,2),
99 p2 = H(1,0)*pt.x + H(1,1)*pt.y + H(1,2),
100 p3 = H(2,0)*pt.x + H(2,1)*pt.y + H(2,2),
101 p3_2 = p3*p3;
102 if( p3 )
103 {
104 A(0,0) = H(0,0)/p3 - p1*H(2,0)/p3_2; // fxdx
105 A(0,1) = H(0,1)/p3 - p1*H(2,1)/p3_2; // fxdy
106
107 A(1,0) = H(1,0)/p3 - p2*H(2,0)/p3_2; // fydx
108 A(1,1) = H(1,1)/p3 - p2*H(2,1)/p3_2; // fydx
109 }
110 else
111 A.setTo(Scalar::all(numeric_limits<double>::max()));
112}

135EllipticKeyPoint::EllipticKeyPoint()
136{
137 *this = EllipticKeyPoint(Point2f(0,0), Scalar(1, 0, 1) );
138}

140EllipticKeyPoint::EllipticKeyPoint( const Point2f& _center, const Scalar& _ellipse )
141{
142 center = _center;
143 ellipse = _ellipse;
144
145 double a = ellipse[0], b = ellipse[1], c = ellipse[2];
146 double ac_b2 = a*c - b*b;
147 double x1, x2;
148 solveQuadratic(1., -(a+c), ac_b2, x1, x2);
149 axes.width = (float)(1/sqrt(x1));
150 axes.height = (float)(1/sqrt(x2));
151
152 boundingBox.width = (float)sqrt(ellipse[2]/ac_b2);
153 boundingBox.height = (float)sqrt(ellipse[0]/ac_b2);
154}

156Mat_<double> EllipticKeyPoint::getSecondMomentsMatrix( const Scalar& _ellipse )
157{
158 Mat_<double> M(2, 2);
159 M(0,0) = _ellipse[0];
160 M(1,0) = M(0,1) = _ellipse[1];
161 M(1,1) = _ellipse[2];
162 return M;
163}
164
165Mat_<double> EllipticKeyPoint::getSecondMomentsMatrix() const
166{
167 return getSecondMomentsMatrix(ellipse);
168}

170void EllipticKeyPoint::calcProjection( const Mat_<double>& H, EllipticKeyPoint& projection ) const
171{
172 Point2f dstCenter = applyHomography(H, center);
173
174 Mat_<double> invM; invert(getSecondMomentsMatrix(), invM);
175 Mat_<double> Aff; linearizeHomographyAt(H, center, Aff);
176 Mat_<double> dstM; invert(Aff*invM*Aff.t(), dstM);
177
178 projection = EllipticKeyPoint( dstCenter, Scalar(dstM(0,0), dstM(0,1), dstM(1,1)) );
179}

181void EllipticKeyPoint::convert( const vector<KeyPoint>& src, vector<EllipticKeyPoint>& dst )
182{
183 if( !src.empty() )
184 {
185 dst.resize(src.size());
186 for( size_t i = 0; i < src.size(); i++ )
187 {
188 float rad = src[i].size/2;
189 assert( rad );
190 float fac = 1.f/(rad*rad);
191 dst[i] = EllipticKeyPoint( src[i].pt, Scalar(fac, 0, fac) );
192 }
193 }

196void EllipticKeyPoint::convert( const vector<EllipticKeyPoint>& src, vector<KeyPoint>& dst )
197{
198 if( !src.empty() )
199 {
200 dst.resize(src.size());
201 for( size_t i = 0; i < src.size(); i++ )
202 {
203 Size_<float> axes = src[i].axes;
204 float rad = sqrt(axes.height*axes.width);
205 dst[i] = KeyPoint(src[i].center, 2*rad );
206 }
207 }

210void EllipticKeyPoint::calcProjection( const vector<EllipticKeyPoint>& src, const Mat_<double>& H, vector<EllipticKeyPoint>& dst )
211{
212 if( !src.empty() )
213 {
214 assert( !H.empty() && H.cols == 3 && H.rows == 3);
215 dst.resize(src.size());
216 vector<EllipticKeyPoint>::const_iterator srcIt = src.begin();
217 vector<EllipticKeyPoint>::iterator dstIt = dst.begin();
218 for( ; srcIt != src.end(); ++srcIt, ++dstIt )
219 srcIt->calcProjection(H, *dstIt);
220 }

223static void filterEllipticKeyPointsByImageSize( vector<EllipticKeyPoint>& keypoints, const Size& imgSize )
224{
225 if( !keypoints.empty() )
226 {
227 vector<EllipticKeyPoint> filtered;
228 filtered.reserve(keypoints.size());
229 vector<EllipticKeyPoint>::const_iterator it = keypoints.begin();
230 for( int i = 0; it != keypoints.end(); ++it, i++ )
231 {
232 if( it->center.x + it->boundingBox.width < imgSize.width &&
233 it->center.x - it->boundingBox.width > 0 &&
234 it->center.y + it->boundingBox.height < imgSize.height &&
235 it->center.y - it->boundingBox.height > 0 )
236 filtered.push_back(*it);
237 }
238 keypoints.assign(filtered.begin(), filtered.end());
239 }

242struct IntersectAreaCounter
243{
244 IntersectAreaCounter( float _dr, int _minx,
245 int _miny, int _maxy,
246 const Point2f& _diff,
247 const Scalar& _ellipse1, const Scalar& _ellipse2 ) :
248 dr(_dr), bua(0), bna(0), minx(_minx), miny(_miny), maxy(_maxy),
249 diff(_diff), ellipse1(_ellipse1), ellipse2(_ellipse2) {}
250 IntersectAreaCounter( const IntersectAreaCounter& counter, Split )
251 {
252 *this = counter;
253 bua = 0;
254 bna = 0;
255 }
256
257 void operator()( const BlockedRange& range )
258 {
259 CV_Assert( miny < maxy );
260 CV_Assert( dr > FLT_EPSILON );
261
262 int temp_bua = bua, temp_bna = bna;
263 for( int i = range.begin(); i != range.end(); i++ )
264 {
265 float rx1 = minx + i*dr;
266 float rx2 = rx1 - diff.x;
267 for( float ry1 = (float)miny; ry1 <= (float)maxy; ry1 += dr )
268 {
269 float ry2 = ry1 - diff.y;
270 //compute the distance from the ellipse center
271 float e1 = (float)(ellipse1[0]*rx1*rx1 + 2*ellipse1[1]*rx1*ry1 + ellipse1[2]*ry1*ry1);
272 float e2 = (float)(ellipse2[0]*rx2*rx2 + 2*ellipse2[1]*rx2*ry2 + ellipse2[2]*ry2*ry2);
273 //compute the area
274 if( e1<1 && e2<1 ) temp_bna++;
275 if( e1<1 || e2<1 ) temp_bua++;
276 }
277 }
278 bua = temp_bua;
279 bna = temp_bna;
280 }
281
282 void join( IntersectAreaCounter& ac )
283 {
284 bua += ac.bua;
285 bna += ac.bna;
286 }
287
288 float dr;
289 int bua, bna;
290
291 int minx;
292 int miny, maxy;
293
294 Point2f diff;
295 Scalar ellipse1, ellipse2;
296
297};

299struct SIdx
300{
301 SIdx() : S(-1), i1(-1), i2(-1) {}
302 SIdx(float _S, int _i1, int _i2) : S(_S), i1(_i1), i2(_i2) {}
303 float S;
304 int i1;
305 int i2;
306
307 bool operator<(const SIdx& v) const { return S > v.S; }
308
309 struct UsedFinder
310 {
311 UsedFinder(const SIdx& _used) : used(_used) {}
312 const SIdx& used;
313 bool operator()(const SIdx& v) const { return (v.i1 == used.i1 || v.i2 == used.i2); }
314 UsedFinder& operator=(const UsedFinder&);
315 };

318static void computeOneToOneMatchedOverlaps( const vector<EllipticKeyPoint>& keypoints1, const vector<EllipticKeyPoint>& keypoints2t,
319 bool commonPart, vector<SIdx>& overlaps, float minOverlap )
320{
321 CV_Assert( minOverlap >= 0.f );
322 overlaps.clear();
323 if( keypoints1.empty() || keypoints2t.empty() )
324 return;
325
326 overlaps.clear();
327 overlaps.reserve(cvRound(keypoints1.size() * keypoints2t.size() * 0.01));
328
329 for( size_t i1 = 0; i1 < keypoints1.size(); i1++ )
330 {
331 EllipticKeyPoint kp1 = keypoints1[i1];
332 float maxDist = sqrt(kp1.axes.width*kp1.axes.height),
333 fac = 30.f/maxDist;
334 if( !commonPart )
335 fac=3;
336
337 maxDist = maxDist*4;
338 fac = 1.f/(fac*fac);
339
340 EllipticKeyPoint keypoint1a = EllipticKeyPoint( kp1.center, Scalar(fac*kp1.ellipse[0], fac*kp1.ellipse[1], fac*kp1.ellipse[2]) );
341
342 for( size_t i2 = 0; i2 < keypoints2t.size(); i2++ )
343 {
344 EllipticKeyPoint kp2 = keypoints2t[i2];
345 Point2f diff = kp2.center - kp1.center;
346
347 if( norm(diff) < maxDist )
348 {
349 EllipticKeyPoint keypoint2a = EllipticKeyPoint( kp2.center, Scalar(fac*kp2.ellipse[0], fac*kp2.ellipse[1], fac*kp2.ellipse[2]) );
350 //find the largest eigenvalue
351 int maxx = (int)ceil(( keypoint1a.boundingBox.width > (diff.x+keypoint2a.boundingBox.width)) ?
352 keypoint1a.boundingBox.width : (diff.x+keypoint2a.boundingBox.width));
353 int minx = (int)floor((-keypoint1a.boundingBox.width < (diff.x-keypoint2a.boundingBox.width)) ?
354 -keypoint1a.boundingBox.width : (diff.x-keypoint2a.boundingBox.width));
355
356 int maxy = (int)ceil(( keypoint1a.boundingBox.height > (diff.y+keypoint2a.boundingBox.height)) ?
357 keypoint1a.boundingBox.height : (diff.y+keypoint2a.boundingBox.height));
358 int miny = (int)floor((-keypoint1a.boundingBox.height < (diff.y-keypoint2a.boundingBox.height)) ?
359 -keypoint1a.boundingBox.height : (diff.y-keypoint2a.boundingBox.height));
360 int mina = (maxx-minx) < (maxy-miny) ? (maxx-minx) : (maxy-miny) ;
361
362 //compute the area
363 float dr = (float)mina/50.f;
364 int N = (int)floor((float)(maxx - minx) / dr);
365 IntersectAreaCounter ac( dr, minx, miny, maxy, diff, keypoint1a.ellipse, keypoint2a.ellipse );
366 parallel_reduce( BlockedRange(0, N+1), ac );
367 if( ac.bna > 0 )
368 {
369 float ov = (float)ac.bna / (float)ac.bua;
370 if( ov >= minOverlap )
371 overlaps.push_back(SIdx(ov, (int)i1, (int)i2));
372 }
373 }
374 }
375 }
376
377 sort( overlaps.begin(), overlaps.end() );
378
379 typedef vector<SIdx>::iterator It;
380
381 It pos = overlaps.begin();
382 It end = overlaps.end();
383
384 while(pos != end)
385 {
386 It prev = pos++;
387 end = std::remove_if(pos, end, SIdx::UsedFinder(*prev));
388 }
389 overlaps.erase(pos, overlaps.end());
390}

392static void calculateRepeatability( const Mat& img1, const Mat& img2, const Mat& H1to2,
393 const vector<KeyPoint>& _keypoints1, const vector<KeyPoint>& _keypoints2,
394 float& repeatability, int& correspondencesCount,
395 Mat* thresholdedOverlapMask=0 )
396{
397 vector<EllipticKeyPoint> keypoints1, keypoints2, keypoints1t, keypoints2t;
398 EllipticKeyPoint::convert( _keypoints1, keypoints1 );
399 EllipticKeyPoint::convert( _keypoints2, keypoints2 );
400
401 // calculate projections of key points
402 EllipticKeyPoint::calcProjection( keypoints1, H1to2, keypoints1t );
403 Mat H2to1; invert(H1to2, H2to1);
404 EllipticKeyPoint::calcProjection( keypoints2, H2to1, keypoints2t );
405
406 float overlapThreshold;
407 bool ifEvaluateDetectors = thresholdedOverlapMask == 0;
408 if( ifEvaluateDetectors )
409 {
410 overlapThreshold = 1.f - 0.4f;
411
412 // remove key points from outside of the common image part
413 Size sz1 = img1.size(), sz2 = img2.size();
414 filterEllipticKeyPointsByImageSize( keypoints1, sz1 );
415 filterEllipticKeyPointsByImageSize( keypoints1t, sz2 );
416 filterEllipticKeyPointsByImageSize( keypoints2, sz2 );
417 filterEllipticKeyPointsByImageSize( keypoints2t, sz1 );
418 }
419 else
420 {
421 overlapThreshold = 1.f - 0.5f;
422
423 thresholdedOverlapMask->create( (int)keypoints1.size(), (int)keypoints2t.size(), CV_8UC1 );
424 thresholdedOverlapMask->setTo( Scalar::all(0) );
425 }
426 size_t size1 = keypoints1.size(), size2 = keypoints2t.size();
427 size_t minCount = MIN( size1, size2 );
428
429 // calculate overlap errors
430 vector<SIdx> overlaps;
431 computeOneToOneMatchedOverlaps( keypoints1, keypoints2t, ifEvaluateDetectors, overlaps, overlapThreshold/*min overlap*/ );
432
433 correspondencesCount = -1;
434 repeatability = -1.f;
435 if( overlaps.empty() )
436 return;
437
438 if( ifEvaluateDetectors )
439 {
440 // regions one-to-one matching
441 correspondencesCount = (int)overlaps.size();
442 repeatability = minCount ? (float)correspondencesCount / minCount : -1;
443 }
444 else
445 {
446 for( size_t i = 0; i < overlaps.size(); i++ )
447 {
448 int y = overlaps[i].i1;
449 int x = overlaps[i].i2;
450 thresholdedOverlapMask->at<uchar>(y,x) = 1;
451 }
452 }
453}

455void cv::evaluateFeatureDetector( const Mat& img1, const Mat& img2, const Mat& H1to2,
456 vector<KeyPoint>* _keypoints1, vector<KeyPoint>* _keypoints2,
457 float& repeatability, int& correspCount,
458 const Ptr<FeatureDetector>& _fdetector )
459{
460 Ptr<FeatureDetector> fdetector(_fdetector);
461 vector<KeyPoint> *keypoints1, *keypoints2, buf1, buf2;
462 keypoints1 = _keypoints1 != 0 ? _keypoints1 : &buf1;
463 keypoints2 = _keypoints2 != 0 ? _keypoints2 : &buf2;
464
465 if( (keypoints1->empty() || keypoints2->empty()) && fdetector.empty() )
466 CV_Error( CV_StsBadArg, "fdetector must not be empty when keypoints1 or keypoints2 is empty" );
467
468 if( keypoints1->empty() )
469 fdetector->detect( img1, *keypoints1 );
470 if( keypoints2->empty() )
471 fdetector->detect( img2, *keypoints2 );
472
473 calculateRepeatability( img1, img2, H1to2, *keypoints1, *keypoints2, repeatability, correspCount );
474}

482static inline float recall( int correctMatchCount, int correspondenceCount )
483{
484 return correspondenceCount ? (float)correctMatchCount / (float)correspondenceCount : -1;
485}
486
487static inline float precision( int correctMatchCount, int falseMatchCount )
488{
489 return correctMatchCount + falseMatchCount ? (float)correctMatchCount / (float)(correctMatchCount + falseMatchCount) : -1;
490}

492void cv::computeRecallPrecisionCurve( const vector<vector<DMatch> >& matches1to2,
493 const vector<vector<uchar> >& correctMatches1to2Mask,
494 vector<Point2f>& recallPrecisionCurve )
495{
496 CV_Assert( matches1to2.size() == correctMatches1to2Mask.size() );
497
498 vector<DMatchForEvaluation> allMatches;
499 int correspondenceCount = 0;
500 for( size_t i = 0; i < matches1to2.size(); i++ )
501 {
502 for( size_t j = 0; j < matches1to2[i].size(); j++ )
503 {
504 DMatchForEvaluation match = matches1to2[i][j];
505 match.isCorrect = correctMatches1to2Mask[i][j] ;
506 allMatches.push_back( match );
507 correspondenceCount += match.isCorrect != 0 ? 1 : 0;
508 }
509 }
510
511 std::sort( allMatches.begin(), allMatches.end() );
512
513 int correctMatchCount = 0, falseMatchCount = 0;
514 recallPrecisionCurve.resize( allMatches.size() );
515 for( size_t i = 0; i < allMatches.size(); i++ )
516 {
517 if( allMatches[i].isCorrect )
518 correctMatchCount++;
519 else
520 falseMatchCount++;
521
522 float r = recall( correctMatchCount, correspondenceCount );
523 float p = precision( correctMatchCount, falseMatchCount );
524 recallPrecisionCurve[i] = Point2f(1-p, r);
525 }
526}

528float cv::getRecall( const vector<Point2f>& recallPrecisionCurve, float l_precision )
529{
530 int nearestPointIndex = getNearestPoint( recallPrecisionCurve, l_precision );
531
532 float recall = -1.f;
533
534 if( nearestPointIndex >= 0 )
535 recall = recallPrecisionCurve[nearestPointIndex].y;
536
537 return recall;
538}

540int cv::getNearestPoint( const vector<Point2f>& recallPrecisionCurve, float l_precision )
541{
542 int nearestPointIndex = -1;
543
544 if( l_precision >= 0 && l_precision <= 1 )
545 {
546 float minDiff = FLT_MAX;
547 for( size_t i = 0; i < recallPrecisionCurve.size(); i++ )
548 {
549 float curDiff = std::fabs(l_precision - recallPrecisionCurve[i].x);
550 if( curDiff <= minDiff )
551 {
552 nearestPointIndex = (int)i;
553 minDiff = curDiff;
554 }
555 }
556 }
557
558 return nearestPointIndex;
559}

561void cv::evaluateGenericDescriptorMatcher( const Mat& img1, const Mat& img2, const Mat& H1to2,
562 vector<KeyPoint>& keypoints1, vector<KeyPoint>& keypoints2,
563 vector<vector<DMatch> >* _matches1to2, vector<vector<uchar> >* _correctMatches1to2Mask,
564 vector<Point2f>& recallPrecisionCurve,
565 const Ptr<GenericDescriptorMatcher>& _dmatcher )
566{
567 Ptr<GenericDescriptorMatcher> dmatcher = _dmatcher;
568 dmatcher->clear();
569
570 vector<vector<DMatch> > *matches1to2, buf1;
571 matches1to2 = _matches1to2 != 0 ? _matches1to2 : &buf1;
572
573 vector<vector<uchar> > *correctMatches1to2Mask, buf2;
574 correctMatches1to2Mask = _correctMatches1to2Mask != 0 ? _correctMatches1to2Mask : &buf2;
575
576 if( keypoints1.empty() )
577 CV_Error( CV_StsBadArg, "keypoints1 must not be empty" );
578
579 if( matches1to2->empty() && dmatcher.empty() )
580 CV_Error( CV_StsBadArg, "dmatch must not be empty when matches1to2 is empty" );
581
582 bool computeKeypoints2ByPrj = keypoints2.empty();
583 if( computeKeypoints2ByPrj )
584 {
585 assert(0);
586 // TODO: add computing keypoints2 from keypoints1 using H1to2
587 }
588
589 if( matches1to2->empty() || computeKeypoints2ByPrj )
590 {
591 dmatcher->clear();
592 dmatcher->radiusMatch( img1, keypoints1, img2, keypoints2, *matches1to2, std::numeric_limits<float>::max() );
593 }
594 float repeatability;
595 int correspCount;
596 Mat thresholdedOverlapMask; // thresholded allOverlapErrors
597 calculateRepeatability( img1, img2, H1to2, keypoints1, keypoints2, repeatability, correspCount, &thresholdedOverlapMask );
598
599 correctMatches1to2Mask->resize(matches1to2->size());
600 for( size_t i = 0; i < matches1to2->size(); i++ )
601 {
602 (*correctMatches1to2Mask)[i].resize((*matches1to2)[i].size());
603 for( size_t j = 0;j < (*matches1to2)[i].size(); j++ )
604 {
605 int indexQuery = (*matches1to2)[i][j].queryIdx;
606 int indexTrain = (*matches1to2)[i][j].trainIdx;
607 (*correctMatches1to2Mask)[i][j] = thresholdedOverlapMask.at<uchar>( indexQuery, indexTrain );
608 }
609 }
610
611 computeRecallPrecisionCurve( *matches1to2, *correctMatches1to2Mask, recallPrecisionCurve );
612}
