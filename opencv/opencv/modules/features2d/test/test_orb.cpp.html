47TEST(Features2D_ORB, _1996)
48{
49 Ptr<FeatureDetector> fd = FeatureDetector::create("ORB");
50 Ptr<DescriptorExtractor> de = DescriptorExtractor::create("ORB");
51
52 Mat image = imread(string(cvtest::TS::ptr()->get_data_path()) + "shared/lena.png");
53 ASSERT_FALSE(image.empty());
54
55 Mat roi(image.size(), CV_8UC1, Scalar(0));
56
57 Point poly[] = {Point(100, 20), Point(300, 50), Point(400, 200), Point(10, 500)};
58 fillConvexPoly(roi, poly, int(sizeof(poly) / sizeof(poly[0])), Scalar(255));
59
60 std::vector<KeyPoint> keypoints;
61 fd->detect(image, keypoints, roi);
62 Mat descriptors;
63 de->compute(image, keypoints, descriptors);
64
65 //image.setTo(Scalar(255,255,255), roi);
66
67 int roiViolations = 0;
68 for(std::vector<KeyPoint>::const_iterator kp = keypoints.begin(); kp != keypoints.end(); ++kp)
69 {
70 int x = cvRound(kp->pt.x);
71 int y = cvRound(kp->pt.y);
72
73 ASSERT_LE(0, x);
74 ASSERT_LE(0, y);
75 ASSERT_GT(image.cols, x);
76 ASSERT_GT(image.rows, y);

92}
