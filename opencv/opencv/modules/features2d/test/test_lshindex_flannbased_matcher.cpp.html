30/*
31 For OpenCV2.4/OpenCV3.0
32
33 Test for Pull Request # 3829
34 https://github.com/opencv/opencv/pull/3829
35
36 This test code creates brute force matcher for accuracy of reference, and the test target matcher.
37 Then, add() and train() transformed query image descriptors, and some outlier images descriptors to both matchers.
38 Then, compared with the query image by match() and findHomography() to detect outlier and calculate accuracy.
39 And each drawMatches() images are saved, if SAVE_DRAW_MATCHES_IMAGES is true.
40 Finally, compare accuracies between the brute force matcher and the test target matcher.
41
42 The lsh algorithm uses std::random_shuffle in lsh_index.h to make the random indexes table.
43 So, in relation to default random seed value of the execution environment or by using "srand(time(0)) function",
44 the match time and accuracy of the match results are different, each time the code ran.
45 And the match time becomes late in relation to the number of the hash collision times.
46*/

96int maskMatchesByTrainImgIdx(const vector<cv::DMatch>& matches, int trainImgIdx, vector<char>& mask)
97{
98 int matchcnt = 0;
99 mask.resize(matches.size());
100 fill(mask.begin(), mask.end(), 0);
101 for (size_t i = 0; i < matches.size(); i++)
102 {
103 if (matches[i].imgIdx == trainImgIdx)
104 {
105 mask[i] = 1;
106 matchcnt++;
107 }
108 }
109 return matchcnt;
110}

112int calcHomographyAndInlierCount(const vector<cv::KeyPoint>& query_kp, const vector<cv::KeyPoint>& train_kp, const vector<cv::DMatch>& match, vector<char> &mask, cv::Mat &homography)
113{
114 // make query and current train image keypoint pairs
115 std::vector<cv::Point2f> srcPoints, dstPoints;
116 for (unsigned int i = 0; i < match.size(); ++i)
117 {
118 if (mask[i] != 0) // is current train image ?
119 {
120 srcPoints.push_back(query_kp[match[i].queryIdx].pt);
121 dstPoints.push_back(train_kp[match[i].trainIdx].pt);
122 }
123 }
124 // calc homography
125 vector<uchar> inlierMask;
126 homography = findHomography(srcPoints, dstPoints, cv::RANSAC, 3.0, inlierMask);
127
128 // update outlier mask
129 int j = 0;
130 for (unsigned int i = 0; i < match.size(); ++i)
131 {
132 if (mask[i] != 0) // is current train image ?
133 {
134 if (inlierMask.size() == 0 || inlierMask[j] == 0) // is outlier ?
135 {
136 mask[i] = 0;
137 }
138 j++;
139 }
140 }
141
142 // count inlier
143 int inlierCnt = 0;
144 for (unsigned int i = 0; i < mask.size(); ++i)
145 {
146 if (mask[i] != 0)
147 {
148 inlierCnt++;
149 }
150 }
151 return inlierCnt;
152}

154void drawDetectedRectangle(cv::Mat& imgResult, const cv::Mat& homography, const cv::Mat& imgQuery)
155{
156 std::vector<cv::Point2f> query_corners(4);
157 query_corners[0] = cv::Point(0, 0);
158 query_corners[1] = cv::Point(imgQuery.cols, 0);
159 query_corners[2] = cv::Point(imgQuery.cols, imgQuery.rows);
160 query_corners[3] = cv::Point(0, imgQuery.rows);
161 std::vector<cv::Point2f> train_corners(4);
162 perspectiveTransform(query_corners, train_corners, homography);
163 line(imgResult, train_corners[0] + query_corners[1], train_corners[1] + query_corners[1], cv::Scalar(0, 255, 0), 4);
164 line(imgResult, train_corners[1] + query_corners[1], train_corners[2] + query_corners[1], cv::Scalar(0, 255, 0), 4);
165 line(imgResult, train_corners[2] + query_corners[1], train_corners[3] + query_corners[1], cv::Scalar(0, 255, 0), 4);
166 line(imgResult, train_corners[3] + query_corners[1], train_corners[0] + query_corners[1], cv::Scalar(0, 255, 0), 4);
167}

169// --------------------------------------------------------------------------------------
170// transform query image, extract&compute, train, matching and save result image function
171// --------------------------------------------------------------------------------------
172typedef struct tagTrainInfo
173{
174 int traindesccnt;
175 double traintime;
176 double matchtime;
177 double accuracy;
178}TrainInfo;
179
180TrainInfo transImgAndTrain(
181 cv::Feature2D *fe,
182 cv::DescriptorMatcher *matcher,
183 const string &matchername,
184 const cv::Mat& imgQuery, const vector<cv::KeyPoint>& query_kp, const cv::Mat& query_desc,
185 const vector<cv::Mat>& imgOutliers, const vector<vector<cv::KeyPoint> >& outliers_kp, const vector<cv::Mat>& outliers_desc, const int totalOutlierDescCnt,
186 const float t, const testparam *tp,
187 const int testno, const bool bVerboseOutput, const bool bSaveDrawMatches)
188{
189 TrainInfo ti;
190
191 // transform query image
192 cv::Mat imgTransform;
193 (tp->transfunc)(t, imgQuery, imgTransform);
194
195 // extract kp and compute desc from transformed query image
196 vector<cv::KeyPoint> trans_query_kp;
197 cv::Mat trans_query_desc;
198#if CV_MAJOR_VERSION==2
199 (*fe)(imgTransform, cv::Mat(), trans_query_kp, trans_query_desc);
200#elif CV_MAJOR_VERSION==3
201 fe->detectAndCompute(imgTransform, Mat(), trans_query_kp, trans_query_desc);
202#endif
203 // add&train transformed query desc and outlier desc
204 matcher->clear();
205 matcher->add(vector<cv::Mat>(1, trans_query_desc));
206 double s = (double)cv::getTickCount();
207 matcher->train();
208 ti.traintime = 1000.0*((double)cv::getTickCount() - s) / cv::getTickFrequency();
209 ti.traindesccnt = trans_query_desc.rows;
210#if defined(TRAIN_WITH_OUTLIER_IMAGES)
211 // same as matcher->add(outliers_desc); matcher->train();
212 for (unsigned int i = 0; i < outliers_desc.size(); ++i)
213 {
214 matcher->add(vector<cv::Mat>(1, outliers_desc[i]));
215 s = (double)cv::getTickCount();
216 matcher->train();
217 ti.traintime += 1000.0*((double)cv::getTickCount() - s) / cv::getTickFrequency();
218 }
219 ti.traindesccnt += totalOutlierDescCnt;
220#endif
221 // matching
222 vector<cv::DMatch> match;
223 s = (double)cv::getTickCount();
224 matcher->match(query_desc, match);
225 ti.matchtime = 1000.0*((double)cv::getTickCount() - s) / cv::getTickFrequency();
226
227 // prepare a directory and variables for save matching images
228 vector<char> mask;
229 cv::Mat imgResult;
230 const char resultDir[] = "result";
231 if (bSaveDrawMatches)
232 {
233 testing::internal::FilePath fp = testing::internal::FilePath(resultDir);
234 fp.CreateFolder();
235 }
236
237 char buff[2048];
238 int matchcnt;
239
240 // save query vs transformed query matching image with detected rectangle
241 matchcnt = maskMatchesByTrainImgIdx(match, (int)0, mask);
242 // calc homography and inlier
243 cv::Mat homography;
244 int inlierCnt = calcHomographyAndInlierCount(query_kp, trans_query_kp, match, mask, homography);
245 ti.accuracy = (double)inlierCnt / (double)mask.size()*100.0;
246 drawMatches(imgQuery, query_kp, imgTransform, trans_query_kp, match, imgResult, cv::Scalar::all(-1), cv::Scalar::all(128), mask, cv::DrawMatchesFlags::DRAW_RICH_KEYPOINTS);
247 if (inlierCnt)
248 {
249 // draw detected rectangle
250 drawDetectedRectangle(imgResult, homography, imgQuery);
251 }
252 // draw status
253 sprintf(buff, "%s accuracy:%-3.2f%% %d descriptors training time:%-3.2fms matching :%-3.2fms", matchername.c_str(), ti.accuracy, ti.traindesccnt, ti.traintime, ti.matchtime);
254 putText(imgResult, buff, cv::Point(0, 12), cv::FONT_HERSHEY_PLAIN, 0.8, cv::Scalar(0., 0., 255.));
255 sprintf(buff, "%s/res%03d_%s_%s%.1f_inlier.png", resultDir, testno, matchername.c_str(), tp->transname.c_str(), t);
256 if (bSaveDrawMatches && !imwrite(buff, imgResult)) cout << "Image " << buff << " can not be saved (may be because directory " << resultDir << " does not exist)." << endl;
257
258#if defined(TRAIN_WITH_OUTLIER_IMAGES)
259 // save query vs outlier matching image(s)
260 for (unsigned int i = 0; i <imgOutliers.size(); ++i)
261 {
262 matchcnt = maskMatchesByTrainImgIdx(match, (int)i + 1, mask);
263 drawMatches(imgQuery, query_kp, imgOutliers[i], outliers_kp[i], match, imgResult, cv::Scalar::all(-1), cv::Scalar::all(128), mask);// , DrawMatchesFlags::DRAW_RICH_KEYPOINTS);
264 sprintf(buff, "query_num:%d train_num:%d matched:%d %d descriptors training time:%-3.2fms matching :%-3.2fms", (int)query_kp.size(), (int)outliers_kp[i].size(), matchcnt, ti.traindesccnt, ti.traintime, ti.matchtime);
265 putText(imgResult, buff, cv::Point(0, 12), cv::FONT_HERSHEY_PLAIN, 0.8, cv::Scalar(0., 0., 255.));
266 sprintf(buff, "%s/res%03d_%s_%s%.1f_outlier%02d.png", resultDir, testno, matchername.c_str(), tp->transname.c_str(), t, i);
267 if (bSaveDrawMatches && !imwrite(buff, imgResult)) cout << "Image " << buff << " can not be saved (may be because directory " << resultDir << " does not exist)." << endl;
268 }
269#endif
270 if (bVerboseOutput)
271 {
272 cout << tp->transname <<" image matching accuracy:" << ti.accuracy << "% " << ti.traindesccnt << " train:" << ti.traintime << "ms match:" << ti.matchtime << "ms" << endl;
273 }
274
275 return ti;
276}

280// --------------------------------------------------------------------------------------
281class CV_FeatureDetectorMatcherBaseTest : public cvtest::BaseTest
282{
283private:
284
285 testparam *tp;
286 double target_accuracy_margin_from_bfmatcher;
287 cv::Feature2D* fe; // feature detector extractor
288
289 cv::DescriptorMatcher* bfmatcher; // brute force matcher for accuracy of reference
290 cv::DescriptorMatcher* flmatcher; // flann matcher to test
291 cv::Mat imgQuery; // query image
292 vector<cv::Mat> imgOutliers; // outlier image
293 vector<cv::KeyPoint> query_kp; // query key points detect from imgQuery
294 cv::Mat query_desc; // query descriptors extract from imgQuery
295 vector<vector<cv::KeyPoint> > outliers_kp;
296 vector<cv::Mat> outliers_desc;
297 int totalOutlierDescCnt;
298
299 string flmatchername;
300
301public:
302
303 //
304 // constructor
305 //
306 CV_FeatureDetectorMatcherBaseTest(testparam* _tp, double _accuracy_margin, cv::Feature2D* _fe,
307 cv::DescriptorMatcher *_flmatcher, string _flmatchername, int norm_type_for_bfmatcher) :
308 tp(_tp),
309 target_accuracy_margin_from_bfmatcher(_accuracy_margin),
310 fe(_fe),
311 flmatcher(_flmatcher),
312 flmatchername(_flmatchername)
313 {
314#if defined(INIT_RANDOM_SEED)
315 // from test/test_eigen.cpp
316 srand((unsigned int)time(0));
317#endif
318 // create brute force matcher for accuracy of reference
319 bfmatcher = new cv::BFMatcher(norm_type_for_bfmatcher);
320 }
321
322 virtual ~CV_FeatureDetectorMatcherBaseTest()
323 {
324 if (bfmatcher)
325 {
326 delete bfmatcher;
327 bfmatcher = NULL;
328 }
329 }
330
331 //
332 // Main Test method
333 //
334 virtual void run(int)
335 {
336 // load query image
337 string strQueryFile = string(cvtest::TS::ptr()->get_data_path()) + "shared/lena.png";
338 imgQuery = cv::imread(strQueryFile, 0);
339 if (imgQuery.empty())
340 {
341 ts->printf(cvtest::TS::LOG, "Image %s can not be read.\n", strQueryFile.c_str());
342 ts->set_failed_test_info(cvtest::TS::FAIL_INVALID_TEST_DATA);
343 return;
344 }
345
346 // load outlier images
347 char* outliers[] = { (char*)"baboon.png", (char*)"fruits.png", (char*)"airplane.png" };
348 for (unsigned int i = 0; i < sizeof(outliers) / sizeof(char*); i++)
349 {
350 string strOutlierFile = string(cvtest::TS::ptr()->get_data_path()) + "shared/" + outliers[i];
351 cv::Mat imgOutlier = cv::imread(strOutlierFile, 0);
352 if (imgQuery.empty())
353 {
354 ts->printf(cvtest::TS::LOG, "Image %s can not be read.\n", strOutlierFile.c_str());
355 ts->set_failed_test_info(cvtest::TS::FAIL_INVALID_TEST_DATA);
356 return;
357 }
358 imgOutliers.push_back(imgOutlier);
359 }
360
361 // extract and compute keypoints and descriptors from query image
362#if CV_MAJOR_VERSION==2
363 (*fe)(imgQuery, cv::Mat(), query_kp, query_desc);
364#elif CV_MAJOR_VERSION==3
365 fe->detectAndCompute(imgQuery, Mat(), query_kp, query_desc);
366#endif
367 // extract and compute keypoints and descriptors from outlier images
368 fe->detect(imgOutliers, outliers_kp);
369 ((cv::DescriptorExtractor*)fe)->compute(imgOutliers, outliers_kp, outliers_desc);
370 totalOutlierDescCnt = 0;
371 for (unsigned int i = 0; i < outliers_desc.size(); ++i) totalOutlierDescCnt += outliers_desc[i].rows;
372
373 if (SHOW_DEBUG_LOG)
374 {
375 cout << query_kp.size() << " keypoints extracted from query image." << endl;
376#if defined(TRAIN_WITH_OUTLIER_IMAGES)
377 cout << totalOutlierDescCnt << " keypoints extracted from outlier image(s)." << endl;
378#endif
379 }
380 // compute brute force matcher accuracy for reference
381 double totalTrainTime = 0.;
382 double totalMatchTime = 0.;
383 double totalAccuracy = 0.;
384 int cnt = 0;
385 for (float t = tp->from; t <= tp->to; t += tp->step, ++testno_for_make_filename, ++cnt)
386 {
387 if (SHOW_DEBUG_LOG) cout << "Test No." << testno_for_make_filename << " BFMatcher " << t;
388
389 TrainInfo ti = transImgAndTrain(fe, bfmatcher, "BFMatcher",
390 imgQuery, query_kp, query_desc,
391 imgOutliers, outliers_kp, outliers_desc,
392 totalOutlierDescCnt,
393 t, tp, testno_for_make_filename, SHOW_DEBUG_LOG, SAVE_DRAW_MATCHES_IMAGES);
394 totalTrainTime += ti.traintime;
395 totalMatchTime += ti.matchtime;
396 totalAccuracy += ti.accuracy;
397 }
398 double bf_average_accuracy = totalAccuracy / cnt;
399 if (SHOW_DEBUG_LOG)
400 {
401 cout << "total training time: " << totalTrainTime << "ms" << endl;
402 cout << "total matching time: " << totalMatchTime << "ms" << endl;
403 cout << "average accuracy:" << bf_average_accuracy << "%" << endl;
404 }
405
406 // test the target matcher
407 totalTrainTime = 0.;
408 totalMatchTime = 0.;
409 totalAccuracy = 0.;
410 cnt = 0;
411 for (float t = tp->from; t <= tp->to; t += tp->step, ++testno_for_make_filename, ++cnt)
412 {
413 if (SHOW_DEBUG_LOG) cout << "Test No." << testno_for_make_filename << " " << flmatchername << " " << t;
414
415 TrainInfo ti = transImgAndTrain(fe, flmatcher, flmatchername,
416 imgQuery, query_kp, query_desc,
417 imgOutliers, outliers_kp, outliers_desc,
418 totalOutlierDescCnt,
419 t, tp, testno_for_make_filename, SHOW_DEBUG_LOG, SAVE_DRAW_MATCHES_IMAGES);
420
421 totalTrainTime += ti.traintime;
422 totalMatchTime += ti.matchtime;
423 totalAccuracy += ti.accuracy;
424 }
425 double average_accuracy = totalAccuracy / cnt;
426 double target_average_accuracy = bf_average_accuracy * target_accuracy_margin_from_bfmatcher;
427
428 if (SHOW_DEBUG_LOG)
429 {
430 cout << "total training time: " << totalTrainTime << "ms" << endl;
431 cout << "total matching time: " << totalMatchTime << "ms" << endl;
432 cout << "average accuracy:" << average_accuracy << "%" << endl;
433 cout << "threshold of the target matcher average accuracy as error :" << target_average_accuracy << "%" << endl;
434 cout << "accuracy degraded " << (100.0 - (average_accuracy / bf_average_accuracy *100.0)) << "% from BFMatcher.(lower percentage is better)" << endl;
435 }
436 // compare accuracies between the brute force matcher and the test target matcher
437 if (average_accuracy < target_average_accuracy)
438 {
439 ts->printf(cvtest::TS::LOG, "Bad average accuracy %f < %f while test %s %s query\n", average_accuracy, target_average_accuracy, flmatchername.c_str(), tp->transname.c_str());
440 ts->set_failed_test_info(cvtest::TS::FAIL_BAD_ACCURACY);
441 }
442 return;
443 }
444
445};

450static void rotate(float deg, const cv::Mat& src, cv::Mat& dst)
451{
452 cv::warpAffine(src, dst, getRotationMatrix2D(cv::Point2f(src.cols / 2.0f, src.rows / 2.0f), deg, 1), src.size(), cv::INTER_CUBIC);
453}
454static void scale(float scale, const cv::Mat& src, cv::Mat& dst)
455{
456 cv::resize(src, dst, cv::Size((int)(src.cols*scale), (int)(src.rows*scale)), cv::INTER_CUBIC);
457}
458static void blur(float k, const cv::Mat& src, cv::Mat& dst)
459{
460 GaussianBlur(src, dst, cv::Size((int)k, (int)k), 0);
461}
