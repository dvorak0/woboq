84int NearestNeighborTest::checkFind( const Mat& data )
85{
86 int code = cvtest::TS::OK;
87 int pointsCount = 1000;
88 float noise = 0.2f;
89
90 RNG rng;
91 Mat points( pointsCount, dims, CV_32FC1 );
92 Mat results( pointsCount, K, CV_32SC1 );
93
94 std::vector<int> fmap( pointsCount );
95 for( int pi = 0; pi < pointsCount; pi++ )
96 {
97 int fi = rng.next() % featuresCount;
98 fmap[pi] = fi;
99 for( int d = 0; d < dims; d++ )
100 points.at<float>(pi, d) = data.at<float>(fi, d) + rng.uniform(0.0f, 1.0f) * noise;
101 }
102
103 code = findNeighbors( points, results );
104
105 if( code == cvtest::TS::OK )
106 {
107 int correctMatches = 0;
108 for( int pi = 0; pi < pointsCount; pi++ )
109 {
110 if( fmap[pi] == results.at<int>(pi, 0) )
111 correctMatches++;
112 }
113
114 double correctPerc = correctMatches / (double)pointsCount;
115 if (correctPerc < .75)
116 {
117 ts->printf( cvtest::TS::LOG, "correct_perc = %d\n", correctPerc );
118 code = cvtest::TS::FAIL_BAD_ACCURACY;
119 }
120 }
121
122 return code;
123}

125void NearestNeighborTest::run( int /*start_from*/ ) {
126 int code = cvtest::TS::OK, tempCode;
127 Mat desc( featuresCount, dims, CV_32FC1 );
128 ts->get_rng().fill( desc, RNG::UNIFORM, minValue, maxValue );
129
130 createModel( desc );
131
132 tempCode = checkGetPoints( desc );
133 if( tempCode != cvtest::TS::OK )
134 {
135 ts->printf( cvtest::TS::LOG, "bad accuracy of GetPoints \n" );
136 code = tempCode;
137 }
138
139 tempCode = checkFindBoxed();
140 if( tempCode != cvtest::TS::OK )
141 {
142 ts->printf( cvtest::TS::LOG, "bad accuracy of FindBoxed \n" );
143 code = tempCode;
144 }
145
146 tempCode = checkFind( desc );
147 if( tempCode != cvtest::TS::OK )
148 {
149 ts->printf( cvtest::TS::LOG, "bad accuracy of Find \n" );
150 code = tempCode;
151 }
152
153 releaseModel();
154
155 ts->set_failed_test_info( code );
156}

178int CV_KDTreeTest_CPP::checkGetPoints( const Mat& data )
179{
180 Mat res1( data.size(), data.type() ),
181 res3( data.size(), data.type() );
182 Mat idxs( 1, data.rows, CV_32SC1 );
183 for( int pi = 0; pi < data.rows; pi++ )
184 {
185 idxs.at<int>(0, pi) = pi;
186 // 1st way
187 const float* point = tr->getPoint(pi);
188 for( int di = 0; di < data.cols; di++ )
189 res1.at<float>(pi, di) = point[di];
190 }
191
192 // 3d way
193 tr->getPoints( idxs, res3 );
194
195 if( norm( res1, data, NORM_L1) != 0 ||
196 norm( res3, data, NORM_L1) != 0)
197 return cvtest::TS::FAIL_BAD_ACCURACY;
198 return cvtest::TS::OK;
199}

201int CV_KDTreeTest_CPP::checkFindBoxed()
202{
203 vector<float> min( dims, static_cast<float>(minValue)), max(dims, static_cast<float>(maxValue));
204 vector<int> indices;
205 tr->findOrthoRange( min, max, indices );
206 // TODO check indices
207 if( (int)indices.size() != featuresCount)
208 return cvtest::TS::FAIL_BAD_ACCURACY;
209 return cvtest::TS::OK;
210}
211
212int CV_KDTreeTest_CPP::findNeighbors( Mat& points, Mat& neighbors )
213{
214 const int emax = 20;
215 Mat neighbors2( neighbors.size(), CV_32SC1 );
216 int j;
217 for( int pi = 0; pi < points.rows; pi++ )
218 {
219 // 1st way
220 Mat nrow = neighbors.row(pi);
221 tr->findNearest( points.row(pi), neighbors.cols, emax, nrow );
222
223 // 2nd way
224 vector<int> neighborsIdx2( neighbors2.cols, 0 );
225 tr->findNearest( points.row(pi), neighbors2.cols, emax, neighborsIdx2 );
226 vector<int>::const_iterator it2 = neighborsIdx2.begin();
227 for( j = 0; it2 != neighborsIdx2.end(); ++it2, j++ )
228 neighbors2.at<int>(pi,j) = *it2;
229 }
230
231 // compare results
232 if( norm( neighbors, neighbors2, NORM_L1 ) != 0 )
233 return cvtest::TS::FAIL_BAD_ACCURACY;
234
235 return cvtest::TS::OK;
236}

240 delete tr;
241}

264int CV_FlannTest::knnSearch( Mat& points, Mat& neighbors )
265{
266 Mat dist( points.rows, neighbors.cols, CV_32FC1);
267 int knn = 1, j;
268
269 // 1st way
270 index->knnSearch( points, neighbors, dist, knn, SearchParams() );
271
272 // 2nd way
273 Mat neighbors1( neighbors.size(), CV_32SC1 );
274 for( int i = 0; i < points.rows; i++ )
275 {
276 float* fltPtr = points.ptr<float>(i);
277 vector<float> query( fltPtr, fltPtr + points.cols );
278 vector<int> indices( neighbors1.cols, 0 );
279 vector<float> dists( dist.cols, 0 );
280 index->knnSearch( query, indices, dists, knn, SearchParams() );
281 vector<int>::const_iterator it = indices.begin();
282 for( j = 0; it != indices.end(); ++it, j++ )
283 neighbors1.at<int>(i,j) = *it;
284 }
285
286 // compare results
287 if( norm( neighbors, neighbors1, NORM_L1 ) != 0 )
288 return cvtest::TS::FAIL_BAD_ACCURACY;
289
290 return cvtest::TS::OK;
291}

293int CV_FlannTest::radiusSearch( Mat& points, Mat& neighbors )
294{
295 Mat dist( 1, neighbors.cols, CV_32FC1);
296 Mat neighbors1( neighbors.size(), CV_32SC1 );
297 float radius = 10.0f;
298 int j;
299
300 // radiusSearch can only search one feature at a time for range search
301 for( int i = 0; i < points.rows; i++ )
302 {
303 // 1st way
304 Mat p( 1, points.cols, CV_32FC1, points.ptr<float>(i) ),
305 n( 1, neighbors.cols, CV_32SC1, neighbors.ptr<int>(i) );
306 index->radiusSearch( p, n, dist, radius, neighbors.cols, SearchParams() );
307
308 // 2nd way
309 float* fltPtr = points.ptr<float>(i);
310 vector<float> query( fltPtr, fltPtr + points.cols );
311 vector<int> indices( neighbors1.cols, 0 );
312 vector<float> dists( dist.cols, 0 );
313 index->radiusSearch( query, indices, dists, radius, neighbors.cols, SearchParams() );
314 vector<int>::const_iterator it = indices.begin();
315 for( j = 0; it != indices.end(); ++it, j++ )
316 neighbors1.at<int>(i,j) = *it;
317 }
318 // compare results
319 if( norm( neighbors, neighbors1, NORM_L1 ) != 0 )
320 return cvtest::TS::FAIL_BAD_ACCURACY;
321
322 return cvtest::TS::OK;
323}

393void CV_FlannSavedIndexTest::createModel(const cv::Mat &data)
394{
395 switch ( cvtest::randInt(ts->get_rng()) % 2 )
396 {
397 //case 0: createIndex( data, LinearIndexParams() ); break; // nothing to save for linear search
398 case 0: createIndex( data, KMeansIndexParams() ); break;
399 case 1: createIndex( data, KDTreeIndexParams() ); break;
400 //case 2: createIndex( data, CompositeIndexParams() ); break; // nothing to save for linear search
401 //case 2: createIndex( data, AutotunedIndexParams() ); break; // possible linear index !
402 default: assert(0);
403 }
404 string filename = tempfile();
405 index->save( filename );
406
407 createIndex( data, SavedIndexParams(filename.c_str()));
408 remove( filename.c_str() );
409}
