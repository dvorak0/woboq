43#include "test_precomp.hpp"

45using namespace cv;
46
47class CV_BRISKTest : public cvtest::BaseTest
48{
49public:
50 CV_BRISKTest();
51 ~CV_BRISKTest();
52protected:
53 void run(int);
54};
55
56CV_BRISKTest::CV_BRISKTest() {}
57CV_BRISKTest::~CV_BRISKTest() {}
58
59void CV_BRISKTest::run( int )
60{
61 Mat image1 = imread(string(ts->get_data_path()) + "inpaint/orig.png");
62 Mat image2 = imread(string(ts->get_data_path()) + "cameracalibration/chess9.png");
63
64 if (image1.empty() || image2.empty())
65 {
66 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_TEST_DATA );
67 return;
68 }
69
70 Mat gray1, gray2;
71 cvtColor(image1, gray1, CV_BGR2GRAY);
72 cvtColor(image2, gray2, CV_BGR2GRAY);
73
74 Ptr<FeatureDetector> detector = Algorithm::create<FeatureDetector>("Feature2D.BRISK");
75
76 vector<KeyPoint> keypoints1;
77 vector<KeyPoint> keypoints2;
78 detector->detect(image1, keypoints1);
79 detector->detect(image2, keypoints2);
80
81 for(size_t i = 0; i < keypoints1.size(); ++i)
82 {
83 const KeyPoint& kp = keypoints1[i];
84 ASSERT_NE(kp.angle, -1);
85 }
86
87 for(size_t i = 0; i < keypoints2.size(); ++i)
88 {
89 const KeyPoint& kp = keypoints2[i];
90 ASSERT_NE(kp.angle, -1);
91 }
92}
93
94TEST(Features2d_BRISK, regression) { CV_BRISKTest test; test.safe_run(); }
