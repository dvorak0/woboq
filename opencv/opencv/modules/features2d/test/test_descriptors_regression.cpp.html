55static void writeMatInBin( const Mat& mat, const string& filename )
56{
57 FILE* f = fopen( filename.c_str(), "wb");
58 if( f )
59 {
60 int type = mat.type();
61 fwrite( (void*)&mat.rows, sizeof(int), 1, f );
62 fwrite( (void*)&mat.cols, sizeof(int), 1, f );
63 fwrite( (void*)&type, sizeof(int), 1, f );
64 int dataSize = (int)(mat.step * mat.rows);
65 fwrite( (void*)&dataSize, sizeof(int), 1, f );
66 fwrite( (void*)mat.data, 1, dataSize, f );
67 fclose(f);
68 }
69}
70
71static Mat readMatFromBin( const string& filename )
72{
73 FILE* f = fopen( filename.c_str(), "rb" );
74 if( f )
75 {
76 int rows, cols, type, dataSize;
77 size_t elements_read1 = fread( (void*)&rows, sizeof(int), 1, f );
78 size_t elements_read2 = fread( (void*)&cols, sizeof(int), 1, f );
79 size_t elements_read3 = fread( (void*)&type, sizeof(int), 1, f );
80 size_t elements_read4 = fread( (void*)&dataSize, sizeof(int), 1, f );
81 CV_Assert(elements_read1 == 1 && elements_read2 == 1 && elements_read3 == 1 && elements_read4 == 1);
82
83 Mat returnMat(rows, cols, type);
84 CV_Assert(returnMat.step * returnMat.rows == (size_t)(dataSize));
85
86 size_t elements_read = fread( (void*)returnMat.data, 1, dataSize, f );
87 CV_Assert(elements_read == (size_t)(dataSize));
88
89 fclose(f);
90
91 return returnMat;
92 }
93 return Mat();
94}

96template<class Distance>
97class CV_DescriptorExtractorTest : public cvtest::BaseTest
98{
99public:
100 typedef typename Distance::ValueType ValueType;
101 typedef typename Distance::ResultType DistanceType;
102
103 CV_DescriptorExtractorTest( const string _name, DistanceType _maxDist, const Ptr<DescriptorExtractor>& _dextractor,
104 Distance d = Distance() ):
105 name(_name), maxDist(_maxDist), dextractor(_dextractor), distance(d) {}
106protected:
107 virtual void createDescriptorExtractor() {}
108
109 void compareDescriptors( const Mat& validDescriptors, const Mat& calcDescriptors )
110 {
111 if( validDescriptors.size != calcDescriptors.size || validDescriptors.type() != calcDescriptors.type() )
112 {
113 ts->printf(cvtest::TS::LOG, "Valid and computed descriptors matrices must have the same size and type.\n");
114 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_TEST_DATA );
115 return;
116 }
117
118 CV_Assert( DataType<ValueType>::type == validDescriptors.type() );
119
120 int dimension = validDescriptors.cols;
121 DistanceType curMaxDist = std::numeric_limits<DistanceType>::min();
122 for( int y = 0; y < validDescriptors.rows; y++ )
123 {
124 DistanceType dist = distance( validDescriptors.ptr<ValueType>(y), calcDescriptors.ptr<ValueType>(y), dimension );
125 if( dist > curMaxDist )
126 curMaxDist = dist;
127 }
128
129 stringstream ss;
130 ss << "Max distance between valid and computed descriptors " << curMaxDist;
131 if( curMaxDist < maxDist )
132 ss << "." << endl;
133 else
134 {
135 ss << ">" << maxDist << " - bad accuracy!"<< endl;
136 ts->set_failed_test_info( cvtest::TS::FAIL_BAD_ACCURACY );
137 }
138 ts->printf(cvtest::TS::LOG, ss.str().c_str() );
139 }
140
141 void emptyDataTest()
142 {
143 assert( !dextractor.empty() );
144
145 // One image.
146 Mat image;
147 vector<KeyPoint> keypoints;
148 Mat descriptors;
149
150 try
151 {
152 dextractor->compute( image, keypoints, descriptors );
153 }
154 catch(...)
155 {
156 ts->printf( cvtest::TS::LOG, "compute() on empty image and empty keypoints must not generate exception (1).\n");
157 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_TEST_DATA );
158 }
159
160 image.create( 50, 50, CV_8UC3 );
161 try
162 {
163 dextractor->compute( image, keypoints, descriptors );
164 }
165 catch(...)
166 {
167 ts->printf( cvtest::TS::LOG, "compute() on nonempty image and empty keypoints must not generate exception (1).\n");
168 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_TEST_DATA );
169 }
170
171 // Several images.
172 vector<Mat> images;
173 vector<vector<KeyPoint> > keypointsCollection;
174 vector<Mat> descriptorsCollection;
175 try
176 {
177 dextractor->compute( images, keypointsCollection, descriptorsCollection );
178 }
179 catch(...)
180 {
181 ts->printf( cvtest::TS::LOG, "compute() on empty images and empty keypoints collection must not generate exception (2).\n");
182 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_TEST_DATA );
183 }
184 }

186 void regressionTest()
187 {
188 assert( !dextractor.empty() );
189
190 // Read the test image.
191 string imgFilename = string(ts->get_data_path()) + FEATURES2D_DIR + "/" + IMAGE_FILENAME;
192
193 Mat img = imread( imgFilename );
194 if( img.empty() )
195 {
196 ts->printf( cvtest::TS::LOG, "Image %s can not be read.\n", imgFilename.c_str() );
197 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_TEST_DATA );
198 return;
199 }
200
201 vector<KeyPoint> keypoints;
202 FileStorage fs( string(ts->get_data_path()) + FEATURES2D_DIR + "/keypoints.xml.gz", FileStorage::READ );
203 if( fs.isOpened() )
204 {
205 read( fs.getFirstTopLevelNode(), keypoints );
206
207 Mat calcDescriptors;
208 double t = (double)getTickCount();
209 dextractor->compute( img, keypoints, calcDescriptors );
210 t = getTickCount() - t;
211 ts->printf(cvtest::TS::LOG, "\nAverage time of computing one descriptor = %g ms.\n", t/((double)cvGetTickFrequency()*1000.)/calcDescriptors.rows);
212
213 if( calcDescriptors.rows != (int)keypoints.size() )
214 {
215 ts->printf( cvtest::TS::LOG, "Count of computed descriptors and keypoints count must be equal.\n" );
216 ts->printf( cvtest::TS::LOG, "Count of keypoints is %d.\n", (int)keypoints.size() );
217 ts->printf( cvtest::TS::LOG, "Count of computed descriptors is %d.\n", calcDescriptors.rows );
218 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_OUTPUT );
219 return;
220 }
221
222 if( calcDescriptors.cols != dextractor->descriptorSize() || calcDescriptors.type() != dextractor->descriptorType() )
223 {
224 ts->printf( cvtest::TS::LOG, "Incorrect descriptor size or descriptor type.\n" );
225 ts->printf( cvtest::TS::LOG, "Expected size is %d.\n", dextractor->descriptorSize() );
226 ts->printf( cvtest::TS::LOG, "Calculated size is %d.\n", calcDescriptors.cols );
227 ts->printf( cvtest::TS::LOG, "Expected type is %d.\n", dextractor->descriptorType() );
228 ts->printf( cvtest::TS::LOG, "Calculated type is %d.\n", calcDescriptors.type() );
229 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_OUTPUT );
230 return;
231 }
232
233 // TODO read and write descriptor extractor parameters and check them
234 Mat validDescriptors = readDescriptors();
235 if( !validDescriptors.empty() )
236 compareDescriptors( validDescriptors, calcDescriptors );
237 else
238 {
239 if( !writeDescriptors( calcDescriptors ) )
240 {
241 ts->printf( cvtest::TS::LOG, "Descriptors can not be written.\n" );
242 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_TEST_DATA );
243 return;
244 }
245 }
246 }
247 else
248 {
249 ts->printf( cvtest::TS::LOG, "Compute and write keypoints.\n" );
250 fs.open( string(ts->get_data_path()) + FEATURES2D_DIR + "/keypoints.xml.gz", FileStorage::WRITE );
251 if( fs.isOpened() )
252 {
253 ORB fd;
254 fd.detect(img, keypoints);
255 write( fs, "keypoints", keypoints );
256 }
257 else
258 {
259 ts->printf(cvtest::TS::LOG, "File for writting keypoints can not be opened.\n");
260 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_TEST_DATA );
261 return;
262 }
263 }
264 }
265
266 void run(int)
267 {
268 createDescriptorExtractor();
269 if( dextractor.empty() )
270 {
271 ts->printf(cvtest::TS::LOG, "Descriptor extractor is empty.\n");
272 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_TEST_DATA );
273 return;
274 }
275
276 emptyDataTest();
277 regressionTest();
278
279 ts->set_failed_test_info( cvtest::TS::OK );
280 }
281
282 virtual Mat readDescriptors()
283 {
284 Mat res = readMatFromBin( string(ts->get_data_path()) + DESCRIPTOR_DIR + "/" + string(name) );
285 return res;
286 }
287
288 virtual bool writeDescriptors( Mat& descs )
289 {
290 writeMatInBin( descs, string(ts->get_data_path()) + DESCRIPTOR_DIR + "/" + string(name) );
291 return true;
292 }
293
294 string name;
295 const DistanceType maxDist;
296 Ptr<DescriptorExtractor> dextractor;
297 Distance distance;
298
299private:
300 CV_DescriptorExtractorTest& operator=(const CV_DescriptorExtractorTest&) { return *this; }
301};
