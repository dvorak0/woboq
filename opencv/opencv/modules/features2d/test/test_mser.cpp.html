65int CV_MserTest::LoadBoxes(const char* path, vector<CvBox2D>& boxes)
66{
67 boxes.clear();
68 FILE* f = fopen(path,"r");
69
70 if (f==NULL)
71 {
72 return 0;
73 }
74
75 while (!feof(f))
76 {
77 CvBox2D box;
78 int values_read = fscanf(f,"%f,%f,%f,%f,%f\n",&box.angle,&box.center.x,&box.center.y,&box.size.width,&box.size.height);
79 CV_Assert(values_read == 5);
80 boxes.push_back(box);
81 }
82 fclose(f);
83 return 1;
84}

86int CV_MserTest::SaveBoxes(const char* path, const vector<CvBox2D>& boxes)
87{
88 FILE* f = fopen(path,"w");
89 if (f==NULL)
90 {
91 return 0;
92 }
93 for (int i=0;i<(int)boxes.size();i++)
94 {
95 fprintf(f,"%f,%f,%f,%f,%f\n",boxes[i].angle,boxes[i].center.x,boxes[i].center.y,boxes[i].size.width,boxes[i].size.height);
96 }
97 fclose(f);
98 return 1;
99}

101int CV_MserTest::CompareBoxes(const vector<CvBox2D>& boxes1,const vector<CvBox2D>& boxes2, float max_rel_diff)
102{
103 if (boxes1.size() != boxes2.size())
104 return 0;
105
106 for (int i=0; i<(int)boxes1.size();i++)
107 {
108 float rel_diff;
109 if (!((boxes1[i].angle == 0.0f) && (abs(boxes2[i].angle) < max_rel_diff)))
110 {
111 float angle_diff = (float)fmod(boxes1[i].angle - boxes2[i].angle, 180);
112 // for angular correctness, it makes no sense to use a "relative" error.
113 // a 1-degree error around 5 degrees is equally bas as around 250 degrees.
114 // in correct cases, angle_diff can now be a bit above 0 or a bit below 180
115 if (angle_diff > 90.0f)
116 {
117 angle_diff -= 180.0f;
118 }
119 rel_diff = (float)fabs(angle_diff);
120 if (rel_diff > max_rel_diff)
121 return i;
122 }
123
124 if (!((boxes1[i].center.x == 0.0f) && (abs(boxes2[i].center.x) < max_rel_diff)))
125 {
126 rel_diff = abs(boxes1[i].center.x-boxes2[i].center.x)/abs(boxes1[i].center.x);
127 if (rel_diff > max_rel_diff)
128 return i;
129 }
130
131 if (!((boxes1[i].center.y == 0.0f) && (abs(boxes2[i].center.y) < max_rel_diff)))
132 {
133 rel_diff = abs(boxes1[i].center.y-boxes2[i].center.y)/abs(boxes1[i].center.y);
134 if (rel_diff > max_rel_diff)
135 return i;
136 }
137 if (!((boxes1[i].size.width == 0.0f) && (abs(boxes2[i].size.width) < max_rel_diff)))
138 {
139 rel_diff = abs(boxes1[i].size.width-boxes2[i].size.width)/abs(boxes1[i].size.width);
140 if (rel_diff > max_rel_diff)
141 return i;
142 }
143
144 if (!((boxes1[i].size.height == 0.0f) && (abs(boxes2[i].size.height) < max_rel_diff)))
145 {
146 rel_diff = abs(boxes1[i].size.height-boxes2[i].size.height)/abs(boxes1[i].size.height);
147 if (rel_diff > max_rel_diff)
148 return i;
149 }
150 }
151
152 return -1;
153}

155void CV_MserTest::run(int)
156{
157 string image_path = string(ts->get_data_path()) + "mser/puzzle.png";
158
159 Mat img = imread( image_path );
160 if (img.empty())
161 {
162 ts->printf( cvtest::TS::LOG, "Unable to open image mser/puzzle.png\n");
163 ts->set_failed_test_info(cvtest::TS::FAIL_MISSING_TEST_DATA);
164 return;
165 }
166
167 Mat yuv;
168 cvtColor(img, yuv, COLOR_BGR2YCrCb);
169 vector<vector<Point> > msers;
170 MSER()(yuv, msers);
171
172 vector<CvBox2D> boxes;
173 vector<CvBox2D> boxes_orig;
174 for ( size_t i = 0; i < msers.size(); i++ )
175 {
176 RotatedRect box = fitEllipse(msers[i]);
177 box.angle=(float)CV_PI/2-box.angle;
178 boxes.push_back(box);
179 }
180
181 string boxes_path = string(ts->get_data_path()) + "mser/boxes.txt";
182 string calc_boxes_path = string(ts->get_data_path()) + "mser/boxes.calc.txt";
183
184 if (!LoadBoxes(boxes_path.c_str(),boxes_orig))
185 {
186 SaveBoxes(boxes_path.c_str(),boxes);
187 ts->printf( cvtest::TS::LOG, "Unable to open data file mser/boxes.txt\n");
188 ts->set_failed_test_info(cvtest::TS::FAIL_MISSING_TEST_DATA);
189 return;
190 }
191
192 const float dissimularity = 0.01f;
193 int n_box = CompareBoxes(boxes_orig,boxes,dissimularity);
194 if (n_box < 0)
195 {
196 ts->set_failed_test_info(cvtest::TS::OK);
197 }
198 else
199 {
200 SaveBoxes(calc_boxes_path.c_str(), boxes);
201 ts->set_failed_test_info(cvtest::TS::FAIL_BAD_ACCURACY);
202 ts->printf( cvtest::TS::LOG, "Incorrect correspondence in box %d\n",n_box);
203 }
204}
