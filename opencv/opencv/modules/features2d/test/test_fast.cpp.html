43#include "test_precomp.hpp"

45using namespace cv;
46
47class CV_FastTest : public cvtest::BaseTest
48{
49public:
50 CV_FastTest();
51 ~CV_FastTest();
52protected:
53 void run(int);
54};
55
56CV_FastTest::CV_FastTest() {}
57CV_FastTest::~CV_FastTest() {}
58
59void CV_FastTest::run( int )
60{
61 for(int type=0; type <= 2; ++type) {
62 Mat image1 = imread(string(ts->get_data_path()) + "inpaint/orig.png");
63 Mat image2 = imread(string(ts->get_data_path()) + "cameracalibration/chess9.png");
64 string xml = string(ts->get_data_path()) + format("fast/result%d.xml", type);
65
66 if (image1.empty() || image2.empty())
67 {
68 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_TEST_DATA );
69 return;
70 }
71
72 Mat gray1, gray2;
73 cvtColor(image1, gray1, CV_BGR2GRAY);
74 cvtColor(image2, gray2, CV_BGR2GRAY);
75
76 vector<KeyPoint> keypoints1;
77 vector<KeyPoint> keypoints2;
78 FASTX(gray1, keypoints1, 30, true, type);
79 FASTX(gray2, keypoints2, (type > 0 ? 30 : 20), true, type);
80
81 for(size_t i = 0; i < keypoints1.size(); ++i)
82 {
83 const KeyPoint& kp = keypoints1[i];
84 cv::circle(image1, kp.pt, cvRound(kp.size/2), CV_RGB(255, 0, 0));
85 }
86
87 for(size_t i = 0; i < keypoints2.size(); ++i)
88 {
89 const KeyPoint& kp = keypoints2[i];
90 cv::circle(image2, kp.pt, cvRound(kp.size/2), CV_RGB(255, 0, 0));
91 }
92
93 Mat kps1(1, (int)(keypoints1.size() * sizeof(KeyPoint)), CV_8U, &keypoints1[0]);
94 Mat kps2(1, (int)(keypoints2.size() * sizeof(KeyPoint)), CV_8U, &keypoints2[0]);
95
96 FileStorage fs(xml, FileStorage::READ);
97 if (!fs.isOpened())
98 {
99 fs.open(xml, FileStorage::WRITE);
100 if (!fs.isOpened())
101 {
102 ts->set_failed_test_info(cvtest::TS::FAIL_INVALID_TEST_DATA);
103 return;
104 }
105 fs << "exp_kps1" << kps1;
106 fs << "exp_kps2" << kps2;
107 fs.release();
108 fs.open(xml, FileStorage::READ);
109 if (!fs.isOpened())
110 {
111 ts->set_failed_test_info(cvtest::TS::FAIL_INVALID_TEST_DATA);
112 return;
113 }
114 }
115
116 Mat exp_kps1, exp_kps2;
117 read( fs["exp_kps1"], exp_kps1, Mat() );
118 read( fs["exp_kps2"], exp_kps2, Mat() );
119 fs.release();
120
121 if ( exp_kps1.size != kps1.size || 0 != norm(exp_kps1, kps1, NORM_L2) ||
122 exp_kps2.size != kps2.size || 0 != norm(exp_kps2, kps2, NORM_L2))
123 {
124 ts->set_failed_test_info(cvtest::TS::FAIL_MISMATCH);
125 return;
126 }
127
128 /*cv::namedWindow("Img1"); cv::imshow("Img1", image1);
129 cv::namedWindow("Img2"); cv::imshow("Img2", image2);
130 cv::waitKey(0);*/
131 }
132
133 ts->set_failed_test_info(cvtest::TS::OK);
134}
