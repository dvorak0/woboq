47static
48void defaultDistribs( Mat& means, vector<Mat>& covs, int type=CV_32FC1 )
49{
50 float mp0[] = {0.0f, 0.0f}, cp0[] = {0.67f, 0.0f, 0.0f, 0.67f};
51 float mp1[] = {5.0f, 0.0f}, cp1[] = {1.0f, 0.0f, 0.0f, 1.0f};
52 float mp2[] = {1.0f, 5.0f}, cp2[] = {1.0f, 0.0f, 0.0f, 1.0f};
53 means.create(3, 2, type);
54 Mat m0( 1, 2, CV_32FC1, mp0 ), c0( 2, 2, CV_32FC1, cp0 );
55 Mat m1( 1, 2, CV_32FC1, mp1 ), c1( 2, 2, CV_32FC1, cp1 );
56 Mat m2( 1, 2, CV_32FC1, mp2 ), c2( 2, 2, CV_32FC1, cp2 );
57 means.resize(3), covs.resize(3);
58
59 Mat mr0 = means.row(0);
60 m0.convertTo(mr0, type);
61 c0.convertTo(covs[0], type);
62
63 Mat mr1 = means.row(1);
64 m1.convertTo(mr1, type);
65 c1.convertTo(covs[1], type);
66
67 Mat mr2 = means.row(2);
68 m2.convertTo(mr2, type);
69 c2.convertTo(covs[2], type);
70}

72// generate points sets by normal distributions
73static
74void generateData( Mat& data, Mat& labels, const vector<int>& sizes, const Mat& _means, const vector<Mat>& covs, int dataType, int labelType )
75{
76 vector<int>::const_iterator sit = sizes.begin();
77 int total = 0;
78 for( ; sit != sizes.end(); ++sit )
79 total += *sit;
80 CV_Assert( _means.rows == (int)sizes.size() && covs.size() == sizes.size() );
81 CV_Assert( !data.empty() && data.rows == total );
82 CV_Assert( data.type() == dataType );
83
84 labels.create( data.rows, 1, labelType );
85
86 randn( data, Scalar::all(-1.0), Scalar::all(1.0) );
87 vector<Mat> means(sizes.size());
88 for(int i = 0; i < _means.rows; i++)
89 means[i] = _means.row(i);
90 vector<Mat>::const_iterator mit = means.begin(), cit = covs.begin();
91 int bi, ei = 0;
92 sit = sizes.begin();
93 for( int p = 0, l = 0; sit != sizes.end(); ++sit, ++mit, ++cit, l++ )
94 {
95 bi = ei;
96 ei = bi + *sit;
97 assert( mit->rows == 1 && mit->cols == data.cols );
98 assert( cit->rows == data.cols && cit->cols == data.cols );
99 for( int i = bi; i < ei; i++, p++ )
100 {
101 Mat r = data.row(i);
102 r = r * (*cit) + *mit;
103 if( labelType == CV_32FC1 )
104 labels.at<float>(p, 0) = (float)l;
105 else if( labelType == CV_32SC1 )
106 labels.at<int>(p, 0) = l;
107 else
108 {
109 CV_DbgAssert(0);
110 }
111 }
112 }
113}

115static
116int maxIdx( const vector<int>& count )
117{
118 int idx = -1;
119 int maxVal = -1;
120 vector<int>::const_iterator it = count.begin();
121 for( int i = 0; it != count.end(); ++it, i++ )
122 {
123 if( *it > maxVal)
124 {
125 maxVal = *it;
126 idx = i;
127 }
128 }
129 assert( idx >= 0);
130 return idx;
131}

133static
134bool getLabelsMap( const Mat& labels, const vector<int>& sizes, vector<int>& labelsMap, bool checkClusterUniq=true )
135{
136 size_t total = 0, nclusters = sizes.size();
137 for(size_t i = 0; i < sizes.size(); i++)
138 total += sizes[i];
139
140 assert( !labels.empty() );
141 assert( labels.total() == total && (labels.cols == 1 || labels.rows == 1));
142 assert( labels.type() == CV_32SC1 || labels.type() == CV_32FC1 );
143
144 bool isFlt = labels.type() == CV_32FC1;
145
146 labelsMap.resize(nclusters);
147
148 vector<bool> buzy(nclusters, false);
149 int startIndex = 0;
150 for( size_t clusterIndex = 0; clusterIndex < sizes.size(); clusterIndex++ )
151 {
152 vector<int> count( nclusters, 0 );
153 for( int i = startIndex; i < startIndex + sizes[clusterIndex]; i++)
154 {
155 int lbl = isFlt ? (int)labels.at<float>(i) : labels.at<int>(i);
156 CV_Assert(lbl < (int)nclusters);
157 count[lbl]++;
158 CV_Assert(count[lbl] < (int)total);
159 }
160 startIndex += sizes[clusterIndex];
161
162 int cls = maxIdx( count );
163 CV_Assert( !checkClusterUniq || !buzy[cls] );
164
165 labelsMap[clusterIndex] = cls;
166
167 buzy[cls] = true;
168 }
169
170 if(checkClusterUniq)
171 {
172 for(size_t i = 0; i < buzy.size(); i++)
173 if(!buzy[i])
174 return false;
175 }
176
177 return true;
178}

180static
181bool calcErr( const Mat& labels, const Mat& origLabels, const vector<int>& sizes, float& err, bool labelsEquivalent, bool checkClusterUniq )
182{
183 err = 0;
184 CV_Assert( !labels.empty() && !origLabels.empty() );
185 CV_Assert( labels.rows == 1 || labels.cols == 1 );
186 CV_Assert( origLabels.rows == 1 || origLabels.cols == 1 );
187 CV_Assert( labels.total() == origLabels.total() );
188 CV_Assert( labels.type() == CV_32SC1 || labels.type() == CV_32FC1 );
189 CV_Assert( origLabels.type() == labels.type() );
190
191 vector<int> labelsMap;
192 bool isFlt = labels.type() == CV_32FC1;
193 if( !labelsEquivalent )
194 {
195 if( !getLabelsMap( labels, sizes, labelsMap, checkClusterUniq ) )
196 return false;
197
198 for( int i = 0; i < labels.rows; i++ )
199 if( isFlt )
200 err += labels.at<float>(i) != labelsMap[(int)origLabels.at<float>(i)] ? 1.f : 0.f;
201 else
202 err += labels.at<int>(i) != labelsMap[origLabels.at<int>(i)] ? 1.f : 0.f;
203 }
204 else
205 {
206 for( int i = 0; i < labels.rows; i++ )
207 if( isFlt )
208 err += labels.at<float>(i) != origLabels.at<float>(i) ? 1.f : 0.f;
209 else
210 err += labels.at<int>(i) != origLabels.at<int>(i) ? 1.f : 0.f;
211 }
212 err /= (float)labels.rows;
213 return true;
214}

217class CV_CvEMTest : public cvtest::BaseTest

219public:
220 CV_CvEMTest() {}
221protected:
222 virtual void run( int start_from );

224 const cv::Mat& trainData, const cv::Mat& trainLabels,
225 const cv::Mat& testData, const cv::Mat& testLabels,
226 const vector<int>& sizes);

229int CV_CvEMTest::runCase( int caseIndex, const CvEMParams& params,
230 const cv::Mat& trainData, const cv::Mat& trainLabels,
231 const cv::Mat& testData, const cv::Mat& testLabels,
232 const vector<int>& sizes )
233{
234 int code = cvtest::TS::OK;
235
236 cv::Mat labels;
237 float err;
238
239 CvEM em;
240 em.train( trainData, Mat(), params, &labels );
241
242 // check train error
243 if( !calcErr( labels, trainLabels, sizes, err , false, false ) )
244 {
245 ts->printf( cvtest::TS::LOG, "Case index %i : Bad output labels.\n", caseIndex );
246 code = cvtest::TS::FAIL_INVALID_OUTPUT;
247 }
248 else if( err > 0.006f )
249 {
250 ts->printf( cvtest::TS::LOG, "Case index %i : Bad accuracy (%f) on train data.\n", caseIndex, err );
251 code = cvtest::TS::FAIL_BAD_ACCURACY;
252 }
253
254 // check test error
255 labels.create( testData.rows, 1, CV_32SC1 );
256 for( int i = 0; i < testData.rows; i++ )
257 {
258 Mat sample = testData.row(i);
259 labels.at<int>(i,0) = (int)em.predict( sample, 0 );
260 }
261 if( !calcErr( labels, testLabels, sizes, err, false, false ) )
262 {
263 ts->printf( cvtest::TS::LOG, "Case index %i : Bad output labels.\n", caseIndex );
264 code = cvtest::TS::FAIL_INVALID_OUTPUT;
265 }
266 else if( err > 0.006f )
267 {
268 ts->printf( cvtest::TS::LOG, "Case index %i : Bad accuracy (%f) on test data.\n", caseIndex, err );
269 code = cvtest::TS::FAIL_BAD_ACCURACY;
270 }
271
272 return code;
273}

275void CV_CvEMTest::run( int /*start_from*/ )
276{
277 int sizesArr[] = { 500, 700, 800 };
278 int pointsCount = sizesArr[0]+ sizesArr[1] + sizesArr[2];
279
280 // Points distribution
281 Mat means;
282 vector<Mat> covs;
283 defaultDistribs( means, covs );
284
285 // train data
286 Mat trainData( pointsCount, 2, CV_32FC1 ), trainLabels;
287 vector<int> sizes( sizesArr, sizesArr + sizeof(sizesArr) / sizeof(sizesArr[0]) );
288 generateData( trainData, trainLabels, sizes, means, covs, CV_32FC1, CV_32SC1 );
289
290 // test data
291 Mat testData( pointsCount, 2, CV_32FC1 ), testLabels;
292 generateData( testData, testLabels, sizes, means, covs, CV_32FC1, CV_32SC1 );
293
294 CvEMParams params;
295 params.nclusters = 3;
296 Mat probs(trainData.rows, params.nclusters, CV_32FC1, cv::Scalar(1));
297 CvMat probsHdr = probs;
298 params.probs = &probsHdr;
299 Mat weights(1, params.nclusters, CV_32FC1, cv::Scalar(1));
300 CvMat weightsHdr = weights;
301 params.weights = &weightsHdr;
302 CvMat meansHdr = means;
303 params.means = &meansHdr;
304 std::vector<CvMat> covsHdrs(params.nclusters);
305 std::vector<const CvMat*> covsPtrs(params.nclusters);
306 for(int i = 0; i < params.nclusters; i++)
307 {
308 covsHdrs[i] = covs[i];
309 covsPtrs[i] = &covsHdrs[i];
310 }
311 params.covs = &covsPtrs[0];
312
313 int code = cvtest::TS::OK;
314 int caseIndex = 0;
315 {
316 params.start_step = cv::EM::START_AUTO_STEP;
317 params.cov_mat_type = cv::EM::COV_MAT_GENERIC;
318 int currCode = runCase(caseIndex++, params, trainData, trainLabels, testData, testLabels, sizes);
319 code = currCode == cvtest::TS::OK ? code : currCode;
320 }
321 {
322 params.start_step = cv::EM::START_AUTO_STEP;
323 params.cov_mat_type = cv::EM::COV_MAT_DIAGONAL;
324 int currCode = runCase(caseIndex++, params, trainData, trainLabels, testData, testLabels, sizes);
325 code = currCode == cvtest::TS::OK ? code : currCode;
326 }
327 {
328 params.start_step = cv::EM::START_AUTO_STEP;
329 params.cov_mat_type = cv::EM::COV_MAT_SPHERICAL;
330 int currCode = runCase(caseIndex++, params, trainData, trainLabels, testData, testLabels, sizes);
331 code = currCode == cvtest::TS::OK ? code : currCode;
332 }
333 {
334 params.start_step = cv::EM::START_M_STEP;
335 params.cov_mat_type = cv::EM::COV_MAT_GENERIC;
336 int currCode = runCase(caseIndex++, params, trainData, trainLabels, testData, testLabels, sizes);
337 code = currCode == cvtest::TS::OK ? code : currCode;
338 }
339 {
340 params.start_step = cv::EM::START_M_STEP;
341 params.cov_mat_type = cv::EM::COV_MAT_DIAGONAL;
342 int currCode = runCase(caseIndex++, params, trainData, trainLabels, testData, testLabels, sizes);
343 code = currCode == cvtest::TS::OK ? code : currCode;
344 }
345 {
346 params.start_step = cv::EM::START_M_STEP;
347 params.cov_mat_type = cv::EM::COV_MAT_SPHERICAL;
348 int currCode = runCase(caseIndex++, params, trainData, trainLabels, testData, testLabels, sizes);
349 code = currCode == cvtest::TS::OK ? code : currCode;
350 }
351 {
352 params.start_step = cv::EM::START_E_STEP;
353 params.cov_mat_type = cv::EM::COV_MAT_GENERIC;
354 int currCode = runCase(caseIndex++, params, trainData, trainLabels, testData, testLabels, sizes);
355 code = currCode == cvtest::TS::OK ? code : currCode;
356 }
357 {
358 params.start_step = cv::EM::START_E_STEP;
359 params.cov_mat_type = cv::EM::COV_MAT_DIAGONAL;
360 int currCode = runCase(caseIndex++, params, trainData, trainLabels, testData, testLabels, sizes);
361 code = currCode == cvtest::TS::OK ? code : currCode;
362 }
363 {
364 params.start_step = cv::EM::START_E_STEP;
365 params.cov_mat_type = cv::EM::COV_MAT_SPHERICAL;
366 int currCode = runCase(caseIndex++, params, trainData, trainLabels, testData, testLabels, sizes);
367 code = currCode == cvtest::TS::OK ? code : currCode;
368 }
369
370 ts->set_failed_test_info( code );
371}

373class CV_CvEMTest_SaveLoad : public cvtest::BaseTest {
374public:
375 CV_CvEMTest_SaveLoad() {}
376protected:
377 virtual void run( int /*start_from*/ )
378 {
379 int code = cvtest::TS::OK;
380
381 Mat samples = Mat(3,1,CV_32F);
382 samples.at<float>(0,0) = 1;
383 samples.at<float>(1,0) = 2;
384 samples.at<float>(2,0) = 3;
385
386 Mat labels(samples.rows, 1, CV_32S);
387
388 CvEMParams params;
389 params.nclusters = 2;
390
391 CvMat samples_c = samples, labels_c = labels;
392
393 CvEM em(&samples_c, 0, params, &labels_c);
394
395 Mat firstResult(samples.rows, 1, CV_32FC1);
396 for( int i = 0; i < samples.rows; i++)
397 firstResult.at<float>(i) = em.predict( samples.row(i) );
398
399 // Write out
400
401 string filename = cv::tempfile(".xml");
402 {
403 FileStorage fs = FileStorage(filename, FileStorage::WRITE);
404 try
405 {
406 em.write(fs.fs, "em");
407 }
408 catch(...)
409 {
410 ts->printf( cvtest::TS::LOG, "Crash in write method.\n" );
411 ts->set_failed_test_info( cvtest::TS::FAIL_EXCEPTION );
412 }
413 }
414
415 em.clear();
416
417 // Read in
418 {
419 FileStorage fs = FileStorage(filename, FileStorage::READ);
420 CV_Assert(fs.isOpened());
421 FileNode fn = fs["em"];
422 try
423 {
424 em.read(fs.fs, (CvFileNode*)fn.node);
425 }
426 catch(...)
427 {
428 ts->printf( cvtest::TS::LOG, "Crash in read method.\n" );
429 ts->set_failed_test_info( cvtest::TS::FAIL_EXCEPTION );
430 }
431 }
432
433 remove( filename.c_str() );
434
435 int errCaseCount = 0;
436 for( int i = 0; i < samples.rows; i++)
437 errCaseCount = std::abs(em.predict(samples.row(i)) - firstResult.at<float>(i)) < FLT_EPSILON ? 0 : 1;
438
439 if( errCaseCount > 0 )
440 {
441 ts->printf( cvtest::TS::LOG, "Different prediction results before writeing and after reading (errCaseCount=%d).\n", errCaseCount );
442 code = cvtest::TS::FAIL_BAD_ACCURACY;
443 }
444
445 ts->set_failed_test_info( code );
446 }
447};
