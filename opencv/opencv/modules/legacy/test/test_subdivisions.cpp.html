95int CV_SubdivTest::read_params( CvFileStorage* fs )
96{
97 int code = cvtest::BaseTest::read_params( fs );
98 int t;
99
100 if( code < 0 )
101 return code;
102
103 test_case_count = cvReadInt( find_param( fs, "test_case_count" ), test_case_count );
104 min_log_point_count = cvReadInt( find_param( fs, "min_log_point_count" ), min_log_point_count );
105 max_log_point_count = cvReadInt( find_param( fs, "max_log_point_count" ), max_log_point_count );
106 min_log_img_size = cvReadInt( find_param( fs, "min_log_img_size" ), min_log_img_size );
107 max_log_img_size = cvReadInt( find_param( fs, "max_log_img_size" ), max_log_img_size );
108
109 min_log_point_count = cvtest::clipInt( min_log_point_count, 1, 10 );
110 max_log_point_count = cvtest::clipInt( max_log_point_count, 1, 10 );
111 if( min_log_point_count > max_log_point_count )
112 CV_SWAP( min_log_point_count, max_log_point_count, t );
113
114 min_log_img_size = cvtest::clipInt( min_log_img_size, 1, 10 );
115 max_log_img_size = cvtest::clipInt( max_log_img_size, 1, 10 );
116 if( min_log_img_size > max_log_img_size )
117 CV_SWAP( min_log_img_size, max_log_img_size, t );
118
119 return 0;
120}

123int CV_SubdivTest::prepare_test_case( int test_case_idx )
124{
125 RNG& rng = ts->get_rng();
126 int code = cvtest::BaseTest::prepare_test_case( test_case_idx );
127 if( code < 0 )
128 return code;
129
130 clear();
131
132 point_count = cvRound(exp((cvtest::randReal(rng)*
133 (max_log_point_count - min_log_point_count) + min_log_point_count)*CV_LOG2));
134 img_size.width = cvRound(exp((cvtest::randReal(rng)*
135 (max_log_img_size - min_log_img_size) + min_log_img_size)*CV_LOG2));
136 img_size.height = cvRound(exp((cvtest::randReal(rng)*
137 (max_log_img_size - min_log_img_size) + min_log_img_size)*CV_LOG2));
138
139 storage = cvCreateMemStorage( 1 << 10 );
140 return 1;
141}

149static inline double sqdist( CvPoint2D32f pt1, CvPoint2D32f pt2 )
150{
151 double dx = pt1.x - pt2.x;
152 double dy = pt1.y - pt2.y;
153
154 return dx*dx + dy*dy;
155}

158static int
159subdiv2DCheck( CvSubdiv2D* subdiv )
160{
161 int i, j, total = subdiv->edges->total;
162 CV_Assert( subdiv != 0 );
163
164 for( i = 0; i < total; i++ )
165 {
166 CvQuadEdge2D* edge = (CvQuadEdge2D*)cvGetSetElem(subdiv->edges,i);
167
168 if( edge && CV_IS_SET_ELEM( edge ))
169 {
170 for( j = 0; j < 4; j++ )
171 {
172 CvSubdiv2DEdge e = (CvSubdiv2DEdge)edge + j;
173 CvSubdiv2DEdge o_next = cvSubdiv2DNextEdge(e);
174 CvSubdiv2DEdge o_prev = cvSubdiv2DGetEdge(e, CV_PREV_AROUND_ORG );
175 CvSubdiv2DEdge d_prev = cvSubdiv2DGetEdge(e, CV_PREV_AROUND_DST );
176 CvSubdiv2DEdge d_next = cvSubdiv2DGetEdge(e, CV_NEXT_AROUND_DST );
177
178 // check points
179 if( cvSubdiv2DEdgeOrg(e) != cvSubdiv2DEdgeOrg(o_next))
180 return 0;
181 if( cvSubdiv2DEdgeOrg(e) != cvSubdiv2DEdgeOrg(o_prev))
182 return 0;
183 if( cvSubdiv2DEdgeDst(e) != cvSubdiv2DEdgeDst(d_next))
184 return 0;
185 if( cvSubdiv2DEdgeDst(e) != cvSubdiv2DEdgeDst(d_prev))
186 return 0;
187 if( j % 2 == 0 )
188 {
189 if( cvSubdiv2DEdgeDst(o_next) != cvSubdiv2DEdgeOrg(d_prev))
190 return 0;
191 if( cvSubdiv2DEdgeDst(o_prev) != cvSubdiv2DEdgeOrg(d_next))
192 return 0;
193 if( cvSubdiv2DGetEdge(cvSubdiv2DGetEdge(cvSubdiv2DGetEdge(
194 e,CV_NEXT_AROUND_LEFT),CV_NEXT_AROUND_LEFT),CV_NEXT_AROUND_LEFT) != e )
195 return 0;
196 if( cvSubdiv2DGetEdge(cvSubdiv2DGetEdge(cvSubdiv2DGetEdge(
197 e,CV_NEXT_AROUND_RIGHT),CV_NEXT_AROUND_RIGHT),CV_NEXT_AROUND_RIGHT) != e)
198 return 0;
199 }
200 }
201 }
202 }
203
204 return 1;
205}

208// the whole testing is done here, run_func() is not utilized in this test
209int CV_SubdivTest::validate_test_results( int /*test_case_idx*/ )
210{
211 int code = cvtest::TS::OK;
212 RNG& rng = ts->get_rng();
213 int j, k, real_count = point_count;
214 double xrange = img_size.width*(1 - FLT_EPSILON);
215 double yrange = img_size.height*(1 - FLT_EPSILON);
216
217 subdiv = cvCreateSubdivDelaunay2D(
218 cvRect( 0, 0, img_size.width, img_size.height ), storage );
219
220 CvSeq* seq = cvCreateSeq( 0, sizeof(*seq), sizeof(CvPoint2D32f), storage );
221 CvSeqWriter writer;
222 cvStartAppendToSeq( seq, &writer );
223
224 // insert random points
225 for( j = 0; j < point_count; j++ )
226 {
227 CvPoint2D32f pt;
228 CvSubdiv2DPoint* point;
229
230 pt.x = (float)(cvtest::randReal(rng)*xrange);
231 pt.y = (float)(cvtest::randReal(rng)*yrange);
232
233 CvSubdiv2DPointLocation loc =
234 cvSubdiv2DLocate( subdiv, pt, 0, &point );
235
236 if( loc == CV_PTLOC_VERTEX )
237 {
238 int index = cvSeqElemIdx( (CvSeq*)subdiv, point );
239 CvPoint2D32f* pt1;
240 cvFlushSeqWriter( &writer );
241 pt1 = (CvPoint2D32f*)cvGetSeqElem( seq, index - 3 );
242
243 if( !pt1 ||
244 fabs(pt1->x - pt.x) > FLT_EPSILON ||
245 fabs(pt1->y - pt.y) > FLT_EPSILON )
246 {
247 ts->printf( cvtest::TS::LOG, "The point #%d: (%.1f,%.1f) is said to coinside with a subdivision vertex, "
248 "however it could be found in a sequence of inserted points\n", j, pt.x, pt.y );
249 code = cvtest::TS::FAIL_INVALID_OUTPUT;
250 goto _exit_;
251 }
252 real_count--;
253 }
254
255 point = cvSubdivDelaunay2DInsert( subdiv, pt );
256 if( point->pt.x != pt.x || point->pt.y != pt.y )
257 {
258 ts->printf( cvtest::TS::LOG, "The point #%d: (%.1f,%.1f) has been incorrectly added\n", j, pt.x, pt.y );
259 code = cvtest::TS::FAIL_INVALID_OUTPUT;
260 goto _exit_;
261 }
262
263 if( (j + 1) % 10 == 0 || j == point_count - 1 )
264 {
265 if( !subdiv2DCheck( subdiv ))
266 {
267 ts->printf( cvtest::TS::LOG, "Subdivision consistency check failed after inserting the point #%d\n", j );
268 code = cvtest::TS::FAIL_INVALID_OUTPUT;
269 goto _exit_;
270 }
271 }
272
273 if( loc != CV_PTLOC_VERTEX )
274 {
275 CV_WRITE_SEQ_ELEM( pt, writer );
276 }
277 }
278
279 if( code < 0 )
280 goto _exit_;
281
282 cvCalcSubdivVoronoi2D( subdiv );
283 seq = cvEndWriteSeq( &writer );
284
285 if( !subdiv2DCheck( subdiv ))
286 {
287 ts->printf( cvtest::TS::LOG, "The subdivision failed consistency check after building the Voronoi tesselation\n" );
288 code = cvtest::TS::FAIL_INVALID_OUTPUT;
289 goto _exit_;
290 }
291
292 for( j = 0; j < MAX((point_count - 5)/10 + 5, 10); j++ )
293 {
294 CvPoint2D32f pt;
295 double minDistance;
296
297 pt.x = (float)(cvtest::randReal(rng)*xrange);
298 pt.y = (float)(cvtest::randReal(rng)*yrange);
299
300 CvSubdiv2DPoint* point = cvFindNearestPoint2D( subdiv, pt );
301 CvSeqReader reader;
302
303 if( !point )
304 {
305 ts->printf( cvtest::TS::LOG, "There is no nearest point (?!) for the point (%.1f, %.1f) in the subdivision\n",
306 pt.x, pt.y );
307 code = cvtest::TS::FAIL_INVALID_OUTPUT;
308 goto _exit_;
309 }
310
311 cvStartReadSeq( seq, &reader );
312 minDistance = sqdist( pt, point->pt );
313
314 for( k = 0; k < seq->total; k++ )
315 {
316 CvPoint2D32f ptt;
317 CV_READ_SEQ_ELEM( ptt, reader );
318
319 double distance = sqdist( pt, ptt );
320 if( minDistance > distance && sqdist(ptt, point->pt) > FLT_EPSILON*1000 )
321 {
322 ts->printf( cvtest::TS::LOG, "The triangulation vertex (%.3f,%.3f) was said to be nearest to (%.3f,%.3f),\n"
323 "whereas another vertex (%.3f,%.3f) is closer\n",
324 point->pt.x, point->pt.y, pt.x, pt.y, ptt.x, ptt.y );
325 code = cvtest::TS::FAIL_BAD_ACCURACY;
326 goto _exit_;
327 }
328 }
329 }
330
331_exit_:
332 if( code < 0 )
333 ts->set_failed_test_info( code );
334
335 return code;
336}
