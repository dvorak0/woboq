5struct CV_EXPORTS L2Fake : public L2<float>

7 enum { normType = NORM_L2 };

10class CV_BruteForceMatcherTest : public cvtest::BaseTest
11{
12public:
13 CV_BruteForceMatcherTest() {}
14protected:
15 void run( int )
16 {
17 const int dimensions = 64;
18 const int descriptorsNumber = 5000;
19
20 Mat train = Mat( descriptorsNumber, dimensions, CV_32FC1);
21 Mat query = Mat( descriptorsNumber, dimensions, CV_32FC1);
22
23 Mat permutation( 1, descriptorsNumber, CV_32SC1 );
24 for( int i=0;i<descriptorsNumber;i++ )
25 permutation.at<int>( 0, i ) = i;
26
27 //RNG rng = RNG( cvGetTickCount() );
28 RNG rng;
29 randShuffle( permutation, 1, &rng );
30
31 float boundary = 500.f;
32 for( int row=0;row<descriptorsNumber;row++ )
33 {
34 for( int col=0;col<dimensions;col++ )
35 {
36 int bit = rng( 2 );
37 train.at<float>( permutation.at<int>( 0, row ), col ) = bit*boundary + rng.uniform( 0.f, boundary );
38 query.at<float>( row, col ) = bit*boundary + rng.uniform( 0.f, boundary );
39 }
40 }
41
42 vector<DMatch> specMatches, genericMatches;
43 BruteForceMatcher<L2<float> > specMatcher;
44 BruteForceMatcher<L2Fake > genericMatcher;
45
46 int64 time0 = cvGetTickCount();
47 specMatcher.match( query, train, specMatches );
48 int64 time1 = cvGetTickCount();
49 genericMatcher.match( query, train, genericMatches );
50 int64 time2 = cvGetTickCount();
51
52 float specMatcherTime = float(time1 - time0)/(float)cvGetTickFrequency();
53 ts->printf( cvtest::TS::LOG, "Matching by matrix multiplication time s: %f, us per pair: %f\n",
54 specMatcherTime*1e-6, specMatcherTime/( descriptorsNumber*descriptorsNumber ) );
55
56 float genericMatcherTime = float(time2 - time1)/(float)cvGetTickFrequency();
57 ts->printf( cvtest::TS::LOG, "Matching without matrix multiplication time s: %f, us per pair: %f\n",
58 genericMatcherTime*1e-6, genericMatcherTime/( descriptorsNumber*descriptorsNumber ) );
59
60 if( (int)specMatches.size() != descriptorsNumber || (int)genericMatches.size() != descriptorsNumber )
61 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_OUTPUT );
62 for( int i=0;i<descriptorsNumber;i++ )
63 {
64 float epsilon = 0.01f;
65 bool isEquiv = fabs( specMatches[i].distance - genericMatches[i].distance ) < epsilon &&
66 specMatches[i].queryIdx == genericMatches[i].queryIdx &&
67 specMatches[i].trainIdx == genericMatches[i].trainIdx;
68 if( !isEquiv || specMatches[i].trainIdx != permutation.at<int>( 0, i ) )
69 {
70 ts->set_failed_test_info( cvtest::TS::FAIL_MISMATCH );
71 break;
72 }
73 }
74
75
76 //Test mask
77 Mat mask( query.rows, train.rows, CV_8UC1 );
78 rng.fill( mask, RNG::UNIFORM, 0, 2 );
79
80
81 time0 = cvGetTickCount();
82 specMatcher.match( query, train, specMatches, mask );
83 time1 = cvGetTickCount();
84 genericMatcher.match( query, train, genericMatches, mask );
85 time2 = cvGetTickCount();
86
87 specMatcherTime = float(time1 - time0)/(float)cvGetTickFrequency();
88 ts->printf( cvtest::TS::LOG, "Matching by matrix multiplication time with mask s: %f, us per pair: %f\n",
89 specMatcherTime*1e-6, specMatcherTime/( descriptorsNumber*descriptorsNumber ) );
90
91 genericMatcherTime = float(time2 - time1)/(float)cvGetTickFrequency();
92 ts->printf( cvtest::TS::LOG, "Matching without matrix multiplication time with mask s: %f, us per pair: %f\n",
93 genericMatcherTime*1e-6, genericMatcherTime/( descriptorsNumber*descriptorsNumber ) );
94
95 if( specMatches.size() != genericMatches.size() )
96 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_OUTPUT );
97
98 for( size_t i=0;i<specMatches.size();i++ )
99 {
100 //float epsilon = 1e-2;
101 float epsilon = 10000000;
102 bool isEquiv = fabs( specMatches[i].distance - genericMatches[i].distance ) < epsilon &&
103 specMatches[i].queryIdx == genericMatches[i].queryIdx &&
104 specMatches[i].trainIdx == genericMatches[i].trainIdx;
105 if( !isEquiv )
106 {
107 ts->set_failed_test_info( cvtest::TS::FAIL_MISMATCH );
108 break;
109 }
110 }
111 }
