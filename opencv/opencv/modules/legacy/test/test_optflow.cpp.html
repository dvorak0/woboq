71Mat copnvert2flow(const Mat& velx, const Mat& vely)
72{
73 Mat flow(velx.size(), CV_32FC2);
74 for(int y = 0 ; y < flow.rows; ++y)
75 for(int x = 0 ; x < flow.cols; ++x)
76 flow.at<Point2f>(y, x) = Point2f(velx.at<float>(y, x), vely.at<float>(y, x));
77 return flow;
78}
79
80void calcOpticalFlowLK( const Mat& prev, const Mat& curr, Size winSize, Mat& flow )
81{
82 Mat velx(prev.size(), CV_32F), vely(prev.size(), CV_32F);
83 CvMat cvvelx = velx; CvMat cvvely = vely;
84 CvMat cvprev = prev; CvMat cvcurr = curr;
85 cvCalcOpticalFlowLK( &cvprev, &cvcurr, winSize, &cvvelx, &cvvely );
86 flow = copnvert2flow(velx, vely);
87}
88
89void calcOpticalFlowBM( const Mat& prev, const Mat& curr, Size bSize, Size shiftSize, Size maxRange, int usePrevious, Mat& flow )
90{
91 Size sz((curr.cols - bSize.width + shiftSize.width)/shiftSize.width, (curr.rows - bSize.height + shiftSize.height)/shiftSize.height);
92 Mat velx(sz, CV_32F), vely(sz, CV_32F);
93
94 CvMat cvvelx = velx; CvMat cvvely = vely;
95 CvMat cvprev = prev; CvMat cvcurr = curr;
96 cvCalcOpticalFlowBM( &cvprev, &cvcurr, bSize, shiftSize, maxRange, usePrevious, &cvvelx, &cvvely);
97 flow = copnvert2flow(velx, vely);
98}

100void calcOpticalFlowHS( const Mat& prev, const Mat& curr, int usePrevious, double lambda, TermCriteria criteria, Mat& flow)
101{
102 Mat velx(prev.size(), CV_32F), vely(prev.size(), CV_32F);
103 CvMat cvvelx = velx; CvMat cvvely = vely;
104 CvMat cvprev = prev; CvMat cvcurr = curr;
105 cvCalcOpticalFlowHS( &cvprev, &cvcurr, usePrevious, &cvvelx, &cvvely, lambda, criteria );
106 flow = copnvert2flow(velx, vely);
107}

109void calcAffineFlowPyrLK( const Mat& prev, const Mat& curr,
110 const vector<Point2f>& prev_features, vector<Point2f>& curr_features,
111 vector<uchar>& status, vector<float>& track_error, vector<float>& matrices,
112 TermCriteria criteria = TermCriteria(TermCriteria::COUNT+TermCriteria::EPS,30, 0.01),
113 Size win_size = Size(15, 15), int level = 3, int flags = 0)
114{
115 CvMat cvprev = prev;
116 CvMat cvcurr = curr;
117
118 size_t count = prev_features.size();
119 curr_features.resize(count);
120 status.resize(count);
121 track_error.resize(count);
122 matrices.resize(count * 6);
123
124 cvCalcAffineFlowPyrLK( &cvprev, &cvcurr, 0, 0,
125 (const CvPoint2D32f*)&prev_features[0], (CvPoint2D32f*)&curr_features[0], &matrices[0],
126 (int)count, win_size, level, (char*)&status[0], &track_error[0], criteria, flags );
127}

129double showFlowAndCalcError(const string& name, const Mat& gray, const Mat& flow,
130 const Rect& where, const Point& d,
131 bool showImages = false, bool writeError = false)
132{
133 const int mult = 16;
134
135 if (showImages)
136 {
137 Mat tmp, cflow;
138 resize(gray, tmp, gray.size() * mult, 0, 0, INTER_NEAREST);
139 cvtColor(tmp, cflow, CV_GRAY2BGR);
140
141 const float m2 = 0.3f;
142 const float minVel = 0.1f;
143
144 for(int y = 0; y < flow.rows; ++y)
145 for(int x = 0; x < flow.cols; ++x)
146 {
147 Point2f f = flow.at<Point2f>(y, x);
148
149 if (f.x * f.x + f.y * f.y > minVel * minVel)
150 {
151 Point p1 = Point(x, y) * mult;
152 Point p2 = Point(cvRound((x + f.x*m2) * mult), cvRound((y + f.y*m2) * mult));
153
154 line(cflow, p1, p2, CV_RGB(0, 255, 0));
155 circle(cflow, Point(x, y) * mult, 2, CV_RGB(255, 0, 0));
156 }
157 }
158
159 rectangle(cflow, (where.tl() + d) * mult, (where.br() + d - Point(1,1)) * mult, CV_RGB(0, 0, 255));
160 namedWindow(name, 1); imshow(name, cflow);
161 }
162
163 double angle = atan2((float)d.y, (float)d.x);
164 double error = 0;
165
166 bool all = true;
167 Mat inner = flow(where);
168 for(int y = 0; y < inner.rows; ++y)
169 {
170 for(int x = 0; x < inner.cols; ++x)
171 {
172 const Point2f f = inner.at<Point2f>(y, x);
173
174 if (f.x == 0 && f.y == 0)
175 continue;
176
177 all = false;
178
179 double a = atan2(f.y, f.x);
180 error += fabs(angle - a);
181 }
182 }
183
184 double res = all ? numeric_limits<double>::max() : error / (inner.cols * inner.rows);
185
186 if (writeError)
187 cout << "Error " + name << " = " << res << endl;
188
189 return res;
190}

193Mat generateImage(const Size& sz, bool doBlur = true)
194{
195 RNG rng;
196 Mat mat(sz, CV_8U);
197 mat = Scalar(0);
198 for(int y = 0; y < mat.rows; ++y)
199 for(int x = 0; x < mat.cols; ++x)
200 mat.at<uchar>(y, x) = (uchar)rng;
201 if (doBlur)
202 blur(mat, mat, Size(3, 3));
203 return mat;
204}
205
206Mat generateSample(const Size& sz)
207{
208 Mat smpl(sz, CV_8U);
209 smpl = Scalar(0);
210 Point sc(smpl.cols/2, smpl.rows/2);
211 rectangle(smpl, Point(0,0), sc - Point(1,1), Scalar(255), CV_FILLED);
212 rectangle(smpl, sc, Point(smpl.cols, smpl.rows), Scalar(255), CV_FILLED);
213 return smpl;
214}

216bool CV_OptFlowTest::runDense(const Point& d)
217{
218 Size matSize(40, 40);
219 Size movSize(8, 8);
220
221 Mat smpl = generateSample(movSize);
222 Mat prev = generateImage(matSize);
223 Mat curr = prev.clone();
224
225 Rect rect(Point(prev.cols/2, prev.rows/2) - Point(movSize.width/2, movSize.height/2), movSize);
226
227 Mat flowLK, flowBM, flowHS, flowFB, flowFB_G, flowBM_received, m1;
228
229 m1 = prev(rect); smpl.copyTo(m1);
230 m1 = curr(Rect(rect.tl() + d, rect.br() + d)); smpl.copyTo(m1);
231
232 calcOpticalFlowLK( prev, curr, Size(15, 15), flowLK);
233 calcOpticalFlowBM( prev, curr, Size(15, 15), Size(1, 1), Size(15, 15), 0, flowBM_received);
234 calcOpticalFlowHS( prev, curr, 0, 5, TermCriteria(TermCriteria::MAX_ITER, 400, 0), flowHS);
235 calcOpticalFlowFarneback( prev, curr, flowFB, 0.5, 3, std::max(d.x, d.y) + 10, 100, 6, 2, 0);
236 calcOpticalFlowFarneback( prev, curr, flowFB_G, 0.5, 3, std::max(d.x, d.y) + 10, 100, 6, 2, OPTFLOW_FARNEBACK_GAUSSIAN);
237
238 flowBM.create(prev.size(), CV_32FC2);
239 flowBM = Scalar(0);
240 Point origin((flowBM.cols - flowBM_received.cols)/2, (flowBM.rows - flowBM_received.rows)/2);
241 Mat wcp = flowBM(Rect(origin, flowBM_received.size()));
242 flowBM_received.copyTo(wcp);
243
244 double errorLK = showFlowAndCalcError("LK", prev, flowLK, rect, d);
245 double errorBM = showFlowAndCalcError("BM", prev, flowBM, rect, d);
246 double errorFB = showFlowAndCalcError("FB", prev, flowFB, rect, d);
247 double errorFBG = showFlowAndCalcError("FBG", prev, flowFB_G, rect, d);
248 double errorHS = showFlowAndCalcError("HS", prev, flowHS, rect, d); (void)errorHS;
249 //waitKey();
250
251 const double thres = 0.2;
252 if (errorLK > thres || errorBM > thres || errorFB > thres || errorFBG > thres /*|| errorHS > thres */)
253 {
254 ts->set_failed_test_info(cvtest::TS::FAIL_MISMATCH);
255 return false;
256 }
257 return true;
258}

261bool CV_OptFlowTest::runSparse()
262{
263 Mat prev = imread(string(ts->get_data_path()) + "optflow/rock_1.bmp", 0);
264 Mat next = imread(string(ts->get_data_path()) + "optflow/rock_2.bmp", 0);
265
266 if (prev.empty() || next.empty())
267 {
268 ts->set_failed_test_info( cvtest::TS::FAIL_INVALID_TEST_DATA );
269 return false;
270 }
271
272 Mat cprev, cnext;
273 cvtColor(prev, cprev, CV_GRAY2BGR);
274 cvtColor(next, cnext, CV_GRAY2BGR);
275
276 vector<Point2f> prev_pts;
277 vector<Point2f> next_ptsOpt;
278 vector<Point2f> next_ptsAff;
279 vector<uchar> status_Opt;
280 vector<uchar> status_Aff;
281 vector<float> error;
282 vector<float> matrices;
283
284 Size netSize(10, 10);
285 Point2f center = Point(prev.cols/2, prev.rows/2);
286
287 for(int i = 0 ; i < netSize.width; ++i)
288 for(int j = 0 ; j < netSize.width; ++j)
289 {
290 Point2f p(i * float(prev.cols)/netSize.width, j * float(prev.rows)/netSize.height);
291 prev_pts.push_back((p - center) * 0.5f + center);
292 }
293
294 calcOpticalFlowPyrLK( prev, next, prev_pts, next_ptsOpt, status_Opt, error );
295 calcAffineFlowPyrLK ( prev, next, prev_pts, next_ptsAff, status_Aff, error, matrices);
296
297 const double expected_shift = 25;
298 const double thres = 1;
299 for(size_t i = 0; i < prev_pts.size(); ++i)
300 {
301 circle(cprev, prev_pts[i], 2, CV_RGB(255, 0, 0));
302
303 if (status_Opt[i])
304 {
305 circle(cnext, next_ptsOpt[i], 2, CV_RGB(0, 0, 255));
306 Point2f shift = prev_pts[i] - next_ptsOpt[i];
307
308 double n = sqrt(shift.ddot(shift));
309 if (fabs(n - expected_shift) > thres)
310 {
311 ts->set_failed_test_info(cvtest::TS::FAIL_MISMATCH);
312 return false;
313 }
314 }
315
316 if (status_Aff[i])
317 {
318 circle(cnext, next_ptsAff[i], 4, CV_RGB(0, 255, 0));
319 Point2f shift = prev_pts[i] - next_ptsAff[i];
320
321 double n = sqrt(shift.ddot(shift));
322 if (fabs(n - expected_shift) > thres)
323 {
324 ts->set_failed_test_info(cvtest::TS::FAIL_MISMATCH);
325 return false;
326 }
327 }
328
329 }
330
331 /*namedWindow("P"); imshow("P", cprev);
332 namedWindow("N"); imshow("N", cnext);
333 waitKey();*/
334
335 return true;
336}

339void CV_OptFlowTest::run( int /* start_from */)
340{
341
342 if (!runDense(Point(3, 0)))
343 return;
344
345 if (!runDense(Point(0, 3)))
346 return;
347
348 //if (!runDense(Point(3, 3))) return; //probably LK works incorrectly in this case.
349
350 if (!runSparse())
351 return;
352
353 ts->set_failed_test_info(cvtest::TS::OK);
354}
