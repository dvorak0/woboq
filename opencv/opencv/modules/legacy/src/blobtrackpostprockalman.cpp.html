111CvBlobTrackPostProcKalman::CvBlobTrackPostProcKalman()
112{
113 m_ModelNoise = 1e-6f;
114 m_DataNoisePos = 1e-6f;
115 m_DataNoiseSize = 1e-1f;
116
117 #if STATE_NUM>6
118 m_DataNoiseSize *= (float)pow(20.,2.);
119 #else
120 m_DataNoiseSize /= (float)pow(20.,2.);
121 #endif
122
123 AddParam("ModelNoise",&m_ModelNoise);
124 AddParam("DataNoisePos",&m_DataNoisePos);
125 AddParam("DataNoiseSize",&m_DataNoiseSize);
126
127 m_Frame = 0;
128 m_pKalman = cvCreateKalman(STATE_NUM,4);
129 memcpy( m_pKalman->transition_matrix->data.fl, A, sizeof(A));
130 memcpy( m_pKalman->measurement_matrix->data.fl, H, sizeof(H));
131
132 cvSetIdentity( m_pKalman->process_noise_cov, cvRealScalar(m_ModelNoise) );
133 cvSetIdentity( m_pKalman->measurement_noise_cov, cvRealScalar(m_DataNoisePos) );
134 CV_MAT_ELEM(*m_pKalman->measurement_noise_cov, float, 2,2) = m_DataNoiseSize;
135 CV_MAT_ELEM(*m_pKalman->measurement_noise_cov, float, 3,3) = m_DataNoiseSize;
136 cvSetIdentity( m_pKalman->error_cov_post, cvRealScalar(1));
137 cvZero(m_pKalman->state_post);
138 cvZero(m_pKalman->state_pre);
139
140 SetModuleName("Kalman");
141}

156CvBlob* CvBlobTrackPostProcKalman::Process(CvBlob* pBlob)
157{
158 CvBlob* pBlobRes = &m_Blob;
159 float Z[4];
160 CvMat Zmat = cvMat(4,1,CV_32F,Z);
161 m_Blob = pBlob[0];
162
163 if(m_Frame < 2)
164 { /* First call: */
165 m_pKalman->state_post->data.fl[0+4] = CV_BLOB_X(pBlob)-m_pKalman->state_post->data.fl[0];
166 m_pKalman->state_post->data.fl[1+4] = CV_BLOB_Y(pBlob)-m_pKalman->state_post->data.fl[1];
167 if(m_pKalman->DP>6)
168 {
169 m_pKalman->state_post->data.fl[2+4] = CV_BLOB_WX(pBlob)-m_pKalman->state_post->data.fl[2];
170 m_pKalman->state_post->data.fl[3+4] = CV_BLOB_WY(pBlob)-m_pKalman->state_post->data.fl[3];
171 }
172 m_pKalman->state_post->data.fl[0] = CV_BLOB_X(pBlob);
173 m_pKalman->state_post->data.fl[1] = CV_BLOB_Y(pBlob);
174 m_pKalman->state_post->data.fl[2] = CV_BLOB_WX(pBlob);
175 m_pKalman->state_post->data.fl[3] = CV_BLOB_WY(pBlob);
176 }
177 else
178 { /* Nonfirst call: */
179 cvKalmanPredict(m_pKalman,0);
180 Z[0] = CV_BLOB_X(pBlob);
181 Z[1] = CV_BLOB_Y(pBlob);
182 Z[2] = CV_BLOB_WX(pBlob);
183 Z[3] = CV_BLOB_WY(pBlob);
184 cvKalmanCorrect(m_pKalman,&Zmat);
185 cvMatMulAdd(m_pKalman->measurement_matrix, m_pKalman->state_post, NULL, &Zmat);
186 CV_BLOB_X(pBlobRes) = Z[0];
187 CV_BLOB_Y(pBlobRes) = Z[1];
188// CV_BLOB_WX(pBlobRes) = Z[2];
189// CV_BLOB_WY(pBlobRes) = Z[3];
190 }
191 m_Frame++;
192 return pBlobRes;
193}

283CvBlob* CvBlobTrackPredictKalman::Predict()
284{
285 if(m_Frame >= 2)
286 {
287 cvKalmanPredict(m_pKalman,0);
288 m_BlobPredict.x = m_pKalman->state_pre->data.fl[0];
289 m_BlobPredict.y = m_pKalman->state_pre->data.fl[1];
