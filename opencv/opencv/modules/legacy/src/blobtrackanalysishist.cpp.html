108 void AddBlob(CvBlob* pBlob)
109 {
110 float FV[MAX_FV_SIZE+1];
111 DefBlobFVN* pFVBlob = (DefBlobFVN*)m_BlobList.GetBlobByID(CV_BLOB_ID(pBlob));
112
113 if(!m_ClearFlag) Clear();
114
115 if(pFVBlob==NULL)
116 {
117 DefBlobFVN BlobNew;
118 BlobNew.blob = pBlob[0];
119 BlobNew.LastFrame = m_Frame;
120 BlobNew.state = 0;;
121 BlobNew.FrameNum = 0;
122 m_BlobList.AddBlob((CvBlob*)&BlobNew);
123 pFVBlob = (DefBlobFVN*)m_BlobList.GetBlobByID(CV_BLOB_ID(pBlob));
124 } /* Add new record if necessary. */
125
126 pFVBlob->blob = pBlob[0];
127
128 /* Shift: */
129 for(int i=(BLOB_NUM-1); i>0; --i)
130 {
131 pFVBlob->BlobSeq[i] = pFVBlob->BlobSeq[i-1];
132 }
133
134 pFVBlob->BlobSeq[0] = pBlob[0];
135
136 if(m_Dim>0)
137 { /* Calculate FV position: */
138 FV[0] = CV_BLOB_X(pBlob);
139 FV[1] = CV_BLOB_Y(pBlob);
140 }
141
142 if(m_Dim<=2)
143 { /* Add new FV if position is enough: */
144 *(int*)(FV+m_Dim) = CV_BLOB_ID(pBlob);
145 cvSeqPush( m_pFVSeq, FV );
146 }
147 else if(pFVBlob->FrameNum > BLOB_NUM)
148 { /* Calculate velocity for more complex FV: */
149 float AverVx = 0;
150 float AverVy = 0;
151 { /* Average velocity: */
152 CvBlob* pBlobSeq = pFVBlob->BlobSeq;
153 for(int i=1;i<BLOB_NUM;++i)
154 {
155 AverVx += CV_BLOB_X(pBlobSeq+i-1)-CV_BLOB_X(pBlobSeq+i);
156 AverVy += CV_BLOB_Y(pBlobSeq+i-1)-CV_BLOB_Y(pBlobSeq+i);
157 }
158 AverVx /= BLOB_NUM-1;
159 AverVy /= BLOB_NUM-1;
160
161 FV[2] = AverVx;
162 FV[3] = AverVy;
163 }
164
165 if(m_Dim>4)
166 { /* State duration: */
167 float T = (CV_BLOB_WX(pBlob)+CV_BLOB_WY(pBlob))*0.01f;
168
169 if( fabs(AverVx) < T && fabs(AverVy) < T)
170 pFVBlob->state++;
171 else
172 pFVBlob->state=0;
173 FV[4] = (float)pFVBlob->state;
174 } /* State duration. */
175
176 /* Add new FV: */
177 *(int*)(FV+m_Dim) = CV_BLOB_ID(pBlob);
178 cvSeqPush( m_pFVSeq, FV );
179
180 } /* If velocity is calculated. */
181
182 pFVBlob->FrameNum++;
183 pFVBlob->LastFrame = m_Frame;
184 }; /* AddBlob */

186 void Process(IplImage* pImg, IplImage* /*pFG*/)
187 {
188 int i;
189 if(!m_ClearFlag) Clear();
190 for(i=m_BlobList.GetBlobNum(); i>0; --i)
191 { /* Delete unused blob: */
192 DefBlobFVN* pFVBlob = (DefBlobFVN*)m_BlobList.GetBlob(i-1);
193 if(pFVBlob->LastFrame < m_Frame)
194 {
195 m_BlobList.DelBlob(i-1);
196 }
197 } /* Check next blob in list. */
198
199 m_FVMin[0] = 0;
200 m_FVMin[1] = 0;
201 m_FVMax[0] = (float)(pImg->width-1);
202 m_FVMax[1] = (float)(pImg->height-1);
203 m_FVVar[0] = m_FVMax[0]*0.01f;
204 m_FVVar[1] = m_FVMax[1]*0.01f;
205 m_FVVar[2] = (float)(pImg->width-1)/1440.0f;
206 m_FVMax[2] = (float)(pImg->width-1)*0.02f;
207 m_FVMin[2] = -m_FVMax[2];
208 m_FVVar[3] = (float)(pImg->width-1)/1440.0f;
209 m_FVMax[3] = (float)(pImg->height-1)*0.02f;
210 m_FVMin[3] = -m_FVMax[3];
211 m_FVMax[4] = 25*32.0f; /* max state is 32 sec */
212 m_FVMin[4] = 0;
213 m_FVVar[4] = 10;
214
215 m_Frame++;
216 m_ClearFlag = 0;
217 };

822 virtual void Process(IplImage* pImg, IplImage* pFG)
823 {
824 m_pFVGen->Process(pImg, pFG);
825 int SK = m_SmoothKernel[0];
826
827 for(int i=0; i<m_pFVGen->GetFVNum(); ++i)
828 {
829 int BlobID = 0;
830 float* pFV = m_pFVGen->GetFV(i,&BlobID);
831 float* pFVMax = m_pFVGen->GetFVMax();
832 float* pFVMin = m_pFVGen->GetFVMin();
833 DefTrackFG* pF = (DefTrackFG*)m_TrackFGList.GetBlobByID(BlobID);
834 int HistVal = 1;
835
836 if(pFV==NULL) break;
837
838 pF->LastFrame = m_Frame;
839
840 { /* Binarize FV: */
841 int j;
842 for(j=0; j<m_Dim; ++j)
843 {
844 int index;
845 float f0 = pFVMin?pFVMin[j]:0;
846 float f1 = pFVMax?pFVMax[j]:1;
847 assert(f1>f0);
848 index = cvRound((m_BinNum-1)*(pFV[j]-f0)/(f1-f0));
849 if(index<0)index=0;
850 if(index>=m_BinNum)index=m_BinNum-1;
851 m_pFVi[j] = index;
852 }
853 }
854
855 HistVal = m_HistMat.GetVal(m_pFVi);/* get bin value*/
856 pF->state = 0;
857 { /* Calculate state: */
858 float T = m_HistMat.m_Max*m_AbnormalThreshold; /* calc threshold */
859
860 if(m_TrackNum>0) T = 256.0f * m_TrackNum*m_AbnormalThreshold;
861 if(T>0)
862 {
863 pF->state = (T - HistVal)/(T*0.2f) + 0.5f;
864 }
865 if(pF->state<0)pF->state=0;
866 if(pF->state>1)pF->state=1;
867 }
868
869 { /* If it is a new FV then add it to trajectory histogram: */
870 int flag = 1;
871 int r = m_SmoothRadius;
872
873 // printf("BLob %3d NEW FV [", CV_BLOB_ID(pF));
874 // for(i=0;i<m_Dim;++i) printf("%d,", m_pFVi[i]);
875 // printf("]");
876
877 for(int k=0; k<m_Dim; ++k)
878 {
879 m_pFViVar[k]=-r;
880 }
881
882 while(flag)
883 {
884 float dist = 0;
885 int HistAdd = 0;
886 int good = 1;
887 for(int k=0; k<m_Dim; ++k)
888 {
889 m_pFViVarRes[k] = m_pFVi[k]+m_pFViVar[k];
890 if(m_pFViVarRes[k]<0) good= 0;
891 if(m_pFViVarRes[k]>=m_BinNum) good= 0;
892 dist += m_pFViVar[k]*m_pFViVar[k];
893 }/* Calculate next dimension. */
894
895 if(SK=='G' || SK=='g')
896 {
897 double dist2 = dist/(r*r);
898 HistAdd = cvRound(256*exp(-dist2)); /* Hist Add for (dist=1) = 25.6*/
899 }
900 else if(SK=='L' || SK=='l')
901 {
902 dist = (float)(sqrt(dist)/(r+1));
903 HistAdd = cvRound(256*(1-dist));
904 }
905 else
906 {
907 HistAdd = 255; /* Flat smoothing. */
908 }
909
910 if(good && HistAdd>0)
911 { /* Update histogram: */
912 assert(pF->pHist);
913 pF->pHist->SetMax(m_pFViVarRes, HistAdd);
914 } /* Update histogram. */
915
916 int idx = 0;
917 for( ; idx<m_Dim; ++idx)
918 { /* Next config: */
919 if((m_pFViVar[idx]++) < r)
920 break;
921 m_pFViVar[idx] = -r;
922 } /* Increase next dimension variable. */
923 if(idx==m_Dim)break;
924 } /* Next variation. */
925 } /* If new FV. */
926 } /* Next FV. */
927
928 { /* Check all blobs on list: */
929 int i;
930 for(i=m_TrackFGList.GetBlobNum(); i>0; --i)
931 { /* Add histogram and delete blob from list: */
932 DefTrackFG* pF = (DefTrackFG*)m_TrackFGList.GetBlob(i-1);
933 if(pF->LastFrame+3 < m_Frame && pF->pHist)
934 {
935 m_HistMat.Add(pF->pHist);
936 delete pF->pHist;
937 m_TrackNum++;
938 m_TrackFGList.DelBlob(i-1);
939 }
940 }/* next blob */
941 }
942
943 m_Frame++;
944
945 if(m_Wnd)
946 { /* Debug output: */
947 int* idxs = NULL;
948 int Val = 0;
949 IplImage* pI = cvCloneImage(pImg);
950
951 cvZero(pI);
952
953 for(Val = m_HistMat.GetNext(&idxs,1); idxs; Val=m_HistMat.GetNext(&idxs,0))
954 { /* Draw all elements: */
955 if(!idxs) break;
956 if(Val == 0) continue;
957
958 float vf = (float)Val/(m_HistMat.m_Max?m_HistMat.m_Max:1);
959 int x = cvRound((float)(pI->width-1)*(float)idxs[0] / (float)m_BinNum);
960 int y = cvRound((float)(pI->height-1)*(float)idxs[1] / (float)m_BinNum);
961
962 cvCircle(pI, cvPoint(x,y), cvRound(vf*pI->height/(m_BinNum*2)),CV_RGB(255,0,0),CV_FILLED);
963 if(m_Dim > 3)
964 {
965 int dx = -2*(idxs[2]-m_BinNum/2);
966 int dy = -2*(idxs[3]-m_BinNum/2);
967 cvLine(pI,cvPoint(x,y),cvPoint(x+dx,y+dy),CV_RGB(0,cvRound(vf*255),1));
968 }
969 if( m_Dim==4 &&
970 m_pFVGen->GetFVMax()[0]==m_pFVGen->GetFVMax()[2] &&
971 m_pFVGen->GetFVMax()[1]==m_pFVGen->GetFVMax()[3])
972 {
973 int x1 = cvRound((float)(pI->width-1)*(float)idxs[2] / (float)m_BinNum);
974 int y1 = cvRound((float)(pI->height-1)*(float)idxs[3] / (float)m_BinNum);
975 cvCircle(pI, cvPoint(x1,y1), cvRound(vf*pI->height/(m_BinNum*2)),CV_RGB(0,0,255),CV_FILLED);
976 }
977 } /* Draw all elements. */
978
979 for(int i=m_TrackFGList.GetBlobNum();i>0;--i)
980 {
981 DefTrackFG* pF = (DefTrackFG*)m_TrackFGList.GetBlob(i-1);
982 DefMat* pHist = pF?pF->pHist:NULL;
983
984 if(pHist==NULL) continue;
985
986 for(Val = pHist->GetNext(&idxs,1);idxs;Val=pHist->GetNext(&idxs,0))
987 { /* Draw all elements: */
988 float vf;
989 int x,y;
990
991 if(!idxs) break;
992 if(Val == 0) continue;
993
994 vf = (float)Val/(pHist->m_Max?pHist->m_Max:1);
995 x = cvRound((float)(pI->width-1)*(float)idxs[0] / (float)m_BinNum);
996 y = cvRound((float)(pI->height-1)*(float)idxs[1] / (float)m_BinNum);
997
998 cvCircle(pI, cvPoint(x,y), cvRound(2*vf),CV_RGB(0,0,cvRound(255*vf)),CV_FILLED);
999 if(m_Dim > 3)
1000 {
1001 int dx = -2*(idxs[2]-m_BinNum/2);
1002 int dy = -2*(idxs[3]-m_BinNum/2);
1003 cvLine(pI,cvPoint(x,y),cvPoint(x+dx,y+dy),CV_RGB(0,0,255));
1004 }
1005 if( m_Dim==4 &&
1006 m_pFVGen->GetFVMax()[0]==m_pFVGen->GetFVMax()[2] &&
1007 m_pFVGen->GetFVMax()[1]==m_pFVGen->GetFVMax()[3])
1008 { /* if SS feature vector */
1009 int x1 = cvRound((float)(pI->width-1)*(float)idxs[2] / (float)m_BinNum);
1010 int y1 = cvRound((float)(pI->height-1)*(float)idxs[3] / (float)m_BinNum);
1011 cvCircle(pI, cvPoint(x1,y1), cvRound(vf*pI->height/(m_BinNum*2)),CV_RGB(0,0,255),CV_FILLED);
1012 }
1013 } /* Draw all elements. */
1014 } /* Next track. */
1015
1016 //cvNamedWindow("Hist",0);
1017 //cvShowImage("Hist", pI);
1018 cvReleaseImage(&pI);
1019 }
