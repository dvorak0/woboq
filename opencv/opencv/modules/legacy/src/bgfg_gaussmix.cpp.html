45static void CV_CDECL
46icvReleaseGaussianBGModel( CvGaussBGModel** bg_model )
47{
48 if( !bg_model )
49 CV_Error( CV_StsNullPtr, "" );
50
51 if( *bg_model )
52 {
53 delete (cv::BackgroundSubtractorMOG*)((*bg_model)->mog);
54 cvReleaseImage( &(*bg_model)->background );
55 cvReleaseImage( &(*bg_model)->foreground );
56 memset( *bg_model, 0, sizeof(**bg_model) );
57 delete *bg_model;
58 *bg_model = 0;
59 }
60}

63static int CV_CDECL
64icvUpdateGaussianBGModel( IplImage* curr_frame, CvGaussBGModel* bg_model, double learningRate )
65{
66 cv::Mat image = cv::cvarrToMat(curr_frame), mask = cv::cvarrToMat(bg_model->foreground);
67
68 cv::BackgroundSubtractorMOG* mog = (cv::BackgroundSubtractorMOG*)(bg_model->mog);
69 CV_Assert(mog != 0);
70
71 (*mog)(image, mask, learningRate);
72 bg_model->countFrames++;
73
74 return 0;
75}

77CV_IMPL CvBGStatModel*
78cvCreateGaussianBGModel( IplImage* first_frame, CvGaussBGStatModelParams* parameters )
79{
80 CvGaussBGStatModelParams params;
81
82 CV_Assert( CV_IS_IMAGE(first_frame) );
83
84 //init parameters
85 if( parameters == NULL )
86 { // These constants are defined in cvaux/include/cvaux.h
87 params.win_size = CV_BGFG_MOG_WINDOW_SIZE;
88 params.bg_threshold = CV_BGFG_MOG_BACKGROUND_THRESHOLD;
89
90 params.std_threshold = CV_BGFG_MOG_STD_THRESHOLD;
91 params.weight_init = CV_BGFG_MOG_WEIGHT_INIT;
92
93 params.variance_init = CV_BGFG_MOG_SIGMA_INIT*CV_BGFG_MOG_SIGMA_INIT;
94 params.minArea = CV_BGFG_MOG_MINAREA;
95 params.n_gauss = CV_BGFG_MOG_NGAUSSIANS;
96 }
97 else
98 params = *parameters;
99
100 CvGaussBGModel* bg_model = new CvGaussBGModel;
101 memset( bg_model, 0, sizeof(*bg_model) );
102 bg_model->type = CV_BG_MODEL_MOG;
103 bg_model->release = (CvReleaseBGStatModel)icvReleaseGaussianBGModel;
104 bg_model->update = (CvUpdateBGStatModel)icvUpdateGaussianBGModel;
105
106 bg_model->params = params;
107
108 cv::BackgroundSubtractorMOG* mog =
109 new cv::BackgroundSubtractorMOG(params.win_size,
110 params.n_gauss,
111 params.bg_threshold,
112 params.variance_init);
113
114 bg_model->mog = mog;
115
116 CvSize sz = cvGetSize(first_frame);
117 bg_model->background = cvCreateImage(sz, IPL_DEPTH_8U, first_frame->nChannels);
118 bg_model->foreground = cvCreateImage(sz, IPL_DEPTH_8U, 1);
119
120 bg_model->countFrames = 0;
121
122 icvUpdateGaussianBGModel( first_frame, bg_model, 1 );
123
124 return (CvBGStatModel*)bg_model;
125}

170/*//Implementation of the Gaussian mixture model background subtraction from:
171 //
172 //"Improved adaptive Gausian mixture model for background subtraction"
173 //Z.Zivkovic
174 //International Conference Pattern Recognition, UK, August, 2004
175 //http://www.zoranz.net/Publications/zivkovic2004ICPR.pdf
176 //The code is very fast and performs also shadow detection.
177 //Number of Gausssian components is adapted per pixel.
178 //
179 // and
180 //
181 //"Efficient Adaptive Density Estimapion per Image Pixel for the Task of Background Subtraction"
182 //Z.Zivkovic, F. van der Heijden
183 //Pattern Recognition Letters, vol. 27, no. 7, pages 773-780, 2006.
184 //
185 //The algorithm similar to the standard Stauffer&Grimson algorithm with
186 //additional selection of the number of the Gaussian components based on:
187 //
188 //"Recursive unsupervised learning of finite mixture models "
189 //Z.Zivkovic, F.van der Heijden
190 //IEEE Trans. on Pattern Analysis and Machine Intelligence, vol.26, no.5, pages 651-656, 2004
191 //http://www.zoranz.net/Publications/zivkovic2004PAMI.pdf
192 //
193 //
194 //Example usage with as cpp class
195 // BackgroundSubtractorMOG2 bg_model;
196 //For each new image the model is updates using:
197 // bg_model(img, fgmask);
198 //
199 //Example usage as part of the CvBGStatModel:
200 // CvBGStatModel* bg_model = cvCreateGaussianBGModel2( first_frame );
201 //
202 // //update for each frame
203 // cvUpdateBGStatModel( tmp_frame, bg_model );//segmentation result is in bg_model->foreground
204 //
205 // //release at the program termination
206 // cvReleaseBGStatModel( &bg_model );
207 //

215/*
216 Interface of Gaussian mixture algorithm from:
217
218 "Improved adaptive Gausian mixture model for background subtraction"
219 Z.Zivkovic
220 International Conference Pattern Recognition, UK, August, 2004
221 http://www.zoranz.net/Publications/zivkovic2004ICPR.pdf
222
223 Advantages:
224 -fast - number of Gausssian components is constantly adapted per pixel.
225 -performs also shadow detection (see bgfg_segm_test.cpp example)
226
227 */

230#define CV_BG_MODEL_MOG2 3 /* "Mixture of Gaussians 2". */

233/* default parameters of gaussian background detection algorithm */
234#define CV_BGFG_MOG2_STD_THRESHOLD 4.0f /* lambda=2.5 is 99% */
235#define CV_BGFG_MOG2_WINDOW_SIZE 500 /* Learning rate; alpha = 1/CV_GBG_WINDOW_SIZE */
236#define CV_BGFG_MOG2_BACKGROUND_THRESHOLD 0.9f /* threshold sum of weights for background test */
237#define CV_BGFG_MOG2_STD_THRESHOLD_GENERATE 3.0f /* lambda=2.5 is 99% */
238#define CV_BGFG_MOG2_NGAUSSIANS 5 /* = K = number of Gaussians in mixture */
239#define CV_BGFG_MOG2_VAR_INIT 15.0f /* initial variance for new components*/
240#define CV_BGFG_MOG2_VAR_MIN 4.0f
241#define CV_BGFG_MOG2_VAR_MAX 5*CV_BGFG_MOG2_VAR_INIT
242#define CV_BGFG_MOG2_MINAREA 15.0f /* for postfiltering */
243
244/* additional parameters */
245#define CV_BGFG_MOG2_CT 0.05f /* complexity reduction prior constant 0 - no reduction of number of components*/
246#define CV_BGFG_MOG2_SHADOW_VALUE 127 /* value to use in the segmentation mask for shadows, sot 0 not to do shadow detection*/
247#define CV_BGFG_MOG2_SHADOW_TAU 0.5f /* Tau - shadow threshold, see the paper for explanation*/
248
249typedef struct CvGaussBGStatModel2Params
250{
251 //image info
252 int nWidth;
253 int nHeight;
254 int nND;//number of data dimensions (image channels)
255
256 bool bPostFiltering;//defult 1 - do postfiltering - will make shadow detection results also give value 255
257 double minArea; // for postfiltering
258
259 bool bInit;//default 1, faster updates at start
260
261 /////////////////////////
262 //very important parameters - things you will change
263 ////////////////////////
264 float fAlphaT;
265 //alpha - speed of update - if the time interval you want to average over is T
266 //set alpha=1/T. It is also usefull at start to make T slowly increase
267 //from 1 until the desired T
268 float fTb;
269 //Tb - threshold on the squared Mahalan. dist. to decide if it is well described
270 //by the background model or not. Related to Cthr from the paper.
271 //This does not influence the update of the background. A typical value could be 4 sigma
272 //and that is Tb=4*4=16;
273
274 /////////////////////////
275 //less important parameters - things you might change but be carefull
276 ////////////////////////
277 float fTg;
278 //Tg - threshold on the squared Mahalan. dist. to decide
279 //when a sample is close to the existing components. If it is not close
280 //to any a new component will be generated. I use 3 sigma => Tg=3*3=9.
281 //Smaller Tg leads to more generated components and higher Tg might make
282 //lead to small number of components but they can grow too large
283 float fTB;//1-cf from the paper
284 //TB - threshold when the component becomes significant enough to be included into
285 //the background model. It is the TB=1-cf from the paper. So I use cf=0.1 => TB=0.
286 //For alpha=0.001 it means that the mode should exist for approximately 105 frames before
287 //it is considered foreground
288 float fVarInit;
289 float fVarMax;
290 float fVarMin;
291 //initial standard deviation for the newly generated components.
292 //It will will influence the speed of adaptation. A good guess should be made.
293 //A simple way is to estimate the typical standard deviation from the images.
294 //I used here 10 as a reasonable value
295 float fCT;//CT - complexity reduction prior
296 //this is related to the number of samples needed to accept that a component
297 //actually exists. We use CT=0.05 of all the samples. By setting CT=0 you get
298 //the standard Stauffer&Grimson algorithm (maybe not exact but very similar)
299
300 //even less important parameters
301 int nM;//max number of modes - const - 4 is usually enough
302
303 //shadow detection parameters
304 bool bShadowDetection;//default 1 - do shadow detection
305 unsigned char nShadowDetection;//do shadow detection - insert this value as the detection result
306 float fTau;
307 // Tau - shadow threshold. The shadow is detected if the pixel is darker
308 //version of the background. Tau is a threshold on how much darker the shadow can be.
309 //Tau= 0.5 means that if pixel is more than 2 times darker then it is not shadow
310 //See: Prati,Mikic,Trivedi,Cucchiarra,"Detecting Moving Shadows...",IEEE PAMI,2003.
311} CvGaussBGStatModel2Params;

344//shadow detection performed per pixel
345// should work for rgb data, could be usefull for gray scale and depth data as well
346// See: Prati,Mikic,Trivedi,Cucchiarra,"Detecting Moving Shadows...",IEEE PAMI,2003.
347CV_INLINE int _icvRemoveShadowGMM(float* data, int nD,
348 unsigned char nModes,
349 CvPBGMMGaussian* pGMM,
350 float m_fTb,
351 float m_fTB,
352 float m_fTau)
353{
354 float tWeight = 0;
355 float numerator, denominator;
356 // check all the components marked as background:
357 for (int iModes=0;iModes<nModes;iModes++)
358 {
359
360 CvPBGMMGaussian g=pGMM[iModes];
361
362 numerator = 0.0f;
363 denominator = 0.0f;
364 for (int iD=0;iD<nD;iD++)
365 {
366 numerator += data[iD] * g.mean[iD];
367 denominator += g.mean[iD]* g.mean[iD];
368 }
369
370 // no division by zero allowed
371 if (denominator == 0)
372 {
373 return 0;
374 };
375 float a = numerator / denominator;
376
377 // if tau < a < 1 then also check the color distortion
378 if ((a <= 1) && (a >= m_fTau))
379 {
380
381 float dist2a=0.0f;
382
383 for (int iD=0;iD<nD;iD++)
384 {
385 float dD= a*g.mean[iD] - data[iD];
386 dist2a += (dD*dD);
387 }
388
389 if (dist2a<m_fTb*g.variance*a*a)
390 {
391 return 2;
392 }
393 };
394
395 tWeight += g.weight;
396 if (tWeight > m_fTB)
397 {
398 return 0;
399 };
400 };
401 return 0;
402}

404//update GMM - the base update function performed per pixel
405//
406//"Efficient Adaptive Density Estimapion per Image Pixel for the Task of Background Subtraction"
407//Z.Zivkovic, F. van der Heijden
408//Pattern Recognition Letters, vol. 27, no. 7, pages 773-780, 2006.
409//
410//The algorithm similar to the standard Stauffer&Grimson algorithm with
411//additional selection of the number of the Gaussian components based on:
412//
413//"Recursive unsupervised learning of finite mixture models "
414//Z.Zivkovic, F.van der Heijden
415//IEEE Trans. on Pattern Analysis and Machine Intelligence, vol.26, no.5, pages 651-656, 2004
416//http://www.zoranz.net/Publications/zivkovic2004PAMI.pdf

423CV_INLINE int _icvUpdateGMM(float* data, int nD,
424 unsigned char* pModesUsed,
425 CvPBGMMGaussian* pGMM,
426 int m_nM,
427 float m_fAlphaT,
428 float m_fTb,
429 float m_fTB,
430 float m_fTg,
431 float m_fVarInit,
432 float m_fVarMax,
433 float m_fVarMin,
434 float m_fPrune)
435{
436 //calculate distances to the modes (+ sort)
437 //here we need to go in descending order!!!
438 bool bBackground=0;//return value -> true - the pixel classified as background
439
440 //internal:
441 bool bFitsPDF=0;//if it remains zero a new GMM mode will be added
442 float m_fOneMinAlpha=1-m_fAlphaT;
443 unsigned char nModes=*pModesUsed;//current number of modes in GMM
444 float totalWeight=0.0f;
445
446 //////
447 //go through all modes
448 int iMode=0;
449 CvPBGMMGaussian* pGauss=pGMM;
450 for (;iMode<nModes;iMode++,pGauss++)
451 {
452 float weight = pGauss->weight;//need only weight if fit is found
453 weight=m_fOneMinAlpha*weight+m_fPrune;
454
455 ////
456 //fit not found yet
457 if (!bFitsPDF)
458 {
459 //check if it belongs to some of the remaining modes
460 float var=pGauss->variance;
461
462 //calculate difference and distance
463 float dist2=0.0f;
464#if (CV_BGFG_MOG2_NDMAX==1)
465 float dData=pGauss->mean[0]-data[0];
466 dist2=dData*dData;
467#else
468 float dData[CV_BGFG_MOG2_NDMAX];
469
470 for (int iD=0;iD<nD;iD++)
471 {
472 dData[iD]=pGauss->mean[iD]-data[iD];
473 dist2+=dData[iD]*dData[iD];
474 }
475#endif
476 //background? - m_fTb - usually larger than m_fTg
477 if ((totalWeight<m_fTB)&&(dist2<m_fTb*var))
478 bBackground=1;
479
480 //check fit
481 if (dist2<m_fTg*var)
482 {
483 /////
484 //belongs to the mode - bFitsPDF becomes 1
485 bFitsPDF=1;
486
487 //update distribution
488
489 //update weight
490 weight+=m_fAlphaT;
491
492 float k = m_fAlphaT/weight;
493
494 //update mean
495#if (CV_BGFG_MOG2_NDMAX==1)
496 pGauss->mean[0]-=k*dData;
497#else
498 for (int iD=0;iD<nD;iD++)
499 {
500 pGauss->mean[iD]-=k*dData[iD];
501 }
502#endif
503
504 //update variance
505 float varnew = var + k*(dist2-var);
506 //limit the variance
507 pGauss->variance = MIN(m_fVarMax,MAX(varnew,m_fVarMin));
508
509 //sort
510 //all other weights are at the same place and
511 //only the matched (iModes) is higher -> just find the new place for it
512 for (int iLocal = iMode;iLocal>0;iLocal--)
513 {
514 //check one up
515 if (weight < (pGMM[iLocal-1].weight))
516 {
517 break;
518 }
519 else
520 {
521 //swap one up
522 CvPBGMMGaussian temp = pGMM[iLocal];
523 pGMM[iLocal] = pGMM[iLocal-1];
524 pGMM[iLocal-1] = temp;
525 pGauss--;
526 }
527 }
528 //belongs to the mode - bFitsPDF becomes 1
529 /////
530 }
531 }//!bFitsPDF)
532
533 //check prune
534 if (weight<-m_fPrune)
535 {
536 weight=0.0;
537 nModes--;
538 }
539
540 pGauss->weight=weight;//update weight by the calculated value
541 totalWeight+=weight;
542 }
543 //go through all modes
544 //////
545
546 //renormalize weights
547 for (iMode = 0; iMode < nModes; iMode++)
548 {
549 pGMM[iMode].weight = pGMM[iMode].weight/totalWeight;
550 }
551
552 //make new mode if needed and exit
553 if (!bFitsPDF)
554 {
555 if (nModes==m_nM)
556 {
557 //replace the weakest
558 pGauss=pGMM+m_nM-1;
559 }
560 else
561 {
562 //add a new one
563 pGauss=pGMM+nModes;
564 nModes++;
565 }
566
567 if (nModes==1)
568 {
569 pGauss->weight=1;
570 }
571 else
572 {
573 pGauss->weight=m_fAlphaT;
574
575 //renormalize all weights
576 for (iMode = 0; iMode < nModes-1; iMode++)
577 {
578 pGMM[iMode].weight *=m_fOneMinAlpha;
579 }
580 }
581
582 //init
583 memcpy(pGauss->mean,data,nD*sizeof(float));
584 pGauss->variance=m_fVarInit;
585
586 //sort
587 //find the new place for it
588 for (int iLocal = nModes-1;iLocal>0;iLocal--)
589 {
590 //check one up
591 if (m_fAlphaT < (pGMM[iLocal-1].weight))
592 {
593 break;
594 }
595 else
596 {
597 //swap one up
598 CvPBGMMGaussian temp = pGMM[iLocal];
599 pGMM[iLocal] = pGMM[iLocal-1];
600 pGMM[iLocal-1] = temp;
601 }
602 }
603 }
604
605 //set the number of modes
606 *pModesUsed=nModes;
607
608 return bBackground;
609}

615// a bit more efficient implementation for common case of 3 channel (rgb) images
616CV_INLINE int _icvUpdateGMM_C3(float r,float g, float b,
617 unsigned char* pModesUsed,
618 CvPBGMMGaussian* pGMM,
619 int m_nM,
620 float m_fAlphaT,
621 float m_fTb,
622 float m_fTB,
623 float m_fTg,
624 float m_fVarInit,
625 float m_fVarMax,
626 float m_fVarMin,
627 float m_fPrune)
628{
629 //calculate distances to the modes (+ sort)
630 //here we need to go in descending order!!!
631 bool bBackground=0;//return value -> true - the pixel classified as background
632
633 //internal:
634 bool bFitsPDF=0;//if it remains zero a new GMM mode will be added
635 float m_fOneMinAlpha=1-m_fAlphaT;
636 unsigned char nModes=*pModesUsed;//current number of modes in GMM
637 float totalWeight=0.0f;
638
639 //////
640 //go through all modes
641 int iMode=0;
642 CvPBGMMGaussian* pGauss=pGMM;
643 for (;iMode<nModes;iMode++,pGauss++)
644 {
645 float weight = pGauss->weight;//need only weight if fit is found
646 weight=m_fOneMinAlpha*weight+m_fPrune;
647
648 ////
649 //fit not found yet
650 if (!bFitsPDF)
651 {
652 //check if it belongs to some of the remaining modes
653 float var=pGauss->variance;
654
655 //calculate difference and distance
656 float muR = pGauss->mean[0];
657 float muG = pGauss->mean[1];
658 float muB = pGauss->mean[2];
659
660 float dR=muR - r;
661 float dG=muG - g;
662 float dB=muB - b;
663
664 float dist2=(dR*dR+dG*dG+dB*dB);
665
666 //background? - m_fTb - usually larger than m_fTg
667 if ((totalWeight<m_fTB)&&(dist2<m_fTb*var))
668 bBackground=1;
669
670 //check fit
671 if (dist2<m_fTg*var)
672 {
673 /////
674 //belongs to the mode - bFitsPDF becomes 1
675 bFitsPDF=1;
676
677 //update distribution
678
679 //update weight
680 weight+=m_fAlphaT;
681
682 float k = m_fAlphaT/weight;
683
684 //update mean
685 pGauss->mean[0] = muR - k*(dR);
686 pGauss->mean[1] = muG - k*(dG);
687 pGauss->mean[2] = muB - k*(dB);
688
689 //update variance
690 float varnew = var + k*(dist2-var);
691 //limit the variance
692 pGauss->variance = MIN(m_fVarMax,MAX(varnew,m_fVarMin));
693
694 //sort
695 //all other weights are at the same place and
696 //only the matched (iModes) is higher -> just find the new place for it
697 for (int iLocal = iMode;iLocal>0;iLocal--)
698 {
699 //check one up
700 if (weight < (pGMM[iLocal-1].weight))
701 {
702 break;
703 }
704 else
705 {
706 //swap one up
707 CvPBGMMGaussian temp = pGMM[iLocal];
708 pGMM[iLocal] = pGMM[iLocal-1];
709 pGMM[iLocal-1] = temp;
710 pGauss--;
711 }
712 }
713 //belongs to the mode - bFitsPDF becomes 1
714 /////
715 }
716
717 }//!bFitsPDF)
718
719 //check prunning
720 if (weight<-m_fPrune)
721 {
722 weight=0.0;
723 nModes--;
724 }
725
726 pGauss->weight=weight;
727 totalWeight+=weight;
728 }
729 //go through all modes
730 //////
731
732 //renormalize weights
733 for (iMode = 0; iMode < nModes; iMode++)
734 {
735 pGMM[iMode].weight = pGMM[iMode].weight/totalWeight;
736 }
737
738 //make new mode if needed and exit
739 if (!bFitsPDF)
740 {
741 if (nModes==m_nM)
742 {
743 //replace the weakest
744 pGauss=pGMM+m_nM-1;
745 }
746 else
747 {
748 //add a new one
749 pGauss=pGMM+nModes;
750 nModes++;
751 }
752
753 if (nModes==1)
754 {
755 pGauss->weight=1;
756 }
757 else
758 {
759 pGauss->weight=m_fAlphaT;
760
761 //renormalize all weights
762 for (iMode = 0; iMode < nModes-1; iMode++)
763 {
764 pGMM[iMode].weight *=m_fOneMinAlpha;
765 }
766 }
767
768 //init
769 pGauss->mean[0]=r;
770 pGauss->mean[1]=g;
771 pGauss->mean[2]=b;
772
773 pGauss->variance=m_fVarInit;
774
775 //sort
776 //find the new place for it
777 for (int iLocal = nModes-1;iLocal>0;iLocal--)
778 {
779 //check one up
780 if (m_fAlphaT < (pGMM[iLocal-1].weight))
781 {
782 break;
783 }
784 else
785 {
786 //swap one up
787 CvPBGMMGaussian temp = pGMM[iLocal];
788 pGMM[iLocal] = pGMM[iLocal-1];
789 pGMM[iLocal-1] = temp;
790 }
791 }
792 }
793
794 //set the number of modes
795 *pModesUsed=nModes;
796
797 return bBackground;
798}
