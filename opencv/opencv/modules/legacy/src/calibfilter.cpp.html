91bool CvCalibFilter::SetEtalon( CvCalibEtalonType type, double* params,
92 int pointCount, CvPoint2D32f* _points )
93{
94 int i, arrSize;
95
96 Stop();
97
98 for( i = 0; i < MAX_CAMERAS; i++ )
99 cvFree( latestPoints + i );
100
101 if( type == CV_CALIB_ETALON_USER || type != etalonType )
102 {
103 if (etalonParams != NULL)
104 {
105 cvFree( &etalonParams );
106 }
107 }
108
109 etalonType = type;
110
111 switch( etalonType )
112 {
113 case CV_CALIB_ETALON_CHESSBOARD:
114 etalonParamCount = 3;
115 if( !params || cvRound(params[0]) != params[0] || params[0] < 3 ||
116 cvRound(params[1]) != params[1] || params[1] < 3 || params[2] <= 0 )
117 {
118 assert(0);
119 return false;
120 }
121
122 pointCount = cvRound((params[0] - 1)*(params[1] - 1));
123 break;
124
125 case CV_CALIB_ETALON_USER:
126 etalonParamCount = 0;
127
128 if( !_points || pointCount < 4 )
129 {
130 assert(0);
131 return false;
132 }
133 break;
134
135 default:
136 assert(0);
137 return false;
138 }
139
140 if( etalonParamCount > 0 )
141 {
142 arrSize = etalonParamCount * sizeof(etalonParams[0]);
143 etalonParams = (double*)cvAlloc( arrSize );
144 }
145
146 arrSize = pointCount * sizeof(etalonPoints[0]);
147
148 if( etalonPointCount != pointCount )
149 {
150 if (etalonPoints != NULL)
151 {
152 cvFree( &etalonPoints );
153 }
154 etalonPointCount = pointCount;
155 etalonPoints = (CvPoint2D32f*)cvAlloc( arrSize );
156 }
157
158 switch( etalonType )
159 {
160 case CV_CALIB_ETALON_CHESSBOARD:
161 {
162 int etalonWidth = cvRound( params[0] ) - 1;
163 int etalonHeight = cvRound( params[1] ) - 1;
164 int x, y, k = 0;
165
166 etalonParams[0] = etalonWidth;
167 etalonParams[1] = etalonHeight;
168 etalonParams[2] = params[2];
169
170 for( y = 0; y < etalonHeight; y++ )
171 for( x = 0; x < etalonWidth; x++ )
172 {
173 etalonPoints[k++] = cvPoint2D32f( (etalonWidth - 1 - x)*params[2],
174 y*params[2] );
175 }
176 }
177 break;
178
179 case CV_CALIB_ETALON_USER:
180 if (params != NULL)
181 {
182 memcpy( etalonParams, params, arrSize );
183 }
184 if (_points != NULL)
185 {
186 memcpy( etalonPoints, _points, arrSize );
187 }
188 break;
189
190 default:
191 assert(0);
192 return false;
193 }
194
195 return true;
196}

255void CvCalibFilter::Stop( bool calibrate )
256{
257 int i, j;
258 isCalibrated = false;
259
260 // deallocate undistortion maps
261 for( i = 0; i < cameraCount; i++ )
262 {
263 cvReleaseMat( &undistMap[i][0] );
264 cvReleaseMat( &undistMap[i][1] );
265 cvReleaseMat( &rectMap[i][0] );
266 cvReleaseMat( &rectMap[i][1] );
267 }
268
269 if( calibrate && framesAccepted > 0 )
270 {
271 int n = framesAccepted;
272 CvPoint3D32f* buffer =
273 (CvPoint3D32f*)cvAlloc( n * etalonPointCount * sizeof(buffer[0]));
274 CvMat mat;
275 float* rotMatr = (float*)cvAlloc( n * 9 * sizeof(rotMatr[0]));
276 float* transVect = (float*)cvAlloc( n * 3 * sizeof(transVect[0]));
277 int* counts = (int*)cvAlloc( n * sizeof(counts[0]));
278
279 cvInitMatHeader( &mat, 1, sizeof(CvCamera)/sizeof(float), CV_32FC1, 0 );
280 memset( cameraParams, 0, cameraCount * sizeof(cameraParams[0]));
281
282 for( i = 0; i < framesAccepted; i++ )
283 {
284 counts[i] = etalonPointCount;
285 for( j = 0; j < etalonPointCount; j++ )
286 buffer[i * etalonPointCount + j] = cvPoint3D32f( etalonPoints[j].x,
287 etalonPoints[j].y, 0 );
288 }
289
290 for( i = 0; i < cameraCount; i++ )
291 {
292 cvCalibrateCamera( framesAccepted, counts,
293 imgSize, points[i], buffer,
294 cameraParams[i].distortion,
295 cameraParams[i].matrix,
296 transVect, rotMatr, 0 );
297
298 cameraParams[i].imgSize[0] = (float)imgSize.width;
299 cameraParams[i].imgSize[1] = (float)imgSize.height;
300
301// cameraParams[i].focalLength[0] = cameraParams[i].matrix[0];
302// cameraParams[i].focalLength[1] = cameraParams[i].matrix[4];
303
304// cameraParams[i].principalPoint[0] = cameraParams[i].matrix[2];
305// cameraParams[i].principalPoint[1] = cameraParams[i].matrix[5];
306
307 memcpy( cameraParams[i].rotMatr, rotMatr, 9 * sizeof(rotMatr[0]));
308 memcpy( cameraParams[i].transVect, transVect, 3 * sizeof(transVect[0]));
309
310 mat.data.ptr = (uchar*)(cameraParams + i);
311
312 /* check resultant camera parameters: if there are some INF's or NAN's,
313 stop and reset results */
314 if( !cvCheckArr( &mat, CV_CHECK_RANGE | CV_CHECK_QUIET, -10000, 10000 ))
315 break;
316 }
317
318
319
320 isCalibrated = i == cameraCount;
321
322 {/* calibrate stereo cameras */
323 if( cameraCount == 2 )
324 {
325 stereo.camera[0] = &cameraParams[0];
326 stereo.camera[1] = &cameraParams[1];
327
328 icvStereoCalibration( framesAccepted, counts,
329 imgSize,
330 points[0],points[1],
331 buffer,
332 &stereo);
333 }
334
335 }
336
337 cvFree( &buffer );
338 cvFree( &counts );
339 cvFree( &rotMatr );
340 cvFree( &transVect );
341 }
342
343 framesAccepted = 0;
344}

353bool CvCalibFilter::FindEtalon( CvMat** mats )
354{
355 bool result = true;
356
357 if( !mats || etalonPointCount == 0 )
358 {
359 assert(0);
360 result = false;
361 }
362
363 if( result )
364 {
365 int i, tempPointCount0 = etalonPointCount*2;
366
367 for( i = 0; i < cameraCount; i++ )
368 {
369 if( !latestPoints[i] )
370 latestPoints[i] = (CvPoint2D32f*)
371 cvAlloc( tempPointCount0*2*sizeof(latestPoints[0]));
372 }
373
374 for( i = 0; i < cameraCount; i++ )
375 {
376 CvSize size;
377 int tempPointCount = tempPointCount0;
378 bool found = false;
379
380 if( !CV_IS_MAT(mats[i]) && !CV_IS_IMAGE(mats[i]))
381 {
382 assert(0);
383 break;
384 }
385
386 size = cvGetSize(mats[i]);
387
388 if( size.width != imgSize.width || size.height != imgSize.height )
389 {
390 imgSize = size;
391 }
392
393 if( !grayImg || grayImg->width != imgSize.width ||
394 grayImg->height != imgSize.height )
395 {
396 cvReleaseMat( &grayImg );
397 cvReleaseMat( &tempImg );
398 grayImg = cvCreateMat( imgSize.height, imgSize.width, CV_8UC1 );
399 tempImg = cvCreateMat( imgSize.height, imgSize.width, CV_8UC1 );
400 }
401
402 if( !storage )
403 storage = cvCreateMemStorage();
404
405 switch( etalonType )
406 {
407 case CV_CALIB_ETALON_CHESSBOARD:
408 if( CV_MAT_CN(cvGetElemType(mats[i])) == 1 )
409 cvCopy( mats[i], grayImg );
410 else
411 cvCvtColor( mats[i], grayImg, CV_BGR2GRAY );
412 found = cvFindChessBoardCornerGuesses( grayImg, tempImg, storage,
413 cvSize( cvRound(etalonParams[0]),
414 cvRound(etalonParams[1])),
415 latestPoints[i], &tempPointCount ) != 0;
416 if( found )
417 cvFindCornerSubPix( grayImg, latestPoints[i], tempPointCount,
418 cvSize(5,5), cvSize(-1,-1),
419 cvTermCriteria(CV_TERMCRIT_ITER|CV_TERMCRIT_EPS,10,0.1));
420 break;
421 default:
422 assert(0);
423 result = false;
424 break;
425 }
426
427 latestCounts[i] = found ? tempPointCount : -tempPointCount;
428 result = result && found;
429 }
430 }
431
432 if( storage )
433 cvClearMemStorage( storage );
434
435 return result;
436}

439bool CvCalibFilter::Push( const CvPoint2D32f** pts )
440{
441 bool result = true;
442 int i, newMaxPoints = etalonPointCount*(MAX(framesAccepted,framesTotal) + 1);
443
444 isCalibrated = false;
445
446 if( !pts )
447 {
448 for( i = 0; i < cameraCount; i++ )
449 if( latestCounts[i] <= 0 )
450 return false;
451 pts = (const CvPoint2D32f**)latestPoints;
452 }
453
454 for( i = 0; i < cameraCount; i++ )
455 {
456 if( !pts[i] )
457 {
458 assert(0);
459 break;
460 }
461
462 if( maxPoints < newMaxPoints )
463 {
464 CvPoint2D32f* prev = points[i];
465 cvFree( points + i );
466 points[i] = (CvPoint2D32f*)cvAlloc( newMaxPoints * sizeof(prev[0]));
467 memcpy( points[i], prev, maxPoints * sizeof(prev[0]));
468 }
469
470 memcpy( points[i] + framesAccepted*etalonPointCount, pts[i],
471 etalonPointCount*sizeof(points[0][0]));
472 }
473
474 if( maxPoints < newMaxPoints )
475 maxPoints = newMaxPoints;
476
477 result = i == cameraCount;
478
479 if( ++framesAccepted >= framesTotal )
480 Stop( true );
481 return result;
482}

485bool CvCalibFilter::GetLatestPoints( int idx, CvPoint2D32f** pts,
486 int* count, bool* found )
487{
488 int n;
489
490 if( (unsigned)idx >= (unsigned)cameraCount ||

492 {
493 assert(0);
494 return false;
495 }
496
497 n = latestCounts[idx];
498

502
503 return true;

512
513void CvCalibFilter::DrawPoints( CvMat** dstarr )
514{
515 int i, j;
516
517 if( !dstarr )
518 {
519 assert(0);
520 return;
521 }
522
523 for( i = 0; i < cameraCount; i++ )
524 {
525 if( dstarr[i] && latestCounts[i] )
526 {
527 CvMat dst_stub, *dst;
528 int count = 0;
529 bool found = false;
530 CvPoint2D32f* pts = 0;
531
532 GetLatestPoints( i, &pts, &count, &found );
533
534 dst = cvGetMat( dstarr[i], &dst_stub );
535
536 static const CvScalar line_colors[] =
537 {
538 {{0,0,255}},
539 {{0,128,255}},
540 {{0,200,200}},
541 {{0,255,0}},
542 {{200,200,0}},
543 {{255,0,0}},
544 {{255,0,255}}
545 };
546
547 const int colorCount = sizeof(line_colors)/sizeof(line_colors[0]);
548 const int r = 4;
549 CvScalar color = line_colors[0];
550 CvPoint prev_pt = { 0, 0};
551
552 for( j = 0; j < count; j++ )
553 {
554 CvPoint pt;
555 pt.x = cvRound(pts[j].x);
556 pt.y = cvRound(pts[j].y);
557
558 if( found )
559 {
560 if( etalonType == CV_CALIB_ETALON_CHESSBOARD )
561 color = line_colors[(j/cvRound(etalonParams[0]))%colorCount];
562 else
563 color = CV_RGB(0,255,0);
564
565 if( j != 0 )
566 cvLine( dst, prev_pt, pt, color, 1, CV_AA );
567 }
568
569 cvLine( dst, cvPoint( pt.x - r, pt.y - r ),
570 cvPoint( pt.x + r, pt.y + r ), color, 1, CV_AA );
571
572 cvLine( dst, cvPoint( pt.x - r, pt.y + r),
573 cvPoint( pt.x + r, pt.y - r), color, 1, CV_AA );
574
575 cvCircle( dst, pt, r+1, color, 1, CV_AA );
576
577 prev_pt = pt;
578 }
579 }
580 }
581}

702bool CvCalibFilter::LoadCameraParams( const char* filename )
703{
704 int i, j;
705 int d = 0;
706 FILE* f = fopen( filename, "r" );
707
708 isCalibrated = false;
709
710 if( !f ) return false;
711
712 if( fscanf( f, "%d", &d ) != 1 || d <= 0 || d > 10 )
713 return false;
714
715 SetCameraCount( d );
716
717 for( i = 0; i < cameraCount; i++ )
718 {
719 for( j = 0; j < (int)(sizeof(cameraParams[i])/sizeof(float)); j++ )
720 {
721 int values_read = fscanf( f, "%f", &((float*)(cameraParams + i))[j] );
722 CV_Assert(values_read == 1);
723 }
724 }
725
726
727 /* Load stereo params */
728
729 /* load quad */
730 for( i = 0; i < 2; i++ )
731 {
732 for( j = 0; j < 4; j++ )
733 {
734 int values_read = fscanf(f, "%f ", &(stereo.quad[i][j].x) );
735 CV_Assert(values_read == 1);
736 values_read = fscanf(f, "%f ", &(stereo.quad[i][j].y) );
737 CV_Assert(values_read == 1);
738 }
739 }
740
741 /* Load coeffs */
742 for( i = 0; i < 2; i++ )
743 {
744 for( j = 0; j < 9; j++ )
745 {
746 int values_read = fscanf(f, "%lf ", &(stereo.coeffs[i][j/3][j%3]) );
747 CV_Assert(values_read == 1);
748 }
749 }
750
751
752
753
754 fclose(f);
755
756 stereo.warpSize = cvSize( cvRound(cameraParams[0].imgSize[0]), cvRound(cameraParams[0].imgSize[1]));
757
758 isCalibrated = true;
759
760 return true;
761}

768
769bool CvCalibFilter::Rectify( CvMat** srcarr, CvMat** dstarr )
770{
771 int i;
772
773 if( !srcarr || !dstarr )
774 {
775 assert(0);
776 return false;
777 }
778
779 if( isCalibrated && cameraCount == 2 )
780 {
781 for( i = 0; i < cameraCount; i++ )
782 {
783 if( srcarr[i] && dstarr[i] )
784 {
785 IplImage src_stub, *src;
786 IplImage dst_stub, *dst;
787
788 src = cvGetImage( srcarr[i], &src_stub );
789 dst = cvGetImage( dstarr[i], &dst_stub );
790
791 if( src->imageData == dst->imageData )
792 {
793 if( !undistImg ||
794 undistImg->width != src->width ||
795 undistImg->height != src->height ||
796 CV_MAT_CN(undistImg->type) != src->nChannels )
797 {
798 cvReleaseMat( &undistImg );
799 undistImg = cvCreateMat( src->height, src->width,
800 CV_8U + (src->nChannels-1)*8 );
801 }
802 cvCopy( src, undistImg );
803 src = cvGetImage( undistImg, &src_stub );
804 }
805
806 cvZero( dst );
807
808 if( !rectMap[i][0] || rectMap[i][0]->width != src->width ||
809 rectMap[i][0]->height != src->height )
810 {
811 cvReleaseMat( &rectMap[i][0] );
812 cvReleaseMat( &rectMap[i][1] );
813 rectMap[i][0] = cvCreateMat(stereo.warpSize.height,stereo.warpSize.width,CV_32FC1);
814 rectMap[i][1] = cvCreateMat(stereo.warpSize.height,stereo.warpSize.width,CV_32FC1);
815 cvComputePerspectiveMap(stereo.coeffs[i], rectMap[i][0], rectMap[i][1]);
816 }
817 cvRemap( src, dst, rectMap[i][0], rectMap[i][1] );
818 }
819 }
820 }
821 else
822 {
823 for( i = 0; i < cameraCount; i++ )
824 {
825 if( srcarr[i] != dstarr[i] )
826 cvCopy( srcarr[i], dstarr[i] );
827 }
828 }
829
830 return true;
831}

839bool CvCalibFilter::Undistort( CvMat** srcarr, CvMat** dstarr )
840{
841 int i;
842
843 if( !srcarr || !dstarr )
844 {
845 assert(0);
846 return false;
847 }
848
849 if( isCalibrated )
850 {
851 for( i = 0; i < cameraCount; i++ )
852 {
853 if( srcarr[i] && dstarr[i] )
854 {
855 CvMat src_stub, *src;
856 CvMat dst_stub, *dst;
857
858 src = cvGetMat( srcarr[i], &src_stub );
859 dst = cvGetMat( dstarr[i], &dst_stub );
860
861 if( src->data.ptr == dst->data.ptr )
862 {
863 if( !undistImg || undistImg->width != src->width ||
864 undistImg->height != src->height ||
865 CV_ARE_TYPES_EQ( undistImg, src ))
866 {
867 cvReleaseMat( &undistImg );
868 undistImg = cvCreateMat( src->height, src->width, src->type );
869 }
870
871 cvCopy( src, undistImg );
872 src = undistImg;
873 }
874
875 #if 1
876 {
877 CvMat A = cvMat( 3, 3, CV_32FC1, cameraParams[i].matrix );
878 CvMat k = cvMat( 1, 4, CV_32FC1, cameraParams[i].distortion );
879
880 if( !undistMap[i][0] || undistMap[i][0]->width != src->width ||
881 undistMap[i][0]->height != src->height )
882 {
883 cvReleaseMat( &undistMap[i][0] );
884 cvReleaseMat( &undistMap[i][1] );
885 undistMap[i][0] = cvCreateMat( src->height, src->width, CV_32FC1 );
886 undistMap[i][1] = cvCreateMat( src->height, src->width, CV_32FC1 );
887 cvInitUndistortMap( &A, &k, undistMap[i][0], undistMap[i][1] );
888 }
889
890 cvRemap( src, dst, undistMap[i][0], undistMap[i][1] );
891 #else
892 cvUndistort2( src, dst, &A, &k );
893 #endif
894 }
895 }
896 }
897 }
898 else
899 {
900 for( i = 0; i < cameraCount; i++ )
901 {
902 if( srcarr[i] != dstarr[i] )
903 cvCopy( srcarr[i], dstarr[i] );
904 }
905 }
906
907
908 return true;
909}
