56// J.S. Beis and D.G. Lowe. Shape indexing using approximate nearest-neighbor search
57// in highdimensional spaces. In Proc. IEEE Conf. Comp. Vision Patt. Recog.,
58// pages 1000--1006, 1997. http://citeseer.ist.psu.edu/beis97shape.html
59#undef __deref
60#undef __valuetype

62template < class __valuetype, class __deref >
63class CvKDTree {
64public:
65 typedef __deref deref_type;
66 typedef typename __deref::scalar_type scalar_type;
67 typedef typename __deref::accum_type accum_type;
68
69private:
70 struct node {
71 int dim; // split dimension; >=0 for nodes, -1 for leaves
72 __valuetype value; // if leaf, value of leaf
73 int left, right; // node indices of left and right branches
74 scalar_type boundary; // left if deref(value,dim)<=boundary, otherwise right
75 };
76 typedef std::vector < node > node_array;
77
78 __deref deref; // requires operator() (__valuetype lhs,int dim)
79
80 node_array nodes; // node storage
81 int point_dim; // dimension of points (the k in kd-tree)
82 int root_node; // index of root node, -1 if empty tree
83
84 // for given set of point indices, compute dimension of highest variance
85 template < class __instype, class __valuector >
86 int dimension_of_highest_variance(__instype * first, __instype * last,
87 __valuector ctor) {
88 assert(last - first > 0);
89
90 accum_type maxvar = -std::numeric_limits < accum_type >::max();
91 int maxj = -1;
92 for (int j = 0; j < point_dim; ++j) {
93 accum_type mean = 0;
94 for (__instype * k = first; k < last; ++k)
95 mean += deref(ctor(*k), j);
96 mean /= last - first;
97 accum_type var = 0;
98 for (__instype * k = first; k < last; ++k) {
99 accum_type diff = accum_type(deref(ctor(*k), j)) - mean;
100 var += diff * diff;
101 }
102 var /= last - first;
103
104 assert(maxj != -1 || var >= maxvar);
105
106 if (var >= maxvar) {
107 maxvar = var;
108 maxj = j;
109 }
110 }
111
112 return maxj;
113 }
114
115 // given point indices and dimension, find index of median; (almost) modifies [first,last)
116 // such that points_in[first,median]<=point[median], points_in(median,last)>point[median].
117 // implemented as partial quicksort; expected linear perf.
118 template < class __instype, class __valuector >
119 __instype * median_partition(__instype * first, __instype * last,
120 int dim, __valuector ctor) {
121 assert(last - first > 0);
122 __instype *k = first + (last - first) / 2;
123 median_partition(first, last, k, dim, ctor);
124 return k;
125 }
126
127 template < class __instype, class __valuector >
128 struct median_pr {
129 const __instype & pivot;
130 int dim;
131 __deref deref;
132 __valuector ctor;
133 median_pr(const __instype & _pivot, int _dim, __deref _deref, __valuector _ctor)
134 : pivot(_pivot), dim(_dim), deref(_deref), ctor(_ctor) {
135 }
136 bool operator() (const __instype & lhs) const {
137 return deref(ctor(lhs), dim) <= deref(ctor(pivot), dim);
138 }
139 private:
140 median_pr& operator=(const median_pr&);
141 };
142
143 template < class __instype, class __valuector >
144 void median_partition(__instype * first, __instype * last,
145 __instype * k, int dim, __valuector ctor) {
146 int pivot = (int)((last - first) / 2);
147
148 std::swap(first[pivot], last[-1]);
149 __instype *middle = std::partition(first, last - 1,
150 median_pr < __instype, __valuector >
151 (last[-1], dim, deref, ctor));
152 std::swap(*middle, last[-1]);
153
154 if (middle < k)
155 median_partition(middle + 1, last, k, dim, ctor);
156 else if (middle > k)
157 median_partition(first, middle, k, dim, ctor);
158 }
159
160 // insert given points into the tree; return created node
161 template < class __instype, class __valuector >
162 int insert(__instype * first, __instype * last, __valuector ctor) {
163 if (first == last)
164 return -1;
165 else {
166
167 int dim = dimension_of_highest_variance(first, last, ctor);
168 __instype *median = median_partition(first, last, dim, ctor);
169
170 __instype *split = median;
171 for (; split != last && deref(ctor(*split), dim) ==
172 deref(ctor(*median), dim); ++split) {}
173
174 if (split == last) { // leaf
175 int nexti = -1;
176 for (--split; split >= first; --split) {
177 int i = (int)nodes.size();
178 node & n = *nodes.insert(nodes.end(), node());
179 n.dim = -1;
180 n.value = ctor(*split);
181 n.left = -1;
182 n.right = nexti;
183 nexti = i;
184 }
185
186 return nexti;
187 } else { // node
188 int i = (int)nodes.size();
189 // note that recursive insert may invalidate this ref
190 node & n = *nodes.insert(nodes.end(), node());
191
192 n.dim = dim;
193 n.boundary = deref(ctor(*median), dim);
194
195 int left = insert(first, split, ctor);
196 nodes[i].left = left;
197 int right = insert(split, last, ctor);
198 nodes[i].right = right;
199
200 return i;
201 }
202 }
203 }
204
205 // run to leaf; linear search for p;
206 // if found, remove paths to empty leaves on unwind
207 bool remove(int *i, const __valuetype & p) {
208 if (*i == -1)
209 return false;
210 node & n = nodes[*i];
211 bool r;
212
213 if (n.dim >= 0) { // node
214 if (deref(p, n.dim) <= n.boundary) // left
215 r = remove(&n.left, p);
216 else // right
217 r = remove(&n.right, p);
218
219 // if terminal, remove this node
220 if (n.left == -1 && n.right == -1)
221 *i = -1;
222
223 return r;
224 } else { // leaf
225 if (n.value == p) {
226 *i = n.right;
227 return true;
228 } else
229 return remove(&n.right, p);
230 }
231 }

233public:
234 struct identity_ctor {
235 const __valuetype & operator() (const __valuetype & rhs) const {
236 return rhs;
237 }
238 };
239
240 // initialize an empty tree
241 CvKDTree(__deref _deref = __deref())
242 : deref(_deref), root_node(-1) {
243 }
244 // given points, initialize a balanced tree
245 CvKDTree(__valuetype * first, __valuetype * last, int _point_dim,
246 __deref _deref = __deref())
247 : deref(_deref) {
248 set_data(first, last, _point_dim, identity_ctor());
249 }
250 // given points, initialize a balanced tree
251 template < class __instype, class __valuector >
252 CvKDTree(__instype * first, __instype * last, int _point_dim,
253 __valuector ctor, __deref _deref = __deref())
254 : deref(_deref) {
255 set_data(first, last, _point_dim, ctor);
256 }
257
258 void set_deref(__deref _deref) {
259 deref = _deref;
260 }
261
262 void set_data(__valuetype * first, __valuetype * last, int _point_dim) {
263 set_data(first, last, _point_dim, identity_ctor());
264 }
265 template < class __instype, class __valuector >
266 void set_data(__instype * first, __instype * last, int _point_dim,
267 __valuector ctor) {
268 point_dim = _point_dim;
269 nodes.clear();
270 nodes.reserve(last - first);
271 root_node = insert(first, last, ctor);
272 }
273
274 int dims() const {
275 return point_dim;
276 }
277
278 // remove the given point
279 bool remove(const __valuetype & p) {
280 return remove(&root_node, p);
281 }

283 void print() const {
284 print(root_node);
285 }
286 void print(int i, int indent = 0) const {
287 if (i == -1)
288 return;
289 for (int j = 0; j < indent; ++j)
290 std::cout << " ";
291 const node & n = nodes[i];
292 if (n.dim >= 0) {
293 std::cout << "node " << i << ", left " << nodes[i].left << ", right " <<
294 nodes[i].right << ", dim " << nodes[i].dim << ", boundary " <<
295 nodes[i].boundary << std::endl;
296 print(n.left, indent + 3);
297 print(n.right, indent + 3);
298 } else
299 std::cout << "leaf " << i << ", value = " << nodes[i].value << std::endl;
300 }

302 ////////////////////////////////////////////////////////////////////////////////////////
303 // bbf search
304public:
305 struct bbf_nn { // info on found neighbors (approx k nearest)
306 const __valuetype *p; // nearest neighbor
307 accum_type dist; // distance from d to query point
308 bbf_nn(const __valuetype & _p, accum_type _dist)
309 : p(&_p), dist(_dist) {
310 }
311 bool operator<(const bbf_nn & rhs) const {
312 return dist < rhs.dist;
313 }
314 };
315 typedef std::vector < bbf_nn > bbf_nn_pqueue;
316private:
317 struct bbf_node { // info on branches not taken
318 int node; // corresponding node
319 accum_type dist; // minimum distance from bounds to query point
320 bbf_node(int _node, accum_type _dist)
321 : node(_node), dist(_dist) {
322 }
323 bool operator<(const bbf_node & rhs) const {
324 return dist > rhs.dist;
325 }
326 };
327 typedef std::vector < bbf_node > bbf_pqueue;
328 mutable bbf_pqueue tmp_pq;
329
330 // called for branches not taken, as bbf walks to leaf;
331 // construct bbf_node given minimum distance to bounds of alternate branch
332 void pq_alternate(int alt_n, bbf_pqueue & pq, scalar_type dist) const {
333 if (alt_n == -1)
334 return;
335
336 // add bbf_node for alternate branch in priority queue
337 pq.push_back(bbf_node(alt_n, dist));
338 std::push_heap(pq.begin(), pq.end());
339 }
340
341 // called by bbf to walk to leaf;
342 // takes one step down the tree towards query point d
343 template < class __desctype >
344 int bbf_branch(int i, const __desctype * d, bbf_pqueue & pq) const {
345 const node & n = nodes[i];
346 // push bbf_node with bounds of alternate branch, then branch
347 if (d[n.dim] <= n.boundary) { // left
348 pq_alternate(n.right, pq, n.boundary - d[n.dim]);
349 return n.left;
350 } else { // right
351 pq_alternate(n.left, pq, d[n.dim] - n.boundary);
352 return n.right;
353 }
354 }
355
356 // compute euclidean distance between two points
357 template < class __desctype >
358 accum_type distance(const __desctype * d, const __valuetype & p) const {
359 accum_type dist = 0;
360 for (int j = 0; j < point_dim; ++j) {
361 accum_type diff = accum_type(d[j]) - accum_type(deref(p, j));
362 dist += diff * diff;
363 } return (accum_type) sqrt(dist);
364 }
365
366 // called per candidate nearest neighbor; constructs new bbf_nn for
367 // candidate and adds it to priority queue of all candidates; if
368 // queue len exceeds k, drops the point furthest from query point d.
369 template < class __desctype >
370 void bbf_new_nn(bbf_nn_pqueue & nn_pq, int k,
371 const __desctype * d, const __valuetype & p) const {
372 bbf_nn nn(p, distance(d, p));
373 if ((int) nn_pq.size() < k) {
374 nn_pq.push_back(nn);
375 std::push_heap(nn_pq.begin(), nn_pq.end());
376 } else if (nn_pq[0].dist > nn.dist) {
377 std::pop_heap(nn_pq.begin(), nn_pq.end());
378 nn_pq.end()[-1] = nn;
379 std::push_heap(nn_pq.begin(), nn_pq.end());
380 }
381 assert(nn_pq.size() < 2 || nn_pq[0].dist >= nn_pq[1].dist);
382 }
383
384public:
385 // finds (with high probability) the k nearest neighbors of d,
386 // searching at most emax leaves/bins.
387 // ret_nn_pq is an array containing the (at most) k nearest neighbors
388 // (see bbf_nn structure def above).
389 template < class __desctype >
390 int find_nn_bbf(const __desctype * d, int k, int emax, bbf_nn_pqueue & ret_nn_pq) const
391 {
392 assert(k > 0);
393 ret_nn_pq.clear();
394
395 if (root_node == -1)
396 return 0;
397
398 // add root_node to bbf_node priority queue;
399 // iterate while queue non-empty and emax>0
400 tmp_pq.clear();
401 tmp_pq.push_back(bbf_node(root_node, 0));
402 while (tmp_pq.size() && emax > 0)
403 {
404
405 // from node nearest query point d, run to leaf
406 std::pop_heap(tmp_pq.begin(), tmp_pq.end());
407 bbf_node bbf(tmp_pq.end()[-1]);
408 tmp_pq.erase(tmp_pq.end() - 1);
409
410 int i;
411 for (i = bbf.node; i != -1 && nodes[i].dim >= 0; i = bbf_branch(i, d, tmp_pq)) {}
412
413 if (i != -1)
414 {
415
416 // add points in leaf/bin to ret_nn_pq
417 do {
418 bbf_new_nn(ret_nn_pq, k, d, nodes[i].value);
419 } while (-1 != (i = nodes[i].right));
420
421 --emax;
422 }
423 }
424
425 tmp_pq.clear();
426 return (int)ret_nn_pq.size();
427 }

429 ////////////////////////////////////////////////////////////////////////////////////////
430 // orthogonal range search
431private:
432 void find_ortho_range(int i, scalar_type * bounds_min,
433 scalar_type * bounds_max,
434 std::vector < __valuetype > &inbounds) const {
435 if (i == -1)
436 return;
437 const node & n = nodes[i];
438 if (n.dim >= 0) { // node
439 if (bounds_min[n.dim] <= n.boundary)
440 find_ortho_range(n.left, bounds_min, bounds_max, inbounds);
441 if (bounds_max[n.dim] > n.boundary)
442 find_ortho_range(n.right, bounds_min, bounds_max, inbounds);
443 } else { // leaf
444 do {
445 inbounds.push_back(nodes[i].value);
446 } while (-1 != (i = nodes[i].right));
447 }
448 }
449public:
450 // return all points that lie within the given bounds; inbounds is cleared
451 int find_ortho_range(scalar_type * bounds_min,
452 scalar_type * bounds_max,
453 std::vector < __valuetype > &inbounds) const {
454 inbounds.clear();
455 find_ortho_range(root_node, bounds_min, bounds_max, inbounds);
456 return (int)inbounds.size();
457 }
458};
