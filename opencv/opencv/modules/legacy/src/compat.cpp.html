95/* Changes RNG range while preserving RNG state */
96void cvRandSetRange( CvRandState* state, double param1, double param2, int index)
97{
98 if( !state )
99 {
100 cvError( CV_StsNullPtr, "cvRandSetRange", "Null pointer to RNG state", "cvcompat.h", 0 );
101 return;
102 }
103
104 if( (unsigned)(index + 1) > 4 )
105 {
106 cvError( CV_StsOutOfRange, "cvRandSetRange", "index is not in -1..3", "cvcompat.h", 0 );
107 return;
108 }
109
110 if( index < 0 )
111 {
112 state->param[0].val[0] = state->param[0].val[1] =
113 state->param[0].val[2] = state->param[0].val[3] = param1;
114 state->param[1].val[0] = state->param[1].val[1] =
115 state->param[1].val[2] = state->param[1].val[3] = param2;
116 }
117 else
118 {
119 state->param[0].val[index] = param1;
120 state->param[1].val[index] = param2;
121 }
122}

125void cvRandInit( CvRandState* state, double param1, double param2,
126 int seed, int disttype)
127{
128 if( !state )
129 {
130 cvError( CV_StsNullPtr, "cvRandInit", "Null pointer to RNG state", "cvcompat.h", 0 );
131 return;
132 }
133
134 if( disttype != CV_RAND_UNI && disttype != CV_RAND_NORMAL )
135 {
136 cvError( CV_StsBadFlag, "cvRandInit", "Unknown distribution type", "cvcompat.h", 0 );
137 return;
138 }
139
140 state->state = (uint64)(seed ? seed : -1);
141 state->disttype = disttype;
142 cvRandSetRange( state, param1, param2, -1 );
143}

261void cvMinAreaRect( CvPoint* points, int n, int, int, int, int,
262 CvPoint2D32f* anchor, CvPoint2D32f* vect1, CvPoint2D32f* vect2 )
263{
264 CvMat mat = cvMat( 1, n, CV_32SC2, points );
265 CvBox2D box = cvMinAreaRect2( &mat, 0 );
266 CvPoint2D32f pt[4];
267
268 cvBoxPoints( box, pt );
269 *anchor = pt[0];
270 vect1->x = pt[1].x - pt[0].x;
271 vect1->y = pt[1].y - pt[0].y;
272 vect2->x = pt[3].x - pt[0].x;
273 vect2->y = pt[3].y - pt[0].y;
274}

302/* Projects 2d points to one of standard coordinate planes
303 (i.e. removes one of coordinates) */
304void cvProject3D( CvPoint3D32f* points3D, int count,
305 CvPoint2D32f* points2D, int xIndx, int yIndx)
306{
307 CvMat src = cvMat( 1, count, CV_32FC3, points3D );
308 CvMat dst = cvMat( 1, count, CV_32FC2, points2D );
309 float m[6] = {0,0,0,0,0,0};
310 CvMat M = cvMat( 2, 3, CV_32F, m );
311
312 assert( (unsigned)xIndx < 3 && (unsigned)yIndx < 3 );
313 m[xIndx] = m[yIndx+3] = 1.f;
314
315 cvTransform( &src, &dst, &M, NULL );
316}

450/* the two functions below have quite hackerish implementations, use with care
451 (or, which is better, switch to cvUndistortInitMap and cvRemap instead */
452void cvUnDistortInit( const CvArr*,
453 CvArr* undistortion_map,
454 const float* A, const float* k,
455 int)
456{
457 union { uchar* ptr; float* fl; } data;
458 CvSize sz;
459 cvGetRawData( undistortion_map, &data.ptr, 0, &sz );
460 assert( sz.width >= 8 );
461 /* just save the intrinsic parameters to the map */
462 data.fl[0] = A[0]; data.fl[1] = A[4];
463 data.fl[2] = A[2]; data.fl[3] = A[5];
464 data.fl[4] = k[0]; data.fl[5] = k[1];
465 data.fl[6] = k[2]; data.fl[7] = k[3];
466}
467
468void cvUnDistort( const CvArr* src, CvArr* dst,
469 const CvArr* undistortion_map, int )
470{
471 union { uchar* ptr; float* fl; } data;
472 float a[] = {0,0,0,0,0,0,0,0,1};
473 CvSize sz;
474 cvGetRawData( undistortion_map, &data.ptr, 0, &sz );
475 assert( sz.width >= 8 );
476 a[0] = data.fl[0]; a[4] = data.fl[1];
477 a[2] = data.fl[2]; a[5] = data.fl[3];
478 cvUnDistortOnce( src, dst, a, data.fl + 4, 1 );
479}

482/* Find fundamental matrix */
483void cvFindFundamentalMatrix( int* points1, int* points2, int numpoints, int, float* matrix )
484{
485 CvMat* pointsMat1;
486 CvMat* pointsMat2;
487 CvMat fundMatr = cvMat(3,3,CV_32F,matrix);
488 int i, curr = 0;
489
490 pointsMat1 = cvCreateMat(3,numpoints,CV_64F);
491 pointsMat2 = cvCreateMat(3,numpoints,CV_64F);
492
493 for( i = 0; i < numpoints; i++ )
494 {
495 cvmSet(pointsMat1,0,i,points1[curr]);//x
496 cvmSet(pointsMat1,1,i,points1[curr+1]);//y
497 cvmSet(pointsMat1,2,i,1.0);
498
499 cvmSet(pointsMat2,0,i,points2[curr]);//x
500 cvmSet(pointsMat2,1,i,points2[curr+1]);//y
501 cvmSet(pointsMat2,2,i,1.0);
502 curr += 2;
503 }
504
505 cvFindFundamentalMat(pointsMat1,pointsMat2,&fundMatr,CV_FM_RANSAC,1,0.99,0);
506
507 cvReleaseMat(&pointsMat1);
508 cvReleaseMat(&pointsMat2);
509}

521/* Calibrates camera using multiple views of calibration pattern */
522void cvCalibrateCamera( int image_count, int* _point_counts,
523 CvSize image_size, CvPoint2D32f* _image_points, CvPoint3D32f* _object_points,
524 float* _distortion_coeffs, float* _camera_matrix, float* _translation_vectors,
525 float* _rotation_matrices, int flags )
526{
527 int i, total = 0;
528 CvMat point_counts = cvMat( image_count, 1, CV_32SC1, _point_counts );
529 CvMat image_points, object_points;
530 CvMat dist_coeffs = cvMat( 4, 1, CV_32FC1, _distortion_coeffs );
531 CvMat camera_matrix = cvMat( 3, 3, CV_32FC1, _camera_matrix );
532 CvMat rotation_matrices = cvMat( image_count, 9, CV_32FC1, _rotation_matrices );
533 CvMat translation_vectors = cvMat( image_count, 3, CV_32FC1, _translation_vectors );
534
535 for( i = 0; i < image_count; i++ )
536 total += _point_counts[i];
537
538 image_points = cvMat( total, 1, CV_32FC2, _image_points );
539 object_points = cvMat( total, 1, CV_32FC3, _object_points );
540
541 cvCalibrateCamera2( &object_points, &image_points, &point_counts, image_size,
542 &camera_matrix, &dist_coeffs, &rotation_matrices, &translation_vectors,
543 flags );
544}

573/* Find 3d position of object given intrinsic camera parameters,
574 3d model of the object and projection of the object into view plane */
575void cvFindExtrinsicCameraParams( int point_count,
576 CvSize, CvPoint2D32f* _image_points,
577 CvPoint3D32f* _object_points, float* focal_length,
578 CvPoint2D32f principal_point, float* _distortion_coeffs,
579 float* _rotation_vector, float* _translation_vector )
580{
581 CvMat image_points = cvMat( point_count, 1, CV_32FC2, _image_points );
582 CvMat object_points = cvMat( point_count, 1, CV_32FC3, _object_points );
583 CvMat dist_coeffs = cvMat( 4, 1, CV_32FC1, _distortion_coeffs );
584 float a[9];
585 CvMat camera_matrix = cvMat( 3, 3, CV_32FC1, a );
586 CvMat rotation_vector = cvMat( 1, 1, CV_32FC3, _rotation_vector );
587 CvMat translation_vector = cvMat( 1, 1, CV_32FC3, _translation_vector );
588
589 a[0] = focal_length[0]; a[4] = focal_length[1];
590 a[2] = principal_point.x; a[5] = principal_point.y;
591 a[1] = a[3] = a[6] = a[7] = 0.f;
592 a[8] = 1.f;
593
594 cvFindExtrinsicCameraParams2( &object_points, &image_points, &camera_matrix,
595 &dist_coeffs, &rotation_vector, &translation_vector, 0 );
596}

599/* Variant of the previous function that takes double-precision parameters */
600void cvFindExtrinsicCameraParams_64d( int point_count,
601 CvSize, CvPoint2D64f* _image_points,
602 CvPoint3D64f* _object_points, double* focal_length,
603 CvPoint2D64f principal_point, double* _distortion_coeffs,
604 double* _rotation_vector, double* _translation_vector )
605{

607 CvMat object_points = cvMat( point_count, 1, CV_64FC3, _object_points );
608 CvMat dist_coeffs = cvMat( 4, 1, CV_64FC1, _distortion_coeffs );
609 double a[9];
610 CvMat camera_matrix = cvMat( 3, 3, CV_64FC1, a );
611 CvMat rotation_vector = cvMat( 1, 1, CV_64FC3, _rotation_vector );
612 CvMat translation_vector = cvMat( 1, 1, CV_64FC3, _translation_vector );
613
614 a[0] = focal_length[0]; a[4] = focal_length[1];
615 a[2] = principal_point.x; a[5] = principal_point.y;

620 &dist_coeffs, &rotation_vector, &translation_vector, 0 );

634/* Does reprojection of 3d object points to the view plane */
635void cvProjectPoints( int point_count, CvPoint3D64f* _object_points,
636 double* _rotation_vector, double* _translation_vector,
637 double* focal_length, CvPoint2D64f principal_point,
638 double* _distortion, CvPoint2D64f* _image_points,
639 double* _deriv_points_rotation_matrix,
640 double* _deriv_points_translation_vect,
641 double* _deriv_points_focal,
642 double* _deriv_points_principal_point,
643 double* _deriv_points_distortion_coeffs )
644{
645 CvMat object_points = cvMat( point_count, 1, CV_64FC3, _object_points );
646 CvMat image_points = cvMat( point_count, 1, CV_64FC2, _image_points );
647 CvMat rotation_vector = cvMat( 3, 1, CV_64FC1, _rotation_vector );
648 CvMat translation_vector = cvMat( 3, 1, CV_64FC1, _translation_vector );
649 double a[9];
650 CvMat camera_matrix = cvMat( 3, 3, CV_64FC1, a );
651 CvMat dist_coeffs = cvMat( 4, 1, CV_64FC1, _distortion );
652 CvMat dpdr = cvMat( 2*point_count, 3, CV_64FC1, _deriv_points_rotation_matrix );
653 CvMat dpdt = cvMat( 2*point_count, 3, CV_64FC1, _deriv_points_translation_vect );
654 CvMat dpdf = cvMat( 2*point_count, 2, CV_64FC1, _deriv_points_focal );
655 CvMat dpdc = cvMat( 2*point_count, 2, CV_64FC1, _deriv_points_principal_point );
656 CvMat dpdk = cvMat( 2*point_count, 4, CV_64FC1, _deriv_points_distortion_coeffs );
657
658 a[0] = focal_length[0]; a[4] = focal_length[1];
659 a[2] = principal_point.x; a[5] = principal_point.y;
660 a[1] = a[3] = a[6] = a[7] = 0.;
661 a[8] = 1.;
