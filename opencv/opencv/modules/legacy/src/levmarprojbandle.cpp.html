74static void icvComputeDerivateProj(CvMat *points4D,CvMat *projMatr, CvMat *status, CvMat *derivProj)
75{
76 /* Compute derivate for given projection matrix points and status of points */
77
78 CV_FUNCNAME( "icvComputeDerivateProj" );
79 __BEGIN__;
80
81
82 /* ----- Test input params for errors ----- */
83 if( points4D == 0 || projMatr == 0 || status == 0 || derivProj == 0)
84 {
85 CV_ERROR( CV_StsNullPtr, "Some of parameters is a NULL pointer" );
86 }
87
88 if( !CV_IS_MAT(points4D) )
89 {
90 CV_ERROR( CV_StsUnsupportedFormat, "points4D must be a matrix 4xN" );
91 }
92
93 /* Compute number of points */
94 int numPoints;
95 numPoints = points4D->cols;
96
97 if( numPoints < 1 )
98 {
99 CV_ERROR( CV_StsOutOfRange, "Number of points4D must be more than zero" );
100 }
101
102 if( points4D->rows != 4 )
103 {
104 CV_ERROR( CV_StsOutOfRange, "Number of coordinates of points4D must be 4" );
105 }
106
107 if( !CV_IS_MAT(projMatr) )
108 {
109 CV_ERROR( CV_StsUnsupportedFormat, "projMatr must be a matrix 3x4" );
110 }
111
112 if( projMatr->rows != 3 || projMatr->cols != 4 )
113 {
114 CV_ERROR( CV_StsOutOfRange, "Size of projection matrix (projMatr) must be 3x4" );
115 }
116
117 if( !CV_IS_MAT(status) )
118 {
119 CV_ERROR( CV_StsUnsupportedFormat, "Status must be a matrix 1xN" );
120 }
121
122 if( status->rows != 1 || status->cols != numPoints )
123 {
124 CV_ERROR( CV_StsOutOfRange, "Size of status of points must be 1xN" );
125 }
126
127 if( !CV_IS_MAT(derivProj) )
128 {
129 CV_ERROR( CV_StsUnsupportedFormat, "derivProj must be a matrix VisN x 12" );
130 }
131
132 if( derivProj->cols != 12 )
133 {
134 CV_ERROR( CV_StsOutOfRange, "derivProj must be a matrix VisN x 12" );
135 }
136 /* ----- End test ----- */
137
138 /* Allocate memory for derivates */
139
140 double p[12];
141 /* Copy projection matrix */
142 for(int i = 0; i < 12; i++ )
143 {
144 p[i] = cvmGet(projMatr,i/4,i%4);
145 }
146
147 /* Fill deriv matrix */
148 int currVisPoint;
149 int currPoint;
150
151 currVisPoint = 0;
152 for( currPoint = 0; currPoint < numPoints; currPoint++ )
153 {
154 if( cvmGet(status,0,currPoint) > 0 )
155 {
156 double X[4];
157 X[0] = cvmGet(points4D,0,currVisPoint);
158 X[1] = cvmGet(points4D,1,currVisPoint);
159 X[2] = cvmGet(points4D,2,currVisPoint);
160 X[3] = cvmGet(points4D,3,currVisPoint);
161
162 /* Compute derivate for this point */
163
164 double piX[3];
165 piX[0] = X[0]*p[0] + X[1]*p[1] + X[2]*p[2] + X[3]*p[3];
166 piX[1] = X[0]*p[4] + X[1]*p[5] + X[2]*p[6] + X[3]*p[7];
167 piX[2] = X[0]*p[8] + X[1]*p[9] + X[2]*p[10] + X[3]*p[11];
168
169 /* fill derivate by point */
170
171 double tmp3 = 1/(piX[2]*piX[2]);
172
173 double tmp1 = -piX[0]*tmp3;
174 double tmp2 = -piX[1]*tmp3;
175
176 /* fill derivate by projection matrix */
177 for(int i = 0; i < 4; i++ )
178 {
179 /* derivate for x */
180 cvmSet(derivProj,currVisPoint*2,i,X[i]/piX[2]);//x' p1i
181 cvmSet(derivProj,currVisPoint*2,4+i,0);//x' p1i
182 cvmSet(derivProj,currVisPoint*2,8+i,X[i]*tmp1);//x' p3i
183
184 /* derivate for y */
185 cvmSet(derivProj,currVisPoint*2+1,i,0);//y' p2i
186 cvmSet(derivProj,currVisPoint*2+1,4+i,X[i]/piX[2]);//y' p2i
187 cvmSet(derivProj,currVisPoint*2+1,8+i,X[i]*tmp2);//y' p3i
188 }
189
190 currVisPoint++;
191 }
192 }
193
194 if( derivProj->rows != currVisPoint * 2 )
195 {
196 CV_ERROR( CV_StsOutOfRange, "derivProj must be a matrix 2VisN x 12" );
197 }
198
199
200 __END__;
201 return;
202}

232static void icvComputeDerivatePoints(CvMat *points4D,CvMat *projMatr, CvMat *presPoints, CvMat *derivPoint)
233{
234
235 CV_FUNCNAME( "icvComputeDerivatePoints" );
236 __BEGIN__;
237
238 /* ----- Test input params for errors ----- */
239 if( points4D == 0 || projMatr == 0 || presPoints == 0 || derivPoint == 0)
240 {
241 CV_ERROR( CV_StsNullPtr, "Some of parameters is a NULL pointer" );
242 }
243
244 if( !CV_IS_MAT(points4D) )
245 {
246 CV_ERROR( CV_StsUnsupportedFormat, "points4D must be a matrix N x 4" );
247 }
248
249 int numPoints;
250 numPoints = presPoints->cols;
251
252 if( numPoints < 1 )
253 {
254 CV_ERROR( CV_StsOutOfRange, "Number of points must be more than zero" );
255 }
256
257 if( points4D->rows != 4 )
258 {
259 CV_ERROR( CV_StsOutOfRange, "points4D must be a matrix N x 4" );
260 }
261
262 if( !CV_IS_MAT(projMatr) )
263 {
264 CV_ERROR( CV_StsUnsupportedFormat, "projMatr must be a matrix 3x4" );
265 }
266
267 if( projMatr->rows != 3 || projMatr->cols != 4 )
268 {
269 CV_ERROR( CV_StsOutOfRange, "Size of projection matrix (projMatr) must be 3x4" );
270 }
271
272 if( !CV_IS_MAT(presPoints) )
273 {
274 CV_ERROR( CV_StsUnsupportedFormat, "Status must be a matrix 1xN" );
275 }
276
277 if( presPoints->rows != 1 || presPoints->cols != numPoints )
278 {
279 CV_ERROR( CV_StsOutOfRange, "Size of presPoints status must be 1xN" );
280 }
281
282 if( !CV_IS_MAT(derivPoint) )
283 {
284 CV_ERROR( CV_StsUnsupportedFormat, "derivPoint must be a matrix 2 x 4VisNum" );
285 }
286 /* ----- End test ----- */
287
288 /* Compute derivates by points */
289
290 double p[12];
291 for(int i = 0; i < 12; i++ )
292 {
293 p[i] = cvmGet(projMatr,i/4,i%4);
294 }
295
296 int currVisPoint;
297 int currProjPoint;
298
299 currVisPoint = 0;
300 for( currProjPoint = 0; currProjPoint < numPoints; currProjPoint++ )
301 {
302 if( cvmGet(presPoints,0,currProjPoint) > 0 )
303 {
304 double X[4];
305 X[0] = cvmGet(points4D,0,currProjPoint);
306 X[1] = cvmGet(points4D,1,currProjPoint);
307 X[2] = cvmGet(points4D,2,currProjPoint);
308 X[3] = cvmGet(points4D,3,currProjPoint);
309
310 double piX[3];
311 piX[0] = X[0]*p[0] + X[1]*p[1] + X[2]*p[2] + X[3]*p[3];
312 piX[1] = X[0]*p[4] + X[1]*p[5] + X[2]*p[6] + X[3]*p[7];
313 piX[2] = X[0]*p[8] + X[1]*p[9] + X[2]*p[10] + X[3]*p[11];
314
315 double tmp3 = 1/(piX[2]*piX[2]);
316
317 for(int j = 0; j < 2; j++ )//for x and y
318 {
319 for(int i = 0; i < 4; i++ )// for X,Y,Z,W
320 {
321 cvmSet( derivPoint,
322 j, currVisPoint*4+i,
323 (p[j*4+i]*piX[2]-p[8+i]*piX[j]) * tmp3 );
324 }
325 }
326 currVisPoint++;
327 }
328 }
329
330 if( derivPoint->rows != 2 || derivPoint->cols != currVisPoint*4 )
331 {
332 CV_ERROR( CV_StsUnsupportedFormat, "derivPoint must be a matrix 2 x 4VisNum" );
333 }
334
335 __END__;
336 return;
337}

366static void icvComputeMatrixVAll(int numImages,CvMat **pointDeriv,CvMat **presPoints, CvMat **matrV)
367{
368 int *shifts = 0;
369
370 CV_FUNCNAME( "icvComputeMatrixVAll" );
371 __BEGIN__;
372
373 /* ----- Test input params for errors ----- */
374 if( numImages < 1 )
375 {
376 CV_ERROR( CV_StsOutOfRange, "Number of images must more than zero" );
377 }
378 if( pointDeriv == 0 || presPoints == 0 || matrV == 0 )
379 {
380 CV_ERROR( CV_StsNullPtr, "Some of parameters is a NULL pointer" );
381 }
382 /* !!! not tested all parameters */
383 /* ----- End test ----- */
384
385 /* Compute all matrices U */
386 int currImage;
387 int currPoint;
388 int numPoints;
389 numPoints = presPoints[0]->cols;
390 CV_CALL(shifts = (int*)cvAlloc(sizeof(int)*numImages));
391 memset(shifts,0,sizeof(int)*numImages);
392
393 for( currPoint = 0; currPoint < numPoints; currPoint++ )//For each point (matrix V)
394 {
395 int i,j;
396
397 for( i = 0; i < 4; i++ )
398 {
399 for( j = 0; j < 4; j++ )
400 {
401 double sum = 0;
402 for( currImage = 0; currImage < numImages; currImage++ )
403 {
404 if( cvmGet(presPoints[currImage],0,currPoint) > 0 )
405 {
406 sum += cvmGet(pointDeriv[currImage],0,shifts[currImage]*4+i) *
407 cvmGet(pointDeriv[currImage],0,shifts[currImage]*4+j);
408
409 sum += cvmGet(pointDeriv[currImage],1,shifts[currImage]*4+i) *
410 cvmGet(pointDeriv[currImage],1,shifts[currImage]*4+j);
411 }
412 }
413
414 cvmSet(matrV[currPoint],i,j,sum);
415 }
416 }
417
418
419 /* shift position of visible points */
420 for( currImage = 0; currImage < numImages; currImage++ )
421 {
422 if( cvmGet(presPoints[currImage],0,currPoint) > 0 )
423 {
424 shifts[currImage]++;
425 }
426 }
427 }
428
429 __END__;
430 cvFree( &shifts);
431
432 return;
433}

507
