102float
103CvEM::predict( const CvMat* _sample, CvMat* _probs ) const
104{
105 Mat prbs0 = cvarrToMat(_probs), prbs = prbs0, sample = cvarrToMat(_sample);
106 int cls = static_cast<int>(emObj.predict(sample, _probs ? _OutputArray(prbs) : cv::noArray())[1]);
107 if(_probs)
108 {
109 if( prbs.data != prbs0.data )
110 {
111 CV_Assert( prbs.size == prbs0.size );
112 prbs.convertTo(prbs0, prbs0.type());
113 }
114 }
115 return (float)cls;
116}

118void CvEM::set_mat_hdrs()
119{
120 if(emObj.isTrained())
121 {
122 meansHdr = emObj.get<Mat>("means");
123 int K = emObj.get<int>("nclusters");
124 covsHdrs.resize(K);
125 covsPtrs.resize(K);
126 const std::vector<Mat>& covs = emObj.get<vector<Mat> >("covs");
127 for(size_t i = 0; i < covsHdrs.size(); i++)
128 {
129 covsHdrs[i] = covs[i];
130 covsPtrs[i] = &covsHdrs[i];
131 }
132 weightsHdr = emObj.get<Mat>("weights");
133 probsHdr = probs;
134 }
135}

137static
138void init_params(const CvEMParams& src,
139 Mat& prbs, Mat& weights,
140 Mat& means, vector<Mat>& covsHdrs)
141{
142 prbs = src.probs;
143 weights = src.weights;
144 means = src.means;
145
146 if(src.covs)
147 {
148 covsHdrs.resize(src.nclusters);
149 for(size_t i = 0; i < covsHdrs.size(); i++)
150 covsHdrs[i] = src.covs[i];
151 }
152}

154bool CvEM::train( const CvMat* _samples, const CvMat* _sample_idx,
155 CvEMParams _params, CvMat* _labels )
156{
157 CV_Assert(_sample_idx == 0);
158 Mat samples = cvarrToMat(_samples), labels0, labels;
159 if( _labels )
160 labels0 = labels = cvarrToMat(_labels);
161
162 bool isOk = train(samples, Mat(), _params, _labels ? &labels : 0);
163 CV_Assert( labels0.data == labels.data );
164
165 return isOk;
166}

200bool CvEM::train( const Mat& _samples, const Mat& _sample_idx,
201 CvEMParams _params, Mat* _labels )
202{
203 CV_Assert(_sample_idx.empty());
204 Mat prbs, weights, means, logLikelihoods;
205 std::vector<Mat> covshdrs;
206 init_params(_params, prbs, weights, means, covshdrs);
207
208 emObj = EM(_params.nclusters, _params.cov_mat_type, _params.term_crit);
209 bool isOk = false;
210 if( _params.start_step == EM::START_AUTO_STEP )
211 isOk = emObj.train(_samples,
212 logLikelihoods, _labels ? _OutputArray(*_labels) : cv::noArray(), probs);
213 else if( _params.start_step == EM::START_E_STEP )
214 isOk = emObj.trainE(_samples, means, covshdrs, weights,
215 logLikelihoods, _labels ? _OutputArray(*_labels) : cv::noArray(), probs);
216 else if( _params.start_step == EM::START_M_STEP )
217 isOk = emObj.trainM(_samples, prbs,
218 logLikelihoods, _labels ? _OutputArray(*_labels) : cv::noArray(), probs);
219 else
220 CV_Error(CV_StsBadArg, "Bad start type of EM algorithm");
221
222 if(isOk)
223 {
224 logLikelihood = sum(logLikelihoods).val[0];
225 set_mat_hdrs();
226 }
227
228 return isOk;
229}
