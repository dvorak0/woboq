61// cv3dTrackerCalibrateCameras //
62/////////////////////////////////
63CV_IMPL CvBool cv3dTrackerCalibrateCameras(int num_cameras,
64 const Cv3dTrackerCameraIntrinsics camera_intrinsics[], // size is num_cameras
65 CvSize etalon_size,
66 float square_size,
67 IplImage *samples[], // size is num_cameras
68 Cv3dTrackerCameraInfo camera_info[]) // size is num_cameras
69{
70 CV_FUNCNAME("cv3dTrackerCalibrateCameras");
71 const int num_points = etalon_size.width * etalon_size.height;
72 int cameras_done = 0; // the number of cameras whose positions have been determined
73 CvPoint3D32f *object_points = NULL; // real-world coordinates of checkerboard points
74 CvPoint2D32f *points = NULL; // 2d coordinates of checkerboard points as seen by a camera
75 IplImage *gray_img = NULL; // temporary image for color conversion
76 IplImage *tmp_img = NULL; // temporary image used by FindChessboardCornerGuesses
77 int c, i, j;
78
79 if (etalon_size.width < 3 || etalon_size.height < 3)
80 CV_ERROR(CV_StsBadArg, "Chess board size is invalid");
81
82 for (c = 0; c < num_cameras; c++)
83 {
84 // CV_CHECK_IMAGE is not available in the cvaux library
85 // so perform the checks inline.
86
87 //CV_CALL(CV_CHECK_IMAGE(samples[c]));
88
89 if( samples[c] == NULL )
90 CV_ERROR( CV_HeaderIsNull, "Null image" );
91
92 if( samples[c]->dataOrder != IPL_DATA_ORDER_PIXEL && samples[c]->nChannels > 1 )
93 CV_ERROR( CV_BadOrder, "Unsupported image format" );
94
95 if( samples[c]->maskROI != 0 || samples[c]->tileInfo != 0 )
96 CV_ERROR( CV_StsBadArg, "Unsupported image format" );
97
98 if( samples[c]->imageData == 0 )
99 CV_ERROR( CV_BadDataPtr, "Null image data" );
100
101 if( samples[c]->roi &&
102 ((samples[c]->roi->xOffset | samples[c]->roi->yOffset
103 | samples[c]->roi->width | samples[c]->roi->height) < 0 ||
104 samples[c]->roi->xOffset + samples[c]->roi->width > samples[c]->width ||
105 samples[c]->roi->yOffset + samples[c]->roi->height > samples[c]->height ||
106 (unsigned) (samples[c]->roi->coi) > (unsigned) (samples[c]->nChannels)))
107 CV_ERROR( CV_BadROISize, "Invalid ROI" );
108
109 // End of CV_CHECK_IMAGE inline expansion
110
111 if (samples[c]->depth != IPL_DEPTH_8U)
112 CV_ERROR(CV_BadDepth, "Channel depth of source image must be 8");
113
114 if (samples[c]->nChannels != 3 && samples[c]->nChannels != 1)
115 CV_ERROR(CV_BadNumChannels, "Source image must have 1 or 3 channels");
116 }
117
118 CV_CALL(object_points = (CvPoint3D32f *)cvAlloc(num_points * sizeof(CvPoint3D32f)));
119 CV_CALL(points = (CvPoint2D32f *)cvAlloc(num_points * sizeof(CvPoint2D32f)));
120
121 // fill in the real-world coordinates of the checkerboard points
122 FillObjectPoints(object_points, etalon_size, square_size);
123
124 for (c = 0; c < num_cameras; c++)
125 {
126 CvSize image_size = cvSize(samples[c]->width, samples[c]->height);
127 IplImage *img;
128
129 // The input samples are not required to all have the same size or color
130 // format. If they have different sizes, the temporary images are
131 // reallocated as necessary.
132 if (samples[c]->nChannels == 3)
133 {
134 // convert to gray
135 if (gray_img == NULL || gray_img->width != samples[c]->width ||
136 gray_img->height != samples[c]->height )
137 {
138 if (gray_img != NULL)
139 cvReleaseImage(&gray_img);
140 CV_CALL(gray_img = cvCreateImage(image_size, IPL_DEPTH_8U, 1));
141 }
142
143 CV_CALL(cvCvtColor(samples[c], gray_img, CV_BGR2GRAY));
144
145 img = gray_img;
146 }
147 else
148 {
149 // no color conversion required
150 img = samples[c];
151 }
152
153 if (tmp_img == NULL || tmp_img->width != samples[c]->width ||
154 tmp_img->height != samples[c]->height )
155 {
156 if (tmp_img != NULL)
157 cvReleaseImage(&tmp_img);
158 CV_CALL(tmp_img = cvCreateImage(image_size, IPL_DEPTH_8U, 1));
159 }
160
161 int count = num_points;
162 bool found = cvFindChessBoardCornerGuesses(img, tmp_img, 0,
163 etalon_size, points, &count) != 0;
164 if (count == 0)
165 continue;
166
167 // If found is true, it means all the points were found (count = num_points).
168 // If found is false but count is non-zero, it means that not all points were found.
169
170 cvFindCornerSubPix(img, points, count, cvSize(5,5), cvSize(-1,-1),
171 cvTermCriteria(CV_TERMCRIT_ITER|CV_TERMCRIT_EPS, 10, 0.01f));
172
173 // If the image origin is BL (bottom-left), fix the y coordinates
174 // so they are relative to the true top of the image.
175 if (samples[c]->origin == IPL_ORIGIN_BL)
176 {
177 for (i = 0; i < count; i++)
178 points[i].y = samples[c]->height - 1 - points[i].y;
179 }
180
181 if (found)
182 {
183 // Make sure x coordinates are increasing and y coordinates are decreasing.
184 // (The y coordinate of point (0,0) should be the greatest, because the point
185 // on the checkerboard that is the origin is nearest the bottom of the image.)
186 // This is done after adjusting the y coordinates according to the image origin.
187 if (points[0].x > points[1].x)
188 {
189 // reverse points in each row
190 for (j = 0; j < etalon_size.height; j++)
191 {
192 CvPoint2D32f *row = &points[j*etalon_size.width];
193 for (i = 0; i < etalon_size.width/2; i++)
194 std::swap(row[i], row[etalon_size.width-i-1]);
195 }
196 }
197
198 if (points[0].y < points[etalon_size.width].y)
199 {
200 // reverse points in each column
201 for (i = 0; i < etalon_size.width; i++)
202 {
203 for (j = 0; j < etalon_size.height/2; j++)
204 std::swap(points[i+j*etalon_size.width],
205 points[i+(etalon_size.height-j-1)*etalon_size.width]);
206 }
207 }
208 }
209
210 DrawEtalon(samples[c], points, count, etalon_size, found);
211
212 if (!found)
213 continue;
214
215 float rotVect[3];
216 float rotMatr[9];
217 float transVect[3];
218
219 cvFindExtrinsicCameraParams(count,
220 image_size,
221 points,
222 object_points,
223 const_cast<float *>(camera_intrinsics[c].focal_length),
224 camera_intrinsics[c].principal_point,
225 const_cast<float *>(camera_intrinsics[c].distortion),
226 rotVect,
227 transVect);
228
229 // Check result against an arbitrary limit to eliminate impossible values.
230 // (If the chess board were truly that far away, the camera wouldn't be able to
231 // see the squares.)
232 if (transVect[0] > 1000*square_size
233 || transVect[1] > 1000*square_size
234 || transVect[2] > 1000*square_size)
235 {
236 // ignore impossible results
237 continue;
238 }
239
240 CvMat rotMatrDescr = cvMat(3, 3, CV_32FC1, rotMatr);
241 CvMat rotVectDescr = cvMat(3, 1, CV_32FC1, rotVect);
242
243 /* Calc rotation matrix by Rodrigues Transform */
244 cvRodrigues2( &rotVectDescr, &rotMatrDescr );
245
246 //combine the two transformations into one matrix
247 //order is important! rotations are not commutative
248 float tmat[4][4] = { { 1.f, 0.f, 0.f, 0.f },
249 { 0.f, 1.f, 0.f, 0.f },
250 { 0.f, 0.f, 1.f, 0.f },
251 { transVect[0], transVect[1], transVect[2], 1.f } };
252
253 float rmat[4][4] = { { rotMatr[0], rotMatr[1], rotMatr[2], 0.f },
254 { rotMatr[3], rotMatr[4], rotMatr[5], 0.f },
255 { rotMatr[6], rotMatr[7], rotMatr[8], 0.f },
256 { 0.f, 0.f, 0.f, 1.f } };
257
258
259 MultMatrix(camera_info[c].mat, tmat, rmat);
260
261 // change the transformation of the cameras to put them in the world coordinate
262 // system we want to work with.
263
264 // Start with an identity matrix; then fill in the values to accomplish
265 // the desired transformation.
266 float smat[4][4] = { { 1.f, 0.f, 0.f, 0.f },
267 { 0.f, 1.f, 0.f, 0.f },
268 { 0.f, 0.f, 1.f, 0.f },
269 { 0.f, 0.f, 0.f, 1.f } };
270
271 // First, reflect through the origin by inverting all three axes.
272 smat[0][0] = -1.f;
273 smat[1][1] = -1.f;
274 smat[2][2] = -1.f;
275 MultMatrix(tmat, camera_info[c].mat, smat);
276
277 // Scale x and y coordinates by the focal length (allowing for non-square pixels
278 // and/or non-symmetrical lenses).
279 smat[0][0] = 1.0f / camera_intrinsics[c].focal_length[0];
280 smat[1][1] = 1.0f / camera_intrinsics[c].focal_length[1];
281 smat[2][2] = 1.0f;
282 MultMatrix(camera_info[c].mat, smat, tmat);
283
284 camera_info[c].principal_point = camera_intrinsics[c].principal_point;
285 camera_info[c].valid = true;
286
287 cameras_done++;
288 }
289
290exit:
291 cvReleaseImage(&gray_img);
292 cvReleaseImage(&tmp_img);
293 cvFree(&object_points);
294 cvFree(&points);
295
296 return cameras_done == num_cameras;
297}

299// fill in the real-world coordinates of the checkerboard points
300static void FillObjectPoints(CvPoint3D32f *obj_points, CvSize etalon_size, float square_size)
301{
302 int x, y, i;
303
304 for (y = 0, i = 0; y < etalon_size.height; y++)
305 {
306 for (x = 0; x < etalon_size.width; x++, i++)
307 {
308 obj_points[i].x = square_size * x;
309 obj_points[i].y = square_size * y;
310 obj_points[i].z = 0;
311 }
312 }
313}

316// Mark the points found on the input image
317// The marks are drawn multi-colored if all the points were found.
318static void DrawEtalon(IplImage *img, CvPoint2D32f *corners,
319 int corner_count, CvSize etalon_size, int draw_ordered)
320{
321 const int r = 4;
322 int i;
323 int x, y;
324 CvPoint prev_pt = { 0, 0 };
325 static const CvScalar rgb_colors[] = {
326 {{0,0,255}},
327 {{0,128,255}},
328 {{0,200,200}},
329 {{0,255,0}},
330 {{200,200,0}},
331 {{255,0,0}},
332 {{255,0,255}} };
333 static const CvScalar gray_colors[] = {
334 {{80}}, {{120}}, {{160}}, {{200}}, {{100}}, {{140}}, {{180}}
335 };
336 const CvScalar* colors = img->nChannels == 3 ? rgb_colors : gray_colors;
337
338 CvScalar color = colors[0];
339 for (y = 0, i = 0; y < etalon_size.height; y++)
340 {
341 if (draw_ordered)
342 color = colors[y % ARRAY_SIZEOF(rgb_colors)];
343
344 for (x = 0; x < etalon_size.width && i < corner_count; x++, i++)
345 {
346 CvPoint pt;
347 pt.x = cvRound(corners[i].x);
348 pt.y = cvRound(corners[i].y);
349 if (img->origin == IPL_ORIGIN_BL)
350 pt.y = img->height - 1 - pt.y;
351
352 if (draw_ordered)
353 {
354 if (i != 0)
355 cvLine(img, prev_pt, pt, color, 1, CV_AA);
356 prev_pt = pt;
357 }
358
359 cvLine( img, cvPoint(pt.x - r, pt.y - r),
360 cvPoint(pt.x + r, pt.y + r), color, 1, CV_AA );
361 cvLine( img, cvPoint(pt.x - r, pt.y + r),
362 cvPoint(pt.x + r, pt.y - r), color, 1, CV_AA );
363 cvCircle( img, pt, r+1, color, 1, CV_AA );
364 }
365 }

369static CvPoint3D32f midpoint(const CvPoint3D32f &p1, const CvPoint3D32f &p2)
370{
371 return cvPoint3D32f((p1.x+p2.x)/2, (p1.y+p2.y)/2, (p1.z+p2.z)/2);

398//////////////////////////////
399// cv3dTrackerLocateObjects //
400//////////////////////////////
401CV_IMPL int cv3dTrackerLocateObjects(int num_cameras, int num_objects,
402 const Cv3dTrackerCameraInfo camera_info[], // size is num_cameras
403 const Cv3dTracker2dTrackedObject tracking_info[], // size is num_objects*num_cameras
404 Cv3dTrackerTrackedObject tracked_objects[]) // size is num_objects
405{
406 /*CV_FUNCNAME("cv3dTrackerLocateObjects");*/
407 int found_objects = 0;
408
409 // count how many cameras could see each object
410 std::map<int, int> count;
411 for (int c = 0; c < num_cameras; c++)
412 {
413 if (!camera_info[c].valid)
414 continue;
415
416 for (int i = 0; i < num_objects; i++)
417 {
418 const Cv3dTracker2dTrackedObject *o = &tracking_info[c*num_objects+i];
419 if (o->id != -1)
420 count[o->id]++;
421 }
422 }
423
424 // process each object that was seen by at least two cameras
425 for (std::map<int, int>::iterator i = count.begin(); i != count.end(); i++)
426 {
427 if (i->second < 2)
428 continue; // ignore object seen by only one camera
429 int id = i->first;
430
431 // find an approximation of the objects location for each pair of cameras that
432 // could see this object, and average them
433 CvPoint3D32f total = cvPoint3D32f(0, 0, 0);
434 int weight = 0;
435
436 for (int c1 = 0; c1 < num_cameras-1; c1++)
437 {
438 if (!camera_info[c1].valid)
439 continue;
440
441 const Cv3dTracker2dTrackedObject *o1 = find(&tracking_info[c1*num_objects],
442 num_objects, id);
443 if (o1 == NULL)
444 continue; // this camera didn't see this object
445
446 CvPoint3D32f p1a = CAMERA_POS(camera_info[c1]);
447 CvPoint3D32f p1b = ImageCStoWorldCS(camera_info[c1], o1->p);
448
449 for (int c2 = c1 + 1; c2 < num_cameras; c2++)
450 {
451 if (!camera_info[c2].valid)
452 continue;
453
454 const Cv3dTracker2dTrackedObject *o2 = find(&tracking_info[c2*num_objects],
455 num_objects, id);
456 if (o2 == NULL)
457 continue; // this camera didn't see this object
458
459 CvPoint3D32f p2a = CAMERA_POS(camera_info[c2]);
460 CvPoint3D32f p2b = ImageCStoWorldCS(camera_info[c2], o2->p);
461
462 // these variables are initialized simply to avoid erroneous error messages
463 // from the compiler
464 CvPoint3D32f r1 = cvPoint3D32f(0, 0, 0);
465 CvPoint3D32f r2 = cvPoint3D32f(0, 0, 0);
466
467 // find the intersection of the two lines (or the points of closest
468 // approach, if they don't intersect)
469 if (!intersection(p1a, p1b, p2a, p2b, r1, r2))
470 continue;
471
472 total += midpoint(r1, r2);
473 weight++;
474 }
475 }
476
477 CvPoint3D32f center = total/weight;
478 tracked_objects[found_objects++] = cv3dTrackerTrackedObject(id, center);
479 }
480
481 return found_objects;
482}

486// Compute the determinant of the 3x3 matrix represented by 3 row vectors.
487static inline double det(CvPoint3D32f v1, CvPoint3D32f v2, CvPoint3D32f v3)
488{
489 return v1.x*v2.y*v3.z + v1.z*v2.x*v3.y + v1.y*v2.z*v3.x
490 - v1.z*v2.y*v3.x - v1.x*v2.z*v3.y - v1.y*v2.x*v3.z;
491}

509// Find the intersection of two lines, or if they don't intersect,
510// the points of closest approach.
511// The lines are defined by (o1,p1) and (o2, p2).
512// If they intersect, r1 and r2 will be the same.
513// Returns false on error.
514static bool intersection(CvPoint3D32f o1, CvPoint3D32f p1,
515 CvPoint3D32f o2, CvPoint3D32f p2,
516 CvPoint3D32f &r1, CvPoint3D32f &r2)
517{
518 CvPoint3D32f x = o2 - o1;
519 CvPoint3D32f d1 = p1 - o1;
520 CvPoint3D32f d2 = p2 - o2;
521
522 CvPoint3D32f cross = cvPoint3D32f(d1.y*d2.z - d1.z*d2.y,
523 d1.z*d2.x - d1.x*d2.z,
524 d1.x*d2.y - d1.y*d2.x);
525 double den = cross.x*cross.x + cross.y*cross.y + cross.z*cross.z;
526
527 if (den < EPS)
528 return false;
529
530 double t1 = det(x, d2, cross) / den;
531 double t2 = det(x, d1, cross) / den;
532
533 r1 = o1 + d1 * t1;
534 r2 = o2 + d2 * t2;
535
536 return true;
537}

539// Convert from image to camera space by transforming point p in
540// the image plane by the camera matrix.
541static CvPoint3D32f ImageCStoWorldCS(const Cv3dTrackerCameraInfo &camera_info, CvPoint2D32f p)
542{
543 float tp[4];
544 tp[0] = (float)p.x - camera_info.principal_point.x;
545 tp[1] = (float)p.y - camera_info.principal_point.y;
546 tp[2] = 1.f;
547 tp[3] = 1.f;
548
549 float tr[4];
550 //multiply tp by mat to get tr
551 MultVectorMatrix(tr, tp, camera_info.mat);
552
553 return cvPoint3D32f(tr[0]/tr[3], tr[1]/tr[3], tr[2]/tr[3]);
554}

556// Multiply affine transformation m1 by the affine transformation m2 and
557// return the result in rm.
558static void MultMatrix(float rm[4][4], const float m1[4][4], const float m2[4][4])
559{
560 for (int i=0; i<=3; i++)
561 for (int j=0; j<=3; j++)
562 {
563 rm[i][j]= 0.0;
564 for (int k=0; k <= 3; k++)
565 rm[i][j] += m1[i][k]*m2[k][j];
566 }
567}
568
569// Multiply the vector v by the affine transformation matrix m and return the
570// result in rv.
571void MultVectorMatrix(float rv[4], const float v[4], const float m[4][4])
572{
573 for (int i=0; i<=3; i++)
574 {
575 rv[i] = 0.f;
576 for (int j=0;j<=3;j++)
577 rv[i] += v[j] * m[j][i];
578 }
