100void PatchGenerator::generateRandomTransform(Point2f srcCenter, Point2f dstCenter,
101 Mat& transform, RNG& rng, bool inverse) const
102{
103 double lambda1 = rng.uniform(lambdaMin, lambdaMax);
104 double lambda2 = rng.uniform(lambdaMin, lambdaMax);
105 double theta = rng.uniform(thetaMin, thetaMax);
106 double phi = rng.uniform(phiMin, phiMax);
107
108 // Calculate random parameterized affine transformation A,
109 // A = T(patch center) * R(theta) * R(phi)' *
110 // S(lambda1, lambda2) * R(phi) * T(-pt)
111 double st = sin(theta);
112 double ct = cos(theta);
113 double sp = sin(phi);
114 double cp = cos(phi);
115 double c2p = cp*cp;
116 double s2p = sp*sp;
117
118 double A = lambda1*c2p + lambda2*s2p;
119 double B = (lambda2 - lambda1)*sp*cp;
120 double C = lambda1*s2p + lambda2*c2p;
121
122 double Ax_plus_By = A*srcCenter.x + B*srcCenter.y;
123 double Bx_plus_Cy = B*srcCenter.x + C*srcCenter.y;
124
125 transform.create(2, 3, CV_64F);
126 Mat_<double>& T = (Mat_<double>&)transform;
127 T(0,0) = A*ct - B*st;
128 T(0,1) = B*ct - C*st;
129 T(0,2) = -ct*Ax_plus_By + st*Bx_plus_Cy + dstCenter.x;
130 T(1,0) = A*st + B*ct;
131 T(1,1) = B*st + C*ct;
132 T(1,2) = -st*Ax_plus_By - ct*Bx_plus_Cy + dstCenter.y;
133
134 if( inverse )
135 invertAffineTransform(T, T);
136}

149void PatchGenerator::operator ()(const Mat& image, const Mat& T,
150 Mat& patch, Size patchSize, RNG& rng) const
151{
152 patch.create( patchSize, image.type() );
153 if( backgroundMin != backgroundMax )
154 {
155 rng.fill(patch, RNG::UNIFORM, Scalar::all(backgroundMin), Scalar::all(backgroundMax));
156 warpAffine(image, patch, T, patchSize, INTER_LINEAR, BORDER_TRANSPARENT);
157 }
158 else
159 warpAffine(image, patch, T, patchSize, INTER_LINEAR, BORDER_CONSTANT, Scalar::all(backgroundMin));
160
161 int ksize = randomBlur ? (unsigned)rng % 9 - 5 : 0;
162 if( ksize > 0 )
163 {
164 ksize = ksize*2 + 1;
165 GaussianBlur(patch, patch, Size(ksize, ksize), 0, 0);
166 }
167
168 if( noiseRange > 0 )
169 {
170 AutoBuffer<uchar> _noiseBuf( patchSize.width*patchSize.height*image.elemSize() );
171 Mat noise(patchSize, image.type(), (uchar*)_noiseBuf);
172 int delta = image.depth() == CV_8U ? 128 : image.depth() == CV_16U ? 32768 : 0;
173 rng.fill(noise, RNG::NORMAL, Scalar::all(delta), Scalar::all(noiseRange));
174 if( backgroundMin != backgroundMax )
175 addWeighted(patch, 1, noise, 1, -delta, patch);
176 else
177 {
178 for( int i = 0; i < patchSize.height; i++ )
179 {
180 uchar* prow = patch.ptr<uchar>(i);
181 const uchar* nrow = noise.ptr<uchar>(i);
182 for( int j = 0; j < patchSize.width; j++ )
183 if( prow[j] != backgroundMin )
184 prow[j] = saturate_cast<uchar>(prow[j] + nrow[j] - delta);
185 }
186 }
187 }
188}

190void PatchGenerator::warpWholeImage(const Mat& image, Mat& matT, Mat& buf,
191 Mat& warped, int border, RNG& rng) const
192{
193 Mat_<double> T = matT;
194 Rect roi(INT_MAX, INT_MAX, INT_MIN, INT_MIN);
195
196 for( int k = 0; k < 4; k++ )
197 {
198 Point2f pt0, pt1;
199 pt0.x = (float)(k == 0 || k == 3 ? 0 : image.cols);
200 pt0.y = (float)(k < 2 ? 0 : image.rows);
201 pt1.x = (float)(T(0,0)*pt0.x + T(0,1)*pt0.y + T(0,2));
202 pt1.y = (float)(T(1,0)*pt0.x + T(1,1)*pt0.y + T(1,2));
203
204 roi.x = std::min(roi.x, cvFloor(pt1.x));
205 roi.y = std::min(roi.y, cvFloor(pt1.y));
206 roi.width = std::max(roi.width, cvCeil(pt1.x));
207 roi.height = std::max(roi.height, cvCeil(pt1.y));
208 }
209
210 roi.width -= roi.x - 1;
211 roi.height -= roi.y - 1;
212 int dx = border - roi.x;
213 int dy = border - roi.y;
214
215 if( (roi.width+border*2)*(roi.height+border*2) > buf.cols )
216 buf.create(1, (roi.width+border*2)*(roi.height+border*2), image.type());
217
218 warped = Mat(roi.height + border*2, roi.width + border*2,
219 image.type(), buf.data);
220
221 T(0,2) += dx;
222 T(1,2) += dy;
223 (*this)(image, T, warped, warped.size(), rng);
224
225 if( T.data != matT.data )
226 T.convertTo(matT, matT.type());
227}

256static void getDiscreteCircle(int R, vector<Point>& circle, vector<int>& filledHCircle)
257{
258 int x = R, y = 0;
259 for( ;;y++ )
260 {
261 x = cvRound(std::sqrt((double)R*R - y*y));
262 if( x < y )
263 break;
264 circle.push_back(Point(x,y));
265 if( x == y )
266 break;
267 }
268
269 int i, n8 = (int)circle.size() - (x == y), n8_ = n8 - (x != y), n4 = n8 + n8_, n = n4*4;
270 CV_Assert(n8 > 0);
271 circle.resize(n);
272
273 for( i = 0; i < n8; i++ )
274 {
275 Point p = circle[i];
276 circle[i+n4] = Point(-p.y, p.x);
277 circle[i+n4*2] = Point(-p.x, -p.y);
278 circle[i+n4*3] = Point(p.y, -p.x);
279 }
280
281 for( i = n8; i < n4; i++ )
282 {
283 Point p = circle[n4 - i], q = Point(p.y, p.x);
284 circle[i] = q;
285 circle[i+n4] = Point(-q.y, q.x);
286 circle[i+n4*2] = Point(-q.x, -q.y);
287 circle[i+n4*3] = Point(q.y, -q.x);
288 }
289
290 // the filled upper half of the circle is encoded as sequence of integers,
291 // i-th element is the coordinate of right-most circle point in each horizontal line y=i.
292 // the left-most point will be -filledHCircle[i].
293 for( i = 0, y = -1; i < n4; i++ )
294 {
295 Point p = circle[i];
296 if( p.y != y )
297 {
298 filledHCircle.push_back(p.x);
299 y = p.y;
300 if( y == R )
301 break;
302 }
303 }
304}

313void LDetector::getMostStable2D(const Mat& image, vector<KeyPoint>& keypoints,
314 int maxPoints, const PatchGenerator& _patchGenerator) const
315{
316 PatchGenerator patchGenerator = _patchGenerator;
317 patchGenerator.backgroundMin = patchGenerator.backgroundMax = 128;
318
319 Mat warpbuf, warped;
320 Mat matM(2, 3, CV_64F), _iM(2, 3, CV_64F);
321 double *M = (double*)matM.data, *iM = (double*)_iM.data;
322 RNG& rng = theRNG();
323 int i, k;
324 vector<KeyPoint> tempKeypoints;
325 double d2 = clusteringDistance*clusteringDistance;
326 keypoints.clear();
327
328 // TODO: this loop can be run in parallel, for that we need
329 // a separate accumulator keypoint lists for different threads.
330 for( i = 0; i < nViews; i++ )
331 {
332 // 1. generate random transform
333 // 2. map the source image corners and compute the ROI in canvas
334 // 3. select the ROI in canvas, adjust the transformation matrix
335 // 4. apply the transformation
336 // 5. run keypoint detector in pyramids
337 // 6. map each point back and update the lists of most stable points
338
339 if(verbose && (i+1)*progressBarSize/nViews != i*progressBarSize/nViews)
340 putchar('.');
341
342 if( i > 0 )
343 patchGenerator.generateRandomTransform(Point2f(), Point2f(), matM, rng);
344 else
345 {
346 // identity transformation
347 M[0] = M[4] = 1;
348 M[1] = M[3] = M[2] = M[5] = 0;
349 }
350
351 patchGenerator.warpWholeImage(image, matM, warpbuf, warped, cvCeil(baseFeatureSize*0.5+radius), rng);
352 (*this)(warped, tempKeypoints, maxPoints*3);
353 invertAffineTransform(matM, _iM);
354
355 int j, sz0 = (int)tempKeypoints.size(), sz1;
356 for( j = 0; j < sz0; j++ )
357 {
358 KeyPoint kpt1 = tempKeypoints[j];
359 KeyPoint kpt0((float)(iM[0]*kpt1.pt.x + iM[1]*kpt1.pt.y + iM[2]),
360 (float)(iM[3]*kpt1.pt.x + iM[4]*kpt1.pt.y + iM[5]),
361 kpt1.size, -1.f, 1.f, kpt1.octave);
362 float r = kpt1.size*0.5f;
363 if( kpt0.pt.x < r || kpt0.pt.x >= image.cols - r ||
364 kpt0.pt.y < r || kpt0.pt.y >= image.rows - r )
365 continue;
366
367 sz1 = (int)keypoints.size();
368 for( k = 0; k < sz1; k++ )
369 {
370 KeyPoint kpt = keypoints[k];
371 if( kpt.octave != kpt0.octave )
372 continue;
373 double dx = kpt.pt.x - kpt0.pt.x, dy = kpt.pt.y - kpt0.pt.y;
374 if( dx*dx + dy*dy <= d2*(1 << kpt.octave*2) )
375 {
376 keypoints[k] = KeyPoint((kpt.pt.x*kpt.response + kpt0.pt.x)/(kpt.response+1),
377 (kpt.pt.y*kpt.response + kpt0.pt.y)/(kpt.response+1),
378 kpt.size, -1.f, kpt.response + 1, kpt.octave);
379 break;
380 }
381 }
382 if( k == sz1 )
383 keypoints.push_back(kpt0);
384 }
385 }
386
387 if( verbose )
388 putchar('\n');
389
390 if( (int)keypoints.size() > maxPoints )
391 {
392 sort(keypoints, CmpKeypointScores());
393 keypoints.resize(maxPoints);
394 }
395}

398static inline int computeLResponse(const uchar* ptr, const int* cdata, int csize)
399{
400 int i, csize2 = csize/2, sum = -ptr[0]*csize;
401 for( i = 0; i < csize2; i++ )
402 {
403 int ofs = cdata[i];
404 sum += ptr[ofs] + ptr[-ofs];
405 }
406 return sum;
407}

410static Point2f adjustCorner(const float* fval, float& fvaln)
411{
412 double bx = (fval[3] - fval[5])*0.5;
413 double by = (fval[2] - fval[7])*0.5;
414 double Axx = fval[3] - fval[4]*2 + fval[5];
415 double Axy = (fval[0] - fval[2] - fval[6] + fval[8])*0.25;
416 double Ayy = fval[1] - fval[4]*2 + fval[7];
417 double D = Axx*Ayy - Axy*Axy;
418 D = D != 0 ? 1./D : 0;
419 double dx = (bx*Ayy - by*Axy)*D;
420 double dy = (by*Axx - bx*Axy)*D;
421 dx = std::min(std::max(dx, -1.), 1.);
422 dy = std::min(std::max(dy, -1.), 1.);
423 fvaln = (float)(fval[4] + (bx*dx + by*dy)*0.5);
424 if(fvaln*fval[4] < 0 || std::abs(fvaln) < std::abs(fval[4]))
425 fvaln = fval[4];
426
427 return Point2f((float)dx, (float)dy);
428}

437void LDetector::operator()(const vector<Mat>& pyr, vector<KeyPoint>& keypoints, int maxCount, bool scaleCoords) const
438{
439 const int lthreshold = 3;
440 int L, x, y, i, j, k, tau = lthreshold;
441 Mat scoreBuf(pyr[0].size(), CV_16S), maskBuf(pyr[0].size(), CV_8U);
442 int scoreElSize = (int)scoreBuf.elemSize();
443 vector<Point> circle0;
444 vector<int> fhcircle0, circle, fcircle_s, fcircle;
445 getDiscreteCircle(radius, circle0, fhcircle0);
446 CV_Assert(fhcircle0.size() == (size_t)(radius+1) && circle0.size() % 2 == 0);
447 keypoints.clear();
448
449 for( L = 0; L < nOctaves; L++ )
450 {
451 // Pyramidal keypoint detector body:
452 // 1. build next pyramid layer
453 // 2. scan points, check the circular neighborhood, compute the score
454 // 3. do non-maxima suppression
455 // 4. adjust the corners (sub-pix)
456 double cscale = scaleCoords ? 1 << L : 1;
457 Size layerSize = pyr[L].size();
458 if( layerSize.width < radius*2 + 3 || layerSize.height < radius*2 + 3 )
459 break;
460 Mat scoreLayer(layerSize, scoreBuf.type(), scoreBuf.data);
461 Mat maskLayer(layerSize, maskBuf.type(), maskBuf.data);
462 const Mat& pyrLayer = pyr[L];
463 int sstep = (int)(scoreLayer.step/sizeof(short));
464 int mstep = (int)maskLayer.step;
465
466 int csize = (int)circle0.size(), csize2 = csize/2;
467 circle.resize(csize*3);
468 for( i = 0; i < csize; i++ )
469 circle[i] = circle[i+csize] = circle[i+csize*2] = (int)((-circle0[i].y)*pyrLayer.step + circle0[i].x);
470 fcircle.clear();
471 fcircle_s.clear();
472 for( i = -radius; i <= radius; i++ )
473 {
474 x = fhcircle0[std::abs(i)];
475 for( j = -x; j <= x; j++ )
476 {
477 fcircle_s.push_back(i*sstep + j);
478 fcircle.push_back((int)(i*pyrLayer.step + j));
479 }
480 }
481 int nsize = (int)fcircle.size();
482 const int* cdata = &circle[0];
483 const int* ndata = &fcircle[0];
484 const int* ndata_s = &fcircle_s[0];
485
486 for( y = 0; y < radius; y++ )
487 {
488 memset( scoreLayer.ptr<short>(y), 0, layerSize.width*scoreElSize );
489 memset( scoreLayer.ptr<short>(layerSize.height-y-1), 0, layerSize.width*scoreElSize );
490 memset( maskLayer.ptr<uchar>(y), 0, layerSize.width );
491 memset( maskLayer.ptr<uchar>(layerSize.height-y-1), 0, layerSize.width );
492 }
493
494 int vradius = (int)(radius*pyrLayer.step);
495
496 for( y = radius; y < layerSize.height - radius; y++ )
497 {
498 const uchar* img = pyrLayer.ptr<uchar>(y) + radius;
499 short* scores = scoreLayer.ptr<short>(y);
500 uchar* mask = maskLayer.ptr<uchar>(y);
501
502 for( x = 0; x < radius; x++ )
503 {
504 scores[x] = scores[layerSize.width - 1 - x] = 0;
505 mask[x] = mask[layerSize.width - 1 - x] = 0;
506 }
507
508 for( x = radius; x < layerSize.width - radius; x++, img++ )
509 {
510 int val0 = *img;
511 if( (std::abs(val0 - img[radius]) < tau && std::abs(val0 - img[-radius]) < tau) ||
512 (std::abs(val0 - img[vradius]) < tau && std::abs(val0 - img[-vradius]) < tau))
513 {
514 scores[x] = 0;
515 mask[x] = 0;
516 continue;
517 }
518
519 for( k = 0; k < csize; k++ )
520 {
521 if( std::abs(val0 - img[cdata[k]]) < tau &&
522 (std::abs(val0 - img[cdata[k + csize2]]) < tau ||
523 std::abs(val0 - img[cdata[k + csize2 - 1]]) < tau ||
524 std::abs(val0 - img[cdata[k + csize2 + 1]]) < tau ||
525 std::abs(val0 - img[cdata[k + csize2 - 2]]) < tau ||
526 std::abs(val0 - img[cdata[k + csize2 + 2]]) < tau/* ||
527 std::abs(val0 - img[cdata[k + csize2 - 3]]) < tau ||
528 std::abs(val0 - img[cdata[k + csize2 + 3]]) < tau*/) )
529 break;
530 }
531
532 if( k < csize )
533 {
534 scores[x] = 0;
535 mask[x] = 0;
536 }
537 else
538 {
539 scores[x] = (short)computeLResponse(img, cdata, csize);
540 mask[x] = 1;
541 }
542 }
543 }
544
545 for( y = radius+1; y < layerSize.height - radius-1; y++ )
546 {
547 const uchar* img = pyrLayer.ptr<uchar>(y) + radius+1;
548 short* scores = scoreLayer.ptr<short>(y) + radius+1;
549 const uchar* mask = maskLayer.ptr<uchar>(y) + radius+1;
550
551 for( x = radius+1; x < layerSize.width - radius-1; x++, img++, scores++, mask++ )
552 {
553 int val0 = *scores;
554 if( !*mask || std::abs(val0) < lthreshold ||
555 (mask[-1] + mask[1] + mask[-mstep-1] + mask[-mstep] + mask[-mstep+1]+
556 mask[mstep-1] + mask[mstep] + mask[mstep+1] < 3))
557 continue;
558 bool recomputeZeroScores = radius*2 < y && y < layerSize.height - radius*2 &&
559 radius*2 < x && x < layerSize.width - radius*2;
560
561 if( val0 > 0 )
562 {
563 for( k = 0; k < nsize; k++ )
564 {
565 int val = scores[ndata_s[k]];
566 if( val == 0 && recomputeZeroScores )
567 scores[ndata_s[k]] = (short)(val =
568 computeLResponse(img + ndata[k], cdata, csize));
569 if( val0 < val )
570 break;
571 }
572 }
573 else
574 {
575 for( k = 0; k < nsize; k++ )
576 {
577 int val = scores[ndata_s[k]];
578 if( val == 0 && recomputeZeroScores )
579 scores[ndata_s[k]] = (short)(val =
580 computeLResponse(img + ndata[k], cdata, csize));
581 if( val0 > val )
582 break;
583 }
584 }
585 if( k < nsize )
586 continue;
587 float fval[9], fvaln = 0;
588 for( int i1 = -1; i1 <= 1; i1++ )
589 for( int j1 = -1; j1 <= 1; j1++ )
590 {
591 fval[(i1+1)*3 + j1 + 1] = (float)(scores[sstep*i1+j1] ? scores[sstep*i1+j1] :
592 computeLResponse(img + pyrLayer.step*i1 + j1, cdata, csize));
593 }
594 Point2f pt = adjustCorner(fval, fvaln);
595 pt.x += x;
596 pt.y += y;
597 keypoints.push_back(KeyPoint((float)(pt.x*cscale), (float)(pt.y*cscale),
598 (float)(baseFeatureSize*cscale), -1, fvaln, L));
599 }
600 }
601 }

795void FernClassifier::prepare(int _nclasses, int _patchSize, int _signatureSize,
796 int _nstructs, int _structSize,
797 int _nviews, int _compressionMethod)
798{
799 clear();
800
801 CV_Assert( _nclasses > 1 && _patchSize >= 5 && _nstructs > 0 &&
802 _nviews > 0 && _structSize > 0 &&
803 (_compressionMethod == COMPRESSION_NONE ||
804 _compressionMethod == COMPRESSION_RANDOM_PROJ ||
805 _compressionMethod == COMPRESSION_PCA) );
806
807 nclasses = _nclasses;
808 patchSize = Size(_patchSize, _patchSize);
809 nstructs = _nstructs;
810 structSize = _structSize;
811 signatureSize = _compressionMethod == COMPRESSION_NONE ? nclasses : std::min(_signatureSize, nclasses);
812 compressionMethod = signatureSize == nclasses ? COMPRESSION_NONE : _compressionMethod;
813
814 leavesPerStruct = 1 << structSize;
815
816 int i, nfeatures = structSize*nstructs;
817
818 features = vector<Feature>( nfeatures );
819 posteriors = vector<float>( leavesPerStruct*nstructs*nclasses, 1.f );
820 classCounters = vector<int>( nclasses, leavesPerStruct );
821
822 CV_Assert( patchSize.width <= 256 && patchSize.height <= 256 );
823 RNG& rng = theRNG();
824
825 for( i = 0; i < nfeatures; i++ )
826 {
827 int x1 = (unsigned)rng % patchSize.width;
828 int y1 = (unsigned)rng % patchSize.height;
829 int x2 = (unsigned)rng % patchSize.width;
830 int y2 = (unsigned)rng % patchSize.height;
831 features[i] = Feature(x1, y1, x2, y2);
832 }
833}

843void FernClassifier::train(const vector<vector<Point2f> >& points,
844 const vector<Mat>& refimgs,
845 const vector<vector<int> >& labels,
846 int _nclasses, int _patchSize,
847 int _signatureSize, int _nstructs,
848 int _structSize, int _nviews, int _compressionMethod,
849 const PatchGenerator& patchGenerator)
850{
851 CV_Assert( points.size() == refimgs.size() );
852 int numPoints = calcNumPoints( points );
853 _nclasses = (!labels.empty() && _nclasses>0) ? _nclasses : numPoints;
854 CV_Assert( labels.empty() || labels.size() == points.size() );
855
856
857 prepare(_nclasses, _patchSize, _signatureSize, _nstructs,
858 _structSize, _nviews, _compressionMethod);
859
860 // pass all the views of all the samples through the generated trees and accumulate
861 // the statistics (posterior probabilities) in leaves.
862 Mat patch;
863 RNG& rng = theRNG();
864
865 int globalPointIdx = 0;
866 for( size_t imgIdx = 0; imgIdx < points.size(); imgIdx++ )
867 {
868 const Point2f* imgPoints = &points[imgIdx][0];
869 const int* imgLabels = labels.empty() ? 0 : &labels[imgIdx][0];
870 for( size_t pointIdx = 0; pointIdx < points[imgIdx].size(); pointIdx++, globalPointIdx++ )
871 {
872 Point2f pt = imgPoints[pointIdx];
873 const Mat& src = refimgs[imgIdx];
874 int classId = imgLabels==0 ? globalPointIdx : imgLabels[pointIdx];
875 if( verbose && (globalPointIdx+1)*progressBarSize/numPoints != globalPointIdx*progressBarSize/numPoints )
876 putchar('.');
877 CV_Assert( 0 <= classId && classId < nclasses );
878 classCounters[classId] += _nviews;
879 for( int v = 0; v < _nviews; v++ )
880 {
881 patchGenerator(src, pt, patch, patchSize, rng);
882 for( int f = 0; f < nstructs; f++ )
883 posteriors[getLeaf(f, patch)*nclasses + classId]++;
884 }
885 }
886 }
887 if( verbose )
888 putchar('\n');
889
890 finalize(rng);
891}

894void FernClassifier::trainFromSingleView(const Mat& image,
895 const vector<KeyPoint>& keypoints,
896 int _patchSize, int _signatureSize,
897 int _nstructs, int _structSize,
898 int _nviews, int _compressionMethod,
899 const PatchGenerator& patchGenerator)
900{
901 prepare((int)keypoints.size(), _patchSize, _signatureSize, _nstructs,
902 _structSize, _nviews, _compressionMethod);
903 int i, j, k, nsamples = (int)keypoints.size(), maxOctave = 0;
904 for( i = 0; i < nsamples; i++ )
905 {
906 classCounters[i] = _nviews;
907 maxOctave = std::max(maxOctave, keypoints[i].octave);
908 }
909
910 double maxScale = patchGenerator.lambdaMax*2;
911 Mat canvas(cvRound(std::max(image.cols,image.rows)*maxScale + patchSize.width*2 + 10),
912 cvRound(std::max(image.cols,image.rows)*maxScale + patchSize.width*2 + 10), image.type());
913 Mat noisebuf;
914 vector<Mat> pyrbuf(maxOctave+1), pyr(maxOctave+1);
915 Point2f center0((image.cols-1)*0.5f, (image.rows-1)*0.5f),
916 center1((canvas.cols - 1)*0.5f, (canvas.rows - 1)*0.5f);
917 Mat matM(2, 3, CV_64F);
918 double *M = (double*)matM.data;
919 RNG& rng = theRNG();
920
921 Mat patch(patchSize, CV_8U);
922
923 for( i = 0; i < _nviews; i++ )
924 {
925 patchGenerator.generateRandomTransform(center0, center1, matM, rng);
926
927 CV_Assert(matM.type() == CV_64F);
928 Rect roi(INT_MAX, INT_MAX, INT_MIN, INT_MIN);
929
930 for( k = 0; k < 4; k++ )
931 {
932 Point2f pt0, pt1;
933 pt0.x = (float)(k == 0 || k == 3 ? 0 : image.cols);
934 pt0.y = (float)(k < 2 ? 0 : image.rows);
935 pt1.x = (float)(M[0]*pt0.x + M[1]*pt0.y + M[2]);
936 pt1.y = (float)(M[3]*pt0.x + M[4]*pt0.y + M[5]);
937
938 roi.x = std::min(roi.x, cvFloor(pt1.x));
939 roi.y = std::min(roi.y, cvFloor(pt1.y));
940 roi.width = std::max(roi.width, cvCeil(pt1.x));
941 roi.height = std::max(roi.height, cvCeil(pt1.y));
942 }
943
944 roi.width -= roi.x + 1;
945 roi.height -= roi.y + 1;
946
947 Mat canvas_roi(canvas, roi);
948 M[2] -= roi.x;
949 M[5] -= roi.y;
950
951 Size size = canvas_roi.size();
952 rng.fill(canvas_roi, RNG::UNIFORM, Scalar::all(0), Scalar::all(256));
953 warpAffine( image, canvas_roi, matM, size, INTER_LINEAR, BORDER_TRANSPARENT);
954
955 pyr[0] = canvas_roi;
956 for( j = 1; j <= maxOctave; j++ )
957 {
958 size = Size((size.width+1)/2, (size.height+1)/2);
959 if( pyrbuf[j].cols < size.width*size.height )
960 pyrbuf[j].create(1, size.width*size.height, image.type());
961 pyr[j] = Mat(size, image.type(), pyrbuf[j].data);
962 pyrDown(pyr[j-1], pyr[j]);
963 }
964
965 if( patchGenerator.noiseRange > 0 )
966 {
967 const int noiseDelta = 128;
968 if( noisebuf.cols < pyr[0].cols*pyr[0].rows )
969 noisebuf.create(1, pyr[0].cols*pyr[0].rows, image.type());
970 for( j = 0; j <= maxOctave; j++ )
971 {
972 Mat noise(pyr[j].size(), image.type(), noisebuf.data);
973 rng.fill(noise, RNG::UNIFORM, Scalar::all(-patchGenerator.noiseRange + noiseDelta),
974 Scalar::all(patchGenerator.noiseRange + noiseDelta));
975 addWeighted(pyr[j], 1, noise, 1, -noiseDelta, pyr[j]);
976 }
977 }
978
979 for( j = 0; j < nsamples; j++ )
980 {
981 KeyPoint kpt = keypoints[j];
982 float scale = 1.f/(1 << kpt.octave);
983 Point2f pt((float)((M[0]*kpt.pt.x + M[1]*kpt.pt.y + M[2])*scale),
984 (float)((M[3]*kpt.pt.x + M[4]*kpt.pt.y + M[5])*scale));
985 getRectSubPix(pyr[kpt.octave], patchSize, pt, patch, patch.type());
986 for( int f = 0; f < nstructs; f++ )
987 posteriors[getLeaf(f, patch)*nclasses + j]++;
988 }
989
990 if( verbose && (i+1)*progressBarSize/_nviews != i*progressBarSize/_nviews )
991 putchar('.');
992 }
993 if( verbose )
994 putchar('\n');
995
996 finalize(rng);
997}

1008int FernClassifier::operator()(const Mat& patch, vector<float>& signature) const
1009{
1010 if( posteriors.empty() )
1011 CV_Error(CV_StsNullPtr,
1012 "The descriptor has not been trained or "
1013 "the floating-point posteriors have been deleted");
1014 CV_Assert(patch.size() == patchSize);
1015
1016 int i, j, sz = signatureSize;
1017 signature.resize(sz);
1018 float* s = &signature[0];
1019
1020 for( j = 0; j < sz; j++ )
1021 s[j] = 0;
1022
1023 for( i = 0; i < nstructs; i++ )
1024 {
1025 int lf = getLeaf(i, patch);
1026 const float* ldata = &posteriors[lf*signatureSize];
1027 for( j = 0; j <= sz - 4; j += 4 )
1028 {
1029 float t0 = s[j] + ldata[j];
1030 float t1 = s[j+1] + ldata[j+1];
1031 s[j] = t0; s[j+1] = t1;
1032 t0 = s[j+2] + ldata[j+2];
1033 t1 = s[j+3] + ldata[j+3];
1034 s[j+2] = t0; s[j+3] = t1;
1035 }
1036 for( ; j < sz; j++ )
1037 s[j] += ldata[j];
1038 }
1039
1040 j = 0;
1041 if( signatureSize == nclasses && compressionMethod == COMPRESSION_NONE )
1042 {
1043 for( i = 1; i < nclasses; i++ )
1044 if( s[j] < s[i] )
1045 j = i;
1046 }
1047 return j;
1048}

1051void FernClassifier::finalize(RNG&)
1052{
1053 int i, j, k, n = nclasses;
1054 vector<double> invClassCounters(n);
1055 Mat_<double> _temp(1, n);
1056 double* temp = &_temp(0,0);
1057
1058 for( i = 0; i < n; i++ )
1059 invClassCounters[i] = 1./classCounters[i];
1060
1061 for( i = 0; i < nstructs; i++ )
1062 {
1063 for( j = 0; j < leavesPerStruct; j++ )
1064 {
1065 float* P = &posteriors[(i*leavesPerStruct + j)*nclasses];
1066 double sum = 0;
1067 for( k = 0; k < n; k++ )
1068 sum += P[k]*invClassCounters[k];
1069 sum = 1./sum;
1070 for( k = 0; k < n; k++ )
1071 temp[k] = P[k]*invClassCounters[k]*sum;
1072 log(_temp, _temp);
1073 for( k = 0; k < n; k++ )
1074 P[k] = (float)temp[k];
1075 }
1076 }
1077
1078#if 0
1079 // do the first pass over the data.
1080 if( compressionMethod == COMPRESSION_RANDOM_PROJ )
1081 {
1082 // In case of random projection
1083 // we generate a random m x n matrix consisting of -1's and 1's
1084 // (called Bernoulli matrix) and multiply it by each vector
1085 // of posterior probabilities.
1086 // the product is stored back into the same input vector.
1087
1088 Mat_<uchar> csmatrix;
1089 if( m < n )
1090 {
1091 // generate random Bernoulli matrix:
1092 // -1's are replaced with 0's and 1's stay 1's.
1093 csmatrix.create(m, n);
1094 rng.fill(csmatrix, RNG::UNIFORM, Scalar::all(0), Scalar::all(2));
1095 }
1096 vector<float> dst(m);
1097
1098 for( i = 0; i < totalLeaves; i++ )
1099 {
1100 int S = sampleCounters[i];
1101 if( S == 0 )
1102 continue;
1103
1104 float scale = 1.f/(S*(m < n ? std::sqrt((float)m) : 1.f));
1105 const int* leaf = (const int*)&posteriors[i*n];
1106 float* out_leaf = (float*)&posteriors[i*m];
1107
1108 for( j = 0; j < m; j++ )
1109 {
1110 float val = 0;
1111 if( m < n )
1112 {
1113 const uchar* csrow = csmatrix.ptr(j);
1114 // Because -1's in the Bernoulli matrix are encoded as 0's,
1115 // the real dot product value will be
1116 // A - B, where A is the sum of leaf[j]'s for which csrow[j]==1 and
1117 // B is the sum of leaf[j]'s for which csrow[j]==0.
1118 // since A + B = S, then A - B = A - (S - A) = 2*A - S.
1119 int A = 0;
1120 for( k = 0; k < n; k++ )
1121 A += leaf[k] & -(int)csrow[k];
1122 val = (A*2 - S)*scale;
1123 }
1124 else
1125 val = leaf[j]*scale;
1126 dst[j] = val;
1127 }
1128
1129 // put the vector back (since it's shorter than the original, we can do it in-place)
1130 for( j = 0; j < m; j++ )
1131 out_leaf[j] = dst[j];
1132 }
1133 }
1134 else if( compressionMethod == COMPRESSION_PCA )
1135 {
1136 // In case of PCA we do 3 passes over the data:
1137 // first, we compute the mean vector
1138 // second, we compute the covariation matrix
1139 // then we do eigen decomposition of the matrix and construct the PCA
1140 // projection matrix
1141 // and on the third pass we actually do PCA compression.
1142
1143 int nonEmptyLeaves = 0;
1144 Mat_<double> _mean(1, n), _vec(1, n), _dvec(m, 1),
1145 _cov(n, n), _evals(n, 1), _evects(n, n);
1146 _mean = 0.;
1147 double* mean = &_mean(0,0);
1148 double* vec = &_vec(0,0);
1149 double* dvec = &_dvec(0,0);
1150
1151 for( i = 0; i < totalLeaves; i++ )
1152 {
1153 int S = sampleCounters[i];
1154 if( S == 0 )
1155 continue;
1156 float invS = 1.f/S;
1157 const int* leaf = (const int*)&posteriors[0] + i*n;
1158 float* out_leaf = (float*)&posteriors[0] + i*n;
1159
1160 for( j = 0; j < n; j++ )
1161 {
1162 float t = leaf[j]*invS;
1163 out_leaf[j] = t;
1164 mean[j] += t;
1165 }
1166 nonEmptyLeaves++;
1167 }
1168
1169 CV_Assert( nonEmptyLeaves >= ntrees );
1170 _mean *= 1./nonEmptyLeaves;
1171
1172 for( i = 0; i < totalLeaves; i++ )
1173 {
1174 int S = sampleCounters[i];
1175 if( S == 0 )
1176 continue;
1177 const float* leaf = (const float*)&posteriors[0] + i*n;
1178 for( j = 0; j < n; j++ )
1179 vec[j] = leaf[j] - mean[j];
1180 gemm(_vec, _vec, 1, _cov, 1, _cov, GEMM_1_T);
1181 }
1182
1183 _cov *= 1./nonEmptyLeaves;
1184 eigen(_cov, _evals, _evects);
1185 // use the first m eigenvectors (stored as rows of the eigenvector matrix)
1186 // as the projection matrix in PCA
1187 _evects = _evects(Range(0, m), Range::all());
1188
1189 for( i = 0; i < totalLeaves; i++ )
1190 {
1191 int S = sampleCounters[i];
1192 if( S == 0 )
1193 continue;
1194 const float* leaf = (const float*)&posteriors[0] + i*n;
1195 float* out_leaf = (float*)&posteriors[0] + i*m;
1196
1197 for( j = 0; j < n; j++ )
1198 vec[j] = leaf[j] - mean[j];
1199 gemm(_evects, _vec, 1, Mat(), 0, _dvec, GEMM_2_T);
1200
1201 for( j = 0; j < m; j++ )
1202 out_leaf[j] = (float)dvec[j];
1203 }
1204 }
1205 else
1206 CV_Error( CV_StsBadArg,
1207 "Unknown compression method; use COMPRESSION_RANDOM_PROJ or COMPRESSION_PCA" );
1208
1209 // and shrink the vector
1210 posteriors.resize(totalLeaves*m);
1211#endif

1506bool PlanarObjectDetector::operator()(const vector<Mat>& pyr, const vector<KeyPoint>& keypoints,
1507 Mat& matH, vector<Point2f>& corners, vector<int>* pairs) const
1508{
1509 int i, j, m = (int)modelPoints.size(), n = (int)keypoints.size();
1510 vector<int> bestMatches(m, -1);
1511 vector<float> maxLogProb(m, -FLT_MAX);
1512 vector<float> signature;
1513 vector<Point2f> fromPt, toPt;
1514
1515 for( i = 0; i < n; i++ )
1516 {
1517 KeyPoint kpt = keypoints[i];
1518 CV_Assert(0 <= kpt.octave && kpt.octave < (int)pyr.size());
1519 kpt.pt.x /= (float)(1 << kpt.octave);
1520 kpt.pt.y /= (float)(1 << kpt.octave);
1521 int k = fernClassifier(pyr[kpt.octave], kpt.pt, signature);
1522 if( k >= 0 && (bestMatches[k] < 0 || signature[k] > maxLogProb[k]) )
1523 {
1524 maxLogProb[k] = signature[k];
1525 bestMatches[k] = i;
1526 }
1527 }
1528
1529 if(pairs)
1530 pairs->resize(0);
1531
1532 for( i = 0; i < m; i++ )
1533 if( bestMatches[i] >= 0 )
1534 {
1535 fromPt.push_back(modelPoints[i].pt);
1536 toPt.push_back(keypoints[bestMatches[i]].pt);
1537 }
1538
1539 if( fromPt.size() < 4 )
1540 return false;
1541
1542 vector<uchar> mask;
1543 matH = findHomography(fromPt, toPt, RANSAC, 10, mask);
1544 if( matH.data )
1545 {
1546 const Mat_<double>& H = matH;
1547 corners.resize(4);
1548 for( i = 0; i < 4; i++ )
1549 {
1550 Point2f pt((float)(modelROI.x + (i == 0 || i == 3 ? 0 : modelROI.width)),
1551 (float)(modelROI.y + (i <= 1 ? 0 : modelROI.height)));
1552 double w = 1./(H(2,0)*pt.x + H(2,1)*pt.y + H(2,2));
1553 corners[i] = Point2f((float)((H(0,0)*pt.x + H(0,1)*pt.y + H(0,2))*w),
1554 (float)((H(1,0)*pt.x + H(1,1)*pt.y + H(1,2))*w));
1555 }
1556 }
1557
1558 if( pairs )
1559 {
1560 for( i = j = 0; i < m; i++ )
1561 if( bestMatches[i] >= 0 && mask[j++] )
1562 {
1563 pairs->push_back(i);
1564 pairs->push_back(bestMatches[i]);
1565 }
1566 }
1567
1568 return matH.data != 0;
1569}
