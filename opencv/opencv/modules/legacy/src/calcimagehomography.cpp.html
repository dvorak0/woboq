44/****************************************************************************************\
45
46 calculate image homography
47
48\****************************************************************************************/
49
50CV_IMPL void
51cvCalcImageHomography( float* line, CvPoint3D32f* _center,
52 float* _intrinsic, float* _homography )
53{
54 double norm_xy, norm_xz, xy_sina, xy_cosa, xz_sina, xz_cosa, nx1, plane_dist;
55 float _ry[3], _rz[3], _r_trans[9];
56 CvMat rx = cvMat( 1, 3, CV_32F, line );
57 CvMat ry = cvMat( 1, 3, CV_32F, _ry );
58 CvMat rz = cvMat( 1, 3, CV_32F, _rz );
59 CvMat r_trans = cvMat( 3, 3, CV_32F, _r_trans );
60 CvMat center = cvMat( 3, 1, CV_32F, _center );
61
62 float _sub[9];
63 CvMat sub = cvMat( 3, 3, CV_32F, _sub );
64 float _t_trans[3];
65 CvMat t_trans = cvMat( 3, 1, CV_32F, _t_trans );
66
67 CvMat intrinsic = cvMat( 3, 3, CV_32F, _intrinsic );
68 CvMat homography = cvMat( 3, 3, CV_32F, _homography );
69
70 if( !line || !_center || !_intrinsic || !_homography )
71 CV_Error( CV_StsNullPtr, "" );
72
73 norm_xy = cvSqrt( line[0] * line[0] + line[1] * line[1] );
74 xy_cosa = line[0] / norm_xy;
75 xy_sina = line[1] / norm_xy;
76
77 norm_xz = cvSqrt( line[0] * line[0] + line[2] * line[2] );
78 xz_cosa = line[0] / norm_xz;
79 xz_sina = line[2] / norm_xz;
80
81 nx1 = -xz_sina;
82
83 _rz[0] = (float)(xy_cosa * nx1);
84 _rz[1] = (float)(xy_sina * nx1);
85 _rz[2] = (float)xz_cosa;
86 cvScale( &rz, &rz, 1./cvNorm(&rz,0,CV_L2) );
87
88 /* new axe y */
89 cvCrossProduct( &rz, &rx, &ry );
90 cvScale( &ry, &ry, 1./cvNorm( &ry, 0, CV_L2 ) );
91
92 /* transpone rotation matrix */
93 memcpy( &_r_trans[0], line, 3*sizeof(float));
94 memcpy( &_r_trans[3], _ry, 3*sizeof(float));
95 memcpy( &_r_trans[6], _rz, 3*sizeof(float));
96
97 /* calculate center distanse from arm plane */
98 plane_dist = cvDotProduct( &center, &rz );
99
100 /* calculate (I - r_trans)*center */
101 cvSetIdentity( &sub );
102 cvSub( &sub, &r_trans, &sub );
103 cvMatMul( &sub, &center, &t_trans );
104
105 cvMatMul( &t_trans, &rz, &sub );
106 cvScaleAdd( &sub, cvRealScalar(1./plane_dist), &r_trans, &sub ); /* ? */
107
108 cvMatMul( &intrinsic, &sub, &r_trans );
109 cvInvert( &intrinsic, &sub, CV_SVD );
110 cvMatMul( &r_trans, &sub, &homography );
111}
