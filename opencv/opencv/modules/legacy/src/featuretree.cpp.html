45void cvReleaseFeatureTree(CvFeatureTree* tr)
46{
47 delete tr;
48}
49
50// desc is m x d set of candidate points.
51// results is m x k set of row indices of matching points.
52// dist is m x k distance to matching points.
53void cvFindFeatures(CvFeatureTree* tr, const CvMat* desc,
54 CvMat* results, CvMat* dist, int k, int emax)
55{
56 tr->FindFeatures(desc, k, emax, results, dist);
57}
58
59int cvFindFeaturesBoxed(CvFeatureTree* tr,
60 CvMat* bounds_min, CvMat* bounds_max,
61 CvMat* results)
62{
63 return tr->FindOrthoRange(bounds_min, bounds_max, results);
64}
