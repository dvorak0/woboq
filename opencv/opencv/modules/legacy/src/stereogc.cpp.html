88// truncTab[x+255] = MAX(x-255,0)
89static uchar icvTruncTab[512];

91static int icvCutoffSqrTab[256];

93static void icvInitStereoConstTabs()
94{
95 static volatile int initialized = 0;
96 if( !initialized )
97 {
98 int i;
99 for( i = 0; i < 512; i++ )
100 icvTruncTab[i] = (uchar)MIN(MAX(i-255,0),255);
101 for( i = 0; i < 256; i++ )
102 icvCutoffSqrTab[i] = MIN(i*i, CUTOFF);
103 initialized = 1;
104 }
105}

107static void icvInitStereoTabs( CvStereoGCState2* state2 )
108{
109 int i, K = state2->K;
110
111 for( i = 0; i <= CUTOFF; i++ )
112 state2->dataCostFuncTab[i] = MIN(i*DENOMINATOR - K, 0);
113
114 for( i = 0; i < CUTOFF*2 + 1; i++ )
115 state2->smoothnessR[i] = MIN(abs(i-CUTOFF), state2->interactionRadius);
116
117 for( i = 0; i < 512; i++ )
118 {
119 int diff = abs(i - 255);
120 state2->smoothnessGrayDiff[i] = diff < state2->Ithreshold ? state2->lambda1 : state2->lambda2;
121 }
122}

125static int icvGCResizeOrphansBuf( GCVtx**& orphans, int norphans )
126{
127 int i, newNOrphans = MAX(norphans*3/2, 256);
128 GCVtx** newOrphans = (GCVtx**)cvAlloc( newNOrphans*sizeof(orphans[0]) );
129 for( i = 0; i < norphans; i++ )
130 newOrphans[i] = orphans[i];
131 cvFree( &orphans );
132 orphans = newOrphans;
133 return newNOrphans;
134}

136static int64 icvGCMaxFlow( GCVtx* vtx, int nvtx, GCEdge* edges, GCVtx**& _orphans, int& _maxOrphans )
137{
138 const int TERMINAL = -1, ORPHAN = -2;
139 GCVtx stub, *nilNode = &stub, *first = nilNode, *last = nilNode;
140 int i, k;
141 int curr_ts = 0;
142 int64 flow = 0;
143 int norphans = 0, maxOrphans = _maxOrphans;
144 GCVtx** orphans = _orphans;
145 stub.next = nilNode;
146
147 // initialize the active queue and the graph vertices
148 for( i = 0; i < nvtx; i++ )
149 {
150 GCVtx* v = vtx + i;
151 v->ts = 0;
152 if( v->weight != 0 )
153 {
154 last = last->next = v;
155 v->dist = 1;
156 v->parent = TERMINAL;
157 v->t = v->weight < 0;
158 }
159 else
160 {
161 v->next = NULL;
162 v->parent = 0;
163 }
164 }
165
166 first = first->next;
167 last->next = nilNode;
168 nilNode->next = 0;
169
170 // run the search-path -> augment-graph -> restore-trees loop
171 for(;;)
172 {
173 GCVtx* v, *u;
174 int e0 = -1, ei = 0, ej = 0, min_weight, weight;
175 uchar vt;
176
177 // grow S & T search trees, find an edge connecting them
178 while( first != nilNode )
179 {
180 v = first;
181 if( v->parent )
182 {
183 vt = v->t;
184 for( ei = v->first; ei != 0; ei = edges[ei].next )
185 {
186 if( edges[ei^vt].weight == 0 )
187 continue;
188 u = edges[ei].dst;
189 if( !u->parent )
190 {
191 u->t = vt;
192 u->parent = ei ^ 1;
193 u->ts = v->ts;
194 u->dist = v->dist + 1;
195 if( !u->next )
196 {
197 u->next = nilNode;
198 last = last->next = u;
199 }
200 continue;
201 }
202
203 if( u->t != vt )
204 {
205 e0 = ei ^ vt;
206 break;
207 }
208
209 if( u->dist > v->dist+1 && u->ts <= v->ts )
210 {
211 // reassign the parent
212 u->parent = ei ^ 1;
213 u->ts = v->ts;
214 u->dist = v->dist + 1;
215 }
216 }
217 if( e0 > 0 )
218 break;
219 }
220 // exclude the vertex from the active list
221 first = first->next;
222 v->next = 0;
223 }
224
225 if( e0 <= 0 )
226 break;
227
228 // find the minimum edge weight along the path
229 min_weight = edges[e0].weight;
230 assert( min_weight > 0 );
231 // k = 1: source tree, k = 0: destination tree
232 for( k = 1; k >= 0; k-- )
233 {
234 for( v = edges[e0^k].dst;; v = edges[ei].dst )
235 {
236 if( (ei = v->parent) < 0 )
237 break;
238 weight = edges[ei^k].weight;
239 min_weight = MIN(min_weight, weight);
240 assert( min_weight > 0 );
241 }
242 weight = abs(v->weight);
243 min_weight = MIN(min_weight, weight);
244 assert( min_weight > 0 );
245 }
246
247 // modify weights of the edges along the path and collect orphans
248 edges[e0].weight -= min_weight;
249 edges[e0^1].weight += min_weight;
250 flow += min_weight;
251
252 // k = 1: source tree, k = 0: destination tree
253 for( k = 1; k >= 0; k-- )
254 {
255 for( v = edges[e0^k].dst;; v = edges[ei].dst )
256 {
257 if( (ei = v->parent) < 0 )
258 break;
259 edges[ei^(k^1)].weight += min_weight;
260 if( (edges[ei^k].weight -= min_weight) == 0 )
261 {
262 if( norphans >= maxOrphans )
263 maxOrphans = icvGCResizeOrphansBuf( orphans, norphans );
264 orphans[norphans++] = v;
265 v->parent = ORPHAN;
266 }
267 }
268
269 v->weight = (short)(v->weight + min_weight*(1-k*2));
270 if( v->weight == 0 )
271 {
272 if( norphans >= maxOrphans )
273 maxOrphans = icvGCResizeOrphansBuf( orphans, norphans );
274 orphans[norphans++] = v;
275 v->parent = ORPHAN;
276 }
277 }
278
279 // restore the search trees by finding new parents for the orphans
280 curr_ts++;
281 while( norphans > 0 )
282 {
283 GCVtx* v1 = orphans[--norphans];
284 int d, min_dist = INT_MAX;
285 e0 = 0;
286 vt = v1->t;
287
288 for( ei = v1->first; ei != 0; ei = edges[ei].next )
289 {
290 if( edges[ei^(vt^1)].weight == 0 )
291 continue;
292 u = edges[ei].dst;
293 if( u->t != vt || u->parent == 0 )
294 continue;
295 // compute the distance to the tree root
296 for( d = 0;; )
297 {
298 if( u->ts == curr_ts )
299 {
300 d += u->dist;
301 break;
302 }
303 ej = u->parent;
304 d++;
305 if( ej < 0 )
306 {
307 if( ej == ORPHAN )
308 d = INT_MAX-1;
309 else
310 {
311 u->ts = curr_ts;
312 u->dist = 1;
313 }
314 break;
315 }
316 u = edges[ej].dst;
317 }
318
319 // update the distance
320 if( ++d < INT_MAX )
321 {
322 if( d < min_dist )
323 {
324 min_dist = d;
325 e0 = ei;
326 }
327 for( u = edges[ei].dst; u->ts != curr_ts; u = edges[u->parent].dst )
328 {
329 u->ts = curr_ts;
330 u->dist = --d;
331 }
332 }
333 }
334
335 if( (v1->parent = e0) > 0 )
336 {
337 v1->ts = curr_ts;
338 v1->dist = min_dist;
339 continue;
340 }
341
342 /* no parent is found */
343 v1->ts = 0;
344 for( ei = v1->first; ei != 0; ei = edges[ei].next )
345 {
346 u = edges[ei].dst;
347 ej = u->parent;
348 if( u->t != vt || !ej )
349 continue;
350 if( edges[ei^(vt^1)].weight && !u->next )
351 {
352 u->next = nilNode;
353 last = last->next = u;
354 }
355 if( ej > 0 && edges[ej].dst == v1 )
356 {
357 if( norphans >= maxOrphans )
358 maxOrphans = icvGCResizeOrphansBuf( orphans, norphans );
359 orphans[norphans++] = u;
360 u->parent = ORPHAN;
361 }
362 }
363 }
364 }
365
366 _orphans = orphans;
367 _maxOrphans = maxOrphans;
368
369 return flow;
370}

373CvStereoGCState* cvCreateStereoGCState( int numberOfDisparities, int maxIters )
374{
375 CvStereoGCState* state = 0;
376
377 state = (CvStereoGCState*)cvAlloc( sizeof(*state) );
378 memset( state, 0, sizeof(*state) );
379 state->minDisparity = 0;
380 state->numberOfDisparities = numberOfDisparities;
381 state->maxIters = maxIters <= 0 ? 3 : maxIters;
382 state->Ithreshold = 5;
383 state->interactionRadius = 1;
384 state->K = state->lambda = state->lambda1 = state->lambda2 = -1.f;
385 state->occlusionCost = OCCLUSION_PENALTY;
386
387 return state;
388}

409// ||I(x) - J(x')|| =
410// min(CUTOFF,
411// min(
412// max(
413// max(minJ(x') - I(x), 0),
414// max(I(x) - maxJ(x'), 0)),
415// max(
416// max(minI(x) - J(x'), 0),
417// max(J(x') - maxI(x), 0)))**2) ==
418// min(CUTOFF,
419// min(
420// max(minJ(x') - I(x), 0) +
421// max(I(x) - maxJ(x'), 0),
422//
423// max(minI(x) - J(x'), 0) +
424// max(J(x') - maxI(x), 0)))**2)
425// where (I, minI, maxI) and
426// (J, minJ, maxJ) are stored as interleaved 3-channel images.
427// minI, maxI are computed from I,
428// minJ, maxJ are computed from J - see icvInitGraySubPix.
429static inline int icvDataCostFuncGraySubpix( const uchar* a, const uchar* b )
430{
431 int va = a[0], vb = b[0];
432 int da = icvTruncTab[b[1] - va + 255] + icvTruncTab[va - b[2] + 255];
433 int db = icvTruncTab[a[1] - vb + 255] + icvTruncTab[vb - a[2] + 255];
434 return icvCutoffSqrTab[MIN(da,db)];
435}

437static inline int icvSmoothnessCostFunc( int da, int db, int maxR, const int* stabR, int scale )
438{
439 return da == db ? 0 : (da == OCCLUDED || db == OCCLUDED ? maxR : stabR[da - db])*scale;
440}

442static void icvInitGraySubpix( const CvMat* left, const CvMat* right,
443 CvMat* left3, CvMat* right3 )
444{
445 int k, x, y, rows = left->rows, cols = left->cols;
446
447 for( k = 0; k < 2; k++ )
448 {
449 const CvMat* src = k == 0 ? left : right;
450 CvMat* dst = k == 0 ? left3 : right3;
451 int sstep = src->step;
452
453 for( y = 0; y < rows; y++ )
454 {
455 const uchar* sptr = src->data.ptr + sstep*y;
456 const uchar* sptr_prev = y > 0 ? sptr - sstep : sptr;
457 const uchar* sptr_next = y < rows-1 ? sptr + sstep : sptr;
458 uchar* dptr = dst->data.ptr + dst->step*y;
459 int v_prev = sptr[0];
460
461 for( x = 0; x < cols; x++, dptr += 3 )
462 {
463 int v = sptr[x], v1, minv = v, maxv = v;
464
465 v1 = (v + v_prev)/2;
466 minv = MIN(minv, v1); maxv = MAX(maxv, v1);
467 v1 = (v + sptr_prev[x])/2;
468 minv = MIN(minv, v1); maxv = MAX(maxv, v1);
469 v1 = (v + sptr_next[x])/2;
470 minv = MIN(minv, v1); maxv = MAX(maxv, v1);
471 if( x < cols-1 )
472 {
473 v1 = (v + sptr[x+1])/2;
474 minv = MIN(minv, v1); maxv = MAX(maxv, v1);
475 }
476 v_prev = v;
477 dptr[0] = (uchar)v;
478 dptr[1] = (uchar)minv;
479 dptr[2] = (uchar)maxv;
480 }
481 }
482 }
483}

485// Optimal K is computed as avg_x(k-th-smallest_d(||I(x)-J(x+d)||)),
486// where k = number_of_disparities*0.25.
487static float
488icvComputeK( CvStereoGCState* state )
489{
490 int x, y, x1, d, i, j, rows = state->left->rows, cols = state->left->cols, n = 0;
491 int mind = state->minDisparity, nd = state->numberOfDisparities, maxd = mind + nd;
492 int k = MIN(MAX((nd + 2)/4, 3), nd), delta, t, sum = 0;
493 std::vector<int> _arr(k+1);
494 int *arr = &_arr[0];
495
496 for( y = 0; y < rows; y++ )
497 {
498 const uchar* lptr = state->left->data.ptr + state->left->step*y;
499 const uchar* rptr = state->right->data.ptr + state->right->step*y;
500
501 for( x = 0; x < cols; x++ )
502 {
503 for( d = maxd-1, i = 0; d >= mind; d-- )
504 {
505 x1 = x - d;
506 if( (unsigned)x1 >= (unsigned)cols )
507 continue;
508 delta = icvDataCostFuncGraySubpix( lptr + x*3, rptr + x1*3 );
509 if( i < k )
510 arr[i++] = delta;
511 else
512 for( i = 0; i < k; i++ )
513 if( delta < arr[i] )
514 CV_SWAP( arr[i], delta, t );
515 }
516 delta = arr[0];
517 for( j = 1; j < i; j++ )
518 delta = MAX(delta, arr[j]);
519 sum += delta;
520 n++;
521 }
522 }
523
524 return (float)sum/n;
525}

527static int64 icvComputeEnergy( const CvStereoGCState* state, const CvStereoGCState2* state2,
528 bool allOccluded )
529{
530 int x, y, rows = state->left->rows, cols = state->left->cols;
531 int64 E = 0;
532 const int* dtab = state2->dataCostFuncTab;
533 int maxR = state2->interactionRadius;
534 const int* stabR = state2->smoothnessR + CUTOFF;
535 const int* stabI = state2->smoothnessGrayDiff + 255;
536 const uchar* left = state->left->data.ptr;
537 const uchar* right = state->right->data.ptr;
538 short* dleft = state->dispLeft->data.s;
539 short* dright = state->dispRight->data.s;
540 int step = state->left->step;
541 int dstep = (int)(state->dispLeft->step/sizeof(short));
542
543 assert( state->left->step == state->right->step &&
544 state->dispLeft->step == state->dispRight->step );
545
546 if( allOccluded )
547 return (int64)OCCLUSION_PENALTY*rows*cols*2;
548
549 for( y = 0; y < rows; y++, left += step, right += step, dleft += dstep, dright += dstep )
550 {
551 for( x = 0; x < cols; x++ )
552 {
553 int d = dleft[x], x1, d1;
554 if( d == OCCLUDED )
555 E += OCCLUSION_PENALTY;
556 else
557 {
558 x1 = x + d;
559 if( (unsigned)x1 >= (unsigned)cols )
560 continue;
561 d1 = dright[x1];
562 if( d == -d1 )
563 E += dtab[icvDataCostFuncGraySubpix( left + x*3, right + x1*3 )];
564 }
565
566 if( x < cols-1 )
567 {
568 d1 = dleft[x+1];
569 E += icvSmoothnessCostFunc(d, d1, maxR, stabR, stabI[left[x*3] - left[x*3+3]] );
570 }
571 if( y < rows-1 )
572 {
573 d1 = dleft[x+dstep];
574 E += icvSmoothnessCostFunc(d, d1, maxR, stabR, stabI[left[x*3] - left[x*3+step]] );
575 }
576
577 d = dright[x];
578 if( d == OCCLUDED )
579 E += OCCLUSION_PENALTY;
580
581 if( x < cols-1 )
582 {
583 d1 = dright[x+1];
584 E += icvSmoothnessCostFunc(d, d1, maxR, stabR, stabI[right[x*3] - right[x*3+3]] );
585 }
586 if( y < rows-1 )
587 {
588 d1 = dright[x+dstep];
589 E += icvSmoothnessCostFunc(d, d1, maxR, stabR, stabI[right[x*3] - right[x*3+step]] );
590 }
591 assert( E >= 0 );
592 }
593 }
594
595 return E;
596}

598static inline void icvAddEdge( GCVtx *x, GCVtx* y, GCEdge* edgeBuf, int nedges, int w, int rw )
599{
600 GCEdge *xy = edgeBuf + nedges, *yx = xy + 1;
601
602 assert( x != 0 && y != 0 );
603 xy->dst = y;
604 xy->next = x->first;
605 xy->weight = (short)w;
606 x->first = nedges;
607
608 yx->dst = x;
609 yx->next = y->first;
610 yx->weight = (short)rw;
611 y->first = nedges+1;
612}

614static inline int icvAddTWeights( GCVtx* vtx, int sourceWeight, int sinkWeight )
615{
616 int w = vtx->weight;
617 if( w > 0 )
618 sourceWeight += w;
619 else
620 sinkWeight -= w;
621 vtx->weight = (short)(sourceWeight - sinkWeight);
622 return MIN(sourceWeight, sinkWeight);
623}

625static inline int icvAddTerm( GCVtx* x, GCVtx* y, int A, int B, int C, int D,
626 GCEdge* edgeBuf, int& nedges )
627{
628 int dE = 0, w;
629
630 assert(B - A + C - D >= 0);
631 if( B < A )
632 {
633 dE += icvAddTWeights(x, D, B);
634 dE += icvAddTWeights(y, 0, A - B);
635 if( (w = B - A + C - D) != 0 )
636 {
637 icvAddEdge( x, y, edgeBuf, nedges, 0, w );
638 nedges += 2;
639 }
640 }
641 else if( C < D )
642 {
643 dE += icvAddTWeights(x, D, A + D - C);
644 dE += icvAddTWeights(y, 0, C - D);
645 if( (w = B - A + C - D) != 0 )
646 {
647 icvAddEdge( x, y, edgeBuf, nedges, w, 0 );
648 nedges += 2;
649 }
650 }
651 else
652 {
653 dE += icvAddTWeights(x, D, A);
654 if( B != A || C != D )
655 {
656 icvAddEdge( x, y, edgeBuf, nedges, B - A, C - D );
657 nedges += 2;
658 }
659 }
660 return dE;
661}

663static int64 icvAlphaExpand( int64 Eprev, int alpha, CvStereoGCState* state, CvStereoGCState2* state2 )
664{
665 GCVtx *var, *var1;
666 int64 E = 0;
667 int delta, E00=0, E0a=0, Ea0=0, Eaa=0;
668 int k, a, d, d1, x, y, x1, y1, rows = state->left->rows, cols = state->left->cols;
669 int nvtx = 0, nedges = 2;
670 GCVtx* vbuf = (GCVtx*)state->vtxBuf->data.ptr;
671 GCEdge* ebuf = (GCEdge*)state->edgeBuf->data.ptr;
672 int maxR = state2->interactionRadius;
673 const int* dtab = state2->dataCostFuncTab;
674 const int* stabR = state2->smoothnessR + CUTOFF;
675 const int* stabI = state2->smoothnessGrayDiff + 255;
676 const uchar* left0 = state->left->data.ptr;
677 const uchar* right0 = state->right->data.ptr;
678 short* dleft0 = state->dispLeft->data.s;
679 short* dright0 = state->dispRight->data.s;
680 GCVtx** pleft0 = (GCVtx**)state->ptrLeft->data.ptr;
681 GCVtx** pright0 = (GCVtx**)state->ptrRight->data.ptr;
682 int step = state->left->step;
683 int dstep = (int)(state->dispLeft->step/sizeof(short));
684 int pstep = (int)(state->ptrLeft->step/sizeof(GCVtx*));
685 int aa[] = { alpha, -alpha };
686
687 //double t = (double)cvGetTickCount();
688
689 assert( state->left->step == state->right->step &&
690 state->dispLeft->step == state->dispRight->step &&
691 state->ptrLeft->step == state->ptrRight->step );
692 for( k = 0; k < 2; k++ )
693 {
694 ebuf[k].dst = 0;
695 ebuf[k].next = 0;
696 ebuf[k].weight = 0;
697 }
698
699 for( y = 0; y < rows; y++ )
700 {
701 const uchar* left = left0 + step*y;
702 const uchar* right = right0 + step*y;
703 const short* dleft = dleft0 + dstep*y;
704 const short* dright = dright0 + dstep*y;
705 GCVtx** pleft = pleft0 + pstep*y;
706 GCVtx** pright = pright0 + pstep*y;
707 const uchar* lr[] = { left, right };
708 const short* dlr[] = { dleft, dright };
709 GCVtx** plr[] = { pleft, pright };
710
711 for( k = 0; k < 2; k++ )
712 {
713 a = aa[k];
714 for( y1 = y+(y>0); y1 <= y+(y<rows-1); y1++ )
715 {
716 const short* disp = (k == 0 ? dleft0 : dright0) + y1*dstep;
717 GCVtx** ptr = (k == 0 ? pleft0 : pright0) + y1*pstep;
718 for( x = 0; x < cols; x++ )
719 {
720 GCVtx* v = ptr[x] = &vbuf[nvtx++];
721 v->first = 0;
722 v->weight = disp[x] == (short)(OCCLUDED ? -OCCLUSION_PENALTY2 : 0);
723 }
724 }
725 }
726
727 for( x = 0; x < cols; x++ )
728 {
729 d = dleft[x];
730 x1 = x + d;
731 var = pleft[x];
732
733 // (left + x, right + x + d)
734 if( d != alpha && d != OCCLUDED && (unsigned)x1 < (unsigned)cols )
735 {
736 var1 = pright[x1];
737 d1 = dright[x1];
738 if( d == -d1 )
739 {
740 assert( var1 != 0 );
741 delta = IS_BLOCKED(alpha, d) ? INFINITY : 0;
742 //add inter edge
743 E += icvAddTerm( var, var1,
744 dtab[icvDataCostFuncGraySubpix( left + x*3, right + x1*3 )],
745 delta, delta, 0, ebuf, nedges );
746 }
747 else if( IS_BLOCKED(alpha, d) )
748 E += icvAddTerm( var, var1, 0, INFINITY, 0, 0, ebuf, nedges );
749 }
750
751 // (left + x, right + x + alpha)
752 x1 = x + alpha;
753 if( (unsigned)x1 < (unsigned)cols )
754 {
755 var1 = pright[x1];
756 d1 = dright[x1];
757
758 E0a = IS_BLOCKED(d, alpha) ? INFINITY : 0;
759 Ea0 = IS_BLOCKED(-d1, alpha) ? INFINITY : 0;
760 Eaa = dtab[icvDataCostFuncGraySubpix( left + x*3, right + x1*3 )];
761 E += icvAddTerm( var, var1, 0, E0a, Ea0, Eaa, ebuf, nedges );
762 }
763
764 // smoothness
765 for( k = 0; k < 2; k++ )
766 {
767 GCVtx** p = plr[k];
768 const short* disp = dlr[k];
769 const uchar* img = lr[k] + x*3;
770 int scale;
771 var = p[x];
772 d = disp[x];
773 a = aa[k];
774
775 if( x < cols - 1 )
776 {
777 var1 = p[x+1];
778 d1 = disp[x+1];
779 scale = stabI[img[0] - img[3]];
780 E0a = icvSmoothnessCostFunc( d, a, maxR, stabR, scale );
781 Ea0 = icvSmoothnessCostFunc( a, d1, maxR, stabR, scale );
782 E00 = icvSmoothnessCostFunc( d, d1, maxR, stabR, scale );
783 E += icvAddTerm( var, var1, E00, E0a, Ea0, 0, ebuf, nedges );
784 }
785
786 if( y < rows - 1 )
787 {
788 var1 = p[x+pstep];
789 d1 = disp[x+dstep];
790 scale = stabI[img[0] - img[step]];
791 E0a = icvSmoothnessCostFunc( d, a, maxR, stabR, scale );
792 Ea0 = icvSmoothnessCostFunc( a, d1, maxR, stabR, scale );
793 E00 = icvSmoothnessCostFunc( d, d1, maxR, stabR, scale );
794 E += icvAddTerm( var, var1, E00, E0a, Ea0, 0, ebuf, nedges );
795 }
796 }
797
798 // visibility term
799 if( d != OCCLUDED && IS_BLOCKED(alpha, -d))
800 {
801 x1 = x + d;
802 if( (unsigned)x1 < (unsigned)cols )
803 {
804 if( d != -dleft[x1] )
805 {
806 var1 = pleft[x1];
807 E += icvAddTerm( var, var1, 0, INFINITY, 0, 0, ebuf, nedges );
808 }
809 }
810 }
811 }
812 }
813
814 //t = (double)cvGetTickCount() - t;
815 ebuf[0].weight = ebuf[1].weight = 0;
816 E += icvGCMaxFlow( vbuf, nvtx, ebuf, state2->orphans, state2->maxOrphans );
817
818 if( E < Eprev )
819 {
820 for( y = 0; y < rows; y++ )
821 {
822 short* dleft = dleft0 + dstep*y;
823 short* dright = dright0 + dstep*y;
824 GCVtx** pleft = pleft0 + pstep*y;
825 GCVtx** pright = pright0 + pstep*y;
826 for( x = 0; x < cols; x++ )
827 {
828 GCVtx* var2 = pleft[x];
829 if( var2 && var2->parent && var2->t )
830 dleft[x] = (short)alpha;
831
832 var2 = pright[x];
833 if( var2 && var2->parent && var2->t )
834 dright[x] = (short)-alpha;
835 }
836 }
837 }
838
839 return MIN(E, Eprev);
840}

843CV_IMPL void cvFindStereoCorrespondenceGC( const CvArr* _left, const CvArr* _right,
844 CvArr* _dispLeft, CvArr* _dispRight, CvStereoGCState* state, int useDisparityGuess )
845{
846 CvStereoGCState2 state2;
847 state2.orphans = 0;
848 state2.maxOrphans = 0;
849
850 CvMat lstub, *left = cvGetMat( _left, &lstub );
851 CvMat rstub, *right = cvGetMat( _right, &rstub );
852 CvMat dlstub, *dispLeft = cvGetMat( _dispLeft, &dlstub );
853 CvMat drstub, *dispRight = cvGetMat( _dispRight, &drstub );
854 CvSize size;
855 int iter, i, nZeroExpansions = 0;
856 CvRNG rng = cvRNG(-1);
857 int64 E;
858
859 CV_Assert( state != 0 );
860 CV_Assert( CV_ARE_SIZES_EQ(left, right) && CV_ARE_TYPES_EQ(left, right) &&
861 CV_MAT_TYPE(left->type) == CV_8UC1 );
862 CV_Assert( !dispLeft ||
863 (CV_ARE_SIZES_EQ(dispLeft, left) && CV_MAT_CN(dispLeft->type) == 1) );
864 CV_Assert( !dispRight ||
865 (CV_ARE_SIZES_EQ(dispRight, left) && CV_MAT_CN(dispRight->type) == 1) );
866
867 size = cvGetSize(left);
868 if( !state->left || state->left->width != size.width || state->left->height != size.height )
869 {
870 int pcn = (int)(sizeof(GCVtx*)/sizeof(int));
871 int vcn = (int)(sizeof(GCVtx)/sizeof(int));
872 int ecn = (int)(sizeof(GCEdge)/sizeof(int));
873 cvReleaseMat( &state->left );
874 cvReleaseMat( &state->right );
875 cvReleaseMat( &state->ptrLeft );
876 cvReleaseMat( &state->ptrRight );
877 cvReleaseMat( &state->dispLeft );
878 cvReleaseMat( &state->dispRight );
879
880 state->left = cvCreateMat( size.height, size.width, CV_8UC3 );
881 state->right = cvCreateMat( size.height, size.width, CV_8UC3 );
882 state->dispLeft = cvCreateMat( size.height, size.width, CV_16SC1 );
883 state->dispRight = cvCreateMat( size.height, size.width, CV_16SC1 );
884 state->ptrLeft = cvCreateMat( size.height, size.width, CV_32SC(pcn) );
885 state->ptrRight = cvCreateMat( size.height, size.width, CV_32SC(pcn) );
886 state->vtxBuf = cvCreateMat( 1, size.height*size.width*2, CV_32SC(vcn) );
887 state->edgeBuf = cvCreateMat( 1, size.height*size.width*12 + 16, CV_32SC(ecn) );
888 }
889
890 if( !useDisparityGuess )
891 {
892 cvSet( state->dispLeft, cvScalarAll(OCCLUDED));
893 cvSet( state->dispRight, cvScalarAll(OCCLUDED));
894 }
895 else
896 {
897 CV_Assert( dispLeft && dispRight );
898 cvConvert( dispLeft, state->dispLeft );
899 cvConvert( dispRight, state->dispRight );
900 }
901
902 state2.Ithreshold = state->Ithreshold;
903 state2.interactionRadius = state->interactionRadius;
904 state2.lambda = cvRound(state->lambda*DENOMINATOR);
905 state2.lambda1 = cvRound(state->lambda1*DENOMINATOR);
906 state2.lambda2 = cvRound(state->lambda2*DENOMINATOR);
907 state2.K = cvRound(state->K*DENOMINATOR);
908
909 icvInitStereoConstTabs();
910 icvInitGraySubpix( left, right, state->left, state->right );
911
912 std::vector<int> disp(state->numberOfDisparities);
913 CvMat _disp = cvMat( 1, (int)disp.size(), CV_32S, &disp[0] );
914 cvRange( &_disp, state->minDisparity, state->minDisparity + state->numberOfDisparities );
915 cvRandShuffle( &_disp, &rng );
916
917 if( state2.lambda < 0 && (state2.K < 0 || state2.lambda1 < 0 || state2.lambda2 < 0) )
918 {
919 float L = icvComputeK(state)*0.2f;
920 state2.lambda = cvRound(L*DENOMINATOR);
921 }
922
923 if( state2.K < 0 )
924 state2.K = state2.lambda*5;
925 if( state2.lambda1 < 0 )
926 state2.lambda1 = state2.lambda*3;
927 if( state2.lambda2 < 0 )
928 state2.lambda2 = state2.lambda;
929
930 icvInitStereoTabs( &state2 );
931
932 E = icvComputeEnergy( state, &state2, !useDisparityGuess );
933 for( iter = 0; iter < state->maxIters; iter++ )
934 {
935 for( i = 0; i < state->numberOfDisparities; i++ )
936 {
937 int alpha = disp[i];
938 int64 Enew = icvAlphaExpand( E, -alpha, state, &state2 );
939 if( Enew < E )
940 {
941 nZeroExpansions = 0;
942 E = Enew;
943 }
944 else if( ++nZeroExpansions >= state->numberOfDisparities )
945 break;
946 }
947 }
948
949 if( dispLeft )
950 cvConvert( state->dispLeft, dispLeft );
951 if( dispRight )
952 cvConvert( state->dispRight, dispRight );
953
954 cvFree( &state2.orphans );
955}
