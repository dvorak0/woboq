50public:
51 typedef enum { PDT_GAUSS=1, PDT_BERNOULLI, PDT_DBFRIENDLY } PHI_DISTR_TYPE;
52 ~CSMatrixGenerator();
53 static float* getCSMatrix(int m, int n, PHI_DISTR_TYPE dt); // do NOT free returned pointer

56private:
57 static float *cs_phi_; // matrix for compressive sensing
58 static int cs_phi_m_, cs_phi_n_;

61float* CSMatrixGenerator::getCSMatrix(int m, int n, PHI_DISTR_TYPE dt)
62{
63 assert(m <= n);
64
65 if (cs_phi_m_!=m || cs_phi_n_!=n || cs_phi_==NULL) {
66 if (cs_phi_) delete [] cs_phi_;
67 cs_phi_ = new float[m*n];
68 }
69
70 #if 0 // debug - load the random matrix from a file (for reproducability of results)
71 //assert(m == 176);
72 //assert(n == 500);
73 //const char *phi = "/u/calonder/temp/dim_red/kpca_phi.txt";
74 const char *phi = "/u/calonder/temp/dim_red/debug_phi.txt";
75 std::ifstream ifs(phi);
76 for (size_t i=0; i<m*n; ++i) {
77 if (!ifs.good()) {
78 printf("[ERROR] RandomizedTree::makeRandomMeasMatrix: problem reading '%s'\n", phi);
79 exit(0);
80 }
81 ifs >> cs_phi[i];
82 }
83 ifs.close();
84
85 static bool warned=false;
86 if (!warned) {
87 printf("[NOTE] RT: reading %ix%i PHI matrix from '%s'...\n", m, n, phi);
88 warned=true;
89 }
90
91 return;
92 #endif
93
94 float *cs_phi = cs_phi_;
95
96 if (m == n) {
97 // special case - set to 0 for safety
98 memset(cs_phi, 0, m*n*sizeof(float));
99 printf("[WARNING] %s:%i: square CS matrix (-> no reduction)\n", __FILE__, __LINE__);
100 }
101 else {
102 cv::RNG rng(23);
103
104 // par is distr param, cf 'Favorable JL Distributions' (Baraniuk et al, 2006)
105 if (dt == PDT_GAUSS) {
106 float par = (float)(1./m);
107 for (int i=0; i<m*n; ++i)
108 *cs_phi++ = (float)rng.gaussian(par);
109 }
110 else if (dt == PDT_BERNOULLI) {
111 float par = (float)(1./sqrt((float)m));
112 for (int i=0; i<m*n; ++i)
113 *cs_phi++ = (rng(2)==0 ? par : -par);
114 }
115 else if (dt == PDT_DBFRIENDLY) {
116 float par = (float)sqrt(3./m);
117 for (int i=0; i<m*n; ++i) {
118 int r = rng(6);
119 *cs_phi++ = (r==0 ? par : (r==1 ? -par : 0.f));
120 }
121 }
122 else
123 throw("PHI_DISTR_TYPE not implemented.");
124 }
125
126 return cs_phi_;
127}

149// sum up 50 byte vectors of length 176
150// assume 4 bits max for input vector values
151// final shift is 2 bits right
152// temp buffer should be twice as long as signature
153// sig and buffer need not be initialized
154inline void sum_50t_176c(uchar **pp, uchar *sig, unsigned short *temp)
155{
156#if CV_SSE2
157 __m128i acc, *acc1, *acc2, *acc3, *acc4, tzero;
158 __m128i *ssig, *ttemp;
159
160 ssig = (__m128i *)sig;
161 ttemp = (__m128i *)temp;
162
163 // empty ttemp[]
164 tzero = _mm_set_epi32(0, 0, 0, 0);
165 for (int i=0; i<22; i++)
166 ttemp[i] = tzero;
167
168 for (int j=0; j<48; j+=16)
169 {
170 // empty ssig[]
171 for (int i=0; i<11; i++)
172 ssig[i] = tzero;
173
174 for (int i=j; i<j+16; i+=4) // 4 columns at a time, to 16
175 {
176 acc1 = (__m128i *)pp[i];
177 acc2 = (__m128i *)pp[i+1];
178 acc3 = (__m128i *)pp[i+2];
179 acc4 = (__m128i *)pp[i+3];
180
181 // add next four columns
182 acc = _mm_adds_epu8(acc1[0],acc2[0]);
183 acc = _mm_adds_epu8(acc,acc3[0]);
184 acc = _mm_adds_epu8(acc,acc4[1]);
185 ssig[0] = _mm_adds_epu8(acc,ssig[0]);
186 // add four columns
187 acc = _mm_adds_epu8(acc1[1],acc2[1]);
188 acc = _mm_adds_epu8(acc,acc3[1]);
189 acc = _mm_adds_epu8(acc,acc4[1]);
190 ssig[1] = _mm_adds_epu8(acc,ssig[1]);
191 // add four columns
192 acc = _mm_adds_epu8(acc1[2],acc2[2]);
193 acc = _mm_adds_epu8(acc,acc3[2]);
194 acc = _mm_adds_epu8(acc,acc4[2]);
195 ssig[2] = _mm_adds_epu8(acc,ssig[2]);
196 // add four columns

198 acc = _mm_adds_epu8(acc,acc3[3]);
199 acc = _mm_adds_epu8(acc,acc4[3]);
200 ssig[3] = _mm_adds_epu8(acc,ssig[3]);
201 // add four columns
202 acc = _mm_adds_epu8(acc1[4],acc2[4]);
203 acc = _mm_adds_epu8(acc,acc3[4]);
204 acc = _mm_adds_epu8(acc,acc4[4]);
205 ssig[4] = _mm_adds_epu8(acc,ssig[4]);
206 // add four columns
207 acc = _mm_adds_epu8(acc1[5],acc2[5]);
208 acc = _mm_adds_epu8(acc,acc3[5]);
209 acc = _mm_adds_epu8(acc,acc4[5]);
210 ssig[5] = _mm_adds_epu8(acc,ssig[5]);
211 // add four columns
212 acc = _mm_adds_epu8(acc1[6],acc2[6]);
213 acc = _mm_adds_epu8(acc,acc3[6]);
214 acc = _mm_adds_epu8(acc,acc4[6]);
215 ssig[6] = _mm_adds_epu8(acc,ssig[6]);
216 // add four columns
217 acc = _mm_adds_epu8(acc1[7],acc2[7]);
218 acc = _mm_adds_epu8(acc,acc3[7]);
219 acc = _mm_adds_epu8(acc,acc4[7]);
220 ssig[7] = _mm_adds_epu8(acc,ssig[7]);
221 // add four columns
222 acc = _mm_adds_epu8(acc1[8],acc2[8]);
223 acc = _mm_adds_epu8(acc,acc3[8]);
224 acc = _mm_adds_epu8(acc,acc4[8]);
225 ssig[8] = _mm_adds_epu8(acc,ssig[8]);
226 // add four columns
227 acc = _mm_adds_epu8(acc1[9],acc2[9]);
228 acc = _mm_adds_epu8(acc,acc3[9]);
229 acc = _mm_adds_epu8(acc,acc4[9]);
230 ssig[9] = _mm_adds_epu8(acc,ssig[9]);
231 // add four columns
232 acc = _mm_adds_epu8(acc1[10],acc2[10]);
233 acc = _mm_adds_epu8(acc,acc3[10]);
234 acc = _mm_adds_epu8(acc,acc4[10]);
235 ssig[10] = _mm_adds_epu8(acc,ssig[10]);
236 }
237
238 // unpack to ttemp buffer and add
239 ttemp[0] = _mm_add_epi16(_mm_unpacklo_epi8(ssig[0],tzero),ttemp[0]);
240 ttemp[1] = _mm_add_epi16(_mm_unpackhi_epi8(ssig[0],tzero),ttemp[1]);
241 ttemp[2] = _mm_add_epi16(_mm_unpacklo_epi8(ssig[1],tzero),ttemp[2]);
242 ttemp[3] = _mm_add_epi16(_mm_unpackhi_epi8(ssig[1],tzero),ttemp[3]);
243 ttemp[4] = _mm_add_epi16(_mm_unpacklo_epi8(ssig[2],tzero),ttemp[4]);
244 ttemp[5] = _mm_add_epi16(_mm_unpackhi_epi8(ssig[2],tzero),ttemp[5]);
245 ttemp[6] = _mm_add_epi16(_mm_unpacklo_epi8(ssig[3],tzero),ttemp[6]);
246 ttemp[7] = _mm_add_epi16(_mm_unpackhi_epi8(ssig[3],tzero),ttemp[7]);
247 ttemp[8] = _mm_add_epi16(_mm_unpacklo_epi8(ssig[4],tzero),ttemp[8]);
248 ttemp[9] = _mm_add_epi16(_mm_unpackhi_epi8(ssig[4],tzero),ttemp[9]);
249 ttemp[10] = _mm_add_epi16(_mm_unpacklo_epi8(ssig[5],tzero),ttemp[10]);
250 ttemp[11] = _mm_add_epi16(_mm_unpackhi_epi8(ssig[5],tzero),ttemp[11]);
251 ttemp[12] = _mm_add_epi16(_mm_unpacklo_epi8(ssig[6],tzero),ttemp[12]);
252 ttemp[13] = _mm_add_epi16(_mm_unpackhi_epi8(ssig[6],tzero),ttemp[13]);
253 ttemp[14] = _mm_add_epi16(_mm_unpacklo_epi8(ssig[7],tzero),ttemp[14]);
254 ttemp[15] = _mm_add_epi16(_mm_unpackhi_epi8(ssig[7],tzero),ttemp[15]);
255 ttemp[16] = _mm_add_epi16(_mm_unpacklo_epi8(ssig[8],tzero),ttemp[16]);
256 ttemp[17] = _mm_add_epi16(_mm_unpackhi_epi8(ssig[8],tzero),ttemp[17]);
257 ttemp[18] = _mm_add_epi16(_mm_unpacklo_epi8(ssig[9],tzero),ttemp[18]);
258 ttemp[19] = _mm_add_epi16(_mm_unpackhi_epi8(ssig[9],tzero),ttemp[19]);
259 ttemp[20] = _mm_add_epi16(_mm_unpacklo_epi8(ssig[10],tzero),ttemp[20]);
260 ttemp[21] = _mm_add_epi16(_mm_unpackhi_epi8(ssig[10],tzero),ttemp[21]);
261 }
262
263 // create ssignature from 16-bit result
264 ssig[0] =_mm_packus_epi16(_mm_srai_epi16(ttemp[0],2),_mm_srai_epi16(ttemp[1],2));
265 ssig[1] =_mm_packus_epi16(_mm_srai_epi16(ttemp[2],2),_mm_srai_epi16(ttemp[3],2));
266 ssig[2] =_mm_packus_epi16(_mm_srai_epi16(ttemp[4],2),_mm_srai_epi16(ttemp[5],2));
267 ssig[3] =_mm_packus_epi16(_mm_srai_epi16(ttemp[6],2),_mm_srai_epi16(ttemp[7],2));
268 ssig[4] =_mm_packus_epi16(_mm_srai_epi16(ttemp[8],2),_mm_srai_epi16(ttemp[9],2));
269 ssig[5] =_mm_packus_epi16(_mm_srai_epi16(ttemp[10],2),_mm_srai_epi16(ttemp[11],2));
270 ssig[6] =_mm_packus_epi16(_mm_srai_epi16(ttemp[12],2),_mm_srai_epi16(ttemp[13],2));
271 ssig[7] =_mm_packus_epi16(_mm_srai_epi16(ttemp[14],2),_mm_srai_epi16(ttemp[15],2));
272 ssig[8] =_mm_packus_epi16(_mm_srai_epi16(ttemp[16],2),_mm_srai_epi16(ttemp[17],2));
273 ssig[9] =_mm_packus_epi16(_mm_srai_epi16(ttemp[18],2),_mm_srai_epi16(ttemp[19],2));
274 ssig[10] =_mm_packus_epi16(_mm_srai_epi16(ttemp[20],2),_mm_srai_epi16(ttemp[21],2));
275#else
276 (void)pp;
277 (void)sig;
278 (void)temp;
279 CV_Error( CV_StsNotImplemented, "Not supported without SSE2" );
280#endif

306int RandomizedTree::getIndex(uchar* patch_data) const
307{
308 int index = 0;
309 for (int d = 0; d < depth_; ++d) {
310 int child_offset = nodes_[index](patch_data);
311 index = 2*index + 1 + child_offset;
312 }
313 return (int)(index - nodes_.size());
314}

324void RandomizedTree::train(std::vector<BaseKeypoint> const& base_set,
325 RNG &rng, PatchGenerator &make_patch,
326 int _depth, int views, size_t reduced_num_dim,
327 int num_quant_bits)
328{
329 init((int)base_set.size(), _depth, rng);
330
331 Mat patch;
332
333 // Estimate posterior probabilities using random affine views
334 std::vector<BaseKeypoint>::const_iterator keypt_it;
335 int class_id = 0;
336 Size patchSize(PATCH_SIZE, PATCH_SIZE);
337 for (keypt_it = base_set.begin(); keypt_it != base_set.end(); ++keypt_it, ++class_id) {
338 for (int i = 0; i < views; ++i) {
339 make_patch( Mat(keypt_it->image), Point(keypt_it->x, keypt_it->y ), patch, patchSize, rng );
340 IplImage iplPatch = patch;
341 addExample(class_id, getData(&iplPatch));
342 }
343 }
344
345 finalize(reduced_num_dim, num_quant_bits);
346}

348void RandomizedTree::allocPosteriorsAligned(int num_leaves, int num_classes)
349{
350 freePosteriors(3);
351
352 posteriors_ = new float*[num_leaves]; //(float**) malloc(num_leaves*sizeof(float*));
353 for (int i=0; i<num_leaves; ++i) {
354 posteriors_[i] = (float*)cvAlloc(num_classes*sizeof(posteriors_[i][0]));

359 for (int i=0; i<num_leaves; ++i) {

362 }

364 classes_ = num_classes;

367void RandomizedTree::freePosteriors(int which)
368{
369 if (posteriors_ && (which&1)) {

371 if (posteriors_[i])
372 cvFree( &posteriors_[i] );
373 delete [] posteriors_;
374 posteriors_ = NULL;
375 }
376
377 if (posteriors2_ && (which&2)) {
378 for (int i=0; i<num_leaves_; ++i)
379 cvFree( &posteriors2_[i] );
380 delete [] posteriors2_;
381 posteriors2_ = NULL;
382 }
383
384 classes_ = -1;

387void RandomizedTree::init(int num_classes, int _depth, RNG &rng)
388{
389 depth_ = _depth;
390 num_leaves_ = 1 << _depth; // 2**d
391 int num_nodes = num_leaves_ - 1; // 2**d - 1
392
393 // Initialize probabilities and counts to 0
394 allocPosteriorsAligned(num_leaves_, num_classes); // will set classes_ correctly
395 for (int i = 0; i < num_leaves_; ++i)
396 memset((void*)posteriors_[i], 0, num_classes*sizeof(float));
397 leaf_counts_.resize(num_leaves_);
398
399 for (int i = 0; i < num_leaves_; ++i)
400 memset((void*)posteriors2_[i], 0, num_classes*sizeof(uchar));
401
402 createNodes(num_nodes, rng);
403}

413// returns the p% percentile of data (length n vector)
414static float percentile(float *data, int n, float p)
415{
416 assert(n>0);
417 assert(p>=0 && p<=1);
418 std::vector<float> vec(data, data+n);
419 std::sort(vec.begin(), vec.end());
420 int ix = (int)(p*(n-1));
421 return vec[ix];
422}

424void RandomizedTree::finalize(size_t reduced_num_dim, int num_quant_bits)
425{
426 // Normalize by number of patches to reach each leaf
427 for (int index = 0; index < num_leaves_; ++index) {
428 float* posterior = posteriors_[index];
429 assert(posterior != NULL);
430 int count = leaf_counts_[index];
431 if (count != 0) {
432 float normalizer = 1.0f / count;
433 for (int c = 0; c < classes_; ++c) {
434 *posterior *= normalizer;
435 ++posterior;
436 }
437 }
438 }
439 leaf_counts_.clear();
440
441 // apply compressive sensing
442 if ((int)reduced_num_dim != classes_)
443 compressLeaves(reduced_num_dim);
444 else {
445 static bool notified = false;
446 if (!notified)
447 printf("\n[OK] NO compression to leaves applied, dim=%i\n", (int)reduced_num_dim);
448 notified = true;
449 }
450
451 // convert float-posteriors to char-posteriors (quantization step)
452 makePosteriors2(num_quant_bits);
453}
454
455void RandomizedTree::compressLeaves(size_t reduced_num_dim)
456{
457 static bool warned = false;
458 if (!warned) {
459 printf("\n[OK] compressing leaves with phi %i x %i\n", (int)reduced_num_dim, (int)classes_);
460 warned = true;
461 }
462
463 static bool warned2 = false;
464 if ((int)reduced_num_dim == classes_) {
465 if (!warned2)
466 printf("[WARNING] RandomizedTree::compressLeaves: not compressing because reduced_dim == classes()\n");
467 warned2 = true;
468 return;
469 }
470
471 // DO NOT FREE RETURNED POINTER
472 float *cs_phi = CSMatrixGenerator::getCSMatrix((int)reduced_num_dim, classes_, CSMatrixGenerator::PDT_BERNOULLI);
473
474 float *cs_posteriors = new float[num_leaves_ * reduced_num_dim]; // temp, num_leaves_ x reduced_num_dim
475 for (int i=0; i<num_leaves_; ++i) {
476 float *post = getPosteriorByIndex(i);
477 float *prod = &cs_posteriors[i*reduced_num_dim];
478 Mat A( (int)reduced_num_dim, classes_, CV_32FC1, cs_phi );
479 Mat X( classes_, 1, CV_32FC1, post );
480 Mat Y( (int)reduced_num_dim, 1, CV_32FC1, prod );
481 Y = A*X;
482 }
483
484 // copy new posteriors
485 freePosteriors(3);
486 allocPosteriorsAligned(num_leaves_, (int)reduced_num_dim);
487 for (int i=0; i<num_leaves_; ++i)
488 memcpy(posteriors_[i], &cs_posteriors[i*reduced_num_dim], reduced_num_dim*sizeof(float));
489 classes_ = (int)reduced_num_dim;
490
491 delete [] cs_posteriors;
492}

494void RandomizedTree::makePosteriors2(int num_quant_bits)
495{
496 int N = (1<<num_quant_bits) - 1;
497
498 float perc[2];
499 estimateQuantPercForPosteriors(perc);
500
501 assert(posteriors_ != NULL);
502 for (int i=0; i<num_leaves_; ++i)
503 quantizeVector(posteriors_[i], classes_, N, perc, posteriors2_[i]);
504
505 // printf("makePosteriors2 quantization bounds: %.3e, %.3e (num_leaves=%i, N=%i)\n",
506 // perc[0], perc[1], num_leaves_, N);
507}

509void RandomizedTree::estimateQuantPercForPosteriors(float perc[2])
510{
511 // _estimate_ percentiles for this tree
512 // TODO: do this more accurately
513 assert(posteriors_ != NULL);
514 perc[0] = perc[1] = .0f;
515 for (int i=0; i<num_leaves_; i++) {
516 perc[0] += percentile(posteriors_[i], classes_, GET_LOWER_QUANT_PERC());
517 perc[1] += percentile(posteriors_[i], classes_, GET_UPPER_QUANT_PERC());
518 }
519 perc[0] /= num_leaves_;
520 perc[1] /= num_leaves_;
521}

544void RandomizedTree::quantizeVector(float *vec, int dim, int N, float bnds[2], int clamp_mode)
545{
546 float map_bnd[2] = {0.f,(float)N}; // bounds of quantized target interval we're mapping to
547 for (int k=0; k<dim; ++k, ++vec) {
548 *vec = float(int((*vec - bnds[0])/(bnds[1] - bnds[0])*(map_bnd[1] - map_bnd[0]) + map_bnd[0]));
549 // 0: clamp both, lower and upper values
550 if (clamp_mode == 0) *vec = (*vec<map_bnd[0]) ? map_bnd[0] : ((*vec>map_bnd[1]) ? map_bnd[1] : *vec);
551 // 1: clamp lower values only
552 else if (clamp_mode == 1) *vec = (*vec<map_bnd[0]) ? map_bnd[0] : *vec;
553 // 2: clamp upper values only
554 else if (clamp_mode == 2) *vec = (*vec>map_bnd[1]) ? map_bnd[1] : *vec;
555 // 4: no clamping
556 else if (clamp_mode == 4) ; // yep, nothing
557 else {
558 printf("clamp_mode == %i is not valid (%s:%i).\n", clamp_mode, __FILE__, __LINE__);
559 exit(1);
560 }
561 }

565void RandomizedTree::quantizeVector(float *vec, int dim, int N, float bnds[2], uchar *dst)
566{
567 int map_bnd[2] = {0, N}; // bounds of quantized target interval we're mapping to
568 int tmp;
569 for (int k=0; k<dim; ++k) {
570 tmp = int((*vec - bnds[0])/(bnds[1] - bnds[0])*(map_bnd[1] - map_bnd[0]) + map_bnd[0]);
571 *dst = (uchar)((tmp<0) ? 0 : ((tmp>N) ? N : tmp));
572 ++vec;
573 ++dst;
574 }
575}

774{

776 uchar buffer[RandomizedTree::PATCH_SIZE * RandomizedTree::PATCH_SIZE];
