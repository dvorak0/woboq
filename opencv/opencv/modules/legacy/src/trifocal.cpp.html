120static void icvNormalizePoints( CvMat* points, CvMat* normPoints,CvMat* cameraMatr )
121{
122 /* Normalize image points using camera matrix */
123
124 CV_FUNCNAME( "icvNormalizePoints" );
125 __BEGIN__;
126
127 /* Test for null pointers */
128 if( points == 0 || normPoints == 0 || cameraMatr == 0 )
129 {
130 CV_ERROR( CV_StsNullPtr, "Some of parameters is a NULL pointer" );
131 }
132
133 if( !CV_IS_MAT(points) || !CV_IS_MAT(normPoints) || !CV_IS_MAT(cameraMatr) )
134 {
135 CV_ERROR( CV_StsUnsupportedFormat, "Input parameters must be a matrices" );
136 }
137
138 int numPoints;
139 numPoints = points->cols;
140 if( numPoints <= 0 || numPoints != normPoints->cols )
141 {
142 CV_ERROR( CV_StsUnmatchedSizes, "Number of points must be the same and more than 0" );
143 }
144
145 if( normPoints->rows != 2 || normPoints->rows != points->rows )
146 {
147 CV_ERROR( CV_StsUnmatchedSizes, "Points must have 2 coordinates" );
148 }
149
150 if(cameraMatr->rows != 3 || cameraMatr->cols != 3)
151 {
152 CV_ERROR( CV_StsUnmatchedSizes, "Size of camera matrix must be 3x3" );
153 }
154
155 double fx,fy,cx,cy;
156
157 fx = cvmGet(cameraMatr,0,0);
158 fy = cvmGet(cameraMatr,1,1);
159 cx = cvmGet(cameraMatr,0,2);
160 cy = cvmGet(cameraMatr,1,2);
161
162 int i;
163 for( i = 0; i < numPoints; i++ )
164 {
165 cvmSet(normPoints, 0, i, (cvmGet(points,0,i) - cx) / fx );
166 cvmSet(normPoints, 1, i, (cvmGet(points,1,i) - cy) / fy );
167 }
168
169 __END__;
170
171 return;
172}

179int icvComputeProjectMatrices6Points( CvMat* points1,CvMat* points2,CvMat* points3,
180 CvMat* projMatr1,CvMat* projMatr2,CvMat* projMatr3/*,
181 CvMat* points4D*/)
182{
183 /* Test input data correctness */
184
185 int numSol = 0;
186
187 CV_FUNCNAME( "icvComputeProjectMatrices6Points" );
188 __BEGIN__;
189
190 /* Test for null pointers */
191 if( points1 == 0 || points2 == 0 || points3 == 0 ||
192 projMatr1 == 0 || projMatr2 == 0 || projMatr3 == 0 )
193 {
194 CV_ERROR( CV_StsNullPtr, "Some of parameters is a NULL pointer" );
195 }
196
197 if( !CV_IS_MAT(points1) || !CV_IS_MAT(points2) || !CV_IS_MAT(points3) ||
198 !CV_IS_MAT(projMatr1) || !CV_IS_MAT(projMatr2) || !CV_IS_MAT(projMatr3) )
199 {
200 CV_ERROR( CV_StsUnsupportedFormat, "Input parameters must be a matrices" );
201 }
202
203 if( (points1->cols != points2->cols) || (points1->cols != points3->cols) || (points1->cols != 6) /* || (points4D->cols !=6) */)
204 {
205 CV_ERROR( CV_StsUnmatchedSizes, "Number of points must be same and == 6" );
206 }
207
208 if( points1->rows != 2 || points2->rows != 2 || points3->rows != 2 )
209 {
210 CV_ERROR( CV_StsUnmatchedSizes, "Number of points coordinates must be 2" );
211 }
212
213 if( projMatr1->cols != 4 || projMatr2->cols != 4 || projMatr3->cols != 4 ||
214 (!(projMatr1->rows == 3 && projMatr2->rows == 3 && projMatr3->rows == 3) &&
215 !(projMatr1->rows == 9 && projMatr2->rows == 9 && projMatr3->rows == 9)) )
216 {
217 CV_ERROR( CV_StsUnmatchedSizes, "Size of project matrix must be 3x4 or 9x4 (for 3 matrices)" );
218 }
219
220#if 0
221 if( points4D->row != 4 )
222 {
223 CV_ERROR( CV_StsUnmatchedSizes, "Number of coordinates of points4D must be 4" );
224 }
225#endif
226
227 /* Find transform matrix for each camera */
228 int i;
229 CvMat* points[3];
230 points[0] = points1;
231 points[1] = points2;
232 points[2] = points3;
233
234 CvMat* projMatrs[3];
235 projMatrs[0] = projMatr1;
236 projMatrs[1] = projMatr2;
237 projMatrs[2] = projMatr3;
238
239 CvMat transMatr;
240 double transMatr_dat[9];
241 transMatr = cvMat(3,3,CV_64F,transMatr_dat);
242
243 CvMat corrPoints1;
244 CvMat corrPoints2;
245
246 double corrPoints_dat[3*3*2];/* 3-point(images) by 3-coordinates by 2-correspondence*/
247
248 corrPoints1 = cvMat(3,3,CV_64F,corrPoints_dat); /* 3-coordinates for each of 3-points(3-image) */
249 corrPoints2 = cvMat(3,3,CV_64F,corrPoints_dat+9);/* 3-coordinates for each of 3-points(3-image) */
250
251 for( i = 0; i < 3; i++ )/* for each image */
252 {
253 /* Get last 4 points for computing transformation */
254 CvMat tmpPoints;
255 /* find base points transform for last four points on i-th image */
256 cvGetSubRect(points[i],&tmpPoints,cvRect(2,0,4,2));
257 icvFindBaseTransform(&tmpPoints,&transMatr);
258
259 {/* We have base transform. Compute error scales for three first points */
260 CvMat trPoint;
261 double trPoint_dat[3*3];
262 trPoint = cvMat(3,3,CV_64F,trPoint_dat);
263 /* fill points */
264 for( int kk = 0; kk < 3; kk++ )
265 {
266 cvmSet(&trPoint,0,kk,cvmGet(points[i],0,kk+2));
267 cvmSet(&trPoint,1,kk,cvmGet(points[i],1,kk+2));
268 cvmSet(&trPoint,2,kk,1);
269 }
270
271 /* Transform points */
272 CvMat resPnts;
273 double resPnts_dat[9];
274 resPnts = cvMat(3,3,CV_64F,resPnts_dat);
275 cvmMul(&transMatr,&trPoint,&resPnts);
276 }
277
278 /* Transform two first points */
279 for( int j = 0; j < 2; j++ )
280 {
281 CvMat pnt;
282 double pnt_dat[3];
283 pnt = cvMat(3,1,CV_64F,pnt_dat);
284 pnt_dat[0] = cvmGet(points[i],0,j);
285 pnt_dat[1] = cvmGet(points[i],1,j);
286 pnt_dat[2] = 1.0;
287
288 CvMat trPnt;
289 double trPnt_dat[3];
290 trPnt = cvMat(3,1,CV_64F,trPnt_dat);
291
292 cvmMul(&transMatr,&pnt,&trPnt);
293
294 /* Collect transformed points */
295 corrPoints_dat[j * 9 + 0 * 3 + i] = trPnt_dat[0];/* x */
296 corrPoints_dat[j * 9 + 1 * 3 + i] = trPnt_dat[1];/* y */
297 corrPoints_dat[j * 9 + 2 * 3 + i] = trPnt_dat[2];/* w */
298 }
299 }
300
301 /* We have computed corr points. Now we can compute generators for reduced fundamental matrix */
302
303 /* Compute generators for reduced fundamental matrix from 3 pair of collect points */
304 CvMat fundReduceCoef1;
305 CvMat fundReduceCoef2;
306 double fundReduceCoef1_dat[5];
307 double fundReduceCoef2_dat[5];
308
309 fundReduceCoef1 = cvMat(1,5,CV_64F,fundReduceCoef1_dat);
310 fundReduceCoef2 = cvMat(1,5,CV_64F,fundReduceCoef2_dat);
311
312 GetGeneratorReduceFundSolution(&corrPoints1, &corrPoints2, &fundReduceCoef1, &fundReduceCoef2);
313
314 /* Choose best solutions for two generators. We can get 3 solutions */
315 CvMat resFundReduceCoef;
316 double resFundReduceCoef_dat[3*5];
317
318 resFundReduceCoef = cvMat(3,5,CV_64F,resFundReduceCoef_dat);
319
320 numSol = GetGoodReduceFundamMatrFromTwo(&fundReduceCoef1, &fundReduceCoef2,&resFundReduceCoef);
321
322 int maxSol;
323 maxSol = projMatrs[0]->rows / 3;
324
325 int currSol;
326 for( currSol = 0; (currSol < numSol && currSol < maxSol); currSol++ )
327 {
328 /* For current solution compute projection matrix */
329 CvMat fundCoefs;
330 cvGetSubRect(&resFundReduceCoef, &fundCoefs, cvRect(0,currSol,5,1));
331
332 CvMat projMatrCoefs;
333 double projMatrCoefs_dat[4];
334 projMatrCoefs = cvMat(1,4,CV_64F,projMatrCoefs_dat);
335
336 GetProjMatrFromReducedFundamental(&fundCoefs,&projMatrCoefs);
337 /* we have computed coeffs for reduced project matrix */
338
339 CvMat objPoints;
340 double objPoints_dat[4*6];
341 objPoints = cvMat(4,6,CV_64F,objPoints_dat);
342 cvZero(&objPoints);
343
344 /* fill object points */
345 for( i =0; i < 4; i++ )
346 {
347 objPoints_dat[i*6] = 1;
348 objPoints_dat[i*6+1] = projMatrCoefs_dat[i];
349 objPoints_dat[i*7+2] = 1;
350 }
351
352 int currCamera;
353 for( currCamera = 0; currCamera < 3; currCamera++ )
354 {
355
356 CvMat projPoints;
357 double projPoints_dat[3*6];
358 projPoints = cvMat(3,6,CV_64F,projPoints_dat);
359
360 /* fill projected points for current camera */
361 for( i = 0; i < 6; i++ )/* for each points for current camera */
362 {
363 projPoints_dat[6*0+i] = cvmGet(points[currCamera],0,i);/* x */
364 projPoints_dat[6*1+i] = cvmGet(points[currCamera],1,i);/* y */
365 projPoints_dat[6*2+i] = 1;/* w */
366 }
367
368 /* compute project matrix for current camera */
369 CvMat projMatrix;
370 double projMatrix_dat[3*4];
371 projMatrix = cvMat(3,4,CV_64F,projMatrix_dat);
372
373 icvComputeProjectMatrix(&objPoints,&projPoints,&projMatrix);
374
375 /* Add this matrix to result */
376 CvMat tmpSubRes;
377 cvGetSubRect(projMatrs[currCamera],&tmpSubRes,cvRect(0,currSol*3,4,3));
378 cvConvert(&projMatrix,&tmpSubRes);
379 }
380
381 /* We know project matrices. And we can reconstruct 6 3D-points if need */
382#if 0
383 if( points4D )
384 {
385 if( currSol < points4D->rows / 4 )
386 {
387 CvMat tmpPoints4D;
388 double tmpPoints4D_dat[4*6];
389 tmpPoints4D = cvMat(4,6,CV_64F,tmpPoints4D_dat);
390
391 icvReconstructPointsFor3View( &wProjMatr[0], &wProjMatr[1], &wProjMatr[2],
392 points1, points2, points3,
393 &tmpPoints4D);
394
395 CvMat tmpSubRes;
396 cvGetSubRect(points4D,tmpSubRes,cvRect(0,currSol*4,6,4));
397 cvConvert(tmpPoints4D,points4D);
398 }
399 }
400#endif
401
402 }/* for all sollutions */
403
404 __END__;
405 return numSol;
406}

409static int icvGetRandNumbers(int range,int count,int* arr)
410{
411 /* Generate random numbers [0,range-1] */
412
413 CV_FUNCNAME( "icvGetRandNumbers" );
414 __BEGIN__;
415
416 /* Test input data */
417 if( arr == 0 )
418 {
419 CV_ERROR( CV_StsNullPtr, "Parameter 'arr' is a NULL pointer" );
420 }
421
422
423 /* Test for errors input data */
424 if( range < count || range <= 0 )
425 {
426 CV_ERROR( CV_StsOutOfRange, "Can't generate such numbers. Count must be <= range and range must be > 0" );
427 }
428
429 int i,j;
430 int newRand;
431 for( i = 0; i < count; i++ )
432 {
433
434 int haveRep = 0;/* firstly we have not repeats */
435 do
436 {
437 /* generate new number */
438 newRand = rand()%range;
439 haveRep = 0;
440 /* Test for repeats in previous numbers */
441 for( j = 0; j < i; j++ )
442 {
443 if( arr[j] == newRand )
444 {
445 haveRep = 1;
446 break;
447 }
448 }
449 } while(haveRep);
450
451 /* We have good random number */
452 arr[i] = newRand;
453 }
454 __END__;
455 return 1;
456}

505static void icvProject4DPoints(CvMat* points4D,CvMat* projMatr, CvMat* projPoints)
506{
507
508 CvMat* tmpProjPoints = 0;
509
510 CV_FUNCNAME( "icvProject4DPoints" );
511
512 __BEGIN__;
513
514 if( points4D == 0 || projMatr == 0 || projPoints == 0)
515 {
516 CV_ERROR( CV_StsNullPtr, "Some of parameters is a NULL pointer" );
517 }
518
519 if( !CV_IS_MAT(points4D) || !CV_IS_MAT(projMatr) || !CV_IS_MAT(projPoints) )
520 {
521 CV_ERROR( CV_StsUnsupportedFormat, "Input parameters must be a matrices" );
522 }
523
524 int numPoints;
525 numPoints = points4D->cols;
526 if( numPoints < 1 )
527 {
528 CV_ERROR( CV_StsOutOfRange, "Number of points4D must be more than zero" );
529 }
530
531 if( numPoints != projPoints->cols )
532 {
533 CV_ERROR( CV_StsUnmatchedSizes, "Number of points must be the same");
534 }
535
536 if( projPoints->rows != 2 )
537 {
538 CV_ERROR( CV_StsUnmatchedSizes, "Number of coordinates of projected points must be 2");
539 }
540
541 if( points4D->rows != 4 )
542 {
543 CV_ERROR(CV_StsUnmatchedSizes, "Number of coordinates of 4D points must be 4");
544 }
545
546 if( projMatr->cols != 4 || projMatr->rows != 3 )
547 {
548 CV_ERROR( CV_StsUnmatchedSizes, "Size of projection matrix must be 3x4");
549 }
550
551
552 CV_CALL( tmpProjPoints = cvCreateMat(3,numPoints,CV_64F) );
553
554 cvmMul(projMatr,points4D,tmpProjPoints);
555
556 /* Scale points */
557 int i;
558 for( i = 0; i < numPoints; i++ )
559 {
560 double scale,x,y;
561
562 scale = cvmGet(tmpProjPoints,2,i);
563 x = cvmGet(tmpProjPoints,0,i);
564 y = cvmGet(tmpProjPoints,1,i);
565
566 if( fabs(scale) > 1e-7 )
567 {
568 x /= scale;
569 y /= scale;
570 }
571 else
572 {
573 x = 1e8;
574 y = 1e8;
575 }
576
577 cvmSet(projPoints,0,i,x);
578 cvmSet(projPoints,1,i,y);
579 }
580
581 __END__;
582
583 cvReleaseMat(&tmpProjPoints);
584
585 return;
586}
