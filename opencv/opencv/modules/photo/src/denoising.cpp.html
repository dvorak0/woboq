48void cv::fastNlMeansDenoising( InputArray _src, OutputArray _dst, float h,
49 int templateWindowSize, int searchWindowSize)
50{
51 Mat src = _src.getMat();
52 _dst.create(src.size(), src.type());
53 Mat dst = _dst.getMat();
54
55#ifdef HAVE_TEGRA_OPTIMIZATION
56 if(tegra::fastNlMeansDenoising(src, dst, h, templateWindowSize, searchWindowSize))
57 return;
58#endif
59
60 switch (src.type()) {
61 case CV_8U:
62 parallel_for_(cv::Range(0, src.rows),
63 FastNlMeansDenoisingInvoker<uchar>(
64 src, dst, templateWindowSize, searchWindowSize, h));
65 break;
66 case CV_8UC2:
67 parallel_for_(cv::Range(0, src.rows),
68 FastNlMeansDenoisingInvoker<cv::Vec2b>(
69 src, dst, templateWindowSize, searchWindowSize, h));
70 break;
71 case CV_8UC3:
72 parallel_for_(cv::Range(0, src.rows),
73 FastNlMeansDenoisingInvoker<cv::Vec3b>(
74 src, dst, templateWindowSize, searchWindowSize, h));
75 break;
76 default:
77 CV_Error(CV_StsBadArg,
78 "Unsupported image format! Only CV_8UC1, CV_8UC2 and CV_8UC3 are supported");
79 }
80}

82void cv::fastNlMeansDenoisingColored( InputArray _src, OutputArray _dst,
83 float h, float hForColorComponents,
84 int templateWindowSize, int searchWindowSize)
85{
86 Mat src = _src.getMat();
87 _dst.create(src.size(), src.type());
88 Mat dst = _dst.getMat();
89
90 if (src.type() != CV_8UC3) {
91 CV_Error(CV_StsBadArg, "Type of input image should be CV_8UC3!");
92 return;
93 }
94
95 Mat src_lab;
96 cvtColor(src, src_lab, CV_LBGR2Lab);
97
98 Mat l(src.size(), CV_8U);
99 Mat ab(src.size(), CV_8UC2);
100 Mat l_ab[] = { l, ab };
101 int from_to[] = { 0,0, 1,1, 2,2 };
102 mixChannels(&src_lab, 1, l_ab, 2, from_to, 3);
103
104 fastNlMeansDenoising(l, l, h, templateWindowSize, searchWindowSize);
105 fastNlMeansDenoising(ab, ab, hForColorComponents, templateWindowSize, searchWindowSize);
106
107 Mat l_ab_denoised[] = { l, ab };
108 Mat dst_lab(src.size(), src.type());
109 mixChannels(l_ab_denoised, 2, &dst_lab, 1, from_to, 3);
110
111 cvtColor(dst_lab, dst, CV_Lab2LBGR);
112}

114static void fastNlMeansDenoisingMultiCheckPreconditions(
115 const std::vector<Mat>& srcImgs,
116 int imgToDenoiseIndex, int temporalWindowSize,
117 int templateWindowSize, int searchWindowSize)
118{
119 int src_imgs_size = (int)srcImgs.size();
120 if (src_imgs_size == 0) {
121 CV_Error(CV_StsBadArg, "Input images vector should not be empty!");
122 }
123
124 if (temporalWindowSize % 2 == 0 ||
125 searchWindowSize % 2 == 0 ||
126 templateWindowSize % 2 == 0) {
127 CV_Error(CV_StsBadArg, "All windows sizes should be odd!");
128 }
129
130 int temporalWindowHalfSize = temporalWindowSize / 2;
131 if (imgToDenoiseIndex - temporalWindowHalfSize < 0 ||
132 imgToDenoiseIndex + temporalWindowHalfSize >= src_imgs_size)
133 {
134 CV_Error(CV_StsBadArg,
135 "imgToDenoiseIndex and temporalWindowSize "
136 "should be chosen corresponding srcImgs size!");
137 }
138
139 for (int i = 1; i < src_imgs_size; i++) {
140 if (srcImgs[0].size() != srcImgs[i].size() || srcImgs[0].type() != srcImgs[i].type()) {
141 CV_Error(CV_StsBadArg, "Input images should have the same size and type!");
142 }
143 }
144}

146void cv::fastNlMeansDenoisingMulti( InputArrayOfArrays _srcImgs, OutputArray _dst,
147 int imgToDenoiseIndex, int temporalWindowSize,
148 float h, int templateWindowSize, int searchWindowSize)
149{
150 vector<Mat> srcImgs;
151 _srcImgs.getMatVector(srcImgs);
152
153 fastNlMeansDenoisingMultiCheckPreconditions(
154 srcImgs, imgToDenoiseIndex,
155 temporalWindowSize, templateWindowSize, searchWindowSize
156 );
157 _dst.create(srcImgs[0].size(), srcImgs[0].type());
158 Mat dst = _dst.getMat();
159
160 switch (srcImgs[0].type()) {
161 case CV_8U:
162 parallel_for_(cv::Range(0, srcImgs[0].rows),
163 FastNlMeansMultiDenoisingInvoker<uchar>(
164 srcImgs, imgToDenoiseIndex, temporalWindowSize,
165 dst, templateWindowSize, searchWindowSize, h));
166 break;
167 case CV_8UC2:
168 parallel_for_(cv::Range(0, srcImgs[0].rows),
169 FastNlMeansMultiDenoisingInvoker<cv::Vec2b>(
170 srcImgs, imgToDenoiseIndex, temporalWindowSize,
171 dst, templateWindowSize, searchWindowSize, h));
172 break;
173 case CV_8UC3:
174 parallel_for_(cv::Range(0, srcImgs[0].rows),
175 FastNlMeansMultiDenoisingInvoker<cv::Vec3b>(
176 srcImgs, imgToDenoiseIndex, temporalWindowSize,
177 dst, templateWindowSize, searchWindowSize, h));
178 break;
179 default:
180 CV_Error(CV_StsBadArg,
181 "Unsupported matrix format! Only uchar, Vec2b, Vec3b are supported");
182 }
183}

185void cv::fastNlMeansDenoisingColoredMulti( InputArrayOfArrays _srcImgs, OutputArray _dst,
186 int imgToDenoiseIndex, int temporalWindowSize,
187 float h, float hForColorComponents,
188 int templateWindowSize, int searchWindowSize)
189{
190 vector<Mat> srcImgs;
191 _srcImgs.getMatVector(srcImgs);
192
193 fastNlMeansDenoisingMultiCheckPreconditions(
194 srcImgs, imgToDenoiseIndex,
195 temporalWindowSize, templateWindowSize, searchWindowSize
196 );
197
198 _dst.create(srcImgs[0].size(), srcImgs[0].type());
199 Mat dst = _dst.getMat();
200
201 int src_imgs_size = (int)srcImgs.size();
202
203 if (srcImgs[0].type() != CV_8UC3) {
204 CV_Error(CV_StsBadArg, "Type of input images should be CV_8UC3!");
205 return;
206 }
207
208 int from_to[] = { 0,0, 1,1, 2,2 };
209
210 // TODO convert only required images
211 vector<Mat> src_lab(src_imgs_size);
212 vector<Mat> l(src_imgs_size);
213 vector<Mat> ab(src_imgs_size);
214 for (int i = 0; i < src_imgs_size; i++) {
215 src_lab[i] = Mat::zeros(srcImgs[0].size(), CV_8UC3);
216 l[i] = Mat::zeros(srcImgs[0].size(), CV_8UC1);
217 ab[i] = Mat::zeros(srcImgs[0].size(), CV_8UC2);
218 cvtColor(srcImgs[i], src_lab[i], CV_LBGR2Lab);
219
220 Mat l_ab[] = { l[i], ab[i] };
221 mixChannels(&src_lab[i], 1, l_ab, 2, from_to, 3);
222 }
223
224 Mat dst_l;
225 Mat dst_ab;
226
227 fastNlMeansDenoisingMulti(
228 l, dst_l, imgToDenoiseIndex, temporalWindowSize,
229 h, templateWindowSize, searchWindowSize);
230
231 fastNlMeansDenoisingMulti(
232 ab, dst_ab, imgToDenoiseIndex, temporalWindowSize,
233 hForColorComponents, templateWindowSize, searchWindowSize);
234
235 Mat l_ab_denoised[] = { dst_l, dst_ab };
236 Mat dst_lab(srcImgs[0].size(), srcImgs[0].type());
237 mixChannels(l_ab_denoised, 2, &dst_lab, 1, from_to, 3);
238
239 cvtColor(dst_lab, dst, CV_Lab2LBGR);
240}
