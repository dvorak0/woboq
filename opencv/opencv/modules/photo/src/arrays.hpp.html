45template <class T> struct Array2d {
46 T* a;
47 int n1,n2;
48 bool needToDeallocArray;
49
50 Array2d(const Array2d& array2d):
51 a(array2d.a), n1(array2d.n1), n2(array2d.n2), needToDeallocArray(false)
52 {
53 if (array2d.needToDeallocArray) {
54 // copy constructor for self allocating arrays not supported
55 throw new exception();
56 }
57 }
58
59 Array2d(T* _a, int _n1, int _n2):
60 a(_a), n1(_n1), n2(_n2), needToDeallocArray(false) {}
61
62 Array2d(int _n1, int _n2):
63 n1(_n1), n2(_n2), needToDeallocArray(true)
64 {
65 a = new T[n1*n2];
66 }
67
68 ~Array2d() {
69 if (needToDeallocArray) {
70 delete[] a;
71 }
72 }
73
74 T* operator [] (int i) {
75 return a + i*n2;
76 }
77
78 inline T* row_ptr(int i) {
79 return (*this)[i];
80 }
81};

83template <class T> struct Array3d {
84 T* a;
85 int n1,n2,n3;
86 bool needToDeallocArray;
87
88 Array3d(T* _a, int _n1, int _n2, int _n3):
89 a(_a), n1(_n1), n2(_n2), n3(_n3), needToDeallocArray(false) {}
90
91 Array3d(int _n1, int _n2, int _n3):
92 n1(_n1), n2(_n2), n3(_n3), needToDeallocArray(true)
93 {
94 a = new T[n1*n2*n3];
95 }
96
97 ~Array3d() {
98 if (needToDeallocArray) {
99 delete[] a;
100 }
101 }
102
103 Array2d<T> operator [] (int i) {
104 Array2d<T> array2d(a + i*n2*n3, n2, n3);
105 return array2d;
106 }
107
108 inline T* row_ptr(int i1, int i2) {
109 return a + i1*n2*n3 + i2*n3;
110 }
111};

113template <class T> struct Array4d {
114 T* a;
115 int n1,n2,n3,n4;
116 bool needToDeallocArray;
117 int steps[4];
118
119 void init_steps() {
120 steps[0] = n2*n3*n4;
121 steps[1] = n3*n4;
122 steps[2] = n4;
123 steps[3] = 1;
124 }
125
126 Array4d(T* _a, int _n1, int _n2, int _n3, int _n4):
127 a(_a), n1(_n1), n2(_n2), n3(_n3), n4(_n4), needToDeallocArray(false)
128 {
129 init_steps();
130 }
131
132 Array4d(int _n1, int _n2, int _n3, int _n4):
133 n1(_n1), n2(_n2), n3(_n3), n4(_n4), needToDeallocArray(true)
134 {
135 a = new T[n1*n2*n3*n4];
136 init_steps();
137 }
138
139 ~Array4d() {
140 if (needToDeallocArray) {
141 delete[] a;
142 }
143 }
144
145 Array3d<T> operator [] (int i) {
146 Array3d<T> array3d(a + i*n2*n3*n4, n2, n3, n4);
147 return array3d;
148 }
149
150 inline T* row_ptr(int i1, int i2, int i3) {
151 return a + i1*n2*n3*n4 + i2*n3*n4 + i3*n4;
152 }
153
154 inline int step_size(int dimension) {
155 return steps[dimension];
156 }
157};
