105template <class T>
106FastNlMeansMultiDenoisingInvoker<T>::FastNlMeansMultiDenoisingInvoker(
107 const vector<Mat>& srcImgs,
108 int imgToDenoiseIndex,
109 int temporalWindowSize,
110 cv::Mat& dst,
111 int template_window_size,
112 int search_window_size,
113 const float h) : dst_(dst), extended_srcs_(srcImgs.size())
114{
115 CV_Assert(srcImgs.size() > 0);
116 CV_Assert(srcImgs[0].channels() == sizeof(T));
117
118 rows_ = srcImgs[0].rows;
119 cols_ = srcImgs[0].cols;
120
121 template_window_half_size_ = template_window_size / 2;
122 search_window_half_size_ = search_window_size / 2;
123 temporal_window_half_size_ = temporalWindowSize / 2;
124
125 template_window_size_ = template_window_half_size_ * 2 + 1;
126 search_window_size_ = search_window_half_size_ * 2 + 1;
127 temporal_window_size_ = temporal_window_half_size_ * 2 + 1;
128
129 border_size_ = search_window_half_size_ + template_window_half_size_;
130 for (int i = 0; i < temporal_window_size_; i++) {
131 copyMakeBorder(
132 srcImgs[imgToDenoiseIndex - temporal_window_half_size_ + i], extended_srcs_[i],
133 border_size_, border_size_, border_size_, border_size_, cv::BORDER_DEFAULT);
134 }
135 main_extended_src_ = extended_srcs_[temporal_window_half_size_];
136
137 const int max_estimate_sum_value =
138 temporal_window_size_ * search_window_size_ * search_window_size_ * 255;
139
140 fixed_point_mult_ = numeric_limits<int>::max() / max_estimate_sum_value;
141
142 // precalc weight for every possible l2 dist between blocks
143 // additional optimization of precalced weights to replace division(averaging) by binary shift
144 int template_window_size_sq = template_window_size_ * template_window_size_;
145 almost_template_window_size_sq_bin_shift = 0;
146 while (1 << almost_template_window_size_sq_bin_shift < template_window_size_sq) {
147 almost_template_window_size_sq_bin_shift++;
148 }
149
150 int almost_template_window_size_sq = 1 << almost_template_window_size_sq_bin_shift;
151 double almost_dist2actual_dist_multiplier =
152 ((double) almost_template_window_size_sq) / template_window_size_sq;
153
154 int max_dist = 255 * 255 * sizeof(T);
155 int almost_max_dist = (int) (max_dist / almost_dist2actual_dist_multiplier + 1);
156 almost_dist2weight.resize(almost_max_dist);
157
158 const double WEIGHT_THRESHOLD = 0.001;
159 for (int almost_dist = 0; almost_dist < almost_max_dist; almost_dist++) {
160 double dist = almost_dist * almost_dist2actual_dist_multiplier;
161 int weight = cvRound(fixed_point_mult_ * std::exp(-dist / (h * h * sizeof(T))));
162
163 if (weight < WEIGHT_THRESHOLD * fixed_point_mult_) {
164 weight = 0;
165 }
166
167 almost_dist2weight[almost_dist] = weight;
168 }
169 CV_Assert(almost_dist2weight[0] == fixed_point_mult_);
170 // additional optimization init end
171
172 if (dst_.empty()) {
173 dst_ = Mat::zeros(srcImgs[0].size(), srcImgs[0].type());
174 }
175}

177template <class T>
178void FastNlMeansMultiDenoisingInvoker<T>::operator() (const Range& range) const {
179 int row_from = range.start;
180 int row_to = range.end - 1;
181
182 Array3d<int> dist_sums(temporal_window_size_, search_window_size_, search_window_size_);
183
184 // for lazy calc optimization
185 Array4d<int> col_dist_sums(
186 template_window_size_, temporal_window_size_, search_window_size_, search_window_size_);
187
188 int first_col_num = -1;
189
190 Array4d<int> up_col_dist_sums(
191 cols_, temporal_window_size_, search_window_size_, search_window_size_);
192
193 for (int i = row_from; i <= row_to; i++) {
194 for (int j = 0; j < cols_; j++) {
195 int search_window_y = i - search_window_half_size_;
196 int search_window_x = j - search_window_half_size_;
197
198 // calc dist_sums
199 if (j == 0) {
200 calcDistSumsForFirstElementInRow(i, dist_sums, col_dist_sums, up_col_dist_sums);
201 first_col_num = 0;
202
203 } else { // calc cur dist_sums using previous dist_sums
204 if (i == row_from) {
205 calcDistSumsForElementInFirstRow(i, j, first_col_num,
206 dist_sums, col_dist_sums, up_col_dist_sums);
207
208 } else {
209 int ay = border_size_ + i;
210 int ax = border_size_ + j + template_window_half_size_;
211
212 int start_by =
213 border_size_ + i - search_window_half_size_;
214
215 int start_bx =
216 border_size_ + j - search_window_half_size_ + template_window_half_size_;
217
218 T a_up = main_extended_src_.at<T>(ay - template_window_half_size_ - 1, ax);
219 T a_down = main_extended_src_.at<T>(ay + template_window_half_size_, ax);
220
221 // copy class member to local variable for optimization
222 int search_window_size = search_window_size_;
223
224 for (int d = 0; d < temporal_window_size_; d++) {
225 Mat cur_extended_src = extended_srcs_[d];
226 Array2d<int> cur_dist_sums = dist_sums[d];
227 Array2d<int> cur_col_dist_sums = col_dist_sums[first_col_num][d];
228 Array2d<int> cur_up_col_dist_sums = up_col_dist_sums[j][d];
229 for (int y = 0; y < search_window_size; y++) {
230 int* dist_sums_row = cur_dist_sums.row_ptr(y);
231
232 int* col_dist_sums_row = cur_col_dist_sums.row_ptr(y);
233
234 int* up_col_dist_sums_row = cur_up_col_dist_sums.row_ptr(y);
235
236 const T* b_up_ptr =
237 cur_extended_src.ptr<T>(start_by - template_window_half_size_ - 1 + y);
238 const T* b_down_ptr =
239 cur_extended_src.ptr<T>(start_by + template_window_half_size_ + y);
240
241 for (int x = 0; x < search_window_size; x++) {
242 dist_sums_row[x] -= col_dist_sums_row[x];
243
244 col_dist_sums_row[x] = up_col_dist_sums_row[x] +
245 calcUpDownDist(
246 a_up, a_down,
247 b_up_ptr[start_bx + x], b_down_ptr[start_bx + x]
248 );
249
250 dist_sums_row[x] += col_dist_sums_row[x];
251
252 up_col_dist_sums_row[x] = col_dist_sums_row[x];
253
254 }
255 }
256 }
257 }
258
259 first_col_num = (first_col_num + 1) % template_window_size_;
260 }
261
262 // calc weights
263 int weights_sum = 0;
264
265 int estimation[3];
266 for (size_t channel_num = 0; channel_num < sizeof(T); channel_num++) {
267 estimation[channel_num] = 0;
268 }
269 for (int d = 0; d < temporal_window_size_; d++) {
270 const Mat& esrc_d = extended_srcs_[d];
271 for (int y = 0; y < search_window_size_; y++) {
272 const T* cur_row_ptr = esrc_d.ptr<T>(border_size_ + search_window_y + y);
273
274 int* dist_sums_row = dist_sums.row_ptr(d, y);
275
276 for (int x = 0; x < search_window_size_; x++) {
277 int almostAvgDist =
278 dist_sums_row[x] >> almost_template_window_size_sq_bin_shift;
279
280 int weight = almost_dist2weight[almostAvgDist];
281 weights_sum += weight;
282
283 T p = cur_row_ptr[border_size_ + search_window_x + x];
284 incWithWeight(estimation, weight, p);
285 }
286 }
287 }
288
289 for (size_t channel_num = 0; channel_num < sizeof(T); channel_num++)
290 estimation[channel_num] = ((unsigned)estimation[channel_num] + weights_sum / 2) / weights_sum;
291
292 dst_.at<T>(i,j) = saturateCastFromArray<T>(estimation);
293
294 }
295 }
296}

298template <class T>
299inline void FastNlMeansMultiDenoisingInvoker<T>::calcDistSumsForFirstElementInRow(
300 int i,
301 Array3d<int>& dist_sums,
302 Array4d<int>& col_dist_sums,
303 Array4d<int>& up_col_dist_sums) const
304{
305 int j = 0;
306
307 for (int d = 0; d < temporal_window_size_; d++) {
308 Mat cur_extended_src = extended_srcs_[d];
309 for (int y = 0; y < search_window_size_; y++) {
310 for (int x = 0; x < search_window_size_; x++) {
311 dist_sums[d][y][x] = 0;
312 for (int tx = 0; tx < template_window_size_; tx++) {
313 col_dist_sums[tx][d][y][x] = 0;
314 }
315
316 int start_y = i + y - search_window_half_size_;
317 int start_x = j + x - search_window_half_size_;
318
319 int* dist_sums_ptr = &dist_sums[d][y][x];
320 int* col_dist_sums_ptr = &col_dist_sums[0][d][y][x];
321 int col_dist_sums_step = col_dist_sums.step_size(0);
322 for (int tx = -template_window_half_size_; tx <= template_window_half_size_; tx++) {
323 for (int ty = -template_window_half_size_; ty <= template_window_half_size_; ty++) {
324 int dist = calcDist<T>(
325 main_extended_src_.at<T>(
326 border_size_ + i + ty, border_size_ + j + tx),
327 cur_extended_src.at<T>(
328 border_size_ + start_y + ty, border_size_ + start_x + tx)
329 );
330
331 *dist_sums_ptr += dist;
332 *col_dist_sums_ptr += dist;
333 }
334 col_dist_sums_ptr += col_dist_sums_step;
335 }
336
337 up_col_dist_sums[j][d][y][x] = col_dist_sums[template_window_size_ - 1][d][y][x];
338 }
339 }
340 }
341}

343template <class T>
344inline void FastNlMeansMultiDenoisingInvoker<T>::calcDistSumsForElementInFirstRow(
345 int i,
346 int j,
347 int first_col_num,
348 Array3d<int>& dist_sums,
349 Array4d<int>& col_dist_sums,
350 Array4d<int>& up_col_dist_sums) const
351{
352 int ay = border_size_ + i;
353 int ax = border_size_ + j + template_window_half_size_;
354
355 int start_by = border_size_ + i - search_window_half_size_;
356 int start_bx = border_size_ + j - search_window_half_size_ + template_window_half_size_;
357
358 int new_last_col_num = first_col_num;
359
360 for (int d = 0; d < temporal_window_size_; d++) {
361 Mat cur_extended_src = extended_srcs_[d];
362 for (int y = 0; y < search_window_size_; y++) {
363 for (int x = 0; x < search_window_size_; x++) {
364 dist_sums[d][y][x] -= col_dist_sums[first_col_num][d][y][x];
365
366 col_dist_sums[new_last_col_num][d][y][x] = 0;
367 int by = start_by + y;
368 int bx = start_bx + x;
369
370 int* col_dist_sums_ptr = &col_dist_sums[new_last_col_num][d][y][x];
371 for (int ty = -template_window_half_size_; ty <= template_window_half_size_; ty++) {
372 *col_dist_sums_ptr +=
373 calcDist<T>(
374 main_extended_src_.at<T>(ay + ty, ax),
375 cur_extended_src.at<T>(by + ty, bx)
376 );
377 }
378
379 dist_sums[d][y][x] += col_dist_sums[new_last_col_num][d][y][x];
380
381 up_col_dist_sums[j][d][y][x] = col_dist_sums[new_last_col_num][d][y][x];
382 }
383 }
384 }
