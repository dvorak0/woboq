99inline int getNearestPowerOf2(int value)
100{
101 int p = 0;
102 while( 1 << p < value) ++p;
103 return p;
104}

106template <class T>
107FastNlMeansDenoisingInvoker<T>::FastNlMeansDenoisingInvoker(
108 const cv::Mat& src,
109 cv::Mat& dst,
110 int template_window_size,
111 int search_window_size,
112 const float h) : src_(src), dst_(dst)
113{
114 CV_Assert(src.channels() == sizeof(T)); //T is Vec1b or Vec2b or Vec3b
115
116 template_window_half_size_ = template_window_size / 2;
117 search_window_half_size_ = search_window_size / 2;
118 template_window_size_ = template_window_half_size_ * 2 + 1;
119 search_window_size_ = search_window_half_size_ * 2 + 1;
120
121 border_size_ = search_window_half_size_ + template_window_half_size_;
122 copyMakeBorder(src_, extended_src_,
123 border_size_, border_size_, border_size_, border_size_, cv::BORDER_DEFAULT);
124
125 const int max_estimate_sum_value = search_window_size_ * search_window_size_ * 255;
126 fixed_point_mult_ = numeric_limits<int>::max() / max_estimate_sum_value;
127
128 // precalc weight for every possible l2 dist between blocks
129 // additional optimization of precalced weights to replace division(averaging) by binary shift
130
131 CV_Assert(template_window_size_ <= 46340 ); // sqrt(INT_MAX)
132 int template_window_size_sq = template_window_size_ * template_window_size_;
133 almost_template_window_size_sq_bin_shift_ = getNearestPowerOf2(template_window_size_sq);
134 double almost_dist2actual_dist_multiplier = ((double)(1 << almost_template_window_size_sq_bin_shift_)) / template_window_size_sq;
135
136 int max_dist = 255 * 255 * sizeof(T);
137 int almost_max_dist = (int) (max_dist / almost_dist2actual_dist_multiplier + 1);
138 almost_dist2weight_.resize(almost_max_dist);
139
140 const double WEIGHT_THRESHOLD = 0.001;
141 for (int almost_dist = 0; almost_dist < almost_max_dist; almost_dist++) {
142 double dist = almost_dist * almost_dist2actual_dist_multiplier;
143 int weight = cvRound(fixed_point_mult_ * std::exp(-dist / (h * h * sizeof(T))));
144
145 if (weight < WEIGHT_THRESHOLD * fixed_point_mult_)
146 weight = 0;
147
148 almost_dist2weight_[almost_dist] = weight;
149 }
150 CV_Assert(almost_dist2weight_[0] == fixed_point_mult_);
151 // additional optimization init end
152
153 if (dst_.empty()) {
154 dst_ = Mat::zeros(src_.size(), src_.type());
155 }
156}

158template <class T>
159void FastNlMeansDenoisingInvoker<T>::operator() (const Range& range) const {
160 int row_from = range.start;
161 int row_to = range.end - 1;
162
163 Array2d<int> dist_sums(search_window_size_, search_window_size_);
164
165 // for lazy calc optimization
166 Array3d<int> col_dist_sums(template_window_size_, search_window_size_, search_window_size_);
167
168 int first_col_num = -1;
169 Array3d<int> up_col_dist_sums(src_.cols, search_window_size_, search_window_size_);
170
171 for (int i = row_from; i <= row_to; i++) {
172 for (int j = 0; j < src_.cols; j++) {
173 int search_window_y = i - search_window_half_size_;
174 int search_window_x = j - search_window_half_size_;
175
176 // calc dist_sums
177 if (j == 0) {
178 calcDistSumsForFirstElementInRow(i, dist_sums, col_dist_sums, up_col_dist_sums);
179 first_col_num = 0;
180
181 } else { // calc cur dist_sums using previous dist_sums
182 if (i == row_from) {
183 calcDistSumsForElementInFirstRow(i, j, first_col_num,
184 dist_sums, col_dist_sums, up_col_dist_sums);
185
186 } else {
187 int ay = border_size_ + i;
188 int ax = border_size_ + j + template_window_half_size_;
189
190 int start_by =
191 border_size_ + i - search_window_half_size_;
192
193 int start_bx =
194 border_size_ + j - search_window_half_size_ + template_window_half_size_;
195
196 T a_up = extended_src_.at<T>(ay - template_window_half_size_ - 1, ax);
197 T a_down = extended_src_.at<T>(ay + template_window_half_size_, ax);
198
199 // copy class member to local variable for optimization
200 int search_window_size = search_window_size_;
201
202 for (int y = 0; y < search_window_size; y++) {
203 int* dist_sums_row = dist_sums.row_ptr(y);
204
205 int* col_dist_sums_row = col_dist_sums.row_ptr(first_col_num,y);
206
207 int* up_col_dist_sums_row = up_col_dist_sums.row_ptr(j, y);
208
209 const T* b_up_ptr =
210 extended_src_.ptr<T>(start_by - template_window_half_size_ - 1 + y);
211
212 const T* b_down_ptr =
213 extended_src_.ptr<T>(start_by + template_window_half_size_ + y);
214
215 for (int x = 0; x < search_window_size; x++) {
216 dist_sums_row[x] -= col_dist_sums_row[x];
217
218 col_dist_sums_row[x] =
219 up_col_dist_sums_row[x] +
220 calcUpDownDist(
221 a_up, a_down,
222 b_up_ptr[start_bx + x], b_down_ptr[start_bx + x]
223 );
224
225 dist_sums_row[x] += col_dist_sums_row[x];
226
227 up_col_dist_sums_row[x] = col_dist_sums_row[x];
228
229 }
230 }
231 }
232
233 first_col_num = (first_col_num + 1) % template_window_size_;
234 }
235
236 // calc weights
237 int weights_sum = 0;
238
239 int estimation[3];
240 for (size_t channel_num = 0; channel_num < sizeof(T); channel_num++) {
241 estimation[channel_num] = 0;
242 }
243
244 for (int y = 0; y < search_window_size_; y++) {
245 const T* cur_row_ptr = extended_src_.ptr<T>(border_size_ + search_window_y + y);
246 int* dist_sums_row = dist_sums.row_ptr(y);
247 for (int x = 0; x < search_window_size_; x++) {
248 int almostAvgDist =
249 dist_sums_row[x] >> almost_template_window_size_sq_bin_shift_;
250
251 int weight = almost_dist2weight_[almostAvgDist];
252 weights_sum += weight;
253
254 T p = cur_row_ptr[border_size_ + search_window_x + x];
255 incWithWeight(estimation, weight, p);
256 }
257 }
258
259 for (size_t channel_num = 0; channel_num < sizeof(T); channel_num++)
260 estimation[channel_num] = ((unsigned)estimation[channel_num] + weights_sum/2) / weights_sum;
261
262 dst_.at<T>(i,j) = saturateCastFromArray<T>(estimation);
263 }
264 }
265}

267template <class T>
268inline void FastNlMeansDenoisingInvoker<T>::calcDistSumsForFirstElementInRow(
269 int i,
270 Array2d<int>& dist_sums,
271 Array3d<int>& col_dist_sums,
272 Array3d<int>& up_col_dist_sums) const
273{
274 int j = 0;
275
276 for (int y = 0; y < search_window_size_; y++) {
277 for (int x = 0; x < search_window_size_; x++) {
278 dist_sums[y][x] = 0;
279 for (int tx = 0; tx < template_window_size_; tx++) {
280 col_dist_sums[tx][y][x] = 0;
281 }
282
283 int start_y = i + y - search_window_half_size_;
284 int start_x = j + x - search_window_half_size_;
285
286 for (int ty = -template_window_half_size_; ty <= template_window_half_size_; ty++) {
287 for (int tx = -template_window_half_size_; tx <= template_window_half_size_; tx++) {
288 int dist = calcDist<T>(extended_src_,
289 border_size_ + i + ty, border_size_ + j + tx,
290 border_size_ + start_y + ty, border_size_ + start_x + tx);
291
292 dist_sums[y][x] += dist;
293 col_dist_sums[tx + template_window_half_size_][y][x] += dist;
294 }
295 }
296
297 up_col_dist_sums[j][y][x] = col_dist_sums[template_window_size_ - 1][y][x];
298 }
299 }
300}

302template <class T>
303inline void FastNlMeansDenoisingInvoker<T>::calcDistSumsForElementInFirstRow(
304 int i,
305 int j,
306 int first_col_num,
307 Array2d<int>& dist_sums,
308 Array3d<int>& col_dist_sums,
309 Array3d<int>& up_col_dist_sums) const
310{
311 int ay = border_size_ + i;
312 int ax = border_size_ + j + template_window_half_size_;
313
314 int start_by = border_size_ + i - search_window_half_size_;
315 int start_bx = border_size_ + j - search_window_half_size_ + template_window_half_size_;
316
317 int new_last_col_num = first_col_num;
318
319 for (int y = 0; y < search_window_size_; y++) {
320 for (int x = 0; x < search_window_size_; x++) {
321 dist_sums[y][x] -= col_dist_sums[first_col_num][y][x];
322
323 col_dist_sums[new_last_col_num][y][x] = 0;
324 int by = start_by + y;
325 int bx = start_bx + x;
326 for (int ty = -template_window_half_size_; ty <= template_window_half_size_; ty++) {
327 col_dist_sums[new_last_col_num][y][x] +=
328 calcDist<T>(extended_src_, ay + ty, ax, by + ty, bx);
329 }
330
331 dist_sums[y][x] += col_dist_sums[new_last_col_num][y][x];
332
333 up_col_dist_sums[j][y][x] = col_dist_sums[new_last_col_num][y][x];
334 }
335 }
