52template <typename T> static inline int calcDist(const T a, const T b);
53
54template <> inline int calcDist(const uchar a, const uchar b) {
55 return (a-b) * (a-b);
56}
57
58template <> inline int calcDist(const Vec2b a, const Vec2b b) {
59 return (a[0]-b[0])*(a[0]-b[0]) + (a[1]-b[1])*(a[1]-b[1]);
60}
61
62template <> inline int calcDist(const Vec3b a, const Vec3b b) {
63 return (a[0]-b[0])*(a[0]-b[0]) + (a[1]-b[1])*(a[1]-b[1]) + (a[2]-b[2])*(a[2]-b[2]);
64}
65
66template <typename T> static inline int calcDist(const Mat& m, int i1, int j1, int i2, int j2) {
67 const T a = m.at<T>(i1, j1);
68 const T b = m.at<T>(i2, j2);
69 return calcDist<T>(a,b);
70}
71
72template <typename T> static inline int calcUpDownDist(T a_up, T a_down, T b_up, T b_down) {
73 return calcDist(a_down,b_down) - calcDist(a_up, b_up);
74}
75
76template <> inline int calcUpDownDist(uchar a_up, uchar a_down, uchar b_up, uchar b_down) {
77 int A = a_down - b_down;
78 int B = a_up - b_up;
79 return (A-B)*(A+B);
80}

82template <typename T> static inline void incWithWeight(int* estimation, int weight, T p);
83
84template <> inline void incWithWeight(int* estimation, int weight, uchar p) {
85 estimation[0] += weight * p;
86}
87
88template <> inline void incWithWeight(int* estimation, int weight, Vec2b p) {
89 estimation[0] += weight * p[0];
90 estimation[1] += weight * p[1];
91}
92
93template <> inline void incWithWeight(int* estimation, int weight, Vec3b p) {
94 estimation[0] += weight * p[0];
95 estimation[1] += weight * p[1];
96 estimation[2] += weight * p[2];
97}
98
99template <typename T> static inline T saturateCastFromArray(int* estimation);
100
101template <> inline uchar saturateCastFromArray(int* estimation) {
102 return saturate_cast<uchar>(estimation[0]);
103}
104
105template <> inline Vec2b saturateCastFromArray(int* estimation) {
106 Vec2b res;
107 res[0] = saturate_cast<uchar>(estimation[0]);
108 res[1] = saturate_cast<uchar>(estimation[1]);
109 return res;
110}
111
112template <> inline Vec3b saturateCastFromArray(int* estimation) {
113 Vec3b res;
114 res[0] = saturate_cast<uchar>(estimation[0]);
115 res[1] = saturate_cast<uchar>(estimation[1]);
116 res[2] = saturate_cast<uchar>(estimation[2]);
117 return res;
118}
