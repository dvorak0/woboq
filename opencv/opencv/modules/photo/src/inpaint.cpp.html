55inline float
56min4( float a, float b, float c, float d )
57{
58 a = MIN(a,b);
59 c = MIN(c,d);
60 return MIN(a,c);

69typedef struct CvHeapElem
70{
71 float T;
72 int i,j;
73 struct CvHeapElem* prev;
74 struct CvHeapElem* next;
75}
76CvHeapElem;

79class CvPriorityQueueFloat
80{
81protected:
82 CvHeapElem *mem,*empty,*head,*tail;
83 int num,in;
84
85public:
86 bool Init( const CvMat* f )
87 {
88 int i,j;
89 for( i = num = 0; i < f->rows; i++ )
90 {
91 for( j = 0; j < f->cols; j++ )
92 num += CV_MAT_ELEM(*f,uchar,i,j)!=0;
93 }
94 if (num<=0) return false;
95 mem = (CvHeapElem*)cvAlloc((num+2)*sizeof(CvHeapElem));
96 if (mem==NULL) return false;
97
98 head = mem;
99 head->i = head->j = -1;
100 head->prev = NULL;
101 head->next = mem+1;
102 head->T = -FLT_MAX;
103 empty = mem+1;
104 for (i=1; i<=num; i++) {
105 mem[i].prev = mem+i-1;
106 mem[i].next = mem+i+1;
107 mem[i].i = -1;
108 mem[i].T = FLT_MAX;
109 }
110 tail = mem+i;
111 tail->i = tail->j = -1;
112 tail->prev = mem+i-1;
113 tail->next = NULL;
114 tail->T = FLT_MAX;
115 return true;
116 }
117
118 bool Add(const CvMat* f) {
119 int i,j;
120 for (i=0; i<f->rows; i++) {
121 for (j=0; j<f->cols; j++) {
122 if (CV_MAT_ELEM(*f,uchar,i,j)!=0) {
123 if (!Push(i,j,0)) return false;
124 }
125 }
126 }
127 return true;
128 }
129
130 bool Push(int i, int j, float T) {
131 CvHeapElem *tmp=empty,*add=empty;
132 if (empty==tail) return false;
133 while (tmp->prev->T>T) tmp = tmp->prev;
134 if (tmp!=empty) {
135 add->prev->next = add->next;
136 add->next->prev = add->prev;
137 empty = add->next;
138 add->prev = tmp->prev;
139 add->next = tmp;
140 add->prev->next = add;
141 add->next->prev = add;
142 } else {
143 empty = empty->next;
144 }
145 add->i = i;
146 add->j = j;
147 add->T = T;
148 in++;
149 // printf("push i %3d j %3d T %12.4e in %4d\n",i,j,T,in);
150 return true;
151 }
152
153 bool Pop(int *i, int *j) {
154 CvHeapElem *tmp=head->next;
155 if (empty==tmp) return false;
156 *i = tmp->i;
157 *j = tmp->j;
158 tmp->prev->next = tmp->next;
159 tmp->next->prev = tmp->prev;
160 tmp->prev = empty->prev;
161 tmp->next = empty;
162 tmp->prev->next = tmp;
163 tmp->next->prev = tmp;
164 empty = tmp;
165 in--;
166 // printf("pop i %3d j %3d T %12.4e in %4d\n",tmp->i,tmp->j,tmp->T,in);
167 return true;
168 }
169
170 bool Pop(int *i, int *j, float *T) {
171 CvHeapElem *tmp=head->next;
172 if (empty==tmp) return false;
173 *i = tmp->i;
174 *j = tmp->j;
175 *T = tmp->T;
176 tmp->prev->next = tmp->next;
177 tmp->next->prev = tmp->prev;
178 tmp->prev = empty->prev;
179 tmp->next = empty;
180 tmp->prev->next = tmp;
181 tmp->next->prev = tmp;
182 empty = tmp;
183 in--;
184 // printf("pop i %3d j %3d T %12.4e in %4d\n",tmp->i,tmp->j,tmp->T,in);
185 return true;
186 }
187
188 CvPriorityQueueFloat(void) {
189 num=in=0;
190 mem=empty=head=tail=NULL;
191 }
192
193 ~CvPriorityQueueFloat(void)
194 {
195 cvFree( &mem );
196 }
197};

211static float FastMarching_solve(int i1,int j1,int i2,int j2, const CvMat* f, const CvMat* t)
212{
213 double sol, a11, a22, m12;
214 a11=CV_MAT_ELEM(*t,float,i1,j1);
215 a22=CV_MAT_ELEM(*t,float,i2,j2);
216 m12=MIN(a11,a22);
217
218 if( CV_MAT_ELEM(*f,uchar,i1,j1) != INSIDE )
219 if( CV_MAT_ELEM(*f,uchar,i2,j2) != INSIDE )
220 if( fabs(a11-a22) >= 1.0 )
221 sol = 1+m12;
222 else
223 sol = (a11+a22+sqrt((double)(2-(a11-a22)*(a11-a22))))*0.5;
224 else
225 sol = 1+a11;
226 else if( CV_MAT_ELEM(*f,uchar,i2,j2) != INSIDE )
227 sol = 1+a22;
228 else
229 sol = 1+m12;
230
231 return (float)sol;
232}

237static void
238icvCalcFMM(const CvMat *f, CvMat *t, CvPriorityQueueFloat *Heap, bool negate) {
239 int i, j, ii = 0, jj = 0, q;
240 float dist;
241
242 while (Heap->Pop(&ii,&jj)) {
243
244 unsigned known=(negate)?CHANGE:KNOWN;
245 CV_MAT_ELEM(*f,uchar,ii,jj) = (uchar)known;
246
247 for (q=0; q<4; q++) {
248 i=0; j=0;
249 if (q==0) {i=ii-1; j=jj;}
250 else if(q==1) {i=ii; j=jj-1;}
251 else if(q==2) {i=ii+1; j=jj;}
252 else {i=ii; j=jj+1;}
253 if ((i<=0)||(j<=0)||(i>f->rows)||(j>f->cols)) continue;
254
255 if (CV_MAT_ELEM(*f,uchar,i,j)==INSIDE) {
256 dist = min4(FastMarching_solve(i-1,j,i,j-1,f,t),
257 FastMarching_solve(i+1,j,i,j-1,f,t),
258 FastMarching_solve(i-1,j,i,j+1,f,t),
259 FastMarching_solve(i+1,j,i,j+1,f,t));
260 CV_MAT_ELEM(*t,float,i,j) = dist;
261 CV_MAT_ELEM(*f,uchar,i,j) = BAND;
262 Heap->Push(i,j,dist);
263 }
264 }
265 }
266
267 if (negate) {
268 for (i=0; i<f->rows; i++) {
269 for(j=0; j<f->cols; j++) {
270 if (CV_MAT_ELEM(*f,uchar,i,j) == CHANGE) {
271 CV_MAT_ELEM(*f,uchar,i,j) = KNOWN;
272 CV_MAT_ELEM(*t,float,i,j) = -CV_MAT_ELEM(*t,float,i,j);
273 }
274 }
275 }
276 }
277}

280static void
281icvTeleaInpaintFMM(const CvMat *f, CvMat *t, CvMat *out, int range, CvPriorityQueueFloat *Heap ) {
282 int i = 0, j = 0, ii = 0, jj = 0, k, l, q, color = 0;
283 float dist;
284
285 if (CV_MAT_CN(out->type)==3) {
286
287 while (Heap->Pop(&ii,&jj)) {
288
289 CV_MAT_ELEM(*f,uchar,ii,jj) = KNOWN;
290 for(q=0; q<4; q++) {
291 if (q==0) {i=ii-1; j=jj;}
292 else if(q==1) {i=ii; j=jj-1;}
293 else if(q==2) {i=ii+1; j=jj;}
294 else if(q==3) {i=ii; j=jj+1;}
295 if ((i<=1)||(j<=1)||(i>t->rows-1)||(j>t->cols-1)) continue;
296
297 if (CV_MAT_ELEM(*f,uchar,i,j)==INSIDE) {
298 dist = min4(FastMarching_solve(i-1,j,i,j-1,f,t),
299 FastMarching_solve(i+1,j,i,j-1,f,t),
300 FastMarching_solve(i-1,j,i,j+1,f,t),
301 FastMarching_solve(i+1,j,i,j+1,f,t));
302 CV_MAT_ELEM(*t,float,i,j) = dist;
303
304 for (color=0; color<=2; color++) {
305 CvPoint2D32f gradI,gradT,r;
306 float Ia=0,Jx=0,Jy=0,s=1.0e-20f,w,dst,lev,dir,sat;
307
308 if (CV_MAT_ELEM(*f,uchar,i,j+1)!=INSIDE) {
309 if (CV_MAT_ELEM(*f,uchar,i,j-1)!=INSIDE) {
310 gradT.x=(float)((CV_MAT_ELEM(*t,float,i,j+1)-CV_MAT_ELEM(*t,float,i,j-1)))*0.5f;
311 } else {
312 gradT.x=(float)((CV_MAT_ELEM(*t,float,i,j+1)-CV_MAT_ELEM(*t,float,i,j)));
313 }
314 } else {
315 if (CV_MAT_ELEM(*f,uchar,i,j-1)!=INSIDE) {
316 gradT.x=(float)((CV_MAT_ELEM(*t,float,i,j)-CV_MAT_ELEM(*t,float,i,j-1)));
317 } else {
318 gradT.x=0;
319 }
320 }
321 if (CV_MAT_ELEM(*f,uchar,i+1,j)!=INSIDE) {
322 if (CV_MAT_ELEM(*f,uchar,i-1,j)!=INSIDE) {
323 gradT.y=(float)((CV_MAT_ELEM(*t,float,i+1,j)-CV_MAT_ELEM(*t,float,i-1,j)))*0.5f;
324 } else {
325 gradT.y=(float)((CV_MAT_ELEM(*t,float,i+1,j)-CV_MAT_ELEM(*t,float,i,j)));
326 }
327 } else {
328 if (CV_MAT_ELEM(*f,uchar,i-1,j)!=INSIDE) {
329 gradT.y=(float)((CV_MAT_ELEM(*t,float,i,j)-CV_MAT_ELEM(*t,float,i-1,j)));
330 } else {
331 gradT.y=0;
332 }
333 }
334 for (k=i-range; k<=i+range; k++) {
335 int km=k-1+(k==1),kp=k-1-(k==t->rows-2);
336 for (l=j-range; l<=j+range; l++) {
337 int lm=l-1+(l==1),lp=l-1-(l==t->cols-2);
338 if (k>0&&l>0&&k<t->rows-1&&l<t->cols-1) {
339 if ((CV_MAT_ELEM(*f,uchar,k,l)!=INSIDE)&&
340 ((l-j)*(l-j)+(k-i)*(k-i)<=range*range)) {
341 r.y = (float)(i-k);
342 r.x = (float)(j-l);
343
344 dst = (float)(1./(VectorLength(r)*sqrt((double)VectorLength(r))));
345 lev = (float)(1./(1+fabs(CV_MAT_ELEM(*t,float,k,l)-CV_MAT_ELEM(*t,float,i,j))));
346
347 dir=VectorScalMult(r,gradT);
348 if (fabs(dir)<=0.01) dir=0.000001f;
349 w = (float)fabs(dst*lev*dir);
350
351 if (CV_MAT_ELEM(*f,uchar,k,l+1)!=INSIDE) {
352 if (CV_MAT_ELEM(*f,uchar,k,l-1)!=INSIDE) {
353 gradI.x=(float)((CV_MAT_3COLOR_ELEM(*out,uchar,km,lp+1,color)-CV_MAT_3COLOR_ELEM(*out,uchar,km,lm-1,color)))*2.0f;
354 } else {
355 gradI.x=(float)((CV_MAT_3COLOR_ELEM(*out,uchar,km,lp+1,color)-CV_MAT_3COLOR_ELEM(*out,uchar,km,lm,color)));
356 }
357 } else {
358 if (CV_MAT_ELEM(*f,uchar,k,l-1)!=INSIDE) {
359 gradI.x=(float)((CV_MAT_3COLOR_ELEM(*out,uchar,km,lp,color)-CV_MAT_3COLOR_ELEM(*out,uchar,km,lm-1,color)));
360 } else {
361 gradI.x=0;
362 }
363 }
364 if (CV_MAT_ELEM(*f,uchar,k+1,l)!=INSIDE) {
365 if (CV_MAT_ELEM(*f,uchar,k-1,l)!=INSIDE) {
366 gradI.y=(float)((CV_MAT_3COLOR_ELEM(*out,uchar,kp+1,lm,color)-CV_MAT_3COLOR_ELEM(*out,uchar,km-1,lm,color)))*2.0f;
367 } else {
368 gradI.y=(float)((CV_MAT_3COLOR_ELEM(*out,uchar,kp+1,lm,color)-CV_MAT_3COLOR_ELEM(*out,uchar,km,lm,color)));
369 }
370 } else {
371 if (CV_MAT_ELEM(*f,uchar,k-1,l)!=INSIDE) {
372 gradI.y=(float)((CV_MAT_3COLOR_ELEM(*out,uchar,kp,lm,color)-CV_MAT_3COLOR_ELEM(*out,uchar,km-1,lm,color)));
373 } else {
374 gradI.y=0;
375 }
376 }
377 Ia += (float)w * (float)(CV_MAT_3COLOR_ELEM(*out,uchar,km,lm,color));
378 Jx -= (float)w * (float)(gradI.x*r.x);
379 Jy -= (float)w * (float)(gradI.y*r.y);
380 s += w;
381 }
382 }
383 }
384 }
385 sat = (float)((Ia/s+(Jx+Jy)/(sqrt(Jx*Jx+Jy*Jy)+1.0e-20f)+0.5f));
386 {
387 CV_MAT_3COLOR_ELEM(*out,uchar,i-1,j-1,color) = cv::saturate_cast<uchar>(sat);
388 }
389 }
390
391 CV_MAT_ELEM(*f,uchar,i,j) = BAND;
392 Heap->Push(i,j,dist);
393 }
394 }
395 }
396
397 } else if (CV_MAT_CN(out->type)==1) {
398
399 while (Heap->Pop(&ii,&jj)) {
400
401 CV_MAT_ELEM(*f,uchar,ii,jj) = KNOWN;
402 for(q=0; q<4; q++) {
403 if (q==0) {i=ii-1; j=jj;}
404 else if(q==1) {i=ii; j=jj-1;}
405 else if(q==2) {i=ii+1; j=jj;}
406 else if(q==3) {i=ii; j=jj+1;}
407 if ((i<=1)||(j<=1)||(i>t->rows-1)||(j>t->cols-1)) continue;
408
409 if (CV_MAT_ELEM(*f,uchar,i,j)==INSIDE) {
410 dist = min4(FastMarching_solve(i-1,j,i,j-1,f,t),
411 FastMarching_solve(i+1,j,i,j-1,f,t),
412 FastMarching_solve(i-1,j,i,j+1,f,t),
413 FastMarching_solve(i+1,j,i,j+1,f,t));
414 CV_MAT_ELEM(*t,float,i,j) = dist;
415
416 for (color=0; color<=0; color++) {
417 CvPoint2D32f gradI,gradT,r;
418 float Ia=0,Jx=0,Jy=0,s=1.0e-20f,w,dst,lev,dir,sat;
419
420 if (CV_MAT_ELEM(*f,uchar,i,j+1)!=INSIDE) {
421 if (CV_MAT_ELEM(*f,uchar,i,j-1)!=INSIDE) {
422 gradT.x=(float)((CV_MAT_ELEM(*t,float,i,j+1)-CV_MAT_ELEM(*t,float,i,j-1)))*0.5f;
423 } else {
424 gradT.x=(float)((CV_MAT_ELEM(*t,float,i,j+1)-CV_MAT_ELEM(*t,float,i,j)));
425 }
426 } else {
427 if (CV_MAT_ELEM(*f,uchar,i,j-1)!=INSIDE) {
428 gradT.x=(float)((CV_MAT_ELEM(*t,float,i,j)-CV_MAT_ELEM(*t,float,i,j-1)));
429 } else {
430 gradT.x=0;
431 }
432 }
433 if (CV_MAT_ELEM(*f,uchar,i+1,j)!=INSIDE) {
434 if (CV_MAT_ELEM(*f,uchar,i-1,j)!=INSIDE) {
435 gradT.y=(float)((CV_MAT_ELEM(*t,float,i+1,j)-CV_MAT_ELEM(*t,float,i-1,j)))*0.5f;
436 } else {
437 gradT.y=(float)((CV_MAT_ELEM(*t,float,i+1,j)-CV_MAT_ELEM(*t,float,i,j)));
438 }
439 } else {
440 if (CV_MAT_ELEM(*f,uchar,i-1,j)!=INSIDE) {
441 gradT.y=(float)((CV_MAT_ELEM(*t,float,i,j)-CV_MAT_ELEM(*t,float,i-1,j)));
442 } else {
443 gradT.y=0;
444 }
445 }
446 for (k=i-range; k<=i+range; k++) {
447 int km=k-1+(k==1),kp=k-1-(k==t->rows-2);
448 for (l=j-range; l<=j+range; l++) {
449 int lm=l-1+(l==1),lp=l-1-(l==t->cols-2);
450 if (k>0&&l>0&&k<t->rows-1&&l<t->cols-1) {
451 if ((CV_MAT_ELEM(*f,uchar,k,l)!=INSIDE)&&
452 ((l-j)*(l-j)+(k-i)*(k-i)<=range*range)) {
453 r.y = (float)(i-k);
454 r.x = (float)(j-l);
455
456 dst = (float)(1./(VectorLength(r)*sqrt(VectorLength(r))));
457 lev = (float)(1./(1+fabs(CV_MAT_ELEM(*t,float,k,l)-CV_MAT_ELEM(*t,float,i,j))));
458
459 dir=VectorScalMult(r,gradT);
460 if (fabs(dir)<=0.01) dir=0.000001f;
461 w = (float)fabs(dst*lev*dir);
462
463 if (CV_MAT_ELEM(*f,uchar,k,l+1)!=INSIDE) {
464 if (CV_MAT_ELEM(*f,uchar,k,l-1)!=INSIDE) {
465 gradI.x=(float)((CV_MAT_ELEM(*out,uchar,km,lp+1)-CV_MAT_ELEM(*out,uchar,km,lm-1)))*2.0f;
466 } else {
467 gradI.x=(float)((CV_MAT_ELEM(*out,uchar,km,lp+1)-CV_MAT_ELEM(*out,uchar,km,lm)));
468 }
469 } else {
470 if (CV_MAT_ELEM(*f,uchar,k,l-1)!=INSIDE) {
471 gradI.x=(float)((CV_MAT_ELEM(*out,uchar,km,lp)-CV_MAT_ELEM(*out,uchar,km,lm-1)));
472 } else {
473 gradI.x=0;
474 }
475 }
476 if (CV_MAT_ELEM(*f,uchar,k+1,l)!=INSIDE) {
477 if (CV_MAT_ELEM(*f,uchar,k-1,l)!=INSIDE) {
478 gradI.y=(float)((CV_MAT_ELEM(*out,uchar,kp+1,lm)-CV_MAT_ELEM(*out,uchar,km-1,lm)))*2.0f;
479 } else {
480 gradI.y=(float)((CV_MAT_ELEM(*out,uchar,kp+1,lm)-CV_MAT_ELEM(*out,uchar,km,lm)));
481 }
482 } else {
483 if (CV_MAT_ELEM(*f,uchar,k-1,l)!=INSIDE) {
484 gradI.y=(float)((CV_MAT_ELEM(*out,uchar,kp,lm)-CV_MAT_ELEM(*out,uchar,km-1,lm)));
485 } else {
486 gradI.y=0;
487 }
488 }
489 Ia += (float)w * (float)(CV_MAT_ELEM(*out,uchar,km,lm));
490 Jx -= (float)w * (float)(gradI.x*r.x);
491 Jy -= (float)w * (float)(gradI.y*r.y);
492 s += w;
493 }
494 }
495 }
496 }
497 sat = (float)((Ia/s+(Jx+Jy)/(sqrt(Jx*Jx+Jy*Jy)+1.0e-20f)+0.5f));
498 {
499 CV_MAT_ELEM(*out,uchar,i-1,j-1) = cv::saturate_cast<uchar>(sat);
500 }
501 }
502
503 CV_MAT_ELEM(*f,uchar,i,j) = BAND;
504 Heap->Push(i,j,dist);
505 }
506 }
507 }
508 }

542 for (l=j-range; l<=j+range; l++) {
543 int lm=l-1+(l==1),lp=l-1-(l==f->cols-2);
544 if (k>0&&l>0&&k<f->rows-1&&l<f->cols-1) {
545 if ((CV_MAT_ELEM(*f,uchar,k,l)!=INSIDE)&&
546 ((l-j)*(l-j)+(k-i)*(k-i)<=range*range)) {
547 r.y=(float)(k-i);
548 r.x=(float)(l-j);
549
550 dst = 1/(VectorLength(r)*VectorLength(r)+1);
551
552 if (CV_MAT_ELEM(*f,uchar,k+1,l)!=INSIDE) {
553 if (CV_MAT_ELEM(*f,uchar,k-1,l)!=INSIDE) {
554 gradI.x=(float)(abs(CV_MAT_3COLOR_ELEM(*out,uchar,kp+1,lm,color)-CV_MAT_3COLOR_ELEM(*out,uchar,kp,lm,color))+
555 abs(CV_MAT_3COLOR_ELEM(*out,uchar,kp,lm,color)-CV_MAT_3COLOR_ELEM(*out,uchar,km-1,lm,color)));
556 } else {
557 gradI.x=(float)(abs(CV_MAT_3COLOR_ELEM(*out,uchar,kp+1,lm,color)-CV_MAT_3COLOR_ELEM(*out,uchar,kp,lm,color)))*2.0f;
558 }
559 } else {
560 if (CV_MAT_ELEM(*f,uchar,k-1,l)!=INSIDE) {
561 gradI.x=(float)(abs(CV_MAT_3COLOR_ELEM(*out,uchar,kp,lm,color)-CV_MAT_3COLOR_ELEM(*out,uchar,km-1,lm,color)))*2.0f;
562 } else {
563 gradI.x=0;
564 }
565 }
566 if (CV_MAT_ELEM(*f,uchar,k,l+1)!=INSIDE) {
567 if (CV_MAT_ELEM(*f,uchar,k,l-1)!=INSIDE) {
568 gradI.y=(float)(abs(CV_MAT_3COLOR_ELEM(*out,uchar,km,lp+1,color)-CV_MAT_3COLOR_ELEM(*out,uchar,km,lm,color))+
569 abs(CV_MAT_3COLOR_ELEM(*out,uchar,km,lm,color)-CV_MAT_3COLOR_ELEM(*out,uchar,km,lm-1,color)));
570 } else {
571 gradI.y=(float)(abs(CV_MAT_3COLOR_ELEM(*out,uchar,km,lp+1,color)-CV_MAT_3COLOR_ELEM(*out,uchar,km,lm,color)))*2.0f;
572 }
573 } else {
574 if (CV_MAT_ELEM(*f,uchar,k,l-1)!=INSIDE) {
575 gradI.y=(float)(abs(CV_MAT_3COLOR_ELEM(*out,uchar,km,lm,color)-CV_MAT_3COLOR_ELEM(*out,uchar,km,lm-1,color)))*2.0f;
576 } else {
577 gradI.y=0;
578 }
579 }
580
581 gradI.x=-gradI.x;
582 dir=VectorScalMult(r,gradI);
583
584 if (fabs(dir)<=0.01) {
585 dir=0.000001f;
586 } else {
587 dir = (float)fabs(VectorScalMult(r,gradI)/sqrt(VectorLength(r)*VectorLength(gradI)));
588 }
589 w = dst*dir;
590 Ia += (float)w * (float)(CV_MAT_3COLOR_ELEM(*out,uchar,km,lm,color));
591 s += w;
592 }
593 }
594 }
595 }
596 CV_MAT_3COLOR_ELEM(*out,uchar,i-1,j-1,color) = cv::saturate_cast<uchar>((double)Ia/s);
597 }
598
599 CV_MAT_ELEM(*f,uchar,i,j) = BAND;
600 Heap->Push(i,j,dist);
601 }
602 }
603 }
604
605 } else if (CV_MAT_CN(out->type)==1) {
606
607 while (Heap->Pop(&ii,&jj)) {
608
609 CV_MAT_ELEM(*f,uchar,ii,jj) = KNOWN;
610 for(q=0; q<4; q++) {
611 if (q==0) {i=ii-1; j=jj;}
612 else if(q==1) {i=ii; j=jj-1;}
613 else if(q==2) {i=ii+1; j=jj;}
614 else if(q==3) {i=ii; j=jj+1;}
615 if ((i<=1)||(j<=1)||(i>t->rows-1)||(j>t->cols-1)) continue;
616
617 if (CV_MAT_ELEM(*f,uchar,i,j)==INSIDE) {
618 dist = min4(FastMarching_solve(i-1,j,i,j-1,f,t),
619 FastMarching_solve(i+1,j,i,j-1,f,t),
620 FastMarching_solve(i-1,j,i,j+1,f,t),
621 FastMarching_solve(i+1,j,i,j+1,f,t));
622 CV_MAT_ELEM(*t,float,i,j) = dist;
623
624 {
625 CvPoint2D32f gradI,r;
626 float Ia=0,s=1.0e-20f,w,dst,dir;
627
628 for (k=i-range; k<=i+range; k++) {
629 int km=k-1+(k==1),kp=k-1-(k==t->rows-2);
630 for (l=j-range; l<=j+range; l++) {
631 int lm=l-1+(l==1),lp=l-1-(l==t->cols-2);
632 if (k>0&&l>0&&k<t->rows-1&&l<t->cols-1) {
633 if ((CV_MAT_ELEM(*f,uchar,k,l)!=INSIDE)&&
634 ((l-j)*(l-j)+(k-i)*(k-i)<=range*range)) {
635 r.y=(float)(i-k);
636 r.x=(float)(j-l);
637
638 dst = 1/(VectorLength(r)*VectorLength(r)+1);
639
640 if (CV_MAT_ELEM(*f,uchar,k+1,l)!=INSIDE) {
641 if (CV_MAT_ELEM(*f,uchar,k-1,l)!=INSIDE) {
642 gradI.x=(float)(abs(CV_MAT_ELEM(*out,uchar,kp+1,lm)-CV_MAT_ELEM(*out,uchar,kp,lm))+
643 abs(CV_MAT_ELEM(*out,uchar,kp,lm)-CV_MAT_ELEM(*out,uchar,km-1,lm)));
644 } else {
645 gradI.x=(float)(abs(CV_MAT_ELEM(*out,uchar,kp+1,lm)-CV_MAT_ELEM(*out,uchar,kp,lm)))*2.0f;
646 }
647 } else {
648 if (CV_MAT_ELEM(*f,uchar,k-1,l)!=INSIDE) {
649 gradI.x=(float)(abs(CV_MAT_ELEM(*out,uchar,kp,lm)-CV_MAT_ELEM(*out,uchar,km-1,lm)))*2.0f;
650 } else {
651 gradI.x=0;
652 }
653 }
654 if (CV_MAT_ELEM(*f,uchar,k,l+1)!=INSIDE) {
655 if (CV_MAT_ELEM(*f,uchar,k,l-1)!=INSIDE) {
656 gradI.y=(float)(abs(CV_MAT_ELEM(*out,uchar,km,lp+1)-CV_MAT_ELEM(*out,uchar,km,lm))+
657 abs(CV_MAT_ELEM(*out,uchar,km,lm)-CV_MAT_ELEM(*out,uchar,km,lm-1)));
658 } else {
659 gradI.y=(float)(abs(CV_MAT_ELEM(*out,uchar,km,lp+1)-CV_MAT_ELEM(*out,uchar,km,lm)))*2.0f;
660 }
661 } else {
662 if (CV_MAT_ELEM(*f,uchar,k,l-1)!=INSIDE) {
663 gradI.y=(float)(abs(CV_MAT_ELEM(*out,uchar,km,lm)-CV_MAT_ELEM(*out,uchar,km,lm-1)))*2.0f;
664 } else {
665 gradI.y=0;
666 }
667 }
668
669 gradI.x=-gradI.x;
670 dir=VectorScalMult(r,gradI);
671
672 if (fabs(dir)<=0.01) {
673 dir=0.000001f;
674 } else {
675 dir = (float)fabs(VectorScalMult(r,gradI)/sqrt(VectorLength(r)*VectorLength(gradI)));
676 }
677 w = dst*dir;
678 Ia += (float)w * (float)(CV_MAT_ELEM(*out,uchar,km,lm));
679 s += w;
680 }
681 }
682 }
683 }
684 CV_MAT_ELEM(*out,uchar,i-1,j-1) = cv::saturate_cast<uchar>((double)Ia/s);
685 }
686
687 CV_MAT_ELEM(*f,uchar,i,j) = BAND;
688 Heap->Push(i,j,dist);
689 }
690 }
691 }
692
693 }

720void
721cvInpaint( const CvArr* _input_img, const CvArr* _inpaint_mask, CvArr* _output_img,
722 double inpaintRange, int flags )
723{
724 cv::Ptr<CvMat> mask, band, f, t, out;
725 cv::Ptr<CvPriorityQueueFloat> Heap, Out;
726 cv::Ptr<IplConvKernel> el_cross, el_range;
727
728 CvMat input_hdr, mask_hdr, output_hdr;
729 CvMat* input_img, *inpaint_mask, *output_img;
730 int range=cvRound(inpaintRange);
731 int erows, ecols;
732
733 input_img = cvGetMat( _input_img, &input_hdr );
734 inpaint_mask = cvGetMat( _inpaint_mask, &mask_hdr );
735 output_img = cvGetMat( _output_img, &output_hdr );
736
737 if( !CV_ARE_SIZES_EQ(input_img,output_img) || !CV_ARE_SIZES_EQ(input_img,inpaint_mask))
738 CV_Error( CV_StsUnmatchedSizes, "All the input and output images must have the same size" );
739
740 if( (CV_MAT_TYPE(input_img->type) != CV_8UC1 &&
741 CV_MAT_TYPE(input_img->type) != CV_8UC3) ||
742 !CV_ARE_TYPES_EQ(input_img,output_img) )
743 CV_Error( CV_StsUnsupportedFormat,
744 "Only 8-bit 1-channel and 3-channel input/output images are supported" );
745
746 if( CV_MAT_TYPE(inpaint_mask->type) != CV_8UC1 )
747 CV_Error( CV_StsUnsupportedFormat, "The mask must be 8-bit 1-channel image" );
748
749 range = MAX(range,1);
750 range = MIN(range,100);
751
752 ecols = input_img->cols + 2;
753 erows = input_img->rows + 2;
754
755 f = cvCreateMat(erows, ecols, CV_8UC1);
756 t = cvCreateMat(erows, ecols, CV_32FC1);
757 band = cvCreateMat(erows, ecols, CV_8UC1);
758 mask = cvCreateMat(erows, ecols, CV_8UC1);
759 el_cross = cvCreateStructuringElementEx(3,3,1,1,CV_SHAPE_CROSS,NULL);
760
761 cvCopy( input_img, output_img );
762 cvSet(mask,cvScalar(KNOWN,0,0,0));
763 COPY_MASK_BORDER1_C1(inpaint_mask,mask,uchar);
764 SET_BORDER1_C1(mask,uchar,0);
765 cvSet(f,cvScalar(KNOWN,0,0,0));
766 cvSet(t,cvScalar(1.0e6f,0,0,0));
767 cvDilate(mask,band,el_cross,1); // image with narrow band
768 Heap=new CvPriorityQueueFloat;
769 if (!Heap->Init(band))
770 return;
771 cvSub(band,mask,band,NULL);
772 SET_BORDER1_C1(band,uchar,0);
773 if (!Heap->Add(band))
774 return;
775 cvSet(f,cvScalar(BAND,0,0,0),band);
776 cvSet(f,cvScalar(INSIDE,0,0,0),mask);
777 cvSet(t,cvScalar(0,0,0,0),band);
778
779 if( flags == CV_INPAINT_TELEA )
780 {
781 out = cvCreateMat(erows, ecols, CV_8UC1);
782 el_range = cvCreateStructuringElementEx(2*range+1,2*range+1,
783 range,range,CV_SHAPE_RECT,NULL);
784 cvDilate(mask,out,el_range,1);
785 cvSub(out,mask,out,NULL);
786 Out=new CvPriorityQueueFloat;
787 if (!Out->Init(out))
788 return;
789 if (!Out->Add(band))
790 return;
791 cvSub(out,band,out,NULL);
792 SET_BORDER1_C1(out,uchar,0);
793 icvCalcFMM(out,t,Out,true);
794 icvTeleaInpaintFMM(mask,t,output_img,range,Heap);
795 }
796 else if (flags == CV_INPAINT_NS) {
797 icvNSInpaintFMM(mask,t,output_img,range,Heap);
798 } else {
799 CV_Error( CV_StsBadArg, "The flags argument must be one of CV_INPAINT_TELEA or CV_INPAINT_NS" );
800 }
801}

803void cv::inpaint( InputArray _src, InputArray _mask, OutputArray _dst,
804 double inpaintRange, int flags )
805{
806 Mat src = _src.getMat(), mask = _mask.getMat();
807 _dst.create( src.size(), src.type() );
808 CvMat c_src = src, c_mask = mask, c_dst = _dst.getMat();
809 cvInpaint( &c_src, &c_mask, &c_dst, inpaintRange, flags );
810}
