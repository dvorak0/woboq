58/*
59 * Intrinsic data types required by the file format:
60 *
61 * 8-bit quantities int8/uint8
62 * 16-bit quantities int16/uint16
63 * 32-bit quantities int32/uint32
64 * 64-bit quantities int64/uint64
65 * strings unsigned char*
66 */
67
68typedef TIFF_INT8_T int8;
69typedef TIFF_UINT8_T uint8;
70
71typedef TIFF_INT16_T int16;
72typedef TIFF_UINT16_T uint16;
73
74typedef TIFF_INT32_T int32;
75typedef TIFF_UINT32_T uint32;
76
77typedef TIFF_INT64_T int64;
78typedef TIFF_UINT64_T uint64;
79
80/*
81 * Some types as promoted in a variable argument list
82 * We use uint16_vap rather then directly using int, because this way
83 * we document the type we actually want to pass through, conceptually,
84 * rather then confusing the issue by merely stating the type it gets
85 * promoted to
86 */
87
88typedef int uint16_vap;

90/*
91 * TIFF header.
92 */
93typedef struct {
94 uint16 tiff_magic; /* magic number (defines byte order) */
95 uint16 tiff_version; /* TIFF version number */
96} TIFFHeaderCommon;
97typedef struct {
98 uint16 tiff_magic; /* magic number (defines byte order) */
99 uint16 tiff_version; /* TIFF version number */
100 uint32 tiff_diroff; /* byte offset to first directory */
101} TIFFHeaderClassic;
102typedef struct {
103 uint16 tiff_magic; /* magic number (defines byte order) */
104 uint16 tiff_version; /* TIFF version number */
105 uint16 tiff_offsetsize; /* size of offsets, should be 8 */
106 uint16 tiff_unused; /* unused word, should be 0 */
107 uint64 tiff_diroff; /* byte offset to first directory */
108} TIFFHeaderBig;

120/*
121 * Tag data type information.
122 *
123 * Note: RATIONALs are the ratio of two 32-bit integer values.
124 */
125typedef enum {
126 TIFF_NOTYPE = 0, /* placeholder */
127 TIFF_BYTE = 1, /* 8-bit unsigned integer */
128 TIFF_ASCII = 2, /* 8-bit bytes w/ last byte null */
129 TIFF_SHORT = 3, /* 16-bit unsigned integer */
130 TIFF_LONG = 4, /* 32-bit unsigned integer */
131 TIFF_RATIONAL = 5, /* 64-bit unsigned fraction */
132 TIFF_SBYTE = 6, /* !8-bit signed integer */
133 TIFF_UNDEFINED = 7, /* !8-bit untyped data */
134 TIFF_SSHORT = 8, /* !16-bit signed integer */
135 TIFF_SLONG = 9, /* !32-bit signed integer */
136 TIFF_SRATIONAL = 10, /* !64-bit signed fraction */
137 TIFF_FLOAT = 11, /* !32-bit IEEE floating point */
138 TIFF_DOUBLE = 12, /* !64-bit IEEE floating point */
139 TIFF_IFD = 13, /* %32-bit unsigned integer (offset) */
140 TIFF_LONG8 = 16, /* BigTIFF 64-bit unsigned integer */
141 TIFF_SLONG8 = 17, /* BigTIFF 64-bit signed integer */
142 TIFF_IFD8 = 18 /* BigTIFF 64-bit unsigned integer (offset) */
143} TIFFDataType;
