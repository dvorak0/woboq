332/**
333 * GDBusInterfaceVTable:
334 * @method_call: Function for handling incoming method calls.
335 * @get_property: Function for getting a property.
336 * @set_property: Function for setting a property.
337 *
338 * Virtual table for handling properties and method calls for a D-Bus
339 * interface.
340 *
341 * Since 2.38, if you want to handle getting/setting D-Bus properties
342 * asynchronously, give %NULL as your get_property() or set_property()
343 * function. The D-Bus call will be directed to your @method_call function,
344 * with the provided @interface_name set to "org.freedesktop.DBus.Properties".
345 *
346 * The usual checks on the validity of the calls is performed. For
347 * `Get` calls, an error is automatically returned if the property does
348 * not exist or the permissions do not allow access. The same checks are
349 * performed for `Set` calls, and the provided value is also checked for
350 * being the correct type.
351 *
352 * For both `Get` and `Set` calls, the #GDBusMethodInvocation
353 * passed to the @method_call handler can be queried with
354 * g_dbus_method_invocation_get_property_info() to get a pointer
355 * to the #GDBusPropertyInfo of the property.
356 *
357 * If you have readable properties specified in your interface info,
358 * you must ensure that you either provide a non-%NULL @get_property()
359 * function or provide implementations of both the `Get` and `GetAll`
360 * methods on org.freedesktop.DBus.Properties interface in your @method_call
361 * function. Note that the required return type of the `Get` call is
362 * `(v)`, not the type of the property. `GetAll` expects a return value
363 * of type `a{sv}`.
364 *
365 * If you have writable properties specified in your interface info,
366 * you must ensure that you either provide a non-%NULL @set_property()
367 * function or provide an implementation of the `Set` call. If implementing
368 * the call, you must return the value of type %G_VARIANT_TYPE_UNIT.
369 *
370 * Since: 2.26
371 */
372struct _GDBusInterfaceVTable
373{
374 GDBusInterfaceMethodCallFunc method_call;
375 GDBusInterfaceGetPropertyFunc get_property;
376 GDBusInterfaceSetPropertyFunc set_property;
377
378 /*< private >*/
379 /* Padding for future expansion - also remember to update
380 * gdbusconnection.c:_g_dbus_interface_vtable_copy() when
381 * changing this.
382 */
383 gpointer padding[8];
384};

428/**
429 * GDBusSubtreeIntrospectFunc:
430 * @connection: A #GDBusConnection.
431 * @sender: The unique bus name of the remote caller.
432 * @object_path: The object path that was registered with g_dbus_connection_register_subtree().
433 * @node: A node that is a child of @object_path (relative to @object_path) or %NULL for the root of the subtree.
434 * @user_data: The @user_data #gpointer passed to g_dbus_connection_register_subtree().
435 *
436 * The type of the @introspect function in #GDBusSubtreeVTable.
437 *
438 * Subtrees are flat. @node, if non-%NULL, is always exactly one
439 * segment of the object path (ie: it never contains a slash).
440 *
441 * This function should return %NULL to indicate that there is no object
442 * at this node.
443 *
444 * If this function returns non-%NULL, the return value is expected to
445 * be a %NULL-terminated array of pointers to #GDBusInterfaceInfo
446 * structures describing the interfaces implemented by @node. This
447 * array will have g_dbus_interface_info_unref() called on each item
448 * before being freed with g_free().
449 *
450 * The difference between returning %NULL and an array containing zero
451 * items is that the standard DBus interfaces will returned to the
452 * remote introspector in the empty array case, but not in the %NULL
453 * case.
454 *
455 * Returns: A %NULL-terminated array of pointers to #GDBusInterfaceInfo, or %NULL.
456 *
457 * Since: 2.26
458 */
459typedef GDBusInterfaceInfo ** (*GDBusSubtreeIntrospectFunc) (GDBusConnection *connection,
460 const gchar *sender,
461 const gchar *object_path,
462 const gchar *node,
463 gpointer user_data);

569/**
570 * GDBusMessageFilterFunction:
571 * @connection: (transfer none): A #GDBusConnection.
572 * @message: (transfer full): A locked #GDBusMessage that the filter function takes ownership of.
573 * @incoming: %TRUE if it is a message received from the other peer, %FALSE if it is
574 * a message to be sent to the other peer.
575 * @user_data: User data passed when adding the filter.
576 *
577 * Signature for function used in g_dbus_connection_add_filter().
578 *
579 * A filter function is passed a #GDBusMessage and expected to return
580 * a #GDBusMessage too. Passive filter functions that don't modify the
581 * message can simply return the @message object:
582 * |[
583 * static GDBusMessage *
584 * passive_filter (GDBusConnection *connection
585 * GDBusMessage *message,
586 * gboolean incoming,
587 * gpointer user_data)
588 * {
589 * /<!-- -->* inspect @message *<!-- -->/
590 * return message;
591 * }
592 * ]|
593 * Filter functions that wants to drop a message can simply return %NULL:
594 * |[
595 * static GDBusMessage *
596 * drop_filter (GDBusConnection *connection
597 * GDBusMessage *message,
598 * gboolean incoming,
599 * gpointer user_data)
600 * {
601 * if (should_drop_message)
602 * {
603 * g_object_unref (message);
604 * message = NULL;
605 * }
606 * return message;
607 * }
608 * ]|
609 * Finally, a filter function may modify a message by copying it:
610 * |[
611 * static GDBusMessage *
612 * modifying_filter (GDBusConnection *connection
613 * GDBusMessage *message,
614 * gboolean incoming,
615 * gpointer user_data)
616 * {
617 * GDBusMessage *copy;
618 * GError *error;
619 *
620 * error = NULL;
621 * copy = g_dbus_message_copy (message, &error);
622 * /<!-- -->* handle @error being is set *<!-- -->/
623 * g_object_unref (message);
624 *
625 * /<!-- -->* modify @copy *<!-- -->/
626 *
627 * return copy;
628 * }
629 * ]|
630 * If the returned #GDBusMessage is different from @message and cannot
631 * be sent on @connection (it could use features, such as file
632 * descriptors, not compatible with @connection), then a warning is
633 * logged to <emphasis>standard error</emphasis>. Applications can
634 * check this ahead of time using g_dbus_message_to_blob() passing a
635 * #GDBusCapabilityFlags value obtained from @connection.
636 *
637 * Returns: (transfer full) (allow-none): A #GDBusMessage that will be freed with
638 * g_object_unref() or %NULL to drop the message. Passive filter
639 * functions can simply return the passed @message object.
640 *
641 * Since: 2.26
642 */
643typedef GDBusMessage *(*GDBusMessageFilterFunction) (GDBusConnection *connection,
644 GDBusMessage *message,
645 gboolean incoming,
646 gpointer user_data);
