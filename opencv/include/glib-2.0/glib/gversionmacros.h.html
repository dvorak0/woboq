118/* evaluates to the current stable version; for development cycles,
119 * this means the next stable target
120 */
121#if (GLIB_MINOR_VERSION % 2)
122#define GLIB_VERSION_CUR_STABLE (G_ENCODE_VERSION (GLIB_MAJOR_VERSION, GLIB_MINOR_VERSION + 1))
123#else
124#define GLIB_VERSION_CUR_STABLE (G_ENCODE_VERSION (GLIB_MAJOR_VERSION, GLIB_MINOR_VERSION))
125#endif
126
127/* evaluates to the previous stable version */
128#if (GLIB_MINOR_VERSION % 2)
129#define GLIB_VERSION_PREV_STABLE (G_ENCODE_VERSION (GLIB_MAJOR_VERSION, GLIB_MINOR_VERSION - 1))
130#else
131#define GLIB_VERSION_PREV_STABLE (G_ENCODE_VERSION (GLIB_MAJOR_VERSION, GLIB_MINOR_VERSION - 2))
132#endif

134/**
135 * GLIB_VERSION_MIN_REQUIRED:
136 *
137 * A macro that should be defined by the user prior to including
138 * the glib.h header.
139 * The definition should be one of the predefined GLib version
140 * macros: %GLIB_VERSION_2_26, %GLIB_VERSION_2_28,...
141 *
142 * This macro defines the earliest version of GLib that the package is
143 * required to be able to compile against.
144 *
145 * If the compiler is configured to warn about the use of deprecated
146 * functions, then using functions that were deprecated in version
147 * %GLIB_VERSION_MIN_REQUIRED or earlier will cause warnings (but
148 * using functions deprecated in later releases will not).
149 *
150 * Since: 2.32
151 */
152/* If the package sets GLIB_VERSION_MIN_REQUIRED to some future
153 * GLIB_VERSION_X_Y value that we don't know about, it will compare as
154 * 0 in preprocessor tests.
155 */
156#ifndef GLIB_VERSION_MIN_REQUIRED
157# define GLIB_VERSION_MIN_REQUIRED (GLIB_VERSION_CUR_STABLE)
158#elif GLIB_VERSION_MIN_REQUIRED == 0
159# undef GLIB_VERSION_MIN_REQUIRED
160# define GLIB_VERSION_MIN_REQUIRED (GLIB_VERSION_CUR_STABLE + 2)
161#endif
162
163/**
164 * GLIB_VERSION_MAX_ALLOWED:
165 *
166 * A macro that should be defined by the user prior to including
167 * the glib.h header.
168 * The definition should be one of the predefined GLib version
169 * macros: %GLIB_VERSION_2_26, %GLIB_VERSION_2_28,...
170 *
171 * This macro defines the latest version of the GLib API that the
172 * package is allowed to make use of.
173 *
174 * If the compiler is configured to warn about the use of deprecated
175 * functions, then using functions added after version
176 * %GLIB_VERSION_MAX_ALLOWED will cause warnings.
177 *
178 * Unless you are using GLIB_CHECK_VERSION() or the like to compile
179 * different code depending on the GLib version, then this should be
180 * set to the same value as %GLIB_VERSION_MIN_REQUIRED.
181 *
182 * Since: 2.32
183 */
184#if !defined (GLIB_VERSION_MAX_ALLOWED) || (GLIB_VERSION_MAX_ALLOWED == 0)
185# undef GLIB_VERSION_MAX_ALLOWED
186# define GLIB_VERSION_MAX_ALLOWED (GLIB_VERSION_CUR_STABLE)
187#endif
188
189/* sanity checks */
190#if GLIB_VERSION_MIN_REQUIRED > GLIB_VERSION_CUR_STABLE
191#error "GLIB_VERSION_MIN_REQUIRED must be <= GLIB_VERSION_CUR_STABLE"
192#endif
193#if GLIB_VERSION_MAX_ALLOWED < GLIB_VERSION_MIN_REQUIRED
194#error "GLIB_VERSION_MAX_ALLOWED must be >= GLIB_VERSION_MIN_REQUIRED"
195#endif
196#if GLIB_VERSION_MIN_REQUIRED < GLIB_VERSION_2_26
197#error "GLIB_VERSION_MIN_REQUIRED must be >= GLIB_VERSION_2_26"
198#endif

200/* These macros are used to mark deprecated functions in GLib headers,
201 * and thus have to be exposed in installed headers. But please
202 * do *not* use them in other projects. Instead, use G_DEPRECATED
203 * or define your own wrappers around it.
204 */
205#define GLIB_AVAILABLE_IN_ALL _GLIB_EXTERN
