22#ifndef __PANGO_ITEM_H__
23#define __PANGO_ITEM_H__
24
25#include <pango/pango-types.h>
26
27G_BEGIN_DECLS
28
29typedef struct _PangoAnalysis PangoAnalysis;
30typedef struct _PangoItem PangoItem;
31
32/* TODO: if more flags are needed, turn this into a real PangoAnalysisFlags enum */
33/**
34 * PANGO_ANALYSIS_FLAG_CENTERED_BASELINE:
35 *
36 * Whether the segment should be shifted to center around the baseline.
37 * Used in vertical writing directions mostly.
38 *
39 * Since: 1.16
40 */
41#define PANGO_ANALYSIS_FLAG_CENTERED_BASELINE (1 << 0)
42
43/**
44 * PangoAnalysis:
45 * @shape_engine: the engine for doing rendering-system-dependent processing.
46 * @lang_engine: the engine for doing rendering-system-independent processing.
47 * @font: the font for this segment.
48 * @level: the bidirectional level for this segment.
49 * @gravity: the glyph orientation for this segment (A #PangoGravity).
50 * @flags: boolean flags for this segment (currently only one) (Since: 1.16).
51 * @script: the detected script for this segment (A #PangoScript) (Since: 1.18).
52 * @language: the detected language for this segment.
53 * @extra_attrs: extra attributes for this segment.
54 *
55 * The #PangoAnalysis structure stores information about
56 * the properties of a segment of text.
57 */
58struct _PangoAnalysis
59{
60 PangoEngineShape *shape_engine;
61 PangoEngineLang *lang_engine;
62 PangoFont *font;
63
64 guint8 level;
65 guint8 gravity; /* PangoGravity */
66 guint8 flags;
67
68 guint8 script; /* PangoScript */
69 PangoLanguage *language;
70
71 GSList *extra_attrs;
72};
73
74/**
75 * PangoItem:
76 *
77 * The #PangoItem structure stores information abouta segment of text.
78 */
79struct _PangoItem
80{
81 gint offset;
82 gint length;
83 gint num_chars;
84 PangoAnalysis analysis;
85};
