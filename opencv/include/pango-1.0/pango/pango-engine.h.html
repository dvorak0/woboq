115/**
116 * PangoEngineLangClass:

128 /*< private >*/

132 void (*script_break) (PangoEngineLang *engine,
133 const char *text,
134 int len,
135 PangoAnalysis *analysis,
136 PangoLogAttr *attrs,
137 int attrs_len);
138};

159/**
160 * PangoEngineShape
161 *
162 * The #PangoEngineShape class is implemented by engines that
163 * customize the rendering-system dependent part of the
164 * Pango pipeline for a particular script or language.
165 * A #PangoEngineShape implementation is then specific to both
166 * a particular rendering system or group of rendering systems
167 * and to a particular script. For instance, there is one
168 * #PangoEngineShape implementation to handle shaping Arabic
169 * for Fontconfig-based backends.
170 **/
171struct _PangoEngineShape
172{
173 PangoEngine parent_instance;
174};
175
176/**
177 * PangoEngineShapeClass:
178 * @script_shape: Given a font, a piece of text, and a #PangoAnalysis
179 * structure, converts characters to glyphs and positions the
180 * resulting glyphs. The results are stored in the #PangoGlyphString
181 * that is passed in. (The implementation should resize it
182 * appropriately using pango_glyph_string_set_size()). All fields
183 * of the @log_clusters and @glyphs array must be filled in, with
184 * the exception that Pango will automatically generate
185 * <literal>glyphs->glyphs[i].attr.is_cluster_start</literal>
186 * using the @log_clusters array. Each input character must occur in one
187 * of the output logical clusters;
188 * if no rendering is desired for a character, this may involve
189 * inserting glyphs with the #PangoGlyph ID #PANGO_GLYPH_EMPTY, which
190 * is guaranteed never to render. If the shaping fails for any reason,
191 * the shaper should return with an empty (zero-size) glyph string.
192 * If the shaper has not set the size on the glyph string yet, simply
193 * returning signals the failure too.
194 * @covers: Returns the characters that this engine can cover
195 * with a given font for a given language. If not overridden, the default
196 * implementation simply returns the coverage information for the
197 * font itself unmodified.
198 *
199 * Class structure for #PangoEngineShape
200 **/
201struct _PangoEngineShapeClass
202{
203 /*< private >*/
204 PangoEngineClass parent_class;
205
206 /*< public >*/
207 void (*script_shape) (PangoEngineShape *engine,
208 PangoFont *font,
209 const char *item_text,
210 unsigned int item_length,
211 const PangoAnalysis *analysis,
212 PangoGlyphString *glyphs,
213 const char *paragraph_text,
214 unsigned int paragraph_length);
215 PangoCoverageLevel (*covers) (PangoEngineShape *engine,
216 PangoFont *font,
217 PangoLanguage *language,
218 gunichar wc);
219};

226/**
227 * PangoEngineScriptInfo
228 * @script: a #PangoScript. The value %PANGO_SCRIPT_COMMON has
229 * the special meaning here of "all scripts"
230 * @langs: a semicolon separated list of languages that this
231 * engine handles for this script. This may be empty,
232 * in which case the engine is saying that it is a
233 * fallback choice for all languages for this range,
234 * but should not be used if another engine
235 * indicates that it is specific for the language for
236 * a given code point. An entry in this list of "*"
237 * indicates that this engine is specific to all
238 * languages for this range.
239 *
240 * The #PangoEngineScriptInfo structure contains
241 * information about how the shaper covers a particular script.
242 */
243struct _PangoEngineScriptInfo
244{
245 PangoScript script;
246 const gchar *langs;
247};

249/**
250 * PangoEngineInfo:
251 * @id: a unique string ID for the engine.
252 * @engine_type: a string identifying the engine type.
253 * @render_type: a string identifying the render type.
254 * @scripts: array of scripts this engine supports.
255 * @n_scripts: number of items in @scripts.
256 *
257 * The #PangoEngineInfo structure contains information about a particular
258 * engine. It contains the following fields:

262 const gchar *id;

322#define PANGO_ENGINE_DEFINE_TYPE(name, prefix, class_init, instance_init, parent_type) \
323static GType prefix ## _type; \
324static void \
325prefix ## _register_type (GTypeModule *module) \
326{ \
327 const GTypeInfo object_info = \
328 { \
329 sizeof (name ## Class), \
330 (GBaseInitFunc) NULL, \
331 (GBaseFinalizeFunc) NULL, \
332 (GClassInitFunc) class_init, \
333 (GClassFinalizeFunc) NULL, \
334 NULL, /* class_data */ \
335 sizeof (name), \
336 0, /* n_prelocs */ \
337 (GInstanceInitFunc) instance_init, \
338 NULL /* value_table */ \
339 }; \
340 \
341 prefix ## _type = g_type_module_register_type (module, parent_type, \
342 # name, \
343 &object_info, 0); \
344}

346/**
347 * PANGO_ENGINE_LANG_DEFINE_TYPE:
348 * @name: Name of the the type to register (for example:, <literal>ArabicEngineFc</literal>
349 * @prefix: Prefix for symbols that will be defined (for example:, <literal>arabic_engine_fc</literal>
350 * @class_init: Class initialization function for the new type, or %NULL
351 * @instance_init: Instance initialization function for the new type, or %NULL
352 *
353 * Outputs the necessary code for GObject type registration for a
354 * #PangoEngineLang class defined in a module. Two static symbols
355 * are defined.
356 *
357 * <programlisting>
358 * static GType <replaceable>prefix</replaceable>_type;
359 * static void <replaceable>prefix</replaceable>_register_type (GTypeModule module);
360 * </programlisting>
361 *
362 * The <function><replaceable>prefix</replaceable>_register_type()</function>
363 * function should be called in your script_engine_init() function for
364 * each type that your module implements, and then your script_engine_create()
365 * function can create instances of the object as follows:
366 *
367 * <informalexample><programlisting>
368 * PangoEngine *engine = g_object_new (<replaceable>prefix</replaceable>_type, NULL);
369 * </programlisting></informalexample>
370 **/
371#define PANGO_ENGINE_LANG_DEFINE_TYPE(name, prefix, class_init, instance_init) \
372 PANGO_ENGINE_DEFINE_TYPE (name, prefix, \
373 class_init, instance_init, \
374 PANGO_TYPE_ENGINE_LANG)

406/* Macro used for possibly builtin Pango modules. Not useful
407 * for externally build modules. If we are compiling a module standalone,
408 * then we name the entry points script_engine_list, etc. But if we
409 * are compiling it for inclusion directly in Pango, then we need them to
410 * to have distinct names for this module, so we prepend a prefix.
411 *
412 * The two intermediate macros are to deal with details of the C
413 * preprocessor; token pasting tokens must be function arguments,
414 * and macro substitution isn't used on function arguments that
415 * are used for token pasting.
416 */
417#ifdef PANGO_MODULE_PREFIX
418#define PANGO_MODULE_ENTRY(func) _PANGO_MODULE_ENTRY2(PANGO_MODULE_PREFIX,func)
419#define _PANGO_MODULE_ENTRY2(prefix,func) _PANGO_MODULE_ENTRY3(prefix,func)
420#define _PANGO_MODULE_ENTRY3(prefix,func) prefix##_script_engine_##func
421#else
422#define PANGO_MODULE_ENTRY(func) script_engine_##func
423#endif
