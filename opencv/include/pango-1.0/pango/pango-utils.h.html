115/* Encode a Pango version as an integer */
116/**
117 * PANGO_VERSION_ENCODE:
118 * @major: the major component of the version number
119 * @minor: the minor component of the version number
120 * @micro: the micro component of the version number
121 *
122 * This macro encodes the given Pango version into an integer. The numbers
123 * returned by %PANGO_VERSION and pango_version() are encoded using this macro.
124 * Two encoded version numbers can be compared as integers.
125 */
126#define PANGO_VERSION_ENCODE(major, minor, micro) ( \
127 ((major) * 10000) \
128 + ((minor) * 100) \
129 + ((micro) * 1))
130
131/* Encoded version of Pango at compile-time */
132/**
133 * PANGO_VERSION:
134 *
135 * The version of Pango available at compile-time, encoded using PANGO_VERSION_ENCODE().
136 */
137/**
138 * PANGO_VERSION_STRING:
139 *
140 * A string literal containing the version of Pango available at compile-time.
141 */
142/**
143 * PANGO_VERSION_MAJOR:
144 *
145 * The major component of the version of Pango available at compile-time.
146 */
147/**
148 * PANGO_VERSION_MINOR:
149 *
150 * The minor component of the version of Pango available at compile-time.
151 */
152/**
153 * PANGO_VERSION_MICRO:
154 *
155 * The micro component of the version of Pango available at compile-time.
156 */
157#define PANGO_VERSION PANGO_VERSION_ENCODE( \
158 PANGO_VERSION_MAJOR, \
159 PANGO_VERSION_MINOR, \
160 PANGO_VERSION_MICRO)
161
162/* Check that compile-time Pango is as new as required */
163/**
164 * PANGO_VERSION_CHECK:
165 * @major: the major component of the version number
166 * @minor: the minor component of the version number
167 * @micro: the micro component of the version number
168 *
169 * Checks that the version of Pango available at compile-time is not older than
170 * the provided version number.
171 */
172#define PANGO_VERSION_CHECK(major,minor,micro) \
173 (PANGO_VERSION >= PANGO_VERSION_ENCODE(major,minor,micro))
174
