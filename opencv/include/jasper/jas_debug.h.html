89/* Output debugging information to standard error provided that the debug
90 level is set sufficiently high. */
91#if defined(DEBUG)
92#define JAS_DBGLOG(n, x) \
93 ((jas_getdbglevel() >= (n)) ? (jas_eprintf x) : 0)
94#else
95#define JAS_DBGLOG(n, x)
96#endif
97
98/* Get the library debug level. */
99int jas_getdbglevel(void);
100
101/* Set the library debug level. */
102int jas_setdbglevel(int dbglevel);
103
104/* Perform formatted output to standard error. */
105int jas_eprintf(const char *fmt, ...);
106
107/* Dump memory to a stream. */
108int jas_memdump(FILE *out, void *data, size_t len);
109

111}
