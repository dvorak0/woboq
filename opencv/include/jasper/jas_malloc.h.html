90#if defined(DEBUG_MEMALLOC)
91/* This is somewhat of a hack, but it's a useful hack. :-) */
92/* Use my own custom memory allocator for debugging. */
93#include "../../../../local/src/memalloc.h"
94#define jas_malloc MEMALLOC
95#define jas_free MEMFREE
96#define jas_realloc MEMREALLOC
97#define jas_calloc MEMCALLOC
98#define jas_alloc2(a, b) MEMALLOC((a)*(b))
99#define jas_alloc3(a, b, c) MEMALLOC((a)*(b)*(c))
100#define jas_realloc2(p, a, b) MEMREALLOC((p), (a)*(b))
101#endif

103/******************************************************************************\
104* Functions.
105\******************************************************************************/
106
107#if !defined(DEBUG_MEMALLOC)
108
109/* Allocate memory. */
110void *jas_malloc(size_t size);
111
112/* Free memory. */
113void jas_free(void *ptr);
114
115/* Resize a block of allocated memory. */
116void *jas_realloc(void *ptr, size_t size);
117
118/* Allocate a block of memory and initialize the contents to zero. */
119void *jas_calloc(size_t nmemb, size_t size);
120
121/* size-checked double allocation .*/
122void *jas_alloc2(size_t, size_t);
123
124void *jas_alloc3(size_t, size_t, size_t);
125
126void *jas_realloc2(void *, size_t, size_t);
127#endif
