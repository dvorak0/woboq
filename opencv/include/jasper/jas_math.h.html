91/* Compute the absolute value. */
92#define JAS_ABS(x) \
93 (((x) >= 0) ? (x) : (-(x)))
94
95/* Compute the minimum of two values. */
96#define JAS_MIN(x, y) \
97 (((x) < (y)) ? (x) : (y))
98
99/* Compute the maximum of two values. */
100#define JAS_MAX(x, y) \
101 (((x) > (y)) ? (x) : (y))
102
103/* Compute the remainder from division (where division is defined such
104 that the remainder is always nonnegative). */
105#define JAS_MOD(x, y) \
106 (((x) < 0) ? (((-x) % (y)) ? ((y) - ((-(x)) % (y))) : (0)) : ((x) % (y)))
107
108/* Compute the integer with the specified number of least significant bits
109 set to one. */
110#define JAS_ONES(n) \
111 ((1 << (n)) - 1)
