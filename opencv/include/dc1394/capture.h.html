36/**
37 * The capture policy.
38 *
39 * Can be blocking (wait for a frame forever) or polling (returns if no frames is in the ring buffer)
40 */
41typedef enum {
42 DC1394_CAPTURE_POLICY_WAIT=672,
43 DC1394_CAPTURE_POLICY_POLL
44} dc1394capture_policy_t;
45#define DC1394_CAPTURE_POLICY_MIN DC1394_CAPTURE_POLICY_WAIT
46#define DC1394_CAPTURE_POLICY_MAX DC1394_CAPTURE_POLICY_POLL
47#define DC1394_CAPTURE_POLICY_NUM (DC1394_CAPTURE_POLICY_MAX - DC1394_CAPTURE_POLICY_MIN + 1)
48
49/**
50 * Capture flags. Currently limited to switching automatic functions on/off: channel allocation, bandwidth allocation and automatic
51 * starting of ISO transmission
52 */
53#define DC1394_CAPTURE_FLAGS_CHANNEL_ALLOC 0x00000001U
54#define DC1394_CAPTURE_FLAGS_BANDWIDTH_ALLOC 0x00000002U
55#define DC1394_CAPTURE_FLAGS_DEFAULT 0x00000004U /* a reasonable default value: do bandwidth and channel allocation */
56#define DC1394_CAPTURE_FLAGS_AUTO_ISO 0x00000008U /* automatically start iso before capture and stop it after */
57
58#ifdef __cplusplus
59extern "C" {
60#endif
61
62/***************************************************************************
63 Capture Functions
64 ***************************************************************************/
65
66/**
67 * Setup the capture, using a ring buffer of a certain size (num_dma_buffers) and certain options (flags)
68 */
69dc1394error_t dc1394_capture_setup(dc1394camera_t *camera, uint32_t num_dma_buffers, uint32_t flags);
70
71/**
72 * Stop the capture
73 */
74dc1394error_t dc1394_capture_stop(dc1394camera_t *camera);
75
76/**
77 * Gets a file descriptor to be used for select(). Must be called after dc1394_capture_setup().
78 */
79int dc1394_capture_get_fileno (dc1394camera_t * camera);
80
81/**
82 * Captures a video frame. The returned struct contains the image buffer, among others. This image buffer SHALL NOT be freed, as it represents an area
83 * in the memory that belongs to the system.
84 */
85dc1394error_t dc1394_capture_dequeue(dc1394camera_t * camera, dc1394capture_policy_t policy, dc1394video_frame_t **frame);
86
87/**
88 * Returns a frame to the ring buffer once it has been used.
89 */
90dc1394error_t dc1394_capture_enqueue(dc1394camera_t * camera, dc1394video_frame_t * frame);
91
92/**
93 * Returns DC1394_TRUE if the given frame (previously dequeued) has been
94 * detected to be corrupt (missing data, corrupted data, overrun buffer, etc.).
95 * Note that certain types of corruption may go undetected in which case
96 * DC1394_FALSE will be returned. The ability to detect corruption also
97 * varies between platforms. Note that corrupt frames still need to be
98 * enqueued with dc1394_capture_enqueue() when no longer needed by the user.
99 */
100dc1394bool_t dc1394_capture_is_frame_corrupt (dc1394camera_t * camera,
101 dc1394video_frame_t * frame);
102
103#ifdef __cplusplus
104}
