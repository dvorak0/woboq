46/**
47 * GdkPixbufLoader:
48 *
49 * The GdkPixbufLoader struct contains only private
50 * fields.
51 */
52typedef struct _GdkPixbufLoader GdkPixbufLoader;
53struct _GdkPixbufLoader
54{
55 GObject parent_instance;
56
57 /*< private >*/
58 gpointer priv;
59};

61typedef struct _GdkPixbufLoaderClass GdkPixbufLoaderClass;
62struct _GdkPixbufLoaderClass
63{
64 GObjectClass parent_class;
65
66 void (*size_prepared) (GdkPixbufLoader *loader,
67 int width,
68 int height);
69
70 void (*area_prepared) (GdkPixbufLoader *loader);
71
72 /* Last known frame needs a redraw for x, y, width, height */
73 void (*area_updated) (GdkPixbufLoader *loader,
74 int x,
75 int y,
76 int width,
77 int height);
78
79 void (*closed) (GdkPixbufLoader *loader);
80};
