52G_BEGIN_DECLS
53
54#define ATK_TYPE_MISC (atk_misc_get_type ())
55#define ATK_IS_MISC(obj) G_TYPE_CHECK_INSTANCE_TYPE ((obj), ATK_TYPE_MISC)
56#define ATK_MISC(obj) G_TYPE_CHECK_INSTANCE_CAST ((obj), ATK_TYPE_MISC, AtkMisc)
57#define ATK_MISC_CLASS(klass) (G_TYPE_CHECK_CLASS_CAST ((klass), ATK_TYPE_MISC, AtkMiscClass))
58#define ATK_IS_MISC_CLASS(klass) (G_TYPE_CHECK_CLASS_TYPE ((klass), ATK_TYPE_MISC))
59#define ATK_MISC_GET_CLASS(obj) (G_TYPE_INSTANCE_GET_CLASS ((obj), ATK_TYPE_MISC, AtkMiscClass))
60
61
62#ifndef _TYPEDEF_ATK_MISC_
63#define _TYPEDEF_ATK_MISC_
64typedef struct _AtkMisc AtkMisc;
65typedef struct _AtkMiscClass AtkMiscClass;
66#endif
67
68struct _AtkMisc
69{
70 GObject parent;
71};
72
73/*
74 * Singleton instance - only the ATK implementation layer for
75 * a given GUI toolkit/application instance should touch this
76 * symbol directly.
77 */
78ATK_VAR AtkMisc *atk_misc_instance;
79
80struct _AtkMiscClass
81{
82 GObjectClass parent;
83 void (* threads_enter) (AtkMisc *misc);
84 void (* threads_leave) (AtkMisc *misc);
85 gpointer vfuncs[32]; /* future bincompat */
86};
87GType atk_misc_get_type (void);
88
89/*
90 * Wrapper for thread lock, i.e. take the thread mutex for the GUI toolkit,
91 * if one exists.
92 * (This method is implemented by the toolkit ATK implementation layer;
93 * for instance, for GTK+, GAIL implements this via GDK_THREADS_ENTER).
94 */
95void atk_misc_threads_enter (AtkMisc *misc);
96
97/*
98 * Wrapper for thread lock, i.e. release the thread mutex for the GUI toolkit,
99 * if one exists.
100 * (This method is implemented by the toolkit ATK implementation layer;
101 * for instance, for GTK+, GAIL implements this via GDK_THREADS_LEAVE).
102 */
103void atk_misc_threads_leave (AtkMisc *misc);
104
105/*
106 * Obtain (singleton) instance of AtkMisc.
107 */
108const AtkMisc *atk_misc_get_instance (void);
109
110G_END_DECLS
