32/*
33 * The AtkDocument interface should be supported by any object that is a container
34 * for 'document content' as opposed to a collection of user interface elements.
35 *
36 */
37
38#define ATK_TYPE_DOCUMENT (atk_document_get_type ())
39#define ATK_IS_DOCUMENT(obj) G_TYPE_CHECK_INSTANCE_TYPE ((obj), ATK_TYPE_DOCUMENT)
40#define ATK_DOCUMENT(obj) G_TYPE_CHECK_INSTANCE_CAST ((obj), ATK_TYPE_DOCUMENT, AtkDocument)
41#define ATK_DOCUMENT_GET_IFACE(obj) (G_TYPE_INSTANCE_GET_INTERFACE ((obj), ATK_TYPE_DOCUMENT, AtkDocumentIface))
42
43#ifndef _TYPEDEF_ATK_DOCUMENT_
44#define _TYPEDEF_ATK_DOCUMENT_
45typedef struct _AtkDocument AtkDocument;
46#endif
47typedef struct _AtkDocumentIface AtkDocumentIface;
48
49/**
50 * AtkDocumentIface:
51 * @get_document_locale: gets locale. This virtual function is
52 * deprecated since 2.7.90 and it should not be overriden.
53 */
54struct _AtkDocumentIface
55{
56 GTypeInterface parent;
57 const gchar* ( *get_document_type) (AtkDocument *document);
58 gpointer ( *get_document) (AtkDocument *document);
59
60 const gchar* ( *get_document_locale) (AtkDocument *document);
61 AtkAttributeSet * ( *get_document_attributes) (AtkDocument *document);
62 const gchar* ( *get_document_attribute_value) (AtkDocument *document,
63 const gchar *attribute_name);
64 gboolean ( *set_document_attribute) (AtkDocument *document,
65 const gchar *attribute_name,
66 const gchar *attribute_value);
67};
68
69GType atk_document_get_type (void);
70
71const gchar* atk_document_get_document_type (AtkDocument *document);
72gpointer atk_document_get_document (AtkDocument *document);
73
74G_DEPRECATED
75const gchar* atk_document_get_locale (AtkDocument *document);
76
77AtkAttributeSet* atk_document_get_attributes (AtkDocument *document);
78const gchar* atk_document_get_attribute_value (AtkDocument *document,
79 const gchar *attribute_name);
80gboolean atk_document_set_attribute_value (AtkDocument *document,
81 const gchar *attribute_name,
82 const gchar *attribute_value);
83
84G_END_DECLS
