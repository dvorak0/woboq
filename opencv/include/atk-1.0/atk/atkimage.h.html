32/*
33 * The AtkImage interface should be supported by any object that has an
34 * associated image. This interface provides the standard mechanism for
35 * an assistive technology to get descriptive information about images.
36 */
37
38#define ATK_TYPE_IMAGE (atk_image_get_type ())
39#define ATK_IS_IMAGE(obj) G_TYPE_CHECK_INSTANCE_TYPE ((obj), ATK_TYPE_IMAGE)
40#define ATK_IMAGE(obj) G_TYPE_CHECK_INSTANCE_CAST ((obj), ATK_TYPE_IMAGE, AtkImage)
41#define ATK_IMAGE_GET_IFACE(obj) (G_TYPE_INSTANCE_GET_INTERFACE ((obj), ATK_TYPE_IMAGE, AtkImageIface))
42
43#ifndef _TYPEDEF_ATK_IMAGE_
44#define _TYPEDEF_ATK_IMAGE_
45typedef struct _AtkImage AtkImage;
46#endif
47typedef struct _AtkImageIface AtkImageIface;
48
49struct _AtkImageIface
50{
51 GTypeInterface parent;
52 void ( *get_image_position) (AtkImage *image,
53 gint *x,
54 gint *y,
55 AtkCoordType coord_type);
56 const gchar* ( *get_image_description) (AtkImage *image);
57 void ( *get_image_size) (AtkImage *image,
58 gint *width,
59 gint *height);
60 gboolean ( *set_image_description) (AtkImage *image,
61 const gchar *description);
62 const gchar* ( *get_image_locale) (AtkImage *image);
63};
64
65GType atk_image_get_type (void);
66
67const gchar* atk_image_get_image_description (AtkImage *image);
68
69void atk_image_get_image_size (AtkImage *image,
70 gint *width,
71 gint *height);
72
73gboolean atk_image_set_image_description (AtkImage *image,
74 const gchar *description);
75void atk_image_get_image_position (AtkImage *image,
76 gint *x,
77 gint *y,
78 AtkCoordType coord_type);
79
80const gchar* atk_image_get_image_locale (AtkImage *image);
