46/**
47 * AtkEventListener:
48 * @obj: An #AtkObject instance for whom the callback will be called when
49 * the specified event (e.g. 'focus:') takes place.
50 *
51 * A function which is called when an object emits a matching event,
52 * as used in #atk_add_focus_tracker.
53 * Currently the only events for which object-specific handlers are
54 * supported are events of type "focus:". Most clients of ATK will prefer to
55 * attach signal handlers for the various ATK signals instead.
56 *
57 * see atk_add_focus_tracker.
58 **/
59typedef void (*AtkEventListener) (AtkObject* obj);
60/**
61 * AtkEventListenerInit:
62 *
63 * An #AtkEventListenerInit function is a special function that is
64 * called in order to initialize the per-object event registration system
65 * used by #AtkEventListener, if any preparation is required.
66 *
67 * see atk_focus_tracker_init.
68 **/
69typedef void (*AtkEventListenerInit) (void);
70/**
71 * AtkKeySnoopFunc:
72 * @event: an AtkKeyEventStruct containing information about the key event for which
73 * notification is being given.
74 * @user_data: a block of data which will be passed to the event listener, on notification.
75 *
76 * An #AtkKeySnoopFunc is a type of callback which is called whenever a key event occurs,
77 * if registered via atk_add_key_event_listener. It allows for pre-emptive
78 * interception of key events via the return code as described below.
79 *
80 * Returns: TRUE (nonzero) if the event emission should be stopped and the event
81 * discarded without being passed to the normal GUI recipient; FALSE (zero) if the
82 * event dispatch to the client application should proceed as normal.
83 *
84 * see atk_add_key_event_listener.
85 **/
86typedef gint (*AtkKeySnoopFunc) (AtkKeyEventStruct *event,
87 gpointer user_data);
88
89/**
90 * AtkKeyEventStruct:
91 * @type: An AtkKeyEventType, generally one of ATK_KEY_EVENT_PRESS or ATK_KEY_EVENT_RELEASE
92 * @state: A bitmask representing the state of the modifier keys immediately after the event takes place.
93 * The meaning of the bits is currently defined to match the bitmask used by GDK in
94 * GdkEventType.state, see
95 * http://developer.gnome.org/doc/API/2.0/gdk/gdk-Event-Structures.html#GdkEventKey
96 * @keyval: A guint representing a keysym value corresponding to those used by GDK and X11: see
97 * /usr/X11/include/keysymdef.h.
98 * @length: The length of member #string.
99 * @string: A string containing one of the following: either a string approximating the text that would
100 * result from this keypress, if the key is a control or graphic character, or a symbolic name for this keypress.
101 * Alphanumeric and printable keys will have the symbolic key name in this string member, for instance "A". "0",
102 * "semicolon", "aacute". Keypad keys have the prefix "KP".
103 * @keycode: The raw hardware code that generated the key event. This field is raraly useful.
104 * @timestamp: A timestamp in milliseconds indicating when the event occurred.
105 * These timestamps are relative to a starting point which should be considered arbitrary,
106 * and only used to compare the dispatch times of events to one another.
107 *
108 * Encapsulates information about a key event.
109 **/
110struct _AtkKeyEventStruct {
111 gint type;
112 guint state;
113 guint keyval;
114 gint length;
115 gchar *string;
116 guint16 keycode;
117 guint32 timestamp;
118};

120/**
121 *AtkKeyEventType:
122 *@ATK_KEY_EVENT_PRESS: specifies a key press event
123 *@ATK_KEY_EVENT_RELEASE: specifies a key release event
124 *@ATK_KEY_EVENT_LAST_DEFINED: Not a valid value; specifies end of enumeration
125 *
126 *Specifies the type of a keyboard evemt.
127 **/
128typedef enum
129{
130 ATK_KEY_EVENT_PRESS,
131 ATK_KEY_EVENT_RELEASE,
132 ATK_KEY_EVENT_LAST_DEFINED
133} AtkKeyEventType;

190/* --- GType boilerplate --- */
191/* convenience macros for atk type implementations, which for a type GtkGadgetAccessible will:
192 * - prototype: static void gtk_gadget_accessible_class_init (GtkGadgetClass *klass);
193 * - prototype: static void gtk_gadget_accessible_init (GtkGadget *self);
194 * - define: static gpointer gtk_gadget_accessible_parent_class = NULL;
195 * gtk_gadget_accessible_parent_class is initialized prior to calling gtk_gadget_class_init()
196 * - implement: GType gtk_gadget_accessible_get_type (void) { ... }
197 * - support custom code in gtk_gadget_accessible_get_type() after the type is registered.
198 *
199 * macro arguments: TypeName, type_name, TYPE_PARENT, CODE
200 * example: ATK_DEFINE_TYPE_WITH_CODE (GtkGadgetAccessible, gtk_gadget_accessible, GTK_TYPE_GADGET,
201 * G_IMPLEMENT_INTERFACE (ATK_TYPE_TABLE, gtk_gadget_accessible_table_iface_init))
202 */

204/**
205 * ATK_DEFINE_TYPE:
206 * @TN: The name of the new type, in Camel case.
207 * @t_n: The name of the new type, in lowercase, with words separated by '_'.
208 * @T_P: The #GType of the parent type.
209 *
210 * A convenience macro for type ATK implementations, which declares a class
211 * initialization function, an instance initialization function (see #GTypeInfo
212 * for information about these) and a static variable named
213 * @t_n _parent_class pointing to the parent class. Furthermore, it
214 * defines a _get_type() function.
215 *
216 * Since: 1.22
217 */
218#define ATK_DEFINE_TYPE(TN, t_n, T_P) ATK_DEFINE_TYPE_EXTENDED (TN, t_n, T_P, 0, {})
219
220/**
221 * ATK_DEFINE_TYPE_WITH_CODE:
222 * @TN: The name of the new type, in Camel case.
223 * @t_n: The name of the new type in lowercase, with words separated by '_'.
224 * @T_P: The #GType of the parent type.
225 * @_C_: Custom code that gets inserted in the _get_type() function.
226 *
227 * A convenience macro for ATK type implementations.
228 * Similar to ATK_DEFINE_TYPE(), but allows you to insert custom code into the
229 * _get_type() function, e.g. interface implementations via G_IMPLEMENT_INTERFACE().
230 *
231 * Since: 1.22
232 */
233#define ATK_DEFINE_TYPE_WITH_CODE(TN, t_n, T_P, _C_) _ATK_DEFINE_TYPE_EXTENDED_BEGIN (TN, t_n, T_P, 0) {_C_;} _ATK_DEFINE_TYPE_EXTENDED_END()

262/**
263 * ATK_DEFINE_TYPE_EXTENDED:
264 * @TN: The name of the new type, in Camel case.
265 * @t_n: The name of the new type, in lowercase, with words separated by '_'.
266 * @T_P: The #GType of the parent type.
267 * @_f_: #GTypeFlags to pass to g_type_register_static()
268 * @_C_: Custom code that gets inserted in the _get_type() function.
269 *
270 * The most general convenience macro for ATK type implementations, on which
271 * ATK_DEFINE_TYPE(), etc are based.
272 *
273 * Since: 1.22
274 */
275#define ATK_DEFINE_TYPE_EXTENDED(TN, t_n, T_P, _f_, _C_) _ATK_DEFINE_TYPE_EXTENDED_BEGIN (TN, t_n, T_P, _f_) {_C_;} _ATK_DEFINE_TYPE_EXTENDED_END()
276
277#define _ATK_DEFINE_TYPE_EXTENDED_BEGIN(TypeName, type_name, TYPE, flags) \
278\
279static void type_name##_init (TypeName *self); \
280static void type_name##_class_init (TypeName##Class *klass); \
281static gpointer type_name##_parent_class = NULL; \
282static void type_name##_class_intern_init (gpointer klass) \
283{ \
284 type_name##_parent_class = g_type_class_peek_parent (klass); \
285 type_name##_class_init ((TypeName##Class*) klass); \
286} \
287\
288GType \
289type_name##_get_type (void) \
290{ \
291 static volatile gsize g_define_type_id__volatile = 0; \
292 if (g_once_init_enter (&g_define_type_id__volatile)) \
293 { \
294 AtkObjectFactory *factory; \
295 GType derived_type; \
296 GTypeQuery query; \
297 GType derived_atk_type; \
298 GType g_define_type_id; \
299\
300 /* Figure out the size of the class and instance we are deriving from */ \
301 derived_type = g_type_parent (TYPE); \
302 factory = atk_registry_get_factory (atk_get_default_registry (), \
303 derived_type); \
304 derived_atk_type = atk_object_factory_get_accessible_type (factory); \
305 g_type_query (derived_atk_type, &query); \
306\
307 g_define_type_id = \
308 g_type_register_static_simple (derived_atk_type, \
309 g_intern_static_string (#TypeName), \
310 query.class_size, \
311 (GClassInitFunc) type_name##_class_intern_init, \
312 query.instance_size, \
313 (GInstanceInitFunc) type_name##_init, \
314 (GTypeFlags) flags); \
315 { /* custom code follows */
316#define _ATK_DEFINE_TYPE_EXTENDED_END() \
317 /* following custom code */ \
318 } \
319 g_once_init_leave (&g_define_type_id__volatile, g_define_type_id); \
320 } \
321 return g_define_type_id__volatile; \
322} /* closes type_name##_get_type() */
