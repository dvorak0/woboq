20#ifndef __GTK_ACCESSIBLE_H__
21#define __GTK_ACCESSIBLE_H__

27#include <atk/atk.h>
28#include <gtk/gtkwidget.h>
29
30G_BEGIN_DECLS
31
32#define GTK_TYPE_ACCESSIBLE (gtk_accessible_get_type ())
33#define GTK_ACCESSIBLE(obj) (G_TYPE_CHECK_INSTANCE_CAST ((obj), GTK_TYPE_ACCESSIBLE, GtkAccessible))
34#define GTK_ACCESSIBLE_CLASS(klass) (G_TYPE_CHECK_CLASS_CAST ((klass), GTK_TYPE_ACCESSIBLE, GtkAccessibleClass))
35#define GTK_IS_ACCESSIBLE(obj) (G_TYPE_CHECK_INSTANCE_TYPE ((obj), GTK_TYPE_ACCESSIBLE))
36#define GTK_IS_ACCESSIBLE_CLASS(klass) (G_TYPE_CHECK_CLASS_TYPE ((klass), GTK_TYPE_ACCESSIBLE))
37#define GTK_ACCESSIBLE_GET_CLASS(obj) (G_TYPE_INSTANCE_GET_CLASS ((obj), GTK_TYPE_ACCESSIBLE, GtkAccessibleClass))
38
39typedef struct _GtkAccessible GtkAccessible;
40typedef struct _GtkAccessibleClass GtkAccessibleClass;
41
42 /*
43 * This object is a thin wrapper, in the GTK+ namespace, for AtkObject
44 */
45struct _GtkAccessible
46{
47 AtkObject parent;
48
49 /*
50 * The GtkWidget whose properties and features are exported via this
51 * accessible instance.
52 */
53 GtkWidget *GSEAL (widget);
54};
55
56struct _GtkAccessibleClass
57{
58 AtkObjectClass parent_class;
59
60 void (*connect_widget_destroyed) (GtkAccessible *accessible);
61
62 /* Padding for future expansion */
63 void (*_gtk_reserved1) (void);
64 void (*_gtk_reserved2) (void);
65 void (*_gtk_reserved3) (void);
66 void (*_gtk_reserved4) (void);
67};
68
69GType gtk_accessible_get_type (void) G_GNUC_CONST;
70
71void gtk_accessible_set_widget (GtkAccessible *accessible,
72 GtkWidget *widget);
73GtkWidget* gtk_accessible_get_widget (GtkAccessible *accessible);
74void gtk_accessible_connect_widget_destroyed (GtkAccessible *accessible);
75
76G_END_DECLS
77
78#endif /* __GTK_ACCESSIBLE_H__ */
