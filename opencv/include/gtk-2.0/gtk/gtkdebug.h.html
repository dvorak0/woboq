38typedef enum {
39 GTK_DEBUG_MISC = 1 << 0,
40 GTK_DEBUG_PLUGSOCKET = 1 << 1,
41 GTK_DEBUG_TEXT = 1 << 2,
42 GTK_DEBUG_TREE = 1 << 3,
43 GTK_DEBUG_UPDATES = 1 << 4,
44 GTK_DEBUG_KEYBINDINGS = 1 << 5,
45 GTK_DEBUG_MULTIHEAD = 1 << 6,
46 GTK_DEBUG_MODULES = 1 << 7,
47 GTK_DEBUG_GEOMETRY = 1 << 8,
48 GTK_DEBUG_ICONTHEME = 1 << 9,
49 GTK_DEBUG_PRINTING = 1 << 10,
50 GTK_DEBUG_BUILDER = 1 << 11
51} GtkDebugFlag;
52
53#ifdef G_ENABLE_DEBUG
54
55#define GTK_NOTE(type,action) G_STMT_START { \
56 if (gtk_debug_flags & GTK_DEBUG_##type) \
57 { action; }; } G_STMT_END
58
59#else /* !G_ENABLE_DEBUG */
60
61#define GTK_NOTE(type, action)
62
63#endif /* G_ENABLE_DEBUG */
