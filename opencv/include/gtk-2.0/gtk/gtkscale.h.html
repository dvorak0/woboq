48typedef struct _GtkScale GtkScale;
49typedef struct _GtkScaleClass GtkScaleClass;
50
51struct _GtkScale
52{
53 GtkRange range;
54
55 gint GSEAL (digits);
56 guint GSEAL (draw_value) : 1;
57 guint GSEAL (value_pos) : 2;
58};
59
60struct _GtkScaleClass
61{
62 GtkRangeClass parent_class;
63
64 gchar* (* format_value) (GtkScale *scale,
65 gdouble value);
66
67 void (* draw_value) (GtkScale *scale);
68
69 void (* get_layout_offsets) (GtkScale *scale,
70 gint *x,
71 gint *y);
72
73 /* Padding for future expansion */
74 void (*_gtk_reserved1) (void);
75 void (*_gtk_reserved2) (void);
76 void (*_gtk_reserved3) (void);
77};
