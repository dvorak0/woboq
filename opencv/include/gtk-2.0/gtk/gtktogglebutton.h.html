48typedef struct _GtkToggleButton GtkToggleButton;

51struct _GtkToggleButton
52{
53 GtkButton button;
54
55 guint GSEAL (active) : 1;
56 guint GSEAL (draw_indicator) : 1;
57 guint GSEAL (inconsistent) : 1;
58};

61{

63

67 void (*_gtk_reserved1) (void);
68 void (*_gtk_reserved2) (void);
69 void (*_gtk_reserved3) (void);
70 void (*_gtk_reserved4) (void);
71};
