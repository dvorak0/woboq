48typedef struct _GtkMisc GtkMisc;
49typedef struct _GtkMiscClass GtkMiscClass;

51struct _GtkMisc
52{
53 GtkWidget widget;
54
55 gfloat GSEAL (xalign);
56 gfloat GSEAL (yalign);
57
58 guint16 GSEAL (xpad);
59 guint16 GSEAL (ypad);
60};
61
62struct _GtkMiscClass
63{
64 GtkWidgetClass parent_class;
65};
