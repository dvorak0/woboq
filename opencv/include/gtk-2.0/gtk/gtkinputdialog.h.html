56typedef struct _GtkInputDialog GtkInputDialog;
57typedef struct _GtkInputDialogClass GtkInputDialogClass;
58
59struct _GtkInputDialog
60{
61 GtkDialog dialog;
62
63 GtkWidget *GSEAL (axis_list);
64 GtkWidget *GSEAL (axis_listbox);
65 GtkWidget *GSEAL (mode_optionmenu);
66
67 GtkWidget *GSEAL (close_button);
68 GtkWidget *GSEAL (save_button);
69
70 GtkWidget *GSEAL (axis_items[GDK_AXIS_LAST]);
71 GdkDevice *GSEAL (current_device);
72
73 GtkWidget *GSEAL (keys_list);
74 GtkWidget *GSEAL (keys_listbox);
75};
76
77struct _GtkInputDialogClass
78{
79 GtkDialogClass parent_class;
80
81 void (* enable_device) (GtkInputDialog *inputd,
82 GdkDevice *device);
83 void (* disable_device) (GtkInputDialog *inputd,
84 GdkDevice *device);
85
86 /* Padding for future expansion */
87 void (*_gtk_reserved1) (void);
88 void (*_gtk_reserved2) (void);
89 void (*_gtk_reserved3) (void);
90 void (*_gtk_reserved4) (void);
91};
