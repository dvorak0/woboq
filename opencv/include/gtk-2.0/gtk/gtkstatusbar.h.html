39#define GTK_TYPE_STATUSBAR (gtk_statusbar_get_type ())
40#define GTK_STATUSBAR(obj) (G_TYPE_CHECK_INSTANCE_CAST ((obj), GTK_TYPE_STATUSBAR, GtkStatusbar))
41#define GTK_STATUSBAR_CLASS(klass) (G_TYPE_CHECK_CLASS_CAST ((klass), GTK_TYPE_STATUSBAR, GtkStatusbarClass))
42#define GTK_IS_STATUSBAR(obj) (G_TYPE_CHECK_INSTANCE_TYPE ((obj), GTK_TYPE_STATUSBAR))
43#define GTK_IS_STATUSBAR_CLASS(klass) (G_TYPE_CHECK_CLASS_TYPE ((klass), GTK_TYPE_STATUSBAR))
44#define GTK_STATUSBAR_GET_CLASS(obj) (G_TYPE_INSTANCE_GET_CLASS ((obj), GTK_TYPE_STATUSBAR, GtkStatusbarClass))
45
46
47typedef struct _GtkStatusbar GtkStatusbar;
48typedef struct _GtkStatusbarClass GtkStatusbarClass;
49
50struct _GtkStatusbar
51{
52 GtkHBox parent_widget;
53
54 GtkWidget *GSEAL (frame);
55 GtkWidget *GSEAL (label);
56
57 GSList *GSEAL (messages);
58 GSList *GSEAL (keys);
59
60 guint GSEAL (seq_context_id);
61 guint GSEAL (seq_message_id);
62
63 GdkWindow *GSEAL (grip_window);
64
65 guint GSEAL (has_resize_grip) : 1;
66};
67
68struct _GtkStatusbarClass
69{
70 GtkHBoxClass parent_class;
71
72 gpointer reserved;
73
74 void (*text_pushed) (GtkStatusbar *statusbar,
75 guint context_id,
76 const gchar *text);
77 void (*text_popped) (GtkStatusbar *statusbar,
78 guint context_id,
79 const gchar *text);
80
81 /* Padding for future expansion */
82 void (*_gtk_reserved1) (void);
83 void (*_gtk_reserved2) (void);
84 void (*_gtk_reserved3) (void);
85 void (*_gtk_reserved4) (void);
86};
87
88
89GType gtk_statusbar_get_type (void) G_GNUC_CONST;
90GtkWidget* gtk_statusbar_new (void);
91/* If you don't want to use contexts, 0 is a predefined global
92 * context_id you can pass to push/pop/remove
93 */
94guint gtk_statusbar_get_context_id (GtkStatusbar *statusbar,
95 const gchar *context_description);
96/* Returns message_id used for gtk_statusbar_remove */
97guint gtk_statusbar_push (GtkStatusbar *statusbar,
98 guint context_id,
99 const gchar *text);
100void gtk_statusbar_pop (GtkStatusbar *statusbar,
101 guint context_id);
102void gtk_statusbar_remove (GtkStatusbar *statusbar,
103 guint context_id,
104 guint message_id);
105void gtk_statusbar_remove_all (GtkStatusbar *statusbar,
106 guint context_id);
107
108
109void gtk_statusbar_set_has_resize_grip (GtkStatusbar *statusbar,
110 gboolean setting);
111gboolean gtk_statusbar_get_has_resize_grip (GtkStatusbar *statusbar);

113GtkWidget* gtk_statusbar_get_message_area (GtkStatusbar *statusbar);
