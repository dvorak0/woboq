32#define GTK_TYPE_FILE_FILTER (gtk_file_filter_get_type ())
33#define GTK_FILE_FILTER(obj) (G_TYPE_CHECK_INSTANCE_CAST ((obj), GTK_TYPE_FILE_FILTER, GtkFileFilter))
34#define GTK_IS_FILE_FILTER(obj) (G_TYPE_CHECK_INSTANCE_TYPE ((obj), GTK_TYPE_FILE_FILTER))

36typedef struct _GtkFileFilter GtkFileFilter;
37typedef struct _GtkFileFilterInfo GtkFileFilterInfo;
38
39typedef enum {
40 GTK_FILE_FILTER_FILENAME = 1 << 0,
41 GTK_FILE_FILTER_URI = 1 << 1,
42 GTK_FILE_FILTER_DISPLAY_NAME = 1 << 2,
43 GTK_FILE_FILTER_MIME_TYPE = 1 << 3
44} GtkFileFilterFlags;
45
46typedef gboolean (*GtkFileFilterFunc) (const GtkFileFilterInfo *filter_info,
47 gpointer data);
48
49struct _GtkFileFilterInfo
50{
51 GtkFileFilterFlags contains;
52
53 const gchar *filename;
54 const gchar *uri;
55 const gchar *display_name;
56 const gchar *mime_type;
57};

61GtkFileFilter * gtk_file_filter_new (void);

63 const gchar *name);

65
66void gtk_file_filter_add_mime_type (GtkFileFilter *filter,
67 const gchar *mime_type);
68void gtk_file_filter_add_pattern (GtkFileFilter *filter,
69 const gchar *pattern);
70void gtk_file_filter_add_pixbuf_formats (GtkFileFilter *filter);
71void gtk_file_filter_add_custom (GtkFileFilter *filter,
72 GtkFileFilterFlags needed,
73 GtkFileFilterFunc func,
74 gpointer data,
75 GDestroyNotify notify);
76
77GtkFileFilterFlags gtk_file_filter_get_needed (GtkFileFilter *filter);
78gboolean gtk_file_filter_filter (GtkFileFilter *filter,
