42typedef struct _GtkExpander GtkExpander;
43typedef struct _GtkExpanderClass GtkExpanderClass;
44typedef struct _GtkExpanderPrivate GtkExpanderPrivate;

46struct _GtkExpander
47{
48 GtkBin bin;
49
50 GtkExpanderPrivate *GSEAL (priv);
51};

53struct _GtkExpanderClass
54{
55 GtkBinClass parent_class;
56
57 /* Key binding signal; to get notification on the expansion
58 * state connect to notify:expanded.
59 */
60 void (* activate) (GtkExpander *expander);
61};
