52typedef struct _GtkScrolledWindow GtkScrolledWindow;
53typedef struct _GtkScrolledWindowClass GtkScrolledWindowClass;
54
55struct _GtkScrolledWindow
56{
57 GtkBin container;
58
59 /*< public >*/
60 GtkWidget *GSEAL (hscrollbar);
61 GtkWidget *GSEAL (vscrollbar);
62
63 /*< private >*/
64 guint GSEAL (hscrollbar_policy) : 2;
65 guint GSEAL (vscrollbar_policy) : 2;
66 guint GSEAL (hscrollbar_visible) : 1;
67 guint GSEAL (vscrollbar_visible) : 1;
68 guint GSEAL (window_placement) : 2;
69 guint GSEAL (focus_out) : 1; /* Flag used by ::move-focus-out implementation */
70
71 guint16 GSEAL (shadow_type);
72};
73
74struct _GtkScrolledWindowClass
75{
76 GtkBinClass parent_class;
77
78 gint scrollbar_spacing;
79
80 /* Action signals for keybindings. Do not connect to these signals
81 */
82
83 /* Unfortunately, GtkScrollType is deficient in that there is
84 * no horizontal/vertical variants for GTK_SCROLL_START/END,
85 * so we have to add an additional boolean flag.
86 */
87 gboolean (*scroll_child) (GtkScrolledWindow *scrolled_window,
88 GtkScrollType scroll,
89 gboolean horizontal);
90
91 void (* move_focus_out) (GtkScrolledWindow *scrolled_window,
92 GtkDirectionType direction);
93
94 /* Padding for future expansion */
95 void (*_gtk_reserved1) (void);
96 void (*_gtk_reserved2) (void);
97 void (*_gtk_reserved3) (void);
98 void (*_gtk_reserved4) (void);
99};
