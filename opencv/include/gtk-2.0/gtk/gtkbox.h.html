67/**
68 * GtkBoxChild:
69 * @widget: the child widget, packed into the GtkBox.
70 * @padding: the number of extra pixels to put between this child and its
71 * neighbors, set when packed, zero by default.
72 * @expand: flag indicates whether extra space should be given to this child.
73 * Any extra space given to the parent GtkBox is divided up among all children
74 * with this attribute set to %TRUE; set when packed, %TRUE by default.
75 * @fill: flag indicates whether any extra space given to this child due to its
76 * @expand attribute being set is actually allocated to the child, rather than
77 * being used as padding around the widget; set when packed, %TRUE by default.
78 * @pack: one of #GtkPackType indicating whether the child is packed with
79 * reference to the start (top/left) or end (bottom/right) of the GtkBox.
80 * @is_secondary: %TRUE if the child is secondary
81 *
82 * The #GtkBoxChild holds a child widget of #GtkBox and describes how the child
83 * is to be packed into the #GtkBox. All fields of this #GtkBoxChild should be
84 * considered read-only and they should never be set directly by an application.
85 * Use gtk_box_query_child_packing() and gtk_box_set_child_packing() to query
86 * and set the #GtkBoxChild.padding, #GtkBoxChild.expand, #GtkBoxChild.fill and
87 * #GtkBoxChild.pack fields.
88 *
89 * Deprecated: 2.22: Use gtk_container_get_children() instead.
90 */
91#if !defined (GTK_DISABLE_DEPRECATED) || defined (GTK_COMPILATION)
92typedef struct _GtkBoxChild GtkBoxChild;
93struct _GtkBoxChild
94{
95 GtkWidget *widget;
96 guint16 padding;
97 guint expand : 1;
98 guint fill : 1;
99 guint pack : 1;
100 guint is_secondary : 1;
101};
102#endif
