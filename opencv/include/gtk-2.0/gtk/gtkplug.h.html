48typedef struct _GtkPlug GtkPlug;

52struct _GtkPlug
53{
54 GtkWindow window;
55
56 GdkWindow *GSEAL (socket_window);
57 GtkWidget *GSEAL (modality_window);
58 GtkWindowGroup *GSEAL (modality_group);
59 GHashTable *GSEAL (grabbed_keys);
60
61 guint GSEAL (same_app) : 1;
62};

64struct _GtkPlugClass
65{
66 GtkWindowClass parent_class;
67
68 void (*embedded) (GtkPlug *plug);

70 /* Padding for future expansion */

72 void (*_gtk_reserved2) (void);
73 void (*_gtk_reserved3) (void);
74 void (*_gtk_reserved4) (void);
75};
