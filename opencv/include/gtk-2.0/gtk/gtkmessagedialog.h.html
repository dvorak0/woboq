71/**
72 * GtkButtonsType:
73 * @GTK_BUTTONS_NONE: no buttons at all
74 * @GTK_BUTTONS_OK: an OK button
75 * @GTK_BUTTONS_CLOSE: a Close button
76 * @GTK_BUTTONS_CANCEL: a Cancel button
77 * @GTK_BUTTONS_YES_NO: Yes and No buttons
78 * @GTK_BUTTONS_OK_CANCEL: OK and Cancel buttons
79 *
80 * Prebuilt sets of buttons for the dialog. If
81 * none of these choices are appropriate, simply use %GTK_BUTTONS_NONE
82 * then call gtk_dialog_add_buttons().
83 * <note>
84 * Please note that %GTK_BUTTONS_OK, %GTK_BUTTONS_YES_NO
85 * and %GTK_BUTTONS_OK_CANCEL are discouraged by the
86 * <ulink url="http://library.gnome.org/devel/hig-book/stable/">GNOME HIG</ulink>.
87 * </note>
88 */
89typedef enum
90{
91 GTK_BUTTONS_NONE,
92 GTK_BUTTONS_OK,
93 GTK_BUTTONS_CLOSE,
94 GTK_BUTTONS_CANCEL,
95 GTK_BUTTONS_YES_NO,
96 GTK_BUTTONS_OK_CANCEL
97} GtkButtonsType;
