50typedef struct _GtkViewport GtkViewport;
51typedef struct _GtkViewportClass GtkViewportClass;
52
53struct _GtkViewport
54{
55 GtkBin bin;
56
57 GtkShadowType GSEAL (shadow_type);
58 GdkWindow *GSEAL (view_window);
59 GdkWindow *GSEAL (bin_window);
60 GtkAdjustment *GSEAL (hadjustment);
61 GtkAdjustment *GSEAL (vadjustment);
62};
63
64struct _GtkViewportClass
65{
66 GtkBinClass parent_class;
67
68 void (*set_scroll_adjustments) (GtkViewport *viewport,
69 GtkAdjustment *hadjustment,
70 GtkAdjustment *vadjustment);
71};
