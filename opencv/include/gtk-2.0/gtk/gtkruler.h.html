60typedef struct _GtkRuler GtkRuler;
61typedef struct _GtkRulerClass GtkRulerClass;
62typedef struct _GtkRulerMetric GtkRulerMetric;
63
64/* All distances below are in 1/72nd's of an inch. (According to
65 * Adobe that's a point, but points are really 1/72.27 in.)
66 */
67struct _GtkRuler
68{
69 GtkWidget widget;
70
71 GdkPixmap *GSEAL (backing_store);
72 GdkGC *GSEAL (non_gr_exp_gc); /* unused */
73 GtkRulerMetric *GSEAL (metric);
74 gint GSEAL (xsrc);
75 gint GSEAL (ysrc);
76 gint GSEAL (slider_size);
77
78 /* The upper limit of the ruler (in points) */
79 gdouble GSEAL (lower);
80 /* The lower limit of the ruler */
81 gdouble GSEAL (upper);
82 /* The position of the mark on the ruler */
83 gdouble GSEAL (position);
84 /* The maximum size of the ruler */
85 gdouble GSEAL (max_size);
86};
87
88struct _GtkRulerClass
89{
90 GtkWidgetClass parent_class;
91
92 void (* draw_ticks) (GtkRuler *ruler);
93 void (* draw_pos) (GtkRuler *ruler);
94
95 /* Padding for future expansion */
96 void (*_gtk_reserved1) (void);
97 void (*_gtk_reserved2) (void);
98 void (*_gtk_reserved3) (void);
99 void (*_gtk_reserved4) (void);
100};
101
102struct _GtkRulerMetric
103{
104 gchar *metric_name;
105 gchar *abbrev;
106 /* This should be points_per_unit. This is the size of the unit
107 * in 1/72nd's of an inch and has nothing to do with screen pixels */
108 gdouble pixels_per_unit;
109 gdouble ruler_scale[10];
110 gint subdivide[5]; /* five possible modes of subdivision */
111};
