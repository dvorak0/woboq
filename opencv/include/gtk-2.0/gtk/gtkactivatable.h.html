39typedef struct _GtkActivatable GtkActivatable; /* Dummy typedef */
40typedef struct _GtkActivatableIface GtkActivatableIface;
41
42
43/**
44 * GtkActivatableIface:
45 * @update: Called to update the activatable when its related action's properties change.
46 * You must check the #GtkActivatable:use-action-appearance property only apply action
47 * properties that are meant to effect the appearance accordingly.
48 * @sync_action_properties: Called to update the activatable completely, this is called internally when
49 * #GtkActivatable::related-action property is set or unset and by the implementor when
50 * #GtkActivatable::use-action-appearance changes.<note><para>This method can be called
51 * with a %NULL action at times</para></note>
52 *
53 * Since: 2.16
54 */
55
56struct _GtkActivatableIface
57{
58 GTypeInterface g_iface;
59
60 /* virtual table */
61 void (* update) (GtkActivatable *activatable,
62 GtkAction *action,
63 const gchar *property_name);
64 void (* sync_action_properties) (GtkActivatable *activatable,
65 GtkAction *action);
66};
