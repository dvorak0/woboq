39//-----------------------------------------------------------------------------
40//
41// Optional Standard Attributes -- these attributes are "optional"
42// because not every image file header has them, but they define a
43// "standard" way to represent commonly used data in the file header.
44//
45// For each attribute, with name "foo", and type "T", the following
46// functions are automatically generated via macros:
47//
48// void addFoo (Header &header, const T &value);
49// bool hasFoo (const Header &header);
50// const TypedAttribute<T> & fooAttribute (const Header &header);
51// TypedAttribute<T> & fooAttribute (Header &header);
52// const T & foo (const Header &Header);
53// T & foo (Header &Header);
54//
55//-----------------------------------------------------------------------------
56

68#define IMF_STD_ATTRIBUTE_DEF(name,suffix,type) \
69 \
70 void add##suffix (Header &header, const type &v); \
71 bool has##suffix (const Header &header); \
72 const TypedAttribute<type> & name##Attribute (const Header &header); \
73 TypedAttribute<type> & name##Attribute (Header &header); \
74 const type & name (const Header &header); \
75 type & name (Header &header);

240//
241// wrapmodes -- determines how texture map images are extrapolated.
242// If an OpenEXR file is used as a texture map for 3D rendering,
243// texture coordinates (0.0, 0.0) and (1.0, 1.0) correspond to
244// the upper left and lower right corners of the data window.
245// If the image is mapped onto a surface with texture coordinates
246// outside the zero-to-one range, then the image must be extrapolated.
247// This attribute tells the renderer how to do this extrapolation.
248// The attribute contains either a pair of comma-separated keywords,
249// to specify separate extrapolation modes for the horizontal and
250// vertical directions; or a single keyword, to specify extrapolation
251// in both directions (e.g. "clamp,periodic" or "clamp"). Extra white
252// space surrounding the keywords is allowed, but should be ignored
253// by the renderer ("clamp, black " is equivalent to "clamp,black").
254// The keywords listed below are predefined; some renderers may support
255// additional extrapolation modes:
256//
257// black pixels outside the zero-to-one range are black
258//
259// clamp texture coordinates less than 0.0 and greater
260// than 1.0 are clamped to 0.0 and 1.0 respectively
261//
262// periodic the texture image repeats periodically
263//
264// mirror the texture image repeats periodically, but
265// every other instance is mirrored
266//
267
268IMF_STD_ATTRIBUTE_DEF (wrapmodes, Wrapmodes, std::string)
269
270
271//
272// framesPerSecond -- defines the nominal playback frame rate for image
273// sequences, in frames per second. Every image in a sequence should
274// have a framesPerSecond attribute, and the attribute value should be
275// the same for all images in the sequence. If an image sequence has
276// no framesPerSecond attribute, playback software should assume that
277// the frame rate for the sequence is 24 frames per second.
278//
279// In order to allow exact representation of NTSC frame and field rates,
280// framesPerSecond is stored as a rational number. A rational number is
281// a pair of integers, n and d, that represents the value n/d.
282//
283// For the exact values of commonly used frame rates, please see header
284// file ImfFramesPerSecond.h.
285//
286
287IMF_STD_ATTRIBUTE_DEF (framesPerSecond, FramesPerSecond, Rational)
