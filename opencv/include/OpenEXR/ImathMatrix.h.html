1317template <class T>
1318Matrix33<T>
1319Matrix33<T>::gjInverse (bool singExc) const throw (Iex::MathExc)
1320{
1321 int i, j, k;
1322 Matrix33 s;
1323 Matrix33 t (*this);
1324
1325 // Forward elimination
1326
1327 for (i = 0; i < 2 ; i++)
1328 {
1329 int pivot = i;
1330
1331 T pivotsize = t[i][i];
1332
1333 if (pivotsize < 0)
1334 pivotsize = -pivotsize;
1335
1336 for (j = i + 1; j < 3; j++)
1337 {
1338 T tmp = t[j][i];
1339
1340 if (tmp < 0)
1341 tmp = -tmp;
1342
1343 if (tmp > pivotsize)
1344 {
1345 pivot = j;
1346 pivotsize = tmp;
1347 }
1348 }
1349
1350 if (pivotsize == 0)
1351 {
1352 if (singExc)
1353 throw ::Imath::SingMatrixExc ("Cannot invert singular matrix.");
1354
1355 return Matrix33();
1356 }
1357
1358 if (pivot != i)
1359 {
1360 for (j = 0; j < 3; j++)
1361 {
1362 T tmp;
1363
1364 tmp = t[i][j];
1365 t[i][j] = t[pivot][j];
1366 t[pivot][j] = tmp;
1367
1368 tmp = s[i][j];
1369 s[i][j] = s[pivot][j];
1370 s[pivot][j] = tmp;
1371 }
1372 }
1373
1374 for (j = i + 1; j < 3; j++)
1375 {
1376 T f = t[j][i] / t[i][i];
1377
1378 for (k = 0; k < 3; k++)
1379 {
1380 t[j][k] -= f * t[i][k];
1381 s[j][k] -= f * s[i][k];
1382 }
1383 }
1384 }
1385
1386 // Backward substitution
1387
1388 for (i = 2; i >= 0; --i)
1389 {
1390 T f;
1391
1392 if ((f = t[i][i]) == 0)
1393 {
1394 if (singExc)
1395 throw ::Imath::SingMatrixExc ("Cannot invert singular matrix.");
1396
1397 return Matrix33();
1398 }
1399
1400 for (j = 0; j < 3; j++)
1401 {
1402 t[i][j] /= f;
1403 s[i][j] /= f;
1404 }
1405
1406 for (j = 0; j < i; j++)
1407 {
1408 f = t[j][i];
1409
1410 for (k = 0; k < 3; k++)
1411 {
1412 t[j][k] -= f * t[i][k];

1414 }
1415 }
1416 }
1417
1418 return s;
1419}
