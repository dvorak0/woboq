39//-----------------------------------------------------------------------------
40//
41// class TileDescription and enum LevelMode
42//
43//-----------------------------------------------------------------------------
44
45namespace Imf {
46
47
48enum LevelMode
49{
50 ONE_LEVEL = 0,
51 MIPMAP_LEVELS = 1,
52 RIPMAP_LEVELS = 2,
53
54 NUM_LEVELMODES // number of different level modes
55};
56
57
58enum LevelRoundingMode
59{
60 ROUND_DOWN = 0,
61 ROUND_UP = 1,
62
63 NUM_ROUNDINGMODES // number of different rounding modes
64};
65
66
67class TileDescription
68{
69 public:
70
71 unsigned int xSize; // size of a tile in the x dimension
72 unsigned int ySize; // size of a tile in the y dimension
73 LevelMode mode;
74 LevelRoundingMode roundingMode;
75
76 TileDescription (unsigned int xs = 32,
77 unsigned int ys = 32,
78 LevelMode m = ONE_LEVEL,
79 LevelRoundingMode r = ROUND_DOWN)
80 :
81 xSize (xs),
82 ySize (ys),
83 mode (m),
84 roundingMode (r)
85 {
86 // empty
87 }
88
89 bool
90 operator == (const TileDescription &other) const
91 {
92 return xSize == other.xSize &&
93 ySize == other.ySize &&
94 mode == other.mode &&
95 roundingMode == other.roundingMode;
96 }
97};
98
99
100} // namespace Imf
