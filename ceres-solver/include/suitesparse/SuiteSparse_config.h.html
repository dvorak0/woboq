5/* Configuration file for SuiteSparse: a Suite of Sparse matrix packages
6 * (AMD, COLAMD, CCOLAMD, CAMD, CHOLMOD, UMFPACK, CXSparse, and others).
7 *
8 * SuiteSparse_config.h provides the definition of the long integer. On most
9 * systems, a C program can be compiled in LP64 mode, in which long's and
10 * pointers are both 64-bits, and int's are 32-bits. Windows 64, however, uses
11 * the LLP64 model, in which int's and long's are 32-bits, and long long's and
12 * pointers are 64-bits.
13 *
14 * SuiteSparse packages that include long integer versions are
15 * intended for the LP64 mode. However, as a workaround for Windows 64
16 * (and perhaps other systems), the long integer can be redefined.
17 *
18 * If _WIN64 is defined, then the __int64 type is used instead of long.
19 *
20 * The long integer can also be defined at compile time. For example, this
21 * could be added to SuiteSparse_config.mk:
22 *
23 * CFLAGS = -O -D'SuiteSparse_long=long long' \
24 * -D'SuiteSparse_long_max=9223372036854775801' -D'SuiteSparse_long_idd="lld"'
25 *
26 * This file defines SuiteSparse_long as either long (on all but _WIN64) or
27 * __int64 on Windows 64. The intent is that a SuiteSparse_long is always a
28 * 64-bit integer in a 64-bit code. ptrdiff_t might be a better choice than
29 * long; it is always the same size as a pointer.
30 *
31 * This file also defines the SUITESPARSE_VERSION and related definitions.
32 *
33 * Copyright (c) 2012, Timothy A. Davis. No licensing restrictions apply
34 * to this file or to the SuiteSparse_config directory.
35 * Author: Timothy A. Davis.
36 */
37

52#ifndef SuiteSparse_long
53
54#ifdef _WIN64
55
56#define SuiteSparse_long __int64
57#define SuiteSparse_long_max _I64_MAX
58#define SuiteSparse_long_idd "I64d"
59
60#else
61
62#define SuiteSparse_long long
63#define SuiteSparse_long_max LONG_MAX
64#define SuiteSparse_long_idd "ld"
65
66#endif
67#define SuiteSparse_long_id "%" SuiteSparse_long_idd
68#endif

133/* === SuiteSparse version ================================================== */
134/* ========================================================================== */
135
136/* SuiteSparse is not a package itself, but a collection of packages, some of
137 * which must be used together (UMFPACK requires AMD, CHOLMOD requires AMD,
138 * COLAMD, CAMD, and CCOLAMD, etc). A version number is provided here for the
139 * collection itself. The versions of packages within each version of
140 * SuiteSparse are meant to work together. Combining one packge from one
141 * version of SuiteSparse, with another package from another version of
142 * SuiteSparse, may or may not work.
143 *
144 * SuiteSparse contains the following packages:
145 *
146 * SuiteSparse_config version 4.2.1 (version always the same as SuiteSparse)
147 * AMD version 2.3.1
148 * BTF version 1.2.0
149 * CAMD version 2.3.1
150 * CCOLAMD version 2.8.0
151 * CHOLMOD version 2.1.2
152 * COLAMD version 2.8.0
153 * CSparse version 3.1.2
154 * CXSparse version 3.1.2
155 * KLU version 1.2.1
156 * LDL version 2.1.0
157 * RBio version 2.1.1
158 * SPQR version 1.3.1 (full name is SuiteSparseQR)
159 * UMFPACK version 5.6.2
160 * MATLAB_Tools various packages & M-files
161 *
162 * Other package dependencies:
163 * BLAS required by CHOLMOD and UMFPACK
164 * LAPACK required by CHOLMOD
165 * METIS 4.0.1 required by CHOLMOD (optional) and KLU (optional)
166 */
167
168
169int SuiteSparse_version /* returns SUITESPARSE_VERSION */
170(
171 /* output, not defined on input. Not used if NULL. Returns
172 the three version codes in version [0..2]:
173 version [0] is SUITESPARSE_MAIN_VERSION
174 version [1] is SUITESPARSE_SUB_VERSION
175 version [2] is SUITESPARSE_SUBSUB_VERSION
176 */
177 int version [3]
178) ;
179
180/* Versions prior to 4.2.0 do not have the above function. The following
181 code fragment will work with any version of SuiteSparse:
182
183 #ifdef SUITESPARSE_HAS_VERSION_FUNCTION
184 v = SuiteSparse_version (NULL) ;
185 #else
186 v = SUITESPARSE_VERSION ;
187 #endif
188*/
189#define SUITESPARSE_HAS_VERSION_FUNCTION
190
191#define SUITESPARSE_DATE "April 25, 2013"
192#define SUITESPARSE_VER_CODE(main,sub) ((main) * 1000 + (sub))
193#define SUITESPARSE_MAIN_VERSION 4
194#define SUITESPARSE_SUB_VERSION 2
195#define SUITESPARSE_SUBSUB_VERSION 1
196#define SUITESPARSE_VERSION \
197 SUITESPARSE_VER_CODE(SUITESPARSE_MAIN_VERSION,SUITESPARSE_SUB_VERSION)
198

201#endif
