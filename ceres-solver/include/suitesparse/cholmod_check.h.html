13/* CHOLMOD Check module.
14 *
15 * Routines that check and print the 5 basic data types in CHOLMOD, and 3 kinds
16 * of integer vectors (subset, perm, and parent), and read in matrices from a
17 * file:
18 *
19 * cholmod_check_common check/print the Common object
20 * cholmod_print_common
21 *
22 * cholmod_check_sparse check/print a sparse matrix in column-oriented form
23 * cholmod_print_sparse
24 *
25 * cholmod_check_dense check/print a dense matrix
26 * cholmod_print_dense
27 *
28 * cholmod_check_factor check/print a Cholesky factorization
29 * cholmod_print_factor
30 *
31 * cholmod_check_triplet check/print a sparse matrix in triplet form
32 * cholmod_print_triplet
33 *
34 * cholmod_check_subset check/print a subset (integer vector in given range)
35 * cholmod_print_subset
36 *
37 * cholmod_check_perm check/print a permutation (an integer vector)
38 * cholmod_print_perm
39 *
40 * cholmod_check_parent check/print an elimination tree (an integer vector)
41 * cholmod_print_parent
42 *
43 * cholmod_read_triplet read a matrix in triplet form (any Matrix Market
44 * "coordinate" format, or a generic triplet format).
45 *
46 * cholmod_read_sparse read a matrix in sparse form (same file format as
47 * cholmod_read_triplet).
48 *
49 * cholmod_read_dense read a dense matrix (any Matrix Market "array"
50 * format, or a generic dense format).
51 *
52 * cholmod_write_sparse write a sparse matrix to a Matrix Market file.
53 *
54 * cholmod_write_dense write a dense matrix to a Matrix Market file.
55 *
56 * cholmod_print_common and cholmod_check_common are the only two routines that
57 * you may call after calling cholmod_finish.
58 *
59 * Requires the Core module. Not required by any CHOLMOD module, except when
60 * debugging is enabled (in which case all modules require the Check module).
61 *
62 * See cholmod_read.c for a description of the file formats supported by the
63 * cholmod_read_* routines.
64 */
65

363/* -------------------------------------------------------------------------- */
364/* cholmod_read_matrix: read a sparse or dense matrix from a file */
365/* -------------------------------------------------------------------------- */
366
367void *cholmod_read_matrix
368(
369 /* ---- input ---- */
370 FILE *f, /* file to read from, must already be open */
371 int prefer, /* If 0, a sparse matrix is always return as a
372 * cholmod_triplet form. It can have any stype
373 * (symmetric-lower, unsymmetric, or
374 * symmetric-upper).
375 * If 1, a sparse matrix is returned as an unsymmetric
376 * cholmod_sparse form (A->stype == 0), with both
377 * upper and lower triangular parts present.
378 * This is what the MATLAB mread mexFunction does,
379 * since MATLAB does not have an stype.
380 * If 2, a sparse matrix is returned with an stype of 0
381 * or 1 (unsymmetric, or symmetric with upper part
382 * stored).
383 * This argument has no effect for dense matrices.
384 */
385 /* ---- output---- */
386 int *mtype, /* CHOLMOD_TRIPLET, CHOLMOD_SPARSE or CHOLMOD_DENSE */
387 /* --------------- */
388 cholmod_common *Common
389) ;
390
391void *cholmod_l_read_matrix (FILE *, int, int *, cholmod_common *) ;
