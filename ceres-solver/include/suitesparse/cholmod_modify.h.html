14/* CHOLMOD Modify module.
15 *
16 * Sparse Cholesky modification routines: update / downdate / rowadd / rowdel.
17 * Can also modify a corresponding solution to Lx=b when L is modified. This
18 * module is most useful when applied on a Cholesky factorization computed by
19 * the Cholesky module, but it does not actually require the Cholesky module.
20 * The Core module can create an identity Cholesky factorization (LDL' where
21 * L=D=I) that can then by modified by these routines.
22 *
23 * Primary routines:
24 * -----------------
25 *
26 * cholmod_updown multiple rank update/downdate
27 * cholmod_rowadd add a row to an LDL' factorization
28 * cholmod_rowdel delete a row from an LDL' factorization
29 *
30 * Secondary routines:
31 * -------------------
32 *
33 * cholmod_updown_solve update/downdate, and modify solution to Lx=b
34 * cholmod_updown_mark update/downdate, and modify solution to partial Lx=b
35 * cholmod_updown_mask update/downdate for LPDASA
36 * cholmod_rowadd_solve add a row, and update solution to Lx=b
37 * cholmod_rowadd_mark add a row, and update solution to partial Lx=b
38 * cholmod_rowdel_solve delete a row, and downdate Lx=b
39 * cholmod_rowdel_mark delete a row, and downdate solution to partial Lx=b
40 *
41 * Requires the Core module. Not required by any other CHOLMOD module.
42 */
43
44#ifndef CHOLMOD_MODIFY_H
45#define CHOLMOD_MODIFY_H
46
47#include "cholmod_core.h"

49/* -------------------------------------------------------------------------- */
50/* cholmod_updown: multiple rank update/downdate */
51/* -------------------------------------------------------------------------- */
52
53/* Compute the new LDL' factorization of LDL'+CC' (an update) or LDL'-CC'
54 * (a downdate). The factor object L need not be an LDL' factorization; it
55 * is converted to one if it isn't. */
56
57int cholmod_updown
58(
59 /* ---- input ---- */
60 int update, /* TRUE for update, FALSE for downdate */
61 cholmod_sparse *C, /* the incoming sparse update */
62 /* ---- in/out --- */
63 cholmod_factor *L, /* factor to modify */
64 /* --------------- */
65 cholmod_common *Common
66) ;
67
68int cholmod_l_updown (int, cholmod_sparse *, cholmod_factor *,
69 cholmod_common *) ;
70
71/* -------------------------------------------------------------------------- */
72/* cholmod_updown_solve: update/downdate, and modify solution to Lx=b */
73/* -------------------------------------------------------------------------- */
74
75/* Does the same as cholmod_updown, except that it also updates/downdates the
76 * solution to Lx=b+DeltaB. x and b must be n-by-1 dense matrices. b is not
77 * need as input to this routine, but a sparse change to b is (DeltaB). Only
78 * entries in DeltaB corresponding to columns modified in L are accessed; the
79 * rest must be zero. */
80
81int cholmod_updown_solve
82(
83 /* ---- input ---- */
84 int update, /* TRUE for update, FALSE for downdate */
85 cholmod_sparse *C, /* the incoming sparse update */
86 /* ---- in/out --- */
87 cholmod_factor *L, /* factor to modify */
88 cholmod_dense *X, /* solution to Lx=b (size n-by-1) */
89 cholmod_dense *DeltaB, /* change in b, zero on output */
90 /* --------------- */
91 cholmod_common *Common
92) ;
93
94int cholmod_l_updown_solve (int, cholmod_sparse *, cholmod_factor *,
95 cholmod_dense *, cholmod_dense *, cholmod_common *) ;
96
97/* -------------------------------------------------------------------------- */
98/* cholmod_updown_mark: update/downdate, and modify solution to partial Lx=b */
99/* -------------------------------------------------------------------------- */
100
101/* Does the same as cholmod_updown_solve, except only part of L is used in
102 * the update/downdate of the solution to Lx=b. This routine is an "expert"
103 * routine. It is meant for use in LPDASA only. See cholmod_updown.c for
104 * a description of colmark. */
105
106int cholmod_updown_mark
107(
108 /* ---- input ---- */
109 int update, /* TRUE for update, FALSE for downdate */
110 cholmod_sparse *C, /* the incoming sparse update */
111 int *colmark, /* int array of size n. See cholmod_updown.c */
112 /* ---- in/out --- */
113 cholmod_factor *L, /* factor to modify */
114 cholmod_dense *X, /* solution to Lx=b (size n-by-1) */
115 cholmod_dense *DeltaB, /* change in b, zero on output */
116 /* --------------- */
117 cholmod_common *Common
118) ;
119
120int cholmod_l_updown_mark (int, cholmod_sparse *, SuiteSparse_long *,
121 cholmod_factor *, cholmod_dense *, cholmod_dense *, cholmod_common *) ;
122
123/* -------------------------------------------------------------------------- */
124/* cholmod_updown_mask: update/downdate, for LPDASA */
125/* -------------------------------------------------------------------------- */
126
127/* Does the same as cholmod_updown_mark, except has an additional "mask"
128 * argument. This routine is an "expert" routine. It is meant for use in
129 * LPDASA only. See cholmod_updown.c for a description of mask. */
130
131int cholmod_updown_mask

134 int update, /* TRUE for update, FALSE for downdate */

137 int *mask, /* size n */
138 /* ---- in/out --- */
139 cholmod_factor *L, /* factor to modify */

150/* -------------------------------------------------------------------------- */
151/* cholmod_rowadd: add a row to an LDL' factorization (a rank-2 update) */
152/* -------------------------------------------------------------------------- */
153
154/* cholmod_rowadd adds a row to the LDL' factorization. It computes the kth
155 * row and kth column of L, and then updates the submatrix L (k+1:n,k+1:n)
156 * accordingly. The kth row and column of L must originally be equal to the
157 * kth row and column of the identity matrix. The kth row/column of L is
158 * computed as the factorization of the kth row/column of the matrix to
159 * factorize, which is provided as a single n-by-1 sparse matrix R. */
160
161int cholmod_rowadd
162(
163 /* ---- input ---- */
164 size_t k, /* row/column index to add */
165 cholmod_sparse *R, /* row/column of matrix to factorize (n-by-1) */
166 /* ---- in/out --- */
167 cholmod_factor *L, /* factor to modify */
168 /* --------------- */
169 cholmod_common *Common
170) ;
171
172int cholmod_l_rowadd (size_t, cholmod_sparse *, cholmod_factor *,
173 cholmod_common *) ;
174
175/* -------------------------------------------------------------------------- */
176/* cholmod_rowadd_solve: add a row, and update solution to Lx=b */
177/* -------------------------------------------------------------------------- */
178
179/* Does the same as cholmod_rowadd, and also updates the solution to Lx=b
180 * See cholmod_updown for a description of how Lx=b is updated. There is on
181 * additional parameter: bk specifies the new kth entry of b. */
182
183int cholmod_rowadd_solve
184(
185 /* ---- input ---- */
186 size_t k, /* row/column index to add */
187 cholmod_sparse *R, /* row/column of matrix to factorize (n-by-1) */
188 double bk [2], /* kth entry of the right-hand-side b */
189 /* ---- in/out --- */
190 cholmod_factor *L, /* factor to modify */
191 cholmod_dense *X, /* solution to Lx=b (size n-by-1) */
192 cholmod_dense *DeltaB, /* change in b, zero on output */
193 /* --------------- */
194 cholmod_common *Common
195) ;
196
197int cholmod_l_rowadd_solve (size_t, cholmod_sparse *, double *,
198 cholmod_factor *, cholmod_dense *, cholmod_dense *, cholmod_common *) ;
199

207
208int cholmod_rowadd_mark

215 /* ---- in/out --- */

227/* -------------------------------------------------------------------------- */
228/* cholmod_rowdel: delete a row from an LDL' factorization (a rank-2 update) */
229/* -------------------------------------------------------------------------- */
230
231/* Sets the kth row and column of L to be the kth row and column of the identity
232 * matrix, and updates L(k+1:n,k+1:n) accordingly. To reduce the running time,
233 * the caller can optionally provide the nonzero pattern (or an upper bound) of
234 * kth row of L, as the sparse n-by-1 vector R. Provide R as NULL if you want
235 * CHOLMOD to determine this itself, which is easier for the caller, but takes
236 * a little more time.
237 */
238
239int cholmod_rowdel
240(
241 /* ---- input ---- */
242 size_t k, /* row/column index to delete */
243 cholmod_sparse *R, /* NULL, or the nonzero pattern of kth row of L */
244 /* ---- in/out --- */
245 cholmod_factor *L, /* factor to modify */
246 /* --------------- */
247 cholmod_common *Common
248) ;
249
250int cholmod_l_rowdel (size_t, cholmod_sparse *, cholmod_factor *,
251 cholmod_common *) ;
252
253/* -------------------------------------------------------------------------- */
254/* cholmod_rowdel_solve: delete a row, and downdate Lx=b */
255/* -------------------------------------------------------------------------- */
256
257/* Does the same as cholmod_rowdel, but also downdates the solution to Lx=b.
258 * When row/column k of A is "deleted" from the system A*y=b, this can induce
259 * a change to x, in addition to changes arising when L and b are modified.
260 * If this is the case, the kth entry of y is required as input (yk) */
261
262int cholmod_rowdel_solve
263(
264 /* ---- input ---- */
265 size_t k, /* row/column index to delete */
266 cholmod_sparse *R, /* NULL, or the nonzero pattern of kth row of L */
267 double yk [2], /* kth entry in the solution to A*y=b */
268 /* ---- in/out --- */
269 cholmod_factor *L, /* factor to modify */
270 cholmod_dense *X, /* solution to Lx=b (size n-by-1) */
271 cholmod_dense *DeltaB, /* change in b, zero on output */
272 /* --------------- */
273 cholmod_common *Common
274) ;
275
276int cholmod_l_rowdel_solve (size_t, cholmod_sparse *, double *,
277 cholmod_factor *, cholmod_dense *, cholmod_dense *, cholmod_common *) ;
278
