14/* CHOLMOD MatrixOps module.
15 *
16 * Basic operations on sparse and dense matrices.
17 *
18 * cholmod_drop A = entries in A with abs. value >= tol
19 * cholmod_norm_dense s = norm (X), 1-norm, inf-norm, or 2-norm
20 * cholmod_norm_sparse s = norm (A), 1-norm or inf-norm
21 * cholmod_horzcat C = [A,B]
22 * cholmod_scale A = diag(s)*A, A*diag(s), s*A or diag(s)*A*diag(s)
23 * cholmod_sdmult Y = alpha*(A*X) + beta*Y or alpha*(A'*X) + beta*Y
24 * cholmod_ssmult C = A*B
25 * cholmod_submatrix C = A (i,j), where i and j are arbitrary vectors
26 * cholmod_vertcat C = [A ; B]
27 *
28 * A, B, C: sparse matrices (cholmod_sparse)
29 * X, Y: dense matrices (cholmod_dense)
30 * s: scalar or vector
31 *
32 * Requires the Core module. Not required by any other CHOLMOD module.
33 */
34

103/* -------------------------------------------------------------------------- */
104/* cholmod_scale: A = diag(s)*A, A*diag(s), s*A or diag(s)*A*diag(s) */
105/* -------------------------------------------------------------------------- */
106
107/* scaling modes, selected by the scale input parameter: */
108#define CHOLMOD_SCALAR 0 /* A = s*A */
109#define CHOLMOD_ROW 1 /* A = diag(s)*A */
110#define CHOLMOD_COL 2 /* A = A*diag(s) */
111#define CHOLMOD_SYM 3 /* A = diag(s)*A*diag(s) */
112
113int cholmod_scale
114(
115 /* ---- input ---- */
116 cholmod_dense *S, /* scale factors (scalar or vector) */
117 int scale, /* type of scaling to compute */
118 /* ---- in/out --- */
119 cholmod_sparse *A, /* matrix to scale */
120 /* --------------- */
121 cholmod_common *Common
122) ;
123
124int cholmod_l_scale (cholmod_dense *, int, cholmod_sparse *, cholmod_common *) ;

170/* -------------------------------------------------------------------------- */
171/* cholmod_submatrix: C = A (r,c), where i and j are arbitrary vectors */
172/* -------------------------------------------------------------------------- */
173
174/* rsize < 0 denotes ":" in MATLAB notation, or more precisely 0:(A->nrow)-1.
175 * In this case, r can be NULL. An rsize of zero, or r = NULL and rsize >= 0,
176 * denotes "[ ]" in MATLAB notation (the empty set).
177 * Similar rules hold for csize.
178 */
179
180cholmod_sparse *cholmod_submatrix
181(
182 /* ---- input ---- */
183 cholmod_sparse *A, /* matrix to subreference */
184 int *rset, /* set of row indices, duplicates OK */
185 SuiteSparse_long rsize, /* size of r; rsize < 0 denotes ":" */
186 int *cset, /* set of column indices, duplicates OK */
187 SuiteSparse_long csize, /* size of c; csize < 0 denotes ":" */
188 int values, /* if TRUE compute the numerical values of C */
189 int sorted, /* if TRUE then return C with sorted columns */
190 /* --------------- */
191 cholmod_common *Common
192) ;
193
194cholmod_sparse *cholmod_l_submatrix (cholmod_sparse *, SuiteSparse_long *,
195 SuiteSparse_long, SuiteSparse_long *, SuiteSparse_long, int, int,
196 cholmod_common *) ;
