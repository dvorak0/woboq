18// === spqr_symbolic ===========================================================
19// =============================================================================
20
21// The contents of this object do not change during numeric factorization. The
22// Symbolic object depends only on the pattern of the input matrix, and not its
23// values. These contents also do not change with column pivoting for rank
24// detection. This makes parallelism easier to manage, since all threads can
25// have access to this object without synchronization.
26//
27// The total size of the Symbolic object is (10 + 2*m + anz + 2*n + 5*nf + rnz)
28// Long's, where the user's input A matrix is m-by-n with anz nonzeros, nf <=
29// MIN(m,n) is the number of frontal matrices, and rnz <= nnz(R) is the number
30// of column indices used to represent the supernodal form of R (one Long per
31// non-pivotal column index in the leading row of each block of R).
32
33struct spqr_symbolic
34{
35
36 // -------------------------------------------------------------------------
37 // row-form of the input matrix and its permutations
38 // -------------------------------------------------------------------------
39
40 // During symbolic analysis, the nonzero pattern of S = A(P,Q) is
41 // constructed, where A is the user's input matrix. Its numerical values
42 // are also constructed, but they do not become part of the Symbolic
43 // object. The matrix S is stored in row-oriented form. The rows of S are
44 // sorted according to their leftmost column index (via PLinv). Column
45 // indices in each row of S are in strictly ascending order, even though
46 // the input matrix A need not be sorted.
47
48 SuiteSparse_long m, n, anz ; // S is m-by-n with anz entries
49
50 SuiteSparse_long *Sp ; // size m+1, row pointers of S
51
52 SuiteSparse_long *Sj ; // size anz = Sp [n], column indices of S
53
54 SuiteSparse_long *Qfill ; // size n, fill-reducing column permutation.
55 // Qfill [k] = j if column k of A is column j of S.
56
57 SuiteSparse_long *PLinv ; // size m, inverse row permutation that places
58 // S=A(P,Q) in increasing order of leftmost column
59 // index. PLinv [i] = k if row i of A is row k of S.
60
61 SuiteSparse_long *Sleft ; // size n+2. The list of rows of S whose
62 // leftmost column index is j is given by
63 // Sleft [j] ... Sleft [j+1]-1. This can be empty (that is, Sleft
64 // [j] can equal Sleft [j+1]). Sleft [n] is the number of
65 // non-empty rows of S, and Sleft [n+1] == m. That is, Sleft [n]
66 // ... Sleft [n+1]-1 gives the empty rows of S, if any.
67
68 // -------------------------------------------------------------------------
69 // frontal matrices: pattern and tree
70 // -------------------------------------------------------------------------
71
72 // Each frontal matrix is fm-by-fn, with fnpiv pivot columns. The fn
73 // column indices are given by a set of size fnpiv pivot columns, defined
74 // by Super, followed by the pattern Rj [ Rp[f] ... Rp[f+1]-1 ].
75
76 // The row indices of the front are not kept. If the Householder vectors
77 // are not kept, the row indices are not needed. If the Householder
78 // vectors are kept, the row indices are computed dynamically during
79 // numerical factorization.
80
81 SuiteSparse_long nf ; // number of frontal matrices; nf <= MIN (m,n)
82 SuiteSparse_long maxfn ; // max # of columns in any front
83
84 // parent, child, and childp define the row merge tree or etree (A'A)
85 SuiteSparse_long *Parent ; // size nf+1
86 SuiteSparse_long *Child ; // size nf+1
87 SuiteSparse_long *Childp ; // size nf+2
88
89 // The parent of a front f is Parent [f], or EMPTY if f=nf.
90 // A list of children of f can be obtained in the list
91 // Child [Childp [f] ... Childp [f+1]-1].
92
93 // Node nf in the tree is a placeholder; it does not represent a frontal
94 // matrix. All roots of the frontal "tree" (may be a forest) have the
95 // placeholder node nf as their parent. Thus, the tree of nodes 0:nf is
96 // truly a tree, with just one parent (node nf).
97
98 SuiteSparse_long *Super ; // size nf+1. Super [f] gives the first
99 // pivot column in the front F. This refers to a column of S. The
100 // number of expected pivot columns in F is thus
101 // Super [f+1] - Super [f].
102
103 SuiteSparse_long *Rp ; // size nf+1
104 SuiteSparse_long *Rj ; // size rjsize; compressed supernodal form of R
105
106 SuiteSparse_long *Post ; // size nf+1, post ordering of frontal tree.
107 // f=Post[k] gives the kth node in the postordered tree
108
109 SuiteSparse_long rjsize ; // size of Rj
110
111 SuiteSparse_long do_rank_detection ; // TRUE: allow for tol >= 0.
112 // FALSE: ignore tol
113
114 // the rest depends on whether or not rank-detection is allowed:
115 SuiteSparse_long maxstack ; // max stack size (sequential case)
116 SuiteSparse_long hisize ; // size of Hii
117
118 SuiteSparse_long keepH ; // TRUE if H is present
119
120 SuiteSparse_long *Hip ; // size nf+1. If H is kept, the row indices
121 // of frontal matrix f are in Hii [Hip [f] ... Hip [f] + Hm [f]],
122 // where Hii and Hm are stored in the numeric object.
123
124 // There is one block row of R per frontal matrix.
125 // The fn column indices of R are given by Rj [Rp [f] ... Rp [f+1]-1],
126 // where the first fp column indices are Super [f] ... Super [f+1]-1.
127 // The remaining column indices in Rj [...] are non-pivotal, and are
128 // in the range Super [f+1] to n. The number of rows of R is at
129 // most fp, but can be less if dead columns appear in the matrix.
130 // The number of columns in the contribution block C is always
131 // cn = fn - fp, where fn = Rp [f+1] - Rp [f].
132
133 SuiteSparse_long ntasks ; // number of tasks in task graph
134 SuiteSparse_long ns ; // number of stacks
135
136 // -------------------------------------------------------------------------
137 // the rest of the QR symbolic object is present only if ntasks > 1
138 // -------------------------------------------------------------------------
139
140 // Task tree (nodes 0:ntasks), including placeholder node
141 SuiteSparse_long *TaskChildp ; // size ntasks+2
142 SuiteSparse_long *TaskChild ; // size ntasks+1
143
144 SuiteSparse_long *TaskStack ; // size ntasks+1
145
146 // list of fronts for each task
147 SuiteSparse_long *TaskFront ; // size nf+1
148 SuiteSparse_long *TaskFrontp ; // size ntasks+2
149
150 SuiteSparse_long *On_stack ; // size nf+1, front f is on
151 // stack On_stack [f]
152
153 // size of each stack
154 SuiteSparse_long *Stack_maxstack ; // size ns+2
155
156} ;

163// The Numeric object contains the numerical values of the triangular/
164// trapezoidal factor R, and optionally the Householder vectors H if they
165// are kept.
166
167template <typename Entry> struct spqr_numeric
168{
169
170 // -------------------------------------------------------------------------
171 // Numeric R factor
172 // -------------------------------------------------------------------------
173
174 Entry **Rblock ; // size nf. R [f] is an (Entry *) pointer to the
175 // R block for front F. It is an upper trapezoidal
176 // of size Rm(f)-by-Rn(f), but only the upper
177 // triangular part is stored in column-packed format.
178
179 Entry **Stacks ; // size ns; an array of stacks holding the R and H
180 // factors and the current frontal matrix F at the head.
181 // This is followed by empty space, then the C blocks of
182 // prior frontal matrices at the bottom. When the
183 // factorization is complete, only the R and H part at
184 // the head of each stack is left.
185
186 SuiteSparse_long *Stack_size ; // size ns; Stack_size [s] is the size
187 // of Stacks [s]
188
189 SuiteSparse_long hisize ; // size of Hii
190
191 SuiteSparse_long n ; // A is m-by-n
192 SuiteSparse_long m ;
193 SuiteSparse_long nf ; // number of frontal matrices
194 SuiteSparse_long ntasks ; // # of tasks in task graph actually used
195 SuiteSparse_long ns ; // number of stacks actually used
196 SuiteSparse_long maxstack ; // size of sequential stack, if used
197
198 // -------------------------------------------------------------------------
199 // for rank detection and m < n case
200 // -------------------------------------------------------------------------
201
202 char *Rdead ; // size n, Rdead [k] = 1 if k is a dead pivot column,
203 // Rdead [k] = 0 otherwise. If no columns are dead,
204 // this is NULL. If m < n, then at least m-n columns
205 // will be dead.
206
207 SuiteSparse_long rank ; // number of live pivot columns
208 SuiteSparse_long rank1 ; // number of live pivot columns in first ntol
209 // columns of A
210
211 SuiteSparse_long maxfrank ; // max number of rows in any R block
212
213 double norm_E_fro ; // 2-norm of w, the vector of dead column 2-norms
214
215 // -------------------------------------------------------------------------
216 // for keeping Householder vectors
217 // -------------------------------------------------------------------------
218
219 // The factorization is R = (H_s * ... * H_2 * H_1) * P_H
220 // where P_H is the permutation HPinv, and H_1, ... H_s are the Householder
221 // vectors (s = rjsize).
222
223 SuiteSparse_long keepH ; // TRUE if H is present
224
225 SuiteSparse_long rjsize ; // size of Hstair and HTau
226
227 SuiteSparse_long *HStair ; // size rjsize. The list Hstair [Rp [f] ...
228 // Rp [f+1]-1] gives the staircase for front F
229
230 Entry *HTau ; // size rjsize. The list HTau [Rp [f] ... Rp [f+1]-1]
231 // gives the Householder coefficients for front F
232
233 SuiteSparse_long *Hii ; // size hisize, row indices of H.
234
235 SuiteSparse_long *HPinv ; // size m. HPinv [i] = k if row i of A and H
236 // is row k of R. This permutation includes
237 // QRsym->PLinv, and the permutation constructed via
238 // pivotal row ordering during factorization.
239
240 SuiteSparse_long *Hm ; // size nf, Hm [f] = # of rows in front F
241 SuiteSparse_long *Hr ; // size nf, Hr [f] = # of rows in R block of
242 // front F
243 SuiteSparse_long maxfm ; // max (Hm [0:nf-1]), computed only if H kept
244
245} ;

252// A combined symbolic+numeric QR factorization of A or [A B],
253// with singletons
254
255template <typename Entry> struct SuiteSparseQR_factorization
256{
257
258 // QR factorization of A or [A Binput] after singletons have been removed
259 double tol ; // tol used
260 spqr_symbolic *QRsym ;
261 spqr_numeric <Entry> *QRnum ;
262
263 // singletons, in compressed-row form; R is n1rows-by-n
264 SuiteSparse_long *R1p ; // size n1rows+1
265 SuiteSparse_long *R1j ;
266 Entry *R1x ;
267 SuiteSparse_long r1nz ; // nnz (R1)
268
269 // combined singleton and fill-reducing permutation
270 SuiteSparse_long *Q1fill ;
271 SuiteSparse_long *P1inv ;
272 SuiteSparse_long *HP1inv ; // NULL if n1cols == 0, in which case
273 // QRnum->HPinv serves in its place.
274
275 // Rmap and RmapInv are NULL if QR->rank == A->ncol
276 SuiteSparse_long *Rmap ; // size n. Rmap [j] = k if column j of R is
277 // the kth live column and where k < QR->rank;
278 // otherwise, if j is a dead column, then
279 // k >= QR->rank.
280
281 SuiteSparse_long *RmapInv ;
282
283 SuiteSparse_long n1rows ; // number of singleton rows of [A B]
284 SuiteSparse_long n1cols ; // number of singleton columns of [A B]
285
286 SuiteSparse_long narows ; // number of rows of A
287 SuiteSparse_long nacols ; // number of columns of A
288 SuiteSparse_long bncols ; // number of columns of B
289 SuiteSparse_long rank ; // rank estimate of A (n1rows + QRnum->rank1),
290 // ranges from 0 to min(m,n)
291
292 int allow_tol ; // if TRUE, do rank detection
293} ;

300// SuiteSparseQR Sparse QR factorization and solve
301// SuiteSparseQR_qmult Q*X, Q'*X, X*Q, or X*Q' for X full or sparse
302
303// returns rank(A) estimate, or EMPTY on failure
304template <typename Entry> SuiteSparse_long SuiteSparseQR
305(
306 // inputs, not modified
307 int ordering, // all, except 3:given treated as 0:fixed
308 double tol, // only accept singletons above tol
309
310 SuiteSparse_long econ, // number of rows of C and R to return; a value
311 // less than the rank r of A is treated as r, and
312 // a value greater than m is treated as m.
313
314 int getCTX, // if 0: return Z = C of size econ-by-bncols
315 // if 1: return Z = C' of size bncols-by-econ
316 // if 2: return Z = X of size econ-by-bncols
317
318 cholmod_sparse *A, // m-by-n sparse matrix
319
320 // B is either sparse or dense. If Bsparse is non-NULL, B is sparse and
321 // Bdense is ignored. If Bsparse is NULL and Bdense is non-NULL, then B is
322 // dense. B is not present if both are NULL.
323 cholmod_sparse *Bsparse,
324 cholmod_dense *Bdense,
325
326 // output arrays, neither allocated nor defined on input.
327
328 // Z is the matrix C, C', or X
329 cholmod_sparse **Zsparse,
330 cholmod_dense **Zdense,
331 cholmod_sparse **R, // the R factor
332 SuiteSparse_long **E, // size n; fill-reducing ordering of A.
333 cholmod_sparse **H, // the Householder vectors (m-by-nh)
334 SuiteSparse_long **HPinv,// size m; row permutation for H
335 cholmod_dense **HTau, // size nh, Householder coefficients
336
337 // workspace and parameters
338 cholmod_common *cc
339) ;
