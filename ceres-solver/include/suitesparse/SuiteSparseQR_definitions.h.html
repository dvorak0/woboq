5/* Core definitions for both C and C++ programs. */
6
7#ifndef SUITESPARSEQR_DEFINITIONS_H
8#define SUITESPARSEQR_DEFINITIONS_H
9
10/* ordering options */
11#define SPQR_ORDERING_FIXED 0
12#define SPQR_ORDERING_NATURAL 1
13#define SPQR_ORDERING_COLAMD 2
14#define SPQR_ORDERING_GIVEN 3 /* only used for C/C++ interface */
15#define SPQR_ORDERING_CHOLMOD 4 /* CHOLMOD best-effort (COLAMD, METIS,...)*/
16#define SPQR_ORDERING_AMD 5 /* AMD(A'*A) */
17#define SPQR_ORDERING_METIS 6 /* metis(A'*A) */
18#define SPQR_ORDERING_DEFAULT 7 /* SuiteSparseQR default ordering */
19#define SPQR_ORDERING_BEST 8 /* try COLAMD, AMD, and METIS; pick best */
20#define SPQR_ORDERING_BESTAMD 9 /* try COLAMD and AMD; pick best */
21
22/* Let [m n] = size of the matrix after pruning singletons. The default
23 * ordering strategy is to use COLAMD if m <= 2*n. Otherwise, AMD(A'A) is
24 * tried. If there is a high fill-in with AMD then try METIS(A'A) and take
25 * the best of AMD and METIS. METIS is not tried if it isn't installed. */
26
27/* tol options */
28#define SPQR_DEFAULT_TOL (-2) /* if tol <= -2, the default tol is used */
29#define SPQR_NO_TOL (-1) /* if -2 < tol < 0, then no tol is used */
30
31/* for qmult, method can be 0,1,2,3: */
32#define SPQR_QTX 0
33#define SPQR_QX 1
34#define SPQR_XQT 2
35#define SPQR_XQ 3
36
37/* system can be 0,1,2,3: Given Q*R=A*E from SuiteSparseQR_factorize: */
38#define SPQR_RX_EQUALS_B 0 /* solve R*X=B or X = R\B */
39#define SPQR_RETX_EQUALS_B 1 /* solve R*E'*X=B or X = E*(R\B) */
40#define SPQR_RTX_EQUALS_B 2 /* solve R'*X=B or X = R'\B */
41#define SPQR_RTX_EQUALS_ETB 3 /* solve R'*X=E'*B or X = R'\(E'*B) */
42

44/* === SuiteSparseQR version ================================================ */
45/* ========================================================================== */

47/*
48 All versions of SuiteSparseQR will include the following definitions.
49 As an example, to test if the version you are using is 1.3 or later:
50
51 if (SPQR_VERSION >= SPQR_VER_CODE (1,3)) ...
52
53 This also works during compile-time:
54
55 #if SPQR_VERSION >= SPQR_VER_CODE (1,3)
56 printf ("This is version 1.3 or later\n") ;
57 #else
58 printf ("This is version is earlier than 1.3\n") ;
59 #endif
60 */
61
62#define SPQR_DATE "Jun 20, 2012"
63#define SPQR_VER_CODE(main,sub) ((main) * 1000 + (sub))
64#define SPQR_MAIN_VERSION 1
65#define SPQR_SUB_VERSION 3
66#define SPQR_SUBSUB_VERSION 1
67#define SPQR_VERSION SPQR_VER_CODE(SPQR_MAIN_VERSION,SPQR_SUB_VERSION)
68
69#endif
