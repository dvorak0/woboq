14/* CHOLMOD Supernodal module.
15 *
16 * Supernodal analysis, factorization, and solve. The simplest way to use
17 * these routines is via the Cholesky module. It does not provide any
18 * fill-reducing orderings, but does accept the orderings computed by the
19 * Cholesky module. It does not require the Cholesky module itself, however.
20 *
21 * Primary routines:
22 * -----------------
23 * cholmod_super_symbolic supernodal symbolic analysis
24 * cholmod_super_numeric supernodal numeric factorization
25 * cholmod_super_lsolve supernodal Lx=b solve
26 * cholmod_super_ltsolve supernodal L'x=b solve
27 *
28 * Prototypes for the BLAS and LAPACK routines that CHOLMOD uses are listed
29 * below, including how they are used in CHOLMOD.
30 *
31 * BLAS routines:
32 * --------------
33 * dtrsv solve Lx=b or L'x=b, L non-unit diagonal, x and b stride-1
34 * dtrsm solve LX=B or L'X=b, L non-unit diagonal
35 * dgemv y=y-A*x or y=y-A'*x (x and y stride-1)
36 * dgemm C=A*B', C=C-A*B, or C=C-A'*B
37 * dsyrk C=tril(A*A')
38 *
39 * LAPACK routines:
40 * ----------------
41 * dpotrf LAPACK: A=chol(tril(A))
42 *
43 * Requires the Core module, and two external packages: LAPACK and the BLAS.
44 * Optionally used by the Cholesky module.
45 */
46
47#ifndef CHOLMOD_SUPERNODAL_H
48#define CHOLMOD_SUPERNODAL_H

52/* -------------------------------------------------------------------------- */
53/* cholmod_super_symbolic */
54/* -------------------------------------------------------------------------- */
55
56/* Analyzes A, AA', or A(:,f)*A(:,f)' in preparation for a supernodal numeric
57 * factorization. The user need not call this directly; cholmod_analyze is
58 * a "simple" wrapper for this routine.
59 */
60
61int cholmod_super_symbolic
62(
63 /* ---- input ---- */
64 cholmod_sparse *A, /* matrix to analyze */
65 cholmod_sparse *F, /* F = A' or A(:,f)' */
66 int *Parent, /* elimination tree */
67 /* ---- in/out --- */
68 cholmod_factor *L, /* simplicial symbolic on input,
69 * supernodal symbolic on output */
70 /* --------------- */
71 cholmod_common *Common
72) ;
73
74int cholmod_l_super_symbolic (cholmod_sparse *, cholmod_sparse *,
75 SuiteSparse_long *, cholmod_factor *, cholmod_common *) ;
76
77/* -------------------------------------------------------------------------- */
78/* cholmod_super_symbolic2 */
79/* -------------------------------------------------------------------------- */
80
81/* Analyze for supernodal Cholesky or multifrontal QR. CHOLMOD itself always
82 * analyzes for supernodal Cholesky, of course. This "for_cholesky = TRUE"
83 * option is used by SuiteSparseQR only. Added for V1.7 */
84
85int cholmod_super_symbolic2
86(
87 /* ---- input ---- */
88 int for_cholesky, /* Cholesky if TRUE, QR if FALSE */
89 cholmod_sparse *A, /* matrix to analyze */
90 cholmod_sparse *F, /* F = A' or A(:,f)' */
91 int *Parent, /* elimination tree */
92 /* ---- in/out --- */
93 cholmod_factor *L, /* simplicial symbolic on input,
94 * supernodal symbolic on output */
95 /* --------------- */
96 cholmod_common *Common
97) ;
98
99int cholmod_l_super_symbolic2 (int, cholmod_sparse *, cholmod_sparse *,
100 SuiteSparse_long *, cholmod_factor *, cholmod_common *) ;
101
102/* -------------------------------------------------------------------------- */
103/* cholmod_super_numeric */
104/* -------------------------------------------------------------------------- */
105
106/* Computes the numeric LL' factorization of A, AA', or A(:,f)*A(:,f)' using
107 * a BLAS-based supernodal method. The user need not call this directly;
108 * cholmod_factorize is a "simple" wrapper for this routine.
109 */
110
111int cholmod_super_numeric
112(
113 /* ---- input ---- */
114 cholmod_sparse *A, /* matrix to factorize */
115 cholmod_sparse *F, /* F = A' or A(:,f)' */
116 double beta [2], /* beta*I is added to diagonal of matrix to factorize */
117 /* ---- in/out --- */
118 cholmod_factor *L, /* factorization */
119 /* --------------- */
120 cholmod_common *Common
121) ;
122
123int cholmod_l_super_numeric (cholmod_sparse *, cholmod_sparse *, double *,
124 cholmod_factor *, cholmod_common *) ;
125
126/* -------------------------------------------------------------------------- */
127/* cholmod_super_lsolve */
128/* -------------------------------------------------------------------------- */
129
130/* Solve Lx=b where L is from a supernodal numeric factorization. The user
131 * need not call this routine directly. cholmod_solve is a "simple" wrapper
132 * for this routine. */
133
134int cholmod_super_lsolve
135(
136 /* ---- input ---- */
137 cholmod_factor *L, /* factor to use for the forward solve */
138 /* ---- output ---- */
139 cholmod_dense *X, /* b on input, solution to Lx=b on output */
140 /* ---- workspace */
141 cholmod_dense *E, /* workspace of size nrhs*(L->maxesize) */
142 /* --------------- */
143 cholmod_common *Common
144) ;
145
146int cholmod_l_super_lsolve (cholmod_factor *, cholmod_dense *, cholmod_dense *,
147 cholmod_common *) ;
148
149/* -------------------------------------------------------------------------- */
150/* cholmod_super_ltsolve */
151/* -------------------------------------------------------------------------- */
152
153/* Solve L'x=b where L is from a supernodal numeric factorization. The user
154 * need not call this routine directly. cholmod_solve is a "simple" wrapper
155 * for this routine. */
156
157int cholmod_super_ltsolve
158(
159 /* ---- input ---- */
160 cholmod_factor *L, /* factor to use for the backsolve */
161 /* ---- output ---- */
162 cholmod_dense *X, /* b on input, solution to L'x=b on output */
163 /* ---- workspace */
164 cholmod_dense *E, /* workspace of size nrhs*(L->maxesize) */
165 /* --------------- */
166 cholmod_common *Common
167) ;
168
169int cholmod_l_super_ltsolve (cholmod_factor *, cholmod_dense *, cholmod_dense *,
170 cholmod_common *) ;
