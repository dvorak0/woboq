31#ifndef CERES_EXAMPLES_RANDOM_H_
32#define CERES_EXAMPLES_RANDOM_H_
33
34#include <math.h>
35#include <stdlib.h>
36
37namespace ceres {
38namespace examples {
39
40// Return a random number sampled from a uniform distribution in the range
41// [0,1].
42inline double RandDouble() {
43 double r = static_cast<double>(rand());
44 return r / RAND_MAX;
45}
46
47// Marsaglia Polar method for generation standard normal (pseudo)
48// random numbers http://en.wikipedia.org/wiki/Marsaglia_polar_method
49inline double RandNormal() {
50 double x1, x2, w;
51 do {
52 x1 = 2.0 * RandDouble() - 1.0;
53 x2 = 2.0 * RandDouble() - 1.0;
54 w = x1 * x1 + x2 * x2;
55 } while ( w >= 1.0 || w == 0.0 );
56
57 w = sqrt((-2.0 * log(w)) / w);
58 return x1 * w;
59}
