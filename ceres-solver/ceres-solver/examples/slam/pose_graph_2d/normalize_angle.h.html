31#ifndef CERES_EXAMPLES_POSE_GRAPH_2D_NORMALIZE_ANGLE_H_
32#define CERES_EXAMPLES_POSE_GRAPH_2D_NORMALIZE_ANGLE_H_
33
34#include <cmath>
35
36#include "ceres/ceres.h"
37
38namespace ceres {
39namespace examples {
40
41// Normalizes the angle in radians between [-pi and pi).
42template <typename T>
43inline T NormalizeAngle(const T& angle_radians) {
44 // Use ceres::floor because it is specialized for double and Jet types.
45 T two_pi(2.0 * M_PI);
46 return angle_radians -
47 two_pi * ceres::floor((angle_radians + T(M_PI)) / two_pi);
48}
49
50} // namespace examples
51} // namespace ceres
