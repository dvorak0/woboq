81#define BEGIN_MGH_PROBLEM(name, num_parameters, num_residuals) \
82 struct name { \
83 static const int kNumParameters = num_parameters; \
84 static const double initial_x[kNumParameters]; \
85 static const double lower_bounds[kNumParameters]; \
86 static const double upper_bounds[kNumParameters]; \
87 static const double constrained_optimal_cost; \
88 static const double unconstrained_optimal_cost; \
89 static CostFunction* Create() { \
90 if (FLAGS_use_numeric_diff) { \
91 ceres::NumericDiffOptions options; \
92 SetNumericDiffOptions(&options); \
93 if (FLAGS_numeric_diff_method == "central") { \
94 return new NumericDiffCostFunction<name, \
95 ceres::CENTRAL, \
96 num_residuals, \
97 num_parameters>( \
98 new name, ceres::TAKE_OWNERSHIP, num_residuals, options); \
99 } else if (FLAGS_numeric_diff_method == "forward") { \
100 return new NumericDiffCostFunction<name, \
101 ceres::FORWARD, \
102 num_residuals, \
103 num_parameters>( \
104 new name, ceres::TAKE_OWNERSHIP, num_residuals, options); \
105 } else if (FLAGS_numeric_diff_method == "ridders") { \
106 return new NumericDiffCostFunction<name, \
107 ceres::RIDDERS, \
108 num_residuals, \
109 num_parameters>( \
110 new name, ceres::TAKE_OWNERSHIP, num_residuals, options); \
111 } else { \
112 LOG(ERROR) << "Invalid numeric diff method specified"; \
113 return NULL; \
114 } \
115 } else { \
116 return new AutoDiffCostFunction<name, \
117 num_residuals, \
118 num_parameters>(new name); \
119 } \
120 } \
121 template <typename T> \
122 bool operator()(const T* const x, T* residual) const {
123
124#define END_MGH_PROBLEM return true; } }; // NOLINT

538template<typename TestProblem> bool Solve(bool is_constrained, int trial) {
539 double x[TestProblem::kNumParameters];
540 for (int i = 0; i < TestProblem::kNumParameters; ++i) {
541 x[i] = pow(10, trial) * TestProblem::initial_x[i];
542 }
543
544 Problem problem;
545 problem.AddResidualBlock(TestProblem::Create(), NULL, x);
546 double optimal_cost = TestProblem::unconstrained_optimal_cost;
547
548 if (is_constrained) {
549 for (int i = 0; i < TestProblem::kNumParameters; ++i) {
550 problem.SetParameterLowerBound(x, i, TestProblem::lower_bounds[i]);
551 problem.SetParameterUpperBound(x, i, TestProblem::upper_bounds[i]);
552 }
553 optimal_cost = TestProblem::constrained_optimal_cost;
554 }
555
556 Solver::Options options;
557 options.parameter_tolerance = 1e-18;
558 options.function_tolerance = 1e-18;
559 options.gradient_tolerance = 1e-18;
560 options.max_num_iterations = 1000;
561 options.linear_solver_type = DENSE_QR;
562 Solver::Summary summary;
563 Solve(options, &problem, &summary);
564
565 const double kMinLogRelativeError = 4.0;
566 const double log_relative_error = -std::log10(
567 std::abs(2.0 * summary.final_cost - optimal_cost) /
568 (optimal_cost > 0.0 ? optimal_cost : 1.0));
569
570 const bool success = log_relative_error >= kMinLogRelativeError;
571 LOG(INFO) << "Expected : " << optimal_cost
572 << " actual: " << 2.0 * summary.final_cost
573 << " " << success
574 << " in " << summary.total_time_in_seconds
575 << " seconds";
576 return success;
577}
