46// This block of quasi-repeated code calls the user-supplied functor, which may
47// take a variable number of arguments. This is accomplished by specializing the
48// struct based on the size of the trailing parameters; parameters with 0 size
49// are assumed missing.
50template<typename Functor, typename T, int N0, int N1, int N2, int N3, int N4,
51 int N5, int N6, int N7, int N8, int N9>
52struct VariadicEvaluate {
53 static bool Call(const Functor& functor, T const *const *input, T* output) {
54 return functor(input[0],
55 input[1],
56 input[2],
57 input[3],
58 input[4],
59 input[5],
60 input[6],
61 input[7],
62 input[8],
63 input[9],
64 output);
65 }
66};
