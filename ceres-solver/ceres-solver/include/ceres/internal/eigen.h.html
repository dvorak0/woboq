31#ifndef CERES_INTERNAL_EIGEN_H_
32#define CERES_INTERNAL_EIGEN_H_
33
34#include "Eigen/Core"
35
36namespace ceres {
37
38typedef Eigen::Matrix<double, Eigen::Dynamic, 1> Vector;
39typedef Eigen::Matrix<double,
40 Eigen::Dynamic,
41 Eigen::Dynamic,
42 Eigen::RowMajor> Matrix;
43typedef Eigen::Map<Vector> VectorRef;
44typedef Eigen::Map<Matrix> MatrixRef;
45typedef Eigen::Map<const Vector> ConstVectorRef;
46typedef Eigen::Map<const Matrix> ConstMatrixRef;
47
48// Column major matrices for DenseSparseMatrix/DenseQRSolver
49typedef Eigen::Matrix<double,
50 Eigen::Dynamic,
51 Eigen::Dynamic,
52 Eigen::ColMajor> ColMajorMatrix;
53
54typedef Eigen::Map<ColMajorMatrix, 0,
55 Eigen::Stride<Eigen::Dynamic, 1> > ColMajorMatrixRef;
56
57typedef Eigen::Map<const ColMajorMatrix,
58 0,
59 Eigen::Stride<Eigen::Dynamic, 1> > ConstColMajorMatrixRef;
60
61
62
63// C++ does not support templated typdefs, thus the need for this
64// struct so that we can support statically sized Matrix and Maps.
65template <int num_rows = Eigen::Dynamic, int num_cols = Eigen::Dynamic>
66struct EigenTypes {
67 typedef Eigen::Matrix <double, num_rows, num_cols, Eigen::RowMajor>
68 Matrix;
69
70 typedef Eigen::Map<
71 Eigen::Matrix<double, num_rows, num_cols, Eigen::RowMajor> >
72 MatrixRef;
73
74 typedef Eigen::Matrix <double, num_rows, 1>
75 Vector;
76
77 typedef Eigen::Map <
78 Eigen::Matrix<double, num_rows, 1> >
79 VectorRef;
80
81
82 typedef Eigen::Map<
83 const Eigen::Matrix<double, num_rows, num_cols, Eigen::RowMajor> >
84 ConstMatrixRef;
85
86 typedef Eigen::Map <
87 const Eigen::Matrix<double, num_rows, 1> >
88 ConstVectorRef;
89};
90
91} // namespace ceres
92
93#endif // CERES_INTERNAL_EIGEN_H_
