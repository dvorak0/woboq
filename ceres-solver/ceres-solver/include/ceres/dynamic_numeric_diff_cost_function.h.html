50// This numeric diff implementation differs from the one found in
51// numeric_diff_cost_function.h by supporting numericdiff on cost
52// functions with variable numbers of parameters with variable
53// sizes. With the other implementation, all the sizes (both the
54// number of parameter blocks and the size of each block) must be
55// fixed at compile time.
56//
57// The functor API differs slightly from the API for fixed size
58// numeric diff; the expected interface for the cost functors is:
59//
60// struct MyCostFunctor {
61// bool operator()(double const* const* parameters, double* residuals) const {
62// // Use parameters[i] to access the i'th parameter block.
63// }
64// }
65//
66// Since the sizing of the parameters is done at runtime, you must
67// also specify the sizes after creating the
68// DynamicNumericDiffCostFunction. For example:
69//
70// DynamicAutoDiffCostFunction<MyCostFunctor, CENTRAL> cost_function(
71// new MyCostFunctor());
72// cost_function.AddParameterBlock(5);
73// cost_function.AddParameterBlock(10);
74// cost_function.SetNumResiduals(21);
75template <typename CostFunctor, NumericDiffMethodType method = CENTRAL>
76class DynamicNumericDiffCostFunction : public DynamicCostFunction {
77 public:
78 explicit DynamicNumericDiffCostFunction(
79 const CostFunctor* functor,
80 Ownership ownership = TAKE_OWNERSHIP,
81 const NumericDiffOptions& options = NumericDiffOptions())
82 : functor_(functor),
83 ownership_(ownership),
84 options_(options) {
85 }
86
87 virtual ~DynamicNumericDiffCostFunction() {
88 if (ownership_ != TAKE_OWNERSHIP) {
89 functor_.release();
90 }
91 }
92
93 virtual bool Evaluate(double const* const* parameters,
94 double* residuals,
95 double** jacobians) const {
96 using internal::NumericDiff;
97 CHECK_GT(num_residuals(), 0)
98 << "You must call DynamicNumericDiffCostFunction::SetNumResiduals() "
99 << "before DynamicNumericDiffCostFunction::Evaluate().";
100
101 const std::vector<int32>& block_sizes = parameter_block_sizes();
102 CHECK(!block_sizes.empty())
103 << "You must call DynamicNumericDiffCostFunction::AddParameterBlock() "
104 << "before DynamicNumericDiffCostFunction::Evaluate().";
105
106 const bool status = EvaluateCostFunctor(parameters, residuals);
107 if (jacobians == NULL || !status) {
108 return status;
109 }
110
111 // Create local space for a copy of the parameters which will get mutated.
112 int parameters_size = accumulate(block_sizes.begin(), block_sizes.end(), 0);
113 std::vector<double> parameters_copy(parameters_size);
114 std::vector<double*> parameters_references_copy(block_sizes.size());
115 parameters_references_copy[0] = &parameters_copy[0];
116 for (size_t block = 1; block < block_sizes.size(); ++block) {
117 parameters_references_copy[block] = parameters_references_copy[block - 1]
118 + block_sizes[block - 1];
119 }
120
121 // Copy the parameters into the local temp space.
122 for (size_t block = 0; block < block_sizes.size(); ++block) {
123 memcpy(parameters_references_copy[block],
124 parameters[block],
125 block_sizes[block] * sizeof(*parameters[block]));
126 }
127
128 for (size_t block = 0; block < block_sizes.size(); ++block) {
129 if (jacobians[block] != NULL &&
130 !NumericDiff<CostFunctor, method, DYNAMIC,
131 DYNAMIC, DYNAMIC, DYNAMIC, DYNAMIC, DYNAMIC,
132 DYNAMIC, DYNAMIC, DYNAMIC, DYNAMIC, DYNAMIC,
133 DYNAMIC, DYNAMIC>::EvaluateJacobianForParameterBlock(
134 functor_.get(),
135 residuals,
136 options_,
137 this->num_residuals(),
138 block,
139 block_sizes[block],
140 &parameters_references_copy[0],
141 jacobians[block])) {
142 return false;
143 }
144 }
145 return true;
146 }
147
148 private:
149 bool EvaluateCostFunctor(double const* const* parameters,
150 double* residuals) const {
151 return EvaluateCostFunctorImpl(functor_.get(),
152 parameters,
153 residuals,
154 functor_.get());
155 }
156
157 // Helper templates to allow evaluation of a functor or a
158 // CostFunction.
159 bool EvaluateCostFunctorImpl(const CostFunctor* functor,
160 double const* const* parameters,
161 double* residuals,
162 const void* /* NOT USED */) const {
163 return (*functor)(parameters, residuals);
164 }
165
166 bool EvaluateCostFunctorImpl(const CostFunctor* functor,
167 double const* const* parameters,
168 double* residuals,
169 const CostFunction* /* NOT USED */) const {
170 return functor->Evaluate(parameters, residuals, NULL);
171 }
172
173 internal::scoped_ptr<const CostFunctor> functor_;
174 Ownership ownership_;
175 NumericDiffOptions options_;
176};
