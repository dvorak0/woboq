31#ifndef CERES_PUBLIC_VERSION_H_
32#define CERES_PUBLIC_VERSION_H_
33
34#define CERES_VERSION_MAJOR 1
35#define CERES_VERSION_MINOR 13
36#define CERES_VERSION_REVISION 0
37
38// Classic CPP stringifcation; the extra level of indirection allows the
39// preprocessor to expand the macro before being converted to a string.
40#define CERES_TO_STRING_HELPER(x) #x
41#define CERES_TO_STRING(x) CERES_TO_STRING_HELPER(x)
42
43// The Ceres version as a string; for example "1.9.0".
44#define CERES_VERSION_STRING CERES_TO_STRING(CERES_VERSION_MAJOR) "." \
45 CERES_TO_STRING(CERES_VERSION_MINOR) "." \
46 CERES_TO_STRING(CERES_VERSION_REVISION)
47
48#endif // CERES_PUBLIC_VERSION_H_
