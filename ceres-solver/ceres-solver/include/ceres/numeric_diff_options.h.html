32#ifndef CERES_PUBLIC_NUMERIC_DIFF_OPTIONS_H_
33#define CERES_PUBLIC_NUMERIC_DIFF_OPTIONS_H_
34
35namespace ceres {
36
37// Options pertaining to numeric differentiation (e.g., convergence criteria,
38// step sizes).
39struct CERES_EXPORT NumericDiffOptions {
40 NumericDiffOptions() {
41 relative_step_size = 1e-6;
42 ridders_relative_initial_step_size = 1e-2;
43 max_num_ridders_extrapolations = 10;
44 ridders_epsilon = 1e-12;
45 ridders_step_shrink_factor = 2.0;
46 }
47
48 // Numeric differentiation step size (multiplied by parameter block's
49 // order of magnitude). If parameters are close to zero, the step size
50 // is set to sqrt(machine_epsilon).
51 double relative_step_size;
52
53 // Initial step size for Ridders adaptive numeric differentiation (multiplied
54 // by parameter block's order of magnitude).
55 // If parameters are close to zero, Ridders' method sets the step size
56 // directly to this value. This parameter is separate from
57 // "relative_step_size" in order to set a different default value.
58 //
59 // Note: For Ridders' method to converge, the step size should be initialized
60 // to a value that is large enough to produce a significant change in the
61 // function. As the derivative is estimated, the step size decreases.
62 double ridders_relative_initial_step_size;
63
64 // Maximal number of adaptive extrapolations (sampling) in Ridders' method.
65 int max_num_ridders_extrapolations;
66
67 // Convergence criterion on extrapolation error for Ridders adaptive
68 // differentiation. The available error estimation methods are defined in
69 // NumericDiffErrorType and set in the "ridders_error_method" field.
70 double ridders_epsilon;
71
72 // The factor in which to shrink the step size with each extrapolation in
73 // Ridders' method.
74 double ridders_step_shrink_factor;
75};
76
77} // namespace ceres
78
79#endif // CERES_PUBLIC_NUMERIC_DIFF_OPTIONS_H_
