45class BlockJacobiPreconditionerTest : public ::testing::Test {
46 protected:
47 void SetUpFromProblemId(int problem_id) {
48 scoped_ptr<LinearLeastSquaresProblem> problem(
49 CreateLinearLeastSquaresProblemFromId(problem_id));
50
51 CHECK_NOTNULL(problem.get());
52 A.reset(down_cast<BlockSparseMatrix*>(problem->A.release()));
53 D.reset(problem->D.release());
54
55 Matrix dense_a;
56 A->ToDenseMatrix(&dense_a);
57 dense_ata = dense_a.transpose() * dense_a;
58 dense_ata += VectorRef(D.get(), A->num_cols())
59 .array().square().matrix().asDiagonal();
60 }
61
62 void VerifyDiagonalBlocks(const int problem_id) {
63 SetUpFromProblemId(problem_id);
64
65 BlockJacobiPreconditioner pre(*A);
66 pre.Update(*A, D.get());
67 BlockRandomAccessDiagonalMatrix* m =
68 const_cast<BlockRandomAccessDiagonalMatrix*>(&pre.matrix());
69 EXPECT_EQ(m->num_rows(), A->num_cols());
70 EXPECT_EQ(m->num_cols(), A->num_cols());
71
72 const CompressedRowBlockStructure* bs = A->block_structure();
73 for (int i = 0; i < bs->cols.size(); ++i) {
74 const int block_size = bs->cols[i].size;
75 int r, c, row_stride, col_stride;
76 CellInfo* cell_info = m->GetCell(i, i,
77 &r, &c,
78 &row_stride, &col_stride);
79 MatrixRef m(cell_info->values, row_stride, col_stride);
80 Matrix actual_block_inverse = m.block(r, c, block_size, block_size);
81 Matrix expected_block = dense_ata.block(bs->cols[i].position,
82 bs->cols[i].position,
83 block_size,
84 block_size);
85 const double residual = (actual_block_inverse * expected_block -
86 Matrix::Identity(block_size, block_size)).norm();
87 EXPECT_NEAR(residual, 0.0, 1e-12) << "Block: " << i;
88 }
89 }
90
91 scoped_ptr<BlockSparseMatrix> A;
92 scoped_array<double> D;
93 Matrix dense_ata;
94};
95
96TEST_F(BlockJacobiPreconditionerTest, SmallProblem) {
97 VerifyDiagonalBlocks(2);
98}
99
100TEST_F(BlockJacobiPreconditionerTest, LargeProblem) {
101 VerifyDiagonalBlocks(3);
102}
