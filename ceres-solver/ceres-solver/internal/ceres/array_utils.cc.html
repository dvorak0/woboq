46bool IsArrayValid(const int size, const double* x) {
47 if (x != NULL) {
48 for (int i = 0; i < size; ++i) {
49 if (!IsFinite(x[i]) || (x[i] == kImpossibleValue)) {
50 return false;
51 }
52 }
53 }
54 return true;
55}
56
57int FindInvalidValue(const int size, const double* x) {
58 if (x == NULL) {
59 return size;
60 }
61
62 for (int i = 0; i < size; ++i) {
63 if (!IsFinite(x[i]) || (x[i] == kImpossibleValue)) {
64 return i;
65 }
66 }
67
68 return size;
69}
70
71void InvalidateArray(const int size, double* x) {
72 if (x != NULL) {
73 for (int i = 0; i < size; ++i) {
74 x[i] = kImpossibleValue;
75 }
76 }
77}
78
79void AppendArrayToString(const int size, const double* x, string* result) {
80 for (int i = 0; i < size; ++i) {
81 if (x == NULL) {
82 StringAppendF(result, "Not Computed ");
83 } else {
84 if (x[i] == kImpossibleValue) {
85 StringAppendF(result, "Uninitialized ");
86 } else {
87 StringAppendF(result, "%12g ", x[i]);
88 }
89 }
90 }
91}
92
93void MapValuesToContiguousRange(const int size, int* array) {
94 std::vector<int> unique_values(array, array + size);
95 std::sort(unique_values.begin(), unique_values.end());
96 unique_values.erase(std::unique(unique_values.begin(),
97 unique_values.end()),
98 unique_values.end());
99
100 for (int i = 0; i < size; ++i) {
101 array[i] = std::lower_bound(unique_values.begin(),
102 unique_values.end(),
103 array[i]) - unique_values.begin();
104 }
105}
