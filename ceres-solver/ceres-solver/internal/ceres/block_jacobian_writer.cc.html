49// Given the residual block ordering, build a lookup table to determine which
50// per-parameter jacobian goes where in the overall program jacobian.
51//
52// Since we expect to use a Schur type linear solver to solve the LM step, take
53// extra care to place the E blocks and the F blocks contiguously. E blocks are
54// the first num_eliminate_blocks parameter blocks as indicated by the parameter
55// block ordering. The remaining parameter blocks are the F blocks.
56//
57// TODO(keir): Consider if we should use a boolean for each parameter block
58// instead of num_eliminate_blocks.
59void BuildJacobianLayout(const Program& program,
60 int num_eliminate_blocks,
61 vector<int*>* jacobian_layout,
62 vector<int>* jacobian_layout_storage) {
63 const vector<ResidualBlock*>& residual_blocks = program.residual_blocks();
64
65 // Iterate over all the active residual blocks and determine how many E blocks
66 // are there. This will determine where the F blocks start in the jacobian
67 // matrix. Also compute the number of jacobian blocks.
68 int f_block_pos = 0;
69 int num_jacobian_blocks = 0;
70 for (int i = 0; i < residual_blocks.size(); ++i) {
71 ResidualBlock* residual_block = residual_blocks[i];
72 const int num_residuals = residual_block->NumResiduals();
73 const int num_parameter_blocks = residual_block->NumParameterBlocks();
74
75 // Advance f_block_pos over each E block for this residual.
76 for (int j = 0; j < num_parameter_blocks; ++j) {
77 ParameterBlock* parameter_block = residual_block->parameter_blocks()[j];
78 if (!parameter_block->IsConstant()) {
79 // Only count blocks for active parameters.
80 num_jacobian_blocks++;
81 if (parameter_block->index() < num_eliminate_blocks) {
82 f_block_pos += num_residuals * parameter_block->LocalSize();
83 }
84 }
85 }
86 }
87
88 // We now know that the E blocks are laid out starting at zero, and the F
89 // blocks are laid out starting at f_block_pos. Iterate over the residual
90 // blocks again, and this time fill the jacobian_layout array with the
91 // position information.
92
93 jacobian_layout->resize(program.NumResidualBlocks());
94 jacobian_layout_storage->resize(num_jacobian_blocks);
95
96 int e_block_pos = 0;
97 int* jacobian_pos = &(*jacobian_layout_storage)[0];
98 for (int i = 0; i < residual_blocks.size(); ++i) {
99 const ResidualBlock* residual_block = residual_blocks[i];
100 const int num_residuals = residual_block->NumResiduals();
101 const int num_parameter_blocks = residual_block->NumParameterBlocks();
102
103 (*jacobian_layout)[i] = jacobian_pos;
104 for (int j = 0; j < num_parameter_blocks; ++j) {
105 ParameterBlock* parameter_block = residual_block->parameter_blocks()[j];
106 const int parameter_block_index = parameter_block->index();
107 if (parameter_block->IsConstant()) {
108 continue;
109 }
110 const int jacobian_block_size =
111 num_residuals * parameter_block->LocalSize();
112 if (parameter_block_index < num_eliminate_blocks) {
113 *jacobian_pos = e_block_pos;
114 e_block_pos += jacobian_block_size;
115 } else {
116 *jacobian_pos = f_block_pos;
117 f_block_pos += jacobian_block_size;
118 }
119 jacobian_pos++;
120 }
121 }
122}

126BlockJacobianWriter::BlockJacobianWriter(const Evaluator::Options& options,
127 Program* program)
128 : program_(program) {
129 CHECK_GE(options.num_eliminate_blocks, 0)
130 << "num_eliminate_blocks must be greater than 0.";
131
132 BuildJacobianLayout(*program,
133 options.num_eliminate_blocks,
134 &jacobian_layout_,
135 &jacobian_layout_storage_);
136}

138// Create evaluate prepareres that point directly into the final jacobian. This
139// makes the final Write() a nop.
140BlockEvaluatePreparer* BlockJacobianWriter::CreateEvaluatePreparers(
141 int num_threads) {
142 int max_derivatives_per_residual_block =
143 program_->MaxDerivativesPerResidualBlock();
144
145 BlockEvaluatePreparer* preparers = new BlockEvaluatePreparer[num_threads];
146 for (int i = 0; i < num_threads; i++) {
147 preparers[i].Init(&jacobian_layout_[0], max_derivatives_per_residual_block);
148 }
149 return preparers;
150}

152SparseMatrix* BlockJacobianWriter::CreateJacobian() const {
153 CompressedRowBlockStructure* bs = new CompressedRowBlockStructure;
154
155 const vector<ParameterBlock*>& parameter_blocks =
156 program_->parameter_blocks();
157
158 // Construct the column blocks.
159 bs->cols.resize(parameter_blocks.size());
160 for (int i = 0, cursor = 0; i < parameter_blocks.size(); ++i) {
161 CHECK_NE(parameter_blocks[i]->index(), -1);
162 CHECK(!parameter_blocks[i]->IsConstant());
163 bs->cols[i].size = parameter_blocks[i]->LocalSize();
164 bs->cols[i].position = cursor;
165 cursor += bs->cols[i].size;
166 }
167
168 // Construct the cells in each row.
169 const vector<ResidualBlock*>& residual_blocks = program_->residual_blocks();
170 int row_block_position = 0;
171 bs->rows.resize(residual_blocks.size());
172 for (int i = 0; i < residual_blocks.size(); ++i) {
173 const ResidualBlock* residual_block = residual_blocks[i];
174 CompressedRow* row = &bs->rows[i];
175
176 row->block.size = residual_block->NumResiduals();
177 row->block.position = row_block_position;
178 row_block_position += row->block.size;
179
180 // Size the row by the number of active parameters in this residual.
181 const int num_parameter_blocks = residual_block->NumParameterBlocks();
182 int num_active_parameter_blocks = 0;
183 for (int j = 0; j < num_parameter_blocks; ++j) {
184 if (residual_block->parameter_blocks()[j]->index() != -1) {
185 num_active_parameter_blocks++;
186 }
187 }
188 row->cells.resize(num_active_parameter_blocks);
189
190 // Add layout information for the active parameters in this row.
191 for (int j = 0, k = 0; j < num_parameter_blocks; ++j) {
192 const ParameterBlock* parameter_block =
193 residual_block->parameter_blocks()[j];
194 if (!parameter_block->IsConstant()) {
195 Cell& cell = row->cells[k];
196 cell.block_id = parameter_block->index();
197 cell.position = jacobian_layout_[i][k];
198
199 // Only increment k for active parameters, since there is only layout
200 // information for active parameters.
201 k++;
202 }
203 }
204
205 sort(row->cells.begin(), row->cells.end(), CellLessThan);
206 }
207
208 BlockSparseMatrix* jacobian = new BlockSparseMatrix(bs);
209 CHECK_NOTNULL(jacobian);
210 return jacobian;
211}
