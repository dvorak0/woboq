50SparseMatrix* DynamicCompressedRowJacobianWriter::CreateJacobian() const {
51 // Initialize `jacobian` with zero number of `max_num_nonzeros`.
52 const int num_residuals = program_->NumResiduals();
53 const int num_effective_parameters = program_->NumEffectiveParameters();
54
55 DynamicCompressedRowSparseMatrix* jacobian =
56 new DynamicCompressedRowSparseMatrix(num_residuals,
57 num_effective_parameters,
58 0);
59
60 vector<int>* row_blocks = jacobian->mutable_row_blocks();
61 for (int i = 0; i < jacobian->num_rows(); ++i) {
62 row_blocks->push_back(1);
63 }
64
65 vector<int>* col_blocks = jacobian->mutable_col_blocks();
66 for (int i = 0; i < jacobian->num_cols(); ++i) {
67 col_blocks->push_back(1);
68 }
69
70 return jacobian;
71}

73void DynamicCompressedRowJacobianWriter::Write(int residual_id,
74 int residual_offset,
75 double **jacobians,
76 SparseMatrix* base_jacobian) {
77 DynamicCompressedRowSparseMatrix* jacobian =
78 down_cast<DynamicCompressedRowSparseMatrix*>(base_jacobian);
79
80 // Get the `residual_block` of interest.
81 const ResidualBlock* residual_block =
82 program_->residual_blocks()[residual_id];
83 const int num_residuals = residual_block->NumResiduals();
84
85 vector<pair<int, int> > evaluated_jacobian_blocks;
86 CompressedRowJacobianWriter::GetOrderedParameterBlocks(
87 program_, residual_id, &evaluated_jacobian_blocks);
88
89 // `residual_offset` is the residual row in the global jacobian.
90 // Empty the jacobian rows.
91 jacobian->ClearRows(residual_offset, num_residuals);
92
93 // Iterate over each parameter block.
94 for (int i = 0; i < evaluated_jacobian_blocks.size(); ++i) {
95 const ParameterBlock* parameter_block =
96 program_->parameter_blocks()[evaluated_jacobian_blocks[i].first];
97 const int parameter_block_jacobian_index =
98 evaluated_jacobian_blocks[i].second;
99 const int parameter_block_size = parameter_block->LocalSize();
100
101 // For each parameter block only insert its non-zero entries.
102 for (int r = 0; r < num_residuals; ++r) {
103 for (int c = 0; c < parameter_block_size; ++c) {
104 const double& v = jacobians[parameter_block_jacobian_index][
105 r * parameter_block_size + c];
106 // Only insert non-zero entries.
107 if (v != 0.0) {
108 jacobian->InsertEntry(
109 residual_offset + r, parameter_block->delta_offset() + c, v);
110 }
111 }
112 }
113 }
114}
