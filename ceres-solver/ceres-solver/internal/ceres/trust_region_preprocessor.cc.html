56ParameterBlockOrdering* CreateDefaultLinearSolverOrdering(
57 const Program& program) {
58 ParameterBlockOrdering* ordering = new ParameterBlockOrdering;
59 const vector<ParameterBlock*>& parameter_blocks =
60 program.parameter_blocks();
61 for (int i = 0; i < parameter_blocks.size(); ++i) {
62 ordering->AddElementToGroup(
63 const_cast<double*>(parameter_blocks[i]->user_state()), 0);
64 }
65 return ordering;
66}

68// Check if all the user supplied values in the parameter blocks are
69// sane or not, and if the program is feasible or not.
70bool IsProgramValid(const Program& program, std::string* error) {
71 return (program.ParameterBlocksAreFinite(error) &&
72 program.IsFeasible(error));
73}

75void AlternateLinearSolverAndPreconditionerForSchurTypeLinearSolver(
76 Solver::Options* options) {
77 if (!IsSchurType(options->linear_solver_type)) {
78 return;
79 }
80
81 const LinearSolverType linear_solver_type_given = options->linear_solver_type;
82 const PreconditionerType preconditioner_type_given =
83 options->preconditioner_type;
84 options->linear_solver_type = LinearSolver::LinearSolverForZeroEBlocks(
85 linear_solver_type_given);
86
87 std::string message;
88 if (linear_solver_type_given == ITERATIVE_SCHUR) {
89 options->preconditioner_type = Preconditioner::PreconditionerForZeroEBlocks(
90 preconditioner_type_given);
91
92 message =
93 StringPrintf(
94 "No E blocks. Switching from %s(%s) to %s(%s).",
95 LinearSolverTypeToString(linear_solver_type_given),
96 PreconditionerTypeToString(preconditioner_type_given),
97 LinearSolverTypeToString(options->linear_solver_type),
98 PreconditionerTypeToString(options->preconditioner_type));
99 } else {
100 message =
101 StringPrintf(
102 "No E blocks. Switching from %s to %s.",
103 LinearSolverTypeToString(linear_solver_type_given),
104 LinearSolverTypeToString(options->linear_solver_type));
105 }
106
107 VLOG_IF(1, options->logging_type != SILENT) << message;
108}

110// For Schur type and SPARSE_NORMAL_CHOLESKY linear solvers, reorder
111// the program to reduce fill-in and increase cache coherency.
112bool ReorderProgram(PreprocessedProblem* pp) {
113 Solver::Options& options = pp->options;
114 if (IsSchurType(options.linear_solver_type)) {
115 return ReorderProgramForSchurTypeLinearSolver(
116 options.linear_solver_type,
117 options.sparse_linear_algebra_library_type,
118 pp->problem->parameter_map(),
119 options.linear_solver_ordering.get(),
120 pp->reduced_program.get(),
121 &pp->error);
122 }
123
124 if (options.linear_solver_type == SPARSE_NORMAL_CHOLESKY &&
125 !options.dynamic_sparsity) {
126 return ReorderProgramForSparseNormalCholesky(
127 options.sparse_linear_algebra_library_type,
128 *options.linear_solver_ordering,
129 pp->reduced_program.get(),
130 &pp->error);
131 }
132
133 return true;
134}

136// Configure and create a linear solver object. In doing so, if a
137// sparse direct factorization based linear solver is being used, then
138// find a fill reducing ordering and reorder the program as needed
139// too.
140bool SetupLinearSolver(PreprocessedProblem* pp) {
141 Solver::Options& options = pp->options;
142 if (options.linear_solver_ordering.get() == NULL) {
143 // If the user has not supplied a linear solver ordering, then we
144 // assume that they are giving all the freedom to us in choosing
145 // the best possible ordering. This intent can be indicated by
146 // putting all the parameter blocks in the same elimination group.
147 options.linear_solver_ordering.reset(
148 CreateDefaultLinearSolverOrdering(*pp->reduced_program));
149 } else {
150 // If the user supplied an ordering, then check if the first
151 // elimination group is still non-empty after the reduced problem
152 // has been constructed.
153 //
154 // This is important for Schur type linear solvers, where the
155 // first elimination group is special -- it needs to be an
156 // independent set.
157 //
158 // If the first elimination group is empty, then we cannot use the
159 // user's requested linear solver (and a preconditioner as the
160 // case may be) so we must use a different one.
161 ParameterBlockOrdering* ordering = options.linear_solver_ordering.get();
162 const int min_group_id = ordering->MinNonZeroGroup();
163 ordering->Remove(pp->removed_parameter_blocks);
164 if (IsSchurType(options.linear_solver_type) &&
165 min_group_id != ordering->MinNonZeroGroup()) {
166 AlternateLinearSolverAndPreconditionerForSchurTypeLinearSolver(
167 &options);
168 }
169 }
170
171 // Reorder the program to reduce fill in and improve cache coherency
172 // of the Jacobian.
173 if (!ReorderProgram(pp)) {
174 return false;
175 }
176
177 // Configure the linear solver.
178 pp->linear_solver_options = LinearSolver::Options();
179 pp->linear_solver_options.min_num_iterations =
180 options.min_linear_solver_iterations;
181 pp->linear_solver_options.max_num_iterations =
182 options.max_linear_solver_iterations;
183 pp->linear_solver_options.type = options.linear_solver_type;
184 pp->linear_solver_options.preconditioner_type = options.preconditioner_type;
185 pp->linear_solver_options.visibility_clustering_type =
186 options.visibility_clustering_type;
187 pp->linear_solver_options.sparse_linear_algebra_library_type =
188 options.sparse_linear_algebra_library_type;
189 pp->linear_solver_options.dense_linear_algebra_library_type =
190 options.dense_linear_algebra_library_type;
191 pp->linear_solver_options.use_explicit_schur_complement =
192 options.use_explicit_schur_complement;
193 pp->linear_solver_options.dynamic_sparsity = options.dynamic_sparsity;
194 pp->linear_solver_options.num_threads = options.num_linear_solver_threads;
195
196 // Ignore user's postordering preferences and force it to be true if
197 // cholmod_camd is not available. This ensures that the linear
198 // solver does not assume that a fill-reducing pre-ordering has been
199 // done.
200 pp->linear_solver_options.use_postordering = options.use_postordering;
201 if (options.linear_solver_type == SPARSE_SCHUR &&
202 options.sparse_linear_algebra_library_type == SUITE_SPARSE &&
203 !SuiteSparse::IsConstrainedApproximateMinimumDegreeOrderingAvailable()) {
204 pp->linear_solver_options.use_postordering = true;
205 }
206
207 OrderingToGroupSizes(options.linear_solver_ordering.get(),
208 &pp->linear_solver_options.elimination_groups);
209
210 // Schur type solvers expect at least two elimination groups. If
211 // there is only one elimination group, then it is guaranteed that
212 // this group only contains e_blocks. Thus we add a dummy
213 // elimination group with zero blocks in it.
214 if (IsSchurType(pp->linear_solver_options.type) &&
215 pp->linear_solver_options.elimination_groups.size() == 1) {
216 pp->linear_solver_options.elimination_groups.push_back(0);
217 }
218
219 pp->linear_solver.reset(LinearSolver::Create(pp->linear_solver_options));
220 return (pp->linear_solver.get() != NULL);
221}

223// Configure and create the evaluator.
224bool SetupEvaluator(PreprocessedProblem* pp) {
225 const Solver::Options& options = pp->options;
226 pp->evaluator_options = Evaluator::Options();
227 pp->evaluator_options.linear_solver_type = options.linear_solver_type;
228 pp->evaluator_options.num_eliminate_blocks = 0;
229 if (IsSchurType(options.linear_solver_type)) {
230 pp->evaluator_options.num_eliminate_blocks =
231 options
232 .linear_solver_ordering
233 ->group_to_elements().begin()
234 ->second.size();
235 }
236
237 pp->evaluator_options.num_threads = options.num_threads;
238 pp->evaluator_options.dynamic_sparsity = options.dynamic_sparsity;
239 pp->evaluator.reset(Evaluator::Create(pp->evaluator_options,
240 pp->reduced_program.get(),
241 &pp->error));
242
243 return (pp->evaluator.get() != NULL);
244}

246// If the user requested inner iterations, then find an inner
247// iteration ordering as needed and configure and create a
248// CoordinateDescentMinimizer object to perform the inner iterations.
249bool SetupInnerIterationMinimizer(PreprocessedProblem* pp) {
250 Solver::Options& options = pp->options;
251 if (!options.use_inner_iterations) {
252 return true;
253 }
254
255 // With just one parameter block, the outer iteration of the trust
256 // region method and inner iterations are doing exactly the same
257 // thing, and thus inner iterations are not needed.
258 if (pp->reduced_program->NumParameterBlocks() == 1) {
259 LOG(WARNING) << "Reduced problem only contains one parameter block."
260 << "Disabling inner iterations.";
261 return true;
262 }
263
264 if (options.inner_iteration_ordering.get() != NULL) {
265 // If the user supplied an ordering, then remove the set of
266 // inactive parameter blocks from it
267 options.inner_iteration_ordering->Remove(pp->removed_parameter_blocks);
268 if (options.inner_iteration_ordering->NumElements() == 0) {
269 LOG(WARNING) << "No remaining elements in the inner iteration ordering.";
270 return true;
271 }
272
273 // Validate the reduced ordering.
274 if (!CoordinateDescentMinimizer::IsOrderingValid(
275 *pp->reduced_program,
276 *options.inner_iteration_ordering,
277 &pp->error)) {
278 return false;
279 }
280 } else {
281 // The user did not supply an ordering, so create one.
282 options.inner_iteration_ordering.reset(
283 CoordinateDescentMinimizer::CreateOrdering(*pp->reduced_program));
284 }
285
286 pp->inner_iteration_minimizer.reset(new CoordinateDescentMinimizer);
287 return pp->inner_iteration_minimizer->Init(*pp->reduced_program,
288 pp->problem->parameter_map(),
289 *options.inner_iteration_ordering,
290 &pp->error);
291}

293// Configure and create a TrustRegionMinimizer object.
294void SetupMinimizerOptions(PreprocessedProblem* pp) {
295 const Solver::Options& options = pp->options;
296
297 SetupCommonMinimizerOptions(pp);
298 pp->minimizer_options.is_constrained =
299 pp->reduced_program->IsBoundsConstrained();
300 pp->minimizer_options.jacobian.reset(pp->evaluator->CreateJacobian());
301 pp->minimizer_options.inner_iteration_minimizer =
302 pp->inner_iteration_minimizer;
303
304 TrustRegionStrategy::Options strategy_options;
305 strategy_options.linear_solver = pp->linear_solver.get();
306 strategy_options.initial_radius =
307 options.initial_trust_region_radius;
308 strategy_options.max_radius = options.max_trust_region_radius;
309 strategy_options.min_lm_diagonal = options.min_lm_diagonal;
310 strategy_options.max_lm_diagonal = options.max_lm_diagonal;
311 strategy_options.trust_region_strategy_type =
312 options.trust_region_strategy_type;
313 strategy_options.dogleg_type = options.dogleg_type;
314 pp->minimizer_options.trust_region_strategy.reset(
315 CHECK_NOTNULL(TrustRegionStrategy::Create(strategy_options)));
316}

323bool TrustRegionPreprocessor::Preprocess(const Solver::Options& options,
324 ProblemImpl* problem,
325 PreprocessedProblem* pp) {
326 CHECK_NOTNULL(pp);
327 pp->options = options;
328 ChangeNumThreadsIfNeeded(&pp->options);
329
330 pp->problem = problem;
331 Program* program = problem->mutable_program();
332 if (!IsProgramValid(*program, &pp->error)) {
333 return false;
334 }
335
336 pp->reduced_program.reset(
337 program->CreateReducedProgram(&pp->removed_parameter_blocks,
338 &pp->fixed_cost,
339 &pp->error));
340
341 if (pp->reduced_program.get() == NULL) {
342 return false;
343 }
344
345 if (pp->reduced_program->NumParameterBlocks() == 0) {
346 // The reduced problem has no parameter or residual blocks. There
347 // is nothing more to do.
348 return true;
349 }
350
351 if (!SetupLinearSolver(pp) ||
352 !SetupEvaluator(pp) ||
353 !SetupInnerIterationMinimizer(pp)) {
354 return false;
355 }
356
357 SetupMinimizerOptions(pp);
358 return true;
359}
