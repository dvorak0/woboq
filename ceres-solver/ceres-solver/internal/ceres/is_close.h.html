32
33#ifndef CERES_INTERNAL_IS_CLOSE_H_
34#define CERES_INTERNAL_IS_CLOSE_H_
35
36namespace ceres {
37namespace internal {
38// Returns true if x and y have a relative (unsigned) difference less than
39// relative_precision and false otherwise. Stores the relative and absolute
40// difference in relative/absolute_error if non-NULL. If one of the two values
41// is exactly zero, the absolute difference will be compared, and relative_error
42// will be set to the absolute difference.
43bool IsClose(double x,
44 double y,
45 double relative_precision,
46 double *relative_error,
47 double *absolute_error);
48} // namespace internal
49} // namespace ceres
50
51#endif // CERES_INTERNAL_IS_CLOSE_H_
