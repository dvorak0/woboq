45bool IsProgramValid(const Program& program, std::string* error) {
46 if (program.IsBoundsConstrained()) {
47 *error = "LINE_SEARCH Minimizer does not support bounds.";
48 return false;
49 }
50 return program.ParameterBlocksAreFinite(error);
51}
52
53bool SetupEvaluator(PreprocessedProblem* pp) {
54 pp->evaluator_options = Evaluator::Options();
55 // This ensures that we get a Block Jacobian Evaluator without any
56 // requirement on orderings.
57 pp->evaluator_options.linear_solver_type = CGNR;
58 pp->evaluator_options.num_eliminate_blocks = 0;
59 pp->evaluator_options.num_threads = pp->options.num_threads;
60 pp->evaluator.reset(Evaluator::Create(pp->evaluator_options,
61 pp->reduced_program.get(),
62 &pp->error));
63 return (pp->evaluator.get() != NULL);
64}

71bool LineSearchPreprocessor::Preprocess(const Solver::Options& options,
72 ProblemImpl* problem,
73 PreprocessedProblem* pp) {
74 CHECK_NOTNULL(pp);
75 pp->options = options;
76 ChangeNumThreadsIfNeeded(&pp->options);
77
78 pp->problem = problem;
79 Program* program = problem->mutable_program();
80 if (!IsProgramValid(*program, &pp->error)) {
81 return false;
82 }
83
84 pp->reduced_program.reset(
85 program->CreateReducedProgram(&pp->removed_parameter_blocks,
86 &pp->fixed_cost,
87 &pp->error));
88
89 if (pp->reduced_program.get() == NULL) {
90 return false;
91 }
92
93 if (pp->reduced_program->NumParameterBlocks() == 0) {
94 return true;
95 }
96
97 if (!SetupEvaluator(pp)) {
98 return false;
99 }
100
101 SetupCommonMinimizerOptions(pp);
102 return true;
103}
