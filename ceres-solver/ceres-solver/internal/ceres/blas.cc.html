48namespace ceres {
49namespace internal {
50
51void BLAS::SymmetricRankKUpdate(int num_rows,
52 int num_cols,
53 const double* a,
54 bool transpose,
55 double alpha,
56 double beta,
57 double* c) {
58#ifdef CERES_NO_LAPACK
59 LOG(FATAL) << "Ceres was built without a BLAS library.";
60#else
61 char uplo = 'L';
62 char trans = transpose ? 'T' : 'N';
63 int n = transpose ? num_cols : num_rows;
64 int k = transpose ? num_rows : num_cols;
65 int lda = k;
66 int ldc = n;
67 dsyrk_(&uplo,
68 &trans,
69 &n,
70 &k,
71 &alpha,
72 const_cast<double*>(a),
73 &lda,
74 &beta,
75 c,
76 &ldc);
77#endif
78}
