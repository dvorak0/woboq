47// Templated base class for the CostFunction signatures.
48template <int kNumResiduals, int N0, int N1, int N2>
49class MockCostFunctionBase : public
50SizedCostFunction<kNumResiduals, N0, N1, N2> {
51 public:
52 virtual bool Evaluate(double const* const* parameters,
53 double* residuals,
54 double** jacobians) const {
55 // Do nothing. This is never called.
56 return true;
57 }
58};

60class UnaryCostFunction : public MockCostFunctionBase<2, 1, 0, 0> {};
61class BinaryCostFunction : public MockCostFunctionBase<2, 1, 1, 0> {};
62class TernaryCostFunction : public MockCostFunctionBase<2, 1, 1, 1> {};

64TEST(_, ReorderResidualBlockNormalFunction) {
65 ProblemImpl problem;
66 double x;
67 double y;
68 double z;
69
70 problem.AddParameterBlock(&x, 1);
71 problem.AddParameterBlock(&y, 1);
72 problem.AddParameterBlock(&z, 1);
73
74 problem.AddResidualBlock(new UnaryCostFunction(), NULL, &x);
75 problem.AddResidualBlock(new BinaryCostFunction(), NULL, &z, &x);
76 problem.AddResidualBlock(new BinaryCostFunction(), NULL, &z, &y);
77 problem.AddResidualBlock(new UnaryCostFunction(), NULL, &z);
78 problem.AddResidualBlock(new BinaryCostFunction(), NULL, &x, &y);
79 problem.AddResidualBlock(new UnaryCostFunction(), NULL, &y);
80
81 ParameterBlockOrdering* linear_solver_ordering = new ParameterBlockOrdering;
82 linear_solver_ordering->AddElementToGroup(&x, 0);
83 linear_solver_ordering->AddElementToGroup(&y, 0);
84 linear_solver_ordering->AddElementToGroup(&z, 1);
85
86 Solver::Options options;
87 options.linear_solver_type = DENSE_SCHUR;
88 options.linear_solver_ordering.reset(linear_solver_ordering);
89
90 const vector<ResidualBlock*>& residual_blocks =
91 problem.program().residual_blocks();
92
93 vector<ResidualBlock*> expected_residual_blocks;
94
95 // This is a bit fragile, but it serves the purpose. We know the
96 // bucketing algorithm that the reordering function uses, so we
97 // expect the order for residual blocks for each e_block to be
98 // filled in reverse.
99 expected_residual_blocks.push_back(residual_blocks[4]);
100 expected_residual_blocks.push_back(residual_blocks[1]);
101 expected_residual_blocks.push_back(residual_blocks[0]);
102 expected_residual_blocks.push_back(residual_blocks[5]);
103 expected_residual_blocks.push_back(residual_blocks[2]);
104 expected_residual_blocks.push_back(residual_blocks[3]);
105
106 Program* program = problem.mutable_program();
107 program->SetParameterOffsetsAndIndex();
108
109 std::string message;
110 EXPECT_TRUE(LexicographicallyOrderResidualBlocks(
111 2,
112 problem.mutable_program(),
113 &message));
114 EXPECT_EQ(residual_blocks.size(), expected_residual_blocks.size());
115 for (int i = 0; i < expected_residual_blocks.size(); ++i) {
116 EXPECT_EQ(residual_blocks[i], expected_residual_blocks[i]);
117 }
118}

142TEST(_, ApplyOrderingNormal) {
143 ProblemImpl problem;
144 double x;
145 double y;
146 double z;
147
148 problem.AddParameterBlock(&x, 1);
149 problem.AddParameterBlock(&y, 1);
150 problem.AddParameterBlock(&z, 1);
151
152 ParameterBlockOrdering linear_solver_ordering;
153 linear_solver_ordering.AddElementToGroup(&x, 0);
154 linear_solver_ordering.AddElementToGroup(&y, 2);
155 linear_solver_ordering.AddElementToGroup(&z, 1);
156
157 Program* program = problem.mutable_program();
158 std::string message;
159
160 EXPECT_TRUE(ApplyOrdering(problem.parameter_map(),
161 linear_solver_ordering,
162 program,
163 &message));
164 const vector<ParameterBlock*>& parameter_blocks = program->parameter_blocks();
165
166 EXPECT_EQ(parameter_blocks.size(), 3);
167 EXPECT_EQ(parameter_blocks[0]->user_state(), &x);
168 EXPECT_EQ(parameter_blocks[1]->user_state(), &z);
169 EXPECT_EQ(parameter_blocks[2]->user_state(), &y);
170}

172#ifndef CERES_NO_SUITESPARSE
173class ReorderProgramForSparseNormalCholeskyUsingSuiteSparseTest :
174 public ::testing::Test {
175 protected:
176 void SetUp() {
177 problem_.AddResidualBlock(new UnaryCostFunction(), NULL, &x_);
178 problem_.AddResidualBlock(new BinaryCostFunction(), NULL, &z_, &x_);
179 problem_.AddResidualBlock(new BinaryCostFunction(), NULL, &z_, &y_);
180 problem_.AddResidualBlock(new UnaryCostFunction(), NULL, &z_);
181 problem_.AddResidualBlock(new BinaryCostFunction(), NULL, &x_, &y_);
182 problem_.AddResidualBlock(new UnaryCostFunction(), NULL, &y_);
183 }
184
185 void ComputeAndValidateOrdering(
186 const ParameterBlockOrdering& linear_solver_ordering) {
187 Program* program = problem_.mutable_program();
188 vector<ParameterBlock*> unordered_parameter_blocks =
189 program->parameter_blocks();
190
191 std::string error;
192 EXPECT_TRUE(ReorderProgramForSparseNormalCholesky(
193 ceres::SUITE_SPARSE,
194 linear_solver_ordering,
195 program,
196 &error));
197 const vector<ParameterBlock*>& ordered_parameter_blocks =
198 program->parameter_blocks();
199 EXPECT_EQ(ordered_parameter_blocks.size(),
200 unordered_parameter_blocks.size());
201
202 EXPECT_THAT(unordered_parameter_blocks,
203 ::testing::UnorderedElementsAreArray(ordered_parameter_blocks));
204 }
205
206 ProblemImpl problem_;
207 double x_;
208 double y_;
209 double z_;
210};
