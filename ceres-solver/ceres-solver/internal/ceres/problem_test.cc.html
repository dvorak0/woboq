59// Trivial cost function that accepts a single argument.
60class UnaryCostFunction : public CostFunction {
61 public:
62 UnaryCostFunction(int num_residuals, int32 parameter_block_size) {
63 set_num_residuals(num_residuals);
64 mutable_parameter_block_sizes()->push_back(parameter_block_size);
65 }
66 virtual ~UnaryCostFunction() {}
67
68 virtual bool Evaluate(double const* const* parameters,
69 double* residuals,
70 double** jacobians) const {
71 for (int i = 0; i < num_residuals(); ++i) {
72 residuals[i] = 1;
73 }
74 return true;
75 }
76};

399// Make the dynamic problem tests (e.g. for removing residual blocks)
400// parameterized on whether the low-latency mode is enabled or not.
401//
402// This tests against ProblemImpl instead of Problem in order to inspect the
403// state of the resulting Program; this is difficult with only the thin Problem
404// interface.
405struct DynamicProblem : public ::testing::TestWithParam<bool> {
406 DynamicProblem() {
407 Problem::Options options;
408 options.enable_fast_removal = GetParam();
409 problem.reset(new ProblemImpl(options));
410 }
411
412 ParameterBlock* GetParameterBlock(int block) {
413 return problem->program().parameter_blocks()[block];
414 }
415 ResidualBlock* GetResidualBlock(int block) {
416 return problem->program().residual_blocks()[block];
417 }
418
419 bool HasResidualBlock(ResidualBlock* residual_block) {
420 bool have_residual_block = true;
421 if (GetParam()) {
422 have_residual_block &=
423 (problem->residual_block_set().find(residual_block) !=
424 problem->residual_block_set().end());
425 }
426 have_residual_block &=
427 find(problem->program().residual_blocks().begin(),
428 problem->program().residual_blocks().end(),
429 residual_block) != problem->program().residual_blocks().end();
430 return have_residual_block;
431 }
432
433 int NumResidualBlocks() {
434 // Verify that the hash set of residuals is maintained consistently.
435 if (GetParam()) {
436 EXPECT_EQ(problem->residual_block_set().size(),
437 problem->NumResidualBlocks());
438 }
439 return problem->NumResidualBlocks();
440 }

922 // Compute the size of a.

926 }
927 ASSERT_EQ(size, b.size());
928
929 // Sort a.
930 vector<T> a_sorted(size);
931 copy(a, a + size, a_sorted.begin());
932 sort(a_sorted.begin(), a_sorted.end());
933
934 // Sort b.
935 vector<T> b_sorted(b);
936 sort(b_sorted.begin(), b_sorted.end());
937
938 // Compare.
939 for (int i = 0; i < size; ++i) {
940 EXPECT_EQ(a_sorted[i], b_sorted[i]);

1061template <int kNumResiduals, int kNumParameterBlocks>
1062class QuadraticCostFunction : public CostFunction {
1063 public:
1064 QuadraticCostFunction() {
1065 CHECK_GT(kNumResiduals, 0);
1066 CHECK_GT(kNumParameterBlocks, 0);
1067 set_num_residuals(kNumResiduals);
1068 for (int i = 0; i < kNumParameterBlocks; ++i) {
1069 mutable_parameter_block_sizes()->push_back(kNumResiduals);
1070 }
1071 }
1072
1073 virtual bool Evaluate(double const* const* parameters,
1074 double* residuals,
1075 double** jacobians) const {
1076 for (int i = 0; i < kNumResiduals; ++i) {
1077 residuals[i] = i;
1078 for (int j = 0; j < kNumParameterBlocks; ++j) {
1079 residuals[i] -= (j + 1.0) * parameters[j][i] * parameters[j][i];
1080 }
1081 }
1082
1083 if (jacobians == NULL) {
1084 return true;
1085 }
1086
1087 for (int j = 0; j < kNumParameterBlocks; ++j) {
1088 if (jacobians[j] != NULL) {
1089 MatrixRef(jacobians[j], kNumResiduals, kNumResiduals) =
1090 (-2.0 * (j + 1.0) *
1091 ConstVectorRef(parameters[j], kNumResiduals)).asDiagonal();
1092 }
1093 }
1094
1095 return true;
1096 }
1097};

1102 m.resize(input.num_rows, input.num_cols);

1104 for (int row = 0; row < input.num_rows; ++row) {

1106 const int col = input.cols[j];

1108 }
