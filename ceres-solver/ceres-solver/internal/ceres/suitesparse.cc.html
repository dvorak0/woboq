58cholmod_sparse* SuiteSparse::CreateSparseMatrix(TripletSparseMatrix* A) {
59 cholmod_triplet triplet;
60
61 triplet.nrow = A->num_rows();
62 triplet.ncol = A->num_cols();
63 triplet.nzmax = A->max_num_nonzeros();
64 triplet.nnz = A->num_nonzeros();
65 triplet.i = reinterpret_cast<void*>(A->mutable_rows());
66 triplet.j = reinterpret_cast<void*>(A->mutable_cols());
67 triplet.x = reinterpret_cast<void*>(A->mutable_values());
68 triplet.stype = 0; // Matrix is not symmetric.
69 triplet.itype = CHOLMOD_INT;
70 triplet.xtype = CHOLMOD_REAL;
71 triplet.dtype = CHOLMOD_DOUBLE;
72
73 return cholmod_triplet_to_sparse(&triplet, triplet.nnz, &cc_);
74}

77cholmod_sparse* SuiteSparse::CreateSparseMatrixTranspose(
78 TripletSparseMatrix* A) {
79 cholmod_triplet triplet;
80
81 triplet.ncol = A->num_rows(); // swap row and columns
82 triplet.nrow = A->num_cols();
83 triplet.nzmax = A->max_num_nonzeros();
84 triplet.nnz = A->num_nonzeros();
85
86 // swap rows and columns
87 triplet.j = reinterpret_cast<void*>(A->mutable_rows());
88 triplet.i = reinterpret_cast<void*>(A->mutable_cols());
89 triplet.x = reinterpret_cast<void*>(A->mutable_values());
90 triplet.stype = 0; // Matrix is not symmetric.
91 triplet.itype = CHOLMOD_INT;
92 triplet.xtype = CHOLMOD_REAL;
93 triplet.dtype = CHOLMOD_DOUBLE;
94
95 return cholmod_triplet_to_sparse(&triplet, triplet.nnz, &cc_);
96}

98cholmod_sparse SuiteSparse::CreateSparseMatrixTransposeView(
99 CompressedRowSparseMatrix* A) {
100 cholmod_sparse m;
101 m.nrow = A->num_cols();
102 m.ncol = A->num_rows();
103 m.nzmax = A->num_nonzeros();
104 m.nz = NULL;
105 m.p = reinterpret_cast<void*>(A->mutable_rows());
106 m.i = reinterpret_cast<void*>(A->mutable_cols());
107 m.x = reinterpret_cast<void*>(A->mutable_values());
108 m.z = NULL;
109
110 if (A->storage_type() == CompressedRowSparseMatrix::LOWER_TRIANGULAR) {
111 m.stype = 1;
112 } else if (A->storage_type() == CompressedRowSparseMatrix::UPPER_TRIANGULAR) {
113 m.stype = -1;
114 } else {
115 m.stype = 0;
116 }
117
118 m.itype = CHOLMOD_INT;
119 m.xtype = CHOLMOD_REAL;
120 m.dtype = CHOLMOD_DOUBLE;
121 m.sorted = 1;
122 m.packed = 1;
123
124 return m;
125}

127cholmod_dense* SuiteSparse::CreateDenseVector(const double* x,
128 int in_size,
129 int out_size) {
130 CHECK_LE(in_size, out_size);
131 cholmod_dense* v = cholmod_zeros(out_size, 1, CHOLMOD_REAL, &cc_);
132 if (x != NULL) {
133 memcpy(v->x, x, in_size*sizeof(*x));
134 }
135 return v;
136}

138cholmod_factor* SuiteSparse::AnalyzeCholesky(cholmod_sparse* A,
139 string* message) {
140 // Cholmod can try multiple re-ordering strategies to find a fill
141 // reducing ordering. Here we just tell it use AMD with automatic
142 // matrix dependence choice of supernodal versus simplicial
143 // factorization.
144 cc_.nmethods = 1;
145 cc_.method[0].ordering = CHOLMOD_AMD;
146 cc_.supernodal = CHOLMOD_AUTO;
147
148 cholmod_factor* factor = cholmod_analyze(A, &cc_);
149 if (VLOG_IS_ON(2)) {
150 cholmod_print_common(const_cast<char*>("Symbolic Analysis"), &cc_);
151 }
152
153 if (cc_.status != CHOLMOD_OK) {
154 *message = StringPrintf("cholmod_analyze failed. error code: %d",
155 cc_.status);
156 return NULL;
157 }
158
159 return CHECK_NOTNULL(factor);
160}

162cholmod_factor* SuiteSparse::BlockAnalyzeCholesky(
163 cholmod_sparse* A,
164 const vector<int>& row_blocks,
165 const vector<int>& col_blocks,
166 string* message) {
167 vector<int> ordering;
168 if (!BlockAMDOrdering(A, row_blocks, col_blocks, &ordering)) {
169 return NULL;
170 }
171 return AnalyzeCholeskyWithUserOrdering(A, ordering, message);
172}

174cholmod_factor* SuiteSparse::AnalyzeCholeskyWithUserOrdering(
175 cholmod_sparse* A,
176 const vector<int>& ordering,
177 string* message) {
178 CHECK_EQ(ordering.size(), A->nrow);
179
180 cc_.nmethods = 1;
181 cc_.method[0].ordering = CHOLMOD_GIVEN;
182
183 cholmod_factor* factor =
184 cholmod_analyze_p(A, const_cast<int*>(&ordering[0]), NULL, 0, &cc_);
185 if (VLOG_IS_ON(2)) {
186 cholmod_print_common(const_cast<char*>("Symbolic Analysis"), &cc_);
187 }
188 if (cc_.status != CHOLMOD_OK) {
189 *message = StringPrintf("cholmod_analyze failed. error code: %d",
190 cc_.status);
191 return NULL;
192 }
193
194 return CHECK_NOTNULL(factor);
195}

217bool SuiteSparse::BlockAMDOrdering(const cholmod_sparse* A,
218 const vector<int>& row_blocks,
219 const vector<int>& col_blocks,
220 vector<int>* ordering) {
221 const int num_row_blocks = row_blocks.size();
222 const int num_col_blocks = col_blocks.size();
223
224 // Arrays storing the compressed column structure of the matrix
225 // incoding the block sparsity of A.
226 vector<int> block_cols;
227 vector<int> block_rows;
228
229 CompressedColumnScalarMatrixToBlockMatrix(reinterpret_cast<const int*>(A->i),
230 reinterpret_cast<const int*>(A->p),
231 row_blocks,
232 col_blocks,
233 &block_rows,
234 &block_cols);
235
236 cholmod_sparse_struct block_matrix;
237 block_matrix.nrow = num_row_blocks;
238 block_matrix.ncol = num_col_blocks;
239 block_matrix.nzmax = block_rows.size();
240 block_matrix.p = reinterpret_cast<void*>(&block_cols[0]);
241 block_matrix.i = reinterpret_cast<void*>(&block_rows[0]);
242 block_matrix.x = NULL;
243 block_matrix.stype = A->stype;
244 block_matrix.itype = CHOLMOD_INT;
245 block_matrix.xtype = CHOLMOD_PATTERN;
246 block_matrix.dtype = CHOLMOD_DOUBLE;
247 block_matrix.sorted = 1;
248 block_matrix.packed = 1;
249
250 vector<int> block_ordering(num_row_blocks);
251 if (!cholmod_amd(&block_matrix, NULL, 0, &block_ordering[0], &cc_)) {
252 return false;
253 }
254
255 BlockOrderingToScalarOrdering(row_blocks, block_ordering, ordering);
256 return true;
257}

259LinearSolverTerminationType SuiteSparse::Cholesky(cholmod_sparse* A,
260 cholmod_factor* L,
261 string* message) {
262 CHECK_NOTNULL(A);
263 CHECK_NOTNULL(L);
264
265 // Save the current print level and silence CHOLMOD, otherwise
266 // CHOLMOD is prone to dumping stuff to stderr, which can be
267 // distracting when the error (matrix is indefinite) is not a fatal
268 // failure.
269 const int old_print_level = cc_.print;
270 cc_.print = 0;
271
272 cc_.quick_return_if_not_posdef = 1;
273 int cholmod_status = cholmod_factorize(A, L, &cc_);
274 cc_.print = old_print_level;
275
276 // TODO(sameeragarwal): This switch statement is not consistent. It
277 // treats all kinds of CHOLMOD failures as warnings. Some of these
278 // like out of memory are definitely not warnings. The problem is
279 // that the return value Cholesky is two valued, but the state of
280 // the linear solver is really three valued. SUCCESS,
281 // NON_FATAL_FAILURE (e.g., indefinite matrix) and FATAL_FAILURE
282 // (e.g. out of memory).
283 switch (cc_.status) {
284 case CHOLMOD_NOT_INSTALLED:
285 *message = "CHOLMOD failure: Method not installed.";
286 return LINEAR_SOLVER_FATAL_ERROR;
287 case CHOLMOD_OUT_OF_MEMORY:
288 *message = "CHOLMOD failure: Out of memory.";
289 return LINEAR_SOLVER_FATAL_ERROR;
290 case CHOLMOD_TOO_LARGE:
291 *message = "CHOLMOD failure: Integer overflow occurred.";
292 return LINEAR_SOLVER_FATAL_ERROR;
293 case CHOLMOD_INVALID:
294 *message = "CHOLMOD failure: Invalid input.";
295 return LINEAR_SOLVER_FATAL_ERROR;
296 case CHOLMOD_NOT_POSDEF:
297 *message = "CHOLMOD warning: Matrix not positive definite.";
298 return LINEAR_SOLVER_FAILURE;
299 case CHOLMOD_DSMALL:
300 *message = "CHOLMOD warning: D for LDL' or diag(L) or "
301 "LL' has tiny absolute value.";
302 return LINEAR_SOLVER_FAILURE;
303 case CHOLMOD_OK:
304 if (cholmod_status != 0) {
305 return LINEAR_SOLVER_SUCCESS;
306 }
307
308 *message = "CHOLMOD failure: cholmod_factorize returned false "
309 "but cholmod_common::status is CHOLMOD_OK."
310 "Please report this to ceres-solver@googlegroups.com.";
311 return LINEAR_SOLVER_FATAL_ERROR;
312 default:
313 *message =
314 StringPrintf("Unknown cholmod return code: %d. "
315 "Please report this to ceres-solver@googlegroups.com.",
316 cc_.status);
317 return LINEAR_SOLVER_FATAL_ERROR;
318 }
319
320 return LINEAR_SOLVER_FATAL_ERROR;
321}

323cholmod_dense* SuiteSparse::Solve(cholmod_factor* L,
324 cholmod_dense* b,
325 string* message) {
326 if (cc_.status != CHOLMOD_OK) {
327 *message = "cholmod_solve failed. CHOLMOD status is not CHOLMOD_OK";
328 return NULL;
329 }
330
331 return cholmod_solve(CHOLMOD_A, L, b, &cc_);
332}

339bool SuiteSparse::ConstrainedApproximateMinimumDegreeOrdering(
340 cholmod_sparse* matrix,
341 int* constraints,
342 int* ordering) {
343#ifndef CERES_NO_CAMD
344 return cholmod_camd(matrix, NULL, 0, constraints, ordering, &cc_);
345#else
346 LOG(FATAL) << "Congratulations you have found a bug in Ceres."
347 << "Ceres Solver was compiled with SuiteSparse "
348 << "version 4.1.0 or less. Calling this function "
349 << "in that case is a bug. Please contact the"
350 << "the Ceres Solver developers.";
351 return false;
352#endif
353}
