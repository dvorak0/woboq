47const double kE = 2.71828182845904523536;
48
49typedef Jet<double, 2> J;
50
51// Convenient shorthand for making a jet.
52J MakeJet(double a, double v0, double v1) {
53 J z;
54 z.a = a;
55 z.v[0] = v0;
56 z.v[1] = v1;
57 return z;
58}
59
60// On a 32-bit optimized build, the mismatch is about 1.4e-14.
61double const kTolerance = 1e-13;
62
63void ExpectJetsClose(const J &x, const J &y) {
64 ExpectClose(x.a, y.a, kTolerance);
65 ExpectClose(x.v[0], y.v[0], kTolerance);
66 ExpectClose(x.v[1], y.v[1], kTolerance);
67}

516TEST(Jet, JetsInEigenMatrices) {
517 J x = MakeJet(2.3, -2.7, 1e-3);
518 J y = MakeJet(1.7, 0.5, 1e+2);
519 J z = MakeJet(5.3, -4.7, 1e-3);
520 J w = MakeJet(9.7, 1.5, 10.1);
521
522 Eigen::Matrix<J, 2, 2> M;
523 Eigen::Matrix<J, 2, 1> v, r1, r2;
524
525 M << x, y, z, w;
526 v << x, z;
527
528 // Check that M * v == (v^T * M^T)^T
529 r1 = M * v;
530 r2 = (v.transpose() * M.transpose()).transpose();
531
532 ExpectJetsClose(r1(0), r2(0));
533 ExpectJetsClose(r1(1), r2(1));
534}

593TEST(JetTraitsTest, MatrixScalarBinaryOps) {
594 const J x = MakeJet(2.3, -2.7, 1e-3);
595 const J y = MakeJet(1.7, 0.5, 1e+2);
596 const J z = MakeJet(5.3, -4.7, 1e-3);
597 const J w = MakeJet(9.7, 1.5, 10.1);
598
599 Eigen::Matrix<J, 2, 2> M;
600 Eigen::Vector2d v;
601
602 M << x, y, z, w;
603 v << 0.6, -2.1;
604
605 // Check that M * v == M * v.cast<J>().
606 const Eigen::Matrix<J, 2, 1> r1 = M * v;
607 const Eigen::Matrix<J, 2, 1> r2 = M * v.cast<J>();
608
609 ExpectJetsClose(r1(0), r2(0));
610 ExpectJetsClose(r1(1), r2(1));
611
612 // Check that M * a == M * T(a).
613 const double a = 3.1;
614 const Eigen::Matrix<J, 2, 2> r3 = M * a;
615 const Eigen::Matrix<J, 2, 2> r4 = M * J(a);
616
617 ExpectJetsClose(r3(0, 0), r4(0, 0));
618 ExpectJetsClose(r3(1, 0), r4(1, 0));
619 ExpectJetsClose(r3(0, 1), r4(0, 1));
620 ExpectJetsClose(r3(1, 1), r4(1, 1));
621}
