53enum LinearSolverTerminationType {
54 // Termination criterion was met.
55 LINEAR_SOLVER_SUCCESS,
56
57 // Solver ran for max_num_iterations and terminated before the
58 // termination tolerance could be satisfied.
59 LINEAR_SOLVER_NO_CONVERGENCE,
60
61 // Solver was terminated due to numerical problems, generally due to
62 // the linear system being poorly conditioned.
63 LINEAR_SOLVER_FAILURE,
64
65 // Solver failed with a fatal error that cannot be recovered from,
66 // e.g. CHOLMOD ran out of memory when computing the symbolic or
67 // numeric factorization or an underlying library was called with
68 // the wrong arguments.
69 LINEAR_SOLVER_FATAL_ERROR
70};

75// Abstract base class for objects that implement algorithms for
76// solving linear systems
77//
78// Ax = b
79//
80// It is expected that a single instance of a LinearSolver object
81// maybe used multiple times for solving multiple linear systems with
82// the same sparsity structure. This allows them to cache and reuse
83// information across solves. This means that calling Solve on the
84// same LinearSolver instance with two different linear systems will
85// result in undefined behaviour.
86//
87// Subclasses of LinearSolver use two structs to configure themselves.
88// The Options struct configures the LinearSolver object for its
89// lifetime. The PerSolveOptions struct is used to specify options for
90// a particular Solve call.
91class LinearSolver {
92 public:
93 struct Options {
94 Options()
95 : type(SPARSE_NORMAL_CHOLESKY),
96 preconditioner_type(JACOBI),
97 visibility_clustering_type(CANONICAL_VIEWS),
98 dense_linear_algebra_library_type(EIGEN),
99 sparse_linear_algebra_library_type(SUITE_SPARSE),
100 use_postordering(false),
101 dynamic_sparsity(false),
102 use_explicit_schur_complement(false),
103 min_num_iterations(1),
104 max_num_iterations(1),
105 num_threads(1),
106 residual_reset_period(10),
107 row_block_size(Eigen::Dynamic),
108 e_block_size(Eigen::Dynamic),
109 f_block_size(Eigen::Dynamic) {
110 }
111
112 LinearSolverType type;
113 PreconditionerType preconditioner_type;
114 VisibilityClusteringType visibility_clustering_type;
115 DenseLinearAlgebraLibraryType dense_linear_algebra_library_type;
116 SparseLinearAlgebraLibraryType sparse_linear_algebra_library_type;
117
118 // See solver.h for information about these flags.
119 bool use_postordering;
120 bool dynamic_sparsity;
121 bool use_explicit_schur_complement;
122
123 // Number of internal iterations that the solver uses. This
124 // parameter only makes sense for iterative solvers like CG.
125 int min_num_iterations;
126 int max_num_iterations;
127
128 // If possible, how many threads can the solver use.
129 int num_threads;
130
131 // Hints about the order in which the parameter blocks should be
132 // eliminated by the linear solver.
133 //
134 // For example if elimination_groups is a vector of size k, then
135 // the linear solver is informed that it should eliminate the
136 // parameter blocks 0 ... elimination_groups[0] - 1 first, and
137 // then elimination_groups[0] ... elimination_groups[1] - 1 and so
138 // on. Within each elimination group, the linear solver is free to
139 // choose how the parameter blocks are ordered. Different linear
140 // solvers have differing requirements on elimination_groups.
141 //
142 // The most common use is for Schur type solvers, where there
143 // should be at least two elimination groups and the first
144 // elimination group must form an independent set in the normal
145 // equations. The first elimination group corresponds to the
146 // num_eliminate_blocks in the Schur type solvers.
147 std::vector<int> elimination_groups;
148
149 // Iterative solvers, e.g. Preconditioned Conjugate Gradients
150 // maintain a cheap estimate of the residual which may become
151 // inaccurate over time. Thus for non-zero values of this
152 // parameter, the solver can be told to recalculate the value of
153 // the residual using a |b - Ax| evaluation.
154 int residual_reset_period;
155
156 // If the block sizes in a BlockSparseMatrix are fixed, then in
157 // some cases the Schur complement based solvers can detect and
158 // specialize on them.
159 //
160 // It is expected that these parameters are set programmatically
161 // rather than manually.
162 //
163 // Please see schur_complement_solver.h and schur_eliminator.h for
164 // more details.
165 int row_block_size;
166 int e_block_size;
167 int f_block_size;
168 };

170 // Options for the Solve method.
171 struct PerSolveOptions {
172 PerSolveOptions()
173 : D(NULL),
174 preconditioner(NULL),
175 r_tolerance(0.0),
176 q_tolerance(0.0) {
177 }
178
179 // This option only makes sense for unsymmetric linear solvers
180 // that can solve rectangular linear systems.
181 //
182 // Given a matrix A, an optional diagonal matrix D as a vector,
183 // and a vector b, the linear solver will solve for
184 //
185 // | A | x = | b |
186 // | D | | 0 |
187 //
188 // If D is null, then it is treated as zero, and the solver returns
189 // the solution to
190 //
191 // A x = b
192 //
193 // In either case, x is the vector that solves the following
194 // optimization problem.
195 //
196 // arg min_x ||Ax - b||^2 + ||Dx||^2
197 //
198 // Here A is a matrix of size m x n, with full column rank. If A
199 // does not have full column rank, the results returned by the
200 // solver cannot be relied on. D, if it is not null is an array of
201 // size n. b is an array of size m and x is an array of size n.
202 double * D;
203
204 // This option only makes sense for iterative solvers.
205 //
206 // In general the performance of an iterative linear solver
207 // depends on the condition number of the matrix A. For example
208 // the convergence rate of the conjugate gradients algorithm
209 // is proportional to the square root of the condition number.
210 //
211 // One particularly useful technique for improving the
212 // conditioning of a linear system is to precondition it. In its
213 // simplest form a preconditioner is a matrix M such that instead
214 // of solving Ax = b, we solve the linear system AM^{-1} y = b
215 // instead, where M is such that the condition number k(AM^{-1})
216 // is smaller than the conditioner k(A). Given the solution to
217 // this system, x = M^{-1} y. The iterative solver takes care of
218 // the mechanics of solving the preconditioned system and
219 // returning the corrected solution x. The user only needs to
220 // supply a linear operator.
221 //
222 // A null preconditioner is equivalent to an identity matrix being
223 // used a preconditioner.
224 LinearOperator* preconditioner;
225
226
227 // The following tolerance related options only makes sense for
228 // iterative solvers. Direct solvers ignore them.
229
230 // Solver terminates when
231 //
232 // |Ax - b| <= r_tolerance * |b|.
233 //
234 // This is the most commonly used termination criterion for
235 // iterative solvers.
236 double r_tolerance;
237
238 // For PSD matrices A, let
239 //
240 // Q(x) = x'Ax - 2b'x
241 //
242 // be the cost of the quadratic function defined by A and b. Then,
243 // the solver terminates at iteration i if
244 //
245 // i * (Q(x_i) - Q(x_i-1)) / Q(x_i) < q_tolerance.
246 //
247 // This termination criterion is more useful when using CG to
248 // solve the Newton step. This particular convergence test comes
249 // from Stephen Nash's work on truncated Newton
250 // methods. References:
251 //
252 // 1. Stephen G. Nash & Ariela Sofer, Assessing A Search
253 // Direction Within A Truncated Newton Method, Operation
254 // Research Letters 9(1990) 219-221.
255 //
256 // 2. Stephen G. Nash, A Survey of Truncated Newton Methods,
257 // Journal of Computational and Applied Mathematics,
258 // 124(1-2), 45-59, 2000.
259 //
260 double q_tolerance;
261 };
262
263 // Summary of a call to the Solve method. We should move away from
264 // the true/false method for determining solver success. We should
265 // let the summary object do the talking.
266 struct Summary {
267 Summary()
268 : residual_norm(0.0),
269 num_iterations(-1),
270 termination_type(LINEAR_SOLVER_FAILURE) {
271 }
272
273 double residual_norm;
274 int num_iterations;
275 LinearSolverTerminationType termination_type;
276 std::string message;
277 };
278
279 // If the optimization problem is such that there are no remaining
280 // e-blocks, a Schur type linear solver cannot be used. If the
281 // linear solver is of Schur type, this function implements a policy
282 // to select an alternate nearest linear solver to the one selected
283 // by the user. The input linear_solver_type is returned otherwise.
284 static LinearSolverType LinearSolverForZeroEBlocks(
285 LinearSolverType linear_solver_type);
286
287 virtual ~LinearSolver();
288
289 // Solve Ax = b.
290 virtual Summary Solve(LinearOperator* A,
291 const double* b,
292 const PerSolveOptions& per_solve_options,
293 double* x) = 0;
294
295 // The following two methods return copies instead of references so
296 // that the base class implementation does not have to worry about
297 // life time issues. Further, these calls are not expected to be
298 // frequent or performance sensitive.
299 virtual std::map<std::string, int> CallStatistics() const {
300 return std::map<std::string, int>();
301 }
302
303 virtual std::map<std::string, double> TimeStatistics() const {
304 return std::map<std::string, double>();
305 }
306
307 // Factory
308 static LinearSolver* Create(const Options& options);
309};

311// This templated subclass of LinearSolver serves as a base class for
312// other linear solvers that depend on the particular matrix layout of
313// the underlying linear operator. For example some linear solvers
314// need low level access to the TripletSparseMatrix implementing the
315// LinearOperator interface. This class hides those implementation
316// details behind a private virtual method, and has the Solve method
317// perform the necessary upcasting.
318template <typename MatrixType>
319class TypedLinearSolver : public LinearSolver {
320 public:
321 virtual ~TypedLinearSolver() {}
322 virtual LinearSolver::Summary Solve(
323 LinearOperator* A,
324 const double* b,
325 const LinearSolver::PerSolveOptions& per_solve_options,
326 double* x) {
327 ScopedExecutionTimer total_time("LinearSolver::Solve", &execution_summary_);
328 CHECK_NOTNULL(A);
329 CHECK_NOTNULL(b);
330 CHECK_NOTNULL(x);
331 return SolveImpl(down_cast<MatrixType*>(A), b, per_solve_options, x);
332 }
333
334 virtual std::map<std::string, int> CallStatistics() const {
335 return execution_summary_.calls();
336 }
337
338 virtual std::map<std::string, double> TimeStatistics() const {
339 return execution_summary_.times();
340 }
341
342 private:
343 virtual LinearSolver::Summary SolveImpl(
344 MatrixType* A,
345 const double* b,
346 const LinearSolver::PerSolveOptions& per_solve_options,
347 double* x) = 0;
348
349 ExecutionSummary execution_summary_;
350};

352// Linear solvers that depend on acccess to the low level structure of

354typedef TypedLinearSolver<BlockSparseMatrix> BlockSparseMatrixSolver; // NOLINT
355typedef TypedLinearSolver<CompressedRowSparseMatrix> CompressedRowSparseMatrixSolver; // NOLINT
