77VisibilityBasedPreconditioner::VisibilityBasedPreconditioner(
78 const CompressedRowBlockStructure& bs,
79 const Preconditioner::Options& options)
80 : options_(options),
81 num_blocks_(0),
82 num_clusters_(0),
83 factor_(NULL) {
84 CHECK_GT(options_.elimination_groups.size(), 1);
85 CHECK_GT(options_.elimination_groups[0], 0);
86 CHECK(options_.type == CLUSTER_JACOBI ||
87 options_.type == CLUSTER_TRIDIAGONAL)
88 << "Unknown preconditioner type: " << options_.type;
89 num_blocks_ = bs.cols.size() - options_.elimination_groups[0];
90 CHECK_GT(num_blocks_, 0)
91 << "Jacobian should have atleast 1 f_block for "
92 << "visibility based preconditioning.";
93
94 // Vector of camera block sizes
95 block_size_.resize(num_blocks_);
96 for (int i = 0; i < num_blocks_; ++i) {
97 block_size_[i] = bs.cols[i + options_.elimination_groups[0]].size;
98 }
99
100 const time_t start_time = time(NULL);
101 switch (options_.type) {
102 case CLUSTER_JACOBI:
103 ComputeClusterJacobiSparsity(bs);
104 break;
105 case CLUSTER_TRIDIAGONAL:
106 ComputeClusterTridiagonalSparsity(bs);
107 break;
108 default:
109 LOG(FATAL) << "Unknown preconditioner type";
110 }
111 const time_t structure_time = time(NULL);
112 InitStorage(bs);
113 const time_t storage_time = time(NULL);
114 InitEliminator(bs);
115 const time_t eliminator_time = time(NULL);
116
117 // Allocate temporary storage for a vector used during
118 // RightMultiply.
119 tmp_rhs_ = CHECK_NOTNULL(ss_.CreateDenseVector(NULL,
120 m_->num_rows(),
121 m_->num_rows()));
122 const time_t init_time = time(NULL);
123 VLOG(2) << "init time: "
124 << init_time - start_time
125 << " structure time: " << structure_time - start_time
126 << " storage time:" << storage_time - structure_time
127 << " eliminator time: " << eliminator_time - storage_time;
128}

130VisibilityBasedPreconditioner::~VisibilityBasedPreconditioner() {
131 if (factor_ != NULL) {
132 ss_.Free(factor_);
133 factor_ = NULL;
134 }
135 if (tmp_rhs_ != NULL) {
136 ss_.Free(tmp_rhs_);
137 tmp_rhs_ = NULL;
138 }
139}

141// Determine the sparsity structure of the CLUSTER_JACOBI
142// preconditioner. It clusters cameras using their scene
143// visibility. The clusters form the diagonal blocks of the
144// preconditioner matrix.
145void VisibilityBasedPreconditioner::ComputeClusterJacobiSparsity(
146 const CompressedRowBlockStructure& bs) {
147 vector<set<int> > visibility;
148 ComputeVisibility(bs, options_.elimination_groups[0], &visibility);
149 CHECK_EQ(num_blocks_, visibility.size());
150 ClusterCameras(visibility);
151 cluster_pairs_.clear();
152 for (int i = 0; i < num_clusters_; ++i) {
153 cluster_pairs_.insert(make_pair(i, i));
154 }
155}
156
157// Determine the sparsity structure of the CLUSTER_TRIDIAGONAL
158// preconditioner. It clusters cameras using using the scene
159// visibility and then finds the strongly interacting pairs of
160// clusters by constructing another graph with the clusters as
161// vertices and approximating it with a degree-2 maximum spanning
162// forest. The set of edges in this forest are the cluster pairs.
163void VisibilityBasedPreconditioner::ComputeClusterTridiagonalSparsity(
164 const CompressedRowBlockStructure& bs) {
165 vector<set<int> > visibility;
166 ComputeVisibility(bs, options_.elimination_groups[0], &visibility);
167 CHECK_EQ(num_blocks_, visibility.size());
168 ClusterCameras(visibility);
169
170 // Construct a weighted graph on the set of clusters, where the
171 // edges are the number of 3D points/e_blocks visible in both the
172 // clusters at the ends of the edge. Return an approximate degree-2
173 // maximum spanning forest of this graph.
174 vector<set<int> > cluster_visibility;
175 ComputeClusterVisibility(visibility, &cluster_visibility);
176 scoped_ptr<WeightedGraph<int> > cluster_graph(
177 CHECK_NOTNULL(CreateClusterGraph(cluster_visibility)));
178 scoped_ptr<WeightedGraph<int> > forest(
179 CHECK_NOTNULL(Degree2MaximumSpanningForest(*cluster_graph)));
180 ForestToClusterPairs(*forest, &cluster_pairs_);
181}

183// Allocate storage for the preconditioner matrix.
184void VisibilityBasedPreconditioner::InitStorage(
185 const CompressedRowBlockStructure& bs) {
186 ComputeBlockPairsInPreconditioner(bs);
187 m_.reset(new BlockRandomAccessSparseMatrix(block_size_, block_pairs_));
188}

190// Call the canonical views algorithm and cluster the cameras based on
191// their visibility sets. The visibility set of a camera is the set of
192// e_blocks/3D points in the scene that are seen by it.
193//
194// The cluster_membership_ vector is updated to indicate cluster
195// memberships for each camera block.
196void VisibilityBasedPreconditioner::ClusterCameras(
197 const vector<set<int> >& visibility) {
198 scoped_ptr<WeightedGraph<int> > schur_complement_graph(
199 CHECK_NOTNULL(CreateSchurComplementGraph(visibility)));
200
201 HashMap<int, int> membership;
202
203 if (options_.visibility_clustering_type == CANONICAL_VIEWS) {
204 vector<int> centers;
205 CanonicalViewsClusteringOptions clustering_options;
206 clustering_options.size_penalty_weight =
207 kCanonicalViewsSizePenaltyWeight;
208 clustering_options.similarity_penalty_weight =
209 kCanonicalViewsSimilarityPenaltyWeight;
210 ComputeCanonicalViewsClustering(clustering_options,
211 *schur_complement_graph,
212 &centers,
213 &membership);
214 num_clusters_ = centers.size();
215 } else if (options_.visibility_clustering_type == SINGLE_LINKAGE) {
216 SingleLinkageClusteringOptions clustering_options;
217 clustering_options.min_similarity =
218 kSingleLinkageMinSimilarity;
219 num_clusters_ = ComputeSingleLinkageClustering(clustering_options,
220 *schur_complement_graph,
221 &membership);
222 } else {
223 LOG(FATAL) << "Unknown visibility clustering algorithm.";
224 }
225
226 CHECK_GT(num_clusters_, 0);
227 VLOG(2) << "num_clusters: " << num_clusters_;
228 FlattenMembershipMap(membership, &cluster_membership_);
229}

231// Compute the block sparsity structure of the Schur complement
232// matrix. For each pair of cameras contributing a non-zero cell to
233// the schur complement, determine if that cell is present in the
234// preconditioner or not.
235//
236// A pair of cameras contribute a cell to the preconditioner if they
237// are part of the same cluster or if the the two clusters that they
238// belong have an edge connecting them in the degree-2 maximum
239// spanning forest.
240//
241// For example, a camera pair (i,j) where i belonges to cluster1 and
242// j belongs to cluster2 (assume that cluster1 < cluster2).
243//
244// The cell corresponding to (i,j) is present in the preconditioner
245// if cluster1 == cluster2 or the pair (cluster1, cluster2) were
246// connected by an edge in the degree-2 maximum spanning forest.
247//
248// Since we have already expanded the forest into a set of camera
249// pairs/edges, including self edges, the check can be reduced to
250// checking membership of (cluster1, cluster2) in cluster_pairs_.
251void VisibilityBasedPreconditioner::ComputeBlockPairsInPreconditioner(
252 const CompressedRowBlockStructure& bs) {
253 block_pairs_.clear();
254 for (int i = 0; i < num_blocks_; ++i) {
255 block_pairs_.insert(make_pair(i, i));
256 }
257
258 int r = 0;
259 const int num_row_blocks = bs.rows.size();
260 const int num_eliminate_blocks = options_.elimination_groups[0];
261
262 // Iterate over each row of the matrix. The block structure of the
263 // matrix is assumed to be sorted in order of the e_blocks/point
264 // blocks. Thus all row blocks containing an e_block/point occur
265 // contiguously. Further, if present, an e_block is always the first
266 // parameter block in each row block. These structural assumptions
267 // are common to all Schur complement based solvers in Ceres.
268 //
269 // For each e_block/point block we identify the set of cameras
270 // seeing it. The cross product of this set with itself is the set
271 // of non-zero cells contibuted by this e_block.
272 //
273 // The time complexity of this is O(nm^2) where, n is the number of
274 // 3d points and m is the maximum number of cameras seeing any
275 // point, which for most scenes is a fairly small number.
276 while (r < num_row_blocks) {
277 int e_block_id = bs.rows[r].cells.front().block_id;
278 if (e_block_id >= num_eliminate_blocks) {
279 // Skip the rows whose first block is an f_block.
280 break;
281 }
282
283 set<int> f_blocks;
284 for (; r < num_row_blocks; ++r) {
285 const CompressedRow& row = bs.rows[r];
286 if (row.cells.front().block_id != e_block_id) {
287 break;
288 }
289
290 // Iterate over the blocks in the row, ignoring the first block
291 // since it is the one to be eliminated and adding the rest to
292 // the list of f_blocks associated with this e_block.
293 for (int c = 1; c < row.cells.size(); ++c) {
294 const Cell& cell = row.cells[c];
295 const int f_block_id = cell.block_id - num_eliminate_blocks;
296 CHECK_GE(f_block_id, 0);
297 f_blocks.insert(f_block_id);
298 }
299 }
300
301 for (set<int>::const_iterator block1 = f_blocks.begin();
302 block1 != f_blocks.end();
303 ++block1) {
304 set<int>::const_iterator block2 = block1;
305 ++block2;
306 for (; block2 != f_blocks.end(); ++block2) {
307 if (IsBlockPairInPreconditioner(*block1, *block2)) {
308 block_pairs_.insert(make_pair(*block1, *block2));
309 }
310 }
311 }
312 }
313
314 // The remaining rows which do not contain any e_blocks.
315 for (; r < num_row_blocks; ++r) {
316 const CompressedRow& row = bs.rows[r];
317 CHECK_GE(row.cells.front().block_id, num_eliminate_blocks);
318 for (int i = 0; i < row.cells.size(); ++i) {
319 const int block1 = row.cells[i].block_id - num_eliminate_blocks;
320 for (int j = 0; j < row.cells.size(); ++j) {
321 const int block2 = row.cells[j].block_id - num_eliminate_blocks;
322 if (block1 <= block2) {
323 if (IsBlockPairInPreconditioner(block1, block2)) {
324 block_pairs_.insert(make_pair(block1, block2));
325 }
326 }
327 }
328 }
329 }
330
331 VLOG(1) << "Block pair stats: " << block_pairs_.size();
332}

334// Initialize the SchurEliminator.
335void VisibilityBasedPreconditioner::InitEliminator(
336 const CompressedRowBlockStructure& bs) {
337 LinearSolver::Options eliminator_options;
338 eliminator_options.elimination_groups = options_.elimination_groups;
339 eliminator_options.num_threads = options_.num_threads;
340 eliminator_options.e_block_size = options_.e_block_size;
341 eliminator_options.f_block_size = options_.f_block_size;
342 eliminator_options.row_block_size = options_.row_block_size;
343 eliminator_.reset(SchurEliminatorBase::Create(eliminator_options));
344 const bool kFullRankETE = true;
345 eliminator_->Init(
346 eliminator_options.elimination_groups[0], kFullRankETE, &bs);
347}

349// Update the values of the preconditioner matrix and factorize it.
350bool VisibilityBasedPreconditioner::UpdateImpl(const BlockSparseMatrix& A,
351 const double* D) {
352 const time_t start_time = time(NULL);
353 const int num_rows = m_->num_rows();
354 CHECK_GT(num_rows, 0);
355
356 // We need a dummy rhs vector and a dummy b vector since the Schur
357 // eliminator combines the computation of the reduced camera matrix
358 // with the computation of the right hand side of that linear
359 // system.
360 //
361 // TODO(sameeragarwal): Perhaps its worth refactoring the
362 // SchurEliminator::Eliminate function to allow NULL for the rhs. As
363 // of now it does not seem to be worth the effort.
364 Vector rhs = Vector::Zero(m_->num_rows());
365 Vector b = Vector::Zero(A.num_rows());
366
367 // Compute a subset of the entries of the Schur complement.
368 eliminator_->Eliminate(&A, b.data(), D, m_.get(), rhs.data());
369
370 // Try factorizing the matrix. For CLUSTER_JACOBI, this should
371 // always succeed modulo some numerical/conditioning problems. For
372 // CLUSTER_TRIDIAGONAL, in general the preconditioner matrix as
373 // constructed is not positive definite. However, we will go ahead
374 // and try factorizing it. If it works, great, otherwise we scale
375 // all the cells in the preconditioner corresponding to the edges in
376 // the degree-2 forest and that guarantees positive
377 // definiteness. The proof of this fact can be found in Lemma 1 in
378 // "Visibility Based Preconditioning for Bundle Adjustment".
379 //
380 // Doing the factorization like this saves us matrix mass when
381 // scaling is not needed, which is quite often in our experience.
382 LinearSolverTerminationType status = Factorize();
383
384 if (status == LINEAR_SOLVER_FATAL_ERROR) {
385 return false;
386 }
387
388 // The scaling only affects the tri-diagonal case, since
389 // ScaleOffDiagonalBlocks only pays attenion to the cells that
390 // belong to the edges of the degree-2 forest. In the CLUSTER_JACOBI
391 // case, the preconditioner is guaranteed to be positive
392 // semidefinite.
393 if (status == LINEAR_SOLVER_FAILURE && options_.type == CLUSTER_TRIDIAGONAL) {
394 VLOG(1) << "Unscaled factorization failed. Retrying with off-diagonal "
395 << "scaling";
396 ScaleOffDiagonalCells();
397 status = Factorize();
398 }
399
400 VLOG(2) << "Compute time: " << time(NULL) - start_time;
401 return (status == LINEAR_SOLVER_SUCCESS);
402}
403
404// Consider the preconditioner matrix as meta-block matrix, whose
405// blocks correspond to the clusters. Then cluster pairs corresponding
406// to edges in the degree-2 forest are off diagonal entries of this
407// matrix. Scaling these off-diagonal entries by 1/2 forces this
408// matrix to be positive definite.
409void VisibilityBasedPreconditioner::ScaleOffDiagonalCells() {
410 for (set<pair<int, int> >::const_iterator it = block_pairs_.begin();
411 it != block_pairs_.end();
412 ++it) {
413 const int block1 = it->first;
414 const int block2 = it->second;
415 if (!IsBlockPairOffDiagonal(block1, block2)) {
416 continue;
417 }
418
419 int r, c, row_stride, col_stride;
420 CellInfo* cell_info = m_->GetCell(block1, block2,
421 &r, &c,
422 &row_stride, &col_stride);
423 CHECK(cell_info != NULL)
424 << "Cell missing for block pair (" << block1 << "," << block2 << ")"
425 << " cluster pair (" << cluster_membership_[block1]
426 << " " << cluster_membership_[block2] << ")";
427
428 // Ah the magic of tri-diagonal matrices and diagonal
429 // dominance. See Lemma 1 in "Visibility Based Preconditioning
430 // For Bundle Adjustment".
431 MatrixRef m(cell_info->values, row_stride, col_stride);
432 m.block(r, c, block_size_[block1], block_size_[block2]) *= 0.5;
433 }
434}

436// Compute the sparse Cholesky factorization of the preconditioner
437// matrix.
438LinearSolverTerminationType VisibilityBasedPreconditioner::Factorize() {
439 // Extract the TripletSparseMatrix that is used for actually storing
440 // S and convert it into a cholmod_sparse object.
441 cholmod_sparse* lhs = ss_.CreateSparseMatrix(
442 down_cast<BlockRandomAccessSparseMatrix*>(
443 m_.get())->mutable_matrix());
444
445 // The matrix is symmetric, and the upper triangular part of the
446 // matrix contains the values.
447 lhs->stype = 1;
448
449 // TODO(sameeragarwal): Refactor to pipe this up and out.
450 std::string status;
451
452 // Symbolic factorization is computed if we don't already have one handy.
453 if (factor_ == NULL) {
454 factor_ = ss_.BlockAnalyzeCholesky(lhs, block_size_, block_size_, &status);
455 }
456
457 const LinearSolverTerminationType termination_type =
458 (factor_ != NULL)
459 ? ss_.Cholesky(lhs, factor_, &status)
460 : LINEAR_SOLVER_FATAL_ERROR;
461
462 ss_.Free(lhs);
463 return termination_type;
464}

466void VisibilityBasedPreconditioner::RightMultiply(const double* x,
467 double* y) const {
468 CHECK_NOTNULL(x);
469 CHECK_NOTNULL(y);
470 SuiteSparse* ss = const_cast<SuiteSparse*>(&ss_);
471
472 const int num_rows = m_->num_rows();
473 memcpy(CHECK_NOTNULL(tmp_rhs_)->x, x, m_->num_rows() * sizeof(*x));
474 // TODO(sameeragarwal): Better error handling.
475 std::string status;
476 cholmod_dense* solution =
477 CHECK_NOTNULL(ss->Solve(factor_, tmp_rhs_, &status));
478 memcpy(y, solution->x, sizeof(*y) * num_rows);
479 ss->Free(solution);
480}

486// Classify camera/f_block pairs as in and out of the preconditioner,
487// based on whether the cluster pair that they belong to is in the
488// preconditioner or not.
489bool VisibilityBasedPreconditioner::IsBlockPairInPreconditioner(
490 const int block1,
491 const int block2) const {
492 int cluster1 = cluster_membership_[block1];
493 int cluster2 = cluster_membership_[block2];
494 if (cluster1 > cluster2) {
495 swap(cluster1, cluster2);
496 }
497 return (cluster_pairs_.count(make_pair(cluster1, cluster2)) > 0);
498}
499
500bool VisibilityBasedPreconditioner::IsBlockPairOffDiagonal(
501 const int block1,
502 const int block2) const {
503 return (cluster_membership_[block1] != cluster_membership_[block2]);
504}

506// Convert a graph into a list of edges that includes self edges for
507// each vertex.
508void VisibilityBasedPreconditioner::ForestToClusterPairs(
509 const WeightedGraph<int>& forest,
510 HashSet<pair<int, int> >* cluster_pairs) const {
511 CHECK_NOTNULL(cluster_pairs)->clear();
512 const HashSet<int>& vertices = forest.vertices();
513 CHECK_EQ(vertices.size(), num_clusters_);
514
515 // Add all the cluster pairs corresponding to the edges in the
516 // forest.
517 for (HashSet<int>::const_iterator it1 = vertices.begin();
518 it1 != vertices.end();
519 ++it1) {
520 const int cluster1 = *it1;
521 cluster_pairs->insert(make_pair(cluster1, cluster1));
522 const HashSet<int>& neighbors = forest.Neighbors(cluster1);
523 for (HashSet<int>::const_iterator it2 = neighbors.begin();
524 it2 != neighbors.end();
525 ++it2) {
526 const int cluster2 = *it2;
527 if (cluster1 < cluster2) {
528 cluster_pairs->insert(make_pair(cluster1, cluster2));
529 }
530 }
531 }
532}

534// The visibilty set of a cluster is the union of the visibilty sets
535// of all its cameras. In other words, the set of points visible to
536// any camera in the cluster.
537void VisibilityBasedPreconditioner::ComputeClusterVisibility(
538 const vector<set<int> >& visibility,
539 vector<set<int> >* cluster_visibility) const {
540 CHECK_NOTNULL(cluster_visibility)->resize(0);
541 cluster_visibility->resize(num_clusters_);
542 for (int i = 0; i < num_blocks_; ++i) {
543 const int cluster_id = cluster_membership_[i];
544 (*cluster_visibility)[cluster_id].insert(visibility[i].begin(),
545 visibility[i].end());
546 }
547}

549// Construct a graph whose vertices are the clusters, and the edge
550// weights are the number of 3D points visible to cameras in both the
551// vertices.
552WeightedGraph<int>* VisibilityBasedPreconditioner::CreateClusterGraph(
553 const vector<set<int> >& cluster_visibility) const {
554 WeightedGraph<int>* cluster_graph = new WeightedGraph<int>;
555
556 for (int i = 0; i < num_clusters_; ++i) {
557 cluster_graph->AddVertex(i);
558 }
559
560 for (int i = 0; i < num_clusters_; ++i) {
561 const set<int>& cluster_i = cluster_visibility[i];
562 for (int j = i+1; j < num_clusters_; ++j) {
563 vector<int> intersection;
564 const set<int>& cluster_j = cluster_visibility[j];
565 set_intersection(cluster_i.begin(), cluster_i.end(),
566 cluster_j.begin(), cluster_j.end(),
567 back_inserter(intersection));
568
569 if (intersection.size() > 0) {
570 // Clusters interact strongly when they share a large number
571 // of 3D points. The degree-2 maximum spanning forest
572 // alorithm, iterates on the edges in decreasing order of
573 // their weight, which is the number of points shared by the
574 // two cameras that it connects.
575 cluster_graph->AddEdge(i, j, intersection.size());
576 }
577 }
578 }
579 return cluster_graph;
580}

582// Canonical views clustering returns a HashMap from vertices to
583// cluster ids. Convert this into a flat array for quick lookup. It is
584// possible that some of the vertices may not be associated with any
585// cluster. In that case, randomly assign them to one of the clusters.
586//
587// The cluster ids can be non-contiguous integers. So as we flatten
588// the membership_map, we also map the cluster ids to a contiguous set
589// of integers so that the cluster ids are in [0, num_clusters_).
590void VisibilityBasedPreconditioner::FlattenMembershipMap(
591 const HashMap<int, int>& membership_map,
592 vector<int>* membership_vector) const {
593 CHECK_NOTNULL(membership_vector)->resize(0);
594 membership_vector->resize(num_blocks_, -1);
595
596 HashMap<int, int> cluster_id_to_index;
597 // Iterate over the cluster membership map and update the
598 // cluster_membership_ vector assigning arbitrary cluster ids to
599 // the few cameras that have not been clustered.
600 for (HashMap<int, int>::const_iterator it = membership_map.begin();
601 it != membership_map.end();
602 ++it) {
603 const int camera_id = it->first;
604 int cluster_id = it->second;
605
606 // If the view was not clustered, randomly assign it to one of the
607 // clusters. This preserves the mathematical correctness of the
608 // preconditioner. If there are too many views which are not
609 // clustered, it may lead to some quality degradation though.
610 //
611 // TODO(sameeragarwal): Check if a large number of views have not
612 // been clustered and deal with it?
613 if (cluster_id == -1) {
614 cluster_id = camera_id % num_clusters_;
615 }
616
617 const int index = FindWithDefault(cluster_id_to_index,
618 cluster_id,
619 cluster_id_to_index.size());
620
621 if (index == cluster_id_to_index.size()) {
622 cluster_id_to_index[cluster_id] = index;
623 }
624
625 CHECK_LT(index, num_clusters_);
626 membership_vector->at(camera_id) = index;
627 }
