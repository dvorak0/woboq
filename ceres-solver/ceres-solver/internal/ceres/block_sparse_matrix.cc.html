49BlockSparseMatrix::BlockSparseMatrix(
50 CompressedRowBlockStructure* block_structure)
51 : num_rows_(0),
52 num_cols_(0),
53 num_nonzeros_(0),
54 values_(NULL),
55 block_structure_(block_structure) {
56 CHECK_NOTNULL(block_structure_.get());
57
58 // Count the number of columns in the matrix.
59 for (int i = 0; i < block_structure_->cols.size(); ++i) {
60 num_cols_ += block_structure_->cols[i].size;
61 }
62
63 // Count the number of non-zero entries and the number of rows in
64 // the matrix.
65 for (int i = 0; i < block_structure_->rows.size(); ++i) {
66 int row_block_size = block_structure_->rows[i].block.size;
67 num_rows_ += row_block_size;
68
69 const vector<Cell>& cells = block_structure_->rows[i].cells;
70 for (int j = 0; j < cells.size(); ++j) {
71 int col_block_id = cells[j].block_id;
72 int col_block_size = block_structure_->cols[col_block_id].size;
73 num_nonzeros_ += col_block_size * row_block_size;
74 }
75 }
76
77 CHECK_GE(num_rows_, 0);
78 CHECK_GE(num_cols_, 0);
79 CHECK_GE(num_nonzeros_, 0);
80 VLOG(2) << "Allocating values array with "
81 << num_nonzeros_ * sizeof(double) << " bytes."; // NOLINT
82 values_.reset(new double[num_nonzeros_]);
83 CHECK_NOTNULL(values_.get());
84}
85
86void BlockSparseMatrix::SetZero() {
87 std::fill(values_.get(), values_.get() + num_nonzeros_, 0.0);
88}

90void BlockSparseMatrix::RightMultiply(const double* x, double* y) const {
91 CHECK_NOTNULL(x);
92 CHECK_NOTNULL(y);
93
94 for (int i = 0; i < block_structure_->rows.size(); ++i) {
95 int row_block_pos = block_structure_->rows[i].block.position;
96 int row_block_size = block_structure_->rows[i].block.size;
97 const vector<Cell>& cells = block_structure_->rows[i].cells;
98 for (int j = 0; j < cells.size(); ++j) {
99 int col_block_id = cells[j].block_id;
100 int col_block_size = block_structure_->cols[col_block_id].size;
101 int col_block_pos = block_structure_->cols[col_block_id].position;
102 MatrixVectorMultiply<Eigen::Dynamic, Eigen::Dynamic, 1>(
103 values_.get() + cells[j].position, row_block_size, col_block_size,
104 x + col_block_pos,
105 y + row_block_pos);
106 }
107 }
108}

110void BlockSparseMatrix::LeftMultiply(const double* x, double* y) const {
111 CHECK_NOTNULL(x);
112 CHECK_NOTNULL(y);
113
114 for (int i = 0; i < block_structure_->rows.size(); ++i) {
115 int row_block_pos = block_structure_->rows[i].block.position;
116 int row_block_size = block_structure_->rows[i].block.size;
117 const vector<Cell>& cells = block_structure_->rows[i].cells;
118 for (int j = 0; j < cells.size(); ++j) {
119 int col_block_id = cells[j].block_id;
120 int col_block_size = block_structure_->cols[col_block_id].size;
121 int col_block_pos = block_structure_->cols[col_block_id].position;
122 MatrixTransposeVectorMultiply<Eigen::Dynamic, Eigen::Dynamic, 1>(
123 values_.get() + cells[j].position, row_block_size, col_block_size,
124 x + row_block_pos,
125 y + col_block_pos);
126 }
127 }
128}

130void BlockSparseMatrix::SquaredColumnNorm(double* x) const {
131 CHECK_NOTNULL(x);
132 VectorRef(x, num_cols_).setZero();
133 for (int i = 0; i < block_structure_->rows.size(); ++i) {
134 int row_block_size = block_structure_->rows[i].block.size;
135 const vector<Cell>& cells = block_structure_->rows[i].cells;
136 for (int j = 0; j < cells.size(); ++j) {
137 int col_block_id = cells[j].block_id;
138 int col_block_size = block_structure_->cols[col_block_id].size;
139 int col_block_pos = block_structure_->cols[col_block_id].position;
140 const MatrixRef m(values_.get() + cells[j].position,
141 row_block_size, col_block_size);
142 VectorRef(x + col_block_pos, col_block_size) += m.colwise().squaredNorm();
143 }
144 }
145}

147void BlockSparseMatrix::ScaleColumns(const double* scale) {
148 CHECK_NOTNULL(scale);
149
150 for (int i = 0; i < block_structure_->rows.size(); ++i) {
151 int row_block_size = block_structure_->rows[i].block.size;
152 const vector<Cell>& cells = block_structure_->rows[i].cells;
153 for (int j = 0; j < cells.size(); ++j) {
154 int col_block_id = cells[j].block_id;
155 int col_block_size = block_structure_->cols[col_block_id].size;
156 int col_block_pos = block_structure_->cols[col_block_id].position;
157 MatrixRef m(values_.get() + cells[j].position,
158 row_block_size, col_block_size);
159 m *= ConstVectorRef(scale + col_block_pos, col_block_size).asDiagonal();
160 }
161 }

164void BlockSparseMatrix::ToDenseMatrix(Matrix* dense_matrix) const {
165 CHECK_NOTNULL(dense_matrix);
166
167 dense_matrix->resize(num_rows_, num_cols_);
168 dense_matrix->setZero();
169 Matrix& m = *dense_matrix;
170
171 for (int i = 0; i < block_structure_->rows.size(); ++i) {
172 int row_block_pos = block_structure_->rows[i].block.position;
173 int row_block_size = block_structure_->rows[i].block.size;
174 const vector<Cell>& cells = block_structure_->rows[i].cells;
175 for (int j = 0; j < cells.size(); ++j) {
176 int col_block_id = cells[j].block_id;
177 int col_block_size = block_structure_->cols[col_block_id].size;
178 int col_block_pos = block_structure_->cols[col_block_id].position;
179 int jac_pos = cells[j].position;
180 m.block(row_block_pos, col_block_pos, row_block_size, col_block_size)
181 += MatrixRef(values_.get() + jac_pos, row_block_size, col_block_size);
182 }
183 }

186void BlockSparseMatrix::ToTripletSparseMatrix(
187 TripletSparseMatrix* matrix) const {
188 CHECK_NOTNULL(matrix);
189
190 matrix->Reserve(num_nonzeros_);
191 matrix->Resize(num_rows_, num_cols_);
192 matrix->SetZero();
193
194 for (int i = 0; i < block_structure_->rows.size(); ++i) {
195 int row_block_pos = block_structure_->rows[i].block.position;
196 int row_block_size = block_structure_->rows[i].block.size;
197 const vector<Cell>& cells = block_structure_->rows[i].cells;
198 for (int j = 0; j < cells.size(); ++j) {
199 int col_block_id = cells[j].block_id;
200 int col_block_size = block_structure_->cols[col_block_id].size;
201 int col_block_pos = block_structure_->cols[col_block_id].position;
202 int jac_pos = cells[j].position;
203 for (int r = 0; r < row_block_size; ++r) {
204 for (int c = 0; c < col_block_size; ++c, ++jac_pos) {
205 matrix->mutable_rows()[jac_pos] = row_block_pos + r;
206 matrix->mutable_cols()[jac_pos] = col_block_pos + c;
207 matrix->mutable_values()[jac_pos] = values_[jac_pos];
208 }
209 }
210 }
211 }
212 matrix->set_num_nonzeros(num_nonzeros_);
213}

222void BlockSparseMatrix::ToTextFile(FILE* file) const {
223 CHECK_NOTNULL(file);
224 for (int i = 0; i < block_structure_->rows.size(); ++i) {
225 const int row_block_pos = block_structure_->rows[i].block.position;
226 const int row_block_size = block_structure_->rows[i].block.size;
227 const vector<Cell>& cells = block_structure_->rows[i].cells;
228 for (int j = 0; j < cells.size(); ++j) {
229 const int col_block_id = cells[j].block_id;
230 const int col_block_size = block_structure_->cols[col_block_id].size;
231 const int col_block_pos = block_structure_->cols[col_block_id].position;
232 int jac_pos = cells[j].position;
233 for (int r = 0; r < row_block_size; ++r) {
234 for (int c = 0; c < col_block_size; ++c) {
235 fprintf(file, "% 10d % 10d %17f\n",
236 row_block_pos + r,
237 col_block_pos + c,
238 values_[jac_pos++]);
239 }
240 }
241 }
242 }
