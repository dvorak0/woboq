35namespace internal {
36
37// Fixed sized struct for storing an evaluation.
38struct ExpectedEvaluation {
39 int num_rows;
40 int num_cols;
41 double cost;
42 const double residuals[50];
43 const double gradient[50];
44 const double jacobian[200];
45};
46
47// Compare two evaluations.
48void CompareEvaluations(int expected_num_rows,
49 int expected_num_cols,
50 double expected_cost,
51 const double* expected_residuals,
52 const double* expected_gradient,
53 const double* expected_jacobian,
54 const double actual_cost,
55 const double* actual_residuals,
56 const double* actual_gradient,
57 const double* actual_jacobian);

59} // namespace internal
60} // namespace ceres
