40Minimizer* Minimizer::Create(MinimizerType minimizer_type) {
41 if (minimizer_type == TRUST_REGION) {
42 return new TrustRegionMinimizer;
43 }
44
45 if (minimizer_type == LINE_SEARCH) {
46 return new LineSearchMinimizer;
47 }
48
49 LOG(FATAL) << "Unknown minimizer_type: " << minimizer_type;
50 return NULL;
51}

56bool Minimizer::RunCallbacks(const Minimizer::Options& options,
57 const IterationSummary& iteration_summary,
58 Solver::Summary* summary) {
59 const bool is_not_silent = !options.is_silent;
60 CallbackReturnType status = SOLVER_CONTINUE;
61 int i = 0;
62 while (status == SOLVER_CONTINUE && i < options.callbacks.size()) {
63 status = (*options.callbacks[i])(iteration_summary);
64 ++i;
65 }
66 switch (status) {
67 case SOLVER_CONTINUE:
68 return true;
69 case SOLVER_TERMINATE_SUCCESSFULLY:
70 summary->termination_type = USER_SUCCESS;
71 summary->message =
72 "User callback returned SOLVER_TERMINATE_SUCCESSFULLY.";
73 VLOG_IF(1, is_not_silent) << "Terminating: " << summary->message;
74 return false;
75 case SOLVER_ABORT:
76 summary->termination_type = USER_FAILURE;
77 summary->message = "User callback returned SOLVER_ABORT.";
78 VLOG_IF(1, is_not_silent) << "Terminating: " << summary->message;
79 return false;
80 default:
81 LOG(FATAL) << "Unknown type of user callback status";
82 }
83 return false;
84}
