41#define CASESTR(x) case x: return #x
42#define STRENUM(x) if (value == #x) { *type = x; return true;}

44static void UpperCase(string* input) {
45 std::transform(input->begin(), input->end(), input->begin(), ::toupper);
46}
47
48const char* LinearSolverTypeToString(LinearSolverType type) {
49 switch (type) {
50 CASESTR(DENSE_NORMAL_CHOLESKY);
51 CASESTR(DENSE_QR);
52 CASESTR(SPARSE_NORMAL_CHOLESKY);
53 CASESTR(DENSE_SCHUR);
54 CASESTR(SPARSE_SCHUR);
55 CASESTR(ITERATIVE_SCHUR);
56 CASESTR(CGNR);
57 default:
58 return "UNKNOWN";
59 }
60}

62bool StringToLinearSolverType(string value, LinearSolverType* type) {

67 STRENUM(DENSE_SCHUR);
68 STRENUM(SPARSE_SCHUR);

340bool IsSchurType(LinearSolverType type) {
341 return ((type == SPARSE_SCHUR) ||
342 (type == DENSE_SCHUR) ||
343 (type == ITERATIVE_SCHUR));
344}

346bool IsSparseLinearAlgebraLibraryTypeAvailable(
347 SparseLinearAlgebraLibraryType type) {
348 if (type == SUITE_SPARSE) {
349#ifdef CERES_NO_SUITESPARSE
350 return false;
351#else
352 return true;
353#endif
354 }
355
356 if (type == CX_SPARSE) {
357#ifdef CERES_NO_CXSPARSE
358 return false;
359#else
360 return true;
361#endif
362 }
363
364 if (type == EIGEN_SPARSE) {
365#ifdef CERES_USE_EIGEN_SPARSE
366 return true;
367#else
368 return false;
369#endif
370 }
371
372 LOG(WARNING) << "Unknown sparse linear algebra library " << type;
373 return false;
374}
375
376bool IsDenseLinearAlgebraLibraryTypeAvailable(
377 DenseLinearAlgebraLibraryType type) {
378 if (type == EIGEN) {
379 return true;
380 }
381 if (type == LAPACK) {
382#ifdef CERES_NO_LAPACK
383 return false;
384#else
385 return true;
386#endif
387 }
388
389 LOG(WARNING) << "Unknown dense linear algebra library " << type;
390 return false;
391}
