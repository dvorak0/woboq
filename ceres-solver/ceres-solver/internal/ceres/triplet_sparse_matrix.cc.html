45TripletSparseMatrix::TripletSparseMatrix()

48 max_num_nonzeros_(0),
49 num_nonzeros_(0),
50 rows_(NULL),
51 cols_(NULL),
52 values_(NULL) {}

56TripletSparseMatrix::TripletSparseMatrix(int num_rows,
57 int num_cols,
58 int max_num_nonzeros)
59 : num_rows_(num_rows),
60 num_cols_(num_cols),
61 max_num_nonzeros_(max_num_nonzeros),
62 num_nonzeros_(0),
63 rows_(NULL),
64 cols_(NULL),
65 values_(NULL) {
66 // All the sizes should at least be zero
67 CHECK_GE(num_rows, 0);
68 CHECK_GE(num_cols, 0);
69 CHECK_GE(max_num_nonzeros, 0);
70 AllocateMemory();
71}

73TripletSparseMatrix::TripletSparseMatrix(const int num_rows,
74 const int num_cols,
75 const std::vector<int>& rows,
76 const std::vector<int>& cols,
77 const std::vector<double>& values)
78 : num_rows_(num_rows),
79 num_cols_(num_cols),
80 max_num_nonzeros_(values.size()),
81 num_nonzeros_(values.size()),
82 rows_(NULL),
83 cols_(NULL),
84 values_(NULL) {
85 // All the sizes should at least be zero
86 CHECK_GE(num_rows, 0);
87 CHECK_GE(num_cols, 0);
88 CHECK_EQ(rows.size(), cols.size());
89 CHECK_EQ(rows.size(), values.size());
90 AllocateMemory();
91 std::copy(rows.begin(), rows.end(), rows_.get());
92 std::copy(cols.begin(), cols.end(), cols_.get());
93 std::copy(values.begin(), values.end(), values_.get());
94}

109TripletSparseMatrix& TripletSparseMatrix::operator=(
110 const TripletSparseMatrix& rhs) {
111 num_rows_ = rhs.num_rows_;
112 num_cols_ = rhs.num_cols_;
113 num_nonzeros_ = rhs.num_nonzeros_;
114 max_num_nonzeros_ = rhs.max_num_nonzeros_;
115 AllocateMemory();
116 CopyData(rhs);
117 return *this;
118}

120bool TripletSparseMatrix::AllTripletsWithinBounds() const {
121 for (int i = 0; i < num_nonzeros_; ++i) {
122 if ((rows_[i] < 0) || (rows_[i] >= num_rows_) ||
123 (cols_[i] < 0) || (cols_[i] >= num_cols_))
124 return false;
125 }
126 return true;
127}

129void TripletSparseMatrix::Reserve(int new_max_num_nonzeros) {
130 CHECK_LE(num_nonzeros_, new_max_num_nonzeros)
131 << "Reallocation will cause data loss";
132
133 // Nothing to do if we have enough space already.
134 if (new_max_num_nonzeros <= max_num_nonzeros_)
135 return;
136
137 int* new_rows = new int[new_max_num_nonzeros];
138 int* new_cols = new int[new_max_num_nonzeros];
139 double* new_values = new double[new_max_num_nonzeros];
140
141 for (int i = 0; i < num_nonzeros_; ++i) {
142 new_rows[i] = rows_[i];
143 new_cols[i] = cols_[i];
144 new_values[i] = values_[i];
145 }
146
147 rows_.reset(new_rows);
148 cols_.reset(new_cols);
149 values_.reset(new_values);
150
151 max_num_nonzeros_ = new_max_num_nonzeros;
152}
153
154void TripletSparseMatrix::SetZero() {
155 std::fill(values_.get(), values_.get() + max_num_nonzeros_, 0.0);
156 num_nonzeros_ = 0;
157}
158
159void TripletSparseMatrix::set_num_nonzeros(int num_nonzeros) {
160 CHECK_GE(num_nonzeros, 0);
161 CHECK_LE(num_nonzeros, max_num_nonzeros_);
162 num_nonzeros_ = num_nonzeros;
163}

165void TripletSparseMatrix::AllocateMemory() {
166 rows_.reset(new int[max_num_nonzeros_]);
167 cols_.reset(new int[max_num_nonzeros_]);
168 values_.reset(new double[max_num_nonzeros_]);
169}

171void TripletSparseMatrix::CopyData(const TripletSparseMatrix& orig) {
172 for (int i = 0; i < num_nonzeros_; ++i) {
173 rows_[i] = orig.rows_[i];
174 cols_[i] = orig.cols_[i];
175 values_[i] = orig.values_[i];
176 }
177}

179void TripletSparseMatrix::RightMultiply(const double* x, double* y) const {
180 for (int i = 0; i < num_nonzeros_; ++i) {
181 y[rows_[i]] += values_[i]*x[cols_[i]];
182 }
183}
184
185void TripletSparseMatrix::LeftMultiply(const double* x, double* y) const {
186 for (int i = 0; i < num_nonzeros_; ++i) {
187 y[cols_[i]] += values_[i]*x[rows_[i]];
188 }
189}

191void TripletSparseMatrix::SquaredColumnNorm(double* x) const {
192 CHECK_NOTNULL(x);
193 VectorRef(x, num_cols_).setZero();
194 for (int i = 0; i < num_nonzeros_; ++i) {
195 x[cols_[i]] += values_[i] * values_[i];
196 }
197}

199void TripletSparseMatrix::ScaleColumns(const double* scale) {
200 CHECK_NOTNULL(scale);
201 for (int i = 0; i < num_nonzeros_; ++i) {
202 values_[i] = values_[i] * scale[cols_[i]];
203 }
204}

206void TripletSparseMatrix::ToDenseMatrix(Matrix* dense_matrix) const {
207 dense_matrix->resize(num_rows_, num_cols_);
208 dense_matrix->setZero();
209 Matrix& m = *dense_matrix;
210 for (int i = 0; i < num_nonzeros_; ++i) {
211 m(rows_[i], cols_[i]) += values_[i];
212 }
213}

215void TripletSparseMatrix::AppendRows(const TripletSparseMatrix& B) {
216 CHECK_EQ(B.num_cols(), num_cols_);
217 Reserve(num_nonzeros_ + B.num_nonzeros_);
218 for (int i = 0; i < B.num_nonzeros_; ++i) {
219 rows_.get()[num_nonzeros_] = B.rows()[i] + num_rows_;
220 cols_.get()[num_nonzeros_] = B.cols()[i];
221 values_.get()[num_nonzeros_++] = B.values()[i];
222 }
223 num_rows_ = num_rows_ + B.num_rows();
224}
225
226void TripletSparseMatrix::AppendCols(const TripletSparseMatrix& B) {
227 CHECK_EQ(B.num_rows(), num_rows_);
228 Reserve(num_nonzeros_ + B.num_nonzeros_);
229 for (int i = 0; i < B.num_nonzeros_; ++i, ++num_nonzeros_) {
230 rows_.get()[num_nonzeros_] = B.rows()[i];
231 cols_.get()[num_nonzeros_] = B.cols()[i] + num_cols_;
232 values_.get()[num_nonzeros_] = B.values()[i];
233 }
234 num_cols_ = num_cols_ + B.num_cols();
235}

238void TripletSparseMatrix::Resize(int new_num_rows, int new_num_cols) {
239 if ((new_num_rows >= num_rows_) && (new_num_cols >= num_cols_)) {
240 num_rows_ = new_num_rows;
241 num_cols_ = new_num_cols;
242 return;
243 }
244
245 num_rows_ = new_num_rows;
246 num_cols_ = new_num_cols;
247
248 int* r_ptr = rows_.get();
249 int* c_ptr = cols_.get();
250 double* v_ptr = values_.get();
251
252 int dropped_terms = 0;
253 for (int i = 0; i < num_nonzeros_; ++i) {
254 if ((r_ptr[i] < num_rows_) && (c_ptr[i] < num_cols_)) {
255 if (dropped_terms) {
256 r_ptr[i-dropped_terms] = r_ptr[i];
257 c_ptr[i-dropped_terms] = c_ptr[i];
258 v_ptr[i-dropped_terms] = v_ptr[i];
259 }
260 } else {
261 ++dropped_terms;
262 }
263 }
264 num_nonzeros_ -= dropped_terms;
265}

267TripletSparseMatrix* TripletSparseMatrix::CreateSparseDiagonalMatrix(
268 const double* values, int num_rows) {
269 TripletSparseMatrix* m =
270 new TripletSparseMatrix(num_rows, num_rows, num_rows);
271 for (int i = 0; i < num_rows; ++i) {
272 m->mutable_rows()[i] = i;
273 m->mutable_cols()[i] = i;
274 m->mutable_values()[i] = values[i];
275 }
276 m->set_num_nonzeros(num_rows);
277 return m;
278}

281 CHECK_NOTNULL(file);
282 for (int i = 0; i < num_nonzeros_; ++i) {

287TripletSparseMatrix* TripletSparseMatrix::CreateRandomMatrix(
288 const TripletSparseMatrix::RandomMatrixOptions& options) {
289 CHECK_GT(options.num_rows, 0);
290 CHECK_GT(options.num_cols, 0);
291 CHECK_GT(options.density, 0.0);
292 CHECK_LE(options.density, 1.0);
293
294 std::vector<int> rows;
295 std::vector<int> cols;
296 std::vector<double> values;
297 while (rows.empty()) {
298 rows.clear();
299 cols.clear();
300 values.clear();
301 for (int r = 0; r < options.num_rows; ++r) {
302 for (int c = 0; c < options.num_cols; ++c) {
303 if (RandDouble() <= options.density) {
304 rows.push_back(r);
305 cols.push_back(c);
306 values.push_back(RandNormal());
307 }
308 }
309 }
310 }
311
312 return new TripletSparseMatrix(
313 options.num_rows, options.num_cols, rows, cols, values);
314}
