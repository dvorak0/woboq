56// Different sparse linear algebra libraries prefer different storage
57// orders for the input matrix. This trait class helps choose the
58// ordering based on the sparse linear algebra backend being used.
59//
60// The storage order is lower-triangular by default. It is only
61// SuiteSparse which prefers an upper triangular matrix. Saves a whole
62// matrix copy in the process.
63//
64// Note that this is the storage order for a compressed row sparse
65// matrix. All the sparse linear algebra libraries take compressed
66// column sparse matrices as input. We map these matrices to into
67// compressed column sparse matrices before calling them and in the
68// process, transpose them.
69//
70// TODO(sameeragarwal): This does not account for post ordering, where
71// the optimal storage order maybe different. Either get rid of post
72// ordering support entirely, or investigate making this trait class
73// richer.
74
75CompressedRowSparseMatrix::StorageType StorageTypeForSparseLinearAlgebraLibrary(
76 SparseLinearAlgebraLibraryType sparse_linear_algebra_library_type) {
77 if (sparse_linear_algebra_library_type == SUITE_SPARSE) {
78 return CompressedRowSparseMatrix::UPPER_TRIANGULAR;
79 }
80 return CompressedRowSparseMatrix::LOWER_TRIANGULAR;
81}

85#ifdef CERES_USE_EIGEN_SPARSE
86// A templated factorized and solve function, which allows us to use
87// the same code independent of whether a AMD or a Natural ordering is
88// used.
89template <typename SimplicialCholeskySolver, typename SparseMatrixType>
90LinearSolver::Summary SimplicialLDLTSolve(const SparseMatrixType& lhs,
91 const bool do_symbolic_analysis,
92 SimplicialCholeskySolver* solver,
93 double* rhs_and_solution,
94 EventLogger* event_logger) {
95 LinearSolver::Summary summary;
96 summary.num_iterations = 1;
97 summary.termination_type = LINEAR_SOLVER_SUCCESS;
98 summary.message = "Success.";
99
100 if (do_symbolic_analysis) {
101 solver->analyzePattern(lhs);
102 if (VLOG_IS_ON(2)) {
103 std::stringstream ss;
104 solver->dumpMemory(ss);
105 VLOG(2) << "Symbolic Analysis\n" << ss.str();
106 }
107 event_logger->AddEvent("Analyze");
108 if (solver->info() != Eigen::Success) {
109 summary.termination_type = LINEAR_SOLVER_FATAL_ERROR;
110 summary.message = "Eigen failure. Unable to find symbolic factorization.";
111 return summary;
112 }
113 }
114
115 solver->factorize(lhs);
116 event_logger->AddEvent("Factorize");
117 if (solver->info() != Eigen::Success) {
118 summary.termination_type = LINEAR_SOLVER_FAILURE;
119 summary.message = "Eigen failure. Unable to find numeric factorization.";
120 return summary;
121 }
122
123 const Vector rhs = VectorRef(rhs_and_solution, lhs.cols());
124
125 VectorRef(rhs_and_solution, lhs.cols()) = solver->solve(rhs);
126 event_logger->AddEvent("Solve");
127 if (solver->info() != Eigen::Success) {
128 summary.termination_type = LINEAR_SOLVER_FAILURE;
129 summary.message = "Eigen failure. Unable to do triangular solve.";
130 return summary;
131 }
132
133 return summary;
134}

136#endif // CERES_USE_EIGEN_SPARSE

160LinearSolver::Summary SparseNormalCholeskySolver::SolveImpl(
161 CompressedRowSparseMatrix* A,
162 const double* b,
163 const LinearSolver::PerSolveOptions& per_solve_options,
164 double* x) {
165 const int num_cols = A->num_cols();
166 VectorRef(x, num_cols).setZero();
167 A->LeftMultiply(b, x);
168
169 if (per_solve_options.D != NULL) {
170 // Temporarily append a diagonal block to the A matrix, but undo
171 // it before returning the matrix to the user.
172 scoped_ptr<CompressedRowSparseMatrix> regularizer;
173 if (A->col_blocks().size() > 0) {
174 regularizer.reset(CompressedRowSparseMatrix::CreateBlockDiagonalMatrix(
175 per_solve_options.D, A->col_blocks()));
176 } else {
177 regularizer.reset(
178 new CompressedRowSparseMatrix(per_solve_options.D, num_cols));
179 }
180 A->AppendRows(*regularizer);
181 }
182
183 if (outer_product_.get() == NULL) {
184 outer_product_.reset(
185 CompressedRowSparseMatrix::CreateOuterProductMatrixAndProgram(
186 *A,
187 StorageTypeForSparseLinearAlgebraLibrary(
188 options_.sparse_linear_algebra_library_type),
189 &pattern_));
190 }
191
192 CompressedRowSparseMatrix::ComputeOuterProduct(
193 *A, pattern_, outer_product_.get());
194
195 LinearSolver::Summary summary;
196 switch (options_.sparse_linear_algebra_library_type) {
197 case SUITE_SPARSE:
198 summary = SolveImplUsingSuiteSparse(x);
199 break;
200 case CX_SPARSE:
201 summary = SolveImplUsingCXSparse(x);
202 break;
203 case EIGEN_SPARSE:
204 summary = SolveImplUsingEigen(x);
205 break;
206 default:
207 LOG(FATAL) << "Unknown sparse linear algebra library : "
208 << options_.sparse_linear_algebra_library_type;
209 }
210
211 if (per_solve_options.D != NULL) {
212 A->DeleteRows(num_cols);
213 }
214
215 return summary;
216}

218LinearSolver::Summary SparseNormalCholeskySolver::SolveImplUsingEigen(
219 double* rhs_and_solution) {
220#ifndef CERES_USE_EIGEN_SPARSE
221
222 LinearSolver::Summary summary;
223 summary.num_iterations = 0;
224 summary.termination_type = LINEAR_SOLVER_FATAL_ERROR;
225 summary.message =
226 "SPARSE_NORMAL_CHOLESKY cannot be used with EIGEN_SPARSE "
227 "because Ceres was not built with support for "
228 "Eigen's SimplicialLDLT decomposition. "
229 "This requires enabling building with -DEIGENSPARSE=ON.";
230 return summary;
231
232#else
233
234 EventLogger event_logger("SparseNormalCholeskySolver::Eigen::Solve");
235
236 // Map outer_product_ to an upper triangular column major matrix.
237 //
238 // outer_product_ is a compressed row sparse matrix and in lower
239 // triangular form, when mapped to a compressed column sparse
240 // matrix, it becomes an upper triangular matrix.
241 Eigen::MappedSparseMatrix<double, Eigen::ColMajor> lhs(
242 outer_product_->num_rows(),
243 outer_product_->num_rows(),
244 outer_product_->num_nonzeros(),
245 outer_product_->mutable_rows(),
246 outer_product_->mutable_cols(),
247 outer_product_->mutable_values());
248
249 bool do_symbolic_analysis = false;
250
251 // If using post ordering or an old version of Eigen, we cannot
252 // depend on a preordered jacobian, so we work with a SimplicialLDLT
253 // decomposition with AMD ordering.
254 if (options_.use_postordering || !EIGEN_VERSION_AT_LEAST(3, 2, 2)) {
255 if (amd_ldlt_.get() == NULL) {
256 amd_ldlt_.reset(new SimplicialLDLTWithAMDOrdering);
257 do_symbolic_analysis = true;
258 }
259
260 return SimplicialLDLTSolve(lhs,
261 do_symbolic_analysis,
262 amd_ldlt_.get(),
263 rhs_and_solution,
264 &event_logger);
265 }
266
267#if EIGEN_VERSION_AT_LEAST(3, 2, 2)
268 // The common case
269 if (natural_ldlt_.get() == NULL) {
270 natural_ldlt_.reset(new SimplicialLDLTWithNaturalOrdering);
271 do_symbolic_analysis = true;
272 }
273
274 return SimplicialLDLTSolve(lhs,
275 do_symbolic_analysis,
276 natural_ldlt_.get(),
277 rhs_and_solution,
278 &event_logger);
279#endif
280
281#endif // EIGEN_USE_EIGEN_SPARSE
282}

284LinearSolver::Summary SparseNormalCholeskySolver::SolveImplUsingCXSparse(
285 double* rhs_and_solution) {
286#ifdef CERES_NO_CXSPARSE
287
288 LinearSolver::Summary summary;
289 summary.num_iterations = 0;
290 summary.termination_type = LINEAR_SOLVER_FATAL_ERROR;
291 summary.message =
292 "SPARSE_NORMAL_CHOLESKY cannot be used with CX_SPARSE "
293 "because Ceres was not built with support for CXSparse. "
294 "This requires enabling building with -DCXSPARSE=ON.";
295
296 return summary;
297
298#else
299
300 EventLogger event_logger("SparseNormalCholeskySolver::CXSparse::Solve");
301 LinearSolver::Summary summary;
302 summary.num_iterations = 1;
303 summary.termination_type = LINEAR_SOLVER_SUCCESS;
304 summary.message = "Success.";
305
306 // Map outer_product_ to an upper triangular column major matrix.
307 //
308 // outer_product_ is a compressed row sparse matrix and in lower
309 // triangular form, when mapped to a compressed column sparse
310 // matrix, it becomes an upper triangular matrix.
311 cs_di lhs = cxsparse_.CreateSparseMatrixTransposeView(outer_product_.get());
312
313 event_logger.AddEvent("Setup");
314
315 // Compute symbolic factorization if not available.
316 if (cxsparse_factor_ == NULL) {
317 if (options_.use_postordering) {
318 cxsparse_factor_ = cxsparse_.BlockAnalyzeCholesky(
319 &lhs, outer_product_->col_blocks(), outer_product_->col_blocks());
320 } else {
321 cxsparse_factor_ = cxsparse_.AnalyzeCholeskyWithNaturalOrdering(&lhs);
322 }
323 }
324 event_logger.AddEvent("Analysis");
325
326 if (cxsparse_factor_ == NULL) {
327 summary.termination_type = LINEAR_SOLVER_FATAL_ERROR;
328 summary.message =
329 "CXSparse failure. Unable to find symbolic factorization.";
330 } else if (!cxsparse_.SolveCholesky(
331 &lhs, cxsparse_factor_, rhs_and_solution)) {
332 summary.termination_type = LINEAR_SOLVER_FAILURE;
333 summary.message = "CXSparse::SolveCholesky failed.";
334 }
335 event_logger.AddEvent("Solve");
336
337 return summary;
338#endif
339}

341LinearSolver::Summary SparseNormalCholeskySolver::SolveImplUsingSuiteSparse(
342 double* rhs_and_solution) {
343#ifdef CERES_NO_SUITESPARSE
344
345 LinearSolver::Summary summary;
346 summary.num_iterations = 0;
347 summary.termination_type = LINEAR_SOLVER_FATAL_ERROR;
348 summary.message =
349 "SPARSE_NORMAL_CHOLESKY cannot be used with SUITE_SPARSE "
350 "because Ceres was not built with support for SuiteSparse. "
351 "This requires enabling building with -DSUITESPARSE=ON.";
352 return summary;
353
354#else
355
356 EventLogger event_logger("SparseNormalCholeskySolver::SuiteSparse::Solve");
357 LinearSolver::Summary summary;
358 summary.termination_type = LINEAR_SOLVER_SUCCESS;
359 summary.num_iterations = 1;
360 summary.message = "Success.";
361
362 // Map outer_product_ to a lower triangular column major matrix.
363 //
364 // outer_product_ is a compressed row sparse matrix and in upper
365 // triangular form, when mapped to a compressed column sparse
366 // matrix, it becomes a lower triangular matrix.
367 const int num_cols = outer_product_->num_cols();
368 cholmod_sparse lhs =
369 ss_.CreateSparseMatrixTransposeView(outer_product_.get());
370 event_logger.AddEvent("Setup");
371
372 if (factor_ == NULL) {
373 if (options_.use_postordering) {
374 factor_ = ss_.BlockAnalyzeCholesky(
375 &lhs,
376 outer_product_->col_blocks(),
377 outer_product_->col_blocks(),
378 &summary.message);
379 } else {
380 factor_ = ss_.AnalyzeCholeskyWithNaturalOrdering(&lhs, &summary.message);
381 }
382 }
383
384 event_logger.AddEvent("Analysis");
385
386 if (factor_ == NULL) {
387 summary.termination_type = LINEAR_SOLVER_FATAL_ERROR;
388 // No need to set message as it has already been set by the
389 // symbolic analysis routines above.
390 return summary;
391 }
392
393 summary.termination_type = ss_.Cholesky(&lhs, factor_, &summary.message);
394 if (summary.termination_type != LINEAR_SOLVER_SUCCESS) {
395 return summary;
396 }
397
398 cholmod_dense* rhs =
399 ss_.CreateDenseVector(rhs_and_solution, num_cols, num_cols);
400 cholmod_dense* solution = ss_.Solve(factor_, rhs, &summary.message);
401 event_logger.AddEvent("Solve");
402
403 ss_.Free(rhs);
404 if (solution != NULL) {
405 memcpy(rhs_and_solution, solution->x, num_cols * sizeof(*rhs_and_solution));
406 ss_.Free(solution);
407 } else {
408 // No need to set message as it has already been set by the
409 // numeric factorization routine above.
410 summary.termination_type = LINEAR_SOLVER_FAILURE;
411 }
412
413 event_logger.AddEvent("Teardown");
414 return summary;
415#endif
416}
