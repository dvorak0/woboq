48using std::vector;
49
50typedef HashMap<int, int> IntMap;
51typedef HashSet<int> IntSet;
52
53class CanonicalViewsClustering {
54 public:
55 CanonicalViewsClustering() {}
56
57 // Compute the canonical views clustering of the vertices of the
58 // graph. centers will contain the vertices that are the identified
59 // as the canonical views/cluster centers, and membership is a map
60 // from vertices to cluster_ids. The i^th cluster center corresponds
61 // to the i^th cluster. It is possible depending on the
62 // configuration of the clustering algorithm that some of the
63 // vertices may not be assigned to any cluster. In this case they
64 // are assigned to a cluster with id = kInvalidClusterId.
65 void ComputeClustering(const CanonicalViewsClusteringOptions& options,
66 const WeightedGraph<int>& graph,
67 vector<int>* centers,
68 IntMap* membership);
69
70 private:
71 void FindValidViews(IntSet* valid_views) const;
72 double ComputeClusteringQualityDifference(const int candidate,
73 const vector<int>& centers) const;
74 void UpdateCanonicalViewAssignments(const int canonical_view);
75 void ComputeClusterMembership(const vector<int>& centers,
76 IntMap* membership) const;
77
78 CanonicalViewsClusteringOptions options_;
79 const WeightedGraph<int>* graph_;
80 // Maps a view to its representative canonical view (its cluster
81 // center).
82 IntMap view_to_canonical_view_;
83 // Maps a view to its similarity to its current cluster center.
84 HashMap<int, double> view_to_canonical_view_similarity_;
85 CERES_DISALLOW_COPY_AND_ASSIGN(CanonicalViewsClustering);
86};
87
88void ComputeCanonicalViewsClustering(
89 const CanonicalViewsClusteringOptions& options,
90 const WeightedGraph<int>& graph,
91 vector<int>* centers,
92 IntMap* membership) {
93 time_t start_time = time(NULL);
94 CanonicalViewsClustering cv;
95 cv.ComputeClustering(options, graph, centers, membership);
96 VLOG(2) << "Canonical views clustering time (secs): "
97 << time(NULL) - start_time;
98}

100// Implementation of CanonicalViewsClustering
101void CanonicalViewsClustering::ComputeClustering(
102 const CanonicalViewsClusteringOptions& options,
103 const WeightedGraph<int>& graph,
104 vector<int>* centers,
105 IntMap* membership) {
106 options_ = options;
107 CHECK_NOTNULL(centers)->clear();
108 CHECK_NOTNULL(membership)->clear();
109 graph_ = &graph;
110
111 IntSet valid_views;
112 FindValidViews(&valid_views);
113 while (valid_views.size() > 0) {
114 // Find the next best canonical view.
115 double best_difference = -std::numeric_limits<double>::max();
116 int best_view = 0;
117
118 // TODO(sameeragarwal): Make this loop multi-threaded.
119 for (IntSet::const_iterator view = valid_views.begin();
120 view != valid_views.end();
121 ++view) {
122 const double difference =
123 ComputeClusteringQualityDifference(*view, *centers);
124 if (difference > best_difference) {
125 best_difference = difference;
126 best_view = *view;
127 }
128 }
129
130 CHECK_GT(best_difference, -std::numeric_limits<double>::max());
131
132 // Add canonical view if quality improves, or if minimum is not
133 // yet met, otherwise break.
134 if ((best_difference <= 0) &&
135 (centers->size() >= options_.min_views)) {
136 break;
137 }
138
139 centers->push_back(best_view);
140 valid_views.erase(best_view);
141 UpdateCanonicalViewAssignments(best_view);
142 }
143
144 ComputeClusterMembership(*centers, membership);
145}
146
147// Return the set of vertices of the graph which have valid vertex
148// weights.
149void CanonicalViewsClustering::FindValidViews(
150 IntSet* valid_views) const {
151 const IntSet& views = graph_->vertices();
152 for (IntSet::const_iterator view = views.begin();
153 view != views.end();
154 ++view) {
155 if (graph_->VertexWeight(*view) != WeightedGraph<int>::InvalidWeight()) {
156 valid_views->insert(*view);
157 }
158 }
159}

161// Computes the difference in the quality score if 'candidate' were
162// added to the set of canonical views.
163double CanonicalViewsClustering::ComputeClusteringQualityDifference(
164 const int candidate,
165 const vector<int>& centers) const {
166 // View score.
167 double difference =
168 options_.view_score_weight * graph_->VertexWeight(candidate);
169
170 // Compute how much the quality score changes if the candidate view
171 // was added to the list of canonical views and its nearest
172 // neighbors became members of its cluster.
173 const IntSet& neighbors = graph_->Neighbors(candidate);
174 for (IntSet::const_iterator neighbor = neighbors.begin();
175 neighbor != neighbors.end();
176 ++neighbor) {
177 const double old_similarity =
178 FindWithDefault(view_to_canonical_view_similarity_, *neighbor, 0.0);
179 const double new_similarity = graph_->EdgeWeight(*neighbor, candidate);
180 if (new_similarity > old_similarity) {
181 difference += new_similarity - old_similarity;
182 }
183 }
184
185 // Number of views penalty.
186 difference -= options_.size_penalty_weight;
187
188 // Orthogonality.
189 for (int i = 0; i < centers.size(); ++i) {
190 difference -= options_.similarity_penalty_weight *
191 graph_->EdgeWeight(centers[i], candidate);
192 }
193
194 return difference;
195}
196
197// Reassign views if they're more similar to the new canonical view.
198void CanonicalViewsClustering::UpdateCanonicalViewAssignments(
199 const int canonical_view) {
200 const IntSet& neighbors = graph_->Neighbors(canonical_view);
201 for (IntSet::const_iterator neighbor = neighbors.begin();
202 neighbor != neighbors.end();
203 ++neighbor) {
204 const double old_similarity =
205 FindWithDefault(view_to_canonical_view_similarity_, *neighbor, 0.0);
206 const double new_similarity =
207 graph_->EdgeWeight(*neighbor, canonical_view);
208 if (new_similarity > old_similarity) {
209 view_to_canonical_view_[*neighbor] = canonical_view;
210 view_to_canonical_view_similarity_[*neighbor] = new_similarity;
211 }
212 }
213}
214
215// Assign a cluster id to each view.
216void CanonicalViewsClustering::ComputeClusterMembership(
217 const vector<int>& centers,
218 IntMap* membership) const {
219 CHECK_NOTNULL(membership)->clear();
220
221 // The i^th cluster has cluster id i.
222 IntMap center_to_cluster_id;
223 for (int i = 0; i < centers.size(); ++i) {
224 center_to_cluster_id[centers[i]] = i;
225 }
226
227 static const int kInvalidClusterId = -1;
228
229 const IntSet& views = graph_->vertices();
230 for (IntSet::const_iterator view = views.begin();
231 view != views.end();
232 ++view) {
233 IntMap::const_iterator it =
234 view_to_canonical_view_.find(*view);
235 int cluster_id = kInvalidClusterId;
236 if (it != view_to_canonical_view_.end()) {
237 cluster_id = FindOrDie(center_to_cluster_id, it->second);
238 }
239
240 InsertOrDie(membership, *view, cluster_id);
241 }
