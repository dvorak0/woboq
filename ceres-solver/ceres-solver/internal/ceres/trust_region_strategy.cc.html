33#include "ceres/trust_region_strategy.h"
34#include "ceres/dogleg_strategy.h"
35#include "ceres/levenberg_marquardt_strategy.h"
36
37namespace ceres {
38namespace internal {
39
40TrustRegionStrategy::~TrustRegionStrategy() {}
41
42TrustRegionStrategy* TrustRegionStrategy::Create(const Options& options) {
43 switch (options.trust_region_strategy_type) {
44 case LEVENBERG_MARQUARDT:
45 return new LevenbergMarquardtStrategy(options);
46 case DOGLEG:
47 return new DoglegStrategy(options);
48 default:
49 LOG(FATAL) << "Unknown trust region strategy: "
50 << options.trust_region_strategy_type;
51 }
52
53 LOG(FATAL) << "Unknown trust region strategy: "
54 << options.trust_region_strategy_type;
55 return NULL;
56}
