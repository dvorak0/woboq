59bool CXSparse::SolveCholesky(cs_di* A,
60 cs_dis* symbolic_factorization,
61 double* b) {
62 // Make sure we have enough scratch space available.
63 if (scratch_size_ < A->n) {
64 if (scratch_size_ > 0) {
65 cs_di_free(scratch_);
66 }
67 scratch_ =
68 reinterpret_cast<CS_ENTRY*>(cs_di_malloc(A->n, sizeof(CS_ENTRY)));
69 scratch_size_ = A->n;
70 }
71
72 // Solve using Cholesky factorization
73 csn* numeric_factorization = cs_di_chol(A, symbolic_factorization);
74 if (numeric_factorization == NULL) {
75 LOG(WARNING) << "Cholesky factorization failed.";
76 return false;
77 }
78
79 // When the Cholesky factorization succeeded, these methods are
80 // guaranteed to succeeded as well. In the comments below, "x"
81 // refers to the scratch space.
82 //
83 // Set x = P * b.
84 cs_di_ipvec(symbolic_factorization->pinv, b, scratch_, A->n);
85 // Set x = L \ x.
86 cs_di_lsolve(numeric_factorization->L, scratch_);
87 // Set x = L' \ x.
88 cs_di_ltsolve(numeric_factorization->L, scratch_);
89 // Set b = P' * x.
90 cs_di_pvec(symbolic_factorization->pinv, scratch_, b, A->n);
91
92 // Free Cholesky factorization.
93 cs_di_nfree(numeric_factorization);
94 return true;
95}

107cs_dis* CXSparse::BlockAnalyzeCholesky(cs_di* A,
108 const vector<int>& row_blocks,
109 const vector<int>& col_blocks) {
110 const int num_row_blocks = row_blocks.size();
111 const int num_col_blocks = col_blocks.size();
112
113 vector<int> block_rows;
114 vector<int> block_cols;
115 CompressedColumnScalarMatrixToBlockMatrix(A->i,
116 A->p,
117 row_blocks,
118 col_blocks,
119 &block_rows,
120 &block_cols);
121 cs_di block_matrix;
122 block_matrix.m = num_row_blocks;
123 block_matrix.n = num_col_blocks;
124 block_matrix.nz = -1;
125 block_matrix.nzmax = block_rows.size();
126 block_matrix.p = &block_cols[0];
127 block_matrix.i = &block_rows[0];
128 block_matrix.x = NULL;
129
130 int* ordering = cs_amd(1, &block_matrix);
131 vector<int> block_ordering(num_row_blocks, -1);
132 std::copy(ordering, ordering + num_row_blocks, &block_ordering[0]);
133 cs_free(ordering);
134
135 vector<int> scalar_ordering;
136 BlockOrderingToScalarOrdering(row_blocks, block_ordering, &scalar_ordering);
137
138 cs_dis* symbolic_factorization =
139 reinterpret_cast<cs_dis*>(cs_calloc(1, sizeof(cs_dis)));
140 symbolic_factorization->pinv = cs_pinv(&scalar_ordering[0], A->n);
141 cs* permuted_A = cs_symperm(A, symbolic_factorization->pinv, 0);
142
143 symbolic_factorization->parent = cs_etree(permuted_A, 0);
144 int* postordering = cs_post(symbolic_factorization->parent, A->n);
145 int* column_counts = cs_counts(permuted_A,
146 symbolic_factorization->parent,
147 postordering,
148 0);
149 cs_free(postordering);
150 cs_spfree(permuted_A);
151
152 symbolic_factorization->cp = (int*) cs_malloc(A->n+1, sizeof(int));
153 symbolic_factorization->lnz = cs_cumsum(symbolic_factorization->cp,
154 column_counts,
155 A->n);
156 symbolic_factorization->unz = symbolic_factorization->lnz;
157
158 cs_free(column_counts);
159
160 if (symbolic_factorization->lnz < 0) {
161 cs_sfree(symbolic_factorization);
162 symbolic_factorization = NULL;
163 }
164
165 return symbolic_factorization;
166}

168cs_di CXSparse::CreateSparseMatrixTransposeView(CompressedRowSparseMatrix* A) {
169 cs_di At;
170 At.m = A->num_cols();
171 At.n = A->num_rows();
172 At.nz = -1;
173 At.nzmax = A->num_nonzeros();
174 At.p = A->mutable_rows();
175 At.i = A->mutable_cols();
176 At.x = A->mutable_values();
177 return At;
178}

180cs_di* CXSparse::CreateSparseMatrix(TripletSparseMatrix* tsm) {
181 cs_di_sparse tsm_wrapper;
182 tsm_wrapper.nzmax = tsm->num_nonzeros();
183 tsm_wrapper.nz = tsm->num_nonzeros();
184 tsm_wrapper.m = tsm->num_rows();
185 tsm_wrapper.n = tsm->num_cols();
186 tsm_wrapper.p = tsm->mutable_cols();
187 tsm_wrapper.i = tsm->mutable_rows();
188 tsm_wrapper.x = tsm->mutable_values();
189
190 return cs_compress(&tsm_wrapper);
191}

193void CXSparse::ApproximateMinimumDegreeOrdering(cs_di* A, int* ordering) {
194 int* cs_ordering = cs_amd(1, A);
195 std::copy(cs_ordering, cs_ordering + A->m, ordering);
196 cs_free(cs_ordering);
197}
