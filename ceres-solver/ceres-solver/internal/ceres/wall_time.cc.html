45namespace ceres {
46namespace internal {
47
48double WallTimeInSeconds() {
49#ifdef CERES_USE_OPENMP
50 return omp_get_wtime();
51#else
52#ifdef _WIN32
53 return static_cast<double>(std::time(NULL));
54#else
55 timeval time_val;
56 gettimeofday(&time_val, NULL);
57 return (time_val.tv_sec + time_val.tv_usec * 1e-6);
58#endif
59#endif
60}
61
62EventLogger::EventLogger(const std::string& logger_name)
63 : start_time_(WallTimeInSeconds()),
64 last_event_time_(start_time_),
65 events_("") {
66 StringAppendF(&events_,
67 "\n%s\n Delta Cumulative\n",
68 logger_name.c_str());
69}
70
71EventLogger::~EventLogger() {
72 if (VLOG_IS_ON(3)) {
73 AddEvent("Total");
74 VLOG(2) << "\n" << events_ << "\n";
75 }
76}
77
78void EventLogger::AddEvent(const std::string& event_name) {
79 if (!VLOG_IS_ON(3)) {
80 return;
81 }
82
83 const double current_time = WallTimeInSeconds();
84 const double relative_time_delta = current_time - last_event_time_;
85 const double absolute_time_delta = current_time - start_time_;
86 last_event_time_ = current_time;
87
88 StringAppendF(&events_,
89 " %30s : %10.5f %10.5f\n",
90 event_name.c_str(),
91 relative_time_delta,
92 absolute_time_delta);
93}
