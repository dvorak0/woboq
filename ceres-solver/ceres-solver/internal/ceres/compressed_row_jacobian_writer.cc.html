52void CompressedRowJacobianWriter::PopulateJacobianRowAndColumnBlockVectors(
53 const Program* program, CompressedRowSparseMatrix* jacobian) {
54 const vector<ParameterBlock*>& parameter_blocks =
55 program->parameter_blocks();
56 vector<int>& col_blocks = *(jacobian->mutable_col_blocks());
57 col_blocks.resize(parameter_blocks.size());
58 for (int i = 0; i < parameter_blocks.size(); ++i) {
59 col_blocks[i] = parameter_blocks[i]->LocalSize();
60 }
61
62 const vector<ResidualBlock*>& residual_blocks =
63 program->residual_blocks();
64 vector<int>& row_blocks = *(jacobian->mutable_row_blocks());
65 row_blocks.resize(residual_blocks.size());
66 for (int i = 0; i < residual_blocks.size(); ++i) {
67 row_blocks[i] = residual_blocks[i]->NumResiduals();
68 }
69}

71void CompressedRowJacobianWriter::GetOrderedParameterBlocks(
72 const Program* program,
73 int residual_id,
74 vector<pair<int, int> >* evaluated_jacobian_blocks) {
75 const ResidualBlock* residual_block =
76 program->residual_blocks()[residual_id];
77 const int num_parameter_blocks = residual_block->NumParameterBlocks();
78
79 for (int j = 0; j < num_parameter_blocks; ++j) {
80 const ParameterBlock* parameter_block =
81 residual_block->parameter_blocks()[j];
82 if (!parameter_block->IsConstant()) {
83 evaluated_jacobian_blocks->push_back(
84 make_pair(parameter_block->index(), j));
85 }
86 }
87 sort(evaluated_jacobian_blocks->begin(), evaluated_jacobian_blocks->end());
88}

90SparseMatrix* CompressedRowJacobianWriter::CreateJacobian() const {
91 const vector<ResidualBlock*>& residual_blocks =
92 program_->residual_blocks();
93
94 int total_num_residuals = program_->NumResiduals();
95 int total_num_effective_parameters = program_->NumEffectiveParameters();
96
97 // Count the number of jacobian nonzeros.
98 int num_jacobian_nonzeros = 0;
99 for (int i = 0; i < residual_blocks.size(); ++i) {
100 ResidualBlock* residual_block = residual_blocks[i];
101 const int num_residuals = residual_block->NumResiduals();
102 const int num_parameter_blocks = residual_block->NumParameterBlocks();
103 for (int j = 0; j < num_parameter_blocks; ++j) {
104 ParameterBlock* parameter_block = residual_block->parameter_blocks()[j];
105 if (!parameter_block->IsConstant()) {
106 num_jacobian_nonzeros += num_residuals * parameter_block->LocalSize();
107 }
108 }
109 }
110
111 // Allocate storage for the jacobian with some extra space at the end.
112 // Allocate more space than needed to store the jacobian so that when the LM
113 // algorithm adds the diagonal, no reallocation is necessary. This reduces
114 // peak memory usage significantly.
115 CompressedRowSparseMatrix* jacobian =
116 new CompressedRowSparseMatrix(
117 total_num_residuals,
118 total_num_effective_parameters,
119 num_jacobian_nonzeros + total_num_effective_parameters);
120
121 // At this stage, the CompressedRowSparseMatrix is an invalid state. But this
122 // seems to be the only way to construct it without doing a memory copy.
123 int* rows = jacobian->mutable_rows();
124 int* cols = jacobian->mutable_cols();
125
126 // Initialize crsb rows and cols.
127 std::vector<int>& crsb_rows = *jacobian->mutable_crsb_rows();
128 std::vector<int>& crsb_cols = *jacobian->mutable_crsb_cols();
129 crsb_rows.resize(residual_blocks.size() + 1);
130 crsb_rows[0] = 0;
131
132 int row_pos = 0;
133 rows[0] = 0;
134 for (int i = 0; i < residual_blocks.size(); ++i) {
135 const ResidualBlock* residual_block = residual_blocks[i];
136 const int num_parameter_blocks = residual_block->NumParameterBlocks();
137
138 // Count the number of derivatives for a row of this residual block and
139 // build a list of active parameter block indices.
140 int num_derivatives = 0;
141 vector<int> parameter_indices;
142 for (int j = 0; j < num_parameter_blocks; ++j) {
143 ParameterBlock* parameter_block = residual_block->parameter_blocks()[j];
144 if (!parameter_block->IsConstant()) {
145 parameter_indices.push_back(parameter_block->index());
146 num_derivatives += parameter_block->LocalSize();
147 }
148 }
149
150 // Sort the parameters by their position in the state vector.
151 sort(parameter_indices.begin(), parameter_indices.end());
152 if (adjacent_find(parameter_indices.begin(), parameter_indices.end()) !=
153 parameter_indices.end()) {
154 std::string parameter_block_description;
155 for (int j = 0; j < num_parameter_blocks; ++j) {
156 ParameterBlock* parameter_block = residual_block->parameter_blocks()[j];
157 parameter_block_description +=
158 parameter_block->ToString() + "\n";
159 }
160 LOG(FATAL) << "Ceres internal error: "
161 << "Duplicate parameter blocks detected in a cost function. "
162 << "This should never happen. Please report this to "
163 << "the Ceres developers.\n"
164 << "Residual Block: " << residual_block->ToString() << "\n"
165 << "Parameter Blocks: " << parameter_block_description;
166 }
167
168 // Populate crsb rows and cols.
169 crsb_rows[i + 1] = crsb_rows[i] + parameter_indices.size();
170 std::copy(parameter_indices.begin(), parameter_indices.end(),
171 std::back_inserter(crsb_cols));
172
173 // Update the row indices.
174 const int num_residuals = residual_block->NumResiduals();
175 for (int j = 0; j < num_residuals; ++j) {
176 rows[row_pos + j + 1] = rows[row_pos + j] + num_derivatives;
177 }
178
179 // Iterate over parameter blocks in the order which they occur in the
180 // parameter vector. This code mirrors that in Write(), where jacobian
181 // values are updated.
182 int col_pos = 0;
183 for (int j = 0; j < parameter_indices.size(); ++j) {
184 ParameterBlock* parameter_block =
185 program_->parameter_blocks()[parameter_indices[j]];
186 const int parameter_block_size = parameter_block->LocalSize();
187
188 for (int r = 0; r < num_residuals; ++r) {
189 // This is the position in the values array of the jacobian where this
190 // row of the jacobian block should go.
191 const int column_block_begin = rows[row_pos + r] + col_pos;
192
193 for (int c = 0; c < parameter_block_size; ++c) {
194 cols[column_block_begin + c] = parameter_block->delta_offset() + c;
195 }
196 }
197 col_pos += parameter_block_size;
198 }
199 row_pos += num_residuals;
200 }
201 CHECK_EQ(num_jacobian_nonzeros, rows[total_num_residuals]);
202
203 PopulateJacobianRowAndColumnBlockVectors(program_, jacobian);
204
205 return jacobian;
206}

208void CompressedRowJacobianWriter::Write(int residual_id,
209 int residual_offset,
210 double **jacobians,
211 SparseMatrix* base_jacobian) {
212 CompressedRowSparseMatrix* jacobian =
213 down_cast<CompressedRowSparseMatrix*>(base_jacobian);
214
215 double* jacobian_values = jacobian->mutable_values();
216 const int* jacobian_rows = jacobian->rows();
217
218 const ResidualBlock* residual_block =
219 program_->residual_blocks()[residual_id];
220 const int num_residuals = residual_block->NumResiduals();
221
222 vector<pair<int, int> > evaluated_jacobian_blocks;
223 GetOrderedParameterBlocks(program_, residual_id, &evaluated_jacobian_blocks);
224
225 // Where in the current row does the jacobian for a parameter block begin.
226 int col_pos = 0;
227
228 // Iterate over the jacobian blocks in increasing order of their
229 // positions in the reduced parameter vector.
230 for (int i = 0; i < evaluated_jacobian_blocks.size(); ++i) {
231 const ParameterBlock* parameter_block =
232 program_->parameter_blocks()[evaluated_jacobian_blocks[i].first];
233 const int argument = evaluated_jacobian_blocks[i].second;
234 const int parameter_block_size = parameter_block->LocalSize();
235
236 // Copy one row of the jacobian block at a time.
237 for (int r = 0; r < num_residuals; ++r) {
238 // Position of the r^th row of the current jacobian block.
239 const double* block_row_begin =
240 jacobians[argument] + r * parameter_block_size;
241
242 // Position in the values array of the jacobian where this
243 // row of the jacobian block should go.
244 double* column_block_begin =
245 jacobian_values + jacobian_rows[residual_offset + r] + col_pos;
246
247 std::copy(block_row_begin,
248 block_row_begin + parameter_block_size,
249 column_block_begin);
250 }
251 col_pos += parameter_block_size;
252 }
253}
