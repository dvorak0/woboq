51void CompareMatrices(const SparseMatrix* a, const SparseMatrix* b) {
52 EXPECT_EQ(a->num_rows(), b->num_rows());
53 EXPECT_EQ(a->num_cols(), b->num_cols());
54
55 int num_rows = a->num_rows();
56 int num_cols = a->num_cols();
57
58 for (int i = 0; i < num_cols; ++i) {
59 Vector x = Vector::Zero(num_cols);
60 x(i) = 1.0;
61
62 Vector y_a = Vector::Zero(num_rows);
63 Vector y_b = Vector::Zero(num_rows);
64
65 a->RightMultiply(x.data(), y_a.data());
66 b->RightMultiply(x.data(), y_b.data());
67
68 EXPECT_EQ((y_a - y_b).norm(), 0);
69 }
70}

72class CompressedRowSparseMatrixTest : public ::testing::Test {
73 protected:
74 virtual void SetUp() {
75 scoped_ptr<LinearLeastSquaresProblem> problem(
76 CreateLinearLeastSquaresProblemFromId(1));
77
78 CHECK_NOTNULL(problem.get());
79
80 tsm.reset(down_cast<TripletSparseMatrix*>(problem->A.release()));
81 crsm.reset(CompressedRowSparseMatrix::FromTripletSparseMatrix(*tsm));
82
83 num_rows = tsm->num_rows();
84 num_cols = tsm->num_cols();
85
86 vector<int>* row_blocks = crsm->mutable_row_blocks();
87 row_blocks->resize(num_rows);
88 std::fill(row_blocks->begin(), row_blocks->end(), 1);
89
90 vector<int>* col_blocks = crsm->mutable_col_blocks();
91 col_blocks->resize(num_cols);
92 std::fill(col_blocks->begin(), col_blocks->end(), 1);
93
94 // With all blocks of size 1, crsb_rows and crsb_cols are equivalent to
95 // rows and cols.
96 std::copy(crsm->rows(),
97 crsm->rows() + crsm->num_rows() + 1,
98 std::back_inserter(*crsm->mutable_crsb_rows()));
99 std::copy(crsm->cols(),
100 crsm->cols() + crsm->num_nonzeros(),
101 std::back_inserter(*crsm->mutable_crsb_cols()));
102 }
103
104 int num_rows;
105 int num_cols;
106
107 scoped_ptr<TripletSparseMatrix> tsm;
108 scoped_ptr<CompressedRowSparseMatrix> crsm;
109};

185TEST_F(CompressedRowSparseMatrixTest, AppendAndDeleteBlockDiagonalMatrix) {
186 int num_diagonal_rows = crsm->num_cols();
187
188 scoped_array<double> diagonal(new double[num_diagonal_rows]);
189 for (int i = 0; i < num_diagonal_rows; ++i) {
190 diagonal[i] = i;
191 }
192
193 vector<int> row_and_column_blocks;
194 row_and_column_blocks.push_back(1);
195 row_and_column_blocks.push_back(2);
196 row_and_column_blocks.push_back(2);
197
198 const vector<int> pre_row_blocks = crsm->row_blocks();
199 const vector<int> pre_col_blocks = crsm->col_blocks();
200
201 const vector<int> pre_crsb_rows = crsm->crsb_rows();
202 const vector<int> pre_crsb_cols = crsm->crsb_cols();
203
204 scoped_ptr<CompressedRowSparseMatrix> appendage(
205 CompressedRowSparseMatrix::CreateBlockDiagonalMatrix(
206 diagonal.get(), row_and_column_blocks));
207 LOG(INFO) << appendage->row_blocks().size();
208
209 crsm->AppendRows(*appendage);
210
211 const vector<int> post_row_blocks = crsm->row_blocks();
212 const vector<int> post_col_blocks = crsm->col_blocks();
213
214 vector<int> expected_row_blocks = pre_row_blocks;
215 expected_row_blocks.insert(expected_row_blocks.end(),
216 row_and_column_blocks.begin(),
217 row_and_column_blocks.end());
218
219 vector<int> expected_col_blocks = pre_col_blocks;
220
221 EXPECT_EQ(expected_row_blocks, crsm->row_blocks());
222 EXPECT_EQ(expected_col_blocks, crsm->col_blocks());
223
224 EXPECT_EQ(crsm->crsb_cols().size(),
225 pre_crsb_cols.size() + row_and_column_blocks.size());
226 EXPECT_EQ(crsm->crsb_rows().size(),
227 pre_crsb_rows.size() + row_and_column_blocks.size());
228 for (int i = 0; i < row_and_column_blocks.size(); ++i) {
229 EXPECT_EQ(crsm->crsb_rows()[i + pre_crsb_rows.size()],
230 pre_crsb_rows.back() + i + 1);
231 EXPECT_EQ(crsm->crsb_cols()[i + pre_crsb_cols.size()], i);
232 }
233
234 crsm->DeleteRows(num_diagonal_rows);
235 EXPECT_EQ(crsm->row_blocks(), pre_row_blocks);
236 EXPECT_EQ(crsm->col_blocks(), pre_col_blocks);
237
238 EXPECT_EQ(crsm->crsb_rows(), pre_crsb_rows);
239 EXPECT_EQ(crsm->crsb_cols(), pre_crsb_cols);
240}

271TEST(CompressedRowSparseMatrix, CreateBlockDiagonalMatrix) {
272 vector<int> blocks;
273 blocks.push_back(1);
274 blocks.push_back(2);
275 blocks.push_back(2);
276
277 Vector diagonal(5);
278 for (int i = 0; i < 5; ++i) {
279 diagonal(i) = i + 1;
280 }
281
282 scoped_ptr<CompressedRowSparseMatrix> matrix(
283 CompressedRowSparseMatrix::CreateBlockDiagonalMatrix(diagonal.data(),
284 blocks));
285
286 EXPECT_EQ(matrix->num_rows(), 5);
287 EXPECT_EQ(matrix->num_cols(), 5);
288 EXPECT_EQ(matrix->num_nonzeros(), 9);
289 EXPECT_EQ(blocks, matrix->row_blocks());
290 EXPECT_EQ(blocks, matrix->col_blocks());
291
292 Vector x(5);
293 Vector y(5);
294
295 x.setOnes();
296 y.setZero();
297 matrix->RightMultiply(x.data(), y.data());
298 for (int i = 0; i < diagonal.size(); ++i) {
299 EXPECT_EQ(y[i], diagonal[i]);
300 }
301
302 y.setZero();
303 matrix->LeftMultiply(x.data(), y.data());
304 for (int i = 0; i < diagonal.size(); ++i) {
305 EXPECT_EQ(y[i], diagonal[i]);
306 }
307
308 Matrix dense;
309 matrix->ToDenseMatrix(&dense);
310 EXPECT_EQ((dense.diagonal() - diagonal).norm(), 0.0);
311}

313TEST(CompressedRowSparseMatrix, Transpose) {
314 // 0 1 0 2 3 0
315 // 4 6 7 0 0 8
316 // 9 10 0 11 12 0
317 // 13 0 14 15 9 0
318 // 0 16 17 0 0 0
319
320 // Block structure:
321 // A A A A B B
322 // A A A A B B
323 // A A A A B B
324 // C C C C D D
325 // C C C C D D
326 // C C C C D D
327
328 CompressedRowSparseMatrix matrix(5, 6, 30);
329 int* rows = matrix.mutable_rows();
330 int* cols = matrix.mutable_cols();
331 double* values = matrix.mutable_values();
332 matrix.mutable_row_blocks()->push_back(3);
333 matrix.mutable_row_blocks()->push_back(3);
334 matrix.mutable_col_blocks()->push_back(4);
335 matrix.mutable_col_blocks()->push_back(2);
336
337 matrix.mutable_crsb_rows()->push_back(0);
338 matrix.mutable_crsb_rows()->push_back(2);
339 matrix.mutable_crsb_rows()->push_back(4);
340 matrix.mutable_crsb_cols()->push_back(0);
341 matrix.mutable_crsb_cols()->push_back(1);
342 matrix.mutable_crsb_cols()->push_back(0);
343 matrix.mutable_crsb_cols()->push_back(1);
344
345 rows[0] = 0;
346 cols[0] = 1;
347 cols[1] = 3;
348 cols[2] = 4;
349
350 rows[1] = 3;
351 cols[3] = 0;
352 cols[4] = 1;
353 cols[5] = 2;
354 cols[6] = 5;
355
356 rows[2] = 7;
357 cols[7] = 0;
358 cols[8] = 1;
359 cols[9] = 3;
360 cols[10] = 4;
361
362 rows[3] = 11;
363 cols[11] = 0;
364 cols[12] = 2;
365 cols[13] = 3;
366 cols[14] = 4;
367
368 rows[4] = 15;
369 cols[15] = 1;
370 cols[16] = 2;
371 rows[5] = 17;
372
373 std::copy(values, values + 17, cols);
374
375 scoped_ptr<CompressedRowSparseMatrix> transpose(matrix.Transpose());
376
377 ASSERT_EQ(transpose->row_blocks().size(), matrix.col_blocks().size());
378 for (int i = 0; i < transpose->row_blocks().size(); ++i) {
379 EXPECT_EQ(transpose->row_blocks()[i], matrix.col_blocks()[i]);
380 }
381
382 ASSERT_EQ(transpose->col_blocks().size(), matrix.row_blocks().size());
383 for (int i = 0; i < transpose->col_blocks().size(); ++i) {
384 EXPECT_EQ(transpose->col_blocks()[i], matrix.row_blocks()[i]);
385 }
386
387 Matrix dense_matrix;
388 matrix.ToDenseMatrix(&dense_matrix);
389
390 Matrix dense_transpose;
391 transpose->ToDenseMatrix(&dense_transpose);
392 EXPECT_NEAR((dense_matrix - dense_transpose.transpose()).norm(), 0.0, 1e-14);
393}

395TEST(CompressedRowSparseMatrix, ComputeOuterProduct) {
396 // "Randomly generated seed."
397 SetRandomState(29823);
398 const int kMaxNumRowBlocks = 10;
399 const int kMaxNumColBlocks = 10;
400 const int kNumTrials = 10;
401
402 // Create a random matrix, compute its outer product using Eigen and
403 // ComputeOuterProduct. Convert both matrices to dense matrices and
404 // compare their upper triangular parts.
405 for (int num_row_blocks = 1; num_row_blocks < kMaxNumRowBlocks;
406 ++num_row_blocks) {
407 for (int num_col_blocks = 1; num_col_blocks < kMaxNumColBlocks;
408 ++num_col_blocks) {
409 for (int trial = 0; trial < kNumTrials; ++trial) {
410 CompressedRowSparseMatrix::RandomMatrixOptions options;
411 options.num_row_blocks = num_row_blocks;
412 options.num_col_blocks = num_col_blocks;
413 options.min_row_block_size = 1;
414 options.max_row_block_size = 5;
415 options.min_col_block_size = 1;
416 options.max_col_block_size = 10;
417 options.block_density = std::max(0.1, RandDouble());
418
419 VLOG(2) << "num row blocks: " << options.num_row_blocks;
420 VLOG(2) << "num col blocks: " << options.num_col_blocks;
421 VLOG(2) << "min row block size: " << options.min_row_block_size;
422 VLOG(2) << "max row block size: " << options.max_row_block_size;
423 VLOG(2) << "min col block size: " << options.min_col_block_size;
424 VLOG(2) << "max col block size: " << options.max_col_block_size;
425 VLOG(2) << "block density: " << options.block_density;
426
427 scoped_ptr<CompressedRowSparseMatrix> random_matrix(
428 CompressedRowSparseMatrix::CreateRandomMatrix(options));
429
430 Eigen::MappedSparseMatrix<double, Eigen::RowMajor> mapped_random_matrix(
431 random_matrix->num_rows(),
432 random_matrix->num_cols(),
433 random_matrix->num_nonzeros(),
434 random_matrix->mutable_rows(),
435 random_matrix->mutable_cols(),
436 random_matrix->mutable_values());
437
438 Matrix expected_outer_product =
439 mapped_random_matrix.transpose() * mapped_random_matrix;
440
441 // Use compressed row lower triangular matrix, which will then
442 // get mapped to a compressed column upper triangular matrix.
443 vector<int> program;
444 scoped_ptr<CompressedRowSparseMatrix> outer_product(
445 CompressedRowSparseMatrix::CreateOuterProductMatrixAndProgram(
446 *random_matrix,
447 CompressedRowSparseMatrix::LOWER_TRIANGULAR,
448 &program));
449 CompressedRowSparseMatrix::ComputeOuterProduct(
450 *random_matrix, program, outer_product.get());
451
452 EXPECT_EQ(outer_product->row_blocks(), random_matrix->col_blocks());
453 EXPECT_EQ(outer_product->col_blocks(), random_matrix->col_blocks());
454
455 Matrix actual_outer_product =
456 Eigen::MappedSparseMatrix<double, Eigen::ColMajor>(
457 outer_product->num_rows(),
458 outer_product->num_rows(),
459 outer_product->num_nonzeros(),
460 outer_product->mutable_rows(),
461 outer_product->mutable_cols(),
462 outer_product->mutable_values());
463 expected_outer_product.triangularView<Eigen::StrictlyLower>().setZero();
464 actual_outer_product.triangularView<Eigen::StrictlyLower>().setZero();
465
466 EXPECT_EQ(actual_outer_product.rows(), actual_outer_product.cols());
467 EXPECT_EQ(expected_outer_product.rows(), expected_outer_product.cols());
468 EXPECT_EQ(actual_outer_product.rows(), expected_outer_product.rows());
469
470 const double diff_norm =
471 (actual_outer_product - expected_outer_product).norm() /
472 expected_outer_product.norm();
473 EXPECT_NEAR(diff_norm, 0.0, std::numeric_limits<double>::epsilon())
474 << "expected: \n"
475 << expected_outer_product << "\nactual: \n"
476 << actual_outer_product;
477 }
478 }
479 }

482TEST(CompressedRowSparseMatrix, FromTripletSparseMatrix) {
483 TripletSparseMatrix::RandomMatrixOptions options;
484 options.num_rows = 5;
485 options.num_cols = 7;
486 options.density = 0.5;
487
