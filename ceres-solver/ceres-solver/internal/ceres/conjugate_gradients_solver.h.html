32// semidefinite linear systems.

34#ifndef CERES_INTERNAL_CONJUGATE_GRADIENTS_SOLVER_H_
35#define CERES_INTERNAL_CONJUGATE_GRADIENTS_SOLVER_H_
36
37#include "ceres/linear_solver.h"
38#include "ceres/internal/macros.h"
39
40namespace ceres {
41namespace internal {
42
43class LinearOperator;
44
45// This class implements the now classical Conjugate Gradients
46// algorithm of Hestenes & Stiefel for solving postive semidefinite
47// linear sytems. Optionally it can use a preconditioner also to
48// reduce the condition number of the linear system and improve the
49// convergence rate. Modern references for Conjugate Gradients are the
50// books by Yousef Saad and Trefethen & Bau. This implementation of CG
51// has been augmented with additional termination tests that are
52// needed for forcing early termination when used as part of an
53// inexact Newton solver.
54//
55// For more details see the documentation for
56// LinearSolver::PerSolveOptions::r_tolerance and
57// LinearSolver::PerSolveOptions::q_tolerance in linear_solver.h.
58class ConjugateGradientsSolver : public LinearSolver {
59 public:
60 explicit ConjugateGradientsSolver(const LinearSolver::Options& options);
61 virtual Summary Solve(LinearOperator* A,
62 const double* b,
63 const LinearSolver::PerSolveOptions& per_solve_options,
64 double* x);
65
66 private:
67 const LinearSolver::Options options_;
68 CERES_DISALLOW_COPY_AND_ASSIGN(ConjugateGradientsSolver);
69};
70
71} // namespace internal
72} // namespace ceres

74#endif // CERES_INTERNAL_CONJUGATE_GRADIENTS_SOLVER_H_
