31#ifndef CERES_INTERNAL_LINE_SEARCH_DIRECTION_H_
32#define CERES_INTERNAL_LINE_SEARCH_DIRECTION_H_
33
34#include "ceres/internal/eigen.h"
35#include "ceres/line_search_minimizer.h"
36#include "ceres/types.h"
37
38namespace ceres {
39namespace internal {
40
41class LineSearchDirection {
42 public:
43 struct Options {
44 Options()
45 : num_parameters(0),
46 type(LBFGS),
47 nonlinear_conjugate_gradient_type(FLETCHER_REEVES),
48 function_tolerance(1e-12),
49 max_lbfgs_rank(20),
50 use_approximate_eigenvalue_bfgs_scaling(true) {
51 }
52
53 int num_parameters;
54 LineSearchDirectionType type;
55 NonlinearConjugateGradientType nonlinear_conjugate_gradient_type;
56 double function_tolerance;
57 int max_lbfgs_rank;
58 bool use_approximate_eigenvalue_bfgs_scaling;
59 };
60
61 static LineSearchDirection* Create(const Options& options);
62
63 virtual ~LineSearchDirection() {}
64 virtual bool NextDirection(const LineSearchMinimizer::State& previous,
65 const LineSearchMinimizer::State& current,
66 Vector* search_direction) = 0;
67};
68
69} // namespace internal
70} // namespace ceres
71
72#endif // CERES_INTERNAL_LINE_SEARCH_DIRECTION_H_
