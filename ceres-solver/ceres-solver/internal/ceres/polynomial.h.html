43// All polynomials are assumed to be the form
44//
45// sum_{i=0}^N polynomial(i) x^{N-i}.
46//
47// and are given by a vector of coefficients of size N + 1.
48
49// Evaluate the polynomial at x using the Horner scheme.
50inline double EvaluatePolynomial(const Vector& polynomial, double x) {
51 double v = 0.0;
52 for (int i = 0; i < polynomial.size(); ++i) {
53 v = v * x + polynomial(i);
54 }
55 return v;
56}
57
58// Use the companion matrix eigenvalues to determine the roots of the
59// polynomial.
60//
61// This function returns true on success, false otherwise.
62// Failure indicates that the polynomial is invalid (of size 0) or
63// that the eigenvalues of the companion matrix could not be computed.
64// On failure, a more detailed message will be written to LOG(ERROR).
65// If real is not NULL, the real parts of the roots will be returned in it.
66// Likewise, if imaginary is not NULL, imaginary parts will be returned in it.
67bool FindPolynomialRoots(const Vector& polynomial,
68 Vector* real,
69 Vector* imaginary);
70
71// Return the derivative of the given polynomial. It is assumed that
72// the input polynomial is at least of degree zero.
73Vector DifferentiatePolynomial(const Vector& polynomial);
74
75// Find the minimum value of the polynomial in the interval [x_min,
76// x_max]. The minimum is obtained by computing all the roots of the
77// derivative of the input polynomial. All real roots within the
78// interval [x_min, x_max] are considered as well as the end points
79// x_min and x_max. Since polynomials are differentiable functions,
80// this ensures that the true minimum is found.
81void MinimizePolynomial(const Vector& polynomial,
82 double x_min,
83 double x_max,
84 double* optimal_x,
85 double* optimal_value);
86
87// Structure for storing sample values of a function.
88//
89// Clients can use this struct to communicate the value of the
90// function and or its gradient at a given point x.
91struct FunctionSample {
92 FunctionSample()
93 : x(0.0),
94 value(0.0),
95 value_is_valid(false),
96 gradient(0.0),
97 gradient_is_valid(false) {
98 }
99 std::string ToDebugString() const;
100
101 double x;
102 double value; // value = f(x)
103 bool value_is_valid;
104 double gradient; // gradient = f'(x)
105 bool gradient_is_valid;
106};
107
108// Given a set of function value and/or gradient samples, find a
109// polynomial whose value and gradients are exactly equal to the ones
110// in samples.
111//
112// Generally speaking,
113//
114// degree = # values + # gradients - 1
115//
116// Of course its possible to sample a polynomial any number of times,
117// in which case, generally speaking the spurious higher order
118// coefficients will be zero.
119Vector FindInterpolatingPolynomial(const std::vector<FunctionSample>& samples);
120
121// Interpolate the function described by samples with a polynomial,
122// and minimize it on the interval [x_min, x_max]. Depending on the
123// input samples, it is possible that the interpolation or the root
124// finding algorithms may fail due to numerical difficulties. But the
125// function is guaranteed to return its best guess of an answer, by
126// considering the samples and the end points as possible solutions.
127void MinimizeInterpolatingPolynomial(const std::vector<FunctionSample>& samples,
128 double x_min,
129 double x_max,
130 double* optimal_x,
131 double* optimal_value);
