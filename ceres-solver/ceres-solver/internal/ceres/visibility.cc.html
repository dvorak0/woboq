58void ComputeVisibility(const CompressedRowBlockStructure& block_structure,
59 const int num_eliminate_blocks,
60 vector< set<int> >* visibility) {
61 CHECK_NOTNULL(visibility);
62
63 // Clear the visibility vector and resize it to hold a
64 // vector for each camera.
65 visibility->resize(0);
66 visibility->resize(block_structure.cols.size() - num_eliminate_blocks);
67
68 for (int i = 0; i < block_structure.rows.size(); ++i) {
69 const vector<Cell>& cells = block_structure.rows[i].cells;
70 int block_id = cells[0].block_id;
71 // If the first block is not an e_block, then skip this row block.
72 if (block_id >= num_eliminate_blocks) {
73 continue;
74 }
75
76 for (int j = 1; j < cells.size(); ++j) {
77 int camera_block_id = cells[j].block_id - num_eliminate_blocks;
78 DCHECK_GE(camera_block_id, 0);
79 DCHECK_LT(camera_block_id, visibility->size());
80 (*visibility)[camera_block_id].insert(block_id);
81 }
82 }
83}

85WeightedGraph<int>* CreateSchurComplementGraph(
86 const vector<set<int> >& visibility) {
87 const time_t start_time = time(NULL);
88 // Compute the number of e_blocks/point blocks. Since the visibility
89 // set for each e_block/camera contains the set of e_blocks/points
90 // visible to it, we find the maximum across all visibility sets.
91 int num_points = 0;
92 for (int i = 0; i < visibility.size(); i++) {
93 if (visibility[i].size() > 0) {
94 num_points = max(num_points, (*visibility[i].rbegin()) + 1);
95 }
96 }
97
98 // Invert the visibility. The input is a camera->point mapping,
99 // which tells us which points are visible in which
100 // cameras. However, to compute the sparsity structure of the Schur
101 // Complement efficiently, its better to have the point->camera
102 // mapping.
103 vector<set<int> > inverse_visibility(num_points);
104 for (int i = 0; i < visibility.size(); i++) {
105 const set<int>& visibility_set = visibility[i];
106 for (set<int>::const_iterator it = visibility_set.begin();
107 it != visibility_set.end();
108 ++it) {
109 inverse_visibility[*it].insert(i);
110 }
111 }
112
113 // Map from camera pairs to number of points visible to both cameras
114 // in the pair.
115 HashMap<pair<int, int>, int > camera_pairs;
116
117 // Count the number of points visible to each camera/f_block pair.
118 for (vector<set<int> >::const_iterator it = inverse_visibility.begin();
119 it != inverse_visibility.end();
120 ++it) {
121 const set<int>& inverse_visibility_set = *it;
122 for (set<int>::const_iterator camera1 = inverse_visibility_set.begin();
123 camera1 != inverse_visibility_set.end();
124 ++camera1) {
125 set<int>::const_iterator camera2 = camera1;
126 for (++camera2; camera2 != inverse_visibility_set.end(); ++camera2) {
127 ++(camera_pairs[make_pair(*camera1, *camera2)]);
128 }
129 }
130 }
131
132 WeightedGraph<int>* graph = new WeightedGraph<int>;
133
134 // Add vertices and initialize the pairs for self edges so that self
135 // edges are guaranteed. This is needed for the Canonical views
136 // algorithm to work correctly.
137 static const double kSelfEdgeWeight = 1.0;
138 for (int i = 0; i < visibility.size(); ++i) {
139 graph->AddVertex(i);
140 graph->AddEdge(i, i, kSelfEdgeWeight);
141 }
142
143 // Add an edge for each camera pair.
144 for (HashMap<pair<int, int>, int>::const_iterator it = camera_pairs.begin();
145 it != camera_pairs.end();
146 ++it) {
147 const int camera1 = it->first.first;
148 const int camera2 = it->first.second;
149 CHECK_NE(camera1, camera2);
150
151 const int count = it->second;
152 // Static cast necessary for Windows.
153 const double weight = static_cast<double>(count) /
154 (sqrt(static_cast<double>(
155 visibility[camera1].size() * visibility[camera2].size())));
156 graph->AddEdge(camera1, camera2, weight);
157 }
158
159 VLOG(2) << "Schur complement graph time: " << (time(NULL) - start_time);
160 return graph;
161}
