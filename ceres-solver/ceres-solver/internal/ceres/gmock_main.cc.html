43// MS C++ compiler/linker has a bug on Windows (not on Windows CE), which
44// causes a link error when _tmain is defined in a static library and UNICODE
45// is enabled. For this reason instead of _tmain, main function is used on
46// Windows. See the following link to track the current status of this bug:
47// http://connect.microsoft.com/VisualStudio/feedback/ViewFeedback.aspx?FeedbackID=394464 // NOLINT
48#if GTEST_OS_WINDOWS_MOBILE
49# include <tchar.h> // NOLINT
50
51GTEST_API_ int _tmain(int argc, TCHAR** argv) {
52#else
53GTEST_API_ int main(int argc, char** argv) {
54#endif // GTEST_OS_WINDOWS_MOBILE
55 std::cout << "Running main() from gmock_main.cc\n";
56 google::InitGoogleLogging(argv[0]);
57 // Since Google Mock depends on Google Test, InitGoogleMock() is
58 // also responsible for initializing Google Test. Therefore there's
59 // no need for calling testing::InitGoogleTest() separately.
60 testing::InitGoogleMock(&argc, argv);
61 // On Windows, gtest passes additional non-gflags command line flags to
62 // death-tests, specifically --gtest_filter & --gtest_internal_run_death_test
63 // in order that these unknown (to gflags) flags do not invoke an error in
64 // gflags, InitGoogleTest() (called by InitGoogleMock()) must be called
65 // before ParseCommandLineFlags() to handle & remove them before gflags
66 // parses the remaining flags.
67 CERES_GFLAGS_NAMESPACE::ParseCommandLineFlags(&argc, &argv, true);
68 return RUN_ALL_TESTS();
69}
