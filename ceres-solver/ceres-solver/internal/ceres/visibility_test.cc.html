56TEST(VisibilityTest, SimpleMatrix) {
57 // A = [1 0 0 0 0 1
58 // 1 0 0 1 0 0
59 // 0 1 1 0 0 0
60 // 0 1 0 0 1 0]
61
62 int num_cols = 6;
63 int num_eliminate_blocks = 2;
64 CompressedRowBlockStructure bs;
65
66 // Row 1
67 {
68 bs.rows.push_back(CompressedRow());
69 CompressedRow& row = bs.rows.back();
70 row.block.size = 2;
71 row.block.position = 0;
72 row.cells.push_back(Cell(0, 0));
73 row.cells.push_back(Cell(5, 0));
74 }
75
76 // Row 2
77 {
78 bs.rows.push_back(CompressedRow());
79 CompressedRow& row = bs.rows.back();
80 row.block.size = 2;
81 row.block.position = 2;
82 row.cells.push_back(Cell(0, 1));
83 row.cells.push_back(Cell(3, 1));
84 }
85
86 // Row 3
87 {
88 bs.rows.push_back(CompressedRow());
89 CompressedRow& row = bs.rows.back();
90 row.block.size = 2;
91 row.block.position = 4;
92 row.cells.push_back(Cell(1, 2));
93 row.cells.push_back(Cell(2, 2));
94 }
95
96 // Row 4
97 {
98 bs.rows.push_back(CompressedRow());
99 CompressedRow& row = bs.rows.back();
100 row.block.size = 2;
101 row.block.position = 6;
102 row.cells.push_back(Cell(1, 3));
103 row.cells.push_back(Cell(4, 3));
104 }
105 bs.cols.resize(num_cols);
106
107 vector< set<int> > visibility;
108 ComputeVisibility(bs, num_eliminate_blocks, &visibility);
109 ASSERT_EQ(visibility.size(), num_cols - num_eliminate_blocks);
110 for (int i = 0; i < visibility.size(); ++i) {
111 ASSERT_EQ(visibility[i].size(), 1);
112 }
113
114 scoped_ptr<WeightedGraph<int> > graph(CreateSchurComplementGraph(visibility));
115 EXPECT_EQ(graph->vertices().size(), visibility.size());
116 for (int i = 0; i < visibility.size(); ++i) {
117 EXPECT_EQ(graph->VertexWeight(i), 1.0);
118 }
119
120 for (int i = 0; i < visibility.size(); ++i) {
121 for (int j = i; j < visibility.size(); ++j) {
122 double edge_weight = 0.0;
123 if ((i == 1 && j == 3) || (i == 0 && j == 2) || (i == j)) {
124 edge_weight = 1.0;
125 }
126
127 EXPECT_EQ(graph->EdgeWeight(i, j), edge_weight)
128 << "Edge: " << i << " " << j
129 << " weight: " << graph->EdgeWeight(i, j)
130 << " expected weight: " << edge_weight;
131 }
132 }
133}

136TEST(VisibilityTest, NoEBlocks) {
137 // A = [1 0 0 0 0 0
138 // 1 0 0 0 0 0
139 // 0 1 0 0 0 0
140 // 0 1 0 0 0 0]
141
142 int num_cols = 6;
143 int num_eliminate_blocks = 2;
144 CompressedRowBlockStructure bs;
145
146 // Row 1
147 {
148 bs.rows.push_back(CompressedRow());
149 CompressedRow& row = bs.rows.back();
150 row.block.size = 2;
151 row.block.position = 0;
152 row.cells.push_back(Cell(0, 0));
153 }
154
155 // Row 2
156 {
157 bs.rows.push_back(CompressedRow());
158 CompressedRow& row = bs.rows.back();
159 row.block.size = 2;
160 row.block.position = 2;
161 row.cells.push_back(Cell(0, 1));
162 }
163
164 // Row 3
165 {
166 bs.rows.push_back(CompressedRow());
167 CompressedRow& row = bs.rows.back();
168 row.block.size = 2;
169 row.block.position = 4;
170 row.cells.push_back(Cell(1, 2));
171 }
172
173 // Row 4
174 {
175 bs.rows.push_back(CompressedRow());
176 CompressedRow& row = bs.rows.back();
177 row.block.size = 2;
178 row.block.position = 6;
179 row.cells.push_back(Cell(1, 3));
180 }
181 bs.cols.resize(num_cols);
182
183 vector<set<int> > visibility;
184 ComputeVisibility(bs, num_eliminate_blocks, &visibility);
185 ASSERT_EQ(visibility.size(), num_cols - num_eliminate_blocks);
186 for (int i = 0; i < visibility.size(); ++i) {
187 ASSERT_EQ(visibility[i].size(), 0);
188 }
189
190 scoped_ptr<WeightedGraph<int> > graph(CreateSchurComplementGraph(visibility));
191 EXPECT_EQ(graph->vertices().size(), visibility.size());
192 for (int i = 0; i < visibility.size(); ++i) {
193 EXPECT_EQ(graph->VertexWeight(i), 1.0);
194 }
195
196 for (int i = 0; i < visibility.size(); ++i) {
197 for (int j = i; j < visibility.size(); ++j) {
198 double edge_weight = 0.0;
199 if (i == j) {
200 edge_weight = 1.0;
201 }
202 EXPECT_EQ(graph->EdgeWeight(i, j), edge_weight)
203 << "Edge: " << i << " " << j
204 << " weight: " << graph->EdgeWeight(i, j)
205 << " expected weight: " << edge_weight;
206 }
207 }
