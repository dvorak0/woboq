39void HouseholderTestHelper(const Vector& x) {
40 const double kTolerance = 1e-14;
41
42 // Check to ensure that H * x = ||x|| * [0 ... 0 1]'.
43 Vector v(x.rows());
44 double beta;
45 ComputeHouseholderVector(x, &v, &beta);
46 Vector result = x - beta * v * (v.transpose() * x);
47
48 Vector expected_result(x.rows());
49 expected_result.setZero();
50 expected_result(x.rows() - 1) = 1;
51 expected_result *= x.norm();
52
53 for (int i = 0; i < x.rows(); ++i) {
54 EXPECT_NEAR(expected_result[i], result[i], kTolerance);
55 }
56}
57
58TEST(HouseholderVector, ZeroPositive) {
59 Vector x(3);
60 x << 0.0, 0.0, 0.25;
61
62 HouseholderTestHelper(x);
63}
