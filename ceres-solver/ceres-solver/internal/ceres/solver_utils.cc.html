38namespace ceres {
39namespace internal {

41#define CERES_EIGEN_VERSION \
42 CERES_TO_STRING(EIGEN_WORLD_VERSION) "." \
43 CERES_TO_STRING(EIGEN_MAJOR_VERSION) "." \
44 CERES_TO_STRING(EIGEN_MINOR_VERSION)
45
46std::string VersionString() {
47 std::string value = std::string(CERES_VERSION_STRING);
48 value += "-eigen-(" + std::string(CERES_EIGEN_VERSION) + ")";
49
50#ifdef CERES_NO_LAPACK
51 value += "-no_lapack";
52#else
53 value += "-lapack";
54#endif
55
56#ifndef CERES_NO_SUITESPARSE
57 value += "-suitesparse-(" + std::string(CERES_SUITESPARSE_VERSION) + ")";
58#endif
59
60#ifndef CERES_NO_CXSPARSE
61 value += "-cxsparse-(" + std::string(CERES_CXSPARSE_VERSION) + ")";
62#endif
63
64#ifdef CERES_USE_EIGEN_SPARSE
65 value += "-eigensparse";
66#endif
67
68#ifdef CERES_RESTRUCT_SCHUR_SPECIALIZATIONS
69 value += "-no_schur_specializations";
70#endif
71
72#ifdef CERES_USE_OPENMP
73 value += "-openmp";
74#else
75 value += "-no_openmp";
76#endif
77
78#ifdef CERES_NO_CUSTOM_BLAS
79 value += "-no_custom_blas";
80#endif
81
82 return value;
83}
