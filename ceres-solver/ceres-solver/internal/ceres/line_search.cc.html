91LineSearch* LineSearch::Create(const LineSearchType line_search_type,
92 const LineSearch::Options& options,
93 string* error) {
94 LineSearch* line_search = NULL;
95 switch (line_search_type) {
96 case ceres::ARMIJO:
97 line_search = new ArmijoLineSearch(options);
98 break;
99 case ceres::WOLFE:
100 line_search = new WolfeLineSearch(options);
101 break;
102 default:
103 *error = string("Invalid line search algorithm type: ") +
104 LineSearchTypeToString(line_search_type) +
105 string(", unable to create line search.");
106 return NULL;
107 }
108 return line_search;
109}

121void LineSearchFunction::Init(const Vector& position,
122 const Vector& direction) {
123 position_ = position;
124 direction_ = direction;
125}
126
127bool LineSearchFunction::Evaluate(double x, double* f, double* g) {
128 scaled_direction_ = x * direction_;
129 if (!evaluator_->Plus(position_.data(),
130 scaled_direction_.data(),
131 evaluation_point_.data())) {
132 return false;
133 }
134
135 if (g == NULL) {
136 return (evaluator_->Evaluate(evaluation_point_.data(),
137 f, NULL, NULL, NULL) &&
138 IsFinite(*f));
139 }
140
141 if (!evaluator_->Evaluate(evaluation_point_.data(),
142 f, NULL, gradient_.data(), NULL)) {
143 return false;
144 }
145
146 *g = direction_.dot(gradient_);
147 return IsFinite(*f) && IsFinite(*g);
148}
149
150double LineSearchFunction::DirectionInfinityNorm() const {
151 return direction_.lpNorm<Eigen::Infinity>();
152}

163void LineSearchFunction::TimeStatistics(
164 double* cost_evaluation_time_in_seconds,
165 double* gradient_evaluation_time_in_seconds) const {
166 const map<string, double> evaluator_time_statistics =
167 evaluator_->TimeStatistics();
168 *cost_evaluation_time_in_seconds =
169 FindWithDefault(evaluator_time_statistics, "Evaluator::Residual", 0.0) -
170 initial_evaluator_residual_time_in_seconds;
171 // Strictly speaking this will slightly underestimate the time spent
172 // evaluating the gradient of the line search univariate cost function as it
173 // does not count the time spent performing the dot product with the direction
174 // vector. However, this will typically be small by comparison, and also
175 // allows direct subtraction of the timing information from the totals for
176 // the evaluator returned in the solver summary.
177 *gradient_evaluation_time_in_seconds =
178 FindWithDefault(evaluator_time_statistics, "Evaluator::Jacobian", 0.0) -
179 initial_evaluator_jacobian_time_in_seconds;
180}

182void LineSearch::Search(double step_size_estimate,
183 double initial_cost,
184 double initial_gradient,
185 Summary* summary) const {
186 const double start_time = WallTimeInSeconds();
187 *CHECK_NOTNULL(summary) = LineSearch::Summary();
188
189 summary->cost_evaluation_time_in_seconds = 0.0;
190 summary->gradient_evaluation_time_in_seconds = 0.0;
191 summary->polynomial_minimization_time_in_seconds = 0.0;
192
193 options().function->ResetTimeStatistics();
194 this->DoSearch(step_size_estimate, initial_cost, initial_gradient, summary);
195 options().function->
196 TimeStatistics(&summary->cost_evaluation_time_in_seconds,
197 &summary->gradient_evaluation_time_in_seconds);
198
199 summary->total_time_in_seconds = WallTimeInSeconds() - start_time;
200}

202// Returns step_size \in [min_step_size, max_step_size] which minimizes the
203// polynomial of degree defined by interpolation_type which interpolates all
204// of the provided samples with valid values.
205double LineSearch::InterpolatingPolynomialMinimizingStepSize(
206 const LineSearchInterpolationType& interpolation_type,
207 const FunctionSample& lowerbound,
208 const FunctionSample& previous,
209 const FunctionSample& current,
210 const double min_step_size,
211 const double max_step_size) const {
212 if (!current.value_is_valid ||
213 (interpolation_type == BISECTION &&
214 max_step_size <= current.x)) {
215 // Either: sample is invalid; or we are using BISECTION and contracting
216 // the step size.
217 return std::min(std::max(current.x * 0.5, min_step_size), max_step_size);
218 } else if (interpolation_type == BISECTION) {
219 CHECK_GT(max_step_size, current.x);
220 // We are expanding the search (during a Wolfe bracketing phase) using
221 // BISECTION interpolation. Using BISECTION when trying to expand is
222 // strictly speaking an oxymoron, but we define this to mean always taking
223 // the maximum step size so that the Armijo & Wolfe implementations are
224 // agnostic to the interpolation type.
225 return max_step_size;
226 }
227 // Only check if lower-bound is valid here, where it is required
228 // to avoid replicating current.value_is_valid == false
229 // behaviour in WolfeLineSearch.
230 CHECK(lowerbound.value_is_valid)
231 << std::scientific << std::setprecision(kErrorMessageNumericPrecision)
232 << "Ceres bug: lower-bound sample for interpolation is invalid, "
233 << "please contact the developers!, interpolation_type: "
234 << LineSearchInterpolationTypeToString(interpolation_type)
235 << ", lowerbound: " << lowerbound << ", previous: " << previous
236 << ", current: " << current;
237
238 // Select step size by interpolating the function and gradient values
239 // and minimizing the corresponding polynomial.
240 vector<FunctionSample> samples;
241 samples.push_back(lowerbound);
242
243 if (interpolation_type == QUADRATIC) {
244 // Two point interpolation using function values and the
245 // gradient at the lower bound.
246 samples.push_back(ValueSample(current.x, current.value));
247
248 if (previous.value_is_valid) {
249 // Three point interpolation, using function values and the
250 // gradient at the lower bound.
251 samples.push_back(ValueSample(previous.x, previous.value));
252 }
253 } else if (interpolation_type == CUBIC) {
254 // Two point interpolation using the function values and the gradients.
255 samples.push_back(current);
256
257 if (previous.value_is_valid) {
258 // Three point interpolation using the function values and
259 // the gradients.
260 samples.push_back(previous);
261 }
262 } else {
263 LOG(FATAL) << "Ceres bug: No handler for interpolation_type: "
264 << LineSearchInterpolationTypeToString(interpolation_type)
265 << ", please contact the developers!";
266 }
267
268 double step_size = 0.0, unused_min_value = 0.0;
269 MinimizeInterpolatingPolynomial(samples, min_step_size, max_step_size,
270 &step_size, &unused_min_value);
271 return step_size;
272}

277void ArmijoLineSearch::DoSearch(const double step_size_estimate,
278 const double initial_cost,
279 const double initial_gradient,
280 Summary* summary) const {
281 CHECK_GE(step_size_estimate, 0.0);
282 CHECK_GT(options().sufficient_decrease, 0.0);
283 CHECK_LT(options().sufficient_decrease, 1.0);
284 CHECK_GT(options().max_num_iterations, 0);
285 LineSearchFunction* function = options().function;
286
287 // Note initial_cost & initial_gradient are evaluated at step_size = 0,
288 // not step_size_estimate, which is our starting guess.
289 const FunctionSample initial_position =
290 ValueAndGradientSample(0.0, initial_cost, initial_gradient);
291
292 FunctionSample previous = ValueAndGradientSample(0.0, 0.0, 0.0);
293 previous.value_is_valid = false;
294
295 FunctionSample current = ValueAndGradientSample(step_size_estimate, 0.0, 0.0);
296 current.value_is_valid = false;
297
298 // As the Armijo line search algorithm always uses the initial point, for
299 // which both the function value and derivative are known, when fitting a
300 // minimizing polynomial, we can fit up to a quadratic without requiring the
301 // gradient at the current query point.
302 const bool interpolation_uses_gradient_at_current_sample =
303 options().interpolation_type == CUBIC;
304 const double descent_direction_max_norm = function->DirectionInfinityNorm();
305
306 ++summary->num_function_evaluations;
307 if (interpolation_uses_gradient_at_current_sample) {
308 ++summary->num_gradient_evaluations;
309 }
310 current.value_is_valid =
311 function->Evaluate(current.x,
312 &current.value,
313 interpolation_uses_gradient_at_current_sample
314 ? &current.gradient : NULL);
315 current.gradient_is_valid =
316 interpolation_uses_gradient_at_current_sample && current.value_is_valid;
317 while (!current.value_is_valid ||
318 current.value > (initial_cost
319 + options().sufficient_decrease
320 * initial_gradient
321 * current.x)) {
322 // If current.value_is_valid is false, we treat it as if the cost at that
323 // point is not large enough to satisfy the sufficient decrease condition.
324 ++summary->num_iterations;
325 if (summary->num_iterations >= options().max_num_iterations) {
326 summary->error =
327 StringPrintf("Line search failed: Armijo failed to find a point "
328 "satisfying the sufficient decrease condition within "
329 "specified max_num_iterations: %d.",
330 options().max_num_iterations);
331 LOG_IF(WARNING, !options().is_silent) << summary->error;
332 return;
333 }
334
335 const double polynomial_minimization_start_time = WallTimeInSeconds();
336 const double step_size =
337 this->InterpolatingPolynomialMinimizingStepSize(
338 options().interpolation_type,
339 initial_position,
340 previous,
341 current,
342 (options().max_step_contraction * current.x),
343 (options().min_step_contraction * current.x));
344 summary->polynomial_minimization_time_in_seconds +=
345 (WallTimeInSeconds() - polynomial_minimization_start_time);
346
347 if (step_size * descent_direction_max_norm < options().min_step_size) {
348 summary->error =
349 StringPrintf("Line search failed: step_size too small: %.5e "
350 "with descent_direction_max_norm: %.5e.", step_size,
351 descent_direction_max_norm);
352 LOG_IF(WARNING, !options().is_silent) << summary->error;
353 return;
354 }
355
356 previous = current;
357 current.x = step_size;
358
359 ++summary->num_function_evaluations;
360 if (interpolation_uses_gradient_at_current_sample) {
361 ++summary->num_gradient_evaluations;
362 }
363 current.value_is_valid =
364 function->Evaluate(current.x,
365 &current.value,
366 interpolation_uses_gradient_at_current_sample
367 ? &current.gradient : NULL);
368 current.gradient_is_valid =
369 interpolation_uses_gradient_at_current_sample && current.value_is_valid;
370 }
371
372 summary->optimal_step_size = current.x;
373 summary->success = true;
374}

379void WolfeLineSearch::DoSearch(const double step_size_estimate,
380 const double initial_cost,
381 const double initial_gradient,
382 Summary* summary) const {
383 // All parameters should have been validated by the Solver, but as
384 // invalid values would produce crazy nonsense, hard check them here.
385 CHECK_GE(step_size_estimate, 0.0);
386 CHECK_GT(options().sufficient_decrease, 0.0);
387 CHECK_GT(options().sufficient_curvature_decrease,
388 options().sufficient_decrease);
389 CHECK_LT(options().sufficient_curvature_decrease, 1.0);
390 CHECK_GT(options().max_step_expansion, 1.0);
391
392 // Note initial_cost & initial_gradient are evaluated at step_size = 0,
393 // not step_size_estimate, which is our starting guess.
394 const FunctionSample initial_position =
395 ValueAndGradientSample(0.0, initial_cost, initial_gradient);
396
397 bool do_zoom_search = false;
398 // Important: The high/low in bracket_high & bracket_low refer to their
399 // _function_ values, not their step sizes i.e. it is _not_ required that
400 // bracket_low.x < bracket_high.x.
401 FunctionSample solution, bracket_low, bracket_high;
402
403 // Wolfe bracketing phase: Increases step_size until either it finds a point
404 // that satisfies the (strong) Wolfe conditions, or an interval that brackets
405 // step sizes which satisfy the conditions. From Nocedal & Wright [1] p61 the
406 // interval: (step_size_{k-1}, step_size_{k}) contains step lengths satisfying
407 // the strong Wolfe conditions if one of the following conditions are met:
408 //
409 // 1. step_size_{k} violates the sufficient decrease (Armijo) condition.
410 // 2. f(step_size_{k}) >= f(step_size_{k-1}).
411 // 3. f'(step_size_{k}) >= 0.
412 //
413 // Caveat: If f(step_size_{k}) is invalid, then step_size is reduced, ignoring
414 // this special case, step_size monotonically increases during bracketing.
415 if (!this->BracketingPhase(initial_position,
416 step_size_estimate,
417 &bracket_low,
418 &bracket_high,
419 &do_zoom_search,
420 summary)) {
421 // Failed to find either a valid point, a valid bracket satisfying the Wolfe
422 // conditions, or even a step size > minimum tolerance satisfying the Armijo
423 // condition.
424 return;
425 }
426
427 if (!do_zoom_search) {
428 // Either: Bracketing phase already found a point satisfying the strong
429 // Wolfe conditions, thus no Zoom required.
430 //
431 // Or: Bracketing failed to find a valid bracket or a point satisfying the
432 // strong Wolfe conditions within max_num_iterations, or whilst searching
433 // shrank the bracket width until it was below our minimum tolerance.
434 // As these are 'artificial' constraints, and we would otherwise fail to
435 // produce a valid point when ArmijoLineSearch would succeed, we return the
436 // point with the lowest cost found thus far which satsifies the Armijo
437 // condition (but not the Wolfe conditions).
438 summary->optimal_step_size = bracket_low.x;
439 summary->success = true;
440 return;
441 }
442
443 VLOG(3) << std::scientific << std::setprecision(kErrorMessageNumericPrecision)
444 << "Starting line search zoom phase with bracket_low: "
445 << bracket_low << ", bracket_high: " << bracket_high
446 << ", bracket width: " << fabs(bracket_low.x - bracket_high.x)
447 << ", bracket abs delta cost: "
448 << fabs(bracket_low.value - bracket_high.value);
449
450 // Wolfe Zoom phase: Called when the Bracketing phase finds an interval of
451 // non-zero, finite width that should bracket step sizes which satisfy the
452 // (strong) Wolfe conditions (before finding a step size that satisfies the
453 // conditions). Zoom successively decreases the size of the interval until a
454 // step size which satisfies the Wolfe conditions is found. The interval is
455 // defined by bracket_low & bracket_high, which satisfy:
456 //
457 // 1. The interval bounded by step sizes: bracket_low.x & bracket_high.x
458 // contains step sizes that satsify the strong Wolfe conditions.
459 // 2. bracket_low.x is of all the step sizes evaluated *which satisifed the
460 // Armijo sufficient decrease condition*, the one which generated the
461 // smallest function value, i.e. bracket_low.value <
462 // f(all other steps satisfying Armijo).
463 // - Note that this does _not_ (necessarily) mean that initially
464 // bracket_low.value < bracket_high.value (although this is typical)
465 // e.g. when bracket_low = initial_position, and bracket_high is the
466 // first sample, and which does not satisfy the Armijo condition,
467 // but still has bracket_high.value < initial_position.value.
468 // 3. bracket_high is chosen after bracket_low, s.t.
469 // bracket_low.gradient * (bracket_high.x - bracket_low.x) < 0.
470 if (!this->ZoomPhase(initial_position,
471 bracket_low,
472 bracket_high,
473 &solution,
474 summary) && !solution.value_is_valid) {
475 // Failed to find a valid point (given the specified decrease parameters)
476 // within the specified bracket.
477 return;
478 }
479 // Ensure that if we ran out of iterations whilst zooming the bracket, or
480 // shrank the bracket width to < tolerance and failed to find a point which
481 // satisfies the strong Wolfe curvature condition, that we return the point
482 // amongst those found thus far, which minimizes f() and satisfies the Armijo
483 // condition.
484 solution =
485 solution.value_is_valid && solution.value <= bracket_low.value
486 ? solution : bracket_low;
487
488 summary->optimal_step_size = solution.x;
489 summary->success = true;
490}
491
492// Returns true if either:
493//
494// A termination condition satisfying the (strong) Wolfe bracketing conditions
495// is found:
496//
497// - A valid point, defined as a bracket of zero width [zoom not required].
498// - A valid bracket (of width > tolerance), [zoom required].
499//
500// Or, searching was stopped due to an 'artificial' constraint, i.e. not
501// a condition imposed / required by the underlying algorithm, but instead an
502// engineering / implementation consideration. But a step which exceeds the
503// minimum step size, and satsifies the Armijo condition was still found,
504// and should thus be used [zoom not required].
505//
506// Returns false if no step size > minimum step size was found which
507// satisfies at least the Armijo condition.
508bool WolfeLineSearch::BracketingPhase(
509 const FunctionSample& initial_position,
510 const double step_size_estimate,
511 FunctionSample* bracket_low,
512 FunctionSample* bracket_high,
513 bool* do_zoom_search,
514 Summary* summary) const {
515 LineSearchFunction* function = options().function;
516
517 FunctionSample previous = initial_position;
518 FunctionSample current = ValueAndGradientSample(step_size_estimate, 0.0, 0.0);
519 current.value_is_valid = false;
520
521 const double descent_direction_max_norm =
522 function->DirectionInfinityNorm();
523
524 *do_zoom_search = false;
525 *bracket_low = initial_position;
526
527 // As we require the gradient to evaluate the Wolfe condition, we always
528 // calculate it together with the value, irrespective of the interpolation
529 // type. As opposed to only calculating the gradient after the Armijo
530 // condition is satisifed, as the computational saving from this approach
531 // would be slight (perhaps even negative due to the extra call). Also,
532 // always calculating the value & gradient together protects against us
533 // reporting invalid solutions if the cost function returns slightly different
534 // function values when evaluated with / without gradients (due to numerical
535 // issues).
536 ++summary->num_function_evaluations;
537 ++summary->num_gradient_evaluations;
538 current.value_is_valid =
539 function->Evaluate(current.x,
540 &current.value,
541 &current.gradient);
542 current.gradient_is_valid = current.value_is_valid;
543
544 while (true) {
545 ++summary->num_iterations;
546
547 if (current.value_is_valid &&
548 (current.value > (initial_position.value
549 + options().sufficient_decrease
550 * initial_position.gradient
551 * current.x) ||
552 (previous.value_is_valid && current.value > previous.value))) {
553 // Bracket found: current step size violates Armijo sufficient decrease
554 // condition, or has stepped past an inflection point of f() relative to
555 // previous step size.
556 *do_zoom_search = true;
557 *bracket_low = previous;
558 *bracket_high = current;
559 VLOG(3) << std::scientific
560 << std::setprecision(kErrorMessageNumericPrecision)
561 << "Bracket found: current step (" << current.x
562 << ") violates Armijo sufficient condition, or has passed an "
563 << "inflection point of f() based on value.";
564 break;
565 }
566
567 if (current.value_is_valid &&
568 fabs(current.gradient) <=
569 -options().sufficient_curvature_decrease * initial_position.gradient) {
570 // Current step size satisfies the strong Wolfe conditions, and is thus a
571 // valid termination point, therefore a Zoom not required.
572 *bracket_low = current;
573 *bracket_high = current;
574 VLOG(3) << std::scientific
575 << std::setprecision(kErrorMessageNumericPrecision)
576 << "Bracketing phase found step size: " << current.x
577 << ", satisfying strong Wolfe conditions, initial_position: "
578 << initial_position << ", current: " << current;
579 break;
580
581 } else if (current.value_is_valid && current.gradient >= 0) {
582 // Bracket found: current step size has stepped past an inflection point
583 // of f(), but Armijo sufficient decrease is still satisfied and
584 // f(current) is our best minimum thus far. Remember step size
585 // monotonically increases, thus previous_step_size < current_step_size
586 // even though f(previous) > f(current).
587 *do_zoom_search = true;
588 // Note inverse ordering from first bracket case.
589 *bracket_low = current;
590 *bracket_high = previous;
591 VLOG(3) << "Bracket found: current step (" << current.x
592 << ") satisfies Armijo, but has gradient >= 0, thus have passed "
593 << "an inflection point of f().";
594 break;
595
596 } else if (current.value_is_valid &&
597 fabs(current.x - previous.x) * descent_direction_max_norm
598 < options().min_step_size) {
599 // We have shrunk the search bracket to a width less than our tolerance,
600 // and still not found either a point satisfying the strong Wolfe
601 // conditions, or a valid bracket containing such a point. Stop searching
602 // and set bracket_low to the size size amongst all those tested which
603 // minimizes f() and satisfies the Armijo condition.
604 LOG_IF(WARNING, !options().is_silent)
605 << "Line search failed: Wolfe bracketing phase shrank "
606 << "bracket width: " << fabs(current.x - previous.x)
607 << ", to < tolerance: " << options().min_step_size
608 << ", with descent_direction_max_norm: "
609 << descent_direction_max_norm << ", and failed to find "
610 << "a point satisfying the strong Wolfe conditions or a "
611 << "bracketing containing such a point. Accepting "
612 << "point found satisfying Armijo condition only, to "
613 << "allow continuation.";
614 *bracket_low = current;
615 break;
616
617 } else if (summary->num_iterations >= options().max_num_iterations) {
618 // Check num iterations bound here so that we always evaluate the
619 // max_num_iterations-th iteration against all conditions, and
620 // then perform no additional (unused) evaluations.
621 summary->error =
622 StringPrintf("Line search failed: Wolfe bracketing phase failed to "
623 "find a point satisfying strong Wolfe conditions, or a "
624 "bracket containing such a point within specified "
625 "max_num_iterations: %d", options().max_num_iterations);
626 LOG_IF(WARNING, !options().is_silent) << summary->error;
627 // Ensure that bracket_low is always set to the step size amongst all
628 // those tested which minimizes f() and satisfies the Armijo condition
629 // when we terminate due to the 'artificial' max_num_iterations condition.
630 *bracket_low =
631 current.value_is_valid && current.value < bracket_low->value
632 ? current : *bracket_low;
633 break;
634 }
635 // Either: f(current) is invalid; or, f(current) is valid, but does not
636 // satisfy the strong Wolfe conditions itself, or the conditions for
637 // being a boundary of a bracket.
638
639 // If f(current) is valid, (but meets no criteria) expand the search by
640 // increasing the step size.
641 const double max_step_size =
642 current.value_is_valid
643 ? (current.x * options().max_step_expansion) : current.x;
644
645 // We are performing 2-point interpolation only here, but the API of
646 // InterpolatingPolynomialMinimizingStepSize() allows for up to
647 // 3-point interpolation, so pad call with a sample with an invalid
648 // value that will therefore be ignored.
649 const FunctionSample unused_previous;
650 DCHECK(!unused_previous.value_is_valid);
651 // Contracts step size if f(current) is not valid.
652 const double polynomial_minimization_start_time = WallTimeInSeconds();
653 const double step_size =
654 this->InterpolatingPolynomialMinimizingStepSize(
655 options().interpolation_type,
656 previous,
657 unused_previous,
658 current,
659 previous.x,
660 max_step_size);
661 summary->polynomial_minimization_time_in_seconds +=
662 (WallTimeInSeconds() - polynomial_minimization_start_time);
663 if (step_size * descent_direction_max_norm < options().min_step_size) {
664 summary->error =
665 StringPrintf("Line search failed: step_size too small: %.5e "
666 "with descent_direction_max_norm: %.5e", step_size,
667 descent_direction_max_norm);
668 LOG_IF(WARNING, !options().is_silent) << summary->error;
669 return false;
670 }
671
672 previous = current.value_is_valid ? current : previous;
673 current.x = step_size;
674
675 ++summary->num_function_evaluations;
676 ++summary->num_gradient_evaluations;
677 current.value_is_valid =
678 function->Evaluate(current.x,
679 &current.value,
680 &current.gradient);
681 current.gradient_is_valid = current.value_is_valid;
682 }
683
684 // Ensure that even if a valid bracket was found, we will only mark a zoom
685 // as required if the bracket's width is greater than our minimum tolerance.
686 if (*do_zoom_search &&
687 fabs(bracket_high->x - bracket_low->x) * descent_direction_max_norm
688 < options().min_step_size) {
689 *do_zoom_search = false;
690 }
691
692 return true;
693}
694
695// Returns true iff solution satisfies the strong Wolfe conditions. Otherwise,
696// on return false, if we stopped searching due to the 'artificial' condition of
697// reaching max_num_iterations, solution is the step size amongst all those
698// tested, which satisfied the Armijo decrease condition and minimized f().
699bool WolfeLineSearch::ZoomPhase(const FunctionSample& initial_position,
700 FunctionSample bracket_low,
701 FunctionSample bracket_high,
702 FunctionSample* solution,
703 Summary* summary) const {
704 LineSearchFunction* function = options().function;
705
706 CHECK(bracket_low.value_is_valid && bracket_low.gradient_is_valid)
707 << std::scientific << std::setprecision(kErrorMessageNumericPrecision)
708 << "Ceres bug: f_low input to Wolfe Zoom invalid, please contact "
709 << "the developers!, initial_position: " << initial_position
710 << ", bracket_low: " << bracket_low
711 << ", bracket_high: "<< bracket_high;
712 // We do not require bracket_high.gradient_is_valid as the gradient condition
713 // for a valid bracket is only dependent upon bracket_low.gradient, and
714 // in order to minimize jacobian evaluations, bracket_high.gradient may
715 // not have been calculated (if bracket_high.value does not satisfy the
716 // Armijo sufficient decrease condition and interpolation method does not
717 // require it).
718 //
719 // We also do not require that: bracket_low.value < bracket_high.value,
720 // although this is typical. This is to deal with the case when
721 // bracket_low = initial_position, bracket_high is the first sample,
722 // and bracket_high does not satisfy the Armijo condition, but still has
723 // bracket_high.value < initial_position.value.
724 CHECK(bracket_high.value_is_valid)
725 << std::scientific << std::setprecision(kErrorMessageNumericPrecision)
726 << "Ceres bug: f_high input to Wolfe Zoom invalid, please "
727 << "contact the developers!, initial_position: " << initial_position
728 << ", bracket_low: " << bracket_low
729 << ", bracket_high: "<< bracket_high;
730
731 if (bracket_low.gradient * (bracket_high.x - bracket_low.x) >= 0) {
732 // The third condition for a valid initial bracket:
733 //
734 // 3. bracket_high is chosen after bracket_low, s.t.
735 // bracket_low.gradient * (bracket_high.x - bracket_low.x) < 0.
736 //
737 // is not satisfied. As this can happen when the users' cost function
738 // returns inconsistent gradient values relative to the function values,
739 // we do not CHECK_LT(), but we do stop processing and return an invalid
740 // value.
741 summary->error =
742 StringPrintf("Line search failed: Wolfe zoom phase passed a bracket "
743 "which does not satisfy: bracket_low.gradient * "
744 "(bracket_high.x - bracket_low.x) < 0 [%.8e !< 0] "
745 "with initial_position: %s, bracket_low: %s, bracket_high:"
746 " %s, the most likely cause of which is the cost function "
747 "returning inconsistent gradient & function values.",
748 bracket_low.gradient * (bracket_high.x - bracket_low.x),
749 initial_position.ToDebugString().c_str(),
750 bracket_low.ToDebugString().c_str(),
751 bracket_high.ToDebugString().c_str());
752 LOG_IF(WARNING, !options().is_silent) << summary->error;
753 solution->value_is_valid = false;
754 return false;
755 }
756
757 const int num_bracketing_iterations = summary->num_iterations;
758 const double descent_direction_max_norm = function->DirectionInfinityNorm();
759
760 while (true) {
761 // Set solution to bracket_low, as it is our best step size (smallest f())
762 // found thus far and satisfies the Armijo condition, even though it does
763 // not satisfy the Wolfe condition.
764 *solution = bracket_low;
765 if (summary->num_iterations >= options().max_num_iterations) {
766 summary->error =
767 StringPrintf("Line search failed: Wolfe zoom phase failed to "
768 "find a point satisfying strong Wolfe conditions "
769 "within specified max_num_iterations: %d, "
770 "(num iterations taken for bracketing: %d).",
771 options().max_num_iterations, num_bracketing_iterations);
772 LOG_IF(WARNING, !options().is_silent) << summary->error;
773 return false;
774 }
775 if (fabs(bracket_high.x - bracket_low.x) * descent_direction_max_norm
776 < options().min_step_size) {
777 // Bracket width has been reduced below tolerance, and no point satisfying
778 // the strong Wolfe conditions has been found.
779 summary->error =
780 StringPrintf("Line search failed: Wolfe zoom bracket width: %.5e "
781 "too small with descent_direction_max_norm: %.5e.",
782 fabs(bracket_high.x - bracket_low.x),
783 descent_direction_max_norm);
784 LOG_IF(WARNING, !options().is_silent) << summary->error;
785 return false;
786 }
787
788 ++summary->num_iterations;
789 // Polynomial interpolation requires inputs ordered according to step size,
790 // not f(step size).
791 const FunctionSample& lower_bound_step =
792 bracket_low.x < bracket_high.x ? bracket_low : bracket_high;
793 const FunctionSample& upper_bound_step =
794 bracket_low.x < bracket_high.x ? bracket_high : bracket_low;
795 // We are performing 2-point interpolation only here, but the API of
796 // InterpolatingPolynomialMinimizingStepSize() allows for up to
797 // 3-point interpolation, so pad call with a sample with an invalid
798 // value that will therefore be ignored.
799 const FunctionSample unused_previous;
800 DCHECK(!unused_previous.value_is_valid);
801 const double polynomial_minimization_start_time = WallTimeInSeconds();
802 solution->x =
803 this->InterpolatingPolynomialMinimizingStepSize(
804 options().interpolation_type,
805 lower_bound_step,
806 unused_previous,
807 upper_bound_step,
808 lower_bound_step.x,
809 upper_bound_step.x);
810 summary->polynomial_minimization_time_in_seconds +=
811 (WallTimeInSeconds() - polynomial_minimization_start_time);
812 // No check on magnitude of step size being too small here as it is
813 // lower-bounded by the initial bracket start point, which was valid.
814 //
815 // As we require the gradient to evaluate the Wolfe condition, we always
816 // calculate it together with the value, irrespective of the interpolation
817 // type. As opposed to only calculating the gradient after the Armijo
818 // condition is satisifed, as the computational saving from this approach
819 // would be slight (perhaps even negative due to the extra call). Also,
820 // always calculating the value & gradient together protects against us
821 // reporting invalid solutions if the cost function returns slightly
822 // different function values when evaluated with / without gradients (due
823 // to numerical issues).
824 ++summary->num_function_evaluations;
825 ++summary->num_gradient_evaluations;
826 solution->value_is_valid =
827 function->Evaluate(solution->x,
828 &solution->value,
829 &solution->gradient);
830 solution->gradient_is_valid = solution->value_is_valid;
831 if (!solution->value_is_valid) {
832 summary->error =
833 StringPrintf("Line search failed: Wolfe Zoom phase found "
834 "step_size: %.5e, for which function is invalid, "
835 "between low_step: %.5e and high_step: %.5e "
836 "at which function is valid.",
837 solution->x, bracket_low.x, bracket_high.x);
838 LOG_IF(WARNING, !options().is_silent) << summary->error;
839 return false;
840 }
841
842 VLOG(3) << "Zoom iteration: "
843 << summary->num_iterations - num_bracketing_iterations
844 << ", bracket_low: " << bracket_low
845 << ", bracket_high: " << bracket_high
846 << ", minimizing solution: " << *solution;
847
848 if ((solution->value > (initial_position.value
849 + options().sufficient_decrease
850 * initial_position.gradient
851 * solution->x)) ||
852 (solution->value >= bracket_low.value)) {
853 // Armijo sufficient decrease not satisfied, or not better
854 // than current lowest sample, use as new upper bound.
855 bracket_high = *solution;
856 continue;
857 }
858
859 // Armijo sufficient decrease satisfied, check strong Wolfe condition.
860 if (fabs(solution->gradient) <=
861 -options().sufficient_curvature_decrease * initial_position.gradient) {
862 // Found a valid termination point satisfying strong Wolfe conditions.
863 VLOG(3) << std::scientific
864 << std::setprecision(kErrorMessageNumericPrecision)
865 << "Zoom phase found step size: " << solution->x
866 << ", satisfying strong Wolfe conditions.";
867 break;
868
869 } else if (solution->gradient * (bracket_high.x - bracket_low.x) >= 0) {
870 bracket_high = bracket_low;
871 }
872
873 bracket_low = *solution;
874 }
875 // Solution contains a valid point which satisfies the strong Wolfe
876 // conditions.
877 return true;
878}
