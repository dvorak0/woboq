46SchurJacobiPreconditioner::SchurJacobiPreconditioner(
47 const CompressedRowBlockStructure& bs,
48 const Preconditioner::Options& options)
49 : options_(options) {
50 CHECK_GT(options_.elimination_groups.size(), 1);
51 CHECK_GT(options_.elimination_groups[0], 0);
52 const int num_blocks = bs.cols.size() - options_.elimination_groups[0];
53 CHECK_GT(num_blocks, 0)
54 << "Jacobian should have atleast 1 f_block for "
55 << "SCHUR_JACOBI preconditioner.";
56
57 std::vector<int> blocks(num_blocks);
58 for (int i = 0; i < num_blocks; ++i) {
59 blocks[i] = bs.cols[i + options_.elimination_groups[0]].size;
60 }
61
62 m_.reset(new BlockRandomAccessDiagonalMatrix(blocks));
63 InitEliminator(bs);
64}

69// Initialize the SchurEliminator.
70void SchurJacobiPreconditioner::InitEliminator(
71 const CompressedRowBlockStructure& bs) {
72 LinearSolver::Options eliminator_options;
73 eliminator_options.elimination_groups = options_.elimination_groups;
74 eliminator_options.num_threads = options_.num_threads;
75 eliminator_options.e_block_size = options_.e_block_size;
76 eliminator_options.f_block_size = options_.f_block_size;
77 eliminator_options.row_block_size = options_.row_block_size;
78 eliminator_.reset(SchurEliminatorBase::Create(eliminator_options));
79 const bool kFullRankETE = true;
80 eliminator_->Init(
81 eliminator_options.elimination_groups[0], kFullRankETE, &bs);
82}

84// Update the values of the preconditioner matrix and factorize it.
85bool SchurJacobiPreconditioner::UpdateImpl(const BlockSparseMatrix& A,
86 const double* D) {
87 const int num_rows = m_->num_rows();
88 CHECK_GT(num_rows, 0);
89
90 // We need a dummy rhs vector and a dummy b vector since the Schur
91 // eliminator combines the computation of the reduced camera matrix
92 // with the computation of the right hand side of that linear
93 // system.
94 //
95 // TODO(sameeragarwal): Perhaps its worth refactoring the
96 // SchurEliminator::Eliminate function to allow NULL for the rhs. As
97 // of now it does not seem to be worth the effort.
98 Vector rhs = Vector::Zero(m_->num_rows());
99 Vector b = Vector::Zero(A.num_rows());
100
101 // Compute a subset of the entries of the Schur complement.
102 eliminator_->Eliminate(&A, b.data(), D, m_.get(), rhs.data());
103 m_->Invert();
104 return true;
105}

107void SchurJacobiPreconditioner::RightMultiply(const double* x,
108 double* y) const {
109 m_->RightMultiply(x, y);
110}

112int SchurJacobiPreconditioner::num_rows() const {
