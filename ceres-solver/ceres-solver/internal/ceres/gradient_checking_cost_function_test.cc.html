59// Pick a (non-quadratic) function whose derivative are easy:
60//
61// f = exp(- a' x).
62// df = - f a.
63//
64// where 'a' is a vector of the same size as 'x'. In the block
65// version, they are both block vectors, of course.
66template<int bad_block = 1, int bad_variable = 2>
67class TestTerm : public CostFunction {
68 public:
69 // The constructor of this function needs to know the number
70 // of blocks desired, and the size of each block.
71 TestTerm(int arity, int const *dim) : arity_(arity) {
72 // Make 'arity' random vectors.
73 a_.resize(arity_);
74 for (int j = 0; j < arity_; ++j) {
75 a_[j].resize(dim[j]);
76 for (int u = 0; u < dim[j]; ++u) {
77 a_[j][u] = 2.0 * RandDouble() - 1.0;
78 }
79 }
80
81 for (int i = 0; i < arity_; i++) {
82 mutable_parameter_block_sizes()->push_back(dim[i]);
83 }
84 set_num_residuals(1);
85 }
86
87 bool Evaluate(double const* const* parameters,
88 double* residuals,
89 double** jacobians) const {
90 // Compute a . x.
91 double ax = 0;
92 for (int j = 0; j < arity_; ++j) {
93 for (int u = 0; u < parameter_block_sizes()[j]; ++u) {
94 ax += a_[j][u] * parameters[j][u];
95 }
96 }
97
98 // This is the cost, but also appears as a factor
99 // in the derivatives.
100 double f = *residuals = exp(-ax);
101
102 // Accumulate 1st order derivatives.
103 if (jacobians) {
104 for (int j = 0; j < arity_; ++j) {
105 if (jacobians[j]) {
106 for (int u = 0; u < parameter_block_sizes()[j]; ++u) {
107 // See comments before class.
108 jacobians[j][u] = - f * a_[j][u];
109
110 if (bad_block == j && bad_variable == u) {
111 // Whoopsiedoopsie! Deliberately introduce a faulty jacobian entry
112 // like what happens when users make an error in their jacobian
113 // computations. This should get detected.
114 LOG(INFO) << "Poisoning jacobian for parameter block " << j
115 << ", row 0, column " << u;
116 jacobians[j][u] += 500;
117 }
118 }
119 }
120 }
121 }
122
123 return true;
124 }
125
126 private:
127 int arity_;
128 vector<vector<double> > a_;
129};

131TEST(GradientCheckingCostFunction, ResidualsAndJacobiansArePreservedTest) {
132 srand(5);
133
134 // Test with 3 blocks of size 2, 3 and 4.
135 int const arity = 3;
136 int const dim[arity] = { 2, 3, 4 };
137
138 // Make a random set of blocks.
139 vector<double*> parameters(arity);
140 for (int j = 0; j < arity; ++j) {
141 parameters[j] = new double[dim[j]];
142 for (int u = 0; u < dim[j]; ++u) {
143 parameters[j][u] = 2.0 * RandDouble() - 1.0;
144 }
145 }
146
147 double original_residual;
148 double residual;
149 vector<double*> original_jacobians(arity);
150 vector<double*> jacobians(arity);
151
152 for (int j = 0; j < arity; ++j) {
153 // Since residual is one dimensional the jacobians have the same
154 // size as the parameter blocks.
155 jacobians[j] = new double[dim[j]];
156 original_jacobians[j] = new double[dim[j]];
157 }
158
159 const double kRelativeStepSize = 1e-6;
160 const double kRelativePrecision = 1e-4;
161
162 TestTerm<-1, -1> term(arity, dim);
163 GradientCheckingIterationCallback callback;
164 scoped_ptr<CostFunction> gradient_checking_cost_function(
165 CreateGradientCheckingCostFunction(&term, NULL,
166 kRelativeStepSize,
167 kRelativePrecision,
168 "Ignored.", &callback));
169 term.Evaluate(&parameters[0],
170 &original_residual,
171 &original_jacobians[0]);
172
173 gradient_checking_cost_function->Evaluate(&parameters[0],
174 &residual,
175 &jacobians[0]);
176 EXPECT_EQ(original_residual, residual);
177
178 for (int j = 0; j < arity; j++) {
179 for (int k = 0; k < dim[j]; ++k) {
180 EXPECT_EQ(original_jacobians[j][k], jacobians[j][k]);
181 }
182
183 delete[] parameters[j];
184 delete[] jacobians[j];
185 delete[] original_jacobians[j];
186 }
187}

189TEST(GradientCheckingCostFunction, SmokeTest) {
190 srand(5);
191
192 // Test with 3 blocks of size 2, 3 and 4.
193 int const arity = 3;
194 int const dim[arity] = { 2, 3, 4 };
195
196 // Make a random set of blocks.
197 vector<double*> parameters(arity);
198 for (int j = 0; j < arity; ++j) {
199 parameters[j] = new double[dim[j]];
200 for (int u = 0; u < dim[j]; ++u) {
201 parameters[j][u] = 2.0 * RandDouble() - 1.0;
202 }
203 }
204
205 double residual;
206 vector<double*> jacobians(arity);
207 for (int j = 0; j < arity; ++j) {
208 // Since residual is one dimensional the jacobians have the same size as the
209 // parameter blocks.
210 jacobians[j] = new double[dim[j]];
211 }
212
213 const double kRelativeStepSize = 1e-6;
214 const double kRelativePrecision = 1e-4;
215
216 // Should have one term that's bad, causing everything to get dumped.
217 LOG(INFO) << "Bad gradient";
218 {
219 TestTerm<1, 2> term(arity, dim);
220 GradientCheckingIterationCallback callback;
221 scoped_ptr<CostFunction> gradient_checking_cost_function(
222 CreateGradientCheckingCostFunction(&term, NULL,
223 kRelativeStepSize,
224 kRelativePrecision,
225 "Fuzzy banana", &callback));
226 EXPECT_TRUE(
227 gradient_checking_cost_function->Evaluate(&parameters[0], &residual,
228 &jacobians[0]));
229 EXPECT_TRUE(callback.gradient_error_detected());
230 EXPECT_TRUE(callback.error_log().find("Fuzzy banana") != std::string::npos);
231 EXPECT_TRUE(callback.error_log().find("(1,0,2) Relative error worse than")
232 != std::string::npos);
233 }
234
235 // The gradient is correct, so no errors are reported.
236 LOG(INFO) << "Good gradient";
237 {
238 TestTerm<-1, -1> term(arity, dim);
239 GradientCheckingIterationCallback callback;
240 scoped_ptr<CostFunction> gradient_checking_cost_function(
241 CreateGradientCheckingCostFunction(&term, NULL,
242 kRelativeStepSize,
243 kRelativePrecision,
244 "Fuzzy banana", &callback));
245 EXPECT_TRUE(
246 gradient_checking_cost_function->Evaluate(&parameters[0], &residual,
247 &jacobians[0]));
248 EXPECT_FALSE(callback.gradient_error_detected());
249 }
250
251 for (int j = 0; j < arity; j++) {
252 delete[] parameters[j];
253 delete[] jacobians[j];
254 }

257// The following three classes are for the purposes of defining
258// function signatures. They have dummy Evaluate functions.
259
260// Trivial cost function that accepts a single argument.
261class UnaryCostFunction : public CostFunction {
262 public:
263 UnaryCostFunction(int num_residuals, int32 parameter_block_size) {
264 set_num_residuals(num_residuals);
265 mutable_parameter_block_sizes()->push_back(parameter_block_size);
266 }
267 virtual ~UnaryCostFunction() {}
268
269 virtual bool Evaluate(double const* const* parameters,
270 double* residuals,
271 double** jacobians) const {
272 for (int i = 0; i < num_residuals(); ++i) {
273 residuals[i] = 1;
274 }
275 return true;
276 }
277};

279// Trivial cost function that accepts two arguments.
280class BinaryCostFunction: public CostFunction {
281 public:
282 BinaryCostFunction(int num_residuals,
283 int32 parameter_block1_size,
284 int32 parameter_block2_size) {
285 set_num_residuals(num_residuals);
286 mutable_parameter_block_sizes()->push_back(parameter_block1_size);
287 mutable_parameter_block_sizes()->push_back(parameter_block2_size);
288 }
289
290 virtual bool Evaluate(double const* const* parameters,
291 double* residuals,
292 double** jacobians) const {
293 for (int i = 0; i < num_residuals(); ++i) {
294 residuals[i] = 2;
295 }
296 return true;
297 }
298};

300// Trivial cost function that accepts three arguments.

313 virtual bool Evaluate(double const* const* parameters,

316 for (int i = 0; i < num_residuals(); ++i) {
317 residuals[i] = 3;

323// Verify that the two ParameterBlocks are formed from the same user
324// array and have the same LocalParameterization object.
325void ParameterBlocksAreEquivalent(const ParameterBlock* left,
326 const ParameterBlock* right) {
327 CHECK_NOTNULL(left);
328 CHECK_NOTNULL(right);
329 EXPECT_EQ(left->user_state(), right->user_state());
330 EXPECT_EQ(left->Size(), right->Size());
331 EXPECT_EQ(left->Size(), right->Size());
332 EXPECT_EQ(left->LocalSize(), right->LocalSize());
333 EXPECT_EQ(left->local_parameterization(), right->local_parameterization());
334 EXPECT_EQ(left->IsConstant(), right->IsConstant());
335}

337TEST(GradientCheckingProblemImpl, ProblemDimensionsMatch) {
338 // Parameter blocks with arbitrarily chosen initial values.
339 double x[] = {1.0, 2.0, 3.0};
340 double y[] = {4.0, 5.0, 6.0, 7.0};
341 double z[] = {8.0, 9.0, 10.0, 11.0, 12.0};
342 double w[] = {13.0, 14.0, 15.0, 16.0};
343
344 ProblemImpl problem_impl;
345 problem_impl.AddParameterBlock(x, 3);
346 problem_impl.AddParameterBlock(y, 4);
347 problem_impl.SetParameterBlockConstant(y);
348 problem_impl.AddParameterBlock(z, 5);
349 problem_impl.AddParameterBlock(w, 4, new QuaternionParameterization);
350 problem_impl.AddResidualBlock(new UnaryCostFunction(2, 3), NULL, x);
351 problem_impl.AddResidualBlock(new BinaryCostFunction(6, 5, 4) ,
352 NULL, z, y);
353 problem_impl.AddResidualBlock(new BinaryCostFunction(3, 3, 5),
354 new TrivialLoss, x, z);
355 problem_impl.AddResidualBlock(new BinaryCostFunction(7, 5, 3),
356 NULL, z, x);
357 problem_impl.AddResidualBlock(new TernaryCostFunction(1, 5, 3, 4),
358 NULL, z, x, y);
359
360 GradientCheckingIterationCallback callback;
361 scoped_ptr<ProblemImpl> gradient_checking_problem_impl(
362 CreateGradientCheckingProblemImpl(&problem_impl, 1.0, 1.0, &callback));
363
364 // The dimensions of the two problems match.
365 EXPECT_EQ(problem_impl.NumParameterBlocks(),
366 gradient_checking_problem_impl->NumParameterBlocks());
367 EXPECT_EQ(problem_impl.NumResidualBlocks(),
368 gradient_checking_problem_impl->NumResidualBlocks());
369
370 EXPECT_EQ(problem_impl.NumParameters(),
371 gradient_checking_problem_impl->NumParameters());
372 EXPECT_EQ(problem_impl.NumResiduals(),
373 gradient_checking_problem_impl->NumResiduals());
374
375 const Program& program = problem_impl.program();
376 const Program& gradient_checking_program =
377 gradient_checking_problem_impl->program();
378
379 // Since we added the ParameterBlocks and ResidualBlocks explicitly,
380 // they should be in the same order in the two programs. It is
381 // possible that may change due to implementation changes to
382 // Program. This is not expected to be the case and writing code to
383 // anticipate that possibility not worth the extra complexity in
384 // this test.
385 for (int i = 0; i < program.parameter_blocks().size(); ++i) {
386 ParameterBlocksAreEquivalent(
387 program.parameter_blocks()[i],
388 gradient_checking_program.parameter_blocks()[i]);
389 }
390
391 for (int i = 0; i < program.residual_blocks().size(); ++i) {
392 // Compare the sizes of the two ResidualBlocks.
393 const ResidualBlock* original_residual_block =
394 program.residual_blocks()[i];
395 const ResidualBlock* new_residual_block =
396 gradient_checking_program.residual_blocks()[i];
397 EXPECT_EQ(original_residual_block->NumParameterBlocks(),
398 new_residual_block->NumParameterBlocks());
399 EXPECT_EQ(original_residual_block->NumResiduals(),
400 new_residual_block->NumResiduals());
401 EXPECT_EQ(original_residual_block->NumScratchDoublesForEvaluate(),
402 new_residual_block->NumScratchDoublesForEvaluate());
403
404 // Verify that the ParameterBlocks for the two residuals are equivalent.
405 for (int j = 0; j < original_residual_block->NumParameterBlocks(); ++j) {
406 ParameterBlocksAreEquivalent(
407 original_residual_block->parameter_blocks()[j],
408 new_residual_block->parameter_blocks()[j]);
409 }
410 }
