49Evaluator::~Evaluator() {}
50
51Evaluator* Evaluator::Create(const Evaluator::Options& options,
52 Program* program,
53 std::string* error) {
54 switch (options.linear_solver_type) {
55 case DENSE_QR:
56 case DENSE_NORMAL_CHOLESKY:
57 return new ProgramEvaluator<ScratchEvaluatePreparer,
58 DenseJacobianWriter>(options,
59 program);
60 case DENSE_SCHUR:
61 case SPARSE_SCHUR:
62 case ITERATIVE_SCHUR:
63 case CGNR:
64 return new ProgramEvaluator<BlockEvaluatePreparer,
65 BlockJacobianWriter>(options,
66 program);
67 case SPARSE_NORMAL_CHOLESKY:
68 if (options.dynamic_sparsity) {
69 return new ProgramEvaluator<ScratchEvaluatePreparer,
70 DynamicCompressedRowJacobianWriter,
71 DynamicCompressedRowJacobianFinalizer>(
72 options, program);
73 } else {
74 return new ProgramEvaluator<ScratchEvaluatePreparer,
75 CompressedRowJacobianWriter>(options,
76 program);
77 }
78
79 default:
80 *error = "Invalid Linear Solver Type. Unable to create evaluator.";
81 return NULL;
82 }
83}
