31#include "gtest/gtest.h"
32#include "ceres/iteration_callback.h"
33#include "ceres/minimizer.h"
34#include "ceres/solver.h"
35
36namespace ceres {
37namespace internal {
38
39class FakeIterationCallback : public IterationCallback {
40 public:
41 virtual ~FakeIterationCallback() {}
42 virtual CallbackReturnType operator()(const IterationSummary& summary) {
43 return SOLVER_CONTINUE;
44 }
45};
46
47TEST(Minimizer, InitializationCopiesCallbacks) {
48 FakeIterationCallback callback0;
49 FakeIterationCallback callback1;
50
51 Solver::Options solver_options;
52 solver_options.callbacks.push_back(&callback0);
53 solver_options.callbacks.push_back(&callback1);
54
55 Minimizer::Options minimizer_options(solver_options);
56 ASSERT_EQ(2, minimizer_options.callbacks.size());
57
58 EXPECT_EQ(minimizer_options.callbacks[0], &callback0);
59 EXPECT_EQ(minimizer_options.callbacks[1], &callback1);
60}
61
62class AbortingIterationCallback : public IterationCallback {
63 public:
64 virtual ~AbortingIterationCallback() {}
65 virtual CallbackReturnType operator()(const IterationSummary& summary) {
66 return SOLVER_ABORT;
67 }
68};
69
70TEST(Minimizer, UserAbortUpdatesSummaryMessage) {
71 AbortingIterationCallback callback;
72 Solver::Options solver_options;
73 solver_options.callbacks.push_back(&callback);
74 Minimizer::Options minimizer_options(solver_options);
75 Solver::Summary summary;
76 Minimizer::RunCallbacks(minimizer_options, IterationSummary(), &summary);
77 EXPECT_EQ(summary.message, "User callback returned SOLVER_ABORT.");
78}
79
80class SucceedingIterationCallback : public IterationCallback {
81 public:
82 virtual ~SucceedingIterationCallback() {}
83 virtual CallbackReturnType operator()(const IterationSummary& summary) {
84 return SOLVER_TERMINATE_SUCCESSFULLY;
85 }
86};
87
88TEST(Minimizer, UserSuccessUpdatesSummaryMessage) {
89 SucceedingIterationCallback callback;
90 Solver::Options solver_options;
91 solver_options.callbacks.push_back(&callback);
92 Minimizer::Options minimizer_options(solver_options);
93 Solver::Summary summary;
94 Minimizer::RunCallbacks(minimizer_options, IterationSummary(), &summary);
95 EXPECT_EQ(summary.message,
96 "User callback returned SOLVER_TERMINATE_SUCCESSFULLY.");
97}
