42namespace ceres {
43namespace internal {
44
45class Program;
46class LinearSolver;
47
48// Given a Program, and a ParameterBlockOrdering which partitions
49// (non-exhaustively) the Hessian matrix into independent sets,
50// perform coordinate descent on the parameter blocks in the
51// ordering. The independent set structure allows for all parameter
52// blocks in the same independent set to be optimized in parallel, and
53// the order of the independent set determines the order in which the
54// parameter block groups are optimized.
55//
56// The minimizer assumes that none of the parameter blocks in the
57// program are constant.
58class CoordinateDescentMinimizer : public Minimizer {
59 public:
60 bool Init(const Program& program,
61 const ProblemImpl::ParameterMap& parameter_map,
62 const ParameterBlockOrdering& ordering,
63 std::string* error);
64
65 // Minimizer interface.
66 virtual ~CoordinateDescentMinimizer();
67 virtual void Minimize(const Minimizer::Options& options,
68 double* parameters,
69 Solver::Summary* summary);
70
71 // Verify that each group in the ordering forms an independent set.
72 static bool IsOrderingValid(const Program& program,
73 const ParameterBlockOrdering& ordering,
74 std::string* message);
75
76 // Find a recursive decomposition of the Hessian matrix as a set
77 // of independent sets of decreasing size and invert it. This
78 // seems to work better in practice, i.e., Cameras before
79 // points.
80 static ParameterBlockOrdering* CreateOrdering(const Program& program);
81
82 private:
83 void Solve(Program* program,
84 LinearSolver* linear_solver,
85 double* parameters,
86 Solver::Summary* summary);
87
88 std::vector<ParameterBlock*> parameter_blocks_;
89 std::vector<std::vector<ResidualBlock*> > residual_blocks_;
90 // The optimization is performed in rounds. In each round all the
91 // parameter blocks that form one independent set are optimized in
92 // parallel. This array, marks the boundaries of the independent
93 // sets in parameter_blocks_.
94 std::vector<int> independent_set_offsets_;
95
96 Evaluator::Options evaluator_options_;
97};
