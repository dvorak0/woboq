31// A preconditioned conjugate gradients solver
32// (ConjugateGradientsSolver) for positive semidefinite linear
33// systems.

35// We have also augmented the termination criterion used by this
36// solver to support not just residual based termination but also
37// termination based on decrease in the value of the quadratic model
38// that CG optimizes.

55bool IsZeroOrInfinity(double x) {
56 return ((x == 0.0) || (IsInfinite(x)));
57}

66LinearSolver::Summary ConjugateGradientsSolver::Solve(
67 LinearOperator* A,
68 const double* b,
69 const LinearSolver::PerSolveOptions& per_solve_options,
70 double* x) {
71 CHECK_NOTNULL(A);
72 CHECK_NOTNULL(x);
73 CHECK_NOTNULL(b);
74 CHECK_EQ(A->num_rows(), A->num_cols());
75
76 LinearSolver::Summary summary;
77 summary.termination_type = LINEAR_SOLVER_NO_CONVERGENCE;
78 summary.message = "Maximum number of iterations reached.";
79 summary.num_iterations = 0;
80
81 const int num_cols = A->num_cols();
82 VectorRef xref(x, num_cols);
83 ConstVectorRef bref(b, num_cols);
84
85 const double norm_b = bref.norm();
86 if (norm_b == 0.0) {
87 xref.setZero();
88 summary.termination_type = LINEAR_SOLVER_SUCCESS;
89 summary.message = "Convergence. |b| = 0.";
90 return summary;
91 }
92
93 Vector r(num_cols);
94 Vector p(num_cols);
95 Vector z(num_cols);
96 Vector tmp(num_cols);
97
98 const double tol_r = per_solve_options.r_tolerance * norm_b;
99
100 tmp.setZero();
101 A->RightMultiply(x, tmp.data());
102 r = bref - tmp;
103 double norm_r = r.norm();
104 if (options_.min_num_iterations == 0 && norm_r <= tol_r) {
105 summary.termination_type = LINEAR_SOLVER_SUCCESS;
106 summary.message =
107 StringPrintf("Convergence. |r| = %e <= %e.", norm_r, tol_r);
108 return summary;
109 }
110
111 double rho = 1.0;
112
113 // Initial value of the quadratic model Q = x'Ax - 2 * b'x.
114 double Q0 = -1.0 * xref.dot(bref + r);
115
116 for (summary.num_iterations = 1;; ++summary.num_iterations) {
117 // Apply preconditioner
118 if (per_solve_options.preconditioner != NULL) {
119 z.setZero();
120 per_solve_options.preconditioner->RightMultiply(r.data(), z.data());
121 } else {
122 z = r;
123 }
124
125 double last_rho = rho;
126 rho = r.dot(z);
127 if (IsZeroOrInfinity(rho)) {
128 summary.termination_type = LINEAR_SOLVER_FAILURE;
129 summary.message = StringPrintf("Numerical failure. rho = r'z = %e.", rho);
130 break;
131 }
132
133 if (summary.num_iterations == 1) {
134 p = z;
135 } else {
136 double beta = rho / last_rho;
137 if (IsZeroOrInfinity(beta)) {
138 summary.termination_type = LINEAR_SOLVER_FAILURE;
139 summary.message = StringPrintf(
140 "Numerical failure. beta = rho_n / rho_{n-1} = %e, "
141 "rho_n = %e, rho_{n-1} = %e", beta, rho, last_rho);
142 break;
143 }
144 p = z + beta * p;
145 }
146
147 Vector& q = z;
148 q.setZero();
149 A->RightMultiply(p.data(), q.data());
150 const double pq = p.dot(q);
151 if ((pq <= 0) || IsInfinite(pq)) {
152 summary.termination_type = LINEAR_SOLVER_NO_CONVERGENCE;
153 summary.message = StringPrintf(
154 "Matrix is indefinite, no more progress can be made. "
155 "p'q = %e. |p| = %e, |q| = %e",
156 pq, p.norm(), q.norm());
157 break;
158 }
159
160 const double alpha = rho / pq;
161 if (IsInfinite(alpha)) {
162 summary.termination_type = LINEAR_SOLVER_FAILURE;
163 summary.message =
164 StringPrintf("Numerical failure. alpha = rho / pq = %e, "
165 "rho = %e, pq = %e.", alpha, rho, pq);
166 break;
167 }
168
169 xref = xref + alpha * p;
170
171 // Ideally we would just use the update r = r - alpha*q to keep
172 // track of the residual vector. However this estimate tends to
173 // drift over time due to round off errors. Thus every
174 // residual_reset_period iterations, we calculate the residual as
175 // r = b - Ax. We do not do this every iteration because this
176 // requires an additional matrix vector multiply which would
177 // double the complexity of the CG algorithm.
178 if (summary.num_iterations % options_.residual_reset_period == 0) {
179 tmp.setZero();
180 A->RightMultiply(x, tmp.data());
181 r = bref - tmp;
182 } else {
183 r = r - alpha * q;
184 }
185
186 // Quadratic model based termination.
187 // Q1 = x'Ax - 2 * b' x.
188 const double Q1 = -1.0 * xref.dot(bref + r);
189
190 // For PSD matrices A, let
191 //
192 // Q(x) = x'Ax - 2b'x
193 //
194 // be the cost of the quadratic function defined by A and b. Then,
195 // the solver terminates at iteration i if
196 //
197 // i * (Q(x_i) - Q(x_i-1)) / Q(x_i) < q_tolerance.
198 //
199 // This termination criterion is more useful when using CG to
200 // solve the Newton step. This particular convergence test comes
201 // from Stephen Nash's work on truncated Newton
202 // methods. References:
203 //
204 // 1. Stephen G. Nash & Ariela Sofer, Assessing A Search
205 // Direction Within A Truncated Newton Method, Operation
206 // Research Letters 9(1990) 219-221.
207 //
208 // 2. Stephen G. Nash, A Survey of Truncated Newton Methods,
209 // Journal of Computational and Applied Mathematics,
210 // 124(1-2), 45-59, 2000.
211 //
212 const double zeta = summary.num_iterations * (Q1 - Q0) / Q1;
213 if (zeta < per_solve_options.q_tolerance &&
214 summary.num_iterations >= options_.min_num_iterations) {
215 summary.termination_type = LINEAR_SOLVER_SUCCESS;
216 summary.message =
217 StringPrintf("Iteration: %d Convergence: zeta = %e < %e. |r| = %e",
218 summary.num_iterations,
219 zeta,
220 per_solve_options.q_tolerance,
221 r.norm());
222 break;
223 }
224 Q0 = Q1;
225
226 // Residual based termination.
227 norm_r = r. norm();
228 if (norm_r <= tol_r &&
229 summary.num_iterations >= options_.min_num_iterations) {
230 summary.termination_type = LINEAR_SOLVER_SUCCESS;
231 summary.message =
232 StringPrintf("Iteration: %d Convergence. |r| = %e <= %e.",
233 summary.num_iterations,
234 norm_r,
235 tol_r);
236 break;
237 }
238
239 if (summary.num_iterations >= options_.max_num_iterations) {
240 break;
241 }
242 }
243
244 return summary;
245}
