65LinearSolver::Summary IterativeSchurComplementSolver::SolveImpl(
66 BlockSparseMatrix* A,
67 const double* b,
68 const LinearSolver::PerSolveOptions& per_solve_options,
69 double* x) {
70 EventLogger event_logger("IterativeSchurComplementSolver::Solve");
71
72 CHECK_NOTNULL(A->block_structure());
73 const int num_eliminate_blocks = options_.elimination_groups[0];
74 // Initialize a ImplicitSchurComplement object.
75 if (schur_complement_ == NULL) {
76 DetectStructure(*(A->block_structure()),
77 num_eliminate_blocks,
78 &options_.row_block_size,
79 &options_.e_block_size,
80 &options_.f_block_size);
81 schur_complement_.reset(new ImplicitSchurComplement(options_));
82 }
83 schur_complement_->Init(*A, per_solve_options.D, b);
84
85 const int num_schur_complement_blocks =
86 A->block_structure()->cols.size() - num_eliminate_blocks;
87 if (num_schur_complement_blocks == 0) {
88 VLOG(2) << "No parameter blocks left in the schur complement.";
89 LinearSolver::Summary cg_summary;
90 cg_summary.num_iterations = 0;
91 cg_summary.termination_type = LINEAR_SOLVER_SUCCESS;
92 schur_complement_->BackSubstitute(NULL, x);
93 return cg_summary;
94 }
95
96 // Initialize the solution to the Schur complement system to zero.
97 reduced_linear_system_solution_.resize(schur_complement_->num_rows());
98 reduced_linear_system_solution_.setZero();
99
100 // Instantiate a conjugate gradient solver that runs on the Schur
101 // complement matrix with the block diagonal of the matrix F'F as
102 // the preconditioner.
103 LinearSolver::Options cg_options;
104 cg_options.min_num_iterations = options_.min_num_iterations;
105 cg_options.max_num_iterations = options_.max_num_iterations;
106 ConjugateGradientsSolver cg_solver(cg_options);
107 LinearSolver::PerSolveOptions cg_per_solve_options;
108
109 cg_per_solve_options.r_tolerance = per_solve_options.r_tolerance;
110 cg_per_solve_options.q_tolerance = per_solve_options.q_tolerance;
111
112 Preconditioner::Options preconditioner_options;
113 preconditioner_options.type = options_.preconditioner_type;
114 preconditioner_options.visibility_clustering_type =
115 options_.visibility_clustering_type;
116 preconditioner_options.sparse_linear_algebra_library_type =
117 options_.sparse_linear_algebra_library_type;
118 preconditioner_options.num_threads = options_.num_threads;
119 preconditioner_options.row_block_size = options_.row_block_size;
120 preconditioner_options.e_block_size = options_.e_block_size;
121 preconditioner_options.f_block_size = options_.f_block_size;
122 preconditioner_options.elimination_groups = options_.elimination_groups;
123
124 switch (options_.preconditioner_type) {
125 case IDENTITY:
126 break;
127 case JACOBI:
128 preconditioner_.reset(
129 new SparseMatrixPreconditionerWrapper(
130 schur_complement_->block_diagonal_FtF_inverse()));
131 break;
132 case SCHUR_JACOBI:
133 if (preconditioner_.get() == NULL) {
134 preconditioner_.reset(
135 new SchurJacobiPreconditioner(*A->block_structure(),
136 preconditioner_options));
137 }
138 break;
139 case CLUSTER_JACOBI:
140 case CLUSTER_TRIDIAGONAL:
141 if (preconditioner_.get() == NULL) {
142 preconditioner_.reset(
143 new VisibilityBasedPreconditioner(*A->block_structure(),
144 preconditioner_options));
145 }
146 break;
147 default:
148 LOG(FATAL) << "Unknown Preconditioner Type";
149 }
150
151 bool preconditioner_update_was_successful = true;
152 if (preconditioner_.get() != NULL) {
153 preconditioner_update_was_successful =
154 preconditioner_->Update(*A, per_solve_options.D);
155 cg_per_solve_options.preconditioner = preconditioner_.get();
156 }
157 event_logger.AddEvent("Setup");
158
159 LinearSolver::Summary cg_summary;
160 cg_summary.num_iterations = 0;
161 cg_summary.termination_type = LINEAR_SOLVER_FAILURE;
162
163 // TODO(sameeragarwal): Refactor preconditioners to return a more
164 // sane message.
165 cg_summary.message = "Preconditioner update failed.";
166 if (preconditioner_update_was_successful) {
167 cg_summary = cg_solver.Solve(schur_complement_.get(),
168 schur_complement_->rhs().data(),
169 cg_per_solve_options,
170 reduced_linear_system_solution_.data());
171 if (cg_summary.termination_type != LINEAR_SOLVER_FAILURE &&
172 cg_summary.termination_type != LINEAR_SOLVER_FATAL_ERROR) {
173 schur_complement_->BackSubstitute(
174 reduced_linear_system_solution_.data(), x);
175 }
176 }
177 event_logger.AddEvent("Solve");
178 return cg_summary;
179}
