31// Base classes for access to an linear operator.
32
33#ifndef CERES_INTERNAL_LINEAR_OPERATOR_H_
34#define CERES_INTERNAL_LINEAR_OPERATOR_H_
35
36#include "ceres/types.h"
37
38namespace ceres {
39namespace internal {
40
41// This is an abstract base class for linear operators. It supports
42// access to size information and left and right multiply operators.
43class LinearOperator {
44 public:
45 virtual ~LinearOperator();
46
47 // y = y + Ax;
48 virtual void RightMultiply(const double* x, double* y) const = 0;
49 // y = y + A'x;
50 virtual void LeftMultiply(const double* x, double* y) const = 0;
51
52 virtual int num_rows() const = 0;
53 virtual int num_cols() const = 0;
54};
55
56} // namespace internal
57} // namespace ceres
58
59#endif // CERES_INTERNAL_LINEAR_OPERATOR_H_
