37#include "ceres/internal/port.h"

39namespace ceres {
40
41inline void SetRandomState(int state) {
42 srand(state);
43}
44
45inline int Uniform(int n) {
46 return rand() % n;
47}
48
49inline double RandDouble() {
50 double r = static_cast<double>(rand());
51 return r / RAND_MAX;
52}
53
54// Box-Muller algorithm for normal random number generation.
55// http://en.wikipedia.org/wiki/Box-Muller_transform
56inline double RandNormal() {
57 double x1, x2, w;
58 do {
59 x1 = 2.0 * RandDouble() - 1.0;
60 x2 = 2.0 * RandDouble() - 1.0;
61 w = x1 * x1 + x2 * x2;
62 } while ( w >= 1.0 || w == 0.0 );
63
64 w = sqrt((-2.0 * log(w)) / w);
65 return x1 * w;
66}
