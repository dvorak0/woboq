48class SchurComplementSolverTest : public ::testing::Test {
49 protected:
50 void SetUpFromProblemId(int problem_id) {
51 scoped_ptr<LinearLeastSquaresProblem> problem(
52 CreateLinearLeastSquaresProblemFromId(problem_id));
53
54 CHECK_NOTNULL(problem.get());
55 A.reset(down_cast<BlockSparseMatrix*>(problem->A.release()));
56 b.reset(problem->b.release());
57 D.reset(problem->D.release());
58
59 num_cols = A->num_cols();
60 num_rows = A->num_rows();
61 num_eliminate_blocks = problem->num_eliminate_blocks;
62
63 x.resize(num_cols);
64 sol.resize(num_cols);
65 sol_d.resize(num_cols);
66
67 LinearSolver::Options options;
68 options.type = DENSE_QR;
69
70 scoped_ptr<LinearSolver> qr(LinearSolver::Create(options));
71
72 TripletSparseMatrix triplet_A(A->num_rows(),
73 A->num_cols(),
74 A->num_nonzeros());
75 A->ToTripletSparseMatrix(&triplet_A);
76
77 // Gold standard solutions using dense QR factorization.
78 DenseSparseMatrix dense_A(triplet_A);
79 qr->Solve(&dense_A, b.get(), LinearSolver::PerSolveOptions(), sol.data());
80
81 // Gold standard solution with appended diagonal.
82 LinearSolver::PerSolveOptions per_solve_options;
83 per_solve_options.D = D.get();
84 qr->Solve(&dense_A, b.get(), per_solve_options, sol_d.data());
85 }
86
87 void ComputeAndCompareSolutions(
88 int problem_id,
89 bool regularization,
90 ceres::LinearSolverType linear_solver_type,
91 ceres::DenseLinearAlgebraLibraryType dense_linear_algebra_library_type,
92 ceres::SparseLinearAlgebraLibraryType sparse_linear_algebra_library_type,
93 bool use_postordering) {
94 SetUpFromProblemId(problem_id);
95 LinearSolver::Options options;
96 options.elimination_groups.push_back(num_eliminate_blocks);
97 options.elimination_groups.push_back(
98 A->block_structure()->cols.size() - num_eliminate_blocks);
99 options.type = linear_solver_type;
100 options.dense_linear_algebra_library_type =
101 dense_linear_algebra_library_type;
102 options.sparse_linear_algebra_library_type =
103 sparse_linear_algebra_library_type;
104 options.use_postordering = use_postordering;
105 DetectStructure(*A->block_structure(),
106 num_eliminate_blocks,
107 &options.row_block_size,
108 &options.e_block_size,
109 &options.f_block_size);
110
111 scoped_ptr<LinearSolver> solver(LinearSolver::Create(options));
112
113 LinearSolver::PerSolveOptions per_solve_options;
114 LinearSolver::Summary summary;
115 if (regularization) {
116 per_solve_options.D = D.get();
117 }
118
119 summary = solver->Solve(A.get(), b.get(), per_solve_options, x.data());
120 EXPECT_EQ(summary.termination_type, LINEAR_SOLVER_SUCCESS);
121
122 if (regularization) {
123 ASSERT_NEAR((sol_d - x).norm() / num_cols, 0, 1e-10)
124 << "Expected solution: " << sol_d.transpose()
125 << " Actual solution: " << x.transpose();
126 } else {
127 ASSERT_NEAR((sol - x).norm() / num_cols, 0, 1e-10)
128 << "Expected solution: " << sol.transpose()
129 << " Actual solution: " << x.transpose();
130 }
131 }
132
133 int num_rows;
134 int num_cols;
135 int num_eliminate_blocks;
136
137 scoped_ptr<BlockSparseMatrix> A;
138 scoped_array<double> b;
139 scoped_array<double> D;
140 Vector x;
141 Vector sol;
142 Vector sol_d;
143};
