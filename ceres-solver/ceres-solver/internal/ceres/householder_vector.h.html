31#ifndef CERES_PUBLIC_HOUSEHOLDER_VECTOR_H_
32#define CERES_PUBLIC_HOUSEHOLDER_VECTOR_H_
33
34#include "Eigen/Core"
35#include "glog/logging.h"
36
37namespace ceres {
38namespace internal {
39
40// Algorithm 5.1.1 from 'Matrix Computations' by Golub et al. (Johns Hopkins
41// Studies in Mathematical Sciences) but using the nth element of the input
42// vector as pivot instead of first. This computes the vector v with v(n) = 1
43// and beta such that H = I - beta * v * v^T is orthogonal and
44// H * x = ||x||_2 * e_n.
45template <typename Scalar>
46void ComputeHouseholderVector(const Eigen::Matrix<Scalar, Eigen::Dynamic, 1>& x,
47 Eigen::Matrix<Scalar, Eigen::Dynamic, 1>* v,
48 Scalar* beta) {
49 CHECK_NOTNULL(beta);
50 CHECK_NOTNULL(v);
51 CHECK_GT(x.rows(), 1);
52 CHECK_EQ(x.rows(), v->rows());
53
54 Scalar sigma = x.head(x.rows() - 1).squaredNorm();
55 *v = x;
56 (*v)(v->rows() - 1) = Scalar(1.0);
57
58 *beta = Scalar(0.0);
59 const Scalar& x_pivot = x(x.rows() - 1);
60
61 if (sigma <= Scalar(std::numeric_limits<double>::epsilon())) {
62 if (x_pivot < Scalar(0.0)) {
63 *beta = Scalar(2.0);
64 }
65 return;
66 }
67
68 const Scalar mu = sqrt(x_pivot * x_pivot + sigma);
69 Scalar v_pivot = Scalar(1.0);
70
71 if (x_pivot <= Scalar(0.0)) {
72 v_pivot = x_pivot - mu;
73 } else {
74 v_pivot = -sigma / (x_pivot + mu);
75 }
76
77 *beta = Scalar(2.0) * v_pivot * v_pivot / (sigma + v_pivot * v_pivot);
78
79 v->head(v->rows() - 1) /= v_pivot;
80}
