42class BlockSparseMatrix;
43struct CompressedRowBlockStructure;

45// A block Jacobi preconditioner. This is intended for use with
46// conjugate gradients, or other iterative symmetric solvers. To use
47// the preconditioner, create one by passing a BlockSparseMatrix "A"
48// to the constructor. This fixes the sparsity pattern to the pattern
49// of the matrix A^TA.
50//
51// Before each use of the preconditioner in a solve with conjugate gradients,
52// update the matrix by running Update(A, D). The values of the matrix A are
53// inspected to construct the preconditioner. The vector D is applied as the
54// D^TD diagonal term.
55class BlockJacobiPreconditioner : public BlockSparseMatrixPreconditioner {
56 public:
57 // A must remain valid while the BlockJacobiPreconditioner is.
58 explicit BlockJacobiPreconditioner(const BlockSparseMatrix& A);
59 virtual ~BlockJacobiPreconditioner();
60
61 // Preconditioner interface
62 virtual void RightMultiply(const double* x, double* y) const;
63 virtual int num_rows() const { return m_->num_rows(); }
64 virtual int num_cols() const { return m_->num_rows(); }
65
66 const BlockRandomAccessDiagonalMatrix& matrix() const { return *m_; }
67 private:
68 virtual bool UpdateImpl(const BlockSparseMatrix& A, const double* D);
69
70 scoped_ptr<BlockRandomAccessDiagonalMatrix> m_;
71};
