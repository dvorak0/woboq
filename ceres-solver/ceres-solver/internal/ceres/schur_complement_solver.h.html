59class BlockSparseMatrix;
60
61// Base class for Schur complement based linear least squares
62// solvers. It assumes that the input linear system Ax = b can be
63// partitioned into
64//
65// E y + F z = b
66//
67// Where x = [y;z] is a partition of the variables. The paritioning
68// of the variables is such that, E'E is a block diagonal
69// matrix. Further, the rows of A are ordered so that for every
70// variable block in y, all the rows containing that variable block
71// occur as a vertically contiguous block. i.e the matrix A looks like
72//
73// E F
74// A = [ y1 0 0 0 | z1 0 0 0 z5]
75// [ y1 0 0 0 | z1 z2 0 0 0]
76// [ 0 y2 0 0 | 0 0 z3 0 0]
77// [ 0 0 y3 0 | z1 z2 z3 z4 z5]
78// [ 0 0 y3 0 | z1 0 0 0 z5]
79// [ 0 0 0 y4 | 0 0 0 0 z5]
80// [ 0 0 0 y4 | 0 z2 0 0 0]
81// [ 0 0 0 y4 | 0 0 0 0 0]
82// [ 0 0 0 0 | z1 0 0 0 0]
83// [ 0 0 0 0 | 0 0 z3 z4 z5]
84//
85// This structure should be reflected in the corresponding
86// CompressedRowBlockStructure object associated with A. The linear
87// system Ax = b should either be well posed or the array D below
88// should be non-null and the diagonal matrix corresponding to it
89// should be non-singular.
90//
91// SchurComplementSolver has two sub-classes.
92//
93// DenseSchurComplementSolver: For problems where the Schur complement
94// matrix is small and dense, or if CHOLMOD/SuiteSparse is not
95// installed. For structure from motion problems, this is solver can
96// be used for problems with upto a few hundred cameras.
97//
98// SparseSchurComplementSolver: For problems where the Schur
99// complement matrix is large and sparse. It requires that
100// CHOLMOD/SuiteSparse be installed, as it uses CHOLMOD to find a
101// sparse Cholesky factorization of the Schur complement. This solver
102// can be used for solving structure from motion problems with tens of
103// thousands of cameras, though depending on the exact sparsity
104// structure, it maybe better to use an iterative solver.
105//
106// The two solvers can be instantiated by calling
107// LinearSolver::CreateLinearSolver with LinearSolver::Options::type
108// set to DENSE_SCHUR and SPARSE_SCHUR
109// respectively. LinearSolver::Options::elimination_groups[0] should be
110// at least 1.
111class SchurComplementSolver : public BlockSparseMatrixSolver {
112 public:
113 explicit SchurComplementSolver(const LinearSolver::Options& options)
114 : options_(options) {
115 CHECK_GT(options.elimination_groups.size(), 1);
116 CHECK_GT(options.elimination_groups[0], 0);
117 }
118
119 // LinearSolver methods
120 virtual ~SchurComplementSolver() {}
121 virtual LinearSolver::Summary SolveImpl(
122 BlockSparseMatrix* A,
123 const double* b,
124 const LinearSolver::PerSolveOptions& per_solve_options,
125 double* x);
126
127 protected:
128 const LinearSolver::Options& options() const { return options_; }
129
130 const BlockRandomAccessMatrix* lhs() const { return lhs_.get(); }
131 void set_lhs(BlockRandomAccessMatrix* lhs) { lhs_.reset(lhs); }
132 const double* rhs() const { return rhs_.get(); }
133 void set_rhs(double* rhs) { rhs_.reset(rhs); }
134
135 private:
136 virtual void InitStorage(const CompressedRowBlockStructure* bs) = 0;
137 virtual LinearSolver::Summary SolveReducedLinearSystem(
138 const LinearSolver::PerSolveOptions& per_solve_options,
139 double* solution) = 0;
140
141 LinearSolver::Options options_;
142
143 scoped_ptr<SchurEliminatorBase> eliminator_;
144 scoped_ptr<BlockRandomAccessMatrix> lhs_;
145 scoped_array<double> rhs_;
146
147 CERES_DISALLOW_COPY_AND_ASSIGN(SchurComplementSolver);
148};

150// Dense Cholesky factorization based solver.
151class DenseSchurComplementSolver : public SchurComplementSolver {
152 public:
153 explicit DenseSchurComplementSolver(const LinearSolver::Options& options)
154 : SchurComplementSolver(options) {}
155 virtual ~DenseSchurComplementSolver() {}
156
157 private:
158 virtual void InitStorage(const CompressedRowBlockStructure* bs);
159 virtual LinearSolver::Summary SolveReducedLinearSystem(
160 const LinearSolver::PerSolveOptions& per_solve_options,
161 double* solution);
162
163 CERES_DISALLOW_COPY_AND_ASSIGN(DenseSchurComplementSolver);
164};

166// Sparse Cholesky factorization based solver.
167class SparseSchurComplementSolver : public SchurComplementSolver {
168 public:
169 explicit SparseSchurComplementSolver(const LinearSolver::Options& options);
170 virtual ~SparseSchurComplementSolver();
171
172 private:
173 virtual void InitStorage(const CompressedRowBlockStructure* bs);
174 virtual LinearSolver::Summary SolveReducedLinearSystem(
175 const LinearSolver::PerSolveOptions& per_solve_options,
176 double* solution);
177 LinearSolver::Summary SolveReducedLinearSystemUsingSuiteSparse(
178 const LinearSolver::PerSolveOptions& per_solve_options,
179 double* solution);
180 LinearSolver::Summary SolveReducedLinearSystemUsingCXSparse(
181 const LinearSolver::PerSolveOptions& per_solve_options,
182 double* solution);
183 LinearSolver::Summary SolveReducedLinearSystemUsingEigen(
184 const LinearSolver::PerSolveOptions& per_solve_options,
185 double* solution);
186 LinearSolver::Summary SolveReducedLinearSystemUsingConjugateGradients(
187 const LinearSolver::PerSolveOptions& per_solve_options,
188 double* solution);
189
190 // Size of the blocks in the Schur complement.
191 std::vector<int> blocks_;
192
193 SuiteSparse ss_;
194 // Symbolic factorization of the reduced linear system. Precomputed
195 // once and reused in subsequent calls.
196 cholmod_factor* factor_;
197
198 CXSparse cxsparse_;
199 // Cached factorization
200 cs_dis* cxsparse_factor_;
201
202#ifdef CERES_USE_EIGEN_SPARSE
203
204 // The preprocessor gymnastics here are dealing with the fact that
205 // before version 3.2.2, Eigen did not support a third template
206 // parameter to specify the ordering.
207#if EIGEN_VERSION_AT_LEAST(3,2,2)
208 typedef Eigen::SimplicialLDLT<Eigen::SparseMatrix<double>, Eigen::Lower,
209 Eigen::NaturalOrdering<int> >
210 SimplicialLDLT;
211#else
212 typedef Eigen::SimplicialLDLT<Eigen::SparseMatrix<double>, Eigen::Lower>
213 SimplicialLDLT;
214#endif
215
216 scoped_ptr<SimplicialLDLT> simplicial_ldlt_;
217#endif
218
219 scoped_ptr<BlockRandomAccessDiagonalMatrix> preconditioner_;
220 CERES_DISALLOW_COPY_AND_ASSIGN(SparseSchurComplementSolver);
221};
