53class UnaryCostFunction: public CostFunction {
54 public:
55 UnaryCostFunction(const int num_residuals,
56 const int32 parameter_block_size,
57 const double* jacobian)
58 : jacobian_(jacobian, jacobian + num_residuals * parameter_block_size) {
59 set_num_residuals(num_residuals);
60 mutable_parameter_block_sizes()->push_back(parameter_block_size);
61 }
62
63 virtual bool Evaluate(double const* const* parameters,
64 double* residuals,
65 double** jacobians) const {
66 for (int i = 0; i < num_residuals(); ++i) {
67 residuals[i] = 1;
68 }
69
70 if (jacobians == NULL) {
71 return true;
72 }
73
74 if (jacobians[0] != NULL) {
75 copy(jacobian_.begin(), jacobian_.end(), jacobians[0]);
76 }
77
78 return true;
79 }
80
81 private:
82 vector<double> jacobian_;
83};

86class BinaryCostFunction: public CostFunction {
87 public:
88 BinaryCostFunction(const int num_residuals,
89 const int32 parameter_block1_size,
90 const int32 parameter_block2_size,
91 const double* jacobian1,
92 const double* jacobian2)
93 : jacobian1_(jacobian1,
94 jacobian1 + num_residuals * parameter_block1_size),
95 jacobian2_(jacobian2,
96 jacobian2 + num_residuals * parameter_block2_size) {
97 set_num_residuals(num_residuals);
98 mutable_parameter_block_sizes()->push_back(parameter_block1_size);
99 mutable_parameter_block_sizes()->push_back(parameter_block2_size);
100 }
101
102 virtual bool Evaluate(double const* const* parameters,
103 double* residuals,
104 double** jacobians) const {
105 for (int i = 0; i < num_residuals(); ++i) {
106 residuals[i] = 2;
107 }
108
109 if (jacobians == NULL) {
110 return true;
111 }
112
113 if (jacobians[0] != NULL) {
114 copy(jacobian1_.begin(), jacobian1_.end(), jacobians[0]);
115 }
116
117 if (jacobians[1] != NULL) {
118 copy(jacobian2_.begin(), jacobian2_.end(), jacobians[1]);
119 }
120
121 return true;
122 }
123
124 private:
125 vector<double> jacobian1_;
126 vector<double> jacobian2_;
127};

129// x_plus_delta = delta * x;
130class PolynomialParameterization : public LocalParameterization {
131 public:
132 virtual ~PolynomialParameterization() {}
133
134 virtual bool Plus(const double* x,
135 const double* delta,
136 double* x_plus_delta) const {
137 x_plus_delta[0] = delta[0] * x[0];
138 x_plus_delta[1] = delta[0] * x[1];
139 return true;
140 }
141
142 virtual bool ComputeJacobian(const double* x, double* jacobian) const {
143 jacobian[0] = x[0];
144 jacobian[1] = x[1];
145 return true;
146 }
147
148 virtual int GlobalSize() const { return 2; }
149 virtual int LocalSize() const { return 1; }
150};

152TEST(CovarianceImpl, ComputeCovarianceSparsity) {
153 double parameters[10];
154
155 double* block1 = parameters;
156 double* block2 = block1 + 1;
157 double* block3 = block2 + 2;
158 double* block4 = block3 + 3;
159
160 ProblemImpl problem;
161
162 // Add in random order
163 Vector junk_jacobian = Vector::Zero(10);
164 problem.AddResidualBlock(
165 new UnaryCostFunction(1, 1, junk_jacobian.data()), NULL, block1);
166 problem.AddResidualBlock(
167 new UnaryCostFunction(1, 4, junk_jacobian.data()), NULL, block4);
168 problem.AddResidualBlock(
169 new UnaryCostFunction(1, 3, junk_jacobian.data()), NULL, block3);
170 problem.AddResidualBlock(
171 new UnaryCostFunction(1, 2, junk_jacobian.data()), NULL, block2);
172
173 // Sparsity pattern
174 //
175 // Note that the problem structure does not imply this sparsity
176 // pattern since all the residual blocks are unary. But the
177 // ComputeCovarianceSparsity function in its current incarnation
178 // does not pay attention to this fact and only looks at the
179 // parameter block pairs that the user provides.
180 //
181 // X . . . . . X X X X
182 // . X X X X X . . . .
183 // . X X X X X . . . .
184 // . . . X X X . . . .
185 // . . . X X X . . . .
186 // . . . X X X . . . .
187 // . . . . . . X X X X
188 // . . . . . . X X X X
189 // . . . . . . X X X X
190 // . . . . . . X X X X
191
192 int expected_rows[] = {0, 5, 10, 15, 18, 21, 24, 28, 32, 36, 40};
193 int expected_cols[] = {0, 6, 7, 8, 9,
194 1, 2, 3, 4, 5,
195 1, 2, 3, 4, 5,
196 3, 4, 5,
197 3, 4, 5,
198 3, 4, 5,
199 6, 7, 8, 9,
200 6, 7, 8, 9,
201 6, 7, 8, 9,
202 6, 7, 8, 9};
203
204
205 vector<pair<const double*, const double*> > covariance_blocks;
206 covariance_blocks.push_back(make_pair(block1, block1));
207 covariance_blocks.push_back(make_pair(block4, block4));
208 covariance_blocks.push_back(make_pair(block2, block2));
209 covariance_blocks.push_back(make_pair(block3, block3));
210 covariance_blocks.push_back(make_pair(block2, block3));
211 covariance_blocks.push_back(make_pair(block4, block1)); // reversed
212
213 Covariance::Options options;
214 CovarianceImpl covariance_impl(options);
215 EXPECT_TRUE(covariance_impl
216 .ComputeCovarianceSparsity(covariance_blocks, &problem));
217
218 const CompressedRowSparseMatrix* crsm = covariance_impl.covariance_matrix();
219
220 EXPECT_EQ(crsm->num_rows(), 10);
221 EXPECT_EQ(crsm->num_cols(), 10);
222 EXPECT_EQ(crsm->num_nonzeros(), 40);
223
224 const int* rows = crsm->rows();
225 for (int r = 0; r < crsm->num_rows() + 1; ++r) {
226 EXPECT_EQ(rows[r], expected_rows[r])
227 << r << " "
228 << rows[r] << " "
229 << expected_rows[r];
230 }
231
232 const int* cols = crsm->cols();
233 for (int c = 0; c < crsm->num_nonzeros(); ++c) {
234 EXPECT_EQ(cols[c], expected_cols[c])
235 << c << " "
236 << cols[c] << " "
237 << expected_cols[c];
238 }
239}

489 void ComputeAndCompareCovarianceBlocksInTangentOrAmbientSpace(
490 const Covariance::Options& options,
491 bool lift_covariance_to_ambient_space,
492 const double* expected_covariance) {
493 // Generate all possible combination of block pairs and check if the
494 // covariance computation is correct.
495 for (int i = 0; i <= 64; ++i) {
496 vector<pair<const double*, const double*> > covariance_blocks;
497 if (i & 1) {
498 covariance_blocks.push_back(all_covariance_blocks_[0]);
499 }
500
501 if (i & 2) {
502 covariance_blocks.push_back(all_covariance_blocks_[1]);
503 }
504
505 if (i & 4) {
506 covariance_blocks.push_back(all_covariance_blocks_[2]);
507 }
508
509 if (i & 8) {
510 covariance_blocks.push_back(all_covariance_blocks_[3]);
511 }
512
513 if (i & 16) {
514 covariance_blocks.push_back(all_covariance_blocks_[4]);
515 }
516
517 if (i & 32) {
518 covariance_blocks.push_back(all_covariance_blocks_[5]);
519 }
520
521 Covariance covariance(options);
522 EXPECT_TRUE(covariance.Compute(covariance_blocks, &problem_));
523
524 for (int i = 0; i < covariance_blocks.size(); ++i) {
525 const double* block1 = covariance_blocks[i].first;
526 const double* block2 = covariance_blocks[i].second;
527 // block1, block2
528 GetCovarianceBlockAndCompare(block1,
529 block2,
530 lift_covariance_to_ambient_space,
531 covariance,
532 expected_covariance);
533 // block2, block1
534 GetCovarianceBlockAndCompare(block2,
535 block1,
536 lift_covariance_to_ambient_space,
537 covariance,
538 expected_covariance);
539 }
540 }
541 }
542
543 void GetCovarianceBlockAndCompare(const double* block1,
544 const double* block2,
545 bool lift_covariance_to_ambient_space,
546 const Covariance& covariance,
547 const double* expected_covariance) {
548 const BoundsMap& column_bounds = lift_covariance_to_ambient_space ?
549 column_bounds_ : local_column_bounds_;
550 const int row_begin = FindOrDie(column_bounds, block1).first;
551 const int row_end = FindOrDie(column_bounds, block1).second;
552 const int col_begin = FindOrDie(column_bounds, block2).first;
553 const int col_end = FindOrDie(column_bounds, block2).second;
554
555 Matrix actual(row_end - row_begin, col_end - col_begin);
556 if (lift_covariance_to_ambient_space) {
557 EXPECT_TRUE(covariance.GetCovarianceBlock(block1,
558 block2,
559 actual.data()));
560 } else {
561 EXPECT_TRUE(covariance.GetCovarianceBlockInTangentSpace(block1,
562 block2,
563 actual.data()));
564 }
565
566 int dof = 0; // degrees of freedom = sum of LocalSize()s
567 for (BoundsMap::const_iterator iter = column_bounds.begin();
568 iter != column_bounds.end(); ++iter) {
569 dof = std::max(dof, iter->second.second);
570 }
571 ConstMatrixRef expected(expected_covariance, dof, dof);
572 double diff_norm = (expected.block(row_begin,
573 col_begin,
574 row_end - row_begin,
575 col_end - col_begin) - actual).norm();
576 diff_norm /= (row_end - row_begin) * (col_end - col_begin);
577
578 const double kTolerance = 1e-5;
579 EXPECT_NEAR(diff_norm, 0.0, kTolerance)
580 << "rows: " << row_begin << " " << row_end << " "
581 << "cols: " << col_begin << " " << col_end << " "
582 << "\n\n expected: \n " << expected.block(row_begin,
583 col_begin,
584 row_end - row_begin,
585 col_end - col_begin)
586 << "\n\n actual: \n " << actual
587 << "\n\n full expected: \n" << expected;
588 }
589
590 double parameters_[6];
591 Problem problem_;
592 vector<pair<const double*, const double*> > all_covariance_blocks_;
593 BoundsMap column_bounds_;
594 BoundsMap local_column_bounds_;
595};
