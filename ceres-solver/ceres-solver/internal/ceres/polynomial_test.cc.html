48// For IEEE-754 doubles, machine precision is about 2e-16.
49const double kEpsilon = 1e-13;
50const double kEpsilonLoose = 1e-9;
51
52// Return the constant polynomial p(x) = 1.23.
53Vector ConstantPolynomial(double value) {
54 Vector poly(1);
55 poly(0) = value;
56 return poly;
57}
58
59// Return the polynomial p(x) = poly(x) * (x - root).
60Vector AddRealRoot(const Vector& poly, double root) {
61 Vector poly2(poly.size() + 1);
62 poly2.setZero();
63 poly2.head(poly.size()) += poly;
64 poly2.tail(poly.size()) -= root * poly;
65 return poly2;
66}
67
68// Return the polynomial
69// p(x) = poly(x) * (x - real - imag*i) * (x - real + imag*i).
70Vector AddComplexRootPair(const Vector& poly, double real, double imag) {
71 Vector poly2(poly.size() + 2);
72 poly2.setZero();
73 // Multiply poly by x^2 - 2real + abs(real,imag)^2
74 poly2.head(poly.size()) += poly;
75 poly2.segment(1, poly.size()) -= 2 * real * poly;
76 poly2.tail(poly.size()) += (real*real + imag*imag) * poly;
77 return poly2;
78}
79
80// Sort the entries in a vector.
81// Needed because the roots are not returned in sorted order.
82Vector SortVector(const Vector& in) {
83 Vector out(in);
84 std::sort(out.data(), out.data() + out.size());
85 return out;
86}

88// Run a test with the polynomial defined by the N real roots in roots_real.
89// If use_real is false, NULL is passed as the real argument to
90// FindPolynomialRoots. If use_imaginary is false, NULL is passed as the
91// imaginary argument to FindPolynomialRoots.
92template<int N>
93void RunPolynomialTestRealRoots(const double (&real_roots)[N],
94 bool use_real,
95 bool use_imaginary,
96 double epsilon) {
97 Vector real;
98 Vector imaginary;
99 Vector poly = ConstantPolynomial(1.23);
100 for (int i = 0; i < N; ++i) {
101 poly = AddRealRoot(poly, real_roots[i]);
102 }
103 Vector* const real_ptr = use_real ? &real : NULL;
104 Vector* const imaginary_ptr = use_imaginary ? &imaginary : NULL;
105 bool success = FindPolynomialRoots(poly, real_ptr, imaginary_ptr);
106
107 EXPECT_EQ(success, true);
108 if (use_real) {
109 EXPECT_EQ(real.size(), N);
110 real = SortVector(real);
111 ExpectArraysClose(N, real.data(), real_roots, epsilon);
112 }
113 if (use_imaginary) {
114 EXPECT_EQ(imaginary.size(), N);
115 const Vector zeros = Vector::Zero(N);
116 ExpectArraysClose(N, imaginary.data(), zeros.data(), epsilon);
117 }
118}

121TEST(Polynomial, InvalidPolynomialOfZeroLengthIsRejected) {
122 // Vector poly(0) is an ambiguous constructor call, so
123 // use the constructor with explicit column count.
124 Vector poly(0, 1);
125 Vector real;
126 Vector imag;
127 bool success = FindPolynomialRoots(poly, &real, &imag);
128
129 EXPECT_EQ(success, false);
130}

173TEST(Polynomial, QuadraticPolynomialWithComplexRootsWorks) {
174 Vector real;
175 Vector imag;
176
177 Vector poly = ConstantPolynomial(1.23);
178 poly = AddComplexRootPair(poly, 42.42, 4.2);
179 bool success = FindPolynomialRoots(poly, &real, &imag);
180
181 EXPECT_EQ(success, true);
182 EXPECT_EQ(real.size(), 2);
183 EXPECT_EQ(imag.size(), 2);
184 ExpectClose(real(0), 42.42, kEpsilon);
185 ExpectClose(real(1), 42.42, kEpsilon);
186 ExpectClose(std::abs(imag(0)), 4.2, kEpsilon);
187 ExpectClose(std::abs(imag(1)), 4.2, kEpsilon);
188 ExpectClose(std::abs(imag(0) + imag(1)), 0.0, kEpsilon);
189}
