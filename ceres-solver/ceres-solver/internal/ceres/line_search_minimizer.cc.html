66// TODO(sameeragarwal): I think there is a small bug here, in that if
67// the evaluation fails, then the state can contain garbage. Look at
68// this more carefully.
69bool Evaluate(Evaluator* evaluator,
70 const Vector& x,
71 LineSearchMinimizer::State* state,
72 std::string* message) {
73 if (!evaluator->Evaluate(x.data(),
74 &(state->cost),
75 NULL,
76 state->gradient.data(),
77 NULL)) {
78 *message = "Gradient evaluation failed.";
79 return false;
80 }
81
82 Vector negative_gradient = -state->gradient;
83 Vector projected_gradient_step(x.size());
84 if (!evaluator->Plus(x.data(),
85 negative_gradient.data(),
86 projected_gradient_step.data())) {
87 *message = "projected_gradient_step = Plus(x, -gradient) failed.";
88 return false;
89 }
90
91 state->gradient_squared_norm = (x - projected_gradient_step).squaredNorm();
92 state->gradient_max_norm =
93 (x - projected_gradient_step).lpNorm<Eigen::Infinity>();
94 return true;
95}

99void LineSearchMinimizer::Minimize(const Minimizer::Options& options,
100 double* parameters,
101 Solver::Summary* summary) {
102 const bool is_not_silent = !options.is_silent;
103 double start_time = WallTimeInSeconds();
104 double iteration_start_time = start_time;
105
106 Evaluator* evaluator = CHECK_NOTNULL(options.evaluator.get());
107 const int num_parameters = evaluator->NumParameters();
108 const int num_effective_parameters = evaluator->NumEffectiveParameters();
109
110 summary->termination_type = NO_CONVERGENCE;
111 summary->num_successful_steps = 0;
112 summary->num_unsuccessful_steps = 0;
113
114 VectorRef x(parameters, num_parameters);
115
116 State current_state(num_parameters, num_effective_parameters);
117 State previous_state(num_parameters, num_effective_parameters);
118
119 Vector delta(num_effective_parameters);
120 Vector x_plus_delta(num_parameters);
121
122 IterationSummary iteration_summary;
123 iteration_summary.iteration = 0;
124 iteration_summary.step_is_valid = false;
125 iteration_summary.step_is_successful = false;
126 iteration_summary.cost_change = 0.0;
127 iteration_summary.gradient_max_norm = 0.0;
128 iteration_summary.gradient_norm = 0.0;
129 iteration_summary.step_norm = 0.0;
130 iteration_summary.linear_solver_iterations = 0;
131 iteration_summary.step_solver_time_in_seconds = 0;
132
133 // Do initial cost and Jacobian evaluation.
134 if (!Evaluate(evaluator, x, &current_state, &summary->message)) {
135 summary->termination_type = FAILURE;
136 summary->message = "Initial cost and jacobian evaluation failed. "
137 "More details: " + summary->message;
138 LOG_IF(WARNING, is_not_silent) << "Terminating: " << summary->message;
139 return;
140 }
141
142 summary->initial_cost = current_state.cost + summary->fixed_cost;
143 iteration_summary.cost = current_state.cost + summary->fixed_cost;
144
145 iteration_summary.gradient_max_norm = current_state.gradient_max_norm;
146 iteration_summary.gradient_norm = sqrt(current_state.gradient_squared_norm);
147
148 if (iteration_summary.gradient_max_norm <= options.gradient_tolerance) {
149 summary->message = StringPrintf("Gradient tolerance reached. "
150 "Gradient max norm: %e <= %e",
151 iteration_summary.gradient_max_norm,
152 options.gradient_tolerance);
153 summary->termination_type = CONVERGENCE;
154 VLOG_IF(1, is_not_silent) << "Terminating: " << summary->message;
155 return;
156 }
157
158 iteration_summary.iteration_time_in_seconds =
159 WallTimeInSeconds() - iteration_start_time;
160 iteration_summary.cumulative_time_in_seconds =
161 WallTimeInSeconds() - start_time
162 + summary->preprocessor_time_in_seconds;
163 summary->iterations.push_back(iteration_summary);
164
165 LineSearchDirection::Options line_search_direction_options;
166 line_search_direction_options.num_parameters = num_effective_parameters;
167 line_search_direction_options.type = options.line_search_direction_type;
168 line_search_direction_options.nonlinear_conjugate_gradient_type =
169 options.nonlinear_conjugate_gradient_type;
170 line_search_direction_options.max_lbfgs_rank = options.max_lbfgs_rank;
171 line_search_direction_options.use_approximate_eigenvalue_bfgs_scaling =
172 options.use_approximate_eigenvalue_bfgs_scaling;
173 scoped_ptr<LineSearchDirection> line_search_direction(
174 LineSearchDirection::Create(line_search_direction_options));
175
176 LineSearchFunction line_search_function(evaluator);
177
178 LineSearch::Options line_search_options;
179 line_search_options.interpolation_type =
180 options.line_search_interpolation_type;
181 line_search_options.min_step_size = options.min_line_search_step_size;
182 line_search_options.sufficient_decrease =
183 options.line_search_sufficient_function_decrease;
184 line_search_options.max_step_contraction =
185 options.max_line_search_step_contraction;
186 line_search_options.min_step_contraction =
187 options.min_line_search_step_contraction;
188 line_search_options.max_num_iterations =
189 options.max_num_line_search_step_size_iterations;
190 line_search_options.sufficient_curvature_decrease =
191 options.line_search_sufficient_curvature_decrease;
192 line_search_options.max_step_expansion =
193 options.max_line_search_step_expansion;
194 line_search_options.is_silent = options.is_silent;
195 line_search_options.function = &line_search_function;
196
197 scoped_ptr<LineSearch>
198 line_search(LineSearch::Create(options.line_search_type,
199 line_search_options,
200 &summary->message));
201 if (line_search.get() == NULL) {
202 summary->termination_type = FAILURE;
203 LOG_IF(ERROR, is_not_silent) << "Terminating: " << summary->message;
204 return;
205 }
206
207 LineSearch::Summary line_search_summary;
208 int num_line_search_direction_restarts = 0;
209
210 while (true) {
211 if (!RunCallbacks(options, iteration_summary, summary)) {
212 break;
213 }
214
215 iteration_start_time = WallTimeInSeconds();
216 if (iteration_summary.iteration >= options.max_num_iterations) {
217 summary->message = "Maximum number of iterations reached.";
218 summary->termination_type = NO_CONVERGENCE;
219 VLOG_IF(1, is_not_silent) << "Terminating: " << summary->message;
220 break;
221 }
222
223 const double total_solver_time = iteration_start_time - start_time +
224 summary->preprocessor_time_in_seconds;
225 if (total_solver_time >= options.max_solver_time_in_seconds) {
226 summary->message = "Maximum solver time reached.";
227 summary->termination_type = NO_CONVERGENCE;
228 VLOG_IF(1, is_not_silent) << "Terminating: " << summary->message;
229 break;
230 }
231
232 iteration_summary = IterationSummary();
233 iteration_summary.iteration = summary->iterations.back().iteration + 1;
234 iteration_summary.step_is_valid = false;
235 iteration_summary.step_is_successful = false;
236
237 bool line_search_status = true;
238 if (iteration_summary.iteration == 1) {
239 current_state.search_direction = -current_state.gradient;
240 } else {
241 line_search_status = line_search_direction->NextDirection(
242 previous_state,
243 current_state,
244 &current_state.search_direction);
245 }
246
247 if (!line_search_status &&
248 num_line_search_direction_restarts >=
249 options.max_num_line_search_direction_restarts) {
250 // Line search direction failed to generate a new direction, and we
251 // have already reached our specified maximum number of restarts,
252 // terminate optimization.
253 summary->message =
254 StringPrintf("Line search direction failure: specified "
255 "max_num_line_search_direction_restarts: %d reached.",
256 options.max_num_line_search_direction_restarts);
257 summary->termination_type = FAILURE;
258 LOG_IF(WARNING, is_not_silent) << "Terminating: " << summary->message;
259 break;
260 } else if (!line_search_status) {
261 // Restart line search direction with gradient descent on first iteration
262 // as we have not yet reached our maximum number of restarts.
263 CHECK_LT(num_line_search_direction_restarts,
264 options.max_num_line_search_direction_restarts);
265
266 ++num_line_search_direction_restarts;
267 LOG_IF(WARNING, is_not_silent)
268 << "Line search direction algorithm: "
269 << LineSearchDirectionTypeToString(
270 options.line_search_direction_type)
271 << ", failed to produce a valid new direction at "
272 << "iteration: " << iteration_summary.iteration
273 << ". Restarting, number of restarts: "
274 << num_line_search_direction_restarts << " / "
275 << options.max_num_line_search_direction_restarts
276 << " [max].";
277 line_search_direction.reset(
278 LineSearchDirection::Create(line_search_direction_options));
279 current_state.search_direction = -current_state.gradient;
280 }
281
282 line_search_function.Init(x, current_state.search_direction);
283 current_state.directional_derivative =
284 current_state.gradient.dot(current_state.search_direction);
285
286 // TODO(sameeragarwal): Refactor this into its own object and add
287 // explanations for the various choices.
288 //
289 // Note that we use !line_search_status to ensure that we treat cases when
290 // we restarted the line search direction equivalently to the first
291 // iteration.
292 const double initial_step_size =
293 (iteration_summary.iteration == 1 || !line_search_status)
294 ? std::min(1.0, 1.0 / current_state.gradient_max_norm)
295 : std::min(1.0, 2.0 * (current_state.cost - previous_state.cost) /
296 current_state.directional_derivative);
297 // By definition, we should only ever go forwards along the specified search
298 // direction in a line search, most likely cause for this being violated
299 // would be a numerical failure in the line search direction calculation.
300 if (initial_step_size < 0.0) {
301 summary->message =
302 StringPrintf("Numerical failure in line search, initial_step_size is "
303 "negative: %.5e, directional_derivative: %.5e, "
304 "(current_cost - previous_cost): %.5e",
305 initial_step_size, current_state.directional_derivative,
306 (current_state.cost - previous_state.cost));
307 summary->termination_type = FAILURE;
308 LOG_IF(WARNING, is_not_silent) << "Terminating: " << summary->message;
309 break;
310 }
311
312 line_search->Search(initial_step_size,
313 current_state.cost,
314 current_state.directional_derivative,
315 &line_search_summary);
316 if (!line_search_summary.success) {
317 summary->message =
318 StringPrintf("Numerical failure in line search, failed to find "
319 "a valid step size, (did not run out of iterations) "
320 "using initial_step_size: %.5e, initial_cost: %.5e, "
321 "initial_gradient: %.5e.",
322 initial_step_size, current_state.cost,
323 current_state.directional_derivative);
324 LOG_IF(WARNING, is_not_silent) << "Terminating: " << summary->message;
325 summary->termination_type = FAILURE;
326 break;
327 }
328
329 current_state.step_size = line_search_summary.optimal_step_size;
330 delta = current_state.step_size * current_state.search_direction;
331
332 previous_state = current_state;
333 iteration_summary.step_solver_time_in_seconds =
334 WallTimeInSeconds() - iteration_start_time;
335
336 const double x_norm = x.norm();
337
338 if (!evaluator->Plus(x.data(), delta.data(), x_plus_delta.data())) {
339 summary->termination_type = FAILURE;
340 summary->message =
341 "x_plus_delta = Plus(x, delta) failed. This should not happen "
342 "as the step was valid when it was selected by the line search.";
343 LOG_IF(WARNING, is_not_silent) << "Terminating: " << summary->message;
344 break;
345 }
346
347 if (!Evaluate(evaluator,
348 x_plus_delta,
349 &current_state,
350 &summary->message)) {
351 summary->termination_type = FAILURE;
352 summary->message =
353 "Step failed to evaluate. This should not happen as the step was "
354 "valid when it was selected by the line search. More details: " +
355 summary->message;
356 LOG_IF(WARNING, is_not_silent) << "Terminating: " << summary->message;
357 break;
358 }
359
360 // Compute the norm of the step in the ambient space.
361 iteration_summary.step_norm = (x_plus_delta - x).norm();
362 x = x_plus_delta;
363
364 iteration_summary.gradient_max_norm = current_state.gradient_max_norm;
365 iteration_summary.gradient_norm = sqrt(current_state.gradient_squared_norm);
366 iteration_summary.cost_change = previous_state.cost - current_state.cost;
367 iteration_summary.cost = current_state.cost + summary->fixed_cost;
368
369 iteration_summary.step_is_valid = true;
370 iteration_summary.step_is_successful = true;
371 iteration_summary.step_size = current_state.step_size;
372 iteration_summary.line_search_function_evaluations =
373 line_search_summary.num_function_evaluations;
374 iteration_summary.line_search_gradient_evaluations =
375 line_search_summary.num_gradient_evaluations;
376 iteration_summary.line_search_iterations =
377 line_search_summary.num_iterations;
378 iteration_summary.iteration_time_in_seconds =
379 WallTimeInSeconds() - iteration_start_time;
380 iteration_summary.cumulative_time_in_seconds =
381 WallTimeInSeconds() - start_time
382 + summary->preprocessor_time_in_seconds;
383 summary->iterations.push_back(iteration_summary);
384
385 // Iterations inside the line search algorithm are considered
386 // 'steps' in the broader context, to distinguish these inner
387 // iterations from from the outer iterations of the line search
388 // minimizer. The number of line search steps is the total number
389 // of inner line search iterations (or steps) across the entire
390 // minimization.
391 summary->num_line_search_steps += line_search_summary.num_iterations;
392 summary->line_search_cost_evaluation_time_in_seconds +=
393 line_search_summary.cost_evaluation_time_in_seconds;
394 summary->line_search_gradient_evaluation_time_in_seconds +=
395 line_search_summary.gradient_evaluation_time_in_seconds;
396 summary->line_search_polynomial_minimization_time_in_seconds +=
397 line_search_summary.polynomial_minimization_time_in_seconds;
398 summary->line_search_total_time_in_seconds +=
399 line_search_summary.total_time_in_seconds;
400 ++summary->num_successful_steps;
401
402 const double step_size_tolerance = options.parameter_tolerance *
403 (x_norm + options.parameter_tolerance);
404 if (iteration_summary.step_norm <= step_size_tolerance) {
405 summary->message =
406 StringPrintf("Parameter tolerance reached. "
407 "Relative step_norm: %e <= %e.",
408 (iteration_summary.step_norm /
409 (x_norm + options.parameter_tolerance)),
410 options.parameter_tolerance);
411 summary->termination_type = CONVERGENCE;
412 VLOG_IF(1, is_not_silent) << "Terminating: " << summary->message;
413 return;
414 }
415
416 if (iteration_summary.gradient_max_norm <= options.gradient_tolerance) {
417 summary->message = StringPrintf("Gradient tolerance reached. "
418 "Gradient max norm: %e <= %e",
419 iteration_summary.gradient_max_norm,
420 options.gradient_tolerance);
421 summary->termination_type = CONVERGENCE;
422 VLOG_IF(1, is_not_silent) << "Terminating: " << summary->message;
423 break;
424 }
425
426 const double absolute_function_tolerance =
427 options.function_tolerance * previous_state.cost;
428 if (fabs(iteration_summary.cost_change) <= absolute_function_tolerance) {
429 summary->message =
430 StringPrintf("Function tolerance reached. "
431 "|cost_change|/cost: %e <= %e",
432 fabs(iteration_summary.cost_change) /
433 previous_state.cost,
434 options.function_tolerance);
435 summary->termination_type = CONVERGENCE;
436 VLOG_IF(1, is_not_silent) << "Terminating: " << summary->message;
437 break;
438 }
439 }
