69class BlockRandomAccessSparseMatrixAdapter : public LinearOperator {
70 public:
71 explicit BlockRandomAccessSparseMatrixAdapter(
72 const BlockRandomAccessSparseMatrix& m)
73 : m_(m) {
74 }
75
76 virtual ~BlockRandomAccessSparseMatrixAdapter() {}
77
78 // y = y + Ax;
79 virtual void RightMultiply(const double* x, double* y) const {
80 m_.SymmetricRightMultiply(x, y);
81 }
82
83 // y = y + A'x;
84 virtual void LeftMultiply(const double* x, double* y) const {
85 m_.SymmetricRightMultiply(x, y);
86 }
87
88 virtual int num_rows() const { return m_.num_rows(); }
89 virtual int num_cols() const { return m_.num_rows(); }
90
91 private:
92 const BlockRandomAccessSparseMatrix& m_;
93};

123LinearSolver::Summary SchurComplementSolver::SolveImpl(
124 BlockSparseMatrix* A,
125 const double* b,
126 const LinearSolver::PerSolveOptions& per_solve_options,
127 double* x) {
128 EventLogger event_logger("SchurComplementSolver::Solve");
129
130 if (eliminator_.get() == NULL) {
131 InitStorage(A->block_structure());
132 DetectStructure(*A->block_structure(),
133 options_.elimination_groups[0],
134 &options_.row_block_size,
135 &options_.e_block_size,
136 &options_.f_block_size);
137 eliminator_.reset(CHECK_NOTNULL(SchurEliminatorBase::Create(options_)));
138 const bool kFullRankETE = true;
139 eliminator_->Init(
140 options_.elimination_groups[0], kFullRankETE, A->block_structure());
141 };
142 std::fill(x, x + A->num_cols(), 0.0);
143 event_logger.AddEvent("Setup");
144
145 eliminator_->Eliminate(A, b, per_solve_options.D, lhs_.get(), rhs_.get());
146 event_logger.AddEvent("Eliminate");
147
148 double* reduced_solution = x + A->num_cols() - lhs_->num_cols();
149 const LinearSolver::Summary summary =
150 SolveReducedLinearSystem(per_solve_options, reduced_solution);
151 event_logger.AddEvent("ReducedSolve");
152
153 if (summary.termination_type == LINEAR_SOLVER_SUCCESS) {
154 eliminator_->BackSubstitute(A, b, per_solve_options.D, reduced_solution, x);
155 event_logger.AddEvent("BackSubstitute");
156 }
157
158 return summary;
159}

161// Initialize a BlockRandomAccessDenseMatrix to store the Schur
162// complement.
163void DenseSchurComplementSolver::InitStorage(
164 const CompressedRowBlockStructure* bs) {
165 const int num_eliminate_blocks = options().elimination_groups[0];
166 const int num_col_blocks = bs->cols.size();
167
168 vector<int> blocks(num_col_blocks - num_eliminate_blocks, 0);
169 for (int i = num_eliminate_blocks, j = 0;
170 i < num_col_blocks;
171 ++i, ++j) {
172 blocks[j] = bs->cols[i].size;
173 }
174
175 set_lhs(new BlockRandomAccessDenseMatrix(blocks));
176 set_rhs(new double[lhs()->num_rows()]);
177}

179// Solve the system Sx = r, assuming that the matrix S is stored in a
180// BlockRandomAccessDenseMatrix. The linear system is solved using
181// Eigen's Cholesky factorization.
182LinearSolver::Summary
183DenseSchurComplementSolver::SolveReducedLinearSystem(
184 const LinearSolver::PerSolveOptions& per_solve_options,
185 double* solution) {
186 LinearSolver::Summary summary;
187 summary.num_iterations = 0;
188 summary.termination_type = LINEAR_SOLVER_SUCCESS;
189 summary.message = "Success.";
190
191 const BlockRandomAccessDenseMatrix* m =
192 down_cast<const BlockRandomAccessDenseMatrix*>(lhs());
193 const int num_rows = m->num_rows();
194
195 // The case where there are no f blocks, and the system is block
196 // diagonal.
197 if (num_rows == 0) {
198 return summary;
199 }
200
201 summary.num_iterations = 1;
202
203 if (options().dense_linear_algebra_library_type == EIGEN) {
204 Eigen::LLT<Matrix, Eigen::Upper> llt =
205 ConstMatrixRef(m->values(), num_rows, num_rows)
206 .selfadjointView<Eigen::Upper>()
207 .llt();
208 if (llt.info() != Eigen::Success) {
209 summary.termination_type = LINEAR_SOLVER_FAILURE;
210 summary.message =
211 "Eigen failure. Unable to perform dense Cholesky factorization.";
212 return summary;
213 }
214
215 VectorRef(solution, num_rows) = llt.solve(ConstVectorRef(rhs(), num_rows));
216 } else {
217 VectorRef(solution, num_rows) = ConstVectorRef(rhs(), num_rows);
218 summary.termination_type =
219 LAPACK::SolveInPlaceUsingCholesky(num_rows,
220 m->values(),
221 solution,
222 &summary.message);
223 }
224
225 return summary;
226}

247// Determine the non-zero blocks in the Schur Complement matrix, and
248// initialize a BlockRandomAccessSparseMatrix object.
249void SparseSchurComplementSolver::InitStorage(
250 const CompressedRowBlockStructure* bs) {
251 const int num_eliminate_blocks = options().elimination_groups[0];
252 const int num_col_blocks = bs->cols.size();
253 const int num_row_blocks = bs->rows.size();
254
255 blocks_.resize(num_col_blocks - num_eliminate_blocks, 0);
256 for (int i = num_eliminate_blocks; i < num_col_blocks; ++i) {
257 blocks_[i - num_eliminate_blocks] = bs->cols[i].size;
258 }
259
260 set<pair<int, int> > block_pairs;
261 for (int i = 0; i < blocks_.size(); ++i) {
262 block_pairs.insert(make_pair(i, i));
263 }
264
265 int r = 0;
266 while (r < num_row_blocks) {
267 int e_block_id = bs->rows[r].cells.front().block_id;
268 if (e_block_id >= num_eliminate_blocks) {
269 break;
270 }
271 vector<int> f_blocks;
272
273 // Add to the chunk until the first block in the row is
274 // different than the one in the first row for the chunk.
275 for (; r < num_row_blocks; ++r) {
276 const CompressedRow& row = bs->rows[r];
277 if (row.cells.front().block_id != e_block_id) {
278 break;
279 }
280
281 // Iterate over the blocks in the row, ignoring the first
282 // block since it is the one to be eliminated.
283 for (int c = 1; c < row.cells.size(); ++c) {
284 const Cell& cell = row.cells[c];
285 f_blocks.push_back(cell.block_id - num_eliminate_blocks);
286 }
287 }
288
289 sort(f_blocks.begin(), f_blocks.end());
290 f_blocks.erase(unique(f_blocks.begin(), f_blocks.end()), f_blocks.end());
291 for (int i = 0; i < f_blocks.size(); ++i) {
292 for (int j = i + 1; j < f_blocks.size(); ++j) {
293 block_pairs.insert(make_pair(f_blocks[i], f_blocks[j]));
294 }
295 }
296 }
297
298 // Remaing rows do not contribute to the chunks and directly go
299 // into the schur complement via an outer product.
300 for (; r < num_row_blocks; ++r) {
301 const CompressedRow& row = bs->rows[r];
302 CHECK_GE(row.cells.front().block_id, num_eliminate_blocks);
303 for (int i = 0; i < row.cells.size(); ++i) {
304 int r_block1_id = row.cells[i].block_id - num_eliminate_blocks;
305 for (int j = 0; j < row.cells.size(); ++j) {
306 int r_block2_id = row.cells[j].block_id - num_eliminate_blocks;
307 if (r_block1_id <= r_block2_id) {
308 block_pairs.insert(make_pair(r_block1_id, r_block2_id));
309 }
310 }
311 }
312 }
313
314 set_lhs(new BlockRandomAccessSparseMatrix(blocks_, block_pairs));
315 set_rhs(new double[lhs()->num_rows()]);
316}

318LinearSolver::Summary
319SparseSchurComplementSolver::SolveReducedLinearSystem(
320 const LinearSolver::PerSolveOptions& per_solve_options,
321 double* solution) {
322 if (options().type == ITERATIVE_SCHUR) {
323 CHECK(options().use_explicit_schur_complement);
324 return SolveReducedLinearSystemUsingConjugateGradients(per_solve_options,
325 solution);
326 }
327
328 switch (options().sparse_linear_algebra_library_type) {
329 case SUITE_SPARSE:
330 return SolveReducedLinearSystemUsingSuiteSparse(per_solve_options,
331 solution);
332 case CX_SPARSE:
333 return SolveReducedLinearSystemUsingCXSparse(per_solve_options,
334 solution);
335 case EIGEN_SPARSE:
336 return SolveReducedLinearSystemUsingEigen(per_solve_options,
337 solution);
338 default:
339 LOG(FATAL) << "Unknown sparse linear algebra library : "
340 << options().sparse_linear_algebra_library_type;
341 }
342
343 return LinearSolver::Summary();
344}

346// Solve the system Sx = r, assuming that the matrix S is stored in a
347// BlockRandomAccessSparseMatrix. The linear system is solved using
348// CHOLMOD's sparse cholesky factorization routines.
349LinearSolver::Summary
350SparseSchurComplementSolver::SolveReducedLinearSystemUsingSuiteSparse(
351 const LinearSolver::PerSolveOptions& per_solve_options,
352 double* solution) {
353#ifdef CERES_NO_SUITESPARSE
354
355 LinearSolver::Summary summary;
356 summary.num_iterations = 0;
357 summary.termination_type = LINEAR_SOLVER_FATAL_ERROR;
358 summary.message = "Ceres was not built with SuiteSparse support. "
359 "Therefore, SPARSE_SCHUR cannot be used with SUITE_SPARSE";
360 return summary;
361
362#else
363
364 LinearSolver::Summary summary;
365 summary.num_iterations = 0;
366 summary.termination_type = LINEAR_SOLVER_SUCCESS;
367 summary.message = "Success.";
368
369 TripletSparseMatrix* tsm =
370 const_cast<TripletSparseMatrix*>(
371 down_cast<const BlockRandomAccessSparseMatrix*>(lhs())->matrix());
372 const int num_rows = tsm->num_rows();
373
374 // The case where there are no f blocks, and the system is block
375 // diagonal.
376 if (num_rows == 0) {
377 return summary;
378 }
379
380 summary.num_iterations = 1;
381 cholmod_sparse* cholmod_lhs = NULL;
382 if (options().use_postordering) {
383 // If we are going to do a full symbolic analysis of the schur
384 // complement matrix from scratch and not rely on the
385 // pre-ordering, then the fastest path in cholmod_factorize is the
386 // one corresponding to upper triangular matrices.
387
388 // Create a upper triangular symmetric matrix.
389 cholmod_lhs = ss_.CreateSparseMatrix(tsm);
390 cholmod_lhs->stype = 1;
391
392 if (factor_ == NULL) {
393 factor_ = ss_.BlockAnalyzeCholesky(cholmod_lhs,
394 blocks_,
395 blocks_,
396 &summary.message);
397 }
398 } else {
399 // If we are going to use the natural ordering (i.e. rely on the
400 // pre-ordering computed by solver_impl.cc), then the fastest
401 // path in cholmod_factorize is the one corresponding to lower
402 // triangular matrices.
403
404 // Create a upper triangular symmetric matrix.
405 cholmod_lhs = ss_.CreateSparseMatrixTranspose(tsm);
406 cholmod_lhs->stype = -1;
407
408 if (factor_ == NULL) {
409 factor_ = ss_.AnalyzeCholeskyWithNaturalOrdering(cholmod_lhs,
410 &summary.message);
411 }
412 }
413
414 if (factor_ == NULL) {
415 ss_.Free(cholmod_lhs);
416 summary.termination_type = LINEAR_SOLVER_FATAL_ERROR;
417 // No need to set message as it has already been set by the
418 // symbolic analysis routines above.
419 return summary;
420 }
421
422 summary.termination_type =
423 ss_.Cholesky(cholmod_lhs, factor_, &summary.message);
424
425 ss_.Free(cholmod_lhs);
426
427 if (summary.termination_type != LINEAR_SOLVER_SUCCESS) {
428 // No need to set message as it has already been set by the
429 // numeric factorization routine above.
430 return summary;
431 }
432
433 cholmod_dense* cholmod_rhs =
434 ss_.CreateDenseVector(const_cast<double*>(rhs()), num_rows, num_rows);
435 cholmod_dense* cholmod_solution = ss_.Solve(factor_,
436 cholmod_rhs,
437 &summary.message);
438 ss_.Free(cholmod_rhs);
439
440 if (cholmod_solution == NULL) {
441 summary.message =
442 "SuiteSparse failure. Unable to perform triangular solve.";
443 summary.termination_type = LINEAR_SOLVER_FAILURE;
444 return summary;
445 }
446
447 VectorRef(solution, num_rows)
448 = VectorRef(static_cast<double*>(cholmod_solution->x), num_rows);
449 ss_.Free(cholmod_solution);
450 return summary;
451#endif // CERES_NO_SUITESPARSE
452}

454// Solve the system Sx = r, assuming that the matrix S is stored in a
455// BlockRandomAccessSparseMatrix. The linear system is solved using
456// CXSparse's sparse cholesky factorization routines.
457LinearSolver::Summary
458SparseSchurComplementSolver::SolveReducedLinearSystemUsingCXSparse(
459 const LinearSolver::PerSolveOptions& per_solve_options,
460 double* solution) {
461#ifdef CERES_NO_CXSPARSE
462
463 LinearSolver::Summary summary;
464 summary.num_iterations = 0;
465 summary.termination_type = LINEAR_SOLVER_FATAL_ERROR;
466 summary.message = "Ceres was not built with CXSparse support. "
467 "Therefore, SPARSE_SCHUR cannot be used with CX_SPARSE";
468 return summary;
469
470#else
471
472 LinearSolver::Summary summary;
473 summary.num_iterations = 0;
474 summary.termination_type = LINEAR_SOLVER_SUCCESS;
475 summary.message = "Success.";
476
477 // Extract the TripletSparseMatrix that is used for actually storing S.
478 TripletSparseMatrix* tsm =
479 const_cast<TripletSparseMatrix*>(
480 down_cast<const BlockRandomAccessSparseMatrix*>(lhs())->matrix());
481 const int num_rows = tsm->num_rows();
482
483 // The case where there are no f blocks, and the system is block
484 // diagonal.
485 if (num_rows == 0) {
486 return summary;
487 }
488
489 cs_di* lhs = CHECK_NOTNULL(cxsparse_.CreateSparseMatrix(tsm));
490 VectorRef(solution, num_rows) = ConstVectorRef(rhs(), num_rows);
491
492 // Compute symbolic factorization if not available.
493 if (cxsparse_factor_ == NULL) {
494 cxsparse_factor_ = cxsparse_.BlockAnalyzeCholesky(lhs, blocks_, blocks_);
495 }
496
497 if (cxsparse_factor_ == NULL) {
498 summary.termination_type = LINEAR_SOLVER_FATAL_ERROR;
499 summary.message =
500 "CXSparse failure. Unable to find symbolic factorization.";
501 } else if (!cxsparse_.SolveCholesky(lhs, cxsparse_factor_, solution)) {
502 summary.termination_type = LINEAR_SOLVER_FAILURE;
503 summary.message = "CXSparse::SolveCholesky failed.";
504 }
505
506 cxsparse_.Free(lhs);
507 return summary;
508#endif // CERES_NO_CXPARSE
509}

511// Solve the system Sx = r, assuming that the matrix S is stored in a
512// BlockRandomAccessSparseMatrix. The linear system is solved using
513// Eigen's sparse cholesky factorization routines.
514LinearSolver::Summary
515SparseSchurComplementSolver::SolveReducedLinearSystemUsingEigen(
516 const LinearSolver::PerSolveOptions& per_solve_options,
517 double* solution) {
518#ifndef CERES_USE_EIGEN_SPARSE
519
520 LinearSolver::Summary summary;
521 summary.num_iterations = 0;
522 summary.termination_type = LINEAR_SOLVER_FATAL_ERROR;
523 summary.message =
524 "SPARSE_SCHUR cannot be used with EIGEN_SPARSE. "
525 "Ceres was not built with support for "
526 "Eigen's SimplicialLDLT decomposition. "
527 "This requires enabling building with -DEIGENSPARSE=ON.";
528 return summary;
529
530#else
531 EventLogger event_logger("SchurComplementSolver::EigenSolve");
532 LinearSolver::Summary summary;
533 summary.num_iterations = 0;
534 summary.termination_type = LINEAR_SOLVER_SUCCESS;
535 summary.message = "Success.";
536
537 // Extract the TripletSparseMatrix that is used for actually storing S.
538 TripletSparseMatrix* tsm =
539 const_cast<TripletSparseMatrix*>(
540 down_cast<const BlockRandomAccessSparseMatrix*>(lhs())->matrix());
541 const int num_rows = tsm->num_rows();
542
543 // The case where there are no f blocks, and the system is block
544 // diagonal.
545 if (num_rows == 0) {
546 return summary;
547 }
548
549 // This is an upper triangular matrix.
550 scoped_ptr<CompressedRowSparseMatrix> crsm(
551 CompressedRowSparseMatrix::FromTripletSparseMatrix(*tsm));
552 // Map this to a column major, lower triangular matrix.
553 Eigen::MappedSparseMatrix<double, Eigen::ColMajor> eigen_lhs(
554 crsm->num_rows(),
555 crsm->num_rows(),
556 crsm->num_nonzeros(),
557 crsm->mutable_rows(),
558 crsm->mutable_cols(),
559 crsm->mutable_values());
560 event_logger.AddEvent("ToCompressedRowSparseMatrix");
561
562 // Compute symbolic factorization if one does not exist.
563 if (simplicial_ldlt_.get() == NULL) {
564 simplicial_ldlt_.reset(new SimplicialLDLT);
565 // This ordering is quite bad. The scalar ordering produced by the
566 // AMD algorithm is quite bad and can be an order of magnitude
567 // worse than the one computed using the block version of the
568 // algorithm.
569 simplicial_ldlt_->analyzePattern(eigen_lhs);
570 if (VLOG_IS_ON(2)) {
571 std::stringstream ss;
572 simplicial_ldlt_->dumpMemory(ss);
573 VLOG(2) << "Symbolic Analysis\n"
574 << ss.str();
575 }
576 event_logger.AddEvent("Analysis");
577 if (simplicial_ldlt_->info() != Eigen::Success) {
578 summary.termination_type = LINEAR_SOLVER_FATAL_ERROR;
579 summary.message =
580 "Eigen failure. Unable to find symbolic factorization.";
581 return summary;
582 }
583 }
584
585 simplicial_ldlt_->factorize(eigen_lhs);
586 event_logger.AddEvent("Factorize");
587 if (simplicial_ldlt_->info() != Eigen::Success) {
588 summary.termination_type = LINEAR_SOLVER_FAILURE;
589 summary.message = "Eigen failure. Unable to find numeric factoriztion.";
590 return summary;
591 }
592
593 VectorRef(solution, num_rows) =
594 simplicial_ldlt_->solve(ConstVectorRef(rhs(), num_rows));
595 event_logger.AddEvent("Solve");
596 if (simplicial_ldlt_->info() != Eigen::Success) {
597 summary.termination_type = LINEAR_SOLVER_FAILURE;
598 summary.message = "Eigen failure. Unable to do triangular solve.";
599 }
600
601 return summary;
602#endif // CERES_USE_EIGEN_SPARSE
603}

605LinearSolver::Summary
606SparseSchurComplementSolver::SolveReducedLinearSystemUsingConjugateGradients(
607 const LinearSolver::PerSolveOptions& per_solve_options,
608 double* solution) {
609 const int num_rows = lhs()->num_rows();
610 // The case where there are no f blocks, and the system is block
611 // diagonal.
612 if (num_rows == 0) {
613 LinearSolver::Summary summary;
614 summary.num_iterations = 0;
615 summary.termination_type = LINEAR_SOLVER_SUCCESS;
616 summary.message = "Success.";
617 return summary;
618 }
619
620 // Only SCHUR_JACOBI is supported over here right now.
621 CHECK_EQ(options().preconditioner_type, SCHUR_JACOBI);
622
623 if (preconditioner_.get() == NULL) {
624 preconditioner_.reset(new BlockRandomAccessDiagonalMatrix(blocks_));
625 }
626
627 BlockRandomAccessSparseMatrix* sc =
628 down_cast<BlockRandomAccessSparseMatrix*>(
629 const_cast<BlockRandomAccessMatrix*>(lhs()));
630
631 // Extract block diagonal from the Schur complement to construct the
632 // schur_jacobi preconditioner.
633 for (int i = 0; i < blocks_.size(); ++i) {
634 const int block_size = blocks_[i];
635
636 int sc_r, sc_c, sc_row_stride, sc_col_stride;
637 CellInfo* sc_cell_info =
638 CHECK_NOTNULL(sc->GetCell(i, i,
639 &sc_r, &sc_c,
640 &sc_row_stride, &sc_col_stride));
641 MatrixRef sc_m(sc_cell_info->values, sc_row_stride, sc_col_stride);
642
643 int pre_r, pre_c, pre_row_stride, pre_col_stride;
644 CellInfo* pre_cell_info = CHECK_NOTNULL(
645 preconditioner_->GetCell(i, i,
646 &pre_r, &pre_c,
647 &pre_row_stride, &pre_col_stride));
648 MatrixRef pre_m(pre_cell_info->values, pre_row_stride, pre_col_stride);
649
650 pre_m.block(pre_r, pre_c, block_size, block_size) =
651 sc_m.block(sc_r, sc_c, block_size, block_size);
652 }
653 preconditioner_->Invert();
654
655 VectorRef(solution, num_rows).setZero();
656
657 scoped_ptr<LinearOperator> lhs_adapter(
658 new BlockRandomAccessSparseMatrixAdapter(*sc));
659 scoped_ptr<LinearOperator> preconditioner_adapter(
660 new BlockRandomAccessDiagonalMatrixAdapter(*preconditioner_));
661
662
663 LinearSolver::Options cg_options;
664 cg_options.min_num_iterations = options().min_num_iterations;
665 cg_options.max_num_iterations = options().max_num_iterations;
666 ConjugateGradientsSolver cg_solver(cg_options);
667
668 LinearSolver::PerSolveOptions cg_per_solve_options;
669 cg_per_solve_options.r_tolerance = per_solve_options.r_tolerance;
670 cg_per_solve_options.q_tolerance = per_solve_options.q_tolerance;
671 cg_per_solve_options.preconditioner = preconditioner_adapter.get();
672
673 return cg_solver.Solve(lhs_adapter.get(),
674 rhs(),
