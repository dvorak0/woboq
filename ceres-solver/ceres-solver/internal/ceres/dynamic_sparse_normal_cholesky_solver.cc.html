56DynamicSparseNormalCholeskySolver::DynamicSparseNormalCholeskySolver(
57 const LinearSolver::Options& options)
58 : options_(options) {}
59
60LinearSolver::Summary DynamicSparseNormalCholeskySolver::SolveImpl(
61 CompressedRowSparseMatrix* A,
62 const double* b,
63 const LinearSolver::PerSolveOptions& per_solve_options,
64 double* x) {
65 const int num_cols = A->num_cols();
66 VectorRef(x, num_cols).setZero();
67 A->LeftMultiply(b, x);
68
69 if (per_solve_options.D != NULL) {
70 // Temporarily append a diagonal block to the A matrix, but undo
71 // it before returning the matrix to the user.
72 scoped_ptr<CompressedRowSparseMatrix> regularizer;
73 if (!A->col_blocks().empty()) {
74 regularizer.reset(CompressedRowSparseMatrix::CreateBlockDiagonalMatrix(
75 per_solve_options.D, A->col_blocks()));
76 } else {
77 regularizer.reset(
78 new CompressedRowSparseMatrix(per_solve_options.D, num_cols));
79 }
80 A->AppendRows(*regularizer);
81 }
82
83 LinearSolver::Summary summary;
84 switch (options_.sparse_linear_algebra_library_type) {
85 case SUITE_SPARSE:
86 summary = SolveImplUsingSuiteSparse(A, x);
87 break;
88 case CX_SPARSE:
89 summary = SolveImplUsingCXSparse(A, x);
90 break;
91 case EIGEN_SPARSE:
92 summary = SolveImplUsingEigen(A, x);
93 break;
94 default:
95 LOG(FATAL) << "Unknown sparse linear algebra library : "
96 << options_.sparse_linear_algebra_library_type;
97 }
98
99 if (per_solve_options.D != NULL) {
100 A->DeleteRows(num_cols);
101 }
102
103 return summary;
104}

106LinearSolver::Summary DynamicSparseNormalCholeskySolver::SolveImplUsingEigen(
107 CompressedRowSparseMatrix* A, double* rhs_and_solution) {
108#ifndef CERES_USE_EIGEN_SPARSE
109
110 LinearSolver::Summary summary;
111 summary.num_iterations = 0;
112 summary.termination_type = LINEAR_SOLVER_FATAL_ERROR;
113 summary.message =
114 "SPARSE_NORMAL_CHOLESKY cannot be used with EIGEN_SPARSE "
115 "because Ceres was not built with support for "
116 "Eigen's SimplicialLDLT decomposition. "
117 "This requires enabling building with -DEIGENSPARSE=ON.";
118 return summary;
119
120#else
121
122 EventLogger event_logger("DynamicSparseNormalCholeskySolver::Eigen::Solve");
123
124 Eigen::MappedSparseMatrix<double, Eigen::RowMajor> a(A->num_rows(),
125 A->num_cols(),
126 A->num_nonzeros(),
127 A->mutable_rows(),
128 A->mutable_cols(),
129 A->mutable_values());
130
131 Eigen::SparseMatrix<double> lhs = a.transpose() * a;
132 Eigen::SimplicialLDLT<Eigen::SparseMatrix<double> > solver;
133
134 LinearSolver::Summary summary;
135 summary.num_iterations = 1;
136 summary.termination_type = LINEAR_SOLVER_SUCCESS;
137 summary.message = "Success.";
138
139 solver.analyzePattern(lhs);
140 if (VLOG_IS_ON(2)) {
141 std::stringstream ss;
142 solver.dumpMemory(ss);
143 VLOG(2) << "Symbolic Analysis\n" << ss.str();
144 }
145
146 event_logger.AddEvent("Analyze");
147 if (solver.info() != Eigen::Success) {
148 summary.termination_type = LINEAR_SOLVER_FATAL_ERROR;
149 summary.message = "Eigen failure. Unable to find symbolic factorization.";
150 return summary;
151 }
152
153 solver.factorize(lhs);
154 event_logger.AddEvent("Factorize");
155 if (solver.info() != Eigen::Success) {
156 summary.termination_type = LINEAR_SOLVER_FAILURE;
157 summary.message = "Eigen failure. Unable to find numeric factorization.";
158 return summary;
159 }
160
161 const Vector rhs = VectorRef(rhs_and_solution, lhs.cols());
162 VectorRef(rhs_and_solution, lhs.cols()) = solver.solve(rhs);
163 event_logger.AddEvent("Solve");
164 if (solver.info() != Eigen::Success) {
165 summary.termination_type = LINEAR_SOLVER_FAILURE;
166 summary.message = "Eigen failure. Unable to do triangular solve.";
167 return summary;
168 }
169
170 return summary;
171#endif // CERES_USE_EIGEN_SPARSE
172}

174LinearSolver::Summary DynamicSparseNormalCholeskySolver::SolveImplUsingCXSparse(
175 CompressedRowSparseMatrix* A, double* rhs_and_solution) {
176#ifdef CERES_NO_CXSPARSE
177
178 LinearSolver::Summary summary;
179 summary.num_iterations = 0;
180 summary.termination_type = LINEAR_SOLVER_FATAL_ERROR;
181 summary.message =
182 "SPARSE_NORMAL_CHOLESKY cannot be used with CX_SPARSE "
183 "because Ceres was not built with support for CXSparse. "
184 "This requires enabling building with -DCXSPARSE=ON.";
185
186 return summary;
187
188#else
189 EventLogger event_logger(
190 "DynamicSparseNormalCholeskySolver::CXSparse::Solve");
191
192 LinearSolver::Summary summary;
193 summary.num_iterations = 1;
194 summary.termination_type = LINEAR_SOLVER_SUCCESS;
195 summary.message = "Success.";
196
197 CXSparse cxsparse;
198
199 // Wrap the augmented Jacobian in a compressed sparse column matrix.
200 cs_di a_transpose = cxsparse.CreateSparseMatrixTransposeView(A);
201
202 // Compute the normal equations. J'J delta = J'f and solve them
203 // using a sparse Cholesky factorization. Notice that when compared
204 // to SuiteSparse we have to explicitly compute the transpose of Jt,
205 // and then the normal equations before they can be
206 // factorized. CHOLMOD/SuiteSparse on the other hand can just work
207 // off of Jt to compute the Cholesky factorization of the normal
208 // equations.
209 cs_di* a = cxsparse.TransposeMatrix(&a_transpose);
210 cs_di* lhs = cxsparse.MatrixMatrixMultiply(&a_transpose, a);
211 cxsparse.Free(a);
212 event_logger.AddEvent("NormalEquations");
213
214 cs_dis* factor = cxsparse.AnalyzeCholesky(lhs);
215 event_logger.AddEvent("Analysis");
216
217 if (factor == NULL) {
218 summary.termination_type = LINEAR_SOLVER_FATAL_ERROR;
219 summary.message = "CXSparse::AnalyzeCholesky failed.";
220 } else if (!cxsparse.SolveCholesky(lhs, factor, rhs_and_solution)) {
221 summary.termination_type = LINEAR_SOLVER_FAILURE;
222 summary.message = "CXSparse::SolveCholesky failed.";
223 }
224 event_logger.AddEvent("Solve");
225
226 cxsparse.Free(lhs);
227 cxsparse.Free(factor);
228 event_logger.AddEvent("TearDown");
229 return summary;
230#endif
231}

233LinearSolver::Summary
234DynamicSparseNormalCholeskySolver::SolveImplUsingSuiteSparse(
235 CompressedRowSparseMatrix* A, double* rhs_and_solution) {
236#ifdef CERES_NO_SUITESPARSE
237
238 LinearSolver::Summary summary;
239 summary.num_iterations = 0;
240 summary.termination_type = LINEAR_SOLVER_FATAL_ERROR;
241 summary.message =
242 "SPARSE_NORMAL_CHOLESKY cannot be used with SUITE_SPARSE "
243 "because Ceres was not built with support for SuiteSparse. "
244 "This requires enabling building with -DSUITESPARSE=ON.";
245 return summary;
246
247#else
248
249 EventLogger event_logger(
250 "DynamicSparseNormalCholeskySolver::SuiteSparse::Solve");
251 LinearSolver::Summary summary;
252 summary.termination_type = LINEAR_SOLVER_SUCCESS;
253 summary.num_iterations = 1;
254 summary.message = "Success.";
255
256 SuiteSparse ss;
257 const int num_cols = A->num_cols();
258 cholmod_sparse lhs = ss.CreateSparseMatrixTransposeView(A);
259 event_logger.AddEvent("Setup");
260 cholmod_factor* factor = ss.AnalyzeCholesky(&lhs, &summary.message);
261 event_logger.AddEvent("Analysis");
262
263 if (factor == NULL) {
264 summary.termination_type = LINEAR_SOLVER_FATAL_ERROR;
265 return summary;
266 }
267
268 summary.termination_type = ss.Cholesky(&lhs, factor, &summary.message);
269 if (summary.termination_type == LINEAR_SOLVER_SUCCESS) {
270 cholmod_dense* rhs =
271 ss.CreateDenseVector(rhs_and_solution, num_cols, num_cols);
272 cholmod_dense* solution = ss.Solve(factor, rhs, &summary.message);
273 event_logger.AddEvent("Solve");
274 ss.Free(rhs);
275 if (solution != NULL) {
276 memcpy(
277 rhs_and_solution, solution->x, num_cols * sizeof(*rhs_and_solution));
278 ss.Free(solution);
279 } else {
280 summary.termination_type = LINEAR_SOLVER_FAILURE;
281 }
282 }
283
284 ss.Free(factor);
285 event_logger.AddEvent("Teardown");
286 return summary;
287
288#endif
289}
