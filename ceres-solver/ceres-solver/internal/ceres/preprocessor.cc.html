42Preprocessor* Preprocessor::Create(MinimizerType minimizer_type) {
43 if (minimizer_type == TRUST_REGION) {
44 return new TrustRegionPreprocessor;
45 }
46
47 if (minimizer_type == LINE_SEARCH) {
48 return new LineSearchPreprocessor;
49 }
50
51 LOG(FATAL) << "Unknown minimizer_type: " << minimizer_type;
52 return NULL;
53}

58void ChangeNumThreadsIfNeeded(Solver::Options* options) {
59#ifndef CERES_USE_OPENMP
60 if (options->num_threads > 1) {
61 LOG(WARNING)
62 << "OpenMP support is not compiled into this binary; "
63 << "only options.num_threads = 1 is supported. Switching "
64 << "to single threaded mode.";
65 options->num_threads = 1;
66 }
67
68 // Only the Trust Region solver currently uses a linear solver.
69 if (options->minimizer_type == TRUST_REGION &&
70 options->num_linear_solver_threads > 1) {
71 LOG(WARNING)
72 << "OpenMP support is not compiled into this binary; "
73 << "only options.num_linear_solver_threads=1 is supported. Switching "
74 << "to single threaded mode.";
75 options->num_linear_solver_threads = 1;
76 }
77#endif // CERES_USE_OPENMP
78}

80void SetupCommonMinimizerOptions(PreprocessedProblem* pp) {
81 const Solver::Options& options = pp->options;
82 Program* program = pp->reduced_program.get();
83
84 // Assuming that the parameter blocks in the program have been
85 // reordered as needed, extract them into a contiguous vector.
86 pp->reduced_parameters.resize(program->NumParameters());
87 double* reduced_parameters = pp->reduced_parameters.data();
88 program->ParameterBlocksToStateVector(reduced_parameters);
89
90 Minimizer::Options& minimizer_options = pp->minimizer_options;
91 minimizer_options = Minimizer::Options(options);
92 minimizer_options.evaluator = pp->evaluator;
93
94 if (options.logging_type != SILENT) {
95 pp->logging_callback.reset(
96 new LoggingCallback(options.minimizer_type,
97 options.minimizer_progress_to_stdout));
98 minimizer_options.callbacks.insert(minimizer_options.callbacks.begin(),
99 pp->logging_callback.get());
100 }
101
102 if (options.update_state_every_iteration) {
103 pp->state_updating_callback.reset(
104 new StateUpdatingCallback(program, reduced_parameters));
105 // This must get pushed to the front of the callbacks so that it
106 // is run before any of the user callbacks.
107 minimizer_options.callbacks.insert(minimizer_options.callbacks.begin(),
108 pp->state_updating_callback.get());
109 }
110}
