49// Helper functor used by the constructor for reordering the contents
50// of a TripletSparseMatrix. This comparator assumes thay there are no
51// duplicates in the pair of arrays rows and cols, i.e., there is no
52// indices i and j (not equal to each other) s.t.
53//
54// rows[i] == rows[j] && cols[i] == cols[j]
55//
56// If this is the case, this functor will not be a StrictWeakOrdering.
57struct RowColLessThan {
58 RowColLessThan(const int* rows, const int* cols) : rows(rows), cols(cols) {}
59
60 bool operator()(const int x, const int y) const {
61 if (rows[x] == rows[y]) {
62 return (cols[x] < cols[y]);
63 }
64 return (rows[x] < rows[y]);
65 }
66
67 const int* rows;
68 const int* cols;
69};

71void TransposeForCompressedRowSparseStructure(const int num_rows,
72 const int num_cols,
73 const int num_nonzeros,
74 const int* rows,
75 const int* cols,
76 const double* values,
77 int* transpose_rows,
78 int* transpose_cols,
79 double* transpose_values) {
80 // Explicitly zero out transpose_rows.
81 std::fill(transpose_rows, transpose_rows + num_cols + 1, 0);
82
83 // Count the number of entries in each column of the original matrix
84 // and assign to transpose_rows[col + 1].
85 for (int idx = 0; idx < num_nonzeros; ++idx) {
86 ++transpose_rows[cols[idx] + 1];
87 }
88
89 // Compute the starting position for each row in the transpose by
90 // computing the cumulative sum of the entries of transpose_rows.
91 for (int i = 1; i < num_cols + 1; ++i) {
92 transpose_rows[i] += transpose_rows[i - 1];
93 }
94
95 // Populate transpose_cols and (optionally) transpose_values by
96 // walking the entries of the source matrices. For each entry that
97 // is added, the value of transpose_row is incremented allowing us
98 // to keep track of where the next entry for that row should go.
99 //
100 // As a result transpose_row is shifted to the left by one entry.
101 for (int r = 0; r < num_rows; ++r) {
102 for (int idx = rows[r]; idx < rows[r + 1]; ++idx) {
103 const int c = cols[idx];
104 const int transpose_idx = transpose_rows[c]++;
105 transpose_cols[transpose_idx] = r;
106 if (values != NULL && transpose_values != NULL) {
107 transpose_values[transpose_idx] = values[idx];
108 }
109 }
110 }
111
112 // This loop undoes the left shift to transpose_rows introduced by
113 // the previous loop.
114 for (int i = num_cols - 1; i > 0; --i) {
115 transpose_rows[i] = transpose_rows[i - 1];
116 }
117 transpose_rows[0] = 0;
118}

120void AddRandomBlock(const int num_rows,
121 const int num_cols,
122 const int row_block_begin,
123 const int col_block_begin,
124 std::vector<int>* rows,
125 std::vector<int>* cols,

167CompressedRowSparseMatrix* CompressedRowSparseMatrix::FromTripletSparseMatrix(
168 const TripletSparseMatrix& input, bool transpose) {
169 int num_rows = input.num_rows();
170 int num_cols = input.num_cols();
171 const int* rows = input.rows();
172 const int* cols = input.cols();
173 const double* values = input.values();
174
175 if (transpose) {
176 std::swap(num_rows, num_cols);
177 std::swap(rows, cols);
178 }
179
180 // index is the list of indices into the TripletSparseMatrix input.
181 vector<int> index(input.num_nonzeros(), 0);
182 for (int i = 0; i < input.num_nonzeros(); ++i) {
183 index[i] = i;
184 }
185
186 // Sort index such that the entries of m are ordered by row and ties
187 // are broken by column.
188 std::sort(index.begin(), index.end(), RowColLessThan(rows, cols));
189
190 VLOG(1) << "# of rows: " << num_rows << " # of columns: " << num_cols
191 << " num_nonzeros: " << input.num_nonzeros() << ". Allocating "
192 << ((num_rows + 1) * sizeof(int) + // NOLINT
193 input.num_nonzeros() * sizeof(int) + // NOLINT
194 input.num_nonzeros() * sizeof(double)); // NOLINT
195
196 CompressedRowSparseMatrix* output =
197 new CompressedRowSparseMatrix(num_rows, num_cols, input.num_nonzeros());
198
199 // Copy the contents of the cols and values array in the order given
200 // by index and count the number of entries in each row.
201 int* output_rows = output->mutable_rows();
202 int* output_cols = output->mutable_cols();
203 double* output_values = output->mutable_values();
204
205 output_rows[0] = 0;
206 for (int i = 0; i < index.size(); ++i) {
207 const int idx = index[i];
208 ++output_rows[rows[idx] + 1];
209 output_cols[i] = cols[idx];
210 output_values[i] = values[idx];
211 }
212
213 // Find the cumulative sum of the row counts.
214 for (int i = 1; i < num_rows + 1; ++i) {
215 output_rows[i] += output_rows[i - 1];
216 }
217
218 CHECK_EQ(output->num_nonzeros(), input.num_nonzeros());
219 return output;
220}

222CompressedRowSparseMatrix::CompressedRowSparseMatrix(const double* diagonal,
223 int num_rows) {
224 CHECK_NOTNULL(diagonal);
225
226 num_rows_ = num_rows;
227 num_cols_ = num_rows;
228 storage_type_ = UNSYMMETRIC;
229 rows_.resize(num_rows + 1);
230 cols_.resize(num_rows);
231 values_.resize(num_rows);
232
233 rows_[0] = 0;
234 for (int i = 0; i < num_rows_; ++i) {
235 cols_[i] = i;
236 values_[i] = diagonal[i];
237 rows_[i + 1] = i + 1;
238 }
239
240 CHECK_EQ(num_nonzeros(), num_rows);
241}

249void CompressedRowSparseMatrix::RightMultiply(const double* x,
250 double* y) const {
251 CHECK_NOTNULL(x);
252 CHECK_NOTNULL(y);
253
254 for (int r = 0; r < num_rows_; ++r) {
255 for (int idx = rows_[r]; idx < rows_[r + 1]; ++idx) {
256 y[r] += values_[idx] * x[cols_[idx]];
257 }
258 }
259}

261void CompressedRowSparseMatrix::LeftMultiply(const double* x, double* y) const {
262 CHECK_NOTNULL(x);
263 CHECK_NOTNULL(y);
264
265 for (int r = 0; r < num_rows_; ++r) {
266 for (int idx = rows_[r]; idx < rows_[r + 1]; ++idx) {
267 y[cols_[idx]] += values_[idx] * x[r];
268 }
269 }
270}

272void CompressedRowSparseMatrix::SquaredColumnNorm(double* x) const {
273 CHECK_NOTNULL(x);
274
275 std::fill(x, x + num_cols_, 0.0);
276 for (int idx = 0; idx < rows_[num_rows_]; ++idx) {
277 x[cols_[idx]] += values_[idx] * values_[idx];
278 }

301void CompressedRowSparseMatrix::DeleteRows(int delta_rows) {
302 CHECK_GE(delta_rows, 0);
303 CHECK_LE(delta_rows, num_rows_);
304
305 num_rows_ -= delta_rows;
306 rows_.resize(num_rows_ + 1);
307
308 // The rest of the code updates the block information. Immediately
309 // return in case of no block information.
310 if (row_blocks_.empty()) {
311 return;
312 }
313
314 // Sanity check for compressed row sparse block information
315 CHECK_EQ(crsb_rows_.size(), row_blocks_.size() + 1);
316 CHECK_EQ(crsb_rows_.back(), crsb_cols_.size());
317
318 // Walk the list of row blocks until we reach the new number of rows
319 // and the drop the rest of the row blocks.
320 int num_row_blocks = 0;
321 int num_rows = 0;
322 while (num_row_blocks < row_blocks_.size() && num_rows < num_rows_) {
323 num_rows += row_blocks_[num_row_blocks];
324 ++num_row_blocks;
325 }
326
327 row_blocks_.resize(num_row_blocks);
328
329 // Update compressed row sparse block (crsb) information.
330 CHECK_EQ(num_rows, num_rows_);
331 crsb_rows_.resize(num_row_blocks + 1);
332 crsb_cols_.resize(crsb_rows_[num_row_blocks]);
333}

335void CompressedRowSparseMatrix::AppendRows(const CompressedRowSparseMatrix& m) {
336 CHECK_EQ(m.num_cols(), num_cols_);
337
338 CHECK((row_blocks_.empty() && m.row_blocks().empty()) ||
339 (!row_blocks_.empty() && !m.row_blocks().empty()))
340 << "Cannot append a matrix with row blocks to one without and vice versa."
341 << "This matrix has : " << row_blocks_.size() << " row blocks."
342 << "The matrix being appended has: " << m.row_blocks().size()
343 << " row blocks.";
344
345 if (m.num_rows() == 0) {
346 return;
347 }
348
349 if (cols_.size() < num_nonzeros() + m.num_nonzeros()) {
350 cols_.resize(num_nonzeros() + m.num_nonzeros());
351 values_.resize(num_nonzeros() + m.num_nonzeros());
352 }
353
354 // Copy the contents of m into this matrix.
355 DCHECK_LT(num_nonzeros(), cols_.size());
356 if (m.num_nonzeros() > 0) {
357 std::copy(m.cols(), m.cols() + m.num_nonzeros(), &cols_[num_nonzeros()]);
358 std::copy(
359 m.values(), m.values() + m.num_nonzeros(), &values_[num_nonzeros()]);
360 }
361
362 rows_.resize(num_rows_ + m.num_rows() + 1);
363 // new_rows = [rows_, m.row() + rows_[num_rows_]]
364 std::fill(rows_.begin() + num_rows_,
365 rows_.begin() + num_rows_ + m.num_rows() + 1,
366 rows_[num_rows_]);
367
368 for (int r = 0; r < m.num_rows() + 1; ++r) {
369 rows_[num_rows_ + r] += m.rows()[r];
370 }
371
372 num_rows_ += m.num_rows();
373
374 // The rest of the code updates the block information. Immediately
375 // return in case of no block information.
376 if (row_blocks_.empty()) {
377 return;
378 }
379
380 // Sanity check for compressed row sparse block information
381 CHECK_EQ(crsb_rows_.size(), row_blocks_.size() + 1);
382 CHECK_EQ(crsb_rows_.back(), crsb_cols_.size());
383
384 row_blocks_.insert(
385 row_blocks_.end(), m.row_blocks().begin(), m.row_blocks().end());
386
387 // The rest of the code updates the compressed row sparse block
388 // (crsb) information.
389 const int num_crsb_nonzeros = crsb_cols_.size();
390 const int m_num_crsb_nonzeros = m.crsb_cols_.size();
391 crsb_cols_.resize(num_crsb_nonzeros + m_num_crsb_nonzeros);
392 std::copy(&m.crsb_cols()[0],
393 &m.crsb_cols()[0] + m_num_crsb_nonzeros,
394 &crsb_cols_[num_crsb_nonzeros]);
395
396 const int num_crsb_rows = crsb_rows_.size() - 1;
397 const int m_num_crsb_rows = m.crsb_rows_.size() - 1;
398 crsb_rows_.resize(num_crsb_rows + m_num_crsb_rows + 1);
399 std::fill(crsb_rows_.begin() + num_crsb_rows,
400 crsb_rows_.begin() + num_crsb_rows + m_num_crsb_rows + 1,
401 crsb_rows_[num_crsb_rows]);
402
403 for (int r = 0; r < m_num_crsb_rows + 1; ++r) {
404 crsb_rows_[num_crsb_rows + r] += m.crsb_rows()[r];
405 }
406}

437CompressedRowSparseMatrix* CompressedRowSparseMatrix::CreateBlockDiagonalMatrix(
438 const double* diagonal, const vector<int>& blocks) {
439 int num_rows = 0;
440 int num_nonzeros = 0;
441 for (int i = 0; i < blocks.size(); ++i) {
442 num_rows += blocks[i];
443 num_nonzeros += blocks[i] * blocks[i];
444 }
445
446 CompressedRowSparseMatrix* matrix =
447 new CompressedRowSparseMatrix(num_rows, num_rows, num_nonzeros);
448
449 int* rows = matrix->mutable_rows();
450 int* cols = matrix->mutable_cols();
451 double* values = matrix->mutable_values();
452 std::fill(values, values + num_nonzeros, 0.0);
453
454 int idx_cursor = 0;
455 int col_cursor = 0;
456 for (int i = 0; i < blocks.size(); ++i) {
457 const int block_size = blocks[i];
458 for (int r = 0; r < block_size; ++r) {
459 *(rows++) = idx_cursor;
460 values[idx_cursor + r] = diagonal[col_cursor + r];
461 for (int c = 0; c < block_size; ++c, ++idx_cursor) {
462 *(cols++) = col_cursor + c;
463 }
464 }
465 col_cursor += block_size;
466 }
467 *rows = idx_cursor;
468
469 *matrix->mutable_row_blocks() = blocks;
470 *matrix->mutable_col_blocks() = blocks;
471
472 // Fill compressed row sparse block (crsb) information.
473 vector<int>& crsb_rows = *matrix->mutable_crsb_rows();
474 vector<int>& crsb_cols = *matrix->mutable_crsb_cols();
475 for (int i = 0; i < blocks.size(); ++i) {
476 crsb_rows.push_back(i);
477 crsb_cols.push_back(i);
478 }
479 crsb_rows.push_back(blocks.size());
480
481 CHECK_EQ(idx_cursor, num_nonzeros);
482 CHECK_EQ(col_cursor, num_rows);
483 return matrix;
484}

486CompressedRowSparseMatrix* CompressedRowSparseMatrix::Transpose() const {
487 CompressedRowSparseMatrix* transpose =
488 new CompressedRowSparseMatrix(num_cols_, num_rows_, num_nonzeros());
489
490 switch (storage_type_) {
491 case UNSYMMETRIC:
492 transpose->set_storage_type(UNSYMMETRIC);
493 break;
494 case LOWER_TRIANGULAR:
495 transpose->set_storage_type(UPPER_TRIANGULAR);
496 break;
497 case UPPER_TRIANGULAR:
498 transpose->set_storage_type(LOWER_TRIANGULAR);
499 break;
500 default:
501 LOG(FATAL) << "Unknown storage type: " << storage_type_;
502 };
503
504 TransposeForCompressedRowSparseStructure(num_rows(),
505 num_cols(),
506 num_nonzeros(),
507 rows(),
508 cols(),
509 values(),
510 transpose->mutable_rows(),
511 transpose->mutable_cols(),
512 transpose->mutable_values());
513
514 // The rest of the code updates the block information. Immediately
515 // return in case of no block information.
516 if (row_blocks_.empty()) {
517 return transpose;
518 }
519
520 // Sanity check for compressed row sparse block information
521 CHECK_EQ(crsb_rows_.size(), row_blocks_.size() + 1);
522 CHECK_EQ(crsb_rows_.back(), crsb_cols_.size());
523
524 *(transpose->mutable_row_blocks()) = col_blocks_;
525 *(transpose->mutable_col_blocks()) = row_blocks_;
526
527 // The rest of the code updates the compressed row sparse block
528 // (crsb) information.
529 vector<int>& transpose_crsb_rows = *transpose->mutable_crsb_rows();
530 vector<int>& transpose_crsb_cols = *transpose->mutable_crsb_cols();
531
532 transpose_crsb_rows.resize(col_blocks_.size() + 1);
533 transpose_crsb_cols.resize(crsb_cols_.size());
534 TransposeForCompressedRowSparseStructure(row_blocks().size(),
535 col_blocks().size(),
536 crsb_cols().size(),
537 crsb_rows().data(),
538 crsb_cols().data(),
539 NULL,
540 transpose_crsb_rows.data(),
541 transpose_crsb_cols.data(),
542 NULL);
543
544 return transpose;
545}

548// A ProductTerm is a term in the block outer product of a matrix with
549// itself.
550struct ProductTerm {
551 ProductTerm(const int row, const int col, const int index)
552 : row(row), col(col), index(index) {}
553
554 bool operator<(const ProductTerm& right) const {
555 if (row == right.row) {
556 if (col == right.col) {
557 return index < right.index;
558 }
559 return col < right.col;
560 }
561 return row < right.row;
562 }
563
564 int row;
565 int col;
566 int index;
567};
568
569// Create outer product matrix based on the block product information.
570// The input block product is already sorted. This function does not
571// set the sparse rows/cols information. Instead, it only collects the
572// nonzeros for each compressed row and puts in row_nnz. The caller of
573// this function will traverse the block product in a second round to
574// generate the sparse rows/cols information. This function also
575// computes the block offset information for the outer product matrix,
576// which is used in outer product computation.
577CompressedRowSparseMatrix* CreateOuterProductMatrix(
578 const int num_cols,
579 const CompressedRowSparseMatrix::StorageType storage_type,
580 const vector<int>& blocks,
581 const vector<ProductTerm>& product,
582 vector<int>* row_nnz) {
583 // Count the number of unique product term, which in turn is the
584 // number of non-zeros in the outer product. Also count the number
585 // of non-zeros in each row.
586 row_nnz->resize(blocks.size());
587 std::fill(row_nnz->begin(), row_nnz->end(), 0);
588 (*row_nnz)[product[0].row] = blocks[product[0].col];
589 int num_nonzeros = blocks[product[0].row] * blocks[product[0].col];
590 for (int i = 1; i < product.size(); ++i) {
591 // Each (row, col) block counts only once.
592 // This check depends on product sorted on (row, col).
593 if (product[i].row != product[i - 1].row ||
594 product[i].col != product[i - 1].col) {
595 (*row_nnz)[product[i].row] += blocks[product[i].col];
596 num_nonzeros += blocks[product[i].row] * blocks[product[i].col];
597 }
598 }
599
600 CompressedRowSparseMatrix* matrix =
601 new CompressedRowSparseMatrix(num_cols, num_cols, num_nonzeros);
602 matrix->set_storage_type(storage_type);
603
604 *(matrix->mutable_row_blocks()) = blocks;
605 *(matrix->mutable_col_blocks()) = blocks;
606
607 // Compute block offsets for outer product matrix, which is used in
608 // ComputeOuterProduct.
609 vector<int>* block_offsets = matrix->mutable_block_offsets();
610 block_offsets->resize(blocks.size() + 1);
611 (*block_offsets)[0] = 0;
612 for (int i = 0; i < blocks.size(); ++i) {
613 (*block_offsets)[i + 1] = (*block_offsets)[i] + blocks[i];
614 }
615
616 return matrix;
617}

619CompressedRowSparseMatrix* CompressAndFillProgram(
620 const int num_cols,
621 const CompressedRowSparseMatrix::StorageType storage_type,
622 const vector<int>& blocks,
623 const vector<ProductTerm>& product,
624 vector<int>* program) {
625 CHECK_GT(product.size(), 0);
626
627 vector<int> row_nnz;
628 CompressedRowSparseMatrix* matrix = CreateOuterProductMatrix(
629 num_cols, storage_type, blocks, product, &row_nnz);
630
631 const vector<int>& block_offsets = matrix->block_offsets();
632
633 int* crsm_rows = matrix->mutable_rows();
634 std::fill(crsm_rows, crsm_rows + num_cols + 1, 0);
635 int* crsm_cols = matrix->mutable_cols();
636 std::fill(crsm_cols, crsm_cols + matrix->num_nonzeros(), 0);
637
638 CHECK_NOTNULL(program)->clear();
639 program->resize(product.size());
640
641 // Non zero elements are not stored consecutively across rows in a block.
642 // We seperate nonzero into three categories:
643 // nonzeros in all previous row blocks counted in nnz
644 // nonzeros in current row counted in row_nnz
645 // nonzeros in previous col blocks of current row counted in col_nnz
646 //
647 // Give an element (j, k) within a block such that j and k
648 // represent the relative position to the starting row and starting col of
649 // the block, the row and col for the element is
650 // block_offsets[current.row] + j
651 // block_offsets[current.col] + k
652 // The total number of nonzero to the element is
653 // nnz + row_nnz[current.row] * j + col_nnz + k
654 //
655 // program keeps col_nnz for block product, which is used later for
656 // outer product computation.
657 //
658 // There is no special handling for diagonal blocks as we generate
659 // BLOCK triangular matrix (diagonal block is full block) instead of
660 // standard triangular matrix.
661 int nnz = 0;
662 int col_nnz = 0;
663
664 // Process first product term.
665 for (int j = 0; j < blocks[product[0].row]; ++j) {
666 crsm_rows[block_offsets[product[0].row] + j + 1] = row_nnz[product[0].row];
667 for (int k = 0; k < blocks[product[0].col]; ++k) {
668 crsm_cols[row_nnz[product[0].row] * j + k] =
669 block_offsets[product[0].col] + k;
670 }
671 }
672
673 (*program)[product[0].index] = 0;
674
675 // Process rest product terms.
676 for (int i = 1; i < product.size(); ++i) {
677 const ProductTerm& previous = product[i - 1];
678 const ProductTerm& current = product[i];
679
680 // Sparsity structure is updated only if the term is not a repeat.
681 if (previous.row != current.row || previous.col != current.col) {
682 col_nnz += blocks[previous.col];
683 if (previous.row != current.row) {
684 nnz += col_nnz * blocks[previous.row];
685 col_nnz = 0;
686
687 for (int j = 0; j < blocks[current.row]; ++j) {
688 crsm_rows[block_offsets[current.row] + j + 1] = row_nnz[current.row];
689 }
690 }
691
692 for (int j = 0; j < blocks[current.row]; ++j) {
693 for (int k = 0; k < blocks[current.col]; ++k) {
694 crsm_cols[nnz + row_nnz[current.row] * j + col_nnz + k] =
695 block_offsets[current.col] + k;
696 }
697 }
698 }
699
700 (*program)[current.index] = col_nnz;
701 }
702
703 for (int i = 1; i < num_cols + 1; ++i) {
704 crsm_rows[i] += crsm_rows[i - 1];
705 }
706
707 return matrix;
708}
709
710// input is a matrix of dimesion <row_block_size, input_cols>
711// output is a matrix of dimension <col_block1_size, output_cols>
712//
713// Implement block multiplication O = I1' * I2.
714// I1 is block(0, col_block1_begin, row_block_size, col_block1_size) of input
715// I2 is block(0, col_block2_begin, row_block_size, col_block2_size) of input
716// O is block(0, 0, col_block1_size, col_block2_size) of output
717void ComputeBlockMultiplication(const int row_block_size,
718 const int col_block1_size,
719 const int col_block2_size,
720 const int col_block1_begin,
721 const int col_block2_begin,
722 const int input_cols,
723 const double* input,
724 const int output_cols,
725 double* output) {
726 for (int r = 0; r < row_block_size; ++r) {
727 for (int idx1 = 0; idx1 < col_block1_size; ++idx1) {
728 for (int idx2 = 0; idx2 < col_block2_size; ++idx2) {
729 output[output_cols * idx1 + idx2] +=
730 input[input_cols * r + col_block1_begin + idx1] *
731 input[input_cols * r + col_block2_begin + idx2];
732 }
733 }
734 }
735}
736} // namespace

738CompressedRowSparseMatrix*
739CompressedRowSparseMatrix::CreateOuterProductMatrixAndProgram(
740 const CompressedRowSparseMatrix& m,
741 const CompressedRowSparseMatrix::StorageType storage_type,
742 vector<int>* program) {
743 CHECK(storage_type == LOWER_TRIANGULAR || storage_type == UPPER_TRIANGULAR);
744 CHECK_NOTNULL(program)->clear();
745 CHECK_GT(m.num_nonzeros(), 0)
746 << "Congratulations, you found a bug in Ceres. Please report it.";
747
748 vector<ProductTerm> product;
749 const vector<int>& col_blocks = m.col_blocks();
750 const vector<int>& crsb_rows = m.crsb_rows();
751 const vector<int>& crsb_cols = m.crsb_cols();
752
753 // Give input matrix m in Compressed Row Sparse Block format
754 // (row_block, col_block)
755 // represent each block multiplication
756 // (row_block, col_block1)' X (row_block, col_block2)
757 // by its product term index and sort the product terms
758 // (col_block1, col_block2, index)
759 //
760 // Due to the compression on rows, col_block is accessed through idx to
761 // crsb_cols. So col_block is accessed as crsb_cols[idx] in the code.
762 for (int row_block = 1; row_block < crsb_rows.size(); ++row_block) {
763 for (int idx1 = crsb_rows[row_block - 1]; idx1 < crsb_rows[row_block];
764 ++idx1) {
765 if (storage_type == LOWER_TRIANGULAR) {
766 for (int idx2 = crsb_rows[row_block - 1]; idx2 <= idx1; ++idx2) {
767 product.push_back(
768 ProductTerm(crsb_cols[idx1], crsb_cols[idx2], product.size()));
769 }
770 } else { // Upper triangular matrix.
771 for (int idx2 = idx1; idx2 < crsb_rows[row_block]; ++idx2) {
772 product.push_back(
773 ProductTerm(crsb_cols[idx1], crsb_cols[idx2], product.size()));
774 }
775 }
776 }
777 }
778
779 sort(product.begin(), product.end());
780 return CompressAndFillProgram(
781 m.num_cols(), storage_type, col_blocks, product, program);
782}
783
784// Give input matrix m in Compressed Row Sparse Block format
785// (row_block, col_block)
786// compute outer product m' * m as sum of block multiplications
787// (row_block, col_block1)' X (row_block, col_block2)
788//
789// Given row_block of the input matrix m, we use m_row_begin to represent
790// the starting row of the row block and m_row_nnz to represent number of
791// nonzero in each row of the row block, then the rows belonging to
792// the row block can be represented as a dense matrix starting at
793// m.values() + m.rows()[m_row_begin]
794// with dimension
795// <m.row_blocks()[row_block], m_row_nnz>
796//
797// Then each input matrix block (row_block, col_block) can be represented as
798// a block of above dense matrix starting at position
799// (0, m_col_nnz)
800// with size
801// <m.row_blocks()[row_block], m.col_blocks()[col_block]>
802// where m_col_nnz is the number of nonzero before col_block in each row.
803//
804// The outer product block is represented similarly with m_row_begin,
805// m_row_nnz, m_col_nnz, etc. replaced by row_begin, row_nnz, col_nnz,
806// etc. The difference is, m_row_begin and m_col_nnz is counted
807// during the traverse of block multiplication, while row_begin and
808// col_nnz are got from pre-computed block_offsets and program.
809//
810// Due to the compression on rows, col_block is accessed through
811// idx to crsb_col vector. So col_block is accessed as crsb_col[idx]
812// in the code.
813//
814// Note this function produces a triangular matrix in block unit (i.e.
815// diagonal block is a normal block) instead of standard triangular matrix.
816// So there is no special handling for diagonal blocks.
817void CompressedRowSparseMatrix::ComputeOuterProduct(
818 const CompressedRowSparseMatrix& m,
819 const vector<int>& program,
820 CompressedRowSparseMatrix* result) {
821 CHECK(result->storage_type() == LOWER_TRIANGULAR ||
822 result->storage_type() == UPPER_TRIANGULAR);
823 result->SetZero();
824 double* values = result->mutable_values();
825 const int* rows = result->rows();
826 const vector<int>& block_offsets = result->block_offsets();
827
828 int cursor = 0;
829 const double* m_values = m.values();
830 const int* m_rows = m.rows();
831 const vector<int>& row_blocks = m.row_blocks();
832 const vector<int>& col_blocks = m.col_blocks();
833 const vector<int>& crsb_rows = m.crsb_rows();
834 const vector<int>& crsb_cols = m.crsb_cols();
835 const StorageType storage_type = result->storage_type();
836#define COL_BLOCK1 (crsb_cols[idx1])
837#define COL_BLOCK2 (crsb_cols[idx2])
838
839 // Iterate row blocks.
840 for (int row_block = 0, m_row_begin = 0; row_block < row_blocks.size();
841 m_row_begin += row_blocks[row_block++]) {
842 // Non zeros are not stored consecutively across rows in a block.
843 // The gaps between rows is the number of nonzeros of the
844 // input matrix compressed row.
845 const int m_row_nnz = m_rows[m_row_begin + 1] - m_rows[m_row_begin];
846
847 // Iterate (col_block1 x col_block2).
848 for (int idx1 = crsb_rows[row_block], m_col_nnz1 = 0;
849 idx1 < crsb_rows[row_block + 1];
850 m_col_nnz1 += col_blocks[COL_BLOCK1], ++idx1) {
851 // Non zeros are not stored consecutively across rows in a
852 // block. The gaps between rows is the number of nonzeros of the
853 // outer product matrix compressed row.
854 const int row_begin = block_offsets[COL_BLOCK1];
855 const int row_nnz = rows[row_begin + 1] - rows[row_begin];
856 if (storage_type == LOWER_TRIANGULAR) {
857 for (int idx2 = crsb_rows[row_block], m_col_nnz2 = 0; idx2 <= idx1;
858 m_col_nnz2 += col_blocks[COL_BLOCK2], ++idx2, ++cursor) {
859 int col_nnz = program[cursor];
860 ComputeBlockMultiplication(row_blocks[row_block],
861 col_blocks[COL_BLOCK1],
862 col_blocks[COL_BLOCK2],
863 m_col_nnz1,
864 m_col_nnz2,
865 m_row_nnz,
866 m_values + m_rows[m_row_begin],
867 row_nnz,
868 values + rows[row_begin] + col_nnz);
869 }
870 } else {
871 for (int idx2 = idx1, m_col_nnz2 = m_col_nnz1;
872 idx2 < crsb_rows[row_block + 1];
873 m_col_nnz2 += col_blocks[COL_BLOCK2], ++idx2, ++cursor) {
874 int col_nnz = program[cursor];
875 ComputeBlockMultiplication(row_blocks[row_block],
876 col_blocks[COL_BLOCK1],
877 col_blocks[COL_BLOCK2],
878 m_col_nnz1,
879 m_col_nnz2,
880 m_row_nnz,
881 m_values + m_rows[m_row_begin],
882 row_nnz,
883 values + rows[row_begin] + col_nnz);
884 }
885 }
886 }
887 }
888
889#undef COL_BLOCK1
890#undef COL_BLOCK2
891
892 CHECK_EQ(cursor, program.size());
893}

895CompressedRowSparseMatrix* CompressedRowSparseMatrix::CreateRandomMatrix(
896 const CompressedRowSparseMatrix::RandomMatrixOptions& options) {
897 CHECK_GT(options.num_row_blocks, 0);
898 CHECK_GT(options.min_row_block_size, 0);
899 CHECK_GT(options.max_row_block_size, 0);
900 CHECK_LE(options.min_row_block_size, options.max_row_block_size);
901 CHECK_GT(options.num_col_blocks, 0);
902 CHECK_GT(options.min_col_block_size, 0);
903 CHECK_GT(options.max_col_block_size, 0);
904 CHECK_LE(options.min_col_block_size, options.max_col_block_size);
905 CHECK_GT(options.block_density, 0.0);
906 CHECK_LE(options.block_density, 1.0);
907
908 vector<int> row_blocks;
909 vector<int> col_blocks;
910
911 // Generate the row block structure.
912 for (int i = 0; i < options.num_row_blocks; ++i) {
913 // Generate a random integer in [min_row_block_size, max_row_block_size]
914 const int delta_block_size =
915 Uniform(options.max_row_block_size - options.min_row_block_size);
916 row_blocks.push_back(options.min_row_block_size + delta_block_size);
917 }
918
919 // Generate the col block structure.
920 for (int i = 0; i < options.num_col_blocks; ++i) {
921 // Generate a random integer in [min_col_block_size, max_col_block_size]
922 const int delta_block_size =
923 Uniform(options.max_col_block_size - options.min_col_block_size);
924 col_blocks.push_back(options.min_col_block_size + delta_block_size);
925 }
926
927 vector<int> crsb_rows;
928 vector<int> crsb_cols;
929 vector<int> tsm_rows;
930 vector<int> tsm_cols;
931 vector<double> tsm_values;
932
933 // For ease of construction, we are going to generate the
934 // CompressedRowSparseMatrix by generating it as a
935 // TripletSparseMatrix and then converting it to a
936 // CompressedRowSparseMatrix.
937
938 // It is possible that the random matrix is empty which is likely
939 // not what the user wants, so do the matrix generation till we have
940 // at least one non-zero entry.
941 while (tsm_values.empty()) {
942 crsb_rows.clear();
943 crsb_cols.clear();
944 tsm_rows.clear();
945 tsm_cols.clear();
946 tsm_values.clear();
947
948 int row_block_begin = 0;
949 for (int r = 0; r < options.num_row_blocks; ++r) {
950 int col_block_begin = 0;
951 crsb_rows.push_back(crsb_cols.size());
952 for (int c = 0; c < options.num_col_blocks; ++c) {
953 // Randomly determine if this block is present or not.
954 if (RandDouble() <= options.block_density) {
955 AddRandomBlock(row_blocks[r],
956 col_blocks[c],
957 row_block_begin,
958 col_block_begin,
959 &tsm_rows,
960 &tsm_cols,
961 &tsm_values);
962 // Add the block to the block sparse structure.
963 crsb_cols.push_back(c);
964 }
965 col_block_begin += col_blocks[c];
966 }
967 row_block_begin += row_blocks[r];
968 }
969 crsb_rows.push_back(crsb_cols.size());
970 }
971
972 const int num_rows = std::accumulate(row_blocks.begin(), row_blocks.end(), 0);
973 const int num_cols = std::accumulate(col_blocks.begin(), col_blocks.end(), 0);
974 const bool kDoNotTranspose = false;
975 CompressedRowSparseMatrix* matrix =
976 CompressedRowSparseMatrix::FromTripletSparseMatrix(
977 TripletSparseMatrix(
978 num_rows, num_cols, tsm_rows, tsm_cols, tsm_values),
979 kDoNotTranspose);
980 (*matrix->mutable_row_blocks()) = row_blocks;
981 (*matrix->mutable_col_blocks()) = col_blocks;
982 (*matrix->mutable_crsb_rows()) = crsb_rows;
983 (*matrix->mutable_crsb_cols()) = crsb_cols;
984 matrix->set_storage_type(CompressedRowSparseMatrix::UNSYMMETRIC);
985 return matrix;
986}
